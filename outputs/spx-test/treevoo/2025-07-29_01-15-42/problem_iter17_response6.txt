```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Incorporate Volume Impact
    volume_adjusted_spread = intraday_high_low_spread * df['volume']
    volume_adjusted_change = close_to_open_change * df['volume']
    
    # Analyze Volume Trends
    moving_average_volume = df['volume'].rolling(window=20).mean()
    
    # Combine Weighed Components
    sum_volume_adjusted_spreads = volume_adjusted_spread + volume_adjusted_change
    
    # Identify Positive and Negative Price-Volume Imbalances
    price_volume_imbalance = (df['high'] - df['low']) / moving_average_volume
    threshold = 1.5  # Example threshold
    positive_imbalance = price_volume_imbalance > threshold
    negative_imbalance = price_volume_imbalance < -threshold
    
    # Generate Preliminary Alpha Signal
    preliminary_alpha_signal = pd.Series(0, index=df.index)
    preliminary_alpha_signal[positive_imbalance] = 1
    preliminary_alpha_signal[negative_imbalance] = -1
    
    # Calculate Percentage Change of the Close Price from the Previous Day
    percentage_change_close = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Analyze the Spread Between High and Low Prices
    daily_volatility = df['high'] - df['low']
    lagged_daily_volatility = daily_volatility - daily_volatility.shift(1)
    
    # Consider the Relationship Between Closing Price and Volume (PVT)
    pvt = (df['close'] - df['close'].shift(1)) * df['volume']
    moving_average_pvt = pvt.rolling(window=20).mean()
    pvt_diff = pvt - moving_average_pvt
    
    # Explore Patterns in the Amount Data
    amount_trend = df['amount'] - df['amount'].rolling(window=20).mean()
    
    # Combine Multiple Indicators to Form a Composite Factor
    alpha_score = (
        0.4 * percentage_change_close + 
        0.3 * pvt_diff + 
        0.3 * amount_trend
    )
    
    # Generate Adjusted Price-Volume Signal
    adjusted_price_volume_signal = (df['high'] - df['low']) / df['volume']
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Apply Volatility Filter
    median_volatility = realized_volatility.median()
    volatility_filter = (realized_volatility < median_volatility)
    alpha_signal = preliminary_alpha_signal.copy()
    alpha_signal[~volatility_filter] = 0
    
    # Generate Final Alpha Signal
    momentum_score = 0.5 * (alpha_score + adjusted_price_volume_signal)
    alpha_signal[positive_imbalance] += momentum_score
    alpha_signal[negative_imbalance] -= momentum_score
    alpha_signal[(~positive_imbalance) & (~negative_imbalance)] = momentum_score
    
    return alpha_signal
```
