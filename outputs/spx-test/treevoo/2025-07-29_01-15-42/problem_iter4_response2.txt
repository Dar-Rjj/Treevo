import pandas as pd

def heuristics_v2(df, threshold=0.5):
    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Compute Moving Average Volume
    df['Moving_Average_Volume'] = df['volume'].rolling(window=20).mean()
    
    # Generate Alpha Signal
    df['Volume_Adjusted_Price_Range'] = df['Daily_Price_Range'] / df['Moving_Average_Volume']
    
    def set_alpha_factor_value(row):
        if row['Volume_Adjusted_Price_Range'] > threshold:
            return 1
        elif row['Volume_Adjusted_Price_Range'] < -threshold:
            return -1
        else:
            return 0
    
    df['Alpha_Factor'] = df.apply(set_alpha_factor_value, axis=1)
    
    return df['Alpha_Factor']
