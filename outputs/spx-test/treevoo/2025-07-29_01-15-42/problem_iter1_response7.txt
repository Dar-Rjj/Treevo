```python
import pandas as pd

def heuristics_v2(df, sma_window=10):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Smooth with Simple Moving Average (SMA)
    df['sma_return'] = df['close_return'].rolling(window=sma_window).mean()
    
    # Summarize Volume over the SMA period
    df['total_volume'] = df['volume'].rolling(window=sma_window).sum()
    
    # Calculate Ratio of Momentum to Volume
    df['volume_adjusted_momentum'] = df['sma_return'] / df['total_volume']
    
    return df['volume_adjusted_momentum']
```
