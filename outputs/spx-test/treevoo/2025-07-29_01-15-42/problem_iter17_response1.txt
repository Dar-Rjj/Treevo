```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] / df['close'].shift(28)) - 1
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(10)) - 1
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Close-to-Close Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Smooth with Exponential Moving Average (EMA)
    ema_5 = daily_return.ewm(span=5, adjust=False).mean()
    total_volume = df['volume'].rolling(window=5).sum()
    total_amount = df['amount'].rolling(window=5).sum()
    
    # Adjust for Volume and Amount
    volume_amount_adjusted_momentum = ema_5 / (total_volume + total_amount)
    
    # Calculate Daily Price Range
    price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'] - df['close'].shift(1)).apply(lambda x: max(x, 0))
    losses = (df['close'] - df['close'].shift(1)).apply(lambda x: abs(min(x, 0)))
    smoothed_gains = gains.ewm(span=14, adjust=False).mean()
    smoothed_losses = losses.ewm(span=14, adjust=False).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Weighted Average of Combined Components
    weighted_average = (combined_momentum * 0.3 + 
                        high_low_spread * 0.2 + 
                        close_to_open_movement * 0.1 + 
                        realized_volatility * 0.1 + 
                        price_range_ratio * 0.15 + 
                        rtsi * 0.15)
    
    # Final Factor
    volume_acceleration = (df['volume'] / df['volume'].shift(10))
    final_factor = weighted_average * volume_acceleration * volume_amount_adjusted_momentum
    
    return final_factor
```
