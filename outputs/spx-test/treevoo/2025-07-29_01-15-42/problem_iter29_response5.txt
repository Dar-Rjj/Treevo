import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the ratio of high to low price
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the ratio of close to open price
    df['close_open_ratio'] = df['close'] / df['open']
    
    # Calculate the relative volume as a ratio of current volume to the average volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the above ratios and percentage changes into a single factor
    df['factor'] = (df['close_change'] * 0.5 + 
                    (df['high_low_ratio'] - 1) * 0.3 +
                    (df['close_open_ratio'] - 1) * 0.1 +
                    (df['volume_ratio'] - 1) * 0.1)
    
    return df['factor'].dropna()
