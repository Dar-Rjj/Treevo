import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Moving Average Volume (Last 20 days)
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    adjusted_pv_signal = daily_price_change / moving_avg_volume
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility (Close price, 20-day rolling window)
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust by Average Volume
    avg_volume = moving_avg_volume
    adjusted_high_low_range = high_low_range / avg_volume
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_open_movement = df['close'] - df['open']
    
    # Weighted Average of Combined Components
    weighted_avg = (0.4 * combined_momentum) + \
                   (0.3 * intraday_spread) + \
                   (0.2 * close_open_movement) + \
                   (0.1 * adjusted_high_low_range)
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] - df['volume'].shift(10)
    
    # Adjust by Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.01
    positive_imbalance = (adjusted_pv_signal > threshold)
    negative_imbalance = (adjusted_pv_signal < -threshold)
    
    # Generate Preliminary Alpha Signal
    preliminary_alpha = (positive_imbalance).astype(int) - (negative_imbalance).astype(int)
    
    # Calculate Momentum Score
    momentum_score = combined_momentum - combined_momentum.shift(1)
    
    # Final Alpha Signal Based on Combined Indicators
    median_realized_volatility = realized_volatility.median()
    final_factor = (
        ((adjusted_pv_signal > threshold) & (realized_volatility < median_realized_volatility) & (combined_momentum > 0)).astype(int) - 
        ((adjusted_pv_signal < -threshold) & (realized_volatility < median_realized_volatility) & (combined_momentum < 0)).astype(int)
    )
    
    # Generate Final Alpha Signal
    final_alpha = final_factor.copy()
    final_alpha[positive_imbalance] += momentum_score
    final_alpha[negative_imbalance] -= momentum_score
    
    return final_alpha
