```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    df['Close_to_Open_Change'] = df['close'] - df['open']
    
    # Analyze Volume Trends
    df['Moving_Average_Volume'] = df['volume'].rolling(window=30).mean()
    
    # Generate Volume-Adjusted Indicators
    df['Volume_Adjusted_Price_Range'] = df['Daily_Price_Range'] / df['Moving_Average_Volume']
    df['Volume_Adjusted_High_Low_Spread'] = (df['high'] - df['low']) * df['volume']
    df['Volume_Adjusted_Close_to_Open_Change'] = df['Close_to_Open_Change'] * df['volume']
    
    # Calculate Realized Volatility
    df['Realized_Volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['open']
    df['Rolling_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=10).mean()
    
    # Final Alpha Signal Based on Combined Indicators
    threshold = 0.5  # Example threshold value
    median_volatility = df['Realized_Volatility'].median()
    
    def generate_alpha_signal(row):
        if row['Volume_Adjusted_Price_Range'] > threshold and row['Realized_Volatility'] < median_volatility:
            return 1
        elif row['Volume_Adjusted_Price_Range'] < -threshold and row['Realized_Volatility'] < median_volatility:
            return -1
        else:
            if row['Volume_Adjusted_High_Low_Spread'] > 0 and row['Volume_Adjusted_Close_to_Open_Change'] > 0:
                imbalance = 1
            elif row['Volume_Adjusted_High_Low_Spread'] < 0 and row['Volume_Adjusted_Close_to_Open_Change'] < 0:
                imbalance = -1
            else:
                imbalance = 0
            
            momentum_score = row['Combined_Value'] - df.loc[row.name - pd.Timedelta(days=1), 'Combined_Value']
            
            if imbalance == 1:
                return 1 + momentum_score
            elif imbalance == -1:
                return -1 - momentum_score
            else:
                return momentum_score
    
    df['Combined_Value'] = df['Volume_Adjusted_Price_Range'] + df['Volume_Adjusted_High_Low_Spread'] + df['Volume_Adjusted_Close_to_Open_Change']
    df['Alpha_Factor'] = df.apply(generate_alpha_signal, axis=1)
    
    # Introduce New Indicator: Volume-Weighted Close
    df['Volume_Weighted_Close'] = df['close'] * df['volume']
    df['Moving_Average_Volume_Weighted_Close'] = df['Volume_Weighted_Close'].rolling(window=20).mean()
    
    return df['Alpha_Factor']
```
