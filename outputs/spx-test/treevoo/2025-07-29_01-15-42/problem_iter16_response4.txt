```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open']
    
    # Smooth with EMA (e.g., 10 days)
    ema_window = 10
    df['ema_close_to_open_return'] = df['close_to_open_return'].ewm(span=ema_window, adjust=False).mean()
    
    # Adjust for Volume and Amount
    df['total_volume_ema_period'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount_ema_period'] = df['amount'].rolling(window=ema_window).sum()
    df['adjusted_momentum_score'] = df['ema_close_to_open_return'] / (df['total_volume_ema_period'] + df['total_amount_ema_period'])
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['daily_price_range'] / df['close']
    
    # Compute RTSI
    gains = df['close'].diff().apply(lambda x: max(x, 0))
    losses = df['close'].diff().apply(lambda x: abs(min(x, 0)))
    
    ema_gains = gains.ewm(span=14, adjust=False).mean()
    ema_losses = losses.ewm(span=14, adjust=False).mean()
    
    df['rTSI'] = 100 * (ema_gains / (ema_gains + ema_losses))
    
    # Calculate Daily Volume Momentum
    df['daily_volume_change'] = df['volume'].diff()
    ema_volume_window = 5
    df['ema_daily_volume_change'] = df['daily_volume_change'].ewm(span=ema_volume_window, adjust=False).mean()
    
    # Combine Factors
    df['combined_factor'] = (df['price_range_ratio'] * df['rTSI']) + df['adjusted_momentum_score'] + df['ema_daily_volume_change']
    
    return df['combined_factor']
```
