```python
import pandas as pd

def heuristics_v2(df):
    # Calculate price momentum
    price_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate price volatility
    price_volatility = df['close'].rolling(window=20).std()
    
    # Detect price trend reversal
    trend_reversal = (df['close'] - df['low'].shift(10)) - (df['high'].shift(10) - df['close'].shift(10))
    
    # Measure volume momentum
    volume_momentum = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    
    # Assess volume relative to price
    volume_price_ratio = (df['volume'] / df['close']) - (df['volume'].shift(20) / df['close'].shift(20))
    
    # Create a hybrid factor
    hybrid_factor = price_momentum * volume_momentum
    
    # Construct a divergence indicator
    divergence_indicator = price_momentum - volume_momentum
    
    # Evaluate true range
    true_range = df[['high', 'low']].diff(axis=1).abs().max(axis=1).combine(
        (df['high'] - df['close'].shift(1)).abs(), max
    ).combine(
        (df['low'] - df['close'].shift(1)).abs(), max
    )
    
    # Assess normalized true range
    sma_true_range = true_range.rolling(window=20).mean()
    normalized_true_range = true_range / sma_true_range
    
    # Compute buy/sell pressure
    buy_sell_pressure = (df['close'] - df['open']) * df['volume']
    
    # Determine net buying/selling intensity
    net_buying_selling_intensity = buy_sell_pressure.rolling(window=20).sum()
    
    # Estimate liquidity
    liquidity = (df['high'] - df['low']) / df['volume']
    
    # Analyze market depth
    market_depth = (df['high'] - df['low']) / ((df['open'] + df['close']) / 2)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        price_momentum * 0.2 +
        price_volatility * 0.1 +
        trend_reversal * 0.1 +
        volume_momentum * 0.2 +
        volume_price_ratio * 0.1 +
        hybrid_factor * 0.1 +
        divergence_indicator * 0.1 +
        normalized_true_range * 0.1 +
        net_buying_selling_intensity * 0.1 +
        liquidity * 0.1 +
        market_depth * 0.1
    )
    
    return alpha_factor
```
