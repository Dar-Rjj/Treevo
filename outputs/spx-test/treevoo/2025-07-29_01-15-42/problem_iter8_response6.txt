import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] / df['low']
    
    # Compute Relative True Strength Index (RTSI)
    df['close_shifted'] = df['close'].shift(1)
    df['gain'] = df.apply(lambda row: row['close'] - row['close_shifted'] if row['close'] > row['close_shifted'] else 0, axis=1)
    df['loss'] = df.apply(lambda row: abs(row['close'] - row['close_shifted']) if row['close'] < row['close_shifted'] else 0, axis=1)
    df['ema_gain'] = df['gain'].ewm(span=14).mean()
    df['ema_loss'] = df['loss'].ewm(span=14).mean()
    df['rtsi'] = (df['ema_gain'] / (df['ema_gain'] + df['ema_loss'])) * 100
    
    # Combine Intraday and Open-to-Close Returns
    df['open_to_close_return'] = df['close'] / df['open']
    df['weighted_intraday_return'] = df['intraday_return'] * 0.5
    df['weighted_open_to_close_return'] = df['open_to_close_return'] * 0.5
    df['combined_returns'] = df['weighted_intraday_return'] + df['weighted_open_to_close_return']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['daily_price_range'] / df['open']
    
    # Calculate Volume Momentum
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_momentum'] = df['volume'] - df['volume_ma_20']
    
    # Combine Price and Volume Intensity
    df['price_volume_intensity'] = (df['price_range_ratio'] * df['rtsi']) + df['volume_momentum']
    
    # Calculate Daily Price Change
    df['close_to_close_return'] = df['close'].diff()
    df['sma_return'] = df['close_to_close_return'].rolling(window=5).mean()
    
    # Adjust for Volume
    df['volume_sum'] = df['volume'].rolling(window=5).sum()
    df['volume_adjusted_momentum'] = df['sma_return'] / df['volume_sum']
    
    # Combine Intensity and Momentum Factors
    df['final_alpha_factor'] = (df['daily_price_range'] * df['rtsi']) + (df['price_volume_intensity'] * df['volume_adjusted_momentum'])
    
    return df['final_alpha_factor']
