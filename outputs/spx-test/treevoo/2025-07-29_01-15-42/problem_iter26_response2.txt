```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    long_term_momentum = df['close'].pct_change(28)
    short_term_momentum = df['close'].pct_change(7)
    combined_momentum = 0.5 * (long_term_momentum + short_term_momentum)
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=10).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    ema_smoothed_close_to_open_return = close_to_open_return.ewm(span=10).mean()
    
    # Adjust for Volume and Amount
    total_volume = df['volume'].rolling(window=10).sum()
    total_amount = (df['close'] * df['volume']).rolling(window=10).sum()
    adjusted_ema_return = ema_smoothed_close_to_open_return / (total_volume + total_amount)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    gains = df['close'].diff().where(lambda x: x > 0, 0)
    losses = -df['close'].diff().where(lambda x: x < 0, 0)
    ema_gains = gains.ewm(span=14).mean()
    ema_losses = losses.ewm(span=14).mean()
    rtsi = (ema_gains / (ema_gains + ema_losses)) * 100
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.35 * combined_momentum +
        0.1 * intraday_high_low_spread +
        0.1 * adjusted_ema_return +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.2 * rtsi
    )
    
    # Final Factor
    volume_acceleration = df['volume'] / df['volume'].shift(10)
    final_factor = weighted_average * volume_acceleration * (df['volume'] / df['volume'].shift(1))
    
    return final_factor
```
