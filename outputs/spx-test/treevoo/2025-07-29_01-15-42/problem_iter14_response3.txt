```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Moving Average Volume
    moving_average_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    adjusted_price_volume_signal = daily_price_change / moving_average_volume
    
    # Calculate Realized Volatility
    realized_volatility = (df['close'] / df['close'].shift(1) - 1).rolling(window=20).std()
    
    # Compute Momentum
    momentum = df['close'] - df['close'].shift(10)
    
    # Define Threshold and Median Volatility
    threshold = adjusted_price_volume_signal.abs().mean()
    median_volatility = realized_volatility.median()
    
    # Initialize Alpha Factor Series
    alpha_factor = pd.Series(index=df.index, dtype=float)
    
    # Final Alpha Signal Based on Combined Indicators
    alpha_factor[(adjusted_price_volume_signal > threshold) & 
                 (realized_volatility < median_volatility) & 
                 (momentum > 0)] = 1
    
    alpha_factor[(adjusted_price_volume_signal < -threshold) & 
                 (realized_volatility < median_volatility) & 
                 (momentum < 0)] = -1
    
    alpha_factor[(~((adjusted_price_volume_signal > threshold) | 
                    (adjusted_price_volume_signal < -threshold)) | 
                  (realized_volatility >= median_volatility) | 
                  ((momentum <= 0) & (momentum >= 0)))] = 0
    
    return alpha_factor
```
