```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    long_term_momentum = df['close'].pct_change(periods=28)
    short_term_momentum = df['close'].pct_change(periods=10)
    combined_momentum = (long_term_momentum + short_term_momentum) / 2
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    ema_smoothed_close_to_open_return = close_to_open_return.ewm(span=10).mean()
    
    # Adjust for Volume and Amount
    total_volume_ema_period = df['volume'].rolling(window=10).sum()
    total_amount_ema_period = df['amount'].rolling(window=10).sum()
    adjusted_ema_smoothed_return = ema_smoothed_close_to_open_return / (total_volume_ema_period + total_amount_ema_period)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    positive_differences = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    negative_differences = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    smoothed_gains = positive_differences.ewm(span=14).mean()
    smoothed_losses = negative_differences.ewm(span=14).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.3,
        'intraday_high_low_spread': 0.2,
        'adjusted_ema_smoothed_return': 0.1,
        'realized_volatility': 0.1,
        'price_range_ratio': 0.15,
        'rtsi': 0.15
    }
    weighted_average = (
        combined_momentum * weights['combined_momentum'] +
        intraday_high_low_spread * weights['intraday_high_low_spread'] +
        adjusted_ema_smoothed_return * weights['adjusted_ema_smoothed_return'] +
        realized_volatility * weights['realized_volatility'] +
        price_range_ratio * weights['price_range_ratio'] +
        rtsi * weights['rtsi']
    )
    
    # Final Factor
    volume_change = df['volume'].pct_change()
    final_factor = weighted_average * volume_change
    
    return final_factor
```
