```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(14)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(5)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=5).std()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=5).mean()
    adjusted_high_low_range = high_low_range / avg_volume
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Movement
    close_open_movement = (df['close'] - df['open']) / df['open']
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.4,
        'intraday_high_low_spread': 0.2,
        'close_open_movement': 0.2,
        'adjusted_high_low_range': 0.2
    }
    weighted_average = (
        combined_momentum * weights['combined_momentum'] +
        intraday_high_low_spread * weights['intraday_high_low_spread'] +
        close_open_movement * weights['close_open_movement'] +
        adjusted_high_low_range * weights['adjusted_high_low_range']
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(5)
    
    # Adjust by Volume Trend
    volume_change = df['volume'].pct_change()
    final_factor = weighted_average * volume_acceleration * volume_change
    
    # Analyze Volume Trends
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Combine Price Range and Volume Analysis
    price_range_over_vol = high_low_range / moving_avg_volume
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.01
    positive_imbalance = (price_range_over_vol > threshold).astype(int)
    negative_imbalance = (price_range_over_vol < -threshold).astype(int)
    
    # Generate Preliminary Alpha Signal
    preliminary_alpha_signal = positive_imbalance - negative_imbalance
    
    # Calculate Momentum Score
    momentum_score = preliminary_alpha_signal.diff()
    
    # Generate Final Alpha Signal
    final_alpha_signal = final_factor.copy()
    final_alpha_signal[positive_imbalance == 1] += momentum_score
    final_alpha_signal[negative_imbalance == 1] -= momentum_score
    final_alpha_signal[preliminary_alpha_signal == 0] = final_factor
    
    return final_alpha_signal
```
