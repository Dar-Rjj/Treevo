```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns with weights
    combined_return = 0.5 * intraday_return + 0.5 * open_to_close_return
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Smooth with Simple Moving Average (SMA) with a window of 5 days
    sma_window = 5
    sma_smoothed_return = close_to_close_return.rolling(window=sma_window).mean()
    
    # Summarize Volume over the SMA period
    total_volume = df['volume'].rolling(window=sma_window).sum()
    
    # Calculate Ratio of Momentum to Volume
    volume_adjusted_momentum = sma_smoothed_return / total_volume
    
    # Integrate Combined Returns and Volume Adjusted Momentum Score
    factor_values = combined_return * volume_adjusted_momentum
    
    return factor_values
```
