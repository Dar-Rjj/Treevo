```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Movements
    price_range = df['high'] - df['low']
    
    # Analyze Volume Trends
    ma_volume = df['volume'].rolling(window=20).mean()
    
    # Combine Price Range and Volume Analysis
    volume_adjusted_price_range = price_range / ma_volume
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 1.5 * np.std(volume_adjusted_price_range)
    positive_imbalance = (volume_adjusted_price_range > threshold).astype(int)
    negative_imbalance = (volume_adjusted_price_range < -threshold).astype(int) * -1
    
    # Generate Preliminary Alpha Signal
    preliminary_alpha = positive_imbalance + negative_imbalance
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open = df['open'] - df['close'].shift(1)
    
    # Weighted Average of Intraday Components
    weighted_intraday = 0.6 * intraday_spread + 0.4 * close_to_open.abs()
    
    # Calculate Momentum Score
    momentum_score = weighted_intraday - weighted_intraday.shift(1)
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std() * np.sqrt(252)
    
    # Final Alpha Signal Based on Combined Indicators
    median_volatility = realized_volatility.median()
    final_alpha = pd.Series(index=df.index, dtype=float)

    for i in range(len(df)):
        if volume_adjusted_price_range.iloc[i] > threshold and realized_volatility.iloc[i] < median_volatility:
            final_alpha.iloc[i] = preliminary_alpha.iloc[i] + momentum_score.iloc[i]
        elif volume_adjusted_price_range.iloc[i] < -threshold and realized_volatility.iloc[i] < median_volatility:
            final_alpha.iloc[i] = preliminary_alpha.iloc[i] - momentum_score.iloc[i]
        elif preliminary_alpha.iloc[i] == 0:
            final_alpha.iloc[i] = momentum_score.iloc[i]
        else:
            final_alpha.iloc[i] = 0.0

    return final_alpha
```
