```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Momentum
    long_term_mom = (df['close'] / df['close'].shift(28)) - 1
    short_term_mom = (df['close'] / df['close'].shift(10)) - 1
    combined_momentum = (long_term_mom + short_term_mom) / 2
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = (df['high'] - df['low'])
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_ret = (df['open'].shift(-1) / df['close']) - 1
    ema_close_to_open_ret = close_to_open_ret.ewm(span=10, min_periods=10).mean()
    
    # Adjust for Volume and Amount
    agg_volume = df['volume'].rolling(window=10).sum()
    agg_amount = df['amount'].rolling(window=10).sum()
    adjusted_ema_return = ema_close_to_open_ret / (agg_volume + agg_amount)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff() > 0) * df['close'].diff()
    losses = (df['close'].diff() < 0) * df['close'].diff().abs()
    smoothed_gains = gains.ewm(span=14, min_periods=14).mean()
    smoothed_losses = losses.ewm(span=14, min_periods=14).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Weighted Average of Combined Components
    weighted_avg = (
        0.3 * combined_momentum +
        0.2 * intraday_high_low_spread +
        0.1 * adjusted_ema_return +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Final Factor
    volume_change = df['volume'] / df['volume'].shift(1)
    final_factor = weighted_avg * volume_change
    final_factor = final_factor / (agg_volume + agg_amount)

    return final_factor
```
