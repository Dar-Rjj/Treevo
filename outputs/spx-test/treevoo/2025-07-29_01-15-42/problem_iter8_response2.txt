```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open = df['close'] - df['open'].shift(1)
    
    # Incorporate Volume Impact: Adjust Weights Based on Volume
    volume_adj_close_to_open = close_to_open / df['volume']
    
    # Calculate Realized Volatility using a rolling window
    realized_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Range by Volume
    range_by_volume = (df['high'] - df['low']) / df['volume']
    
    # Incorporate Momentum Factor
    momentum = df['close'].pct_change(periods=126).rolling(window=20).mean()  # 126 days is approximately 6 months
    
    # Combine with Adjusted Range
    momentum_adjusted_range = range_by_volume * momentum
    
    # Calculate Final Alpha Factor
    final_alpha_factor = (intraday_spread + volume_adj_close_to_open) / (realized_volatility + momentum_adjusted_range)
    
    return final_alpha_factor
```
