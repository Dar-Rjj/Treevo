```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_open_change = df['close'] - df['open']
    
    # Incorporate Volume Impact
    volume_adjusted_high_low = high_low_spread * df['volume']
    volume_adjusted_close_open = close_open_change * df['volume']
    
    # Combine Weighed Components
    combined_value = volume_adjusted_high_low + volume_adjusted_close_open
    
    # Calculate Momentum Score
    momentum_score = combined_value - combined_value.shift(1)
    
    return momentum_score.dropna()
```
