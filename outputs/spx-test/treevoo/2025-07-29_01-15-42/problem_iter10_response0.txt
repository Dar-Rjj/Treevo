```python
import pandas as pd
import numpy as np

def heuristics(data):
    # Compute Daily Price Range
    data['daily_price_range'] = data['high'] - data['low']
    
    # Calculate Moving Average Volume (last 20 days)
    data['moving_avg_volume'] = data['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    data['volume_adjusted_price_range'] = data['daily_price_range'] / data['moving_avg_volume']
    
    # Calculate Realized Volatility (close price, rolling window of 20 days)
    data['realized_volatility'] = data['close'].rolling(window=20).std()
    median_volatility = data['realized_volatility'].median()
    
    # Set Threshold
    threshold = 0.5
    
    # Final Alpha Signal Based on Combined Indicators
    conditions = [
        (data['volume_adjusted_price_range'] > threshold) & (data['realized_volatility'] < median_volatility),
        (data['volume_adjusted_price_range'] < -threshold) & (data['realized_volatility'] < median_volatility)
    ]
    choices = [1, -1]
    data['alpha_factor'] = np.select(conditions, choices, default=0)
    
    return data['alpha_factor']
```
