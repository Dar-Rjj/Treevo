```python
import pandas as pd

def heuristics(data):
    # Calculate Daily Close-to-Close Return
    data['close_return'] = data['close'].diff()
    
    # Smooth with Simple Moving Average (SMA) - 5 days window
    data['sma_close_return'] = data['close_return'].rolling(window=5).mean()
    
    # Summarize Volume over the SMA period
    data['volume_sum'] = data['volume'].rolling(window=5).sum()
    
    # Summarize Amount over the SMA period
    data['amount_sum'] = data['amount'].rolling(window=5).sum()
    
    # Combine summarized volume and amount by summing them
    data['combined_vol_amt'] = data['volume_sum'] + data['amount_sum']
    
    # Calculate Ratio of Momentum to Combined Volume and Amount
    data['adj_momentum_score'] = data['sma_close_return'] / data['combined_vol_amt']
    
    # Calculate Daily Price Range
    data['price_range'] = data['high'] - data['low']
    
    # Calculate Price Range Ratio
    data['price_range_ratio'] = data['price_range'] / data['open']
    
    # Compute Relative True Strength Index (RTSI)
    data['gain'] = data['close'].diff().apply(lambda x: x if x > 0 else 0)
    data['loss'] = data['close'].diff().apply(lambda x: -x if x < 0 else 0)
    
    # Smooth Gains and Losses with 14-period Exponential Moving Average
    data['ema_gain'] = data['gain'].ewm(span=14, adjust=False).mean()
    data['ema_loss'] = data['loss'].ewm(span=14, adjust=False).mean()
    
    # Calculate RTSI
    data['rtsi'] = 100 * (data['ema_gain'] / (data['ema_gain'] + data['ema_loss']))
    
    # Combine Factors
    data['price_rtsi_product'] = data['price_range_ratio'] * data['rtsi']
    data['alpha_factor'] = data['price_rtsi_product'] + data['adj_momentum_score']
    
    # Multiply the combined result by Volume and Amount Adjusted Momentum Score
    data['final_alpha_factor'] = data['alpha_factor'] * data['adj_momentum_score']
    
    return data['final_alpha_factor'].dropna()
```
