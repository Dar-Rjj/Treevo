```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day average volume and the ratio of today's volume to this average
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    df['vol_ratio'] = df['volume'] / df['avg_vol_5d']

    # Calculate the difference between today's close and the 5-day average close
    df['avg_close_5d'] = df['close'].rolling(window=5).mean()
    df['close_diff'] = df['close'] - df['avg_close_5d']

    # Calculate the high-low spread for the day
    df['high_low_spread'] = df['high'] - df['low']

    # Calculate the open-close change
    df['open_close_change'] = df['close'] - df['open']

    # Calculate the current day's range and the previous day's range
    df['range_today'] = df['high'] - df['low']
    df['range_yesterday'] = df['range_today'].shift(1)

    # Calculate the interaction between the current day's trading volume and price action
    df['price_move_magnitude'] = (df['high'] - df['low']).abs()
    df['vol_price_interaction'] = (df['volume'] / df['price_move_magnitude'])

    # Combine all the sub-thoughts into a single factor
    df['alpha_factor'] = (
        df['vol_ratio'] + 
        df['close_diff'] + 
        df['high_low_spread'] + 
        df['open_close_change'] + 
        (df['range_today'] - df['range_yesterday']) + 
        df['vol_price_interaction']
    )

    return df['alpha_factor']
```
