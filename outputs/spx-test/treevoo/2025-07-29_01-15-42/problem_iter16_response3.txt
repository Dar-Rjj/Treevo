```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weighted Average of Combined Components
    weighted_average = (combined_momentum * 0.4
                        + intraday_spread * 0.3
                        + close_to_open_movement * 0.2
                        + realized_volatility * 0.1)
    
    # Volume Acceleration
    volume_acceleration = df['volume'].pct_change(10)
    
    # Final Factor
    final_factor = weighted_average * (1 + volume_acceleration)
    
    return final_factor
```
