```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movements
    high_prices = df['high']
    low_prices = df['low']
    daily_price_range = high_prices - low_prices
    
    # Analyze Volume Trends
    daily_volume = df['volume']
    moving_average_volume = daily_volume.rolling(window=20).mean()
    
    # Combine Price Range and Volume Analysis
    volume_adjusted_price_range = daily_price_range / moving_average_volume
    threshold = 1.5  # Example threshold, can be adjusted based on backtesting
    
    # Generate Alpha Signal
    alpha_signal = pd.Series(0, index=df.index)
    alpha_signal[volume_adjusted_price_range > threshold] = 1
    alpha_signal[volume_adjusted_price_range < -threshold] = -1
    
    return alpha_signal
```
