```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates an alpha factor based on a thought tree using open, high, low, close, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: Series with index as date and values as the alpha factor
    """
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / daily_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / daily_range
    
    # Calculate the volume-weighted buying pressure
    volume_weighted_buying_pressure = buying_pressure * df['volume']
    
    # Calculate the volume-weighted selling pressure
    volume_weighted_selling_pressure = selling_pressure * df['volume']
    
    # Calculate the net volume-weighted pressure
    net_volume_weighted_pressure = volume_weighted_buying_pressure - volume_weighted_selling_pressure
    
    # Calculate the price change
    price_change = df['close'].pct_change()
    
    # Calculate the momentum
    momentum = price_change.rolling(window=5).sum()
    
    # Calculate the final alpha factor
    alpha_factor = net_volume_weighted_pressure + momentum
    
    return alpha_factor
```
