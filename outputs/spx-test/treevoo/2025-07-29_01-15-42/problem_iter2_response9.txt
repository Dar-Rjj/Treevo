```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = df['high'] / df['low']
    
    # Compute Relative True Strength Index (RTSI)
    close_diff = df['close'].diff()
    gains = close_diff.clip(lower=0)
    losses = -close_diff.clip(upper=0)
    avg_gain = gains.rolling(window=14).mean()
    avg_loss = losses.rolling(window=14).mean()
    rtsi = (avg_gain / (avg_gain + avg_loss)) * 100
    
    # Combine Intraday and Open-to-Close Returns
    open_to_close_return = df['open'] / df['close']
    combined_return = 0.5 * (intraday_return + open_to_close_return)
    
    # Calculate Volume Impact
    volume_avg = df['volume'].rolling(window=5).mean()
    volume_adjusted_return = combined_return * volume_avg
    
    # Generate Final Alpha Factor
    final_alpha_factor = daily_price_range * rtsi * volume_adjusted_return
    
    return final_alpha_factor
```
