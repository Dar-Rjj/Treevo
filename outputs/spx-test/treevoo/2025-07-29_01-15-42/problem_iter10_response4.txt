import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] / df['close'].shift(14) - 1
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] / df['close'].shift(5) - 1
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=14).std()
    
    # Adjust High-to-Low Range by Average Volume
    average_volume = df['volume'].rolling(window=14).mean()
    adjusted_high_low_range = high_low_range / average_volume
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_open_movement = df['close'] - df['open']
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.3,
        'intraday_high_low_spread': 0.2,
        'close_open_movement': 0.2,
        'adjusted_high_low_range': 0.3
    }
    weighted_average = (
        combined_momentum * weights['combined_momentum'] +
        intraday_high_low_spread * weights['intraday_high_low_spread'] +
        close_open_movement * weights['close_open_movement'] +
        adjusted_high_low_range * weights['adjusted_high_low_range']
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'] / df['volume'].shift(5)
    
    # Final Factor
    final_factor = weighted_average * volume_acceleration
    final_factor = final_factor * (df['volume'] / df['volume'].shift(1))
    
    return final_factor
