```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open']
    
    # Smooth with EMA (10 days)
    ema_window = 10
    df['ema_close_to_open'] = df['close_to_open_return'].ewm(span=ema_window, min_periods=ema_window).mean()
    
    # Adjust for Volume and Amount
    total_volume = df['volume'].rolling(window=ema_window, min_periods=ema_window).sum()
    total_amount = (df['close'] * df['volume']).rolling(window=ema_window, min_periods=ema_window).sum()
    df['adjusted_ema_close_to_open'] = df['ema_close_to_open'] / (total_volume + total_amount)
    
    # Combine Long-Term and Short-Term Momentum
    long_term_momentum = df['close'].pct_change(periods=60)
    short_term_momentum = df['close'].pct_change(periods=10)
    df['combined_momentum'] = long_term_momentum + short_term_momentum
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Compute RTSI
    df['gain'] = df['close'].diff().apply(lambda x: max(x, 0))
    df['loss'] = df['close'].diff().apply(lambda x: -min(x, 0))
    gain_ema = df['gain'].ewm(span=14, min_periods=14).mean()
    loss_ema = df['loss'].ewm(span=14, min_periods=14).mean()
    df['rtpsi'] = (gain_ema / (gain_ema + loss_ema)) * 100
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['close'].rolling(window=10, min_periods=10).std()
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.4,
        'price_range_ratio': 0.2,
        'rtpsi': 0.2,
        'realized_volatility': 0.2
    }
    weighted_average = sum(df[col] * weight for col, weight in weights.items())
    
    # Final Factor
    df['alpha_factor'] = weighted_average * df['adjusted_ema_close_to_open']
    
    # Identify Short-Term Reversal
    df['short_term_reversal'] = df['close'].pct_change(periods=5) * df['close'].pct_change(periods=4) < 0
    
    # Examine Volume-Price Relationship
    df['positive_day'] = (df['close'] > df['open']).astype(int)
    df['volume_price_relation'] = (df['positive_day'] * df['volume']).rolling(window=20, min_periods=20).sum() / df['volume'].rolling(window=20, min_periods=20).sum()
    
    # Analyze Price Movement Characteristics
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    df['average_true_range'] = df['true_range'].rolling(window=14, min_periods=14).mean()
    df['above_prev_high'] = (df['close'] > df['high'].shift(1)).rolling(window=30, min_periods=30).sum()
    
    return df['alpha_factor']
```
