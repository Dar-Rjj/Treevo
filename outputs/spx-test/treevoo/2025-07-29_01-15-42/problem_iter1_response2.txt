```python
def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_return = 0.5 * intraday_return + 0.5 * open_to_close_return
    
    # Calculate Average Volume Impact
    volume_rolling_avg = df['volume'].rolling(window=5).mean()
    
    # Adjust Combined Return by Volume Average
    adjusted_return = combined_return / volume_rolling_avg
    
    return adjusted_return
```
