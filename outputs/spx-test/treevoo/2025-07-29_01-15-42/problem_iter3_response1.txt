```python
def heuristics_v2(df, sma_window=10):
    import pandas as pd
    
    # Compute Close-to-Close Return
    df['return'] = df['close'].pct_change()
    
    # Smooth with Simple Moving Average (SMA)
    df['sma_return'] = df['return'].rolling(window=sma_window).mean()
    
    # Summarize Volume over the SMA period
    df['sum_volume'] = df['volume'].rolling(window=sma_window).sum()
    
    # Summarize Amount over the SMA period
    df['sum_amount'] = df['amount'].rolling(window=sma_window).sum()
    
    # Combine summarized volume and amount by summing them
    df['combined_vol_amount'] = df['sum_volume'] + df['sum_amount']
    
    # Calculate Ratio of Momentum to Combined Volume and Amount
    df['adjusted_momentum_score'] = df['sma_return'] / df['combined_vol_amount']
    
    return df['adjusted_momentum_score']
```
