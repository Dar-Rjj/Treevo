```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(14)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(5)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility (using rolling window of 20 days for example)
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Adjust by Volume
    average_volume = df['volume'].rolling(window=20).mean()
    adjusted_high_to_low_range = high_to_low_range / average_volume
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = (df['close'] - df['open']) / df['open']
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.3,
        'intraday_high_low_spread': 0.2,
        'close_to_open_movement': 0.2,
        'adjusted_high_to_low_range': 0.3
    }
    
    weighted_average = (
        combined_momentum * weights['combined_momentum'] +
        intraday_high_low_spread * weights['intraday_high_low_spread'] +
        close_to_open_movement * weights['close_to_open_movement'] +
        adjusted_high_to_low_range * weights['adjusted_high_to_low_range']
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(5)
    
    # Adjust by Volume Trend
    volume_change = df['volume'].pct_change()
    
    # Final Factor
    final_factor = weighted_average * volume_acceleration * (1 + volume_change)
    
    # Integrate Open-Price Reversal
    open_price_reversal = (df['open'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Adjust Final Factor
    final_factor += open_price_reversal * 0.5  # Weighted by Open-Price Reversal
    
    return final_factor
```
