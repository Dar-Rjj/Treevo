```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Movements
    df['price_range'] = df['high'] - df['low']
    
    # Analyze Volume Trends
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Combine Price Range and Volume Analysis
    df['vol_adj_price_range'] = df['price_range'] / df['avg_volume_20']
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.5  # Example threshold, can be adjusted
    df['imbalance'] = 0
    df.loc[df['vol_adj_price_range'] > threshold, 'imbalance'] = 1
    df.loc[df['vol_adj_price_range'] < -threshold, 'imbalance'] = -1
    
    # Generate Preliminary Alpha Signal
    df['prelim_alpha'] = df['imbalance']
    
    # Calculate Intraday High-Low Spread and Close-to-Open Movement
    df['intraday_high_low_spread'] = (df['high'] + df['low']) / 2 - df['close'].shift(1)
    df['close_to_open'] = df['open'] - df['close'].shift(1)
    
    # Weighted Average of Intraday Components
    df['weighted_intraday'] = 0.6 * df['intraday_high_low_spread'] + 0.4 * df['close_to_open']
    
    # Calculate Momentum Score
    df['momentum_score'] = df['weighted_intraday'].diff()
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Final Alpha Signal Based on Combined Indicators
    median_volatility = df['realized_volatility'].median()
    df['alpha_factor'] = 0
    df.loc[(df['vol_adj_price_range'] > threshold) & (df['realized_volatility'] < median_volatility), 'alpha_factor'] = df['prelim_alpha'] + df['momentum_score']
    df.loc[(df['vol_adj_price_range'] < -threshold) & (df['realized_volatility'] < median_volatility), 'alpha_factor'] = df['prelim_alpha'] - df['momentum_score']
    df.loc[df['imbalance'] == 0, 'alpha_factor'] = df['momentum_score']
    
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust by Volume
    df['vol_adj_high_low_range'] = df['high_low_range'] / df['avg_volume_20']
    
    # Incorporate Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['adj_high_low_range_vol_momentum'] = df['vol_adj_high_low_range'] * df['volume_change']
    
    # Compute Relative True Strength Index (RTSI)
    df['gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rTSI'] = (df['avg_gain'] / (df['avg_gain'] + df['avg_loss'])) * 100
    
    # Calculate Daily Price Change
    df['close_to_close_return'] = df['close'].diff()
    
    # Adjust for Volume
    df['sum_volume_20'] = df['volume'].rolling(window=20).sum()
    df['volume_adjusted_momentum'] = df['close_to_close_return'] / df['sum_volume_20']
    
    # Combine Factors
    df['alpha_factor'] = df['adj_high_low_range_vol_momentum'] * df['rTSI'] * df['volume_adjusted_momentum']
    
    return df['alpha_factor']
```
