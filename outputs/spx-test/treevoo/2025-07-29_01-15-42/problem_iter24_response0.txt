```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(-1)
    ema_close_to_open = close_to_open_return.ewm(span=10).mean()
    
    # Adjust for Volume and Amount
    total_volume_ema = df['volume'].rolling(window=10).sum()
    total_amount_ema = df['amount'].rolling(window=10).sum()
    volume_amount_adjusted_ema = ema_close_to_open / (total_volume_ema + total_amount_ema)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'] - df['close'].shift(1)).apply(lambda x: max(x, 0))
    losses = (df['close'] - df['close'].shift(1)).apply(lambda x: -min(x, 0))
    smoothed_gains = gains.ewm(span=14).mean()
    smoothed_losses = losses.ewm(span=14).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Calculate EMA Smoothed Daily Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    ema_volume_change = volume_change.ewm(span=5).mean()
    
    # Calculate EMA Smoothed Daily Amount Change
    amount_change = df['amount'] - df['amount'].shift(1)
    ema_amount_change = amount_change.ewm(span=5).mean()
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.3 * combined_momentum +
        0.2 * intraday_high_low_spread +
        0.1 * volume_amount_adjusted_ema +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Final Factor
    volume_acceleration = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    final_factor = weighted_average * volume_acceleration * (1 + ema_volume_change) * (1 + ema_amount_change)
    
    return final_factor
```
