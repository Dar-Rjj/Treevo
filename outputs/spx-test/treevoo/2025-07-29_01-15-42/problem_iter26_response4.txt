```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Indicators
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['Daily_Return'] = df['close'].diff()

    # Calculate Moving Averages
    df['20D_MA_Volume'] = df['volume'].rolling(window=20).mean()
    df['Summarized_Volume'] = df['volume'].rolling(window=5).sum()  # Example: 5-day summary

    # Generate Alpha Signal
    # Volume-Adjusted Price Range
    df['Volume_Adjusted_Price_Range'] = df['Daily_Price_Range'] / df['20D_MA_Volume']

    # Volume-Adjusted Momentum
    df['SMA_Daily_Return'] = df['Daily_Return'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = df['SMA_Daily_Return'] / df['Summarized_Volume']

    # Intraday and Open-to-Close Returns
    df['Intraday_Return'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['Open_to_Close_Return'] = (df['close'] - df['open']) / df['open']

    # Combine Returns
    df['Weighted_Intraday_Return'] = df['Intraday_Return'] * 0.5
    df['Weighted_Open_to_Close_Return'] = df['Open_to_Close_Return'] * 0.5
    df['Combined_Return'] = df['Weighted_Intraday_Return'] + df['Weighted_Open_to_Close_Return']

    # Dynamic Threshold
    df['Std_Volume_Adjusted_Price_Range'] = df['Volume_Adjusted_Price_Range'].rolling(window=20).std()
    df['Dynamic_Threshold'] = 2 * df['Std_Volume_Adjusted_Price_Range']

    # Synthesize Final Alpha Factor
    conditions = [
        (df['Volume_Adjusted_Price_Range'] > df['Dynamic_Threshold']),
        (df['Volume_Adjusted_Price_Range'] < -df['Dynamic_Threshold'])
    ]
    choices = [1, -1]
    
    df['Alpha_Factor'] = np.select(conditions, choices, default=df['Volume_Adjusted_Momentum'] + df['Combined_Return'])

    return df['Alpha_Factor']
```
