```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    df['close_open_change'] = df['close'] - df['open']
    
    # Incorporate Volume Impact
    df['volume_adjusted_high_low_spread'] = df['high_low_spread'] * df['volume']
    df['volume_adjusted_close_open_change'] = df['close_open_change'] * df['volume']
    
    # Analyze Volume Trends
    df['moving_avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Combine Weighed Components
    df['sum_volume_adjusted_spreads'] = (df['volume_adjusted_high_low_spread'] + 
                                         df['volume_adjusted_close_open_change'])
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.05
    df['price_range_over_moving_avg_volume'] = df['high_low_spread'] / df['moving_avg_volume']
    df['imbalance_signal'] = 0
    df.loc[df['price_range_over_moving_avg_volume'] > threshold, 'imbalance_signal'] = 1
    df.loc[df['price_range_over_moving_avg_volume'] < -threshold, 'imbalance_signal'] = -1
    
    # Generate Preliminary Alpha Signal
    df['prelim_alpha_signal'] = df['imbalance_signal']
    
    # Calculate Percentage Change of the Close Price from the Previous Day
    df['pct_change_close'] = df['close'].pct_change()
    
    # Analyze the Spread Between High and Low Prices
    df['lagged_high_low_spread'] = df['high_low_spread'].shift(1)
    df['daily_volatility'] = df['high_low_spread'] - df['lagged_high_low_spread']
    
    # Consider the Relationship Between Closing Price and Volume (PVT)
    df['pvt'] = (df['close'] - df['close'].shift(1)) * df['volume']
    df['pvt_ma'] = df['pvt'].rolling(window=20).mean()
    df['pvt_diff'] = df['pvt'] - df['pvt_ma']
    
    # Explore Patterns in the Amount Data
    df['amount_ma'] = df['amount'].rolling(window=20).mean()
    df['amount_trend'] = df['amount'] - df['amount_ma']
    
    # Create an Alpha Score Based on the Combination of the Percentage Change, PVT, and the Amount Trend
    df['momentum_score'] = (0.3 * df['pct_change_close'] + 
                            0.4 * df['pvt_diff'] + 
                            0.3 * df['amount_trend'])
    
    # Adjust Alpha Signal
    df['adjusted_price_volume_signal'] = df['high_low_spread'] / df['volume']
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['close'].pct_change().rolling(window=20).std() * (252 ** 0.5)
    
    # Apply Volatility Filter
    median_volatility = df['realized_volatility'].median()
    df['filtered_alpha_signal'] = df['prelim_alpha_signal']
    df.loc[df['realized_volatility'] >= median_volatility, 'filtered_alpha_signal'] = 0
    
    # Generate Final Alpha Signal
    df['final_alpha_signal'] = df['filtered_alpha_signal']
    df.loc[df['filtered_alpha_signal'] == 1, 'final_alpha_signal'] += df['momentum_score']
    df.loc[df['filtered_alpha_signal'] == -1, 'final_alpha_signal'] -= df['momentum_score']
    df.loc[df['filtered_alpha_signal'] == 0, 'final_alpha_signal'] = df['momentum_score']
    
    # Incorporate Intraday Momentum
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open']
    df['final_alpha_signal'] += df['intraday_momentum'].apply(lambda x: x if x > 0 else -x)
    
    # Analyze Intraday Volume Trends
    df['intraday_volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df.loc[df['intraday_volume_ratio'] > 1.5, 'final_alpha_signal'] *= 1.2
    df.loc[df['intraday_volume_ratio'] < 0.5, 'final_alpha_signal'] *= 0.8
    
    return df['final_alpha_signal']
```
