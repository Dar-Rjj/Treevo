import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Adjusted Range
    high_to_low_range = df['high'] - df['low']
    realized_volatility = df['close'].rolling(window=20).std()
    avg_volume = df['volume'].rolling(window=20).mean()
    high_to_low_adjusted_range = high_to_low_range / avg_volume / realized_volatility

    # Calculate Momentum Component
    sma_short = df['close'].rolling(window=5).mean()
    sma_long = df['close'].rolling(window=20).mean()
    momentum_diff = sma_short - sma_long
    trend_direction = np.where(momentum_diff > 0, 1, -1)

    # Calculate Volume Component
    ema_volume_short = df['volume'].ewm(span=3, adjust=False).mean()
    ema_volume_long = df['volume'].ewm(span=10, adjust=False).mean()
    volume_diff = ema_volume_short - ema_volume_long
    volume_trend = np.where(volume_diff > 0, 1, -1)

    # Combine Momentum, Volume, and High-to-Low Adjusted Range
    combined_signal = (momentum_diff * volume_diff * high_to_low_adjusted_range) * (trend_direction * volume_trend)

    # Generate Integrated Alpha Factor
    buy_signals = np.where(combined_signal > 0, combined_signal, 0)
    sell_signals = np.where(combined_signal < 0, -combined_signal, 0)
    alpha_factor = buy_signals.sum() - sell_signals.sum()

    return pd.Series(alpha_factor, index=df.index, name='alpha_factor')
