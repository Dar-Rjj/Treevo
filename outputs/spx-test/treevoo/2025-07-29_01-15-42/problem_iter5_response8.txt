```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Smooth with Simple Moving Average (SMA) - 5 days
    sma_window = 5
    df['sma_close_return'] = df['close_return'].rolling(window=sma_window).mean()
    
    # Summarize Volume over the SMA period
    df['total_volume_sma'] = df['volume'].rolling(window=sma_window).sum()
    
    # Summarize Amount over the SMA period
    df['total_amount_sma'] = df['amount'].rolling(window=sma_window).sum()
    
    # Calculate Ratio of Momentum to Combined Volume and Amount
    df['combined_vol_amount'] = df['total_volume_sma'] + df['total_amount_sma']
    df['vol_amount_adj_momentum'] = df['sma_close_return'] / df['combined_vol_amount']
    
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['gain'] = df['close'].diff().apply(lambda x: max(x, 0))
    df['loss'] = df['close'].diff().apply(lambda x: max(-x, 0))
    
    # Smooth Gains and Losses - 14-period Exponential Moving Average
    ema_window = 14
    df['ema_gain'] = df['gain'].ewm(span=ema_window, adjust=False).mean()
    df['ema_loss'] = df['loss'].ewm(span=ema_window, adjust=False).mean()
    
    # Calculate RTSI
    df['rtsi'] = (df['ema_gain'] / (df['ema_gain'] + df['ema_loss'])) * 100
    
    # Calculate Volume Momentum
    vol_ema_window = 20
    df['vol_mom'] = df['volume'] - df['volume'].rolling(window=vol_ema_window).mean()
    
    # Combine Factors
    df['combined_factor'] = (df['price_range_ratio'] * df['rtsi']) + df['vol_mom']
    df['alpha_factor'] = df['combined_factor'] * df['vol_amount_adj_momentum']
    
    return df['alpha_factor']
```
