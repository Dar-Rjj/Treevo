```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] / df['close'].shift(28)) - 1
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(10)) - 1
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    daily_returns = df['close'].pct_change()
    realized_volatility = daily_returns.rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(-1)
    ema_close_to_open_return = close_to_open_return.ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume and Amount
    total_volume_ema = df['volume'].rolling(window=10).sum()
    total_amount_ema = df['amount'].rolling(window=10).sum()
    adjusted_ema_return = ema_close_to_open_return / (total_volume_ema + total_amount_ema)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    gains = df['close'].diff().clip(lower=0)
    losses = -df['close'].diff().clip(upper=0)
    smoothed_gains = gains.ewm(span=14, adjust=False).mean()
    smoothed_losses = losses.ewm(span=14, adjust=False).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Calculate EMA Smoothed Daily Volume Change
    daily_volume_change = df['volume'].diff()
    ema_daily_volume_change = daily_volume_change.ewm(span=5, adjust=False).mean()
    
    # Calculate EMA Smoothed Daily Amount Change
    daily_amount_change = df['amount'].diff()
    ema_daily_amount_change = daily_amount_change.ewm(span=5, adjust=False).mean()
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.3 * combined_momentum +
        0.2 * intraday_high_low_spread +
        0.1 * adjusted_ema_return +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Final Factor
    volume_acceleration = (df['volume'] / df['volume'].shift(10)) - 1
    final_factor = weighted_average * volume_acceleration
    final_factor = final_factor * (1 + (df['volume'] / df['volume'].shift(1)))
    final_factor = final_factor * (1 + (df['amount'] / df['amount'].shift(1)))
    
    return final_factor
```
