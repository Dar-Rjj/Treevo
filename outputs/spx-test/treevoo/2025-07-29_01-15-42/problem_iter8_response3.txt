```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Incorporate Volume Impact
    volume_adjusted_spread = intraday_high_low_spread * df['volume']
    volume_adjusted_change = close_to_open_change * df['volume']
    
    # Analyze Volume Trends
    moving_average_volume = df['volume'].rolling(window=20).mean()
    
    # Identify Positive and Negative Price-Volume Imbalances
    price_volume_imbalance = (df['high'] - df['low']) / moving_average_volume
    positive_imbalance = price_volume_imbalance > 1
    negative_imbalance = price_volume_imbalance < -1
    
    # Generate Preliminary Alpha Signal
    preliminary_alpha = pd.Series(0, index=df.index)
    preliminary_alpha[positive_imbalance] = 1
    preliminary_alpha[negative_imbalance] = -1
    
    # Calculate Momentum Score
    combined_value = volume_adjusted_spread + volume_adjusted_change
    momentum_score = combined_value.diff(periods=1)
    
    # Generate Final Alpha Signal
    final_alpha = pd.Series(0, index=df.index)
    final_alpha[positive_imbalance] = preliminary_alpha[positive_imbalance] + momentum_score[positive_imbalance]
    final_alpha[negative_imbalance] = preliminary_alpha[negative_imbalance] - momentum_score[negative_imbalance]
    final_alpha[(~positive_imbalance) & (~negative_imbalance)] = momentum_score[(~positive_imbalance) & (~negative_imbalance)]
    
    return final_alpha
```
