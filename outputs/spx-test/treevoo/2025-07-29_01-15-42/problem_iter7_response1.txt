```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(14)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(5)
    
    # Combine Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Intraday High-Low Spread
    intraday_high_low = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Movement
    close_to_open = (df['open'] - df['close']) / df['close']
    
    # Weights for the components
    weights = {
        'combined_momentum': 0.5,
        'intraday_high_low': 0.3,
        'close_to_open': 0.2
    }
    
    # Weighted Average of Combined Momentum, Intraday High-Low, and Close-to-Open
    weighted_factor = (
        weights['combined_momentum'] * combined_momentum +
        weights['intraday_high_low'] * intraday_high_low +
        weights['close_to_open'] * close_to_open
    )
    
    # Adjust by Volume Acceleration
    volume_acceleration = (df['volume'].pct_change() + 1).fillna(1)
    adjusted_factor = weighted_factor * volume_acceleration
    
    return adjusted_factor
```
