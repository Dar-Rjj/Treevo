```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate Average True Range for Volatility
    df['true_range'] = df[['high', 'low']].apply(lambda x: np.max(x) - np.min(x), axis=1)
    lookback_period = 14
    df['avg_true_range'] = df['true_range'].rolling(window=lookback_period).mean()

    # Determine Smoothing Factor based on Volatility
    df['smoothing_factor'] = 2 / (1 + df['avg_true_range'])

    # Apply Exponential Smoothing
    df['smoothed_return'] = df['daily_return'].ewm(alpha=df['smoothing_factor']).mean()

    # Calculate Moving Average of Returns
    moving_average_lookback = 50
    df['moving_avg_return'] = df['daily_return'].rolling(window=moving_average_lookback).mean()

    # Calculate Trend Strength
    df['trend_strength'] = df['smoothed_return'] - df['moving_avg_return']

    return df['trend_strength']
```
