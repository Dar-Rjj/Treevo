```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel alpha factor based on the given market data.
    
    Parameters:
    df (pandas.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pandas.Series: A Series of alpha factor values indexed by date.
    """
    
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the ratio of high to low
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the volume weighted close price
    vwap = (df['close'] * df['volume']) / df['volume'].rolling(window=5).mean()
    
    # Calculate the average true range
    atr = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine the factors into a single alpha factor
    alpha_factor = pct_change_close * high_low_ratio * vwap / (atr + 1e-6)
    
    return alpha_factor
```
