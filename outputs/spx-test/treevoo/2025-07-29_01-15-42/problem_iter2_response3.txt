```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['close_prev'] = df['close'].shift(1)
    df['gain'] = np.where(df['close'] > df['close_prev'], df['close'] - df['close_prev'], 0)
    df['loss'] = np.where(df['close'] < df['close_prev'], abs(df['close'] - df['close_prev']), 0)
    
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    
    df['rTSI'] = (df['avg_gain'] / (df['avg_gain'] + df['avg_loss'])) * 100
    
    # Calculate Volume Volatility
    df['volatility'] = df['volume'].rolling(window=20).std() / df['volume'].rolling(window=20).mean()
    
    # Combine Price and Volume Intensity
    df['intensity'] = df['price_range_ratio'] * df['rTSI'] * df['volatility']
    
    # Calculate Daily Price Change
    df['daily_return'] = df['close'].pct_change()
    
    # Smooth with Simple Moving Average (SMA) over 5 days
    df['sma_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Summarize Volume over the SMA period
    df['sma_volume'] = df['volume'].rolling(window=5).sum()
    
    # Calculate Ratio of Momentum to Volume
    df['momentum_to_volume'] = df['sma_return'] / df['sma_volume']
    
    # Combine Intensity and Momentum Factors
    df['alpha_factor'] = df['intensity'] * df['momentum_to_volume']
    
    return df['alpha_factor'].dropna()
```
