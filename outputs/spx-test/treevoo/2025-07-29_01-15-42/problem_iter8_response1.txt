```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Compute Moving Average Volume (Last 20 Days)
    df['moving_avg_vol'] = df['volume'].rolling(window=20).mean()
    
    # Generate Volume-Adjusted Price Range
    df['vol_adj_price_range'] = df['daily_price_range'] / df['moving_avg_vol']
    
    # Calculate Long-Term Price Momentum
    df['long_term_mom'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Calculate Short-Term Price Momentum
    df['short_term_mom'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Combine Long-Term and Short-Term Momentum
    df['combined_mom'] = df['long_term_mom'] + df['short_term_mom']
    
    # Calculate Intraday Price Range
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close to Range Midpoint Deviation
    df['range_midpoint'] = (df['high'] + df['low']) / 2
    df['close_to_midpoint_dev'] = df['close'] - df['range_midpoint']
    
    # Incorporate Volume and Amount Trends
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Calculate Close-to-Open Movement
    df['close_to_open_movement'] = (df['close'] - df['open']) / df['open']
    
    # Weighted Average of Momentum, Intraday, and Volume-Amount Components
    weights = {
        'combined_mom': 0.3,
        'intraday_high_low_spread': 0.2,
        'close_to_open_movement': 0.2,
        'vol_adj_price_range': 0.1,
        'close_to_midpoint_dev': 0.2
    }
    
    df['weighted_avg'] = (
        df['combined_mom'] * weights['combined_mom'] +
        df['intraday_high_low_spread'] * weights['intraday_high_low_spread'] +
        df['close_to_open_movement'] * weights['close_to_open_movement'] +
        df['vol_adj_price_range'] * weights['vol_adj_price_range'] +
        df['close_to_midpoint_dev'] * weights['close_to_midpoint_dev']
    )
    
    # Calculate Volume Acceleration
    df['vol_acceleration'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Generate Alpha Signal
    df['mom_vol_adj_signal'] = df['weighted_avg'] * df['vol_acceleration']
    
    # Final Alpha Factor
    threshold = 0.05
    df['alpha_factor'] = 0
    df.loc[df['mom_vol_adj_signal'] > threshold, 'alpha_factor'] = df['mom_vol_adj_signal']
    df.loc[df['mom_vol_adj_signal'] < -threshold, 'alpha_factor'] = -df['mom_vol_adj_signal']
    
    # Refine Final Alpha Factor
    vol_adj_threshold = 0.02
    df.loc[(df['vol_adj_price_range'] > vol_adj_threshold) & (df['alpha_factor'] != 0), 'alpha_factor'] = df['alpha_factor'] * 1
    df.loc[(df['vol_adj_price_range'] < -vol_adj_threshold) & (df['alpha_factor'] != 0), 'alpha_factor'] = df['alpha_factor'] * -1
    
    return df['alpha_factor']
```
