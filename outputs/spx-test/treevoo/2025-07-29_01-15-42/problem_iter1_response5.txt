```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    df['close_open_movement'] = df['close'] - df['open']

    # Assign weights
    weight_high_low_spread = 0.7
    weight_close_open_movement = 0.3
    
    # Weighted Average of Intraday Components
    df['weighted_intraday'] = (df['high_low_spread'] * weight_high_low_spread) + (df['close_open_movement'] * weight_close_open_movement)
    
    # Calculate Momentum Score
    df['momentum_score'] = df['weighted_intraday'] - df['weighted_intraday'].shift(1)

    return df['momentum_score']
```
