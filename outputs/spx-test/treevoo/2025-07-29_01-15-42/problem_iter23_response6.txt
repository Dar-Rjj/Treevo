```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA50'] = df['close'].rolling(window=50).mean()
    df['SMA200'] = df['close'].rolling(window=200).mean()
    
    # Identify Golden Cross and Death Cross
    df['GoldenCross'] = (df['SMA50'] > df['SMA200']) & (df['SMA50'].shift(1) <= df['SMA200'].shift(1))
    df['DeathCross'] = (df['SMA50'] < df['SMA200']) & (df['SMA50'].shift(1) >= df['SMA200'].shift(1))
    
    # Compute RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Determine Overbought and Oversold Conditions
    df['Overbought'] = df['RSI'] > 70
    df['Oversold'] = df['RSI'] < 30
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['CumulativeVolume'] = df['volume'].cumsum()
    df['CumulativePriceVolume'] = (df['close'] * df['volume']).cumsum()
    df['VWAP'] = df['CumulativePriceVolume'] / df['CumulativeVolume']
    
    # Compare VWAP with Current Price
    df['PriceAboveVWAP'] = df['close'] > df['VWAP']
    df['PriceBelowVWAP'] = df['close'] < df['VWAP']
    
    # Measure Standard Deviation of Close Prices
    df['StdDev'] = df['close'].rolling(window=20).std()
    
    # Analyze High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Evaluate Price Volatility Trends
    df['IncreaseInStdDev'] = df['StdDev'] > df['StdDev'].shift(1)
    df['DecreaseInHighLowRange'] = df['HighLowRange'] < df['HighLowRange'].shift(1)
    
    # Combine Factors for Alpha Score
    df['AlphaScore'] = (
        0.4 * (df['GoldenCross'] * 1 - df['DeathCross'] * 1) +
        0.3 * (df['Oversold'] * 1 - df['Overbought'] * 1) +
        0.2 * (df['PriceAboveVWAP'] * 1 - df['PriceBelowVWAP'] * 1) +
        0.1 * (df['IncreaseInStdDev'] * 1 - df['DecreaseInHighLowRange'] * 1)
    )
    
    return df['AlphaScore']
```
