```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open']
    
    # Smooth with EMA (10 days)
    ema_window = 10
    df['ema_close_to_open_return'] = df['close_to_open_return'].ewm(span=ema_window, adjust=False).mean()
    
    # Adjust for Volume and Amount
    df['total_volume_ema'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount_ema'] = (df['close'] * df['volume']).rolling(window=ema_window).sum()
    df['adjusted_return'] = df['ema_close_to_open_return'] / (df['total_volume_ema'] + df['total_amount_ema'])
    
    # Combine Long-Term and Short-Term Momentum
    long_term_momentum = df['close'] / df['close'].shift(28)
    short_term_momentum = df['close'] / df['close'].shift(7)
    df['combined_momentum'] = long_term_momentum + short_term_momentum
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Compute Relative Strength Index (RSI)
    df['price_change'] = df['close'].diff()
    df['gain'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['price_change'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['gain'].ewm(alpha=1/14, adjust=False).mean()
    df['avg_loss'] = df['loss'].ewm(alpha=1/14, adjust=False).mean()
    df['rsi'] = 100 - (100 / (1 + (df['avg_gain'] / df['avg_loss'])))
    
    # Calculate Realized Volatility (15-day rolling window)
    df['realized_volatility'] = df['close'].pct_change().rolling(window=15).std()
    
    # Integrate Volume Shock Factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_shock_factor'] = df['volume_change'] / df['volume_change'].rolling(window=30).std()
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.4,
        'price_range_ratio': 0.2,
        'rsi': 0.2,
        'realized_volatility': 0.1,
        'volume_shock_factor': 0.1
    }
    
    df['weighted_average'] = (
        df['combined_momentum'] * weights['combined_momentum'] +
        df['price_range_ratio'] * weights['price_range_ratio'] +
        df['rsi'] * weights['rsi'] +
        df['realized_volatility'] * weights['realized_volatility'] +
        df['volume_shock_factor'] * weights['volume_shock_factor']
    )
    
    # Final Factor
    df['alpha_factor'] = df['weighted_average'] * df['adjusted_return']
    
    return df['alpha_factor']
```
