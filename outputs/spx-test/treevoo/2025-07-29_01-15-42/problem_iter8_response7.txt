```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['close_shifted'] = df['close'].shift(1)
    df['gains'] = np.where(df['close'] > df['close_shifted'], df['close'] - df['close_shifted'], 0)
    df['losses'] = np.where(df['close'] < df['close_shifted'], df['close_shifted'] - df['close'], 0)
    df['avg_gain'] = df['gains'].rolling(window=14).mean()
    df['avg_loss'] = df['losses'].rolling(window=14).mean()
    df['rtsi'] = (df['avg_gain'] / (df['avg_gain'] + df['avg_loss'])) * 100
    
    # Calculate Volume Volatility
    df['vol_20d_mean'] = df['volume'].rolling(window=20).mean()
    df['vol_20d_std'] = df['volume'].rolling(window=20).std()
    df['vol_volatility'] = df['vol_20d_std'] / df['vol_20d_mean']
    
    # Combine Price and Volume Intensity
    df['intensity'] = df['price_range_ratio'] * df['rtsi'] * df['vol_volatility']
    
    # Analyze Volume Trends
    df['vol_moving_avg'] = df['volume'].rolling(window=20).mean()
    
    # Adjust Daily Price Range by Moving Average Volume
    df['vol_adjusted_price_range'] = df['price_range'] / df['vol_moving_avg']
    
    # Identify Positive and Negative Price-Volume Imbalances
    threshold = 0.5
    df['imbalance'] = np.where(df['vol_adjusted_price_range'] > threshold, 1,
                               np.where(df['vol_adjusted_price_range'] < -threshold, -1, 0))
    
    # Generate Initial Alpha Signal
    df['alpha_signal'] = df['imbalance']
    
    # Calculate Intraday Price Range
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Close to Range Midpoint Deviation
    df['close_to_midpoint_deviation'] = df['close'] - df['intraday_midpoint']
    
    # Incorporate Volume and Amount Trends
    df['vol_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Combine Volume-Adjusted Price Range and Intraday Reversal
    df['combined_signal'] = df['vol_adjusted_price_range'] * df['vol_change'] * df['amount_change']
    
    # Adjust Alpha Signal
    df['final_alpha_factor'] = df['alpha_signal'] + (df['close_to_midpoint_deviation'] * df['vol_change'])
    
    # Combine Intensity and Adjusted Alpha Factor
    df['alpha_factor'] = df['intensity'] * df['final_alpha_factor']
    
    return df['alpha_factor']
```
