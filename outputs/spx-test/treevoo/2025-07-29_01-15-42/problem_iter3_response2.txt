import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = df['close'] - df['open']
    
    # Incorporate Volume Impact
    volume_impact = (df['volume'] - df['volume'].mean()) / df['volume'].std()
    
    # Weighted Average of Intraday Components
    weights = {
        'intraday_high_low_spread': 0.6,
        'close_to_open_movement': 0.4
    }
    
    weighted_high_low = weights['intraday_high_low_spread'] * intraday_high_low_spread
    weighted_close_open = weights['close_to_open_movement'] * close_to_open_movement
    
    intraday_indicator = weighted_high_low + weighted_close_open
    intraday_indicator_with_volume = intraday_indicator * volume_impact
    
    # Calculate Momentum Score
    momentum_score = intraday_indicator_with_volume - intraday_indicator_with_volume.shift(1)
    
    return momentum_score
