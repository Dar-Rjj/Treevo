import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Realized Volatility
    rolling_window = 10
    realized_volatility = df['close'].rolling(window=rolling_window).std()
    
    # Adjust by Volume
    avg_volume = df['volume'].rolling(window=rolling_window).mean()
    high_to_low_adjusted_range = high_to_low_range / avg_volume
    
    # Calculate Momentum Component
    sma_short_term = df['close'].rolling(window=5).mean()
    sma_long_term = df['close'].rolling(window=20).mean()
    momentum_diff = sma_short_term - sma_long_term
    trend_direction = np.sign(momentum_diff)
    
    # Calculate Volume Component
    ema_short_term = df['volume'].ewm(span=3, adjust=False).mean()
    ema_long_term = df['volume'].ewm(span=10, adjust=False).mean()
    volume_diff = ema_short_term - ema_long_term
    volume_trend = np.sign(volume_diff)
    
    # Combine Momentum, Volume, and High-to-Low Adjusted Range
    combined_signal = trend_direction * volume_trend * high_to_low_adjusted_range
    
    # Generate Integrated Alpha Factor
    buy_signals = (trend_direction > 0) & (volume_trend > 0)
    sell_signals = (trend_direction < 0) & (volume_trend < 0)
    integrated_alpha_factor = (buy_signals * combined_signal).sum() - (sell_signals * combined_signal).sum()
    
    # Add Weighted Open Price Influence
    weighted_open_price = df['open'] * 0.5  # Example weighted factor
    integrated_alpha_factor *= weighted_open_price
    
    return pd.Series(integrated_alpha_factor, index=df.index)
