import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Average True Range for volatility
    df['true_range'] = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=lookback_period).mean()
    
    # Determine Smoothing Factor based on Volatility
    smoothing_factor = 2 / (1 + lookback_period)
    
    # Apply Exponential Smoothing to Daily Return
    df['smoothed_daily_return'] = df['daily_return'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Moving Average of Returns
    df['moving_avg_return'] = df['daily_return'].rolling(window=lookback_period).mean()
    
    # Generate Trend Strength Signal
    df['trend_strength_signal'] = df['smoothed_daily_return'] - df['moving_avg_return']
    
    # Introduce Price Momentum
    df['price_momentum'] = df['close'].pct_change(periods=lookback_period)
    
    # Combine Trend and Momentum
    df['alpha_factor'] = df['trend_strength_signal'] * df['price_momentum']
    
    return df['alpha_factor']
