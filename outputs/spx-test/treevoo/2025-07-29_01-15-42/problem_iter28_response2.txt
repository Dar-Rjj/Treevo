```python
import pandas as pd

def heuristics_v2(df):
    n = 10
    m = 60
    p = 5
    q = 20
    r = 14
    s = 30
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'].diff(n)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'].diff(m)
    
    # Short-Term Reversal
    momentum_shift = short_term_momentum.diff().rolling(window=p).sum()
    short_term_reversal = (momentum_shift > 0) & (short_term_momentum < 0) | (momentum_shift < 0) & (short_term_momentum > 0)
    short_term_reversal = short_term_reversal.astype(int)
    
    # Volume-Price Relationship
    volume_increase = (df['volume'] > df['volume'].shift(1)) & (df['close'] > df['open'])
    volume_price_relationship = volume_increase.rolling(window=q).mean()
    
    # Average True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift()).abs()
    tr3 = (df['low'] - df['close'].shift()).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    average_true_range = true_range.rolling(window=r).mean()
    
    # Days with Close Above Previous High
    close_above_high = (df['close'] > df['high'].shift())
    days_above_high = close_above_high.rolling(window=s).sum()
    
    # Combine all factors
    factor = (short_term_momentum + long_term_momentum + short_term_reversal + 
              volume_price_relationship + average_true_range + days_above_high)
    
    return factor
```
