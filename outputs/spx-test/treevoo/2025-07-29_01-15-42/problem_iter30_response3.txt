```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day simple moving average (SMA) of the closing prices
    sma_14 = df['close'].rolling(window=14).mean()
    
    # Calculate raw momentum by subtracting 14-day SMA from current day's closing price
    raw_momentum = df['close'] - sma_14
    
    # Smooth the momentum using an exponential moving average (EMA) with a span of 28
    smoothed_momentum = raw_momentum.ewm(span=28, adjust=False).mean()
    
    # Compute the relative strength as the ratio of the asset's close price to its lowest price over the last 52 weeks
    min_price_52_weeks = df['close'].rolling(window=252).min()  # 252 trading days in a year
    relative_strength = df['close'] / min_price_52_weeks
    
    # Multiply the smoothed momentum by the relative strength to create a more robust alpha factor
    enhanced_momentum = smoothed_momentum * relative_strength
    
    # Calculate the volume-weighted average price (VWAP) over the last 30 days
    vwap = (df['close'] * df['volume']).rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    
    # Adjust the final alpha factor value by multiplying it with the VWAP
    alpha_factor = enhanced_momentum * vwap
    
    return alpha_factor
```
