```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].shift(30) / df['close']
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].shift(5) / df['close']
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=25).std()
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Daily Close-to-Open Movement
    close_to_open_movement = df['open'] / df['close'].shift(1)
    
    # Smooth with Exponential Moving Average (EMA)
    ema_smoothed_return = df['close'].pct_change().ewm(span=7).mean()
    
    # Adjust for Volume and Amount
    summarized_volume = df['volume'].rolling(window=7).sum()
    summarized_amount = df['amount'].rolling(window=7).sum()
    volume_and_amount_combined = summarized_volume + summarized_amount
    momentum_to_volume_amount_ratio = ema_smoothed_return / volume_and_amount_combined
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff() > 0).astype(float) * df['close'].diff()
    losses = (df['close'].diff() < 0).astype(float) * df['close'].diff().abs()
    smoothed_gains = gains.ewm(span=14).mean()
    smoothed_losses = losses.ewm(span=14).mean()
    rtsi = 100 * (smoothed_gains / (smoothed_gains + smoothed_losses))
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.35 * combined_momentum +
        0.2 * intraday_spread +
        0.1 * close_to_open_movement +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.1 * rtsi
    )
    
    # Final Factor
    volume_acceleration = df['volume'].pct_change(periods=15)
    volume_change = df['volume'].pct_change()
    final_factor = weighted_average * volume_acceleration * (1 + volume_change) * momentum_to_volume_amount_ratio
    
    return final_factor
```
