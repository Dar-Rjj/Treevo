```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Movement
    close_to_open = df['close'] / df['open'] - 1
    
    # Calculate Daily Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Smooth with Exponential Moving Average (EMA)
    ema_window = 5
    ema_smoothed_returns = close_to_close_return.ewm(span=ema_window).mean()
    
    # Adjust for Volume and Amount
    total_volume = df['volume'].rolling(window=ema_window).sum()
    total_amount = df['amount'].rolling(window=ema_window).sum()
    combined_volume_amount = total_volume + total_amount
    volume_amount_adjusted_momentum = ema_smoothed_returns / combined_volume_amount
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff().clip(lower=0)).ewm(span=14).mean()
    losses = (-df['close'].diff().clip(upper=0)).ewm(span=14).mean()
    rtsi = 100 * (gains / (gains + losses))
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.3 * combined_momentum +
        0.2 * high_low_spread +
        0.1 * close_to_open +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Final Factor
    volume_acceleration = df['volume'].pct_change(periods=10)
    volume_change = df['volume'].pct_change()
    final_factor = weighted_average * (1 + volume_acceleration) * (1 + volume_change) * volume_amount_adjusted_momentum
    
    return final_factor
```
