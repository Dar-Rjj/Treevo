```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'] - df['open']
    
    # Calculate Exponential Moving Average Volume
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Generate Adjusted Price-Volume Signal
    df['adjusted_price_vol_signal'] = df['daily_price_change'] / df['ema_volume']
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = (df['close'] / df['close'].shift(28)) - 1
    
    # Calculate Medium-Term Price Momentum
    df['medium_term_momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = (df['close'] / df['close'].shift(10)) - 1
    
    # Combine Momentum Factors
    df['combined_momentum'] = df['long_term_momentum'] + df['medium_term_momentum'] + df['short_term_momentum']
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['close'].rolling(window=20).std()
    
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Adjust High-to-Low Range by Average Volume
    df['adjusted_high_low_range'] = df['high_low_range'] / df['ema_volume']
    
    # Weighted Average of Combined Components
    df['weighted_average'] = (df['combined_momentum'] * 0.3 +
                              df['high_low_range'] * 0.2 +
                              df['daily_price_change'] * 0.2 +
                              df['adjusted_high_low_range'] * 0.3)
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] / df['volume'].shift(15)) - 1
    
    # Final Alpha Signal Based on Combined Indicators
    threshold = 0.05
    median_volatility = df['realized_volatility'].median()
    
    conditions = [
        (df['adjusted_price_vol_signal'] > threshold) & (df['realized_volatility'] < median_volatility),
        (df['adjusted_price_vol_signal'] < -threshold) & (df['realized_volatility'] < median_volatility)
    ]
    choices = [1, -1]
    
    df['alpha_factor'] = np.select(conditions, choices, default=df['weighted_average'] * df['volume_acceleration'])
    
    return df['alpha_factor']
```
