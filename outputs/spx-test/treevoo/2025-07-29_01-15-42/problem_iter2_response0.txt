```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Smooth with Simple Moving Average (SMA)
    window_length = 5
    df['sma_close_return'] = df['close_return'].rolling(window=window_length).mean()
    
    # Adjust for Volume: Summarize Volume over the SMA period and Calculate Ratio of Momentum to Volume
    df['volume_summary'] = df['volume'].rolling(window=window_length).sum()
    df['volume_adjusted_momentum'] = df['sma_close_return'] / df['volume_summary']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    intraday_weight = 0.5
    open_to_close_weight = 0.5
    df['combined_return'] = (intraday_weight * df['intraday_return'] + 
                             open_to_close_weight * df['open_to_close_return'])
    
    # Adjust Combined Return by Volume
    df['volume_average'] = df['volume'].rolling(window=window_length).mean()
    df['volume_adjusted_combined_return'] = df['combined_return'] / df['volume_average']
    
    # Combine Volume Adjusted Momentum and Volume Adjusted Intraday & Open-to-Close Returns
    df['final_factor'] = df['volume_adjusted_momentum'] + df['volume_adjusted_combined_return']
    
    return df['final_factor'].dropna()
```
