```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Moving Average Volume
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    volume_adjusted_price_change = daily_price_change / moving_avg_volume
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Define threshold and median volatility
    threshold = 0.5  # Example threshold, can be adjusted
    median_volatility = realized_volatility.median()
    
    # Final Alpha Signal Based on Combined Indicators
    alpha_factor = pd.Series(index=df.index)
    for date in df.index:
        if volume_adjusted_price_change.loc[date] > threshold and realized_volatility.loc[date] < median_volatility:
            alpha_factor.loc[date] = 1
        elif volume_adjusted_price_change.loc[date] < -threshold and realized_volatility.loc[date] < median_volatility:
            alpha_factor.loc[date] = -1
        else:
            alpha_factor.loc[date] = 0
    
    return alpha_factor
```
