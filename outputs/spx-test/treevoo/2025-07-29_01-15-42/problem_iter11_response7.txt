```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the range of the day (high - low)
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the difference between high and close
    df['high_close_diff'] = df['high'] - df['close']
    
    # Calculate the difference between open and low
    df['open_low_diff'] = df['open'] - df['low']
    
    # Calculate the factor as a weighted sum of the calculated features
    factor = (df['close_pct_change'] * 0.4) + \
             (df['daily_range'] / df['close'] * 0.3) + \
             (df['high_close_diff'] / df['close'] * 0.2) + \
             (df['open_low_diff'] / df['close'] * 0.1)
    
    return factor.dropna()
```
