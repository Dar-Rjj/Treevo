```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(21)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(7)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Movement
    close_to_open_movement = (df['close'] - df['open']) / df['close']
    
    # Weighted Average of Momentum and Intraday Components
    weights = {
        'combined_momentum': 0.5,
        'intraday_high_low_spread': 0.3,
        'close_to_open_movement': 0.2
    }
    weighted_average = (
        combined_momentum * weights['combined_momentum'] +
        intraday_high_low_spread * weights['intraday_high_low_spread'] +
        close_to_open_movement * weights['close_to_open_movement']
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = df['volume'].pct_change(7)
    
    # Integrate Amount Data
    amount_acceleration = df['amount'].pct_change(7)
    
    # Multiply Weighted Average by Volume and Amount Acceleration
    factor = weighted_average * volume_acceleration * amount_acceleration
    
    return factor
```
