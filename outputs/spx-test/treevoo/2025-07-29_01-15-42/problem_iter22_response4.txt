```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Moving Average Volume (Last 20 Days)
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    adjusted_price_volume_signal = daily_price_change / moving_avg_volume
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = (df['close'] / df['close'].shift(28)) - 1
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(10)) - 1
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility (Close price with a rolling window of 20 days)
    close_returns = df['close'].pct_change()
    realized_volatility = close_returns.rolling(window=20).std() * np.sqrt(252)
    
    # Smooth with Exponential Moving Average (EMA) (5 days)
    ema_smoothed_return = close_returns.ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume
    volume_sum = df['volume'].rolling(window=5).sum()
    adjusted_ema_smoothed_return = ema_smoothed_return / volume_sum
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust High-to-Low Range by Average Volume (same rolling window of 20 days)
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    adjusted_high_low_range = high_low_range / avg_volume_20_days
    
    # Weighted Average of Combined Components
    weighted_combined_factor = (
        0.4 * combined_momentum +
        0.3 * high_low_range +
        0.2 * daily_price_change +
        0.1 * adjusted_high_low_range
    )
    
    # Calculate Volume Acceleration
    volume_acceleration = (df['volume'] / df['volume'].shift(10))
    
    # Set Thresholds
    threshold = 0.01
    median_combined_factor = weighted_combined_factor.median()
    median_realized_volatility = realized_volatility.median()
    
    # Final Alpha Signal Based on Combined Indicators
    alpha_factor = pd.Series(index=df.index, dtype='float64')
    for t in df.index:
        if adjusted_price_volume_signal[t] > threshold and weighted_combined_factor[t] > median_combined_factor and realized_volatility[t] < median_realized_volatility:
            alpha_factor[t] = 1
        elif adjusted_price_volume_signal[t] < -threshold and weighted_combined_factor[t] < median_combined_factor and realized_volatility[t] < median_realized_volatility:
            alpha_factor[t] = -1
        else:
            alpha_factor[t] = weighted_combined_factor[t] * volume_acceleration[t]
    
    return alpha_factor
```
