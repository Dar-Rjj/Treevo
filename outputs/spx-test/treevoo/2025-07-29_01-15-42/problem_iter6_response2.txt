```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = df['price_range'] / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    df['close_prev'] = df['close'].shift(1)
    df['gain'] = df['close'] - df['close_prev']
    df['gain_pos'] = df['gain'].apply(lambda x: max(x, 0))
    df['loss_neg'] = -df['gain'].apply(lambda x: min(x, 0))
    
    df['gain_ema'] = df['gain_pos'].rolling(window=14).mean()
    df['loss_ema'] = df['loss_neg'].rolling(window=14).mean()
    
    df['rtsi'] = (df['gain_ema'] / (df['gain_ema'] + df['loss_ema'])) * 100
    
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Smooth with Simple Moving Average (SMA)
    sma_window = 5
    df['sma_return'] = df['price_change'].rolling(window=sma_window).mean()
    
    # Adjust for Volume
    df['total_volume'] = df['volume'].rolling(window=sma_window).sum()
    df['vol_adj_momentum'] = df['sma_return'] / df['total_volume']
    
    # Introduce Trend Filter
    df['long_term_trend'] = df['close'].rolling(window=200).mean()
    df['trend_filter'] = (df['close'] > df['long_term_trend']).astype(int) * 2 - 1
    
    # Combine Factors
    df['alpha_factor'] = (df['price_range_ratio'] * df['rtsi'] * df['vol_adj_momentum'] * df['trend_filter'])
    
    return df['alpha_factor']
```
