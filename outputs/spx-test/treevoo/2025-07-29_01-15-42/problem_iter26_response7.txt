```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A series of factor values indexed by date.
    """
    
    # Calculate the relative difference between the close and open price
    rel_diff = (df['close'] - df['open']) / df['open']
    
    # Calculate the range of the day
    daily_range = (df['high'] - df['low']) / df['low']
    
    # Calculate the volume-to-range ratio
    vol_to_range_ratio = df['volume'] / daily_range
    
    # Generate the alpha factor
    factor = rel_diff * vol_to_range_ratio
    
    return factor
```
