```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].pct_change(28)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].pct_change(10)
    
    # Combine Long-Term and Short-Term Momentum
    combined_momentum = long_term_momentum + short_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].rolling(window=20).std()
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate EMA Smoothed Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    ema_close_to_open_return = close_to_open_return.ewm(span=10, adjust=False).mean()
    
    # Adjust for Volume and Amount
    total_volume = df['volume'].rolling(window=10).sum()
    total_amount = df['amount'].rolling(window=10).sum()
    volume_amount_adjustment = ema_close_to_open_return / (total_volume + total_amount)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['close']
    
    # Compute Relative True Strength Index (RTSI)
    gains = (df['close'].diff() > 0) * df['close'].diff()
    losses = (df['close'].diff() < 0) * abs(df['close'].diff())
    smoothed_gains = gains.ewm(span=14, adjust=False).mean()
    smoothed_losses = losses.ewm(span=14, adjust=False).mean()
    rtsi = 100 * (smoothed_gains / (smoothed_gains + smoothed_losses))
    
    # Calculate EMA Smoothed Daily Volume Change
    daily_volume_change = df['volume'].diff()
    ema_daily_volume_change = daily_volume_change.ewm(span=5, adjust=False).mean()
    
    # Calculate EMA Smoothed Daily Amount Change
    daily_amount_change = df['amount'].diff()
    ema_daily_amount_change = daily_amount_change.ewm(span=5, adjust=False).mean()
    
    # Weighted Average of Combined Components
    weighted_average = (
        0.3 * combined_momentum +
        0.2 * intraday_spread +
        0.1 * volume_amount_adjustment +
        0.1 * realized_volatility +
        0.15 * price_range_ratio +
        0.15 * rtsi
    )
    
    # Volume Acceleration
    volume_acceleration = (df['volume'] / df['volume'].shift(10)).fillna(1.0)
    
    # Adjust by Volume and Amount Change
    volume_change = df['volume'].diff().fillna(0)
    amount_change = df['amount'].diff().fillna(0)
    volume_amount_adjustment_factor = (volume_change + amount_change) / (df['volume'] + df['amount'])
    
    # Incorporate EMA Adjusted Momentum
    final_factor = weighted_average * volume_acceleration * (1 + volume_amount_adjustment_factor)
    
    return final_factor
```
