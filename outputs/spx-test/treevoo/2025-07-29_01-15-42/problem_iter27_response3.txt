```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open']
    
    # Smooth with EMA (10 days)
    ema_window = 10
    df['ema_close_to_open'] = df['close_to_open_return'].ewm(span=ema_window, adjust=False).mean()
    
    # Adjust for Volume and Amount
    df['total_volume_ema'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount_ema'] = df['amount'].rolling(window=ema_window).sum()
    df['adjusted_ema_return'] = df['ema_close_to_open'] / (df['total_volume_ema'] + df['total_amount_ema'])
    
    # Combine Long-Term and Short-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(28)) / df['close'].shift(28)
    short_term_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['combined_momentum'] = long_term_momentum + short_term_momentum
    
    # Calculate Price Range Ratio
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Compute Relative Strength Index (RSI) (14-period)
    df['diff'] = df['close'].diff()
    df['gain'] = df['diff'].apply(lambda x: max(x, 0))
    df['loss'] = df['diff'].apply(lambda x: -min(x, 0))
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rsi'] = 100 - (100 / (1 + (df['avg_gain'] / df['avg_loss'])))
    
    # Calculate Realized Volatility (20-day rolling window)
    df['realized_volatility'] = df['close'].rolling(window=20).std()
    
    # Integrate Volume Shock Factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_std'] = df['volume'].rolling(window=30).std()
    df['volume_shock_factor'] = df['volume_change'] / df['volume_std']
    
    # Weighted Average of Combined Components
    weights = {
        'combined_momentum': 0.3,
        'price_range_ratio': 0.2,
        'rsi': 0.1,
        'realized_volatility': 0.1,
        'volume_shock_factor': 0.3
    }
    
    df['weighted_average'] = (
        df['combined_momentum'] * weights['combined_momentum'] +
        df['price_range_ratio'] * weights['price_range_ratio'] +
        df['rsi'] * weights['rsi'] +
        df['realized_volatility'] * weights['realized_volatility'] +
        df['volume_shock_factor'] * weights['volume_shock_factor']
    )
    
    # Final Factor
    df['alpha_factor'] = df['weighted_average'] * df['adjusted_ema_return']
    
    return df['alpha_factor']
```
