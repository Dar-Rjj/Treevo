```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()

    # Smooth Returns with EMA (5 days)
    df['ema_close_return'] = df['close_to_close_return'].ewm(span=5, adjust=False).mean()

    # Adjust for Volume and Amount
    ema_window = 5
    df['total_volume'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount'] = df['amount'].rolling(window=ema_window).sum()
    df['adjusted_momentum_1'] = df['ema_close_return'] / (df['total_volume'] + df['total_amount'])

    # Calculate Daily Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open']

    # Smooth with EMA (10 days)
    df['ema_close_open_return'] = df['close_to_open_return'].ewm(span=10, adjust=False).mean()

    # Adjust for Volume and Amount
    ema_window = 10
    df['total_volume_2'] = df['volume'].rolling(window=ema_window).sum()
    df['total_amount_2'] = df['amount'].rolling(window=ema_window).sum()
    df['adjusted_momentum_2'] = df['ema_close_open_return'] / (df['total_volume_2'] + df['total_amount_2'])

    # Calculate Long-Term Price Momentum (t-28 to t)
    df['long_term_momentum'] = df['close'] / df['close'].shift(28) - 1

    # Calculate Short-Term Price Momentum (t-10 to t)
    df['short_term_momentum'] = df['close'] / df['close'].shift(10) - 1

    # Combine Long-Term and Short-Term Momentum
    df['combined_momentum'] = df['long_term_momentum'] + df['short_term_momentum']

    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Relative True Strength Index (RTSI)
    df['gain'] = np.where(df['close'] > df['close'].shift(), df['close'] - df['close'].shift(), 0)
    df['loss'] = np.where(df['close'] < df['close'].shift(), df['close'].shift() - df['close'], 0)
    df['ema_gain'] = df['gain'].ewm(span=14, adjust=False).mean()
    df['ema_loss'] = df['loss'].ewm(span=14, adjust=False).mean()
    df['rTSI'] = 100 * df['ema_gain'] / (df['ema_gain'] + df['ema_loss'])

    # Calculate Intraday Volatility (ATR)
    df['tr'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()

    # Calculate Realized Volatility (20 days)
    df['realized_volatility'] = df['close'].pct_change().rolling(window=20).std()

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Movement
    df['close_to_open_movement'] = df['close'] - df['open'].shift(-1)

    # Weighted Average of Combined Components
    df['weighted_average'] = (
        0.3 * df['combined_momentum'] +
        0.2 * df['intraday_high_low_spread'] +
        0.1 * df['close_to_open_movement'] +
        0.2 * (df['price_range'] / df['close']) +
        0.1 * df['rTSI'] +
        0.1 * df['realized_volatility']
    )

    # Adjust by Volume and Amount
    df['volume_change'] = df['volume'] / df['volume'].shift() - 1
    df['amount_change'] = df['amount'] / df['amount'].shift() - 1
    df['adjusted_factor'] = df['weighted_average'] * (df['volume_change'] + df['amount_change'])

    # Final Factor
    return df['adjusted_factor']
```
