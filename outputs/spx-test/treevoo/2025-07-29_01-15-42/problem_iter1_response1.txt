```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Price Range Ratio
    price_range_ratio = daily_price_range / df['open']
    
    # Compute Relative True Strength Index (RTSI)
    close_diff = df['close'].diff()
    gains = close_diff.where(close_diff > 0, 0).rolling(window=14).sum()
    losses = (-close_diff).where(close_diff < 0, 0).rolling(window=14).sum()
    smoothed_gains = gains.rolling(window=14).mean()
    smoothed_losses = losses.rolling(window=14).mean()
    rtsi = (smoothed_gains / (smoothed_gains + smoothed_losses)) * 100
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std() / df['volume'].rolling(window=20).mean()
    
    # Combine Factors
    alpha_factor = (price_range_ratio * rtsi) * volume_volatility
    
    return alpha_factor
```
