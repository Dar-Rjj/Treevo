```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['open']
    
    # Calculate Moving Average Volume
    df['MA_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Generate Adjusted Price-Volume Signal
    df['Price_Volume_Signal'] = df['Daily_Price_Change'] / df['MA_Volume_20']
    
    # Calculate Long-Term Price Momentum
    df['Long_Term_Momentum'] = (df['close'] / df['close'].shift(28)) - 1
    
    # Calculate Short-Term Price Momentum
    df['Short_Term_Momentum'] = (df['close'] / df['close'].shift(10)) - 1
    
    # Combine Long-Term and Short-Term Momentum
    df['Combined_Momentum'] = df['Long_Term_Momentum'] + df['Short_Term_Momentum']
    
    # Calculate Realized Volatility
    df['Realized_Volatility'] = df['close'].pct_change().rolling(window=20).std() * np.sqrt(252)
    
    # Calculate Daily Close-to-Close Return
    df['Close_to_Close_Return'] = df['close'].pct_change()
    
    # Smooth with Exponential Moving Average (EMA) over 5 days
    df['EMA_5'] = df['Close_to_Close_Return'].ewm(span=5, adjust=False).mean()
    
    # Summarize Volume over the EMA period
    df['Sum_Volume_5'] = df['volume'].rolling(window=5).sum()
    
    # Summarize Amount over the EMA period
    df['Sum_Amount_5'] = df['amount'].rolling(window=5).sum()
    
    # Calculate Ratio of Momentum to Combined Volume and Amount
    df['Combined_VA'] = df['Sum_Volume_5'] + df['Sum_Amount_5']
    df['Momentum_VA_Ratio'] = df['Combined_Momentum'] / df['Combined_VA']
    
    # Calculate High-to-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Adjust High-to-Low Range by Average Volume
    df['Adj_High_Low_Range'] = df['High_Low_Range'] / df['MA_Volume_20']
    
    # Weighted Average of Combined Components
    df['Weighted_Average'] = (0.4 * df['Combined_Momentum']) + (0.3 * df['Intraday_High_Low_Spread']) + (0.2 * df['Daily_Price_Change']) + (0.1 * df['Adj_High_Low_Range'])
    
    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['volume'] / df['volume'].shift(10)
    
    # Final Alpha Signal Based on Combined Indicators
    threshold = 0.01
    median_momentum = df['Combined_Momentum'].median()
    median_volatility = df['Realized_Volatility'].median()
    
    conditions = [
        (df['Price_Volume_Signal'] > threshold) & (df['Combined_Momentum'] > median_momentum) & (df['Realized_Volatility'] < median_volatility),
        (df['Price_Volume_Signal'] < -threshold) & (df['Combined_Momentum'] < median_momentum) & (df['Realized_Volatility'] < median_volatility)
    ]
    choices = [1, -1]
    
    df['Alpha_Factor'] = np.select(conditions, choices, default=df['Weighted_Average'] * df['Volume_Acceleration'])
    
    return df['Alpha_Factor']
```
