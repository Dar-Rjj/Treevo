```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Intraday Range Percentage
    range_percentage = intraday_range / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['close']
    
    # Combine Factors
    recent_trend = df['close'] - df['close'].rolling(window=20).mean()
    combined_factor = (volume_intensity * close_position) + recent_trend + range_percentage + momentum_shift
    
    return combined_factor
```
