import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement).astype(int)
    potential_reversal = ((open_close_movement > 0.6 * high_low_range) & (df['vwap'] < df['close'])).astype(int)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (high_low_range / df['close'])
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    n_period = 10  # Example period, can be adjusted
    volume_trend = df['volume'].rolling(window=n_period).sum()
    avg_volume = df['volume'].rolling(window=n_period).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Incorporate Amount and Volume Interaction
    amount_per_unit_volume = df['amount'] / df['volume']
    volume_weighted_price_movement = ((df['close'] - df['open']) / df['open']) * (df['amount'] / df['volume'])
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        weighted_ratio +
        positive_momentum * 0.5 +
        potential_reversal * 0.3 +
        intraday_range_percentage * 0.2 +
        close_open_momentum_shift * 0.4 +
        volume_activity_ratio * 0.6 +
        volume_weighted_price_movement * 0.7
    )
    
    return alpha_factor
