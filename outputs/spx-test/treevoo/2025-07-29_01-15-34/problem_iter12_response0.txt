```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Close-to-Open Momentum Shift
    close_to_open_momentum = (df['close'] - df['open']) / df['close'].shift(1)
    
    # Volume Activity Ratio
    volume_sum_n = df['volume'].rolling(window=n).sum()
    volume_ema_n = df['volume'].ewm(span=n, adjust=False).mean()
    volume_activity_ratio = volume_sum_n / volume_ema_n
    
    # Price Volatility Factor
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    log_return_volatility = daily_log_return.rolling(window=n).std()
    
    # Relative Strength Index (RSI)
    gain = np.maximum(0, df['close'].diff())
    loss = -np.minimum(0, df['close'].diff())
    avg_gain = gain.rolling(window=n).mean()
    avg_loss = loss.rolling(window=n).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        intraday_range +
        close_to_open_momentum +
        volume_activity_ratio +
        log_return_volatility +
        rsi
    ) / 5  # Normalize by the number of components
    
    return alpha_factor
```
