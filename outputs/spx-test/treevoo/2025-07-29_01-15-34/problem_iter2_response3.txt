```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close, multiplied by Volume
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = high_low_range > open_close_movement
    potential_reversal = open_close_movement > 0.6 * high_low_range
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=n).sum()
    average_volume = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_trend / average_volume
    
    # Combine all the components into a single factor
    factor = (
        weighted_ratio + 
        100 * intraday_range_percentage + 
        100 * close_open_momentum_shift + 
        10 * volume_activity_ratio
    )
    
    return factor
```
