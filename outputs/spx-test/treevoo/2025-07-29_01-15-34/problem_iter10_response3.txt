```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    close_to_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    volume_activity_ratio = df['volume'].rolling(window=n).sum() / df['volume'].rolling(window=n).mean()
    
    # Price Volatility Factor
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_volatility_factor = daily_return / daily_return.rolling(window=n).std()
    
    # Final Alpha Factor
    alpha_factor = (intraday_range + close_to_open_momentum) * volume_activity_ratio * price_volatility_factor
    
    return alpha_factor
```
