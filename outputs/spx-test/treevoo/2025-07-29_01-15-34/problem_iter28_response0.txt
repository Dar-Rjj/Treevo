import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute the ratio of current day's volume to the 20-day moving average volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Determine the range of the day and compare it with the previous day's range
    df['range'] = df['high'] - df['low']
    df['range_change'] = df['range'] - df['range'].shift(1)
    
    # Calculate the percentage of the day price spent above and below the opening price
    conditions = [
        (df['close'] > df['open']),
        (df['close'] < df['open'])
    ]
    choices_high = [1, 0]
    choices_low = [0, 1]
    df['above_open_pct'] = (df['high'] - df['open']) / (df['high'] - df['low'])
    df['below_open_pct'] = (df['open'] - df['low']) / (df['high'] - df['low'])
    
    # Compute where the closing price falls within the day's range
    df['close_position_in_range'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Evaluate the change in the amount traded from one day to another
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    # Create a composite factor by combining the above indicators
    # This is a simple average of normalized indicators for demonstration.
    # In practice, you might want to use more sophisticated combinations or machine learning models.
    df['composite_factor'] = (
        df['volume_ratio'] + 
        df['range_change'] + 
        df['above_open_pct'] * (df['above_open_pct'] > 0.5) + df['below_open_pct'] * (df['below_open_pct'] > 0.5) +
        df['close_position_in_range'] +
        df['amount_change']
    ) / 5
    
    return df['composite_factor']
