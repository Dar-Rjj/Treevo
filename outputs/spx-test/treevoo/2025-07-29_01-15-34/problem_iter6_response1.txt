import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = volume_trend - avg_volume
    
    # Calculate Intraday Volume Intensity
    past_10_day_vol_sum = df['volume'].rolling(window=10).sum()
    intraday_vol_intensity = df['volume'] / past_10_day_vol_sum
    
    # Combine Factors
    factor = (intraday_vol_intensity * close_position) + (df['close'] - df['close'].rolling(window=20).mean())
    
    return factor
