```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    volume_sum = df['volume'].rolling(window=n).sum()
    volume_ma = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_sum / volume_ma
    
    # Price Volatility Factor
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_return.rolling(window=n).std()
    price_volatility_factor = daily_return / volatility
    
    # Combine all factors
    alpha_factor = (intraday_range + momentum_shift + volume_activity_ratio + price_volatility_factor) / 4
    return alpha_factor
```
