```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / df['close']
    
    # Calculate Smoothed High-Low Price Momentum
    smoothed_momentum = (high_low_diff * adjusted_volume).rolling(window=5).mean()
    
    return smoothed_momentum
```
