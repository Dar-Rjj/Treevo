```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    weighted_ratio = intraday_range / (open_close_movement + 1e-6)
    trend_strength = (intraday_range > open_close_movement).astype(int)
    reversal_signal = (open_close_movement > 0.6 * intraday_range).astype(int)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    combined_factor = daily_log_return * volume_change
    
    # Compute Average of Combined Factor over 10 Days
    avg_combined_factor = combined_factor.rolling(window=10).mean()
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'] + 1e-6)
    
    # Adjust for Recent Price Trend using EMA
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend = df['close'] - ema_20
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Combine Factors
    combined_factors = (intraday_volume_intensity * closing_position) - recent_price_trend
    
    return combined_factors
```
