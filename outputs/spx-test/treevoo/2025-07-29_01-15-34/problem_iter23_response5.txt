def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    pd.Series: Series with index (date) representing the factor values.
    """
    # Calculate the percentage change in the closing price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the closing price and the average price
    price_diff = df['close'] - avg_price
    
    # Calculate the volume-weighted price difference
    volume_weighted_diff = price_diff * df['volume']
    
    # Calculate the 10-day moving sum of the volume-weighted price difference
    moving_sum = volume_weighted_diff.rolling(window=10).sum()
    
    # Calculate the final alpha factor
    alpha_factor = close_pct_change * moving_sum.shift(1)
    
    return alpha_factor
