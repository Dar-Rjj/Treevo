```python
def heuristics_v2(df):
    # Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio
    weighted_ratio = (intraday_range / open_close_movement) * df['volume']
    
    # Trend and Reversal Signals
    positive_momentum = (intraday_range > open_close_movement).astype(int)
    potential_reversal = ((open_close_movement > 0.6 * intraday_range)).astype(int)
    
    # Combine signals into a predictive factor
    alpha_factor = weighted_ratio * (positive_momentum - potential_reversal)
    
    return alpha_factor
```
