```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on (open, high, low, close) prices and volume.
    
    :param df: A pandas DataFrame with index (date) and columns (open, high, low, close, volume)
    :return: A pandas Series representing the alpha factor values
    """
    # Calculate price range
    price_range = df['high'] - df['low']
    
    # Calculate buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the balance of buying and selling pressure
    balance_pressure = buying_pressure - selling_pressure
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the volume ratio
    volume_ratio = df['volume'] / avg_volume
    
    # Combine the balance of pressure and volume ratio to create the final alpha factor
    alpha_factor = balance_pressure * volume_ratio
    
    return alpha_factor
```
