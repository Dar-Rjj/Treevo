```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Range Impact
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Volume Surge Ratio
    n_period = 10
    vol_sum = df['volume'].rolling(window=n_period).sum()
    vol_max = df['volume'].rolling(window=n_period).max()
    volume_surge_ratio = vol_sum / vol_max
    
    # Daily Return
    daily_return = (df['close'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Exponentially Weighted Moving Average of Daily Return Standard Deviation
    span = 30
    ewma_daily_return_std = daily_return.ewm(span=span, min_periods=span).std()
    
    # Combine all the components into a single alpha factor
    alpha_factor = (
        0.4 * intraday_range +
        0.3 * momentum_shift +
        0.2 * volume_surge_ratio -
        0.1 * ewma_daily_return_std
    )
    
    return alpha_factor
```
