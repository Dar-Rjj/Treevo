import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Price Change
    intraday_price_change = (df['close'] - df['open']) / intraday_range
    
    # Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    close_to_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    volume_activity_ratio = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    
    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_volatility = daily_return / daily_return.rolling(window=20).std()
    
    # Combine Intraday Components
    combined_intraday_components = (intraday_volume_intensity * closing_position) + intraday_price_change + close_to_open_momentum_shift
    
    # Adjust for Recent Price Trend
    recent_price_trend = df['close'].ewm(span=20).mean() - df['close']
    
    # Incorporate Daily Return Volatility
    final_factor = combined_intraday_components * daily_return_volatility + recent_price_trend
    
    return final_factor
