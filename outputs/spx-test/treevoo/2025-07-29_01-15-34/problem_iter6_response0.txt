```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement).astype(int)
    potential_reversal = (open_close_movement > 0.7 * high_low_range).astype(int)
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()  # N-period Sum
    average_volume = df['volume'].rolling(window=10).mean()  # N-period Moving Average
    volume_activity_ratio = volume_trend / average_volume
    
    # Incorporate Amount and Volume Interaction
    amount_per_unit_volume = df['amount'] / df['volume']
    volume_weighted_price_movement = ((df['close'] - df['open']) / df['open']) * amount_per_unit_volume
    
    # Integrated Alpha Factor
    alpha_factor = (
        weighted_ratio + 
        positive_momentum - 
        potential_reversal + 
        intraday_range_percentage + 
        close_open_momentum_shift + 
        volume_activity_ratio + 
        volume_weighted_price_movement
    )
    
    return alpha_factor
```
