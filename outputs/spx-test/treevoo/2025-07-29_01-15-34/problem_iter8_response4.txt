```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    df['Combined_Factor'] = df['Daily_Log_Return'] * df['Volume_Change']
    
    # Compute Average of Combined Factor over 10 Days
    df['Average_Combined_Factor_10D'] = df['Combined_Factor'].rolling(window=10).mean()
    
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    df['Open_to_Close_Movement'] = df['close'] - df['open']
    
    # Calculate Closing Position in Intraday Range
    df['Closing_Position'] = df['Open_to_Close_Movement'] / df['Intraday_High_Low_Range']
    
    # Calculate Intraday Volume Intensity
    df['Intraday_Volume_Intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Combine Intraday Factors
    df['EMA_20D'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Price_Trend_Adjustment'] = df['EMA_20D'] - df['close']
    df['Combined_Intraday_Factors'] = df['Intraday_Volume_Intensity'] * df['Closing_Position'] + df['Price_Trend_Adjustment']
    
    # Predictive Factor
    df['Weighted_Ratio'] = (df['Intraday_High_Low_Range'] / df['Open_to_Close_Movement']) * df['volume']
    df['Trend_Reversal_Signal'] = (df['Intraday_High_Low_Range'] > df['Open_to_Close_Movement']).astype(int) - (df['Open_to_Close_Movement'] > 0.6 * df['Intraday_High_Low_Range']).astype(int)
    df['Predictive_Factor'] = df['Weighted_Ratio'] * df['Trend_Reversal_Signal']
    
    # Integrate All Factors
    final_alpha_factor = (0.4 * df['Average_Combined_Factor_10D'] +
                          0.3 * df['Combined_Intraday_Factors'] +
                          0.3 * df['Predictive_Factor'])
    
    return final_alpha_factor
```
