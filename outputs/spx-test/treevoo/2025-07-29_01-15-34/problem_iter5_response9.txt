```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    df['combined_factor'] = df['daily_log_return'] * df['volume_change']
    
    # Compute Average of Combined Factor over 10 Days
    df['avg_combined_factor_10d'] = df['combined_factor'].rolling(window=10).mean()
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Calculate Intraday Volume Intensity
    df['volume_intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Combine Intraday Factors
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_adjustment'] = df['ema_20'] - df['close']
    df['intraday_factors'] = df['volume_intensity'] * df['closing_position'] + df['price_trend_adjustment']
    
    # Integrate All Factors
    df['alpha_factor'] = 0.5 * df['avg_combined_factor_10d'] + 0.5 * df['intraday_factors']
    
    return df['alpha_factor']
```
