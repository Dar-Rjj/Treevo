```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = high_low_range / np.abs(open_close_movement)
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > 1.5 * np.abs(open_close_movement)).astype(int)
    potential_reversal = ((np.abs(open_close_movement) > 0.8 * high_low_range) & (open_close_movement < 0)).astype(int)
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Volume Activity Ratio
    volume_sum = df['volume'].rolling(window=20).sum()
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_activity_ratio = volume_sum / volume_ma
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = df['close'] - ema_20
    
    # Calculate Short-Term Price Momentum
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_5 - ema_10
    
    # Composite Alpha Factor
    weighted_sum = (weighted_ratio * 0.3 + 
                    closing_position * 0.2 + 
                    volume_activity_ratio * 0.2 + 
                    short_term_momentum * 0.3)
    
    # Incorporate Recent Volatility
    recent_volatility = df['close'].rolling(window=20).std()
    trend_adjustment = ema_adjustment / recent_volatility
    
    # Final Alpha Factor
    alpha_factor = weighted_sum + trend_adjustment
    
    return alpha_factor
```
