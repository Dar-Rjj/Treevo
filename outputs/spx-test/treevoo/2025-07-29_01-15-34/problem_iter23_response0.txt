```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on open, high, low, close, and volume.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Calculate the normalized volume
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Heuristic Alpha Factor
    alpha_factor = (buying_pressure - selling_pressure) * volume_ratio
    
    return alpha_factor
```
