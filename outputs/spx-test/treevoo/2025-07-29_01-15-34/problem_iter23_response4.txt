def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
        
    Returns:
        pd.Series: Alpha factor values indexed by date
    """
    # Calculate the difference between the high and low price
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the volume-weighted price change
    vol_weighted_price_change = (df['close'] - df['open']) / df['volume']
    
    # Combine the three factors
    alpha_factor = (price_range * close_change) + vol_weighted_price_change
    
    return alpha_factor
