```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Volume Activity Ratio
    n = 5  # N-period for volume trend and moving average
    volume_trend = df['volume'].rolling(window=n).sum()
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Calculate Price Momentum Score
    short_term_momentum = df['close'].pct_change().rolling(window=5).sum()
    long_term_momentum = df['close'].pct_change(periods=7)  # Weekly return
    price_momentum_score = 0.6 * short_term_momentum + 0.4 * long_term_momentum
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_change = daily_volume_change.rolling(window=5).mean()
    long_term_volume_activity = df['volume'].rolling(window=20).sum()
    volume_activity_score = 0.6 * short_term_volume_change + 0.4 * long_term_volume_activity
    
    # Calculate Intraday Volume Intensity
    median_10_day_volume = df['volume'].rolling(window=10).median()
    intraday_volume_intensity = df['volume'] / median_10_day_volume
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend_adjustment = ema_20 - df['close']
    
    # Composite Alpha Factor
    weighted_sum_scores = 0.6 * price_momentum_score + 0.4 * volume_activity_score
    composite_alpha_factor = (intraday_volume_intensity * closing_position + 
                              weighted_sum_scores + recent_price_trend_adjustment)
    
    return composite_alpha_factor
```
