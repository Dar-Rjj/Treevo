```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum().rank(pct=True) * 100
    df['monthly_return'] = df['close'].pct_change(periods=20)
    df['long_term_momentum'] = df['monthly_return'].rank(pct=True) * 100
    df['price_momentum_score'] = (df['short_term_momentum'] + df['long_term_momentum']) / 2

    # Calculate Volume Activity Score
    df['volume_change'] = df['volume'].pct_change()
    df['short_term_volume'] = df['volume_change'].rolling(window=5).mean().rank(pct=True) * 100
    df['long_term_volume'] = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    df['volume_activity_score'] = (df['short_term_volume'] + df['long_term_volume']) / 2

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']

    # Adjust for Recent Price Trend
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_adjustment'] = df['ema_20'] - df['close']

    # Weighted Sum of Scores
    df['weighted_sum_scores'] = (df['price_momentum_score'] * 0.5 +
                                 df['volume_activity_score'] * 0.3 +
                                 df['intraday_range'] * 0.2)

    # Final Alpha Factor Combination
    df['combined_factors'] = (df['closing_position'] * df['volume_activity_score'] * df['price_trend_adjustment'])

    # Add a Sentiment Score (Assuming a sentiment score is available in the dataframe, else it can be computed)
    # Here we assume 'sentiment' is a column in the dataframe
    df['sentiment_score'] = df['sentiment'].rank(pct=True) * 100
    df['final_alpha_factor'] = (df['combined_factors'] * df['weighted_sum_scores'] +
                                df['sentiment_score'] * 0.2)

    return df['final_alpha_factor']
```
