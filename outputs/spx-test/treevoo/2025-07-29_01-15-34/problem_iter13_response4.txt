```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_high_low_range
    
    # Calculate Intraday Volume Intensity
    past_10_day_volume_sum = df['volume'].rolling(window=10).sum()
    intraday_volume_intensity = df['volume'] / past_10_day_volume_sum
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] / df['low']) / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum_shift = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()  # N-period Sum for Volume Trend
    average_volume = df['volume'].rolling(window=10).mean()  # N-period Moving Average for Volume
    volume_activity_ratio = volume_trend - average_volume
    
    # Combine Factors
    factor = (intraday_volume_intensity * closing_position) + \
             (df['close'] - df['close'].rolling(window=20).mean()) + \
             intraday_range_percentage + close_to_open_momentum_shift

    return factor
```
