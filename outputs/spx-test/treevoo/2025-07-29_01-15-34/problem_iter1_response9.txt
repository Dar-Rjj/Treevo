```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['volume_weighted_price'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Short-Term Price Trend
    short_term_trend = df['volume_weighted_price'].shift(1).rolling(window=3).mean() - df['volume_weighted_price'].shift(3)
    
    # Calculate Long-Term Price Trend
    long_term_trend = df['volume_weighted_price'].shift(1).rolling(window=16).mean() - df['volume_weighted_price'].shift(20)
    
    # Reversal Momentum
    reversal_momentum = (long_term_trend - short_term_trend) * df['volume_weighted_price']
    
    return reversal_momentum
```
