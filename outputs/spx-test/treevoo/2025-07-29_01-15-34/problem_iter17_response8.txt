```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates an alpha factor based on a heuristic approach.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    
    # Calculate the difference between the high and low price
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in closing price
    close_change = df['close'].pct_change()
    
    # Calculate the volume weighted average price change
    vwap_change = (df['close'] * df['volume']).pct_change() / df['volume'].pct_change()
    
    # Heuristic Alpha Factor
    alpha_factor = (price_range * close_change) + vwap_change
    
    return alpha_factor
```
