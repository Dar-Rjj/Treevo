```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    df['Combined_Factor'] = df['Daily_Log_Return'] * df['Volume_Change']
    
    # Compute Average of Combined Factor over 10 Days
    df['Avg_Combined_10D'] = df['Combined_Factor'].rolling(window=10).mean()
    
    # Calculate Intraday Range Percentage
    df['Intraday_Range_Percentage'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    df['Close_to_Open_Momentum'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    N = 10  # Number of periods for volume trend
    df['Volume_Trend'] = df['volume'].rolling(window=N).sum()
    df['Average_Volume'] = df['volume'].rolling(window=N).mean()
    df['Volume_Activity_Ratio'] = df['Volume_Trend'] / df['Average_Volume']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Avg_Combined_10D'] * df['Intraday_Range_Percentage'] * df['Close_to_Open_Momentum'] * df['Volume_Activity_Ratio']
    
    return df['Alpha_Factor']
```
