```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on the thought tree.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index (date) and columns ['open', 'high', 'low', 'close', 'volume'].
        
    Returns:
        pd.Series: A Series with index (date) representing the alpha factor values.
    """
    
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the range (high - low)
    price_range = df['high'] - df['low']
    
    # Calculate the normalized volume (volume / avg_price)
    norm_volume = df['volume'] / avg_price
    
    # Heuristic factor: (avg_price * norm_volume) / price_range
    factor = (avg_price * norm_volume) / price_range
    
    return factor
```
