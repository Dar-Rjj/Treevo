import pandas as pd

def heuristics_v2(df):
    """
    The function heuristics_v2 takes a DataFrame with columns: open, high, low, close, and volume.
    It returns a Series with the calculated alpha factor.

    :param df: pandas DataFrame with market data
    :return: pandas Series with the calculated alpha factor
    """
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow, using the typical price as a proxy for the 'amount'
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flows
    positive_money_flow = money_flow[df['close'] > df['close'].shift(1)]
    negative_money_flow = money_flow[df['close'] < df['close'].shift(1)]
    
    # Calculate the money flow ratio
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    
    # Calculate the alpha factor
    alpha_factor = 100 - (100 / (1 + money_flow_ratio))
    
    return alpha_factor
