```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['Daily_Return'] = df['close'].pct_change()
    df['Short_Term_Momentum'] = df['Daily_Return'].rolling(window=5).sum().rank(pct=True) * 100
    df['Monthly_Return'] = df['close'].pct_change(periods=20)
    df['Long_Term_Momentum'] = df['Monthly_Return'].rank(pct=True) * 100
    df['Price_Momentum_Score'] = (df['Short_Term_Momentum'] * 0.5 + df['Long_Term_Momentum'] * 0.5)

    # Calculate Volume Activity Score
    df['Daily_Volume_Change'] = df['volume'].pct_change()
    df['Short_Term_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=5).mean().rank(pct=True) * 100
    df['Long_Term_Volume_Activity'] = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    df['Volume_Activity_Score'] = (df['Short_Term_Volume_Change'] * 0.5 + df['Long_Term_Volume_Activity'] * 0.5)

    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    df['Closing_Position'] = (df['close'] - df['open']) / df['Intraday_High_Low_Range']

    # Calculate Intraday Volume Intensity
    df['Average_10Day_Volume'] = df['volume'].rolling(window=10).mean()
    df['Intraday_Volume_Intensity'] = df['volume'] / df['Average_10Day_Volume']

    # Adjust for Recent Price Trend
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Recent_Price_Trend'] = df['EMA_20'] - df['close']

    # Combine Factors
    df['Adjusted_Intraday_Volume'] = df['Intraday_Volume_Intensity'] * df['Closing_Position']
    df['Combined_Factors'] = df['Adjusted_Intraday_Volume'] + df['Recent_Price_Trend']
    
    # Weighted Sum of Scores
    df['Weighted_Scores'] = (df['Price_Momentum_Score'] * 0.6 + 
                             df['Volume_Activity_Score'] * 0.3 + 
                             df['Intraday_High_Low_Range'] * 0.1)

    # Final Alpha Factor Combination
    df['Final_Alpha_Factor'] = df['Combined_Factors'] * df['Weighted_Scores']

    return df['Final_Alpha_Factor']
```
