```python
import pandas as pd

def heuristics_v2(df):
    # Simple Moving Averages
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    
    # Momentum Indicators
    df['short_term_momentum'] = df['5_day_SMA'] - df['20_day_SMA']
    df['medium_term_momentum'] = df['20_day_SMA'] - df['50_day_SMA']

    # Rate of Change (RoC) Indicators
    df['12_day_RoC'] = df['close'].pct_change(periods=12)
    df['26_day_RoC'] = df['close'].pct_change(periods=26)

    # On-Balance Volume (OBV) Trend Analysis
    obv = (df['volume'] * (df['close'] > df['close'].shift())).cumsum()
    df['OBV_14_day_slope'] = obv.rolling(window=14).apply(lambda x: (x[-1] - x[0]) / 14, raw=True)

    # Volume-Price Trend (VPT) Analysis
    vpt = ((df['close'] - df['close'].shift()) / df['close'].shift()) * df['volume']
    df['VPT_20_day_diff_from_10MA'] = vpt - vpt.rolling(window=10).mean()

    # Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['21_day_CMF'] = money_flow_volume.rolling(window=21).sum() / df['volume'].rolling(window=21).sum()

    # Inside Bar Pattern Detection
    inside_bar_count = 0
    for i in range(1, len(df)):
        if (df.loc[df.index[i], 'high'] <= df.loc[df.index[i-1], 'high']) and (df.loc[df.index[i], 'low'] >= df.loc[df.index[i-1], 'low']):
            inside_bar_count += 1
        else:
            inside_bar_count = 0
        df.loc[df.index[i], 'inside_bar_count'] = inside_bar_count

    # Engulfing Candlestick Pattern
    df['engulfing_pattern'] = (df['open'] < df['close'].shift()) & (df['close'] > df['open'].shift()) & (df['close'] > df['open'].shift()) | \
                              (df['open'] > df['close'].shift()) & (df['close'] < df['open'].shift()) & (df['close'] < df['open'].shift())

    # True Range (TR) Based Volatility Indicator
    tr = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift()), abs(df['low'] - df['close'].shift())], axis=1).max(axis=1)
    df['14_day_ATR'] = tr.rolling(window=14).mean()

    # Bollinger Bands Width
    df['20_day_BB_middle'] = df['close'].rolling(window=20).mean()
    df['20_day_BB_std'] = df['close'].rolling(window=20).std()
    df['20_day_BB_upper'] = df['20_day_BB_middle'] + 2 * df['20_day_BB_std']
    df['20_day_BB_lower'] = df['20_day_BB_middle'] - 2 * df['20_day_BB_std']
    df['BB_width'] = (df['20_day_BB_upper'] - df['20_day_BB_lower']) / df['20_day_BB_middle']

    # Alpha Factor
    alpha_factor = (df['short_term_momentum'] + df['medium_term_momentum'] + df['12_day_RoC'] + df['26_day_RoC'] + 
                    df['OBV_14_day_slope'] + df['VPT_20_day_diff_from_10MA'] + df['21_day_CMF'] + 
                    df['inside_bar_count'] + df['engulfing_pattern'] + df['14_day_ATR'] + df['BB_width']).fillna(0)
    
    return alpha_factor
```
