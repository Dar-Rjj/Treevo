```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    volume_sum = df['volume'].rolling(window=n).sum()
    volume_ma = df['volume'].rolling(window=n).mean()
    volume_ratio = volume_sum / volume_ma
    
    # Price Volatility Factor
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_std = daily_return.rolling(window=n).std()
    volatility_factor = daily_return / daily_return_std
    ema_daily_return = daily_return.ewm(span=n, adjust=False).mean()
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        0.25 * intraday_range + 
        0.25 * momentum_shift + 
        0.25 * volume_ratio + 
        0.25 * ema_daily_return
    )
    
    return alpha_factor
```
