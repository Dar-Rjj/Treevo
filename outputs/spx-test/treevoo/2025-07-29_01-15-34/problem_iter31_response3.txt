```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['sum_daily_returns_5d'] = df['daily_return'].rolling(window=5).sum()
    df['quarterly_return'] = df['close'].pct_change(periods=60)  # Assuming 60 days for a quarter
    df['price_momentum_score'] = 0.5 * df['sum_daily_returns_5d'] + 0.5 * df['quarterly_return']

    # Calculate Volume Activity Score
    df['volume_change'] = df['volume'].pct_change()
    df['avg_volume_change_5d'] = df['volume_change'].rolling(window=5).mean()
    df['sum_volume_30d'] = df['volume'].rolling(window=30).sum()
    df['volume_activity_score'] = 0.5 * df['avg_volume_change_5d'] + 0.5 * (df['sum_volume_30d'] / df['volume'])

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']

    # Combine Factors
    df['intraday_volume_intensity'] = df['volume'] / df['intraday_range']
    df['combined_factors'] = df['intraday_volume_intensity'] * df['closing_position']

    # Adjust for Recent Price Trend
    df['ema_20d'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_adjustment'] = df['ema_20d'] - df['close']

    # Weighted Sum of Scores
    df['weighted_sum_scores'] = 0.6 * df['price_momentum_score'] + 0.3 * df['volume_activity_score'] + 0.1 * df['intraday_range']

    # Final Alpha Factor Combination
    df['final_alpha_factor'] = (df['combined_factors'] * df['weighted_sum_scores']) + df['price_trend_adjustment']

    # Add a Sentiment Score (assuming a 'sentiment' column exists in the DataFrame)
    df['final_alpha_factor'] += df['sentiment']

    return df['final_alpha_factor']
```
