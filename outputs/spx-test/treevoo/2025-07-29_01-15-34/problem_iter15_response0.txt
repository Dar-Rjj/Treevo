```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()
    df['monthly_return'] = df['close'].pct_change(periods=20)
    
    # Convert to a Rank (0-100)
    df['short_term_momentum_rank'] = (df['short_term_momentum'].rank(pct=True) * 100).clip(0, 100)
    df['long_term_momentum_rank'] = (df['monthly_return'].rank(pct=True) * 100).clip(0, 100)
    
    # Combine short-term and long-term momentum
    df['price_momentum_score'] = df['short_term_momentum_rank'] + df['long_term_momentum_rank']
    
    # Calculate Volume Activity Score
    df['daily_volume_change'] = df['volume'].pct_change()
    df['avg_daily_volume_change'] = df['daily_volume_change'].rolling(window=5).mean()
    df['long_term_volume_sum'] = df['volume'].rolling(window=20).sum()
    
    # Convert to a Rank (0-100)
    df['short_term_volume_activity_rank'] = (df['avg_daily_volume_change'].rank(pct=True) * 100).clip(0, 100)
    df['long_term_volume_activity_rank'] = (df['long_term_volume_sum'].rank(pct=True) * 100).clip(0, 100)
    
    # Combine short-term and long-term volume activity
    df['volume_activity_score'] = df['short_term_volume_activity_rank'] + df['long_term_volume_activity_rank']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Intraday Volume Intensity
    avg_10d_vol = df['volume'].rolling(window=10).mean()
    df['intraday_volume_intensity'] = df['volume'] / avg_10d_vol
    
    # Adjust for Recent Price Trend
    df['ema_20'] = df['close'].ewm(span=20).mean()
    df['recent_price_trend'] = df['close'] - df['ema_20']
    
    # Weighted Sum of Scores
    df['weighted_scores'] = (df['price_momentum_score'] * 0.6 +
                             df['volume_activity_score'] * 0.3 +
                             df['intraday_range'] * 0.1)
    
    # Final Alpha Factor Combination
    df['combined_factors'] = (df['intraday_volume_intensity'] * df['closing_position'] +
                              df['recent_price_trend'])
    
    # Add a Sentiment Score
    df['sentiment_score'] = 50  # Placeholder for sentiment score calculation
    df['final_alpha_factor'] = (df['combined_factors'] * df['weighted_scores'] +
                                df['sentiment_score'] * 0.2)
    
    return df['final_alpha_factor']
```
