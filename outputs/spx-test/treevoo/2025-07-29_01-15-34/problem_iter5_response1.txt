```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = high_low_range / (open_close_movement + 1e-8)  # Avoid division by zero
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement).astype(int)
    potential_reversal = ((open_close_movement > 0.6 * high_low_range) & (open_close_movement > 0)).astype(int)
    
    # Combine Indicators for Predictive Factor
    intraday_factor = positive_momentum * weighted_ratio + potential_reversal * (1 - weighted_ratio)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    combined_factor = daily_log_return * volume_change
    
    # Compute Average of Combined Factor over 10 Days
    avg_combined_factor = combined_factor.rolling(window=10, min_periods=1).mean()
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10, min_periods=1).sum()
    avg_volume = df['volume'].rolling(window=10, min_periods=1).mean()
    volume_activity_ratio = volume_trend / (avg_volume + 1e-8)  # Avoid division by zero
    
    # Final Alpha Factor
    alpha_factor = (
        intraday_factor * 0.3 +
        avg_combined_factor * 0.4 +
        intraday_range_percentage * 0.1 +
        close_open_momentum * 0.1 +
        volume_activity_ratio * 0.1
    )
    
    return alpha_factor
```
