```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    high_low_open_close_ratio = intraday_high_low / open_close
    positive_momentum = (intraday_high_low > open_close).astype(int)
    potential_reversal = (open_close > 0.6 * intraday_high_low).astype(int)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    combined_factor = daily_log_return * volume_change
    
    # Compute Average of Combined Factor over 10 Days
    avg_combined_factor = combined_factor.rolling(window=10).mean()
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    n_period_sum_volume = df['volume'].rolling(window=10).sum()
    n_period_avg_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = n_period_sum_volume / n_period_avg_volume
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Adjust for Recent Price Trend
    recent_price_trend = df['close'] - df['close'].ewm(span=220).mean()
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Price Change
    intraday_price_change = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_volatility = daily_return / daily_return.rolling(window=10).std()
    
    # Combine Intraday Components
    combined_intraday_components = (
        intraday_volume_intensity * closing_position + 
        intraday_price_change + 
        close_open_momentum_shift
    )
    
    # Incorporate Daily Return Volatility
    adjusted_intraday_components = combined_intraday_components * daily_return_volatility + recent_price_trend
    
    # Final Factor
    final_factor = (
        combined_intraday_components * recent_price_trend + 
        avg_combined_factor
    )
    
    return final_factor
```
