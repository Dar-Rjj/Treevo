```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Price Movement Analysis
    df['High_Low_Range'] = df['high'] - df['low']
    df['Open_Close_Movement'] = df['close'] - df['open']
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['VWAP'] = (df['volume'] * df['Typical_Price']).cumsum() / df['volume'].cumsum()
    df['High_Low_Open_Close_Ratio'] = df['High_Low_Range'] / df['Open_Close_Movement'].abs()
    df['VWAP_Deviation'] = df['close'] - df['VWAP']
    
    # Trend and Reversal Signals
    df['Positive_Momentum'] = (df['High_Low_Range'] > 1.5 * df['Open_Close_Movement'])
    df['Potential_Reversal'] = ((df['Open_Close_Movement'] > 0.8 * df['High_Low_Range']) & (df['VWAP_Deviation'] < 0) & (df['Open_Close_Movement'] < 0))
    
    # Intraday Range Analysis
    df['Range_Percentage'] = (df['high'] - df['low']) / df['close']
    df['Closing_Position'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Volume Analysis
    df['Volume_Intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()
    df['Volume_Trend'] = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=20).mean()
    
    # Recent Price Trend Adjustment
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Close_to_EMA'] = df['close'] - df['EMA_20']
    
    # Close-to-Open Momentum Shift
    df['Close_to_Open_Momentum'] = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Intraday Range Percentage Enhanced
    df['Intraday_Range_Percentage_Enhanced'] = (df['high'] - df['low']) / df['close']
    
    # Adjust for Recent Price Trend
    df['Close_to_EMA_Adjusted'] = df['close'] - df['EMA_20']
    
    # Composite Alpha Factor
    df['Price_Momentum_Score'] = df['close'].pct_change().fillna(0).rolling(window=5).sum()
    df['Volume_Activity_Score'] = df['volume'].pct_change().fillna(0).rolling(window=5).mean()
    df['Short_term_Price_Movement'] = df['close'].pct_change().fillna(0).rolling(window=5).sum()
    df['Long_term_Price_Movement'] = df['close'].pct_change(periods=5).fillna(0)
    df['Short_term_Volume_Change'] = df['volume'].pct_change().fillna(0).rolling(window=5).mean()
    df['Long_term_Volume_Activity'] = df['volume'].rolling(window=20).sum()
    
    df['Weighted_Sum_Scores'] = 0.6 * df['Price_Momentum_Score'] + 0.4 * df['Volume_Activity_Score']
    df['Intraday_Volume_Intensity_Closing_Position'] = df['Volume_Intensity'] * df['Closing_Position']
    df['Final_Alpha_Factor'] = (df['Weighted_Sum_Scores'] 
                                + df['Intraday_Volume_Intensity_Closing_Position'] 
                                + df['Intraday_Range_Percentage_Enhanced'] 
                                + df['Close_to_Open_Momentum'])
    
    return df['Final_Alpha_Factor']
```
