import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Volume Activity Ratio
    n_period_volume_sum = df['volume'].rolling(window=20).sum()
    n_period_volume_ma = df['volume'].rolling(window=20).mean()
    volume_activity_ratio = n_period_volume_sum / n_period_volume_ma
    
    # Intraday Volume Intensity
    avg_10_day_volume = df['volume'].rolling(window=10).mean()
    intraday_volume_intensity = df['volume'] / avg_10_day_volume
    
    # Price Volatility Factor
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    n_period_std_dev = daily_return.rolling(window=20).std()
    price_volatility_factor = daily_return / n_period_std_dev
    
    # Combine Factors
    factor_product = intraday_volume_intensity * closing_position
    recent_price_trend = df['close'].ewm(span=20, adjust=False).mean() - df['close']
    combined_factor = factor_product + recent_price_trend

    return combined_factor
