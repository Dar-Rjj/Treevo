```python
def heuristics_v2(df):
    # Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Predictive Factor
    weighted_ratio = (intraday_high_low_range / open_close_movement) * df['volume']
    
    # Trend and Reversal Signals
    positive_momentum_signal = (intraday_high_low_range > open_close_movement).astype(int)
    potential_reversal_signal = (open_close_movement > 0.6 * intraday_high_low_range).astype(int)
    
    # Combining signals with the weighted ratio
    factor = weighted_ratio * (positive_momentum_signal - potential_reversal_signal)
    
    return factor
```
