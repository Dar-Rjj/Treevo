```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Range Percentage
    df['intraday_range'] = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Close-to-Open Momentum Shift
    df['close_to_open_momentum'] = (df['close'] - df['open']) / df['close'].shift(1)
    
    # Volume Activity Ratio
    n = 10
    df['volume_sum_n'] = df['volume'].rolling(window=n).sum()
    df['volume_ema_n'] = df['volume'].ewm(span=n, adjust=False).mean()
    df['volume_activity_ratio'] = df['volume_sum_n'] / df['volume_ema_n']
    
    # Price Volatility Factor
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility_factor'] = df['log_return'].rolling(window=n).std()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=n).mean()
    avg_loss = loss.rolling(window=n).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Range Percentage Enhanced
    df['intraday_range_enhanced'] = (df['high'] - df['low']) / df['open']
    
    # Close-to-Open Momentum Shift Enhanced
    df['close_to_open_momentum_enhanced'] = (df['close'] - df['open']) / df['close']
    
    # Adjust for Recent Price Trend
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = df['close'] - df['ema_20']
    
    # Composite Alpha Factor Components
    df['short_term_price_movement'] = df['log_return'].rolling(window=5).sum()
    df['long_term_price_movement'] = np.log(df['close'] / df['close'].shift(20))
    df['short_term_volume_change'] = (df['volume'] - df['volume'].shift(1)).rolling(window=5).mean()
    df['long_term_volume_activity'] = df['volume'].rolling(window=20).sum()
    
    # Weighted Sum of Scores
    price_momentum_score = 0.6 * (df['short_term_price_movement'] + df['long_term_price_movement'])
    volume_activity_score = 0.4 * (df['short_term_volume_change'] + df['long_term_volume_activity'])
    composite_alpha = price_momentum_score + volume_activity_score
    
    # Final Alpha Factor
    df['final_alpha'] = (
        df['intraday_volume_intensity'] * df['closing_position'] +
        df['intraday_range_enhanced'] +
        df['close_to_open_momentum_enhanced'] +
        composite_alpha
    )
    
    return df['final_alpha']
```
