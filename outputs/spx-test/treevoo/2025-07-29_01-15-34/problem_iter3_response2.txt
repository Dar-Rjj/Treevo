```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['open']

    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=n).sum()
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_trend / avg_volume

    # Incorporate Price Volatility Factor
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    return_volatility = daily_return.rolling(window=n).std()
    normalized_volatility = daily_return / return_volatility

    # Combine all factors into a single alpha factor
    alpha_factor = (
        intraday_range + 
        momentum_shift + 
        volume_activity_ratio + 
        normalized_volatility
    )

    return alpha_factor
```
