```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Volume Adjusted Close Price
    volume_adjusted_close = df['close'] * df['volume']
    
    # Integrate with 10-day Moving Average of Close
    close_10_day_ma = df['close'].rolling(window=10).mean()
    integrated_close = (volume_adjusted_close + close_10_day_ma) / 2
    
    # Scale High-to-Low Range by Integrated Close
    scaled_high_low_range = high_low_range / integrated_close
    
    # Calculate 10-day Return
    ten_day_return = df['close'].pct_change(periods=10)
    
    # Integrate Momentum with Scaled High-to-Low Range
    alpha_factor = scaled_high_low_range * ten_day_return
    
    return alpha_factor
```
