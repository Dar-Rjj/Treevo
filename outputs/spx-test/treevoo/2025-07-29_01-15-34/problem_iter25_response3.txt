import pandas as pd

def heuristics_v2(df):
    # Intraday Price Movement Analysis
    high_low_range = df['high'] - df['low']
    open_close_movement = df['close'] - df['open']
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    high_low_to_open_close_ratio = high_low_range / open_close_movement.abs()
    vwap_deviation = df['close'] - vwap
    positive_momentum = (high_low_range > 1.5 * open_close_movement).astype(int)
    potential_reversal = ((open_close_movement > 0.8 * high_low_range) & (vwap_deviation < 0) & (open_close_movement < 0)).astype(int)
    
    # Intraday Range Analysis
    range_percentage = (df['high'] - df['low']) / df['close']
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Volume Analysis
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).median()
    volume_trend = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    
    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = df['close'] - ema_20
    
    # Additional Price Analysis
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    ma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = ma_20 + 2 * std_20
    lower_band = ma_20 - 2 * std_20
    bollinger_band_signal = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Composite Alpha Factor
    weighted_sum = (high_low_to_open_close_ratio * 0.3 +
                    vwap_deviation * 0.3 +
                    volume_intensity * 0.2 +
                    closing_position * 0.1 +
                    rsi * 0.1)
    final_alpha = weighted_sum + ema_adjustment + bollinger_band_signal
    
    return final_alpha
