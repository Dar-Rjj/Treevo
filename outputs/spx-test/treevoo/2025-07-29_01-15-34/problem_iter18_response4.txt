```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum().rank(pct=True) * 100
    df['monthly_return'] = df['close'].pct_change(periods=20)
    df['long_term_momentum'] = df['monthly_return'].rank(pct=True) * 100
    df['price_momentum_score'] = (df['short_term_momentum'] + df['long_term_momentum']) / 2

    # Calculate Volume Activity Score
    df['daily_volume_change'] = df['volume'].pct_change()
    df['avg_daily_volume_change'] = df['daily_volume_change'].rolling(window=5).mean().rank(pct=True) * 100
    df['sum_20d_volume'] = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    df['volume_activity_score'] = (df['avg_daily_volume_change'] + df['sum_20d_volume']) / 2

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']

    # Combine Factors
    df['intraday_volume_intensity'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_adjustment'] = df['close'] - df['ema_20']
    
    df['weighted_sum_scores'] = (df['price_momentum_score'] * 0.5 +
                                 df['volume_activity_score'] * 0.3 +
                                 df['intraday_range'] * 0.2)

    df['combined_factors'] = (df['intraday_volume_intensity'] * df['closing_position'] +
                              df['price_trend_adjustment'])

    # Final Alpha Factor Combination
    df['sentiment_score'] = np.random.randint(0, 100, size=len(df))  # Placeholder for actual sentiment score
    df['final_alpha_factor'] = (df['combined_factors'] * df['weighted_sum_scores'] +
                                df['sentiment_score'] * 0.2)

    return df['final_alpha_factor']
```
