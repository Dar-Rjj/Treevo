```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    weighted_ratio = high_low_range / open_close_movement
    trend_strength = (high_low_range > open_close_movement).astype(int)
    potential_reversal = (open_close_movement > 0.6 * high_low_range).astype(int)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    price_momentum_volume = daily_log_return * volume_change
    
    # Compute Average of Combined Factor over 10 Days
    average_factor_10d = price_momentum_volume.rolling(window=10).mean()
    
    # Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Intraday Volume Intensity
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_intensity = df['volume'] / avg_volume_10d
    
    # Price Volatility Factor
    daily_return_std = daily_log_return.rolling(window=10).std()
    price_volatility_factor = daily_log_return / daily_return_std
    
    # Adjust for Recent Price Trend
    recent_trend_adjustment = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Volume Spike Indicator
    volume_spike = (df['volume'] - df['volume'].shift(1)) / avg_volume_10d
    
    # Integrate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Intraday Components
    intraday_components = (volume_intensity * closing_position) + intraday_range_percentage + close_open_momentum_shift
    
    # Incorporate Daily Return Volatility
    adjusted_intraday_components = intraday_components * price_volatility_factor + recent_trend_adjustment
    
    # Final Factor
    final_factor = adjusted_intraday_components + average_factor_10d * rsi
    
    return final_factor
```
