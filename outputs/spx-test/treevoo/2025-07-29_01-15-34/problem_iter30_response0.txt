```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['Daily_Return'] = df['close'].pct_change()
    short_term_return = df['Daily_Return'].rolling(window=5).sum()
    long_term_return = df['close'].pct_change(periods=20)
    
    short_term_rank = (short_term_return.rank(pct=True) * 100).fillna(0)
    long_term_rank = (long_term_return.rank(pct=True) * 100).fillna(0)
    
    price_momentum_score = (short_term_rank + long_term_rank) / 2
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change().fillna(0)
    avg_daily_volume_change = daily_volume_change.rolling(window=5).mean()
    sum_20_day_volume = df['volume'].rolling(window=20).sum()
    
    short_term_volume_rank = (avg_daily_volume_change.rank(pct=True) * 100).fillna(0)
    long_term_volume_rank = (sum_20_day_volume.rank(pct=True) * 100).fillna(0)
    
    volume_activity_score = (short_term_volume_rank + long_term_volume_rank) / 2
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position_in_range = (df['close'] - df['open']) / intraday_high_low_range
    
    # Calculate Intraday Volume Intensity
    sum_10_day_volume = df['volume'].rolling(window=10).sum()
    intraday_volume_intensity = df['volume'] / sum_10_day_volume
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend = df['close'] - ema_20
    
    # Combine Factors
    combined_factors = (
        intraday_volume_intensity * close_position_in_range +
        recent_price_trend
    )
    
    # Weighted Sum of Scores
    weighted_scores = (
        price_momentum_score * 0.6 +
        volume_activity_score * 0.3 +
        intraday_high_low_range * 0.1
    )
    
    # Final Alpha Factor Combination
    final_alpha_factor = (
        combined_factors * weighted_scores
    )
    
    # Add a Sentiment Score (assuming sentiment scores are available)
    if 'sentiment' in df.columns:
        sentiment_rank = (df['sentiment'].rank(pct=True) * 100).fillna(0)
        final_alpha_factor += sentiment_rank * 0.2
    
    return final_alpha_factor
```
