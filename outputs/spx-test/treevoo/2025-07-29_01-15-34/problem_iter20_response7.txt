import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Intraday Range Intensity
    range_intensity = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Open-to-Close Momentum Shift
    open_to_close_momentum = (df['close'] - df['open']) / df['close'].shift(1)
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=n).sum()
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Volume Activity Ratio
    volume_n_period_sum = df['volume'].rolling(window=n).sum()
    volume_ema = df['volume'].ewm(span=n, adjust=False).mean()
    volume_act_ratio = volume_n_period_sum / volume_ema
    
    # Calculate Price Momentum Score
    daily_return = df['close'].pct_change()
    short_term_momentum = daily_return.rolling(window=5).sum()
    weekly_return = df['close'].resample('W').ffill().pct_change()
    price_momentum_score = 0.6 * short_term_momentum + 0.4 * weekly_return
    
    # Daily Return Volatility Adjusted
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    atr = df['high'] - df['low']
    daily_vol_adjusted_return = daily_log_return / atr.rolling(window=n).mean()
    
    # Price Volatility Factor
    daily_log_return_vol = np.log(df['close'] / df['close'].shift(1)).rolling(window=n).std()
    daily_gain = np.maximum(0, df['close'] - df['close'].shift(1))
    daily_loss = np.minimum(0, df['close'] - df['close'].shift(1))
    avg_gain = daily_gain.rolling(window=n).mean()
    avg_loss = -daily_loss.rolling(window=n).mean()
    rsi = 100 - (100 / (1 + (avg_gain / avg_loss)))
    
    # Composite Alpha Factor
    price_momentum_final = 0.6 * short_term_momentum + 0.4 * weekly_return
    volume_activity_final = 0.5 * volume_activity_ratio + 0.5 * volume_act_ratio
    composite_alpha_factor = 0.6 * price_momentum_final + 0.4 * volume_activity_final

    return pd.Series(composite_alpha_factor, index=df.index)
