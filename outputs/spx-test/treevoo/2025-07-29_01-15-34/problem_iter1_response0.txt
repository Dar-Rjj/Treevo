```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Short-term Price Movement
    df['short_term_return'] = df['daily_return'].rolling(window=5).sum()
    df['short_term_price_score'] = 100 * (df['short_term_return'] - df['short_term_return'].min()) / (df['short_term_return'].max() - df['short_term_return'].min())
    
    # Long-term Price Movement
    df['weekly_return'] = df['close'].pct_change(5)
    df['long_term_price_score'] = 100 * (df['weekly_return'] - df['weekly_return'].min()) / (df['weekly_return'].max() - df['weekly_return'].min())
    
    # Combine Price Momentum Score
    df['price_momentum_score'] = 0.5 * (df['short_term_price_score'] + df['long_term_price_score'])
    
    # Compute Daily Volume Change
    df['daily_volume_change'] = df['volume'].pct_change()
    
    # Short-term Volume Change
    df['avg_daily_volume_change'] = df['daily_volume_change'].rolling(window=5).mean()
    df['short_term_volume_score'] = 100 * (df['avg_daily_volume_change'] - df['avg_daily_volume_change'].min()) / (df['avg_daily_volume_change'].max() - df['avg_daily_volume_change'].min())
    
    # Long-term Volume Activity
    df['sum_20_days_volume'] = df['volume'].rolling(window=20).sum()
    df['long_term_volume_score'] = 100 * (df['sum_20_days_volume'] - df['sum_20_days_volume'].min()) / (df['sum_20_days_volume'].max() - df['sum_20_days_volume'].min())
    
    # Combine Volume Activity Score
    df['volume_activity_score'] = 0.5 * (df['short_term_volume_score'] + df['long_term_volume_score'])
    
    # Composite Alpha Factor
    df['alpha_factor'] = 0.6 * df['price_momentum_score'] + 0.4 * df['volume_activity_score']
    
    return df['alpha_factor']
```
