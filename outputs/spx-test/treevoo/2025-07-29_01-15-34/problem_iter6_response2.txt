```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Adjust for Recent Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    price_trend = sma_20 - df['close']
    
    # Combine Factors
    factor_values = volume_intensity * close_position + price_trend
    
    return factor_values
```
