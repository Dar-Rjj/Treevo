import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_high_low_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum = (df['close'] - df['open']) / df['close']
    
    # Calculate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    average_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend - average_volume
    
    # Adjust for Recent Price Trend
    recent_price_trend = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    combined_factor = (
        (volume_intensity * closing_position) +
        recent_price_trend +
        intraday_range_percentage +
        close_to_open_momentum
    )
    
    return combined_factor
