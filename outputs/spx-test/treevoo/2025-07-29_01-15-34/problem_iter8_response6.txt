    close_to_open_momentum = (df['close'] - df['open']) / df['open']

    volume_activity_ratio = volume_trend / (average_volume + 1e-8)  # add small constant to avoid division by zero

    alpha_factor = (
        weighted_ratio * 0.3 +
        (positive_momentum * 0.2 - potential_reversal * 0.1) +
        average_combined_factor * 0.2 +
