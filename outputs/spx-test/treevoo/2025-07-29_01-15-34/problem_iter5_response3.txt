```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    df['Combined_Factor'] = df['Daily_Log_Return'] * df['Volume_Change']
    
    # Compute Average of Combined Factor over 10 Days
    df['Avg_Combined_10D'] = df['Combined_Factor'].rolling(window=10).mean()
    
    # Calculate Intraday Range Percentage
    df['Intraday_Range_Percentage'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    df['Close_Open_Momentum_Shift'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    df['Volume_Trend'] = df['volume'].rolling(window=20).sum()
    df['Avg_Volume_20D'] = df['volume'].rolling(window=20).mean()
    df['Volume_Activity_Ratio'] = df['Volume_Trend'] / df['Avg_Volume_20D']
    
    # Calculate Volatility
    df['Volatility_20D'] = df['Daily_Log_Return'].rolling(window=20).std()
    
    # Determine Momentum Strength
    df['Momentum_Strength'] = df['Daily_Log_Return'].rolling(window=20).mean() / df['Volatility_20D']
    
    # Apply Volume Trend Adjustment
    volume_trend_20D = (df['volume'] - df['volume'].shift(1)).rolling(window=20).mean()
    df['Adjusted_Momentum_Strength'] = df['Momentum_Strength'] * (1 + volume_trend_20D / 100)
    df.loc[volume_trend_20D <= 0, 'Adjusted_Momentum_Strength'] = df['Momentum_Strength'] * (1 - volume_trend_20D / 100)
    
    # Smooth the Adjusted Momentum Strength
    df['Smoothed_Adjusted_Momentum'] = df['Adjusted_Momentum_Strength'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Avg_Combined_10D'] + df['Intraday_Range_Percentage'] + 
                          df['Close_Open_Momentum_Shift'] + df['Volume_Activity_Ratio'] + 
                          df['Smoothed_Adjusted_Momentum'])
    
    return df['Alpha_Factor']
```
