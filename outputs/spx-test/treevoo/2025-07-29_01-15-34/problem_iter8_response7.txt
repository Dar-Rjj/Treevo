import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position_in_range = (df['close'] - df['open']) / high_low_range
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    N = 10
    volume_trend = df['volume'].rolling(window=N).sum()
    avg_volume = df['volume'].rolling(window=N).mean()
    volume_activity_ratio = volume_trend - avg_volume
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Combine Indicators for Predictive Factor
    high_low_to_open_close_weighted = (high_low_range / (df['close'] - df['open'])) * df['volume']
    
    # Compute Volume-Weighted Intraday Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_weighted_intraday_momentum = intraday_return * df['volume']
    
    # Combine Factors
    factor = intraday_volume_intensity * close_position_in_range
    recent_price_trend = df['close'] - df['close'].rolling(window=20).mean()
    final_factor = factor + recent_price_trend
    
    return final_factor
