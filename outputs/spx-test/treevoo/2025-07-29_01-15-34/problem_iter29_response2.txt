import pandas as pd

def heuristics_v2(df):
    # Intraday Price Movement Analysis
    high_low_range = df['high'] - df['low']
    open_close_movement = df['close'] - df['open']
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    high_low_to_open_close_ratio = high_low_range / open_close_movement.abs()
    vwap_deviation = df['close'] - vwap
    positive_momentum = (high_low_range > 1.5 * open_close_movement.abs()).astype(int)
    potential_reversal = ((open_close_movement > 0.8 * high_low_range) & (vwap_deviation < 0) & (open_close_movement < 0)).astype(int)

    # Intraday Range Analysis
    range_percentage = high_low_range / df['close']
    closing_position_in_range = (df['close'] - df['open']) / high_low_range

    # Volume Analysis
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    volume_trend = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=20).mean()

    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = df['close'] - ema_20

    # Close-to-Open Momentum Shift
    close_to_open_momentum = (df['close'] - df['open']) / df['open'].shift(1)

    # Intraday Range Percentage Enhanced
    intraday_range_percentage_enhanced = (df['high'] - df['low']) / df['close']

    # Composite Alpha Factor
    price_momentum_score = (df['close'].pct_change().rolling(window=5).sum() * 0.6) + (df['close'].pct_change(periods=5).rolling(window=5).sum() * 0.4)
    volume_activity_score = (df['volume'].pct_change().rolling(window=5).mean() * 0.6) + (df['volume'].rolling(window=20).sum() * 0.4)
    weighted_sum = (price_momentum_score * 0.6) + (volume_activity_score * 0.4)
    
    # Incorporate Recent Volatility
    recent_volatility = df['close'].rolling(window=20).std()
    ema_adjustment_normalized = ema_adjustment / recent_volatility

    # Final Alpha Factor
    alpha_factor = weighted_sum + (volume_intensity * closing_position_in_range) + intraday_range_percentage_enhanced + close_to_open_momentum + ema_adjustment_normalized

    return alpha_factor
