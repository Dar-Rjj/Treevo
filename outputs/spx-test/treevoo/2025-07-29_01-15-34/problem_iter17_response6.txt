```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength of close price to open price
    df['rs'] = df['close'] / df['open']
    
    # Calculate the weighted average of high and low prices using volume as weight
    df['weighted_avg'] = (df['high'] + df['low']) / 2 * df['volume']
    
    # Create a factor that captures the momentum of the weighted average
    df['factor'] = df['weighted_avg'].pct_change(periods=5)
    
    # Scale the factor by the relative strength
    df['factor'] = df['factor'] * df['rs']
    
    return df['factor']
```
