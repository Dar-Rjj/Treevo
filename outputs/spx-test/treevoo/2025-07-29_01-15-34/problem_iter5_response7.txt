```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close and multiply by volume
    intraday_factor = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = high_low_range > open_close_movement
    potential_reversal = open_close_movement > 0.6 * high_low_range
    
    # Calculate Intraday Range Percentage
    intraday_range_pct = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    n_period = 5  # for example, 5-day period
    volume_trend = df['volume'].rolling(window=n_period).sum()
    avg_volume = df['volume'].rolling(window=n_period).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Calculate Price Momentum Score
    daily_return = df['close'] / df['close'].shift(1) - 1
    short_term_price_momentum = daily_return.rolling(window=5).sum().clip(lower=0, upper=None)
    short_term_price_score = (short_term_price_momentum - short_term_price_momentum.min()) / (short_term_price_momentum.max() - short_term_price_momentum.min()) * 100
    
    weekly_return = df['close'] / df['close'].shift(5) - 1
    long_term_price_score = (weekly_return - weekly_return.min()) / (weekly_return.max() - weekly_return.min()) * 100
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'] / df['volume'].shift(1) - 1
    short_term_volume_activity = daily_volume_change.rolling(window=5).mean().clip(lower=0, upper=None)
    short_term_volume_score = (short_term_volume_activity - short_term_volume_activity.min()) / (short_term_volume_activity.max() - short_term_volume_activity.min()) * 100
    
    long_term_volume_sum = df['volume'].rolling(window=20).sum()
    long_term_volume_score = (long_term_volume_sum - long_term_volume_sum.min()) / (long_term_volume_sum.max() - long_term_volume_sum.min()) * 100
    
    # Composite Alpha Factor
    price_momentum_score = 0.6 * short_term_price_score + 0.4 * long_term_price_score
    volume_activity_score = 0.6 * short_term_volume_score + 0.4 * long_term_volume_score
    composite_alpha = 0.6 * price_momentum_score + 0.4 * volume_activity_score
    
    return composite_alpha
```
