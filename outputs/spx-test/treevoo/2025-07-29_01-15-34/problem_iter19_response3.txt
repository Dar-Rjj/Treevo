```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14):
    # Intraday Range Intensity
    df['Intraday_Range_Intensity'] = (df['High'] - df['Low']) / ((df['High'] + df['Low']) / 2)
    
    # Open-to-Close Momentum Shift
    df['Open_to_Close_Momentum_Shift'] = (df['Close'] - df['Open']) / df['Close'].shift(1)
    
    # Volume Activity Ratio
    df['Volume_Sum_N'] = df['Volume'].rolling(window=n).sum()
    df['Volume_EMA_N'] = df['Volume'].ewm(span=n, adjust=False).mean()
    df['Volume_Activity_Ratio'] = df['Volume_Sum_N'] / df['Volume_EMA_N']
    
    # Daily Return Volatility Adjusted
    df['Log_Return'] = np.log(df['Close'] / df['Close'].shift(1))
    df['True_Range'] = df[['High', 'Low']].sub(df['Close'].shift(1), axis=0).abs().max(axis=1)
    df['Average_True_Range'] = df['True_Range'].rolling(window=n).mean()
    df['Daily_Return_Vol_Adjusted'] = df['Log_Return'] / df['Average_True_Range']
    
    # Price Volatility Factor
    df['Return_Std'] = df['Log_Return'].rolling(window=n).std()
    
    # Relative Strength Index (RSI)
    df['Gain'] = np.where(df['Close'] > df['Close'].shift(1), df['Close'] - df['Close'].shift(1), 0)
    df['Loss'] = np.where(df['Close'] < df['Close'].shift(1), df['Close'].shift(1) - df['Close'], 0)
    df['Average_Gain'] = df['Gain'].rolling(window=n).mean()
    df['Average_Loss'] = df['Loss'].rolling(window=n).mean()
    df['RSI'] = 100 - (100 / (1 + (df['Average_Gain'] / df['Average_Loss'])))

    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Intraday_Range_Intensity'] +
        df['Open_to_Close_Momentum_Shift'] +
        df['Volume_Activity_Ratio'] +
        df['Daily_Return_Vol_Adjusted'] +
        df['Return_Std'] +
        df['RSI']
    ) / 6  # Average of all the factors

    return df['Alpha_Factor']
```
