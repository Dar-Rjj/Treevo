```python
import pandas as pd

def heuristics_v2(df):
    # Compute the 21-day EMA of the close price
    df['EMA_21_close'] = df['close'].ewm(span=21, adjust=False).mean()
    
    # Subtract the 21-day EMA from the current day's close price
    raw_alpha_factor = df['close'] - df['EMA_21_close']
    
    # Compute the 14-day EMA of the volume
    df['EMA_14_volume'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Subtract today's volume from the 14-day EMA of volume
    volume_factor = df['volume'] - df['EMA_14_volume']
    
    # Multiply the raw alpha factor by the volume factor
    combined_factor = raw_alpha_factor * volume_factor
    
    return combined_factor
```
