```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / df['close']
    
    # Calculate High-Low Price Momentum
    high_low_momentum = high_low_diff * adjusted_volume
    
    # Calculate Smoothed High-Low Price Momentum (5-day Moving Average)
    smoothed_high_low_momentum = high_low_momentum.rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Combine Intraday Return with Volume
    combined_intraday_return = intraday_return * df['volume']
    
    # Compute Volume-Weighted Intraday Momentum
    volume_weighted_intraday_momentum = combined_intraday_return / intraday_high_low_range
    
    # Calculate Previous Day's Volume
    previous_day_volume = df['volume'].shift(1)
    
    # Integrated Alpha Factor
    integrated_alpha_factor = (smoothed_high_low_momentum + volume_weighted_intraday_momentum) / previous_day_volume
    
    return integrated_alpha_factor
```
