```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Volume-Weighted Price
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Short-Term Price Trend (t-1 to t-3)
    df['short_term_trend'] = df['vwap'].shift(1) - df['vwap'].shift(3)
    
    # Long-Term Price Trend (t-5 to t-20)
    df['long_term_trend'] = df['vwap'].shift(5) - df['vwap'].shift(20)
    
    # Reversal Momentum
    df['reversal_momentum'] = (df['long_term_trend'] - df['short_term_trend']) * df['vwap']
    
    return df['reversal_momentum'].dropna()
```
