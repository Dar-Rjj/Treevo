```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / high_low_range
    
    # Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Price Change
    price_change = (df['close'] - df['open']) / high_low_range
    
    # Intraday Range Percentage
    range_percentage = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Volume-Weighted Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['close'].shift(1)
    close_open_momentum *= volume_intensity
    
    # Volume Activity Ratio
    n_period_sum_volume = df['volume'].rolling(window=10).sum()
    n_period_ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    volume_activity_ratio = n_period_sum_volume / n_period_ema_volume
    
    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    return_volatility = daily_return.rolling(window=20).std()
    daily_return_volatility = daily_return / return_volatility
    
    # Combine Intraday Components
    intraday_components = (volume_intensity * close_position) + price_change + close_open_momentum
    
    # Adjust for Recent Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    recent_trend_adjustment = df['close'] - sma_20
    
    # Incorporate Daily Return Volatility
    adjusted_intraday = (intraday_components * daily_return_volatility) + recent_trend_adjustment
    
    # Trend Reversal Potential
    rsi = df['close'].diff().mask(df['close'].diff() < 0, 0).rolling(window=14).mean() / df['close'].diff().abs().rolling(window=14).mean()
    rsi = 100 - (100 / (1 + rsi))
    overbought_flag = (rsi > 70).astype(int)
    oversold_flag = (rsi < 30).astype(int)
    roc = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    trend_reversal_potential = 0
    trend_reversal_potential += (overbought_flag & (roc < 0)).astype(int) * 0.5
    trend_reversal_potential -= (oversold_flag & (roc > 0)).astype(int) * 0.5
    
    # Final Alpha Factor
    alpha_factor = (
        adjusted_intraday +
        trend_reversal_potential +
        range_percentage +
        close_open_momentum
    )
    
    return alpha_factor
```
