```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # High-Low to Open-Close Ratio
    high_low_to_open_close_ratio = high_low_range / abs(open_close_movement).replace(0, 1e-6)
    
    # Closing Position in Intraday Range
    closing_position_in_intraday_range = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Volume Activity Ratio
    n_period = 10
    volume_activity_ratio = df['volume'].rolling(window=n_period).sum() / df['volume'].rolling(window=n_period).mean()
    
    # Intraday Volume Intensity
    median_volume = df['volume'].rolling(window=10).median()
    intraday_volume_intensity = df['volume'] / median_volume
    
    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend_adjustment = df['close'] - ema_20
    
    # Recent Volatility
    recent_volatility = df['close'].rolling(window=20).std()
    trend_adjustment_with_volatility = recent_price_trend_adjustment / recent_volatility.replace(0, 1e-6)
    
    # Composite Alpha Factor
    weighted_sum = (
        0.3 * high_low_to_open_close_ratio +
        0.3 * vwap_deviation +
        0.2 * intraday_volume_intensity +
        0.2 * closing_position_in_intraday_range
    )
    
    final_alpha_factor = weighted_sum + trend_adjustment_with_volatility
    
    return final_alpha_factor
```
