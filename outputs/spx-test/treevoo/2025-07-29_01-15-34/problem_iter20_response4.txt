```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pandas.DataFrame): A DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pandas.Series: A Series with index as date and values as the calculated alpha factor.
    """
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average price from the previous day
    avg_price_change = avg_price.pct_change()
    
    # Calculate the ratio of volume to the average of the previous 5 days' volume
    vol_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Combine the above metrics into a single alpha factor
    alpha_factor = avg_price_change * vol_ratio * price_range
    
    return alpha_factor
```
