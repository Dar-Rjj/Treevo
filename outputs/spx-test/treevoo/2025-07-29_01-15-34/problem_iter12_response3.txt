import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()
    df['monthly_return'] = df['close'].pct_change(periods=20)
    
    df['short_term_momentum_rank'] = df['short_term_momentum'].rank(pct=True) * 100
    df['long_term_momentum_rank'] = df['monthly_return'].rank(pct=True) * 100
    df['price_momentum_score'] = (df['short_term_momentum_rank'] + df['long_term_momentum_rank']) / 2

    # Calculate Volume Activity Score
    df['daily_volume_change'] = df['volume'].pct_change()
    df['avg_daily_volume_change'] = df['daily_volume_change'].rolling(window=5).mean()
    df['sum_of_volumes_20_days'] = df['volume'].rolling(window=20).sum()
    
    df['short_term_volume_rank'] = df['avg_daily_volume_change'].rank(pct=True) * 100
    df['long_term_volume_rank'] = df['sum_of_volumes_20_days'].rank(pct=True) * 100
    df['volume_activity_score'] = (df['short_term_volume_rank'] + df['long_term_volume_rank']) / 2

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']

    # Calculate Intraday Volume Intensity
    df['avg_volume_10_days'] = df['volume'].rolling(window=10).mean()
    df['intraday_volume_intensity'] = df['volume'] / df['avg_volume_10_days']

    # Adjust for Recent Price Trend
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['recent_price_trend'] = df['close'] - df['ema_20']

    # Combine Factors
    df['weighted_scores'] = (df['price_momentum_score'] * 0.5 +
                             df['volume_activity_score'] * 0.3 +
                             df['intraday_range'] * 0.2)

    df['combined_factors'] = (df['intraday_volume_intensity'] * df['closing_position'] +
                              df['recent_price_trend'])

    # Final Alpha Factor Combination
    df['final_alpha_factor'] = (df['combined_factors'] * df['weighted_scores'])
    
    # Add a Sentiment Score (Assuming we have a sentiment score column)
    df['sentiment_score'] = 50  # Placeholder value, replace with actual sentiment score
    df['final_alpha_factor'] += df['sentiment_score'] * 0.1

    return df['final_alpha_factor']
