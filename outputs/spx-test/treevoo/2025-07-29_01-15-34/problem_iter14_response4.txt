```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position_in_range = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    average_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend - average_volume
    
    # Combine Factors
    factor = (volume_intensity * close_position_in_range) + \
             (df['close'] - df['close'].ewm(span=20, adjust=False).mean()) + \
             intraday_range_percentage + \
             close_open_momentum + \
             volume_activity_ratio
    
    return factor
```
