```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Short-term Price Movement
    daily_returns = df['close'].pct_change()
    short_term_price_movement = daily_returns.rolling(window=5).sum()
    
    # Calculate Short-term Volume Change
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_change = daily_volume_change.rolling(window=5).mean()
    
    # Combine Factors
    avg_10_day_volume = df['volume'].rolling(window=10).mean()
    intraday_volume_intensity = df['volume'] / avg_10_day_volume
    closing_position_adjusted = close_position * intraday_volume_intensity
    
    # Adjust for Recent Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    recent_trend_adjustment = sma_20 - df['close']
    
    # Weighted Sum of Scores
    weighted_price_movement = 0.6 * short_term_price_movement
    weighted_volume_change = 0.4 * short_term_volume_change
    weighted_sum = weighted_price_movement + weighted_volume_change
    
    # Final Alpha Factor Combination
    final_factor = (closing_position_adjusted + recent_trend_adjustment) * weighted_sum
    
    return final_factor
```
