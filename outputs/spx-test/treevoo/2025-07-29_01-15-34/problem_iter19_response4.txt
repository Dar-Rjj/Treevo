import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=n).sum()
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Calculate Price Momentum Score
    daily_return = df['close'].pct_change()
    short_term_momentum = daily_return.rolling(window=5).sum()
    weekly_return = df['close'].pct_change(periods=5)
    price_momentum_score = 0.6 * short_term_momentum + 0.4 * weekly_return
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_change = daily_volume_change.rolling(window=5).mean()
    volume_activity_score = short_term_volume_change
    
    # Composite Alpha Factor
    composite_alpha_factor = 0.6 * price_momentum_score + 0.4 * volume_activity_score
    
    return composite_alpha_factor
