```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Calculate Typical Price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * typical_price).cumsum() / df['volume'].cumsum()
    
    # VWAP Deviation from Close
    vwap_deviation = df['close'] - vwap
    
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = high_low_range / abs(open_close_movement).replace(0, 1)  # Avoid division by zero
    
    # Positive Momentum
    positive_momentum = high_low_range > 1.5 * abs(open_close_movement)
    
    # Potential Reversal
    potential_reversal = (abs(open_close_movement) > 0.8 * high_low_range) & (vwap_deviation < 0) & (open_close_movement < 0)
    
    # Momentum Continuation
    momentum_continuation = (high_low_range > 2 * abs(open_close_movement)) & (vwap > df['close'])
    
    # Reversal Confirmation
    reversal_confirmation = (open_close_movement < -0.5 * high_low_range) & (vwap < df['close'])
    
    # Combine Indicators for Predictive Factor
    factor = (weighted_ratio * positive_momentum) + (vwap_deviation * potential_reversal) + (momentum_continuation * reversal_confirmation)
    
    return factor
```
