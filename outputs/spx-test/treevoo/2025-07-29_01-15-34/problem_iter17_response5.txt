```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    high_low_open_close_ratio = high_low_range / open_close_movement
    trend_strength = (high_low_range > open_close_movement).astype(int)
    potential_reversal = (open_close_movement > 0.6 * high_low_range).astype(int)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    combined_price_volume = daily_log_return * volume_change
    
    # Compute Average of Combined Factor over 10 Days
    avg_combined_factor = combined_price_volume.rolling(window=10).mean()
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = high_low_range / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Calculate Closing Position in Intraday Range
    closing_position_in_range = (df['close'] - df['open']) / high_low_range
    
    # Adjust for Recent Price Trend
    recent_price_trend = df['close'] - df['close'].ewm(span=220).mean()
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Price Change
    intraday_price_change = (df['close'] - df['open']) / high_low_range
    
    # Volume Activity Ratio
    volume_activity_ratio = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    
    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_volatility = daily_return / daily_return.rolling(window=10).std()
    
    # Combine Intraday Components
    combined_intraday_components = (
        intraday_volume_intensity * closing_position_in_range +
        intraday_price_change +
        close_to_open_momentum
    )
    
    # Incorporate Daily Return Volatility
    adjusted_intraday_components = combined_intraday_components * daily_return_volatility + recent_price_trend
    
    # Final Factor
    final_factor = (
        combined_intraday_components * recent_price_trend +
        avg_combined_factor
    )
    
    return final_factor
```
