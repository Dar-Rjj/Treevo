```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_high_low_range
    
    # Calculate Intraday Volume Intensity
    volume_sum_10_days = df['volume'].rolling(window=10).sum()
    volume_median_10_days = df['volume'].rolling(window=10).median()
    intraday_volume_intensity = df['volume'] / ((volume_sum_10_days + volume_median_10_days) / 2)
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = intraday_high_low_range / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum_shift = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    volume_average = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend - volume_average
    
    # Combine Factors
    factor = (intraday_volume_intensity * closing_position) + \
             (df['close'] - df['close'].rolling(window=20).mean()) + \
             intraday_range_percentage + close_to_open_momentum_shift
    
    return factor
```
