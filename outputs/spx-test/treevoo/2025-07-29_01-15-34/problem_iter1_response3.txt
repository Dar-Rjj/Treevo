```python
def heuristics_v2(data, n=10):
    # Calculate Intraday Range Percentage
    intraday_range = (data['high'] - data['low']) / data['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum = (data['close'] - data['open']) / data['open']
    
    # Summarize Volume Trend
    volume_trend = data['volume'].rolling(window=n).sum()
    
    # Calculate N-period Moving Average of Volume
    avg_volume = data['volume'].rolling(window=n).mean()
    
    # Integrate Volume Activity Ratio
    volume_activity_ratio = volume_trend / avg_volume
    
    # Combine all components into a single alpha factor
    alpha_factor = intraday_range + close_to_open_momentum + volume_activity_ratio
    
    return alpha_factor
```
