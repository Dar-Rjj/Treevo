import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = high_low_range > open_close_movement
    potential_reversal = open_close_movement > 0.6 * high_low_range
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = high_low_range / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    n_period_sum_volume = df['volume'].rolling(window=5).sum()
    n_period_avg_volume = df['volume'].rolling(window=5).mean()
    volume_activity_ratio = n_period_sum_volume / n_period_avg_volume
    
    # Calculate Price Momentum Score
    daily_return = df['close'].pct_change()
    short_term_price_momentum = (daily_return.rolling(window=5).sum() / 5 + 1).pow(1/5) * 100
    long_term_price_momentum = (df['close'] / df['close'].shift(5)).pow(1/5) * 100
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_activity = (daily_volume_change.rolling(window=5).mean() + 1).pow(1/5) * 100
    long_term_volume_activity = (df['volume'].rolling(window=20).sum() / df['volume'].rolling(window=20).mean()) * 100
    
    # Price Volatility Factor
    price_volatility = daily_return.rolling(window=5).std()
    
    # Composite Alpha Factor
    price_momentum_score = 0.6 * short_term_price_momentum + 0.4 * long_term_price_momentum
    volume_activity_score = 0.6 * short_term_volume_activity + 0.4 * long_term_volume_activity
    composite_alpha_factor = 0.6 * price_momentum_score + 0.4 * volume_activity_score
    final_alpha_factor = composite_alpha_factor / (price_volatility + 1e-6)
    
    return final_alpha_factor
