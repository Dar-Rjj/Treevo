```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    average_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend / average_volume
    
    # Calculate Intraday Volume Intensity
    recent_average_volume = df['volume'].rolling(window=10).mean()
    intraday_volume_intensity = df['volume'] / recent_average_volume
    
    # Combine Factors
    closing_position_weighted = close_position * intraday_volume_intensity
    sma_20 = df['close'].rolling(window=20).mean()
    recent_price_trend = sma_20 - df['close']
    factor = closing_position_weighted * (intraday_range + momentum_shift) + recent_price_trend
    
    return factor
```
