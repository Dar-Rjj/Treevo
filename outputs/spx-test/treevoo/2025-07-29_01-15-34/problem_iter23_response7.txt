```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of open and close prices, using volume as weights
    w_avg_price = (df['open'] * df['volume'] + df['close'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the difference between high and low price, representing the daily price range
    price_range = df['high'] - df['low']
    
    # Calculate a simple moving average of the weighted average price over 5 days
    ma_5d = w_avg_price.rolling(window=5).mean()
    
    # Factor: compare the current day's weighted average price to the 5-day moving average
    factor_value = (w_avg_price - ma_5d) / price_range
    
    return factor_value
```
