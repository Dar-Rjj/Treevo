import pandas as pd

def heuristics_v2(df):
    # Compute the 50-day rolling average of the close price
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Subtract the 50-day SMA from the current day's close price to create the raw alpha factor
    raw_alpha = df['close'] - sma_50
    
    # Compute the 20-day EMA of the volume
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Divide today's volume by the 20-day EMA of volume to create the volume factor
    volume_factor = df['volume'] / ema_20_volume
    
    # Generate the final composite alpha factor by multiplying the two sub-factors
    composite_alpha = raw_alpha * volume_factor
    
    return composite_alpha
