```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    short_term_momentum = df['daily_return'].rolling(window=5).sum().rank(pct=True) * 100
    long_term_momentum = (df['close'].pct_change(periods=20)).rank(pct=True) * 100
    price_momentum_score = (short_term_momentum + long_term_momentum) / 2

    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_activity = daily_volume_change.rolling(window=5).mean().rank(pct=True) * 100
    long_term_volume_activity = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    volume_activity_score = (short_term_volume_activity + long_term_volume_activity) / 2

    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_high_low_range

    # Calculate Intraday Volume Intensity
    avg_10d_vol = df['volume'].rolling(window=10).mean()
    intraday_volume_intensity = df['volume'] / avg_10d_vol

    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend = ema_20 - df['close']

    # Combine Factors
    combined_factors = (intraday_volume_intensity * closing_position) + recent_price_trend

    # Weighted Sum of Scores
    weighted_sum = (price_momentum_score * 0.5) + (volume_activity_score * 0.4) + (intraday_high_low_range * 0.1)

    # Final Alpha Factor Combination
    final_alpha_factor = combined_factors * weighted_sum

    return final_alpha_factor
```
