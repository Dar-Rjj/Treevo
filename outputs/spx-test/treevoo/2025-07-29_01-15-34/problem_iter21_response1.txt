```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Volume Activity Ratio
    volume_sum = df['volume'].rolling(window=10).sum()
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    volume_activity_ratio = volume_sum / volume_ema
    
    # Price Volatility Factor
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    log_return_std = daily_log_return.rolling(window=10).std()
    
    gain = (df['close'] - df['close'].shift(1)).clip(lower=0)
    loss = (df['close'] - df['close'].shift(1)).clip(upper=0).abs()
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rsi = 100 - (100 / (1 + (avg_gain / avg_loss)))
    
    price_volatility_factor = (log_return_std + rsi) / 2
    
    # Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Intraday Volume Intensity
    past_10_day_volume_sum = df['volume'].rolling(window=10).sum()
    intraday_volume_intensity = df['volume'] / past_10_day_volume_sum
    
    # Intraday Range Percentage Enhanced
    intraday_range_enhanced = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift Enhanced
    close_open_momentum_enhanced = (df['close'] - df['open']) / df['open']
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend = ema_20 - df['close']
    
    # Composite Alpha Factor
    short_term_price_movement = df['close'].pct_change().rolling(window=5).sum()
    long_term_price_movement = df['close'].pct_change(periods=5).rolling(window=5).mean()
    short_term_volume_change = df['volume'].pct_change().rolling(window=5).mean()
    long_term_volume_activity = df['volume'].rolling(window=20).sum()
    
    price_momentum_score = 0.6 * short_term_price_movement + 0.4 * long_term_price_movement
    volume_activity_score = 0.6 * short_term_volume_change + 0.4 * long_term_volume_activity
    
    composite_alpha_factor = (
        0.6 * price_momentum_score + 
        0.4 * volume_activity_score + 
        intraday_volume_intensity * closing_position + 
        intraday_range_enhanced + 
        close_open_momentum_enhanced + 
        recent_price_trend
    )
    
    return composite_alpha_factor
```
