```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day SMA of closing prices
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Use the difference between the current day's close price and the 50-day SMA as an alpha factor
    raw_alpha_factor = df['close'] - sma_50
    
    # Calculate the 20-day EMA of volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Create a volume-based alpha factor
    volume_factor = df['volume'] / ema_volume
    
    # Combine the SMA and volume factors
    combined_alpha_factor = raw_alpha_factor * volume_factor
    
    return combined_alpha_factor
```
