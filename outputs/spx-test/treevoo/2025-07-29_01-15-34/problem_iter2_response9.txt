```python
import pandas as pd

def heuristics_v2(df, n_period=5, long_term_period=20):
    # Calculate Intraday Range Percentage
    df['intraday_range'] = (df['high'] - df['low']) / df['close']

    # Calculate Close-to-Open Momentum Shift
    df['close_to_open'] = (df['close'] - df['open']) / df['open']

    # Integrate Volume Activity Ratio
    df['volume_trend'] = df['volume'].rolling(window=n_period).sum()
    df['avg_volume'] = df['volume'].rolling(window=n_period).mean()
    df['volume_activity_ratio'] = df['volume_trend'] / df['avg_volume']

    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()
    df['weekly_return'] = df['close'].pct_change(periods=5)
    df['price_momentum_score'] = 100 * (df['short_term_momentum'] + df['weekly_return'])

    # Calculate Volume Activity Score
    df['daily_volume_change'] = df['volume'].pct_change()
    df['avg_daily_volume_change'] = df['daily_volume_change'].rolling(window=5).mean()
    df['long_term_volume_sum'] = df['volume'].rolling(window=long_term_period).sum()
    df['volume_activity_score'] = 100 * (df['avg_daily_volume_change'] + df['long_term_volume_sum'])

    # Composite Alpha Factor
    df['composite_factor'] = 0.6 * df['price_momentum_score'] + 0.4 * df['volume_activity_score']

    return df['composite_factor']
```
