```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the ratio of price change to daily range
    df['change_to_range_ratio'] = df['price_change'] / df['daily_range']
    
    # Calculate the weighted price change by volume
    df['weighted_price_change'] = df['price_change'] * df['volume']
    
    # Calculate the factor: (weighted price change * change to range ratio) / close
    df['factor'] = (df['weighted_price_change'] * df['change_to_range_ratio']) / df['close']
    
    # Return the alpha factor as a pandas Series
    return df['factor']
```
