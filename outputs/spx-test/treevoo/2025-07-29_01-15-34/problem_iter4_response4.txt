```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    # Weighted Ratio of High-Low to Open-Close, multiplied by volume
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    factor = pd.Series(index=df.index, dtype=float)
    
    # Positive Momentum if High-Low > Open-Close
    positive_momentum = (high_low_range > open_close_movement)
    
    # Potential Reversal if Open-Close > 0.6 * High-Low
    potential_reversal = (open_close_movement > 0.6 * high_low_range)
    
    # Assign the factor values based on the conditions
    factor[positive_momentum & ~potential_reversal] = weighted_ratio[positive_momentum & ~potential_reversal]
    factor[~positive_momentum & potential_reversal] = -weighted_ratio[~positive_momentum & potential_reversal]
    factor[~positive_momentum & ~potential_reversal] = 0
    
    return factor
```
