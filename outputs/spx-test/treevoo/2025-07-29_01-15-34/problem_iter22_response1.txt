```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['close'].shift(1)
    
    # Volume Activity Ratio
    n_period = 10
    volume_sum = df['volume'].rolling(window=n_period).sum()
    volume_ema = df['volume'].ewm(span=n_period, adjust=False).mean()
    volume_activity_ratio = volume_sum / volume_ema
    
    # Price Volatility Factor
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    log_return_std = daily_log_return.rolling(window=n_period).std()
    
    # Relative Strength Index (RSI)
    gain = (df['close'] - df['close'].shift(1)).apply(lambda x: max(0, x))
    loss = (df['close'] - df['close'].shift(1)).apply(lambda x: min(0, x) * -1)
    avg_gain = gain.rolling(window=n_period).mean()
    avg_loss = loss.rolling(window=n_period).mean()
    rsi = 100 - (100 / (1 + (avg_gain / avg_loss)))
    
    # Closing Position in Intraday Range
    close_in_intraday_range = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Intraday Volume Intensity
    past_10_day_volume_sum = df['volume'].rolling(window=10).sum()
    intraday_volume_intensity = df['volume'] / past_10_day_volume_sum
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend_adjustment = ema_20 - df['close']
    
    # Short-term and Long-term Price Movement
    daily_return = df['close'].pct_change()
    short_term_price_movement = daily_return.rolling(window=5).sum()
    weekly_return = df['close'].pct_change(periods=5)
    long_term_price_movement = weekly_return
    
    # Short-term and Long-term Volume Change
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_change = daily_volume_change.rolling(window=5).mean()
    long_term_volume_activity = df['volume'].rolling(window=20).sum()
    
    # Composite Alpha Factor
    price_score = 0.6 * (short_term_price_movement + long_term_price_movement)
    volume_score = 0.4 * (short_term_volume_change + long_term_volume_activity)
    composite_alpha = intraday_range + close_open_momentum + price_score + volume_score + price_trend_adjustment + close_in_intraday_range + intraday_volume_intensity
    
    return composite_alpha
```
