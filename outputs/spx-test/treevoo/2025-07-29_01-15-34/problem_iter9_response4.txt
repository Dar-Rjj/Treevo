import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close multiplied by Volume
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement)
    potential_reversal = (open_close_movement > 0.5 * high_low_range)
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    n_period_sum_volume = df['volume'].rolling(window=5).sum()
    n_period_avg_volume = df['volume'].rolling(window=5).mean()
    volume_activity_ratio = n_period_sum_volume / n_period_avg_volume
    
    # Calculate Price Momentum Score
    daily_return = df['close'].pct_change()
    short_term_price_momentum = daily_return.rolling(window=5).sum().map(lambda x: min(max(x, -1), 1) * 50 + 50)
    weekly_return = df['close'].pct_change(periods=5)
    long_term_price_momentum = weekly_return.map(lambda x: min(max(x, -1), 1) * 50 + 50)
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_change = daily_volume_change.rolling(window=5).mean().map(lambda x: min(max(x, -1), 1) * 50 + 50)
    long_term_volume_activity = df['volume'].rolling(window=20).sum().map(lambda x: min(max(x, df['volume'].quantile(0.25)), df['volume'].quantile(0.75)) * 50 + 50)
    
    # Composite Alpha Factor
    price_momentum_score = 0.7 * (short_term_price_momentum + long_term_price_momentum) / 2
    volume_activity_score = 0.3 * (short_term_volume_change + long_term_volume_activity) / 2
    composite_alpha_factor = price_momentum_score + volume_activity_score
    
    return composite_alpha_factor
