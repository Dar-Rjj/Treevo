```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Intraday Price Change
    intraday_price_change = (df['close'] - df['open']) / intraday_range
    
    # Combine Intraday Components
    intraday_components = (volume_intensity * closing_position) + intraday_price_change
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend_adjustment = ema_20 - df['close']
    
    # Calculate Price Momentum Score
    daily_returns = df['close'].pct_change()
    short_term_momentum = daily_returns.rolling(window=5).sum().rank(pct=True) * 100
    long_term_momentum = (df['close'] / df['close'].shift(20)).rank(pct=True) * 100
    price_momentum_score = (short_term_momentum + long_term_momentum) / 2
    
    # Calculate Volume Activity Score
    daily_volume_changes = df['volume'].pct_change()
    short_term_volume_activity = daily_volume_changes.rolling(window=5).mean().rank(pct=True) * 100
    long_term_volume_activity = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    volume_activity_score = (short_term_volume_activity + long_term_volume_activity) / 2
    
    # Weighted Sum of Scores
    weighted_sum_scores = (price_momentum_score * 0.6) + (volume_activity_score * 0.3) + (intraday_range * 0.1)
    
    # Final Alpha Factor Combination
    alpha_factor = (intraday_components * weighted_sum_scores) + price_trend_adjustment
    
    return alpha_factor
```
