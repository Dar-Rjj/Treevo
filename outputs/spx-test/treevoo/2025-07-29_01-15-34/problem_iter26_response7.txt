```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day Simple Moving Average (SMA) of closing prices
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Use the difference between the current day's close price and the 50-day SMA as an alpha factor
    raw_alpha_factor = df['close'] - sma_50
    
    # Calculate the 20-day Exponential Moving Average (EMA) of volume
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Create a volume-based alpha factor by comparing today's volume to its 20-day EMA
    volume_factor = df['volume'] / ema_20_volume
    
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Combine the SMA, volume, and RSI factors to create a composite alpha factor
    intermediate_composite_alpha = raw_alpha_factor * volume_factor
    final_composite_alpha_factor = intermediate_composite_alpha * rsi_14
    
    return final_composite_alpha_factor
```
