import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=n).sum()
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_trend - avg_volume
    
    # Combine Indicators for Predictive Factor
    weighted_ratio = (intraday_range / (df['close'] - df['open'])) * df['volume']
    trend_strength_signal = (intraday_range > (df['close'] - df['open']))
    reversal_signal = (df['close'] - df['open']) > 0.6 * intraday_range
    
    # Compute Volume-Weighted Intraday Momentum
    intraday_return = (df['close'] - df['open'])
    volume_weighted_intraday_momentum = intraday_return * df['volume']
    
    # Adjust for Recent Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    recent_price_trend = df['close'] - sma_20
    
    # Final Alpha Factor
    alpha_factor = (closing_position * volume_weighted_intraday_momentum) + recent_price_trend
    
    return alpha_factor
