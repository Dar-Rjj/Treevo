import pandas as pd

def heuristics_v2(df):
    # Calculate the relative close price to the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    rel_close_avg = df['close'] / avg_high_low
    
    # Determine if the close is above or below the high-low average
    is_above_avg = (df['close'] > avg_high_low).astype(int)
    
    # Adjust the factor by the volume, giving more weight to days with higher volume
    weighted_factor = rel_close_avg * df['volume']
    
    # Apply a heuristic adjustment based on whether the close is above or below the average
    adjusted_factor = weighted_factor * (is_above_avg * 2 - 1)

    return adjusted_factor
