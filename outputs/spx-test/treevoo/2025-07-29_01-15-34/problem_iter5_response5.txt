```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range Percentage
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Volume Activity Ratio
    n = 10
    df['volume_trend'] = df['volume'].rolling(window=n).sum()
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_activity_ratio'] = df['volume_trend'] / df['average_volume']
    
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()
    df['short_term_score'] = (df['short_term_momentum'] - df['short_term_momentum'].min()) / (df['short_term_momentum'].max() - df['short_term_momentum'].min()) * 100
    
    df['weekly_return'] = df['close'].pct_change(periods=5)
    df['long_term_score'] = (df['weekly_return'] - df['weekly_return'].min()) / (df['weekly_return'].max() - df['weekly_return'].min()) * 100
    
    # Calculate Volume Activity Score
    df['daily_volume_change'] = df['volume'].pct_change()
    df['avg_daily_vol_change'] = df['daily_volume_change'].rolling(window=5).mean()
    df['short_term_vol_score'] = (df['avg_daily_vol_change'] - df['avg_daily_vol_change'].min()) / (df['avg_daily_vol_change'].max() - df['avg_daily_vol_change'].min()) * 100
    
    df['long_term_vol_sum'] = df['volume'].rolling(window=20).sum()
    df['long_term_vol_score'] = (df['long_term_vol_sum'] - df['long_term_vol_sum'].min()) / (df['long_term_vol_sum'].max() - df['long_term_vol_sum'].min()) * 100
    
    # Calculate Intraday Volume Intensity
    df['intraday_vol_intensity'] = df['volume'] / df['volume'].rolling(window=10).median()
    
    # Adjust for Recent Price Trend
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend_adjustment'] = df['close'] - df['ema_20']
    
    # Composite Alpha Factor
    df['price_momentum_score'] = df['short_term_score'] * 0.6 + df['long_term_score'] * 0.4
    df['volume_activity_score'] = df['short_term_vol_score'] * 0.6 + df['long_term_vol_score'] * 0.4
    df['intraday_factor'] = df['intraday_vol_intensity'] * df['closing_position']
    
    df['alpha_factor'] = df['price_momentum_score'] * 0.6 + df['volume_activity_score'] * 0.4 + df['intraday_factor'] * df['price_trend_adjustment']
    
    return df['alpha_factor']
```
