```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['Daily_Return'] = df['close'].pct_change()
    df['Short_Term_Momentum'] = df['Daily_Return'].rolling(window=5).sum()
    df['Weekly_Return'] = df['close'].pct_change(periods=5)
    
    # Convert to a Rank (0-100) for short-term and long-term momentum
    df['Short_Term_Momentum_Rank'] = (df['Short_Term_Momentum'].rank(pct=True) * 100).clip(0, 100)
    df['Long_Term_Momentum_Rank'] = (df['Weekly_Return'].rank(pct=True) * 100).clip(0, 100)
    
    # Calculate Volume Activity Score
    df['Daily_Volume_Change'] = df['volume'].pct_change()
    df['Average_Daily_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=5).mean()
    df['Total_Volume_20_Days'] = df['volume'].rolling(window=20).sum()
    
    # Convert to a Rank (0-100) for volume activity
    df['Short_Term_Volume_Rank'] = (df['Average_Daily_Volume_Change'].rank(pct=True) * 100).clip(0, 100)
    df['Long_Term_Volume_Rank'] = (df['Total_Volume_20_Days'].rank(pct=True) * 100).clip(0, 100)
    
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['Closing_Position'] = (df['close'] - df['open']) / df['Intraday_High_Low_Range']
    
    # Calculate Intraday Volume Intensity
    df['Average_Volume_10_Days'] = df['volume'].rolling(window=10).mean()
    df['Intraday_Volume_Intensity'] = df['volume'] / df['Average_Volume_10_Days']
    
    # Adjust for Recent Price Trend
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Price_Trend_Adjustment'] = df['SMA_20'] - df['close']
    
    # Combine Factors
    df['Combined_Factors'] = df['Intraday_Volume_Intensity'] * df['Closing_Position'] * df['Price_Trend_Adjustment']
    
    # Weighted Sum of Scores
    df['Weighted_Score'] = (df['Short_Term_Momentum_Rank'] * 0.7) + (df['Long_Term_Momentum_Rank'] * 0.3)
    df['Volume_Activity_Score'] = (df['Short_Term_Volume_Rank'] * 0.5) + (df['Long_Term_Volume_Rank'] * 0.5)
    
    # Final Alpha Factor Combination
    df['Final_Alpha_Factor'] = df['Combined_Factors'] * (df['Weighted_Score'] * 0.7 + df['Volume_Activity_Score'] * 0.3)
    
    return df['Final_Alpha_Factor'].dropna()
```
