```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP Deviation from Close
    vwap_deviation = df['close'] - vwap
    
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = high_low_range / np.abs(open_close_movement).replace(0, np.nan)
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Volume Activity Ratio
    n_period = 10
    volume_sum = df['volume'].rolling(window=n_period).sum()
    volume_ma = df['volume'].rolling(window=n_period).mean()
    volume_activity_ratio = volume_sum / volume_ma
    
    # Intraday Volume Intensity
    median_vol_10d = df['volume'].rolling(window=10).median()
    intraday_volume_intensity = df['volume'] / median_vol_10d
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = df['close'] - ema_20
    
    # Short-Term Price Momentum
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_5 - ema_10
    
    # Long-Term Momentum
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_momentum = ema_20 - ema_50
    
    # Incorporate Recent Volatility
    recent_volatility = df['close'].rolling(window=20).std()
    volatility_adjusted_ema = ema_adjustment / recent_volatility
    
    # Composite Alpha Factor
    weighted_sum = (weighted_ratio * 0.3 +
                    vwap_deviation * 0.3 +
                    intraday_volume_intensity * 0.2 +
                    close_position * 0.2)
    
    trend_adjusted_alpha = weighted_sum + volatility_adjusted_ema
    final_alpha_factor = trend_adjusted_alpha + short_term_momentum + long_term_momentum
    
    return final_alpha_factor
```
