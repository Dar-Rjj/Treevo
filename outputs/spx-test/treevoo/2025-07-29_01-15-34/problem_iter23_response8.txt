import pandas as pd

def heuristics_v2(df):
    # Intraday Price Movement Analysis
    high_low_range = df['high'] - df['low']
    open_close_movement = df['close'] - df['open']
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * typical_price).cumsum() / df['volume'].cumsum()
    high_low_to_open_close_ratio = high_low_range / abs(open_close_movement)
    vwap_deviation = df['close'] - vwap
    
    positive_momentum = high_low_range > 1.5 * abs(open_close_movement)
    potential_reversal = open_close_movement > 0.8 * high_low_range
    strong_reversal = (vwap_deviation < 0) & (open_close_movement < 0)

    # Calculate Intraday High-Low Range and Closing Position
    intraday_high_low_range = df['high'] - df['low']
    closing_position_in_range = (df['close'] - df['open']) / intraday_high_low_range

    # Volume Analysis
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).median()
    volume_trend = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=20).mean()

    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend_adjustment = ema_20 - df['close']

    # Additional Price Analysis
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=13, adjust=False).mean()
    ema_down = down.ewm(com=13, adjust=False).mean()
    rsi = 100 - (100 / (1 + (ema_up / ema_down)))
    
    ma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = ma_20 + 2 * std_20
    lower_band = ma_20 - 2 * std_20

    # Combine Factors
    combined_factors = volume_intensity * closing_position_in_range
    combined_factors += recent_price_trend_adjustment
    weighted_sum_of_scores = 0.6 * high_low_to_open_close_ratio + 0.3 * volume_trend + 0.1 * intraday_high_low_range

    # Final Alpha Factor Combination
    final_alpha_factor = combined_factors * weighted_sum_of_scores
    sentiment_score = 0.2 * 50  # Assuming a fixed sentiment score for simplicity
    final_alpha_factor += sentiment_score
    final_alpha_factor += recent_price_trend_adjustment + (df['close'] - ma_20) / std_20

    return final_alpha_factor
