```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_sum_10 = df['volume'].rolling(window=10).sum()
    volume_median_10 = df['volume'].rolling(window=10).median()
    volume_intensity = df['volume'] / ((volume_sum_10 + volume_median_10) / 2)
    
    # Combine Factors: Multiply Intraday Volume Intensity with Closing Position
    combined_factor = volume_intensity * close_position
    
    # Adjust for Recent Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    trend_adjustment = df['close'] - sma_20
    
    # Final Factor
    alpha_factor = combined_factor * trend_adjustment
    
    return alpha_factor
```
