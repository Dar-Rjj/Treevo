```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum().rank(pct=True) * 100
    df['monthly_return'] = df['close'].pct_change(periods=20)
    df['long_term_momentum'] = df['monthly_return'].rank(pct=True) * 100
    
    # Calculate Volume Activity Score
    df['volume_change'] = df['volume'].pct_change()
    df['short_term_volume'] = df['volume_change'].rolling(window=5).mean().rank(pct=True) * 100
    df['long_term_volume'] = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = (df['high'] - df['low'])
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Adjust for Recent Price Trend using EMA
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = df['close'] - df['ema_20']
    
    # Weighted Sum of Scores
    df['price_momentum_score'] = 0.6 * (df['short_term_momentum'] + df['long_term_momentum']) / 2
    df['volume_activity_score'] = 0.3 * (df['short_term_volume'] + df['long_term_volume']) / 2
    df['weighted_scores'] = df['price_momentum_score'] + df['volume_activity_score'] + 0.1 * df['intraday_range']
    
    # Combine Factors
    df['combined_factors'] = df['weighted_scores'] * df['price_trend']
    
    # Add a Sentiment Score
    # Assuming 'sentiment_score' is available in the DataFrame
    # If not, it should be computed or fetched from an external source
    df['alpha_factor'] = df['combined_factors'] + 0.2 * df['sentiment_score']
    
    return df['alpha_factor']
```
