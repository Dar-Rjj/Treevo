```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).median()
    
    # Combine closing position with volume intensity
    combined_factor = close_position * volume_intensity
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_trend = ema_20 - df['close']
    
    # Final factor combining all components
    final_factor = combined_factor + recent_trend
    
    return final_factor
```
