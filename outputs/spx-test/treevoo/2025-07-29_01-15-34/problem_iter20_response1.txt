```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on price and volume data.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'volume'.
        
    Returns:
        pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the difference between high and low prices
    range_diff = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of volume to the 5-day moving average of volume
    vol_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Combine the factors
    factor = (range_diff * close_pct_change) * vol_ratio
    
    return factor
```
