```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Adjusted Volume
    adjusted_volume = df['volume'] / df['close']
    
    # Calculate High-Low Price Momentum
    high_low_momentum = high_low_diff * adjusted_volume
    
    # Calculate Smoothed High-Low Price Momentum
    smoothed_high_low_momentum = high_low_momentum.rolling(window=5).mean()
    
    return smoothed_high_low_momentum
```
