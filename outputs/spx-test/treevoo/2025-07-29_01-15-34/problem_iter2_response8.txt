```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    # Weighted Ratio of High-Low to Open-Close, multiplied by Volume
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement)
    potential_reversal = (open_close_movement > 0.6 * high_low_range)
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Compute Volume-Weighted Intraday Momentum
    volume_weighted_intraday_momentum = (intraday_return * df['volume']) + high_low_range

    # Combine all the above factors into one predictive factor
    factor = (
        weighted_ratio * 
        positive_momentum.astype(int) * 
        potential_reversal.astype(int) * 
        volume_weighted_intraday_momentum
    )
    
    return factor
```
