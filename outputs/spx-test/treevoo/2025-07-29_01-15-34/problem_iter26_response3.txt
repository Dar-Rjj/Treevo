```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the close price over 5 days
    rolling_mean_close = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between the current close and the 5-day rolling mean
    diff_close_rolling_mean = df['close'] - rolling_mean_close
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the sum of the volume over the last 5 days
    rolling_sum_volume = df['volume'].rolling(window=5).sum()
    
    # Alpha factor: (difference between the current close and the 5-day rolling mean) * (daily return) / (rolling sum of volume)
    alpha_factor = (diff_close_rolling_mean * daily_return) / rolling_sum_volume
    
    return alpha_factor
```
