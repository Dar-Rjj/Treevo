```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Volume Adjusted Close Price
    volume_adjusted_close = df['close'] * df['volume']
    
    # Integrate with 10-day Moving Average of Close
    close_10_day_ma = df['close'].rolling(window=10).mean()
    integrated_close = (volume_adjusted_close + close_10_day_ma) / 2
    
    # Scale the High-to-Low Range by the Integrated Close Price
    factor_values = high_to_low_range / integrated_close
    
    return factor_values
```
