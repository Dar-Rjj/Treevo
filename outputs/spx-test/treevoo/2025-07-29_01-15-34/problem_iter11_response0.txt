```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / high_low_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Intraday Price Change
    price_change = (df['close'] - df['open']) / high_low_range
    
    # Combine Intraday Components
    intraday_components = (volume_intensity * close_position) + price_change
    
    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    trend_adjustment = ema_20 - df['close']
    
    # Combine Indicators for Predictive Factor
    weighted_ratio = high_low_range / (df['close'] - df['open']).abs()
    vwap_deviation = vwap - df['close']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > 1.5 * (df['close'] - df['open']))
    potential_reversal = ((df['close'] - df['open']) > 0.8 * high_low_range)
    strong_reversal = (vwap_deviation < 0) & (df['close'] - df['open'] < 0)
    
    predictive_factor = weighted_ratio + vwap_deviation
    predictive_factor[positive_momentum] += 1
    predictive_factor[potential_reversal] -= 0.5
    predictive_factor[strong_reversal] -= 1
    
    # Final Alpha Factor
    alpha_factor = intraday_components + trend_adjustment + predictive_factor
    
    return alpha_factor
```
