```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum and Trend Following
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_Crossover'] = df['SMA_20'] - df['SMA_50']
    
    df['ROC_14'] = (df['close'] / df['close'].shift(14)) - 1
    
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['EMA_Crossover'] = df['EMA_10'] - df['EMA_30']
    
    # Volatility
    df['TR'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1
    )
    df['ATR_20'] = df['TR'].rolling(window=20).mean()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Price Patterns
    df['Bullish_Engulfing'] = (df['open'] > df['close'].shift(1)) & (df['close'] < df['open'].shift(1))
    df['Bearish_Engulfing'] = (df['open'] < df['close'].shift(1)) & (df['close'] > df['open'].shift(1))
    
    df['Resistance_Level_Break'] = df['high'] > df['high'].rolling(window=30).max().shift(1)
    df['Support_Level_Break'] = df['low'] < df['low'].rolling(window=30).min().shift(1)
    
    # Volume Analysis
    df['OBV'] = ((df['close'] - df['close'].shift(1)) * df['volume']).cumsum()
    
    df['VWAP_t'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Market Sentiment
    df['Price_to_Volume_Ratio'] = (df['close'] / df['open']) / (df['volume'] / df['volume'].shift(1))
    
    # Combined Alpha Factor
    df['Alpha_Factor'] = (
        df['SMA_Crossover'] + 
        df['ROC_14'] + 
        df['EMA_Crossover'] + 
        df['ATR_20'] + 
        df['RSI_14'] + 
        df['Bullish_Engulfing'].astype(int) - 
        df['Bearish_Engulfing'].astype(int) + 
        df['Resistance_Level_Break'].astype(int) - 
        df['Support_Level_Break'].astype(int) + 
        df['OBV'] + 
        df['VWAP_t'] + 
        df['Price_to_Volume_Ratio']
    )
    
    return df['Alpha_Factor']
```
