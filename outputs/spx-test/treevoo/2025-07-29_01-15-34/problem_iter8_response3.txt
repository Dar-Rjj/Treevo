```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Intraday Range Percentage
    range_percentage = intraday_range / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    average_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend - average_volume
    
    # Combine Factors
    factor = (volume_intensity * closing_position) + \
             (df['close'] - df['close'].rolling(window=20).mean()) + \
             range_percentage + momentum_shift + volume_activity_ratio
    
    return factor
```
