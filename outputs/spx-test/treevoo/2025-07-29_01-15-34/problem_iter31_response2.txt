```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # High-Low to Open-Close Ratio
    high_low_open_close_ratio = high_low_range / open_close_movement.abs()
    
    # VWAP Deviation
    vwap_deviation = df['close'] - vwap
    
    # Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).median()
    
    # Trend and Reversal Signals
    positive_momentum = (high_low_range > 1.5 * open_close_movement.abs()).astype(int)
    strong_reversal = ((open_close_movement < 0) & (vwap_deviation < 0)).astype(int)
    
    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    trend_adjustment = df['close'] - ema_20
    
    # Composite Alpha Factor
    weighted_sum = (high_low_open_close_ratio * 0.3 +
                    vwap_deviation * 0.3 +
                    volume_intensity * 0.2 +
                    closing_position * 0.2)
    
    # Incorporate Recent Volatility
    recent_volatility = df['close'].rolling(window=20).std()
    trend_adjustment_normalized = trend_adjustment / recent_volatility
    
    # Final Alpha Factor
    alpha_factor = weighted_sum + trend_adjustment + trend_adjustment_normalized
    
    return alpha_factor
```
