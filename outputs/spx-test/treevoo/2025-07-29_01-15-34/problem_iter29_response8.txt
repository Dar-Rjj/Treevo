```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day Weighted Moving Average (WMA) of closing prices
    weights = pd.Series(range(1, 51), dtype=float)
    wma_50 = df['close'].rolling(window=50).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)

    # Use the difference between the current day's close price and the 50-day WMA as an initial alpha factor
    raw_alpha_factor = df['close'] - wma_50

    # Incorporate volume into the alpha factor to account for trading activity
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    volume_factor = df['volume'] / ema_20_volume

    # Combine the WMA and volume factors to create a composite alpha factor
    intermediate_alpha_factor = raw_alpha_factor * volume_factor

    # Incorporate high and low prices to capture intraday volatility
    true_range = df[['high', 'low', 'close']].pct_change(axis=1).abs().max(axis=1)
    atr_10 = true_range.rolling(window=10).mean()
    volatility_factor = (df['high'] - df['low']) / atr_10

    # Combine the WMA, volume, and volatility factors to create a comprehensive composite alpha factor
    final_alpha_factor = intermediate_alpha_factor * volatility_factor

    return final_alpha_factor
```
