import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close multiplied by Volume
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement).astype(int)
    potential_reversal = ((open_close_movement > 0.6 * high_low_range) & (open_close_movement > 0)).astype(int)
    
    # Calculate Intraday Range Percentage
    intraday_range_pct = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=5).sum()
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_activity_ratio = volume_trend / avg_volume
    
    # Calculate Price Momentum Score
    daily_return = df['close'].pct_change()
    short_term_price_score = (daily_return.rolling(window=5).sum() / 5) * 100
    weekly_return = df['close'].pct_change(5)
    long_term_price_score = (weekly_return + 1) ** (252/5) - 1
    price_momentum_score = 0.6 * short_term_price_score + 0.4 * long_term_price_score
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_score = (daily_volume_change.rolling(window=5).mean() / 5) * 100
    long_term_volume_score = (df['volume'].rolling(window=20).sum() / 20) * 100
    volume_activity_score = 0.6 * short_term_volume_score + 0.4 * long_term_volume_score
    
    # Price Volatility Factor
    daily_return_vol = df['close'].pct_change()
    daily_return_std = daily_return_vol.rolling(window=5).std()
    price_volatility_factor = daily_return_vol / daily_return_std
    
    # Composite Alpha Factor
    composite_alpha = 0.6 * price_momentum_score + 0.4 * volume_activity_score
    final_alpha_factor = composite_alpha / price_volatility_factor
    
    return final_alpha_factor
