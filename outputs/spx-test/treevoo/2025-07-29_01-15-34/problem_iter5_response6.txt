import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = df['close'].apply(lambda x: (x / df['close'].shift(1)).fillna(1)).apply(np.log)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    df['combined_factor'] = df['log_return'] * df['volume_change']
    
    # Compute Average of Combined Factor over 10 Days
    df['avg_combined_factor_10d'] = df['combined_factor'].rolling(window=10).mean()
    
    # Calculate Intraday Range Percentage
    df['intraday_range_pct'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate Close-to-Open Momentum Shift
    df['close_to_open_momentum'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    N = 20
    df['volume_trend'] = df['volume'].rolling(window=N).sum()
    df['avg_volume_20d'] = df['volume'].rolling(window=N).mean()
    df['volume_activity_ratio'] = df['volume_trend'] / df['avg_volume_20d']
    
    # Calculate Intraday Volume Intensity
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['intraday_volume_intensity'] = df['volume'] / df['avg_volume_10d']
    
    # Combine Factors
    df['factor'] = df['intraday_volume_intensity'] * df['closing_position'] * (df['close'] - df['close'].rolling(window=20).mean())
    
    return df['factor']
