```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Calculate Intraday Volume Intensity
    df['volume_intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Price Change
    df['intraday_price_change'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Intraday Range Percentage
    df['intraday_range_percentage'] = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    df['momentum_shift'] = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    df['volume_activity_ratio'] = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    
    # Daily Return Volatility
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volatility'] = df['daily_return'] / df['daily_return'].rolling(window=20).std()
    
    # Combine Intraday Components
    df['intraday_components'] = (df['volume_intensity'] * df['closing_position']) + df['intraday_price_change'] + df['momentum_shift']
    
    # Adjust for Recent Price Trend
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['price_trend'] = df['close'] - df['ema_20']
    
    # Incorporate Daily Return Volatility
    df['incorporated_volatility'] = df['intraday_components'] * df['volatility'] + df['price_trend']
    
    # Add VWAP Component
    vwap_numerator = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap_denominator = df['volume'].rolling(window=10).sum()
    df['vwap'] = vwap_numerator.rolling(window=10).sum() / vwap_denominator
    df['vwap_component'] = df['close'] - df['vwap']
    
    # Final Alpha Factor
    df['final_factor'] = df['incorporated_volatility'] + df['vwap_component']
    df['final_factor'] *= df['volume_activity_ratio']
    
    # Incorporate Volume Spike Indicator
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].rolling(window=10).mean()
    df['final_factor'] += df['volume_spike']
    
    # Integrate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    df['final_factor'] *= rsi
    
    return df['final_factor']
```
