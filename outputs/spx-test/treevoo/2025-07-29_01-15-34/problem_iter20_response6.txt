```python
def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the range as the difference between high and low
    price_range = df['high'] - df['low']
    
    # Define a factor which is the ratio of the average price to the price range
    factor = avg_price / price_range

    return factor
```
