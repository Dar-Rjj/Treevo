```python
import pandas as pd

def heuristics_v2(df):
    # Compute the 50-day rolling average of high prices
    df['high_50_day_rolling_avg'] = df['high'].rolling(window=50).mean()
    
    # Calculate the raw alpha factor
    df['raw_alpha_factor'] = df['close'] - df['high_50_day_rolling_avg']
    
    # Compute the 20-day EMA of volume
    df['volume_20_day_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Create a volume-based alpha factor
    df['volume_factor'] = df['volume'] / df['volume_20_day_ema']
    
    # Combine the momentum and volume factors
    df['composite_alpha_factor'] = df['raw_alpha_factor'] * df['volume_factor']
    
    return df['composite_alpha_factor']
```
