```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_sum = df['volume'].rolling(window=10).sum()
    volume_median = df['volume'].rolling(window=10).median()
    volume_intensity = df['volume'] / ((volume_sum + volume_median) / 2)
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum = (df['close'] - df['open']) / df['close']
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    volume_average = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = volume_trend - volume_average
    
    # Adjust for Recent Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    recent_price_trend = sma_20 - df['close']
    
    # Combine Factors
    factor = (volume_intensity * closing_position +
              recent_price_trend +
              intraday_range_percentage +
              close_to_open_momentum)
    
    return factor
```
