```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Identify Trend Strength and Reversal Signals
    trend_strength = high_low_range > open_close_movement
    potential_reversal = open_close_movement > 0.6 * high_low_range
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Change
    combined_factor = daily_log_return * volume_change
    avg_combined_factor = combined_factor.rolling(window=10).mean()
    
    # Intraday Range Percentage
    intraday_range_pct = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    close_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Volume Activity Ratio
    n_period_sum_volume = df['volume'].rolling(window=10).sum()
    n_period_avg_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = n_period_sum_volume / n_period_avg_volume
    
    # Calculate Intraday Volume Intensity
    intraday_vol_intensity = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Price Volatility Factor
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility_factor = daily_return.rolling(window=10).std()
    price_volatility_factor = daily_return / volatility_factor
    
    # Adjust for Recent Price Trend
    recent_price_trend = df['close'] - df['close'].ewm(span=20).mean()
    
    # Incorporate Volume Spike Indicator
    volume_spike_indicator = (df['volume'] - df['volume'].shift(1)) / df['volume'].rolling(window=10).mean()
    
    # Integrate Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Intraday Components
    intraday_components = (intraday_vol_intensity * closing_position) + intraday_range_pct + close_open_momentum_shift
    
    # Incorporate Daily Return Volatility
    intraday_components_with_volatility = intraday_components * price_volatility_factor + recent_price_trend
    
    # Final Factor
    final_factor = intraday_components_with_volatility * rsi + (daily_log_return * volume_change)
    
    return final_factor
```
