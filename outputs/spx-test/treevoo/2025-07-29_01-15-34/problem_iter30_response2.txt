import pandas as pd

def heuristics_v2(df):
    # Price-Based Factors
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    df['ROC_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5) * 100
    df['ROC_10'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10) * 100
    df['ROC_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20) * 100

    def rsi(series, period):
        delta = series.diff()
        gain = delta.where(delta > 0, 0)
        loss = -delta.where(delta < 0, 0)
        avg_gain = gain.rolling(window=period).mean()
        avg_loss = loss.rolling(window=period).mean()
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))

    df['RSI_5'] = rsi(df['close'], 5)
    df['RSI_10'] = rsi(df['close'], 10)
    df['RSI_20'] = rsi(df['close'], 20)

    # Volume-Based Factors
    df['OBV_Simple'] = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0)) * df['volume']
    df['OBV_Trend'] = df['OBV_Simple'].rolling(window=5).mean()

    df['Vol_Change_5'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5) * 100
    df['Vol_Change_10'] = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10) * 100
    df['Vol_Change_20'] = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20) * 100

    # Combined Factors
    df['PV_Ratio'] = df['close'] / df['volume']
    df['PV_Ratio_MA'] = df['PV_Ratio'].rolling(window=5).mean()

    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_MA'] = df['High_Low_Range'].rolling(window=5).mean()

    # Custom Factors
    df['Open_Close_Spread'] = df['open'] - df['close']
    df['Open_Close_Spread_MA'] = df['Open_Close_Spread'].rolling(window=5).mean()

    df['Close_Vol_Correl'] = df['close'].rolling(window=5).corr(df['volume'])
    df['Close_Vol_Correl_10'] = df['close'].rolling(window=10).corr(df['volume'])
    df['Close_Vol_Correl_20'] = df['close'].rolling(window=20).corr(df['volume'])

    # Novel Alpha Factor
    df['Novel_Alpha_Factor'] = (df['SMA_5'] - df['SMA_20']) + (df['RSI_10'] - 50) + (df['Vol_Change_5'] - df['Vol_Change_20']) + (df['High_Low_Range_MA'] / df['close']) + (df['Close_Vol_Correl_10'] - df['Close_Vol_Correl_20'])

    return df['Novel_Alpha_Factor']
