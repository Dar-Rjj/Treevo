```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Price Change
    intraday_price_change = (df['close'] - df['open']) / intraday_range
    
    # Intraday Range Percentage
    intraday_range_percentage = intraday_range / df['close']
    
    # Close-to-Open Momentum Shift
    close_to_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    volume_activity_ratio = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    
    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_volatility = daily_return / daily_return.rolling(window=10).std()
    
    # Combine Intraday Components
    combined_intraday = (volume_intensity * closing_position) + intraday_price_change + close_to_open_momentum_shift
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_trend_adjustment = ema_20 - df['close']
    
    # Incorporate Daily Return Volatility
    intraday_adjusted = combined_intraday * daily_return_volatility + recent_trend_adjustment
    
    # Trend Reversal Potential
    rsi_14 = 100 - (100 / (1 + df['close'].diff().rolling(window=14).apply(lambda x: x[x > 0].mean() / x[x < 0].abs().mean())))
    roc_14 = df['close'].pct_change(periods=14)
    trend_reversal_potential = 0
    trend_reversal_potential += np.where(rsi_14 > 70, 1, 0) * np.where(roc_14 < 0, 1, 0)
    trend_reversal_potential -= np.where(rsi_14 < 30, 1, 0) * np.where(roc_14 > 0, 1, 0)
    
    # Momentum Reversal Factor
    sma_5 = df['close'].rolling(window=5).mean()
    momentum_reversal_factor = (df['close'] - sma_5) / sma_5
    
    # Final Alpha Factor
    final_alpha_factor = 0.4 * intraday_adjusted + 0.3 * trend_reversal_potential + 0.3 * momentum_reversal_factor
    
    return final_alpha_factor
```
