```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Short-term Price Movement
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()
    df['rank_short_term_momentum'] = df['short_term_momentum'].rank(pct=True) * 100
    
    # Long-term Price Movement
    df['weekly_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['rank_weekly_return'] = df['weekly_return'].rank(pct=True) * 100
    
    # Combine Price Momentum Scores
    df['price_momentum_score'] = df['rank_short_term_momentum'] + df['rank_weekly_return']
    
    # Calculate Volume Changes
    df['daily_volume_change'] = df['volume'].pct_change()
    
    # Short-term Volume Change
    df['avg_daily_vol_change'] = df['daily_volume_change'].rolling(window=5).mean()
    df['rank_avg_daily_vol_change'] = df['avg_daily_vol_change'].rank(pct=True) * 100
    
    # Long-term Volume Activity
    df['sum_vol_20_days'] = df['volume'].rolling(window=20).sum()
    df['rank_sum_vol_20_days'] = df['sum_vol_20_days'].rank(pct=True) * 100
    
    # Combine Volume Activity Scores
    df['volume_activity_score'] = df['rank_avg_daily_vol_change'] + df['rank_sum_vol_20_days']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position_in_range'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Intraday Volume Intensity
    median_vol_10_days = df['volume'].rolling(window=10).median()
    df['intraday_vol_intensity'] = df['volume'] / median_vol_10_days
    
    # Adjust for Recent Price Trend
    df['ema_20'] = df['close'].ewm(span=20).mean()
    df['price_trend_adjustment'] = df['close'] - df['ema_20']
    
    # Combine Factors
    df['combined_factors'] = df['intraday_vol_intensity'] * df['closing_position_in_range'] * df['price_trend_adjustment']
    
    # Weighted Sum of Scores
    df['weighted_scores'] = 0.7 * df['price_momentum_score'] + 0.3 * df['volume_activity_score']
    
    # Final Alpha Factor Combination
    df['final_alpha_factor'] = df['combined_factors'] * df['weighted_scores']
    
    return df['final_alpha_factor']
```
