```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']

    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()

    # Combine Indicators for Predictive Factor
    weighted_ratio = high_low_range / np.abs(open_close_movement)
    
    vwap_deviation = vwap - df['close']
    
    positive_momentum = (high_low_range > 1.5 * np.abs(open_close_movement)).astype(int)
    potential_reversal = ((np.abs(open_close_movement) > 0.8 * high_low_range) & (vwap_deviation < 0) & (open_close_movement < 0)).astype(int)
    momentum_continuation = ((high_low_range > 2 * np.abs(open_close_movement)) & (vwap > df['close'])).astype(int)
    reversal_confirmation = ((open_close_movement < -0.5 * high_low_range) & (vwap < df['close'])).astype(int)

    # Calculate Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']

    # Calculate Closing Position in Intraday Range
    closing_position_in_range = (df['close'] - df['open']) / (df['high'] - df['low'])

    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=20).sum()
    average_volume = df['volume'].rolling(window=20).mean()
    volume_activity_ratio = volume_trend / average_volume

    # Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()

    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend = df['close'] - ema_20

    # Incorporate Recent Volatility
    std_20 = df['close'].rolling(window=20).std()
    recent_volatility_adjustment = recent_price_trend / std_20

    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_volatility = daily_return / daily_return.rolling(window=20).std()

    # Trend Reversal Potential
    rsi = df['close'].diff().apply(lambda x: np.where(x > 0, x, 0)).rolling(window=14).mean() / \
          np.abs(df['close'].diff()).rolling(window=14).mean()
    overbought_flag = (rsi > 70).astype(int)
    oversold_flag = (rsi < 30).astype(int)
    roc_14 = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    trend_reversal_potential = (overbought_flag & (roc_14 < 0)).astype(int) - (oversold_flag & (roc_14 > 0)).astype(int)

    # Momentum Reversal Factor
    sma_5 = df['close'].rolling(window=5).mean()
    momentum_reversal_factor = (df['close'] - sma_5) / sma_5

    # Final Alpha Factor
    weighted_sum = (weighted_ratio * 0.3) + (vwap_deviation * 0.3) + (intraday_volume_intensity * 0.2) + (closing_position_in_range * 0.2)
    trend_adjustment = recent_volatility_adjustment
    combined_intraday_components = (weighted_sum + trend_adjustment) * daily_return_volatility
    final_alpha = combined_intraday_components + trend_reversal_potential + momentum_reversal_factor

    return final_alpha
```
