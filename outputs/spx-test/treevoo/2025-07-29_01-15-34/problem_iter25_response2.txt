```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Intraday Volume Intensity
    volume_10day_sum = df['volume'].rolling(window=10).sum()
    intraday_volume_intensity = df['volume'] / volume_10day_sum
    
    # Intraday Price Change
    intraday_price_change = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Close-to-Open Momentum Shift
    close_to_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Combined Intraday Components
    combined_intraday_components = (intraday_volume_intensity * closing_position) + \
                                   intraday_price_change + \
                                   close_to_open_momentum
    
    # Recent Price Trend
    recent_price_trend = df['close'].ewm(span=20).mean() - df['close']
    
    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_volatility = daily_return / daily_return.rolling(window=20).std()
    
    # Adjusted Intraday Components
    adjusted_intraday_components = combined_intraday_components * daily_return_volatility + recent_price_trend
    
    # Trend Reversal Potential
    rsi = df['close'].rolling(window=14).apply(lambda x: 100 - (100 / (1 + (x.diff().clip(lower=0).sum() / -x.diff().clip(upper=0).sum()))))
    roc = df['close'].pct_change(periods=14)
    trend_reversal_potential = np.where((rsi > 70) & (roc < 0), 1, 0) + np.where((rsi < 30) & (roc > 0), -1, 0)
    
    # Momentum Reversal Factor
    sma_5_days = df['close'].rolling(window=5).mean()
    momentum_reversal_factor = (df['close'] - sma_5_days) / sma_5_days
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * adjusted_intraday_components + 0.3 * trend_reversal_potential + 0.2 * momentum_reversal_factor
    
    return final_alpha_factor
```
