```python
import pandas as pd

def heuristics_v2(df):
    # Momentum and Trend Following
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['SMA_50'] - df['SMA_200']
    
    df['ROC_10'] = (df['close'] / df['close'].shift(10)) - 1
    
    # Volatility
    df['TR'] = df.apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Price Patterns
    df['Bullish_Engulfing'] = (df['open'] > df['close'].shift(1)) & (df['close'] < df['open'].shift(1)).astype(int)
    
    df['Resistance_Level_Break'] = (df['high'] > df['high'].rolling(window=30).max()).astype(int)
    
    # Volume Analysis
    df['OBV'] = ((df['close'] - df['close'].shift(1)) * df['volume']).cumsum()
    
    # Liquidity
    df['Bid_Ask_Spread'] = (df['ask_price'] - df['bid_price']).abs()
    
    # Market Sentiment
    df['Price_to_Volume_Ratio'] = (df['close'] / df['open']) / (df['volume'] / df['volume'].shift(1))
    
    # Fundamental Analysis
    df['Earnings_Surprise'] = (df['actual_earnings'] - df['estimated_earnings']) / df['estimated_earnings']
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['SMA_Crossover'] + 
        df['ROC_10'] + 
        df['ATR_14'] + 
        df['Bullish_Engulfing'] + 
        df['Resistance_Level_Break'] + 
        df['OBV'] + 
        df['Bid_Ask_Spread'] + 
        df['Price_to_Volume_Ratio'] + 
        df['Earnings_Surprise']
    )
    
    return alpha_factor
```
