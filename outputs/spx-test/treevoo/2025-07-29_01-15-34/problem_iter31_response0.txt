import pandas as pd

def heuristics_v2(df):
    # Intraday Price Movement Analysis
    high_low_range = df['high'] - df['low']
    open_close_movement = df['close'] - df['open']
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * typical_price).cumsum() / df['volume'].cumsum()
    high_low_to_open_close_ratio = high_low_range / open_close_movement.abs()
    vwap_deviation = df['close'] - vwap

    # Intraday Range Analysis
    range_percentage = (df['high'] - df['low']) / df['close']
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])

    # Volume Analysis
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).median()

    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = ema_20 - df['close']

    # Composite Alpha Factor
    weighted_sum = (high_low_to_open_close_ratio * 0.3 
                    + vwap_deviation * 0.3 
                    + volume_intensity * 0.2 
                    + closing_position * 0.1 
                    + (df['close'].diff() / df['close']).rolling(window=14).mean() * 0.1)  # Using RSI-approximation here

    # Momentum Indicator
    roc_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_indicator = weighted_sum + roc_10

    # Final Alpha
    final_alpha = momentum_indicator + ema_adjustment

    return final_alpha
