import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum().rank(pct=True) * 100
    df['monthly_return'] = df['close'].pct_change(periods=20)
    df['long_term_momentum'] = df['monthly_return'].rank(pct=True) * 100
    df['price_momentum_score'] = (df['short_term_momentum'] + df['long_term_momentum']) / 2

    # Calculate Volume Activity Score
    df['volume_change'] = df['volume'].pct_change()
    df['short_term_volume_activity'] = df['volume_change'].rolling(window=5).mean().rank(pct=True) * 100
    df['long_term_volume_activity'] = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    df['volume_activity_score'] = (df['short_term_volume_activity'] + df['long_term_volume_activity']) / 2

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']

    # Calculate Intraday Volume Intensity
    df['intraday_volume_intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()

    # Combine Factors
    df['recent_trend'] = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    combined_factors = (df['intraday_volume_intensity'] * df['closing_position'] +
                        df['recent_trend'] + df['intraday_range'])

    # Weighted Sum of Scores
    df['weighted_sum_scores'] = (df['price_momentum_score'] * 0.6 +
                                 df['volume_activity_score'] * 0.3 +
                                 (df['intraday_range'] / df['intraday_range'].max()) * 0.1)

    # Final Alpha Factor Combination
    final_factor = combined_factors * df['weighted_sum_scores']
    
    # Add a Sentiment Score (assuming a placeholder function for sentiment score)
    def compute_sentiment_score():
        # Placeholder for an actual news sentiment computation
        return np.random.normal(size=len(df))
    
    sentiment_score = compute_sentiment_score()
    final_factor += sentiment_score * 0.2

    return final_factor.dropna()
