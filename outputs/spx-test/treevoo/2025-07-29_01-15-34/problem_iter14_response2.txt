```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14):
    # Intraday Range Intensity
    df['range_intensity'] = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    
    # Open-to-Close Momentum Shift
    df['prev_close'] = df['close'].shift(1)
    df['momentum_shift'] = (df['close'] - df['open']) / df['prev_close']
    
    # Volume Activity Ratio
    df['vol_sum_n'] = df['volume'].rolling(window=n).sum()
    df['vol_ema_n'] = df['volume'].ewm(span=n, adjust=False).mean()
    df['activity_ratio'] = df['vol_sum_n'] / df['vol_ema_n']
    
    # Daily Return Volatility Adjusted
    df['log_return'] = np.log(df['close'] / df['prev_close'])
    df['tr'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['atr'] = df['tr'].rolling(window=n).mean()
    df['vol_adjusted_return'] = df['log_return'] / df['atr']
    
    # Price Volatility Factor
    df['log_return_std'] = df['log_return'].rolling(window=n).std()
    df['gain'] = df['close'].diff().apply(lambda x: max(x, 0))
    df['loss'] = -df['close'].diff().apply(lambda x: min(x, 0))
    df['avg_gain'] = df['gain'].rolling(window=n).mean()
    df['avg_loss'] = df['loss'].rolling(window=n).mean()
    df['rs'] = df['avg_gain'] / df['avg_loss']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['range_intensity'] +
        df['momentum_shift'] +
        df['activity_ratio'] +
        df['vol_adjusted_return'] +
        df['log_return_std'] +
        df['rsi']
    )
    
    return df['alpha_factor'].dropna()
```
