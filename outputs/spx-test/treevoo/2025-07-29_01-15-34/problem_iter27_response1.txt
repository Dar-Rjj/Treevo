```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    short_term_returns = df['close'].pct_change().rolling(window=5).sum()
    short_term_rank = (short_term_returns.rank(pct=True) * 100).fillna(0)

    long_term_return = df['close'].pct_change(periods=63)
    long_term_rank = (long_term_return.rank(pct=True) * 100).fillna(0)

    price_momentum_score = (short_term_rank + long_term_rank) / 2

    # Calculate Volume Activity Score
    short_term_volume_change = df['volume'].pct_change().rolling(window=5).mean()
    short_term_vol_rank = (short_term_volume_change.rank(pct=True) * 100).fillna(0)

    long_term_volume = df['volume'].rolling(window=30).sum()
    long_term_vol_rank = (long_term_volume.rank(pct=True) * 100).fillna(0)

    volume_activity_score = (short_term_vol_rank + long_term_vol_rank) / 2

    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_high_low_range

    # Combine Factors
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend = ema_20 - df['close']

    combined_factors = (volume_activity_score * closing_position) + recent_price_trend

    # Weighted Sum of Scores
    weighted_scores = (price_momentum_score * 0.6) + (volume_activity_score * 0.3) + (intraday_high_low_range * 0.1)

    # Final Alpha Factor Combination
    final_alpha_factor = (combined_factors * weighted_scores)

    # Add a Sentiment Score
    news_sentiment = 50  # Placeholder for actual sentiment score
    sentiment_score = (news_sentiment / 100) * 100  # Convert to rank 0-100

    final_alpha_factor += sentiment_score * 0.1

    return final_alpha_factor
```
