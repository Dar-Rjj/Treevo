```python
import pandas as pd

def heuristics(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum().rank(pct=True) * 100
    df['monthly_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['long_term_momentum'] = df['monthly_return'].rank(pct=True) * 100
    df['price_momentum_score'] = 0.6 * df['short_term_momentum'] + 0.4 * df['long_term_momentum']

    # Calculate Volume Activity Score
    df['daily_volume_change'] = df['volume'].pct_change()
    df['avg_vol_change_5d'] = df['daily_volume_change'].rolling(window=5).mean().rank(pct=True) * 100
    df['total_vol_20d'] = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    df['volume_activity_score'] = 0.6 * df['avg_vol_change_5d'] + 0.4 * df['total_vol_20d']

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Calculate Intraday Volume Intensity
    df['intraday_vol_intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Calculate Intraday Range Percentage
    df['intraday_range_percentage'] = df['intraday_range'] / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    df['close_open_momentum_shift'] = (df['close'] - df['open']) / df['close']
    
    # Calculate Intraday Gap
    df['intraday_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = (
        df['intraday_vol_intensity'] * df['closing_position'] +
        (df['close'] - df['close'].ewm(span=20).mean()) +
        df['intraday_range_percentage'] + 
        df['close_open_momentum_shift'] + 
        df['intraday_gap']
    )
    
    # Weighted Sum of Scores
    df['weighted_scores'] = (
        0.6 * df['price_momentum_score'] + 
        0.3 * df['volume_activity_score'] + 
        0.1 * df['intraday_range']
    )
    
    # Final Alpha Factor Combination
    df['final_alpha_factor'] = df['combined_factors'] * df['weighted_scores']
    
    return df['final_alpha_factor']
```
