```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Compute Moving Average of Returns
    moving_avg_period = 20
    df['moving_avg_returns'] = df['daily_return'].rolling(window=moving_avg_period).mean()
    
    # Calculate Volatility
    volatility_period = 20
    df['volatility'] = df['daily_return'].rolling(window=volatility_period).std()
    
    # Determine Momentum Strength
    df['momentum_strength'] = df['moving_avg_returns'] / df['volatility']
    
    # Apply Volume Trend Adjustment
    df['volume_trend'] = (df['volume'] - df['volume'].shift(1)).rolling(window=moving_avg_period).mean()
    df['adjusted_momentum_strength'] = df.apply(
        lambda row: row['momentum_strength'] * (1 + row['volume_trend'] / 100) if row['volume_trend'] > 0 else row['momentum_strength'] * (1 - row['volume_trend'] / 100),
        axis=1
    )
    
    # Smooth the Adjusted Momentum Strength
    smoothing_period = 5
    df['smoothed_adjusted_momentum_strength'] = df['adjusted_momentum_strength'].rolling(window=smoothing_period).mean()
    
    return df['smoothed_adjusted_momentum_strength']
```
