```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']

    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * typical_price).cumsum() / df['volume'].cumsum()

    # Combine Indicators for Predictive Factor
    weighted_ratio = high_low_range / np.abs(open_close_movement)
    
    # VWAP Deviation from Close
    vwap_deviation = df['close'] - vwap
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > 1.5 * np.abs(open_close_movement))
    potential_reversal = (np.abs(open_close_movement) > 0.8 * high_low_range) & (vwap_deviation < 0) & (open_close_movement < 0)

    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])

    # Integrate Volume Activity Ratio
    n_period_volume = df['volume'].rolling(window=10).sum()
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = n_period_volume / volume_ma

    # Calculate Intraday Volume Intensity
    median_10_day_volume = df['volume'].rolling(window=10).median()
    intraday_volume_intensity = df['volume'] / median_10_day_volume

    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = df['close'] - ema_20

    # Introduce Intraday Volatility Factor
    intraday_std = df[['high', 'low', 'open', 'close']].std(axis=1)
    twenty_day_std = df['close'].rolling(window=20).std()
    intraday_to_twenty_day_vol_ratio = intraday_std / twenty_day_std

    # Composite Alpha Factor
    weighted_sum = (weighted_ratio * 0.3) + (vwap_deviation * 0.3) + (intraday_volume_intensity * 0.2) + (closing_position * 0.2)
    trend_adjustment = ema_adjustment / twenty_day_std
    recent_volatility = intraday_to_twenty_day_vol_ratio * 0.1
    final_alpha_factor = weighted_sum + trend_adjustment + recent_volatility

    return final_alpha_factor
```
