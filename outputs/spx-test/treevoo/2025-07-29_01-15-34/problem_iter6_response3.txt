```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    close_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Short-term Price Movement
    daily_returns = df['close'].pct_change()
    short_term_price_movement = daily_returns.rolling(window=5).sum()
    
    # Calculate Short-term Volume Change
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_change = daily_volume_change.rolling(window=5).mean()
    
    # Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Adjust for Recent Price Trend using EMA
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_trend_adjustment = ema_20 - df['close']
    
    # Combine Factors
    combined_factors = volume_intensity * close_position + recent_trend_adjustment
    
    # Weighted Sum of Scores
    weighted_score = 0.7 * short_term_price_movement + 0.3 * short_term_volume_change
    
    # Final Alpha Factor Combination
    final_alpha_factor = combined_factors * weighted_score
    
    return final_alpha_factor
```
