```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_high_low_range
    
    # Calculate Intraday Volume Intensity
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Intraday Price Change
    intraday_price_change = (df['close'] - df['open']) / intraday_high_low_range
    
    # Intraday Range Percentage
    intraday_range_percentage = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    volume_activity_ratio = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()
    
    # Daily Return Volatility
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    daily_return_volatility = daily_return / daily_return.rolling(window=10).std()
    
    # Combine Intraday Components
    combined_intraday = (volume_intensity * closing_position + 
                         intraday_price_change + 
                         momentum_shift)
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_trend_adjustment = ema_20 - df['close']
    
    # Incorporate Daily Return Volatility
    incorporated_volatility = combined_intraday * daily_return_volatility + recent_trend_adjustment
    
    # Add Volume Weighted Average Price (VWAP) Component
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_component = df['close'] - vwap
    
    # Final Alpha Factor
    alpha_factor = (incorporated_volatility + vwap_component) * volume_activity_ratio
    
    return alpha_factor
```
