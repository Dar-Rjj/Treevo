```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Volume Activity Ratio
    volume_trend = df['volume'].rolling(window=10).sum()
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = volume_trend / avg_volume
    
    # Calculate Price Momentum Score
    daily_return = df['close'].pct_change()
    short_term_momentum = daily_return.rolling(window=5).sum() * 100
    weekly_return = df['close'].pct_change(periods=7) * 100
    long_term_momentum = (weekly_return + 1).pow(1/7) * 100
    price_momentum_score = (short_term_momentum + long_term_momentum) / 2
    
    # Calculate Volume Activity Score
    daily_volume_change = df['volume'].pct_change()
    short_term_volume_activity = daily_volume_change.rolling(window=5).mean() * 100
    long_term_volume_activity = df['volume'].rolling(window=20).sum() / df['volume'].rolling(window=20).median() * 100
    volume_activity_score = (short_term_volume_activity + long_term_volume_activity) / 2
    
    # Calculate Intraday Volume Intensity
    median_10_day_volume = df['volume'].rolling(window=10).median()
    intraday_volume_intensity = df['volume'] / median_10_day_volume
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    recent_price_trend = ema_20 - df['close']
    
    # Composite Alpha Factor
    weighted_scores = price_momentum_score * 0.6 + volume_activity_score * 0.4
    final_alpha_factor = (intraday_volume_intensity * closing_position + recent_price_trend) * weighted_scores
    
    return final_alpha_factor
```
