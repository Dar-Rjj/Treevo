```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / intraday_range
    
    # Calculate Intraday Volume Intensity
    intraday_volume_intensity = df['volume'] / df['volume'].rolling(window=10).sum()
    
    # Combine Factors: Multiply Intraday Volume Intensity with Closing Position
    combined_factor = intraday_volume_intensity * closing_position
    
    # Adjust for Recent Price Trend
    sma_20 = df['close'].rolling(window=20).mean()
    recent_price_trend_adjustment = df['close'] - sma_20
    
    # Add Intraday Range Percentage and Close-to-Open Momentum Shift
    intraday_range_percentage = intraday_range / df['open']
    close_to_open_momentum_shift = (df['close'] - df['open']) / df['close']
    
    # Final Alpha Factor
    alpha_factor = (combined_factor + 
                    recent_price_trend_adjustment + 
                    intraday_range_percentage + 
                    close_to_open_momentum_shift)
    
    return alpha_factor
```
