```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP Deviation from Close
    vwap_deviation = vwap - df['close']
    
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = high_low_range / np.abs(open_close_movement)
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > 1.5 * np.abs(open_close_movement))
    potential_reversal = (np.abs(open_close_movement) > 0.8 * high_low_range) & (vwap_deviation < 0) & (open_close_movement < 0)
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = high_low_range / df['close']
    
    # Calculate Closing Position in Intraday Range
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Volume Activity Ratio
    n_period_volume_sum = df['volume'].rolling(window=10).sum()
    n_period_volume_mean = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = n_period_volume_sum / n_period_volume_mean
    
    # Calculate Intraday Volume Intensity
    median_10d_volume = df['volume'].rolling(window=10).median()
    intraday_volume_intensity = df['volume'] / median_10d_volume
    
    # Adjust for Recent Price Trend
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = df['close'] - ema_20
    
    # Incorporate Recent Volatility
    recent_volatility = df['close'].rolling(window=20).std()
    ema_adjustment_volatility = ema_adjustment / recent_volatility
    
    # Composite Alpha Factor
    weighted_sum = (weighted_ratio * 0.3 +
                    vwap_deviation * 0.3 +
                    intraday_volume_intensity * 0.2 +
                    closing_position * 0.2)
    
    trend_adjustment = ema_adjustment_volatility
    final_alpha_factor = weighted_sum + trend_adjustment
    
    return final_alpha_factor
```
