import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum().rank(pct=True) * 100
    df['monthly_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['long_term_momentum'] = df['monthly_return'].rank(pct=True) * 100
    df['price_momentum_score'] = (0.6 * df['short_term_momentum']) + (0.4 * df['long_term_momentum'])

    # Calculate Volume Activity Score
    df['daily_volume_change'] = df['volume'].pct_change()
    df['short_term_vol_change'] = df['daily_volume_change'].rolling(window=5).mean().rank(pct=True) * 100
    df['long_term_vol_activity'] = df['volume'].rolling(window=20).sum().rank(pct=True) * 100
    df['volume_activity_score'] = (0.6 * df['short_term_vol_change']) + (0.4 * df['long_term_vol_activity'])

    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['open']) / df['intraday_range']

    # Calculate Intraday Volume Intensity
    df['intraday_vol_intensity'] = df['volume'] / df['volume'].rolling(window=10).sum()

    # Calculate Intraday Range Percentage
    df['intraday_range_pct'] = df['intraday_range'] / df['open']

    # Calculate Close-to-Open Momentum Shift
    df['close_to_open_momentum'] = (df['close'] - df['open']) / df['close']

    # Combine Factors
    df['combined_factors'] = (df['intraday_vol_intensity'] * df['closing_position']) + \
                             (df['close'] - df['close'].ewm(span=20, adjust=False).mean()) + \
                             df['intraday_range_pct'] + df['close_to_open_momentum']

    # Weighted Sum of Scores
    df['weighted_sum_scores'] = (0.6 * df['price_momentum_score']) + \
                                (0.3 * df['volume_activity_score']) + \
                                (0.1 * df['intraday_range'])

    # Final Alpha Factor Combination
    df['alpha_factor'] = df['combined_factors'] * df['weighted_sum_scores']
    
    # Add a Sentiment Score (Placeholder for actual sentiment score calculation)
    df['sentiment_score'] = 50  # Placeholder value, replace with actual sentiment score
    df['alpha_factor'] += 0.2 * df['sentiment_score']

    return df['alpha_factor']
