```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Moving Average of Returns
    moving_avg_period = 20
    df['moving_avg_returns'] = df['daily_returns'].rolling(window=moving_avg_period).mean()
    
    # Calculate Volatility
    volatility_period = 20
    df['volatility'] = df['daily_returns'].rolling(window=volatility_period).std()
    
    # Determine Momentum Strength
    df['momentum_strength'] = df['moving_avg_returns'] / df['volatility']
    
    # Apply Adaptive Weighting
    volume_normalization_period = 20
    df['normalized_volume'] = df['volume'] / df['volume'].rolling(window=volume_normalization_period).mean()
    df['adaptive_weighted_momentum'] = df['momentum_strength'] * df['normalized_volume']
    
    # Smooth the Adaptive Weights
    smoothing_period = 5
    df['smoothed_adaptive_weights'] = df['adaptive_weighted_momentum'].rolling(window=smoothing_period).mean()
    
    return df['smoothed_adaptive_weights'].dropna()
```
