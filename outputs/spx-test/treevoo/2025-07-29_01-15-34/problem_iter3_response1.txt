```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Short-term Price Movement
    df['Sum_Daily_Returns_5D'] = df['Daily_Return'].rolling(window=5).sum()
    df['Short_Term_Price_Rank'] = df['Sum_Daily_Returns_5D'].rank(pct=True) * 100
    
    # Long-term Price Movement
    df['Weekly_Return'] = df['close'].pct_change(periods=5)
    df['Long_Term_Price_Rank'] = df['Weekly_Return'].rank(pct=True) * 100
    
    # Compute Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'].pct_change()
    
    # Short-term Volume Change
    df['Avg_Daily_Volume_Change_5D'] = df['Daily_Volume_Change'].rolling(window=5).mean()
    df['Short_Term_Volume_Rank'] = df['Avg_Daily_Volume_Change_5D'].rank(pct=True) * 100
    
    # Long-term Volume Activity
    df['Sum_Volumes_20D'] = df['volume'].rolling(window=20).sum()
    df['Long_Term_Volume_Rank'] = df['Sum_Volumes_20D'].rank(pct=True) * 100
    
    # Combine Ranks
    df['Price_Momentum_Score'] = (df['Short_Term_Price_Rank'] + df['Long_Term_Price_Rank']) / 2
    df['Volume_Activity_Score'] = (df['Short_Term_Volume_Rank'] + df['Long_Term_Volume_Rank']) / 2
    
    # Composite Alpha Factor
    df['Composite_Alpha_Factor'] = 0.7 * df['Price_Momentum_Score'] + 0.3 * df['Volume_Activity_Score']
    
    return df['Composite_Alpha_Factor']
```
