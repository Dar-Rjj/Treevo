```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Intraday Range Percentage
    intraday_range = (df['high'] - df['low']) / df['close']
    
    # Close-to-Open Momentum Shift
    momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Volume Activity Ratio
    volume_n_sum = df['volume'].rolling(window=n).sum()
    volume_n_ma = df['volume'].rolling(window=n).mean()
    volume_activity_ratio = volume_n_sum / volume_n_ma
    
    # Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Daily Return Volatility
    daily_return_volatility = daily_return / daily_return.rolling(window=n).std()

    # Combine all factors into a single alpha factor
    alpha_factor = intraday_range + momentum_shift + volume_activity_ratio + daily_return_volatility
    
    return alpha_factor
```
