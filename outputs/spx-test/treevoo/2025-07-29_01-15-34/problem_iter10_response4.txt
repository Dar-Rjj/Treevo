```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement)
    potential_reversal = (open_close_movement > 0.7 * high_low_range)
    
    # Combine Indicators for Predictive Factor
    combined_factor = (weighted_ratio * positive_momentum) - (weighted_ratio * potential_reversal)
    
    # Calculate Intraday Range Percentage
    intraday_range_percentage = high_low_range / df['close']
    
    # Calculate Close-to-Open Momentum Shift
    close_to_open_momentum_shift = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Activity Ratio
    n_period_sum_volume = df['volume'].rolling(window=10).sum()
    n_period_avg_volume = df['volume'].rolling(window=10).mean()
    volume_activity_ratio = n_period_sum_volume / n_period_avg_volume
    
    # Incorporate Amount and Volume Interaction
    amount_per_volume = df['amount'] / df['volume']
    volume_weighted_amount = (close_to_open_momentum_shift * amount_per_volume) * df['volume']
    
    # Final Alpha Factor
    alpha_factor = (combined_factor + intraday_range_percentage + close_to_open_momentum_shift 
                    + volume_activity_ratio + volume_weighted_amount)
    
    return alpha_factor
```
