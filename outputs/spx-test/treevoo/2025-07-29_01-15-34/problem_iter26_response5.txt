```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values indexed by date.
    """
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the percentage of the close price within the daily range
    close_percent_range = (df['close'] - df['low']) / price_range
    
    # Calculate the ratio of volume to the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Combine the two signals
    factor = close_percent_range * volume_ratio
    
    return factor
```
