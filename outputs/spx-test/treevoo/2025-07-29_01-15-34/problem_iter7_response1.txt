```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Open to Close Movement
    open_close_movement = df['close'] - df['open']
    
    # Combine Indicators for Predictive Factor
    # Weighted Ratio of High-Low to Open-Close
    weighted_ratio = (high_low_range / open_close_movement) * df['volume']
    
    # Identify Trend Strength and Reversal Signals
    positive_momentum = (high_low_range > open_close_movement)
    potential_reversal = (open_close_movement > 0.6 * high_low_range)
    
    # Compute Volume-Weighted Intraday Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_weighted_intraday_momentum = intraday_return * df['volume']
    
    # Final Alpha Factor
    alpha_factor = (
        weighted_ratio * positive_momentum + 
        volume_weighted_intraday_momentum * (~potential_reversal)
    )
    
    return alpha_factor
```
