```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Price Movement Analysis
    high_low_range = df['high'] - df['low']
    open_close_movement = df['close'] - df['open']
    
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_deviation = df['close'] - vwap
    
    movement_ratio = high_low_range / open_close_movement.abs()
    positive_momentum = (high_low_range > 1.5 * open_close_movement.abs())
    potential_reversal = (open_close_movement.abs() > 0.8 * high_low_range)
    strong_reversal = (vwap_deviation < 0) & (open_close_movement < 0)

    # Intraday Range Analysis
    range_percentage = (df['high'] - df['low']) / df['close']
    closing_position = (df['close'] - df['open']) / (df['high'] - df['low'])

    # Volume Analysis
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).median()
    volume_trend = df['volume'].rolling(window=5).sum() / df['volume'].rolling(window=10).mean()

    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = ema_20 - df['close']

    # Composite Alpha Factor
    weighted_sum = (movement_ratio * 0.3 +
                    vwap_deviation * 0.4 +
                    volume_intensity * 0.1 +
                    closing_position * 0.2)
    
    trend_strength = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    final_alpha = weighted_sum + ema_adjustment + trend_strength

    return final_alpha
```
