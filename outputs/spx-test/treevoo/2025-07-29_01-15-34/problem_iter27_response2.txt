import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate Exponential Moving Averages
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate Day-to-Day Price Change
    df['Price_Change_Day'] = df['close'].diff()

    # Calculate Week-to-Week Price Change
    df['Price_Change_Week'] = df['close'].diff(periods=5)

    # Calculate Month-to-Month Price Change
    df['Price_Change_Month'] = df['close'].diff(periods=20)

    # Calculate True Range
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x['high'], x['close'].shift(1)) - min(x['low'], x['close'].shift(1)), axis=1)

    # Calculate Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()

    # Calculate Day-to-Day Volume Change
    df['Volume_Change_Day'] = df['volume'].diff()

    # Calculate Week-to-Week Volume Change
    df['Volume_Change_Week'] = df['volume'].diff(periods=5)

    # Calculate Month-to-Month Volume Change
    df['Volume_Change_Month'] = df['volume'].diff(periods=20)

    # Calculate Volume Weighted Average Price (VWAP) over a 5-day period
    df['VWAP_5'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate Volume Weighted Average Price (VWAP) over a 10-day period
    df['VWAP_10'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Calculate Volume Weighted Average Price (VWAP) over a 20-day period
    df['VWAP_20'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Calculate Rate of Change (ROC)
    df['ROC_5'] = df['close'].pct_change(periods=5)
    df['ROC_10'] = df['close'].pct_change(periods=10)
    df['ROC_20'] = df['close'].pct_change(periods=20)

    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))

    # Calculate Moving Average Convergence Divergence (MACD)
    df['MACD_line'] = df['EMA_12'] - df['EMA_26']
    df['MACD_signal'] = df['MACD_line'].ewm(span=9, adjust=False).mean()
    df['MACD_histogram'] = df['MACD_line'] - df['MACD_signal']

    # Calculate Historical Volatility
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['HV_5'] = df
