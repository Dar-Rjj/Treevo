```python
import pandas as pd

def heuristics_v2(df):
    """
    A function to generate an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series with index as date and values as the generated alpha factor.
    """

    # Calculate the average of open, high, low, and close prices
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the price movement
    price_movement = df['close'] - df['open']

    # Generate the alpha factor
    # The heuristic here is that a higher price movement relative to the price range,
    # combined with a higher trading volume, may indicate strong momentum.
    alpha_factor = (price_movement / price_range) * df['volume'] * avg_price

    return alpha_factor
```
