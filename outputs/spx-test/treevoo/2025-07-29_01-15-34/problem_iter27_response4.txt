```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Price Movement Analysis
    high_low_range = df['high'] - df['low']
    open_close_movement = df['close'] - df['open']
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * typical_price).cumsum() / df['volume'].cumsum()
    high_low_open_close_ratio = high_low_range / open_close_movement.abs()
    vwap_deviation = df['close'] - vwap
    positive_momentum = (high_low_range > 1.5 * open_close_movement.abs()).astype(int)
    potential_reversal = ((open_close_movement > 0.8 * high_low_range) & (vwap_deviation < 0) & (open_close_movement < 0)).astype(int)

    # Intraday Range Analysis
    range_percentage = high_low_range / df['close']
    closing_position = open_close_movement / high_low_range

    # Volume Analysis
    volume_intensity = df['volume'] / df['volume'].rolling(window=10).median()
    volume_trend = df['volume'].rolling(window=10).sum() / df['volume'].rolling(window=10).mean()

    # Recent Price Trend Adjustment
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_adjustment = ema_20 - df['close']

    # Additional Price Analysis
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20

    # Composite Alpha Factor
    weighted_sum = (high_low_open_close_ratio * 0.3 +
                    vwap_deviation * 0.3 +
                    volume_intensity * 0.2 +
                    closing_position * 0.1 +
                    rsi * 0.1)

    roc_10 = df['close'].pct_change(periods=10)
    momentum_indicator = roc_10
    final_alpha = weighted_sum + ema_adjustment + (upper_band - df['close']) + (df['close'] - lower_band) + momentum_indicator

    return final_alpha
```
