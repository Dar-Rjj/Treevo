```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Score
    df['Daily_Return'] = df['close'].pct_change()
    df['Sum_Daily_Return_5d'] = df['Daily_Return'].rolling(window=5).sum()
    df['Price_Momentum_Short_Term_Rank'] = (df['Sum_Daily_Return_5d'].rank(pct=True) * 100).clip(0, 100)
    
    df['Monthly_Return'] = df['close'].pct_change(periods=20)
    df['Price_Momentum_Long_Term_Rank'] = (df['Monthly_Return'].rank(pct=True) * 100).clip(0, 100)
    
    # Calculate Volume Activity Score
    df['Daily_Volume_Change'] = df['volume'].pct_change()
    df['Average_Daily_Volume_Change_5d'] = df['Daily_Volume_Change'].rolling(window=5).mean()
    df['Volume_Activity_Short_Term_Rank'] = (df['Average_Daily_Volume_Change_5d'].rank(pct=True) * 100).clip(0, 100)
    
    df['Sum_Volume_20d'] = df['volume'].rolling(window=20).sum()
    df['Volume_Activity_Long_Term_Rank'] = (df['Sum_Volume_20d'].rank(pct=True) * 100).clip(0, 100)
    
    # Calculate Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['Closing_Position'] = (df['close'] - df['open']) / df['Intraday_High_Low_Range']
    
    # Calculate Intraday Volume Intensity
    df['Sum_Volume_10d'] = df['volume'].rolling(window=10).sum()
    df['Intraday_Volume_Intensity'] = df['volume'] / df['Sum_Volume_10d']
    
    # Calculate Intraday Range Percentage
    df['Intraday_Range_Percentage'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Momentum Shift
    df['Close_to_Open_Momentum_Shift'] = (df['close'] - df['open']) / df['close']
    
    # Calculate Intraday Gap
    df['Intraday_Gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Factors
    df['Combined_Factors'] = (
        df['Intraday_Volume_Intensity'] * df['Closing_Position'] +
        (df['close'] - df['close'].ewm(span=20, adjust=False).mean()) +
        df['Intraday_Range_Percentage'] + df['Close_to_Open_Momentum_Shift'] + df['Intraday_Gap']
    )
    
    # Weighted Sum of Scores
    df['Weighted_Score'] = (
        0.6 * df['Price_Momentum_Short_Term_Rank'] +
        0.3 * df['Volume_Activity_Short_Term_Rank'] +
        0.1 * df['Intraday_High_Low_Range']
    )
    
    # Final Alpha Factor Combination
    df['Final_Alpha_Factor'] = (
        df['Combined_Factors'] * df['Weighted_Score'] +
        0.2 * 50  # Placeholder for sentiment score, replace with actual sentiment score
    )
    
    return df['Final_Alpha_Factor']
```
