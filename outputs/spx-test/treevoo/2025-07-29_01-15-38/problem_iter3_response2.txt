```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_volume_5d
    volume_adjusted_momentum = short_term_momentum * relative_volume
    
    # Integrate Amount Influence
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    relative_amount = df['amount'] / avg_amount_5d
    final_factor = volume_adjusted_momentum * relative_amount
    
    return final_factor
```
