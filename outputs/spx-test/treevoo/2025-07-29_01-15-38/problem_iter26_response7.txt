```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_volume_5
    volume_adjusted_momentum = short_term_momentum * relative_volume
    
    # Weight by Volume Surge
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_surge_weight = 1.0
    volume_surge_weight[df['volume'] > 1.5 * avg_volume_10] = 1.5
    volume_surge_weight[(df['volume'] > avg_volume_10) & (df['volume'] <= 1.5 * avg_volume_10)] = 1.2
    weighted_momentum = volume_adjusted_momentum * volume_surge_weight
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    price_range_filter = (daily_price_range > 1.5 * avg_daily_price_range_20)
    weighted_momentum[price_range_filter] = 0.0
    
    # Apply Volatility Filter
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev_20 = daily_return.rolling(window=20).std()
    volatility_filter = (std_dev_20 > 0.03)
    weighted_momentum[volatility_filter] = 0.0
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=21).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Introduce High-Low Range Indicator
    high_low_range = df['high'] - df['low']
    avg_high_low_range_5 = high_low_range.rolling(window=5).mean()
    normalized_high_low_range = high_low_range / avg_high_low_range_5
    
    # Introduce Open-Close Range Indicator
    open_close_range = abs(df['open'] - df['close'])
    avg_open_close_range_5 = open_close_range.rolling(window=5).mean()
    normalized_open_close_range = open_close_range / avg_open_close_range_5
    
    # Final Alpha Factor
    final_factor = (weighted_momentum * momentum_oscillator * 
                    normalized_high_low_range * normalized_open_close_range)
    
    return final_factor
```
