```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily high-low spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Analyze the trend in the high-low spread
    df['high_low_spread_change'] = df['high_low_spread'].diff()
    
    # Evaluate the ratio (Close - Open) / (High - Low)
    df['close_open_ratio'] = (df['close'] - df['open']) / df['high_low_spread']
    
    # Determine the close-open difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Assess the Volume-to-Price Movement Correlation
    df['volume_change'] = df['volume'].diff()
    df['price_change'] = df['close'].diff()
    df['volume_price_corr'] = df['volume_change'] * df['price_change']
    
    # Evaluate the Volume-to-Amount Ratio
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Measure the change in volume from one period to the next
    df['volume_momentum'] = df['volume'].diff()
    
    # Combine high-low spread trends with volume momentum
    df['composite_factor'] = df['high_low_spread_change'] * df['volume_momentum']
    
    # Synthesize close-open relation with volume-to-amount ratio
    df['synthesized_factor'] = df['close_open_diff'] * df['volume_to_amount_ratio']
    
    # Final alpha factor
    df['alpha_factor'] = df['composite_factor'] + df['synthesized_factor']
    
    return df['alpha_factor']
```
