```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Weighted High-Low Spread
    df['high_low_spread'] = (df['high'] - df['low']) * df['volume']
    
    # Compute 5-Day Price Momentum
    df['5_day_price_momentum'] = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).sum() / 5
    df['relative_volume'] = df['volume'] / df['5_day_avg_volume']
    df['adjusted_5_day_momentum'] = df['5_day_price_momentum'] * df['relative_volume']
    
    # Analyze Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    df['volume_trend'] = df['volume_change'].apply(lambda x: 2 if x > 1.5 else -2 if x < 0.5 else 0)
    
    # Combine Factors Based on Volume Classification
    df['combined_factor'] = (df['intraday_return'] + 
                             df['high_low_spread'] + 
                             df['adjusted_5_day_momentum']) * df['volume_trend']
    
    # Apply Cumulative Movement
    df['cumulative_factor'] = df['combined_factor'].rolling(window=5).sum()
    
    # Integration of Additional Factors
    df['daily_high_low_range'] = df['high'] - df['low']
    df['5_day_avg_high_low_range'] = df['daily_high_low_range'].rolling(window=5).sum() / 5
    df['daily_volume_to_avg_volume_ratio'] = df['volume'] / df['5_day_avg_volume']
    
    # Combine Integrated Factors
    df['integrated_factor'] = (df['intraday_return'] + 
                               df['high_low_spread'] + 
                               df['5_day_price_momentum'] * df['volume_trend'] + 
                               df['daily_high_low_range'] - 
                               df['5_day_avg_high_low_range'] + 
                               df['daily_volume_to_avg_volume_ratio'])
    
    # Apply Cumulative Movement
    df['cumulative_integrated_factor'] = df['integrated_factor'].rolling(window=5).sum()
    
    # Advanced Factor Synthesis
    df['5_day_high_low_range_momentum'] = df['daily_high_low_range'].rolling(window=5).sum() / 5
    df['5_day_close_open_momentum'] = (df['close'] - df['open']).rolling(window=5).sum() / 5
    df['combined_price_volume_momentum'] = (df['5_day_price_momentum'] + 
                                            df['5_day_avg_high_low_range']) * df['relative_volume']
    df['volume_adjusted_close_open'] = (df['close'] - df['open']) * df['relative_volume']
    
    # Integrate All Factors
    df['advanced_factor'] = (df['intraday_return'] + 
                             df['high_low_spread'] + 
                             df['adjusted_5_day_momentum'] + 
                             df['daily_high_low_range'] - 
                             df['5_day_avg_high_low_range'] + 
                             df['daily_volume_to_avg_volume_ratio'] + 
                             df['volume_adjusted_close_open']) * df['volume_trend']
    
    # Apply Enhanced Cumulative Movement
    df['cumulative_advanced_factor'] = df['advanced_factor'].rolling(window=10).sum()
    
    return df['cumulative_advanced_factor']
```
