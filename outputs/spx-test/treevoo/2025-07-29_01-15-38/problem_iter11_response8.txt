```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = 0.6 * intraday_returns + 0.4 * open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = pd.Series(index=df.index, dtype='float64')
    volume_trend[volume_ratio > 2] = 3
    volume_trend[(volume_ratio >= 0.5) & (volume_ratio <= 2)] = 1
    volume_trend[volume_ratio < 0.5] = -3
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_returns * volume_trend
    
    # Short-Term Price Change
    short_term_price_change = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume
    avg_5_day_volume = df['volume'].rolling(window=5).mean().shift(1)
    short_term_adjusted_momentum = short_term_price_change * (df['volume'] / avg_5_day_volume)
    
    # Long-Term Price Momentum
    long_term_price_change = df['close'] - df['close'].shift(20)
    
    # Price Momentum Oscillator
    price_momentum_oscillator = short_term_adjusted_momentum - long_term_price_change
    
    # 10-Day Volume Surge
    current_10_day_avg_vol = df['volume'].rolling(window=10).mean()
    previous_10_day_avg_vol = df['volume'].rolling(window=10).mean().shift(10)
    
    volume_surge_weight = pd.Series(index=df.index, dtype='float64')
    volume_surge_weight[current_10_day_avg_vol > 1.5 * previous_10_day_avg_vol] = 1.5
    volume_surge_weight[(current_10_day_avg_vol > previous_10_day_avg_vol) & (current_10_day_avg_vol <= 1.5 * previous_10_day_avg_vol)] = 1.2
    volume_surge_weight[~((current_10_day_avg_vol > 1.5 * previous_10_day_avg_vol) | (current_10_day_avg_vol > previous_10_day_avg_vol))] = 1
    
    # Daily Price Range
    daily_price_range = df['high'] - df['low']
    avg_20_day_price_range = daily_price_range.rolling(window=20).mean()
    
    # Final Factor Calculation
    factor_value = price_momentum_oscillator * volume_surge_weight
    factor_value[daily_price_range > 1.5 * avg_20_day_price_range] = 0
    
    return factor_value
```
