import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Calculate Price Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Weight by Volume Surge
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    volume_weight = (volume_ratio > 1.5) * 1.5 + ((volume_ratio > 1) & (volume_ratio <= 1.5)) * 1.2 + (volume_ratio <= 1) * 1
    
    # Integrate Price Range
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    normalized_price_range = price_range / df['close']
    
    # Combine Indicators
    combined_indicator = momentum_oscillator * volume_weight * normalized_price_range
    
    # Calculate Volume-Adjusted High-Low Spread
    adjusted_high_low_spread = (df['high'] - df['low']) * (df['volume'] / df['volume'].rolling(window=10).mean())
    
    # Calculate High-Low Range Momentum
    high_low_momentum = adjusted_high_low_spread.diff().rolling(window=10).sum()
    
    # Calculate 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    
    # Combine Momentum Factors
    combined_momentum = high_low_momentum + adjusted_five_day_momentum
    
    # Filter by Intraday Return
    intraday_return = df['close'] - df['open']
    alpha_factor = combined_momentum.where(intraday_return > 0, 0)
    
    return alpha_factor
