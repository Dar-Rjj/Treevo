```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']
    
    # Analyze Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    df['volume_trend'] = 0
    df.loc[df['volume_ratio'] > 1.5, 'volume_trend'] = 2
    df.loc[(df['volume_ratio'] >= 0.8) & (df['volume_ratio'] <= 1.5), 'volume_trend'] = 0
    df.loc[df['volume_ratio'] < 0.8, 'volume_trend'] = -2
    
    # Adjust Combined Returns Based on Volume Classification
    df['adjusted_returns'] = df['combined_returns'] * df['volume_trend']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['volume_adjustment'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Weighted High-Low Spread
    df['weighted_high_low_spread'] = df['high_low_spread'] * df['volume_adjustment']
    
    # Integrate Volume with Intraday Dynamics
    df['volume_intraday_dynamics'] = (df['high'] - df['low']) * df['volume_adjustment']
    
    # Compute Hybrid Factor
    df['hybrid_factor'] = df['intraday_returns'] + df['weighted_high_low_spread']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum
    df['short_term_momentum'] = df['high_low_range'].rolling(window=5).sum()
    
    # Calculate Medium-Term High-Low Range Momentum
    df['medium_term_momentum'] = df['high_low_range'].rolling(window=10).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    df['long_term_momentum'] = df['high_low_range'].rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['momentum_oscillator'] = (df['short_term_momentum'] - df['medium_term_momentum']) / df['long_term_momentum']
    
    # Weight by Volume Surge
    average_volume = df['volume'].rolling(window=10).mean()
    df['volume_surge_ratio'] = df['volume'] / average_volume
    df['volume_weight'] = 0
    df.loc[df['volume_surge_ratio'] > 2, 'volume_weight'] = 2
    df.loc[(df['volume_surge_ratio'] > 1.7) & (df['volume_surge_ratio'] <= 2), 'volume_weight'] = 1.5
    df.loc[(df['volume_surge_ratio'] > 1) & (df['volume_surge_ratio'] <= 1.7), 'volume_weight'] = 1.2
    df.loc[df['volume_surge_ratio'] <= 1, 'volume_weight'] = 1
    
    # Filter by Close Price Increase
    df['filtered_oscillator'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'filtered_oscillator'] = df['momentum_oscillator']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['filtered_oscillator'] * df['volume_weight'] + df['hybrid_factor']
    
    return df['alpha_factor']
```
