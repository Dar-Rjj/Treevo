```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['AvgClose_5'] = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    df['AvgVolume_5'] = df['volume'].rolling(window=5).mean()
    df['AdjustedMomentum'] = df['AvgClose_5'] * (df['volume'] / df['AvgVolume_5'])
    
    # Apply Price Range Filter
    df['DailyPriceRange'] = df['high'] - df['low']
    df['AvgDailyPriceRange_20'] = df['DailyPriceRange'].rolling(window=20).mean()
    df['FilteredMomentum'] = df['AdjustedMomentum'].where(df['DailyPriceRange'] <= 1.5 * df['AvgDailyPriceRange_20'], 0)
    
    # Apply Volatility Filter
    df['DailyReturn'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['StdDev_DailyReturns_20'] = df['DailyReturn'].rolling(window=20).std()
    df['FilteredMomentum'] = df['FilteredMomentum'].where(df['StdDev_DailyReturns_20'] <= 0.03, 0)
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum
    df['ShortTermRangeMomentum'] = df['HighLowRange'].diff().rolling(window=5).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    df['LongTermRangeMomentum'] = df['HighLowRange'].diff().rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['RangeMomentumOscillator'] = df['ShortTermRangeMomentum'] - df['LongTermRangeMomentum']
    
    # Weight by Volume Surge
    df['AvgVolume_10'] = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / df['AvgVolume_10']
    df['VolumeWeight'] = volume_ratio.where(volume_ratio > 1.5, 1).where(volume_ratio > 1, 1.2).fillna(1.5)
    
    # Filter by Close Price Increase
    df['PriceIncrease'] = df['close'].diff() > 0
    df['FilteredOscillator'] = df['RangeMomentumOscillator'].where(df['PriceIncrease'], 0)
    
    # Final Alpha Factor
    df['FinalFactor'] = df['FilteredMomentum'] * df['FilteredOscillator'] * df['VolumeWeight']
    
    return df['FinalFactor']
```
