```python
import pandas as pd

def heuristics_v2(df):
    # 5-Day Price Momentum
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['price_momentum'] = df['close'] / df['5_day_avg_close'] - 1
    
    # 5-Day Average Volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Relative Volume
    df['relative_volume'] = df['volume'] / df['5_day_avg_volume']
    
    # Adjusted Momentum
    df['adjusted_momentum'] = df['price_momentum'] * df['relative_volume']
    
    # Intraday Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    
    # Open-to-Close Returns
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']
    
    # Combined Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']
    
    # Analyze Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    conditions = [
        (df['volume_ratio'] > 2.0),
        (df['volume_ratio'] >= 1.0) & (df['volume_ratio'] <= 2.0),
        (df['volume_ratio'] >= 0.7) & (df['volume_ratio'] < 1.0),
        (df['volume_ratio'] >= 0.5) & (df['volume_ratio'] < 0.7),
        (df['volume_ratio'] < 0.5)
    ]
    choices = [3, 2, 1, -1, -3]
    df['volume_trend_classification'] = pd.np.select(conditions, choices, default=0)
    
    # Adjust Combined Returns Based on Volume Classification
    df['adjusted_combined_returns'] = df['combined_returns'] * df['volume_trend_classification']
    
    # Short-Term Momentum
    df['short_term_momentum'] = df['close'].rolling(window=10).mean()
    
    # Long-Term Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Integrate Volume with Intraday Dynamics
    df['high_low_spread_adjusted'] = (df['high'] - df['low']) * df['relative_volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['adjusted_combined_returns'] * df['momentum_oscillator'] + 
                          df['high_low_spread_adjusted']) * df['volume_trend_classification']
    
    return df['alpha_factor']
```
