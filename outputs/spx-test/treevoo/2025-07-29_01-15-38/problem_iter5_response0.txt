```python
def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / average_volume_5_days
    volume_adjusted_momentum = short_term_momentum * relative_volume
    
    # Integrate Amount Influence
    average_amount_5_days = df['amount'].rolling(window=5).mean()
    relative_amount = df['amount'] / average_amount_5_days
    final_factor = volume_adjusted_momentum * relative_amount
    
    return final_factor
```
