```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume Fluctuations
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_volume_5
    adjusted_momentum = short_term_momentum * relative_volume
    
    # Weight by Volume Surge
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_weight = 2.0 * (df['volume'] > 2 * avg_volume_10) + 1.5 * ((df['volume'] <= 2 * avg_volume_10) & (df['volume'] > 1.5 * avg_volume_10)) + 1.0 * (df['volume'] <= 1.5 * avg_volume_10)
    weighted_momentum = adjusted_momentum * volume_weight
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    factor_value = weighted_momentum
    factor_value[daily_price_range > 2 * avg_daily_price_range_20] = 0
    
    return factor_value
```
