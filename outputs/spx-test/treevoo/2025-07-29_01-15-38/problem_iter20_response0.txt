```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Compute 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).sum() / 5
    
    # Compute 10-Day Price Momentum
    ten_day_momentum = df['close'].rolling(window=10).sum() / 10
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    adjusted_ten_day_momentum = ten_day_momentum * relative_volume
    
    # Analyze Volume Change
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    def classify_volume_trend(ratio):
        if ratio > 1.5:
            return 2
        elif ratio < 0.5:
            return -2
        else:
            return 0
    volume_trend_classification = volume_change_ratio.apply(classify_volume_trend)
    
    # Combine Factors Based on Volume Classification
    combined_factor = (
        intraday_return + 
        weighted_high_low_spread + 
        adjusted_five_day_momentum + 
        adjusted_ten_day_momentum
    ) * volume_trend_classification
    
    # Apply Cumulative Movement
    cumulative_factor = combined_factor.rolling(window=10).sum()
    
    return cumulative_factor
```
