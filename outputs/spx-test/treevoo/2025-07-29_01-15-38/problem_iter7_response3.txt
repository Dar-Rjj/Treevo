```python
def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum
    df['range_diff_1d'] = df['high_low_range'] - df['high_low_range'].shift(1)
    df['short_term_momentum'] = df['range_diff_1d'].rolling(window=5).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    df['long_term_momentum'] = df['range_diff_1d'].rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Weight by Volume Surge
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / df['avg_volume_10d']
    df['volume_weight'] = 1
    df.loc[volume_ratio > 1.5, 'volume_weight'] = 1.5
    df.loc[(volume_ratio > 1) & (volume_ratio <= 1.5), 'volume_weight'] = 1.2
    
    # Filter by Close Price Increase
    df['factor_value'] = df['momentum_oscillator'] * df['volume_weight']
    df.loc[df['close'] <= df['close'].shift(1), 'factor_value'] = 0
    
    return df['factor_value']
```
