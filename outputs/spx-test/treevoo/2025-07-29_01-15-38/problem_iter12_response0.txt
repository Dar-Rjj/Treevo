```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    price_momentum = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_volume_5d
    adjusted_momentum = price_momentum * relative_volume
    
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = volume_ratio.apply(lambda x: 3 if x > 2.0 else (1 if 0.7 <= x <= 2.0 else -3))
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_returns * volume_trend
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].rolling(window=8).mean()
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=26).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Integrate Volume with Intraday Dynamics
    high_low_spread = (df['high'] - df['low']) * relative_volume
    
    # Weight by Volume Surge
    avg_volume_15d = df['volume'].rolling(window=15).mean()
    previous_avg_volume_15d = avg_volume_15d.shift(1)
    volume_surge_weight = df['volume'].apply(lambda x, pa: 2.0 if x > 2.0 * pa else (1.5 if x > pa else 1.0), args=(previous_avg_volume_15d,))
    
    # Final Alpha Factor
    final_factor = (adjusted_combined_returns * momentum_oscillator) + (high_low_spread * volume_surge_weight)
    
    return final_factor
```
