```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume Fluctuations
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    relative_vol = df['volume'] / avg_vol_5
    adjusted_short_term_momentum = short_term_momentum * relative_vol
    
    # Weight by Volume Surge
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_weight = 1.0
    volume_weight[df['volume'] > 1.5 * avg_vol_10] = 1.5
    volume_weight[df['volume'] > avg_vol_10] = 1.2
    weighted_momentum = adjusted_short_term_momentum * volume_weight
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_price_range_20 = daily_price_range.rolling(window=20).mean()
    factor_value = weighted_momentum
    factor_value[daily_price_range > 1.5 * avg_price_range_20] = 0
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=21).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = adjusted_short_term_momentum - (df['close'] - long_term_momentum)
    
    # Introduce High-Low Range Indicator
    daily_high_low_range = df['high'] - df['low']
    avg_high_low_range_5 = daily_high_low_range.rolling(window=5).mean()
    normalized_high_low_range = daily_high_low_range / avg_high_low_range_5
    
    # Final Alpha Factor
    final_factor = factor_value * momentum_oscillator * normalized_high_low_range
    
    return final_factor
```
