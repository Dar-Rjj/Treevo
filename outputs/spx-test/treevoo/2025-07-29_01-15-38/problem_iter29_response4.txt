```python
import pandas as pd

def heuristics_v2(df, window_size=10, n_days=5):
    # Calculate price change and intraday return
    df['price_change'] = df['close'].diff()
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['price_range'] = df['high'] - df['low']

    # Calculate volume change and adjustment
    df['volume_change'] = df['volume'].diff()
    df['mean_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_adjustment'] = df['volume'] / df['mean_volume'].shift(1)

    # Combine price and volume factors
    df['price_volume_factor'] = df['price_change'] * df['volume_change']
    df['pricerange_volume_ratio'] = df['price_range'] / df['volume']

    # Intraday momentum
    df['intraday_momentum'] = (df['high'] - df['low']) * df['volume_adjustment']

    # Multi-day momentum
    df['n_day_return'] = (df['close'] / df['close'].shift(n_days)) - 1
    df['five_day_momentum'] = df['close'].rolling(window=5).sum() / 5
    df['relative_volume'] = df['volume'] / (df['volume'].rolling(window=5).sum() / 5)
    df['adjusted_five_day_momentum'] = df['five_day_momentum'] * df['relative_volume']

    # Short-term reversal and sentiment
    df['short_term_reversal'] = (df['close'].shift(1) / df['close'].shift(2)) - 1
    df['positive_closes'] = df['price_change'].gt(0).rolling(window=n_days).sum()
    df['negative_closes'] = df['price_change'].lt(0).rolling(window=n_days).sum()
    df['sentiment_score'] = df['positive_closes'] - df['negative_closes']

    # Technical indicators
    df['sma'] = df['close'].rolling(window=window_size).mean()
    gain = df['price_change'].apply(lambda x: max(x, 0)).rolling(window=window_size).mean()
    loss = df['price_change'].apply(lambda x: abs(min(x, 0))).rolling(window=window_size).mean()
    df['rsi'] = 100 - (100 / (1 + (gain / loss)))

    # Cumulative movement
    df['cumulative_movement'] = df['alpha_factor'].rolling(window=window_size).sum()

    # Classify volume trend
    df['volume_trend_classification'] = df['volume'].pct_change().apply(
        lambda x: 2 if x > 1.5 else (-2 if x < 0.5 else 0)
    )

    # Final alpha factor synthesis
    df['alpha_factor'] = (
        1 * df['price_volume_factor'] +
        1 * df['pricerange_volume_ratio'] +
        1 * df['intraday_momentum'] +
        1 * df['adjusted_five_day_momentum'] +
        1 * df['n_day_return'] +
        1 * df['short_term_reversal'] +
        1 * df['sentiment_score'] +
        1 * (df['sma'] - df['close']) +
        1 * df['rsi'] +
        1 * (df['cumulative_movement'] * df['volume_trend_classification'])
    )

    return df['alpha_factor']
```
