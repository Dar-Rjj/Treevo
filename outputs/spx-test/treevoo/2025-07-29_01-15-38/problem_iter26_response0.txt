```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']
    
    # Analyze Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    conditions = [
        (df['volume_ratio'] > 2),
        (df['volume_ratio'] >= 1.6) & (df['volume_ratio'] <= 2),
        (df['volume_ratio'] >= 1.5) & (df['volume_ratio'] < 1.6),
        (df['volume_ratio'] >= 0.85) & (df['volume_ratio'] < 1.5),
        (df['volume_ratio'] >= 0.5) & (df['volume_ratio'] < 0.85),
        (df['volume_ratio'] >= 0.3) & (df['volume_ratio'] < 0.5),
        (df['volume_ratio'] < 0.3)
    ]
    choices = [3, 2, 1, 0, -1, -2, -3]
    df['volume_trend_classification'] = pd.Series(pd.np.select(conditions, choices, default=0), index=df.index)
    
    # Adjust Combined Returns Based on Volume Classification
    df['adjusted_combined_returns'] = df['combined_returns'] * df['volume_trend_classification']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['volume_adjustment'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Weighted High-Low Spread
    df['weighted_high_low_spread'] = df['intraday_momentum'] * df['volume_adjustment']
    
    # Integrate Volume with Intraday Dynamics
    df['integrated_volume_intraday'] = (df['high'] - df['low']) * df['volume_adjustment']
    
    # Compute 5-Day Price Momentum
    df['5_day_price_momentum'] = df['close'].rolling(window=5).mean() - df['close'].shift(5)
    
    # Incorporate Volume Influence
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / df['5_day_avg_volume']
    df['adjusted_5_day_momentum'] = df['5_day_price_momentum'] * df['relative_volume']
    
    # Adjust Integrated Returns Based on Volume Classification
    df['integrated_returns'] = (df['adjusted_combined_returns'] + 
                                df['weighted_high_low_spread'] + 
                                df['adjusted_5_day_momentum']) * df['volume_trend_classification']
    
    # Apply Cumulative Movement
    df['cumulative_movement'] = df['integrated_returns'].rolling(window=5).sum()
    
    return df['cumulative_movement']
```
