```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the weighted sum of volume and price difference
    weighted_sum = df['volume'] * price_diff
    
    # Generate the alpha factor
    alpha_factor = (weighted_sum / avg_price).rolling(window=5).mean()
    
    return alpha_factor
```
