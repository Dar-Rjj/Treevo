```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5DayMomentum'] = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['5DayAvgVolume'] = df['volume'].rolling(window=5).mean()
    df['RelativeVolume'] = df['volume'] / df['5DayAvgVolume']
    df['AdjustedMomentum'] = df['5DayMomentum'] * df['RelativeVolume']
    
    # Calculate Volume Change Indicator
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['PriceMomentum'] = df['SMA_10'] - df['close']
    
    # Combine Indicators
    df['CombinedIndicator'] = df['VolumeChange'] * df['PriceMomentum']
    df['FilteredIndicator'] = df['CombinedIndicator'].apply(lambda x: 1 if x > 0 else 0)
    
    # Calculate Short-Term and Long-Term Momentum
    df['ShortTermMomentum'] = df['close'].rolling(window=5).mean()
    df['LongTermMomentum'] = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Oscillator
    df['MomentumOscillator'] = df['ShortTermMomentum'] - df['LongTermMomentum']
    
    # Weight by Volume Surge
    df['VolumeRatio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['VolumeWeight'] = df['VolumeRatio'].apply(
        lambda x: 1.5 if x > 1.5 else (1.2 if x > 1 else 1)
    )
    
    # Introduce High-Low Range Indicator
    df['DailyHLRange'] = df['high'] - df['low']
    df['5DayAvgHLRange'] = df['DailyHLRange'].rolling(window=5).mean()
    df['NormalizedHLRange'] = df['DailyHLRange'] / df['5DayAvgHLRange']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['FilteredIndicator'] * 
        df['MomentumOscillator'] * 
        df['VolumeWeight'] * 
        df['NormalizedHLRange']
    )
    
    return df['FinalAlphaFactor']
```
