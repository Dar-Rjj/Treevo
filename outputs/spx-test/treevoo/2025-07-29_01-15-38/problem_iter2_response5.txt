```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Calculate Volume-Adjusted High-Low Spread
    mean_volume = df['volume'].rolling(window=N).mean()
    adjusted_volume = df['volume'] / mean_volume
    high_low_spread = df['high'] - df['low']
    volume_adjusted_spread = adjusted_volume * high_low_spread
    
    # Calculate High-Low Range Momentum
    momentum = volume_adjusted_spread.diff().rolling(window=N).sum()
    
    # Filter by Intraday Return
    intraday_return = df['close'] - df['open']
    momentum = momentum.where(intraday_return > 0, 0)
    
    return momentum
```
