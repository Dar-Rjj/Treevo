import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend_classification = volume_ratio.apply(
        lambda x: 2 if x > 1.5 else (0 if 0.8 <= x <= 1.5 else -2)
    )
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_returns * volume_trend_classification
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].shift(1)
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = high_low_spread * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    intraday_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute Hybrid Factor
    hybrid_factor = adjusted_combined_returns + weighted_high_low_spread
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum
    short_term_momentum = high_low_range.rolling(window=3).sum()
    
    # Calculate Medium-Term High-Low Range Momentum
    medium_term_momentum = high_low_range.rolling(window=7).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    long_term_momentum = high_low_range.rolling(window=14).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    momentum_oscillator = (short_term_momentum - medium_term_momentum) / long_term_momentum
    
    # Weight by Volume Surge
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_surge_ratio = df['volume'] / avg_volume_5_days
    volume_weight = volume_surge_ratio.apply(
        lambda x: 2.5 if x > 2 else (1.8 if x > 1.8 else (1.3 if x > 1 else 1))
    )
    
    # Filter by Close Price Increase
    close_price_increase = (df['close'] > df['close'].shift(1)).astype(int)
    filtered_momentum_oscillator = momentum_oscillator * close_price_increase
    
    # Final Alpha Factor
    final_alpha_factor = (filtered_momentum_oscillator * volume_weight) + hybrid_factor
    
    return final_alpha_factor
