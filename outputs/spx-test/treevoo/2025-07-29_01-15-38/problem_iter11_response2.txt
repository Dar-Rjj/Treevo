```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    momentum_5 = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = momentum_5 * (df['volume'] / avg_volume_5)
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    price_range_filter = (daily_price_range > 1.5 * avg_daily_price_range_20)
    
    # Apply Volatility Filter
    daily_returns = df['close'].pct_change()
    std_dev_daily_returns_20 = daily_returns.rolling(window=20).std()
    volatility_filter = (std_dev_daily_returns_20 > 0.03)
    
    # Combine all filters
    factor = volume_adjusted_momentum.copy()
    factor[price_range_filter] = 0
    factor[volatility_filter] = 0
    
    return factor
```
