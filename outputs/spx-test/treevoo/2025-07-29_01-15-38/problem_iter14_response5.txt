```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume Fluctuations
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    relative_vol = df['volume'] / avg_vol_5
    adjusted_momentum = short_term_momentum * relative_vol
    
    # Weight by Volume Surge
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_weight = 1.5 * (df['volume'] > 1.5 * avg_vol_10) + \
                    1.2 * (df['volume'] > avg_vol_10) + \
                    1 * (df['volume'] <= avg_vol_10)
    momentum_weighted = adjusted_momentum * volume_weight
    
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_intraday_returns = 0.6 * intraday_returns + 0.4 * open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = 2 * (volume_ratio > 1.5) + 0 * ((volume_ratio >= 0.5) & (volume_ratio <= 1.5)) - 2 * (volume_ratio < 0.5)
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_intraday_returns + momentum_weighted * volume_trend
    
    # Calculate Enhanced Volume Volatility
    avg_vol_M = df['volume'].rolling(window=30).mean()
    volatility_signal = df['volume'] - avg_vol_M
    rolling_std_vol = df['volume'].rolling(window=5).std()
    enhanced_volume_volatility = volatility_signal * rolling_std_vol
    
    # Compute 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    five_day_avg_vol = df['volume'].rolling(window=5).mean()
    relative_vol = df['volume'] / five_day_avg_vol
    adjusted_five_day_momentum = five_day_momentum * relative_vol
    
    # Integrate Volume with Intraday Dynamics
    high_low_spread = (df['high'] - df['low'])
    weighted_high_low_spread = high_low_spread * relative_vol
    
    # Compute Hybrid Factor
    hybrid_factor = (intraday_return + 
                     combined_intraday_returns + 
                     adjusted_five_day_momentum + 
                     enhanced_volume_volatility + 
                     weighted_high_low_spread)
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_range_20 = daily_price_range.rolling(window=20).mean()
    factor_value = hybrid_factor.where(daily_price_range <= 1.5 * avg_range_20, 0)
    
    # Apply Cumulative Movement
    ewma_factor = factor_value.ewm(span=10).mean()
    
    return ewma_factor
```
