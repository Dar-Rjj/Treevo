import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    mean_volume = df['volume'].mean()
    volume_adjustment = df['volume'] / mean_volume
    
    # Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    high_low_with_volume = (df['high'] - df['low']) * volume_adjustment
    
    # 5-Day Price Momentum
    five_day_price_momentum = df['close'].rolling(window=5).mean()
    
    # 5-Day Average Volume
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Relative Volume
    relative_volume = df['volume'] / five_day_avg_volume
    
    # Adjust 5-Day Momentum
    adjusted_five_day_momentum = five_day_price_momentum * relative_volume
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    def classify_volume_trend(volume_ratio):
        if volume_ratio > 1.5:
            return 2
        elif volume_ratio >= 0.5:
            return 0
        else:
            return -2
    volume_trend_classification = volume_ratio.apply(classify_volume_trend)
    
    # Combine Returns Based on Volume Classification
    combined_returns = (intraday_return + weighted_high_low_spread + adjusted_five_day_momentum) * volume_trend_classification
    
    # Apply Cumulative Movement
    cumulative_factor = combined_returns.rolling(window=5).sum()
    
    return cumulative_factor
