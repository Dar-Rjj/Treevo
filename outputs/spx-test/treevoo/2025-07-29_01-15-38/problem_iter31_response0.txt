```python
import pandas as pd

def heuristics_v2(df):
    # 5-Day Price Momentum
    df['5_Day_Price_Momentum'] = df['close'].rolling(window=5).sum() / 5
    
    # 5-Day Average Volume
    df['5_Day_Average_Volume'] = df['volume'].rolling(window=5).sum() / 5
    
    # Relative Volume
    df['Relative_Volume'] = df['volume'] / df['5_Day_Average_Volume']
    
    # Adjusted 5-Day Momentum
    df['Adjusted_5_Day_Momentum'] = df['5_Day_Price_Momentum'] * df['Relative_Volume']
    
    # Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['open']
    
    # Open-to-Close Return
    df['Open_to_Close_Return'] = (df['close'] - df['open']) / df['open']
    
    # Combined Return
    df['Combined_Return'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    
    # Classify Volume Trend
    volume_ratio = df['volume'] / df['volume'].shift(1)
    conditions = [
        (volume_ratio > 2.0),
        (volume_ratio >= 1.8) & (volume_ratio <= 2.0),
        (volume_ratio >= 1.6) & (volume_ratio < 1.8),
        (volume_ratio >= 0.9) & (volume_ratio < 1.6),
        (volume_ratio >= 0.7) & (volume_ratio < 0.9),
        (volume_ratio >= 0.5) & (volume_ratio < 0.7),
        (volume_ratio < 0.5)
    ]
    choices = [3, 2, 1, 0, -1, -2, -3]
    df['Volume_Trend_Classification'] = pd.np.select(conditions, choices, default=0)
    
    # Adjust Combined Returns Based on Volume Classification
    df['Adjusted_Combined_Return'] = df['Combined_Return'] * df['Volume_Trend_Classification']
    
    # Momentum Oscillator
    df['Momentum_Oscillator'] = (df['close'].rolling(window=8).mean()) - (df['close'].rolling(window=26).mean())
    
    # Integrate Volume with Intraday Dynamics
    df['High_Low_Spread_Adjusted'] = (df['high'] - df['low']) * df['Relative_Volume']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Adjusted_Combined_Return'] * df['Momentum_Oscillator']) + df['High_Low_Spread_Adjusted']
    df['Alpha_Factor'] = df['Alpha_Factor'] * df['Volume_Trend_Classification']
    
    # Apply Cumulative Movement
    df['Cumulative_Alpha_Factor'] = df['Alpha_Factor'].rolling(window=5).sum()
    
    return df['Cumulative_Alpha_Factor']
```
