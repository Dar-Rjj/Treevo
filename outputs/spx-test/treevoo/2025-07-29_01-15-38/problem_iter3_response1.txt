```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Price Momentum Indicator
    n = 10
    df['ema_close'] = df['close'].ewm(span=n, adjust=False).mean()
    df['price_momentum'] = df['close'] - df['ema_close']
    
    # Combine Indicators with Enhanced Filtering
    df['combined_indicator'] = df['volume_change'] * df['price_momentum']
    
    # Apply a Dual Threshold for Filtering
    upper_threshold = 0.5
    lower_threshold = -0.5
    
    conditions = [
        (df['combined_indicator'] > upper_threshold),
        (df['combined_indicator'] < lower_threshold)
    ]
    
    choices = [1, -1]
    
    df['factor'] = pd.np.select(conditions, choices, default=0)
    
    return df['factor']
```
