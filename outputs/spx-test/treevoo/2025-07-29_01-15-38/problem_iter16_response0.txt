```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['ema_10'] - df['close']
    
    # Combine Volume and Price Momentum
    df['volume_price_momentum'] = df['volume_change'] * df['price_momentum']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum
    df['short_term_range_momentum'] = (df['high_low_range'] - df['high_low_range'].shift(1)).rolling(window=5).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    df['long_term_range_momentum'] = (df['high_low_range'] - df['high_low_range'].shift(1)).rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['range_momentum_oscillator'] = df['short_term_range_momentum'] - df['long_term_range_momentum']
    
    # Weight by Volume Surge
    df['vol_avg_10'] = df['volume'].rolling(window=10).mean().shift(1)
    df['volume_ratio'] = df['volume'] / df['vol_avg_10']
    df['volume_weight'] = 1
    df.loc[df['volume_ratio'] > 1.5, 'volume_weight'] = 1.5
    df.loc[(df['volume_ratio'] > 1) & (df['volume_ratio'] <= 1.5), 'volume_weight'] = 1.2
    
    # Filter by Close Price Increase
    df['filtered_oscillator'] = df['range_momentum_oscillator']
    df.loc[df['close'] <= df['close'].shift(1), 'filtered_oscillator'] = 0
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['volume_price_momentum'] * df['filtered_oscillator'] * df['volume_weight'])
    
    return df['alpha_factor']
```
