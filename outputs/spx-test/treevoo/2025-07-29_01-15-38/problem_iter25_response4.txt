```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Create Price-Volume Factor
    df['price_volume_factor'] = df['price_change'] * df['volume_change']
    
    # Create Price-Range to Volume Ratio
    df['pricerange_volume_ratio'] = df['price_range'] / df['volume']
    
    # Calculate 5-Day Return
    n = 5
    df['n_day_return'] = (df['close'] / df['close'].shift(n)) - 1
    
    # Identify Short-Term Reversals
    df['short_term_reversal'] = (df['close'].shift(1) / df['close'].shift(2)) - 1
    
    # Calculate Number of Positive and Negative Closes
    positive_closes = (df['price_change'] > 0).rolling(window=20).sum()
    negative_closes = (df['price_change'] < 0).rolling(window=20).sum()
    
    # Create Sentiment Score
    df['sentiment_score'] = positive_closes - negative_closes
    
    # Simple Moving Average (SMA)
    window_size = 20
    df['sma'] = df['close'].rolling(window=window_size).mean()
    
    # Relative Strength Index (RSI)
    gains = df['price_change'].clip(lower=0).rolling(window=14).mean()
    losses = df['price_change'].clip(upper=0).abs().rolling(window=14).mean()
    df['rsi'] = 100 - (100 / (1 + (gains / losses)))
    
    # Final Alpha Factor Synthesis
    w1, w2, w3, w4, w5, w6, w7 = 1, 1, 1, 1, 1, 1, 1
    df['alpha_factor'] = (w1 * df['price_volume_factor'] + 
                          w2 * df['pricerange_volume_ratio'] + 
                          w3 * df['n_day_return'] + 
                          w4 * df['short_term_reversal'] + 
                          w5 * df['sentiment_score'] + 
                          w6 * (df['sma'] - df['close']) + 
                          w7 * df['rsi'])
    
    return df['alpha_factor']
```
