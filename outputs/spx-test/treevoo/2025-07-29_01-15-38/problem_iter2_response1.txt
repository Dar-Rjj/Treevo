```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    n_days = 10
    df['sma_close'] = df['close'].rolling(window=n_days).mean()
    df['price_momentum'] = df['sma_close'] - df['close']
    
    # Combine Indicators
    df['combined_indicator'] = df['volume_change'] * df['price_momentum']
    threshold = 0  # Example threshold
    df['filtered_indicator'] = df.apply(lambda row: 1 if row['combined_indicator'] > threshold else 0, axis=1)
    
    # Calculate Short-Term Momentum
    short_term_days = 5
    df['short_term_momentum'] = df['close'].rolling(window=short_term_days).mean()
    
    # Calculate Long-Term Momentum
    long_term_days = 20
    df['long_term_momentum'] = df['close'].rolling(window=long_term_days).mean()
    
    # Calculate Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Weight by Volume Surge
    df['prev_volume_avg'] = df['volume'].shift(1).rolling(window=10).mean()
    df['volume_weight'] = df.apply(lambda row: 1.5 if row['volume'] > 1.5 * row['prev_volume_avg'] 
                                    else (1.2 if row['volume'] > row['prev_volume_avg'] else 1), axis=1)
    
    # Final Alpha Factor
    df['final_factor'] = df['filtered_indicator'] * df['momentum_oscillator'] * df['volume_weight']
    
    return df['final_factor']
```
