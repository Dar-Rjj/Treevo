```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-Day Price Momentum
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['price_momentum'] = (df['close'] - df['5_day_avg_close'])

    # Calculate the Average Volume over the last 5 days
    df['avg_volume_5_day'] = df['volume'].rolling(window=5).mean()

    # Adjust the price momentum by relative volume
    df['adjusted_price_momentum'] = df['price_momentum'] * (df['volume'] / df['avg_volume_5_day'])

    return df['adjusted_price_momentum']
```
