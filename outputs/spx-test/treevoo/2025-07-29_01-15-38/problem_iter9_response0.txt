def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Short-Term and Long-Term High-Low Range Momentum
    short_term_momentum = (high_low_range - high_low_range.shift(1)).rolling(window=7).sum()
    long_term_momentum = (high_low_range - high_low_range.shift(1)).rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Calculate Volume Weight
    volume_ratio = df['volume'].rolling(window=10).mean()
    volume_weight = (df['volume'] > 1.5 * volume_ratio).astype(int) * 1.5 + \
                    ((df['volume'] > volume_ratio) & (df['volume'] <= 1.5 * volume_ratio)).astype(int) * 1.2 + \
                    (df['volume'] <= volume_ratio).astype(int)
    
    # Filter by Close Price Increase
    close_increase_filter = (df['close'] > df['close'].shift(1)).astype(int)
    filtered_momentum_oscillator = momentum_oscillator * close_increase_filter
    
    # Incorporate Close Price Trend
    close_price_trend = df['close'].rolling(window=7).mean()
    trend_factor = (df['close'] > close_price_trend).astype(int) * 1.2 + \
                   (df['close'] <= close_price_trend).astype(int) * 0.8
    
    # Final Alpha Factor
    final_factor = filtered_momentum_oscillator * volume_weight * trend_factor
    
    return final_factor
