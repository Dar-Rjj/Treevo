```python
import pandas as pd

def heuristics_v2(df, N=20, M=20):
    # Calculate Price Momentum
    avg_close = df['close'].rolling(window=N).mean()
    recent_close = df['close']
    momentum_signal = recent_close - avg_close

    # Calculate Volume Volatility
    avg_volume = df['volume'].rolling(window=M).mean()
    recent_volume = df['volume']
    volatility_signal = recent_volume - avg_volume

    # Combine Momentum and Volatility Signals
    combined_signal = momentum_signal * volatility_signal

    return combined_signal
```
