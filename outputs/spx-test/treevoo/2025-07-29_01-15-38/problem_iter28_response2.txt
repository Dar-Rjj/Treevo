```python
import pandas as pd

def heuristics_v2(df):
    # Compute the ratio of close price to volume on each day
    df['close_to_volume_ratio'] = df['close'] / df['volume']
    
    # Calculate the rolling average of close price to volume ratio over a window
    df['rolling_close_to_volume_ratio'] = df['close_to_volume_ratio'].rolling(window=20).mean()
    
    # Calculate the daily price range (High - Low)
    df['daily_price_range'] = df['high'] - df['low']
    
    # Compute the rolling average of the daily price range
    df['rolling_daily_price_range'] = df['daily_price_range'].rolling(window=20).mean()
    
    # Calculate the change in close price from one day to the next
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the rolling average of the daily return
    df['rolling_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate the difference between close and open prices
    df['daily_gap'] = df['close'] - df['open']
    
    # Compute the rolling average of the daily gap (Close - Open)
    df['rolling_daily_gap'] = df['daily_gap'].rolling(window=20).mean()
    
    # Calculate the daily volume change
    df['daily_volume_change'] = df['volume'].pct_change()
    
    # Compute the rolling average of the daily volume change
    df['rolling_daily_volume_change'] = df['daily_volume_change'].rolling(window=20).mean()
    
    # Calculate the standard deviation of daily returns over a window
    df['rolling_std_daily_return'] = df['daily_return'].rolling(window=20).std()
    
    # Combine all the features into a single alpha factor
    df['alpha_factor'] = (
        df['rolling_close_to_volume_ratio'] +
        df['rolling_daily_price_range'] +
        df['rolling_daily_return'] +
        df['rolling_daily_gap'] +
        df['rolling_daily_volume_change'] +
        df['rolling_std_daily_return']
    )
    
    return df['alpha_factor']
```
