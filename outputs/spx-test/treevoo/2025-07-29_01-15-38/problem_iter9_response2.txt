```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = pd.Series(index=df.index, data=0.0)
    volume_trend[volume_ratio > 1.5] = 2
    volume_trend[(volume_ratio >= 0.5) & (volume_ratio <= 1.5)] = 0
    volume_trend[volume_ratio < 0.5] = -2
    
    # Assess Transaction Amount Impact
    amount_ratio = df['amount'] / df['amount'].shift(1)
    
    # Categorize Amount Trend
    amount_trend = pd.Series(index=df.index, data=0.0)
    amount_trend[amount_ratio > 1.5] = 2
    amount_trend[(amount_ratio >= 0.5) & (amount_ratio <= 1.5)] = 0
    amount_trend[amount_ratio < 0.5] = -2
    
    # Adjust Combined Returns Based on Volume and Amount Classification
    adjusted_returns = combined_returns * (volume_trend + amount_trend)
    
    # Apply Weighted Moving Average
    weights = [5, 4, 3, 2, 1]
    weighted_factor = adjusted_returns.rolling(window=len(weights)).apply(lambda x: (x * weights).sum() / sum(weights), raw=True)
    
    return weighted_factor
```
