```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    N = 10
    df['sma_close'] = df['close'].rolling(window=N).mean()
    df['price_momentum'] = df['sma_close'] - df['close']
    
    # Combine Indicators
    df['combined_indicator'] = df['volume_change'] * df['price_momentum']
    
    # Apply a Threshold for Filtering
    threshold = 0
    df['factor'] = (df['combined_indicator'] > threshold).astype(int)
    
    return df['factor']
```
