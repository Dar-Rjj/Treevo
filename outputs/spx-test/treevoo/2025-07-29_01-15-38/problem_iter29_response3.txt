import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Compute 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = volume_ratio.apply(lambda x: 2 if x > 1.5 else -2 if x < 0.5 else 0)
    
    # Combine Factors Based on Volume Classification
    combined_factors = (
        intraday_return +
        weighted_high_low_spread +
        adjusted_five_day_momentum
    ) * volume_trend
    
    # Apply Cumulative Movement
    cumulative_movement = combined_factors.rolling(window=5).sum()
    
    # Integration of Additional Factors
    daily_high_low_range = df['high'] - df['low']
    five_day_avg_high_low_range = daily_high_low_range.rolling(window=5).sum() / 5
    daily_vol_to_avg_vol_ratio = df['volume'] / five_day_avg_volume
    
    # Combine Integrated Factors
    integrated_factors = (
        intraday_return +
        weighted_high_low_spread +
        five_day_momentum +
        daily_high_low_range -
        five_day_avg_high_low_range +
        daily_vol_to_avg_vol_ratio
    ) * volume_trend
    
    # Apply Cumulative Movement
    integrated_cumulative_movement = integrated_factors.rolling(window=5).sum()
    
    return integrated_cumulative_movement
