```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_price_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / average_volume_5_days
    adjusted_short_term_momentum = short_term_price_momentum * relative_volume
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    mean_volume = df['volume'].mean()
    volume_adjustment = df['volume'] / mean_volume
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Combine Adjusted Factors
    combined_factor = adjusted_short_term_momentum + weighted_high_low_spread
    
    return combined_factor
```
