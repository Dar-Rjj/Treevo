import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the relative range (High - Low) / Close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the average volume over a 5-day window
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Compute the factor: (close % change * relative range) / average volume
    df['factor'] = (df['close_pct_change'] * df['relative_range']) / df['avg_volume_5d']
    
    # Return the factor as a pandas Series
    return df['factor']
