```python
import pandas as pd

def heuristics_v2(data, n=5):
    # Calculate Intraday Returns
    intraday_returns = (data['high'] - data['low']) / data['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (data['close'] - data['open']) / data['open']
    
    # Combine Normalized Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = data['volume'] / data['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = pd.Series(0, index=data.index)
    volume_trend[volume_ratio > 1.5] = 2
    volume_trend[(volume_ratio > 0.5) & (volume_ratio <= 1.5)] = 0
    volume_trend[volume_ratio <= 0.5] = -2
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_returns = combined_returns * volume_trend
    
    # Apply Weighted Moving Average
    weights = list(range(n, 0, -1))
    wma = adjusted_returns.rolling(window=n).apply(lambda x: (x * weights[:len(x)]).sum() / sum(weights[:len(x)]), raw=False)
    
    return wma
```
