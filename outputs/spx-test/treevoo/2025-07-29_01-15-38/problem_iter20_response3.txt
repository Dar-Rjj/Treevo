```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with (date) as index and columns [open, high, low, close, volume].
        
    Returns:
        pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to the average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the difference between close and open prices
    close_open_diff = df['close'] - df['open']
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Combine the factors using a weighted sum
    factor = 0.5 * close_to_avg_ratio + 0.3 * close_open_diff + 0.2 * volume_change
    
    return factor
```
