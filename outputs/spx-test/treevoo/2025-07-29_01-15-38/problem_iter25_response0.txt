```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Daily Trend
    daily_trend = (df['close'] - df['open']) / df['open']
    ema_10_daily_trend = daily_trend.ewm(span=10, adjust=False).mean()
    adjusted_daily_trend = daily_trend / ema_10_daily_trend

    # Incorporate Volatility
    volatility = df['high'] - df['low']
    adjusted_volatility = adjusted_daily_trend / volatility

    # Integrate Volume-Adjusted Momentum
    volume_adjusted_momentum = (df['volume'] * adjusted_volatility).rolling(window=5).sum()

    # Long-Term Trend Indicator
    sma_20_close = df['close'].rolling(window=20).mean()
    sma_50_close = df['close'].rolling(window=50).mean()
    long_term_trend = (sma_20_close - sma_50_close) * volume_adjusted_momentum

    # Conditional Adjustment for Trend
    sensitivity = 1.0  # Positive or negative sensitivity
    trend_condition = (sma_20_close > sma_50_close)
    long_term_trend_adjusted = long_term_trend.where(trend_condition, long_term_trend * (-sensitivity))

    # Complementary Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open'] + daily_trend

    # Volume Classification
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_classification = pd.cut(volume_ratio, bins=[-float('inf'), 0.3, 0.8, 1.5, float('inf')], labels=[-2, 0, 1, 2])
    
    # Final Factor Adjustment
    final_factor = (long_term_trend_adjusted + intraday_returns) * volume_classification
    final_factor_cumulative = final_factor.rolling(window=20).sum()  # Cumulative Movement Over a Window

    return final_factor_cumulative
```
