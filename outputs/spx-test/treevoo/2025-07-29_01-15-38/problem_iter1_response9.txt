```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Integrate Volume with Intraday Dynamics
    daily_volume_sum = df['volume'].sum()
    normalized_volume = df['volume'] / daily_volume_sum
    adjusted_volume_range = normalized_volume * intraday_range
    
    # Compute Hybrid Factor
    min_adjusted_volume_range = adjusted_volume_range.min()
    hybrid_factor = intraday_return + (adjusted_volume_range - min_adjusted_volume_range)
    
    # Apply Cumulative Movement
    window = 5  # Example window size
    cumulative_hybrid_factor = hybrid_factor.rolling(window=window).sum()
    
    return cumulative_hybrid_factor
```
