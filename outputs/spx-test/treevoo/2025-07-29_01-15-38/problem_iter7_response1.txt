```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = short_term_momentum * (df['volume'] / avg_volume_5)
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    
    # Set the factor to 0 if the condition is met
    condition = (daily_price_range > 1.5 * avg_daily_price_range_20)
    volume_adjusted_momentum[condition] = 0
    
    return volume_adjusted_momentum
```
