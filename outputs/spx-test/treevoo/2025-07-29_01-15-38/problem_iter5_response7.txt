```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()

    # Adjust for Volume Fluctuations
    avg_5_day_vol = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_5_day_vol
    adjusted_short_term_momentum = short_term_momentum * relative_volume

    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()

    # Calculate Price Momentum Oscillator
    price_momentum_oscillator = short_term_momentum - long_term_momentum

    # Weight by Volume Surge
    vol_10_day_avg = df['volume'].rolling(window=10).mean()
    prev_vol_10_day_avg = vol_10_day_avg.shift(1)

    volume_surge_weight = 1
    condition_1 = (df['volume'] > 1.5 * prev_vol_10_day_avg)
    condition_2 = (df['volume'] > prev_vol_10_day_avg) & (df['volume'] <= 1.5 * prev_vol_10_day_avg)

    volume_surge_weight[condition_1] = 1.5
    volume_surge_weight[condition_2] = 1.2

    weighted_price_momentum_oscillator = price_momentum_oscillator * volume_surge_weight

    # Combine Indicators
    combined_indicator = weighted_price_momentum_oscillator * adjusted_short_term_momentum

    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_20_day_price_range = daily_price_range.rolling(window=20).mean()
    
    # Set factor value to 0 if price range is too high
    factor_value = combined_indicator
    factor_value[daily_price_range > 1.5 * avg_20_day_price_range] = 0

    return factor_value
```
