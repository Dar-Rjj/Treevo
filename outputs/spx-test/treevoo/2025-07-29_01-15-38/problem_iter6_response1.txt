```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Short-Term (5 days) High-Low Range Momentum
    df['short_term_momentum'] = (df['high_low_range'] - df['high_low_range'].shift(1)).rolling(window=5).sum()

    # Calculate Long-Term (20 days) High-Low Range Momentum
    df['long_term_momentum'] = (df['high_low_range'] - df['high_low_range'].shift(1)).rolling(window=20).sum()

    # Calculate High-Low Range Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']

    # Weight by Volume Surge
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean().shift(1)
    df['volume_weight'] = 1
    df.loc[df['volume_ratio'] > 1.5, 'volume_weight'] = 1.5
    df.loc[(df['volume_ratio'] <= 1.5) & (df['volume_ratio'] > 1), 'volume_weight'] = 1.2

    # Integrate Close Price Velocity
    df['close_change'] = df['close'] - df['close'].shift(1)
    df['velocity_factor'] = 1
    df.loc[df['close_change'] > 0, 'velocity_factor'] = 1.2
    df.loc[df['close_change'] <= 0, 'velocity_factor'] = 0.8

    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_oscillator'] * df['volume_weight'] * df['velocity_factor']

    return df['alpha_factor']
```
