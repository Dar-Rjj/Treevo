```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    mean_volume = df['volume'].expanding().mean()
    volume_adjustment = df['volume'] / mean_volume
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_volume_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    
    # Analyze Volume Change
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend_classification = pd.Series(index=df.index, dtype='float64')
    volume_trend_classification[volume_change_ratio > 1.5] = 2
    volume_trend_classification[(volume_change_ratio >= 0.5) & (volume_change_ratio <= 1.5)] = 0
    volume_trend_classification[volume_change_ratio < 0.5] = -2
    
    # Adjust Combined Returns Based on Volume Classification
    combined_returns = (intraday_return + 
                        weighted_high_low_spread + 
                        adjusted_five_day_momentum) * volume_trend_classification
    
    # Apply Cumulative Movement
    factor_values = combined_returns.cumsum()
    
    return factor_values
```
