```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate 5-day moving average of daily returns
    df['ma_5_daily_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Calculate 20-day standard deviation of daily returns
    df['std_20_daily_return'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate 20-day moving average of daily returns
    df['ma_20_daily_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate difference between 5-day and 20-day moving averages of daily returns
    df['ma_diff_5_20'] = df['ma_5_daily_return'] - df['ma_20_daily_return']
    
    # Calculate volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate volume ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate 20-day price-volume correlation
    df['price_volume_corr'] = df['close'].rolling(window=20).corr(df['volume'])
    
    # Calculate gap size
    df['gap_size'] = df['open'] - df['close'].shift(1)
    
    # Calculate intraday range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate close location value
    df['close_location_value'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate true range
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), 
        axis=1
    )
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['ma_diff_5_20'] + 
        df['volume_ratio'] * df['price_volume_corr'] + 
        df['gap_size'] + 
        df['intraday_range'] * df['close_location_value'] + 
        df['true_range']
    )
    
    return df['alpha_factor']
```
