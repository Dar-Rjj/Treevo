```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5_day_price_momentum'] = df['close'].rolling(window=5).sum() / 5
    
    # Compute 5-Day Average Volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).sum() / 5
    
    # Determine Relative Volume
    df['relative_volume'] = df['volume'] / df['5_day_avg_volume']
    
    # Adjust Momentum
    df['adjusted_momentum'] = df['5_day_price_momentum'] * df['relative_volume']
    
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['price_momentum'] = df['sma_10'] - df['close']
    
    # Combine Indicators
    df['combined_indicator'] = (df['volume_change'] * df['price_momentum'])
    df['combined_indicator_filtered'] = df['combined_indicator'].apply(lambda x: 1 if x > 0 else 0)
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Weight by Volume Surge
    df['10_day_vol_avg'] = df['volume'].rolling(window=10).mean()
    df['prev_10_day_vol_avg'] = df['10_day_vol_avg'].shift(1)
    
    def volume_weight(row):
        if row['volume'] > 1.5 * row['prev_10_day_vol_avg']:
            return 1.5
        elif row['volume'] > row['prev_10_day_vol_avg']:
            return 1.2
        else:
            return 1.0
    
    df['volume_weight'] = df.apply(volume_weight, axis=1)
    
    # Final Alpha Factor
    df['final_factor'] = df['combined_indicator_filtered'] * df['momentum_oscillator'] * df['volume_weight']
    
    return df['final_factor']
```
