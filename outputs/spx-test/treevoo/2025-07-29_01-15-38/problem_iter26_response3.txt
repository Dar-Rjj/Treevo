```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series with the generated alpha factor values indexed by (date).
    """
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the ratio of average price to the price range
    price_ratio = avg_price / price_range
    
    # Calculate the normalized volume change
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Combine the price ratio and the volume change to create the heuristic
    heuristic = (price_ratio * volume_change).rolling(window=5).mean()
    
    return heuristic
```
