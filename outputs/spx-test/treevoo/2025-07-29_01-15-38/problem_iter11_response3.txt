```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Adjust for Volume Fluctuations
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_vol_5
    adjusted_short_term_momentum = short_term_momentum * relative_volume

    # Weight by Volume Surge
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_surge_weight = 1 + (df['volume'] > 1.5 * avg_vol_10) * 0.5 + (df['volume'] > avg_vol_10) * 0.2
    weighted_adjusted_momentum = adjusted_short_term_momentum * volume_surge_weight

    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_price_range_20 = daily_price_range.rolling(window=20).mean()
    price_range_filter = (daily_price_range <= 1.5 * avg_price_range_20)
    combined_indicator = weighted_adjusted_momentum * price_range_filter

    # Calculate Short-Term and Long-Term Momentum
    short_term_avg = df['close'].rolling(window=6).mean()
    long_term_avg = df['close'].rolling(window=21).mean()

    # Calculate Momentum Oscillator
    momentum_oscillator = short_term_avg - long_term_avg

    # Introduce High-Low Range Indicator
    daily_high_low_range = df['high'] - df['low']
    avg_high_low_range_5 = daily_high_low_range.rolling(window=5).mean()
    normalized_high_low_range = daily_high_low_range / avg_high_low_range_5

    # Final Alpha Factor
    final_factor = (combined_indicator * momentum_oscillator * normalized_high_low_range).fillna(0)

    return final_factor
```
