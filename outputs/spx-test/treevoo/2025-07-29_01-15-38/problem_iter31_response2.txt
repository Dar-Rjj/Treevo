```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily price change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate the average close price over a short period (e.g., 5 days)
    n_days = 5
    df['avg_close'] = df['close'].rolling(window=n_days).mean()
    
    # Determine the difference in volume from previous day
    df['volume_change'] = df['volume'].diff()
    
    # Find a rolling sum of volume over a set window (e.g., 5 days)
    df['rolling_volume_sum'] = df['volume'].rolling(window=n_days).sum()
    
    # Compute daily price range
    df['price_range'] = df['high'] - df['low']
    
    # Evaluate the gap between open and close prices
    df['open_close_gap'] = (df['open'] - df['close']).abs()
    
    # Measure how often the close price is above or below the daily midpoint
    df['close_above_midpoint'] = (df['close'] > (df['high'] + df['low']) / 2).astype(int)
    
    # Develop a ratio of price change to volume change
    df['price_vol_ratio'] = df['daily_price_change'] / df['volume_change']
    df['price_vol_ratio'].fillna(0, inplace=True)  # handle division by zero
    
    # Create an indicator that combines daily volatility and trading volume
    avg_volume = df['volume'].rolling(window=n_days).mean()
    df['volatility_volume_indicator'] = df['price_range'] / avg_volume
    df['volatility_volume_indicator'].fillna(0, inplace=True)  # handle division by zero
    
    # Combine all the features into a single alpha factor
    alpha_factor = (
        df['daily_price_change'] +
        df['avg_close'] +
        df['volume_change'] +
        df['rolling_volume_sum'] +
        df['price_range'] +
        df['open_close_gap'] +
        df['close_above_midpoint'] +
        df['price_vol_ratio'] +
        df['volatility_volume_indicator']
    )
    
    return alpha_factor
```
