```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['price_momentum'] = df['sma_10'] - df['close']
    
    # Combine Indicators
    df['combined_value'] = df['volume_change'] * df['price_momentum']
    df['factor'] = (df['combined_value'] > 0).astype(int)
    
    return df['factor']
```
