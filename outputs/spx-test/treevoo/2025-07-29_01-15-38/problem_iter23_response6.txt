```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Weighted High-Low Spread
    df['weighted_high_low_spread'] = (df['high'] - df['low']) * df['volume']
    
    # Compute 5-Day Price Momentum
    df['5day_momentum'] = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['5day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / df['5day_avg_volume']
    df['adjusted_5day_momentum'] = df['5day_momentum'] * df['relative_volume']
    
    # Analyze Volume Change
    df['volume_change_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    conditions = [
        (df['volume_change_ratio'] > 1.5),
        (df['volume_change_ratio'] < 0.5),
        (df['volume_change_ratio'] >= 0.5) & (df['volume_change_ratio'] <= 1.5)
    ]
    choices = [2, -2, 0]
    df['volume_trend_classification'] = pd.np.select(conditions, choices, default=0)
    
    # Combine Factors Based on Volume Classification
    df['combined_factors'] = (df['intraday_return'] + 
                              df['weighted_high_low_spread'] + 
                              df['adjusted_5day_momentum']) * df['volume_trend_classification']
    
    # Apply Cumulative Movement
    df['cumulative_factor'] = df['combined_factors'].rolling(window=5).sum()
    
    # Incorporate Technical Indicators
    df['sma'] = df['close'].rolling(window=5).mean()
    df['ema'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Relative Strength Index (RSI)
    df['price_change'] = df['close'].diff(1)
    df['gains'] = df['price_change'].apply(lambda x: x if x > 0 else 0).rolling(window=5).mean()
    df['losses'] = df['price_change'].apply(lambda x: -x if x < 0 else 0).rolling(window=5).mean()
    df['rsi'] = 100 - (100 / (1 + (df['gains'] / df['losses'])))
    
    # Final Alpha Factor Synthesis
    alpha_factor = (
        1 * df['cumulative_factor'] + 
        0 * df['pricerange_volume_ratio'] + 
        0 * df['n_day_return'] + 
        0 * df['short_term_reversal'] + 
        0 * df['sentiment_score'] + 
        0 * (df['sma'] - df['close']) + 
        0 * (df['ema'] - df['close']) + 
        0 * df['rsi']
    )
    
    return alpha_factor
```
