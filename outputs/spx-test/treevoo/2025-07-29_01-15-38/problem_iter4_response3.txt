```python
def heuristics_v2(df):
    # Compute SMA of High Prices for the last 14 days
    sma_high = df['high'].rolling(window=14).mean()
    
    # Compute SMA of Low Prices for the last 14 days
    sma_low = df['low'].rolling(window=14).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = (sma_high - sma_low) / (sma_high - sma_low).abs()
    
    # Calculate 5-day Rolling Standard Deviation of Volume
    vol_std = df['volume'].rolling(window=5).std()
    
    # Weight the Momentum Oscillator
    wimo = momentum_oscillator * vol_std
    
    return wimo
```
