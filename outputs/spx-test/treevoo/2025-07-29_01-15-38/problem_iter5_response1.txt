import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].mean()
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    
    # Compute Hybrid Factor
    hybrid_factor = intraday_return + weighted_high_low_spread + adjusted_five_day_momentum
    hybrid_factor = hybrid_factor.rolling(window=5).sum()
    
    return hybrid_factor
