```python
import pandas as pd

def heuristics(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_volume_intraday = (df['high'] - df['low']) * volume_adjustment
    
    # Compute 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend_classification = volume_ratio.apply(
        lambda x: 2 if x > 1.5 else (-2 if x < 0.7 else 0)
    )
    
    # Combine Returns Based on Volume Classification
    combined_returns = (intraday_return + weighted_high_low_spread + adjusted_five_day_momentum) * volume_trend_classification
    
    # Apply Cumulative Movement
    cumulative_movement = combined_returns.rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum
    short_term_high_low_momentum = (df['high'] - df['low']).rolling(window=5).sum()
    medium_term_high_low_momentum = (df['high'] - df['low']).rolling(window=10).sum()
    long_term_high_low_momentum = (df['high'] - df['low']).rolling(window=20).sum()
    high_low_range_momentum_oscillator = (short_term_high_low_momentum - medium_term_high_low_momentum) / long_term_high_low_momentum
    
    # Weight by Volume Surge
    avg_10_day_volume = df['volume'].rolling(window=10).mean()
    volume_ratio_10 = df['volume'] / avg_10_day_volume
    volume_surge_weight = volume_ratio_10.apply(
        lambda x: 2 if x > 2 else (1.5 if x > 1.5 and x <= 2 else (1.2 if x > 1 and x <= 1.5 else 1))
    )
    
    # Filter by Close Price Increase
    close_price_increase_filter = (df['close'] > df['close'].shift(1)).astype(int)
    filtered_momentum_oscillator = high_low_range_momentum_oscillator * volume_surge_weight * close_price_increase_filter
    
    # Final Alpha Factor
    final_alpha_factor = filtered_momentum_oscillator + cumulative_movement
    
    return final_alpha_factor
```
