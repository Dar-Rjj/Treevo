```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Price Volatility
    df['price_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Calculate Volume Volatility
    df['volume_volatility'] = df['volume_change'] / df['volume'].shift(1)
    
    # Create Price-Volume Factor
    df['price_volume_factor'] = df['price_change'] * df['volume_change']
    
    # Create Price-Range to Volume Ratio
    df['pricerange_volume_ratio'] = df['price_range'] / df['volume']
    
    # Create Price-Volume Volatility Factor
    df['pricevol_volumevol_factor'] = df['price_volatility'] * df['volume_volatility']
    
    # Calculate N-Day Return (e.g., 5-day return)
    df['n_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Identify Short-Term Reversals
    df['short_term_reversal'] = (df['close'].shift(1) / df['close'].shift(2)) - 1
    
    # Calculate Number of Positive and Negative Closes
    df['positive_closes'] = df['price_change'].gt(0).rolling(window=20).sum()
    df['negative_closes'] = df['price_change'].lt(0).rolling(window=20).sum()
    
    # Create Sentiment Score
    df['sentiment_score'] = df['positive_closes'] - df['negative_closes']
    
    # Simple Moving Average (SMA) with window size 20
    df['sma'] = df['close'].rolling(window=20).mean()
    
    # Relative Strength Index (RSI) with window size 14
    df['gains'] = df['price_change'].apply(lambda x: max(x, 0)).rolling(window=14).mean()
    df['losses'] = df['price_change'].apply(lambda x: abs(min(x, 0))).rolling(window=14).mean()
    df['rsi'] = 100 - (100 / (1 + (df['gains'] / df['losses'])))
    
    # Final Alpha Factor Synthesis
    df['alpha_factor'] = (
        0.15 * df['price_volume_factor'] +
        0.10 * df['pricerange_volume_ratio'] +
        0.10 * df['pricevol_volumevol_factor'] +
        0.20 * df['n_day_return'] +
        0.05 * df['short_term_reversal'] +
        0.10 * df['sentiment_score'] +
        0.15 * (df['sma'] - df['close']) +
        0.15 * df['rsi']
    )
    
    return df['alpha_factor']
```
