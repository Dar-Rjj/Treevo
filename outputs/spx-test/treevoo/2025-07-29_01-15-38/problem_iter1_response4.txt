```python
def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].mean()
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = (df['high'] - df['low']) * volume_adjustment
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Combine the factors to create the final alpha factor
    alpha_factor = (intraday_momentum + weighted_high_low_spread) * intraday_return
    
    return alpha_factor
```
