```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5_day_momentum'] = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['5_day_avg_vol'] = df['volume'].rolling(window=5).mean()
    df['relative_vol'] = df['volume'] / df['5_day_avg_vol']
    df['vol_adjusted_momentum'] = df['5_day_momentum'] * df['relative_vol']

    # Calculate Intraday Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']

    # Combine Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']

    # Analyze Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)

    # Classify Volume Trend
    conditions = [
        (df['volume_change'] > 2.0),
        (df['volume_change'] >= 1.6) & (df['volume_change'] <= 2.0),
        (df['volume_change'] >= 1.5) & (df['volume_change'] < 1.6),
        (df['volume_change'] >= 0.85) & (df['volume_change'] < 1.5),
        (df['volume_change'] >= 0.7) & (df['volume_change'] < 0.85),
        (df['volume_change'] >= 0.5) & (df['volume_change'] < 0.7),
        (df['volume_change'] < 0.5)
    ]
    choices = [3, 2, 1, 0, -1, -2, -3]
    df['volume_trend_classification'] = pd.Series(pd.np.select(conditions, choices, default=0), index=df.index)

    # Adjust Combined Returns Based on Volume Classification
    df['adjusted_combined_returns'] = df['combined_returns'] * df['volume_trend_classification']

    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].rolling(window=8).mean()

    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].rolling(window=26).mean()

    # Calculate Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']

    # Integrate Volume with Intraday Dynamics
    df['high_low_spread_vol_adj'] = (df['high'] - df['low']) * df['relative_vol']

    # Final Alpha Factor
    df['alpha_factor'] = (df['adjusted_combined_returns'] * df['momentum_oscillator'] 
                          + df['high_low_spread_vol_adj'] * df['volume_trend_classification'])

    # Apply Cumulative Movement
    df['cumulative_alpha_factor'] = df['alpha_factor'].rolling(window=5).sum()

    return df['cumulative_alpha_factor']
```
