```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    df['5_day_momentum'] = df['5_day_avg_close'].pct_change(periods=5)
    
    # Incorporate Volume Influence
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / df['5_day_avg_volume']
    df['adjusted_5_day_momentum'] = df['5_day_momentum'] * df['relative_volume']
    
    # Combine Intraday and Open-to-Close Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']
    
    # Classify Volume Trend
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend_classification'] = df['volume_ratio'].apply(
        lambda x: 3 if x > 2.0 else
        2 if 1.8 <= x <= 2.0 else
        1 if 1.6 <= x < 1.8 else
        0 if 0.9 <= x < 1.6 else
       -1 if 0.7 <= x < 0.9 else
       -2 if 0.5 <= x < 0.7 else
       -3
    )
    df['adjusted_combined_returns'] = df['combined_returns'] * df['volume_trend_classification']
    
    # Integrate Volume with Intraday Dynamics
    df['intraday_dynamic'] = (df['high'] - df['low']) * df['relative_volume']
    
    # Calculate Momentum Oscillator
    df['short_term_momentum'] = df['close'].rolling(window=8).mean()
    df['long_term_momentum'] = df['close'].rolling(window=26).mean()
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Compute Enhanced Price Momentum
    df['N_day_avg_close'] = df['close'].rolling(window=14).mean()
    df['momentum_signal'] = df['close'] - df['N_day_avg_close']
    df['SMA_high'] = df['high'].rolling(window=14).mean()
    df['SMA_low'] = df['low'].rolling(window=14).mean()
    df['intraday_momentum_oscillator'] = (df['SMA_high'] - df['SMA_low']) / (df['SMA_high'] - df['SMA_low']).abs()
    df['enhanced_price_momentum'] = df['momentum_signal'] + df['intraday_momentum_oscillator']
    
    # Calculate Enhanced Volume Volatility
    df['M_day_avg_volume'] = df['volume'].rolling(window=14).mean()
    df['volatility_signal'] = df['volume'] - df['M_day_avg_volume']
    df['5_day_vol_std'] = df['volume'].rolling(window=5).std()
    df['enhanced_volume_volatility'] = df['volatility_signal'] * df['5_day_vol_std']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_combined_returns'] * df['momentum_oscillator'] +
                                df['intraday_dynamic'] +
                                df['enhanced_volume_volatility'] * df['volume_trend_classification'])
    
    # Apply Cumulative Movement
    df['cumulative_factor'] = df['final_alpha_factor'].rolling(window=5).sum()
    
    return df['cumulative_factor']
```
