```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Momentum
    df['ShortTermMomentum'] = df['close'].rolling(window=5).mean()
    
    # Relative Volume
    df['AvgVolume5'] = df['volume'].rolling(window=5).mean()
    df['RelativeVolume'] = df['volume'] / df['AvgVolume5']
    
    # Adjusted Momentum
    df['AdjustedMomentum'] = df['ShortTermMomentum'] * df['RelativeVolume']
    
    # Daily Price Range
    df['DailyPriceRange'] = df['high'] - df['low']
    
    # Average Daily Price Range over 20 days
    df['AvgRange20'] = df['DailyPriceRange'].rolling(window=20).mean()
    
    # Apply Price Range Filter
    df['FactorValue'] = df.apply(lambda row: 0 if row['DailyPriceRange'] > 1.5 * row['AvgRange20'] else row['AdjustedMomentum'], axis=1)
    
    return df['FactorValue']
```
