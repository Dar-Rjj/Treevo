```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum from Close Prices
    df['Momentum'] = df['close'] - df['close'].shift(10)
    
    # Volume-Adjust the Momentum
    mean_vol_10_to_t = df['volume'].rolling(window=10).mean()
    df['Volume_Adjusted_Momentum'] = df['Momentum'] * (df['volume'] / mean_vol_10_to_t)
    
    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Calculate Average Daily Price Range over past 20 days
    df['AvgRange_20'] = df['Daily_Price_Range'].rolling(window=20).mean()
    
    # Apply Price Volatility Filter
    df['Factor_Value'] = 0
    condition = df['Daily_Price_Range'] <= 1.5 * df['AvgRange_20']
    df.loc[condition, 'Factor_Value'] = df.loc[condition, 'Volume_Adjusted_Momentum']
    
    return df['Factor_Value']
```
