def heuristics_v2(data):
    # Extract necessary columns
    open_prices = data['open']
    high_prices = data['high']
    low_prices = data['low']
    close_prices = data['close']
    volume = data['volume']

    # Calculate the range of the day
    daily_range = high_prices - low_prices

    # Calculate the upper and lower bounds of the true range
    upper_bound = high_prices - close_prices.shift(1)
    lower_bound = close_prices.shift(1) - low_prices
    true_range = (upper_bound > lower_bound).multiply(upper_bound, fill_value=0) + \
                 (lower_bound >= upper_bound).multiply(lower_bound, fill_value=0)

    # Calculate the buying pressure
    buying_pressure = (close_prices - low_prices) / (high_prices - low_prices) * volume

    # Calculate the factor value
    factor_value = (buying_pressure.rolling(window=5).mean() / daily_range).fillna(0)

    return factor_value
