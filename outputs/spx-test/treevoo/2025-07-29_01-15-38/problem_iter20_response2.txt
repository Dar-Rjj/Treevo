```python
import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Daily trend strength
    daily_trend_strength = (data['close'] - data['open']) / data['open']
    
    # 10-day EMA of (Close - Open)
    smoothed_trend = (data['close'] - data['open']).ewm(span=10).mean()
    
    # Ratio of daily to smoothed trend
    trend_ratio = (data['close'] - data['open']) / smoothed_trend
    
    # Adjusting for volatility
    volatility_adjusted_trend = trend_ratio / (data['high'] - data['low'])
    
    # Incorporating volume-adjusted momentum over 5 days
    volume_adjusted_momentum = (volatility_adjusted_trend * data['volume']).rolling(window=5).sum()
    
    # Long-term trend indicator
    sma_20 = data['close'].rolling(window=20).mean()
    sma_50 = data['close'].rolling(window=50).mean()
    long_term_trend = sma_20 - sma_50
    
    # Conditional adjustment for uptrend or downtrend
    sensitivity = np.where(sma_20 > sma_50, 1.2, 0.8)
    primary_factor = volume_adjusted_momentum * long_term_trend * sensitivity
    
    # Complementary factor: Intraday returns
    intraday_returns = (data['high'] - data['low']) / data['open']
    open_to_close_returns = (data['close'] - data['open']) / data['open']
    combined_intraday_returns = intraday_returns + open_to_close_returns
    
    # Volume classification
    volume_ratio = data['volume'] / data['volume'].shift(1)
    volume_trend_classification = np.select(
        [volume_ratio > 1.5, (volume_ratio >= 0.8) & (volume_ratio <= 1.5),
         (volume_ratio >= 0.3) & (volume_ratio < 0.8), volume_ratio < 0.3],
        [2, 1, 0, -2],
        default=np.nan
    )
    
    # Adjusted combined returns based on volume trend classification
    adjusted_combined_returns = combined_intraday_returns * volume_trend_classification
    
    # Final factor adjustment
    final_factor = primary_factor + adjusted_combined_returns
    cumulative_final_factor = final_factor.rolling(window=20).sum()
    
    return cumulative_final_factor
```
