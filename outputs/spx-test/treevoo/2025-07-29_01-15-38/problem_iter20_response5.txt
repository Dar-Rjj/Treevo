import pandas as pd

def heuristics_v2(df, N=14, M=5):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'].div(df['volume'].shift(1))
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_volume_intraday_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute Enhanced Price Momentum
    average_close_N_days = df['close'].rolling(window=N).mean()
    recent_close = df['close']
    momentum_signal = recent_close - average_close_N_days
    
    sma_high_prices = df['high'].rolling(window=14).mean()
    sma_low_prices = df['low'].rolling(window=14).mean()
    intraday_momentum_oscillator = (sma_high_prices - sma_low_prices) / (sma_high_prices - sma_low_prices).abs()
    enhanced_price_momentum = momentum_signal + intraday_momentum_oscillator
    
    # Analyze Volume Change
    volume_change = volume_adjustment
    
    # Classify Volume Trend
    volume_trend_classification = volume_change.apply(lambda x: 2 if x > 2 else (0 if x >= 0.7 else -2))
    
    # Adjust Combined Returns Based on Volume Classification
    combined_returns = intraday_return + weighted_high_low_spread + 5_day_price_momentum
    adjusted_combined_returns = combined_returns * volume_trend_classification
    
    # Calculate Enhanced Volume Volatility
    average_volume_M_days = df['volume'].rolling(window=M).mean()
    volatility_signal = df['volume'] - average_volume_M_days
    rolling_std_volume = df['volume'].rolling(window=5).std()
    enhanced_volume_volatility = volatility_signal * rolling_std_volume
    
    # Compute 5-Day Price Momentum
    five_day_price_momentum = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    five_day_average_volume = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / five_day_average_volume
    adjusted_5_day_momentum = five_day_price_momentum * relative_volume
    
    # Compute Hybrid Factor
    hybrid_factor = (
        intraday_return

        + adjusted_5_day_momentum
