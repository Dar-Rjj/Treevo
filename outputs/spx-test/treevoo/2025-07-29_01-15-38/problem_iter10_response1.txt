```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    adjusted_momentum = short_term_momentum * (df['volume'] / avg_volume_5)
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    
    # Set factor to 0 if DailyPriceRange_t > 1.5 * AvgDailyPriceRange_20
    filter_condition = daily_price_range > 1.5 * avg_daily_price_range_20
    adjusted_momentum[filter_condition] = 0
    
    return adjusted_momentum
```
