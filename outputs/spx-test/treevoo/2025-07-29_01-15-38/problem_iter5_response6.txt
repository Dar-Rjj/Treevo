```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    df['intraday_returns'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Returns
    df['open_to_close_returns'] = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / df['avg_volume_5_days']
    df['volume_adjusted_momentum'] = df['short_term_momentum'] * df['relative_volume']
    
    # Weight Combined Returns by Volume Trend
    df['volume_trend'] = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    df['weighted_returns'] = df['combined_returns'] * df['volume_trend']
    
    # Integrate Volume-Adjusted Short-Term Momentum and Weighted Returns
    df['integrated_momentum'] = df['volume_adjusted_momentum'] + df['weighted_returns']
    
    # Apply Price Range Filter
    df['daily_price_range'] = df['high'] - df['low']
    df['avg_price_range_20_days'] = df['daily_price_range'].rolling(window=20).mean()
    
    # Exclude stocks with excessive volatility
    df['alpha_factor'] = df.apply(lambda row: 0 if row['daily_price_range'] > 1.5 * row['avg_price_range_20_days'] else row['integrated_momentum'], axis=1)
    
    return df['alpha_factor']
```
