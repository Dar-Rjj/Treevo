```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    mean_volume_10days = df['volume'].rolling(window=10).mean()
    volume_adjustment = df['volume'] / mean_volume_10days
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_intraday = (df['high'] - df['low']) * volume_adjustment
    
    # Compute 5-Day Price Momentum
    five_day_price_momentum = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_price_momentum * relative_volume
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = pd.Series(index=df.index, dtype=float)
    volume_trend[volume_ratio > 1.5] = 2
    volume_trend[(volume_ratio >= 0.5) & (volume_ratio <= 1.5)] = 0
    volume_trend[volume_ratio < 0.5] = -2
    
    # Adjust Combined Returns Based on Volume Classification
    combined_returns = (intraday_return + weighted_high_low_spread + 
                        adjusted_five_day_momentum) * volume_trend
    
    # Apply Cumulative Movement
    cumulative_movement = combined_returns.rolling(window=10).sum()
    
    # Incorporate Amount Influence
    average_amount_5days = df['amount'].rolling(window=5).mean()
    relative_amount = df['amount'] / average_amount_5days
    final_factor = cumulative_movement * relative_amount
    
    return final_factor
```
