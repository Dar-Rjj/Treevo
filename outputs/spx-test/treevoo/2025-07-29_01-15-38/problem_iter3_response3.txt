```python
import pandas as pd

def heuristics_v2(df, N=10, M=10):
    # Calculate Price Momentum
    df['avg_close'] = df['close'].rolling(window=N).mean()
    df['recent_close'] = df['close']
    df['price_momentum'] = df['recent_close'] - df['avg_close']
    
    # Calculate Volume Change
    df['avg_volume'] = df['volume'].rolling(window=M).mean()
    df['recent_volume'] = df['volume']
    df['volume_change'] = df['recent_volume'] / df['avg_volume']
    
    # Combine Momentum and Volume Change
    df['combined_alpha_factor'] = df['price_momentum'] * df['volume_change']
    
    return df['combined_alpha_factor']
```
