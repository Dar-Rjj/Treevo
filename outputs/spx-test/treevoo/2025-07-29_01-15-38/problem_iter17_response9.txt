```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5DayMomentum'] = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    df['5DayAvgVolume'] = df['volume'].rolling(window=5).mean()
    df['VolumeAdjustedMomentum'] = df['5DayMomentum'] * (df['volume'] / df['5DayAvgVolume'])
    
    # Weight by Volume Surge
    df['10DayAvgVolume'] = df['volume'].rolling(window=10).mean()
    volume_weight = [1.5 if vol > 1.5 * avg_vol else 1.2 if vol > avg_vol else 1 for vol, avg_vol in zip(df['volume'], df['10DayAvgVolume'])]
    df['VolumeWeightedMomentum'] = df['VolumeAdjustedMomentum'] * volume_weight
    
    # Apply Price Range Filter
    df['DailyPriceRange'] = df['high'] - df['low']
    df['AvgDailyPriceRange_20'] = df['DailyPriceRange'].rolling(window=20).mean()
    price_range_filter = df['DailyPriceRange'] > 1.5 * df['AvgDailyPriceRange_20']
    df.loc[price_range_filter, 'VolumeWeightedMomentum'] = 0
    
    # Apply Volatility Filter
    df['DailyReturn'] = df['close'].pct_change()
    df['StdDev_DailyReturns_20'] = df['DailyReturn'].rolling(window=20).std()
    volatility_filter = df['StdDev_DailyReturns_20'] > 0.03
    df.loc[volatility_filter, 'VolumeWeightedMomentum'] = 0
    
    # Calculate Intraday and Open-to-Close Returns
    df['IntradayReturn'] = (df['high'] - df['low']) / df['open']
    df['OpenToCloseReturn'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = 0.6 * df['IntradayReturn'] + 0.4 * df['OpenToCloseReturn']
    
    # Analyze Volume Change
    df['VolumeRatio'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend_classification = [3 if ratio > 2 else 1 if 0.5 <= ratio <= 2 else -3 for ratio in df['VolumeRatio']]
    df['VolumeTrend'] = volume_trend_classification
    
    # Adjust Combined Returns Based on Volume Classification
    df['AdjustedCombinedReturns'] = combined_returns * df['VolumeTrend']
    
    # Integrate Volume with Intraday Dynamics
    df['HighLowDiff'] = df['high'] - df['low']
    df['VolumeAdjustedHighLow'] = df['HighLowDiff'] * df['VolumeTrend']
    
    # Calculate Long-Term Momentum
    df['20DayMomentum'] = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Oscillator
    df['MomentumOscillator'] = df['5DayMomentum'] - df['20DayMomentum']
    
    # Introduce High-Low Range Indicator
    df['DailyHighLowRange'] = df['high'] - df['low']
    df['5DayAvgHighLowRange'] = df['DailyHighLowRange'].rolling(window=5).mean()
    df['NormalizedHighLowRange'] = df['DailyHighLowRange'] / df['5DayAvgHighLowRange']
    
    # Combine Indicators and Momentum Oscillator
    df['IntegratedIndicator'] = df['VolumeWeightedMomentum'] * df['MomentumOscillator'] * df['NormalizedHighLowRange'] * df['VolumeTrend']
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['IntegratedIndicator']
    df.loc[df['VolumeWeightedMomentum'] == 0, 'AlphaFactor'] = 0
    
    return df['AlphaFactor']
```
