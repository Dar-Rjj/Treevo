import pandas as pd

def heuristics_v2(df):
    # Short-Term Adjusted Momentum
    short_term_price_change = df['close'].pct_change(periods=5)
    avg_5_day_vol = df['volume'].rolling(window=5).mean()
    vol_adjustment = df['volume'] / avg_5_day_vol
    short_term_adj_momentum = short_term_price_change * vol_adjustment
    
    # Long-Term Price Momentum
    long_term_price_change = df['close'].pct_change(periods=20)
    
    # Price Momentum Oscillator
    price_momentum_oscillator = short_term_price_change - long_term_price_change
    
    # 10-Day Volume Surge
    current_10_day_vol_avg = df['volume'].rolling(window=10).mean()
    previous_10_day_vol_avg = df['volume'].shift(10).rolling(window=10).mean()
    volume_surge_weight = (current_10_day_vol_avg > 1.5 * previous_10_day_vol_avg) * 1.5 + \
                          (current_10_day_vol_avg > previous_10_day_vol_avg) * 0.2 + \
                          (current_10_day_vol_avg <= previous_10_day_vol_avg) * 1.0
    
    # Daily Price Range
    daily_price_range = df['high'] - df['low']
    avg_20_day_price_range = daily_price_range.rolling(window=20).mean()
    
    # Apply Price Range Filter and Volume Surge Weighting
    factor_value = price_momentum_oscillator * volume_surge_weight
    factor_value[daily_price_range > 1.5 * avg_20_day_price_range] = 0
    
    return factor_value
