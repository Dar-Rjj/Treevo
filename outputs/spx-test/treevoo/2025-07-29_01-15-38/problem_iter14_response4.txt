```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day average money flow
    n = 14
    average_money_flow = money_flow.rolling(window=n).mean()
    
    # Calculate the money flow ratio
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_average_money_flow = positive_money_flow.rolling(window=n).sum()
    negative_average_money_flow = negative_money_flow.rolling(window=n).sum().abs()
    money_flow_ratio = positive_average_money_flow / negative_average_money_flow
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_flow_ratio))
    
    # Calculate the 5-day moving average of the close price
    ma_close = df['close'].rolling(window=5).mean()
    
    # Calculate the alpha factor
    alpha_factor = (money_flow_index - 50) * (df['close'] - ma_close)
    
    return alpha_factor
```
