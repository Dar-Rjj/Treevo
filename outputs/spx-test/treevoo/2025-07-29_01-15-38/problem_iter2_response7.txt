```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    momentum = df['close'] - df['close'].shift(10)
    
    # Volume-Adjust the Momentum
    volume_adjusted_momentum = momentum * (df['volume'] / df['volume'].rolling(window=10).mean())
    
    # Apply Price Volatility Filter
    daily_price_range = df['high'] - df['low']
    avg_price_range_20 = daily_price_range.rolling(window=20).mean()
    price_volatility_filter = daily_price_range > 1.5 * avg_price_range_20
    volume_adjusted_momentum[price_volatility_filter] = 0
    
    # Calculate Volume Volatility
    M = 20  # Define the rolling window size for volume volatility
    average_volume_M = df['volume'].rolling(window=M).mean()
    recent_volume = df['volume']
    volume_volatility = recent_volume - average_volume_M
    
    # Generate Volatility Signal and final factor
    volatility_signal = (volume_volatility > 0).astype(int)  # Assuming positive volume volatility is a signal
    final_factor = volume_adjusted_momentum * volatility_signal
    
    return final_factor
```
