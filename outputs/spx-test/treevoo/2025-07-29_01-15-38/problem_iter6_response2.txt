```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum (7 days)
    df['short_term_momentum'] = df['high_low_range'].diff().rolling(window=7).sum()
    
    # Calculate Long-Term High-Low Range Momentum (20 days)
    df['long_term_momentum'] = df['high_low_range'].diff().rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Volume Ratio and compare with previous 10-day average
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean().shift(1)
    
    # Assign volume weight
    conditions = [
        (df['volume_ratio'] > 1.5),
        (df['volume_ratio'] > 1) & (df['volume_ratio'] <= 1.5),
        (df['volume_ratio'] <= 1)
    ]
    choices = [1.5, 1.2, 1]
    df['volume_weight'] = pd.np.select(conditions, choices, default=1)
    
    # Filter by Close Price Increase
    df['factor_value'] = df['momentum_oscillator'] * df['volume_weight'] * (df['close'] > df['close'].shift(1)).astype(int)
    
    # Output the final factor value
    return df['factor_value']
```
