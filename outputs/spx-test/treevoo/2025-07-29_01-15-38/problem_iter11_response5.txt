import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    hl_range = df['high'] - df['low']
    
    # Short-Term High-Low Range Momentum
    short_term_mom = (hl_range - hl_range.shift(1)).rolling(window=5).sum()
    
    # Long-Term High-Low Range Momentum
    long_term_mom = (hl_range - hl_range.shift(1)).rolling(window=20).sum()
    
    # High-Low Range Momentum Oscillator
    mom_oscillator = short_term_mom - long_term_mom
    
    # Weight by Volume Surge
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    volume_weight = [1.5 if v > 1.5 * av else 1.2 if v > av else 1 for v, av in zip(df['volume'], avg_vol_10d)]
    
    # Integrate Close Price Velocity
    close_price_change = df['close'] - df['close'].shift(1)
    velocity_factor = [1.2 if c > 0 else 0.8 for c in close_price_change]
    
    # Incorporate Open-Price Momentum
    open_price_diff = df['open'] - df['open'].shift(1)
    momentum_adjuster = [0.5 if o > 0 else -0.5 for o in open_price_diff]
    
    # Final Alpha Factor
    final_factor = (mom_oscillator * volume_weight) * velocity_factor + momentum_adjuster
    
    return pd.Series(final_factor, index=df.index, name='alpha_factor')
