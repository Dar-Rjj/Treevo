```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']

    # Adjust for Volume
    mean_volume = df['volume'].expanding().mean()
    volume_adjustment = df['volume'] / mean_volume

    # Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment

    # 5-Day Price Momentum
    five_day_price_momentum = df['close'].rolling(window=5).mean()

    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_5_day_momentum = five_day_price_momentum * relative_volume

    # Analyze Volume Change
    prev_volume = df['volume'].shift(1)
    volume_ratio = df['volume'] / prev_volume

    # Classify Volume Trend
    volume_trend_classification = volume_ratio.apply(
        lambda x: 2 if x > 1.5 else (-2 if x < 0.5 else 0)
    )

    # Combine Returns Based on Volume Classification
    combined_returns = (intraday_return + weighted_high_low_spread + adjusted_5_day_momentum) * volume_trend_classification

    # Apply Cumulative Movement
    cumulative_movement = combined_returns.rolling(window=5).sum()

    return cumulative_movement
```
