```python
import pandas as pd

def heuristics_v2(df, N=30, M=14):
    # Calculate Enhanced Price Momentum
    df['avg_close_N'] = df['close'].rolling(window=N).mean()
    df['momentum_signal'] = df['close'] - df['avg_close_N']
    
    df['sma_high_14'] = df['high'].rolling(window=14).mean()
    df['sma_low_14'] = df['low'].rolling(window=14).mean()
    df['intraday_momentum_oscillator'] = (df['sma_high_14'] - df['sma_low_14']) / (df['sma_high_14'] - df['sma_low_14']).abs()
    df['enhanced_momentum'] = df['momentum_signal'] + df['intraday_momentum_oscillator']

    # Calculate Enhanced Volume Volatility
    df['avg_volume_M'] = df['volume'].rolling(window=M).mean()
    df['volatility_signal'] = df['volume'] - df['avg_volume_M']
    df['volume_std_5'] = df['volume'].rolling(window=5).std()
    df['enhanced_volatility'] = df['volatility_signal'] * df['volume_std_5']

    # Integrate Enhanced Momentum and Volatility Signals
    df['final_alpha_factor'] = df['enhanced_momentum'] * df['enhanced_volatility']

    return df['final_alpha_factor']
```
