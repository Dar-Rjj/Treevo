```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Combined Intraday Returns
    combined_returns = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Calculate Volume-Adjusted Short-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    relative_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    adjusted_momentum = short_term_momentum * relative_volume
    
    # Weight Combined Returns by Volume Trend
    volume_trend = df['volume'].diff().apply(lambda x: 1 if x > 0 else -1)
    weighted_returns = combined_returns * volume_trend
    
    # Integrate Adjusted Momentum and Weighted Returns
    integrated_momentum = adjusted_momentum + weighted_returns
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_range_20 = daily_price_range.rolling(window=20).mean()
    filter_condition = (daily_price_range > 1.5 * avg_range_20)
    factor_values = integrated_momentum.where(~filter_condition, 0)
    
    return factor_values
```
