```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    relative_vol = df['volume'] / avg_vol_5
    volume_adjusted_momentum = short_term_momentum * relative_vol
    
    # Weight by Volume Surge
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_surge_weight = (df['volume'] > 1.5 * avg_vol_10) * 1.5 + \
                          ((df['volume'] > avg_vol_10) & (df['volume'] <= 1.5 * avg_vol_10)) * 1.2 + \
                          (df['volume'] <= avg_vol_10) * 1.0
    volume_weighted_momentum = volume_adjusted_momentum * volume_surge_weight
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    price_range_filtered = volume_weighted_momentum.copy()
    price_range_filtered[daily_price_range > 1.5 * avg_daily_price_range_20] = 0
    
    # Apply Volatility Filter
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    volatility_filtered = price_range_filtered.copy()
    volatility_filtered[std_dev_20 > 0.03] = 0
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Introduce High-Low Range Indicator
    daily_high_low_range = df['high'] - df['low']
    avg_high_low_range_5 = daily_high_low_range.rolling(window=5).mean()
    normalized_high_low_range = daily_high_low_range / avg_high_low_range_5
    
    # Final Alpha Factor
    final_factor = (volatility_filtered * momentum_oscillator * volume_surge_weight * normalized_high_low_range)
    
    return final_factor
```
