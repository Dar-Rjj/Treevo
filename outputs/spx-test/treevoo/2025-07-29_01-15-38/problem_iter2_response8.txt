```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / average_volume_5_days
    volume_adjusted_momentum = short_term_momentum * relative_volume
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    
    # Set factor value to 0 if price range on day t is greater than 1.5 * AvgRange_20, otherwise use the Volume-Adjusted Short-Term Momentum
    factor_value = volume_adjusted_momentum
    factor_value[daily_price_range > 1.5 * avg_daily_price_range_20] = 0
    
    return factor_value
```
