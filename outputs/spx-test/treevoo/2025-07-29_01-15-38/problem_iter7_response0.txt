```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = ema_10 - df['close']
    
    # Combine and Filter Indicators
    df['combined_value'] = df['volume_change'] * df['price_momentum']
    
    upper_threshold = df['combined_value'].quantile(0.8)
    lower_threshold = df['combined_value'].quantile(0.2)
    
    df['factor'] = 0
    df.loc[df['combined_value'] > upper_threshold, 'factor'] = 1
    df.loc[df['combined_value'] < lower_threshold, 'factor'] = -1
    
    return df['factor']
```
