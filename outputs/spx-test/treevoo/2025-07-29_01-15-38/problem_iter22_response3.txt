```python
def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['AvgClose_5'] = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    df['AvgVolume_5'] = df['volume'].rolling(window=5).mean()
    df['AdjustedMomentum'] = df['AvgClose_5'] * (df['volume'] / df['AvgVolume_5'])
    
    # Apply Price Range Filter
    df['DailyPriceRange'] = df['high'] - df['low']
    df['AvgDailyPriceRange_20'] = df['DailyPriceRange'].rolling(window=20).mean()
    df.loc[df['DailyPriceRange'] > 1.5 * df['AvgDailyPriceRange_20'], 'AdjustedMomentum'] = 0
    
    # Apply Volatility Filter
    df['DailyReturn'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['StdDev_DailyReturns_20'] = df['DailyReturn'].rolling(window=20).std()
    df.loc[df['StdDev_DailyReturns_20'] > 0.03, 'AdjustedMomentum'] = 0
    
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum
    df['ShortTermHLRangeMomentum'] = df['HighLowRange'].diff().rolling(window=5).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    df['LongTermHLRangeMomentum'] = df['HighLowRange'].diff().rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['HLRangeMomentumOscillator'] = df['ShortTermHLRangeMomentum'] - df['LongTermHLRangeMomentum']
    
    # Weight by Volume Surge
    df['VolumeRatio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['VolumeWeight'] = 1.0
    df.loc[df['VolumeRatio'] > 1.5, 'VolumeWeight'] = 1.5
    df.loc[(df['VolumeRatio'] > 1) & (df['VolumeRatio'] <= 1.5), 'VolumeWeight'] = 1.2
    
    # Filter by Close Price Increase
    df['PriceIncrease'] = (df['close'] > df['close'].shift(1)).astype(int)
    df['HLRangeMomentumOscillator'] = df['HLRangeMomentumOscillator'] * df['PriceIncrease']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['AdjustedMomentum'] * df['HLRangeMomentumOscillator'] * df['VolumeWeight']
    
    return df['FinalAlphaFactor']
```
