```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = pd.Series(index=df.index, dtype='float64')
    volume_trend[volume_ratio > 1.5] = 2
    volume_trend[(volume_ratio >= 0.7) & (volume_ratio <= 1.5)] = 0
    volume_trend[volume_ratio < 0.7] = -2
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_returns * volume_trend
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].shift(1)
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = high_low_spread * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_intraday_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute Hybrid Factor
    hybrid_factor = combined_returns + weighted_high_low_spread
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum
    short_term_momentum = high_low_range.rolling(window=5).sum()
    
    # Calculate Medium-Term High-Low Range Momentum
    medium_term_momentum = high_low_range.rolling(window=10).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    long_term_momentum = high_low_range.rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    momentum_oscillator = (short_term_momentum - medium_term_momentum) / long_term_momentum
    
    # Weight by Volume Surge
    average_volume = df['volume'].rolling(window=10).mean()
    volume_surge = df['volume'] / average_volume
    volume_weight = pd.Series(index=df.index, dtype='float64')
    volume_weight[volume_surge > 2] = 2
    volume_weight[(volume_surge > 1.5) & (volume_surge <= 2)] = 1.5
    volume_weight[(volume_surge > 1) & (volume_surge <= 1.5)] = 1.2
    volume_weight[volume_surge <= 1] = 1
    
    # Filter by Close Price Increase
    close_price_increase = df['close'] > df['close'].shift(1)
    filtered_momentum_oscillator = momentum_oscillator.where(close_price_increase, 0)
    
    # Final Alpha Factor
    final_alpha_factor = (filtered_momentum_oscillator * volume_weight) + hybrid_factor
    
    return final_alpha_factor
```
