```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns for open, high, low, close, volume.
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the price range (high - low)
    price_range = df['high'] - df['low']
    
    # Calculate the ratio of average price to price range
    price_ratio = avg_price / price_range
    
    # Calculate the volume adjusted ratio
    volume_adjusted_ratio = price_ratio * df['volume']
    
    # Calculate the final factor value
    factor_value = (volume_adjusted_ratio - volume_adjusted_ratio.rolling(window=10).mean()) / volume_adjusted_ratio.rolling(window=10).std()
    
    return factor_value
```
