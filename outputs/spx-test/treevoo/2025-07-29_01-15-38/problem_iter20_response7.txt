```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Analyze Volume Change
    volume_change = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = pd.Series(index=df.index, dtype=float)
    volume_trend[volume_change > 1.5] = 2
    volume_trend[(volume_change >= 0.8) & (volume_change <= 1.5)] = 1
    volume_trend[(volume_change >= 0.3) & (volume_change < 0.8)] = 0
    volume_trend[volume_change < 0.3] = -2
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_returns * volume_trend
    
    # Calculate 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    five_day_average_volume = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / five_day_average_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].rolling(window=8).mean()
    long_term_momentum = df['close'].rolling(window=26).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Integrate Volume with Intraday Dynamics
    high_low_spread = (df['high'] - df['low']) * relative_volume
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_returns * momentum_oscillator + high_low_spread
    final_alpha_factor *= volume_trend
    
    # Calculate Enhanced Price Momentum
    N = 14
    average_close_N_days = df['close'].rolling(window=N).mean()
    recent_close = df['close']
    momentum_signal = recent_close - average_close_N_days
    
    sma_high = df['high'].rolling(window=14).mean()
    sma_low = df['low'].rolling(window=14).mean()
    intraday_momentum_oscillator = (sma_high - sma_low) / abs(sma_high - sma_low)
    enhanced_price_momentum = momentum_signal + intraday_momentum_oscillator
    
    # Compute Enhanced Volume Volatility
    M = 5
    average_volume_M_days = df['volume'].rolling(window=M).mean()
    recent_volume = df['volume']
    volatility_signal = recent_volume - average_volume_M_days
    five_day_volatility = df['volume'].rolling(window=5).std()
    enhanced_volume_volatility = volatility_signal * five_day_volatility
    
    # Compute Hybrid Factor
    hybrid_factor = adjusted_combined_returns + adjusted_five_day_momentum + enhanced_price_momentum + enhanced_volume_volatility
    
    return hybrid_factor
```
