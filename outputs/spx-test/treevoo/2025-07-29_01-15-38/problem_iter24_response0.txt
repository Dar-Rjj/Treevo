```python
def heuristics_v2(df):
    # Calculate the difference between the high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the percentage change of the close price from the previous day
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the ratio of the close price to the average of high and low
    df['close_avg_ratio'] = df['close'] / ((df['high'] + df['low']) / 2)
    
    # Calculate the factor as a combination of the above heuristics
    df['factor'] = df['high_low_diff'] * df['close_pct_change'] * df['close_avg_ratio']
    
    return df['factor']
```
