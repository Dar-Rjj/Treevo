```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum (5-day)
    short_term_momentum = df['close'].rolling(window=5).mean()

    # Calculate Long-Term Price Momentum (20-day)
    long_term_momentum = df['close'].rolling(window=20).mean()

    # Calculate Price Momentum Oscillator
    price_momentum_oscillator = short_term_momentum - long_term_momentum

    # Calculate 10-day Volume Average
    volume_avg_10_days = df['volume'].rolling(window=10).mean().shift(1)

    # Calculate Volume Ratio
    volume_ratio = df['volume'] / volume_avg_10_days

    # Assign Volume Surge Weights
    volume_weight = pd.Series(1, index=df.index)
    volume_weight[volume_ratio > 1.5] = 1.5
    volume_weight[(volume_ratio <= 1.5) & (volume_ratio > 1)] = 1.2

    # Calculate Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(1)

    # Combine Indicators
    combined_indicator = price_momentum_oscillator * volume_weight * volume_change

    # Apply a Threshold for Filtering
    threshold = 0.1
    alpha_factor = (combined_indicator > threshold).astype(int)

    return alpha_factor
```
