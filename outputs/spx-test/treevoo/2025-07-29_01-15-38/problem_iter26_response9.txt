import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5DayPriceMomentum'] = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['5DayAverageVolume'] = df['volume'].rolling(window=5).mean()
    df['RelativeVolume'] = df['volume'] / df['5DayAverageVolume']
    df['Adjusted5DayMomentum'] = df['5DayPriceMomentum'] * df['RelativeVolume']
    
    # Calculate Intraday Returns
    df['IntradayReturns'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    df['OpenToCloseReturns'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['CombinedReturns'] = df['IntradayReturns'] + df['OpenToCloseReturns']
    
    # Analyze Volume Change
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    conditions = [
        (df['VolumeChange'] > 2.0),
        (df['VolumeChange'] >= 1.8) & (df['VolumeChange'] <= 2.0),
        (df['VolumeChange'] >= 1.6) & (df['VolumeChange'] < 1.8),
        (df['VolumeChange'] >= 0.9) & (df['VolumeChange'] < 1.6),
        (df['VolumeChange'] >= 0.7) & (df['VolumeChange'] < 0.9),
        (df['VolumeChange'] >= 0.5) & (df['VolumeChange'] < 0.7),
        (df['VolumeChange'] < 0.5)
    ]
    choices = [3, 2, 1, 0, -1, -2, -3]
    df['VolumeTrend'] = pd.np.select(conditions, choices, default=0)
    
    # Adjust Combined Returns Based on Volume Classification
    df['AdjustedCombinedReturns'] = df['CombinedReturns'] * df['VolumeTrend']
    
    # Integrate Volume with Intraday Dynamics
    df['HighLowSpread_RelativeVolume'] = (df['high'] - df['low']) * df['RelativeVolume']

    df['ShortTermMomentum'] = df['close'].rolling(window=8).mean()
    
    # Calculate Long-Term Momentum
    df['LongTermMomentum'] = df['close'].rolling(window=26).mean()
    
    # Calculate Momentum Oscillator
    df['MomentumOscillator'] = df['ShortTermMomentum'] - df['LongTermMomentum']

    df['AverageClose_N'] = df['close'].rolling(window=14).mean()

    df['SMA_High'] = df['high'].rolling(window=14).mean()

    df['SMA_Low'] = df['low'].rolling(window=14).mean()

    df['SMA_HighMinusLow'] = df['SMA_High'] - df['SMA_Low']
