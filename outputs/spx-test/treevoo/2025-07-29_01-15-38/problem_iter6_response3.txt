```python
import pandas as pd

def heuristics_v2(df):
    # Compute EMA of High Prices using the last 20 days
    ema_high = df['high'].ewm(span=20).mean()
    
    # Compute EMA of Low Prices using the last 20 days
    ema_low = df['low'].ewm(span=20).mean()
    
    # Calculate Intraday Momentum Oscillator
    intraday_momentum_oscillator = (ema_high - ema_low) / (ema_high - ema_low).abs()
    
    # Calculate 10-day Rolling Standard Deviation of Volume
    volume_std = df['volume'].rolling(window=10).std()
    
    # Weight the Intraday Momentum Oscillator by the 10-day Rolling Standard Deviation of Volume
    wimvo = intraday_momentum_oscillator * volume_std
    
    return wimvo
```
