```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of high and low prices using volume as the weight
    weighted_avg = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the difference between the close price and the weighted average
    close_to_wavg_diff = df['close'] - weighted_avg
    
    # Calculate the rolling mean of the close to weighted average difference for the last 10 days
    rolling_mean = close_to_wavg_diff.rolling(window=10).mean()
    
    # Calculate the factor value as the ratio of the current close to weighted average difference to its rolling mean
    factor_values = close_to_wavg_diff / rolling_mean
    
    return factor_values
```
