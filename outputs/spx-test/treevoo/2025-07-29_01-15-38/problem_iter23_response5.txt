```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend_classification = volume_ratio.apply(lambda x: 2 if x > 1.5 else (
                                                      1 if 0.8 <= x <= 1.5 else (
                                                      0 if 0.3 <= x < 0.8 else -2)))
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_returns * volume_trend_classification
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_volume_5_days
    adjusted_short_term_momentum = short_term_momentum * relative_volume
    
    # Weight by Volume Surge
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_surge_weight = (df['volume'] > 1.5 * avg_volume_10_days).apply(lambda x: 1.5 if x else (
                                                                            1.2 if df['volume'] > avg_volume_10_days else 1))
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_price_range_20 = daily_price_range.rolling(window=20).mean()
    price_range_filter = (daily_price_range > 1.5 * avg_daily_price_range_20).apply(lambda x: 0 if x else 1)
    
    # Apply Volatility Filter
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev_daily_returns_20 = daily_returns.rolling(window=20).std()
    volatility_filter = (std_dev_daily_returns_20 > 0.03).apply(lambda x: 0 if x else 1)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Integrate Volume with Intraday Dynamics
    high_low_spread = (df['high'] - df['low']) * relative_volume
    
    # Final Alpha Factor
    alpha_factor = (adjusted_combined_returns * momentum_oscillator + high_low_spread) * volume_trend_classification
    alpha_factor = alpha_factor * price_range_filter * volatility_filter * volume_surge_weight
    
    return alpha_factor
```
