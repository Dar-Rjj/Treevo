```python
def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5_Day_Avg_Close'] = df['close'].rolling(window=5).mean()
    df['Price_Momentum'] = df['close'] / df['5_Day_Avg_Close'] - 1

    # Adjust by Relative Volume
    df['5_Day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Momentum'] = df['Price_Momentum'] * (df['volume'] / df['5_Day_Avg_Volume'])

    # Apply Price Range Filter
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['AvgRange_20'] = df['Daily_Price_Range'].rolling(window=20).mean()
    
    # Exclude Stocks with Excessive Volatility
    df['Factor'] = df.apply(lambda row: 0 if row['Daily_Price_Range'] > 1.5 * row['AvgRange_20'] else row['Volume_Adjusted_Momentum'], axis=1)
    
    return df['Factor']
```
