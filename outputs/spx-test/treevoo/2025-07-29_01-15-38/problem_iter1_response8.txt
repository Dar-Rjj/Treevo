```python
import pandas as pd

def heuristics_v2(df):
    # Calculate SMA of High Prices for the last 14 days
    sma_high = df['high'].rolling(window=14).mean()
    
    # Calculate SMA of Low Prices for the last 14 days
    sma_low = df['low'].rolling(window=14).mean()
    
    # Calculate Momentum Oscillator
    momentum_oscillator = (sma_high - sma_low) / (sma_high - sma_low).abs()
    
    # Calculate Intraday Trading Volume Fluctuation (5-day Rolling Standard Deviation of Volume)
    volume_fluctuation = df['volume'].rolling(window=5).std()
    
    # Weight the Momentum Oscillator with Volume Fluctuation
    wimo = momentum_oscillator * volume_fluctuation
    
    return wimo
```
