```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Returns
    intraday_returns = df['high'] - df['low']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Calculate Volume Trend
    volume_trend = np.where(df['volume'] > df['volume'].shift(1), 1, -1)
    
    # Weight by Volume Trend
    weighted_returns = combined_returns * volume_trend
    
    # Apply Exponential Moving Average
    alpha = 0.1  # Smoothing factor
    factor_values = weighted_returns.ewm(alpha=alpha, adjust=False).mean()
    
    return factor_values
```
