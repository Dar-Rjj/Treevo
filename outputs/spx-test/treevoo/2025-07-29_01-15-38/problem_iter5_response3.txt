```python
import pandas as pd

def heuristics_v2(df, N=30, M=60):
    # Calculate Enhanced Price Momentum
    average_close = df['close'].rolling(window=N).mean()
    recent_close = df['close']
    momentum_signal = recent_close - average_close
    
    sma_high = df['high'].rolling(window=14).mean()
    sma_low = df['low'].rolling(window=14).mean()
    intraday_momentum_oscillator = (sma_high - sma_low) / (sma_high - sma_low).abs()
    enhanced_momentum = momentum_signal + intraday_momentum_oscillator
    
    # Incorporate Volume Influence on Enhanced Momentum
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / avg_volume_5_days
    adjusted_enhanced_momentum = enhanced_momentum * relative_volume
    
    # Calculate Enhanced Volume Volatility
    average_volume = df['volume'].rolling(window=M).mean()
    recent_volume = df['volume']
    volatility_signal = recent_volume - average_volume
    volume_fluctuation = df['volume'].rolling(window=5).std()
    enhanced_volatility = volatility_signal * volume_fluctuation
    
    # Integrate Adjusted Momentum and Volatility Signals
    final_alpha_factor = adjusted_enhanced_momentum * enhanced_volatility
    return final_alpha_factor
```
