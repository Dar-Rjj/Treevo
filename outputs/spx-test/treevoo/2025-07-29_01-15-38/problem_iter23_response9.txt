```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in the average price
    avg_price_change = avg_price.pct_change()
    
    # Calculate the ratio of close to open prices
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (avg_price_change * close_open_ratio) - volume_change
    
    return alpha_factor
```
