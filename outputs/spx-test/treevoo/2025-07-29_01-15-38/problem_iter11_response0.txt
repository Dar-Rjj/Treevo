```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume Fluctuations
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    relative_vol = df['volume'] / avg_vol_5
    df['adjusted_momentum'] = df['short_term_momentum'] * relative_vol
    
    # Weight by Volume Surge
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_weight = pd.Series(1, index=df.index)
    volume_weight[df['volume'] > 1.5 * avg_vol_10] = 1.5
    volume_weight[(df['volume'] > avg_vol_10) & (df['volume'] <= 1.5 * avg_vol_10)] = 1.2
    df['weighted_momentum'] = df['adjusted_momentum'] * volume_weight
    
    # Combine Intraday and Open-to-Close Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']
    combined_returns = 0.6 * df['intraday_returns'] + 0.4 * df['open_to_close_returns']
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend = pd.Series(0, index=df.index)
    volume_trend[volume_ratio > 2] = 3
    volume_trend[(volume_ratio >= 0.5) & (volume_ratio <= 2)] = 1
    volume_trend[volume_ratio < 0.5] = -3
    df['combined_returns_adjusted'] = combined_returns * volume_trend
    
    # Integrate Volume with Intraday Dynamics
    intraday_momentum = df['high'] - df['low']
    volume_adjustment = df['volume'] / df['volume'].shift(1)
    high_low_spread_weighted = intraday_momentum * volume_adjustment
    hybrid_factor = df['combined_returns_adjusted'] + high_low_spread_weighted
    
    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_daily_range_20 = daily_price_range.rolling(window=20).mean()
    factor_value = hybrid_factor.copy()
    factor_value[daily_price_range > 1.5 * avg_daily_range_20] = 0
    
    # Apply Cumulative Movement
    ewm_factor = factor_value.ewm(span=20, adjust=False).mean()
    
    return ewm_factor
```
