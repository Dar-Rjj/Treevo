```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    mean_volume = df['volume'].mean()
    volume_adjustment = df['volume'] / mean_volume
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_factor = (df['high'] - df['low']) * volume_adjustment
    
    # Compute Hybrid Factor
    hybrid_factor = intraday_return + weighted_high_low_spread + integrated_factor
    
    # Apply Cumulative Movement
    cumulative_hybrid_factor = hybrid_factor.rolling(window=5).sum()
    
    return cumulative_hybrid_factor
```
