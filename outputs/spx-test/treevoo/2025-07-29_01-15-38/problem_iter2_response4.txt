def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['High'] - df['Low']
    
    # Calculate Short-Term and Long-Term High-Low Range Momentum
    df['Range_Momentum_Diff'] = df['High_Low_Range'].diff()
    df['Short_Term_HL_Range_Momentum'] = df['Range_Momentum_Diff'].rolling(window=5).sum()
    df['Long_Term_HL_Range_Momentum'] = df['Range_Momentum_Diff'].rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['HL_Range_Oscillator'] = df['Short_Term_HL_Range_Momentum'] - df['Long_Term_HL_Range_Momentum']

    # Weight by Volume Surge
    df['Volume_Ratio'] = df['Volume'] / df['Volume'].rolling(window=10).mean().shift(1)
    conditions = [
        (df['Volume_Ratio'] > 1.5),
        (df['Volume_Ratio'] > 1) & (df['Volume_Ratio'] <= 1.5),
        (df['Volume_Ratio'] <= 1)
    ]
    choices = [1.5, 1.2, 1]
    df['Volume_Weight'] = np.select(conditions, choices, default=1)
    
    # Filter by Close Price Increase
    df['Close_Price_Increase'] = (df['Close'] > df['Close'].shift(1)).astype(int)

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['HL_Range_Oscillator'] * df['Volume_Weight'] * df['Close_Price_Increase']

    return df['Final_Alpha_Factor']
