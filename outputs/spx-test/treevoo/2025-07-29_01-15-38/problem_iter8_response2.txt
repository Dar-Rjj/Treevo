```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    N = 10
    df['ema_close'] = df['close'].ewm(span=N, adjust=False).mean()
    df['price_momentum'] = df['ema_close'] - df['close']
    
    # Combine Indicators
    df['combined_indicators'] = df['volume_change'] * df['price_momentum']
    upper_threshold = df['combined_indicators'].quantile(0.75)
    lower_threshold = df['combined_indicators'].quantile(0.25)
    df['filtered_indicators'] = df['combined_indicators'].apply(
        lambda x: 1 if x > upper_threshold else (-1 if x < lower_threshold else 0)
    )
    
    # Calculate Short-Term Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Momentum
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum Oscillator
    df['momentum_oscillator'] = short_term_ema - long_term_ema
    
    # Weight by Volume Surge
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean().shift(1)
    df['volume_weight'] = df['volume_ratio'].apply(
        lambda x: 1.8 if x > 1.5 else (1.3 if x > 1 else 1)
    )
    
    # Final Alpha Factor
    df['final_factor'] = df['filtered_indicators'] * df['momentum_oscillator'] * df['volume_weight']
    
    return df['final_factor']
```
