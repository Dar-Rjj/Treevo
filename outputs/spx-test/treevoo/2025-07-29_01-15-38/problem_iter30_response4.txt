```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5_day_price_momentum'] = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / df['5_day_avg_volume']
    df['adjusted_5_day_momentum'] = df['5_day_price_momentum'] * df['relative_volume']

    # Calculate Intraday Returns
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']
    
    # Analyze Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    conditions = [
        (df['volume_ratio'] > 2.0),
        (df['volume_ratio'].between(1.8, 2.0)),
        (df['volume_ratio'].between(1.6, 1.8)),
        (df['volume_ratio'].between(0.9, 1.6)),
        (df['volume_ratio'].between(0.7, 0.9)),
        (df['volume_ratio'].between(0.5, 0.7)),
        (df['volume_ratio'] < 0.5)
    ]
    choices = [3, 2, 1, 0, -1, -2, -3]
    df['volume_trend_classification'] = pd.np.select(conditions, choices, default=0)
    
    # Adjust Combined Returns Based on Volume Classification
    df['adjusted_combined_returns'] = df['combined_returns'] * df['volume_trend_classification']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['close'].rolling(window=8).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['close'].rolling(window=26).mean()
    
    # Calculate Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Integrate Volume with Intraday Dynamics
    df['high_low_spread_vol_adjusted'] = (df['high'] - df['low']) * df['relative_volume']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_combined_returns'] * df['momentum_oscillator'] +
        df['high_low_spread_vol_adjusted'] * df['volume_trend_classification']
    )
    
    # Apply Cumulative Movement
    df['cumulative_factor'] = df['final_alpha_factor'].rolling(window=5).sum()

    # Enhance with Volume-Weighted Return
    df['volume_weighted_intraday_return'] = (df['high'] - df['low']) / (df['open'] * df['relative_volume'])
    df['volume_weighted_open_to_close_return'] = (df['close'] - df['open']) / (df['open'] * df['relative_volume'])
    df['combined_volume_weighted_returns'] = df['volume_weighted_intraday_return'] + df['volume_weighted_open_to_close_return']
    df['adjusted_combined_with_volume_weighted'] = df['combined_returns'] + df['combined_volume_weighted_returns']
    
    # Enhance with Volume-Adjusted Momentum
    df['volume_adjusted_short_term_momentum'] = (df['close'] * df['volume']).rolling(window=8).sum() / df['volume'].rolling(window=8).sum()
    df['volume_adjusted_long_term_momentum'] = (df['close'] * df['volume']).rolling(window=26).sum() / df['volume'].rolling(window=26).sum()
    df['volume_adjusted_momentum_oscillator'] = df['volume_adjusted_short_term_momentum'] - df['volume_adjusted_long_term_momentum']

    # Final Alpha Factor
    df['final_alpha_factor_enhanced'] = (
        df['adjusted_combined_with_volume_weighted'] * df['volume_adjusted_momentum_oscillator'] +
        df['high_low_spread_vol_adjusted'] * df['volume_trend_classification']
    )

    return df['final_alpha_factor_enhanced']
```
