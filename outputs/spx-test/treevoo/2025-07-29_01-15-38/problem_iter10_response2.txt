```python
def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Short-Term High-Low Range Momentum (5 days)
    df['short_term_momentum'] = df['high_low_range'].diff().rolling(window=5).sum()
    
    # Calculate Long-Term High-Low Range Momentum (20 days)
    df['long_term_momentum'] = df['high_low_range'].diff().rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Volume Ratio and Weight by Volume Surge
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / avg_volume_10_days
    df['volume_weight'] = 1.5 * (df['volume_ratio'] > 1.5) + 1.2 * ((df['volume_ratio'] > 1) & (df['volume_ratio'] <= 1.5)) + 1 * (df['volume_ratio'] <= 1)
    
    # Integrate Close Price Velocity
    df['close_price_change'] = df['close'].diff()
    df['velocity_factor'] = 1.2 * (df['close_price_change'] > 0) + 0.8 * (df['close_price_change'] <= 0)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum_oscillator'] * df['volume_weight'] * df['velocity_factor']
    
    return df['alpha_factor']
```
