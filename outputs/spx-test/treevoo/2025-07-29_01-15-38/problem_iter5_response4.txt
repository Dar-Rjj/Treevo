```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Combined Intraday Returns
    combined_returns = (df['high'] - df['low']) + (df['close'] - df['open'])

    # Calculate Volume Trend Indicator
    volume_trend = np.where(df['volume'] > df['volume'].shift(1), 1, -1)

    # Adjust Combined Returns by Volume Trend
    adjusted_returns = combined_returns * volume_trend

    # Compute SMA of High and Low Prices for Momentum Oscillator
    sma_high = df['high'].rolling(window=14).mean()
    sma_low = df['low'].rolling(window=14).mean()
    momentum_oscillator = (sma_high - sma_low) / np.abs(sma_high - sma_low)

    # Combine Adjusted Returns and Momentum Oscillator
    composite_factor = adjusted_returns + momentum_oscillator

    # Weight by 5-day Rolling Standard Deviation of Volume
    rolling_std_volume = df['volume'].rolling(window=5).std()
    weighted_composite = composite_factor * rolling_std_volume

    # Apply Exponential Moving Average to Final Composite Factor
    alpha = 0.3  # Smoothing factor
    ewm_factor = weighted_composite.ewm(alpha=alpha, adjust=False).mean()

    return ewm_factor
```
