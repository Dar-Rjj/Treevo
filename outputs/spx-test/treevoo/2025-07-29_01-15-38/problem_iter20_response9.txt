```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['AvgClose_5'] = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    df['AvgVolume_5'] = df['volume'].rolling(window=5).mean()
    df['RelativeVolume'] = df['volume'] / df['AvgVolume_5']
    df['AdjustedMomentum'] = df['AvgClose_5'] * df['RelativeVolume']
    
    # Weight by Volume Surge
    df['AvgVol_10'] = df['volume'].rolling(window=10).mean()
    df['VolumeSurgeWeight'] = 1.0
    df.loc[df['volume'] > 1.5 * df['AvgVol_10'], 'VolumeSurgeWeight'] = 1.5
    df.loc[(df['volume'] > df['AvgVol_10']) & (df['volume'] <= 1.5 * df['AvgVol_10']), 'VolumeSurgeWeight'] = 1.2
    
    # Apply Price Range Filter
    df['DailyPriceRange'] = df['high'] - df['low']
    df['AvgDailyPriceRange_20'] = df['DailyPriceRange'].rolling(window=20).mean()
    df.loc[df['DailyPriceRange'] > 1.5 * df['AvgDailyPriceRange_20'], 'AdjustedMomentum'] = 0
    
    # Apply Volatility Filter
    df['DailyReturn'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['StdDev_DailyReturns_20'] = df['DailyReturn'].rolling(window=20).std()
    df.loc[df['StdDev_DailyReturns_20'] > 0.03, 'AdjustedMomentum'] = 0
    
    # Calculate Long-Term Price Momentum
    df['AvgClose_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Oscillator
    df['MomentumOscillator'] = df['AvgClose_5'] - df['AvgClose_20']
    
    # Introduce High-Low Range Indicator
    df['HighLowRange'] = df['high'] - df['low']
    df['AvgHighLowRange_5'] = df['HighLowRange'].rolling(window=5).mean()
    df['NormalizedHighLowRange'] = df['HighLowRange'] / df['AvgHighLowRange_5']
    
    # Calculate Short-Term High-Low Range Momentum
    df['ShortTermHighLowMomentum'] = (df['HighLowRange'] - df['HighLowRange'].shift(1)).rolling(window=5).sum()
    
    # Calculate Long-Term High-Low Range Momentum
    df['LongTermHighLowMomentum'] = (df['HighLowRange'] - df['HighLowRange'].shift(1)).rolling(window=20).sum()
    
    # Calculate High-Low Range Momentum Oscillator
    df['HighLowRangeMomentumOscillator'] = df['ShortTermHighLowMomentum'] - df['LongTermHighLowMomentum']
    
    # Filter by Close Price Increase
    df['FilteredMomentumOscillator'] = df['MomentumOscillator'] * (df['close'] > df['close'].shift(1))
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['AdjustedMomentum'] * 
        df['VolumeSurgeWeight'] * 
        df['FilteredMomentumOscillator'] * 
        df['NormalizedHighLowRange'] * 
        df['HighLowRangeMomentumOscillator']
    )
    
    return df['FinalAlphaFactor']
```
