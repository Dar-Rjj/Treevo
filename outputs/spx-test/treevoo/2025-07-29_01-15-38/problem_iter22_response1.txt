```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Average Close Prices
    df['5_day_avg_close'] = df['close'].rolling(window=5).mean()
    
    # Relative Volume (Current Day's Volume / 5-Day Avg Volume)
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['relative_volume'] = df['volume'] / df['5_day_avg_volume']
    
    # Adjusted 5-Day Price Momentum
    df['adjusted_5_day_momentum'] = df['5_day_avg_close'] * df['relative_volume']
    
    # Intraday Returns: (High - Low) / Open
    df['intraday_returns'] = (df['high'] - df['low']) / df['open']
    
    # Open-to-Close Returns: (Close - Open) / Open
    df['open_to_close_returns'] = (df['close'] - df['open']) / df['open']
    
    # Sum Intraday and Open-to-Close Returns
    df['combined_returns'] = df['intraday_returns'] + df['open_to_close_returns']
    
    # Volume Ratio: Current Day's Volume / Previous Day's Volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Volume Trend Classification
    conditions = [
        (df['volume_ratio'] > 2.0),
        (df['volume_ratio'] >= 0.7) & (df['volume_ratio'] <= 2.0),
        (df['volume_ratio'] < 0.7)
    ]
    choices = [3, 1, -3]
    df['volume_trend_classification'] = pd.np.select(conditions, choices, default=1)
    
    # Adjust Combined Returns Based on Volume Trend
    df['adjusted_combined_returns'] = df['combined_returns'] * df['volume_trend_classification']
    
    # Short-Term Momentum: Average Close prices from Day t to Day (t-7)
    df['short_term_momentum'] = df['close'].rolling(window=8).mean()
    
    # Long-Term Momentum: Average Close prices from Day t to Day (t-25)
    df['long_term_momentum'] = df['close'].rolling(window=26).mean()
    
    # Subtract Long-Term from Short-Term Momentum
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # High-Low Spread adjusted by Relative Volume
    df['high_low_spread_adjusted'] = (df['high'] - df['low']) * df['relative_volume']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_combined_returns'] * df['momentum_oscillator']) + df['high_low_spread_adjusted']
    
    return df['final_alpha_factor']
```
