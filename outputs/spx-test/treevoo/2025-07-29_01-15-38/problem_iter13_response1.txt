```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Combined Returns
    intraday_returns = df['high'] - df['low']
    open_to_close_returns = df['close'] - df['open']
    combined_returns = (intraday_returns + open_to_close_returns)

    # Calculate Volume-Adjusted Short-Term Momentum
    average_close = df['close'].rolling(window=5).mean()
    relative_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = average_close * relative_volume

    # Weight Combined Returns by Volume Trend
    volume_trend = df['volume'].diff().apply(lambda x: 1 if x > 0 else -1)
    weighted_returns = combined_returns * volume_trend

    # Integrate Volume-Adjusted Short-Term Momentum and Weighted Returns
    integrated_momentum = (volume_adjusted_momentum + weighted_returns)

    # Apply Price Range Filter
    daily_price_range = df['high'] - df['low']
    avg_range_20 = daily_price_range.rolling(window=20).mean()
    factor_value = integrated_momentum
    factor_value[(daily_price_range > 1.5 * avg_range_20)] = 0

    return factor_value
```
