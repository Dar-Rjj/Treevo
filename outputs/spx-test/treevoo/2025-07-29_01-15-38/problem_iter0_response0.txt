def heuristics_v2(df):
    """
    Compute the Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: Series of Volume Weighted Close-to-Open Return values, indexed by date.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
