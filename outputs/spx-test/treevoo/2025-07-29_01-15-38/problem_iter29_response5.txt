```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close price to volume
    df['close_to_volume'] = df['close'] / df['volume']
    
    # Calculate the rolling average and sum of close price to volume ratio
    df['avg_close_to_volume'] = df['close_to_volume'].rolling(window=5).mean()
    df['sum_close_to_volume'] = df['close_to_volume'].rolling(window=5).sum()
    
    # Calculate the daily price range (High - Low)
    df['price_range'] = df['high'] - df['low']
    
    # Compute the rolling average and sum of the daily price range
    df['avg_price_range'] = df['price_range'].rolling(window=5).mean()
    df['sum_price_range'] = df['price_range'].rolling(window=5).sum()
    
    # Calculate the change in close price from one day to the next
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the rolling average and sum of the daily return
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()
    df['sum_daily_return'] = df['daily_return'].rolling(window=5).sum()
    
    # Calculate the difference between close and open prices
    df['gap'] = df['close'] - df['open']
    
    # Compute the rolling average and sum of the daily gap
    df['avg_gap'] = df['gap'].rolling(window=5).mean()
    df['sum_gap'] = df['gap'].rolling(window=5).sum()
    
    # Calculate the daily volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Compute the rolling average and sum of the daily volume change
    df['avg_volume_change'] = df['volume_change'].rolling(window=5).mean()
    df['sum_volume_change'] = df['volume_change'].rolling(window=5).sum()
    
    # Calculate the standard deviation of daily returns over a window
    df['std_daily_return'] = df['daily_return'].rolling(window=5).std()
    
    # Compute the rolling sum of the absolute value of daily returns
    df['sum_abs_daily_return'] = df['daily_return'].abs().rolling(window=5).sum()
    
    # Calculate the ratio of close price to amount on each day
    df['close_to_amount'] = df['close'] / df['amount']
    
    # Compute the rolling average and sum of close price to amount ratio
    df['avg_close_to_amount'] = df['close_to_amount'].rolling(window=5).mean()
    df['sum_close_to_amount'] = df['close_to_amount'].rolling(window=5).sum()
    
    # Calculate the product of daily price range and daily volume
    df['range_volume_product'] = df['price_range'] * df['volume']
    
    # Compute the rolling average and sum of the product of daily price range and daily volume
    df['avg_range_volume_product'] = df['range_volume_product'].rolling(window=5).mean()
    df['sum_range_volume_product'] = df['range_volume_product'].rolling(window=5).sum()
    
    # Generate the alpha factor by combining multiple features
    df['alpha_factor'] = (
        df['avg_close_to_volume'] + 
        df['sum_close_to_volume'] + 
        df['avg_price_range'] + 
        df['sum_price_range'] + 
        df['avg_daily_return'] + 
        df['sum_daily_return'] + 
        df['avg_gap'] + 
        df['sum_gap'] + 
        df['avg_volume_change'] + 
        df['sum_volume_change'] + 
        df['std_daily_return'] + 
        df['sum_abs_daily_return'] + 
        df['avg_close_to_amount'] + 
        df['sum_close_to_amount'] + 
        df['avg_range_volume_product'] + 
        df['sum_range_volume_product']
    )
    
    return df['alpha_factor']
```
