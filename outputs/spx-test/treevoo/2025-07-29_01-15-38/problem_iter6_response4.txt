```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum (5-day)
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum (20-day)
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Calculate Price Momentum Oscillator
    price_momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Weight by Volume Surge
    volume_ratio = df['volume'].rolling(window=10).mean()
    previous_volume_ratio = df['volume'].shift(1).rolling(window=10).mean()
    
    volume_surge_weight = (
        (df['volume'] > 1.5 * previous_volume_ratio) * 1.5 +
        ((df['volume'] <= 1.5 * previous_volume_ratio) & (df['volume'] > previous_volume_ratio)) * 1.2 +
        (df['volume'] <= previous_volume_ratio) * 1.0
    )
    
    # Calculate Volume Change Indicator
    volume_change = df['volume'] / df['volume'].shift(1)
    
    # Integrate Price Range
    price_range = (df['high'] - df['low']).rolling(window=5).mean() / df['close']
    
    # Combine Indicators
    combined_value = (price_momentum_oscillator * volume_surge_weight * volume_change * price_range)
    
    # Apply a Threshold for Filtering
    threshold = 0.5
    factor_values = (combined_value > threshold).astype(int)
    
    return factor_values
```
