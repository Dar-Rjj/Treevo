```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate Volume-Adjusted High-Low Spread
    mean_volume = df['volume'].rolling(window=N).mean()
    adjusted_volume = df['volume'] / mean_volume
    high_low_spread = df['high'] - df['low']
    volume_adjusted_high_low = high_low_spread * adjusted_volume

    # Calculate High-Low Range Momentum
    high_low_momentum = volume_adjusted_high_low.diff().rolling(window=N).sum()

    # Calculate 5-Day Price Momentum
    close_5_day_avg = df['close'].rolling(window=5).mean()

    # Incorporate Volume Influence
    volume_5_day_avg = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / volume_5_day_avg
    adjusted_price_momentum = close_5_day_avg * relative_volume

    # Combine Momentum Factors
    combined_momentum = high_low_momentum + adjusted_price_momentum

    # Filter by Intraday Return
    intraday_return = df['close'] - df['open']
    combined_momentum[intraday_return <= 0] = 0

    return combined_momentum
```
