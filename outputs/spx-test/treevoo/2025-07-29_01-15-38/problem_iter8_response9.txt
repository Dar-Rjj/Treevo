```python
import pandas as pd

def heuristics_v2(df, N=14, M=5):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].mean()
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_volume_intraday_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute Enhanced Price Momentum
    avg_close_N_days = df['close'].rolling(window=N).mean()
    momentum_signal = df['close'] - avg_close_N_days
    sma_high = df['high'].rolling(window=14).mean()
    sma_low = df['low'].rolling(window=14).mean()
    intraday_momentum_oscillator = (sma_high - sma_low) / (sma_high - sma_low).abs()
    enhanced_price_momentum = momentum_signal + intraday_momentum_oscillator
    
    # Calculate Enhanced Volume Volatility
    avg_volume_M_days = df['volume'].rolling(window=M).mean()
    volatility_signal = df['volume'] - avg_volume_M_days
    rolling_std_volume_5_days = df['volume'].rolling(window=5).std()
    enhanced_volume_volatility = volatility_signal * rolling_std_volume_5_days
    
    # Compute 5-Day Price Momentum
    price_momentum_5_days = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    avg_volume_5_days = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / avg_volume_5_days
    adjusted_5_day_momentum = price_momentum_5_days * relative_volume
    
    # Compute Hybrid Factor
    hybrid_factor = (intraday_return + 
                     weighted_high_low_spread + 
                     adjusted_5_day_momentum + 
                     enhanced_price_momentum + 
                     enhanced_volume_volatility)
    
    # Apply Cumulative Movement
    hybrid_factor = hybrid_factor.rolling(window=5).sum()
    
    return hybrid_factor
```
