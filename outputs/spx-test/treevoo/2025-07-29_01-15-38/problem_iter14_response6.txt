```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday and Open-to-Close Returns
    intraday_returns = df['high'] - df['low']
    open_to_close_returns = df['close'] - df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Calculate 5-Day Price Momentum
    price_momentum = df['close'].rolling(window=5).mean()
    
    # Adjust for Volume Fluctuations
    relative_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = price_momentum * relative_volume
    
    # Weight Combined Returns by Volume Trend
    volume_trend = (df['volume'] > df['volume'].shift(1)).astype(int) * 2 - 1
    weighted_returns = combined_returns * volume_trend
    
    # Integrate Volume-Adjusted Short-Term Momentum and Weighted Returns
    integrated_momentum = volume_adjusted_momentum + weighted_returns
    
    # Calculate Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    sma_10 = df['close'].rolling(window=10).mean()
    price_momentum_indicator = sma_10 - df['close']
    
    # Combine Indicators
    combined_indicators = volume_change * price_momentum_indicator
    threshold = 0
    combined_indicators = (combined_indicators > threshold).astype(int)
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).sum() / 5
    long_term_momentum = df['close'].rolling(window=20).sum() / 20
    
    # Calculate Momentum Oscillator
    momentum_oscillator = short_term_momentum - long_term_momentum
    
    # Weight by Volume Surge
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    volume_weight = (volume_ratio > 1.5) * 1.5 + ((volume_ratio > 1) & (volume_ratio <= 1.5)) * 1.2 + (volume_ratio <= 1) * 1
    
    # Introduce High-Low Range Indicator
    daily_high_low_range = df['high'] - df['low']
    avg_high_low_range = daily_high_low_range.rolling(window=5).mean()
    normalized_high_low_range = daily_high_low_range / avg_high_low_range
    
    # Final Alpha Factor
    final_factor = (integrated_momentum + combined_indicators) * momentum_oscillator * volume_weight * normalized_high_low_range
    
    return final_factor
```
