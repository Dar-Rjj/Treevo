```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    range_diff = high_low_range.diff()
    range_momentum = range_diff.rolling(window=N).sum()
    
    # Filter by Close Price Increase
    close_increase = (df['close'].diff() > 0).astype(int)
    
    # Apply the momentum value if close price increased, otherwise set to 0
    factor_values = range_momentum * close_increase
    
    return factor_values
```
