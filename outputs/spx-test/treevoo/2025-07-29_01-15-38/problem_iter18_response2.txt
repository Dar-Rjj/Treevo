```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Adjusted Momentum
    adjusted_momentum = short_term_momentum * relative_volume
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate AvgRange_20
    avg_range_20 = daily_price_range.rolling(window=20).mean()
    
    # Apply Enhanced Price Range Filter
    price_range_filter = (daily_price_range > 1.5 * avg_range_20)
    adjusted_momentum[price_range_filter] = 0
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Apply Volume Filter
    volume_filter = (volume_ratio < 0.8)
    adjusted_momentum[volume_filter] = 0
    
    return adjusted_momentum
```
