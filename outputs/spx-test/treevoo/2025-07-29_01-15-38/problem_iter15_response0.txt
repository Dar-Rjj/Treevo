```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].shift(1)
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    intraday_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute Enhanced Price Momentum
    avg_close_5_days = df['close'].rolling(window=5).mean()
    momentum_signal = df['close'] - avg_close_5_days
    sma_high_14 = df['high'].rolling(window=14).mean()
    sma_low_14 = df['low'].rolling(window=14).mean()
    sma_diff = (sma_high_14 - sma_low_14) / (sma_high_14 - sma_low_14).abs()
    enhanced_price_momentum = momentum_signal + sma_diff
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend_classification = pd.Series(index=df.index, dtype=float)
    volume_trend_classification[volume_ratio > 2] = 2
    volume_trend_classification[(volume_ratio >= 0.7) & (volume_ratio <= 2)] = 0
    volume_trend_classification[volume_ratio < 0.7] = -2
    
    # Adjust Combined Returns Based on Volume Classification
    combined_returns = intraday_return + weighted_high_low_spread + enhanced_price_momentum
    adjusted_returns = combined_returns * volume_trend_classification
    
    # Calculate Enhanced Volume Volatility
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volatility_signal = df['volume'] - avg_volume_5_days
    rolling_std_volume_5_days = df['volume'].rolling(window=5).std()
    enhanced_volume_volatility = volatility_signal * rolling_std_volume_5_days
    
    # Compute 5-Day Price Momentum
    price_momentum_5_days = df['close'].rolling(window=5).sum() / 5
    
    # Incorporate Volume Influence
    avg_volume_5_days = df['volume'].rolling(window=5).sum() / 5
    relative_volume = df['volume'] / avg_volume_5_days
    adjusted_5_day_momentum = price_momentum_5_days * relative_volume
    
    # Compute Hybrid Factor
    hybrid_factor = (intraday_return + weighted_high_low_spread + 
                     enhanced_price_momentum + enhanced_volume_volatility + 
                     adjusted_5_day_momentum)
    hybrid_factor = hybrid_factor.cumsum()
    
    return hybrid_factor
```
