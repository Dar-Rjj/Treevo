```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Price Momentum Indicator
    n_days = 10
    df['sma_close'] = df['close'].rolling(window=n_days).mean()
    df['price_momentum'] = df['sma_close'] - df['close']
    
    # Calculate High-Low Range Momentum
    df['high_low_range_momentum'] = (df['high_low_range'] - df['high_low_range'].shift(1)).rolling(window=n_days).sum()
    
    # Combine Indicators
    df['combined_value'] = df['volume_change'] * df['price_momentum'] + df['high_low_range_momentum']
    
    # Filter by Close Price Increase
    df['combined_value'] = df.apply(lambda row: row['combined_value'] if row['close'] > row['close'].shift(1) else 0, axis=1)
    
    # Apply a Threshold for Filtering
    threshold = 0.5
    df['alpha_factor'] = df['combined_value'].apply(lambda x: 1 if x > threshold else 0)
    
    return df['alpha_factor']
```
