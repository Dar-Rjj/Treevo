```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Adjust for Volume Fluctuations
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    relative_vol = df['volume'] / avg_vol_5
    adjusted_momentum = short_term_momentum * relative_vol
    
    # Weight by Volume Surge
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    volume_surge_weight = 1.5 * (df['volume'] > 1.5 * avg_vol_10) + \
                          1.2 * ((df['volume'] <= 1.5 * avg_vol_10) & (df['volume'] > avg_vol_10)) + \
                          1 * (df['volume'] <= avg_vol_10)
    
    # Combine Indicators
    combined_indicator = adjusted_momentum * volume_surge_weight
    
    # Apply Price Range Filter
    price_range = df['high'] - df['low']
    avg_price_range_20 = price_range.rolling(window=20).mean()
    factor_value = combined_indicator.where(price_range <= 1.5 * avg_price_range_20, 0)
    
    return factor_value
```
