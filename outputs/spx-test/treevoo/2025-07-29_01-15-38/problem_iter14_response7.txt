```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5DayPriceMomentum'] = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    df['5DayAvgVolume'] = df['volume'].rolling(window=5).mean()
    df['RelativeVolume'] = df['volume'] / df['5DayAvgVolume']
    df['Adjusted5DayPriceMomentum'] = df['5DayPriceMomentum'] * df['RelativeVolume']
    
    # Calculate Intraday Return
    df['IntradayReturn'] = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    df['IntradayMomentum'] = df['high'] - df['low']
    
    # Adjust for Volume
    df['VolumeAdjustment'] = df['volume'] / df['volume'].mean()
    
    # Calculate Weighted High-Low Spread
    df['WeightedHighLowSpread'] = df['IntradayMomentum'] * df['VolumeAdjustment']
    
    # Combine Indicators
    df['CombinedIndicators'] = (df['IntradayReturn'] + 
                                df['WeightedHighLowSpread'] + 
                                df['Adjusted5DayPriceMomentum'])
    df['FilteredCombinedIndicators'] = df['CombinedIndicators'].apply(lambda x: 1 if x > 0 else 0)
    
    # Calculate Short-Term and Long-Term Momentum
    df['ShortTermMomentum'] = df['close'].rolling(window=5).mean()
    df['LongTermMomentum'] = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Oscillator
    df['MomentumOscillator'] = df['ShortTermMomentum'] - df['LongTermMomentum']
    
    # Weight by Volume Surge
    df['VolumeRatio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['VolumeSurgeWeight'] = df['VolumeRatio'].apply(
        lambda x: 1.5 if x > 1.5 else (1.2 if x > 1 else 1)
    )
    
    # Introduce High-Low Range Indicator
    df['DailyHighLowRange'] = df['high'] - df['low']
    df['5DayAvgHighLowRange'] = df['DailyHighLowRange'].rolling(window=5).mean()
    df['NormalizedHighLowRange'] = df['DailyHighLowRange'] / df['5DayAvgHighLowRange']
    
    # Analyze Volume Change
    df['VolumeChangeRatio'] = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    df['VolumeTrend'] = df['VolumeChangeRatio'].apply(
        lambda x: 2 if x > 1.5 else (-2 if x < 0.5 else 0)
    )
    
    # Adjust Combined Returns Based on Volume Classification
    df['AdjustedCombinedIndicators'] = (df['IntradayReturn'] + 
                                        df['WeightedHighLowSpread'] + 
                                        df['Adjusted5DayPriceMomentum']) * df['VolumeTrend']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (df['FilteredCombinedIndicators'] * 
                              df['MomentumOscillator'] * 
                              df['VolumeSurgeWeight'] * 
                              df['NormalizedHighLowRange'])
    
    return df['FinalAlphaFactor']
```
