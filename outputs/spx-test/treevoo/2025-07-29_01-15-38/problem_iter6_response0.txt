```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    
    # Calculate Price Momentum Indicator
    n_ema = 10
    df['ema_close'] = df['close'].ewm(span=n_ema, adjust=False).mean()
    df['price_momentum'] = df['ema_close'] - df['close']
    
    # Combine Indicators
    df['combined_indicator'] = df['volume_change'] * df['price_momentum']
    threshold = df['combined_indicator'].quantile(0.75)
    df['combined_indicator_filtered'] = (df['combined_indicator'] > threshold).astype(int)
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_ema = 5
    long_term_ema = 20
    df['short_term_momentum'] = df['close'].ewm(span=short_term_ema, adjust=False).mean()
    df['long_term_momentum'] = df['close'].ewm(span=long_term_ema, adjust=False).mean()
    
    # Calculate Momentum Oscillator
    df['momentum_oscillator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Weight by Volume Surge
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean().shift(1)
    df['volume_weight'] = 1
    df.loc[df['volume_ratio'] > 1.5, 'volume_weight'] = 1.8
    df.loc[(df['volume_ratio'] > 1) & (df['volume_ratio'] <= 1.5), 'volume_weight'] = 1.3
    
    # Final Alpha Factor
    df['final_factor'] = df['combined_indicator_filtered'] * df['momentum_oscillator'] * df['volume_weight']
    
    return df['final_factor']
```
