def heuristics_v2(df):
    # Calculate Intraday Returns
    intraday_returns = (df['high'] - df['low']) / df['open']
    
    # Calculate Open-to-Close Returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Open-to-Close Returns
    combined_returns = intraday_returns + open_to_close_returns
    
    # Analyze Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    def classify_volume_trend(volume_ratio):
        if volume_ratio > 1.6:
            return 2
        elif 1.5 <= volume_ratio <= 1.6:
            return 1
        elif 0.85 <= volume_ratio < 1.5:
            return 0
        elif 0.5 <= volume_ratio < 0.85:
            return -1
        else:
            return -2
    volume_trend_classification = volume_ratio.apply(classify_volume_trend)
    
    # Adjust Combined Returns Based on Volume Classification
    adjusted_combined_returns = combined_returns * volume_trend_classification
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume
    volume_adjustment = df['volume'] / df['volume'].shift(1)
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = intraday_momentum * volume_adjustment
    
    # Integrate Volume with Intraday Dynamics
    integrated_intraday_dynamics = (df['high'] - df['low']) * volume_adjustment
    
    # Compute 5-Day Price Momentum
    five_day_price_momentum = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_price_momentum * relative_volume
    
    # Adjust Combined Returns Based on Volume Classification
    final_factor = (adjusted_combined_returns + weighted_high_low_spread + adjusted_five_day_momentum) * volume_trend_classification
    
    # Apply Cumulative Movement
    cumulative_factor = final_factor.rolling(window=5).sum()
    
    return cumulative_factor
