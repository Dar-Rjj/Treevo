```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Daily Trend
    daily_trend = (df['close'] - df['open']) / ((df['close'] - df['open']).ewm(span=10).mean())
    
    # Incorporate Volatility
    volatility_adjusted_trend = daily_trend / (df['high'] - df['low'])
    
    # Integrate Volume-Adjusted Momentum
    volume_adjusted_momentum = (df['volume'] * volatility_adjusted_trend).rolling(window=5).sum()
    
    # Long-Term Trend Adjustment
    long_term_trend = (df['close'].rolling(window=20).mean() - df['close'].rolling(window=50).mean()) * volume_adjusted_momentum
    
    # Conditional Trend Sensitivity
    sensitivity = 1.2 if df['close'].rolling(window=20).mean() > df['close'].rolling(window=50).mean() else 0.8
    trend_sensitivity_adjusted = long_term_trend * sensitivity
    
    # Final Factor Calculation
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_trend_classification = (
        (volume_ratio > 1.5) * 2 
        + ((volume_ratio >= 0.8) & (volume_ratio <= 1.5)) * 1 
        + ((volume_ratio >= 0.3) & (volume_ratio < 0.8)) * 0 
        + (volume_ratio < 0.3) * (-2)
    )
    
    final_factor = trend_sensitivity_adjusted * volume_trend_classification
    
    return final_factor
```
