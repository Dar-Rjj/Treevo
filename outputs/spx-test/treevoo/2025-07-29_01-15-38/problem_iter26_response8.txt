```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Weighted High-Low Spread
    weighted_high_low_spread = (df['high'] - df['low']) * df['volume']
    
    # Compute 5-Day Price Momentum
    five_day_momentum = df['close'].rolling(window=5).mean()
    
    # Incorporate Volume Influence
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    relative_volume = df['volume'] / five_day_avg_volume
    adjusted_five_day_momentum = five_day_momentum * relative_volume
    
    # Analyze Volume Change
    volume_change_ratio = df['volume'] / df['volume'].shift(1)
    
    # Classify Volume Trend
    volume_trend_classification = volume_change_ratio.apply(
        lambda x: 2 if x > 1.5 else (-2 if x < 0.5 else 0)
    )
    
    # Daily High-Low Range
    daily_high_low_range = df['high'] - df['low']
    
    # 5-Day Average High-Low Range
    five_day_avg_high_low_range = daily_high_low_range.rolling(window=5).mean()
    
    # Daily Volume-to-Average Volume Ratio
    daily_vol_to_avg_vol_ratio = df['volume'] / five_day_avg_volume
    
    # Combine Factors Based on Volume Classification
    combined_factor = (
        intraday_return +
        weighted_high_low_spread +
        adjusted_five_day_momentum
    ) * volume_trend_classification + daily_high_low_range - five_day_avg_high_low_range + daily_vol_to_avg_vol_ratio
    
    # Apply Cumulative Movement
    cumulative_factor = combined_factor.rolling(window=5).sum()
    
    return cumulative_factor
```
