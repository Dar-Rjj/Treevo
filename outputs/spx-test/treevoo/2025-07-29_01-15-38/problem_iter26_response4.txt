```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change from the previous day
    price_change = df['close'].diff()
    
    # Calculate the volume change from the previous day
    volume_change = df['volume'].diff()
    
    # Calculate a signal based on the relationship between price change and volume change
    signal = (price_change / df['close'].shift(1)) * (volume_change / df['volume'].shift(1))
    
    # Combine the signal with the average price
    factor = signal * avg_price
    
    return factor
```
