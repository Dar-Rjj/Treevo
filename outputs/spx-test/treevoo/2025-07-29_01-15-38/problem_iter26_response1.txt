```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the buying pressure
    df['buying_pressure'] = (2 * df['close'] - df['low'] - df['high']) / df['price_range']
    
    # Calculate the relative strength
    df['relative_strength'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the factor value
    df['factor_value'] = df['buying_pressure'] * df['relative_strength'] * df['volume']
    
    # Return the factor values
    return df['factor_value']
```
