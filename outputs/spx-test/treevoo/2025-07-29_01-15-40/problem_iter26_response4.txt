```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    df['High_Low_Momentum_Score'] = df['Intraday_High_Low_Spread'].diff()
    df['High_Low_Momentum_Score'] = df['High_Low_Momentum_Score'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    df['High_Low_Momentum_Score_Adjusted'] = df['High_Low_Momentum_Score'] * df['volume']
    
    # Calculate Price Momentum over 5 days
    period = 5
    df['Price_Change'] = df['close'].pct_change(periods=period)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=period).sum()
    
    # Calculate Volume Trend over 5 days
    df['Volume_Change'] = df['volume'].pct_change(periods=period)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=period).sum()
    
    # Enhance Market Sentiment Analysis
    # Calculate Range Momentum
    df['Daily_Range'] = df['high'] - df['low']
    df['Range_Momentum'] = df['Daily_Range'].diff()
    df['Range_Momentum'] = df['Range_Momentum'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Open_Close_Movement'] = df['Open_Close_Diff'].rolling(window=period).sum()
    
    # Combine Factors
    high_low_coeff = 1.0
    price_momentum_coeff = 0.8
    volume_trend_coeff = 0.6
    range_momentum_coeff = 0.4
    open_close_movement_coeff = 0.2
    
    df['Factor_Value'] = (df['High_Low_Momentum_Score_Adjusted'] * high_low_coeff +
                          df['Price_Momentum'] * price_momentum_coeff +
                          df['Volume_Trend'] * volume_trend_coeff +
                          df['Range_Momentum'] * range_momentum_coeff +
                          df['Open_Close_Movement'] * open_close_movement_coeff)
    
    return df['Factor_Value'].dropna()
```
