```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=5, M=5, K=5):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Calculate Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=N).mean()

    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(df['intraday_high_low_spread'] > df['avg_intraday_high_low_spread'], 1, -1) * df['volume']

    # Calculate Price Momentum over the past 5 days
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_momentum'] = df['price_change'].rolling(window=5).sum()

    # Calculate Volume Trend over the past 5 days
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_trend'] = df['volume_change'].rolling(window=5).sum()

    # Enhance Market Sentiment
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=M).mean()
    df['range_momentum_score'] = np.where(df['daily_range'] > df['avg_daily_range'], 1, -1)

    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_sum'] = df['open_close_diff'].rolling(window=5).sum()

    # Incorporate Intraday Volatility using Garman-Klass formula
    df['intraday_volatility'] = 0.5 * (np.log(df['high'] / df['low']) ** 2) - (2 * np.log(2) - 1) * (np.log(df['close'] / df['open']) ** 2)
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=K).mean()
    df['intraday_volatility_score'] = np.where(df['intraday_volatility'] > df['avg_intraday_volatility'], 1, -1)

    # Combine Factors
    factor = (
        0.3 * df['high_low_momentum_score'] + 
        0.2 * df['price_momentum'] + 
        0.1 * df['volume_trend'] + 
        0.2 * df['range_momentum_score'] + 
        0.1 * df['open_close_sum'] + 
        0.1 * df['intraday_volatility_score']
    )

    # Incorporate Open-Price Change
    df['open_price_change'] = df['open'] - df['open'].shift(5)
    factor += 0.1 * df['open_price_change']

    return factor
```
