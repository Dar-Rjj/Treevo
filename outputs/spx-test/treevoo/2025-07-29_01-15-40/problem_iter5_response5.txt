import pandas as pd
import numpy as np

def heuristics_v2(df, N=5, M=20):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread over past N days
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=N).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(df['high_low_spread'] > df['avg_high_low_spread'], 1, -1)
    
    # Adjust by Volume
    df['high_low_momentum_score_vol_adj'] = df['high_low_momentum_score'] * df['volume']
    
    # Calculate 5-Day Momentum
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    df['20_day_volume_ma'] = df['volume'].rolling(window=M).mean()
    
    # Compute Volume Deviation from MA
    df['volume_dev_from_ma'] = df['volume'] - df['20_day_volume_ma']
    
    # Calculate Simple Moving Average (SMA) of Close
    df['sma_close'] = df['close'].rolling(window=N).mean()
    
    # Integrate Volume into Momentum
    # Compute Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] + df['open']) / 2 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Combine SMA and VWAP
    df['integrated_momentum'] = (df['sma_close'] + df['vwap']) / 2
    
    # Calculate 5-Day Price Range
    df['5_day_price_range'] = df['high'] - df['low']
    
    # Calculate Price Range and Volume Synchronicity
    df['price_range_vol_synchronicity'] = df['5_day_price_range'] * df['volume_dev_from_ma']
    
    # Calculate Momentum and Volume Synchronicity
    df['momentum_vol_synchronicity'] = df['5_day_momentum'] * df['volume_dev_from_ma']
    
    # Combine Synchronicities
    df['combined_synchronicities'] = df['price_range_vol_synchronicity'] + df['momentum_vol_synchronicity']
    
    # Apply Exponential Smoothing
    df['smoothed_combined_synchronicities'] = df['combined_synchronicities'].ewm(span=N).mean()
    
    # Calculate Price Reversal Indicator
    df['price_reversal_indicator'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Composite Factor Construction
    df['composite_factor'] = (df['high_low_momentum_score_vol_adj'] * df['integrated_momentum'] * df['price_reversal_indicator'] +
                              df['smoothed_combined_synchronicities'])
    
    # Return the final factor
    return df['composite_factor']
