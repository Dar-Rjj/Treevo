```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['Avg_High-Low_Spread'] = df['High-Low_Spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['High-Low_Momentum_Score'] = (df['High-Low_Spread'] - df['Avg_High-Low_Spread']) * df['volume']
    
    # Calculate Simple Moving Average (SMA) of Close for N days
    df['SMA_Close'] = df['close'].rolling(window=n).mean()
    
    # Calculate Price Reversal Indicator
    df['Open-Close_Spread'] = df['open'] - df['close']
    df['Price_Reversal_Indicator'] = df['High-Low_Spread'] - df['Open-Close_Spread']
    
    # Integrate Volume into Momentum: Compute Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine SMA and VWAP
    df['Integrated_Momentum'] = (df['SMA_Close'] + df['VWAP']) / 2
    
    # Composite Factor Construction
    df['Composite_Factor'] = df['High-Low_Momentum_Score'] * df['Integrated_Momentum'] * df['Price_Reversal_Indicator']
    
    # Calculate Price Momentum
    df['Price_Change'] = df['close'] - df['close'].shift(n)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=n).sum()
    
    # Calculate Volume Trend
    df['Volume_Change'] = df['volume'] - df['volume'].shift(n)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=n).sum()
    
    # Combine Momentum and Volume Interaction
    price_momentum_coeff = 0.7
    volume_trend_coeff = 0.3
    df['Weighted_Sum'] = (price_momentum_coeff * df['Price_Momentum']) + (volume_trend_coeff * df['Volume_Trend'])
    df['Enhanced_Factor_MV'] = df['Weighted_Sum'] * (df['high'] - df['low'])
    
    # Calculate Intraday Open-Close Spread
    df['Intraday_Open-Close_Spread'] = df['open'] - df['close']
    
    # Compute Average Intraday Open-Close Spread over past N days
    df['Avg_Open-Close_Spread'] = df['Intraday_Open-Close_Spread'].rolling(window=n).mean()
    
    # Compute Open-Close Momentum Score
    df['Open-Close_Momentum_Score'] = (df['Intraday_Open-Close_Spread'] - df['Avg_Open-Close_Spread']) * df['volume']
    
    # Enhance Composite Factor
    df['Enhanced_Composite_Factor'] = (df['Open-Close_Momentum_Score'] * df['High-Low_Momentum_Score'] *
                                       df['Integrated_Momentum'] * df['Price_Reversal_Indicator'])
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Enhanced_Composite_Factor'] + df['Enhanced_Factor_MV']) / 2
    
    return df['Final_Alpha_Factor']
```
