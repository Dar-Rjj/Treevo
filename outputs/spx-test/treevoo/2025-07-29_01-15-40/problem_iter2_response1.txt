```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios with a 10-day window
    df['rolling_avg_high_low'] = df['high_low_ratio'].rolling(window=10).mean()
    
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Exponential Weighted Moving Average (EWMA) of Log Return with a 21-day lookback
    df['ewma_log_return'] = df['log_return'].ewm(span=21, adjust=False).mean()
    
    # Calculate Volume Relative Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price and Volume Information
    df['integrated_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Combine Signals
    df['combined_momentum'] = df['rolling_avg_high_low'] * df['integrated_momentum']
    
    return df['combined_momentum']
```
