```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Ratio Momentum
    high_low_ratio = df['high'] / df['low']
    high_low_rolling_mean = high_low_ratio.rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_change = df['close'].pct_change(periods=5)
    price_momentum = price_change.sum()
    
    # Integrate Volume into Momentum
    vwap = (df[['close', 'open']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Price Reversal Indicator
    high_low_spread = df['high'] - df['low']
    open_close_spread = abs(df['open'] - df['close'])
    price_reversal = high_low_spread - open_close_spread
    
    # Calculate Upward and Downward Movements
    upward_movement = df['high'] - df['close'].shift(1)
    downward_movement = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    net_directional = pd.Series(0, index=df.index)
    net_directional[upward_movement > downward_movement] = 1
    net_directional[upward_movement < downward_movement] = -1
    
    # Calculate Cumulative Directional Movement Adjusted (CDMA)
    cdma = net_directional * df['volume']
    cdma = cdma.rolling(window=5).sum()
    
    # Additional Trend Strength Indicator
    true_range = df['high'] - df['low']
    atr = true_range.rolling(window=14).mean()
    trend_strength = cdma * atr
    
    # Composite Factor Construction
    composite_factor = (high_low_rolling_mean + price_momentum) * cdma
    composite_factor_refined = (high_low_rolling_mean + price_momentum + trend_strength) * cdma
    
    return composite_factor_refined
```
