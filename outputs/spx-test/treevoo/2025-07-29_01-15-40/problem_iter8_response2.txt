```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Cumulative Directional Movement
    df['Upward'] = df['high'].diff().apply(lambda x: max(x, 0))
    df['Downward'] = df['low'].diff().apply(lambda x: min(x, 0))
    df['Net_Movement'] = df[['Upward', 'Downward']].apply(lambda x: x['Upward'] if x['Upward'] > -x['Downward'] else x['Downward'], axis=1)
    df['Cumulative_Directional_Movement'] = df['Net_Movement'].rolling(window=5).sum()

    # Calculate 5-Day Price Momentum
    df['5Day_Price_Momentum'] = df['close'] - df['close'].shift(5)

    # Calculate 20-Day Volume Moving Average
    df['20Day_Volume_MA'] = df['volume'].rolling(window=20).mean()

    # Compute Volume Deviation from MA
    df['Volume_Deviation'] = df['volume'] - df['20Day_Volume_MA']

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # Calculate 5-Day High-Low Spread
    df['5Day_High_Low_Spread'] = df['high'] - df['low']

    # Calculate Open-Close Spread
    df['Open_Close_Spread'] = df['open'] - df['close']

    # Calculate Price Reversal Indicator
    df['Price_Reversal_Indicator'] = df['5Day_High_Low_Spread'] - df['Open_Close_Spread']

    # Compute 5-Day Price Range and Volume Synchronicity
    df['5Day_Price_Range_Volume_Sync'] = df['5Day_High_Low_Spread'] * df['Volume_Deviation']

    # Compute 5-Day Momentum and Volume Synchronicity
    df['5Day_Momentum_Volume_Sync'] = df['5Day_Price_Momentum'] * df['Volume_Deviation']

    # Combine Synchronicities
    df['Combined_Synchronicities'] = df['5Day_Price_Range_Volume_Sync'] + df['5Day_Momentum_Volume_Sync']

    # Apply Exponential Smoothing
    df['Smoothed_Synchronicities'] = df['Combined_Synchronicities'].ewm(span=5, adjust=False).mean()

    # Adjust by Volume
    df['Adjusted_Synchronicities'] = df['Smoothed_Synchronicities'] * df['volume']

    # Combine Directional, Momentum, and Synchronicity
    df['Integrated_Directional_Momentum_Sync'] = df['Cumulative_Directional_Movement'] + df['Adjusted_Synchronicities']

    # Final Factor Construction
    df['Final_Factor'] = df['Integrated_Directional_Momentum_Sync'] * df['Price_Reversal_Indicator']

    return df['Final_Factor']
```
