```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: A Series of alpha factor values, indexed by date.
    """
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the difference between the high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the price momentum as the difference between the current close and the previous close
    price_momentum = df['close'].diff()
    
    # Calculate the volume change as the difference between the current volume and the previous volume
    volume_change = df['volume'].diff()
    
    # Calculate the heuristic factor
    heuristic_factor = (avg_price * price_momentum) / (price_range * volume_change)
    
    # Handle division by zero
    heuristic_factor = heuristic_factor.replace([float('inf'), -float('inf')], 0).fillna(0)
    
    return heuristic_factor
```
