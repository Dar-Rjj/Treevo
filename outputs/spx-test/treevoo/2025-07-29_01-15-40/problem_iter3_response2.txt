```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    momentum_5d = df['close'].diff(5)
    
    # Calculate 20-Day Volume Moving Average
    volume_ma_20d = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    volume_deviation = df['volume'] - volume_ma_20d
    
    # Calculate 5-Day Price Range
    price_range_5d = df['high'] - df['low']
    
    # Calculate Price Range and Volume Synchronicity
    range_volume_sync = price_range_5d * volume_deviation
    
    # Calculate Momentum and Volume Synchronicity
    momentum_volume_sync = momentum_5d * volume_deviation
    
    # Combine Synchronicities
    combined_sync = momentum_volume_sync + range_volume_sync
    
    # Apply Exponential Smoothing
    alpha = 0.2  # Smoothing factor
    smoothed_sync = combined_sync.ewm(alpha=alpha, adjust=False).mean()
    
    return smoothed_sync
```
