```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pandas.DataFrame): DataFrame with index as date and columns as open, high, low, close, volume.
    
    Returns:
    pandas.Series: A Series with index as date and values as the calculated alpha factor.
    """
    
    # Calculate the body size of the candle
    body_size = df['close'] - df['open']
    
    # Calculate the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the upper and lower wicks
    upper_wick = df['high'] - df[['open', 'close']].max(axis=1)
    lower_wick = df[['open', 'close']].min(axis=1) - df['low']
    
    # Determine the relative size of the wicks to the daily range
    upper_wick_ratio = upper_wick / daily_range
    lower_wick_ratio = lower_wick / daily_range
    
    # Calculate the alpha factor
    # The factor considers the dominance of the upper or lower wick
    # and the relative size of the body compared to the daily range
    alpha_factor = (upper_wick_ratio - lower_wick_ratio) * (body_size / daily_range)
    
    return alpha_factor
```
