import pandas as pd

def heuristics_v2(df, n=5, m=10, p=5):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['Average_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score and Adjust by Volume
    df['High_Low_Momentum_Score'] = (df['Intraday_High_Low_Spread'] - df['Average_Intraday_High_Low_Spread']) * df['volume']
    
    # Calculate Price Momentum over the period
    df['Price_Change'] = df['close'].diff(periods=1)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=n).sum()
    
    # Calculate Volume Trend over the period
    df['Volume_Change'] = df['volume'].diff(periods=1)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Compute Average Daily Range over past M days
    df['Average_Daily_Range'] = df['Daily_Range'].rolling(window=m).mean()
    
    # Calculate Range Momentum
    df['Range_Momentum'] = df['Daily_Range'] - df['Average_Daily_Range']
    
    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Price_Difference'] = df['close'] - df['open']
    df['Open_Price_Movement'] = df['Open_Close_Price_Difference'].rolling(window=n).sum()
    
    # Calculate Intraday Price Volatility
    df['Intraday_Price_Volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1).rolling(window=p).mean()
    
    # Compute Price Volatility Score and Adjust by Volume
    df['Average_Price_Volatility'] = df['Intraday_Price_Volatility'].rolling(window=m).mean()
    df['Price_Volatility_Score'] = (df['Intraday_Price_Volatility'] - df['Average_Price_Volatility']) * df['volume']
    
    # Combine Factors using a weighted sum
    df['Alpha_Factor'] = (0.2 * df['High_Low_Momentum_Score'] +
                          0.3 * df['Price_Momentum'] +
                          0.1 * df['Volume_Trend'] +
                          0.1 * df['Range_Momentum'] +
                          0.2 * df['Open_Price_Movement'] +
                          0.1 * df['Price_Volatility_Score'])
    
    return df['Alpha_Factor'].dropna()
