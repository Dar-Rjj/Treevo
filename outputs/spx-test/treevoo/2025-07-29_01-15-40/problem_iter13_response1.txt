```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return
    df['ewma_log_return'] = df['log_return'].ewm(span=21, adjust=False).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combined Momentum
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Determine Net Directional Movement
    df['upward_move'] = df['high'] - df['close'].shift(1)
    df['downward_move'] = df['low'] - df['close'].shift(1)
    df['net_directional_movement'] = np.where(df['upward_move'] > df['downward_move'], 1,
                                              np.where(df['upward_move'] < df['downward_move'], -1, 0))
    
    # Adjusted Cumulative Directional Movement
    df['cumulative_directional_movement'] = df['net_directional_movement'].rolling(window=21).sum() * df['volume']
    
    # Add Trend Strength Indicator
    df['daily_range'] = df['high'] - df['low']
    df['average_daily_range'] = df['daily_range'].rolling(window=21).mean()
    df['trend_strength'] = np.where(df['daily_range'] > df['average_daily_range'], 1, 0)
    
    # Integrated Factor
    df['integrated_factor'] = (df['combined_momentum'] + df['cumulative_directional_movement']) * df['trend_strength']
    
    return df['integrated_factor']
```
