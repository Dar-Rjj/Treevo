```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return with 21-day lookback period
    df['ewma_log_return'] = df['log_return'].ewm(span=21, adjust=False).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combined Momentum
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Price Momentum over a 5-day period
    df['price_change'] = df['close'] - df['close'].shift(5)
    df['price_momentum'] = df['price_change'].rolling(window=5).sum()
    
    # Volume Trend over a 5-day period
    df['volume_trend'] = df['volume'] - df['volume'].shift(5)
    df['volume_trend'] = df['volume_trend'].rolling(window=5).sum()
    
    # Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] + df['open']) / 2 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Combine SMA and VWAP
    df['sma_close'] = df['close'].rolling(window=20).mean()
    df['combined_price_volume'] = (df['sma_close'] + df['vwap']) / 2
    
    # Price Reversal Indicator
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['open'] - df['close']
    df['price_reversal'] = df['high_low_spread'] - df['open_close_spread']
    
    # Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Net Directional Movement
    df['net_directional_movement'] = np.where(df['upward_movement'] > df['downward_movement'], 1,
                                              np.where(df['upward_movement'] < df['downward_movement'], -1, 0))
    
    # Cumulative Directional Movement Adjusted (CDMA)
    df['cdma'] = df['net_directional_movement'] * df['volume']
    df['cdma'] = df['cdma'].rolling(window=10).sum()
    
    # Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Rolling Average of High-Low Ratios
    df['rolling_high_low_ratio'] = df['high_low_ratio'].rolling(window=10).mean()
    
    # Volume Weighted Factor
    df['volume_weight'] = df['volume'] / df['close']
    df['volume_weighted_high_low_ratio'] = df['high_low_ratio'] * df['volume_weight']
    df['rolling_volume_weighted_high_low_ratio'] = df['volume_weighted_high_low_ratio'].rolling(window=10).mean()
    
    # Composite Factor Construction
    df['composite_factor'] = (df['price_momentum'] + df['volume_trend'] + df['cdma']) * df['combined_momentum']
    
    return df['composite_factor']
```
