```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, k=10):
    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=n).mean()
    
    # High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(df['intraday_high_low_spread'] > df['avg_intraday_high_low_spread'], 1, -1)
    df['high_low_momentum_score'] *= df['volume']
    
    # Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(n)
    df['volume_trend'] = df['volume_change'].rolling(window=n).sum()
    
    # Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Average Daily Range over past M days
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    
    # Range Momentum
    df['range_momentum_score'] = np.where(df['daily_range'] > df['avg_daily_range'], 1, -1)
    
    # Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=n).sum()
    
    # Intraday Volatility using Garman-Klass formula
    df['intraday_volatility'] = 0.5 * (np.log(df['high'] / df['low'])**2) - (2 * np.log(2) - 1) * (np.log(df['close'] / df['open'])**2)
    
    # Average Intraday Volatility over past K days
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=k).mean()
    
    # Intraday Volatility Score
    df['intraday_volatility_score'] = np.where(df['intraday_volatility'] > df['avg_intraday_volatility'], 1, -1)
    
    # Open-Price Momentum
    df['open_price_change'] = df['open'] - df['open'].shift(n)
    df['open_price_momentum'] = df['open_price_change'].rolling(window=n).sum()
    
    # Combine Factors
    df['factor'] = (
        0.2 * df['high_low_momentum_score'] + 
        0.2 * df['price_momentum'] + 
        0.1 * df['volume_trend'] + 
        0.1 * df['range_momentum_score'] + 
        0.1 * df['open_close_movement'] + 
        0.1 * df['intraday_volatility_score'] + 
        0.2 * df['open_price_momentum']
    )
    
    return df['factor']
```
