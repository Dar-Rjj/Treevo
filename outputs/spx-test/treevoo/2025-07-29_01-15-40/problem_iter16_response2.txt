```python
import pandas as pd

def heuristics_v2(df, period=5, high_low_coeff=0.3, price_mom_coeff=0.25, volume_trend_coeff=0.2, range_mom_coeff=0.25):
    # Calculate Intraday High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    avg_high_low_spread = df['HighLowSpread'].rolling(window=period).mean()
    high_low_momentum_score = (df['HighLowSpread'] - avg_high_low_spread) * df['volume']
    high_low_momentum_score = high_low_momentum_score / high_low_momentum_score.abs().mean()  # Normalize to make it comparable
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff(period)
    
    # Calculate Volume Trend
    volume_trend = df['volume'].diff(period)
    
    # Enhance Market Sentiment Analysis
    df['DailyRange'] = df['high'] - df['low']
    avg_daily_range = df['DailyRange'].rolling(window=period).mean()
    range_momentum = df['DailyRange'] - avg_daily_range
    open_close_diff = df['close'] - df['open']
    
    # Combine all the components
    factor_values = (
        high_low_coeff * high_low_momentum_score +
        price_mom_coeff * price_momentum +
        volume_trend_coeff * volume_trend +
        range_mom_coeff * range_momentum
    )
    
    return factor_values
```
