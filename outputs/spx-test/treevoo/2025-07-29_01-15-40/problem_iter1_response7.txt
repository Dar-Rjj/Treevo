```python
import pandas as pd

def heuristics_v2(df):
    # Determine the period for momentum and volume trend
    period = 5

    # Calculate Price Momentum
    price_changes = df['close'] - df['close'].shift(period)
    price_momentum = price_changes.rolling(window=period).sum()

    # Calculate Volume Trend
    volume_changes = df['volume'] - df['volume'].shift(period)
    volume_trend = volume_changes.rolling(window=period).sum()

    # Normalize Price Momentum
    price_momentum_normalized = price_momentum / price_momentum.abs()
    
    # Normalize Volume Trend
    volume_trend_normalized = volume_trend / volume_trend.abs()
    
    # Define coefficients for weighted sum
    price_coefficient = 0.6
    volume_coefficient = 0.4
    
    # Weighted Sum
    weighted_sum = (price_momentum_normalized * price_coefficient) + (volume_trend_normalized * volume_coefficient)
    
    # Adjust for Market Sentiment using High-Low Range
    high_low_range = df['high'].rolling(window=period).max() - df['low'].rolling(window=period).min()
    final_factor = weighted_sum + (high_low_range / df['close'])

    return final_factor
```
