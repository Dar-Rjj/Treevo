```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    df['5DayMomentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    df['20DayVolMA'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    df['VolDevFromMA'] = df['volume'] - df['20DayVolMA']
    
    # Calculate Price Momentum over a period of 5 days
    df['PriceMomentum'] = (df['close'] - df['close'].shift(5)).rolling(window=5).sum()
    
    # Calculate Volume Trend over a period of 5 days
    df['VolumeTrend'] = (df['volume'] - df['volume'].shift(5)).rolling(window=5).sum()
    
    # Combine Momentum and Volume Synchronicity
    df['MomentumVolSync'] = df['5DayMomentum'] * df['VolDevFromMA']
    
    # Apply Exponential Smoothing to reduce noise
    alpha = 0.6  # Smoothing factor
    df['SmoothedSync'] = df['MomentumVolSync'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Final Factor
    weight_smoothed_sync = 0.7
    weight_volume_trend = 0.3
    df['FinalFactor'] = (weight_smoothed_sync * df['SmoothedSync'] + 
                         weight_volume_trend * df['VolumeTrend'])
    
    # Adjust for Market Sentiment using High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    df['FinalAlphaFactor'] = df['FinalFactor'] * df['HighLowRange']
    
    return df['FinalAlphaFactor']
```
