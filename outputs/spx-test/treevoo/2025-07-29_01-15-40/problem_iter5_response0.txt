```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios (e.g., 10-day window)
    rolling_avg_high_low_ratio = high_low_ratio.rolling(window=10).mean()
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate True Range
    prev_close = df['close'].shift(1)
    high_low_diff = df['high'] - df['low']
    high_prev_close_diff = (df['high'] - prev_close).abs()
    low_prev_close_diff = (df['low'] - prev_close).abs()
    true_range = pd.concat([high_low_diff, high_prev_close_diff, low_prev_close_diff], axis=1).max(axis=1)
    
    # Compute Intraday Volatility Adjusted Return
    vol_adj_intraday_return = intraday_return / true_range
    
    # Integrate Price and Volume Information
    volume_change = df['volume'] / df['volume'].shift(1)
    integrated_signal = vol_adj_intraday_return * volume_change
    
    # Combine Signals
    combined_momentum = rolling_avg_high_low_ratio * integrated_signal
    
    return combined_momentum
```
