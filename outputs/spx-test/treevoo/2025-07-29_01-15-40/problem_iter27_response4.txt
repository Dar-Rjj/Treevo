```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=5, m_days=5, k_days=5):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=n_days).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(
        df['intraday_high_low_spread'] > df['avg_intraday_high_low_spread'], 1, -1
    )
    df['high_low_momentum_score'] *= df['volume']
    
    # Calculate Price Momentum over the period
    df['price_change'] = df['close'] - df['close'].shift(n_days)
    df['price_momentum'] = df['price_change'].rolling(window=n_days).sum()
    
    # Calculate Volume Trend over the period
    df['volume_change'] = df['volume'] - df['volume'].shift(n_days)
    df['volume_trend'] = df['volume_change'].rolling(window=n_days).sum()
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m_days).mean()
    
    # Calculate Range Momentum
    df['range_momentum'] = np.where(
        df['daily_range'] > df['avg_daily_range'], 1, -1
    )
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_momentum'] = df['open_close_diff'].rolling(window=n_days).sum()
    
    # Incorporate Intraday Volatility
    df['intraday_volatility'] = 0.5 * (df['high'] - df['low']) ** 2 + 0.069 * (df['close'] - df['open']) * (df['high'] + df['low'] - 2 * df['close'])
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=k_days).mean()
    
    # Compute Intraday Volatility Score
    df['volatility_score'] = np.where(
        df['intraday_volatility'] > df['avg_intraday_volatility'], 1, -1
    )
    
    # Calculate Open-Price Momentum
    df['open_price_change'] = df['open'] - df['open'].shift(n_days)
    df['open_price_momentum'] = df['open_price_change'].rolling(window=n_days).sum()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Combine Factors with weighted sum
    df['alpha_factor'] = (
        0.2 * df['high_low_momentum_score'] +
        0.2 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.1 * df['range_momentum'] +
        0.1 * df['open_close_momentum'] +
        0.1 * df['volatility_score'] +
        0.1 * df['open_price_momentum'] +
        0.1 * df['intraday_return']
    )
    
    return df['alpha_factor']
```
