```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    volume_weighted_mean = (high_low_ratio.rolling(window=rolling_window).sum() * df['volume'].rolling(window=rolling_window).sum()) / df['volume'].rolling(window=rolling_window).sum()
    
    # Smooth the Volume-Weighted Mean using Exponential Smoothing
    smoothing_factor = 0.2
    smoothed_factor = volume_weighted_mean.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return smoothed_factor
```
