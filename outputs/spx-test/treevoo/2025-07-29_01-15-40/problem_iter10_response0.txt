```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['HighLowRatio'] = df['high'] / df['low']

    # Calculate Rolling Average of High-Low Ratios
    df['RollingAvgHighLow'] = df['HighLowRatio'].rolling(window=10).mean()

    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']

    # Calculate True Range
    df['TrueRange1'] = df['high'] - df['low']
    df['TrueRange2'] = (df['high'] - df['close'].shift(1)).abs()
    df['TrueRange3'] = (df['low'] - df['close'].shift(1)).abs()
    df['TrueRange'] = df[['TrueRange1', 'TrueRange2', 'TrueRange3']].max(axis=1)

    # Compute Intraday Volatility Adjusted Return
    df['VolatilityAdjustedReturn'] = df['IntradayReturn'] / df['TrueRange']

    # Integrate Price and Volume Information
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    df['IntegratedSignal'] = df['VolatilityAdjustedReturn'] * df['VolumeChange']

    # Combine Signals
    df['CombinedMomentum'] = df['RollingAvgHighLow'] * df['IntegratedSignal']

    return df['CombinedMomentum']
```
