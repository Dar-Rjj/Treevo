```python
import pandas as pd

def heuristics(df):
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']

    # Calculate Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    df['volume_weighted_high_low_ratio'] = (df['high_low_ratio'] * df['volume']).rolling(window=rolling_window).sum() / df['volume'].rolling(window=rolling_window).sum()

    # Calculate Rolling Standard Deviation of Close Prices
    df['std_close'] = df['close'].rolling(window=rolling_window).std()

    # Adjust the Volume-Weighted Mean by Volatility
    df['alpha_factor'] = df['volume_weighted_high_low_ratio'] / df['std_close']

    return df['alpha_factor']
```
