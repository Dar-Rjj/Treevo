```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the (open, high, low, close) prices
    df['avg_price'] = df[['open', 'high', 'low', 'close']].mean(axis=1)
    
    # Calculate the difference between the close price and the average price
    df['close_avg_diff'] = df['close'] - df['avg_price']
    
    # Calculate the momentum of the close price
    df['momentum'] = df['close'].pct_change()
    
    # Compute a factor based on the combination of the close-avg difference and momentum
    df['factor'] = df['close_avg_diff'] * df['momentum']
    
    # Return the factor as a pandas Series
    return df['factor']
```
