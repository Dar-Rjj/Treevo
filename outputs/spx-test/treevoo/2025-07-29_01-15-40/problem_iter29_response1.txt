import pandas as pd

def heuristics_v2(df, n=5, m=10, k=5):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = (df['high_low_spread'] - df['avg_high_low_spread']).apply(lambda x: 1 if x > 0 else -1)
    df['high_low_momentum_score'] *= df['volume']
    
    # Calculate Price Momentum over the past n days
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Calculate Volume Trend over the past n days
    df['volume_change'] = df['volume'] - df['volume'].shift(n)
    df['volume_trend'] = df['volume_change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    df['range_momentum'] = (df['daily_range'] - df['avg_daily_range']).apply(lambda x: 1 if x > 0 else -1)
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_momentum'] = df['open_close_diff'].rolling(window=n).sum()
    
    # Incorporate Intraday Volatility using Garman-Klass formula
    df['intraday_volatility'] = 0.5 * (df['high'] - df['low'])**2 / (4 * np.log(2)) - (2 * np.log(2) - 1) * (df['close'] - df['open'])**2
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=k).mean()
    df['intraday_volatility_score'] = (df['intraday_volatility'] - df['avg_intraday_volatility']).apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Open-Price Momentum over the past n days
    df['open_price_change'] = df['open'] - df['open'].shift(n)
    df['open_price_momentum'] = df['open_price_change'].rolling(window=n).sum()
    
    # Calculate Open to Low Price Difference
    df['open_to_low_diff'] = df['open'] - df['low']
    df['open_to_low_momentum'] = df['open_to_low_diff'].rolling(window=n).sum()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Combine Factors with weighted sum
    factor = (0.2 * df['high_low_momentum_score'] 
              + 0.2 * df['price_momentum'] 
              + 0.1 * df['volume_trend'] 
              + 0.1 * df['range_momentum'] 
              + 0.1 * df['open_close_momentum'] 
              + 0.2 * df['intraday_volatility_score'] 
              + 0.1 * df['open_price_momentum'] 
              + 0.1 * df['intraday_return'])
    
    return factor.dropna()
