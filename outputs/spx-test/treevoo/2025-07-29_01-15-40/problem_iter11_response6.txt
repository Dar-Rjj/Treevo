```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Ratio Momentum
    df['high_low_ratio'] = df['high'] / df['low']
    high_low_momentum = df['high_low_ratio'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    n_days = 5
    price_change = df['close'].diff(periods=n_days)
    price_momentum = price_change.rolling(window=n_days).sum()
    
    # Integrate Volume into Momentum
    df['vwap'] = (df[['close', 'open']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    sma = df['close'].rolling(window=10).mean()
    vwap_adjusted_momentum = sma + df['vwap']
    
    # Calculate Price Reversal Indicator
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['open'] - df['close']
    price_reversal_indicator = high_low_spread - open_close_spread
    
    # Calculate Upward and Downward Movements
    upward_movement = df['high'] - df['close'].shift(1)
    downward_movement = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    net_directional_movement = pd.Series(index=df.index, dtype='float64')
    net_directional_movement[upward_movement > downward_movement] = 1
    net_directional_movement[upward_movement < downward_movement] = -1
    net_directional_movement[(upward_movement == downward_movement)] = 0
    
    # Calculate Cumulative Directional Movement Adjusted (CDMA)
    cdma = (net_directional_movement * df['volume']).rolling(window=10).sum()
    
    # Composite Factor Construction
    weighted_sum = 0.5 * high_low_momentum + 0.5 * price_momentum
    final_factor = weighted_sum * cdma
    
    return final_factor
```
