```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5DayMomentum'] = df['close'].diff(5)
    
    # Calculate 20-Day Volume Moving Average
    df['20DayVolMA'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    df['VolumeDeviation'] = df['volume'] - df['20DayVolMA']
    
    # Calculate 5-Day High-Low Spread
    df['5DayHighLowSpread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['OpenCloseSpread'] = df['open'] - df['close']
    
    # Calculate Price Reversal Indicator
    df['PriceReversal'] = df['5DayHighLowSpread'] - df['OpenCloseSpread']
    
    # Compute 5-Day Price Range and Volume Synchronicity
    df['PriceRangeVolumeSync'] = df['5DayHighLowSpread'] * df['VolumeDeviation']
    
    # Compute 5-Day Momentum and Volume Synchronicity
    df['MomentumVolumeSync'] = df['5DayMomentum'] * df['VolumeDeviation']
    
    # Combine Synchronicities
    df['CombinedSynchronicities'] = df['PriceRangeVolumeSync'] + df['MomentumVolumeSync']
    
    # Apply Exponential Smoothing
    alpha = 0.2
    df['SmoothedSynchronicities'] = df['CombinedSynchronicities'].ewm(alpha=alpha, adjust=False).mean()
    
    # Final Factor Construction
    # Add 10-Day Price Momentum
    df['10DayMomentum'] = df['close'].diff(10)
    df['FinalFactor'] = (df['SmoothedSynchronicities'] * df['PriceReversal']) + df['10DayMomentum']
    
    return df['FinalFactor']
```
