```python
import pandas as pd

def heuristics_v2(df, n_days=5, m_days=5, p_days=5):
    # Calculate Intraday High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread
    df['AvgHighLowSpread'] = df['HighLowSpread'].rolling(window=n_days).mean()
    
    # Compute High-Low Momentum Score
    df['HighLowMomentumScore'] = (df['HighLowSpread'] - df['AvgHighLowSpread']) * df['volume']
    
    # Calculate Price Momentum
    df['PriceMomentum'] = df['close'].diff(periods=n_days)
    
    # Calculate Volume Trend
    df['VolumeTrend'] = df['volume'].diff(periods=n_days)
    
    # Enhance Market Sentiment Analysis
    df['DailyRange'] = df['high'] - df['low']
    df['AvgDailyRange'] = df['DailyRange'].rolling(window=m_days).mean()
    df['RangeMomentum'] = (df['DailyRange'] - df['AvgDailyRange'])
    
    # Integrate Open-Price and Close-Price Movement
    df['OpenCloseMovement'] = df['close'] - df['open']
    df['OpenCloseMovementSum'] = df['OpenCloseMovement'].rolling(window=n_days).sum()
    
    # Calculate Open to Low Price Difference
    df['OpenLowDifference'] = df['open'] - df['low']
    df['OpenLowDifferenceSum'] = df['OpenLowDifference'].rolling(window=n_days).sum()
    
    # Calculate Intraday Price Volatility
    intraday_prices = df[['open', 'high', 'low', 'close']]
    df['IntradayVolatility'] = intraday_prices.rolling(window=p_days).std().mean(axis=1)
    
    # Compute Average Intraday Price Volatility
    df['AvgIntradayVolatility'] = df['IntradayVolatility'].rolling(window=n_days).mean()
    
    # Compute Price Volatility Score
    df['PriceVolatilityScore'] = (df['IntradayVolatility'] - df['AvgIntradayVolatility']) * df['volume']
    
    # Incorporate Open Price Movement Factor
    df['OpenPriceChange'] = df['open'].diff(periods=n_days)
    
    # Include Close to Low Price Difference
    df['CloseLowDifference'] = df['close'] - df['low']
    df['CloseLowDifferenceSum'] = df['CloseLowDifference'].rolling(window=n_days).sum()
    
    # Combine Factors
    combined_factor = (
        0.2 * df['HighLowMomentumScore'] +
        0.2 * df['PriceMomentum'] +
        0.1 * df['VolumeTrend'] +
        0.1 * df['RangeMomentum'] +
        0.1 * df['OpenCloseMovementSum'] +
        0.1 * df['OpenLowDifferenceSum'] +
        0.1 * df['PriceVolatilityScore'] +
        0.1 * df['OpenPriceChange'] +
        0.1 * df['CloseLowDifferenceSum']
    )
    
    return combined_factor
```
