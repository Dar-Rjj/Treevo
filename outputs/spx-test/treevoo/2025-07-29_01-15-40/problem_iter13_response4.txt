```python
import pandas as pd

def heuristics_v2(df, n_days=5, price_momentum_coeff=0.5, volume_trend_coeff=0.5):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['Average_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=n_days).mean()
    
    # Compute High-Low Momentum Score
    df['High_Low_Momentum_Score'] = (df['Intraday_High_Low_Spread'] - df['Average_Intraday_High_Low_Spread']) * df['volume']
    
    # Calculate Simple Moving Average (SMA) of Close
    df['SMA_Close'] = df['close'].rolling(window=n_days).mean()
    
    # Calculate Price Reversal Indicator
    df['Price_Reversal_Indicator'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Integrate Volume into Momentum: Compute Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] + df['open']) / 2 * df['volume'] / df['volume'].rolling(window=n_days).sum()
    
    # Combine SMA and VWAP for Integrated Momentum
    df['Integrated_Momentum'] = (df['SMA_Close'] + df['VWAP']) / 2
    
    # Composite Factor Construction
    df['Composite_Factor'] = df['High_Low_Momentum_Score'] * df['Integrated_Momentum'] * df['Price_Reversal_Indicator']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(n_days).rolling(window=n_days).sum()
    
    # Calculate Volume Trend
    df['Volume_Trend'] = df['volume'].pct_change(n_days).rolling(window=n_days).sum()
    
    # Combine Momentum and Volume Interaction
    df['Weighted_Price_Momentum'] = df['Price_Momentum'] * price_momentum_coeff
    df['Weighted_Volume_Trend'] = df['Volume_Trend'] * volume_trend_coeff
    df['Enhanced_Factor'] = df['Weighted_Price_Momentum'] + df['Weighted_Volume_Trend'] + df['Composite_Factor']
    
    return df['Enhanced_Factor']
```
