```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the volume-weighted price
    vwp = (df['close'] * df['volume']) / df['volume'].sum()
    
    # Calculate the factor by combining the above components
    factor = (close_open_diff / avg_high_low) * vwp
    
    return factor
```
