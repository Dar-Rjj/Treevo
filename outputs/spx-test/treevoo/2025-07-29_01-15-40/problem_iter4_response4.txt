```python
import pandas as pd

def heuristics_v2(df):
    # Compute High Minus Prior High as Upward
    df['upward'] = df['high'] - df['high'].shift(1)
    
    # Compute Low Minus Prior Low as Downward
    df['downward'] = df['low'].shift(1) - df['low']
    
    # If Upward > Downward, Use Upward; Otherwise, Use Downward
    df['net_movement'] = df.apply(lambda row: row['upward'] if row['upward'] > row['downward'] else row['downward'], axis=1)
    
    # Sum Net Movements Over a Period (e.g., 14 days)
    df['cdmi'] = df['net_movement'].rolling(window=14).sum()
    
    return df['cdmi']
```
