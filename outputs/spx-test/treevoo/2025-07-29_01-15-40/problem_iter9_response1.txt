```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']
    
    # Compute High Minus Prior High, Low Minus Prior Low
    df['high_minus_prior_high'] = df['high'] - df['high'].shift(1)
    df['low_minus_prior_low'] = df['low'] - df['low'].shift(1)
    
    # Determine Net Intraday and Directional Movements
    df['upward_movement'] = df[['range', 'spread', 'high_minus_prior_high']].max(axis=1)
    df['downward_movement'] = df[['range', 'spread', 'low_minus_prior_low']].min(axis=1)
    df['net_movement'] = df.apply(lambda row: row['upward_movement'] if row['upward_movement'] > row['downward_movement'] else row['downward_movement'], axis=1)
    
    # Combine Intraday Movements with Net Movements
    df['combined_movement'] = df['range'] + df['spread'] + df['net_movement']
    
    # Adjust for Volume Activity
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['weighted_combined_movement'] = df['combined_movement'] * (1 + df['volume_change'])
    
    # Calculate CIMVDMI
    df['cimvdmi'] = df['weighted_combined_movement'].rolling(window=10).mean()
    
    return df['cimvdmi']
```
