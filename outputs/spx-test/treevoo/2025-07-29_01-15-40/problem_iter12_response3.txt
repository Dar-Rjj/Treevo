```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the range (high - low)
    df['range'] = df['high'] - df['low']
    
    # Calculate the buying pressure (high + close) / 2 - (open + low) / 2
    df['buying_pressure'] = (df['high'] + df['close']) / 2 - (df['open'] + df['low']) / 2
    
    # Calculate the selling pressure (open + low) / 2 - (high + close) / 2
    df['selling_pressure'] = (df['open'] + df['low']) / 2 - (df['high'] + df['close']) / 2
    
    # Calculate the average volume over the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['buying_pressure'] - df['selling_pressure']) * df['close_pct_change'] * (df['volume'] / df['avg_volume_5d'])
    
    return df['alpha_factor'].dropna()
```
