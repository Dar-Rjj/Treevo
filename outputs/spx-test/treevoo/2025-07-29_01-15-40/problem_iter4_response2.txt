```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    range_high_low = df['high'] - df['low']
    spread_open_close = df['close'] - df['open']
    intraday_movement = range_high_low + spread_open_close
    
    # Confirm with Volume Activity
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Increase or decrease the momentum score based on volume change
    momentum_score = intraday_movement.copy()
    momentum_score[volume_change > 0] += volume_change
    momentum_score[volume_change < 0] -= abs(volume_change)
    
    return momentum_score
```
