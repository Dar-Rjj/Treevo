import pandas as pd

def heuristics_v2(df):
    # EMA Crossover
    df['EMA12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['EMA_Crossover'] = df['EMA12'] - df['EMA26']
    
    # LWMA Crossover
    def lwma(series, n):
        weights = pd.Series(range(1, n+1))
        return (weights * series.rolling(n).apply(lambda x: x[::-1], raw=True)).sum() / weights.sum()
    df['LWMA14'] = lwma(df['close'], 14)
    df['LWMA28'] = lwma(df['close'], 28)
    df['LWMA_Crossover'] = df['LWMA14'] - df['LWMA28']
    
    # Rate of Change (ROC) for 10, 50, and 100 periods
    for n in [10, 50, 100]:
        df[f'ROC_{n}'] = df['close'].pct_change(periods=n)
        
    # Momentum
    for n in [10, 50, 100]:
        df[f'Momentum_{n}'] = df['close'] - df['close'].shift(n)
        
    # On-Balance-Volume (OBV)
    df['OBV'] = 0
    c = df.apply(lambda x: 1 if x['close'] > x['close'].shift(1) else -1 if x['close'] < x['close'].shift(1) else 0, axis=1)
    df['OBV'] = df['volume'] * c
    df['OBV'] = df['OBV'].cumsum()
    
    # Volume-Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['high'] + df['low'] + df['close']) / 3
    df['Cum_Vol_Typ_Price'] = (df['TypicalPrice'] * df['volume']).cumsum()
    df['Cum_Volume'] = df['volume'].cumsum()
    df['VWAP'] = df['Cum_Vol_Typ_Price'] / df['Cum_Volume']
    df['VWAP_Factor'] = df['close'] - df['VWAP']
    
    # Chaikin Money Flow (CMF)
    df['MoneyFlowMultiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['MoneyFlowVolume'] = df['MoneyFlowMultiplier'] * df['volume']
    df['20day_CMFL'] = df['MoneyFlowVolume'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['CMF'] = df['20day_CMFL']
    
    # ATR over a period
    df['TR'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()

    df['SMA20'] = df['close'].rolling(window=20).mean()

    df['BBandLower'] = df['SMA20'] - 2 * df['StdDev20']
    df['BBandFactor'] = (df['close'] - df['SMA20']) / (df['BBandUpper'] - df['BBandLower'])
