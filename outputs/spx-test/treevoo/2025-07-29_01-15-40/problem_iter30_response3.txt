def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
        data (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
        
    Returns:
        pd.Series: Alpha factor values indexed by date
    """
    # Calculate the range of the day
    daily_range = data['high'] - data['low']
    
    # Calculate the percentage change from open to close
    pct_change_open_close = (data['close'] - data['open']) / data['open']
    
    # Calculate the weighted average price
    weighted_avg_price = (data['high'] + data['low'] + 2 * data['close']) / 4
    
    # Calculate the difference between the weighted average price and the close price
    diff_weighted_avg_close = weighted_avg_price - data['close']
    
    # Combine the above factors into a single alpha factor
    alpha_factor = (pct_change_open_close * daily_range) + diff_weighted_avg_close
    
    return alpha_factor
