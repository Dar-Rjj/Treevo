```python
import pandas as pd

def heuristics_v2(df, n_days=5):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate True Range
    df['true_range'] = df.apply(
        lambda x: max(
            abs(x['high'] - x['low']),
            abs(x['high'] - df['close'].shift(1).loc[x.name]),
            abs(x['low'] - df['close'].shift(1).loc[x.name])
        ),
        axis=1
    )
    
    # Compute Intraday Volatility Adjusted Return
    df['volatility_adjusted_return'] = df['intraday_return'] / df['true_range']
    
    # Cumulative Momentum Over Multiple Days
    df['cumulative_momentum'] = df['volatility_adjusted_return'].rolling(window=n_days).sum()
    
    return df['cumulative_momentum'].dropna()
```
