```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    avg_intraday_high_low_spread = df['intraday_high_low_spread'].rolling(window=5).mean()
    high_low_momentum_score = (df['intraday_high_low_spread'] > avg_intraday_high_low_spread).astype(int) * 2 - 1
    high_low_momentum_score_adjusted = high_low_momentum_score * df['volume']

    # Calculate Price Momentum
    price_change = df['close'].diff(periods=5)
    price_momentum = price_change.rolling(window=5).sum()

    # Calculate Volume Trend
    volume_change = df['volume'].diff(periods=5)
    volume_trend = volume_change.rolling(window=5).sum()

    # Enhance Market Sentiment Analysis
    # Calculate Range Momentum
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=5).mean()
    range_momentum_score = (daily_range > avg_daily_range).astype(int) * 2 - 1

    # Integrate Open-Price and Close-Price Movement
    open_close_price_diff = df['close'] - df['open']
    open_close_price_movement = open_close_price_diff.rolling(window=5).sum()

    # Combine Factors with weighted sum
    factor = (
        0.3 * high_low_momentum_score_adjusted + 
        0.3 * price_momentum + 
        0.2 * volume_trend + 
        0.1 * range_momentum_score + 
        0.1 * open_close_price_movement
    )

    return factor
```
