```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']

    # Determine Net Intraday and Directional Movements
    df['upward'] = df['high'] - df['close'].shift(1)
    df['downward'] = df['low'] - df['close'].shift(1)
    df['net_movement'] = np.where(df['upward'] > df['downward'], df['upward'], df['downward'])
    df['combined_momentum'] = (df['range'] + df['spread'] + df['net_movement']) * df['volume']

    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['combined_momentum'] *= df['volume_change']

    # Calculate Intraday High-Low Spread
    df['intraday_hl_spread'] = df['high'] - df['low']
    df['avg_intraday_hl_spread'] = df['intraday_hl_spread'].rolling(window=21).mean()

    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(df['intraday_hl_spread'] > df['avg_intraday_hl_spread'], 1, -1) * df['volume']

    # Calculate Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))

    # EWMA of Log Return
    df['ewma_log_return'] = df['log_return'].ewm(span=21).mean()

    # Integrate Price and Volume
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['integrated_momentum'] = df['ewma_log_return'] * df['volume_change']

    # Calculate Upward and Downward Movements
    df['upward'] = df['high'] - df['close'].shift(1)
    df['downward'] = df['low'] - df['close'].shift(1)

    # Determine Net Directional Movement
    df['net_directional_movement'] = np.where(df['upward'] > df['downward'], 1, np.where(df['upward'] < df['downward'], -1, 0))

    # Calculate Adjusted Cumulative Directional Movement
    df['adjusted_cumulative_directional_movement'] = df['net_directional_movement'].rolling(window=21).sum() * df['volume']

    # Add Trend Strength Indicator
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=21).mean()
    df['trend_strength_indicator'] = np.where(df['daily_range'] > df['avg_daily_range'], 1, 0)

    # Integrate Volume into Momentum
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['sma_close'] = df['close'].rolling(window=21).mean()
    df['integrated_vwap_sma'] = (df['sma_close'] + df['vwap']) / 2

    # Composite Factor Construction
    df['composite_factor'] = (df['high_low_momentum_score'] * df['integrated_momentum'] * 
                              (1 - df['net_directional_movement']) * 
                              (df['combined_momentum'] + df['adjusted_cumulative_directional_movement']) * 
                              df['trend_strength_indicator'])

    # Add a Volatility Component
    df['volatility'] = df['log_return'].rolling(window=21).std()
    df['final_factor'] = df['composite_factor'] / df['volatility']

    return df['final_factor'].dropna()
```
