```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to open price
    close_to_open = df['close'] / df['open']
    
    # Calculate the ratio of high to low price
    high_to_low = df['high'] / df['low']
    
    # Calculate the relative change in volume
    volume_change = df['volume'].pct_change()
    
    # Combine the three ratios into a single alpha factor
    # The intuition is that a higher close-to-open and high-to-low ratio
    # along with an increase in volume could indicate a positive sentiment
    alpha_factor = (close_to_open + high_to_low) * (1 + volume_change)
    
    return alpha_factor
```
