```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Sub-Thought 1: High-Low Price Momentum with Volume-Weighted Component
    df['high_low_ratio'] = df['high'] / df['low']
    df['volume_weighted_high_low'] = (df['high_low_ratio'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['smoothed_high_low'] = df['volume_weighted_high_low'].ewm(alpha=0.2).mean()

    # Sub-Thought 2: Simple Moving Averages and Momentum
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_momentum'] = df['sma_5'] - df['sma_20']
    df['adjusted_high_low'] = df['high_low_ratio'] + df['sma_momentum']

    # Sub-Thought 3: Price Volatility Analysis
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)), abs(x['low'] - df['close'].shift(1))), axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    df['atr_7'] = df['true_range'].rolling(window=7).mean()
    df['volatility_ratio'] = df['atr_7'] / df['atr_14']
    df['adjusted_volatility_high_low'] = df['high_low_ratio'] * df['volatility_ratio']

    # Sub-Thought 4: Evaluate Volume Changes for Market Sentiment
    df['volume_change'] = df['volume'].pct_change()
    df['volume_trend'] = df['volume_change'].rolling(window=5).mean()
    df['volume_adjusted_high_low'] = df['high_low_ratio'] * (1 + df['volume_trend'])

    # Sub-Thought 5: Trading Range Analysis
    df['daily_trading_range'] = df['high'] - df['low']
    df['avg_trading_range_10'] = df['daily_trading_range'].rolling(window=10).mean()
    df['trading_range_signal'] = (df['daily_trading_range'] > df['avg_trading_range_10']).astype(int)
    df['adjusted_trading_range_high_low'] = df['high_low_ratio'] * (1 + df['trading_range_signal'])

    # Sub-Thought 6: Open and Close Prices Impact on Next Day's Return
    df['overnight_return'] = df['open'].pct_change().shift(-1)
    df['intraday_return'] = (df['close'] / df['open'] - 1) * 100
    df['combined_return'] = df['overnight_return'] + df['intraday_return']
    df['return_adjusted_high_low'] = df['high_low_ratio'] * (1 + df['combined_return'])

    # Final Alpha Factor
    df['final_factor'] = (
        df['smoothed_high_low'] * 
        df['adjusted_high_low'] * 
        df['adjusted_volatility_high_low'] * 
        df['volume_adjusted_high_low'] * 
        df['adjusted_trading_range_high_low'] * 
        df['return_adjusted_high_low']
    )

    return df['final_factor']
```
