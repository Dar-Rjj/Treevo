```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    pos_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    neg_money_flow = -money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the money ratio
    money_ratio = pos_money_flow.rolling(window=14).sum() / neg_money_flow.rolling(window=14).sum()
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_ratio))
    
    # Calculate the average true range
    true_range = df[['high' - 'low', abs('high' - df['close'].shift(1)), abs('low' - df['close'].shift(1))]].max(axis=1)
    average_true_range = true_range.rolling(window=14).mean()
    
    # Calculate the factor
    factor = money_flow_index / average_true_range
    
    return factor
```
