```python
import pandas as pd

def heuristics_v2(df, N=5, M=5, hl_coeff=0.3, price_momentum_coeff=0.2, volume_trend_coeff=0.1, range_momentum_coeff=0.1, open_close_diff_coeff=0.15, open_low_diff_coeff=0.15):
    # Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Average Intraday High-Low Spread
    df['Avg_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=N).mean()
    
    # High-Low Momentum Score
    df['High_Low_Momentum_Score'] = (df['High_Low_Spread'] > df['Avg_High_Low_Spread']).astype(int) * 2 - 1
    df['High_Low_Momentum_Score_Adjusted'] = df['High_Low_Momentum_Score'] * df['volume']
    
    # Price Momentum
    df['Price_Change'] = df['close'].pct_change()
    df['Price_Momentum'] = df['Price_Change'].rolling(window=N).sum()
    
    # Volume Trend
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=N).sum()
    
    # Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Average Daily Range
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=M).mean()
    
    # Range Momentum
    df['Range_Momentum'] = (df['Daily_Range'] > df['Avg_Daily_Range']).astype(int) * 2 - 1
    
    # Open to Close Price Difference
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Open_Close_Diff_Sum'] = df['Open_Close_Diff'].rolling(window=N).sum()
    
    # Open to Low Price Difference
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['Open_Low_Diff_Sum'] = df['Open_Low_Diff'].rolling(window=N).sum()
    
    # Combine Factors
    df['Factor'] = (hl_coeff * df['High_Low_Momentum_Score_Adjusted'] +
                    price_momentum_coeff * df['Price_Momentum'] +
                    volume_trend_coeff * df['Volume_Trend'] +
                    range_momentum_coeff * df['Range_Momentum'] +
                    open_close_diff_coeff * df['Open_Close_Diff_Sum'] +
                    open_low_diff_coeff * df['Open_Low_Diff_Sum'])
    
    return df['Factor'].dropna()
```
