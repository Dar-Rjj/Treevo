import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']
    
    # Determine Net Intraday and Directional Movements
    df['net_movement'] = df[['range', 'spread']].max(axis=1)
    df['net_movement_weighted'] = df['net_movement'] * df['volume'].rolling(window=5).mean()
    
    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['combined_momentum'] = df['net_movement_weighted'] * df['volume_change']
    
    # Calculate Intraday High-Low Spread
    df['hl_spread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread
    df['avg_hl_spread'] = df['hl_spread'].rolling(window=5).mean()
    
    # Compute High-Low Momentum Score
    df['hl_momentum_score'] = (df['hl_spread'] - df['avg_hl_spread']) * df['volume']
    
    # Calculate Log Return
    df['log_return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    
    # EWMA of Log Return
    df['ewma_log_return'] = df['log_return'].ewm(span=21).mean()
    
    # Integrate Price and Volume
    df['combined_momentum_ewma'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    df['net_directional_movement'] = df.apply(
        lambda row: 1 if row['upward_movement'] > row['downward_movement'] else (-1 if row['upward_movement'] < row['downward_movement'] else 0),
        axis=1
    )
    
    # Calculate Adjusted Cumulative Directional Movement
    df['cumulative_directional_movement'] = df['net_directional_movement'].rolling(window=21).sum() * df['volume']
    
    # Add Trend Strength Indicator
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=21).mean()
    df['trend_strength'] = (df['daily_range'] > df['avg_daily_range']).astype(int)
    
    # Integrate Volume into Momentum
    df['vwap'] = (df['close'] + df['open']) / 2 * df['volume']
    df['vwap'] = df['vwap'].rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Composite Factor Construction
    df['composite_factor'] = (
        df['hl_momentum_score'] * df['combined_momentum_ewma'] *
        df['trend_strength'] * df['cumulative_directional_movement']
    )
    
    # Add a Volatility Component
    df['std_dev_log_return'] = df['log_return'].rolling(window=21).std()
    df['final_factor'] = df['composite_factor'] * (1 / df['std_dev_log_return'])
    
    return df['final_factor']
