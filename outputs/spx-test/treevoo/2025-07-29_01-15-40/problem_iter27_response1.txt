```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['IntradaySpread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    high_low_moving_avg = df['IntradaySpread'].rolling(window=5).mean()
    df['HighLowMomentumScore'] = np.where(df['IntradaySpread'] > high_low_moving_avg, 1, -1)
    df['HighLowMomentumScore'] *= np.log(df['volume'])
    
    # Calculate Price Momentum over 5 days
    price_change = df['close'] - df['close'].shift(5)
    df['PriceMomentum'] = price_change.rolling(window=5).sum()
    
    # Calculate Volume Trend over 5 days
    volume_change = df['volume'] - df['volume'].shift(5)
    df['VolumeTrend'] = volume_change.rolling(window=5).sum()
    
    # Enhance Market Sentiment Analysis
    daily_range = df['high'] - df['low']
    range_moving_avg = daily_range.rolling(window=5).mean()
    df['RangeMomentum'] = np.where(daily_range > range_moving_avg, 1, -1)

    open_close_diff = df['close'] - df['open']
    adjusted_open_close_diff = open_close_diff / df['volume']
    df['AdjustedOpenCloseMovement'] = adjusted_open_close_diff.rolling(window=5).sum()

    # Combine Factors with Weighted Sum
    factor = (df['HighLowMomentumScore'] * 0.2 +
              df['PriceMomentum'] * 0.3 +
              df['VolumeTrend'] * 0.1 +
              df['RangeMomentum'] * 0.2 +
              df['AdjustedOpenCloseMovement'] * 0.2)
    
    return factor
```
