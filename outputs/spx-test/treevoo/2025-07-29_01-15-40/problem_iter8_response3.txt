```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios
    rolling_window = 10
    df['Rolling_Avg_High_Low_Ratio'] = df['High_Low_Ratio'].rolling(window=rolling_window).mean()
    
    # Price Trend Component
    ma_10 = df['close'].rolling(window=10).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    df['Price_Trend'] = ma_10 - ma_50
    
    # Volume Trend Component
    sum_vol_10 = df['volume'].rolling(window=10).sum()
    sum_vol_50 = df['volume'].rolling(window=50).sum()
    df['Volume_Trend'] = sum_vol_10 - sum_vol_50
    
    # Combined Indicator
    combined_indicator = df['Price_Trend'] * df['Volume_Trend']
    
    # Adjust for Volatility
    std_20 = df['close'].rolling(window=20).std()
    df['Adjusted_Combined_Indicator'] = combined_indicator / std_20
    
    # Synthesize Indicators
    weights = {'Rolling_Avg_High_Low_Ratio': 0.5, 'Adjusted_Combined_Indicator': 0.5}
    df['Synthesized_Indicator'] = (weights['Rolling_Avg_High_Low_Ratio'] * df['Rolling_Avg_High_Low_Ratio'] + 
                                   weights['Adjusted_Combined_Indicator'] * df['Adjusted_Combined_Indicator'])
    
    # Calculate 5-Day Price Momentum
    df['5_Day_Momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    df['20_Day_Vol_MA'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    df['Volume_Deviation'] = df['volume'] - df['20_Day_Vol_MA']
    
    # Calculate 5-Day High-Low Spread
    df['5_Day_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['Open_Close_Spread'] = df['open'] - df['close']
    
    # Calculate Price Reversal Indicator
    df['Price_Reversal_Indicator'] = df['5_Day_High_Low_Spread'] - df['Open_Close_Spread']
    
    # Compute 5-Day Price Range and Volume Synchronicity
    df['5_Day_Price_Range_Sync'] = df['5_Day_High_Low_Spread'] * df['Volume_Deviation']
    
    # Compute 5-Day Momentum and Volume Synchronicity
    df['5_Day_Momentum_Sync'] = df['5_Day_Momentum'] * df['Volume_Deviation']
    
    # Combine Synchronicities
    df['Combined_Synchronicities'] = df['5_Day_Price_Range_Sync'] + df['5_Day_Momentum_Sync']
    
    # Apply Exponential Smoothing
    alpha = 0.2
    df['Exponentially_Smoothed_Sync'] = df['Combined_Synchronicities'].ewm(alpha=alpha).mean()
    
    # Integrated Momentum and Reversal Factor
    df['Final_Factor'] = df['Exponentially_Smoothed_Sync'] * df['Price_Reversal_Indicator']
    
    return df['Final_Factor']
```
