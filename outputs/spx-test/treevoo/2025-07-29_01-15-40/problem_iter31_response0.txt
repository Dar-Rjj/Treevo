```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    avg_spread = df['Intraday_High_Low_Spread'].rolling(window=5).mean()
    df['High_Low_Momentum_Score'] = (df['Intraday_High_Low_Spread'] - avg_spread).apply(lambda x: 1 if x > 0 else -1)
    df['High_Low_Momentum_Score_Adjusted'] = df['High_Low_Momentum_Score'] * df['volume']

    # Calculate Price Momentum
    period = 5
    df['Price_Change'] = df['close'].diff(period)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=period).sum()

    # Calculate Volume Trend
    df['Volume_Change'] = df['volume'].diff(period)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=period).sum()

    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    avg_range = df['Daily_Range'].rolling(window=5).mean()
    df['Range_Momentum'] = (df['Daily_Range'] - avg_range).apply(lambda x: 1 if x > 0 else -1)
    
    df['Open_Close_Price_Diff'] = df['close'] - df['open']
    df['Open_Price_Movement'] = df['Open_Close_Price_Diff'].rolling(window=period).sum()

    # Calculate Intraday Price Volatility
    df['Intraday_Volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1, ddof=0)
    avg_volatility = df['Intraday_Volatility'].rolling(window=period).mean()
    df['Volatility_Score'] = (df['Intraday_Volatility'] - avg_volatility).apply(lambda x: 1 if x > 0 else -1)
    df['Volatility_Score_Adjusted'] = df['Volatility_Score'] * df['volume']

    # Combine Factors
    alpha_factor = (
        0.3 * df['High_Low_Momentum_Score_Adjusted'] +
        0.2 * df['Price_Momentum'] +
        0.15 * df['Volume_Trend'] +
        0.1 * df['Range_Momentum'] +
        0.1 * df['Open_Price_Movement'] +
        0.15 * df['Volatility_Score_Adjusted']
    )

    return alpha_factor
```
