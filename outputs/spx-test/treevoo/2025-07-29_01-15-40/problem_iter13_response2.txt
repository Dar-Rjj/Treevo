```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['HighLowRatio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios
    window = 10
    df['RollingHighLowAvg'] = df['HighLowRatio'].rolling(window=window).mean()
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    
    # Calculate True Range
    df['TrueRange'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], 
                                                               abs(x['high'] - df['close'].shift(1)), 
                                                               abs(x['low'] - df['close'].shift(1))), axis=1)
    
    # Compute Intraday Volatility Adjusted Return
    df['IntradayVolAdjReturn'] = df['IntradayReturn'] / df['TrueRange']
    
    # Integrate Price and Volume Information
    df['VolumeChange'] = df['volume'] / df['volume'].shift(1)
    df['IntegratedSignal'] = df['IntradayVolAdjReturn'] * df['VolumeChange']
    
    # Combine Signals
    df['CombinedMomentum'] = df['RollingHighLowAvg'] * df['IntegratedSignal']
    
    return df['CombinedMomentum']
```
