```python
import pandas as pd

def heuristics_v2(df):
    # Price Trend Component
    ma_10 = df['close'].rolling(window=10).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    price_trend = ma_10 - ma_50
    
    # Volume Trend Component
    vol_10 = df['volume'].rolling(window=10).sum()
    vol_50 = df['volume'].rolling(window=50).sum()
    volume_trend = vol_10 - vol_50
    
    # Combined Indicator
    combined_indicator = price_trend * volume_trend
    
    # Adjust for Volatility
    sd_20 = df['close'].rolling(window=20).std()
    adjusted_indicator = combined_indicator / sd_20
    
    return adjusted_indicator
```
