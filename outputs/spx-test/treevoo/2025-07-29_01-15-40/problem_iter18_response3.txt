```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, p=10):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # Compute Average Intraday High-Low Spread over past N days
    df['Avg_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=n).mean()

    # Compute High-Low Momentum Score
    df['High_Low_Momentum_Score'] = (df['Intraday_High_Low_Spread'] - df['Avg_Intraday_High_Low_Spread']).apply(lambda x: 1 if x > 0 else -1)
    df['High_Low_Momentum_Score'] *= df['volume']

    # Calculate Price Momentum over the past 5 days
    df['Price_Momentum'] = df['close'] - df['close'].shift(n)

    # Calculate Volume Trend over the past 5 days
    df['Volume_Trend'] = df['volume'] - df['volume'].shift(n)

    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=m).mean()
    df['Range_Momentum_Score'] = (df['Daily_Range'] - df['Avg_Daily_Range']).apply(lambda x: 1 if x > 0 else -1)

    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Price_Difference'] = df['close'] - df['open']
    df['Open_Close_Price_Trend'] = df['Open_Close_Price_Difference'].rolling(window=n).sum()

    # Intraday Price Volatility
    df['Intraday_Price_Volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1).rolling(window=p).mean()
    df['Avg_Intraday_Price_Volatility'] = df['Intraday_Price_Volatility'].rolling(window=p).mean()
    df['Price_Volatility_Score'] = (df['Intraday_Price_Volatility'] - df['Avg_Intraday_Price_Volatility']).apply(lambda x: 1 if x > 0 else -1)
    df['Price_Volatility_Score'] *= df['volume']

    # Combine Factors
    df['Alpha_Factor'] = (
        0.3 * df['High_Low_Momentum_Score'] +
        0.2 * df['Price_Momentum'] +
        0.2 * df['Volume_Trend'] +
        0.1 * df['Range_Momentum_Score'] +
        0.1 * df['Open_Close_Price_Trend'] +
        0.1 * df['Price_Volatility_Score']
    )

    return df['Alpha_Factor']
```
