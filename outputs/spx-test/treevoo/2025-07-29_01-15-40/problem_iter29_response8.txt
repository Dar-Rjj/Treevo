```python
import pandas as pd

def heuristics_v2(df, N=10, M=10, P=10):
    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Average Intraday High-Low Spread
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=N).mean()
    
    # High-Low Momentum Score
    df['high_low_momentum_score'] = (df['intraday_high_low_spread'] - df['avg_intraday_high_low_spread']).apply(lambda x: 1 if x > 0 else -1)
    df['high_low_momentum_score'] *= df['volume']
    
    # Price Momentum
    df['price_change'] = df['close'].diff(5)
    df['price_momentum'] = df['price_change'].rolling(window=5).sum()
    
    # Volume Trend
    df['volume_change'] = df['volume'].diff(5)
    df['volume_trend'] = df['volume_change'].rolling(window=5).sum()
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=M).mean()
    df['range_momentum'] = (df['daily_range'] - df['avg_daily_range']).apply(lambda x: 1 if x > 0 else -1)
    df['open_close_price_diff'] = df['close'] - df['open']
    df['open_price_movement'] = df['open_close_price_diff'].rolling(window=5).sum()
    
    # Intraday Price Volatility
    df['intraday_prices'] = df[['open', 'high', 'low', 'close']].mean(axis=1)
    df['intraday_volatility'] = df['intraday_prices'].rolling(window=P).std()
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=N).mean()
    df['price_volatility_score'] = (df['intraday_volatility'] - df['avg_intraday_volatility']).apply(lambda x: 1 if x > 0 else -1)
    df['price_volatility_score'] *= df['volume']
    
    # Combine Factors
    alpha_factor = (0.2 * df['high_low_momentum_score'] + 
                    0.2 * df['price_momentum'] + 
                    0.2 * df['volume_trend'] + 
                    0.2 * df['range_momentum'] + 
                    0.1 * df['open_price_movement'] + 
                    0.1 * df['price_volatility_score'])
    
    return alpha_factor
```
