def heuristics_v2(df):
    """
    Generate an alpha factor based on price and volume data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series with the generated alpha factor values, indexed by date.
    """
    # Calculate the intraday return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate the intraday range
    intraday_range = (df['high'] - df['low']) / df['low']
    
    # Calculate the volume weighted intraday return
    volume_weighted_return = intraday_return * (df['volume'] / df['volume'].rolling(window=5).mean())
    
    # Calculate the final alpha factor
    alpha_factor = volume_weighted_return / intraday_range
    
    return alpha_factor
