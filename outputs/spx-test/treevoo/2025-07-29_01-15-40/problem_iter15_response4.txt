import pandas as pd

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = (df['close'] / df['close'].shift(1)).apply(lambda x: 0 if x <= 0 else math.log(x))
    
    # EWMA of Log Return with 21-day Lookback Period
    df['ewma_log_return'] = df['log_return'].ewm(span=21, adjust=False).mean()
    
    # Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combined Momentum
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Rolling Average of High-Low Ratios with 10-day window
    df['rolling_high_low_ratio'] = df['high_low_ratio'].rolling(window=10).mean()
    
    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # True Range
    df['true_range'] = df[['high', 'low', df['close'].shift(1)]].max(axis=1) - df[['high', 'low', df['close'].shift(1)]].min(axis=1)
    
    # Intraday Volatility Adjusted Return
    df['intraday_vol_adj_return'] = df['intraday_return'] / df['true_range']
    
    # Volume Relative Change
    df['volume_relative_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Integrated Intraday Signal
    df['integrated_intraday_signal'] = df['intraday_vol_adj_return'] * df['volume_relative_change']
    
    # Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Rolling Average of Daily Ranges with 10-day window
    df['rolling_daily_range'] = df['daily_range'].rolling(window=10).mean()
    
    # Combine Signals
    df['combined_intraday_signal'] = df['rolling_high_low_ratio'] * df['integrated_intraday_signal']
    df['combined_momentum_signal'] = df['rolling_daily_range'] * df['combined_momentum']
    
    # Integrated Factor
    df['integrated_factor'] = df['combined_momentum_signal'] + df['combined_intraday_signal']
    
    return df['integrated_factor']
