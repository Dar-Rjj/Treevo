```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    avg_intraday_high_low_spread = df['intraday_high_low_spread'].rolling(window=5).mean()
    high_low_momentum_score = (df['intraday_high_low_spread'] > avg_intraday_high_low_spread).astype(int) * 2 - 1
    df['high_low_momentum_score'] = high_low_momentum_score * df['volume']
    
    # Calculate Price Momentum
    period = 5
    df['price_change'] = df['close'] - df['close'].shift(period)
    df['price_momentum'] = df['price_change'].rolling(window=period).sum()
    
    # Calculate Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(period)
    df['volume_trend'] = df['volume_change'].rolling(window=period).sum()
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    avg_daily_range = df['daily_range'].rolling(window=5).mean()
    range_momentum = (df['daily_range'] > avg_daily_range).astype(int) * 2 - 1
    df['range_momentum'] = range_momentum
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_price_movement'] = df['open_close_diff'].rolling(window=period).sum()
    
    # Calculate Intraday Price Volatility
    intraday_prices = df[['open', 'high', 'low', 'close']]
    df['intraday_volatility'] = intraday_prices.std(axis=1).rolling(window=period).mean()
    
    # Compute Price Volatility Score
    avg_intraday_volatility = df['intraday_volatility'].rolling(window=5).mean()
    price_volatility_score = (df['intraday_volatility'] > avg_intraday_volatility).astype(int) * 2 - 1
    df['price_volatility_score'] = price_volatility_score * df['volume']
    
    # Combine Factors
    alpha_factor = (
        0.2 * df['high_low_momentum_score'] +
        0.3 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.1 * df['range_momentum'] +
        0.2 * df['open_price_movement'] +
        0.1 * df['price_volatility_score']
    )
    
    return alpha_factor
```
