```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    volume_weighted_mean = (high_low_ratio.rolling(window=rolling_window).sum() * df['volume']) / df['volume'].rolling(window=rolling_window).sum()
    
    # Apply Exponential Smoothing to the Volume-Weighted Mean
    smoothing_factor = 0.2
    smoothed_volume_weighted_mean = volume_weighted_mean.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Incorporate Volatility Adjustment
    rolling_std = df['close'].rolling(window=rolling_window).std()
    volatility_adjusted_factor = smoothed_volume_weighted_mean / rolling_std
    
    return volatility_adjusted_factor
```
