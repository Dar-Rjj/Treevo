```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    df['High_Low_Momentum_Score'] = (df['Intraday_High_Low_Spread'] > df['Intraday_High_Low_Spread'].shift(1)).astype(int) * 2 - 1
    df['High_Low_Momentum_Score_Adjusted'] = df['High_Low_Momentum_Score'] * df['volume']
    
    # Calculate Price Momentum over 5 days
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=5).sum()
    
    # Calculate Volume Trend over 5 days
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=5).sum()
    
    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    df['Range_Momentum'] = (df['Daily_Range'] > df['Daily_Range'].shift(1)).astype(int) * 2 - 1
    
    df['Open_Close_Price_Difference'] = df['close'] - df['open']
    
    # Combine Factors with weighted sum
    alpha_factor = (
        0.3 * df['High_Low_Momentum_Score_Adjusted'] +
        0.25 * df['Price_Momentum'] +
        0.2 * df['Volume_Trend'] +
        0.15 * df['Range_Momentum'] +
        0.1 * df['Open_Close_Price_Difference']
    )
    
    return alpha_factor
```
