```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the buying pressure
    df['buying_pressure'] = (2 * df['close'] - df['low'] - df['high']) / df['range']
    
    # Calculate the average volume over the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the relative volume
    df['relative_volume'] = df['volume'] / df['avg_volume_5d']
    
    # Calculate the heuristic factor
    df['heuristic_factor'] = df['buying_pressure'] * df['relative_volume']
    
    return df['heuristic_factor']
```
