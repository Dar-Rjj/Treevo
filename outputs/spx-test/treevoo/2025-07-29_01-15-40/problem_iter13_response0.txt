```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame, period: int = 5, hl_coeff: float = 0.3, price_coeff: float = 0.4, volume_coeff: float = 0.2, sentiment_coeff: float = 0.1) -> pd.Series:
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    avg_intraday_high_low_spread = intraday_high_low_spread.rolling(window=period).mean()
    high_low_momentum_score = (intraday_high_low_spread - avg_intraday_high_low_spread).apply(lambda x: 1 if x > 0 else -1)
    high_low_momentum_score_adj = high_low_momentum_score * df['volume']
    
    # Calculate Price Momentum
    price_changes = df['close'].diff(periods=period)
    price_momentum = price_changes.rolling(window=period).sum()
    
    # Calculate Volume Trend
    volume_changes = df['volume'].diff(periods=period)
    volume_trend = volume_changes.rolling(window=period).sum()
    
    # Enhance Market Sentiment Analysis
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=period).mean()
    range_momentum_score = (daily_range - avg_daily_range).apply(lambda x: 1 if x > 0 else -1)
    open_close_movement = (df['close'] - df['open']).rolling(window=period).sum()
    sentiment_analysis = range_momentum_score + open_close_movement
    
    # Combine High-Low Momentum, Price Momentum, Volume Trend, and Enhanced Market Sentiment
    factor_values = (hl_coeff * high_low_momentum_score_adj +
                     price_coeff * price_momentum +
                     volume_coeff * volume_trend +
                     sentiment_coeff * sentiment_analysis)
    
    return factor_values
```
