```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    momentum_5d = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    volume_ma_20d = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    volume_deviation = df['volume'] - volume_ma_20d
    
    # Combine Momentum and Volume Synchronicity
    synchronicity = momentum_5d * volume_deviation
    
    # Apply Exponential Smoothing
    alpha = 0.1  # Smoothing factor
    smoothed_synchronicity = synchronicity.ewm(alpha=alpha, adjust=False).mean()
    
    # Incorporate High-Low Range for Market Sentiment
    high_low_range = df['high'] - df['low']
    market_sentiment = high_low_range / df['close']
    
    # Calculate Final Factor
    final_factor = (smoothed_synchronicity + market_sentiment) * volume_deviation
    
    return final_factor
```
