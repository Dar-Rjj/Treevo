```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']
    
    # Compute High Minus Prior High and Low Minus Prior Low
    df['high_minus_prior_high'] = df['high'] - df['high'].shift(1)
    df['low_minus_prior_low'] = df['low'] - df['low'].shift(1)
    
    # Determine Net Intraday and Directional Movements
    df['net_movement'] = 0
    df.loc[df['high_minus_prior_high'] > df['low_minus_prior_low'], 'net_movement'] = df['high_minus_prior_high']
    df.loc[df['high_minus_prior_high'] <= df['low_minus_prior_low'], 'net_movement'] = df['low_minus_prior_low']
    
    # Combine Intraday Movements with Net Movements, weighted by average volume
    df['combined_momentum'] = (df['range'] + df['spread'] + df['net_movement']) * df['volume'].rolling(window=2).mean()
    
    # Adjust for Volume Activity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['volume_ratio']
    
    # Calculate CIMVDMI
    df['CIMVDMI'] = df['adjusted_combined_momentum'].cumsum()
    
    return df['CIMVDMI']
```
