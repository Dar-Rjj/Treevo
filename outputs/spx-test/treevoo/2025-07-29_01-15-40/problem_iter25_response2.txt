```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Alpha Factor 1: Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Alpha Factor 2: Close Position in Range
    df['close_position_in_range'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Alpha Factor 3: Volume to Amount Ratio
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Alpha Factor 4: Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Alpha Factor 5: Price Momentum (over N days, let's say N=5 for this example)
    N = 5
    df['price_momentum'] = (df['close'] - df['close'].shift(N)) / df['close'].shift(N)
    
    # Helper function to calculate EMA
    def ema(series, window):
        return series.ewm(span=window, adjust=False).mean()
    
    # Alpha Factor 6: MACD Line and Signal Line
    df['macd_line'] = ema(df['close'], 12) - ema(df['close'], 26)
    df['signal_line'] = ema(df['macd_line'], 9)
    
    # Alpha Factor 7: MACD Histogram
    df['macd_histogram'] = df['macd_line'] - df['signal_line']
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        df['daily_range'] +
        df['close_position_in_range'] +
        df['volume_to_amount_ratio'] +
        df['volume_change'] +
        df['price_momentum'] +
        df['macd_histogram']
    )
    
    return alpha_factor
```
