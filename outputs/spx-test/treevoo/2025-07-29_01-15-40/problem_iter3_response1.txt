```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    df['5D_Momentum'] = df['close'] - df['close'].shift(5)

    # Calculate 20-Day Volume Moving Average
    df['20D_Vol_MA'] = df['volume'].rolling(window=20).mean()

    # Compute Volume Deviation from MA
    df['Vol_Dev'] = df['volume'] - df['20D_Vol_MA']

    # Calculate Momentum and Volume Synchronicity
    df['Sync'] = df['5D_Momentum'] * df['Vol_Dev']

    # Apply Exponential Smoothing
    df['Smoothed_Sync'] = df['Sync'].ewm(span=5, adjust=False).mean()

    # Introduce Price Volatility Factor
    df['10D_Price_Vol'] = df['close'].rolling(window=10).std()
    df['Adjusted_Sync'] = df['Smoothed_Sync'] / df['10D_Price_Vol']

    # Incorporate High-Low Price Range
    df['HL_Range'] = df['high'] - df['low']
    df['5D_Avg_HL_Range'] = df['HL_Range'].rolling(window=5).mean()

    # Final Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_Sync'] * df['5D_Avg_HL_Range']

    return df['Alpha_Factor']
```
