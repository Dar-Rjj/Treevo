```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, p=10):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score and Adjust by Volume
    df['high_low_momentum_score'] = (df['high_low_spread'] - df['avg_high_low_spread']).apply(lambda x: 1 if x > 0 else -1) * df['volume']
    
    # Calculate Price Momentum over the past N days
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Calculate Volume Trend over the past N days
    df['volume_change'] = df['volume'] - df['volume'].shift(n)
    df['volume_trend'] = df['volume_change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    
    # Calculate Range Momentum
    df['range_momentum'] = (df['daily_range'] - df['avg_daily_range']).apply(lambda x: 1 if x > 0 else -1)
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_momentum'] = df['open_close_diff'].rolling(window=n).sum()
    
    # Calculate Open to Low Price Difference
    df['open_low_diff'] = df['open'] - df['low']
    df['open_low_momentum'] = df['open_low_diff'].rolling(window=n).sum()
    
    # Calculate Intraday Price Volatility over a rolling window of P days
    df['intraday_prices'] = (df[['open', 'high', 'low', 'close']].mean(axis=1))
    df['intraday_volatility'] = df['intraday_prices'].rolling(window=p).std()
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=p).mean()
    
    # Compute Price Volatility Score and Adjust by Volume
    df['price_volatility_score'] = (df['intraday_volatility'] - df['avg_intraday_volatility']).apply(lambda x: 1 if x > 0 else -1) * df['volume']
    
    # Combine Factors using a weighted sum
    df['alpha_factor'] = (
        0.3 * df['high_low_momentum_score'] +
        0.2 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.1 * df['range_momentum'] +
        0.15 * df['open_close_momentum'] +
        0.05 * df['open_low_momentum'] +
        0.1 * df['price_volatility_score']
    )
    
    return df['alpha_factor']
```
