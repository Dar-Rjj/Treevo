```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    n_days = 5
    price_changes = df['close'].pct_change()
    price_momentum = price_changes.rolling(window=n_days).sum()

    # Calculate Volume Trend
    volume_changes = df['volume'].pct_change()
    volume_trend = volume_changes.rolling(window=n_days).sum()

    # Integrate Volume into Momentum
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    integrated_momentum = (price_momentum + vwap) / 2

    # Calculate Price Reversal Indicator
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['open'] - df['close']
    price_reversal = high_low_spread - open_close_spread

    # Combine Price Momentum and Volume Trend
    momentum_coefficient = 0.6
    volume_coefficient = 0.4
    combined_signal = (price_momentum * momentum_coefficient) + (volume_trend * volume_coefficient)

    # Composite Factor Construction
    factor = (combined_signal + integrated_momentum) * price_reversal

    return factor
```
