```python
def heuristics_v2(data):
    # Calculate Price Trend Component
    ten_day_ma = data['close'].rolling(window=10).mean()
    fifty_day_ma = data['close'].rolling(window=50).mean()
    price_trend = fifty_day_ma - ten_day_ma

    # Calculate Volume Trend Component
    ten_day_vol_sum = data['volume'].rolling(window=10).sum()
    fifty_day_vol_sum = data['volume'].rolling(window=50).sum()
    volume_trend = fifty_day_vol_sum - ten_day_vol_sum

    # Combine Price and Volume Components
    combined_trend = price_trend * volume_trend
    
    # Adjust for Volatility
    twenty_day_sd = data['close'].rolling(window=20).std()
    adjusted_trend = combined_trend / twenty_day_sd
    
    return adjusted_trend
```
