```python
import pandas as pd

def heuristics_v2(df, n=5, m=5, p=5):
    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=n).mean()
    
    # High-Low Momentum Score
    df['high_low_momentum_score'] = df.apply(lambda row: 1 if row['intraday_high_low_spread'] > row['avg_intraday_high_low_spread'] else -1, axis=1)
    df['high_low_momentum_score'] *= df['volume']
    
    # Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(m)
    df['volume_trend'] = df['volume_change'].rolling(window=m).sum()
    
    # Enhance Market Sentiment Analysis
    # Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Average Daily Range over past M days
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    
    # Range Momentum
    df['range_momentum'] = df.apply(lambda row: 1 if row['daily_range'] > row['avg_daily_range'] else -1, axis=1)
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_price_movement'] = df['open_close_diff'].rolling(window=m).sum()
    
    # Intraday Price Volatility
    df['intraday_prices'] = df[['open', 'high', 'low', 'close']].std(axis=1)
    df['avg_intraday_volatility'] = df['intraday_prices'].rolling(window=p).mean()
    
    # Price Volatility Score
    df['price_volatility_score'] = df.apply(lambda row: 1 if row['intraday_prices'] > row['avg_intraday_volatility'] else -1, axis=1)
    df['price_volatility_score'] *= df['volume']
    
    # Combine Factors
    factor = (
        0.3 * df['high_low_momentum_score'] +
        0.2 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.2 * df['range_momentum'] +
        0.1 * df['open_price_movement'] +
        0.1 * df['price_volatility_score']
    )
    
    return factor
```
