```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    range_high_low = df['high'] - df['low']
    spread_open_close = df['close'] - df['open']
    
    # Combine Range and Spread
    intraday_movement = range_high_low + spread_open_close
    
    # Weight by Average Volume
    volume_weighted_intraday_movement = intraday_movement * df['volume']
    
    # Confirm with Volume Activity
    volume_change = df['volume'].diff()
    
    # Increase or Decrease Momentum Score based on Volume Change
    momentum_score = volume_weighted_intraday_movement
    momentum_score[volume_change > 0] += (df['volume'] - df['volume'].shift(1))[volume_change > 0]
    momentum_score[volume_change < 0] -= (df['volume'].shift(1) - df['volume'])[volume_change < 0]
    
    return momentum_score
```
