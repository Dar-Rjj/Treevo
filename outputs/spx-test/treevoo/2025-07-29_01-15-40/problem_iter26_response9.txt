```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, p=5):
    # Calculate Intraday High-Low Spread
    df['intraday_hl_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread
    df['avg_intraday_hl_spread'] = df['intraday_hl_spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['hl_momentum_score'] = (df['intraday_hl_spread'] - df['avg_intraday_hl_spread']).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    df['hl_momentum_score_adj'] = df['hl_momentum_score'] * df['volume']
    
    # Calculate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Calculate Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(n)
    df['volume_trend'] = df['volume_change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    df['range_momentum'] = (df['daily_range'] - df['avg_daily_range']).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=n).sum()
    
    # Calculate Open to Low Price Difference
    df['open_low_diff'] = df['open'] - df['low']
    df['open_low_movement'] = df['open_low_diff'].rolling(window=n).sum()
    
    # Calculate Intraday Price Volatility
    df['intraday_volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1).rolling(window=p).mean()
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=m).mean()
    df['volatility_score'] = (df['intraday_volatility'] - df['avg_intraday_volatility']).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    df['volatility_score_adj'] = df['volatility_score'] * df['volume']
    
    # Combine Factors
    df['factor_value'] = (
        0.2 * df['hl_momentum_score_adj'] +
        0.2 * df['price_momentum'] +
        0.2 * df['volume_trend'] +
        0.1 * df['range_momentum'] +
        0.1 * df['open_close_movement'] +
        0.1 * df['open_low_movement'] +
        0.1 * df['volatility_score_adj']
    )
    
    return df['factor_value']
```
