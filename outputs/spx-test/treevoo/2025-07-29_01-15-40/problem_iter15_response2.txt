```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close to open price
    close_to_open = df['close'] / df['open']
    
    # Calculate the ratio of high to low price
    high_to_low = df['high'] / df['low']
    
    # Calculate the ratio of volume today vs yesterday
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Heuristic: If the close is higher than the open and the high is significantly higher than the low, 
    # and the volume has increased compared to the previous day, it might suggest positive momentum.
    factor = (close_to_open - 1) * (high_to_low - 1) * (volume_ratio - 1)
    
    return factor
```
