```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    avg_intraday_high_low_spread = intraday_high_low_spread.rolling(window=10).mean()
    high_low_momentum_score = (intraday_high_low_spread > avg_intraday_high_low_spread).astype(int) * 2 - 1
    high_low_momentum_score *= df['volume']
    
    # Calculate Price Momentum
    price_change = df['close'].diff(periods=5)
    price_momentum = price_change.rolling(window=5).sum()
    
    # Calculate Volume Trend
    volume_change = df['volume'].diff(periods=5)
    volume_trend = volume_change.rolling(window=5).sum()
    
    # Enhance Market Sentiment Analysis
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=10).mean()
    range_momentum = (daily_range > avg_daily_range).astype(int) * 2 - 1
    
    open_close_diff = df['close'] - df['open']
    open_close_momentum = open_close_diff.rolling(window=5).sum()
    
    # Incorporate Intraday Volatility
    intraday_volatility = 0.5 * np.log(df['high'] / df['low'])**2 - (2 * np.log(2) - 1) * np.log(df['close'] / df['open'])**2
    avg_intraday_volatility = intraday_volatility.rolling(window=10).mean()
    intraday_volatility_score = (intraday_volatility > avg_intraday_volatility).astype(int) * 2 - 1
    
    # Calculate Daily High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate Volume-Weighted Rolling Average of High-Low Ratios
    high_low_ratio_rolling = high_low_ratio.rolling(window=10).mean()
    volume_weighted_mean = (high_low_ratio * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Incorporate Volatility Adjustment
    close_std = df['close'].rolling(window=10).std()
    adjusted_volume_weighted_mean = volume_weighted_mean / close_std
    
    # Combine Factors
    alpha_factor = (
        0.2 * high_low_momentum_score +
        0.2 * price_momentum +
        0.2 * volume_trend +
        0.1 * range_momentum +
        0.1 * open_close_momentum +
        0.1 * intraday_volatility_score +
        0.1 * adjusted_volume_weighted_mean
    )
    
    return alpha_factor
```
