def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios
    rolling_window = 10
    df['Rolling_Avg_High_Low_Ratio'] = df['High_Low_Ratio'].rolling(window=rolling_window).mean()
    
    # Calculate Daily Volume Weight
    df['Volume_Weight'] = df['volume'] * df['close']
    
    # Calculate Volume Weighted High-Low Ratio
    df['Volume_Weighted_High_Low_Ratio'] = df['High_Low_Ratio'] * (df['volume'] / df['Volume_Weight'])
    
    # Calculate Rolling Average of Volume Weighted High-Low Ratios
    df['Volume_Weighted_Rolling_Avg_High_Low_Ratio'] = df['Volume_Weighted_High_Low_Ratio'].rolling(window=rolling_window).mean()
    
    return df['Volume_Weighted_Rolling_Avg_High_Low_Ratio']
