```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['5D_Momentum'] = df['close'].diff(5)
    
    # Calculate 20-Day Volume Moving Average
    df['20D_Vol_MA'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    df['Vol_Dev'] = df['volume'] - df['20D_Vol_MA']
    
    # Calculate 5-Day High-Low Spread
    df['5D_HL_Spread'] = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    df['OC_Spread'] = df['open'] - df['close']
    
    # Calculate Price Reversal Indicator
    df['Price_Reversal'] = df['5D_HL_Spread'] - df['OC_Spread']
    
    # Compute 5-Day Price Range and Volume Synchronicity
    df['5D_PR_Vol_Sync'] = df['5D_HL_Spread'] * df['Vol_Dev']
    
    # Compute 5-Day Momentum and Volume Synchronicity
    df['5D_Mom_Vol_Sync'] = df['5D_Momentum'] * df['Vol_Dev']
    
    # Combine Synchronicities
    df['Combined_Sync'] = df['5D_PR_Vol_Sync'] + df['5D_Mom_Vol_Sync']
    
    # Apply Exponential Smoothing
    df['Smoothed_Combined_Sync'] = df['Combined_Sync'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Construction
    df['Final_Factor'] = df['Smoothed_Combined_Sync'] * df['Price_Reversal']
    
    return df['Final_Factor']
```
