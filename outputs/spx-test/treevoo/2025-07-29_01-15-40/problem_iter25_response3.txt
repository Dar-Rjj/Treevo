import pandas as pd

def heuristics_v2(df):
    # Exponential Moving Average (EMA) Crossover
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    ema_crossover = ema_12 - ema_26

    # Linear Weighted Moving Average (LWMA) Crossover
    def lwma(series, n):
        weights = list(range(1, n+1))
        return series.rolling(window=n, center=False).apply(lambda x: (x * weights).sum() / sum(weights), raw=True)
    
    lwma_14 = lwma(df['close'], 14)
    lwma_28 = lwma(df['close'], 28)
    lwma_crossover = lwma_14 - lwma_28

    # Simple Moving Average (SMA) Difference
    sma_200 = df['close'].rolling(window=200).mean()
    sma_difference = df['close'] - sma_200

    # Rate of Change (ROC)
    roc_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    roc_50 = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    roc_100 = (df['close'] - df['close'].shift(100)) / df['close'].shift(100)

    # Momentum
    momentum_10 = df['close'] - df['close'].shift(10)
    momentum_50 = df['close'] - df['close'].shift(50)
    momentum_100 = df['close'] - df['close'].shift(100)

    # On-Balance-Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()

    # Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    vwap_difference = df['close'] - vwap

    # True Range (TR)
    tr = pd.DataFrame({
        'hl': df['high'] - df['low'],
        'h-pc': abs(df['high'] - df['close'].shift(1)),
        'l-pc': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)

    # ATR over a period
    atr_14 = tr.rolling(window=14).mean()

    # Bollinger Bands
    bbands_middle = df['close'].rolling(window=20).mean()
    bbands_std = df['close'].rolling(window=20).std()
    bbands_upper = bbands_middle + 2 * bbands_std
    bbands_lower = bbands_middle - 2 * bbands_std
    bbands_factor = (df['close'] - bbands_middle) / (bbands_upper - bbands_lower)

    # Relative Strength Index (RSI)
    delta = df['close'].diff()

    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Stochastic Oscillator
    low_n = df['low'].rolling(window=14).min()
    high_n = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - low_n) / (high_n - low_n)
