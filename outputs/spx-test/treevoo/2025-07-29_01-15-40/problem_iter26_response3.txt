```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    df['Volume_Weighted_High_Low'] = (df['High_Low_Ratio'] * df['volume']).rolling(window=rolling_window).mean() / df['volume'].rolling(window=rolling_window).mean()
    smoothing_factor = 0.2
    df['Volume_Weighted_High_Low'] = df['Volume_Weighted_High_Low'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Daily Open-Close Ratio
    df['Open_Close_Ratio'] = df['close'] / df['open']
    
    # Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Average Intraday High-Low Spread
    intraday_spread_window = 10
    df['Avg_Intraday_Spread'] = df['Intraday_Spread'].rolling(window=intraday_spread_window).mean()
    
    # High-Low Momentum Score
    df['High_Low_Momentum_Score'] = np.where(df['Intraday_Spread'] > df['Avg_Intraday_Spread'], 1, -1)
    df['High_Low_Momentum_Score'] *= df['volume']
    
    # Price Momentum
    price_momentum_period = 5
    df['Price_Momentum'] = df['close'] - df['close'].shift(price_momentum_period)
    df['Price_Momentum'] = df['Price_Momentum'].rolling(window=price_momentum_period).sum()
    
    # Volume Trend
    volume_trend_period = 5
    df['Volume_Trend'] = df['volume'] - df['volume'].shift(volume_trend_period)
    df['Volume_Trend'] = df['Volume_Trend'].rolling(window=volume_trend_period).sum()
    
    # Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Average Daily Range
    daily_range_window = 10
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=daily_range_window).mean()
    
    # Range Momentum
    df['Range_Momentum'] = np.where(df['Daily_Range'] > df['Avg_Daily_Range'], 1, -1)
    
    # Open to Close Price Difference
    df['Open_Close_Price_Diff'] = df['close'] - df['open']
    
    # Close to Open Price Ratio
    df['Close_Open_Price_Ratio'] = df['close'] / df['open']
    
    # Open to Low Price Difference
    df['Open_Low_Price_Diff'] = df['open'] - df['low']
    
    # Low to Open Price Ratio
    df['Low_Open_Price_Ratio'] = df['low'] / df['open']
    
    # Open to High Price Difference
    df['Open_High_Price_Diff'] = df['open'] - df['high']
    
    # High to Open Price Ratio
    df['High_Open_Price_Ratio'] = df['high'] / df['open']
    
    # Combine Factors
    alpha_factor = (
        0.2 * df['High_Low_Momentum_Score'] +
        0.15 * df['Volume_Weighted_High_Low'] +
        0.2 * df['Price_Momentum'] +
        0.1 * df['Volume_Trend'] +
        0.1 * df['Range_Momentum'] +
        0.05 * df['Open_Close_Price_Diff'] +
        0.05 * df['Close_Open_Price_Ratio'] +
        0.05 * df['Open_Low_Price_Diff'] +
        0.05 * df['Low_Open_Price_Ratio'] +
        0.05 * df['Open_High_Price_Diff'] +
        0.05 * df['High_Open_Price_Ratio']
    )
    
    return alpha_factor
```
