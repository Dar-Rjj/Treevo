```python
def heuristics_v2(df):
    # Compute Upward and Downward Movements
    df['Upward'] = df['high'] - df['close'].shift(1)
    df['Downward'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    df['NetDM'] = 0
    df.loc[df['Upward'] > df['Downward'], 'NetDM'] = 1
    df.loc[df['Upward'] < df['Downward'], 'NetDM'] = -1
    
    # Calculate CDMA
    df['CDMA'] = (df['NetDM'] * df['volume']).cumsum()
    
    return df['CDMA']
```
