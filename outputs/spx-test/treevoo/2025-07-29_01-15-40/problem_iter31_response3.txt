```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=20, k=20):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = (df['high_low_spread'] - df['avg_high_low_spread']).apply(lambda x: 1 if x > 0 else -1)
    df['high_low_momentum_score'] *= df['volume']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(n)
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].shift(n)
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    df['range_momentum'] = (df['daily_range'] - df['avg_daily_range']).apply(lambda x: 1 if x > 0 else -1)
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=n).sum()
    
    # Incorporate Intraday Volatility
    df['intraday_volatility'] = 0.5 * (np.log(df['high']) - np.log(df['low']))**2 - (2 * np.log(2) - 1) * (np.log(df['close']) - np.log(df['open']))**2
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=k).mean()
    df['intraday_volatility_score'] = (df['intraday_volatility'] - df['avg_intraday_volatility']).apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Factors
    df['alpha_factor'] = (
        0.2 * df['high_low_momentum_score'] +
        0.3 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.1 * df['range_momentum'] +
        0.2 * df['open_close_movement'] +
        0.1 * df['intraday_volatility_score']
    )
    
    return df['alpha_factor']
```
