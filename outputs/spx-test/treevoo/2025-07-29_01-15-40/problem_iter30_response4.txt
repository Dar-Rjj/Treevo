```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=5, p=5):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['Average_Intraday_HL_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['HL_Momentum_Score'] = np.where(df['Intraday_High_Low_Spread'] > df['Average_Intraday_HL_Spread'], 1, -1)
    df['HL_Momentum_Score'] *= df['volume']
    
    # Calculate Price Momentum
    df['Price_Change'] = df['close'] - df['close'].shift(n)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=n).sum()
    
    # Calculate Volume Trend
    df['Volume_Change'] = df['volume'] - df['volume'].shift(n)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    df['Average_Daily_Range'] = df['Daily_Range'].rolling(window=m).mean()
    df['Range_Momentum'] = np.where(df['Daily_Range'] > df['Average_Daily_Range'], 1, -1)
    
    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Open_Close_Movement'] = df['Open_Close_Diff'].rolling(window=n).sum()
    
    # Calculate Intraday Price Volatility
    df['Intraday_Volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1, ddof=0)
    df['Average_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=p).mean()
    
    # Compute Price Volatility Score
    df['Volatility_Score'] = np.where(df['Intraday_Volatility'] > df['Average_Intraday_Volatility'], 1, -1)
    df['Volatility_Score'] *= df['volume']
    
    # Incorporate Open-Price to High-Price Movement
    df['Open_High_Diff'] = df['high'] - df['open']
    df['Open_High_Movement'] = df['Open_High_Diff'].rolling(window=n).sum()
    
    # Combine Factors
    df['Factor'] = (df['HL_Momentum_Score'] * 0.2 +
                    df['Price_Momentum'] * 0.2 +
                    df['Volume_Trend'] * 0.1 +
                    df['Range_Momentum'] * 0.1 +
                    df['Open_Close_Movement'] * 0.2 +
                    df['Open_High_Movement'] * 0.1 +
                    df['Volatility_Score'] * 0.1)
    
    return df['Factor']
```
