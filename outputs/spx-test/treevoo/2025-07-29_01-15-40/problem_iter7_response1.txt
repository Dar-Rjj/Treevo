```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread over past N days
    df['Avg_Intraday_HL_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=N).mean()
    
    # Compute High-Low Momentum Score
    df['HL_Momentum_Score'] = (df['Intraday_High_Low_Spread'] - df['Avg_Intraday_HL_Spread']).apply(lambda x: 1 if x > 0 else -1)
    df['HL_Momentum_Score'] = df['HL_Momentum_Score'] * df['volume']
    
    # Calculate Simple Moving Average (SMA) of Close
    df['SMA_Close'] = df['close'].rolling(window=N).mean()
    
    # Calculate Price Reversal Indicator
    df['Price_Reversal_Indicator'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Compute Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine SMA and VWAP
    df['Integrated_Momentum'] = df['SMA_Close'] + df['VWAP']
    
    # Composite Factor Construction
    df['Composite_Factor'] = df['HL_Momentum_Score'] * df['Integrated_Momentum'] * df['Price_Reversal_Indicator']
    
    return df['Composite_Factor']
```
