import pandas as pd

def heuristics_v2(df, N=5, M=5, K=5):
    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=N).mean()
    
    # High-Low Momentum Score
    df['high_low_momentum_score'] = (df['intraday_high_low_spread'] > df['avg_intraday_high_low_spread']).astype(int) * 2 - 1
    df['high_low_momentum_score'] *= df['volume']
    
    # Price Momentum
    df['price_change'] = df['close'].diff()
    df['price_momentum'] = df['price_change'].rolling(window=N).sum()
    
    # Volume Trend
    df['volume_change'] = df['volume'].diff()
    df['volume_trend'] = df['volume_change'].rolling(window=N).sum()

    # Daily Range
    df['daily_range'] = df['high'] - df['low']
    # Average Daily Range over past M days
    df['avg_daily_range'] = df['daily_range'].rolling(window=M).mean()
    # Range Momentum Score
    df['range_momentum_score'] = (df['daily_range'] > df['avg_daily_range']).astype(int) * 2 - 1
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_momentum'] = df['open_close_diff'].rolling(window=N).sum()
