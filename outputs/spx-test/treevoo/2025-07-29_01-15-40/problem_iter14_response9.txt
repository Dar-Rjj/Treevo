```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Ratio Momentum
    high_low_ratio = df['high'] / df['low']
    high_low_momentum = high_low_ratio.rolling(window=10).mean()
    
    # Calculate Price Momentum
    price_change = df['close'].diff(periods=5)
    price_momentum = price_change.rolling(window=5).sum()
    
    # Integrate Volume into Momentum
    vwap = (df['close'] + df['open']) / 2 * df['volume'] / df['volume'].rolling(window=5).sum()
    vwap = vwap.rolling(window=5).sum()
    momentum_with_volume = (price_momentum + vwap) / 2
    
    # Calculate Price Reversal Indicator
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['open'] - df['close']
    price_reversal = high_low_spread - open_close_spread
    
    # Calculate Upward and Downward Movements
    upward_movement = df['high'] - df['close'].shift(1)
    downward_movement = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    net_directional_movement = (upward_movement > downward_movement).astype(int) - (upward_movement < downward_movement).astype(int)
    
    # Calculate Cumulative Directional Movement Adjusted (CDMA)
    cdma = (net_directional_movement * df['volume']).rolling(window=5).sum()
    
    # Composite Factor Construction
    composite_factor = (high_low_momentum + price_momentum) / 2 * cdma
    
    return composite_factor
```
