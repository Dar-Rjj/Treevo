```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, high_low_coef=0.3, price_mom_coef=0.2, volume_trend_coef=0.1, range_mom_coef=0.2, open_close_coef=0.1, open_low_coef=0.1):
    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']

    # Compute Average Intraday High-Low Spread over past N days
    df['Avg_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=n).mean()

    # Compute High-Low Momentum Score
    df['High_Low_Momentum'] = (df['High_Low_Spread'] > df['Avg_High_Low_Spread']).astype(int) * 2 - 1
    df['High_Low_Momentum_Adjusted'] = df['High_Low_Momentum'] * df['volume']

    # Calculate Price Momentum over the last 5 days
    df['Price_Change'] = df['close'].diff(periods=5)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=5).sum()

    # Calculate Volume Trend over the last 5 days
    df['Volume_Change'] = df['volume'].diff(periods=5)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=5).sum()

    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=m).mean()
    df['Range_Momentum'] = (df['Daily_Range'] > df['Avg_Daily_Range']).astype(int) * 2 - 1

    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Open_Low_Diff'] = df['open'] - df['low']
    df['Open_Close_Movement'] = df['Open_Close_Diff'].rolling(window=5).sum()
    df['Open_Low_Movement'] = df['Open_Low_Diff'].rolling(window=5).sum()

    # Combine Factors
    df['Factor_Value'] = (high_low_coef * df['High_Low_Momentum_Adjusted'] +
                          price_mom_coef * df['Price_Momentum'] +
                          volume_trend_coef * df['Volume_Trend'] +
                          range_mom_coef * df['Range_Momentum'] +
                          open_close_coef * df['Open_Close_Movement'] +
                          open_low_coef * df['Open_Low_Movement'])

    return df['Factor_Value']
```
