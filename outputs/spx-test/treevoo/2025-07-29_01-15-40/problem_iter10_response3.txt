```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Compute Rolling Average of High-Low Ratios with a 10-day window
    rolling_window = 10
    df['high_low_momentum'] = df['high_low_ratio'].rolling(window=rolling_window).mean()
    
    # Return the high-low momentum factor as a pandas Series
    return df['high_low_momentum']
```
