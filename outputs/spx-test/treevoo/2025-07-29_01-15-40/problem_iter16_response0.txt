```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios (10 days)
    rolling_high_low_avg = high_low_ratio.rolling(window=10).mean()
    
    # Calculate 5-Day Momentum
    five_day_momentum = df['close'].diff(5)
    
    # Calculate 20-Day Volume Moving Average
    volume_20_day_ma = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    volume_deviation = df['volume'] - volume_20_day_ma
    
    # Combine Momentum and Volume Synchronicity
    combined_synchronicity = five_day_momentum * volume_deviation
    
    # Apply Exponential Smoothing to Combined Synchronicity
    alpha = 0.2  # Smoothing factor
    smoothed_synchronicity = combined_synchronicity.ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Final Factor
    final_factor = 0.7 * smoothed_synchronicity + 0.3 * rolling_high_low_avg
    
    return final_factor
```
