```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, p=10, q=5, 
                  hl_weight=0.3, pm_weight=0.2, vt_weight=0.1, rm_weight=0.1, opm_weight=0.1, pvs_weight=0.1, ohm_weight=0.1):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread
    df['Avg_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['High_Low_Momentum_Score'] = (df['Intraday_High_Low_Spread'] - df['Avg_Intraday_High_Low_Spread']).apply(lambda x: 1 if x > 0 else -1)
    df['High_Low_Momentum_Score'] *= df['volume']
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(n)
    
    # Calculate Volume Trend
    df['Volume_Trend'] = df['volume'] - df['volume'].shift(n)
    
    # Enhance Market Sentiment Analysis
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Compute Average Daily Range
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=m).mean()
    
    # Calculate Range Momentum
    df['Range_Momentum'] = (df['Daily_Range'] - df['Avg_Daily_Range']).apply(lambda x: 1 if x > 0 else -1)
    
    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Price_Diff'] = df['close'] - df['open']
    df['Open_Close_Price_Diff_Sum'] = df['Open_Close_Price_Diff'].rolling(window=n).sum()
    
    # Calculate Intraday Price Volatility
    df['Intraday_Price_Volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1).rolling(window=p).mean()
    
    # Compute Price Volatility Score
    df['Price_Volatility_Score'] = (df['Intraday_Price_Volatility'] - df['Intraday_Price_Volatility'].rolling(window=n).mean()).apply(lambda x: 1 if x > 0 else -1)
    df['Price_Volatility_Score'] *= df['volume']
    
    # Calculate Intraday Open-High Spread
    df['Intraday_Open_High_Spread'] = df['high'] - df['open']
    
    # Compute Average Intraday Open-High Spread
    df['Avg_Intraday_Open_High_Spread'] = df['Intraday_Open_High_Spread'].rolling(window=q).mean()
    
    # Compute Open-High Momentum Score
    df['Open_High_Momentum_Score'] = (df['Intraday_Open_High_Spread'] - df['Avg_Intraday_Open_High_Spread']).apply(lambda x: 1 if x > 0 else -1)
    df['Open_High_Momentum_Score'] *= df['volume']
    
    # Combine Factors
    df['Factor_Value'] = (
        hl_weight * df['High_Low_Momentum_Score'] +
        pm_weight * df['Price_Momentum'] +
        vt_weight * df['Volume_Trend'] +
        rm_weight * df['Range_Momentum'] +
        opm_weight * df['Open_Close_Price_Diff_Sum'] +
        pvs_weight * df['Price_Volatility_Score'] +
        ohm_weight * df['Open_High_Momentum_Score']
    )
    
    return df['Factor_Value']
```
