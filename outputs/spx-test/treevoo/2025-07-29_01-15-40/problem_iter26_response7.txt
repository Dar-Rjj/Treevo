```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, p=10):
    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['Avg_Intraday_HL_Spread'] = df['Intraday_HL_Spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['HL_Momentum_Score'] = (df['Intraday_HL_Spread'] - df['Avg_Intraday_HL_Spread']).apply(lambda x: 1 if x > 0 else -1)
    df['HL_Momentum_Score_Adjusted'] = df['HL_Momentum_Score'] * df['volume']
    
    # Calculate Price Momentum
    df['Price_Change'] = df['close'].pct_change(periods=n)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=n).sum()
    
    # Calculate Volume Trend
    df['Volume_Change'] = df['volume'].pct_change(periods=n)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=m).mean()
    df['Range_Momentum'] = (df['Daily_Range'] - df['Avg_Daily_Range']).apply(lambda x: 1 if x > 0 else -1)
    df['Open_Close_Price_Diff'] = df['close'] - df['open']
    df['Open_Close_Price_Movement'] = df['Open_Close_Price_Diff'].rolling(window=n).sum()
    
    # Calculate Open to Low Price Difference
    df['Open_Low_Price_Diff'] = df['low'] - df['open']
    df['Open_Low_Price_Movement'] = df['Open_Low_Price_Diff'].rolling(window=n).sum()
    
    # Calculate Intraday Price Volatility
    intraday_prices = df[['open', 'high', 'low', 'close']].std(axis=1)
    df['Intraday_Volatility'] = intraday_prices.rolling(window=p).mean()
    df['Avg_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=n).mean()
    df['Volatility_Score'] = (df['Intraday_Volatility'] - df['Avg_Intraday_Volatility']).apply(lambda x: 1 if x > 0 else -1)
    df['Volatility_Score_Adjusted'] = df['Volatility_Score'] * df['volume']
    
    # Incorporate Open Price Movement Factor
    df['Open_Price_Change'] = df['open'].pct_change(periods=n)
    df['Open_Price_Movement'] = df['Open_Price_Change'].rolling(window=n).sum()
    
    # Combine Factors with Coefficients
    coefficient_hl_mom = 0.2
    coefficient_price_mom = 0.2
    coefficient_volume_trend = 0.1
    coefficient_range_mom = 0.1
    coefficient_open_close = 0.1
    coefficient_open_low = 0.1
    coefficient_volatility_score = 0.1
    coefficient_open_price = 0.1
    
    df['Factor'] = (coefficient_hl_mom * df['HL_Momentum_Score_Adjusted'] +
                    coefficient_price_mom * df['Price_Momentum'] +
                    coefficient_volume_trend * df['Volume_Trend'] +
                    coefficient_range_mom * df['Range_Momentum'] +
                    coefficient_open_close * df['Open_Close_Price_Movement'] +
                    coefficient_open_low * df['Open_Low_Price_Movement'] +
                    coefficient_volatility_score * df['Volatility_Score_Adjusted'] +
                    coefficient_open_price * df['Open_Price_Movement'])
    
    return df['Factor']
```
