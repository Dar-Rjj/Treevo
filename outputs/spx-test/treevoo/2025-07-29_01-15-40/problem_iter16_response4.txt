```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return with 21-day lookback period
    df['ewma_log_return'] = df['log_return'].ewm(span=21).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combined Momentum: Multiply EWMA Log Return by Volume Change
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Price Momentum over 5 days
    df['price_momentum'] = df['close'].pct_change().rolling(window=5).sum()
    
    # Calculate Volume Trend over 5 days
    df['volume_trend'] = df['volume'].pct_change().rolling(window=5).sum()
    
    # Calculate Price Reversal Indicator
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['open'] - df['close']
    df['price_reversal'] = df['high_low_spread'] - df['open_close_spread']
    
    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    conditions = [
        (df['upward_movement'] > df['downward_movement']),
        (df['upward_movement'] < df['downward_movement'])
    ]
    choices = [1, -1]
    df['net_directional_movement'] = np.select(conditions, choices, default=0)
    
    # Calculate Cumulative Directional Movement Adjusted (CDMA) over 5 days
    df['cdma'] = df['net_directional_movement'] * df['volume']
    df['cdma_5d'] = df['cdma'].rolling(window=5).sum()
    
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios over 10 days
    df['high_low_ratio_avg'] = df['high_low_ratio'].rolling(window=10).mean()
    
    # Calculate Daily Volume Weight
    total_volume = df['volume'].rolling(window=10).sum()
    df['volume_weight'] = df['volume'] / total_volume
    
    # Calculate Rolling Average of Volume Weighted High-Low Ratios over 10 days
    df['weighted_high_low_ratio'] = df['high_low_ratio'] * df['volume_weight']
    df['weighted_high_low_ratio_avg'] = df['weighted_high_low_ratio'].rolling(window=10).mean()
    
    # Composite Factor Construction
    df['composite_factor'] = (
        df['combined_momentum'] + 
        df['price_momentum'] + 
        df['volume_trend'] + 
        df['cdma_5d'] + 
        df['weighted_high_low_ratio_avg']
    )
    
    return df['composite_factor']
```
