```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return
    df['ewma_log_return'] = df['log_return'].ewm(span=21, min_periods=21).mean()
    
    # Integrate Price and Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Price Momentum over 5 days
    df['price_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Trend over 5 days
    df['volume_trend'] = df['volume'] - df['volume'].shift(5)
    
    # Calculate Price Reversal Indicator
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['open'] - df['close'].shift(1)
    df['price_reversal'] = df['high_low_spread'] - df['open_close_spread']
    
    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    df['net_directional'] = np.where(df['upward_movement'] > df['downward_movement'], 1,
                                     np.where(df['upward_movement'] < df['downward_movement'], -1, 0))
    
    # Calculate Cumulative Directional Movement Adjusted (CDMA)
    df['cdma'] = df['net_directional'].rolling(window=5, min_periods=5).sum() * df['volume']
    
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Rolling Average of High-Low Ratios
    df['rolling_high_low_ratio'] = df['high_low_ratio'].rolling(window=10, min_periods=10).mean()
    
    # Incorporate Volume Weighted Factor
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=10, min_periods=10).sum()
    df['volume_weighted_rolling_high_low_ratio'] = (df['high_low_ratio'] * df['volume_weight']).rolling(window=10, min_periods=10).sum()
    
    # Calculate 5-Day Momentum
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    df['20_day_vol_ma'] = df['volume'].rolling(window=20, min_periods=20).mean()
    
    # Compute Volume Deviation from MA
    df['volume_deviation'] = df['volume'] - df['20_day_vol_ma']
    
    # Combine Momentum and Volume Synchronicity
    df['synchronicity'] = df['5_day_momentum'] * df['volume_deviation']
    
    # Apply Exponential Smoothing to Combined Synchronicity
    df['smoothed_synchronicity'] = df['synchronicity'].ewm(span=10, min_periods=10).mean()
    
    # Composite Factor Construction
    df['composite_factor'] = (df['combined_momentum'] + 
                              df['volume_trend'] + 
                              df['cdma'] + 
                              df['smoothed_synchronicity'] + 
                              df['volume_weighted_rolling_high_low_ratio'])
    
    return df['composite_factor']
```
