```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Price Movement
    range_ = df['high'] - df['low']
    spread_ = df['close'] - df['open']
    intraday_movement = range_ + spread_

    # Volume Confirmation
    volume_change = df['volume'].diff()
    momentum_score = volume_change.apply(lambda x: 1 if x > 0 else -1)

    # Price Trend Component
    ma_10 = df['close'].rolling(window=10).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    price_trend = ma_10 - ma_50

    # Volume Trend Component
    vol_10_sum = df['volume'].rolling(window=10).sum()
    vol_50_sum = df['volume'].rolling(window=50).sum()
    volume_trend = vol_10_sum - vol_50_sum

    # Combined Indicator
    combined_intraday_volume = intraday_movement * momentum_score
    combined_price_volume_trend = price_trend * volume_trend
    final_indicator = combined_intraday_volume + combined_price_volume_trend

    return final_indicator
```
