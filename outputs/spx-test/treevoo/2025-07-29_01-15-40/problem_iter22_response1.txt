import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Volume-Weighted Rolling Average of High-Low Ratios
    window = 10
    df['Volume_Weighted_High_Low_Ratio'] = (df['High_Low_Ratio'] * df['volume']).rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
    
    # Apply Exponential Smoothing
    alpha = 0.2
    df['Smoothed_Volume_Weighted_High_Low_Ratio'] = df['Volume_Weighted_High_Low_Ratio'].ewm(alpha=alpha, adjust=False).mean()
    
    return df['Smoothed_Volume_Weighted_High_Low_Ratio']
