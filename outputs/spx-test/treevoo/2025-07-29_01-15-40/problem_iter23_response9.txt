import pandas as pd

def heuristics_v2(df):
    # Momentum Sub-Thoughts
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()
    df['momentum'] = df['5_day_SMA'] - df['20_day_SMA']
    
    # Volatility Sub-Thoughts
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_close_prev'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['14_day_ATR'] = df['true_range'].rolling(window=14).mean()
    df['7_day_ATR'] = df['true_range'].rolling(window=7).mean()
    df['volatility_ratio'] = df['7_day_ATR'] / df['14_day_ATR']
    
    # Volume Sub-Thoughts
    df['volume_change'] = df['volume'].pct_change()
    df['5_day_vol_change'] = df['volume_change'].rolling(window=5).mean()
    df['volume_signal'] = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    
    # Trading Range Sub-Thoughts
    df['daily_trading_range'] = df['high'] - df['low']
    df['10_day_TR_MA'] = df['daily_trading_range'].rolling(window=10).mean()
    df['range_signal'] = (df['daily_trading_range'] > df['10_day_TR_MA']).astype(int)
    
    # Open and Close Prices Sub-Thoughts
    df['overnight_return'] = df['open'].pct_change()
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['combined_return'] = df['overnight_return'] + df['intraday_return']
    
    # Composite Alpha Signal
    df['composite_alpha'] = (
        0.4 * df['momentum'] + 
        0.3 * df['volatility_ratio'] + 
        0.2 * df['volume_signal'] + 
        0.1 * df['range_signal'] +
        0.1 * df['combined_return']
    )
    
    return df['composite_alpha'].dropna()
