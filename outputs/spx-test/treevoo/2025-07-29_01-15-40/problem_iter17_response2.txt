```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios
    df['Rolling_High_Low_Ratio_Avg'] = df['High_Low_Ratio'].rolling(window=10).mean()
    
    # Calculate 5-Day Momentum
    df['5D_Momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Intraday High-Low Range
    df['Intraday_HL_Range'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Range
    df['Avg_Intraday_HL_Range'] = df['Intraday_HL_Range'].rolling(window=10).mean()
    
    # Compute Intraday High-Low Range Score
    df['HL_Range_Score'] = np.where(df['Intraday_HL_Range'] > df['Avg_Intraday_HL_Range'], 1, -1)
    
    # Adjust Intraday High-Low Range Score by 5-Day Momentum
    df['Adjusted_HL_Range_Score'] = df['HL_Range_Score'] * df['5D_Momentum']
    
    # Calculate 20-Day Volume Moving Average
    df['20D_Volume_MA'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    df['Volume_Deviation'] = df['volume'] - df['20D_Volume_MA']
    
    # Adjust by Volume
    df['Volume_Adjusted_Score'] = df['Adjusted_HL_Range_Score'] * df['Volume_Deviation']
    
    # Calculate 5-Day Close Price Standard Deviation
    df['5D_Close_STD'] = df['close'].rolling(window=5).std()
    
    # Adjust by Price Volatility
    df['Price_Vol_Adj_Score'] = df['Volume_Adjusted_Score'] / df['5D_Close_STD']
    
    # Apply Weighted Moving Average Smoothing
    weights = np.array([0.1, 0.2, 0.3, 0.4])
    df['WMA_Smoothed_Score'] = df['Price_Vol_Adj_Score'].rolling(window=4).apply(lambda x: np.sum(weights * x), raw=False)
    
    # Combine Momentum and Volume Synchronicity
    df['Synchronicity'] = df['5D_Momentum'] * df['Volume_Deviation']
    
    # Apply Exponential Smoothing to Combined Synchronicity
    df['Exp_Smoothed_Synchronicity'] = df['Synchronicity'].ewm(span=5, adjust=False).mean()
    
    # Enhance Market Sentiment
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=10).mean()
    df['Market_Sentiment'] = np.where(df['Daily_Range'] > df['Avg_Daily_Range'], 1, -1)
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Open_Close_Sum'] = df['Open_Close_Diff'].rolling(window=10).sum()
    df['Enhanced_Sentiment'] = df['Market_Sentiment'] * df['Open_Close_Sum']
    
    # Calculate Final Factor
    df['Final_Factor'] = (df['WMA_Smoothed_Score'] + df['Exp_Smoothed_Synchronicity']) + df['Enhanced_Sentiment']
    df['Final_Factor'] /= df['Rolling_High_Low_Ratio_Avg'].rolling(window=10).std()
    
    return df['Final_Factor']
```
