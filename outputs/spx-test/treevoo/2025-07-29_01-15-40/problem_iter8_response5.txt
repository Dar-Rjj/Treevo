```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread over past N days
    N = 10
    avg_intraday_high_low_spread = intraday_high_low_spread.rolling(window=N).mean()
    
    # Compute High-Low Momentum Score
    high_low_momentum_score = (intraday_high_low_spread - avg_intraday_high_low_spread) * df['volume']
    
    # Calculate 5-Day Momentum
    five_day_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    twenty_day_vol_ma = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    volume_deviation_from_ma = df['volume'] - twenty_day_vol_ma
    
    # Calculate 5-Day Price Range
    five_day_price_range = df['high'] - df['low']
    
    # Calculate Price Range and Volume Synchronicity
    price_range_volume_sync = five_day_price_range * volume_deviation_from_ma
    
    # Calculate Momentum and Volume Synchronicity
    momentum_volume_sync = five_day_momentum * volume_deviation_from_ma
    
    # Combine Synchronicities
    combined_synchronicities = momentum_volume_sync + price_range_volume_sync
    
    # Apply Exponential Smoothing
    smoothed_synchronicities = combined_synchronicities.ewm(span=10, adjust=False).mean()
    
    # Price Trend Component
    ten_day_ma_close = df['close'].rolling(window=10).mean()
    fifty_day_ma_close = df['close'].rolling(window=50).mean()
    price_trend = ten_day_ma_close - fifty_day_ma_close
    
    # Volume Trend Component
    ten_day_sum_volume = df['volume'].rolling(window=10).sum()
    fifty_day_sum_volume = df['volume'].rolling(window=50).sum()
    volume_trend = ten_day_sum_volume - fifty_day_sum_volume
    
    # Combined Indicator
    combined_indicator = price_trend * volume_trend
    twenty_day_sd_close = df['close'].rolling(window=20).std()
    adjusted_combined_indicator = combined_indicator / twenty_day_sd_close
    
    # Integrate Volume into Momentum
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    integrated_momentum = (ten_day_ma_close + vwap) / 2
    
    # Composite Factor Construction
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['open'] - df['close']
    price_reversal_indicator = high_low_spread - open_close_spread
    composite_factor = high_low_momentum_score * integrated_momentum * price_reversal_indicator
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_synchronicities + adjusted_combined_indicator + composite_factor
    return final_alpha_factor
```
