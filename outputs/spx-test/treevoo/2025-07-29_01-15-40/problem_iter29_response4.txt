```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Compute High-Low Momentum Score
    df['High_Low_Momentum'] = (df['Intraday_HL_Spread'] > df['Intraday_HL_Spread'].shift(1)).astype(int) * 2 - 1
    df['High_Low_Momentum'] *= df['volume']
    
    # Calculate Price Momentum over 5 days
    period = 5
    df['Price_Change'] = df['close'] - df['close'].shift(period)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=period).sum()
    
    # Calculate Volume Trend over 5 days
    df['Volume_Change'] = df['volume'] - df['volume'].shift(period)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=period).sum()
    
    # Enhance Market Sentiment Analysis
    df['Range_Momentum'] = (df['Intraday_HL_Spread'] > df['Intraday_HL_Spread'].shift(1)).astype(int) * 2 - 1
    
    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Open_Close_Movement'] = df['Open_Close_Diff'].rolling(window=period).sum()
    
    # Combine Factors with weighted sum
    alpha_factor = (
        0.3 * df['High_Low_Momentum'] +
        0.2 * df['Price_Momentum'] +
        0.2 * df['Volume_Trend'] +
        0.1 * df['Range_Momentum'] +
        0.2 * df['Open_Close_Movement']
    )
    
    return alpha_factor
```
