```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios (10 days)
    df['avg_high_low_ratio'] = df['high_low_ratio'].rolling(window=10).mean()
    
    # Calculate Daily Volume Weight
    df['volume_weight'] = df['volume'] / df['close']
    
    # Calculate Rolling Average of Volume Weighted High-Low Ratios (10 days)
    df['vol_weighted_avg_high_low_ratio'] = (df['high_low_ratio'] * df['volume_weight']).rolling(window=10).mean() / df['volume_weight'].rolling(window=10).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread (10 days)
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=10).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = df['intraday_high_low_spread'] - df['avg_intraday_high_low_spread']
    df['high_low_momentum_score'] *= df['volume']
    
    # Calculate Price Momentum (5 days)
    df['price_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Volume Trend (5 days)
    df['volume_trend'] = df['volume'] - df['volume'].shift(5)
    
    # Combine High-Low Momentum, Price Momentum, and Volume Trend
    df['combined_factor'] = 0.4 * df['high_low_momentum_score'] + 0.3 * df['price_momentum'] + 0.3 * df['volume_trend']
    
    # Final Factor
    return df['combined_factor']
```
