import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    momentum_5d = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    volume_ma_20d = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    volume_deviation = df['volume'] - volume_ma_20d
    
    # Calculate 5-Day Price Range
    price_range_5d = df['high'] - df['low']
    
    # Calculate Price Range and Volume Synchronicity
    synchronicity_price_volume = price_range_5d * volume_deviation
    
    # Calculate Momentum and Volume Synchronicity
    synchronicity_momentum_volume = momentum_5d * volume_deviation
    
    # Combine Synchronicities
    combined_synchronicities = synchronicity_price_volume + synchronicity_momentum_volume
    
    # Apply Exponential Smoothing to reduce noise
    smoothed_synchronicities = combined_synchronicities.ewm(span=10, adjust=False).mean()
    
    # Price Trend Component
    ma_close_10d = df['close'].rolling(window=10).mean()
    ma_close_50d = df['close'].rolling(window=50).mean()
    price_trend = ma_close_10d - ma_close_50d
    
    # Volume Trend Component
    sum_volume_10d = df['volume'].rolling(window=10).sum()
    sum_volume_50d = df['volume'].rolling(window=50).sum()
    volume_trend = sum_volume_10d - sum_volume_50d
    
    # Combined Indicator
    combined_indicator = price_trend * volume_trend
    
    # Adjust for Volatility
    std_close_20d = df['close'].rolling(window=20).std()
    adjusted_combined_indicator = combined_indicator / std_close_20d
    
    # Final Alpha Factor
    final_alpha_factor = smoothed_synchronicities + adjusted_combined_indicator
    
    return final_alpha_factor
