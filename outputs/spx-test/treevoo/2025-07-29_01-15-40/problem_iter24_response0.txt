```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of close to open price
    close_to_open_ratio = df['close'] / df['open']
    
    # Calculate the ratio of close to high price
    close_to_high_ratio = df['close'] / df['high']
    
    # Calculate the ratio of close to low price
    close_to_low_ratio = df['close'] / df['low']
    
    # Calculate the average of the three ratios
    avg_ratio = (close_to_open_ratio + close_to_high_ratio + close_to_low_ratio) / 3
    
    # Calculate the weighted sum of the percentage change and the average ratio
    factor_values = 0.5 * close_pct_change + 0.5 * (avg_ratio - 1)
    
    return factor_values
```
