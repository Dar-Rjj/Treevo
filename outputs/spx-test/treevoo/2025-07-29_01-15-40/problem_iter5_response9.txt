```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios
    rolling_window = 10
    df['rolling_high_low_ratio'] = df['high_low_ratio'].rolling(window=rolling_window).mean()
    
    # Incorporate Volume Weighted Factor
    df['volume_weight'] = df['volume'] * df['close']
    df['volume_weighted_high_low_ratio'] = df['high_low_ratio'] * df['volume_weight']
    df['rolling_volume_weighted_high_low_ratio'] = df['volume_weighted_high_low_ratio'].rolling(window=rolling_window).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread
    average_window = 10
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=average_window).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = (df['intraday_high_low_spread'] - df['avg_intraday_high_low_spread']).apply(lambda x: 1 if x > 0 else -1)
    df['high_low_momentum_score'] *= df['volume']
    
    # Calculate Price Momentum
    price_momentum_period = 5
    df['price_change'] = df['close'] - df['close'].shift(price_momentum_period)
    df['price_momentum'] = df['price_change'].rolling(window=price_momentum_period).sum()
    
    # Calculate Volume Trend
    volume_trend_period = 5
    df['volume_change'] = df['volume'] - df['volume'].shift(volume_trend_period)
    df['volume_trend'] = df['volume_change'].rolling(window=volume_trend_period).sum()
    
    # Combine High-Low Momentum, Price Momentum, and Volume Trend
    df['market_sentiment'] = df['high'] - df['low']
    df['weighted_sum'] = (
        0.4 * df['high_low_momentum_score'] +
        0.3 * df['price_momentum'] +
        0.3 * df['volume_trend']
    )
    
    # Calculate Final Factor
    df['final_factor'] = df['weighted_sum'] + df['market_sentiment']
    
    return df['final_factor']
```
