```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    N = 5  # Number of days for high-low momentum
    M = 5  # Number of days for market sentiment
    K = 5  # Number of days for intraday volatility
    P = 5  # Number of days for price and volume trend
    O = 5  # Number of days for open-price movement

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Calculate Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=N).mean()

    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = (df['intraday_high_low_spread'] - df['avg_intraday_high_low_spread']).apply(lambda x: 1 if x > 0 else -1)
    df['high_low_momentum_score'] = df['high_low_momentum_score'] * df['volume']

    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(P)
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].shift(P)

    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=M).mean()
    df['range_momentum'] = (df['daily_range'] - df['avg_daily_range']).apply(lambda x: 1 if x > 0 else -1)
    df['open_to_close_diff'] = df['close'] - df['open']
    df['open_close_momentum'] = df['open_to_close_diff'].rolling(window=P).sum()

    # Incorporate Intraday Volatility
    df['intraday_volatility'] = 0.5 * (np.log(df['high']) - np.log(df['low']))**2 - (2 * np.log(2) - 1) * (np.log(df['close']) - np.log(df['open']))**2
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=K).mean()
    df['intraday_volatility_score'] = (df['intraday_volatility'] - df['avg_intraday_volatility']).apply(lambda x: 1 if x > 0 else -1)

    # Combine all factors with weighted sum
    df['alpha_factor'] = (
        0.3 * df['high_low_momentum_score'] +
        0.2 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.15 * df['range_momentum'] +
        0.1 * df['open_close_momentum'] +
        0.15 * df['intraday_volatility_score']
    )

    # Incorporate Open-Price Change
    df['open_price_change'] = df['open'] - df['open'].shift(O)
    df['open_price_momentum'] = df['open_price_change'].rolling(window=O).sum()
    df['alpha_factor'] += 0.05 * df['open_price_momentum']

    return df['alpha_factor']
```
