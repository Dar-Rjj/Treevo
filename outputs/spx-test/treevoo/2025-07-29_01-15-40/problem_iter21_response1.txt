```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']
    
    # Determine Net Intraday and Directional Movements
    df['upward'] = (df['close'] > df['open']) * (df['close'] - df['open'])
    df['downward'] = (df['close'] < df['open']) * (df['open'] - df['close'])
    df['net_movement'] = df['upward'] - df['downward']
    
    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['combined_momentum'] = (df['range'] + df['spread'] + df['net_movement']) * df['volume'].rolling(window=21).mean()
    df['combined_momentum'] = df['combined_momentum'] * df['volume_change']
    
    # Calculate Intraday High-Low Spread and Average Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    avg_high_low_spread = df['high_low_spread'].rolling(window=21).mean()
    df['high_low_momentum'] = (df['high_low_spread'] - avg_high_low_spread) * df['volume']
    
    # Calculate Log Return and EWMA of Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['ewma_log_return'] = df['log_return'].ewm(span=21).mean()
    
    # Integrate Price and Volume
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['integrated_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Upward and Downward Movements
    df['upward_move'] = df['high'] - df['close'].shift(1)
    df['downward_move'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    df['net_directional_movement'] = 0
    df.loc[df['upward_move'] > df['downward_move'], 'net_directional_movement'] = 1
    df.loc[df['upward_move'] < df['downward_move'], 'net_directional_movement'] = -1
    
    # Calculate Adjusted Cumulative Directional Movement
    df['cumulative_directional_movement'] = df['net_directional_movement'].rolling(window=21).sum() * df['volume']
    
    # Add Trend Strength Indicator
    df['daily_range'] = df['high'] - df['low']
    avg_daily_range = df['daily_range'].rolling(window=21).mean()
    df['trend_strength'] = 0
    df.loc[df['daily_range'] > avg_daily_range, 'trend_strength'] = 1
    
    # Integrate Volume into Momentum
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['sma_21'] = df['close'].rolling(window=21).mean()
    df['integrated_momentum'] = df['integrated_momentum'] * (df['sma_21'] + df['vwap'])
    
    # Composite Factor Construction
    df['composite_factor'] = (df['high_low_momentum'] * df['integrated_momentum'] * df['trend_strength'] 
                              + df['combined_momentum'] + df['cumulative_directional_movement'])
    
    # Add a Volatility Component
    df['volatility'] = df['log_return'].rolling(window=21).std()
    df['final_factor'] = df['composite_factor'] / df['volatility']
    
    # Incorporate Market Sentiment
    df['rsi'] = 100 - (100 / (1 + (df['close'].diff().clip(lower=0).rolling(window=14).mean() 
                                 / df['close'].diff().clip(upper=0).rolling(window=14).mean().abs())))
    df.loc[df['rsi'] < 30, 'final_factor'] *= 1.1
    df.loc[df['rsi'] > 70, 'final_factor'] *= 0.9
    
    return df['final_factor']
```
