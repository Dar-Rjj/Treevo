```python
import pandas as pd

def heuristics_v2(df, n_days=5, m_days=5, high_low_coeff=0.3, price_mom_coeff=0.2, volume_trend_coeff=0.1, range_mom_coeff=0.2, open_close_coeff=0.1, open_low_coeff=0.1):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=n_days).mean()
    
    # Compute High-Low Momentum Score and Adjust by Volume
    df['high_low_momentum_score'] = (df['high_low_spread'] > df['avg_high_low_spread']).astype(int) * df['volume'] - (df['high_low_spread'] < df['avg_high_low_spread']).astype(int) * df['volume']
    
    # Calculate Price Momentum over the period
    df['price_change'] = df['close'] - df['close'].shift(n_days)
    df['price_momentum'] = df['price_change'].rolling(window=n_days).sum()
    
    # Calculate Volume Trend over the period
    df['volume_change'] = df['volume'] - df['volume'].shift(n_days)
    df['volume_trend'] = df['volume_change'].rolling(window=n_days).sum()
    
    # Enhance Market Sentiment Analysis
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    # Compute Average Daily Range
    df['avg_daily_range'] = df['daily_range'].rolling(window=m_days).mean()
    # Calculate Range Momentum
    df['range_momentum'] = (df['daily_range'] > df['avg_daily_range']).astype(int) - (df['daily_range'] < df['avg_daily_range']).astype(int)
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=n_days).sum()
    df['open_low_diff'] = df['low'] - df['open']
    df['open_low_movement'] = df['open_low_diff'].rolling(window=n_days).sum()
    
    # Combine Factors
    df['alpha_factor'] = (
        high_low_coeff * df['high_low_momentum_score'] +
        price_mom_coeff * df['price_momentum'] +
        volume_trend_coeff * df['volume_trend'] +
        range_mom_coeff * df['range_momentum'] +
        open_close_coeff * df['open_close_movement'] +
        open_low_coeff * df['open_low_movement']
    )
    
    return df['alpha_factor']
```
