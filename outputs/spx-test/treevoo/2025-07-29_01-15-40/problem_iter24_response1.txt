```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    volume_weighted_high_low_ratio = (high_low_ratio.rolling(window=rolling_window).apply(
        lambda x: np.average(x, weights=df.loc[x.index, 'volume']), raw=False))
    
    # Apply Exponential Smoothing to the Volume-Weighted High-Low Ratios
    smoothing_factor = 0.2
    volume_weighted_smoothed = volume_weighted_high_low_ratio.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Daily Close Price Momentum
    close_momentum_window = 5
    close_price_momentum = df['close'].pct_change(periods=close_momentum_window)
    
    # Combine the two components
    alpha_factor = (volume_weighted_smoothed + close_price_momentum) / 2
    
    return alpha_factor
```
