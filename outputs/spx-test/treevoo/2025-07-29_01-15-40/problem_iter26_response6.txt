```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    N = 5  # Number of days for averaging intraday high-low spread
    M = 5  # Number of days for averaging daily range
    P = 5  # Rolling window for price volatility
    coeffs = [0.3, 0.2, 0.1, 0.2, 0.1, 0.1]  # Coefficients for combining factors

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=N).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(
        df['intraday_high_low_spread'] > df['avg_intraday_high_low_spread'], 
        1, -1)
    
    # Adjust by Volume
    df['high_low_momentum_score'] *= df['volume']

    # Calculate Price Momentum over the last 5 days
    df['price_momentum'] = df['close'] - df['close'].shift(5)

    # Calculate Volume Trend over the last 5 days
    df['volume_trend'] = df['volume'] - df['volume'].shift(5)

    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=M).mean()
    
    # Calculate Range Momentum
    df['range_momentum_score'] = np.where(
        df['daily_range'] > df['avg_daily_range'], 
        1, -1)
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=5).sum()

    # Calculate Intraday Price Volatility
    df['intraday_prices'] = (df[['open', 'high', 'low', 'close']].mean(axis=1))
    df['intraday_price_volatility'] = df['intraday_prices'].rolling(window=P).std()
    df['avg_intraday_price_volatility'] = df['intraday_price_volatility'].rolling(window=P).mean()
    
    # Compute Price Volatility Score
    df['price_volatility_score'] = np.where(
        df['intraday_price_volatility'] > df['avg_intraday_price_volatility'], 
        1, -1)
    
    # Adjust by Volume
    df['price_volatility_score'] *= df['volume']

    # Combine Factors
    df['factor'] = (
        coeffs[0] * df['high_low_momentum_score'] + 
        coeffs[1] * df['price_momentum'] + 
        coeffs[2] * df['volume_trend'] + 
        coeffs[3] * df['range_momentum_score'] + 
        coeffs[4] * df['open_close_movement'] + 
        coeffs[5] * df['price_volatility_score']
    )

    return df['factor']
```
