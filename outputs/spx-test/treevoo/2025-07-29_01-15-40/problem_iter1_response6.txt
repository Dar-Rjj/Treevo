```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    df['5D_Momentum'] = df['close'].diff(5)
    
    # Calculate 20-Day Volume Moving Average
    df['20D_Volume_MA'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    df['Volume_Deviation'] = df['volume'] - df['20D_Volume_MA']
    
    # Calculate Momentum and Volume Synchronicity
    df['Momentum_Volume_Sync'] = df['5D_Momentum'] * df['Volume_Deviation']
    
    # Apply Exponential Smoothing
    df['Alpha_Factor'] = df['Momentum_Volume_Sync'].ewm(span=10, adjust=False).mean()
    
    return df['Alpha_Factor']
```
