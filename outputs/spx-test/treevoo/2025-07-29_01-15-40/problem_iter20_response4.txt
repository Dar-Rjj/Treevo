```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, m=10, alpha=0.2):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(df['high_low_spread'] > df['avg_high_low_spread'], 1, -1)
    df['volume_adjusted_high_low_momentum'] = df['high_low_momentum_score'] * df['volume']
    
    # Calculate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Calculate Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(n)
    df['volume_trend'] = df['volume_change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    
    # Calculate Range Momentum
    df['range_momentum_score'] = np.where(df['daily_range'] > df['avg_daily_range'], 1, -1)
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=n).sum()
    
    df['open_low_diff'] = df['open'] - df['low']
    df['open_low_movement'] = df['open_low_diff'].rolling(window=n).sum()
    
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Volume-Weighted Rolling Average of High-Low Ratios
    df['weighted_high_low_ratio'] = (df['high_low_ratio'] * df['volume']).rolling(window=m).sum() / df['volume'].rolling(window=m).sum()
    
    # Smooth the Volume-Weighted Mean
    df['smoothed_weighted_high_low_ratio'] = df['weighted_high_low_ratio'].ewm(alpha=alpha).mean()
    
    # Combine Factors
    df['alpha_factor'] = (
        0.3 * df['volume_adjusted_high_low_momentum'] +
        0.2 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.1 * df['range_momentum_score'] +
        0.15 * df['open_close_movement'] +
        0.15 * df['open_low_movement'] +
        0.05 * df['smoothed_weighted_high_low_ratio']
    )
    
    return df['alpha_factor']
```
