```python
import pandas as pd

def heuristics_v2(df):
    # Determine the period for momentum and volume trend
    period = 5
    
    # Calculate Price Momentum
    price_changes = df['close'].diff(period)
    price_momentum = price_changes.rolling(window=period).sum()
    
    # Calculate Volume Trend
    volume_changes = df['volume'].diff(period)
    volume_trend = volume_changes.rolling(window=period).sum()
    
    # Combine Price Momentum and Volume Trend with weights
    price_coefficient = 0.7
    volume_coefficient = 0.3
    weighted_sum = (price_momentum * price_coefficient) + (volume_trend * volume_coefficient)
    
    # Adjust for market sentiment using high and low prices
    high_low_range = df['high'] - df['low']
    high_low_adjustment = high_low_range.rolling(window=period).mean()
    
    # Final factor
    final_factor = weighted_sum + high_low_adjustment
    
    return final_factor
```
