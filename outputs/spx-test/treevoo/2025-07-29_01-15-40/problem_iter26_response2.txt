```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the difference between high and low for each day (Daily Range)
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the percentage of the daily range that the close is above the low
    df['close_above_low'] = (df['close'] - df['low']) / df['daily_range']
    
    # Calculate the volume weighted by the daily range
    df['range_weighted_volume'] = df['volume'] * df['daily_range']
    
    # Calculate the factor as a combination of the above metrics
    factor = (
        0.4 * df['close_change'] + 
        0.3 * df['close_above_low'] + 
        0.3 * (df['range_weighted_volume'] / df['volume'].rolling(window=20).mean())
    )
    
    return factor
```
