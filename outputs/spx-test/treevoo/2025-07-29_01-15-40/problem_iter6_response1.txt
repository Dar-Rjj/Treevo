```python
import pandas as pd

def heuristics_v2(df, n=5, hl_coeff=0.5, price_coeff=0.3, volume_coeff=0.2):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = 0
    df.loc[df['intraday_high_low_spread'] > df['avg_intraday_high_low_spread'], 'high_low_momentum_score'] = 1
    df.loc[df['intraday_high_low_spread'] < df['avg_intraday_high_low_spread'], 'high_low_momentum_score'] = -1
    
    # Adjust by Volume
    df['high_low_momentum_score'] *= df['volume']
    
    # Calculate Price Momentum
    df['price_change'] = df['close'].pct_change(periods=n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Calculate Volume Trend
    df['volume_change'] = df['volume'].pct_change(periods=n)
    df['volume_trend'] = df['volume_change'].rolling(window=n).sum()
    
    # Combine High-Low Momentum, Price Momentum, and Volume Trend
    df['market_sentiment'] = (df['high'] - df['low']) / df['close']
    df['weighted_sum'] = (hl_coeff * df['high_low_momentum_score']) + (price_coeff * df['price_momentum']) + (volume_coeff * df['volume_trend'])
    
    # Incorporate Open-Price Change
    df['open_price_change'] = df['open'].pct_change(periods=n)
    df['open_price_momentum'] = df['open_price_change'].rolling(window=n).sum()
    
    # Calculate Final Factor
    df['final_factor'] = df['weighted_sum'] + df['open_price_momentum']
    
    return df['final_factor']
```
