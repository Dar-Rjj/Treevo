import pandas as pd

def heuristics_v2(df):
    # Compute Upward and Downward Movements
    df['Upward'] = df['high'].diff().apply(lambda x: max(x, 0))
    df['Downward'] = df['low'].diff().apply(lambda x: min(x, 0))

    # Determine Net Movement
    def net_movement(row):
        if row['Upward'] > abs(row['Downward']):
            return row['Upward']
        else:
            return row['Downward']

    df['NetMovement'] = df.apply(net_movement, axis=1)

    # Calculate CDMI by summing up the NetMovements over a period (e.g., 14 days)
    df['CDMI'] = df['NetMovement'].rolling(window=14).sum()

    # Return the CDMI factor
    return df['CDMI']
