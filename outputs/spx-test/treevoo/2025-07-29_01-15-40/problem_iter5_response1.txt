```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']

    # Calculate Rolling Average of High-Low Ratios (10 days)
    df['Rolling_High_Low_Ratio'] = df['High_Low_Ratio'].rolling(window=10).mean()

    # Price Trend Component
    df['10_day_MA_Close'] = df['close'].rolling(window=10).mean()
    df['50_day_MA_Close'] = df['close'].rolling(window=50).mean()
    df['Price_Trend'] = df['10_day_MA_Close'] - df['50_day_MA_Close']

    # Volume Trend Component
    df['10_day_Sum_Volume'] = df['volume'].rolling(window=10).sum()
    df['50_day_Sum_Volume'] = df['volume'].rolling(window=50).sum()
    df['Volume_Trend'] = df['10_day_Sum_Volume'] - df['50_day_Sum_Volume']

    # Combined Indicator
    df['Combined_Indicator'] = df['Price_Trend'] * df['Volume_Trend']

    # Adjust for Volatility
    df['20_day_SD_Close'] = df['close'].rolling(window=20).std()
    df['Adjusted_Combined_Indicator'] = df['Combined_Indicator'] / df['20_day_SD_Close']

    # Synthesize Indicators
    df['Final_Indicator'] = 0.5 * df['Rolling_High_Low_Ratio'] + 0.5 * df['Adjusted_Combined_Indicator']

    # Drop rows with NaN values
    df.dropna(inplace=True)

    return df['Final_Indicator']
```
