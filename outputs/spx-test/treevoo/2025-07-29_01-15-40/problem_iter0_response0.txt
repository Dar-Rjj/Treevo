```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: Series of the alpha factor values indexed by date
    """
    # Calculate close-to-open return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume']

    return volume_weighted_return
```
