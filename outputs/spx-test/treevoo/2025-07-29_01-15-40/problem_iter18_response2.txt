import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']

    # Determine Net Intraday and Directional Movements
    df['upward'] = (df['high'] - df.shift(1)['close']).apply(lambda x: max(x, 0))
    df['downward'] = (df.shift(1)['close'] - df['low']).apply(lambda x: max(x, 0))
    df['net_movement'] = df.apply(lambda row: row['upward'] if row['upward'] > row['downward'] else -row['downward'], axis=1)

    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] / df.shift(1)['volume']
    df['combined_momentum'] = (df['range'] + df['spread'] + df['net_movement']) * (df['volume'] / df['volume'].mean())

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Calculate Rolling Average Intraday High-Low Spread
    df['rolling_avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=21).mean()

    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = df.apply(lambda row: 1 if row['intraday_high_low_spread'] > row['rolling_avg_intraday_high_low_spread'] else -1, axis=1)
    df['high_low_momentum_score'] *= df['volume']

    # Calculate Log Return
    df['log_return'] = (df['close'] / df.shift(1)['close']).apply(np.log)

    # EWMA of Log Return
    df['ewma_log_return'] = df['log_return'].ewm(span=21, min_periods=21).mean()

    # Integrate Price and Volume
    df['integrated_momentum'] = df['ewma_log_return'] * df['volume_change']

    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df.shift(1)['close']
    df['downward_movement'] = df.shift(1)['close'] - df['low']

    # Determine Net Directional Movement
    df['net_directional_movement'] = df.apply(lambda row: 1 if row['upward_movement'] > row['downward_movement'] else (-1 if row['upward_movement'] < row['downward_movement'] else 0), axis=1)

    # Calculate Cumulative Directional Movement
    df['cumulative_directional_movement'] = df['net_directional_movement'].rolling(window=21).sum() * df['volume']

    # Add Trend Strength Indicator
    df['daily_range'] = df['high'] - df['low']
    df['rolling_avg_daily_range'] = df['daily_range'].rolling(window=21).mean()
    df['trend_strength_indicator'] = (df['daily_range'] > df['rolling_avg_daily_range']).astype(int)

    # Integrate Volume into Momentum
    df['vwap'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])
    df['sma_vwap'] = (df['vwap'] + df['close'].rolling(window=21).mean()) / 2

    # Composite Factor Construction
    df['composite_factor'] = (df['high_low_momentum_score'] * df['integrated_momentum'] * df['sma_vwap'] + 
                              df['combined_momentum'] + df['cumulative_directional_movement']) * df['trend_strength_indicator']

    # Add a Volatility Component
    df['volatility'] = df['log_return'].rolling(window=21).std()
    df['final_factor'] = df['composite_factor'] / df['volatility']

    return df['final_factor']
