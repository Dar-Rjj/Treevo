import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Enhanced Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    smoothing_factor = 0.2
    additional_smoothing_factor = 0.1
    
    # Compute the Enhanced Volume-Weighted Mean
    def enhanced_volume_weighted_mean(high_low_ratios, volumes, smoothing_factor):
        weights = (volumes / volumes.sum()) * smoothing_factor
        return (high_low_ratios * weights).sum() / weights.sum()
    
    df['enhanced_vol_weighted_mean'] = df['high_low_ratio'].rolling(window=rolling_window).apply(
        lambda x: enhanced_volume_weighted_mean(x, df.loc[x.index, 'volume'], smoothing_factor),
        raw=False
    )
    
    # Apply Additional Smoothing
    df['smoothed_vol_weighted_mean'] = df['enhanced_vol_weighted_mean'].ewm(alpha=additional_smoothing_factor).mean()
    
    # Calculate Daily Open-Close Ratio
    df['open_close_ratio'] = df['open'] / df['close']
    
    # Incorporate Open-Close Ratio into Overall Momentum Score
    df['combined_ratio'] = (df['high_low_ratio'] + df['open_close_ratio']) / 2
    df['volume_weighted_combined_ratio'] = df['combined_ratio'] * df['volume']
    
    # Normalize by Volume (optional but not allowed as per the instruction)
    # df['normalized_combined_ratio'] = df['volume_weighted_combined_ratio'] / df['volume']
    
    # Return the factor values
    return df['volume_weighted_combined_ratio']
