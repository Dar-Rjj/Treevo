```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Compute Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    df['Volume_Weighted_HL_Ratio'] = (df['High_Low_Ratio'].rolling(window=rolling_window).apply(
        lambda x: np.average(x, weights=df.loc[x.index, 'volume']), raw=False
    ))
    smoothing_factor = 0.2
    df['Smoothed_Volume_Weighted_HL_Ratio'] = df['Volume_Weighted_HL_Ratio'].ewm(alpha=smoothing_factor).mean()
    
    # Calculate High-Low Momentum
    n_days = 5
    df['Avg_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=n_days).mean()
    df['High_Low_Momentum'] = np.where(df['Intraday_High_Low_Spread'] > df['Avg_Intraday_High_Low_Spread'], 
                                       1, -1) * df['volume']
    
    # Calculate Price Momentum
    price_momentum_period = 5
    df['Price_Change'] = df['close'].diff(periods=price_momentum_period)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=price_momentum_period).sum()
    
    # Enhance Market Sentiment Analysis
    m_days = 5
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=m_days).mean()
    df['Range_Momentum'] = np.where(df['Daily_Range'] > df['Avg_Daily_Range'], 1, -1)
    
    df['Open_Close_Price_Diff'] = df['close'] - df['open']
    df['Open_Close_Price_Sum'] = df['Open_Close_Price_Diff'].rolling(window=m_days).sum()
    
    # Incorporate Intraday Volatility
    k_days = 5
    df['Intraday_Volatility'] = 0.5 * np.log(df['high'] / df['low'])**2 - (2 * np.log(2) - 1) * np.log(df['close'] / df['open'])**2
    df['Avg_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=k_days).mean()
    df['Intraday_Volatility_Score'] = np.where(df['Intraday_Volatility'] > df['Avg_Intraday_Volatility'], 1, -1)
    
    # Combine Factors
    hl_momentum_coeff = 0.3
    price_momentum_coeff = 0.2
    volume_weighted_hl_ratio_coeff = 0.1
    range_momentum_coeff = 0.1
    open_close_price_sum_coeff = 0.1
    intraday_volatility_score_coeff = 0.2
    
    df['Alpha_Factor'] = (
        hl_momentum_coeff * df['High_Low_Momentum'] +
        price_momentum_coeff * df['Price_Momentum'] +
        volume_weighted_hl_ratio_coeff * df['Smoothed_Volume_Weighted_HL_Ratio'] +
        range_momentum_coeff * df['Range_Momentum'] +
        open_close_price_sum_coeff * df['Open_Close_Price_Sum'] +
        intraday_volatility_score_coeff * df['Intraday_Volatility_Score']
    )
    
    return df['Alpha_Factor']
```
