```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return
    df['ewma_log_return'] = df['log_return'].ewm(span=21, min_periods=21).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combined Momentum
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']
    df['high_minus_prior_high'] = df['high'] - df['high'].shift(1)
    df['low_minus_prior_low'] = df['low'] - df['low'].shift(1)
    
    # Determine Net Intraday and Directional Movements
    df['net_intraday_movement'] = np.where(df['high_minus_prior_high'] > df['low_minus_prior_low'],
                                           df['high_minus_prior_high'],
                                           df['low_minus_prior_low'])
    
    # Combine Intraday Movements with Net Movements
    df['combined_intraday_movements'] = (df['range'] + df['spread'] + df['net_intraday_movement']) * df['volume'].rolling(window=21).mean()
    
    # Adjust for Volume Activity
    df['adjusted_combined_momentum'] = np.where(df['volume'] > df['volume'].shift(1),
                                                df['combined_momentum'] * (df['volume'] / df['volume'].shift(1)),
                                                df['combined_momentum'] * (df['volume'].shift(1) / df['volume']))
    
    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    conditions = [
        (df['upward_movement'] > df['downward_movement']),
        (df['upward_movement'] < df['downward_movement'])
    ]
    choices = [1, -1]
    df['net_directional_movement'] = np.select(conditions, choices, default=0)
    
    # Calculate Adjusted Cumulative Directional Movement
    df['adjusted_cumulative_directional_movement'] = df['net_directional_movement'] * df['volume']
    df['cumulative_adjusted_directional_movement'] = df['adjusted_cumulative_directional_movement'].cumsum()
    
    # Integrated Factor
    df['integrated_factor'] = df['adjusted_combined_momentum'] + df['cumulative_adjusted_directional_movement']
    
    # CIMVDMI
    df['cimvdm'] = df['integrated_factor'].cumsum()
    
    return df['cimvdm']
```
