import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # EWMA of Log Return with 21-day lookback period
    df['ewma_log_return'] = df['log_return'].ewm(span=21).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combined Momentum (Multiply EWMA Log Return by Volume Change)
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Price Momentum over a 5-day period
    df['price_change'] = df['close'] - df['close'].shift(5)
    df['price_momentum'] = df['price_change'].rolling(window=5).sum()
    
    # Calculate Volume Trend over a 5-day period
    df['volume_trend'] = df['volume'] - df['volume'].shift(5)
    df['volume_trend'] = df['volume_trend'].rolling(window=5).sum()
    
    # Compute Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] + df['open']) / 2 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()
    
    # Combine SMA and VWAP
    df['sma_close'] = df['close'].rolling(window=5).mean()
    df['integrated_volume'] = (df['sma_close'] + df['vwap']) / 2
    
    # Calculate Price Reversal Indicator
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['close'] - df['open']
    df['price_reversal'] = df['high_low_spread'] - df['open_close_spread']
    
    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    df['net_directional_movement'] = np.where(
        df['upward_movement'] > df['downward_movement'], 1,
        np.where(df['upward_movement'] < df['downward_movement'], -1, 0)
    )
    
    # Calculate Cumulative Directional Movement Adjusted (CDMA)
    df['cdma'] = df['net_directional_movement'] * df['volume']
    df['cdma'] = df['cdma'].rolling(window=5).sum()
    
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios over 10 days
    df['rolling_high_low_ratio'] = df['high_low_ratio'].rolling(window=10).mean()
    
    # Incorporate Volume Weighted Factor
    df['volume_weight'] = df['volume'] * df['close']
    df['volume_weighted_high_low_ratio'] = (df['high_low_ratio'] * df['volume_weight']).rolling(window=10).sum() / df['volume_weight'].rolling(window=10).sum()
    
    # Volume-Weighted Price Reversal
    df['volume_weighted_price_reversal'] = df['price_reversal'] * df['volume']
    df['volume_weighted_price_reversal'] = df['volume_weighted_price_reversal'].rolling(window=5).sum()
    
    # Composite Factor Construction
    df['composite_factor'] = (
        df['price_momentum'] + df['volume_trend']
    ) * df['cdma'] + df['volume_weighted_price_reversal'] + df['combined_momentum']
    
    return df['composite_factor']
