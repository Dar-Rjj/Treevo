```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['HighLow_Ratio'] = df['high'] / df['low']
    
    # Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    df['Volume_Weighted_HighLow_Ratio'] = (df['HighLow_Ratio'] * df['volume']).rolling(window=rolling_window).sum() / df['volume'].rolling(window=rolling_window).sum()
    
    # Apply Exponential Smoothing to the Volume-Weighted High-Low Ratio
    smoothing_factor = 0.2
    df['Smoothed_Volume_Weighted_HighLow_Ratio'] = df['Volume_Weighted_HighLow_Ratio'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Daily Open-Close Ratio
    df['OpenClose_Ratio'] = df['close'] / df['open']
    
    # Combine both factors into a single alpha factor
    df['Alpha_Factor'] = df['Smoothed_Volume_Weighted_HighLow_Ratio'] * df['OpenClose_Ratio']
    
    # Return the alpha factor
    return df['Alpha_Factor']
```
