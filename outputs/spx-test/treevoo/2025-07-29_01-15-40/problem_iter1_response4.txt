```python
import pandas as pd

def heuristics_v2(df, N=50):
    # Calculate Simple Moving Average (SMA) of Close
    sma = df['close'].rolling(window=N).mean()
    
    # Calculate Price Reversal Indicator
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['open'] - df['close']
    price_reversal_indicator = high_low_spread - open_close_spread
    
    # Integrate Volume into Momentum
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    integrated_momentum = (sma + vwap) / 2
    
    # Composite Factor Construction
    final_factor = integrated_momentum * price_reversal_indicator
    
    return final_factor
```
