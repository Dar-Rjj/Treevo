```python
import pandas as pd

def heuristics_v2(df, n=5, m=5, k=5, high_low_coeff=0.3, price_mom_coeff=0.2, vol_trend_coeff=0.1, range_mom_coeff=0.15, open_close_coeff=0.1, intraday_vol_coeff=0.15):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=n).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = df.apply(lambda row: 1 if row['intraday_high_low_spread'] > row['avg_intraday_high_low_spread'] else -1, axis=1)
    df['high_low_momentum_score'] *= df['volume']
    
    # Calculate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_change'].rolling(window=n).sum()
    
    # Calculate Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(n)
    df['volume_trend'] = df['volume_change'].rolling(window=n).sum()
    
    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    df['range_momentum'] = df.apply(lambda row: 1 if row['daily_range'] > row['avg_daily_range'] else -1, axis=1)
    
    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=n).sum()
    
    # Incorporate Intraday Volatility
    df['intraday_volatility'] = 0.5 * (df['high'] - df['low'])**2 / (df['high'] + df['low'])
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=k).mean()
    df['intraday_volatility_score'] = df.apply(lambda row: 1 if row['intraday_volatility'] > row['avg_intraday_volatility'] else -1, axis=1)
    
    # Combine Factors
    df['alpha_factor'] = (
        high_low_coeff * df['high_low_momentum_score'] +
        price_mom_coeff * df['price_momentum'] +
        vol_trend_coeff * df['volume_trend'] +
        range_mom_coeff * df['range_momentum'] +
        open_close_coeff * df['open_close_movement'] +
        intraday_vol_coeff * df['intraday_volatility_score']
    )
    
    return df['alpha_factor']
```
