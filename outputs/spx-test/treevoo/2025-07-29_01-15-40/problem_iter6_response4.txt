```python
import pandas as pd

def heuristics_v2(df):
    # Compute Upward and Downward Movements
    df['Upward'] = df['high'] - df['close'].shift(1)
    df['Downward'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    conditions = [
        (df['Upward'] > df['Downward']),
        (df['Upward'] < df['Downward'])
    ]
    choices = [1, -1]
    df['Net_Directional_Movement'] = pd.np.select(conditions, choices, default=0)
    
    # Weight by Volume and Return
    df['Daily_Return'] = df['close'].pct_change()
    df['Weighted_Net_Directional_Movement'] = df['Net_Directional_Movement'] * df['volume'] * df['Daily_Return'].abs()
    
    # Calculate WCDMA over a period (e.g., 14 days)
    df['WCDMA'] = df['Weighted_Net_Directional_Movement'].rolling(window=14).sum()
    
    return df['WCDMA']
```
