```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, hl_coeff=0.3, price_mom_coeff=0.2, vol_trend_coeff=0.1, range_mom_coeff=0.2, open_price_change_coeff=0.2):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Calculate Average Intraday High-Low Spread over past N days
    df['avg_intraday_hl_spread'] = df['intraday_high_low_spread'].rolling(window=m).mean()

    # Compute High-Low Momentum Score
    df['hl_momentum_score'] = (df['intraday_high_low_spread'] - df['avg_intraday_hl_spread']) * df['volume']

    # Calculate Price Momentum over the past 5 days
    df['price_momentum'] = df['close'] - df['close'].shift(n)
    df['price_momentum'] = df['price_momentum'].rolling(window=n).sum()

    # Calculate Volume Trend over the past 5 days
    df['volume_trend'] = df['volume'] - df['volume'].shift(n)
    df['volume_trend'] = df['volume_trend'].rolling(window=n).sum()

    # Enhance Market Sentiment Analysis
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=m).mean()
    df['range_momentum'] = (df['daily_range'] - df['avg_daily_range'])

    # Integrate Open-Price and Close-Price Movement
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_movement'] = df['open_close_diff'].rolling(window=n).sum()

    # Incorporate Open-Price Change
    df['open_price_change'] = df['open'] - df['open'].shift(n)
    df['open_price_change'] = df['open_price_change'].rolling(window=n).sum()

    # Combine all factors using weighted sum
    df['alpha_factor'] = (
        hl_coeff * df['hl_momentum_score'] +
        price_mom_coeff * df['price_momentum'] +
        vol_trend_coeff * df['volume_trend'] +
        range_mom_coeff * df['range_momentum'] +
        open_price_change_coeff * df['open_price_change']
    )

    return df['alpha_factor']
```
