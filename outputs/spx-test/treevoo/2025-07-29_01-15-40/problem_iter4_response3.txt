```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return with 21-day Lookback Period
    df['ewma_log_return'] = df['log_return'].ewm(span=21, min_periods=21).mean()
    
    # Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combined Momentum
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    return df['combined_momentum']
```
