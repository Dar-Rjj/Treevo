```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over the last 5 days
    df['Avg_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=5).mean()
    
    # Compute High-Low Momentum Score
    df['High_Low_Momentum_Score'] = np.where(df['Intraday_High_Low_Spread'] > df['Avg_Intraday_High_Low_Spread'], 1, -1)
    
    # Adjust by Volume
    df['Adjusted_High_Low_Momentum_Score'] = df['High_Low_Momentum_Score'] * df['volume']
    
    # Calculate Price Momentum over the last 5 days
    df['Price_Change'] = df['close'] - df['close'].shift(5)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=5).sum()
    
    # Calculate Volume Trend over the last 5 days
    df['Volume_Change'] = df['volume'] - df['volume'].shift(5)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=5).sum()
    
    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    df['Avg_Daily_Range'] = df['Daily_Range'].rolling(window=5).mean()
    df['Range_Momentum'] = np.where(df['Daily_Range'] > df['Avg_Daily_Range'], 1, -1)
    
    # Integrate Open-Price and Close-Price Movement
    df['Open_Close_Price_Diff'] = df['close'] - df['open']
    df['Open_Price_Movement'] = df['Open_Close_Price_Diff'].rolling(window=5).sum()
    
    # Incorporate Intraday Volatility
    df['Intraday_Volatility'] = 0.5 * (df['high'] - df['low']).apply(lambda x: np.log(x**2))
    df['Avg_Intraday_Volatility'] = df['Intraday_Volatility'].rolling(window=5).mean()
    df['Intraday_Volatility_Score'] = np.where(df['Intraday_Volatility'] > df['Avg_Intraday_Volatility'], 1, -1)
    
    # Combine Factors with Coefficients
    df['Factor_Value'] = (
        0.2 * df['Adjusted_High_Low_Momentum_Score'] +
        0.2 * df['Price_Momentum'] +
        0.1 * df['Volume_Trend'] +
        0.2 * df['Range_Momentum'] +
        0.2 * df['Open_Price_Movement'] +
        0.1 * df['Intraday_Volatility_Score']
    )
    
    return df['Factor_Value']
```
