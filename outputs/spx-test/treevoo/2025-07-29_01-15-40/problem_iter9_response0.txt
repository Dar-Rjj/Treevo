```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    df['5DayMomentum'] = df['close'] - df['close'].shift(5)

    # Calculate Intraday High-Low Range
    df['IntradayRange'] = df['high'] - df['low']

    # Calculate Average Intraday High-Low Range (N=10)
    df['AvgIntradayRange'] = df['IntradayRange'].rolling(window=10).mean()

    # Compute Intraday High-Low Range Score
    df['IntradayRangeScore'] = df.apply(lambda row: 1 if row['IntradayRange'] > row['AvgIntradayRange'] else (-1 if row['IntradayRange'] < row['AvgIntradayRange'] else 0), axis=1)

    # Adjust Intraday High-Low Range Score by 5-Day Momentum
    df['AdjustedIntradayScore'] = df['IntradayRangeScore'] * df['5DayMomentum']

    # Calculate 20-Day Volume Moving Average
    df['20DayVolumeMA'] = df['volume'].rolling(window=20).mean()

    # Compute Volume Deviation from MA
    df['VolumeDeviation'] = df['volume'] - df['20DayVolumeMA']

    # Adjust by Volume
    df['VolumeAdjustedScore'] = df['AdjustedIntradayScore'] * df['VolumeDeviation']

    # Calculate 5-Day Close Price Standard Deviation
    df['5DayCloseStd'] = df['close'].rolling(window=5).std()

    # Adjust by Price Volatility
    df['FinalScore'] = df['VolumeAdjustedScore'] / df['5DayCloseStd']

    # Apply Weighted Moving Average Smoothing (weights: 0.4, 0.3, 0.2, 0.1)
    weights = [0.4, 0.3, 0.2, 0.1]
    df['AlphaFactor'] = df['FinalScore'].rolling(window=len(weights)).apply(lambda x: sum(w * v for w, v in zip(weights, x)), raw=False)

    return df['AlphaFactor']
```
