```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios (10 days)
    df['rolling_high_low_ratio'] = df['high_low_ratio'].rolling(window=10).mean()
    
    # Calculate 5-Day Momentum
    df['momentum_5d'] = df['close'] - df['close'].shift(5)
    
    # Calculate 20-Day Volume Moving Average
    df['volume_ma_20d'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Deviation from MA
    df['volume_deviation'] = df['volume'] - df['volume_ma_20d']
    
    # Combine Momentum and Volume Synchronicity
    df['synchronicity'] = df['momentum_5d'] * df['volume_deviation']
    
    # Apply Exponential Smoothing to Combined Synchronicity
    alpha = 0.2  # Smoothing factor
    df['smoothed_synchronicity'] = df['synchronicity'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Final Factor
    final_factor = (
        0.6 * df['smoothed_synchronicity'] + 
        0.4 * df['rolling_high_low_ratio'] * (df['high'] - df['low'])
    )
    
    return final_factor
```
