```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Volume-Weighted High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    df['volume_weighted_high_low_ratio'] = (df['high_low_ratio'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['smoothed_high_low_ratio'] = df['volume_weighted_high_low_ratio'].ewm(alpha=0.2, adjust=False).mean()
    
    # High-Low Momentum Score
    avg_high_low_spread = df['high_low_spread'].rolling(window=10).mean()
    df['high_low_momentum_score'] = (df['high_low_spread'] > avg_high_low_spread).astype(int) * 2 - 1
    df['high_low_momentum_score'] *= df['volume']
    
    # Price Momentum
    period = 5
    df['price_change'] = df['close'].diff(period)
    df['price_momentum'] = df['price_change'].rolling(window=period).sum()
    
    # Volume Trend
    df['volume_change'] = df['volume'].diff(period)
    df['volume_trend'] = df['volume_change'].rolling(window=period).sum()
    
    # Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Average Daily Range
    m = 10
    df['average_daily_range'] = df['daily_range'].rolling(window=m).mean()
    
    # Range Momentum
    df['range_momentum'] = (df['daily_range'] > df['average_daily_range']).astype(int) * 2 - 1
    
    # Open to Close Price Difference
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_momentum'] = df['open_close_diff'].rolling(window=period).sum()
    
    # Open to Low Price Difference
    df['open_low_diff'] = df['open'] - df['low']
    df['open_low_momentum'] = df['open_low_diff'].rolling(window=period).sum()
    
    # Relative Strength Index (RSI)
    n = 14
    df['gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['gain'].rolling(window=n).mean()
    df['avg_loss'] = df['loss'].rolling(window=n).mean()
    df['rs'] = df['avg_gain'] / df['avg_loss']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Combine Factors
    df['alpha_factor'] = (
        0.1 * df['smoothed_high_low_ratio'] +
        0.2 * df['high_low_momentum_score'] +
        0.2 * df['price_momentum'] +
        0.1 * df['volume_trend'] +
        0.1 * df['range_momentum'] +
        0.1 * df['open_close_momentum'] +
        0.1 * df['open_low_momentum'] +
        0.1 * df['rsi']
    )
    
    return df['alpha_factor'].dropna()
```
