def heuristics_v2(data, N=5):
    # Calculate Intraday Return
    data['intraday_return'] = (data['close'] - data['open']) / data['open']
    
    # Calculate True Range
    data['true_range'] = data.apply(
        lambda x: max(
            abs(x['high'] - x['low']),
            abs(x['high'] - data.shift(1).loc[x.name, 'close']),
            abs(data.shift(1).loc[x.name, 'close'] - x['low'])
        ), axis=1
    )
    
    # Compute Intraday Volatility Adjusted Return
    data['vol_adjusted_return'] = data['intraday_return'] / data['true_range']
    
    # Cumulative Momentum Over Multiple Days
    data['cumulative_vol_adj_return'] = data['vol_adjusted_return'].rolling(window=N).sum()
    
    return data['cumulative_vol_adj_return']
