def heuristics_v2(df):
    """
    Generate an alpha factor from a pandas DataFrame containing market data.
    
    Parameters:
    - df: A pandas DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    - A pandas Series of the generated alpha factor, indexed by (date).
    """
    # Calculate the intraday price movement
    intraday_movement = df['close'] - df['open']
    
    # Calculate the buying and selling pressure
    buying_pressure = df['high'] - df['close']
    selling_pressure = df['open'] - df['low']
    
    # Heuristic: Difference in buying and selling pressures adjusted by intraday movement
    alpha_factor = (buying_pressure - selling_pressure) * intraday_movement / (df['high'] - df['low'])
    
    return alpha_factor
