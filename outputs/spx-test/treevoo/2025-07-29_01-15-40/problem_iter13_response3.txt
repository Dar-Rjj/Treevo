```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return with 21-day lookback period
    df['ewma_log_return'] = df['log_return'].ewm(span=21, adjust=False).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combined Momentum
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    conditions = [
        (df['upward_movement'] > df['downward_movement']),
        (df['upward_movement'] < df['downward_movement'])
    ]
    choices = [1, -1]
    df['net_directional_movement'] = np.select(conditions, choices, default=0)
    
    # Calculate Adjusted Cumulative Directional Movement
    df['cumulative_directional_movement'] = df['net_directional_movement'].rolling(window=21).sum() * df['volume']
    
    # Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Average Daily Range over 21 Days
    df['average_daily_range'] = df['daily_range'].rolling(window=21).mean()
    
    # Trend Strength Indicator
    df['trend_strength'] = np.where(df['daily_range'] > df['average_daily_range'], 1, 0)
    
    # Integrated Factor
    df['integrated_factor'] = (df['combined_momentum'] + df['cumulative_directional_movement']) * df['trend_strength']
    
    return df['integrated_factor']
```
