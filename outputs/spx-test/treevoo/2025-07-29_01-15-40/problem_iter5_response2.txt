```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # EWMA of Log Return with 21-day Lookback Period
    df['ewma_log_return'] = df['log_return'].ewm(span=21, adjust=False).mean()
    
    # Integrate Price and Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Upward and Downward Movements
    df['upward_movement'] = df['high'] - df['close'].shift(1)
    df['downward_movement'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    conditions = [
        (df['upward_movement'] > df['downward_movement']),
        (df['upward_movement'] < df['downward_movement'])
    ]
    choices = [1, -1]
    df['net_directional_movement'] = np.select(conditions, choices, default=0)
    
    # Calculate Adjusted Cumulative Directional Movement
    df['cumulative_directional_movement'] = (df['net_directional_movement'].rolling(window=21).sum() * df['volume']).fillna(0)
    
    # Integrated Factor
    df['integrated_factor'] = df['combined_momentum'] + df['cumulative_directional_movement']
    
    return df['integrated_factor']
```
