    lwma_14 = df['close'].rolling(window=14).apply(lambda x: (x * range(1, 15)).sum() / sum(range(1, 15)), raw=True)
    lwma_28 = df['close'].rolling(window=28).apply(lambda x: (x * range(1, 29)).sum() / sum(range(1, 29)), raw=True)
