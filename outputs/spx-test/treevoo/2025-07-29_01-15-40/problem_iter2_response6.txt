```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread over past N days
    df['avg_intraday_high_low_spread'] = df['intraday_high_low_spread'].rolling(window=N).mean()
    
    # Compute High-Low Momentum Score
    high_low_momentum_score = (df['intraday_high_low_spread'] > df['avg_intraday_high_low_spread']).astype(int) - (df['intraday_high_low_spread'] < df['avg_intraday_high_low_spread']).astype(int)
    df['high_low_momentum_score'] = high_low_momentum_score * df['volume']
    
    # Calculate Simple Moving Average (SMA) of Close
    df['sma_close'] = df['close'].rolling(window=N).mean()
    
    # Calculate Price Reversal Indicator
    df['price_reversal_indicator'] = (df['high'] - df['low']) - (df['open'] - df['close'])
    
    # Integrate Volume into Momentum
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine SMA and VWAP
    df['integrated_momentum'] = (df['sma_close'] + df['vwap']) / 2
    
    # Composite Factor Construction
    df['composite_factor'] = df['high_low_momentum_score'] * df['integrated_momentum'] * df['price_reversal_indicator']
    
    return df['composite_factor']
```
