```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages of closing prices
    short_window = 10
    long_window = 50
    df['SMA_short'] = df['close'].rolling(window=short_window).mean()
    df['SMA_long'] = df['close'].rolling(window=long_window).mean()

    # Generate momentum signal
    df['momentum_signal'] = (df['SMA_short'] > df['SMA_long']).astype(int) - (df['SMA_short'] < df['SMA_long']).astype(int)

    # Calculate the percentage change in closing prices from one day to the next
    df['price_change_pct'] = df['close'].pct_change() * 100
    df['price_change_signal'] = (df['price_change_pct'] > 0).astype(int) - (df['price_change_pct'] < 0).astype(int)

    # Calculate the average volume over a recent period
    volume_window = 20
    df['avg_volume'] = df['volume'].rolling(window=volume_window).mean()

    # Identify days where the current volume is significantly higher or lower than the average volume
    threshold = 1.5
    df['volume_signal'] = ((df['volume'] / df['avg_volume']) > threshold).astype(int) - ((df['volume'] / df['avg_volume']) < 1/threshold).astype(int)

    # Combine price and volume signals to create a composite alpha factor
    df['composite_signal'] = df['momentum_signal'] + df['price_change_signal'] + df['volume_signal']

    return df['composite_signal']
```
