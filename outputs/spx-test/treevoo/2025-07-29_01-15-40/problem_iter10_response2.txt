```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Momentum
    df['5_Day_Momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Average Intraday High-Low Spread over past N days
    N = 20
    df['Avg_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=N).mean()
    
    # Compute High-Low Momentum Score
    df['High_Low_Momentum_Score'] = (df['Intraday_High_Low_Spread'] > df['Avg_Intraday_High_Low_Spread']).astype(int) * 2 - 1
    
    # Adjust High-Low Momentum Score by 5-Day Momentum
    df['Adjusted_HL_Momentum_Score'] = df['High_Low_Momentum_Score'] * df['5_Day_Momentum']
    
    # Compute Volume Deviation from 20-Day MA
    df['Volume_Deviation'] = df['volume'] - df['volume'].rolling(window=20).mean()
    
    # Adjust by Volume
    df['Volume_Adjusted_Score'] = df['Adjusted_HL_Momentum_Score'] * df['Volume_Deviation']
    
    # Calculate 5-Day Close Price Standard Deviation
    df['5_Day_Close_STD'] = df['close'].rolling(window=5).std()
    
    # Adjust by Price Volatility
    df['Price_Vol_Adjusted_Score'] = df['Volume_Adjusted_Score'] / df['5_Day_Close_STD']
    
    # Apply Exponential Smoothing
    alpha = 0.2
    df['Smoothed_Score'] = df['Price_Vol_Adjusted_Score'].ewm(alpha=alpha, adjust=False).mean()
    
    return df['Smoothed_Score']
```
