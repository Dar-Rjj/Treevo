```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    momentum_period = 5
    volume_period = 5
    price_coefficient = 0.7
    volume_coefficient = 0.3

    # Calculate Price Momentum
    df['price_change'] = df['close'].pct_change()
    df['price_momentum'] = df['price_change'].rolling(window=momentum_period).sum()

    # Calculate Volume Trend
    df['volume_change'] = df['volume'].pct_change()
    df['volume_trend'] = df['volume_change'].rolling(window=volume_period).sum()

    # Combine Price Momentum and Volume Trend
    df['weighted_price_momentum'] = df['price_momentum'] * price_coefficient
    df['weighted_volume_trend'] = df['volume_trend'] * volume_coefficient
    df['combined_factor'] = df['weighted_price_momentum'] + df['weighted_volume_trend']

    # Incorporate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['final_factor'] = df['combined_factor'] + df['high_low_range']

    return df['final_factor']
```
