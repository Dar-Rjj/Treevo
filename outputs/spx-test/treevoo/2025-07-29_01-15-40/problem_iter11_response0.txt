import pandas as pd

def heuristics_v2(df):
    # Calculate Log Return
    df['log_return'] = (df['close'] / df['close'].shift(1)).apply(np.log)
    
    # EWMA of Log Return with 21-day lookback period
    df['ewma_log_return'] = df['log_return'].ewm(span=21, min_periods=21).mean()
    
    # Integrate Price and Volume
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['combined_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Daily High-Low Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios with a 10-day window
    df['rolling_high_low_ratio'] = df['high_low_ratio'].rolling(window=10, min_periods=10).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate True Range
    df['true_range'] = df.apply(
        lambda x: max(abs(x['high'] - x['low']), abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), 
        axis=1
    )
    
    # Compute Intraday Volatility Adjusted Return
    df['intraday_vol_adjusted_return'] = df['intraday_return'] / df['true_range']
    
    # Integrate Intraday and Volume Information
    df['volume_relative_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['integrated_intraday_signal'] = df['intraday_vol_adjusted_return'] * df['volume_relative_change']
    
    # Combine Signals
    df['combined_signal'] = df['rolling_high_low_ratio'] * df['integrated_intraday_signal']
    
    # Integrated Factor
    df['integrated_factor'] = df['combined_momentum'] + df['combined_signal']
    
    return df['integrated_factor']
