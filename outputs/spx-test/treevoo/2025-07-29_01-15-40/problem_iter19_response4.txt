```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['HighLowRatio'] = df['high'] / df['low']
    
    # Calculate Rolling Average of High-Low Ratios (10 days)
    df['RollingAvgHighLowRatio'] = df['HighLowRatio'].rolling(window=10).mean()
    
    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = df['high'] - df['low']
    
    # Calculate Average Intraday High-Low Spread (10 days)
    df['AvgIntradayHighLowSpread'] = df['IntradayHighLowSpread'].rolling(window=10).mean()
    
    # Compute High-Low Momentum Score
    df['HighLowMomentumScore'] = (df['IntradayHighLowSpread'] - df['AvgIntradayHighLowSpread']) * df['volume']
    
    # Calculate Price Momentum (5 days)
    df['PriceChange'] = df['close'] - df['close'].shift(5)
    df['PriceMomentum'] = df['PriceChange'].rolling(window=5).sum()
    
    # Enhance Market Sentiment Analysis
    # Calculate Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate Average Daily Range (10 days)
    df['AvgDailyRange'] = df['DailyRange'].rolling(window=10).mean()
    
    # Calculate Range Momentum
    df['RangeMomentum'] = (df['DailyRange'] - df['AvgDailyRange'])
    
    # Integrate Open-Price and Close-Price Movement
    df['OpenCloseDiff'] = df['close'] - df['open']
    
    # Combine High-Low Momentum, Price Momentum, and Enhanced Market Sentiment
    alpha_factor = 0.4 * df['HighLowMomentumScore'] + 0.3 * df['PriceMomentum'] + 0.3 * df['RangeMomentum']
    
    return alpha_factor
```
