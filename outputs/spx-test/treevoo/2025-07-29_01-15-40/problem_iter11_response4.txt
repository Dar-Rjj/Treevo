```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movements and Directional Changes
    df['range'] = df['high'] - df['low']
    df['spread'] = df['close'] - df['open']
    
    # Determine Net Intraday and Directional Movements
    df['upward_move'] = (df['high'] - df['close'].shift(1)).fillna(0)
    df['downward_move'] = (df['low'] - df['close'].shift(1)).fillna(0)
    df['net_movement'] = np.where(df['upward_move'] > df['downward_move'], df['upward_move'], df['downward_move'])
    df['combined_momentum'] = (df['range'] + df['spread'] + df['net_movement']) * df['volume'].rolling(window=21).mean()
    
    # Adjust for Volume Activity
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['combined_momentum_adjusted'] = df['combined_momentum'] * df['volume_change']
    
    # Calculate Intraday High-Low Spread and Average
    df['high_low_spread'] = df['high'] - df['low']
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=21).mean()
    
    # Compute High-Low Momentum Score
    df['high_low_momentum_score'] = np.where(df['high_low_spread'] > df['avg_high_low_spread'], 1, -1) * df['volume']
    
    # Calculate Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # EWMA of Log Return
    df['ewma_log_return'] = df['log_return'].ewm(span=21).mean()
    
    # Integrate Price and Volume
    df['integrated_momentum'] = df['ewma_log_return'] * df['volume_change']
    
    # Calculate Upward and Downward Movements
    df['upward_move_close'] = df['high'] - df['close'].shift(1)
    df['downward_move_close'] = df['low'] - df['close'].shift(1)
    
    # Determine Net Directional Movement
    df['net_directional_movement'] = np.where(df['upward_move_close'] > df['downward_move_close'], 1, 
                                              np.where(df['upward_move_close'] < df['downward_move_close'], -1, 0))
    
    # Calculate Adjusted Cumulative Directional Movement
    df['cumulative_directional_movement'] = df['net_directional_movement'].rolling(window=21).sum() * df['volume']
    
    # Add Trend Strength Indicator
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=21).mean()
    df['trend_strength'] = np.where(df['daily_range'] > df['avg_daily_range'], 1, 0)
    
    # Integrate Volume into Momentum
    df['vwap'] = (df['close'] * df['volume']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    df['integrated_vwap'] = df['vwap'] * df['trend_strength']
    
    # Composite Factor Construction
    df['composite_factor'] = (df['high_low_momentum_score'] * df['integrated_momentum'] * df['cumulative_directional_movement']) * df['trend_strength']
    
    return df['composite_factor']
```
