```python
import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df):
    # Calculate Daily High-Low Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Volume-Weighted Rolling Average of High-Low Ratios
    rolling_window = 10
    df['Volume_Weighted_High_Low_Ratio'] = (df['High_Low_Ratio'] * df['volume']).rolling(window=rolling_window).mean() / df['volume'].rolling(window=rolling_window).mean()
    df['Smoothed_Volume_Weighted_High_Low_Ratio'] = df['Volume_Weighted_High_Low_Ratio'].ewm(alpha=0.2, adjust=False).mean()
    
    # Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    n_days = 5
    df['Average_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=n_days).mean()
    
    # High-Low Momentum Score
    df['High_Low_Momentum_Score'] = np.where(df['Intraday_High_Low_Spread'] > df['Average_Intraday_High_Low_Spread'], 1, -1)
    df['High_Low_Momentum_Score'] *= df['volume']
    
    # Price Momentum
    df['Price_Change'] = df['close'].pct_change(periods=5)
    df['Price_Momentum'] = df['Price_Change'].rolling(window=5).sum()
    
    # Volume Trend
    df['Volume_Change'] = df['volume'].pct_change(periods=5)
    df['Volume_Trend'] = df['Volume_Change'].rolling(window=5).sum()
    
    # Enhance Market Sentiment Analysis
    df['Daily_Range'] = df['high'] - df['low']
    m_days = 10
    df['Average_Daily_Range'] = df['Daily_Range'].rolling(window=m_days).mean()
    df['Range_Momentum'] = np.where(df['Daily_Range'] > df['Average_Daily_Range'], 1, -1)
    
    # Open-Price and Close-Price Movement
    df['Open_Close_Price_Diff'] = df['close'] - df['open']
    df['Open_Close_Price_Movement'] = df['Open_Close_Price_Diff'].rolling(window=5).sum()
    
    # Intraday Price Volatility
    p_days = 5
    df['Intraday_Prices'] = df[['open', 'high', 'low', 'close']].std(axis=1)
    df['Average_Intraday_Price_Volatility'] = df['Intraday_Prices'].rolling(window=p_days).mean()
    df['Price_Volatility_Score'] = np.where(df['Intraday_Prices'] > df['Average_Intraday_Price_Volatility'], 1, -1)
    df['Price_Volatility_Score'] *= df['volume']
    
    # Intraday Open-High Spread
    df['Intraday_Open_High_Spread'] = df['high'] - df['open']
    q_days = 5
    df['Average_Intraday_Open_High_Spread'] = df['Intraday_Open_High_Spread'].rolling(window=q_days).mean()
    df['Open_High_Momentum_Score'] = np.where(df['Intraday_Open_High_Spread'] > df['Average_Intraday_Open_High_Spread'], 1, -1)
    df['Open_High_Momentum_Score'] *= df['volume']
    
    # Combine Factors
    coefficients = {
        'Smoothed_Volume_Weighted_High_Low_Ratio': 0.3,
        'High_Low_Momentum_Score': 0.2,
        'Price_Momentum': 0.1,
        'Volume_Trend': 0.1,
        'Range_Momentum': 0.1,
        'Open_Close_Price_Movement': 0.1,
        'Price_Volatility_Score': 0.1,
        'Open_High_Momentum_Score': 0.1
    }
    
    factor = 0
    for col, coef in coefficients.items():
        factor += df[col] * coef
    
    return factor
```
