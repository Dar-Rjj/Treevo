```python
def heuristics_v2(df):
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change from the previous day's close to the current day's close
    price_change = df['close'].diff()
    
    # Calculate the volume change from the previous day
    volume_change = df['volume'].diff()
    
    # Factor: If the price change is positive and the volume has increased, return the average price
    # Otherwise, return the negative of the average price
    factor_values = avg_price.where((price_change > 0) & (volume_change > 0), -avg_price)
    
    return factor_values
```
