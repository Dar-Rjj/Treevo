```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    n = 10  # Number of days for raw momentum
    m = 20  # Number of days for volume moving average
    threshold = 2.0  # Volume spike threshold
    past_days = 5  # Number of days for average volume calculation
    volatility_window = 10  # Number of days for standard deviation calculation
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['close'].diff()
    
    # Identify Volume Spike
    df['Volume_MA'] = df['volume'].rolling(window=m).mean()
    df['Volume_Spike'] = (df['volume'] > threshold * df['Volume_MA']).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (1 - 0.5 * df['Volume_Spike'])
    
    # Calculate Intraday High-Low Spread Factor
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Combined_Intraday_Factor'] = df['Intraday_Spread'] + df['Close_to_Open_Return']
    
    # Calculate Weighted Price Movement
    df['Weighted_Price_Movement'] = (df['close'] - df['open']) * df['volume']
    df['Average_Volume'] = df['volume'].rolling(window=past_days).mean()
    df['Weighted_Price_Movement_Norm'] = df['Weighted_Price_Movement'] / df['Average_Volume']
    
    # Calculate Enhanced Weighted Price Movement
    df['Enhanced_Weighted_Price_Movement'] = (df['close'] - df['open']) * df['volume'] * (df['high'] - df['low'])
    df['Enhanced_Weighted_Price_Movement_Norm'] = df['Enhanced_Weighted_Price_Movement'] / df['Average_Volume']
    
    # Final Alpha Factor
    df['Final_Alpha'] = (df['Adjusted_Momentum'] * df['Combined_Intraday_Factor'] * 
                         df['Enhanced_Weighted_Price_Movement_Norm'])
    
    # Adjust for Volatility
    df['Volatility'] = df['close'].rolling(window=volatility_window).std()
    df['Final_Alpha_Adjusted'] = df['Final_Alpha'] / (1 + df['Volatility'])
    
    return df['Final_Alpha_Adjusted']
```
