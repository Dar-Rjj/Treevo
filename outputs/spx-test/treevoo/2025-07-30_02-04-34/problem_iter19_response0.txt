```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=30, volume_spike_threshold=1.5):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).mean()
    
    # Identify Volume Spikes
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > df['volume_ma'] * volume_spike_threshold).astype(int)
    
    # Adjust Momentum by Volume Spike
    momentum_dampening_factor = 0.5
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * (1 - momentum_dampening_factor))
    
    # Calculate Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).mean() / 5
    
    # Compute Volume Weight
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['average_volume']
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['combined_momentum'] = df['smoothed_price_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread Factor
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    df['intraday_factor'] = df['intraday_high_low_spread'] + df['close_to_open_return']
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['average_past_5_days_volume'] = df['volume'].rolling(window=5).mean()
    df['normalized_weighted_price_movement'] = df['weighted_price_movement'] / df['average_past_5_days_volume']
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'].diff()
    df['ema_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean().shift(1)
    df['close_price_trend'] = df['close_price_change'] - df['ema_close_price_change']
    
    # Introduce Reversal Indicator
    df['long_term_momentum'] = df['daily_price_change'].rolling(window=20).mean() / 20
    df['short_term_momentum'] = df['daily_price_change'].rolling(window=5).mean() / 5
    df['reversal_indicator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['adjusted_momentum'] * df['intraday_factor'] * 
                          df['normalized_weighted_price_movement'] * 
                          df['close_price_trend'] + 
                          df['reversal_indicator'])
    
    return df['alpha_factor']
```
