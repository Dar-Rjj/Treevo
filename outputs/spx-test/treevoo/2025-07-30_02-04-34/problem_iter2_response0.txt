import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']

    # Smooth Intraday Return using EMA
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['diff_intraday_return'] = df['intraday_return'] - df['ema_intraday_return'].shift(1)

    # Detect Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    avg_volume_change = df['volume_change'].rolling(window=20).mean()
    std_volume_change = df['volume_change'].rolling(window=20).std()
    df['volume_spike'] = (df['volume_change'] > 1.5 * avg_volume_change + 1.5 * std_volume_change).astype(int)

    # Combine Intraday Momentum and Volume Spike
    df['combined_factor'] = df['diff_intraday_return'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=3).mean().shift(1)

    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) - df['open']

    # Compute Next Day Open to Current Day Close
    df['next_day_open_to_close'] = df['open'].shift(-1) - df['close']

    # Final Alpha Signal
    df['alpha_signal'] = (
        df['smoothed_combined_factor'] * 
        df['intraday_momentum'] * 
        df['volume']
    )

    return df['alpha_signal'].dropna()
