```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Price Momentum
    df['return'] = (df['close'].shift(1) - df['close'].shift(n)) / df['close'].shift(n)
    
    # Compute Weighted Average Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Adjust Momentum by Volume
    df['vol_adjustment'] = df['volume'] / df['avg_volume']
    df['adjusted_momentum'] = df['return'] * df['vol_adjustment']
    
    return df['adjusted_momentum'].dropna()
```
