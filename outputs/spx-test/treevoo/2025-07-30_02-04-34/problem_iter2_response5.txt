```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread Factor
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    combined_intraday_factor = intraday_high_low_spread + close_to_open_return
    
    # Calculate Price Momentum
    n_days = 5
    price_momentum = (df['close'].shift(1) - df['close'].shift(n_days)) / df['close'].shift(n_days)
    
    # Adjust Momentum by Volume
    weighted_average_volume = df['volume'].rolling(window=n_days).mean()
    normalized_volume = df['volume'] / weighted_average_volume
    adjusted_price_momentum = price_momentum * normalized_volume
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume']
    average_volume = df['volume'].rolling(window=5).mean()
    normalized_weighted_price_movement = weighted_price_movement / average_volume
    
    # Final Alpha Factor
    final_alpha = combined_intraday_factor * normalized_weighted_price_movement * adjusted_price_momentum
    
    # Adjust for Volatility
    std_dev_close = df['close'].rolling(window=10).std()
    final_alpha = final_alpha / (1 + std_dev_close)
    
    return final_alpha
```
