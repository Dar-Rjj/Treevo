```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_momentum'] = df['open'] - df['high_low_diff']
    
    # Compute Next Day Open to Current Day Close
    df['next_day_open'] = df['open'].shift(-1)
    df['next_day_open_to_current_close'] = df['next_day_open'] - df['close']
    
    # Combine Both for Alpha Signal
    df['alpha_signal'] = (df['intraday_momentum'] * df['volume']) + df['next_day_open_to_current_close']
    
    # Drop intermediate columns
    df.drop(['high_low_diff', 'intraday_momentum', 'next_day_open'], axis=1, inplace=True)
    
    return df['alpha_signal']
```
