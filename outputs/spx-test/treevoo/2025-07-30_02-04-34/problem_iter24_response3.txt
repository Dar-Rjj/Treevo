```python
import pandas as pd

def heuristics_v2(df):
    # Short-term momentum (5-day)
    short_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-term momentum (20-day)
    long_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Momentum difference
    momentum_diff = short_momentum - long_momentum
    
    # Volume change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Average volume over the last 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Volume variance over the past 10 days
    volume_variance = df['volume'].rolling(window=10).var()
    
    # Normalized volume variance
    normalized_vol_var = volume_variance / df['volume']
    
    # Volume factor
    volume_factor = volume_change / avg_volume - normalized_vol_var
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Open-Close Spread
    open_close_spread = df['close'] - df['open']
    
    # Gap Up or Down
    gap_up_down = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Price action factor
    price_action_factor = open_close_spread / high_low_range + gap_up_down
    
    # Composite factor
    composite_factor = 0.4 * momentum_diff + 0.3 * volume_factor + 0.3 * price_action_factor
    
    return composite_factor
```
