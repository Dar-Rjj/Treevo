```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Smooth Intraday Return
    intraday_return_ema = intraday_return.ewm(span=7, adjust=False).mean().shift(1)
    
    # Calculate Intraday Reversal
    intraday_reversal = (df['close'] - df['open']) / df['open']
    
    # Smooth Intraday Reversal
    intraday_reversal_ema = intraday_reversal.ewm(span=7, adjust=False).mean().shift(1)
    
    # Detect Volume Spikes
    volume_change = df['volume'].pct_change()
    avg_volume_change = volume_change.rolling(window=30).mean()
    volume_spike = (volume_change > 2 * avg_volume_change).astype(int)
    
    # Incorporate Close Price Trend
    close_price_change = df['close'].pct_change()
    close_price_change_ema = close_price_change.ewm(span=7, adjust=False).mean().shift(1)
    
    # Combine Intraday Factors
    intraday_momentum_difference = intraday_return_ema - intraday_reversal_ema
    combined_intraday_factor = intraday_momentum_difference * volume_spike + close_price_change_ema
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate Raw Momentum
    n = 7
    raw_momentum = daily_price_change.rolling(window=n).sum() / n
    
    # Adjust Momentum by Volume Spike
    adjusted_momentum = raw_momentum * (1 - volume_spike * 0.1)
    
    # Calculate Smoothed Price Momentum
    smoothed_price_momentum = daily_price_change.rolling(window=5).sum() / 5
    
    # Compute Volume Weight
    average_volume = df['volume'].rolling(window=n).mean()
    normalized_volume = df['volume'] / average_volume
    volume_weight = normalized_volume.rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    base_momentum_signal = smoothed_price_momentum * volume_weight
    
    # Calculate Intraday High-Low Spread Factor
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    combined_intraday_high_low = high_low_spread + close_to_open_return
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume'] / average_volume
    
    # Final Alpha Factor
    final_alpha = (adjusted_momentum * combined_intraday_factor * weighted_price_movement + base_momentum_signal)
    
    # Adjust for Volatility
    std_dev_close = df['close'].rolling(window=10).std()
    final_alpha = final_alpha / (1 + std_dev_close)
    
    return final_alpha
```
