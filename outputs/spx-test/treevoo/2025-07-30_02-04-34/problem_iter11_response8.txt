```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of money flow
    mf_10_day_avg = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of volume
    vol_10_day_avg = df['volume'].rolling(window=10).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = mf_10_day_avg / vol_10_day_avg
    
    # Calculate the 10-day standard deviation of the close price
    close_std_10_day = df['close'].rolling(window=10).std()
    
    # Calculate the normalized money flow ratio
    normalized_money_flow_ratio = money_flow_ratio / close_std_10_day
    
    # Calculate the factor value
    factor = normalized_money_flow_ratio
    
    return factor
```
