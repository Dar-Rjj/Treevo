```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Raw Momentum
    raw_momentum = df['close'].diff()
    
    # Compute Weighted Average Volume
    weighted_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Adjust Price Momentum by Volume
    adjusted_volume = df['volume'] / weighted_avg_volume
    volume_adjusted_momentum = raw_momentum * adjusted_volume
    
    # Incorporate Volume Intensity
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_intensity = 1 / avg_volume_20
    intensity_adjusted_momentum = raw_momentum * volume_intensity
    
    # Detect Volume Spikes
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    volume_change = df['volume'] / moving_avg_volume
    volume_spike = (volume_change > 3) & (df['volume'] > moving_avg_volume)
    
    # Adjust Momentum by Volume Spike
    momentum_with_spike = raw_momentum.where(~volume_spike, other=raw_momentum * 0.7)
    
    # Combine Intraday Factors
    intraday_factor = intraday_high_low_spread + close_open_spread
    
    # Compute Next Day Open to Current Day Close
    next_day_open = df['open'].shift(-1)
    open_close_diff = df['close'] - next_day_open
    
    # Combine for Initial Alpha Signal
    initial_alpha_signal = open_close_diff * df['volume']
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Compute Smoothed Price Momentum
    smoothed_price_momentum = daily_price_change.rolling(window=7).mean()
    
    # Compute Weighted Volume Indicator
    weighted_volume_indicator = (volume_change * volume_change.sign()).rolling(window=7).mean()
    
    # Calculate Weighted Price Movement
    money_flow = (df['close'] - df['open']) * df['volume']
    avg_volume_7 = df['volume'].rolling(window=7).mean()
    weighted_price_movement = money_flow / avg_volume_7
    
    # Calculate Intraday Volatility
    intraday_range = df['high'] - df['low']
    intraday_volatility = intraday_range.rolling(window=7).mean()
    
    # Final Alpha Signal
    final_alpha_signal = (
        (momentum_with_spike * smoothed_price_momentum * weighted_volume_indicator) +
        initial_alpha_signal
    ) * intraday_factor * weighted_price_movement * intraday_volatility
    
    return final_alpha_signal
```
