```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 7-day Price Return
    df['price_return_7d'] = (df['close'].shift(1) - df['close'].shift(7)) / df['close'].shift(7)
    
    # Calculate 7-day Volume Change
    df['volume_change_7d'] = df['volume'].shift(1) - df['volume'].shift(7)
    
    # Adjust Momentum by Volume
    df['volume_adjusted_momentum'] = df['price_return_7d'] * df['volume_change_7d']
    
    # Apply Volume-Adjusted Momentum to Range
    df['enhanced_high_low_range'] = df['high_low_range'] * df['volume_adjusted_momentum']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = abs(df['volume'] - df['volume'].shift(1))
    
    # Normalize the Volume Shock
    df['volume_shock_normalized'] = df['daily_volume_change'] / df['volume'].rolling(window=5).mean()
    
    # Adjust Intraday Reversal
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_shock_normalized']
    
    # Combine with Enhanced High-Low Range Factor
    df['combined_factor'] = df['enhanced_high_low_range'] + df['adjusted_intraday_return']
    
    # Calculate Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['close'].diff().rolling(window=5).sum() / 5
    
    # Compute Weighted Volume Indicator
    df['volume_change_sign'] = df['volume'].diff() * df['volume'].diff().apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    df['weighted_volume_indicator'] = df['volume_change_sign'].rolling(window=5).mean()
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (df['smoothed_price_momentum'] * df['weighted_volume_indicator'] + 
                                df['combined_factor'])
    
    return df['final_alpha_signal']
```
