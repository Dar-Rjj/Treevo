import pandas as pd

def heuristics_v2(df, lookback_period=21):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate 7-day return (Price Momentum)
    df['close_lag_1'] = df['close'].shift(1)
    df['close_lag_7'] = df['close'].shift(7)
    df['momentum'] = (df['close_lag_1'] / df['close_lag_7']) - 1
    
    # Apply Momentum to High-Low Range
    df['weighted_high_low_range'] = df['high_low_range'] * df['momentum']
    
    # Sum the weighted high-low range over the lookback period
    df['factor'] = df['weighted_high_low_range'].rolling(window=lookback_period).sum()
    
    # Drop the intermediate columns
    df.drop(columns=['high_low_range', 'close_lag_1', 'close_lag_7', 'momentum', 'weighted_high_low_range'], inplace=True)
    
    return df['factor']
