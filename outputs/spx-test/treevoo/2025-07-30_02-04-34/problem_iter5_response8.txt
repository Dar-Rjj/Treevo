```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['avg_5d_volume'] = df['volume'].rolling(window=5).mean()
    df['momentum_vol_adjusted'] = (df['momentum_indicator'] * df['volume']) / df['avg_5d_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_20d_volume_change'] = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = (df['volume_change'] > 1.5 * df['avg_20d_volume_change']).astype(int)
    
    # Combine Factors
    df['volume_shock_normalized'] = df['volume'] / df['avg_5d_volume']
    df['alpha_factor'] = df['momentum_vol_adjusted'] * df['volume_spike'] + df['intraday_return'] * df['volume_shock_normalized']
    
    return df['alpha_factor']
```
