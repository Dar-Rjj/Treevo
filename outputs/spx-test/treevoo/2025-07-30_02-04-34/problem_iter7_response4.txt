```python
import pandas as pd

def heuristics_v2(df, n=10):
    """
    Calculate a price momentum adjusted by volume.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as (date)
    - n: Number of days to look back for the momentum calculation
    
    Returns:
    - A pandas Series with the factor values indexed by (date)
    """
    
    # Calculate Price Momentum
    close_prices = df['close']
    return_t_n = close_prices.pct_change(periods=n).shift(1)  # Compute the return from t-1 to t-n
    
    # Adjust Momentum by Volume
    volumes = df['volume']
    sum_of_volumes = volumes.rolling(window=n).sum()
    volume_adjusted_momentum = (return_t_n * volumes).div(sum_of_volumes)
    
    return volume_adjusted_momentum
```
