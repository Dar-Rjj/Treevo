import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Smooth Intraday Return using EMA with span 5
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_momentum_diff'] = df['intraday_return'] - df['ema_intraday_return']

    # Incorporate Volume Intensity
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['inverse_avg_volume'] = 1 / df['avg_volume_5d']
    df['intraday_return_vol_adj'] = df['intraday_return'] * df['inverse_avg_volume']

    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    avg_vol_change = df['volume_change'].rolling(window=20).mean()
    std_vol_change = df['volume_change'].rolling(window=20).std()
    df['volume_spike'] = (df['volume_change'] > 1.5 * avg_vol_change + std_vol_change).astype(int)

    # Combine Factors
    df['combined_factor'] = df['intraday_momentum_diff'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=3).mean()

    # Final factor
    df['final_factor'] = df['smoothed_combined_factor'] * df['inverse_avg_volume']

    # Return the final factor
    return df['final_factor'].dropna()
