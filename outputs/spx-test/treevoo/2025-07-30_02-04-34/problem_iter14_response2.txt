import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the percentage change in closing prices
    df['close_pct_change'] = df['close'].pct_change().fillna(0)
    
    # Calculate the 10-day rolling mean of the daily price range
    df['price_range_10d_avg'] = df['price_range'].rolling(window=10).mean()
    
    # Calculate the 5-day rolling standard deviation of the percentage change in closing prices
    df['close_pct_change_5d_std'] = df['close_pct_change'].rolling(window=5).std().fillna(0)
    
    # Calculate the factor as a combination of these features
    df['factor'] = (df['close_pct_change'] / df['close_pct_change_5d_std']) * df['price_range_10d_avg']
    
    return df['factor']
