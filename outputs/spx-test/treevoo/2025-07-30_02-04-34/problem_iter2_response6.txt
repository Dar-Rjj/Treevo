```python
import pandas as pd

def heuristics_v2(df, n=10, m=20, spike_threshold=2.0):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(1)
    
    # Identify Volume Spikes
    volume_ma = df['volume'].rolling(window=m).mean()
    volume_spike = (df['volume'] > (spike_threshold * volume_ma)).astype(int)
    
    # Adjust Momentum by Volume Spike
    adjusted_momentum = raw_momentum * (1 - 0.5 * volume_spike)  # Dampen momentum by 50% if there's a volume spike
    
    # Calculate Intraday High-Low Spread Factor
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    combined_intraday_factor = intraday_high_low_spread + close_to_open_return
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume']
    average_volume = df['volume'].rolling(window=5).mean()
    normalized_weighted_price_movement = weighted_price_movement / average_volume
    
    # Final Alpha Factor
    final_alpha = adjusted_momentum * combined_intraday_factor * normalized_weighted_price_movement
    
    # Adjust for Volatility
    volatility = df['close'].rolling(window=10).std()
    final_alpha = final_alpha / (1 + volatility)
    
    return final_alpha
```
