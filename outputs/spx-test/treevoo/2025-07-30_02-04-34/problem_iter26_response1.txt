import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: Alpha factor values indexed by (date).
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the relative position of the close price within the day's range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the 10-day moving average of the close price
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day rolling standard deviation of the close price
    df['close_std_30'] = df['close'].rolling(window=30).std()
    
    # Calculate the volume weighted by the close position
    df['volume_weighted'] = df['volume'] * df['close_position']
    
    # Calculate the 10-day moving average of the volume weighted
    df['volume_weighted_ma_10'] = df['volume_weighted'].rolling(window=10).mean()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (
        df['close_pct_change'] + 
        (df['close'] - df['close_ma_10']) / df['close_std_30'] +
        (df['volume_weighted'] - df['volume_weighted_ma_10']) / df['volume_weighted_ma_10']
    )
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
