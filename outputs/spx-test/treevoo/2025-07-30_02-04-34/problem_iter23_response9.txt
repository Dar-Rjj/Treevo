```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return with EMA (span=5)
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    
    # Calculate Difference between actual and smoothed return
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Calculate Average Volume over last 5 periods
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Multiply Intraday Momentum Difference by Inverse of Average Volume
    df['intraday_momentum_vol_adj'] = df['intraday_momentum_diff'] * (1 / df['avg_volume_5'])
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    avg_volume_change = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = (df['volume_change'] > 1.75 * avg_volume_change).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_intraday_momentum_vol'] = df['intraday_momentum_vol_adj'] * df['volume_spike']
    
    # Smooth the combined factor again, using simple moving average (window=4)
    df['smoothed_combined_intraday_momentum_vol'] = df['combined_intraday_momentum_vol'].rolling(window=4).mean().shift(1)
    
    # Multiply final smoothed factor by Inverse of Average Volume
    df['final_smoothed_intraday_momentum_vol'] = df['smoothed_combined_intraday_momentum_vol'] * (1 / df['avg_volume_5'])
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['volume_impact'] = (df['volume'] / df['avg_volume_5'])
    
    # Final Alpha Signal
    df['adjusted_price_momentum'] = df['momentum_indicator'] * df['volume_impact']
    df['smoothed_price_momentum'] = df['momentum_indicator'].ewm(span=5, adjust=False).mean()
    df['weighted_volume_indicator'] = df['volume_impact'].rolling(window=5).mean()
    df['alpha_factor'] = (df['adjusted_price_momentum'] * df['smoothed_price_momentum'] * df['weighted_volume_indicator']) + df['final_smoothed_intraday_momentum_vol']
    
    return df['alpha_factor']
```
