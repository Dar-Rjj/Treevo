import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Intraday_Momentum'] = df['open'] - df['High_Low_Diff']
    
    # Compute Next Day Open to Current Day Close
    df['Next_Day_Open'] = df['open'].shift(-1)
    df['Open_Close_Diff'] = df['Next_Day_Open'] - df['close']
    
    # Combine for Initial Alpha Signal
    df['Initial_Alpha_Signal'] = (df['Intraday_Momentum'] * df['volume']).diff().fillna(0)
    
    # Smooth Intraday Return
    df['Smoothed_Intraday_Return'] = df['Intraday_Momentum'].ewm(span=5, adjust=False).mean().shift(1)
    
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].diff()
    
    # Compute Smoothed Price Momentum
    df['Smoothed_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).sum() / 5
    
    # Compute Weighted Volume Indicator
    df['Weighted_Volume_Indicator'] = (df['Volume_Change'] * df['Volume_Change'].apply(lambda x: 1 if x > 0 else -1)).rolling(window=5).mean()
    
    # Identify Spike
    df['Volume_Spike'] = (df['Volume_Change'] > 1.5 * df['Volume_Change'].rolling(window=20).mean()).astype(int)
    
    # Incorporate Volume Intensity
    df['Average_Volume'] = df['volume'].rolling(window=10).mean()
    df['Inverse_Average_Volume'] = 1 / df['Average_Volume']
    df['Volume_Intensity_Adjusted'] = df['Intraday_Momentum'] * df['Inverse_Average_Volume']
    
    # Combine Factors
    df['Combined_Factor'] = df['Intraday_Momentum'] * df['Volume_Spike']
    df['SMA_Combined_Factor'] = df['Combined_Factor'].rolling(window=3).mean().shift(1)
    df['Final_Smoothed_Factor'] = df['SMA_Combined_Factor'] * df['Inverse_Average_Volume']
    
    # Final Alpha Signal
    df['Final_Alpha_Signal'] = df['Smoothed_Price_Momentum'] * df['Weighted_Volume_Indicator'] + df['Initial_Alpha_Signal']
    
    return df['Final_Alpha_Signal']
