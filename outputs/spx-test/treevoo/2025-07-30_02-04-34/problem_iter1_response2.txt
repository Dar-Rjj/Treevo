```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['price_change'].rolling(window=5).mean()
    
    # Compute Weighted Volume Indicator
    df['weighted_volume_indicator'] = (df['volume_change'] * df['volume_change'].apply(lambda x: 1 if x > 0 else -1)).rolling(window=5).mean()
    
    # Combine Indicators
    df['alpha_factor'] = 0 + (df['smoothed_price_momentum'] * df['weighted_volume_indicator'])
    
    return df['alpha_factor']
```
