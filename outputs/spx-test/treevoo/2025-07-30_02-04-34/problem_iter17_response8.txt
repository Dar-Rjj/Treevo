```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Smoothed Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5).mean().shift(1)
    
    # Incorporate Volume Intensity
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_intensity'] = 1 / df['average_volume']
    df['intraday_momentum_difference'] = df['intraday_return'] * df['volume_intensity']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change()
    df['avg_volume_change'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = (df['volume_change'] > 1.75 * df['avg_volume_change']).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_factor'] = df['intraday_momentum_difference'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=4).mean().shift(1)
    
    # Integrate Next Day Open to Current Day Close
    df['next_day_open'] = df['open'].shift(-1)
    df['next_day_open_to_current_close'] = df['close'] - df['next_day_open']
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (
        df['intraday_high_low_spread'] +
        df['close_open_spread'] +
        df['smoothed_intraday_return'] +
        df['smoothed_combined_factor'] +
        df['next_day_open_to_current_close']
    )
    df['final_alpha_signal'] = df['final_alpha_signal'].ewm(span=3).mean().shift(1)
    
    return df['final_alpha_signal']
```
