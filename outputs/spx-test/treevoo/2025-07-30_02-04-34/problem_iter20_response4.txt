```python
import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the provided DataFrame.
    The factor is a combination of intraday price momentum and volume-adjusted price change.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the intraday price momentum
    intraday_momentum = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume-adjusted price change
    volume_adjusted_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Combine the two components to form the alpha factor
    alpha_factor = (intraday_momentum + volume_adjusted_price_change).rolling(window=5).mean()
    
    return alpha_factor
```
