```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Price Momentum
    df['price_momentum'] = (df['close'].shift(1) - df['close'].shift(n)) / df['close'].shift(n)
    
    # Compute Weighted Average Volume and Amount
    weighted_avg_volume = df['volume'].rolling(window=n).mean()
    weighted_avg_amount = df['amount'].rolling(window=n).mean()
    
    # Adjust Volume and Amount
    adjusted_volume = df['volume'] - weighted_avg_volume
    adjusted_amount = df['amount'] - weighted_avg_amount
    
    # Compute the ratio of Adjusted Volume to Adjusted Amount
    adj_ratio = adjusted_volume / adjusted_amount
    
    # Multiply Price Momentum by Adjusted Volume and Amount
    df['adjusted_price_momentum'] = df['price_momentum'] * adj_ratio
    
    return df['adjusted_price_momentum']
```
