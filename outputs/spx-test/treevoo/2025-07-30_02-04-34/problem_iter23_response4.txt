```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return with EMA
    ema_intraday_return = intraday_return.ewm(span=5, adjust=False).mean().shift(1)
    intraday_momentum_diff = intraday_return - ema_intraday_return
    
    # Incorporate Volume Intensity
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_intensity = 1 / avg_volume_5
    intraday_momentum_vol_adj = intraday_momentum_diff * volume_intensity
    
    # Detect Volume Spikes
    volume_change = df['volume'].pct_change()
    avg_vol_change_25 = volume_change.rolling(window=25).mean()
    vol_spike = (volume_change > 1.75 * avg_vol_change_25).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    combined_factor = intraday_momentum_vol_adj * vol_spike
    combined_factor_smoothed = combined_factor.rolling(window=4).mean().shift(1)
    
    # Multiply final smoothed factor by Inverse of Average Volume
    combined_factor_final = combined_factor_smoothed * (1 / avg_volume_5)
    
    # Calculate Momentum Indicator
    momentum_indicator = intraday_high_low_spread / close_open_spread
    
    # Integrate Volume Impact
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_impact = (df['volume'] / avg_volume_5_days) * momentum_indicator
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate Raw Momentum
    raw_momentum = daily_price_change.rolling(window=5).sum() / 5
    
    # Adjust Momentum by Volume Spike
    adjusted_momentum = raw_momentum * (1 - 0.3 * vol_spike)
    
    # Compute Smoothed Price Momentum
    smoothed_price_momentum = daily_price_change.rolling(window=5).sum() / 5
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume'] / avg_volume_5_days
    
    # Incorporate Close Price Trend
    close_price_change = df['close'].diff()
    ema_close_price_change = close_price_change.ewm(span=5, adjust=False).mean().shift(1)
    smoothed_close_price_change = close_price_change - ema_close_price_change
    
    # Introduce Reversal Indicator
    long_term_momentum = daily_price_change.rolling(window=20).sum() / 20
    short_term_momentum = daily_price_change.rolling(window=5).sum() / 5
    reversal_indicator = short_term_momentum - long_term_momentum
    
    # Final Alpha Signal
    alpha_signal = (adjusted_momentum * smoothed_price_momentum * weighted_price_movement 
                    + combined_factor_final + smoothed_close_price_change + reversal_indicator)
    
    return alpha_signal
```
