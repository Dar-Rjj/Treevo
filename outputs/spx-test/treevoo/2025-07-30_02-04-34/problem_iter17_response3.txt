```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return using EMA with span 5
    smoothed_intraday_return = intraday_return.ewm(span=5, adjust=False).mean().shift(1)
    intraday_momentum_difference = intraday_return - smoothed_intraday_return
    
    # Incorporate Volume Intensity
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    inverse_avg_volume = 1 / avg_volume_5_days
    intraday_momentum_diff_volume_adj = intraday_momentum_difference * inverse_avg_volume
    
    # Detect Volume Spikes
    volume_change = df['volume'].pct_change()
    avg_vol_change_25_days = volume_change.rolling(window=25).mean()
    volume_spike = volume_change > 1.75 * avg_vol_change_25_days
    
    # Combine Intraday Momentum and Volume Impact
    combined_factor = intraday_momentum_difference * volume_spike
    combined_factor_smoothed = combined_factor.rolling(window=4).mean().shift(1)
    final_combined_factor = combined_factor_smoothed * inverse_avg_volume
    
    # Calculate Momentum Indicator
    momentum_indicator = intraday_high_low_spread / close_open_spread
    
    # Integrate Volume Impact
    avg_volume_last_5_days = df['volume'].rolling(window=5).mean()
    weighted_volume_indicator = df['volume'] / avg_volume_last_5_days
    integrated_momentum = momentum_indicator * weighted_volume_indicator
    
    # Final Alpha Signal
    adjusted_price_momentum = intraday_momentum_difference * smoothed_intraday_return
    final_alpha_signal = (adjusted_price_momentum * integrated_momentum) + final_combined_factor
    
    return final_alpha_signal
```
