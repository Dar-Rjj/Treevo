```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, long_term_momentum_days=20, volatility_threshold=0.05, reduction_factor=0.7):
    # Calculate Raw Momentum
    daily_price_change = df['close'].diff()
    raw_momentum = daily_price_change.rolling(window=n).mean()

    # Adjust Momentum by Volume Spike
    volume_spike = (df['volume'] > df['volume'].rolling(window=n).mean() * 2).astype(int)
    adjusted_momentum = raw_momentum * (1 - volume_spike * 0.3)  # Dampen momentum by 30% if volume spike

    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']

    # Combine Intraday Factors
    combined_intraday_factor = (intraday_high_low_spread + close_open_spread) * 0.5

    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume']
    average_volume = df['volume'].rolling(window=5).mean()
    weighted_price_movement_normalized = weighted_price_movement / average_volume

    # Introduce Reversal Indicator
    long_term_momentum = daily_price_change.rolling(window=long_term_momentum_days).mean()
    short_term_momentum = daily_price_change.rolling(window=5).mean()
    reversal_indicator = short_term_momentum - long_term_momentum

    # Final Alpha Factor
    final_alpha_factor = (
        adjusted_momentum * combined_intraday_factor * 
        weighted_price_movement_normalized + 
        df['close'].pct_change().rolling(window=5).mean() + 
        reversal_indicator
    )

    # Introduce Volatility Filter
    volatility = df['close'].rolling(window=long_term_momentum_days).std()
    final_alpha_factor = final_alpha_factor * (volatility < volatility_threshold) * reduction_factor + \
                         final_alpha_factor * (volatility >= volatility_threshold)

    return final_alpha_factor
```
