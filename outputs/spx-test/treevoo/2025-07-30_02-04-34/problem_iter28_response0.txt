```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=25, volume_spike_threshold=1.75):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Daily_Price_Change'].rolling(window=n).mean()
    
    # Identify Volume Spikes
    df['Volume_MA'] = df['volume'].rolling(window=m).mean()
    df['Volume_Spike'] = (df['volume'] > df['Volume_MA'] * volume_spike_threshold).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['Adjusted_Raw_Momentum'] = df['Raw_Momentum'] * (1 - 0.5 * df['Volume_Spike'])
    
    # Calculate Smoothed Price Momentum
    df['Smoothed_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Compute Volume Weight
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Weight'] = df['volume'] / df['Average_Volume']
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['Smoothed_Price_Momentum_Weighted'] = df['Smoothed_Price_Momentum'] * df['Volume_Weight']
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['Close_Open_Spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return
    df['Smoothed_Intraday_Return'] = df['Intraday_Return'].ewm(span=5, adjust=False).mean().shift(1)
    df['Intraday_Momentum_Difference'] = df['Intraday_Return'] - df['Smoothed_Intraday_Return']
    
    # Incorporate Volume Intensity
    df['Average_Volume_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_Intensity'] = 1 / df['Average_Volume_5']
    df['Intraday_Momentum_Difference_Adjusted'] = df['Intraday_Momentum_Difference'] * df['Volume_Intensity']
    
    # Detect Volume Spikes
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Spike'] = (df['Volume_Change'] > df['Volume_Change'].rolling(window=25).mean() * 1.75).astype(int)
    df['Combined_Intraday_Momentum'] = df['Intraday_Momentum_Difference_Adjusted'] * df['Volume_Spike']
    
    # Smooth the combined factor again
    df['Smoothed_Combined_Intraday_Momentum'] = df['Combined_Intraday_Momentum'].rolling(window=4).mean().shift(1)
    
    # Calculate Momentum Indicator
    df['Momentum_Indicator'] = df['Intraday_High_Low_Spread'] / df['Close_Open_Spread']
    
    # Integrate Volume Impact
    df['Average_Volume_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_Impact'] = (df['volume'] / df['Average_Volume_5']).replace([np.inf, -np.inf], 0)
    df['Momentum_Indicator_Weighted'] = df['Momentum_Indicator'] * df['Volume_Impact']
    
    # Final Alpha Signal
    df['Final_Alpha_Factor'] = (df['Adjusted_Raw_Momentum'] * df['Smoothed_Price_Momentum_Weighted'] + 
                                df['Smoothed_Combined_Intraday_Momentum'] + 
                                df['Momentum_Indicator_Weighted'])
    
    return df['Final_Alpha_Factor']
```
