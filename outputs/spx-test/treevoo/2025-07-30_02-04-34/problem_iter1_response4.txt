```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Momentum Indicator
    momentum_indicator = intraday_high_low_spread / close_open_spread
    
    # Calculate 5-Day Average Volume
    average_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Integrate Volume Impact
    volume_impact = (momentum_indicator * df['volume']) / average_volume_5d
    
    return volume_impact
```
