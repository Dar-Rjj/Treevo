```python
import pandas as pd

def heuristics_v2(df, m=10, reduction_factor=0.5, volume_spike_threshold=2.0):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'].pct_change()
    
    # Calculate the moving average of the volume over the last m days
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    
    # Determine if today's volume is a spike
    df['volume_spike'] = (df['volume'] / df['volume_ma']) > volume_spike_threshold
    
    # Adjust Momentum Using Volume Spike
    df['adjusted_momentum'] = df['raw_momentum']
    df.loc[df['volume_spike'], 'adjusted_momentum'] = df['raw_momentum'] * reduction_factor
    
    return df['adjusted_momentum']
```
