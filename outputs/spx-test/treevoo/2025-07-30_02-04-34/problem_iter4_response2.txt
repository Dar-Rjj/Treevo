import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Smooth Intraday Return with EMA (span 5)
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5).mean()
    df['smoothed_intraday_return'] = df['smoothed_intraday_return'].shift(1)  # Exclude latest value to prevent look-ahead bias
    
    # Detect Volume Spikes
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['avg_volume_change'] = df['volume_change'].rolling(window=20).mean()
    df['spike_indicator'] = (df['volume_change'] > 1.5 * df['avg_volume_change']).astype(int)
    
    # Combine Factors
    df['combined_factor'] = (df['intraday_return'] - df['smoothed_intraday_return']) * df['spike_indicator']
    df['final_factor'] = df['combined_factor'].rolling(window=3).mean().shift(1)  # Exclude the most recent value in SMA to prevent future data leakage
    
    return df['final_factor'].dropna()
