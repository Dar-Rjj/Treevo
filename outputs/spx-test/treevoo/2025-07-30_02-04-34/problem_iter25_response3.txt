import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Adjust Raw Momentum based on Volume Spike
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change'] = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = df['volume_change'] > 1.5 * df['avg_volume_change']
    df['adjusted_momentum'] = np.where(df['volume_spike'], 
                                        df['short_term_momentum'] * 0.8, 
                                        df['short_term_momentum'])
    
    # Detect Volume Spikes
    df['volume_spike'] = df['volume'] > 1.5 * df['volume'].shift(1)
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    
    # Compute Weighted Volume Indicator
    df['weighted_volume_indicator'] = (df['volume_change'] * np.sign(df['volume_change'])).rolling(window=5).mean()
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['close_price_trend'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Smooth Intraday Return
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=14, adjust=False).mean()
    df['diff_intraday_return'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Integrate Volume Impact
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_impact'] = df['intraday_return'] / df['avg_volume']
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = df['intraday_high_low_spread'] + df['close_open_spread']
    
    # Initial Alpha Signal
    df['initial_alpha_signal'] = (df['adjusted_momentum'] * df['combined_intraday_factors'] *
                                  df['smoothed_price_momentum'] * df['weighted_volume_indicator'])
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['initial_alpha_signal'] * df['close_price_trend'] * 
                                df['volume_impact'])
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    df['final_alpha_factor'] = df['final_alpha_factor'] / (1 + df['volatility'])
    
    return df['final_alpha_factor']
