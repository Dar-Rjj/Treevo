import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the given market features.
    The factor is a combination of price and volume information, aiming to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the calculated alpha factor, indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate the 5-day moving average of the daily return
    df['ma_5_day_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the daily return
    df['ma_20_day_return'] = df['daily_return'].rolling(window=20).mean()
    
    # Calculate the 5-day standard deviation of the daily return
    df['std_5_day_return'] = df['daily_return'].rolling(window=5).std()
    
    # Calculate the 20-day standard deviation of the daily return
    df['std_20_day_return'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate the 5-day moving average of the volume
    df['ma_5_day_volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the volume
    df['ma_20_day_volume'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the 5-day standard deviation of the volume
    df['std_5_day_volume'] = df['volume'].rolling(window=5).std()
    
    # Calculate the 20-day standard deviation of the volume
    df['std_20_day_volume'] = df['volume'].rolling(window=20).std()
    
    # Calculate the volatility ratio
    df['volatility_ratio'] = df['std_5_day_return'] / df['std_20_day_return']
    
    # Calculate the volume ratio
    df['volume_ratio'] = df['ma_5_day_volume'] / df['ma_20_day_volume']
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['ma_5_day_return'] - df['ma_20_day_return']) * df['volatility_ratio'] * df['volume_ratio']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
