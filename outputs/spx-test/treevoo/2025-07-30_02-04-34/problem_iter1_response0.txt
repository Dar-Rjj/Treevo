```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Average Volume over the last 5 periods
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Take the inverse of the average volume
    inverse_avg_volume = 1 / avg_volume
    
    # Multiply Intraday Return by the Inverse of Average Volume
    factor = intraday_return * inverse_avg_volume
    
    return factor
```
