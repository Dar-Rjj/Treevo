```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    n = 10  # Number of days for momentum
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).sum() / n
    
    # Identify Volume Spikes
    m = 30  # Number of days for moving average
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['volume_ma']).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['adjusted_raw_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * 0.5)
    
    # Calculate Smoothed Price Momentum
    smooth_window = 5
    df['smoothed_momentum'] = df['daily_price_change'].rolling(window=smooth_window).sum() / smooth_window
    
    # Compute Volume Weight
    volume_average = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / volume_average
    df['volume_weight'] = df['volume_weight'].rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['base_momentum_signal'] = df['smoothed_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean().shift(1)
    df['intraday_return_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Incorporate Dynamic Volume Intensity
    df['median_volume'] = df['volume'].rolling(window=5).median()
    df['dynamic_volume_weight'] = df['volume'] / df['median_volume']
    df['dynamic_volume_weight'] = df['dynamic_volume_weight'].rolling(window=5).median()
    df['weighted_intraday_return'] = df['intraday_return_diff'] * df['dynamic_volume_weight']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change()
    avg_volume_change = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = (df['volume_change'] > 1.75 * avg_volume_change).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_intraday_momentum'] = df['weighted_intraday_return'] * df['volume_spike']
    df['combined_intraday_momentum'] = df['combined_intraday_momentum'].rolling(window=4).mean().shift(1)
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['avg_5_days_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_impact'] = (df['momentum_indicator'] * df['volume']) / df['avg_5_days_volume']
    
    # Final Alpha Signal
    df['final_alpha'] = df['adjusted_raw_momentum'] * df['smoothed_momentum'] * df['volume_impact'] + df['combined_intraday_momentum']
    
    return df['final_alpha']
```
