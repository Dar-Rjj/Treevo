```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Smooth Intraday Return with EMA (span=7)
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=7, adjust=False).mean()
    df['diff_intraday_smoothed'] = df['intraday_return'] - df['smoothed_intraday_return'].shift(1)
    
    # Incorporate Volume Intensity
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['inverse_avg_volume'] = 1 / df['avg_volume_10']
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * df['inverse_avg_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change_30'] = df['volume_change'].rolling(window=30).mean()
    df['volume_spike'] = df['volume_change'] > 2 * df['avg_volume_change_30']
    df['volume_spike'] = df['volume_spike'].astype(int)  # Convert to 0/1 for multiplication
    
    # Combine Factors
    df['combined_factor'] = df['diff_intraday_smoothed'] * df['volume_spike']
    df['final_smoothed_factor'] = df['combined_factor'].rolling(window=5).mean().shift(1)
    df['final_factor'] = df['final_smoothed_factor'] * df['inverse_avg_volume']
    
    return df['final_factor'].dropna()
```
