import pandas as pd

def heuristics_v2(df, n=10, m=5, volume_spike_threshold=1.5):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    raw_momentum = df['daily_price_change'].rolling(window=n).sum() / n
    
    # Identify Volume Spikes
    avg_volume = df['volume'].rolling(window=m).mean()
    volume_spike = (df['volume'] > volume_spike_threshold * avg_volume).astype(int)
    
    # Adjust Momentum by Volume Spike
    adjusted_momentum = raw_momentum * (1 - 0.3 * volume_spike)  # Dampen by 30% if spike
    
    # Calculate Smoothed Price Momentum
    smoothed_price_momentum = df['daily_price_change'].rolling(window=5).sum() / 5
    
    # Compute Volume Weight
    avg_volume_n_days = df['volume'].rolling(window=n).mean()
    normalized_volume = (df['volume'] / avg_volume_n_days).rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    base_momentum_signal = smoothed_price_momentum * normalized_volume
    
    # Calculate Intraday High-Low Spread Factor
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    combined_intraday_factor = intraday_high_low_spread + close_to_open_return
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume']
    average_volume_5_days = df['volume'].rolling(window=5).sum() / 5
    weighted_price_movement_normalized = weighted_price_movement / average_volume_5_days
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum * combined_intraday_factor * weighted_price_movement_normalized) + base_momentum_signal
    
    # Adjust for Volatility
    std_dev_close = df['close'].rolling(window=10).std()
    final_alpha_factor_adjusted = final_alpha_factor / (1 + std_dev_close)
    
    return final_alpha_factor_adjusted
