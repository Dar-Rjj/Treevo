import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor from the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, volume) indexed by date.
    
    Returns:
    pd.Series: A Series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['10_day_ma'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the closing price
    df['10_day_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day moving average of the daily returns
    df['10_day_return_ma'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the daily returns
    df['10_day_return_std'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the normalized distance of the current close price from its 10-day moving average
    df['normalized_distance'] = (df['close'] - df['10_day_ma']) / df['10_day_std']
    
    # Calculate the volatility-adjusted momentum factor
    df['volatility_adjusted_momentum'] = df['10_day_return_ma'] / df['10_day_return_std']
    
    # Combine the factors to create the final alpha factor
    df['alpha_factor'] = df['normalized_distance'] * df['volatility_adjusted_momentum']
    
    return df['alpha_factor']
