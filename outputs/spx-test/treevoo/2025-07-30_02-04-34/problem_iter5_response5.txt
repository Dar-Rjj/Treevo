```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Logarithmic Price Return
    df['daily_log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change Rate
    df['volume_change_rate'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Compute Smoothed Logarithmic Price Momentum
    df['smoothed_log_momentum'] = df['daily_log_return'].rolling(window=7).mean()
    
    # Compute Exponential Weighted Volume Indicator
    df['ew_volume_indicator'] = df['volume_change_rate'] * np.sign(df['volume_change_rate'])
    df['ew_volume_indicator'] = df['ew_volume_indicator'].ewm(alpha=0.5, adjust=False).mean()
    
    # Combine Enhanced Indicators
    df['intermediate_alpha_factor'] = df['smoothed_log_momentum'] * df['ew_volume_indicator']
    
    # Adjust Intermediate Alpha Factor by Volume
    weighted_avg_volume = df['volume'].rolling(window=7).mean()
    df['adjusted_volume'] = df['volume'] / weighted_avg_volume
    df['final_alpha_factor'] = df['intermediate_alpha_factor'] * df['adjusted_volume']
    
    return df['final_alpha_factor']
```
