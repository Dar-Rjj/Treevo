```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day rate of change in close prices
    df['close_roc_5'] = (df['close'] / df['close'].shift(5) - 1)
    
    # Calculate the 10-day simple moving average of close prices
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Determine if the current close price is above or below the SMA
    df['close_above_sma_10'] = (df['close'] > df['sma_10']).astype(int)
    
    # Calculate the difference between current volume and the previous day's volume
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the 5-day average volume
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the 5-day average volume
    df['volume_ratio_5'] = df['volume'] / df['avg_volume_5']
    
    # Calculate the daily price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the 5-day average of daily ranges
    df['avg_price_range_5'] = df['price_range'].rolling(window=5).mean()
    
    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (df['close_roc_5'] + 
                          df['close_above_sma_10'] * 0.5 + 
                          df['volume_ratio_5'] * 0.3 - 
                          df['avg_price_range_5'] * 0.2)
    
    return df['alpha_factor']
```
