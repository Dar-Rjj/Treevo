import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=25, spike_threshold=1.75):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).mean()
    
    # Identify Volume Spikes
    df['vol_avg_m'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = df['volume'] > (spike_threshold * df['vol_avg_m'])
    
    # Adjust Momentum by Volume Spike
    df['adjusted_raw_momentum'] = df.apply(lambda row: row['raw_momentum'] * 0.5 if row['volume_spike'] else row['raw_momentum'], axis=1)
    
    # Calculate Smoothed Price Momentum
    df['smoothed_momentum'] = df['daily_price_change'].rolling(window=5).mean() / 5
    
    # Compute Volume Weight
    df['avg_volume_n'] = df['volume'].rolling(window=n).mean()
    df['normalized_volume'] = df['volume'] / df['avg_volume_n']
    df['volume_weight'] = df['normalized_volume'].rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['base_momentum_signal'] = df['smoothed_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return using EMA
    df['ema_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean().shift(1)  # Exclude latest value
    df['intraday_momentum_diff'] = df['intraday_return'] - df['ema_intraday_return']
    
    # Incorporate Volume Intensity
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['inverse_avg_volume'] = 1 / df['avg_volume_5']
    df['intraday_momentum_impact'] = df['intraday_momentum_diff'] * df['inverse_avg_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change()
    df['avg_vol_change'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike_flag'] = df['volume_change'] > (spike_threshold * df['avg_vol_change'])
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_intraday_vol_impact'] = df['intraday_momentum_impact'] * df['volume_spike_flag'].astype(int)
    df['final_smoothed_factor'] = df['combined_intraday_vol_impact'].rolling(window=4).mean().shift(1)  # Exclude latest value
    df['final_smoothed_factor_inv_vol'] = df['final_smoothed_factor'] * df['inverse_avg_volume']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['weighted_volume_indicator'] = df['momentum_indicator'] * df['volume'] / df['avg_volume_5_days']
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (df['adjusted_raw_momentum'] * df['smoothed_momentum'] * df['weighted_volume_indicator']) + df['final_smoothed_factor_inv_vol']
    
    return df['final_alpha_signal']
