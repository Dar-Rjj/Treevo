import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    n = 10  # number of days for momentum
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).sum() / n
    
    # Identify Volume Spikes
    m = 20  # number of days for moving average
    df['volume_moving_avg'] = df['volume'].rolling(window=m).mean()
    spike_threshold = 1.5  # threshold for volume spike
    df['volume_spike'] = (df['volume'] > spike_threshold * df['volume_moving_avg']).astype(int)
    
    # Adjust Momentum by Volume Spike
    dampening_factor = 0.7  # multiplier for raw momentum
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * (1 - dampening_factor))
    
    # Calculate Smoothed Price Momentum
    smooth_window = 5
    df['smoothed_momentum'] = df['daily_price_change'].rolling(window=smooth_window).sum() / smooth_window
    
    # Compute Volume Weight
    volume_average = df['volume'].rolling(window=n).mean()
    df['normalized_volume'] = df['volume'] / volume_average
    df['volume_weight'] = df['normalized_volume'].rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['base_momentum_signal'] = df['smoothed_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return with EMA
    span = 5
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=span, adjust=False).mean().shift(1)  # no look-ahead bias
    df['intraday_momentum_difference'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Incorporate Volume Intensity
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    df['inverse_avg_volume'] = 1 / avg_volume_5_days
    df['volume_intensity_impact'] = df['intraday_momentum_difference'] * df['inverse_avg_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change()
    avg_volume_change = df['volume_change'].rolling(window=25).mean()
    df['volume_spike_flag'] = (df['volume_change'] > 1.75 * avg_volume_change).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_intraday_vol_impact'] = df['volume_intensity_impact'] * df['volume_spike_flag']
    df['final_smoothed_factor'] = df['combined_intraday_vol_impact'].rolling(window=4).mean().shift(1)  # no future data leakage
    df['final_smoothed_factor'] *= df['inverse_avg_volume']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    avg_5_days_volume = df['volume'].rolling(window=5).mean()
    df['volume_impact'] = df['volume'] / avg_5_days_volume
    df['weighted_momentum_indicator'] = df['momentum_indicator'] * df['volume_impact']
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (df['adjusted_momentum'] * df['smoothed_momentum'] * df['volume_weight']) + df['final_smoothed_factor'] + df['weighted_momentum_indicator']
    
    return df['final_alpha_signal'].dropna()
