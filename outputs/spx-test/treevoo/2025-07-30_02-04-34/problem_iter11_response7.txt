import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, spike_threshold=2.0):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Daily_Price_Change'].rolling(window=n).sum() / n
    
    # Identify Volume Spikes
    df['Volume_MA'] = df['volume'].rolling(window=m).mean()
    df['Volume_Spike'] = (df['volume'] > df['Volume_MA'] * spike_threshold).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['Adjusted_Momentum'] = df['Raw_Momentum'] * (1 - 0.5 * df['Volume_Spike'])
    
    # Calculate Smoothed Price Momentum
    df['Smoothed_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).sum() / 5
    
    # Compute Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    df['Normalized_Volume'] = df['volume'] / avg_volume
    df['Volume_Weight'] = df['Normalized_Volume'].rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['Combined_Momentum'] = df['Smoothed_Price_Momentum'] * df['Volume_Weight']
    
    # Calculate Intraday High-Low Spread Factor
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    df['Combined_Intraday_Factor'] = df['Intraday_High_Low_Spread'] + df['Close_to_Open_Return']
    
    # Calculate Weighted Price Movement
    df['Weighted_Price_Movement'] = (df['close'] - df['open']) * df['volume']
    avg_past_5_days_volume = df['volume'].rolling(window=5).sum() / 5
    df['Weighted_Price_Movement_Normalized'] = df['Weighted_Price_Movement'] / avg_past_5_days_volume
    
    # Incorporate Close Price Trend
    df['Close_Price_Change'] = df['close'].pct_change()
    df['Smoothed_Close_Price_Change'] = df['Close_Price_Change'].ewm(span=5, adjust=False).mean().shift(1)  # Exclude latest value
    df['Close_Price_Trend'] = df['Close_Price_Change'] - df['Smoothed_Close_Price_Change']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_Momentum'] * df['Combined_Intraday_Factor'] * df['Weighted_Price_Movement_Normalized'] + df['Close_Price_Trend']
    
    return df['Alpha_Factor']
