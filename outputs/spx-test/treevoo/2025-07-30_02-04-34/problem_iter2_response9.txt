```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['price_change'].rolling(window=5).mean()
    
    # Compute Volume Weight
    n_days = 5
    df['volume_weight'] = df['volume'].rolling(window=n_days).mean()
    df['normalized_volume'] = df['volume'] / df['volume_weight']
    df['average_normalized_volume'] = df['normalized_volume'].rolling(window=n_days).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['base_momentum_signal'] = df['smoothed_price_momentum'] * df['average_normalized_volume']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Weighted Volume Indicator
    df['weighted_volume_indicator'] = (df['volume_change'] * df['volume_change'].apply(lambda x: 1 if x > 0 else -1)).rolling(window=5).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['base_momentum_signal'] + df['weighted_volume_indicator']
    
    return df['final_alpha_factor']
```
