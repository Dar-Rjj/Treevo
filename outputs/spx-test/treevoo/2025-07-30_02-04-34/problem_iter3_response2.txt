```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Logarithmic Price Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Change Rate
    df['volume_change_rate'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Compute Smoothed Logarithmic Price Momentum
    df['smoothed_log_momentum'] = df['log_return'].rolling(window=7).sum() / 7
    
    # Compute Exponential Weighted Volume Indicator
    df['volume_indicator'] = df['volume_change_rate'] * np.sign(df['volume_change_rate'])
    df['exp_weighted_volume_indicator'] = df['volume_indicator'].ewm(alpha=0.5, adjust=False).mean()
    
    # Combine Enhanced Indicators
    df['alpha_factor'] = df['smoothed_log_momentum'] * df['exp_weighted_volume_indicator']
    
    return df['alpha_factor']
```
