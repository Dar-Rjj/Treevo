import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Intraday Momentum Difference
    high_low_diff = df['high'] - df['low']
    intraday_momentum_diff = high_low_diff / df['open']
    
    # Smooth Intraday Return using EMA
    ema_intraday_momentum = intraday_momentum_diff.ewm(span=5, adjust=False).mean()
    smoothed_intraday_return = intraday_momentum_diff - ema_intraday_momentum.shift(1)
    
    # Incorporate Volume Intensity
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_intensity = 1 / avg_volume
    intraday_momentum_adjusted = smoothed_intraday_return * volume_intensity
    
    # Detect Volume Spikes
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change = volume_change.rolling(window=20).mean()
    volume_spike = (volume_change > 1.5 * avg_volume_change).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    combined_factor = intraday_momentum_adjusted * volume_spike
    smoothed_combined_factor = combined_factor.rolling(window=3).mean().shift(1)
    final_combined_factor = smoothed_combined_factor * volume_intensity
    
    # Integrate Next Day Open to Current Day Close
    next_day_open = df['open'].shift(-1)
    open_close_diff = df['close'] - next_day_open
    
    # Final Alpha Signal
    alpha_signal = final_combined_factor + open_close_diff
    
    return alpha_signal
