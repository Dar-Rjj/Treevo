```python
import pandas as pd

def heuristics_v2(df, n=10, m=20, volume_spike_threshold=2.0, momentum_dampening_factor=0.5):
    # Calculate Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(n)
    
    # Identify Volume Spikes
    volume_moving_average = df['volume'].rolling(window=m).mean()
    volume_spike = (df['volume'] > volume_moving_average * volume_spike_threshold)
    
    # Adjust Momentum by Volume Spike
    adjusted_momentum = raw_momentum.copy()
    adjusted_momentum[volume_spike] = raw_momentum[volume_spike] * momentum_dampening_factor
    
    return adjusted_momentum
```
