import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Raw Momentum (n=5 days)
    n = 5
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).mean()
    
    # Identify Volume Spikes (m=20 days)
    m = 20
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['volume_ma']).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - 0.3 * df['volume_spike'])
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Smooth Intraday Return (EMA with span=14)
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=14, adjust=False).mean().shift(1)
    df['intraday_return_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Integrate Volume Impact
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['intraday_return_vol_impact'] = df['intraday_return'] / avg_volume_20
    
    # Combine Intraday Factors
    df['combined_intraday_factor'] = df['intraday_high_low_spread'] + df['close_open_spread']
    
    # Compute Smoothed Price Momentum (5 days)
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    df['weighted_price_movement_normalized'] = df['weighted_price_movement'] / avg_volume_5
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean().shift(1)
    df['close_price_change_diff'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Introduce Reversal Indicator
    df['long_term_momentum'] = df['daily_price_change'].rolling(window=20).mean()
    df['short_term_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    df['reversal_indicator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['adjusted_momentum'] * df['combined_intraday_factor'] * 
                          df['weighted_price_movement_normalized'] + 
                          df['close_price_change_diff'] + 
                          df['reversal_indicator'])
    
    return df['alpha_factor']
