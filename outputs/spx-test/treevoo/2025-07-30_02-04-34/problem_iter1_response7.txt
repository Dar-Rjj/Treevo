```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    close_prices = df['close']
    returns = (close_prices - close_prices.shift(n)) / close_prices.shift(n)
    
    # Compute Weighted Average Volume
    volumes = df['volume']
    weighted_avg_volume = volumes.rolling(window=n).mean()
    
    # Normalize Volume by dividing each day's volume by the Weighted Average Volume
    normalized_volume = volumes / weighted_avg_volume
    
    # Multiply the normalized volume by the computed Price Momentum
    adjusted_momentum = returns * normalized_volume
    
    return adjusted_momentum
```
