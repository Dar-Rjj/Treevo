```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = df['close'] - df['open']
    
    # Smooth Intraday Reversal with EMA (exclude latest value)
    ema_reversal = intraday_reversal.ewm(span=7, adjust=False).mean().shift(1)
    
    # Calculate Volume Change
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_change = df['volume'] / avg_volume_10d
    
    # Identify Spike
    volume_spike = (volume_change > 2)
    
    # Combine Intraday Reversal and Volume Spike
    adjusted_reversal = (intraday_reversal - ema_reversal) * volume_spike
    combined_factor = adjusted_reversal.rolling(window=5).mean().shift(1)
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate Raw Momentum
    raw_momentum = daily_price_change.rolling(window=5).sum() / 5
    
    # Adjust Raw Momentum by Volume Spike
    adjusted_momentum = raw_momentum * (1 - 0.5 * volume_spike)
    
    # Calculate Smoothed Price Momentum
    smoothed_momentum = daily_price_change.rolling(window=5).sum() / 5
    
    # Compute Volume Weight
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    normalized_volume = df['volume'] / avg_volume_5d
    volume_weight = normalized_volume.rolling(window=5).mean()
    
    # Base Momentum Signal
    base_momentum_signal = smoothed_momentum * volume_weight
    
    # Calculate Intraday High-Low Spread Factor
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    combined_intraday_factor = high_low_spread + close_open_return
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume'] / avg_volume_5d
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_momentum * combined_intraday_factor * weighted_price_movement + base_momentum_signal)
    
    # Adjust for Volatility
    std_dev_close = df['close'].rolling(window=10).std()
    final_alpha_factor = final_alpha_factor / (1 + std_dev_close)
    
    return final_alpha_factor
```
