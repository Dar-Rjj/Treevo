```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Smooth Intraday Return with EMA
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=14, adjust=False).mean()
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Integrate Volume Impact
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_impact'] = df['intraday_return'] * (1 / df['avg_volume_20'])
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change_50'] = df['volume_change'].rolling(window=50).mean()
    df['volume_spike'] = (df['volume_change'] > 3 * df['avg_volume_change_50']).astype(int)
    
    # Combine Factors
    df['combined_factor'] = df['intraday_momentum_diff'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=10).mean().shift(1)
    df['final_combined_factor'] = df['smoothed_combined_factor'] * (1 / df['avg_volume_20'])
    
    # Adjust Raw Momentum
    df['adjusted_momentum'] = df['intraday_momentum_diff'] * (1 - 0.5 * df['volume_spike'])
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean().shift(1)
    df['close_price_trend'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Final Alpha Factor
    df['final_alpha'] = df['adjusted_momentum'] * df['final_combined_factor'] * df['close_price_trend']
    df['close_volatility_10'] = df['close'].rolling(window=10).std()
    df['final_alpha_adjusted'] = df['final_alpha'] / (1 + df['close_volatility_10'])
    
    return df['final_alpha_adjusted'].dropna()
```
