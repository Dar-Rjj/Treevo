```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['intraday_high_low_spread'] / df['open']
    
    # Smooth Intraday Return using EMA with span 5
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_momentum_difference'] = df['intraday_return'] - df['smoothed_intraday_return'].shift(1)
    
    # Incorporate Volume Intensity
    df['average_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['inverse_average_volume'] = 1 / df['average_volume_5d']
    df['intraday_momentum_vol_impact'] = df['intraday_momentum_difference'] * df['inverse_average_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change()
    df['avg_volume_change_25d'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = (df['volume_change'] > 1.75 * df['avg_volume_change_25d']).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_intraday_mom_vol_impact'] = df['intraday_momentum_vol_impact'] * df['volume_spike']
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Adjust Momentum by Volume Spike
    dampening_factor = 0.5  # Example multiplier less than 1
    df['adjusted_smoothed_momentum'] = df['smoothed_price_momentum'] * (1 - dampening_factor * df['volume_spike'])
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['average_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['weighted_volume_indicator'] = df['weighted_price_movement'] / df['average_volume_5d']
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'].diff()
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['smoothed_close_price_diff'] = df['close_price_change'] - df['smoothed_close_price_change'].shift(1)
    
    # Introduce Reversal Indicator
    df['long_term_momentum'] = df['daily_price_change'].rolling(window=20).mean()
    df['short_term_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    df['reversal_indicator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Final Alpha Signal
    df['alpha_signal'] = (
        df['adjusted_smoothed_momentum'] * df['weighted_volume_indicator'] + 
        df['combined_intraday_mom_vol_impact'] + 
        df['smoothed_close_price_diff'] + 
        df['reversal_indicator']
    )
    
    return df['alpha_signal'].dropna()
```
