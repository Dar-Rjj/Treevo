```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Momentum Indicator
    momentum_indicator = intraday_high_low_spread / close_open_spread
    
    # Integrate Volume Impact
    # Calculate 5-day average volume
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    
    # Adjust Momentum Indicator by Volume
    momentum_vol_adjusted = (momentum_indicator * df['volume']) / avg_5_day_volume
    
    # Detect Volume Spikes
    # Calculate daily volume change
    volume_change = df['volume'].pct_change()
    
    # Calculate 20-day average volume change
    avg_20_day_vol_change = volume_change.rolling(window=20).mean()
    
    # Identify volume spikes
    volume_spike = (volume_change > 1.5 * avg_20_day_vol_change).astype(int)
    
    # Combine Factors
    combined_factor = momentum_vol_adjusted * volume_spike
    
    # Smooth the combined factor with a 3-day simple moving average, excluding the most recent value
    smooth_combined_factor = combined_factor.rolling(window=3).mean().shift(1)
    
    return smooth_combined_factor
```
