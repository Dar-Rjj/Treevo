```python
import pandas as pd

def heuristics_v2(df, n=20, m=10):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).sum() / n
    
    # Identify Volume Spikes
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['volume_ma']).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * 0.3)
    
    # Calculate Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    
    # Compute Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['norm_volume'] = df['volume'] / df['avg_volume']
    df['volume_weight'] = df['norm_volume'].rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['base_momentum_signal'] = df['smoothed_price_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean().shift(1)
    df['intraday_return_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Incorporate Volume Intensity
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['inv_avg_volume_5'] = 1 / df['avg_volume_5']
    df['intraday_momentum_diff_vol'] = df['intraday_return_diff'] * df['inv_avg_volume_5']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['avg_volume_change'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike_detect'] = (df['volume_change'] > 1.75 * df['avg_volume_change']).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['intraday_momentum_vol_impact'] = df['intraday_momentum_diff_vol'] * df['volume_spike_detect']
    df['intraday_momentum_vol_impact_smooth'] = df['intraday_momentum_vol_impact'].rolling(window=4).mean().shift(1)
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['weighted_volume_indicator'] = (df['momentum_indicator'] * df['volume']) / df['avg_volume_5_days']
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (df['adjusted_momentum'] * df['base_momentum_signal'] * df['weighted_volume_indicator'] + 
                                df['intraday_momentum_vol_impact_smooth'])
    
    return df['final_alpha_signal']
```
