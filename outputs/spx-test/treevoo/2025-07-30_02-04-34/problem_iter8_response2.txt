```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Smooth Intraday Return using EMA (span=5)
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return'].shift(1)
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    avg_volume_change = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = (df['volume_change'] > 1.5 * avg_volume_change).astype(int)
    
    # Combine Intraday Momentum and Volume Spike
    df['combined_factor'] = df['intraday_momentum_diff'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=3).mean().shift(1)
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] / df['close'].shift(1) - 1
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean().shift(1)
    df['close_price_change_diff'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['avg_5_day_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_momentum'] = df['momentum_indicator'] * df['volume'] / df['avg_5_day_volume']
    
    # Final Alpha Signal
    df['final_alpha'] = df['smoothed_combined_factor'] * df['close_price_change_diff'] * df['intraday_momentum_diff'] * df['volume_adjusted_momentum']
    
    return df['final_alpha']
```
