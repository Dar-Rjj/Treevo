```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    close_prices = df['close']
    price_momentum = (close_prices.shift(1) - close_prices.shift(n)) / close_prices.shift(n)
    
    # Compute Weighted Average Volume and Amount
    volume = df['volume']
    amount = df['amount']
    weighted_avg_volume = volume.rolling(window=n).mean()
    weighted_avg_amount = amount.rolling(window=n).mean()
    
    # Adjust Volume and Amount
    adjusted_volume = volume - weighted_avg_volume
    adjusted_amount = amount - weighted_avg_amount
    
    # Multiply Price Momentum by Adjusted Volume and Amount
    factor = price_momentum * (adjusted_volume + adjusted_amount)
    
    return factor
```
