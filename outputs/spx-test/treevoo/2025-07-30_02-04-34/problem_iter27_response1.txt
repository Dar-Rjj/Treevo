import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_change_pct = df['close'].pct_change()
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Combine the factors
    alpha_factor = (price_range / df['close']) + close_change_pct + (vwap - df['close']) + (ma_5 - ma_20)
    
    return alpha_factor
