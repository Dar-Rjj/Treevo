```python
import pandas as pd

def heuristics_v2(df, n=20, m=10, spike_threshold=2.0):
    # Calculate Raw Momentum
    df['raw_momentum'] = (df['close'].shift(1) - df['close'].shift(n)) / df['close'].shift(n)
    
    # Identify Volume Spikes
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > spike_threshold * df['volume_ma'])
    
    # Adjust Momentum by Volume Spike
    df['adjusted_momentum'] = df['raw_momentum']
    df.loc[df['volume_spike'], 'adjusted_momentum'] *= 0.5  # Dampen the momentum if there's a volume spike
    
    # Adjust Momentum by Volume
    df['weighted_avg_volume'] = df['volume'].rolling(window=n).sum() / n
    df['normalized_volume'] = df['volume'] / df['weighted_avg_volume']
    
    # Final adjusted momentum
    df['final_adjusted_momentum'] = df['adjusted_momentum'] * df['normalized_volume']
    
    return df['final_adjusted_momentum']
```
