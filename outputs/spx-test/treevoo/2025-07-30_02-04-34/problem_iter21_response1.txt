```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()

    # Calculate Raw Momentum over past n days
    n = 5
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).sum() / n

    # Identify Volume Spikes
    m = 20
    df['volume_moving_avg'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['volume_moving_avg']).astype(int)

    # Adjust Momentum by Volume Spike
    momentum_multiplier = 0.7  # Dampening factor
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * (1 - momentum_multiplier))

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']

    # Smooth Intraday Return using EMA
    span = 14
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=span, adjust=False).mean()
    df['smoothed_diff_intraday_return'] = df['intraday_return'] - df['smoothed_intraday_return']

    # Integrate Volume Impact
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['intraday_return_volume_impact'] = df['intraday_return'] / df['avg_volume_20']

    # Combine Intraday Factors
    df['combined_intraday_factor'] = df['intraday_high_low_spread'] + df['close_open_spread']

    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5

    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['normalized_weighted_price_movement'] = df['weighted_price_movement'] / df['avg_volume_5']

    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'].diff()
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['smoothed_diff_close_price_change'] = df['close_price_change'] - df['smoothed_close_price_change']

    # Introduce Reversal Indicator
    df['long_term_momentum'] = df['daily_price_change'].rolling(window=20).sum() / 20
    df['short_term_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    df['reversal_indicator'] = df['short_term_momentum'] - df['long_term_momentum']

    # Include Volatility Adjustment
    df['historical_volatility'] = df['close'].rolling(window=20).std()
    df['volatility_adjusted_alpha'] = df['adjusted_momentum'] * df['combined_intraday_factor'] * df['normalized_weighted_price_movement'] * df['smoothed_diff_close_price_change'] * df['reversal_indicator'] / df['historical_volatility']

    return df['volatility_adjusted_alpha'].dropna()
```
