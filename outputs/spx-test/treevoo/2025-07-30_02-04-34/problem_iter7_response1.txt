```python
import pandas as pd

def heuristics_v2(df, n=10, m=30, volume_spike_threshold=2.0):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'].diff()

    # Identify Volume Spikes
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > df['volume_ma'] * volume_spike_threshold).astype(int)

    # Adjust Raw Momentum
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * 0.5)

    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']

    # Combine Intraday Factors
    df['combined_intraday_factors'] = df['intraday_high_low_spread'] + df['close_to_open_return']

    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['average_volume'] = df['volume'].rolling(window=5).sum() / 5
    df['normalized_weighted_price_movement'] = df['weighted_price_movement'] / df['average_volume']

    # Final Alpha Factor
    df['final_alpha'] = df['adjusted_momentum'] * df['combined_intraday_factors'] * df['normalized_weighted_price_movement']

    # Adjust for Volatility
    df['close_std'] = df['close'].rolling(window=10).std()
    df['final_alpha'] = df['final_alpha'] / (1 + df['close_std'])

    return df['final_alpha']
```
