```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, spike_threshold=2.0):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).mean()
    
    # Identify Volume Spikes
    df['volume_moving_avg'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > df['volume_moving_avg'] * spike_threshold).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - 0.5 * df['volume_spike'])
    
    # Calculate Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Compute Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['normalized_volume'] = df['volume'] / df['avg_volume']
    df['volume_weight'] = df['normalized_volume'].rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['weighted_smoothed_momentum'] = df['smoothed_price_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread Factor
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    df['combined_intraday_factor'] = df['intraday_spread'] + df['close_to_open_return']
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume'] / df['avg_volume']
    
    # Incorporate Next Day Open to Current Day Close
    df['next_day_open_to_close'] = df['open'].shift(-1) - df['close']
    df['next_day_open_to_close'] = df['next_day_open_to_close'].shift(1)  # Avoid look-ahead bias
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_momentum'] * 
                                 df['combined_intraday_factor'] * 
                                 df['weighted_price_movement'] * 
                                 df['next_day_open_to_close'])
    
    # Close Price Trend
    df['close_price_change'] = df['close'].diff()
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['close_price_trend'] = df['close_price_change'] - df['smoothed_close_price_change'].shift(1)
    
    # Add the base momentum signal to the final alpha factor
    df['final_alpha_factor'] += df['smoothed_price_momentum']
    
    # Integrate Volume Intensity
    df['average_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_intensity'] = 1 / df['average_volume_5_days']
    df['final_alpha_factor'] *= df['volume_intensity']
    
    return df['final_alpha_factor']
```
