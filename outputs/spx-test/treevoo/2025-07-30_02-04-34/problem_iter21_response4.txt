```python
import pandas as pd

def heuristics_v2(df, n=5, m=30, volume_spike_threshold=2.0, dampening_multiplier=0.8):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Raw Momentum
    raw_momentum = df['close'].diff()
    
    # Adjust Price Momentum by Volume
    weighted_avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_volume = df['volume'] / weighted_avg_volume
    price_momentum_by_volume = raw_momentum * adjusted_volume
    
    # Incorporate Volume Intensity
    avg_volume_5_periods = df['volume'].rolling(window=5).mean()
    inverse_avg_volume = 1 / avg_volume_5_periods
    momentum_by_inverse_volume = raw_momentum * inverse_avg_volume
    
    # Detect Volume Spikes
    volume_ma = df['volume'].rolling(window=m).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_change = volume_change.rolling(window=30).mean()
    volume_spike = (df['volume'] > volume_ma + volume_spike_threshold * avg_volume_change)
    
    # Adjust Momentum by Volume Spike
    adjusted_momentum = momentum_by_inverse_volume.where(~volume_spike, momentum_by_inverse_volume * dampening_multiplier)
    
    # Combine Intraday Factors
    intraday_factor = intraday_high_low_spread + close_open_spread
    
    # Compute Next Day Open to Current Day Close
    next_day_open_to_close = df['open'].shift(-1) - df['close']
    
    # Combine for Initial Alpha Signal
    initial_alpha_signal = (next_day_open_to_close * df['volume']).fillna(0)
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Compute Smoothed Price Momentum
    smoothed_price_momentum = daily_price_change.rolling(window=7).mean()
    
    # Compute Weighted Volume Indicator
    signed_volume_change = volume_change * volume_change.apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    weighted_volume_indicator = signed_volume_change.rolling(window=7).mean()
    
    # Calculate Weighted Price Movement
    money_flow = (df['close'] - df['open']) * df['volume']
    avg_volume_7_days = df['volume'].rolling(window=7).mean()
    weighted_price_movement = money_flow / avg_volume_7_days
    
    # Calculate Intraday Volatility
    intraday_range = df['high'] - df['low']
    intraday_volatility = intraday_range.rolling(window=7).mean()
    
    # Final Alpha Signal
    alpha_signal = (
        adjusted_momentum * smoothed_price_momentum * 
        weighted_volume_indicator + initial_alpha_signal * 
        intraday_factor * weighted_price_movement * 
        intraday_volatility
    )
    
    return alpha_signal
```
