import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Raw Momentum
    n = 5
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).sum() / n
    
    # Identify Volume Spikes
    m = 20
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    volume_spike_threshold = 2.0
    df['is_volume_spike'] = (df['volume'] > volume_spike_threshold * df['volume_ma']).astype(int)
    
    # Adjust Momentum by Volume Spike
    momentum_dampening_factor = 0.8
    df['adjusted_momentum'] = df.apply(lambda row: row['raw_momentum'] * momentum_dampening_factor if row['is_volume_spike'] else row['raw_momentum'], axis=1)
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return
    span = 5
    df['intraday_return_ema'] = df['intraday_return'].ewm(span=span, adjust=False).mean().shift(1)
    df['intraday_return_difference'] = df['intraday_return'] - df['intraday_return_ema']
    
    # Incorporate Volume Intensity
    average_volume = df['volume'].rolling(window=5).mean()
    df['inverse_average_volume'] = 1 / average_volume
    df['intraday_momentum_difference'] = df['intraday_return_difference'] * df['inverse_average_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_change_threshold = 2 * df['volume_change'].rolling(window=25).mean()
    df['is_volume_spike'] = (df['volume_change'] > volume_change_threshold).astype(int)
    
    # Combine Intraday Factors
    intraday_scaling_factor = 0.5
    df['combined_intraday_factor'] = (df['intraday_high_low_spread'] + df['close_open_spread']) * intraday_scaling_factor
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['average_volume_5_days'] = df['volume'].rolling(window=5).sum() / 5
    df['weighted_price_movement_normalized'] = df['weighted_price_movement'] / df['average_volume_5_days']
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean().shift(1)
    df['close_price_change_difference'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Introduce Reversal Indicator
    df['long_term_momentum'] = df['daily_price_change'].rolling(window=20).sum() / 20
    df['short_term_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    df['reversal_indicator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['adjusted_momentum'] * df['combined_intraday_factor'] * 
        df['weighted_price_movement_normalized'] + 
        df['close_price_change_difference'] + 
        df['reversal_indicator']
    )
