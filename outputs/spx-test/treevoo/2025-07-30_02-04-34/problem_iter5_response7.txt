```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 7-day Price Momentum
    price_momentum = (df['close'].shift(1) / df['close'].shift(7)) - 1
    
    # Calculate 7-day Volume Momentum
    volume_momentum = (df['volume'].shift(1) / df['volume'].shift(7)) - 1
    
    # Adjust Momentum by Volume
    volume_adjusted_momentum = price_momentum * volume_momentum
    
    # Apply Volume-Adjusted Momentum to High-Low Range
    weighted_high_low_range = high_low_range * volume_adjusted_momentum
    enhanced_high_low_range = weighted_high_low_range.rolling(window=7).sum()
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)).abs()
    
    # Normalize the Volume Shock
    normalized_volume_shock = daily_volume_change / df['volume'].rolling(window=5).mean()
    
    # Adjust Intraday Reversal by Volume Shock
    adjusted_intraday_reversal = intraday_return * normalized_volume_shock
    
    # Combine with Enhanced High-Low Range Factor
    final_alpha_factor = enhanced_high_low_range + adjusted_intraday_reversal
    
    return final_alpha_factor
```
