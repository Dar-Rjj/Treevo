import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close, volume) prices.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    pd.Series: Alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day rolling standard deviation of the returns
    df['std_10_return'] = df['return'].rolling(window=10).std()
    
    # Calculate the momentum factor over the last 60 days
    df['momentum_60'] = df['close'] / df['close'].shift(60)
    
    # Calculate the relative strength index (RSI) over the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the factor
    factor = (
        (df['ma_10_close'] / df['ma_30_close']) * 
        (df['momentum_60']) * 
        (1 / (df['std_10_return'])) * 
        (df['rsi_14'])
    )
    
    return factor.dropna()
