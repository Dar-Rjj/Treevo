import pandas as pd

def heuristics_v2(df):
    n = 10  # Number of days for momentum calculation
    df['Return'] = (df['close'].shift(1) - df['close'].shift(n)) / df['close'].shift(n)
    
    # Adjust Price Momentum by Volume
    weighted_avg_volume = df['volume'].rolling(window=n).mean()
    adjusted_volume = df['volume'].shift(1) / weighted_avg_volume
    price_momentum_adjusted = df['Return'] * adjusted_volume
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) - df['open']
    
    # Compute Next Day Open to Current Day Close
    next_day_open_to_close = df['close'] - df['open'].shift(-1)
    
    # Combine for Initial Alpha Signal
    initial_alpha_signal = (next_day_open_to_close * df['volume']).shift(1)
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Smoothed Price Momentum
    smoothed_price_momentum = daily_price_change.rolling(window=5).sum() / 5
    
    # Compute Weighted Volume Indicator
    weighted_volume_indicator = (volume_change * (volume_change > 0).astype(int) - volume_change * (volume_change < 0).astype(int)).rolling(window=5).mean()
    
    # Final Alpha Signal
    final_alpha_signal = (price_momentum_adjusted * smoothed_price_momentum * weighted_volume_indicator) + initial_alpha_signal
    
    return final_alpha_signal
