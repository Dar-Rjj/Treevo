import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day average volume
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the difference between today's volume and the 30-day average volume
    df['volume_diff'] = df['volume'] - df['avg_volume_30']
    
    # Calculate the sum of signed power (power=2) of the last 5 close prices minus the open price
    df['close_minus_open'] = df['close'] - df['open']
    df['signed_power_2'] = df['close_minus_open'].apply(lambda x: x**2 if x > 0 else -x**2)
    df['sum_signed_power_2'] = df['signed_power_2'].rolling(window=5).sum()
    
    # Calculate the daily high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate the absolute value of today’s return (close - open)
    df['abs_return'] = (df['close'] - df['open']).abs()
    
    # Combine the daily high-low range with the absolute value of today’s return
    df['combined_indicator'] = df['high_low_range'] + df['abs_return']
    
    # Combine all subthoughts into a final factor
    df['alpha_factor'] = (df['volume_diff'] / df['avg_volume_30']) * df['sum_signed_power_2'] * df['combined_indicator']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
