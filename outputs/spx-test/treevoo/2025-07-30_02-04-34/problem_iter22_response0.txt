```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages (SMA) for different periods
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between short-term and long-term SMA
    df['SMA_diff'] = df['SMA_5'] - df['SMA_20']
    
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the average daily return over a fixed period (e.g., 5, 10 days)
    df['avg_daily_return_5'] = df['daily_return'].rolling(window=5).mean()
    df['avg_daily_return_10'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the standard deviation of daily returns over a fixed period (e.g., 5, 10 days) for volatility
    df['volatility_5'] = df['daily_return'].rolling(window=5).std()
    df['volatility_10'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the ratio of current day's volume to the average volume over a fixed period (e.g., 5, 10 days)
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio_5'] = df['volume'] / df['avg_volume_5']
    df['volume_ratio_10'] = df['volume'] / df['avg_volume_10']
    
    # Calculate the correlation between daily volume and daily return
    df['volume_return_corr'] = df['volume'].rolling(window=5).corr(df['daily_return'])
    
    # Calculate the range (high - low) and analyze its trends
    df['range'] = df['high'] - df['low']
    df['avg_range_5'] = df['range'].rolling(window=5).mean()
    df['avg_range_10'] = df['range'].rolling(window=10).mean()
    
    # Calculate the relative position of the closing price within the high-low range
    df['relative_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the amount (volume * close) and analyze its trends
    df['amount'] = df['volume'] * df['close']
    df['avg_amount_5'] = df['amount'].rolling(window=5).mean()
    df['avg_amount_10'] = df['amount'].rolling(window=10).mean()
    
    # Calculate the correlation between the amount and the close price
    df['amount_close_corr'] = df['amount'].rolling(window=5).corr(df['close'])
    
    # Combine multiple factors to create a composite indicator
    df['composite_factor'] = (df['SMA_diff'] + df['avg_daily_return_5'] - df['volatility_5'] 
                             + df['volume_ratio_5'] * df['volume_return_corr'] + df['relative_position'] 
                             + df['amount_close_corr'])
    
    # Normalize the composite factor
    df['normalized_composite_factor'] = (df['composite_factor'] - df['composite_factor'].mean()) / df['composite_factor'].std()
    
    return df['normalized_composite_factor']
```
