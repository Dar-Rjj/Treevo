import pandas as pd

def heuristics_v2(df, n=10, m=30):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n).mean()
    
    # Identify Volume Spikes
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    volume_spike_threshold = 1.5  # Example threshold, can be adjusted
    df['volume_spike'] = df['volume'] > (df['volume_ma'] * volume_spike_threshold)
    
    # Adjust Momentum by Volume Spike
    momentum_dampen_multiplier = 0.7  # Example multiplier, can be adjusted
    df['adjusted_momentum'] = df.apply(
        lambda row: row['raw_momentum'] * momentum_dampen_multiplier if row['volume_spike'] else row['raw_momentum'],
        axis=1
    )
    
    # Calculate Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Compute Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['normalized_volume'] = df['volume'] / df['avg_volume']
    df['volume_weight'] = df['normalized_volume'].rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['weighted_momentum'] = df['smoothed_price_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread Factor
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    df['combined_intraday_factor'] = df['intraday_spread'] + df['close_to_open_return']
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume'] / df['avg_volume']
    
    # Incorporate Next Day Open to Current Day Close
    df['next_day_open_to_close'] = df['open'].shift(-1) - df['close']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['adjusted_momentum'] *
        df['combined_intraday_factor'] *
        df['weighted_price_movement']
    )
    
    # Close Price Trend
    df['close_price_change'] = df['close'].pct_change()
    df['ema_close_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['close_trend_diff'] = df['close_price_change'] - df['ema_close_change']
    
    # Add the base momentum signal to the final alpha factor
    df['final_alpha_factor'] = df['alpha_factor'] + df['close_trend_diff']
    
    # Integrate Volume Intensity
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['intraday_momentum_difference'] = df['combined_intraday_factor'] - df['close_trend_diff']
    df['final_alpha_factor'] += df['intraday_momentum_difference'] / df['avg_volume_5']
    
    return df['final_alpha_factor']
