```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'volume' and index 'date'.
    
    Returns:
    pd.Series: A pandas Series indexed by 'date' representing the factor values.
    """
    # Calculate the intraday range
    intraday_range = df['high'] - df['low']
    
    # Calculate the closing price relative to the intraday range
    close_to_range = (df['close'] - df['low']) / intraday_range
    
    # Calculate the volume weighted by the closing price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate the 10-day moving average of the volume weighted close
    moving_avg_10d = volume_weighted_close.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the volume
    moving_avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Normalize the 10-day moving average of the volume weighted close by the 10-day moving average of the volume
    normalized_moving_avg_10d = moving_avg_10d / moving_avg_volume_10d
    
    # Final alpha factor: combination of the closing price relative to the intraday range and the normalized moving average
    alpha_factor = (close_to_range + normalized_moving_avg_10d) / 2
    
    return alpha_factor
```
