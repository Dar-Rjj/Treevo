```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Smooth Intraday Return with EMA
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    avg_volume_change = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = (df['volume_change'] > 1.5 * avg_volume_change).astype(int)
    
    # Combine Factors
    df['raw_factor'] = df['intraday_momentum_diff'] * df['volume_spike']
    df['final_factor'] = df['raw_factor'].rolling(window=3, min_periods=1).mean().shift(1)
    
    return df['final_factor'].dropna()
```
