```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Smooth Intraday Return using EMA
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=14, adjust=False).mean().shift(1)
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Integrate Volume Impact
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['inv_avg_volume'] = 1 / df['avg_volume_20']
    df['intraday_vol_impact'] = df['intraday_return'] * df['inv_avg_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_vol_change_50'] = df['volume_change'].rolling(window=50).mean()
    df['vol_spike'] = (df['volume_change'] > 3 * df['avg_vol_change_50']).astype(int)
    
    # Combine Factors
    df['combined_factor'] = df['intraday_momentum_diff'] * df['vol_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=10).mean().shift(1)
    df['final_intraday_factor'] = df['smoothed_combined_factor'] * df['inv_avg_volume']
    
    # Calculate Price Momentum
    n = 14
    df['price_momentum'] = (df['close'].shift(1) - df['close'].shift(n)) / df['close'].shift(n)
    
    # Adjust Price Momentum by Volume
    df['weighted_avg_volume'] = df['volume'].rolling(window=n).sum() / n
    df['adjusted_price_momentum'] = df['price_momentum'] * (df['volume'] / df['weighted_avg_volume'])
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    
    # Compute Weighted Volume Indicator
    df['weighted_volume_indicator'] = (df['volume_change'] * df['volume_change'].apply(lambda x: 1 if x > 0 else -1)).rolling(window=5).mean()
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (df['adjusted_price_momentum'] * df['smoothed_price_momentum'] * df['weighted_volume_indicator']) + df['final_intraday_factor']
    
    return df['final_alpha_signal']
```
