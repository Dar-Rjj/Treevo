```python
import pandas as pd

def heuristics(df, n=5):
    # Calculate Price Momentum
    close_prices = df['close']
    price_momentum = (close_prices - close_prices.shift(n)) / close_prices.shift(n)
    
    # Adjust Momentum by Volume and Amount
    volumes = df['volume']
    amounts = df['amount']
    
    # Compute Weighted Average Volume and Amount
    weighted_avg_volume = volumes.rolling(window=n).mean()
    weighted_avg_amount = amounts.rolling(window=n).mean()
    
    # Normalize Volume and Amount
    normalized_volume = volumes / weighted_avg_volume
    normalized_amount = amounts / weighted_avg_amount
    
    # Multiply Price Momentum by Normalized Volume and Amount
    adjusted_momentum = price_momentum * normalized_volume * normalized_amount
    
    return adjusted_momentum
```
