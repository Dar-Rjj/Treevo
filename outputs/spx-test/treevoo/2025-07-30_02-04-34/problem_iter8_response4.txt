import pandas as pd

def heuristics_v2(df, n=10, m=10, spike_threshold=1.5, momentum_dampening=0.5):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'].diff()
    
    # Adjust Price Momentum by Volume
    weighted_avg_volume = df['volume'].rolling(window=n).mean()
    df['adjusted_volume'] = df['volume'] / weighted_avg_volume
    df['adjusted_momentum'] = df['raw_momentum'] * df['adjusted_volume']
    
    # Identify Volume Spikes
    moving_avg_volume = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = df['volume'] > (spike_threshold * moving_avg_volume)
    
    # Adjust Momentum by Volume Spike
    df['final_momentum'] = df['raw_momentum']
    df.loc[df['volume_spike'], 'final_momentum'] *= momentum_dampening
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low'] - (df['open'] - df['close'])
    
    # Combine Intraday Factors
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['close_to_open_return'] = df['close'] - df['open']
    df['combined_intraday_factor'] = df['intraday_high_low_spread'] + df['close_to_open_return']
    
    # Compute Next Day Open to Current Day Close
    df['next_day_open_to_close'] = df['open'].shift(-1) - df['close']
    
    # Combine for Initial Alpha Signal
    df['initial_alpha_signal'] = df['final_momentum'] * df['volume'] + df['next_day_open_to_close']
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    
    # Compute Weighted Volume Indicator
    df['weighted_volume_indicator'] = (df['volume_change'] * df['volume_change'].apply(lambda x: 1 if x > 0 else -1)).rolling(window=5).mean()
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume'] / (df['volume'].rolling(window=5).mean())
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        (df['final_momentum'] * df['smoothed_price_momentum'] * df['weighted_volume_indicator']) +
        df['initial_alpha_signal'] * df['combined_intraday_factor'] * df['weighted_price_movement']
    )
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    df['alpha_factor'] /= 1 + df['volatility']
    
    return df['alpha_factor'].dropna()
