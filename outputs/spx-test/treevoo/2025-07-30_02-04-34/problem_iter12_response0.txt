```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']

    # Smooth Intraday Return with EMA (span=14)
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=14, adjust=False).mean()
    df['intraday_return_diff'] = df['intraday_return'] - df['smoothed_intraday_return']

    # Incorporate Volume Intensity
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['inverse_average_volume'] = 1 / df['average_volume']
    df['volume_integrated_return'] = df['intraday_return'] * df['inverse_average_volume']

    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change()
    average_volume_change = df['volume_change'].rolling(window=50).mean()
    df['volume_spike'] = (df['volume_change'] > 3 * average_volume_change).astype(int)

    # Combine Factors
    df['combined_factor'] = df['intraday_return_diff'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=10).mean()
    df['final_factor'] = df['smoothed_combined_factor'] * df['inverse_average_volume']

    return df['final_factor']
```
