```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate the Moving Average of Volume over Last 5 Days
    moving_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Adjust Intraday Return by Volume Shock
    volume_shock_ratio = daily_volume_change / moving_avg_volume
    adjusted_intraday_return = intraday_return * volume_shock_ratio
    
    return adjusted_intraday_return
```
