import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = df['open'] - high_low_diff

    # Compute Next Day Open to Current Day Close
    next_day_open = df['open'].shift(-1)
    open_close_diff = next_day_open - df['close']

    # Combine for Initial Alpha Signal
    initial_alpha_signal = (intraday_momentum * df['volume']).rolling(window=5).sum()

    # Smooth Intraday Return
    intraday_return_ema = df['close'].pct_change().ewm(span=5, adjust=False).mean().shift(1)

    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()

    # Calculate Volume Change
    volume_change = df['volume'].diff()

    # Compute Smoothed Price Momentum
    smoothed_price_momentum = daily_price_change.rolling(window=5).sum() / 5

    # Compute Weighted Volume Indicator
    weighted_volume_indicator = (volume_change * np.sign(volume_change)).rolling(window=5).mean()

    # Identify Spike
    average_volume_change_last_20_days = volume_change.rolling(window=20).mean()
    spike = (volume_change > 1.5 * average_volume_change_last_20_days).astype(int)

    # Combine Factors
    combined_factor = (intraday_momentum * spike).rolling(window=3).mean().shift(1)

    # Final Alpha Signal
    final_alpha_signal = (smoothed_price_momentum * weighted_volume_indicator + initial_alpha_signal).shift(1)

    return final_alpha_signal
