```python
import pandas as pd

def heuristics_v2(df):
    # Parameters
    n_days_momentum = 5
    volume_spike_multiplier = 0.8
    intraday_scaling_factor = 0.5
    short_term_days = 5
    long_term_days = 20
    volatility_threshold_low = 0.05
    volatility_threshold_high = 0.1
    volatility_amplify_factor = 1.3
    volatility_reduce_factor = 0.7
    
    # Calculate Raw Momentum
    df['daily_price_change'] = df['close'].pct_change()
    df['raw_momentum'] = df['daily_price_change'].rolling(window=n_days_momentum).sum() / n_days_momentum
    
    # Adjust Momentum by Volume Spike
    df['volume_spike'] = df['volume'] > (df['volume'].shift(1) * 1.5)
    df['adjusted_momentum'] = df.apply(lambda row: row['raw_momentum'] * volume_spike_multiplier if row['volume_spike'] else row['raw_momentum'], axis=1)
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Combine Intraday Factors
    df['combined_intraday_factor'] = (df['intraday_high_low_spread'] + df['close_open_spread']) * intraday_scaling_factor
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    df['normalized_weighted_price_movement'] = df['weighted_price_movement'] / df['average_volume']
    
    # Introduce Reversal Indicator
    df['long_term_momentum'] = df['daily_price_change'].rolling(window=long_term_days).sum() / long_term_days
    df['short_term_momentum'] = df['daily_price_change'].rolling(window=short_term_days).sum() / short_term_days
    df['reversal_indicator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['adjusted_momentum'] * df['combined_intraday_factor'] * 
                                df['normalized_weighted_price_movement'] + 
                                df['reversal_indicator'])
    
    # Introduce Volatility Filter
    df['volatility'] = df['close'].rolling(window=long_term_days).std()
    df['final_alpha_factor'] = df.apply(
        lambda row: row['final_alpha_factor'] * volatility_reduce_factor if row['volatility'] < volatility_threshold_low else (
            row['final_alpha_factor'] * volatility_amplify_factor if row['volatility'] > volatility_threshold_high else row['final_alpha_factor']
        ), axis=1
    )
    
    return df['final_alpha_factor']
```
