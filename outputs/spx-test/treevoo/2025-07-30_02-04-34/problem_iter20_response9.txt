```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return using EMA with a span of 5
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_momentum_difference'] = df['intraday_return'] - df['smoothed_intraday_return'].shift(1)
    
    # Incorporate Volume Intensity
    df['average_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['inverse_average_volume'] = 1 / df['average_volume_5_days']
    df['intraday_momentum_weighted_by_volume'] = df['intraday_momentum_difference'] * df['inverse_average_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['average_volume_change_25_days'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = (df['volume_change'] > 1.75 * df['average_volume_change_25_days']).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_factor'] = df['intraday_momentum_weighted_by_volume'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=4).mean().shift(1)
    df['final_combined_factor'] = df['smoothed_combined_factor'] * df['inverse_average_volume']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['momentum_indicator_weighted_volume'] = df['momentum_indicator'] * (df['volume'] / df['average_volume_5_days'])
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (df['momentum_indicator_weighted_volume'] * df['smoothed_intraday_return'].shift(1) + 
                                df['final_combined_factor'])
    
    return df['final_alpha_signal']
```
