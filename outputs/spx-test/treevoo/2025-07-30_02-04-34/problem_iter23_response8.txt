```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low']) / 2).cumsum() / df['volume'].cumsum()
    
    # Calculate the moving average of the VWAP over a 10-day window
    vwap_10_day_ma = vwap.rolling(window=10).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day window
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (price_range * pct_change_close * (vwap - vwap_10_day_ma) * (rsi - 50)).rolling(window=5).mean()
    
    return alpha_factor
```
