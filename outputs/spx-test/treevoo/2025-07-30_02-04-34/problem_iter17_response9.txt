import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=25, intraday_ema_span=5, volume_smoothing_window=4):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['price_change'].rolling(window=n).mean()
    
    # Identify Volume Spikes
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > 1.75 * df['volume_ma']).astype(float)
    
    # Adjust Momentum by Volume Spike
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * 0.5)
    
    # Calculate Smoothed Price Momentum
    df['smoothed_momentum'] = df['price_change'].rolling(window=5).mean() / 5
    
    # Compute Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = (df['volume'] / avg_volume).rolling(window=n).mean()
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['base_momentum_signal'] = df['smoothed_momentum'] * df['volume_weight']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=intraday_ema_span).mean().shift(1)
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Incorporate Volume Intensity
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['inverse_avg_volume'] = 1 / df['avg_volume_5']
    df['intraday_momentum_adjusted'] = df['intraday_momentum_diff'] * df['inverse_avg_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change()
    df['volume_spike_threshold'] = df['volume_change'].rolling(window=m).std() * 1.75
    df['volume_spike'] = (df['volume_change'] > df['volume_spike_threshold']).astype(float)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_intraday_factor'] = df['intraday_momentum_adjusted'] * df['volume_spike']
    df['smoothed_combined_intraday_factor'] = df['combined_intraday_factor'].rolling(window=volume_smoothing_window).mean().shift(1)
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['volume_5_day_avg'] = df['volume'].rolling(window=5).mean()
    df['weighted_volume_indicator'] = df['momentum_indicator'] * df['volume'] / df['volume_5_day_avg']
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (df['adjusted_momentum'] * df['smoothed_momentum'] * df['volume_weight']) + df['smoothed_combined_intraday_factor']
    
    return df['final_alpha_signal'].dropna()
