```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, volume_spike_threshold=2.0):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'].diff()
    
    # Identify Volume Spikes
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > df['volume_ma'] * volume_spike_threshold).astype(int)
    
    # Adjust Raw Momentum
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - 0.5 * df['volume_spike'])
    
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Combine Intraday Factors
    df['intraday_factors'] = df['intraday_spread'] + df['close_to_open_return']
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    df['weighted_price_movement_normalized'] = df['weighted_price_movement'] / df['average_volume']
    
    # Integrate Volume Impact
    df['volume_impact'] = df['volume'] / df['average_volume']
    df['intraday_momentum'] = df['intraday_factors'] * df['volume_impact']
    
    # Combine for Initial Alpha Signal
    df['initial_alpha_signal'] = df['adjusted_momentum'] + df['intraday_momentum']
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['close'].diff().rolling(window=5).mean()
    
    # Compute Weighted Volume Indicator
    df['volume_change'] = df['volume'].pct_change()
    df['volume_change_sign'] = np.sign(df['volume_change'])
    df['weighted_volume_indicator'] = (df['volume_change'] * df['volume_change_sign']).rolling(window=5).mean()
    
    # Introduce New Features
    df['close_to_previous_close_return'] = df['close'] - df['close'].shift(1)
    df['close_to_previous_open_return'] = df['close'] - df['open'].shift(1)
    df['new_features'] = df['close_to_previous_close_return'] + df['close_to_previous_open_return']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_momentum'] *
        df['intraday_factors'] *
        df['weighted_price_movement_normalized'] *
        df['initial_alpha_signal'] *
        df['smoothed_price_momentum'] *
        df['weighted_volume_indicator'] *
        df['new_features']
    )
    
    # Adjust for Volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    df['final_alpha_factor'] = df['final_alpha_factor'] / (1 + df['volatility'])
    
    return df['final_alpha_factor']
```
