import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Smoothed Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    smoothed_intraday_return = intraday_return.ewm(span=5, adjust=False).mean().shift(1)
    
    # Incorporate Volume Intensity
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_intensity = 1 / avg_volume_5_days
    intraday_momentum_difference = (df['high'] - df['low'])
    volume_adjusted_momentum = intraday_momentum_difference * volume_intensity
    
    # Detect Volume Spikes
    volume_change = df['volume'].diff()
    avg_volume_change = volume_change.rolling(window=20).mean()
    volume_spike = (volume_change > 2 * avg_volume_change).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    combined_factor = volume_adjusted_momentum * volume_spike
    smoothed_combined_factor = combined_factor.rolling(window=4).mean().shift(1)
    
    # Integrate Next Day Open to Current Day Close
    next_day_open = df['open'].shift(-1)
    open_close_diff = df['close'] - next_day_open
    
    # Final Alpha Signal
    alpha_signal = (smoothed_intraday_return + smoothed_combined_factor + open_close_diff).dropna()
    alpha_signal = alpha_signal[alpha_signal.abs() > 0.01]  # Apply a threshold to filter out small signals
    
    return alpha_signal
