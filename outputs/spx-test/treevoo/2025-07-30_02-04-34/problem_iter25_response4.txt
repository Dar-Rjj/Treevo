```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Smoothed Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5).mean().shift(1)
    
    # Incorporate Volume Intensity
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_intensity'] = 1 / df['avg_volume_10']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change_20'] = df['volume_change'].rolling(window=20).mean()
    df['vol_spike'] = (df['volume_change'] > 3 * df['avg_volume_change_20']).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['intraday_momentum_vol_impact'] = df['intraday_return'] * df['vol_spike']
    df['smoothed_intraday_momentum_vol_impact'] = df['intraday_momentum_vol_impact'].rolling(window=4).mean().shift(1)
    
    # Analyze Price and Volume Relationship
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['divergence'] = ((df['price_change'] * df['volume_change']) < 0).astype(int)
    
    # Final Alpha Signal
    df['alpha_signal'] = df['smoothed_intraday_return'] + df['smoothed_intraday_momentum_vol_impact']
    df['alpha_signal'] = df['alpha_signal'].where(df['alpha_signal'].abs() > 0.01, 0)  # Apply threshold
    df['alpha_signal'] = df['alpha_signal'] * (1 - df['divergence'])  # Adjust based on divergences
    
    return df['alpha_signal']
```
