```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-Based Factor
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['momentum_difference'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Volume-Based Factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_change_factor'] = df['volume_change'] / df['average_volume']
    
    # Price Action Factor
    df['daily_range'] = df['high'] - df['low']
    df['daily_spread'] = df['close'] - df['open']
    df['price_action_factor'] = df['daily_spread'] / df['daily_range']
    
    # Intraday Analysis
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['intraday_close_open_spread'] = df['close'] - df['open']
    df['smoothed_intraday_return'] = (df['high'] - df['low']) / df['open']
    df['smoothed_intraday_return_ema'] = df['smoothed_intraday_return'].ewm(span=5, adjust=False).mean().shift(1)
    df['inverse_average_volume'] = 1 / df['average_volume']
    df['smoothed_intraday_return_weighted'] = df['smoothed_intraday_return_ema'] * df['inverse_average_volume']
    
    # Detect volume spikes
    df['volume_change_last_25'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = df['volume_change'] > 1.75 * df['volume_change_last_25']
    df['volume_spike_smoothed'] = df['volume_spike'].rolling(window=4).mean().shift(1)
    df['intraday_momentum_volume_impact'] = df['smoothed_intraday_return_weighted'] * df['volume_spike_smoothed']
    
    # Moving Averages
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_100'] = df['close'].ewm(span=100, adjust=False).mean()
    df['smacd'] = df['sma_5'] - df['sma_20']
    
    # Composite Factor
    df['composite_factor'] = (
        0.4 * df['momentum_difference'] +
        0.3 * df['volume_change_factor'] +
        0.1 * df['price_action_factor'] +
        0.2 * df['intraday_momentum_volume_impact']
    )
    
    # Final Alpha Signal
    df['adjusted_price_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['smoothed_price_momentum'] = df['adjusted_price_momentum'].ewm(span=5, adjust=False).mean()
    df['weighted_volume_indicator'] = df['volume_change_factor'] * df['inverse_average_volume']
    df['final_alpha_signal'] = (
        (df['adjusted_price_momentum'] * df['smoothed_price_momentum']) *
        df['weighted_volume_indicator'] +
        df['intraday_momentum_volume_impact']
    )
    
    return df['final_alpha_signal']
```
