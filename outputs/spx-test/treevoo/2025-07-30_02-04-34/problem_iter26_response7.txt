```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in close prices over 5 days
    df['roc_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate the 10-day simple moving average (SMA) of close prices
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Determine if the current close price is above or below the SMA
    df['above_sma_10'] = (df['close'] > df['sma_10']).astype(int)
    
    # Use the sum of signed power (power=2) of the last 5 close prices minus the open price to detect trend strength or weakness
    df['trend_strength'] = ((df['close'] - df['open']) ** 2).rolling(window=5).sum()
    
    # Calculate the difference between current volume and the previous day's volume
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the difference between today's volume and the average volume over the past 30 days
    df['volume_unusual'] = df['volume'] - df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of today's volume to the 5-day average volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Measure the daily price range
    df['daily_range'] = df['high'] - df['low']
    
    # Combine the daily high-low range with the absolute value of todayâ€™s return to form a combined indicator of market action
    df['market_action'] = df['daily_range'] + abs(df['close'] - df['open'])
    
    # Calculate the 5-day average of daily ranges
    df['avg_daily_range_5'] = df['daily_range'].rolling(window=5).mean()
    
    # Synthesize the alpha factor
    alpha_factor = (
        df['roc_5'] * 0.3 +
        df['above_sma_10'] * 0.2 +
        df['trend_strength'] * 0.1 +
        df['volume_diff'] * 0.1 +
        df['volume_unusual'] * 0.1 +
        df['volume_ratio'] * 0.1 +
        df['market_action'] * 0.1
    )
    
    return alpha_factor
```
