import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor from the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A series of factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 21-day moving average of the close price
    df['ma_21'] = df['close'].rolling(window=21).mean()
    
    # Calculate the 21-day rolling standard deviation of the close price
    df['std_21'] = df['close'].rolling(window=21).std()
    
    # Calculate the 21-day moving average of the volume
    df['vol_ma_21'] = df['volume'].rolling(window=21).mean()
    
    # Calculate the normalized range (high - low) / close
    df['normalized_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the momentum factor as the 21-day return
    df['momentum_21'] = df['close'].pct_change(periods=21)
    
    # Calculate the relative strength factor as the ratio of the 21-day moving average to the 50-day moving average
    df['ma_50'] = df['close'].rolling(window=50).mean()
    df['relative_strength'] = df['ma_21'] / df['ma_50']
    
    # Calculate the alpha factor as a combination of the above factors
    df['alpha_factor'] = (
        df['daily_return'] * 0.3 +
        (df['close'] - df['ma_21']) / df['std_21'] * 0.3 +
        df['normalized_range'] * 0.2 +
        df['momentum_21'] * 0.1 +
        df['relative_strength'] * 0.1
    )
    
    return df['alpha_factor']
