import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel and interpretable alpha factor based on the provided DataFrame.
    The factor is designed to help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the relative range
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the buying pressure
    df['buying_pressure'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the weighted buying pressure
    df['weighted_buying_pressure'] = df['buying_pressure'] * df['volume']
    
    # Calculate the 5-day moving average of the weighted buying pressure
    df['ma_weighted_buying_pressure'] = df['weighted_buying_pressure'].rolling(window=5).mean()
    
    # Calculate the 10-day rolling standard deviation of the return
    df['std_return'] = df['return'].rolling(window=10).std()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['ma_weighted_buying_pressure'] / df['range']) * df['std_return']
    
    # Drop any rows with NaN values
    df.dropna(inplace=True)
    
    return df['alpha_factor']
