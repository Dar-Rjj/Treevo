```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)).abs()
    
    # Normalize the Volume Shock by dividing by average volume over last 5 periods
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    normalized_volume_shock = daily_volume_change / avg_volume_5d
    
    # Multiply Intraday Return by this normalized Volume Shock
    factor = intraday_return * normalized_volume_shock
    
    return factor
```
