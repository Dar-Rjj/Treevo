```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Smooth Intraday Return using EMA with span 5
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change'] = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = df['volume_change'] > 1.5 * df['avg_volume_change']
    df['volume_spike'] = df['volume_spike'].astype(int)
    
    # Combine Factors
    df['combined_factor'] = df['intraday_momentum_diff'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=3, min_periods=1).mean().shift(1)
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Momentum Indicator
    df['momentum_indicator'] = df['intraday_high_low_spread'] / df['close_open_spread']
    
    # Integrate Volume Impact
    df['avg_5_day_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_impact'] = (df['momentum_indicator'] * df['volume']) / df['avg_5_day_volume']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_combined_factor'] + df['volume_impact']
    
    return df['final_alpha_factor']
```
