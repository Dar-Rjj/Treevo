import pandas as pd

def heuristics_v2(df, n=10, m=20, threshold=2.0):
    # Calculate Raw Momentum
    df['raw_momentum'] = df['close'].diff()
    
    # Identify Volume Spike
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > (threshold * df['volume_ma'])).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * 0.5)
    
    # Calculate Intraday High-Low Spread Factor
    df['high_low_spread'] = df['high'] - df['low']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    df['intraday_factor'] = df['high_low_spread'] + (df['close'] - df['open'])
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    df['weighted_price_movement_avg'] = df['weighted_price_movement'] / df['avg_volume']
    
    # Final Alpha Factor
    df['std_dev_close'] = df['close'].rolling(window=10).std()
    df['final_alpha'] = (df['adjusted_momentum'] * df['intraday_factor'] * df['weighted_price_movement_avg']) / (1 + df['std_dev_close'])
    
    return df['final_alpha']
