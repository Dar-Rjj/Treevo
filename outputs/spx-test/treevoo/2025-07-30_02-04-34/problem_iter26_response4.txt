```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Smooth Intraday Return with EMA
    df['intraday_return_ema'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_momentum_diff'] = df['intraday_return'] - df['intraday_return_ema']
    
    # Incorporate Volume Intensity
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    df['inverse_avg_volume'] = 1 / df['avg_volume']
    df['intraday_momentum_vol_impact'] = df['intraday_momentum_diff'] * df['inverse_avg_volume']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change'] = df['volume_change'].rolling(window=25).mean()
    df['is_volume_spike'] = df['volume_change'] > 1.75 * df['avg_volume_change']
    
    # Combine Intraday Momentum and Volume Impact
    df['intraday_momentum_vol_spike'] = df['intraday_momentum_vol_impact'] * df['is_volume_spike']
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    
    # Adjust Momentum by Volume Spike
    df['adjusted_momentum'] = df['smoothed_price_momentum'] * (0.8 if df['is_volume_spike'].any() else 1)
    
    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['weighted_volume_indicator'] = df['weighted_price_movement'] / df['avg_volume']
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['smoothed_close_price_diff'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Introduce Reversal Indicator
    df['long_term_momentum'] = df['daily_price_change'].rolling(window=20).sum() / 20
    df['short_term_momentum'] = df['daily_price_change'].rolling(window=5).sum() / 5
    df['reversal_indicator'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Final Alpha Signal
    df['alpha_signal'] = (
        df['adjusted_momentum'] * df['smoothed_price_momentum'] * 
        df['weighted_volume_indicator'] + 
        df['intraday_momentum_vol_spike'] + 
        df['smoothed_close_price_diff'] + 
        df['reversal_indicator']
    )
    
    return df['alpha_signal']
```
