```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread Factor
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    combined_intraday_factor = high_low_spread + close_open_return
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume']
    
    # Calculate Average Volume over the past 5 days
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Normalize Weighted Price Movement by Average Volume
    normalized_weighted_price_movement = weighted_price_movement / average_volume
    
    # Final Alpha Factor
    final_alpha_factor = combined_intraday_factor * normalized_weighted_price_movement
    
    # Adjust for Volatility
    std_close_prices = df['close'].rolling(window=10).std()
    adjusted_final_alpha_factor = final_alpha_factor / (1 + std_close_prices)
    
    return adjusted_final_alpha_factor
```
