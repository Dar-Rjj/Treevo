import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-Based Factor
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Volume-Based Factor
    volume_change = df['volume'] - df['volume'].shift(1)
    average_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / df['volume'].shift(1)
    enhanced_volume_change = volume_change * volume_ratio
    
    # Price Action Factor
    daily_range = df['high'] - df['low']
    daily_spread = df['close'] - df['open']
    high_close_spread = df['high'] - df['close']
    price_action_factor = (daily_spread / daily_range) - high_close_spread
    
    # Intraday Analysis
    intraday_high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    smoothed_intraday_return = ((df['high'] - df['low']) / df['close']).ewm(span=5).mean().shift(1)
    average_volume_5 = df['volume'].rolling(window=5).mean()
    intraday_momentum_difference = short_term_momentum - long_term_momentum
    volume_change_prev = df['volume'] - df['volume'].shift(1)
    avg_volume_change = volume_change_prev.rolling(window=25).mean()
    volume_spike = (volume_change_prev > 1.75 * avg_volume_change).astype(int)
    intraday_momentum_difference_by_inv_avg_volume = intraday_momentum_difference / average_volume_5
    intraday_momentum_difference_by_volume_spike = intraday_momentum_difference * volume_spike
    smoothed_intraday_factor = intraday_momentum_difference_by_volume_spike.rolling(window=4).mean().shift(1)
    
    # Composite Factor
    composite_factor = 0.4 * momentum_difference + 0.3 * enhanced_volume_change + 0.3 * price_action_factor
    
    # Final Alpha Signal
    adjusted_price_momentum = momentum_difference.ewm(span=5).mean().shift(1)
    weighted_volume_indicator = enhanced_volume_change.ewm(span=5).mean().shift(1)
    final_alpha_factor = (adjusted_price_momentum * smoothed_intraday_return * weighted_volume_indicator) + smoothed_intraday_factor
    
    return final_alpha_factor
