```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff().abs()
    
    # Calculate the Moving Average of Volume over Last 5 Days
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Volume Shock
    df['volume_shock'] = df['daily_volume_change'] / df['volume_ma_5']
    
    # Multiply Intraday Return by Volume Shock
    df['intraday_reversal_momentum'] = df['intraday_return'] * df['volume_shock']
    
    return df['intraday_reversal_momentum']
```
