```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Smoothed Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    smoothed_intraday_return = intraday_return.ewm(span=5, adjust=False).mean().shift(1)
    
    # Calculate the rate of change in close prices over 5 days
    close_price_change_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate the 10-day simple moving average (SMA) of close prices
    sma_10 = df['close'].rolling(window=10).mean()
    above_sma_10 = (df['close'] > sma_10).astype(int)
    
    # Sum of signed power (power=2) of the last 5 close prices minus the open price
    trend_strength = (df['close'] - df['open']).rolling(window=5).sum() ** 2
    
    # Analyze volume changes to identify significant trading activity
    volume_change = df['volume'] - df['volume'].shift(1)
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    unusual_volume = df['volume'] - avg_volume_30
    
    # Evaluate the relationship between volume and price
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume_5
    
    # Multiply Intraday Momentum Difference by Inverse of Average Volume
    inverse_avg_volume_5 = 1 / avg_volume_5
    intraday_momentum_diff = intraday_return * inverse_avg_volume_5
    
    # Detect Volume Spikes
    avg_volume_change_20 = volume_change.rolling(window=20).mean()
    volume_spike = (volume_change > 2 * avg_volume_change_20).astype(int)
    
    # Measure the daily price range (High[t] - Low[t])
    daily_range = df['high'] - df['low']
    
    # Combined indicator of market action
    combined_market_action = daily_range + abs(close_open_spread)
    
    # Calculate the 5-day average of daily ranges
    avg_daily_range_5 = daily_range.rolling(window=5).mean()
    
    # Integrate Next Day Open to Current Day Close
    next_day_open = df['open'].shift(-1)
    next_day_open_minus_close = next_day_open - df['close']
    
    # Final Alpha Signal
    alpha_signal = (
        intraday_high_low_spread +
        close_open_spread +
        smoothed_intraday_return +
        close_price_change_5d +
        above_sma_10 +
        trend_strength +
        unusual_volume +
        volume_ratio +
        intraday_momentum_diff +
        volume_spike +
        combined_market_action +
        avg_daily_range_5 +
        next_day_open_minus_close
    ).shift(1)  # Shift to prevent future data leakage
    
    # Apply a threshold to filter out small signals
    alpha_signal = alpha_signal[alpha_signal.abs() > 0.01]
    
    return alpha_signal
```
