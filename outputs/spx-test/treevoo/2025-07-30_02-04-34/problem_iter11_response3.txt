```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Smooth Intraday Return
    ema_intraday = intraday_momentum.ewm(span=5, adjust=False).mean().shift(1)
    intraday_return_diff = intraday_momentum - ema_intraday
    
    # Detect Volume Spikes
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    avg_vol_change = volume_change.rolling(window=20).mean()
    volume_spike = (volume_change > 1.5 * avg_vol_change).astype(int)
    
    # Combine Intraday Momentum and Volume Spike
    combined_factor = intraday_return_diff * volume_spike
    combined_factor = combined_factor.rolling(window=3).mean().shift(1)
    
    # Incorporate Close Price Trend
    close_price_change = df['close'].pct_change()
    ema_close_price = close_price_change.ewm(span=5, adjust=False).mean().shift(1)
    close_price_diff = close_price_change - ema_close_price
    
    # Integrate with the combined factor
    combined_factor *= close_price_diff
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Momentum Indicator
    momentum_indicator = high_low_spread / close_open_spread
    
    # Integrate Volume Impact
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    volume_adjusted_momentum = momentum_indicator * df['volume'] / avg_5_day_volume
    combined_factor *= volume_adjusted_momentum
    
    # Final Alpha Signal
    final_alpha = combined_factor * intraday_momentum * df['volume']
    
    return final_alpha
```
