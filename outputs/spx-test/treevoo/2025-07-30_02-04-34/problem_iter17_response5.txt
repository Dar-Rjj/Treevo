```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-term momentum (5-day)
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Long-term momentum (20-day)
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Momentum difference
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Volume change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Average volume over the last 5 days
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Volume change factor
    volume_change_factor = volume_change / average_volume
    
    # High-low range
    high_low_range = df['high'] - df['low']
    
    # Open-close spread
    open_close_spread = df['close'] - df['open']
    
    # Price action factor
    price_action_factor = open_close_spread / high_low_range
    
    # Simple Moving Averages
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    
    # Exponential Moving Averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    
    # Rate of Change (RoC) using closing prices
    roc = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume moving average (20-day)
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    
    # Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Composite factor
    composite_factor = (
        0.4 * momentum_difference + 
        0.3 * volume_change_factor + 
        0.3 * price_action_factor
    )
    
    # SMA and EMA crossover factor
    sma_ema_crossover = (sma_50 - ema_50) / df['close']
    
    # RSI and RoC combination factor
    rsi_roc_combination = 0.5 * rsi + 0.5 * roc
    
    # Final alpha factor
    alpha_factor = composite_factor + sma_ema_crossover + rsi_roc_combination
    
    return alpha_factor
```
