import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Smoothed Intraday Return
    smoothed_intraday_return = ((df['high'] - df['low']) / df['close']).ewm(span=5).mean().shift(1)
    
    # Incorporate Volume Intensity
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_intensity = (intraday_high_low_spread / avg_volume_5d).shift(1)
    
    # Detect Volume Spikes
    volume_change = df['volume'].pct_change()
    avg_vol_change_25d = volume_change.rolling(window=25).mean()
    vol_spike = (volume_change > 1.75 * avg_vol_change_25d).astype(int).shift(1)
    
    # Combine Intraday Momentum and Volume Impact
    combined_momentum_volume = (smoothed_intraday_return * vol_spike).rolling(window=4).mean().shift(1)
    
    # Calculate Momentum Indicator
    momentum_indicator = (intraday_high_low_spread / close_open_spread).shift(1)
    
    # Integrate Volume Impact
    weighted_volume_indicator = (df['volume'] / avg_volume_5d).shift(1)
    
    # Integrate Next Day Open to Current Day Close
    next_day_open_to_close = (df['open'].shift(-1) - df['close']).shift(1)
    
    # Final Alpha Signal
    final_alpha_signal = (
        (momentum_indicator * smoothed_intraday_return * weighted_volume_indicator) +
        combined_momentum_volume
    )
    
    return final_alpha_signal
