import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['close']
    
    # Smooth Intraday Return using EMA
    ema_intraday_return = intraday_return.ewm(span=14, adjust=False).mean().shift(1)
    intraday_momentum_diff = intraday_return - ema_intraday_return
    
    # Integrate Volume Impact
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    intraday_return_vol_adj = intraday_return * (1 / avg_volume_20)
    
    # Detect Volume Spikes
    volume_change = df['volume'].pct_change()
    avg_volume_change_50 = volume_change.rolling(window=50).mean()
    volume_spike = (volume_change > 3 * avg_volume_change_50).astype(int)
    
    # Combine Factors
    combined_factor = intraday_momentum_diff * volume_spike
    smoothed_combined_factor = combined_factor.rolling(window=10).mean().shift(1)
    final_smoothed_factor = smoothed_combined_factor * (1 / avg_volume_20)
    
    # Calculate Raw Momentum
    raw_momentum = df['close'].diff()
    
    # Adjust Raw Momentum
    adjusted_momentum = raw_momentum * (1 if volume_spike == 1 else 1)
    
    # Combine Intraday Factors
    intraday_high_low_spread = df['high'] - df['low']
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Weighted Price Movement
    weighted_price_movement = (df['close'] - df['open']) * df['volume']
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    weighted_price_movement_adj = weighted_price_movement / avg_volume_5
    
    # Integrate Volume Impact
    intraday_momentum_vol_adj = intraday_return * (df['volume'] / avg_volume_5)
    
    # Initial Alpha Signal
    initial_alpha_signal = final_smoothed_factor * df['volume']
    
    # Compute Smoothed Price Momentum
    smoothed_price_momentum = df['close'].diff().rolling(window=5).mean()
    
    # Compute Weighted Volume Indicator
    volume_change_sign = np.sign(volume_change)
    weighted_volume_indicator = (volume_change * volume_change_sign).rolling(window=5).mean()
    
    # Incorporate Close Price Trend
    close_price_change = df['close'].diff()
    ema_close_price_change = close_price_change.ewm(span=5, adjust=False).mean().shift(1)
    close_price_trend = close_price_change - ema_close_price_change
    
    # Final Alpha Factor
    final_alpha = (
        adjusted_momentum * 
        (intraday_high_low_spread + close_to_open_return) * 
        weighted_price_movement_adj * 
        initial_alpha_signal * 
        smoothed_price_momentum * 
        weighted_volume_indicator * 
        close_price_trend
    )
    
    # Adjust for Volatility
    std_close_10 = df['close'].rolling(window=10).std()
    final_alpha_vol_adj = final_alpha / (1 + std_close_10)
    
    return final_alpha_vol_adj
