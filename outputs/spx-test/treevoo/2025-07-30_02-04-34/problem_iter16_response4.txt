```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Momentum
    n = 10  # Number of days for raw momentum
    df['raw_momentum'] = df['close'].diff()

    # Identify Volume Spikes
    m = 10  # Number of days for volume moving average
    df['volume_ma'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['volume_ma']).astype(int)

    # Adjust Raw Momentum
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - 0.5 * df['volume_spike'])

    # Combine Intraday Factors
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['close_to_open_return'] = df['close'] - df['open']

    # Calculate Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    df['average_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['weighted_price_movement_normalized'] = df['weighted_price_movement'] / df['average_volume_5_days']

    # Integrate Volume Impact
    df['volume_impact'] = df['volume'] / df['average_volume_5_days']

    # Initial Alpha Signal
    df['initial_alpha_signal'] = df['adjusted_momentum'] * df['volume']

    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['close'].diff().rolling(window=5).sum() / 5

    # Compute Weighted Volume Indicator
    df['volume_change'] = df['volume'].diff()
    df['signed_volume_change'] = df['volume_change'] * df['volume_change'].apply(lambda x: 1 if x >= 0 else -1)
    df['weighted_volume_indicator'] = df['signed_volume_change'].rolling(window=5).mean()

    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'].diff()
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['close_price_trend'] = df['close_price_change'] - df['smoothed_close_price_change']

    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['adjusted_momentum'] * 
        (df['intraday_high_low_spread'] + df['close_to_open_return']) * 
        df['weighted_price_movement_normalized'] * 
        df['initial_alpha_signal'] * 
        df['smoothed_price_momentum'] * 
        df['weighted_volume_indicator'] * 
        df['close_price_trend']
    )

    # Adjust for Volatility
    df['std_dev_close'] = df['close'].rolling(window=10).std()
    df['final_alpha_factor'] = df['final_alpha_factor'] / (1 + df['std_dev_close'])

    return df['final_alpha_factor']
```
