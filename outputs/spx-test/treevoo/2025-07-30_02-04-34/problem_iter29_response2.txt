import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-Based Factor
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['momentum_difference'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Volume and Price Change Analysis
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    df['unusual_volume'] = df['volume'] - df['avg_volume_30']
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['enhanced_volume_change'] = df['volume_change'] * df['volume_ratio']
    
    # Enhanced Price Action Factor
    df['daily_range'] = df['high'] - df['low']
    df['daily_spread'] = df['close'] - df['open']
    df['high_close_spread'] = df['high'] - df['close']
    df['price_action_factor'] = (df['daily_spread'] / df['daily_range']) - df['high_close_spread']
    df['combined_indicator'] = df['daily_range'] + abs(df['daily_spread'])
    
    # Intraday Analysis
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['close_open_spread'] = df['close'] - df['open']
    df['intraday_return_smoothed'] = df['intraday_high_low_spread'] / df['close']
    df['intraday_return_smoothed_ema'] = df['intraday_return_smoothed'].ewm(span=5, adjust=False).mean().shift(1)
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['intraday_momentum_difference'] = df['momentum_difference'] * (1 / df['avg_volume_5'])
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change_25'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = (df['volume_change'] > 1.75 * df['avg_volume_change_25']).astype(int)
    df['intraday_momentum_vol_spike'] = df['intraday_momentum_difference'] * df['volume_spike']
    df['intraday_combined_factor'] = df['intraday_momentum_vol_spike'].rolling(window=4).mean().shift(1)
    
    # Composite Factor
    df['composite_factor'] = (0.4 * df['momentum_difference'] +
                              0.3 * df['enhanced_volume_change'] +
                              0.3 * df['price_action_factor'])
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] - df['close'].shift(1)
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean().shift(1)
    df['close_price_change_diff'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Introduce Reversal Indicator
    df['long_term_momentum_reversal'] = df['close'].pct_change().rolling(window=20).sum() / 20
    df['short_term_momentum_reversal'] = df['close'].pct_change().rolling(window=5).sum() / 5
    df['reversal_indicator'] = df['short_term_momentum_reversal'] - df['long_term_momentum_reversal']
    
    # Final Alpha Signal
    df['adjusted_price_momentum'] = df['momentum_difference'] * df['smoothed_close_price_change']
    df['weighted_volume_indicator'] = df['enhanced_volume_change'] * df['intraday_combined_factor']
    df['final_alpha_signal'] = (df['adjusted_price_momentum'] * df['weighted_volume_indicator'] +
                                df['intraday_combined_factor'] +
                                df['close_price_change_diff'] +
                                df['reversal_indicator'])
    
    return df['final_alpha_signal']
