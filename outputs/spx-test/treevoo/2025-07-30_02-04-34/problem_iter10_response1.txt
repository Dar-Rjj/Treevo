```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Smooth Intraday Return
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, adjust=False).mean()
    df['intraday_return_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['avg_volume_change'] = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = (df['volume_change'] > 1.5 * df['avg_volume_change']).astype(int)
    
    # Combine Intraday Momentum and Volume Spike
    df['intraday_momentum_signal'] = df['intraday_return_diff'] * df['volume_spike']
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'] / df['close'].shift(1)
    df['smoothed_close_price_change'] = df['close_price_change'].ewm(span=5, adjust=False).mean()
    df['close_price_change_diff'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Combine All Factors
    df['combined_factors'] = df['intraday_momentum_signal'] * df['close_price_change_diff']
    
    # Final Alpha Signal
    df['final_alpha_factor'] = df['combined_factors'] * df['volume']
    
    return df['final_alpha_factor']
```
