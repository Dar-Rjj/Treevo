```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    
    # Smooth Intraday Return with EMA (span=5)
    df['intraday_return_smoothed'] = df['intraday_return'].ewm(span=5).mean()
    df['intraday_momentum_diff'] = df['intraday_return'] - df['intraday_return_smoothed'].shift(1)
    
    # Incorporate Volume Intensity
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_intensity'] = 1 / df['avg_volume_5d']
    df['intraday_momentum_diff_volume_adjusted'] = df['intraday_momentum_diff'] * df['volume_intensity']
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['avg_volume_change_25d'] = df['volume_change'].rolling(window=25).mean()
    df['volume_spike'] = (df['volume_change'] > 1.75 * df['avg_volume_change_25d']).astype(int)
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_factor'] = df['intraday_momentum_diff_volume_adjusted'] * df['volume_spike']
    
    # Smooth the combined factor with a simple moving average (window=4)
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=4).mean().shift(1)
    
    # Final Alpha Signal
    df['alpha_signal'] = df['smoothed_combined_factor'] * df['volume_intensity']
    
    return df['alpha_signal']
```
