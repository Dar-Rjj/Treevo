import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the computed alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the daily return
    df['ma_30_daily_return'] = df['daily_return'].rolling(window=30).mean()
    
    # Calculate the 10-day moving average of the volume
    df['ma_10_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the volume
    df['ma_30_volume'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14_close'] = 100 - (100 / (1 + rs))
    
    # Heuristic factor: (10-day MA of daily return - 30-day MA of daily return) * (RSI - 50) * (10-day MA of volume / 30-day MA of volume)
    df['heuristic_factor'] = (df['ma_10_daily_return'] - df['ma_30_daily_return']) * (df['rsi_14_close'] - 50) * (df['ma_10_volume'] / df['ma_30_volume'])
    
    # Return the heuristic factor as a Series
    return df['heuristic_factor']
