import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low'] = df['high'] - df['low']
    
    # Calculate Close-Open Spread
    df['close_open_spread'] = df['close'] - df['open']
    
    # Calculate Raw Momentum (difference between current and previous close)
    df['raw_momentum'] = df['close'].diff()
    
    # Adjust Price Momentum by Volume
    n = 5
    df['weighted_avg_volume'] = df['volume'].rolling(window=n).mean()
    df['adjusted_volume'] = df['volume'] / df['weighted_avg_volume']
    df['price_momentum_adjusted_by_volume'] = df['raw_momentum'] * df['adjusted_volume']
    
    # Incorporate Volume Intensity
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['volume_intensity'] = 1 / df['avg_volume_5']
    df['volume_intensity_momentum'] = df['raw_momentum'] * df['volume_intensity']
    
    # Detect Volume Spikes
    m = 20
    df['volume_ma_20'] = df['volume'].rolling(window=m).mean()
    df['volume_spike'] = df['volume'] > 1.5 * df['volume_ma_20']
    df['dampened_momentum'] = df['raw_momentum'].where(~df['volume_spike'], df['raw_momentum'] * 0.9)
    
    # Combine Intraday Factors
    df['intraday_factor'] = df['intraday_high_low'] + df['close_open_spread']
    
    # Compute Next Day Open to Current Day Close
    df['next_day_open'] = df['open'].shift(-1)
    df['next_day_open_current_close'] = df['next_day_open'] - df['close']
    
    # Combine for Initial Alpha Signal
    df['initial_alpha_signal'] = (df['intraday_factor'] + df['next_day_open_current_close']) * df['volume']
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()

    df['money_flow'] = (df['close'] - df['open']) * df['volume']

    # Calculate Intraday Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['intraday_range'].rolling(window=5).mean()
    
    # Final Alpha Signal
    df['final_alpha_signal'] = (
        df['dampened_momentum'] * 
        df['smoothed_price_momentum'] * 
        df['weighted_volume_indicator'] + 
        df['initial_alpha_signal'] * 
        df['intraday_factor'] * 
        df['weighted_price_movement'] *

    )
