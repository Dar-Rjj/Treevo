```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']

    # Smooth Intraday Return with EMA (span=7, exclude latest value)
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=7, adjust=False).mean().shift(1)

    # Difference between actual and smoothed return
    df['intraday_momentum_diff'] = df['intraday_return'] - df['smoothed_intraday_return']

    # Incorporate Volume Intensity
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['inverse_avg_volume'] = 1 / df['avg_volume']
    df['volume_integrated_return'] = df['intraday_return'] * df['inverse_avg_volume']

    # Detect Volume Spikes
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change'] = df['volume_change'].rolling(window=30).mean()
    df['volume_spike'] = (df['volume_change'] > 2 * df['avg_volume_change']).astype(int)

    # Combine Factors
    df['combined_factor'] = df['intraday_momentum_diff'] * df['volume_spike']
    
    # Smooth combined factor using simple moving average (window=5, exclude most recent value)
    df['smoothed_combined_factor'] = df['combined_factor'].rolling(window=5).mean().shift(1)
    
    # Multiply final smoothed factor by Inverse of Average Volume
    df['final_factor'] = df['smoothed_combined_factor'] * df['inverse_avg_volume']

    return df['final_factor'].dropna()
```
