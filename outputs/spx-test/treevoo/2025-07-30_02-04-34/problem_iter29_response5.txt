```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate short-term and long-term momentum
    df['short_term_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['long_term_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['momentum_difference'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Raw momentum
    df['raw_momentum'] = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # Smoothed price momentum
    df['smoothed_price_momentum'] = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # Volume-Adjusted Momentum and Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['average_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_change_factor'] = df['volume_change'] / df['average_volume_5d']
    
    # Volume spike detection
    df['average_volume_change_20d'] = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)) > 2 * df['average_volume_change_20d']
    df['adjusted_raw_momentum'] = np.where(df['volume_spike'], df['raw_momentum'] * 0.5, df['raw_momentum'])
    
    # Price Action and Intraday Analysis
    df['daily_range'] = df['high'] - df['low']
    df['daily_spread'] = df['close'] - df['open']
    df['price_action_factor'] = df['daily_spread'] / df['daily_range']
    
    # Intraday return
    df['intraday_return'] = (df['high'] - df['low']) / df['close']
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5, min_periods=5).mean().shift(1)
    
    # Multiply smoothed intraday return by inverse of average volume
    df['inverse_avg_volume'] = 1 / df['average_volume_5d']
    df['intraday_return_impact'] = df['smoothed_intraday_return'] * df['inverse_avg_volume']
    
    # Dynamic Volume Intensity and Median Volume
    df['dynamic_volume_weight'] = df['volume'] / df['volume'].rolling(window=5).median()
    df['median_dynamic_volume_weight'] = df['dynamic_volume_weight'].rolling(window=5).median()
    
    # Combine factors
    df['composite_alpha'] = (
        0.3 * df['smoothed_price_momentum'] +
        0.2 * df['adjusted_raw_momentum'] +
        0.1 * df['price_action_factor'] +
        0.4 * df['intraday_return_impact'] * df['median_dynamic_volume_weight']
    )
    
    return df['composite_alpha']
```
