```python
import pandas as pd

def heuristics_v2(df):
    # Fundamental Analysis
    close_open_ratio = df['close'] / df['open']
    daily_returns = df['close'].pct_change()
    ma_daily_returns = daily_returns.rolling(window=10).mean()
    
    # Technical Indicators
    roc_5_days = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    price_range_vol = (df['high'] - df['low']) / df['volume']
    
    # Volume Analysis
    vol_close_open_diff = df['volume'] / (df['close'] - df['open'])
    amount_roc_7_days = (df['amount'] - df['amount'].shift(7)) / df['amount'].shift(7)
    corr_amount_roc = df['close'].pct_change(7).rolling(window=30).corr(amount_roc_7_days)
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        close_open_ratio * ma_daily_returns +
        roc_5_days + 
        price_range_vol + 
        vol_close_open_diff * corr_amount_roc
    )
    
    return alpha_factor
```
