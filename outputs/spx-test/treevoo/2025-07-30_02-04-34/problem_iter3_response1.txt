```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate 7-day Price Momentum
    price_momentum = df['close'].pct_change(periods=7)

    # Calculate 7-day Volume Momentum
    volume_momentum = df['volume'].pct_change(periods=7)

    # Adjust Price Momentum by Volume Momentum
    adjusted_momentum = price_momentum * volume_momentum

    # Apply Volume-Adjusted Momentum to High-Low Range
    weighted_range = high_low_range * adjusted_momentum

    # Sum the weighted range over a lookback period (e.g., 20 days)
    lookback_period = 20
    factor = weighted_range.rolling(window=lookback_period).sum()

    return factor
```
