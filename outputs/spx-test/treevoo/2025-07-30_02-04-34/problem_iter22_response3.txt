```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, spike_threshold=1.5, momentum_dampener=0.5):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['Daily_Price_Change'].rolling(window=n).sum() / n
    
    # Identify Volume Spikes
    df['Volume_MA'] = df['volume'].rolling(window=m).mean()
    df['Volume_Spike'] = (df['volume'] > df['Volume_MA'] * spike_threshold).astype(int)
    
    # Adjust Momentum by Volume Spike
    df['Adjusted_Momentum'] = df.apply(
        lambda row: row['Raw_Momentum'] * momentum_dampener if row['Volume_Spike'] == 1 else row['Raw_Momentum'],
        axis=1
    )
    
    # Calculate Smoothed Price Momentum
    df['Smoothed_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).sum() / 5
    
    # Compute Volume Weight
    df['Volume_Weight'] = df['volume'].rolling(window=n).sum() / n
    
    # Combine Smoothed Price Momentum and Volume Weight
    df['Combined_Smoothed_Momentum'] = df['Smoothed_Price_Momentum'] * df['Volume_Weight']
    
    # Calculate Intraday High-Low Spread Factor
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread Factor and Close-to-Open Return
    df['Combined_Intraday_Factor'] = df['Intraday_High_Low_Spread'] + df['Close_to_Open_Return']
    
    # Calculate Weighted Price Movement
    df['Weighted_Price_Movement'] = (df['close'] - df['open']) * df['volume']
    df['Average_Volume'] = df['volume'].rolling(window=5).sum() / 5
    df['Normalized_Weighted_Price_Movement'] = df['Weighted_Price_Movement'] / df['Average_Volume']
    
    # Incorporate Next Day Open to Current Day Close
    df['Next_Day_Open_to_Close'] = df['open'].shift(-1) - df['close']
    
    # Final Alpha Factor
    df['Base_Momentum_Signal'] = df['Adjusted_Momentum'] * df['Combined_Intraday_Factor'] * df['Normalized_Weighted_Price_Movement']
    df['Close_Price_Change'] = df['close'] - df['close'].shift(1)
    df['EMA_Close_Price_Change'] = df['Close_Price_Change'].ewm(span=5, adjust=False).mean().shift(1)
    df['Difference_Close_Price_Change'] = df['Close_Price_Change'] - df['EMA_Close_Price_Change']
    df['Final_Alpha_Factor'] = df['Base_Momentum_Signal'] + df['Difference_Close_Price_Change']
    
    # Integrate Volume Intensity
    df['Volume_Intensity'] = df['volume'].rolling(window=5).sum() / 5
    df['Final_Alpha_Factor'] *= df['Volume_Intensity']

    return df['Final_Alpha_Factor']
```
