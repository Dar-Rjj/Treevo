```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['open']) / df['open']
    
    # Raw Momentum
    df['raw_momentum'] = df['close'].diff()
    
    # Detect Volume Spikes
    df['volume_change'] = df['volume'].pct_change().fillna(0)
    avg_volume_change = df['volume_change'].rolling(window=20).mean()
    df['volume_spike'] = df['volume_change'] > 2 * avg_volume_change
    
    # Adjust Raw Momentum
    df['adjusted_momentum'] = df['raw_momentum'] * (1 - df['volume_spike'] * 0.5)
    
    # Weighted Price Movement
    df['weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    
    # Integrate Volume Impact
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    df['volume_impact'] = (df['intraday_high_low_spread'] * df['volume']) / avg_5_day_volume
    
    # Smooth Intraday Return
    df['smoothed_intraday_return'] = df['intraday_return'].ewm(span=5).mean()
    df['intraday_return_diff'] = df['intraday_return'] - df['smoothed_intraday_return']
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = df['intraday_high_low_spread'] + df['intraday_return_diff']
    
    # Initial Alpha Signal
    df['initial_alpha_signal'] = df['adjusted_momentum'] * df['combined_intraday_factors'] * df['weighted_price_movement']
    
    # Compute Smoothed Price Momentum
    df['smoothed_price_momentum'] = df['close'].diff().rolling(window=5).sum() / 5
    
    # Compute Weighted Volume Indicator
    df['volume_change_sign'] = np.sign(df['volume_change'])
    df['weighted_volume_indicator'] = (df['volume_change'] * df['volume_change_sign']).rolling(window=5).mean()
    
    # Incorporate Close Price Trend
    df['close_price_change'] = df['close'].diff()
    df['smoothed_close_price_change'] = df['close_price_change'].shift(1).ewm(span=7).mean()
    df['close_price_trend'] = df['close_price_change'] - df['smoothed_close_price_change']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['initial_alpha_signal'] * 
        df['smoothed_price_momentum'] * 
        df['weighted_volume_indicator'] * 
        df['close_price_trend']
    )
    
    # Adjust for Volatility
    df['std_close'] = df['close'].rolling(window=10).std()
    df['final_alpha_factor'] = df['final_alpha_factor'] / (1 + (df['std_close'] * 1.5))
    
    # Combine Intraday Momentum and Volume Impact
    df['combined_intraday_momentum'] = df['intraday_return_diff'] * df['volume_spike']
    df['smoothed_combined_factor'] = df['combined_intraday_momentum'].shift(1).rolling(window=4).mean()
    df['inverse_avg_volume'] = 1 / avg_5_day_volume
    df['final_alpha_factor'] *= df['smoothed_combined_factor'] * df['inverse_avg_volume']
    
    return df['final_alpha_factor']
```
