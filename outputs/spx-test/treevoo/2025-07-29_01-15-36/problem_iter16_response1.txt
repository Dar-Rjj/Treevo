```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, volume_threshold=50000):
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['High_Low_Diff'] = df['high'] - df['low']
    recent_high_low_mean = df['High_Low_Diff'].rolling(window=n).mean()
    recent_high_low_std = df['High_Low_Diff'].rolling(window=n).std()
    
    # Adjust Intraday Return by Intraday Volatility
    df['Intraday_Volatility'] = recent_high_low_mean + recent_high_low_std
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['Intraday_Volatility']
    
    # Determine Range-to-Range Ratio
    df['Prev_High_Low_Range'] = df['High_Low_Diff'].shift(1)
    df['Range_to_Range_Ratio'] = df['High_Low_Diff'] / df['Prev_High_Low_Range']
    
    # Integrate Volume Sensitivity
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Adjusted_Range_to_Range_Ratio'] = df['Range_to_Range_Ratio'] * df['Volume_Ratio']
    
    # Combine Adjusted Intraday Return and Adjusted Range-to-Range Ratio
    df['Combined_Intraday_Factors'] = df['Adjusted_Intraday_Return'] * df['Adjusted_Range_to_Range_Ratio']
    
    # Detect Volume Shock
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Shock'] = df['Volume_Change'] > volume_threshold
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['Price_Change'] = df['close'].diff()
    df['Upward_Change'] = df['Price_Change'].apply(lambda x: x if x > 0 else 0)
    df['Downward_Change'] = -df['Price_Change'].apply(lambda x: x if x < 0 else 0)
    avg_up = df['Upward_Change'].rolling(window=n).mean()
    avg_down = df['Downward_Change'].rolling(window=n).mean()
    df['RS'] = avg_up / avg_down.replace(0, np.nan)
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Volume-Adjusted Momentum
    df['Volume_Adjusted_Momentum'] = (df['RSI'] * df['volume']) + (df['Intraday_Return'] * df['Volume_Ratio'])
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Volume_Adjusted_Momentum'] * df['Combined_Intraday_Factors']
    
    return df['Final_Factor']
```
