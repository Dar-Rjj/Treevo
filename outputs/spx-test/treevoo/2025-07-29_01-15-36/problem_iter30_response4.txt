```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate EMAs
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Momentum indicator
    momentum = short_ema - long_ema
    
    # Volume change
    volume_change = df['volume'].pct_change()
    
    # RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # ATR
    tr1 = df['high'] - df['low']
    tr2 = np.abs(df['high'] - df['close'].shift())
    tr3 = np.abs(df['low'] - df['close'].shift())
    true_range = np.max(np.vstack((tr1, tr2, tr3)), axis=0)
    atr = pd.Series(true_range).rolling(window=14).mean()
    
    # Daily return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Percentage range
    percentage_range = (df['high'] - df['low']) / df['close']
    
    # Cumulative volume
    cumulative_volume = (df['high'] - df['low']).mul(df['volume']).cumsum()
    
    # Volume-price trend
    vpt = (df['close'].diff() * df['volume']).cumsum()
    
    # Composite alpha factor
    composite_alpha_factor = (
        momentum * volume_change +
        rsi +
        daily_return +
        percentage_range +
        cumulative_volume +
        vpt
    ) / atr
    
    return composite_alpha_factor
```
