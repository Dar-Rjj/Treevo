```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    intraday_combined = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_momentum = close_to_range_ratio * df['volume']
    
    # Final Combination
    intermediate_alpha = intraday_combined * volume_weighted_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Accumulation Distribution
    # Calculate Volume Trend
    volume_trend = df['volume'].diff(1)
    
    # Calculate Average True Range Over 20 Days
    true_range = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    atr_20 = true_range.rolling(window=20).mean()
    
    # Calculate Accumulation Distribution Line
    accumulation_distribution = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    accumulation_distribution = (accumulation_distribution * df['volume']).rolling(window=20).sum()
    
    # Calculate Price Volatility
    price_volatility = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    final_alpha = intermediate_alpha * volume_trend
    final_alpha = final_alpha * (atr_20 * price_volatility)
    final_alpha = final_alpha * accumulation_distribution
    
    return final_alpha
```
