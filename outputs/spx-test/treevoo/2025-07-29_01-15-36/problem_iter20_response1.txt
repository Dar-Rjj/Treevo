```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices over 20 days
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_20
    
    # Calculate Average Volume Over 20 Days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range_20 = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine Volume and Price Volatility
    combined_vol_price_vol = avg_volume_20 * avg_high_low_range_20
    
    # Weight Momentum Factor by Combined Volume & Price Volatility
    weighted_momentum_factor = momentum_factor * combined_vol_price_vol
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range = close_to_range_ratio * df['volume']
    
    # Final Combination
    final_combination = combined_intraday_indicators * volume_weighted_close_to_range
    
    # Adjust for Volume Impact, Price Volatility, and Volume Shock
    # Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Average True Range Over 20 Days
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    avg_true_range_20 = true_range.rolling(window=20).mean()
    
    # Volume Shock
    volume_shock = df['volume'] - df['volume'].shift(5)
    
    # Final Adjustment
    final_adjustment = final_combination * volume_trend * (1 / avg_true_range_20) + volume_shock
    
    return final_adjustment
```
