```python
def heuristics_v2(data):
    # Calculate the price momentum based on the close price
    data['price_momentum'] = data['close'].pct_change(periods=1)
    
    # Calculate the average volume over the last 5 days
    data['avg_volume_5d'] = data['volume'].rolling(window=5).mean()
    
    # Calculate the volatility of the high and low prices
    data['volatility_hl'] = (data['high'] - data['low']) / data['close']
    
    # Combine the factors to create the final heuristic
    data['heuristic_factor'] = (data['price_momentum'] * 
                                data['avg_volume_5d'] * 
                                data['volatility_hl'])
    
    return data['heuristic_factor'].dropna()
```
