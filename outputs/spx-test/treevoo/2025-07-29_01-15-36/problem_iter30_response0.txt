```python
import pandas as pd

def heuristics_v2(df):
    """
    The function `heuristics_v2` generates a novel and interpretable alpha factor based on the given
    DataFrame, which includes the columns: open, high, low, close, and volume. The factor is calculated
    using a combination of price and volume data to predict future stock returns.

    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, volume).

    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """

    # Calculate the average price for the day
    avg_price = (df['high'] + df['low'] + df['close']) / 3

    # Calculate the change in average price from the previous day
    avg_price_change = avg_price - avg_price.shift(1)

    # Calculate the volume change from the previous day
    volume_change = df['volume'] - df['volume'].shift(1)

    # Calculate the factor as a weighted combination of price and volume changes
    factor = 0.7 * avg_price_change + 0.3 * volume_change

    return factor
```
