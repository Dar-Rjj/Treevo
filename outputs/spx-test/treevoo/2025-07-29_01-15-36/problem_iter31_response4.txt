```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) of Close prices
    close_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum Factor
    momentum = close_ema - df['close']
    
    # Calculate EMA of Volume
    volume_ema = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate EMA of High-Low Range
    high_low_range = df['high'] - df['low']
    high_low_ema = high_low_range.ewm(span=20, adjust=False).mean()
    
    # Weight by Combined Volume & Price Volatility
    combined_weight = volume_ema * high_low_ema
    
    # Final factor
    factor = momentum * combined_weight
    
    return factor
```
