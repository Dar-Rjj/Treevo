```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=20).mean()

    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_close
    
    # Calculate Average Volume Over 20 Days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine Volume and Price Volatility
    combined_volatility = avg_volume * avg_high_low_range
    
    # Weight Momentum by Combined Volume & Price Volatility
    weighted_momentum = momentum_factor * combined_volatility
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = intraday_high_low_spread * open_to_close_momentum + open_to_close_momentum
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Price Range
    ma_price_range = daily_price_range.rolling(window=20).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    up_changes = (df['close'] - df['close'].shift(1)).clip(lower=0)
    down_changes = (df['close'].shift(1) - df['close']).clip(lower=0)
    avg_up_change = up_changes.rolling(window=14).mean()
    avg_down_change = down_changes.rolling(window=14).mean()
    rs = avg_up_change / avg_down_change
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = ma_price_range * df['volume'] + rsi * df['volume']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / intraday_high_low_spread
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = df['volume'] * close_to_range_ratio
    
    # Final Combination
    final_combination = combined_intraday_indicators * volume_weighted_close_to_range_momentum + volume_adjusted_momentum
    
    # Adjust for Volume Impact
    volume_trend = df['volume'] - df['volume'].shift(1)
    final_alpha_factor = final_combination * volume_trend
    
    return final_alpha_factor
```
