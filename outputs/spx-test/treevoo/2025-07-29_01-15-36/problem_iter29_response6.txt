```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price within the day's range
    df['day_range'] = df['high'] - df['low']
    df['close_position'] = (df['close'] - df['low']) / df['day_range']
    
    # Calculate the average true range (ATR) over a 14-day window
    df['true_range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the buying pressure and selling pressure
    df['buying_pressure'] = df['close'] - df[['open', 'low']].min(axis=1)
    df['selling_pressure'] = df[['high', 'open']].max(axis=1) - df['close']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_position'] * df['avg_true_range'] * 
                          (df['buying_pressure'] - df['selling_pressure']))
    
    return df['alpha_factor']
```
