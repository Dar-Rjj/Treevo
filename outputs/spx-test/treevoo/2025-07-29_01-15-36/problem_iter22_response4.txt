```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    mean_high_low_diff = high_low_diff.rolling(window=20).mean()
    std_high_low_diff = high_low_diff.rolling(window=20).std()
    intraday_volatility = mean_high_low_diff + std_high_low_diff
    
    # Adjust Intraday Return by Intraday Volatility
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Determine Range-to-Range Ratio
    range_to_range_ratio = (df['high'] - df['low']).div(df['high'].shift(1) - df['low'].shift(1))
    
    # Integrate Volume Sensitivity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_range_to_range_ratio = range_to_range_ratio * volume_ratio
    
    # Combine Adjusted Intraday Return and Adjusted Range-to-Range Ratio
    combined_intraday_factors = adjusted_intraday_return * adjusted_range_to_range_ratio
    
    # Detect Volume Shock
    volume_change = df['volume'] - df['volume'].shift(1)
    sudden_volume_increase = volume_change > (volume_change.rolling(window=20).std() * 2)
    
    # Calculate N-Day Relative Strength Index (RSI)
    up_down = df['close'].diff()
    up = up_down.clip(lower=0)
    down = -up_down.clip(upper=0)
    avg_up = up.rolling(window=14).mean()
    avg_down = down.rolling(window=14).mean()
    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_20
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (rsi * df['volume']) + (intraday_return * volume_ratio)
    
    # Final Combination
    final_combination = volume_adjusted_momentum * combined_intraday_factors * momentum_factor
    
    # Adjust for Volume Impact and Enhanced Price Volatility
    volume_trend = df['volume'] - df['volume'].shift(1)
    tr = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    atr = tr.rolling(window=20).mean()
    final_adjustment = final_combination * volume_trend * (df['volume'] / atr)
    
    return final_adjustment
```
