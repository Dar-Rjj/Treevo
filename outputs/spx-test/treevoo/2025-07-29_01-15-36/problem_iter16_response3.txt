```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['daily_price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (e.g., 5 days)
    window_size = 5
    df['sum_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI) (e.g., 14 days)
    n_days = 14
    df['price_change'] = df['close'].diff()
    df['upward_change'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['downward_change'] = -df['price_change'].apply(lambda x: x if x < 0 else 0)
    
    df['ema_up'] = df['upward_change'].ewm(span=n_days, adjust=False).mean()
    df['ema_down'] = df['downward_change'].ewm(span=n_days, adjust=False).mean()
    
    df['rs'] = df['ema_up'] / df['ema_down'].abs()
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Combine Intraday Indicators
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['open_to_close_momentum'] = df['close'] - df['open']
    df['combined_intraday_indicators'] = df['intraday_high_low_spread'] * df['open_to_close_momentum'] + df['open_to_close_momentum']
    
    # Compute Average True Range
    df['true_range'] = df[['high' - 'low', abs('high' - 'close').shift(1), abs('low' - 'close').shift(1)]].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=n_days).mean()
    
    # Determine Volume Trend
    m_days = 5
    df['volume_trend'] = (df['volume'].rolling(window=m_days).sum() > df['volume'].shift(m_days).rolling(window=m_days).sum()).astype(int) * 2 - 1
    
    # Calculate Price Momentum
    p_days = 5
    df['price_momentum'] = df['close'] - df['close'].rolling(window=p_days).mean()
    
    # Enhance Volatility Measure
    q_days = 20
    df['volatility'] = df['close'].rolling(window=q_days).std() * 1.5
    
    # Combine Factors
    df['combined_factors'] = (df['intraday_high_low_spread'] * df['atr']) * df['volume_trend'] + df['price_momentum']
    
    # Final Combination
    df['final_combination'] = df['combined_intraday_indicators'] * df['volume_weighted_momentum'] + df['sum_volume_weighted_momentum']
    
    # Final Adjustment
    r_days = 10
    df['trend_adjustment'] = (df['close'] - df['close'].shift(r_days)).rolling(window=r_days).mean()
    df['final_adjustment'] = df['final_combination'] * df['volatility'] + df['trend_adjustment']
    
    # Integrate All Components
    df['alpha_factor'] = df['rsi'] + df['final_adjustment']
    
    return df['alpha_factor']
```
