```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    df['ema_price_range'] = df['price_range'].ewm(span=n, adjust=False).mean()
    
    # Calculate Upward and Downward Price Changes
    df['up_change'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['down_change'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    
    # Calculate Exponential Average Upward and Downward Change
    df['ema_up_change'] = df['up_change'].ewm(span=n, adjust=False).mean()
    df['ema_down_change'] = df['down_change'].ewm(span=n, adjust=False).mean()
    
    # Compute RS (Relative Strength)
    df['rs'] = df['ema_up_change'] / df['ema_down_change'].abs()
    
    # Calculate RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Volume Adjusted Momentum
    df['volume_adjusted_ema_price_range'] = df['ema_price_range'] * df['volume']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume']
    
    # Combine Results
    df['factor'] = df['volume_adjusted_ema_price_range'] + df['volume_adjusted_rsi']
    
    return df['factor']
```
