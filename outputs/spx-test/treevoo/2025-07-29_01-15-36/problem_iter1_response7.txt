```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Sum of 10-day and 20-day Daily Returns
    df['momentum_10d'] = df['daily_return'].rolling(window=10).sum()
    df['momentum_20d'] = df['daily_return'].rolling(window=20).sum()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Determine Sign of Volume Change
    df['volume_sign'] = df['volume_change'].apply(lambda x: 1 if x > 0 else -1)
    
    # Price Momentum with Volume Confirmation
    df['price_momentum_with_volume_confirmation'] = (df['momentum_10d'] + df['momentum_20d']) * df['volume_sign']
    
    return df['price_momentum_with_volume_confirmation']
```
