```python
def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_range = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    
    volume_shock_threshold = 2.0  # Example threshold, can be adjusted
    amount_shock_threshold = 2.0  # Example threshold, can be adjusted
    
    # Identify Sudden Increase in Volume and Amount
    volume_shock = (df['volume_change'] / df['volume'].shift(1)) > volume_shock_threshold
    amount_shock = (df['amount_change'] / df['amount'].shift(1)) > amount_shock_threshold
    
    # Combine Intraday Momentum and Volume/Amount Shock
    factor = (intraday_range * volume_shock) + (intraday_range * amount_shock)
    
    return factor
```
