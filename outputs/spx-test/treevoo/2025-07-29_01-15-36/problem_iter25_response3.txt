```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Final Intraday Combination
    final_intraday_combination = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact and Enhanced Price Volatility
    # Calculate Volume Trend
    volume_trend = df['volume'].diff()
    
    # Calculate Average True Range Over 20 Days
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    average_true_range_20d = df['true_range'].rolling(window=20).mean()
    
    # Calculate Average Volume Over 20 Days
    average_volume_20d = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    average_high_low_range_20d = high_low_range.rolling(window=20).mean()
    
    # Weight by Combined Volume & Enhanced Price Volatility
    volume_volatility_weight = average_volume_20d * average_high_low_range_20d
    
    # Final Adjustment
    final_adjustment = final_intraday_combination * volume_trend * volume_volatility_weight
    
    # Calculate Simple Moving Average (SMA) of Close prices
    sma_20d = df['close'].rolling(window=20).mean()
    
    # Calculate SMA Momentum
    sma_momentum = sma_20d - df['close']
    
    # Integrate RSI and SMA Momentum
    # Calculate N-Day Relative Strength Index (RSI)
    up_price_changes = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    down_price_changes = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    
    n = 14  # RSI period
    average_up = up_price_changes.rolling(window=n).mean()
    average_down = down_price_changes.rolling(window=n).mean().abs()
    
    relative_strength = average_up / average_down
    rsi = 100 - (100 / (1 + relative_strength))
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (average_true_range_20d * df['volume']) + (rsi * df['volume'])
    
    # Final Momentum Combination
    final_momentum_combination = final_adjustment * volume_adjusted_momentum + sma_momentum
    
    return final_momentum_combination
```
