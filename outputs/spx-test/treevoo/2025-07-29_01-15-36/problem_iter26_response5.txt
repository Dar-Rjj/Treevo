```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low for a volatility measure
    df['volatility'] = df['high'] - df['low']
    
    # Calculate the momentum based on the close price of the current and previous day
    df['momentum'] = df['close'].pct_change()
    
    # Define a new factor that combines volatility and momentum
    # This factor is intended to capture stocks with high momentum but lower volatility
    df['alpha_factor'] = (df['momentum'] / (1 + df['volatility']))
    
    # Drop the intermediate columns
    df.drop(columns=['volatility', 'momentum'], inplace=True)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
