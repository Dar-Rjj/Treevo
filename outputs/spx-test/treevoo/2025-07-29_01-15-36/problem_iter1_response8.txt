```python
import pandas as pd

def heuristics_v2(df, n=14, m=10, p=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Average True Range
    true_range = df[['high', 'low', 'close']].copy()
    true_range['prev_close'] = df['close'].shift(1)
    true_range['tr1'] = true_range['high'] - true_range['low']
    true_range['tr2'] = (true_range['high'] - true_range['prev_close']).abs()
    true_range['tr3'] = (true_range['low'] - true_range['prev_close']).abs()
    true_range['true_range'] = true_range[['tr1', 'tr2', 'tr3']].max(axis=1)
    avg_true_range = true_range['true_range'].rolling(window=n).mean()
    
    # Determine Volume Trend
    volume_sum_last_m_days = df['volume'].rolling(window=m).sum()
    volume_sum_prev_m_days = df['volume'].shift(m).rolling(window=m).sum()
    volume_trend = (volume_sum_last_m_days > volume_sum_prev_m_days).astype(int) * 2 - 1
    
    # Calculate Price Momentum
    close_avg = df['close'].rolling(window=p).mean()
    price_momentum = df['close'] - close_avg
    
    # Combine Factors
    factor = (high_low_range * avg_true_range * volume_trend) + price_momentum
    
    return factor
```
