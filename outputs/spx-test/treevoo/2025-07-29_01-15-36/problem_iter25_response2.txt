```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum = df['close'] - sma_close
    
    # Calculate Average Volume Over 20 Days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine Volume and Price Volatility
    volume_price_volatility = avg_volume * avg_high_low_range
    
    # Weight Momentum by Combined Volume & Price Volatility
    weighted_momentum = momentum * volume_price_volatility
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate 20-Day Moving Average of Price Range
    ma_price_range = (df['high'] - df['low']).rolling(window=20).sum() / 20
    
    # Calculate N-Day Relative Strength Index (RSI)
    n = 14
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    avg_gain = gain.rolling(window=n).mean()
    avg_loss = loss.rolling(window=n).mean()
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (ma_price_range * df['volume']) + (rsi * df['volume'])
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_spread
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = combined_intraday * close_to_range_ratio
    
    # Final Combination
    final_combination = (combined_intraday * volume_weighted_close_to_range_momentum) + volume_adjusted_momentum
    
    # Adjust for Volume Impact
    volume_trend = df['volume'].diff(1)
    final_factor = final_combination * volume_trend
    
    return final_factor
```
