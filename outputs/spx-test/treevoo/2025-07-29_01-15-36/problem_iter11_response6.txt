```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average price of the day
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the buying pressure, an indication of demand
    buying_pressure = df['close'] - df['open']
    
    # Calculate the relative strength, which is a comparison of the day's gains to the day's losses
    gain_loss_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Combine the indicators into a composite factor
    composite_factor = (avg_price * buying_pressure * gain_loss_ratio) / df['volume']
    
    return composite_factor
```
