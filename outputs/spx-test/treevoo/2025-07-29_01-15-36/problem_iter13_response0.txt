```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_crr = close_to_range_ratio * df['volume']
    
    # Final Combination
    final_combination = combined_intraday * volume_weighted_crr
    
    # Adjust for Volume Impact and Price Volatility
    # Volume Trend
    volume_trend = df['volume'].diff()
    
    # Average High-Low Range Over 20 Days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Final Adjustment
    alpha_factor = final_combination * volume_trend * (1 / avg_high_low_range)
    
    return alpha_factor
```
