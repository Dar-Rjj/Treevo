import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    mean_high_low_diff = df['high_low_diff'].rolling(window=20).mean()
    std_high_low_diff = df['high_low_diff'].rolling(window=20).std()
    
    # Adjust Intraday Return by Intraday Volatility
    df['intraday_volatility'] = mean_high_low_diff + std_high_low_diff
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Determine Range-to-Range Ratio
    df['range_to_range_ratio'] = df['high_low_diff'] / df['high_low_diff'].shift(1)
    
    # Integrate Volume Sensitivity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_range_to_range_ratio'] = df['range_to_range_ratio'] * df['volume_ratio']
    
    # Combine Adjusted Intraday Return and Adjusted Range-to-Range Ratio
    df['combined_intraday_factor'] = df['adjusted_intraday_return'] * df['adjusted_range_to_range_ratio']
    
    # Detect Volume Shock
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_shock'] = df['volume_change'] > 0.5 * df['volume'].shift(1)
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['price_change'] = df['close'].diff()
    df['upward_change'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['downward_change'] = -df['price_change'].apply(lambda x: x if x < 0 else 0)
    avg_upward = df['upward_change'].rolling(window=14).mean()
    avg_downward = df['downward_change'].rolling(window=14).mean()
    RS = avg_upward / avg_downward
    df['rsi'] = 100 - (100 / (1 + RS))
    
    # Calculate Exponential Moving Average (EMA) of Close prices
    df['ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum Factor
    df['momentum_factor'] = df['close'] - df['ema_close']
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Combine Volume and Price Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_high_low_range = df['high_low_diff'].rolling(window=20).mean()
    combined_volatility = avg_volume * avg_high_low_range
    
    # Combined Weighted Momentum
    df['weighted_momentum'] = (df['rsi'] * df['volume']) + \
                              (df['intraday_return'] * df['volume_ratio']) + \
                              (df['momentum_factor'] * combined_volatility) + \
                              (df['obv'] * df['adjusted_range_to_range_ratio'])
    
    # Final Factor Calculation
    df['final_factor'] = df['weighted_momentum'] * df['combined_intraday_factor']
    
    return df['final_factor']
