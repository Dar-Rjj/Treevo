```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    n = 5  # Set the number of days for EMA
    ema_price_range = daily_price_range.ewm(span=n, adjust=False).mean()
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / daily_price_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_c2r_momentum = df['volume'] * close_to_range_ratio
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    window_size = 5
    summarized_volume_weighted_c2r_momentum = volume_weighted_c2r_momentum.rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    up_change = df['close'].diff().where(lambda x: x > 0, 0)
    down_change = -df['close'].diff().where(lambda x: x < 0, 0)
    ema_up = up_change.ewm(span=n, adjust=False).mean()
    ema_down = down_change.ewm(span=n, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema = ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = volume_adjusted_ema + volume_adjusted_rsi
    
    # Final Combination
    final_combination = (combined_intraday * summarized_volume_weighted_c2r_momentum) * volume_adjusted_momentum
    
    # Adjust for Volume Impact
    volume_trend = df['volume'].diff()
    final_factor = final_combination * volume_trend
    
    return final_factor
```
