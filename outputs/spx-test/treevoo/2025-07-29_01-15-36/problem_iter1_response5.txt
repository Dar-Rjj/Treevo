```python
def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_momentum = df['volume'] * close_to_range_ratio
    
    # Summarize over a Window (e.g., 5 days)
    window_size = 5
    factor_values = volume_weighted_momentum.rolling(window=window_size).sum()
    
    return factor_values
```
