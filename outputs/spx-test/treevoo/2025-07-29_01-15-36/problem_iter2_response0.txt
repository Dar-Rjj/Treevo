```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices
    df['SMA'] = df['close'].rolling(window=20).mean()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-Weighted Moving Average of Returns
    N = 20
    df['volume_sum'] = df['volume'].rolling(window=N).sum()
    df['volume_weighted_returns'] = (df['daily_return'] * df['volume']).rolling(window=N).sum()
    df['VWMA'] = df['volume_weighted_returns'] / df['volume_sum']
    
    # Incorporate Price Volatility
    df['avg_high_low_range'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['price_volatility'] = df['avg_volume'] * df['avg_high_low_range']
    
    # Incorporate Trading Range
    df['true_range'] = df.apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df.shift(1)['close']), abs(x['low'] - df.shift(1)['close'])), axis=1)
    df['ma_true_range'] = df['true_range'].rolling(window=N).mean()
    
    # Standardize VWMA
    df['VWMA_mean'] = df['VWMA'].rolling(window=N).mean()
    df['VWMA_std'] = df['VWMA'].rolling(window=N).std()
    df['VWMA_standardized'] = (df['VWMA'] - df['VWMA_mean']) / df['VWMA_std']
    
    # Combine Indicators
    threshold = 0.5
    df['signal'] = 0
    df.loc[(df['VWMA_standardized'] > 0) & (df['price_volatility'] + df['ma_true_range'] > threshold), 'signal'] = 1
    df.loc[(df['VWMA_standardized'] < 0) & (df['price_volatility'] + df['ma_true_range'] < -threshold), 'signal'] = -1
    
    return df['signal']
```
