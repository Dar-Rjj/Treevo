```python
import pandas as pd

def heuristics_v2(df, n=14, volume_increase_threshold=1.5, amount_increase_threshold=1.5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['n_day_avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate N-Day RSI
    df['up_change'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['down_change'] = -df['close'].diff().apply(lambda x: x if x < 0 else 0)
    df['avg_gain'] = df['up_change'].rolling(window=n).mean()
    df['avg_loss'] = df['down_change'].rolling(window=n).mean()
    df['rsi'] = 100 - (100 / (1 + df['avg_gain'] / df['avg_loss']))
    
    # Intraday Momentum
    df['intraday_range'] = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    df['volume_shock'] = (df['volume_change'] > volume_increase_threshold).astype(int)
    df['amount_shock'] = (df['amount_change'] > amount_increase_threshold).astype(int)
    
    # Calculate N-Day Exponential Moving Average of Volume
    df['n_day_ema_volume'] = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Volume Trend
    df['volume_trend'] = (df['volume'] > df['n_day_ema_volume']).astype(int) * 2 - 1
    
    # Adjust Momentum Factors with Volume and Amount Shock
    df['volume_adjusted_n_day_avg_price_range'] = df['n_day_avg_price_range'] * df['volume_shock']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume_shock']
    df['volume_adjusted_intraday_range'] = df['intraday_range'] * df['volume_shock']
    df['amount_adjusted_intraday_range'] = df['intraday_range'] * df['amount_shock']
    
    # Calculate N-Day Trade Imbalance
    mid_price = (df['high'] + df['low']) / 2
    df['buy_volume'] = (df['volume'] * (df['close'] >= mid_price)).rolling(window=n).sum()
    df['sell_volume'] = (df['volume'] * (df['close'] < mid_price)).rolling(window=n).sum()
    df['trade_imbalance'] = df['buy_volume'] - df['sell_volume']
    
    # Combine All Adjusted Momentum Factors
    df['combined_factors'] = (df['volume_adjusted_n_day_avg_price_range'] 
                              + df['volume_adjusted_rsi'] 
                              + df['volume_adjusted_intraday_range'] 
                              + df['amount_adjusted_intraday_range'])
    
    # Incorporate Volume Trend and Trade Imbalance
    df['combined_factors'] *= df['volume_trend'] * df['trade_imbalance']
    
    # Calculate N-Day Moving Average of Close Prices
    df['n_day_avg_close'] = df['close'].rolling(window=n).mean()
    
    # Adjust Combined Factors with N-Day Moving Average of Close Prices
    df['final_factor'] = df['combined_factors'] / df['n_day_avg_close']
    
    return df['final_factor']
```
