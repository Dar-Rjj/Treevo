```python
import pandas as pd

def heuristics_v2(df, n=14, m=10, p=5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Average True Range
    true_range = df[['high', 'close']].shift(1).max(axis=1) - df[['low', 'close']].shift(1).min(axis=1)
    average_true_range = true_range.rolling(window=n).mean()
    
    # Determine Volume Trend
    volume_trend = (df['volume'].rolling(window=m).sum() > df['volume'].shift(m).rolling(window=m).sum()).astype(int) * 2 - 1
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].rolling(window=p).mean()
    
    # Calculate N-Day Moving Average of Price Range
    moving_avg_price_range = high_low_range.rolling(window=n).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Factors
    combined_factors = (high_low_range * average_true_range) * volume_trend + price_momentum
    volume_adjusted_price_range = moving_avg_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    
    # Final Alpha Factor
    alpha_factor = combined_factors + volume_adjusted_price_range + volume_adjusted_rsi
    
    return alpha_factor
```
