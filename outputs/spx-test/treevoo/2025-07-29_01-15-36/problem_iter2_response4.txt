```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Average True Range
    true_range = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': (df['high'] - df['close']).abs(),
        'l-pc': (df['low'] - df['close']).abs()
    }).max(axis=1)
    avg_true_range = true_range.rolling(window=14).mean()
    
    # Determine Volume Trend
    volume_sum_last_10 = df['volume'].rolling(window=10).sum()
    volume_sum_prev_10 = df['volume'].shift(10).rolling(window=10).sum()
    volume_trend = (volume_sum_last_10 > volume_sum_prev_10).astype(int) * 2 - 1
    
    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_close
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    avg_high_low_range_20 = high_low_range.rolling(window=20).mean()
    combined_volatility = avg_volume_20 * avg_high_low_range_20
    weighted_momentum = momentum_factor / combined_volatility
    
    # Calculate Price Momentum
    avg_close_10 = df['close'].rolling(window=10).mean()
    price_momentum = df['close'] - avg_close_10
    
    # Combine Factors
    factor = (high_low_range * avg_true_range) * volume_trend + price_momentum
    alpha_factor = factor * weighted_momentum
    
    return alpha_factor
```
