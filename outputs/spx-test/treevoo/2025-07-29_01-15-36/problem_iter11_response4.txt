```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = df['volume'] * close_to_range_ratio
    
    # Final Combination
    final_combination = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(1)
    average_high_low_range_20_days = high_low_range.rolling(window=20).mean()
    combined_volume_price_volatility = df['volume'] * average_high_low_range_20_days
    
    adjusted_final_combination = final_combination * volume_trend * combined_volume_price_volatility + volume_trend
    
    # Compute Average True Range
    true_range = df['high'] - df['low']
    average_true_range = true_range.rolling(window=14).mean()
    
    # Determine Volume Trend
    m = 10
    volume_trend_indicator = df['volume'].rolling(window=m).sum() - df['volume'].rolling(window=m).sum().shift(m)
    
    # Calculate Price Momentum
    p = 10
    price_momentum = df['close'] - df['close'].rolling(window=p).mean()
    
    # Calculate N-Day Moving Average of Price Range
    n = 20
    n_day_moving_average_price_range = high_low_range.rolling(window=n).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    price_change = df['close'].diff()
    up_changes = price_change.where(price_change > 0, 0)
    down_changes = -price_change.where(price_change < 0, 0)
    
    avg_up = up_changes.rolling(window=n).mean()
    avg_down = down_changes.rolling(window=n).mean()
    
    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Factors
    combined_factors = high_low_range * average_true_range
    combined_factors = combined_factors * volume_trend_indicator.apply(lambda x: 2 if x > 0 else -2)
    combined_factors = combined_factors + price_momentum
    combined_factors = combined_factors + (n_day_moving_average_price_range * df['volume'])
    combined_factors = combined_factors + (rsi * df['volume'])
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_final_combination + combined_factors
    
    return final_alpha_factor
```
