import pandas as pd

def heuristics_v2(df):
    # Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    mean_high_low = df['high_low_diff'].rolling(window=10).mean()
    std_high_low = df['high_low_diff'].rolling(window=10).std()
    df['intraday_volatility'] = (df['high_low_diff'] - mean_high_low) / std_high_low
    
    # Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Adjusted Intraday Return
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Volume-Adjusted Intraday Return
    df['volume_adjusted_intraday_return'] = df['adjusted_intraday_return'] * df['volume']
    
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Volume and Amount Shock Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    volume_shock_threshold = 500000  # Set a threshold for volume increase
    amount_shock_threshold = 10000000  # Set a threshold for amount increase
    df['volume_shock_indicator'] = (df['volume_change'] > volume_shock_threshold).astype(int)
    df['amount_shock_indicator'] = (df['amount_change'] > amount_shock_threshold).astype(int)
    
    # Combined Intraday Momentum and Shocks
    df['combined_intraday_momentum_and_shocks'] = (df['intraday_range'] * df['volume_shock_indicator']) + (df['intraday_range'] * df['amount_shock_indicator'])
    
    # Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # N-Day Moving Average of Price Range
    n = 5
    df['n_day_moving_average_price_range'] = df['daily_price_range'].rolling(window=n).mean()
    
    # Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['daily_price_range']
    
    # Volume-Weighted Close-to-Range Ratio
    df['volume_weighted_close_to_range_ratio'] = df['close_to_range_ratio'] * df['volume']
    
    # Window Sum of Volume-Weighted Close-to-Range Ratio
    window_size = 5
    df['window_sum_volume_weighted_close_to_range_ratio'] = df['volume_weighted_close_to_range_ratio'].rolling(window=window_size).sum()
    
    # N-Day Relative Strength Index (RSI)
    df['upward_price_change'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['downward_price_change'] = (df['close'] < df['close'].shift(1)) * (df['close'].shift(1) - df['close'])
    n = 14
    df['average_upward_change'] = df['upward_price_change'].rolling(window=n).mean()
    df['average_downward_change'] = df['downward_price_change'].rolling(window=n).mean().abs()
    df['rs'] = df['average_upward_change'] / df['average_downward_change']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Volume-Adjusted RSI
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume']
    
    # Volume-Adjusted N-Day Moving Average of Price Range
    df['volume_adjusted_n_day_moving_average_price_range'] = df['n_day_moving_average_price_range'] * df['volume']
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['volume_adjusted_n_day_moving_average_price_range'] + df['volume_adjusted_rsi']) / 2
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['volume_adjusted_intraday_return'] + 
                                df['combined_intraday_momentum_and_shocks'] + 
                                df['volume_adjusted_momentum']) / 3
    
    return df['final_alpha
