```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term SMA
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the momentum indicator (difference between short-term and long-term SMA)
    df['momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Calculate the percentage change in volume from the previous day
    df['volume_change'] = df['volume'].pct_change()
    
    # Combine the momentum indicator with the volume change to form a composite alpha factor
    df['composite_factor'] = df['momentum'] * df['volume_change']
    
    # Calculate the Relative Strength Index (RSI) over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Adjust the composite alpha factor by adding the RSI value
    df['composite_factor'] += df['RSI']
    
    # Calculate the Average True Range (ATR) over a 14-day period
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': abs(df['high'] - df['close'].shift(1)),
        'l-pc': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    df['ATR'] = tr.rolling(window=14).mean()
    
    # Adjust the composite alpha factor by dividing it by the ATR
    df['composite_factor'] /= df['ATR']
    
    # Calculate the percentage change in the open-to-close price
    df['open_close_change'] = (df['close'] - df['open']) / df['open']
    
    # Adjust the composite alpha factor by adding the open-to-close price change
    df['composite_factor'] += df['open_close_change']
    
    # Calculate the daily price range as (high - low)
    df['price_range'] = df['high'] - df['low']
    
    # Normalize the daily price range by the average true range (ATR)
    df['normalized_price_range'] = df['price_range'] / df['ATR']
    
    # Adjust the composite alpha factor by subtracting the normalized price range
    df['composite_factor'] -= df['normalized_price_range']
    
    # Calculate the percentage change in the amount traded from the previous day
    df['amount_change'] = df['amount'].pct_change()
    
    # Adjust the composite alpha factor by adding the percentage change in the amount
    df['composite_factor'] += df['amount_change']
    
    # Return the composite alpha factor as a pandas Series
    return df['composite_factor']
```
