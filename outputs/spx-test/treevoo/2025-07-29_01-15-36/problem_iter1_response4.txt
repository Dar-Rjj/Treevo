def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Previous Day's High-Low Range
    prev_high_low_range = high_low_range.shift(1)
    
    # Determine Range-to-Range Ratio
    range_to_range_ratio = high_low_range / prev_high_low_range
    
    # Integrate Volume Sensitivity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_range_to_range_ratio = range_to_range_ratio * volume_ratio
    
    # Final Factor Calculation
    close_open_return = df['close'] - df['open']
    factor_values = adjusted_range_to_range_ratio + close_open_return
    
    return factor_values
