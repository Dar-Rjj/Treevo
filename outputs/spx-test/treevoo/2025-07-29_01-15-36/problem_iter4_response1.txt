```python
import pandas as pd

def heuristics_v2(df, n=14, m=10, p=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Compute Average True Range
    true_range = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    average_true_range = true_range.rolling(window=n).sum() / n

    # Determine Volume Trend
    volume_trend = (df['volume'].rolling(window=m).sum().diff() > 0).astype(int)
    volume_trend = volume_trend.replace({1: 1, 0: -1})

    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].rolling(window=p).mean()

    # Combine Factors
    factor = (high_low_range * average_true_range) * volume_trend + price_momentum

    return factor
```
