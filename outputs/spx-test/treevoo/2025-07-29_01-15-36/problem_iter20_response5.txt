import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_momentum = combined_intraday_indicators * close_to_range_ratio * df['volume']
    
    # Intermediate Alpha Factor
    intermediate_alpha_factor = volume_weighted_momentum
    
    # Adjust for Volume Impact and Enhanced Price Volatility
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average True Range Over 20 Days
    tr = np.maximum(np.maximum(df['high'] - df['low'], np.abs(df['high'] - df['close'].shift(1))), 
                    np.abs(df['low'] - df['close'].shift(1)))
    atr_20 = tr.rolling(window=20).mean()
    
    # Calculate Average Volume Over 20 Days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range_20 = high_low_range.rolling(window=20).mean()
    
    # Combined Volume & Enhanced Price Volatility
    combined_volatility = avg_volume_20 * avg_high_low_range_20
    
    # Calculate Simple Moving Average (SMA) of Close prices over 20 days
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_20
    
    # Final Adjustment
    final_adjustment = intermediate_alpha_factor * volume_trend / combined_volatility
    alpha_factor = final_adjustment * momentum_factor
    
    return alpha_factor
