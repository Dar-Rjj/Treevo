```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a 5-Day Window
    summarized_volume_weighted_momentum = volume_weighted_close_to_range_momentum.rolling(window=5).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    price_diff = df['close'].diff(1)
    up_changes = np.where(price_diff > 0, price_diff, 0)
    down_changes = np.where(price_diff < 0, -price_diff, 0)
    ema_up = up_changes.ewm(span=14, adjust=False).mean()
    ema_down = down_changes.ewm(span=14, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    ema_price_range_volume = (intraday_spread * df['volume']).ewm(span=14, adjust=False).mean()
    ema_rsi_volume = (rsi * df['volume']).ewm(span=14, adjust=False).mean()
    volume_adjusted_momentum = ema_price_range_volume + ema_rsi_volume
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = (df['close'] - sma_20) * (df['volume'].rolling(window=20).mean() * (df['high'] - df['low']).rolling(window=20).mean())
    
    # Final Combination
    final_combination = (
        combined_intraday_indicators * 
        volume_weighted_close_to_range_momentum * 
        summarized_volume_weighted_momentum * 
        volume_adjusted_momentum + 
        momentum_factor
    )
    
    # Adjust for Volume Impact and Price Volatility
    volume_trend = df['volume'].diff(1)
    average_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    volume_acceleration = volume_trend.diff(1)
    
    final_adjustment = (
        final_combination * 
        volume_trend * 
        (1 / average_high_low_range) * 
        volume_acceleration
    )

    return final_adjustment
```
