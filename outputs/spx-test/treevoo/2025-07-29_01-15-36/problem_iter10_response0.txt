```python
import pandas as pd

def heuristics_v2(df, n=14, m=5, p=20, q=20, r=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Average True Range
    true_range = df[['high', 'low']].apply(lambda x: max(x['high'], df['close'].shift(1)) - min(x['low'], df['close'].shift(1)), axis=1)
    avg_true_range = true_range.rolling(window=n).mean()
    
    # Determine Volume Trend
    volume_trend = (df['volume'].rolling(window=m).sum() - df['volume'].shift(m).rolling(window=m).sum()).apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].rolling(window=p).mean()
    
    # Enhance Volatility Measure
    close_std = df['close'].rolling(window=q).std() * 1.5
    
    # Combine Factors
    combined_factors = (high_low_range * avg_true_range * volume_trend) + price_momentum
    
    # Final Adjustment
    final_adjustment = (combined_factors * close_std) + (df['close'] - df['close'].shift(r)).apply(lambda x: 1 if x > 0 else -1)
    
    return final_adjustment
```
