import pandas as pd
import numpy as np

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate N-Day RSI
    df['up_change'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['down_change'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    avg_gain = df['up_change'].rolling(window=n).mean()
    avg_loss = df['down_change'].rolling(window=n).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'].diff()
    df['amount_change'] = df['amount'].diff()
    volume_shock = (df['volume_change'] > 0.5 * df['volume'].shift(1)).astype(int)
    amount_shock = (df['amount_change'] > 0.5 * df['amount'].shift(1)).astype(int)
    
    # Calculate N-Day Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = (df['volume'] > df['ema_volume']).astype(int) * 2 - 1
    
    # Adjust Momentum Factors with Volume and Amount Shock
    df['adjusted_avg_price_range'] = df['avg_price_range'] * volume_shock
    df['adjusted_rsi'] = df['rsi'] * volume_shock
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] * (volume_shock + amount_shock)
    
    # Calculate N-Day Trade Imbalance
    midpoint = (df['high'] + df['low']) / 2
    df['buy_volume'] = df.apply(lambda row: row['volume'] if row['close'] >= midpoint[row.name] else 0, axis=1)
    df['sell_volume'] = df.apply(lambda row: row['volume'] if row['close'] < midpoint[row.name] else 0, axis=1)
    df['trade_imbalance'] = df['buy_volume'].rolling(window=n).sum() - df['sell_volume'].rolling(window=n).sum()
    
    # Combine All Adjusted Momentum Factors
    df['combined_momentum'] = df['adjusted_avg_price_range'] + df['adjusted_rsi'] + df['adjusted_intraday_momentum']
    df['combined_momentum'] = df['combined_momentum'] * df['volume_trend'] * df['trade_imbalance']
    
    # Calculate N-Day Moving Average of Close Prices
    df['ma_close'] = df['close'].rolling(window=n).mean()
    
    # Integrate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['ma_high_low_diff'] = df['high_low_diff'].rolling(window=n).mean()
    df['combined_momentum'] = df['combined_momentum'] * df['ma_high_low_diff']
    
    # Adjust Combined Factors with N-Day Moving Average of Close Prices
    df['final_factor'] = df['combined_momentum'] / df['ma_close']
    
    return df['final_factor']
