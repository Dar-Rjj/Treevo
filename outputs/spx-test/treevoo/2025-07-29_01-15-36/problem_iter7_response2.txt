```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    ema_price_range = df['price_range'].ewm(span=n, adjust=False).mean()
    
    # Determine Upward and Downward Price Changes
    df['up_change'] = np.where(df['close'] > df['close'].shift(), df['close'] - df['close'].shift(), 0)
    df['down_change'] = np.where(df['close'] < df['close'].shift(), df['close'].shift() - df['close'], 0)
    
    # Calculate Exponential Moving Averages of Upward and Downward Changes
    ema_up_change = df['up_change'].ewm(span=n, adjust=False).mean()
    ema_down_change = df['down_change'].ewm(span=n, adjust=False).mean()
    
    # Compute RS (Relative Strength)
    rs = ema_up_change / ema_down_change.abs()
    
    # Calculate RSI
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_momentum = df['volume'] * (ema_price_range + rsi)
    
    return volume_adjusted_momentum
```
