```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    df['momentum_factor'] = df['SMA_20'] - df['close']
    
    # Calculate Combined Volume & Price Volatility
    avg_volume_220 = df['volume'].rolling(window=220).mean()
    avg_high_low_20 = (df['high'] - df['low']).rolling(window=20).mean()
    combined_vol_price_vol = avg_volume_220 * avg_high_low_20
    
    # Weight Momentum Factor by Combined Volume & Price Volatility
    df['weighted_momentum_factor'] = df['momentum_factor'] / combined_vol_price_vol
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    mean_high_low_diff = high_low_diff.rolling(window=20).mean()
    std_high_low_diff = high_low_diff.rolling(window=20).std()
    df['intraday_volatility'] = (df['intraday_high_low_spread'] - mean_high_low_diff) / std_high_low_diff
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Determine Range-to-Range Ratio
    df['previous_high_low_range'] = df['high'].shift(1) - df['low'].shift(1)
    df['current_high_low_range'] = df['high'] - df['low']
    df['range_to_range_ratio'] = df['current_high_low_range'] / df['previous_high_low_range']
    
    # Integrate Volume Sensitivity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_range_to_range_ratio'] = df['range_to_range_ratio'] * df['volume_ratio']
    
    # Detect Volume Shock
    volume_change = df['volume'] - df['volume'].shift(1)
    df['volume_shock'] = (volume_change > 1.5 * df['volume'].shift(1)).astype(int)
    
    # Combine Intraday Indicators
    df['combined_intraday_indicators'] = (df['intraday_high_low_spread'] * df['intraday_return']) + df['intraday_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['high_low_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_close_to_range_momentum'] = df['combined_intraday_indicators'] * df['close_to_range_ratio']
    
    # Final Combination
    df['alpha_factor'] = (df['combined_intraday_indicators'] * df['volume_weighted_close_to_range_momentum']) * df['weighted_momentum_factor']
    
    # Adjust for Volume Impact and Price Volatility
    volume_trend = df['volume'] - df['volume'].shift(1)
    avg_high_low_range_20 = (df['high'] - df['low']).rolling(window=20).mean()
    df['final_alpha_factor'] = df['alpha_factor'] * volume_trend / avg_high_low_range_20
    
    return df['final_alpha_factor']
```
