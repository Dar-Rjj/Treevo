```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    ema_price_range = df['price_range'].ewm(span=n, adjust=False).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['price_change'] = df['close'].diff()
    df['up_change'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['down_change'] = df['price_change'].apply(lambda x: -x if x < 0 else 0)
    
    ema_up = df['up_change'].ewm(span=n, adjust=False).mean()
    ema_down = df['down_change'].ewm(span=n, adjust=False).mean()
    
    rs = ema_up / ema_down.abs()
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema_price_range = ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    
    # Combine Results
    factor_values = volume_adjusted_ema_price_range + volume_adjusted_rsi
    
    return factor_values
```
