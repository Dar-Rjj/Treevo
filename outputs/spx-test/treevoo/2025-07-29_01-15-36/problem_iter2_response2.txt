```python
def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Determine Range-to-Range Ratio
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    df['range_to_range_ratio'] = df['high_low_range'] / df['prev_high_low_range']
    
    # Integrate Volume Sensitivity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_range_to_range_ratio'] = df['range_to_range_ratio'] * df['volume_ratio']
    
    # Calculate Intraday Momentum
    df['intraday_range'] = df['high'] - df['low']
    
    # Detect Volume Shock
    volume_shock_threshold = 2.0  # Example threshold, can be adjusted
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    df['volume_shock'] = (df['volume_change'] > volume_shock_threshold).astype(int)
    
    # Combine Adjusted Range-to-Range Ratio and Intraday Range
    df['combined_momentum'] = df['adjusted_range_to_range_ratio'] * df['intraday_range']
    
    # Final Factor Calculation
    df['final_factor'] = df['combined_momentum'] * df['volume_shock']
    
    return df['final_factor']
```
