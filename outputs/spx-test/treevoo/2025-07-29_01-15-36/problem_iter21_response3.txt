```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    window_size = 5
    summarized_volume_weighted_momentum = volume_weighted_close_to_range_momentum.rolling(window=window_size).sum()
    
    # Calculate N-Day Exponential Moving Average of Price Range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=window_size, adjust=False).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    upward_change = df['close'].diff().clip(lower=0)
    downward_change = -df['close'].diff().clip(upper=0)
    ema_upward_change = upward_change.ewm(span=window_size, adjust=False).mean()
    ema_downward_change = downward_change.ewm(span=window_size, adjust=False).mean()
    rs = ema_upward_change / ema_downward_change
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema = ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = volume_adjusted_ema + volume_adjusted_rsi
    
    # Final Combination
    final_combination = (combined_intraday_indicators * volume_weighted_close_to_range_momentum *
                         summarized_volume_weighted_momentum * volume_adjusted_momentum)
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'].diff()
    average_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    volume_acceleration = df['volume'].diff().diff()
    
    # Final Adjustment
    alpha_factor = (final_combination * volume_trend * (1 / average_high_low_range) * volume_acceleration)
    
    return alpha_factor
```
