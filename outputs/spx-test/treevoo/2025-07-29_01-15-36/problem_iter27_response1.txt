```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    intraday_indicator = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_ctr = close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (e.g., 5 days)
    window_size = 5
    summarized_volume_weighted_ctr = volume_weighted_ctr.rolling(window=window_size).sum()
    
    # Calculate N-Day Exponential Moving Average of Price Range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=window_size, adjust=False).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=window_size, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=window_size, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema = ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = volume_adjusted_ema + volume_adjusted_rsi
    
    # Final Combination
    final_combination = (intraday_indicator * volume_weighted_ctr * summarized_volume_weighted_ctr * volume_adjusted_momentum)
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'].diff()
    avg_high_low_range_20_days = (df['high'] - df['low']).rolling(window=20).mean()
    volume_acceleration = (df['volume'].diff().diff())
    price_volatility = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    alpha_factor = (final_combination * volume_trend * (1 / avg_high_low_range_20_days) * volume_acceleration + price_volatility) * volume_weighted_ctr
    
    return alpha_factor
```
