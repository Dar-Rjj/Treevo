```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Daily Price Range
    daily_range = df['high'] - df['low']
    
    # N-Day Avg Price Range
    n_day_avg_range = daily_range.rolling(window=n).mean()
    
    # N-Day RSI
    delta = df['close'].diff(1)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    avg_up = up.rolling(window=n).mean()
    avg_down = down.rolling(window=n).mean().abs()
    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Adjusted Intraday Return
    adjusted_intraday_return = intraday_return / daily_range
    
    # Volume and Amount Shock
    volume_change = df['volume'].diff(1)
    amount_change = df['amount'].diff(1)
    
    # N-Day Avg Volume
    n_day_avg_volume = df['volume'].rolling(window=n).mean()
    
    # Volume Trend
    volume_trend = (df['volume'] > n_day_avg_volume).astype(int) - (df['volume'] < n_day_avg_volume).astype(int)
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = df['volume'] * (df['close'] - df['open']) / daily_range
    
    # Momentum
    mom_5day = intraday_return.rolling(window=5).mean()
    mom_20day = intraday_return.rolling(window=20).mean()
    momentum = mom_5day - mom_20day
    
    # Combined Adjusted Factors
    combined_factors = (n_day_avg_range * volume_change) + \
                       (rsi * volume_change) + \
                       (daily_range * volume_change) + \
                       (daily_range * amount_change) + \
                       (volume_weighted_intraday_return) * \
                       volume_trend
    
    return combined_factors
```
