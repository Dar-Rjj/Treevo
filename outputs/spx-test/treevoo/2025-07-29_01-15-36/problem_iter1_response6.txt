```python
def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Mean and Standard Deviation of recent High-Low Differences
    mean_high_low_diff = high_low_diff.rolling(window=20).mean()
    std_high_low_diff = high_low_diff.rolling(window=20).std()
    
    # Calculate Intraday Volatility
    intraday_volatility = mean_high_low_diff + std_high_low_diff
    
    # Adjust Intraday Return by Intraday Volatility
    adjusted_return = intraday_return / intraday_volatility
    
    # Weight by Volume
    weighted_adjusted_return = adjusted_return * df['volume']
    
    return weighted_adjusted_return
```
