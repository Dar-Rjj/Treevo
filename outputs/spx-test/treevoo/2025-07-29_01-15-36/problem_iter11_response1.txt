```python
def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    - data (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    - pd.Series: A Series with index as date and values as the generated alpha factor.
    """
    # Calculate the range of the day
    daily_range = data['high'] - data['low']
    
    # Calculate the buying pressure
    buying_pressure = (data['close'] - data['low']) / daily_range
    
    # Calculate the selling pressure
    selling_pressure = (data['high'] - data['close']) / daily_range
    
    # Calculate the volume-weighted buying pressure
    volume_weighted_buying_pressure = buying_pressure * data['volume']
    
    # Calculate the volume-weighted selling pressure
    volume_weighted_selling_pressure = selling_pressure * data['volume']
    
    # Calculate the net volume-weighted pressure
    net_volume_weighted_pressure = volume_weighted_buying_pressure - volume_weighted_selling_pressure
    
    # Calculate the factor
    factor = net_volume_weighted_pressure.rolling(window=10).mean()
    
    return factor
```
