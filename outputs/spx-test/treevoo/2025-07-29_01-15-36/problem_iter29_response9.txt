import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the daily price movement relative to the open
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate the buying pressure as a function of volume and price movement
    df['buying_pressure'] = (df['volume'] * df['price_movement']) / df['daily_range']
    
    # Heuristic: The alpha factor is defined as the ratio of buying pressure to the daily return
    # This heuristic assumes that when the buying pressure is high, the stock might be overbought and vice versa.
    df['alpha_factor'] = df['buying_pressure'] / df['daily_return']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
