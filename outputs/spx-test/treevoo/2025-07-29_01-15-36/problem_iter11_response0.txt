```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['n_day_price_range_ma'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['delta'] = df['close'].diff()
    df['up'] = df['delta'].clip(lower=0)
    df['down'] = -df['delta'].clip(upper=0)
    df['avg_gain'] = df['up'].rolling(window=n).mean()
    df['avg_loss'] = df['down'].rolling(window=n).mean()
    df['rs'] = df['avg_gain'] / df['avg_loss'].replace(0, 1e-6)
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['n_day_price_range_ma'] * df['volume']) + (df['rsi'] * df['volume'])
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['volatility_mean'] = df['high_low_diff'].rolling(window=n).mean()
    df['volatility_std'] = df['high_low_diff'].rolling(window=n).std()
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_intraday_return'] = df['intraday_return'] / df['volatility_std'].replace(0, 1e-6)
    
    # Weight by Volume
    df['weighted_intraday_return'] = df['adjusted_intraday_return'] * df['volume']
    
    # Incorporate Momentum
    df['5_day_return_ma'] = df['intraday_return'].rolling(window=5).mean()
    df['20_day_return_ma'] = df['intraday_return'].rolling(window=20).mean()
    df['momentum_factor'] = df['5_day_return_ma'] - df['20_day_return_ma']
    
    # Integrate Price Trend
    df['5_day_close_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_close_ma'] = df['close'].rolling(window=20).mean()
    df['price_trend_factor'] = df['5_day_close_ma'] - df['20_day_close_ma']
    
    # Combine All Factors
    df['alpha_factor'] = df['volume_adjusted_momentum'] * df['weighted_intraday_return'] * df['momentum_factor'] * df['price_trend_factor']
    
    return df['alpha_factor']
```
