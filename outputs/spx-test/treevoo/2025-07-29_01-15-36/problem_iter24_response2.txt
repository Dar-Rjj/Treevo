```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    close_change = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Calculate a modified volume factor
    volume_factor = df['volume'] * (df['close'] - df['open'])
    
    # Combine the factors to create the alpha factor
    alpha_factor = (price_diff * close_change) + volume_factor
    
    return alpha_factor
```
