```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread and Open-to-Close Momentum
    combined_indicator = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Adjust for Volume Impact
    # Calculate Volume Trend
    volume_trend = df['volume'].pct_change()
    
    # Final Adjustment
    alpha_factor = combined_indicator * volume_trend
    
    return alpha_factor
```
