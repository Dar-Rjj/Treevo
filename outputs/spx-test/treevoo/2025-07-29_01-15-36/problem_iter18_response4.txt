```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_ctr = close_to_range_ratio * df['volume']
    
    # Final Combination
    final_combination = combined_intraday * volume_weighted_ctr
    
    # Adjust for Volume Impact, Price Volatility, and Volume Shock
    # Volume Trend
    df['volume_trend'] = df['volume'] - df['volume'].shift(1)
    
    # Average True Range Over 20 Days
    true_range = df[['high', 'low']].diff(axis=1).max(axis=1).abs() + \
                 (df['high'] - df['close']).abs().shift(1) + \
                 (df['low'] - df['close']).abs().shift(1)
    average_true_range = true_range.rolling(window=20).mean()
    
    # Volume Shock
    volume_shock = df['volume'] - df['volume'].shift(5)
    
    # Final Adjustment
    alpha_factor = (final_combination * df['volume_trend']) * (1 / average_true_range) + volume_shock
    
    return alpha_factor
```
