```python
import pandas as pd

def heuristics_v2(df, n=14, m=10, p=20, q=20, r=5, volatility_factor=1.5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Average True Range
    true_range = df[['high', 'low']].apply(lambda x: max(x[0], x[1]), axis=1) - df['close'].shift(1)
    true_range = true_range.abs()
    atr = true_range.rolling(window=n).mean()
    
    # Determine Volume Trend
    volume_trend = (df['volume'].rolling(window=m).sum() > df['volume'].shift(m).rolling(window=m).sum()).astype(int)
    volume_trend = volume_trend * 2 - 1  # Convert to 1 for increasing, -1 for decreasing
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].rolling(window=p).mean()
    
    # Enhance Volatility Measure
    std_dev_close = df['close'].rolling(window=q).std() * volatility_factor
    
    # Combine Factors
    combined_factors = (high_low_range * atr) * volume_trend + price_momentum
    
    # Final Adjustment
    final_adjustment = std_dev_close * combined_factors
    momentum_adjustment = (df['close'].diff().rolling(window=r).mean() > 0).astype(int) * 2 - 1
    final_factors = final_adjustment * final_adjustment
    
    return final_factors
```
