```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Create a momentum indicator from the difference between short-term and long-term SMA
    df['momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Calculate the percentage change in volume from the previous day
    df['volume_change'] = df['volume'].pct_change()
    
    # Combine the momentum indicator with the volume change to form a composite alpha factor
    df['composite_alpha'] = df['momentum'] * df['volume_change']
    
    # Add the relative strength index (RSI) calculated from the close prices over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Integrate the RSI into the composite alpha factor
    df['composite_alpha_with_rsi'] = df['composite_alpha'] + df['RSI']
    
    # Calculate the average true range (ATR) to measure volatility over a 14-day period
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    df['ATR'] = true_range.rolling(window=14).mean()
    
    # Incorporate the ATR into the composite alpha factor
    df['final_alpha_factor'] = df['composite_alpha_with_rsi'] / df['ATR']
    
    # Return the final alpha factor as a pandas Series
    return df['final_alpha_factor'].dropna()
```
