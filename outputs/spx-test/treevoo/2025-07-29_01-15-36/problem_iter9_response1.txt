import pandas as pd

def heuristics_v2(df, n=14, m=5, p=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Average True Range
    true_range = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    atr = true_range.rolling(window=n).mean()
    
    # Determine Volume Trend
    volume_trend = (df['volume'].rolling(window=m).sum() > df['volume'].shift(m).rolling(window=m).sum()).astype(int) * 4 - 2
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].rolling(window=p).mean()
    
    # Calculate N-Day Moving Average of Price Range
    moving_avg_price_range = high_low_range.rolling(window=n).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Factors
    combined_factors = (high_low_range * atr) * volume_trend + price_momentum
    volume_adjusted_factors = (moving_avg_price_range * df['volume']) + (rsi * df['volume'])
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors + volume_adjusted_factors
    
    return final_alpha_factor
