import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']

    # Combine Intraday Indicators
    combined_intraday_indicators = (high_low_spread * open_close_momentum) + open_close_momentum

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range

    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']

    # Final Combination for Intermediate Alpha Factor
    intermediate_alpha_factor = combined_intraday_indicators * volume_weighted_close_to_range_momentum

    # Adjust for Volume Impact and Enhanced Price Volatility

    # Calculate Volume Trend
    volume_trend = df['volume'].diff()

    # Calculate Average True Range Over 20 Days
    def average_true_range(high, low, close, period=20):
        tr = pd.concat([high - low, abs(high - close.shift(1)), abs(low - close.shift(1))], axis=1).max(axis=1)
        return tr.rolling(window=period).mean()
    atr_20 = average_true_range(df['high'], df['low'], df['close'])

    # Calculate Average Volume Over 20 Days
    avg_volume_20 = df['volume'].rolling(window=20).mean()

    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range_20 = (df['high'] - df['low']).rolling(window=20).mean()

    # Weight by Combined Volume & Enhanced Price Volatility
    combined_volatility_weight = avg_volume_20 * avg_high_low_range_20

    # Final Adjustment
    final_adjustment = intermediate_alpha_factor * volume_trend / combined_volatility_weight

    # Calculate Simple Moving Average (SMA) of Close prices over 20 days
    sma_20 = df['close'].rolling(window=20).mean()

    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_20

    # Combine Adjusted Momentum with SMA Momentum
    alpha_factor = final_adjustment * momentum_factor

    return alpha_factor
