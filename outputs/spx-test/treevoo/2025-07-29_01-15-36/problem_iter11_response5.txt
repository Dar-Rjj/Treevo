```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Price Range
    moving_avg_price_range = daily_price_range.rolling(window=20).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    price_change = df['close'].diff()
    up_changes = price_change.where(price_change > 0, 0)
    down_changes = -price_change.where(price_change < 0, 0)
    
    n = 14  # RSI period
    avg_gain = up_changes.rolling(window=n).mean()
    avg_loss = down_changes.rolling(window=n).mean()
    
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (moving_avg_price_range * df['volume']) + (rsi * df['volume'])
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = df['volume'] * close_to_range_ratio
    
    # Final Combination
    final_combination = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Volume Trend
    volume_trend = df['volume'].diff()
    average_high_low_range_20_days = high_low_range.rolling(window=20).mean()
    
    combined_volume_price_volatility = df['volume'] * average_high_low_range_20_days
    
    final_adjustment = (final_combination * volume_trend) / combined_volume_price_volatility + volume_trend
    
    return final_adjustment
```
