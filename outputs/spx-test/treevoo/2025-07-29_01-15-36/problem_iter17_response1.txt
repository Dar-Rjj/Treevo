import pandas as pd

def heuristics_v2(df, n=14):
    # Daily Price Range
    df['daily_range'] = df['high'] - df['low']
    
    # N-Day Avg Price Range
    df['n_day_avg_range'] = df['daily_range'].rolling(window=n).mean()
    
    # N-Day RSI
    df['up_change'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['down_change'] = -df['close'].diff().apply(lambda x: x if x < 0 else 0)
    df['avg_up'] = df['up_change'].rolling(window=n).mean()
    df['avg_down'] = df['down_change'].rolling(window=n).mean()
    df['rs'] = df['avg_up'] / df['avg_down'].replace(0, 1e-5)
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjusted Intraday Return
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility'].replace(0, 1e-5)
    
    # Volume and Amount Shock
    df['volume_change'] = df['volume'].diff()
    df['amount_change'] = df['amount'].diff()
    
    # N-Day Avg Volume
    df['n_day_avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Volume Trend
    df['volume_trend'] = (df['volume'] > df['n_day_avg_volume']).astype(int) * 2 - 1
    
    # Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['volume'] * df['adjusted_intraday_return']
    
    # Momentum
    df['5_day_ma_intraday_return'] = df['intraday_return'].rolling(window=5).mean()
    df['20_day_ma_intraday_return'] = df['intraday_return'].rolling(window=20).mean()
    df['momentum'] = df['5_day_ma_intraday_return'] - df['20_day_ma_intraday_return']
    
    # Adjusted Factors
    df['n_day_avg_range_volume_shock'] = df['n_day_avg_range'] * df['volume_change']
    df['n_day_rsi_volume_shock'] = df['rsi'] * df['volume_change']
    df['intraday_range_volume_shock'] = df['intraday_volatility'] * df['volume_change']
    df['intraday_range_amount_shock'] = df['intraday_volatility'] * df['amount_change']
    
    # Combined Adjusted Factors
    df['combined_adjusted_factors'] = (df['n_day_avg_range_volume_shock'] + 
                                       df['n_day_rsi_volume_shock'] + 
                                       df['intraday_range_volume_shock'] + 
                                       df['intraday_range_amount_shock'])
    df['combined_adjusted_factors'] += df['volume_weighted_intraday_return']
    df['combined_adjusted_factors'] *= df['volume_trend']
    
    # Calculate the difference between today's close and the previous day's close (Momentum)
    df['momentum_close_diff'] = df['close'].diff()
    
    # Calculate the rate of change over N days (RateOfChange)
    df['rate_of_change'] = df['close'].pct_change(n)
    
    # Determine the average gain or loss over N days (AverageGainLoss)
    df['average_gain_loss'] = df['close'].diff().rolling(window=n).mean()
    
    # Typical Price
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3

    df['cumulative_vwap'] = (df['typical_price'] * df['volume']).cumsum()
