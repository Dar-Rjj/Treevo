```python
import pandas as pd

def heuristics_v2(df, N=14):
    # Momentum
    momentum = df['close'] - df['close'].shift(1)
    
    # Rate of Change
    rate_of_change = (df['close'] - df['close'].shift(N)) / df['close'].shift(N)
    
    # Average Gain or Loss
    gain_loss = df['close'] - df['close'].shift(1)
    average_gain_loss = gain_loss.rolling(window=N).mean()
    
    # Volume Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    cumulative_vwap = (typical_price * df['volume']).cumsum()
    cumulative_volume = df['volume'].cumsum()
    vwap = cumulative_vwap / cumulative_volume
    
    # True Range and Average True Range
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=N).mean()
    
    # Simple Moving Average
    sma = df['close'].rolling(window=N).mean()
    
    # Exponential Moving Average
    ema = df['close'].ewm(span=N, adjust=False).mean()
    
    # Composite Alpha Signal
    alpha_signal = (
        (momentum + rate_of_change + atr) + 
        (df['close'] > vwap).astype(int) + 
        (df['close'] > sma).astype(int)
    )
    
    return alpha_signal
```
