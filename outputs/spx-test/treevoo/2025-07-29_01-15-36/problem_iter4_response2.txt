```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['price_range_ma'] = df['price_range'].rolling(window=n).mean()
    
    # Determine Upward and Downward Price Changes
    df['upward_change'] = (df['close'].diff().clip(lower=0))
    df['downward_change'] = (-df['close'].diff().clip(upper=0))
    
    # Calculate Average Upward and Downward Changes
    df['avg_up'] = df['upward_change'].rolling(window=n).mean()
    df['avg_down'] = df['downward_change'].rolling(window=n).mean()
    
    # Compute RS (Relative Strength)
    df['rs'] = df['avg_up'] / df['avg_down'].replace(0, 1)  # Avoid division by zero
    
    # Calculate RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Calculate Volume Adjusted Momentum
    df['volume_adjusted_price_range'] = df['price_range_ma'] * df['volume']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume']
    
    # Combine Results
    df['alpha_factor'] = df['volume_adjusted_price_range'] + df['volume_adjusted_rsi']
    
    return df['alpha_factor']
```
