import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    mf_multiplier = (typical_price - df['close'].shift(1)) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mf_multiplier * df['volume']
    
    # Calculate the positive and negative money flow
    positive_mf = mf_volume.where(mf_multiplier > 0, 0).rolling(window=14).sum()
    negative_mf = mf_volume.where(mf_multiplier < 0, 0).rolling(window=14).sum()
    
    # Calculate the money ratio
    money_ratio = positive_mf / negative_mf.abs()
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate the alpha factor
    alpha_factor = mfi - mfi.shift(1)
    
    return alpha_factor
