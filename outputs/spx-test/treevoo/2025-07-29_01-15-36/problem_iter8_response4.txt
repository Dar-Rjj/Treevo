```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['up_change'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['down_change'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    df['avg_up_change'] = df['up_change'].rolling(window=n).mean()
    df['avg_down_change'] = df['down_change'].rolling(window=n).mean()
    df['rs'] = df['avg_up_change'] / df['avg_down_change']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=n).std()
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    volume_threshold = 2.0
    amount_threshold = 2.0
    df['volume_shock'] = (df['volume_change'] > volume_threshold).astype(int)
    df['amount_shock'] = (df['amount_change'] > amount_threshold).astype(int)
    
    # Calculate N-Day Moving Average of Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = (df['volume'] > df['avg_volume']).replace({True: 1, False: -1})
    
    # Weight by Volume
    df['weighted_intraday_return'] = df['adjusted_intraday_return'] * df['volume']
    
    # Calculate Momentum
    df['5_day_ma'] = df['intraday_return'].rolling(window=5).mean()
    df['20_day_ma'] = df['intraday_return'].rolling(window=20).mean()
    df['momentum'] = df['5_day_ma'] - df['20_day_ma']
    
    # Adjust Momentum Factors with Volume and Amount Shock
    df['volume_adjusted_avg_price_range'] = df['avg_price_range'] * df['volume_shock']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume_shock']
    df['volume_adjusted_intraday_momentum'] = df['intraday_momentum'] * df['volume_shock']
    df['amount_adjusted_intraday_momentum'] = df['intraday_momentum'] * df['amount_shock']
    
    # Combine All Adjusted Momentum Factors
    df['combined_factor'] = (df['volume_adjusted_avg_price_range'] 
                             + df['volume_adjusted_rsi'] 
                             + df['volume_adjusted_intraday_momentum'] 
                             + df['amount_adjusted_intraday_momentum'] 
                             + df['weighted_intraday_return'])
    
    # Incorporate Volume Trend
    df['final_factor'] = df['combined_factor'] * df['volume_trend']
    
    return df['final_factor']
```
