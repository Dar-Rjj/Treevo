```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    df['MA_5'] = df['close'].rolling(window=5).mean()
    df['MA_20'] = df['close'].rolling(window=20).mean()
    
    # Generate buy/sell signals when short-term MA crosses long-term MA
    df['MA_Crossover'] = (df['MA_5'] > df['MA_20']).astype(int)
    
    # Calculate the percentage change in close prices from t-1 to t
    df['Price_Momentum'] = df['close'].pct_change()
    
    # Calculate the average gain and loss over a period for RSI
    df['Gain'] = df['close'].diff().clip(lower=0)
    df['Loss'] = -df['close'].diff().clip(upper=0)
    df['Avg_Gain'] = df['Gain'].rolling(window=14).mean()
    df['Avg_Loss'] = df['Loss'].rolling(window=14).mean()
    df['RSI'] = 100 - (100 / (1 + (df['Avg_Gain'] / df['Avg_Loss'])))
    
    # Calculate the VWAP for the current day
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    df['Close_VWAP_Diff'] = df['close'] - df['VWAP']
    
    # Calculate the A/D Line
    df['Money_Flow_Multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['Money_Flow_Volume'] = df['Money_Flow_Multiplier'] * df['volume']
    df['A/D_Line'] = df['Money_Flow_Volume'].cumsum()
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['close'].diff() > 0).astype(int) - (df['close'].diff() < 0).astype(int)
    df['OBV'] = (df['OBV'] * df['volume']).cumsum()
    
    # Calculate Bollinger Bands
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['STD_20'] = df['close'].rolling(window=20).std()
    df['BB_Upper'] = df['SMA_20'] + 2 * df['STD_20']
    df['BB_Lower'] = df['SMA_20'] - 2 * df['STD_20']
    df['BB_Position'] = (df['close'] - df['SMA_20']) / (2 * df['STD_20'])
    
    # Calculate the daily range (high - low)
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate the difference between the open and close price
    df['Open_Close_Spread'] = df['open'] - df['close']
    
    # Calculate the ratio of high to low
    df['High_Low_Ratio'] = df['high'] / df['low']
    
    # Calculate the change in volume from t-1 to t
    df['Volume_Change'] = df['volume'].diff()
    
    # Calculate the ratio of close price to volume
    df['Close_Volume_Ratio'] = df['close'] / df['volume']
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['MA_Crossover'] + 
        df['Price_Momentum'] + 
        df['RSI'] + 
        df['Close_VWAP_Diff'] + 
        df['A/D_Line'] + 
        df['OBV'] + 
        df['BB_Position'] + 
        df['Daily_Range'] + 
        df['Open_Close_Spread'] + 
        df['High_Low_Ratio'] + 
        df['Volume_Change'] + 
        df['Close_Volume_Ratio']
    )
    
    return df['Alpha_Factor']
```
