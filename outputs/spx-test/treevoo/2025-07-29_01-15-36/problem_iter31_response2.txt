```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Final Combination for Intermediate Alpha Factor
    intermediate_alpha_factor = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact, Price Volatility, RSI, and Accumulation Distribution
    # Calculate Volume Trend
    volume_trend = df['volume'].diff()
    
    # Calculate Average True Range Over 20 Days
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    average_true_range = true_range.rolling(window=20).mean()
    
    # Calculate Accumulation Distribution Line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    accumulation_distribution_line = (money_flow_multiplier * df['volume']).rolling(window=20).sum()
    
    # Calculate Price Volatility
    price_volatility = df['close'].rolling(window=20).std()
    
    # Calculate N-Day RSI (N=14)
    n = 14
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Final Adjustment for Alpha Factor
    alpha_factor = intermediate_alpha_factor * volume_trend * accumulation_distribution_line * price_volatility * rsi
    
    # Additional Momentum and Volume-Weighted Enhancements
    # Calculate Volume-Weighted Open-to-Close Momentum
    volume_weighted_open_close_momentum = (open_close_momentum * df['volume']).rolling(window=20).sum()
    
    # Calculate Volume-Weighted Intraday High-Low Spread
    volume_weighted_high_low_spread = (high_low_spread * df['volume']).rolling(window=20).sum()
    
    # Combine Enhanced Indicators
    enhanced_intermediate_alpha_factor = volume_weighted_open_close_momentum + volume_weighted_high_low_spread
    
    # Final Synthesis
    final_alpha_factor = (enhanced_intermediate_alpha_factor * alpha_factor) * volume_trend * price_volatility * accumulation_distribution_line * rsi
    
    return final_alpha_factor
```
