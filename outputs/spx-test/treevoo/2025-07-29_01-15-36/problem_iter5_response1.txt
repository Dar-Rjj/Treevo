```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_range = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    volume_change = df['volume'].pct_change()
    amount_change = df['amount'].pct_change()
    volume_shock = (volume_change > 0.5) * 1  # Threshold for Volume Increase
    amount_shock = (amount_change > 0.5) * 1  # Threshold for Amount Increase
    
    # Combine Intraday Momentum and Volume/Amount Shock
    combined_momentum = (intraday_range * volume_shock) + (intraday_range * amount_shock)
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    n = 10
    moving_avg_price_range = daily_price_range.rolling(window=n).mean()
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / daily_price_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range = df['volume'] * close_to_range_ratio
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    window_size = 5
    summarized_volume_weighted_momentum = volume_weighted_close_to_range.rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    n_rsi = 14
    df['price_diff'] = df['close'].diff()
    up_changes = df['price_diff'].apply(lambda x: x if x > 0 else 0)
    down_changes = df['price_diff'].apply(lambda x: -x if x < 0 else 0)
    avg_up = up_changes.rolling(window=n_rsi).mean()
    avg_down = down_changes.rolling(window=n_rsi).mean()
    relative_strength = avg_up / avg_down
    rsi = 100 - (100 / (1 + relative_strength))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_moving_avg = moving_avg_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = volume_adjusted_moving_avg + volume_adjusted_rsi
    
    # Combine Volume-Adjusted Intraday Momentum and Breakout
    final_factor = combined_momentum + summarized_volume_weighted_momentum + volume_adjusted_momentum
    
    return final_factor
```
