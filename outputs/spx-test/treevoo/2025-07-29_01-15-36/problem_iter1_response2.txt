```python
import pandas as pd

def heuristics_v2(df, N=20, true_range_threshold=0.5):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Volume-Weighted Moving Average of Returns
    df['volume_sum'] = df['volume'].rolling(window=N).sum()
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    df['vwr_sum'] = df['volume_weighted_return'].rolling(window=N).sum()
    df['VWMA'] = df['vwr_sum'] / df['volume_sum']
    
    # Calculate True Range
    df['true_range'] = df.apply(lambda x: max(x['high'] - x['low'], 
                                              abs(x['high'] - df['close'].shift(1).loc[x.name]), 
                                              abs(x['low'] - df['close'].shift(1).loc[x.name])), axis=1)
    
    # Calculate N-day Moving Average of True Range
    df['true_range_ma'] = df['true_range'].rolling(window=N).mean()
    
    # Standardize VWMA
    mean_VWMA = df['VWMA'].rolling(window=N).mean()
    std_VWMA = df['VWMA'].rolling(window=N).std()
    df['standardized_VWMA'] = (df['VWMA'] - mean_VWMA) / std_VWMA
    
    # Combine Indicators
    def signal(row):
        if row['standardized_VWMA'] > 0 and row['true_range_ma'] > true_range_threshold:
            return 1
        elif row['standardized_VWMA'] < 0 and row['true_range_ma'] < true_range_threshold:
            return -1
        else:
            return 0

    df['signal'] = df.apply(signal, axis=1)
    
    return df['signal']
```
