```python
import pandas as pd

def heuristics_v2(df, n=14, volume_threshold=1.5, amount_threshold=1.5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['price_change'] = df['close'].diff()
    df['gain'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['price_change'].apply(lambda x: -x if x < 0 else 0)
    avg_gain = df['gain'].rolling(window=n).mean()
    avg_loss = df['loss'].rolling(window=n).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Intraday Momentum
    df['intraday_range'] = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    df['volume_shock'] = (df['volume_change'] > volume_threshold).astype(int)
    df['amount_shock'] = (df['amount_change'] > amount_threshold).astype(int)
    
    # Calculate N-Day Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = (df['volume'] > df['ema_volume']).astype(int) * 2 - 1
    
    # Adjust Momentum Factors with Volume and Amount Shock
    df['adjusted_avg_price_range'] = df['avg_price_range'] * df['volume_shock']
    df['adjusted_rsi'] = df['rsi'] * df['volume_shock']
    df['adjusted_intraday_range'] = df['intraday_range'] * df['volume_shock']
    df['adjusted_intraday_range_by_amount'] = df['intraday_range'] * df['amount_shock']
    
    # Combine All Adjusted Momentum Factors
    df['combined_momentum'] = (df['adjusted_avg_price_range'] + df['adjusted_rsi'] + 
                               df['adjusted_intraday_range'] + df['adjusted_intraday_range_by_amount'])
    
    # Incorporate Volume Trend
    df['final_factor'] = df['combined_momentum'] * df['volume_trend']
    
    return df['final_factor']
```
