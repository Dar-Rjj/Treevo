```python
import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday High-Low Spread
    high_low_spread = data['high'] - data['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = data['close'] - data['open']
    
    # Combine Intraday Indicators
    intraday_indicator = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = data['high'] - data['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = data['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * intraday_indicator * data['volume']
    
    # Calculate Open-to-High Momentum
    open_high_momentum = data['high'] - data['open']
    
    # Calculate Low-to-Close Momentum
    low_close_momentum = data['close'] - data['low']
    
    # Combine Open-to-High and Low-to-Close Momentum
    combined_momentum = open_high_momentum + low_close_momentum
    
    # Final Combination
    intermediate_alpha_factor = volume_weighted_close_to_range_momentum * combined_momentum
    
    # Calculate Volume Trend
    volume_trend = data['volume'] - data['volume'].shift(1)
    
    # Calculate Average True Range Over 20 Days
    true_range = np.maximum.reduce([data['high'] - data['low'],
                                    (data['high'] - data['close']).abs(),
                                    (data['low'] - data['close']).abs()])
    average_true_range = true_range.rolling(window=20).mean()
    
    # Calculate Accumulation Distribution Line
    money_flow_multiplier = ((data['close'] - data['low']) - (data['high'] - data['close'])) / (data['high'] - data['low'])
    money_flow_volume = money_flow_multiplier * data['volume']
    accumulation_distribution_line = money_flow_volume.rolling(window=20).sum()
    
    # Calculate Price Volatility
    price_volatility = data['close'].rolling(window=22).std()
    
    # Calculate Volume-Weighted Price Change
    price_change = data['close'] - data['close'].shift(1)
    volume_weighted_price_change = (price_change * data['volume']).rolling(window=20).sum()
    
    # Final Adjustment
    alpha_factor = (intermediate_alpha_factor * volume_trend * average_true_range * 
                    accumulation_distribution_line * price_volatility * volume_weighted_price_change)
    
    return alpha_factor
```
