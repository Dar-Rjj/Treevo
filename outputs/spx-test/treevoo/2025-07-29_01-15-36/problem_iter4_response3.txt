```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Indicators
    combined_indicator = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Adjust for Volume Impact
    volume_trend = df['volume'].pct_change().fillna(0)  # Use percentage change to get trend, fillna(0) to handle first day
    final_adjustment = combined_indicator * volume_trend
    
    return final_adjustment
```
