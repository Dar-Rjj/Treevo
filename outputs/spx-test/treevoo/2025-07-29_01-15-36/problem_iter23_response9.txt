```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_c2r = close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    window_size = 5
    summarized_volume_weighted_c2r = volume_weighted_c2r.rolling(window=window_size).sum()
    
    # Calculate N-Day Exponential Moving Average of Price Range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=window_size, adjust=False).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=window_size-1, adjust=False).mean()
    ema_down = down.ewm(com=window_size-1, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema = ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = volume_adjusted_ema + volume_adjusted_rsi
    
    # Final Combination
    final_combination = (
        combined_intraday * 
        volume_weighted_c2r * 
        summarized_volume_weighted_c2r * 
        volume_adjusted_momentum
    )
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'].diff(1)
    avg_high_low_range_20d = (df['high'] - df['low']).rolling(window=20).mean()
    volume_acceleration = (df['volume'].diff(1)).diff(1)
    
    alpha_factor = (
        final_combination * 
        volume_trend * 
        (1 / avg_high_low_range_20d) * 
        volume_acceleration
    )
    
    return alpha_factor
```
