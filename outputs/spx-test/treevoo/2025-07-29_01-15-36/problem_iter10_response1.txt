```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    n_days_ema = 14
    ema_price_range = df['price_range'].ewm(span=n_days_ema, adjust=False).mean()
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    window_size = 5
    summarized_momentum = df['volume_weighted_momentum'].rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    n_days_rsi = 14
    df['price_change'] = df['close'].diff()
    df['upward_change'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['downward_change'] = -df['price_change'].apply(lambda x: x if x < 0 else 0)
    
    ema_up = df['upward_change'].ewm(span=n_days_rsi, adjust=False).mean()
    ema_down = df['downward_change'].ewm(span=n_days_rsi, adjust=False).mean()
    
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema = ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = volume_adjusted_ema + volume_adjusted_rsi
    
    # Combine Volume-Adjusted Breakout and Momentum
    factor_values = summarized_momentum + volume_adjusted_momentum
    
    return factor_values
```
