import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = sma_close - df['close']
    
    # Calculate Average Volume Over 20 Days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine Volume and Price Volatility
    combined_volatility = avg_volume * avg_high_low_range
    
    # Weight the Momentum Factor by Combined Volatility
    weighted_momentum = momentum_factor * combined_volatility
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Price Range
    ma_price_range = daily_price_range.rolling(window=20).mean()
    
    # Calculate N-Day Exponential Moving Average of Price Range
    ema_price_range = daily_price_range.ewm(span=20, adjust=False).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (ma_price_range * df['volume']) + (rsi * df['volume'])
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    summarized_volume_weighted_close_to_range_momentum = volume_weighted_close_to_range_momentum.rolling(window=5).sum()
    
    # Final Combination
    final_combination = (combined_intraday_indicators * volume_weighted_close_to_range_momentum * 
                         summarized_volume_weighted_close_to_range_momentum * volume_adjusted_momentum) + volume_adjusted_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'] - df['volume'].shift(1)
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))
    
    # Final Adjustment
    alpha_factor = (final_combination * volume_trend * (1 / avg_high_low_range) * volume_acceleration)
    
    return alpha_factor
