```python
import pandas as pd

def heuristics_v2(df, N=14, MA5=5, MA20=20):
    # Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # N-Day Avg Price Range
    df['N_Day_Avg_Price_Range'] = df['Daily_Price_Range'].rolling(window=N).mean()
    
    # N-Day RSI
    df['Upward_Change'] = df.apply(lambda x: x['close'] - df.shift(1)['close'] if x['close'] > df.shift(1)['close'] else 0, axis=1)
    df['Downward_Change'] = df.apply(lambda x: df.shift(1)['close'] - x['close'] if x['close'] < df.shift(1)['close'] else 0, axis=1)
    df['Avg_Up'] = df['Upward_Change'].rolling(window=N).mean()
    df['Avg_Down'] = df['Downward_Change'].rolling(window=N).mean()
    df['RS'] = df['Avg_Up'] / df['Avg_Down'].abs()
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Intraday Return and Volatility
    df['Intraday_Return'] = df['close'] - df['open']
    df['Intraday_Volatility'] = df['high'] - df['low']
    df['Adjusted_Intraday_Return'] = df['Intraday_Return'] / df['Intraday_Volatility']
    
    # Volume and Amount Shock
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)
    
    # N-Day Avg Volume
    df['N_Day_Avg_Volume'] = df['volume'].rolling(window=N).mean()
    
    # Volume Trend
    df['Volume_Trend'] = df.apply(lambda x: 1 if x['volume'] > x['N_Day_Avg_Volume'] else -1 if x['volume'] < x['N_Day_Avg_Volume'] else 0, axis=1)
    
    # Volume-Weighted Intraday Return
    df['Volume_Weighted_Intraday_Return'] = df['volume'] * df['Adjusted_Intraday_Return']
    
    # Momentum
    df['MA5_Intraday_Return'] = df['Intraday_Return'].rolling(window=MA5).mean()
    df['MA20_Intraday_Return'] = df['Intraday_Return'].rolling(window=MA20).mean()
    df['Momentum'] = df['MA5_Intraday_Return'] - df['MA20_Intraday_Return']
    
    # Combined Adjusted Factors
    df['Combined_Adjusted_Factors'] = (
        (df['N_Day_Avg_Price_Range'] * df['Volume_Change']) +
        (df['RSI'] * df['Volume_Change']) +
        (df['Daily_Price_Range'] * df['Volume_Change']) +
        (df['Daily_Price_Range'] * df['Amount_Change'])
    )
    df['Combined_Adjusted_Factors'] += df['Volume_Weighted_Intraday_Return']
    df['Combined_Adjusted_Factors'] *= df['Volume_Trend']
    
    return df['Combined_Adjusted_Factors']
```
