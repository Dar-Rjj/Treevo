import pandas as pd
import numpy as np

def average_true_range(high, low, close, period=20):
    tr = pd.Series(np.zeros(len(close)))
    for i in range(1, len(close)):
        hl = high[i] - low[i]
        hpc = abs(high[i] - close[i-1])
        lpc = abs(low[i] - close[i-1])
        tr[i] = max(hl, hpc, lpc)
    
    atr = tr.rolling(window=period).mean()
    return atr

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_momentum = close_to_range_ratio * (df['close'] - df['open'])
    
    # Final Combination
    intermediate_alpha_factor = combined_intraday * volume_weighted_momentum
    
    # Adjust for Volume Impact and Enhanced Price Volatility
    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average True Range Over 20 Days
    atr_20 = average_true_range(df['high'], df['low'], df['close'])
    
    # Calculate Volume-Price Trend
    vpt = (df['volume'] * (df['close'] - df['close'].shift(1))) / df['close'].shift(1)
    
    # Final Adjustment
    alpha_factor = intermediate_alpha_factor * vpt
    alpha_factor = alpha_factor * (volume_trend + atr_20)
    
    return alpha_factor
