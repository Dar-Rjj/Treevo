import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices over 20 days
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    df['momentum'] = df['SMA_20'] - df['close']
    
    # Calculate Average Volume Over 20 Days
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    df['avg_high_low_range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine Volume and Price Volatility
    df['vol_price_volatility'] = df['avg_volume_20'] * df['avg_high_low_range_20']
    
    # Weight the Momentum Factor by Combined Volume & Price Volatility
    df['weighted_momentum'] = df['momentum'] * df['vol_price_volatility']
    
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Combine Intraday Indicators
    df['combined_intraday_indicators'] = (df['high_low_spread'] * df['open_close_momentum']) + df['open_close_momentum']
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Price Range
    df['ma_price_range_20'] = df['daily_price_range'].rolling(window=20).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    n = 14
    df['price_change'] = df['close'].diff()
    df['up_change'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['down_change'] = -df['price_change'].apply(lambda x: x if x < 0 else 0)
    
    df['avg_up'] = df['up_change'].rolling(window=n).mean()
    df['avg_down'] = df['down_change'].rolling(window=n).mean()
    
    df['rs'] = df['avg_up'] / df['avg_down'].replace(0, np.nan)
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Volume-Adjusted Momentum
    df['vol_adj_momentum'] = (df['ma_price_range_20'] * df['volume']) + (df['rsi'] * df['volume'])
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['high_low_spread']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['vol_weighted_close_to_range_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Final Combination
    df['final_combination'] = (df['combined_intraday_indicators'] * df['vol_weighted_close_to_range_momentum']) + df['vol_adj_momentum']
    
    # Adjust for Volume Impact, Price Volatility, and Accumulation Distribution
    df['volume_trend'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average True Range Over 20 Days
    df['true_range'] = df[['high', 'close']].shift(1).max(axis=1) - df[['low', 'close']].shift(1).min(axis=1)
    df['avg_true_range_20'] = df['true_range'].rolling(window=20).mean()
    
    # Calculate Accumulation Distribution Line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['accum_dist_line'] = money_flow_volume.rolling(window=20).sum()
    
    # Calculate Price Volatility
    df['price_volatility'] = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    df['alpha_factor'] = (df['final_combination'] * df['volume_trend'] * (df['vol_price_volatility'] + df['avg_true_range_20']) * df['accum_dist_line'] * df['price_volatility'])
    
    return
