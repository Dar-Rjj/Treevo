```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the true range
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    df['true_range'] = df[['true_range', (df['high'] - df['close'].shift(1)).abs(), (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    
    # Calculate the average true range over the last 10 days
    df['avg_true_range_10'] = df['true_range'].rolling(window=10).mean()
    
    # Calculate the percentage change in volume
    df['volume_change_pct'] = df['volume'].pct_change()
    
    # Calculate the high-low difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the close-open difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Heuristic: High volatility and low volume increase -> Potential for mean reversion
    df['alpha_factor'] = (df['avg_true_range_10'] / df['high_low_diff']) * df['volume_change_pct']
    
    return df['alpha_factor']
```
