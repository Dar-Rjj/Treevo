```python
import pandas as pd

def heuristics(df):
    # Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Mean of Recent High-Low Differences (last 20 days for example)
    mean_high_low = high_low_diff.rolling(window=20).mean()
    
    # Standard Deviation of Recent High-Low Differences
    std_high_low = high_low_diff.rolling(window=20).std()
    
    # Intraday Volatility
    intraday_volatility = (high_low_diff - mean_high_low) / std_high_low
    
    # Adjust Intraday Return by Intraday Volatility
    adj_intraday_return = intraday_return / intraday_volatility
    
    # Weighted by Volume
    weighted_factor = adj_intraday_return * df['volume']

    return weighted_factor
```
