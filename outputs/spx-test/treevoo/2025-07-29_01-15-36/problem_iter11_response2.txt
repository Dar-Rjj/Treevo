```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate Upward and Downward Price Changes
    df['up_change'] = (df['close'].diff() > 0) * df['close'].diff()
    df['down_change'] = (df['close'].diff() < 0) * (-df['close'].diff())
    
    # Calculate Smoothed Average Upward and Downward Changes
    df['avg_up'] = df['up_change'].rolling(window=n).mean()
    df['avg_down'] = df['down_change'].rolling(window=n).mean()
    
    # Compute RS (Relative Strength)
    df['rs'] = df['avg_up'] / df['avg_down'].abs()
    
    # Calculate RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Calculate Intraday Momentum
    df['intraday_range'] = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    volume_shock_threshold = 1.5  # Example threshold
    amount_shock_threshold = 1.5  # Example threshold
    df['volume_shock'] = (df['volume_change'] > volume_shock_threshold).astype(int)
    df['amount_shock'] = (df['amount_change'] > amount_shock_threshold).astype(int)
    
    # Calculate N-Day Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Calculate N-Day Moving Average of Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = (df['volume'] > df['ema_volume']).astype(int) * 2 - 1
    
    # Adjust Momentum Factors with Volume and Amount Shock
    df['volume_adjusted_avg_price_range'] = df['avg_price_range'] * df['volume_shock']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume_shock']
    df['volume_adjusted_intraday_range'] = df['intraday_range'] * df['volume_shock']
    df['amount_adjusted_intraday_range'] = df['intraday_range'] * df['amount_shock']
    
    # Calculate N-Day Trade Imbalance
    midpoint = (df['high'] + df['low']) / 2
    buy_volume = (df['close'] >= midpoint) * df['volume']
    sell_volume = (df['close'] < midpoint) * df['volume']
    df['buy_vol_n_day'] = buy_volume.rolling(window=n).sum()
    df['sell_vol_n_day'] = sell_volume.rolling(window=n).sum()
    df['trade_imbalance'] = df['buy_vol_n_day'] - df['sell_vol_n_day']
    
    # Combine All Adjusted Momentum Factors
    df['combined_factor'] = (
        df['volume_adjusted_avg_price_range'] +
        df['volume_adjusted_rsi'] +
        df['volume_adjusted_intraday_range'] +
        df['amount_adjusted_intraday_range']
    ) * df['volume_trend'] * df['trade_imbalance']
    
    return df['combined_factor']
```
