```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_momentum = df['volume'] * close_to_range_ratio
    
    # Final Combination
    final_combination = combined_intraday * volume_weighted_momentum
    
    # Adjust for Volume Impact
    volume_trend = df['volume'].diff()
    
    # Final Adjustment
    final_adjustment = final_combination * volume_trend
    
    return final_adjustment
```
