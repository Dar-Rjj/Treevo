```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    mean_high_low_diff = df['high_low_diff'].rolling(window=5).mean()
    std_high_low_diff = df['high_low_diff'].rolling(window=5).std()
    df['intraday_volatility'] = (df['high_low_diff'] - mean_high_low_diff) / std_high_low_diff
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_return'] = df['adjusted_return'] * df['volume']
    
    # Incorporate Momentum
    df['5d_ma_return'] = df['intraday_return'].rolling(window=5).mean()
    df['20d_ma_return'] = df['intraday_return'].rolling(window=20).mean()
    df['momentum_factor'] = df['5d_ma_return'] - df['20d_ma_return']
    
    # Integrate Price Trend
    df['5d_ma_close'] = df['close'].rolling(window=5).mean()
    df['20d_ma_close'] = df['close'].rolling(window=20).mean()
    df['price_trend_factor'] = df['5d_ma_close'] - df['20d_ma_close']
    
    # Combine Momentum and Price Trend
    df['combined_factor'] = df['momentum_factor'] * df['price_trend_factor']
    
    return df['combined_factor'].dropna()
```
