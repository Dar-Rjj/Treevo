```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple returns
    df['simple_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Logarithmic returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Simple moving average (SMA) over 20 periods
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Exponential moving average (EMA) over 20 periods
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Rate of change (ROC) over 10 periods
    df['roc_10'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Daily range
    df['daily_range'] = df['high'] - df['low']
    
    # True range
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    
    # Average true range (ATR) over 14 periods
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # On-balance volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Chaikin money flow (CMF) over 20 periods
    df['cmf_20'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) * df['volume'] / (df['high'] - df['low'])
    df['cmf_20'] = df['cmf_20'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Money flow index (MFI) over 14 periods
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum()
    df['mfi_14'] = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Factor: Combining multiple indicators
    df['alpha_factor'] = (df['sma_20'] + df['ema_20'] + df['roc_10'] + df['atr_14'] + df['obv'] + df['cmf_20'] + df['mfi_14']) / 7
    
    return df['alpha_factor']
```
