```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    intraday_indicators = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_crr = close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (e.g., 5 days)
    window_size = 5
    summarized_volume_weighted_crr = volume_weighted_crr.rolling(window=window_size).sum()
    
    # Calculate N-Day Exponential Moving Average of Price Range
    n_day_ema_price_range = df['high'] - df['low']
    n_day_ema_price_range = n_day_ema_price_range.ewm(span=window_size, adjust=False).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    price_change = df['close'].diff()
    up = np.maximum(price_change, 0)
    down = -np.minimum(price_change, 0)
    ema_up = up.ewm(span=window_size, adjust=False).mean()
    ema_down = down.ewm(span=window_size, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema = n_day_ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = volume_adjusted_ema + volume_adjusted_rsi
    
    # Calculate Simple Moving Average (SMA) of Close Prices (last 20 days)
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = sma_close - df['close']
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    combined_volatility = avg_volume * avg_high_low_range
    weighted_momentum = momentum_factor * combined_volatility
    
    # Final Combination
    final_combination = (
        (intraday_indicators * volume_weighted_crr) *
        summarized_volume_weighted_crr *
        volume_adjusted_momentum +
        weighted_momentum
    )
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'].diff()
    avg_high_low_range_20_days = (df['high'] - df['low']).rolling(window=20).mean()
    volume_acceleration = (df['volume'].diff() - df['volume'].diff().shift(1))
    final_adjustment = (
        final_combination * volume_trend *
        (1 / avg_high_low_range_20_days) *
        volume_acceleration
    )
    
    return final_adjustment
```
