```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Final Combination
    final_combination = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Accumulation Distribution
    # Calculate Volume Trend
    volume_trend = df['volume'].diff()
    
    # Calculate Average True Range Over 20 Days
    true_range = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    average_true_range = true_range.rolling(window=20).mean()
    
    # Calculate Accumulation Distribution Line
    accumulation_distribution_line = (((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])) * df['volume']
    accumulation_distribution_line = accumulation_distribution_line.rolling(window=20).sum()
    
    # Calculate Price Volatility
    price_volatility = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    adjusted_final_combination = (final_combination * volume_trend) * (average_true_range + price_volatility) * accumulation_distribution_line
    
    # Additional Momentum and Volume-Weighted Enhancements
    # Calculate Volume-Weighted Open-to-Close Momentum
    volume_weighted_open_to_close_momentum = (open_to_close_momentum * df['volume']).rolling(window=20).sum()
    
    # Calculate Volume-Weighted Intraday High-Low Spread
    volume_weighted_intraday_high_low_spread = (intraday_high_low_spread * df['volume']).rolling(window=20).sum()
    
    # Combine Enhanced Indicators
    enhanced_intermediate_alpha_factor = volume_weighted_open_to_close_momentum + volume_weighted_intraday_high_low_spread
    
    # Incorporate Long-Term Momentum
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    long_term_momentum = ma_50 - ma_200
    enhanced_intermediate_alpha_factor *= long_term_momentum
    
    # Final Synthesis
    final_alpha_factor = (enhanced_intermediate_alpha_factor * final_combination) * (volume_trend + price_volatility) * accumulation_distribution_line
    
    return final_alpha_factor
```
