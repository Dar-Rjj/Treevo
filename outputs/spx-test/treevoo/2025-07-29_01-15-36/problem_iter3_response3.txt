import pandas as pd

def heuristics_v2(df, N=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=N).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['up_close'] = df['close'].diff(1)
    df['down_close'] = -df['close'].diff(1)
    df['up_close'] = df['up_close'].clip(lower=0)
    df['down_close'] = df['down_close'].clip(lower=0)
    df['avg_up'] = df['up_close'].rolling(window=N).mean()
    df['avg_down'] = df['down_close'].rolling(window=N).mean()
    df['RS'] = df['avg_up'] / df['avg_down'].replace(0, 1e-8)  # Avoid division by zero
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Calculate N-Day Moving Average of Volume
    df['avg_volume'] = df['volume'].rolling(window=N).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = (df['volume'] > df['avg_volume']).astype(int) - (df['volume'] < df['avg_volume']).astype(int)
    
    # Calculate Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['avg_price_range'] * df['volume'] + 
                                      df['RSI'] * df['volume']) * df['volume_trend']
    
    return df['volume_adjusted_momentum']
