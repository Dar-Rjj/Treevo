```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices over the last 20 days
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor: Subtract Latest Close Price from SMA
    momentum = sma - df['close']
    
    # Calculate Average Volume Over 20 Days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine Volume and Price Volatility
    volume_price_volatility = avg_volume * avg_high_low_range
    
    # Weight the Momentum by Combined Volume & Price Volatility
    weighted_momentum = momentum * volume_price_volatility
    
    # Calculate Intraday Indicators
    intraday_high_low_spread = df['high'] - df['low']
    open_to_close_momentum = df['close'] - df['open']
    combined_intraday_indicators = intraday_high_low_spread * open_to_close_momentum
    
    # Calculate Close-to-Range Ratio
    high_low_range = df['high'] - df['low']
    close_to_range_ratio = df['close'] / high_low_range
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = combined_intraday_indicators * close_to_range_ratio
    
    # Volume-Price Trend
    volume_price_trend = df['volume'] * (df['close'] - df['close'].shift(1))
    
    # Integrate Momentum and Intraday Indicators
    integrated_momentum_intraday = weighted_momentum + combined_intraday_indicators
    
    # Final Alpha Factor
    alpha_factor = integrated_momentum_intraday * volume_weighted_close_to_range_momentum * volume_price_trend
    
    return alpha_factor
```
