```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices over the last 20 days
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Subtract Latest Close Price from SMA to get the momentum factor
    momentum = sma_20 - df['close']
    
    # Calculate Average Volume Over 20 Days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    high_low_range = df['high'] - df['low']
    avg_high_low_range_20 = high_low_range.rolling(window=20).mean()
    
    # Calculate Average Trading Range (High - Low) / Close for each day, then average over the last 20 days
    trading_range = (df['high'] - df['low']) / df['close']
    avg_trading_range_20 = trading_range.rolling(window=20).mean()
    
    # Combine Volume, Price Volatility, and Trading Range
    combined_weight = (avg_volume_20 * avg_high_low_range_20) / avg_trading_range_20
    
    # Weight the momentum factor by the combined weight
    weighted_momentum = momentum * combined_weight
    
    return weighted_momentum
```
