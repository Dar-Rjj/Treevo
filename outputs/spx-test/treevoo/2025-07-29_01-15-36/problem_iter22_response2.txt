```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['HighLowSpread'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['OpenCloseMomentum'] = df['close'] - df['open']
    
    # Combine Intraday Indicators
    df['IntradayCombined'] = (df['HighLowSpread'] * df['OpenCloseMomentum']) + df['OpenCloseMomentum']
    
    # Calculate Close-to-Range Ratio
    df['CloseToRangeRatio'] = df['close'] / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    df['VolumeWeightedCTR'] = df['CloseToRangeRatio'] * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    window_size = 5
    df['SumVolumeWeightedCTR'] = df['VolumeWeightedCTR'].rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    n = 14  # Typical RSI period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=n).mean()
    avg_loss = loss.rolling(window=n).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    df['PriceRangeVol'] = (df['high'] - df['low']) * df['volume']
    df['EMAPriceRangeVol'] = df['PriceRangeVol'].ewm(span=n, adjust=False).mean()
    df['EMARSI'] = df['RSI'] * df['volume']
    df['VolumeAdjustedMomentum'] = df['EMAPriceRangeVol'] + df['EMARSI']
    
    # Final Combination
    df['FinalCombination'] = (df['IntradayCombined'] * df['SumVolumeWeightedCTR']) * df['VolumeAdjustedMomentum']
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    df['VolumeTrend'] = df['volume'] - df['volume'].shift(1)
    df['AvgHighLowRange'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['VolumeAcceleration'] = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))
    
    # Final Adjustment
    df['AlphaFactor'] = (df['FinalCombination'] * df['VolumeTrend']) * (1 / df['AvgHighLowRange']) * df['VolumeAcceleration']
    
    return df['AlphaFactor'].dropna()
```
