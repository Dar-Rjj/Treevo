```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    recent_high_low_mean = df['high_low_diff'].rolling(window=10).mean()
    recent_high_low_std = df['high_low_diff'].rolling(window=10).std()
    df['intraday_volatility'] = (df['high_low_diff'] - recent_high_low_mean) / recent_high_low_std
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Weight by Volume
    df['weighted_adjusted_return'] = df['adjusted_return'] * df['volume']
    
    return df['weighted_adjusted_return']
```
