```python
import pandas as pd

def heuristics_v2(df):
    # Momentum Indicators
    df['price_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    
    # Trend Indicators
    df['sma_short'] = df['close'].rolling(window=10).mean()
    df['sma_long'] = df['close'].rolling(window=30).mean()
    df['sma_crossover'] = df['sma_short'] - df['sma_long']
    df['high_low_range'] = df['high'] - df['low']
    
    # Volatility Indicators
    df['true_range'] = df[['high' - 'low', 
                           (df['high'] - df['close'].shift(1)).abs(), 
                           (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    df['price_volatility'] = df['close'].rolling(window=20).std()
    
    # Relative Strength Indicators
    price_changes = df['close'].diff()
    positive_changes = price_changes.where(price_changes > 0, 0)
    absolute_changes = price_changes.abs()
    df['rsi_inspired'] = positive_changes.rolling(window=14).mean() / absolute_changes.rolling(window=14).mean()
    
    # Money Flow Indicators
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(df['close'] > df['close'].shift(1), 0)
    negative_money_flow = money_flow.where(df['close'] < df['close'].shift(1), 0)
    df['mfi_inspired'] = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    
    # Volume-Weighted Price
    df['vwap_inspired'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Market Sentiment
    df['open_close_spread'] = df['close'] - df['open']
    df['gap_analysis'] = df['open'] - df['close'].shift(1)
    
    # Weighted Momentum with Volume and Price Volatility
    sma_close_20 = df['close'].rolling(window=20).mean()
    momentum_factor = df['close'] - sma_close_20
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    avg_high_low_range_20 = (df['high'] - df['low']).rolling(window=20).mean()
    df['weighted_momentum'] = momentum_factor * (avg_volume_20 * avg_high_low_range_20)
    
    # Final alpha factor
    alpha_factor = (
        df['price_momentum'] +
        df['volume_momentum'] +
        df['sma_crossover'] +
        df['high_low_range'] +
        df['true_range'] +
        df['price_volatility'] +
        df['rsi_inspired'] +
        df['mfi_inspired'] +
        df['vwap_inspired'] +
        df['open_close_spread'] +
        df['gap_analysis'] +
        df['weighted_momentum']
    )
    
    return alpha_factor
```
