```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    mean_high_low = high_low_diff.rolling(window=20).mean()
    std_high_low = high_low_diff.rolling(window=20).std()

    # Adjust Intraday Return by Intraday Volatility
    adjusted_intraday_return = intraday_return / (mean_high_low + std_high_low)

    # Determine Range-to-Range Ratio
    range_to_range_ratio = high_low_diff / high_low_diff.shift(1)

    # Integrate Volume Sensitivity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_range_to_range_ratio = range_to_range_ratio * volume_ratio

    # Combine Adjusted Intraday Return and Adjusted Range-to-Range Ratio
    combined_intraday_factors = adjusted_intraday_return * adjusted_range_to_range_ratio

    # Detect Volume Shock
    volume_change = df['volume'] - df['volume'].shift(1)
    sudden_volume_increase = (volume_change > 0.5 * df['volume'].shift(1)).astype(int)

    # Calculate N-Day Relative Strength Index (RSI)
    up_close = (df['close'] - df['close'].shift(1)).clip(lower=0)
    down_close = (df['close'].shift(1) - df['close']).clip(lower=0)
    avg_up = up_close.rolling(window=14).mean()
    avg_down = down_close.rolling(window=14).mean()
    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))

    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_20 = df['close'].rolling(window=20).mean()

    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_20
    avg_volume_220 = df['volume'].rolling(window=220).mean()
    avg_high_low_range_20 = (df['high'] - df['low']).rolling(window=20).mean()
    combined_volatility = avg_volume_220 * avg_high_low_range_20
    weighted_momentum = momentum_factor * combined_volatility

    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (rsi * df['volume']) + (intraday_return * volume_ratio)

    # Final Combination
    final_combination = volume_adjusted_momentum * combined_intraday_factors * weighted_momentum

    # Adjust for Volume Impact and Enhanced Price Volatility
    volume_trend = df['volume'] - df['volume'].shift(1)
    avg_true_range = (df['high'] - df['low']).rolling(window=20).mean()
    high_freq_trading_indicator = ((df['high'] + df['low'] + 2 * df['close']) / 4) - \
                                  ((df['high'].shift(1) + df['low'].shift(1) + 2 * df['close'].shift(1)) / 4)

    final_alpha_factor = final_combination * volume_trend * avg_true_range * high_freq_trading_indicator

    return final_alpha_factor
```
