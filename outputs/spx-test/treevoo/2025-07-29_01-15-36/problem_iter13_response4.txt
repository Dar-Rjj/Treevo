```python
import pandas as pd

def heuristics_v2(df, N=10, volume_threshold=1.5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['N_day_price_range_avg'] = df['price_range'].rolling(window=N).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=N).std()
    
    # Adjust Intraday Return by Intraday Volatility
    df['adj_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Determine Range-to-Range Ratio
    df['range_to_range_ratio'] = df['price_range'] / df['price_range'].shift(1)
    
    # Integrate Volume Sensitivity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['adj_range_to_range_ratio'] = df['range_to_range_ratio'] * df['volume_ratio']
    
    # Combine Adjusted Intraday Return and Adjusted Range-to-Range Ratio
    df['combined_intraday_factors'] = df['adj_intraday_return'] * df['adj_range_to_range_ratio']
    
    # Detect Volume Shock
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_shock'] = df['volume_change'] > volume_threshold
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['upward_change'] = df.apply(lambda x: x['close'] - x['close'].shift(1) if x['close'] > x['close'].shift(1) else 0, axis=1)
    df['downward_change'] = df.apply(lambda x: x['close'].shift(1) - x['close'] if x['close'] < x['close'].shift(1) else 0, axis=1)
    df['avg_upward_change'] = df['upward_change'].rolling(window=N).mean()
    df['avg_downward_change'] = df['downward_change'].rolling(window=N).mean()
    df['RS'] = df['avg_upward_change'] / df['avg_downward_change'].abs()
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['N_day_price_range_avg'] * df['volume']) + (df['RSI'] * df['volume'])
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_adjusted_momentum'] * df['combined_intraday_factors']
    
    return df['final_factor']
```
