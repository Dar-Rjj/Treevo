import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = sma_close - df['close']
    
    # Weight by Combined Volume & Price Volatility
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    combined_volatility = avg_volume_20 * avg_high_low_range
    weighted_momentum = momentum_factor * combined_volatility
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate N-Day Relative Strength Index (RSI)
    price_change = df['close'].diff()
    up_changes = 0.0 + price_change
    down_changes = 0.0 + price_change
    up_changes[up_changes < 0] = 0
    down_changes[down_changes > 0] = 0
    avg_up = up_changes.rolling(window=20).mean()
    avg_down = down_changes.abs().rolling(window=20).mean()
    rs = avg_up / avg_down
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (avg_high_low_range.rolling(window=20).mean() * df['volume']) + (rsi * df['volume'])
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / intraday_high_low_spread
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = df['volume'] * close_to_range_ratio
    
    # Final Combination
    final_combination = (combined_intraday_indicators * volume_weighted_close_to_range_momentum) + volume_adjusted_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Accumulation Distribution
    volume_trend = df['volume'].diff()
    accumulation_distribution_line = (((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])) * df['volume']
    accumulation_distribution = accumulation_distribution_line.rolling(window=20).sum()
    price_volatility = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    adjusted_factor = (final_combination * volume_trend * combined_volatility * accumulation_distribution * price_volatility).fillna(0)
    
    return adjusted_factor
