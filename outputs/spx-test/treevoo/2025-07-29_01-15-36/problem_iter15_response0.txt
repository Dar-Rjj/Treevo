import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, volume_threshold=1.5, amount_threshold=1.5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    up_gain = delta.where(delta > 0, 0)
    down_loss = -delta.where(delta < 0, 0)
    avg_gain = up_gain.rolling(window=n).mean()
    avg_loss = down_loss.rolling(window=n).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Intraday Momentum
    df['intraday_range'] = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'].pct_change()
    df['amount_change'] = df['amount'].pct_change()
    df['volume_shock'] = (df['volume_change'] > volume_threshold).astype(int)
    df['amount_shock'] = (df['amount_change'] > amount_threshold).astype(int)
    
    # Calculate N-Day Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Calculate N-Day Moving Average of Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = np.where(df['volume'] > df['ema_volume'], 1, -1)
    
    # Adjust Momentum Factors with Volume and Amount Shock
    df['volume_adjusted_avg_price_range'] = df['avg_price_range'] * df['volume_shock']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume_shoc`k]
    df['volume_adjusted_intraday_range'] = df['intraday_range'] * df['volume_shock']
    df['amount_adjusted_intraday_range'] = df['intraday_range'] * df['amount_shock']
    
    # Calculate N-Day Trade Imbalance
    midpoint = (df['high'] + df['low']) / 2
    df['buy_volume'] = df.apply(lambda x: x['volume'] if x['close'] >= midpoint else 0, axis=1)
    df['sell_volume'] = df.apply(lambda x: x['volume'] if x['close'] < midpoint else 0, axis=1)
    df['n_day_buy_volume'] = df['buy_volume'].rolling(window=n).sum()
    df['n_day_sell_volume'] = df['sell_volume'].rolling(window=n).sum()
    df['trade_imbalance'] = df['n_day_buy_volume'] - df['n_day_sell_volume']
    
    # Calculate N-Day Price Velocity
    df['n_day_price_change'] = df['close'].diff(n)
    df['price_velocity'] = df['n_day_price_change'] / n
    
    # Combine All Adjusted Momentum Factors
    df['combined_factor'] = (
        df['volume_adjusted_avg_price_range'] +
        df['volume_adjusted_rsi'] +
        df['volume_adjusted_intraday_range'] +
        df['amount_adjusted_intraday_range']
    ) * df['volume_trend'] * df['trade_imbalance'] * df['price_velocity']
    
    return df['combined_factor']
