```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate N-Day Exponential Moving Average of Price Range
    df['EMA_price_range'] = df['price_range'].ewm(span=20, adjust=False).mean()

    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']

    # Calculate Volume-Weighted Close-to-Range Momentum
    df['vol_weighted_close_to_range_momentum'] = df['volume'] * df['close_to_range_ratio']

    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    df['sum_vol_weighted_momentum'] = df['vol_weighted_close_to_range_momentum'].rolling(window=5).sum()

    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=9, adjust=False).mean()
    ema_down = down.ewm(com=9, adjust=False).mean()
    rs = ema_up / ema_down
    df['RSI'] = 100 - (100 / (1 + rs))

    # Calculate Volume-Adjusted Momentum
    df['vol_adjusted_ema_price_range'] = df['EMA_price_range'] * df['volume']
    df['vol_adjusted_rsi'] = df['RSI'] * df['volume']
    df['vol_adjusted_momentum'] = df['vol_adjusted_ema_price_range'] + df['vol_adjusted_rsi']

    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    df['open_to_close_momentum'] = df['close'] - df['open']

    # Combine Intraday Indicators
    df['combined_intraday_indicators'] = df['intraday_spread'] * df['open_to_close_momentum'] + df['open_to_close_momentum']

    # Calculate Momentum Factor
    df['momentum_factor'] = (df['SMA_20'] - df['close'])
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['avg_high_low_range_20'] = df['price_range'].rolling(window=20).mean()
    df['avg_trading_range_20'] = ((df['high'] - df['low']) / df['close']).rolling(window=20).mean()
    df['combined_volatility'] = df['avg_volume_20'] * df['avg_high_low_range_20'] / df['avg_trading_range_20']
    df['weighted_momentum_factor'] = df['momentum_factor'] * df['combined_volatility']

    # Final Combination
    df['alpha_factor'] = df['combined_intraday_indicators'] * df['sum_vol_weighted_momentum'] + df['weighted_momentum_factor']

    return df['alpha_factor']
```
