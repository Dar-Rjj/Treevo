```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Indicators
    high_low_spread = df['high'] - df['low']
    open_to_close_momentum = df['close'] - df['open']

    # Combine Intraday Indicators
    intraday_composite = (high_low_spread * open_to_close_momentum) + open_to_close_momentum

    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_spread

    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']

    # Summarize Volume-Weighted Close-to-Range Momentum over 5 days
    summarized_volume_weighted = volume_weighted_close_to_range_momentum.rolling(window=5).sum()

    # Calculate N-Day RSI
    df['prev_close'] = df['close'].shift(1)
    upward_change = np.maximum(0, df['close'] - df['prev_close'])
    downward_change = np.maximum(0, df['prev_close'] - df['close'])

    ema_up = upward_change.ewm(span=14, adjust=False).mean()
    ema_down = downward_change.ewm(span=14, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))

    # Volume-Adjusted Momentum
    volume_adjusted_ema = (high_low_spread.ewm(span=14, adjust=False).mean()) * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    combined_volume_adjusted = volume_adjusted_ema + volume_adjusted_rsi

    # Final Combination
    final_combination = (
        intraday_composite * 
        volume_weighted_close_to_range_momentum * 
        summarized_volume_weighted * 
        combined_volume_adjusted
    )

    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    df['prev_volume'] = df['volume'].shift(1)
    df['prev_prev_volume'] = df['volume'].shift(2)
    volume_trend = df['volume'] - df['prev_volume']
    average_high_low_range = (high_low_spread.rolling(window=20).sum() / 20)
    volume_acceleration = (df['volume'] - df['prev_volume']) - (df['prev_volume'] - df['prev_prev_volume'])

    final_adjustment = (
        final_combination * 
        volume_trend * 
        (1 / average_high_low_range) * 
        volume_acceleration
    )

    return final_adjustment
```
