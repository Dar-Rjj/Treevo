def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday High-Low Spread and Open-to-Close Momentum
    combined_indicator = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Adjust for Volume Impact
    volume_trend = df['volume'] / df['volume'].shift(1)
    intermediate_factor = combined_indicator * volume_trend
    
    # Detect Volume Shock
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock_threshold = 2.0  # Example threshold, can be adjusted
    volume_shock_indicator = (volume_change > volume_shock_threshold).astype(float)
    
    # Combine Intermediate Factor and Volume Shock
    alpha_factor = intermediate_factor * volume_shock_indicator
    
    return alpha_factor
