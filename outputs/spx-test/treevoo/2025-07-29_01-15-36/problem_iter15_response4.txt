```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = data['high'] - data['low']

    # Calculate Open-to-Close Momentum
    open_to_close_momentum = data['close'] - data['open']

    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum

    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (data['close'] - data['low']) / (data['high'] - data['low'])

    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * data['volume']

    # Final Combination
    final_combination = combined_intraday_indicators * volume_weighted_close_to_range_momentum

    # Adjust for Volume Impact and Price Volatility
    # Volume Trend
    data['volume_trend'] = data['volume'].diff()

    # Average True Range Over 20 Days
    data['true_range'] = data[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    data['atr_20'] = data['true_range'].rolling(window=20).mean()

    # Final Adjustment
    final_adjustment = (final_combination * data['volume_trend']) / data['atr_20']
    
    return final_adjustment
```
