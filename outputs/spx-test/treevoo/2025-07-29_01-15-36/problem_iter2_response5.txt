```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range (N=5)
    N = 5
    df['avg_price_range'] = df['price_range'].rolling(window=N).mean()
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['vol_weighted_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (Window Size = 5 days)
    df['sum_vol_weighted_momentum'] = df['vol_weighted_momentum'].rolling(window=N).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['up_change'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['down_change'] = -df['close'].diff().apply(lambda x: x if x < 0 else 0)
    
    df['avg_up_change'] = df['up_change'].rolling(window=N).mean()
    df['avg_down_change'] = df['down_change'].rolling(window=N).mean()
    
    df['RS'] = df['avg_up_change'] / df['avg_down_change'].replace(0, 1e-6)  # Avoid division by zero
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Calculate Volume-Adjusted Momentum
    df['vol_adj_avg_price_range'] = df['avg_price_range'] * df['volume']
    df['vol_adj_RSI'] = df['RSI'] * df['volume']
    df['vol_adj_momentum'] = df['vol_adj_avg_price_range'] + df['vol_adj_RSI']
    
    # Combine Volume-Adjusted Breakout and Momentum
    df['alpha_factor'] = df['sum_vol_weighted_momentum'] + df['vol_adj_momentum']
    
    return df['alpha_factor']
```
