```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['DailyReturn'] = df['close'].pct_change()
    
    # Calculate Price Momentum (10-day, 20-day)
    df['PriceMomentum_10'] = df['DailyReturn'].rolling(window=10).sum()
    df['PriceMomentum_20'] = df['DailyReturn'].rolling(window=20).sum()
    
    # Calculate Volume Change and Sign of Volume Change
    df['VolumeChange'] = df['volume'].diff()
    df['VolumeSign'] = df['VolumeChange'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Difference
    df['HighLowDiff'] = df['high'] - df['low']
    
    # Calculate Mean and Standard Deviation of High-Low Differences
    df['MeanHighLowDiff'] = df['HighLowDiff'].rolling(window=10).mean()
    df['StdHighLowDiff'] = df['HighLowDiff'].rolling(window=10).std()
    
    # Adjust Intraday Return by Intraday Volatility
    df['AdjustedIntradayReturn'] = df['IntradayReturn'] / df['StdHighLowDiff']
    
    # Combine Price Momentum and Adjusted Intraday Return
    df['CombinedValue'] = df['PriceMomentum_10'] * df['AdjustedIntradayReturn']
    
    # Weight by Volume
    df['AlphaFactor'] = df['CombinedValue'] * df['volume']
    
    return df['AlphaFactor']
```
