```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Final Combination of Intraday and Volume-Weighted Momentum
    intermediate_alpha_factor = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Accumulation Distribution
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'].diff().fillna(0)
    
    # Calculate Average True Range Over 20 Days
    def average_true_range(high, low, close, period=20):
        tr = np.maximum.reduce([high - low, abs(high - close.shift(1)), abs(low - close.shift(1))])
        return tr.rolling(window=period).mean()
    
    df['average_true_range'] = average_true_range(df['high'], df['low'], df['close'])
    
    # Calculate Accumulation Distribution Line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    accumulation_distribution_line = money_flow_volume.rolling(window=20).sum()
    
    # Final Adjustment
    final_adjustment = intermediate_alpha_factor * df['volume_trend'] * df['average_true_range'] * accumulation_distribution_line
    
    return final_adjustment
```
