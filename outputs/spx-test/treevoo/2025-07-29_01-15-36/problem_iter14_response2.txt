def heuristics_v2(df):
    """
    Generate an alpha factor that could help predict future stock returns.
    
    The function takes a pandas DataFrame with (open, high, low, close, volume) 
    as columns and date as index. It outputs a pandas Series representing the 
    factor values indexed by date.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (open, high, low, close, volume) columns
    
    Returns:
    pd.Series: A Series with the factor values indexed by date.
    """
    # Calculate the price action component
    df['price_action'] = df['close'] - df['open']
    
    # Calculate the range component
    df['range'] = df['high'] - df['low']
    
    # Calculate the ratio of price action to range
    df['price_range_ratio'] = df['price_action'] / df['range']
    
    # Calculate the volume weighted price action
    df['volume_weighted_price_action'] = df['price_action'] * df['volume']
    
    # Calculate the final factor
    df['factor'] = df['price_range_ratio'] * df['volume_weighted_price_action'].shift(1)
    
    return df['factor']
