```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Adjust by Intraday Volatility
    high_low_diff = df['high'] - df['low']
    mean_high_low = high_low_diff.rolling(window=20).mean()
    std_high_low = high_low_diff.rolling(window=20).std()
    adjusted_intraday_return = intraday_return / (mean_high_low + std_high_low)
    
    # Determine Adjusted Range-to-Range Ratio
    high_low_range_ratio = high_low_diff / high_low_diff.shift(1)
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_range_to_range_ratio = high_low_range_ratio * volume_ratio
    
    # Detect Sudden Increase in Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    sudden_volume_increase = (volume_change > volume_change.quantile(0.95)).astype(int)
    
    # Calculate N-Day Exponential Moving Average (EMA) of Close Prices
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum Factor
    momentum_factor = df['close'] - ema_20
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    up = delta.where(delta > 0, 0)
    down = -delta.where(delta < 0, 0)
    ema_up = up.ewm(span=14, adjust=False).mean()
    ema_down = down.ewm(span=14, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Combined Weighted Momentum
    average_volume_20 = df['volume'].rolling(window=20).mean()
    average_high_low_range_20 = (df['high'] - df['low']).rolling(window=20).mean()
    combined_weighted_momentum = (rsi * df['volume'] +
                                  adjusted_intraday_return * volume_ratio +
                                  momentum_factor * (average_volume_20 + average_high_low_range_20) +
                                  obv * adjusted_range_to_range_ratio)
    
    # Final Factor Calculation
    final_factor = combined_weighted_momentum * adjusted_intraday_return
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'] - df['volume'].shift(1)
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))
    final_adjustment = (final_factor * volume_trend *
                        (1 / average_high_low_range_20) * volume_acceleration)
    
    return final_adjustment
```
