import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, m=5, p=10, q=20, r=10):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    df['ema_price_range'] = df['price_range'].ewm(span=n, adjust=False).mean()
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_c2r'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (e.g., 5 days)
    df['sum_volume_weighted_c2r'] = df['volume_weighted_c2r'].rolling(window=m).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/n, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/n, adjust=False).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Average True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    df['atr'] = true_range.rolling(window=n).mean()
    
    # Determine Volume Trend
    volume_sum = df['volume'].rolling(window=m).sum()
    df['volume_trend'] = np.where(volume_sum > volume_sum.shift(m), 1, -1)
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].rolling(window=p).mean()
    
    # Enhance Volatility Measure
    df['volatility'] = df['close'].rolling(window=q).std() * 1.5
    
    # Combine Factors
    df['combined_factors'] = (df['high_low_range'] * df['atr']) * df['volume_trend'] + df['price_momentum']
    
    # Final Adjustment
    df['final_adjustment'] = df['combined_factors'] * df['volatility']
    df['trend_factor'] = np.where(df['close'].rolling(window=r).mean() > df['close'].shift(r).rolling(window=r).mean(), 1, -1)
    df['final_adjustment'] += df['price_momentum'] * df['trend_factor']
    
    # Integrate All Components
    df['factor'] = df['sum_volume_weighted_c2r'] + df['rsi'] + df['final_adjustment']
    
    return df['factor']
