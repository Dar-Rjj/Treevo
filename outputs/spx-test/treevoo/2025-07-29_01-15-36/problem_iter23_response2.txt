```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_ctr = close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (e.g., 5 days)
    window_size = 5
    summarized_volume_weighted_ctr = volume_weighted_ctr.rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI) (e.g., 14 days)
    n = 14
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=n-1, adjust=False).mean()
    ema_down = down.ewm(com=n-1, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    price_range = (df['high'] - df['low']) * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    volume_adjusted_momentum = price_range + volume_adjusted_rsi
    
    # Calculate Simple Moving Average (SMA) of Close Prices (e.g., 20 days)
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = (df['close'] - sma_close) * (df['volume'].rolling(window=20).mean() * (df['high'] - df['low']).rolling(window=20).mean())
    
    # Final Combination
    final_combination = (combined_intraday * summarized_volume_weighted_ctr * volume_adjusted_momentum) + momentum_factor
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'] - df['volume'].shift(1)
    average_high_low_range_20_days = (df['high'] - df['low']).rolling(window=20).mean()
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))
    
    final_factor = final_combination * volume_trend * (1 / average_high_low_range_20_days) * volume_acceleration
    
    return final_factor
```
