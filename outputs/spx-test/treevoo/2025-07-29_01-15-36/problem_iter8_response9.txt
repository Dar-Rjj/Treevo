```python
import pandas as pd

def heuristics_v2(df, N=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=N).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['up_change'] = df['close'].diff().apply(lambda x: max(x, 0))
    df['down_change'] = df['close'].diff().apply(lambda x: -min(x, 0))
    df['avg_up'] = df['up_change'].rolling(window=N).mean()
    df['avg_down'] = df['down_change'].rolling(window=N).mean()
    df['rs'] = df['avg_up'] / df['avg_down']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['mean_high_low_diff'] = df['high_low_diff'].rolling(window=N).mean()
    df['std_high_low_diff'] = df['high_low_diff'].rolling(window=N).std()
    df['intraday_volatility'] = df['std_high_low_diff'] / df['mean_high_low_diff']
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Determine Range-to-Range Ratio
    df['prev_day_price_range'] = df['price_range'].shift(1)
    df['range_to_range_ratio'] = df['price_range'] / df['prev_day_price_range']
    
    # Integrate Volume Sensitivity
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_range_to_range_ratio'] = df['range_to_range_ratio'] * df['volume_ratio']
    
    # Detect Volume Shock
    volume_threshold = 2.0  # Example threshold
    df['volume_shock'] = (df['volume'] / df['volume'].shift(1) > volume_threshold).astype(int)
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['avg_price_range'] * df['volume']) + (df['rsi'] * df['volume'])
    df['volume_trend'] = (df['volume'] > df['volume'].rolling(window=N).mean()).astype(int) - (df['volume'] < df['volume'].rolling(window=N).mean()).astype(int)
    df['volume_adjusted_momentum'] *= df['volume_trend']
    
    # Combine Adjusted Intraday Return, Adjusted Range-to-Range Ratio, and Volume-Adjusted Momentum
    df['combined_momentum'] = df['adjusted_intraday_return'] * df['adjusted_range_to_range_ratio'] * df['volume_adjusted_momentum']
    
    # Final Factor Calculation
    df['final_factor'] = df['combined_momentum'] * df['volume_shock']
    
    return df['final_factor']
```
