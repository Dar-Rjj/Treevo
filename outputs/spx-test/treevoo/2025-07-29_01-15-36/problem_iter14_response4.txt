```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the difference between close and open prices
    close_open_diff = df['close'] - df['open']
    
    # Calculate the signed volume: positive if closing price is higher than opening, negative otherwise
    signed_volume = df['volume'] * (close_open_diff > 0).astype(int) * 2 - 1
    
    # Calculate a factor based on the ratio of close-open difference to the price range multiplied by the signed volume
    factor = (close_open_diff / price_range) * signed_volume
    
    return factor
```
