```python
def heuristics_v2(df):
    # Calculate the ratio of close price to open price
    close_to_open = df['close'] / df['open']
    
    # Calculate the ratio of high price to low price
    high_to_low = df['high'] / df['low']
    
    # Calculate the average of the two ratios
    avg_ratio = (close_to_open + high_to_low) / 2
    
    # Calculate the change in volume compared to the previous day
    volume_change = df['volume'].pct_change()
    
    # The alpha factor is the product of the average ratio and the change in volume
    alpha_factor = avg_ratio * volume_change
    
    return alpha_factor
```
