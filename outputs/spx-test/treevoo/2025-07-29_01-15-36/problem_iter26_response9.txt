```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']
    
    # Final Combination
    final_combination = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact, Price Volatility, and Accumulation Distribution
    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average True Range Over 20 Days
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-c': abs(df['high'] - df['close'].shift(1)),
        'l-c': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    atr_20 = tr.rolling(window=20).mean()
    
    # Calculate Accumulation Distribution Line
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl_20 = adl.rolling(window=20).sum()
    
    # Calculate Price Volatility
    price_volatility = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    adjusted_final_combination = final_combination * volume_trend * (atr_20 + price_volatility) * adl_20
    
    # Additional Momentum and Volume-Weighted Enhancements
    # Calculate Volume-Weighted Open-to-Close Momentum
    vol_weighted_open_to_close_momentum = open_to_close_momentum * df['volume']
    vol_weighted_open_to_close_momentum_20 = vol_weighted_open_to_close_momentum.rolling(window=20).sum()
    
    # Calculate Volume-Weighted Intraday High-Low Spread
    vol_weighted_high_low_spread = intraday_high_low_spread * df['volume']
    vol_weighted_high_low_spread_20 = vol_weighted_high_low_spread.rolling(window=20).sum()
    
    # Combine Enhanced Indicators
    enhanced_intermediate_alpha_factor = vol_weighted_open_to_close_momentum_20 + vol_weighted_high_low_spread_20
    
    # Final Synthesis
    final_alpha_factor = enhanced_intermediate_alpha_factor * adjusted_final_combination * volume_trend * price_volatility * adl_20
    
    return final_alpha_factor
```
