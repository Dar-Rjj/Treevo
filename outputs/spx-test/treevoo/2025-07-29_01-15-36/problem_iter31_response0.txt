```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices over the last 20 days
    sma_20 = df['close'].rolling(window=20).mean()

    # Calculate the Momentum Factor: Subtract the latest close price from the SMA
    momentum = sma_20 - df['close']

    # Calculate the average volume over the last 20 days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the average high-low range over the last 20 days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Weight the momentum by the combined volume and price volatility
    weight = avg_volume * avg_high_low_range
    weighted_momentum = momentum * weight
    
    return weighted_momentum
```
