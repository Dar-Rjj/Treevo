```python
import pandas as pd

def heuristics_v2(df, m, p):
    # Calculate Intraday High-Low Spread and Range
    intraday_range = df['high'] - df['low']

    # Calculate Open-to-Close Momentum
    open_to_close_momentum = df['close'] - df['open']

    # Determine Volume Trend
    volume_trend = (df['volume'].rolling(window=m).sum() > df['volume'].shift(m).rolling(window=m).sum()).astype(int) * 2 - 1

    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].rolling(window=p).mean()

    # Combine Indicators
    combined_indicator = (intraday_range * open_to_close_momentum) + open_to_close_momentum
    combined_indicator = combined_indicator * volume_trend

    # Final Adjustment
    alpha_factor = combined_indicator + price_momentum

    return alpha_factor
```
