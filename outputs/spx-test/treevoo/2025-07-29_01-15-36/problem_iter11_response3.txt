```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    df['ema_price_range'] = df['price_range'].ewm(span=14, adjust=False).mean()
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (e.g., 5 days)
    df['sum_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=5).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_close_momentum'] = df['close'] - df['open']
    
    # Combine Intraday Indicators
    df['combined_intraday_indicators'] = df['high_low_spread'] * df['open_close_momentum'] + df['open_close_momentu']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Average True Range
    df['true_range'] = df[['high' - 'low', 'high' - df['close'].shift(1), df['close'].shift(1) - 'low']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Determine Volume Trend
    df['volume_trend'] = np.where(df['volume'].rolling(window=5).sum() > df['volume'].shift(5).rolling(window=5).sum(), 1, -1)
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].rolling(window=5).mean()
    
    # Enhance Volatility Measure
    df['enhanced_volatility'] = df['close'].rolling(window=14).std() * 1.5
    
    # Combine Factors
    df['combined_factors'] = (df['high_low_range'] * df['atr']) * df['volume_trend'] + df['price_momentum']
    
    # Final Combination
    df['final_combination'] = (df['combined_intraday_indicators'] * df['volume_weighted_momentum']) + df['sum_volume_weighted_momentum']
    
    # Final Adjustment
    df['final_adjustment'] = df['final_combination'] * df['enhanced_volatility'] + df['close'].pct_change().rolling(window=10).sum()
    
    # Integrate All Components
    df['alpha_factor'] = df['final_adjustment'] + df['rsi'] * df['sum_volume_weighted_momentum']
    
    return df['alpha_factor']
```
