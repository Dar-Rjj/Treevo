```python
import pandas as pd

def heuristics(data):
    # Calculate the Exponential Moving Average (EMA) of Close prices
    ema_close = data['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the latest close price
    latest_close = data['close'].iloc[-1]
    
    # Calculate the Momentum Factor
    momentum_factor = latest_close - ema_close
    
    # Calculate the Exponential Moving Average (EMA) of Volume
    ema_volume = data['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the Exponential Moving Average (EMA) of High-Low Range
    high_low_range = data['high'] - data['low']
    ema_high_low_range = high_low_range.ewm(span=20, adjust=False).mean()
    
    # Combine Volume and Price Volatility
    combined_weight = ema_volume * ema_high_low_range
    
    # Weight the Momentum Factor by the Combined Weight
    weighted_momentum = momentum_factor * combined_weight
    
    return weighted_momentum
```
