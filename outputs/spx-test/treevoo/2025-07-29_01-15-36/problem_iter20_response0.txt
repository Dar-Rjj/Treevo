```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the difference between the high and low prices, and normalize it by the open price
    range_normalized = (df['high'] - df['low']) / df['open']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (pct_change_close + vwap.diff() + range_normalized).fillna(0)
    
    return alpha_factor
```
