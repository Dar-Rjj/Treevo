```python
import pandas as pd

def heuristics_v2(df, n=10, window=5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    df['sum_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=window).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['upward_change'] = (df['close'].diff() > 0) * df['close'].diff().fillna(0)
    df['downward_change'] = -(df['close'].diff() < 0) * df['close'].diff().fillna(0)
    df['avg_upward_change'] = df['upward_change'].rolling(window=n).mean()
    df['avg_downward_change'] = df['downward_change'].rolling(window=n).mean()
    df['rs'] = df['avg_upward_change'] / df['avg_downward_change'].abs()
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Volume-Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['avg_price_range'] * df['volume']) + (df['rsi'] * df['volume'])
    
    # Combine Volume-Adjusted Breakout and Momentum
    df['alpha_factor'] = df['sum_volume_weighted_momentum'] + df['volume_adjusted_momentum']
    
    return df['alpha_factor']
```
