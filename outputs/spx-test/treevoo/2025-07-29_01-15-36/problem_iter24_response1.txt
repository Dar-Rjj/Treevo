```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term SMAs
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Create a momentum indicator from the difference of SMAs
    df['momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Calculate the percentage change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Combine the momentum indicator with the volume change
    df['composite_alpha'] = df['momentum'] * df['volume_change']
    
    # Calculate the RSI over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Integrate the RSI into the composite alpha factor
    df['composite_alpha_rsi'] = df['composite_alpha'] + df['RSI']
    
    # Calculate the Average True Range (ATR) over a 14-day period
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR'] = df['true_range'].rolling(window=14).mean()
    
    # Incorporate the ATR into the composite alpha factor
    df['final_alpha'] = df['composite_alpha_rsi'] / df['ATR']
    
    # Return the final alpha factor
    return df['final_alpha']
```
