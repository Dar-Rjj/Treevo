import pandas as pd

def heuristics(df, n=14, volume_threshold=2.0, amount_threshold=2.0):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # N-Day Moving Average of Price Range
    df['n_day_avg_price_range'] = df['daily_price_range'].rolling(window=n).mean()
    
    # N-Day Relative Strength Index (RSI)
    df['upward_change'] = 0
    df['downward_change'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'upward_change'] = df['close'] - df['close'].shift(1)
    df.loc[df['close'] < df['close'].shift(1), 'downward_change'] = df['close'].shift(1) - df['close']
    df['avg_up'] = df['upward_change'].rolling(window=n).mean()
    df['avg_down'] = df['downward_change'].rolling(window=n).mean()
    df['rs'] = df['avg_up'] / df['avg_down'].abs().replace(0, 0.0001)  # Avoid division by zero
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=n).std()
    
    # Adjusted Intraday Return
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Detect Volume and Amount Shock
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['volume_shock'] = (df['volume_change'] / df['volume'].shift(1)) > volume_threshold
    df['amount_shock'] = (df['amount_change'] / df['amount'].shift(1)) > amount_threshold
    
    # N-Day Moving Average of Volume
    df['n_day_avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Volume Trend
    df['volume_trend'] = 0
    df.loc[df['volume'] > df['n_day_avg_volume'], 'volume_trend'] = 1
    df.loc[df['volume'] < df['n_day_avg_volume'], 'volume_trend'] = -1
    
    # Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = df['volume'] * df['adjusted_intraday_return']
    
    # Momentum
    df['5_day_ma_intraday_return'] = df['intraday_return'].rolling(window=5).mean()
    df['20_day_ma_intraday_return'] = df['intraday_return'].rolling(window=20).mean()
    df['momentum'] = df['5_day_ma_intraday_return'] - df['20_day_ma_intraday_return']
    
    # Adjust Momentum Factors
    df['price_range_adj'] = df['n_day_avg_price_range'] * df['volume_shock']
    df['rsi_adj'] = df['rsi'] * df['volume_shock']
    df['intraday_range_adj'] = df['intraday_range'] * df['volume_shock']
    df['intraday_range_amount_adj'] = df['intraday_range'] * df['amount_shock']
    
    # Combine Adjusted Momentum Factors
    df['combined_factors'] = df['price_range_adj'] + df['rsi_adj'] + df['intraday_range_adj'] + df['intraday_range_amount_adj'] + df['volume_weighted_intraday_return']
    
    # Final Factor
    df['final_factor'] = df['combined_factors'] * df['volume_trend']
    
    return df['final_factor']
