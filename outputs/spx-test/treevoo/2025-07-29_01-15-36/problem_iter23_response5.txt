```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_20
    
    # Calculate Average Volume Over 20 Days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate Average High-Low Range Over 20 Days
    avg_high_low_range_20 = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine Volume and Price Volatility
    combined_weight = avg_volume_20 * avg_high_low_range_20
    
    # Weight the Momentum Factor by Combined Volume & Price Volatility
    weighted_momentum_factor = momentum_factor * combined_weight
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Integrate Momentum and Price Change
    alpha_factor = weighted_momentum_factor + price_change
    
    return alpha_factor
```
