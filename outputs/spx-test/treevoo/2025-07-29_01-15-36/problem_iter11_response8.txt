```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, window_size=5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    df['ema_price_range'] = df['price_range'].ewm(span=n, adjust=False).mean()
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    df['sum_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['upward_change'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['downward_change'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    ema_up = df['upward_change'].ewm(span=n, adjust=False).mean()
    ema_down = df['downward_change'].ewm(span=n, adjust=False).mean()
    df['rs'] = ema_up / ema_down
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_ema'] = df['ema_price_range'] * df['volume']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume']
    df['volume_adjusted_momentum'] = df['volume_adjusted_ema'] + df['volume_adjusted_rsi']
    
    # Calculate Intraday High-Low Spread
    df['intraday_spread'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_to_close_momentum'] = df['close'] - df['open']
    
    # Combine Intraday Indicators
    df['combined_intraday'] = df['intraday_spread'] * df['open_to_close_momentum'] + df['open_to_close_momentum']
    
    # Final Combination
    df['final_combination'] = df['combined_intraday'] * df['sum_volume_weighted_momentum']
    
    # Adjust for Volume Impact
    df['volume_trend'] = df['volume'] - df['volume'].shift(1)
    df['alpha_factor'] = df['final_combination'] * df['volume_trend']
    
    return df['alpha_factor']
```
