```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_diff'].rolling(window=20).std()
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_adjusted_return'] = df['adjusted_return'] * df['volume']
    
    # Calculate Momentum
    df['5_day_ma'] = df['volume_weighted_adjusted_return'].rolling(window=5).mean()
    df['20_day_ma'] = df['volume_weighted_adjusted_return'].rolling(window=20).mean()
    df['momentum'] = df['5_day_ma'] - df['20_day_ma']
    
    return df['momentum'].dropna()
```
