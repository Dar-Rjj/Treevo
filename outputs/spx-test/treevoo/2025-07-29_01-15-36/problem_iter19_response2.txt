```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    n = 5  # Time period for momentum and RSI calculations
    short_period = 10  # Short period for moving average
    long_period = 30  # Long period for moving average
    
    # Price-based Momentum
    price_momentum = (data['close'] - data['close'].shift(n)) / data['close'].shift(n)
    
    # Volume-based Momentum
    volume_momentum = (data['volume'] - data['volume'].shift(n)) / data['volume'].shift(n)
    
    # Moving Average Crossover
    sma_short = data['close'].rolling(window=short_period).mean()
    sma_long = data['close'].rolling(window=long_period).mean()
    moving_average_crossover = sma_short - sma_long
    
    # High-Low Range
    high_low_range = data['high'] - data['low']
    
    # True Range
    true_range = pd.concat([
        (data['high'] - data['low']),
        (data['high'] - data['close'].shift(1)).abs(),
        (data['low'] - data['close'].shift(1)).abs()
    ], axis=1).max(axis=1)
    
    # Price Change Volatility
    price_volatility = data['close'].rolling(window=n).std()
    
    # RSI inspired
    close_diff = data['close'].diff()
    positive_changes = close_diff.where(close_diff > 0, 0)
    absolute_changes = close_diff.abs()
    rsi_inspired = positive_changes.rolling(window=n).mean() / absolute_changes.rolling(window=n).mean()
    
    # MFI inspired
    typical_price = (data['high'] + data['low'] + data['close']) / 3
    money_flow = typical_price * data['volume']
    positive_money_flow = money_flow.where(close_diff > 0, 0)
    negative_money_flow = money_flow.where(close_diff < 0, 0)
    money_flow_ratio = positive_money_flow.rolling(window=n).sum() / negative_money_flow.rolling(window=n).sum()
    mfi_inspired = 100 - (100 / (1 + money_flow_ratio))
    
    # VWAP Inspired Factor
    vwap_inspired = (data['close'] * data['volume']).cumsum() / data['volume'].cumsum()
    
    # Open-Close Spread
    open_close_spread = data['close'] - data['open']
    
    # Gap Analysis
    gap_analysis = data['open'] - data['close'].shift(1)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        price_momentum + 
        volume_momentum + 
        moving_average_crossover + 
        high_low_range + 
        true_range + 
        price_volatility + 
        rsi_inspired + 
        mfi_inspired + 
        vwap_inspired + 
        open_close_spread + 
        gap_analysis
    )
    
    return alpha_factor
```
