```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    recent_mean_high_low = df['high_low_diff'].rolling(window=10).mean()
    recent_std_high_low = df['high_low_diff'].rolling(window=10).std()
    df['intraday_volatility'] = recent_std_high_low
    
    # Adjust Intraday Return by Intraday Volatility
    df['adjusted_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Weight by Volume
    df['volume_weighted_adjusted_return'] = df['adjusted_intraday_return'] * df['volume']
    
    # Incorporate Momentum
    df['5_day_moving_avg_return'] = df['intraday_return'].rolling(window=5).mean()
    df['20_day_moving_avg_return'] = df['intraday_return'].rolling(window=20).mean()
    df['momentum_factor'] = df['5_day_moving_avg_return'] - df['20_day_moving_avg_return']
    
    # Integrate Price Trend
    df['5_day_moving_avg_price'] = df['close'].rolling(window=5).mean()
    df['20_day_moving_avg_price'] = df['close'].rolling(window=20).mean()
    df['price_trend_factor'] = df['5_day_moving_avg_price'] - df['20_day_moving_avg_price']
    
    # Combine Momentum and Price Trend
    df['combined_factor'] = df['momentum_factor'] * df['price_trend_factor']
    
    return df['combined_factor']
```
