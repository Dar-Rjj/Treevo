```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    mean_high_low = high_low_diff.rolling(window=10).mean()
    std_high_low = high_low_diff.rolling(window=10).std()
    intraday_volatility = std_high_low
    
    # Adjust Intraday Return by Intraday Volatility
    adjusted_return = intraday_return / intraday_volatility
    
    # Weight by Volume
    volume_weighted_return = adjusted_return * df['volume']
    
    # Incorporate Momentum
    ema_5 = volume_weighted_return.ewm(span=5, adjust=False).mean()
    ema_20 = volume_weighted_return.ewm(span=20, adjust=False).mean()
    momentum_factor = ema_5 - ema_20
    
    # Integrate Price Trend
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    price_trend_factor = sma_10 - sma_30
    
    # Combine Momentum and Price Trend
    combined_factor = momentum_factor * price_trend_factor
    
    return combined_factor
```
