import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Indicators
    intraday_high_low_spread = df['high'] - df['low']
    open_to_close_momentum = df['close'] - df['open']
    combined_intraday_indicators = (intraday_high_low_spread * open_to_close_momentum) + open_to_close_momentum

    # Calculate Close-to-Range Ratio
    high_low_range = df['high'] - df['low']
    close_to_range_ratio = df['close'] / high_low_range

    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = close_to_range_ratio * df['volume']

    # Final Combination
    intermediate_alpha_factor = combined_intraday_indicators * volume_weighted_close_to_range_momentum

    # Adjust for Volume Impact and Enhanced Price Volatility
    volume_trend = df['volume'].diff()
    average_true_range = ATR(df, 20)
    volume_price_trend = df['volume'] * (df['close'] - df['close'].shift(1))

    # Final Adjustment
    alpha_factor = intermediate_alpha_factor * volume_price_trend

    return alpha_factor

def ATR(df, n):
    high_low = df['high'] - df['low']
    high_close = np.abs(df['high'] - df['close'].shift())
    low_close = np.abs(df['low'] - df['close'].shift())
    ranges = pd.concat([high_low, high_close, low_close], axis=1)
    true_range = np.max(ranges, axis=1)
    atr = true_range.rolling(n).mean()
    return atr
