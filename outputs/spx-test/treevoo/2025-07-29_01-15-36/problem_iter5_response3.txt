```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    n = 14  # Number of days for the EMA
    alpha = 2 / (n + 1)
    df['ema_price_range'] = df['price_range'].ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Close-to-Range Ratio
    df['close_to_range_ratio'] = df['close'] / df['price_range']
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    df['volume_weighted_momentum'] = df['volume'] * df['close_to_range_ratio']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window
    window_size = 5
    df['sum_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=window_size).sum()
    
    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    
    avg_gain = gain.ewm(alpha=alpha, adjust=False).mean()
    avg_loss = loss.ewm(alpha=alpha, adjust=False).mean()
    
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    df['volume_adjusted_ema_price_range'] = df['ema_price_range'] * df['volume']
    df['volume_adjusted_rsi'] = df['rsi'] * df['volume']
    
    # Combine Volume-Adjusted Breakout and Momentum
    df['factor_value'] = (df['sum_volume_weighted_momentum'] + 
                          df['volume_adjusted_ema_price_range'] + 
                          df['volume_adjusted_rsi'])
    
    return df['factor_value']
```
