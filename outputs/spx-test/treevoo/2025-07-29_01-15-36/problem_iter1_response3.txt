```python
import pandas as pd

def heuristics(df):
    # Calculate Simple Moving Average (SMA) of Close prices for the last 20 days
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate the momentum factor
    momentum = df['close'] - sma
    
    # Calculate average volume over the last 20 days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the average high-low range over the last 20 days
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine volume and price volatility
    combined_weight = avg_volume * avg_high_low_range
    
    # Weight the momentum factor by the combined weight
    weighted_momentum = momentum * combined_weight
    
    return weighted_momentum
```
