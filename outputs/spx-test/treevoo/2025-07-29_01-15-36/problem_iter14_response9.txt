```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = open_close_momentum * close_to_range_ratio * df['volume']
    
    # Final Combination
    intermediate_alpha_factor = combined_intraday_indicators * volume_weighted_close_to_range_momentum
    
    # Adjust for Volume Impact and Enhanced Price Volatility
    # Calculate Volume Trend
    volume_trend = df['volume'].diff()
    
    # Calculate Average True Range Over 20 Days
    true_range = df[['high', 'low', 'close']].copy()
    true_range['true_range'] = true_range.max(axis=1) - true_range.min(axis=1)
    average_true_range = true_range['true_range'].rolling(window=20).mean()
    
    # Final Adjustment
    alpha_factor = intermediate_alpha_factor * volume_trend * average_true_range
    
    return alpha_factor
```
