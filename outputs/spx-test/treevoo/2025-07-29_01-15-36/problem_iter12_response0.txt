def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    - df: pandas DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    - pandas Series with (date) as index representing the factor values
    """
    # Calculate the relative strength index (RSI) using the close price
    def compute_rsi(prices, window=14):
        delta = prices.diff()
        up_gain = (delta.where(delta > 0, 0)).fillna(0)
        down_loss = (-delta.where(delta < 0, 0)).fillna(0)
        
        avg_gain = up_gain.rolling(window=window).mean()
        avg_loss = down_loss.rolling(window=window).mean()
        
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi
    
    # Calculate the RSI for the close price
    rsi = compute_rsi(df['close'])
    
    # Calculate the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the percentage change in closing price
    pct_change_close = df['close'].pct_change()
    
    # Combine the RSI, average volume, and percentage change in closing price
    factor = (rsi * avg_volume_5d) * pct_change_close
    
    return factor
