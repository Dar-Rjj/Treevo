```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = df['close'] - sma_close
    
    # Weight by Combined Volume & Price Volatility
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    combined_vol_price_vol = avg_volume * avg_high_low_range
    momentum_factor_weighted = momentum_factor * combined_vol_price_vol
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    high_low_diff = df['high'] - df['low']
    mean_high_low_diff = high_low_diff.rolling(window=20).mean()
    std_high_low_diff = high_low_diff.rolling(window=20).std()
    intraday_volatility = mean_high_low_diff + std_high_low_diff
    
    # Adjust Intraday Return by Intraday Volatility
    adjusted_intraday_return = intraday_return / intraday_volatility
    
    # Determine Range-to-Range Ratio
    prev_day_high_low_range = (df['high'].shift(1) - df['low'].shift(1))
    current_day_high_low_range = df['high'] - df['low']
    range_to_range_ratio = current_day_high_low_range / prev_day_high_low_range
    
    # Integrate Volume Sensitivity
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_range_to_range_ratio = range_to_range_ratio * volume_ratio
    
    # Detect Volume Shock
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock = (volume_change > 3 * std_high_low_diff).astype(int)
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (intraday_high_low_spread * intraday_return) + intraday_return
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / (df['high'] - df['low'])
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    vol_weighted_close_to_range_momentum = combined_intraday_indicators * close_to_range_ratio
    
    # Final Combination
    final_combination = (combined_intraday_indicators * vol_weighted_close_to_range_momentum) * momentum_factor_weighted
    
    # Adjust for Volume Impact, Enhanced Price Volatility, and Volume Acceleration
    volume_trend = df['volume'] - df['volume'].shift(1)
    
    # Calculate Average True Range Over 20 Days
    true_range = pd.concat([df['high'] - df['low'], 
                            (df['high'] - df['close'].shift(1)).abs(), 
                            (df['low'] - df['close'].shift(1)).abs()], axis=1).max(axis=1)
    avg_true_range = true_range.rolling(window=20).mean()
    
    # Volume Acceleration
    vol_acceleration = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))
    
    # Final Adjustment
    alpha_factor = (final_combination * volume_trend * avg_true_range * vol_acceleration)
    
    return alpha_factor
```
