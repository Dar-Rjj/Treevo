```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Exponential Moving Average of Price Range
    ema_price_range = df['price_range'].ewm(span=n, adjust=False).mean()
    
    # Calculate N-Day Relative Strength Index (RSI)
    df['up_change'] = 0.0
    df['down_change'] = 0.0
    df.loc[df['close'] > df['close'].shift(1), 'up_change'] = df['close'] - df['close'].shift(1)
    df.loc[df['close'] < df['close'].shift(1), 'down_change'] = df['close'].shift(1) - df['close']
    
    ema_up = df['up_change'].ewm(span=n, adjust=False).mean()
    ema_down = df['down_change'].ewm(span=n, adjust=False).mean()
    
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-Adjusted Momentum
    volume_adjusted_ema_price_range = ema_price_range * df['volume']
    volume_adjusted_rsi = rsi * df['volume']
    
    combined_results = volume_adjusted_ema_price_range + volume_adjusted_rsi
    
    # Introduce N-Day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Adjust Combined Results with N-Day EMA of Volume
    adjusted_combined_results = combined_results / ema_volume
    
    return adjusted_combined_results
```
