```python
import pandas as pd

def heuristics_v2(df):
    n = 14  # Number of days for ATR and RSI
    m = 5   # Number of days for volume trend
    p = 5   # Number of days for price momentum

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Compute Average True Range
    df['True_Range'] = df[['high' - 'low', 'high' - df['close'].shift(1), df['close'].shift(1) - 'low']].max(axis=1)
    df['Average_True_Range'] = df['True_Range'].rolling(window=n).mean()

    # Determine Volume Trend
    df['Volume_Sum_Current'] = df['volume'].rolling(window=m).sum()
    df['Volume_Sum_Prev'] = df['volume'].shift(m).rolling(window=m).sum()
    df['Volume_Trend'] = (df['Volume_Sum_Current'] > df['Volume_Sum_Prev']).astype(int) * 2 - 1

    # Detect Volume Spike
    df['Avg_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Spike'] = (df['volume'] > 2 * df['Avg_Volume']).astype(float) + 1.0

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].rolling(window=p).mean()

    # Calculate N-Day Moving Average of Price Range
    df['N_Day_Price_Range'] = df['High_Low_Range'].rolling(window=n).mean()

    # Calculate N-Day Relative Strength Index (RSI)
    df['Upward_Change'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['Downward_Change'] = -df['close'].diff().apply(lambda x: x if x < 0 else 0)
    df['Avg_Up'] = df['Upward_Change'].rolling(window=n).mean()
    df['Avg_Down'] = df['Downward_Change'].rolling(window=n).mean()
    df['RS'] = df['Avg_Up'] / df['Avg_Down']
    df['RSI'] = 100 - (100 / (1 + df['RS']))

    # Combine Factors
    df['Combined_Factor_1'] = (df['High_Low_Range'] * df['Average_True_Range']) * df['Volume_Trend'] * df['Volume_Spike']
    df['Combined_Factor_2'] = df['Price_Momentum']
    df['Combined_Factor_3'] = (df['N_Day_Price_Range'] * df['volume']) + (df['RSI'] * df['volume'])

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factor_1'] + df['Combined_Factor_2'] + df['Combined_Factor_3']

    return df['Final_Alpha_Factor']
```
