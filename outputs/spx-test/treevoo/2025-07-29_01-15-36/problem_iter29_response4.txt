import pandas as pd

def heuristics_v2(df):
    # Intraday Indicators
    intraday_high_low_spread = df['high'] - df['low']
    open_to_close_momentum = df['close'] - df['open']
    combined_intraday_indicators = intraday_high_low_spread * open_to_close_momentum
    
    # Close-to-Range Ratio
    high_low_range = df['high'] - df['low']
    close_to_range_ratio = df['close'] / high_low_range
    
    # Volume-Weighted Close-to-Range Momentum
    volume_weighted_momentum = combined_intraday_indicators * close_to_range_ratio * df['volume']
    
    # Summarize Volume-Weighted Close-to-Range Momentum over a Window (e.g., 5 days)
    window_size = 5
    summarized_volume_weighted_momentum = volume_weighted_momentum.rolling(window=window_size).sum()
    
    # N-Day Relative Strength Index (RSI)
    n = 14  # Common period for RSI
    price_diff = df['close'].diff()
    up_chg = 0 * price_diff
    down_chg = 0 * price_diff
    up_chg[price_diff > 0] = price_diff[price_diff > 0]
    down_chg[price_diff < 0] = price_diff[price_diff < 0].abs()
    avg_gain = up_chg.ewm(com=n-1, min_periods=n).mean()
    avg_loss = down_chg.ewm(com=n-1, min_periods=n).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted Momentum
    moving_average_price_range = (df['high'] - df['low']).rolling(window=20).mean()
    volume_adjusted_momentum = (moving_average_price_range * df['volume']) + (rsi * df['volume'])
    
    # Volume-Price Trend
    vpt = (df['volume'] * (df['close'] - df['close'].shift(1))).cumsum()
    
    # Final Alpha Factor Combination
    final_alpha_factor = (
        volume_weighted_momentum * vpt *
        combined_intraday_indicators * volume_adjusted_momentum *
        summarized_volume_weighted_momentum + volume_adjusted_momentum
    )
    
    # Adjust for Volume Impact, Price Volatility, and Volume Acceleration
    volume_trend = df['volume'] - df['volume'].shift(1)
    average_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) - (df['volume'].shift(1) - df['volume'].shift(2))
    price_volatility = df['close'].rolling(window=20).std()
    
    # Final Adjustment
    adjusted_alpha_factor = (
        final_alpha_factor * volume_trend * (1 / average_high_low_range) * volume_acceleration + price_volatility
    )
    
    return adjusted_alpha_factor
