import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['Open_Close_Momentum'] = df['close'] - df['open']
    
    # Combine Intraday Indicators
    df['Intraday_Indicator'] = (df['High_Low_Spread'] * df['Open_Close_Momentum']) + df['Open_Close_Momentum']
    
    # Calculate N-Day Moving Average of Price Range
    df['Price_Range'] = df['high'] - df['low']
    df['N_Day_Price_Range_MA'] = df['Price_Range'].rolling(window=n).mean()
    
    # Calculate N-Day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Detect Volume and Amount Shock
    df['Volume_Shock'] = (df['volume'] > df['volume'].shift(1) * 1.5).astype(int)
    df['Amount_Shock'] = (df['amount'] > df['amount'].shift(1) * 1.5).astype(int)
    
    # Calculate N-Day Exponential Moving Average of Volume
    df['Volume_EMA'] = df['volume'].ewm(span=n, adjust=False).mean()
    
    # Calculate Volume Trend
    df['Volume_Trend'] = (df['volume'] > df['Volume_EMA']).astype(int) * 2 - 1
    
    # Adjust Momentum Factors with Volume and Amount Shock
    df['Adjusted_N_Day_Price_Range'] = df['N_Day_Price_Range_MA'] * df['Volume_Shock']
    df['Adjusted_RSI'] = df['RSI'] * df['Volume_Shock']
    df['Volume_Adjusted_Intraday_Spread'] = df['Intraday_Indicator'] * df['Volume_Shock']
    df['Amount_Adjusted_Intraday_Spread'] = df['Intraday_Indicator'] * df['Amount_Shock']
    
    # Calculate N-Day Trade Imbalance
    midpoint = (df['high'] + df['low']) / 2
    df['Buy_Volume'] = df['volume'].where(df['close'] >= midpoint, 0).rolling(window=n).sum()
    df['Sell_Volume'] = df['volume'].where(df['close'] < midpoint, 0).rolling(window=n).sum()
    df['Trade_Imbalance'] = df['Buy_Volume'] - df['Sell_Volume']
    
    # Combine All Adjusted Momentum Factors
    combined_momentum = (
        df['Adjusted_N_Day_Price_Range'] + 
        df['Adjusted_RSI'] + 
        df['Volume_Adjusted_Intraday_Spread'] + 
        df['Amount_Adjusted_Intraday_Spread']
    ) * df['Volume_Trend'] * df['Trade_Imbalance']
    
    # Calculate Simple Moving Average (SMA) of Close prices
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate SMA Momentum
    df['SMA_Momentum'] = df['close'] - df['SMA_20']
    
    # Integrate RSI and SMA Momentum
    df['Volume_Adjusted_Momentum'] = df['N_Day_Price_Range_MA'] * df['volume'] + df['RSI'] * df['volume']
    df['Final_Momentum'] = (combined_momentum * df['Volume_Adjusted_Momentum']) + df['SMA_Momentum']
    
    # Calculate N-Day Accumulation/Distribution Line
    df['Money_Flow_Multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['Money_Flow_Volume'] = df['Money_Flow_Multiplier'] * df['volume']
    df['Accum_Dist_Line'] = df['Money_Flow_Volume'].rolling(window=n).sum()
    
    # Output as Alpha Factor
    return df['Final_Momentum'] + df['Accum_Dist_Line']
