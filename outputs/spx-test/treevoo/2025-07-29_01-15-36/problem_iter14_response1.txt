```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Combine Intraday Indicators
    combined_intraday_indicators = (high_low_spread * open_close_momentum) + open_close_momentum
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Price Range
    moving_avg_price_range = daily_price_range.rolling(window=20).mean()
    
    # Calculate N-Day Relative Strength Index (RSI) with N=14
    N = 14
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=N).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=N).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted Momentum
    volume_adjusted_momentum = (moving_avg_price_range * df['volume']) + (rsi * df['volume'])
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_spread
    
    # Calculate Volume-Weighted Close-to-Range Momentum
    volume_weighted_close_to_range_momentum = df['volume'] * close_to_range_ratio
    
    # Final Combination
    final_combination = (combined_intraday_indicators * volume_weighted_close_to_range_momentum) + volume_adjusted_momentum
    
    # Adjust for Volume Impact
    volume_trend = df['volume'].diff()
    final_alpha_factor = final_combination * volume_trend
    
    return final_alpha_factor
```
