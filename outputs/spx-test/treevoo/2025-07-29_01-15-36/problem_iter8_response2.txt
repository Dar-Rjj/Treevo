```python
import pandas as pd

def heuristics_v2(df, n=14, m=10, p=20):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['High'] - df['Low']

    # Compute Average True Range
    df['True_Range'] = df[['High', 'Close']].max(axis=1) - df[['Low', 'Close']].min(axis=1)
    df['Average_True_Range'] = df['True_Range'].rolling(window=n).mean()

    # Determine Volume Trend
    df['Volume_Sum_Last_m_Days'] = df['Volume'].rolling(window=m).sum()
    df['Volume_Trend'] = (df['Volume_Sum_Last_m_Days'].shift(-m) > df['Volume_Sum_Last_m_Days']).astype(int) * 2 - 1

    # Calculate Price Momentum
    df['Price_Momentum'] = df['Close'] - df['Close'].rolling(window=p).mean()

    # Calculate N-Day Exponential Moving Average of Price Range
    df['Daily_Price_Range'] = df['High'] - df['Low']
    df['EMA_Price_Range'] = df['Daily_Price_Range'].ewm(span=n, adjust=False).mean()

    # Calculate N-Day Relative Strength Index (RSI)
    delta = df['Close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.ewm(span=n, min_periods=n, adjust=False).mean()
    avg_loss = loss.ewm(span=n, min_periods=n, adjust=False).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))

    # Combine Factors
    high_low_range_atr = df['High_Low_Range'] * df['Average_True_Range']
    volume_adjusted_high_low_range_atr = high_low_range_atr * df['Volume_Trend']
    combined_factors = volume_adjusted_high_low_range_atr + df['Price_Momentum']

    ema_price_range_volume = df['EMA_Price_Range'] * df['Volume']
    rsi_volume = df['RSI'] * df['Volume']

    final_alpha_factor = combined_factors + ema_price_range_volume + rsi_volume

    return final_alpha_factor.dropna()
```
