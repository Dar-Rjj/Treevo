```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, p=10, q=60):
    # Using Basic Price Information
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']

    # Utilizing Volume and Amount Data
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['avg_transaction_size'] = df['amount'] / df['volume']

    # Combining Multiple Metrics for Enhanced Factors
    df['price_momentum'] = df['close'] - df['close'].rolling(window=n).mean()
    df['volatility'] = df['daily_return'].rolling(window=n).std()
    df['momentum_vol_increase'] = (df['price_momentum'] * (df['volume_change'] > 0) * (df['daily_return'] > 0))

    pos_intraday_returns = df['intraday_return'][df['intraday_return'] > 0].rolling(window=m).sum()
    neg_intraday_returns = df['intraday_return'][df['intraday_return'] < 0].rolling(window=m).sum().abs()
    df['pos_neg_ratio'] = pos_intraday_returns / neg_intraday_returns

    # Incorporating Market Behavior Patterns
    large_move_threshold = 0.02
    df['large_moves'] = (df['daily_return'].abs() > large_move_threshold).astype(int)
    df['market_excitement'] = df['large_moves'].rolling(window=p).sum()

    df['trend_indicator'] = df['close'].rolling(window=q).apply(lambda x: pd.Series(x).reset_index(drop=True).corr(pd.Series(range(q))), raw=False)

    # Exploring Interactions Between Indicators
    df['return_consistency'] = df['daily_return'] * (df['intraday_return'] > 0).astype(int) + df['daily_return'] * (df['intraday_return'] < 0).astype(int) * -1
    df['trend_vs_momentum'] = df['trend_indicator'] * (df['price_momentum'] > 0).astype(int) - df['trend_indicator'] * (df['price_momentum'] < 0).astype(int)
    
    df['bearish_bullish_signal'] = df['upper_shadow'] / (df['high'] - df['low'])
    df['bearish_bullish_signal'] = 1 / df['bearish_bullish_signal']

    # Advanced Composite Factor Construction
    factor_weights = {
        'price_momentum': 0.3,
        'volatility': -0.2,
        'momentum_vol_increase': 0.4,
        'pos_neg_ratio': 0.1,
        'market_excitement': 0.2,
        'trend_indicator': 0.3,
        'return_consistency': 0.1,
        'trend_vs_momentum': 0.1,
        'bearish_bullish_signal': 0.1
    }

    df['composite_factor'] = 0
    for factor, weight in factor_weights.items():
        df['composite_factor'] += df[factor] * weight

    return df['composite_factor'].fillna(0)
```
