```python
import pandas as pd

def heuristics_v2(df, window_size=5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-to-Range Ratio
    close_to_range_ratio = df['close'] / high_low_range
    
    # Multiply Volume by Close-to-Range Ratio
    volume_weighted_momentum = close_to_range_ratio * df['volume']
    
    # Summarize over a Window
    alpha_factor = volume_weighted_momentum.rolling(window=window_size).sum()
    
    return alpha_factor
```
