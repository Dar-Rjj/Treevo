```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=N).mean()
    
    # Calculate Upward and Downward Price Changes for RSI
    df['up_change'] = df['close'].diff().where(df['close'].diff() > 0, 0)
    df['down_change'] = -df['close'].diff().where(df['close'].diff() < 0, 0)
    
    # Calculate Average Upward and Downward Changes over N days
    df['avg_up_change'] = df['up_change'].rolling(window=N).mean()
    df['avg_down_change'] = df['down_change'].rolling(window=N).mean()
    
    # Compute Relative Strength (RS)
    df['rs'] = df['avg_up_change'] / df['avg_down_change'].replace(0, np.nan)
    
    # Calculate RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Intraday Volatility
    df['high_low_diff'] = df['high'] - df['low']
    df['mean_high_low'] = df['high_low_diff'].rolling(window=N).mean()
    df['std_high_low'] = df['high_low_diff'].rolling(window=N).std()
    df['intraday_volatility'] = df['std_high_low'] / df['mean_high_low']
    
    # Adjust Intraday Return by Intraday Volatility
    df['adj_intraday_return'] = df['intraday_return'] / df['intraday_volatility']
    
    # Calculate N-Day Moving Average of Volume
    df['avg_volume'] = df['volume'].rolling(window=N).mean()
    
    # Calculate Volume Trend
    df['volume_trend'] = np.where(df['volume'] > df['avg_volume'], 1, -1)
    
    # Volume-Adjusted Momentum
    df['vol_adj_mom'] = (df['avg_price_range'] * df['volume']) + (df['rsi'] * df['volume'])
    df['vol_adj_mom'] = df['vol_adj_mom'] * df['volume_trend']
    
    # Weight by Volume
    df['factor'] = df['vol_adj_mom'] * df['volume']
    
    return df['factor']
```
