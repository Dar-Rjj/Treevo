```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices for the last 20 days
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the average volume over the last 20 days
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the average high-low range over the last 20 days
    df['Avg_High_Low_Range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine volume and price volatility (high-low range)
    df['Combined_Weight'] = df['Avg_Volume_20'] * df['Avg_High_Low_Range_20']
    
    # Calculate Momentum Factor: Subtract latest close price from SMA and weight by combined volume & price volatility
    df['Momentum_Factor'] = (df['SMA_20'] - df['close']) * df['Combined_Weight']
    
    # Return the factor values as a pandas Series
    return df['Momentum_Factor']
```
