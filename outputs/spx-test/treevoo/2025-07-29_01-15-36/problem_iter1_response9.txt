```python
def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_range = df['high'] - df['low']
    
    # Detect Volume Shock
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_shock_threshold = 5 * df['volume'].rolling(window=20).std()  # Example threshold, can be adjusted
    volume_shock_indicator = (volume_change > volume_shock_threshold).astype(int)

    # Combine Intraday Momentum and Volume Shock
    combined_factor = intraday_range * volume_shock_indicator
    
    return combined_factor
```
