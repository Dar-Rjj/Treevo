import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate N-Day Moving Average of Price Range
    df['avg_price_range'] = df['price_range'].rolling(window=n).mean()
    
    # Calculate Upward and Downward Price Changes for RSI
    df['up_change'] = 0
    df['down_change'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'up_change'] = df['close'] - df['close'].shift(1)
    df.loc[df['close'] < df['close'].shift(1), 'down_change'] = df['close'].shift(1) - df['close']
    
    # Calculate Average Upward and Downward Change over N days
    df['avg_up_change'] = df['up_change'].rolling(window=n).mean()
    df['avg_down_change'] = df['down_change'].rolling(window=n).mean()
    
    # Compute RS (Relative Strength)
    df['rs'] = df['avg_up_change'] / df['avg_down_change'].abs()
    
    # Calculate RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Calculate Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['avg_price_range'] * df['volume']) + (df['rsi'] * df['volume'])
    
    return df['volume_adjusted_momentum']
