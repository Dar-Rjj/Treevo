import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x[0], x[1]), axis=1) - df['close'].shift(1)
    df['true_range'] = df['true_range'].fillna(0)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_log_return_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_adjusted_factor'] = df['volume_weighted_log_return_sum'] / df['volume'].rolling(window=n).sum()
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Incorporate Volume Weight
    df['volume_weighted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, 1, n))
    weights /= weights.sum()  # Normalize to sum to 1
    df['exp_smoothed_factor'] = df['volume_weighted_atr'].rolling(window=n).apply(lambda x: (x * weights).sum(), raw=False)
    
    # Incorporate Volume Trend
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_trend'] = df['volume'] / df['avg_volume']
    df['vol_trend_adjusted_factor'] = df['exp_smoothed_factor'] * df['volume_trend']
    
    # Introduce Price Momentum
    df['close_sma'] = df['close'].rolling(window=m).mean()
    df['price_momentum'] = df['close'] - df['close_sma']
    
    # Combine Momentum and Volume-Weighted Factors
    df['final_factor'] = df['price_momentum'] * df['vol_trend_adjusted_factor']
    
    return df['final_factor']
