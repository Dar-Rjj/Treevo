```python
import pandas as pd

def heuristics_v2(df):
    # Select High and Low Prices
    high_prices = df['high']
    low_prices = df['low']
    
    # Compute High-Low Difference
    high_low_diff = high_prices - low_prices
    
    # Choose Lookback Period (e.g., 14 days)
    lookback_period = 14
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum
    raw_momentum = high_low_diff - sma_high_low_diff
    
    # Apply Exponential Weighting
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Add Offset to Shift to Positive Range
    min_value = 0
    max_possible_value = (high_prices - low_prices).max()
    offset = abs(min_value) + max_possible_value
    
    # Scale to Desired Range (e.g., 0 to 100) Using Multiplication
    scaled_momentum = (smoothed_momentum + offset) / (2 * max_possible_value) * 100
    
    return scaled_momentum
```
