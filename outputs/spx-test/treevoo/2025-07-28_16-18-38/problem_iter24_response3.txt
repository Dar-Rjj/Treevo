import pandas as pd

def heuristics_v2(df):
    # Parameters
    lookback_period = 14
    smoothing_factor = 0.2
    offset = 0
    volatility_lookback = 20
    volume_weighted_lookback = 30
    
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Calculate Simple Moving Average (SMA) of High-Low Difference
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum
    raw_momentum = high_low_diff - sma_high_low_diff
    
    # Apply Exponential Smoothing
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Add Offset to Shift to Positive Range
    smoothed_momentum += offset
    
    # Scale to Desired Range (0 to 100)
    max_value = (df['high'] - df['low']).max()
    bounded_momentum = (smoothed_momentum / max_value) * 100
    
    # Apply Rolling Window Volatility Filter
    rolling_volatility = high_low_diff.rolling(window=volatility_lookback).std()
    adjusted_momentum = smoothed_momentum / rolling_volatility
    adjusted_momentum[rolling_volatility == 0] = 0  # Ensure non-zero volatility
    
    # Integrate Volume-Weighted Momentum
    volume_weighted_high_low_diff = (df['high'] - df['low']) * df['volume']
    sma_volume_weighted_high_low_diff = volume_weighted_high_low_diff.rolling(window=volume_weighted_lookback).mean()
    volume_weighted_momentum = sma_volume_weighted_high_low_diff / df['volume'].rolling(window=volume_weighted_lookback).mean()
    
    # Final Alpha Factor
    alpha_factor = (adjusted_momentum + volume_weighted_momentum) / 2
    
    return alpha_factor
