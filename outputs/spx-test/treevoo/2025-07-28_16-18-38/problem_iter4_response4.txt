```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Compute High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Weight by Volume
    volume_weighted_high_to_low_return = high_to_low_return * df['volume']
    
    # Compute Close-to-Open Return of Previous Day
    close_to_open_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Sum Volume-Weighted High-to-Low Return and Close-to-Open Return
    combined_return = volume_weighted_high_to_low_return + close_to_open_return
    
    # Accumulate the Combined Value Over N Days
    accumulated_value = combined_return.rolling(window=N).sum()
    
    return accumulated_value
```
