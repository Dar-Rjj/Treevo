```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Volume-Weighted High-to-Low Range
    volume_weighted_high_to_low_range = high_to_low_range * df['volume']
    
    # Calculate 7-day Exponential Moving Average (EMA) of Volume-Weighted High-to-Low Range
    ema_7 = volume_weighted_high_to_low_range.ewm(span=7, adjust=False).mean()
    
    # Calculate 21-day Exponential Moving Average (EMA) of Volume-Weighted High-to-Low Range
    ema_21 = volume_weighted_high_to_low_range.ewm(span=21, adjust=False).mean()
    
    # Compute the Ratio of 7-day EMA to 21-day EMA
    factor = ema_7 / ema_21
    
    return factor
```
