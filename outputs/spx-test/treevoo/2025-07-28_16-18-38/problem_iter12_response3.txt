```python
import pandas as pd

def heuristics_v2(df):
    lookback_period = 14
    smoothing_factor = 0.2
    min_value = 0
    max_value = 100

    # Calculate High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']

    # Calculate Exponential Moving Average (EMA) of High-Low Difference
    df['EMA_High_Low_Diff'] = df['High_Low_Diff'].ewm(span=lookback_period, adjust=False).mean()

    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['High_Low_Diff'] - df['EMA_High_Low_Diff']

    # Apply Smoothing
    df['Smoothed_Momentum'] = df['Raw_Momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Calculate Volume-Weighted High and Low
    df['Volume_Weighted_High'] = (df['high'] * df['volume']).rolling(window=lookback_period).sum()
    df['Volume_Weighted_Low'] = (df['low'] * df['volume']).rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    df['Volume_Weighted_High_Low_Diff'] = (df['Volume_Weighted_High'] - df['Volume_Weighted_Low']) / total_volume

    # Transform to Bounded Range
    offset = -df['Smoothed_Momentum'].min() + min_value
    df['Bounded_Momentum'] = (df['Smoothed_Momentum'] + offset) * (max_value / (df['Smoothed_Momentum'].max() + offset))

    return df['Bounded_Momentum']
```
