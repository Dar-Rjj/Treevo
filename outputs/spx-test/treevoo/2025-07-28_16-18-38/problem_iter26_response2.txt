```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Simple Moving Average (SMA) of Close prices over a certain period
    sma_period = 14
    df['sma_close'] = df['close'].rolling(window=sma_period).mean()
    df['sma_ratio'] = df['close'] / df['sma_close']
    
    # Exponential Moving Average (EMA) of Close prices
    ema_period = 14
    df['ema_close'] = df['close'].ewm(span=ema_period, adjust=False).mean()
    df['ema_ratio'] = df['close'] / df['ema_close']
    
    # Volume ratio
    avg_volume_period = 14
    df['avg_volume'] = df['volume'].rolling(window=avg_volume_period).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Price-volume momentum
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    
    # Large trade indicator
    df['large_trade_indicator'] = (df['volume'] > 2 * df['avg_volume']).astype(int)
    
    # Dynamic Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=sma_period).mean()
    df['raw_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    smoothing_factor = 0.2
    df['dynamic_momentum'] = df['raw_momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Composite factor
    composite_weights = {
        'sma_ratio': 0.15,
        'ema_ratio': 0.15,
        'intraday_volatility': 0.15,
        'volume_ratio': 0.15,
        'price_volume_momentum': 0.15,
        'large_trade_indicator': 0.15,
        'dynamic_momentum': 0.1
    }
    df['composite_factor'] = (
        df['sma_ratio'] * composite_weights['sma_ratio'] +
        df['ema_ratio'] * composite_weights['ema_ratio'] +
        df['intraday_volatility'] * composite_weights['intraday_volatility'] +
        df['volume_ratio'] * composite_weights['volume_ratio'] +
        df['price_volume_momentum'] * composite_weights['price_volume_momentum'] +
        df['large_trade_indicator'] * composite_weights['large_trade_indicator'] +
        df['dynamic_momentum'] * composite_weights['dynamic_momentum']
    )
    
    # Synthesize Momentum Indicators
    df['high_to_low_range'] = df['high'] - df['low']
    df['ema_7_high_low'] = df['high_to_low_range'].ewm(span=7, adjust=False).mean()
    df['ema_21_high_low'] = df['high_to_low_range'].ewm(span=21, adjust=False).mean()
    df['combined_momentum'] = df['ema_7_high_low'] / df['ema_21_high_low'] * df['intraday_volatility']
    
    # Enhanced Components
    df['intraday_volatility_enhanced'] = np.abs(df['close'] - df['open'])
    df['volume_trend'] = df['volume'] - df['volume'].rolling(window=5).mean()
    df['enhanced_components'] = df['intraday_volatility_enhanced'] + df['volume_trend']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['enhanced_components'] * df['combined_momentum'] + df['composite_factor']
    
    return df['final_alpha_factor']
```
