```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Return and weight by volume
    df['high_low_return'] = df['high'] - df['low']
    df['volume_weighted_return'] = df['high_low_return'] * df['volume']
    
    # Incorporate Previous Day's Close-to-Open Movement
    df['close_to_open_return'] = df['close'].shift(1) - df['open']
    df['short_term_momentum'] = df['volume_weighted_return'] + df['close_to_open_return']
    
    # Calculate Moving Average of High-to-Low Difference
    lookback_period = 14
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum
    df['raw_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    
    # Apply Smoothing
    smoothing_factor = 0.2
    df['smoothed_momentum'] = df['raw_momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Accumulate the Combined Value Over N Days
    n_days = 7
    df['accumulated_momentum'] = df['smoothed_momentum'].rolling(window=n_days).sum()
    
    # Combine with Volume-Weighted Close Price
    df['close_volume_product'] = df['close'] * df['volume']
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    df['vwcp'] = (df['close_volume_product'].rolling(window=lookback_period).sum()) / total_volume
    scaling_factor = 0.5
    df['final_dynamic_momentum'] = scaling_factor * (df['accumulated_momentum'] * df['vwcp'])
    
    # Long-Term Price Momentum Adjustment
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['adjusted_close_difference'] = df['close'] - df['10_day_avg_close']
    
    # Combine with Short-Term Momentum
    df['combined_short_term_momentum'] = df['short_term_momentum'] + df['final_dynamic_momentum']
    df['final_alpha_factor'] = df['adjusted_close_difference'] / df['combined_short_term_momentum']
    
    return df['final_alpha_factor']
```
