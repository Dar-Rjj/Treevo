```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, sma_n=10, exp_smoothing_alpha=0.2):
    # Compute High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Volume-Weighted High-to-Low Return
    volume_weighted_hl_return = high_to_low_return * df['volume']
    
    # Close-to-Open Return of Previous Day
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine with Volume-Weighted High-to-Low Return
    combined_return = volume_weighted_hl_return + close_to_open_return
    
    # True Range
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1).shift(1)
    true_range.fillna(0, inplace=True)
    
    # Volume-Weighted Moving Sum of Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volume_weighted_log_returns = log_returns * df['volume']
    moving_sum_log_returns = volume_weighted_log_returns.rolling(window=n).sum()
    
    # Volume-Adjusted Factor
    sum_volume = df['volume'].rolling(window=n).sum()
    volume_adjusted_factor = moving_sum_log_returns / sum_volume
    
    # Average True Range (ATR)
    atr = true_range.rolling(window=n).mean()
    
    # Volume Weight
    volume_weight = atr * volume_adjusted_factor
    
    # Apply Exponential Smoothing
    smoothed_factor = volume_weight.ewm(alpha=exp_smoothing_alpha, adjust=False).mean()
    
    # Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=sma_n).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothed_raw_momentum = raw_momentum.ewm(alpha=exp_smoothing_alpha, adjust=False).mean()
    
    # Relative Strength
    n_day_close_change = df['close'] - df['close'].shift(sma_n)
    sma_n_day_close_change = n_day_close_change.rolling(window=sma_n).mean()
    relative_strength = n_day_close_change - sma_n_day_close_change
    
    # Combine Volume-Weighted High-to-Low Return, True Range Momentum, Dynamic Momentum, and Relative Strength
    combined_factor = combined_return + smoothed_factor + smoothed_raw_momentum + relative_strength
    
    # Incorporate Volume Trend
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_trend_ratio = df['volume'] / avg_volume
    final_factor = combined_factor * volume_trend_ratio
    
    return final_factor
```
