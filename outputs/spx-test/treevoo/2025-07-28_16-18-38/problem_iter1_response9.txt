```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute 7-day Exponential Moving Average (EMA) of High-to-Low Range
    ema_7 = high_low_range.ewm(span=7, adjust=False).mean()
    
    # Compute 21-day Exponential Moving Average (EMA) of High-to-Low Range
    ema_21 = high_low_range.ewm(span=21, adjust=False).mean()
    
    # Compute the Ratio of 7-day EMA to 21-day EMA
    ema_ratio = ema_7 / ema_21
    
    return ema_ratio
```
