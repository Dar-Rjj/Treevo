```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=50, k=10):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'], x['close'].shift(1)) - min(x['low'], x['close'].shift(1)), axis=1
    )
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_sum'] = df['volume'].rolling(window=n).sum()
    df['volume_adjusted_factor'] = df['volume_weighted_sum'] / df['volume_sum']
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Incorporate Volume Weight
    df['volume_weighted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(-1, 0, n))
    weights /= weights.sum()  # Normalize weights
    df['exp_smoothed'] = df['volume_weighted_atr'].rolling(window=n).apply(
        lambda x: np.dot(x, weights), raw=True
    )
    
    # Incorporate Volume Trend
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_trend_ratio'] = df['volume'] / df['average_volume']
    df['exp_smoothed_with_volume_trend'] = df['exp_smoothed'] * df['volume_trend_ratio']
    
    # Introduce Price Momentum Component
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_momentum'] = (df['close'] - df['sma_close']) / df['sma_close']
    df['momentum_adjusted_exp_smoothed'] = df['exp_smoothed_with_volume_trend'] * df['price_momentum']
    
    # Add Volatility Component
    df['std_close'] = df['close'].rolling(window=k).std()
    df['volatility_adjusted'] = df['momentum_adjusted_exp_smoothed'] * df['std_close']
    
    # Combine Momentum and Volume-Weighted Factors
    df['final_alpha_factor'] = df['momentum_adjusted_exp_smoothed'] + df['volatility_adjusted']
    
    return df['final_alpha_factor']
```
