```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pandas.DataFrame): DataFrame with index as date and columns as market features (open, high, low, close, volume).
        
    Returns:
        pandas.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the relative range of the day
    daily_range = df['high'] - df['low']
    relative_high = (df['high'] - df['close']) / daily_range
    relative_low = (df['low'] - df['close']) / daily_range
    
    # Calculate the directional movement
    up_move = df['high'] - df['high'].shift(1)
    down_move = df['low'].shift(1) - df['low']
    pos_dm = up_move.where(up_move > down_move, 0)
    neg_dm = down_move.where(down_move > up_move, 0)
    
    # Calculate the alpha factor
    factor = (relative_high - relative_low) * (pos_dm - neg_dm) * df['volume']
    
    return factor
```
