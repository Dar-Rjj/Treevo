```python
def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Dynamic Momentum Adjustment
    high_low_diff = df['high'] - df['low']
    lookback_period = 14
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothing_factor = 0.2
    smoothed_raw_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Combine with Intraday Movement
    factor_values = smoothed_raw_momentum / intraday_movement
    
    return factor_values
```
