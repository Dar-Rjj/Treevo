```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the high and low price
    price_diff = df['high'] - df['low']
    
    # Calculate the volume weighted price
    vwp = (df['close'] * df['volume']) / df['volume'].rolling(window=5).mean()
    
    # Calculate the price momentum
    price_mom = df['close'].pct_change(periods=20)
    
    # Calculate the factor value
    factor_value = (price_diff / df['close']) + vwp + price_mom
    
    return factor_value
```
