```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, m=30, k=7):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, 1, n))
    weights /= weights.sum()
    df['exp_smoothed_volume_weighted'] = df['volume_weighted_log_return'].rolling(window=n).apply(lambda x: (x * weights[:len(x)]).sum(), raw=False)
    df['exp_smoothed_atr'] = df['atr'].rolling(window=n).apply(lambda x: (x * weights[:len(x)]).sum(), raw=False)
    
    # Incorporate Volume Trend
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_trend_ratio'] = df['volume'] / df['average_volume']
    df['volume_adjusted'] = df['exp_smoothed_volume_weighted'] * df['volume_trend_ratio']
    
    # Introduce Price Momentum Component
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_momentum'] = df['close'] - df['sma_close']
    df['price_momentum_adjusted'] = df['volume_adjusted'] * df['price_momentum']
    
    # Add Volatility Component
    df['volatility'] = df['close'].rolling(window=k).std()
    df['volatility_adjusted'] = df['price_momentum_adjusted'] * df['volatility']
    
    # Analyze Dynamic Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=n).mean()
    df['raw_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    df['smoothed_momentum'] = df['raw_momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['exp_smoothed_volume_weighted'] + 
                          df['exp_smoothed_atr'] + 
                          df['price_momentum_adjusted'] + 
                          df['volatility_adjusted'] + 
                          df['smoothed_momentum']) / 5
    
    return df['alpha_factor']
```
