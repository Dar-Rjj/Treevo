```python
def heuristics_v2(data):
    """
    Generate an alpha factor using a combination of price and volume information.
    
    Parameters:
        data (pd.DataFrame): A DataFrame with the index as date and columns as ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
        pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    daily_return = data['close'].pct_change()
    
    # Calculate the relative range
    range_factor = (data['high'] - data['low']) / data['close']
    
    # Calculate the volume factor
    volume_factor = data['volume'] / data['volume'].rolling(window=5).mean()
    
    # Combine the factors to create the final heuristic
    factor = daily_return * range_factor * volume_factor
    
    return factor
```
