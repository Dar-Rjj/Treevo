import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, alpha=0.1):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df.apply(lambda x: max(x['high'] - x['low'], 
                                              abs(x['high'] - x['close'].shift(1)), 
                                              abs(x['low'] - x['close'].shift(1))), axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_adjusted_factor'] = df['volume_weighted_sum'] / df['volume'].rolling(window=n).sum()
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Incorporate Volume Weight
    df['volume_weighted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Apply Exponential Smoothing
    weights = np.array([alpha * (1 - alpha)**i for i in range(n)])
    weights /= weights.sum()
    df['exponential_smoothed_factor'] = df['volume_weighted_atr'].rolling(window=n).apply(lambda x: (x * weights).sum(), raw=False)
    
    return df['exponential_smoothed_factor'].dropna()
