```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread

    # Synthesize Momentum Indicators
    high_to_low_range = df['high'] - df['low']
    ema_7_high_low = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_to_low_range.ewm(span=21, adjust=False).mean()
    momentum_signal = (ema_7_high_low / ema_21_high_low) * intraday_movement

    # Dynamic Momentum Adjustment
    volume_weighted_high_low = (df['high'] - df['low']) * df['volume']
    sma_14_volume_weighted = volume_weighted_high_low.rolling(window=14).mean()
    raw_momentum = volume_weighted_high_low - sma_14_volume_weighted
    smoothed_raw_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    adjusted_momentum = smoothed_raw_momentum / momentum_signal

    # Incorporate Price and Volume Factors
    price_volatility = df['high'] - df['low']
    price_direction = np.where(df['close'] > df['open'], 1, -1)
    volume_change = df['volume'].pct_change()
    volume_spike = np.where(df['volume'] > df['volume'].rolling(window=5).mean() * 2, 1, 0)
    
    # Incorporate Transaction Amounts
    amount_deviation = (df['amount'] - df['amount'].rolling(window=30).mean()) / df['amount'].rolling(window=30).std()
    transaction_value = df['amount'] / df['volume']
    
    # Construct Momentum and Reversion Indicators
    short_term_momentum = (df['close'] / df['close'].shift(5)) - 1
    num_up_days = (df['close'] > df['close'].shift(1)).rolling(window=5).sum()
    long_term_mean = df['close'].rolling(window=200).mean()
    z_score = (df['close'] - long_term_mean) / df['close'].rolling(window=200).std()
    adjusted_z_score = z_score * df['volume']

    # Combine all factors into a single alpha factor
    alpha_factor = (
        0.3 * adjusted_momentum +
        0.2 * price_volatility +
        0.1 * price_direction +
        0.1 * volume_change +
        0.1 * volume_spike +
        0.1 * amount_deviation +
        0.1 * transaction_value +
        0.1 * short_term_momentum +
        0.1 * num_up_days +
        0.1 * adjusted_z_score
    )

    return alpha_factor
```
