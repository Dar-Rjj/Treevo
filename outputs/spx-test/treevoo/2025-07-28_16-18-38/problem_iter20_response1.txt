```python
def heuristics_v2(df):
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the range of high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the price change from the previous day
    price_change = df['close'].diff()
    
    # Calculate the factor as a combination of the above components
    factor = (avg_price * price_change) / (price_range + 1e-6)
    
    return factor
```
