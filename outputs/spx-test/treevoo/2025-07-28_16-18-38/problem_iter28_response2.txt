```python
import pandas as pd

def heuristics_v2(df):
    # Daily return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Intraday volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    
    # Simple Moving Average (SMA) of Close prices over 14 days
    sma_close = df['close'].rolling(window=14).mean()
    sma_ratio = df['close'] / sma_close
    
    # Exponential Moving Average (EMA) of Close prices for more recent data emphasis
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_ratio = df['close'] / ema_close
    
    # Volume ratio
    sma_volume = df['volume'].rolling(window=14).mean()
    volume_ratio = df['volume'] / sma_volume
    
    # Price-volume momentum
    price_volume_momentum = (df['close'] - df['open']) * df['volume']
    
    # Large trade indicator
    large_trade_indicator = (df['volume'] > 2 * sma_volume).astype(int)
    
    # High-Low Difference
    high_low_diff = df['high'] - df['low']
    # SMA of High-Low Difference
    sma_high_low = high_low_diff.rolling(window=14).mean()
    # Raw Momentum
    raw_momentum = high_low_diff - sma_high_low
    # Smoothing with EMA
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    # Offset and scaling to [0, 100]
    min_value = 0
    max_value = smoothed_momentum.abs().max()
    dynamic_momentum = (smoothed_momentum - min_value) / (max_value - min_value) * 100

    # Composite factor
    composite_factor = (sma_ratio + ema_ratio + intraday_volatility + volume_ratio + price_volume_momentum + large_trade_indicator + dynamic_momentum) / 7
    
    return composite_factor
```
