```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    close_changes = df['close'].diff()
    
    # Initialize EWMA value
    ewma = close_changes.fillna(0).copy()  # Initialize with the first day's close price change
    alpha = 0.1  # Smoothing factor
    
    # Iteratively update EWMA value
    for i in range(1, len(ewma)):
        ewma.iloc[i] = ewma.iloc[i-1] * (1 - alpha) + close_changes.iloc[i] * alpha
    
    # Calculate Volume Multiplier
    volume_sum = df['volume'].rolling(window=len(df), min_periods=1).sum()
    volume_multiplier = df['volume'] / volume_sum
    
    # Volume Weight the EWMA of Close Price Changes
    vwema = ewma * volume_multiplier
    
    return vwema
```
