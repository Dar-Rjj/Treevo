```python
import pandas as pd

def heuristics_v2(df, n_days=14):
    # Calculate True Range
    df['Previous Close'] = df['close'].shift(1)
    df['High-Low'] = df['high'] - df['low']
    df['High-PrevClose'] = (df['high'] - df['Previous Close']).abs()
    df['Low-PrevClose'] = (df['low'] - df['Previous Close']).abs()
    df['True Range'] = df[['High-Low', 'High-PrevClose', 'Low-PrevClose']].max(axis=1)
    
    # Compute Average True Range (ATR)
    df['ATR'] = df['True Range'].rolling(window=n_days).mean()
    
    # Derive Momentum Indicator
    df['Momentum'] = df['close'] - df['close'].shift(n_days)
    
    # Incorporate Volume Weight
    df['Volume Weighted Momentum'] = df['Momentum'] * df['volume']
    
    # Generate the factor
    alpha_factor = df['Volume Weighted Momentum'] / df['ATR']
    
    return alpha_factor
```
