```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Volume-Weighted High-to-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Calculate 7-day and 21-day Exponential Moving Averages (EMA)
    ema_7 = volume_weighted_high_low_range.ewm(span=7, adjust=False).mean()
    ema_21 = volume_weighted_high_low_range.ewm(span=21, adjust=False).mean()
    
    # Compute the Ratio of 7-day EMA to 21-day EMA
    momentum_ratio = ema_7 / ema_21
    
    # Compute Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Amount
    amount_adjusted_return = daily_return * df['amount']
    
    # Calculate 5-day Simple Moving Average (SMA) of Amount-Adjusted Return
    sma_5 = amount_adjusted_return.rolling(window=5).mean()
    
    # Combine Momentum Ratio and SMA of Amount-Adjusted Return
    factor = momentum_ratio * sma_5
    
    return factor
```
