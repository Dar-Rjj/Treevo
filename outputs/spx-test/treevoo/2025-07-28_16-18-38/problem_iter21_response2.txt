```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High and Low
    lookback_period = 14
    df['Volume_Weighted_High'] = (df['high'] * df['volume']).rolling(window=lookback_period).sum()
    df['Volume_Weighted_Low'] = (df['low'] * df['volume']).rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    df['Volume_Weighted_High_Low_Diff'] = (df['Volume_Weighted_High'] - df['Volume_Weighted_Low']) / total_volume

    # Compute Dynamic Momentum
    sma_lookback = 14
    df['High_Low_Diff'] = df['high'] - df['low']
    df['SMA_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=sma_lookback).mean()
    df['Raw_Momentum'] = df['High_Low_Diff'] - df['SMA_High_Low_Diff']
    smoothing_factor = 0.2
    df['Smoothed_Momentum'] = df['Raw_Momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Combine Volume-Weighted High-Low Range and Dynamic Momentum
    scaling_factor = 0.5
    df['Combined_Value'] = (df['Volume_Weighted_High_Low_Diff'] + df['Smoothed_Momentum']) * scaling_factor

    # Transform to Bounded Range Without Normalization
    offset = 0
    max_possible_value = (df['Volume_Weighted_High_Low_Diff'].max() + df['Smoothed_Momentum'].max()) * scaling_factor
    df['Bounded_Combined_Value'] = (df['Combined_Value'] - df['Combined_Value'].min()) / (max_possible_value - df['Combined_Value'].min()) * 100 + offset

    # Compute EMA Components
    ema_7 = df['Bounded_Combined_Value'].ewm(span=7, adjust=False).mean()
    ema_21 = df['Bounded_Combined_Value'].ewm(span=21, adjust=False).mean()

    # Compute Ratio of EMAs
    df['EMA_Ratio'] = ema_7 / ema_21

    # Compute Price Oscillator
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    df['Price_Oscillator'] = close_ema_5 - close_ema_14

    # Compute Amount-Adjusted Return
    df['Daily_Return'] = df['close'].pct_change()
    df['Amount_Adjusted_Return'] = df['Daily_Return'] * df['amount']
    df['SMA_Amount_Adjusted_Return'] = df['Amount_Adjusted_Return'].rolling(window=5).mean()

    # Final Alpha Factor
    df['Alpha_Factor'] = (df['EMA_Ratio'] * df['SMA_Amount_Adjusted_Return']) + df['Price_Oscillator']

    return df['Alpha_Factor']
```
