```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=20, k=5):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    df['volume_sum'] = df['volume'].rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_adjusted_factor'] = df['volume_weighted_sum'] / df['volume_sum']
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Incorporate Volume Weight
    df['volume_adjusted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, n))
    weights /= weights.sum()
    df['weighted_volume_adjusted_atr'] = df['volume_adjusted_atr'].rolling(window=n).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Incorporate Volume Trend
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_trend'] = df['volume'] / df['avg_volume']
    df['volume_trend_adjusted'] = df['weighted_volume_adjusted_atr'] * df['volume_trend']
    
    # Introduce Price Momentum Component
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_momentum'] = (df['close'] - df['sma_close']) / df['sma_close']
    df['momentum_adjusted'] = df['volume_trend_adjusted'] * df['price_momentum']
    
    # Add Volatility Component
    df['volatility'] = df['close'].rolling(window=k).std()
    df['final_factor'] = df['momentum_adjusted'] * df['volatility']
    
    return df['final_factor']
```
