```python
import pandas as pd

def heuristics_v2(df):
    # Define the lookback period and smoothing factor
    lookback_period = 14
    smoothing_factor = 0.2
    scaling_factor = 0.5
    
    # Compute the high-low difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Compute the simple moving average (SMA) of the high-low difference
    df['SMA_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=lookback_period).mean()
    
    # Calculate raw momentum
    df['Raw_Momentum'] = df['High_Low_Diff'] - df['SMA_High_Low_Diff']
    
    # Apply exponential weighting for smoothing
    df['Smoothed_Momentum'] = df['Raw_Momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['Close_Volume'] = df['close'] * df['volume']
    df['Sum_Close_Volume'] = df['Close_Volume'].rolling(window=lookback_period).sum()
    df['Total_Volume'] = df['volume'].rolling(window=lookback_period).sum()
    df['VWAP'] = df['Sum_Close_Volume'] / df['Total_Volume']
    
    # Combine dynamic momentum and VWAP
    df['Factor'] = scaling_factor * (df['Smoothed_Momentum'] * df['VWAP'])
    
    # Return the factor as a pandas Series
    return df['Factor'].dropna()
```
