import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the close price over the last 10 days
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the difference between the current close and its 10-day mean
    df['close_diff_10d_mean'] = df['close'] - df['close_10d_mean']
    
    # Calculate the rolling standard deviation of the close price over the last 10 days
    df['close_10d_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the z-score of the close price
    df['close_zscore'] = df['close_diff_10d_mean'] / df['close_10d_std']
    
    # Calculate the average volume over the last 10 days
    df['volume_10d_mean'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the ratio of the current volume to the 10-day average volume
    df['volume_ratio'] = df['volume'] / df['volume_10d_mean']
    
    # Calculate the alpha factor
    df['alpha_factor'] = df['close_zscore'] * df['volume_ratio']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
