```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Change
    df['close_change'] = df['close'] - df['close'].shift(1)
    
    # Initialize and Update VWEMA of Close Price Changes
    alpha = 0.2  # Smoothing factor
    df['vwap_close_change'] = df['close_change'] * df['volume']
    df['vwap_close_change_sum'] = df['vwap_close_change'].rolling(window=21).sum()
    df['volume_sum'] = df['volume'].rolling(window=21).sum()
    df['vwema'] = df['vwap_close_change'] / df['volume_sum']
    df['vwema'] = df['vwema'].ewm(alpha=alpha, adjust=False).mean()

    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Compute Momentum Using Volume-Weighted High-to-Low Range
    df['high_low_7day_ema'] = df['high_low_range'].ewm(span=7, adjust=False).mean()
    df['high_low_21day_ema'] = df['high_low_range'].ewm(span=21, adjust=False).mean()
    df['momentum_ratio'] = df['high_low_7day_ema'] / df['high_low_21day_ema']

    # Introduce Open Price Influence
    df['open_close_change'] = df['open'] - df['close']
    df['open_momentum_7day_ema'] = df['open_close_change'].ewm(span=7, adjust=False).mean()
    df['open_momentum_21day_ema'] = df['open_close_change'].ewm(span=21, adjust=False).mean()
    df['open_momentum_ratio'] = df['open_momentum_7day_ema'] / df['open_momentum_21day_ema']

    # Integrate VWEMA, Momentum, and Open Price Influence
    df['combined_alpha_factor'] = (df['vwema'] 
                                   * df['momentum_ratio'] 
                                   * df['open_momentum_ratio'])

    return df['combined_alpha_factor']
```
