```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Volume-Weighted High-to-Low Return
    volume_weighted_hl_return = high_to_low_return * df['volume']
    
    # Incorporate Previous Day's Close-to-Open Movement
    df['close_prev_day'] = df['close'].shift(1)
    close_to_open_return = (df['open'] - df['close_prev_day']) / df['close_prev_day']
    combined_return = volume_weighted_hl_return + close_to_open_return
    
    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    lookback_period = 14
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor).mean()
    
    # Introduce Relative Strength
    lookback_period_close = 10
    n_day_close_change = df['close'].pct_change(periods=lookback_period_close)
    sma_n_day_close_change = n_day_close_change.rolling(window=lookback_period_close).mean()
    relative_strength = n_day_close_change - sma_n_day_close_change
    
    # Combine Volume-Weighted High-to-Low Return, Dynamic Momentum, and Relative Strength
    combined_factor = combined_return + smoothed_momentum + relative_strength
    
    # Calculate Cumulative Effect Over N Days
    n_days = 5
    cumulative_factor = combined_factor.rolling(window=n_days).sum()
    
    return cumulative_factor
```
