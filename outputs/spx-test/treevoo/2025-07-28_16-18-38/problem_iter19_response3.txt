```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 5-day and 20-day moving averages of close prices
    df['5day_MA'] = df['close'].rolling(window=5).mean()
    df['20day_MA'] = df['close'].rolling(window=20).mean()
    
    # Calculate trend indicator
    df['trend_indicator'] = df['5day_MA'] - df['20day_MA']
    
    # Calculate daily volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate average 5-day volatility
    df['average_5day_volatility'] = df['daily_volatility'].rolling(window=5).mean()
    
    # Calculate daily volume change
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate average 5-day volume change
    df['average_5day_volume_change'] = df['daily_volume_change'].rolling(window=5).mean()
    
    # Calculate daily amount
    df['daily_amount'] = df['close'] * df['volume']
    
    # Calculate average 5-day daily amount
    df['average_5day_daily_amount'] = df['daily_amount'].rolling(window=5).mean()
    
    # Calculate daily amount change
    df['daily_amount_change'] = (df['daily_amount'] - df['daily_amount'].shift(1)) / df['daily_amount'].shift(1)
    
    # Construct composite factors
    df['price_volume_momentum'] = (df['daily_return'] + df['daily_volatility']) * df['daily_volume_change']
    df['trend_confirmation'] = df['trend_indicator'] * df['daily_amount_change']
    df['sentiment_volatility'] = (df['average_5day_volatility'] + df['average_5day_daily_amount']) * df['trend_indicator']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = df['price_volume_momentum'] + df['trend_confirmation'] + df['sentiment_volatility']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
