```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Dynamic Momentum Adjustment
    lookback_period = 14
    sma_lookback = high_low_spread.rolling(window=lookback_period).mean()
    raw_momentum = high_low_spread - sma_lookback
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor).mean()
    
    # Combine with Intraday Movement
    factor_values = smoothed_momentum / intraday_movement
    
    return factor_values
```
