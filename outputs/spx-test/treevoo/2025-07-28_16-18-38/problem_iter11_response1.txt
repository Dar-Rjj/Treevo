import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-to-Low Return
    high_low_return = (df['high'] - df['low']) / df['low']
    volume_weighted_high_low_return = high_low_return * df['volume']

    # Incorporate Previous Day's Close-to-Open Movement
    close_to_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    combined_short_term_momentum = volume_weighted_high_low_return + close_to_open_return

    # Calculate Daily Close Price Changes
    daily_close_changes = df['close'].diff()

    # Calculate Volume-Weighted Exponentially Weighted Moving Average (VW-EWMA) of Close Price Changes
    alpha = 0.1
    vw_ewma = [daily_close_changes[0]]
    for i in range(1, len(daily_close_changes)):
        prev_vw_ewma = vw_ewma[-1]
        current_change = daily_close_changes[i]
        volume_multiplier = df['volume'][i] / df['volume'].rolling(window=14).sum()[i]
        vw_ewma.append((1 - alpha) * prev_vw_ewma + alpha * current_change * volume_multiplier)
    vw_ewma = pd.Series(vw_ewma, index=daily_close_changes.index)

    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low = high_low_diff.rolling(window=14).mean()
    dynamic_momentum = (high_low_diff - sma_high_low).ewm(alpha=0.2).mean()

    # Calculate Price Trend Strength
    daily_range = df['high'] - df['low']
    atr = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    price_trend_strength = daily_range / atr

    # Integrate Price Trend Strength with Dynamic Momentum
    adjusted_dynamic_momentum = dynamic_momentum * price_trend_strength

    # Combine Short-Term, Long-Term, and Adjusted Dynamic Momentum
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_difference = df['close'] - ten_day_avg_close
    final_alpha_factor = (adjusted_close_difference / combined_short_term_momentum) + adjusted_dynamic_momentum + vw_ewma

    return final_alpha_factor
