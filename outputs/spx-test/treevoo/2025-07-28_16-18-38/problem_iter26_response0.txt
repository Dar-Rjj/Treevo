import pandas as pd

def heuristics_v2(df):
    # Thought 1.1: Enhanced Intraday Movement and Price Direction
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    combined_intraday_indicator = high_low_spread + close_open_spread
    price_direction = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Thought 1.2: Momentum Indicators with Volume Influence and Trend Information
    ema_high_low_7 = high_low_spread.ewm(span=7, adjust=False).mean()
    ema_high_low_21 = high_low_spread.ewm(span=21, adjust=False).mean()
    momentum_ratio = ema_high_low_7 / ema_high_low_21
    adjusted_momentum_factor = momentum_ratio * high_low_spread
    volume_weighted_high_low_diff = high_low_spread * df['volume']
    sma_volume_weighted_high_low_14 = volume_weighted_high_low_diff.rolling(window=14).mean()
    volume_weighted_diff = volume_weighted_high_low_diff - sma_volume_weighted_high_low_14
    smoothed_momentum = volume_weighted_diff.ewm(alpha=0.2, adjust=False).mean()

    sma_close = df['close'].rolling(window=14).mean()
    sma_ratio = df['close'] / sma_close
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_ratio = df['close'] / ema_close
    wma_close = (df['close'] * (15 - df['close'].rolling(window=14).rank())).rolling(window=14).sum() / 105
    wma_ratio = df['close'] / wma_close

    # Thought 1.3: Volume and Amount Analysis for Alpha Generation
    volume_change = df['volume'].pct_change()
    volume_spike = (df['volume'] > df['volume'].rolling(window=30).mean() * 1.5).astype(int)
    transaction_value = df['amount'] / df['volume']
    avg_amount_30 = df['amount'].rolling(window=30).mean()
    amount_deviation = df['amount'] - avg_amount_30
    volume_weighted_momentum = smoothed_momentum * volume_spike
    volume_ratio = df['volume'] / df['volume'].rolling(window=14).mean()
    price_volume_momentum = (df['close'] - df['open']) * df['volume']
    large_trade_indicator = (df['volume'] > df['volume'].rolling(window=30).mean() * 2).astype(int)
    money_flow = (df['close'] - df['open']) * df['amount']

    # Thought 1.4: Advanced Intraday and Momentum Integration
    intraday_volatility = high_low_spread.rolling(window=7).std()
    momentum_oscillator = (df['close'] - sma_close) / sma_close
    intraday_momentum_score = intraday_volatility * momentum_oscillator

    # Final Alpha Factor
    composite_factor = (
        sma_ratio + 
        ema_ratio + 
        wma_ratio + 
        intraday_volatility + 
        high_low_spread / df['close'] + 
        volume_ratio + 
        price_volume_momentum + 
        large_trade_indicator + 
        money_flow
    )
    final_alpha = (
        composite_factor * 
        price_direction * 
        volume_spike * 
        (1 + volume_change) * 
        intraday_momentum_score
    )

    return final_alpha
