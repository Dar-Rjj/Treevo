```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Volume-Weighted High-to-Low Return
    volume_weighted_momentum = high_to_low_return * df['volume']
    
    # Incorporate Previous Day's Close-to-Open Movement
    prev_close_to_open_return = (df['open'] / df['close'].shift(1) - 1)
    combined_momentum = volume_weighted_momentum + prev_close_to_open_return
    
    # Long-Term Price Momentum Adjustment
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_difference = df['close'] - ten_day_avg_close
    
    # Combine with Short-Term Momentum
    final_alpha_factor = adjusted_close_difference / combined_momentum
    
    return final_alpha_factor
```
