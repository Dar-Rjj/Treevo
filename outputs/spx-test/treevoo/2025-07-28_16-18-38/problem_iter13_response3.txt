```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic High-to-Low Range
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    dynamic_high_low_range = high_low_diff - sma_high_low_diff
    smoothed_high_low_range = dynamic_high_low_range.ewm(alpha=0.2).mean()

    # Compute Volume-Weighted High-to-Low Range
    volume_weighted_hl_range = smoothed_high_low_range * df['volume']

    # Compute Momentum
    ema_7d = volume_weighted_hl_range.ewm(span=7, adjust=False).mean()
    ema_21d = volume_weighted_hl_range.ewm(span=21, adjust=False).mean()
    momentum = ema_7d / ema_21d

    # Compute Amount-Adjusted Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    amount_adjusted_return = daily_return * df['amount']
    sma_amount_adjusted_return = amount_adjusted_return.rolling(window=5).mean()

    # Combine the factors
    alpha_factor = momentum + sma_amount_adjusted_return

    return alpha_factor
```
