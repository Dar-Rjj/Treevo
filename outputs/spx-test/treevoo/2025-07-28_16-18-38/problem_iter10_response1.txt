```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    df['close_change'] = df['close'].diff()
    
    # Calculate Exponentially Weighted Moving Average (EWMA) of Close Price Changes
    alpha = 0.1  # Smoothing factor
    ewma = df['close_change'].ewm(alpha=alpha, adjust=False).mean()
    
    # Volume Weight the EWMA of Close Price Changes
    volume_multiplier = df['volume'] / df['volume'].rolling(window=14).sum()
    volume_weighted_ewma = ewma * volume_multiplier
    
    # Integrate High-Low Price Information
    high_low_diff = df['high'] - df['low']
    sma_high_low = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    # Combine Adjusted EWMA and High-Low Momentum
    combined_momentum = volume_weighted_ewma * smoothed_momentum
    final_smoothed_momentum = combined_momentum.ewm(alpha=0.3, adjust=False).mean()
    
    # Calculate Price Momentum
    average_daily_return = df['close'].pct_change()
    price_momentum = average_daily_return.rolling(window=20).mean()
    
    # Calculate Volume Momentum
    average_daily_volume_change = df['volume'].pct_change()
    volume_momentum = average_daily_volume_change.rolling(window=20).mean()
    
    # Generate Combined Alpha Factor
    alpha_factor = final_smoothed_momentum * price_momentum
    alpha_factor = alpha_factor * (1 if volume_momentum[-1] > 0 else -1)
    
    return alpha_factor
```
