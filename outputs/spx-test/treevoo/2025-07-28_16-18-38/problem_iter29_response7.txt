import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    lookback_high_low = 14
    ema_short_span = 7
    ema_long_span = 21
    price_oscillator_ema_short_span = 5
    price_oscillator_ema_long_span = 14
    return_sma_span = 5
    
    # Calculate Hybrid High-to-Low Range
    high = df['high']
    low = df['low']
    
    # Compute Volume-Weighted High and Low
    vol_weighted_high = (df['high'] * df['volume']).rolling(window=lookback_high_low).sum()
    vol_weighted_low = (df['low'] * df['volume']).rolling(window=lookback_high_low).sum()
    total_volume = df['volume'].rolling(window=lookback_high_low).sum()
    vol_weighted_high_low_diff = (vol_weighted_high - vol_weighted_low) / total_volume
    
    # Compute Dynamic Momentum
    high_low_diff = high - low
    sma_high_low_diff = high_low_diff.rolling(window=lookback_high_low).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    dynamic_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    # Combine Volume-Weighted High-Low Range and Dynamic Momentum
    combined_value = 0.5 * (vol_weighted_high_low_diff + dynamic_momentum)
    
    # Transform to Bounded Range Without Normalization
    offset = 0
    max_possible_value = 0.5 * (high.max() - low.min()) + dynamic_momentum.abs().max()
    bounded_combined_value = (combined_value + offset) * (100 / max_possible_value)
    
    # Compute EMA Components
    ema_short = bounded_combined_value.ewm(span=ema_short_span, adjust=False).mean()
    ema_long = bounded_combined_value.ewm(span=ema_long_span, adjust=False).mean()
    
    # Compute Ratio of EMAs
    ema_ratio = ema_short / ema_long
    
    # Compute Price Oscillator
    close_ema_short = df['close'].ewm(span=price_oscillator_ema_short_span, adjust=False).mean()
    close_ema_long = df['close'].ewm(span=price_ocoder_ema_long_span, adjust=False).mean()
    price_oscillator = close_ema_short - close_ema_long
    
    # Compute Amount-Adjusted Return
    daily_return = df['close'].pct_change()
    amount_adjusted_return = daily_return * df['amount']
    amount_adjusted_return_sma = amount_adjusted_return.rolling(window=return_sma_span).mean()
    
    # Final Alpha Factor
    alpha_factor = ema_ratio * amount_adjusted_return_sma + price_oscillator
    
    return alpha_factor
