```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=20):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'], df['close'].shift(1)) - min(x['low'], df['close'].shift(1)), axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_moving_sum'] = (df['volume_weighted_log_return'].rolling(window=window).sum() 
                                        / df['volume'].rolling(window=window).sum())
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=window).mean()
    
    # Incorporate Volume-Weighted Factor
    df['volume_weighted_factor'] = df['volume_weighted_moving_sum'] * df['atr']
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, window))
    weights /= weights.sum()  # Normalize to sum to 1
    
    df['exp_smoothed_volume_weighted_factor'] = (df['volume_weighted_factor']
                                                 .rolling(window=window)
                                                 .apply(lambda x: (x * weights).sum(), raw=True))
    
    return df['exp_smoothed_volume_weighted_factor']
```
