```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Dynamic Momentum
    # Calculate Moving Average of High-Low Difference
    lookback_period = 14
    sma_high_low_diff = high_low_range.rolling(window=lookback_period).mean()
    raw_momentum = high_low_range - sma_high_low_diff
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Compute Momentum Components
    ema_7 = high_low_range.ewm(span=7, adjust=False).mean()
    ema_21 = high_low_range.ewm(span=21, adjust=False).mean()
    
    # Combine Momentum Signals
    ratio_ema = ema_7 / ema_21
    final_momentum_signal = ratio_ema * smoothed_momentum
    
    return final_momentum_signal
```
