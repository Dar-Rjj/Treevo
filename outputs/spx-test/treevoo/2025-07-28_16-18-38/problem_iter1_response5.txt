```python
def heuristics_v2(df):
    # Select High and Low Prices
    high = df['high']
    low = df['low']
    
    # Compute High-Low Difference
    high_low_diff = high - low
    
    # Choose Lookback Period (e.g., 14 days) and Compute Simple Moving Average (SMA)
    lookback_period = 14
    sma_high_low = high_low_diff.rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum
    raw_momentum = high_low_diff - sma_high_low
    
    # Apply Smoothing
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Add Offset to Shift to Positive Range
    min_value = 0
    max_possible_value = smoothed_momentum.max() - smoothed_momentum.min()
    offset = -smoothed_momentum.min()
    shifted_momentum = smoothed_momentum + offset
    
    # Scale to Desired Range (e.g., 0 to 100)
    scaled_momentum = (shifted_momentum / max_possible_value) * 100
    
    # Clip to Ensure Bounded Range
    bounded_momentum = scaled_momentum.clip(lower=min_value, upper=100)
    
    return bounded_momentum
```
