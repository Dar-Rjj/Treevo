```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].diff()
    
    # Calculate Relative Strength
    positive_returns = df['daily_return'].rolling(window=14).apply(lambda x: x[x>0].sum(), raw=False)
    negative_returns = df['daily_return'].rolling(window=14).apply(lambda x: abs(x[x<0].sum()), raw=False)
    relative_strength_ratio = positive_returns / negative_returns
    
    # Calculate RSVMI
    rsvmi = 100 + 100 * np.log(relative_strength_ratio)
    
    # Volume Momentum
    volume_change = df['volume'].diff()
    smoothed_volume_change = volume_change.rolling(window=14).mean()
    rsvmi_with_momentum = rsvmi * smoothed_volume_change
    
    return rsvmi_with_momentum
```
