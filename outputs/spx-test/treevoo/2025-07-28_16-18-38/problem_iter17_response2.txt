```python
import pandas as pd

def heuristics_v2(df):
    # 1.1.1 Calculate the Range (High - Low) as a measure of volatility
    df['range'] = df['high'] - df['low']
    
    # 1.1.2 Introduce a Factor based on the relationship between Close and Open
    df['close_open_diff'] = df['close'] - df['open']
    df['direction'] = (df['close'] > df['open']).astype(int)
    
    # 1.2.1 Create a Factor that measures changes in volume relative to the previous day
    df['volume_change_pct'] = df['volume'].pct_change()
    
    # 1.2.2 Assess the impact of volume spikes on price
    df['volume_spike'] = (df['volume'] > df['volume'].rolling(window=5).mean() * 2).astype(int)
    
    # 1.3.1 Develop a Factor Linking Amount Traded to Stock Performance
    avg_amount = df['amount'].rolling(window=30).mean()
    df['amount_deviation'] = df['amount'] - avg_amount
    
    # 1.3.2 Combine Volume and Amount Data
    df['value_ratio'] = df['amount'] / df['volume']
    
    # 2.1.1 Design a simple momentum factor using the past 5-day returns
    df['5_day_return'] = df['close'].pct_change(5)
    
    # 2.1.2 Investigate the consistency of momentum
    df['up_days'] = (df['close'] > df['close'].shift(1)).rolling(window=5).sum()
    
    # 2.2.1 Use the long-term moving average (e.g., 200 days) as a baseline
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['distance_from_200_ma'] = df['close'] - df['200_day_ma']
    
    # 2.2.2 Consider the rate of deviation from the mean
    df['z_score_200'] = (df['close'] - df['200_day_ma']) / df['close'].rolling(window=200).std()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['range'] + df['close_open_diff'] + df['volume_change_pct'] + 
                          df['volume_spike'] + df['amount_deviation'] + df['value_ratio'] + 
                          df['5_day_return'] + df['up_days'] + df['distance_from_200_ma'] + df['z_score_200'])
    
    return df['alpha_factor']
```
