```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()

    # Transform to Bounded Range Without Normalization
    offset = 0  # Set Minimum Value
    max_possible_value = (df['high'].max() - df['low'].min()) * 1.5  # Estimated Maximum Possible Value
    bounded_momentum = (smoothed_momentum + offset) / max_possible_value * 100

    # Introduce Signal Filtering
    volatility = high_low_diff.rolling(window=20).std()
    adjusted_momentum = smoothed_momentum / (volatility + 1e-6)  # Ensure Non-Zero Volatility
    volume_weighted_high_low = (high_low_diff * df['volume']).rolling(window=30).mean()
    momentum_adjusted = adjusted_momentum * (volume_weighted_high_low / volume_weighted_high_low.max())

    # Use Open, High, Low, Close Prices to Identify Patterns and Trends
    daily_return = (df['close'] - df['open']) / df['open']
    intraday_volatility = (df['high'] - df['low']) / df['open']

    # Explore Moving Averages to Capture Trend Information
    sma_close = df['close'].rolling(window=20).mean()
    sma_ratio = df['close'] / sma_close
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_ratio = df['close'] / ema_close

    # Utilize Volume and Amount Data to Assess Market Liquidity and Interest
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume
    price_volume_momentum = (df['close'] - df['open']) * df['volume']
    large_trade_indicator = (df['volume'] > 2 * avg_volume).astype(int)

    # Combine all factors into a single alpha factor
    alpha_factor = (
        bounded_momentum +
        momentum_adjusted +
        daily_return +
        intraday_volatility +
        sma_ratio +
        ema_ratio +
        volume_ratio +
        price_volume_momentum +
        large_trade_indicator
    )

    return alpha_factor
```
