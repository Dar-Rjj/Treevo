```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the close prices over a 5-day window
    df['close_5d_mean'] = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between the current close and its 5-day mean
    df['close_diff_5d_mean'] = df['close'] - df['close_5d_mean']
    
    # Calculate the standard deviation of the close prices over a 5-day window
    df['close_5d_std'] = df['close'].rolling(window=5).std()
    
    # Calculate the z-score like value (difference divided by std)
    df['close_z_score_5d'] = df['close_diff_5d_mean'] / df['close_5d_std']
    
    # Calculate the average true range (ATR) over a 14-day window
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the factor as the ratio of the z-score to the ATR
    df['factor'] = df['close_z_score_5d'] / df['atr_14']
    
    return df['factor'].dropna()
```
