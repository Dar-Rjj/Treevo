```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)

    # Calculate Relative Strength
    positive_returns = df['daily_return'].rolling(window=14).apply(lambda x: x[x>0].sum(), raw=False)
    negative_returns = df['daily_return'].rolling(window=14).apply(lambda x: np.abs(x[x<0].sum()), raw=False)
    relative_strength_ratio = positive_returns / negative_returns

    # Calculate RSVMI
    rsvmi = 100 + (100 * np.log(relative_strength_ratio))

    # Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    smoothed_volume_change = volume_change.rolling(window=14).mean()

    # Integrate Price and Volume
    average_close_price_14d = df['close'].rolling(window=14).mean()
    price_volume_index = rsvmi * smoothed_volume_change + average_close_price_14d

    average_volume_14d = df['volume'].rolling(window=14).mean()
    
    # Final Alpha Factor
    alpha_factor = price_volume_index - average_volume_14d
    
    return alpha_factor
```
