def heuristics_v2(data):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    - data (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by date.
    
    Returns:
    - pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the true range
    tr = data[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    
    # Calculate the money flow multiplier
    mfm = ((data['close'] - data['low']) - (data['high'] - data['close'])) / (data['high'] - data['low'])
    
    # Calculate the money flow volume
    mf_volume = mfm * data['volume']
    
    # Calculate the 14-day sum of money flow volume
    mf_positive = mf_volume.where(mfm > 0, 0).rolling(window=14).sum()
    mf_negative = mf_volume.where(mfm < 0, 0).rolling(window=14).sum()
    
    # Calculate the money ratio
    mr = mf_positive / mf_negative.abs()
    
    # Combine true range and money ratio to create the alpha factor
    factor = tr * mr
    
    return factor
