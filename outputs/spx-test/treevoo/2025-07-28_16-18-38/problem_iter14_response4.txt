import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High-to-Low Return
    high_low_return = (df['high'] - df['low']) / df['low']
    volume_weighted_high_low_return = high_low_return * df['volume']

    # Incorporate Previous Day's Close-to-Open Movement
    prev_close_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    intermediate_alpha_factor = volume_weighted_high_low_return + prev_close_open_return

    # Calculate Daily Close Price Changes
    close_price_change = df['close'] - df['close'].shift(1)

    # Calculate Volume-Weighted Exponentially Weighted Moving Average (VW-EWMA) of Close Price Changes
    alpha = 0.3
    vw_ewma = [close_price_change[0]]
    for i in range(1, len(close_price_change)):
        vw_ewma_value = (1 - alpha) * vw_ewma[-1] + alpha * close_price_change[i]
        volume_multiplier = df['volume'][i] / df['volume'].rolling(window=14).sum()[i]
        vw_ewma.append(vw_ewma_value * volume_multiplier)
    vw_ewma = pd.Series(vw_ewma, index=df.index)

    # Calculate Dynamic Volume-Weighted Momentum
    high_low_diff = df['high'] - df['low']
    raw_momentum = high_low_diff - high_low_diff.rolling(window=14).mean()
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor).mean()
    dynamic_volume_weighted_momentum = smoothed_momentum.rolling(window=14).sum()

    # Combine with Volume-Weighted Close Price
    vwcp = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    combined_momentum = 0.5 * dynamic_volume_weighted_momentum * vwcp

    # Calculate Price Trend Strength
    daily_range = df['high'] - df['low']
    atr = daily_range.rolling(window=14).mean()
    price_trend_strength = daily_range / atr

    # Integrate Price Trend Strength with Dynamic Momentum
    adjusted_dynamic_momentum = combined_momentum * price_trend_strength

    # Combine Short-Term, Long-Term, and Adjusted Dynamic Momentum
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_difference = df['close'] - ten_day_avg_close
    final_alpha_factor = adjusted_close_difference / (intermediate_alpha_factor + adjusted_dynamic_momentum)

    return final_alpha_factor
