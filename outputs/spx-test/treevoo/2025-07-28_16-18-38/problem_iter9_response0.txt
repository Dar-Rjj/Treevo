```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    df['tr'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'], x.shift(1)['close']) - min(x['low'], x.shift(1)['close']), axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_returns'] = (df['log_return'] * df['volume']).rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_adjusted_factor'] = df['volume_weighted_log_returns'] / df['volume'].rolling(window=n).sum()
    
    # Compute Average True Range (ATR)
    df['atr'] = df['tr'].rolling(window=n).mean()
    
    # Incorporate Volume Weight
    df['volume_weighted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, n))
    weights /= weights.sum()  # Normalize the weights to sum to 1
    df['exponentially_smoothed_factor'] = df['volume_weighted_atr'].rolling(window=n).apply(
        lambda x: (x * weights).sum(), raw=False)
    
    # Incorporate Volume Trend
    df['volume_trend'] = df['volume'] / df['volume'].rolling(window=n).mean()
    df['final_factor'] = df['exponentially_smoothed_factor'] * df['volume_trend']
    
    return df['final_factor']
```
