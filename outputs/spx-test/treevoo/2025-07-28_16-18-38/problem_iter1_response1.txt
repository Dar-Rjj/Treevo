```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average daily return
    df['avg_daily_return'] = df['close'] - df['close'].shift(1)
    
    # Smooth the average daily return to get price momentum
    df['price_momentum'] = df['avg_daily_return'].rolling(window=20).mean()
    
    # Calculate the average daily volume change
    df['avg_daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Smooth the average daily volume change to get volume momentum
    df['volume_momentum'] = df['avg_daily_volume_change'].rolling(window=20).mean()
    
    # Combine price momentum and volume momentum
    df['combined_momentum'] = df['price_momentum'] * df['volume_momentum']
    
    # Adjust for the sign of the volume momentum
    df['combined_momentum_adjusted'] = df['combined_momentum'].where(df['volume_momentum'] > 0, -df['combined_momentum'])
    
    return df['combined_momentum_adjusted'].dropna()
```
