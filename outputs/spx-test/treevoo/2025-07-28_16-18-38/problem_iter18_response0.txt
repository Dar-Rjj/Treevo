```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low and Close-Open Spreads
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread

    # Compute 7-day and 21-day EMA of High-Low Spread
    ema_7_high_low = intraday_movement.ewm(span=7, adjust=False).mean()
    ema_21_high_low = intraday_movement.ewm(span=21, adjust=False).mean()

    # Calculate Raw Momentum
    raw_momentum = intraday_movement.rolling(window=14).mean()
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()

    # Combine Momentum Signals
    momentum_ratio = ema_7_high_low / ema_21_high_low
    combined_momentum = momentum_ratio * smoothed_momentum

    # Accumulate the Combined Value Over N Days (e.g., 21 days)
    accumulated_momentum = combined_momentum.rolling(window=21).sum()

    # Calculate Volume-Weighted Close Price (VWCP)
    vwcp = (df['close'] * df['volume']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()

    # Combine Dynamic and Intraday Momentum with VWCP
    scaled_momentum = combined_momentum * vwcp * 0.5

    # Long-Term Price Momentum Adjustment
    twenty_day_avg_close = df['close'].rolling(window=20).mean()
    adjusted_close_difference = df['close'] - twenty_day_avg_close

    # Final Alpha Factor
    final_alpha_factor = adjusted_close_difference / accumulated_momentum

    return final_alpha_factor
```
