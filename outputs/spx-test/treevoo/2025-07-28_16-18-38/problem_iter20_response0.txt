```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the average of the high and low prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average high/low
    df['avg_high_low_change'] = df['avg_high_low'].pct_change()
    
    # Calculate the ratio of close price to the average of high and low
    df['close_to_avg_ratio'] = df['close'] / df['avg_high_low']
    
    # Calculate the volume weighted close price
    df['vol_weighted_close'] = df['close'] * df['volume']
    
    # Calculate the factor: (close_to_avg_ratio - avg_high_low_change) * vol_weighted_close
    alpha_factor = (df['close_to_avg_ratio'] - df['avg_high_low_change']) * df['vol_weighted_close']
    
    return alpha_factor.dropna()
```
