import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['open']
    
    # Combine Intraday Return and High-Low Range
    df['combined_intraday'] = df['intraday_return'] * df['high_low_range']
    
    # Adjust for Volume: Calculate Volume Impact
    df['volume_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_impact'] = df['volume'] / df['volume_ma']
    df['adjusted_intraday_momentum'] = df['combined_intraday'] * df['volume_impact']
    
    # Integrate Price and Volume Momentum
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['price_momentum_smoothed'] = df['price_momentum'].rolling(window=10).mean()
    
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_momentum_smoothed'] = df['volume_change'].rolling(window=10).mean()
    
    # Adjust Combined Intraday Momentum by Integrated Momentum
    df['momentum_ratio'] = df['price_momentum_smoothed'] / df['volume_momentum_smoothed']
    df['integrated_momentum'] = df['adjusted_intraday_momentum'] * df['momentum_ratio']
    
    # Adjust for the sign of the Volume Momentum
    df['integrated_momentum'] = df['integrated_momentum'] * df['volume_momentum_smoothed'].apply(lambda x: 1 if x > 0 else -1)
