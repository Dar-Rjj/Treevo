import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread and Close-Open Spread
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread

    # Compute 7-day and 21-day EMA of High-Low Spread
    ema_7_high_low = high_low_spread.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_low_spread.ewm(span=21, adjust=False).mean()

    # Calculate Raw Momentum
    raw_momentum = (high_low_spread.rolling(window=14).mean()).diff()
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()

    # Combine Momentum Signals
    momentum_ratio = ema_7_high_low / ema_21_high_low
    combined_momentum = momentum_ratio * intraday_movement
    short_term_momentum = combined_momentum.rolling(window=14).sum().fillna(0)

    # Calculate Volume-Weighted Close Price
    vwcp_numerator = (df['close'] * df['volume']).rolling(window=14).sum()
    vwcp_denominator = df['volume'].rolling(window=14).sum()
    vwcp = vwcp_numerator / vwcp_denominator
    vwcp.fillna(0, inplace=True)

    # Combine Dynamic and Intraday Momentum with VWCP
    vwcp_adjusted_momentum = 0.5 * combined_momentum * vwcp

    # Long-Term Price Momentum Adjustment
    long_term_average_close = df['close'].rolling(window=15).mean()
    adjusted_close_difference = df['close'] - long_term_average_close
    long_term_scaled_difference = 0.3 * adjusted_close_difference / short_term_momentum

    # Final Alpha Factor
    alpha_factor = vwcp_adjusted_momentum + long_term_scaled_difference
    return alpha_factor
