```python
import pandas as pd

def heuristics_v2(df):
    # 1.1.1 Compute the difference between today's high and low for each day
    df['range'] = df['high'] - df['low']

    # 1.1.2 Determine if the stock closed higher or lower than it opened
    df['close_open_diff'] = df['close'] - df['open']
    
    # 1.2.1 Calculate the percentage change in volume from one day to the next
    df['volume_change'] = df['volume'].pct_change()

    # 1.2.2 Identify days with significantly higher volume compared to a recent average
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > df['avg_vol_5d'] * 1.5).astype(int)

    # 1.3.1.1 Calculate the average amount traded over a certain period (e.g., 30 days)
    df['avg_amount_30d'] = df['amount'].rolling(window=30).mean()

    # 1.3.1.2 Determine the deviation of today's amount from this average
    df['amount_deviation'] = df['amount'] - df['avg_amount_30d']

    # 1.3.1.2 Create a ratio of the amount traded to the volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']

    # 1.4.1.1 Calculate the cumulative return over the last 5 days
    df['momentum_5d'] = (df['close'] / df['close'].shift(5)) - 1

    # 1.4.2.1 Calculate the distance between the current close and the 200-day moving average
    df['ma_200d'] = df['close'].rolling(window=200).mean()
    df['reversion_to_mean'] = df['close'] - df['ma_200d']

    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['range'] +
        df['close_open_diff'] +
        df['volume_change'] +
        df['volume_spike'] +
        df['amount_deviation'] +
        df['amount_volume_ratio'] +
        df['momentum_5d'] +
        df['reversion_to_mean']
    )

    return df['alpha_factor']
```
