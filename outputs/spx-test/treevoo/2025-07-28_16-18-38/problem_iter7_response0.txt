```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    df['close_change'] = df['close'].diff()
    
    # Set smoothing factor (alpha) for VW-EWMA
    alpha = 0.2
    
    # Initialize the VW-EWMA series
    df['vw_ewma'] = 0.0
    df.loc[df.index[0], 'vw_ewma'] = df.loc[df.index[0], 'close_change']
    
    # Calculate Volume-Weighted Exponentially Weighted Moving Average (VW-EWMA)
    for i in range(1, len(df)):
        previous_vw_ewma = df.loc[df.index[i-1], 'vw_ewma']
        current_close_change = df.loc[df.index[i], 'close_change']
        volume_multiplier = df.loc[df.index[i], 'volume'] / df['volume'].rolling(window=i+1).sum().iloc[-1]
        
        df.loc[df.index[i], 'vw_ewma'] = (previous_vw_ewma * (1 - alpha) + 
                                          current_close_change * alpha) * volume_multiplier
    
    # Calculate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Choose Lookback Period
    lookback_period = 14
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=lookback_period).mean()
    
    # Subtract SMA from Current Day's High-Low Difference
    df['sma_deviation'] = df['high_low_diff'] - df['sma_high_low_diff']
    
    # Apply Exponential Smoothing
    exp_smoothing_alpha = 0.2
    df['dynamic_momentum'] = df['sma_deviation'].ewm(alpha=exp_smoothing_alpha).mean()
    
    # Final alpha factor
    df['alpha_factor'] = df['vw_ewma'] + df['dynamic_momentum']
    
    return df['alpha_factor']
```
