import pandas as pd
import numpy as np

def heuristics_v2(df, n=14, smoothing_factor=0.2):
    # Compute High-to-Low Return
    df['high_to_low_return'] = (df['high'] - df['low']) / df['low']
    
    # Volume-Weighted High-to-Low Return
    df['volume_weighted_hl_return'] = df['high_to_low_return'] * df['volume']
    
    # Close-to-Open Return of Previous Day
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine with Volume-Weighted High-to-Low Return
    df['combined_return'] = df['volume_weighted_hl_return'] + df['close_to_open_return']
    
    # True Range
    df['true_range'] = df[['high', 'low', df['close'].shift(1)]].apply(lambda x: max(x) - min(x), axis=1)
    
    # Moving Sum of Log Returns weighted by volume
    log_returns = np.log(df['close'] / df['close'].shift(1))
    df['volume_weighted_log_returns'] = log_returns * df['volume']
    df['moving_sum_log_returns'] = df['volume_weighted_log_returns'].rolling(window=n).sum()
    
    # Volume-Adjusted Factor
    df['volume_adjusted_factor'] = df['moving_sum_log_returns'] / df['volume'].rolling(window=n).sum()
    
    # Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Volume-Adjusted ATR
    df['volume_adjusted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, n))
    weights /= weights.sum()
    df['volume_adjusted_atr_smoothed'] = df['volume_adjusted_atr'].rolling(window=n).apply(lambda x: np.dot(x, weights), raw=True)
    
    # Dynamic Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=n).mean()
    df['raw_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    df['smoothed_momentum'] = df['raw_momentum'].ewm(alpha=smoothing_factor).mean()
    
    # Relative Strength
    df['n_day_close_change'] = df['close'] - df['close'].shift(n)
    df['sma_n_day_close_change'] = df['n_day_close_change'].rolling(window=n).mean()
    df['relative_strength'] = df['n_day_close_change'] - df['sma_n_day_close_change']
    
    # Combine all factors
    df['combined_factor'] = df['combined_return'] + df['volume_adjusted_atr_smoothed'] + df['smoothed_momentum'] + df['relative_strength']
    
    # Cumulative Effect Over N Days
    df['cumulative_factor'] = df['combined_factor'].rolling(window=n).sum()
    
    # Incorporate Volume Trend
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_trend_ratio'] = df['volume'] / df['avg_volume']
    df['final_alpha_factor'] = df['cumulative_factor'] * df['volume_trend_ratio']
    
    return df['final_alpha_factor']
