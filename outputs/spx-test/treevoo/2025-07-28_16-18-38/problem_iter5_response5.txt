```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close Price Change
    close_change = df['close'].diff()
    
    # Initialize and Update VWEMA of Close Price Changes
    alpha = 0.1  # Smoothing factor
    vwema = close_change.copy()
    vwema[0] = close_change.iloc[0]
    for i in range(1, len(close_change)):
        vwema[i] = (vwema[i-1] * (1 - alpha)) + (close_change.iloc[i] * alpha)
    
    # Volume Weight the VWEMA
    volume_sum = df['volume'].rolling(window=21, min_periods=1).sum()
    volume_multiplier = df['volume'] / volume_sum
    vwema_weighted = vwema * volume_multiplier
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Momentum Using Volume-Weighted High-to-Low Range
    ema_7 = high_low_range.ewm(span=7, adjust=False).mean()
    ema_21 = high_low_range.ewm(span=21, adjust=False).mean()
    momentum_ratio = ema_7 / ema_21
    
    # Integrate VWEMA and Momentum
    alpha_factor = vwema_weighted * momentum_ratio
    
    return alpha_factor
```
