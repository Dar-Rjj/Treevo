```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    df['close_change'] = df['close'].diff()
    
    # Set smoothing factor (alpha)
    alpha = 0.1
    
    # Calculate Volume-Weighted Exponentially Weighted Moving Average (VW-EWMA) of Close Price Changes
    df['vw_ewma'] = 0
    df.loc[df.index[0], 'vw_ewma'] = df.loc[df.index[0], 'close_change']
    for i in range(1, len(df)):
        prev_vw_ewma = df.loc[df.index[i-1], 'vw_ewma']
        current_close_change = df.loc[df.index[i], 'close_change']
        volume_multiplier = df.loc[df.index[i], 'volume'] / df['volume'].sum()
        df.loc[df.index[i], 'vw_ewma'] = (prev_vw_ewma * (1 - alpha) + current_close_change * alpha) * volume_multiplier
    
    # Calculate Dynamic Momentum
    lookback_period = 14
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=lookback_period).mean()
    df['dynamic_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    df['dynamic_momentum'] = df['dynamic_momentum'].ewm(alpha=0.2).mean()
    
    # Calculate Price Trend Strength
    df['daily_range'] = df['high'] - df['low']
    df['atr'] = df['high'].rolling(window=lookback_period).max() - df['low'].rolling(window=lookback_period).min()
    df['trend_strength'] = df['daily_range'] / df['atr']
    
    # Integrate Price Trend Strength with Dynamic Momentum
    df['intermediate_alpha'] = df['dynamic_momentum'] * df['trend_strength']
    
    # Incorporate Previous Day's Close-to-Open Movement
    df['close_to_open_return'] = df['open'].shift(-1) / df['close'] - 1
    df['short_term_alpha'] = df['intermediate_alpha'] + df['close_to_open_return']
    
    # Long-Term Price Momentum Adjustment
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['adjusted_close_diff'] = df['close'] - df['10_day_avg_close']
    df['final_alpha_factor'] = df['adjusted_close_diff'] / df['short_term_alpha']
    
    return df['final_alpha_factor']
```
