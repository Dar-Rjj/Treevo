```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Volume-Weighted High-to-Low Return
    volume_weighted_hl_return = high_to_low_return * df['volume']
    
    # Compute Close-to-Open Return of Previous Day
    close_to_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Combine with Volume-Weighted High-to-Low Return
    combined_return = volume_weighted_hl_return + close_to_open_return
    
    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_lookback = 14
    sma_high_low = high_low_diff.rolling(window=sma_lookback).mean()
    raw_momentum = high_low_diff - sma_high_low
    smoothing_factor = 0.2
    dynamic_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Combine Volume-Weighted High-to-Low Return and Dynamic Momentum
    final_combined = combined_return + dynamic_momentum
    
    # Calculate Cumulative Effect Over N Days
    n_days = 5
    cumulative_effect = final_combined.rolling(window=n_days).sum()
    
    return cumulative_effect
```
