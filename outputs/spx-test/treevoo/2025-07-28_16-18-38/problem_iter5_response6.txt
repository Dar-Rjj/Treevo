```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Momentum
    hl_diff = df['high'] - df['low']
    ema_hl = hl_diff.ewm(span=21).mean()
    raw_momentum = hl_diff - ema_hl
    smoothed_momentum = raw_momentum.ewm(alpha=0.3).mean()

    # Bounded Transformation
    offset = 0  # Set Minimum Value
    max_possible_value = (df['high'] - df['low']).max()
    bounded_momentum = (smoothed_momentum + offset) / (max_possible_value + offset) * 100
    bounded_momentum = bounded_momentum.clip(0, 100)

    # Calculate Volume-Weighted High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    volume_weighted_return = high_to_low_return * df['volume']

    # Incorporate Previous Day's Close-to-Open Movement
    close_to_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    combined_returns = volume_weighted_return + close_to_open_return

    # Combine and Accumulate the Momentum Values
    combined_momentum = bounded_momentum + combined_returns
    combined_momentum = (combined_momentum - combined_momentum.min()) / (combined_momentum.max() - combined_momentum.min()) * 100

    rolling_sum = combined_momentum.rolling(window=21).sum().fillna(0)
    
    return rolling_sum
```
