```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: A pandas Series indexed by date, representing the computed alpha factor values.
    """
    # Calculate the percentage change of close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of high price to low price
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the volume weighted close price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate the moving average of the volume weighted close price over 10 days
    ma_volume_weighted_close = volume_weighted_close.rolling(window=10).mean()
    
    # Alpha factor: (close price change + high/low ratio) * (volume weighted close price / its 10-day moving average)
    alpha_factor = (close_pct_change + high_low_ratio) * (volume_weighted_close / ma_volume_weighted_close)
    
    return alpha_factor
```
