```python
import pandas as pd

def heuristics_v2(df, N=5):
    # Compute High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Weight by Volume
    volume_weighted_return = high_to_low_return * df['volume']
    
    # Compute Close-to-Open Return of Previous Day
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Sum Volume-Weighted High-to-Low Return and Close-to-Open Return
    combined_return = volume_weighted_return + close_to_open_return
    
    # Accumulate the Combined Value Over N Days
    rolling_sum = combined_return.rolling(window=N).sum()
    
    # 10 Day Average Close
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    
    # Adjusted Close Difference
    adjusted_close_diff = df['close'] - ten_day_avg_close
    
    # Combine with Short-Term Momentum
    final_alpha_factor = adjusted_close_diff / rolling_sum
    
    return final_alpha_factor
```
