```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Return Analysis
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Intraday Movement
    df['high_low_spread'] = df['high'] - df['low']
    df['close_open_spread'] = df['close'] - df['open']
    df['combined_spreads'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Moving Averages to Identify Trends
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['trend_indicator'] = df['5_day_MA'] - df['20_day_MA']
    df['SMA_ratio'] = df['close'] / df['close'].rolling(window=20).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_ratio'] = df['close'] / df['EMA_20']
    
    # Explore Volume and Market Interest
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['large_trade_indicator'] = np.where(df['volume'] > 2 * df['volume'].rolling(window=20).mean(), 1, 0)
    df['5_day_MA_volume'] = df['volume'].rolling(window=5).mean()
    
    # Construct Focused Composite Factors
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    df['trend_confirmation'] = df['trend_indicator'] * ((df['close'] * df['volume']) - (df['close'].shift(1) * df['volume'].shift(1))) / (df['close'].shift(1) * df['volume'].shift(1))
    
    # Enhanced Momentum Indicators
    df['high_to_low_range'] = df['high'] - df['low']
    df['7_day_EMA_HL'] = df['high_to_low_range'].ewm(span=7, adjust=False).mean()
    df['21_day_EMA_HL'] = df['high_to_low_range'].ewm(span=21, adjust=False).mean()
    df['combine_momentum_signals'] = df['7_day_EMA_HL'] / df['21_day_EMA_HL']
    df['enhanced_momentum'] = df['combine_momentum_signals'] * df['combined_spreads']
    
    # Dynamic Momentum Adjustment
    df['VWHL'] = (df['high'] - df['low']) * df['volume']
    df['14_day_SMA_VWHL'] = df['VWHL'].rolling(window=14).mean()
    df['adjusted_VWHL'] = df['VWHL'] - df['14_day_SMA_VWHL']
    df['smoothed_raw_momentum'] = df['adjusted_VWHL'].ewm(span=5, adjust=False).mean()
    df['final_momentum'] = df['smoothed_raw_momentum'] / df['combine_momentum_signals']
    
    # Enhanced Components
    df['intraday_volatility'] = abs(df['close'] - df['open'])
    df['volume_trend'] = df['volume'] - df['5_day_MA_volume']
    df['combined_enhanced_components'] = df['intraday_volatility'] + df['volume_trend']
    
    # Final Alpha Factor
    df['adjusted_momentum'] = df['final_momentum'] * df['combine_momentum_signals']
    df['alpha_factor'] = df['adjusted_momentum'] + df['combined_enhanced_components']
    
    return df['alpha_factor']
```
