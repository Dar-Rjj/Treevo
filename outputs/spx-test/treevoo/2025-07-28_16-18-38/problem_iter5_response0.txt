```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    close_changes = df['close'].diff()

    # Calculate Exponentially Weighted Moving Average (EWMA) of Close Price Changes
    alpha = 0.1  # Smoothing factor
    ewma_close_changes = close_changes.ewm(alpha=alpha, adjust=False).mean()

    # Volume Weight the EWMA of Close Price Changes
    volume_multiplier = df['volume'] / df['volume'].rolling(window=14).sum()
    volume_weighted_ewma = ewma_close_changes * volume_multiplier

    # Integrate High-Low Price Information
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothed_raw_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()

    # Combine Adjusted EWMA and High-Low Momentum
    combined_momentum = volume_weighted_ewma * smoothed_raw_momentum
    final_combined_momentum = combined_momentum.ewm(alpha=0.3, adjust=False).mean()

    # Calculate Price Momentum
    avg_daily_return = df['close'].pct_change()
    smoothed_price_momentum = avg_daily_return.rolling(window=20).mean()

    # Calculate Volume Momentum
    avg_daily_volume_change = df['volume'].pct_change()
    smoothed_volume_momentum = avg_daily_volume_change.rolling(window=20).mean()

    # Generate Combined Alpha Factor
    combined_alpha_factor = final_combined_momentum * smoothed_price_momentum
    combined_alpha_factor = combined_alpha_factor * np.sign(smoothed_volume_momentum)
    final_alpha_factor = combined_alpha_factor.ewm(alpha=0.3, adjust=False).mean()

    return final_alpha_factor
```
