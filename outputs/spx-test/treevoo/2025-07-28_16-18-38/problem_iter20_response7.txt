```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Simple Moving Average (SMA) of Close prices
    sma_window = 14
    df['sma_close'] = df['close'].rolling(window=sma_window).mean()
    df['sma_ratio'] = df['close'] / df['sma_close']
    
    # Exponential Moving Average (EMA) of Close prices
    ema_span = 14
    df['ema_close'] = df['close'].ewm(span=ema_span, adjust=False).mean()
    df['ema_ratio'] = df['close'] / df['ema_close']
    
    # Volume ratio
    df['sma_volume'] = df['volume'].rolling(window=sma_window).mean()
    df['volume_ratio'] = df['volume'] / df['sma_volume']
    
    # Price-volume momentum
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    
    # Large trade indicator
    df['large_trade_indicator'] = np.where(df['volume'] > 2 * df['sma_volume'], 1, 0)
    
    # Dynamic Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=sma_window).mean()
    df['raw_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    smoothing_factor = 0.2
    df['dynamic_momentum'] = df['raw_momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Transform to bounded range without normalization
    min_value = 0
    max_possible_value = df['dynamic_momentum'].max() - df['dynamic_momentum'].min()
    df['bounded_dynamic_momentum'] = (df['dynamic_momentum'] + abs(min_value)) * (100 / max_possible_value)
    
    # Composite factor
    df['composite_factor'] = (df['sma_ratio'] + df['ema_ratio'] + df['intraday_volatility'] + 
                              df['volume_ratio'] + df['price_volume_momentum'] + 
                              df['large_trade_indicator'] + df['bounded_dynamic_momentum']) / 7
    
    return df['composite_factor']
```
