import pandas as pd

def heuristics_v2(df):
    # Calculate Synthesized Intraday Measure
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_range = df['high'] - df['low']
    weighted_intraday_return = intraday_return * (high_low_range / high_low_range.rolling(window=10).mean())
    
    volume_impact = df['volume'] / df['volume'].rolling(window=10).mean()
    synthesized_intraday = weighted_intraday_return * volume_impact
    
    # Integrate Price and Volume Momentum
    avg_daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_momentum = avg_daily_return.rolling(window=10).mean()
    
    avg_daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_momentum = avg_daily_volume_change.rolling(window=10).mean()
    
    integrated_momentum = synthesized_intraday * (price_momentum / volume_momentum)
    integrated_momentum = integrated_momentum * (volume_momentum.apply(lambda x: 1 if x > 0 else -1))
    
    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothed_momentum = raw_momentum.ewm(alpha=0.2).mean()

    vwap = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
