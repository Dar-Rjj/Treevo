```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return and intraday volatility
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Advanced daily return and volatility
    df['advanced_daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['advanced_daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Logarithmic daily return and intraday volatility
    df['logarithmic_daily_return'] = np.log(df['close'] / df['close'].shift(1))
    df['logarithmic_intraday_volatility'] = np.log(df['high'] / df['low'])
    
    # Moving averages
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['WMA_20'] = df['close'].rolling(window=20).apply(lambda x: np.sum(x * np.arange(1, 21)) / np.sum(np.arange(1, 21)), raw=True)
    
    # SMA, EMA, WMA ratios
    df['SMA_ratio'] = df['close'] / df['SMA_20']
    df['EMA_ratio'] = df['close'] / df['EMA_20']
    df['WMA_ratio'] = df['close'] / df['WMA_20']
    
    # Short-term and long-term trends
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['trend_indicator'] = df['5_day_MA'] - df['SMA_20']
    
    # Volume analysis
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # High and low range
    df['daily_price_range'] = df['high'] - df['low']
    df['high_low_spread'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    df['high_low_range_ratio'] = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))
    
    # Composite factors
    df['composite_factor_1'] = (df['SMA_ratio'] + df['EMA_ratio'] + df['WMA_ratio'] + 
                                df['logarithmic_daily_return'] + df['logarithmic_intraday_volatility'])
    df['composite_factor_2'] = (df['daily_return'] + df['advanced_daily_return']) * df['daily_volume_change']
    df['composite_factor_3'] = df['trend_indicator'] * ((df['close'] * df['volume'] - df['close'].shift(1) * df['volume'].shift(1)) / 
                                                        (df['close'].shift(1) * df['volume'].shift(1)))
    df['composite_factor_4'] = (df['daily_return'] / df['intraday_volatility']) * df['daily_volume_change']
    df['composite_factor_5'] = (df['daily_price_range'] / (df['high'] + df['low'])) * (df['volume'] - df['volume'].shift(1))
    
    # Integrated momentum and combined trend and range
    df['integrated_momentum'] = (df['composite_factor_2'] + df['composite_factor_4']) / 2
    df['combined_trend_and_range'] = (df['composite_factor_3'] + df['composite_factor_5']) / 2
    
    # Final factor
    df['heuristics_factor'] = df['integrated_momentum'] + df['combined_trend_and_range']
    
    return df['heuristics_factor']
```
