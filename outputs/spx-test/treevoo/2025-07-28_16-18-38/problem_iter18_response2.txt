```python
import pandas as pd

def heuristics_v2(df, period=14):
    # Momentum-Based Alpha Factor
    ema_close = df['close'].ewm(span=period, adjust=False).mean()
    momentum_factor = df['close'] - ema_close
    
    # Reversal-Based Alpha Factor
    avg_high = df['high'].rolling(window=period).mean()
    reversal_factor = df['low'] / avg_high
    
    # Volume Shock Alpha Factor
    volume_change = df['volume'].pct_change()
    cumulative_volume_change = volume_change.rolling(window=period).sum()
    
    # Price Range Alpha Factor
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=period).mean()
    price_range_factor = daily_range / avg_daily_range
    
    # Open-Close Gap Alpha Factor
    open_close_gap = df['open'] - df['close']
    gap_percentage = open_close_gap / df['close'].shift(1)
    
    # Cumulative Return Alpha Factor
    cumulative_return = (df['close'] / df['close'].shift(period) - 1)
    
    # Combine all factors into a single DataFrame
    factors = pd.DataFrame({
        'momentum_factor': momentum_factor,
        'reversal_factor': reversal_factor,
        'cumulative_volume_change': cumulative_volume_change,
        'price_range_factor': price_range_factor,
        'gap_percentage': gap_percentage,
        'cumulative_return': cumulative_return
    })
    
    return factors
```
