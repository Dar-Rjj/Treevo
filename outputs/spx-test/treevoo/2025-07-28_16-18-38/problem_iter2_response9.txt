import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, ma_vol_window=20):
    # Calculate True Range and Intraday Return
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n_days).mean()
    
    # Derive Momentum Indicator
    df['momentum'] = df['close'] - df['close'].shift(n_days)
    
    # Incorporate Volume Weight
    df['volume_weighted_momentum'] = df['momentum'] * df['volume']
    
    # Adjust for Volume
    df['ma_volume'] = df['volume'].rolling(window=ma_vol_window).mean()
    df['volume_impact'] = df['volume'] / df['ma_volume']
    df['volume_adjusted_momentum'] = df['volume_weighted_momentum'] * df['volume_impact']
    
    # Adjust for Volatility
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['historical_volatility'] = df['log_returns'].rolling(window=n_days).std() * np.sqrt(n_days)
    df['volatility_adjusted_momentum'] = df['volume_adjusted_momentum'] / df['historical_volatility']
    
    return df['volatility_adjusted_momentum'].dropna()
