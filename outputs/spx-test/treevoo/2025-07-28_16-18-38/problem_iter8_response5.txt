```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    sma_high_low = high_low_diff.rolling(window=14).mean()
    
    # Calculate Raw High-to-Low Range
    raw_high_low_range = high_low_diff - sma_high_low
    
    # Apply Exponential Weighting
    smoothing_factor = 0.2
    smoothed_high_low_range = raw_high_low_range.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Compute Volume-Weighted High-to-Low Range
    volume_weighted_range = smoothed_high_low_range * df['volume']
    
    # Calculate 7-day and 21-day Exponential Moving Average (EMA) of Volume-Weighted High-to-Low Range
    ema_7 = volume_weighted_range.ewm(span=7, adjust=False).mean()
    ema_21 = volume_weighted_range.ewm(span=21, adjust=False).mean()
    
    # Compute the Ratio of 7-day EMA to 21-day EMA
    momentum_ratio = ema_7 / ema_21
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Adjust Daily Return by Amount
    amount_adjusted_return = daily_return * df['amount']
    
    # Calculate 5-day Simple Moving Average (SMA) of Amount-Adjusted Return
    sma_amount_adjusted_return = amount_adjusted_return.rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = momentum_ratio * sma_amount_adjusted_return
    
    return alpha_factor
```
