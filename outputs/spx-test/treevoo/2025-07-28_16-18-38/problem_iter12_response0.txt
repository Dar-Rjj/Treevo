```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_sum'] = df['volume'].rolling(window=n).sum()
    df['volume_adjusted_factor'] = df['volume_weighted_sum'] / df['volume_sum']
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).sum() / n
    
    # Incorporate Volume Weight
    df['volume_weighted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, n))
    weights /= weights.sum()
    df['exponentially_smoothed'] = df['volume_weighted_atr'].rolling(window=n).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Incorporate Volume Trend
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_trend'] = df['volume'] / df['average_volume']
    df['volume_trend_factor'] = df['exponentially_smoothed'] * df['volume_trend']
    
    # Introduce Price Momentum
    df['price_momentum'] = df['close'] - df['close'].rolling(window=m).mean()
    
    # Combine Momentum and Volume-Weighted Factors
    df['alpha_factor'] = df['price_momentum'] * df['volume_trend_factor']
    
    return df['alpha_factor']
```
