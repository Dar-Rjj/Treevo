import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Dynamic Momentum
    lookback_period = 14
    sma_high_low_diff = high_to_low_range.rolling(window=lookback_period).mean()
    raw_momentum = high_to_low_range - sma_high_low_diff
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()

    # Compute Momentum Components
    ema_7_days = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_days = high_to_low_range.ewm(span=21, adjust=False).mean()

    # Combine Momentum Signals
    ema_ratio = ema_7_days / ema_21_days
    final_momentum_signal = ema_ratio * smoothed_momentum
    
    return final_momentum_signal
