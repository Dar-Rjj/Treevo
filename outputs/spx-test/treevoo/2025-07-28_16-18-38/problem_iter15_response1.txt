import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Alpha Factor
    sma_period = 20
    df['SMA'] = df['close'].rolling(window=sma_period).mean()
    df['Momentum_Factor'] = df['close'] - df['SMA']
    
    # Reversal-Based Alpha Factor
    df['Reversal_Ratio'] = df['low'] / df['high'].shift(1)
    
    # Volume Shock Alpha Factor
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Price Range Alpha Factor
    df['Daily_Range'] = df['high'] - df['low']
    avg_range_period = 20
    df['Avg_Range'] = df['Daily_Range'].rolling(window=avg_range_period).mean()
    df['Price_Range_Factor'] = df['Daily_Range'] - df['Avg_Range']
    
    # Open-Close Gap Alpha Factor
    df['Open_Close_Gap'] = df['open'] - df['close']
    df['Gap_Sign'] = df['Open_Close_Gap'].apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (df['Momentum_Factor'] + 
                          df['Reversal_Ratio'] + 
                          df['Volume_Change'] + 
                          df['Price_Range_Factor'] + 
                          df['Gap_Sign'])
    
    return df['Alpha_Factor']
