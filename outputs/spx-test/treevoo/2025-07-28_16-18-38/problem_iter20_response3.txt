```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 200-day moving average and distance to it
    df['200_day_MA'] = df['close'].rolling(window=200).mean()
    df['distance_to_200MA'] = df['close'] - df['200_day_MA']
    
    # Calculate short-term and long-term moving averages
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    
    # Calculate trend indicator
    df['trend_indicator'] = df['5_day_MA'] - df['20_day_MA']
    
    # Calculate simple momentum factor
    df['cumulative_return_5_days'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate daily volume change
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate daily amount
    df['daily_amount'] = df['close'] * df['volume']
    
    # Calculate daily amount change
    df['daily_amount_change'] = (df['daily_amount'] - df['daily_amount'].shift(1)) / df['daily_amount'].shift(1)
    
    # Identify days with significantly higher volume
    df['volume_spike'] = df['volume'] > df['volume'].rolling(window=10).mean() * 2
    
    # Construct composite factors
    df['price_volume_momentum'] = (df['daily_return'] + df['daily_range'] / df['close'].shift(1)) * df['daily_volume_change']
    df['trend_confirmation'] = df['trend_indicator'] * df['daily_amount_change']
    df['sentiment_volatility'] = (df['daily_range'].rolling(window=5).mean() + df['daily_amount'].rolling(window=5).mean()) * df['trend_indicator']
    df['reversion_to_mean'] = df['distance_to_200MA'] / df['200_day_MA']
    df['reversal_indicator'] = df['low'] / df['high'].shift(1)
    df['open_close_gap'] = df['open'] - df['close'].shift(1)
    df['gap_sign'] = np.sign(df['open_close_gap'])
    
    # Select the final alpha factor
    df['alpha_factor'] = df['price_volume_momentum'] + df['trend_confirmation'] + df['sentiment_volatility'] + df['reversion_to_mean'] + df['reversal_indicator'] + df['gap_sign']
    
    return df['alpha_factor']
```
