```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Synthesize Momentum Indicators
    high_to_low_range = df['high'] - df['low']
    ema_7_high_low = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_to_low_range.ewm(span=21, adjust=False).mean()
    momentum_signal = (ema_7_high_low / ema_21_high_low) * intraday_movement
    
    # Dynamic Momentum Adjustment
    volume_weighted_high_low_diff = (df['high'] - df['low']) * df['volume']
    sma_14_volume_weighted = volume_weighted_high_low_diff.rolling(window=14).mean()
    raw_momentum = volume_weighted_high_low_diff - sma_14_volume_weighted
    smoothed_raw_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    dynamic_momentum = smoothed_raw_momentum / momentum_signal
    
    # Incorporate Price and Volume Factors
    price_range = df['high'] - df['low']
    price_direction = np.where(df['close'] > df['open'], 1, -1)
    
    volume_change = df['volume'].pct_change()
    volume_spike = df['volume'] > df['volume'].rolling(window=14).mean() * 1.5
    volume_impact = np.where(volume_spike & (df['close'] - df['open'] > 0), 1, 0)
    
    amount_30_day_avg = df['amount'].rolling(window=30).mean()
    amount_deviation = (df['amount'] - amount_30_day_avg) / amount_30_day_avg
    value_of_transactions = df['amount'] / df['volume']
    
    short_term_momentum = (df['close'] / df['close'].shift(5)) - 1
    up_down_days = (df['close'] > df['close'].shift()).rolling(window=5).sum()
    
    long_term_mean = df['close'].rolling(window=200).mean()
    z_score = (df['close'] - long_term_mean) / df['close'].rolling(window=200).std()
    
    # Construct Final Alpha Factor
    alpha_factor = (
        dynamic_momentum * 
        price_direction * 
        volume_impact * 
        amount_deviation * 
        value_of_transactions * 
        short_term_momentum * 
        up_down_days * 
        z_score
    )
    
    return alpha_factor
```
