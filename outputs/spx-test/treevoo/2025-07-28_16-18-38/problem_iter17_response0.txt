import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High and Low
    lookback_period = 14
    df['Volume_Weighted_High'] = (df['high'] * df['volume']).rolling(window=lookback_period).sum()
    df['Volume_Weighted_Low'] = (df['low'] * df['volume']).rolling(window=lookback_period).sum()
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    df['Volume_Weighted_High_Low_Diff'] = (df['Volume_Weighted_High'] - df['Volume_Weighted_Low']) / total_volume

    # Compute Dynamic Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    momentum_sma_window = 14
    df['Momentum_SMA'] = df['High_Low_Diff'].rolling(window=momentum_sma_window).mean()
    df['Raw_Momentum'] = df['High_Low_Diff'] - df['Momentum_SMA']
    smoothing_factor = 0.2
    df['Smoothed_Momentum'] = df['Raw_Momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Combine Volume-Weighted High-Low Range and Dynamic Momentum
    scaling_factor = 0.5
    df['Combined_Value'] = (df['Volume_Weighted_High_Low_Diff'] + df['Smoothed_Momentum']) * scaling_factor

    # Transform to Bounded Range Without Normalization
    offset = 0
    max_possible_value = df['Combined_Value'].max() - df['Combined_Value'].min()
    df['Bounded_Combined_Value'] = (df['Combined_Value'] + offset) * (100 / max_possible_value)

    # Compute EMA Components
    ema_7_span = 7
    ema_21_span = 21
    df['EMA_7'] = df['Bounded_Combined_Value'].ewm(span=ema_7_span, adjust=False).mean()
    df['EMA_21'] = df['Bounded_Combined_Value'].ewm(span=ema_21_span, adjust=False).mean()

    # Compute Ratio of EMAs
    df['EMA_Ratio'] = df['EMA_7'] / df['EMA_21']

    # Compute Amount-Adjusted Return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Amount_Adjusted_Return'] = df['Daily_Return'] * df['amount']
    return_sma_window = 5
    df['Amount_Adjusted_Return_SMA'] = df['Amount_Adjusted_Return'].rolling(window=return_sma_window).mean()

    # Final Alpha Factor
    df['Alpha_Factor'] = df['EMA_Ratio'] * df['Amount_Adjusted_Return_SMA']

    return df['Alpha_Factor'].dropna()
