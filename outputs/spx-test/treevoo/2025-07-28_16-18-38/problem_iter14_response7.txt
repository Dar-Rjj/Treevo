```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume information.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'volume'.
        
    Returns:
        pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the average price
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the price change
    price_change = df['close'].diff()
    
    # Calculate the volume change
    volume_change = df['volume'].diff()
    
    # Calculate the heuristic factor
    factor = (avg_price * price_change) / (volume_change + 1)
    
    return factor
```
