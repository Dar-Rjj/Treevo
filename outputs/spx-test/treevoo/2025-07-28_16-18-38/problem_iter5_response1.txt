```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    enhanced_intraday_movement = high_low_spread + close_open_spread

    # Compute Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    lookback_period = 14
    sma_high_low = high_low_diff.rolling(window=lookback_period).mean()
    raw_momentum = high_low_diff - sma_high_low
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()

    # Compute Momentum Components
    ema_7_high_low = (df['high'] - df['low']).ewm(span=7, adjust=False).mean()
    ema_21_high_low = (df['high'] - df['low']).ewm(span=21, adjust=False).mean()
    
    # Adjusted Close Difference
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_diff = df['close'] - ten_day_avg_close

    # Combine Momentum Signals
    momentum_ratio = ema_7_high_low / ema_21_high_low
    combined_momentum_signals = momentum_ratio * smoothed_momentum

    # Final Momentum Indicator
    final_momentum_indicator = (adjusted_close_diff / enhanced_intraday_movement) * combined_momentum_signals

    return final_momentum_indicator
```
