```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    volume_weighted_high_to_low_return = high_to_low_return * df['volume']

    # Incorporate Previous Day's Close-to-Open Movement
    close_to_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    combined_return = volume_weighted_high_to_low_return + close_to_open_return

    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothed_momentum = raw_momentum.ewm(alpha=0.2).mean()

    # Introduce Relative Strength
    n_day_close_change = df['close'] - df['close'].shift(10)
    sma_n_day_close_change = n_day_close_change.rolling(window=10).mean()
    relative_strength = n_day_close_change - sma_n_day_close_change

    # Combine Volume-Weighted High-to-Low Return, Dynamic Momentum, and Relative Strength
    combined_factor = combined_return + smoothed_momentum + relative_strength

    # Calculate Cumulative Effect Over N Days
    cumulative_factor = combined_factor.rolling(window=10).sum()

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()

    # Combine Dynamic Momentum and VWAP
    dynamic_momentum_vwap = smoothed_momentum * vwap * 0.5

    # Final Alpha Factor
    final_alpha_factor = cumulative_factor + dynamic_momentum_vwap

    return final_alpha_factor
```
