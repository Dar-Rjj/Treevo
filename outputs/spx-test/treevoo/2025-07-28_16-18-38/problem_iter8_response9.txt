```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    df['close_change'] = df['close'].diff()

    # Calculate Exponentially Weighted Moving Average (EWMA) of Close Price Changes
    alpha = 0.1  # Smoothing factor
    ewma = [df['close_change'].iloc[0]]  # Initialize EWMA with the first day's close price change
    for i in range(1, len(df)):
        ewma_next = ewma[-1] * (1 - alpha) + df['close_change'].iloc[i] * alpha
        ewma.append(ewma_next)
    df['ewma_close_change'] = ewma

    # Volume Weight the EWMA of Close Price Changes
    volume_sum = df['volume'].sum()
    df['volume_weighted_ewma'] = df['ewma_close_change'] * (df['volume'] / volume_sum)

    # Integrate High-Low and Close-Open Price Information
    df['high_low_spread'] = df['high'] - df['low']
    df['close_open_spread'] = df['close'] - df['open']
    df['combined_spreads'] = df['high_low_spread'] + df['close_open_spread']

    # Combined Adjusted Momentum and Volume Factor
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['adj_close_diff'] = df['close'] - df['10_day_avg_close']
    df['adj_volume_diff'] = df['volume'] - df['10_day_avg_volume']
    
    final_smoothing_factor = 0.3
    df['final_adjustment'] = (df['adj_close_diff'] / df['combined_spreads']) * df['adj_volume_diff']
    df['final_smoothed_factor'] = df['final_adjustment'].ewm(alpha=final_smoothing_factor).mean()

    return df['final_smoothed_factor']
```
