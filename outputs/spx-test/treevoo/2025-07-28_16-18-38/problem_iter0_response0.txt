```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a volume-weighted close-to-open return factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: Volume-weighted close-to-open return, indexed by date.
    """
    # Calculate the close-to-open return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
