```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()

    # Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Combine Dynamic Momentum and VWAP
    combined_momentum_vwap = 0.5 * (smoothed_momentum * vwap)

    # Calculate True Range and Intraday Return
    true_range = np.maximum.reduce([
        df['high'] - df['low'],
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ])
    intraday_return = (df['close'] - df['open']) / df['open']

    # Compute Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()

    # Derive Momentum Indicator
    momentum_indicator = (df['close'] / df['close'].shift(1)) - 1

    # Calculate Daily Close Price Changes
    close_price_changes = df['close'].diff()

    # Exponentially Weighted Moving Average (EWMA) of Close Price Changes
    ewma_close_changes = close_price_changes.ewm(alpha=0.2, adjust=False).mean()

    # Volume Multiplier
    volume_multiplier = df['volume'] / df['volume'].rolling(window=14).sum()

    # Volume Weight the Momentum Indicator and EWMA
    volume_weighted_momentum = momentum_indicator * volume_multiplier
    volume_weighted_ewma = ewma_close_changes * volume_multiplier

    # Adjust for Volume
    volume_impact = df['volume'] / df['volume'].rolling(window=14).mean()
    volume_adjusted_momentum = volume_weighted_momentum * volume_impact

    # Calculate Historical Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    historical_volatility = log_returns.rolling(window=14).std()

    # Combine and Normalize Intraday Momentum by Volatility
    adjusted_momentum = volume_adjusted_momentum / historical_volatility
    final_factor = 0.3 * adjusted_momentum + 0.3 * volume_weighted_ewma + 0.4 * atr

    return final_factor
```
