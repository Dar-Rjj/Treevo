```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high_low_range', 'high' - df['prev_close'], df['prev_close'] - df['low']]].max(axis=1)
    
    # Calculate Average True Range over last 10 days
    df['average_true_range'] = df['true_range'].rolling(window=10).mean()
    
    # Calculate Intraday Movement (High Low Spread + Close Open Spread)
    df['intraday_movement'] = df['high_low_range'] + (df['close'] - df['open'])
    
    # Compare to Average True Range
    df['momentum_score'] = df['intraday_movement'] / df['average_true_range']
    
    # Adjust for Long-Term Trend (50-Day Moving Average)
    df['50_day_MA'] = df['close'].rolling(window=50).mean()
    trend_diff = df['close'] - df['50_day_MA']
    
    # Final Momentum Score
    df['final_momentum_score'] = (df['momentum_score'] * df['volume']) + trend_diff
    
    return df['final_momentum_score']
```
