import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Thought 1.1: Enhanced Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    combined_intraday_indicator = high_low_spread + close_open_spread
    price_direction = np.where(df['close'] > df['open'], 1, -1)
    
    # Thought 1.2: Momentum Indicators with Volume Influence
    ema_7_high_low = df['high'] - df['low']
    ema_7_high_low = ema_7_high_low.ewm(span=7).mean()
    ema_21_high_low = (df['high'] - df['low']).ewm(span=21).mean()
    momentum_ratio = ema_7_high_low / ema_21_high_low
    adjusted_momentum_factor = momentum_ratio * (df['high'] - df['low'])
    volume_weighted_high_low_diff = (df['high'] - df['low']) * df['volume']
    sma_14_volume_weighted = volume_weighted_high_low_diff.rolling(window=14).mean()
    adjusted_volume_weighted_high_low_diff = volume_weighted_high_low_diff - sma_14_volume_weighted
    
    # Thought 1.3: Volume and Amount Analysis for Alpha Generation
    volume_change = df['volume'].pct_change()
    volume_spike = (df['volume'] > df['volume'].rolling(window=10).mean() * 2).astype(int)
    transaction_value = df['amount'] / df['volume']
    deviation_from_avg_transaction_value = transaction_value - transaction_value.rolling(window=30).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.5 * (adjusted_momentum_factor + adjusted_volume_weighted_high_low_diff) 
        + 0.5 * (price_direction * volume_spike)
    )
    
    return final_alpha_factor
