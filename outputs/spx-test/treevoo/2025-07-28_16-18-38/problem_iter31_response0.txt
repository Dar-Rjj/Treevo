```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Choose Lookback Period (e.g., 14 days)
    lookback = 14
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=lookback).mean()
    
    # Subtract SMA from Current Day's High-Low Difference
    df['mom_diff'] = df['high_low_diff'] - df['sma_high_low_diff']
    
    # Apply Exponential Weighting with Smoothing Factor (e.g., 0.2)
    df['exp_mom_diff'] = df['mom_diff'].ewm(alpha=0.2, adjust=False).mean()
    
    # Transform to Bounded Range Without Normalization
    min_val = 0
    max_val = df['exp_mom_diff'].max()
    df['bounded_mom'] = (df['exp_mom_diff'] - min_val) / max_val * 100
    
    # Introduce Signal Filtering
    vol_lookback = 20
    df['volatility'] = df['high_low_diff'].rolling(window=vol_lookback).std()
    df['filtered_mom'] = df['bounded_mom'] / (df['volatility'] + 1e-6)  # Ensure non-zero volatility
    
    # Integrate Volume-Weighted Momentum
    df['volume_weighted_high_low_diff'] = df['high_low_diff'] * df['volume']
    volume_lookback = 30
    df['sma_volume_weighted_high_low_diff'] = df['volume_weighted_high_low_diff'].rolling(window=volume_lookback).mean()
    
    # Identify Patterns and Trends
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Explore Trend Information
    sma_period = 50
    df['sma_close'] = df['close'].rolling(window=sma_period).mean()
    df['sma_ratio'] = df['close'] / df['sma_close']
    
    ema_period = 20
    df['ema_close'] = df['close'].ewm(span=ema_period, adjust=False).mean()
    df['ema_ratio'] = df['close'] / df['ema_close']
    
    # Assess Market Liquidity and Interest
    volume_sma_period = 30
    df['sma_volume'] = df['volume'].rolling(window=volume_sma_period).mean()
    df['volume_ratio'] = df['volume'] / df['sma_volume']
    
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    df['large_trade_indicator'] = (df['volume'] > 2 * df['sma_volume']).astype(int)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['filtered_mom'] 
        + df['sma_volume_weighted_high_low_diff'] 
        + df['daily_return'] 
        + df['intraday_volatility'] 
        + df['sma_ratio'] 
        + df['ema_ratio'] 
        + df['volume_ratio'] 
        + df['price_volume_momentum'] 
        + df['large_trade_indicator']
    )
    
    return alpha_factor
```
