import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'], df['close'].shift(1)) - min(x['low'], df['close'].shift(1)), axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_moving_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_moving_sum'] = df['volume'].rolling(window=n).sum()
    df['volume_adjusted_factor'] = df['volume_weighted_moving_sum'] / df['volume_moving_sum']
    
    # Compute Average True Range (ATR)
    df['average_true_range'] = df['true_range'].rolling(window=n).mean()
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, n))
    weights /= weights.sum()  # Normalize weights
    
    def exponential_smooth(s, w):
        return (s * w).sum()
    
    df['exp_smoothed_volume_adjusted_factor'] = df['volume_adjusted_factor'].rolling(window=n).apply(lambda x: exponential_smooth(x, weights), raw=True)
    df['exp_smoothed_average_true_range'] = df['average_true_range'].rolling(window=n).apply(lambda x: exponential_smooth(x, weights), raw=True)
    
    # Incorporate Volume Trend
    df['volume_trend_ratio'] = df['volume'] / df['volume'].rolling(window=n).mean()
    df['volume_trend_adjusted_factor'] = df['exp_smoothed_volume_adjusted_factor'] * df['volume_trend_ratio']
    
    # Introduce Price Momentum
    df['price_momentum'] = df['close'] - df['close'].rolling(window=m).mean()
    
    # Combine Momentum and Volume-Weighted Factors
    df['alpha_factor'] = df['price_momentum'] * df['volume_trend_adjusted_factor'] * df['exp_smoothed_average_true_range']
    
    # Return the alpha factor values
    return df['alpha_factor']
