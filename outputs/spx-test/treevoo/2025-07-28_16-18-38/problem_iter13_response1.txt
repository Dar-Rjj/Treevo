```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    close_changes = df['close'].diff()
    
    # Exponentially Weighted Moving Average (EWMA) of Close Changes
    alpha = 0.3  # Smoothing factor
    ewma_close_changes = close_changes.ewm(alpha=alpha, adjust=False).mean()
    
    # Volume-Weight the EWMA
    volume_multiplier = df['volume'] / df['volume'].rolling(window=10).sum()
    volume_weighted_ewma = ewma_close_changes * volume_multiplier
    
    # Integrate Intraday Price Information
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    combined_spreads = high_low_spread + close_open_spread
    
    # Adjusted Momentum and Volume Factor
    avg_close_10 = df['close'].rolling(window=10).mean()
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    adjusted_close_diff = df['close'] - avg_close_10
    adjusted_volume_diff = df['volume'] - avg_volume_10
    
    final_adjustment = (adjusted_close_diff / combined_spreads) * adjusted_volume_diff
    
    # Final Exponential Smoothing
    final_factor = final_adjustment.ewm(alpha=alpha, adjust=False).mean()
    
    return final_factor
```
