```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Change
    df['close_change'] = df['close'].diff()

    # Initialize and Update EWMA of Close Price Changes
    alpha = 0.1  # Smoothing factor
    ewma_close_change = df['close_change'].ewm(alpha=alpha, adjust=False).mean()

    # Volume Weight the EWMA
    volume_sum = df['volume'].rolling(window=len(df), min_periods=1).sum()
    volume_multiplier = df['volume'] / volume_sum
    vwema = ewma_close_change * volume_multiplier

    return vwema
```
