```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Choose Lookback Period (e.g., 14 days)
    lookback_period = 14
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    sma_high_low = high_low_diff.rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum
    raw_momentum = high_low_diff - sma_high_low
    
    # Apply Smoothing with a chosen Smoothing Factor (e.g., 0.2)
    smoothing_factor_1 = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor_1, adjust=False).mean()
    
    # Integrate Volume Impact
    volume_weighted_high_low = (df['high'] - df['low']) * df['volume']
    volume_weighted_sma = volume_weighted_high_low.rolling(window=lookback_period).mean()
    volume_weighted_momentum = volume_weighted_high_low - volume_weighted_sma
    volume_smoothed_momentum = volume_weighted_momentum.ewm(alpha=smoothing_factor_1, adjust=False).mean()
    
    # Incorporate Closing Price
    close_adjusted_high_low = (df['close'] - df['low']) - (df['high'] - df['close'])
    close_adjusted_sma = close_adjusted_high_low.rolling(window=lookback_period).mean()
    close_adjusted_momentum = close_adjusted_high_low - close_adjusted_sma
    close_adjusted_smoothed_momentum = close_adjusted_momentum.ewm(alpha=smoothing_factor_1, adjust=False).mean()
    
    # Combine Factors
    combined_factors = (smoothed_momentum + volume_smoothed_momentum + close_adjusted_smoothed_momentum) / 3
    
    # Apply Final Smoothing with a chosen Final Smoothing Factor (e.g., 0.3)
    final_smoothing_factor = 0.3
    final_alpha_factor = combined_factors.ewm(alpha=final_smoothing_factor, adjust=False).mean()
    
    return final_alpha_factor
```
