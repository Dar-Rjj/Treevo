```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Momentum Adjustment
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_difference = df['close'] - ten_day_avg_close
    
    # Combine with Intraday Movement
    factor_values = adjusted_close_difference / intraday_movement
    
    return factor_values
```
