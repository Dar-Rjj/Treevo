```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Day-to-Day Log Return of Close Price
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    tr = pd.DataFrame({
        'high_low': df['high'] - df['low'],
        'high_prev_close': abs(df['high'] - df['close'].shift(1)),
        'low_prev_close': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    volume_weighted_log_returns = (log_returns * df['volume']).rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    volume_adjusted_factor = volume_weighted_log_returns / df['volume'].rolling(window=n).sum()
    
    # Compute Average True Range (ATR)
    atr = tr.rolling(window=n).mean()
    
    # Derive Momentum Indicator
    momentum_indicator = df['close'] / df['close'].shift(n) - 1
    
    # Incorporate Volume Weight
    volume_weighted_momentum = momentum_indicator * df['volume']
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, n))
    weights /= weights.sum()
    smoothed_volume_adjusted_factor = volume_adjusted_factor.rolling(window=n).apply(lambda x: (x * weights).sum(), raw=True)
    smoothed_atr = atr.rolling(window=n).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Final Alpha Factor
    alpha_factor = smoothed_volume_adjusted_factor * smoothed_atr * volume_weighted_momentum
    
    return alpha_factor
```
