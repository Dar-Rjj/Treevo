```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Composite Momentum Indicator
    composite_momentum = high_low_range + intraday_return
    
    # Adjust for Short-Term vs. Long-Term Trend
    ema_7 = composite_momentum.ewm(span=7, adjust=False).mean()
    ema_21 = composite_momentum.ewm(span=21, adjust=False).mean()
    trend_ratio = ema_7 / ema_21
    
    # Adjust for Volume
    moving_avg_volume = df['volume'].rolling(window=21).mean()
    volume_impact = df['volume'] / moving_avg_volume
    volume_adjusted_momentum = composite_momentum * volume_impact
    
    # Adjust for Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    historical_volatility = log_returns.rolling(window=21).std()
    volatility_adjusted_momentum = volume_adjusted_momentum / historical_volatility
    
    # Combine all adjustments
    final_factor = trend_ratio * volatility_adjusted_momentum
    
    return final_factor
```
