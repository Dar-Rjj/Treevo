```python
import pandas as pd

def heuristics_v2(df):
    # Thought 1.1.1: Calculate the Range (High - Low) as a measure of volatility
    df['range'] = df['high'] - df['low']
    
    # Thought 1.1.2: Introduce a Factor based on the relationship between Close and Open
    df['close_open_diff'] = df['close'] - df['open']
    df['price_movement'] = df['close_open_diff'].apply(lambda x: 1 if x > 0 else -1)
    
    # Thought 1.2.1: Create a Factor that measures changes in volume relative to the previous day
    df['volume_change'] = df['volume'].pct_change()
    
    # Thought 1.2.2: Assess the impact of volume spikes on price
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_spike'] = df['volume'] / df['avg_volume_20']
    
    # Thought 2.1.1: Compare the amount traded on a given day to historical values
    df['avg_amount_30'] = df['amount'].rolling(window=30).mean()
    df['amount_deviation'] = df['amount'] / df['avg_amount_30']
    
    # Thought 2.1.2: Combine Volume and Amount Data
    df['value_ratio'] = df['amount'] / df['volume']
    
    # Thought 3.1.1: Design a simple momentum factor using the past 5-day returns
    df['returns_1d'] = df['close'].pct_change()
    df['momentum_5d'] = df['returns_1d'].rolling(window=5).sum()
    
    # Thought 3.1.2: Measure the number of up (or down) days in the last 5 trading days
    df['up_days'] = df['returns_1d'].apply(lambda x: 1 if x > 0 else 0).rolling(window=5).sum()
    
    # Thought 3.2.1: Use the long-term moving average (e.g., 200 days) as a baseline
    df['ma_200'] = df['close'].rolling(window=200).mean()
    df['reversion_to_mean'] = df['close'] - df['ma_200']
    
    # Thought 3.2.2: Compute the z-score of the current close relative to the 200-day mean
    df['std_200'] = df['close'].rolling(window=200).std()
    df['z_score'] = (df['close'] - df['ma_200']) / df['std_200']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['range'] * df['price_movement'] * df['volume_change'] * df['volume_spike'] * 
                          df['amount_deviation'] * df['value_ratio'] * df['momentum_5d'] * 
                          df['up_days'] * df['reversion_to_mean'] * df['z_score'])
    
    return df['alpha_factor']
```
