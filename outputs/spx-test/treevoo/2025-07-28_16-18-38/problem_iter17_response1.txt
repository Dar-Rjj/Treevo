```python
import pandas as pd

def heuristics_v2(df):
    # Thought 1.1.1: Calculate the Daily Range (High - Low) as a Volatility Measure
    df['daily_range'] = df['high'] - df['low']
    
    # Thought 1.1.2: Introduce a Factor Based on the Relationship Between Close and Open
    df['close_open_direction'] = (df['close'] > df['open']).astype(int)
    
    # Thought 1.1.3: Develop a Reversion to Mean Indicator
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['reversion_to_mean'] = df['close'] - df['200_day_ma']
    
    # Thought 1.2.1: Create a Factor that Measures Changes in Volume Relative to the Previous Day
    df['volume_change'] = df['volume'].pct_change()
    
    # Thought 1.2.2: Assess the Impact of Volume Spikes on Price
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] / df['avg_volume_5_days']) > 2
    
    # Thought 1.2.3: Develop a Factor Linking Amount Traded to Stock Performance
    df['amount_traded_deviation'] = df['amount'] - df['amount'].rolling(window=20).mean()
    
    # Thought 1.2.4: Combine Volume and Amount Data
    df['value_of_transactions'] = df['amount'] / df['volume']
    
    # Thought 2.1: Explore Short-Term Momentum
    df['5_day_momentum'] = df['close'].pct_change(periods=5)
    
    # Thought 2.2: Build a Reversal to Mean Indicator
    df['reversal_ratio'] = df['low'] / df['high'].shift(1)
    
    # Thought 3.1: Develop Momentum-Based Alpha Factor
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['momentum_factor'] = df['close'] - df['sma_50']
    
    # Thought 3.2: Develop Price Range Alpha Factor
    df['average_range_20_days'] = df['daily_range'].rolling(window=20).mean()
    df['range_factor'] = df['daily_range'] / df['average_range_20_days']
    
    # Thought 3.3: Develop Open-Close Gap Alpha Factor
    df['gap'] = df['open'] - df['close'].shift(1)
    df['gap_sign'] = df['gap'].apply(lambda x: 1 if x > 0 else -1)
    
    # Final Alpha Factor: Combine all the factors
    df['alpha_factor'] = (
        0.2 * df['daily_range'] +
        0.1 * df['close_open_direction'] +
        0.1 * df['reversion_to_mean'] +
        0.1 * df['volume_change'] +
        0.05 * df['volume_spike'] +
        0.05 * df['amount_traded_deviation'] +
        0.05 * df['value_of_transactions'] +
        0.1 * df['5_day_momentum'] +
        0.05 * df['reversal_ratio'] +
        0.1 * df['momentum_factor'] +
        0.05 * df['range_factor'] +
        0.05 * df['gap_sign']
    )
    
    return df['alpha_factor']
```
