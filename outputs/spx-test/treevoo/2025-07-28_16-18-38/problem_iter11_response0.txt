```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Synthesize Momentum Indicators
    high_to_low_range = df['high'] - df['low']
    ema_7_high_low = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_to_low_range.ewm(span=21, adjust=False).mean()
    momentum_signal = (ema_7_high_low / ema_21_high_low) * intraday_movement
    
    # Dynamic Momentum Adjustment
    high_low_diff = df['high'] - df['low']
    sma_14_high_low = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_14_high_low
    smoothed_raw_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    # Final Factor
    factor = smoothed_raw_momentum / momentum_signal
    
    return factor
```
