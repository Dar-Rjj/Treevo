```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Momentum Indicators
    ema_7_high_low = high_low_spread.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_low_spread.ewm(span=21, adjust=False).mean()
    momentum_indicator = (ema_7_high_low / ema_21_high_low) * intraday_movement
    
    # Dynamic Momentum Adjustment
    volume_weighted_high_low = (df['high'] - df['low']) * df['volume']
    sma_14_volume_weighted_high_low = volume_weighted_high_low.rolling(window=14).mean()
    subtract_sma = volume_weighted_high_low - sma_14_volume_weighted_high_low
    smoothed_subtract_sma = subtract_sma.ewm(alpha=0.2, adjust=False).mean()
    
    final_factor = smoothed_subtract_sma / momentum_indicator
    
    return final_factor
```
