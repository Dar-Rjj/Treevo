```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Synthesize Momentum Indicators
    high_to_low_range = df['high'] - df['low']
    ema_7_high_low = high_to_low_range.ewm(span=7).mean()
    ema_21_high_low = high_to_low_range.ewm(span=21).mean()
    momentum_signal = (ema_7_high_low / ema_21_high_low) * intraday_movement
    
    # Dynamic Momentum Adjustment
    volume_weighted_high_low = (df['high'] - df['low']) * df['volume']
    sma_14_vol_weighted = volume_weighted_high_low.rolling(window=14).mean()
    raw_momentum = volume_weighted_high_low - sma_14_vol_weighted
    smoothed_momentum = raw_momentum.ewm(alpha=0.2).mean()
    adjusted_momentum = smoothed_momentum / momentum_signal
    
    # Enhanced Components
    intraday_volatility = np.abs(df['close'] - df['open'])
    vol_5_day_ma = df['volume'].rolling(window=5).mean()
    volume_trend = df['volume'] - vol_5_day_ma
    enhanced_components = intraday_volatility + volume_trend
    
    # Final Alpha Factor
    final_factor = (momentum_signal * adjusted_momentum) + (enhanced_components * momentum_signal)
    
    return final_factor
```
