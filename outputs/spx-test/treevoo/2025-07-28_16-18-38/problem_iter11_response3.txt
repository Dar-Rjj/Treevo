import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic Volume-Weighted Momentum
    high_to_low_return = df['high'] - df['low']
    volume_weighted_momentum = high_to_low_return * df['volume']
    
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Simple Moving Average (SMA) of High-Low Difference with a 14-day lookback
    sma_high_low = high_low_diff.rolling(window=14).mean()
    
    # Raw Momentum: Subtract SMA from Current Day's High-Low Difference
    raw_momentum = high_low_diff - sma_high_low
    
    # Apply Exponential Weighting with a smoothing factor of 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=0.2).mean()
    
    # Accumulate the Combined Value Over N Days
    accumulated_momentum = smoothed_momentum.rolling(window=14).sum()

    # Combine with Volume-Weighted Close Price
    close_volume_product = df['close'] * df['volume']
    total_volume = df['volume'].rolling(window=14).sum()
    vwcp = (close_volume_product.rolling(window=14).sum()) / total_volume
    
    # Combine Dynamic Momentum and VWCP
    combined_momentum = (accumulated_momentum * vwcp) * 0.5
    
    # Long-Term Price Momentum Adjustment
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_diff = df['close'] - ten_day_avg_close
    
    # Final Alpha Factor: Divide Adjusted Close Difference by Accumulated Short-Term Momentum
    alpha_factor = adjusted_close_diff / accumulated_momentum

    return alpha_factor
