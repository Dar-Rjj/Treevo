```python
import pandas as pd

def heuristics_v2(df):
    # Daily Return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Open-Close Ratio
    df['Open_Close_Ratio'] = df['open'] / df['close']
    
    # True Range
    high_low = df['high'] - df['low']
    high_close_prev = (df['high'] - df['close'].shift(1)).abs()
    low_close_prev = (df['low'] - df['close'].shift(1)).abs()
    df['True_Range'] = pd.concat([high_low, high_close_prev, low_close_prev], axis=1).max(axis=1)
    
    # Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Amount-to-Volume Ratio
    df['Amount_to_Volume_Ratio'] = df['amount'] / df['volume']
    
    # Moving Average Crossover
    df['Short_MA'] = df['close'].rolling(window=5).mean()
    df['Long_MA'] = df['close'].rolling(window=20).mean()
    df['MA_Crossover'] = df['Short_MA'] - df['Long_MA']
    
    # Price Momentum (10 days and 30 days)
    df['Price_Momentum_10'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['Price_Momentum_30'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Reversal Potential
    df['Reversal_Potential'] = (df['high'] - df['close']) + (df['open'] - df['low'])

    # Select the factors as a final output
    alpha_factors = df[['Daily_Return', 'Open_Close_Ratio', 'True_Range', 'Volume_Change', 'High_Low_Range',
                        'Amount_to_Volume_Ratio', 'MA_Crossover', 'Price_Momentum_10', 'Price_Momentum_30', 'Reversal_Potential']]
    
    return alpha_factors
```
