```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['AvgDailyReturn'] = df['close'].diff()
    df['PriceMomentum'] = df['AvgDailyReturn'].rolling(window=20).mean()
    
    # Calculate Volume Momentum
    df['AvgDailyVolumeChange'] = df['volume'].diff()
    df['VolumeMomentum'] = df['AvgDailyVolumeChange'].rolling(window=20).mean()
    
    # Combine Momentum Indicators
    df['CombinedMomentum'] = df['PriceMomentum'] * df['VolumeMomentum']
    df['CombinedMomentum'] = df['CombinedMomentum'] * df['VolumeMomentum'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate High-to-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Compute Range Momentum
    df['7DayEMA_HighLowRange'] = df['HighLowRange'].ewm(span=7, adjust=False).mean()
    df['21DayEMA_HighLowRange'] = df['HighLowRange'].ewm(span=21, adjust=False).mean()
    
    # Integrate Momentum Factors
    df['EMARatio'] = df['7DayEMA_HighLowRange'] / df['21DayEMA_HighLowRange']
    df['AlphaFactor'] = df['CombinedMomentum'] * df['EMARatio']
    
    return df['AlphaFactor']
```
