```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    df['close_change'] = df['close'].diff()
    
    # Set parameters
    alpha = 0.1  # Smoothing factor for VW-EWMA
    lookback_period = 14  # Lookback period for SMA and ATR
    smoothing_factor = 0.2  # Smoothing factor for Exponential Smoothing
    
    # Initialize VW-EWMA
    df['vw_ewma'] = 0
    df.iloc[0, df.columns.get_loc('vw_ewma')] = df.iloc[0]['close_change']
    
    # Calculate Volume Multiplier
    total_volume = df['volume'].sum()
    df['volume_multiplier'] = df['volume'] / total_volume
    
    # Iteratively update VW-EWMA
    for i in range(1, len(df)):
        prev_ewma = df.iloc[i-1]['vw_ewma']
        current_change = df.iloc[i]['close_change']
        volume_multiplier = df.iloc[i]['volume_multiplier']
        df.at[df.index[i], 'vw_ewma'] = (prev_ewma * (1 - alpha) + current_change * alpha) * volume_multiplier
    
    # Compute High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=lookback_period).mean()
    
    # Subtract SMA from Current Day's High-Low Difference
    df['dynamic_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    
    # Apply Exponential Smoothing to Dynamic Momentum
    df['dynamic_momentum_smoothed'] = df['dynamic_momentum'].ewm(alpha=smoothing_factor).mean()
    
    # Calculate Average True Range (ATR)
    df['true_range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['atr'] = df['true_range'].rolling(window=lookback_period).mean()
    
    # Normalize Daily Range by ATR
    df['daily_range_normalized'] = (df['high'] - df['low']) / df['atr']
    
    # Integrate Price Trend Strength with Dynamic Momentum
    df['final_alpha_factor'] = df['dynamic_momentum_smoothed'] * df['daily_range_normalized']
    
    return df['final_alpha_factor']
```
