```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a heuristic approach.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with the following columns: open, high, low, close, volume.
    
    Returns:
    pd.Series: A Series with the generated alpha factor values.
    """
    # Calculate the difference between the high and low prices
    df['range'] = df['high'] - df['low']
    
    # Calculate the proportion of the closing price within the range
    df['close_proportion'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the proportion of the volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Combine the two proportions to create the alpha factor
    df['alpha_factor'] = df['close_proportion'] * df['volume_change']
    
    return df['alpha_factor']
```
