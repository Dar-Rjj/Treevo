```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute High-to-Low Return
    high_to_low_return = high_to_low_range / df['low']
    
    # Weight by Volume
    volume_weighted_return = high_to_low_return * df['volume']
    
    # Compute Exponential Moving Averages
    ema_7 = volume_weighted_return.ewm(span=7, adjust=False).mean()
    ema_21 = volume_weighted_return.ewm(span=21, adjust=False).mean()
    
    # Incorporate Previous Day's Close-to-Open Movement
    close_to_open_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Combine with Crossover Ratio
    crossover_ratio = ema_7 / ema_21
    alpha_factor = close_to_open_return + crossover_ratio
    
    return alpha_factor
```
