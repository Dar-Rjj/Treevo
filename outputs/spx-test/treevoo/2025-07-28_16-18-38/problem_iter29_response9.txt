```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Identify trend using moving averages
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['trend_indicator'] = df['5_day_MA'] - df['20_day_MA']
    
    # Examine trading volume
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Construct Focused Composite Factors
    df['price_volume_momentum'] = df['daily_return'] * df['daily_volume_change']
    df['trend_confirmation'] = df['trend_indicator'] * ((df['close'] * df['volume']) - (df['close'].shift(1) * df['volume'].shift(1))) / (df['close'].shift(1) * df['volume'].shift(1))
    
    # Incorporate Dynamic Momentum Oscillator
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=14).mean()
    df['raw_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    df['smoothed_momentum'] = 0.2 * df['raw_momentum'] + 0.8 * df['smoothed_momentum'].shift(1)
    min_value = 0
    max_value = df['smoothed_momentum'].max() - df['smoothed_momentum'].min()
    df['dynamic_momentum'] = (df['smoothed_momentum'] - df['smoothed_momentum'].min()) / max_value * 100
    
    # Generate Novel Composite Alpha Factors
    df['adjusted_dynamic_momentum'] = df['dynamic_momentum'] * df['daily_volume_change']
    df['combined_indicator'] = df['trend_confirmation'] * df['adjusted_dynamic_momentum']
    df['enhanced_composite'] = df['price_volume_momentum'] + df['combined_indicator']
    
    return df['enhanced_composite']
```
