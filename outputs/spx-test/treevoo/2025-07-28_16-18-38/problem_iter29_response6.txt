```python
import pandas as pd

def heuristics_v2(df):
    # Daily return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Intraday volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    
    # Simple Moving Average (SMA) of Close prices over a 14-day period
    sma_close = df['close'].rolling(window=14).mean()
    sma_ratio = df['close'] / sma_close
    
    # Exponential Moving Average (EMA) of Close prices
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    ema_ratio = df['close'] / ema_close
    
    # Volume ratio
    sma_volume = df['volume'].rolling(window=14).mean()
    volume_ratio = df['volume'] / sma_volume
    
    # Price-volume momentum
    price_volume_momentum = (df['close'] - df['open']) * df['volume']
    
    # Large trade indicator
    large_trade_indicator = (df['volume'] > 2 * sma_volume).astype(int)
    
    # High-Low Difference
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    dynamic_momentum = (smoothed_momentum - smoothed_momentum.min()) / (smoothed_momentum.max() - smoothed_momentum.min()) * 100
    
    # Define weights for the composite factor
    weights = {
        'sma_ratio': 0.15,
        'ema_ratio': 0.15,
        'intraday_volatility': 0.1,
        'volume_ratio': 0.1,
        'price_volume_momentum': 0.15,
        'large_trade_indicator': 0.1,
        'dynamic_momentum': 0.25
    }
    
    # Calculate the weighted composite factor
    weighted_composite_factor = (
        sma_ratio * weights['sma_ratio'] +
        ema_ratio * weights['ema_ratio'] +
        intraday_volatility * weights['intraday_volatility'] +
        volume_ratio * weights['volume_ratio'] +
        price_volume_momentum * weights['price_volume_momentum'] +
        large_trade_indicator * weights['large_trade_indicator'] +
        dynamic_momentum * weights['dynamic_momentum']
    )
    
    return weighted_composite_factor
```
