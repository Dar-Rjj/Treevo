```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Intraday Movement
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Close_Open_Spread'] = df['close'] - df['open']
    df['Combined_Spreads'] = df['High_Low_Spread'] + df['Close_Open_Spread']
    
    # Moving Averages
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['Trend_Indicator'] = df['5_day_MA'] - df['20_day_MA']
    df['SMA_Ratio'] = df['close'] / df['close'].rolling(window=20).mean()
    df['EMA_Ratio'] = df['close'] / df['close'].ewm(span=20, adjust=False).mean()
    
    # Volume Analysis
    df['Daily_Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Large_Trade_Indicator'] = np.where(df['volume'] > 2 * df['volume'].rolling(window=20).mean(), 1, 0)
    df['5_day_Moving_Average_of_Volume'] = df['volume'].rolling(window=5).mean()
    
    # Composite Factors
    df['Price_Volume_Momentum'] = (df['close'] - df['open']) * df['volume']
    df['Trend_Confirmation'] = df['Trend_Indicator'] * (df['close'] * df['volume'] - df['close'].shift(1) * df['volume'].shift(1)) / (df['close'].shift(1) * df['volume'].shift(1))
    
    # Enhanced Momentum Indicators
    df['High_to_Low_Range'] = df['high'] - df['low']
    df['7_day_EMA_High_Low'] = df['High_to_Low_Range'].ewm(span=7, adjust=False).mean()
    df['21_day_EMA_High_Low'] = df['High_to_Low_Range'].ewm(span=21, adjust=False).mean()
    df['Combine_Momentum_Signals'] = df['7_day_EMA_High_Low'] / df['21_day_EMA_High_Low']
    df['Multiply_by_Intraday_Movement'] = df['Combine_Momentum_Signals'] * df['Combined_Spreads']
    
    # Dynamic Momentum Adjustment
    df['Volume_Weighted_HL_Difference'] = (df['high'] - df['low']) * df['volume']
    df['14_day_SMA_VWHL'] = df['Volume_Weighted_HL_Difference'].rolling(window=14).mean()
    df['Subtract_SMA_from_Current_VWHL'] = df['Volume_Weighted_HL_Difference'] - df['14_day_SMA_VWHL']
    df['Smoothed_Raw_Momentum'] = df['Subtract_SMA_from_Current_VWHL'].ewm(span=14, adjust=False).mean()
    df['Divide_by_Combined_Momentum_Signals'] = df['Smoothed_Raw_Momentum'] / df['Combine_Momentum_Signals']
    
    # Enhanced Components
    df['Intraday_Volatility'] = abs(df['close'] - df['open'])
    df['Volume_Trend'] = df['volume'] - df['5_day_Moving_Average_of_Volume']
    df['Combine_Enhanced_Components'] = df['Intraday_Volatility'] + df['Volume_Trend']
    
    # Final Alpha Factor
    df['Adjusted_Momentum'] = df['Divide_by_Combined_Momentum_Signals']
    df['Final_Alpha_Factor'] = (df['Adjusted_Momentum'] * df['Combine_Momentum_Signals']) + df['Combine_Enhanced_Components']
    
    return df['Final_Alpha_Factor'].dropna()
```
