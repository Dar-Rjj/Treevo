```python
def heuristics_v2(data):
    # Calculate the relative position of the closing price within the high-low range
    data['close_rel_pos'] = (data['close'] - data['low']) / (data['high'] - data['low'])

    # Calculate the relative strength indicator (RSI) based on the close price
    delta = data['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    data['rsi'] = 100 - (100 / (1 + rs))

    # Calculate the money flow index (MFI)
    typical_price = (data['high'] + data['low'] + data['close']) / 3
    money_flow = typical_price * data['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    data['mfi'] = 100 - (100 / (1 + money_ratio))

    # Calculate the factor as a weighted sum of the above indicators
    factor = 0.4 * data['close_rel_pos'] + 0.3 * data['rsi'] + 0.3 * data['mfi']

    return factor
```
