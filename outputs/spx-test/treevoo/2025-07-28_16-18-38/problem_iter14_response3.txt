import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic and Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    dynamic_intraday_movement = high_low_spread + close_open_spread
    
    # Synthesize Momentum Indicators
    ema_7_high_low = high_low_spread.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_low_spread.ewm(span=21, adjust=False).mean()
    raw_momentum = high_low_spread - high_low_spread.rolling(window=14).mean()
    smoothed_momentum = raw_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    combined_momentum_signal = (ema_7_high_low / ema_21_high_low) * dynamic_intraday_movement
    accumulated_momentum = combined_momentum_signal.rolling(window=14).sum().fillna(0)
    
    # Combine with Volume-Weighted Close Price
    vwcp = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    combined_value = combined_momentum_signal * vwcp * 0.5
    
    # Long-Term Price Momentum Adjustment
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_difference = df['close'] - ten_day_avg_close
    final_alpha_factor = adjusted_close_difference / (accumulated_momentum + 1e-6)  # Add small epsilon to avoid division by zero
    
    return final_alpha_factor
