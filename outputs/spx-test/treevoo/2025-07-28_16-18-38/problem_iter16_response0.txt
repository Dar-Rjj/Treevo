```python
import pandas as pd

def heuristics_v2(df, n_days=10, lookback_period=14, smoothing_factor=0.2, scaling_factor=0.5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Weight Intraday Return by the High-Low Range
    weighted_intraday_return = intraday_return * high_low_range
    
    # Adjust for Volume
    moving_average_volume = df['volume'].rolling(window=n_days).mean()
    adjusted_intraday = weighted_intraday_return / moving_average_volume
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['close'].shift(1)).rolling(window=n_days).mean()
    
    # Calculate Volume Momentum
    volume_momentum = (df['volume'] - df['volume'].shift(1)).rolling(window=n_days).mean()
    
    # Adjust Combined Intraday Measure by Integrated Momentum
    integrated_momentum = price_momentum / volume_momentum
    adjusted_intraday = adjusted_intraday * integrated_momentum
    adjusted_intraday = adjusted_intraday * (1 if volume_momentum > 0 else -1)
    
    # Calculate Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    dynamic_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Combine Dynamic Momentum and VWAP
    combined_momentum_vwap = (dynamic_momentum * vwap) * scaling_factor
    
    # Final Alpha Factor
    alpha_factor = adjusted_intraday + combined_momentum_vwap
    return alpha_factor
```
