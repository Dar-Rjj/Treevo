```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=50, k=10):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[1], x[2]), axis=1)
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_moving_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Apply Exponential Smoothing
    weights = np.exp(np.linspace(0, -1, n))
    df['exponentially_smoothed_volume_weighted'] = df['volume_weighted_log_return'].rolling(window=n).apply(
        lambda x: np.sum(x * weights), raw=True)
    df['exponentially_smoothed_atr'] = df['atr'].rolling(window=n).apply(
        lambda x: np.sum(x * weights), raw=True)
    
    # Incorporate Volume Trend
    df['volume_trend_ratio'] = df['volume'] / df['volume'].rolling(window=n).mean()
    df['volume_trend_adjusted'] = df['exponentially_smoothed_volume_weighted'] * df['volume_trend_ratio']
    
    # Introduce Price Momentum Component
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_momentum'] = df['close'] - df['sma_close']
    df['momentum_adjusted'] = df['volume_trend_adjusted'] * df['price_momentum']
    
    # Add Volatility Component
    df['volatility'] = df['close'].rolling(window=k).std()
    df['final_factor'] = df['momentum_adjusted'] * df['volatility']
    
    return df['final_factor']
```
