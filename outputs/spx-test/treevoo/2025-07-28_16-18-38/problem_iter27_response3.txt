```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Thought 1.1: Enhanced Intraday Movement and Price Direction
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    combined_intraday_indicator = high_low_spread + close_open_spread
    price_direction = np.where(df['close'] > df['open'], 1, -1)
    
    # Thought 1.2: Momentum Indicators with Volume Influence
    ema_7_high_low = df['high'].subtract(df['low']).ewm(span=7).mean()
    ema_21_high_low = df['high'].subtract(df['low']).ewm(span=21).mean()
    momentum_ratio = ema_7_high_low / ema_21_high_low
    adjusted_momentum_factor = momentum_ratio * (df['high'] - df['low'])
    volume_weighted_high_low_diff = (df['high'] - df['low']) * df['volume']
    sma_14_vol_weighted_high_low = volume_weighted_high_low_diff.rolling(window=14).mean()
    smoothed_subtraction_result = volume_weighted_high_low_diff - sma_14_vol_weighted_high_low
    smoothed_momentum = smoothed_subtraction_result.ewm(alpha=0.2).mean()
    
    # Thought 1.3: Volume and Amount Analysis for Alpha Generation
    volume_change = df['volume'].pct_change()
    volume_spike = df['volume'] > df['volume'].rolling(window=5).mean() * 1.5
    transaction_value = df['amount'] / df['volume']
    avg_amount_30_days = df['amount'].rolling(window=30).mean()
    deviation_from_avg_amount = df['amount'] - avg_amount_30_days
    volume_weighted_momentum = smoothed_momentum * volume_spike
    
    # Thought 1.4: Advanced Intraday and Momentum Integration
    intraday_volatility = df['high'].subtract(df['low']).rolling(window=7).std()
    momentum_oscillator = (df['close'] - df['close'].rolling(window=14).mean()) / df['close'].rolling(window=14).mean()
    intraday_momentum_score = intraday_volatility * momentum_oscillator
    
    # Thought 1.5: Integrated Trend and Volume Analysis
    trend_strength = (df['close'] - df['close'].rolling(window=50).mean()) / df['close'].rolling(window=50).mean()
    volume_trend = df['volume'].ewm(span=10).mean()
    volume_trend_ratio = df['volume'] / volume_trend
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.3 * momentum_ratio + 
        0.3 * adjusted_momentum_factor + 
        0.2 * volume_weighted_momentum + 
        0.1 * price_direction +
        0.1 * volume_spike.astype(int) +
        0.1 * volume_change +
        0.1 * intraday_momentum_score +
        0.1 * trend_strength +
        0.1 * volume_trend_ratio
    )
    
    return final_alpha_factor
```
