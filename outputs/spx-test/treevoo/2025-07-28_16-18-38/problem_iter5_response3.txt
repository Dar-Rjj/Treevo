import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=14, alpha=0.6):
    # Calculate True Range
    df['tr'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    
    # Compute Average True Range (ATR)
    df['atr'] = df['tr'].rolling(window=n_days).mean()
    
    # Derive Momentum Indicator
    df['momentum'] = df['close'] - df['close'].shift(n_days)
    
    # Calculate Daily Close Price Changes
    df['price_changes'] = df['close'] - df['close'].shift(1)
    
    # Calculate Exponentially Weighted Moving Average (EWMA) of Close Price Changes
    ewma = [df['price_changes'].iloc[0]]
    for i in range(1, len(df)):
        ewma_next = (1 - alpha) * ewma[-1] + alpha * df['price_changes'].iloc[i]
        ewma.append(ewma_next)
    df['ewma'] = ewma
    
    # Volume Weight the Momentum Indicator and EWMA
    df['volume_multiplier'] = df['volume'] / df['volume'].sum()
    df['momentum_vol_weighted'] = df['momentum'] * df['volume_multiplier']
    df['ewma_vol_weighted'] = df['ewma'] * df['volume_multiplier']
    
    # Adjust for Volume
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    df['volume_impact'] = df['volume'] / df['avg_volume']
    df['intraday_momentum'] = df['close'] - df['open']
    df['intraday_momentum_vol_adj'] = df['intraday_momentum'] * df['volume_impact']
    
    # Adjust for Volatility
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['historical_volatility'] = df['log_returns'].rolling(window=n_days).std()
    df['intraday_momentum_vol_adj_norm'] = df['intraday_momentum_vol_adj'] / df['historical_volatility']
    
    # Combine and Normalize Intraday Momentum by Volatility
    df['factor_value'] = (df['intraday_momentum_vol_adj_norm'] + 
                          df['momentum_vol_weighted'] + 
                          df['ewma_vol_weighted']) / df['atr']

    return df['factor_value']
