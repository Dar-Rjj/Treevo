```python
def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the average true range over 14 days
    atr = (df['high'] - df['low']).rolling(window=14).mean()
    
    # Calculate the price change from the previous day
    price_change = df['close'].pct_change()
    
    # Calculate the volume change from the previous day
    volume_change = df['volume'].pct_change()
    
    # Calculate the factor: (daily_range / ATR) * (price_change * volume_change)
    factor = (daily_range / atr) * (price_change * volume_change)
    
    # Fill NaN values with 0
    factor.fillna(0, inplace=True)
    
    return factor
```
