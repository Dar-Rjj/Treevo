import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread

    # Synthesize Momentum Indicators
    high_to_low_range = df['high'] - df['low']
    ema_7_high_low = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_to_low_range.ewm(span=21, adjust=False).mean()
    momentum_ratio = ema_7_high_low / ema_21_high_low
    combined_momentum = momentum_ratio * intraday_movement

    # Volume-Weighted Dynamic Momentum
    daily_close_change = df['close'].diff().fillna(0)
    alpha = 0.1  # Smoothing factor
    vw_ewma = [daily_close_change.iloc[0]]  # Initial VW-EWMA value
    for i in range(1, len(daily_close_change)):
        prev_vw_ewma = vw_ewma[-1]
        current_close_change = daily_close_change.iloc[i]
        volume_multiplier = df['volume'].iloc[i] / df['volume'].sum()
        updated_vw_ewma = (prev_vw_ewma * (1 - alpha) + current_close_change * alpha) * volume_multiplier
        vw_ewma.append(updated_vw_ewma)
    vw_ewma_series = pd.Series(vw_ewma, index=df.index)

    # Dynamic Momentum Adjustment
    high_low_diff = df['high'] - df['low']
    sma_high_low = high_low_diff.rolling(window=14).mean()
    diff_sma = high_low_diff - sma_high_low
    smoothing_factor = 0.2
    exp_weighted_diff = diff_sma.ewm(alpha=smoothing_factor, adjust=False).mean()

    # Integrate Price Trend Strength with Combined Momentum
    daily_range = df['high'] - df['low']
    atr = daily_range.rolling(window=14).mean()  # Simplified ATR
    normalized_daily_range = daily_range / atr
    final_alpha = vw_ewma_series * normalized_daily_range * combined_momentum * exp_weighted_diff

    return final_alpha
