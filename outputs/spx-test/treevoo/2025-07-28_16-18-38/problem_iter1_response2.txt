```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # Momentum Adjustment
    avg_close_10_days = df['close'].rolling(window=10).mean()
    today_adjusted_close = df['close'] - avg_close_10_days
    momentum_adjusted_intraday = today_adjusted_close / (intraday_movement + 1e-6)  # add small constant to avoid division by zero
    
    return momentum_adjusted_intraday
```
