```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate intraday volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Simple Moving Average (SMA) of Close prices over a 14-day period
    sma_period = 14
    df['SMA_Close'] = df['close'].rolling(window=sma_period).mean()
    df['SMA_Ratio'] = df['close'] / df['SMA_Close']
    
    # Calculate Exponential Moving Average (EMA) of Close prices
    ema_period = 14
    df['EMA_Close'] = df['close'].ewm(span=ema_period, adjust=False).mean()
    df['EMA_Ratio'] = df['close'] / df['EMA_Close']
    
    # Calculate Volume ratio
    df['SMA_Volume'] = df['volume'].rolling(window=sma_period).mean()
    df['Volume_Ratio'] = df['volume'] / df['SMA_Volume']
    
    # Calculate Price-volume momentum
    df['Price_Volume_Momentum'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate Large trade indicator
    df['Large_Trade_Indicator'] = (df['volume'] > 2 * df['SMA_Volume']).astype(int)
    
    # Calculate High-Low Difference
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Calculate SMA of High-Low Difference over a 14-day period
    df['SMA_High_Low_Diff'] = df['High_Low_Diff'].rolling(window=sma_period).mean()
    
    # Calculate Raw Momentum
    df['Raw_Momentum'] = df['High_Low_Diff'] - df['SMA_High_Low_Diff']
    
    # Apply Exponential Weighting with a smoothing factor of 0.2
    df['Smoothed_Momentum'] = df['Raw_Momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Transform to Bounded Range Without Normalization
    min_value = 0
    max_possible_value = df['Smoothed_Momentum'].max() - df['Smoothed_Momentum'].min()
    df['Dynamic_Momentum'] = (df['Smoothed_Momentum'] + abs(min_value)) * (100 / max_possible_value)
    
    # Combine factors for more robust signals
    df['Composite_Factor'] = (df['SMA_Ratio'] + df['EMA_Ratio'] + df['Intraday_Volatility'] + 
                              df['Volume_Ratio'] + df['Price_Volume_Momentum'] + df['Large_Trade_Indicator'] + 
                              df['Dynamic_Momentum']) / 7
    
    # Introduce a weighted composite factor
    weights = {
        'SMA_Ratio': 0.15,
        'EMA_Ratio': 0.15,
        'Intraday_Volatility': 0.1,
        'Volume_Ratio': 0.1,
        'Price_Volume_Momentum': 0.15,
        'Large_Trade_Indicator': 0.1,
        'Dynamic_Momentum': 0.25
    }
    
    df['Weighted_Composite_Factor'] = (df['SMA_Ratio'] * weights['SMA_Ratio'] +
                                       df['EMA_Ratio'] * weights['EMA_Ratio'] +
                                       df['Intraday_Volatility'] * weights['Intraday_Volatility'] +
                                       df['Volume_Ratio'] * weights['Volume_Ratio'] +
                                       df['Price_Volume_Momentum'] * weights['Price_Volume_Momentum'] +
                                       df['Large_Trade_Indicator'] * weights['Large_Trade_Indicator'] +
                                       df['Dynamic_Momentum'] * weights['Dynamic_Momentum'])
    
    return df['Weighted_Composite_Factor']
```
