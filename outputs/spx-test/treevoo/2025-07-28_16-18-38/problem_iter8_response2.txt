```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-to-Low Return
    df['high_to_low_return'] = (df['high'] - df['low']) / df['low']
    df['vol_weighted_high_to_low'] = df['high_to_low_return'] * df['volume']

    # Incorporate Previous Day's Close-to-Open Movement
    df['close_to_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['combined_return'] = df['vol_weighted_high_to_low'] + df['close_to_open_return']

    # Calculate Daily Close Price Change
    df['close_change'] = df['close'].diff()

    # Initialize and Update EWMA of Close Price Changes
    ewma_alpha = 0.1
    df['ewma_close_change'] = df['close_change'].ewm(alpha=ewma_alpha, adjust=False).mean()

    # Volume Weight the EWMA
    df['total_volume'] = df['volume'].rolling(window=14).sum()
    df['vol_weighted_ewma'] = (df['ewma_close_change'] * df['volume']) / df['total_volume']

    # Calculate Dynamic Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['sma_high_low_diff'] = df['high_low_diff'].rolling(window=14).mean()
    df['raw_momentum'] = df['high_low_diff'] - df['sma_high_low_diff']
    df['smoothed_momentum'] = df['raw_momentum'].ewm(alpha=0.2, adjust=False).mean()

    # Introduce Relative Strength
    df['n_day_close_change'] = df['close'].pct_change(periods=10)
    df['sma_n_day_close_change'] = df['n_day_close_change'].rolling(window=10).mean()
    df['relative_strength'] = df['n_day_close_change'] - df['sma_n_day_close_change']

    # Compute Momentum Components
    df['7_ema_high_low_range'] = df['high_low_diff'].ewm(span=7, adjust=False).mean()
    df['21_ema_high_low_range'] = df['high_low_diff'].ewm(span=21, adjust=False).mean()

    # Combine Momentum Signals
    df['momentum_ratio'] = df['7_ema_high_low_range'] / df['21_ema_high_low_range']
    df['integrated_momentum'] = df['vol_weighted_ewma'] * df['smoothed_momentum'] * df['momentum_ratio']

    # Combine Volume-Weighted High-to-Low Return, Dynamic Momentum, and Relative Strength
    df['combined_factor'] = df['combined_return'] + df['smoothed_momentum'] + df['relative_strength']

    # Calculate Cumulative Effect Over N Days
    n_days = 10
    df['cumulative_factor'] = df['combined_factor'].rolling(window=n_days).sum()

    return df['cumulative_factor']
```
