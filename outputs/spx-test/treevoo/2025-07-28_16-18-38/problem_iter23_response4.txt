```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    
    # High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Close-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Combined Spreads
    combined_spreads = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Simple Moving Average (SMA) of Close Prices (10-day)
    sma_close = df['close'].rolling(window=10).mean()
    sma_ratio = df['close'] / sma_close
    
    # Exponential Moving Average (EMA) of Close Prices (10-day)
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_ratio = df['close'] / ema_close
    
    # 7-day EMA of High-to-Low Range
    ema_7_high_low = (df['high'] - df['low']).ewm(span=7, adjust=False).mean()
    
    # 21-day EMA of High-to-Low Range
    ema_21_high_low = (df['high'] - df['low']).ewm(span=21, adjust=False).mean()
    
    # Momentum Signal
    momentum_signal = ema_7_high_low / ema_21_high_low
    
    # Momentum Signal Adjusted
    momentum_signal_adjusted = (ema_7_high_low / ema_21_high_low) * (df['high'] - df['low'])
    
    # Volume Ratio
    volume_sma = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / volume_sma
    
    # Price-Volume Momentum
    price_volume_momentum = (df['close'] - df['open']) * df['volume']
    
    # Large Trade Indicator
    large_trade_indicator = np.where(df['volume'] > 2 * volume_sma, 1, 0)
    
    # Volume-Weighted High-Low Difference
    volume_weighted_hl_diff = (df['high'] - df['low']) * df['volume']
    volume_weighted_hl_diff_sma_14 = volume_weighted_hl_diff.rolling(window=14).mean()
    smoothed_raw_momentum = (volume_weighted_hl_diff - volume_weighted_hl_diff_sma_14) * 0.2
    dynamic_momentum_adjustment = smoothed_raw_momentum / (ema_7_high_low / ema_21_high_low)
    
    # Composite Factor
    composite_factor = (sma_ratio + ema_ratio + intraday_volatility + volume_ratio + price_volume_momentum + large_trade_indicator + momentum_signal_adjusted + dynamic_momentum_adjustment)
    
    # 5-Day Momentum
    five_day_momentum = (df['close'] / df['close'].shift(5)) - 1
    
    # Consistency of Momentum
    up_days = df['close'] > df['close'].shift(1)
    consistency_of_momentum = up_days.rolling(window=5).sum()
    
    # Reversion to Mean Indicator
    sma_200_close = df['close'].rolling(window=200).mean()
    distance_from_200_sma = df['close'] - sma_200_close
    z_score = (df['close'] - sma_200_close) / df['close'].rolling(window=200).std()
    adjusted_z_score = z_score * (df['volume'] / df['volume'].rolling(window=200).mean())
    
    # Final Alpha Factor
    alpha_factor = (composite_factor + five_day_momentum + consistency_of_momentum + adjusted_z_score)
    
    return alpha_factor
```
