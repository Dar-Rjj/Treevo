```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday Movement Analysis
    df['high_low_spread'] = df['high'] - df['low']
    df['close_open_spread'] = df['close'] - df['open']
    df['combined_spreads'] = df['high_low_spread'] + df['close_open_spread']
    
    # Moving Averages
    df['sma_close'] = df['close'].rolling(window=20).mean()
    df['sma_ratio'] = df['close'] / df['sma_close']
    df['ema_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_ratio'] = df['close'] / df['ema_close']
    
    # Explore Volume and Market Interest
    df['sma_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['sma_volume']
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    df['large_trade_indicator'] = (df['volume'] > 2 * df['sma_volume']).astype(int)
    
    # Synthesize Momentum Indicators
    df['high_to_low_range'] = df['high'] - df['low']
    df['7d_ema_high_to_low'] = df['high_to_low_range'].ewm(span=7, adjust=False).mean()
    df['21d_ema_high_to_low'] = df['high_to_low_range'].ewm(span=21, adjust=False).mean()
    df['combined_momentum_signals'] = (df['7d_ema_high_to_low'] / df['21d_ema_high_to_low']) * df['combined_spreads']
    
    # Dynamic Momentum Adjustment
    df['volume_weighted_high_low_diff'] = (df['high'] - df['low']) * df['volume']
    df['sma_volume_weighted_high_low_diff'] = df['volume_weighted_high_low_diff'].rolling(window=14).mean()
    df['smoothed_raw_momentum'] = df['volume_weighted_high_low_diff'] - df['sma_volume_weighted_high_low_diff']
    df['smoothed_raw_momentum'] = df['smoothed_raw_momentum'].ewm(alpha=0.2, adjust=False).mean()
    df['adjusted_momentum'] = df['smoothed_raw_momentum'] / df['combined_momentum_signals']
    
    # Enhanced Components
    df['intraday_volatility'] = abs(df['close'] - df['open'])
    df['5d_sma_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_trend'] = df['volume'] - df['5d_sma_volume']
    df['combined_enhanced_components'] = (df['intraday_volatility'] + df['volume_trend']) * df['combined_momentum_signals']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_momentum_signals'] * df['adjusted_momentum']) + df['combined_enhanced_components']
    
    return df['final_alpha_factor']
```
