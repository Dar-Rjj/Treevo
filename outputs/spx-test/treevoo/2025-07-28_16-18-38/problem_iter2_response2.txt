```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Close Price Changes
    df['Close_Change'] = df['close'].diff()
    
    # Calculate Exponentially Weighted Moving Average (EWMA) of Close Price Changes
    alpha = 0.1  # Smoothing factor
    ewma = df['Close_Change'].ewm(alpha=alpha, adjust=False).mean()
    df['EWMA_Close_Change'] = ewma
    
    # Volume Weight the EWMA of Close Price Changes
    total_volume = df['volume'].sum()
    volume_multiplier = df['volume'] / total_volume
    df['Volume_Weighted_EWMA'] = df['EWMA_Close_Change'] * volume_multiplier
    
    # Calculate Price Momentum
    daily_return = df['close'].pct_change()
    price_momentum = daily_return.rolling(window=50).mean()
    df['Price_Momentum'] = price_momentum
    
    # Calculate Volume Momentum
    volume_change = df['volume'].diff()
    volume_momentum = volume_change.rolling(window=50).mean()
    df['Volume_Momentum'] = volume_momentum
    
    # Combine Adjusted EWMA and Momentum Indicators
    df['Factor'] = df['Volume_Weighted_EWMA'] * df['Price_Momentum']
    df['Factor'] *= df['Volume_Momentum'].apply(lambda x: 1 if x > 0 else -1)
    
    return df['Factor']
```
