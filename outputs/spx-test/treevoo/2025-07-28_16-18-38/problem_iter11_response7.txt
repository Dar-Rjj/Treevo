```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread

    # Synthesize Momentum Indicators with Volume Weighting
    high_to_low_range = df['high'] - df['low']
    ema_7_high_to_low = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_high_to_low = high_to_low_range.ewm(span=21, adjust=False).mean()
    combined_momentum_signals = (ema_7_high_to_low / ema_21_high_to_low) * intraday_movement

    # Dynamic Momentum Adjustment
    high_low_diff = df['high'] - df['low']
    sma_14_high_low_diff = high_low_diff.rolling(window=14).mean()
    smoothed_raw_momentum = (high_low_diff - sma_14_high_low_diff).ewm(alpha=0.2, adjust=False).mean()
    adjusted_combined_momentum = smoothed_raw_momentum / combined_momentum_signals

    # Integrate High-Low and Close-Open Price Information
    combined_spreads = high_low_spread + close_open_spread

    # Combined Adjusted Momentum and Volume Factor
    avg_close_10 = df['close'].rolling(window=10).mean()
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    adjusted_close_diff = df['close'] - avg_close_10
    adjusted_volume_diff = df['volume'] - avg_volume_10
    final_adjustment = (adjusted_close_diff / combined_spreads) * adjusted_volume_diff
    final_factor = final_adjustment.ewm(alpha=0.3, adjust=False).mean()

    return final_factor
```
