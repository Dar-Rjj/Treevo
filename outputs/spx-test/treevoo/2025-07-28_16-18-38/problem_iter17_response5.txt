import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic and Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    dynamic_intraday_movement = high_low_spread + close_open_spread

    # Synthesize Momentum Indicators
    ema_7_high_low = df['high'].ewm(span=7, adjust=False).mean() - df['low'].ewm(span=7, adjust=False).mean()
    ema_21_high_low = df['high'].ewm(span=21, adjust=False).mean() - df['low'].ewm(span=21, adjust=False).mean()
    
    raw_momentum = (df['high'] - df['low']).rolling(window=14).apply(lambda x: x[-1] - x.mean(), raw=True)
    smoothed_momentum = 0.2 * raw_momentum + 0.8 * raw_momentum.shift(1)
    combined_momentum = (ema_7_high_low / ema_21_high_low) * dynamic_intraday_movement
    accumulated_momentum = combined_momentum.rolling(window=14).sum()

    # Volume-Weighted Close Price
    vwcp_numerator = (df['close'] * df['volume']).rolling(window=14).sum()
    vwcp_denominator = df['volume'].rolling(window=14).sum()
    vwcp = vwcp_numerator / vwcp_denominator
    combined_momentum_vwcp = 0.5 * combined_momentum * vwcp

    # Long-Term Price Momentum Adjustment
    ten_day_avg_close = df['close'].rolling(window=10).mean()
    adjusted_close_diff = df['close'] - ten_day_avg_close
    long_term_adjustment = 0.3 * (adjusted_close_diff / accumulated_momentum)

    # Final Alpha Factor
    final_alpha_factor = combined_momentum_vwcp + long_term_adjustment
    return final_alpha_factor
