import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Choose Lookback Period (e.g., 14 days)
    lookback_period = 14
    
    # Compute Simple Moving Average (SMA) of High-Low Differences
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    
    # Subtract SMA from Current Day's High-Low Difference
    raw_momentum = high_low_diff - sma_high_low_diff
    
    # Apply Exponential Weighting
    smoothing_factor = 0.2
    exp_weighted_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Transform to Bounded Range Without Normalization
    # Add Offset to Shift to Positive Range
    min_value = 0
    max_possible_value = (df['high'] - df['low']).max() * 2  # Assuming the maximum possible value is twice the max high-low difference
    offset = abs(exp_weighted_momentum.min())
    
    # Scale to Desired Range (e.g., 0 to 100) Using Multiplication
    bounded_range_momentum = (exp_weighted_momentum + offset) / (max_possible_value + offset) * 100
    
    return bounded_range_momentum
