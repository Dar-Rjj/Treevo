```python
import pandas as pd

def heuristics_v2(df):
    # 1.1.1 Calculate the Range (High - Low) as a measure of volatility
    df['range'] = df['high'] - df['low']
    
    # 1.1.2.1 Determine if the stock closed higher or lower than it opened
    df['close_open_diff'] = df['close'] - df['open']
    
    # 1.1.2.2.1 Use the percentage change from open to close
    df['open_close_pct_change'] = (df['close'] - df['open']) / df['open']
    
    # 1.2.1.1 Calculate the percentage change in volume from one day to the next
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # 1.2.2.1 Identify days with significantly higher volume compared to a recent average
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = df['volume'] > (df['avg_volume_5'] * 1.5)
    
    # 1.3.1.1 Calculate the difference between today's open and yesterday's close
    df['open_prev_close_diff'] = df['open'] - df['close'].shift(1)
    
    # 2.1.1.1 Calculate the average amount traded over a certain period (e.g., 30 days)
    df['avg_amount_30'] = df['amount'].rolling(window=30).mean()
    
    # 2.1.1.2 Evaluate the deviation of the current day's amount from this average
    df['amount_deviation'] = (df['amount'] - df['avg_amount_30']) / df['avg_amount_30']
    
    # 2.1.2.1 Create a ratio of the amount traded to the volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # 3.1.1.1 Calculate the cumulative return over the last 5 days
    df['momentum_5_day'] = df['close'].pct_change().rolling(window=5).sum()
    
    # 3.1.2.1 Apply higher weights to more recent returns
    df['weighted_momentum_5_day'] = df['close'].pct_change().ewm(span=5, adjust=False).sum()
    
    # 3.2.1.1 Calculate the distance between the current close and the 200-day moving average
    df['200_day_ma'] = df['close'].rolling(window=200).mean()
    df['reversion_to_mean'] = df['close'] - df['200_day_ma']
    
    # 3.2.2.1 Calculate the standard deviation of returns over the same long-term period
    df['std_200_day'] = df['close'].pct_change().rolling(window=200).std()
    
    # 3.2.2.2 Use the Z-score to measure the deviation from the mean
    df['z_score'] = (df['close'] - df['200_day_ma']) / df['std_200_day']
    
    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (
        df['range'] + 
        df['open_close_pct_change'] + 
        df['volume_pct_change'] + 
        df['open_prev_close_diff'] + 
        df['amount_deviation'] + 
        df['amount_volume_ratio'] + 
        df['momentum_5_day'] + 
        df['weighted_momentum_5_day'] + 
        df['reversion_to_mean'] + 
        df['z_score']
    )
    
    return df['alpha_factor']
```
