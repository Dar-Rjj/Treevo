```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Dynamic Momentum
    high_low_diff = high_low_range
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    dynamic_momentum = (high_low_diff - sma_high_low_diff).ewm(alpha=0.2, adjust=False).mean()
    
    # Compute EMA Components
    ema_7_day = high_low_range.ewm(span=7, adjust=False).mean()
    ema_21_day = high_low_range.ewm(span=21, adjust=False).mean()
    
    # Combine Momentum Signals
    ema_ratio = ema_7_day / ema_21_day
    combined_momentum = ema_ratio * dynamic_momentum
    
    return combined_momentum
```
