import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=20, k=5, p=30):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['true_range'] = df['true_range'].combine_first(df['true_range'].shift(1))
    
    # Compute Volume-Weighted Moving Sum of Log Returns
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['volume_weighted_log_return_sum'] = df['volume_weighted_log_return'].rolling(window=n).sum()
    
    # Compute Volume-Adjusted Factor
    df['volume_sum'] = df['volume'].rolling(window=n).sum()
    df['volume_adjusted_factor'] = df['volume_weighted_log_return_sum'] / df['volume_sum']
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Incorporate Volume Weight
    df['volume_weighted_atr'] = df['atr'] * df['volume_adjusted_factor']
    
    # Apply Exponential Smoothing
    alpha = 2 / (n + 1)
    df['exponentially_smoothed'] = df['volume_weighted_atr'].ewm(alpha=alpha, adjust=False).mean()
    
    # Incorporate Volume Trend
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_trend_ratio'] = df['volume'] / df['average_volume']
    df['volume_trend_adjusted'] = df['exponentially_smoothed'] * df['volume_trend_ratio']
    
    # Introduce Price Momentum Component
    df['sma_close'] = df['close'].rolling(window=m).mean()
    df['price_momentum'] = (df['close'] - df['sma_close']) / df['sma_close']
    df['momentum_adjusted'] = df['volume_trend_adjusted'] * df['price_momentum']
    
    # Add Volatility Component
    df['volatility'] = df['close'].rolling(window=k).std()
    df['volatility_adjusted'] = df['momentum_adjusted'] * df['volatility']
    
    # Introduce Cumulative Volume Impact
    df['cumulative_volume'] = df['volume'].rolling(window=p).sum()
    df['cumulative_volume_impact'] = df['cumulative_volume'] / n
    df['final_factor'] = df['volatility_adjusted'] * df['cumulative_volume_impact']
    
    return df['final_factor']
