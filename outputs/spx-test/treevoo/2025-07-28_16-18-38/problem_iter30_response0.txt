import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Choose Lookback Period (e.g., 14 days)
    lookback_period = 14
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum
    raw_momentum = high_low_diff - sma_high_low_diff
    
    # Apply Exponential Weighting with Smoothing Factor (e.g., 0.2)
    smoothing_factor = 0.2
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Add Offset to Shift to Positive Range
    min_value = 0
    max_possible_value = smoothed_momentum.max() - smoothed_momentum.min()
    offset = -smoothed_momentum.min()
    
    # Scale to Desired Range (e.g., 0 to 100)
    scaled_momentum = (smoothed_momentum + offset) * (100 / max_possible_value)
    
    # Apply Square Root to Smooth the Distribution
    bounded_momentum = np.sqrt(scaled_momentum)
    
    # Ensure Bounded Range Through Scaling
    final_momentum = bounded_momentum * (100 / np.sqrt(100))
    
    return final_momentum
