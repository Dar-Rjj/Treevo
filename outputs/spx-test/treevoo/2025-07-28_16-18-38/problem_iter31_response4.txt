```python
import pandas as pd

def heuristics_v2(df):
    # Daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday movement analysis
    df['high_low_spread'] = df['high'] - df['low']
    df['close_open_spread'] = df['close'] - df['open']
    df['combined_spreads'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Moving averages
    sma_close = df['close'].rolling(window=20).mean()
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # SMA and EMA ratios
    df['sma_ratio'] = df['close'] / sma_close
    df['ema_ratio'] = df['close'] / ema_close
    
    # Volume related factors
    sma_volume = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / sma_volume
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    df['large_trade_indicator'] = (df['volume'] > 2 * sma_volume).astype(int)
    
    # Dynamic momentum
    hl_diff = df['high'] - df['low']
    sma_hl_diff = hl_diff.rolling(window=20).mean()
    df['dynamic_momentum'] = (hl_diff - sma_hl_diff).ewm(span=20, adjust=False).mean()
    
    # Enhanced components
    df['intraday_volatility'] = abs(df['close'] - df['open'])
    volume_trend = df['volume'] - df['volume'].rolling(window=5).mean()
    df['enhanced_factor'] = (df['intraday_volatility'] + volume_trend) * (df['sma_ratio'] + df['ema_ratio'])
    
    # Final alpha factor
    df['alpha_factor'] = df['enhanced_factor'] + df['dynamic_momentum']
    
    return df['alpha_factor']
```
