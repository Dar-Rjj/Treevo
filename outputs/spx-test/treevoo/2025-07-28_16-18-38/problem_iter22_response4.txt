```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Daily return - (Close - Open) / Open
    daily_return = (data['close'] - data['open']) / data['open']
    
    # Intraday volatility - (High - Low) / Open
    intraday_volatility = (data['high'] - data['low']) / data['open']
    
    # Simple Moving Average (SMA) of Close prices over a 10-day period
    sma_close = data['close'].rolling(window=10).mean()
    sma_ratio = data['close'] / sma_close
    
    # Exponential Moving Average (EMA) of Close prices with a span of 10 days
    ema_close = data['close'].ewm(span=10, adjust=False).mean()
    ema_ratio = data['close'] / ema_close
    
    # Volume ratio - Current Volume / SMA(Volume)
    sma_volume = data['volume'].rolling(window=10).mean()
    volume_ratio = data['volume'] / sma_volume
    
    # Price-volume momentum - (Close - Open) * Volume
    price_volume_momentum = (data['close'] - data['open']) * data['volume']
    
    # Large trade indicator - If (Volume > 2 * SMA(Volume), 1, 0)
    large_trade_indicator = (data['volume'] > 2 * sma_volume).astype(int)
    
    # Composite factor
    composite_factor = (sma_ratio + ema_ratio + intraday_volatility + 
                        volume_ratio + price_volume_momentum + large_trade_indicator)
    
    return composite_factor
```
