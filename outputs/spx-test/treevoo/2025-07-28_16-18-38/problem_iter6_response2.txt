```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Movement
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    intraday_movement = high_low_spread + close_open_spread
    
    # 10 Day Average Close and Volume
    avg_10day_close = df['close'].rolling(window=10).mean()
    avg_10day_volume = df['volume'].rolling(window=10).mean()
    
    # Adjusted Close Difference and Volume Difference
    adjusted_close_diff = df['close'] - avg_10day_close
    adjusted_volume_diff = df['volume'] - avg_10day_volume
    
    # Combined Momentum and Volume Factor
    combined_factor = (adjusted_close_diff / intraday_movement) * adjusted_volume_diff
    
    return combined_factor
```
