import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust for Volume
    # Calculate Volume Impact
    moving_avg_volume = df['volume'].rolling(window=20).mean()
    volume_impact = df['volume'] / moving_avg_volume
    
    # Weight Intraday Momentum by Volume Impact
    weighted_intraday_momentum = intraday_return * volume_impact
    
    # Adjust for Volatility
    # Calculate Historical Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    historical_volatility = log_returns.rolling(window=20).std() * np.sqrt(252)
    
    # Apply Adjustment to Intraday Momentum
    adjusted_intraday_momentum = weighted_intraday_momentum / historical_volatility
    
    # Return the alpha factor
    return adjusted_intraday_momentum
