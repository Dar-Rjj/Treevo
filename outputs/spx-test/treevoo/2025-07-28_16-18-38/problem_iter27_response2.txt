import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average True Range (ATR) over 14 days
    df['true_range'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate High-Low Ratio
    df['high_low_ratio'] = (df['high'] - df['close'].shift(1)) / (df['low'] - df['close'].shift(1))
    
    # Calculate Distance to 200-day Moving Average
    df['sma_200'] = df['close'].rolling(window=200).mean()
    df['distance_to_sma_200'] = df['close'] - df['sma_200']
    
    # Calculate Short-term, Medium-term, and Long-term trends
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Trend Indicator: 5-day MA - 20-day MA
    df['trend_indicator'] = df['sma_5'] - df['sma_20']
    
    # Simple Momentum Factor: Cumulative return over the last 5 days
    df['simple_momentum_factor'] = df['close'].pct_change().rolling(window=5).sum()
    
    # Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate Directional Movement Index (DMI) over 14 days
    df['dm_pos'] = df['high'].diff().clip(lower=0)
    df['dm_neg'] = -df['low'].diff().clip(upper=0)
    df['tr'] = df['true_range']
    atr = df['tr'].rolling(window=14).sum()
    df['dip_14'] = (df['dm_pos'].rolling(window=14).sum() / atr) * 100
    df['din_14'] = (df['dm_neg'].rolling(window=14).sum() / atr) * 100
    df['dmi_14'] = df['dip_14'] - df['din_14']
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    df['daily_amount'] = df['close'] * df['volume']
    
    # Calculate Daily Amount Change
    df['daily_amount_change'] = (df['daily_amount'] - df['daily_amount'].shift(1)) / df['daily_amount'].shift(1)
    
    # On-Balance Volume (OBV)
    df['obv'] = np.where(df['close'] > df['close'].shift(1), df['volume'], np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
