import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Movement Analysis
    high_low_spread = df['high'] - df['low']
    close_open_spread = df['close'] - df['open']
    combined_spreads = high_low_spread + close_open_spread
    
    # Moving Averages
    sma_close = df['close'].rolling(window=20).mean()
    sma_ratio = df['close'] / sma_close
    
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_ratio = df['close'] / ema_close
    
    # Volume and Market Interest
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume
    
    price_volume_momentum = (df['close'] - df['open']) * df['volume']
    
    large_trade_indicator = (df['volume'] > 2 * avg_volume).astype(int)
    
    # Synthesize Momentum Indicators
    high_to_low_range = df['high'] - df['low']
    ema_7_high_low = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_high_low = high_to_low_range.ewm(span=21, adjust=False).mean()
    combined_momentum_signals = (ema_7_high_low / ema_21_high_low) * combined_spreads
    
    # Dynamic Momentum Adjustment
    volume_weighted_high_low = (df['high'] - df['low']) * df['volume']
    sma_volume_weighted_high_low = volume_weighted_high_low.rolling(window=14).mean()
    raw_momentum = volume_weighted_high_low - sma_volume_weighted_high_low
    smoothed_raw_momentum = raw_momentum.ewm(span=14, adjust=False).mean()
    adjusted_momentum = smoothed_raw_momentum / combined_momentum_signals
    
    # Enhanced Components
    intraday_volatility = abs(df['high'] - df['low'])
    volume_trend = df['volume'] - df['volume'].rolling(window=5).mean()
    combined_enhanced_components = (intraday_volatility + volume_trend) * combined_momentum_signals
    
    # Final Alpha Factor
    final_alpha_factor = (combined_momentum_signals * adjusted_momentum) + combined_enhanced_components
    
    return final_alpha_factor
