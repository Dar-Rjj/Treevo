```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Hybrid High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Volume-Weighted Hybrid High-to-Low Range
    vol_weighted_high_low_range = high_low_range * df['volume']
    
    # Compute Dynamic Momentum
    high_low_diff = df['high'] - df['low']
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    raw_momentum = high_low_diff - sma_high_low_diff
    smoothing_factor = 0.2
    dynamic_momentum = raw_momentum.ewm(alpha=smoothing_factor).mean()
    
    # Combine Volume-Weighted High-to-Low Range and Dynamic Momentum
    scaling_factor = 0.5
    combined_value = (vol_weighted_high_low_range * dynamic_momentum) * scaling_factor
    
    # Compute Momentum
    ema_7_day = combined_value.ewm(span=7, adjust=False).mean()
    ema_21_day = combined_value.ewm(span=21, adjust=False).mean()
    momentum_ratio = ema_7_day / ema_21_day
    
    # Compute Amount-Adjusted Return
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    amount_adjusted_return = daily_return * df['amount']
    sma_amount_adjusted_return = amount_adjusted_return.rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = momentum_ratio * sma_amount_adjusted_return
    return alpha_factor
```
