import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Choose Lookback Period and Compute EMA of High-Low Difference
    lookback_period = 21
    ema_high_low_diff = high_low_diff.ewm(span=lookback_period, adjust=False).mean()
    
    # Calculate Raw Momentum
    raw_momentum = high_low_diff - ema_high_low_diff
    
    # Apply Smoothing
    smoothing_factor = 0.3
    smoothed_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Add Offset to Shift to Positive Range
    offset = -smoothed_momentum.min() + 0  # Set Minimum Value to 0
    shifted_momentum = smoothed_momentum + offset
    
    # Scale to Desired Range (0 to 100)
    max_value = shifted_momentum.max()
    bounded_momentum = (shifted_momentum / max_value) * 100
    
    # Clip to Ensure Bounded Range
    bounded_momentum = np.clip(bounded_momentum, 0, 100)
    
    return pd.Series(bounded_momentum, index=df.index, name='DynamicMomentumOscillator')
