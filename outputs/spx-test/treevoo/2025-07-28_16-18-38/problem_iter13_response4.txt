```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Compute High-to-Low Return
    high_to_low_return = data['high'] - data['low']
    
    # Volume-Weighted High-to-Low Return
    volume_weighted_hl_return = high_to_low_return * data['volume']
    
    # Compute Close-to-Open Return (using previous day's close and current day's open)
    close_prev = data['close'].shift(1)
    close_to_open_return = close_prev - data['open']
    
    # Sum of Volume-Weighted High-to-Low Return and Close-to-Open Return
    short_term_momentum = volume_weighted_hl_return + close_to_open_return
    
    # 10 Day Average Close
    ten_day_avg_close = data['close'].rolling(window=10).mean()
    
    # Adjusted Close Difference
    adjusted_close_diff = data['close'] - ten_day_avg_close
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_close_diff / short_term_momentum)

    return final_alpha_factor
```
