```python
import pandas as pd

def heuristics_v2(df):
    # Thought 1.1.1: Calculate the Range (High - Low) as a measure of volatility
    df['range'] = df['high'] - df['low']
    
    # Thought 1.1.2: Introduce a Factor based on the relationship between Close and Open
    df['close_open_diff'] = df['close'] - df['open']
    df['price_movement'] = (df['close'] > df['open']).astype(int)
    
    # Thought 1.2.1: Create a Factor that measures changes in volume relative to the previous day
    df['volume_change'] = df['volume'].pct_change()
    
    # Thought 1.2.2: Assess the impact of volume spikes on price
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > df['avg_volume_5']).astype(int)
    
    # Thought 1.3.1: Calculate Simple Moving Averages (SMA)
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Thought 1.3.2: Analyze the Cross-Over Points
    df['sma_cross'] = (df['sma_50'] > df['sma_200']).astype(int).diff()
    
    # Thought 2.1.1: Compare the amount traded on a given day to historical values
    df['avg_amount_30'] = df['amount'].rolling(window=30).mean()
    df['amount_deviation'] = df['amount'] / df['avg_amount_30'] - 1
    
    # Thought 2.1.2: Combine Volume and Amount Data
    df['value_ratio'] = df['amount'] / df['volume']
    
    # Thought 3.1.1: Design a simple momentum factor using the past 5-day returns
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Thought 3.1.2: Investigate the consistency of momentum
    df['up_days_5'] = (df['close'] > df['open']).rolling(window=5).sum()
    
    # Thought 3.2.1: Use the long-term moving average (e.g., 200 days) as a baseline
    df['reversion_distance'] = df['close'] - df['sma_200']
    
    # Thought 3.2.2: Consider the rate of deviation from the mean
    df['z_score'] = (df['close'] - df['sma_200']) / df['close'].rolling(window=200).std()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['range'] + 
        df['close_open_diff'] * df['price_movement'] + 
        df['volume_change'] * df['volume_spike'] + 
        df['sma_cross'] * df['sma_50'] + 
        df['amount_deviation'] * df['value_ratio'] + 
        df['momentum_5d'] * df['up_days_5'] + 
        df['reversion_distance'] * df['z_score']
    )
    
    return alpha_factor
```
