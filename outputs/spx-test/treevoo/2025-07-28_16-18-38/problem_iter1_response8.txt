import pandas as pd
import numpy as np

def heuristics_v2(df, window=10, alpha=0.5):
    # Calculate Day-to-Day Log Return of Close Price
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Volume-Weighted Moving Sum
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    
    # Initialize the VWEMA factor
    vwema_factor = []
    
    for t in range(window, len(df)):
        # Compute Volume-Adjusted Factor
        volume_adjusted_factor = df['volume_weighted_log_return'][t-window+1:t+1].sum() / df['volume'][t-window+1:t+1].sum()
        
        # Apply Exponential Smoothing
        weights = np.array([alpha * (1 - alpha)**i for i in range(window)])
        smoothed_factor = (volume_adjusted_factor * weights).sum()
        
        vwema_factor.append(smoothed_factor)
    
    # Create a Series with the VWEMA factor
    vwema_series = pd.Series(vwema_factor, index=df.index[window:])
    
    return vwema_series
