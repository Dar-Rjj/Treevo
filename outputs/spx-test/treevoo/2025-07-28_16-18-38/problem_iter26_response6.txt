import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Identify trend using moving averages
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['trend_indicator'] = df['5_day_MA'] - df['20_day_MA']
    
    # Analyze daily volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Examine trading volume
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Evaluate high and low range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Construct focused composite factors
    df['price_volume_momentum'] = (df['daily_return'] + df['daily_volatility']) * df['daily_volume_change']
    df['trend_confirmation'] = df['trend_indicator'] * (df['close'] * df['volume'] - df['close'].shift(1) * df['volume'].shift(1)) / (df['close'].shift(1) * df['volume'].shift(1))
    df['volatility_adjusted_momentum'] = (df['daily_return'] / df['daily_volatility']) * df['daily_volume_change']
    df['range_adjusted_volume'] = (df['daily_price_range'] / (df['high'] + df['low'])) * (df['volume'] - df['volume'].shift(1))
    
    # Incorporate open and close prices
    df['open_close_diff'] = (df['close'] - df['open']) / df['open']
    df['high_low_spread'] = (df['high'] - df['low']) / (df['open'] + df['close'])
    
    # Use open, high, low, close prices to identify patterns and trends
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    df['range_over_close'] = (df['high'] - df['low']) / df['close']
    
    # Explore moving averages to capture trend information
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_ratio'] = df['close'] / df['SMA_20']
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_ratio'] = df['close'] / df['EMA_20']
    df['WMA_20'] = df['close'].rolling(window=20).apply(lambda x: (x * pd.Series(range(1, 21))).sum() / 210)
    df['WMA_ratio'] = df['close'] / df['WMA_20']
    
    # Utilize volume and amount data to assess market liquidity and interest
    df['volume_SMA_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_SMA_20']
    df['price_volume_momentum'] = (df['close'] - df['open']) * df['volume']
    df['large_trade_indicator'] = (df['volume'] > 2 * df['volume_SMA_20']).astype(int)
    df['money_flow'] = (df['close'] - df['open']) * df['amount']
    
    # Combine factors for more robust signals
    df['composite_factor'] = (df['SMA_ratio'] + df['EMA_ratio'] + df['WMA_ratio'] + df['intraday_volatility'] + 
                              df['range_over_close'] + df['volume_ratio'] + df['price_volume_momentum'] + 
                              df['large_trade_indicator'] + df['money_flow'])
