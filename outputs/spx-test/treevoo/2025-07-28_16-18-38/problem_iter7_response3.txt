import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Compute Dynamic Momentum
    high_low_diff = high_to_low_range
    sma_high_low_diff = high_low_diff.rolling(window=14).mean()
    dynamic_momentum = high_low_diff - sma_high_low_diff
    smoothed_dynamic_momentum = dynamic_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    # Compute Momentum Components
    ema_7_day = high_to_low_range.ewm(span=7, adjust=False).mean()
    ema_21_day = high_to_low_range.ewm(span=21, adjust=False).mean()
    
    # Combine Momentum Signals
    ema_ratio = ema_7_day / ema_21_day
    synthesized_momentum_indicator = ema_ratio * smoothed_dynamic_momentum
    
    return synthesized_momentum_indicator
