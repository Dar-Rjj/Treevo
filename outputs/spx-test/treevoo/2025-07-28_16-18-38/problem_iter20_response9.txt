```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the change in close price
    df['close_change'] = df['close'].diff()
    
    # Calculate the average true range
    df['true_range'] = df[['high'-'low', 'high'-'close_shift', 'close_shift'-'low']].max(axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate money flow ratio
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df[df['close'] > df['close'].shift(1)]['money_flow']
    df['negative_money_flow'] = df[df['close'] < df['close'].shift(1)]['money_flow']
    df['money_ratio'] = df['positive_money_flow'].rolling(window=14).sum() / df['negative_money_flow'].rolling(window=14).sum()
    
    # Combine factors
    factor_value = df['close_change'] / df['avg_true_range'] * df['money_ratio']
    
    return factor_value.dropna()
```
