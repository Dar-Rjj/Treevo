```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Difference
    high_low_diff = df['high'] - df['low']
    
    # Choose Lookback Period
    lookback_period = 14
    
    # Compute Simple Moving Average (SMA) of High-Low Difference
    sma_high_low_diff = high_low_diff.rolling(window=lookback_period).mean()
    
    # Calculate Raw Momentum
    raw_momentum = high_low_diff - sma_high_low_diff
    
    # Apply Exponential Weighting with Smoothing Factor
    smoothing_factor = 0.2
    dynamic_momentum = raw_momentum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Calculate Volume-Weighted Close Price (VWCP)
    vwcp_numerator = (df['close'] * df['volume']).rolling(window=lookback_period).sum()
    vwcp_denominator = df['volume'].rolling(window=lookback_period).sum()
    vwcp = vwcp_numerator / vwcp_denominator
    
    # Combine Dynamic Momentum and VWCP
    scaling_factor = 0.5
    combined_factor = (dynamic_momentum * vwcp) * scaling_factor
    
    return combined_factor
```
