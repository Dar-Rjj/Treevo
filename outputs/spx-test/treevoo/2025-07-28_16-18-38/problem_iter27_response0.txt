import pandas as pd

def heuristics_v2(df):
    # Calculate logarithmic daily return: log(close_t / close_t-1)
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Identify trend using exponential moving averages
    df['short_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['trend_indicator'] = df['short_ema'] - df['long_ema']
    
    # Analyze intraday volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Examine trading volume
    df['daily_volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume']
    
    # Construct composite factors
    df['price_volume_momentum'] = (df['log_return'] + df['intraday_volatility']) * df['daily_volume_change']
    df['trend_confirmation'] = df['trend_indicator'] * ((df['close'] * df['volume']) - (df['close'].shift(1) * df['volume'].shift(1))) / (df['open'] * df['volume'].shift(1))
    
    # Combine the two composite factors
    df['alpha_factor'] = df['price_volume_momentum'] + df['trend_confirmation']
    
    return df['alpha_factor']
