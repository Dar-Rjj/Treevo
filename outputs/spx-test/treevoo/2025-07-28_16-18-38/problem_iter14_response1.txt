```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High and Low
    lookback_period = 14
    df['Volume_Weighted_High'] = (df['high'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    df['Volume_Weighted_Low'] = (df['low'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Volume-Weighted High-Low Difference
    df['Volume_Weighted_HL_Diff'] = (df['Volume_Weighted_High'] - df['Volume_Weighted_Low']) / df['volume'].rolling(window=lookback_period).sum()

    # Compute Dynamic Momentum
    df['High_Low_Diff'] = df['high'] - df['low']
    df['HL_EMA'] = df['High_Low_Diff'].ewm(span=lookback_period, adjust=False).mean()
    df['Raw_Momentum'] = df['High_Low_Diff'] - df['HL_EMA']
    smoothing_factor = 0.2
    df['Smoothed_Momentum'] = df['Raw_Momentum'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Combine Volume-Weighted High-Low Range and Dynamic Momentum
    scaling_factor = 0.5
    df['Combined_Value'] = (df['Volume_Weighted_HL_Diff'] * df['Smoothed_Momentum']) * scaling_factor

    # Transform to Bounded Range Without Normalization
    offset = 0
    max_value = (df['Volume_Weighted_HL_Diff'].max() * df['Smoothed_Momentum'].max()) * scaling_factor
    df['Bounded_Range'] = (df['Combined_Value'] + offset) * (100 / max_value)

    # Compute EMA Components
    df['7_day_EMA'] = df['Bounded_Range'].ewm(span=7, adjust=False).mean()
    df['21_day_EMA'] = df['Bounded_Range'].ewm(span=21, adjust=False).mean()

    # Compute Ratio of EMAs
    df['EMA_Ratio'] = df['7_day_EMA'] / df['21_day_EMA']

    # Compute Amount-Adjusted Return
    df['Daily_Return'] = df['close'] / df['close'].shift(1) - 1
    df['Amount_Adjusted_Return'] = df['Daily_Return'] * df['amount']
    df['SMA_5_Amount_Adjusted_Return'] = df['Amount_Adjusted_Return'].rolling(window=5).mean()

    # Final Alpha Factor
    df['Alpha_Factor'] = df['EMA_Ratio'] * df['SMA_5_Amount_Adjusted_Return']

    return df['Alpha_Factor']
```
