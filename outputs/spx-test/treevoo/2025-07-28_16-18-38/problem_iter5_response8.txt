import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']

    # Compute Volume-Weighted High-to-Low Range
    volume_weighted_hl_range = high_to_low_range * df['volume']

    # Compute 7-day EMA of Volume-Weighted High-to-Low Range
    ema_7 = volume_weighted_hl_range.ewm(span=7, adjust=False).mean()

    # Compute 21-day EMA of Volume-Weighted High-to-Low Range
    ema_21 = volume_weighted_hl_range.ewm(span=21, adjust=False).mean()

    # Compute the Ratio of 7-day EMA to 21-day EMA
    ema_ratio = ema_7 / ema_21

    # Calculate Day-to-Day Log Return of Close Price
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Calculate True Range
    true_range = np.maximum(np.abs(df['high'] - df['low']),
                            np.maximum(np.abs(df['high'] - df['close'].shift(1)),
                                       np.abs(df['low'] - df['close'].shift(1))))

    # Compute Volume-Weighted Moving Sum of Log Returns
    window = 7
    volume_weighted_log_returns = (log_return * df['volume']).rolling(window=window).sum()

    # Compute Volume-Adjusted Factor
    volume_adjusted_factor = volume_weighted_log_returns / df['volume'].rolling(window=window).sum()

    # Compute Average True Range (ATR)
    atr = true_range.rolling(window=window).mean()

    # Derive Momentum Indicator
    momentum_indicator = df['close'] / df['close'].shift(window)

    # Incorporate Volume Weight
    volume_weighted_momentum = momentum_indicator * df['volume']

    # Apply Exponential Smoothing
    weights = np.exp(-np.arange(window) / window)
    smoothed_volume_adj_factor = (volume_adjusted_factor.rolling(window=window).apply(lambda x: (x * weights).sum(), raw=True))
    smoothed_atr = (atr.rolling(window=window).apply(lambda x: (x * weights).sum(), raw=True))

    # Combine all factors into a single alpha factor
    alpha_factor = ema_ratio * smoothed_volume_adj_factor * smoothed_atr

    return alpha_factor.dropna()
