import pandas as pd

def heuristics_v2(df):
    # Calculate EMA Crossover Signal
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=30, adjust=False).mean()
    ema_crossover_signal = ema_short - ema_long
    
    # Calculate Exponentially Weighted Volume
    ew_volume = df['volume'].ewm(com=0.5).mean()
    
    # Relative Difference Between Close Price and EW Volume
    close_ew_volume_diff = df['close'] / ew_volume
    
    # Combine Factors
    heuristics_matrix = 0.7 * ema_crossover_signal + 0.3 * close_ew_volume_diff
    return heuristics_matrix
