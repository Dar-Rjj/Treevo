def heuristics_v2(df):
    # Calculate the squared difference between consecutive days' close prices
    sq_diff = df['close'].diff().pow(2)
    # Sum the squared differences and the close prices over a 30-day rolling window
    sq_diff_sum = sq_diff.rolling(window=30).sum()
    close_sum_sqrt = df['close'].rolling(window=30).sum().sqrt()
    # Calculate the ratio of the sum of squared differences to the square root of the sum of close prices
    heuristics_matrix = sq_diff_sum / close_sum_sqrt
    return heuristics_matrix
