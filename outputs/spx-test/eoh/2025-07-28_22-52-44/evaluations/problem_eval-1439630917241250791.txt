def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' close prices
    abs_diff = df['close'].diff().abs()
    # Calculate the 25-day EMA of the absolute differences and the close prices
    ema_abs_diff = abs_diff.ewm(span=25, adjust=False).mean()
    ema_close = df['close'].ewm(span=25, adjust=False).mean()
    # Calculate the ratio of the EMA of absolute differences to the EMA of close prices
    heuristics_matrix = ema_abs_diff / ema_close
    return heuristics_matrix
