def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Adjust the high-low difference by the volume
    vol_adjusted_hl_diff = hl_diff * df['volume']
    # Calculate the 20-day simple moving average of the close price
    sma_close = df['close'].rolling(window=20).mean()
    # Calculate the ratio of the volume-adjusted high-low difference to the 20-day SMA of the close price
    heuristics_matrix = vol_adjusted_hl_diff / sma_close
    return heuristics_matrix
