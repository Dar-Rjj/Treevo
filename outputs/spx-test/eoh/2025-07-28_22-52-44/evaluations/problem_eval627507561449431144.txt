def heuristics_v2(df):
    # Calculate the ratio of (close - open) to (high - low)
    momentum_factor = (df['close'] - df['open']) / (df['high'] - df['low'])
    # Apply a 20-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = momentum_factor.ewm(span=20).mean()
    return heuristics_matrix
