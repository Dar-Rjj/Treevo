def heuristics_v2(df):
    # Calculate the range (high - low)
    price_range = df['high'] - df['low']
    
    # Calculate the 5-day moving average of the volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    
    # Calculate the volume-adjusted price range
    volume_adjusted_price_range = price_range / avg_volume_5
    
    # Apply a 5-day moving average to smooth out the factor
    heuristics_matrix = volume_adjusted_price_range.rolling(window=5).mean()
    
    return heuristics_matrix
