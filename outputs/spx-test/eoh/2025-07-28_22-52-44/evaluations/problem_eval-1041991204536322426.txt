def heuristics_v2(df):
    # Calculate the open-close gap and smooth it
    open_close_gap = abs(df['close'] - df['open']) / ((df['close'] + df['open']) / 2)
    smoothed_gap = open_close_gap.ewm(span=10).mean()
    
    # Calculate the 30-day price rate of change
    roc = df['close'].pct_change(periods=30)
    
    # Combine the factors
    heuristics_matrix = 0.7 * smoothed_gap + 0.3 * roc
    
    return heuristics_matrix
