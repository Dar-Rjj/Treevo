def heuristics_v2(df):
    # Calculate the inverse distance weights for a 10-day window
    weights = 1 / (pd.Series(range(1, 11)))
    # Normalize the weights so they sum to 1
    weights /= weights.sum()
    # Calculate the weighted sum of volume over a 10-day rolling window
    weighted_volume = df['volume'].rolling(window=10).apply(lambda x: (x * weights).sum(), raw=True)
    # Calculate the 10-day simple moving average of closing prices
    close_sma = df['close'].rolling(window=10).mean()
    # Generate the heuristics matrix by dividing the weighted volume by the SMA of closing prices
    heuristics_matrix = weighted_volume / close_sma
    return heuristics_matrix
