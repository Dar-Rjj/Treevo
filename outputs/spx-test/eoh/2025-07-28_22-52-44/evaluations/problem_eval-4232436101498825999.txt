def heuristics_v2(df):
    # Calculate the squared difference between consecutive days' close prices
    squared_diff = df['close'].diff().pow(2)
    # Sum the squared differences and the squared close prices over a 20-day rolling window
    squared_diff_sum = squared_diff.rolling(window=20).sum()
    close_squared_sum = df['close'].pow(2).rolling(window=20).sum()
    # Calculate the ratio of the sum of squared differences to the sum of squared close prices
    heuristics_matrix = squared_diff_sum / close_squared_sum
    return heuristics_matrix
