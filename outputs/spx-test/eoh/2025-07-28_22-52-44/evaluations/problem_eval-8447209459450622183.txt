def heuristics_v2(df):
    # Calculate the smoothed close price
    smoothed_close = df['close'].rolling(window=14).mean().pct_change(periods=21)
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=21)
    
    # Calculate the volume change factor
    volume_change = df['volume'].pct_change(periods=7)
    
    # Combine the factors with new weights
    heuristics_matrix = 0.5 * smoothed_close + 0.3 * momentum + 0.2 * volume_change
    
    return heuristics_matrix
