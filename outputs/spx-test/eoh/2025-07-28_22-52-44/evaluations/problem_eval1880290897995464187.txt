def heuristics_v2(df):
    # Calculate the 10-day average of the trading volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    # Calculate the 30-day moving average of the close prices
    ma_close_30 = df['close'].rolling(window=30).mean()
    # Compute the ratio of the 10-day average volume to the 30-day moving average of the close prices
    heuristics_matrix = avg_volume_10 / ma_close_30
    return heuristics_matrix
