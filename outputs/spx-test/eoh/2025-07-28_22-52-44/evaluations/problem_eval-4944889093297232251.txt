import numpy as np

def heuristics_v2(df):
    oc_diff = np.abs(df['open'] - df['close'])
    hl_diff = df['high'] - df['low']
    weights = np.log(df['volume'])
    weighted_oc_diff_sum = (oc_diff * weights).rolling(window=20).sum()
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=20).sum()
    heuristics_matrix = weighted_oc_diff_sum / weighted_hl_diff_sum
    return heuristics_matrix
