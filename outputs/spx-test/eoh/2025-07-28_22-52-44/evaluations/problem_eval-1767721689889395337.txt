def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the standard deviation of daily returns over a 15-day rolling window
    std_dev = daily_returns.rolling(window=15).std()
    # Calculate the 15-day rolling sum of volume
    vol_sum = df['volume'].rolling(window=15).sum()
    # Calculate the weighted sum of the standard deviation using volume as weight
    heuristics_matrix = (std_dev * df['volume']) / vol_sum
    return heuristics_matrix
