import numpy as np

def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Calculate the inverse of the cube root of the close price as the weight
    weights = 1 / np.cbrt(df['close'])
    # Weighted sum of the high-low differences over a 30-day rolling window
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=30).sum()
    # Weighted sum of the squared close prices over the same 30-day rolling window
    weighted_close_squared_sum = ((df['close'] ** 2) * weights).rolling(window=30).sum()
    # Calculate the ratio of the weighted sum of high-low differences to the weighted sum of squared close prices
    heuristics_matrix = weighted_hl_diff_sum / weighted_close_squared_sum
    return heuristics_matrix
