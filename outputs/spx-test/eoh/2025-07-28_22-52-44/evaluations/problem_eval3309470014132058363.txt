def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Calculate the ratio of the high-low difference to the close price
    hl_close_ratio = hl_diff / df['close']
    # Calculate the exponential moving average of the ratio over a 20-day window
    heuristics_matrix = hl_close_ratio.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
