def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the 20-day mean of daily returns
    mean_returns = daily_returns.rolling(window=20).mean()
    # Calculate the 20-day standard deviation of daily returns
    std_returns = daily_returns.rolling(window=20).std()
    # Calculate the ratio of the mean return to the standard deviation
    heuristics_matrix = mean_returns / std_returns
    return heuristics_matrix
