import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Adjust returns by volume
    adjusted_returns = daily_returns * df['volume']
    # Calculate the rolling standard deviation of adjusted returns over a 30-day window
    heuristics_matrix = adjusted_returns.rolling(window=30).std()
    return heuristics_matrix
