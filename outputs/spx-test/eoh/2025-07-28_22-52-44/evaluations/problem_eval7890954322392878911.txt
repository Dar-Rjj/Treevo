def heuristics_v2(df):
    df['momentum'] = df['close'] - df['open']
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_strength'] = df['volume'] / df['avg_volume_5d']
    df['heuristic_factor'] = (df['momentum'] * df['volume_strength']).shift(1)  # Shift to predict next day
    heuristics_matrix = df['heuristic_factor'].dropna()
    return heuristics_matrix
