def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' close prices
    abs_diff = df['close'].diff().abs()
    # Sum the absolute differences and calculate the mean of the close prices over a 20-day rolling window
    abs_diff_sum = abs_diff.rolling(window=20).sum()
    close_mean = df['close'].rolling(window=20).mean()
    # Calculate the ratio of the sum of absolute differences to the mean of close prices
    heuristics_matrix = abs_diff_sum / close_mean
    return heuristics_matrix
