def heuristics_v2(df):
    # Calculate the 10-day rolling average of volume
    avg_volume = df['volume'].rolling(window=10).mean()
    # Calculate the 10-day rolling maximum of close prices
    max_close = df['close'].rolling(window=10).max()
    # Calculate the ratio of the average volume to the maximum close price
    heuristics_matrix = avg_volume / max_close
    return heuristics_matrix
