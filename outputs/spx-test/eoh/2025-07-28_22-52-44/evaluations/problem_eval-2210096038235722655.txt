def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    # Calculate the mean and standard deviation of daily returns over a 10-day window
    df['mean_daily_return'] = df['daily_return'].rolling(window=10).mean()
    df['std_daily_return'] = df['daily_return'].rolling(window=10).std()
    # Adjust the mean daily return by its standard deviation to create the volatility-adjusted momentum
    df['vam_factor'] = df['mean_daily_return'] / (df['std_daily_return'] + 1e-5)  # Adding a small value to avoid division by zero
    # Prepare the heuristics matrix
    heuristics_matrix = df['vam_factor'].dropna()
    return heuristics_matrix
