def heuristics_v2(df):
    short_window = 5
    long_window = 20
    price_moving_avg_short = df['close'].rolling(window=short_window).mean()
    price_moving_avg_long = df['close'].rolling(window=long_window).mean()
    volume_change = df['volume'].pct_change().fillna(0)
    heuristics_matrix = (price_moving_avg_short - price_moving_avg_long) * (1 + volume_change)
    return heuristics_matrix
