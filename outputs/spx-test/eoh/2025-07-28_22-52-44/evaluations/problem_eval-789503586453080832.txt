def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Multiply the high-low differences by their corresponding volumes
    hl_diff_weighted = hl_diff * df['volume']
    # Sum the weighted high-low differences over a 20-day rolling window
    hl_diff_sum = hl_diff_weighted.rolling(window=20).sum()
    # Multiply the close prices by their corresponding volumes
    close_weighted = df['close'] * df['volume']
    # Sum the weighted close prices over the same 20-day rolling window
    close_sum = close_weighted.rolling(window=20).sum()
    # Calculate the ratio of the sum of weighted high-low differences to the sum of weighted close prices
    heuristics_matrix = hl_diff_sum / close_sum
    return heuristics_matrix
