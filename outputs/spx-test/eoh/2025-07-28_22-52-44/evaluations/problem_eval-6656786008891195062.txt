def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = np.cbrt(df['volume'])
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=10).sum()
    close_sma = df['close'].rolling(window=10).mean()
    heuristics_matrix = weighted_hl_diff_sum / close_sma
    return heuristics_matrix
