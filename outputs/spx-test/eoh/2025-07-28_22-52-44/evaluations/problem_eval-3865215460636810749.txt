def heuristics_v2(df):
    # Calculate the product of high and low prices for each day
    hl_product = df['high'] * df['low']
    # Sum the high-low products over a 15-day rolling window
    hl_product_sum = hl_product.rolling(window=15).sum()
    # Sum the square of close prices over the same 15-day rolling window
    close_square_sum = (df['close'] ** 2).rolling(window=15).sum()
    # Calculate the ratio of the sum of high-low products to the sum of the square of close prices
    heuristics_matrix = hl_product_sum / close_square_sum
    return heuristics_matrix
