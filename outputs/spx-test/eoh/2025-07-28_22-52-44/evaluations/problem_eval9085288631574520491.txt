def heuristics_v2(df):
    # Modified RSI with exponential moving average for gain and loss
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Adjusted MACD
    ema_12_volume = (df['volume'] * df['close']).ewm(span=12, adjust=False).mean() / df['volume'].ewm(span=12, adjust=False).mean()
    ema_26_volume = (df['volume'] * df['close']).ewm(span=26, adjust=False).mean() / df['volume'].ewm(span=26, adjust=False).mean()
    macd_volume_adjusted = ema_12_volume - ema_26_volume
    signal_volume_adjusted = macd_volume_adjusted.rolling(window=9).mean()
    
    # Adjusted Momentum Indicator: Rate of Change in the Difference between Close and VWAP
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    diff_close_vwap = df['close'] - vwap
    roc_diff = diff_close_vwap.pct_change(periods=14).fillna(0)
    
    # Combine factors
    heuristics_matrix = 0.5 * (rsi - 50) + 0.3 * (macd_volume_adjusted - signal_volume_adjusted) + 0.2 * roc_diff
    return heuristics_matrix
