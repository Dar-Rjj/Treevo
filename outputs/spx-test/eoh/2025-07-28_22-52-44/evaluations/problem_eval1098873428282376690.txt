def heuristics_v2(df):
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Average Daily Return Over the Past 30 Days
    avg_daily_return_30 = daily_return.rolling(window=30).mean()
    
    # Ratio of Daily Return to Average Daily Return
    return_ratio = daily_return / avg_daily_return_30
    
    # Maximum Volume Over the Past 50 Days
    max_volume_50 = df['volume'].rolling(window=50).max()
    
    # Ratio of Current Volume to Maximum Volume
    volume_ratio = df['volume'] / max_volume_50
    
    # Combine Factors
    heuristics_matrix = 0.7 * return_ratio + 0.3 * volume_ratio
    return heuristics_matrix
