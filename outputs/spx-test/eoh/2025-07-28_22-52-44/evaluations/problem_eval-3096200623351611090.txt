import numpy as np
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Calculate the inverse of the square root of the close price as the weight
    weights = 1 / np.sqrt(df['close'])
    # Exponentially weighted moving average of the high-low differences over a 20-day rolling window
    ewma_hl_diff = (hl_diff * weights).ewm(span=20).mean()
    # Exponentially weighted moving average of the close prices over the same 20-day rolling window
    ewma_close = (df['close'] * weights).ewm(span=20).mean()
    # Calculate the ratio of the exponentially weighted moving average of high-low differences to the exponentially weighted moving average of close prices
    heuristics_matrix = ewma_hl_diff / ewma_close
    return heuristics_matrix
