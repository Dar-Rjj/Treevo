def heuristics_v2(df):
    # Calculate the cumulative sum of volume over a 20-day window
    vol_cumsum = df['volume'].rolling(window=20).sum()
    # Calculate the standard deviation of close prices over a 20-day window
    close_std = df['close'].rolling(window=20).std()
    # Generate the heuristic matrix by dividing the cumulative sum of volume by the standard deviation of close prices
    heuristics_matrix = vol_cumsum / close_std
    return heuristics_matrix
