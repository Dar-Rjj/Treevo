def heuristics_v2(df):
    # Calculate the 5-day simple moving average (SMA) of the close price
    df['sma_5'] = df['close'].rolling(window=5).mean()
    # Calculate the 20-day exponential moving average (EMA) of the close price
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    # Compute the difference between the close price and EMA to capture momentum
    df['price_deviation_from_ema'] = df['close'] - df['ema_20']
    # Determine the 5-day SMA of the daily return
    df['daily_return'] = df['close'].pct_change()
    df['sma_daily_return_5'] = df['daily_return'].rolling(window=5).mean()
    # Construct the heuristics matrix integrating both factors
    heuristics_matrix = (df['sma_daily_return_5'] + df['price_deviation_from_ema'])
    return heuristics_matrix
