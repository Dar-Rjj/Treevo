import numpy as np

def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Calculate the inverse of the square root of the close price as the weight
    weights = 1 / np.sqrt(df['close'])
    # Weighted sum of the high-low differences over a 20-day rolling window
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=20).sum()
    # Weighted sum of the close prices over the same 20-day rolling window
    weighted_close_sum = (df['close'] * weights).rolling(window=20).sum()
    # Calculate the ratio of the weighted sum of high-low differences to the weighted sum of close prices
    heuristics_matrix = weighted_hl_diff_sum / weighted_close_sum
    return heuristics_matrix
