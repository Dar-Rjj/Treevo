def heuristics_v2(df):
    # Calculate the ratio of volume to close price
    volume_close_ratio = df['volume'] / df['close']
    
    # Calculate the 15-day EMA of the volume to close price ratio
    ema_15 = volume_close_ratio.ewm(span=15, adjust=False).mean()
    
    # Calculate the 30-day SMA of the volume to close price ratio
    sma_30 = volume_close_ratio.rolling(window=30).mean()
    
    # Subtract the 30-day SMA from the 15-day EMA to get the final heuristics matrix
    heuristics_matrix = ema_15 - sma_30
    
    return heuristics_matrix
