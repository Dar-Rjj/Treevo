import numpy as np

def heuristics_v2(df):
    co_diff = (df['close'] - df['open']).abs()
    weights = np.log(df['volume'] + 1)  # Adding 1 to handle log(0) case
    weighted_co_diff_sum = (co_diff * weights).rolling(window=20).sum()
    weighted_high_sum = (df['high'] * weights).rolling(window=20).sum()
    heuristics_matrix = weighted_co_diff_sum / weighted_high_sum
    return heuristics_matrix
