def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the standard deviation and mean of absolute daily returns over a 30-day rolling window
    std_dev = daily_returns.rolling(window=30).std()
    abs_mean = daily_returns.abs().rolling(window=30).mean()
    # Calculate the ratio of standard deviation to the mean of absolute daily returns
    heuristics_matrix = std_dev / abs_mean
    return heuristics_matrix
