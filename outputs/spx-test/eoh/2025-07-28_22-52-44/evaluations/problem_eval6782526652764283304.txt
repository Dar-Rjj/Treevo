def heuristics_v2(df):
    # Calculate the squared difference between consecutive days' close prices
    sq_diff = df['close'].diff() ** 2
    # Sum the squared differences and the close prices over a 20-day rolling window
    sq_diff_sum = sq_diff.rolling(window=20).sum()
    close_sum = df['close'].rolling(window=20).sum()
    # Calculate the ratio of the sum of squared differences to the square of the sum of close prices
    heuristics_matrix = sq_diff_sum / (close_sum ** 2)
    return heuristics_matrix
