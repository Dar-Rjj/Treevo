def heuristics_v2(df):
    # Scale daily high-low differences with the volume
    hl_scaled = (df['high'] - df['low']) * df['volume']
    # Sum the scaled high-low differences over a 20-day rolling window
    hl_scaled_sum = hl_scaled.rolling(window=20).sum()
    # Sum the close prices over the same 20-day rolling window
    close_sum = df['close'].rolling(window=20).sum()
    # Calculate the ratio of the sum of scaled high-low differences to the sum of close prices
    heuristics_matrix = hl_scaled_sum / close_sum
    return heuristics_matrix
