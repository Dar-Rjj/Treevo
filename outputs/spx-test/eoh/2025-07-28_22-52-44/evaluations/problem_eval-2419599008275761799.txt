def heuristics_v2(df):
    # Calculate the 5-day moving average of close prices
    ma_5 = df['close'].rolling(window=5).mean()
    # Calculate the ratio of each day's closing price to its 5-day moving average
    ratio = df['close'] / ma_5
    # Calculate the cumulative product of these ratios over a 60-day window
    heuristics_matrix = ratio.rolling(window=60).apply(lambda x: x.prod(), raw=True)
    return heuristics_matrix
