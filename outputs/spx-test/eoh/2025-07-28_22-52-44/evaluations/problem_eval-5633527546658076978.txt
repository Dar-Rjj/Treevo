def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = 1 / (df['close'] ** (1/2))
    weighted_hl_diff_ema = (hl_diff * weights).ewm(span=20, adjust=False).mean()
    sma_close = df['close'].rolling(window=20).mean()
    heuristics_matrix = weighted_hl_diff_ema / sma_close
    return heuristics_matrix
