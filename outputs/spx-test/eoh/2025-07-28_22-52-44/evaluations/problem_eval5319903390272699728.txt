def heuristics_v2(df):
    # Calculate the squared difference between consecutive days' close prices
    sq_diff = df['close'].diff() ** 2
    # Sum the squared differences and the squares of close prices over a 20-day rolling window
    sq_diff_sum = sq_diff.rolling(window=20).sum()
    close_sq_sum = (df['close'] ** 2).rolling(window=20).sum()
    # Calculate the product of the sums of the squares of close prices and then take its square root
    sqrt_close_sq_product = (close_sq_sum).apply(lambda x: x ** 0.5)
    # Calculate the ratio of the sum of squared differences to the square root of the product of the sums of the squares of close prices
    heuristics_matrix = sq_diff_sum / sqrt_close_sq_product
    return heuristics_matrix
