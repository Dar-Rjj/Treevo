import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in the close price
    pct_change = df['close'].pct_change()
    # Calculate the 7-day exponentially weighted moving average of the percentage change
    ewma_pct_change = pct_change.ewm(span=7).mean()
    # Calculate the 7-day exponentially weighted moving average of the volume
    ewma_volume = df['volume'].ewm(span=7).mean()
    # Combine the two EWMA series with a weight of 0.5 for each
    heuristics_matrix = (ewma_pct_change + ewma_volume) / 2
    return heuristics_matrix
