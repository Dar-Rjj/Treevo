def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' close prices
    abs_diff = df['close'].diff().abs()
    # Sum the absolute differences and the close prices over a 20-day rolling window
    abs_diff_sum = abs_diff.rolling(window=20).sum()
    close_sum = df['close'].rolling(window=20).sum()
    # Calculate the ratio of the sum of absolute differences to the sum of close prices
    heuristics_matrix = abs_diff_sum / close_sum
    return heuristics_matrix
