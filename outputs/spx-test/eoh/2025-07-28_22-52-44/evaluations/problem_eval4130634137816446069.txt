def heuristics_v2(df):
    # Calculate the ratio of (close - open) to (high - low)
    score_factor = (df['close'] - df['open']) / (df['high'] - df['low'])
    # Apply a 10-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = score_factor.ewm(span=10).mean()
    return heuristics_matrix
