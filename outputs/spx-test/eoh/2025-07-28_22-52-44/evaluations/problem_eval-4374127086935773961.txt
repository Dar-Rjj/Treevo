import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the log of the volume
    log_volume = np.log(df['volume'])
    # Calculate the difference between high and low
    hl_diff = df['high'] - df['low']
    # Calculate the rolling correlation over a 30-day window
    heuristics_matrix = hl_diff.rolling(window=30).corr(log_volume)
    return heuristics_matrix
