def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = 1 / (df['close'] ** (1/2))
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=20).sum()
    weighted_close_sum = (df['close'] * weights).rolling(window=20).sum()
    heuristics_matrix = weighted_hl_diff_sum / weighted_close_sum
    return heuristics_matrix
