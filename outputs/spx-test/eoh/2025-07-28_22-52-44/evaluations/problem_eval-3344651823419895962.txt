def heuristics_v2(df):
    # Calculate short-term (10 days) and long-term (30 days) VWAP
    vwap_short = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    vwap_long = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    
    # Calculate the VWPO
    vwpo = vwap_short - vwap_long
    
    heuristics_matrix = vwpo
    return heuristics_matrix
