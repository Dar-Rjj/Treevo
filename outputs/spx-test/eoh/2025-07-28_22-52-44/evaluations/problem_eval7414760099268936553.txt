def heuristics_v2(df):
    # Calculate the logarithmic difference between high and low prices
    log_diff = (df['high'] / df['low']).apply(np.log)
    # Sum the logarithmic differences and the close prices over a 30-day rolling window
    log_diff_sum = log_diff.rolling(window=30).sum()
    close_sum = df['close'].rolling(window=30).sum()
    # Calculate the ratio of the sum of logarithmic differences to the sum of close prices
    heuristics_matrix = log_diff_sum / close_sum
    return heuristics_matrix
