def heuristics_v2(df):
    # Calculate the standard deviation of daily closing prices over a 20-day window as a measure of price volatility
    df['price_volatility'] = df['close'].rolling(window=20).std()
    
    # Calculate the ratio of current volume to the average volume over the last 10 days to identify volume spikes
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_spike'] = df['volume'] / df['avg_volume_10d']
    
    # Compute the heuristic factor by multiplying price volatility with the volume spike indicator
    heuristics_matrix = df['price_volatility'] * df['volume_spike']
    
    return heuristics_matrix
