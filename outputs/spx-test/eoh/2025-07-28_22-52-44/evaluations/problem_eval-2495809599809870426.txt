def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' high and low prices
    abs_diff = (df['high'] - df['low']).diff().abs()
    # Sum the absolute differences and the volume over a 10-day rolling window
    abs_diff_sum = abs_diff.rolling(window=10).sum()
    volume_sum = df['volume'].rolling(window=10).sum()
    # Calculate the ratio of the sum of absolute differences to the sum of volume
    heuristics_matrix = abs_diff_sum / volume_sum
    return heuristics_matrix
