def heuristics_v2(df):
    # Calculate the daily price change
    price_change = df['close'] - df['open']
    # Weight the price change by the volume
    volume_weighted_change = price_change * df['volume']
    # Sum the volume-weighted price changes over a 20-day rolling window
    heuristics_matrix = volume_weighted_change.rolling(window=20).sum()
