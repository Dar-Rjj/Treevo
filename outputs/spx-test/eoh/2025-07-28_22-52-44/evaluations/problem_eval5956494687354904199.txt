import pandas as pd

def heuristics_v2(df):
    # Calculate the range (high - low) and then divide by the open price
    volatility_factor = (df['high'] - df['low']) / df['open']
    # Apply a 10-day moving average to smooth out the factor
    heuristics_matrix = volatility_factor.rolling(window=10).mean()
    return heuristics_matrix
