def heuristics_v2(df):
    # Calculate Smoothed RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    smooth_rsi = rsi.rolling(window=5).mean()
    
    # Calculate Adjusted MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.rolling(window=9).mean()
    adjusted_macd = macd - signal
    
    # Custom Momentum Indicator: Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14)
    
    # Combine factors
    heuristics_matrix = 0.5 * (smooth_rsi - 50) + 0.3 * adjusted_macd + 0.2 * roc
    return heuristics_matrix
