import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the close and open prices, then divide by the high price
    price_difference_factor = (df['close'] - df['open']) / df['high']
    # Apply a 10-day exponential moving average to smooth out the factor
    heuristics_matrix = price_difference_factor.ewm(span=10, adjust=False).mean()
    return heuristics_matrix
