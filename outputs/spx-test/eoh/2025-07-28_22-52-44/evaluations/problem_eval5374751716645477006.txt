import numpy as np

def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = np.log(df['close'])
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=30).sum()
    weighted_close_sum = (df['close'] * weights).rolling(window=30).sum()
    heuristics_matrix = weighted_hl_diff_sum / weighted_close_sum
    return heuristics_matrix
