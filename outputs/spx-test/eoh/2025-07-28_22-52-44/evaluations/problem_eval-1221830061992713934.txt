def heuristics_v2(df):
    # Sum the volume over a 15-day rolling window
    volume_sum = df['volume'].rolling(window=15).sum()
    # Sum the close prices over the same 15-day rolling window
    close_sum = df['close'].rolling(window=15).sum()
    # Calculate the ratio of the sum of volumes to the sum of close prices
    heuristics_matrix = volume_sum / close_sum
    return heuristics_matrix
