def heuristics_v2(df):
    # Calculate the difference between open and close prices
    oc_diff = df['open'] - df['close']
    # Calculate the 10-day moving average of the (open - close) differences
    oc_ma = oc_diff.rolling(window=10).mean()
    # Calculate the (high - low) range
    hl_range = df['high'] - df['low']
    # Calculate the 10-day standard deviation of the (high - low) range
    hl_std = hl_range.rolling(window=10).std()
    # Compute the heuristic factor as the ratio of the moving average of (open - close) differences to the standard deviation of (high - low) range
    heuristics_matrix = oc_ma / hl_std
    return heuristics_matrix
