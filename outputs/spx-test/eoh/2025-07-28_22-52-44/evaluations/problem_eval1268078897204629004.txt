def heuristics_v2(df):
    # Calculate the 5-day moving average of the closing price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the adaptive momentum as the difference between the current close and the 5-day MA
    adaptive_momentum = df['close'] - ma_5
    
    # Calculate the volatility-adjusted volume factor
    volatility_adjusted_volume = (df['high'] - df['low']) * df['volume']
    
    # Combine the factors into a single heuristic score for each day
    heuristics_matrix = adaptive_momentum + volatility_adjusted_volume
    
    return heuristics_matrix
