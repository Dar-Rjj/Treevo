def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = 1 / df['close']
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=10).sum()
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = weighted_hl_diff_sum / ema_close
    return heuristics_matrix
