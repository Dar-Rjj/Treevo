def heuristics_v2(df):
    # Calculate the 60-day moving average of the close prices
    ma_60 = df['close'].rolling(window=60).mean()
    # Calculate the absolute difference between daily close price and the 60-day moving average
    abs_diff = (df['close'] - ma_60).abs()
    # Sum the absolute differences over a 30-day rolling window
    heuristics_matrix = abs_diff.rolling(window=30).sum()
    return heuristics_matrix
