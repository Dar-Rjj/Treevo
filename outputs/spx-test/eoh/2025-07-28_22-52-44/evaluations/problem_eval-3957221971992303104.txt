import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Calculate the inverse of the square root of the close price as the weight
    weights = 1 / np.sqrt(df['close'])
    # Weighted sum of the high-low differences using exponential moving average over a 20-day window
    weighted_hl_diff_ema = (hl_diff * weights).ewm(span=20, adjust=False).mean()
    # Weighted sum of the close prices using exponential moving average over the same 20-day window
    weighted_close_ema = (df['close'] * weights).ewm(span=20, adjust=False).mean()
    # Calculate the ratio of the weighted exponential moving average of high-low differences to the weighted EMA of close prices
    heuristics_matrix = weighted_hl_diff_ema / weighted_close_ema
    return heuristics_matrix
