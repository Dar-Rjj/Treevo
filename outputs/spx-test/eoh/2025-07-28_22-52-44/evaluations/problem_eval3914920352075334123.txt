def heuristics_v2(df):
    # Calculate the absolute difference between close and open prices for each day
    co_diff = (df['close'] - df['open']).abs()
    # Sum the close-open differences over a 20-day rolling window
    co_diff_sum = co_diff.rolling(window=20).sum()
    # Sum the volume over the same 20-day rolling window
    vol_sum = df['volume'].rolling(window=20).sum()
    # Calculate the ratio of the sum of close-open differences to the sum of volume
    heuristics_matrix = co_diff_sum / vol_sum
    return heuristics_matrix
