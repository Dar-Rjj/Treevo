import pandas as pd

def heuristics_v2(df):
    # Calculate the range (high - low) and then divide by the close price
    volatility_factor = (df['high'] - df['low']) / df['close']
    # Apply an exponential weighted moving average to emphasize recent data
    heuristics_matrix = volatility_factor.ewm(span=5, adjust=False).mean()
    return heuristics_matrix
