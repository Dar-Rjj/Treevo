def heuristics_v2(df):
    # Calculate the 20-day moving average of the closing prices
    ma_20 = df['close'].rolling(window=20).mean()
    # Calculate the absolute difference between the closing prices and their 20-day moving average
    abs_diff_ma = (df['close'] - ma_20).abs()
    # Calculate the average of these absolute differences over a 20-day rolling window
    heuristics_matrix = abs_diff_ma.rolling(window=20).mean()
    return heuristics_matrix
