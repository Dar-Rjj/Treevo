import pandas as pd

def heuristics_v2(df):
    weights = [1/len(df.columns)] * len(df.columns)  # Assuming equal initial importance
    corr_matrix = df.rolling(window=30).corr()
    heuristics_matrix = corr_matrix.groupby(level=0).apply(lambda x: (x.values * weights).sum())
    return heuristics_matrix
