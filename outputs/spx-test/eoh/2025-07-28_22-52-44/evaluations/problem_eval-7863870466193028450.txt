import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of the absolute difference between close and open prices to the average of high and low prices
    volatility_factor = abs(df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
    # Apply a 14-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = volatility_factor.ewm(span=14).mean()
    return heuristics_matrix
