import pandas as pd

def heuristics_v2(df):
    # Calculate the closing price divided by the average of high and low prices
    stability_factor = df['close'] / ((df['high'] + df['low']) / 2)
    # Apply a 10-day exponentially weighted moving average
    heuristics_matrix = stability_factor.ewm(span=10).mean()
    return heuristics_matrix
