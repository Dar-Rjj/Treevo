def heuristics_v2(df):
    df['MACD_Price'] = df['close'].ewm(span=12).mean() - df['close'].ewm(span=26).mean()
    df['MACD_Volume'] = df['volume'].ewm(span=12).mean() - df['volume'].ewm(span=26).mean()
    weighted_macd = df['MACD_Price'] * 0.7 + df['MACD_Volume'] * 0.3
    smoothed_result = weighted_macd.ewm(span=9).mean()
    heuristics_matrix = smoothed_result
    return heuristics_matrix
