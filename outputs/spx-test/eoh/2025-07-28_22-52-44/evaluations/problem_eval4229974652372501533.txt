def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' high and low prices
    abs_diff = (df['high'] - df['low']).abs()
    # Sum the absolute differences and the close prices over a 30-day rolling window
    abs_diff_sum = abs_diff.rolling(window=30).sum()
    close_sum = df['close'].rolling(window=30).sum()
    # Calculate the ratio of the sum of absolute differences to the sum of close prices
    heuristics_matrix = abs_diff_sum / close_sum
    return heuristics_matrix
