def heuristics_v2(df):
    # Calculate the difference between high and low prices as volatility measure
    daily_volatility = df['high'] - df['low']
    
    # Calculate the ratio of close price to open price as a momentum indicator
    close_open_ratio = df['close'] / df['open']
    
    # Identify days where the closing price is below the opening price (bearish day)
    bearish_day = (df['close'] < df['open']).astype(int)
    
    # Compute the change in volume compared to the previous day to gauge interest
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Combine the factors into a single heuristic score for each day
    heuristics_matrix = (daily_volatility + close_open_ratio + bearish_day * 10 + volume_change) / 4
    
    return heuristics_matrix
