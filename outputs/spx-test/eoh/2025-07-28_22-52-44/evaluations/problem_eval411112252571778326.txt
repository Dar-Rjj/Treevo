def heuristics_v2(df):
    # Calculate the 20-day rate of change in the closing price
    roc_20 = df['close'].pct_change(periods=20)
    
    # Calculate the 10-day average volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Ratio of 10-day average volume to the high price
    avg_vol_high_ratio = avg_volume_10 / df['high']
    
    # Combine the factors
    heuristics_matrix = 0.7 * roc_20 + 0.3 * avg_vol_high_ratio
    return heuristics_matrix
