import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day EMA of the volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    # Calculate the 20-day SMA of the close prices
    sma_close = df['close'].rolling(window=20).mean()
    # Calculate the ratio of the 5-day EMA of the volume to the 20-day SMA of the close prices
    heuristics_matrix = ema_volume / sma_close
    return heuristics_matrix
