def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' high and low prices
    abs_diff = (df['high'] - df['low']).abs()
    # Define a linearly decreasing weight for a 20-day rolling window
    weights = np.arange(1, 21)[::-1]
    # Apply the weights on both abs_diff and close prices
    abs_diff_weighted = abs_diff.rolling(window=20).apply(lambda x: (x * weights).sum(), raw=True)
    close_weighted = df['close'].rolling(window=20).apply(lambda x: (x * weights).sum(), raw=True)
    # Calculate the ratio of the weighted sum of absolute differences to the weighted sum of close prices
    heuristics_matrix = abs_diff_weighted / close_weighted
    return heuristics_matrix
