def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()
    df['avg_price_10d'] = df['close'].rolling(window=10).mean()
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['heuristic_factor'] = (df['price_change'] * df['avg_volume_10d']) + (df['volume_change'] * df['avg_price_10d'])
    heuristics_matrix = df['heuristic_factor']
    return heuristics_matrix
