def heuristics_v2(df):
    # Calculate the 5-day exponentially weighted moving average of the closing price
    ewma_5 = df['close'].ewm(span=5).mean()
    
    # Calculate the adaptive momentum as the difference between the current close and the 5-day EWMA
    adaptive_momentum = df['close'] - ewma_5
    
    # Calculate the volatility-adjusted volume factor
    volatility_adjusted_volume = (df['high'] - df['low']) * df['volume']
    
    # Combine the factors into a single heuristic score for each day
    heuristics_matrix = adaptive_momentum + volatility_adjusted_volume
    
    return heuristics_matrix
