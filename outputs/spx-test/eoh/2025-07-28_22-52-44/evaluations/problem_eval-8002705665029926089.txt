import pandas as pd

def heuristics_v2(df):
    # Calculate the difference (close - open) and then divide by the average volume
    price_change_factor = (df['close'] - df['open']) / df['volume']
    # Apply a 15-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = price_change_factor.ewm(span=15).mean()
    return heuristics_matrix
