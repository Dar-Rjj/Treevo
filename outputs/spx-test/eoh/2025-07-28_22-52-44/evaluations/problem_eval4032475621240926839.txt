import numpy as np

def heuristics_v2(df):
    # Calculate the logarithmic difference between consecutive days' close prices
    log_diff = np.log(df['close']).diff().abs()
    # Sum the logarithmic differences and the close prices over a 20-day rolling window
    log_diff_sum = log_diff.rolling(window=20).sum()
    close_sum = df['close'].rolling(window=20).sum()
    # Calculate the ratio of the sum of logarithmic differences to the sum of close prices
    heuristics_matrix = log_diff_sum / close_sum
    return heuristics_matrix
