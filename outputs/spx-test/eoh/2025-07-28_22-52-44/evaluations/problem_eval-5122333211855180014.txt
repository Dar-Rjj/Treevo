import pandas as pd

def heuristics_v2(df):
    # Calculate the difference (close - open) and then divide by the average of high and low
    momentum_factor = (df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
    # Apply a 7-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = momentum_factor.ewm(span=7).mean()
    return heuristics_matrix
