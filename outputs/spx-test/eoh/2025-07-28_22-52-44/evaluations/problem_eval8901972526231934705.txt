import pandas as pd

def heuristics_v2(df):
    # Calculate the range (high - low) and then divide by the close price
    volatility_factor = (df['high'] - df['low']) / df['close']
    # Apply a 20-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = volatility_factor.ewm(span=20).mean()
    return heuristics_matrix
