import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily percentage change in closing price
    close_change = df['close'].pct_change()
    # Calculate the natural logarithm of the volume
    log_volume = np.log(df['volume'])
    # Compute the product of the two to get the VWCC factor
    heuristics_matrix = close_change * log_volume
    return heuristics_matrix
