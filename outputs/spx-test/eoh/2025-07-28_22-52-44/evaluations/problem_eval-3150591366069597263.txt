def heuristics_v2(df):
    # Calculate the rate of change in the close price
    roc = df['close'].pct_change(periods=10)
    
    # Calculate the oscillator based on the difference between high and low
    oscillator = (df['high'] - df['low']) / df['close']
    
    # Calculate the accumulation/distribution line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line = money_flow_volume.cumsum()
    
    # Combine the factors with new weights
    heuristics_matrix = 0.4 * roc + 0.3 * oscillator + 0.3 * ad_line.pct_change(periods=10)
    
    return heuristics_matrix
