import pandas as pd

def heuristics_v2(df):
    # Calculate the range (high - low) and then divide by the average of high and low
    volatility_factor = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2)
    # Apply a 7-day moving average to smooth out the factor
    heuristics_matrix = volatility_factor.rolling(window=7).mean()
    return heuristics_matrix
