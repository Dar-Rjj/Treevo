def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = 1 / (df['close'].apply(np.log))
    weighted_hl_diff_ema = (hl_diff * weights).ewm(span=30, adjust=False).mean()
    sma_close = df['close'].rolling(window=30).mean()
    heuristics_matrix = weighted_hl_diff_ema / sma_close
    return heuristics_matrix
