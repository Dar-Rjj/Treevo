def heuristics_v2(df):
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=14)
    
    # Calculate the volume change factor
    volume_change = df['volume'].pct_change(periods=28)
    
    # Combine the factors with new weights
    heuristics_matrix = 0.6 * momentum + 0.4 * volume_change
    
    return heuristics_matrix
