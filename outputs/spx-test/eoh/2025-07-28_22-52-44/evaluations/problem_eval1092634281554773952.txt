import pandas as pd

def heuristics_v2(df):
    # Calculate the range (high - low) and then divide by the average of open and close prices
    volatility_factor = (df['high'] - df['low']) / ((df['open'] + df['close']) / 2)
    # Apply a 10-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = volatility_factor.ewm(span=10).mean()
    return heuristics_matrix
