def heuristics_v2(df):
    co_diff = (df['close'] - df['open']).abs()
    co_diff_weighted = co_diff * df['volume']
    co_diff_sum = co_diff_weighted.rolling(window=10).sum()
    high_weighted = df['high'] * df['volume']
    high_sum = high_weighted.rolling(window=10).sum()
    heuristics_matrix = co_diff_sum / high_sum
    return heuristics_matrix
