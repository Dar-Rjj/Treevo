import numpy as np

def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Calculate the weight as the log of the close price
    weights = np.log(df['close'])
    # Weighted sum of the high-low differences over a 10-day rolling window
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=10).sum()
    # Weighted sum of the close prices over the same 10-day rolling window
    weighted_close_sum = (df['close'] * weights).rolling(window=10).sum()
    # Calculate the ratio of the weighted sum of high-low differences to the weighted sum of close prices
    heuristics_matrix = weighted_hl_diff_sum / weighted_close_sum
    return heuristics_matrix
