def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = np.log(df['close'])
    weighted_hl_diff_sum = (hl_diff * weights).rolling(window=10).sum()
    sma_close = df['close'].rolling(window=10).mean()
    heuristics_matrix = weighted_hl_diff_sum / sma_close
    return heuristics_matrix
