def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' close prices
    abs_diff = df['close'].diff().abs()
    # Sum the absolute differences and the absolute close prices over a 30-day rolling window
    abs_diff_sum = abs_diff.rolling(window=30).sum()
    close_abs_sum = df['close'].abs().rolling(window=30).sum()
    # Calculate the ratio of the sum of absolute differences to the sum of absolute close prices
    heuristics_matrix = abs_diff_sum / close_abs_sum
    return heuristics_matrix
