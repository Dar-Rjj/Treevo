import pandas as pd
import ta.momentum as momentum

def heuristics_v2(df):
    rsi = momentum.RSIIndicator(df['close'], window=14).rsi()
    rsi_volume_product = df['volume'] * rsi
    rsi_volume_sum = rsi_volume_product.rolling(window=30).sum()
    volume_sum = df['volume'].rolling(window=30).sum()
    heuristics_matrix = rsi_volume_sum / volume_sum
    return heuristics_matrix
