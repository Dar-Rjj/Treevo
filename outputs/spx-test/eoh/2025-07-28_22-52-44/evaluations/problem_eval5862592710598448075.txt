def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    # Calculate the ratio of volume to price difference
    volume_ratio = df['volume'] / price_diff
    # Sum the ratios over a 15-day rolling window
    heuristics_matrix = volume_ratio.rolling(window=15).sum()
    return heuristics_matrix
