def heuristics_v2(df):
    # Calculate the squared difference between consecutive days' close prices
    sq_diff = df['close'].diff().pow(2)
    # Sum the squared differences and the squared close prices over a 30-day rolling window
    sq_diff_sum = sq_diff.rolling(window=30).sum()
    close_sq_sum = df['close'].pow(2).rolling(window=30).sum()
    # Calculate the ratio of the sum of squared differences to the sum of squared close prices
    heuristics_matrix = sq_diff_sum / close_sq_sum
    return heuristics_matrix
