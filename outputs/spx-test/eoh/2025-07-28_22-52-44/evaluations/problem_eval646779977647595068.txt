def heuristics_v2(df):
    # Calculate the 50-day and 200-day moving averages of the close price
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the difference between the 50-day and 200-day moving averages
    ma_diff = ma_50 - ma_200
    
    # Calculate the 10-day rate of change (ROC) of volume
    roc_volume = df['volume'].pct_change(periods=10)
    
    # Combine the moving average difference with the ROC of volume
    heuristics_matrix = ma_diff + roc_volume
    
    return heuristics_matrix
