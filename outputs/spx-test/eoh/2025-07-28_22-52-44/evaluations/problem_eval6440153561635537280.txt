def heuristics_v2(df):
    # Calculate the 10-day simple moving average (SMA) of the close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    # Compute the volume-weighted average of high and low prices
    df['vwap'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    # Construct the heuristics matrix integrating both factors
    heuristics_matrix = (df['vwap'] - df['sma_10'])
    return heuristics_matrix
