import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of the closing price to the average of high and low
    price_relation_factor = df['close'] / ((df['high'] + df['low']) / 2)
    # Apply a 7-day exponential moving average to smooth out the factor
    heuristics_matrix = price_relation_factor.ewm(span=7).mean()
    return heuristics_matrix
