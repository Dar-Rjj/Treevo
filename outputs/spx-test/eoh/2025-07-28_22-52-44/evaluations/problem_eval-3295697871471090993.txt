def heuristics_v2(df):
    # Calculate the ratio of (high - open) to (close - low)
    volatility_factor = (df['high'] - df['open']) / (df['close'] - df['low'])
    # Apply a 15-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = volatility_factor.ewm(span=15).mean()
    return heuristics_matrix
