def heuristics_v2(df):
    def dynamic_range_momentum(row):
        return (row['high'] - row['low']) / row['close']
    
    # Apply the DRM calculation across each row
    drm_values = df.apply(dynamic_range_momentum, axis=1)
    
    # Smooth the DRM values with a 5-day moving average to create ADRM
    heuristics_matrix = drm_values.rolling(window=5).mean()
    
    return heuristics_matrix
