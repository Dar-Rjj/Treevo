def heuristics_v2(df):
    hl_diff = df['high'] - df['low']
    weights = 1 / (df['close'] ** (1/3))
    weighted_hl_diff_ema = (hl_diff * weights).ewm(span=10).mean()
    ema_close = df['close'].ewm(span=10).mean()
    heuristics_matrix = weighted_hl_diff_ema / ema_close
    return heuristics_matrix
