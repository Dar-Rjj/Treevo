import pandas as pd

def heuristics_v2(df):
    # Calculate the range (high - low) and then divide by the average of open and close prices
    volatility_factor = (df['high'] - df['low']) / ((df['open'] + df['close']) / 2)
    # Apply a 5-day cumulative sum to the factor
    cumulative_sum = volatility_factor.rolling(window=5).sum()
    # Apply a 3-day moving average to smooth out the cumulative sum
    heuristics_matrix = cumulative_sum.rolling(window=3).mean()
    return heuristics_matrix
