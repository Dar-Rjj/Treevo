import pandas as pd

def heuristics_v2(df):
    # Calculate the cumulative sum of daily trading volume
    cum_volume = df['volume'].cumsum()
    # Calculate the 30-day simple moving average of the closing price
    sma_30_close = df['close'].rolling(window=30).mean()
    # Generate the heuristic factor
    heuristics_matrix = cum_volume / sma_30_close
    return heuristics_matrix
