def heuristics_v2(df):
    # Adjust volume by the difference between close and open prices
    adjusted_volume = (df['close'] - df['open']) * df['volume']
    # Calculate the cumulative sum of the adjusted volume
    cumsum_adjusted_volume = adjusted_volume.cumsum()
    # Apply a 10-day simple moving average to smooth the series
    heuristics_matrix = cumsum_adjusted_volume.rolling(window=10).mean()
    return heuristics_matrix
