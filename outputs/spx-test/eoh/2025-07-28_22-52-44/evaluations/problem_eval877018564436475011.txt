def heuristics_v2(df):
    # Calculate the 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    # Compute the ratio of the close price to its 10-day moving average
    close_ma_ratio = df['close'] / ma_10
    # Calculate the cumulative product of this ratio
    heuristics_matrix = close_ma_ratio.cumprod()
    return heuristics_matrix
