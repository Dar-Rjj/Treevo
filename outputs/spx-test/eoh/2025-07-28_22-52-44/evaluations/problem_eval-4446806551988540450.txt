def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' close prices
    abs_diff = df['close'].diff().abs()
    # Calculate the weight as the average volume over a 20-day rolling window
    vol_avg = df['volume'].rolling(window=20).mean()
    # Multiply the absolute differences by the volume average to get the weighted differences
    weighted_abs_diff = abs_diff * vol_avg
    # Sum the weighted absolute differences and the close prices over a 20-day rolling window
    weighted_abs_diff_sum = weighted_abs_diff.rolling(window=20).sum()
    close_sum = df['close'].rolling(window=20).sum()
    # Calculate the ratio of the sum of weighted absolute differences to the sum of close prices
    heuristics_matrix = weighted_abs_diff_sum / close_sum
    return heuristics_matrix
