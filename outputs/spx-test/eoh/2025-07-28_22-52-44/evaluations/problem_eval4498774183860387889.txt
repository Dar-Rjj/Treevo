import pandas as pd

def heuristics_v2(df):
    # Momentum - 10 day return
    momentum = df['close'].pct_change(10)
    
    # Volatility - Standard deviation over the last 30 days
    volatility = df['close'].rolling(window=30).std()
    
    # Volume Shock - Percentage change in volume compared to the previous day
    volume_shock = df['volume'].pct_change()
    
    # Composite Alpha Factor: Equal weighting of the above components
    heuristics_matrix = (momentum + volatility + volume_shock) / 3
    
    return heuristics_matrix
