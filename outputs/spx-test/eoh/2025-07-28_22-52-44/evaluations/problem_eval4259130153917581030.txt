import pandas as pd

def heuristics_v2(df):
    # Calculate RSI with a 20-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate MACD with 9 and 21 EMAs
    ema_9 = df['close'].ewm(span=9, adjust=False).mean()
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    macd = ema_9 - ema_21
    signal = macd.rolling(window=9).mean()
    
    # Custom Momentum Indicator: Difference between close and VWAP over 10 days
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    vwap_10 = vwap.rolling(window=10).mean()
    momentum_indicator = df['close'] - vwap_10
    
    # Combine factors
    heuristics_matrix = 0.5 * (rsi - 50) + 0.3 * (macd - signal) + 0.2 * momentum_indicator
    return heuristics_matrix
