import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the log difference between high and low
    log_diff = np.log(df['high']) - np.log(df['low'])
    # Normalize by the average of open and close prices
    volatility_factor = log_diff / ((df['open'] + df['close']) / 2)
    # Apply a 10-day exponentially weighted moving average to smooth out the factor
    heuristics_matrix = volatility_factor.ewm(span=10).mean()
    return heuristics_matrix
