def heuristics_v2(df):
    # Calculate the logarithmic difference between consecutive days' close prices
    log_diff = np.log(df['close']).diff().abs()
    # Sum the logarithmic differences over a 30-day rolling window
    log_diff_sum = log_diff.rolling(window=30).sum()
    # Calculate the standard deviation of closing prices and the mean of the trading volume over a 30-day rolling window
    close_std = df['close'].rolling(window=30).std()
    volume_mean = df['volume'].rolling(window=30).mean()
    # Calculate the ratio of the sum of logarithmic differences to the product of the standard deviation and mean volume
    heuristics_matrix = log_diff_sum / (close_std * volume_mean)
    return heuristics_matrix
