def heuristics_v2(df):
    # Calculate the momentum factor with a 50-day lookback
    momentum = df['close'].pct_change(periods=50)
    
    # Calculate the volume change factor with a 14-day lookback
    volume_change = df['volume'].pct_change(periods=14)
    
    # Combine the factors with new weights
    heuristics_matrix = 0.8 * momentum + 0.2 * volume_change
    
    return heuristics_matrix
