def heuristics_v2(df):
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Calculate Volatility (Standard Deviation of Daily Returns)
    volatility = daily_returns.rolling(window=20).std()
    
    # Calculate Cumulative Difference Between Close and Open Prices
    cum_close_open_diff = (df['close'] - df['open']).cumsum()
    
    # Combine Factors
    heuristics_matrix = 0.6 * volatility + 0.4 * cum_close_open_diff
    return heuristics_matrix
