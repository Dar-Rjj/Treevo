def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate daily volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Create a factor based on the interaction of daily return and volume change
    df['interaction_factor'] = df['daily_return'] * df['volume_change']
    
    # Calculate 10-day momentum
    df['momentum_10d'] = df['close'].pct_change(10)
    
    # Combine the interaction factor with momentum
    df['heuristics_factor'] = df['interaction_factor'] + df['momentum_10d']
    
    # Return the heuristics factor as a Series
    heuristics_matrix = df['heuristics_factor'].dropna()
    return heuristics_matrix
