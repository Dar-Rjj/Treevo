def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Weighted Moving Average
    wma_short = df['close'].rolling(window=10).mean()
    wma_long = df['close'].rolling(window=30).mean()
    
    # Difference Between Short-Term and Long-Term WMA
    wma_diff_signal = wma_short - wma_long
    
    # Average Volume Over the Past 20 Days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Ratio of Close Price to Average Volume
    close_avg_volume_ratio = df['close'] / avg_volume_20
    
    # Combine Factors
    heuristics_matrix = 0.6 * wma_diff_signal + 0.4 * close_avg_volume_ratio
    return heuristics_matrix
