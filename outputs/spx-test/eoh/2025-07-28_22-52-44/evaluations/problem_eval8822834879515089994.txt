def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    # Calculate the product of the daily return and volume
    return_volume_product = daily_return * df['volume']
    # Calculate the cumulative sum over a 20-day rolling window
    heuristics_matrix = return_volume_product.rolling(window=20).sum()
    return heuristics_matrix
