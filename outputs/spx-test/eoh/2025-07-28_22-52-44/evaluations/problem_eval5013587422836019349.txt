def heuristics_v2(df):
    # Calculate the logarithmic difference between consecutive days' close prices
    log_diff = np.log(df['close']).diff().abs()
    # Weight the logarithmic differences by the volume
    weighted_log_diff = log_diff * df['volume']
    # Sum the weighted logarithmic differences and the close prices over a 20-day rolling window
    weighted_log_diff_sum = weighted_log_diff.rolling(window=20).sum()
    close_sum = df['close'].rolling(window=20).sum()
    # Calculate the ratio of the sum of weighted logarithmic differences to the sum of close prices
    heuristics_matrix = weighted_log_diff_sum / close_sum
    return heuristics_matrix
