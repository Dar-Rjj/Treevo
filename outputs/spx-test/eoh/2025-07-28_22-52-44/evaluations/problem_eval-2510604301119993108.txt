def heuristics_v2(df):
    # Calculate the difference between high and low prices for each day
    hl_diff = df['high'] - df['low']
    # Sum the high-low differences over a 20-day rolling window, weighted by volume
    hl_diff_sum = (hl_diff * df['volume']).rolling(window=20).sum()
    # Sum the close prices over the same 20-day rolling window, weighted by volume
    close_sum = (df['close'] * df['volume']).rolling(window=20).sum()
    # Calculate the ratio of the sum of high-low differences to the sum of close prices
    heuristics_matrix = hl_diff_sum / close_sum
    return heuristics_matrix
