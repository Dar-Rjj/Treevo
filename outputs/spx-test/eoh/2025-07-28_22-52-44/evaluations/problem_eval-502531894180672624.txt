def heuristics_v2(df):
    # Calculate the 14-day standard deviation of the close price as a measure of volatility
    volatility = df['close'].rolling(window=14).std()
    
    # Calculate the 14-day moving average of the close price
    ma_14 = df['close'].rolling(window=14).mean()
    
    # Calculate the relative strength of the close price compared to its moving average
    relative_strength = (df['close'] - ma_14) / ma_14
    
    # Combine the factors
    heuristics_matrix = 0.5 * volatility + 0.5 * relative_strength
    
    return heuristics_matrix
