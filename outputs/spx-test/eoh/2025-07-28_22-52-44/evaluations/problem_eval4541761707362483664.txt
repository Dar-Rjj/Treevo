def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' close prices
    abs_diff = df['close'].diff().abs()
    # Calculate the mean of absolute differences over a 20-day rolling window
    abs_diff_mean = abs_diff.rolling(window=20).mean()
    # Calculate the standard deviation of close prices over a 20-day rolling window
    close_std = df['close'].rolling(window=20).std()
    # Calculate the ratio of the mean of absolute differences to the standard deviation of close prices
    heuristics_matrix = abs_diff_mean / close_std
    return heuristics_matrix
