def heuristics_v2(df):
    # Calculate the absolute difference between consecutive days' close prices
    abs_diff = df['close'].diff().abs()
    # Calculate the weight (volume) for each day
    weights = df['volume']
    # Sum the weighted absolute differences and the sum of squared close prices over a 20-day rolling window
    weighted_abs_diff_sum = (abs_diff * weights).rolling(window=20).sum()
    close_squared_sum = (df['close'] ** 2).rolling(window=20).sum()
    # Calculate the ratio of the weighted sum of absolute differences to the sum of squared close prices
    heuristics_matrix = weighted_abs_diff_sum / close_squared_sum
    return heuristics_matrix
