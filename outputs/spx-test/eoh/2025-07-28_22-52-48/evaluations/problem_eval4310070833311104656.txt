def heuristics_v2(df):
    median_close_20 = df['close'].rolling(window=20).median()
    min_close_20 = df['close'].rolling(window=20).min()
    ratio_median = np.log(df['close'] / median_close_20)
    ratio_min = -np.sqrt(df['close'] / min_close_20)
    heuristics_matrix = (ratio_median + ratio_min).dropna()
    return heuristics_matrix
