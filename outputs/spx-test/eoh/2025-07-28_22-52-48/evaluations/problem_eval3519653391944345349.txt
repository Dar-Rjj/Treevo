def heuristics_v2(df):
    ema_close_20 = df['close'].ewm(span=20).mean()
    max_close_20 = df['close'].rolling(window=20).max()
    ratio_ema = df['close'] / ema_close_20
    log_diff_max = (df['close'] - max_close_20).apply(lambda x: np.log1p(x) if x >= 0 else -np.log1p(-x))
    heuristics_matrix = (ratio_ema + log_diff_max).dropna()
    return heuristics_matrix
