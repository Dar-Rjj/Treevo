def heuristics_v2(df):
    atr_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    cvw_close_50 = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    ratio_atr = df['close'] / atr_20
    ratio_cvw = -df['close'] / cvw_close_50
    heuristics_matrix = (ratio_atr + ratio_cvw).dropna()
    return heuristics_matrix
