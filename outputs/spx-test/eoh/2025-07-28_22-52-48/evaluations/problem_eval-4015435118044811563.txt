def heuristics_v2(df):
    def compute_roc(price_series, n=10):
        roc = ((price_series / price_series.shift(n)) - 1) * 100
        return roc

    def compute_atr(df, n=14):
        high_low = df['high'] - df['low']
        high_close = (df['high'] - df['close'].shift()).abs()
        low_close = (df['low'] - df['close'].shift()).abs()
        ranges = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
        atr = ranges.rolling(window=n).mean()
        return atr
    
    close_prices = df['close']
    roc_values = compute_roc(close_prices)
    atr_values = compute_atr(df)
    
    # Heuristic matrix: combination of ROC and ATR
    heuristics_matrix = (roc_values + atr_values)/2
    return heuristics_matrix
