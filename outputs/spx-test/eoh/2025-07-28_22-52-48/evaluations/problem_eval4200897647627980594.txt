def heuristics_v2(df):
    ma_close_30 = df['close'].rolling(window=30).mean()
    max_close_30 = df['close'].rolling(window=30).max()
    ratio_ma = df['close'] / ma_close_30
    log_ratio_max = -np.log(df['close'] / max_close_30)
    heuristics_matrix = (ratio_ma + log_ratio_max) / 2
    heuristics_matrix = heuristics_matrix.dropna()
    return heuristics_matrix
