def heuristics_v2(df):
    median_close_30 = df['close'].rolling(window=30).median()
    median_close_60 = df['close'].rolling(window=60).median()
    log_ratio_min = np.log(df['close'] / median_close_30)
    log_ratio_max = -np.log(df['close'] / median_close_60)
    heuristics_matrix = (log_ratio_min + log_ratio_max).dropna()
    return heuristics_matrix
