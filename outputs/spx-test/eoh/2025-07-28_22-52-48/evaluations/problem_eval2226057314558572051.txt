import numpy as np

def heuristics_v2(df):
    avg_close_30 = df['close'].rolling(window=30).mean()
    median_close_30 = df['close'].rolling(window=30).median()
    ratio_avg = df['close'] / avg_close_30
    log_ratio_median = np.log(df['close'] / median_close_30)
    heuristics_matrix = (ratio_avg + log_ratio_median).dropna()
    return heuristics_matrix
