def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) for close prices
    roc_10 = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day standard deviation of daily log returns
    returns = df['close'].pct_change()
    std_dev_30 = returns.rolling(window=30).std()
    
    # Define weights for ROC and standard deviation
    weight_roc = 0.6
    weight_std_dev = 0.4
    
    # Create the heuristic factor
    heuristics_matrix = (weight_roc * roc_10) + (weight_std_dev * std_dev_30)
    
    return heuristics_matrix
