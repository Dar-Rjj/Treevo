import pandas as pd

def heuristics_v2(df):
    def compute_rsi(close_series, periods=14):
        delta = close_series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=periods).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=periods).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    def compute_wma(close_series, periods=30):
        weights = pd.Series(range(1, periods + 1))
        wma = close_series.rolling(window=periods).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
        return wma

    close_prices = df['close']

    rsi_values = compute_rsi(close_prices)
    wma_values = compute_wma(close_prices)

    heuristics_matrix = (rsi_values + wma_values) / 2
    return heuristics_matrix
