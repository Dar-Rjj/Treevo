def heuristics_v2(df):
    median_close_10 = df['close'].rolling(window=10).median()
    iqr_close_10 = df['close'].rolling(window=10).quantile(0.75) - df['close'].rolling(window=10).quantile(0.25)
    heuristic_val = (np.log(df['close']) - np.log(median_close_10)) / iqr_close_10
    heuristics_matrix = heuristic_val.dropna()
    return heuristics_matrix
