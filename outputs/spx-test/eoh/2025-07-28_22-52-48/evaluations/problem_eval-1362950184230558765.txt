def heuristics_v2(df):
    ema_close_20 = df['close'].ewm(span=20).mean()
    max_close_20 = df['close'].rolling(window=20).max()
    min_close_20 = df['close'].rolling(window=20).min()
    ratio_ema = df['close'] / ema_close_20
    diff_max_min = max_close_20 - min_close_20
    heuristics_matrix = (ratio_ema + diff_max_min).dropna()
    return heuristics_matrix
