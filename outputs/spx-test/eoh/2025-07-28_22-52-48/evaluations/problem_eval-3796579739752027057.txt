import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day Exponential Moving Average (EMA) of the 'volume'
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the percentage change in the 'close' price
    pct_change_close = df['close'].pct_change(periods=1)
    
    # Define weights for the combination
    weight_ema_volume = 0.6
    weight_pct_change = 0.4
    
    # Create the heuristics matrix
    heuristics_matrix = (weight_ema_volume * ema_10_volume) + (weight_pct_change * pct_change_close)
    
    return heuristics_matrix
