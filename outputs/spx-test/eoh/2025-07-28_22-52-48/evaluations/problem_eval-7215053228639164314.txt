def heuristics_v2(df):
    min_close_20 = df['close'].rolling(window=20).min()
    max_close_20 = df['close'].rolling(window=20).max()
    log_ratio_min = (df['close'] / min_close_20).apply(np.log)
    log_ratio_max = (df['close'] / max_close_20).apply(np.log)
    heuristics_matrix = (log_ratio_min - log_ratio_max).dropna()
    return heuristics_matrix
