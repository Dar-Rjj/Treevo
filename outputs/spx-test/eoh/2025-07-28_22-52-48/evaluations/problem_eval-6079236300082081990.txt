def heuristics_v2(df):
    avg_close_14 = df['close'].rolling(window=14).mean()
    min_close_14 = df['close'].rolling(window=14).min()
    vol_change_14 = (df['volume'] - df['volume'].shift(14)) / df['volume'].shift(14)
    heuristics_matrix = (avg_close_14 / min_close_14) - vol_change_14
    return heuristics_matrix
