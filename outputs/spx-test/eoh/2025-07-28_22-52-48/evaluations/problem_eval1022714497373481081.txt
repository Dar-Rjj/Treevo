def heuristics_v2(df):
    # Calculate short-term (5-day), medium-term (10-day), and long-term (20-day) moving averages
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Compute the differences between each pair of moving averages
    diff_ma_5_10 = ma_5 - ma_10
    diff_ma_10_20 = ma_10 - ma_20
    
    # Define new weights for the combination
    weight_ma_5 = 0.3
    weight_ma_10 = 0.2
    weight_ma_20 = 0.2
    weight_diff_5_10 = 0.15
    weight_diff_10_20 = 0.15
    
    # Create the heuristic factor
    heuristics_matrix = (weight_ma_5 * ma_5) + (weight_ma_10 * ma_10) + (weight_ma_20 * ma_20) + \
                        (weight_diff_5_10 * diff_ma_5_10) + (weight_diff_10_20 * diff_ma_10_20)
    
    return heuristics_matrix
