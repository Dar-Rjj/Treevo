def heuristics_v2(df):
    min_close_20 = df['close'].rolling(window=20).min()
    max_close_20 = df['close'].rolling(window=20).max()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    ratio_min = df['close'] / min_close_20
    ratio_max = -df['close'] / max_close_20
    volume_ratio = df['volume'] / avg_volume_20
    heuristics_matrix = (ratio_min + ratio_max) * volume_ratio
    return heuristics_matrix
