def heuristics_v2(df):
    ema_20_price = df['close'].ewm(span=20, adjust=False).mean()
    ema_50_price = df['close'].ewm(span=50, adjust=False).mean()
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = ((ema_20_price - ema_50_price) * (df['volume'] / ema_20_volume)).dropna()
    return heuristics_matrix
