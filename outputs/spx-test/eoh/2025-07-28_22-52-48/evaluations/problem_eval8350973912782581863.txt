def heuristics_v2(df):
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    diff_ma = ma_5 - ma_20
    prod_ma = ma_5 * ma_20
    weight_prod = 0.5
    weight_diff = 0.5
    heuristics_matrix = (weight_prod * prod_ma) + (weight_diff * diff_ma)
    return heuristics_matrix
