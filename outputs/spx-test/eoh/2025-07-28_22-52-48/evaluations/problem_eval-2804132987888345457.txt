def heuristics_v2(df):
    def compute_high_low_diff(series_high, series_low, window=14):
        high_max = series_high.rolling(window=window).max()
        low_min = series_low.rolling(window=window).min()
        diff = high_max - low_min
        return diff

    vol_mean = df['volume'].rolling(window=14).mean()
    high_low_diff = compute_high_low_diff(df['high'], df['low'])
    heuristics_matrix = (high_low_diff / vol_mean).dropna()
    return heuristics_matrix
