def heuristics_v2(df):
    sma_12 = df['close'].rolling(window=12).mean()
    sma_26 = df['close'].rolling(window=26).mean()
    sma_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / sma_volume_20
    heuristics_matrix = ((sma_12 - sma_26) * volume_ratio).dropna()
    return heuristics_matrix
