def heuristics_v2(df):
    min_close_10 = df['close'].rolling(window=10).min()
    max_close_30 = df['close'].rolling(window=30).max()
    ratio_min = df['close'] / min_close_10
    ratio_max = -df['close'] / max_close_30
    heuristics_matrix = (ratio_min + ratio_max).dropna()
    return heuristics_matrix
