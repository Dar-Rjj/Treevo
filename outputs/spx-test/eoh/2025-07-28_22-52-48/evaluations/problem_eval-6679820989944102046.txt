def heuristics_v2(df):
    min_close_30 = df['close'].rolling(window=30).min()
    max_close_30 = df['close'].rolling(window=30).max()
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    ratio_min = (df['close'] / min_close_30) ** 3
    ratio_max = - (df['close'] / max_close_30) ** 3
    heuristics_matrix = (ratio_min + ratio_max) * np.log(avg_volume_10)
    return heuristics_matrix
