def heuristics_v2(df):
    min_close_20 = df['close'].rolling(window=20).min()
    max_close_20 = df['close'].rolling(window=20).max()
    std_close_20 = df['close'].rolling(window=20).std()
    ratio_min = df['close'] / min_close_20
    ratio_max = df['close'] / max_close_20
    avg_ratio = (ratio_min + ratio_max) / 2
    std_ratio = df['close'] / std_close_20
    heuristics_matrix = avg_ratio - std_ratio
    return heuristics_matrix
