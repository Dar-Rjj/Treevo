import pandas as pd
import numpy as np

def heuristics_v2(df):
    median_close_10 = df['close'].rolling(window=10).median()
    max_close_30 = df['close'].rolling(window=30).max()
    ratio_median = np.log(df['close'] / median_close_10)
    ratio_max = -np.sqrt(df['close'] / max_close_30)
    heuristics_matrix = (ratio_median + ratio_max).dropna()
    return heuristics_matrix
