def heuristics_v2(df):
    median_high_low_20 = (df['high'].rolling(window=20).median() + df['low'].rolling(window=20).median()) / 2
    mad_close_20 = df['close'].rolling(window=20).apply(lambda x: np.fabs(x - x.mean()).mean(), raw=True)
    heuristic_val = (median_high_low_20 - np.log(df['close'])) / mad_close_20
    heuristics_matrix = heuristic_val.dropna()
    return heuristics_matrix
