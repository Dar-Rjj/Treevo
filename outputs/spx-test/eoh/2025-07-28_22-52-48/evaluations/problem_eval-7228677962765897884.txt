def heuristics_v2(df):
    avg_high_low_60 = (df['high'].rolling(window=60).mean() + df['low'].rolling(window=60).mean()) / 2
    std_close_60 = df['close'].rolling(window=60).std()
    heuristic_val = (avg_high_low_60 - np.log(df['close'])) / std_close_60
    heuristics_matrix = heuristic_val.dropna()
    return heuristics_matrix
