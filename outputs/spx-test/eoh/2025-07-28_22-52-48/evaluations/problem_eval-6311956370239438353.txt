def heuristics_v2(df):
    med_close_10 = df['close'].rolling(window=10).median()
    med_close_50 = df['close'].rolling(window=50).median()
    ratio_med_10 = (df['close'] / med_close_10).apply(np.log)
    ratio_med_50 = - (df['close'] / med_close_50).apply(np.log)
    heuristics_matrix = (ratio_med_10 + ratio_med_50).dropna()
    return heuristics_matrix
