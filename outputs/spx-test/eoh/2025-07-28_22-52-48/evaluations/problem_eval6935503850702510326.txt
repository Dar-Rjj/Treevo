def heuristics_v2(df):
    sma_close_20 = df['close'].rolling(window=20).mean()
    min_close_20 = df['close'].rolling(window=20).min()
    ratio_sma = df['close'] / sma_close_20
    ratio_min = -df['close'] / min_close_20
    heuristics_matrix = (ratio_sma + ratio_min).dropna()
    return heuristics_matrix
