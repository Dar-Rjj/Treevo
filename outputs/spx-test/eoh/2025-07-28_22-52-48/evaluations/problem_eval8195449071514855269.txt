def heuristics_v2(df):
    close_prices = df['close']
    sma_50 = close_prices.rolling(window=50).mean()
    ratio_sma = close_prices / sma_50
    daily_returns = close_prices.pct_change()
    std_dev_14 = daily_returns.rolling(window=14).std()
    log_std_dev = -np.log(std_dev_14)
    heuristics_matrix = (ratio_sma + log_std_dev).dropna()
    return heuristics_matrix
