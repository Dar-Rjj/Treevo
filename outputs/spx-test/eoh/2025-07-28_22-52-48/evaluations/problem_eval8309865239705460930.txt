def heuristics_v2(df):
    log_returns = np.log(df['close'] / df['close'].shift(1))
    weights = 1 / df['volume']
    normalized_weights = weights / weights.rolling(window=10).sum()
    heuristics_matrix = (log_returns.rolling(window=10) * normalized_weights.rolling(window=10)).sum().dropna()
    return heuristics_matrix
