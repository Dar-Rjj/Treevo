def heuristics_v2(df):
    min_close_60 = df['close'].rolling(window=60).min()
    max_close_60 = df['close'].rolling(window=60).max()
    ratio_min = (df['close'] / min_close_60) ** 3
    ratio_max = - (df['close'] / max_close_60) ** 3
    heuristics_matrix = (ratio_min + ratio_max).dropna()
    return heuristics_matrix
