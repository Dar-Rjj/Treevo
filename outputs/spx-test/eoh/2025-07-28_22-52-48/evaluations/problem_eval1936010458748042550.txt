def heuristics_v2(df):
    avg_high_low_20 = (df['high'].rolling(window=20).mean() + df['low'].rolling(window=20).mean()) / 2
    std_close_20 = df['close'].rolling(window=20).std()
    heuristic_val = (avg_high_low_20 - np.log(df['close'])) / std_close_20
    heuristics_matrix = heuristic_val.dropna()
    return heuristics_matrix
