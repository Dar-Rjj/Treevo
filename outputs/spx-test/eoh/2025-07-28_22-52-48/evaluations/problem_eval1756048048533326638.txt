def heuristics_v2(df):
    ema_close_20 = df['close'].ewm(span=20).mean()
    max_close_20 = df['close'].rolling(window=20).max()
    ratio_ema = df['close'] / ema_close_20
    ratio_max = -df['close'] / max_close_20
    heuristics_matrix = (ratio_ema + ratio_max).dropna()
    return heuristics_matrix
