def heuristics_v2(df):
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_close_50 = df['close'].ewm(span=50, adjust=False).mean()
    ratio_ema_20 = df['close'] / ema_close_20
    ratio_ema_50 = -df['close'] / ema_close_50
    heuristics_matrix = (ratio_ema_20 + ratio_ema_50).dropna()
    return heuristics_matrix
