def heuristics_v2(df):
    def compute_obv(close_prices, volumes):
        obv = [0]
        for i in range(1, len(close_prices)):
            if close_prices[i] > close_prices[i-1]:
                obv.append(obv[-1] + volumes[i])
            elif close_prices[i] < close_prices[i-1]:
                obv.append(obv[-1] - volumes[i])
            else:
                obv.append(obv[-1])
        return pd.Series(obv, index=close_prices.index)

    def compute_rate_of_change(price_series, periods=1):
        roc = (price_series.diff(periods) / price_series.shift(periods)) * 100
        return roc

    close_prices = df['close']
    volumes = df['volume']
    
    obv_values = compute_obv(close_prices, volumes)
    roc_values = compute_rate_of_change(close_prices)
    
    heuristics_matrix = obv_values.diff() * roc_values
    return heuristics_matrix
