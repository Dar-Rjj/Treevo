def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    std_10 = df['daily_return'].rolling(window=10).std()
    ma_high_10 = df['high'].rolling(window=10).mean()
    log_diff = (df['close'] / ma_high_10).apply(np.log)
    heuristics_matrix = (std_10 + log_diff).dropna()
    return heuristics_matrix
