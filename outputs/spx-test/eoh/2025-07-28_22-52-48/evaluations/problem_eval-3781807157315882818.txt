def heuristics_v2(df):
    sma_close_10 = df['close'].rolling(window=10).mean()
    max_high_10 = df['high'].rolling(window=10).max()
    diff_sma_close = sma_close_10 - df['close']
    ratio_high_sma = df['close'] / max_high_10
    heuristics_matrix = (diff_sma_close + ratio_high_sma).dropna()
    return heuristics_matrix
