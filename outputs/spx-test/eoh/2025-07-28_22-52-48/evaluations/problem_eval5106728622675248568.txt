def heuristics_v2(df):
    ema_close_30 = df['close'].ewm(span=30).mean()
    max_close_30 = df['close'].rolling(window=30).max()
    ratio_ema = df['close'] / ema_close_30
    ratio_max = -((df['close'] / max_close_30) ** 0.5)
    heuristics_matrix = (ratio_ema + ratio_max).dropna()
    return heuristics_matrix
