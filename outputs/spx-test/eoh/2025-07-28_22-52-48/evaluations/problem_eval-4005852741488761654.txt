import pandas as pd

def heuristics_v2(df):
    # Calculate short-term (5-day) and long-term (20-day) moving averages
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Compute the difference between short-term and long-term moving averages
    diff_ma = ma_5 - ma_20
    
    # Define weights for the combination of moving averages and their differences
    weight_ma_5 = 0.4
    weight_ma_20 = 0.3
    weight_diff = 0.3
    
    # Create the heuristic factor
    heuristics_matrix = (weight_ma_5 * ma_5) + (weight_ma_20 * ma_20) + (weight_diff * diff_ma)
    
    return heuristics_matrix
