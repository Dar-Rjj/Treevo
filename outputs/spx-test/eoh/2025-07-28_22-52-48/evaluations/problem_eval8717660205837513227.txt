import pandas as pd

def heuristics_v2(df):
    short_window_high = 7
    long_window_high = 28
    short_window_low = 9
    long_window_low = 30
    df['short_high_ma'] = df['high'].rolling(window=short_window_high).mean()
    df['long_high_ma'] = df['high'].rolling(window=long_window_high).mean()
    df['short_low_ma'] = df['low'].rolling(window=short_window_low).mean()
    df['long_low_ma'] = df['low'].rolling(window=long_window_low).mean()
    heuristics_matrix = (df['short_high_ma'] - df['long_high_ma']) * 1.5 - (df['short_low_ma'] - df['long_low_ma'])
    return heuristics_matrix
