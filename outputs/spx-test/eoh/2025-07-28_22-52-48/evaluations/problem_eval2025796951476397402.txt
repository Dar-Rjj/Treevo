def heuristics_v2(df):
    std_close_30 = df['close'].rolling(window=30).std()
    max_close_30 = df['close'].rolling(window=30).max()
    min_close_30 = df['close'].rolling(window=30).min()
    range_diff = max_close_30 - min_close_30
    ratio_std = df['close'] / std_close_30
    heuristics_matrix = (range_diff + ratio_std).dropna()
    return heuristics_matrix
