def heuristics_v2(df):
    std_close_60 = df['close'].rolling(window=60).std()
    avg_volume_50 = df['volume'].rolling(window=50).mean()
    ratio_close = df['close'] / std_close_60
    ratio_volume = df['volume'] / avg_volume_50
    heuristics_matrix = (ratio_close + ratio_volume).dropna()
    return heuristics_matrix
