def heuristics_v2(df):
    avg_high_low_10 = (df['high'].rolling(window=10).mean() + df['low'].rolling(window=10).mean()) / 2
    std_close_10 = df['close'].rolling(window=10).std()
    heuristic_val = (avg_high_low_10 - df['close'] ** (1/3)) / std_close_10
    heuristics_matrix = heuristic_val.dropna()
    return heuristics_matrix
