def heuristics_v2(df):
    min_close_30 = df['close'].rolling(window=30).min()
    max_close_30 = df['close'].rolling(window=30).max()
    ratio_min = df['close'] / min_close_30
    ratio_max = -df['close'] / max_close_30
    heuristics_matrix = 0.6 * ratio_min + 0.4 * ratio_max
    return heuristics_matrix
