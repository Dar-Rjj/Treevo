def heuristics_v2(df):
    # Calculate short-term and long-term moving averages for close prices
    ma_short = df['close'].rolling(window=5).mean()
    ma_long = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the short-term and long-term MAs
    ma_diff = ma_short - ma_long
    
    # Calculate the percentage change in volume over a 5-day period
    vol_change = df['volume'].pct_change(periods=5)
    
    # Generate a heuristic by combining MA difference and volume change
    heuristics_matrix = (ma_diff * vol_change).fillna(0)
    
    return heuristics_matrix
