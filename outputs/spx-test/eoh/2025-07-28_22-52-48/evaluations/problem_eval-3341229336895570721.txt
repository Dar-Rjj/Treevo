def heuristics_v2(df):
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    diff_ma = ma_5 - ma_20
    roc_ma_5 = ma_5.pct_change()
    
    weight_ma_5 = 0.6
    weight_ma_20 = 0.2
    weight_diff = 0.1
    weight_roc = 0.1
    
    heuristics_matrix = (weight_ma_5 * ma_5) + (weight_ma_20 * ma_20) + (weight_diff * diff_ma) + (weight_roc * roc_ma_5)
    
    return heuristics_matrix
