import pandas as pd

def heuristics_v2(df):
    # Calculate short-term (5-day) and long-term (20-day) exponential moving averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the momentum of the short-term EMA
    momentum_ema_5 = ema_5 - ema_5.shift(1)
    
    # Define weights for the combination of EMAs and their momentum
    weight_ema_5 = 0.4
    weight_ema_20 = 0.3
    weight_momentum = 0.3
    
    # Create the heuristic factor
    heuristics_matrix = (weight_ema_5 * ema_5) + (weight_ema_20 * ema_20) + (weight_momentum * momentum_ema_5)
    
    return heuristics_matrix
