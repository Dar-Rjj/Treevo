import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['return'] = df['close'].pct_change()
    
    # Calculate short-term (5-day) and long-term (20-day) EMA of returns
    ema_5_return = df['return'].ewm(span=5, adjust=False).mean()
    ema_20_return = df['return'].ewm(span=20, adjust=False).mean()
    
    # Compute the rate of change of the short-term EMA
    roc_ema_5 = ema_5_return.pct_change()
    
    # Define weights for the combination of EMAs and their rate of change
    weight_ema_5 = 0.5
    weight_ema_20 = 0.3
    weight_roc_ema_5 = 0.2
    
    # Create the heuristic factor
    heuristics_matrix = (weight_ema_5 * ema_5_return) + (weight_ema_20 * ema_20_return) + (weight_roc_ema_5 * roc_ema_5)
    
    return heuristics_matrix
