def heuristics_v2(df):
    short_window = 5
    long_window = 20
    df['short_close_ma'] = df['close'].rolling(window=short_window).mean()
    df['long_close_ma'] = df['close'].rolling(window=long_window).mean()
    df['volume_strength'] = df['volume'] / df['volume'].rolling(window=long_window).mean()
    heuristics_matrix = (df['short_close_ma'] - df['long_close_ma']) * df['volume_strength']
    return heuristics_matrix
