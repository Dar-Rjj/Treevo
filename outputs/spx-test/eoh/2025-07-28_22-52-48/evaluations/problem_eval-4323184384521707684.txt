import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    df['short_high_ma'] = df['high'].rolling(window=short_window).mean()
    df['long_high_ma'] = df['high'].rolling(window=long_window).mean()
    df['short_low_ma'] = df['low'].rolling(window=short_window).mean()
    df['long_low_ma'] = df['low'].rolling(window=long_window).mean()
    heuristics_matrix = (df['short_high_ma'] - df['long_high_ma']) - (df['short_low_ma'] - df['long_low_ma'])
    return heuristics_matrix
