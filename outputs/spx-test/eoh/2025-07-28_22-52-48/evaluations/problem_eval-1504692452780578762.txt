def heuristics_v2(df):
    min_close_40 = df['close'].rolling(window=40).min()
    max_close_40 = df['close'].rolling(window=40).max()
    ratio_min = (df['close'] / min_close_40) ** 0.5
    ratio_max = - (df['close'] / max_close_40) ** 0.5
    heuristics_matrix = (ratio_min + ratio_max).dropna()
    return heuristics_matrix
