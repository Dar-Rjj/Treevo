import pandas as pd

def heuristics_v2(df):
    def compute_rs(close_series, periods=14):
        gain = (close_series.diff(1) > 0).astype(int)
        loss = (close_series.diff(1) < 0).astype(int)
        avg_gain = gain.rolling(window=periods, min_periods=periods).mean()
        avg_loss = loss.rolling(window=periods, min_periods=periods).mean()
        rs = avg_gain / avg_loss
        return rs

    def compute_mfi(high, low, close, volume, periods=14):
        typical_price = (high + low + close) / 3
        money_flow = typical_price * volume
        positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
        negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
        mfi_ratio = positive_money_flow.rolling(window=periods, min_periods=periods).sum() / \
                    negative_money_flow.rolling(window=periods, min_periods=periods).sum()
        mfi = 100 - (100 / (1 + mfi_ratio))
        return mfi

    close_prices = df['close']
    high_prices = df['high']
    low_prices = df['low']
    volumes = df['volume']

    rs_values = compute_rs(close_prices)
    mfi_values = compute_mfi(high_prices, low_prices, close_prices, volumes)

    heuristics_matrix = (rs_values + mfi_values) / 2
    return heuristics_matrix
