import pandas as pd

def heuristics_v2(df):
    ma_10 = df['close'].rolling(window=10).mean()
    ma_30 = df['close'].rolling(window=30).mean()
    diff_ma = ma_10 - ma_30
    roc_ma_10 = ma_10.pct_change()
    
    weight_ma_10 = 0.5
    weight_ma_30 = 0.2
    weight_diff = 0.2
    weight_roc = 0.1
    
    heuristics_matrix = (weight_ma_10 * ma_10) + (weight_ma_30 * ma_30) + (weight_diff * diff_ma) + (weight_roc * roc_ma_10)
    
    return heuristics_matrix
