def heuristics_v2(df):
    close_prices = df['close']
    
    # Calculate the 50-day Simple Moving Average (SMA)
    sma_50 = close_prices.rolling(window=50).mean()
    
    # Calculate the log returns
    log_returns = np.log(close_prices / close_prices.shift(1))
    
    # Calculate the 20-day standard deviation of the log returns
    std_log_returns = log_returns.rolling(window=20).std()
    
    # Heuristic matrix: combination of SMA and standard deviation of log returns
    heuristics_matrix = (sma_50 + std_log_returns).dropna()
    
    return heuristics_matrix
