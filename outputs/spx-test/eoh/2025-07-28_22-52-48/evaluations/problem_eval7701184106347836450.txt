def heuristics_v2(df):
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    ratio_volume = df['volume'] / avg_volume_30
    
    # Calculate the slope using linear regression on the last 10 days' close prices
    def calculate_slope(series):
        x = np.arange(len(series))
        y = series.values
        return np.polyfit(x, y, 1)[0]
    
    slope = df['close'].rolling(window=10).apply(calculate_slope)
    
    heuristics_matrix = (ratio_volume + slope).dropna()
    return heuristics_matrix
