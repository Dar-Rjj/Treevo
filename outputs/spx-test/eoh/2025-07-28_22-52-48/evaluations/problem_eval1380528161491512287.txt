def heuristics_v2(df):
    avg_high_low_30 = (df['high'].rolling(window=30).mean() + df['low'].rolling(window=30).mean()) / 2
    std_close_30 = df['close'].rolling(window=30).std()
    heuristic_val = (avg_high_low_30 - df['close'] ** 0.5) / std_close_30
    heuristics_matrix = heuristic_val.dropna()
    return heuristics_matrix
