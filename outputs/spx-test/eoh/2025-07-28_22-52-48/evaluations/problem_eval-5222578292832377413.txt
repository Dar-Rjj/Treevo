def heuristics_v2(df):
    high_low_diff = df['high'] - df['low']
    roc_high_low = high_low_diff.pct_change()
    vol_change = df['volume'].pct_change()
    log_return = np.log(df['close']).diff()
    
    weight_roc = 0.5
    weight_vol = 0.3
    weight_log = 0.2
    
    heuristics_matrix = (weight_roc * roc_high_low) + (weight_vol * vol_change) + (weight_log * log_return)
    return heuristics_matrix
