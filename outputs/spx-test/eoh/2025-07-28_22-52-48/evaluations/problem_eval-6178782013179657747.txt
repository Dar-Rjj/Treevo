import pandas as pd

def heuristics_v2(df):
    avg_close_20 = df['close'].rolling(window=20).mean()
    ewm_avg_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    ratio_avg = df['close'] / avg_close_20
    ratio_ewma = df['close'] / ewm_avg_close_20
    heuristics_matrix = (ratio_avg + ratio_ewma) / 2
    return heuristics_matrix
