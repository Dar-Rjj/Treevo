def heuristics_v2(df):
    ema_20_price = df['close'].ewm(span=20, adjust=False).mean()
    ema_50_price = df['close'].ewm(span=50, adjust=False).mean()
    roc_10_volume = df['volume'].pct_change(periods=10)
    heuristics_matrix = (ema_20_price / ema_50_price) * (1 + roc_10_volume).dropna()
    return heuristics_matrix
