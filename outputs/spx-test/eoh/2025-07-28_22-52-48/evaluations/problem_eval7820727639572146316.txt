def heuristics_v2(df):
    ma_close_30 = df['close'].rolling(window=30).mean()
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    ratio_ma = df['close'] / ma_close_30
    diff_volume = df['volume'] - avg_volume_30
    heuristics_matrix = (ratio_ma + diff_volume).dropna()
    return heuristics_matrix
