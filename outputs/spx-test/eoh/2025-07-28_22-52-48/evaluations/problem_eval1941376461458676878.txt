import pandas as pd

def heuristics_v2(df):
    short_span = 5
    long_span = 20
    df['short_high_ewma'] = df['high'].ewm(span=short_span, adjust=False).mean()
    df['long_high_ewma'] = df['high'].ewm(span=long_span, adjust=False).mean()
    df['short_low_ewma'] = df['low'].ewm(span=short_span, adjust=False).mean()
    df['long_low_ewma'] = df['low'].ewm(span=long_span, adjust=False).mean()
    heuristics_matrix = (df['short_high_ewma'] - df['long_high_ewma']) - (df['short_low_ewma'] - df['long_low_ewma'])
    return heuristics_matrix
