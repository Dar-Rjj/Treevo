def heuristics_v2(df):
    ma_close_20 = df['close'].rolling(window=20).mean()
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    ratio_ma = df['close'] / ma_close_20
    ratio_ema = -df['close'] / ema_close_20
    heuristics_matrix = (ratio_ma + ratio_ema) / 2
    return heuristics_matrix
