def heuristics_v2(df):
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_50_close = df['close'].ewm(span=50, adjust=False).mean()
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (ema_20_close - ema_50_close) * ema_20_volume
    return heuristics_matrix
