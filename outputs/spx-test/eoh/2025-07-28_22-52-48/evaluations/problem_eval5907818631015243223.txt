import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    df['short_high_ema'] = df['high'].ewm(span=short_window, adjust=False).mean()
    df['long_high_ema'] = df['high'].ewm(span=long_window, adjust=False).mean()
    df['short_low_ema'] = df['low'].ewm(span=short_window, adjust=False).mean()
    df['long_low_ema'] = df['low'].ewm(span=long_window, adjust=False).mean()
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=long_window).mean()
    heuristics_matrix = ((df['short_high_ema'] - df['long_high_ema']) - (df['short_low_ema'] - df['long_low_ema'])) * df['volume_ratio']
    return heuristics_matrix
