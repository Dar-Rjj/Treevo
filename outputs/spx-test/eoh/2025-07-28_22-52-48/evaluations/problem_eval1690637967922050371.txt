def heuristics_v2(df):
    short_window_close = 5
    long_window_close = 20
    short_window_volume = 6
    long_window_volume = 30
    df['short_close_ma'] = df['close'].rolling(window=short_window_close).mean()
    df['long_close_ma'] = df['close'].rolling(window=long_window_close).mean()
    df['short_volume_ma'] = df['volume'].rolling(window=short_window_volume).mean()
    df['long_volume_ma'] = df['volume'].rolling(window=long_window_volume).mean()
    heuristics_matrix = (df['short_close_ma'] - df['long_close_ma']) * 1.5 + (df['short_volume_ma'] - df['long_volume_ma'])
    return heuristics_matrix
