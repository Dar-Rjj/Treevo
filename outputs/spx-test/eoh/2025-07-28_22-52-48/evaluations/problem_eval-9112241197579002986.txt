def heuristics_v2(df):
    sma_close_30 = df['close'].rolling(window=30).mean()
    max_close_30 = df['close'].rolling(window=30).max()
    ratio_sma = df['close'] / sma_close_30
    ratio_max = -df['close'] / max_close_30
    heuristics_matrix = (ratio_sma + ratio_max).dropna()
    return heuristics_matrix
