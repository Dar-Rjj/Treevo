import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low for each day
    price_diff = df['high'] - df['low']
    # Calculate the historical volatility (standard deviation) of the close price over a 30-day window
    volatility = df['close'].rolling(window=30).std()
    # Compute the weighted sum of price difference and volume, using the inverse of volatility as weight
    weighted_sum = (price_diff + df['volume']) / (volatility + 1e-6)  # Adding small constant to avoid division by zero
    heuristics_matrix = pd.Series(weighted_sum, index=df.index)
    return heuristics_matrix
