def heuristics_v2(df):
    mean_close_30 = df['close'].rolling(window=30).mean()
    std_close_30 = df['close'].rolling(window=30).std()
    ratio_mean = df['close'] / mean_close_30
    inverse_std = 1 / std_close_30
    heuristics_matrix = 0.7 * ratio_mean + 0.3 * inverse_std
    return heuristics_matrix
