import pandas as pd

def heuristics_v2(df):
    # Calculate short-term (5-day) and long-term (20-day) exponential moving averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the rate of change in volume over a 5-day period
    roc_vol = df['volume'].pct_change(periods=5)
    
    # Generate the heuristic factor by combining EMA difference and ROC of volume
    heuristics_matrix = (ema_5 - ema_20) * roc_vol.fillna(0)
    
    return heuristics_matrix
