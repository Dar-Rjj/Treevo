def heuristics_v2(df):
    sma_close_20 = df['close'].rolling(window=20).mean()
    max_close_20 = df['close'].rolling(window=20).max()
    ratio_sma = df['close'] / sma_close_20
    log_ratio_max = -np.log(df['close'] / max_close_20)
    heuristics_matrix = (ratio_sma + log_ratio_max).dropna()
    return heuristics_matrix
