def heuristics_v2(df):
    min_close_5 = df['close'].rolling(window=5).min()
    max_close_5 = df['close'].rolling(window=5).max()
    min_close_20 = df['close'].rolling(window=20).min()
    max_close_20 = df['close'].rolling(window=20).max()
    diff_min = min_close_5 - min_close_20
    diff_max = max_close_5 - max_close_20
    sum_diff = diff_min + diff_max
    heuristics_matrix = df['close'] / sum_diff
    return heuristics_matrix
