import pandas as pd

def heuristics_v2(df):
    df['price_diff'] = df['close'] - df['open']
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ma_10']
    price_diff_ma = df['price_diff'].rolling(window=5).mean()
    
    # Heuristic matrix: combination of price difference mean, and volume ratio
    heuristics_matrix = price_diff_ma * df['volume_ratio']
    return heuristics_matrix
