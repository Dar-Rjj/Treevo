def heuristics_v2(df):
    ma_close_30 = df['close'].rolling(window=30).mean()
    ma_close_5 = df['close'].rolling(window=5).mean()
    ma_close_20 = df['close'].rolling(window=20).mean()
    ratio_ma = df['close'] / ma_close_30
    momentum = ma_close_5 - ma_close_20
    heuristics_matrix = (ratio_ma * momentum).dropna()
    return heuristics_matrix
