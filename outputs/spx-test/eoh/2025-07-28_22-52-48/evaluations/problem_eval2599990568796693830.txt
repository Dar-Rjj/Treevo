import pandas as pd

def heuristics_v2(df):
    def compute_percentage_change(price_series, periods=14):
        pct_change = (price_series - price_series.shift(periods)) / price_series.shift(periods)
        return pct_change
    
    def compute_volume_ratio(volume_series, periods=14):
        avg_volume = volume_series.rolling(window=periods).mean()
        volume_ratio = volume_series / avg_volume
        return volume_ratio
    
    def compute_volatility(price_series, periods=14):
        daily_returns = price_series.pct_change()
        volatility = daily_returns.rolling(window=periods).std()
        return volatility
    
    close_prices = df['close']
    volume = df['volume']
    
    percentage_change = compute_percentage_change(close_prices)
    volume_ratio = compute_volume_ratio(volume)
    volatility = compute_volatility(close_prices)
    
    # Heuristic matrix: combination of percentage change, volume ratio, and volatility
    heuristics_matrix = (percentage_change + volume_ratio + volatility) / 3
    return heuristics_matrix
