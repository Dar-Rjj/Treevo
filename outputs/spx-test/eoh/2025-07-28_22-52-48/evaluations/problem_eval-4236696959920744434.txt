import pandas as pd

def heuristics_v2(df):
    # Calculate the True Range
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    
    # Calculate the 10-day Average True Range (ATR)
    atr_10 = df['tr'].rolling(window=10).mean()
    
    # Calculate the 20-day moving average of the volume
    ma_20_volume = df['volume'].rolling(window=20).mean()
    
    # Create the heuristic factor
    heuristics_matrix = (atr_10 / df['close']) * ma_20_volume
    
    return heuristics_matrix
