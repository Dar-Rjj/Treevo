def heuristics_v2(df):
    def compute_roc(series, n=10):
        return series.pct_change(n)
    
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    roc_volume = compute_roc(df['volume'])
    heuristic_val = roc_volume + (df['close'] - ema_50)
    heuristics_matrix = heuristic_val.dropna()
    return heuristics_matrix
