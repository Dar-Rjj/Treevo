def heuristics_v2(df):
    mean_close_20 = df['close'].rolling(window=20).mean()
    std_close_20 = df['close'].rolling(window=20).std()
    ratio_mean = df['close'] / mean_close_20
    ratio_std = -df['close'] / std_close_20
    heuristics_matrix = (ratio_mean + ratio_std) / 2
    return heuristics_matrix
