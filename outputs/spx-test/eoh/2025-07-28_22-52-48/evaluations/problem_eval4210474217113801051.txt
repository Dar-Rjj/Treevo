import numpy as np

def heuristics_v2(df):
    median_close_30 = df['close'].rolling(window=30).median()
    log_ratio = np.log(df['close'] / median_close_30)
    log_returns = np.log(df['close']).diff().fillna(0)
    sma_log_returns_5 = log_returns.rolling(window=5).mean()
    heuristics_matrix = log_ratio * np.sign(sma_log_returns_5).dropna()
    return heuristics_matrix
