def heuristics_v2(df):
    min_close_10 = df['close'].rolling(window=10).min()
    max_close_10 = df['close'].rolling(window=10).max()
    ratio_min = (df['close'] / min_close_10) ** 4
    ratio_max = - (df['close'] / max_close_10) ** 4
    heuristics_matrix = (ratio_min + ratio_max).dropna()
    return heuristics_matrix
