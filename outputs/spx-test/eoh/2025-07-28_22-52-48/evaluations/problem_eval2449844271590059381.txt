def heuristics_v2(df):
    avg_close_20 = df['close'].rolling(window=20).mean()
    median_close_20 = df['close'].rolling(window=20).median()
    exp_diff_avg = (df['close'] - avg_close_20).apply(np.exp)
    exp_diff_median = -(df['close'] - median_close_20).apply(np.exp)
    heuristics_matrix = (exp_diff_avg + exp_diff_median).dropna()
    return heuristics_matrix
