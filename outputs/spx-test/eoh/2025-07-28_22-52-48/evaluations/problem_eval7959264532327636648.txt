import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 5-day rolling standard deviation of volume
    df['volatility'] = df['volume'].rolling(window=5).std()
    
    # Combine the daily return and volatility into a single heuristic
    df['heuristic'] = df['daily_return'] * df['volatility']
    
    # Apply a 3-day weighted moving average to smooth the heuristic
    weights = [0.5, 0.3, 0.2]
    df['heuristics_matrix'] = df['heuristic'].rolling(window=3).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Return the heuristics matrix
    return heuristics_matrix
