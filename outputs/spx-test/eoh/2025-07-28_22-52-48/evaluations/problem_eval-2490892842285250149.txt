    # Calculate simple moving averages
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate relative strength index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(delta > 0, 0).fillna(0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(delta < 0, 0).fillna(0).rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Construct the heuristic matrix with weights
    heuristics_matrix = 0.4 * (sma_5 - sma_10) + 0.3 * rsi + 0.3 * mfi
    
    return heuristics_matrix
