def heuristics_v2(df):
    min_close_30 = df['close'].rolling(window=30).min()
    max_close_30 = df['close'].rolling(window=30).max()
    ema_close_30 = df['close'].ewm(span=30, adjust=False).mean()
    ratio_min = df['close'] / min_close_30
    ratio_max = -df['close'] / max_close_30
    heuristics_matrix = (ratio_min + ratio_max + ema_close_30).dropna()
    return heuristics_matrix
