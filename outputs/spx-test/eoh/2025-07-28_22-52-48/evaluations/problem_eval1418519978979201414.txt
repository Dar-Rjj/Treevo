def heuristics_v2(df):
    median_close_30 = df['close'].rolling(window=30).median()
    mean_close_30 = df['close'].rolling(window=30).mean()
    ratio_median = df['close'] / median_close_30
    ratio_mean = -df['close'] / mean_close_30
    heuristics_matrix = (ratio_median + ratio_mean).dropna()
    return heuristics_matrix
