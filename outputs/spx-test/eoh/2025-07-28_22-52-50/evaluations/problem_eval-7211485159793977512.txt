def heuristics_v2(df):
    std_returns_50_day = df['close'].pct_change().rolling(window=50).std()
    ema_close_50_day = df['close'].ewm(span=50, adjust=False).mean()
    high_low_diff_50_day = (df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()) / ema_close_50_day
    heuristics_matrix = (std_returns_50_day / ema_close_50_day) * (1 + high_low_diff_50_day)
    return heuristics_matrix
