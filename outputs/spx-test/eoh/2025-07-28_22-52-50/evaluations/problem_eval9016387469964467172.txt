def heuristics_v2(df):
    ema_close_20_day = df['close'].ewm(span=20, adjust=False).mean()
    volume_ratio_20_day = df['volume'] / df['volume'].rolling(window=20).median()
    open_close_diff_20_day = (df['open'] - df['close']) / df['close']
    heuristics_matrix = ema_close_20_day * (1 + volume_ratio_20_day + open_close_diff_20_day)
    return heuristics_matrix
