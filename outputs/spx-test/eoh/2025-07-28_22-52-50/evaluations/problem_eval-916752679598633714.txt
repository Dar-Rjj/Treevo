def heuristics_v2(df):
    log_high_low_20_day = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()).apply(lambda x: np.log(x) if x > 0 else 0)
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    heuristics_matrix = log_high_low_20_day * avg_volume_20_day
    return heuristics_matrix
