import pandas as pd

def heuristics_v2(df):
    # Calculate moving averages
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the returns of each MA over the last 10 periods
    ma_5_ret = (ma_5 - ma_5.shift(10)) / ma_5.shift(10)
    ma_10_ret = (ma_10 - ma_10.shift(10)) / ma_10.shift(10)
    ma_20_ret = (ma_20 - ma_20.shift(10)) / ma_20.shift(10)
    
    # Determine weights based on the returns of each MA
    total_ret = ma_5_ret + ma_10_ret + ma_20_ret
    weight_5 = ma_5_ret / total_ret
    weight_10 = ma_10_ret / total_ret
    weight_20 = ma_20_ret / total_ret
    
    # Combine the MAs using the calculated weights
    combined_ma = (ma_5 * weight_5) + (ma_10 * weight_10) + (ma_20 * weight_20)
    
    # Calculate recent price volatility
    recent_volatility = df['close'].pct_change().rolling(window=5).std()
    
    # Apply a simple heuristic adjustment based on recent price volatility
    heuristics_matrix = combined_ma + (recent_volatility * 0.5)
    
    return pd.Series(heuristics_matrix, index=df.index)
```
