def heuristics_v2(df):
    std_close_10_day = df['close'].rolling(window=10).std()
    mean_close_10_day = df['close'].rolling(window=10).mean()
    max_high_10_day = df['high'].rolling(window=10).max()
    min_low_10_day = df['low'].rolling(window=10).min()
    heuristics_matrix = (std_close_10_day / mean_close_10_day) / (max_high_10_day - min_low_10_day)
    return heuristics_matrix
