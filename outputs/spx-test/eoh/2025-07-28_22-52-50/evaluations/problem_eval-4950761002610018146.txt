import pandas as pd

def heuristics_v2(df):
    # Calculate RSI
    rsi_period = 14
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_period).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate OBV
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    obv -= (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Calculate standard deviation of closing prices
    std_period = 20
    std = df['close'].rolling(window=std_period).std()
    
    # Combined heuristic: RSI, OBV, and standard deviation
    heuristics_matrix = rsi + obv + std
    
    return heuristics_matrix
