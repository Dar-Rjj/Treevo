def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    mean_close_20_day = df['close'].rolling(window=20).mean()
    ema_high_low_diff_20_day = ((df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / mean_close_20_day).ewm(span=20, adjust=False).mean()
    atr_20_day = df[['high', 'low', 'close']].rolling(window=20).apply(lambda x: np.mean([np.max(x) - np.min(y) for y in [x[i:i+2] for i in range(len(x)-1)]]), raw=False)
    heuristics_matrix = (np.log(std_close_20_day / mean_close_20_day)) * (1 + ema_high_low_diff_20_day) * (1 + atr_20_day / mean_close_20_day)
    return heuristics_matrix
