import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day Simple Moving Average (SMA)
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Compute the ratio of the closing price to its 20-day SMA
    close_to_sma_ratio = df['close'] / sma_20
    
    # Calculate the 10-day cumulative sum of the difference between high and low prices
    hi_lo_diff_cumsum = (df['high'] - df['low']).rolling(window=10).sum()
    
    # Composite factor
    heuristics_matrix = close_to_sma_ratio + hi_lo_diff_cumsum
    
    return heuristics_matrix
