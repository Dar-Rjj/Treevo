import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low for each day
    range_diff = df['high'] - df['low']
    
    # Compute the ratio of volume to the daily price range
    vol_to_range_ratio = df['volume'] / range_diff
    
    # Calculate the Exponential Moving Average (EMA) of the computed ratio
    ema_factor = vol_to_range_ratio.ewm(span=10, adjust=False).mean()
    
    # Return the EMA factor as a Series
    heuristics_matrix = pd.Series(ema_factor, name='Heuristics_Factor')
    return heuristics_matrix
