def heuristics_v2(df):
    std_close_30_day = df['close'].rolling(window=30).std()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    high_low_diff_30_day = (df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()) / sma_close_30_day
    heuristics_matrix = (std_close_30_day / sma_close_30_day) * (1 + high_low_diff_30_day)
    return heuristics_matrix
