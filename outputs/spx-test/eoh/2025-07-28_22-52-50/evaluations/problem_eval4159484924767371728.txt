def heuristics_v2(df):
    momentum_20_day = df['close'] / df['close'].shift(20) - 1
    atr_20_day = df['high'] - df['low']
    atr_20_day = atr_20_day.rolling(window=20).mean()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    heuristics_matrix = momentum_20_day * (atr_20_day / sma_close_20_day)
    return heuristics_matrix
