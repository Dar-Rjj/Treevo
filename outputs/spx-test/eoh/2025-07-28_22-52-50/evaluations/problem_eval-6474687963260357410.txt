import pandas as pd

def heuristics_v2(df):
    # Calculate RSI
    rsi_period = 14
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_period).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))

    # Calculate PPO
    short_ema_period = 12
    long_ema_period = 26
    df['EMA_Short'] = df['close'].ewm(span=short_ema_period, adjust=False).mean()
    df['EMA_Long'] = df['close'].ewm(span=long_ema_period, adjust=False).mean()
    ppo_line = (df['EMA_Short'] - df['EMA_Long']) / df['EMA_Long'] * 100

    # Calculate the ratio of the highest high to the lowest low over the last 20 days
    high_low_ratio = df['high'].rolling(window=20).max() / df['low'].rolling(window=20).min()

    # Combined heuristic: RSI, PPO, and high/low ratio
    heuristics_matrix = df['RSI'] + ppo_line + high_low_ratio

    return heuristics_matrix
