def heuristics_v2(df):
    short_window = 10
    long_window = 30
    df['momentum'] = df['close'].pct_change(periods=short_window)
    df['avg_volume'] = df['volume'].rolling(window=long_window).mean()
    df['relative_strength'] = df['close'] / df['avg_volume']
    df['heuristic_factor'] = df['momentum'] * df['relative_strength']
    heuristics_matrix = df['heuristic_factor'].dropna()
    return heuristics_matrix
