import pandas as pd

def heuristics_v2(df):
    open_prices = df['open']
    close_prices = df['close']
    volume = df['volume']
    
    # Calculate the short-term (5 days) and long-term (20 days) moving averages for open and close prices
    short_open_ma = open_prices.rolling(window=5).mean()
    long_open_ma = open_prices.rolling(window=20).mean()
    short_close_ma = close_prices.rolling(window=5).mean()
    long_close_ma = close_prices.rolling(window=20).mean()
    
    # Momentum indicator: 14-day rate of change (ROC) of volume
    volume_momentum = (volume / volume.shift(14)) - 1
    
    # Combined heuristic factor
    heuristics_matrix = 0.5 * ((short_open_ma - long_open_ma) + (short_close_ma - long_close_ma)) + 0.5 * volume_momentum
    
    return heuristics_matrix
