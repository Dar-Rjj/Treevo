import numpy as np
import pandas as pd

def heuristics_v2(df):
    std_close_30_day = df['close'].rolling(window=30).std()
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    high_low_diff_30_day = (df['close'].rolling(window=30).max() - df['close'].rolling(window=30).min()) / ema_close_30_day
    heuristics_matrix = (std_close_30_day / ema_close_30_day) * np.log(1 + high_low_diff_30_day)
    return heuristics_matrix
