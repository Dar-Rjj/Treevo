def heuristics_v2(df):
    std_close_50_day = df['close'].rolling(window=50).std()
    sma_close_50_day = df['close'].rolling(window=50).mean()
    high_low_diff_50_day = (df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()) / sma_close_50_day
    heuristics_matrix = (std_close_50_day / sma_close_50_day) * np.sqrt(1 + high_low_diff_50_day)
    return heuristics_matrix
