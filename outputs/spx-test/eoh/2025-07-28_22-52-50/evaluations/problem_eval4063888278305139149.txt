def heuristics_v2(df):
    std_close_10_day = df['close'].rolling(window=10).std()
    mean_close_10_day = df['close'].rolling(window=10).mean()
    sma_high_low_diff_15_day = ((df['high'].rolling(window=15).max() - df['low'].rolling(window=15).min()) / mean_close_10_day).rolling(window=15).mean()
    heuristics_matrix = (std_close_10_day / mean_close_10_day) * (1 + sma_high_low_diff_15_day)
    return heuristics_matrix
