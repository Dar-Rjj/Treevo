def heuristics_v2(df):
    vwap_30_day = (df['close'] * df['volume']).rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns_30_day = log_returns.rolling(window=30).std()
    heuristics_matrix = (df['close'] - vwap_30_day) / std_log_returns_30_day
    return heuristics_matrix
