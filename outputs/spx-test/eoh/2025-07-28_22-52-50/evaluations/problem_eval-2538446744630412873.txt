def heuristics_v2(df):
    # Calculate 10-day standard deviation of daily returns
    df['Return'] = df['close'].pct_change()
    std_returns_10_day = df['Return'].rolling(window=10).std()

    # Calculate Directional Indicator (DI) as the difference between close and open, smoothed over 10 days
    df['DI'] = (df['close'] - df['open']).ewm(span=10, adjust=False).mean()

    # Combine the two measures to form the heuristics matrix
    heuristics_matrix = std_returns_10_day + df['DI']

    return heuristics_matrix
