def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    ema_close_20_day = df['close'].ewm(span=20, adjust=False).mean()
    high_low_log_diff_20_day = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()).apply(lambda x: math.log(1 + x))
    heuristics_matrix = (std_close_20_day / ema_close_20_day) * (1 + high_low_log_diff_20_day)
    return heuristics_matrix
