import pandas as pd

def heuristics_v2(df):
    mad_close_20_day = df['close'].rolling(window=20).apply(lambda x: pd.Series(x).mad(), raw=True)
    ema_close_20_day = df['close'].ewm(span=20, adjust=False).mean()
    high_low_diff_20_day = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / ema_close_20_day
    heuristics_matrix = (mad_close_20_day / ema_close_20_day) * (1 + high_low_diff_20_day)
    return heuristics_matrix
