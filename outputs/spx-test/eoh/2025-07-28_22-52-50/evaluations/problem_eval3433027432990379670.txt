def heuristics_v2(df):
    mad_close_30_day = df['close'].rolling(window=30).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    high_low_diff_30_day = (df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()) / ema_close_30_day
    heuristics_matrix = (mad_close_30_day / ema_close_30_day) * (1 + high_low_diff_30_day)
    return heuristics_matrix
