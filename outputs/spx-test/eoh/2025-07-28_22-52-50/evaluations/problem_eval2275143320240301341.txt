def heuristics_v2(df):
    ema_close_20_day = df['close'].ewm(span=20, adjust=False).mean()
    ema_high_low_20_day = (df['high'].ewm(span=20, adjust=False).mean() + df['low'].ewm(span=20, adjust=False).mean()) / 2
    heuristics_matrix = ema_close_20_day / ema_high_low_20_day
    return heuristics_matrix
