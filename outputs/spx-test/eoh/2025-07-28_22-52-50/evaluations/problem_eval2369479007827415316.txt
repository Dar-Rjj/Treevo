def heuristics_v2(df):
    ema_range_20_day = (df['high'] - df['low']).ewm(span=20, adjust=False).mean()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    cv_volume_20_day = df['volume'].rolling(window=20).std() / df['volume'].rolling(window=20).mean()
    heuristics_matrix = (ema_range_20_day / sma_close_20_day) * (1 + cv_volume_20_day)
    return heuristics_matrix
