def heuristics_v2(df):
    atr_14_day = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    roc_close_14_day = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    rsi_14_day = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / df['close'].diff(1).clip(upper=0).rolling(window=14).mean().abs())))
    heuristics_matrix = (atr_14_day / roc_close_14_day) * (1 + rsi_14_day / 100)
    return heuristics_matrix
