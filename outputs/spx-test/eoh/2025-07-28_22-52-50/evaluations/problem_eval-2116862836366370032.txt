def heuristics_v2(df):
    ema_close_30_day = df['close'].ewm(span=30).mean()
    std_close_30_day = df['close'].ewm(span=30).std()
    cv_close_30_day = std_close_30_day / ema_close_30_day
    ema_range_30_day = (df['high'] - df['low']).ewm(span=30).mean()
    heuristics_matrix = cv_close_30_day / ema_range_30_day
    return heuristics_matrix
