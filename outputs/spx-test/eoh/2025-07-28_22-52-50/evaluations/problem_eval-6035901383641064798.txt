def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    high_low_diff_20_day = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / sma_close_20_day
    vol_pct_change_20_day = df['volume'].pct_change(periods=20).abs()
    heuristics_matrix = (std_close_20_day / sma_close_20_day) * (1 + high_low_diff_20_day + vol_pct_change_20_day)
    return heuristics_matrix
