def heuristics_v2(df):
    atr_30_day = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    volume_diff_30_day = (df['volume'].rolling(window=30).max() - df['volume'].rolling(window=30).min()) / sma_close_30_day
    heuristics_matrix = (atr_30_day / sma_close_30_day) * (1 + volume_diff_30_day)
    return heuristics_matrix
