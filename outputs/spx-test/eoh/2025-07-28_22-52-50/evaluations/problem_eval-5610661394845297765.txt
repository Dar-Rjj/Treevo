def heuristics_v2(df):
    high_low_ratio_20_day = df['high'].rolling(window=20).mean() / df['low'].rolling(window=20).mean()
    log_return_20_day = np.log(df['close']).diff().rolling(window=20).mean()
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    vol_ratio_20_day = df['volume'] / avg_volume_20_day
    heuristics_matrix = (high_low_ratio_20_day * log_return_20_day) * (1 + vol_ratio_20_day)
    return heuristics_matrix
