def heuristics_v2(df):
    atr_14_day = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    avg_close = df['close'].rolling(window=14).mean()
    vol_change = df['volume'].pct_change(periods=14)
    heuristics_matrix = (atr_14_day / avg_close) * (1 + vol_change)
    return heuristics_matrix
