def heuristics_v2(df):
    momentum_20_day = df['close'].pct_change(periods=20)
    avg_volume_50_day = df['volume'].rolling(window=50).mean()
    price_range_50_day = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    adjustment_factor = avg_volume_50_day / price_range_50_day
    heuristics_matrix = momentum_20_day * adjustment_factor
    return heuristics_matrix
