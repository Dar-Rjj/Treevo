def heuristics_v2(df):
    std_close_30_day = df['close'].rolling(window=30).std()
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    high_low_diff_15_day = (df['high'].rolling(window=15).max() - df['low'].rolling(window=15).min()) / ema_close_30_day
    heuristics_matrix = (std_close_30_day / ema_close_30_day) * (1 + high_low_diff_15_day)
    return heuristics_matrix
