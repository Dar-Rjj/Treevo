def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    ema_close_40_day = df['close'].ewm(span=40, adjust=False).mean()
    high_low_diff_10_day = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / ema_close_40_day
    heuristics_matrix = (std_close_20_day / ema_close_40_day) * (1 + high_low_diff_10_day)
    return heuristics_matrix
