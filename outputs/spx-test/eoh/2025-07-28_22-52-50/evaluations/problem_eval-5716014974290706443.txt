import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change of the closing price
    close_rate_of_change = df['close'].pct_change()
    
    # Compute the 20-day standard deviation of the trading volume
    vol_std_20 = df['volume'].rolling(window=20).std()
    
    # Composite factor
    heuristics_matrix = close_rate_of_change + vol_std_20
    
    return heuristics_matrix
