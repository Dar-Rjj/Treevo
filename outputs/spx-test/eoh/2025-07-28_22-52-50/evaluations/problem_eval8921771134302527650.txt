def heuristics_v2(df):
    range_close_30_day = df['close'].rolling(window=30).max() - df['close'].rolling(window=30).min()
    cumsum_volume_30_day = df['volume'].rolling(window=30).sum()
    high_low_diff_30_day = (df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()) / (df['close'].rolling(window=30).max() - df['close'].rolling(window=30).min())
    heuristics_matrix = (range_close_30_day / cumsum_volume_30_day) * (1 + high_low_diff_30_day)
    return heuristics_matrix
