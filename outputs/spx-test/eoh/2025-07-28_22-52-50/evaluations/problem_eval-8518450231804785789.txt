import pandas as pd
import numpy as np

def heuristics_v2(df):
    ema_close_100_day = df['close'].ewm(span=100, adjust=False).mean()
    sma_close_50_day = df['close'].rolling(window=50).mean()
    high_low_diff_60_day = (df['high'].rolling(window=60).max() - df['low'].rolling(window=60).min()) / sma_close_50_day
    heuristics_matrix = (ema_close_100_day / sma_close_50_day) * np.sqrt(1 + high_low_diff_60_day)
    return heuristics_matrix
