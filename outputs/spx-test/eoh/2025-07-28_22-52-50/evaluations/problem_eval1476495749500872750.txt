def heuristics_v2(df):
    std_close_60_day = df['close'].rolling(window=60).std()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    high_low_diff_10_day = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / sma_close_30_day
    heuristics_matrix = (std_close_60_day / sma_close_30_day) * (1 + high_low_diff_10_day)
    return heuristics_matrix
