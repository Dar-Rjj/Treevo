def heuristics_v2(df):
    std_close_30_day = df['close'].rolling(window=30).std()
    mean_close_30_day = df['close'].rolling(window=30).mean()
    avg_range_10_day = (df['high'] - df['low']).rolling(window=10).mean()
    heuristics_matrix = (std_close_30_day / mean_close_30_day) / avg_range_10_day
    return heuristics_matrix
