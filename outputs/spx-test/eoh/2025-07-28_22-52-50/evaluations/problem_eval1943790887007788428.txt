def heuristics_v2(df):
    ema_close_75_day = df['close'].ewm(span=75, adjust=False).mean()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    high_low_diff_45_day = (df['high'].rolling(window=45).max() - df['low'].rolling(window=45).min()) / sma_close_30_day
    heuristics_matrix = (ema_close_75_day / sma_close_30_day) * np.sqrt(1 + high_low_diff_45_day)
    return heuristics_matrix
