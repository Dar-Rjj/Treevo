import pandas as pd

def heuristics_v2(df):
    # Calculate short and long moving averages
    short_ma = df['close'].rolling(window=15).mean()
    long_ma = df['close'].rolling(window=60).mean()
    
    # Compute RSI (Relative Strength Index)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate ROC (Rate of Change)
    roc = df['close'].pct_change(periods=20)
    
    # Composite factor
    heuristics_matrix = (short_ma - long_ma) + rsi + roc
    
    return heuristics_matrix
