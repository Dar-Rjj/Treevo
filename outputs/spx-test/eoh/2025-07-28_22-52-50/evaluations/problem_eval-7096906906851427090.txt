def heuristics_v2(df):
    ema_close_50_day = df['close'].ewm(span=50, adjust=False).mean()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    high_low_diff_30_day = (df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()) / sma_close_20_day
    avg_volume_30_day = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30_day
    heuristics_matrix = (ema_close_50_day / sma_close_20_day) * (1 + high_low_diff_30_day) * volume_ratio
    return heuristics_matrix
