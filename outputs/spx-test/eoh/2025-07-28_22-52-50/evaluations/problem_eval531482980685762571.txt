def heuristics_v2(df):
    ema_close_20_day = df['close'].ewm(span=20, adjust=False).mean()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    avg_volume_30_day = df['volume'].rolling(window=30).mean()
    volume_diff_30_day = (avg_volume_30_day - df['volume']) / df['volume']
    heuristics_matrix = (ema_close_20_day / sma_close_30_day) * (1 + volume_diff_30_day)
    return heuristics_matrix
