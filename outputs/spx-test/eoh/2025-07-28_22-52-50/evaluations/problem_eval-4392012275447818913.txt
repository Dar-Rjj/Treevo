def heuristics_v2(df):
    pct_change_10_day = df['close'].pct_change(periods=10)
    avg_volume_30_day = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30_day
    sma_close_20_day = df['close'].rolling(window=20).mean()
    sma_close_50_day = df['close'].rolling(window=50).mean()
    sma_diff = sma_close_20_day - sma_close_50_day
    heuristics_matrix = (pct_change_10_day + volume_ratio + sma_diff)
    return heuristics_matrix
