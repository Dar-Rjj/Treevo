def heuristics_v2(df):
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=30).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=30).mean()
    rs = gain / loss
    rsi_30_day = 100 - (100 / (1 + rs))
    heuristics_matrix = ema_close_30_day * (rsi_30_day / 50)
    return heuristics_matrix
