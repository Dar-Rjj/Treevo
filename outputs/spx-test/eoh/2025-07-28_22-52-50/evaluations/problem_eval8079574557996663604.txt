import numpy as np

def heuristics_v2(df):
    atr_14_day = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    ema_close_25_day = df['close'].ewm(span=25, adjust=False).mean()
    high_low_diff_25_day = (df['high'].rolling(window=25).max() - df['low'].rolling(window=25).min()) / ema_close_25_day
    heuristics_matrix = (atr_14_day / ema_close_25_day) * np.sqrt(high_low_diff_25_day)
    return heuristics_matrix
