import pandas as pd

def heuristics_v2(df):
    # Calculate the highest high over a lookback period
    lookback_period = 252
    df['High_H'] = df['high'].rolling(window=lookback_period).max()
    
    # Ratio of close to high
    df['Close_High_Ratio'] = df['close'] / df['High_H']
    
    # Accumulation Distribution Line (ADL)
    df['ADL'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['ADL'] = df['ADL'].cumsum()
    
    # Combined heuristic: Close to High ratio adjusted by ADL
    heuristics_matrix = df['Close_High_Ratio'] + df['ADL']
    
    return heuristics_matrix
