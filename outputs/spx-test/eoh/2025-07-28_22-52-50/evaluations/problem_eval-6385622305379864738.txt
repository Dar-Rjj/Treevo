def heuristics_v2(df):
    daily_returns = df['close'].pct_change()
    positive_returns = daily_returns[daily_returns > 0].rolling(window=20).sum()
    negative_returns = daily_returns[daily_returns < 0].rolling(window=20).sum().abs()
    heuristics_matrix = positive_returns / negative_returns
    return heuristics_matrix
