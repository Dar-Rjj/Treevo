def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))

    # Calculate % change in volume
    df['Volume_Change'] = df['volume'].pct_change()

    # Calculate SMA crossovers
    sma_short_period = 10
    sma_long_period = 50
    df['SMA_Short'] = df['close'].rolling(window=sma_short_period).mean()
    df['SMA_Long'] = df['close'].rolling(window=sma_long_period).mean()
    sma_crossover = df['SMA_Short'] - df['SMA_Long']

    # Combined heuristic: RSI adjusted by Volume Change and SMA crossover
    heuristics_matrix = df['RSI'] + df['Volume_Change'] * 100 + sma_crossover

    return heuristics_matrix
