def heuristics_v2(df):
    mad_close_20_day = df['close'].rolling(window=20).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    sma_close_20_day = df['close'].rolling(window=20).mean()
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    vol_change_20_day = (df['volume'] - avg_volume_20_day) / avg_volume_20_day
    heuristics_matrix = (mad_close_20_day / sma_close_20_day) * (1 + vol_change_20_day)
    return heuristics_matrix
