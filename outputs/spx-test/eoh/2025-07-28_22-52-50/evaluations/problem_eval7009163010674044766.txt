import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving average of volume
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate rate of change in the closing price
    roc_close = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combined heuristic: EMA of volume, ROC of close, and RSI
    heuristics_matrix = ema_volume * roc_close * rsi
    
    return heuristics_matrix
