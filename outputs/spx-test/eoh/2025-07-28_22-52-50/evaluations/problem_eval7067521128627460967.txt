import pandas as pd

def heuristics_v2(df):
    def moving_average(series, window=10):
        return series.rolling(window=window).mean()

    def exponential_moving_average(series, span=10):
        return series.ewm(span=span, adjust=False).mean()

    def price_change_ratio(series):
        return (series - series.shift(1)) / series.shift(1)

    def volume_change_ratio(series):
        return (series - series.shift(1)) / series.shift(1)

    close = df['close']
    volume = df['volume']

    ma_10 = moving_average(close, 10)
    ema_10 = exponential_moving_average(close, 10)
    daily_return = price_change_ratio(close)
    volume_change = volume_change_ratio(volume)

    heuristics_matrix = pd.concat([ma_10, ema_10, daily_return, volume_change], axis=1)
    heuristics_matrix.columns = ['MA_10', 'EMA_10', 'Daily_Return', 'Volume_Change']

    return heuristics_matrix
