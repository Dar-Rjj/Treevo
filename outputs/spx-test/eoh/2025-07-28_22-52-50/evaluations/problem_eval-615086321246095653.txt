def heuristics_v2(df):
    volume_momentum = df['volume'].pct_change(periods=10)
    sma_close_50_day = df['close'].rolling(window=50).mean()
    price_ratio = df['close'] / sma_close_50_day
    high_low_change = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / df['close'].shift(10)
    heuristics_matrix = volume_momentum * price_ratio * (1 + high_low_change)
    return heuristics_matrix
