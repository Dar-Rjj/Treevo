def heuristics_v2(df):
    short_window = 20
    long_window = 50
    momentum_window = 5
    volume_avg_window = 10
    df['short_mavg'] = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    df['long_mavg'] = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    df['volume_wavg'] = df['volume'].rolling(window=volume_avg_window, min_periods=volume_avg_window).mean()
    df['price_momentum'] = df['close'].pct_change(momentum_window)
    df['heuristic_factor'] = (df['short_mavg'] - df['long_mavg']) + df['volume_wavg'] * df['price_momentum']
    heuristics_matrix = df['heuristic_factor'].dropna()
    return heuristics_matrix
