import pandas as pd

def heuristics_v2(df):
    # Calculate short and long exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()

    # Compute RSI (Relative Strength Index)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate ROC (Rate of Change)
    roc = df['close'].pct_change(periods=12)

    # Composite factor
    heuristics_matrix = (short_ema - long_ema) + rsi + roc

    return heuristics_matrix
