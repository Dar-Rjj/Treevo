def heuristics_v2(df):
    ma_close_10_day = df['close'].rolling(window=10).mean()
    ma_close_50_day = df['close'].rolling(window=50).mean()
    cum_abs_change_30_day = (df['close'] - df['close'].shift(1)).abs().rolling(window=30).sum()
    heuristics_matrix = (ma_close_10_day - ma_close_50_day) / cum_abs_change_30_day
    return heuristics_matrix
