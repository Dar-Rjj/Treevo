def heuristics_v2(df):
    # Calculate VWAP
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['TP_Volume'] = df['Typical_Price'] * df['volume']
    vwap_30_day = df['TP_Volume'].rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=30).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=30).mean()
    rs = gain / loss
    rsi_30_day = 100 - (100 / (1 + rs))
    
    # Heuristic calculation
    heuristics_matrix = (df['close'] / vwap_30_day) * (rsi_30_day / 50)
    
    return heuristics_matrix
