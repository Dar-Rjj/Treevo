def heuristics_v2(df):
    ema_close_10_day = df['close'].ewm(span=10, adjust=False).mean()
    sma_close_10_day = df['close'].rolling(window=10).mean()
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    heuristics_matrix = (ema_close_10_day / sma_close_10_day) * rsi
    return heuristics_matrix
