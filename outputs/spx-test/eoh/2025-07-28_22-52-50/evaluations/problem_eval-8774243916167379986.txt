import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    
    # Momentum: 10-day return
    momentum = close.pct_change(10)
    
    # Mean Reversion: Difference between price and its 30-day moving average
    mean_reversion = close - close.rolling(window=30).mean()
    
    # Volume Trend: 10-day change in volume
    volume_trend = volume.pct_change(10)
    
    # Composite Heuristics Matrix
    heuristics_matrix = (momentum + mean_reversion + volume_trend) / 3
    
    return heuristics_matrix
