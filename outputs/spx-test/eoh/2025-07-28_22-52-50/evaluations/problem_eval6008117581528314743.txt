def heuristics_v2(df):
    ema_close_10_day = df['close'].ewm(span=10).mean()
    std_close_10_day = df['close'].ewm(span=10).std()
    cv_close_10_day = std_close_10_day / ema_close_10_day
    ema_range_10_day = (df['high'] - df['low']).ewm(span=10).mean()
    heuristics_matrix_10_day = cv_close_10_day / ema_range_10_day
    
    ema_close_30_day = df['close'].ewm(span=30).mean()
    std_close_30_day = df['close'].ewm(span=30).std()
    cv_close_30_day = std_close_30_day / ema_close_30_day
    ema_range_30_day = (df['high'] - df['low']).ewm(span=30).mean()
    heuristics_matrix_30_day = cv_close_30_day / ema_range_30_day
    
    heuristics_matrix = (heuristics_matrix_10_day + heuristics_matrix_30_day) / 2
    return heuristics_matrix
