import numpy as np

def heuristics_v2(df):
    std_close_25_day = df['close'].rolling(window=25).std()
    sma_close_25_day = df['close'].rolling(window=25).mean()
    high_low_diff_25_day = (df['high'].rolling(window=25).max() - df['low'].rolling(window=25).min()) / sma_close_25_day
    heuristics_matrix = (std_close_25_day / sma_close_25_day) * np.log(1 + high_low_diff_25_day)
    return heuristics_matrix
