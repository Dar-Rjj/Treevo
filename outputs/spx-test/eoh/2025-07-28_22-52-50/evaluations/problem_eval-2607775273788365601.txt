def heuristics_v2(df):
    high_40_day = df['high'].rolling(window=40).max()
    low_40_day = df['low'].rolling(window=40).min()
    ema_close_40_day = df['close'].ewm(span=40, adjust=False).mean()
    close_diff_40_day = df['close'] - ema_close_40_day
    cv_volume_40_day = df['volume'].rolling(window=40).std() / df['volume'].rolling(window=40).mean()
    heuristics_matrix = (high_40_day / low_40_day) * (1 + close_diff_40_day) * (1 + cv_volume_40_day)
    return heuristics_matrix
