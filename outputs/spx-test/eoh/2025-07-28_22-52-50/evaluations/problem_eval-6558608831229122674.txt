def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    mean_close_20_day = df['close'].rolling(window=20).mean()
    avg_range_20_day = (df['high'] - df['low']).rolling(window=20).mean()
    heuristics_matrix = (std_close_20_day / mean_close_20_day) / avg_range_20_day
    return heuristics_matrix
