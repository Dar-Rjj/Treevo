def heuristics_v2(df):
    sma_close_50_day = df['close'].rolling(window=50).mean()
    sma_volume_20_day = df['volume'].rolling(window=20).mean()
    close_diff = df['close'] - sma_close_50_day
    volume_ratio = df['volume'] / sma_volume_20_day
    heuristics_matrix = close_diff * volume_ratio
    return heuristics_matrix
