def heuristics_v2(df):
    mad_close_20_day = df['close'].rolling(window=20).apply(lambda x: np.median(np.abs(x - np.median(x))), raw=True)
    mean_close_20_day = df['close'].rolling(window=20).mean()
    sma_high_low_diff_20_day = ((df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / mean_close_20_day).rolling(window=20).mean()
    heuristics_matrix = (mad_close_20_day / mean_close_20_day) * (1 + sma_high_low_diff_20_day)
    return heuristics_matrix
