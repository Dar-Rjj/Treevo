def heuristics_v2(df):
    ema_close_20_day = df['close'].ewm(span=20, adjust=False).mean()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    high_low_diff_20_day = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / sma_close_20_day
    heuristics_matrix = (ema_close_20_day / sma_close_20_day) * (1 + high_low_diff_20_day)
    return heuristics_matrix
