def heuristics_v2(df):
    ema_close_20_day = df['close'].ewm(span=20, adjust=False).mean()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    cv_close_20_day = 2 * (df['close'].rolling(window=20).std() / df['close'].rolling(window=20).mean())
    heuristics_matrix = (ema_close_20_day / sma_close_20_day) * (1 + cv_close_20_day)
    return heuristics_matrix
