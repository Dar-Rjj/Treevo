def heuristics_v2(df):
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    skew_close_30_day = df['close'].rolling(window=30).skew()
    heuristics_matrix = (ema_close_30_day / sma_close_30_day) * (1 + 3 * skew_close_30_day)
    return heuristics_matrix
