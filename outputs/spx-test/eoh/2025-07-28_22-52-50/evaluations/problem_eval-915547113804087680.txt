def heuristics_v2(df):
    mad_close_30_day = df['close'].rolling(window=30).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)
    sma_close_30_day = df['close'].rolling(window=30).mean()
    vol_change_30_day = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    heuristics_matrix = (mad_close_30_day / sma_close_30_day) * (1 + vol_change_30_day)
    return heuristics_matrix
