def heuristics_v2(df):
    mad_returns_30_day = df['close'].pct_change().rolling(window=30).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    sma_close_30_day = df['close'].rolling(window=30).mean()
    high_low_diff_20_day = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / sma_close_30_day
    heuristics_matrix = (mad_returns_30_day / sma_close_30_day) * (1 + high_low_diff_20_day)
    return heuristics_matrix
