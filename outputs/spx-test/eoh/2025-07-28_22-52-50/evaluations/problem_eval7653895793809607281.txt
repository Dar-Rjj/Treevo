def heuristics_v2(df):
    roc_close_60_day = df['close'].pct_change(periods=60)
    avg_volume_60_day = df['volume'].rolling(window=60).mean()
    median_volume_60_day = df['volume'].rolling(window=60).median()
    heuristics_matrix = roc_close_60_day * (avg_volume_60_day / median_volume_60_day)
    return heuristics_matrix
