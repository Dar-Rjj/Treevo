def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    high_low_diff_20_day = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / sma_close_20_day
    log_ratio = (std_close_20_day / sma_close_20_day).apply(lambda x: math.log(x) if x > 0 else 0)
    exp_adjustment = np.exp(high_low_diff_20_day * 0.1)
    heuristics_matrix = log_ratio * (1 + exp_adjustment)
    return heuristics_matrix
