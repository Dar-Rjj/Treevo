import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    
    # Calculate the short-term (10 days) and long-term (30 days) Exponential Moving Averages for close prices
    short_close_ema = close_prices.ewm(span=10, adjust=False).mean()
    long_close_ema = close_prices.ewm(span=30, adjust=False).mean()
    
    # Average Trading Volume over 20 days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Combined heuristic factor
    heuristics_matrix = (short_close_ema - long_close_ema) + avg_volume
    
    return heuristics_matrix
