import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    
    # Calculate the short-term (5 days) and long-term (20 days) Exponential Moving Averages for close prices
    short_close_ema = close_prices.ewm(span=5, adjust=False).mean()
    long_close_ema = close_prices.ewm(span=20, adjust=False).mean()
    
    # Intraday Volatility: Standard Deviation of daily high-low ranges over 20 days
    intraday_volatility = (df['high'] - df['low']).rolling(window=20).std()
    
    # Combined heuristic factor
    heuristics_matrix = (short_close_ema - long_close_ema) + intraday_volatility
    
    return heuristics_matrix
