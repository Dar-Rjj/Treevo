def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate Momentum
    momentum_period = 10
    momentum = df['close'] - df['close'].shift(momentum_period)

    # Calculate the ratio of the current close to the highest high over the last 20 days
    highest_high_20_day = df['high'].rolling(window=20).max()
    close_to_highest_high_ratio = df['close'] / highest_high_20_day

    # Combined heuristic: RSI adjusted by Momentum and the ratio of the current close to the highest high
    heuristics_matrix = rsi + (momentum * 0.01) * close_to_highest_high_ratio

    return heuristics_matrix
