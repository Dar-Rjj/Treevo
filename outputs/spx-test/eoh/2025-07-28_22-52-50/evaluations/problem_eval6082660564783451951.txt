def heuristics_v2(df):
    sma_close_10_day = df['close'].rolling(window=10).mean()
    ema_close_10_day = df['close'].ewm(span=10, adjust=False).mean()
    cv_close_10_day = df['close'].rolling(window=10).std() / df['close'].rolling(window=10).mean()
    heuristics_matrix = (sma_close_10_day / ema_close_10_day) * (1 + cv_close_10_day)
    return heuristics_matrix
