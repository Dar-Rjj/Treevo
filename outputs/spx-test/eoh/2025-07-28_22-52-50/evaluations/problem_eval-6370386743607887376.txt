def heuristics_v2(df):
    atr_20_day = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi_20_day = 100 - (100 / (1 + rs))
    heuristics_matrix = (atr_20_day / sma_close_20_day) * (rsi_20_day / 50)
    return heuristics_matrix
