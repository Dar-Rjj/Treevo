def heuristics_v2(df):
    ema_close_50_day = df['close'].ewm(span=50, adjust=False).mean()
    ema_volume_50_day = df['volume'].ewm(span=50, adjust=False).mean()
    ema_range_50_day = (df['high'] - df['low']).ewm(span=50, adjust=False).mean()
    heuristics_matrix = ema_close_50_day / (ema_volume_50_day + ema_range_50_day)
    return heuristics_matrix
