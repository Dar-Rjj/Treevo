def heuristics_v2(df):
    std_close_30_day = df['close'].rolling(window=30).std()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    high_low_diff_15_day = (df['high'].rolling(window=15).max() - df['low'].rolling(window=15).min()) / sma_close_20_day
    heuristics_matrix = (std_close_30_day / sma_close_20_day) * (1 + high_low_diff_15_day)
    return heuristics_matrix
