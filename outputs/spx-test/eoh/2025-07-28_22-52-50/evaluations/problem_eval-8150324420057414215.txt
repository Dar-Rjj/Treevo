def heuristics_v2(df):
    std_close_30_day = df['close'].rolling(window=30).std()
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    avg_volume_30_day = df['volume'].rolling(window=30).mean()
    vol_change_30_day = (df['volume'] - avg_volume_30_day) / avg_volume_30_day
    heuristics_matrix = (std_close_30_day / ema_close_30_day) * (1 + vol_change_30_day)
    return heuristics_matrix
