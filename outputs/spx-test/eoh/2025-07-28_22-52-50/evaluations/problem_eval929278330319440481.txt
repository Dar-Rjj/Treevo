def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    vol_change_20_day = (df['volume'] - avg_volume_20_day) / avg_volume_20_day
    heuristics_matrix = (std_close_20_day / sma_close_20_day) * (1 + vol_change_20_day)
    return heuristics_matrix
