def heuristics_v2(df):
    roc_close_5_day = df['close'].pct_change(periods=5)
    sma_close_20_day = df['close'].rolling(window=20).mean()
    high_low_log_diff_20_day = (df['high'].rolling(window=20).max() / df['low'].rolling(window=20).min()).apply(np.log)
    heuristics_matrix = (roc_close_5_day / sma_close_20_day) * (1 + high_low_log_diff_20_day)
    return heuristics_matrix
