def heuristics_v2(df):
    high_close_50_day = df['close'].rolling(window=50).max()
    low_close_50_day = df['close'].rolling(window=50).min()
    high_volume_50_day = df['volume'].rolling(window=50).max()
    low_volume_50_day = df['volume'].rolling(window=50).min()
    heuristics_matrix = (high_close_50_day / low_close_50_day) * (high_volume_50_day - low_volume_50_day)
    return heuristics_matrix
