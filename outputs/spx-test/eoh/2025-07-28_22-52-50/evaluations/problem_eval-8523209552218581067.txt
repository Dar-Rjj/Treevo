import pandas as pd

def heuristics_v2(df):
    # Calculate short and long exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Compute ADOSC (Accumulation/Distribution Oscillator)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    adosc = money_flow_volume.rolling(window=14).sum() - money_flow_volume.rolling(window=28).sum()
    
    # Calculate Stochastic Oscillator
    low_min = df['low'].rolling(window=14).min()
    high_max = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - low_min) / (high_max - low_min)
    
    # Composite factor
    heuristics_matrix = (short_ema - long_ema) + adosc + k_percent
    
    return heuristics_matrix
