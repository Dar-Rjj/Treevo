def heuristics_v2(df):
    close_pct_change = df['close'].pct_change()
    log_returns = np.log(df['close'] / df['close'].shift(1))
    vol_30_day = log_returns.rolling(window=30).std()
    vwap_30_day = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    heuristics_matrix = close_pct_change + vol_30_day + vwap_30_day
    return heuristics_matrix
