def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    avg_high_low_20_day = (df['high'].rolling(window=20).mean() + df['low'].rolling(window=20).mean()) / 2
    vwap_20_day = (df['volume'] * df['close']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    heuristics_matrix = (std_close_20_day / avg_high_low_20_day) * vwap_20_day
    return heuristics_matrix
