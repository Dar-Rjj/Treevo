def heuristics_v2(df):
    std_close_20_day = df['close'].rolling(window=20).std()
    mean_close_20_day = df['close'].rolling(window=20).mean()
    ema_high_low_diff_20_day = ((df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / mean_close_20_day).ewm(span=20, adjust=False).mean()
    heuristics_matrix = (std_close_20_day / mean_close_20_day) * (1 + ema_high_low_diff_20_day)
    return heuristics_matrix
