def heuristics_v2(df):
    roc_5_day = df['close'].pct_change(5)
    roc_10_day = df['close'].pct_change(10)
    roc_20_day = df['close'].pct_change(20)
    rsi_14_day = 100 - (100 / (1 + df['close'].diff().rolling(window=14).mean() / df['close'].diff().abs().rolling(window=14).mean()))
    weighted_roc = (roc_5_day * 0.4 + roc_10_day * 0.3 + roc_20_day * 0.2) * rsi_14_day
    heuristics_matrix = weighted_roc
    return heuristics_matrix
