def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc_period = 10
    df['ROC'] = (df['close'] - df['close'].shift(roc_period)) / df['close'].shift(roc_period) * 100

    # Calculate Cumulative Volume
    cumulative_volume = df['volume'].rolling(window=roc_period).sum()

    # Combined heuristic: ROC adjusted by cumulative volume
    heuristics_matrix = df['ROC'] + (cumulative_volume / 1000000) * 0.01

    return heuristics_matrix
