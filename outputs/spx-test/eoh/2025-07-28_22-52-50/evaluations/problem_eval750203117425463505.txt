def heuristics_v2(df):
    avg_daily_return_10_day = df['close'].pct_change().rolling(window=10).mean()
    high_low_ratio_10_day = df['high'].rolling(window=10).max() / df['low'].rolling(window=10).min()
    heuristics_matrix = avg_daily_return_10_day * high_low_ratio_10_day
    return heuristics_matrix
