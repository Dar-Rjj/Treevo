    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate 20-day SMA of closing price
    sma_20 = df['close'].rolling(window=20, min_periods=1).mean()
    
    # Calculate volume ratio: current day's volume divided by the 30-day average volume
    avg_volume_30 = df['volume'].rolling(window=30, min_periods=1).mean()
    volume_ratio = df['volume'] / avg_volume_30
    
    # Composite heuristic
    heuristics_matrix = 0.5 * (rsi / 100) + 0.3 * (sma_20 / df['close']) + 0.2 * volume_ratio
    
    return heuristics_matrix
