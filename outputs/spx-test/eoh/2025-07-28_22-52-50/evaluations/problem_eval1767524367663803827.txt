def heuristics_v2(df):
    ema_close_10_day = df['close'].ewm(span=10, adjust=False).mean()
    ema_close_50_day = df['close'].ewm(span=50, adjust=False).mean()
    std_close_20_day = df['close'].rolling(window=20).std()
    heuristics_matrix = (ema_close_10_day - ema_close_50_day) / std_close_20_day
    return heuristics_matrix
