import pandas as pd

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    df['short_mavg'] = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    df['long_mavg'] = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    df['volume_change'] = df['volume'].pct_change()
    df['price_volatility'] = df['close'].rolling(window=short_window, min_periods=short_window).std()
    df['heuristic_factor'] = (df['short_mavg'] - df['long_mavg']) + df['volume_change'] + df['price_volatility']
    heuristics_matrix = df['heuristic_factor'].dropna()
    return heuristics_matrix
