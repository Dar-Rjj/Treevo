def heuristics_v2(df):
    wma_volume_60_day = df['volume'].rolling(window=60).apply(lambda x: (x * range(1, 61)).sum() / sum(range(1, 61)), raw=False)
    sma_close_40_day = df['close'].rolling(window=40).mean()
    close_cv_30_day = df['close'].rolling(window=30).std() / df['close'].rolling(window=30).mean()
    heuristics_matrix = (wma_volume_60_day / sma_close_40_day) * (1 + close_cv_30_day)
    return heuristics_matrix
