def heuristics_v2(df):
    roc_close_30_day = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    std_volume_30_day = df['volume'].rolling(window=30).std()
    mean_volume_30_day = df['volume'].rolling(window=30).mean()
    cv_volume_30_day = std_volume_30_day / mean_volume_30_day
    heuristics_matrix = roc_close_30_day * cv_volume_30_day
    return heuristics_matrix
