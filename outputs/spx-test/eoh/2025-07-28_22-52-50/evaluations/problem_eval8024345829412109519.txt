def heuristics_v2(df):
    atr_10_day = (df['high'] - df['low']).rolling(window=10).mean()
    std_close_10_day = df['close'].rolling(window=10).std()
    mean_close_10_day = df['close'].rolling(window=10).mean()
    heuristics_matrix = atr_10_day / (std_close_10_day / mean_close_10_day)
    return heuristics_matrix
