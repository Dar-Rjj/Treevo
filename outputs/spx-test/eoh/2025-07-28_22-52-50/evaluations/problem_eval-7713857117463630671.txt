def heuristics_v2(df):
    high_50_day = df['high'].rolling(window=50).max()
    low_50_day = df['low'].rolling(window=50).min()
    sma_close_50_day = df['close'].rolling(window=50).mean()
    ema_close_50_day = df['close'].ewm(span=50, adjust=False).mean()
    adjustment_factor = sma_close_50_day - ema_close_50_day
    heuristics_matrix = (high_50_day / low_50_day) * (1 + adjustment_factor)
    return heuristics_matrix
