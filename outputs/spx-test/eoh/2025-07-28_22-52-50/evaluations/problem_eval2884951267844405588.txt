def heuristics_v2(df):
    ema_close_200_day = df['close'].ewm(span=200, adjust=False).mean()
    sma_close_50_day = df['close'].rolling(window=50).mean()
    high_low_diff_60_day = (df['high'].rolling(window=60).max() - df['low'].rolling(window=60).min()) / sma_close_50_day
    heuristics_matrix = (ema_close_200_day / sma_close_50_day) * (1 + high_low_diff_60_day)
    return heuristics_matrix
