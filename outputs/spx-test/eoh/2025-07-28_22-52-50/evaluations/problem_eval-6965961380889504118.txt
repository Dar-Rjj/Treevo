def heuristics_v2(df):
    sma_close_30_day = df['close'].rolling(window=30).mean()
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    diff_sma_close = sma_close_30_day - df['close']
    high_low_diff_30_day = (df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()) / ema_close_30_day
    heuristics_matrix = (diff_sma_close / ema_close_30_day) * (1 + high_low_diff_30_day)
    return heuristics_matrix
