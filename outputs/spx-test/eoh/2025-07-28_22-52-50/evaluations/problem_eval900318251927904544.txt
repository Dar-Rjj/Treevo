def heuristics_v2(df):
    var_close_20_day = df['close'].rolling(window=20).var()
    sma_close_60_day = df['close'].rolling(window=60).mean()
    high_low_diff_10_day = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / sma_close_60_day
    heuristics_matrix = (var_close_20_day / sma_close_60_day) * (1 + high_low_diff_10_day)
    return heuristics_matrix
