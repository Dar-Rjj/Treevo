def heuristics_v2(df):
    atr_50_day = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    sma_close_50_day = df['close'].rolling(window=50).mean()
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=50).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=50).mean()
    rs = gain / loss
    rsi_50_day = 100 - (100 / (1 + rs))
    heuristics_matrix = (atr_50_day / sma_close_50_day) * (rsi_50_day / 50)
    return heuristics_matrix
