def heuristics_v2(df):
    atr_30_day = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=30).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=30).mean()
    rs = gain / loss
    rsi_30_day = 100 - (100 / (1 + rs))
    heuristics_matrix = (atr_30_day / sma_close_30_day) * (rsi_30_day / 50)
    return heuristics_matrix
