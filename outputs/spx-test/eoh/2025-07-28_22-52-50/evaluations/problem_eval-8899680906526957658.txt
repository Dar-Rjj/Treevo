def heuristics_v2(df):
    ema_close_30_day = df['close'].ewm(span=30, adjust=False).mean()
    sma_close_30_day = df['close'].rolling(window=30).mean()
    cv_close_30_day = df['close'].rolling(window=30).std() / df['close'].rolling(window=30).mean()
    heuristics_matrix = (ema_close_30_day / sma_close_30_day) * (1 + 2 * cv_close_30_day)
    return heuristics_matrix
