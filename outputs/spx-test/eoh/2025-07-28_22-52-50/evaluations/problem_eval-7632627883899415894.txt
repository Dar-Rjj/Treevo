def heuristics_v2(df):
    ema_close_100_day = df['close'].ewm(span=100, adjust=False).mean()
    sma_close_50_day = df['close'].rolling(window=50).mean()
    avg_volume_30_day = df['volume'].rolling(window=30).mean()
    heuristics_matrix = (ema_close_100_day / sma_close_50_day) * avg_volume_30_day
    return heuristics_matrix
