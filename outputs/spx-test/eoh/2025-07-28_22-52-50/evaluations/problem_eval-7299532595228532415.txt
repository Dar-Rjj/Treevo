def heuristics_v2(df):
    std_close_30_day = df['close'].rolling(window=30).std()
    wma_close_30_day = df['close'].rolling(window=30).apply(lambda x: (x * df.loc[x.index, 'volume']).sum() / df.loc[x.index, 'volume'].sum(), raw=False)
    high_low_diff_15_day = (df['high'].rolling(window=15).max() - df['low'].rolling(window=15).min()) / wma_close_30_day
    heuristics_matrix = (std_close_30_day / wma_close_30_day) * (1 + high_low_diff_15_day)
    return heuristics_matrix
