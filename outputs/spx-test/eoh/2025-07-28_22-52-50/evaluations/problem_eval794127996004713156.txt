def heuristics_v2(df):
    vwap_20_day = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    sma_close_20_day = df['close'].rolling(window=20).mean()
    avg_volume_20_day = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=20).std()
    heuristics_matrix = (vwap_20_day / sma_close_20_day) * (1 + avg_volume_20_day)
    return heuristics_matrix
