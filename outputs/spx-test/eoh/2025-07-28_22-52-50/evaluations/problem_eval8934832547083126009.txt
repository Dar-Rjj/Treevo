def heuristics_v2(df):
    ema_volume_30_day = df['volume'].ewm(span=30).mean()
    abs_return = df['close'].pct_change().abs()
    ema_abs_return_30_day = abs_return.ewm(span=30).mean()
    ema_close_30_day = df['close'].ewm(span=30).mean()
    heuristics_matrix = (ema_volume_30_day + ema_abs_return_30_day) / ema_close_30_day
    return heuristics_matrix
