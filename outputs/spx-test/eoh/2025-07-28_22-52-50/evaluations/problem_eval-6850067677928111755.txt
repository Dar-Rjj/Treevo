import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day Exponential Moving Average (EMA)
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    
    # Compute the Stochastic Oscillator
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    
    # Calculate the 12-day Volume Rate of Change (VROC)
    vroc = df['volume'].pct_change(periods=12)
    
    # Composite factor
    heuristics_matrix = ema_14 + k + vroc
    
    return heuristics_matrix
