def heuristics_v2(df):
    daily_returns = df['close'].pct_change()
    pos_returns = daily_returns[daily_returns > 0].rolling(window=50).sum()
    neg_returns = daily_returns[daily_returns < 0].rolling(window=50).sum()
    heuristics_matrix = (pos_returns + abs(neg_returns)) / 50
    return heuristics_matrix
