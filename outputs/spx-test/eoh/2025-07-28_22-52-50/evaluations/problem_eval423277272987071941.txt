def heuristics_v2(df):
    momentum_60_day = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    avg_volume_60_day = df['volume'].rolling(window=60).mean()
    avg_close_60_day = df['close'].rolling(window=60).mean()
    heuristics_matrix = momentum_60_day * (avg_volume_60_day / avg_close_60_day)
    return heuristics_matrix
