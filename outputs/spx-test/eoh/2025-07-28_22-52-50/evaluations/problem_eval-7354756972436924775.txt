import pandas as pd

def heuristics_v2(df):
    # Calculate Modified RSI
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    rsi_period = 14
    gain = up.ewm(com=rsi_period-1, adjust=False).mean()
    loss = down.abs().ewm(com=rsi_period-1, adjust=False).mean()
    rs = gain / loss
    df['M_RSI'] = 100 - (100 / (1 + rs))

    # Calculate PPO
    short_ema_period = 12
    long_ema_period = 26
    df['EMA_Short'] = df['close'].ewm(span=short_ema_period, adjust=False).mean()
    df['EMA_Long'] = df['close'].ewm(span=long_ema_period, adjust=False).mean()
    df['PPO'] = ((df['EMA_Short'] - df['EMA_Long']) / df['EMA_Long']) * 100

    # Calculate Bollinger Bands
    sma_period = 20
    stdev_periods = 2
    df['SMA'] = df['close'].rolling(window=sma_period).mean()
    df['StdDev'] = df['close'].rolling(window=sma_period).std()
    df['Upper_Band'] = df['SMA'] + (df['StdDev'] * stdev_periods)
    df['Lower_Band'] = df['SMA'] - (df['StdDev'] * stdev_periods)
    df['BB_Width'] = (df['Upper_Band'] - df['Lower_Band']) / df['SMA']

    # Combined heuristic: M_RSI, PPO, and BB_Width
    heuristics_matrix = df['M_RSI'] + df['PPO'] + df['BB_Width']

    return heuristics_matrix
