def heuristics_v2(df):
    vwap_10_day = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    cma_close_60_day = df['close'].expanding().mean()
    vwap_momentum_30_day = vwap_10_day.pct_change(30)
    heuristics_matrix = (vwap_10_day / cma_close_60_day) * (1 + vwap_momentum_30_day)
    return heuristics_matrix
