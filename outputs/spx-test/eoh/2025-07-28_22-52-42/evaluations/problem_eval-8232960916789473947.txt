import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and low prices
    df['log_close'] = np.log(df['close'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 30-day Simple Moving Average (SMA) for log_close
    df['SMA_30_log_close'] = df['log_close'].rolling(window=30).mean()
    
    # Calculate the 20-day Exponential Moving Average (EMA) for log_low
    df['EMA_20_log_low'] = df['log_low'].ewm(span=20, adjust=False).mean()
    
    # Create the heuristic: difference between EMA_20_log_low and SMA_30_log_close
    heuristics_matrix = df['EMA_20_log_low'] - df['SMA_30_log_close']
    
    return heuristics_matrix
