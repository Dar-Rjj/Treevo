def heuristics_v2(df):
    # Momentum: 15-day change in closing price
    momentum = df['close'] - df['close'].shift(15)
    
    # Volatility: 30-day standard deviation of closing price
    vol = df['close'].rolling(window=30).std()
    
    # Volume ratio: current volume compared to its 30-day moving average
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    vol_ratio = df['volume'] / avg_vol_30d
    
    # Directional movement: difference between 5-day high and low
    high_5d = df['high'].rolling(window=5).max()
    low_5d = df['low'].rolling(window=5).min()
    dir_movement = (high_5d - low_5d) / df['close']
    
    # Composite heuristic with different weights
    heuristics_matrix = (momentum * 0.5 + (vol * vol_ratio) * 0.2 + dir_movement * 0.3)
    
    return heuristics_matrix
