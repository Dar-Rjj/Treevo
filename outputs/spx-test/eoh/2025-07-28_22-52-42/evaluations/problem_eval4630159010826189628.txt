import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close prices and volume
    df['log_close'] = np.log(df['close'])
    df['log_volume'] = np.log(df['volume'])
    
    # Calculate the 20-day Simple Moving Average (SMA) for log_close
    df['SMA_20_log_close'] = df['log_close'].rolling(window=20).mean()
    
    # Calculate the 20-day cumulative sum for log_volume
    df['cumsum_20_log_volume'] = df['log_volume'].rolling(window=20).sum()
    
    # Create the heuristic: ratio of cumsum_20_log_volume to SMA_20_log_close
    heuristics_matrix = df['cumsum_20_log_volume'] / df['SMA_20_log_close']
    
    return heuristics_matrix
