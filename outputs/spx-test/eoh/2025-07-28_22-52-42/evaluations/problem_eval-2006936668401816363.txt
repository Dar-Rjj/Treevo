def heuristics_v2(df):
    # Calculate the natural logarithm of the open and close prices
    df['log_open'] = np.log(df['open'])
    df['log_close'] = np.log(df['close'])

    # Calculate the 15-day Exponential Moving Average (EMA) for log_open
    df['EMA_15_log_open'] = df['log_open'].ewm(span=15, adjust=False).mean()

    # Calculate the 30-day Moving Average (MA) for log_close
    df['MA_30_log_close'] = df['log_close'].rolling(window=30).mean()

    # Create the heuristic: difference between EMA_15_log_open and MA_30_log_close
    heuristics_matrix = df['EMA_15_log_open'] - df['MA_30_log_close']

    return heuristics_matrix
