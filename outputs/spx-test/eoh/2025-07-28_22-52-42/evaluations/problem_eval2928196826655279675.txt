import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 5-day moving average of the ratio of closing prices to their high prices
    close_high_ratio_rolling = (df['close'] / df['high']).rolling(window=5).mean()
    # Apply natural logarithm to the volume
    ln_volume = np.log(df['volume'])
    # Generate the heuristics matrix as the product of the 5-day moving average of the close-high ratio and the natural logarithm of volume
    heuristics_matrix = close_high_ratio_rolling * ln_volume
    return heuristics_matrix
