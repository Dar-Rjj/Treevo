import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the log difference of consecutive days' closing prices
    log_close_diff = np.log(df['close']).diff().shift(-1)
    # Calculate the log of the volume
    log_volume = np.log(df['volume'])
    # Compute the difference between these two values
    diff_matrix = log_close_diff - log_volume
    # Apply a sigmoid transformation to the difference matrix
    heuristics_matrix = 1 / (1 + np.exp(-diff_matrix))
    return heuristics_matrix
