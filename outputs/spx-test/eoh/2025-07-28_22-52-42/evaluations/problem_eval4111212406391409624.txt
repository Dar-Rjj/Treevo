def heuristics_v2(df):
    # Calculate the natural logarithm of the high prices
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 40-day Simple Moving Average (SMA) for volume
    df['SMA_40_volume'] = df['volume'].rolling(window=40).mean()
    
    # Calculate the 20-day Simple Moving Average (SMA) for log_high
    df['SMA_20_log_high'] = df['log_high'].rolling(window=20).mean()
    
    # Create the heuristic: ratio of SMA_40_volume to SMA_20_log_high
    heuristics_matrix = df['SMA_40_volume'] / df['SMA_20_log_high']
    
    return heuristics_matrix
