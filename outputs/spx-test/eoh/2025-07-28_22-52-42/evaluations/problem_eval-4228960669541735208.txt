def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 30-day Simple Moving Average (SMA) for log_close
    df['SMA_30_log_close'] = df['log_close'].rolling(window=30).mean()
    
    # Calculate the 10-day Exponential Moving Average (EMA) for log_high
    df['EMA_10_log_high'] = df['log_high'].ewm(span=10, adjust=False).mean()
    
    # Create the heuristic: difference between SMA_30_log_close and EMA_10_log_high
    heuristics_matrix = df['SMA_30_log_close'] - df['EMA_10_log_high']
    
    return heuristics_matrix
