import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 30-day Cumulative Moving Average (CMA) for log_high
    df['CMA_30_log_high'] = df['log_high'].expanding().mean()
    
    # Calculate the 30-day Geometric Mean (GM) for log_close
    def geometric_mean(x):
        return np.exp(np.log(x).mean())
    df['GM_30_log_close'] = df['log_close'].rolling(window=30).apply(geometric_mean, raw=True)
    
    # Create the heuristic: difference between CMA_30_log_high and GM_30_log_close
    heuristics_matrix = df['CMA_30_log_high'] - df['GM_30_log_close']
    
    return heuristics_matrix
