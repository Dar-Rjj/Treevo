def heuristics_v2(df):
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    # Calculate the 10-day moving average of volume
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    # Generate the heuristics matrix by multiplying the price difference with the ratio of 10-day volume MA to current volume
    heuristics_matrix = price_diff * (volume_ma_10 / df['volume'])
    return heuristics_matrix
