import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the open and low prices
    df['log_open'] = np.log(df['open'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 20-day Exponential Moving Average (EMA) for log_low
    df['EMA_20_log_low'] = df['log_low'].ewm(span=20, adjust=False).mean()
    
    # Define weights for Weighted Moving Average (WMA) with an exponentially decreasing weight scheme
    exp_weights = np.exp(np.linspace(1, -1, 20))
    exp_weights /= exp_weights.sum()
    
    # Calculate the 20-day Weighted Moving Average (WMA) for log_open
    df['WMA_20_log_open'] = df['log_open'].rolling(window=20).apply(lambda x: (x * exp_weights[::-1]).sum(), raw=True)
    
    # Create the heuristic: difference between EMA_20_log_low and WMA_20_log_open
    heuristics_matrix = df['EMA_20_log_low'] - df['WMA_20_log_open']
    
    return heuristics_matrix
