import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day standard deviation of high and low prices
    df['std_14_high_low'] = (df['high'] - df['low']).rolling(window=14).std()
    # Calculate the 14-day Exponential Moving Average (EMA) for close prices
    df['EMA_14_close'] = df['close'].ewm(span=14, adjust=False).mean()
    # Create the heuristic: 14-day standard deviation of high and low prices minus EMA_14_close
    heuristics_matrix = df['std_14_high_low'] - df['EMA_14_close']
    
    return heuristics_matrix
