import pandas as pd

def heuristics_v2(df):
    # Short and long moving average windows
    short_window = 10
    long_window = 50
    
    # Calculate simple moving averages
    SMA_short = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    SMA_long = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    
    # Momentum: difference between current price and the price from n days ago
    momentum_10 = df['close'] - df['close'].shift(10)
    momentum_50 = df['close'] - df['close'].shift(50)
    
    # Combine the factors into a DataFrame
    heuristics_matrix = pd.DataFrame({
        'SMA_short': SMA_short,
        'SMA_long': SMA_long,
        'Momentum_10': momentum_10,
        'Momentum_50': momentum_50
    })
    
    # Return only the complete rows (i.e., without NaN values introduced by the calculations)
    return heuristics_matrix
