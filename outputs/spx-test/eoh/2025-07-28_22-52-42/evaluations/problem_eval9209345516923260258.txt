def heuristics_v2(df):
    # Calculate the log return of the close price over a short period (e.g., 5 days)
    log_return = df['close'].apply(np.log).diff(5)
    
    # Calculate the long-term average volume (e.g., 60 days)
    avg_vol_long_term = df['volume'].rolling(window=60).mean()
    
    # Compute the ratio of the current day's volume to the long-term average volume
    vol_ratio = df['volume'] / avg_vol_long_term
    
    # Generate the heuristics matrix as a combination of the log return and the volume ratio
    heuristics_matrix = log_return * vol_ratio

    return heuristics_matrix
