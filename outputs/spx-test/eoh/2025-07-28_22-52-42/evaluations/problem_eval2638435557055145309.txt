import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in daily high and low
    roc_high_low = (df['high'] - df['low']) / df['low'].shift(1)
    
    # Compute the 5-day moving average of the trading volume
    volume_ma_5 = df['volume'].rolling(window=5).mean()
    
    # Heuristic matrix: product of the rate of change in daily high and low and the 5-day moving average of the trading volume
    heuristics_matrix = roc_high_low * volume_ma_5
    
    return heuristics_matrix
