import pandas as pd
import numpy as np

def heuristics_v2(df):
    df['log_close'] = np.log(df['close'])
    df['log_volume'] = np.log(df['volume'])
    df['SMA_30_log_close'] = df['log_close'].rolling(window=30).mean()
    df['cumsum_10_log_volume'] = df['log_volume'].rolling(window=10).sum()
    heuristics_matrix = df['cumsum_10_log_volume'] / df['SMA_30_log_close']
    return heuristics_matrix
