import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the 15-day Simple Moving Average (SMA) for high_low_diff
    df['SMA_15_high_low_diff'] = df['high_low_diff'].rolling(window=15).mean()
    
    # Calculate the 30-day cumulative sum of volume
    df['cumsum_volume_30'] = df['volume'].rolling(window=30).sum()
    
    # Create the heuristic: ratio of SMA_15_high_low_diff to cumsum_volume_30
    heuristics_matrix = df['SMA_15_high_low_diff'] / df['cumsum_volume_30']
    
    return heuristics_matrix
