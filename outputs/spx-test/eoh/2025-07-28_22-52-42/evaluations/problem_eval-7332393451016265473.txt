def heuristics_v2(df):
    # Calculate the average price for each day
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    # Calculate the ratio of the difference between closing and opening prices to the average price
    price_diff_ratio = (df['close'] - df['open']) / avg_price
    # Apply square root to the volume
    sqrt_volume = np.sqrt(df['volume'])
    # Generate the heuristics matrix as the product of the price difference ratio and the square root of volume
    heuristics_matrix = price_diff_ratio * sqrt_volume
    return heuristics_matrix
