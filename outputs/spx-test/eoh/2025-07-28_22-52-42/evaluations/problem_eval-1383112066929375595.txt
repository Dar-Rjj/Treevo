import pandas as pd
import numpy as np

def heuristics_v2(df):
    ema_diff = df['high'].ewm(span=10, adjust=False).mean() - df['low'].ewm(span=10, adjust=False).mean()
    log_volume_change = np.log(df['volume'] / df['volume'].shift(1))
    heuristics_matrix = ema_diff * log_volume_change
    return heuristics_matrix
