def heuristics_v2(df):
    # Calculate the difference between closing and opening prices
    price_diff = df['close'] - df['open']
    # Apply a non-linear transformation (square) to the price difference
    transformed_price_diff = price_diff ** 2
    # Calculate the rolling sum of volume over a 5-day window
    volume_sum = df['volume'].rolling(window=5).sum()
    # Generate the heuristics matrix as the product of the two values
    heuristics_matrix = transformed_price_diff * volume_sum
    return heuristics_matrix
