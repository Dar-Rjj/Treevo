def heuristics_v2(df):
    # Momentum: 21-day change in closing price
    momentum = df['close'] - df['close'].shift(21)
    
    # Volatility: 20-day average true range
    tr = df[['high', 'low']].rolling(window=2).apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), raw=True)
    atr = tr.rolling(window=20).mean()
    
    # Volume ratio: current volume compared to its 30-day moving average
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    vol_ratio = df['volume'] / avg_vol_30d
    
    # Directional movement: difference between 20-day high and low
    high_20d = df['high'].rolling(window=20).max()
    low_20d = df['low'].rolling(window=20).min()
    dir_movement = (high_20d - low_20d) / df['close']
    
    # Relative Strength Index (RSI): 14-day RSI
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Composite heuristic
    heuristics_matrix = (momentum + (atr * vol_ratio) + dir_movement + rsi) / 4
    
    return heuristics_matrix
