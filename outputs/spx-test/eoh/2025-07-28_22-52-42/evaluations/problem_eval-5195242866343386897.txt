import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the open and close prices
    df['log_open'] = np.log(df['open'])
    df['log_close'] = np.log(df['close'])
    
    # Calculate the 50-day Simple Moving Average (SMA) for log_open and log_close
    df['SMA_50_log_open'] = df['log_open'].rolling(window=50).mean()
    df['SMA_50_log_close'] = df['log_close'].rolling(window=50).mean()
    
    # Create the heuristic: difference between SMA_50_log_open and SMA_50_log_close
    heuristics_matrix = df['SMA_50_log_open'] - df['SMA_50_log_close']
    
    return heuristics_matrix
