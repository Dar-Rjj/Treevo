import pandas as pd
import numpy as np

def heuristics_v2(df):
    df['log_close'] = np.log(df['close'])
    ema_20_log_close = df['log_close'].ewm(span=20, adjust=False).mean()
    ema_50_log_close = df['log_close'].ewm(span=50, adjust=False).mean()
    diff_ema = ema_20_log_close - ema_50_log_close
    std_log_close = df['log_close'].rolling(window=20).std()
    heuristics_matrix = diff_ema / std_log_close
    return heuristics_matrix
