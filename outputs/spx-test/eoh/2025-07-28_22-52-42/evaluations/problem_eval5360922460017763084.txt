import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily price range
    price_range = df['high'] - df['low']
    # Calculate the 10-day EMA of the price range
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    # Calculate the 10-day standard deviation of the closing prices
    std_close = df['close'].rolling(window=10).std()
    # Generate the heuristics matrix as the EMA of the price range divided by the standard deviation of closing prices
    heuristics_matrix = ema_price_range / std_close
    return heuristics_matrix
