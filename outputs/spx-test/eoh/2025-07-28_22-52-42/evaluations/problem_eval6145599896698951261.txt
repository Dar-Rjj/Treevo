def heuristics_v2(df):
    def calculate_roc(series, n):
        return (series - series.shift(n)) / series.shift(n) * 100

    features = df.columns
    ma_short = df.rolling(window=10).mean()
    ma_long = df.rolling(window=50).mean()
    roc = df.apply(lambda x: calculate_roc(x, 10))
    
    heuristics_matrix = (ma_short - ma_long) + roc
    return heuristics_matrix
