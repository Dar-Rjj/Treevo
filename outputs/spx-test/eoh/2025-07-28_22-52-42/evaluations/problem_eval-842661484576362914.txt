import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 20-day Simple Moving Average (SMA) for log_high
    df['SMA_20_log_high'] = df['log_high'].rolling(window=20).mean()
    
    # Define weights for Weighted Moving Average (WMA)
    weights = np.arange(1, 21)
    weights = weights / weights.sum()
    
    # Calculate the 20-day Weighted Moving Average (WMA) for log_close
    df['WMA_20_log_close'] = df['log_close'].rolling(window=20).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Create the heuristic: difference between SMA_20_log_high and WMA_20_log_close
    heuristics_matrix = df['SMA_20_log_high'] - df['WMA_20_log_close']
    
    return heuristics_matrix
