def heuristics_v2(df):
    roc = df['close'].pct_change(periods=20)
    adosc = (df['high'] - df['close'].shift(1)) / (df['high'] - df['low']) * df['volume']
    adosc = adosc.rolling(window=14).sum() - adosc.rolling(window=28).sum()
    heuristics_matrix = (roc + adosc) / 2
    return heuristics_matrix
