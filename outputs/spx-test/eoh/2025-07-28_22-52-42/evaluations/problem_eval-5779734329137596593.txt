def heuristics_v2(df):
    # Calculate the average close price over a 5-day window
    avg_close = df['close'].rolling(window=5).mean()
    # Calculate the difference between the current day's close price and the 5-day average close price
    diff_close = df['close'] - avg_close
    # Form the heuristics matrix by dividing the difference in close prices by the volume
    heuristics_matrix = diff_close / df['volume']
    return heuristics_matrix
