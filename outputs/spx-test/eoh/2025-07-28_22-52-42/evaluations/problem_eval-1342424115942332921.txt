def heuristics_v2(df):
    # Calculate the ratio of the difference between high and low prices to the trading volume
    price_diff_to_volume = (df['high'] - df['low']) / df['volume']
    # Apply a logarithmic transformation to the price difference to volume ratio
    log_transformed_price_diff_to_volume = np.log(price_diff_to_volume + 1)
    # Generate the heuristics matrix as the sum of the log-transformed value and its square
    heuristics_matrix = log_transformed_price_diff_to_volume + log_transformed_price_diff_to_volume.pow(2)
    return heuristics_matrix
