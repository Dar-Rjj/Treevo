import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the log returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute the 14-day momentum (difference between current and 14 days ago close)
    df['momentum_14'] = df['close'] - df['close'].shift(14)
    
    # Compute the 14-day standard deviation of log returns
    df['std_log_returns_14'] = df['log_returns'].rolling(window=14).std()
    
    # Heuristic matrix: sum of momentum and standard deviation
    heuristics_matrix = df['momentum_14'] + df['std_log_returns_14']
    
    return heuristics_matrix
