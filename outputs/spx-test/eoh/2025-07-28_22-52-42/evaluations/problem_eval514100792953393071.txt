def heuristics_v2(df):
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 50-day moving average of volume
    vol_50d = df['volume'].rolling(window=50).mean()
    # Calculate the 20-day moving average of volume
    vol_20d = df['volume'].rolling(window=20).mean()
    # Volume ratio: 50-day MA of volume / 20-day MA of volume
    vol_ratio = vol_50d / vol_20d
    
    # Calculate the 14-day ATR
    tr = pd.DataFrame(index=df.index, columns=['tr'])
    tr['tr'] = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), abs(df['low'] - df['close'].shift(1)))
    atr = tr.rolling(window=14).mean()
    
    # Composite heuristic
    heuristics_matrix = (rsi + vol_ratio + atr['tr']) / 3
    
    return heuristics_matrix
