import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the open and low prices
    df['log_open'] = np.log(df['open'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 10-day standard deviation (STD) for log_open and log_low
    df['STD_10_log_open'] = df['log_open'].rolling(window=10).std()
    df['STD_10_log_low'] = df['log_low'].rolling(window=10).std()
    
    # Create the heuristic: product of STD_10_log_open and STD_10_log_low
    heuristics_matrix = df['STD_10_log_open'] * df['STD_10_log_low']
    
    return heuristics_matrix
