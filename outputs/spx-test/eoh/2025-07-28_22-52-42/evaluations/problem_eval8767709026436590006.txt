def heuristics_v2(df):
    # Calculate the long-term momentum (e.g., 240-day) of the close price
    long_momentum = df['close'].pct_change(240)
    
    # Calculate the standard deviation of the volume over a recent period (e.g., 30 days)
    vol_stability = df['volume'].rolling(window=30).std()
    
    # Generate the heuristics matrix as a combination of long-term momentum and volume stability
    heuristics_matrix = (long_momentum - vol_stability) / 2

    return heuristics_matrix
