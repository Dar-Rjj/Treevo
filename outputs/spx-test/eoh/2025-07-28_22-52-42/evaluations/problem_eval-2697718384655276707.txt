def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) for the close price
    roc_close = df['close'].pct_change(10)
    
    # Calculate the ratio of the current volume to the maximum volume over the past 60 days
    max_vol_60d = df['volume'].rolling(window=60).max()
    vol_ratio = df['volume'] / max_vol_60d
    
    # Compute the average of the last 10 high prices
    avg_high_10d = df['high'].rolling(window=10).mean()
    
    # Generate the heuristics matrix as a combination of ROC, volume ratio, and average high
    heuristics_matrix = (roc_close + vol_ratio + avg_high_10d) / 3
    
    return heuristics_matrix
