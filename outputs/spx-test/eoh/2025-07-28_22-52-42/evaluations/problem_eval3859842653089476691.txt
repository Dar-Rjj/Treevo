def heuristics_v2(df):
    # Exponential Moving Averages
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_diff = ema_10 - ema_30
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume Ratio: Today's volume compared to the maximum volume over the past 20 days
    max_vol_20d = df['volume'].rolling(window=20).max()
    vol_ratio = df['volume'] / max_vol_20d
    
    # Composite heuristic with different weights
    heuristics_matrix = (ema_diff * 0.5 + rsi * 0.3 + vol_ratio * 0.2)
    
    return heuristics_matrix
