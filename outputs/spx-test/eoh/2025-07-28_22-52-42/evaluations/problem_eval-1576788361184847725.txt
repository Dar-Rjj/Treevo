import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])

    # Calculate the 30-day Exponential Moving Average (EMA) for log_high
    df['EMA_30_log_high'] = df['log_high'].ewm(span=30, adjust=False).mean()

    # Define weights for Weighted Moving Average (WMA) with a linearly decreasing weight scheme
    lin_weights = np.linspace(30, 1, 30)
    lin_weights /= lin_weights.sum()

    # Calculate the 30-day Weighted Moving Average (WMA) for log_close
    df['WMA_30_log_close'] = df['log_close'].rolling(window=30).apply(lambda x: (x * lin_weights[::-1]).sum(), raw=True)

    # Create the heuristic: ratio of EMA_30_log_high to WMA_30_log_close
    heuristics_matrix = df['EMA_30_log_high'] / df['WMA_30_log_close']

    return heuristics_matrix
