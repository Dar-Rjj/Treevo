def heuristics_v2(df):
    # Calculate the ratio of the difference between high and low prices to the trading volume
    price_diff_to_volume = (df['high'] - df['low']) / df['volume']
    # Apply a non-linear transformation by squaring the price difference to volume ratio
    transformed_price_diff_to_volume = (price_diff_to_volume ** 2)
    # Generate the heuristics matrix as the sum of the transformed value with itself to emphasize the signal
    heuristics_matrix = transformed_price_diff_to_volume + transformed_price_diff_to_volume
    return heuristics_matrix
