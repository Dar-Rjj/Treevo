import pandas as pd

def heuristics_v2(df):
    def adx(series, period=14):
        high = df['high']
        low = df['low']
        up_move = high - high.shift(1)
        down_move = low.shift(1) - low
        plus_dm = pd.Series(0, index=up_move.index)
        minus_dm = pd.Series(0, index=down_move.index)
        plus_dm[(up_move > down_move) & (up_move > 0)] = up_move
        minus_dm[(down_move > up_move) & (down_move > 0)] = down_move
        tr = pd.concat([high - low, abs(high - df['close'].shift(1)), abs(low - df['close'].shift(1))], axis=1).max(axis=1)
        tr_smooth = tr.rolling(window=period).sum()
        plus_di = 100 * (plus_dm.rolling(window=period).sum() / tr_smooth)
        minus_di = 100 * (minus_dm.rolling(window=period).sum() / tr_smooth)
        dx = 100 * (abs(plus_di - minus_di) / (plus_di + minus_di))
        adx_values = dx.rolling(window=period).mean()
        return adx_values

    def obv(series):
        prev_obv = 0
        obv = [prev_obv]
        for i in range(1, len(df)):
            if df['close'][i] > df['close'][i-1]:
                current_obv = prev_obv + df['volume'][i]
            elif df['close'][i] < df['close'][i-1]:
                current_obv = prev_obv - df['volume'][i]
            else:
                current_obv = prev_obv
            obv.append(current_obv)
            prev_obv = current_obv
        return pd.Series(obv, index=df.index)

    adx_values = adx(df['close'])
    obv_values = obv(df['close'])

    heuristics_matrix = (adx_values + obv_values) / 2
    
    return heuristics_matrix
