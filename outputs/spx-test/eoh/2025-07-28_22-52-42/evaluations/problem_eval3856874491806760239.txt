def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 30-day Simple Moving Average (SMA) for log_high
    df['SMA_30_log_high'] = df['log_high'].rolling(window=30).mean()
    
    # Calculate the 30-day Exponentially Weighted Moving Average (EWMA) for log_close
    df['EWMA_30_log_close'] = df['log_close'].ewm(span=30, adjust=False).mean()
    
    # Create the heuristic: ratio between SMA_30_log_high and EWMA_30_log_close
    heuristics_matrix = df['SMA_30_log_high'] / df['EWMA_30_log_close']
    
    return heuristics_matrix
