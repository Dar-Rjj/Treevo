import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the open and close prices
    df['log_open'] = np.log(df['open'])
    df['log_close'] = np.log(df['close'])
    
    # Calculate the 20-day Exponential Moving Average (EMA) for log_open
    df['EMA_20_log_open'] = df['log_open'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 10-day Standard Deviation (STD) for log_close
    df['STD_10_log_close'] = df['log_close'].rolling(window=10).std()
    
    # Create the heuristic: difference between EMA_20_log_open and STD_10_log_close
    heuristics_matrix = df['EMA_20_log_open'] - df['STD_10_log_close']
    
    return heuristics_matrix
