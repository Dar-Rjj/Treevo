def heuristics_v2(df):
    # Calculate the natural logarithm of the open prices and volume
    df['log_open'] = np.log(df['open'])
    df['log_volume'] = np.log(df['volume'])
    
    # Calculate the 10-day cumulative sum for log_volume
    df['cumsum_10_log_volume'] = df['log_volume'].rolling(window=10).sum()
    
    # Calculate the 40-day cumulative sum for log_open
    df['cumsum_40_log_open'] = df['log_open'].rolling(window=40).sum()
    
    # Create the heuristic: ratio of cumsum_10_log_volume to cumsum_40_log_open
    heuristics_matrix = df['cumsum_10_log_volume'] / df['cumsum_40_log_open']
    
    return heuristics_matrix
