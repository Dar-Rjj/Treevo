def heuristics_v2(df):
    # Calculate the short-term momentum (e.g., 5-day) of the close price
    short_momentum = df['close'].pct_change(5)
    
    # Calculate the average volume over a recent period (e.g., 10 days) and the long-term period (e.g., 60 days)
    avg_vol_recent = df['volume'].rolling(window=10).mean()
    avg_vol_long_term = df['volume'].rolling(window=60).mean()
    
    # Compute the relative difference in volume
    vol_diff = (avg_vol_recent - avg_vol_long_term) / avg_vol_long_term
    
    # Generate the heuristics matrix as a combination of momentum and volume difference
    heuristics_matrix = (short_momentum + vol_diff) / 2

    return heuristics_matrix
