import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 20-day EMA of the daily returns
    df['EMA_20_daily_return'] = df['daily_return'].ewm(span=20).mean()
    
    # Calculate the 20-day standard deviation of the volume
    df['std_20_volume'] = df['volume'].rolling(window=20).std()
    
    # Generate the heuristics matrix as the EMA of the daily return adjusted by the standard deviation of the volume
    heuristics_matrix = df['EMA_20_daily_return'] / df['std_20_volume']
    
    return heuristics_matrix
