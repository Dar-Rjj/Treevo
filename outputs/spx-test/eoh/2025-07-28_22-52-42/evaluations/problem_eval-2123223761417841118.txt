    # Calculate 20-day simple moving average of close prices
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate standard deviation (volatility) of close prices over 20 days
    std_close = df['close'].rolling(window=20).std()
    
    # Adjusted volume: Volume * (Close - SMA(Close)) / Volatility
    adj_volume = df['volume'] * (df['close'] - sma_close) / std_close
    
    # Heuristic value: Rolling sum of adjusted volume over 10 days
    heuristics_matrix = adj_volume.rolling(window=10).sum()
    
    return heuristics_matrix
