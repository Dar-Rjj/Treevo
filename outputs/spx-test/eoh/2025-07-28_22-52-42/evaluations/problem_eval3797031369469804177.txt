def heuristics_v2(df):
    # Calculate the 50-day Simple Moving Average (SMA) for volume
    df['SMA_50_volume'] = df['volume'].rolling(window=50).mean()
    
    # Calculate the 20-day Exponential Moving Average (EMA) for close
    df['EMA_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Create the heuristic: ratio of SMA_50_volume to EMA_20_close
    heuristics_matrix = df['SMA_50_volume'] / df['EMA_20_close']
    
    return heuristics_matrix
