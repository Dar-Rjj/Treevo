import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day rate of change (ROC) for the close price
    roc_close = df['close'].pct_change(5)
    
    # Calculate the ratio of the current volume to the standard deviation of volume over the past 30 days
    std_vol_30d = df['volume'].rolling(window=30).std()
    vol_ratio = df['volume'] / std_vol_30d
    
    # Compute the cumulative sum of the last 5 day's returns
    cum_return_5d = df['close'].pct_change().rolling(window=5).sum()
    
    # Generate the heuristics matrix as a combination of ROC, volume ratio, and cumulative return
    heuristics_matrix = (roc_close + vol_ratio + cum_return_5d) / 3
    
    return heuristics_matrix
