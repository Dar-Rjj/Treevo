def heuristics_v2(df):
    # Rate of change: 10-day percentage change in closing price
    roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Relative Strength Index (RSI): 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Moving average: 50-day simple moving average
    sma_50 = df['close'].rolling(window=50).mean()
    ma_diff = df['close'] - sma_50
    
    # Composite heuristic
    heuristics_matrix = roc + rsi + ma_diff
    
    return heuristics_matrix
