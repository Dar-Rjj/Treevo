def heuristics_v2(df):
    # Calculate 20-day price change as a proxy for momentum
    df['momentum_20d'] = df['close'].pct_change(20)
    
    # Estimate volatility using the 14-day average true range (ATR)
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['atr_14d'] = df['true_range'].rolling(window=14).mean()
    
    # Measure liquidity using the percentage change in volume compared to its 10-day minimum
    df['min_vol_10d'] = df['volume'].rolling(window=10).min()
    df['vol_liq'] = (df['volume'] - df['min_vol_10d']) / df['min_vol_10d']
    
    # Composite heuristic
    heuristics_matrix = df['momentum_20d'] + df['atr_14d'] * 100 + df['vol_liq']
    
    return heuristics_matrix
