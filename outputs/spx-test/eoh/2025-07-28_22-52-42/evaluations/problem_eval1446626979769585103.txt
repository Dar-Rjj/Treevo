def heuristics_v2(df):
    sma_10_volume = df['volume'].rolling(window=10).mean()
    sma_30_volume = df['volume'].rolling(window=30).mean()
    diff_sma_volume = sma_10_volume - sma_30_volume
    std_close = df['close'].rolling(window=20).std()
    heuristics_matrix = diff_sma_volume / std_close
    return heuristics_matrix
