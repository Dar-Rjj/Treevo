def heuristics_v2(df):
    # Price acceleration: 15-day change in the rate of closing price change
    price_change = df['close'].diff()
    acceleration = price_change - price_change.shift(15)
    
    # Relative Strength Index (RSI): 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Accumulation/Distribution Line: measures buying and selling pressure
    adl = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    adl *= df['volume']
    adl = adl.cumsum()
    
    # Composite heuristic with different weights
    heuristics_matrix = (acceleration * 0.4 + rsi * 0.3 + adl * 0.3)
    
    return heuristics_matrix
