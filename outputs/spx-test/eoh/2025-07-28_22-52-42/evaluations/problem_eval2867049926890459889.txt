import pandas as pd

def heuristics_v2(df):
    def dmi(series, period=14):
        high = df['high']
        low = df['low']
        up_move = high - high.shift(1)
        down_move = low.shift(1) - low
        plus_dm = pd.Series(0, index=up_move.index)
        minus_dm = pd.Series(0, index=down_move.index)
        plus_dm[(up_move > down_move) & (up_move > 0)] = up_move
        minus_dm[(down_move > up_move) & (down_move > 0)] = down_move
        tr = pd.concat([high - low, abs(high - df['close'].shift(1)), abs(low - df['close'].shift(1))], axis=1).max(axis=1)
        tr_smooth = tr.rolling(window=period).sum()
        plus_di = 100 * (plus_dm.rolling(window=period).sum() / tr_smooth)
        minus_di = 100 * (minus_dm.rolling(window=period).sum() / tr_smooth)
        dmi = plus_di - minus_di
        return dmi.ewm(span=period, adjust=False).mean()

    def mfi(series, period=14):
        typical_price = (df['high'] + df['low'] + df['close']) / 3
        raw_money_flow = typical_price * df['volume']
        positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
        negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
        money_flow_ratio = positive_money_flow.rolling(window=period).sum() / negative_money_flow.rolling(window=period).sum()
        mfi = 100 - (100 / (1 + money_flow_ratio))
        return mfi

    dmi_values = dmi(df['close'])
    mfi_values = mfi(df['close'])

    heuristics_matrix = (dmi_values + mfi_values) / 2
    
    return heuristics_matrix
