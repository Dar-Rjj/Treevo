def heuristics_v2(df):
    # Rate of change: 5-day percentage change in closing price
    roc = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Volume ratio: current volume compared to its 50-day minimum
    min_vol_50d = df['volume'].rolling(window=50).min()
    vol_ratio = df['volume'] / min_vol_50d
    
    # Volatility: 14-day average true range
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': abs(df['high'] - df['close'].shift(1)),
        'l-pc': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Composite heuristic
    heuristics_matrix = (roc + vol_ratio + atr) / 3
    
    return heuristics_matrix
