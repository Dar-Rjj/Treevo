import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the low prices and volume
    df['log_low'] = np.log(df['low'])
    df['log_volume'] = np.log(df['volume'])
    
    # Calculate the 50-day Simple Moving Average (SMA) for log_low
    df['SMA_50_log_low'] = df['log_low'].rolling(window=50).mean()
    
    # Calculate the 50-day cumulative sum for log_volume
    df['cumsum_50_log_volume'] = df['log_volume'].rolling(window=50).sum()
    
    # Create the heuristic: ratio of cumsum_50_log_volume to SMA_50_log_low
    heuristics_matrix = df['cumsum_50_log_volume'] / df['SMA_50_log_low']
    
    return heuristics_matrix
