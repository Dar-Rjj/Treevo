import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum of closing prices over a 5-day window
    close_momentum = df['close'].pct_change(5)
    # Calculate the change in trading volume over a 5-day window
    volume_change = df['volume'].diff(5)
    # Generate the heuristics matrix as the product of the close momentum and the volume change
    heuristics_matrix = close_momentum * volume_change
    return heuristics_matrix
