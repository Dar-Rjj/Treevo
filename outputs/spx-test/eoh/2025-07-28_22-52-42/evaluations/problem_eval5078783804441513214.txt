import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and low prices
    df['log_close'] = np.log(df['close'])
    df['log_low'] = np.log(df['low'])

    # Calculate the 20-day Exponential Moving Average (EMA) for log_low
    df['EMA_20_log_low'] = df['log_low'].ewm(span=20, adjust=False).mean()

    # Define weights for Weighted Moving Average (WMA) with an exponentially decreasing weight scheme
    exp_weights = np.exp(np.linspace(0, -1, 20))
    exp_weights /= exp_weights.sum()

    # Calculate the 20-day Weighted Moving Average (WMA) for log_close
    df['WMA_20_log_close'] = df['log_close'].rolling(window=20).apply(lambda x: (x * exp_weights[::-1]).sum(), raw=True)

    # Create the heuristic: ratio of EMA_20_log_low to WMA_20_log_close
    heuristics_matrix = df['EMA_20_log_low'] / df['WMA_20_log_close']

    return heuristics_matrix
