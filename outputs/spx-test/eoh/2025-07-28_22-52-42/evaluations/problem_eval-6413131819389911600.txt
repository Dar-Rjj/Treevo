def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day Moving Average (MA) for daily returns
    df['MA_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 5-day and 20-day rolling sum of volume
    df['sum_vol_5'] = df['volume'].rolling(window=5).sum()
    df['sum_vol_20'] = df['volume'].rolling(window=20).sum()
    
    # Calculate the ratio of 5-day rolling sum of volume to 20-day rolling sum of volume
    vol_ratio = df['sum_vol_5'] / df['sum_vol_20']
    
    # Create the heuristic: product of MA_10_daily_return and vol_ratio
    heuristics_matrix = df['MA_10_daily_return'] * vol_ratio
    
    return heuristics_matrix
