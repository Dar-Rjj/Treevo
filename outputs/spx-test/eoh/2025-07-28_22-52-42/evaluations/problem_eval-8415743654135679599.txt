import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving average
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    # Calculate momentum
    momentum = df['close'] - df['close'].shift(1)
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    # Volume change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    heuristics_matrix = pd.DataFrame({
        'SMA_5': sma_5,
        'SMA_10': sma_10,
        'Momentum': momentum,
        'Price_Range': price_range,
        'Volume_Change': volume_change
    })
    
    # Convert to Series for output
    heuristics_series = heuristics_matrix.apply(tuple, axis=1)
    heuristics_series.index = df.index  # Ensure the index is date
    
    return heuristics_matrix
