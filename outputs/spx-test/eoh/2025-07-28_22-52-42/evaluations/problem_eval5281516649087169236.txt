import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 5-day moving average of the high prices
    ma_5_high = df['high'].rolling(window=5).mean()
    
    # Calculate the 10-day moving average of the low prices
    ma_10_low = df['low'].rolling(window=10).mean()
    
    # Calculate the 30-day sum of the differences between the daily high and low prices
    sum_30_range = (df['high'] - df['low']).rolling(window=30).sum()
    
    # Generate the heuristics matrix as the product of the (ma_5_high - ma_10_low) and sum_30_range
    heuristics_matrix = (ma_5_high - ma_10_low) * sum_30_range
    
    return heuristics_matrix
