import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the open and high prices
    df['log_open'] = np.log(df['open'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 20-day Exponential Moving Average (EMA) for log_high
    df['EMA_20_log_high'] = df['log_high'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 50-day Exponential Moving Average (EMA) for log_open
    df['EMA_50_log_open'] = df['log_open'].ewm(span=50, adjust=False).mean()
    
    # Create the heuristic: difference between EMA_20_log_high and EMA_50_log_open
    heuristics_matrix = df['EMA_20_log_high'] - df['EMA_50_log_open']
    
    return heuristics_matrix
