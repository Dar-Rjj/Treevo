def heuristics_v2(df):
    # Momentum: 21-day change in closing price
    momentum = df['close'] - df['close'].shift(21)
    
    # Volatility: 30-day standard deviation of closing price
    vol = df['close'].rolling(window=30).std()
    
    # Volume ratio: current volume compared to its 50-day moving average
    avg_vol_50d = df['volume'].rolling(window=50).mean()
    vol_ratio = df['volume'] / avg_vol_50d
    
    # Directional movement: difference between 30-day high and low
    high_30d = df['high'].rolling(window=30).max()
    low_30d = df['low'].rolling(window=30).min()
    dir_movement = (high_30d - low_30d) / df['close']
    
    # Composite heuristic
    heuristics_matrix = (momentum + (vol * vol_ratio) + dir_movement) / 3
    
    return heuristics_matrix
