import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day Exponential Moving Average (EMA) for high and low prices
    ema_10_high = df['high'].ewm(span=10, adjust=False).mean()
    ema_10_low = df['low'].ewm(span=10, adjust=False).mean()
    
    # Calculate the difference between 10-day EMA of high and low prices
    diff_ema_high_low = ema_10_high - ema_10_low
    
    # Calculate 5-day Exponential Moving Average (EMA) for volume
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Generate the heuristics matrix by dividing the difference in EMA of high and low by the EMA of volume
    heuristics_matrix = diff_ema_high_low / ema_5_volume
    
    return heuristics_matrix
