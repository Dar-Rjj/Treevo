def heuristics_v2(df):
    # Calculate the 20-day Exponentially Weighted Moving Average (EWMA) for volume
    df['EWMA_20_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 10-day Simple Moving Average (SMA) for high prices
    df['SMA_10_high'] = df['high'].rolling(window=10).mean()
    
    # Calculate the 5-day Rate of Change (ROC) for closing prices
    df['ROC_5_close'] = df['close'].pct_change(periods=5)
    
    # Create the heuristic: EWMA_20_volume divided by SMA_10_high, then multiplied by ROC_5_close
    heuristics_matrix = (df['EWMA_20_volume'] / df['SMA_10_high']) * df['ROC_5_close']
    
    return heuristics_matrix
