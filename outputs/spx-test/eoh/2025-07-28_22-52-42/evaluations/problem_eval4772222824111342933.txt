def heuristics_v2(df):
    # Calculate the ratio of the difference between high and low prices
    hl_ratio = (df['high'] - df['low']) / df[['high', 'low']].mean(axis=1)
    # Calculate the rolling mean of volume over a 10-day window
    volume_mean = df['volume'].rolling(window=10).mean()
    # Apply a non-linear transformation (square root) to the hl_ratio and inverse of volume_mean
    transformed_hl_ratio = hl_ratio ** 0.5
    transformed_volume_mean = 1 / volume_mean
    # Generate the heuristics matrix as the product of the two transformed values
    heuristics_matrix = transformed_hl_ratio * transformed_volume_mean
    return heuristics_matrix
