def heuristics_v2(df):
    # Momentum: 21-day rate of change in closing price
    momentum = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    
    # Volatility: 25-day standard deviation of volume
    vol = df['volume'].rolling(window=25).std()
    
    # Directional movement: 10-day average true range
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=10).mean()
    
    # Composite heuristic
    heuristics_matrix = (momentum + (vol / df['volume']) + atr) / 3
    
    return heuristics_matrix
