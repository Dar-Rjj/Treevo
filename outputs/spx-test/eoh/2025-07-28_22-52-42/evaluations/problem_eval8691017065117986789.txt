import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the ratio of consecutive days' closing prices to their opening prices
    close_open_ratio = (df['close'] / df['open']).pct_change().shift(-1)
    # Apply natural logarithm to the volume
    log_volume = np.log(df['volume'])
    # Generate the heuristics matrix as the product of the close-open ratio and the log of volume
    heuristics_matrix = close_open_ratio * log_volume
    return heuristics_matrix
