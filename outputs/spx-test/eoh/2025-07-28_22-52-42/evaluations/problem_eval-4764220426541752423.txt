def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 20-day Exponential Moving Average (EMA) for log_high
    df['EMA_20_log_high'] = df['log_high'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 20-day Simple Moving Average (SMA) for log_close
    df['SMA_20_log_close'] = df['log_close'].rolling(window=20).mean()
    
    # Create the heuristic: difference between EMA_20_log_high and SMA_20_log_close
    heuristics_matrix = df['EMA_20_log_high'] - df['SMA_20_log_close']
    
    return heuristics_matrix
