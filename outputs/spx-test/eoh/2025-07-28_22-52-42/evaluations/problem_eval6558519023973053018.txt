def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    # Measure the difference between open and close
    df['open_close_diff'] = df['close'] - df['open']
    # Calculate a ratio indicating where the closing price is relative to the day's range
    df['close_range_ratio'] = (df['close'] - df['low']) / df['range']
    # Generate an indicator of the relationship between the open price and the day's midpoint
    df['open_midpoint_diff'] = df['open'] - (df['high'] + df['low']) / 2
    # Aggregate the indicators into a single heuristic value
    heuristics_matrix = 0.5 * df['open_close_diff'] + 0.3 * df['close_range_ratio'] + 0.2 * df['open_midpoint_diff']
    return heuristics_matrix
