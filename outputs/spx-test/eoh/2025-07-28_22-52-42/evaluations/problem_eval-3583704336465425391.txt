import pandas as pd
import numpy as pd

def heuristics_v2(df):
    # Calculate the natural logarithm of the low and volume
    df['log_low'] = np.log(df['low'])
    df['log_volume'] = np.log(df['volume'])

    # Calculate the 20-day Exponential Moving Average (EMA) for log_volume
    df['EMA_20_log_volume'] = df['log_volume'].ewm(span=20, adjust=False).mean()

    # Define weights for Weighted Moving Average (WMA) with an exponentially decreasing weight scheme
    exp_weights = np.exp(-np.linspace(0, 1, 15))
    exp_weights /= exp_weights.sum()

    # Calculate the 15-day Weighted Moving Average (WMA) for log_low
    df['WMA_15_log_low'] = df['log_low'].rolling(window=15).apply(lambda x: (x * exp_weights[::-1]).sum(), raw=True)

    # Create the heuristic: ratio of EMA_20_log_volume to WMA_15_log_low
    heuristics_matrix = df['EMA_20_log_volume'] / df['WMA_15_log_low']

    return heuristics_matrix
