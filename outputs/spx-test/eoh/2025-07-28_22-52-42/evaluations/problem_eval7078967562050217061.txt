def heuristics_v2(df):
    # Calculate the natural logarithm of the close, high, and low prices
    df['log_high'] = np.log(df['high'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 30-day Exponential Moving Average (EMA) for log_high
    df['EMA_30_log_high'] = df['log_high'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 30-day Simple Moving Average (SMA) for log_low
    df['SMA_30_log_low'] = df['log_low'].rolling(window=30).mean()
    
    # Create the heuristic: difference between EMA_30_log_high and SMA_30_log_low
    heuristics_matrix = df['EMA_30_log_high'] - df['SMA_30_log_low']
    
    return heuristics_matrix
