import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close prices and volume
    df['log_close'] = np.log(df['close'])
    df['log_volume'] = np.log(df['volume'])
    
    # Calculate the 20-day Exponentially Weighted Moving Average (EWMA) for log_close
    df['EWMA_20_log_close'] = df['log_close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 10-day Simple Moving Average (SMA) for log_volume
    df['SMA_10_log_volume'] = df['log_volume'].rolling(window=10).mean()
    
    # Create the heuristic: difference between EWMA_20_log_close and SMA_10_log_volume
    heuristics_matrix = df['EWMA_20_log_close'] - df['SMA_10_log_volume']
    
    return heuristics_matrix
