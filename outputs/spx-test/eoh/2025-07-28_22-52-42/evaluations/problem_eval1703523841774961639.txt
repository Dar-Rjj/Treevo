import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the weighted average of high and low prices
    weighted_avg = 0.7 * df['high'] + 0.3 * df['low']
    # Divide the weighted average by the closing price
    price_ratio = weighted_avg / df['close']
    # Take the square root of the volume
    sqrt_volume = np.sqrt(df['volume'])
    # Generate the heuristics matrix as the product of the price ratio and the square root of volume
    heuristics_matrix = price_ratio * sqrt_volume
    return heuristics_matrix
