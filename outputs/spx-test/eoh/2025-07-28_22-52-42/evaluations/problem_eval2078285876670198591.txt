def heuristics_v2(df):
    # Rate of change in Average True Range (ATR): 14-day percentage change in ATR
    df['tr'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    atr = df['tr'].rolling(window=14).mean()
    atr_roc = atr.pct_change(14)
    
    # Accumulation/Distribution Line: cumulative flow of money into or out of a security
    df['mfm'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['mfv'] = df['mfm'] * df['volume']
    adl = df['mfv'].cumsum()
    
    # Relative Strength Index (RSI): 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Composite heuristic
    heuristics_matrix = (atr_roc + adl + rsi) / 3
    
    return heuristics_matrix
