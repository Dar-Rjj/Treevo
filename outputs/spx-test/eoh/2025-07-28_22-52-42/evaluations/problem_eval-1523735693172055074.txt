import pandas as pd

def heuristics_v2(df):
    # Calculate simple return
    df['simple_return'] = df['close'].pct_change()
    # Momentum: 12-months simple moving average return
    df['momentum'] = df['simple_return'].rolling(window=12).mean()
    # Volatility: standard deviation of daily returns over the last 12 months
    df['volatility'] = df['simple_return'].rolling(window=12).std()
    # Liquidity: ratio of volume to the 12-months average volume
    df['avg_volume_12m'] = df['volume'].rolling(window=12).mean()
    df['liquidity'] = df['volume'] / df['avg_volume_12m']
    # Composite heuristic, assuming equal weighting for simplicity
    df['heuristic'] = (df['momentum'] + 1/df['volatility'] + df['liquidity']) / 3
    heuristics_matrix = df['heuristic'].dropna()
    
    return heuristics_matrix
