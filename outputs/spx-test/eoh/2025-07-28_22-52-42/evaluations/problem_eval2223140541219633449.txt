import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the open and close prices
    df['log_open'] = np.log(df['open'])
    df['log_close'] = np.log(df['close'])

    # Calculate the 30-day Exponential Moving Average (EMA) for log_close
    df['EMA_30_log_close'] = df['log_close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 5-day Standard Deviation (STD) for log_open
    df['STD_5_log_open'] = df['log_open'].rolling(window=5).std()

    # Create the heuristic: difference between EMA_30_log_close and STD_5_log_open
    heuristics_matrix = df['EMA_30_log_close'] - df['STD_5_log_open']
    
    return heuristics_matrix
