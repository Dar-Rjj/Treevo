def heuristics_v2(df):
    # Calculate the 20-day rate of change of the close price
    roc_20 = df['close'].pct_change(20)
    
    # Calculate the 30-day standard deviation of the daily returns
    daily_returns = df['close'].pct_change()
    std_dev_30 = daily_returns.rolling(window=30).std()
    
    # Generate the heuristics matrix as a combination of ROC and standard deviation
    heuristics_matrix = 0.6 * roc_20 + 0.4 * std_dev_30

    return heuristics_matrix
