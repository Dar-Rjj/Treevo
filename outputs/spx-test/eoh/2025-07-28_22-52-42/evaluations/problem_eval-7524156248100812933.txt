import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of consecutive days' closing prices
    close_ratio = df['close'].pct_change().shift(-1)
    # Calculate the absolute change in volume
    abs_volume_change = df['volume'].diff().abs()
    # Apply a non-linear transformation to the close ratio and absolute volume change
    transformed_close_ratio = close_ratio ** 2
    # Generate the 10-day rolling standard deviation of the closing price and apply a square root transformation
    std_dev_10day = (df['close'].rolling(window=10).std()) ** 0.5
    # Generate the heuristics matrix as the product of the two transformed values and the 10-day rolling standard deviation
    heuristics_matrix = transformed_close_ratio * (1 / abs_volume_change) * std_dev_10day
    return heuristics_matrix
