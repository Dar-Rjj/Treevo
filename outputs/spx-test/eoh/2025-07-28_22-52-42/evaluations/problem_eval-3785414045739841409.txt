import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in closing prices
    close_rate_of_change = df['close'].pct_change()
    # Calculate the standard deviation of volume
    vol_std = df['volume'].rolling(window=5).std()
    # Calculate the mean of volume
    vol_mean = df['volume'].rolling(window=5).mean()
    # Calculate the coefficient of variation of volume
    vol_cv = vol_std / vol_mean
    # Generate the heuristics matrix as the product of the close rate of change and the coefficient of variation of volume
    heuristics_matrix = close_rate_of_change * vol_cv
    return heuristics_matrix
