def heuristics_v2(df):
    # Calculate the 5-day rate of change (ROC) for the close price
    roc_close = df['close'].pct_change(5)
    
    # Calculate the ratio of the current volume to the minimum volume over the past 30 days
    min_vol_30d = df['volume'].rolling(window=30).min()
    vol_ratio = df['volume'] / min_vol_30d
    
    # Compute the average of the last 5 high prices
    avg_high_5d = df['high'].rolling(window=5).mean()
    
    # Generate the heuristics matrix as a combination of ROC, volume ratio, and average high
    heuristics_matrix = (roc_close + vol_ratio + avg_high_5d) / 3
    
    return heuristics_matrix
