import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the high and low prices over a 20-day period
    avg_high_low = (df['high'] + df['low']) / 2
    # Calculate the ratio of the close price to the average of the high and low prices
    close_to_avg_ratio = df['close'] / avg_high_low
    # Compute the standard deviation of the ratio over a 20-day period
    heuristics_matrix = close_to_avg_ratio.rolling(window=20).std()
    return heuristics_matrix
