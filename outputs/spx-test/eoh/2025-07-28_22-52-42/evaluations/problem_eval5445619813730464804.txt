import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the ratio of consecutive days' closing prices to their low prices
    close_low_ratio = (df['close'] / df['low']).pct_change().shift(-1)
    # Apply square root to the volume
    sqrt_volume = np.sqrt(df['volume'])
    # Generate the heuristics matrix as the product of the close-low ratio and the square root of volume
    heuristics_matrix = close_low_ratio * sqrt_volume
    return heuristics_matrix
