import numpy as np

def heuristics_v2(df):
    # Calculate the ratio of the difference between high and low prices to the trading volume
    price_diff_to_volume = (df['high'] - df['low']) / df['volume']
    # Apply a logarithmic non-linear transformation to the price difference to volume ratio
    log_transformed_price_diff_to_volume = np.log1p(price_diff_to_volume)
    # Generate the heuristics matrix as the sum of the transformed value and the original price difference to volume ratio
    heuristics_matrix = log_transformed_price_diff_to_volume + price_diff_to_volume
    return heuristics_matrix
