def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])

    # Calculate the 50-day Simple Moving Average (SMA) for log_high
    df['SMA_50_log_high'] = df['log_high'].rolling(window=50).mean()

    # Calculate the 20-day Exponentially Weighted Moving Average (EWMA) for log_close with a span of 20
    df['EWMA_20_log_close'] = df['log_close'].ewm(span=20, adjust=False).mean()

    # Create the heuristic: difference between SMA_50_log_high and EWMA_20_log_close
    heuristics_matrix = df['SMA_50_log_high'] - df['EWMA_20_log_close']

    return heuristics_matrix
