import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day rate of change of the closing price
    roc_5 = df['close'].pct_change(periods=5)
    # Calculate the 20-day standard deviation of the trading volume
    std_volume = df['volume'].rolling(window=20).std()
    # Generate the heuristics matrix by multiplying the ROC with the standard deviation of volume
    heuristics_matrix = roc_5 * std_volume
    return heuristics_matrix
