def heuristics_v2(df):
    # Momentum: 14-day change in closing price
    momentum = df['close'] - df['close'].shift(14)
    
    # Volatility: 20-day standard deviation of closing price
    vol = df['close'].rolling(window=20).std()
    
    # Volume ratio: current volume compared to its 30-day moving average
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    vol_ratio = df['volume'] / avg_vol_30d
    
    # Directional movement: difference between 20-day high and low
    high_20d = df['high'].rolling(window=20).max()
    low_20d = df['low'].rolling(window=20).min()
    dir_movement = (high_20d - low_20d) / df['close']
    
    # Composite heuristic
    heuristics_matrix = (momentum + (vol * vol_ratio) + dir_movement) / 3
    
    return heuristics_matrix
