import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the difference between closing and opening prices
    price_diff = df['close'] - df['open']
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    # Apply square root to the volume
    sqrt_volume = np.sqrt(df['volume'])
    # Generate the heuristics matrix as the product of the normalized price difference and the square root of volume
    heuristics_matrix = (price_diff / avg_price) * sqrt_volume
    return heuristics_matrix
