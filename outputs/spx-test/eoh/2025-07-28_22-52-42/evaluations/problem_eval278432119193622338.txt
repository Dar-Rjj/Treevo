def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=5)  # Calculate 5-day momentum
    rel_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)  # Relative change in volume
    heuristics_matrix = momentum + rel_vol_change  # Combine momentum and relative volume change
    return heuristics_matrix
