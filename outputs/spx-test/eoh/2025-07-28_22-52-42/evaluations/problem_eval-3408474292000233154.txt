import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 20-day Exponential Moving Average (EMA) for log_high
    df['EMA_20_log_high'] = df['log_high'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 50-day Exponential Moving Average (EMA) for log_close
    df['EMA_50_log_close'] = df['log_close'].ewm(span=50, adjust=False).mean()
    
    # Create the heuristic: ratio between EMA_20_log_high and EMA_50_log_close
    heuristics_matrix = df['EMA_20_log_high'] / df['EMA_50_log_close']
    
    return heuristics_matrix
