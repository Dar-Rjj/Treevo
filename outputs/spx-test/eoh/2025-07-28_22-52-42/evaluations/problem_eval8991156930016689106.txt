import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the adjusted close and volume
    df['log_adj_close'] = np.log(df['adj_close'])
    df['log_volume'] = np.log(df['volume'])
    
    # Calculate the 30-day Simple Moving Average (SMA) for log_volume
    df['SMA_30_log_volume'] = df['log_volume'].rolling(window=30).mean()
    
    # Define weights for Weighted Moving Average (WMA) with linear decay
    weights = np.linspace(1, 0.1, 30)
    weights = weights / weights.sum()
    
    # Calculate the 30-day Weighted Moving Average (WMA) for log_adj_close
    df['WMA_30_log_adj_close'] = df['log_adj_close'].rolling(window=30).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Create the heuristic: difference between SMA_30_log_volume and WMA_30_log_adj_close
    heuristics_matrix = df['SMA_30_log_volume'] - df['WMA_30_log_adj_close']
    
    return heuristics_matrix
