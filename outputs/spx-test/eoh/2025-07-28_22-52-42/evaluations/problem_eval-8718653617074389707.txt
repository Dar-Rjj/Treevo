import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the open and low prices
    df['log_open'] = np.log(df['open'])
    df['log_low'] = np.log(df['low'])

    # Calculate the 30-day Simple Moving Average (SMA) for log_low
    df['SMA_30_log_low'] = df['log_low'].rolling(window=30).mean()

    # Define weights for Weighted Moving Average (WMA)
    weights = np.exp(np.linspace(0, 1, 30))
    weights = weights / weights.sum()

    # Calculate the 30-day Weighted Moving Average (WMA) for log_open
    df['WMA_30_log_open'] = df['log_open'].rolling(window=30).apply(lambda x: (x * weights).sum(), raw=True)

    # Create the heuristic: difference between SMA_30_log_low and WMA_30_log_open
    heuristics_matrix = df['SMA_30_log_low'] - df['WMA_30_log_open']

    return heuristics_matrix
