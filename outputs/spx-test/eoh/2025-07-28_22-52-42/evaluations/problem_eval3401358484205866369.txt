import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the logarithmic difference between high and low prices
    log_diff_high_low = np.log(df['high'] / df['low'])
    # Calculate the exponential moving average of volume over a 5-day window
    exp_moving_avg_volume = df['volume'].ewm(span=5, adjust=False).mean()
    # Generate the heuristics matrix as the product of the two values
    heuristics_matrix = log_diff_high_low * exp_moving_avg_volume
    return heuristics_matrix
