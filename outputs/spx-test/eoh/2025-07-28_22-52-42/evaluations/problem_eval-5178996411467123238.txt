import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of consecutive days' closing prices
    close_ratio = df['close'].pct_change().shift(-1)
    # Calculate the rolling mean of volume over a 5-day window
    volume_mean = df['volume'].rolling(window=5).mean()
    # Apply a non-linear transformation to the close ratio and volume mean
    transformed_close_ratio = (close_ratio - close_ratio.min()) / (close_ratio.max() - close_ratio.min())
    transformed_volume_mean = (volume_mean - volume_mean.min()) / (volume_mean.max() - volume_mean.min())
    # Generate the heuristics matrix as the product of the two transformed values
    heuristics_matrix = transformed_close_ratio * transformed_volume_mean
    return heuristics_matrix
