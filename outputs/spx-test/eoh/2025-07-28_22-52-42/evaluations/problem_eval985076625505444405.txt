import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the low prices
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 10-day Simple Moving Average (SMA) for log_low
    df['SMA_10_log_low'] = df['log_low'].rolling(window=10).mean()
    
    # Calculate the percentage change in closing prices
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the 20-day cumulative sum of the percentage change in closing prices
    df['cum_sum_20_pct_change_close'] = df['pct_change_close'].rolling(window=20).sum()
    
    # Create the heuristic: difference between SMA_10_log_low and cum_sum_20_pct_change_close
    heuristics_matrix = df['SMA_10_log_low'] - df['cum_sum_20_pct_change_close']
    
    return heuristics_matrix
