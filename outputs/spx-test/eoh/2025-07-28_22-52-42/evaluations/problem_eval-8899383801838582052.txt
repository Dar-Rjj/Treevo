import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 20-day Exponential Moving Average (EMA) for high prices
    df['EMA_20_high'] = df['high'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 100-day Exponential Moving Average (EMA) for low prices
    df['EMA_100_low'] = df['low'].ewm(span=100, adjust=False).mean()
    
    # Create the heuristic: difference between EMA_20_high and EMA_100_low
    heuristics_matrix = df['EMA_20_high'] - df['EMA_100_low']
    
    return heuristics_matrix
