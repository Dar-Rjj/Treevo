import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])

    # Calculate the 20-day Exponential Moving Average (EMA) for log_high
    df['EMA_20_log_high'] = df['log_high'].ewm(span=20, adjust=False).mean()

    # Define weights for Weighted Moving Average (WMA) with a quadratically decreasing weight scheme
    quad_weights = np.power(np.linspace(20, 1, 20), 2)
    quad_weights /= quad_weights.sum()

    # Calculate the 20-day Weighted Moving Average (WMA) for log_close
    df['WMA_20_log_close'] = df['log_close'].rolling(window=20).apply(lambda x: (x * quad_weights[::-1]).sum(), raw=True)

    # Create the heuristic: ratio of EMA_20_log_high to WMA_20_log_close
    heuristics_matrix = df['EMA_20_log_high'] / df['WMA_20_log_close']

    return heuristics_matrix
