import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 30-day momentum of close prices
    momentum_30 = df['close'].pct_change(30)
    
    # Calculate the 30-day standard deviation of volume
    std_volume_30 = df['volume'].rolling(window=30).std()
    
    # Generate the heuristics matrix as the product of the momentum and standard deviation
    heuristics_matrix = momentum_30 * std_volume_30
    
    return heuristics_matrix
