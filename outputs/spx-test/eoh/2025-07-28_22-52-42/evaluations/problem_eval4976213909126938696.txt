def heuristics_v2(df):
    # Calculate the 5-day rate of change (ROC) for the closing price to capture momentum
    roc_close = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day rolling standard deviation of the volume to capture volatility
    vol_volatility = df['volume'].rolling(window=10).std()
    
    # Generate the heuristics matrix as the product of ROC of close and the volatility of volume
    heuristics_matrix = roc_close * vol_volatility
    
    return heuristics_matrix
