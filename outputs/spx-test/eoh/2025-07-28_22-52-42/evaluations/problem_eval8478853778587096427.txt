import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of closing price to the sum of high and low prices
    price_ratio = df['close'] / (df['high'] + df['low'])
    # Apply the square root to the volume
    sqrt_volume = df['volume'].apply(lambda x: x ** 0.5)
    # Generate the initial heuristics matrix by multiplying the two values
    heuristics_matrix = price_ratio * sqrt_volume
    # Apply a non-linear transformation to the heuristics matrix
    transformed_heuristics_matrix = heuristics_matrix.apply(lambda x: x ** 3 if x > 1 else x)
    return heuristics_matrix
