import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the log difference of consecutive days' closing prices
    close_log_diff = np.log(df['close']).diff().shift(-1)
    # Calculate the 5-day rolling mean of the natural log of volume
    volume_log_mean = np.log(df['volume']).rolling(window=5).mean()
    # Apply a non-linear transformation: square root of the absolute value of the sum of close_log_diff and volume_log_mean
    heuristics_matrix = (close_log_diff + volume_log_mean).apply(lambda x: np.sqrt(np.abs(x)))
    return heuristics_matrix
