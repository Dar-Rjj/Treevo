def heuristics_v2(df):
    # Calculate the natural logarithm of the close, high, and low prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the average of the logarithms of the high and low prices
    df['avg_log_high_low'] = (df['log_high'] + df['log_low']) / 2
    
    # Calculate the 10-day Exponential Moving Average (EMA) of the average log high and low
    df['EMA_10_avg_log_high_low'] = df['avg_log_high_low'].ewm(span=10, adjust=False).mean()
    
    # Create the heuristic: difference between log_close and EMA_10_avg_log_high_low
    heuristics_matrix = df['log_close'] - df['EMA_10_avg_log_high_low']
    
    return heuristics_matrix
