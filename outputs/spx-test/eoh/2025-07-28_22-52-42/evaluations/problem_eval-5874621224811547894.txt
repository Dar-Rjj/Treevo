import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the log difference of consecutive days' closing prices
    close_log_diff = np.log(df['close']).diff().shift(-1)
    # Calculate the 5-day moving average of volume
    volume_mean_5d = df['volume'].rolling(window=5).mean()
    # Apply an exponential transformation to the close log difference and take the reciprocal of the volume mean
    transformed_close_log_diff = np.exp(close_log_diff)
    transformed_volume_mean = 1 / volume_mean_5d
    # Generate the heuristics matrix as the product of the two transformed values
    heuristics_matrix = transformed_close_log_diff * transformed_volume_mean
    return heuristics_matrix
