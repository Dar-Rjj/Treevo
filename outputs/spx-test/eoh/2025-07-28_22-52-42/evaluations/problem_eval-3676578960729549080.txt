import pandas as pd

def heuristics_v2(df):
    def aroon_oscillator(series, period=14):
        high_max = series.rolling(window=period).max()
        low_min = series.rolling(window=period).min()
        periods_since_high = period - (series == high_max).rolling(window=period).apply(lambda x: x.argmax(), raw=True)
        periods_since_low = period - (series == low_min).rolling(window=period).apply(lambda x: x.argmin(), raw=True)
        aroon_up = 100 * (period - periods_since_high) / period
        aroon_down = 100 * (period - periods_since_low) / period
        return aroon_up - aroon_down

    roc = df['close'].pct_change(periods=12)
    aroon_values = aroon_oscillator(df['close'])
    volume_ratio = df['volume'] / df['volume'].rolling(window=30).mean()
    
    heuristics_matrix = (roc + aroon_values) * volume_ratio
    
    return heuristics_matrix
