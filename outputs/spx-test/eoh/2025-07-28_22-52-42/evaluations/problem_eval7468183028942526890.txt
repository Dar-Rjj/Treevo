def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) for the close price
    roc_close = df['close'].pct_change(10)
    
    # Calculate the ratio of the current volume to the maximum volume over the past 20 days
    max_vol_20d = df['volume'].rolling(window=20).max()
    vol_ratio = df['volume'] / max_vol_20d
    
    # Compute the average of the last 10 low prices
    avg_low_10d = df['low'].rolling(window=10).mean()
    
    # Generate the heuristics matrix as a combination of ROC, volume ratio, and average low
    heuristics_matrix = (roc_close + vol_ratio + avg_low_10d) / 3
    
    return heuristics_matrix
