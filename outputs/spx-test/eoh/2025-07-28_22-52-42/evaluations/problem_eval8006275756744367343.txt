def heuristics_v2(df):
    # Calculate RSI over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume ratio: current volume compared to its 30-day moving average
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    vol_ratio = df['volume'] / avg_vol_30d
    
    # Daily price range as a percentage of the closing price
    price_range_percent = (df['high'] - df['low']) / df['close']
    
    # Composite heuristic
    heuristics_matrix = (rsi + vol_ratio + price_range_percent) / 3
    
    return heuristics_matrix
