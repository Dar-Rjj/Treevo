import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of consecutive days' closing prices
    close_ratio = df['close'].pct_change().shift(-1)
    # Calculate the 2-day moving average of the volume
    vol_ma_2 = df['volume'].rolling(window=2).mean()
    # Calculate the 10-day standard deviation of the HL difference
    hl_diff_std = (df['high'] - df['low']).rolling(window=10).std()
    # Generate the heuristics matrix
    heuristics_matrix = (close_ratio / vol_ma_2) * hl_diff_std.pow(3)
    return heuristics_matrix
