import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day rate of change for high and low prices
    df['roc_10_high'] = df['high'].pct_change(10)
    df['roc_10_low'] = df['low'].pct_change(10)
    
    # Calculate the 5-day sum of volumes
    df['sum_5_volume'] = df['volume'].rolling(window=5).sum()
    
    # Create the heuristic: product of 10-day ROC of high and low prices and 5-day sum of volume
    heuristics_matrix = (df['roc_10_high'] * df['roc_10_low']) * df['sum_5_volume']
    
    return heuristics_matrix
