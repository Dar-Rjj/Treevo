import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of consecutive days' closing prices
    close_ratio = df['close'].pct_change().shift(-1)
    # Calculate the difference between high and low prices
    hl_diff = df['high'] - df['low']
    # Calculate the 5-day standard deviation of volume
    vol_std = df['volume'].rolling(window=5).std()
    # Generate the heuristics matrix as the product of the close ratio divided by the HL difference and the square root of volume standard deviation
    heuristics_matrix = (close_ratio / hl_diff) * vol_std.pow(0.5)
    return heuristics_matrix
