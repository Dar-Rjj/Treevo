import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and low prices
    df['log_close'] = np.log(df['close'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 50-day Simple Moving Average (SMA) for log_low
    df['SMA_50_log_low'] = df['log_low'].rolling(window=50).mean()
    
    # Define weights for Weighted Moving Average (WMA) with exponential decay
    weights = np.exp(np.linspace(0.1, 1, 50))
    weights = weights / weights.sum()
    
    # Calculate the 50-day Weighted Moving Average (WMA) for log_close
    df['WMA_50_log_close'] = df['log_close'].rolling(window=50).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Create the heuristic: difference between SMA_50_log_low and WMA_50_log_close
    heuristics_matrix = df['SMA_50_log_low'] - df['WMA_50_log_close']
    
    return heuristics_matrix
