import pandas as pd

def heuristics_v2(df):
    # Calculate the signed volume flow
    df['signed_volume_flow'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate the 20-day cumulative sum of the signed volume flow
    heuristics_matrix = df['signed_volume_flow'].rolling(window=20).sum()
    
    return heuristics_matrix
