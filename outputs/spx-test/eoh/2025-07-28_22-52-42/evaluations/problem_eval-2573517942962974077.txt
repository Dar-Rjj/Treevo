def heuristics_v2(df):
    # Calculate the rate of change in volume over a 10-day window
    vol_roc = df['volume'].pct_change(10)
    
    # Calculate the difference between high and low prices
    price_spread = df['high'] - df['low']
    
    # Compute the average price spread over a 20-day window
    avg_price_spread = price_spread.rolling(window=20).mean()
    
    # Generate the heuristics matrix as a combination of volume rate of change and price spread
    heuristics_matrix = (vol_roc + avg_price_spread) / 2

    return heuristics_matrix
