import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day and 200-day simple moving averages for the close price
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    
    # Calculate the RSI over a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the percentage change in volume compared to the 50-day average
    df['Volume_50'] = df['volume'].rolling(window=50).mean()
    df['Volume_Change'] = (df['volume'] - df['Volume_50']) / df['Volume_50']
    
    # Alpha factor: Combine SMA ratio, RSI, and volume change
    df['Alpha_Factor'] = (df['SMA_50'] / df['SMA_200']) + rsi + df['Volume_Change']
    
    # Return the alpha factor as a Series
    heuristics_matrix = df['Alpha_Factor']
    return heuristics_matrix
