def heuristics_v2(df):
    # Momentum: 10-day change in closing price
    momentum = df['close'] - df['close'].shift(10)
    
    # Volatility: 15-day standard deviation of closing price
    vol = df['close'].rolling(window=15).std()
    
    # Volume ratio: current volume compared to its 25-day moving average
    avg_vol_25d = df['volume'].rolling(window=25).mean()
    vol_ratio = df['volume'] / avg_vol_25d
    
    # Price volatility: 10-day average true range (ATR) normalized by close price
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=10).mean()
    price_volatility = atr / df['close']
    
    # Composite heuristic
    heuristics_matrix = (momentum + (vol * vol_ratio) + price_volatility) / 3
    
    return heuristics_matrix
