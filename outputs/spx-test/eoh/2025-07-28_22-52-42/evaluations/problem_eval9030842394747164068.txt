def heuristics_v2(df):
    # Calculate the daily percentage change in close prices
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the 20-day moving average of the closing price
    df['MA_20_close'] = df['close'].rolling(window=20).mean()
    
    # Derive a momentum signal from the ratio of the current close to the 20-day MA
    df['momentum_signal'] = df['close'] / df['MA_20_close']
    
    # Create the heuristic: Volume-adjusted momentum factor
    heuristics_matrix = df['pct_change_close'] * df['volume'] * df['momentum_signal']
    
    return heuristics_matrix
