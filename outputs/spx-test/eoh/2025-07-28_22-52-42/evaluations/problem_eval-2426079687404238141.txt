def heuristics_v2(df):
    # Calculate the natural logarithm of the volume and the daily trading range
    df['log_volume'] = np.log(df['volume'])
    df['log_range'] = np.log(df['high'] - df['low'])
    
    # Calculate the 30-day Simple Moving Average (SMA) for log_range
    df['SMA_30_log_range'] = df['log_range'].rolling(window=30).mean()
    
    # Define weights for Weighted Moving Average (WMA) with exponential decay
    weights = np.exp(np.linspace(0.1, 1, 30))
    weights = weights / weights.sum()
    
    # Calculate the 30-day Weighted Moving Average (WMA) for log_volume
    df['WMA_30_log_volume'] = df['log_volume'].rolling(window=30).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Create the heuristic: difference between SMA_30_log_range and WMA_30_log_volume
    heuristics_matrix = df['SMA_30_log_range'] - df['WMA_30_log_volume']
    
    return heuristics_matrix
