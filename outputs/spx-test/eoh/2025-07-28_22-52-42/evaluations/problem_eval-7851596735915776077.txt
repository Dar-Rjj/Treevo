import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of today's closing price to yesterday's
    close_ratio = df['close'] / df['close'].shift(1)
    # Calculate the standard deviation of volume over a 5-day window
    volume_std = df['volume'].rolling(window=5).std()
    # Apply a square root transformation to the ratio of close ratios divided by volume std
    heuristics_matrix = (close_ratio / volume_std).apply(lambda x: x**0.5)
    return heuristics_matrix
