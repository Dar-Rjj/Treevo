import numpy as np

def heuristics_v2(df):
    # Calculate the absolute difference between high and low prices
    price_diff = abs(df['high'] - df['low'])
    # Multiply by the square root of the trading volume
    modified_price_diff_to_volume = price_diff * np.sqrt(df['volume'])
    # Apply a logarithmic transformation
    transformed_value = np.log1p(modified_price_diff_to_volume)
    # Generate the heuristics matrix from the transformed value
    heuristics_matrix = transformed_value
    return heuristics_matrix
