import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and volume
    df['log_close'] = np.log(df['close'])
    df['log_volume'] = np.log(df['volume'])

    # Calculate the 20-day and 5-day Simple Moving Averages (SMA) for log_volume
    df['SMA_20_log_volume'] = df['log_volume'].rolling(window=20).mean()
    df['SMA_5_log_volume'] = df['log_volume'].rolling(window=5).mean()

    # Create the heuristic: difference between the sum of SMAs of log_volume and log_close
    heuristics_matrix = (df['SMA_20_log_volume'] + df['SMA_5_log_volume']) - df['log_close']

    return heuristics_matrix
