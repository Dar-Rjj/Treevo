import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the difference between closing and opening prices
    close_open_diff = df['close'] - df['open']
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    # Generate the price movement factor
    price_movement_factor = close_open_diff / avg_high_low
    # Apply the square root to the volume
    sqrt_volume = np.sqrt(df['volume'])
    # Generate the heuristics matrix as the product of the price movement factor and the square root of volume
    heuristics_matrix = price_movement_factor * sqrt_volume
    return heuristics_matrix
