import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the weighted average of high and low prices
    high_low_avg = (df['high'] + df['low']) / 2
    # Compute the difference between the high-low average and the opening price
    hl_open_diff = high_low_avg - df['open']
    # Use the natural logarithm of volume plus one as the weight
    ln_volume_plus_1 = np.log(df['volume'] + 1)
    # Generate the heuristics matrix as the product of the high-low-open difference and the natural logarithm of volume plus one
    heuristics_matrix = hl_open_diff * ln_volume_plus_1
    return heuristics_matrix
