def heuristics_v2(df):
    # Calculate the 5-day exponential moving average (EMA) for the close price
    ema_close_5d = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the ratio of the current volume to the average volume over the past 30 days
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    vol_ratio = df['volume'] / avg_vol_30d
    
    # Compute the standard deviation of the high prices over the last 10 days
    std_high_10d = df['high'].rolling(window=10).std()
    
    # Generate the heuristics matrix as a combination of EMA, volume ratio, and standard deviation of high
    heuristics_matrix = (ema_close_5d + vol_ratio + std_high_10d) / 3
    
    return heuristics_matrix
