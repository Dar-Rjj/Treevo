def heuristics_v2(df):
    # Calculate the ratio of close price to average volume over a 10-day window
    df['Close_Vol_Ratio'] = df['close'] / df['volume'].rolling(window=10).mean()
    # Calculate the 5-day moving average of the Close_Vol_Ratio
    df['MA_5_Close_Vol_Ratio'] = df['Close_Vol_Ratio'].rolling(window=5).mean()
    # Create the heuristic: difference between the Close_Vol_Ratio and its 5-day MA
    heuristics_matrix = df['Close_Vol_Ratio'] - df['MA_5_Close_Vol_Ratio']
    return heuristics_matrix
