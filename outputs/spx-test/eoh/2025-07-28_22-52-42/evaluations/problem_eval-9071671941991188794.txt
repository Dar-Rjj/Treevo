import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and low prices
    df['log_close'] = np.log(df['close'])
    df['log_low'] = np.log(df['low'])

    # Calculate the 50-day Exponential Moving Average (EMA) for log_close
    df['EMA_50_log_close'] = df['log_close'].ewm(span=50, adjust=False).mean()

    # Calculate the 50-day Simple Moving Average (SMA) for log_low
    df['SMA_50_log_low'] = df['log_low'].rolling(window=50).mean()

    # Create the heuristic: difference between SMA_50_log_low and EMA_50_log_close
    heuristics_matrix = df['SMA_50_log_low'] - df['EMA_50_log_close']

    return heuristics_matrix
