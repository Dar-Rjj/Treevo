import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close, high, and low prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the average of the logarithms of the high and low prices over a 20-day window
    df['avg_log_high_low'] = (df['log_high'] + df['log_low']) / 2
    df['MA_20_avg_log_high_low'] = df['avg_log_high_low'].rolling(window=20).mean()
    
    # Create the heuristic: weighted difference between log_close and MA_20_avg_log_high_low
    heuristics_matrix = df['log_close'] - 1.5 * df['MA_20_avg_log_high_low']
    
    return heuristics_matrix
