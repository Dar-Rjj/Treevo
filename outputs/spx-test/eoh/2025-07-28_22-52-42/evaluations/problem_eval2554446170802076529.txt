import pandas as pd

def heuristics_v2(df):
    ema_volume_15 = df['volume'].ewm(span=15, adjust=False).mean()
    ema_close_60 = df['close'].ewm(span=60, adjust=False).mean()
    momentum_30 = df['close'] / df['close'].shift(30)
    heuristics_matrix = (ema_volume_15 / ema_close_60) * momentum_30
    return heuristics_matrix
