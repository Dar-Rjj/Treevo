import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of consecutive days' closing prices to the next day's opening price
    close_open_ratio = df['close'] / df['open'].shift(-1)
    # Calculate the 5-day moving average of volumes
    volume_mean_5d = df['volume'].rolling(window=5).mean()
    # Calculate the ratio of the 5-day moving average of volumes to the current day's volume
    volume_ratio = volume_mean_5d / df['volume']
    # Generate the heuristics matrix as the product of the two calculated ratios
    heuristics_matrix = close_open_ratio * volume_ratio
    return heuristics_matrix
