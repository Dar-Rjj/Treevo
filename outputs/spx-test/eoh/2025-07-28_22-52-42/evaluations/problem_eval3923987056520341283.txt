def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    # Calculate the product of the daily return and the trading volume
    return_volume_product = daily_return * df['volume']
    # Compute the cumulative sum of the product
    heuristics_matrix = return_volume_product.cumsum()
    return heuristics_matrix
