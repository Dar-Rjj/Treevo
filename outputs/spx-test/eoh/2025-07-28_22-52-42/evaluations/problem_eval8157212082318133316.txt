import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the average of high and low prices
    high_low_avg = (df['high'] + df['low']) / 2
    # Use the square root of volume as the weight
    sqrt_volume = np.sqrt(df['volume'])
    # Generate the heuristics matrix as the product of the high-low average and the square root of volume
    heuristics_matrix = high_low_avg * sqrt_volume
    return heuristics_matrix
