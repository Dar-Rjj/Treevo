def heuristics_v2(df):
    # Calculate the short-term EMA (e.g., 10-day) of the close price
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the long-term EMA (e.g., 50-day) of the close price
    ema_long = df['close'].ewm(span=50, adjust=False).mean()
    
    # Compute the EMA difference
    ema_diff = ema_short - ema_long
    
    # Calculate the standard deviation of volume over a 30-day window
    vol_std = df['volume'].rolling(window=30).std()
    
    # Generate the heuristics matrix as a combination of EMA difference and volume standard deviation
    heuristics_matrix = ema_diff + vol_std

    return heuristics_matrix
