import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of the difference between high and low prices
    price_diff_ratio = (df['high'] - df['low']) / df['close'].shift(1)
    # Calculate the rolling standard deviation of volume over a 10-day window
    volume_std = df['volume'].rolling(window=10).std()
    # Calculate the EMA of closing prices over a 10-day window
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    # Generate the factor as the product of the price difference ratio, volume std, and EMA of close
    heuristics_matrix = price_diff_ratio * volume_std * ema_close
    return heuristics_matrix
