import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and low prices
    df['log_close'] = np.log(df['close'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 20-day Exponential Moving Average (EMA) for log_low
    df['EMA_20_log_low'] = df['log_low'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 20-day Standard Deviation of the logarithm of the high prices
    df['std_20_log_high'] = df['log_high'] = np.log(df['high'])
    df['std_20_log_high'] = df['log_high'].rolling(window=20).std()
    
    # Create the heuristic: (log_close - EMA_20_log_low) / std_20_log_high
    heuristics_matrix = (df['log_close'] - df['EMA_20_log_low']) / df['std_20_log_high']
    
    return heuristics_matrix
