import pandas as pd

def heuristics_v2(df):
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ratio_ema = ema_5 / ema_20
    roc_volume = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    heuristics_matrix = roc_volume - ratio_ema
    return heuristics_matrix
