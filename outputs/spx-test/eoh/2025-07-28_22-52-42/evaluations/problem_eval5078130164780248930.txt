import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for close prices with a 10-period lag
    df['ROC_close'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate the 15-day Exponential Moving Average (EMA) for high and low prices
    df['EMA_15_high'] = df['high'].ewm(span=15, adjust=False).mean()
    df['EMA_15_low'] = df['low'].ewm(span=15, adjust=False).mean()
    
    # Compute the difference between the EMAs
    ema_diff = df['EMA_15_high'] - df['EMA_15_low']
    
    # Generate the heuristics matrix as a weighted sum of ROC and EMA difference
    heuristics_matrix = 0.6 * df['ROC_close'] + 0.4 * ema_diff
    
    return heuristics_matrix
