def heuristics_v2(df):
    # Rate of Change: 14-day percentage change in closing price
    roc = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Adjusted Volatility: 20-day standard deviation of closing price, weighted by the difference between current volume and 30-day average volume
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    vol_adj = df['close'].rolling(window=20).std() * (df['volume'] - avg_vol_30d) / avg_vol_30d
    
    # Smoothed Directional Movement: exponential moving average of the difference between 20-day high and low
    high_20d = df['high'].rolling(window=20).max()
    low_20d = df['low'].rolling(window=20).min()
    dir_movement_smoothed = ((high_20d - low_20d) / df['close']).ewm(span=20, adjust=False).mean()
    
    # Composite heuristic
    heuristics_matrix = (roc + vol_adj + dir_movement_smoothed) / 3
    
    return heuristics_matrix
