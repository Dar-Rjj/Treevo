def heuristics_v2(df):
    # Average trading range: 20-day average of (high - low)
    avg_trading_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Cumulative sum of daily returns over 10 days
    daily_returns = df['close'].pct_change()
    cum_sum_returns = daily_returns.rolling(window=10).sum()
    
    # Composite heuristic
    heuristics_matrix = (avg_trading_range / cum_sum_returns).dropna()
    
    return heuristics_matrix
