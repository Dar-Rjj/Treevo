def heuristics_v2(df):
    # Rate of change in closing prices over 10 days
    roc_close = df['close'].pct_change(periods=10)
    
    # Smoothed trading volume using Exponential Moving Average (EMA) over 50 days
    ema_volume = df['volume'].ewm(span=50, adjust=False).mean()
    
    # Distance from the current closing price to its 200-day moving average
    ma_200d = df['close'].rolling(window=200).mean()
    dist_to_ma = df['close'] - ma_200d
    
    # Composite heuristic
    heuristics_matrix = roc_close * ema_volume + dist_to_ma
    
    return heuristics_matrix
