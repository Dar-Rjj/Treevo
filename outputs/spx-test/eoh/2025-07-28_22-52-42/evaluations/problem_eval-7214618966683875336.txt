import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])

    # Calculate the 30-day Exponential Moving Average (EMA) for log_close
    df['EMA_30_log_close'] = df['log_close'].ewm(span=30, adjust=False).mean()

    # Calculate the 30-day Simple Moving Average (SMA) for log_high
    df['SMA_30_log_high'] = df['log_high'].rolling(window=30).mean()

    # Create the heuristic: ratio of EMA_30_log_close to SMA_30_log_high
    heuristics_matrix = df['EMA_30_log_close'] / df['SMA_30_log_high']

    return heuristics_matrix
