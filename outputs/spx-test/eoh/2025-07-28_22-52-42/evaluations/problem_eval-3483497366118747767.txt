import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    def roc(series, period=14):
        return (series / series.shift(period) - 1) * 100
    
    # Calculate the Accumulation/Distribution (AD) Index
    def ad(series_high, series_low, series_close, series_volume):
        money_flow_multiplier = ((series_close - series_low) - (series_high - series_close)) / (series_high - series_low)
        money_flow_volume = money_flow_multiplier * series_volume
        return money_flow_volume.cumsum()
    
    roc_values = roc(df['close'])
    ad_values = ad(df['high'], df['low'], df['close'], df['volume'])

    heuristics_matrix = (roc_values + ad_values) / 2
    
    return heuristics_matrix
