import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the close and high prices
    df['log_close'] = np.log(df['close'])
    df['log_high'] = np.log(df['high'])
    
    # Calculate the 30-day Simple Moving Average (SMA) for log_high
    df['SMA_30_log_high'] = df['log_high'].rolling(window=30).mean()
    
    # Define weights for Weighted Moving Average (WMA) with exponential decay
    weights = np.exp(np.linspace(0.1, 1, 30))
    weights = weights / weights.sum()
    
    # Calculate the 30-day Weighted Moving Average (WMA) for log_close
    df['WMA_30_log_close'] = df['log_close'].rolling(window=30).apply(lambda x: (x * weights).sum(), raw=True)
    
    # Create the heuristic: difference between SMA_30_log_high and WMA_30_log_close
    heuristics_matrix = df['SMA_30_log_high'] - df['WMA_30_log_close']
    
    return heuristics_matrix
