import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the difference between consecutive days' closing and opening prices
    close_open_diff = (df['close'] - df['open']).shift(-1)
    # Apply square root to the volume
    sqrt_volume = np.sqrt(df['volume'])
    # Generate the heuristics matrix as the product of the close-open difference and the square root of volume
    heuristics_matrix = close_open_diff * sqrt_volume
    return heuristics_matrix
