import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the high and low prices
    df['log_high'] = np.log(df['high'])
    df['log_low'] = np.log(df['low'])
    
    # Calculate the 15-day Exponential Moving Average (EMA) for log_high
    df['EMA_15_log_high'] = df['log_high'].ewm(span=15, adjust=False).mean()
    
    # Calculate the 5-day Standard Deviation (STD) for log_low
    df['STD_5_log_low'] = df['log_low'].rolling(window=5).std()
    
    # Create the heuristic: ratio of EMA_15_log_high to STD_5_log_low
    heuristics_matrix = df['EMA_15_log_high'] / df['STD_5_log_low']
    
    return heuristics_matrix
