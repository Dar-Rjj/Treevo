import pandas as pd

def heuristics_v2(df):
    # Calculate the range (difference between high and low) for each day
    df['range'] = df['high'] - df['low']
    # Sum the volume over a 20-day window
    df['volume_20d_sum'] = df['volume'].rolling(window=20).sum()
    # Generate the heuristics matrix as the product of the daily range and the 20-day volume sum
    heuristics_matrix = df['range'] * df['volume_20d_sum']
    return heuristics_matrix
