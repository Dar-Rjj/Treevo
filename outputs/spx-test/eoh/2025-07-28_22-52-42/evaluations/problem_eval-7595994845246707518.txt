def heuristics_v2(df):
    # Calculate the ratio of consecutive days' high and low prices
    hl_ratio = df['high'] / df['low'].shift(1)
    # Calculate the rolling sum of volume over a 10-day window
    volume_sum = df['volume'].rolling(window=10).sum()
    # Generate the heuristics matrix as the product of the two values
    heuristics_matrix = hl_ratio * volume_sum
    return heuristics_matrix
