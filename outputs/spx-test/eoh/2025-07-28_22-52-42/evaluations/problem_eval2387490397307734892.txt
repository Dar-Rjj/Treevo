import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the natural logarithm of the low and volume
    df['log_low'] = np.log(df['low'])
    df['log_volume'] = np.log(df['volume'])
    
    # Calculate the 15-day Simple Moving Average (SMA) for log_low
    df['SMA_15_log_low'] = df['log_low'].rolling(window=15).mean()
    
    # Calculate the 5-day sum of log_volume
    df['SUM_5_log_volume'] = df['log_volume'].rolling(window=5).sum()
    
    # Create the heuristic: ratio between SMA_15_log_low and SUM_5_log_volume
    heuristics_matrix = df['SMA_15_log_low'] / df['SUM_5_log_volume']
    
    return heuristics_matrix
