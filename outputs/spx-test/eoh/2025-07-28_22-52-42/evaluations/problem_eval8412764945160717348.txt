def heuristics_v2(df):
    # Rate of change: 50-day percentage change in closing price
    roc_50d = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    
    # Average True Range: 14-day ATR
    tr = df['high'] - df['low']
    atr_14d = tr.rolling(window=14).mean()
    
    # Volume ratio: current volume compared to its 50-day moving average
    avg_vol_50d = df['volume'].rolling(window=50).mean()
    vol_ratio = df['volume'] / avg_vol_50d
    
    # Composite heuristic
    heuristics_matrix = (roc_50d + (atr_14d * vol_ratio))
    
    return heuristics_matrix
