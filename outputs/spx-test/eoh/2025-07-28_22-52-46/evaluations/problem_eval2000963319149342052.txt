def heuristics_v2(df):
    # Trend Following Factor: Calculate the 50-day moving average and compare it to the current close price
    df['trend_following'] = (df['close'] - df['close'].rolling(window=50).mean()) / df['close']
    
    # Volatility Factor: Calculate the 14-day Average True Range (ATR)
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = df['high'] - df['close'].shift(1)
    df['low_close_prev'] = df['low'] - df['close'].shift(1)
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Liquidity Factor: Relative Volume, which is the current volume divided by the 20-day average volume
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] / df['avg_vol_20']
    
    # Composite Heuristic Matrix by combining all factors
    heuristics_matrix = df['trend_following'] + df['atr'] + df['relative_volume']
    
    return heuristics_matrix
