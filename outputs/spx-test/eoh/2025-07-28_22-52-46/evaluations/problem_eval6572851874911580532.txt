def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()

    # Calculate the percentage change in the ratio of high to low prices
    df['high_low_ratio'] = (df['high'] / df['low']).pct_change()
    
    # Compute simple moving averages for close price
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Calculate SMA long-term trend
    df['long_term_trend'] = df['sma_50'] - df['sma_200']
    
    # 10-day standard deviation of the closing price
    df['std_10'] = df['close'].rolling(window=10).std()
    
    # Rolling 90-day correlations with future daily returns
    corr_high_low = df['high_low_ratio'].rolling(window=90).corr(df['daily_return'].shift(-1))
    corr_long_term_trend = df['long_term_trend'].rolling(window=90).corr(df['daily_return'].shift(-1))
    corr_std_10 = df['std_10'].rolling(window=90).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['high_low_ratio'] * corr_high_low +
                         df['long_term_trend'] * corr_long_term_trend +
                         df['std_10'] * corr_std_10)
    
    return heuristics_matrix
