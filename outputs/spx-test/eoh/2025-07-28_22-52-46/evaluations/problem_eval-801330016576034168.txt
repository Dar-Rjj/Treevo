import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day EMA of the volume
    df['volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the ratio of the close price to the 20-day EMA of the volume
    df['close_to_volume_ema_ratio'] = df['close'] / df['volume_ema_20']
    
    # Calculate the 20-day standard deviation of the close prices
    df['close_std_20'] = df['close'].rolling(window=20).std()
    
    # Calculate the weight for each day based on the standard deviation
    df['weight'] = df['close_std_20'] / df['close_std_20'].sum()
    
    # Calculate the weighted moving average of the close-to-volume-EMA ratio
    heuristics_matrix = (df['close_to_volume_ema_ratio'] * df['weight']).rolling(window=20).sum()
    
    return heuristics_matrix
