def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    
    # Calculate 20-day momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_50'] - df['sma_100']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical Sharpe ratios
    sharpe_ratios = {'momentum': (df['momentum'] * df['daily_return'].shift(-1)).mean() / df['momentum'].std(),
                     'sma_crossover': (df['sma_crossover'] * df['daily_return'].shift(-1)).mean() / df['sma_crossover'].std(),
                     'volume_trend': (df['volume_trend'] * df['daily_return'].shift(-1)).mean() / df['volume_trend'].std()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * sharpe_ratios['momentum'] +
                         df['sma_crossover'] * sharpe_ratios['sma_crossover'] +
                         df['volume_trend'] * sharpe_ratios['volume_trend'])
    
    return heuristics_matrix
