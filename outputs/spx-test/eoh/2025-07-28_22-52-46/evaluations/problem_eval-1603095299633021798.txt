def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate ROC for 10 and 30 days
    df['roc_10'] = df['close'].pct_change(periods=10)
    df['roc_30'] = df['close'].pct_change(periods=30)
    
    # Compute ATR
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': abs(df['high'] - df['close'].shift(1)),
        'l-pc': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    df['atr'] = tr.rolling(window=14).mean()
    
    # Compute SMA of volume
    df['sma_volume_10'] = df['volume'].rolling(window=10).mean()
    
    # ATR to SMA volume ratio
    df['atr_sma_volume_ratio'] = df['atr'] / df['sma_volume_10']
    
    # Rolling 60-day correlation coefficients
    corr_coeff_rsi = df['rsi'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_coeff_roc_10 = df['roc_10'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_coeff_roc_30 = df['roc_30'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_coeff_atr_sma_volume_ratio = df['atr_sma_volume_ratio'].rolling(window=60).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['rsi'] * corr_coeff_rsi +
                         df['roc_10'] * corr_coeff_roc_10 +
                         df['roc_30'] * corr_coeff_roc_30 +
                         df['atr_sma_volume_ratio'] * corr_coeff_atr_sma_volume_ratio)
    
    return heuristics_matrix
