def heuristics_v2(df):
    # Calculate the 14-day rate of change in closing prices
    df['roc'] = df['close'].pct_change(periods=14)
    
    # Calculate the Adjusted Chaikin Money Flow (ACMF) over 20 days
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    weights = pd.Series(range(1, 21), index=df.index[-20:])
    acmf = (money_flow_volume.rolling(window=20).apply(lambda x: (x * weights).sum(), raw=True)) / (df['volume'].rolling(window=20).apply(lambda x: (x * weights).sum(), raw=True))
    
    # Generate the heuristic matrix
    heuristics_matrix = df['roc'] + acmf
    
    return heuristics_matrix
