def heuristics_v2(df):
    # Calculate the daily price change
    price_change = df['close'] - df['open']
    
    # Multiply the price change by the volume to capture the strength of the movement
    strength = price_change * df['volume']
    
    # Apply a 14-day moving average to smooth the series
    heuristics_matrix = strength.rolling(window=14).mean()
    
    return heuristics_matrix
