def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_20'] - df['sma_50']
    
    # Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Historical return correlations
    corr_with_returns = {'sma_crossover': df['sma_crossover'].corr(df['daily_return'].shift(-1)),
                         'volatility': df['volatility'].corr(df['daily_return'].shift(-1)),
                         'rsi': df['rsi'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['sma_crossover'] * corr_with_returns['sma_crossover'] +
                         df['volatility'] * corr_with_returns['volatility'] +
                         df['rsi'] * corr_with_returns['rsi'])
    
    return heuristics_matrix
