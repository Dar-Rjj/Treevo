def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate rate of change
    df['roc_5'] = df['close'].pct_change(5)
    df['roc_20'] = df['close'].pct_change(20)
    
    # Calculate Information Coefficients (IC)
    ic_roc_5 = (df['roc_5'] * df['daily_return'].shift(-1)).mean()
    ic_roc_20 = (df['roc_20'] * df['daily_return'].shift(-1)).mean()
    
    # Volatility-adjusted volume trend
    df['vol_adj_volume_trend'] = (df['volume'] / df['close'].rolling(window=20).std()).pct_change(10)
    
    # Generate heuristic matrix
    heuristics_matrix = (df['roc_5'] * ic_roc_5 +
                         df['roc_20'] * ic_roc_20 +
                         df['vol_adj_volume_trend'])
    
    return heuristics_matrix
