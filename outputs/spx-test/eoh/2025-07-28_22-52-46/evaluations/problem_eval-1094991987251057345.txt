def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate price changes
    df['price_change_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['price_change_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Volume trends
    df['volume_trend_5'] = df['volume'].rolling(window=5).mean() / df['volume'].rolling(window=20).mean()
    df['volume_trend_20'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical Information ratios
    info_ratios = {
        'price_change_5': (df['price_change_5'] * df['daily_return'].shift(-1)).mean() / (df['price_change_5'] * df['daily_return'].shift(-1)).std(),
        'price_change_20': (df['price_change_20'] * df['daily_return'].shift(-1)).mean() / (df['price_change_20'] * df['daily_return'].shift(-1)).std(),
        'sma_crossover': (df['sma_crossover'] * df['daily_return'].shift(-1)).mean() / (df['sma_crossover'] * df['daily_return'].shift(-1)).std(),
        'volume_trend_5': (df['volume_trend_5'] * df['daily_return'].shift(-1)).mean() / (df['volume_trend_5'] * df['daily_return'].shift(-1)).std(),
        'volume_trend_20': (df['volume_trend_20'] * df['daily_return'].shift(-1)).mean() / (df['volume_trend_20'] * df['daily_return'].shift(-1)).std()
    }
    
    # Generate heuristic matrix
    heuristics_matrix = (df['price_change_5'] * info_ratios['price_change_5'] +
                         df['price_change_20'] * info_ratios['price_change_20'] +
                         df['sma_crossover'] * info_ratios['sma_crossover'] +
                         df['volume_trend_5'] * info_ratios['volume_trend_5'] +
                         df['volume_trend_20'] * info_ratios['volume_trend_20'])
    
    return heuristics_matrix
