def heuristics_v2(df):
    # Calculate the 14-day percentage change in closing price
    pct_change = df['close'].pct_change(periods=14)
    
    # Calculate the 14-day rolling standard deviation for both volume and percentage change
    vol_std = df['volume'].rolling(window=14).std()
    pct_change_std = pct_change.rolling(window=14).std()
    
    # Adjust weights based on standard deviations, with higher volatility leading to lower weight
    vol_weight = 1 / (vol_std + 1e-6)  # Adding a small constant to avoid division by zero
    pct_change_weight = 1 / (pct_change_std + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Scale weights so that they sum to 1
    total_weight = vol_weight + pct_change_weight
    vol_weight_scaled = (vol_weight / total_weight)
    pct_change_weight_scaled = (pct_change_weight / total_weight)
    
    # Calculate the heuristic by taking the weighted sum of the volume and the percentage change
    heuristics_matrix = (df['volume'] * vol_weight_scaled) + (pct_change * pct_change_weight_scaled)
    
    return heuristics_matrix
