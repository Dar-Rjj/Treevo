import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Exponential Moving Averages
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_volume_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate ROC
    df['roc_close'] = df['close'].pct_change(periods=5)
    
    # Calculate EMA Volume to Close Price Ratio
    df['volume_to_price_ratio'] = df['ema_volume_10'] / df['ema_close_10']
    
    # Calculate Kurtosis
    kurtosis = {'volume_to_price_ratio': df['volume_to_price_ratio'].kurt(),
                'roc_close': df['roc_close'].kurt()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['volume_to_price_ratio'] * kurtosis['volume_to_price_ratio'] + 
                         df['roc_close'] * kurtosis['roc_close'])
    
    return heuristics_matrix
