def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # 50-day moving average
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Distance from 50-day MA
    df['dist_from_ma50'] = df['close'] - df['ma_50']
    
    # Volume rate of change
    df['volume_roc'] = df['volume'].pct_change()
    
    # Volatility (std of returns over 30 days)
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Correlations with future daily returns
    corr_dist_from_ma50 = df['dist_from_ma50'].rolling(window=30).corr(df['daily_return'].shift(-1))
    corr_volume_roc = df['volume_roc'].rolling(window=30).corr(df['daily_return'].shift(-1))
    corr_volatility = df['volatility'].rolling(window=30).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['dist_from_ma50'] * corr_dist_from_ma50 +
                         df['volume_roc'] * corr_volume_roc +
                         df['volatility'] * corr_volatility)
    
    return heuristics_matrix
