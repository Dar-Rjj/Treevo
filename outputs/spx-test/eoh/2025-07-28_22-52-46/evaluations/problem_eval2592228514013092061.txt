def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 10-day EMA for daily return
    df['ema_return'] = df['daily_return'].ewm(span=10, adjust=False).mean()
    
    # Calculate 10-day EMA for volume
    df['ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Generate heuristic matrix
    heuristics_matrix = df['ema_return'] + df['ema_volume']
    
    return heuristics_matrix
