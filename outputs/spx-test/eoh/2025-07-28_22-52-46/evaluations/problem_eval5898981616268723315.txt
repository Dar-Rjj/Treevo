import pandas as pd

def heuristics_v2(df):
    # Calculate the ratio of close price to volume
    price_volume_ratio = df['close'] / df['volume']
    
    # Apply an exponential moving average (EMA) with a span of 5 days to smooth the series
    heuristics_matrix = price_volume_ratio.ewm(span=5, adjust=False).mean()
    
    return heuristics_matrix
