import pandas as pd

def heuristics_v2(df):
    # Compute the numerator (High - Low) and the denominator (Close)
    num = df['high'] - df['low']
    denom = df['close']
    
    # Calculate the base heuristic
    base_heuristic = num / denom
    
    # Apply a 5-day moving average to smooth the series
    heuristics_matrix = base_heuristic.rolling(window=5).mean()
    
    return heuristics_matrix
