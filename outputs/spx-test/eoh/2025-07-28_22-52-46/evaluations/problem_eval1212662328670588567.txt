def heuristics_v2(df):
    # Calculate the 20-day MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    
    # Calculate the ratio of the current day's close to the 20-day minimum close
    min_close_20 = df['close'].rolling(window=20).min()
    close_ratio = df['close'] / min_close_20
    
    # Combine MACD and close price ratio
    heuristics_matrix = macd + close_ratio
    
    return heuristics_matrix
