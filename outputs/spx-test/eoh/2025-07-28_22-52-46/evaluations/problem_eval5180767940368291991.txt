def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentums
    df['momentum_5'] = df['close'] - df['close'].shift(5)
    df['momentum_20'] = df['close'] - df['close'].shift(20)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Volume trends
    df['volume_trend_5'] = df['volume'].rolling(window=5).mean() / df['volume'].rolling(window=20).mean()
    df['volume_trend_20'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical Sharpe ratios
    sharpe_ratios = {
        'momentum_5': (df['momentum_5'] * df['daily_return'].shift(-1)).mean() / df['momentum_5'].std(),
        'momentum_20': (df['momentum_20'] * df['daily_return'].shift(-1)).mean() / df['momentum_20'].std(),
        'sma_crossover': (df['sma_crossover'] * df['daily_return'].shift(-1)).mean() / df['sma_crossover'].std(),
        'volume_trend_5': (df['volume_trend_5'] * df['daily_return'].shift(-1)).mean() / df['volume_trend_5'].std(),
        'volume_trend_20': (df['volume_trend_20'] * df['daily_return'].shift(-1)).mean() / df['volume_trend_20'].std()
    }
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum_5'] * sharpe_ratios['momentum_5'] +
                         df['momentum_20'] * sharpe_ratios['momentum_20'] +
                         df['sma_crossover'] * sharpe_ratios['sma_crossover'] +
                         df['volume_trend_5'] * sharpe_ratios['volume_trend_5'] +
                         df['volume_trend_20'] * sharpe_ratios['volume_trend_20'])
    
    return heuristics_matrix
