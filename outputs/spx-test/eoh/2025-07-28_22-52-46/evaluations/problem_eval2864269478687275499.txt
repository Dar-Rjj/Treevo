import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day Exponential Moving Average of closing prices
    ema = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-day rolling standard deviation of closing prices as a measure of volatility
    volatility = df['close'].rolling(window=14).std()
    
    # Generate the heuristic matrix
    heuristics_matrix = ema + volatility
    
    return heuristics_matrix
