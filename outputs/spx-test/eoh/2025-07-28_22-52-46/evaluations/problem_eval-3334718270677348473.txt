import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    ma_short = df['close'].rolling(window=10).mean()
    ma_long = df['close'].rolling(window=50).mean()
    
    # Price momentum over a 20-day period
    momentum = df['close'] / df['close'].shift(20) - 1
    
    # Historical performance of each component (simplified as their mean)
    perf_ma_short = ma_short.mean()
    perf_ma_long = ma_long.mean()
    perf_momentum = momentum.mean()
    
    # Total performance for weight calculation
    total_perf = perf_ma_short + perf_ma_long + perf_momentum
    
    # Weights based on historical performance
    weight_ma_short = perf_ma_short / total_perf
    weight_ma_long = perf_ma_long / total_perf
    weight_momentum = perf_momentum / total_perf
    
    # Compute the heuristics matrix
    heuristics_matrix = (weight_ma_short * ma_short + 
                         weight_ma_long * ma_long + 
                         weight_momentum * momentum)
    
    return heuristics_matrix
