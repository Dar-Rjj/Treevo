def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_15'] = df['close'].rolling(window=15).mean()
    df['sma_40'] = df['close'].rolling(window=40).mean()
    
    # Calculate momentums
    df['momentum_7'] = df['close'] - df['close'].shift(7)
    df['momentum_25'] = df['close'] - df['close'].shift(25)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_15'] - df['sma_40']
    
    # Volume trends
    df['volume_trend_7'] = df['volume'].rolling(window=7).mean() / df['volume'].rolling(window=25).mean()
    df['volume_trend_25'] = df['volume'].rolling(window=25).mean() / df['volume'].rolling(window=40).mean()
    
    # Historical Sharpe ratios
    sharpe_ratios = {
        'momentum_7': (df['momentum_7'] * df['daily_return'].shift(-1)).mean() / df['momentum_7'].std(),
        'momentum_25': (df['momentum_25'] * df['daily_return'].shift(-1)).mean() / df['momentum_25'].std(),
        'sma_crossover': (df['sma_crossover'] * df['daily_return'].shift(-1)).mean() / df['sma_crossover'].std(),
        'volume_trend_7': (df['volume_trend_7'] * df['daily_return'].shift(-1)).mean() / df['volume_trend_7'].std(),
        'volume_trend_25': (df['volume_trend_25'] * df['daily_return'].shift(-1)).mean() / df['volume_trend_25'].std()
    }
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum_7'] * sharpe_ratios['momentum_7'] +
                         df['momentum_25'] * sharpe_ratios['momentum_25'] +
                         df['sma_crossover'] * sharpe_ratios['sma_crossover'] +
                         df['volume_trend_7'] * sharpe_ratios['volume_trend_7'] +
                         df['volume_trend_25'] * sharpe_ratios['volume_trend_25'])
    
    return heuristics_matrix
