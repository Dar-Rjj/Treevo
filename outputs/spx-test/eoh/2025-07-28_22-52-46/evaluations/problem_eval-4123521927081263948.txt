def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day price range
    df['price_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate 14-day Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_flow_sum = positive_flow.rolling(window=14).sum()
    negative_flow_sum = negative_flow.rolling(window=14).sum()
    mfi_ratio = positive_flow_sum / negative_flow_sum
    df['mfi'] = 100 - (100 / (1 + mfi_ratio))
    
    # Calculate 10-day Rate of Change (ROC)
    df['roc'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10) * 100
    
    # Historical Sortino ratios
    sortino_ratios = {
        'price_range': (df['price_range'] * df['daily_return'].shift(-1)).mean() / df['price_range'][df['daily_return'].shift(-1) < 0].std(),
        'mfi': (df['mfi'] * df['daily_return'].shift(-1)).mean() / df['mfi'][df['daily_return'].shift(-1) < 0].std(),
        'roc': (df['roc'] * df['daily_return'].shift(-1)).mean() / df['roc'][df['daily_return'].shift(-1) < 0].std()
    }
    
    # Generate heuristic matrix
    heuristics_matrix = (df['price_range'] * sortino_ratios['price_range'] +
                         df['mfi'] * sortino_ratios['mfi'] +
                         df['roc'] * sortino_ratios['roc'])
    
    return heuristics_matrix
