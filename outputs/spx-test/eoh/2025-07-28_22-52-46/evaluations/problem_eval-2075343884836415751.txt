import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate price relative to SMA
    df['price_to_sma_20'] = df['close'] / df['sma_20']
    df['price_to_sma_100'] = df['close'] / df['sma_100']
    
    # Volume trend
    df['volume_trend_20'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=50).mean()
    df['volume_trend_50'] = df['volume'].rolling(window=50).mean() / df['volume'].rolling(window=100).mean()
    
    # Volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Historical return correlations
    corr_with_returns = {'momentum': df['momentum'].corr(df['daily_return'].shift(-1)), 
                         'price_to_sma_20': df['price_to_sma_20'].corr(df['daily_return'].shift(-1)),
                         'price_to_sma_100': df['price_to_sma_100'].corr(df['daily_return'].shift(-1)),
                         'volume_trend_20': df['volume_trend_20'].corr(df['daily_return'].shift(-1)),
                         'volume_trend_50': df['volume_trend_50'].corr(df['daily_return'].shift(-1)),
                         'volatility': df['volatility'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * corr_with_returns['momentum'] +
                         df['price_to_sma_20'] * corr_with_returns['price_to_sma_20'] +
                         df['price_to_sma_100'] * corr_with_returns['price_to_sma_100'] +
                         df['volume_trend_20'] * corr_with_returns['volume_trend_20'] +
                         df['volume_trend_50'] * corr_with_returns['volume_trend_50'] +
                         df['volatility'] * corr_with_returns['volatility'])
    
    return heuristics_matrix
