import pandas as pd

def heuristics_v2(df):
    def calc_rsi(data, window=14):
        delta = data['close'].diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    def calc_volatility_change(data, short_window=5, long_window=20):
        data[f'SD_{short_window}'] = data['close'].rolling(window=short_window).std()
        data[f'SD_{long_window}'] = data['close'].rolling(window=long_window).std()
        data['Vol_Change'] = (data[f'SD_{short_window}'] - data[f'SD_{long_window}']) / data[f'SD_{long_window}']
        return data

    df = df.copy()
    df['RSI'] = calc_rsi(df)
    df = calc_volatility_change(df)

    df['Alpha_Factor'] = df['RSI'] + df['Vol_Change']

    heuristics_matrix = df['Alpha_Factor']

    return heuristics_matrix
