def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Rolling 60-day historical Information ratios
    info_ratios_momentum = (df['momentum'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['momentum'].rolling(window=60).std() + 1e-8)
    info_ratios_sma_crossover = (df['sma_crossover'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['sma_crossover'].rolling(window=60).std() + 1e-8)
    info_ratios_volume_trend = (df['volume_trend'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['volume_trend'].rolling(window=60).std() + 1e-8)
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * info_ratios_momentum +
                         df['sma_crossover'] * info_ratios_sma_crossover +
                         df['volume_trend'] * info_ratios_volume_trend)
    
    return heuristics_matrix
