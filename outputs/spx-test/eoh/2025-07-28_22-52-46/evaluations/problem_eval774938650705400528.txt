def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    df['momentum'] = df['close'] - df['close'].shift(5)
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    sharpe_ratios_momentum = (df['momentum'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / df['momentum'].rolling(window=60).std()
    sharpe_ratios_sma_crossover = (df['sma_crossover'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / df['sma_crossover'].rolling(window=60).std()
    sharpe_ratios_volume_trend = (df['volume_trend'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / df['volume_trend'].rolling(window=60).std()

    heuristics_matrix = (df['momentum'] * sharpe_ratios_momentum +
                         df['sma_crossover'] * sharpe_ratios_sma_crossover +
                         df['volume_trend'] * sharpe_ratios_volume_trend)
    
    return heuristics_matrix
