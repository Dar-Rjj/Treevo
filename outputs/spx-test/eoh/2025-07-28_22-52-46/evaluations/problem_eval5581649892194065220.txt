import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate price relative to SMA
    df['price_to_sma_10'] = df['close'] / df['sma_10']
    df['price_to_sma_30'] = df['close'] / df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical return correlations
    corr_with_returns = {'momentum': df['momentum'].corr(df['daily_return'].shift(-1)), 
                         'price_to_sma_10': df['price_to_sma_10'].corr(df['daily_return'].shift(-1)),
                         'price_to_sma_30': df['price_to_sma_30'].corr(df['daily_return'].shift(-1)),
                         'volume_trend': df['volume_trend'].corr(df['daily_return'].shift(-1))}
    
    # Historical volatility
    vol_momentum = df['momentum'].rolling(window=30).std()
    vol_price_to_sma_10 = df['price_to_sma_10'].rolling(window=30).std()
    vol_price_to_sma_30 = df['price_to_sma_30'].rolling(window=30).std()
    vol_volume_trend = df['volume_trend'].rolling(window=30).std()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * (corr_with_returns['momentum'] / vol_momentum) +
                         df['price_to_sma_10'] * (corr_with_returns['price_to_sma_10'] / vol_price_to_sma_10) +
                         df['price_to_sma_30'] * (corr_with_returns['price_to_sma_30'] / vol_price_to_sma_30) +
                         df['volume_trend'] * (corr_with_returns['volume_trend'] / vol_volume_trend))
    
    return heuristics_matrix
