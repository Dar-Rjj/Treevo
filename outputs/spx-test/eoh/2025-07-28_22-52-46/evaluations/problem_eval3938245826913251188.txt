def heuristics_v2(df):
    # Compute the numerator (|High - Low|) and the denominator sqrt(Open * Close)
    num = abs(df['high'] - df['low'])
    denom = (df['open'] * df['close']).apply(np.sqrt)
    
    # Calculate the base heuristic
    base_heuristic = num / denom
    
    # Apply a 5-day exponentially weighted moving average to smooth the series
    heuristics_matrix = base_heuristic.ewm(span=5, adjust=False).mean()
    
    return heuristics_matrix
