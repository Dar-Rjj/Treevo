def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Rolling 30-day historical mean returns
    mean_momentum = df['momentum'].rolling(window=30).mean()
    mean_sma_crossover = df['sma_crossover'].rolling(window=30).mean()
    mean_volume_trend = df['volume_trend'].rolling(window=30).mean()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * mean_momentum +
                         df['sma_crossover'] * mean_sma_crossover +
                         df['volume_trend'] * mean_volume_trend)
    
    return heuristics_matrix
