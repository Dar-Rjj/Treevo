import pandas as pd

    def calc_rsi(data, window=14):
        delta = data['close'].diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    def calc_dist_to_high(data, window=20):
        data['20_day_high'] = data['high'].rolling(window=window).max()
        data['Dist_to_High'] = (data['close'] - data['20_day_high']) / data['20_day_high']
        return data['Dist_to_High']

    def calc_volume_ratio(data, window=50):
        data['Avg_Volume_50'] = data['volume'].rolling(window=window).mean()
        data['Volume_Ratio'] = data['volume'] / data['Avg_Volume_50']
        return data['Volume_Ratio']

    df['RSI'] = calc_rsi(df)
    df['Dist_to_High'] = calc_dist_to_high(df)
    df['Volume_Ratio'] = calc_volume_ratio(df)

    df['Alpha_Factor'] = df['RSI'] + df['Dist_to_High'] + df['Volume_Ratio']
    heuristics_matrix = df['Alpha_Factor']

    return heuristics_matrix
