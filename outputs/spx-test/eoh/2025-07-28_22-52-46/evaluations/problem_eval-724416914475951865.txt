def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Bollinger Bands width
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['std_20'] = df['close'].rolling(window=20).std()
    df['bb_width'] = 2 * df['std_20']
    
    # Calculate 20-day high-low price range
    df['high_low_range'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Rolling 60-day historical Information Ratios
    information_ratio_rsi = (df['rsi'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['rsi'] * df['daily_return'].shift(-1)).rolling(window=60).std()
    information_ratio_bb_width = (df['bb_width'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['bb_width'] * df['daily_return'].shift(-1)).rolling(window=60).std()
    information_ratio_high_low_range = (df['high_low_range'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['high_low_range'] * df['daily_return'].shift(-1)).rolling(window=60).std()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['rsi'] * information_ratio_rsi +
                         df['bb_width'] * information_ratio_bb_width +
                         df['high_low_range'] * information_ratio_high_low_range)
    
    return heuristics_matrix
