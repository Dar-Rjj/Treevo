import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Price to volume ratio
    df['price_to_volume'] = df['close'] / df['volume']
    
    # Daily return volatility
    df['return_volatility'] = df['daily_return'].rolling(window=14).std()
    
    # Exponential moving averages for weights
    rsi_ema = df['rsi'].ewm(span=60, adjust=False).mean()
    price_to_volume_ema = df['price_to_volume'].ewm(span=60, adjust=False).mean()
    return_volatility_ema = df['return_volatility'].ewm(span=60, adjust=False).mean()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['rsi'] * rsi_ema +
                         df['price_to_volume'] * price_to_volume_ema +
                         df['return_volatility'] * return_volatility_ema)
    
    return heuristics_matrix
