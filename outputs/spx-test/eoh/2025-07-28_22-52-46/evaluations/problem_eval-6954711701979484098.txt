def heuristics_v2(df):
    # Compute the numerator (High - Low) and the denominator (Open + Close)
    num = df['high'] - df['low']
    denom = df['open'] + df['close']
    
    # Calculate the base heuristic
    base_heuristic = num / denom
    
    # Apply a 14-day Exponentially Weighted Moving Average to smooth the series
    heuristics_matrix = base_heuristic.ewm(span=14).mean()
    
    return heuristics_matrix
