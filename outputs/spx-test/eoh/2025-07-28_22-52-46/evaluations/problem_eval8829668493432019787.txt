def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_100'] = df['close'].rolling(window=100).mean()
    
    # Calculate 20 and 50-day momentums
    df['momentum_20'] = df['close'] - df['close'].shift(20)
    df['momentum_50'] = df['close'] - df['close'].shift(50)
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical correlations
    correlations = {'momentum_20': df['momentum_20'].corr(df['daily_return'].shift(-1)),
                    'momentum_50': df['momentum_50'].corr(df['daily_return'].shift(-1)),
                    'sma_100': -df['sma_100'].corr(df['daily_return'].shift(-1)),  # Negative because SMA is inversely related to returns
                    'volume_trend': df['volume_trend'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum_20'] * correlations['momentum_20'] +
                         df['momentum_50'] * correlations['momentum_50'] +
                         df['sma_100'] * correlations['sma_100'] +
                         df['volume_trend'] * correlations['volume_trend'])
    
    return heuristics_matrix
