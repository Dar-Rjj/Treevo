import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Aroon
    df['aroon_up'] = 100 * df['close'].rolling(window=25).apply(lambda x: (x.argmax() + 1) / 25, raw=True)
    df['aroon_down'] = 100 * df['close'].rolling(window=25).apply(lambda x: (x.argmin() + 1) / 25, raw=True)
    df['aroon'] = df['aroon_up'] - df['aroon_down']
    
    # Weighted moving average of volume
    df['volume_wma'] = df['volume'].rolling(window=10).mean()
    
    # Correlations with future daily returns
    corr_rsi = df['rsi'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_aroon = df['aroon'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_volume_wma = df['volume_wma'].rolling(window=60).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['rsi'] * corr_rsi +
                         df['aroon'] * corr_aroon +
                         df['volume_wma'] * corr_volume_wma)
    
    return heuristics_matrix
