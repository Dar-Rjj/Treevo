import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate price relative to SMA
    df['price_to_sma_10'] = df['close'] / df['sma_10']
    df['price_to_sma_30'] = df['close'] / df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical return volatilities
    vol_with_returns = {'momentum': df['momentum'].std(), 
                        'price_to_sma_10': df['price_to_sma_10'].std(),
                        'price_to_sma_30': df['price_to_sma_30'].std(),
                        'volume_trend': df['volume_trend'].std()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * vol_with_returns['momentum'] +
                         df['price_to_sma_10'] * vol_with_returns['price_to_sma_10'] +
                         df['price_to_sma_30'] * vol_with_returns['price_to_sma_30'] +
                         df['volume_trend'] * vol_with_returns['volume_trend'])
    
    return heuristics_matrix
