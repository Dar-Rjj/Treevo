def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day and 60-day price to volume ratios
    df['price_to_volume_20'] = df['close'] / df['volume'].rolling(window=20).mean()
    df['price_to_volume_60'] = df['close'] / df['volume'].rolling(window=60).mean()
    
    # Rolling 30-day historical Sharpe ratios for the price to volume ratios
    sharpe_ratios_price_to_volume_20 = (df['price_to_volume_20'] * df['daily_return'].shift(-1)).rolling(window=30).mean() / df['price_to_volume_20'].rolling(window=30).std()
    sharpe_ratios_price_to_volume_60 = (df['price_to_volume_60'] * df['daily_return'].shift(-1)).rolling(window=30).mean() / df['price_to_volume_60'].rolling(window=30).std()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['price_to_volume_20'] * sharpe_ratios_price_to_volume_20 +
                         df['price_to_volume_60'] * sharpe_ratios_price_to_volume_60)
    
    return heuristics_matrix
