def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day price volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Calculate the ratio of the latest close to the highest high over the past 50 days
    df['high_ratio'] = df['close'] / df['high'].rolling(window=50).max()
    
    # Calculate the ratio of the latest close to the lowest low over the past 50 days
    df['low_ratio'] = df['close'] / df['low'].rolling(window=50).min()
    
    # Historical Information Ratios
    information_ratios = {
        'volatility': (df['volatility'] * df['daily_return'].shift(-1)).mean() / (df['volatility'] * df['daily_return'].shift(-1)).std(),
        'high_ratio': (df['high_ratio'] * df['daily_return'].shift(-1)).mean() / (df['high_ratio'] * df['daily_return'].shift(-1)).std(),
        'low_ratio': (df['low_ratio'] * df['daily_return'].shift(-1)).mean() / (df['low_ratio'] * df['daily_return'].shift(-1)).std()
    }
    
    # Generate heuristic matrix
    heuristics_matrix = (df['volatility'] * information_ratios['volatility'] +
                         df['high_ratio'] * information_ratios['high_ratio'] +
                         df['low_ratio'] * information_ratios['low_ratio'])
    
    return heuristics_matrix
