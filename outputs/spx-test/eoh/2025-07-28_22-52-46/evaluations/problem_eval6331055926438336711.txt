def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute exponential moving averages for close price
    df['ema_10'] = df['close'].ewm(span=10).mean()
    df['ema_30'] = df['close'].ewm(span=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate EMA crossover
    df['ema_crossover'] = df['ema_10'] - df['ema_30']
    
    # Volume-to-price ratio
    df['vpr'] = df['volume'] / df['close']
    
    # Rolling 60-day historical correlations with future returns
    corr_momentum = (df['momentum']).rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_ema_crossover = (df['ema_crossover']).rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_vpr = (df['vpr']).rolling(window=60).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * corr_momentum +
                         df['ema_crossover'] * corr_ema_crossover +
                         df['vpr'] * corr_vpr)
    
    return heuristics_matrix
