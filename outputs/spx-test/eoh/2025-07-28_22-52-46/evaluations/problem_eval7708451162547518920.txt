import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-period directional movement
    df['directional_movement'] = df['close'].diff(10)
    
    # Calculate the 10-period volume-price ratio
    df['volume_price_ratio'] = (df['volume'] / df['close']).rolling(window=10).mean()
    
    # Calculate the mean reversion score
    df['mean_reversion_score'] = (df['close'] - df['close'].rolling(window=50).mean()) / df['close'].rolling(window=50).std()
    
    # Historical return correlations
    corr_with_returns = {'directional_movement': df['directional_movement'].corr(df['daily_return'].shift(-1)), 
                         'volume_price_ratio': df['volume_price_ratio'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['directional_movement'] * corr_with_returns['directional_movement'] +
                         df['volume_price_ratio'] * corr_with_returns['volume_price_ratio']) * abs(df['mean_reversion_score'])
    
    return heuristics_matrix
