import pandas as pd

def heuristics_v2(df):
    def calc_rsi(data, window=14):
        delta = data['close'].diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        data['RSI'] = rsi
        return data

    def calc_volume_change(data):
        data['Volume_Change'] = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)
        return data

    df = calc_rsi(df.copy())
    df = calc_volume_change(df)

    df['Alpha_Factor'] = df['RSI'] + df['Volume_Change']
    heuristics_matrix = df['Alpha_Factor']

    return heuristics_matrix
