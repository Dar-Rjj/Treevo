def heuristics_v2(df):
    # Calculate 10-day moving average of the close price
    df['10_day_avg_close'] = df['close'].rolling(window=10).mean()
    # Compute the difference between the current close and its 10-day moving average
    heuristics_matrix = df['close'] - df['10_day_avg_close']
    return heuristics_matrix
