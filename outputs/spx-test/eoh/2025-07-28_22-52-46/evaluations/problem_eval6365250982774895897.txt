def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    df['momentum'] = df['close'] - df['close'].shift(5)
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    corr_momentum = (df['momentum']).rolling(window=30).corr(df['daily_return'].shift(-1))
    corr_sma_crossover = (df['sma_crossover']).rolling(window=30).corr(df['daily_return'].shift(-1))
    corr_volume_trend = (df['volume_trend']).rolling(window=30).corr(df['daily_return'].shift(-1))
    weights_sum = corr_momentum + corr_sma_crossover + corr_volume_trend
    heuristics_matrix = (df['momentum'] * (corr_momentum / weights_sum) +
                         df['sma_crossover'] * (corr_sma_crossover / weights_sum) +
                         df['volume_trend'] * (corr_volume_trend / weights_sum))
    return heuristics_matrix
