def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Exponentially Weighted Moving Average (EWMA)
    df['ewma_10'] = df['daily_return'].ewm(span=10, adjust=False).mean()
    
    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Rate of Change (ROC) in volume
    df['roc_volume'] = df['volume'].pct_change(periods=10)
    
    # 60-day correlation with future returns
    corr_ewma = df['ewma_10'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_rsi = df['rsi'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_roc_volume = df['roc_volume'].rolling(window=60).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['ewma_10'] * corr_ewma +
                         df['rsi'] * corr_rsi +
                         df['roc_volume'] * corr_roc_volume)
    
    return heuristics_matrix
