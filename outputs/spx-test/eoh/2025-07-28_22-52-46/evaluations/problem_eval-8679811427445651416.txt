import pandas as pd

def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate ADX
    tr = pd.DataFrame({'H-L': df['high'] - df['low'],
                       'H-Cp': abs(df['high'] - df['close'].shift(1)),
                       'L-Cp': abs(df['low'] - df['close'].shift(1))})
    tr['tr'] = tr.max(axis=1)
    atr = tr['tr'].rolling(window=14).mean()
    
    dm_pos = (df['high'].diff().fillna(0) + 0.000000001).where(lambda x: x > 0, 0)
    dm_neg = (df['low'].diff().fillna(0) + 0.000000001).where(lambda x: x < 0, 0)
    
    di_pos = (dm_pos.rolling(window=14).sum() / atr) * 100
    di_neg = (abs(dm_neg.rolling(window=14).sum()) / atr) * 100
    dx = (abs(di_pos - di_neg) / (di_pos + di_neg)) * 100
    adx = dx.rolling(window=14).mean()
    
    # Generate heuristic matrix
    heuristics_matrix = rsi + adx
    
    return heuristics_matrix
