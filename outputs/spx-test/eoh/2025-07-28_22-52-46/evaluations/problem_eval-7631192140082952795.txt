def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    # Compute exponential moving averages for close price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate RSI
    delta = df['close'].diff()
    up, down = delta.clip(lower=0), delta.clip(upper=0).abs()
    roll_up = up.ewm(com=13, adjust=False).mean()
    roll_down = down.ewm(com=13, adjust=False).mean()
    rs = roll_up / roll_down
    df['rsi'] = 100.0 - (100.0 / (1.0 + rs))
    
    # Volume percentage change
    df['volume_change'] = df['volume'].pct_change()
    
    # EMA crossover
    df['ema_crossover'] = df['ema_10'] - df['ema_30']
    
    # Rolling 60-day historical Information Ratios
    info_ratios_ema_cross = (df['ema_crossover'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['ema_crossover'] * df['daily_return'].shift(-1)).rolling(window=60).std()
    info_ratios_rsi = (df['rsi'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['rsi'] * df['daily_return'].shift(-1)).rolling(window=60).std()
    info_ratios_volume_change = (df['volume_change'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / (df['volume_change'] * df['daily_return'].shift(-1)).rolling(window=60).std()
    
    heuristics_matrix = (df['ema_crossover'] * info_ratios_ema_cross +
                         df['rsi'] * info_ratios_rsi +
                         df['volume_change'] * info_ratios_volume_change)
    
    return heuristics_matrix
