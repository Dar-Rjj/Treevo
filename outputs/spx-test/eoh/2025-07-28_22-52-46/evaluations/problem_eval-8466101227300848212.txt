def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    df['momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    df['ema_crossover'] = df['ema_10'] - df['ema_30']
    
    df['volume_trend'] = df['volume'].rolling(window=5).mean() / df['volume'].rolling(window=20).mean()

    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    sharpe_ratios_momentum = (df['momentum'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / df['momentum'].rolling(window=60).std()
    sharpe_ratios_ema_crossover = (df['ema_crossover'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / df['ema_crossover'].rolling(window=60).std()
    sharpe_ratios_volume_trend = (df['volume_trend'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / df['volume_trend'].rolling(window=60).std()
    sharpe_ratios_rsi = (df['rsi'] * df['daily_return'].shift(-1)).rolling(window=60).mean() / df['rsi'].rolling(window=60).std()
    
    heuristics_matrix = (df['momentum'].rank(pct=True) * sharpe_ratios_momentum +
                         df['ema_crossover'].rank(pct=True) * sharpe_ratios_ema_crossover +
                         df['volume_trend'].rank(pct=True) * sharpe_ratios_volume_trend +
                         df['rsi'].rank(pct=True) * sharpe_ratios_rsi)
    
    return heuristics_matrix
