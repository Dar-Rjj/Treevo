import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day Average True Range (ATR)
    df['tr'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    atr = df['tr'].rolling(window=14).mean()
    
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Generate the heuristic matrix
    heuristics_matrix = atr + rsi
    
    return heuristics_matrix
