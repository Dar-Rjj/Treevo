def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate MACD
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = df['ema_12'] - df['ema_26']
    
    # Calculate Aroon Oscillator
    df['aroon_up'] = 100 * (df['high'].rolling(window=25).apply(lambda x: (x.argmax() + 1) / 25))
    df['aroon_down'] = 100 * (df['low'].rolling(window=25).apply(lambda x: (x.argmin() + 1) / 25))
    df['aroon_oscillator'] = df['aroon_up'] - df['aroon_down']
    
    # Historical Information Ratios
    ir_ratios = {
        'macd': (df['macd'] * df['daily_return'].shift(-1)).mean() / (df['macd'] * df['daily_return'].shift(-1)).std(),
        'aroon_oscillator': (df['aroon_oscillator'] * df['daily_return'].shift(-1)).mean() / (df['aroon_oscillator'] * df['daily_return'].shift(-1)).std()
    }
    
    # Generate heuristic matrix
    heuristics_matrix = (df['macd'] * ir_ratios['macd'] + df['aroon_oscillator'] * ir_ratios['aroon_oscillator'])
    
    return heuristics_matrix
