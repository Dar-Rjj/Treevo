def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate price-to-volume ratio
    df['price_volume_ratio'] = df['close'] / df['volume']
    
    # Calculate volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Historical 60-day correlations
    corr_momentum = (df['momentum'] * df['daily_return'].shift(-1)).rolling(window=60).corr(df['daily_return'])
    corr_price_volume_ratio = (df['price_volume_ratio'] * df['daily_return'].shift(-1)).rolling(window=60).corr(df['daily_return'])
    corr_volatility = (df['volatility'] * df['daily_return'].shift(-1)).rolling(window=60).corr(df['daily_return'])
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * corr_momentum +
                         df['price_volume_ratio'] * corr_price_volume_ratio +
                         df['volatility'] * corr_volatility)
    
    return heuristics_matrix
