def heuristics_v2(df):
    # Momentum Factor: Calculate the 10-day return
    mom_factor = df['close'].pct_change(10)
    
    # Volatility Factor: Calculate the 20-day standard deviation of daily returns
    vol_factor = df['close'].pct_change().rolling(window=20).std()
    
    # Liquidity Factor: Volume-to-price ratio as a proxy for liquidity
    liq_factor = df['volume'] / df['close']
    
    # Composite Heuristic Matrix by combining all factors
    heuristics_matrix = (mom_factor + vol_factor + liq_factor) / 3
    
    return heuristics_matrix
