import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day rate of change in closing prices
    df['roc'] = df['close'].pct_change(periods=20)
    
    # Calculate the Chaikin Money Flow (CMF) over 20 days
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Generate the heuristic matrix
    heuristics_matrix = df['roc'] + cmf
    
    return heuristics_matrix
