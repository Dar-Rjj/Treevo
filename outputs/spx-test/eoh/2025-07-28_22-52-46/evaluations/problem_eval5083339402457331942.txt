def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute exponential moving averages for close price
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_100'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate 20-day momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate EMA crossover
    df['ema_crossover'] = df['ema_50'] - df['ema_100']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=60).mean()
    
    # Historical Sharpe ratios
    sharpe_ratios = {'momentum': (df['momentum'] * df['daily_return'].shift(-1)).mean() / df['momentum'].std(),
                     'ema_crossover': (df['ema_crossover'] * df['daily_return'].shift(-1)).mean() / df['ema_crossover'].std(),
                     'volume_trend': (df['volume_trend'] * df['daily_return'].shift(-1)).mean() / df['volume_trend'].std()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * sharpe_ratios['momentum'] +
                         df['ema_crossover'] * sharpe_ratios['ema_crossover'] +
                         df['volume_trend'] * sharpe_ratios['volume_trend'])
    
    return heuristics_matrix
