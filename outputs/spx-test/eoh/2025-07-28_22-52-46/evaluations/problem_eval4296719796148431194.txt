def heuristics_v2(df):
    def calculate_momentum(data, window=10):
        return (data['close'].pct_change(periods=window)).fillna(0)

    def calculate_volatility(data, window=10):
        return (data['close'].pct_change().rolling(window=window).std()).fillna(0)

    momentum = calculate_momentum(df)
    volatility = calculate_volatility(df)
    
    # Heuristic scoring: higher momentum with lower volatility scores higher
    heuristics_matrix = (momentum - volatility)
    
    return heuristics_matrix
