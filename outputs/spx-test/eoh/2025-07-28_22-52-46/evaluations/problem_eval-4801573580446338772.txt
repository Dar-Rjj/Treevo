def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical Sortino ratios
    sortino_ratios = {'momentum': (df['momentum'] * df['daily_return'].shift(-1)).mean() / df['momentum'][df['momentum'] < 0].std(),
                      'sma_crossover': (df['sma_crossover'] * df['daily_return'].shift(-1)).mean() / df['sma_crossover'][df['sma_crossover'] < 0].std(),
                      'volume_trend': (df['volume_trend'] * df['daily_return'].shift(-1)).mean() / df['volume_trend'][df['volume_trend'] < 0].std()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * sortino_ratios['momentum'] +
                         df['sma_crossover'] * sortino_ratios['sma_crossover'] +
                         df['volume_trend'] * sortino_ratios['volume_trend'])
    
    return heuristics_matrix
