    df['Alpha_Factor'] = (df['Short_MA'] - df['Long_MA']) + df['Volume_Diff'] * df['Momentum']
    
    # Aggregate the factors into a heuristics matrix
    heuristics_matrix = df[['Alpha_Factor']].copy()
