import pandas as pd

def heuristics_v2(df):
    # Momentum of price change over the last 10 days
    momentum_10 = df['close'].pct_change(periods=10)
    
    # Strength of volume spikes compared to the 5-day moving average
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    volume_spike_strength = (df['volume'] - avg_vol_5) / avg_vol_5
    
    # Relative position of the closing price within the daily range
    relative_close = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Combine features into a single heuristic
    heuristics_matrix = momentum_10 + volume_spike_strength + relative_close
    
    return heuristics_matrix
