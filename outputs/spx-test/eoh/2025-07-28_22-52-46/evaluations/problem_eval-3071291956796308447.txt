def heuristics_v2(df):
    # Calculate daily return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate modified momentum (e.g., 10-day change in log prices)
    df['Log_Price'] = np.log(df['close'])
    df['Momentum'] = df['Log_Price'].diff(10)
    
    # Calculate volatility (standard deviation of daily returns over the last 10 days)
    df['Volatility'] = df['Daily_Return'].rolling(window=10).std()
    
    # Calculate ratio of current volume to 10-day average volume
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Adjust for price range: (High - Low) / Close over the last 10 days
    df['Price_Range_Adjustment'] = (df['high'] - df['low']) / df['close']
    df['Avg_Price_Range'] = df['Price_Range_Adjustment'].rolling(window=10).mean()
    
    # Combine the factors into a heuristic score
    df['Heuristic_Score'] = df['Momentum'] * df['Volume_Ratio'] / (df['Volatility'] + df['Avg_Price_Range'])
    
    # Drop rows with NaN values resulted from calculations
    df.dropna(inplace=True)
    
    # Extract the heuristics score as a series
    heuristics_matrix = df['Heuristic_Score']
    
    return heuristics_matrix
