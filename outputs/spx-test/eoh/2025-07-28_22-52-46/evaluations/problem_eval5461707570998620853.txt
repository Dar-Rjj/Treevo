def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate exponentially smoothed momentum
    df['momentum'] = df['close'].ewm(span=5, adjust=False).mean() - df['close'].shift(5)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Rolling 60-day historical correlations with future returns
    corr_momentum = df['momentum'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_sma_crossover = df['sma_crossover'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_rsi = df['rsi'].rolling(window=60).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * corr_momentum +
                         df['sma_crossover'] * corr_sma_crossover +
                         df['rsi'] * corr_rsi)
    
    return heuristics_matrix
