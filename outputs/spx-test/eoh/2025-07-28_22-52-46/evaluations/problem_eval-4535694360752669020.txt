def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate ATR
    df['tr'] = df[['high', 'low']].apply(lambda x: max(x[0] - x[1], abs(x[0] - df['close'].shift()), abs(x[1] - df['close'].shift())), axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Volatility
    vol_momentum = df['momentum'].rolling(window=60).std()
    vol_rsi = df['rsi'].rolling(window=60).std()
    vol_atr = df['atr'].rolling(window=60).std()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] / vol_momentum +
                         df['rsi'] / vol_rsi +
                         df['atr'] / vol_atr)
    
    return heuristics_matrix
