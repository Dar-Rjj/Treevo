import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day and 10-day moving averages of the closing prices
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Compute the difference between the two moving averages
    diff_ma = ma_5 - ma_10
    
    # Calculate the 20-day average volume
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Scale the difference in moving averages by the 20-day average volume
    heuristics_matrix = (diff_ma / avg_volume).dropna()
    
    return heuristics_matrix
