import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate price relative to SMA
    df['price_to_sma_10'] = df['close'] / df['sma_10']
    df['price_to_sma_30'] = df['close'] / df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Rolling 60-day historical betas with future returns
    beta_with_returns = {
        'momentum': df['momentum'].rolling(window=60).cov(df['daily_return'].shift(-1)) / df['momentum'].rolling(window=60).var(),
        'price_to_sma_10': df['price_to_sma_10'].rolling(window=60).cov(df['daily_return'].shift(-1)) / df['price_to_sma_10'].rolling(window=60).var(),
        'price_to_sma_30': df['price_to_sma_30'].rolling(window=60).cov(df['daily_return'].shift(-1)) / df['price_to_sma_30'].rolling(window=60).var(),
        'volume_trend': df['volume_trend'].rolling(window=60).cov(df['daily_return'].shift(-1)) / df['volume_trend'].rolling(window=60).var()
    }
    
    # Generate heuristic matrix using rolling betas
    heuristics_matrix = (df['momentum'] * beta_with_returns['momentum'] +
                         df['price_to_sma_10'] * beta_with_returns['price_to_sma_10'] +
                         df['price_to_sma_30'] * beta_with_returns['price_to_sma_30'] +
                         df['volume_trend'] * beta_with_returns['volume_trend'])
    
    return heuristics_matrix
