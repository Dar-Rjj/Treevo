import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day rate of change in closing prices
    df['roc'] = df['close'].pct_change(periods=20)
    
    # Calculate the 20-day Accumulation/Distribution Oscillator (ADOSC)
    ad = ((df['close'] - df['low']) - (df['high'] - df['close'])) * df['volume'] / (df['high'] - df['low'])
    adosc = ad.rolling(window=20).mean()
    
    # Generate the heuristic matrix
    heuristics_matrix = df['roc'] + adosc
    
    return heuristics_matrix
