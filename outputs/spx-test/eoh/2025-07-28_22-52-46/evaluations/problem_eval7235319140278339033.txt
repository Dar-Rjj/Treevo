import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate EMA for close price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate momentum based on EMA
    df['momentum_ema'] = df['ema_10'] - df['ema_30']
    
    # Calculate EMA crossover signal
    df['ema_crossover'] = df['ema_10'] - df['ema_30']
    
    # Volume-to-price ratio
    df['volume_to_price_ratio'] = df['volume'] / df['close']
    
    # Modified RSI with adjusted window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    df['rsi_mod'] = 100 - (100 / (1 + rs))
    
    # Rolling 30-day historical Sharpe ratios
    sharpe_ratios_momentum_ema = (df['momentum_ema'] * df['daily_return'].shift(-1)).rolling(window=30).mean() / df['momentum_ema'].rolling(window=30).std()
    sharpe_ratios_ema_crossover = (df['ema_crossover'] * df['daily_return'].shift(-1)).rolling(window=30).mean() / df['ema_crossover'].rolling(window=30).std()
    sharpe_ratios_volume_to_price = (df['volume_to_price_ratio'] * df['daily_return'].shift(-1)).rolling(window=30).mean() / df['volume_to_price_ratio'].rolling(window=30).std()
    sharpe_ratios_rsi_mod = (df['rsi_mod'] * df['daily_return'].shift(-1)).rolling(window=30).mean() / df['rsi_mod'].rolling(window=30).std()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum_ema'].rank(pct=True) * sharpe_ratios_momentum_ema +
                         df['ema_crossover'].rank(pct=True) * sharpe_ratios_ema_crossover +
                         df['volume_to_price_ratio'].rank(pct=True) * sharpe_ratios_volume_to_price +
                         df['rsi_mod'].rank(pct=True) * sharpe_ratios_rsi_mod)
    
    return heuristics_matrix
