def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate 20-day momentum
    df['momentum_20'] = df['close'] - df['close'].shift(20)
    
    # Calculate 5/20-day and 10/30-day SMA crossovers
    df['sma_crossover_5_20'] = df['sma_5'] - df['sma_20']
    df['sma_crossover_10_30'] = df['sma_10'] - df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=14).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical Sharpe ratios
    sharpe_ratios = {'momentum_20': (df['momentum_20'] * df['daily_return'].shift(-1)).mean() / df['momentum_20'].std(),
                     'sma_crossover_5_20': (df['sma_crossover_5_20'] * df['daily_return'].shift(-1)).mean() / df['sma_crossover_5_20'].std(),
                     'sma_crossover_10_30': (df['sma_crossover_10_30'] * df['daily_return'].shift(-1)).mean() / df['sma_crossover_10_30'].std(),
                     'volume_trend': (df['volume_trend'] * df['daily_return'].shift(-1)).mean() / df['volume_trend'].std()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum_20'] * sharpe_ratios['momentum_20'] +
                         df['sma_crossover_5_20'] * sharpe_ratios['sma_crossover_5_20'] +
                         df['sma_crossover_10_30'] * sharpe_ratios['sma_crossover_10_30'] +
                         df['volume_trend'] * sharpe_ratios['volume_trend'])
    
    return heuristics_matrix
