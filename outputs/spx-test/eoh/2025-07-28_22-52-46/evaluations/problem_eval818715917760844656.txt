def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['sma_15'] = df['close'].rolling(window=15).mean()
    df['sma_45'] = df['close'].rolling(window=45).mean()
    df['momentum'] = df['close'] - df['close'].shift(7)
    df['sma_crossover'] = df['sma_15'] - df['sma_45']
    df['volume_trend'] = df['volume'].rolling(window=15).mean() / df['volume'].rolling(window=45).mean()
    sharpe_ratios_momentum = (df['momentum'] * df['daily_return'].shift(-1)).rolling(window=90).mean() / df['momentum'].rolling(window=90).std()
    sharpe_ratios_sma_crossover = (df['sma_crossover'] * df['daily_return'].shift(-1)).rolling(window=90).mean() / df['sma_crossover'].rolling(window=90).std()
    sharpe_ratios_volume_trend = (df['volume_trend'] * df['daily_return'].shift(-1)).rolling(window=90).mean() / df['volume_trend'].rolling(window=90).std()
    heuristics_matrix = (df['momentum'] * sharpe_ratios_momentum + df['sma_crossover'] * sharpe_ratios_sma_crossover + df['volume_trend'] * sharpe_ratios_volume_trend)
    return heuristics_matrix
