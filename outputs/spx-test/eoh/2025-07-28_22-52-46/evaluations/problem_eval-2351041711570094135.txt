def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Rolling 30-day betas
    beta_momentum = (df['momentum'].rolling(window=30).cov(df['daily_return'].shift(-1)) / df['daily_return'].shift(-1).rolling(window=30).var())
    beta_sma_crossover = (df['sma_crossover'].rolling(window=30).cov(df['daily_return'].shift(-1)) / df['daily_return'].shift(-1).rolling(window=30).var())
    beta_volume_trend = (df['volume_trend'].rolling(window=30).cov(df['daily_return'].shift(-1)) / df['daily_return'].shift(-1).rolling(window=30).var())
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * beta_momentum *
                         df['sma_crossover'] * beta_sma_crossover *
                         df['volume_trend'] * beta_volume_trend).apply(lambda x: x ** (1/3) if x > 0 else 0)
    
    return heuristics_matrix
