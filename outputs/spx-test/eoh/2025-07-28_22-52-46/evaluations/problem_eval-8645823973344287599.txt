import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()

    # Calculate EMA
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))

    # Calculate ROC of volume
    df['roc_volume_10'] = df['volume'].pct_change(10)

    # Calculate 60-day mean reversion strengths
    mean_reversion_ema = (df['close'] - df['ema_20']).rolling(window=60).mean().abs()
    mean_reversion_rsi = (50 - df['rsi_14']).rolling(window=60).mean().abs()
    mean_reversion_roc = (df['roc_volume_10']).rolling(window=60).mean().abs()

    # Generate heuristic matrix
    heuristics_matrix = (df['ema_20'] * mean_reversion_ema +
                         df['rsi_14'] * mean_reversion_rsi +
                         df['roc_volume_10'] * mean_reversion_roc)
    
    return heuristics_matrix
