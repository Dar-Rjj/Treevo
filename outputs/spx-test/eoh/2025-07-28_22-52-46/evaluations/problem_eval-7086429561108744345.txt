def heuristics_v2(df):
    # Compute the numerator (|High - Low|) and the denominator (Open + Close)/2
    num = abs(df['high'] - df['low'])
    denom = (df['open'] + df['close']) / 2
    
    # Calculate the base heuristic
    base_heuristic = num / denom
    
    # Calculate the 10-day standard deviation of the volume
    vol_std = df['volume'].rolling(window=10).std()
    
    # Incorporate the volume standard deviation into the heuristic
    heuristics_matrix = base_heuristic * vol_std
    
    return heuristics_matrix
