import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute exponential moving averages for close price
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_60'] = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate price relative to EMA
    df['price_to_ema_20'] = df['close'] / df['ema_20']
    df['price_to_ema_60'] = df['close'] / df['ema_60']
    
    # Volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Standard deviations of features
    std_devs = {'momentum': df['momentum'].std(),
                'price_to_ema_20': df['price_to_ema_20'].std(),
                'price_to_ema_60': df['price_to_ema_60'].std(),
                'volume_change': df['volume_change'].std()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * std_devs['momentum'] +
                         df['price_to_ema_20'] * std_devs['price_to_ema_20'] +
                         df['price_to_ema_60'] * std_devs['price_to_ema_60'] +
                         df['volume_change'] * std_devs['volume_change'])
    
    return heuristics_matrix
