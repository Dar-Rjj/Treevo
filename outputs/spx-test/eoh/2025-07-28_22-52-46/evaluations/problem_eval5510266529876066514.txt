import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate momentum with exponential smoothing
    df['momentum'] = df['close'].ewm(span=5, adjust=False).mean() - df['close'].shift(5)
    
    # Calculate price relative to SMA
    df['price_to_sma_5'] = df['close'] / df['sma_5']
    df['price_to_sma_20'] = df['close'] / df['sma_20']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=5).mean() / df['volume'].rolling(window=20).mean()
    
    # Historical return correlations
    corr_with_returns = {'momentum': df['momentum'].corr(df['daily_return'].shift(-1)), 
                         'price_to_sma_5': df['price_to_sma_5'].corr(df['daily_return'].shift(-1)),
                         'price_to_sma_20': df['price_to_sma_20'].corr(df['daily_return'].shift(-1)),
                         'volume_trend': df['volume_trend'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * corr_with_returns['momentum'] +
                         df['price_to_sma_5'] * corr_with_returns['price_to_sma_5'] +
                         df['price_to_sma_20'] * corr_with_returns['price_to_sma_20'] +
                         df['volume_trend'] * corr_with_returns['volume_trend'])
    
    return heuristics_matrix
