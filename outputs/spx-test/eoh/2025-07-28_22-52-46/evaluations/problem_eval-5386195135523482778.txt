import pandas as pd
import talib

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_60'] = df['close'].rolling(window=60).mean()
    
    # Calculate RSI
    df['rsi'] = talib.RSI(df['close'], timeperiod=14)
    
    # Calculate MACD
    macd, signal, _ = talib.MACD(df['close'], fastperiod=12, slowperiod=26, signalperiod=9)
    df['macd'] = macd
    df['signal'] = signal
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=60).mean()
    
    # Historical return correlations
    corr_with_returns = {
        'rsi': df['rsi'].corr(df['daily_return'].shift(-1)),
        'macd': df['macd'].corr(df['daily_return'].shift(-1)),
        'signal': df['signal'].corr(df['daily_return'].shift(-1)),
        'volume_trend': df['volume_trend'].corr(df['daily_return'].shift(-1))
    }
    
    # Generate heuristic matrix
    heuristics_matrix = (df['rsi'] * corr_with_returns['rsi'] +
                         df['macd'] * corr_with_returns['macd'] +
                         df['signal'] * corr_with_returns['signal'] +
                         df['volume_trend'] * corr_with_returns['volume_trend'])
    
    return heuristics_matrix
