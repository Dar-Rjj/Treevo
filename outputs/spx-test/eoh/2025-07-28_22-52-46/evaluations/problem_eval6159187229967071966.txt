def heuristics_v2(df):
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['ma_30'] = df['close'].rolling(window=30).mean()
    df['ma_60'] = df['close'].rolling(window=60).mean()
    
    df['ratio_10'] = df['close'] / df['ma_10']
    df['ratio_30'] = df['close'] / df['ma_30']
    df['ratio_60'] = df['close'] / df['ma_60']
    
    df['average_ratio'] = (df['ratio_10'] + df['ratio_30'] + df['ratio_60']) / 3
    
    heuristics_matrix = df['average_ratio'].ewm(span=5, adjust=False).mean()
    
    return heuristics_matrix
