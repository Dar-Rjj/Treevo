def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Compute ATR
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Custom Weighted Moving Average
    n = 20
    weights = list(range(1, n+1))
    df['wma_close'] = (df['close'].rolling(window=n).apply(lambda prices: (prices * weights).sum() / sum(weights), raw=True))
    
    # Heuristic Matrix
    heuristics_matrix = (df['rsi'] - 50) + (100 / df['atr']) + df['wma_close']
    
    return heuristics_matrix
