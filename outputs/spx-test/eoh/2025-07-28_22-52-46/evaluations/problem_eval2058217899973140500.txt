def heuristics_v2(df):
    # Compute the ratio of the absolute change in close price to the 50-day average volume
    abs_change_close = abs(df['close'] - df['close'].shift(1))
    avg_50_day_volume = df['volume'].rolling(window=50).mean()
    ratio = abs_change_close / avg_50_day_volume
    
    # Apply a 20-day exponential moving average to smooth the series
    heuristics_matrix = ratio.ewm(span=20).mean()
    
    return heuristics_matrix
