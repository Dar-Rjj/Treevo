def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_100'] = df['close'].rolling(window=100).mean()
    
    # Calculate 20-day and 50-day momentum
    df['momentum_20'] = df['close'] - df['close'].shift(20)
    df['momentum_50'] = df['close'] - df['close'].shift(50)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_50'] - df['sma_100']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical correlations
    correlations = {'momentum_20': (df['momentum_20'] * df['daily_return'].shift(-1)).corr(),
                    'momentum_50': (df['momentum_50'] * df['daily_return'].shift(-1)).corr(),
                    'sma_crossover': (df['sma_crossover'] * df['daily_return'].shift(-1)).corr(),
                    'volume_trend': (df['volume_trend'] * df['daily_return'].shift(-1)).corr()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum_20'] * correlations['momentum_20'] +
                         df['momentum_50'] * correlations['momentum_50'] +
                         df['sma_crossover'] * correlations['sma_crossover'] +
                         df['volume_trend'] * correlations['volume_trend'])
    
    return heuristics_matrix
