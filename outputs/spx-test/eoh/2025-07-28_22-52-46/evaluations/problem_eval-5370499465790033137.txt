def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 20-day price volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Calculate the ratio of the latest close to the highest high over the past 50 days
    df['high_ratio'] = df['close'] / df['high'].rolling(window=50).max()
    
    # Calculate 10-day momentum
    df['momentum'] = df['close'].pct_change(periods=10)
    
    # Historical correlations
    correlations = {'volatility': df['volatility'].corr(df['daily_return'].shift(-1)),
                    'high_ratio': df['high_ratio'].corr(df['daily_return'].shift(-1)),
                    'momentum': df['momentum'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['volatility'] * correlations['volatility'] +
                         df['high_ratio'] * correlations['high_ratio'] +
                         df['momentum'] * correlations['momentum'])
    
    return heuristics_matrix
