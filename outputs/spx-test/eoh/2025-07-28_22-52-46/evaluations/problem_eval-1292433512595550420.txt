def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute exponential moving averages for close price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate EMA crossover
    df['ema_crossover'] = df['ema_10'] - df['ema_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical mean returns
    mean_returns = {'ema_crossover': df['ema_crossover'].mean(),
                    'rsi': df['rsi'].mean(),
                    'volume_trend': df['volume_trend'].mean()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['ema_crossover'] * mean_returns['ema_crossover'] +
                         df['rsi'] * mean_returns['rsi'] +
                         df['volume_trend'] * mean_returns['volume_trend'])
    
    return heuristics_matrix
