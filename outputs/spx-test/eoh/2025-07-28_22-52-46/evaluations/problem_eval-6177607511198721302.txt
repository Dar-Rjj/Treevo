import pandas as pd

def heuristics_v2(df):
    # Compute the numerator (High - Low) and the denominator (Open + Close)/2
    num = df['high'] - df['low']
    denom = (df['open'] + df['close']) / 2
    
    # Calculate the base heuristic
    base_heuristic = num / denom
    
    # Apply a 3-day moving average to smooth the series
    heuristics_matrix = base_heuristic.rolling(window=3).mean()
    
    return heuristics_matrix
