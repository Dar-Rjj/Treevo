import pandas as pd

def heuristics_v2(df):
    # Momentum: difference between the 10-day and 50-day simple moving averages
    sma_10 = df['close'].rolling(window=10).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    momentum = sma_10 - sma_50
    
    # Price action relative to historical trading range
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    trading_range_position = (df['close'] - low_20) / (high_20 - low_20)
    
    # Combine features into a single heuristic
    heuristics_matrix = momentum + (trading_range_position * 100)
    
    return heuristics_matrix
