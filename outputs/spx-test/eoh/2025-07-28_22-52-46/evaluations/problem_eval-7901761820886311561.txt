def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_10'] - df['sma_30']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical Information ratios
    info_ratios = {'momentum': (df['momentum'] * df['daily_return'].shift(-1)).mean() / (df['momentum'] * df['daily_return']).std(),
                   'sma_crossover': (df['sma_crossover'] * df['daily_return'].shift(-1)).mean() / (df['sma_crossover'] * df['daily_return']).std(),
                   'volume_trend': (df['volume_trend'] * df['daily_return'].shift(-1)).mean() / (df['volume_trend'] * df['daily_return']).std()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * info_ratios['momentum'] +
                         df['sma_crossover'] * info_ratios['sma_crossover'] +
                         df['volume_trend'] * info_ratios['volume_trend'])
    
    return heuristics_matrix
