def heuristics_v2(df):
    # Compute the numerator (|High - Low|) and the denominator (Open + Close)/2
    num = abs(df['high'] - df['low'])
    denom = (df['open'] + df['close']) / 2
    
    # Calculate the base heuristic
    base_heuristic = num / denom
    
    # Apply a 10-day Exponentially Weighted Moving Average to smooth the series
    heuristics_matrix = base_heuristic.ewm(span=10).mean()
    
    return heuristics_matrix
