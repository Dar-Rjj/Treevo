import pandas as pd

def heuristics_v2(df):
    # Calculate the Average True Range (ATR)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the Rate of Change (ROC) of volume
    roc_volume = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Volatility of each component (simplified as their standard deviation)
    vol_atr = atr.std()
    vol_roc_volume = roc_volume.std()
    
    # Total volatility for weight calculation
    total_vol = vol_atr + vol_roc_volume
    
    # Weights based on volatility
    weight_atr = vol_atr / total_vol
    weight_roc_volume = vol_roc_volume / total_vol
    
    # Compute the heuristics matrix
    heuristics_matrix = (weight_atr * atr + 
                         weight_roc_volume * roc_volume)
    
    return heuristics_matrix
