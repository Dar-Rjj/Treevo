def heuristics_v2(df):
    # Momentum: 21-day rate of change (ROC)
    roc_21 = df['close'].pct_change(periods=21)
    
    # Volume relative to 30-day average volume
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    relative_volume = df['volume'] / avg_volume_30
    
    # Combine features into a single heuristic
    heuristics_matrix = roc_21 + relative_volume
    
    return heuristics_matrix
