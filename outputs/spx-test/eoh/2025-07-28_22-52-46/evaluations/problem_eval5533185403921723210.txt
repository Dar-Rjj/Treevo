def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_30'] = df['close'].rolling(window=30).mean()
    df['sma_60'] = df['close'].rolling(window=60).mean()
    
    # Calculate 10-day momentum
    df['momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_30'] - df['sma_60']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=20).mean()
    
    # Historical information coefficients
    ic_momentum = df['momentum'].corr(df['daily_return'].shift(-1))
    ic_sma_crossover = df['sma_crossover'].corr(df['daily_return'].shift(-1))
    ic_volume_trend = df['volume_trend'].corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * ic_momentum +
                         df['sma_crossover'] * ic_sma_crossover +
                         df['volume_trend'] * ic_volume_trend)
    
    return heuristics_matrix
