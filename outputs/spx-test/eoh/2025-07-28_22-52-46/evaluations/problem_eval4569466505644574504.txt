import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate momentum (e.g., 10-day momentum)
    df['Momentum'] = df['close'].pct_change(10)
    
    # Calculate volatility (standard deviation of daily returns over the last 10 days)
    df['Volatility'] = df['Daily_Return'].rolling(window=10).std()
    
    # Calculate weighted moving average of volume
    df['WMA_Volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Combine the factors into a heuristic score
    df['Heuristic_Score'] = df['Momentum'] * df['WMA_Volume'] / df['Volatility']
    
    # Drop rows with NaN values resulted from calculations
    df.dropna(inplace=True)
    
    # Extract the heuristics score as a series
    heuristics_matrix = df['Heuristic_Score']
    
    return heuristics_matrix
