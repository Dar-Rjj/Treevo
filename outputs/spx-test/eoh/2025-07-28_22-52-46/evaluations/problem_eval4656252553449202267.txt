def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate price-to-volume ratio
    df['price_to_volume'] = df['close'] / df['volume']
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate close to highest high ratio
    df['high_n'] = df['high'].rolling(window=14).max()
    df['close_to_high'] = df['close'] / df['high_n']
    
    # Historical MAE
    mae = {'price_to_volume': (df['price_to_volume'] - df['daily_return'].shift(-1)).abs().mean(),
           'rsi': (df['rsi'] - df['daily_return'].shift(-1)).abs().mean(),
           'close_to_high': (df['close_to_high'] - df['daily_return'].shift(-1)).abs().mean()}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['price_to_volume'] * mae['price_to_volume'] +
                         df['rsi'] * mae['rsi'] +
                         df['close_to_high'] * mae['close_to_high'])
    
    return heuristics_matrix
