import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate volatility
    df['volatility'] = df['close'].rolling(window=50).std()
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Price-to-volume ratio
    df['price_to_volume'] = df['close'] / df['volume']
    
    # Rolling 50-day historical Sharpe ratios
    sharpe_ratios_volatility = (df['volatility'] * df['daily_return'].shift(-1)).rolling(window=50).mean() / df['volatility'].rolling(window=50).std()
    sharpe_ratios_rsi = (df['rsi'] * df['daily_return'].shift(-1)).rolling(window=50).mean() / df['rsi'].rolling(window=50).std()
    sharpe_ratios_price_to_volume = (df['price_to_volume'] * df['daily_return'].shift(-1)).rolling(window=50).mean() / df['price_to_volume'].rolling(window=50).std()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['volatility'] * sharpe_ratios_volatility +
                         df['rsi'] * sharpe_ratios_rsi +
                         df['price_to_volume'] * sharpe_ratios_price_to_volume)
    
    return heuristics_matrix
