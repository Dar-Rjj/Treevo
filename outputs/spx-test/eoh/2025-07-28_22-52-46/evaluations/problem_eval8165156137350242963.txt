def heuristics_v2(df):
    # Compute the 5-day and 20-day percentage change in closing price
    df['price_change_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['price_change_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Compute the 5-day and 20-day change in volume
    df['volume_change_5'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    df['volume_change_20'] = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    
    # Combine the changes to form the heuristic matrix
    heuristics_matrix = (df['price_change_5'] * df['volume_change_5']) - (df['price_change_20'] * df['volume_change_20'])
    
    return heuristics_matrix
