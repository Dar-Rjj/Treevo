import pandas as pd

def heuristics_v2(df):
    # Compute the numerator (Close) and the denominator (High + Low)/2
    num = df['close']
    denom = (df['high'] + df['low']) / 2
    
    # Calculate the base heuristic
    base_heuristic = num / denom
    
    # Apply a 5-day exponential moving average to smooth the series
    heuristics_matrix = base_heuristic.ewm(span=5, adjust=False).mean()
    
    return heuristics_matrix
