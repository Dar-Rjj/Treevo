def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute simple moving averages for close price
    df['sma_15'] = df['close'].rolling(window=15).mean()
    df['sma_60'] = df['close'].rolling(window=60).mean()
    
    # Calculate momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate SMA crossover
    df['sma_crossover'] = df['sma_15'] - df['sma_60']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=60).mean()
    
    # Rolling 60-day historical correlations with future returns
    corr_momentum = df['momentum'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_sma_crossover = df['sma_crossover'].rolling(window=60).corr(df['daily_return'].shift(-1))
    corr_volume_trend = df['volume_trend'].rolling(window=60).corr(df['daily_return'].shift(-1))
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * corr_momentum +
                         df['sma_crossover'] * corr_sma_crossover +
                         df['volume_trend'] * corr_volume_trend)
    
    return heuristics_matrix
