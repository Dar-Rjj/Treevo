def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute exponential moving averages for close price
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_100'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate 20-day momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate EMA crossover
    df['ema_crossover'] = df['ema_50'] - df['ema_100']
    
    # Volume trend
    df['volume_trend'] = df['volume'].rolling(window=20).mean() / df['volume'].rolling(window=30).mean()
    
    # Historical correlations
    correlations = {'momentum': df['momentum'].corr(df['daily_return'].shift(-1)),
                    'ema_crossover': df['ema_crossover'].corr(df['daily_return'].shift(-1)),
                    'volume_trend': df['volume_trend'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * correlations['momentum'] +
                         df['ema_crossover'] * correlations['ema_crossover'] +
                         df['volume_trend'] * correlations['volume_trend'])
    
    return heuristics_matrix
