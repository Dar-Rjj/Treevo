def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate 10-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate 20-day Exponential Moving Average
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the difference between the latest close and 20-day EMA
    df['ema_diff'] = df['close'] - df['ema_20']
    
    # Calculate the ratio of the latest volume to its 50-day average
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=50).mean()
    
    # Historical correlations with future returns
    correlations = {'rsi': df['rsi'].corr(df['daily_return'].shift(-1)),
                    'ema_diff': df['ema_diff'].corr(df['daily_return'].shift(-1)),
                    'volume_ratio': df['volume_ratio'].corr(df['daily_return'].shift(-1))}
    
    # Generate heuristic matrix
    heuristics_matrix = (df['rsi'] * correlations['rsi'] +
                         df['ema_diff'] * correlations['ema_diff'] +
                         df['volume_ratio'] * correlations['volume_ratio'])
    
    return heuristics_matrix
