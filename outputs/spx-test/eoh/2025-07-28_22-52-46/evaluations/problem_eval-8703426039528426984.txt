import pandas as pd

def heuristics_v2(df):
    # Compute daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Compute momentum and mean reversion signals
    df['momentum'] = df['close'].pct_change(20)
    df['mean_reversion'] = (df['close'] - df['close'].rolling(window=20).mean()) / df['close'].rolling(window=20).std()
    
    # Compute volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Historical volatility
    df['volatility_momentum'] = df['momentum'].rolling(window=50).std()
    df['volatility_mean_reversion'] = df['mean_reversion'].rolling(window=50).std()
    df['volatility_volume_change'] = df['volume_change'].rolling(window=50).std()
    
    # Generate heuristic matrix
    heuristics_matrix = (df['momentum'] * (1 / df['volatility_momentum']) +
                         df['mean_reversion'] * (1 / df['volatility_mean_reversion']) +
                         df['volume_change'] * (1 / df['volatility_volume_change']))
    
    return heuristics_matrix
