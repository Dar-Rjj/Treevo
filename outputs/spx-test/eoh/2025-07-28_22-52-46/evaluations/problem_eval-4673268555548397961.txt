def heuristics_v2(df):
    # Compute the standard deviation of daily returns over the last 30 days to measure volatility
    df['volatility'] = df['close'].pct_change().rolling(window=30).std()
    
    # Calculate the 50-day high
    df['high_50'] = df['high'].rolling(window=50).max()
    
    # Calculate the adjusted close price by dividing the close price by its volatility
    df['adjusted_close'] = df['close'] / df['volatility']
    
    # Calculate the ratio of the closing price to the 50-day high
    df['close_to_high_ratio'] = df['close'] / df['high_50']
    
    # Generate the heuristic matrix
    heuristics_matrix = df['adjusted_close'] + df['close_to_high_ratio']
    
    return heuristics_matrix
