```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day return as a momentum indicator
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 15
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the alpha factor as the product of momentum and the inverse of volatility, 
    # weighted by the smoothed volume and adjusted by RSI
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    alpha_factor = (df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema_15']) * (df['rsi_14'] / 50)
    
    return alpha_factor
```
