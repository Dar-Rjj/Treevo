```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the average volume over the last 5 days
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    # Calculate the factor as the ratio of close price difference to the average volume
    factor = df['close_diff'] / (df['avg_volume_5'] + 1e-7)
    return factor
```
