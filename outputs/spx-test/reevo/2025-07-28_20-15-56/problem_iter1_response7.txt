```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and average of open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    diff = df['close'] - avg_open_close
    
    # Calculate a volume-weighted indicator, which considers the ratio of today's volume to the 5-day moving average of volume
    volume_ma_5d = df['volume'].rolling(window=5).mean()
    volume_indicator = df['volume'] / volume_ma_5d
    
    # The final heuristic factor is the product of the price difference and the volume indicator
    alpha_factor = diff * volume_indicator
    return alpha_factor
```
