```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 10
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(window=14).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_20d'] + 1e-7)
    
    # Calculate the 10-day and 30-day EMA of the close price for dual smoothing
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the adaptive metric as the difference between the EMAs
    df['adaptive_metric'] = df['ema_10'] - df['ema_30']
    
    # Calculate the 10-day and 30-day EMA of the volume for dual smoothing
    df['v_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['v_ema_30'] = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Calculate the adaptive volume metric as the difference between the EMAs
    df['adaptive_volume_metric'] = df['v_ema_10'] - df['v_ema_30']
    
    # Calculate the money flow index (MFI) as a liquidity measure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(df['close'] > df['close'].shift(1), 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(df['close'] < df['close'].shift(1), 0).rolling(window=14).sum()
    money_ratio = positive_money_flow / (negative_money_flow + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Combine the factors to create the alpha factor
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema_10'] *
                    df['rsi_14'] * df['market_sentiment'] * df['adaptive_metric'] * df['adaptive_volume_metric'] * df['mfi_14'])
    
    return alpha_factor
```
