```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, and close and open to capture the day's range and price movement
    daily_range = df['high'] - df['low']
    price_movement = (df['close'] - df['open']).abs()

    # Momentum component: 3-day and 10-day percentage change in close price
    three_day_momentum = df['close'].pct_change(3).fillna(0)
    ten_day_momentum = df['close'].pct_change(10).fillna(0)

    # Volume-Price Trend: a factor that reflects the relationship between volume and price changes
    vpt = ((df['close'] - df['close'].shift(1)) / df['close'].shift(1)) * df['volume']

    # Average of the last 5 days' and 20 days' Volume-Price Trends to smooth out the values
    vpt_5d_avg = vpt.rolling(window=5).mean().fillna(0)
    vpt_20d_avg = vpt.rolling(window=20).mean().fillna(0)

    # Volatility component: 20-day standard deviation of the daily log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std().fillna(0)

    # Combining all the components into a single alpha factor
    alpha_factor = (price_movement / (daily_range + 1e-7)) * three_day_momentum * ten_day_momentum * vpt_5d_avg * vpt_20d_avg / (volatility + 1e-7)
    return alpha_factor
```
