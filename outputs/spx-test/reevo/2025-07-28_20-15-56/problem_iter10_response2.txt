```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 15-day return as a longer-term momentum indicator for better trend identification
    df['momentum_15d'] = df['close'].pct_change(periods=15)
    
    # Calculate the 60-day rolling standard deviation of log returns as a more stable and robust volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_60d'] = df['log_returns'].rolling(window=60).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 20 for smoother volume signal
    df['smoothed_volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(window=14).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_60d'] + 1e-7)
    
    # Calculate the alpha factor as the product of longer-term momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI and market sentiment
    alpha_factor = (df['momentum_15d'] * df['inv_volatility'] * df['smoothed_volume_ema_20'] *
                    df['rsi_14'] * df['market_sentiment'])
    
    return alpha_factor
```
