```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close (momentum)
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over the last 5 days
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of close price difference to the average volume
    factor_momentum = df['close_diff'] / (df['avg_volume_5'] + 1e-7)
    
    # Calculate the 20-day rolling standard deviation of the close price (volatility)
    df['close_volatility_20'] = df['close'].rolling(window=20).std()
    
    # Normalize the momentum factor by the volatility
    factor_volatility_adjusted = factor_momentum / (df['close_volatility_20'] + 1e-7)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors: momentum, volatility-adjusted, and RSI
    alpha_factor = factor_momentum + factor_volatility_adjusted + (rsi - 50)
    
    return alpha_factor
```
