```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 10
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the adaptive window size based on the trend of the 5-day momentum
    df['trend_strength'] = df['momentum_5d'].rolling(window=5).mean().abs()
    df['adaptive_window'] = (df['trend_strength'] * 20 + 10).astype(int).clip(lower=10, upper=30)
    
    # Adjust the momentum and volatility factors using the adaptive window
    df['momentum_adaptive'] = df['close'].pct_change(periods=df['adaptive_window'])
    df['volatility_adaptive'] = df['log_returns'].rolling(window=df['adaptive_window']).std()
    
    # Calculate the alpha factor as the product of adaptive momentum and the inverse of adaptive volatility, 
    # weighted by the smoothed volume
    df['inv_volatility_adaptive'] = 1 / (df['volatility_adaptive'] + 1e-7)
    alpha_factor = df['momentum_adaptive'] * df['inv_volatility_adaptive'] * df['smoothed_volume_ema_10']
    
    return alpha_factor
```
