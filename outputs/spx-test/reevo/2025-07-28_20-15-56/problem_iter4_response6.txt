```python
def heuristics_v2(df: pd.DataFrame, sector_trends: pd.Series, macro_indicators: pd.Series) -> pd.Series:
    # Calculate the 10-day return as a momentum indicator
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 15
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) over an adaptive window
    def adaptive_rsi(close, window=14):
        delta = close.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / (loss + 1e-7)
        rsi = 100 - (100 / (1 + rs))
        return rsi
    
    # Use an adaptive window for RSI calculation
    rsi_window = 14 + (df['volatility_30d'] * 5).astype(int)
    df['rsi_adaptive'] = adaptive_rsi(df['close'], rsi_window)
    
    # Integrate sector-specific trends
    df['sector_trend'] = sector_trends.reindex(df.index, method='ffill')
    
    # Integrate macroeconomic indicators
    df['macro_indicator'] = macro_indicators.reindex(df.index, method='ffill')
    
    # Calculate the alpha factor as the product of momentum and the inverse of volatility, 
    # weighted by the smoothed volume, adjusted by RSI, and influenced by sector and macroeconomic trends
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    alpha_factor = (df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema_15']) * (df['rsi_adaptive'] / 50) * (df['sector_trend'] + df['macro_indicator'])
    
    return alpha_factor
```
