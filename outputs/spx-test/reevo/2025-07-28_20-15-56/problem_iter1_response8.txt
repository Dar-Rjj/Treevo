```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, smoothed by closing price to reduce noise impact
    price_range_smoothed = (df['high'] - df['low']) * (1 + df['close'].pct_change())
    
    # Volume-weighted close price
    volume_weighted_close = df['close'] * df['volume']
    
    # Shift the volume weighted close price for 1 day to represent past value
    volume_weighted_close_lag1 = volume_weighted_close.shift(1)
    
    # Calculate today's return based on the volume weighted close
    ret_today_volume_weighted = (volume_weighted_close - volume_weighted_close_lag1) / volume_weighted_close_lag1
    
    # Heuristic factor is a combination of smoothed price range with volume weighted returns 
    heuristic_factor = (df['close'] - df['open']) / (price_range_smoothed) * ret_today_volume_weighted

    return heuristic_factor
```
