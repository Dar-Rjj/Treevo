```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 7-day return as a momentum indicator
    df['momentum_7d'] = df['close'].pct_change(periods=7)
    
    # Calculate the 30-day rolling standard deviation of log returns as a more robust volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 15 for better smoothing
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 20-day average trading volume to identify trading volume patterns
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the 200-day moving average as a long-term mean reversion signal
    df['ma_200d'] = df['close'].rolling(window=200).mean()
    
    # Calculate the distance from the 200-day moving average as a mean reversion factor
    df['dist_from_ma_200d'] = (df['close'] - df['ma_200d']) / df['close']
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, and average trading volume
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    alpha_factor = (df['momentum_7d'] * df['inv_volatility'] * df['smoothed_volume_ema_15'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_20d'] * (1 + df['dist_from_ma_200d']))
    
    return alpha_factor
```
