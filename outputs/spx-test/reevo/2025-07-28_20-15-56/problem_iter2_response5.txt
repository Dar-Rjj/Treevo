def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and average of open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    diff = df['close'] - avg_open_close

    # Calculate a volume-weighted indicator, which considers the ratio of today's volume to the 5-day moving average of volume
    volume_ma_5d = df['volume'].rolling(window=5).mean()
    volume_indicator = df['volume'] / volume_ma_5d

    # Calculate the adaptive window size for the moving average based on volatility
    volatility = df['high'] - df['low']
    adaptive_window = (volatility.rolling(window=10).mean() * 5).astype(int).clip(lower=5, upper=20)

    # Calculate the trend strength using the adaptive moving average
    trend_strength = df['close'] - df['close'].rolling(window=adaptive_window).mean()

    # Combine the factors into a single alpha factor
    alpha_factor = diff * volume_indicator * trend_strength

    return alpha_factor
