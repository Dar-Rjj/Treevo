```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price, then smooth it with an exponential moving average
    df['price_diff'] = df['close'] - df['open']
    df['ema_price_diff'] = df['price_diff'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio of volume to 5-day average volume
    df['vol_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Final factor is a combination of smoothed price changes and volume ratios
    df['alpha_factor'] = df['ema_price_diff'] * df['vol_ratio']
    
    return df['alpha_factor']
```
