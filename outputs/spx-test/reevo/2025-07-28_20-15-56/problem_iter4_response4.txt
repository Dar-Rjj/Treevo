```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std().fillna(0)
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 10
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the money flow index (MFI) as a volume-weighted momentum indicator
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(delta > 0, 0)
    negative_money_flow = money_flow.where(delta < 0, 0)
    mfi_ratio = positive_money_flow.rolling(window=14).sum() / (negative_money_flow.rolling(window=14).sum() + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + mfi_ratio))
    
    # Calculate the alpha factor as a weighted combination of various indicators
    df['inv_volatility'] = 1 / (df['volatility_20d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema_10'] *
                    df['rsi_14'] * df['market_sentiment'] * df['mfi_14'])
    
    # Normalize the alpha factor to improve interpretability
    alpha_factor = (alpha_factor - alpha_factor.mean()) / (alpha_factor.std() + 1e-7)
    
    return alpha_factor
```
