```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a short-term momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day Exponential Moving Average (EMA) of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_10d_ema'] = df['log_returns'].ewm(span=10, adjust=False).std()
    
    # Smoothed volume using EMA with a span of 10
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 10-day average trading volume to identify short-term trading volume patterns
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 50-day EMA as a medium-term mean reversion signal
    df['ema_50d'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the distance from the 50-day EMA as a mean reversion factor
    df['dist_from_ema_50d'] = (df['close'] - df['ema_50d']) / df['close']
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_10d_ema'] + 1e-7)
    
    # Combine all factors into an alpha factor
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema_10'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_10d'] * (1 + df['dist_from_ema_50d']))
    
    # Ensure the alpha factor is not NaN
    alpha_factor = alpha_factor.fillna(0)
    
    # Add a price-to-volume ratio to capture the relationship between price and volume
    df['price_to_volume_ratio'] = df['close'] / df['volume']
    
    # Adjust the alpha factor with the price-to-volume ratio
    alpha_factor = alpha_factor * df['price_to_volume_ratio']
    
    return alpha_factor
```
