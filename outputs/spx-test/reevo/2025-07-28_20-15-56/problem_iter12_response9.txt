```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day return as a longer-term momentum indicator
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day rolling standard deviation of log returns as a more stable volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 20
    df['smoothed_volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 20-day average trading volume to identify trading volume patterns
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the 52-week high and low for mean reversion
    df['52w_high'] = df['close'].rolling(window=252).max()
    df['52w_low'] = df['close'].rolling(window=252).min()
    df['mean_reversion'] = (df['close'] - df['52w_low']) / (df['52w_high'] - df['52w_low'])
    
    # Calculate the 60-day volume-weighted moving average price
    df['vwap_60d'] = (df['close'] * df['volume']).rolling(window=60).sum() / df['volume'].rolling(window=60).sum()
    
    # Calculate the 100-day EMA of the close price for long-term smoothing
    df['ema_100'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, adjusted by the RSI, market sentiment, average trading volume,
    # mean reversion, 60-day volume-weighted moving average price, and 100-day EMA
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    alpha_factor = (df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema_20'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_20d'] * df['mean_reversion'] * df['vwap_60d'] * df['ema_100'])
    
    return alpha_factor
```
