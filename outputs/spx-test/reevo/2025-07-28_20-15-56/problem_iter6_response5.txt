```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 7-day return as a short-term momentum indicator
    df['momentum_7d'] = df['close'].pct_change(periods=7)
    
    # Calculate the 15-day rolling standard deviation of log returns as a mid-term volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_15d'] = df['log_returns'].rolling(window=15).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 12
    df['smoothed_volume_ema_12'] = df['volume'].ewm(span=12, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) for overbought/oversold signals
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the price-to-volume ratio as a liquidity measure
    df['price_to_volume_ratio'] = df['close'] / df['volume']
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, and price-to-volume ratio
    df['inv_volatility'] = 1 / (df['volatility_15d'] + 1e-7)
    alpha_factor = (df['momentum_7d'] * df['inv_volatility'] * df['smoothed_volume_ema_12'] *
                    (df['rsi_14'] / 50) * df['market_sentiment'] * df['price_to_volume_ratio'])
    
    return alpha_factor
```
