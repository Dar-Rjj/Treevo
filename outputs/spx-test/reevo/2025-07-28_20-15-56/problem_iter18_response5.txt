```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day return as a momentum indicator
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 15
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 52-week high-to-low ratio to capture long-term mean reversion
    df['52w_high_low_ratio'] = df['high'].rolling(window=252).max() / df['low'].rolling(window=252).min()
    
    # Simplify the alpha factor by focusing on key, interpretable drivers
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    alpha_factor = (df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema_15'] *
                    df['market_sentiment'] * df['52w_high_low_ratio'])
    
    return alpha_factor
```
