```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day return as a more responsive momentum indicator
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Calculate the 10-day Exponential Moving Average (EMA) of log returns for a more responsive volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_10d_ema'] = df['log_returns'].ewm(span=10, adjust=False).std()
    
    # Smoothed volume using EMA with a span of 5 for more responsiveness
    df['smoothed_volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 5-day average trading volume to identify shorter-term volume patterns
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the 20-day EMA as a more responsive long-term mean reversion signal
    df['ema_20d'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the distance from the 20-day EMA as a mean reversion factor
    df['dist_from_ema_20d'] = (df['close'] - df['ema_20d']) / df['close']
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_10d_ema'] + 1e-7)
    
    # Combine all factors into an alpha factor
    alpha_factor = (df['momentum_3d'] * df['inv_volatility'] * df['smoothed_volume_ema_5'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_5d'] * (1 + df['dist_from_ema_20d']))
    
    return alpha_factor
```
