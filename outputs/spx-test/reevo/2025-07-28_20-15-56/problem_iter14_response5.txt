```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day return as momentum indicators
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Calculate the 10-day and 30-day rolling standard deviation of log returns as volatility factors
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_10d'] = df['log_returns'].rolling(window=10).std()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with spans of 10 and 20
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['smoothed_volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) with windows of 14 and 28
    delta = df['close'].diff()
    gain_14 = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss_14 = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs_14 = gain_14 / (loss_14 + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs_14))
    
    gain_28 = (delta.where(delta > 0, 0)).rolling(window=28).mean()
    loss_28 = (-delta.where(delta < 0, 0)).rolling(window=28).mean()
    rs_28 = gain_28 / (loss_28 + 1e-7)
    df['rsi_28'] = 100 - (100 / (1 + rs_28))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 10-day and 20-day average trading volume to identify trading volume patterns
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the alpha factor as a combination of short-term and long-term momentum,
    # adjusted by inverse volatility, smoothed volume, RSI, market sentiment, and average trading volume
    df['inv_volatility_10d'] = 1 / (df['volatility_10d'] + 1e-7)
    df['inv_volatility_30d'] = 1 / (df['volatility_30d'] + 1e-7)
    
    alpha_factor = (df['momentum_5d'] * df['inv_volatility_10d'] * df['smoothed_volume_ema_10'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_10d']) + \
                   (df['momentum_20d'] * df['inv_volatility_30d'] * df['smoothed_volume_ema_20'] *
                    df['rsi_28'] * df['market_sentiment'] * df['avg_volume_20d'])
    
    return alpha_factor
```
