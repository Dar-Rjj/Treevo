```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with an adaptive span based on recent volume changes
    df['smoothed_volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure with an adaptive window
    def rsi(data, window):
        delta = data.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / (loss + 1e-7)
        return 100 - (100 / (1 + rs))
    
    df['rsi_14'] = rsi(df['close'], 14)
    df['rsi_28'] = rsi(df['close'], 28)
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 10-day average trading volume to identify trading volume patterns
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 50-day moving average as a long-term mean reversion signal
    df['ma_50d'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance from the 50-day moving average as a mean reversion factor
    df['dist_from_ma_50d'] = (df['close'] - df['ma_50d']) / df['close']
    
    # Calculate the VWAP (Volume-Weighted Average Price) with an adaptive window
    vwap_window = 20
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=vwap_window).sum() / df['volume'].rolling(window=vwap_window).sum()
    
    # Incorporate additional financial metrics: Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_sum / (negative_money_flow_sum + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, average trading volume, VWAP, and MFI
    df['inv_volatility'] = 1 / (df['volatility_20d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema'] *
                    (df['rsi_14'] + df['rsi_28']) / 2 * df['market_sentiment'] * df['avg_volume_10d'] * 
                    (1 + df['dist_from_ma_50d']) * df['vwap'] * df['mfi_14'])
    
    return alpha_factor
```
