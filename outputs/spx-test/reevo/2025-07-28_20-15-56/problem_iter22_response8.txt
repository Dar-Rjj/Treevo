```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day return as a longer-term momentum indicator
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Calculate the 60-day rolling standard deviation of log returns for a more robust volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_60d'] = df['log_returns'].rolling(window=60).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 30 for better smoothing
    df['smoothed_volume_ema_30'] = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 14-day Relative Strength Index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 50-day average trading volume to identify medium-term volume patterns
    df['avg_volume_50d'] = df['volume'].rolling(window=50).mean()
    
    # Calculate the 200-day moving average as a long-term mean reversion signal
    df['ma_200d'] = df['close'].rolling(window=200).mean()
    
    # Calculate the distance from the 200-day moving average as a mean reversion factor
    df['dist_from_ma_200d'] = (df['close'] - df['ma_200d']) / df['close']
    
    # Calculate the VWAP (Volume-Weighted Average Price) with an adaptive window
    vwap_window = 50
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=vwap_window).sum() / df['volume'].rolling(window=vwap_window).sum()
    
    # Incorporate additional financial metrics: Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_sum / (negative_money_flow_sum + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, average trading volume, VWAP, and MFI
    df['inv_volatility'] = 1 / (df['volatility_60d'] + 1e-7)
    alpha_factor = (df['momentum_20d'] * df['inv_volatility'] * df['smoothed_volume_ema_30'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_50d'] * (1 + df['dist_from_ma_200d']) * df['vwap'] * df['mfi_14'])
    
    return alpha_factor
```
