def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with an adaptive span
    df['smoothed_volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure with a dynamic period
    def rsi(series, periods=14):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=periods).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=periods).mean()
        rs = gain / (loss + 1e-7)
        return 100 - (100 / (1 + rs))
    
    df['rsi_14'] = rsi(df['close'], 14)
    df['rsi_5'] = rsi(df['close'], 5)
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 10-day average trading volume to identify trading volume patterns
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 50-day moving average as a long-term mean reversion signal
    df['ma_50d'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance from the 50-day moving average as a mean reversion factor
    df['dist_from_ma_50d'] = (df['close'] - df['ma_50d']) / df['close']
    
    # Calculate the Money Flow Index (MFI) as a liquidity measure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    df['mfi_14'] = 100 - (100 / (1 + (positive_money_flow_sum / (negative_money_flow_sum + 1e-7))))
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, and average trading volume
    df['inv_volatility'] = 1 / (df['volatility_20d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema'] *
                    (df['rsi_14'] + df['rsi_5']) * df['market_sentiment'] * df['avg_volume_10d'] * (1 + df['dist_from_ma_50d']) * df['mfi_14'])
    
    # Introduce a dynamic adjustment factor based on the ratio of current volume to average volume
    df['volume_ratio'] = df['volume'] / df['avg_volume_10d']
    
    # Adjust the alpha factor with the volume ratio for a more robust liquidity measure
    alpha_factor = alpha_factor * (1 + df['volume_ratio'])
    
    # Handle NaNs by forward filling
    alpha_factor.fillna(method='ffill', inplace=True)
    
    return alpha_factor
