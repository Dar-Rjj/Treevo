def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Average True Range (ATR) over 14 days to measure volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_14 = df['true_range'].rolling(window=14).mean().fillna(0)

    # Momentum component: 5-day and 20-day moving averages of the close price
    sma_5 = df['close'].rolling(window=5).mean().fillna(0)
    sma_20 = df['close'].rolling(window=20).mean().fillna(0)
    momentum = (sma_5 - sma_20) / sma_20

    # Volume-Price Trend: a factor that reflects the relationship between volume and price changes
    vpt = ((df['close'] - df['close'].shift(1)) / df['close'].shift(1)) * df['volume']

    # Average of the last 5 days' Volume-Price Trends to smooth out the values
    vpt_5d_avg = vpt.rolling(window=5).mean().fillna(0)

    # Price Movement: capture the day's price movement as a percentage of the daily range
    daily_range = df['high'] - df['low']
    price_movement = (df['close'] - df['open']).abs() / (daily_range + 1e-7)

    # Calculate the Relative Strength Index (RSI) over 14 days for an additional momentum indicator
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Bollinger Bands: calculate 20-day moving average and 2 standard deviations
    sma_20 = df['close'].rolling(window=20).mean().fillna(0)
    std_20 = df['close'].rolling(window=20).std().fillna(0)
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    bb_value = (df['close'] - sma_20) / (2 * std_20 + 1e-7)

    # Money Flow Index (MFI) over 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (positive_money_flow_sum / (negative_money_flow_sum + 1e-7))))

    # Chaikin Money Flow (CMF) over 20 days
    cmf_numerator = (df['close'] - df['low']) - (df['high'] - df['close'])
    cmf_denominator = df['high'] - df['low']
    cmf_volume = cmf_numerator / (cmf_denominator + 1e-7) * df['volume']
    cmf = cmf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Combining all the components into a single alpha factor
    alpha_factor = (price_movement * momentum * vpt_5d_avg * rsi * bb_value * mfi * cmf) / (atr_14 + 1e-7)
    return alpha_factor
