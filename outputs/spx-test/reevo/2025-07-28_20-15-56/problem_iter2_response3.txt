```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day simple moving averages of the closing price
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the distance (in percentage) of today's close from its 5-day and 20-day SMAs
    distance_from_sma_5 = (df['close'] - sma_5) / sma_5 * 100
    distance_from_sma_20 = (df['close'] - sma_20) / sma_20 * 100
    
    # Measure volume change compared to the 5-day average, as a proxy for abnormal trading activity
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_change = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the money flow index (MFI) as a measure of buying pressure over volume and price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(delta > 0, 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(delta < 0, 0).rolling(window=14).sum()
    money_ratio = positive_money_flow / (negative_money_flow + 1e-7)
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Combine the factors into a single alpha factor
    factor = (distance_from_sma_5 + distance_from_sma_20) / 2 + 0.5 * volume_change + 0.2 * (rsi - 50) + 0.2 * (mfi - 50)
    
    return factor
```
