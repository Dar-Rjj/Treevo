```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day return as a momentum indicator
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Calculate the 5-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_5d'] = df['log_returns'].rolling(window=5).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 3
    df['smoothed_volume_ema_3'] = df['volume'].ewm(span=3, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI and market sentiment
    df['inv_volatility'] = 1 / (df['volatility_5d'] + 1e-7)
    alpha_factor = (df['momentum_3d'] * df['inv_volatility'] * df['smoothed_volume_ema_3'] *
                    df['rsi_14'] * df['market_sentiment'])
    
    # Ensure the alpha factor is not NaN
    alpha_factor = alpha_factor.fillna(0)
    
    return alpha_factor
```
