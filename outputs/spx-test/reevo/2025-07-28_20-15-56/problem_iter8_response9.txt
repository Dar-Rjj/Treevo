```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day return as a shorter term momentum indicator
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Calculate the 5-day rolling standard deviation of log returns for a more responsive volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_5d'] = df['log_returns'].rolling(window=5).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 5
    df['smoothed_volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the money flow index (MFI) to incorporate trading volume
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(delta > 0, 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(delta < 0, 0).rolling(window=14).sum()
    money_ratio = positive_money_flow / (negative_money_flow + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate the inverse of the 5-day volatility
    df['inv_volatility_5d'] = 1 / (df['volatility_5d'] + 1e-7)
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, and MFI
    alpha_factor = (df['momentum_3d'] * df['inv_volatility_5d'] * df['smoothed_volume_ema_5'] *
                    df['rsi_14'] * df['market_sentiment'] * df['mfi_14'])
    
    return alpha_factor
```
