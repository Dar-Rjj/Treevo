```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day return as shorter and longer momentum indicators
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std()
    
    # Smoothed volume using triple Exponential Moving Average (EMA) with spans of 10, 20, and 30
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['smoothed_volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['smoothed_volume_ema_30'] = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the money flow index (MFI) to incorporate trading volume
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(delta > 0, 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(delta < 0, 0).rolling(window=14).sum()
    money_ratio = positive_money_flow / (negative_money_flow + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_20d'] + 1e-7)
    
    # Introduce a new adaptive weighting factor based on the relative strength of the momentum
    df['momentum_strength_5d'] = (df['momentum_5d'] - df['momentum_5d'].rolling(window=20).mean()) / df['momentum_5d'].rolling(window=20).std()
    df['momentum_strength_10d'] = (df['momentum_10d'] - df['momentum_10d'].rolling(window=20).mean()) / df['momentum_10d'].rolling(window=20).std()
    
    # Combine the smoothed volumes
    df['combined_smoothed_volume'] = (df['smoothed_volume_ema_10'] + df['smoothed_volume_ema_20'] + df['smoothed_volume_ema_30']) / 3
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the combined smoothed volume, and adjusted by the RSI, market sentiment, MFI, and momentum strength
    alpha_factor = (df['momentum_5d'] * df['momentum_10d'] * df['inv_volatility'] * 
                    df['combined_smoothed_volume'] * df['rsi_14'] * df['market_sentiment'] * df['mfi_14'] * 
                    df['momentum_strength_5d'] * df['momentum_strength_10d'])
    
    return alpha_factor
```
