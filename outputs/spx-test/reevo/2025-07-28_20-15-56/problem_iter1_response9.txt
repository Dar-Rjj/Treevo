```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close to capture short term momentum
    df['mom_1day'] = df['close'].diff(1)
    
    # Calculate the average true range (ATR) over a 14-day window as a measure of volatility
    df['h-l'] = abs(df['high']-df['low'])
    df['h-pc'] = abs(df['high']-df['close'].shift(1))
    df['l-pc'] = abs(df['low']-df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-pc', 'l-pc']].max(axis=1, skipna=False)
    df['atr14'] = df['tr'].rolling(window=14).mean()
    
    # Combine momentum and normalized ATR to create a new factor
    factor_value = df['mom_1day'] / (df['atr14'] + 1e-7)
    
    return factor_value
```
