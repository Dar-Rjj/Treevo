```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over the last 5 days
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of the close price difference to the average volume
    factor = df['close_diff'] / (df['avg_volume_5'] + 1e-7)
    
    # Add a simple moving average crossover component
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    sma_crossover = df['sma_5'] - df['sma_20']
    
    # Combine the factors into a single alpha factor
    alpha_factor = factor * sma_crossover
    
    return alpha_factor
```
