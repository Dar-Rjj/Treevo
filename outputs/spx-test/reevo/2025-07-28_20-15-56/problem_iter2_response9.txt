```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing and opening price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in volume compared to the previous day
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the 3-day simple moving average of the close price
    sma_3d_close = df['close'].rolling(window=3).mean().fillna(0)
    
    # Calculate the 5-day weighted moving average of the close price
    weights = np.array([0.1, 0.2, 0.3, 0.2, 0.1])
    wma_5d_close = df['close'].rolling(window=5).apply(lambda x: np.sum(weights * x), raw=True).fillna(0)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs)).fillna(0)
    
    # Combine the factors to create the alpha factor
    alpha_factor = (close_open_diff / avg_high_low) * volume_change * (sma_3d_close - wma_5d_close) * rsi
    
    return alpha_factor
```
