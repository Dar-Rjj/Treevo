```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a simple moving average of the closing price over 5 days.
    sma_5 = df['close'].rolling(window=5).mean()
    # Calculate the distance (in percentage) of today's close from its 5-day SMA
    distance_from_sma = (df['close'] - sma_5) / sma_5 * 100
    # Measure volume change compared to the 5-day average, as a proxy for abnormal trading activity
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_change = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    # Generate the factor by combining the relative position of the stock price and unusual volume
    factor = distance_from_sma + 0.5 * volume_change
    return factor
```
