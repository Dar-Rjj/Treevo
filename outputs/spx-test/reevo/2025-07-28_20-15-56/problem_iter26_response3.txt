```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 7-day return as a momentum indicator
    df['momentum_7d'] = df['close'].pct_change(periods=7)
    
    # Calculate the 25-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_25d'] = df['log_returns'].rolling(window=25).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with an adaptive span
    df['smoothed_volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 60-day moving average as a long-term mean reversion signal
    df['ma_60d'] = df['close'].rolling(window=60).mean()
    
    # Calculate the distance from the 60-day moving average as a mean reversion factor
    df['dist_from_ma_60d'] = (df['close'] - df['ma_60d']) / df['close']
    
    # Calculate the Money Flow Index (MFI) as a liquidity measure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    df['mfi_14'] = 100 - (100 / (1 + (positive_money_flow_sum / (negative_money_flow_sum + 1e-7))))
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, and mean reversion
    df['inv_volatility'] = 1 / (df['volatility_25d'] + 1e-7)
    alpha_factor = (df['momentum_7d'] * df['inv_volatility'] * df['smoothed_volume_ema'] *
                    df['rsi_14'] * df['market_sentiment'] * (1 + df['dist_from_ma_60d']) * df['mfi_14'])
    
    return alpha_factor
```
