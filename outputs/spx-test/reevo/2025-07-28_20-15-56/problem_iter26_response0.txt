```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day return as a more stable momentum indicator
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day rolling standard deviation of log returns for a more stable volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with an adaptive span
    df['smoothed_volume_ema'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 20-day average trading volume to identify trading volume patterns
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the 100-day moving average as a long-term mean reversion signal
    df['ma_100d'] = df['close'].rolling(window=100).mean()
    
    # Calculate the distance from the 100-day moving average as a mean reversion factor
    df['dist_from_ma_100d'] = (df['close'] - df['ma_100d']) / df['close']
    
    # Calculate the Money Flow Index (MFI) as a liquidity measure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    df['mfi_14'] = 100 - (100 / (1 + (positive_money_flow_sum / (negative_money_flow_sum + 1e-7))))
    
    # Calculate the Volume Weighted Average Price (VWAP) as a liquidity measure
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, and average trading volume
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    alpha_factor = (df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_20d'] * (1 + df['dist_from_ma_100d']) * df['mfi_14'] * df['vwap'])
    
    return alpha_factor
```
