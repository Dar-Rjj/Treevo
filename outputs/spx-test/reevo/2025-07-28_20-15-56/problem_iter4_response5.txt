```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day return as a longer-term momentum indicator
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Calculate the 60-day rolling standard deviation of log returns for a longer-term volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_60d'] = df['log_returns'].rolling(window=60).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 20
    df['smoothed_volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the Money Flow Index (MFI) to incorporate market sentiment
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow[df['close'] > df['close'].shift(1)]
    negative_money_flow = money_flow[df['close'] < df['close'].shift(1)]
    money_ratio = positive_money_flow.rolling(window=14).sum() / (negative_money_flow.rolling(window=14).sum() + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate the alpha factor as the product of momentum and the inverse of volatility, 
    # weighted by the smoothed volume, adjusted by RSI, and incorporating MFI
    df['inv_volatility'] = 1 / (df['volatility_60d'] + 1e-7)
    alpha_factor = (df['momentum_20d'] * df['inv_volatility'] * df['smoothed_volume_ema_20']) * (df['rsi_14'] / 50) * (df['mfi_14'] / 50)
    
    return alpha_factor
```
