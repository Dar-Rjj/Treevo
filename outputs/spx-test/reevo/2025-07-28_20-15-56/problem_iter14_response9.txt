```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive 5-day return as a momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 10
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 10-day average trading volume to identify trading volume patterns
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the Volume Weighted Average Price (VWAP) for the last 10 days
    df['vwap_10d'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 60-day simple moving average as a longer-term trend indicator
    df['sma_60d'] = df['close'].rolling(window=60).mean()
    
    # Calculate the 20-day skewness of log returns as an advanced statistical measure
    df['skewness_20d'] = df['log_returns'].rolling(window=20).apply(lambda x: x.skew(), raw=True)
    
    # Calculate the 20-day kurtosis of log returns as an advanced statistical measure
    df['kurtosis_20d'] = df['log_returns'].rolling(window=20).apply(lambda x: x.kurt(), raw=True)
    
    # Include a macroeconomic indicator (e.g., interest rate) if available in the DataFrame
    if 'interest_rate' in df.columns:
        df['macro_indicator'] = df['interest_rate']
    else:
        df['macro_indicator'] = 1.0  # Placeholder value if no macroeconomic data is available
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, adjusted by the RSI, market sentiment, VWAP, SMA, skewness, kurtosis, and macroeconomic indicator
    df['inv_volatility'] = 1 / (df['volatility_20d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema_10'] *
                    df['rsi_14'] * df['market_sentiment'] * df['vwap_10d'] * df['sma_60d'] *
                    df['skewness_20d'] * df['kurtosis_20d'] * df['macro_indicator'])
    
    return alpha_factor
```
