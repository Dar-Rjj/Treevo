```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day return as a short-term momentum indicator
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Calculate the 10-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_10d'] = df['log_returns'].rolling(window=10).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 5
    df['smoothed_volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 5-day average trading volume to identify trading volume patterns
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the 26-week high and low for mean reversion
    df['26w_high'] = df['close'].rolling(window=126).max()
    df['26w_low'] = df['close'].rolling(window=126).min()
    df['mean_reversion'] = (df['close'] - df['26w_low']) / (df['26w_high'] - df['26w_low'])
    
    # Calculate the 15-day volume-weighted moving average price
    df['vwap_15d'] = (df['close'] * df['volume']).rolling(window=15).sum() / df['volume'].rolling(window=15).sum()
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_10d'] + 1e-7)
    
    # Calculate the alpha factor as the product of short-term momentum, the inverse of volatility,
    # weighted by the smoothed volume, adjusted by the RSI, market sentiment, average trading volume,
    # mean reversion, and 15-day volume-weighted moving average price
    alpha_factor = (df['momentum_3d'] * df['inv_volatility'] * df['smoothed_volume_ema_5'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_5d'] * df['mean_reversion'] * df['vwap_15d'])
    
    return alpha_factor
```
