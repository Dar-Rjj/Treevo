    df['log_returns'] = np.log(df['close']).diff()

    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    df['volatility_60d'] = df['log_returns'].rolling(window=60).std()

    df['smoothed_volume_ema_12'] = df['volume'].ewm(span=12, adjust=False).mean()
