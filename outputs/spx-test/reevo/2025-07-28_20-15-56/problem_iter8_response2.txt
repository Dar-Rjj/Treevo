```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day return as a longer-term momentum indicator
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day rolling standard deviation of log returns as a more stable volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 15
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(window=14).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    
    # Calculate the alpha factor as the product of longer-term momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI and market sentiment
    alpha_factor = (df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema_15'] *
                    df['rsi_14'] * df['market_sentiment'])
    
    return alpha_factor
```
