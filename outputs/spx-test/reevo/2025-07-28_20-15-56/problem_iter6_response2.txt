def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day returns as momentum indicators
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Calculate the 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 10 and 30
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['smoothed_volume_ema_30'] = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Adaptive window for volatility based on the smoothed volume
    df['adaptive_volatility'] = df['volatility_20d'] * (df['smoothed_volume_ema_10'] / df['smoothed_volume_ema_30'])
    df['inv_adaptive_volatility'] = 1 / (df['adaptive_volatility'] + 1e-7)
    
    # Incorporate adaptive windows for RSI
    df['rsi_10'] = df['close'].rolling(window=10).apply(lambda x: rsi(x, 10), raw=True)
    df['rsi_30'] = df['close'].rolling(window=30).apply(lambda x: rsi(x, 30), raw=True)
    
    # Adaptive RSI factor
    df['adaptive_rsi'] = df['rsi_10'] * (df['smoothed_volume_ema_10'] / df['smoothed_volume_ema_30']) + df['rsi_30'] * (1 - df['smoothed_volume_ema_10'] / df['smoothed_volume_ema_30'])
    
    # Calculate the alpha factor as the product of combined momentum, the inverse of adaptive volatility,
    # weighted by the smoothed volume, and adjusted by the adaptive RSI and market sentiment
    combined_momentum = df['momentum_5d'] * df['momentum_20d']
    alpha_factor = (combined_momentum * df['inv_adaptive_volatility'] * df['smoothed_volume_ema_10'] *
                    df['adaptive_rsi'] * df['market_sentiment'])
    
    return alpha_factor

def rsi(prices, n):
    delta = prices.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=n).mean()
    avg_loss = loss.rolling(window=n).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    return rsi
