```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day returns as momentum indicators
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Calculate the 10-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_10d'] = df['log_returns'].rolling(window=10).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 5, 15, and 30
    df['smoothed_volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    df['smoothed_volume_ema_30'] = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the alpha factor as the product of short-term and long-term momentum, 
    # weighted by the smoothed volume and adjusted by RSI
    df['inv_volatility'] = 1 / (df['volatility_10d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['momentum_20d'] * df['inv_volatility'] * 
                    (df['smoothed_volume_ema_5'] + df['smoothed_volume_ema_15'] + df['smoothed_volume_ema_30']) / 3) * (df['rsi_14'] / 50)
    
    return alpha_factor
```
