```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low as a measure of daily volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate the momentum by comparing the current close to the 5-day moving average of the close
    df['momentum'] = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate the money flow index (MFI) as a measure of buying pressure over volume and price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow[df['close'] > df['close'].shift(1)]
    negative_money_flow = raw_money_flow[df['close'] < df['close'].shift(1)]
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Combine the factors into a single alpha factor
    alpha_factor = df['momentum'] * (mfi - 50) / (daily_volatility + 1e-7)
    
    return alpha_factor
```
