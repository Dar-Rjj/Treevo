```python
def heuristics_v2(df: pd.DataFrame, sector_performance: pd.Series, macroeconomic_indicators: pd.Series) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the adaptive 20-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_20d'] = df['log_returns'].rolling(window=20, min_periods=10).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 10
    df['smoothed_volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14, min_periods=7).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14, min_periods=7).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 10-day average trading volume to identify trading volume patterns
    df['avg_volume_10d'] = df['volume'].rolling(window=10, min_periods=5).mean()
    
    # Incorporate sector performance
    df = df.join(sector_performance, how='left')
    df['sector_performance'] = df['sector_performance'].fillna(method='ffill').fillna(method='bfill')
    
    # Integrate macroeconomic indicators
    df = df.join(macroeconomic_indicators, how='left')
    df['macroeconomic_indicators'] = df['macroeconomic_indicators'].fillna(method='ffill').fillna(method='bfill')
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, average trading volume,
    # sector performance, and macroeconomic indicators
    df['inv_volatility'] = 1 / (df['volatility_20d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema_10'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_10d'] *
                    df['sector_performance'] * df['macroeconomic_indicators'])
    
    return alpha_factor
```
