```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a short-term momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day rolling standard deviation of log returns for a more sensitive volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_10d'] = df['log_returns'].rolling(window=10).std()
    
    # Smoothed volume using Exponential Moving Average (EMA) with a span of 5 for more responsive smoothing
    df['smoothed_volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 14-day Relative Strength Index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 50-day moving average as a medium-term mean reversion signal
    df['ma_50d'] = df['close'].rolling(window=50).mean()
    
    # Calculate the distance from the 50-day moving average as a mean reversion factor
    df['dist_from_ma_50d'] = (df['close'] - df['ma_50d']) / df['close']
    
    # Calculate the 100-day moving average as a longer-term mean reversion signal
    df['ma_100d'] = df['close'].rolling(window=100).mean()
    
    # Calculate the distance from the 100-day moving average as a longer-term mean reversion factor
    df['dist_from_ma_100d'] = (df['close'] - df['ma_100d']) / df['close']
    
    # Calculate the Money Flow Index (MFI) as a liquidity measure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_sum / (negative_money_flow_sum + 1e-7)
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))
    
    # Calculate the alpha factor as the product of momentum, the inverse of volatility,
    # weighted by the smoothed volume, and adjusted by the RSI, market sentiment, and MFI
    df['inv_volatility'] = 1 / (df['volatility_10d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['smoothed_volume_ema_5'] *
                    df['rsi_14'] * df['market_sentiment'] * df['mfi_14'] * (1 + df['dist_from_ma_50d']) *
                    (1 + df['dist_from_ma_100d']))
    
    return alpha_factor
```
