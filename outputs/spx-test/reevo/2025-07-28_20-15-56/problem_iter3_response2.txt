```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day return as a momentum indicator for a longer term trend
    df['momentum_10d'] = df['close'].pct_change(periods=10)

    # Calculate the 30-day rolling standard deviation of log returns as a volatility factor
    # with a focus on a slightly longer term volatility
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()

    # Smoothed volume using Exponential Moving Average (EMA) with a span of 15
    # to better capture medium-term trends in trading activity
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()

    # Calculate the Relative Strength Index (RSI) over a 14-day period to measure market sentiment
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))

    # Calculate the alpha factor as the product of momentum and the inverse of volatility,
    # weighted by the smoothed volume and adjusted by the RSI
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)
    alpha_factor = df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema_15'] * (1 - df['rsi_14d'] / 100)

    return alpha_factor
```
