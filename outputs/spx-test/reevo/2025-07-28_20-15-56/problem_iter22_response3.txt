```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day return as a momentum indicator
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day Exponential Moving Average (EMA) of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d_ema'] = df['log_returns'].ewm(span=30, adjust=False).std()
    
    # Smoothed volume using EMA with a span of 20
    df['smoothed_volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the 20-day average trading volume to identify trading volume patterns
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the 100-day EMA as a long-term mean reversion signal
    df['ema_100d'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the distance from the 100-day EMA as a mean reversion factor
    df['dist_from_ema_100d'] = (df['close'] - df['ema_100d']) / df['close']
    
    # Calculate the inverse of volatility
    df['inv_volatility'] = 1 / (df['volatility_30d_ema'] + 1e-7)
    
    # Combine all factors into an alpha factor
    alpha_factor = (df['momentum_10d'] * df['inv_volatility'] * df['smoothed_volume_ema_20'] *
                    df['rsi_14'] * df['market_sentiment'] * df['avg_volume_20d'] * (1 + df['dist_from_ema_100d']))
    
    # Ensure the alpha factor is not NaN
    alpha_factor = alpha_factor.fillna(0)
    
    return alpha_factor
```
