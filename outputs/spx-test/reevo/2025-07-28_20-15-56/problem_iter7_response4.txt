```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a shorter momentum indicator
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day rolling standard deviation of log returns as a more adaptive volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_10d'] = df['log_returns'].rolling(window=10).std()
    
    # Apply dual EMA smoothing to volume, first with a span of 10, then 5
    df['volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_ema_5'] = df['volume_ema_10'].ewm(span=5, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) with a period of 14 for trend measure
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Integrate market sentiment using the ratio of high and low prices
    df['market_sentiment'] = df['high'] / df['low']
    
    # Calculate the alpha factor incorporating momentum, inverse volatility, smoothed volume,
    # RSI, and market sentiment. Use the dual EMA smoothed volume.
    df['inv_volatility'] = 1 / (df['volatility_10d'] + 1e-7)
    alpha_factor = (df['momentum_5d'] * df['inv_volatility'] * df['volume_ema_5'] *
                    df['rsi_14'] * df['market_sentiment'])
    
    return alpha_factor
```
