```python
def heuristics_v2(df: pd.DataFrame, macroeconomic_df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    import numpy as np
    from sklearn.ensemble import RandomForestRegressor

    # Calculate the 10-day and 30-day returns as momentum indicators for short and long-term trends
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    df['momentum_30d'] = df['close'].pct_change(periods=30)

    # Calculate the 30-day rolling standard deviation of log returns as a volatility factor
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()

    # Smoothed volume using Exponential Moving Average (EMA) with a span of 15
    df['smoothed_volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()

    # Calculate the Relative Strength Index (RSI) over a 14-day period to measure market sentiment
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))

    # Calculate the exponential moving average of the close price with a span of 20
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Inverse of the 30-day volatility
    df['inv_volatility'] = 1 / (df['volatility_30d'] + 1e-7)

    # Merge macroeconomic data with the financial data
    df = df.merge(macroeconomic_df, left_index=True, right_index=True, how='left')

    # Feature selection using Random Forest
    X = df[['momentum_10d', 'momentum_30d', 'inv_volatility', 'smoothed_volume_ema_15', 'rsi_14d', 'ema_20']]
    y = df['close'].pct_change().shift(-1).fillna(0)  # Target is the next day's return
    model = RandomForestRegressor(n_estimators=100, random_state=42)
    model.fit(X, y)
    feature_importances = model.feature_importances_

    # Apply adaptive weighting based on feature importance
    weights = feature_importances / feature_importances.sum()
    alpha_factor = (df['momentum_10d'] * weights[0] + 
                    df['momentum_30d'] * weights[1] + 
                    df['inv_volatility'] * weights[2] + 
                    df['smoothed_volume_ema_15'] * weights[3] + 
                    (1 - df['rsi_14d'] / 100) * weights[4] + 
                    df['ema_20'] * weights[5])

    return alpha_factor
```
