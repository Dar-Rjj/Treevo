```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day rolling mean of close and volume
    close_rolling_mean_5 = df['close'].rolling(window=5).mean()
    close_rolling_mean_20 = df['close'].rolling(window=20).mean()
    volume_rolling_mean_5 = df['volume'].rolling(window=5).mean()
    volume_rolling_mean_20 = df['volume'].rolling(window=20).mean()

    # Momentum factor: difference between today's close and the 5-day moving average of close
    momentum_factor = df['close'] - close_rolling_mean_5

    # Long-term trend factor: difference between today's close and the 20-day moving average of close
    long_term_trend_factor = df['close'] - close_rolling_mean_20

    # Volume change factor: difference between today's volume and the 5-day moving average of volume
    volume_change = df['volume'] - volume_rolling_mean_5

    # Close ratio: today's close price to the 5-day moving average of close prices
    close_ratio = df['close'] / close_rolling_mean_5

    # Volume ratio: today's volume to the 5-day moving average of volume
    volume_ratio = df['volume'] / volume_rolling_mean_5

    # Long-term close ratio: today's close price to the 20-day moving average of close prices
    long_term_close_ratio = df['close'] / close_rolling_mean_20

    # Long-term volume ratio: today's volume to the 20-day moving average of volume
    long_term_volume_ratio = df['volume'] / volume_rolling_mean_20

    # Price range: average of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()

    # ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()

    # Interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * volume_change
    interaction_term_2 = close_ratio * volume_ratio
    interaction_term_3 = price_range * atr
    interaction_term_4 = momentum_factor * close_ratio
    interaction_term_5 = volume_change * volume_ratio
    interaction_term_6 = price_range * close_ratio
    interaction_term_7 = long_term_trend_factor * long_term_close_ratio
    interaction_term_8 = long_term_trend_factor * long_term_volume_ratio

    # Dynamic weights for the factors
    dynamic_weight_1 = 0.2 + 0.1 * (long_term_trend_factor / long_term_close_ratio)
    dynamic_weight_2 = 0.2 + 0.1 * (momentum_factor / close_ratio)
    dynamic_weight_3 = 0.2 + 0.1 * (price_range / atr)
    dynamic_weight_4 = 0.2 + 0.1 * (volume_change / volume_ratio)
    dynamic_weight_5 = 0.2 + 0.1 * (long_term_trend_factor / long_term_volume_ratio)

    # Combine the factors with dynamic weights
    alpha_factor = (
        interaction_term_1 * dynamic_weight_1 +
        interaction_term_2 * dynamic_weight_2 +
        interaction_term_3 * dynamic_weight_3 +
        interaction_term_4 * dynamic_weight_4 +
        interaction_term_5 * dynamic_weight_5 +
        interaction_term_6 * dynamic_weight_1 +
        interaction_term_7 * dynamic_weight_2 +
        interaction_term_8 * dynamic_weight_3
    )

    return alpha_factor.dropna()
```
