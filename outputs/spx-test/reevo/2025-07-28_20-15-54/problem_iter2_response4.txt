```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor as the difference between today's close and the 5-day moving average of close, 
    # normalized by the 5-day rolling standard deviation of close
    momentum_factor = (df['close'] - df['close'].rolling(window=5).mean()) / (df['close'].rolling(window=5).std() + 1e-7)
    
    # Calculate the volume change factor as the difference between today's volume and the 5-day moving average of volume,
    # normalized by the 5-day rolling standard deviation of volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate a relative strength factor as the ratio of the close price to the 5-day moving average of close prices
    relative_strength = df['close'] / df['close'].rolling(window=5).mean()
    
    # Combine the momentum, volume change, and relative strength factors in a heuristic way
    alpha_factor = momentum_factor * volume_change * relative_strength
    
    return alpha_factor
```
