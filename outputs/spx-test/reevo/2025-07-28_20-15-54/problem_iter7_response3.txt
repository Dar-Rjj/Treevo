def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive EMA for close prices with a half-life of 5 days
    alpha = 1 - np.exp(np.log(0.5) / 5)
    adaptive_ema = df['close'].ewm(alpha=alpha, adjust=False).mean()
    
    # Momentum factor based on the difference between today's close and the EMA of close
    momentum_factor = (df['close'] - adaptive_ema) / (df['high'].rolling(window=5).mean() + 1e-7)
    
    # Volume change factor comparing today's volume to the exponential moving average of volume
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()
    volume_change = (df['volume'] - volume_ema) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Close price ratio to the adaptive EMA of close prices
    close_ratio = df['close'] / (adaptive_ema + 1e-7)
    
    # Price range over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Normalized price range by the adaptive EMA of close
    price_range_factor = price_range / (adaptive_ema + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Normalize ATR by the adaptive EMA of close
    atr_factor = atr / (adaptive_ema + 1e-7)
    
    # Calculate the RSI (Relative Strength Index) for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rsi = 100 - (100 / (1 + (gain / (loss + 1e-7))))
    
    # Interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * atr_factor * rsi
    interaction_term_2 = volume_change * close_ratio * rsi
    interaction_term_3 = close_ratio * price_range_factor * rsi
    
    # Combine the factors in a heuristic way
    alpha_factor = interaction_term_1 * interaction_term_2 * interaction_term_3
    
    return alpha_factor.dropna()
