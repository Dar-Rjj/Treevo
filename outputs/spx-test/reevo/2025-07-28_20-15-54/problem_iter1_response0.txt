```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the moving average for 'close' over 5 days window
    df['close_5ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate a smoothed volume over a 10 days period
    df['volume_10ma'] = df['volume'].rolling(window=10).mean()
    
    # Heuristic factor: (Current Close - 5 Day Moving Average of Close) / (10 Day Average Volume + 1)
    factor = (df['close'] - df['close_5ma']) / (df['volume_10ma'] + 1)
    
    return factor
```
