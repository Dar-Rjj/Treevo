```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day rolling standard deviation of the close price to incorporate volatility
    df['close_std_10'] = df['close'].rolling(window=10).std()
    df['close_std_30'] = df['close'].rolling(window=30).std()
    
    # Calculate the 10-day and 30-day average true range (ATR) for robust volatility measure
    df['tr_high_low'] = df['high'] - df['low']
    df['tr_high_close_prev'] = (df['high'] - df['close'].shift()).abs()
    df['tr_low_close_prev'] = (df['low'] - df['close'].shift()).abs()
    df['tr'] = df[['tr_high_low', 'tr_high_close_prev', 'tr_low_close_prev']].max(axis=1)
    df['atr_10'] = df['tr'].rolling(window=10).mean()
    df['atr_30'] = df['tr'].rolling(window=30).mean()
    
    # Calculate the difference between today's close and yesterday's close for momentum
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Factor: normalized short-term momentum (difference from yesterday divided by 10-day std)
    alpha_factor_short = df['close_diff'] / df['close_std_10']
    
    # Calculate the 10-day and 30-day exponential moving average of the close price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Factor: relative EMA (ratio of 10-day EMA to 30-day EMA)
    alpha_factor_ema_ratio = df['ema_10'] / df['ema_30']
    
    # Factor: normalized ATR ratio (10-day ATR divided by 30-day ATR)
    alpha_factor_atr_ratio = df['atr_10'] / df['atr_30']
    
    # Calculate the 100-day and 200-day simple moving average for long-term trend
    df['sma_100'] = df['close'].rolling(window=100).mean()
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # Factor: long-term trend (ratio of 100-day SMA to 200-day SMA)
    alpha_factor_sma_ratio = df['sma_100'] / df['sma_200']
    
    # Calculate the 50-day and 150-day exponential moving average for mid-term trend
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['ema_150'] = df['close'].ewm(span=150, adjust=False).mean()
    
    # Factor: mid-term trend (ratio of 50-day EMA to 150-day EMA)
    alpha_factor_ema_mid_ratio = df['ema_50'] / df['ema_150']
    
    # Combine the factors to create a more robust alpha factor
    alpha_factor_combined = (alpha_factor_short + alpha_factor_ema_ratio + alpha_factor_atr_ratio + 
                             alpha_factor_sma_ratio + alpha_factor_ema_mid_ratio) / 5
    
    return alpha_factor_combined.dropna()
```
