```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and the average of high and low, capturing price inefficiency
    price_diff = (df['close'] - (df['high'] + df['low']) / 2)
    
    # Scale the price difference by volume to incorporate liquidity
    scaled_price_diff = price_diff * df['volume']
    
    # Calculate the exponential moving average of the close price to capture the trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the difference between the close price and the EMA, capturing the deviation from the trend
    trend_deviation = df['close'] - ema_10
    
    # Calculate the true range for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR as a measure of volatility
    volatility_factor = atr * scaled_price_diff
    
    # Combine the scaled price difference, trend deviation, and volatility factor
    factor = scaled_price_diff + trend_deviation + volatility_factor
    
    return factor.dropna()
```
