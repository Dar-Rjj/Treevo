```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the simple momentum factor as the difference between today's close and yesterday's close, 
    # normalized by the average of last 5 days' high prices
    momentum_factor = (df['close'] - df['close'].shift(1)) / (df['high'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the volume change factor comparing today's volume to the 5-day average volume, 
    # normalized by the 5-day standard deviation of volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the 5-day average close price
    close_ratio = df['close'] / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the 5-day average of the range between high and low prices
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Normalize the price range by the 5-day average close price
    price_range_factor = price_range / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility using a 14-day EMA
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Normalize ATR by the 5-day average close price
    atr_factor = atr / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the RSI (Relative Strength Index) for momentum using a 14-day EMA
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rsi = 100 - (100 / (1 + (gain / (loss + 1e-7))))
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * atr_factor * rsi
    interaction_term_2 = volume_change * close_ratio * rsi
    interaction_term_3 = close_ratio * price_range_factor * rsi
    
    # Calculate the 30-day exponentially weighted moving average of the close price
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 30-day exponentially weighted moving average of the volume
    ema_volume_30 = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 30-day exponentially weighted moving average of the high minus low
    ema_high_low_30 = (df['high'] - df['low']).ewm(span=30, adjust=False).mean()
    
    # Incorporate long-term trends
    long_trend_factor = (df['close'] - ema_30) / (ema_high_low_30 + 1e-7)
    
    # Combine the factors in a heuristic way
    alpha_factor = (interaction_term_1 + interaction_term_2 + interaction_term_3 + long_trend_factor) * (rsi / 100)
    
    return alpha_factor.dropna()
```
