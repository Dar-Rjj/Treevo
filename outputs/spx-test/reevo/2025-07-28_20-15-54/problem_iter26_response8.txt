```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by the average of last 5 days' high prices
    momentum_factor = (df['close'] - df['close'].shift(1)) / (df['high'].rolling(window=5).mean() + 1e-7)
    
    # Calculate a volume change factor comparing today's volume to the EMA of last 5 days' volume
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()
    volume_change = (df['volume'] - volume_ema) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the EMA of the last 5 days' close prices
    close_ema = df['close'].ewm(span=5, adjust=False).mean()
    close_ratio = df['close'] / (close_ema + 1e-7)
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Factor: normalized price range
    price_range_factor = price_range / (close_ema + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility using EMA
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Normalize ATR by the EMA close price over the last 5 days
    atr_factor = atr / (close_ema + 1e-7)
    
    # Calculate the relative strength index (RSI) for momentum using EMA
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rsi = 100 - (100 / (1 + (gain / (loss + 1e-7))))
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * atr_factor
    interaction_term_2 = volume_change * close_ratio
    interaction_term_3 = close_ratio * price_range_factor
    interaction_term_4 = rsi * atr_factor
    
    # Adaptive weighting based on the recent performance of the factors with EMA
    adaptive_weight_1 = (interaction_term_1.shift(1) * df['close'].pct_change().shift(-1)).ewm(span=20, adjust=False).mean()
    adaptive_weight_2 = (interaction_term_2.shift(1) * df['close'].pct_change().shift(-1)).ewm(span=20, adjust=False).mean()
    adaptive_weight_3 = (interaction_term_3.shift(1) * df['close'].pct_change().shift(-1)).ewm(span=20, adjust=False).mean()
    adaptive_weight_4 = (interaction_term_4.shift(1) * df['close'].pct_change().shift(-1)).ewm(span=20, adjust=False).mean()
    
    # Combine the factors with adaptive weights
    alpha_factor = (adaptive_weight_1 * interaction_term_1 + 
                    adaptive_weight_2 * interaction_term_2 + 
                    adaptive_weight_3 * interaction_term_3 + 
                    adaptive_weight_4 * interaction_term_4)
    
    # Add a sector-specific indicator (assuming 'sector' is a column in the DataFrame)
    if 'sector' in df.columns:
        sector_mean_close = df.groupby('sector')['close'].transform('mean')
        sector_close_ratio = df['close'] / (sector_mean_close + 1e-7)
        alpha_factor += sector_close_ratio
    
    # Incorporate macroeconomic data (assuming 'macro_data' is a column in the DataFrame)
    if 'macro_data' in df.columns:
        macro_impact = (df['macro_data'] - df['macro_data'].shift(1)) / (df['macro_data'].shift(1) + 1e-7)
        alpha_factor += macro_impact
    
    return alpha_factor.dropna()
```
