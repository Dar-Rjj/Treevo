```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the average of high and low, capturing price inefficiency
    price_diff = (df['close'] - (df['high'] + df['low']) / 2)
    
    # Scale the price difference by volume to incorporate liquidity
    scaled_price_diff = price_diff * df['volume']
    
    # Calculate the 10-day and 20-day exponential moving averages of the close price to capture short- and long-term trends
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the deviations from the short- and long-term trends
    trend_deviation_10 = df['close'] - ema_10
    trend_deviation_20 = df['close'] - ema_20
    
    # Calculate the 5-day and 10-day exponentially weighted standard deviations of the close price to capture short- and long-term volatility
    ewstd_5 = df['close'].ewm(span=5, adjust=False).std()
    ewstd_10 = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day and 20-day exponentially weighted moving averages of volume to capture short- and long-term liquidity trends
    ema_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the deviations from the short- and long-term volume trends
    volume_trend_deviation_10 = df['volume'] - ema_volume_10
    volume_trend_deviation_20 = df['volume'] - ema_volume_20
    
    # Combine the scaled price difference, trend deviations, and volume trend deviations
    # and scale by the exponentially weighted standard deviations for robustness
    factor = (
        (scaled_price_diff + trend_deviation_10 + trend_deviation_20 + 
         volume_trend_deviation_10 + volume_trend_deviation_20) / 
        (ewstd_5 + ewstd_10 + 1e-7)
    )
    
    return factor.dropna()
```
