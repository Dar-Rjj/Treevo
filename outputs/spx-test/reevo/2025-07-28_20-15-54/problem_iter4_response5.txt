```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by the standard deviation of last 5 days' high prices for a volatility-adjusted momentum
    momentum_factor = (df['close'] - df['close'].shift(1)) / (df['high'].rolling(window=5).std() + 1e-7)
    
    # Calculate a volume change factor comparing today's volume to the average of last 10 days' volume, scaling with the standard deviation
    volume_change = (df['volume'] - df['volume'].rolling(window=10).mean()) / (df['volume'].rolling(window=10).std() + 1e-7)
    
    # Incorporate trend information via a moving average comparison: if the short-term MA is above the long-term MA, assign a positive trend score, else negative
    ma_short = df['close'].rolling(window=5).mean()
    ma_long = df['close'].rolling(window=20).mean()
    trend_score = (ma_short > ma_long).astype(int) * 2 - 1  # 1 for uptrend, -1 for downtrend
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors in a heuristic way, incorporating RSI, trend information, and adjusting for volatility
    alpha_factor = (momentum_factor * volume_change * trend_score * rsi) / (df['high'].rolling(window=5).std() + 1e-7)
    
    return alpha_factor
```
