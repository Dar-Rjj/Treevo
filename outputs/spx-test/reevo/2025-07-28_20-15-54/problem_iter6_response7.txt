```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum factor as the difference between today's close and the 5-day moving average of close
    momentum_factor = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day volume change factor as the difference between today's volume and the 5-day moving average of volume
    volume_change = df['volume'] - df['volume'].rolling(window=5).mean()
    
    # Calculate a 5-day relative strength factor as the ratio of the close price to the 5-day moving average of close prices
    relative_strength = df['close'] / df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day range factor as the difference between high and low, normalized by the 5-day rolling standard deviation of close
    range_factor = (df['high'] - df['low']) / (df['close'].rolling(window=5).std() + 1e-7)
    
    # Calculate the 5-day volatility factor as the 5-day rolling standard deviation of the close price
    volatility_factor = df['close'].rolling(window=5).std()
    
    # Combine the factors in a heuristic way to create a more robust alpha factor
    alpha_factor = momentum_factor * volume_change * relative_strength * range_factor * (1 / (volatility_factor + 1e-7))
    
    return alpha_factor.dropna()
```
