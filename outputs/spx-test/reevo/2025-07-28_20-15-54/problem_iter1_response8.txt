```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the opening price, then divide by the average true range over the last 5 days
    atr = (df['high'] - df['low']).rolling(window=5).mean()  # Average True Range
    return (df['close'] - df['open']) / (atr + 1e-7)
```
