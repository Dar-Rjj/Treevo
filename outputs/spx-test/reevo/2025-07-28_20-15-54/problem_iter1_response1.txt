```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices, then scale it by volume
    diff_scaled_by_volume = (df['close'] - df['open']) * df['volume']
    
    # Calculate the exponential moving average of the scaled price difference over a 10-day window
    ema_diff_scaled = diff_scaled_by_volume.ewm(span=10, adjust=False).mean()
    
    # The factor is the ratio of the current day's scaled difference to the 10-day EMA
    factor_values = diff_scaled_by_volume / (ema_diff_scaled + 1e-7)
    
    return factor_values
```
