```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day rolling standard deviation of the close price to incorporate volatility
    df['close_std_10'] = df['close'].rolling(window=10).std()
    df['close_std_30'] = df['close'].rolling(window=30).std()
    
    # Calculate the difference between today's close and yesterday's close for momentum
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Factor: normalized short-term momentum (difference from yesterday divided by 10-day std)
    alpha_factor_short = df['close_diff'] / df['close_std_10']
    
    # Calculate the 10-day and 30-day exponential moving average of the close price
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Factor: relative EMA (ratio of 10-day EMA to 30-day EMA)
    alpha_factor_ema_ratio = df['ema_10'] / df['ema_30']
    
    # Combine the factors to create a more robust alpha factor
    alpha_factor_combined = alpha_factor_short * alpha_factor_ema_ratio
    
    return alpha_factor_combined.dropna()
```
