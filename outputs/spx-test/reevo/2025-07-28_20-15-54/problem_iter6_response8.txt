def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and the average of the last 5 days' close, then divide by the average of last 5 days' high prices
    momentum_factor = (df['close'] - df['close'].rolling(window=5).mean()) / (df['high'].rolling(window=5).mean() + 1e-7)
    
    # Calculate a volume change factor comparing today's volume to the average of last 5 days' volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the average of the last 5 days' close prices
    close_ratio = df['close'] / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ratio of today's volume to the average of the last 5 days' volume
    volume_ratio = df['volume'] / (df['volume'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Factor: normalized price range
    price_range_factor = price_range / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    atr = true_range.rolling(window=5).mean()
    atr_factor = atr / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * volume_change
    interaction_term_2 = close_ratio * volume_ratio
    interaction_term_3 = price_range_factor * atr_factor
    
    # Incorporate longer-term trend information via a moving average comparison: if the short-term MA is above the long-term MA, assign a positive trend score, else negative
    ma_short = df['close'].rolling(window=10).mean()
    ma_long = df['close'].rolling(window=30).mean()
    trend_score = (ma_short > ma_long).astype(int) * 2 - 1  # 1 for uptrend, -1 for downtrend
    
    # Calculate the RSI (Relative Strength Index) for additional momentum information
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the MACD (Moving Average Convergence Divergence) for trend following
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_diff = macd_line - signal_line
    
    # Incorporate seasonal adjustment factors
    df['day_of_week'] = df.index.dayofweek
    seasonal_adjustment = df['day_of_week'].apply(lambda x: 1 if x in [4, 5] else 0)  # Adjust for Fridays and Mondays
    
    # Combine the factors in a heuristic way, incorporating trend information, RSI, and MACD
    alpha_factor = (interaction_term_1 * interaction_term_2 * interaction_term_3) * trend_score * rsi * macd_diff * (1 + seasonal_adjustment)
    
    return alpha_factor.dropna()
