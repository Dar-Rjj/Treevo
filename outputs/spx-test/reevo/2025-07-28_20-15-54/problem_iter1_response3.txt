```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close to capture the daily price movement
    daily_change = df['close'].diff(1)
    
    # Determine if the close is in the upper half of the range (high - low). This captures an upward trend.
    is_upper_half = (df['close'] - df['low']) > 0.5 * (df['high'] - df['low'])

    # Volume change relative to the average volume over the last 5 days, which can indicate increasing or decreasing interest in the stock
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_change = (df['volume'] - avg_volume_5d) / (avg_volume_5d + 1e-7)

    # Final alpha factor: Combine the daily price change, the position of the close price in the day's range, and the volume change
    alpha_factor = daily_change * is_upper_half.astype(int) * volume_change
    
    return alpha_factor
```
