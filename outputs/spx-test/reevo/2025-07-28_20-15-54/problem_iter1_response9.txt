```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low over the last 5 days to capture volatility
    volatility_proxy = (df['high'].rolling(window=5).mean() - df['low'].rolling(window=5).mean()).fillna(0)
    
    # Compute the ratio of close to the average of high and low over the last 5 days, 
    # this can capture whether the stock is closing in the upper or lower end of its range.
    avg_range = (df['high'].rolling(window=5).mean() + df['low'].rolling(window=5).mean()) / 2
    close_to_avg = df['close'] / avg_range
    
    # Factor that combines the volatility proxy and the close-to-average-range information
    factor = (close_to_avg - 1) * volatility_proxy
    
    return factor
```
