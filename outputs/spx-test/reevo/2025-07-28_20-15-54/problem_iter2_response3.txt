```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close to capture momentum
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average true range (ATR) over the last 14 days for robust volatility measure
    df['tr_high_low'] = df['high'] - df['low']
    df['tr_high_close_prev'] = (df['high'] - df['close'].shift()).abs()
    df['tr_low_close_prev'] = (df['low'] - df['close'].shift()).abs()
    df['tr'] = df[['tr_high_low', 'tr_high_close_prev', 'tr_low_close_prev']].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the ratio of close price to the ATR, scaled by the volume to consider liquidity
    factor_volume_scaled = df['close'] / df['atr_14'] * df['volume']
    
    # Incorporate momentum by considering the 5-day moving average of the close price
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['momentum_factor'] = df['close'] - df['ma_5']
    
    # Combine the momentum factor with the volume-scaled ATR factor
    factor = (factor_volume_scaled + df['momentum_factor']) / 2
    
    return factor
```
