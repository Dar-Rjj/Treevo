```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day average true range
    atr = (df[['high', 'low']].rolling(window=5).max() - df[['high', 'low']].rolling(window=5).min()).mean(axis=1)
    
    # Calculate the price momentum as the difference between the current close and the 5-day moving average of the close
    momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate the volume momentum as the difference between the current volume and the 5-day moving average of the volume
    volume_momentum = df['volume'] - df['volume'].rolling(window=5).mean()
    
    # Heuristic factor: (Price Momentum * Volume Momentum) / (Average True Range + 1e-7)
    factor = (momentum * volume_momentum) / (atr + 1e-7)
    
    return factor
```
