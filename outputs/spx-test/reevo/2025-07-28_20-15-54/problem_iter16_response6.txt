def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor as the difference between today's close and the 5-day moving average of close
    momentum_factor = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate the volume change factor comparing today's volume to the 5-day moving average of volume
    volume_change = df['volume'] - df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of today's close price to the 5-day moving average of close prices
    close_ratio = df['close'] / df['close'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the 5-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate dynamic volatility measures
    ewstd_10 = df['close'].ewm(span=10, adjust=False).std()  # 10-day exponentially weighted standard deviation
    ewstd_20 = df['close'].ewm(span=20, adjust=False).std()  # 20-day exponentially weighted standard deviation
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * volume_change
    interaction_term_2 = close_ratio * volume_ratio
    interaction_term_3 = price_range * atr
    interaction_term_4 = momentum_factor * close_ratio
    interaction_term_5 = volume_change * volume_ratio
    interaction_term_6 = price_range * close_ratio
    interaction_term_7 = momentum_factor * ewstd_10
    interaction_term_8 = volume_change * ewstd_20
    interaction_term_9 = close_ratio * ewstd_10
    interaction_term_10 = volume_ratio * ewstd_20
    
    # Combine the factors in a heuristic way
    alpha_factor = (interaction_term_1 + interaction_term_2 + interaction_term_3 + 
                    interaction_term_4 + interaction_term_5 + interaction_term_6 + 
                    interaction_term_7 + interaction_term_8 + interaction_term_9 + interaction_term_10)
    
    return alpha_factor.dropna()
