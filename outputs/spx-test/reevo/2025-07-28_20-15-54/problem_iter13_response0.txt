def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive EMA for momentum
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()
    ema_momentum = ema_short - ema_long
    
    # Calculate the RSI for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rsi = 100 - (100 / (1 + (gain / (loss + 1e-7))))
    
    # Calculate the Money Flow Index (MFI) for momentum and volume
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(delta > 0, 0)
    negative_money_flow = raw_money_flow.where(delta < 0, 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / (negative_money_flow.rolling(window=14).sum() + 1e-7)
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate a volume change factor comparing today's volume to the average of last 5 days' volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the average of the last 5 days' close prices
    close_ratio = df['close'] / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Factor: normalized price range
    price_range_factor = price_range / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Normalize ATR by the average close price over the last 5 days
    atr_factor = atr / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = ema_momentum * rsi * mfi
    interaction_term_2 = volume_change * close_ratio * rsi
    interaction_term_3 = close_ratio * price_range_factor * rsi
    
    # Combine the factors in a heuristic way
    alpha_factor = interaction_term_1 * interaction_term_2 * interaction_term_3
    
    return alpha_factor.dropna()
