```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Calculate the rolling standard deviation of the close price over a 10-day window
    df['close_std_10'] = df['close'].rolling(window=10).std()
    
    # Factor: normalized momentum (difference from yesterday divided by 10-day std)
    alpha_factor = df['close_diff'] / df['close_std_10']
    
    return alpha_factor.dropna()
```
