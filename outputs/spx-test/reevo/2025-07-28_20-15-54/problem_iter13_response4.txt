def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive EMA for close prices
    fast_ema = df['close'].ewm(span=12, adjust=False).mean()
    slow_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = fast_ema - slow_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Momentum factor using MACD
    momentum_factor = macd_line - signal_line
    
    # Calculate the RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rsi = 100 - (100 / (1 + (gain / (loss + 1e-7))))
    
    # Calculate the MFI (Money Flow Index)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Volume change factor comparing today's volume to the average of the last 5 days' volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Close price ratio factor
    close_ratio = df['close'] / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Price range factor
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    price_range_factor = price_range / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    atr_factor = atr / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Interaction terms
    interaction_term_1 = momentum_factor * atr_factor * rsi
    interaction_term_2 = volume_change * close_ratio * rsi
    interaction_term_3 = close_ratio * price_range_factor * rsi
    interaction_term_4 = mfi * rsi * close_ratio
    
    # Combine the factors in a heuristic way
    alpha_factor = interaction_term_1 * interaction_term_2 * interaction_term_3 * interaction_term_4
    
    return alpha_factor.dropna()
