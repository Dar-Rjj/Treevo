def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the average of high and low
    price_diff = (df['close'] - (df['high'] + df['low']) / 2)
    
    # Scale the price difference by volume
    scaled_price_diff = price_diff * df['volume']
    
    return scaled_price_diff
