```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor as the difference between today's close and the 5-day moving average of closes
    momentum_factor = (df['close'] - df['close'].rolling(window=5).mean()) / (df['close'].rolling(window=5).std() + 1e-7)
    
    # Calculate a volume change factor comparing today's volume to the 5-day moving average of volumes
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the volatility factor as the standard deviation of the daily returns over the last 5 days
    daily_returns = df['close'].pct_change()
    volatility_factor = daily_returns.rolling(window=5).std()
    
    # Combine the momentum, volume change, and volatility factors in a heuristic way
    alpha_factor = momentum_factor * volume_change / (volatility_factor + 1e-7)
    
    return alpha_factor
```
