```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day momentum factor as the difference between today's close and the 3-day moving average of close
    momentum_factor = df['close'] - df['close'].rolling(window=3).mean()
    
    # Calculate the 3-day volume change factor as the difference between today's volume and the 3-day moving average of volume
    volume_change = df['volume'] - df['volume'].rolling(window=3).mean()
    
    # Calculate a 3-day relative strength factor as the ratio of the close price to the 3-day moving average of close prices
    relative_strength = df['close'] / df['close'].rolling(window=3).mean()
    
    # Calculate the 3-day range factor as the difference between high and low, normalized by the 3-day rolling standard deviation of close
    range_factor = (df['high'] - df['low']) / (df['close'].rolling(window=3).std() + 1e-7)
    
    # Combine the momentum, volume change, relative strength, and range factors in a heuristic way
    alpha_factor = momentum_factor * volume_change * relative_strength * range_factor
    
    return alpha_factor
```
