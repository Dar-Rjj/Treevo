```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by the average of last 5 days' high prices
    momentum_factor = (df['close'] - df['close'].shift(1)) / (df['high'].rolling(window=5).mean() + 1e-7)
    
    # Calculate a volume change factor comparing today's volume to the average of last 5 days' volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the average of the last 5 days' close prices
    close_ratio = df['close'] / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ratio of today's volume to the average of the last 5 days' volume
    volume_ratio = df['volume'] / (df['volume'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Factor: normalized price range
    price_range_factor = price_range / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Normalize ATR by the average close price over the last 5 days
    atr_factor = atr / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Relative strength compared to the 5-day mean
    relative_strength = (df['close'] - df['close'].rolling(window=5).mean()) / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * atr_factor
    interaction_term_2 = volume_change * close_ratio
    interaction_term_3 = volume_ratio * price_range_factor
    interaction_term_4 = relative_strength * close_ratio
    
    # Combine the factors in a heuristic way
    alpha_factor = (interaction_term_1 * interaction_term_2 * interaction_term_3 * interaction_term_4).dropna()
    
    return alpha_factor
```
