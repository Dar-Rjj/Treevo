def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using an adaptive window for the mean and standard deviation
    momentum_factor = (df['close'] - df['close'].shift(1)) / (df['high'].rolling(window=5).mean().ewm(span=5, adjust=False).mean() + 1e-7)
    
    # Calculate a volume change factor comparing today's volume to the exponential moving average of last 5 days' volume
    volume_change = (df['volume'] - df['volume'].ewm(span=5, adjust=False).mean()) / (df['volume'].rolling(window=5).std().ewm(span=5, adjust=False).mean() + 1e-7)
    
    # Calculate the ratio of today's close price to the exponential moving average of the last 5 days' close prices
    close_ratio = df['close'] / (df['close'].ewm(span=5, adjust=False).mean() + 1e-7)
    
    # Calculate the ratio of today's volume to the exponential moving average of the last 5 days' volume
    volume_ratio = df['volume'] / (df['volume'].ewm(span=5, adjust=False).mean() + 1e-7)
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Factor: normalized price range
    price_range_factor = price_range / (df['close'].ewm(span=5, adjust=False).mean() + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Normalize ATR by the exponential moving average close price over the last 5 days
    atr_factor = atr / (df['close'].ewm(span=5, adjust=False).mean() + 1e-7)
    
    # Integrate additional interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * atr_factor
    interaction_term_2 = volume_change * close_ratio
    interaction_term_3 = volume_ratio * price_range_factor
    interaction_term_4 = momentum_factor * close_ratio
    interaction_term_5 = volume_change * price_range_factor
    interaction_term_6 = atr_factor * close_ratio
    
    # Combine the factors in a heuristic way
    alpha_factor = (interaction_term_1 * interaction_term_2 * interaction_term_3 * 
                    interaction_term_4 * interaction_term_5 * interaction_term_6).ewm(span=5, adjust=False).mean()
    
    return alpha_factor.dropna()
