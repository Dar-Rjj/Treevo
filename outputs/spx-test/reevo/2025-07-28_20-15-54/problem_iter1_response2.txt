```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close prices
    price_diff = df['close'] - df['close'].shift(1)
    
    # Calculate the volume weighted by the absolute change in price
    volume_weighted_change = (df['volume'] * price_diff.abs()) / df['volume'].rolling(window=5).mean()
    
    # Calculate the smoothed high-minus-low ratio for the last 5 days
    smooth_high_low = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()) / (df['close'].rolling(window=5).std() + 1e-7)
    
    # Combine the volume weighted change and the smoothed high-minus-low ratio
    factor_values = volume_weighted_change * smooth_high_low
    
    return factor_values
```
