```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using an adaptive window based on the 5-day moving average of closes
    adaptive_window = df['close'].rolling(window=5).std() * 10
    adaptive_momentum = (df['close'] - df['close'].rolling(window=adaptive_window.astype(int)).mean()) / (df['close'].rolling(window=5).std() + 1e-7)
    
    # Calculate a volume change factor comparing today's volume to an adaptive moving average of volumes
    adaptive_volume_window = df['volume'].rolling(window=5).std() * 10
    volume_change = (df['volume'] - df['volume'].rolling(window=adaptive_volume_window.astype(int)).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the volatility factor as the standard deviation of the daily returns over an adaptive window
    daily_returns = df['close'].pct_change()
    adaptive_volatility_window = daily_returns.rolling(window=5).std() * 10
    volatility_factor = daily_returns.rolling(window=adaptive_volatility_window.astype(int)).std()
    
    # Incorporate high-low range as a measure of daily price movement
    high_low_range = (df['high'] - df['low']) / df['close']
    
    # Combine the adaptive momentum, volume change, volatility, and high-low range factors in a heuristic way
    alpha_factor = adaptive_momentum * volume_change / (volatility_factor + 1e-7) * high_low_range
    
    return alpha_factor.dropna()
```
