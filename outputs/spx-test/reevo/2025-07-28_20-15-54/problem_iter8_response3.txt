```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the average of high and low, capturing price inefficiency
    price_diff = (df['close'] - (df['high'] + df['low']) / 2)
    
    # Scale the price difference by volume to incorporate liquidity
    scaled_price_diff = price_diff * df['volume']
    
    # Calculate the exponential moving average of the close price to capture the trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the difference between the close price and the EMA, capturing the deviation from the trend
    trend_deviation = df['close'] - ema_10
    
    # Calculate the 3-day exponential moving average of the close price for a shorter-term trend
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the deviation from the short-term trend
    short_trend_deviation = df['close'] - ema_3
    
    # Calculate the 5-day exponentially weighted standard deviation of the close price to capture volatility
    ewstd_5 = df['close'].ewm(span=5, adjust=False).std()
    
    # Incorporate the relative strength as the ratio of the close price to the 10-day EMA
    relative_strength = df['close'] / ema_10
    
    # Combine the scaled price difference, the trend deviation, the short-term trend deviation, and the relative strength
    # and scale by the exponentially weighted standard deviation to normalize the factor
    factor = (scaled_price_diff + trend_deviation + short_trend_deviation + relative_strength) / (ewstd_5 + 1e-7)
    
    return factor.dropna()
```
