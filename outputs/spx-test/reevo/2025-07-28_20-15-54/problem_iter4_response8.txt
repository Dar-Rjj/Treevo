```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor as the difference between today's close and the average of the last 5 days' close, normalized by the standard deviation of the last 5 days' close
    momentum_factor = (df['close'] - df['close'].rolling(window=5).mean()) / (df['close'].rolling(window=5).std() + 1e-7)
    
    # Calculate the volume change factor comparing today's volume to the average of last 5 days' volume, normalized by the standard deviation of the last 5 days' volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the average of the last 5 days' close prices
    close_ratio = df['close'] / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ratio of today's volume to the average of the last 5 days' volume
    volume_ratio = df['volume'] / (df['volume'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the range of high and low prices over the last 5 days, normalized by the average close price
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    price_range_factor = price_range / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility, normalized by the average close price
    atr = (df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)).rolling(window=5).mean()
    atr_factor = atr / (df['close'].rolling(window=5).mean() + 1e-7)
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * volume_change
    interaction_term_2 = close_ratio * volume_ratio
    interaction_term_3 = price_range_factor * atr_factor
    
    # Combine the factors in a heuristic way
    alpha_factor = interaction_term_1 * interaction_term_2 * interaction_term_3
    
    return alpha_factor.dropna()
```
