def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive EMA for close price using a 14-day period
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the momentum factor as the difference between today's close and the EMA of close
    momentum_factor = (df['close'] - ema_close) / (df['high'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the volume change factor comparing today's volume to the average of last 5 days' volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the EMA of close prices
    close_ratio = df['close'] / (ema_close + 1e-7)
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Factor: normalized price range
    price_range_factor = price_range / (ema_close + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Normalize ATR by the EMA of close price
    atr_factor = atr / (ema_close + 1e-7)
    
    # Calculate the relative strength index (RSI) for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rsi = 100 - (100 / (1 + (gain / (loss + 1e-7))))
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * atr_factor * rsi
    interaction_term_2 = volume_change * close_ratio * rsi
    interaction_term_3 = close_ratio * price_range_factor * rsi
    
    # Combine the factors in a heuristic way
    alpha_factor = interaction_term_1 * interaction_term_2 * interaction_term_3
    
    return alpha_factor.dropna()
