def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the difference between today's close and 5 days ago close, normalized by the 5-day ATR
    momentum_factor = (df['close'] - df['close'].shift(5)) / talib.ATR(df['high'], df['low'], df['close'], timeperiod=5).fillna(1e-7)
    
    # Calculate a volume change factor comparing today's volume to the average of last 5 days' volume
    volume_change = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Calculate the ratio of today's close price to the 5-day exponential moving average of close prices
    close_ratio = df['close'] / df['close'].ewm(span=5).mean()
    
    # Calculate the ratio of today's volume to the 5-day exponential moving average of volume
    volume_ratio = df['volume'] / df['volume'].ewm(span=5).mean()
    
    # Calculate the range of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Factor: normalized price range by the 5-day ATR
    price_range_factor = price_range / talib.ATR(df['high'], df['low'], df['close'], timeperiod=5).fillna(1e-7)
    
    # Integrate interaction terms for more comprehensive signals
    interaction_term_1 = momentum_factor * volume_change
    interaction_term_2 = close_ratio * volume_ratio
    interaction_term_3 = price_range_factor * volume_change
    
    # Combine the factors in a heuristic way
    alpha_factor = interaction_term_1 * interaction_term_2 * interaction_term_3
    
    return alpha_factor.dropna()
