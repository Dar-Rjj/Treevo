```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return (price momentum)
    price_momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the average true range over 5 days (volatility measure)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    avg_true_range = true_range.rolling(window=5).mean()
    
    # Calculate the volume-weighted change in close price
    price_diff = df['close'] - df['close'].shift(1)
    volume_weighted_change = (df['volume'] * price_diff.abs()) / df['volume'].rolling(window=5).mean()
    
    # Smoothed high-minus-low ratio over 5 days
    smooth_high_low = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()) / (df['close'].rolling(window=5).std() + 1e-7)
    
    # Combine the price momentum, volume-weighted change, and smoothed high-minus-low ratio
    factor_values = price_momentum * volume_weighted_change * smooth_high_low / (avg_true_range + 1e-7)
    
    return factor_values
```
