```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive EMA for close price with a span of 5 days
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Momentum factor: difference between today's close and EMA of close, normalized by the EMA of high prices
    momentum_factor = (df['close'] - ema_close) / (df['high'].ewm(span=5, adjust=False).mean() + 1e-7)
    
    # Volume change factor: difference between today's volume and EMA of volume, normalized by the standard deviation of volume
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()
    volume_std = df['volume'].ewm(span=5, adjust=False).std()
    volume_change = (df['volume'] - volume_ema) / (volume_std + 1e-7)
    
    # Close ratio: today's close price to the EMA of the last 5 days' close prices
    close_ratio = df['close'] / (ema_close + 1e-7)
    
    # Price range: average of high and low prices over the last 5 days
    price_range = (df['high'] - df['low']).ewm(span=5, adjust=False).mean()
    price_range_factor = price_range / (ema_close + 1e-7)
    
    # Calculate the ATR (Average True Range) for robust volatility
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    atr_factor = atr / (ema_close + 1e-7)
    
    # Calculate the relative strength index (RSI) for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rsi = 100 - (100 / (1 + (gain / (loss + 1e-7))))
    
    # Combine the factors in a simplified way
    alpha_factor = momentum_factor * volume_change * close_ratio * price_range_factor * atr_factor * (rsi / 100)
    
    return alpha_factor.dropna()
```
