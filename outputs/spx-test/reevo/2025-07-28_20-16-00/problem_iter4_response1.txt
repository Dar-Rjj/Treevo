```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate a simple moving average for the close price over 10 days as a basic trend indicator
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Determine if today's close is above the 10-day SMA, indicating an upward momentum
    momentum = (df['close'] > sma_10).astype(int)
    
    # Calculate the volume-weighted return of the last 5 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last 14 days
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1], x[2]) - min(x[0], x[1], x[2]), axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the 20-day EMA (Exponential Moving Average) to capture medium-term trends
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 50-day EMA to capture longer-term trends
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the MACD (Moving Average Convergence Divergence)
    macd = ema_20 - ema_50
    
    # Calculate the MACD Signal Line (9-day EMA of the MACD)
    macd_signal = macd.ewm(span=9, adjust=False).mean()
    
    # Calculate the MACD Histogram
    macd_hist = macd - macd_signal
    
    # Create a dynamic weight for each component
    volatility_weight = (1 / (daily_volatility + 1e-7))
    momentum_weight = momentum
    rsi_weight = (rsi / 100)
    atr_weight = (1 / (atr + 1e-7))
    macd_weight = (macd_hist / (macd_hist.abs().rolling(window=14).max() + 1e-7))
    
    # Combine the components with dynamic weighting
    factor = (volatility_weight + momentum_weight + rsi_weight + atr_weight + macd_weight) * weighted_return
    
    return factor
```
