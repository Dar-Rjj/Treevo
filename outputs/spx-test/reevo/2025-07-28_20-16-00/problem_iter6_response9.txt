```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate a simple moving average for the close price over 10 days as a more stable trend indicator
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Determine if today's close is above the 10-day SMA, indicating an upward momentum
    momentum = (df['close'] > sma_10).astype(int)
    
    # Calculate the volume-weighted return of the last 5 days for a smoother signal
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last 14 days
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Calculate the MACD (Moving Average Convergence Divergence) to capture market sentiment
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.rolling(window=9).mean()
    macd_diff = macd - signal
    
    # Create a factor combining volatility (as inverse, higher volatility might lead to reversion), momentum, RSI, ATR, weighted return, volume ratio, and MACD
    factor = (1 / (daily_volatility + 1e-7)) * momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return * volume_ratio * macd_diff
    
    return factor
```
