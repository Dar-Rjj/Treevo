```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily volatility as the difference between high and low
    daily_volatility = df['high'] - df['low']
    
    # Calculate a 5-day weighted moving average (WMA) for the close price
    weights = np.arange(1, 6)
    wma_5 = df['close'].rolling(window=5).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
    
    # Determine if today's close is above the 5-day WMA, indicating an upward momentum
    momentum = (df['close'] > wma_5).astype(int)
    
    # Calculate the relative strength (RS) as the ratio of the average gain to the average loss over the last 5 days
    gain = df['close'].diff().clip(lower=0)
    loss = -df['close'].diff().clip(upper=0)
    avg_gain = gain.rolling(window=5).mean()
    avg_loss = loss.rolling(window=5).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    
    # Create a factor combining volatility (as inverse), momentum, and relative strength
    factor = (1 / (daily_volatility + 1e-7)) * momentum * rs
    
    return factor
```
