```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Adaptive Exponential Moving Averages (EMA) for the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Determine if today's close is above the 5-day EMA, indicating an upward momentum
    momentum = (df['close'] > ema_5).astype(int)
    
    # Volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # RSI (Relative Strength Index) using EMA for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # ATR (Average True Range) for the last 10 days using EMA
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.ewm(span=10, adjust=False).mean()
    
    # Dynamic weighting based on the ratio of the current day's volume to the 10-day average volume
    avg_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_ratio = df['volume'] / avg_volume_10
    
    # Macro sentiment indicator (simplified as a 20-day moving average of the close price)
    macro_sentiment = df['close'].rolling(window=20).mean()
    
    # Sector-specific indicator (simplified as a 30-day moving average of the close price)
    sector_indicator = df['close'].rolling(window=30).mean()
    
    # Longer-term market sentiment indicator (simplified as a 50-day moving average of the close price)
    long_term_sentiment = df['close'].rolling(window=50).mean()
    
    # Create a factor combining volatility (as inverse), momentum, RSI, ATR, volume ratio, and macro sentiment
    factor = (1 / (daily_volatility + 1e-7)) * momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return * volume_ratio * (macro_sentiment / df['close'])
    
    # Incorporate the sector-specific and longer-term market sentiment into the factor
    factor *= (sector_indicator / df['close']) * (long_term_sentiment / df['close'])
    
    # Introduce a trend-following component using the 5-day and 10-day EMAs
    trend_following = (ema_5 > ema_10).astype(int)
    factor *= trend_following
    
    return factor
```
