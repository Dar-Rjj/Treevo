```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices, then adjust it based on the average true range and volume
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Calculate the factor value
    price_diff = (df['close'] - df['open']) / (df['ATR_14'] + 1e-7)
    
    # Incorporate the volume into the factor
    volume_factor = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the price difference and volume factor
    factor_value = price_diff * volume_factor
    
    return factor_value
```
