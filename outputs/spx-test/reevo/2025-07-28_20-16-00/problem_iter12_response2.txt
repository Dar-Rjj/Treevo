```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive ATR over 14 days, using a smoothing factor
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.rolling(window=14).mean()
    adaptive_atr = (atr.shift(1) * 13 + true_range) / 14

    # Calculate the multi-period momentum by comparing the current close to the 5-day and 20-day SMAs
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    short_momentum = (df['close'] > sma_5).astype(int)
    long_momentum = (df['close'] > sma_20).astype(int)

    # Determine volume shocks by calculating the ratio of today's volume to the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_shock = (df['volume'] / avg_volume_20).clip(lower=0.5, upper=2.0)

    # Calculate the RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()

    # Incorporate mean reversion using Bollinger Bands
    bollinger_mid = df['close'].rolling(window=20).mean()
    bollinger_std = df['close'].rolling(window=20).std()
    bollinger_upper = bollinger_mid + 2 * bollinger_std
    bollinger_lower = bollinger_mid - 2 * bollinger_std
    mean_reversion = (df['close'] - bollinger_mid) / (2 * bollinger_std)

    # Add another technical indicator: Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.rolling(window=9).mean()
    macd_histogram = macd_line - signal_line

    # Calculate On-Balance Volume (OBV) as an additional inter-market relationship
    obv = (df['close'].diff().gt(0) * df['volume']).cumsum()

    # Calculate the percentage change in OBV
    obv_change = obv.pct_change().fillna(0)

    # Combine the factors into a single alpha factor
    factor = (1 / (adaptive_atr + 1e-7)) * (short_momentum + long_momentum) * (rsi / 100) * weighted_return * volume_shock * (1 - abs(mean_reversion)) * macd_histogram * obv_change

    return factor
```
