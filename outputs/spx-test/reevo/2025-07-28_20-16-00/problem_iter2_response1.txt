def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change in closing prices to capture recent price action
    close_change = df['close'].pct_change()
    
    # Calculate the average true range (ATR) as a measure of volatility
    avg_true_range = ((df['high'] - df['low']) + (df['high'] - df['close'].shift(1)).abs() + (df['close'].shift(1) - df['low']).abs()) / 3
    
    # Calculate a weighted momentum factor, giving more weight to very recent changes
    weighted_momentum = (close_change.rolling(window=3).mean() * 3.0 + close_change.rolling(window=5).mean() * 2.0 + close_change.rolling(window=10).mean()) / 6.0
    
    # Incorporate volume by calculating the percentage change in volume to reflect market activity
    volume_change = df['volume'].pct_change()
    
    # Generate the alpha factor, with an interaction between price momentum, volatility, and volume
    alpha_factor = weighted_momentum * (avg_true_range.mean() / avg_true_range) * volume_change
    return alpha_factor
