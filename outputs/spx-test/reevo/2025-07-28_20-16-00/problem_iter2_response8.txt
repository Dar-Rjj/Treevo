```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rate of change in closing prices over a 10 day period
    close_change_10d = df['close'].pct_change(periods=10)
    
    # Calculate the true range (as the max of high - low, high - previous close, previous close - low)
    true_range = (df['high'] - df['low']).combine((df['high'] - df['close'].shift(1)).abs(), max).combine((df['close'].shift(1) - df['low']).abs(), max)
    
    # Calculate the average true range (ATR) over a 14 day window
    avg_true_range_14d = true_range.rolling(window=14).mean()
    
    # Compute a volatility-adjusted momentum by normalizing the 10-day price change with ATR
    vol_adjusted_momentum = close_change_10d / (avg_true_range_14d + 1e-7)
    
    # Calculate the percentage change in volume over a 5 day period to capture short-term volume trends
    volume_change_5d = df['volume'].pct_change(periods=5)
    
    # Generate a factor for the interaction between volatility-adjusted momentum and volume change
    alpha_factor = vol_adjusted_momentum * volume_change_5d
    
    return alpha_factor
```
