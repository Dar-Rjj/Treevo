```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price of today and the opening price of yesterday
    price_diff = df['close'] - df['open'].shift(1)
    
    # Calculate the average trading volume over the last 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day simple moving average of the close price
    sma_close_5d = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day simple moving average of the high price
    sma_high_5d = df['high'].rolling(window=5).mean()
    
    # Calculate the 5-day simple moving average of the low price
    sma_low_5d = df['low'].rolling(window=5).mean()
    
    # Calculate the 10-day exponential moving average of the close price
    ema_close_10d = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponential moving average of the high price
    ema_high_10d = df['high'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponential moving average of the low price
    ema_low_10d = df['low'].ewm(span=10, adjust=False).mean()
    
    # Alpha factor: weighted difference by the average volume, adjusted by the ratio of the current close to the 5-day SMA of the close
    alpha_factor = (price_diff / (avg_volume + 1e-7)) * (df['close'] / sma_close_5d)
    
    # Further adjust the alpha factor by considering the high and low prices relative to their 5-day SMAs
    alpha_factor *= (sma_high_5d / sma_low_5d)
    
    # Incorporate the 10-day EMA of high and low prices
    alpha_factor *= (ema_high_10d / ema_low_10d)
    
    return alpha_factor
```
