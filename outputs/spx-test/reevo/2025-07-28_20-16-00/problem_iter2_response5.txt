```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day price momentum
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day volatility (standard deviation of daily returns)
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=10).std()
    
    # Calculate the liquidity factor as the average trading volume over the last 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day relative strength index (RSI) as a technical indicator
    gains = daily_returns.where(daily_returns > 0, 0)
    losses = -daily_returns.where(daily_returns < 0, 0)
    avg_gain = gains.rolling(window=5).mean()
    avg_loss = losses.rolling(window=5).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Alpha factor: combine momentum, inverse volatility, and RSI, weighted by liquidity
    alpha_factor = (momentum * (1 / (volatility + 1e-7)) * rsi) * (1 / (avg_volume + 1e-7))
    
    return alpha_factor
```
