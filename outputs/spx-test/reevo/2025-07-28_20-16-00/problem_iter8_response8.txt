def heuristics_v2(df: pd.DataFrame, macroeconomic_data: pd.DataFrame, sentiment_data: pd.DataFrame) -> pd.Series:
    # Calculate the true range, refining the calculation to include the previous day's close
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    prev_close = df['close'].shift(1)
    true_range = (df['high'] - df['low']).combine(prev_close - df['low'], max).combine(prev_close - df['high'], max)
    
    # Calculate the ATR (Average True Range) for the last 14 days
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate a simple moving average for the close price over 5 days as a basic trend indicator
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Determine if today's close is above the 5-day SMA, indicating an upward momentum
    momentum = (df['close'] > sma_5).astype(int)
    
    # Calculate the volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Create a factor combining volatility (as inverse, higher volatility might lead to reversion), momentum, RSI, ATR, and volume ratio
    factor = (1 / (daily_volatility + 1e-7)) * momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return * volume_ratio
    
    # Introduce a regime-based adjustment factor
    regime_factor = (df['close'].pct_change().rolling(window=14).mean() > 0).astype(int)
    
    # Incorporate macroeconomic indicators using a linear combination
    macroeconomic_factor = (macroeconomic_data['GDP_growth'] + macroeconomic_data['inflation_rate']) / 2
    
    # Use machine learning for dynamic weighting
    from sklearn.linear_model import LinearRegression
    model = LinearRegression()
    X = df[['close', 'volume', 'open', 'high', 'low']].join(macroconomic_data).dropna()
    y = df['close'].pct_change().shift(-1).fillna(0)[X.index]
    model.fit(X, y)
    dynamic_weighting = pd.Series(model.predict(X), index=X.index)
    
    # Incorporate market sentiment data
    sentiment_factor = (sentiment_data['positive_sentiment'] - sentiment_data['negative_sentiment']) / sentiment_data['total_sentiment']
    
    # Final factor incorporating the regime-based adjustment, macroeconomic factors, dynamic weighting, and sentiment data
    final_factor = factor * regime_factor * macroeconomic_factor * dynamic_weighting * sentiment_factor
    
    return final_factor
