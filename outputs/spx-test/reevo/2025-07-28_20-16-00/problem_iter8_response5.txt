```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change in closing prices
    close_change = df['close'].pct_change()
    
    # Calculate the true range (high - low, high - previous close, or previous close - low)
    true_range = (df['high'] - df['low']).abs().fillna(0)
    prev_close = df['close'].shift(1)
    true_range = true_range.where(true_range > (df['high'] - prev_close).abs(), (df['high'] - prev_close).abs())
    true_range = true_range.where(true_range > (prev_close - df['low']).abs(), (prev_close - df['low']).abs())
    
    # Calculate the average true range over the last 14 days for more stability
    avg_true_range = true_range.rolling(window=14).mean()
    
    # Calculate a weighted momentum factor, giving more weight to recent changes
    weighted_momentum = (close_change.rolling(window=5).mean() * 3.0 + 
                         close_change.rolling(window=10).mean() * 2.0 + 
                         close_change.rolling(window=20).mean()) / 6.0
    
    # Incorporate volume by calculating the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the standard deviation of returns as a measure of volatility
    volatility = close_change.rolling(window=20).std()
    
    # Calculate the relative strength index (RSI) for the closing prices
    gain = df['close'].diff().clip(lower=0).rolling(window=14).mean()
    loss = df['close'].diff().clip(upper=0).abs().rolling(window=14).mean()
    rsi = 100 - (100 / (1 + (gain / loss)))
    
    # Generate the alpha factor, with an interaction between price momentum, volatility, and volume, and RSI
    alpha_factor = (weighted_momentum * (avg_true_range.mean() / avg_true_range) * volume_change * (volatility.mean() / volatility)) * (100 - rsi)
    
    return alpha_factor
```
