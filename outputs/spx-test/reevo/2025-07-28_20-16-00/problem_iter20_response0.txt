```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate an exponentially weighted moving average for the close price to emphasize recent prices
    ewma_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Determine if today's close is above the 5-day EWMA, indicating an upward momentum
    momentum = (df['close'] > ewma_5).astype(int)
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days using exponential weighting
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine key trends: inverse of daily volatility, momentum, and RSI
    factor = (1 / (daily_volatility + 1e-7)) * momentum * (rsi / 100)
    
    return factor
```
