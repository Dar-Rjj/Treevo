```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    daily_return = df['close'].diff()
    # Calculate the average true range over a 14-day period for a volatility factor
    atr = ta.volatility.AverageTrueRange(df['high'], df['low'], df['close'], window=14).average_true_range()
    # Use the ratio of the daily return to the average true range as a measure of recent move strength
    move_strength = (daily_return / atr).shift(1)
    # Consider volume increase from the previous day as a signal of strong activity
    volume_increase = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    # Combine both factors into one, with a focus on days where the stock had a strong move and high relative volume
    factor = move_strength * (1 + volume_increase)
    return factor
```
