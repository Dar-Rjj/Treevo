```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive ATR over 14 days, with a smoothing factor to adapt to recent volatility changes
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr_14 = true_range.ewm(span=14, adjust=False).mean()
    
    # Dynamic lookback for multi-period momentum: short-term (5-10 days) and medium-term (20-30 days)
    sma_5_10 = df['close'].rolling(window=5).mean() + (df['close'].rolling(window=10).mean() - df['close'].rolling(window=5).mean()) * (df['volume'] / df['volume'].rolling(window=14).mean())
    sma_20_30 = df['close'].rolling(window=20).mean() + (df['close'].rolling(window=30).mean() - df['close'].rolling(window=20).mean()) * (df['volume'] / df['volume'].rolling(window=14).mean())
    short_momentum = (df['close'] > sma_5_10).astype(int)
    medium_momentum = (sma_5_10 > sma_20_30).astype(int)
    
    # Volume shock: significant increase in volume compared to the 14-day average
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    volume_shock = (df['volume'] / avg_volume_14) > 1.5
    
    # Weighted return over the last 3-7 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() + (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=7).sum()
    weighted_return /= (df['volume'].rolling(window=3).sum() + df['volume'].rolling(window=7).sum())
    
    # RSI (Relative Strength Index) for the last 14-28 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Liquidity factor: high volume relative to the daily range
    liquidity = df['volume'] / (df['high'] - df['low'])
    
    # Market sentiment: change in open price relative to the previous close
    market_sentiment = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volatility clustering: exponential moving average of the daily range
    vol_cluster = (df['high'] - df['low']).ewm(span=14, adjust=False).mean()
    
    # Combine factors: adaptive ATR, dynamic multi-period momentum, volume shock, RSI, weighted return, liquidity, and market sentiment
    factor = (1 / (atr_14 + 1e-7)) * (short_momentum + medium_momentum) * volume_shock * (rsi / 100) * weighted_return * liquidity * market_sentiment * (1 / (vol_cluster + 1e-7))
    
    return factor
```
