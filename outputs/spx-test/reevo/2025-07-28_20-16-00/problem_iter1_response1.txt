```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average trading volume over a 5-day window
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of the current day's volume to the 5-day average volume
    volume_ratio = df['volume'] / avg_volume_5d
    
    # The factor is the product of the price change and the normalized volume
    factor = diff_close_open * volume_ratio
    
    return factor
```
