```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices, then 
    # normalize by dividing by the range (high - low). This factor aims to capture how the stock closed
    # relative to its intra-day range in a way that slightly penalizes wide ranging days.
    mid_price = (df['high'] + df['low']) / 2
    normalized_close = (df['close'] - mid_price) / (df['high'] - df['low'] + 1e-7)
    return normalized_close
```
