```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the change in closing prices
    close_change = df['close'].pct_change()
    
    # Calculate the average true range (high - low or high - previous close or previous close - low)
    avg_true_range = ((df['high'] - df['low']) + (df['high'] - df['close'].shift(1)).abs() + (df['close'].shift(1) - df['low']).abs()) / 3
    
    # Calculate a weighted momentum factor, giving more weight to recent changes
    weighted_momentum = (close_change.rolling(window=5).mean() * 2.0 + close_change.rolling(window=10).mean()) / 3.0
    
    # Incorporate volume by calculating the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the standard deviation of returns as a measure of volatility
    volatility = close_change.rolling(window=20).std()
    
    # Generate the alpha factor, with an interaction between price momentum, volatility, and volume
    alpha_factor = (weighted_momentum * (volatility.mean() / volatility) * volume_change) * (avg_true_range / df['close'])
    
    # Incorporate relative strength by calculating the ratio of the current close to the 50-day SMA of the close
    relative_strength = df['close'] / df['close'].rolling(window=50).mean()
    
    # Adaptive window for momentum based on the ATR
    adaptive_window = (avg_true_range.rolling(window=10).mean() * 10).astype(int)
    adaptive_momentum = close_change.rolling(window=adaptive_window, min_periods=1).mean()
    
    # Integrate macroeconomic indicators (e.g., S&P 500 index, 10-year Treasury yield)
    # Assuming 'macro' is a column in the DataFrame representing a macroeconomic indicator
    if 'macro' in df.columns:
        macro_indicator = df['macro']
        macro_change = macro_indicator.pct_change()
        macro_volatility = macro_change.rolling(window=20).std()
        macro_adjustment = (macro_change * (macro_volatility.mean() / macro_volatility))
    else:
        macro_adjustment = 1.0
    
    # Combine all factors into the final alpha factor
    alpha_factor = alpha_factor * relative_strength * adaptive_momentum * macro_adjustment
    
    return alpha_factor
```
