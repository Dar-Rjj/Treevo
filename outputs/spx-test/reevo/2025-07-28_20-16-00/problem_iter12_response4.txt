```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate a simple moving average for the close price over 5 and 20 days as basic trend indicators
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Determine if today's close is above the 5-day SMA, indicating an upward momentum
    short_momentum = (df['close'] > sma_5).astype(int)
    
    # Determine if today's close is above the 20-day SMA, indicating a longer-term upward momentum
    long_momentum = (df['close'] > sma_20).astype(int)
    
    # Calculate the volume-weighted return of the last 3 and 10 days
    weighted_return_3 = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    weighted_return_10 = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last 14 days
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Integrate a sentiment factor based on the change in volume compared to the previous day
    volume_change_sentiment = df['volume'].pct_change().fillna(0)
    
    # Create a factor combining volatility (as inverse, higher volatility might lead to reversion), short-term and long-term momentum, RSI, ATR, volume-weighted returns, volume ratio, and sentiment
    factor = (1 / (daily_volatility + 1e-7)) * (short_momentum * 0.6 + long_momentum * 0.4) * (rsi / 100) * (1 / (atr + 1e-7)) * (weighted_return_3 * 0.7 + weighted_return_10 * 0.3) * volume_ratio * (1 + volume_change_sentiment)
    
    return factor
```
