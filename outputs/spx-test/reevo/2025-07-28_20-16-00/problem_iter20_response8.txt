```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Exponential Moving Average (EMA) for the close price over 5 days as a more reactive trend indicator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Determine if today's close is above the 5-day EMA, indicating an upward momentum
    momentum = (df['close'] > ema_5).astype(int)
    
    # Calculate the volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days using EMA
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last 14 days using EMA
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].ewm(span=14, adjust=False).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Introduce a macroeconomic sentiment indicator (simplified as a moving average of the close price over 20 days)
    macro_sentiment = df['close'].rolling(window=20).mean()
    
    # Calculate the MACD (Moving Average Convergence Divergence)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line
    
    # Calculate the Bollinger Bands
    bb_middle = df['close'].rolling(window=20).mean()
    bb_std = df['close'].rolling(window=20).std()
    bb_upper = bb_middle + 2 * bb_std
    bb_lower = bb_middle - 2 * bb_std
    bb_percent = (df['close'] - bb_lower) / (bb_upper - bb_lower)
    
    # Create a factor combining volatility (as inverse, higher volatility might lead to reversion), momentum, RSI, ATR, volume ratio, macro sentiment, MACD, and Bollinger Bands
    factor = (1 / (daily_volatility + 1e-7)) * momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return * volume_ratio * (macro_sentiment / df['close']) * macd_histogram * bb_percent
    
    return factor
```
