```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between open and close price
    diff_open_close = (df['close'] - df['open']) / df['open']
    # Calculate the average of volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    # Create a factor that combines the percentage change in close price relative to open
    # and the current day's volume compared to the 5-day average volume
    factor = diff_open_close * (df['volume'] / avg_volume_5d)
    return factor
```
