```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Exponential Moving Average (EMA) for the close price over 5 days as a more reactive trend indicator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Adaptive EMA with a span that changes based on the recent volatility
    adaptive_span = 5 + (daily_volatility.rolling(window=5).std() * 10)
    adaptive_ema = df['close'].ewm(span=adaptive_span, adjust=False).mean()
    
    # Determine if today's close is above the adaptive EMA, indicating an upward momentum
    adaptive_momentum = (df['close'] > adaptive_ema).astype(int)
    
    # Multi-period momentum: calculate the average of the momentum over the last 3, 5, and 10 days
    momentum_3 = (df['close'] > df['close'].shift(3)).astype(int)
    momentum_5 = (df['close'] > df['close'].shift(5)).astype(int)
    momentum_10 = (df['close'] > df['close'].shift(10)).astype(int)
    multi_period_momentum = (momentum_3 + adaptive_momentum + momentum_5 + momentum_10) / 4
    
    # Volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # RSI (Relative Strength Index) for the last 14 days using EMA
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # ATR (Average True Range) for the last 14 days using EMA
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].ewm(span=14, adjust=False).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Macro sentiment (simplified as a moving average of the close price over 20 days)
    macro_sentiment = df['close'].rolling(window=20).mean()
    
    # Sector-specific indicator (simplified as a 50-day moving average of the close price)
    sector_indicator = df['close'].rolling(window=50).mean()
    
    # Longer-term market sentiment (simplified as a 100-day moving average of the close price)
    long_term_sentiment = df['close'].rolling(window=100).mean()
    
    # Create a factor combining volatility (as inverse), multi-period momentum, RSI, ATR, volume ratio, and macro/sector/long-term sentiments
    factor = (1 / (daily_volatility + 1e-7)) * multi_period_momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return * volume_ratio * (macro_sentiment / df['close']) * (sector_indicator / df['close']) * (long_term_sentiment / df['close'])
    
    return factor
```
