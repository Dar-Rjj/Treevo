```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Exponential Moving Average (EMA) for the close price over 5 days as a more reactive trend indicator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Determine if today's close is above the 5-day EMA, indicating an upward momentum
    momentum = (df['close'] > ema_5).astype(int)
    
    # Calculate the volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days using EMA
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last 14 days using EMA
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].ewm(span=14, adjust=False).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Introduce a macroeconomic sentiment indicator (simplified as a moving average of the close price over 20 days)
    macro_sentiment_20 = df['close'].rolling(window=20).mean()
    
    # Create a factor combining volatility (as inverse, higher volatility might lead to reversion), momentum, RSI, ATR, volume ratio, and macro sentiment
    factor = (1 / (daily_volatility + 1e-7)) * momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return * volume_ratio * (macro_sentiment_20 / df['close'])
    
    # Add a simple adaptive lookback mechanism by smoothing the factor with a 3-day EMA
    factor = factor.ewm(span=3, adjust=False).mean()
    
    # Introduce a dynamic lookback for the RSI and ATR to adapt to market conditions
    rsi_adaptive = (rsi.rolling(window=14).mean() / 100)
    atr_adaptive = (atr.rolling(window=14).mean() + 1e-7)
    
    # Recompute the factor with adaptive RSI and ATR
    factor = (1 / (daily_volatility + 1e-7)) * momentum * rsi_adaptive * (1 / atr_adaptive) * weighted_return * volume_ratio * (macro_sentiment_20 / df['close'])
    
    # Smooth the final factor with a 3-day EMA for better stability
    factor = factor.ewm(span=3, adjust=False).mean()
    
    # Integrate additional macroeconomic indicators (e.g., 50-day moving average of the close price)
    macro_sentiment_50 = df['close'].rolling(window=50).mean()
    macro_factor = (macro_sentiment_50 / df['close'])
    
    # Combine the macroeconomic factor with the existing factor
    factor = factor * macro_factor
    
    # Integrate a longer-term market sentiment indicator (simplified as a 100-day moving average of the close price)
    long_term_sentiment = df['close'].rolling(window=100).mean()
    long_term_factor = (long_term_sentiment / df['close'])
    
    # Combine the longer-term factor with the existing factor
    factor = factor * long_term_factor
    
    return factor
```
