```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Exponential moving average for the close price over 5 days as an adaptive trend indicator
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Determine if today's close is above the 5-day EMA, indicating an upward momentum
    short_momentum = (df['close'] > ema_5).astype(int)
    
    # Calculate the volume-weighted return of the last 3 days
    weighted_return_3 = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days using exponential smoothing
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).ewm(com=13, adjust=False).mean()
    loss = -delta.where(delta < 0, 0).ewm(com=13, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last 14 days using exponential smoothing
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].ewm(span=14, adjust=False).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Integrate a sentiment factor based on the change in volume compared to the previous day
    volume_change_sentiment = df['volume'].pct_change().fillna(0)
    
    # Create a factor combining volatility (as inverse, higher volatility might lead to reversion), short-term momentum, RSI, ATR, volume-weighted returns, volume ratio, and sentiment
    factor = (1 / (daily_volatility + 1e-7)) * short_momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return_3 * volume_ratio * (1 + volume_change_sentiment)
    
    return factor
```
