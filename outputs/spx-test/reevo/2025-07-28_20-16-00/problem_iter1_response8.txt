```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and the 5-day moving average of the close prices.
    ma5 = df['close'].rolling(window=5).mean()
    diff_ma5 = df['close'] - ma5
    
    # Calculate the volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # Calculate the momentum factor, which is the sum of the past 5-day differences in closing prices
    momentum_5d = df['close'].diff().rolling(window=5).sum()
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (diff_ma5 + weighted_return + momentum_5d) / 3
    return alpha_factor
```
