```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate simple moving averages for the close price over 5 and 20 days as trend indicators
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Determine if today's close is above the 5-day SMA, indicating an upward momentum
    short_term_momentum = (df['close'] > sma_5).astype(int)
    long_term_momentum = (df['close'] > sma_20).astype(int)
    
    # Calculate the volume-weighted return of the last 3 and 10 days
    weighted_return_3 = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    weighted_return_10 = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last 14 days
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Refine the true range by incorporating the previous day's close
    refined_true_range = (df[['high', 'low', 'close']].shift(1).apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1) + true_range) / 2
    atr_refined = refined_true_range.rolling(window=14).mean()
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the 14-day average volume
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    volume_ratio = df['volume'] / avg_volume_14
    
    # Incorporate mean reversion using Bollinger Bands
    bollinger_mid = df['close'].rolling(window=20).mean()
    bollinger_std = df['close'].rolling(window=20).std()
    bollinger_upper = bollinger_mid + 2 * bollinger_std
    bollinger_lower = bollinger_mid - 2 * bollinger_std
    mean_reversion = (df['close'] - bollinger_mid) / (2 * bollinger_std)
    
    # Add another technical indicator: Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.rolling(window=9).mean()
    macd_histogram = macd_line - signal_line
    
    # Create a factor combining volatility (as inverse, higher volatility might lead to reversion), momentum, RSI, ATR, and volume ratio
    factor = (1 / (daily_volatility + 1e-7)) * (short_term_momentum * 0.7 + long_term_momentum * 0.3) * (rsi / 100) * (1 / (atr_refined + 1e-7)) * (weighted_return_3 * 0.5 + weighted_return_10 * 0.5) * volume_ratio * (1 - abs(mean_reversion)) * macd_histogram
    
    return factor
```
