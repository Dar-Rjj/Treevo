```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    price_diff = df['close'] - df['open']
    
    # Calculate the average of high and low price
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the weighted price movement
    weighted_price_movement = price_diff * (df['volume'] / df['volume'].rolling(window=5).mean())
    
    # Calculate the relative strength indicator (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors
    factor_values = weighted_price_movement + (rsi - 50) * avg_price
    return factor_values
```
