```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive ATR over 14 days, with a smoothing factor to adapt to recent volatility changes
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr_14 = true_range.ewm(span=14, adjust=False).mean()
    
    # Multi-period momentum: short-term (5 days) and medium-term (20 days)
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    short_momentum = (df['close'] > sma_5).astype(int)
    medium_momentum = (sma_5 > sma_20).astype(int)
    
    # Volume shock: significant increase in volume compared to the 14-day average
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    volume_shock = (df['volume'] / avg_volume_14) > 1.5
    
    # Weighted return over the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # RSI (Relative Strength Index) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine factors: adaptive ATR, multi-period momentum, volume shock, RSI, and weighted return
    factor = (1 / (atr_14 + 1e-7)) * (short_momentum + medium_momentum) * volume_shock * (rsi / 100) * weighted_return
    
    return factor
```
