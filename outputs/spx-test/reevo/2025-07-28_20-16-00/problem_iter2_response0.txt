```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the change in closing prices
    close_change = df['close'].pct_change()
    
    # Calculate the average true range (ATR)
    avg_true_range = ((df['high'] - df['low']) + (df['high'] - df['close'].shift(1)).abs() + (df['close'].shift(1) - df['low']).abs()) / 3
    
    # Calculate a weighted momentum factor, giving more weight to recent changes
    weighted_momentum = (close_change.rolling(window=5).mean() * 2.0 + close_change.rolling(window=10).mean()) / 3.0
    
    # Incorporate volume by calculating the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the relative strength index (RSI) for momentum
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Generate the alpha factor, with an interaction between price momentum, RSI, volatility (measured by ATR), and volume
    alpha_factor = weighted_momentum * rsi * (avg_true_range.mean() / avg_true_range) * volume_change
    return alpha_factor
```
