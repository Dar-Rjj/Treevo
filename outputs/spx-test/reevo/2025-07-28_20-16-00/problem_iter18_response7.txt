```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, to gauge the day's volatility
    daily_volatility = df['high'] - df['low']
    
    # Exponential Moving Average (EMA) for the close price over an adaptive window as a more reactive trend indicator
    ema_window = 5 + (df['volume'].rolling(window=14).std() / df['volume'].mean()).fillna(5).astype(int)
    ema = df['close'].ewm(span=ema_window, adjust=False).mean()
    
    # Determine if today's close is above the adaptive EMA, indicating an upward momentum
    momentum = (df['close'] > ema).astype(int)
    
    # Calculate the volume-weighted return of the last 3 days
    weighted_return = (df['close'].pct_change().fillna(0) * df['volume']).rolling(window=3).sum() / df['volume'].rolling(window=3).sum()
    
    # Calculate the RSI (Relative Strength Index) for the last adaptive window using EMA
    rsi_window = 14 + (df['volume'].rolling(window=28).std() / df['volume'].mean()).fillna(14).astype(int)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=rsi_window, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=rsi_window, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ATR (Average True Range) for the last adaptive window using EMA
    atr_window = 14 + (df['volume'].rolling(window=28).std() / df['volume'].mean()).fillna(14).astype(int)
    true_range = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[2]) - min(x[1], x[2]), axis=1)
    atr = true_range.ewm(span=atr_window, adjust=False).mean()
    
    # Calculate the dynamic weighting based on the ratio of the current day's volume to the adaptive average volume
    avg_volume_window = 14 + (df['volume'].rolling(window=28).std() / df['volume'].mean()).fillna(14).astype(int)
    avg_volume = df['volume'].ewm(span=avg_volume_window, adjust=False).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Create a factor combining inverse adaptive volatility, momentum, RSI, inverse ATR, and volume ratio
    alpha_factor = (1 / (daily_volatility + 1e-7)) * momentum * (rsi / 100) * (1 / (atr + 1e-7)) * weighted_return * volume_ratio
    
    return alpha_factor
```
