```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, representing the daily trading range
    daily_range = df['high'] - df['low']

    # Compute an adjusted volume factor; this aims to measure volume intensity
    volume_intensity = df['volume'] / (daily_range + 1e-7)

    # Define a volatility-like feature by taking the standard deviation of the close price over the last few days
    rolling_std = df['close'].rolling(window=5).std()

    # Create a momentum indicator by computing the ratio of the current close price to the 10-day moving average
    momentum = df['close'] / df['close'].rolling(window=10).mean()

    # Calculate the relative strength index (RSI) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Adjust for liquidity using the volume and trading range
    liquidity_adjustment = df['volume'] / (daily_range + 1e-7)

    # Integrate market sentiment by calculating the change in volume
    volume_change = df['volume'].pct_change().fillna(0)

    # Use advanced statistical methods: Calculate the exponential weighted moving average (EWMA) of the close price
    ewma_close = df['close'].ewm(span=10, adjust=False).mean()

    # Combine all components into a single interpretable alpha factor
    alpha_factor = (momentum * rsi * liquidity_adjustment * volume_change * (df['close'] / ewma_close)) / (rolling_std + 1e-7)

    return alpha_factor
```
