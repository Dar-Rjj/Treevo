```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the high and low for each day, giving a sense of the day's range
    daily_range = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate the positive price change: today's close minus the open
    positive_change = df['close'] - df['open']
    
    # Compute the volume weighted average price (VWAP) over the last 10 periods
    total_volume = df['volume'].rolling(window=10).sum()
    vwap = ((df['close']*df['volume']).rolling(window=10).sum() / total_volume).fillna(0)
    
    # Factor: Average positive change relative to the VWAP, adjusted by the daily range
    factor_value = (positive_change / (vwap * daily_range)).fillna(0)
    
    return factor_value
```
