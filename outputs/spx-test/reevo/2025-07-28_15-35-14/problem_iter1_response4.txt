```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and opening price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted price
    volume_weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Generate the alpha factor
    alpha_factor = close_open_diff * (avg_high_low - df['close']) / (volume_weighted_price + 1e-7)
    
    return alpha_factor
```
