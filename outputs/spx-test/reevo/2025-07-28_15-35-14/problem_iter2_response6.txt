def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range as the difference between high and low prices
    daily_range = df['high'] - df['low']
    
    # Compute an adjusted volume factor to measure volume intensity
    volume_intensity = df['volume'] / (daily_range + 1e-7)
    
    # Define a short-term volatility feature using the standard deviation of the close price over the last 5 days
    rolling_std_short = df['close'].rolling(window=5).std()
    
    # Define a medium-term volatility feature using the standard deviation of the close price over the last 20 days
    rolling_std_medium = df['close'].rolling(window=20).std()
    
    # Create a short-term momentum indicator by computing the ratio of the current close price to the 10-day moving average
    momentum_short = df['close'] / df['close'].rolling(window=10).mean()
    
    # Create a medium-term momentum indicator by computing the ratio of the current close price to the 30-day moving average
    momentum_medium = df['close'] / df['close'].rolling(window=30).mean()
    
    # Calculate the RSI (Relative Strength Index) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Combine all components into a single interpretable alpha factor
    alpha_factor = (daily_range * volume_intensity * momentum_short * momentum_medium * rsi) / (rolling_std_short * rolling_std_medium + 1e-7)
    
    return alpha_factor
