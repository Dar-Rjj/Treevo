```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the average volume over the last 5 days
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    # Calculate the relative strength as the ratio of the daily close difference to the average volume
    df['relative_strength'] = df['close_diff'] / (df['avg_vol_5d'] + 1e-7)
    # Return the relative strength factor
    return df['relative_strength']
```
