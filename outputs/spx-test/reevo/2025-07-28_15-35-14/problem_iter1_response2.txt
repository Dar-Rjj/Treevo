def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average of the open and close prices
    avg_open_close = (df['open'] + df['close']) / 2
    
    # Calculate the difference between high and low prices, adding a small epsilon to avoid division by zero
    range_high_low = (df['high'] - df['low'] + 1e-7)
    
    # Calculate the weighted average of the price over the day
    weighted_price = (df['volume']*avg_open_close) / df['volume'].rolling(window=5).mean()
    
    # The heuristic is the change in the weighted price over the last 5 days
    heuristic = weighted_price.pct_change(periods=5)
    return heuristic.fillna(0)
