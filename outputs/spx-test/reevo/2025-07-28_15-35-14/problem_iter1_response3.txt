def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the average of the volume over the last 5 days
    df['volume_avg_5d'] = df['volume'].rolling(window=5).mean()
    # Calculate the relative strength of the stock by dividing the close price by the volume
    df['relative_strength'] = df['close'] / (df['volume_avg_5d'] + 1e-7)
    # Calculate the momentum as the 10-day moving average of the close price differences
    df['momentum_10d'] = df['close_diff'].rolling(window=10).mean()
    # The alpha factor is a combination of relative strength and momentum
    return (df['relative_strength'] * df['momentum_10d'])
