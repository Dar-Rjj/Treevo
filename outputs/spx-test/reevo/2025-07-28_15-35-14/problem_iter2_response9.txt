```txt
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing and opening price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted price
    volume_weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate the rolling mean of the closing price
    rolling_close_mean = df['close'].rolling(window=5).mean()
    
    # Calculate the rolling mean of the volume
    rolling_volume_mean = df['volume'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of the closing price
    rolling_close_std = df['close'].rolling(window=5).std()
    
    # Calculate the rolling standard deviation of the volume
    rolling_volume_std = df['volume'].rolling(window=10).std()
    
    # Generate the alpha factor
    alpha_factor = (close_open_diff * (avg_high_low - df['close']) / (volume_weighted_price + 1e-7)) * \
                   (rolling_close_mean / (rolling_close_std + 1e-7)) * \
                   (rolling_volume_mean / (rolling_volume_std + 1e-7))
    
    return alpha_factor
```
