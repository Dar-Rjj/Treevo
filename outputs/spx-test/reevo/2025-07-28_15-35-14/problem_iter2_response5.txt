```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day simple moving averages of the close price
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the 10-day and 30-day SMAs to identify trend strength
    df['trend_strength'] = df['SMA_10'] - df['SMA_30']
    
    # Calculate the 5-day average of the volume
    df['volume_avg_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the relative strength by dividing the close price by the 5-day volume average
    df['relative_strength'] = df['close'] / (df['volume_avg_5d'] + 1e-7)
    
    # Calculate the 10-day momentum of the close price
    df['momentum_10d'] = df['close_diff'] = df['close'].diff().rolling(window=10).mean()
    
    # Calculate the standard deviation of the close price over the last 20 days to measure volatility
    df['volatility_20d'] = df['close'].rolling(window=20).std()
    
    # The alpha factor is a combination of trend strength, relative strength, momentum, and volatility
    return (df['trend_strength'] * 0.4) + (df['relative_strength'] * 0.3) + (df['momentum_10d'] * 0.2) + (1 / df['volatility_20d'] * 0.1)
```
