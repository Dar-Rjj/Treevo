```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the body of the candlestick
    body = (df['close'] - df['open']).abs()
    
    # Calculate the upper and lower shadows
    upper_shadow = df['high'] - df[['open', 'close']].max(axis=1)
    lower_shadow = df[['open', 'close']].min(axis=1) - df['low']
    
    # Calculate the true range
    true_range = df['high'] - df['low']
    
    # Calculate the volatility as the standard deviation of the true range over the last 5 days
    volatility = true_range.rolling(window=5).std()
    
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted price
    volume_weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Generate the alpha factor
    alpha_factor = (body * (avg_high_low - df['close']) / (volume_weighted_price + 1e-7)) * (volatility + 1e-7)
    
    return alpha_factor
```
