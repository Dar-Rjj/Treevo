```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day simple moving average of the price movement (close - open)
    price_movement = (df['close'] - df['open']).rolling(window=5).mean()
    
    # Calculate the 10-day exponential moving average of the volume trend (pct change)
    volume_trend = df['volume'].pct_change().ewm(span=10, adjust=False).mean()
    
    # Compute the weighted sum of both, with a focus on the price movement
    return (price_movement * 0.7 + volume_trend * 0.3)
```
