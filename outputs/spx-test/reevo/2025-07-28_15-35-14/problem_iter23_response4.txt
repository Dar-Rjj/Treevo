def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import numpy as np
    import pandas as pd

    # Momentum - Calculate the 10-day and 30-day Hull Moving Averages (HMA) of the close price
    def hma(series, period):
        wma_half = series.rolling(window=period//2).mean()
        wma_full = series.rolling(window=period).mean()
        return (2 * wma_half - wma_full).rolling(window=int(np.sqrt(period))).mean()

    df['hma_10d'] = hma(df['close'], 10)
    df['hma_30d'] = hma(df['close'], 30)
    # Calculate the momentum as the difference between the two HMAs
    df['momentum'] = df['hma_10d'] - df['hma_30d']

    # Volatility - Calculate the average true range (ATR) over the last 30 days with a more sophisticated method
    df['high_low'] = df['high'] - df['low']
    df['high_close'] = np.abs(df['high'] - df['close'].shift())
    df['low_close'] = np.abs(df['low'] - df['close'].shift())
    df['tr'] = df[['high_low', 'high_close', 'low_close']].max(axis=1)
    df['atr_30d'] = df['tr'].ewm(span=30, adjust=False).mean()

    # Liquidity - Calculate the average volume over the last 20 days
    df['avg_vol_20d'] = df['volume'].rolling(window=20).mean()

    # Price-Volume Relationship - Calculate the 10-day change in volume
    df['volume_change_10d'] = df['volume'].pct_change(periods=10)

    # Relative Strength - Calculate the ratio of the daily log return to the average volume
    df['log_returns'] = np.log(df['close']).diff()
    df['relative_strength'] = df['log_returns'] / (df['avg_vol_20d'] + 1e-7)

    # Market Sentiment - Calculate the 10-day moving average of the high and low prices
    df['ma_high_10d'] = df['high'].rolling(window=10).mean()
    df['ma_low_10d'] = df['low'].rolling(window=10).mean()
    df['market_sentiment'] = df['ma_high_10d'] - df['ma_low_10d']

    # Adaptive Weights - Use a simple HMA cross-over to determine the trend direction
    df['trend_direction'] = (df['hma_10d'] > df['hma_30d']).astype(int)

    # Dynamic Price-Volume Interaction - Adjust relative strength based on the trend direction
    df['dynamic_relative_strength'] = df['relative_strength'] * (1 + 0.5 * (df['trend_direction'] - 0.5))

    # Regime-Based Adjustments - Apply different weights based on the volatility regime
    df['vol_regime'] = (df['atr_30d'] > df['atr_30d'].quantile(0.5)).astype(int)

    market_sentiment_weight = 0.1 + 0.05 * (df['vol_regime'] - 0.5)

        df['dynamic_relative_strength'] * dynamic_relative_strength_weight +

    )
