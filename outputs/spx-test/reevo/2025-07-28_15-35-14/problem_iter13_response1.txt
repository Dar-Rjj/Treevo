```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum - Calculate the 10-day and 30-day Exponential Moving Averages of the close price with adaptive windows
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30d'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the adaptive window based on the standard deviation of log returns
    df['log_returns'] = np.log(df['close']).diff()
    df['std_log_returns'] = df['log_returns'].rolling(window=30).std()
    df['adaptive_window'] = (df['std_log_returns'] * 10 + 10).astype(int)
    
    # Recalculate EMAs with adaptive windows
    df['ema_adaptive_10d'] = df.apply(lambda row: df['close'].ewm(span=row['adaptive_window'], adjust=False).mean().iloc[-1], axis=1)
    df['ema_adaptive_30d'] = df.apply(lambda row: df['close'].ewm(span=row['adaptive_window']*3, adjust=False).mean().iloc[-1], axis=1)
    
    # Calculate the momentum as the difference between the two adaptive EMAs
    df['momentum'] = df['ema_adaptive_10d'] - df['ema_adaptive_30d']
    
    # Volatility - Calculate the Average True Range (ATR) over the last 30 days
    df['tr'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)), abs(x['low'] - df['close'].shift(1))), axis=1)
    df['atr'] = df['tr'].rolling(window=30).mean()
    
    # Liquidity - Calculate the average volume over the last 20 days
    df['avg_vol_20d'] = df['volume'].rolling(window=20).mean()
    
    # Price-Volume Relationship - Calculate the 10-day change in volume
    df['volume_change_10d'] = df['volume'].pct_change(periods=10)
    
    # Relative Strength - Calculate the ratio of the daily close difference to the average volume
    df['close_diff'] = df['close'].diff()
    df['relative_strength'] = df['close_diff'] / (df['avg_vol_20d'] + 1e-7)
    
    # Market Sentiment - Calculate the 10-day moving average of the high and low prices
    df['ma_high_10d'] = df['high'].rolling(window=10).mean()
    df['ma_low_10d'] = df['low'].rolling(window=10).mean()
    df['market_sentiment'] = df['ma_high_10d'] - df['ma_low_10d']
    
    # Adaptive Weights - Use a simple EMA cross-over to determine the trend direction
    df['trend_direction'] = (df['ema_adaptive_10d'] > df['ema_adaptive_30d']).astype(int)
    
    # Dynamic Price-Volume Interaction - Adjust relative strength based on the trend direction
    df['dynamic_relative_strength'] = df['relative_strength'] * (1 + 0.5 * (df['trend_direction'] - 0.5))
    
    # Regime-Based Adjustments - Apply different weights based on the volatility regime
    df['vol_regime'] = (df['atr'] > df['atr'].quantile(0.5)).astype(int)
    df['composite_factor'] = (
        df['momentum'] * (0.4 + 0.1 * (df['vol_regime'] - 0.5)) -
        df['atr'] * (0.2 - 0.05 * (df['vol_regime'] - 0.5)) +
        df['dynamic_relative_strength'] * (0.2 + 0.1 * (df['vol_regime'] - 0.5)) +
        df['volume_change_10d'] * (0.1 - 0.05 * (df['vol_regime'] - 0.5)) +
        df['market_sentiment'] * (0.1 + 0.05 * (df['vol_regime'] - 0.5))
    )
    
    # Return the composite factor
    return df['composite_factor']
```
