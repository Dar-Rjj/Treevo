```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices, to understand the price movement of the day
    df['price_movement'] = df['close'] - df['open']
    # Calculate the true range, which is the maximum of three values: 
    #     1. Current high less the current low, 
    #     2. The absolute value of the current high less the previous close, 
    #     3. The absolute value of the current low less the previous close.
    # This helps in understanding the volatility during that day.
    df['true_range'] = df.apply(lambda x: max(x['high']-x['low'], abs(x['high']-df.shift(1).loc[x.name]['close']), abs(x['low']-df.shift(1).loc[x.name]['close'])), axis=1)
    
    # An attempt to measure the strength of the daily price movement by comparing the daily price change with the daily true range
    alpha_factor = df['price_movement'] / (df['true_range'] + 1e-7)
    return alpha_factor
```
