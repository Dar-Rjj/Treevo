```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum - Calculate the 10-day and 30-day exponential moving averages of the close price
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30d'] = df['close'].ewm(span=30, adjust=False).mean()
    # Calculate the momentum as the difference between the two exponential moving averages
    df['momentum'] = df['ema_10d'] - df['ema_30d']

    # Volatility - Calculate the standard deviation of the daily log returns over the last 30 days
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility'] = df['log_returns'].rolling(window=30).std()

    # Liquidity - Calculate the average volume over the last 20 days
    df['avg_vol_20d'] = df['volume'].rolling(window=20).mean()

    # Price-Volume Relationship - Calculate the 10-day change in volume
    df['volume_change_10d'] = df['volume'].pct_change(periods=10)

    # Relative Strength - Calculate the ratio of the daily close difference to the average volume
    df['close_diff'] = df['close'].diff()
    df['relative_strength'] = df['close_diff'] / (df['avg_vol_20d'] + 1e-7)

    # Market Sentiment - Calculate the 10-day moving average of the high and low prices
    df['ma_high_10d'] = df['high'].rolling(window=10).mean()
    df['ma_low_10d'] = df['low'].rolling(window=10).mean()
    df['market_sentiment'] = df['ma_high_10d'] - df['ma_low_10d']

    # Adaptive Weights - Adjust weights based on the current market regime
    df['regime'] = np.where(df['momentum'] > 0, 'bull', 'bear')
    df['momentum_weight'] = np.where(df['regime'] == 'bull', 0.5, 0.3)
    df['volatility_weight'] = np.where(df['regime'] == 'bull', 0.1, 0.2)
    df['relative_strength_weight'] = np.where(df['regime'] == 'bull', 0.2, 0.3)
    df['volume_change_weight'] = np.where(df['regime'] == 'bull', 0.1, 0.1)
    df['market_sentiment_weight'] = np.where(df['regime'] == 'bull', 0.1, 0.1)

    # Composite Factor - Combine momentum, volatility, liquidity, relative strength, and market sentiment with adaptive weights
    df['composite_factor'] = (df['momentum'] * df['momentum_weight'] - 
                              df['volatility'] * df['volatility_weight'] + 
                              df['relative_strength'] * df['relative_strength_weight'] + 
                              df['volume_change_10d'] * df['volume_change_weight'] + 
                              df['market_sentiment'] * df['market_sentiment_weight'])

    # Return the composite factor
    return df['composite_factor']
```
