```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, smoothed over a 5-day window to capture short-term volatility
    short_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Compute the momentum as the rate of change of closing price over the last 30 days, representing medium-term trend
    momentum_30 = df['close'].pct_change(periods=30)
    
    # Calculate the ratio of current volume to the 10-day average volume, indicating unusual trading activity
    volume_surprise = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Alpha factor combines short-term volatility, momentum, and volume surprise
    factor = (momentum_30 * volume_surprise) / (short_volatility + 1e-7)
    
    return factor
```
