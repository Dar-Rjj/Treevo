```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of the daily range (high - low)
    daily_range = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the positive price change (close - open)
    positive_change = (df['close'] - df['open']).rolling(window=10).mean()
    
    # Compute the 10-day volume weighted average price (VWAP)
    total_volume = df['volume'].rolling(window=10).sum()
    vwap = ((df['close']*df['volume']).rolling(window=10).sum() / total_volume).fillna(0)
    
    # Calculate the 30-day standard deviation of returns as a measure of volatility
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=30).std()
    
    # Calculate the 20-day momentum as the percentage change in close price
    momentum = df['close'].pct_change(periods=20)
    
    # Factor: Combine the positive change, VWAP, daily range, and adjust by volatility and momentum
    factor_value = (positive_change / (vwap * daily_range)) * (volatility + 1) * (momentum + 1)
    factor_value = factor_value.fillna(0)
    
    return factor_value
```
