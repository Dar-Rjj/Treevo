def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    diff = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate the logarithmic return of the volume
    log_volume_return = np.log(df['volume'] / df['volume'].shift(1))
    
    # Compute the weighted sum of both, with a focus on the price movement
    return (diff * 0.75) + (log_volume_return * 0.25)
