def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum - Calculate the 10-day and 30-day moving averages of the close price
    df['ma_10d'] = df['close'].rolling(window=10).mean()
    df['ma_30d'] = df['close'].rolling(window=30).mean()
    # Calculate the momentum as the difference between the two moving averages
    df['momentum'] = df['ma_10d'] - df['ma_30d']

    # Volatility - Calculate the standard deviation of the daily log returns over the last 30 days
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=30).std()

    # Liquidity - Calculate the average volume over the last 20 days
    df['avg_vol_20d'] = df['volume'].rolling(window=20).mean()

    # Price-Volume Relationship - Calculate the 10-day change in volume
    df['volume_change_10d'] = df['volume'].pct_change(periods=10)

    # Relative Strength - Calculate the ratio of the daily close difference to the average volume
    df['close_diff'] = df['close'].diff()
    df['relative_strength'] = df['close_diff'] / (df['avg_vol_20d'] + 1e-7)

    # Market Microstructure - Calculate the bid-ask spread proxy using high and low prices
    df['bid_ask_spread_proxy'] = df['high'] - df['low']

    # Advanced Statistical Feature - Calculate the Hurst exponent for the last 30 days
    def hurst_exponent(price_series, lags_count=10):
        lags = range(2, lags_count)
        tau = [np.sqrt(np.std(np.subtract(price_series[lag:], price_series[:-lag]))) for lag in lags]
        poly = np.polyfit(np.log(lags), np.log(tau), 1)
        return poly[0] * 2.0

    df['hurst_exponent'] = df['close'].rolling(window=30).apply(hurst_exponent, raw=True)

    # Composite Factor - Combine momentum, volatility, liquidity, relative strength, bid-ask spread, and Hurst exponent
    df['composite_factor'] = (
        0.3 * df['momentum'] 
        - 0.2 * df['volatility'] 
        + 0.1 * df['relative_strength'] 
        + 0.1 * df['volume_change_10d'] 
        - 0.1 * df['bid_ask_spread_proxy'] 
        + 0.2 * df['hurst_exponent']
    )

    # Return the composite factor
    return df['composite_factor']
