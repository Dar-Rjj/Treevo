def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum of the stock by comparing the current close price to the 10-period moving average
    df['Momentum'] = (df['close'] - df['close'].rolling(window=10).mean()) / df['close'].rolling(window=10).std()
    # Adjust the momentum for the volume, as higher volumes may indicate stronger moves
    return df['Momentum'] * (df['volume'] / df['volume'].rolling(window=10).mean())
