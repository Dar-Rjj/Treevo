```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 5-day and 20-day rolling standard deviation of the returns to use as a volatility measure
    df['volatility_5'] = df['daily_return'].rolling(window=5).std()
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate the volume-weighted price change
    df['volume_scaled_price_change'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate the 5-day and 20-day moving average of the volume-scaled price change
    df['ma_volume_scaled_price_change_5'] = df['volume_scaled_price_change'].rolling(window=5).mean()
    df['ma_volume_scaled_price_change_20'] = df['volume_scaled_price_change'].rolling(window=20).mean()
    
    # Calculate the 10-day and 30-day moving average of the closing price
    df['ma_close_10'] = df['close'].rolling(window=10).mean()
    df['ma_close_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the current day's close to the 10-day and 30-day moving average of the closing price
    df['close_ma_ratio_10'] = df['close'] / df['ma_close_10']
    df['close_ma_ratio_30'] = df['close'] / df['ma_close_30']
    
    # Integrate trend strength by calculating the difference between the 10-day and 30-day moving averages of the closing prices
    df['trend_strength'] = df['ma_close_10'] - df['ma_close_30']
    
    # Combine the factors: volatility, volume-scaled price change, close-to-MA ratio, and trend strength
    alpha_factor = (df['volatility_5'] + df['volatility_20'] +
                    (df['ma_volume_scaled_price_change_5'] * df['close_ma_ratio_10']) + 
                    (df['ma_volume_scaled_price_change_20'] * df['close_ma_ratio_30']) + 
                    df['trend_strength'])
    
    return alpha_factor
```
