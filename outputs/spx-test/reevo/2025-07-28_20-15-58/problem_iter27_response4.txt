```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the logarithmic daily return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 5-day rolling standard deviation of the log returns to use as a volatility measure
    df['volatility'] = df['log_return'].rolling(window=5).std()
    
    # Calculate the volume-weighted price change
    df['volume_scaled_price_change'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate the 5-day moving average of the volume-scaled price change
    df['ma_volume_scaled_price_change'] = df['volume_scaled_price_change'].rolling(window=5).mean()
    
    # Calculate the 10-day and 30-day moving averages of the closing price
    df['ma_close_10'] = df['close'].rolling(window=10).mean()
    df['ma_close_30'] = df['close'].rolling(window=30).mean()
    
    # Integrate trend strength by calculating the difference between the 10-day and 30-day moving averages of the closing prices
    df['trend_strength'] = df['ma_close_10'] - df['ma_close_30']
    
    # Adaptive window for volatility based on the 5-day momentum
    df['adaptive_volatility'] = df['volatility'] * (1 + df['log_return'].abs().rolling(window=5).sum())
    
    # Dynamic weighting for the factors
    df['weight_log_return'] = df['log_return'].abs().rolling(window=5).mean()
    df['weight_volume_scaled'] = df['volume_scaled_price_change'].abs().rolling(window=5).mean()
    df['weight_trend_strength'] = df['trend_strength'].abs().rolling(window=5).mean()
    
    # Exponential Moving Average (EMA) for smoother trends
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_close_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # EMA-based trend strength
    df['ema_trend_strength'] = df['ema_close_10'] - df['ema_close_30']
    
    # Market sentiment with a simple indicator
    df['market_sentiment'] = df['close'] - df['open']
    df['market_sentiment'] = df['market_sentiment'].apply(lambda x: 1 if x > 0 else -1)
    
    # Price change sentiment
    df['price_change_sentiment'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Liquidity factor
    df['liquidity'] = df['volume'] / (df['high'] - df['low'])
    
    # Sector-specific indicators (example: relative strength compared to a sector index)
    df['sector_index'] = df['close']  # Assume 'close' is the sector index for simplicity
    df['relative_strength'] = df['close'] / df['sector_index']
    
    # Enhanced factor combining dynamic weights, EMA-based trend strength, market sentiment, liquidity, and sector-specific indicators
    alpha_factor = (df['weight_log_return'] * df['adaptive_volatility'] +
                    df['weight_volume_scaled'] * df['ma_volume_scaled_price_change'] +
                    df['weight_trend_strength'] * df['ema_trend_strength'] +
                    df['market_sentiment'] * df['price_change_sentiment'] * df['liquidity'] +
                    df['relative_strength'])
    
    return alpha_factor
```
