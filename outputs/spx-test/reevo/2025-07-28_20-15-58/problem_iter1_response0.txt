```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then scale by the volume
    df['close_diff'] = df['close'].diff()
    df['volume_scaled_close_diff'] = df['close_diff'] * df['volume']
    
    # Calculate a 5-day moving average of the volume-scaled close difference
    df['ma_volume_scaled_close_diff'] = df['volume_scaled_close_diff'].rolling(window=5).mean()
    
    # Calculate the ratio of the current day's high to the 5-day moving average of the highs
    df['high_ma_high'] = df['high'] / df['high'].rolling(window=5).mean()
    
    # Combine the two factors to create the final alpha factor
    alpha_factor = df['ma_volume_scaled_close_diff'] * df['high_ma_high']
    
    return alpha_factor
```
