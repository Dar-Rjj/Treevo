```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 5-day rolling standard deviation of the returns to use as a volatility measure
    df['volatility'] = df['daily_return'].rolling(window=5).std()
    
    # Calculate the volume-weighted price change
    df['volume_scaled_price_change'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate the 5-day moving average of the volume-scaled price change
    df['ma_volume_scaled_price_change'] = df['volume_scaled_price_change'].rolling(window=5).mean()
    
    # Calculate the 10-day and 30-day moving averages of the closing price
    df['ma_close_10'] = df['close'].rolling(window=10).mean()
    df['ma_close_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the current day's close to the 10-day moving average of the closing price
    df['close_ma_ratio'] = df['close'] / df['ma_close_10']
    
    # Integrate trend strength by calculating the difference between the 10-day and 30-day moving averages of the closing prices
    df['trend_strength'] = df['ma_close_10'] - df['ma_close_30']
    
    # Calculate the liquidity as the average volume over the last 5 days
    df['liquidity'] = df['volume'].rolling(window=5).mean()
    
    # Adaptive window for momentum: 50-day and 200-day moving averages
    df['ma_close_50'] = df['close'].rolling(window=50).mean()
    df['ma_close_200'] = df['close'].rolling(window=200).mean()
    
    # Momentum factor: difference between 50-day and 200-day moving averages
    df['momentum_factor'] = df['ma_close_50'] - df['ma_close_200']
    
    # Integrate market sentiment: positive if the 50-day MA is above the 200-day MA, negative otherwise
    df['market_sentiment'] = df['ma_close_50'] > df['ma_close_200']
    df['market_sentiment'] = df['market_sentiment'].astype(int)
    
    # Combine the factors: volatility, volume-scaled price change, close-to-MA ratio, trend strength, and market sentiment
    alpha_factor = (df['volatility'] + df['ma_volume_scaled_price_change'] * df['close_ma_ratio'] + df['trend_strength'] + df['momentum_factor'] * df['market_sentiment'])
    
    # Normalize by liquidity
    alpha_factor = alpha_factor / df['liquidity']
    
    return alpha_factor
```
