def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and the average of the previous 5 days' closes
    avg_5day_close = df['close'].rolling(window=5).mean()
    diff_from_avg = df['close'] - avg_5day_close

    # Calculate the volume-weighted price change
    price_change = df['close'] - df['open']
    volume_weighted_price_change = price_change * df['volume']

    # Alpha factor: ratio of the volume-weighted price change to the standard deviation of the last 5 days' price changes
    std_5day_price_change = price_change.rolling(window=5).std(ddof=0)
    alpha_factor = volume_weighted_price_change / (std_5day_price_change + 1e-7)

    return alpha_factor
