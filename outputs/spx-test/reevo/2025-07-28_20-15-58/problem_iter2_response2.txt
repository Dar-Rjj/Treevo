```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    avg_5day_close = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between today's close and the 5-day moving average
    diff_from_avg = df['close'] - avg_5day_close
    
    # Calculate the volume-weighted price change
    price_change = df['close'] - df['open']
    volume_weighted_price_change = price_change * df['volume']
    
    # Calculate the 5-day standard deviation of the price change
    std_5day_price_change = price_change.rolling(window=5).std(ddof=0)
    
    # Calculate the 5-day moving average of the volume
    avg_5day_volume = df['volume'].rolling(window=5).mean()
    
    # Alpha factor: ratio of the volume-weighted price change to the product of the standard deviation of the last 5 days' price changes and the 5-day average volume
    alpha_factor = volume_weighted_price_change / ((std_5day_price_change + 1e-7) * (avg_5day_volume + 1e-7))
    
    # Incorporate the difference from the 5-day moving average as a component of the alpha factor
    alpha_factor += diff_from_avg / (std_5day_price_change + 1e-7)
    
    return alpha_factor
```
