```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day moving average of the closing price
    df['ma_3_close'] = df['close'].rolling(window=3).mean()
    
    # Calculate the difference between the closing price and the 3-day moving average
    df['close_ma_diff'] = df['close'] - df['ma_3_close']
    
    # Calculate the standard deviation of the past 3 days' closing prices
    df['std_3_close'] = df['close'].rolling(window=3).std()
    
    # The factor is the normalized difference between the close and 3-day MA, weighted by the inverse of the 3-day std
    alpha_factor = df['close_ma_diff'] / (df['std_3_close'] + 1e-7)
    
    # Weight the factor by the volume to account for liquidity
    alpha_factor_weighted = alpha_factor * (df['volume'] / df['volume'].rolling(window=3).mean())
    
    return alpha_factor_weighted
```
