```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 5-day and 20-day rolling standard deviation of the returns to use as a volatility measure
    df['volatility_5'] = df['daily_return'].rolling(window=5).std()
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    
    # Calculate the volume-weighted price change
    df['volume_scaled_price_change'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate the 5-day and 20-day moving average of the volume-scaled price change
    df['ma_volume_scaled_price_change_5'] = df['volume_scaled_price_change'].rolling(window=5).mean()
    df['ma_volume_scaled_price_change_20'] = df['volume_scaled_price_change'].rolling(window=20).mean()
    
    # Calculate the 10-day and 30-day moving average of the closing price
    df['ma_close_10'] = df['close'].rolling(window=10).mean()
    df['ma_close_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the current day's close to the 10-day and 30-day moving average of the closing price
    df['close_ma_ratio_10'] = df['close'] / df['ma_close_10']
    df['close_ma_ratio_30'] = df['close'] / df['ma_close_30']
    
    # Integrate trend strength by calculating the difference between the 10-day and 30-day moving averages of the closing prices
    df['trend_strength'] = df['ma_close_10'] - df['ma_close_30']
    
    # Calculate the momentum factor using the 10-day and 30-day returns
    df['momentum_10'] = df['close'].pct_change(periods=10)
    df['momentum_30'] = df['close'].pct_change(periods=30)
    
    # Calculate the liquidity factor using the 20-day average volume
    df['liquidity'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the mean reversion factor using the 5-day and 20-day returns
    df['mean_reversion_5'] = (df['close'] - df['close'].rolling(window=5).mean()) / df['close'].rolling(window=5).std()
    df['mean_reversion_20'] = (df['close'] - df['close'].rolling(window=20).mean()) / df['close'].rolling(window=20).std()
    
    # Dynamic weighting based on volatility
    weight_volatility = df['volatility_5'] / (df['volatility_5'] + df['volatility_20'] + 1e-7)
    weight_momentum = df['momentum_10'] / (df['momentum_10'] + df['momentum_30'] + 1e-7)
    weight_liquidity = df['liquidity'] / (df['liquidity'].max() + 1e-7)
    weight_mean_reversion = (df['mean_reversion_5'] + df['mean_reversion_20']) / 2
    
    # Combine the factors with dynamic weighting
    alpha_factor = (
        weight_volatility * (df['volatility_5'] + df['volatility_20']) +
        weight_momentum * (df['momentum_10'] + df['momentum_30']) +
        weight_liquidity * (df['ma_volume_scaled_price_change_5'] * df['close_ma_ratio_10'] + df['ma_volume_scaled_price_change_20'] * df['close_ma_ratio_30']) +
        weight_mean_reversion * df['trend_strength']
    )
    
    return alpha_factor
```
