```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and previous day's close
    df['close_diff'] = df['close'].diff()
    # Calculate the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    # Alpha factor: (Today's close - Yesterday's close) / 5-day average volume
    return df['close_diff'] / (df['avg_volume_5d'] + 1e-7)
```
