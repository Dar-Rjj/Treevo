```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a momentum factor with price change and volume change
    df['price_momentum'] = df['close'].pct_change()
    df['volume_momentum'] = df['volume'].pct_change()
    
    # Combine the two momentum indicators
    factor = df['price_momentum'] * df['volume_momentum']
    return factor
```
