```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then smooth it with a 5-day moving average
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    smoothed_return = daily_return.rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the 20-day moving average of volume, then apply exponential smoothing
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume
    exp_smoothed_vol = volume_ratio.ewm(span=7).mean()
    
    # Combine the smoothed daily return and exponentially smoothed volume ratio
    factor_value = smoothed_return * exp_smoothed_vol
    
    return factor_value
```
