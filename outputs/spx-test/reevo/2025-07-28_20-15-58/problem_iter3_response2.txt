```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then scale by the volume
    df['close_diff'] = df['close'].diff()
    df['volume_scaled_close_diff'] = df['close_diff'] * df['volume']
    
    # Calculate a 5-day moving average of the volume-scaled close difference
    df['ma_volume_scaled_close_diff_5'] = df['volume_scaled_close_diff'].rolling(window=5).mean()
    
    # Calculate the ratio of the current day's high to the 5-day moving average of the highs
    df['high_ma_high_5'] = df['high'] / df['high'].rolling(window=5).mean()
    
    # Calculate a 20-day moving average of the volume-scaled close difference for longer-term trend
    df['ma_volume_scaled_close_diff_20'] = df['volume_scaled_close_diff'].rolling(window=20).mean()
    
    # Calculate the ratio of the current day's high to the 20-day moving average of the highs
    df['high_ma_high_20'] = df['high'] / df['high'].rolling(window=20).mean()
    
    # Integrate the 5-day and 20-day factors to create a more robust alpha factor
    alpha_factor = (df['ma_volume_scaled_close_diff_5'] * df['high_ma_high_5']) + (df['ma_volume_scaled_close_diff_20'] * df['high_ma_high_20'])
    
    return alpha_factor
```
