```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    price_diff = df['close'] - df['close'].shift(1)
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change ratio over the average price
    price_change_ratio = price_diff / (avg_price + 1e-7)
    
    # Calculate the weighted volume, emphasizing the volume when the closing price is greater than the opening price
    weighted_volume = df['volume'] * (df['close'] > df['open'])
    
    # Calculate the 5-day momentum of the close price
    momentum_5d = df['close'] - df['close'].shift(5)
    
    # Calculate the volatility as the standard deviation of the last 20 days' price changes
    volatility_20d = price_diff.rolling(window=20).std()
    
    # Generate the final factor by combining the price change ratio, weighted volume, 5-day momentum, and 20-day volatility
    factor = (price_change_ratio * weighted_volume) + (momentum_5d / (volatility_20d + 1e-7))
    
    return factor
```
