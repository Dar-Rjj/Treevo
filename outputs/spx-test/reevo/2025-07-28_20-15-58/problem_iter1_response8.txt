```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices, representing the price movement during the day.
    price_movement = df['close'] - df['open']
    
    # Calculate average true range (a measure of volatility) over the last 14 days as a proxy for market interest or noise.
    atr = ta.volatility.AverageTrueRange(df['high'], df['low'], df['close'], window=14).average_true_range()
    
    # Generate a factor that reflects the relative importance of daily price movements compared to recent volatility,
    # capturing momentum in low-volatility environments and mean reversion signals in high-volatility scenarios.
    factor = price_movement / (atr + 1e-7)
    
    return factor
```
