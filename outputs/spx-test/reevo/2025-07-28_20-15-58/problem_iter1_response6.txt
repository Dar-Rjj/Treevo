```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average trading volume over the last 5 days
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    
    # Alpha factor: ratio of close price change to average volume
    # This factor captures the relationship between price momentum and average trading activity
    alpha_factor = df['close_diff'] / (df['avg_vol_5d'] + 1e-7)
    
    return alpha_factor
```
