```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then scale by the volume
    df['close_diff'] = df['close'].diff()
    df['volume_scaled_close_diff'] = df['close_diff'] * df['volume']
    
    # Adaptive window for moving average based on ATR (Average True Range)
    atr_window = 14
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs().max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=atr_window).mean()
    adaptive_window = (df['atr_14'] / df['true_range']).astype(int)
    adaptive_window = adaptive_window.clip(lower=5, upper=20)  # Clip to a reasonable range

    # Calculate a moving average of the volume-scaled close difference with adaptive window
    df['ma_volume_scaled_close_diff'] = df['volume_scaled_close_diff'].rolling(window=adaptive_window, min_periods=5).mean()
    
    # Calculate the ratio of the current day's high to the adaptive moving average of the highs
    df['high_ma_high'] = df['high'] / df['high'].rolling(window=adaptive_window, min_periods=5).mean()
    
    # Calculate the ratio of the current day's low to the adaptive moving average of the lows
    df['low_ma_low'] = df['low'] / df['low'].rolling(window=adaptive_window, min_periods=5).mean()
    
    # Calculate the 5-day volatility (standard deviation of closing prices)
    df['std_5_close'] = df['close'].rolling(window=5).std()
    
    # Calculate relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the market sentiment as the ratio of the close price to the open price
    df['market_sentiment'] = df['close'] / df['open']
    
    # Momentum factor: 10-day and 50-day moving averages
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    df['ma_50_close'] = df['close'].rolling(window=50).mean()
    df['momentum_factor'] = (df['ma_10_close'] - df['ma_50_close']) / df['std_5_close']
    
    # Liquidity factor: Volume ratio (current volume / 5-day average volume)
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Market breadth: Ratio of the number of advancing issues to declining issues
    df['advancing_issues'] = df['close'] > df['close'].shift(1)
    df['declining_issues'] = df['close'] < df['close'].shift(1)
    df['market_breadth'] = df['advancing_issues'].rolling(window=5).sum() / (df['declining_issues'].rolling(window=5).sum() + 1e-7)
    
    # Combine the factors to create the final alpha factor
    alpha_factor = (df['ma_volume_scaled_close_diff'] 
                    * df['high_ma_high'] 
                    * df['low_ma_low'] 
                    / (df['std_5_close'] + 1e-7)
                    * df['rsi'] 
                    * df['market_sentiment']
                    * df['momentum_factor']
                    * df['volume_ratio']
                    * df['market_breadth'])
    
    return alpha_factor
```
