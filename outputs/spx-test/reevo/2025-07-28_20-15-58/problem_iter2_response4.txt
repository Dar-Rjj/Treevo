```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 5-day standard deviation of daily returns as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=5).std(ddof=0)
    
    # Calculate the 10-day moving average of the close price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the trend strength by taking the difference between the current close and the 10-day moving average
    df['trend_strength'] = df['close'] - df['ma_10_close']
    
    # Calculate the volume-weighted price change
    df['price_change'] = df['close'] - df['open']
    df['volume_weighted_price_change'] = df['price_change'] * df['volume']
    
    # Combine the factors to create the final alpha factor
    df['alpha_factor'] = (df['trend_strength'] * df['volume_weighted_price_change']) / (df['volatility'] + 1e-7)
    
    return df['alpha_factor']
```
