```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the logarithmic daily return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the 5-day rolling standard deviation of the log returns to use as a volatility measure
    df['volatility'] = df['log_return'].rolling(window=5).std()
    
    # Calculate the volume-weighted price change
    df['volume_scaled_price_change'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate the 5-day exponential moving average of the volume-scaled price change
    df['ema_volume_scaled_price_change'] = df['volume_scaled_price_change'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 10-day and 30-day moving averages of the closing price
    df['ma_close_10'] = df['close'].rolling(window=10).mean()
    df['ma_close_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the current day's close to the 10-day moving average of the closing price
    df['close_ma_ratio'] = df['close'] / df['ma_close_10']
    
    # Integrate trend strength by calculating the difference between the 10-day and 30-day moving averages of the closing prices
    df['trend_strength'] = df['ma_close_10'] - df['ma_close_30']
    
    # Calculate the 5-day momentum using log returns
    df['momentum_5'] = df['log_return'].rolling(window=5).sum()
    
    # Adaptive volatility based on the 5-day momentum
    df['adaptive_volatility'] = df['volatility'] * (1 + df['momentum_5'].abs())
    
    # Dynamic weighting for the factors
    df['weight_log_return'] = df['log_return'].abs().rolling(window=5).mean()
    df['weight_volume_scaled'] = df['volume_scaled_price_change'].abs().rolling(window=5).mean()
    df['weight_trend_strength'] = df['trend_strength'].abs().rolling(window=5).mean()
    
    # Additional factor: 5-day EMA of the high price
    df['ema_high_5'] = df['high'].ewm(span=5, adjust=False).mean()
    
    # Additional factor: 5-day EMA of the low price
    df['ema_low_5'] = df['low'].ewm(span=5, adjust=False).mean()
    
    # Relative strength factor
    df['relative_strength'] = df['close'] / df['close'].rolling(window=14).min()
    
    # Market sentiment factor: difference between 5-day EMA of high and low
    df['market_sentiment'] = df['ema_high_5'] - df['ema_low_5']
    
    # Combine the factors with dynamic weights
    alpha_factor = (df['weight_log_return'] * df['adaptive_volatility'] +
                    df['weight_volume_scaled'] * df['ema_volume_scaled_price_change'] * df['close_ma_ratio'] +
                    df['weight_trend_strength'] * df['trend_strength'] +
                    df['weight_log_return'] * df['market_sentiment'] +
                    df['relative_strength'])
    
    return alpha_factor
```
