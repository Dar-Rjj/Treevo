```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()

    # Momentum factor: 5-day and 30-day momentum
    df['momentum_5'] = df['close'] / df['close'].shift(5) - 1
    df['momentum_30'] = df['close'] / df['close'].shift(30) - 1

    # Volatility factor: 5-day rolling standard deviation of the returns
    df['volatility'] = df['daily_return'].rolling(window=5).std()

    # Volume-scaled price change
    df['volume_scaled_price_change'] = (df['close'] - df['open']) * df['volume']
    
    # Adaptive window for moving average based on volatility
    df['ma_window'] = (df['volatility'] * 100).round().clip(5, 30)
    df['ma_close_adaptive'] = df.apply(lambda row: df['close'].rolling(window=int(row['ma_window']), min_periods=1).mean(), axis=1)

    # Close-to-MA ratio using adaptive window
    df['close_ma_ratio'] = df['close'] / df['ma_close_adaptive']

    # Trend strength: difference between 10-day and 30-day moving averages of the closing prices
    df['trend_strength'] = df['close'].rolling(window=10).mean() - df['close'].rolling(window=30).mean()

    # Liquidity factor: volume ratio to 20-day moving average of volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / (df['avg_volume_20'] + 1e-7)

    # Sentiment factor: ratio of today's close to the high-low range
    df['sentiment'] = df['close'] / (df['high'] - df['low'])

    # Combine the factors: momentum, volatility, volume-scaled price change, close-to-MA ratio, trend strength, liquidity, and sentiment
    alpha_factor = (
        df['momentum_5'] + 
        df['momentum_30'] + 
        df['volatility'] + 
        df['volume_scaled_price_change'] * df['close_ma_ratio'] + 
        df['trend_strength'] + 
        df['volume_ratio'] + 
        df['sentiment']
    )

    return alpha_factor
```
