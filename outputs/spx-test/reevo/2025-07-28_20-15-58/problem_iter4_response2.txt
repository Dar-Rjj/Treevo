```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 2-day moving average of the closing price
    df['ma_2_close'] = df['close'].rolling(window=2).mean()
    
    # Calculate the difference between the closing price and the 2-day moving average
    df['close_ma_diff'] = df['close'] - df['ma_2_close']
    
    # Calculate the standard deviation of the past 2 days' closing prices
    df['std_2_close'] = df['close'].rolling(window=2).std()
    
    # The factor is the difference between the close and 2-day MA, weighted by the inverse of the 2-day std
    alpha_factor = df['close_ma_diff'] / (df['std_2_close'] + 1e-7)
    
    # Weight the factor by the volume to account for liquidity
    alpha_factor_weighted = alpha_factor * (df['volume'] / df['volume'].rolling(window=2).mean())
    
    return alpha_factor_weighted
```
