```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day simple moving average of close prices
    df['ma5_close'] = df['close'].rolling(window=5).mean()
    df['ma20_close'] = df['close'].rolling(window=20).mean()

    # Determine the trend strength by comparing the two moving averages
    df['trend_strength'] = (df['ma5_close'] - df['ma20_close']) / df['ma20_close']

    # Compute the average true range (ATR) as a measure of volatility
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['atr5'] = df['true_range'].rolling(window=5).mean()

    # Scale the close price difference by volume and ATR
    df['volume_scaled_close_diff'] = df['close'].diff() * df['volume'] / df['atr5']

    # Calculate the 5-day moving average of the scaled close difference
    df['ma_volume_scaled_close_diff'] = df['volume_scaled_close_diff'].rolling(window=5).mean()

    # Integrate the trend strength with the volume-scaled close difference
    alpha_factor = df['ma_volume_scaled_close_diff'] * df['trend_strength']

    return alpha_factor
```
