```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the 5-day moving average of the closing price
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    df['close_ma_diff'] = df['close'] - df['ma_5_close']
    
    # Calculate the standard deviation of the past 5 days' closing prices
    df['std_5_close'] = df['close'].rolling(window=5).std()
    
    # The factor is the normalized difference between the close and 5-day MA, weighted by the inverse of the 5-day std
    alpha_factor = df['close_ma_diff'] / (df['std_5_close'] + 1e-7)
    
    return alpha_factor
```
