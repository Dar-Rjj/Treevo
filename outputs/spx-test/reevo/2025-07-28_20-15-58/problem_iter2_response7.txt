def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and the average of the previous 5 days' closes
    avg_5day_close = df['close'].rolling(window=5).mean()
    diff_from_avg = df['close'] - avg_5day_close

    # Calculate the volume-weighted price change
    price_change = df['close'] - df['open']
    volume_weighted_price_change = price_change * df['volume']

    # Calculate the standard deviation of the last 5 days' price changes
    std_5day_price_change = price_change.rolling(window=5).std(ddof=0)

    # Calculate the ratio of today's volume to the 20-day moving average of volume
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / (avg_volume + 1e-7)

    # Combine the volume-weighted price change, volatility, and recent price trends
    alpha_factor = (volume_weighted_price_change / (std_5day_price_change + 1e-7)) * diff_from_avg * volume_ratio

    return alpha_factor
