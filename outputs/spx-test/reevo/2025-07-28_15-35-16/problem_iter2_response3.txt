```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and previous close price
    df['price_diff'] = df['close'].diff().fillna(0)
    
    # Calculate the weighted sum of price movement using volume as a weight
    df['volume_weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate a relative strength factor, by dividing the cumulative moving average of volume-weighted price movement
    # with a moving standard deviation of the close prices. This factor aims to capture recent momentum.
    df['rel_strength'] = df['volume_weighted_price_movement'].rolling(window=15).mean() / (df['close'].rolling(window=15).std() + 1e-7)
    
    # Factor to consider the impact of gaps in the trading (difference between this day's open and last day's close)
    df['gap_factor'] = abs(df['open'] - df['close'].shift(1))
    
    # A combined factor that looks at both the gap and the relative strength
    df['combined_factor'] = df['gap_factor'] * df['rel_strength']
    
    # Incorporate volatility: use the true range as a measure of daily volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'], x['close'].shift(1)) - min(x['low'], x['close'].shift(1)), axis=1)
    df['volatility_factor'] = df['true_range'].rolling(window=10).mean()
    
    # Final alpha factor: combine the gap, relative strength, and volatility factors
    df['alpha_factor'] = df['combined_factor'] / (df['volatility_factor'] + 1e-7)

    return df['alpha_factor']
```
