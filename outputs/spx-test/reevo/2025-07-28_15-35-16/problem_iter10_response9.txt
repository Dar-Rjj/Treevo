def heuristics_v2(df: pd.DataFrame, macroeconomic_indicators: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = df['close'].pct_change(periods=2).rolling(window=5, min_periods=1).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Exponential smoothing of the RSI
    rsi_smoothed = rsi.ewm(span=10, adjust=False, min_periods=1).mean()
    
    # Add a market sentiment factor (e.g., based on a simple moving average of the close price)
    market_sentiment = df['close'].rolling(window=20, min_periods=1).mean() - close_price
    
    # Prepare features for dynamic weighting using machine learning
    features = pd.concat([price_diff, momentum_factor, vwap, rsi_smoothed, market_sentiment, macroeconomic_indicators], axis=1)
    features.columns = ['price_diff', 'momentum_factor', 'vwap', 'rsi_smoothed', 'market_sentiment', 'macro_indicator']
    
    # Use a simple linear model for adaptive factor selection (this can be replaced with more sophisticated models)
    from sklearn.linear_model import LinearRegression
    model = LinearRegression()
    X = features.shift(1).dropna()
    y = close_price.reindex(X.index).diff().shift(-1)
    model.fit(X, y)
    weights = pd.Series(model.coef_, index=X.columns)
    
    # Normalize the weights to sum to 1
    weights /= weights.sum()
    
    # Integrate the factors into a new alpha factor with dynamic weights
    alpha_factor = (weights['price_diff'] * price_diff) + (weights['momentum_factor'] * momentum_factor) + \
                   (weights['vwap'] * (vwap - close_price)) + (weights['rsi_smoothed'] * (rsi_smoothed - 50)) + \
                   (weights['market_sentiment'] * market_sentiment) + (weights['macro_indicator'] * macroeconomic_indicators)
    
    return alpha_factor
