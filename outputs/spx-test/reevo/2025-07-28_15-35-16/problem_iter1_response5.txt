```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate moving averages for different periods
    ma_short = df['close'].rolling(window=5).mean()
    ma_long = df['close'].rolling(window=20).mean()
    
    # Calculate the momentum of the close price
    momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate the difference between moving averages and add a momentum bias
    alpha_factor = (ma_short - ma_long) * momentum
    
    return alpha_factor
```
