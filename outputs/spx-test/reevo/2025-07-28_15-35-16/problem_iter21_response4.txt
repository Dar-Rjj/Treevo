def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    momentum_factor = close_price.pct_change(periods=2).rolling(window=5, min_periods=1).mean()
    
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    rsi_smoothed = rsi.ewm(span=10, adjust=False, min_periods=1).mean()
    
    volatility = close_price.rolling(window=20, min_periods=1).std()
    adaptive_weight = 0.1 + 0.9 * (volatility / volatility.mean())
    
    alpha_factor = (adaptive_weight * 0.4 * price_diff) + \
                   (adaptive_weight * 0.3 * momentum_factor) + \
                   (adaptive_weight * 0.2 * (vwap - close_price)) + \
                   (adaptive_weight * 0.1 * (rsi_smoothed - 50))
    
    market_sentiment = close_price.rolling(window=20, min_periods=1).mean() - close_price
    alpha_factor += (adaptive_weight * 0.1 * market_sentiment)
    
    macroeconomic_factor = close_price.rolling(window=10, min_periods=1).mean() - close_price.rolling(window=30, min_periods=1).mean()
    alpha_factor += (adaptive_weight * 0.05 * macroeconomic_factor)
    
    trend_factor = close_price.rolling(window=50, min_periods=1).mean() - close_price
    alpha_factor += (adaptive_weight * 0.05 * trend_factor)
    
    open_close_diff = df['open'] - df['close']
    alpha_factor += (adaptive_weight * 0.05 * open_close_diff)
    
    price_velocity = (close_price - close_price.shift(1)).rolling(window=20, min_periods=1).mean()
    alpha_factor += (adaptive_weight * 0.05 * price_velocity)
    
    volume_change = df['volume'].pct_change(periods=5).rolling(window=5, min_periods=1).mean()
    alpha_factor += (adaptive_weight * 0.05 * volume_change)
    
    sector_indicator = (close_price.rolling(window=60, min_periods=1).mean() - close_price.rolling(window=120, min_periods=1).mean())
    alpha_factor += (adaptive_weight * 0.05 * sector_indicator)
    
    # Add a new factor based on the range of high and low prices
    range_factor = (df['high'] - df['low']).rolling(window=20, min_periods=1).mean()
    alpha_factor += (adaptive_weight * 0.05 * range_factor)
    
    return alpha_factor
