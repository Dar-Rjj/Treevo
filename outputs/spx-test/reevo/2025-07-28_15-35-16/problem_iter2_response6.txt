```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average of the close price over 10 and 30 days
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the difference between today's close and the 10-day EMA
    df['close_diff_ema_10'] = df['close'] - df['ema_10']
    
    # Calculate the ratio of the 10-day EMA to the 30-day EMA
    df['ema_ratio'] = df['ema_10'] / df['ema_30']
    
    # Calculate the exponential moving standard deviation of the close price over 15 days
    df['ema_std_15'] = df['close'].ewm(span=15, adjust=False).std()
    
    # Calculate a volatility factor by normalizing the close price difference with the 15-day EMA standard deviation
    df['volatility_factor'] = df['close_diff_ema_10'] / (df['ema_std_15'] + 1e-7)
    
    # Calculate the 10-day exponential moving average of volume
    df['volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the gap factor as the absolute difference between today's open and yesterday's close
    df['gap_factor'] = abs(df['open'] - df['close'].shift(1))
    
    # A combined factor that incorporates the EMA ratio, volatility factor, and gap factor
    alpha_factor = df['ema_ratio'] * df['volatility_factor'] * df['gap_factor'] / (df['volume_ema_10'] + 1e-7)
    
    return alpha_factor
```
