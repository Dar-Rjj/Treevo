```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 1-day momentum as the difference between today's close and yesterday's close
    df['momentum_1d'] = df['close'] - df['close'].shift(1)
    
    # Calculate the 5-day average volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day volatility (standard deviation of returns)
    df['volatility_5d'] = df['close'].pct_change().rolling(window=5).std()
    
    # Calculate the liquidity as the ratio of volume to volatility
    df['liquidity'] = df['volume'] / df['volatility_5d']
    
    # Alpha factor: (Momentum / 5-Day Volatility) * (High / Low) * Liquidity
    alpha_factor = (df['momentum_1d'] / df['volatility_5d']) * (df['high'] / df['low']) * df['liquidity']
    return alpha_factor
```
