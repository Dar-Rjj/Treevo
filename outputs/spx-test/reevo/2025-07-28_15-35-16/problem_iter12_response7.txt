```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = df['close'].pct_change(periods=2).rolling(window=5, min_periods=1).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Exponential smoothing of the RSI
    rsi_smoothed = rsi.ewm(span=10, adjust=False, min_periods=1).mean()
    
    # Dynamic adaptive weighting of the factors
    rolling_volatility = df['close'].rolling(window=20, min_periods=1).std()
    weight_price_diff = 0.4 + 0.1 * (rolling_volatility / rolling_volatility.max())
    weight_momentum = 0.3 - 0.1 * (rolling_volatility / rolling_volatility.max())
    weight_vwap = 0.15
    weight_rsi = 0.1
    weight_market_volatility = 0.05
    
    # Integrate the price difference, smoothed momentum, VWAP, and smoothed RSI into a new alpha factor with dynamic weights
    alpha_factor = (weight_price_diff * price_diff) + (weight_momentum * momentum_factor) + \
                   (weight_vwap * (vwap - close_price)) + (weight_rsi * (rsi_smoothed - 50))
    
    # Add a market sentiment factor (e.g., based on a simple moving average of the close price)
    market_sentiment = df['close'].rolling(window=20, min_periods=1).mean() - close_price
    weight_market_sentiment = 0.1
    
    # Incorporate the market sentiment factor into the alpha factor
    alpha_factor = (alpha_factor * 0.9) + (weight_market_sentiment * market_sentiment)
    
    # Add a market volatility factor (e.g., based on the exponential moving average of the standard deviation of the close price)
    market_volatility = df['close'].rolling(window=20, min_periods=1).std().ewm(span=10, adjust=False, min_periods=1).mean()
    alpha_factor += (weight_market_volatility * market_volatility)
    
    return alpha_factor
```
