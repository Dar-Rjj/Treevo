```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, which represents the price range of the day
    price_range = df['high'] - df['low']
    
    # Calculate the closing price as a percentage of the day's high
    close_to_high = (df['close'] - df['low']) / (price_range + 1e-7)
    
    # Calculate momentum indicators by looking at the change in closing price over different timeframes
    momentum_5d = df['close'].pct_change(periods=5).fillna(0)
    momentum_14d = df['close'].pct_change(periods=14).fillna(0)
    momentum_28d = df['close'].pct_change(periods=28).fillna(0)
    
    # Calculate trend indicators based on moving averages
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    
    trend_5_20 = (ma_5 - ma_20) / ma_20
    trend_20_50 = (ma_20 - ma_50) / ma_50
    
    # Integrate volatility using the standard deviation of the last 20 days' closing prices
    volatility = df['close'].rolling(window=20).std().fillna(0)
    
    # Incorporate trading volume by calculating the percentage change in volume over different timeframes
    volume_momentum_1d = df['volume'].pct_change(periods=1).fillna(0)
    volume_momentum_5d = df['volume'].pct_change(periods=5).fillna(0)
    volume_momentum_20d = df['volume'].pct_change(periods=20).fillna(0)
    
    # Combine momentum, trend, and volume signals into a refined momentum factor
    refined_momentum = (close_to_high + momentum_5d + momentum_14d + momentum_28d + volume_momentum_1d + volume_momentum_5d + volume_momentum_20d) / 7
    
    # Combine these signals into an alpha factor, normalized by volatility
    alpha_factor = (refined_momentum * (trend_5_20 + trend_20_50) / 2) / (volatility + 1e-7)
    
    return alpha_factor
```
