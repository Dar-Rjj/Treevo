```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average trading volume over a 10-day window
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the ratio of current volume to the 10-day average volume
    volume_ratio = df['volume'] / avg_volume_10d
    
    # The factor is designed to capture the price movement combined with the relative volume
    factor = diff_close_open * volume_ratio
    
    return factor
```
