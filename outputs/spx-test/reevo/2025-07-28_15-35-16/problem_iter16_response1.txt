```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = df['close'].pct_change(periods=2).rolling(window=5, min_periods=1).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Exponential smoothing of the RSI
    rsi_smoothed = rsi.ewm(span=10, adjust=False, min_periods=1).mean()
    
    # Determine the market trend
    market_trend = close_price.rolling(window=60, min_periods=1).mean().pct_change()
    
    # Define dynamic weights based on market trend
    weight_price_diff = 0.4 * (1 + 0.1 * market_trend)
    weight_momentum = 0.3 * (1 + 0.1 * market_trend)
    weight_vwap = 0.15 * (1 + 0.1 * market_trend)
    weight_rsi = 0.15 * (1 + 0.1 * market_trend)
    
    # Integrate the key factors into a new alpha factor with dynamic weights
    alpha_factor = (weight_price_diff * price_diff) + (weight_momentum * momentum_factor) + \
                   (weight_vwap * (vwap - close_price)) + (weight_rsi * (rsi_smoothed - 50))
    
    return alpha_factor
```
