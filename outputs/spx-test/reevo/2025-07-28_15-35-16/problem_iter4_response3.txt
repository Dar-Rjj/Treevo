```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a short-term momentum factor based on the change in the close price over 5 periods
    momentum_5d = close_price.pct_change(periods=5).fillna(0)
    
    # Compute a medium-term momentum factor based on the change in the close price over 20 periods
    momentum_20d = close_price.pct_change(periods=20).fillna(0)
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Integrate the price difference, short-term and medium-term momentum, VWAP, and RSI into a new alpha factor
    alpha_factor = (price_diff * momentum_5d) + (vwap - close_price) + (rsi - 50) + (momentum_20d * 0.5)
    
    return alpha_factor
```
