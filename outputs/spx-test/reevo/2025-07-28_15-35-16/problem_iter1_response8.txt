```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by today's range (high - low)
    df['price_change'] = df['close'].diff().fillna(0)
    factor = df['price_change'] / (df['high'] - df['low'] + 1e-7)
    return factor
```
