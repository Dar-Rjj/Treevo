```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then normalized by the 5-day average of the true range (high - low)
    df['true_range'] = df['high'] - df['low']
    avg_true_range_5 = df['true_range'].rolling(window=5).mean()
    momentum = (df['close'].diff(1)) / (avg_true_range_5 + 1e-7)
    return momentum
```
