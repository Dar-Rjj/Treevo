def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = close_price.pct_change(periods=2).rolling(window=5, min_periods=1).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Exponential smoothing of the RSI
    rsi_smoothed = rsi.ewm(span=10, adjust=False, min_periods=1).mean()
    
    # Adaptive weighting based on recent volatility
    volatility = close_price.rolling(window=20, min_periods=1).std()
    adaptive_weight = 0.1 + 0.9 * (volatility / volatility.mean())
    
    # Combine factors with dynamic weights
    alpha_factor = (adaptive_weight * 0.45 * price_diff) + \
                   (adaptive_weight * 0.35 * momentum_factor) + \
                   (adaptive_weight * 0.2 * (vwap - close_price)) + \
                   (adaptive_weight * 0.1 * (rsi_smoothed - 50))
    
    # Add a market sentiment factor (e.g., based on a simple moving average of the close price)
    market_sentiment = close_price.rolling(window=20, min_periods=1).mean() - close_price
    alpha_factor += (adaptive_weight * 0.1 * market_sentiment)
    
    # Simplify macroeconomic factor (e.g., based on the difference between the 10-day and 30-day moving averages of the close price)
    macroeconomic_factor = close_price.rolling(window=10, min_periods=1).mean() - close_price.rolling(window=30, min_periods=1).mean()
    alpha_factor += (adaptive_weight * 0.075 * macroeconomic_factor)
    
    # Introduce a trend-following factor using the 50-day moving average
    trend_factor = close_price.rolling(window=50, min_periods=1).mean() - close_price
    alpha_factor += (adaptive_weight * 0.075 * trend_factor)
    
    # Add a new factor based on the difference between the open and close price
    open_close_diff = df['open'] - df['close']
    alpha_factor += (adaptive_weight * 0.075 * open_close_diff)
    
    # Incorporate a price velocity factor
    price_velocity = (close_price - close_price.shift(1)).rolling(window=20, min_periods=1).mean()
    alpha_factor += (adaptive_weight * 0.075 * price_velocity)
    
    # Apply robust smoothing for stability and performance
    alpha_factor = alpha_factor.ewm(span=10, adjust=False, min_periods=1).mean()
    
    return alpha_factor
