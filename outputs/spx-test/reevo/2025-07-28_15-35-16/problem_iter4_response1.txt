```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change as a percentage of the previous day's close
    df['price_change_pct'] = (df['close'].diff().fillna(0)) / (df['close'].shift(1) + 1e-7)
    
    # Calculate the average true range (ATR) to use as a volatility measure
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the momentum by taking the 14-day return
    df['momentum_14'] = (df['close'] / df['close'].shift(14)).fillna(1)
    
    # Create a factor that combines the price change percentage, ATR, and momentum
    df['factor'] = (df['price_change_pct'] * df['momentum_14']) / (df['atr_14'] + 1e-7)

    return df['factor']
```
