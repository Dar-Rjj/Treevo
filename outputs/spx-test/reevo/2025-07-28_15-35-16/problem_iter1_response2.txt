```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the moving average of volume over the last 10 days
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    # Calculate a factor that is the ratio of the close price difference to the 10-day moving average of volume
    # This factor could be interpreted as the price change momentum relative to the trading volume
    alpha_factor = df['close_diff'] / (df['volume_ma_10'] + 1e-7)
    return alpha_factor
```
