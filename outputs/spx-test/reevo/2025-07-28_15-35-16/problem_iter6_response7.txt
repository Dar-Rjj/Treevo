```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, which represents the price range of the day
    price_range = df['high'] - df['low']
    
    # Calculate the closing price as a percentage of the day's high
    close_to_high = (df['close'] - df['low']) / (price_range + 1e-7)
    
    # Calculate a momentum indicator by looking at the change in closing price over the last 5 days
    momentum_5d = df['close'].pct_change(periods=5).fillna(0)
    
    # Calculate a trend indicator based on the moving average of the last 20 days vs. the last 5 days
    ma_20 = df['close'].rolling(window=20).mean()
    ma_5 = df['close'].rolling(window=5).mean()
    trend = (ma_5 - ma_20) / ma_20
    
    # Integrate volatility using the standard deviation of the last 20 days' closing prices
    volatility = df['close'].rolling(window=20).std().fillna(0)
    
    # Incorporate trading volume by calculating the percentage change in volume
    volume_momentum = df['volume'].pct_change(periods=1).fillna(0)
    
    # Refined momentum by combining close-to-high, 5-day momentum, and volume momentum
    refined_momentum = (close_to_high + momentum_5d + volume_momentum) / 3
    
    # Calculate the Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the Relative Strength Index (RSI) for the close price
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the On-Balance Volume (OBV) to incorporate trading volume
    obv = (df['volume'] * (df['close'] > df['close'].shift()).astype(int)).cumsum()
    
    # Calculate the Exponential Moving Average (EMA) of the close price
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the Aroon Up and Aroon Down indicators to identify the direction of the trend
    aroon_up = (df['high'].rolling(window=25).apply(lambda x: (x.argmax() - x.index[-1]) / 24, raw=True) * 100).fillna(0)
    aroon_down = (df['low'].rolling(window=25).apply(lambda x: (x.argmin() - x.index[-1]) / 24, raw=True) * 100).fillna(0)
    aroon_oscillator = aroon_up - aroon_down
    
    # Dynamic weighting based on market context
    weight_momentum = (refined_momentum + aroon_oscillator / 100) / 2
    weight_trend = trend
    weight_volatility = 1 / (volatility + 1e-7)
    weight_rsi = (rsi - 50) / 50
    weight_obv = (obv - obv.shift(1)) / obv.shift(1)
    
    # Combine these signals into an alpha factor with dynamic weighting
    alpha_factor = (weight_momentum * weight_trend * weight_volatility * weight_rsi * weight_obv) + (vwap - df['close'])
    
    return alpha_factor
```
