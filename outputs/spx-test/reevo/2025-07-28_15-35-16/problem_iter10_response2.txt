def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods with an adaptive window
    momentum_factor = close_price.pct_change(periods=2).rolling(window=5, min_periods=1).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14, min_periods=1).mean()
    avg_loss = loss.rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Exponential smoothing of the RSI
    rsi_smoothed = rsi.ewm(span=10, adjust=False).mean()
    
    # Calculate the rate of change (ROC) for the close price over a 10-period window
    roc = close_price.pct_change(periods=10).fillna(0)
    
    # Calculate the adaptive moving average convergence divergence (MACD)
    ema_12 = close_price.ewm(span=12, adjust=False).mean()
    ema_26 = close_price.ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_hist = macd_line - signal_line
    
    # Calculate the adaptive weighting of the factors based on their recent performance
    def adaptive_weight(factor_series):
        return factor_series.rolling(window=20, min_periods=1).corr(close_price)
    
    weight_price_diff = adaptive_weight(price_diff)
    weight_momentum = adaptive_weight(momentum_factor)
    weight_vwap = adaptive_weight(vwap)
    weight_rsi = adaptive_weight(rsi_smoothed)
    weight_roc = adaptive_weight(roc)
    weight_macd = adaptive_weight(macd_hist)
    
    # Normalize the weights to sum to 1
    total_weight = weight_price_diff + weight_momentum + weight_vwap + weight_rsi + weight_roc + weight_macd
    weight_price_diff /= total_weight
    weight_momentum /= total_weight
    weight_vwap /= total_weight
    weight_rsi /= total_weight
    weight_roc /= total_weight
    weight_macd /= total_weight
    
    # Integrate the price difference, momentum, VWAP, smoothed RSI, ROC, and MACD into a new alpha factor with adaptive weights
    alpha_factor = (weight_price_diff * price_diff) + (weight_momentum * momentum_factor) + \
                   (weight_vwap * (vwap - close_price)) + (weight_rsi * (rsi_smoothed - 50)) + \
                   (weight_roc * roc) + (weight_macd * macd_hist)
    
    return alpha_factor
