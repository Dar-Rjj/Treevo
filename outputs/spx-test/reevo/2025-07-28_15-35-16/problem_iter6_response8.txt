def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = df['close'].pct_change(periods=2)
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the volatility (standard deviation of the close price over a window)
    volatility = close_price.rolling(window=20).std()
    
    # Calculate the rolling correlation between close price and volume
    corr_close_volume = df[['close', 'volume']].rolling(window=50).corr().iloc[::2, -1]
    
    # Calculate the moving average convergence divergence (MACD)
    ema_12 = close_price.ewm(span=12, adjust=False).mean()
    ema_26 = close_price.ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.ewm(span=9, adjust=False).mean()
    macd_hist = macd - signal
    
    # Integrate the price difference, momentum, VWAP, RSI, volatility, MACD histogram, and rolling correlation into a new alpha factor
    alpha_factor = (
        (price_diff * momentum_factor) +
        (vwap - close_price) +
        (rsi - 50) +
        (volatility / close_price) +
        macd_hist +
        corr_close_volume
    )
    
    return alpha_factor
