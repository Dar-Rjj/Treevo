```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and 5-day moving average of closing price
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    df['diff_close_ma_5'] = df['close'] - df['ma_5_close']
    
    # Calculate the exponential moving volume with a span of 7 days
    df['ema_7_vol'] = df['volume'].ewm(span=7, adjust=False).mean()

    # Novel alpha factor: Ratio of the difference between close and MA(5 close) to EMA(7 volume)
    # This factor incorporates both price momentum (through comparison with a short-term moving average) 
    # and trading activity (through volume's exponential moving average), aiming to capture situations 
    # where increased trading activity accompanies a deviation from recent price trends.
    alpha_factor = df['diff_close_ma_5'] / (df['ema_7_vol'] + 1e-7)
    
    return alpha_factor
```
