```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by today's range (high - low)
    df['price_change'] = df['close'].diff().fillna(0)
    price_volatility_factor = df['price_change'] / (df['high'] - df['low'] + 1e-7)

    # Calculate the ratio of today's volume to the 10-day moving average of volume
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean().fillna(0)
    volume_ratio_factor = df['volume'] / (df['volume_ma_10'] + 1e-7)

    # Calculate the 10-day standard deviation of the close price to measure price volatility
    df['price_volatility_10'] = df['close'].rolling(window=10).std().fillna(0)

    # Combine the factors into a single alpha factor
    alpha_factor = (price_volatility_factor * volume_ratio_factor) / (df['price_volatility_10'] + 1e-7)
    return alpha_factor
```
