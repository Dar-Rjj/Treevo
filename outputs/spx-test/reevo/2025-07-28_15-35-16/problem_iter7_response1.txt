def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = df['close'].pct_change(periods=2).rolling(window=5).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Exponential smoothing of the RSI
    rsi_smoothed = rsi.ewm(span=10, adjust=False).mean()
    
    # Introduce a new feature: ratio of close to VWAP
    close_vwap_ratio = close_price / vwap
    
    # Dynamic weighting of the factors based on recent performance
    weight_price_diff = 0.4
    weight_momentum = 0.3
    weight_vwap = 0.2
    weight_rsi = 0.1
    weight_close_vwap_ratio = 0.1
    
    # Adjust weights dynamically based on recent performance
    recent_performance = (close_price.pct_change().rolling(window=5).mean()).rank(pct=True)
    weight_price_diff *= recent_performance
    weight_momentum *= 1 - recent_performance
    weight_vwap *= recent_performance
    weight_rsi *= 1 - recent_performance
    weight_close_vwap_ratio *= recent_performance
    
    # Integrate the price difference, smoothed momentum, VWAP, smoothed RSI, and close to VWAP ratio into a new alpha factor with dynamic weights
    alpha_factor = (weight_price_diff * price_diff) + (weight_momentum * momentum_factor) + \
                   (weight_vwap * (vwap - close_price)) + (weight_rsi * (rsi_smoothed - 50)) + \
                   (weight_close_vwap_ratio * (close_vwap_ratio - 1))
    
    return alpha_factor
