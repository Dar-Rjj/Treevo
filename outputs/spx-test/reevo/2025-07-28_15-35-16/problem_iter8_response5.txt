def heuristics_v2(df: pd.DataFrame, macroeconomic_data: pd.Series, sentiment_scores: pd.Series) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = df['close'].pct_change(periods=2)
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the exponential moving average (EMA) of the close price
    ema_close = close_price.ewm(span=5, adjust=False).mean()
    
    # Calculate the rate of change (ROC) for the close price
    roc = (close_price - close_price.shift(10)) / close_price.shift(10)
    
    # Calculate the Bollinger Bands
    bb_mean = close_price.rolling(window=20).mean()
    bb_std = close_price.rolling(window=20).std()
    upper_band = bb_mean + 2 * bb_std
    lower_band = bb_mean - 2 * bb_std
    bollinger_percentage = (close_price - lower_band) / (upper_band - lower_band)
    
    # Calculate the volatility
    volatility = close_price.rolling(window=20).std()
    
    # Calculate the trend strength using the Average Directional Index (ADX)
    plus_dm = df['high'].diff()
    minus_dm = df['low'].diff()
    plus_dm[plus_dm < 0] = 0
    minus_dm[minus_dm > 0] = 0
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    plus_di = 100 * (plus_dm.rolling(window=14).sum() / tr.rolling(window=14).sum())
    minus_di = 100 * (minus_dm.abs().rolling(window=14).sum() / tr.rolling(window=14).sum())
    dx = (plus_di - minus_di).abs() / (plus_di + minus_di)
    adx = dx.rolling(window=14).mean()
    
    # Calculate the mean reversion signal
    mean_reversion_signal = (close_price - close_price.rolling(window=20).mean()) / close_price.rolling(window=20).std()
    
    # Integrate macroeconomic indicators and sentiment scores into the alpha factor
    macroeconomic_indicator = macroeconomic_data
    sentiment_factor = sentiment_scores
    
    # Integrate all factors into a new alpha factor
    alpha_factor = (price_diff * momentum_factor) + (vwap - close_price) + (rsi - 50) + (ema_close - close_price) + \
                   roc + bollinger_percentage + (volatility / close_price) + adx + mean_reversion_signal + \
                   macroeconomic_indicator + sentiment_factor
    
    return alpha_factor
