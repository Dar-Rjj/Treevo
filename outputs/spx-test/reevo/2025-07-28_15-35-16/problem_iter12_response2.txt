def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods with an adaptive window
    momentum_window = 5
    momentum_factor = df['close'].pct_change(periods=2).rolling(window=momentum_window, min_periods=1).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price with an adaptive window
    rsi_window = 14
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=rsi_window, min_periods=1).mean()
    avg_loss = loss.rolling(window=rsi_window, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Exponential smoothing of the RSI with an adaptive window
    rsi_smooth_window = 10
    rsi_smoothed = rsi.ewm(span=rsi_smooth_window, adjust=False, min_periods=1).mean()
    
    # Calculate the rate of change (ROC) for the close price over a 10-period window
    roc_window = 10
    roc = close_price.pct_change(periods=roc_window).fillna(0)
    
    # Calculate the adaptive moving average convergence divergence (MACD)
    ema_12 = close_price.ewm(span=12, adjust=False, min_periods=1).mean()
    ema_26 = close_price.ewm(span=26, adjust=False, min_periods=1).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False, min_periods=1).mean()
    macd_hist = macd_line - signal_line
    
    # Calculate the volatility
    volatility_window = 20
    volatility = close_price.rolling(window=volatility_window, min_periods=1).std()
    
    # Adaptive weighting based on volatility
    weight_price_diff = 0.3 * (1 - (volatility / volatility.max()))
    weight_momentum = 0.2 * (1 - (volatility / volatility.max()))
    weight_vwap = 0.2 * (1 - (volatility / volatility.max()))
    weight_rsi = 0.1 * (1 - (volatility / volatility.max()))
    weight_macd = 0.1 * (1 - (volatility / volatility.max()))
    weight_roc = 0.1 * (1 - (volatility / volatility.max()))
    
    # Integrate the price difference, smoothed momentum, VWAP, smoothed RSI, ROC, and MACD into a new alpha factor with adaptive weights
    alpha_factor = (weight_price_diff * price_diff) + (weight_momentum * momentum_factor) + \
                   (weight_vwap * (vwap - close_price)) + (weight_rsi * (rsi_smoothed - 50)) + \
                   (weight_roc * roc) + (weight_macd * macd_hist)
    
    # Add a market sentiment factor (e.g., based on a simple moving average of the close price) with an adaptive window
    market_sentiment_window = 20
    market_sentiment = df['close'].rolling(window=market_sentiment_window, min_periods=1).mean() - close_price
    weight_market_sentiment = 0.1 * (1 - (volatility / volatility.max()))
    
    # Incorporate the market sentiment factor into the alpha factor
    alpha_factor = (alpha_factor * (1 - weight_market_sentiment)) + (weight_market_sentiment * market_sentiment)
    
    return alpha_factor
