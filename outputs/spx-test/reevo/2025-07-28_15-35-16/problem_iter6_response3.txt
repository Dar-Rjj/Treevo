```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor_2d = df['close'].pct_change(periods=2)
    momentum_factor_5d = df['close'].pct_change(periods=5)
    momentum_factor_10d = df['close'].pct_change(periods=10)
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the volatility (standard deviation of the close price over a window)
    volatility = close_price.rolling(window=20).std()
    
    # Integrate the price difference, multiple momentum factors, VWAP, RSI, and volatility into a new alpha factor
    alpha_factor = (
        0.2 * price_diff +
        0.1 * momentum_factor_2d +
        0.1 * momentum_factor_5d +
        0.1 * momentum_factor_10d +
        0.2 * (vwap - close_price) +
        0.2 * (rsi - 50) +
        0.1 * (volatility / close_price)
    )
    
    return alpha_factor
```
