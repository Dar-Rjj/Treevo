```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the average of high and low prices
    close_price = df['close']
    avg_high_low = (df['high'] + df['low']) / 2
    price_diff = close_price - avg_high_low
    
    # Compute a momentum factor based on the change in the close price over two periods
    momentum_factor = df['close'].pct_change(periods=2)
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = close_price.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the exponential moving average (EMA) of the close price with a 7-day span
    ema_7 = close_price.ewm(span=7, adjust=False).mean()
    
    # Calculate the 5-day and 20-day simple moving averages of the closing price
    sma_5 = close_price.rolling(window=5).mean()
    sma_20 = close_price.rolling(window=20).mean()
    
    # Calculate the ratio of the 5-day to 20-day SMA
    sma_ratio = sma_5 / sma_20
    
    # Calculate the 7-day exponential moving average of the volume
    ema_vol_7 = df['volume'].ewm(span=7, adjust=False).mean()
    
    # Integrate the price difference, momentum, VWAP, RSI, EMA, and SMA ratio into a new alpha factor
    alpha_factor = (price_diff * momentum_factor) + (vwap - close_price) + (rsi - 50) + (ema_7 - close_price) + (sma_ratio - 1) + (ema_vol_7 - df['volume'])
    
    return alpha_factor
```
