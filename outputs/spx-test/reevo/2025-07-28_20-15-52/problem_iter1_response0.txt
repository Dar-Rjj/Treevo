def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    close_diff = df['close'].diff()
    # Calculate the 7-day exponentially weighted moving average of the volume
    ewma_volume = df['volume'].ewm(span=7).mean()
    # Calculate the high-low range
    hl_range = (df['high'] - df['low']) / (df['close'] + 1e-7)
    # Factor: normalized change in close prices relative to the recent trading range and volume flow
    return (close_diff / (hl_range * ewma_volume))
