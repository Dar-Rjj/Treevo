```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day returns
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)

    # Calculate the 20-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()

    # Calculate the 30-day volatility
    thirty_day_volatility = df['close'].pct_change().rolling(window=30).std()

    # Create a factor that combines 5-day return, 20-day return, 20-day average volume, and 30-day volatility
    factor = (five_day_return + twenty_day_return) * (avg_20_day_volume / (thirty_day_volatility + 1e-6))

    return factor
```
