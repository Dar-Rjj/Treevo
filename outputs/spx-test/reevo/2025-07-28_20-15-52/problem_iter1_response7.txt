```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a simple moving average of the closing price over 5 days
    sma_5 = df['close'].rolling(window=5).mean()
    # Calculate the difference between the current close and the 5-day SMA, as a percentage of the 5-day SMA
    sma_diff_perc = (df['close'] - sma_5) / sma_5
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    # Combine the above indicators into a single factor
    factor = sma_diff_perc * daily_return + (df['high'] - df['low']) / df['close']
    return factor
```
