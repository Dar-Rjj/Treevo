```python
def heuristics_v2(df: pd.DataFrame, sector_volatility: pd.Series) -> pd.Series:
    # Calculate the 5-day log return
    five_day_log_return = (df['close'].pct_change(periods=5)).apply(lambda x: np.log(1 + x))
    
    # Calculate the 20-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the 20-day volatility (standard deviation of log returns)
    volatility = (df['close'].pct_change().apply(lambda x: np.log(1 + x))).rolling(window=20).std()
    
    # Create a factor that combines 5-day log return, 20-day average volume, and 20-day volatility
    factor = (five_day_log_return * avg_20_day_volume) / volatility
    
    # Adjust for sector-specific volatility
    factor = factor / sector_volatility
    
    return factor
```
