```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate momentum as the average of the last 5 closing prices
    momentum = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between current close and its 10-day moving average as a form of mean reversion
    mean_reversion = df['close'] - df['close'].rolling(window=10).mean()
    
    # Calculate volatility as the standard deviation over the last 5 closing prices, indicative of market uncertainty
    volatility = df['close'].rolling(window=5).std()
    
    # Construct an alpha factor that combines momentum, mean reversion, and inversely weighted by recent volatility
    alpha_factor = (momentum + mean_reversion) / (volatility + 1e-7)
    
    return alpha_factor
```
