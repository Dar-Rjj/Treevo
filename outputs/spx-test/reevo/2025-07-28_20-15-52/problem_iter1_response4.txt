```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between open and close
    open_close_diff = df['open'] - df['close']
    # Calculate the average of high and low
    avg_high_low = (df['high'] + df['low']) / 2
    # Calculate the smoothed volume by taking the rolling mean
    smooth_volume = df['volume'].rolling(window=5).mean()
    # Create a factor that combines these features
    factor = open_close_diff * avg_high_low / smooth_volume
    return factor
```
