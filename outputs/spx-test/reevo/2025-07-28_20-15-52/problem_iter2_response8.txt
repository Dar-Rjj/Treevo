```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate momentum by looking at the 10-day and 30-day price changes
    short_momentum = df['close'].pct_change(periods=10)
    long_momentum = df['close'].pct_change(periods=30)

    # Calculate volatility using the standard deviation of returns over the last 30 days
    volatility = df['close'].pct_change().rolling(window=30).std()

    # Calculate the volume trend by comparing the current volume to the 14-day exponentially weighted moving average
    ewma_volume_14 = df['volume'].ewm(span=14).mean()
    volume_trend = df['volume'] / ewma_volume_14

    # Calculate the high-low range
    hl_range = (df['high'] - df['low']) / (df['close'] + 1e-7)

    # Combine the signals in a meaningful way to create the alpha factor
    # Momentum signal is stronger when volatility is low and volume is high
    alpha_factor = (short_momentum - long_momentum) * (1 / (volatility + 1e-7)) * volume_trend / hl_range

    return alpha_factor
```
