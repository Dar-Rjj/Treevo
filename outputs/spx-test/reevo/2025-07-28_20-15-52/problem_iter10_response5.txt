```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate various returns
    five_day_return = df['close'].pct_change(periods=5)
    ten_day_return = df['close'].pct_change(periods=10)
    thirty_day_return = df['close'].pct_change(periods=30)
    sixty_day_return = df['close'].pct_change(periods=60)
    fifty_day_return = df['close'].pct_change(periods=50)
    two_hundred_day_return = df['close'].pct_change(periods=200)

    # Calculate average volumes
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_40_day_volume = df['volume'].rolling(window=40).mean()
    avg_80_day_volume = df['volume'].rolling(window=80).mean()

    # Volume trend factor
    volume_trend = (avg_20_day_volume - avg_80_day_volume) / avg_80_day_volume

    # Volatilities
    volatility_30 = df['close'].pct_change().rolling(window=30).std()
    volatility_90 = df['close'].pct_change().rolling(window=90).std()

    # Moving averages
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()

    # Relative strength
    relative_strength = fifty_day_return / two_hundred_day_return

    # Momentum signal
    momentum_signal = (df['close'] / ma_50) - 1

    # Mean reversion signal
    mean_reversion_signal = (df['close'] - ma_200) / ma_200

    # Seasonality factor
    monthly_mean = df['close'].resample('M').mean().reindex(df.index, method='ffill')
    seasonality_factor = (df['close'] - monthly_mean) / monthly_mean

    # Combine the signals
    combined_signal = (five_day_return + ten_day_return + thirty_day_return + sixty_day_return) * volume_trend * relative_strength / (volatility_30 / volatility_90)
    combined_signal = combined_signal * momentum_signal * mean_reversion_signal * seasonality_factor

    return combined_signal
```
