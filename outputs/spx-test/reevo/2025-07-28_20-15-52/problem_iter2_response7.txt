```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the opening price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the difference between the high and low price of the day
    diff_high_low = (df['high'] - df['low'] + 1e-7)
    
    # Calculate the weighted volume, higher volume days are given more importance
    weighted_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the Exponential Moving Average (EMA) for recent trends
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the recent volatility using the standard deviation of the last 5 days' close prices
    recent_volatility = df['close'].rolling(window=5).std()
    
    # Combine the signals in a meaningful way to create the alpha factor
    alpha_factor = (diff_close_open / diff_high_low) * weighted_volume * (df['close'] - ema_close) / (recent_volatility + 1e-7)
    
    return alpha_factor
```
