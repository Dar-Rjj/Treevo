```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return
    five_day_return = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the 10-day volatility
    ten_day_volatility = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day momentum
    thirty_day_momentum = df['close'].pct_change(periods=30)
    
    # Create a weighted return by combining 5-day and 30-day returns
    weighted_return = (five_day_return + 0.5 * thirty_day_momentum) / 1.5
    
    # Create a factor that combines the weighted return, 20-day average volume, and adjusts for 10-day volatility
    factor = weighted_return * avg_20_day_volume / ten_day_volatility
    
    return factor
```
