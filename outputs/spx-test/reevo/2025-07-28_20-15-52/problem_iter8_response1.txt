```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 100-day returns
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)
    hundred_day_return = df['close'].pct_change(periods=100)

    # Calculate the 10-day, 20-day, and 60-day average volumes
    avg_10_day_volume = df['volume'].rolling(window=10).mean()
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Create a volume trend factor
    volume_trend = (avg_10_day_volume - avg_60_day_volume) / avg_60_day_volume

    # Calculate the 30-day and 100-day volatility
    short_term_volatility = df['close'].pct_change().rolling(window=30).std()
    long_term_volatility = df['close'].pct_change().rolling(window=100).std()

    # Combine the 5-day, 20-day, and 100-day returns, volume trend, and normalize by both short-term and long-term volatilities
    combined_returns = (five_day_return + twenty_day_return + hundred_day_return) / 3
    factor = combined_returns * volume_trend / (short_term_volatility + long_term_volatility + 1e-7)
    
    return factor
```
