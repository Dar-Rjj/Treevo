```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 10-day, and 30-day returns
    five_day_return = df['close'].pct_change(periods=5)
    ten_day_return = df['close'].pct_change(periods=10)
    thirty_day_return = df['close'].pct_change(periods=30)

    # Calculate the 20-day and 60-day average volumes
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Create a volume trend factor
    volume_trend = (avg_20_day_volume - avg_60_day_volume) / avg_60_day_volume

    # Calculate the 30-day and 90-day volatilities
    volatility_30 = df['close'].pct_change().rolling(window=30).std()
    volatility_90 = df['close'].pct_change().rolling(window=90).std()

    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 10-day and 30-day relative strength
    relative_strength_10_over_30 = df['close'].rolling(window=10).mean() / df['close'].rolling(window=30).mean()

    # Calculate the 50-day and 200-day moving averages
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()

    # Long-term trend indicator
    long_term_trend = df['close'] / ma_50

    # Short-term mean reversion
    short_term_mean_reversion = (df['close'] - ma_50) / ma_50

    # Dynamic weighting based on the ratio of 30-day to 90-day volatilities
    dynamic_weighting = (volatility_30 / volatility_90)

    # Integrate diverse indicators and use dynamic weighting
    factor = (
        (five_day_return + ten_day_return + thirty_day_return) * 
        volume_trend * 
        rsi * 
        relative_strength_10_over_30 * 
        long_term_trend * 
        short_term_mean_reversion / 
        dynamic_weighting
    )

    return factor
```
