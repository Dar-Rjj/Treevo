```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_change'] = df['close'].diff()
    # Calculate the simple moving average of volume over 10 days
    df['volume_sma_10'] = df['volume'].rolling(window=10).mean()
    # Determine if the current day's high is above the 30-day high
    df['is_new_high'] = df['high'] > df['high'].rolling(window=30).max().shift(1)
    # Alpha factor combining change in closing price, trading volume, and new high signal
    alpha = df['close_change'] * (df['volume'] / df['volume_sma_10']) + df['is_new_high'].astype(int) * df['close_change']
    return alpha
```
