```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return
    five_day_return = df['close'].pct_change(periods=5)
    
    # Calculate the 20-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the 20-day volatility
    twenty_day_volatility = df['close'].rolling(window=20).std()
    
    # Create a factor that combines 5-day return, 20-day average volume, and 20-day volatility
    factor = (five_day_return * avg_20_day_volume) / twenty_day_volatility
    
    # Adjust for market regime by using a dynamic lookback period based on 20-day volatility
    dynamic_lookback = (twenty_day_volatility > twenty_day_volatility.shift(1)).astype(int)
    factor = factor * (1 + dynamic_lookback)
    
    return factor
```
