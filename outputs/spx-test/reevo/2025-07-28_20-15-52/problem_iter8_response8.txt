```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day momentum (price change over 50 days)
    fifty_day_momentum = df['close'].pct_change(periods=50)
    
    # Calculate the 20-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the 60-day volatility (standard deviation of returns)
    sixty_day_volatility = df['close'].pct_change().rolling(window=60).std()
    
    # Calculate the 200-day moving average to consider long-term trend
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the 10-day and 30-day moving averages to capture short to medium term trends
    ma_10 = df['close'].rolling(window=10).mean()
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Incorporate additional factors:
    # 20-day price return
    twenty_day_return = df['close'].pct_change(periods=20)
    
    # 5-day average volume
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the 50-day exponential moving average (EMA) for a smoother trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Create a factor that combines 50-day momentum, 20-day average volume, 60-day volatility, 
    # 200-day moving average, 10-day and 30-day moving averages, RSI, 20-day return, 5-day average volume, and 50-day EMA
    factor = (fifty_day_momentum * avg_20_day_volume * twenty_day_return * avg_5_day_volume) / (sixty_day_volatility + 1) * (df['close'] / ma_200) * (ma_10 / ma_30) * (100 / (rsi + 1)) * (df['close'] / ema_50)
    
    return factor
```
