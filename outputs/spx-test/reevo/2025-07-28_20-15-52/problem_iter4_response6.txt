```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the opening price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the range of the day's trading (high-low)
    trading_range = (df['high'] - df['low'] + 1e-7)
    
    # Weighted volume, more weight to higher volume days relative to the 20-day average
    weighted_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) for capturing recent trend
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Recent volatility using the standard deviation of the last 5 days' close prices
    recent_volatility = df['close'].rolling(window=5).std()
    
    # Calculate the change in EMA compared to the closing price
    ema_change = (df['close'] - ema_close) / (recent_volatility + 1e-7)
    
    # Calculate a smoothed momentum indicator over the last 5 days
    smooth_momentum = df['close'].pct_change(periods=5).ewm(span=5, adjust=False).mean()
    
    # Combine the signals in a meaningful way to create the alpha factor
    alpha_factor = (diff_close_open / trading_range) * weighted_volume * ema_change * smooth_momentum
    
    return alpha_factor
```
