```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a simple moving average of the closing price over 5 days
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between the current close and the 5-day SMA, as a percentage of the 5-day SMA
    sma_diff_perc = (df['close'] - sma_5) / sma_5
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the relative strength index (RSI) over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the average true range (ATR) as a measure of volatility over 14 days
    tr = df[['high', 'low']].apply(lambda x: np.max(np.abs(x - x.shift())), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine the above indicators into a single factor
    factor = sma_diff_perc * daily_return + (df['high'] - df['low']) / df['close'] + rsi / 100 - atr / df['close']
    return factor
```
