```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day returns
    five_day_return = df['close'].pct_change(periods=5)
    ten_day_return = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day and 60-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()
    
    # Calculate the 30-day relative strength compared to the 10-day
    rs_30_over_10 = df['close'].rolling(window=30).mean() / df['close'].rolling(window=10).mean()
    
    # Calculate the 20-day trend of the 5-day volume
    vol_trend = df['volume'].rolling(window=5).mean().pct_change(periods=20)
    
    # Calculate the 10-day over 30-day relative strength
    rs_10_over_30 = df['close'].rolling(window=10).mean() / df['close'].rolling(window=30).mean()
    
    # Construct an alpha factor that combines 5-day return, 10-day return, 20-day average volume, 60-day average volume,
    # 30-day over 10-day relative strength, 20-day volume trend, and 10-day over 30-day relative strength
    alpha_factor = (five_day_return * 
                    ten_day_return * 
                    (avg_20_day_volume / avg_60_day_volume) * 
                    rs_30_over_10 * 
                    (1 + vol_trend) * 
                    rs_10_over_30)
    
    return alpha_factor
```
