```python
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

def heuristics_v2(df: pd.DataFrame, macroeconomic_data: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day returns
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)

    # Calculate the 50-day and 200-day moving averages to capture long-term trends
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()

    # Long-term trend factor
    long_term_trend = (ma_50 - ma_200) / (df['close'] + 1e-7)

    # Calculate the 20-day average volume to represent liquidity
    avg_20_day_volume = df['volume'].rolling(window=20).mean()

    # Calculate the 30-day volatility
    thirty_day_volatility = df['close'].pct_change().rolling(window=30).std()

    # Short-term and long-term return ratio
    return_ratio = (five_day_return / (twenty_day_return + 1e-7))

    # Liquidity adjusted by volatility
    liquidity_adjusted_volatility = (avg_20_day_volume / (thirty_day_volatility + 1e-6))

    # Incorporate seasonality
    df['month'] = df.index.month
    monthly_seasonality = df.groupby('month')['close'].transform(lambda x: (x - x.mean()) / x.std())

    # Merge macroeconomic data with the stock data
    combined_df = df.join(macroeconomic_data, how='left')

    # Use machine learning for dynamic factor weighting
    X = combined_df[['five_day_return', 'long_term_trend', 'liquidity_adjusted_volatility', 'monthly_seasonality']].dropna()
    y = combined_df['close'].pct_change().shift(-1).loc[X.index]

    model = LinearRegression()
    model.fit(X, y)
    weights = model.coef_

    # Combine all factors to create the alpha factor using learned weights
    alpha_factor = (X * weights).sum(axis=1)

    return alpha_factor
```
