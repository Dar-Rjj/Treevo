```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 60-day returns
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)
    sixty_day_return = df['close'].pct_change(periods=60)

    # Calculate the 20-day and 60-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Volume trend factor
    vol_trend = (avg_20_day_volume - avg_60_day_volume) / (avg_60_day_volume + 1e-7)

    # Relative strength factors
    rs_60_over_20 = df['close'].ewm(span=60, adjust=False).mean() / df['close'].ewm(span=20, adjust=False).mean()
    rs_30_over_10 = df['close'].ewm(span=30, adjust=False).mean() / df['close'].ewm(span=10, adjust=False).mean()

    # Volatility adjusted returns
    short_volatility = df['close'].pct_change().rolling(window=5).std()
    vol_adj_five_day_return = five_day_return / (short_volatility + 1e-7)

    medium_volatility = df['close'].pct_change().rolling(window=20).std()
    vol_adj_twenty_day_return = twenty_day_return / (medium_volatility + 1e-7)

    long_volatility = df['close'].pct_change().rolling(window=60).std()
    vol_adj_sixty_day_return = sixty_day_return / (long_volatility + 1e-7)

    # RSI calculation
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    avg_gain = up.rolling(window=14).mean()
    avg_loss = down.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Long-term trend using EMA
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    long_term_trend = (ema_50 - ema_200) / (df['close'] + 1e-7)

    # Momentum factors
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_20 = df['close'].pct_change(periods=20)

    # Construct the alpha factor
    alpha_factor = (five_day_return * 
                    twenty_day_return * 
                    sixty_day_return * 
                    (1 + vol_trend) * 
                    rs_60_over_20 * 
                    rs_30_over_10 * 
                    vol_adj_five_day_return * 
                    vol_adj_twenty_day_return * 
                    vol_adj_sixty_day_return * 
                    (1 - rsi / 100) * 
                    (1 + long_term_trend) * 
                    (momentum_10 + momentum_20))

    return alpha_factor
```
