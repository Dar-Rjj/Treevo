```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    price_diff = df['close'].diff(1)
    
    # Compute exponential moving average of the price differences to identify trends
    ema_price_diff = price_diff.ewm(span=5, adjust=False).mean()
    
    # Calculate the ratio of current volume over the 10-day simple moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the EMA of price differences and the volume ratio as an alpha factor
    alpha_factor = (ema_price_diff * volume_ratio).fillna(0)
    return alpha_factor
```
