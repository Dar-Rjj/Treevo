```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 100-day returns
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)
    hundred_day_return = df['close'].pct_change(periods=100)

    # Calculate the 10-day, 20-day, and 60-day average volumes
    avg_10_day_volume = df['volume'].rolling(window=10).mean()
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Create a volume trend factor
    volume_trend = (avg_20_day_volume - avg_60_day_volume) / (avg_60_day_volume + 1e-7)

    # Calculate the 30-day and 100-day volatility
    short_term_volatility = df['close'].pct_change().rolling(window=30).std()
    long_term_volatility = df['close'].pct_change().rolling(window=100).std()

    # Combine the 5-day, 20-day, and 100-day returns, volume trend, and normalize by both short-term and long-term volatilities
    combined_returns = (five_day_return + twenty_day_return + hundred_day_return) / 3
    factor = combined_returns * volume_trend / (short_term_volatility + long_term_volatility + 1e-7)

    # Calculate the 50-day and 200-day exponential moving averages to capture long-term trends
    ma_50 = df['close'].ewm(span=50, adjust=False).mean()
    ma_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Long-term trend factor
    long_term_trend = (ma_50 - ma_200) / (df['close'] + 1e-7)

    # Calculate the 10-day and 30-day relative strength
    rs_30_over_10 = df['close'].ewm(span=30, adjust=False).mean() / df['close'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day and 20-day volume-weighted returns
    vw_return_10 = (df['close'] * df['volume']).ewm(span=10, adjust=False).mean() / df['volume'].ewm(span=10, adjust=False).mean()
    vw_return_20 = (df['close'] * df['volume']).ewm(span=20, adjust=False).mean() / df['volume'].ewm(span=20, adjust=False).mean()

    # Calculate the 10-day and 20-day momentum
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_20 = df['close'].pct_change(periods=20)

    # Adjust the alpha factor by incorporating the 10-day and 20-day volume-weighted returns and momentum
    adjusted_factor = factor * long_term_trend * rs_30_over_10 * (vw_return_10 + vw_return_20) * (momentum_10 + momentum_20)

    # Incorporate macroeconomic data if available
    if 'macro_indicator' in df.columns:
        macro_effect = df['macro_indicator'].pct_change(periods=10)
        adjusted_factor = adjusted_factor * macro_effect

    return adjusted_factor
```
