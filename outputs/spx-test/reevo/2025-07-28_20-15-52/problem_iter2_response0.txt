```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the difference between today's close and the close 10 days ago
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate the volatility as the standard deviation of the last 20 days' closing prices
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate the relative strength as the ratio of today's close to the lowest low in the last 20 days
    relative_strength = df['close'] / df['low'].rolling(window=20).min()
    
    # Combine the signals in a meaningful way to create the alpha factor
    alpha_factor = (momentum / (volatility + 1e-7)) * relative_strength
    
    return alpha_factor
```
