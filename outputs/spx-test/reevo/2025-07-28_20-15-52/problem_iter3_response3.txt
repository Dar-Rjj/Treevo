```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day returns
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)

    # Calculate the 20-day and 60-day average volumes
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Calculate the 20-day and 60-day volatility
    twenty_day_volatility = df['close'].rolling(window=20).std()
    sixty_day_volatility = df['close'].rolling(window=60).std()

    # Create a market regime indicator using the 20-day and 60-day returns
    market_regime = (twenty_day_return - five_day_return) / (sixty_day_volatility - twenty_day_volatility)

    # Combine 5-day return, 20-day volume, and market regime
    factor = (five_day_return * avg_20_day_volume) * market_regime

    # Simplify by focusing on key indicators
    factor = factor / (avg_60_day_volume + 1)  # Add 1 to avoid division by zero

    return factor
```
