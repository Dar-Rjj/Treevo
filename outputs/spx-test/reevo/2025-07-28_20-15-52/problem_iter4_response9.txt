```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the 10-day standard deviation of daily returns as a measure of volatility
    volatility = daily_return.rolling(window=10).std()
    
    # Calculate the difference between today's close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the difference between the high and low price of the day, adding a small constant to avoid division by zero
    diff_high_low = (df['high'] - df['low'] + 1e-7)
    
    # Calculate the weighted volume, higher volume days are given more importance
    weighted_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Compute the ratio of close to open price, adjusted for the day's range, to capture market depth
    market_depth_adjusted = (diff_close_open / diff_high_low)
    
    # Calculate the momentum as the difference between today's close and the close 10 days ago
    momentum = df['close'] - df['close'].shift(10)
    
    # Calculate the liquidity as the ratio of volume to the average volume over the last 20 days
    liquidity = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the relative strength as the ratio of today's close to the lowest low in the last 20 days
    relative_strength = df['close'] / df['low'].rolling(window=20).min()
    
    # Combine the signals in a meaningful way to create the alpha factor, incorporating volatility, volume trends, and momentum
    alpha_factor = (market_depth_adjusted * weighted_volume * momentum * liquidity) * (relative_strength / (volatility + 1e-7)).fillna(0)
    
    return alpha_factor
```
