```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a weighted moving average of the close price with more weight on recent prices
    n = 5  # number of days
    weights = np.arange(1, n+1)  # increasing weights for more recent days
    wma_close = df['close'].rolling(window=n).apply(lambda x: np.sum(weights*x) / np.sum(weights), raw=True)
    
    # Calculate the relative difference between the current close and the weighted moving average
    factor = (df['close'] - wma_close) / wma_close
    
    return factor
```
