```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day momentum (price change over 50 days)
    fifty_day_momentum = df['close'].pct_change(periods=50)
    
    # Calculate the 20-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the 60-day volatility (standard deviation of returns)
    volatility = df['close'].pct_change().rolling(window=60).std()
    
    # Calculate the 200-day moving average to consider long-term trend
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the 10-day Exponential Moving Average (EMA) for capturing recent trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day EMA of volume
    ema_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio of the 10-day EMA to the 200-day moving average
    ema_ratio = ema_10 / ma_200
    
    # Create a factor that combines 50-day momentum, 20-day average volume, 60-day volatility, 10-day EMA, and 200-day moving average
    factor = (fifty_day_momentum * avg_20_day_volume) / (volatility + 1e-7) * (ema_ratio * (ema_volume_10 / avg_20_day_volume))
    
    return factor
```
