```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return
    five_day_return = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day average volume
    avg_10_day_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the 20-day momentum
    momentum_20_day = df['close'] - df['close'].shift(20)
    
    # Create a factor that combines 5-day return, 10-day average volume, and 20-day momentum
    alpha_factor = (five_day_return * avg_10_day_volume * momentum_20_day).fillna(0)
    
    return alpha_factor
```
