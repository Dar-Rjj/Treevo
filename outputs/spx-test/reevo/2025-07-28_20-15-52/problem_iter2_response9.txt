```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a 5-day and 20-day simple moving average of the closing price
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Relative strength: difference between 5-day and 20-day SMA, normalized by 20-day SMA
    relative_strength = (sma_5 - sma_20) / sma_20
    
    # Volatility: standard deviation of daily returns over the past 30 days
    daily_return = df['close'].pct_change()
    volatility = daily_return.rolling(window=30).std()
    
    # Momentum: the rate of change in the closing price over the last 10 days
    momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Alpha factor combining relative strength, volatility, and momentum
    alpha = (relative_strength + momentum) / (1 + volatility)
    
    return alpha
```
