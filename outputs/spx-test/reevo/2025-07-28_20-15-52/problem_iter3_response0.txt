```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day return
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)
    
    # Calculate the 20-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the 20-day volatility
    twenty_day_volatility = df['close'].rolling(window=20).std()
    
    # Create a momentum factor that adjusts for volatility
    momentum_factor = (five_day_return + twenty_day_return) / twenty_day_volatility
    
    # Create a volume factor
    volume_factor = avg_20_day_volume / df['volume']
    
    # Combine the momentum and volume factors
    combined_factor = momentum_factor * volume_factor
    
    return combined_factor
```
