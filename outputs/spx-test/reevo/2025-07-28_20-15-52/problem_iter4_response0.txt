```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return
    five_day_return = df['close'].pct_change(periods=5)
    # Calculate the 10-day average volume
    avg_10_day_volume = df['volume'].rolling(window=10).mean()
    # Create a factor that combines 5-day return with 10-day average volume
    factor = five_day_return * avg_10_day_volume
    return factor
```
