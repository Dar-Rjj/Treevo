```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate short-term momentum as the average of the last 5 closing prices
    short_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate medium-term momentum as the average of the last 20 closing prices
    medium_momentum = df['close'].rolling(window=20).mean()
    
    # Calculate long-term momentum as the average of the last 60 closing prices
    long_momentum = df['close'].rolling(window=60).mean()
    
    # Calculate mean reversion as the difference between current close and its 10-day moving average
    mean_reversion = df['close'] - df['close'].rolling(window=10).mean()
    
    # Calculate volatility as the standard deviation over the last 5 closing prices
    short_volatility = df['close'].rolling(window=5).std()
    
    # Calculate longer-term volatility as the standard deviation over the last 20 closing prices
    long_volatility = df['close'].rolling(window=20).std()
    
    # Construct an alpha factor that combines short, medium, and long-term momentum, mean reversion,
    # and inversely weighted by a blend of short and long-term volatility
    alpha_factor = (short_momentum + medium_momentum + long_momentum + mean_reversion) / (short_volatility + long_volatility + 1e-7)
    
    return alpha_factor
```
