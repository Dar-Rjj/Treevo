```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 60-day returns for momentum
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)
    sixty_day_return = df['close'].pct_change(periods=60)

    # Calculate weighted returns
    weighted_return = (five_day_return * 0.4 + twenty_day_return * 0.3 + sixty_day_return * 0.3)

    # Calculate the 20-day and 60-day average volume for volume trend
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Calculate the 20-day and 60-day volatility
    twenty_day_volatility = df['close'].rolling(window=20).std()
    sixty_day_volatility = df['close'].rolling(window=60).std()

    # Create a factor that combines weighted return, volume trend, and volatility adjustment
    factor = (weighted_return * (avg_20_day_volume / avg_60_day_volume)) / (twenty_day_volatility + sixty_day_volatility)

    return factor
```
