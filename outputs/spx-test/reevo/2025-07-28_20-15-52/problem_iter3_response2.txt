```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day returns for momentum
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)

    # Calculate the 20-day and 60-day average volume for volume trend
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Calculate the 20-day volatility
    twenty_day_volatility = df['close'].rolling(window=20).std()

    # Create a factor that combines 5-day return, 20-day return, and 20-day average volume
    # Adjusted by 20-day volatility
    factor = (five_day_return + twenty_day_return) * (avg_20_day_volume / avg_60_day_volume) / twenty_day_volatility

    return factor
```
