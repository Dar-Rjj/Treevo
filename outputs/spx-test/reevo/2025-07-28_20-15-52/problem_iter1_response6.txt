```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average of the high and low price
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the weighted change in volume
    df['volume_change'] = df['volume'].pct_change().fillna(0)
    
    # Alpha factor: normalized close price difference multiplied by average price and volume change
    alpha_factor = (df['close_diff'] / df['avg_price']) * (1 + df['volume_change'])
    
    return alpha_factor
```
