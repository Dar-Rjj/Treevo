```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average of the high and low price
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the weighted change in volume
    df['volume_change'] = df['volume'].pct_change().fillna(0)
    
    # Calculate the volatility (standard deviation of log returns over a 10-day window)
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=10).std()
    
    # Calculate the leverage (ratio of volume to average volume over a 10-day window)
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['leverage'] = df['volume'] / df['avg_volume_10']
    
    # Alpha factor: combine close price difference, average price, volume change, and volatility
    alpha_factor = (df['close_diff'] / df['avg_price']) * (1 + df['volume_change']) * (1 / (1 + df['volatility'])) * df['leverage']
    
    return alpha_factor
```
