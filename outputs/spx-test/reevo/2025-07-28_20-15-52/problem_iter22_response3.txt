def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 60-day returns
    five_day_return = df['close'].pct_change(periods=5)
    twenty_day_return = df['close'].pct_change(periods=20)
    sixty_day_return = df['close'].pct_change(periods=60)

    # Calculate the 20-day and 60-day average volume
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    avg_60_day_volume = df['volume'].rolling(window=60).mean()

    # Calculate the 60-day relative strength compared to the 20-day
    rs_60_over_20 = df['close'].rolling(window=60).mean() / df['close'].rolling(window=20).mean()

    # Calculate the 20-day trend of the 5-day volume
    vol_trend = df['volume'].rolling(window=5).mean().pct_change(periods=20)

    # Calculate the 10-day and 30-day relative strength
    rs_30_over_10 = df['close'].rolling(window=30).mean() / df['close'].rolling(window=10).mean()

    # Calculate the volatility adjusted 5-day return
    short_volatility = df['close'].rolling(window=5).std()
    vol_adj_five_day_return = five_day_return / (short_volatility + 1e-7)

    # Calculate the volatility adjusted 20-day return
    long_volatility = df['close'].rolling(window=20).std()
    vol_adj_twenty_day_return = twenty_day_return / (long_volatility + 1e-7)

    # Calculate the volatility adjusted 60-day return
    long_long_volatility = df['close'].rolling(window=60).std()
    vol_adj_sixty_day_return = sixty_day_return / (long_long_volatility + 1e-7)

    # Calculate the 14-day RSI
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    avg_gain = up.ewm(span=14, adjust=False).mean()
    avg_loss = down.ewm(span=14, adjust=False).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Dynamic weighting based on the standard deviation of returns
    weight_5 = (five_day_return.std() + 1e-7) / (five_day_return.std() + twenty_day_return.std() + sixty_day_return.std() + 1e-7)
    weight_20 = (twenty_day_return.std() + 1e-7) / (five_day_return.std() + twenty_day_return.std() + sixty_day_return.std() + 1e-7)
    weight_60 = (sixty_day_return.std() + 1e-7) / (five_day_return.std() + twenty_day_return.std() + sixty_day_return.std() + 1e-7)

    # Adaptive exponential smoothing for the 5-day, 20-day, and 60-day returns
    exp_smooth_5 = five_day_return.ewm(span=5, adjust=False).mean()
    exp_smooth_20 = twenty_day_return.ewm(span=20, adjust=False).mean()
    exp_smooth_60 = sixty_day_return.ewm(span=60, adjust=False).mean()

    # Simplify the factor combinations
    factor_returns = (weight_5 * exp_smooth_5 + 
                      weight_20 * exp_smooth_20 + 
                      weight_60 * exp_smooth_60)
    
    factor_vol_adj = (vol_adj_five_day_return * weight_5 + 
                      vol_adj_twenty_day_return * weight_20 + 
                      vol_adj_sixty_day_return * weight_60)
    
    factor_rsi = (1 + rsi / 100)
    
    factor_volume = (avg_20_day_volume / avg_60_day_volume) * (1 + vol_trend)
    
    factor_relative_strength = rs_60_over_20 * rs_30_over_10

    # Construct the final alpha factor
    alpha_factor = factor_returns * factor_volume * factor_relative_strength * factor_vol_adj * factor_rsi

    return alpha_factor
