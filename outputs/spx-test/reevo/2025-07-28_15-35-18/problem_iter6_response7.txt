```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving averages of the closing price for long-term trends
    avg_close_10_days = df['close'].rolling(window=10).mean()
    avg_close_50_days = df['close'].rolling(window=50).mean()
    
    # Mean reversion signal: Difference between the current close and the 10-day moving average, scaled by the 10-day standard deviation
    std_close_10_days = df['close'].rolling(window=10).std()
    mean_reversion_signal = (df['close'] - avg_close_10_days) / std_close_10_days
    
    # Long-term trend: Ratio of the 10-day moving average to the 50-day moving average
    long_term_trend = avg_close_10_days / avg_close_50_days
    
    # Volume spike: Ratio of the current day's volume to the 5-day maximum volume
    max_volume_5_days = df['volume'].rolling(window=5).max()
    volume_spike = df['volume'] / (max_volume_5_days + 1e-7)
    
    # Relative strength: Ratio of the current close to the 10-day minimum close
    min_close_10_days = df['close'].rolling(window=10).min()
    relative_strength = df['close'] / min_close_10_days
    
    # Momentum: Calculate the percentage change in the closing price over a 20-day period
    momentum = df['close'].pct_change(periods=20)
    
    # Combine the mean reversion, long-term trend, volume spike, relative strength, and momentum into a single factor
    factor = (mean_reversion_signal * long_term_trend * volume_spike * relative_strength * momentum).fillna(0)
    return factor
```
