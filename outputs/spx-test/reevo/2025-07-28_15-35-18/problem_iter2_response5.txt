```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average and standard deviation of the closing price for volatility
    avg_close_10_days = df['close'].rolling(window=10).mean()
    std_close_10_days = df['close'].rolling(window=10).std()
    
    # Mean reversion signal: Difference between the current close and the 10-day moving average, scaled by the standard deviation
    mean_reversion_signal = (df['close'] - avg_close_10_days) / std_close_10_days
    
    # Calculate the 5-day maximum volume and the current day's volume ratio to it
    max_volume_5_days = df['volume'].rolling(window=5).max()
    volume_ratio = df['volume'] / (max_volume_5_days + 1e-7)
    
    # Relative strength: Ratio of the current close to the 10-day minimum close
    min_close_10_days = df['close'].rolling(window=10).min()
    relative_strength = df['close'] / min_close_10_days
    
    # Combine the mean reversion, volume ratio, and relative strength into a single factor
    factor = (mean_reversion_signal * volume_ratio * relative_strength).fillna(0)
    return factor
```
