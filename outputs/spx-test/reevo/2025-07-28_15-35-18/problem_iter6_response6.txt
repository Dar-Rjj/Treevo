def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 10-day exponential moving average of the close
    multi_period_momentum = df['close'] - df['close'].ewm(span=10, adjust=False).mean()

    # Compute the exponential moving average (EMA) of the volume over 10 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the high to the 10-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the low to the 10-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day rolling standard deviation of the close price for volatility
    std_close = df['close'].rolling(window=10).std()

    # Calculate the adaptive smoothing factor based on the 5-day rate of change in the close price
    adaptive_smoothing = 1 / (1 + df['close'].pct_change(periods=5).abs().ewm(span=5, adjust=False).mean())

    # Apply adaptive smoothing to the multi-period momentum
    smoothed_multi_period_momentum = multi_period_momentum.ewm(span=adaptive_smoothing, adjust=False).mean()

    # Combine the four components into a single factor using a multiplicative approach
    factor_values = (smoothed_multi_period_momentum * 
                     ema_volume * 
                     relative_high_ema * 
                     relative_low_ema * 
                     (1 + std_close))

    return factor_values
