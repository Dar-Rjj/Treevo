```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 7-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=7).mean()

    # Compute the exponential moving average (EMA) of the volume over 3 days for a more responsive liquidity measure
    ema_volume_short = df['volume'].ewm(span=3, adjust=False).mean()
    # Compute the exponential moving average (EMA) of the volume over 10 days for a more stable liquidity measure
    ema_volume_long = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the high to the 7-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=7, adjust=False).mean()

    # Calculate the ratio of the low to the 7-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=7, adjust=False).mean()

    # Calculate the 7-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=7, adjust=False).mean()

    # Calculate the 7-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=7, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 7-day EMA of the high and the 7-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=7, adjust=False).mean() - df['low'].ewm(span=7, adjust=False).mean()

    # Introduce a volatility factor using the 7-day EMA of the absolute returns
    volatility_factor_7 = abs(df['close'].pct_change()).ewm(span=7, adjust=False).mean()
    # Introduce a longer-term volatility factor using the 20-day EMA of the absolute returns
    volatility_factor_20 = abs(df['close'].pct_change()).ewm(span=20, adjust=False).mean()

    # Interaction term between short and long EMA volumes
    volume_interaction = ema_volume_short * ema_volume_long

    # Price range as a measure of intraday volatility
    price_range = df['high'] - df['low']

    # Incorporate the inter-period relationship: the ratio of today's close to the previous day's close
    inter_period_relationship = df['close'] / df['close'].shift(1)

    # Macro indicator: 30-day EMA of the close price
    macro_indicator = df['close'].ewm(span=30, adjust=False).mean()

    # Trend factor: the ratio of the 7-day EMA of the close to the 20-day EMA of the close
    trend_factor = df['close'].ewm(span=7, adjust=False).mean() / df['close'].ewm(span=20, adjust=False).mean()

    # Market sentiment indicator: the ratio of the current close to the 100-day EMA of the close
    sentiment_indicator = df['close'] / df['close'].ewm(span=100, adjust=False).mean()

    # Robust volatility measure: the 15-day standard deviation of returns
    robust_volatility = df['close'].pct_change().rolling(window=15).std()

    # Combine the components into a single factor
    factor_values = (multi_period_momentum * 
                     volume_interaction * 
                     relative_high_ema * 
                     relative_low_ema * 
                     (ema_open / ema_close) * 
                     bid_ask_spread_proxy * 
                     (volatility_factor_7 + volatility_factor_20) * 
                     price_range * 
                     inter_period_relationship * 
                     macro_indicator * 
                     trend_factor * 
                     sentiment_indicator * 
                     robust_volatility)

    return factor_values
```
