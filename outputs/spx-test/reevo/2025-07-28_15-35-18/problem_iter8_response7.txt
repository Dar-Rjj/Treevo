```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day simple moving average of the close price
    sma_close = df['close'].rolling(window=10).mean()

    # Calculate the multi-period momentum as the difference between today's close and the 10-day SMA of the close
    multi_period_momentum = df['close'] - sma_close

    # Compute the exponential moving average (EMA) of the volume over 5 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate the ratio of the high to the 10-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the low to the 10-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=10, adjust=False).mean()

    # Combine the key indicators into a single factor
    factor_values = multi_period_momentum * ema_volume * relative_high_ema * relative_low_ema

    return factor_values
```
