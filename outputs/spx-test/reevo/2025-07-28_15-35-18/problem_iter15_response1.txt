```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 20-day simple moving average of the close price to capture short and long term trends
    sma_10_close = df['close'].rolling(window=10).mean()
    sma_20_close = df['close'].rolling(window=20).mean()
    
    # Compute the 10-day and 20-day exponential moving average (EMA) of the volume for responsive liquidity measures
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()

    # Calculate the ratio of the high to the 10-day EMA of the high to capture relative strength
    relative_high_ema_10 = df['high'] / df['high'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio of the low to the 10-day EMA of the low to capture relative weakness
    relative_low_ema_10 = df['low'] / df['low'].ewm(span=10, adjust=False).mean()
    
    # Incorporate 10-day EMA of open and close prices
    ema_10_open = df['open'].ewm(span=10, adjust=False).mean()
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 10-day EMA of the high and the 10-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=10, adjust=False).mean() - df['low'].ewm(span=10, adjust=False).mean()

    # Introduce a volatility factor using the 10-day EMA of the absolute returns
    volatility_factor_ema_10 = abs(df['close'].pct_change()).ewm(span=10, adjust=False).mean()
    
    # Introduce an additional volatility measure using 20-day EMA
    volatility_factor_ema_20 = abs(df['close'].pct_change()).ewm(span=20, adjust=False).mean()

    # Interaction terms between SMAs and EMAs
    interaction_sma_ema = (sma_10_close / sma_20_close) * (ema_10_volume / ema_20_volume)

    # Combine the components into a single factor
    factor_values = (
        (df['close'] - sma_20_close) *  # Multi-period momentum
        (ema_10_volume * ema_20_volume) *  # Liquidity measure
        (relative_high_ema_10 * relative_low_ema_10) *  # Relative strength and weakness
        (ema_10_open / ema_10_close) *  # Market depth
        bid_ask_spread_proxy *  # Bid-ask spread proxy
        (volatility_factor_ema_10 + volatility_factor_ema_20) *  # Volatility factors
        interaction_sma_ema  # Interaction terms
    )

    return factor_values
```
