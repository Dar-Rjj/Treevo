```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 10-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=10).mean()

    # Compute the exponential moving average (EMA) of the volume over 15 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=15, adjust=False).mean()

    # Calculate the ratio of the high to the 15-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=15, adjust=False).mean()

    # Calculate the ratio of the low to the 15-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=15, adjust=False).mean()

    # Calculate the 15-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=15, adjust=False).mean()

    # Calculate the 15-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=15, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 15-day EMA of the high and the 15-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=15, adjust=False).mean() - df['low'].ewm(span=15, adjust=False).mean()

    # Introduce a volatility factor using the 15-day EMA of the absolute returns
    volatility_factor = abs(df['close'].pct_change()).ewm(span=15, adjust=False).mean()

    # Incorporate the RSI (Relative Strength Index) with a 14-day window
    rsi = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).ewm(com=13, adjust=False).mean() / df['close'].diff(1).clip(upper=0).abs().ewm(com=13, adjust=False).mean())))

    # Normalize each factor to have a mean of 0 and standard deviation of 1
    multi_period_momentum = (multi_period_momentum - multi_period_momentum.mean()) / multi_period_momentum.std()
    ema_volume = (ema_volume - ema_volume.mean()) / ema_volume.std()
    relative_high_ema = (relative_high_ema - relative_high_ema.mean()) / relative_high_ema.std()
    relative_low_ema = (relative_low_ema - relative_low_ema.mean()) / relative_low_ema.std()
    ema_open = (ema_open - ema_open.mean()) / ema_open.std()
    ema_close = (ema_close - ema_close.mean()) / ema_close.std()
    bid_ask_spread_proxy = (bid_ask_spread_proxy - bid_ask_spread_proxy.mean()) / bid_ask_spread_proxy.std()
    volatility_factor = (volatility_factor - volatility_factor.mean()) / volatility_factor.std()
    rsi = (rsi - rsi.mean()) / rsi.std()

    # Combine the components into a single factor
    factor_values = (multi_period_momentum * 
                     ema_volume * 
                     relative_high_ema * 
                     relative_low_ema * 
                     (ema_open / ema_close) * 
                     bid_ask_spread_proxy * 
                     volatility_factor * 
                     (rsi / 50))

    return factor_values
```
