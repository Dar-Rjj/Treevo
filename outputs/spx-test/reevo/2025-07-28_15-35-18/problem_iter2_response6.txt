```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open, capturing the day's momentum
    daily_momentum = df['close'] - df['open']
    
    # Calculate the volatility as the range between high and low
    daily_volatility = df['high'] - df['low']
    
    # Calculate the volume weighted price (VWP)
    vwp = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate a rolling average of VWP over a 10-day period
    vwp_rolling_avg = vwp.rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the closing price
    ma_5d = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the closing price
    ma_20d = df['close'].rolling(window=20).mean()
    
    # Calculate the 50-day moving average of the closing price
    ma_50d = df['close'].rolling(window=50).mean()
    
    # Calculate the interaction between the daily momentum and the rolling average VWP,
    # normalized by the daily volatility to account for market conditions
    alpha_factor_base = (daily_momentum * vwp_rolling_avg) / (daily_volatility + 1e-7)
    
    # Incorporate the relative position of the current price to its moving averages
    rel_pos_5d = (df['close'] - ma_5d) / (ma_5d + 1e-7)
    rel_pos_20d = (df['close'] - ma_20d) / (ma_20d + 1e-7)
    rel_pos_50d = (df['close'] - ma_50d) / (ma_50d + 1e-7)
    
    # Combine the factors
    alpha_factor = alpha_factor_base * (rel_pos_5d + rel_pos_20d + rel_pos_50d)
    
    return alpha_factor
```
