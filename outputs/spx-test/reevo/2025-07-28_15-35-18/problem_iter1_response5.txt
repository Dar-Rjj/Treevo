```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between 2-day smoothed close price and 5-day smoothed close price
    smooth_close_2 = df['close'].rolling(window=2).mean()
    smooth_close_5 = df['close'].rolling(window=5).mean()
    trend_factor = smooth_close_2 - smooth_close_5
    
    # Introduce a volume-adjusted factor. If the volume is above its 20-day moving average, it amplifies the original signal, otherwise, dampens it.
    volume_trend = df['volume'] > df['volume'].rolling(window=20).mean()
    adjusted_factor = trend_factor * (1 + (volume_trend * 0.5))
    
    return adjusted_factor
```
