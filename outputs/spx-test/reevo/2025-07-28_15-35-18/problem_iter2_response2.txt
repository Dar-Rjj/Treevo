```txt
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, to evaluate the momentum
    daily_momentum = df['close'].diff()

    # Compute an exponential moving average of the volume over 10 days as a more responsive proxy for liquidity
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio of the high to the exponential moving average of the high over the last 10 days
    # This attempts to capture if the stock is trading at higher levels than its recent average
    relative_high = df['high'] / df['high'].ewm(span=10, adjust=False).mean()

    # Calculate the percentage change in the closing price over a 5-day period to capture short-term price trends
    short_term_trend = df['close'].pct_change(periods=5)

    # Combine the four components into a single factor
    factor_values = daily_momentum * ema_volume * relative_high * short_term_trend

    return factor_values
```
