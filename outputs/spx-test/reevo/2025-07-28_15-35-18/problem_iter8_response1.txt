```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 10-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=10).mean()

    # Compute the exponential moving average (EMA) of the volume over 10 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the high to the 10-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the low to the 10-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=10, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 10-day EMA of the high and the 10-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=10, adjust=False).mean() - df['low'].ewm(span=10, adjust=False).mean()

    # Incorporate an adaptive window for volatility based on the 10-day standard deviation of the close price
    volatility = df['close'].rolling(window=10).std()

    # Calculate the 5-day rate of change (ROC) in closing prices for a shorter-term momentum signal
    roc_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Incorporate a more sophisticated liquidity measure: the ratio of the current volume to the 10-day EMA of the volume
    liquidity_ratio = df['volume'] / ema_volume

    # Combine the components into a single factor
    factor_values = (multi_period_momentum * 
                     ema_volume * 
                     relative_high_ema * 
                     relative_low_ema * 
                     (ema_open / ema_close) * 
                     bid_ask_spread_proxy * 
                     roc_5d * 
                     liquidity_ratio / 
                     volatility)

    return factor_values
```
