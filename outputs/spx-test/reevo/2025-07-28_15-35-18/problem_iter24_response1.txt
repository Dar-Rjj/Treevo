```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 10-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=10).mean()

    # Compute the exponential moving average (EMA) of the volume over 5 days for a more responsive liquidity measure
    ema_volume_short = df['volume'].ewm(span=5, adjust=False).mean()
    # Compute the exponential moving average (EMA) of the volume over 20 days for a more stable liquidity measure
    ema_volume_long = df['volume'].ewm(span=20, adjust=False).mean()

    # Interaction term between short and long EMA volumes
    volume_interaction = ema_volume_short * ema_volume_long

    # Introduce a volatility factor using the 10-day EMA of the absolute returns
    volatility_factor_10 = abs(df['close'].pct_change()).ewm(span=10, adjust=False).mean()
    # Introduce a longer-term volatility factor using the 30-day EMA of the absolute returns
    volatility_factor_30 = abs(df['close'].pct_change()).ewm(span=30, adjust=False).mean()

    # Interaction term between the 10-day and 30-day EMA of the absolute returns
    interaction_volatility = volatility_factor_10 * volatility_factor_30

    # Calculate the ratio of the high to the 5-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=5, adjust=False).mean()

    # Calculate the ratio of the low to the 5-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=5, adjust=False).mean()

    # Incorporate trend strength by calculating the 5-day EMA of the close-to-open return
    trend_strength = (df['close'] / df['open']).ewm(span=5, adjust=False).mean()

    # Interaction term between the multi-period momentum and the 5-day EMA of the close-to-open return
    interaction_momentum_trend = multi_period_momentum * trend_strength

    # Price range as a measure of intraday volatility
    price_range = df['high'] - df['low']

    # Incorporate the inter-period relationship: the ratio of today's close to the previous day's close
    inter_period_relationship = df['close'] / df['close'].shift(1)

    # Trend consistency: the number of days the close price has been above/below the 10-day EMA
    trend_consistency = (df['close'] > df['close'].rolling(window=10).mean()).astype(int).rolling(window=10).sum()

    # Sentiment factor: the ratio of the 5-day EMA of the high to the 5-day EMA of the low
    sentiment_factor = df['high'].ewm(span=5, adjust=False).mean() / df['low'].ewm(span=5, adjust=False).mean()

    # Interaction term between the sentiment factor and the inter-period relationship
    interaction_sentiment_ipr = sentiment_factor * inter_period_relationship

    # Combine the components into a single factor with a refined weighting scheme
    factor_values = (
        0.15 * multi_period_momentum + 
        0.10 * volume_interaction + 
        0.10 * (volatility_factor_10 + volatility_factor_30) + 
        0.10 * relative_high_ema + 
        0.10 * relative_low_ema + 
        0.10 * (1 + interaction_momentum_trend) + 
        0.10 * (1 + interaction_volatility) + 
        0.05 * price_range + 
        0.10 * inter_period_relationship + 
        0.10 * trend_consistency + 
        0.10 * (1 + interaction_sentiment_ipr)
    )

    return factor_values
```
