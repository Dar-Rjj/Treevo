```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted average price (VWAP) from the last 5 days
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the percentage change of VWAP over the last 5 days
    vwap_change = vwap.pct_change(periods=5)
    
    # Calculate a volatility factor using the standard deviation of the close prices over the last 10 days
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Combine the vwap change and volatility to form an alpha factor
    alpha_factor = vwap_change / (volatility_factor + 1e-7)
    
    return alpha_factor
```
