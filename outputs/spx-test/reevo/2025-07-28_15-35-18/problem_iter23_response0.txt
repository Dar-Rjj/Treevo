```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 10-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=10).mean()

    # Compute the exponential moving average (EMA) of the volume over 5 days for a more responsive liquidity measure
    ema_volume_short = df['volume'].ewm(span=5, adjust=False).mean()
    # Compute the exponential moving average (EMA) of the volume over 20 days for a more stable liquidity measure
    ema_volume_long = df['volume'].ewm(span=20, adjust=False).mean()

    # Calculate the ratio of the high to the 10-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the low to the 10-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=10, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 10-day EMA of the high and the 10-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=10, adjust=False).mean() - df['low'].ewm(span=10, adjust=False).mean()

    # Introduce a volatility factor using the 10-day EMA of the absolute returns
    volatility_factor_10 = abs(df['close'].pct_change()).ewm(span=10, adjust=False).mean()
    # Introduce a longer-term volatility factor using the 30-day EMA of the absolute returns
    volatility_factor_30 = abs(df['close'].pct_change()).ewm(span=30, adjust=False).mean()

    # Interaction term between short and long EMA volumes
    volume_interaction = ema_volume_short * ema_volume_long

    # Price range as a measure of intraday volatility
    price_range = df['high'] - df['low']

    # Incorporate the inter-period relationship: the ratio of today's close to the previous day's close
    inter_period_relationship = df['close'] / df['close'].shift(1)

    # Trend consistency: the number of days the close price has been above/below the 10-day EMA
    trend_consistency = (df['close'] > df['close'].rolling(window=10).mean()).astype(int).rolling(window=10).sum()

    # Incorporate a sentiment factor: the ratio of the 10-day EMA of the high to the 10-day EMA of the low
    sentiment_factor = df['high'].ewm(span=10, adjust=False).mean() / df['low'].ewm(span=10, adjust=False).mean()

    # Calculate the 5-day EMA of the close price to incorporate short-term trends
    ema_close_5 = df['close'].ewm(span=5, adjust=False).mean()
    # Calculate the 30-day EMA of the close price to incorporate long-term trends
    ema_close_30 = df['close'].ewm(span=30, adjust=False).mean()

    # Trend interaction: the ratio of the 5-day EMA to the 30-day EMA
    trend_interaction = ema_close_5 / ema_close_30

    # Combine the components into a single factor
    factor_values = (multi_period_momentum * 
                     volume_interaction * 
                     relative_high_ema * 
                     relative_low_ema * 
                     (ema_open / ema_close) * 
                     bid_ask_spread_proxy * 
                     (volatility_factor_10 + volatility_factor_30) * 
                     price_range * 
                     inter_period_relationship * 
                     trend_consistency * 
                     sentiment_factor * 
                     trend_interaction)

    return factor_values
```
