```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the true range (high - low)
    true_range = df['high'] - df['low']
    
    # Calculate the price difference (close - open)
    price_diff = df['close'] - df['open']
    
    # Create a factor based on the normalized price movement
    price_movement_factor = price_diff / (true_range + 1e-7)
    
    # Calculate the average true range over the last 5 days
    avg_true_range = true_range.rolling(window=5).mean()
    
    # Volume-weighted moving average of the price movement factor
    volume_weighted_factor = (price_movement_factor * df['volume']).rolling(window=5).mean() / df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day exponential moving average of the close price
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day standard deviation of the close price
    std_close = df['close'].rolling(window=5).std()
    
    # Calculate the volatility factor
    volatility_factor = std_close / (ema_close + 1e-7)
    
    # Calculate the momentum factor as the 5-day percentage change in close price
    momentum_factor = df['close'].pct_change(periods=5)
    
    # Exponentially smooth the factors
    smoothed_volume_weighted_factor = volume_weighted_factor.ewm(span=3, adjust=False).mean()
    smoothed_avg_true_range = avg_true_range.ewm(span=3, adjust=False).mean()
    smoothed_volatility_factor = volatility_factor.ewm(span=3, adjust=False).mean()
    smoothed_momentum_factor = momentum_factor.ewm(span=3, adjust=False).mean()
    
    # Combine the factors with weighted averages
    combined_factor = (smoothed_volume_weighted_factor * 0.4 + 
                       smoothed_avg_true_range * 0.3 + 
                       smoothed_volatility_factor * 0.2 + 
                       smoothed_momentum_factor * 0.1)
    
    return combined_factor
```
