```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum with an adaptive window based on the 10-day simple moving average of the close
    adaptive_window = df['close'].rolling(window=10).std() * 10 + 10
    adaptive_momentum = df['close'] - df['close'].rolling(window=adaptive_window.astype(int)).mean()

    # Compute the exponential moving average (EMA) of the volume over an adaptive window for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=adaptive_window.astype(int), adjust=False).mean()

    # Calculate the ratio of the high to the 10-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the low to the 10-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=10, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 10-day EMA of the high and the 10-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=10, adjust=False).mean() - df['low'].ewm(span=10, adjust=False).mean()

    # Introduce a volatility factor using the 10-day EMA of the absolute returns
    volatility_factor = abs(df['close'].pct_change()).ewm(span=10, adjust=False).mean()

    # Combine the components into a single factor, balancing momentum and liquidity
    factor_values = (adaptive_momentum * 
                     ema_volume * 
                     relative_high_ema * 
                     relative_low_ema * 
                     (ema_open / ema_close) * 
                     bid_ask_spread_proxy / 
                     volatility_factor)

    return factor_values
```
