```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor: 5-day return
    momentum = df['close'].pct_change(periods=5)
    
    # Liquidity factor: Average volume over the last 5 days
    liquidity = df['volume'].rolling(window=5).mean()
    
    # Volatility factor: Standard deviation of returns over the last 10 days
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=10).std()
    
    # True range and price movement factor
    true_range = df['high'] - df['low']
    price_diff = df['close'] - df['open']
    price_movement_factor = price_diff / (true_range + 1e-7)
    
    # Volume-weighted moving average of the price movement factor with adaptive window
    volume_weighted_factor = (price_movement_factor * df['volume']).rolling(window=5).mean() / df['volume'].rolling(window=5).mean()
    
    # Relative strength: Ratio of the current close to the 50-day moving average
    ma_50d = df['close'].rolling(window=50).mean()
    relative_strength = df['close'] / ma_50d
    
    # Trend persistence: 5-day and 20-day moving average crossover
    ma_5d = df['close'].rolling(window=5).mean()
    ma_20d = df['close'].rolling(window=20).mean()
    trend_persistence = (ma_5d - ma_20d) / (ma_20d + 1e-7)
    
    # Market depth: Average volume compared to the 50-day average volume
    avg_volume_50d = df['volume'].rolling(window=50).mean()
    market_depth = df['volume'] / (avg_volume_50d + 1e-7)
    
    # Advanced volatility measure: Parkinson's high-low volatility
    high_low_volatility = np.sqrt((1/4 * np.log(df['high'] / df['low']) ** 2).rolling(window=10).mean()) * np.sqrt(252)
    
    # Adaptive window for trend persistence
    trend_persistence_adaptive = (df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()) / (df['close'].rolling(window=20).mean() + 1e-7)
    
    # Macro indicator: 3-month moving average of the close price
    macro_indicator = df['close'].rolling(window=60).mean()
    
    # Combine factors: Momentum, Liquidity, Volatility, Price Movement, Relative Strength, Trend Persistence, Market Depth, Advanced Volatility, and Macro Indicator
    alpha_factor = (momentum * volume_weighted_factor * relative_strength * trend_persistence_adaptive * market_depth) / (high_low_volatility + 1e-7) * (df['close'] / macro_indicator)
    
    # Smooth the alpha factor using a simple moving average
    smoothed_alpha_factor = alpha_factor.rolling(window=3).mean()
    
    return smoothed_alpha_factor
```
