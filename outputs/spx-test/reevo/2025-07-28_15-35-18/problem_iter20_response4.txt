def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 5-day simple moving average of the close for a more responsive signal
    multi_period_momentum = df['close'] - df['close'].rolling(window=5).mean()

    # Compute the exponential moving average (EMA) of the volume over 5 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()

    # Calculate the ratio of the high to the 5-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=5, adjust=False).mean()

    # Calculate the ratio of the low to the 5-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=5, adjust=False).mean()

    # Calculate the 5-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=5, adjust=False).mean()

    # Calculate the 5-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 5-day EMA of the high and the 5-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=5, adjust=False).mean() - df['low'].ewm(span=5, adjust=False).mean()

    # Introduce a volatility factor using the 5-day EMA of the absolute returns
    volatility_factor = abs(df['close'].pct_change()).ewm(span=5, adjust=False).mean()

    # Incorporate trend strength by calculating the 5-day EMA of the close-to-open return
    trend_strength = (df['close'] / df['open']).ewm(span=5, adjust=False).mean()

    # Calculate the 5-day EMA of the daily return
    daily_return = df['close'].pct_change()
    ema_daily_return = daily_return.ewm(span=5, adjust=False).mean()

    # Interaction term between the multi-period momentum and the EMA of the daily return
    interaction_momentum_ema = multi_period_momentum * ema_daily_return

    # Interaction term between the 5-day EMA of the absolute returns and the bid-ask spread proxy
    interaction_volatility_bid_ask = volatility_factor * bid_ask_spread_proxy

    # Interaction term between the EMA of the volume and the trend strength
    interaction_volume_trend = ema_volume * trend_strength

    # Combine the components into a single factor
    factor_values = (multi_period_momentum *
                     ema_volume *
                     relative_high_ema *
                     relative_low_ema *
                     (ema_open / ema_close) *
                     bid_ask_spread_proxy *
                     volatility_factor *
                     trend_strength *
                     (1 + interaction_momentum_ema) *
                     (1 + interaction_volatility_bid_ask) *
                     (1 + interaction_volume_trend))

    return factor_values
