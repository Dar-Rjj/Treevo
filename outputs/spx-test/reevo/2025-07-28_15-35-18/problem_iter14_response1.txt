```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 3-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=3).mean()

    # Compute the exponential moving average (EMA) of the volume over 3 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=3, adjust=False).mean()

    # Calculate the ratio of the high to the 3-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=3, adjust=False).mean()

    # Calculate the ratio of the low to the 3-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=3, adjust=False).mean()

    # Calculate the 3-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=3, adjust=False).mean()

    # Calculate the 3-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=3, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 3-day EMA of the high and the 3-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=3, adjust=False).mean() - df['low'].ewm(span=3, adjust=False).mean()

    # Introduce a volatility factor using the 3-day EMA of the absolute returns
    volatility_factor = abs(df['close'].pct_change()).ewm(span=3, adjust=False).mean()

    # Incorporate market sentiment or macroeconomic indicators (e.g., VIX index)
    # For demonstration, we'll use a dummy sentiment score (replace with actual data)
    sentiment_score = df['close'].pct_change().rolling(window=3).mean()
    
    # Combine the components into a single factor, balancing contributions
    factor_values = (0.2 * multi_period_momentum + 
                     0.2 * ema_volume + 
                     0.15 * relative_high_ema + 
                     0.15 * relative_low_ema + 
                     0.1 * (ema_open / ema_close) + 
                     0.1 * bid_ask_spread_proxy + 
                     0.1 * volatility_factor + 
                     0.05 * sentiment_score)

    return factor_values
```
