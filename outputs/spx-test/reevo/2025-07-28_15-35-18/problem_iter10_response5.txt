```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 20-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=20).mean()

    # Compute the exponential moving average (EMA) of the volume over 20 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()

    # Calculate the ratio of the high to the 20-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=20, adjust=False).mean()

    # Calculate the ratio of the low to the 20-day EMA of the low to capture relative weakness
    relative_low_ema = df['low'] / df['low'].ewm(span=20, adjust=False).mean()

    # Calculate the 20-day EMA of the open price to incorporate market depth
    ema_open = df['open'].ewm(span=20, adjust=False).mean()

    # Calculate the 20-day EMA of the close price to incorporate market depth
    ema_close = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the bid-ask spread proxy as the difference between the 20-day EMA of the high and the 20-day EMA of the low
    bid_ask_spread_proxy = df['high'].ewm(span=20, adjust=False).mean() - df['low'].ewm(span=20, adjust=False).mean()

    # Incorporate a volatility measure using the standard deviation of returns over the last 20 days
    volatility = df['close'].pct_change().rolling(window=20).std()

    # Exponentially smooth the multi-period momentum
    smoothed_multi_period_momentum = multi_period_momentum.ewm(span=5, adjust=False).mean()

    # Exponentially smooth the EMA volume
    smoothed_ema_volume = ema_volume.ewm(span=5, adjust=False).mean()

    # Exponentially smooth the relative high EMA
    smoothed_relative_high_ema = relative_high_ema.ewm(span=5, adjust=False).mean()

    # Exponentially smooth the relative low EMA
    smoothed_relative_low_ema = relative_low_ema.ewm(span=5, adjust=False).mean()

    # Exponentially smooth the EMA open
    smoothed_ema_open = ema_open.ewm(span=5, adjust=False).mean()

    # Exponentially smooth the EMA close
    smoothed_ema_close = ema_close.ewm(span=5, adjust=False).mean()

    # Exponentially smooth the bid-ask spread proxy
    smoothed_bid_ask_spread_proxy = bid_ask_spread_proxy.ewm(span=5, adjust=False).mean()

    # Exponentially smooth the volatility measure
    smoothed_volatility = volatility.ewm(span=5, adjust=False).mean()

    # Combine the components into a single factor
    factor_values = (smoothed_multi_period_momentum * 
                     smoothed_ema_volume * 
                     smoothed_relative_high_ema * 
                     smoothed_relative_low_ema * 
                     (smoothed_ema_open / smoothed_ema_close) * 
                     smoothed_bid_ask_spread_proxy * 
                     (1 / smoothed_volatility))

    return factor_values
```
