```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close to capture the daily momentum
    daily_momentum = df['close'].diff()
    
    # Calculate the 5-day moving average of the high price
    ma_high_5 = df['high'].rolling(window=5).mean()
    
    # Calculate the 5-day moving average of the low price
    ma_low_5 = df['low'].rolling(window=5).mean()
    
    # Capture the volatility by calculating the standard deviation of the last 5 closing prices
    volatility = df['close'].rolling(window=5).std()
    
    # Heuristic calculation: 
    # The factor is derived as a weighted combination of the daily momentum, relative position to 5-day MA of high and low, and adjusted for the stock's volatility.
    alpha_factor = (daily_momentum + (df['close'] - ma_high_5) - (df['close'] - ma_low_5)) / (volatility + 1e-7)
    
    return alpha_factor
```
