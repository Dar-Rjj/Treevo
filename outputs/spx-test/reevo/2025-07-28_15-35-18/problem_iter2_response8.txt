```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low
    range_diff = (df['high'] - df['low'])
    
    # Calculate the weighted average of close and open prices using volume as weights
    weighted_avg = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])
    
    # Calculate a factor that represents the normalized distance of the weighted average from the low to the range
    factor = (weighted_avg - df['low']) / (range_diff + 1e-7)
    
    # Incorporate momentum by calculating the 5-day and 20-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Momentum factor: difference between the 5-day and 20-day moving averages
    momentum_factor = ma_5 - ma_20
    
    # Combine the normalized distance factor and the momentum factor
    combined_factor = factor + momentum_factor
    
    # Consider multiple time frames for normalization: 5-day and 20-day rolling mean of the combined factor
    combined_factor_5 = combined_factor.rolling(window=5).mean()
    combined_factor_20 = combined_factor.rolling(window=20).mean()
    
    # Final factor: difference between the 5-day and 20-day rolling means of the combined factor
    final_factor = combined_factor_5 - combined_factor_20
    
    return final_factor
```
