```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period momentum as the difference between today's close and the 10-day simple moving average of the close
    multi_period_momentum = df['close'] - df['close'].rolling(window=10).mean()

    # Compute the exponential moving average (EMA) of the volume over 10 days for a more responsive liquidity measure
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate the ratio of the high to the 10-day EMA of the high to capture relative strength
    relative_high_ema = df['high'] / df['high'].ewm(span=10, adjust=False).mean()

    # Calculate the 10-day realized volatility of the close price
    realized_volatility = df['close'].pct_change().rolling(window=10).std() * np.sqrt(252)

    # Calculate the 10-day price pattern as the difference between the 10-day EMA of the close and the 30-day EMA of the close
    price_pattern = df['close'].ewm(span=10, adjust=False).mean() - df['close'].ewm(span=30, adjust=False).mean()

    # Calculate the 10-day bid-ask spread as a proxy for liquidity
    bid_ask_spread = (df['high'] - df['low']) / df['close']

    # Combine the components into a single factor with weights
    factor_values = (multi_period_momentum * 0.3 
                     + ema_volume * 0.2 
                     + relative_high_ema * 0.2 
                     - realized_volatility * 0.1 
                     + price_pattern * 0.1 
                     - bid_ask_spread * 0.1)

    return factor_values
```
