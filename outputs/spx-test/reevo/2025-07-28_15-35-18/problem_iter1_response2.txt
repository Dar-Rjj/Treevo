```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, to evaluate the momentum
    daily_momentum = df['close'].diff()

    # Compute a simple moving average of the volume over 10 days as a proxy for liquidity
    sma_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the ratio of the high to the simple moving average of the high over the last 10 days
    # This attempts to capture if the stock is trading at higher levels than its recent average
    relative_high = df['high'] / df['high'].rolling(window=10).mean()

    # Combine the three components into a single factor
    factor_values = daily_momentum * sma_volume * relative_high

    return factor_values
```
