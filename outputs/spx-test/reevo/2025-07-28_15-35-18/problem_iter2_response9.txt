```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open, capturing the day's momentum
    daily_momentum = df['close'] - df['open']
    
    # Calculate the volatility as the range between high and low
    daily_volatility = df['high'] - df['low']
    
    # Calculate the volume weighted price (VWP)
    vwp = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate a rolling average of VWP over a 10-day period
    vwp_rolling_avg = vwp.rolling(window=10).mean()
    
    # Calculate the 20-day moving average of the close price to capture longer-term trends
    ma_close_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day moving average of the volume to capture longer-term volume trends
    ma_volume_20 = df['volume'].rolling(window=20).mean()
    
    # The alpha factor is then the interaction between the daily momentum, the rolling average VWP,
    # and the longer-term trend, normalized by the daily volatility and adjusted for volume trends
    alpha_factor = (daily_momentum * vwp_rolling_avg * (df['close'] - ma_close_20)) / ((daily_volatility + 1e-7) * (ma_volume_20 + 1e-7))
    
    return alpha_factor
```
