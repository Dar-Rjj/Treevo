```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum using close price
    df['momentum_5'] = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day average volume and the difference from the current volume
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_diff'] = (df['volume'] - df['avg_volume_10']) / df['avg_volume_10']
    
    # Calculate the 20-day moving average of the close price
    df['close_20_ma'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation of the close price
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Combine the factors: momentum, volume difference, and RSI
    factor = df['momentum_5'] + df['volume_diff'] + (df['rsi_14'] - 50) / 50
    
    return factor
```
