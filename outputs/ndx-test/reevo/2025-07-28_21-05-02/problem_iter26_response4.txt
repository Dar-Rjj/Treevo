def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 400-day and 800-day exponential moving averages of the closing price
    df['close_400_ema'] = df['close'].ewm(span=400, adjust=False).mean()
    df['close_800_ema'] = df['close'].ewm(span=800, adjust=False).mean()
    
    # Calculate the 300-day standard deviation of the closing price for a more robust volatility measure
    df['close_300_std'] = df['close'].rolling(window=300).std()
    
    # Calculate the 500-day standard deviation of the volume for a more robust volatility measure
    df['volume_500_std'] = df['volume'].rolling(window=500).std()
    
    # Non-linear transformation: Cube root of the difference between the closing price and the 400-day EMA
    factor_1 = ((df['close'] - df['close_400_ema']) / df['close_300_std']).apply(np.cbrt)
    
    # Non-linear transformation: Exponential of the absolute difference between the 400-day EMA and the 800-day EMA
    factor_2 = abs((df['close_400_ema'] - df['close_800_ema']) / df['close_300_std']).apply(np.exp)
    
    # Non-linear transformation: Square root of the volume divided by its 500-day standard deviation
    factor_3 = (df['volume'] / df['volume_500_std']).apply(np.sqrt)
    
    # Calculate the rate of change over 150 days for the 400-day EMA
    factor_4 = df['close_400_ema'].pct_change(periods=150)
    
    # Calculate the market sentiment as the normalized difference between the 400-day EMA and the 800-day EMA
    market_sentiment = (df['close_400_ema'] - df['close_800_ema']) / df['close_300_std']
    
    # Dynamic weighting: Use the absolute value of the 150-day percentage change in the 400-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.6 * dynamic_weight + 0.2 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.6 * factor_1 + 0.2 * factor_2 + 0.1 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
