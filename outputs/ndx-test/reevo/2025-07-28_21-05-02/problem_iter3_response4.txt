```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day exponential moving average of the closing price
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 100-day exponential moving average of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the 50-day standard deviation of the volume
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Factor 1: Difference between the closing price and the 50-day exponential moving average, divided by the 20-day standard deviation
    factor_1 = (df['close'] - df['close_50_ema']) / df['close_20_std']
    
    # Factor 2: Difference between the 50-day exponential moving average and the 100-day exponential moving average, divided by the 20-day standard deviation
    factor_2 = (df['close_50_ema'] - df['close_100_ema']) / df['close_20_std']
    
    # Factor 3: Volume divided by its 50-day standard deviation
    factor_3 = df['volume'] / df['volume_50_std']
    
    # Combine the factors into a single factor value with weighted contributions
    factor_values = 0.4 * factor_1 + 0.4 * factor_2 + 0.2 * factor_3
    
    return factor_values
```
