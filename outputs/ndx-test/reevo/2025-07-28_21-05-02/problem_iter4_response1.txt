```python
import numpy as np
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 100-day exponential moving average of the closing price
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the 50-day standard deviation of the volume
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Factor 1: Difference between the closing price and the 50-day EMA, divided by the 20-day standard deviation
    factor_1 = (df['close'] - df['close_50_ema']) / df['close_20_std']
    
    # Factor 2: Difference between the 50-day EMA and the 100-day EMA, divided by the 20-day standard deviation
    factor_2 = (df['close_50_ema'] - df['close_100_ema']) / df['close_20_std']
    
    # Factor 3: Volume divided by its 50-day standard deviation, with a non-linear transformation (square root)
    factor_3 = (df['volume'] / df['volume_50_std']).apply(np.sqrt)
    
    # Factor 4: 50-day EMA trend (difference between current and previous day's 50-day EMA)
    factor_4 = df['close_50_ema'].diff()
    
    # Factor 5: Volume trend (difference between current and previous day's volume)
    factor_5 = df['volume'].diff()
    
    # Factor 6: 200-day EMA of the closing price
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    factor_6 = (df['close_100_ema'] - df['close_200_ema']) / df['close_20_std']
    
    # Combine the factors into a single factor value
    factor_values = factor_1 + factor_2 + factor_3 + factor_4 + factor_5 + factor_6
    
    return factor_values
```
