```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    close_diff = df['close'].diff()
    
    # Calculate the average range (high - low) over the last 5 days
    avg_range_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the ratio of close difference to the 5-day average range
    momentum_factor = close_diff / (avg_range_5d + 1e-7)
    
    # Calculate the difference between the closing price and the average of high and low for the day
    avg_high_low = (df['high'] + df['low']) / 2
    deviation_from_avg = df['close'] - avg_high_low
    
    # Weight the deviation by the ratio of volume relative to the moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Create the factor as a weighted deviation, capturing both price action significance and volume relevance
    price_volatility_factor = deviation_from_avg * volume_ratio
    
    # Calculate the volatility (standard deviation) of the closing prices over the last 10 days
    volatility = df['close'].rolling(window=10).std()
    
    # Combine the momentum, price volatility, and volume-weighted factors
    alpha_factor = momentum_factor + price_volatility_factor + (volatility * volume_ratio)
    
    return alpha_factor
```
