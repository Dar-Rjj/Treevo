```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day, 150-day, and 250-day exponential moving average of the closing price
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_150_ema'] = df['close'].ewm(span=150, adjust=False).mean()
    df['close_250_ema'] = df['close'].ewm(span=250, adjust=False).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the 50-day standard deviation of the volume
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Factor 1: Difference between the closing price and the 50-day EMA, divided by the 20-day standard deviation
    factor_1 = (df['close'] - df['close_50_ema']) / df['close_20_std']
    
    # Factor 2: Difference between the 50-day EMA and the 150-day EMA, divided by the 20-day standard deviation
    factor_2 = (df['close_50_ema'] - df['close_150_ema']) / df['close_20_std']
    
    # Factor 3: Difference between the 150-day EMA and the 250-day EMA, divided by the 20-day standard deviation
    factor_3 = (df['close_150_ema'] - df['close_250_ema']) / df['close_20_std']
    
    # Factor 4: Volume divided by its 50-day standard deviation, with a non-linear transformation (square root)
    factor_4 = (df['volume'] / df['volume_50_std']).apply(np.sqrt)
    
    # Factor 5: 50-day EMA trend (difference between current and previous day's 50-day EMA)
    factor_5 = df['close_50_ema'].diff()
    
    # Factor 6: 10-day price momentum
    df['momentum_10d'] = df['close'].diff(10)
    factor_6 = df['momentum_10d'] / df['close_20_std']
    
    # Combine the factors into a single factor value with meaningful weights
    factor_values = (0.2 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4 + 0.1 * factor_5 + 0.1 * factor_6)
    
    return factor_values
```
