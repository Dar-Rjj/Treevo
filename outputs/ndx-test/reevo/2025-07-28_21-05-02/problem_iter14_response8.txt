```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 200-day exponential moving averages of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 75-day standard deviation of the closing price
    df['close_75_std'] = df['close'].rolling(window=75).std()
    
    # Calculate the 100-day standard deviation of the volume
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    
    # Non-linear transformation: Square root of the absolute difference between the closing price and the 100-day EMA
    factor_1 = (df['close'] - df['close_100_ema']).abs().div(df['close_75_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the absolute difference between the 100-day EMA and the 200-day EMA
    factor_2 = (df['close_100_ema'] - df['close_200_ema']).abs().div(df['close_75_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 100-day standard deviation
    factor_3 = df['volume'].div(df['volume_100_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 10 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=10)
    
    # Calculate the market sentiment as the difference between the 100-day EMA and the 200-day EMA
    market_sentiment = df['close_100_ema'] - df['close_200_ema']
    
    # Introduce a new factor: the rate of change over 5 days for the volume
    factor_5 = df['volume'].pct_change(periods=5)
    
    # Include the rate of change over 20 days for the closing price
    factor_6 = df['close'].pct_change(periods=20)
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment and the new factors
    factor_values = 0.3 * factor_1 + 0.25 * factor_2 + 0.2 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment + 0.05 * factor_5 + 0.1 * factor_6
    
    return factor_values
```
