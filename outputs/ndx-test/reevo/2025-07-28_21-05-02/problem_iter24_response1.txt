def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 250-day and 500-day exponential moving averages of the closing price for long-term trends
    df['close_250_ema'] = df['close'].ewm(span=250, adjust=False).mean()
    df['close_500_ema'] = df['close'].ewm(span=500, adjust=False).mean()
    
    # Calculate the 180-day standard deviation of the closing price for a more robust volatility measure
    df['close_180_std'] = df['close'].rolling(window=180).std()
    
    # Calculate the 360-day standard deviation of the volume for a more robust volatility measure
    df['volume_360_std'] = df['volume'].rolling(window=360).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 250-day EMA
    factor_1 = ((df['close'] - df['close_250_ema']) / df['close_180_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 250-day EMA and the 500-day EMA
    factor_2 = ((df['close_250_ema'] - df['close_500_ema']) / df['close_180_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 360-day standard deviation
    factor_3 = (df['volume'] / df['volume_360_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 30 days for the 250-day EMA
    factor_4 = df['close_250_ema'].pct_change(periods=30)
    
    # Calculate the market sentiment as the difference between the 250-day EMA and the 500-day EMA
    market_sentiment = df['close_250_ema'] - df['close_500_ema']
    
    # Dynamic weighting: Use the absolute value of the 30-day percentage change in the 250-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Introduce a new momentum factor: the rate of change over 90 days for the closing price
    factor_5 = df['close'].pct_change(periods=90)
    
    # Introduce a new factor: the absolute value of the difference between the high and low price over 90 days
    df['hl_diff_90'] = (df['high'] - df['low']).rolling(window=90).mean().apply(np.abs)
    factor_6 = df['hl_diff_90']
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.5 * dynamic_weight + 0.3 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.025 * dynamic_weight + 0.025 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment and the new factors
    factor_values = (
        0.5 * factor_1 + 
        0.3 * factor_2 + 
        0.1 * factor_3 + 
        0.05 * factor_4 + 
        0.025 * market_sentiment + 
        0.025 * factor_5 + 
        0.025 * factor_6
    ) * normalized_weight
    
    return factor_values
