```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum (current close - close 5 days ago)
    df['momentum_5d'] = df['close'].diff(5)
    
    # Calculate the 10-day exponential moving average (EMA) of the closing price
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the dynamic volatility using an exponentially weighted standard deviation of the last 10 days' returns
    df['dynamic_volatility_10d'] = df['close'].pct_change().ewm(span=10, adjust=False).std()
    
    # Calculate the volume shock (current volume / average volume over the last 10 days)
    df['volume_shock'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the 5-day exponentially weighted moving average (EWMA) with recent days having more weight
    df['ewma_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the positive difference between the current close price and 5-day EWMA
    df['diff_pos_ewma'] = df[['close', 'ewma_5d']].apply(lambda x: max(x['close'] - x['ewma_5d'], 0), axis=1)
    
    # Non-linear transformation: square root of the positive difference
    df['nonlinear_diff'] = np.sqrt(df['diff_pos_ewma'])
    
    # Calculate the 10-day cumulative return (cumulative return over the past 10 days)
    df['cumulative_return_10d'] = (df['close'] / df['close'].shift(10)) - 1
    
    # Calculate the price range (high - low) over the last 14 days, with exponential weighting
    df['price_range_14d'] = (df['high'] - df['low']).ewm(span=14, adjust=False).mean()
    
    # Factor is the product of the 5-day momentum, 10-day EMA, 10-day dynamic volatility, volume shock, non-linear difference, 10-day cumulative return, and 14-day price range
    factor_value = (df['momentum_5d'] * 
                    (df['close'] - df['ema_10d']) * 
                    df['dynamic_volatility_10d'] * 
                    df['volume_shock'] * 
                    df['nonlinear_diff'] * 
                    df['cumulative_return_10d'] * 
                    df['price_range_14d'])
    
    return factor_value
```
