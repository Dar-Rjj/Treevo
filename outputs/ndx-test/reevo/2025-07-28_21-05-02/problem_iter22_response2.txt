def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 300-day and 600-day exponential moving averages of the closing price
    df['close_300_ema'] = df['close'].ewm(span=300, adjust=False).mean()
    df['close_600_ema'] = df['close'].ewm(span=600, adjust=False).mean()
    
    # Calculate the 300-day standard deviation of the closing price for a more robust volatility measure
    df['close_300_std'] = df['close'].rolling(window=300).std()
    
    # Calculate the 500-day standard deviation of the volume for a more robust volatility measure
    df['volume_500_std'] = df['volume'].rolling(window=500).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 300-day EMA
    factor_1 = ((df['close'] - df['close_300_ema']) / df['close_300_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 300-day EMA and the 600-day EMA
    factor_2 = ((df['close_300_ema'] - df['close_600_ema']) / df['close_300_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 500-day standard deviation
    factor_3 = (df['volume'] / df['volume_500_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 50 days for the 300-day EMA
    factor_4 = df['close_300_ema'].pct_change(periods=50)
    
    # Calculate the market sentiment as the difference between the 300-day EMA and the 600-day EMA
    market_sentiment = df['close_300_ema'] - df['close_600_ema']
    
    # Dynamic weighting: Use the absolute value of the 50-day percentage change in the 300-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.7 * dynamic_weight + 0.15 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.7 * factor_1 + 0.15 * factor_2 + 0.05 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
