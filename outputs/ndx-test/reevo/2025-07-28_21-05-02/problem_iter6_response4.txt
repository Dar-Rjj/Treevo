```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day and 50-day exponential moving averages of the closing price for shorter-term trends
    df['close_20_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the closing price for short-term volatility
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the 20-day standard deviation of the volume for short-term volume anomalies
    df['volume_20_std'] = df['volume'].rolling(window=20).std()
    
    # Short-term momentum factor: Difference between the closing price and the 20-day EMA, divided by the 10-day standard deviation
    short_momentum_factor = (df['close'] - df['close_20_ema']) / df['close_10_std']
    
    # Medium-term momentum factor: Difference between the closing price and the 50-day EMA, divided by the 10-day standard deviation
    medium_momentum_factor = (df['close'] - df['close_50_ema']) / df['close_10_std']
    
    # Trend strength factor: Difference between the 20-day EMA and the 50-day EMA, divided by the 10-day standard deviation
    trend_strength_factor = (df['close_20_ema'] - df['close_50_ema']) / df['close_10_std']
    
    # Volume anomaly factor: Volume divided by its 20-day standard deviation
    volume_anomaly_factor = df['volume'] / df['volume_20_std']
    
    # Volatility factor: 10-day standard deviation of the closing price
    volatility_factor = df['close_10_std']
    
    # Combine the factors into a single factor value with optimized weighted contributions
    factor_values = 0.3 * short_momentum_factor + 0.3 * medium_momentum_factor + 0.2 * trend_strength_factor + 0.1 * volume_anomaly_factor + 0.1 * volatility_factor
    
    return factor_values
```
