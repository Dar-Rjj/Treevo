```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 75-day exponential moving average of the closing price
    df['close_75_ema'] = df['close'].ewm(span=75, adjust=False).mean()
    
    # Calculate the 150-day exponential moving average of the closing price
    df['close_150_ema'] = df['close'].ewm(span=150, adjust=False).mean()
    
    # Calculate the 60-day standard deviation of the closing price
    df['close_60_std'] = df['close'].rolling(window=60).std()
    
    # Calculate the 120-day standard deviation of the volume
    df['volume_120_std'] = df['volume'].rolling(window=120).std()
    
    # Non-linear transformation: Logarithm of the difference between the closing price and the 75-day EMA
    factor_1 = ((df['close'] - df['close_75_ema']) / df['close_60_std']).apply(np.log1p)
    
    # Non-linear transformation: Square root of the difference between the 75-day EMA and the 150-day EMA
    factor_2 = ((df['close_75_ema'] - df['close_150_ema']) / df['close_60_std']).apply(np.sqrt)
    
    # Non-linear transformation: Cube root of the volume divided by its 120-day standard deviation
    factor_3 = (df['volume'] / df['volume_120_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 8 days for the 75-day EMA
    factor_4 = df['close_75_ema'].pct_change(periods=8)
    
    # Calculate the market sentiment as the difference between the 75-day EMA and the 150-day EMA
    market_sentiment = df['close_75_ema'] - df['close_150_ema']
    
    # Introduce a new factor: the rate of change over 5 days for the volume
    factor_5 = df['volume'].pct_change(periods=5)
    
    # Introduce a new momentum factor: the rate of change over 10 days for the closing price
    factor_6 = df['close'].pct_change(periods=10)
    
    # Introduce a new factor: the absolute value of the difference between the high and low price over 30 days
    df['hl_diff_30'] = df['high'] - df['low']
    factor_7 = df['hl_diff_30'].rolling(window=30).mean().apply(np.abs)
    
    # Combine the factors into a single factor value with balanced weighted contributions, including market sentiment and the new factors
    factor_values = (
        0.2 * factor_1 + 
        0.2 * factor_2 + 
        0.15 * factor_3 + 
        0.1 * factor_4 + 
        0.05 * market_sentiment + 
        0.025 * factor_5 + 
        0.025 * factor_6 + 
        0.25 * factor_7
    )
    
    return factor_values
```
