def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 30-day and 100-day exponential moving averages of the closing price
    df['close_30_ema'] = df['close'].ewm(span=30, adjust=False).mean()
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 30-day standard deviation of the closing price for a more responsive volatility measure
    df['close_30_std'] = df['close'].rolling(window=30).std()
    
    # Calculate the 50-day standard deviation of the volume for a more responsive volatility measure
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 30-day EMA
    factor_1 = ((df['close'] - df['close_30_ema']) / df['close_30_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the absolute difference between the 30-day EMA and the 100-day EMA
    factor_2 = (abs(df['close_30_ema'] - df['close_100_ema']) / df['close_30_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 50-day standard deviation
    factor_3 = (df['volume'] / df['volume_50_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 20 days for the 30-day EMA
    factor_4 = df['close_30_ema'].pct_change(periods=20)
    
    # Calculate the market sentiment as the difference between the 30-day EMA and the 100-day EMA
    market_sentiment = df['close_30_ema'] - df['close_100_ema']
    
    # Dynamic weighting: Use the absolute value of the 20-day percentage change in the 30-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.4 * dynamic_weight + 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.4 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
