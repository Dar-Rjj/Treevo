```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 250-day exponential moving averages of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_250_ema'] = df['close'].ewm(span=250, adjust=False).mean()
    
    # Calculate the 50-day and 100-day standard deviations of the closing price and volume
    df['close_50_std'] = df['close'].rolling(window=50).std()
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_50_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 100-day EMA and the 250-day EMA
    factor_2 = ((df['close_100_ema'] - df['close_250_ema']) / df['close_50_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 100-day standard deviation
    factor_3 = (df['volume'] / df['volume_100_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 15 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=15)
    
    # Calculate the market sentiment as the difference between the 100-day EMA and the 250-day EMA
    market_sentiment = df['close_100_ema'] - df['close_250_ema']
    
    # Introduce a new factor: the rate of change over 7 days for the volume
    factor_5 = df['volume'].pct_change(periods=7)
    
    # Calculate the 30-day RSI (Relative Strength Index) of the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=30).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=30).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    df['rsi_30'] = rsi
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment, the new factor, and RSI
    factor_values = 0.3 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.1 * factor_4 + 0.05 * market_sentiment + 0.05 * factor_5 + 0.1 * df['rsi_30']
    
    return factor_values
```
