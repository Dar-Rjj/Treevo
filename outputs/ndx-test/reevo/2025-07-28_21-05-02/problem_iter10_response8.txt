```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day, 50-day, and 100-day exponential moving average of the closing price
    df['close_20_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the closing price
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the 20-day standard deviation of the volume
    df['volume_20_std'] = df['volume'].rolling(window=20).std()
    
    # Factor 1: Difference between the closing price and the 20-day EMA, divided by the 10-day standard deviation
    factor_1 = (df['close'] - df['close_20_ema']) / df['close_10_std']
    
    # Factor 2: Difference between the 20-day EMA and the 50-day EMA, divided by the 10-day standard deviation
    factor_2 = (df['close_20_ema'] - df['close_50_ema']) / df['close_10_std']
    
    # Factor 3: Difference between the 50-day EMA and the 100-day EMA, divided by the 10-day standard deviation
    factor_3 = (df['close_50_ema'] - df['close_100_ema']) / df['close_10_std']
    
    # Factor 4: Volume divided by its 20-day standard deviation, with a non-linear transformation (square root)
    factor_4 = (df['volume'] / df['volume_20_std']).apply(np.sqrt)
    
    # Factor 5: 20-day EMA trend (difference between current and previous day's 20-day EMA)
    factor_5 = df['close_20_ema'].diff()
    
    # Factor 6: 10-day price momentum
    df['momentum_10d'] = df['close'].diff(10)
    factor_6 = df['momentum_10d'] / df['close_10_std']
    
    # Combine the factors into a single factor value with meaningful weights
    factor_values = (0.2 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4 + 0.1 * factor_5 + 0.1 * factor_6)
    
    return factor_values
```
