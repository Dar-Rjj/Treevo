```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 100-day exponentially weighted moving averages of the closing price
    df['close_50_ewma'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_100_ewma'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 20-day exponentially weighted standard deviation of the closing price
    df['close_20_ewstd'] = df['close'].ewm(span=20, adjust=False).std()
    
    # Calculate the 50-day exponentially weighted standard deviation of the volume
    df['volume_50_ewstd'] = df['volume'].ewm(span=50, adjust=False).std()
    
    # Factor 1: Difference between the closing price and the 50-day exponentially weighted moving average, divided by the 20-day exponentially weighted standard deviation
    factor_1 = (df['close'] - df['close_50_ewma']) / df['close_20_ewstd']
    
    # Factor 2: Difference between the 50-day exponentially weighted moving average and the 100-day exponentially weighted moving average, divided by the 20-day exponentially weighted standard deviation
    factor_2 = (df['close_50_ewma'] - df['close_100_ewma']) / df['close_20_ewstd']
    
    # Factor 3: Volume divided by its 50-day exponentially weighted standard deviation, with a non-linear transformation (square root)
    factor_3 = (df['volume'] / df['volume_50_ewstd']).apply(np.sqrt)
    
    # Calculate the 20-day exponentially weighted moving average of the high and low prices
    df['high_20_ewma'] = df['high'].ewm(span=20, adjust=False).mean()
    df['low_20_ewma'] = df['low'].ewm(span=20, adjust=False).mean()
    
    # Factor 4: Difference between the current high and the 20-day exponentially weighted moving average of the high, divided by the 20-day exponentially weighted standard deviation of the closing price
    factor_4 = (df['high'] - df['high_20_ewma']) / df['close_20_ewstd']
    
    # Factor 5: Difference between the current low and the 20-day exponentially weighted moving average of the low, divided by the 20-day exponentially weighted standard deviation of the closing price
    factor_5 = (df['low'] - df['low_20_ewma']) / df['close_20_ewstd']
    
    # Calculate the 20-day exponentially weighted moving average of the open price
    df['open_20_ewma'] = df['open'].ewm(span=20, adjust=False).mean()
    
    # Factor 6: Difference between the current open and the 20-day exponentially weighted moving average of the open, divided by the 20-day exponentially weighted standard deviation of the closing price
    factor_6 = (df['open'] - df['open_20_ewma']) / df['close_20_ewstd']
    
    # Factor 7: Rate of change over 20 days for the 50-day exponentially weighted moving average of the closing price
    factor_7 = df['close_50_ewma'].pct_change(periods=20)
    
    # Combine the factors into a single factor value
    factor_values = factor_1 + factor_2 + factor_3 + factor_4 + factor_5 + factor_6 + factor_7
    
    return factor_values
```
