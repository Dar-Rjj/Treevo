```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, indicative of daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the positive or negative movement of the closing price relative to the open
    price_movement = df['close'] - df['open']
    
    # Calculate a weighted volume based on the day's price movement. Large movements (positive or negative) with high volume are more significant
    weighted_volume = abs(price_movement) * df['volume']
    
    # Calculate the average true range (ATR) over 14 days as a measure of volatility
    atr = (df['high'] - df['low']).rolling(window=14).mean()
    
    # Calculate the 50-day moving average and its difference from the current close price for trend strength
    ma_50 = df['close'].rolling(window=50).mean()
    trend_strength = df['close'] - ma_50
    
    # Calculate the relative strength (RS) as the ratio of the current close price to the 50-day moving average
    rs = df['close'] / ma_50
    
    # The alpha factor is a combination of daily range, weighted volume, ATR, trend strength, and relative strength
    factor = (daily_range / (weighted_volume + 1e-7)) * (atr / (trend_strength + 1e-7)) * rs
    
    return factor
```
