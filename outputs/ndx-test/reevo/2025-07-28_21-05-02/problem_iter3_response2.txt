def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day exponentially weighted moving average of the closing price
    df['close_50_ewma'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 100-day exponentially weighted moving average of the closing price
    df['close_100_ewma'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 20-day exponentially weighted standard deviation of the closing price
    df['close_20_ewstd'] = df['close'].ewm(span=20, adjust=False).std()
    
    # Calculate the 50-day exponentially weighted standard deviation of the volume
    df['volume_50_ewstd'] = df['volume'].ewm(span=50, adjust=False).std()
    
    # Factor 1: Difference between the closing price and the 50-day exponentially weighted moving average, divided by the 20-day exponentially weighted standard deviation
    factor_1 = (df['close'] - df['close_50_ewma']) / df['close_20_ewstd']
    
    # Factor 2: Difference between the 50-day exponentially weighted moving average and the 100-day exponentially weighted moving average, divided by the 20-day exponentially weighted standard deviation
    factor_2 = (df['close_50_ewma'] - df['close_100_ewma']) / df['close_20_ewstd']
    
    # Factor 3: Volume divided by its 50-day exponentially weighted standard deviation
    factor_3 = df['volume'] / df['volume_50_ewstd']
    
    # Combine the factors into a single factor value
    factor_values = factor_1 + factor_2 + factor_3
    
    return factor_values
