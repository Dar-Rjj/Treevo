```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum (current close - close 5 days ago)
    df['momentum_5d'] = df['close'].diff(5)
    
    # Calculate the 10-day exponential moving average (EMA) of the closing price
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 20-day simple moving average (SMA) of the closing price
    df['sma_20d'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 10-day volatility (standard deviation of the last 10 days' returns)
    df['volatility_10d'] = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the volume shock (current volume / average volume over the last 20 days)
    df['volume_shock'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the 5-day weighted moving average with recent days having more weight
    weights = np.array([1, 2, 3, 4, 5])
    df['wma_5d'] = df['close'].rolling(window=5).apply(lambda x: (weights * x).sum() / weights.sum(), raw=True)
    
    # Calculate the positive difference between the current close price and 5-day WMA
    df['diff_pos_wma'] = df[['close', 'wma_5d']].apply(lambda x: max(x['close'] - x['wma_5d'], 0), axis=1)
    
    # Non-linear transformation: square root of the positive difference
    df['nonlinear_diff'] = np.sqrt(df['diff_pos_wma'])
    
    # Calculate the 5-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=5).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=5).mean()
    rs = gain / loss
    df['rsi_5d'] = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=10).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=10).sum()
    df['mfi_10d'] = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Calculate the 10-day Chande momentum oscillator
    df['cmo_10d'] = ((df['close'].diff(1) > 0).rolling(window=10).sum() - (df['close'].diff(1) < 0).rolling(window=10).sum()) / 10 * 100
    
    # Factor is a weighted combination of the 5-day momentum, 10-day EMA, 20-day SMA, 10-day volatility, volume shock, non-linear difference, RSI, MFI, and CMO
    factor_value = (0.2 * df['momentum_5d'] 
                    + 0.1 * (df['close'] - df['ema_10d']) 
                    + 0.1 * (df['close'] - df['sma_20d']) 
                    + 0.1 * df['volatility_10d'] 
                    + 0.1 * df['volume_shock'] 
                    + 0.1 * df['nonlinear_diff'] 
                    + 0.1 * df['rsi_5d'] 
                    + 0.1 * df['mfi_10d'] 
                    + 0.1 * df['cmo_10d'])
    
    return factor_value
```
