def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 200-day exponential moving averages of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the 50-day standard deviation of the closing price
    df['close_50_std'] = df['close'].rolling(window=50).std()

    # Non-linear transformation: Logarithm of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_50_std']).apply(np.log1p)

    # Non-linear transformation: Square root of the difference between the 100-day EMA and the 200-day EMA
    factor_2 = ((df['close_100_ema'] - df['close_200_ema']) / df['close_50_std']).apply(np.sqrt)

    # Non-linear transformation: Cube root of the volume divided by its 100-day standard deviation
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    factor_3 = (df['volume'] / df['volume_100_std']).apply(np.cbrt)

    # Calculate the rate of change over 5 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=5)

    # Calculate the market sentiment as the difference between the 100-day EMA and the 200-day EMA
    market_sentiment = df['close_100_ema'] - df['close_200_ema']

    # Introduce a new factor: the rate of change over 10 days for the volume
    factor_5 = df['volume'].pct_change(periods=10)

    # Combine the factors into a single factor value with weighted contributions, including market sentiment and the new factor
    factor_values = 0.4 * factor_1 + 0.3 * factor_2 + 0.15 * factor_3 + 0.08 * factor_4 + 0.05 * market_sentiment + 0.02 * factor_5

    return factor_values
