```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate average trading price
    avg_trading_price = (df['open'] + df['close']) / 2
    
    # Calculate the difference between the close and the volume-weighted average price
    volatility_proxy = (df['high'] - df['low']).rolling(window=5).mean()
    volume_weighted_price = df['close'] * df['volume']
    normalized_volume_weighted_price = volume_weighted_price / df['volume'].rolling(window=5).mean()
    
    # Factor: (Avg Trading Price - VWAP) / Volatility Proxy
    factor = (avg_trading_price - normalized_volume_weighted_price) / (volatility_proxy + 1e-7)
    
    return factor
```
