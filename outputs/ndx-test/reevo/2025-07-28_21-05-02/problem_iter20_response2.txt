def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 200-day and 500-day exponential moving averages of the closing price
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    df['close_500_ema'] = df['close'].ewm(span=500, adjust=False).mean()
    
    # Calculate the 120-day standard deviation of the closing price
    df['close_120_std'] = df['close'].rolling(window=120).std()
    
    # Calculate the 200-day standard deviation of the volume
    df['volume_200_std'] = df['volume'].rolling(window=200).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 200-day EMA
    factor_1 = ((df['close'] - df['close_200_ema']) / df['close_120_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 200-day EMA and the 500-day EMA
    factor_2 = ((df['close_200_ema'] - df['close_500_ema']) / df['close_120_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 200-day standard deviation
    factor_3 = (df['volume'] / df['volume_200_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 90 days for the 200-day EMA
    factor_4 = df['close_200_ema'].pct_change(periods=90)
    
    # Calculate the market sentiment as the difference between the 200-day EMA and the 500-day EMA
    market_sentiment = df['close_200_ema'] - df['close_500_ema']
    
    # Additional market indicators
    # Calculate the 100-day simple moving average of the high price
    df['high_100_sma'] = df['high'].rolling(window=100).mean()
    
    # Calculate the 100-day simple moving average of the low price
    df['low_100_sma'] = df['low'].rolling(window=100).mean()
    
    # Calculate the 100-day simple moving average of the open price
    df['open_100_sma'] = df['open'].rolling(window=100).mean()
    
    # Non-linear transformation: Square of the difference between the high price and its 100-day SMA
    factor_5 = ((df['high'] - df['high_100_sma']) / df['close_120_std']).apply(np.square)
    
    # Non-linear transformation: Exponential of the difference between the low price and its 100-day SMA
    factor_6 = ((df['low'] - df['low_100_sma']) / df['close_120_std']).apply(np.exp)
    
    # Non-linear transformation: Absolute value of the difference between the open price and its 100-day SMA
    factor_7 = ((df['open'] - df['open_100_sma']) / df['close_120_std']).apply(abs)
    
    # Dynamic weighting: Use the absolute value of the 90-day percentage change in the 200-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.2 * dynamic_weight + 0.15 * dynamic_weight + 0.1 * dynamic_weight + 0.08 * dynamic_weight + 0.07 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.2 * factor_1 + 0.15 * factor_2 + 0.1 * factor_3 + 0.08 * factor_4 + 0.07 * market_sentiment + 0.05 * factor_5 + 0.05 * factor_6 + 0.05 * factor_7) * (dynamic_weight / total_weight)
    
    return factor_values
