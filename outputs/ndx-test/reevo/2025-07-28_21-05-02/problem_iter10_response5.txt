```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 200-day exponential moving average of the closing price
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 400-day exponential moving average of the closing price
    df['close_400_ema'] = df['close'].ewm(span=400, adjust=False).mean()
    
    # Calculate the 100-day standard deviation of the closing price
    df['close_100_std'] = df['close'].rolling(window=100).std()
    
    # Calculate the 200-day standard deviation of the volume
    df['volume_200_std'] = df['volume'].rolling(window=200).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 200-day EMA
    factor_1 = ((df['close'] - df['close_200_ema']) / df['close_100_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the absolute difference between the 200-day EMA and the 400-day EMA
    factor_2 = (np.abs(df['close_200_ema'] - df['close_400_ema']) / df['close_100_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 200-day standard deviation
    factor_3 = (df['volume'] / df['volume_200_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 20 days for the 200-day EMA
    factor_4 = df['close_200_ema'].pct_change(periods=20)
    
    # Calculate the market sentiment as the normalized difference between the 200-day EMA and the 400-day EMA
    market_sentiment = (df['close_200_ema'] - df['close_400_ema']) / df['close_100_std']
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment
    factor_values = 0.3 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.1 * factor_4 + 0.1 * market_sentiment
    
    return factor_values
```
