def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day exponential moving average of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 250-day exponential moving average of the closing price
    df['close_250_ema'] = df['close'].ewm(span=250, adjust=False).mean()
    
    # Calculate the 50-day standard deviation of the closing price
    df['close_50_std'] = df['close'].rolling(window=50).std()
    
    # Calculate the 100-day standard deviation of the volume
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_50_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 100-day EMA and the 250-day EMA
    factor_2 = ((df['close_100_ema'] - df['close_250_ema']) / df['close_50_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 100-day standard deviation
    factor_3 = (df['volume'] / df['volume_100_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 15 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=15)
    
    # Calculate the 30-day simple moving average of the high and low prices
    df['hl_30_sma'] = (df['high'] + df['low']).rolling(window=30).mean()
    
    # Non-linear transformation: Inverse of the 30-day SMA of the high and low prices
    factor_5 = (1 / (df['hl_30_sma'] + 1e-6)).apply(np.log1p)
    
    # Calculate the 20-day money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=20).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=20).sum()
    df['mfi_20d'] = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Non-linear transformation: Sine of the 20-day MFI
    factor_6 = np.sin(df['mfi_20d'])
    
    # Combine the factors into a single factor value with optimized weights
    factor_values = (0.2 * factor_1 
                    + 0.2 * factor_2 
                    + 0.1 * factor_3 
                    + 0.1 * factor_4 
                    + 0.2 * factor_5 
                    + 0.2 * factor_6)
    
    return factor_values
