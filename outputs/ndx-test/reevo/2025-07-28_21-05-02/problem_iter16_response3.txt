```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 200-day exponential moving averages of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 60-day standard deviation of the closing price
    df['close_60_std'] = df['close'].rolling(window=60).std()
    
    # Calculate the 100-day standard deviation of the volume
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_60_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 100-day EMA and the 200-day EMA
    factor_2 = ((df['close_100_ema'] - df['close_200_ema']) / df['close_60_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 100-day standard deviation
    factor_3 = (df['volume'] / df['volume_100_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 30 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=30)
    
    # Calculate the market sentiment as the difference between the 100-day EMA and the 200-day EMA
    market_sentiment = df['close_100_ema'] - df['close_200_ema']
    
    # Additional market indicators
    # Calculate the 50-day simple moving average of the high price
    df['high_50_sma'] = df['high'].rolling(window=50).mean()
    
    # Calculate the 50-day simple moving average of the low price
    df['low_50_sma'] = df['low'].rolling(window=50).mean()
    
    # Calculate the 50-day simple moving average of the open price
    df['open_50_sma'] = df['open'].rolling(window=50).mean()
    
    # Non-linear transformation: Square of the difference between the high price and its 50-day SMA
    factor_5 = ((df['high'] - df['high_50_sma']) / df['close_60_std']).apply(np.square)
    
    # Non-linear transformation: Exponential of the difference between the low price and its 50-day SMA
    factor_6 = ((df['low'] - df['low_50_sma']) / df['close_60_std']).apply(np.exp)
    
    # Non-linear transformation: Absolute value of the difference between the open price and its 50-day SMA
    factor_7 = ((df['open'] - df['open_50_sma']) / df['close_60_std']).apply(abs)
    
    # Dynamic weighting: Use the absolute value of the 30-day percentage change in the 100-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.3 * factor_1 + 0.2 * factor_2 + 0.1 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment + 0.05 * factor_5 + 0.05 * factor_6 + 0.05 * factor_7) * (dynamic_weight / total_weight)
    
    return factor_values
```
