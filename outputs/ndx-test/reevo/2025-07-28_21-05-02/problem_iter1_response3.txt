def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    close_diff = df['close'].diff()
    # Calculate a momentum factor by subtracting the 14-day mean of the close prices from the close prices
    momentum_factor = df['close'] - df['close'].rolling(window=14).mean()
    # Calculate the ratio of volume change to the stock's price movement, which could indicate strong buying or selling pressure
    vol_pressure = df['volume'].pct_change() / (df['close'] - df['open'])
    # Combine the factors into a single alpha factor series
    alpha_factor = (close_diff + momentum_factor) * vol_pressure
    return alpha_factor.dropna()
