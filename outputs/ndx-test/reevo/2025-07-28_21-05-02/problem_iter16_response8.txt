```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 30-day and 120-day exponential moving averages of the closing price
    df['close_30_ema'] = df['close'].ewm(span=30, adjust=False).mean()
    df['close_120_ema'] = df['close'].ewm(span=120, adjust=False).mean()
    
    # Calculate the 15-day standard deviation of the closing price
    df['close_15_std'] = df['close'].rolling(window=15).std()
    
    # Calculate the 30-day standard deviation of the volume
    df['volume_30_std'] = df['volume'].rolling(window=30).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 30-day EMA
    factor_1 = ((df['close'] - df['close_30_ema']) / df['close_15_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 30-day EMA and the 120-day EMA
    factor_2 = ((df['close_30_ema'] - df['close_120_ema']) / df['close_15_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 30-day standard deviation
    factor_3 = (df['volume'] / df['volume_30_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 10 days for the 30-day EMA
    factor_4 = df['close_30_ema'].pct_change(periods=10)
    
    # Calculate the market sentiment as the difference between the 30-day EMA and the 120-day EMA
    market_sentiment = df['close_30_ema'] - df['close_120_ema']
    
    # Dynamic weighting: Use the absolute value of the 10-day percentage change in the 30-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.4 * dynamic_weight + 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.4 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * (dynamic_weight / total_weight)
    
    return factor_values
```
