def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 200-day and 400-day exponential moving averages of the closing price
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    df['close_400_ema'] = df['close'].ewm(span=400, adjust=False).mean()
    
    # Calculate the 150-day standard deviation of the closing price for a more robust volatility measure
    df['close_150_std'] = df['close'].rolling(window=150).std()
    
    # Calculate the 300-day standard deviation of the volume for a more robust volatility measure
    df['volume_300_std'] = df['volume'].rolling(window=300).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 200-day EMA
    factor_1 = ((df['close'] - df['close_200_ema']) / df['close_150_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 200-day EMA and the 400-day EMA
    factor_2 = ((df['close_200_ema'] - df['close_400_ema']) / df['close_150_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 300-day standard deviation
    factor_3 = (df['volume'] / df['volume_300_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 30 days for the 200-day EMA
    factor_4 = df['close_200_ema'].pct_change(periods=30)
    
    # Calculate the market sentiment as the difference between the 200-day EMA and the 400-day EMA
    market_sentiment = df['close_200_ema'] - df['close_400_ema']
    
    # Dynamic weighting: Use the absolute value of the 30-day percentage change in the 200-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = dynamic_weight + 0.5 * dynamic_weight + 0.2 * dynamic_weight + 0.1 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (factor_1 * dynamic_weight + 
                     factor_2 * 0.5 * dynamic_weight + 
                     factor_3 * 0.2 * dynamic_weight + 
                     factor_4 * 0.1 * dynamic_weight + 
                     market_sentiment * 0.1 * dynamic_weight) * normalized_weight
    
    return factor_values
