def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 40-day and 160-day exponential moving averages of the closing price
    df['close_40_ema'] = df['close'].ewm(span=40, adjust=False).mean()
    df['close_160_ema'] = df['close'].ewm(span=160, adjust=False).mean()
    
    # Calculate the 30-day standard deviation of the closing price
    df['close_30_std'] = df['close'].rolling(window=30).std()
    
    # Calculate the 50-day standard deviation of the volume
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 40-day EMA
    factor_1 = ((df['close'] - df['close_40_ema']) / df['close_30_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 40-day EMA and the 160-day EMA
    factor_2 = ((df['close_40_ema'] - df['close_160_ema']) / df['close_30_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 50-day standard deviation
    factor_3 = (df['volume'] / df['volume_50_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 20 days for the 40-day EMA
    factor_4 = df['close_40_ema'].pct_change(periods=20)
    
    # Calculate the market sentiment as the difference between the 40-day EMA and the 160-day EMA
    market_sentiment = df['close_40_ema'] - df['close_160_ema']
    
    # Incorporate additional volatility measure: 20-day standard deviation of the closing price
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Dynamic weighting: Use the absolute value of the 20-day percentage change in the 40-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.4 * dynamic_weight + 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment and additional volatility
    factor_values = (0.4 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * (dynamic_weight / total_weight) + 0.05 * df['close_20_std']
    
    return factor_values
