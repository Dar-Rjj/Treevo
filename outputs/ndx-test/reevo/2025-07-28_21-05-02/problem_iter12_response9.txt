```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 300-day exponential moving average of the closing price
    df['close_300_ema'] = df['close'].ewm(span=300, adjust=False).mean()
    
    # Calculate the 600-day exponential moving average of the closing price
    df['close_600_ema'] = df['close'].ewm(span=600, adjust=False).mean()
    
    # Calculate the 150-day standard deviation of the closing price
    df['close_150_std'] = df['close'].rolling(window=150).std()
    
    # Calculate the 300-day standard deviation of the volume
    df['volume_300_std'] = df['volume'].rolling(window=300).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 300-day EMA
    factor_1 = ((df['close'] - df['close_300_ema']) / df['close_150_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the absolute difference between the 300-day EMA and the 600-day EMA
    factor_2 = (np.abs(df['close_300_ema'] - df['close_600_ema']) / df['close_150_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 300-day standard deviation
    factor_3 = (df['volume'] / df['volume_300_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 30 days for the 300-day EMA
    factor_4 = df['close_300_ema'].pct_change(periods=30)
    
    # Calculate the market sentiment as the normalized difference between the 300-day EMA and the 600-day EMA
    market_sentiment = (df['close_300_ema'] - df['close_600_ema']) / df['close_150_std']
    
    # Incorporate the open, high, and low prices to create a range factor
    df['range'] = df['high'] - df['low']
    range_factor = (df['range'] / df['close_150_std']).apply(np.sqrt)
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment and range factor
    factor_values = 0.25 * factor_1 + 0.25 * factor_2 + 0.15 * factor_3 + 0.10 * factor_4 + 0.10 * market_sentiment + 0.15 * range_factor
    
    return factor_values
```
