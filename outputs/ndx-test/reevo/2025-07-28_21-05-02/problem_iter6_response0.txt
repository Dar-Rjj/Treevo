```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for momentum
    close_diff = df['close'].diff()
    
    # Adaptive moving average using a 50-day EMA to identify the trend
    ma_50_ema = df['close'].ewm(span=50, adjust=False).mean()
    trend_strength = df['close'] - ma_50_ema
    
    # Calculate the relative strength (RS) as the ratio of the current close price to the 50-day EMA
    rs = df['close'] / ma_50_ema
    
    # Calculate the 10-day standard deviation of the closing prices to measure volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate the volume ratio as the ratio of current volume to the 20-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the deviation from the average of high and low for the day
    avg_high_low = (df['high'] + df['low']) / 2
    deviation_from_avg = df['close'] - avg_high_low
    
    # Calculate the 50-day and 100-day EMA for further trend analysis
    ma_100_ema = df['close'].ewm(span=100, adjust=False).mean()
    long_term_trend = ma_50_ema - ma_100_ema
    
    # Volume-weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Market sentiment factor using the ratio of the VWAP to the 50-day EMA
    market_sentiment = vwap / ma_50_ema
    
    # Combine the momentum, trend strength, relative strength, and volume-weighted deviation with market sentiment
    alpha_factor = (close_diff / (volatility + 1e-7)) * rs * trend_strength * (deviation_from_avg * volume_ratio) * market_sentiment * long_term_trend
    
    return alpha_factor
```
