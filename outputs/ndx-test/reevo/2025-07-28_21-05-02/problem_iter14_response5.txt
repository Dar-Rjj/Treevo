```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 100-day exponential moving averages of the closing price
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the 50-day standard deviation of the volume
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Non-linear transformation: Logarithm of the difference between the closing price and the 50-day EMA
    factor_1 = ((df['close'] - df['close_50_ema']) / df['close_20_std']).apply(np.log1p)
    
    # Non-linear transformation: Square root of the difference between the 50-day EMA and the 100-day EMA
    factor_2 = ((df['close_50_ema'] - df['close_100_ema']) / df['close_20_std']).apply(np.sqrt)
    
    # Non-linear transformation: Cube root of the volume divided by its 50-day standard deviation
    factor_3 = (df['volume'] / df['volume_50_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 5 days for the 50-day EMA
    factor_4 = df['close_50_ema'].pct_change(periods=5)
    
    # Calculate the market sentiment as the difference between the 50-day EMA and the 100-day EMA
    market_sentiment = df['close_50_ema'] - df['close_100_ema']
    
    # Introduce a new factor: the rate of change over 5 days for the volume
    factor_5 = df['volume'].pct_change(periods=5)
    
    # Introduce a new momentum factor: the rate of change over 5 days for the closing price
    factor_6 = df['close'].pct_change(periods=5)
    
    # Dynamic weighting: Use the absolute value of the 5-day percentage change in the 50-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.3 * dynamic_weight + 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.025 * dynamic_weight + 0.025 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment and the new factors
    factor_values = (
        0.3 * factor_1 + 
        0.3 * factor_2 + 
        0.2 * factor_3 + 
        0.1 * factor_4 + 
        0.05 * market_sentiment + 
        0.025 * factor_5 + 
        0.025 * factor_6
    ) * (dynamic_weight / total_weight)
    
    return factor_values
```
