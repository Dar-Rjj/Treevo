def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 200-day exponential moving averages of the closing price for medium-term trends
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 50-day standard deviation of the closing price for a more robust volatility measure
    df['close_50_std'] = df['close'].rolling(window=50).std()
    
    # Calculate the 100-day standard deviation of the volume for a more robust volatility measure
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    
    # Non-linear transformation: Cube root of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_50_std']).apply(np.cbrt)
    
    # Non-linear transformation: Exponential of the absolute difference between the 100-day EMA and the 200-day EMA
    factor_2 = abs((df['close_100_ema'] - df['close_200_ema']) / df['close_50_std']).apply(np.exp)
    
    # Non-linear transformation: Square root of the volume divided by its 100-day standard deviation
    factor_3 = (df['volume'] / df['volume_100_std']).apply(np.sqrt)
    
    # Calculate the rate of change over 30 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=30)
    
    # Calculate the market sentiment as the difference between the 100-day EMA and the 200-day EMA
    market_sentiment = df['close_100_ema'] - df['close_200_ema']
    
    # Introduce a new momentum factor: the rate of change over 60 days for the closing price
    factor_5 = df['close'].pct_change(periods=60)
    
    # Introduce a new factor: the absolute value of the difference between the high and low price over 60 days
    df['hl_diff_60'] = (df['high'] - df['low']).rolling(window=60).mean().apply(np.abs)
    factor_6 = df['hl_diff_60']
    
    # Dynamic weighting: Use the absolute value of the 30-day percentage change in the 100-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.4 * dynamic_weight + 0.3 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.025 * dynamic_weight + 0.025 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment and the new factors
    factor_values = (
        0.4 * factor_1 + 
        0.3 * factor_2 + 
        0.1 * factor_3 + 
        0.05 * factor_4 + 
        0.05 * market_sentiment + 
        0.025 * factor_5 + 
        0.025 * factor_6
    ) * normalized_weight
    
    return factor_values
