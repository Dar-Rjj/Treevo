```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day simple moving average of the close price as a measure of momentum
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the daily return to use as a measure of volatility
    daily_return = df['close'].pct_change()
    
    # Calculate the relative strength index (RSI) over a 14-day period to measure momentum
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the positive or negative movement of the closing price relative to the open
    price_movement = df['close'] - df['open']
    
    # Calculate a weighted volume based on the day's price movement. Large movements (positive or negative) with high volume are more significant
    weighted_volume = abs(price_movement) * df['volume']
    
    # Incorporate liquidity by using the ratio of the close price to the average of the high and low prices
    liquidity_ratio = df['close'] / ((df['high'] + df['low']) / 2)
    
    # The alpha factor combines RSI, daily return, weighted volume, and liquidity ratio
    alpha_factor = (rsi + daily_return) * (liquidity_ratio / (weighted_volume + 1e-7)) * (df['close'] / sma_20)
    
    return alpha_factor
```
