```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 100-day exponential moving averages of the closing price
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 20-day standard deviation of the closing price for volatility
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the 50-day standard deviation of the volume for volume anomaly
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Momentum factor: Difference between the closing price and the 50-day EMA, divided by the 20-day standard deviation
    momentum_factor = (df['close'] - df['close_50_ema']) / df['close_20_std']
    
    # Trend strength factor: Difference between the 50-day EMA and the 100-day EMA, divided by the 20-day standard deviation
    trend_strength_factor = (df['close_50_ema'] - df['close_100_ema']) / df['close_20_std']
    
    # Volume anomaly factor: Volume divided by its 50-day standard deviation
    volume_anomaly_factor = df['volume'] / df['volume_50_std']
    
    # Volatility factor: 20-day standard deviation of the closing price
    volatility_factor = df['close_20_std']
    
    # Combine the factors into a single factor value with weighted contributions
    factor_values = 0.4 * momentum_factor + 0.3 * trend_strength_factor + 0.2 * volume_anomaly_factor + 0.1 * volatility_factor
    
    return factor_values
```
