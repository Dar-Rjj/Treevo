```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 120-day exponential moving average of the closing price
    df['close_120_ema'] = df['close'].ewm(span=120, adjust=False).mean()
    
    # Calculate the 240-day exponential moving average of the closing price
    df['close_240_ema'] = df['close'].ewm(span=240, adjust=False).mean()
    
    # Calculate the 60-day standard deviation of the closing price
    df['close_60_std'] = df['close'].rolling(window=60).std()
    
    # Calculate the 120-day standard deviation of the volume
    df['volume_120_std'] = df['volume'].rolling(window=120).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 120-day EMA
    factor_1 = ((df['close'] - df['close_120_ema']) / df['close_60_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 120-day EMA and the 240-day EMA
    factor_2 = ((df['close_120_ema'] - df['close_240_ema']) / df['close_60_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 120-day standard deviation
    factor_3 = (df['volume'] / df['volume_120_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 5 days for the 120-day EMA
    factor_4 = df['close_120_ema'].pct_change(periods=5)
    
    # Calculate the market sentiment as the difference between the 120-day EMA and the 240-day EMA
    market_sentiment = df['close_120_ema'] - df['close_240_ema']
    
    # Non-linear transformation: Exponential of the market sentiment
    factor_5 = market_sentiment.apply(np.exp)
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment
    factor_values = 0.2 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.1 * factor_4 + 0.3 * factor_5
    
    return factor_values
```
