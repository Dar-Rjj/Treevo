def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day exponential moving average of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    
    # Calculate the 200-day exponential moving average of the closing price
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 50-day standard deviation of the closing price
    df['close_50_std'] = df['close'].rolling(window=50).std()
    
    # Calculate the 100-day standard deviation of the volume
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    
    # Calculate the 10-day momentum of the closing price
    df['momentum_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day volatility ratio
    df['volatility_ratio_20'] = df['high'].rolling(window=20).max() / df['low'].rolling(window=20).min()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_50_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 100-day EMA and the 200-day EMA
    factor_2 = ((df['close_100_ema'] - df['close_200_ema']) / df['close_50_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 100-day standard deviation
    factor_3 = (df['volume'] / df['volume_100_std']).apply(np.cbrt)
    
    # Non-linear transformation: Exponential of the 10-day momentum
    factor_4 = df['momentum_10'].apply(np.exp)
    
    # Non-linear transformation: Inverse of the 20-day volatility ratio
    factor_5 = (1 / df['volatility_ratio_20']).apply(np.log1p)
    
    # Combine the factors into a single factor value with weighted contributions
    factor_values = 0.3 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4 + 0.1 * factor_5
    
    return factor_values
