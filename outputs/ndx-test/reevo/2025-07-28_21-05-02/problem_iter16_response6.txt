```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day exponential moving average of the closing price
    df['close_20_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 50-day exponential moving average of the closing price
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the closing price
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the 20-day standard deviation of the volume
    df['volume_20_std'] = df['volume'].rolling(window=20).std()
    
    # Non-linear transformation: Logarithm of the difference between the closing price and the 20-day EMA
    factor_1 = ((df['close'] - df['close_20_ema']) / df['close_10_std']).apply(np.log1p)
    
    # Non-linear transformation: Square root of the difference between the 20-day EMA and the 50-day EMA
    factor_2 = ((df['close_20_ema'] - df['close_50_ema']) / df['close_10_std']).apply(np.sqrt)
    
    # Non-linear transformation: Cube root of the volume divided by its 20-day standard deviation
    factor_3 = (df['volume'] / df['volume_20_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 5 days for the 20-day EMA
    factor_4 = df['close_20_ema'].pct_change(periods=5)
    
    # Calculate the market sentiment as the difference between the 20-day EMA and the 50-day EMA
    market_sentiment = df['close_20_ema'] - df['close_50_ema']
    
    # Introduce a new factor: the rate of change over 5 days for the volume
    factor_5 = df['volume'].pct_change(periods=5)
    
    # Introduce a new momentum factor: the rate of change over 5 days for the closing price
    factor_6 = df['close'].pct_change(periods=5)
    
    # Incorporate the open, high, and low prices to create a range factor
    df['range'] = df['high'] - df['low']
    range_factor = (df['range'] / df['close_10_std']).apply(np.sqrt)
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment and the new factors
    factor_values = (
        0.3 * factor_1 + 
        0.2 * factor_2 + 
        0.15 * factor_3 + 
        0.1 * factor_4 + 
        0.05 * market_sentiment + 
        0.05 * factor_5 + 
        0.05 * factor_6 + 
        0.1 * range_factor
    )
    
    return factor_values
```
