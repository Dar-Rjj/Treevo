```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, indicative of daily price range
    daily_range = df['high'] - df['low']
    
    # Calculate the positive or negative movement of the closing price relative to the open
    price_movement = df['close'] - df['open']
    
    # Calculate a weighted volume based on the day's price movement. Large movements (positive or negative) with high volume are more significant
    weighted_volume = abs(price_movement) * df['volume']
    
    # The alpha factor is the ratio of the daily range to the weighted volume, representing how much the stock moved for a given level of trading activity
    factor = daily_range / (weighted_volume + 1e-7)
    
    return factor
```
