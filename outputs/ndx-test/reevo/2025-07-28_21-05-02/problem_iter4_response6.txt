```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 100-day exponential moving averages of the closing price for trend analysis
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()

    # Calculate the 20-day and 50-day standard deviations of the closing price and volume for volatility normalization
    df['close_20_std'] = df['close'].rolling(window=20).std()
    df['volume_50_std'] = df['volume'].rolling(window=50).std()

    # Factor 1: Momentum factor, difference between the closing price and the 50-day EMA, normalized by the 20-day standard deviation
    factor_1 = (df['close'] - df['close_50_ema']) / df['close_20_std']

    # Factor 2: Trend strength factor, difference between the 50-day EMA and the 100-day EMA, normalized by the 20-day standard deviation
    factor_2 = (df['close_50_ema'] - df['close_100_ema']) / df['close_20_std']

    # Factor 3: Volume momentum factor, current volume divided by its 50-day standard deviation
    factor_3 = df['volume'] / df['volume_50_std']

    # Factor 4: Momentum of the 50-day EMA, rate of change over 5 days
    factor_4 = df['close_50_ema'].pct_change(periods=5)

    # Combine the factors into a single factor value with weighted contributions
    factor_values = 0.3 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4

    return factor_values
```
