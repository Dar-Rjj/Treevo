```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    daily_return = df['close'].diff()
    
    # Calculate the standard deviation of daily returns over a 5-day window
    daily_volatility = daily_return.rolling(window=5).std()
    
    # Calculate the RSI (Relative Strength Index) based on the daily returns with a 14-day period
    rsi = 100 - (100 / (1 + (daily_return.rolling(window=14).apply(lambda x: (x[x>0].sum())/(abs(x[x<0]).sum()) if x[x<0].sum() != 0 else 1)))
    
    # Alpha factor: Daily volatility divided by RSI to get a sense of momentum adjusted for recent price stability
    factor = daily_volatility / (rsi + 1e-7)
    
    return factor
```
