```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 90-day and 180-day exponential moving averages of the closing price
    df['close_90_ema'] = df['close'].ewm(span=90, adjust=False).mean()
    df['close_180_ema'] = df['close'].ewm(span=180, adjust=False).mean()
    
    # Calculate the 60-day standard deviation of the closing price
    df['close_60_std'] = df['close'].rolling(window=60).std()
    
    # Calculate the 90-day standard deviation of the volume
    df['volume_90_std'] = df['volume'].rolling(window=90).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 90-day EMA
    factor_1 = ((df['close'] - df['close_90_ema']) / df['close_60_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 90-day EMA and the 180-day EMA
    factor_2 = ((df['close_90_ema'] - df['close_180_ema']) / df['close_60_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 90-day standard deviation
    factor_3 = (df['volume'] / df['volume_90_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 10 days for the 90-day EMA
    factor_4 = df['close_90_ema'].pct_change(periods=10)
    
    # Calculate the market sentiment as the difference between the 90-day EMA and the 180-day EMA
    market_sentiment = df['close_90_ema'] - df['close_180_ema']
    
    # Momentum factor: Calculate the 20-day momentum of the closing price
    factor_5 = df['close'].pct_change(periods=20)
    
    # Dynamic weighting: Use the absolute value of the 10-day percentage change in the 90-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.3 * dynamic_weight + 0.25 * dynamic_weight + 0.2 * dynamic_weight + 0.1 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment and momentum
    factor_values = (0.3 * factor_1 + 0.25 * factor_2 + 0.2 * factor_3 + 0.1 * factor_4 + 0.1 * market_sentiment + 0.05 * factor_5) * (dynamic_weight / total_weight)
    
    return factor_values
```
