def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 300-day exponential moving averages of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_300_ema'] = df['close'].ewm(span=300, adjust=False).mean()

    # Calculate the 100-day standard deviation of the closing price for a more adaptive volatility measure
    df['close_100_std'] = df['close'].rolling(window=100).std()
    
    # Calculate the 200-day standard deviation of the volume for a more robust volatility measure
    df['volume_200_std'] = df['volume'].rolling(window=200).std()
    
    # Non-linear transformation: Square of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_100_std']).apply(np.square)
    
    # Non-linear transformation: Cube of the difference between the 100-day EMA and the 300-day EMA
    factor_2 = ((df['close_100_ema'] - df['close_300_ema']) / df['close_100_std']).apply(lambda x: x ** 3)
    
    # Non-linear transformation: Exponential of the volume divided by its 200-day standard deviation
    factor_3 = (df['volume'] / df['volume_200_std']).apply(np.exp)
    
    # Calculate the rate of change over 50 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=50)
    
    # Calculate the market sentiment as the ratio of the 100-day EMA to the 300-day EMA
    market_sentiment = df['close_100_ema'] / df['close_300_ema']
    
    # Dynamic weighting: Use the absolute value of the 50-day percentage change in the 100-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.4 * dynamic_weight + 0.3 * dynamic_weight + 0.15 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.4 * factor_1 + 0.3 * factor_2 + 0.15 * factor_3 + 0.1 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
