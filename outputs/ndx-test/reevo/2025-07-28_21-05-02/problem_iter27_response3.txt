def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 400-day and 700-day exponential moving averages of the closing price
    df['close_400_ema'] = df['close'].ewm(span=400, adjust=False).mean()
    df['close_700_ema'] = df['close'].ewm(span=700, adjust=False).mean()
    
    # Calculate the 300-day standard deviation of the closing price for a more robust volatility measure
    df['close_300_std'] = df['close'].rolling(window=300).std()
    
    # Calculate the 500-day standard deviation of the volume for a more robust volatility measure
    df['volume_500_std'] = df['volume'].rolling(window=500).std()
    
    # Non-linear transformation: Cube root of the difference between the closing price and the 400-day EMA
    factor_1 = ((df['close'] - df['close_400_ema']) / df['close_300_std']).apply(np.cbrt)
    
    # Non-linear transformation: Exponential of the absolute difference between the 400-day EMA and the 700-day EMA
    factor_2 = abs((df['close_400_ema'] - df['close_700_ema']) / df['close_300_std']).apply(np.exp)
    
    # Non-linear transformation: Square root of the volume divided by its 500-day standard deviation
    factor_3 = (df['volume'] / df['volume_500_std']).apply(np.sqrt)
    
    # Calculate the rate of change over 90 days for the 400-day EMA
    factor_4 = df['close_400_ema'].pct_change(periods=90)
    
    # Calculate the high-low price difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the 200-day average of the high-low price difference
    df['high_low_200_avg'] = df['high_low_diff'].rolling(window=200).mean()
    
    # Non-linear transformation: Logarithm of the ratio of the current high-low difference to the 200-day average high-low difference
    factor_5 = (df['high_low_diff'] / df['high_low_200_avg']).apply(np.log)
    
    # Calculate the momentum as the percentage change in the close price over 200 days
    factor_6 = df['close'].pct_change(periods=200)
    
    # Dynamic weighting: Use the absolute value of the 90-day percentage change in the 400-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.6 * dynamic_weight + 0.2 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including momentum and high-low difference
    factor_values = (0.6 * factor_1 + 0.2 * factor_2 + 0.1 * factor_3 + 0.05 * factor_4 + 0.05 * factor_5 + 0.05 * factor_6) * normalized_weight
    
    return factor_values
