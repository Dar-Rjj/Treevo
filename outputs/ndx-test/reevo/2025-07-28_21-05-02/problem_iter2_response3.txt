```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day moving average and standard deviation of the closing price
    df['close_20_ma'] = df['close'].rolling(window=20).mean()
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the momentum factor adjusted for volatility
    momentum_volatility_adjusted = (df['close'] - df['close_20_ma']) / df['close_20_std']
    
    # Calculate the difference between the closing price and the average of high and low for the day
    avg_high_low = (df['high'] + df['low']) / 2
    deviation_from_avg = df['close'] - avg_high_low
    
    # Calculate the volume ratio as the current volume divided by the 20-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Create a volume-adjusted price movement factor
    volume_adjusted_price_movement = deviation_from_avg * volume_ratio
    
    # Combine the momentum-volatility-adjusted factor and the volume-adjusted price movement factor
    alpha_factor = momentum_volatility_adjusted + volume_adjusted_price_movement
    
    return alpha_factor
```
