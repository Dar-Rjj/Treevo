def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 15-day and 70-day exponential moving averages of the closing price
    df['close_15_ema'] = df['close'].ewm(span=15, adjust=False).mean()
    df['close_70_ema'] = df['close'].ewm(span=70, adjust=False).mean()
    
    # Calculate the adaptive standard deviation of the closing price
    df['close_adaptive_std'] = df['close'].rolling(window=int(2.5 * np.sqrt(df['close_15_ema']) + 7)).std()
    
    # Calculate the 25-day standard deviation of the volume
    df['volume_25_std'] = df['volume'].rolling(window=25).std()
    
    # Non-linear transformation: Cube root of the difference between the closing price and the 15-day EMA
    factor_1 = ((df['close'] - df['close_15_ema']) / df['close_adaptive_std']).apply(np.cbrt)
    
    # Non-linear transformation: Logarithm of the difference between the 15-day EMA and the 70-day EMA
    factor_2 = ((df['close_15_ema'] - df['close_70_ema']) / df['close_adaptive_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 25-day standard deviation
    factor_3 = (df['volume'] / df['volume_25_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 15 days for the 15-day EMA
    factor_4 = df['close_15_ema'].pct_change(periods=15)
    
    # Calculate the market sentiment as the difference between the 15-day EMA and the 70-day EMA
    market_sentiment = df['close_15_ema'] - df['close_70_ema']
    
    # Incorporate additional volatility measure: 10-day standard deviation of the closing price
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Dynamic weighting: Use the absolute value of the 15-day percentage change in the 15-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.45 * dynamic_weight + 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.025 * dynamic_weight + 0.025 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment and additional volatility
    factor_values = (0.45 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.025 * factor_4 + 0.025 * market_sentiment) * (dynamic_weight / total_weight) + 0.05 * df['close_10_std']
    
    return factor_values
