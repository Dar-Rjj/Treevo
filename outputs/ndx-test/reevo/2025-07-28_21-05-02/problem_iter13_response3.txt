def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 200-day exponential moving averages of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 50-day standard deviation of the closing price
    df['close_50_std'] = df['close'].rolling(window=50).std()
    
    # Calculate the 100-day standard deviation of the volume
    df['volume_100_std'] = df['volume'].rolling(window=100).std()
    
    # Non-linear transformation: Square of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_50_std']).apply(np.square)
    
    # Non-linear transformation: Exponential of the difference between the 100-day EMA and the 200-day EMA
    factor_2 = ((df['close_100_ema'] - df['close_200_ema']) / df['close_50_std']).apply(np.exp)
    
    # Non-linear transformation: Cube root of the volume divided by its 100-day standard deviation
    factor_3 = (df['volume'] / df['volume_100_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 10 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=10)
    
    # Calculate the market sentiment as the difference between the 100-day EMA and the 200-day EMA
    market_sentiment = df['close_100_ema'] - df['close_200_ema']
    
    # Dynamic weighting: Use the absolute value of the 10-day percentage change in the 100-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.4 * dynamic_weight + 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.4 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * (dynamic_weight / total_weight)
    
    return factor_values
