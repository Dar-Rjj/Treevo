def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 150-day and 300-day exponential moving averages of the closing price
    df['close_150_ema'] = df['close'].ewm(span=150, adjust=False).mean()
    df['close_300_ema'] = df['close'].ewm(span=300, adjust=False).mean()
    
    # Calculate the 80-day standard deviation of the closing price
    df['close_80_std'] = df['close'].rolling(window=80).std()
    
    # Calculate the 150-day standard deviation of the volume
    df['volume_150_std'] = df['volume'].rolling(window=150).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 150-day EMA
    factor_1 = ((df['close'] - df['close_150_ema']) / df['close_80_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 150-day EMA and the 300-day EMA
    factor_2 = ((df['close_150_ema'] - df['close_300_ema']) / df['close_80_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 150-day standard deviation
    factor_3 = (df['volume'] / df['volume_150_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 20 days for the 150-day EMA
    factor_4 = df['close_150_ema'].pct_change(periods=20)
    
    # Calculate the market sentiment as the difference between the 150-day EMA and the 300-day EMA
    market_sentiment = df['close_150_ema'] - df['close_300_ema']
    
    # Dynamic weighting: Use the absolute value of the 20-day percentage change in the 150-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    total_weight = 0.4 * dynamic_weight + 0.3 * dynamic_weight + 0.2 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.4 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * (dynamic_weight / total_weight)
    
    # Integrate momentum and volume dynamics
    momentum = df['close'].pct_change(periods=10)
    volume_dynamics = df['volume'].pct_change(periods=10)
    
    # Apply dynamic weighting to the momentum and volume dynamics
    factor_values = 0.7 * factor_values + 0.15 * momentum * dynamic_weight + 0.15 * volume_dynamics * dynamic_weight
    
    return factor_values
