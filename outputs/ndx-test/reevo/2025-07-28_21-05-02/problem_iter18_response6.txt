def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day and 50-day exponential moving averages of the closing price for more responsive signals
    df['close_20_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the closing price for a more responsive volatility measure
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the 20-day standard deviation of the volume for a more responsive volatility measure
    df['volume_20_std'] = df['volume'].rolling(window=20).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 20-day EMA
    factor_1 = ((df['close'] - df['close_20_ema']) / df['close_10_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 20-day EMA and the 50-day EMA
    factor_2 = ((df['close_20_ema'] - df['close_50_ema']) / df['close_10_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 20-day standard deviation
    factor_3 = (df['volume'] / df['volume_20_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 5 days for the 20-day EMA for more responsiveness
    factor_4 = df['close_20_ema'].pct_change(periods=5)
    
    # Calculate the market sentiment as the difference between the 20-day EMA and the 50-day EMA
    market_sentiment = df['close_20_ema'] - df['close_50_ema']
    
    # Dynamic weighting: Use the absolute value of the 5-day percentage change in the 20-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.5 * dynamic_weight + 0.3 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.5 * factor_1 + 0.3 * factor_2 + 0.1 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
