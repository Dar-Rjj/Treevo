```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 30-day, 90-day, and 180-day exponential moving average of the closing price
    df['close_30_ema'] = df['close'].ewm(span=30, adjust=False).mean()
    df['close_90_ema'] = df['close'].ewm(span=90, adjust=False).mean()
    df['close_180_ema'] = df['close'].ewm(span=180, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the closing price
    df['close_10_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the volume
    df['volume_30_std'] = df['volume'].rolling(window=30).std()
    
    # Factor 1: Difference between the closing price and the 30-day EMA, divided by the 10-day standard deviation
    factor_1 = (df['close'] - df['close_30_ema']) / df['close_10_std']
    
    # Factor 2: Difference between the 30-day EMA and the 90-day EMA, divided by the 10-day standard deviation
    factor_2 = (df['close_30_ema'] - df['close_90_ema']) / df['close_10_std']
    
    # Factor 3: Difference between the 90-day EMA and the 180-day EMA, divided by the 10-day standard deviation
    factor_3 = (df['close_90_ema'] - df['close_180_ema']) / df['close_10_std']
    
    # Factor 4: Volume divided by its 30-day standard deviation, with a non-linear transformation (square root)
    factor_4 = (df['volume'] / df['volume_30_std']).apply(np.sqrt)
    
    # Factor 5: 30-day EMA trend (difference between current and previous day's 30-day EMA)
    factor_5 = df['close_30_ema'].diff()
    
    # Factor 6: 7-day price momentum
    df['momentum_7d'] = df['close'].diff(7)
    factor_6 = df['momentum_7d'] / df['close_10_std']
    
    # Combine the factors into a single factor value with meaningful weights
    factor_values = (0.2 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4 + 0.1 * factor_5 + 0.1 * factor_6)
    
    return factor_values
```
