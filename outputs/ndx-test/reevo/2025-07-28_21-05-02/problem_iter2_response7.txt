```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average (EMA) for the close price to capture momentum
    df['ema'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the positive difference between the current close price and EMA
    df['diff_pos_ema'] = df[['close', 'ema']].apply(lambda x: max(x['close'] - x['ema'], 0), axis=1)
    
    # Calculate the volume ratio (VR) over the 5 days period with an exponential weighting
    df['vr_ema'] = df['volume'].ewm(span=5, adjust=False).mean() / df['volume']
    
    # Calculate the market breadth as the difference between the number of advancing and declining issues
    df['advancing'] = (df['close'] > df['open']).astype(int)
    df['declining'] = (df['close'] < df['open']).astype(int)
    df['market_breadth'] = df['advancing'].rolling(window=5).sum() - df['declining'].rolling(window=5).sum()
    
    # Factor is the product of positive close-EMA difference, modified VR, and market breadth
    factor_value = df['diff_pos_ema'] * df['vr_ema'] * df['market_breadth']
    
    return factor_value
```
