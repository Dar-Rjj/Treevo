```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum (current close - close 5 days ago)
    df['momentum_5d'] = df['close'].diff(5)
    
    # Calculate the 10-day exponential moving average (EMA) of the closing price
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the volatility (standard deviation of the last 10 days' returns)
    df['volatility_10d'] = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the volume shock (current volume / average volume over the last 10 days)
    df['volume_shock'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the 5-day weighted moving average with recent days having more weight
    df['wma_5d'] = df['close'].rolling(window=5).apply(lambda x: ((np.arange(1,6)*x).sum()/15), raw=True)
    
    # Calculate the positive difference between the current close price and 5-day WMA
    df['diff_pos_wma'] = df[['close', 'wma_5d']].apply(lambda x: max(x['close'] - x['wma_5d'], 0), axis=1)
    
    # Non-linear transformation: square root of the positive difference
    df['nonlinear_diff'] = np.sqrt(df['diff_pos_wma'])
    
    # Factor is the product of the 5-day momentum, 10-day EMA, 10-day volatility, volume shock, and non-linear difference
    factor_value = df['momentum_5d'] * (df['close'] - df['ema_10d']) * df['volatility_10d'] * df['volume_shock'] * df['nonlinear_diff']
    
    return factor_value
```
