```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day exponential moving average of the closing price for momentum
    df['close_10_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 150-day exponential moving average of the closing price
    df['close_150_ema'] = df['close'].ewm(span=150, adjust=False).mean()
    
    # Calculate the 200-day exponential moving average of the closing price
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 45-day standard deviation of the closing price
    df['close_45_std'] = df['close'].rolling(window=45).std()
    
    # Calculate the 150-day standard deviation of the volume
    df['volume_150_std'] = df['volume'].rolling(window=150).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 150-day EMA
    factor_1 = ((df['close'] - df['close_150_ema']) / df['close_45_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm (log1p) of the difference between the 150-day EMA and the 200-day EMA
    factor_2 = ((df['close_150_ema'] - df['close_200_ema']) / df['close_45_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 150-day standard deviation
    factor_3 = (df['volume'] / df['volume_150_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 7 days for the 10-day EMA to capture short-term momentum
    factor_4 = df['close_10_ema'].pct_change(periods=7)
    
    # Introduce a new factor: the rate of change over 5 days for the volume
    factor_5 = df['volume'].pct_change(periods=5)
    
    # Calculate the market sentiment as the difference between the 150-day EMA and the 200-day EMA
    market_sentiment = df['close_150_ema'] - df['close_200_ema']
    
    # Combine the factors into a single factor value with balanced weighted contributions, including market sentiment and the new factor
    factor_values = 0.2 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4 + 0.1 * market_sentiment + 0.1 * factor_5
    
    return factor_values
```
