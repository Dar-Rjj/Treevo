```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the weighted moving average with recent days having more weight
    df['wma'] = df['close'].rolling(window=5).apply(lambda x: ((np.arange(1,6)*x).sum()/15), raw=True)
    
    # Calculate the positive difference between the current close price and 5-day WMA
    df['diff_pos'] = df[['close', 'wma']].apply(lambda x: max(x['close'] - x['wma'], 0), axis=1)
    
    # Calculate a modified volume ratio (VR) over the 5 days period
    df['vr'] = df['volume'].rolling(window=5).apply(lambda x: (x[-1] / np.mean(x)), raw=True)
    
    # Factor is the product of positive close-WMA difference and modified VR
    factor_value = df['diff_pos'] * df['vr']
    
    return factor_value
```
