```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day, 30-day, and 60-day exponential moving average of the closing price
    df['close_10_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_30_ema'] = df['close'].ewm(span=30, adjust=False).mean()
    df['close_60_ema'] = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the 5-day standard deviation of the closing price
    df['close_5_std'] = df['close'].rolling(window=5).std()
    
    # Calculate the 10-day standard deviation of the volume
    df['volume_10_std'] = df['volume'].rolling(window=10).std()
    
    # Factor 1: Difference between the closing price and the 10-day EMA, divided by the 5-day standard deviation
    factor_1 = (df['close'] - df['close_10_ema']) / df['close_5_std']
    
    # Factor 2: Difference between the 10-day EMA and the 30-day EMA, divided by the 5-day standard deviation
    factor_2 = (df['close_10_ema'] - df['close_30_ema']) / df['close_5_std']
    
    # Factor 3: Difference between the 30-day EMA and the 60-day EMA, divided by the 5-day standard deviation
    factor_3 = (df['close_30_ema'] - df['close_60_ema']) / df['close_5_std']
    
    # Factor 4: Volume divided by its 10-day standard deviation, with a non-linear transformation (cube root)
    factor_4 = (df['volume'] / df['volume_10_std']).apply(np.cbrt)
    
    # Factor 5: 10-day EMA trend (difference between current and previous day's 10-day EMA)
    factor_5 = df['close_10_ema'].diff()
    
    # Factor 6: 5-day price momentum
    df['momentum_5d'] = df['close'].diff(5)
    factor_6 = df['momentum_5d'] / df['close_5_std']
    
    # Combine the factors into a single factor value with meaningful weights
    factor_values = (0.2 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4 + 0.1 * factor_5 + 0.1 * factor_6)
    
    return factor_values
```
