```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day moving average of the closing price
    df['close_50_ma'] = df['close'].rolling(window=50).mean()
    
    # Calculate the 100-day moving average of the closing price
    df['close_100_ma'] = df['close'].rolling(window=100).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    # Calculate the 50-day standard deviation of the volume
    df['volume_50_std'] = df['volume'].rolling(window=50).std()
    
    # Factor 1: Difference between the closing price and the 50-day moving average, divided by the 20-day standard deviation
    factor_1 = (df['close'] - df['close_50_ma']) / df['close_20_std']
    
    # Factor 2: Difference between the 50-day moving average and the 100-day moving average, divided by the 20-day standard deviation
    factor_2 = (df['close_50_ma'] - df['close_100_ma']) / df['close_20_std']
    
    # Factor 3: Volume divided by its 50-day standard deviation
    factor_3 = df['volume'] / df['volume_50_std']
    
    # Combine the factors into a single factor value
    factor_values = factor_1 + factor_2 + factor_3
    
    return factor_values
```
