```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for momentum
    close_diff = df['close'].diff()
    
    # Calculate the 50-day moving average to identify the trend
    ma_50 = df['close'].rolling(window=50).mean()
    trend_strength = df['close'] - ma_50
    
    # Calculate the relative strength (RS) as the ratio of the current close price to the 50-day moving average
    rs = df['close'] / ma_50
    
    # Calculate the 10-day standard deviation of the closing prices to measure volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate the volume ratio as the ratio of current volume to the 20-day moving average of volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the deviation from the average of high and low for the day
    avg_high_low = (df['high'] + df['low']) / 2
    deviation_from_avg = df['close'] - avg_high_low
    
    # Combine the momentum, trend strength, relative strength, and volume-weighted deviation
    alpha_factor = (close_diff / (volatility + 1e-7)) * rs * trend_strength * (deviation_from_avg * volume_ratio)
    
    return alpha_factor
```
