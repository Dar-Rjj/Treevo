```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 250-day exponential moving average of the closing price
    df['close_250_ema'] = df['close'].ewm(span=250, adjust=False).mean()
    
    # Calculate the 600-day exponential moving average of the closing price
    df['close_600_ema'] = df['close'].ewm(span=600, adjust=False).mean()
    
    # Calculate the 150-day standard deviation of the closing price
    df['close_150_std'] = df['close'].rolling(window=150).std()
    
    # Calculate the 250-day standard deviation of the volume
    df['volume_250_std'] = df['volume'].rolling(window=250).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 250-day EMA, normalized by 150-day close std
    factor_1 = ((df['close'] - df['close_250_ema']) / df['close_150_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 250-day EMA and the 600-day EMA, normalized by 150-day close std
    factor_2 = ((df['close_250_ema'] - df['close_600_ema']) / df['close_150_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 250-day standard deviation
    factor_3 = (df['volume'] / df['volume_250_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 20 days for the 250-day EMA
    factor_4 = df['close_250_ema'].pct_change(periods=20)
    
    # Calculate the 20-day momentum of the closing price
    df['momentum_20'] = df['close'].pct_change(periods=20)
    
    # Calculate the 20-day volatility of the closing price
    df['volatility_20'] = df['close'].pct_change().rolling(window=20).std()
    
    # Include a new factor: 20-day rate of change in volume
    df['volume_momentum_20'] = df['volume'].pct_change(periods=20)
    
    # Combine the factors into a single factor value with weighted contributions
    factor_values = 0.3 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.1 * factor_4 + 0.05 * df['momentum_20'] - 0.05 * df['volatility_20'] + 0.05 * df['volume_momentum_20']
    
    return factor_values
```
