def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day exponential moving averages of the closing price for short-term dynamics
    df['close_10_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    df['close_30_ema'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 120-day and 240-day exponential moving averages of the closing price for long-term trends
    df['close_120_ema'] = df['close'].ewm(span=120, adjust=False).mean()
    df['close_240_ema'] = df['close'].ewm(span=240, adjust=False).mean()
    
    # Calculate the 50-day standard deviation of the closing price
    df['close_50_std'] = df['close'].rolling(window=50).std()
    
    # Calculate the 180-day standard deviation of the volume
    df['volume_180_std'] = df['volume'].rolling(window=180).std()
    
    # Non-linear transformation: Logarithm of the difference between the closing price and the 10-day EMA
    factor_1 = ((df['close'] - df['close_10_ema']) / df['close_50_std']).apply(np.log1p)
    
    # Non-linear transformation: Square root of the difference between the 120-day EMA and the 240-day EMA
    factor_2 = ((df['close_120_ema'] - df['close_240_ema']) / df['close_50_std']).apply(np.sqrt)
    
    # Non-linear transformation: Cube root of the volume divided by its 180-day standard deviation
    factor_3 = (df['volume'] / df['volume_180_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 15 days for the 10-day EMA
    factor_4 = df['close_10_ema'].pct_change(periods=15)
    
    # Calculate the market sentiment as the difference between the 10-day EMA and the 30-day EMA
    market_sentiment = df['close_10_ema'] - df['close_30_ema']
    
    # Introduce a new factor: the rate of change over 7 days for the volume
    factor_5 = df['volume'].pct_change(periods=7)
    
    # Introduce a new momentum factor: the rate of change over 20 days for the closing price
    factor_6 = df['close'].pct_change(periods=20)
    
    # Introduce a new factor: the absolute value of the difference between the high and low price over 30 days
    df['hl_diff_30'] = df['high'] - df['low']
    factor_7 = df['hl_diff_30'].rolling(window=30).mean().apply(np.abs)
    
    # Adaptive weighting based on the 10-day EMA's rate of change
    adaptive_weight = abs(factor_4)
    
    # Adjust the adaptive weights to ensure they sum up to 1
    total_weight = 0.2 * adaptive_weight + 0.2 * adaptive_weight + 0.15 * adaptive_weight + 0.1 * adaptive_weight + 0.05 * adaptive_weight + 0.025 * adaptive_weight + 0.025 * adaptive_weight + 0.25 * adaptive_weight
    normalized_weight = adaptive_weight / total_weight
    
    # Combine the factors into a single factor value with adaptively weighted contributions, including market sentiment and the new factors
    factor_values = (
        0.2 * factor_1 + 
        0.2 * factor_2 + 
        0.15 * factor_3 + 
        0.1 * factor_4 + 
        0.05 * market_sentiment + 
        0.025 * factor_5 + 
        0.025 * factor_6 + 
        0.25 * factor_7
    ) * normalized_weight
    
    return factor_values
