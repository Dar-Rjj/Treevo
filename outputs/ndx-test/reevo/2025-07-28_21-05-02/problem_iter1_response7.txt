```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close prices
    price_diff = df['close'] - df['close'].shift(1)
    
    # Calculate a simple moving average of the volume over the last 5 days
    sma_volume = df['volume'].rolling(window=5).mean()
    
    # Generate the alpha factor as the ratio of the price difference to the rolling mean of the volume
    # This attempts to capture momentum adjusted for average trading volume
    alpha_factor = price_diff / (sma_volume + 1e-7)
    
    return alpha_factor
```
