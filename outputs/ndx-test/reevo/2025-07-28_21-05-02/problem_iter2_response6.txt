```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range as the difference between high and low prices
    daily_range = df['high'] - df['low']
    
    # Calculate the positive or negative movement of the closing price relative to the open
    price_movement = df['close'] - df['open']
    
    # Calculate a weighted volume based on the day's price movement. Large movements (positive or negative) with high volume are more significant
    weighted_volume = abs(price_movement) * df['volume']
    
    # Calculate the 14-day moving average of the close price for a momentum indicator
    momentum_14 = df['close'].rolling(window=14).mean()
    
    # Calculate the 14-day standard deviation of the close price for a volatility measure
    volatility_14 = df['close'].rolling(window=14).std()
    
    # Calculate the 14-day average true range to measure market sentiment
    atr_14 = df[['high', 'low', 'close']].rolling(window=14).apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].iloc[-1]), abs(x['low'] - x['close'].iloc[-1])), raw=True)
    
    # Combine the factors into a single alpha factor series
    # The alpha factor is the ratio of the daily range to the weighted volume, adjusted by the momentum, volatility, and ATR
    factor = (daily_range / (weighted_volume + 1e-7)) * (momentum_14 / (volatility_14 + 1e-7)) * (atr_14 / (df['close'] + 1e-7))
    
    return factor
```
