```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 300-day exponential moving average of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_300_ema'] = df['close'].ewm(span=300, adjust=False).mean()
    
    # Calculate the 50-day and 100-day standard deviation of the closing price
    df['close_50_std'] = df['close'].rolling(window=50).std()
    df['close_100_std'] = df['close'].rolling(window=100).std()
    
    # Calculate the 75-day standard deviation of the volume
    df['volume_75_std'] = df['volume'].rolling(window=75).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_50_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the ratio between the 100-day EMA and the 300-day EMA
    factor_2 = (df['close_100_ema'] / df['close_300_ema']).apply(np.log)
    
    # Non-linear transformation: Cube root of the volume divided by its 75-day standard deviation
    factor_3 = (df['volume'] / df['volume_75_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 20 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=20)
    
    # Calculate the market sentiment as the percentage difference between the 100-day EMA and the 300-day EMA
    market_sentiment = (df['close_100_ema'] - df['close_300_ema']) / df['close_300_ema']
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment
    factor_values = 0.4 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.1 * factor_4 + 0.1 * market_sentiment
    
    return factor_values
```
