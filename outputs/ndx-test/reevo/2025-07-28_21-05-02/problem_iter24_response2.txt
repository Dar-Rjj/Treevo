def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 200-day exponential moving averages of the closing price
    df['close_50_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 90-day standard deviation of the closing price for a more robust volatility measure
    df['close_90_std'] = df['close'].rolling(window=90).std()
    
    # Calculate the 180-day standard deviation of the volume for a more robust volatility measure
    df['volume_180_std'] = df['volume'].rolling(window=180).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 50-day EMA
    factor_1 = ((df['close'] - df['close_50_ema']) / df['close_90_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 50-day EMA and the 200-day EMA
    factor_2 = ((df['close_50_ema'] - df['close_200_ema']) / df['close_90_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 180-day standard deviation
    factor_3 = (df['volume'] / df['volume_180_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 60 days for the 50-day EMA
    factor_4 = df['close_50_ema'].pct_change(periods=60)
    
    # Calculate the market sentiment as the difference between the 50-day EMA and the 200-day EMA
    market_sentiment = df['close_50_ema'] - df['close_200_ema']
    
    # Dynamic weighting: Use the absolute value of the 60-day percentage change in the 50-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.7 * dynamic_weight + 0.15 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.7 * factor_1 + 0.15 * factor_2 + 0.05 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
