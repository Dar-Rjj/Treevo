def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 150-day exponential moving average of the closing price
    df['close_150_ema'] = df['close'].ewm(span=150, adjust=False).mean()
    
    # Calculate the 300-day exponential moving average of the closing price
    df['close_300_ema'] = df['close'].ewm(span=300, adjust=False).mean()
    
    # Calculate the 70-day standard deviation of the closing price
    df['close_70_std'] = df['close'].rolling(window=70).std()
    
    # Calculate the 150-day standard deviation of the volume
    df['volume_150_std'] = df['volume'].rolling(window=150).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 150-day EMA
    factor_1 = ((df['close'] - df['close_150_ema']) / df['close_70_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the absolute difference between the 150-day EMA and the 300-day EMA
    factor_2 = (np.abs(df['close_150_ema'] - df['close_300_ema']) / df['close_70_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 150-day standard deviation
    factor_3 = (df['volume'] / df['volume_150_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 10 days for the 150-day EMA
    factor_4 = df['close_150_ema'].pct_change(periods=10)
    
    # Calculate the market sentiment as the normalized difference between the 150-day EMA and the 300-day EMA
    market_sentiment = (df['close_150_ema'] - df['close_300_ema']) / df['close_70_std']
    
    # Combine the factors into a single factor value with balanced weighted contributions, including market sentiment
    factor_values = 0.2 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4 + 0.2 * market_sentiment
    
    return factor_values
