def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 250-day exponential moving averages of the closing price
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_250_ema'] = df['close'].ewm(span=250, adjust=False).mean()
    
    # Calculate the 90-day standard deviation of the closing price for a more adaptive volatility measure
    df['close_90_std'] = df['close'].rolling(window=90).std()
    
    # Non-linear transformation: Cube root of the difference between the closing price and the 100-day EMA
    factor_1 = ((df['close'] - df['close_100_ema']) / df['close_90_std']).apply(np.cbrt)
    
    # Non-linear transformation: Exponential of the absolute difference between the 100-day EMA and the 250-day EMA
    factor_2 = abs((df['close_100_ema'] - df['close_250_ema']) / df['close_90_std']).apply(np.exp)
    
    # Non-linear transformation: Square root of the volume divided by its 150-day standard deviation
    df['volume_150_std'] = df['volume'].rolling(window=150).std()
    factor_3 = (df['volume'] / df['volume_150_std']).apply(np.sqrt)
    
    # Calculate the rate of change over 30 days for the 100-day EMA
    factor_4 = df['close_100_ema'].pct_change(periods=30)
    
    # Calculate the market sentiment as the normalized difference between the 100-day EMA and the 250-day EMA
    market_sentiment = (df['close_100_ema'] - df['close_250_ema']) / df['close_90_std']
    
    # Dynamic weighting: Use the absolute value of the 30-day percentage change in the 100-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.5 * dynamic_weight + 0.2 * dynamic_weight + 0.2 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.5 * factor_1 + 0.2 * factor_2 + 0.2 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
