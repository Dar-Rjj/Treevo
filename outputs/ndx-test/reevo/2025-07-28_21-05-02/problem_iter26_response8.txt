def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 150-day and 400-day exponential moving averages of the closing price
    df['close_150_ema'] = df['close'].ewm(span=150, adjust=False).mean()
    df['close_400_ema'] = df['close'].ewm(span=400, adjust=False).mean()
    
    # Calculate the 120-day standard deviation of the closing price for a more robust volatility measure
    df['close_120_std'] = df['close'].rolling(window=120).std()
    
    # Calculate the 250-day standard deviation of the volume for a more robust volatility measure
    df['volume_250_std'] = df['volume'].rolling(window=250).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 150-day EMA
    factor_1 = ((df['close'] - df['close_150_ema']) / df['close_120_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 150-day EMA and the 400-day EMA
    factor_2 = ((df['close_150_ema'] - df['close_400_ema']) / df['close_120_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 250-day standard deviation
    factor_3 = (df['volume'] / df['volume_250_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 65 days for the 150-day EMA
    factor_4 = df['close_150_ema'].pct_change(periods=65)
    
    # Calculate the market sentiment as the difference between the 150-day EMA and the 400-day EMA
    market_sentiment = df['close_150_ema'] - df['close_400_ema']
    
    # Dynamic weighting: Use the absolute value of the 65-day percentage change in the 150-day EMA to weight the factors
    dynamic_weight = abs(factor_4)
    
    # Adjust the dynamic weights to ensure they sum up to 1
    total_weight = 0.5 * dynamic_weight + 0.3 * dynamic_weight + 0.1 * dynamic_weight + 0.05 * dynamic_weight + 0.05 * dynamic_weight
    normalized_weight = dynamic_weight / total_weight
    
    # Combine the factors into a single factor value with dynamically weighted contributions, including market sentiment
    factor_values = (0.5 * factor_1 + 0.3 * factor_2 + 0.1 * factor_3 + 0.05 * factor_4 + 0.05 * market_sentiment) * normalized_weight
    
    return factor_values
