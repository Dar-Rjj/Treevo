```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the 20-day moving average of the closing price,
    # then divide it by the 20-day standard deviation of the closing price to normalize the factor.
    # This aims to capture the momentum adjusted for volatility.
    df['close_20_ma'] = df['close'].rolling(window=20).mean()
    df['close_20_std'] = df['close'].rolling(window=20).std()
    
    return (df['close'] - df['close_20_ma']) / df['close_20_std']
```
