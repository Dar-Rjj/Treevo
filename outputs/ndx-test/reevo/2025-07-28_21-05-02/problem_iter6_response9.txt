```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 100-day and 200-day exponential moving average of the closing price for long-term trends
    df['close_100_ema'] = df['close'].ewm(span=100, adjust=False).mean()
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the 50-day standard deviation of the closing price to measure volatility
    df['close_50_std'] = df['close'].rolling(window=50).std()

    # Calculate the 100-day standard deviation of the volume to measure volume volatility
    df['volume_100_std'] = df['volume'].rolling(window=100).std()

    # Factor 1: Difference between the closing price and the 100-day exponential moving average, divided by the 50-day standard deviation
    factor_1 = (df['close'] - df['close_100_ema']) / df['close_50_std']

    # Factor 2: Difference between the 100-day exponential moving average and the 200-day exponential moving average, divided by the 50-day standard deviation
    factor_2 = (df['close_100_ema'] - df['close_200_ema']) / df['close_50_std']

    # Factor 3: Volume divided by its 100-day standard deviation, with a square root transformation
    factor_3 = np.sqrt(df['volume'] / df['volume_100_std'])

    # Factor 4: 50-day momentum, which is the current close price minus the 50-day ago close price, divided by the 50-day standard deviation
    df['close_50_lag'] = df['close'].shift(50)
    factor_4 = (df['close'] - df['close_50_lag']) / df['close_50_std']

    # Combine the factors into a single factor value with weighted contributions
    factor_values = 0.3 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.2 * factor_4

    return factor_values
```
