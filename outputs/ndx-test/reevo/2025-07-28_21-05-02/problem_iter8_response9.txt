```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 200-day exponential moving average of the closing price
    df['close_200_ema'] = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 400-day exponential moving average of the closing price
    df['close_400_ema'] = df['close'].ewm(span=400, adjust=False).mean()
    
    # Calculate the 100-day standard deviation of the closing price
    df['close_100_std'] = df['close'].rolling(window=100).std()
    
    # Calculate the 200-day standard deviation of the volume
    df['volume_200_std'] = df['volume'].rolling(window=200).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 200-day EMA
    factor_1 = ((df['close'] - df['close_200_ema']) / df['close_100_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 200-day EMA and the 400-day EMA
    factor_2 = ((df['close_200_ema'] - df['close_400_ema']) / df['close_100_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 200-day standard deviation
    factor_3 = (df['volume'] / df['volume_200_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 20 days for the 200-day EMA
    factor_4 = df['close_200_ema'].pct_change(periods=20)
    
    # Calculate the 20-day momentum of the closing price
    factor_5 = df['close'].pct_change(periods=20)
    
    # Calculate the 10-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = gain / loss
    df['rsi_10d'] = 100 - (100 / (1 + rs))
    
    # Non-linear transformation: Square of the RSI
    factor_6 = df['rsi_10d'].apply(lambda x: x**2)
    
    # Calculate the 10-day money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=10).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=10).sum()
    df['mfi_10d'] = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Non-linear transformation: Cube of the MFI
    factor_7 = df['mfi_10d'].apply(lambda x: x**3)
    
    # Combine the factors into a single factor value with weighted contributions
    factor_values = (0.2 * factor_1 
                    + 0.2 * factor_2 
                    + 0.1 * factor_3 
                    + 0.1 * factor_4 
                    + 0.1 * factor_5 
                    + 0.1 * factor_6 
                    + 0.2 * factor_7)
    
    return factor_values
```
