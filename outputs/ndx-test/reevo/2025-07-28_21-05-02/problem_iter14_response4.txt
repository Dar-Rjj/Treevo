```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 60-day and 120-day exponential moving averages of the closing price
    df['close_60_ema'] = df['close'].ewm(span=60, adjust=False).mean()
    df['close_120_ema'] = df['close'].ewm(span=120, adjust=False).mean()
    
    # Calculate the 30-day and 60-day standard deviations of the closing price and volume
    df['close_30_std'] = df['close'].rolling(window=30).std()
    df['volume_60_std'] = df['volume'].rolling(window=60).std()
    
    # Non-linear transformation: Square root of the difference between the closing price and the 60-day EMA
    factor_1 = ((df['close'] - df['close_60_ema']) / df['close_30_std']).apply(np.sqrt)
    
    # Non-linear transformation: Logarithm of the difference between the 60-day EMA and the 120-day EMA
    factor_2 = ((df['close_60_ema'] - df['close_120_ema']) / df['close_30_std']).apply(np.log1p)
    
    # Non-linear transformation: Cube root of the volume divided by its 60-day standard deviation
    factor_3 = (df['volume'] / df['volume_60_std']).apply(np.cbrt)
    
    # Calculate the rate of change over 5 days for the 60-day EMA
    factor_4 = df['close_60_ema'].pct_change(periods=5)
    
    # Calculate the market sentiment as the difference between the 60-day EMA and the 120-day EMA
    market_sentiment = df['close_60_ema'] - df['close_120_ema']
    
    # Introduce a new factor: the rate of change over 3 days for the volume
    factor_5 = df['volume'].pct_change(periods=3)
    
    # Combine the factors into a single factor value with weighted contributions, including market sentiment and the new factor
    factor_values = 0.3 * factor_1 + 0.3 * factor_2 + 0.2 * factor_3 + 0.1 * factor_4 + 0.05 * market_sentiment + 0.05 * factor_5
    
    return factor_values
```
