```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the 20-day percentage change in closing price
    momentum = df['close'].pct_change(periods=20)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volatility as the standard deviation of the daily returns over a 20-day window
    volatility = daily_return.rolling(window=20).std()
    
    # Calculate the liquidity as the average volume over a 20-day window
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the 50-day and 200-day moving averages of closing prices
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200
    
    # Calculate the 20-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Combine the momentum, volatility, liquidity, trend strength, and high-low range into an alpha factor
    alpha_factor = (momentum / (volatility + 1e-7)) * (df['volume'] / (liquidity + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7))
    
    # Smooth the alpha factor using a weighted 3-day rolling mean
    smoothed_alpha_factor = alpha_factor.rolling(window=3, win_type='triang').mean()
    
    return smoothed_alpha_factor
```
