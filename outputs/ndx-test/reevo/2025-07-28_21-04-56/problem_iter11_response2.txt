def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 12 and 26-day EMA for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.ewm(span=9, adjust=False).mean()
    macd_diff = macd - signal
    
    # Calculate Bollinger Bands with a 20-day window and 2 standard deviations
    middle_band = df['close'].rolling(window=20).mean()
    std_dev = df['close'].rolling(window=20).std()
    upper_band = middle_band + 2 * std_dev
    lower_band = middle_band - 2 * std_dev
    bollinger_percent = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Calculate the 10-day volume-weighted momentum
    momentum = df['close'].pct_change(periods=10)
    volume_weighted_momentum = (momentum * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()
    
    # Calculate the Parkinson's Volatility
    parkinson_volatility = ((df['high'] / df['low']).apply(np.log).rolling(window=10).std()) * (np.sqrt(4.67))
    
    # Combine the RSI, MACD, Bollinger Bands, and volume-weighted momentum into an alpha factor
    alpha_factor = (rsi / 100) * macd_diff * bollinger_percent * (volume_weighted_momentum / (parkinson_volatility + 1e-7))
    
    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
