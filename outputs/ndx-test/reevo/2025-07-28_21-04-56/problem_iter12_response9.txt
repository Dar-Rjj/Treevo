def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 12 and 26-day EMA for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.ewm(span=9, adjust=False).mean()
    macd_diff = macd - signal

    # Calculate Bollinger Bands with a 20-day window and 2 standard deviations
    middle_band = df['close'].rolling(window=20).mean()
    std_dev = df['close'].rolling(window=20).std()
    upper_band = middle_band + 2 * std_dev
    lower_band = middle_band - 2 * std_dev
    bollinger_percent = (df['close'] - lower_band) / (upper_band - lower_band)

    # Calculate the 10-day volume-weighted momentum
    momentum = df['close'].pct_change(periods=10)
    volume_weighted_momentum = (momentum * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()

    # Calculate the 10-day Average True Range (ATR)
    tr = df[['high', 'low', 'close']].join(df[['high', 'low', 'close']].shift(1)).apply(lambda x: max(x[0], x[3]) - min(x[1], x[4]), axis=1)
    atr = tr.rolling(window=10).mean()

    # Calculate the Chaikin Money Flow (CMF) over a 20-day period
    mfv = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    cmf = mfv.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Calculate the On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()

    # Combine the RSI, MACD, Bollinger Bands, volume-weighted momentum, ATR, CMF, and OBV into an alpha factor
    alpha_factor = (rsi / 100) * macd_diff * bollinger_percent * (volume_weighted_momentum / (atr + 1e-7)) * cmf * (obv / df['volume'].rolling(window=20).mean())

    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()

    return smoothed_alpha_factor
