def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 12-day and 26-day EMA for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_hist = macd_line - signal_line

    # Calculate the 10-day volume-weighted momentum
    vw_momentum = (df['close'].pct_change(periods=10) * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()

    # Calculate the 10-day exponentially smoothed high-low range
    hl_range = (df['high'] - df['low']).ewm(span=10, adjust=False).mean()

    # Calculate the 50-day and 200-day moving averages of closing prices
    ma_50 = df['close'].ewm(span=50, adjust=False).mean()
    ma_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200

    # Calculate the 10-day Parkinson's Volatility
    parkinson_volatility = ((df['high'] / df['low']).apply(np.log).rolling(window=10).std()) * (np.sqrt(4.67))

    # Calculate the 30-day Garman-Klass Volatility
    log_hl = (df['high'] / df['low']).apply(np.log)
    log_co = (df['close'] / df['open']).apply(np.log)
    garman_klass_volatility = (0.5 * log_hl**2 - (2 * np.log(2) - 1) * log_co**2).rolling(window=30).sum() / np.log(2)

    # Combine the RSI, MACD histogram, volume-weighted momentum, smoothed high-low range, trend strength, Parkinson's volatility, and Garman-Klass volatility into an alpha factor
    alpha_factor = (rsi / 100 + macd_hist / parkinson_volatility + vw_momentum / hl_range + trend_strength / (garman_klass_volatility + 1e-7)) / 4

    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()

    return smoothed_alpha_factor
