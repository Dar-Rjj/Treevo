```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change of close price from the previous day
    pct_change_close = df['close'].pct_change()
    
    # Calculate the ratio of today's volume to the 5-day average volume
    volume_5d_ma = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / (volume_5d_ma + 1e-7)
    
    # Calculate the high-low spread as a percentage of the open price
    high_low_spread = (df['high'] - df['low']) / (df['open'] + 1e-7)
    
    # Calculate the ratio of close to open price
    close_open_ratio = df['close'] / (df['open'] + 1e-7)
    
    # Combine the factors: percentage change in close, volume ratio, high-low spread, and close-open ratio
    alpha_factor = (pct_change_close * volume_ratio * high_low_spread * close_open_ratio)
    
    return alpha_factor
```
