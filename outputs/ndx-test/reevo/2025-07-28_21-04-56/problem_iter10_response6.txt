def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    import numpy as np
    from sklearn.linear_model import LinearRegression

    # Calculate the 10-day momentum
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the Parkinson's Volatility
    parkinson_volatility = ((df['high'] / df['low']).apply(np.log).rolling(window=10).std()) * (np.sqrt(4.67))
    
    # Calculate the liquidity as the average volume over a 10-day window
    liquidity = df['volume'].rolling(window=10).mean()
    
    # Calculate the 50-day and 200-day moving averages of closing prices
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200
    
    # Calculate the 10-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate the volume-weighted momentum
    volume_weighted_momentum = (momentum * df['volume']).rolling(window=10).mean() / liquidity
    
    # Combine the indicators into a DataFrame
    features = pd.DataFrame({
        'volume_weighted_momentum': volume_weighted_momentum,
        'parkinson_volatility': parkinson_volatility,
        'liquidity': liquidity,
        'trend_strength': trend_strength,
        'hl_range_mean': hl_range_mean
    }).dropna()

    # Fit a linear regression model to dynamically weight the indicators
    X = features.drop('trend_strength', axis=1)
    y = features['trend_strength']
    model = LinearRegression().fit(X, y)
    weights = pd.Series(model.coef_, index=X.columns)

    # Apply the weights to the indicators
    weighted_features = (features * weights).sum(axis=1)

    # Combine the weighted features into an alpha factor
    alpha_factor = (weighted_features / (parkinson_volatility + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7))
    
    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
