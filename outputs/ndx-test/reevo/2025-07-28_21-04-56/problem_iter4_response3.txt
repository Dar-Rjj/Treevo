```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate 1-day return
    close_change = df['close'].pct_change(1)
    
    # Calculate average volume over the last 5 days and 20 days
    volume_avg_5d = df['volume'].rolling(window=5).mean()
    volume_avg_20d = df['volume'].rolling(window=20).mean()
    
    # Calculate price volatility over the last 5 days and 20 days
    price_volatility_5d = (df['high'] - df['low']).rolling(window=5).std()
    price_volatility_20d = (df['high'] - df['low']).rolling(window=20).std()
    
    # Calculate the 10-day and 30-day moving averages of the closing price
    ma_close_10d = df['close'].rolling(window=10).mean()
    ma_close_30d = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the 10-day and 30-day moving averages
    trend_signal = ma_close_10d - ma_close_30d
    
    # Calculate the adaptive trend strength using 5-day and 20-day windows
    trend_strength_5d = (df['close'] - ma_close_10d) / price_volatility_5d
    trend_strength_20d = (df['close'] - ma_close_30d) / price_volatility_20d
    
    # Combine all factors: 1-day return, average volume over 5 and 20 days, daily volatility, and trend signal
    alpha_factor = (close_change * (volume_avg_5d + volume_avg_20d) / (price_volatility_5d + price_volatility_20d)) + trend_signal + (trend_strength_5d + trend_strength_20d)
    
    return alpha_factor.dropna()
```
