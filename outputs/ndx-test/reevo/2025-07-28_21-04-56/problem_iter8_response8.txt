```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 100-day percentage change in closing price (momentum)
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_100 = df['close'].pct_change(periods=100)

    # Calculate the relative strength as the ratio of 10-day momentum to 100-day momentum
    relative_strength = momentum_10 / (momentum_100 + 1e-7)

    # Calculate the daily return
    daily_return = df['close'].pct_change()

    # Calculate the volatility as the standard deviation of the daily returns over a 30-day window
    volatility = daily_return.rolling(window=30).std()

    # Calculate the liquidity as the average volume over a 30-day window
    liquidity = df['volume'].rolling(window=30).mean()

    # Calculate the 100-day and 250-day moving averages of closing prices
    ma_100 = df['close'].rolling(window=100).mean()
    ma_250 = df['close'].rolling(window=250).mean()

    # Calculate the trend strength as the difference between the 100-day and 250-day moving averages
    trend_strength = ma_100 - ma_250

    # Calculate the 30-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=30).mean()

    # Calculate the volume-weighted momentum
    volume_weighted_momentum = (momentum_10 * df['volume']).rolling(window=30).mean() / (liquidity + 1e-7)

    # Calculate the seasonality as the 60-day rolling mean of the daily returns
    seasonality = daily_return.rolling(window=60).mean()

    # Calculate the sentiment as the 30-day cumulative sum of the daily returns
    sentiment = daily_return.rolling(window=30).sum()

    # Combine the volume-weighted momentum, volatility, liquidity, trend strength, high-low range, seasonality, and sentiment into an alpha factor
    alpha_factor = (volume_weighted_momentum / (volatility + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7)) * (relative_strength + 1) * (seasonality + 1) * (sentiment + 1)

    # Smooth the alpha factor using a 5-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=5, adjust=False).mean()

    return smoothed_alpha_factor
```
