```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day exponential moving averages of closing prices
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 10-day rolling standard deviation to represent volatility
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Calculate the difference between the current close and the 10-day EMA
    df['close_ema_10_diff'] = df['close'] - df['ema_10']
    
    # Normalize the difference by the 10-day price range plus a small constant for numerical stability
    df['normalized_close_ema_10_diff'] = df['close_ema_10_diff'] / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the trend strength as the ratio of the 10-day EMA to the 50-day EMA
    df['trend_strength'] = df['ema_10'] / df['ema_50']
    
    # Calculate the liquidity as the 10-day average volume
    df['liquidity_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day momentum as the percentage change in the closing price
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Combine the normalized difference, volatility, trend strength, liquidity, and momentum
    alpha_factor = (df['normalized_close_ema_10_diff'] * df['volatility_10d']) * df['trend_strength'] * df['liquidity_10d'] * (1 + df['momentum_10d'])
    
    return alpha_factor.dropna()
```
