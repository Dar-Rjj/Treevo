```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open, then scale it by the average of high and low to get a scaled daily movement
    scaled_daily_movement = (df['close'] - df['open']) / ((df['high'] + df['low']) / 2 + 1e-7)
    
    # Calculate the relative change in volume compared to the previous day
    volume_change = (df['volume'] - df['volume'].shift(1)) / (df['volume'].shift(1) + 1e-7)
    
    # Combine both factors: Scaled daily movement and relative volume change
    alpha_factor = scaled_daily_movement * volume_change
    
    return alpha_factor
```
