def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day RSI
    def rsi(series, period):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    rsi_10 = rsi(df['close'], 10)
    rsi_30 = rsi(df['close'], 30)
    
    # Calculate the MACD
    exp1 = df['close'].ewm(span=12, adjust=False).mean()
    exp2 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = exp1 - exp2
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_hist = macd_line - signal_line
    
    # Calculate the 50-day and 200-day moving averages of closing prices
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200
    
    # Calculate the 10-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate the volume-weighted momentum
    momentum = df['close'].pct_change(periods=10)
    volume_weighted_momentum = (momentum * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()
    
    # Calculate the Parkinson's Volatility
    parkinson_volatility = ((df['high'] / df['low']).apply(np.log).rolling(window=10).std()) * (np.sqrt(4.67))
    
    # Combine the volume-weighted momentum, RSI, MACD, trend strength, and high-low range into an alpha factor
    alpha_factor = (volume_weighted_momentum / (parkinson_volatility + 1e-7)) * (rsi_10 / rsi_30) * (macd_hist / (trend_strength + 1e-7)) / (hl_range_mean + 1e-7)
    
    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
