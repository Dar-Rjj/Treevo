```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day momentum as the percentage change in closing price
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_30 = df['close'].pct_change(periods=30)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the 10-day and 30-day volatility as the standard deviation of the daily returns
    volatility_10 = daily_return.rolling(window=10).std()
    volatility_30 = daily_return.rolling(window=30).std()
    
    # Calculate the liquidity as the average volume over a 10-day and 30-day window
    liquidity_10 = df['volume'].rolling(window=10).mean()
    liquidity_30 = df['volume'].rolling(window=30).mean()
    
    # Calculate the 50-day and 200-day moving averages of closing prices
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200
    
    # Calculate the 10-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Combine the short-term and long-term momentum, volatility, liquidity, trend strength, and high-low range into an alpha factor
    alpha_factor = (momentum_10 + momentum_30) / (volatility_10 + volatility_30 + 1e-7) * (df['volume'] / (liquidity_10 + liquidity_30 + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7))
    
    # Smooth the alpha factor using a weighted 3-day rolling mean
    weights = np.array([0.5, 0.3, 0.2])
    smoothed_alpha_factor = alpha_factor.rolling(window=3).apply(lambda x: np.dot(x, weights), raw=True)
    
    return smoothed_alpha_factor
```
