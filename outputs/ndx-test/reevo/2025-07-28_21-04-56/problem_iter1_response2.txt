```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    close_change = df['close'].diff(1)
    volume_avg_5d = df['volume'].rolling(window=5).mean()
    price_volatility = (df['high'] - df['low']).rolling(window=5).std()
    
    # Alpha factor combining 1-day return, average volume over 5 days, and daily volatility
    alpha_factor = close_change * volume_avg_5d / price_volatility
    
    return alpha_factor
```
