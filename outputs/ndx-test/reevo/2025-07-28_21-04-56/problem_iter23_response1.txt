def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    import numpy as np

    # Calculate the 14-day RSI using EWM
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 12-day and 26-day EMA for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_hist = macd_line - signal_line

    # Calculate the 10-day volume-weighted momentum using EWM
    vw_momentum = (df['close'].pct_change(periods=10) * df['volume']).ewm(span=10, adjust=False).mean() / df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate the 50-day and 200-day moving averages of closing prices using EWM
    ma_50 = df['close'].ewm(span=50, adjust=False).mean()
    ma_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200

    # Calculate the 3-day and 7-day moving averages of closing prices using EWM
    ma_3 = df['close'].ewm(span=3, adjust=False).mean()
    ma_7 = df['close'].ewm(span=7, adjust=False).mean()
    
    # Calculate the short-term trend strength as the difference between the 3-day and 7-day moving averages
    short_trend_strength = ma_3 - ma_7

    # Calculate the 3-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=3).mean()

    # Calculate the 14-day ATR (Average True Range) using EWM
    tr = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))], axis=1).max(axis=1)
    atr = tr.ewm(span=14, adjust=False).mean()

    # Calculate the 3-day rate of change in volume
    roc_volume = df['volume'].pct_change(periods=3)

    # Calculate the 14-day Williams %R
    highest_high = df['high'].rolling(window=14).max()
    lowest_low = df['low'].rolling(window=14).min()
    williams_r = (highest_high - df['close']) / (highest_high - lowest_low) * -100

    # Incorporate a 14-day Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi = 100 - (100 / (1 + (positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum())))

    # Adaptive Volatility
    volatility_factor = atr / df['close']

    # Mean Reversion
    mean_reversion = (df['close'] - df['close'].rolling(window=50).mean()) / df['close'].rolling(window=50).std()

    # Sentiment Factor
    sentiment_factor = (rsi - 50) / 50

    # Combine the factors into an alpha factor
    alpha_factor = (rsi / 100 + macd_hist + vw_momentum + trend_strength + short_trend_strength + atr + roc_volume + williams_r + mfi + volatility_factor + mean_reversion + sentiment_factor) / 12
