```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the open and close prices
    diff_open_close = (df['close'] - df['open']).abs()
    
    # Calculate the volatility as the average true range over 14 days
    df['true_range'] = df[['high'-'low', 'high'-'close', 'close'-'low']].max(axis=1)
    average_true_range = df['true_range'].rolling(window=14).mean()
    
    # Normalize the price difference by the average true range
    factor = diff_open_close / (average_true_range + 1e-7)
    
    return factor
```
