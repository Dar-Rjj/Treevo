```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 1-day price change
    price_change = df['close'].diff()
    # Calculate the 5-day momentum (price change over 5 days)
    momentum_5d = df['close'] - df['close'].shift(5)
    # Volume weighted by the absolute value of the price change
    volume_weighted_change = df['volume'] * abs(price_change)
    # Calculate the 3-day average of the volume weighted change for smoothing
    smoothed_volume_weighted_change = volume_weighted_change.rolling(window=3).mean()
    # Calculate the 5-day rolling average of the close price to identify trend
    trend_5d = df['close'].rolling(window=5).mean()
    # Calculate liquidity as the 5-day average of the ratio of volume to high-low range
    liquidity = (df['volume'] / (df['high'] - df['low'])).rolling(window=5).mean()
    
    # Combine the factors: trend, momentum, and liquidity
    alpha_factor = (smoothed_volume_weighted_change * momentum_5d) / (trend_5d * liquidity)
    
    return alpha_factor
```
