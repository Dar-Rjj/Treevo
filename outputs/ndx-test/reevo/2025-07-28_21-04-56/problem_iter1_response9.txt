```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, and close and open prices
    high_low_diff = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']

    # Compute the average volume over a rolling window of 10 days
    avg_volume = df['volume'].rolling(window=10).mean()

    # Alpha factor based on the ratio of closing price deviation to high-low range adjusted by relative volume
    alpha_factor = (close_open_diff / (high_low_diff + 1e-7)) * (df['volume'] / avg_volume)
    return alpha_factor
```
