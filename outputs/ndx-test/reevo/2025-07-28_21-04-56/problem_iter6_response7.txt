```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving averages of closing prices
    ma_10 = df['close'].rolling(window=10).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the relative strength as the ratio of the 10-day moving average to the 50-day moving average
    relative_strength = ma_10 / ma_50
    
    # Calculate the 10-day percentage change in closing price (momentum)
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the 10-day rolling standard deviation of daily returns (volatility)
    volatility = daily_return.rolling(window=10).std()
    
    # Calculate the 10-day rolling mean of volume (liquidity)
    liquidity = df['volume'].rolling(window=10).mean()
    
    # Normalize the momentum by volatility
    normalized_momentum = momentum / (volatility + 1e-7)
    
    # Normalize the volume by the 10-day rolling mean of volume
    normalized_volume = df['volume'] / (liquidity + 1e-7)
    
    # Calculate the 10-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Combine the normalized momentum, normalized volume, relative strength, and high-low range into an alpha factor
    alpha_factor = (normalized_momentum * normalized_volume * relative_strength) / (hl_range_mean + 1e-7)
    
    # Smooth the alpha factor using a 5-day rolling mean
    smoothed_alpha_factor = alpha_factor.rolling(window=5).mean()
    
    return smoothed_alpha_factor
```
