def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the RSI with an adaptive window
    def rsi(data, window):
        delta = data.diff(1)
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window).mean()
        roll_down = down.abs().rolling(window).mean()
        RS = roll_up / roll_down
        return 100.0 - (100.0 / (1.0 + RS))
    
    rsi_window = df['close'].rolling(window=14).std().apply(lambda x: max(int(x), 10))
    rsi_factor = rsi(df['close'], rsi_window)
    
    # Calculate the MACD with an adaptive window
    def macd(data, fast, slow, signal):
        exp1 = data.ewm(span=fast, adjust=False).mean()
        exp2 = data.ewm(span=slow, adjust=False).mean()
        macd_line = exp1 - exp2
        signal_line = macd_line.ewm(span=signal, adjust=False).mean()
        return macd_line - signal_line
    
    macd_fast = df['close'].rolling(window=12).std().apply(lambda x: max(int(x), 12))
    macd_slow = df['close'].rolling(window=26).std().apply(lambda x: max(int(x), 26))
    macd_signal = df['close'].rolling(window=9).std().apply(lambda x: max(int(x), 9))
    macd_factor = macd(df['close'], macd_fast, macd_slow, macd_signal)
    
    # Calculate the Bollinger Bands with an adaptive window
    bb_window = df['close'].rolling(window=20).std().apply(lambda x: max(int(x), 20))
    rolling_mean = df['close'].rolling(window=bb_window).mean()
    rolling_std = df['close'].rolling(window=bb_window).std()
    upper_band = rolling_mean + 2 * rolling_std
    lower_band = rolling_mean - 2 * rolling_std
    bb_factor = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Calculate the volume-weighted momentum
    momentum = df['close'].pct_change(periods=10)
    volume_weighted_momentum = (momentum * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()
    
    # Combine the RSI, MACD, Bollinger Bands, and volume-weighted momentum into an alpha factor
    alpha_factor = (rsi_factor + macd_factor + bb_factor + volume_weighted_momentum) / 4.0
    
    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
