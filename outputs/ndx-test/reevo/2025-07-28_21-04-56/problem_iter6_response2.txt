```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day exponential moving averages of closing prices
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the difference between the current close and the 10-day EMA
    close_ema_10_diff = df['close'] - ema_10
    
    # Calculate the 10-day rolling standard deviation to represent volatility
    volatility_10d = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the 10-day momentum
    momentum_10d = df['close'].pct_change(periods=10)
    
    # Calculate the 50-day trend strength as the ratio of the 10-day EMA to the 50-day EMA
    trend_strength = ema_10 / ema_50
    
    # Calculate the liquidity as the ratio of volume to the 10-day rolling mean of volume
    liquidity = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate the volume-weighted momentum
    volume_weighted_momentum = (momentum_10d * df['volume']).rolling(window=10).mean() / liquidity
    
    # Combine the dynamic metrics into an alpha factor
    alpha_factor = (volume_weighted_momentum / (volatility_10d + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7))
    
    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
```
