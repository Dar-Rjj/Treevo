def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving averages of closing prices
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between the current close and the 10-day moving average
    df['close_ma_10_diff'] = df['close'] - df['ma_10']
    
    # Calculate the 10-day rolling standard deviation to represent volatility
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Normalize the difference by the 10-day price range plus a small constant for numerical stability
    df['normalized_close_ma_10_diff'] = df['close_ma_10_diff'] / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap'] /= df['volume'].rolling(window=10).mean()
    
    # Smooth the VWAP using exponential moving average (EMA)
    df['vwap_ema'] = df['vwap'].ewm(span=10, adjust=False).mean()
    
    # Calculate the momentum in the EMA of VWAP
    df['vwap_ema_momentum'] = df['vwap_ema'].pct_change(periods=1)
    
    # Calculate the 10-day mean reversion factor
    df['mean_reversion'] = (df['close'] - df['ma_10']) / df['ma_10']
    
    # Adaptive window for volume-weighted price momentum
    adaptive_window = (df['volume'].rolling(window=10).mean() / df['volume']).round().astype(int) + 5
    adaptive_window = adaptive_window.clip(lower=10, upper=50)
    df['volume_weighted_momentum'] = df['close'].pct_change(periods=adaptive_window)
    
    # Combine the alpha factors: normalized difference, volatility, VWAP EMA momentum, mean reversion, and volume-weighted momentum
    alpha_factor = (
        df['normalized_close_ma_10_diff'] +
        df['volatility_10d'] +
        df['vwap_ema_momentum'] +
        df['mean_reversion'] +
        df['volume_weighted_momentum']
    )
    
    return alpha_factor.dropna()
