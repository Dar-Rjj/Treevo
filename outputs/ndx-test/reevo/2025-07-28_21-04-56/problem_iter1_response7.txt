```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the closing price and the moving average of the last 10 days closing prices.
    df['ma_10'] = df['close'].rolling(window=10).mean()
    alpha_factor = (df['close'] - df['ma_10']) / (df['high'] - df['low'] + 1e-7)
    return alpha_factor
```
