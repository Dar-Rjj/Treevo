```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price, suggesting if the stock closed higher or lower than its opening
    close_open_diff = df['close'] - df['open']
    
    # Calculate the volatility for the day, which is defined as the range (high - low)
    daily_volatility = df['high'] - df['low']
    
    # Average true range over the last 5 days to capture recent trend stability
    tr = df[['high', 'low']].rolling(window=5).std().max(axis=1)
    
    # Heuristic factor: The ratio of daily movement from open to close, adjusted by the average true range, gives a measure
    # of how significant the price movement on a given day is in relation to the overall volatility.
    factor = close_open_diff / (tr + 1e-7)
    
    return factor
```
