```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate daily price change
    daily_price_change = df['close'] - df['open']
    
    # Calculate volume weighted average price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    vwap /= df['volume'].rolling(window=5).mean()
    
    # Smooth the VWAP using exponential moving average (EMA)
    vwap_ema = vwap.ewm(span=5, adjust=False).mean()
    
    # Calculate the momentum in the EMA of VWAP
    vwap_ema_momentum = vwap_ema.pct_change(periods=1)
    
    # Calculate the volatility as the standard deviation of daily returns over a 5-day window
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=5).std()
    
    # Calculate liquidity as the ratio of dollar volume to close price
    liquidity = (df['volume'] * df['close']).rolling(window=5).mean() / df['close']
    
    # Combine the alpha factors: daily price change, momentum in smoothed VWAP, volatility, and liquidity
    alpha_factor = daily_price_change + vwap_ema_momentum - volatility + liquidity
    
    return alpha_factor
```
