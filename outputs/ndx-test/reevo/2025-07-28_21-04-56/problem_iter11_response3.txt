def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 20-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs_10 = gain / loss
    rsi_10 = 100 - (100 / (1 + rs_10))
    
    gain_20 = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss_20 = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs_20 = gain_20 / loss_20
    rsi_20 = 100 - (100 / (1 + rs_20))

    # Calculate the MACD with a short window of 12, a long window of 26, and a signal window of 9
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line

    # Calculate Bollinger Bands with a 20-day window and 2 standard deviations
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    bollinger_percentage = (df['close'] - lower_band) / (upper_band - lower_band)

    # Calculate the momentum as the 10-day percentage change in closing price
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volatility using Parkinson's Volatility
    parkinson_volatility = ((df['high'] / df['low']).apply(np.log).rolling(window=10).std()) * (np.sqrt(4.67))
    
    # Calculate the liquidity as the average volume over a 10-day window
    liquidity = df['volume'].rolling(window=10).mean()
    
    # Calculate the 50-day and 200-day moving averages of closing prices
    ma_50 = df['close'].ewm(span=50, adjust=False).mean()
    ma_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day exponential moving averages
    trend_strength = ma_50 - ma_200
    
    # Calculate the 10-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate the volume-weighted momentum
    volume_weighted_momentum = (momentum * df['volume']).rolling(window=10).mean() / liquidity
    
    # Combine the volume-weighted momentum, Parkinson's volatility, liquidity, trend strength, and high-low range into an alpha factor
    alpha_factor = (volume_weighted_momentum / (parkinson_volatility + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7))
    
    # Incorporate RSI, MACD, and Bollinger Bands into the alpha factor
    alpha_factor = alpha_factor * (rsi_10 - rsi_20) * macd_histogram * bollinger_percentage
    
    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
