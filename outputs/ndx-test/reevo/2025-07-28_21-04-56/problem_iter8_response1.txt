```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the 5-day percentage change in closing price
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volatility as the standard deviation of the daily returns over a 5-day window
    volatility = daily_return.rolling(window=5).std()
    
    # Calculate the liquidity as the average volume over a 5-day window
    liquidity = df['volume'].rolling(window=5).mean()
    
    # Calculate the 20-day and 50-day moving averages of closing prices
    ma_20 = df['close'].rolling(window=20).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the trend strength as the difference between the 20-day and 50-day moving averages
    trend_strength = ma_20 - ma_50
    
    # Calculate the 5-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the 5-day rolling mean of the open-close range
    oc_range_mean = (df['close'] - df['open']).rolling(window=5).mean()
    
    # Combine the momentum, volatility, liquidity, trend strength, high-low range, and open-close range into an alpha factor
    alpha_factor = (momentum / (volatility + 1e-7)) * (df['volume'] / (liquidity + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7)) * (oc_range_mean + 1e-7)
    
    # Smooth the alpha factor using a 3-day exponential moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
```
