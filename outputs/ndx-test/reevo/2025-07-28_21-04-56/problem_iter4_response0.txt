```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day moving average of the close price to capture a longer-term trend
    df['close_50d_ma'] = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between today's close and the 50-day moving average
    df['close_trend_diff'] = df['close'] - df['close_50d_ma']
    
    # Calculate the 10-day exponential moving average of the volume to capture recent liquidity
    df['volume_10d_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day average true range to capture longer-term volatility
    tr = (df['high'] - df['low']).rolling(window=30).mean()
    df['atr_30d'] = tr
    
    # Generate a factor that combines the trend strength, liquidity, and volatility
    trend_strength = df['close_trend_diff'] / (df['atr_30d'] + 1e-7)
    liquidity = 1 / (df['volume_10d_ema'] + 1e-7)
    volatility_adjusted_trend = trend_strength * liquidity
    
    return volatility_adjusted_trend
```
