def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and opening price
    daily_price_change = df['close'] - df['open']
    # Calculate the volume weighted price, to capture the price level influenced by trading volume
    volume_weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    # Calculate a smoothed version of volume weighted price using exponential moving average
    vwp_ema = volume_weighted_price.ewm(span=5, adjust=False).mean()
    # Calculate the momentum in the change of smoothed volume weighted price
    vwp_ema_momentum = vwp_ema.pct_change(periods=1)
    
    # The alpha factor is designed as a combination of daily price change and its sensitivity (momentum) to the change in smoothed volume weighted price
    # Idea: Stocks with positive price changes and growing momentum in their EMA of volume-weighted prices could potentially see further upward movement
    alpha_factor = daily_price_change + vwp_ema_momentum
    
    return alpha_factor
