```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving averages of closing prices
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between the current close and the 10-day moving average
    df['close_ma_10_diff'] = df['close'] - df['ma_10']
    
    # Calculate the 10-day rolling standard deviation to represent volatility
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Normalize the difference by the 10-day price range plus a small constant for numerical stability
    df['normalized_close_ma_10_diff'] = df['close_ma_10_diff'] / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the 20-day and 100-day moving averages of volume
    df['vol_ma_20'] = df['volume'].rolling(window=20).mean()
    df['vol_ma_100'] = df['volume'].rolling(window=100).mean()
    
    # Calculate the liquidity factor as the ratio of the 20-day volume moving average to the 100-day volume moving average
    df['liquidity_factor'] = df['vol_ma_20'] / df['vol_ma_100']
    
    # Incorporate a trend strength measure, such as the ratio of the 10-day moving average to the 50-day moving average
    df['trend_strength'] = df['ma_10'] / df['ma_50']
    
    # Combine the normalized difference with the volatility, liquidity, and trend strength
    alpha_factor = df['normalized_close_ma_10_diff'] * df['volatility_10d'] * df['liquidity_factor'] * df['trend_strength']
    
    return alpha_factor.dropna()
```
