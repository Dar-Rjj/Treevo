```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    df['close_diff'] = df['close'].diff()
    
    # Calculate the 5-day moving average of the volume
    df['volume_5d_ma'] = df['volume'].rolling(window=5).mean()
    
    # Generate a factor that multiplies the price difference by the inverse of the 5-day average volume
    factor = df['close_diff'] / (df['volume_5d_ma'] + 1e-7)
    
    return factor
```
