```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day moving averages of closing prices
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the momentum as the 5-day percentage change in closing price
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volatility as the standard deviation of the daily returns over a 5-day window
    volatility = daily_return.rolling(window=5).std()
    
    # Calculate the liquidity as the average volume over a 5-day window
    liquidity = df['volume'].rolling(window=5).mean()
    
    # Calculate the trend strength as the difference between the 5-day and 10-day moving averages
    trend_strength = ma_5 - ma_10
    
    # Calculate the 5-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine the momentum, volatility, liquidity, trend strength, and high-low range into an alpha factor
    alpha_factor = (momentum / (volatility + 1e-7)) * (df['volume'] / (liquidity + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7))
    
    # Smooth the alpha factor using a 3-day rolling mean
    smoothed_alpha_factor = alpha_factor.rolling(window=3).mean()
    
    return smoothed_alpha_factor
```
