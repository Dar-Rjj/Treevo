```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the 10-day percentage change in closing price
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volatility as the standard deviation of the daily returns over a 10-day window
    volatility = daily_return.rolling(window=10).std()
    
    # Calculate the liquidity as the average volume over a 10-day window
    liquidity = df['volume'].rolling(window=10).mean()
    
    # Combine the momentum, volatility, and liquidity into an alpha factor
    alpha_factor = momentum / (volatility + 1e-7) * (df['volume'] / (liquidity + 1e-7))
    
    # Smooth the alpha factor using a 3-day rolling mean
    smoothed_alpha_factor = alpha_factor.rolling(window=3).mean()
    
    return smoothed_alpha_factor
```
