```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate 1-day return
    close_change = df['close'].diff(1)
    
    # Calculate average volume over the last 5 days
    volume_avg_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate price volatility over the last 5 days
    price_volatility_5d = (df['high'] - df['low']).rolling(window=5).std()
    
    # Calculate the 10-day moving average of the closing price
    ma_close_10d = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    ma_close_30d = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the 10-day and 30-day moving averages
    trend_signal = ma_close_10d - ma_close_30d
    
    # Combine all factors: 1-day return, average volume over 5 days, daily volatility, and trend signal
    alpha_factor = close_change * volume_avg_5d / price_volatility_5d + trend_signal
    
    return alpha_factor
```
