```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate 10-day moving average of the closing price for mean reversion signal
    df['ma_10'] = df['close'].rolling(window=10).mean()
    mean_reversion = (df['close'] - df['ma_10'])

    # Momentum is captured by the difference in today's close and the close 10 days ago
    df['close_lag_10'] = df['close'].shift(10)
    momentum = df['close'] - df['close_lag_10']

    # Normalizing by the daily volatility, which is measured as the standard deviation of the daily range over the last 5 days
    daily_volatility = (df['high'] - df['low']).rolling(window=5).std() + 1e-7  # Adding small constant to avoid division by zero

    # Combining mean reversion, momentum, and normalized by daily volatility
    alpha_factor = (mean_reversion + momentum) / daily_volatility

    return alpha_factor
```
