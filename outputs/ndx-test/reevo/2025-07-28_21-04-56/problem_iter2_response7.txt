```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 30-day moving average of the closing price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day standard deviation of the closing price for volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Calculate the relative strength (RS) as the ratio of the current close to the 30-day moving average
    df['relative_strength'] = df['close'] / df['ma_30']
    
    # Calculate the difference between the current close and the 30-day moving average, normalized by the 10-day volatility
    alpha_factor = (df['close'] - df['ma_30']) / (df['volatility'] + 1e-7)
    
    # Multiply the alpha factor by the relative strength to incorporate momentum
    alpha_factor = alpha_factor * df['relative_strength']
    
    return alpha_factor.dropna()
```
