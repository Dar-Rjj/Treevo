```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 7-day moving averages of closing prices
    ma_3 = df['close'].rolling(window=3).mean()
    ma_7 = df['close'].rolling(window=7).mean()
    
    # Calculate the momentum as the 3-day percentage change in closing price
    momentum = df['close'].pct_change(periods=3)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volatility as the standard deviation of the daily returns over a 3-day window
    volatility = daily_return.rolling(window=3).std()
    
    # Calculate the liquidity as the average volume over a 3-day window
    liquidity = df['volume'].rolling(window=3).mean()
    
    # Calculate the trend strength as the difference between the 3-day and 7-day moving averages
    trend_strength = ma_3 - ma_7
    
    # Calculate the 3-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=3).mean()
    
    # Additional indicator: Money Flow Index (MFI) to incorporate market sentiment
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - 100 / (1 + money_ratio)
    
    # Combine the momentum, volatility, liquidity, trend strength, high-low range, and MFI into an alpha factor
    alpha_factor = (momentum / (volatility + 1e-7)) * (df['volume'] / (liquidity + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7)) * (mfi / 50)
    
    # Smooth the alpha factor using a 2-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=2, adjust=False).mean()
    
    return smoothed_alpha_factor
```
