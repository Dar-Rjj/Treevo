```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate adaptive moving averages for trend analysis
    ma_10 = df['close'].rolling(window=10).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    adaptive_window = (ma_10 - ma_20).apply(lambda x: 10 if x > 0 else 20)
    ma_adaptive = df['close'].rolling(window=adaptive_window, min_periods=10).mean()

    # Trend strength as the difference between the adaptive and 20-day moving averages
    trend_strength = ma_adaptive - ma_20

    # Calculate the 10-day and 20-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi_10 = 100 - (100 / (1 + rs))

    avg_gain_20 = gain.rolling(window=20).mean()
    avg_loss_20 = loss.rolling(window=20).mean()
    rs_20 = avg_gain_20 / avg_loss_20
    rsi_20 = 100 - (100 / (1 + rs_20))

    # Calculate the 10-day and 20-day MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_10 = ema_12 - ema_26
    signal_line_10 = macd_10.ewm(span=9, adjust=False).mean()

    ema_24 = df['close'].ewm(span=24, adjust=False).mean()
    ema_52 = df['close'].ewm(span=52, adjust=False).mean()
    macd_20 = ema_24 - ema_52
    signal_line_20 = macd_20.ewm(span=18, adjust=False).mean()

    # Momentum as the 10-day percentage change in closing price
    momentum = df['close'].pct_change(periods=10)

    # Daily return
    daily_return = df['close'].pct_change()

    # Parkinson's Volatility
    parkinson_volatility = ((df['high'] / df['low']).apply(np.log).rolling(window=10).std()) * (np.sqrt(4.67))

    # Liquidity as the average volume over a 10-day window
    liquidity = df['volume'].rolling(window=10).mean()

    # Volume-weighted momentum
    volume_weighted_momentum = (momentum * df['volume']).rolling(window=10).mean() / liquidity

    # Market regime detection
    regime = (daily_return > 0).rolling(window=30).mean()
    regime_adjustment = regime.apply(lambda x: 1.5 if x > 0.5 else 0.5)

    # Combine the volume-weighted momentum, RSI, MACD, Parkinson's volatility, liquidity, and trend strength into an alpha factor
    alpha_factor = (volume_weighted_momentum / (parkinson_volatility + 1e-7)) * (trend_strength / (rsi_10 - rsi_20 + 1e-7)) * (macd_10 - signal_line_10) * (macd_20 - signal_line_20) * regime_adjustment

    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()

    return smoothed_alpha_factor
```
