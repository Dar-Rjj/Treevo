```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the 5-day average volume
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Calculate the price change momentum
    df['price_momentum'] = df['close'].pct_change(periods=5)
    
    # Combine the factors into a single heuristic
    factor = df['close_diff'] * df['volume_ratio'] * df['price_momentum']
    
    return factor.dropna()
```
