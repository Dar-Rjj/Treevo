```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving averages of closing prices
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the 10-day and 50-day momentum
    df['momentum_10d'] = df['close'] - df['close'].shift(10)
    df['momentum_50d'] = df['close'] - df['close'].shift(50)
    
    # Calculate the 10-day rolling standard deviation to represent volatility
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Normalize the 10-day momentum by the 10-day price range plus a small constant for numerical stability
    df['normalized_momentum_10d'] = df['momentum_10d'] / (df['high'] - df['low'] + 1e-7)
    
    # Combine the normalized momentum with the volatility
    alpha_factor = df['normalized_momentum_10d'] * df['volatility_10d']
    
    # Incorporate the difference between the 10-day and 50-day moving averages
    df['trend_diff'] = df['ma_10'] - df['ma_50']
    
    # Refine the factor by combining it with the trend difference
    alpha_factor = alpha_factor + df['trend_diff']
    
    return alpha_factor.dropna()
```
