```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day moving average of the close price to capture the trend
    df['close_20d_ma'] = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between today's close and the 20-day moving average
    df['close_trend_diff'] = df['close'] - df['close_20d_ma']
    
    # Calculate the 5-day exponential moving average of the volume to capture recent liquidity
    df['volume_5d_ema'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 20-day average true range to capture longer-term volatility
    tr = (df['high'] - df['low']).rolling(window=20).mean()
    df['atr_20d'] = tr
    
    # Generate a factor that combines the trend strength, liquidity, and volatility
    factor = (df['close_trend_diff'] / (df['atr_20d'] + 1e-7)) * (1 / (df['volume_5d_ema'] + 1e-7))
    
    return factor
```
