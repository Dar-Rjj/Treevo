```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 20-day exponential moving averages of closing prices
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 50-day and 200-day exponential moving averages of closing prices for trend strength
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the momentum as the 10-day percentage change in closing price
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the volatility as the standard deviation of the daily returns over a 10-day window
    volatility = daily_return.rolling(window=10).std()
    
    # Calculate the liquidity as the exponential moving average of volume over a 10-day window
    liquidity = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day EMA
    trend_strength = ema_50 - ema_200
    
    # Calculate the 10-day rolling mean of the high-low range
    hl_range_mean = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Calculate the volume-weighted momentum
    volume_weighted_momentum = (momentum * df['volume']).ewm(span=10, adjust=False).mean() / liquidity
    
    # Calculate the price/volume ratio
    price_volume_ratio = df['close'] / df['volume']
    
    # Combine the volume-weighted momentum, volatility, liquidity, trend strength, high-low range, and price/volume ratio into an alpha factor
    alpha_factor = (volume_weighted_momentum / (volatility + 1e-7)) * (trend_strength / (hl_range_mean + 1e-7)) * price_volume_ratio
    
    # Smooth the alpha factor using a 3-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_alpha_factor
```
