```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day moving average of the closing prices to capture longer-term trends
    df['ma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 5-day exponential moving average of the daily returns to capture short-term volatility
    df['daily_returns'] = df['close'].pct_change()
    df['ema_5'] = df['daily_returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate the price range (high - low) and add a small constant to avoid division by zero
    df['price_range'] = df['high'] - df['low'] + 1e-7
    
    # Combine the factors into a single heuristic
    alpha_factor = (df['close'] - df['ma_20']) / df['price_range'] * df['ema_5']
    
    return alpha_factor.dropna()
```
