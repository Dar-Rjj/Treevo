def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 12-day and 26-day EMA for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_hist = macd_line - signal_line

    # Calculate the 10-day volume-weighted momentum
    vw_momentum = (df['close'].pct_change(periods=10) * df['volume']).rolling(window=10).mean() / df['volume'].rolling(window=10).mean()

    # Calculate the 10-day exponentially smoothed high-low range
    hl_range = (df['high'] - df['low']).ewm(span=10, adjust=False).mean()

    # Calculate the 50-day and 200-day moving averages of closing prices
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200

    # Calculate the 10-day Parkinson's Volatility
    parkinson_volatility = ((df['high'] / df['low']).apply(np.log).rolling(window=10).std()) * (np.sqrt(4.67))

    # Calculate the 30-day mean absolute deviation of daily returns
    mad = df['close'].pct_change().rolling(window=30).apply(lambda x: np.mean(np.abs(x - x.mean())), raw=True)

    # Calculate the 20-day cumulative sum of the daily returns
    cum_return = df['close'].pct_change().rolling(window=20).sum()

    # Calculate the 10-day realized volatility
    realized_volatility = df['close'].pct_change().rolling(window=10).std()

    # Combine the RSI, MACD histogram, volume-weighted momentum, smoothed high-low range, trend strength, Parkinson's volatility, mean absolute deviation, cumulative return, and realized volatility into an alpha factor
    alpha_factor = (rsi / 100 + 
                    macd_hist / (parkinson_volatility + 1e-7) + 
                    vw_momentum / (hl_range + 1e-7) + 
                    trend_strength / (df['close'] + 1e-7) + 
                    mad / (realized_volatility + 1e-7) + 
                    cum_return / (realized_volatility + 1e-7)) / 6

    # Smooth the alpha factor using a 2-day exponentially weighted moving average
    smoothed_alpha_factor = alpha_factor.ewm(span=2, adjust=False).mean()

    return smoothed_alpha_factor
