```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of closing prices
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the current close and the 5-day moving average
    df['close_ma_5_diff'] = df['close'] - df['ma_5']
    
    # Calculate the 5-day rolling standard deviation to represent volatility
    df['volatility_5d'] = df['close'].rolling(window=5).std()
    
    # Normalize the difference by the 5-day price range plus a small constant for numerical stability
    df['normalized_close_ma_5_diff'] = df['close_ma_5_diff'] / (df['high'] - df['low'] + 1e-7)
    
    # Calculate the 10-day and 50-day moving averages of volume
    df['vol_ma_10'] = df['volume'].rolling(window=10).mean()
    df['vol_ma_50'] = df['volume'].rolling(window=50).mean()
    
    # Calculate the liquidity factor as the ratio of the 10-day volume moving average to the 50-day volume moving average
    df['liquidity_factor'] = df['vol_ma_10'] / df['vol_ma_50']
    
    # Incorporate a trend strength measure, such as the ratio of the 5-day moving average to the 20-day moving average
    df['trend_strength'] = df['ma_5'] / df['ma_20']
    
    # Combine the normalized difference with the volatility, liquidity, and trend strength
    alpha_factor = df['normalized_close_ma_5_diff'] * df['volatility_5d'] * df['liquidity_factor'] * df['trend_strength']
    
    return alpha_factor.dropna()
```
