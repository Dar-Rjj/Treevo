def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    import numpy as np

    # Calculate the 14-day RSI using EWM
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 12-day and 26-day EMA for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_hist = macd_line - signal_line

    # Calculate the 10-day volume-weighted momentum using EWM
    vw_momentum = (df['close'].pct_change(periods=10) * df['volume']).ewm(span=10, adjust=False).mean() / df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate the 50-day and 200-day moving averages of closing prices using EWM
    ma_50 = df['close'].ewm(span=50, adjust=False).mean()
    ma_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the trend strength as the difference between the 50-day and 200-day moving averages
    trend_strength = ma_50 - ma_200

    # Dynamic Volatility: Calculate the 14-day standard deviation of the log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    dynamic_volatility = log_returns.rolling(window=14).std()

    # Adaptive Sentiment: Calculate the 14-day change in trading volume, adjusted by the 30-day average volume
    volume_change = (df['volume'].pct_change(periods=14) / df['volume'].rolling(window=30).mean()).ewm(span=14, adjust=False).mean()

    # Market Sentiment: Calculate the 5-day relative strength of the close price
    relative_strength = df['close'].pct_change(periods=5)

    # Adaptive Stochastic Oscillator: Calculate the 14-day Stochastic %K and %D with a 3-day smoothing
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    stoch_k = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    stoch_d = stoch_k.rolling(window=3).mean()

    # Trend Reversal Signal: Calculate the 5-day and 10-day rate of change (ROC) for the close price
    roc_5 = df['close'].pct_change(periods=5)
    roc_10 = df['close'].pct_change(periods=10)
    trend_reversal = roc_5 - roc_10

    tr = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift()), abs(df['low'] - df['close'].shift())], axis=1).max(axis=1)
    atr = tr.ewm(span=14, adjust=False).mean()

    ma_14 = df['close'].ewm(span=14, adjust=False).mean()
