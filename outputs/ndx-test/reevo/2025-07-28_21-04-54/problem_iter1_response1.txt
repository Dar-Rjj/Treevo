```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing prices of two consecutive days
    df['close_diff'] = df['close'].diff()
    # Calculate the 5-day moving average of the close price
    df['ma_5'] = df['close'].rolling(window=5).mean()
    # Alpha factor: Ratio of the 5-day moving average change to the absolute value of close price change
    # This could be interpreted as how the recent moving average trend compares with the recent momentum
    alpha_factor = (df['ma_5'].diff() / df['close_diff'].abs()).fillna(0)
    return alpha_factor
```
