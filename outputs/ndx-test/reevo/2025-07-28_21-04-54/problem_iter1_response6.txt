```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close prices
    df['close_diff'] = df['close'].diff()
    # Calculate the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    # Alpha factor: Today's close change divided by the average volume over the past 5 days
    alpha_factor = df['close_diff'] / (df['avg_volume_5d'] + 1e-7)
    return alpha_factor
```
