```python
def heuristics_v2(df: pd.DataFrame, sector_sentiment: pd.Series, macroeconomic_indicators: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Adaptive smoothing for the EMA volatility
    adaptive_ema_volatility = ema_volatility.ewm(span=5, adjust=False).mean()
    
    # Adaptive smoothing for the EMA price range
    adaptive_ema_price_range = ema_price_range.ewm(span=5, adjust=False).mean()
    
    # Integrate sector-specific sentiment
    sector_sentiment_aligned = sector_sentiment.reindex(df.index, method='ffill')
    
    # Integrate macroeconomic indicators
    macroeconomic_factor = macroeconomic_indicators.mean(axis=1)
    
    # Combine the smoothed momentum, adaptive EMA volatility, market sentiment, adaptive EMA price range, sector sentiment, and macroeconomic factor
    # Higher volatility reduces the score, indicating higher risk
    # Market sentiment adjusts the factor, higher sentiment increases the score
    # Adaptive EMA price range adds a measure of intraday volatility
    # Sector sentiment and macroeconomic factors provide additional context
    factor = (smoothed_momentum 
              - (adaptive_ema_volatility / adaptive_ema_volatility.mean()) 
              + (market_sentiment - 1) 
              - (adaptive_ema_price_range / adaptive_ema_price_range.mean())
              + (sector_sentiment_aligned - 1)
              + (macroeconomic_factor - macroeconomic_factor.mean()))
    
    return factor
```
