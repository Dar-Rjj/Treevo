def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Volume-weighted price: calculate the 10-day EMA of the volume-weighted price
    vwp = (df['volume'] * df['close']).ewm(span=10, adjust=False).mean() / ema_volume
    
    # Relative strength: compare the current close to the 10-day EMA of the volume-weighted price
    relative_strength = df['close'] / vwp
    
    # Dynamic weighting: adjust weights based on the current market conditions
    weight_momentum = 1.5 + (ema_volatility / ema_volatility.mean()) * 0.5
    weight_volatility = -1.0
    weight_market_sentiment = 1.0 + (market_sentiment - 1) * 0.5
    weight_price_range = -0.5 - (ema_price_range / ema_price_range.mean()) * 0.1
    weight_relative_strength = 1.0 + (relative_strength - 1) * 0.5
    
    # Adaptive weighting: further adjust weights based on the 10-day EMA of the absolute daily return
    adaptive_weight = (daily_return.abs().ewm(span=10, adjust=False).mean() / daily_return.abs().ewm(span=10, adjust=False).mean().mean())
    weight_momentum *= (1 + adaptive_weight)
    weight_market_sentiment *= (1 + adaptive_weight)
    weight_price_range *= (1 - adaptive_weight)
    weight_relative_strength *= (1 + adaptive_weight)
    
    # Incorporate additional factors: Money flow index (MFI) for liquidity
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(daily_return > 0, 0)
    negative_flow = raw_money_flow.where(daily_return < 0, 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Weight for MFI
    weight_mfi = 1.0 + (mfi - 50) / 50
    
    # Combine the smoothed momentum, normalized EMA volatility, market sentiment, price range, relative strength, and MFI
    factor = (weight_momentum * smoothed_momentum 
              + weight_volatility * (ema_volatility / ema_volatility.mean()) 
              + weight_market_sentiment * (market_sentiment - 1) 
              + weight_price_range * (ema_price_range / ema_price_range.mean()) 
              + weight_relative_strength * (relative_strength - 1)
              + weight_mfi * (mfi - 50) / 50)
    
    # Incorporate recent trend direction
    recent_trend = np.sign(daily_return).rolling(window=5).mean()
    factor *= (1 + recent_trend)
    
    return factor
