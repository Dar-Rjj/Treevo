```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Calculate the average of the last 5 days' close prices
    df['avg_close_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the standard deviation of the last 10 days' close prices
    df['std_close_10'] = df['close'].rolling(window=10).std()
    
    # Alpha factor: Z-score of today's close relative to the last 5 days' average
    alpha_factor = (df['close'] - df['avg_close_5']) / df['std_close_10']
    
    return alpha_factor
```
