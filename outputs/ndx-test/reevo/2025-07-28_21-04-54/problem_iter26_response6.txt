```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Adaptive weight for market sentiment based on the 10-day EMA of the close price
    adaptive_weight = (df['close'].ewm(span=10, adjust=False).mean() / df['close'])
    
    # Relative Strength Index (RSI) calculation
    gain = daily_return.where(daily_return > 0, 0)
    loss = -daily_return.where(daily_return < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # MACD calculation
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line
    
    # Trend factor: 60-day trend
    trend_factor = df['close'].pct_change(60)
    
    # Liquidity factor: volume per price range
    liquidity_factor = df['volume'] / (df['high'] - df['low'])
    
    # On-balance volume (OBV)
    obv = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    obv_ema = obv.ewm(span=20, adjust=False).mean()
    
    # Chaikin Money Flow (CMF)
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    cmf = (mf_multiplier * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine the smoothed momentum, EMA volatility, market sentiment, and price range
    # Higher volatility reduces the score, indicating higher risk
    # Market sentiment adjusts the factor, higher sentiment increases the score
    # Price range adds a measure of intraday volatility
    # Adaptive weight modifies the market sentiment impact
    # RSI and MACD histogram are added to the factor
    # Incorporate trend and liquidity factors
    # Add OBV and CMF for additional insights
    factor = (smoothed_momentum 
              - (ema_volatility / ema_volatility.mean()) 
              + (market_sentiment - 1) * adaptive_weight 
              - (ema_price_range / ema_price_range.mean())
              + (rsi - 50) / 50
              + macd_histogram / macd_histogram.abs().mean()
              + trend_factor * 0.2
              + liquidity_factor * 0.1
              + (obv_ema / obv_ema.mean())
              + (cmf - 0.5) * 2)
    
    return factor
```
