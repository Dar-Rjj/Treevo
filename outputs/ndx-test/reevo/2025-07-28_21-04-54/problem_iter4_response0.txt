```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for stability
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Seasonality: calculate the 6-month (120-day) and 1-year (250-day) moving averages
    ma_120 = df['close'].rolling(window=120).mean()
    ma_250 = df['close'].rolling(window=250).mean()
    
    # Seasonal adjustment: ratio of the 6-month MA to the 1-year MA
    seasonal_adjustment = ma_120 / ma_250 - 1
    
    # Adaptive weights based on the volatility
    adaptive_weights = 1 - (ema_volatility / ema_volatility.max())
    
    # Combine the smoothed momentum, normalized EMA volatility, market sentiment, and seasonal adjustment
    # Higher volatility reduces the score, indicating higher risk
    # Market sentiment adjusts the factor, higher sentiment increases the score
    # Seasonal adjustment adds a long-term trend component
    factor = (smoothed_momentum * adaptive_weights - 
              (ema_volatility / ema_volatility.mean()) + 
              (market_sentiment - 1) + 
              seasonal_adjustment)
    
    # Apply final exponential smoothing to the combined alpha factor for robustness
    factor_smoothed = factor.ewm(span=3).mean()
    
    return factor_smoothed
```
