```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return based on close prices
    daily_return = df['close'].pct_change()
    
    # Calculate a weighted sum of price changes (OHLC) for the day with different weights
    weighted_sum_changes = (1.5*df['open'] + 3*df['high'] - 3*df['low'] + 1.5*df['close']) / 9
    
    # Adjust the weighted sum changes by volume to emphasize the impact of volume
    adjusted_weighted_sum_changes = weighted_sum_changes * df['volume']
    
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Normalize the price range by volume to create a volume-adjusted range
    vol_adjusted_range = price_range / df['volume']
    
    # Combine the daily return, adjusted weighted sum of price changes, and volume-adjusted range
    factor_values = daily_return + adjusted_weighted_sum_changes - vol_adjusted_range
    
    return factor_values
```
