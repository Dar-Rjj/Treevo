def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change from the previous day
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the weighted average of the high and low prices, using the volume as a weight
    df['weighted_avg_price'] = (df['high'] + df['low']) / 2 * df['volume']

    # Calculate a factor that represents the momentum adjusted by the volatility and volume
    df['volatility'] = df['high'] - df['low']
    df['momentum_adjusted'] = (df['price_change'] - df['price_change'].mean()) / (df['volatility'] + 1e-7) * df['volume_change']

    # Return the momentum-adjusted factor
    return df['momentum_adjusted']
