```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between current close and previous close, indicating the momentum
    momentum = df['close'].diff()
    
    # Calculate the percentage change in volume, indicating changes in liquidity or interest
    vol_change = df['volume'].pct_change().fillna(0)
    
    # Calculate a weighted difference of high and low prices, giving more importance to the high price,
    # this could indicate the strength of positive price movement
    weighted_price_diff = (1.5 * df['high'] - 0.5 * df['low']) / (df['close'] + 1e-7)
    
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 5-day moving average of the volume
    vol_ma_5 = df['volume'].rolling(window=5).mean()
    
    # Calculate the 14-day RSI (Relative Strength Index) to measure the speed and change of price movements
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 20-day Bollinger Bands to measure volatility
    ma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = ma_20 + 2 * std_20
    lower_band = ma_20 - 2 * std_20
    bollinger_width = (upper_band - lower_band) / ma_20
    
    # Combine the indicators into an alpha factor
    # Weighing components based on their predictive power
    alpha_factor = (momentum * 0.3 + 
                    2 * vol_change * weighted_price_diff * 0.2 + 
                    (ma_5 - df['close']) / (df['close'] + 1e-7) * 0.15 + 
                    (vol_ma_5 - df['volume']) / (df['volume'] + 1e-7) * 0.1 + 
                    rsi * 0.15 + 
                    bollinger_width * 0.1)
    
    # Apply exponential smoothing to the combined alpha factor
    alpha_factor_smoothed = alpha_factor.ewm(span=3).mean()
    
    return alpha_factor_smoothed
```
