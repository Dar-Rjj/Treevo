```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, which represents the daily trading range
    daily_range = df['high'] - df['low']
    
    # Calculate the change in close price from one day to the next
    price_change = df['close'].diff()
    
    # Determine if the current day's volume is higher than the previous day's volume
    volume_increase = df['volume'].diff() > 0
    
    # Create a factor that is the ratio of the price change to the daily range, adjusted for whether the volume increased
    factor = (price_change / daily_range).where(volume_increase, 0)
    
    return factor
```
