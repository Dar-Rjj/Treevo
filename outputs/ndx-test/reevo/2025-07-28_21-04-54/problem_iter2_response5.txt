```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    close_diff = df['close'].diff()
    
    # Calculate the weighted sum of price changes (OHLC) for the day
    weighted_sum_changes = (df['open'] + 2*df['high'] - 2*df['low'] + df['close']) / 6
    
    # Adjust the weighted sum changes by volume to emphasize the impact of volume
    adjusted_weighted_sum_changes = weighted_sum_changes * df['volume']
    
    # Calculate the standard deviation of the close price over the past 5 days
    close_std_5d = df['close'].rolling(window=5).std()
    
    # Normalize the close price change by its standard deviation
    normalized_close_diff = close_diff / (close_std_5d + 1e-7)
    
    # Calculate the average volume over the past 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Normalize the adjusted weighted sum changes by the average volume
    normalized_adjusted_weighted_sum_changes = adjusted_weighted_sum_changes / (avg_volume_5d + 1e-7)
    
    # Combine the normalized close price change with the normalized adjusted weighted sum changes to create a signal
    factor_values = normalized_close_diff + normalized_adjusted_weighted_sum_changes
    
    return factor_values
```
