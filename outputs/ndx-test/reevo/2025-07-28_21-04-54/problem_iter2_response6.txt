```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day and 30-day simple moving averages of the close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day rolling standard deviation (volatility)
    df['volatility_10'] = df['close'].rolling(window=10).std()
    
    # Calculate the 5-day average volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the change in volume
    df['volume_change'] = df['volume'].diff()
    
    # Alpha factor: (Daily return * (SMA_30 - SMA_10) / Volatility) / (Average Volume + Change in Volume)
    alpha_factor = (df['daily_return'] * (df['sma_30'] - df['sma_10']) / (df['volatility_10'] + 1e-7)) / (df['avg_volume_5d'] + df['volume_change'] + 1e-7)
    
    return alpha_factor
```
