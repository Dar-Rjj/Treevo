```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Trend strength: calculate the 50-day moving average and compare it to the current close
    trend_strength = (df['close'] - df['close'].rolling(window=50).mean()) / df['close'].rolling(window=50).std()
    
    # Adaptive weighting based on recent performance
    weight_momentum = 0.4
    weight_volatility = -0.2
    weight_market_sentiment = 0.2
    weight_price_range = -0.1
    weight_trend = 0.3
    
    # Normalize each component
    normalized_smoothed_momentum = (smoothed_momentum - smoothed_momentum.mean()) / smoothed_momentum.std()
    normalized_ema_volatility = (ema_volatility - ema_volatility.mean()) / ema_volatility.std()
    normalized_market_sentiment = (market_sentiment - market_sentiment.mean()) / market_sentiment.std()
    normalized_ema_price_range = (ema_price_range - ema_price_range.mean()) / ema_price_range.std()
    normalized_trend_strength = (trend_strength - trend_strength.mean()) / trend_strength.std()
    
    # Combine the normalized factors with adaptive weights
    factor = (weight_momentum * normalized_smoothed_momentum 
              + weight_volatility * normalized_ema_volatility 
              + weight_market_sentiment * normalized_market_sentiment 
              + weight_price_range * normalized_ema_price_range
              + weight_trend * normalized_trend_strength)
    
    return factor
```
