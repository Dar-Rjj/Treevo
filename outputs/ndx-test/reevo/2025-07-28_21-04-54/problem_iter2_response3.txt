```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 30-day and 90-day moving averages of the closing price to capture short and long-term trends
    ma_30 = df['close'].rolling(window=30).mean()
    ma_90 = df['close'].rolling(window=90).mean()
    
    # Calculate the difference between the 30-day and 90-day moving averages to identify trend strength
    trend_strength = ma_30 - ma_90
    
    # Calculate the 10-day standard deviation of daily returns to measure volatility
    daily_return = df['close'].diff()
    volatility = daily_return.rolling(window=10).std()
    
    # Calculate a momentum factor as the 60-day cumulative return
    momentum = (df['close'] / df['close'].shift(60)) - 1
    
    # Combine the momentum, trend strength, and normalized volatility
    # Higher volatility reduces the score, indicating higher risk
    # The trend strength is added to favor strong trends
    factor = momentum + trend_strength - (volatility / volatility.mean())
    
    return factor
```
