```python
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression

def heuristics_v2(df: pd.DataFrame, macro_data: pd.DataFrame, sector_trends: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Relative strength: ratio of close to open
    relative_strength = df['close'] / df['open']
    
    # Incorporate macroeconomic indicators
    macro_factors = macro_data.reindex(df.index, method='ffill')
    macro_factors = (macro_factors - macro_factors.mean()) / macro_factors.std()
    
    # Incorporate sector-specific trends
    sector_factors = sector_trends.reindex(df.index, method='ffill')
    sector_factors = (sector_factors - sector_factors.mean()) / sector_factors.std()
    
    # Combine all factors into a DataFrame
    all_factors = pd.DataFrame({
        'smoothed_momentum': smoothed_momentum,
        'ema_volatility': ema_volatility,
        'market_sentiment': market_sentiment,
        'ema_price_range': ema_price_range,
        'relative_strength': relative_strength
    })
    
    # Merge with macro and sector factors
    all_factors = all_factors.join(macro_factors).join(sector_factors)
    
    # Use machine learning to dynamically adjust weights
    model = LinearRegression()
    X = all_factors.shift(1).dropna()
    y = daily_return.loc[X.index]
    model.fit(X, y)
    dynamic_weights = pd.Series(model.coef_, index=X.columns)
    
    # Combine the factors with dynamic weights
    factor = (all_factors * dynamic_weights).sum(axis=1)
    
    # Incorporate market regime changes by adjusting the factor based on the sign of the daily return
    factor *= np.sign(daily_return)
    
    return factor
```
