```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Sector-specific trends: assume sector is provided in the DataFrame
    sector_trend = df.groupby('sector')['close'].transform(lambda x: x.ewm(span=30, adjust=False).mean().pct_change())
    
    # Macro-economic indicators: assume macroeconomic data is available
    macro_indicator = df['macro_indicator']  # Example: interest rates, GDP growth, etc.
    
    # Combine the key indicators: smoothed momentum, normalized EMA volatility, market sentiment, and price range
    # Incorporate sector-specific trends and macroeconomic indicators
    factor = (smoothed_momentum 
              - (ema_volatility / ema_volatility.mean()) 
              + (market_sentiment - 1) 
              - (ema_price_range / ema_price_range.mean())
              + sector_trend
              + macro_indicator)
    
    # Use machine learning for adaptive weights
    from sklearn.linear_model import LinearRegression
    X = pd.concat([smoothed_momentum, ema_volatility, market_sentiment, ema_price_range, sector_trend, macro_indicator], axis=1)
    y = df['close'].pct_change().shift(-1).dropna()  # Future 1-day return
    X = X.dropna().iloc[:-1]  # Align with future return
    model = LinearRegression()
    model.fit(X, y)
    adaptive_weights = model.coef_
    
    # Apply adaptive weights
    factor = (adaptive_weights[0] * smoothed_momentum 
              - adaptive_weights[1] * (ema_volatility / ema_volatility.mean()) 
              + adaptive_weights[2] * (market_sentiment - 1) 
              - adaptive_weights[3] * (ema_price_range / ema_price_range.mean())
              + adaptive_weights[4] * sector_trend
              + adaptive_weights[5] * macro_indicator)
    
    return factor
```
