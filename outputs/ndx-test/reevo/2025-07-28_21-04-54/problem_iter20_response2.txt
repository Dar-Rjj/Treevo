import pandas as pd
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import TimeSeriesSplit, GridSearchCV

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Adaptive weight for market sentiment based on the 10-day EMA of the close price
    adaptive_weight = (df['close'].ewm(span=10, adjust=False).mean() / df['close'])
    
    # Incorporate RSI (Relative Strength Index)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Incorporate MACD (Moving Average Convergence Divergence)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_hist = macd_line - signal_line
    
    # Seasonality: day of the week and month
    df['day_of_week'] = df.index.dayofweek
    df['month'] = df.index.month
    
    # Liquidity: volume to close ratio
    liquidity = df['volume'] / df['close']
    
    # Sector performance: assuming we have a sector column in the dataframe
    if 'sector' not in df.columns:
        df['sector'] = 'Technology'  # Example placeholder for sector data
    
    # Machine learning for dynamic factor weighting
    X = pd.concat([
        smoothed_momentum.rename('smoothed_momentum'),
        (ema_volatility / ema_volatility.mean()).rename('normalized_ema_volatility'),
        (market_sentiment - 1) * adaptive_weight.rename('adjusted_market_sentiment'),
        (ema_price_range / ema_price_range.mean()).rename('normalized_ema_price_range'),
        ((rsi - 50) / 50).rename('normalized_rsi'),
        (macd_hist / macd_hist.abs().mean()).rename('normalized_macd_hist'),
        df['day_of_week'].rename('day_of_week'),
        df['month'].rename('month'),
        liquidity.rename('liquidity'),
        pd.get_dummies(df['sector'], prefix='sector').astype(float)
    ], axis=1).dropna()
    
    y = df['close'].pct_change().shift(-1).loc[X.index]
    
    # Time series cross-validation
    tscv = TimeSeriesSplit(n_splits=3)
    
    # Random Forest Regressor
    rf = RandomForestRegressor(n_estimators=100, random_state=42)
    
    # Grid Search for hyperparameter tuning
    param_grid = {
        'n_estimators': [50, 100, 200],
        'max_depth': [None, 10, 20, 30],
        'min_samples_split': [2, 5, 10],
        'min_samples_leaf': [1, 2, 4]
    }
    
    grid_search = GridSearchCV(rf, param_grid, cv=tscv, n_jobs=-1, scoring='neg_mean_squared_error')
    grid_search.fit(X
