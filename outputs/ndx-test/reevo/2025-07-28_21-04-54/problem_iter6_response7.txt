def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for daily return
    daily_return = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of daily returns for volatility
    ema_volatility = daily_return.ewm(span=10, adjust=False).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for the momentum to add more stability
    smoothed_momentum = momentum.ewm(span=5, adjust=False).mean()
    
    # Volume dynamics: calculate the 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Market sentiment: ratio of volume to its 10-day EMA
    market_sentiment = df['volume'] / ema_volume
    
    # Price range: calculate the 10-day exponential moving average of the high-low range
    price_range = df['high'] - df['low']
    ema_price_range = price_range.ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponential moving average of the open-close range
    open_close_range = df['open'] - df['close']
    ema_open_close_range = open_close_range.ewm(span=10, adjust=False).mean()
    
    # Refined momentum: 60-day cumulative return
    long_term_momentum = (df['close'] / df['close'].shift(60)) - 1
    
    # Exponential smoothing for the long-term momentum to add more stability
    smoothed_long_term_momentum = long_term_momentum.ewm(span=5, adjust=False).mean()
    
    # Dynamic volatility: 20-day exponential moving average of daily returns
    dynamic_volatility = daily_return.ewm(span=20, adjust=False).std()
    
    # Sentiment adjustment: 20-day exponential moving average of market sentiment
    smoothed_market_sentiment = market_sentiment.ewm(span=20, adjust=False).mean()
    
    # Combine the smoothed momentum, normalized dynamic volatility, smoothed market sentiment, price range, and open-close range
    # Higher volatility reduces the score, indicating higher risk
    # Smoothed market sentiment adjusts the factor, higher sentiment increases the score
    # Price range adds a measure of intraday volatility
    # Open-close range adds a measure of day-to-day directional movement
    factor = (smoothed_momentum 
              + 0.5 * smoothed_long_term_momentum 
              - (dynamic_volatility / dynamic_volatility.mean()) 
              + (smoothed_market_sentiment - 1) 
              - (ema_price_range / ema_price_range.mean())
              - (ema_open_close_range / ema_open_close_range.mean()))
    
    # Apply final exponential smoothing to the combined alpha factor
    factor_smoothed = factor.ewm(span=3, adjust=False).mean()
    
    return factor_smoothed
