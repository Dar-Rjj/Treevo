```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 60-day moving averages of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Calculate the difference between the two moving averages to identify long-term trends
    trend_factor = ma_10 - ma_60
    
    # Calculate the 10-day standard deviation of daily returns for volatility
    daily_return = df['close'].diff()
    volatility = daily_return.rolling(window=10).std()
    
    # Calculate a momentum factor as the 30-day cumulative return
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Exponential smoothing for stability
    alpha = 0.2
    smoothed_momentum = momentum.ewm(alpha=alpha).mean()
    
    # Incorporate market sentiment using volume dynamics
    volume_change = df['volume'].pct_change().fillna(0)
    normalized_volume_change = volume_change / volume_change.abs().mean()
    
    # Combine the trend factor, smoothed momentum, normalized volatility, and volume change
    # Higher volatility reduces the score, indicating higher risk
    # Volume change adds a sentiment component
    factor = (trend_factor + smoothed_momentum - (volatility / volatility.mean()) + normalized_volume_change).ewm(span=5).mean()
    
    return factor
```
