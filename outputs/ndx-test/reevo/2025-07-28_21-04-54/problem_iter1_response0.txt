```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive difference between close and open prices,
    # if the current volume is greater than the 5-day moving average of volume.
    vma = df['volume'].rolling(window=5).mean()
    return (df['close'] - df['open']).where(df['volume'] > vma, 0)
```
