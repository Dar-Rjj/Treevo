def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the logarithmic returns to better capture volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the average true range (ATR) as a more sophisticated volatility measure
    df['tr1'] = abs(df['high'] - df['low'])
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    atr = df['true_range'].rolling(window=14).mean()
    
    # Calculate the volume weighted price, which accounts for the impact of large trades
    vwp = (df['open'] + 2*df['high'] + 2*df['low'] + df['close']) * df['volume'] / 6
    
    # Adjust the volume weighted price by the logarithmic returns and ATR
    adjusted_vwp = (vwp.diff() / atr) * log_returns
    
    # Combine the adjusted volume weighted price with the logarithmic returns to create a signal
    factor_values = adjusted_vwp + log_returns

    return factor_values
