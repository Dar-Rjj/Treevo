```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change over the last 5 days
    df['price_change_5d'] = df['close'].pct_change(periods=5)
    df['volume_change_5d'] = df['volume'].pct_change(periods=5)

    # Momentum factor, looking at the trend in price and volume
    df['momentum_factor'] = (df['price_change_5d'] * df['volume_change_5d']).rolling(window=10).mean()

    # Return the momentum factor
    return df['momentum_factor']
```
