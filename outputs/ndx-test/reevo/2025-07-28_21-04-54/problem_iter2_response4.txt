```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change from the previous day
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the weighted average of the high and low prices, using the volume as a weight
    df['weighted_avg_price'] = (df['high'] + df['low']) / 2 * df['volume']

    # Calculate volatility as the difference between high and low prices
    df['volatility'] = df['high'] - df['low']

    # Calculate a mean-reverting component based on the deviation from the moving average
    df['moving_avg'] = df['close'].rolling(window=10).mean()
    df['mean_reversion'] = (df['close'] - df['moving_avg']) / df['volatility']

    # Calculate a momentum-adjusted factor that incorporates mean reversion and volume change
    df['momentum_adjusted'] = (df['price_change'] - df['price_change'].mean()) / (df['volatility'] + 1e-7) * df['volume_change']
    df['factor'] = df['momentum_adjusted'] - df['mean_reversion']

    # Consider multi-period trends by incorporating the trend of the factor over the last 5 days
    df['trend'] = df['factor'].rolling(window=5).mean()
    df['final_factor'] = df['factor'] + df['trend']

    # Return the final factor
    return df['final_factor']
```
