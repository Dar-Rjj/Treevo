def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the 10-day volume-weighted average price (VWAP)
    vwap_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day, 30-day, and 60-day exponentially weighted moving averages of the close price
    ewma_10 = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30 = df['close'].ewm(span=30, adjust=False).mean()
    ewma_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the 10-day and 30-day price volatility using exponential smoothing
    volatility_10 = df['close'].pct_change().ewm(span=10, adjust=False).std()
    volatility_30 = df['close'].pct_change().ewm(span=30, adjust=False).std()
    
    # Calculate the trend strength as the relative position of the 10-day EWMA to the 30-day EWMA
    trend_strength = (ewma_10 - ewma_30) / df['close']
    
    # Calculate the long-term trend strength as the relative position of the 10-day EWMA to the 60-day EWMA
    long_trend_strength = (ewma_10 - ewma_60) / df['close']
    
    # Calculate the 20-day, 50-day, and 100-day high-low ranges to capture short-term, medium-term, and long-term price movements
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    range_50 = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    range_100 = df['high'].rolling(window=100).max() - df['low'].rolling(window=100).min()
    
    # Calculate the 20-day RSI (Relative Strength Index)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(window=20).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=20).mean()
    rs = gain / loss
    rsi_20 = 100 - (100 / (1 + rs))
    
    # Calculate the 20-day and 50-day moving averages of the volume
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_ma_50 = df['volume'].rolling(window=50).mean()
    
    # Calculate the 20-day and 50-day standard deviations of the volume
    volume_std_20 = df['volume'].rolling(window=20).std()
    volume_std_50 = df['volume'].rolling(window=50).std()
    
    # Calculate the volume stability as the inverse of the coefficient of variation of the volume
    volume_stability_20 = (volume_ma_20 / (volume_std_20 + 1e-7))
    volume_stability_50 = (volume_ma_50 / (volume_std_50 + 1e-7))

              * (volatility_30 + 1e-7)**(-1)

              * (long_trend_strength + 1e-7)
              * (range_20 + 1e-7)**(-1)
              * (range_50 + 1e-7)**(-1)
              * (range_100 + 1e-7)**(-1)

              * (volume_stability_20 + 1e-7)
              * (volume_stability_50 + 1e-7))
