```python
def heuristics_v2(df: pd.DataFrame, market_index: pd.Series) -> pd.Series:
    # Calculate the 5-day return as a dynamic momentum indicator
    momentum_5 = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the 20-day return as a longer-term momentum indicator
    momentum_20 = df['close'] / df['close'].shift(20) - 1
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day price volatility using the high and low prices for a more robust measure
    volatility = (df['high'] - df['low']).rolling(window=10).std()
    
    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength with the market index
    relative_strength = (df['close'] / market_index - 1)
    
    # Calculate the interaction between the momentum and the relative position of the 10-day MA to the 30-day MA
    trend_strength = (ma_10 - ma_30) / df['close']
    
    # Combine the momentum, volume, volatility, trend strength, and relative strength into a single factor
    factor = (momentum_5 + momentum_20) * (avg_volume + 1e-7)**(-1) * (volatility + 1e-7)**(-1) * trend_strength * relative_strength
    
    return factor
```
