```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum_5 = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the 20-day return to capture longer-term trend
    momentum_20 = df['close'] / df['close'].shift(20) - 1
    
    # Calculate the average volume over the last 10 days
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day price volatility
    volatility_10 = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the 30-day price volatility
    volatility_30 = df['close'].pct_change().rolling(window=30).std()
    
    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the interaction between the momentum and the relative position of the 10-day MA to the 30-day MA
    trend_strength = (ma_10 - ma_30) / df['close']
    
    # Calculate the 20-day high-low range to capture price movement
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the autocorrelation of daily returns to capture market microstructure
    autocorr_5 = daily_returns.rolling(window=5).apply(lambda x: x.autocorr(lag=1), raw=False)
    
    # Calculate the liquidity ratio as the ratio of the average volume to the close price
    liquidity_ratio = avg_volume_10 / df['close']
    
    # Combine the momentum, volume, volatility, trend strength, range, and liquidity into a single factor
    factor = (
        (momentum_5 * (avg_volume_10 + 1e-7)**(-1) * (volatility_10 + 1e-7)**(-1) * trend_strength * (range_20 + 1e-7)**(-1)) +
        (momentum_20 * (avg_volume_10 + 1e-7)**(-1) * (volatility_30 + 1e-7)**(-1) * trend_strength * (range_20 + 1e-7)**(-1)) +
        (autocorr_5 * (volatility_10 + 1e-7)**(-1) * (liquidity_ratio + 1e-7))
    ).rank(pct=True)
    
    return factor
```
