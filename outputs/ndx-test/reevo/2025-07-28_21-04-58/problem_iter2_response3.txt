```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate percentage changes in price and volume
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the weighted moving average of price and volume changes
    wma_price_change = (df['price_change'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    wma_volume_change = (df['volume_change'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Smooth the WMA with a simple moving average
    smoothed_wma_price_change = wma_price_change.rolling(window=5).mean()
    smoothed_wma_volume_change = wma_volume_change.rolling(window=5).mean()

    # Alpha factor: the product of the smoothed weighted moving averages of price and volume changes
    df['alpha_factor'] = smoothed_wma_price_change * smoothed_wma_volume_change

    return df['alpha_factor']
```
