```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day price volatility
    volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the interaction between the momentum and the relative position of the 10-day MA to the 30-day MA
    trend_strength = (ma_10 - ma_30) / df['close']
    
    # Calculate the 20-day high-low range to capture price movement
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Normalize each component before combining them into a single factor
    momentum_normalized = (momentum - momentum.mean()) / (momentum.std() + 1e-7)
    avg_volume_normalized = (avg_volume - avg_volume.mean()) / (avg_volume.std() + 1e-7)
    volatility_normalized = (volatility - volatility.mean()) / (volatility.std() + 1e-7)
    trend_strength_normalized = (trend_strength - trend_strength.mean()) / (trend_strength.std() + 1e-7)
    range_20_normalized = (range_20 - range_20.mean()) / (range_20.std() + 1e-7)
    
    # Combine the normalized momentum, volume, volatility, trend strength, and range into a single factor
    factor = (momentum_normalized * (avg_volume_normalized + 1e-7)**(-1) * (volatility_normalized + 1e-7)**(-1) * 
              trend_strength_normalized * (range_20_normalized + 1e-7)**(-1))
    
    # Rank the factor values across all stocks on each date
    factor_ranked = factor.groupby(level=0).rank(pct=True)
    
    return factor_ranked
```
