def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day weighted moving average of close prices, with more weight given to recent prices
    weights = np.arange(1, 11)
    wma_10 = df['close'].rolling(window=10).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True)
    
    # Calculate the difference between the current close price and the 10-day weighted moving average
    diff_wma_10 = df['close'] - wma_10
    
    # Calculate 30-day momentum as the rate of change in close price from 30 days ago to today
    momentum_30 = df['close'] / df['close'].shift(30) - 1
    
    # Incorporate volume into the factor, assuming higher trading volume indicates stronger trends
    # Normalize volume by its 30-day moving average to account for general volume levels
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    normalized_volume = df['volume'] / avg_volume_30
    
    # Combine the factors into a single alpha signal
    # The combination assumes that closer proximity to the 10-day WMA, stronger positive 30-day momentum,
    # and higher normalized volume suggest a higher likelihood of future positive returns
    alpha_factor = diff_wma_10 + 5 * momentum_30 + 2 * normalized_volume
    return alpha_factor
