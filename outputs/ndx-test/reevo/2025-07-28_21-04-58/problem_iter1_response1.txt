```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and the previous close, to capture the momentum
    df['close_diff'] = df['close'].diff()
    
    # Create a factor based on the 5-day moving average of the closing price, to identify the trend
    df['close_ma_5'] = df['close'].rolling(window=5).mean()
    
    # Momentum-adjusted close relative to its 5-day moving average
    alpha_factor = (df['close'] - df['close_ma_5']) / df['close_diff'].abs()
    
    return alpha_factor
```
