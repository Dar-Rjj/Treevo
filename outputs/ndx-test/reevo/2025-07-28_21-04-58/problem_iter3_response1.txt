```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the weighted moving average of price and volume changes
    df['wma_price_change'] = df['price_change'].ewm(span=10, adjust=False).mean()
    df['wma_volume_change'] = df['volume_change'].ewm(span=10, adjust=False).mean()

    # Calculate the momentum of the close price
    df['momentum'] = df['close'] - df['close'].shift(30)

    # Integrate sector performance (assuming a column 'sector' exists in the DataFrame)
    df['sector_momentum'] = df.groupby('sector')['momentum'].transform('mean')

    # Calculate volatility as the standard deviation of the last 10 days' returns
    df['volatility'] = df['price_change'].rolling(window=10).std()

    # Combine weighted moving averages, momentum, sector performance, and volatility
    df['alpha_factor'] = (
        df['wma_price_change'] * df['wma_volume_change'] * 
        df['momentum'] * df['sector_momentum'] / df['volatility']
    )

    return df['alpha_factor']
```
