```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving averages of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between the current close price and the 10-day moving average
    diff_ma_10 = df['close'] - ma_10
    
    # Calculate the 30-day momentum as the rate of change in the close price
    momentum_30 = df['close'] / df['close'].shift(30) - 1
    
    # Calculate the 20-day standard deviation of the close price to measure volatility
    vol_20 = df['close'].rolling(window=20).std()
    
    # Calculate the difference between the 10-day MA and the 50-day MA to capture trend strength
    diff_ma_10_50 = ma_10 - ma_50
    
    # Detect volume spikes by calculating the percentage change in volume from the previous day
    volume_spike = (df['volume'] / df['volume'].shift(1)) > 1.5
    
    # Calculate the 10-day average true range (ATR) for additional volatility measure
    df['tr'] = df[['high', 'low']].apply(lambda x: max(x['high'], x['close'].shift()) - min(x['low'], x['close'].shift()), axis=1)
    atr_10 = df['tr'].rolling(window=10).mean()
    
    # Calculate the 10-day and 50-day moving averages of the high and low prices for envelope bands
    ma_high_10 = df['high'].rolling(window=10).mean()
    ma_low_10 = df['low'].rolling(window=10).mean()
    ma_high_50 = df['high'].rolling(window=50).mean()
    ma_low_50 = df['low'].rolling(window=50).mean()
    
    # Calculate the distance between the current high/low and the 10-day/50-day moving averages of high/low
    diff_high_10 = df['high'] - ma_high_10
    diff_low_10 = df['low'] - ma_low_10
    diff_high_50 = df['high'] - ma_high_50
    diff_low_50 = df['low'] - ma_low_50
    
    # Combine all factors into a single alpha signal
    alpha_factor = (diff_ma_10 + 5 * momentum_30 + 0.5 * vol_20 + 2 * diff_ma_10_50 + 3 * volume_spike.astype(int)
                    + 0.5 * atr_10 + 0.2 * diff_high_10 - 0.2 * diff_low_10 + 0.1 * diff_high_50 - 0.1 * diff_low_50)
    
    return alpha_factor
```
