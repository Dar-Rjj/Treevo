def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Apply non-linear transformation to momentum
    nonlinear_momentum = momentum ** 2
    
    # Calculate the volume-weighted average price over the last 10 days
    vwap_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day price volatility using exponential smoothing
    volatility = df['close'].pct_change().ewm(span=10, adjust=False).std()
    
    # Apply non-linear transformation to volatility
    nonlinear_volatility = 1 / (volatility + 1e-7)
    
    # Calculate the 10-day and 60-day exponentially weighted moving averages of the close price
    ewma_10 = df['close'].ewm(span=10, adjust=False).mean()
    ewma_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the trend strength as the relative position of the 10-day EWMA to the 60-day EWMA
    trend_strength = (ewma_10 - ewma_60) / df['close']
    
    # Apply non-linear transformation to trend strength
    nonlinear_trend_strength = np.sqrt(np.abs(trend_strength))
    
    # Calculate the 20-day and 50-day high-low ranges to capture short-term and long-term price movements
    range_20 = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / df['close']
    range_50 = (df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()) / df['close']
    
    # Calculate the 20-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    rsi_20 = 100 - (100 / (1 + rs))
    
    # Calculate the volume ratio (current volume / 10-day average volume)
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the factors into a single alpha signal using additive combinations
    factor = (nonlinear_momentum + vwap_10 + nonlinear_volatility + 
              nonlinear_trend_strength + 1 / (range_20 + 1e-7) + 1 / (range_50 + 1e-7) + 
              (rsi_20 / 100) + (volume_ratio + 1e-7))
    
    # Rank the combined factor to get a normalized score
    factor_ranked = factor.rank(pct=True)
    
    return factor_ranked
