```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the 10-day volume-weighted average price (VWAP)
    vwap_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day exponentially weighted moving average of the close price
    ewma_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponentially weighted moving average of the close price
    ewma_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 60-day exponentially weighted moving average of the close price for longer-term trend
    ewma_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the trend strength as the relative position of the 10-day EWMA to the 30-day and 60-day EWMA
    trend_strength_short = (ewma_10 - ewma_30) / df['close']
    trend_strength_long = (ewma_10 - ewma_60) / df['close']
    trend_strength = trend_strength_short + trend_strength_long
    
    # Calculate the 20-day high-low range to capture short-term price movement
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 50-day high-low range to capture medium-term price movement
    range_50 = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    
    # Calculate the 100-day high-low range to capture long-term price movement
    range_100 = df['high'].rolling(window=100).max() - df['low'].rolling(window=100).min()
    
    # Calculate the 20-day RSI (Relative Strength Index)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(window=20).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=20).mean()
    rs = gain / loss
    rsi_20 = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day realized volatility using logarithmic returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=10).std() * np.sqrt(252)
    
    # Calculate the liquidity factor as the average daily dollar volume
    liquidity = (df['close'] * df['volume']).rolling(window=10).mean()
    
    # Calculate the 20-day and 50-day volume-to-volume ratio
    volume_ratio_20 = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_ratio_50 = df['volume'] / df['volume'].rolling(window=50).mean()
    
    # Combine the factors into a single alpha signal using a non-linear combination
    factor = (momentum 
              * (vwap_10 + 1e-7)**(-1)
              * (realized_volatility + 1e-7)**(-1)
              * (trend_strength + 1e-7)
              * (range_20 + 1e-7)**(-1)
              * (range_50 + 1e-7)**(-1)
              * (range_100 + 1e-7)**(-1)
              * (rsi_20 / 100)
              * (liquidity + 1e-7)**(-1)
              * (volume_ratio_20 + 1e-7)
              * (volume_ratio_50 + 1e-7))
    
    # Rank the factor values across all stocks on each date
    factor_ranked = factor.groupby(level=0).rank(pct=True)
    
    return factor_ranked
```
