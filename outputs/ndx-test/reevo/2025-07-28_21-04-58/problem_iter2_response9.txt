def heuristics_v2(df: pd.DataFrame, sector_performance: pd.Series, market_regime: pd.Series) -> pd.Series:
    # Calculate moving average of close prices over a 10-day window
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the difference between the current close price and the 10-day moving average
    diff_ma_10 = df['close'] - ma_10
    
    # Calculate momentum factor as the rate of change in close price from 30 days ago to today
    momentum_30 = df['close'] / df['close'].shift(30) - 1
    
    # Calculate volatility over the last 30 days
    volatility_30 = df['close'].pct_change().rolling(window=30).std()
    
    # Normalize sector performance to use as a weight
    sector_perf_weight = (sector_performance - sector_performance.mean()) / sector_performance.std()
    
    # Adjust factor weights based on market regime
    if market_regime.name == 'bull':
        factor_weights = {'diff_ma_10': 1, 'momentum_30': 5, 'volatility_30': -2}
    elif market_regime.name == 'bear':
        factor_weights = {'diff_ma_10': 0.5, 'momentum_30': 2, 'volatility_30': -1}
    else:
        factor_weights = {'diff_ma_10': 1, 'momentum_30': 3, 'volatility_30': -1.5}
    
    # Combine the factors into a single alpha signal
    alpha_factor = (factor_weights['diff_ma_10'] * diff_ma_10 
                    + factor_weights['momentum_30'] * momentum_30 
                    + factor_weights['volatility_30'] * volatility_30
                    + sector_perf_weight)
    return alpha_factor
