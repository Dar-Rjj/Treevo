```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and the previous close, to capture the momentum
    df['close_diff'] = df['close'].diff()
    
    # Create a factor based on the 5-day moving average of the closing price, to identify the trend
    df['close_ma_5'] = df['close'].rolling(window=5).mean()
    
    # Create a factor based on the 30-day moving average of the closing price, to identify the longer-term trend
    df['close_ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the volatility as the standard deviation of the past 30 day's price differences
    df['volatility'] = df['close'].rolling(window=30).std()
    
    # Momentum-adjusted close relative to its 5-day moving average, scaled by volatility
    alpha_factor = (df['close'] - df['close_ma_5']) / (df['close_diff'].abs() * df['volatility'])
    
    # Incorporate trading volume as a measure of market activity
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    alpha_factor = alpha_factor * (df['volume'] / df['volume_ma_10'])
    
    return alpha_factor
```
