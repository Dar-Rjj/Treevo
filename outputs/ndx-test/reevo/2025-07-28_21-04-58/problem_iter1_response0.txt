```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate momentum based on the price difference between current and 5 days ago
    momentum = df['close'] - df['close'].shift(5)
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a factor that considers high momentum stocks with relatively lower volume
    factor = (momentum / (avg_volume + 1e-7)).rank(pct=True)
    return factor
```
