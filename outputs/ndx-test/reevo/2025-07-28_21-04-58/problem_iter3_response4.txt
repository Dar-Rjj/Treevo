```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the weighted moving average of price and volume changes
    df['wma_price_change'] = df['price_change'].rolling(window=10).mean()
    df['wma_volume_change'] = df['volume_change'].rolling(window=10).mean()

    # Integrate sector performance (assuming 'sector' column exists)
    df['sector_performance'] = df.groupby('sector')['close'].transform(lambda x: x.pct_change().rolling(window=10).mean())

    # Adjust weights based on market regime (assuming 'market_regime' column exists)
    df['wma_price_change_adjusted'] = df.apply(lambda row: row['wma_price_change'] * 1.5 if row['market_regime'] == 'bull' else row['wma_price_change'] * 0.5, axis=1)
    
    # Alpha factor: the product of the adjusted weighted moving averages of price and volume changes, and sector performance
    df['alpha_factor'] = df['wma_price_change_adjusted'] * df['wma_volume_change'] * df['sector_performance']
    
    return df['alpha_factor']
```
