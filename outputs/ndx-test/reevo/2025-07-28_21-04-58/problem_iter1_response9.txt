```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low for two consecutive days
    df['high_low_diff_t1'] = df['high'].shift(1) - df['low'].shift(1)
    df['high_low_diff_t0'] = df['high'] - df['low']
    
    # Calculate the ratio of today's high-low difference to yesterday's high-low difference
    df['momentum_ratio'] = df['high_low_diff_t0'] / (df['high_low_diff_t1'] + 1e-7)
    
    # Calculate a trend component by taking the difference between closing prices
    df['close_diff'] = df['close'] - df['close'].shift(1)
    
    # Calculate the alpha factor as a combination of momentum and trend components
    df['alpha_factor'] = df['momentum_ratio'] * (df['close_diff'] / (df['close'].shift(1) + 1e-7))
    
    return df['alpha_factor']
```
