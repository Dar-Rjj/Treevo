```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the volume-weighted average price over the last 10 days
    vwap_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day price volatility
    volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the interaction between the momentum and the relative position of the 10-day MA to the 30-day MA
    trend_strength = (ma_10 - ma_30) / df['close']
    
    # Calculate the 20-day high-low range to capture price movement
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 50-day high-low range to capture longer-term price movement
    range_50 = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    
    # Calculate the 20-day volume-weighted average price
    vwap_20 = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate the 60-day high-low range to capture even longer-term price movement
    range_60 = df['high'].rolling(window=60).max() - df['low'].rolling(window=60).min()
    
    # Calculate the 20-day return as a medium-term momentum indicator
    medium_momentum = df['close'] / df['close'].shift(20) - 1
    
    # Combine the momentum, volume-weighted average prices, volatility, trend strength, and ranges into a single factor
    factor = (momentum * (vwap_10 + 1e-7)**(-1) * (vwap_20 + 1e-7)**(-1) * (volatility + 1e-7)**(-1) * 
              trend_strength * (range_20 + 1e-7)**(-1) * (range_50 + 1e-7)**(-1) * (range_60 + 1e-7)**(-1) * medium_momentum)
    
    # Rank the factor values across all stocks on each date
    factor_ranked = factor.groupby(level=0).rank(pct=True)
    
    return factor_ranked
```
