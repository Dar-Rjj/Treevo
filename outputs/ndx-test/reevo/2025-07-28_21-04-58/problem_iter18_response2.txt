def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the 10-day volume-weighted average price (VWAP)
    vwap_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day and 30-day exponentially weighted moving averages of the close price
    ewma_10 = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the trend strength as the relative position of the 10-day EWMA to the 30-day EWMA
    trend_strength = (ewma_10 - ewma_30) / df['close']
    
    # Calculate the 20-day high-low range to capture short-term price movement
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 20-day RSI (Relative Strength Index)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(window=20).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=20).mean()
    rs = gain / loss
    rsi_20 = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day realized volatility using logarithmic returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=10).std() * np.sqrt(252)
    
    # Calculate the liquidity factor as the average daily dollar volume
    liquidity = (df['close'] * df['volume']).rolling(window=10).mean()
    
    # Combine the factors into a single alpha signal using a non-linear combination
    factor = (momentum 
              * (vwap_10 + 1e-7)**(-1)
              * (trend_strength + 1e-7)
              * (rsi_20 / 100)
              * (realized_volatility + 1e-7)**(-1)
              * (liquidity + 1e-7)**(-1))
    
    # Rank the factor values across all stocks on each date
    factor_ranked = factor.groupby(level=0).rank(pct=True)
    
    return factor_ranked
