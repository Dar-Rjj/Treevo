```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 60-day return as a momentum indicator
    short_momentum = df['close'] / df['close'].shift(5) - 1
    long_momentum = df['close'] / df['close'].shift(60) - 1
    
    # Calculate the average volume over an adaptive window (last 10 to 20 days)
    avg_volume = df['volume'].rolling(window=10).mean() + df['volume'].rolling(window=20).mean()
    
    # Calculate the 10-day and 30-day price volatility
    short_volatility = df['close'].pct_change().rolling(window=10).std()
    long_volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Calculate the 10-day, 30-day, and 60-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    ma_30 = df['close'].rolling(window=30).mean()
    ma_60 = df['close'].rolling(window=60).mean()
    
    # Calculate the interaction between the short and long momentum, and the relative position of the 10-day MA to the 30-day and 60-day MA
    trend_strength_short = (ma_10 - ma_30) / df['close']
    trend_strength_long = (ma_10 - ma_60) / df['close']
    
    # Calculate the 20-day high-low range to capture price movement
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Combine the momentum, volume, volatility, trend strength, and range into a single factor
    factor = (
        (short_momentum + long_momentum) * 
        (avg_volume + 1e-7)**(-1) * 
        (short_volatility + long_volatility + 1e-7)**(-1) * 
        (trend_strength_short + trend_strength_long) * 
        (range_20 + 1e-7)**(-1)
    )
    
    # Rank the factor values across all stocks on each date
    factor_ranked = factor.groupby(level=0).rank(pct=True)
    
    return factor_ranked
```
