def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving average of the closing price
    df['close_ma_5'] = df['close'].rolling(window=5).mean()
    df['close_ma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the current close and the 5-day moving average
    df['close_diff_5'] = df['close'] - df['close_ma_5']
    
    # Calculate the 5-day rolling standard deviation of the closing prices to capture volatility
    df['close_std_5'] = df['close'].rolling(window=5).std()
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # Calculate the difference between the VWAP and the 20-day moving average
    df['vwap_diff_20'] = df['vwap'] - df['close_ma_20']
    
    # Calculate the momentum as the difference in the 5-day moving average over the last 10 days
    df['momentum_5_10'] = df['close_ma_5'] - df['close_ma_5'].shift(10)
    
    # Alpha factor combining trend, volatility, and volume-weighted components
    alpha_factor = (df['close_diff_5'] / df['close_std_5']) * (df['vwap_diff_20'] / df['close_ma_20']) * (df['momentum_5_10'] / df['close'])

    return alpha_factor
