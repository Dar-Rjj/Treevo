```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change of volume from the previous day
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # Calculate a factor that combines the relative closing price within the day's range and the volume change
    factor = (df['close'] - df['low']) / (price_range + 1e-7) * (1 + df['volume_pct_change'])
    
    return factor
```
