```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price and volume changes
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the Exponential Moving Average (EMA) of price and volume changes
    df['ema_price_change'] = df['price_change'].ewm(span=10, adjust=False).mean()
    df['ema_volume_change'] = df['volume_change'].ewm(span=10, adjust=False).mean()

    # Calculate the 30-day and 60-day momentum
    df['momentum_30'] = df['close'] / df['close'].shift(30) - 1
    df['momentum_60'] = df['close'] / df['close'].shift(60) - 1

    # Calculate the 30-day volatility
    df['volatility_30'] = df['close'].rolling(window=30).std()

    # Calculate the Relative Strength Index (RSI) for 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))

    # Calculate the Money Flow Index (MFI) for 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_sum / negative_money_flow_sum
    df['mfi_14'] = 100 - (100 / (1 + money_ratio))

    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['obv'] = obv.cumsum()

    # Combine the factors into a single alpha signal
    # The factor assumes that the stronger the positive 30- and 60-day momentum,
    # the lower the volatility, the higher the RSI and MFI, and the higher the OBV,
    # and the higher the product of the EMA of price and volume changes, the higher the factor value
    df['alpha_factor'] = (5 * df['momentum_30'] + 3 * df['momentum_60'] - 2 * df['volatility_30'] + 
                          0.5 * df['rsi_14'] + 0.5 * df['mfi_14'] + 0.2 * df['obv'] +
                          4 * df['ema_price_change'] * df['ema_volume_change'])

    return df['alpha_factor']
```
