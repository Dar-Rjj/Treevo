```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the Exponential Moving Average (EMA) of price and volume changes
    df['ema_price_change'] = df['price_change'].ewm(span=10, adjust=False).mean()
    df['ema_volume_change'] = df['volume_change'].ewm(span=10, adjust=False).mean()

    # Calculate the 30-day and 60-day momentum
    df['momentum_30'] = df['close'] / df['close'].shift(30) - 1
    df['momentum_60'] = df['close'] / df['close'].shift(60) - 1

    # Calculate the 30-day volatility
    df['volatility_30'] = df['close'].rolling(window=30).std()

    # Calculate the Relative Strength Index (RSI) for 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))

    # Combine the factors into a single alpha signal
    # The factor assumes that the stronger the positive 30- and 60-day momentum, 
    # the lower the volatility, the higher the RSI, and the higher the product of 
    # the EMA of price and volume changes, the higher the factor value
    df['alpha_factor'] = (5 * df['momentum_30'] + 3 * df['momentum_60'] - 2 * df['volatility_30'] + 
                          0.5 * df['rsi_14'] + 4 * df['ema_price_change'] * df['ema_volume_change'])

    return df['alpha_factor']
```
