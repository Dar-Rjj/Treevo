```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day price volatility
    volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the 10-day and 30-day moving averages of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength between the 10-day and 30-day moving averages
    trend_strength = (ma_10 - ma_30) / ma_30
    
    # Calculate the dynamic price trend using the rate of change in the 10-day moving average
    dynamic_trend = ma_10.pct_change()
    
    # Combine the momentum, volume, volatility, trend strength, and dynamic trend into a single factor
    factor = (momentum * (avg_volume + 1e-7)**(-1) * (volatility + 1e-7)**(-1) * trend_strength * dynamic_trend).rank(pct=True)
    
    return factor
```
