```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price, then divide it by the average of high and low prices of the day
    # This factor is an attempt to gauge the momentum and strength of price movement in a day, 
    # while also taking into account the overall range. A higher value could indicate a strong upward move.
    return (df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
```
