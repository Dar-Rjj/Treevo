```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day price volatility
    volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the interaction between the momentum and the relative position of the 10-day MA to the 30-day MA
    trend_strength = (ma_10 - ma_30) / df['close']
    
    # Calculate the 20-day high and low
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    
    # Calculate the 20-day range
    range_20 = (high_20 - low_20) / df['close']
    
    # Calculate the RSI (Relative Strength Index) over 14 days
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the momentum, volume, volatility, trend strength, 20-day range, and RSI into a single factor
    factor = (momentum * (avg_volume + 1e-7)**(-1) * (volatility + 1e-7)**(-1) * trend_strength * (range_20 + 1e-7)**(-1) * (rsi / 100)).rank(pct=True)
    
    return factor
```
