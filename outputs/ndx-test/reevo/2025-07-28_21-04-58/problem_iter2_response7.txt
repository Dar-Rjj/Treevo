```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate moving average of close prices over a 10-day window
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate difference between the current close price and the 10-day moving average
    diff_ma_10 = df['close'] - ma_10
    
    # Calculate 30-day momentum
    momentum_30 = df['close'] / df['close'].shift(30) - 1
    
    # Calculate 60-day momentum for a longer-term perspective
    momentum_60 = df['close'] / df['close'].shift(60) - 1
    
    # Calculate the volatility (standard deviation) of the last 30 days' close prices
    volatility_30 = df['close'].rolling(window=30).std()
    
    # Calculate the change in volume from 5 days ago to today
    volume_change_5 = df['volume'] / df['volume'].shift(5) - 1
    
    # Combine the factors into a single alpha signal
    # The factor assumes that the closer the current price is to the 10-day MA, 
    # the stronger the positive 30- and 60-day momentum, the lower the volatility, 
    # and the higher the trading volume change, the higher the factor value
    alpha_factor = diff_ma_10 + 5 * momentum_30 + 3 * momentum_60 - 2 * volatility_30 + 4 * volume_change_5
    return alpha_factor
```
