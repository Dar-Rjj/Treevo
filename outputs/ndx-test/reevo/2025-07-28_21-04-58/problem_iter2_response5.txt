```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the exponential moving average of price and volume changes
    df['ema_price_change'] = df['price_change'].ewm(span=10, adjust=False).mean()
    df['ema_volume_change'] = df['volume_change'].ewm(span=10, adjust=False).mean()

    # Calculate daily volatility
    df['volatility'] = df['close'].pct_change().rolling(window=20).std()

    # Incorporate market sentiment (e.g., using a simple high-low ratio as a proxy)
    df['high_low_ratio'] = (df['high'] - df['low']) / df['close']

    # Alpha factor: the product of the exponential moving averages of price and volume changes, 
    # adjusted by volatility and market sentiment
    df['alpha_factor'] = (df['ema_price_change'] * df['ema_volume_change']) / (df['volatility'] * df['high_low_ratio'])

    return df['alpha_factor']
```
