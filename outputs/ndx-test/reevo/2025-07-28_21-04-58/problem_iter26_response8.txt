def heuristics_v2(df: pd.DataFrame, sector_trends: pd.Series, macro_indicators: pd.Series) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the volume-weighted average price over the last 10 days
    vwap_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day price volatility using exponential smoothing
    volatility = df['close'].pct_change().ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day and 30-day exponentially weighted moving averages of the close price
    ewma_10 = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the trend strength as the relative position of the 10-day EWMA to the 30-day EWMA
    trend_strength = (ewma_10 - ewma_30) / df['close']
    
    # Calculate the 20-day and 50-day high-low ranges to capture short-term and long-term price movements
    range_20 = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    range_50 = df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()
    
    # Calculate the 20-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    rsi_20 = 100 - (100 / (1 + rs))
    
    # Calculate the volume ratio (current volume / 10-day average volume)
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the 60-day exponentially weighted moving average of the close price for longer-term trend
    ewma_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the 60-day high-low range to capture very long-term price movement
    range_60 = df['high'].rolling(window=60).max() - df['low'].rolling(window=60).min()
    
    # Incorporate the 60-day EWMA into the trend strength
    long_trend_strength = (ewma_10 - ewma_60) / df['close']
    
    # Adaptive weights based on sector trends and macroeconomic indicators
    sector_weight = (sector_trends + 1e-7) / (sector_trends + 1e-7).sum()
    macro_weight = (macro_indicators + 1e-7) / (macro_indicators + 1e-7).sum()
    
    # Combine the selected features into a single factor with adaptive weights
    factor = (momentum * sector_weight + vwap_10 * sector_weight + (volatility + 1e-7)**(-1) * macro_weight + 
              trend_strength * sector_weight + (range_20 + 1e-7)**(-1) * macro_weight + (range_50 + 1e-7)**(-1) * macro_weight + 
              (range_60 + 1e-7)**(-1) * macro_weight + (rsi_20 / 100) * sector_weight + (volume_ratio + 1e-7) * sector_weight + 
              long_trend_strength * sector_weight)
