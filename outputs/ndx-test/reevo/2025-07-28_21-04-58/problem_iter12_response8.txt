```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'].pct_change(5)
    
    # Calculate the volume-weighted average price over the last 10 days
    vwap_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    vwap_10_pct = vwap_10.pct_change()
    
    # Calculate the 10-day price volatility using exponential smoothing
    volatility = df['close'].pct_change().ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day and 30-day exponentially weighted moving averages of the close price
    ewma_10 = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30 = df['close'].ewm(span=30, adjust=False).mean()
    trend_strength = (ewma_10 - ewma_30) / df['close']
    
    # Calculate the 20-day and 50-day high-low ranges to capture short-term and long-term price movements
    range_20 = (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / df['close']
    range_50 = (df['high'].rolling(window=50).max() - df['low'].rolling(window=50).min()) / df['close']
    
    # Calculate the 20-day and 50-day volume-weighted average prices
    vwap_20 = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    vwap_50 = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    vwap_20_pct = vwap_20.pct_change()
    vwap_50_pct = vwap_50.pct_change()
    
    # Calculate the 20-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    rsi_20 = 100 - (100 / (1 + rs))
    
    # Calculate the volume ratio (current volume / 10-day average volume)
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the 60-day exponentially weighted moving average of the close price for longer-term trend
    ewma_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the 60-day high-low range to capture very long-term price movement
    range_60 = (df['high'].rolling(window=60).max() - df['low'].rolling(window=60).min()) / df['close']
    
    # Calculate the 60-day volume-weighted average price
    vwap_60 = (df['close'] * df['volume']).rolling(window=60).sum() / df['volume'].rolling(window=60).sum()
    vwap_60_pct = vwap_60.pct_change()
    
    # Combine the momentum, volume-weighted average prices, volatility, trend strength, ranges, RSI, and volume ratio into a single factor
    factor = (momentum * 
              vwap_10_pct * 
              vwap_20_pct * 
              vwap_50_pct * 
              vwap_60_pct * 
              (volatility + 1e-7)**(-1) * 
              trend_strength * 
              range_20 * 
              range_50 * 
              range_60 * 
              (rsi_20 / 100) * 
              (volume_ratio + 1e-7))
    
    # Apply rank transformation for robustness
    factor = factor.rank(pct=True)
    
    return factor
```
