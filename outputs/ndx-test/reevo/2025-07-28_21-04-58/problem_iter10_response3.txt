```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day return as a momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Calculate the 10-day moving average of the close price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    ma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the trend strength as the relative position of the 10-day MA to the 30-day MA, normalized by close price
    trend_strength = (ma_10 - ma_30) / df['close']
    
    # Combine the momentum and trend strength into a single factor
    factor = momentum * trend_strength
    
    # Rank the factor values across all stocks on each date
    factor_ranked = factor.groupby(level=0).rank(pct=True)
    
    return factor_ranked
```
