def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    df['price_diff'] = df['close'].diff()
    # Calculate the weighted moving average of the price difference using volume as weights
    wma_series = (df['price_diff'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    # Calculate a smoothed version of this WMA with another simple moving average
    smoothed_wma = wma_series.rolling(window=5).mean()
    # Return the ratio of today's smooth WMA to its 20-day mean, attempting to capture momentum while smoothing out noise
    return smoothed_wma / smoothed_wma.rolling(window=20).mean()
