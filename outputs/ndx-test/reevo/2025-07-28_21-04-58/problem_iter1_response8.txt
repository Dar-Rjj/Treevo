```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price over a 5 day period
    df['high_low_diff_5d'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate the sum of volume over a 5 day period
    df['volume_sum_5d'] = df['volume'].rolling(window=5).sum()
    
    # Alpha factor: average true range over 5 days normalized by the 5 day volume
    alpha_factor = (df['high_low_diff_5d']) / (df['volume_sum_5d'] + 1e-7)
    
    return alpha_factor
```
