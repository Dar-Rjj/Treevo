```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the weighted moving average of price and volume changes
    df['wma_price_change'] = df['price_change'].rolling(window=10).mean()
    df['wma_volume_change'] = df['volume_change'].rolling(window=10).mean()

    # Calculate the volatility of the close prices
    df['volatility'] = df['close'].rolling(window=10).std()

    # Calculate the momentum of the close prices
    df['momentum'] = df['close'] / df['close'].shift(10)

    # Integrate sector performance (assuming a 'sector' column is available in the DataFrame)
    sector_performance = df.groupby('sector')['close'].transform(lambda x: x.rolling(window=10).mean())

    # Combine factors with weights
    df['alpha_factor'] = (
        0.4 * df['wma_price_change'] +
        0.3 * df['wma_volume_change'] +
        0.2 * df['volatility'] +
        0.1 * df['momentum'] +
        0.1 * sector_performance
    )

    return df['alpha_factor']
```
