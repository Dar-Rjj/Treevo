```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    price_diff = df['close'] - df['close'].shift(1)
    
    # Calculate the average of the absolute values of the past 5 day's price differences
    avg_price_diff = abs(df['close'] - df['close'].shift(1)).rolling(window=5).mean()
    
    # The factor is the ratio of the current price difference to the average price difference
    factor = price_diff / (avg_price_diff + 1e-7)
    return factor
```
