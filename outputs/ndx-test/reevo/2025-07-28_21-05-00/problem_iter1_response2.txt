```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a form of momentum
    momentum = df['close'].diff(1)
    
    # Calculate the average volume over the past 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the price volatility over the last 10 days using the standard deviation of the daily returns
    daily_returns = df['close'].pct_change()
    volatility_10d = daily_returns.rolling(window=10).std()
    
    # Combine the momentum, average volume, and volatility into a composite alpha factor
    # The idea is that positive momentum, high volume, and low volatility might indicate a good buying opportunity
    alpha_factor = momentum * (avg_volume_5d / (volatility_10d + 1e-7))
    
    return alpha_factor
```
