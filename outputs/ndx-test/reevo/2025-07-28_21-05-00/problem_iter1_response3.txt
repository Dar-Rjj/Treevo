```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a percentage change
    df['close_percent_change'] = (df['close'] / df['close'].shift(1) - 1)
    # Calculate average volume over the past 5 days
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    # Alpha factor: weighted combination of percent change in close and deviation from average volume
    df['factor'] = (df['close_percent_change'] * df['avg_vol_5d']) - (df['high'] - df['low'])
    return df['factor']
```
