```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the smoothed momentum using exponential moving average (EMA) over the last 5 days
    smoothed_momentum = df['close'].ewm(span=5, adjust=False).mean().diff(1)
    
    # Calculate the relative strength of the stock by comparing the closing price to the 20-day simple moving average (SMA)
    sma_20d = df['close'].rolling(window=20).mean()
    relative_strength = df['close'] / sma_20d
    
    # Calculate the volume-weighted average price (VWAP) over the last 10 days
    vwap_10d = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the volatility using the standard deviation of the daily returns over the last 10 days
    daily_returns = df['close'].pct_change()
    volatility_10d = daily_returns.rolling(window=10).std()
    
    # Combine the smoothed momentum, relative strength, and smart volume adjustments into a composite alpha factor
    # The idea is that positive smoothed momentum, strong relative strength, and stable VWAP might indicate a good buying opportunity
    alpha_factor = smoothed_momentum * (relative_strength + 1) * (vwap_10d / (volatility_10d + 1e-7))
    
    return alpha_factor
```
