def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    price_change = df['close'] - df['open']
    volume_change = df['volume'].pct_change()

    # Calculate the average of high and low, then subtract the close price
    avg_high_low = (df['high'] + df['low']) / 2
    diff_close_avg = df['close'] - avg_high_low

    # Calculate the momentum factor as the difference between the current close and the 5-day moving average of the close
    momentum_5d = df['close'] - df['close'].rolling(window=5).mean()

    # Calculate the relative strength index (RSI) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volatility as the standard deviation of the price change over the last 10 days
    volatility = price_change.rolling(window=10).std()

    # Calculate the money flow index (MFI) for the last 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(price_change > 0, 0)
    negative_money_flow = raw_money_flow.where(price_change < 0, 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))

    # Calculate the Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line

    # Calculate a shorter-term trend indicator, such as the 20-day moving average
    short_term_trend = df['close'].rolling(window=20).mean()

    # Adaptive weighting based on the short-term trend
    trend_signal = (df['close'] - short_term_trend) / short_term_trend
    adaptive_weight = (trend_signal + 1) / 2  # Normalize to [0, 1]

    # Dynamic weights for each indicator
    weight_price_change = adaptive_weight * 0.2
    weight_volume_change = adaptive_weight * 0.1
    weight_diff_close_avg = adaptive_weight * 0.1
    weight_momentum_5d = adaptive_weight * 0.2
    weight_rsi = adaptive_weight * 0.1
    weight_volatility = adaptive_weight * 0.1
    weight_mfi = adaptive_weight * 0.1
    weight_macd_histogram = adaptive_weight * 0.1
    weight_bid_ask_spread = adaptive_weight * 0.1
    weight_sentiment_score = adaptive_weight * 0.1

    # Integrate multiple indicators to create a robust factor
    factor = (weight_price_change * price_change + 
              weight_volume_change * volume_change + 
              weight_diff_close_avg * diff_close_avg + 
              weight_momentum_5d * momentum_5d + 
              weight_rsi * rsi + 
              weight_volatility * volatility + 
              weight_mfi * mfi + 
              weight_macd_histogram * macd_histogram)

    # Additional dynamic sentiment analysis integration
    sentiment_score = (df['close'] - df['open']).rolling(window=7).mean() / df['close'].rolling(window=7).std()
    factor += weight_sentiment_score * sentiment_score

    # Additional market microstructure indicator
    bid_ask_spread = df['high'] - df['low']
    factor += weight_bid_ask_spread * bid_ask_spread

    # Apply non-linear transformation
    factor = np.sign(factor) * (np.abs(factor) ** 0.5)

    # Exponential moving average to smooth the factor
    factor = factor.ewm(span=5, adjust=False).mean()

    return factor
