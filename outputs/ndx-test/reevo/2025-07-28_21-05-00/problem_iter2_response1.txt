```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].diff()
    
    # Calculate the average of the daily high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 10-day Exponential Moving Average (EMA) of the closing prices
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day Exponential Moving Average (EMA) of the closing prices
    ema_30d = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 10-day standard deviation of the closing prices as a measure of volatility
    std_dev_10d = df['close'].rolling(window=10).std()
    
    # Combine the daily return, average price, EMA volume, and volatility to create an alpha factor
    alpha_factor = (daily_return * avg_price / ema_volume) / (std_dev_10d + 1e-7)
    
    # Apply a trend component by comparing 10-day EMA to 30-day EMA
    trend = ema_10d > ema_30d
    
    # Weight the alpha factor by the trend component
    weighted_alpha_factor = alpha_factor * trend.astype(int)
    
    return weighted_alpha_factor
```
