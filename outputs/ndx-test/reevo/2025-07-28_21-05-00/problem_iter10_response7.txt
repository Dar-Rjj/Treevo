```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    price_change = df['close'] - df['open']
    volume_change = df['volume'].pct_change()

    # Calculate the average of high and low, then subtract the close price
    avg_high_low = (df['high'] + df['low']) / 2
    diff_close_avg = df['close'] - avg_high_low

    # Calculate the momentum factor as the difference between the current close and the 5-day moving average of the close
    momentum_5d = df['close'] - df['close'].rolling(window=5).mean()

    # Calculate the relative strength index (RSI) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volatility as the standard deviation of the price change over the last 10 days
    volatility = price_change.rolling(window=10).std()

    # Calculate the money flow index (MFI) for the last 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(price_change > 0, 0)
    negative_money_flow = raw_money_flow.where(price_change < 0, 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))

    # Calculate the stochastic oscillator
    k_period = 14
    d_period = 3
    min_low = df['low'].rolling(window=k_period).min()
    max_high = df['high'].rolling(window=k_period).max()
    stoch_k = 100 * (df['close'] - min_low) / (max_high - min_low)
    stoch_d = stoch_k.rolling(window=d_period).mean()

    # Calculate the adaptive weighting based on recent performance
    recent_performance = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    adaptive_weight = recent_performance.ewm(span=5, adjust=False).mean()

    # Weigh factors by their predictive power and apply adaptive weighting
    weighted_price_change = (price_change * 0.3) * adaptive_weight
    weighted_volume_change = (volume_change * 0.2) * adaptive_weight
    weighted_diff_close_avg = (diff_close_avg * 0.1) * adaptive_weight
    weighted_momentum = (momentum_5d * 0.1) * adaptive_weight
    weighted_rsi = (rsi * 0.1) * adaptive_weight
    weighted_volatility = (volatility * 0.1) * adaptive_weight
    weighted_mfi = (mfi * 0.1) * adaptive_weight
    weighted_stoch_k = (stoch_k * 0.1) * adaptive_weight

    # Sum the weighted factors
    factor = (weighted_price_change + weighted_volume_change + weighted_diff_close_avg + weighted_momentum + weighted_rsi + weighted_volatility + weighted_mfi + weighted_stoch_k).ewm(span=5, adjust=False).mean()
    
    return factor
```
