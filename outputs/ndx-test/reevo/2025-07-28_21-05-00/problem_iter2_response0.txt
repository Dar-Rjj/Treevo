```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 30-day momentum
    df['momentum_5d'] = df['close'].pct_change(5)
    df['momentum_30d'] = df['close'].pct_change(30)

    # Calculate the average volume over the past 5 days
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()

    # Calculate the 10-day and 30-day standard deviation of returns to measure volatility
    df['volatility_10d'] = df['close'].pct_change().rolling(window=10).std()
    df['volatility_30d'] = df['close'].pct_change().rolling(window=30).std()

    # Calculate the difference between high and low price (range) over a 10-day window
    df['range_10d'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()

    # Calculate the 10-day and 30-day moving average of the closing prices
    df['ma_10d'] = df['close'].rolling(window=10).mean()
    df['ma_30d'] = df['close'].rolling(window=30).mean()

    # Combine momentum, volume, and volatility factors
    df['factor_value'] = (df['momentum_5d'] - df['momentum_30d']) * df['avg_vol_5d'] / (df['volatility_10d'] + 1e-7)

    # Apply a trend component by comparing 10-day MA to 30-day MA
    df['trend'] = df['ma_10d'] > df['ma_30d']

    # Weight the factor value by the trend component
    df['weighted_factor'] = df['factor_value'] * df['trend'].astype(int)

    return df['weighted_factor']
```
