def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the smoothed average of high and low, representing the typical price movement
    smooth_avg_price = (df['high'] + df['low']).rolling(window=5).mean()
    
    # Compute a momentum factor by comparing the current close with the 10-day moving average
    ma_10 = df['close'].rolling(window=10).mean()
    momentum_factor = df['close'] - ma_10
    
    # Volatility factor using the 20-day standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility_factor = log_returns.rolling(window=20).std()
    
    # Combined factor: smoothed price change * momentum factor * volatility factor, scaled by volume
    combined_factor = (smooth_avg_price.diff().fillna(0) * momentum_factor * volatility_factor) * df['volume']
    
    return combined_factor
