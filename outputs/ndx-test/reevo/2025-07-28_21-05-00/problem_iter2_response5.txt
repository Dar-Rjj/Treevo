```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum
    momentum_5d = df['close'] - df['close'].shift(5)
    
    # Calculate the 5-day volatility (standard deviation of log returns)
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility_5d = log_returns.rolling(window=5).std()
    
    # Calculate the 5-day average volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day high-low difference
    high_low_diff_5d = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate the 5-day average daily range
    avg_daily_range_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combine momentum, volatility, and volume
    interaction_factor = (momentum_5d / volatility_5d) * (avg_volume_5d / high_low_diff_5d)
    
    # Incorporate market trend (e.g., 50-day moving average)
    trend_50d = df['close'].rolling(window=50).mean()
    trend_factor = (df['close'] - trend_50d) / trend_50d
    
    # Final alpha factor as a combination of interaction and trend factors
    alpha_factor = interaction_factor + trend_factor
    return alpha_factor
```
