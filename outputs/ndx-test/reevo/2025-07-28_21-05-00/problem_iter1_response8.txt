```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low over the last 5 days.
    high_low_diff_5d = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate the average of the daily range (high - low) over the last 5 days.
    avg_daily_range_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the momentum factor as the difference between the current close and the 5-day moving average of the close.
    momentum_5d = df['close'] - df['close'].rolling(window=5).mean()
    
    # The alpha factor is a combination of the normalized momentum and the ratio of the 5-day high-low difference to the average daily range.
    alpha_factor = (momentum_5d / avg_daily_range_5d) * (high_low_diff_5d / avg_daily_range_5d)
    return alpha_factor
```
