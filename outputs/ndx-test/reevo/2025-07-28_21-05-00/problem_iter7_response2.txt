```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    price_change = df['close'] - df['open']
    volume_change = df['volume'].pct_change()

    # Calculate the average of high and low, then subtract the close price
    avg_high_low = (df['high'] + df['low']) / 2
    diff_close_avg = df['close'] - avg_high_low

    # Calculate the momentum factor as the difference between the current close and the 5-day exponentially weighted moving average of the close
    momentum_5d_ewm = df['close'] - df['close'].ewm(span=5, adjust=False).mean()

    # Calculate the relative strength index (RSI) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volatility as the standard deviation of the price change over the last 10 days
    volatility = price_change.rolling(window=10).std()

    # Calculate the money flow index (MFI) for the last 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(price_change > 0, 0)
    negative_money_flow = raw_money_flow.where(price_change < 0, 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))

    # Adaptive weighting based on recent performance
    weight_price_change = abs(price_change.rolling(window=5).mean()) / abs(price_change.rolling(window=5).std())
    weight_volume_change = abs(volume_change.rolling(window=5).mean()) / abs(volume_change.rolling(window=5).std())
    weight_momentum = abs(momentum_5d_ewm.rolling(window=5).mean()) / abs(momentum_5d_ewm.rolling(window=5).std())
    weight_rsi = abs(rsi.rolling(window=5).mean()) / abs(rsi.rolling(window=5).std())
    weight_volatility = abs(volatility.rolling(window=5).mean()) / abs(volatility.rolling(window=5).std())
    weight_mfi = abs(mfi.rolling(window=5).mean()) / abs(mfi.rolling(window=5).std())

    # Integrate multiple indicators with adaptive weighting to create a robust factor
    factor = (price_change * weight_price_change + 
              volume_change * weight_volume_change + 
              diff_close_avg + 
              momentum_5d_ewm * weight_momentum + 
              rsi * weight_rsi + 
              volatility * weight_volatility + 
              mfi * weight_mfi).ewm(span=5, adjust=False).mean()
    return factor
```
