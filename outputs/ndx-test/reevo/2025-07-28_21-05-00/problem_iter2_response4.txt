```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the average of the daily high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 10-day exponential moving average of the closing prices
    ema_close_10d = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the closing prices
    ema_close_30d = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the momentum component
    momentum = (ema_close_10d - ema_close_30d) / ema_close_30d
    
    # Calculate the volatility as the standard deviation of the past 10 days' returns
    volatility = df['close'].pct_change().rolling(window=10).std()
    
    # Combine the daily return, average price, EMA volume, momentum, and inverse of volatility to create an alpha factor
    alpha_factor = daily_return * avg_price / ema_volume * momentum / (volatility + 1e-7)
    
    return alpha_factor
```
