```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day moving average to capture a longer-term trend.
    df['ma_20d'] = df['close'].rolling(window=20).mean()
    
    # Calculate the momentum factor as the difference between the current close and the 20-day moving average.
    momentum_20d = df['close'] - df['ma_20d']
    
    # Calculate the 20-day standard deviation to use as a robust volatility measure.
    df['std_20d'] = df['close'].rolling(window=20).std()
    
    # Calculate the 20-day high-low difference to capture the range of price movement.
    high_low_diff_20d = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 20-day average true range to capture the average daily range.
    avg_true_range_20d = (df['high'] - df['low']).rolling(window=20).mean()
    
    # The alpha factor is a combination of the normalized momentum, the ratio of the 20-day high-low difference to the average true range, and the ratio of the 20-day standard deviation.
    alpha_factor = (momentum_20d / (df['std_20d'] + 1e-7)) * (high_low_diff_20d / (avg_true_range_20d + 1e-7))
    return alpha_factor
```
