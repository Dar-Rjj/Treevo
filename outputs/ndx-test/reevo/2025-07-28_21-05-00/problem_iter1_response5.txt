```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the current close and the previous close, capturing the momentum
    df['momentum'] = df['close'].diff()
    
    # Calculate the average of the high and low for a true range like indicator which can suggest volatility
    df['avg_true_range'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # A factor that incorporates the momentum over the average true range, attempting to capture significant price movements in more volatile markets
    alpha_factor = df['momentum'] / (df['avg_true_range'] + 1e-7)
    return alpha_factor
```
