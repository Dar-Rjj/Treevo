```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average of the high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate a factor that combines the price change with the average range
    factor = diff_close_open / (avg_high_low + 1e-7)
    
    return factor
```
