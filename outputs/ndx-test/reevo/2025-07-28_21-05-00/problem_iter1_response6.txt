```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    daily_return = df['close'].diff()
    
    # Calculate the average of the daily high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Combine the daily return, average price, and EMA volume to create an alpha factor
    alpha_factor = daily_return * avg_price / ema_volume
    
    return alpha_factor
```
