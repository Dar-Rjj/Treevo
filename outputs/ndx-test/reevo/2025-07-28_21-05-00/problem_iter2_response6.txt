```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return as a percentage change in close price
    df['daily_return'] = (df['close'] / df['close'].shift(1) - 1)
    
    # Calculate the 5-day average volume
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the 10-day standard deviation of daily returns for volatility
    df['volatility_10d'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day momentum as the difference between current and 10 days ago close price
    df['momentum_10d'] = df['close'] - df['close'].shift(10)
    
    # Alpha factor: combine daily return, volatility, and momentum, weighted by 5-day average volume
    df['factor_value'] = (df['daily_return'] * df['avg_vol_5d']) + (df['volatility_10d'] * df['momentum_10d'])
    
    return df['factor_value']
```
