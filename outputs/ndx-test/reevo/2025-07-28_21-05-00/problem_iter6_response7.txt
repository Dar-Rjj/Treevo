```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate price change and volume change
    price_change = df['close'] - df['open']
    volume_change = df['volume'].pct_change()

    # Calculate the average of high and low, then subtract the close price
    avg_high_low = (df['high'] + df['low']) / 2
    diff_close_avg = df['close'] - avg_high_low

    # Calculate the momentum factor as the difference between the current close and the adaptive moving average of the close
    momentum_window = 5 + (df['volume'] > df['volume'].rolling(window=30).mean()).cumsum()
    momentum = df['close'] - df['close'].rolling(window=momentum_window, min_periods=1).mean()

    # Calculate the relative strength index (RSI) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the volatility as the standard deviation of the price change over the last 10 days
    volatility = price_change.rolling(window=10).std()

    # Calculate the money flow index (MFI) for the last 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(price_change > 0, 0)
    negative_money_flow = raw_money_flow.where(price_change < 0, 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))

    # Incorporate liquidity and market cap
    if 'market_cap' in df.columns and 'liquidity' in df.columns:
        liquidity_factor = df['liquidity'].ewm(span=5, adjust=False).mean()
        market_cap_factor = df['market_cap'].ewm(span=5, adjust=False).mean()
    else:
        liquidity_factor = 1
        market_cap_factor = 1

    # Integrate multiple indicators to create a robust factor
    # Apply exponential smoothing to the combined factors
    factor = (price_change * volume_change + diff_close_avg + momentum + rsi + volatility + mfi) * liquidity_factor * market_cap_factor
    factor = factor.ewm(span=5, adjust=False).mean()

    # Add a sector-specific factor (assuming 'sector' is a column in the DataFrame)
    if 'sector' in df.columns:
        sector_factor = df.groupby('sector')['close'].transform(lambda x: x.ewm(span=5, adjust=False).mean())
        factor = factor + sector_factor

    # Consider macroeconomic indicators (assuming 'macro_indicator' is a column in the DataFrame)
    if 'macro_indicator' in df.columns:
        macro_factor = df['macro_indicator'].ewm(span=5, adjust=False).mean()
        factor = factor * macro_factor

    return factor
```
