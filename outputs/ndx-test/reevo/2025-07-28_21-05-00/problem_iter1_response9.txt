```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close, high, and low prices
    diff_close = df['close'].diff().fillna(0)
    diff_high = df['high'].diff().fillna(0)
    diff_low = df['low'].diff().fillna(0)

    # Sum of the absolute differences, which represents the total price movement
    total_movement = abs(diff_close) + abs(diff_high) + abs(diff_low)

    # Calculate average true range over 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()

    # Factor: Ratio of total price movement to average true range, which captures the relative volatility
    factor = total_movement / atr
    return factor
```
