```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price over a 10-day window.
    df['range_10d'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    
    # Calculate the 10-day moving average of the closing prices.
    df['ma_10d'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing prices.
    df['ma_30d'] = df['close'].rolling(window=30).mean()

    # Calculate the distance from the current close to the 10-day MA as a percentage of the 10-day range
    df['factor_value'] = (df['close'] - df['ma_10d']) / (df['range_10d'] + 1e-7)
    
    # Apply a trend component by comparing 10-day MA to 30-day MA
    df['trend'] = df['ma_10d'] > df['ma_30d']
    
    # Weight the factor value by the trend component
    df['weighted_factor'] = df['factor_value'] * df['trend'].astype(int)
    
    return df['weighted_factor']
```
