def heuristics_v2(df):
    # Calculate the exponential moving average of the last 14 days' closing prices
    ema_last_14_close = df['close'].ewm(span=14).mean()
    # Calculate the exponential moving average of the next 14 days' opening prices
    ema_next_14_open = df['open'].shift(-14).ewm(span=14).mean()
    # Calculate the factor as the ratio of the two EMAs
    factor_values = ema_last_14_close / ema_next_14_open
    # Apply a rolling standard deviation for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).std()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
