def heuristics_v2(df):
    def price_range(series, window):
        return series.rolling(window=window).max() - series.rolling(window=window).min()

    def std_dev_returns(series, window):
        return series.pct_change().rolling(window=window).std()
    
    df['Price_Range'] = price_range(df['high'], 20) - price_range(df['low'], 20)
    df['Std_Dev_Returns'] = std_dev_returns(df['close'], 20)
    heuristics_matrix = df['Price_Range'] / df['Std_Dev_Returns']
    return heuristics_matrix
