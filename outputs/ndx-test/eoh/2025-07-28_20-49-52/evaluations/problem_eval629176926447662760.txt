def heuristics_v2(df):
    # Calculate the daily trading range
    df['Trading_Range'] = df['high'] - df['low']
    
    # Calculate the 10-day cumulative sum of the daily trading range
    df['CumSum_TR_10'] = df['Trading_Range'].rolling(window=10).sum()
    
    # Calculate the 50-day cumulative sum of volume
    df['CumSum_Volume_50'] = df['volume'].rolling(window=50).sum()
    
    # Generate the heuristic by dividing the 10-day cumulative sum of the trading range by the 50-day cumulative sum of volume
    heuristics_matrix = df['CumSum_TR_10'] / df['CumSum_Volume_50']
    
    return heuristics_matrix
