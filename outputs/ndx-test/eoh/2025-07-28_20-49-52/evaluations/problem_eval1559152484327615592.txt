def heuristics_v2(df):
    # Calculate the difference between the max and min of the last 7 days' closing prices
    diff_last_7_close = df['close'].rolling(window=7).max() - df['close'].rolling(window=7).min()
    # Calculate the average of the next 7 days' opening prices
    avg_next_7_open = df['open'].shift(-7).rolling(window=7).mean()
    # Compute the factor as the subtraction of the difference from the average
    factor_values = avg_next_7_open - diff_last_7_close
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
