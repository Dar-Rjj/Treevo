def heuristics_v2(df):
    # Calculate the EMA of the last 10 days' closing prices
    ema_last_10_close = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the EMA of the next 10 days' opening prices
    ema_next_10_open = df['open'].shift(-10).ewm(span=10, adjust=False).mean()
    # Calculate the factor as the difference between the two EMAs
    factor_values = ema_last_10_close - ema_next_10_open
    # Apply a rolling mean for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
