def heuristics_v2(df):
    # Calculate the weighted average of high and low prices using volume
    weighted_avg = (df['volume'] * df['high'] + df['volume'] * df['low']) / 2
    # Calculate the factor as the ratio of weighted average to close price
    factor_values = weighted_avg / df['close']
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
