def heuristics_v2(df):
    # Calculate the short-term and long-term averages of the close prices
    short_term_avg = df['close'].rolling(window=5).mean()
    long_term_avg = df['close'].rolling(window=20).mean()
    # Calculate the momentum factor
    momentum_factor = short_term_avg - long_term_avg
    # Apply an exponential moving average for smoothing the factor values
    smoothed_momentum = momentum_factor.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_momentum.dropna()
    return heuristics_matrix
