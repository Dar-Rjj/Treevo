def heuristics_v2(df):
    def std_dev_returns(series, window):
        return series.pct_change().rolling(window=window).std()

    def avg_daily_return(series, window):
        return series.pct_change().rolling(window=window).mean()
    
    df['STD_DEV_10'] = std_dev_returns(df['close'], 10)
    df['AVG_RET_30'] = avg_daily_return(df['close'], 30)
    heuristics_matrix = df['STD_DEV_10'] / df['AVG_RET_30']
    return heuristics_matrix
