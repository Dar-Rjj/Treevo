import pandas as pd

def heuristics_v2(df):
    def sma(series, window):
        return series.rolling(window=window).mean()

    def ema(series, span):
        return series.ewm(span=span, adjust=False).mean()
    
    df['SMA_Volume_30'] = sma(df['volume'], 30)
    df['EMA_Volume_10'] = ema(df['volume'], 10)
    heuristics_matrix = df['EMA_Volume_10'] / df['SMA_Volume_30']
    return heuristics_matrix
