def heuristics_v2(df):
    # Calculate the difference between the maximum high and minimum low prices over the last 10 days
    range_diff = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = range_diff.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
