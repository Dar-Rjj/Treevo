def heuristics_v2(df):
    # Calculate the sum of the last 5 days' closing prices
    sum_last_5_close = df['close'].rolling(window=5).sum()
    # Calculate the sum of the next 5 days' opening prices
    sum_next_5_open = df['open'].shift(-5).rolling(window=5).sum()
    # Calculate the factor as the ratio of the two sums
    factor_values = sum_last_5_close / sum_next_5_open
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
