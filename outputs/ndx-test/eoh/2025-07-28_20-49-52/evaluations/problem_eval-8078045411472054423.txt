def heuristics_v2(df):
    # Calculate the EMA of the last 10 days' close prices
    ema_last_10_close = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the EMA of the next 10 days' open prices
    ema_next_10_open = df['open'].shift(-10).ewm(span=10, adjust=False).mean()
    # Calculate the factor as the ratio of the two EMAs
    factor_values = ema_last_10_close / ema_next_10_open
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=15).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
