import pandas as pd

def heuristics_v2(df):
    # Calculate the Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    # Calculate the arithmetic mean of high, low, and close prices
    arith_mean_price = (df['high'] + df['low'] + df['close']) / 3
    # Calculate the factor as the ratio of VWAP to the arithmetic mean of high, low, and close prices
    factor_values = vwap / arith_mean_price
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
