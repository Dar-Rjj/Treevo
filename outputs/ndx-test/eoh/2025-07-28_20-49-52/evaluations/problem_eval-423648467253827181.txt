def heuristics_v2(df):
    avg_high_low = (df['high'].rolling(window=5).mean() + df['low'].rolling(window=5).mean()) / 2
    close_to_avg_ratio = df['close'] / avg_high_low
    volume_change = df['volume'].pct_change().fillna(0)
    heuristics_matrix = close_to_avg_ratio * volume_change
    return heuristics_matrix
