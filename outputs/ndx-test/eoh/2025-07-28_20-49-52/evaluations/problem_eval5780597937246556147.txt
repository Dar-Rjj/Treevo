def heuristics_v2(df):
    # Calculate the Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    # Calculate the weighted average of high, low, and close prices
    weighted_avg_price = df['high'] * 0.5 + df['low'] * 0.3 + df['close'] * 0.2
    # Calculate the exponential moving average of the weighted average price
    ema_weighted_avg_price = weighted_avg_price.ewm(span=10, adjust=False).mean()
    # Calculate the factor as the ratio of VWAP to the EMA of the weighted average price
    factor_values = vwap / ema_weighted_avg_price
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
