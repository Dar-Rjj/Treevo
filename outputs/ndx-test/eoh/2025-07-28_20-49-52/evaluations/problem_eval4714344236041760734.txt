def heuristics_v2(df):
    # Calculate the 20-day rolling median of the closing price
    rolling_median = df['close'].rolling(window=20).median()
    # Calculate the factor as the difference of the rolling median and the close price
    factor_values = rolling_median - df['close']
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
