def heuristics_v2(df):
    # Calculate the weighted sum of the last 5 days' high and low prices
    weighted_sum_last_5 = df['high'].rolling(window=5).sum() * 0.6 + df['low'].rolling(window=5).sum() * 0.4
    # Calculate the weighted sum of the next 5 days' opening and closing prices
    weighted_sum_next_5 = df['open'].shift(-5).rolling(window=5).sum() * 0.6 + df['close'].shift(-5).rolling(window=5).sum() * 0.4
    # Calculate the factor as the difference between the two weighted sums
    factor_values = weighted_sum_last_5 - weighted_sum_next_5
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=5).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
