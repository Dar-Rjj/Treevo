def heuristics_v2(df):
    # Calculate the median of the last 5 days' close prices
    median_last_5_close = df['close'].rolling(window=5).median()
    # Calculate the median of the next 5 days' open prices
    median_next_5_open = df['open'].shift(-5).rolling(window=5).median()
    # Calculate the factor as the ratio of the two medians
    factor_values = median_last_5_close / median_next_5_open
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
