import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute simple moving averages
    sma_short = df['close'].rolling(window=5).mean()
    sma_long = df['close'].rolling(window=20).mean()
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Compute Price Volume Trend
    pvt = ((df['close'] - df['close'].shift(1)) / df['close'].shift(1)) * df['volume']
    pvt = pvt.cumsum()
    
    # Combine factors into a weighted sum for the final heuristic
    heuristics_matrix = (sma_short - sma_long) + rsi + pvt
    
    return heuristics_matrix
