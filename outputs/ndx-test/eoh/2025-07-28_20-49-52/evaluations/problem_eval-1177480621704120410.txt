def heuristics_v2(df):
    # Calculate the sum of the last 10 days' closing prices
    sum_last_10_close = df['close'].rolling(window=10).sum()
    # Calculate the sum of the next 10 days' high prices
    sum_next_10_high = df['high'].shift(-10).rolling(window=10).sum()
    # Calculate the factor as the ratio of the two sums
    factor_values = sum_last_10_close / sum_next_10_high
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
