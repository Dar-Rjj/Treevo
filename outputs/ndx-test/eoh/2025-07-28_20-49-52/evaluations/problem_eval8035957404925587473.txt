import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day Rate of Change (ROC)
    df['ROC_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day average volume
    df['Avg_Volume_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of today's volume to the 30-day average volume
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_30']
    
    # Generate the final heuristic by combining ROC_10 and Volume_Ratio with weights
    heuristics_matrix = 0.6 * df['ROC_10'] + 0.4 * df['Volume_Ratio']
    
    return heuristics_matrix
