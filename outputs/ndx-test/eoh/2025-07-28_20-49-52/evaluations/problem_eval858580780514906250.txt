import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    short_ma = df['close'].rolling(window=10).mean()
    long_ma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between the short-term and long-term moving averages
    ma_diff = short_ma - long_ma
    
    # Calculate the standard deviation (volatility) of the close prices over a 20-day window
    volatility = df['close'].rolling(window=20).std()
    
    # Generate the heuristic score as a linear combination of the moving average difference and the volatility
    heuristics_matrix = 0.7 * ma_diff - 0.3 * volatility
    
    return heuristics_matrix
