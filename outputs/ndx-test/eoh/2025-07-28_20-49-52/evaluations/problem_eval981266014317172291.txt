import pandas as pd

def heuristics_v2(df):
    # Calculate short and long moving averages for closing prices
    short_window = 5
    long_window = 20
    short_mavg = df['close'].rolling(window=short_window, min_periods=1).mean()
    long_mavg = df['close'].rolling(window=long_window, min_periods=1).mean()

    # Compute volume change
    df['volume_change'] = df['volume'].pct_change().fillna(0)

    # Identify volume shocks
    volume_shock_threshold = 2
    df['volume_shock'] = (df['volume_change'] > volume_shock_threshold).astype(int)

    # Generate signal based on crossovers and volume shocks
    df['signal'] = 0
    df.loc[(short_mavg > long_mavg) & (df['volume_shock'] == 1), 'signal'] = 1
    df.loc[(short_mavg < long_mavg) & (df['volume_shock'] == 1), 'signal'] = -1

    heuristics_matrix = df['signal']
    
    return heuristics_matrix
