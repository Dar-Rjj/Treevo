def heuristics_v2(df):
    # Calculate the sum of the last 5 days' closing prices
    sum_last_5_close = df['close'].rolling(window=5).sum()
    # Calculate the sum of the next 3 days' high prices
    sum_next_3_high = df['high'].shift(-3).rolling(window=3).sum()
    # Calculate the factor as the ratio of the two sums
    factor_values = sum_last_5_close / sum_next_3_high
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
