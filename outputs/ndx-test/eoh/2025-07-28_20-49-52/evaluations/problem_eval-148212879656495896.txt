def heuristics_v2(df):
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    # Calculate the average trading range (high - low) over the last 10 days
    avg_trading_range = (df['high'] - df['low']).rolling(window=10).mean()
    # Calculate the factor as the ratio of average volume to the average trading range
    factor_values = avg_volume / avg_trading_range
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
