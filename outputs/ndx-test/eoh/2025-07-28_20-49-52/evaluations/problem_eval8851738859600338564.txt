def heuristics_v2(df):
    # Calculate the average of the last 10 days' closing prices
    avg_last_10_close = df['close'].rolling(window=10).mean()
    # Calculate the range (max - min) of the next 10 days' prices
    price_range_next_10 = df['high'].shift(-10).rolling(window=10).max() - df['low'].shift(-10).rolling(window=10).min()
    # Calculate the factor as the ratio of the two values
    factor_values = avg_last_10_close / price_range_next_10
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
