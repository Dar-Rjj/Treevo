def heuristics_v2(df):
    # Calculate the modified Volume-Weighted Average Price (VWAP) using open and close prices
    vwap = (df['volume'] * (df['open'] + df['close']) / 2).cumsum() / df['volume'].cumsum()
    # Calculate the factor as the ratio of modified VWAP to high price
    factor_values = vwap / df['high']
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10, min_periods=1).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
