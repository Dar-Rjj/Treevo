def heuristics_v2(df):
    # Calculate the median of the last 7 days' high prices
    median_last_7_high = df['high'].rolling(window=7).median()
    # Calculate the median of the next 7 days' volume
    median_next_7_volume = df['volume'].shift(-7).rolling(window=7).median()
    # Calculate the factor as the ratio of the two medians
    factor_values = median_last_7_high / median_next_7_volume
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
