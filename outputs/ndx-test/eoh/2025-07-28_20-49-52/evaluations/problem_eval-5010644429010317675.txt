def heuristics_v2(df):
    sum_last_5_close = df['close'].rolling(window=5).sum()
    sum_next_5_high = df['high'].shift(-5).rolling(window=5).sum()
    factor_values = sum_last_5_close / sum_next_5_high
    smoothed_factor = factor_values.rolling(window=5).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
