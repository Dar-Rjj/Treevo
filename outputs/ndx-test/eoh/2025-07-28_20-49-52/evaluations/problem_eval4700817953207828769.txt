import pandas as pd

def heuristics_v2(df):
    # Calculate the daily trading range
    df['range'] = df['high'] - df['low']
    # Calculate the Exponential Moving Average of the daily range
    df['EMA_14'] = df['range'].ewm(span=14, adjust=False).mean()
    # Calculate the 20-day standard deviation of the EMA
    heuristics_matrix = df['EMA_14'].rolling(window=20).std().dropna()
    return heuristics_matrix
