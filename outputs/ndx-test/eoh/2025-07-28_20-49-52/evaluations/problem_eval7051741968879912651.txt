def heuristics_v2(df):
    # Calculate the Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    # Calculate the median of the open, high, low, and close prices
    median_price = df[['open', 'high', 'low', 'close']].median(axis=1)
    # Calculate the factor as the ratio of the median price to VWAP
    factor_values = median_price / vwap
    # Apply a weighted moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10, win_type='gaussian').mean(std=2)
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
