def heuristics_v2(df):
    # Calculate the average of the last 10 days' low prices
    avg_last_10_low = df['low'].rolling(window=10).mean()
    # Calculate the average of the next 10 days' volume
    avg_next_10_volume = df['volume'].shift(-10).rolling(window=10).mean()
    # Calculate the factor as the ratio of the two averages
    factor_values = avg_last_10_low / avg_next_10_volume
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
