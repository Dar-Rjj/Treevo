def heuristics_v2(df):
    # Calculate the difference between the highest and lowest prices over the last 14 days
    price_range = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    # Calculate the average of the closing price over the last 14 days
    avg_close = df['close'].rolling(window=14).mean()
    # Calculate the factor as the ratio of the price range to the average closing price
    factor_values = price_range / avg_close
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=7).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
