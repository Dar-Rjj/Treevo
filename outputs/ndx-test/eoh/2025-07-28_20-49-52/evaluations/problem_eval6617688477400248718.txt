import pandas as pd

def heuristics_v2(df):
    window = 20  # Define the lookback window for moving average and std dev
    df['Moving_Avg'] = df['close'].rolling(window=window).mean()
    df['Std_Dev'] = df['close'].pct_change().rolling(window=window).std()
    heuristics_matrix = (df['close'] - df['Moving_Avg']) / df['Std_Dev']
    return heuristics_matrix
