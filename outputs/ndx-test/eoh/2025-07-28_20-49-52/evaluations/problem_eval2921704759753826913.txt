def heuristics_v2(df):
    def sma(series, window):
        return series.rolling(window=window).mean()

    df['PriceChange'] = df['close'] - df['open']
    df['StdDev_PriceChange'] = df['PriceChange'].rolling(window=10).std()
    df['SMA_10_Closing'] = sma(df['close'], 10)
    heuristics_matrix = df['StdDev_PriceChange'] / df['SMA_10_Closing']
    return heuristics_matrix
