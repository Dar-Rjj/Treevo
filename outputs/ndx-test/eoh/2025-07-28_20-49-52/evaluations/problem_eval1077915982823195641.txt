def heuristics_v2(df):
    # Calculate the last 10 days' VWAP
    vwap_last_10 = ((df['close'] + df['open']) / 2 * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    # Calculate the next 10 days' VWAP
    vwap_next_10 = ((df['close'].shift(-10) + df['open'].shift(-10)) / 2 * df['volume'].shift(-10)).rolling(window=10).sum() / df['volume'].shift(-10).rolling(window=10).sum()
    # Calculate the factor as the ratio of the two VWAPs
    factor_values = vwap_last_10 / vwap_next_10
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
