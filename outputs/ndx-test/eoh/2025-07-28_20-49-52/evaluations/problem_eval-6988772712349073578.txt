def heuristics_v2(df):
    # Calculate the average of the last 20 days' closing prices
    avg_last_20_close = df['close'].rolling(window=20).mean()
    # Calculate the difference between the next 20 days' high and low prices
    diff_next_20_high_low = (df['high'].shift(-20) - df['low'].shift(-20)).rolling(window=20).sum()
    # Calculate the factor as the ratio of the two calculated values
    factor_values = avg_last_20_close / diff_next_20_high_low
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=7).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
