def heuristics_v2(df):
    # Calculate the 10-day simple moving average of the closing price
    sma_close_10 = df['close'].rolling(window=10).mean()

    # Calculate the 10-day simple moving average of the volume
    sma_volume_10 = df['volume'].rolling(window=10).mean()

    # Integrate the factors
    heuristics_matrix = (df['close'] - sma_close_10) * (df['volume'] / sma_volume_10)

    return heuristics_matrix
