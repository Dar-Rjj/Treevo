import pandas as pd

def heuristics_v2(df):
    # Calculate Average True Range (ATR) over 14 days
    tr = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    atr_14 = tr.rolling(window=14).mean()

    # Calculate 7-day Momentum Factor
    daily_returns = df['close'].pct_change()
    momentum_factor = daily_returns.rolling(window=7).mean()

    # Integrate the factors
    heuristics_matrix = (df['close'] / atr_14) + momentum_factor

    return heuristics_matrix
