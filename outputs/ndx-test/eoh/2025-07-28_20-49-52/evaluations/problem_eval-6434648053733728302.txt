def heuristics_v2(df):
    # Calculate the average of the last 5 days' high prices
    avg_last_5_high = df['high'].rolling(window=5).mean()
    # Calculate the average of the next 5 days' low prices
    avg_next_5_low = df['low'].shift(-5).rolling(window=5).mean()
    # Calculate the factor as the ratio of the two averages
    factor_values = avg_last_5_high / avg_next_5_low
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
