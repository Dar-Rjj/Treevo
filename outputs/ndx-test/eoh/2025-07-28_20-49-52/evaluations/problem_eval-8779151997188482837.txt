import pandas as pd

def heuristics_v2(df):
    # Calculate Average True Range (ATR) over 14 days
    tr = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    atr_14 = tr.rolling(window=14).mean()

    # Calculate Volatility-Adjusted Price Change
    price_change = df['close'] - df['close'].shift(1)
    volatility_adjusted_price_change = price_change / atr_14

    # Integrate the factors
    heuristics_matrix = (df['close'] / atr_14) + volatility_adjusted_price_change

    return heuristics_matrix
