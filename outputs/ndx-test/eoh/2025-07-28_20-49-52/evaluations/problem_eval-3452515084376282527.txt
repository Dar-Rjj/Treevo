def heuristics_v2(df):
    # Calculate short and long term average volumes
    short_avg_volume = df['volume'].rolling(window=10).mean()
    long_avg_volume = df['volume'].rolling(window=50).mean()
    
    # Calculate the ratio of short to long term average volumes
    volume_ratio = short_avg_volume / long_avg_volume
    
    # Calculate the rate of change in the closing price over a 10 day period
    close_price_change = df['close'].pct_change(periods=10)
    
    # Combine the two signals into a single heuristic
    heuristics_matrix = (volume_ratio + close_price_change) / 2
    
    return heuristics_matrix
