def heuristics_v2(df):
    # Calculate the Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    # Calculate the geometric mean of high, low, and close prices
    geom_mean_price = (df['high'] * df['low'] * df['close']) ** (1/3)
    # Calculate the factor as the ratio of VWAP to the geometric mean of high, low, and close prices
    factor_values = vwap / geom_mean_price
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
