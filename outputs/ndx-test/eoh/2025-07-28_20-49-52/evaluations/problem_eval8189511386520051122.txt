def heuristics_v2(df):
    # Calculate the average of the last 7 days' high prices
    avg_last_7_high = df['high'].rolling(window=7).mean()
    # Calculate the average of the next 7 days' volume
    avg_next_7_volume = df['volume'].shift(-7).rolling(window=7).mean()
    # Calculate the factor as the ratio of the two averages
    factor_values = avg_last_7_high / avg_next_7_volume
    # Apply a weighted moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10, win_type='gaussian').mean(std=2)
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
