def heuristics_v2(df):
    # Calculate the 20-day average volume
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the volume ratio
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_20']
    
    # Calculate the percentage change in closing price over 20 days
    df['Close_Change_20'] = df['close'].pct_change(periods=20)
    
    # Generate the final heuristic by multiplying the Volume_Ratio and Close_Change_20
    heuristics_matrix = df['Volume_Ratio'] * df['Close_Change_20']
    
    return heuristics_matrix
