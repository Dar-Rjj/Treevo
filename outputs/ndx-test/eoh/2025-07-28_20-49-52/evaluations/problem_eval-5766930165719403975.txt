def heuristics_v2(df):
    # Calculate the average return over the last 10 days
    avg_last_10_return = df['close'].pct_change().rolling(window=10).mean()
    # Calculate the standard deviation of daily returns over the next 5 days
    std_next_5_return = df['close'].pct_change().shift(-5).rolling(window=5).std()
    # Calculate the factor as the product of the two values
    factor_values = avg_last_10_return * std_next_5_return
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
