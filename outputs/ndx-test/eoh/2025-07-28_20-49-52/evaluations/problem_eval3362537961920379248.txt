import pandas as pd

def heuristics_v2(df):
    ema_21_close = df['close'].ewm(span=21, adjust=False).mean()
    ema_5_close = df['close'].ewm(span=5, adjust=False).mean()
    ma_21_volume = df['volume'].rolling(window=21).mean()
    heuristics_matrix = (ema_21_close - ema_5_close) / ma_21_volume
    return heuristics_matrix
