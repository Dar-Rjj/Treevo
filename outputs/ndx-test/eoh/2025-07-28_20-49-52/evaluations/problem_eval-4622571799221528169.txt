def heuristics_v2(df):
    # Calculate the Volume-Weighted Average Price (VWAP) of open and close prices
    vwap = (df['volume'] * (df['open'] + df['close']) / 2).cumsum() / df['volume'].cumsum()
    # Calculate the 20-day simple moving average of the closing price
    sma_close = df['close'].rolling(window=20, min_periods=1).mean()
    # Calculate the factor as the difference between VWAP and SMA
    factor_values = vwap - sma_close
    heuristics_matrix = factor_values.dropna()
    return heuristics_matrix
