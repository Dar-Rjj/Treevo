def heuristics_v2(df):
    # Calculate the sum of the last 7 days' closing prices
    sum_last_7_close = df['close'].rolling(window=7).sum()
    # Calculate the sum of the next 7 days' low prices
    sum_next_7_low = df['low'].shift(-7).rolling(window=7).sum()
    # Calculate the factor as the ratio of the two sums
    factor_values = sum_last_7_close / sum_next_7_low
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=14).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
