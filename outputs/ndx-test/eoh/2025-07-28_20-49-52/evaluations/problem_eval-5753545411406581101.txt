def heuristics_v2(df):
    # Calculate the daily price change
    price_change = df['close'].diff()
    # Calculate the EMA of the daily price change
    ema_price_change = price_change.ewm(span=5, adjust=False).mean()
    # Calculate the standard deviation of the EMA over the last 10 days
    std_ema = ema_price_change.rolling(window=10).std()
    heuristics_matrix = std_ema.dropna()
    return heuristics_matrix
