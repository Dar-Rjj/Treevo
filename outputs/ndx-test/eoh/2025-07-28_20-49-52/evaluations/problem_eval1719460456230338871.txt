import pandas as pd

def heuristics_v2(df):
    # Calculate short-term (10 days) and long-term (50 days) simple moving averages for 'close' price
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between the short-term and long-term SMAs
    df['SMA_Diff'] = df['SMA_10'] - df['SMA_50']
    
    # Calculate the difference between current 'close' price and its 10-day SMA
    df['Close_SMA_10_Diff'] = df['close'] - df['SMA_10']
    
    # Generate the final heuristic by combining the SMA_Diff and Close_SMA_10_Diff with weights
    heuristics_matrix = 0.7 * df['SMA_Diff'] + 0.3 * df['Close_SMA_10_Diff']
    
    return heuristics_matrix
