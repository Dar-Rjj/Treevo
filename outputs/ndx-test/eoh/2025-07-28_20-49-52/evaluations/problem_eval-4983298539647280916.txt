def heuristics_v2(df):
    # Calculate the product of the last 5 days' volume
    prod_last_5_volume = df['volume'].rolling(window=5).apply(lambda x: x.prod(), raw=True)
    # Calculate the sum of the next 5 days' low prices
    sum_next_5_low = df['low'].shift(-5).rolling(window=5).sum()
    # Calculate the factor as the ratio of the two calculated values
    factor_values = prod_last_5_volume / sum_next_5_low
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
