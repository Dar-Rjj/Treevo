def heuristics_v2(df):
    # Calculate the 10-period EMA of the close price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the 30-period EMA of the volume
    ema_volume = df['volume'].ewm(span=30, adjust=False).mean()
    # Calculate the factor as the ratio of EMA of the close price to the EMA of the volume
    factor_values = ema_close / ema_volume
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
