def heuristics_v2(df):
    # Calculate the ratio of the difference between high and low prices to the closing price
    factor_values = (df['high'] - df['low']) / df['close']
    # Apply a 10-period exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
