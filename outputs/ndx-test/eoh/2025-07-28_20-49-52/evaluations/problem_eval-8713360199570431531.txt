import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 12-day and 26-day Exponential Moving Averages (EMAs)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the daily return
    daily_return = df['close'].pct_change().fillna(0)
    
    # Generate the heuristic score as the ratio of EMAs scaled by the log of the daily return
    heuristics_matrix = (ema_12 / ema_26) * np.log(1 + daily_return)
    
    return heuristics_matrix
