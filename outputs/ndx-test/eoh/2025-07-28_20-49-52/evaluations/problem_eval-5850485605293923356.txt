def heuristics_v2(df):
    def sma(series, window):
        return series.rolling(window=window).mean()
    
    vol_adj_close = df['close'] * df['volume']
    SMA_10_vol_adj_close = sma(vol_adj_close, 10)
    SMA_5_high_low_range = sma(df['high'] - df['low'], 5)
    heuristics_matrix = SMA_10_vol_adj_close / SMA_5_high_low_range
    return heuristics_matrix
