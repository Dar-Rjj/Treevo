def heuristics_v2(df):
    # Calculate the 5-day and 20-day simple moving averages of the closing price
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    # Calculate the difference between the 5-day and 20-day SMAs
    diff_sma = sma_5 - sma_20
    # Calculate the 60-day standard deviation of the closing price
    std_60 = df['close'].rolling(window=60).std()
    # Form the volatility-adjusted momentum factor
    factor_values = diff_sma / std_60
    heuristics_matrix = factor_values.dropna()
    return heuristics_matrix
