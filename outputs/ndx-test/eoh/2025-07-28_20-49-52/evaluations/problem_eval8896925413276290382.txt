def heuristics_v2(df):
    # Calculate the 20-day average range
    df['Avg_Range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Calculate the current day's range
    df['Day_Range'] = df['high'] - df['low']
    
    # Calculate the range ratio
    df['Range_Ratio'] = df['Day_Range'] / df['Avg_Range_20']
    
    # Calculate the percentage change in closing price over 20 days
    df['Close_Change_20'] = df['close'].pct_change(periods=20)
    
    # Calculate the 20-day SMA of the close change
    df['SMA_Close_Change_20'] = df['Close_Change_20'].rolling(window=20).mean()
    
    # Generate the final heuristic by multiplying the Range_Ratio and SMA_Close_Change_20
    heuristics_matrix = df['Range_Ratio'] * df['SMA_Close_Change_20']
    
    return heuristics_matrix
