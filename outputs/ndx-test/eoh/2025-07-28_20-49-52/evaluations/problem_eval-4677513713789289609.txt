def heuristics_v2(df):
    # Calculate Rate of Change (ROC) for a 10-day period
    roc_period = 10
    df['roc'] = df['close'].pct_change(periods=roc_period).fillna(0)
    
    # Calculate Relative Strength Index (RSI) for a 14-day period
    rsi_period = 14
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_period).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_period).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Generate signal based on ROC and RSI
    df['signal'] = 0
    df.loc[(df['roc'] > 0) & (df['rsi'] < 70), 'signal'] = 1
    df.loc[(df['roc'] < 0) & (df['rsi'] > 30), 'signal'] = -1
    
    heuristics_matrix = df['signal']
    
    return heuristics_matrix
