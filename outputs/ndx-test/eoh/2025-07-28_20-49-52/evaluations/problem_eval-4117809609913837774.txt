def heuristics_v2(df):
    # Calculate the average of the last 15 days' closing prices
    avg_last_15_close = df['close'].rolling(window=15).mean()
    # Calculate the sum of the next 15 days' low and high prices
    sum_next_15_low_high = (df['low'].shift(-15) + df['high'].shift(-15)).rolling(window=15).sum()
    # Calculate the factor as the ratio of the two calculated values
    factor_values = avg_last_15_close / sum_next_15_low_high
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
