def heuristics_v2(df):
    # Calculate the Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    # Calculate the median price over a lookback period
    median_price = df[['high', 'low', 'close']].median(axis=1).rolling(window=10).mean()
    # Calculate the factor as the ratio of VWAP to median price
    factor_values = vwap / median_price
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
