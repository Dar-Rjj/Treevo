def heuristics_v2(df):
    # Calculate the sum of the last 3 days' closing prices
    sum_last_3_close = df['close'].rolling(window=3).sum()
    # Calculate the sum of the next 3 days' low prices
    sum_next_3_low = df['low'].shift(-3).rolling(window=3).sum()
    # Calculate the factor as the ratio of the two sums
    factor_values = sum_last_3_close / sum_next_3_low
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
