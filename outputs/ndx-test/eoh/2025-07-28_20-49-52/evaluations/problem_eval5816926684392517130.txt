def heuristics_v2(df):
    # Calculate the ratio of closing price to volume
    price_volume_ratio = df['close'] / df['volume']
    
    # Calculate the 10-day simple moving average of the price-volume ratio
    smoothed_ratio = price_volume_ratio.rolling(window=10).mean()
    
    # Calculate the 5-day percentage change in the smoothed ratio to get the momentum
    momentum = smoothed_ratio.pct_change(periods=5)
    
    # Apply a 10-day simple moving average for further smoothing
    heuristics_matrix = momentum.rolling(window=10).mean().dropna()

    return heuristics_matrix
