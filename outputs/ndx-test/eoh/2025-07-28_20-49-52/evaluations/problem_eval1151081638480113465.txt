import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) over a 10-day period
    roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate the Average True Range (ATR) over a 14-day period
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    
    # Combine factors into a weighted sum for the final heuristic
    heuristics_matrix = roc + (atr / df['close'])
    
    return heuristics_matrix
