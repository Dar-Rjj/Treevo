def heuristics_v2(df):
    avg_last_10_close = df['close'].rolling(window=10).mean()
    sum_next_5_high_low = (df['high'].shift(-1) + df['low'].shift(-1)).rolling(window=5).sum()
    factor_values = avg_last_10_close / sum_next_5_high_low
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
