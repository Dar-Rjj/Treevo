def heuristics_v2(df):
    # Calculate the sum of the last 10 days' high prices
    sum_last_10_high = df['high'].rolling(window=10).sum()
    # Calculate the sum of the next 10 days' high prices
    sum_next_10_high = df['high'].shift(-10).rolling(window=10).sum()
    # Calculate the factor as the ratio of the two sums
    factor_values = sum_last_10_high / sum_next_10_high
    # Apply a weighted moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10, win_type='gaussian').mean(std=2)
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
