def heuristics_v2(df):
    # Calculate the EMA of the last 5 days' high prices
    ema_last_5_high = df['high'].ewm(span=5, adjust=False).mean()
    # Calculate the EMA of the next 5 days' low prices
    ema_next_5_low = df['low'].shift(-5).ewm(span=5, adjust=False).mean()
    # Calculate the factor as the ratio of the two EMAs
    factor_values = ema_last_5_high / ema_next_5_low
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
