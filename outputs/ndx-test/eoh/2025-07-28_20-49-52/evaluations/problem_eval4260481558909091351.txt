def heuristics_v2(df):
    # Calculate the sum of the absolute differences between high and low prices
    abs_diff_sum = (df['high'] - df['low']).abs().cumsum()
    # Calculate the factor as the ratio of the sum of absolute differences to the total volume
    factor_values = abs_diff_sum / df['volume'].cumsum()
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
