def heuristics_v2(df):
    # Calculate the weighted difference between high and low prices
    weighted_diff = (df['volume'] * (df['high'] - df['low']))
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = weighted_diff.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
