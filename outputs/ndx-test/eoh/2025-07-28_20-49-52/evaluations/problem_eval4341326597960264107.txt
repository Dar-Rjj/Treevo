def heuristics_v2(df):
    # Calculate the cumulative volume over the last 10 days
    cum_volume = df['volume'].rolling(window=10).sum()
    # Calculate the standard deviation of closing prices over the next 10 days
    std_next_10_close = df['close'].shift(-10).rolling(window=10).std()
    # Calculate the factor as the ratio of the two measures
    factor_values = cum_volume / std_next_10_close
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=5, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
