def heuristics_v2(df):
    # Calculate the modified Volume-Weighted Average Price (VWAP) using high and low prices
    vwap = (df['volume'] * (df['high'] + df['low']) / 2).cumsum() / df['volume'].cumsum()
    # Calculate the factor as the ratio of modified VWAP to close price
    factor_values = vwap / df['close']
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
