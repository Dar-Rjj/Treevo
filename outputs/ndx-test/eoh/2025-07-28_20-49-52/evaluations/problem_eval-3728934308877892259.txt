def heuristics_v2(df):
    # Calculate the sum of the last 5 days' high prices
    sum_last_5_high = df['high'].rolling(window=5).sum()
    # Calculate the sum of the next 15 days' opening prices
    sum_next_15_open = df['open'].shift(-15).rolling(window=15).sum()
    # Calculate the factor as the ratio of the two sums
    factor_values = sum_last_5_high / sum_next_15_open
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
