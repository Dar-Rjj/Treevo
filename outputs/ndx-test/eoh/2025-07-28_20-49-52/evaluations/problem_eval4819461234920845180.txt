def heuristics_v2(df):
    # Calculate the 5-day simple moving average of the closing prices
    sma_5_close = df['close'].rolling(window=5).mean()
    # Calculate the 10-day simple moving average of the opening prices
    sma_10_open = df['open'].rolling(window=10).mean()
    # Calculate the factor as the difference between the two SMAs
    factor_values = sma_5_close - sma_10_open
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
