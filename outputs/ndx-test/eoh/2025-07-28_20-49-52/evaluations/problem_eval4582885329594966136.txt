def heuristics_v2(df):
    df['close_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()
    df['high_low_diff'] = (df['high'] - df['low']) / df['close']
    weights = [0.5, 0.3, 0.2]  # Example weights, can be adjusted based on further analysis
    df['heuristic_value'] = df['close_change'] * weights[0] + df['volume_change'] * weights[1] + df['high_low_diff'] * weights[2]
    heuristics_matrix = df['heuristic_value'].dropna()
    return heuristics_matrix
