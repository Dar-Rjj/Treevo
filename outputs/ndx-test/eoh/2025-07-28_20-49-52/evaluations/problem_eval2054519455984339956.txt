def heuristics_v2(df):
    avg_last_5_close = df['close'].rolling(window=5).mean()
    avg_next_5_low = df['low'].shift(-5).rolling(window=5).mean()
    factor_values = avg_last_5_close / avg_next_5_low
    smoothed_factor = factor_values.ewm(span=5, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
