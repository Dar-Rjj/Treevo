def heuristics_v2(df):
    # Calculate the standard deviation of the last 10 days' closing prices
    std_last_10_close = df['close'].rolling(window=10).std()
    # Calculate the standard deviation of the next 10 days' opening prices
    std_next_10_open = df['open'].shift(-10).rolling(window=10).std()
    # Calculate the factor as the ratio of the two standard deviations
    factor_values = std_last_10_close / std_next_10_open
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
