def heuristics_v2(df):
    # Calculate the Volume-Weighted Average Price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    # Calculate the Exponential Moving Average (EMA) of VWAP and close price with a span of 10
    ema_vwap = vwap.ewm(span=10, adjust=False).mean()
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the factor as the ratio of EMA of VWAP to EMA of close price
    factor_values = ema_vwap / ema_close
    # Apply a simple moving average for further smoothing the factor values
    smoothed_factor = factor_values.rolling(window=5).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
