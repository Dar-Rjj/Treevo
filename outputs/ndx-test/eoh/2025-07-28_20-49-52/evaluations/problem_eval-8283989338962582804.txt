import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day weighted moving average (WMA) of the closing price
    weights = pd.Series(range(1, 21))
    df['WMA_20'] = df['close'].rolling(window=20).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)

    # Adjust the WMA with a volume-based factor
    df['volume_factor'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['adjusted_WMA_20'] = df['WMA_20'] * df['volume_factor']

    # Generate the heuristic matrix
    heuristics_matrix = df['adjusted_WMA_20']
    
    return heuristics_matrix
