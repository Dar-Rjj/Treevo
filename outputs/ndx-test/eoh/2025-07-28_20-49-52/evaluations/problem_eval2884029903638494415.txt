def heuristics_v2(df):
    # Calculate the price range (high - low)
    df['Price_Range'] = df['high'] - df['low']
    # Calculate the EMA of the price range with a span of 10 days
    df['EMA_Price_Range'] = df['Price_Range'].ewm(span=10, adjust=False).mean()
    # Calculate the EMA of the volume with a span of 10 days
    df['EMA_Volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    # Generate the final heuristic by adjusting the EMA of the price range by the EMA of volume
    heuristics_matrix = df['EMA_Price_Range'] / df['EMA_Volume']
    return heuristics_matrix
