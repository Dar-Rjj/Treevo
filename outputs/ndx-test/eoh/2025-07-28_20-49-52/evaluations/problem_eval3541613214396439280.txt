def heuristics_v2(df):
    # Calculate the 10 days' VWAP
    vwap_10 = ((df['close'] + df['open']) / 2 * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    # Calculate the 30 days' VWAP
    vwap_30 = ((df['close'] + df['open']) / 2 * df['volume']).rolling(window=30).sum() / df['volume'].rolling(window=30).sum()
    # Calculate the factor as the ratio of the two VWAPs
    factor_values = vwap_10 / vwap_30
    # Apply a 5-day EMA for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=5, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
