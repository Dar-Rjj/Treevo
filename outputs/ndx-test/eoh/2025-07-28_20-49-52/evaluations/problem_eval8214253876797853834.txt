def heuristics_v2(df):
    avg_last_10_close = df['close'].rolling(window=10).mean()
    std_next_10_high = df['high'].shift(-10).rolling(window=10).std()
    factor_values = avg_last_10_close / std_next_10_high
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
