import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day momentum of the closing price
    momentum_20 = df['close'].pct_change(20)
    # Calculate the 10-day Exponential Moving Average (EMA) of the volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    # Create the factor by subtracting the EMA of the volume from the 20-day momentum
    factor_values = momentum_20 - ema_volume
    # Drop any NaN values that may have been introduced by the calculations
    heuristics_matrix = factor_values.dropna()
    return heuristics_matrix
