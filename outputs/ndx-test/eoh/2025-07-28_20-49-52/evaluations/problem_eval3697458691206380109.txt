import pandas as pd

def heuristics_v2(df):
    heuristics_matrix = pd.Series(index=df.index)
    
    # Simple Moving Averages
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Volume-Weighted Price Change
    vw_price_change = (df['close'] - df['open']) * df['volume']
    
    # Price Momentum
    momentum = df['close'] / df['close'].shift(10) - 1
    
    # Combine heuristics into a single factor
    heuristics_matrix = (sma_5 - sma_20) + vw_price_change + momentum
    
    return heuristics_matrix
