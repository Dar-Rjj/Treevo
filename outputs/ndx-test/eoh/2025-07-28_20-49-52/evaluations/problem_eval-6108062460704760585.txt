def heuristics_v2(df):
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    # Normalize the price difference by volume
    norm_price_diff = price_diff / df['volume']
    # Apply a weighted moving average for smoothing the factor values
    smoothed_factor = norm_price_diff.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
