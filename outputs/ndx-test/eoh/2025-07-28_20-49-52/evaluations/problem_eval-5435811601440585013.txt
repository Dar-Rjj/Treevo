def heuristics_v2(df):
    # Calculate the percentage change in closing price over the last 20 days
    pct_change_close = df['close'].pct_change(periods=20)
    # Calculate the weighted sum using volume as weights
    weighted_sum = (df['volume'] * pct_change_close).rolling(window=20).sum()
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = weighted_sum.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
