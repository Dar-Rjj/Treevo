def heuristics_v2(df):
    # Calculate the last 5 days' VWAP
    vwap_last_5 = ((df['close'] + df['open']) / 2 * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    # Calculate the next 15 days' VWAP
    vwap_next_15 = ((df['close'].shift(-15) + df['open'].shift(-15)) / 2 * df['volume'].shift(-15)).rolling(window=15).sum() / df['volume'].shift(-15).rolling(window=15).sum()
    # Calculate the factor as the ratio of the two VWAPs
    factor_values = vwap_last_5 / vwap_next_15
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
