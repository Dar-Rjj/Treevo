def heuristics_v2(df):
    # Calculate the difference between max and min of the last 10 days' closing prices
    diff_close = df['close'].rolling(window=10).max() - df['close'].rolling(window=10).min()
    # Calculate the average of the last 10 days' volume
    avg_last_10_volume = df['volume'].rolling(window=10).mean()
    # Compute the factor as the ratio of the difference in closing prices to the average volume
    factor_values = diff_close / avg_last_10_volume
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
