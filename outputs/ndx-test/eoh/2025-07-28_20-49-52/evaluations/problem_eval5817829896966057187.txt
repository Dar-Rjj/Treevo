def heuristics_v2(df):
    # Calculate the Exponential Moving Average (EMA) of the close price with a span of 10 days
    ema = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the factor as the percentage change between consecutive EMA values
    factor_values = ema.pct_change()
    heuristics_matrix = factor_values.dropna()
    return heuristics_matrix
