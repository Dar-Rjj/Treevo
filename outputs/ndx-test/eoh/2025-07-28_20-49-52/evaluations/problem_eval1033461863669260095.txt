def heuristics_v2(df):
    # Calculate the maximum of the last 3 days' low prices
    max_last_3_low = df['low'].rolling(window=3).max()
    # Calculate the minimum of the next 3 days' high prices
    min_next_3_high = df['high'].shift(-3).rolling(window=3).min()
    # Calculate the factor as the difference between the two
    factor_values = max_last_3_low - min_next_3_high
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=7).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
