def heuristics_v2(df):
    # Calculate the difference between the maximum and minimum high prices over the last 10 days
    diff_last_10_high = df['high'].rolling(window=10).max() - df['high'].rolling(window=10).min()
    # Calculate the average of the next 10 days' high prices
    avg_next_10_high = df['high'].shift(-10).rolling(window=10).mean()
    # Calculate the factor as the ratio of the two values
    factor_values = diff_last_10_high / avg_next_10_high
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
