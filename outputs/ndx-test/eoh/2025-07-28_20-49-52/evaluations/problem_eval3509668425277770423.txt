import pandas as pd

def heuristics_v2(df):
    # Calculate Average True Range (ATR) over 20 days
    tr = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    atr_20 = tr.rolling(window=20).mean()

    # Calculate Volatility-Adjusted Price Change
    price_change = df['close'] - df['close'].shift(1)
    volatility_adjusted_price_change = price_change / atr_20

    # Integrate the factors
    heuristics_matrix = (df['close'] / atr_20) + volatility_adjusted_price_change

    return heuristics_matrix
