def heuristics_v2(df):
    # Calculate the highest high and the lowest low for the last 20 days
    high_last_20 = df['high'].rolling(window=20).max()
    low_last_20 = df['low'].rolling(window=20).min()
    # Calculate the difference between the highest high and the lowest low
    diff_high_low = high_last_20 - low_last_20
    # Calculate the average volume for the last 20 days
    avg_volume = df['volume'].rolling(window=20).mean()
    # Compute the factor as the ratio of the difference to the average volume
    factor_values = diff_high_low / avg_volume
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=20, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
