def heuristics_v2(df):
    # Calculate the last 10 days' VWAP
    vwap_last_10 = ((df['close'] + df['open']) / 2 * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    # Calculate the next 5 days' VWAP
    vwap_next_5 = ((df['close'].shift(-5) + df['open'].shift(-5)) / 2 * df['volume'].shift(-5)).rolling(window=5).sum() / df['volume'].shift(-5).rolling(window=5).sum()
    # Calculate the factor as the ratio of the two VWAPs
    factor_values = vwap_last_10 / vwap_next_5
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=5).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
