def heuristics_v2(df):
    # Calculate the 20-day percentage change in the closing price
    momentum = df['close'].pct_change(periods=20)
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = momentum.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
