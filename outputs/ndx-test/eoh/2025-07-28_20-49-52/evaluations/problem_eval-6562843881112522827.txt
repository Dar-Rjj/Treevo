def heuristics_v2(df):
    # Calculate the average of the last 5 days' closing prices
    avg_last_5_close = df['close'].rolling(window=5).mean()
    # Calculate the average of the next 5 days' opening prices
    avg_next_5_open = df['open'].shift(-5).rolling(window=5).mean()
    # Calculate the factor as the ratio of the two averages
    factor_values = avg_last_5_close / avg_next_5_open
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=5).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
