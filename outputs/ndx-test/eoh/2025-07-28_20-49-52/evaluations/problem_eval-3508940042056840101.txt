def heuristics_v2(df):
    # Calculate the difference between the highest high and lowest low over the last 10 days
    diff_high_low = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    # Calculate the average of the next 5 days' closing prices
    avg_next_5_close = df['close'].shift(-5).rolling(window=5).mean()
    # Calculate the factor as the ratio of the difference to the average of the next 5 days' closing prices
    factor_values = diff_high_low / avg_next_5_close
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
