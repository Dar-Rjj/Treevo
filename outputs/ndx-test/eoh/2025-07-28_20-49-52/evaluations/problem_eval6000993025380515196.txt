def heuristics_v2(df):
    # Calculate the 10-day and 30-day moving averages
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=30).mean()
    
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 20-day average volume
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the volume ratio
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_20']
    
    # Calculate the percentage change in closing price over 10 days
    df['Close_Change_10'] = df['close'].pct_change(periods=10)
    
    # Generate the final heuristic by combining the sentiment score and momentum
    sentiment_score = (sma_short - sma_long) + rsi
    heuristics_matrix = sentiment_score * (df['Volume_Ratio'] * df['Close_Change_10'])
    
    return heuristics_matrix
