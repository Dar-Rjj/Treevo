def heuristics_v2(df):
    # Calculate the 10-day Rate of Change (ROC)
    df['ROC_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the 30-day average volume
    df['Avg_Volume_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of today's volume to the 30-day average volume
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_30']
    
    # Calculate the 20-day Moving Average of the closing price
    df['MA_20'] = df['close'].rolling(window=20).mean()
    
    # Generate the final heuristic by combining ROC_10, Volume_Ratio, and MA_20 with weights
    heuristics_matrix = 0.4 * df['ROC_10'] + 0.3 * df['Volume_Ratio'] + 0.3 * (df['close'] - df['MA_20'])
    
    return heuristics_matrix
