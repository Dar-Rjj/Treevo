def heuristics_v2(df):
    # Calculate the Exponential Moving Average (EMA) of the high and low prices
    ema_high = df['high'].ewm(span=10, adjust=False).mean()
    ema_low = df['low'].ewm(span=10, adjust=False).mean()
    ema_avg = (ema_high + ema_low) / 2
    # Calculate the factor as the ratio of close price to the EMA
    factor_values = df['close'] / ema_avg
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=5, min_periods=1).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
