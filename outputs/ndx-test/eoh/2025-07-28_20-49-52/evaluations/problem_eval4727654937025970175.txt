def heuristics_v2(df):
    # Calculate the sum of the last 5 days' closing prices
    sum_last_5_close = df['close'].rolling(window=5).sum()
    # Calculate the product of the next 5 days' high and low prices
    product_next_5_high_low = (df['high'] * df['low']).shift(-5).rolling(window=5).mean()
    # Calculate the factor as the ratio of the two calculations
    factor_values = sum_last_5_close / product_next_5_high_low
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
