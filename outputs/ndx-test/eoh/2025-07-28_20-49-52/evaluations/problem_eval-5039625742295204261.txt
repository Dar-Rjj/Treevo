import pandas as pd

def heuristics_v2(df):
    # Calculate Average True Range (ATR) over 14 days with EMA
    tr = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    atr_14 = tr.ewm(span=14, adjust=False).mean()

    # Calculate Volatility-Adjusted Price Change with EMA
    price_change = df['close'] - df['close'].shift(1)
    volatility_adjusted_price_change = price_change / atr_14

    # Smooth the volatility-adjusted price change using EMA
    smoothed_volatility_adjusted_price_change = volatility_adjusted_price_change.ewm(span=14, adjust=False).mean()

    # Integrate the factors
    heuristics_matrix = (df['close'] / atr_14) + smoothed_volatility_adjusted_price_change

    return heuristics_matrix
