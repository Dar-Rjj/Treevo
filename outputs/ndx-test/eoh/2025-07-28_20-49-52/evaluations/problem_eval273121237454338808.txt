def heuristics_v2(df):
    # Calculate the maximum of the last 7 days' closing prices
    max_last_7_close = df['close'].rolling(window=7).max()
    # Calculate the minimum of the next 7 days' low prices
    min_next_7_low = df['low'].shift(-7).rolling(window=7).min()
    # Calculate the factor as the difference between the two values
    factor_values = max_last_7_close - min_next_7_low
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
