def heuristics_v2(df):
    # Calculate the 20-day momentum
    df['Momentum_20'] = df['close'].pct_change(periods=20)
    
    # Calculate the 20-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    df['RSI_20'] = 100 - (100 / (1 + rs))
    
    # Adjust the momentum by the RSI
    heuristics_matrix = df['Momentum_20'] * df['RSI_20']
    
    return heuristics_matrix
