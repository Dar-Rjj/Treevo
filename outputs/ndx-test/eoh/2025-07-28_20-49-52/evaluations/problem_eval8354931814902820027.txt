def heuristics_v2(df):
    # Calculate the sum of high and low prices
    sum_high_low = df['high'] + df['low']
    # Calculate the factor as the ratio of the sum of high and low prices to the close price
    factor_values = sum_high_low / df['close']
    # Apply an exponential moving average for smoothing the factor values
    smoothed_factor = factor_values.ewm(span=10, adjust=False).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
