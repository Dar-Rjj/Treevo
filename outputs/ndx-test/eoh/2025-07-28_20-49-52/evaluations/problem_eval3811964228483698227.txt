def heuristics_v2(df):
    # Calculate the average of the last 5 days' high prices
    avg_last_5_high = df['high'].rolling(window=5).mean()
    # Calculate the average of the next 5 days' close prices
    avg_next_5_close = df['close'].shift(-5).rolling(window=5).mean()
    # Calculate the factor as the difference between the two averages
    factor_values = avg_last_5_high - avg_next_5_close
    # Apply a weighted moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10, win_type='gaussian').mean(std=2)
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
