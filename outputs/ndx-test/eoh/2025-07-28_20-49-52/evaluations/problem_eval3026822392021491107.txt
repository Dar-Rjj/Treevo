def heuristics_v2(df):
    # Calculate the median of the last 15 days' volume
    median_last_15_volume = df['volume'].rolling(window=15).median()
    # Calculate the maximum of the next 15 days' low prices
    max_next_15_low = df['low'].shift(-15).rolling(window=15).max()
    # Calculate the factor as the ratio of the two values
    factor_values = median_last_15_volume / max_next_15_low
    # Apply a simple moving average for smoothing the factor values
    smoothed_factor = factor_values.rolling(window=10).mean()
    heuristics_matrix = smoothed_factor.dropna()
    return heuristics_matrix
