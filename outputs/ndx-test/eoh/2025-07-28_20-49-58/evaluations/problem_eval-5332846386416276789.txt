def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate 5-day moving average of volume
    volume_ma5 = df['volume'].rolling(window=5).mean()
    
    # Calculate 20-day simple moving average of the closing price
    sma20 = df['close'].rolling(window=20).mean()
    
    # Combine factors
    heuristics_matrix = roc + (volume_ma5 - sma20)
    
    return heuristics_matrix
