def heuristics_v2(df):
    # Calculate the volume-to-price ratio
    volume_to_price = df['volume'] / df['close']
    # Calculate the 10-day and 30-day simple moving averages of the volume-to-price ratio
    sma_10 = volume_to_price.rolling(window=10).mean()
    sma_30 = volume_to_price.rolling(window=30).mean()
    # Generate the SMA crossover signal; positive if 10-day SMA is above 30-day SMA, negative otherwise
    sma_crossover_signal = (sma_10 > sma_30).astype(int) * 2 - 1
    # Combine the volume-to-price ratio with the SMA crossover signal into a single heuristic
    heuristics_matrix = 0.5 * (volume_to_price + sma_crossover_signal)
    return heuristics_matrix
