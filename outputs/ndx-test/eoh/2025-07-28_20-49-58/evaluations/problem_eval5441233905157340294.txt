import pandas as pd

def heuristics_v2(df):
    def calculate_rsi(column, n):
        delta = column.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    # Calculate RSI over a 14-day period
    rsi = calculate_rsi(df['close'], 14)
    
    # Calculate 20-day SMA of volume
    volume_sma20 = df['volume'].rolling(window=20).mean()
    
    # Combine factors
    heuristics_matrix = rsi + (volume_sma20 / df['volume'])
    
    return heuristics_matrix
