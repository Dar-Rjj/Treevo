def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 15-day period
    roc = calculate_roc(df['close'], 15)
    
    # Calculate 7-day moving average of the closing price
    ma_7 = df['close'].rolling(window=7).mean()
    
    # Calculate VWAP with 7-day moving average
    tp = (df['high'] + df['low'] + ma_7) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Capture the 10-day rolling median of the closing prices
    med_10 = df['close'].rolling(window=10).median()
    
    # Adjust the heuristics matrix with momentum and rolling median
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=5).mean() - med_10
    
    return heuristics_matrix
