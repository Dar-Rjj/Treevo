def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 15-day period
    roc = calculate_roc(df['close'], 15)
    
    # Calculate 30-day EWMA
    ewma = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate 20-day SMA
    sma = df['close'].rolling(window=20).mean()
    
    # Combine factors
    heuristics_matrix = roc + (ewma - sma) / sma
    
    return heuristics_matrix
