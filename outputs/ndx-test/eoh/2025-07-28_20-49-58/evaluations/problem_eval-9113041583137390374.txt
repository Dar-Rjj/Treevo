def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate 10-day SMA of the closing price
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate VWAP with 10-day SMA
    tp = (df['high'] + df['low'] + sma_10) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 10-day standard deviation of the closing price as a measure of volatility
    std_dev_10 = df['close'].rolling(window=10).std()
    
    # Adjust the heuristics matrix with momentum and volatility
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=10).mean() - std_dev_10
    
    return heuristics_matrix
