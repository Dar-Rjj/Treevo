def heuristics_v2(df):
    # Calculate 10-day EMA
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate 30-day EMA
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the alpha factor as the difference between the 10-day and 30-day EMAs
    heuristics_matrix = (ema_10 - ema_30).dropna()
    
    return heuristics_matrix
