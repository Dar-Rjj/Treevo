import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day EMA of the closing price
    ema20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 5-day moving average of VWAP
    vwap_ma5 = vwap.rolling(window=5).mean()
    
    # Calculate 10-day simple moving average of the closing price
    sma10 = df['close'].rolling(window=10).mean()
    
    # Combine factors
    heuristics_matrix = ema20 + (vwap_ma5 / sma10)
    
    return heuristics_matrix
