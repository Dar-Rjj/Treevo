import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 20-day moving average of the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate VWAP with 20-day moving average
    tp = (df['high'] + df['low'] + ma_20) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Combine factors
    heuristics_matrix = roc + (vwap - df['close'])
    
    return heuristics_matrix
