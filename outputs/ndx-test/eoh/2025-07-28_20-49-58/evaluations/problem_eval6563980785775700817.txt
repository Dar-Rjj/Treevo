import pandas as pd

def heuristics_v2(df):
    def calculate_sma(column, n):
        return column.rolling(window=n).mean()

    def calculate_rsi(column, n):
        delta = column.diff()
        gain = delta.where(delta > 0, 0)
        loss = -delta.where(delta < 0, 0)
        avg_gain = gain.rolling(window=n).mean()
        avg_loss = loss.rolling(window=n).mean()
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    # Calculate 20-day SMA of the closing price
    sma_20 = calculate_sma(df['close'], 20)
    
    # Calculate 14-day RSI of the closing price
    rsi_14 = calculate_rsi(df['close'], 14)
    
    # Calculate 50-day and 200-day SMA of the closing price
    sma_50 = calculate_sma(df['close'], 50)
    sma_200 = calculate_sma(df['close'], 200)
    
    # Combine factors
    heuristics_matrix = sma_20 + (sma_50 - sma_200) + rsi_14
    
    return heuristics_matrix
