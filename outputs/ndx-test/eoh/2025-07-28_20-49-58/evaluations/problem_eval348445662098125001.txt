def heuristics_v2(df):
    def calculate_ema(column, span):
        return column.ewm(span=span, adjust=False).mean()

    def calculate_rsi(column, n=14):
        delta = column.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def calculate_mfi(high, low, close, volume, n=14):
        typical_price = (high + low + close) / 3
        raw_money_flow = typical_price * volume
        positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
        negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
        money_ratio = positive_money_flow.rolling(window=n).sum() / negative_money_flow.rolling(window=n).sum()
        return 100 - (100 / (1 + money_ratio))

    # Calculate EMA over a 14-day period
    ema = calculate_ema(df['close'], 14)
    
    # Calculate RSI over a 14-day period
    rsi = calculate_rsi(df['close'])
    
    # Calculate MFI over a 14-day period
    mfi = calculate_mfi(df['high'], df['low'], df['close'], df['volume'])
    
    # Combine factors
    heuristics_matrix = ema + rsi + mfi
    
    return heuristics_matrix
