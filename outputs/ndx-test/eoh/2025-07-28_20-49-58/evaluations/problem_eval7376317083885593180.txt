def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) for closing prices
    roc_10 = df['close'].pct_change(10)
    
    # Calculate the 5-day average of the trading volume
    avg_volume_5 = df['volume'].rolling(window=5, min_periods=5).mean()
    
    # Calculate the 20-day exponential moving average of the closing price
    ema_20_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the alpha factor as the product of ROC and the ratio of 5-day average volume to EMA of closing price
    heuristics_matrix = roc_10 * (avg_volume_5 / ema_20_close)
    
    return heuristics_matrix
