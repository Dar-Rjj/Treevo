def heuristics_v2(df):
    # Calculate 10-day Exponential Moving Average (EMA) of the closing price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate percentage change in volume from the previous day
    vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Aroon Oscillator
    high_rolling = df['high'].rolling(window=25).apply(lambda x: x.argmax() + 1, raw=True)
    low_rolling = df['low'].rolling(window=25).apply(lambda x: x.argmin() + 1, raw=True)
    aroon_up = ((25 - high_rolling) / 25) * 100
    aroon_down = ((25 - low_rolling) / 25) * 100
    aroon_oscillator = aroon_up - aroon_down
    
    # Combine factors into the heuristics matrix
    heuristics_matrix = ema_10 + vol_change + aroon_oscillator
    
    return heuristics_matrix
