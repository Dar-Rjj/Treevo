def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 40-day EWMA
    ewma = df['close'].ewm(span=40, adjust=False).mean()
    
    # Calculate 10-day SMA of volume
    sma_volume = df['volume'].rolling(window=10).mean()
    
    # Combine factors
    heuristics_matrix = roc + (ewma - vwap) / vwap + sma_volume
    
    return heuristics_matrix
