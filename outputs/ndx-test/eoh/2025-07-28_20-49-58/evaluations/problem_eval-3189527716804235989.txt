def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 30-day SMA of VWAP
    vwap_sma30 = vwap.rolling(window=30).mean()
    
    # Calculate 15-day momentum
    momentum_15 = df['close'] - df['close'].shift(15)
    
    # Combine factors
    heuristics_matrix = roc + (vwap_sma30 - df['close']) + momentum_15
    
    return heuristics_matrix
