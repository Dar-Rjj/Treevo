def heuristics_v2(df):
    # Calculate 20-day simple moving average
    sma_20 = df['close'].rolling(window=20).mean()
    # Calculate 50-day simple moving average
    sma_50 = df['close'].rolling(window=50).mean()
    # Compute the ratio of the 20-day SMA to the 50-day SMA
    sma_ratio = sma_20 / sma_50
    
    # Calculate the 10-day change in volume
    vol_change = df['volume'] - df['volume'].shift(10)
    # Calculate the average volume over the past 10 days
    avg_vol = df['volume'].rolling(window=10).mean()
    
    # Adjust the SMAs' ratio with a factor based on recent volume changes
    adjustment_factor = vol_change / avg_vol
    heuristics_matrix = sma_ratio * (1 + adjustment_factor)
    
    return heuristics_matrix
