def heuristics_v2(df):
    # Calculate 10-day EMA
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=30, adjust=False).mean()
    
    # EMA Crossover Signal
    ema_crossover = ema_short - ema_long
    
    # Calculate 20-day standard deviation of the closing price
    std_20 = df['close'].rolling(window=20).std()
    
    # Calculate 5-day ROC of the volume
    roc_volume = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Combine factors
    heuristics_matrix = ema_crossover + std_20 + roc_volume
    
    return heuristics_matrix
