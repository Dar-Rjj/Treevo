import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 5-day moving average of the closing price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate 10-day simple moving average of the closing price
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Combine factors
    heuristics_matrix = roc + (ma_5 / sma_10)
    
    return heuristics_matrix
