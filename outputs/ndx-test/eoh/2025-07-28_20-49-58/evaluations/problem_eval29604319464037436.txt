def heuristics_v2(df):
    # Calculate the difference between daily high and low prices
    hl_diff = df['high'] - df['low']
    
    # Calculate the 10-day average of closing prices
    close_avg_10 = df['close'].rolling(window=10).mean()
    
    # Compute the alpha factor as the ratio of hl_diff to the 10-day average of closing prices
    heuristics_matrix = (hl_diff / close_avg_10).dropna()
    
    return heuristics_matrix
