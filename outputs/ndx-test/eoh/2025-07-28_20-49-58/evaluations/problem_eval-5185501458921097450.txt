def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 60-day period
    roc = calculate_roc(df['close'], 60)
    
    # Calculate 20-day EMA of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Combine factors
    heuristics_matrix = roc + (ema_20 - df['close'])
    
    return heuristics_matrix
