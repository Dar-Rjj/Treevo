def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate VWAP with 10-day moving average
    tp = (df['high'] + df['low'] + ma_10) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Capture the 21-day ATR
    atr = (df['high'] - df['low']).rolling(window=21).mean()
    
    # Adjust the heuristics matrix with 15-day SMA of ROC and ATR
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=15).mean() - atr
    
    return heuristics_matrix
