def heuristics_v2(df):
    # Calculate the 10-day rolling standard deviation of closing prices
    close_std = df['close'].rolling(window=10).std()
    
    # Calculate the True Range
    tr = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    
    # Calculate the 5-day average True Range
    atr_5 = tr.rolling(window=5).mean()
    
    # Compute the alpha factor as the product of the 10-day rolling standard deviation and the 5-day ATR
    heuristics_matrix = close_std * atr_5
    
    return heuristics_matrix
