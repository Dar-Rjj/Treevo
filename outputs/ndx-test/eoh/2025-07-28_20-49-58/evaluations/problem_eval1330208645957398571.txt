def heuristics_v2(df):
    sma_close = df['close'].rolling(window=20).mean()
    range_high_low = df['high'] - df['low']
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    heuristics_matrix = (sma_close * 0.4) + (range_high_low * 0.3) + (ema_volume * 0.3)
    return heuristics_matrix
