import pandas as pd

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    df['short_mavg'] = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    df['long_mavg'] = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    df['volatility'] = df['close'].pct_change().rolling(window=long_window, min_periods=long_window).std()
    heuristics_matrix = (df['short_mavg'] - df['long_mavg']) / df['volatility']
    return heuristics_matrix
