import pandas as pd

def heuristics_v2(df):
    def calculate_ema(column, n):
        return column.ewm(span=n, adjust=False).mean()

    # Calculate EMA over a 10-day period
    ema = calculate_ema(df['close'], 10)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day standard deviation of the closing price
    std_dev = df['close'].rolling(window=20).std()
    
    # Combine factors
    heuristics_matrix = (ema - df['close']) + (vwap - df['close']) / std_dev
    
    return heuristics_matrix
