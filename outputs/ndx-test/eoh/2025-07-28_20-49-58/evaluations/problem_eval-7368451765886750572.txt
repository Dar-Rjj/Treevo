def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 15-day period
    roc = calculate_roc(df['close'], 15)
    
    # Calculate 7-day exponential moving average of the closing price
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    
    # Calculate VWAP with 7-day exponential moving average
    tp = (df['high'] + df['low'] + ema_7) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 7-day standard deviation of the closing price as a measure of volatility
    std_dev_7 = df['close'].rolling(window=7).std()
    
    # Adjust the heuristics matrix with momentum and volatility
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=7).mean() - std_dev_7
    
    return heuristics_matrix
