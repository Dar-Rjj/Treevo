def heuristics_v2(df):
    # Calculate the monthly momentum of the close price
    monthly_momentum = df['close'].pct_change(20)
    # Calculate the RSI based on the last 10 days' close prices
    delta = df['close'].diff()
    avg_gain = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    avg_loss = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi_10 = 100 - (100 / (1 + rs))
    # Combine monthly momentum and RSI into a single heuristic
    heuristics_matrix = 0.6 * monthly_momentum + 0.4 * rsi_10.shift(1)
    return heuristics_matrix
