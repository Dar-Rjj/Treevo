def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    def calculate_rsi(column, n=14):
        delta = column.diff(1)
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window=n).mean()
        roll_down = down.abs().rolling(window=n).mean()
        rs = roll_up / roll_down
        rsi = 100.0 - (100.0 / (1.0 + rs))
        return rsi

    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate Volume-Weighted Moving Average (WMA)
    wma = (df['volume'] * df['close']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate RSI over a 14-day period
    rsi = calculate_rsi(df['close'])

    # Combine factors
    heuristics_matrix = roc + (wma - df['close']) + rsi
    
    return heuristics_matrix
