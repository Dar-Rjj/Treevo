def heuristics_v2(df):
    sma_20 = df['close'].rolling(window=20).mean()
    sma_60 = df['close'].rolling(window=60).mean()
    std_20 = df['close'].rolling(window=20).std()
    std_60 = df['close'].rolling(window=60).std()
    weight_20 = 1 / std_20
    weight_60 = 1 / std_60
    total_weight = weight_20 + weight_60
    weighted_sma_sum = (sma_20 * weight_20 + sma_60 * weight_60) / total_weight
    min_120 = df['close'].rolling(window=120).min()
    momentum = df['close'] - min_120
    heuristics_matrix = weighted_sma_sum + momentum
    return heuristics_matrix
