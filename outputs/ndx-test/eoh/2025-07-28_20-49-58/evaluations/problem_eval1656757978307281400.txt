import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 5-day moving average of VWAP
    vwap_ma5 = vwap.rolling(window=5).mean()
    
    # Calculate 20-day simple moving average of the closing price
    sma20 = df['close'].rolling(window=20).mean()
    
    # Calculate 30-day EMA of ROC
    ema30 = roc.ewm(span=30, adjust=False).mean()
    
    # Combine factors
    heuristics_matrix = roc + (vwap_ma5 - sma20) + ema30
    
    return heuristics_matrix
