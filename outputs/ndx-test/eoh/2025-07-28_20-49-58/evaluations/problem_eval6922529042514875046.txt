import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day EMA of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 5-day standard deviation of the closing price
    std_5 = df['close'].rolling(window=5).std()
    
    # Combine factors
    heuristics_matrix = ema_20 + std_5
    
    return heuristics_matrix
