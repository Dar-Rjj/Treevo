def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate VWAP with 10-day moving average
    tp = (df['high'] + df['low'] + ma_10) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Capture the 5-day volatility of the closing prices
    vol_5 = df['close'].rolling(window=5).std()
    
    # Adjust the heuristics matrix with momentum and volatility
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=5).mean() - vol_5
    
    return heuristics_matrix
