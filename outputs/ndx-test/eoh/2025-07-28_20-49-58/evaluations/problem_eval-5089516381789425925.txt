def heuristics_v2(df):
    def calculate_rsi(column, n):
        delta = column.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    # Calculate RSI over a 14-day period
    rsi = calculate_rsi(df['close'], 14)
    
    # Calculate 20-day moving average of the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate VWAP with 20-day moving average
    tp = (df['high'] + df['low'] + ma_20) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Adjust the heuristics matrix with momentum
    heuristics_matrix = rsi + (vwap - df['close']) + rsi.rolling(window=5).mean()
    
    return heuristics_matrix
