def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate EMA over a 10-day period
    ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine factors, using the average volume as a weight
    weighted_factor = ((vwap / ema) - 1) * df['volume'].mean()
    
    heuristics_matrix = roc + weighted_factor
    
    return heuristics_matrix
