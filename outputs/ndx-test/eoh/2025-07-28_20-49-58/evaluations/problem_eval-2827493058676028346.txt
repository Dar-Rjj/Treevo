def heuristics_v2(df):
    # Calculate 10-day and 30-day EMA
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # EMA Crossover Signal
    ema_crossover = ema_10 - ema_30
    
    # Calculate 20-day moving standard deviation as an adaptive volatility factor
    std_dev_20 = df['close'].rolling(window=20).std()
    
    # Calculate Relative Strength (RS) as the ratio of 10-day EMA to 30-day EMA
    rs = ema_10 / ema_30
    
    # Adjust the heuristics matrix with EMA crossover, RS, and adaptive volatility
    heuristics_matrix = ema_crossover + rs - std_dev_20
    
    return heuristics_matrix
