def heuristics_v2(df):
    # Calculate 20-day EMA of the closing price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 5-day SMA of the volume
    sma_volume = df['volume'].rolling(window=5).mean()
    
    # Combine factors
    heuristics_matrix = sma_volume - ema_close
    
    return heuristics_matrix
