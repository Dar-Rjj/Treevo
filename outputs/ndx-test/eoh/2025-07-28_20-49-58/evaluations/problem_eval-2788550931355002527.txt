def heuristics_v2(df):
    def calculate_momentum(column, n):
        return column - column.shift(n)

    # Calculate Momentum over a 20-day period
    momentum = calculate_momentum(df['close'], 20)
    
    # Calculate RSI over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate 50-day SMA
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Combine factors
    heuristics_matrix = momentum + rsi + (sma_50 - df['close'])
    
    return heuristics_matrix
