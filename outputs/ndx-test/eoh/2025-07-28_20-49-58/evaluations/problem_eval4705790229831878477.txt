import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day EMA of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 5-day SMA of the closing price
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 10-day mean volume
    mean_vol_10 = df['volume'].rolling(window=10).mean()
    
    # Calculate the factor
    heuristics_matrix = (ema_20 - sma_5) * (df['volume'] / mean_vol_10)
    
    return heuristics_matrix
