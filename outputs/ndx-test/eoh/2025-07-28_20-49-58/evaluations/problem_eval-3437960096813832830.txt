def heuristics_v2(df):
    # Calculate the 10-day ROC
    roc_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate the 100-day moving average of the closing price
    ma_100 = df['close'].rolling(window=100).mean()
    
    # Calculate the adjusted ROC by incorporating the difference from the 100-day MA
    heuristics_matrix = roc_10 + (df['close'] - ma_100)
    
    return heuristics_matrix
