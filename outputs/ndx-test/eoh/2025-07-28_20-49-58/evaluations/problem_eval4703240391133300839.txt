def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 10-day momentum of VWAP
    vwap_mom10 = vwap - vwap.shift(10)
    
    # Combine factors
    heuristics_matrix = roc + (vwap_mom10 - df['close'])
    
    return heuristics_matrix
