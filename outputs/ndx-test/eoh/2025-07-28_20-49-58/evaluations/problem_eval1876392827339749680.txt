def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 15-day moving average of the closing price
    ma_15 = df['close'].rolling(window=15).mean()
    
    # Adjusted typical price incorporating the 15-day moving average
    tp = (df['high'] + df['low'] + ma_15) / 3
    
    # Calculate VWAP with the adjusted typical price
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Combine factors
    heuristics_matrix = roc + (vwap - df['close'])
    
    return heuristics_matrix
