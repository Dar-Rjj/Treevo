def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    def calculate_macd(close, short_window=12, long_window=26, signal_window=9):
        short_ema = close.ewm(span=short_window, adjust=False).mean()
        long_ema = close.ewm(span=long_window, adjust=False).mean()
        macd_line = short_ema - long_ema
        signal_line = macd_line.ewm(span=signal_window, adjust=False).mean()
        return signal_line

    roc_short = calculate_roc(df['close'], 5)
    macd_signal = calculate_macd(df['close'])

    heuristics_matrix = 0.5 * (roc_short + macd_signal)
    return heuristics_matrix
