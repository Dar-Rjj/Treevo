def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate 10-day moving average of the volume
    vol_ma10 = df['volume'].rolling(window=10).mean()
    
    # Calculate 10-day standard deviation of the closing price as a measure of volatility
    std_10 = df['close'].rolling(window=10).std()
    
    # Adjust the heuristics matrix with the volatility factor
    heuristics_matrix = roc + (vol_ma10 - df['volume']) + std_10
    
    return heuristics_matrix
