import pandas as pd

def heuristics_v2(df):
    # Calculate EMA over a 20-day period
    ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 5-day moving average of volume
    vol_ma5 = df['volume'].rolling(window=5).mean()
    
    # Calculate 10-day simple moving average of the closing price
    sma10 = df['close'].rolling(window=10).mean()
    
    # Combine factors
    heuristics_matrix = ema + (vol_ma5 - sma10)
    
    return heuristics_matrix
