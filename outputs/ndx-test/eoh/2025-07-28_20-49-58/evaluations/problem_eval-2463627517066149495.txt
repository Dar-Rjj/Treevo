def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate 20-day and 50-day moving averages
    ma_20 = df['close'].rolling(window=20).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Ratio of 20-day MA to 50-day MA
    ma_ratio = ma_20 / ma_50
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 30-day EWMA
    ewma = df['close'].ewm(span=30, adjust=False).mean()
    
    # Combine factors
    heuristics_matrix = roc + ma_ratio + (ewma - vwap) / vwap
    
    return heuristics_matrix
