import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 20-day exponentially weighted moving average of the closing price
    ewma_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the ratio of the closing price to the 20-day EWMA
    close_to_ewma_ratio = df['close'] / ewma_20
    
    # Calculate 5-day ROC of the volume
    roc_volume_5 = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Calculate 14-day RSI on the closing prices
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Combine the factors
    heuristics_matrix = close_to_ewma_ratio - roc_volume_5 + rsi_14
    
    return heuristics_matrix
