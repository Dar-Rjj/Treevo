def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC of volume over a 15-day period
    roc_volume = calculate_roc(df['volume'], 15)
    
    # Calculate 20-day simple moving average of the closing price
    sma20 = df['close'].rolling(window=20).mean()
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate the standard deviation of the closing price over a 10-day period
    std_close = df['close'].rolling(window=10).std()
    
    # Combine factors
    heuristics_matrix = roc_volume + (sma20 - vwap) + std_close
    
    return heuristics_matrix
