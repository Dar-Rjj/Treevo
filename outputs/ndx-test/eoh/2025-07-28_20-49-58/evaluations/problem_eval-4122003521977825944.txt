def heuristics_v2(df):
    # Calculate the 10-day and 60-day exponentially weighted moving average of closing prices
    close_ewma_10 = df['close'].ewm(span=10).mean()
    close_ewma_60 = df['close'].ewm(span=60).mean()
    
    # Compute the combined EWMA as the sum of the 10-day and 60-day EWMAs
    combined_ewma = close_ewma_10 + close_ewma_60
    
    # Compute the alpha factor as the ratio of the current closing price to the combined EWMA of closing prices
    heuristics_matrix = (df['close'] / combined_ewma).dropna()
    
    return heuristics_matrix
