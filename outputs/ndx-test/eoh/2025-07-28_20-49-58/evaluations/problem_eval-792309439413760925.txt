def heuristics_v2(df):
    def calculate_macd(column, n_fast, n_slow):
        ema_fast = column.ewm(span=n_fast, adjust=False).mean()
        ema_slow = column.ewm(span=n_slow, adjust=False).mean()
        return ema_fast - ema_slow

    def calculate_sma(column, n):
        return column.rolling(window=n).mean()

    def calculate_rsi(column, n):
        delta = column.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    # Calculate MACD (26, 12)
    macd = calculate_macd(df['close'], 12, 26)

    # Calculate 50-day SMA and 200-day SMA
    sma_50 = calculate_sma(df['close'], 50)
    sma_200 = calculate_sma(df['close'], 200)
    
    # Calculate 30-day RSI
    rsi = calculate_rsi(df['close'], 30)

    # Combine factors
    heuristics_matrix = macd + (sma_50 - sma_200) + rsi

    return heuristics_matrix
