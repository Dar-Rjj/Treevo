import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    def calculate_ema(column, n):
        return column.ewm(span=n, adjust=False).mean()
    
    def calculate_atr(high, low, close, n):
        tr = pd.DataFrame(index=high.index)
        tr['h-l'] = high - low
        tr['h-pc'] = abs(high - close.shift(1))
        tr['l-pc'] = abs(low - close.shift(1))
        tr['tr'] = tr.max(axis=1)
        atr = tr['tr'].rolling(window=n).mean()
        return atr
    
    # Calculate ROC over a 14-day period
    roc = calculate_roc(df['close'], 14)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day EMA of VWAP
    vwap_ema20 = calculate_ema(vwap, 20)
    
    # Calculate 5-day ATR
    atr_5 = calculate_atr(df['high'], df['low'], df['close'], 5)
    
    # Combine factors
    heuristics_matrix = roc + (vwap_ema20 - df['close']) + atr_5
    
    return heuristics_matrix
