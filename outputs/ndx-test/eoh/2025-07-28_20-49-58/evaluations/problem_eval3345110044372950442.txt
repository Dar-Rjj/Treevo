import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day return
    thirty_day_return = df['close'] / df['close'].shift(30) - 1
    
    # Calculate 15-day and 60-day EMA
    ema_15 = df['close'].ewm(span=15, adjust=False).mean()
    ema_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Combine factors
    heuristics_matrix = thirty_day_return + (ema_15 - ema_60)
    
    return heuristics_matrix
