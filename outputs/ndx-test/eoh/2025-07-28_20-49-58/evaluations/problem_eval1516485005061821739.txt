import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 30-day period of the average daily trading range
    avg_range = df['high'] - df['low']
    roc = calculate_roc(avg_range, 30)
    
    # Calculate 20-day EMA of the closing price
    ema20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Combine factors
    heuristics_matrix = roc + (ema20 - vwap)
    
    return heuristics_matrix
