def heuristics_v2(df):
    # Calculate the 5-day percentage change in high and low prices
    high_change = df['high'].pct_change(5)
    low_change = df['low'].pct_change(5)
    
    # Compute the average of the 5-day percentage change in high and low prices
    avg_change = (high_change + low_change) / 2
    
    # Calculate the 20-day exponential moving average of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the alpha factor as the product of the average change and the 20-day EMA
    heuristics_matrix = avg_change * ema_20
    
    return heuristics_matrix
