def heuristics_v2(df):
    # Calculate the 30-day exponentially weighted moving average of closing prices
    close_ewma_30 = df['close'].ewm(span=30).mean()
    
    # Compute the alpha factor as the ratio of the current closing price to the 30-day EWMA of closing prices
    heuristics_matrix = (df['close'] / close_ewma_30).dropna()
    
    return heuristics_matrix
