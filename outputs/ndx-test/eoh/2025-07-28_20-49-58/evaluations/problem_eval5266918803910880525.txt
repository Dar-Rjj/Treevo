def heuristics_v2(df):
    # Calculate the 60-day average of closing prices
    close_avg_60 = df['close'].rolling(window=60).mean()
    
    # Compute the alpha factor as the ratio of the current closing price to the 60-day average of closing prices
    heuristics_matrix = (df['close'] / close_avg_60).dropna()
    
    return heuristics_matrix
