def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 15-day SMA of VWAP
    vwap_sma15 = vwap.rolling(window=15).mean()
    
    # Calculate 7-day momentum
    momentum_7 = df['close'] - df['close'].shift(7)
    
    # Combine factors
    heuristics_matrix = roc + (vwap_sma15 - df['close']) + momentum_7
    
    return heuristics_matrix
