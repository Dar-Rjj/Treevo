def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 5-day period
    roc = calculate_roc(df['close'], 5)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day SMA of VWAP
    vwap_sma20 = vwap.rolling(window=20).mean()
    
    # Calculate 14-day momentum
    momentum_14 = df['close'] - df['close'].shift(14)
    
    # Combine factors
    heuristics_matrix = roc + (vwap_sma20 - df['close']) + momentum_14
    
    return heuristics_matrix
