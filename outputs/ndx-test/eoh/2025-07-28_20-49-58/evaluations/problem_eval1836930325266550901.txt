def heuristics_v2(df):
    # Calculate EMA over a 12-day period
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    # Calculate EMA over a 26-day period
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    # Compute the difference between EMAs
    ema_diff = ema_12 - ema_26
    # Calculate 10-day average volume
    avg_volume = df['volume'].rolling(window=10).mean()
    # Compute the ratio of current volume to 10-day average volume
    vol_ratio = df['volume'] / avg_volume
    # Combine factors
    heuristics_matrix = ema_diff + vol_ratio
    
    return heuristics_matrix
