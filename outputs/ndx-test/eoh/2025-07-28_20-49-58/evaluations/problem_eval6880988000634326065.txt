def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 15-day period
    roc = calculate_roc(df['close'], 15)
    
    # Calculate 20-day moving average of the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate RSI with 20-day moving average
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_with_ma = 100 - (100 / (1 + rs))
    
    # Adjust the heuristics matrix with momentum
    heuristics_matrix = roc + (rsi_with_ma - df['close']) + roc.rolling(window=15).mean()
    
    return heuristics_matrix
