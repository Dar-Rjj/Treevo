def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate VWAP with 10-day moving average
    tp = (df['high'] + df['low'] + ma_10) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Capture the 30-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_30 = daily_returns.rolling(window=30).std()
    
    # Adjust the heuristics matrix with 20-day SMA of ROC and standard deviation of daily returns
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=20).mean() - std_30
    
    return heuristics_matrix
