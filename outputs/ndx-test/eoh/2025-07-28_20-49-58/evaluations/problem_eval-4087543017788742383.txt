import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    def calculate_rsi(column, n=14):
        delta = column.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate RSI
    rsi = calculate_rsi(df['close'])

    # Combine factors
    heuristics_matrix = roc + (rsi - ma_10)
    
    return heuristics_matrix
