def heuristics_v2(df):
    short_roc_window = 10
    long_sma_window = 30
    
    # Calculate Rate of Change (ROC)
    roc = (df['close'] - df['close'].shift(short_roc_window)) / df['close'].shift(short_roc_window)
    
    # Calculate Long-Term Simple Moving Average (SMA)
    sma_long = df['close'].rolling(window=long_sma_window, min_periods=long_sma_window).mean()
    
    # Calculate the difference between the closing price and its long-term SMA
    diff_price_sma = df['close'] - sma_long
    
    # Combine factors
    heuristics_matrix = roc + diff_price_sma
    
    return heuristics_matrix
