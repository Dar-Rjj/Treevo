def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 10-day simple moving average of the closing price
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate VWAP with 10-day simple moving average
    tp = (df['high'] + df['low'] + sma_10) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate 14-day ATR as a measure of volatility
    tr = df[['high', 'low']].apply(lambda x: x.diff().abs()).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Adjust the heuristics matrix with momentum (ROC, RSI) and volatility (ATR)
    heuristics_matrix = roc + (vwap - df['close']) + rsi - atr
    
    return heuristics_matrix
