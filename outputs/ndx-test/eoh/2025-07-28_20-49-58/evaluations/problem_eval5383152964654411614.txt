def heuristics_v2(df):
    # Calculate 10-day momentum
    momentum_10 = df['close'] - df['close'].shift(10)
    
    # Calculate the highest high over the past 30 days
    highest_high_30 = df['high'].rolling(window=30).max()
    
    # Compute the alpha factor as the ratio of the current closing price to the highest high over the past 30 days
    ratio_to_highest_high = df['close'] / highest_high_30
    
    # Combine factors
    heuristics_matrix = momentum_10 + ratio_to_highest_high
    
    return heuristics_matrix
