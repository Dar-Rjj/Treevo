def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate 20-day EMA of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 14-day RSI
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine factors
    heuristics_matrix = roc + (ema_20 - df['close']) + rsi
    
    return heuristics_matrix
