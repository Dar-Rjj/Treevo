def heuristics_v2(df):
    short_window = 10
    long_window = 30
    signal_window = 9
    rsi_window = 14
    momentum_window = 5
    
    # Calculate short and long moving averages
    ma_short = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    ma_long = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    
    # Calculate MACD
    macd_line = ma_short - ma_long
    signal_line = macd_line.rolling(window=signal_window, min_periods=signal_window).mean()
    macd_hist = macd_line - signal_line
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=rsi_window, min_periods=rsi_window).mean()
    avg_loss = loss.rolling(window=rsi_window, min_periods=rsi_window).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(momentum_window)
    
    # Combine factors
    heuristics_matrix = macd_hist + (rsi - 50) + momentum
    
    return heuristics_matrix
