def heuristics_v2(df):
    # Calculate the 30-day average of closing prices
    close_avg_30 = df['close'].rolling(window=30).mean()
    
    # Compute the alpha factor as the ratio of the current closing price to the 30-day average of closing prices
    heuristics_matrix = (df['close'] / close_avg_30).dropna()
    
    return heuristics_matrix
