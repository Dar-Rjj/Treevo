def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    def calculate_rsi(column, n):
        delta = column.diff()
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window=n, min_periods=1).mean()
        roll_down = down.abs().rolling(window=n, min_periods=1).mean()
        rs = roll_up / roll_down
        rsi = 100.0 - (100.0 / (1.0 + rs))
        return rsi

    # Calculate ROC over a 12-day period
    roc = calculate_roc(df['close'], 12)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 30-day EWMA
    ewma = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate 14-day RSI
    rsi = calculate_rsi(df['close'], 14)
    
    # Combine factors
    heuristics_matrix = (roc + (ewma - vwap) / vwap) * rsi
    
    return heuristics_matrix
