import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 14-day period
    roc = calculate_roc(df['close'], 14)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 7-day moving average of VWAP
    vwap_ma7 = vwap.rolling(window=7).mean()
    
    # Calculate 20-day simple moving average of the closing price
    sma20 = df['close'].rolling(window=20).mean()
    
    # Combine factors
    heuristics_matrix = roc + (vwap_ma7 - sma20)
    
    return heuristics_matrix
