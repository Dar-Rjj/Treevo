def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 15-day period
    roc = calculate_roc(df['close'], 15)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day EMA of VWAP
    vwap_ema20 = vwap.ewm(span=20, adjust=False).mean()
    
    # Calculate 10-day standard deviation of the closing price as a measure of volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Combine factors
    heuristics_matrix = roc + (vwap_ema20 - df['close']) + volatility
    
    return heuristics_matrix
