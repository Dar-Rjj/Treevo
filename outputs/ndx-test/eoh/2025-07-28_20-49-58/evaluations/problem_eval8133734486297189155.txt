def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate VWAP with 10-day moving average
    tp = (df['high'] + df['low'] + ma_10) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Capture the 20-day relative volatility of the closing prices
    cv_20 = df['close'].rolling(window=20).std() / df['close'].rolling(window=20).mean()
    
    # Adjust the heuristics matrix with mean reversion and relative volatility
    heuristics_matrix = -roc + (vwap - df['close']) - roc.rolling(window=5).mean() - cv_20
    
    return heuristics_matrix
