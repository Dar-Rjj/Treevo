def heuristics_v2(df):
    # Calculate the 30-day and 60-day simple moving averages
    sma_30 = df['close'].rolling(window=30).mean()
    sma_60 = df['close'].rolling(window=60).mean()
    
    # Compute the momentum of the closing price over a 15-day period
    momentum = df['close'] - df['close'].shift(15)
    
    # Combine factors
    heuristics_matrix = momentum + (sma_30 - sma_60) / sma_60
    
    return heuristics_matrix
