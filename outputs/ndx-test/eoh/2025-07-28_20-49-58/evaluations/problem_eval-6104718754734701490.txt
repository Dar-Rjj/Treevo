import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum of the close price by comparing it with the close price 10 days ago
    momentum = df['close'].pct_change(10)
    # Calculate the RSI based on the last 14 days' close prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    # Combine momentum and RSI into a single heuristic
    heuristics_matrix = 0.5 * (momentum + rsi.shift(1))
    return heuristics_matrix
