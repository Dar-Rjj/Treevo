import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day momentum
    momentum_20 = df['close'] - df['close'].shift(20)
    
    # Calculate 5-day EMA
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate 14-day SMA
    sma_14 = df['close'].rolling(window=14).mean()
    
    # Combine factors
    heuristics_matrix = momentum_20 + (ema_5 - sma_14)
    
    return heuristics_matrix
