def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate standard deviation over the last 5 days
    std_dev_5 = df['close'].rolling(window=5).std()
    
    # Calculate 14-day EMA
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    
    # Combine factors
    heuristics_matrix = roc + std_dev_5 + (df['close'] - ema_14)
    
    return heuristics_matrix
