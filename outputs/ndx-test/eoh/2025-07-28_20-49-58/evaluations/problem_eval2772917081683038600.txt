def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day SMA of VWAP
    vwap_sma20 = vwap.rolling(window=20).mean()
    
    # Adjust ROC using the ratio of VWAP's 20-day SMA to the closing price
    adjusted_roc = roc * (vwap_sma20 / df['close'])
    
    # Calculate 5-day momentum
    momentum_5 = df['close'] - df['close'].shift(5)
    
    # Combine factors
    heuristics_matrix = adjusted_roc + momentum_5
    
    return heuristics_matrix
