import pandas as pd

def heuristics_v2(df):
    def calculate_rsi(column, n):
        delta = column.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    # Calculate RSI over a 10-day period
    rsi = calculate_rsi(df['close'], 10)
    
    # Calculate 10-day moving average of volume
    volume_ma10 = df['volume'].rolling(window=10).mean()
    
    # Combine factors
    heuristics_matrix = rsi + (df['volume'] / volume_ma10)
    
    return heuristics_matrix
