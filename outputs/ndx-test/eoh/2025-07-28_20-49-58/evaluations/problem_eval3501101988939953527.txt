def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate 5-day moving average of the closing price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate VWAP with 5-day moving average
    tp = (df['high'] + df['low'] + ma_5) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Capture the 20-day return standard deviation
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    
    # Adjust the heuristics matrix with 5-day SMA of ROC and 20-day return standard deviation
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=5).mean() - std_dev_20
    
    return heuristics_matrix
