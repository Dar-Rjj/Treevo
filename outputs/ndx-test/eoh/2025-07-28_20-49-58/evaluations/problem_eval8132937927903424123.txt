def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate 30-day SMA of the closing price
    sma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate VWAP with 30-day SMA
    tp = (df['high'] + df['low'] + sma_30) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 30-day standard deviation of the closing price as a measure of volatility
    std_dev_30 = df['close'].rolling(window=30).std()
    
    # Adjust the heuristics matrix with momentum and volatility
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=10).mean() - std_dev_30
    
    return heuristics_matrix
