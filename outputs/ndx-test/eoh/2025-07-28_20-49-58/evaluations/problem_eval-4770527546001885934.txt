def heuristics_v2(df):
    # Calculate the 50-day average of closing prices
    close_avg_50 = df['close'].rolling(window=50).mean()
    
    # Compute the alpha factor as the ratio of the current closing price to the 50-day average of closing prices
    heuristics_matrix = (df['close'] / close_avg_50).dropna()
    
    return heuristics_matrix
