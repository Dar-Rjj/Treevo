def heuristics_v2(df):
    def calculate_ema(column, span):
        return column.ewm(span=span, adjust=False).mean()

    def calculate_rsi(column, n):
        delta = column.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    # Calculate 5-day and 20-day EMA
    ema5 = calculate_ema(df['close'], 5)
    ema20 = calculate_ema(df['close'], 20)
    
    # Calculate 14-day RSI
    rsi14 = calculate_rsi(df['close'], 14)
    
    # Combine factors
    heuristics_matrix = (ema5 - ema20) + rsi14
    
    return heuristics_matrix
