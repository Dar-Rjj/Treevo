def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 20-day EMA of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate VWAP with 20-day EMA
    tp = (df['high'] + df['low'] + ema_20) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day standard deviation of the closing price as a measure of volatility
    std_dev_20 = df['close'].rolling(window=20).std()
    
    # Adjust the heuristics matrix with momentum and volatility
    heuristics_matrix = roc + (vwap - df['close']) + roc.rolling(window=20).mean() - std_dev_20
    
    return heuristics_matrix
