def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 10-day exponential moving average of VWAP
    vwap_ema10 = vwap.ewm(span=10, adjust=False).mean()
    
    # Combine factors
    heuristics_matrix = roc + (vwap_ema10 - df['close'])
    
    return heuristics_matrix
