import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    def calculate_ema(column, span):
        return column.ewm(span=span, adjust=False).mean()
    
    def calculate_rsi(column, n):
        delta = column.diff()
        up = delta.clip(lower=0)
        down = -delta.clip(upper=0)
        avg_gain = up.rolling(window=n).mean()
        avg_loss = down.rolling(window=n).mean()
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day EMA of VWAP
    vwap_ema20 = calculate_ema(vwap, 20)
    
    # Calculate 14-day RSI
    rsi_14 = calculate_rsi(df['close'], 14)
    
    # Combine factors
    heuristics_matrix = roc + (vwap_ema20 - df['close']) + rsi_14
    
    return heuristics_matrix
