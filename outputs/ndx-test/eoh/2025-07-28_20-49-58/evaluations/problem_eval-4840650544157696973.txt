def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)
    
    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate 10-day exponential moving average of the closing price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate VWAP with 10-day exponential moving average
    tp = (df['high'] + df['low'] + ema_10) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 10-day standard deviation of the high and low prices as a measure of volatility
    std_dev_10 = (df['high'] - df['low']).rolling(window=10).std()
    
    # Adjust the heuristics matrix with 14-day EMA of ROC and volatility
    heuristics_matrix = roc + (vwap - df['close']) + roc.ewm(span=14, adjust=False).mean() - std_dev_10
    
    return heuristics_matrix
