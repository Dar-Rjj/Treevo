import pandas as pd

def heuristics_v2(df):
    short_window = 10
    long_window = 30
    rsi_window = 14
    
    # Calculate short and long moving averages
    ma_short = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    ma_long = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=rsi_window, min_periods=rsi_window).mean()
    avg_loss = loss.rolling(window=rsi_window, min_periods=rsi_window).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine factors
    heuristics_matrix = (ma_short - ma_long) + (rsi - 50)
    
    return heuristics_matrix
