def heuristics_v2(df):
    # Calculate 20-day Exponential Moving Average (EMA)
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Aroon Oscillator
    aroon_up = ((df['high'].rolling(window=25).apply(lambda x: x.argmax(), raw=True)) / 24) * 100
    aroon_down = ((df['low'].rolling(window=25).apply(lambda x: x.argmin(), raw=True)) / 24) * 100
    aroon_oscillator = aroon_up - aroon_down
    
    # Calculate Modified On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int)).cumsum()

    # Combine factors into heuristics matrix
    heuristics_matrix = (ema_20 - df['close']) + aroon_oscillator + (obv - obv.shift(1))
    
    return heuristics_matrix
