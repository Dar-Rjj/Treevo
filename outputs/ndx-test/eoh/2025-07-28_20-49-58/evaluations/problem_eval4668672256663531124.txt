def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day EMA of VWAP
    vwap_ema20 = vwap.ewm(span=20, adjust=False).mean()
    
    # Calculate 10-day SMA of volume
    volume_sma10 = df['volume'].rolling(window=10).mean()
    
    # Combine factors
    heuristics_matrix = roc + (vwap_ema20 / df['close']) + volume_sma10
    
    return heuristics_matrix
