def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 30-day period
    roc = calculate_roc(df['close'], 30)
    
    # Calculate 20-day EMA of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate VWAP with 20-day EMA
    tp = (df['high'] + df['low'] + ema_20) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 10-day standard deviation of the closing price
    std_10 = df['close'].rolling(window=10).std()
    
    # Combine factors
    heuristics_matrix = roc + (vwap - df['close']) + std_10
    
    return heuristics_matrix
