def heuristics_v2(df):
    # Calculate the 10-day percentage change in volume
    volume_change = df['volume'].pct_change(10)
    
    # Calculate the 10-day rate of change in closing prices
    close_roc = df['close'].diff(10) / df['close'].shift(10)
    
    # Compute the alpha factor as the product of volume change and rate of change in closing prices
    heuristics_matrix = volume_change * close_roc
    
    return heuristics_matrix
