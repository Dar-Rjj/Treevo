def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 20-day period
    roc = calculate_roc(df['close'], 20)
    
    # Calculate 5-day moving average of the closing price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate VWAP with 5-day moving average
    tp = (df['high'] + df['low'] + ma_5) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day standard deviation as a measure of volatility
    std_20 = df['close'].rolling(window=20).std()
    
    # Adjust the heuristics matrix with volatility
    heuristics_matrix = roc + (vwap - df['close']) + std_20
    
    return heuristics_matrix
