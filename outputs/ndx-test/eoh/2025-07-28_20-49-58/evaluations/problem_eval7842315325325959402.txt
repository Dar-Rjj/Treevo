import pandas as pd

def heuristics_v2(df):
    # Calculate the relative change in trading volume
    df['volume_change'] = df['volume'].pct_change()
    # Calculate the Average True Range (ATR) over a 14-day window
    df['tr'] = pd.DataFrame({'high-low': df['high'] - df['low'], 
                             'high-close': abs(df['high'] - df['close'].shift()), 
                             'low-close': abs(df['low'] - df['close'].shift())}).max(axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()
    # Combine the volume change with the ATR
    heuristics_matrix = 0.5 * (df['volume_change'] + df['atr'])
    return heuristics_matrix
