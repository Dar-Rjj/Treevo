def heuristics_v2(df):
    def calculate_ewma(column, n):
        return column.ewm(span=n, adjust=False).mean()
    
    # Calculate 20-day EWMA of the closing price
    ewma_20 = calculate_ewma(df['close'], 20)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 20-day SMA of VWAP
    vwap_sma20 = vwap.rolling(window=20).mean()
    
    # Calculate 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine factors
    heuristics_matrix = ewma_20 + (vwap_sma20 / df['close']) + rsi
    
    return heuristics_matrix
