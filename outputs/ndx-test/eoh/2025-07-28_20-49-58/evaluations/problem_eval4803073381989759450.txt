import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    def calculate_rsi(series, n):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    # Calculate ROC over a 5-day period
    roc = calculate_roc(df['close'], 5)
    
    # Calculate RSI over a 21-day period
    rsi = calculate_rsi(df['close'], 21)
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Combine factors
    heuristics_matrix = roc + rsi + (vwap - df['close'])
    
    return heuristics_matrix
