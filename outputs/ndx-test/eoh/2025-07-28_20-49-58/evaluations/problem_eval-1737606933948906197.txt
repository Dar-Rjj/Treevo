def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate VWAP with open price included
    tp = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    vwap = (df['volume'] * tp).cumsum() / df['volume'].cumsum()
    
    # Calculate 50-day EWMA
    ewma = df['close'].ewm(span=50, adjust=False).mean()
    
    # Combine factors
    heuristics_matrix = roc + (ewma - vwap) / vwap
    
    return heuristics_matrix
