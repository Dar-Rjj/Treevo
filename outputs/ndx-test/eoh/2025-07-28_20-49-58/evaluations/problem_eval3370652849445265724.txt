def heuristics_v2(df):
    # Calculate the 25-day minimum close
    min_close_25 = df['close'].rolling(window=25).min()
    
    # Compute the difference between the close and the 25-day minimum close
    diff_close_min = df['close'] - min_close_25
    
    # Calculate the money flow multiplier
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mfm * df['volume']
    
    # Calculate the 10-day Accumulation Distribution Line (ADL)
    adl = mf_volume.rolling(window=10).sum()
    
    # Calculate the 30-day Return on Volume Change
    roc_vol = (df['volume'] - df['volume'].shift(30)) / df['volume'].shift(30)
    
    # Combine factors
    heuristics_matrix = diff_close_min + adl + roc_vol
    
    return heuristics_matrix
