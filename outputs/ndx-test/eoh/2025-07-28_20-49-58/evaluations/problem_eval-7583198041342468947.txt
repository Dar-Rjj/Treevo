import pandas as pd

def heuristics_v2(df):
    # Calculate the 12-day and 26-day exponential moving averages (EMAs)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    # Calculate the MACD line
    macd_line = ema_12 - ema_26
    # Calculate the 9-day EMA of the MACD line for the signal line
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    # Calculate the difference between the MACD line and the signal line
    macd_histogram = macd_line - signal_line
    # Calculate the 50-day and 200-day simple moving averages (SMAs)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    # Generate the SMA crossover signal; positive if 50-day SMA is above 200-day SMA, negative otherwise
    sma_crossover_signal = (sma_50 > sma_200).astype(int) * 2 - 1
    # Combine the MACD histogram and the SMA crossover signal into a single heuristic
    heuristics_matrix = 0.5 * (macd_histogram + sma_crossover_signal)
    return heuristics_matrix
