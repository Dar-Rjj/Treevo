def heuristics_v2(df):
    # Calculate the intraday price range
    price_range = (df['high'] - df['low'])
    # Calculate the volume change
    volume_change = df['volume'].pct_change()
    # Combine the intraday price range and the volume change
    heuristics_matrix = 0.5 * (price_range + volume_change)
    heuristics_matrix = heuristics_matrix.dropna()
    return heuristics_matrix
