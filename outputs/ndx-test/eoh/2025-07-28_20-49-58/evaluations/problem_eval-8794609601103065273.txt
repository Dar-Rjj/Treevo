import pandas as pd

def heuristics_v2(df):
    # Calculate short-term (5 days) and long-term (30 days) moving averages for closing prices
    df['MA_5'] = df['close'].rolling(window=5).mean()
    df['MA_30'] = df['close'].rolling(window=30).mean()
    
    # Daily return calculation
    df['Return'] = df['close'].pct_change()
    
    # Volatility over the last 5 days
    df['Vol_5'] = df['Return'].rolling(window=5).std()
    
    # Constructing the heuristics factor: (MA_5 - MA_30) / Vol_5
    heuristics_matrix = (df['MA_5'] - df['MA_30']) / df['Vol_5']
    
    return heuristics_matrix
