def heuristics_v2(df):
    # Calculate the 30-day minimum closing price
    min_close_30 = df['close'].rolling(window=30).min()
    
    # Calculate the 10-day average trading volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Compute the alpha factor
    heuristics_matrix = (df['close'] - min_close_30) + (df['volume'] / avg_volume_10)
    
    return heuristics_matrix
