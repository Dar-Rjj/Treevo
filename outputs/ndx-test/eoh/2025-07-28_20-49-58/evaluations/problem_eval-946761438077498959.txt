import pandas as pd

def heuristics_v2(df):
    # Compute short and long moving averages for close prices
    df['short_MA'] = df['close'].rolling(window=10).mean()
    df['long_MA'] = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between short and long MAs
    df['MA_diff'] = df['short_MA'] - df['long_MA']
    
    # Compute momentum indicator as the percent change over 12 days
    df['momentum'] = df['close'].pct_change(periods=12)
    
    # Generate score from MA difference and momentum
    df['score'] = df['MA_diff'] * df['momentum']
    
    # Rank the scores to get the heuristics matrix
    heuristics_matrix = df['score'].rank(pct=True)
    
    return heuristics_matrix
