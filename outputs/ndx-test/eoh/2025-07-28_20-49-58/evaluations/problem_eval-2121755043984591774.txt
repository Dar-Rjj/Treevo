def heuristics_v2(df):
    # Calculate the 10-day range (high - low)
    range_10 = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    
    # Calculate the 10-day average daily trading volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Compute the alpha factor as the ratio of the 10-day range to the 10-day average daily trading volume
    heuristics_matrix = (range_10 / avg_volume_10).dropna()
    
    return heuristics_matrix
