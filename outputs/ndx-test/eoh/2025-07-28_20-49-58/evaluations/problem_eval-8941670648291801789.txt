import pandas as pd

def heuristics_v2(df):
    def calculate_roc(column, n):
        return (column - column.shift(n)) / column.shift(n)

    # Calculate ROC over a 10-day period
    roc = calculate_roc(df['close'], 10)
    
    # Calculate modified VWAP using median price
    median_price = (df['high'] + df['low']) / 2
    vwap = (df['volume'] * median_price).cumsum() / df['volume'].cumsum()
    
    # Combine factors
    heuristics_matrix = roc * (vwap - df['close'])
    
    return heuristics_matrix
