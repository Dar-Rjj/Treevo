import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day moving average return
    ma_50 = df['close'].rolling(window=50).mean()
    ma_return = (df['close'] - ma_50) / ma_50
    
    # Calculate volume-weighted standard deviation of the close price
    weighted_close = df['close'] * df['volume']
    vwap = weighted_close.cumsum() / df['volume'].cumsum()
    vol_weighted_std = ((df['close'] - vwap)**2 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine factors
    heuristics_matrix = ma_return + vol_weighted_std
    
    return heuristics_matrix
