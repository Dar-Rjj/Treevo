def heuristics_v2(df):
    # Calculate the 20-day Simple Moving Average (SMA)
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate the standard deviation over the same period
    std_dev = df['close'].rolling(window=20).std()
    
    # Calculate the Bollinger Bands width
    bollinger_width = 2 * std_dev / sma
    
    # Calculate the Chaikin Money Flow (CMF) for a 21-day period
    multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Combine Bollinger Bands width and CMF into a single heuristic measure
    heuristics_matrix = (bollinger_width + cmf) / 2
    
    return heuristics_matrix
