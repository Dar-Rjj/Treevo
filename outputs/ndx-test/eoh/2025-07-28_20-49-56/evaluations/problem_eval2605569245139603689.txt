def heuristics_v2(df):
    # Calculate the 21-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=21)
    
    # Calculate the 60-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev = daily_returns.rolling(window=60).std()
    
    # Combine ROC and standard deviation into a single heuristics measure
    heuristics_matrix = (roc + std_dev) / 2
    
    return heuristics_matrix
