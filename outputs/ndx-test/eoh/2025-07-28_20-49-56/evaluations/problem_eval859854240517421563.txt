def heuristics_v2(df):
    # Calculate the 50-day and 200-day moving averages for the closing price
    short_term_ma = df['close'].rolling(window=50).mean()
    long_term_ma = df['close'].rolling(window=200).mean()

    # Calculate the volatility using the standard deviation of the last 30 days
    volatility = df['close'].rolling(window=30).std()

    # Calculate the momentum as the difference between the current close and the close 60 days ago
    momentum = df['close'] - df['close'].shift(60)

    # Adjust the momentum by the volatility
    adjusted_momentum = momentum / volatility

    # Combine the MA difference and the adjusted momentum
    ma_diff = short_term_ma - long_term_ma
    heuristics_matrix = (ma_diff + adjusted_momentum) / 2

    return heuristics_matrix
