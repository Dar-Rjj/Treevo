def heuristics_v2(df):
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    high_30 = df['high'].rolling(window=30).max()
    low_30 = df['low'].rolling(window=30).min()
    close_to_high = df['close'] / high_30
    close_to_low = df['close'] / low_30
    sma_diff = sma_50 - sma_200
    heuristics_matrix = (sma_diff + (close_to_high - close_to_low)) / 2
    return heuristics_matrix
