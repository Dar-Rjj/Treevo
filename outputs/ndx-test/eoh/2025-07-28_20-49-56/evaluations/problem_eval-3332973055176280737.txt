import pandas as pd

def heuristics_v2(df):
    # Calculate the smoothed Percentage Price Oscillator (PPO)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    ppo_line_raw = ((short_term_ema - long_term_ema) / long_term_ema) * 100
    ppo_line_smoothed = ppo_line_raw.ewm(span=9, adjust=False).mean()
    
    # Calculate the modified Money Flow Index (MFI) using EMA
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).ewm(span=14, adjust=False).mean()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).ewm(span=14, adjust=False).mean()
    mfi_ratio = positive_money_flow / negative_money_flow
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Combine PPO and MFI into a single heuristics measure with different weights
    heuristics_matrix = (ppo_line_smoothed * 0.7 + mfi * 0.3)
    return heuristics_matrix
