def heuristics_v2(df):
    # Calculate the 50-day percentage change in volume
    volume_change = df['volume'].pct_change(50)
    
    # Calculate the 21-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    volatilty = daily_returns.rolling(window=21).std()
    
    # Combine the volume change and volatility
    heuristics_matrix = (volume_change + volatilty) / 2

    return heuristics_matrix
