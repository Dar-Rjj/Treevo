import pandas as pd

def heuristics_v2(df):
    # Calculate short and long term exponential moving averages for the closing price
    short_term_ema = df['close'].ewm(span=15, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=40, adjust=False).mean()

    # Calculate the modified RSI using the median of open, high, low, and close prices
    med_price = df[['open', 'high', 'low', 'close']].median(axis=1)
    delta = med_price.diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Combine the EMA difference and the modified RSI
    ema_diff = short_term_ema - long_term_ema
    heuristics_matrix = (ema_diff + rsi) / 2

    return heuristics_matrix
