def heuristics_v2(df):
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    ppo_line = ((short_term_ema - long_term_ema) / long_term_ema) * 100
    ppo_smoothed = ppo_line.ewm(span=9, adjust=False).mean().ewm(span=9, adjust=False).mean()
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio_10 = positive_money_flow.rolling(window=10).sum() / negative_money_flow.rolling(window=10).sum()
    mfi_ratio_20 = positive_money_flow.rolling(window=20).sum() / negative_money_flow.rolling(window=20).sum()
    mfi_smoothed = (mfi_ratio_10.ewm(span=9, adjust=False).mean() * 0.6 + mfi_ratio_20.ewm(span=9, adjust=False).mean() * 0.4)
    heuristics_matrix = (ppo_smoothed * 0.8 + mfi_smoothed * 0.2)
    return heuristics_matrix
