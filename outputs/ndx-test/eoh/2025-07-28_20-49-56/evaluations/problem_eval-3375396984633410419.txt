import pandas as pd

def heuristics_v2(df):
    # Calculate the 12-day EMA and 26-day EMA for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the MACD line
    macd_line = ema_12 - ema_26
    
    # Calculate the 9-day EMA of the MACD line for signal line
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Smoothed MACD by subtracting the signal line from the MACD line
    smoothed_macd = macd_line - signal_line
    
    # Calculate the price change
    price_change = df['close'].pct_change()
    
    # Calculate the volume change
    volume_change = df['volume'].pct_change()
    
    # Compute the ratio of price change to volume change
    price_vol_ratio = (price_change / volume_change).fillna(0)
    
    # Combine the smoothed MACD and the price-volume ratio
    heuristics_matrix = (smoothed_macd + price_vol_ratio).rename('heuristics')
    
    return heuristics_matrix
