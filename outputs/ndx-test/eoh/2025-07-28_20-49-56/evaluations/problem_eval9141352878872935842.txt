def heuristics_v2(df):
    # Calculate the 20-day Simple Moving Average (SMA)
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate the Chaikin Volatility Index (CVI) for a 10-day period
    df['H-L'] = abs(df['high'] - df['low'])
    cvi = (df['H-L'].rolling(window=10).sum() / df['H-L'].shift(10).rolling(window=10).sum()) * 100
    
    # Combine SMA and CVI into a single heuristics measure
    heuristics_matrix = (sma + cvi) / 2
    
    return heuristics_matrix
