def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 20-day period
    roc = ((df['close'] - df['close'].shift(20)) / df['close'].shift(20)) * 100
    
    # Calculate the Chaikin Money Flow (CMF) for a 20-day period
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = mf_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine ROC and CMF into a single heuristic measure with new weights
    heuristics_matrix = (0.6 * roc + 0.4 * cmf)
    
    return heuristics_matrix
