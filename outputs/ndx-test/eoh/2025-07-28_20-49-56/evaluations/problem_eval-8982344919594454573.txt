import pandas as pd

def heuristics_v2(df):
    # Calculate the True Range
    tr = df['high'] - df['low']
    tr = pd.concat([tr, (df['close'].shift(1) - df['high']).abs(), (df['close'].shift(1) - df['low']).abs()], axis=1).max(axis=1)
    
    # Calculate the Average True Range (ATR) for a 14-day period
    atr = tr.rolling(window=14).mean()
    
    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine ATR and RSI into a single heuristic measure with a weighted sum
    heuristics_matrix = (0.7 * atr + 0.3 * rsi)
    
    return heuristics_matrix
