import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Bollinger Bands and %B Indicator for a 20-day period
    ma20 = df['close'].rolling(window=20).mean()
    std20 = df['close'].rolling(window=20).std(ddof=0)
    upper_band = ma20 + (std20 * 2)
    lower_band = ma20 - (std20 * 2)
    b_percent = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Combine RSI and Bollinger Bands' %B into a single heuristic measure
    heuristics_matrix = (rsi + b_percent) / 2
    
    return heuristics_matrix
