def heuristics_v2(df):
    # Calculate the 50-day EMA for the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the 20-day standard deviation for the closing price
    std_20 = df['close'].rolling(window=20).std()
    
    # Combine the EMA and standard deviation into a single heuristics measure
    heuristics_matrix = ema_50 / std_20
    
    return heuristics_matrix
