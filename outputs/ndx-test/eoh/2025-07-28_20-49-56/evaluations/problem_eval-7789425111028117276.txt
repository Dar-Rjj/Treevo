import pandas as pd

def heuristics_v2(df):
    # Calculate the EMA for short term (9 days) and long term (21 days)
    ema_short = df['close'].ewm(span=9, adjust=False).mean()
    ema_long = df['close'].ewm(span=21, adjust=False).mean()
    
    # Generate the EMA crossover signal
    ema_signal = ema_short - ema_long
    
    # Calculate the daily trading range
    trading_range = df['high'] - df['low']
    
    # Calculate the percentage change in trading volume
    volume_change_pct = df['volume'].pct_change().fillna(0)
    
    # Combine the factors into a composite heuristics measure
    heuristics_matrix = ema_signal + trading_range + volume_change_pct
    
    return heuristics_matrix
