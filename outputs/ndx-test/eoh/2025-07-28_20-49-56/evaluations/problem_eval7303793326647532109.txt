import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum of closing prices over the last 14 days
    momentum = df['close'].pct_change(periods=14).fillna(0)
    
    # Calculate the daily returns
    daily_returns = df['close'].pct_change().fillna(0)
    
    # Compute the volatility as the standard deviation of daily returns over the last 30 days
    volatility = daily_returns.rolling(window=30).std().fillna(0)
    
    # Calculate the Money Flow Multiplier
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    
    # Calculate the Money Flow Volume
    mf_volume = mfm * df['volume']
    
    # Calculate the Accumulation/Distribution Line
    adl = mf_volume.cumsum().fillna(0)
    
    # Combine the factors into a composite heuristics measure
    heuristics_matrix = (momentum + volatility + adl).rename('heuristics')
    
    return heuristics_matrix
