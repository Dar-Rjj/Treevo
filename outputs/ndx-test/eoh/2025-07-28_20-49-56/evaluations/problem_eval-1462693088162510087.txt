import pandas as pd

def heuristics_v2(df):
    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Calculate the Chaikin Money Flow (CMF) for a 21-day period
    mfv = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    cmf = mfv.rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Combine OBV and CMF into a single heuristic measure
    heuristics_matrix = (obv + cmf) / 2
    
    return heuristics_matrix
