import pandas as pd

def heuristics_v2(df):
    # Calculate the Average True Range (ATR) for a 14-day period
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].rolling(window=14).mean()

    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 14-day rolling average volume
    avg_volume = df['volume'].rolling(window=14).mean()

    # Combine ATR and RSI into a single heuristic measure using geometric mean
    heuristics_matrix = (atr * rsi) ** 0.5 * avg_volume

    return heuristics_matrix
