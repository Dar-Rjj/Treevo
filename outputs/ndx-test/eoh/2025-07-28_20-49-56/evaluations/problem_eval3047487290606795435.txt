def heuristics_v2(df):
    # Calculate the 5-day and 20-day EMA of the closing price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Compute the difference between the two EMAs
    ema_diff = ema_5 - ema_20

    # Calculate the 10-day moving average of the volume
    avg_volume = df['volume'].rolling(window=10).mean()

    # Compute the ratio of current volume to the 10-day average volume
    volume_ratio = df['volume'] / avg_volume

    # Combine the EMA difference and volume ratio into a composite score
    heuristics_matrix = 0.7 * ema_diff + 0.3 * volume_ratio

    return heuristics_matrix
