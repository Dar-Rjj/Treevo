def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 14-day period
    roc = ((df['close'] - df['close'].shift(14)) / df['close'].shift(14)) * 100
    
    # Calculate the Chaikin Money Flow (CMF) for a 14-day period
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = mf_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Combine ROC and CMF into a single heuristic measure with weights
    heuristics_matrix = (0.5 * roc + 0.5 * cmf)
    
    return heuristics_matrix
