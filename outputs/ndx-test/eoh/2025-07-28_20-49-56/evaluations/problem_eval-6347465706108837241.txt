import pandas as pd

def heuristics_v2(df):
    # Calculate the Average True Range (ATR) for a 14-day period
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].ewm(span=14, adjust=False).mean()  # Using EMA for smoothing
    
    # Calculate the Money Flow Index (MFI) for a 14-day period with EMA
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio = positive_money_flow.ewm(span=14, adjust=False).mean() / negative_money_flow.ewm(span=14, adjust=False).mean()
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Calculate the Relative Strength Index (RSI) for a 14-day period with EMA
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.ewm(span=14, adjust=False).mean()
    avg_loss = loss.ewm(span=14, adjust=False).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the recent volatility (standard deviation of returns over the last 14 days)
    recent_volatility = df['close'].pct_change().rolling(window=14).std()
    
    # Adjusted weights based on recent volatility
    w_atr = 0.5 + 0.2 * (recent_volatility / recent_volatility.max())
    w_mfi = 0.3 - 0.1 * (recent_volatility / recent_volatility.max())
    w_rsi = 0.2
    
    # Combine ATR, MFI, and RSI into a single heuristic measure with adjusted weights
    heuristics_matrix = (w_atr * atr + w_mfi * mfi + w_rsi * rsi)
    
    return heuristics_matrix
