def heuristics_v2(df):
    # Calculate the Average True Range (ATR) for a 14-day period
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].rolling(window=14).mean()
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the modified MFI for a 7-day period, adjusting by the average daily return
    mfi_ratio = (positive_money_flow.rolling(window=7).sum() / negative_money_flow.rolling(window=7).sum())
    mfi = 100 - (100 / (1 + mfi_ratio))
    avg_return = (df['close'] / df['close'].shift(1) - 1).rolling(window=7).mean()
    mfi_adjusted = mfi + avg_return * 100
    
    # Combine ATR and adjusted MFI into a single heuristic measure
    heuristics_matrix = (atr + mfi_adjusted) / 2
    
    return heuristics_matrix
