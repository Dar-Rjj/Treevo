import pandas as pd

def heuristics_v2(df):
    # Calculate 25-day and 100-day EMA for the closing price
    ema_25 = df['close'].ewm(span=25, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()

    # Calculate the 30-day average volume
    avg_volume_30 = df['volume'].rolling(window=30).mean()

    # Calculate the ratio of the current day's volume to the 30-day average volume
    volume_ratio = df['volume'] / avg_volume_30

    # Calculate the EMA difference
    ema_diff = ema_25 - ema_100

    # Adjust weights based on recent volatility
    vol_ema_diff = ema_diff.rolling(window=5).std()
    vol_volume_ratio = volume_ratio.rolling(window=5).std()
    total_volatility = vol_ema_diff + vol_volume_ratio
    weight_ema_diff = vol_ema_diff / total_volatility
    weight_volume_ratio = vol_volume_ratio / total_volatility

    # Combine the EMA difference and the volume ratio with adjusted weights
    heuristics_matrix = (weight_ema_diff * ema_diff + weight_volume_ratio * volume_ratio)

    return heuristics_matrix
