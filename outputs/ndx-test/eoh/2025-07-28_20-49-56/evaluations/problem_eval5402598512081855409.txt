def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Accumulation/Distribution Line (ADL)
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (mfm * df['volume']).cumsum()
    
    # Combine RSI and ADL into a single heuristic measure
    heuristics_matrix = 0.5 * rsi + 0.5 * adl
    
    return heuristics_matrix
