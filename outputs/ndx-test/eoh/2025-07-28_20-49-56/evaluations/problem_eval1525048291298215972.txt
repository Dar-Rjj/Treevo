def heuristics_v2(df):
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    sma_20_close = df['close'].rolling(window=20).mean()
    momentum = (ema_50 - ema_200) / ema_200
    close_diff = df['close'] - sma_20_close
    heuristics_matrix = (momentum + volume_ratio + close_diff) / 3
    return heuristics_matrix
