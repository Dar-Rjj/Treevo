import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] + \
          (df['close'] < df['close'].shift(1)).astype(int) * -df['volume']
    obv = obv.cumsum()
    
    # Combine RSI and OBV into a single heuristic measure
    heuristics_matrix = (rsi + obv) / 2
    
    return heuristics_matrix
