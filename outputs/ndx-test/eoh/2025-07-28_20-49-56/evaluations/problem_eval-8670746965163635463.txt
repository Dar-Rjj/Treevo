import pandas as pd

def heuristics_v2(df):
    # Calculate the Detrended Price Oscillator (DPO)
    dpo = df['close'].shift(int(20/2)+1) - df['close'].rolling(window=20).mean()
    
    # Calculate the Percentage Price Oscillator (PPO)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    ppo = ((short_term_ema - long_term_ema) / long_term_ema) * 100
    
    # Combine DPO and PPO into a single heuristic measure
    heuristics_matrix = (dpo + ppo) / 2
    
    return heuristics_matrix
