def heuristics_v2(df):
    # Calculate 50-day and 200-day EMA for the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()

    # Calculate the ratio of the current day's volume to the 20-day average volume
    volume_ratio = df['volume'] / avg_volume_20

    # Calculate a simple 10-day momentum (to represent short term trend)
    momentum = df['close'] - df['close'].shift(10)

    # Weight the volume ratio with the momentum
    weighted_volume_ratio = volume_ratio * (momentum / df['close'])

    # Combine the EMA difference and the weighted volume ratio
    ema_diff = ema_50 - ema_200
    heuristics_matrix = (ema_diff + weighted_volume_ratio) / 2

    return heuristics_matrix
