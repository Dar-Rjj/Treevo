def heuristics_v2(df):
    # Calculate the RSI for a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ROC for a 20-day period
    roc = ((df['close'] - df['close'].shift(20)) / df['close'].shift(20)) * 100
    
    # Combine RSI and ROC into a single heuristics measure with specific weights
    heuristics_matrix = (rsi * 0.7 + roc * 0.3)
    return heuristics_matrix
