def heuristics_v2(df):
    # Calculate the Percentage Price Oscillator (PPO) with different spans
    short_term_ema = df['close'].ewm(span=9, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=21, adjust=False).mean()
    ppo_line = ((short_term_ema - long_term_ema) / long_term_ema) * 100
    
    # Calculate the Money Flow Index (MFI) with a different rolling window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio = positive_money_flow.rolling(window=7).sum() / negative_money_flow.rolling(window=7).sum()
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Combine adjusted PPO and MFI into a single heuristics measure
    heuristics_matrix = (ppo_line + mfi) / 2
    return heuristics_matrix
