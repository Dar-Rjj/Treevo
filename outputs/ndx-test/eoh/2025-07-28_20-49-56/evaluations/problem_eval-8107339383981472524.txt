def heuristics_v2(df):
    # Calculate the short and long term EMAs for the closing price
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()

    # Calculate the 5-period EMA for the closing price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate the difference between the closing price and the 5-period EMA
    diff_price_ema5 = df['close'] - ema_5

    # Calculate the stochastic oscillator components
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k = 100 * (diff_price_ema5 - low_14) / (high_14 - low_14)
    
    # Combine the EMA crossover and the stochastic oscillator
    ema_crossover = short_term_ema - long_term_ema
    heuristics_matrix = (ema_crossover + k) / 2

    return heuristics_matrix
