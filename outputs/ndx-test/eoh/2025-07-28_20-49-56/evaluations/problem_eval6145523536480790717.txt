import pandas as pd

def heuristics_v2(df):
    # Calculate the Percentage Price Oscillator (PPO)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    ppo_line = ((short_term_ema - long_term_ema) / long_term_ema) * 100
    
    # Calculate the Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine PPO, MFI, and RSI into a single heuristics measure with different weights
    heuristics_matrix = (ppo_line * 0.5 + mfi * 0.3 + rsi * 0.2)
    return heuristics_matrix
