def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = ((df['close'] - df['close'].shift(10)) / df['close'].shift(10)) * 100
    
    # Calculate the Accumulation/Distribution Line (ADL)
    adl = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    adl = adl * df['volume']
    adl = adl.cumsum()
    
    # Combine ROC and ADL into a single heuristics measure
    heuristics_matrix = (roc + adl) / 2
    return heuristics_matrix
