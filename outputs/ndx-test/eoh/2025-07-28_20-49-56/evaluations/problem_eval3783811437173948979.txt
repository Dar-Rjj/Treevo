import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    short_ma = df['close'].rolling(window=10).mean()
    long_ma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between short and long term MAs
    ma_diff = short_ma - long_ma
    
    # Compute the daily change in volume
    volume_change = df['volume'].pct_change()
    
    # Combine the two signals into a single heuristic
    heuristics_matrix = (ma_diff + volume_change).fillna(0)
    
    return heuristics_matrix
