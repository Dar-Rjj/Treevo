def heuristics_v2(df):
    # Calculate 14-day and 28-day EMA for the closing price
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    ema_28 = df['close'].ewm(span=28, adjust=False).mean()

    # Calculate the 5-day average volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()

    # Calculate the ratio of the current day's volume to the 5-day average volume
    volume_ratio = df['volume'] / avg_volume_5

    # Calculate the standard deviation of the closing prices over the last 10 days
    std_dev_10 = df['close'].rolling(window=10).std()

    # Calculate the percentage change between the 14-day and 28-day EMAs
    ema_percentage_change = (ema_14 - ema_28) / ema_28

    # Adjust the standard deviation by the volume ratio
    adjusted_std_dev = std_dev_10 * volume_ratio

    # Combine the EMA percentage change and the adjusted standard deviation
    heuristics_matrix = (ema_percentage_change + adjusted_std_dev) / 2

    return heuristics_matrix
