import pandas as pd

def heuristics_v2(df):
    # Calculate the Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Calculate the Percentage Price Oscillator (PPO) for 12-day and 26-day EMA
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    ppo = (ema_12 - ema_26) / ema_26 * 100
    
    # Calculate the Average True Range (ATR) for a 14-day period
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].rolling(window=14).mean()
    
    # Combine ATR, ADL, and PPO into a single heuristic measure with weights
    heuristics_matrix = (0.3 * atr + 0.4 * adl + 0.3 * ppo)
    
    return heuristics_matrix
