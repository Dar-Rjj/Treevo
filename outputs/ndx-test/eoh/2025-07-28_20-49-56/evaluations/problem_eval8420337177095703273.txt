def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 14-day period
    roc = ((df['close'] - df['close'].shift(14)) / df['close'].shift(14)) * 100
    
    # Calculate the Chande Momentum Oscillator (CMO) for a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    price_change = typical_price.diff()
    up_change = price_change.where(price_change > 0, 0)
    down_change = -price_change.where(price_change < 0, 0)
    cmo = ((up_change.rolling(window=14).sum() - down_change.rolling(window=14).sum()) 
           / (up_change.rolling(window=14).sum() + down_change.rolling(window=14).sum())) * 100
    
    # Combine ROC and CMO into a single heuristic measure
    heuristics_matrix = (roc + cmo) / 2
    
    return heuristics_matrix
