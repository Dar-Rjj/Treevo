import pandas as pd

def heuristics_v2(df):
    # Calculate the short-term and long-term weighted moving averages
    short_wma = df['close'].ewm(span=10, adjust=False).mean()
    long_wma = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the price change
    price_change = df['close'].pct_change()
    
    # Calculate the volume change
    volume_change = df['volume'].pct_change()
    
    # Compute the ratio of price change to volume change
    price_vol_ratio = (price_change / volume_change).fillna(0)
    
    # Combine the WMA difference and the price-volume ratio
    heuristics_matrix = (short_wma - long_wma + price_vol_ratio).rename('heuristics')
    
    return heuristics_matrix
