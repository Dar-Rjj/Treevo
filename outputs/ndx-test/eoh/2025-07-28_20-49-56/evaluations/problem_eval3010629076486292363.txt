def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for the closing price over a 25-day period
    roc = df['close'].pct_change(periods=25) * 100

    # Calculate the Money Flow Multiplier
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])

    # Calculate the Money Flow Volume
    mf_volume = mfm * df['volume']

    # Calculate the short and long term EMA of the Money Flow Volume
    ema_14 = mf_volume.ewm(span=14, adjust=False).mean()
    ema_29 = mf_volume.ewm(span=29, adjust=False).mean()

    # Calculate the Accumulation/Distribution Oscillator (ADOSC)
    adosc = ema_14 - ema_29

    # Combine the ROC and ADOSC using a weighted average
    heuristics_matrix = (roc * 0.7 + adosc * 0.3)

    return heuristics_matrix
