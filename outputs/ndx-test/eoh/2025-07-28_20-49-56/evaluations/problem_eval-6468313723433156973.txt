import pandas as pd

def heuristics_v2(df):
    def momentum(df, n=20):
        return df['close'].pct_change(n)

    def rsi(df, n=14):
        delta = df['close'].diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def volatility(df, n=30):
        return df['close'].pct_change().rolling(window=n).std()

    df['momentum'] = momentum(df)
    df['rsi'] = rsi(df)
    df['volatility'] = volatility(df)

    heuristics_matrix = (df['momentum'] * 0.5 + df['rsi'] * 0.3 + df['volatility'] * 0.2).rename('heuristics')

    return heuristics_matrix
