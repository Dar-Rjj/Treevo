def heuristics_v2(df):
    # Calculate short and long term moving averages for the volume
    short_term_vol_ma = df['volume'].rolling(window=50).mean()
    long_term_vol_ma = df['volume'].rolling(window=200).mean()

    # Calculate the modified Stochastic Oscillator using the median of high, low, and close prices
    median_price = df[['high', 'low', 'close']].median(axis=1)
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k = 100 * (median_price - lowest_low) / (highest_high - lowest_low)

    # Combine the volume ratio and the modified Stochastic Oscillator
    vol_ratio = short_term_vol_ma / long_term_vol_ma
    heuristics_matrix = (vol_ratio + k) / 2

    return heuristics_matrix
