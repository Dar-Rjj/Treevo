def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'].diff()

    # Compute the 10-day Z-score of the price change for short-term momentum
    z_score = (df['price_change'] - df['price_change'].rolling(window=10).mean()) / df['price_change'].rolling(window=10).std()

    # Count the number of positive and negative trading days over the last 30 days
    positive_days = (df['close'] > df['close'].shift(1)).rolling(window=30).sum()
    negative_days = (df['close'] < df['close'].shift(1)).rolling(window=30).sum()
    
    # Calculate the ratio of positive to negative trading days
    pos_neg_ratio = positive_days / (negative_days + 1e-6)  # Adding small value to avoid division by zero

    # Combine the Z-score and the positive to negative day ratio
    heuristics_matrix = z_score + pos_neg_ratio

    return heuristics_matrix
