def heuristics_v2(df):
    # Calculate 14-day percentage price change
    price_change = (df['close'] - df['close'].shift(14)) / df['close'].shift(14) * 100

    # Calculate the 14-day high and low prices
    high_14 = df['high'].rolling(window=14).max()
    low_14 = df['low'].rolling(window=14).min()

    # Calculate the modified Stochastic Oscillator
    k = 100 * (df['close'] - low_14) / (high_14 - low_14)

    # Combine the percentage price change and the modified Stochastic Oscillator
    heuristics_matrix = (price_change + k) / 2

    return heuristics_matrix
