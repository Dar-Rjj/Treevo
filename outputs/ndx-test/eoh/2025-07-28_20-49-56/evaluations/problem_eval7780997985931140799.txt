def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 14-day period
    roc = (df['close'] - df['close'].shift(14)) / df['close'].shift(14) * 100
    
    # Calculate the Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - low_14) / (high_14 - low_14)
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Combine ROC, Stochastic Oscillator, and OBV into a single heuristic measure with weights
    heuristics_matrix = (0.4 * roc + 0.3 * k_percent + 0.3 * obv)
    
    return heuristics_matrix
