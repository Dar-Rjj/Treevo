def heuristics_v2(df):
    # Calculate the 10-day percentage change in the closing price
    pct_change_10 = df['close'].pct_change(10)
    
    # Calculate the 20-day standard deviation of daily returns for volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Calculate the ratio of the current day's high to low prices
    hl_ratio = df['high'] / df['low']
    
    # Combine the three factors
    heuristics_matrix = (pct_change_10 + volatility + hl_ratio) / 3
    
    return heuristics_matrix
