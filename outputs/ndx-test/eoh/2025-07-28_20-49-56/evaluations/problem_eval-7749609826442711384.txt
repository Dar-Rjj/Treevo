def heuristics_v2(df):
    # Calculate 30-day and 150-day EMA for the closing price
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_150 = df['close'].ewm(span=150, adjust=False).mean()

    # Calculate the 10-day average volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()

    # Calculate the ratio of the current day's volume to the 10-day average volume
    volume_ratio = df['volume'] / avg_volume_10

    # Calculate a simple 20-day momentum (to represent medium term trend)
    momentum = df['close'] - df['close'].shift(20)

    # Weight the volume ratio with the momentum
    weighted_volume_ratio = volume_ratio * (momentum / df['close'])

    # Combine the EMA difference and the weighted volume ratio
    ema_diff = ema_30 - ema_150
    heuristics_matrix = (ema_diff + weighted_volume_ratio) / 2

    return heuristics_matrix
