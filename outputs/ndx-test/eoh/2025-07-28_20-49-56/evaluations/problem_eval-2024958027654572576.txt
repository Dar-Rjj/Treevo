def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 14-day period
    roc = ((df['close'] - df['close'].shift(14)) / df['close'].shift(14)) * 100
    
    # Calculate the Ease of Movement (EOM) for a 14-day period
    distance_moved = (df['high'] + df['low']) / 2 - (df['high'].shift(1) + df['low'].shift(1)) / 2
    box_ratio = (df['volume'] / 1e6) / (df['high'] - df['low'])
    eom = distance_moved / box_ratio
    eom_smoothed = eom.rolling(window=14).mean()
    
    # Combine ROC and EOM into a single heuristic measure
    heuristics_matrix = (roc + eom_smoothed) / 2
    
    return heuristics_matrix
