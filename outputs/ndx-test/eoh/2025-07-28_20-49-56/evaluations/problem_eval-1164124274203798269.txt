def heuristics_v2(df):
    # Calculate 14-day and 28-day ROC for the closing price
    roc_14 = (df['close'] - df['close'].shift(14)) / df['close'].shift(14) * 100
    roc_28 = (df['close'] - df['close'].shift(28)) / df['close'].shift(28) * 100

    # Calculate the 50-day average volume
    avg_volume_50 = df['volume'].rolling(window=50).mean()

    # Calculate the ratio of the current day's volume to the 50-day average volume
    volume_ratio = df['volume'] / avg_volume_50

    # Combine the weighted sum of ROCs and the volume ratio
    weighted_roc = (roc_14 * 0.6 + roc_28 * 0.4)
    heuristics_matrix = (weighted_roc + volume_ratio) / 2

    return heuristics_matrix
