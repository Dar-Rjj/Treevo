def heuristics_v2(df):
    roc = df['close'].pct_change(14) * 100
    positive_volume = df['volume'].where(df['close'] > df['close'].shift(1), 0)
    negative_volume = df['volume'].where(df['close'] < df['close'].shift(1), 0)
    cvd = (positive_volume - negative_volume).cumsum()
    heuristics_matrix = (roc * 0.5 + cvd * 0.5)
    return heuristics_matrix
