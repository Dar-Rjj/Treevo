import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 12-day period
    roc = ((df['close'] - df['close'].shift(12)) / df['close'].shift(12)) * 100
    
    # Calculate the 20-day Simple Moving Average (SMA)
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day Standard Deviation
    std = df['close'].rolling(window=20).std()
    
    # Calculate the Bollinger Bands Width (BBW)
    bbw = (2 * std) / sma
    
    # Combine ROC and BBW into a single heuristics measure
    heuristics_matrix = (roc + bbw) / 2
    
    return heuristics_matrix
