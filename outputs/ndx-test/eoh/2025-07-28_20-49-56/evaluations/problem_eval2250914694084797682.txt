def heuristics_v2(df):
    # Calculate the momentum of the close price
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the change in volume
    volume_change = df['volume'].pct_change(periods=5)
    
    # Combine the two factors
    combined_factor = (momentum + volume_change) / 2
    
    # Apply a rolling window to smooth the factor values
    heuristics_matrix = combined_factor.rolling(window=5).mean()
    
    return heuristics_matrix
