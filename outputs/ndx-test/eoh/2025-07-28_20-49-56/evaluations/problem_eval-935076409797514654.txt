import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = (df['close'] - df['close'].shift(12)) / df['close'].shift(12) * 100
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Combine ROC and OBV into a single heuristics measure with weights
    heuristics_matrix = (roc * 0.6 + obv * 0.4)
    
    return heuristics_matrix
