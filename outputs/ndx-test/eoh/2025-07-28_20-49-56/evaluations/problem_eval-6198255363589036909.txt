def heuristics_v2(df):
    # Calculate the short-term and long-term Exponential Moving Averages (EMA)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the EMA crossover signal
    ema_crossover = short_term_ema - long_term_ema
    
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the volatility as the standard deviation of daily returns over a 14-day window
    volatility = daily_returns.rolling(window=14).std()
    
    # Calculate the volatility-adjusted return
    volatility_adjusted_return = daily_returns / volatility
    
    # Combine the EMA crossover signal and the volatility-adjusted return into a single heuristics measure
    heuristics_matrix = ema_crossover + volatility_adjusted_return
    
    return heuristics_matrix
