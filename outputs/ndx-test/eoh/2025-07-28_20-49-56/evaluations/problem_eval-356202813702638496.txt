def heuristics_v2(df):
    # Calculate 50-day and 200-day SMA for the closing price
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()

    # Calculate the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()

    # Calculate the ratio of the current day's volume to the 20-day average volume
    volume_ratio = df['volume'] / avg_volume_20

    # Calculate the difference between the 50-day and 200-day SMAs
    sma_diff = sma_50 - sma_200

    # Calculate the 10-day rate of change for the closing price
    roc_10 = df['close'].pct_change(periods=10)

    # Combine the SMA difference, the volume ratio, and the 10-day rate of change
    heuristics_matrix = (sma_diff + volume_ratio + roc_10) / 3

    return heuristics_matrix
