import pandas as pd

def heuristics_v2(df):
    # Calculate the Average True Range (ATR) for a 14-day period
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].rolling(window=14).mean()
    
    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine ATR and RSI into a single heuristic measure
    heuristics_matrix = (atr + rsi) / 2
    
    return heuristics_matrix
