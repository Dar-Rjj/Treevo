import pandas as pd

def heuristics_v2(df):
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the 20-day standard deviation of daily returns for volatility
    volatility = daily_returns.rolling(window=20).std()
    
    # Calculate the 20-day average volume
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Compute the relative volume
    relative_volume = df['volume'] / avg_volume
    
    # Combine volatility and relative volume into a single heuristics measure
    heuristics_matrix = (volatility + relative_volume) / 2
    
    return heuristics_matrix
