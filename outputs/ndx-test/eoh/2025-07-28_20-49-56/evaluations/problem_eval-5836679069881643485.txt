import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day percentage change in closing price
    pct_change_30 = df['close'].pct_change(periods=30)
    
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the percentage change and RSI into a single heuristic measure
    heuristics_matrix = (pct_change_30 + rsi) / 2
    
    return heuristics_matrix
