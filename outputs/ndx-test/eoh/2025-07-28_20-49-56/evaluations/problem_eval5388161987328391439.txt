import pandas as pd

def heuristics_v2(df):
    # Calculate the Williams %R for a 14-day period
    highest_high = df['high'].rolling(window=14).max()
    lowest_low = df['low'].rolling(window=14).min()
    williams_r = -100 * ((highest_high - df['close']) / (highest_high - lowest_low))
    
    # Calculate the On-Balance Volume (OBV)
    obv = df['volume'].copy()
    obv[df['close'] < df['close'].shift(1)] = -df['volume'][df['close'] < df['close'].shift(1)]
    obv = obv.cumsum()
    
    # Combine Williams %R and OBV into a single heuristics measure
    heuristics_matrix = (williams_r + obv) / 2
    
    return heuristics_matrix
