def heuristics_v2(df):
    # Calculate the 14-day rate of change (ROC) for the closing price
    roc = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Calculate the 50-day average volume
    avg_volume_50 = df['volume'].rolling(window=50).mean()
    
    # Calculate the ratio of the current day's volume to the 50-day average volume
    volume_ratio = df['volume'] / avg_volume_50
    
    # Combine the ROC and the volume ratio
    heuristics_matrix = (roc + volume_ratio) / 2
    
    return heuristics_matrix
