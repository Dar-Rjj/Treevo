def heuristics_v2(df):
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_250 = df['close'].ewm(span=250, adjust=False).mean()
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30
    ema_diff = ema_100 - ema_250
    heuristics_matrix = (ema_diff + volume_ratio) / 2
    return heuristics_matrix
