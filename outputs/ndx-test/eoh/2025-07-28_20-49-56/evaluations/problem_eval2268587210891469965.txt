import pandas as pd

def heuristics_v2(df):
    # Calculate the Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Calculate the Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.ewm(span=9, adjust=False).mean()
    macd_histogram = macd - signal
    
    # Combine ADL and MACD histogram into a single heuristics measure
    heuristics_matrix = (adl + macd_histogram) / 2
    return heuristics_matrix
