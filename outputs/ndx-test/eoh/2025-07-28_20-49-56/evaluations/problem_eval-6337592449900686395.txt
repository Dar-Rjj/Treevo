def heuristics_v2(df):
    def directional_movement(df, n=14):
        return (df['close'] - df['close'].shift(n)) / df['close'].shift(n)

    def rsi_adjusted_volatility(df, n=14):
        delta = df['close'].diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        volatility = df['close'].pct_change().rolling(window=n).std()
        return rsi * volatility

    df['directional_movement'] = directional_movement(df)
    df['rsi_adjusted_volatility'] = rsi_adjusted_volatility(df)

    heuristics_matrix = (df['directional_movement'] + df['rsi_adjusted_volatility']) / 2

    return heuristics_matrix
