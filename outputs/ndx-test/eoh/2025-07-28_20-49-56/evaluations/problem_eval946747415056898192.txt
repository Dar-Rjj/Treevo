def heuristics_v2(df):
    # Calculate the 21-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=21)
    
    # Calculate the Money Flow Multiplier
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    
    # Calculate the Money Flow Volume
    mf_volume = mfm * df['volume']
    
    # Calculate the 20-day Chaikin Money Flow (CMF)
    cmf = mf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine ROC and CMF into a single heuristic measure
    heuristics_matrix = (roc + cmf) / 2
    
    return heuristics_matrix
