def heuristics_v2(df):
    # Calculate 10-day, 50-day, and 200-day EMA for the closing price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Calculate the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()

    # Calculate the ratio of the current day's volume to the 20-day average volume
    volume_ratio = df['volume'] / avg_volume_20

    # Combine the EMA differences and the volume ratio
    ema_diff_1 = ema_10 - ema_50
    ema_diff_2 = ema_50 - ema_200
    heuristics_matrix = (ema_diff_1 + ema_diff_2 + 2 * volume_ratio) / 4

    return heuristics_matrix
