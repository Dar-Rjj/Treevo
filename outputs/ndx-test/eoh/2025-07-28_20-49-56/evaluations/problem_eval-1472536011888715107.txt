import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Accumulation/Distribution Line (ADL)
    adl_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (adl_multiplier * df['volume']).cumsum()
    
    # Calculate the Rate of Change (ROC) for a 14-day period
    roc = df['close'].pct_change(periods=14).fillna(0)
    
    # Combine RSI, ADL, and ROC into a composite heuristics measure
    heuristics_matrix = (rsi + adl + roc) / 3
    
    return heuristics_matrix
