import pandas as pd

def heuristics_v2(df):
    # Calculate the Average True Range (ATR) for a 10-day period
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].rolling(window=10).mean()
    
    # Calculate the Money Flow Index (MFI) for a 10-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio = positive_money_flow.rolling(window=10).sum() / negative_money_flow.rolling(window=10).sum()
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Calculate the Relative Strength Index (RSI) for a 20-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine ATR, MFI, and RSI into a single heuristic measure with weights
    heuristics_matrix = (0.5 * atr + 0.3 * mfi + 0.2 * rsi)
    
    return heuristics_matrix
