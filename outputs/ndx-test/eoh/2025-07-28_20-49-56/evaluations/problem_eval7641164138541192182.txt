import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = (df['close'] - df['close'].shift(12)) / df['close'].shift(12) * 100
    
    # Calculate the Accumulation/Distribution Oscillator (ADOSC)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    adosc = money_flow_volume.rolling(window=3).mean() - money_flow_volume.rolling(window=10).mean()
    
    # Combine ROC and ADOSC into a single heuristics measure with different weights
    heuristics_matrix = (roc * 0.6 + adosc * 0.4)
    return heuristics_matrix
