def heuristics_v2(df):
    roc = (df['close'] - df['close'].shift(14)) / df['close'].shift(14) * 100
    adl = (df['close'] - df['low']) - (df['high'] - df['close']) * df['volume'] / (df['high'] - df['low'])
    adl_smoothed = adl.ewm(span=14, adjust=False).mean()
    heuristics_matrix = (roc * 0.5 + adl_smoothed * 0.5)
    return heuristics_matrix
