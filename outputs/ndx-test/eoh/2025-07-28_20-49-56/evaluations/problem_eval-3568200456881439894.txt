def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 21-day period
    roc = (df['close'] - df['close'].shift(21)) / df['close'].shift(21) * 100
    
    # Calculate the Accumulation/Distribution Line (ADL)
    adl_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (adl_multiplier * df['volume']).cumsum()
    
    # Combine ROC and ADL into a single heuristic measure
    heuristics_matrix = (roc + adl) / 2
    
    return heuristics_matrix
