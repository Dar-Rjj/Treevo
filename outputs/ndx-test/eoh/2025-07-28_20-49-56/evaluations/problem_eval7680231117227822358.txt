import pandas as pd

def heuristics_v2(df):
    # Calculate the True Range
    tr = df['high'] - df['low']
    tr = pd.concat([tr, (df['close'].shift(1) - df['high']).abs(), (df['close'].shift(1) - df['low']).abs()], axis=1).max(axis=1)
    
    # Calculate the Average True Range (ATR) for a 14-day period
    atr = tr.rolling(window=14).mean()
    
    # Calculate the 12-period and 26-period Exponential Moving Averages
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the MACD line and the 9-day EMA of the MACD line (signal line)
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Combine ATR and MACD into a single heuristic measure
    heuristics_matrix = (atr + macd_line - signal_line) / 2
    
    return heuristics_matrix
