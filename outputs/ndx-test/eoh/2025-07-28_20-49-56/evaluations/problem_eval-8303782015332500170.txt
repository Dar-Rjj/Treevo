def heuristics_v2(df):
    # Calculate the Average True Range (ATR) for a 21-day period
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].rolling(window=21).mean()
    
    # Calculate the Money Flow Index (MFI) for a 21-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio = positive_money_flow.rolling(window=21).sum() / negative_money_flow.rolling(window=21).sum()
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Combine ATR and MFI into a single heuristic measure with weight adjustment
    heuristics_matrix = (atr * 0.3 + mfi * 0.7)
    
    return heuristics_matrix
