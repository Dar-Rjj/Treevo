import pandas as pd

def heuristics_v2(df):
    # Calculate the Percentage Price Oscillator (PPO)
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    ppo_line = ((short_term_ema - long_term_ema) / long_term_ema) * 100
    
    # Calculate the 20-day Simple Moving Average (SMA)
    sma = df['close'].rolling(window=20).mean()
    sma_diff = (df['close'] - sma) / sma
    
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0).rolling(window=14).mean()
    loss = -delta.where(delta < 0, 0).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine PPO, SMA, and RSI into a single heuristics measure with different weights
    heuristics_matrix = (ppo_line * 0.2 + sma_diff * 0.3 + rsi * 0.5)
    return heuristics_matrix
