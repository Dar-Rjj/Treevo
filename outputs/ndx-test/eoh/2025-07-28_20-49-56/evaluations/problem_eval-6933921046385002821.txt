def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for a 14-day period
    roc = df['close'].pct_change(periods=14) * 100
    
    # Calculate the Accumulation/Distribution Line (A/D)
    ad_line = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    ad_line = ad_line.cumsum()
    
    # Calculate the Chaikin Oscillator (CO)
    ad_line_ema3 = ad_line.ewm(span=3, adjust=False).mean()
    ad_line_ema10 = ad_line.ewm(span=10, adjust=False).mean()
    chaikin_oscillator = ad_line_ema3 - ad_line_ema10
    
    # Combine ROC, A/D, and CO into a single heuristic measure with weights
    heuristics_matrix = (0.5 * roc + 0.3 * ad_line + 0.2 * chaikin_oscillator)
    
    return heuristics_matrix
