def heuristics_v2(df):
    # Calculate the RSI for a 10-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ROC for a 30-day period
    roc = ((df['close'] - df['close'].shift(30)) / df['close'].shift(30)) * 100
    
    # Combine RSI and ROC into a single heuristics measure with weights
    heuristics_matrix = (0.7 * rsi + 0.3 * roc)
    return heuristics_matrix
