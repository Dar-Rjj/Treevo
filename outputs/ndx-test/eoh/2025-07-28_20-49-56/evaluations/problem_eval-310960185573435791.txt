def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) of the closing price
    roc_10 = df['close'] / df['close'].shift(10) - 1

    # Calculate the 5-day and 20-day moving average of the volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    avg_volume_20 = df['volume'].rolling(window=20).mean()

    # Compute the ratio of 5-day to 20-day average volume
    volume_ratio = avg_volume_5 / avg_volume_20

    # Combine the ROC and volume ratio into a composite score
    heuristics_matrix = 0.6 * roc_10 + 0.4 * volume_ratio

    return heuristics_matrix
