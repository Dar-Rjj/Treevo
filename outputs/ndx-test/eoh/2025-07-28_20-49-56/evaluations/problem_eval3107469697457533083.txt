def heuristics_v2(df):
    # Calculate short and long term moving averages for the closing price
    short_term_ma = df['close'].rolling(window=50).mean()
    long_term_ma = df['close'].rolling(window=200).mean()

    # Calculate the modified RSI using the weighted average of high, low, and close prices
    avg_price = df['high'] * 0.2 + df['low'] * 0.3 + df['close'] * 0.5
    delta = avg_price.diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Combine the MA difference and the modified RSI
    ma_diff = short_term_ma - long_term_ma
    heuristics_matrix = (ma_diff + rsi) / 2

    return heuristics_matrix
