import pandas as pd

def heuristics_v2(df):
    # Calculate the change in closing price to represent momentum
    momentum = df['close'].pct_change()
    
    # Calculate the rolling standard deviation for volatility
    volatility = df['close'].rolling(window=30).std()
    
    # Calculate the change in volume
    volume_change = df['volume'].pct_change()
    
    # Combine the factors into a composite score
    heuristics_matrix = (momentum + volatility + volume_change) / 3
    
    return heuristics_matrix
