def heuristics_v2(df):
    # Calculate short and long term moving averages for the closing price
    short_term_ma = df['close'].rolling(window=20).mean()
    long_term_ma = df['close'].rolling(window=100).mean()

    # Calculate the modified RSI using the average of high, low, and close prices
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    delta = avg_price.diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=28).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=28).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Combine the MA difference and the modified RSI
    ma_diff = short_term_ma - long_term_ma
    heuristics_matrix = (ma_diff + rsi) / 2

    return heuristics_matrix
