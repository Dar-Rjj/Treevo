import pandas as pd

def calculate_adl(df):
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    adl = money_flow_volume.cumsum()
    return adl

def calculate_dmi(df, window):
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    tr = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    plus_dm = df['high'].diff(1).apply(lambda x: x if x > 0 else 0)
    minus_dm = -df['low'].diff(1).apply(lambda x: x if x < 0 else 0)
    
    smooth_tr = tr.rolling(window=window).sum()
    smooth_plus_dm = plus_dm.rolling(window=window).sum()
    smooth_minus_dm = minus_dm.rolling(window=window).sum()
    
    di_plus = 100 * (smooth_plus_dm / smooth_tr)
    di_minus = 100 * (smooth_minus_dm / smooth_tr)
    
    dx = 100 * (abs(di_plus - di_minus) / (di_plus + di_minus))
    adx = dx.rolling(window=window).mean()
    
    return di_plus - di_minus

def heuristics_v2(df):
    adl = calculate_adl(df)
    dmi = calculate_dmi(df, 14)
    heuristics_matrix = adl + dmi
    return heuristics_matrix
