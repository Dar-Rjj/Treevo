def heuristics_v2(df):
    # Calculate the difference in volume over a 10-day period
    vol_diff = df['volume'].diff(10).fillna(0)
    
    # Calculate the momentum based on the ratio of high price to low price over a 20-day period
    momentum = (df['high'].rolling(window=20).mean() / df['low'].rolling(window=20).mean()) - 1
    
    # Combine the volume difference and the momentum to form the heuristics matrix
    heuristics_matrix = (vol_diff + momentum) / 2
    return heuristics_matrix
