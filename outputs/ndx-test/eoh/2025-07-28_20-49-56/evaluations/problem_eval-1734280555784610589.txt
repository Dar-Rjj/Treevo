def heuristics_v2(df):
    # Calculate the 10-day simple moving average (SMA) of the closing prices
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Measure the momentum of the closing price compared to the SMA
    price_momentum = df['close'] - sma_10
    
    # Intraday volatility as the ratio of the difference between the high and low prices to the close
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Relative strength: the ratio of the difference between the closing and opening prices to the average of the high and low prices
    relative_strength = (df['close'] - df['open']) / ((df['high'] + df['low']) / 2)
    
    # Combine the factors into a composite heuristics measure
    heuristics_matrix = price_momentum + intraday_volatility + relative_strength
    
    return heuristics_matrix
