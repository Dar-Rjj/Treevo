def heuristics_v2(df):
    # Calculate short and long term exponential moving averages for the closing price
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=60, adjust=False).mean()

    # Calculate the modified RSI using the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    delta = avg_price.diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Adjust RSI by incorporating standard deviation as a measure of volatility
    std_dev = avg_price.rolling(window=14).std()
    adjusted_rsi = rsi / std_dev

    # Combine the EMA difference and the volatility-adjusted RSI
    ema_diff = short_term_ema - long_term_ema
    heuristics_matrix = (ema_diff + adjusted_rsi) / 2

    return heuristics_matrix
