import pandas as pd

def heuristics_v2(df):
    # Calculate the Stochastic Oscillator (SO)
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_line = ((df['close'] - lowest_low) / (highest_high - lowest_low)) * 100
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Combine SO and OBV into a single heuristics measure with different weights
    heuristics_matrix = (k_line * 0.6 + obv * 0.4)
    return heuristics_matrix
