def heuristics_v2(df):
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    std_close_20 = df['close'].rolling(window=20).std()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / avg_volume_20
    adjusted_volume_ratio = volume_ratio * (ema_50.diff() / std_close_20)
    heuristics_matrix = adjusted_volume_ratio + ema_50.diff()
    return heuristics_matrix
