def heuristics_v2(df):
    df['H-L'] = abs(df['high'] - df['low'])
    df['H-PC'] = abs(df['high'] - df['close'].shift(1))
    df['L-PC'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['H-L', 'H-PC', 'L-PC']].max(axis=1)
    atr = df['TR'].rolling(window=14).mean()

    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio = positive_money_flow.rolling(window=20).sum() / negative_money_flow.rolling(window=20).sum()
    mfi = 100 - (100 / (1 + mfi_ratio))

    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    std_close = df['close'].rolling(window=20).std()
    weight_atr = 0.4 * (1 + std_close / std_close.max())
    weight_mfi = 0.3 * (1 + std_close / std_close.max())
    weight_rsi = 0.3 * (1 + std_close / std_close.max())

    heuristics_matrix = (weight_atr * atr + weight_mfi * mfi + weight_rsi * rsi)

    return heuristics_matrix
