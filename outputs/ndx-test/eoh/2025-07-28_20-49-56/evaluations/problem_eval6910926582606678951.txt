def heuristics_v2(df):
    # Calculate the daily percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Compute an interaction factor
    price_interact = (df['close'] / ((df['high'] + df['low']) / 2)) * (1 + volume_change)
    
    # Form the heuristics matrix
    heuristics_matrix = price_interact
    
    return heuristics_matrix
