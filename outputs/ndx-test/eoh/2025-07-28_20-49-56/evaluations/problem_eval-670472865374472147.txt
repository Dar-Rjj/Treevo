def heuristics_v2(df):
    short_term_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=26, adjust=False).mean()
    ppo_line = ((short_term_ema - long_term_ema) / long_term_ema) * 100
    ppo_smoothed = ppo_line.ewm(span=9, adjust=False).mean()
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio_14 = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi_ratio_28 = positive_money_flow.rolling(window=28).sum() / negative_money_flow.rolling(window=28).sum()
    mfi_smoothed = (mfi_ratio_14.ewm(span=9, adjust=False).mean() * 0.5 + mfi_ratio_28.ewm(span=9, adjust=False).mean() * 0.5)
    heuristics_matrix = (ppo_smoothed * 0.7 + mfi_smoothed * 0.3)
    return heuristics_matrix
