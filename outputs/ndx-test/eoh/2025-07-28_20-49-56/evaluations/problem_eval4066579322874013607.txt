def heuristics_v2(df):
    # Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_percent = 100 * ((df['close'] - low_14) / (high_14 - low_14))
    
    # Bollinger Bands
    ma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = ma_20 + (2 * std_20)
    lower_band = ma_20 - (2 * std_20)
    bollinger_width = (upper_band - lower_band) / ma_20
    
    # Dynamic Weights
    std_close = df['close'].rolling(window=14).std()
    weight_k = 0.7 * (1 + std_close / std_close.max())
    weight_bbw = 0.3 * (1 + std_close / std_close.max())
    
    heuristics_matrix = (weight_k * k_percent + weight_bbw * bollinger_width)
    
    return heuristics_matrix
