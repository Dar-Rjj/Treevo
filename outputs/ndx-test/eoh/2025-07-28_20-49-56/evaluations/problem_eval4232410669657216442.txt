def heuristics_v2(df):
    # Calculate 30-day and 100-day EMA for the closing price
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()

    # Calculate the 10-day average volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()

    # Calculate the ratio of the current day's volume to the 10-day average volume
    volume_ratio = df['volume'] / avg_volume_10

    # Combine the EMA difference and the volume ratio
    ema_diff = ema_30 - ema_100
    heuristics_matrix = (ema_diff + volume_ratio) / 2

    return heuristics_matrix
