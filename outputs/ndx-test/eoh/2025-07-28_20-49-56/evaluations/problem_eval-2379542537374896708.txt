import pandas as pd

def heuristics_v2(df):
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the 30-day momentum
    momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Calculate the 30-day coefficient of variation for volume to measure consistency
    avg_volume = df['volume'].rolling(window=30).mean()
    std_volume = df['volume'].rolling(window=30).std()
    volume_consistency = std_volume / avg_volume
    
    # Combine momentum and volume consistency into a single heuristics measure
    heuristics_matrix = momentum * (1 / volume_consistency)
    
    return heuristics_matrix
