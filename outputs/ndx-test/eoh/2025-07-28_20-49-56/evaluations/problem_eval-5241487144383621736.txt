import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day simple moving average (SMA)
    sma = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation
    std = df['close'].rolling(window=20).std()
    
    # Calculate the Bollinger Bands width
    bb_width = (sma + 2*std) - (sma - 2*std)
    
    # Calculate the 12-day EMA
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate the EMA growth rate
    ema_growth_rate = (ema_12 - ema_12.shift(1)) / ema_12.shift(1)
    
    # Combine BB width and EMA growth rate into a single heuristics measure
    heuristics_matrix = (bb_width + ema_growth_rate) / 2
    
    return heuristics_matrix
