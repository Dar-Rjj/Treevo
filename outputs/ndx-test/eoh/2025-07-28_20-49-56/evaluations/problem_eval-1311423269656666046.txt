def heuristics_v2(df):
    # Calculate the RSI for a 10-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ROC for a 25-day period
    roc = ((df['close'] - df['close'].shift(25)) / df['close'].shift(25)) * 100
    
    # Combine RSI and ROC into a single heuristics measure, with a bias towards ROC
    heuristics_matrix = (rsi * 0.4 + roc * 0.6)
    return heuristics_matrix
