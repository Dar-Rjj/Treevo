import pandas as pd

def heuristics_v2(df):
    returns = df['close'].pct_change()
    rolling_corr = returns.rolling(window=30).corr(df['volume'])
    short_term_ma = rolling_corr.rolling(window=10).mean()
    long_term_ma = rolling_corr.rolling(window=60).mean()
    heuristics_matrix = short_term_ma - long_term_ma
    return heuristics_matrix
