def heuristics_v2(df):
    # Calculate the rate of change for closing prices
    df['ROC'] = df['close'].pct_change()
    
    # Calculate the Exponential Moving Averages (EMA) required for MACD
    ema12 = df['close'].ewm(span=12, adjust=False).mean()
    ema26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Compute the MACD line and signal line
    macd_line = ema12 - ema26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Generate the DMI by combining ROC and the difference between MACD and Signal Line
    df['DMI'] = df['ROC'] + (macd_line - signal_line)
    
    # Return only the DMI values as a Series
    heuristics_matrix = df['DMI'].dropna()
    return heuristics_matrix
