def heuristics_v2(df):
    recent_sum_close = df['close'].rolling(window=5).sum()
    long_term_sum_close = df['close'].rolling(window=20).sum()
    avg_volume = df['volume'].rolling(window=10).mean()
    
    heuristics_matrix = (recent_sum_close / long_term_sum_close) * avg_volume
    
    return heuristics_matrix
