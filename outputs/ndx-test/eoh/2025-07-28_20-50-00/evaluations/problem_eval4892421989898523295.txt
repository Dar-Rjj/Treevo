import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the 25th and 75th percentiles of the daily volumes
    volume_25th = df['volume'].quantile(0.25)
    volume_75th = df['volume'].quantile(0.75)
    # Ensure the 75th percentile volume is not zero to avoid division by zero
    if volume_75th == 0:
        return pd.Series([0]*len(df), index=df.index)
    # Ratio of 25th to 75th percentile of volume
    volume_ratio = volume_25th / volume_75th
    # Calculate the median of daily returns
    median_daily_returns = daily_returns.median()
    # Compute the heuristic factor
    heuristic_factor = median_daily_returns * (1 + volume_ratio) if volume_75th > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
