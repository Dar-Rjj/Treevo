import pandas as pd

def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['vol_std_30'] = df['volume'].rolling(window=30).std()
    ema_daily_return = df['daily_return'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (ema_daily_return + df['vol_std_30']).dropna()
    return heuristics_matrix
