import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the exponential moving average of daily returns with a span of 10 days
    ema_daily_returns = daily_returns.ewm(span=10, adjust=False).mean()
    # Calculate the mean and standard deviation of the daily volumes
    volume_mean = df['volume'].mean()
    volume_std = df['volume'].std()
    # Calculate the coefficient of variation of daily volumes
    volume_cv = volume_std / volume_mean if volume_mean > 0 else 0
    # Calculate the standard deviation of daily returns
    daily_returns_std = daily_returns.std()
    # Compute the heuristic factor
    denominator = volume_cv + daily_returns_std
    heuristic_factor = ema_daily_returns.mean() / denominator if denominator > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
