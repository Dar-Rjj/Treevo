def heuristics_v2(df):
    avg_volume = df['volume'].rolling(window=10).mean()
    adjusted_high = df['high'] * (df['volume'] / avg_volume)
    adjusted_low = df['low'] * (df['volume'] / avg_volume)
    heuristics_matrix = df['close'] - ((adjusted_high + adjusted_low) / 2).rolling(window=10).mean()
    return heuristics_matrix
