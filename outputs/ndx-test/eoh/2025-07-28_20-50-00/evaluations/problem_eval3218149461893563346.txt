def heuristics_v2(df):
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate 10-day EMA of volume
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Composite heuristic
    heuristics_matrix = price_diff + volume_ema
    
    return heuristics_matrix
