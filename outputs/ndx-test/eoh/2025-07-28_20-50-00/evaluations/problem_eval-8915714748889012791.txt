def heuristics_v2(df):
    sum_volume_20 = df['volume'].rolling(window=20).sum()
    sum_volume_50 = df['volume'].rolling(window=50).sum()
    min_close_200 = df['close'].rolling(window=200).min()
    heuristics_matrix = (sum_volume_20 / sum_volume_50) - (df['close'] - min_close_200)
    return heuristics_matrix
