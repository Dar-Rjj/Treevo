import pandas as pd

def heuristics_v2(df):
    def calculate_ema(series, span):
        return series.ewm(span=span, adjust=False).mean()
    
    ema_short = calculate_ema(df['close'], 10)
    ema_long = calculate_ema(df['close'], 30)
    range_change = (df['high'] - df['low']) / df['open'].shift(1)
    
    heuristics_matrix = (ema_short - ema_long) + range_change
    
    return heuristics_matrix
