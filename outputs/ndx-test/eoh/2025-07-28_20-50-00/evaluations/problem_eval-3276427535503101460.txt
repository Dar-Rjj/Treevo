import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day and 50-day SMA for Close Prices
    sma_10 = df['close'].rolling(window=10).mean()
    sma_50 = df['close'].rolling(window=50).mean()

    # Calculate the difference between 10-day and 50-day SMA
    sma_diff = sma_10 - sma_50

    # Calculate 20-day Standard Deviation for Close Prices
    std_20 = df['close'].rolling(window=20).std()

    # Calculate 10-day Rate of Change (ROC) for Volume
    roc_10_volume = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10) * 100

    # Combine all elements into a final heuristic
    heuristics_matrix = sma_diff + std_20 + roc_10_volume
    
    return heuristics_matrix
