import pandas as pd

def heuristics_v2(df):
    # Calculate the 30-day range (difference between high and low)
    df['range_30'] = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    # Calculate the 5-day momentum of the closing price
    df['momentum_5'] = df['close'].pct_change(periods=5)
    # Combine the range and momentum to form the heuristic factor
    heuristics_matrix = (df['range_30'] + df['momentum_5']).dropna()
    return heuristics_matrix
