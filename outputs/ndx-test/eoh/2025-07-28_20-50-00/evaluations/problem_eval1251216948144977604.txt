import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day Exponential Moving Average (EMA)
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day Standard Deviation
    std_20 = df['close'].rolling(window=20).std()
    
    # Calculate Upper and Lower Bollinger Bands
    upper_band = ema_20 + 2 * std_20
    lower_band = ema_20 - 2 * std_20
    
    # Calculate Bollinger Band Percentage Bandwidth
    bb_percentage_bandwidth = (upper_band - lower_band) / ema_20 * 100
    
    # Calculate Close Price to 20-day EMA Ratio
    close_to_ema_ratio = df['close'] / ema_20
    
    # Combine BB Percentage Bandwidth and Close Price to EMA Ratio into a final heuristic
    heuristics_matrix = bb_percentage_bandwidth * close_to_ema_ratio
    
    return heuristics_matrix
