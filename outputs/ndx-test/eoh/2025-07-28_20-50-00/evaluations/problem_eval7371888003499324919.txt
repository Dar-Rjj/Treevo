import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the 5-day simple moving average (SMA) of volumes to assess recent trading activity
    sma_volume_5 = df['volume'].rolling(window=5).mean()
    # Calculate the percentage change in the SMA of volumes over a 10-day period to capture the trend in trading volume
    volume_sma_change = sma_volume_5.pct_change(periods=10)
    # Compute the heuristic factor
    heuristics_matrix = daily_returns * volume_sma_change
    return heuristics_matrix
