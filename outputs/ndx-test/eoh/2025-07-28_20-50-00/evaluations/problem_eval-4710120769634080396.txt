import pandas as pd

def heuristics_v2(df):
    # Calculate the sum of daily closing prices over a 20-day period
    sum_close = df['close'].rolling(window=20).sum()
    
    # Calculate the sum of daily volumes over a 20-day period
    sum_volume = df['volume'].rolling(window=20).sum()
    
    # Ensure the sum of volumes is not zero to avoid division by zero
    if sum_volume.isnull().any() or (sum_volume == 0).any():
        return pd.Series([0]*len(df), index=df.index)
    
    # Compute the heuristic factor
    heuristic_factor = sum_close / sum_volume
    
    # Create a Series for the heuristic factor, ensuring it is indexed correctly
    heuristics_matrix = pd.Series(heuristic_factor, index=df.index)
    return heuristics_matrix
