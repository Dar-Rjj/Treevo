import pandas as pd

def heuristics_v2(df):
    # Calculate daily return differences (close - open)
    daily_return_diffs = df['close'] - df['open']
    # Calculate the total traded volume
    total_volume = df['volume'].sum()
    # Compute the heuristic factor
    heuristic_factor = daily_return_diffs.sum() / total_volume if total_volume > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
