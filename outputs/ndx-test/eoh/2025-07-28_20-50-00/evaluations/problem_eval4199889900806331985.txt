import pandas as pd

def heuristics_v2(df):
    # Calculate the sum of the differences between daily high and low prices
    sum_high_low_diff = (df['high'] - df['low']).sum()
    # Calculate the average closing price
    avg_close_price = df['close'].mean()
    # Compute the heuristic factor
    heuristic_factor = sum_high_low_diff / avg_close_price if avg_close_price > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
