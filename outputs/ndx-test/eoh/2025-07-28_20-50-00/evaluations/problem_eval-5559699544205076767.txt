import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 60-day simple moving averages of the closing prices
    sma_20 = df['close'].rolling(window=20).mean()
    sma_60 = df['close'].rolling(window=60).mean()
    # Compute the momentum factor based on the difference between the two moving averages
    momentum_factor = sma_20 - sma_60
    # Calculate the 20-day and 60-day exponential moving averages of the volumes
    ema_20_volume = df['volume'].ewm(span=20, adjust=False).mean()
    ema_60_volume = df['volume'].ewm(span=60, adjust=False).mean()
    # Compute the relative strength of volume
    relative_strength_volume = (ema_20_volume - ema_60_volume) / ema_60_volume
    # Ensure the relative strength is not zero to avoid division by zero
    if relative_strength_volume.abs().sum() == 0:
        return pd.Series([0]*len(df), index=df.index)
    # Compute the heuristic factor
    heuristic_factor = (momentum_factor * relative_strength_volume) / relative_strength_volume.abs().sum()
    # Create a Series for the heuristic factor, matching it to each date in the DataFrame
    heuristics_matrix = pd.Series(heuristic_factor, index=df.index)
    return heuristics_matrix
