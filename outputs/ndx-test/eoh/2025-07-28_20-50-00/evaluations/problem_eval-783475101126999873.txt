import pandas as pd

def heuristics_v2(df):
    # Calculate price momentum
    price_momentum = df['close'].pct_change(periods=5)
    
    # Calculate volume change
    volume_change = df['volume'].pct_change(periods=1)
    
    # Calculate daily return
    daily_return = df['close'].pct_change()
    
    # Calculate volatility (standard deviation of daily returns over a window)
    volatility = daily_return.rolling(window=20).std()
    
    # Combine factors into a single score using equal weights
    heuristics_matrix = 0.3 * price_momentum + 0.3 * volume_change + 0.4 * volatility
    
    # Return the heuristics matrix
    return heuristics_matrix
