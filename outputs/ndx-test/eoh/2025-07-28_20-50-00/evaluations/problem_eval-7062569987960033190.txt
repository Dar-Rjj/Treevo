import pandas as pd

def heuristics_v2(df):
    # Calculate MACD
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=13, adjust=False).mean()
    macd = short_ema - long_ema
    signal = macd.ewm(span=4, adjust=False).mean()
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Composite heuristic
    heuristics_matrix = (macd - signal) + rsi
    
    return heuristics_matrix
