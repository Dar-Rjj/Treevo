import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Compute the weighted moving average of daily returns with a window of 10 days, using volume as weight
    wma_daily_returns = (daily_returns * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    # Calculate the 25th and 75th percentiles of the daily close prices
    q25_close = df['close'].quantile(0.25)
    q75_close = df['close'].quantile(0.75)
    # Compute the interquartile range of daily close prices as a measure of price volatility
    iqr_close = q75_close - q25_close
    # Compute the heuristic factor
    heuristic_factor = wma_daily_returns.mean() / iqr_close if iqr_close > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
