import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the weighted moving average of daily returns with a span of 30 days
    wma_daily_returns = daily_returns.rolling(window=30).apply(lambda x: (x * pd.Series(range(1, len(x)+1))).sum() / pd.Series(range(1, len(x)+1)).sum(), raw=True)
    # Calculate the mean and standard deviation of the last 15 days' volumes
    volume_mean = df['volume'][-15:].mean()
    volume_std = df['volume'][-15:].std()
    # Calculate the coefficient of variation of the last 15 days' volumes
    volume_cv = volume_std / volume_mean if volume_mean > 0 else 0
    # Calculate the coefficient of variation of daily returns
    daily_returns_cv = daily_returns.std() / daily_returns.mean() if daily_returns.mean() != 0 else 0
    # Compute the heuristic factor
    denominator = volume_cv + daily_returns_cv
    heuristic_factor = wma_daily_returns.mean() * (1 / denominator) if denominator > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
