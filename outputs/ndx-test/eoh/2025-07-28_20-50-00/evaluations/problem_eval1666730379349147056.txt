def heuristics_v2(df):
    vwap_60 = (df['close'] * df['volume']).rolling(window=60).sum() / df['volume'].rolling(window=60).sum()
    vwap_120 = (df['close'] * df['volume']).rolling(window=120).sum() / df['volume'].rolling(window=120).sum()
    heuristics_matrix = vwap_60 / vwap_120
    return heuristics_matrix
