import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    daily_log_returns = np.log(df['close']).diff()
    # Calculate the average of positive daily log returns
    avg_positive_returns = daily_log_returns[daily_log_returns > 0].mean()
    # Calculate the average of negative daily log returns
    avg_negative_returns = daily_log_returns[daily_log_returns < 0].mean()
    # Compute the heuristic factor
    heuristic_factor = avg_positive_returns / avg_negative_returns if avg_negative_returns != 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
