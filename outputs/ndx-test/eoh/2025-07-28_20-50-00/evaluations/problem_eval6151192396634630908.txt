import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window):
        return series.rolling(window=window).mean()

    def calculate_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    short_window = 10
    long_window = 50
    rsi_window = 14
    
    close_prices = df['close']
    
    short_ma = calculate_moving_average(close_prices, short_window)
    long_ma = calculate_moving_average(close_prices, long_window)
    rsi = calculate_rsi(close_prices, rsi_window)
    
    heuristics_matrix = (short_ma - long_ma) * (70 - rsi)
    return heuristics_matrix
