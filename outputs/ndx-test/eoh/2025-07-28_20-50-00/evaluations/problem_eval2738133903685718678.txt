import pandas as pd

def heuristics_v2(df):
    # Calculate daily return differences
    daily_return_diffs = df['close'] - df['open']
    # Sum of positive daily return differences
    positive_diff_sum = daily_return_diffs[daily_return_diffs > 0].sum()
    # Total sum of absolute daily return differences
    total_abs_diff_sum = daily_return_diffs.abs().sum()
    # Compute the heuristic factor
    heuristic_factor = positive_diff_sum / total_abs_diff_sum if total_abs_diff_sum > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
