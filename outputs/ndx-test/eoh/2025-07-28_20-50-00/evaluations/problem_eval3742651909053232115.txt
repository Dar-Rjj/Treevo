import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the average daily return over a 60-day period
    avg_daily_return = daily_returns.rolling(window=60).mean()
    # Calculate the rolling maximum close price over a 60-day period
    rolling_max = df['close'].rolling(window=60).max()
    # Calculate the drawdown
    drawdown = (df['close'] - rolling_max) / rolling_max
    # Calculate the maximum drawdown over the 60-day period
    max_drawdown = drawdown.rolling(window=60).min()
    # Compute the heuristic factor
    heuristic_factor = avg_daily_return / max_drawdown.abs()
    # Create a Series for the heuristic factor, ensuring it is indexed correctly
    heuristics_matrix = pd.Series(heuristic_factor, index=df.index)
    return heuristics_matrix
