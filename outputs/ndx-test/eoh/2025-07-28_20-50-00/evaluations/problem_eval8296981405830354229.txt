import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Count the number of positive daily returns
    positive_returns_count = (daily_returns > 0).sum()
    
    # Total number of days
    total_days = len(df)
    
    # Calculate the ratio of positive to total daily returns
    positive_ratio = positive_returns_count / total_days
    
    # Calculate the momentum (change in closing price over the last 120 days)
    momentum = (df['close'] - df['close'].shift(120)) / df['close'].shift(120)
    
    # Compute the heuristic factor
    heuristic_factor = positive_ratio * momentum
    
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series(heuristic_factor, index=df.index)
    
    return heuristics_matrix
