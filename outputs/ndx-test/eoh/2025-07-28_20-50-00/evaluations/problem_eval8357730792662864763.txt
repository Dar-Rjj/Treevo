import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 30-day moving average of daily high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    moving_avg_high_low = avg_high_low.rolling(window=30).mean()
    
    # Calculate the difference between the 30-day moving average of daily high and low prices and the current closing price
    diff = moving_avg_high_low - df['close']
    
    # Compute the momentum by calculating the difference in the `diff` values over the last 10 days
    momentum = diff.diff(periods=10)
    
    # Create a Series for the heuristic factor, ensuring it is indexed correctly
    heuristics_matrix = pd.Series(momentum, index=df.index)
    return heuristics_matrix
