import pandas as pd

def heuristics_v2(df):
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the EMA of daily returns
    ema_daily_returns = daily_returns.ewm(span=10, adjust=False).mean()
    # Calculate the daily trading range
    daily_range = df['high'] - df['low']
    # Calculate the standard deviation of daily trading ranges
    daily_range_std = daily_range.std()
    # Compute the heuristic factor
    heuristic_factor = ema_daily_returns / daily_range_std if daily_range_std > 0 else 0
    # Create a Series for the heuristic factor
    heuristics_matrix = pd.Series(heuristic_factor, index=df.index)
    return heuristics_matrix
