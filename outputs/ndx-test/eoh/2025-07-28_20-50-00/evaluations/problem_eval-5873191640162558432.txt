def heuristics_v2(df):
    # Calculate 10-day Momentum for Close Prices
    mom_10_close = df['close'] - df['close'].shift(10)

    # Calculate 20-day Moving Average for Volume
    ma_20_volume = df['volume'].rolling(window=20).mean()

    # Calculate 5-day Rate of Change (ROC) for High Prices
    roc_5_high = (df['high'] - df['high'].shift(5)) / df['high'].shift(5) * 100

    # Combine all elements into a final heuristic
    heuristics_matrix = mom_10_close + ma_20_volume - roc_5_high
    
    return heuristics_matrix
