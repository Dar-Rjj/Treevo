import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day Simple Moving Average (SMA)
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate Close Price to 10-day SMA Ratio
    close_to_sma_ratio = df['close'] / sma_10
    
    # Calculate True Range
    true_range = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs() + (df['close'] - df['high']).abs() + (df['close'] - df['low']).abs()
    
    # Calculate 20-day Average True Range (ATR)
    atr_20 = true_range.rolling(window=20).mean()
    
    # Combine Close Price to SMA Ratio and ATR into a final heuristic
    heuristics_matrix = close_to_sma_ratio * atr_20
    
    return heuristics_matrix
