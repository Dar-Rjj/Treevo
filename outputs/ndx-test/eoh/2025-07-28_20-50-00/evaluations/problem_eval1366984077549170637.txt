import pandas as pd

def heuristics_v2(df):
    # Calculate Rolling Mean and Standard Deviation for Volume
    volume_mean = df['volume'].rolling(window=14).mean()
    volume_std = df['volume'].rolling(window=14).std()

    # Calculate RSI for Close Prices
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate MACD for Close Prices
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_12 - ema_26
    signal = macd.rolling(window=9).mean()
    macd_diff = macd - signal

    # Combine all elements into a final heuristic
    heuristics_matrix = (volume_mean + volume_std + rsi) - macd_diff
    
    return heuristics_matrix
