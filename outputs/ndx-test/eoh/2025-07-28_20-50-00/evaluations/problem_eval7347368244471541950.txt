import pandas as pd

def heuristics_v2(df):
    # Calculate the mean of daily high minus low prices
    avg_high_low_diff = df['high'].mean() - df['low'].mean()
    # Calculate the median of the daily volumes
    volume_median = df['volume'].median()
    # Compute the heuristic factor
    heuristic_factor = avg_high_low_diff - volume_median
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
