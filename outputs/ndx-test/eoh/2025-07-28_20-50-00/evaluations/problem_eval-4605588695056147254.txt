import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the ratio of close to volume
    df['close_to_volume'] = df['close'] / df['volume']
    
    # Compute the weighted moving average using the daily range as weights
    window_size = 10
    df['weights'] = df['range'].rolling(window=window_size).sum()
    df['weighted_close_to_volume'] = (df['close_to_volume'] * df['range']).rolling(window=window_size).sum() / df['weights']
    
    heuristics_matrix = df['weighted_close_to_volume'].dropna()
    return heuristics_matrix
