import pandas as pd

def heuristics_v2(df):
    # Calculate daily high-low price spread
    daily_spread = df['high'] - df['low']
    # Calculate the standard deviation of the daily volumes
    volume_std = df['volume'].std()
    # Calculate the average of daily high-low price spreads
    avg_daily_spread = daily_spread.mean()
    # Compute the heuristic factor
    heuristic_factor = avg_daily_spread / volume_std if volume_std > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
