import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the standard deviation and mean of the daily volumes
    volume_std = df['volume'].std()
    volume_mean = df['volume'].mean()
    # Ensure the mean volume is not zero to avoid division by zero
    if volume_mean == 0:
        return pd.Series([0]*len(df), index=df.index)
    # Coefficient of variation of volume
    volume_cov = volume_std / volume_mean
    # Calculate the average of daily returns
    avg_daily_returns = daily_returns.mean()
    # Compute the heuristic factor
    heuristic_factor = avg_daily_returns * (1/volume_cov) if volume_cov > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
