import pandas as pd

def heuristics_v2(df):
    df['price_vol_ratio'] = df['close'] / df['volume']
    df['high_low_diff'] = df['high'] - df['low']
    weighted_avg = (df['price_vol_ratio'] + 2 * df['high_low_diff']).rolling(window=10).mean()
    heuristics_matrix = weighted_avg
    return heuristics_matrix
