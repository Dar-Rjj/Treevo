import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day Simple Moving Average (SMA)
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate Close Price to 5-day SMA Ratio
    close_to_sma_ratio = df['close'] / sma_5
    
    # Calculate True Range
    true_range = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs() + (df['close'] - df['high']).abs() + (df['close'] - df['low']).abs()
    
    # Calculate 10-day Average True Range (ATR)
    atr_10 = true_range.rolling(window=10).mean()
    
    # Combine Close Price to SMA Ratio and ATR into a final heuristic
    heuristics_matrix = close_to_sma_ratio * atr_10
    
    return heuristics_matrix
