def heuristics_v2(df):
    daily_returns = df['close'].pct_change()
    daily_volume_changes = df['volume'].pct_change()
    corr = daily_returns.rolling(window=60).corr(daily_volume_changes)
    max_high = df['high'].rolling(window=60).max()
    heuristics_matrix = corr * max_high
    return heuristics_matrix
