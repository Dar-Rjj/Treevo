import pandas as pd

def heuristics_v2(df):
    window_size = 10
    df['price_range'] = df['high'].rolling(window=window_size).max() - df['low'].rolling(window=window_size).min()
    df['volume_change'] = df['volume'].pct_change()
    heuristics_matrix = (df['price_range'] * df['volume_change']).shift(1)
    return heuristics_matrix
