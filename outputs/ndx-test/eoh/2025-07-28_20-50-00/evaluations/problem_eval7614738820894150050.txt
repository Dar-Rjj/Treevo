import pandas as pd
from scipy.stats import entropy

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the simple moving average of daily returns with a window of 10 days
    sma_daily_returns = daily_returns.rolling(window=10).mean()
    # Calculate the daily volume changes
    volume_changes = df['volume'].diff().fillna(0)
    # Compute the entropy of daily volume changes
    volume_entropy = entropy(volume_changes.abs())
    # Compute the heuristic factor
    heuristic_factor = sma_daily_returns.mean() / (1 + volume_entropy) if volume_entropy > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
