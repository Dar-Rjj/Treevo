import numpy as np

def heuristics_v2(df):
    # Calculate 50-day and 200-day simple moving averages of the closing price
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    # Calculate the ratio between 50-day and 200-day SMAs
    sma_ratio = sma_50 / sma_200
    # Calculate the natural logarithm of the ratio of today's volume to yesterday's volume
    log_volume_ratio = np.log(df['volume'] / df['volume'].shift(1))
    # Combine SMA ratio and log volume ratio into a heuristic factor
    heuristics_matrix = (sma_ratio + log_volume_ratio).dropna()
    return heuristics_matrix
