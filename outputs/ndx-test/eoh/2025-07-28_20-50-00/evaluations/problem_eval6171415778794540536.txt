import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    # Calculate the mean of the daily range
    avg_daily_range = daily_range.mean()
    # Calculate the mean of the daily close
    avg_daily_close = df['close'].mean()
    # Ensure the mean daily close is not zero to avoid division by zero
    if avg_daily_close == 0:
        return pd.Series([0]*len(df), index=df.index)
    # Compute the heuristic factor
    heuristic_factor = avg_daily_range / avg_daily_close
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
