import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the standard deviation of daily returns
    daily_returns_std = daily_returns.std()
    # Median of daily returns
    median_daily_returns = daily_returns.median()
    # Calculate the 50th percentile of the daily volumes
    volume_50th = df['volume'].quantile(0.5)
    # Calculate the standard deviation of daily volumes
    volume_std = df['volume'].std()
    # Weights calculation
    weight_returns = 1 / (daily_returns_std + 1e-6)  # Adding a small value to avoid division by zero
    weight_volume = 1 / (volume_std + 1e-6)  # Adding a small value to avoid division by zero
    # Compute the heuristic factor
    heuristic_factor = (median_daily_returns * weight_returns) + (volume_50th * weight_volume)
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
