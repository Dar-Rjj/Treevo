import pandas as pd

def heuristics_v2(df):
    # Calculate the daily returns
    daily_returns = df['close'].pct_change().fillna(0)
    # Calculate the EMA of daily returns over a 20-day window
    ema_returns = daily_returns.ewm(span=20, adjust=False).mean()
    # Calculate the coefficient of variation of high and low prices
    cv_price = (df['high'] - df['low']).std() / (df['high'] - df['low']).mean()
    # Compute the heuristic factor
    heuristic_factor = ema_returns / cv_price if cv_price > 0 else 0
    # Create a Series for the heuristic factor
    heuristics_matrix = pd.Series(heuristic_factor, index=df.index)
    return heuristics_matrix
