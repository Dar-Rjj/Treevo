import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Sum of squared daily returns
    sum_squared_returns = (daily_returns ** 2).sum()
    # Calculate the mean of the daily volumes
    avg_volume = df['volume'].mean()
    # Ensure the average volume is not zero to avoid log(0)
    if avg_volume == 0:
        return pd.Series([0]*len(df), index=df.index)
    # Logarithm of average daily volume
    log_avg_volume = np.log(avg_volume)
    # Compute the heuristic factor
    heuristic_factor = sum_squared_returns * log_avg_volume
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
