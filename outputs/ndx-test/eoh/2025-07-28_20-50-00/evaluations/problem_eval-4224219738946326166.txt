import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change().fillna(0) + 1
    # Calculate the cumulative product of daily returns
    cum_product_returns = daily_returns.cumprod()
    # Calculate the difference between consecutive daily volumes
    volume_diff = df['volume'].diff().fillna(0)
    # Calculate the sum of squared differences in volume
    sum_squared_volume_diff = (volume_diff ** 2).sum()
    # Compute the heuristic factor
    heuristic_factor = cum_product_returns.iloc[-1] / sum_squared_volume_diff if sum_squared_volume_diff > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
