import pandas as pd

def heuristics_v2(df):
    df['avg_high_low_20'] = (df['high'] + df['low']) / 2
    df['price_ratio'] = df['close'] / df['avg_high_low_20'].rolling(window=20).mean()
    daily_returns = df['close'].pct_change()
    df['return_diff_10'] = daily_returns.rolling(window=10).max() - daily_returns.rolling(window=10).min()
    heuristics_matrix = (df['price_ratio'] + df['return_diff_10']).dropna()
    return heuristics_matrix
