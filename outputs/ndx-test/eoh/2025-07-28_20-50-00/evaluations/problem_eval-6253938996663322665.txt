import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the exponential moving average of daily returns with a span of 10 days
    ema_daily_returns = daily_returns.ewm(span=10, adjust=False).mean()
    # Calculate the exponential moving average of daily volumes with a span of 10 days
    ema_daily_volumes = df['volume'].ewm(span=10, adjust=False).mean()
    # Combine the EMAs into a weighted average
    combined_ema = (ema_daily_returns + np.log(ema_daily_volumes + 1)) / 2
    # Calculate the standard deviation of daily returns
    daily_returns_std = daily_returns.std()
    # Calculate the standard deviation of log-transformed daily volumes
    daily_volumes_log_std = np.log(df['volume'] + 1).std()
    # Compute the heuristic factor
    denominator = daily_returns_std + daily_volumes_log_std
    heuristic_factor = combined_ema.mean() / denominator if denominator > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
