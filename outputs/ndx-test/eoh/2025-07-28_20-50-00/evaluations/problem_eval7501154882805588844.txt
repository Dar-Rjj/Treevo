def heuristics_v2(df):
    # Calculate price change
    df['price_change'] = df['close'].diff()
    # Simple moving average for short term trend (e.g., 5 days)
    short_rolling = df['close'].rolling(window=5).mean()
    # Simple moving average for long term trend (e.g., 20 days)
    long_rolling = df['close'].rolling(window=20).mean()
    # Short term volatility
    short_volatility = df['close'].rolling(window=5).std()
    # Long term volatility
    long_volatility = df['close'].rolling(window=20).std()
    # Momentum signal: difference between short and long term averages
    momentum_signal = short_rolling - long_rolling
    # Volatility adjusted momentum: momentum normalized by long term volatility
    volatility_adjusted_momentum = momentum_signal / long_volatility
    # Final heuristic matrix: adjusted momentum plus short term volatility
    heuristics_matrix = volatility_adjusted_momentum + short_volatility
    return heuristics_matrix
