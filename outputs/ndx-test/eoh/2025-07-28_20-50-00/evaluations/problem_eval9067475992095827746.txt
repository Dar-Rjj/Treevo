import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the rolling sum of daily returns over a 5-day window
    rolling_sum_daily_returns = daily_returns.rolling(window=5).sum()
    # Calculate the mean of the daily volumes
    volume_mean = df['volume'].mean()
    # Compute the heuristic factor
    heuristic_factor = rolling_sum_daily_returns / volume_mean if volume_mean > 0 else 0
    # Create a Series for the heuristic factor, using the rolling sum values where available
    heuristics_matrix = heuristic_factor.fillna(0)
    return heuristics_matrix
