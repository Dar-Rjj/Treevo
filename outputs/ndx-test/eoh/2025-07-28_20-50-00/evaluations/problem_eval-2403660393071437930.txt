import pandas as pd

def heuristics_v2(df):
    # Calculate CMF
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Bollinger Bands
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    std_dev = typical_price.rolling(window=20).std()
    middle_band = typical_price.rolling(window=20).mean()
    upper_band = middle_band + 2 * std_dev
    lower_band = middle_band - 2 * std_dev
    bollinger_width = (upper_band - lower_band) / middle_band
    
    # Composite heuristic
    heuristics_matrix = cmf + bollinger_width
    
    return heuristics_matrix
