import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day price change
    price_change = df['close'].pct_change(10)
    
    # Calculate the ratio of current volume to the 30-day average volume
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30
    
    # Composite heuristic
    heuristics_matrix = price_change * volume_ratio
    
    return heuristics_matrix
