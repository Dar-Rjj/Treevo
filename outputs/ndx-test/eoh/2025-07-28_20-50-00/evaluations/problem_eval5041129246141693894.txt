import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    # Calculate the simple moving average of daily returns over the last 30 days
    sma_daily_returns = daily_returns.rolling(window=30).mean().iloc[-1]
    # Find the maximum daily return over the last 10 days
    max_daily_return = daily_returns[-10:].max()
    # Compute the heuristic factor
    heuristic_factor = sma_daily_returns / max_daily_return if max_daily_return > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
