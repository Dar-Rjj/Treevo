import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day Exponential Moving Average (EMA)
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 1-day percentage change in EMA
    ema_percentage_change = ema_20.pct_change() * 100
    
    # Calculate the 14-day RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine EMA percentage change and RSI into a final heuristic
    heuristics_matrix = ema_percentage_change * rsi
    
    return heuristics_matrix
