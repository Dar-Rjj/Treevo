import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    daily_log_returns = np.log(df['close']).diff()
    # Compute the exponential moving average of daily log returns with a span of 10 days
    ema_daily_log_returns = daily_log_returns.ewm(span=10, adjust=False).mean()
    # Create a Series for the EMA heuristic factor
    heuristics_matrix = ema_daily_log_returns
    return heuristics_matrix
