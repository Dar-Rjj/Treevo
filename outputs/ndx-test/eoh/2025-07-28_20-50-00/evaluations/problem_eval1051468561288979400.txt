import pandas as pd

def heuristics_v2(df):
    # Calculate the daily high-to-low ratio
    high_low_ratio = df['high'] / df['low']
    # Sum the high-to-low ratios over a 10-day window
    sum_high_low = high_low_ratio.rolling(window=10).sum()
    # Calculate the 10-day average trading volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    # Ensure the average volume is not zero to avoid division by zero
    avg_volume_10[avg_volume_10 == 0] = 1
    # Compute the heuristic factor
    heuristics_matrix = sum_high_low / avg_volume_10
    return heuristics_matrix
