import pandas as pd
import numpy as np

def heuristics_v2(df):
    N = 5  # Number of days to consider for the sum of closing prices
    # Calculate daily log returns
    daily_log_returns = np.log(df['close']).diff()
    # Calculate the mean absolute daily log returns
    mean_abs_daily_log_returns = np.abs(daily_log_returns).mean()
    # Calculate the sum of the last N days' closing prices
    sum_last_N_days_close = df['close'].tail(N).sum()
    # Compute the heuristic factor
    heuristic_factor = mean_abs_daily_log_returns / sum_last_N_days_close if sum_last_N_days_close > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
