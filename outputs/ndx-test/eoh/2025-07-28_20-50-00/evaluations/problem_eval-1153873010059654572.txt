def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['avg_volume_60'] = df['volume'].rolling(window=60).mean()
    df['vol_ratio'] = df['volume'] / df['avg_volume_60']
    ema_daily_return = df['daily_return'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (ema_daily_return + df['vol_ratio']).dropna()
    return heuristics_matrix
