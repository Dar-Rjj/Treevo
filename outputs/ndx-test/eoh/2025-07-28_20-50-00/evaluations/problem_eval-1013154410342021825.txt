import pandas as pd

def heuristics_v2(df):
    # Calculate the average daily range (high - low)
    avg_daily_range = (df['high'] - df['low']).mean()
    # Calculate the total volume traded
    total_volume = df['volume'].sum()
    # Compute the heuristic factor
    heuristic_factor = avg_daily_range / total_volume if total_volume > 0 else 0
    # Create a Series for the heuristic factor, repeating it for each date in the DataFrame
    heuristics_matrix = pd.Series([heuristic_factor]*len(df), index=df.index)
    return heuristics_matrix
