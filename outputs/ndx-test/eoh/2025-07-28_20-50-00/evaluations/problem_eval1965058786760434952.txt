def heuristics_v2(df):
    # Calculate percentage change in volume
    volume_change = df['volume'].pct_change(periods=10)
    
    # Calculate a custom volatility measure
    high_low_diff = df['high'] - df['low']
    avg_true_range = high_low_diff.rolling(window=14).mean()
    
    # Composite heuristic
    heuristics_matrix = volume_change + avg_true_range
    
    return heuristics_matrix
