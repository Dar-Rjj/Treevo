def heuristics_v2(df):
    sma_70 = df['close'].rolling(window=70).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    diff_sma = sma_70 - sma_30
    smoothed_diff = diff_sma.rolling(window=10).mean()
    sma_150 = df['close'].rolling(window=150).mean()
    heuristics_matrix = smoothed_diff / sma_150
    return heuristics_matrix
