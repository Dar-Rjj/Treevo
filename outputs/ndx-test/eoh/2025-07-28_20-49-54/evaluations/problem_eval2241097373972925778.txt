import pandas as pd

def heuristics_v2(df):
    def rsi(data, window):
        delta = data.diff()
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window=window).mean()
        roll_down = down.abs().rolling(window=window).mean()
        rs = roll_up / roll_down
        return 100.0 - (100.0 / (1.0 + rs))
    
    def adl(high, low, close, volume):
        money_flow_multiplier = ((close - low) - (high - close)) / (high - low)
        money_flow_volume = money_flow_multiplier * volume
        return money_flow_volume.cumsum()
    
    # Calculate RSI for the closing price
    rsi_close = rsi(df['close'], 14)
    
    # Calculate the Accumulation/Distribution Line
    adl_line = adl(df['high'], df['low'], df['close'], df['volume'])
    
    # Combine the factors
    heuristics_matrix = rsi_close + adl_line
    
    return heuristics_matrix
