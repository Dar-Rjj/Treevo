def heuristics_v2(df):
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=15).mean()
    daily_return_3day_avg = df['close'].pct_change().rolling(window=3).mean()
    pv_momentum_smoothed = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    sma_ratio = short_sma / long_sma
    heuristics_matrix = (sma_ratio + daily_return_3day_avg + pv_momentum_smoothed)
    return heuristics_matrix
