def heuristics_v2(df):
    # Calculate the 20-period exponential moving average of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 5-period average volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    
    # Calculate the volume ratio (current volume / 5-day average volume)
    volume_ratio = df['volume'] / avg_volume_5
    
    # Combine the EMA and volume ratio
    heuristics_matrix = (ema_20 + volume_ratio).rank(pct=True)
    
    return heuristics_matrix
