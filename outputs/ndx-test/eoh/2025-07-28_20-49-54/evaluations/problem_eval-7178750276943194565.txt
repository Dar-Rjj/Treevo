import pandas as pd

def heuristics_v2(df):
    # Calculate the exponentially weighted moving average of the high price
    ewma_high = df['high'].ewm(span=10).mean()
    
    # Compute the difference between the close price and its EWMA
    ewma_diff = df['close'] - ewma_high
    
    # Calculate the percentage change in trading volume
    volume_change = df['volume'].pct_change()
    
    # Adjust the EWMA difference using the volume change
    adjusted_signal = ewma_diff * (1 + volume_change)
    
    heuristics_matrix = adjusted_signal
    
    return heuristics_matrix
