def heuristics_v2(df):
    # Calculate the 20-period and 50-period Simple Moving Averages (SMA) of the closing price
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the 14-period Average True Range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr_14 = tr.rolling(window=14).mean()
    
    # Compute the momentum as the difference between the 20-period SMA and 50-period SMA
    momentum = sma_20 - sma_50
    
    # Apply a custom heuristic to combine the momentum and ATR
    heuristics_matrix = (momentum + atr_14)
    
    return heuristics_matrix
