import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day rate of change of the closing price
    roc_10 = df['close'].pct_change(periods=10)
    
    # Calculate the 14-day Average True Range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_14 = tr.rolling(window=14).mean()
    
    # Combine the rate of change and ATR
    combined_factor = roc_10 * atr_14
    
    # Rank the combined factor
    heuristics_matrix = combined_factor.rank(pct=True)
    
    return heuristics_matrix
