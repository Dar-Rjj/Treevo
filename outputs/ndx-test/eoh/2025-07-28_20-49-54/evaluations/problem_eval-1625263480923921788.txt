def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=13, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the smoothed price-volume momentum
    smoothed_close = df['close'].rolling(window=5).mean()
    pv_momentum_smoothed = (smoothed_close * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Compute the difference between short and long exponential moving averages
    ema_diff = short_ema - long_ema
    
    # Apply a custom heuristic to combine the EMA difference and smoothed price-volume momentum
    heuristics_matrix = (ema_diff + pv_momentum_smoothed).rank(pct=True)
    
    return heuristics_matrix
