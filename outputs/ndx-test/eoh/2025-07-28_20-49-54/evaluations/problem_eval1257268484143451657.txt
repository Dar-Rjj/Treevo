import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    short_ma = df['close'].rolling(window=10).mean()
    long_ma = df['close'].rolling(window=50).mean()
    
    # Calculate the price-volume relationship
    pv_relationship = df['close'] / df['volume']
    
    # Compute the difference between short and long moving averages
    ma_diff = short_ma - long_ma
    
    # Apply a custom heuristic to combine the moving average difference and price-volume relationship
    heuristics_matrix = (ma_diff + pv_relationship).rank(pct=True)
    
    return heuristics_matrix
