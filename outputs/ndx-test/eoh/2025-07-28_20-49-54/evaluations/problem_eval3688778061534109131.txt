def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Calculate the Exponential Moving Averages (EMA)
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the EMA crossover signal
    ema_crossover = ema_short - ema_long
    
    # Combine ROC and EMA crossover
    heuristics_matrix = roc + ema_crossover
    
    return heuristics_matrix
