import pandas as pd

def heuristics_v2(df):
    close = df['close']
    open_price = df['open']
    high = df['high']
    low = df['low']
    volume = df['volume']
    
    # Calculate simple daily return
    daily_return = (close / close.shift(1)) - 1
    
    # Calculate price change ratio from open to close
    open_to_close_ratio = close / open_price
    
    # Calculate the average of high and low prices
    avg_high_low = (high + low) / 2
    
    # Calculate the trend in volume
    volume_trend = volume - volume.shift(1)
    
    # Combine factors into a single matrix
    heuristics_matrix = pd.DataFrame({
        'daily_return': daily_return,
        'open_to_close_ratio': open_to_close_ratio,
        'avg_high_low': avg_high_low,
        'volume_trend': volume_trend
    }).mean(axis=1)
    
    return heuristics_matrix
