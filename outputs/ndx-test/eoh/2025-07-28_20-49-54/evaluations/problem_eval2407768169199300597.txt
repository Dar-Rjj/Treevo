def heuristics_v2(df):
    # Calculate the 12-period and 26-period exponential moving averages for MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the MACD line and its 9-period EMA (signal line)
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Calculate the 30-day rate of change in volume
    volume_roc = df['volume'].pct_change(periods=30)
    
    # Combine the MACD histogram (difference between MACD line and signal line) with volume ROC
    heuristics_matrix = (macd_line - signal_line) * volume_roc
    
    return heuristics_matrix
