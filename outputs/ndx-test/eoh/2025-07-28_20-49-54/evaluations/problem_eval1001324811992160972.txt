def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the weighted price-volume momentum
    pv_momentum_weighted = (df['close'] * df['volume']).rolling(window=5).sum() / (df['volume'].rolling(window=5).sum() + 1)  # Adding 1 to avoid division by zero
    
    # Compute the difference between short and long exponential moving averages
    ema_diff = short_ema - long_ema
    
    # Apply a custom heuristic to combine the EMA difference and weighted price-volume momentum
    heuristics_matrix = (ema_diff + pv_momentum_weighted).rank(pct=True)
    
    return heuristics_matrix
