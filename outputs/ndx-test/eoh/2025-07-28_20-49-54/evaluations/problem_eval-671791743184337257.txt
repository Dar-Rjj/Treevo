def heuristics_v2(df):
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the high-to-low price ratio
    hl_ratio = df['high'] / df['low']
    
    # Compute the absolute difference between closing price and VWAP
    cp_vwap_diff_abs = abs(df['close'] - vwap)
    
    # Apply a custom heuristic to combine the high-to-low price ratio and the absolute difference between closing price and VWAP
    heuristics_matrix = (hl_ratio * cp_vwap_diff_abs).rank(pct=True)
    
    return heuristics_matrix
