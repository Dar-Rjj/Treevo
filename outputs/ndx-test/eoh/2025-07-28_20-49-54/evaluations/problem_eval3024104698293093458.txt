def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) of the closing price
    roc = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day standard deviation of the volume
    vol_std = df['volume'].rolling(window=20).std()
    
    # Apply a custom heuristic to combine the ROC and standard deviation of volume
    heuristics_matrix = (roc + vol_std).rank(pct=True)
    
    return heuristics_matrix
