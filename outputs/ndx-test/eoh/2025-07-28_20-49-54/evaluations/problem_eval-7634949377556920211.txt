def heuristics_v2(df):
    # Calculate the 10-period Exponential Moving Average (EMA) of the closing price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 14-period True Range (TR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    
    # Calculate the 14-period Exponential Moving Average (EMA) of the True Range (TR)
    ema_tr_14 = tr.ewm(span=14, adjust=False).mean()
    
    # Apply a custom heuristic to combine the EMA and EMA of TR
    heuristics_matrix = (ema_10 + ema_tr_14).rank(pct=True)
    
    return heuristics_matrix
