def heuristics_v2(df):
    adj_price_range = (df['high'] - df['low']) / df['volume']
    price_ratio = adj_price_range / df['close']
    daily_returns = df['close'].pct_change()
    long_volatility = daily_returns.rolling(window=60).std()
    heuristics_matrix = (price_ratio + long_volatility)
    return heuristics_matrix
