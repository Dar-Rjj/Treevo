import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=9, adjust=False).mean()
    long_ema = df['close'].ewm(span=21, adjust=False).mean()

    # Calculate the modified volume ratio of today's volume to the average volume over the past 30 days
    volume_ratio = (df['volume'] / df['volume'].rolling(window=30).mean()).pow(1/2)

    # Compute the difference between short and long exponential moving averages
    ema_diff = short_ema - long_ema

    # Apply a custom heuristic to combine the EMA difference and modified volume ratio
    heuristics_matrix = (ema_diff * volume_ratio).rank(pct=True)

    return heuristics_matrix
