def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the price-volume ratio for 5-day and 10-day windows
    pv_ratio_5 = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    pv_ratio_10 = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Compute the difference between short and long exponential moving averages
    ema_diff = short_ema - long_ema
    
    # Apply a custom heuristic to combine the EMA difference and both price-volume ratios
    heuristics_matrix = (ema_diff + 0.5 * pv_ratio_5 + 0.5 * pv_ratio_10).rank(pct=True)
    
    return heuristics_matrix
