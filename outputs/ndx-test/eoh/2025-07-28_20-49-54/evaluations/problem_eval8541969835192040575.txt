import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the price-volume momentum
    pv_momentum = (df['close'] - df['close'].shift(1)) / (df['volume'] - df['volume'].shift(1))
    
    # Compute the ratio of short to long exponential moving averages
    ema_ratio = short_ema / long_ema
    
    # Apply a custom heuristic to combine the EMA ratio and price-volume momentum
    heuristics_matrix = (ema_ratio + pv_momentum).rank(pct=True)
    
    return heuristics_matrix
