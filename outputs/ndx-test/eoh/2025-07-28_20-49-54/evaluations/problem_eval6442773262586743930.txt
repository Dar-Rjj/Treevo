import pandas as pd

def heuristics_v2(df):
    # Calculate the True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate the Average True Range
    atr = df['TR'].rolling(window=14).mean()
    
    # Calculate the Exponential Moving Average (EMA) of daily returns
    daily_returns = df['close'].pct_change()
    ema = daily_returns.ewm(span=14, adjust=False).mean()
    
    # Combine Volume/ATR ratio and EMA
    volume_atr_ratio = df['volume'] / atr
    heuristics_matrix = volume_atr_ratio + ema
    
    return heuristics_matrix
