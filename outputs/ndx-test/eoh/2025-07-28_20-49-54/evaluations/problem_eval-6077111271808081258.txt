import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the momentum indicator
    momentum = df['close'] - df['close'].shift(14)
    
    # Compute the difference between short and long exponential moving averages
    ema_diff = short_ema - long_ema
    
    # Apply a custom heuristic to combine the EMA difference and momentum
    heuristics_matrix = (ema_diff + momentum).rank(pct=True)
    
    return heuristics_matrix
