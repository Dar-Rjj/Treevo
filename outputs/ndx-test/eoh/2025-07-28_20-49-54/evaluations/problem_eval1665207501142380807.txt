def heuristics_v2(df):
    # Calculate the Average True Range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the ratio of closing price to ATR
    close_to_atr_ratio = df['close'] / atr
    
    # Compute the 14-period Simple Moving Average (SMA) of the trading volume
    sma_volume = df['volume'].rolling(window=14).mean()
    
    # Combine the ratio of closing price to ATR and SMA of trading volume
    heuristics_matrix = (close_to_atr_ratio + sma_volume).rank(pct=True)
    
    return heuristics_matrix
