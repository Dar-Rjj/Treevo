def heuristics_v2(df):
    # Calculate the 20-period Simple Moving Average (SMA) of the closing price
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the ratio of closing price to SMA
    close_sma_ratio = df['close'] / sma_20
    
    # Calculate the 20-period standard deviation of the closing price
    std_20 = df['close'].rolling(window=20).std()
    
    # Apply a custom heuristic to combine the close-to-SMA ratio and the standard deviation
    heuristics_matrix = (close_sma_ratio + std_20).rank(pct=True)
    
    return heuristics_matrix
