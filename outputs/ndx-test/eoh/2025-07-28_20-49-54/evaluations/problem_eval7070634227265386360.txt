def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages
    short_sma = df['close'].rolling(window=20).mean()
    long_sma = df['close'].rolling(window=50).mean()
    
    # Calculate the smoothed price-to-volume ratio
    pv_ratio_smoothed = (df['close'] / df['volume']).rolling(window=10).mean()
    
    # Compute the difference between short and long simple moving averages
    sma_difference = short_sma - long_sma
    
    # Apply a custom heuristic to combine the SMA difference and smoothed price-to-volume ratio
    heuristics_matrix = (sma_difference + pv_ratio_smoothed).rank(pct=True)
    
    return heuristics_matrix
