def heuristics_v2(df):
    # Calculate the percentage change in closing price over a 10-day period
    pct_change_close = df['close'].pct_change(periods=10)
    
    # Calculate the True Range
    tr = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    
    # Calculate the Average True Range (ATR) over the same 10-day period
    atr = tr.rolling(window=10).mean()
    
    # Combine the percentage change in closing price and the ATR
    heuristics_matrix = (pct_change_close + atr).rank(pct=True)
    
    return heuristics_matrix
