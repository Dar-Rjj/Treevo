import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.ewm(span=14, adjust=False).mean() / negative_money_flow.ewm(span=14, adjust=False).mean()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Combine the RSI and MFI with weights
    heuristics_matrix = (rsi * 0.6 + mfi * 0.4)
    
    return heuristics_matrix
