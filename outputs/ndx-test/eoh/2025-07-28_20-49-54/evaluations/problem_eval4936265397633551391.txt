def heuristics_v2(df):
    # Calculate the 10-day and 50-day rate of change in closing prices
    short_roc = df['close'].pct_change(periods=10)
    long_roc = df['close'].pct_change(periods=50)
    
    # Calculate the relative strength of volume
    volume_strength = df['volume'] / df['volume'].rolling(window=60).mean()
    
    # Combine the rate of change and volume strength
    heuristics_matrix = (short_roc - long_roc) * volume_strength
    
    return heuristics_matrix
