import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day return
    df['return_10d'] = df['close'].pct_change(periods=10)
    
    # Calculate the 10-day volatility
    df['volatility_10d'] = df['close'].pct_change().rolling(window=10).std()
    
    # Compute the RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the 10-day return and 10-day volatility
    composite_factor = df['return_10d'] / df['volatility_10d']
    
    # Adjust the composite factor based on RSI
    heuristics_matrix = composite_factor * (100 - rsi) / 100
    
    return heuristics_matrix
