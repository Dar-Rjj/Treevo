def heuristics_v2(df):
    # Calculate the high-to-low price ratio
    hl_ratio = df['high'] / df['low']
    
    # Calculate the exponential moving average of the closing price
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the simple moving average of the volume
    sma_volume = df['volume'].rolling(window=20).mean()
    
    # Compute the difference between the EMA of the closing price and the SMA of the volume
    ema_sma_diff = ema_close - sma_volume
    
    # Combine the high-to-low price ratio and the EMA-SMA difference
    heuristics_matrix = hl_ratio * ema_sma_diff
    
    return heuristics_matrix
