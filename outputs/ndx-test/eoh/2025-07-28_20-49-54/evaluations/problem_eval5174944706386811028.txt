import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the cumulative sum of the difference between closing and opening prices
    cum_sum_diff = (df['close'] - df['open']).cumsum()
    
    # Apply a custom heuristic to combine the RSI and cumulative sum of price differences
    heuristics_matrix = (rsi + cum_sum_diff).rank(pct=True)
    
    return heuristics_matrix
