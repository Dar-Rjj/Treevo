import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages
    short_sma = df['close'].rolling(window=12).mean()
    long_sma = df['close'].rolling(window=26).mean()

    # Calculate the volume ratio of today's volume to the average volume over the past 20 days
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()

    # Compute the difference between short and long simple moving averages
    sma_diff = short_sma - long_sma

    # Apply a custom heuristic to combine the SMA difference and volume ratio
    heuristics_matrix = (sma_diff + volume_ratio).rank(pct=True)

    return heuristics_matrix
