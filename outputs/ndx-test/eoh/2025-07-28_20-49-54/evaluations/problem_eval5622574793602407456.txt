import pandas as pd

def heuristics_v2(df):
    # Compute short and long moving averages for close price
    df['short_mavg'] = df['close'].rolling(window=10).mean()
    df['long_mavg'] = df['close'].rolling(window=50).mean()
    # Calculate the difference between short and long moving averages
    df['mavg_diff'] = df['short_mavg'] - df['long_mavg']
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    # Calculate the average daily volume over 10 days
    df['avg_vol_10d'] = df['volume'].rolling(window=10).mean()
    # Heuristic score based on momentum and volume
    df['heuristic_score'] = (df['mavg_diff'] + df['avg_vol_10d']) / 2
    # Drop rows with NaN values resulting from rolling calculations
    df.dropna(inplace=True)
    # Create heuristics_matrix as a Series
    heuristics_matrix = df['heuristic_score']
    
    return heuristics_matrix
