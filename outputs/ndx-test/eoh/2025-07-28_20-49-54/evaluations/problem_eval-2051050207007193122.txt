import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the adjusted price-volume relationship
    pv_relationship_adjusted = (df['close'] - df['open']) * df['volume']
    
    # Compute the ratio of short to long exponential moving averages
    ema_ratio = short_ema / long_ema
    
    # Combine the EMA ratio and adjusted price-volume relationship
    heuristics_matrix = (ema_ratio + pv_relationship_adjusted).rank(pct=True)
    
    return heuristics_matrix
