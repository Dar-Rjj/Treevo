import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the price-momentum relationship
    pm_relationship = df['close'] / df['close'].shift(1)
    
    # Compute the difference between short and long exponential moving averages
    ema_diff = short_ema - long_ema
    
    # Apply a custom heuristic to combine the EMA difference and price-momentum relationship
    heuristics_matrix = (ema_diff + pm_relationship).rank(pct=True)
    
    return heuristics_matrix
