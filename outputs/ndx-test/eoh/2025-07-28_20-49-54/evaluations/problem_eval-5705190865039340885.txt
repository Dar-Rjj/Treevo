import pandas as pd

def heuristics_v2(df):
    # Calculate the exponentially weighted moving average of the low price
    ewma_low = df['low'].ewm(span=20).mean()
    
    # Compute the difference between the close price and its EWMA
    ewma_diff = df['close'] - ewma_low
    
    # Calculate the percentage change in trading range (high - low)
    range_change = (df['high'] - df['low']).pct_change()
    
    # Adjust the EWMA difference using the range change
    adjusted_signal = ewma_diff * (1 + range_change)
    
    heuristics_matrix = adjusted_signal
    
    return heuristics_matrix
