def heuristics_v2(df):
    # Calculate the 20-period Simple Moving Average (SMA) of the closing price
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the closing price and the SMA
    diff_sma_close = df['close'] - sma_close
    
    # Compute the 5-period standard deviation of the closing price as a measure of recent volatility
    std_5_close = df['close'].rolling(window=5).std()
    
    # Apply a custom heuristic to combine the difference from SMA and the recent volatility
    heuristics_matrix = diff_sma_close + std_5_close
    
    return heuristics_matrix
