def heuristics_v2(df):
    # Calculate the rate of change in the ratio of high to low prices
    roc_hl = (df['high'] / df['low']).pct_change(periods=10)
    
    # Calculate the MACD line and signal line
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    
    # Combine the rate of change and the MACD histogram
    heuristics_matrix = (roc_hl + (macd_line - signal_line)).rank(pct=True)
    
    return heuristics_matrix
