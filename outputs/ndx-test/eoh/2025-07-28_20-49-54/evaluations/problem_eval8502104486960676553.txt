def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages
    short_sma = df['close'].rolling(window=12).mean()
    long_sma = df['close'].rolling(window=24).mean()
    
    # Calculate the price-to-volume ratio for 5-day and 10-day windows
    pv_ratio_5 = (df['close'] / df['volume']).rolling(window=5).sum() / 5
    pv_ratio_10 = (df['close'] / df['volume']).rolling(window=10).sum() / 10
    
    # Compute the ratio between short and long simple moving averages
    sma_ratio = short_sma / long_sma
    
    # Apply a custom heuristic to combine the SMA ratio and both price-to-volume ratios
    heuristics_matrix = (sma_ratio + 0.5 * pv_ratio_5 + 0.5 * pv_ratio_10).rank(pct=True)
    
    return heuristics_matrix
