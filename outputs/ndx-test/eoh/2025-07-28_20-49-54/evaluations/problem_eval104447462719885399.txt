import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Average Directional Movement Index (ADX)
    high_low = df['high'] - df['low']
    high_close = abs(df['high'] - df['close'].shift())
    low_close = abs(df['low'] - df['close'].shift())
    tr = high_low.combine(high_close, max).combine(low_close, max)
    atr = tr.rolling(window=14).mean()
    
    plus_dm = df['high'].diff(1).where(lambda x: x > 0, 0)
    minus_dm = -df['low'].diff(1).where(lambda x: x < 0, 0)
    
    plus_di = 100 * (plus_dm.rolling(window=14).sum() / atr)
    minus_di = 100 * (minus_dm.rolling(window=14).sum() / atr)
    
    dx = 100 * (abs(plus_di - minus_di) / (plus_di + minus_di))
    adx = dx.ewm(span=14, adjust=False).mean()
    
    # Combine RSI and ADX
    heuristics_matrix = (rsi + adx)
    
    return heuristics_matrix
