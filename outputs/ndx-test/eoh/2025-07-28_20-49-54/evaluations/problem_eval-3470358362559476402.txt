def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=13, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the smoothed price-volume momentum
    pv_momentum_smoothed = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Compute the ratio between short and long exponential moving averages
    ema_ratio = short_ema / long_ema
    
    # Apply a custom heuristic to combine the EMA ratio and smoothed price-volume momentum
    heuristics_matrix = (ema_ratio + pv_momentum_smoothed).rank(pct=True)
    
    return heuristics_matrix
