def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    short_ma = df['close'].rolling(window=15).mean()
    long_ma = df['close'].rolling(window=60).mean()
    
    # Calculate the momentum of price changes over a 20-day period
    momentum = df['close'] - df['close'].shift(20)
    
    # Calculate the standard deviation of prices as a measure of volatility
    volatility = df['close'].rolling(window=30).std()
    
    # Adjust the volume by the volatility
    adjusted_volume = df['volume'] / volatility
    
    # Apply a custom heuristic to combine the momentum and the adjusted volume
    heuristics_matrix = (momentum + adjusted_volume).rank(pct=True)
    
    return heuristics_matrix
