import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) with a 20-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Money Flow Index (MFI) with a 20-day window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=20).sum() / negative_money_flow.rolling(window=20).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Combine the RSI and MFI, with RSI having more weight
    heuristics_matrix = (rsi * 0.7 + mfi * 0.3)
    
    return heuristics_matrix
