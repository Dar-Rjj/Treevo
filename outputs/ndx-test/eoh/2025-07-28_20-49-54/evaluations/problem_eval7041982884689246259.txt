import numpy as np

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    short_ma = df['close'].rolling(window=10).mean()
    long_ma = df['close'].rolling(window=50).mean()
    
    # Calculate the price-volume relationship
    pv_relationship = np.log(df['close'] * df['volume'])
    
    # Compute the difference of short and long moving averages
    ma_diff = short_ma - long_ma
    
    # Scale the moving average difference by the log of the price-volume relationship
    scaled_factor = ma_diff * pv_relationship
    
    # Rank the scaled factors
    heuristics_matrix = scaled_factor.rank(pct=True)
    
    return heuristics_matrix
