def heuristics_v2(df):
    # Calculate the 5-day Rate of Change (ROC)
    roc_5 = df['close'].pct_change(periods=5)
    
    # Calculate the 10-day and 20-day Simple Moving Averages (SMA)
    sma_10 = df['close'].rolling(window=10).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the 10-day and 20-day SMA
    sma_diff = sma_10 - sma_20
    
    # Apply a custom heuristic to combine the ROC and SMA difference
    heuristics_matrix = (roc_5 + sma_diff).rank(pct=True)
    
    return heuristics_matrix
