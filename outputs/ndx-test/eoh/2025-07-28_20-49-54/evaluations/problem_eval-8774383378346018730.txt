def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the modified price-momentum relationship
    pm_relationship = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Compute the difference between short and long exponential moving averages
    ema_diff = short_ema - long_ema
    
    # Apply a custom heuristic to combine the EMA difference and modified price-momentum relationship
    heuristics_matrix = (ema_diff + pm_relationship).rank(pct=True)
    
    return heuristics_matrix
