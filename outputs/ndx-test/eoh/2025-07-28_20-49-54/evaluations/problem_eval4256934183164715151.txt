def heuristics_v2(df):
    # Calculate the 10-day momentum
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the average daily return
    avg_daily_return = df['close'].pct_change().rolling(window=10).mean()
    
    # Apply a custom heuristic to combine the momentum and average daily return
    heuristics_matrix = (momentum + avg_daily_return).rank(pct=True)
    
    return heuristics_matrix
