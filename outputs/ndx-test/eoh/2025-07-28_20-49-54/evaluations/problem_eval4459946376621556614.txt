import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day momentum of the closing price
    momentum = df['close'].pct_change(periods=20)
    
    # Calculate the ratio of current volume to the average volume over the last 20 days
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    vol_ratio = df['volume'] / avg_volume_20
    
    # Apply a custom heuristic to combine the momentum and volume ratio
    heuristics_matrix = (momentum + vol_ratio).rank(pct=True)
    
    return heuristics_matrix
