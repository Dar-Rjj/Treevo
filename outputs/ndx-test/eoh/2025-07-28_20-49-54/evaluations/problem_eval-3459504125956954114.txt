import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the momentum of price changes
    price_momentum = df['close'].pct_change(periods=10)
    
    # Calculate the volatility of trading volume
    volume_volatility = df['volume'].rolling(window=10).std()
    
    # Combine the price momentum and volume volatility
    heuristics_matrix = (price_momentum + volume_volatility).rank(pct=True)
    
    return heuristics_matrix
