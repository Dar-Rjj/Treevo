def heuristics_v2(df):
    # Calculate the 14-period Relative Strength Index (RSI) of the closing price
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Calculate the 5-period Rate of Change (ROC) of the volume
    roc_vol_5 = df['volume'].pct_change(periods=5)
    
    # Apply a custom heuristic to combine the RSI and ROC of volume
    heuristics_matrix = (rsi_14 + roc_vol_5).rank(pct=True)
    
    return heuristics_matrix
