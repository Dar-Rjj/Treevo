import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average (EMA) of the close prices
    ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the daily returns
    returns = df['close'].pct_change()
    
    # Calculate the standard deviation of the returns over a 20-day window
    std_dev = returns.rolling(window=20).std()
    
    # Combine the EMA and the standard deviation to form the heuristics matrix
    heuristics_matrix = (ema + std_dev) / 2
    
    return heuristics_matrix
