import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day momentum
    momentum = df['close'].pct_change(10)
    
    # Calculate the 14-day average true range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Compute the ratio of momentum to ATR
    ratio = momentum / atr
    
    # Apply a custom heuristic to rank the stocks
    heuristics_matrix = ratio.rank(pct=True)
    
    return heuristics_matrix
