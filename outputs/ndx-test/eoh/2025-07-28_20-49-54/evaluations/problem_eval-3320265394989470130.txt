def heuristics_v2(df):
    # Calculate the percentage change of the closing price over 14 days
    pct_change_close = df['close'].pct_change(periods=14)
    
    # Calculate the Accumulation/Distribution Line (ADL)
    high_low_diff = df['high'] - df['low']
    close_low_diff = df['close'] - df['low'].shift(1)
    close_high_diff = df['close'] - df['high'].shift(1)
    adl = (close_low_diff / high_low_diff) * df['volume']
    adl = adl.cumsum()
    
    # Combine the percentage change of the closing price and the ADL
    heuristics_matrix = (pct_change_close + adl) / 2
    
    return heuristics_matrix
