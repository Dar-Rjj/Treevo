def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the adjusted price-volume momentum
    pv_momentum_adjusted = (df['close'] * df['volume']).rolling(window=5).mean() / df['volume'].rolling(window=5).mean()
    
    # Compute the ratio between short and long exponential moving averages
    ema_ratio = short_ema / long_ema
    
    # Combine the EMA ratio and adjusted price-volume momentum
    heuristics_matrix = (ema_ratio + pv_momentum_adjusted).rank(pct=True)
    
    return heuristics_matrix
