def heuristics_v2(df):
    # Calculate the 10-period Rate of Change (ROC) of the closing price
    roc_close = df['close'].pct_change(periods=10)
    
    # Calculate the 14-period Relative Strength Index (RSI) of the volume
    delta = df['volume'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Apply a custom heuristic to combine the ROC and RSI
    heuristics_matrix = (roc_close + rsi_volume).rank(pct=True)
    
    return heuristics_matrix
