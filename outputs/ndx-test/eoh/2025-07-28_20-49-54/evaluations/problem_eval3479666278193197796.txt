def heuristics_v2(df):
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the close-to-VWAP ratio
    cv_ratio = df['close'] / vwap
    
    # Compute the exponentially weighted moving average (EWMA) of the closing price
    ewma_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Apply a custom heuristic to combine the close-to-VWAP ratio and the EWMA of the closing price
    heuristics_matrix = (cv_ratio + ewma_close)
    
    return heuristics_matrix
