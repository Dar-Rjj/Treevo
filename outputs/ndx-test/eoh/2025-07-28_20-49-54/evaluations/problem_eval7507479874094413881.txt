def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the ratio of the difference between high and low prices to volume
    hlv_ratio = (df['high'] - df['low']) / df['volume']
    
    # Compute the ratio of short and long EMAs
    ema_ratio = short_ema / long_ema
    
    # Apply a custom heuristic to combine the EMA ratio and the high-low price to volume ratio
    heuristics_matrix = (ema_ratio + hlv_ratio).rank(pct=True)
    
    return heuristics_matrix
