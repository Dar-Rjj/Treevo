import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term momentum
    short_momentum = df['close'].pct_change(10)
    long_momentum = df['close'].pct_change(50)
    
    # Calculate the high-to-close price ratio
    hc_ratio = df['high'] / df['close']
    
    # Combine the momentum and high-to-close price ratio
    heuristics_matrix = (short_momentum - long_momentum + hc_ratio).rank(pct=True)
    
    return heuristics_matrix
