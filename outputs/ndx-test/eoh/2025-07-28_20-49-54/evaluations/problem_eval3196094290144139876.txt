import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change (ROC) over a 14-day period
    roc = df['close'].pct_change(periods=14)
    
    # Calculate the momentum-adjusted price-volume relationship
    momentum_pv = (df['close'] - df['close'].shift(14)) * df['volume']
    
    # Apply a custom heuristic to combine the ROC and momentum-adjusted price-volume relationship
    heuristics_matrix = (roc + momentum_pv).rank(pct=True)
    
    return heuristics_matrix
