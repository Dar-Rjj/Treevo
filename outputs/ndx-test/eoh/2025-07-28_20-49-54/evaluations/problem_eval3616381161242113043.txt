def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    short_ma = df['close'].rolling(window=10).mean()
    long_ma = df['close'].rolling(window=50).mean()
    
    # Calculate the price-volume relationship
    pv_relationship = df['close'] * df['volume']
    
    # Compute the ratio of short and long moving averages
    ma_ratio = short_ma / long_ma
    
    # Apply a custom heuristic to combine the moving average ratio and price-volume relationship
    heuristics_matrix = (ma_ratio + pv_relationship).rank(pct=True)
    
    return heuristics_matrix
