import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum of recent price changes
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the skewness of returns over a 30-day period
    daily_returns = df['close'].pct_change()
    skewness = daily_returns.rolling(window=30).skew()
    
    # Combine momentum and skewness into a single heuristic matrix
    heuristics_matrix = momentum + skewness
    
    return heuristics_matrix
