def heuristics_v2(df):
    # Calculate the close-to-open price ratio
    co_ratio = df['close'] / df['open']
    
    # Compute the Accumulation/Distribution Line
    adl = (df['close'] - df['low']) - (df['high'] - df['close']) * (df['close'] / (df['high'] - df['low'])) * df['volume']
    adl = adl.cumsum()
    
    # Apply a custom heuristic to combine the close-to-open price ratio and the Accumulation/Distribution Line
    heuristics_matrix = (co_ratio + adl).rank(pct=True)
    
    return heuristics_matrix
