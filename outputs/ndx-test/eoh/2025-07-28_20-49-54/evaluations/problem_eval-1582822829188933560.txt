def heuristics_v2(df):
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the high-to-low price ratio
    hl_ratio = df['high'] / df['low']
    
    # Compute the exponential moving average (EMA) of the closing price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Apply a custom heuristic to combine the high-to-low price ratio and the EMA of the closing price
    heuristics_matrix = (hl_ratio + ema_close).rank(pct=True)
    
    return heuristics_matrix
