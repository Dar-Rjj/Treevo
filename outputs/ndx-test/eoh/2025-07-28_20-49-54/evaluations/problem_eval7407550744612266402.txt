import pandas as pd

def heuristics_v2(df):
    # Compute the 20-day high and low
    df['high_20d'] = df['high'].rolling(window=20).max()
    df['low_20d'] = df['low'].rolling(window=20).min()
    
    # Calculate the difference between the 20-day high and low
    df['price_diff_20d'] = df['high_20d'] - df['low_20d']
    
    # Calculate the 20-day average volume
    df['avg_vol_20d'] = df['volume'].rolling(window=20).mean()
    
    # Adjust the price difference by the average volume
    df['heuristic_score'] = df['price_diff_20d'] / (df['avg_vol_20d'] + 1)  # Adding 1 to avoid division by zero
    
    # Drop rows with NaN values resulting from rolling calculations
    df.dropna(inplace=True)
    
    # Create heuristics_matrix as a Series
    heuristics_matrix = df['heuristic_score']
    
    return heuristics_matrix
