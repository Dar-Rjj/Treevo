import pandas as pd

def heuristics_v2(df):
    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Calculate the percentage price change over a 14-day period
    pct_change = df['close'].pct_change(periods=14)
    
    # Combine the OBV and percentage price change
    heuristics_matrix = (obv + pct_change)
    
    return heuristics_matrix
