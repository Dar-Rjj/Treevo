def heuristics_v2(df):
    short_sma = df['close'].rolling(window=20).mean()
    long_sma = df['close'].rolling(window=50).mean()
    pv_momentum_rolling = (df['close'] * df['volume']).rolling(window=14).mean()
    sma_diff = short_sma - long_sma
    heuristics_matrix = (sma_diff + pv_momentum_rolling).rank(pct=True)
    return heuristics_matrix
