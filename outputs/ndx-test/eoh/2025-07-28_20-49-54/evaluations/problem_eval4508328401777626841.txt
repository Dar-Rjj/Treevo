def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages
    short_sma = df['close'].rolling(window=15).mean()
    long_sma = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of price to volume over the last 7 days
    pv_ratio = (df['close'] / df['volume']).rolling(window=7).mean()
    
    # Compute the difference between short and long simple moving averages
    sma_diff = short_sma - long_sma
    
    # Apply a custom heuristic to combine the SMA difference and price-volume ratio
    heuristics_matrix = (sma_diff + pv_ratio).rank(pct=True)
    
    return heuristics_matrix
