import pandas as pd

def heuristics_v2(df):
    # Calculate the standard deviation of the close price over a 20-day window to represent short-term volatility
    price_volatility = df['close'].rolling(window=20).std()
    
    # Calculate the difference between the current volume and the average volume over a 20-day window
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_deviation = df['volume'] - avg_volume_20
    
    # Adjust the price volatility using the volume deviation
    adjusted_volatility = price_volatility * (1 + volume_deviation / avg_volume_20)
    
    heuristics_matrix = adjusted_volatility
    
    return heuristics_matrix
