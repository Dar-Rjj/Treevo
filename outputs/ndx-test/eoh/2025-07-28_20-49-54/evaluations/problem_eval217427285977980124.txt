def heuristics_v2(df):
    short_ema = df['close'].ewm(span=15, adjust=False).mean()
    long_ema = df['close'].ewm(span=30, adjust=False).mean()
    pv_momentum = (df['close'] * df['volume']).rolling(window=7).mean() / df['volume'].rolling(window=7).mean()
    ema_diff = short_ema - long_ema
    heuristics_matrix = (ema_diff + pv_momentum).rank(pct=True)
    return heuristics_matrix
