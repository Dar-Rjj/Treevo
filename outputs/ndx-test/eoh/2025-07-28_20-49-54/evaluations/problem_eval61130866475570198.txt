def heuristics_v2(df):
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=15, adjust=False).mean()
    pv_momentum_smoothed = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    ema_ratio = short_ema / long_ema
    heuristics_matrix = (ema_ratio + pv_momentum_smoothed).rank(pct=True)
    return heuristics_matrix
