def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    
    # Calculate the 10-day rate of change in volume
    volume_change = df['volume'] / df['volume'].shift(10) - 1
    
    # Compute the difference between short and long simple moving averages
    sma_diff = short_sma - long_sma
    
    # Apply a custom heuristic to combine the SMA difference and volume change
    heuristics_matrix = (sma_diff + volume_change).rank(pct=True)
    
    return heuristics_matrix
