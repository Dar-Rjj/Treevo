def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 5-day and 20-day moving averages of the closing price
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_20'] = df['close'].rolling(window=20).mean()
    
    # Generate signals based on moving average crossovers
    df['ma_signal'] = (df['ma_5'] > df['ma_20']).astype(int)
    
    # Calculate the volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Generate a combined heuristic factor
    df['heuristic_factor'] = df['return'] * df['volume_change'] * df['ma_signal']
    
    # Return the heuristic factor as a pandas Series
    heuristics_matrix = df['heuristic_factor'].dropna()
    return heuristics_matrix
