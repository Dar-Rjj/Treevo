def heuristics_v2(df):
    # Calculate the 50-period Simple Moving Average (SMA) of the closing price
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the 20-period Median of the Volume
    median_vol_20 = df['volume'].rolling(window=20).median()
    
    # Apply a custom heuristic to combine the SMA and Median of Volume
    heuristics_matrix = (sma_50 + median_vol_20).rank(pct=True)
    
    return heuristics_matrix
