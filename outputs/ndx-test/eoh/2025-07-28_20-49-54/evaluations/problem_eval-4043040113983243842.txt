import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in closing prices
    roc = df['close'].pct_change(periods=14)
    
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the rolling standard deviation of daily returns as a measure of volatility
    volatility = daily_returns.rolling(window=20).std()
    
    # Apply a custom heuristic to combine the rate of change and volatility
    heuristics_matrix = (roc + volatility).rank(pct=True)
    
    return heuristics_matrix
