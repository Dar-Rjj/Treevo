def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages
    short_sma = df['close'].rolling(window=12).mean()
    long_sma = df['close'].rolling(window=26).mean()
    
    # Calculate the ratio of closing price to the average of high and low prices
    av_ratio = df['close'] / ((df['high'] + df['low']) / 2)
    
    # Compute the difference between short-term and long-term simple moving averages
    sma_diff = short_sma - long_sma
    
    # Apply a custom heuristic to combine the SMA difference and the close-to-average price ratio
    heuristics_matrix = (sma_diff + av_ratio).rank(pct=True)
    
    return heuristics_matrix
