def heuristics_v2(df):
    # Calculate the 20-period Exponential Moving Average (EMA) of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 21-period True Range (TR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    
    # Calculate the 21-period Exponential Moving Average (EMA) of the True Range (TR)
    ema_tr_21 = tr.ewm(span=21, adjust=False).mean()
    
    # Apply a custom heuristic to combine the EMA and EMA of TR
    heuristics_matrix = (ema_20 + ema_tr_21).rank(pct=True)
    
    return heuristics_matrix
