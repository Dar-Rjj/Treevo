import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponentially weighted moving averages
    short_ewma = df['close'].ewm(span=10).mean()
    long_ewma = df['close'].ewm(span=50).mean()
    
    # Calculate the modified price-volume relationship
    pv_relationship = (df['close'] - df['open']) / df['volume']
    
    # Compute the difference between short and long exponentially weighted moving averages
    ewma_diff = short_ewma - long_ewma
    
    # Apply a custom heuristic to combine the exponentially weighted moving average difference and modified price-volume relationship
    heuristics_matrix = (ewma_diff + pv_relationship)
    
    return heuristics_matrix
