def heuristics_v2(df):
    # Calculate the rate of change (ROC) over a 5-day period
    roc_5 = df['close'].pct_change(periods=5)
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Combine ROC and RSI to form the composite factor
    heuristics_matrix = (roc_5 + rsi_14).rank(pct=True)
    
    return heuristics_matrix
