import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day moving average of the closing price
    ma_close = df['close'].rolling(window=10).mean()
    
    # Calculate the 20-day exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine the factors
    heuristics_matrix = ma_close * 0.6 + ema_volume * 0.4
    
    return heuristics_matrix
