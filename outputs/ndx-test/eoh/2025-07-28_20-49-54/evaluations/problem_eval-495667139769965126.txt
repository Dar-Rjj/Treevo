def heuristics_v2(df):
    heuristics_matrix = (df['close'].pct_change()  # Momentum
                        + df['close'].rolling(window=10).std()  # Volatility
                        + (df['close'].rolling(window=5).mean() > df['close'].rolling(window=20).mean()).astype(int))  # Moving Average Crossover
    return heuristics_matrix
