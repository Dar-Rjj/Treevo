import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-period return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-period standard deviation of returns
    std_dev_returns = df['return'].rolling(window=10).std()
    
    # Calculate the 20-day average true range
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    avg_true_range = df['tr'].rolling(window=20).mean()
    
    # Compute the volatility-adjusted momentum indicator
    vol_adj_momentum = (df['close'] - df['close'].shift(20)) / avg_true_range
    
    # Apply a custom heuristic to combine the standard deviation of returns and volatility-adjusted momentum
    heuristics_matrix = (std_dev_returns + vol_adj_momentum).rank(pct=True)
    
    return heuristics_matrix
