def heuristics_v2(df):
    # Calculate the 10-period rate of change of the closing price
    roc_10 = df['close'].pct_change(periods=10)
    
    # Calculate the 14-period Average True Range (ATR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr_14 = tr.rolling(window=14).mean()
    
    # Apply a custom heuristic to combine the ROC and ATR
    heuristics_matrix = (roc_10 + atr_14).rank(pct=True)
    
    return heuristics_matrix
