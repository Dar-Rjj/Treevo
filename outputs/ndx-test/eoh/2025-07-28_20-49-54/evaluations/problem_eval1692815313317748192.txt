import pandas as pd

def heuristics_v2(df):
    # Calculate the volume-adjusted returns
    adj_returns = df['close'].pct_change() * df['volume']
    
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the volume-adjusted returns and RSI
    heuristics_matrix = adj_returns + rsi
    
    return heuristics_matrix
