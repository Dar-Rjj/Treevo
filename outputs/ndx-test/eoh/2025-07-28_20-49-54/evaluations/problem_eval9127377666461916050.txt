def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14)
    
    # Calculate the Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Combine ROC and ADL
    heuristics_matrix = roc + adl
    
    return heuristics_matrix
