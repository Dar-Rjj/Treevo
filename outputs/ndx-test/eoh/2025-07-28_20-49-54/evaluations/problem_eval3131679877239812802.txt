import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the close-to-low price ratio
    cl_ratio = df['close'] / df['low']
    
    # Compute the difference between long and short exponential moving averages
    ema_diff = long_ema - short_ema
    
    # Apply a custom heuristic to combine the EMA difference and close-to-low price ratio
    heuristics_matrix = (ema_diff + cl_ratio).rank(pct=True)
    
    return heuristics_matrix
