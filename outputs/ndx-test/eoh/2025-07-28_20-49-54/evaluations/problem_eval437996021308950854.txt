import pandas as pd

def heuristics_v2(df):
    def ema(data, window):
        return data.ewm(span=window, adjust=False).mean()
    
    # Calculate EMA for volume
    ema_volume = ema(df['volume'], 10)
    
    # Calculate the difference between high and low
    diff_high_low = df['high'] - df['low']
    
    # Calculate EMA for the difference between high and low
    ema_diff_high_low = ema(diff_high_low, 10)
    
    # Calculate a simple time-series momentum indicator
    momentum = df['close'].pct_change(20)
    
    # Combine the factors
    heuristics_matrix = ema_volume + ema_diff_high_low + momentum
    
    return heuristics_matrix
