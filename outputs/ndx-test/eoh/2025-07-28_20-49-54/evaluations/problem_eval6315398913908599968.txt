def heuristics_v2(df):
    price_range = df['high'] - df['low']
    adj_close = df['close'] * (1 + df['close'].pct_change().fillna(0))  # Approximation for demonstration
    price_ratio = price_range / adj_close
    abs_daily_returns = df['close'].pct_change().abs()
    modified_volatility = abs_daily_returns.rolling(window=20).mean()
    heuristics_matrix = (price_ratio + modified_volatility)
    return heuristics_matrix
