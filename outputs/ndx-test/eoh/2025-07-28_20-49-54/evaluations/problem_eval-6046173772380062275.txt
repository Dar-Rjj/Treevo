def heuristics_v2(df):
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the high-to-VWAP ratio
    hv_ratio = df['high'] / vwap
    
    # Compute the simple moving average (SMA) of the closing price
    sma_close = df['close'].rolling(window=10).mean()
    
    # Apply a custom heuristic to combine the high-to-VWAP ratio and the SMA of the closing price
    heuristics_matrix = (hv_ratio + sma_close).rank(pct=True)
    
    return heuristics_matrix
