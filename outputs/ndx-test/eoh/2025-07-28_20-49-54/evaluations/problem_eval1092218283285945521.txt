def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    RSI = 100 - (100 / (1 + RS))
    
    # Calculate MACD and its signal line
    exp1 = df['close'].ewm(span=12, adjust=False).mean()
    exp2 = df['close'].ewm(span=26, adjust=False).mean()
    macd = exp1-exp2
    signal = macd.ewm(span=9, adjust=False).mean()
    
    # Adjust the volume by the MACD
    adjusted_volume = df['volume'] * (macd - signal)
    
    # Apply a custom heuristic to combine the RSI and the MACD-adjusted volume
    heuristics_matrix = (RSI + adjusted_volume).rank(pct=True)
    
    return heuristics_matrix
