def heuristics_v2(df):
    # Calculate the closing-to-opening price ratio
    co_ratio = df['close'] / df['open']
    
    # Compute the short-term and long-term exponential moving averages
    ema_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the difference between the short-term and long-term EMAs
    ema_diff = ema_short - ema_long
    
    # Apply a custom heuristic to combine the closing-to-opening price ratio and the EMA difference
    heuristics_matrix = (co_ratio + ema_diff).rank(pct=True)
    
    return heuristics_matrix
