def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    # Calculate the ratio of closing price to volume
    cpv_ratio = df['close'] / df['volume']
    
    # Compute the ratio of short and long EMAs
    ema_ratio = short_ema / long_ema
    
    # Apply a custom heuristic to combine the EMA ratio and closing price to volume ratio
    heuristics_matrix = (ema_ratio + cpv_ratio).rank(pct=True)
    
    return heuristics_matrix
