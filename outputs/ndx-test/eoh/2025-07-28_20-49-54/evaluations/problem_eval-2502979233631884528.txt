def heuristics_v2(df):
    # Calculate the simple moving averages
    sma_50 = df['close'].rolling(window=50).mean()
    sma_100 = df['close'].rolling(window=100).mean()
    
    # Calculate the high-to-close ratio
    hc_ratio = df['high'] / df['close']
    
    # Compute the difference between the 50-period and 100-period simple moving averages
    sma_diff = sma_50 - sma_100
    
    # Apply a custom heuristic to combine the high-to-close ratio and the difference between SMAs
    heuristics_matrix = (hc_ratio + sma_diff).rank(pct=True)
    
    return heuristics_matrix
