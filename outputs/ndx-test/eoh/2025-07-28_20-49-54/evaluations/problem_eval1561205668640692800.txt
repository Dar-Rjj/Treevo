import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=9, adjust=False).mean()
    long_ema = df['close'].ewm(span=21, adjust=False).mean()
    
    # Calculate the momentum-adjusted price-volume relationship
    momentum = df['close'].pct_change(periods=5)
    pv_relationship_adjusted = (df['close'] * df['volume']) * (1 + momentum)
    
    # Compute the ratio between short and long exponential moving averages
    ema_ratio = short_ema / long_ema
    
    # Apply a custom heuristic to combine the EMA ratio and momentum-adjusted price-volume relationship
    heuristics_matrix = (ema_ratio + pv_relationship_adjusted).rank(pct=True)
    
    return heuristics_matrix
