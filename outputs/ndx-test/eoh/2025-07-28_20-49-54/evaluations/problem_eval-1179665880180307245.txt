def heuristics_v2(df):
    pct_change_close = df['close'].pct_change()
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    vol_ratio = df['volume'] / avg_volume_30
    std_dev_close_10 = df['close'].rolling(window=10).std()
    weight = 1 / (1 + std_dev_close_10)
    heuristics_matrix = (pct_change_close * weight) + (vol_ratio * (1 - weight))
    return heuristics_matrix
