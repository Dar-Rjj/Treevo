def heuristics_v2(df):
    # Calculate the 20-period Exponential Moving Average (EMA) of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 28-period True Range (TR)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    
    # Calculate the 28-period Standard Deviation of the True Range (TR)
    std_tr_28 = tr.rolling(window=28).std()
    
    # Apply a custom heuristic to combine the EMA and Standard Deviation of TR
    heuristics_matrix = (ema_20 + std_tr_28).rank(pct=True)
    
    return heuristics_matrix
