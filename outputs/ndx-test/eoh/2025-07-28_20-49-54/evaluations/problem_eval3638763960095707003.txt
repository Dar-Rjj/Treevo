import pandas as pd

def heuristics_v2(df):
    # Compute short and long term moving averages
    short_ma = df['close'].rolling(window=10).mean()
    long_ma = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between short and long moving averages
    ma_diff = short_ma - long_ma
    
    # Compute the ratio of current volume to the average volume over 20 days
    vol_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine the moving average difference and volume ratio
    heuristics_matrix = (ma_diff + vol_ratio) / 2
    
    return heuristics_matrix
