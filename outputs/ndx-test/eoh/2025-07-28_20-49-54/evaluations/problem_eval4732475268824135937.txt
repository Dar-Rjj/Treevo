import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day percentage change in closing price
    pct_change_10 = df['close'].pct_change(periods=10)
    
    # Calculate the current day's range (high - low)
    daily_range = df['high'] - df['low']
    
    # Calculate the average range over the past 20 days
    avg_range_20 = daily_range.rolling(window=20).mean()
    
    # Calculate the ratio of the current day's range to the average range over the past 20 days
    range_ratio = daily_range / avg_range_20
    
    # Combine the percentage change and range ratio
    heuristics_matrix = pct_change_10 + range_ratio
    
    return heuristics_matrix
