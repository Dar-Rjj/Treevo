def heuristics_v2(df):
    # Calculate short-term and long-term simple moving averages
    short_sma = df['close'].rolling(window=15).mean()
    long_sma = df['close'].rolling(window=30).mean()
    
    # Calculate the weighted sum of price and volume over the last 7 days
    pv_weighted_sum = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Compute the difference between short and long simple moving averages
    sma_diff = short_sma - long_sma
    
    # Apply a custom heuristic to combine the SMA difference and weighted price-volume sum
    heuristics_matrix = (sma_diff + pv_weighted_sum)
    
    return heuristics_matrix
