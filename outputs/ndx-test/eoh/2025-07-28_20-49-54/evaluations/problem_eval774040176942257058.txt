import pandas as pd

def heuristics_v2(df):
    # Calculate the exponentially weighted moving average of the close price
    ewma_close = df['close'].ewm(span=20).mean()
    
    # Compute the difference between the close price and its EWMA
    ewma_diff = df['close'] - ewma_close
    
    # Calculate the percentage change in volume
    vol_change = df['volume'].pct_change()
    
    # Adjust the EWMA difference using the volume change
    adjusted_signal = ewma_diff * (1 + vol_change)
    
    heuristics_matrix = adjusted_signal
    
    return heuristics_matrix
