def heuristics_v2(df):
    # Calculate the rate of change of the closing price over a 14-day period
    roc = df['close'].pct_change(periods=14)
    
    # Calculate the ratio of close to open prices
    close_open_ratio = df['close'] / df['open']
    
    # Apply a custom heuristic to combine the ROC and the close-to-open ratio
    heuristics_matrix = (0.7 * roc + 0.3 * close_open_ratio).rank(pct=True)
    
    return heuristics_matrix
