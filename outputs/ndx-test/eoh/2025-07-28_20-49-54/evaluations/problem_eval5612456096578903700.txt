def heuristics_v2(df):
    short_ema = df['close'].ewm(span=20, adjust=False).mean()
    long_ema = df['close'].ewm(span=60, adjust=False).mean()
    vw_close = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    ema_ratio = short_ema / long_ema
    heuristics_matrix = (ema_ratio + vw_close).rank(pct=True)
    return heuristics_matrix
