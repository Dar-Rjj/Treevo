import pandas as pd
import numpy as np

def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    prev_close = df['close'].shift(1)
    price_ratio = high_low_range / prev_close
    min_volume_30 = df['volume'].rolling(window=30).min()
    vol_log_ratio = np.log(df['volume'] / min_volume_30)
    heuristics_matrix = price_ratio - vol_log_ratio
    return heuristics_matrix
