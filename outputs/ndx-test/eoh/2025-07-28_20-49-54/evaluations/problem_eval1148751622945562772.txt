import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the close-to-average-high-low price ratio
    avg_high_low = (df['high'] + df['low']) / 2
    chl_ratio = df['close'] / avg_high_low
    
    # Apply a custom heuristic to combine the momentum and close-to-average-high-low price ratio
    heuristics_matrix = (momentum + chl_ratio).rank(pct=True)
    
    return heuristics_matrix
