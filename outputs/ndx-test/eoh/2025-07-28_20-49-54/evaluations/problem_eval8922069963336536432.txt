def heuristics_v2(df):
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the low-to-VWAP ratio
    lv_ratio = df['low'] / vwap
    
    # Compute the exponential moving average (EMA) of the closing price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Apply a custom heuristic to combine the low-to-VWAP ratio and the EMA of the closing price
    heuristics_matrix = (lv_ratio + ema_close).rank(pct=True)
    
    return heuristics_matrix
