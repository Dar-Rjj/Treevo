import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Calculate the Volume-Adjusted Volatility
    volatility = df['close'].rolling(window=10).std()
    volume_adjusted_volatility = volatility * df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the ROC and Volume-Adjusted Volatility
    heuristics_matrix = roc + volume_adjusted_volatility
    
    return heuristics_matrix
