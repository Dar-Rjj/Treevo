def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) for the closing price
    roc_close = df['close'].pct_change(periods=10)
    
    # Calculate the 10-day standard deviation of the volume
    vol_std = df['volume'].rolling(window=10).std()
    
    # Apply a custom heuristic to combine the ROC of the closing price and the standard deviation of the volume
    heuristics_matrix = (roc_close + vol_std).rank(pct=True)
    
    return heuristics_matrix
