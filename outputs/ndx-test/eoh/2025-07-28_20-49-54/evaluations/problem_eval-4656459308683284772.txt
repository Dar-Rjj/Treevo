import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term exponential moving averages
    short_ema = df['close'].ewm(span=9, adjust=False).mean()
    long_ema = df['close'].ewm(span=21, adjust=False).mean()
    
    # Calculate the 5-day change in closing price
    momentum_signal = df['close'] / df['close'].shift(5) - 1
    
    # Compute the ratio of short to long exponential moving averages
    ema_ratio = short_ema / long_ema
    
    # Apply a custom heuristic to combine the EMA ratio and momentum signal
    heuristics_matrix = (ema_ratio * momentum_signal).rank(pct=True)
    
    return heuristics_matrix
