def heuristics_v2(df):
    # Calculate 30-day and 150-day simple moving averages
    sma_30 = df['close'].rolling(window=30).mean()
    sma_150 = df['close'].rolling(window=150).mean()
    sma_diff = sma_30 - sma_150

    # Calculate Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Combine SMA difference and CMF into a single heuristic factor
    heuristics_matrix = sma_diff + cmf

    return heuristics_matrix
