def heuristics_v2(df):
    # Calculate 20-day and 60-day rate of change (ROC)
    roc_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20) * 100
    roc_60 = (df['close'] - df['close'].shift(60)) / df['close'].shift(60) * 100
    roc_diff = roc_20 - roc_60

    # Calculate 14-period Commodity Channel Index (CCI)
    tp = (df['high'] + df['low'] + df['close']) / 3
    tp_sma_14 = tp.rolling(window=14).mean()
    md_14 = tp.rolling(window=14).apply(lambda x: np.fabs(x - x.mean()).mean())
    cci_14 = (tp - tp_sma_14) / (0.015 * md_14)

    # Combine ROC difference and CCI into a single heuristic factor
    heuristics_matrix = roc_diff + cci_14

    return heuristics_matrix
