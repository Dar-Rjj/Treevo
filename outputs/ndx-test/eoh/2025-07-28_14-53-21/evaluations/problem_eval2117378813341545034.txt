import pandas as pd

def heuristics_v2(df):
    # Calculate the volume-adjusted close price
    volume_adjusted_close = df['close'] * df['volume']
    
    # Calculate the 10-day momentum of the close price
    momentum = df['close'].diff(periods=10)
    
    # Compute the heuristics
    heuristics_matrix = 0.6 * volume_adjusted_close + 0.4 * momentum
    
    return heuristics_matrix
