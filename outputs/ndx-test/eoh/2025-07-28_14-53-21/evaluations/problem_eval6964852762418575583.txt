import pandas as pd

def heuristics_v2(df):
    def modified_macd(series, fast=13, slow=26, signal=9):
        exp1 = series.ewm(span=fast, adjust=False).mean()
        exp2 = series.ewm(span=slow, adjust=False).mean()
        macd = exp1 - exp2
        signal_line = macd.ewm(span=signal, adjust=False).mean()
        return macd - signal_line

    def rsi(series, periods=20):
        delta = series.diff(1)
        gain = delta.where(delta > 0, 0)
        loss = -delta.where(delta < 0, 0)
        avg_gain = gain.rolling(window=periods, min_periods=1).mean()
        avg_loss = loss.rolling(window=periods, min_periods=1).mean()
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))
    
    close_prices = df['close']
    modified_macd_values = modified_macd(close_prices)
    rsi_values = rsi(close_prices)
    heuristics_matrix = 0.5 * modified_macd_values + 0.5 * rsi_values
    return heuristics_matrix
