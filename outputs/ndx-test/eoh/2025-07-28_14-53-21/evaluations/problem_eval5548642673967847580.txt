def heuristics_v2(df):
    # Calculate 50-day and 200-day exponential moving averages
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Generate EMA crossover signal
    ema_crossover = (ema_50 > ema_200).astype(int)

    # Calculate typical price and money flow
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']

    # Positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)

    # Sum over 14 days
    positive_sum = positive_money_flow.rolling(window=14).sum()
    negative_sum = negative_money_flow.rolling(window=14).sum()

    # Calculate the ratio of positive to negative money flow
    money_flow_ratio = positive_sum / negative_sum

    # Volume condition: only consider days with volume above the 30-day moving average
    volume_condition = (df['volume'] > df['volume'].rolling(window=30).mean()).astype(int)

    # Combine EMA crossover signal, money flow ratio, and volume condition into a single heuristic factor
    heuristics_matrix = ema_crossover * money_flow_ratio * volume_condition

    return heuristics_matrix
