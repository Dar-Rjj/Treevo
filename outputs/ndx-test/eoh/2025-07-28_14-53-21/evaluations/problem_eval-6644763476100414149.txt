import pandas as pd

def heuristics_v2(df):
    def volume_weighted_price(high, low, close, volume):
        vwp = (high + low + close) / 3 * volume
        return vwp

    def rate_of_change(series, window=14):
        roc = (series - series.shift(window)) / series.shift(window) * 100
        return roc

    high = df['high']
    low = df['low']
    close = df['close']
    volume = df['volume']

    vwp = volume_weighted_price(high, low, close, volume)
    roc_values = rate_of_change(vwp)

    heuristics_matrix = 0.5 * vwp + 0.5 * roc_values
    return heuristics_matrix
