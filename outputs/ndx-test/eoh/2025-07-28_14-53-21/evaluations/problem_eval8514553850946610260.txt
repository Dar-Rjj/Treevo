import pandas as pd

def heuristics_v2(df):
    # Calculate 14-day and 28-day Stochastic Oscillators
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_14 = 100 * (df['close'] - low_14) / (high_14 - low_14)
    
    low_28 = df['low'].rolling(window=28).min()
    high_28 = df['high'].rolling(window=28).max()
    k_28 = 100 * (df['close'] - low_28) / (high_28 - low_28)
    
    stoch_osc_diff = k_14 - k_28
    
    # Calculate 30-day Average True Range (ATR)
    tr = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    atr = tr.rolling(window=30).mean()
    
    # Combine Stochastic Oscillator difference and ATR into a single heuristic factor
    heuristics_matrix = (stoch_osc_diff + atr) / 2
    
    return heuristics_matrix
