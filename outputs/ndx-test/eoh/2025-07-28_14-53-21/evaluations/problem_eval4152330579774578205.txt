import pandas as pd
import numpy as np

def heuristics_v2(df):
    price_momentum = df['close'] - df['open']
    trading_range = (df['high'] - df['low']) * df['volume']
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=5).std()
    
    # Calculate the overall standard deviation for each component to use as weights
    weight_pm = 1 / price_momentum.std()
    weight_tr = 1 / trading_range.std()
    weight_vol = 1 / volatility.std()
    
    # Compute the heuristic factor
    heuristics_matrix = (price_momentum * weight_pm) + (trading_range * weight_tr) + (volatility * weight_vol)
    
    return heuristics_matrix
