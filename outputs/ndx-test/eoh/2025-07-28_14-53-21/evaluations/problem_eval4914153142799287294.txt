def heuristics_v2(df):
    # Calculate log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day rolling standard deviation (volatility)
    volatility = df['log_return'].rolling(window=20).std()
    
    # Adjust log returns by volatility
    adjusted_return = df['log_return'] / volatility
    
    # Calculate 10-day weighted moving average of volume
    weights = np.array([i for i in range(1, 11)])
    wma_volume = df['volume'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Combine adjusted return and weighted moving average of volume
    heuristics_matrix = adjusted_return + wma_volume

    return heuristics_matrix
