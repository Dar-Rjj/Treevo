def heuristics_v2(df):
    # Calculate Rate of Change (ROC) over a 10-day period
    roc = df['close'].pct_change(periods=10)
    
    # Calculate On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    obv_normalized = obv / obv.rolling(window=50).mean()
    
    # Calculate Average True Range (ATR) over a 14-day period
    tr = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': np.abs(df['high'] - df['close'].shift(1)),
        'l-pc': np.abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine ROC, normalized OBV, and ATR into a single heuristic factor
    heuristics_matrix = roc + obv_normalized - atr
    
    return heuristics_matrix
