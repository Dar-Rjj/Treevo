def heuristics_v2(df):
    # Calculate momentum as the difference between the 5-day and 20-day moving average of close prices
    momentum = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    
    # Calculate a simple mean reversion signal as the inverse of the recent 5-day return
    mean_reversion = -1 * (df['close'] / df['close'].shift(5) - 1)
    
    # Adjust for trading volume by multiplying the mean reversion and momentum indicators with the ratio of current volume to 5-day average volume
    volume_adjustment = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Combine the momentum, mean reversion, and adjusted volume into a single heuristic factor
    heuristics_matrix = (momentum + mean_reversion) * volume_adjustment
    
    return heuristics_matrix
