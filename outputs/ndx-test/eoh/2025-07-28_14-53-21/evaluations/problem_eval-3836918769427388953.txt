import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day and 200-day simple moving averages
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    sma_ratio = sma_50 / sma_200

    # Calculate Accumulation/Distribution Oscillator (ADOSC)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    adosc_14 = money_flow_volume.rolling(window=14).sum() - money_flow_volume.rolling(window=28).sum()
    adosc_28 = money_flow_volume.rolling(window=28).sum() - money_flow_volume.rolling(window=56).sum()
    adosc_diff = adosc_14 - adosc_28

    # Combine SMA ratio and ADOSC difference into a single heuristic factor
    heuristics_matrix = sma_ratio + adosc_diff

    return heuristics_matrix
