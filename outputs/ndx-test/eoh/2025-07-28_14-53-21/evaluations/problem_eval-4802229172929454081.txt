def heuristics_v2(df):
    # Calculate RSI with a 20-day window
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate MACD with spans of 9 and 18
    exp9 = df['close'].ewm(span=9, adjust=False).mean()
    exp18 = df['close'].ewm(span=18, adjust=False).mean()
    macd = exp9 - exp18
    signal = macd.rolling(window=9).mean()

    # Calculate smoothed volatility as the exponential moving average of the standard deviation of the last 30 days' returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=30).std().ewm(span=5, adjust=False).mean()

    # Combine RSI, MACD signal line, and smoothed volatility into a single heuristic factor with different weights
    heuristics_matrix = 2*rsi + 0.5*signal - 0.75*volatility

    return heuristics_matrix
