import pandas as pd

def heuristics_v2(df):
    price_change = df['close'].diff()
    volume_change = df['volume'].diff()
    momentum = price_change.rolling(window=10).mean()
    volume_trend = volume_change.rolling(window=10).mean()
    heuristics_matrix = (momentum * volume_trend).dropna()
    return heuristics_matrix
