def heuristics_v2(df):
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()

    # Calculate Rate of Change (ROC) over 12 periods
    roc = df['close'].pct_change(periods=12)

    # Combine OBV and ROC into a single heuristic factor
    heuristics_matrix = obv + roc

    return heuristics_matrix
