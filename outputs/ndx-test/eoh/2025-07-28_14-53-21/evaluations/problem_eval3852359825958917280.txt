import pandas as pd

def heuristics_v2(df):
    def macd(series, fast=12, slow=26, signal=9):
        exp1 = series.ewm(span=fast, adjust=False).mean()
        exp2 = series.ewm(span=slow, adjust=False).mean()
        macd_line = exp1 - exp2
        signal_line = macd_line.ewm(span=signal, adjust=False).mean()
        return macd_line - signal_line

    def rsi_vol_adj(series, periods=14, std_periods=20):
        delta = series.diff(1)
        gain = (delta + delta.abs()) / 2
        loss = (delta.abs() - delta) / 2
        avg_gain = gain.rolling(window=periods, min_periods=1).mean()
        avg_loss = loss.rolling(window=periods, min_periods=1).mean()
        rs = avg_gain / avg_loss
        rsi_base = 100 - (100 / (1 + rs))
        std_dev = series.rolling(window=std_periods).std()
        return rsi_base * std_dev

    close_prices = df['close']
    macd_values = macd(close_prices)
    rsi_vol_adj_values = rsi_vol_adj(close_prices)
    heuristics_matrix = 0.5 * macd_values + 0.5 * rsi_vol_adj_values
    return heuristics_matrix
