def heuristics_v2(df):
    # Calculate 10-day and 30-day Stochastic Oscillator
    low_14 = df['low'].rolling(window=10).min()
    high_14 = df['high'].rolling(window=10).max()
    k_10 = 100 * (df['close'] - low_14) / (high_14 - low_14)
    d_10 = k_10.rolling(window=3).mean()

    low_14 = df['low'].rolling(window=30).min()
    high_14 = df['high'].rolling(window=30).max()
    k_30 = 100 * (df['close'] - low_14) / (high_14 - low_14)
    d_30 = k_30.rolling(window=3).mean()

    stoch_diff = d_10 - d_30

    # Calculate 100-day SMA
    sma_100 = df['close'].rolling(window=100).mean()
    
    # Combine Stochastic Oscillator and SMA
    heuristics_matrix = stoch_diff + (sma_100 - df['close'])
    
    return heuristics_matrix
