def heuristics_v2(df):
    # Calculate 30-day and 150-day exponential moving averages
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_150 = df['close'].ewm(span=150, adjust=False).mean()
    ema_diff = ema_30 - ema_150

    # Calculate Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_sum = positive_money_flow.rolling(window=21).sum()
    negative_sum = negative_money_flow.rolling(window=21).sum()
    mfi = 100 - (100 / (1 + (positive_sum / negative_sum)))

    # Combine EMA difference and MFI into a single heuristic factor
    heuristics_matrix = ema_diff + mfi

    return heuristics_matrix
