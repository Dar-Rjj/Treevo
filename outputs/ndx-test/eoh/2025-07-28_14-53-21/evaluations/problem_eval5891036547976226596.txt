def heuristics_v2(df):
    # Calculate 20-day and 100-day exponential moving averages
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_diff_percentage = (ema_20 - ema_100) / ema_100

    # Calculate Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_sum = positive_money_flow.rolling(window=7).sum()
    negative_sum = negative_money_flow.rolling(window=7).sum()
    mfi = 100 - (100 / (1 + (positive_sum / negative_sum)))

    # Combine EMA difference percentage and MFI into a single heuristic factor
    heuristics_matrix = ema_diff_percentage + mfi

    return heuristics_matrix
