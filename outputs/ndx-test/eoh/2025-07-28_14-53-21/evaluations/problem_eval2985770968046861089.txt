import pandas as pd

def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()
    df['momentum_divergence'] = (df['price_change'] > 0) & (df['volume_change'] < 0)
    heuristics_matrix = df['momentum_divergence'].astype(int)
    return heuristics_matrix
