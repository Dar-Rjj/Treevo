import pandas as pd

def heuristics_v2(df):
    def calculate_volatility(series, window=5):
        return series.rolling(window=window).std()

    # Calculate the volume-adjusted close price
    volume_adjusted_close = df['close'] * df['volume']
    
    # Calculate the volatility of the high-low range
    high_low_volatility = calculate_volatility(df['high'] - df['low'])
    
    # Compute the heuristics
    heuristics_matrix = 0.6 * volume_adjusted_close + 0.4 * high_low_volatility
    
    return heuristics_matrix
