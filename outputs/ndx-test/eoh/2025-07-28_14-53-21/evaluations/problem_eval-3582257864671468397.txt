def heuristics_v2(df):
    # Calculate Stochastic Oscillator
    high_14 = df['high'].rolling(window=14).max()
    low_14 = df['low'].rolling(window=14).min()
    k = 100 * (df['close'] - low_14) / (high_14 - low_14)
    
    # Calculate Volume-Weighted Moving Average (VWMA)
    vwma = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Combine Stochastic Oscillator and VWMA into a single heuristic factor with different weights
    heuristics_matrix = 0.6 * k + 0.4 * (vwma - df['close'])

    return heuristics_matrix
