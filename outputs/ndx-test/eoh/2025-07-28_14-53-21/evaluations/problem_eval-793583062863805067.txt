def heuristics_v2(df):
    # Calculate the daily percentage change in the closing price
    price_change = df['close'].pct_change()
    
    # Compute the ratio of the current volume to the maximum volume over a 30-day rolling window
    max_volume_30d = df['volume'].rolling(window=30).max()
    volume_ratio = df['volume'] / max_volume_30d
    
    # Combine the price change with the volume ratio to form the heuristic matrix
    heuristics_matrix = price_change * volume_ratio
    
    return heuristics_matrix
