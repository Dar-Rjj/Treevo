def heuristics_v2(df):
    # Calculate 20-Day Rate of Change
    roc_20 = (df['close'] - df['close'].shift(20)) / df['close'].shift(20) * 100
    
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change()
    
    # Calculate 30-Day Standard Deviation of Daily Returns
    std_dev_30 = daily_returns.rolling(window=30).std()
    
    # Combine ROC and Std Dev into a single heuristic factor
    heuristics_matrix = roc_20 + std_dev_30

    return heuristics_matrix
