def heuristics_v2(df):
    # Calculate Rate of Change (ROC) over 21 days
    roc = df['close'].pct_change(periods=21)
    
    # Calculate Chaikin Money Flow (CMF) over 20 days
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mfm * df['volume']
    cmf = mf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine ROC and CMF to form the heuristics matrix
    heuristics_matrix = roc + cmf
    
    return heuristics_matrix
