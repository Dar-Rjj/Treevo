def heuristics_v2(df):
    roc = (df['close'] - df['close'].shift(20)) / df['close'].shift(20) * 100
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_300 = df['close'].ewm(span=300, adjust=False).mean()
    ema_ratio = ema_100 / ema_300
    heuristics_matrix = roc * ema_ratio
    return heuristics_matrix
