def heuristics_v2(df):
    # Calculate 21-day Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=21).sum() / df['volume'].rolling(window=21).sum()

    # Calculate 50-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=50)

    # Calculate 14-day moving average of the standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_14 = daily_returns.rolling(window=14).std()

    # Combine all factors into a single heuristic factor
    heuristics_matrix = cmf + roc + std_dev_14

    return heuristics_matrix
