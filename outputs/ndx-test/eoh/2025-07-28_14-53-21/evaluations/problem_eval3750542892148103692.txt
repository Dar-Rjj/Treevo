def heuristics_v2(df):
    # Calculate ROC
    roc = df['close'].pct_change(periods=14)
    
    # Calculate VWMA
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap_sum = (tp * df['volume']).rolling(window=14).sum()
    vol_sum = df['volume'].rolling(window=14).sum()
    vwma = vwap_sum / vol_sum
    
    # Combine ROC and VWMA into a single heuristic factor
    heuristics_matrix = 0.5 * roc + 0.5 * (vwma - df['close'])
    
    return heuristics_matrix
