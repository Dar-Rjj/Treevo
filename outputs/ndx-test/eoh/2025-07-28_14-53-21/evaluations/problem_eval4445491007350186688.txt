import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day EMA
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    # Compute the ratio of the closing price to its 50-day EMA
    close_to_ema_ratio = df['close'] / ema_50
    
    # Calculate the 20-day coefficient of variation for volume
    vol_mean = df['volume'].rolling(window=20).mean()
    vol_std = df['volume'].rolling(window=20).std()
    vol_coeff_var = vol_std / vol_mean
    
    # Combine the two factors
    heuristics_matrix = close_to_ema_ratio + vol_coeff_var
    
    return heuristics_matrix
