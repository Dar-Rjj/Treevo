def heuristics_v2(df):
    # Calculate 50-day simple moving average
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between close and 50-day SMA
    diff_close_sma50 = df['close'] - sma_50
    
    # Calculate the Chaikin Money Flow (CMF) over a 20-day period
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate the Average True Range (ATR) over the last 20 days
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=20).mean()
    
    # Combine CMF, the difference between close and 50-day SMA, and ATR into a single heuristic factor
    heuristics_matrix = (cmf / (diff_close_sma50 + 1e-6)) * atr  # Adding 1e-6 to avoid division by zero

    return heuristics_matrix
