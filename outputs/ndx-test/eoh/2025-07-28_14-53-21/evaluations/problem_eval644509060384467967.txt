def heuristics_v2(df):
    # Calculate the 50-day SMA
    sma_50 = df['close'].rolling(window=50).mean()
    # Compute the difference of the closing price to its 50-day SMA
    close_to_sma_diff = df['close'] - sma_50
    
    # Calculate the 20-day minimum for volume
    vol_min = df['volume'].rolling(window=20).min()
    # Compute the ratio of the current day's volume to the 20-day minimum volume
    vol_to_min_ratio = df['volume'] / vol_min
    
    # Combine the two factors
    heuristics_matrix = close_to_sma_diff + vol_to_min_ratio
    
    return heuristics_matrix
