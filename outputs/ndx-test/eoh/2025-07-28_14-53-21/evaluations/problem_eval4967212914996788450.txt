import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change (ROC) over different periods
    roc_short = df['close'].pct_change(periods=10)
    roc_medium = df['close'].pct_change(periods=30)
    roc_long = df['close'].pct_change(periods=90)
    
    # Calculate Chaikin Money Flow (CMF) over 20 periods
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mf_multiplier * df['volume']
    cmf = mf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine ROC and CMF into a single heuristic matrix
    heuristics_matrix = (roc_short * 0.4 + roc_medium * 0.3 + roc_long * 0.2) + cmf * 0.1
    
    return heuristics_matrix
