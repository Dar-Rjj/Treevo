def heuristics_v2(df):
    # Calculate MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_signal_diff = macd_line - signal_line

    # Calculate ATR
    tr = df[['high', 'low']].apply(lambda x: x.max() - x.min(), axis=1)
    atr = tr.rolling(window=14).mean()

    # Dynamic weight allocation based on ATR
    weight_macd = 1 / (1 + atr)
    weight_atr = 1 - weight_macd
    heuristics_matrix = weight_macd * macd_signal_diff + weight_atr * atr

    return heuristics_matrix
