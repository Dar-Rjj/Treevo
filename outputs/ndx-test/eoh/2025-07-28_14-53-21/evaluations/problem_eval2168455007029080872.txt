def heuristics_v2(df):
    # Calculate 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate 20-day Simple Moving Average (SMA)
    sma_20 = df['close'].rolling(window=20).mean()

    # Calculate 5-day Coefficient of Variation (CV) of daily returns
    daily_returns = df['close'].pct_change()
    mean_ret = daily_returns.rolling(window=5).mean()
    std_dev_ret = daily_returns.rolling(window=5).std()
    cv = std_dev_ret / mean_ret

    # Combine all factors into a single heuristic factor
    heuristics_matrix = rsi + sma_20 + cv

    return heuristics_matrix
