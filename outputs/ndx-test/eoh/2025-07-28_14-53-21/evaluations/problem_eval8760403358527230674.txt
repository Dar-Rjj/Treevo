def heuristics_v2(df):
    # Calculate 30-day and 100-day exponential moving averages
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_diff_percentage = (ema_30 - ema_100) / ema_100

    # Calculate 14-day RSI
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Combine EMA difference percentage and RSI into a single heuristic factor
    heuristics_matrix = ema_diff_percentage + rsi

    return heuristics_matrix
