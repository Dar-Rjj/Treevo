def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Smooth the volume using exponential moving average
    smooth_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Combine ROC and smoothed volume into a single heuristic factor
    heuristics_matrix = roc * smooth_volume
    
    return heuristics_matrix
