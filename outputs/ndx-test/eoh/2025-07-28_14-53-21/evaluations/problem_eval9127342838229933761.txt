def heuristics_v2(df):
    weights = [0.5, 0.5]  # Equal weights for high and low
    df['weighted_avg'] = (df['high'] * weights[0] + df['low'] * weights[1])
    df['heuristics_factor'] = (df['close'] - df['weighted_avg']) / df['volume']
    heuristics_matrix = df['heuristics_factor']
    return heuristics_matrix
