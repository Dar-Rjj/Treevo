def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc = df['close'].pct_change(periods=12)
    
    # Calculate Average True Range (ATR)
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    close_to_atr_ratio = df['close'] / atr
    
    # Calculate Chaikin Money Flow (CMF)
    mfv = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mfv * df['volume']
    cmf = mf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine ROC, CMF, and the ratio of closing price to ATR into a single heuristic factor
    heuristics_matrix = roc + cmf + close_to_atr_ratio

    return heuristics_matrix
