def heuristics_v2(df):
    # Calculate 50-day Moving Average of the closing price
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate 21-day Standard Deviation of the high prices
    std_high_21 = df['high'].rolling(window=21).std()
    
    # Combine the 50-day Moving Average and the 21-day Standard Deviation into a single heuristic factor
    heuristics_matrix = ma_50 - std_high_21
    
    return heuristics_matrix
