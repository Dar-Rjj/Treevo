def heuristics_v2(df):
    # Calculate Stochastic Oscillator
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    
    # Calculate Directional Movement Index (DX, ADX not included in this version)
    up_move = df['high'].diff()
    down_move = -df['low'].diff()
    plus_dm = up_move.where((up_move > down_move) & (up_move > 0), 0)
    minus_dm = down_move.where((down_move > up_move) & (down_move > 0), 0)
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    plus_di = 100 * (plus_dm.rolling(window=14).sum() / atr)
    minus_di = 100 * (minus_dm.rolling(window=14).sum() / atr)
    dmi = plus_di - minus_di
    
    # Combine Stochastic Oscillator and DMI into a single heuristic factor
    heuristics_matrix = 0.5 * k + 0.5 * dmi

    return heuristics_matrix
