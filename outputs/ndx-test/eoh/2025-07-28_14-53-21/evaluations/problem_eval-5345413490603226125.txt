def heuristics_v2(df):
    # Calculate RSI with a 7-day period for gain and loss
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=7).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=7).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate MACD with a 12 and 26 day EMA, and a 14-day EMA for the signal line
    exp12 = df['close'].ewm(span=12, adjust=False).mean()
    exp26 = df['close'].ewm(span=26, adjust=False).mean()
    macd = exp12 - exp26
    signal = macd.ewm(span=14, adjust=False).mean()

    # Calculate smoothed volatility as the exponential moving average of the standard deviation of the last 20 days' returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std().ewm(span=5, adjust=False).mean()

    # Combine RSI, MACD signal line, and smoothed volatility into a single heuristic factor
    heuristics_matrix = rsi + signal - volatility

    return heuristics_matrix
