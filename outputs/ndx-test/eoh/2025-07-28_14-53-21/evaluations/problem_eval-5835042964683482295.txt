def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate ATR
    tr = df[['high', 'low']].apply(lambda x: x.max() - x.min(), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine RSI and ATR into a single heuristic factor
    heuristics_matrix = 0.5 * rsi + 0.5 * (atr)
    
    return heuristics_matrix
