import pandas as pd

def heuristics_v2(df):
    close = df['close']
    high = df['high']
    low = df['low']
    volume = df['volume']
    
    # Example Alpha Factor 1: (Close - Low) / (High - Low)
    factor_1 = (close - low) / (high - low)
    
    # Example Alpha Factor 2: (Close - Open) / (High - Low)
    factor_2 = (close - df['open']) / (high - low)
    
    # Example Alpha Factor 3: (Volume - Volume.shift(1)) / Volume
    factor_3 = (volume - volume.shift(1)) / volume
    
    # Combine all factors into a single DataFrame
    heuristics_matrix = pd.DataFrame({
        'Factor_1': factor_1,
        'Factor_2': factor_2,
        'Factor_3': factor_3
    })
    
    # Summarize the factors to create a single series
    heuristics_series = heuristics_matrix.sum(axis=1)
    
    return heuristics_matrix
