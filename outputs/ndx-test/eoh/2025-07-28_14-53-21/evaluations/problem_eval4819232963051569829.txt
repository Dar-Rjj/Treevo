def heuristics_v2(df):
    # Calculate 10-day Rate of Change (ROC)
    roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10) * 100
    
    # Calculate On-Balance Volume (OBV)
    obv = df['volume'].copy()
    obv[df['close'] < df['close'].shift(1)] = -obv
    obv[df['close'] == df['close'].shift(1)] = 0
    obv = obv.cumsum()
    
    # Combine ROC and OBV into a single heuristic factor
    heuristics_matrix = roc + obv

    return heuristics_matrix
