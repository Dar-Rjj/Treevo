def heuristics_v2(df):
    tr = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    tr['hl'] = df['high'] - df['low']
    tr['hc'] = (df['high'] - df['close'].shift()).abs()
    tr['lc'] = (df['low'] - df['close'].shift()).abs()
    tr['true_range'] = tr[['hl', 'hc', 'lc']].max(axis=1)
    atr_14 = tr['true_range'].rolling(window=14).mean()
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mf_multiplier * df['volume']
    cmf_20 = mf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    heuristics_matrix = atr_14 + cmf_20
    return heuristics_matrix
