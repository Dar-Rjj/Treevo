def heuristics_v2(df):
    # Calculate 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the percentage change in RSI
    rsi_change = rsi.pct_change(14)
    
    # Calculate 60-day cumulative return
    cum_return = (df['close'] / df['close'].shift(60)) - 1
    
    # Combine RSI change and 60-day cumulative return into a single heuristic factor
    heuristics_matrix = rsi_change + cum_return

    return heuristics_matrix
