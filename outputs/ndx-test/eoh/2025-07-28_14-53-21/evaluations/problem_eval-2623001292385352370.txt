def heuristics_v2(df):
    # Calculate 50-day and 200-day simple moving averages
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    sma_diff = sma_50 - sma_200

    # Calculate Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Combine SMA difference and CMF into a single heuristic factor
    heuristics_matrix = sma_diff + cmf

    return heuristics_matrix
