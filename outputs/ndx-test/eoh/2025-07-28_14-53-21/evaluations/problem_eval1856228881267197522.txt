def heuristics_v2(df):
    # Calculate Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - low_14) / (high_14 - low_14)
    
    # Calculate ADX
    tr = pd.DataFrame(0, index=df.index, columns=['tr'])
    dmplus = pd.DataFrame(0, index=df.index, columns=['dmplus'])
    dmminus = pd.DataFrame(0, index=df.index, columns=['dmminus'])
    
    tr['tr'] = df[['high', 'low']].diff().abs().max(axis=1)
    tr['tr'].iloc[0] = df['high'].iloc[0] - df['low'].iloc[0]
    dmplus['dmplus'] = df['high'].diff()
    dmminus['dmminus'] = -df['low'].diff()
    
    dmplus.loc[dmplus['dmplus'] <= 0, 'dmplus'] = 0
    dmminus.loc[dmminus['dmminus'] <= 0, 'dmminus'] = 0
    
    atr = tr.rolling(window=14).mean()
    plus_di = 100 * (dmplus.rolling(window=14).sum() / atr['tr'])
    minus_di = 100 * (dmminus.rolling(window=14).sum() / atr['tr'])
    
    dx = 100 * (plus_di - minus_di).abs() / (plus_di + minus_di)
    adx = dx.rolling(window=14).mean()
    
    # Combine Stochastic Oscillator and ADX into a single heuristic factor
    heuristics_matrix = 0.5 * k_percent + 0.5 * adx

    return heuristics_matrix
