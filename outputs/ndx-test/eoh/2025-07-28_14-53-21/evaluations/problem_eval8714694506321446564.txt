def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    RSI = 100 - (100 / (1 + RS))

    # Calculate OBV
    obv = (df['volume'] * ((df['close'] > df['close'].shift(1)) - (df['close'] < df['close'].shift(1)))).cumsum()

    # Combine RSI and OBV into a single heuristic factor
    heuristics_matrix = RSI + obv

    return heuristics_matrix
