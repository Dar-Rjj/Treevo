def heuristics_v2(df):
    # Calculate 50-day and 200-day simple moving averages
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    sma_crossover = sma_50 - sma_200

    # Calculate Accumulation/Distribution Oscillator (ADOSC)
    adosc = (df['high'] - df['close'].shift(1)) / (df['high'] - df['low']) * df['volume']
    adosc_14 = adosc.rolling(window=14).mean()

    # Combine ADOSC and SMA crossover into a single heuristic factor
    heuristics_matrix = adosc_14 + sma_crossover

    return heuristics_matrix
