def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Modified Parabolic SAR
    high = df['high']
    low = df['low']
    close = df['close']
    psar = close.copy()
    psar[0] = high[0]
    ep = high[0]
    acc_factor = 0.02
    max_acc_factor = 0.2
    increment = acc_factor
    direction = 1 if close[1] > psar[0] else -1
    atr = (df['high'] - df['low']).rolling(window=14).mean()  # ATR as measure of volatility

    for i in range(1, len(close)):
        vol_adj = atr[i] / atr.mean()  # Adjust increment by volatility
        if direction == 1:
            psar[i] = psar[i-1] + increment * vol_adj
            if high[i] > ep:
                ep = high[i]
                increment = min(increment + acc_factor, max_acc_factor)
            if psar[i] > low[i] or psar[i] > ep:
                direction = -1
                psar[i] = ep
                ep = low[i]
                increment = acc_factor
        else:
            psar[i] = psar[i-1] - increment * vol_adj
            if low[i] < ep:
                ep = low[i]
                increment = min(increment + acc_factor, max_acc_factor)
            if psar[i] < high[i] or psar[i] < ep:
                direction = 1
                psar[i] = ep
                ep = high[i]
                increment = acc_factor
    
    # Combine RSI and mPSAR into a single heuristic factor
    heuristics_matrix = 0.6 * rsi + 0.4 * (psar - close)

    return heuristics_matrix
