def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate price-to-volume ratio
    price_to_volume = df['close'] / df['volume']

    # Calculate volatility as the standard deviation of the last 20 days' returns
    volatility = df['close'].pct_change().rolling(window=20).std()

    # Combine RSI, price-to-volume, and volatility into a single heuristic factor
    heuristics_matrix = rsi + price_to_volume - volatility

    return heuristics_matrix
