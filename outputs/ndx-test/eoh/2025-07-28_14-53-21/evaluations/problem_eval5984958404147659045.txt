def heuristics_v2(df):
    # Calculate the 10-day Rate of Change (ROC)
    roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10) * 100

    # Calculate the Money Flow Multiplier and Money Flow Volume for ADL
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mfv = mfm * df['volume']

    # Create the Accumulation/Distribution Line
    adl = mfv.cumsum()

    # Combine ROC and ADL into a single heuristic factor
    heuristics_matrix = roc + adl

    return heuristics_matrix
