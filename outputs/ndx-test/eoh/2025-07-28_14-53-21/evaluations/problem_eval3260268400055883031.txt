import pandas as pd

def heuristics_v2(df):
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_crossover = ema_30 - ema_100
    delta = df['close'].diff()
    gain_10 = (delta.where(delta > 0, 0)).rolling(window=10).mean()
    loss_10 = (-delta.where(delta < 0, 0)).rolling(window=10).mean()
    rs_10 = gain_10 / loss_10
    rsi_10 = 100 - (100 / (1 + rs_10))
    gain_20 = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss_20 = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs_20 = gain_20 / loss_20
    rsi_20 = 100 - (100 / (1 + rs_20))
    rsi_diff = rsi_10 - rsi_20
    heuristics_matrix = (ema_crossover + rsi_diff) / 2
    return heuristics_matrix
