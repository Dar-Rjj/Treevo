def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Volume-based Signal
    volume_signal = df['volume'] / df['volume'].rolling(window=14).mean()

    # Combine RSI and Volume Signal into a single heuristic factor
    heuristics_matrix = 0.5 * rsi + 0.5 * volume_signal

    return heuristics_matrix
