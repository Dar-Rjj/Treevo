def heuristics_v2(df):
    # Calculate the average of High and Low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Compute the difference between Close and the average of High and Low
    diff_close_avg_high_low = df['close'] - avg_high_low
    
    # Apply a simple moving average on the computed differences
    heuristics_matrix = diff_close_avg_high_low.rolling(window=10).mean()
    
    return heuristics_matrix
