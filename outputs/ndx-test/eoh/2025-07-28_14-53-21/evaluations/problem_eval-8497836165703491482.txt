def heuristics_v2(df):
    # Calculate 50-day and 200-day simple moving averages
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    sma_diff = sma_50 - sma_200

    # Calculate On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int) - 
           df['volume'] * (df['close'] < df['close'].shift(1)).astype(int)).cumsum()

    # Combine SMA difference and OBV into a single heuristic factor
    heuristics_matrix = sma_diff + obv
    
    return heuristics_matrix
