import pandas as pd

def heuristics_v2(df):
    # Calculate the volume-adjusted close price
    volume_adjusted_close = df['close'] * df['volume']
    
    # Calculate the momentum of the close price
    momentum = df['close'].diff(periods=5)
    
    # Compute the heuristics
    heuristics_matrix = 0.7 * volume_adjusted_close + 0.3 * momentum
    
    return heuristics_matrix
