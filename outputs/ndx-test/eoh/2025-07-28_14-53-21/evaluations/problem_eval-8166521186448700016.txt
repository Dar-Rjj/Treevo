def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (tp * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate ATR
    tr = df[['high', 'low']].apply(lambda x: np.max(x) - np.min(x), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Combine RSI, VWAP, and ATR into a single heuristic factor
    heuristics_matrix = 0.5 * rsi + 0.3 * (vwap - df['close']) + 0.2 * atr
    
    return heuristics_matrix
