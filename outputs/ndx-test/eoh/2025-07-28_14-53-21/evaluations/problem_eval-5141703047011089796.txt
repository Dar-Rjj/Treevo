def heuristics_v2(df):
    # Calculate the 5-day and 20-day standard deviation of close prices as a measure of volatility
    short_volatility = df['close'].rolling(window=5).std()
    long_volatility = df['close'].rolling(window=20).std()
    
    # Volatility difference as a signal
    volatility_signal = short_volatility - long_volatility
    
    # Rate of change in volume over the last 5 days
    volume_roc = (df['volume'] / df['volume'].shift(5) - 1)
    
    # Combine the volatility signal and volume rate of change into a single heuristic factor
    heuristics_matrix = (volatility_signal + volume_roc)
    
    return heuristics_matrix
