def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate VWAP
    tp = (df['high'] + df['low'] + df['close']) / 3
    vwap = (tp * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine RSI and VWAP into a single factor
    rsi_vwap_mix = 0.6 * rsi + 0.4 * (vwap - df['close'])
    
    # Calculate MACD
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd = macd_line - signal_line
    
    # Final heuristic matrix combining the RSI-VWAP mix and MACD
    heuristics_matrix = 0.5 * rsi_vwap_mix + 0.5 * macd
    
    return heuristics_matrix
