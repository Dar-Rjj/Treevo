import pandas as pd

def heuristics_v2(df):
    def price_rate_of_change(series, window=12):
        return series / series.shift(window) - 1
    
    def volume_rate_of_change(series, window=12):
        return series / series.shift(window) - 1

    close_prices = df['close']
    volume = df['volume']

    price_roc = price_rate_of_change(close_prices)
    volume_roc = volume_rate_of_change(volume)

    heuristics_matrix = 0.5 * price_roc + 0.5 * volume_roc
    return heuristics_matrix
