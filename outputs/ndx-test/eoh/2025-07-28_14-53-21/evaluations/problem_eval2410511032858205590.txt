import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day SMA
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Compute the ratio of the closing price to its 10-day SMA
    close_to_sma_ratio = df['close'] / sma_10
    
    # Calculate the difference between the highest and lowest prices over the last 60 days
    high_low_diff = df['high'].rolling(window=60).max() - df['low'].rolling(window=60).min()
    
    # Combine the two factors
    heuristics_matrix = close_to_sma_ratio - high_low_diff
    
    return heuristics_matrix
