def heuristics_v2(df):
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_crossover = ema_20 - ema_100
    delta = df['close'].diff()
    gain_7 = (delta.where(delta > 0, 0)).rolling(window=7).mean()
    loss_7 = (-delta.where(delta < 0, 0)).rolling(window=7).mean()
    rs_7 = gain_7 / loss_7
    rsi_7 = 100 - (100 / (1 + rs_7))
    gain_14 = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss_14 = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs_14 = gain_14 / loss_14
    rsi_14 = 100 - (100 / (1 + rs_14))
    rsi_diff = rsi_7 - rsi_14
    heuristics_matrix = (ema_crossover + rsi_diff) / 2
    return heuristics_matrix
