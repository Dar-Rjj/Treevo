def heuristics_v2(df):
    # Calculate 20-day ADX
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=20).mean()
    up_move = df['high'] - df['high'].shift(1)
    down_move = df['low'].shift(1) - df['low']
    plus_dm = up_move.where((up_move > down_move) & (up_move > 0), 0)
    minus_dm = down_move.where((down_move > up_move) & (down_move > 0), 0)
    plus_di = 100 * (plus_dm.rolling(window=20).sum() / atr)
    minus_di = 100 * (minus_dm.rolling(window=20).sum() / atr)
    dx = 100 * (abs(plus_di - minus_di) / (plus_di + minus_di))
    adx = dx.rolling(window=20).mean()

    # Calculate 15-day EMA of the closing price
    ema_15 = df['close'].ewm(span=15, adjust=False).mean()

    # Calculate 10-day ROC of the trading volume
    roc_volume = df['volume'].pct_change(periods=10)

    # Combine all factors into a single heuristic factor
    heuristics_matrix = adx + ema_15 + roc_volume

    return heuristics_matrix
