def heuristics_v2(df):
    # Calculate the 14-day rate of change (ROC) for the closing price
    roc = df['close'].pct_change(periods=14)
    
    # Determine the highest high and lowest low over the past 28 days
    highest_high = df['high'].rolling(window=28).max()
    lowest_low = df['low'].rolling(window=28).min()
    
    # Create an oscillator based on the difference between the highest high and lowest low
    range_oscillator = highest_high - lowest_low
    
    # Combine ROC and the range oscillator into a single heuristic factor
    heuristics_matrix = roc + range_oscillator

    return heuristics_matrix
