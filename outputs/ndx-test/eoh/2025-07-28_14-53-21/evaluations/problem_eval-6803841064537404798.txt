def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14)
    
    # Calculate Relative Volume (RV)
    average_volume = df['volume'].rolling(window=50).mean()
    rv = df['volume'] / average_volume
    
    # Combine ROC and RV into a single heuristic factor
    heuristics_matrix = 0.7 * roc + 0.3 * rv

    return heuristics_matrix
