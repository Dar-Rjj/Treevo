def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc = df['close'].pct_change(periods=14)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    obv = (obv * df['volume']).cumsum()
    
    # Adjust OBV based on recent volatility
    volatility = df['close'].pct_change().rolling(window=14).std()
    adjusted_obv = obv * (1 + volatility)
    
    # Combine ROC and adjusted OBV into a single heuristic factor
    heuristics_matrix = roc + adjusted_obv

    return heuristics_matrix
