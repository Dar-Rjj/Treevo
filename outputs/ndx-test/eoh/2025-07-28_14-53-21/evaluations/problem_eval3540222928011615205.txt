import pandas as pd

def heuristics_v2(df):
    window = 20
    no_of_std = 2
    
    # Calculate rolling mean and standard deviation
    rolling_mean = df['close'].rolling(window=window).mean()
    rolling_std = df['close'].rolling(window=window).std()
    
    # Calculate upper and lower Bollinger Bands
    upper_band = rolling_mean + (rolling_std * no_of_std)
    lower_band = rolling_mean - (rolling_std * no_of_std)
    
    # Calculate the distance of the close price from the upper and lower bands
    distance_from_upper = df['close'] - upper_band
    distance_from_lower = df['close'] - lower_band
    
    # Compute the heuristics
    heuristics_matrix = (distance_from_upper.where(distance_from_upper > 0, 0) - 
                        distance_from_lower.where(distance_from_lower < 0, 0))
    
    return heuristics_matrix
