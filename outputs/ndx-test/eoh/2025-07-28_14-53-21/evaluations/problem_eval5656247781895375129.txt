def heuristics_v2(df):
    # Calculate 14-day Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()

    # Calculate 30-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=30)

    # Calculate 7-day moving average of the standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_7 = daily_returns.rolling(window=7).std()

    # Combine all factors into a single heuristic factor
    heuristics_matrix = cmf + roc + std_dev_7

    return heuristics_matrix
