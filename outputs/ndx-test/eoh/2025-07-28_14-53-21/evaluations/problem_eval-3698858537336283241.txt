def heuristics_v2(df):
    # Calculate the rate of change in closing price over 14 days
    close_change = df['close'].pct_change(periods=14)
    
    # Calculate the average true range (ATR) over 14 days
    tr = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs()  # True Range
    atr = tr.rolling(window=14).mean()
    
    # Calculate the ATR to volume ratio
    atr_to_volume = atr / df['volume']
    
    # Combine the rate of change and ATR to volume ratio into a single heuristic factor
    heuristics_matrix = close_change + atr_to_volume
    
    return heuristics_matrix
