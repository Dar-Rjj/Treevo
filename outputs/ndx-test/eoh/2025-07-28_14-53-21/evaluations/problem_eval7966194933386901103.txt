def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate ROC
    roc = ((df['close'] - df['close'].shift(14)) / df['close'].shift(14)) * 100

    # Calculate OBV
    obv = (df['volume'] * (df['close'] > df['close'].shift()).astype(int) - 
           df['volume'] * (df['close'] <= df['close'].shift()).astype(int))
    obv = obv.cumsum()

    # Combine RSI, ROC, and OBV into a single heuristic factor
    heuristics_matrix = rsi + roc + obv

    return heuristics_matrix
