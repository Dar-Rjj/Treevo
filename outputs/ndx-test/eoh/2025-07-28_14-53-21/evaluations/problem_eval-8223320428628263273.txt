def heuristics_v2(df):
    # Calculate 100-day and 200-day exponential moving averages
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    ema_diff = ema_100 - ema_200

    # Adjust MFI calculation to a 28-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_sum = positive_money_flow.rolling(window=28).sum()
    negative_sum = negative_money_flow.rolling(window=28).sum()
    mfi = 100 - (100 / (1 + (positive_sum / negative_sum)))

    # Incorporate recent price momentum as an additional signal
    momentum = df['close'] - df['close'].shift(14)

    # Combine EMA difference, MFI, and momentum into a single heuristic factor
    heuristics_matrix = (ema_diff + mfi + momentum) / 3

    return heuristics_matrix
