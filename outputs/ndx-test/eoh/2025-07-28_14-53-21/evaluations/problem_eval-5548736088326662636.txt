def heuristics_v2(df):
    # Calculate 14-period Rate of Change (ROC) of closing prices
    roc = df['close'].pct_change(periods=14) * 100
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['volume'] * ((df['close'] > df['close'].shift(1)).astype(int) - (df['close'] < df['close'].shift(1)).astype(int))).cumsum()
    
    # Combine ROC and OBV into a single heuristic factor
    heuristics_matrix = roc + obv

    return heuristics_matrix
