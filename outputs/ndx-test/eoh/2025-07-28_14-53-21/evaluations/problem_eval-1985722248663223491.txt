def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate MACD
    exp5 = df['close'].ewm(span=5, adjust=False).mean()
    exp35 = df['close'].ewm(span=35, adjust=False).mean()
    macd = exp5 - exp35
    signal = macd.rolling(window=9).mean()

    # Calculate smoothed volatility as the exponential moving average of the standard deviation of the last 20 days' returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std().ewm(span=10, adjust=False).mean()

    # Combine RSI, MACD signal line, and smoothed volatility into a single heuristic factor
    heuristics_matrix = rsi + signal - volatility

    return heuristics_matrix
