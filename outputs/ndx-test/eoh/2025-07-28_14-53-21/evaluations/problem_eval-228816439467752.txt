def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Bollinger Bands
    moving_avg = df['close'].rolling(window=20).mean()
    moving_std = df['close'].rolling(window=20).std()
    upper_band = moving_avg + 2 * moving_std
    lower_band = moving_avg - 2 * moving_std
    
    # Combine RSI and Bollinger Bands into a single heuristic factor
    heuristics_matrix = 0.5 * rsi + 0.5 * ((upper_band - lower_band) / moving_avg)

    return heuristics_matrix
