def heuristics_v2(df):
    roc = (df['close'] - df['close'].shift(21)) / df['close'].shift(21) * 100
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_100 = df['close'].ewm(span=100, adjust=False).mean()
    ema_crossover = ema_30 - ema_100
    heuristics_matrix = roc + ema_crossover
    return heuristics_matrix
