def heuristics_v2(df):
    # Calculate 20-day and 60-day simple moving averages
    sma_20 = df['close'].rolling(window=20).mean()
    sma_60 = df['close'].rolling(window=60).mean()
    sma_diff = sma_20 - sma_60

    # Calculate 14-day Rate of Change (ROC)
    roc_14 = (df['close'] - df['close'].shift(14)) / df['close'].shift(14) * 100

    # Combine SMA difference and ROC into a single heuristic factor
    heuristics_matrix = sma_diff + roc_14

    return heuristics_matrix
