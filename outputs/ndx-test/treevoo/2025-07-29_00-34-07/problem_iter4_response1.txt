```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-day, 5-day, and 20-day price returns
    df['1_day_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['20_day_return'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Average the momentum
    df['average_momentum'] = (df['1_day_return'] + df['5_day_return'] + df['20_day_return']) / 3

    # Calculate daily volume percentage change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Final factor: Adjust average momentum for volume
    df['final_factor'] = df['average_momentum'] * (1 + df['volume_change'])

    return df['final_factor']
```
