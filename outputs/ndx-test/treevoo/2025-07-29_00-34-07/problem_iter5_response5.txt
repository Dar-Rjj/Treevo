```python
import pandas as pd

def heuristics_v2(df):
    # Compute 1-day, 5-day, and 20-day Price Returns
    df['1d_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted Average Momentum
    df['weighted_momentum'] = (0.6 * df['1d_return'] + 
                               0.3 * df['5d_return'] + 
                               0.1 * df['20d_return'])
    
    # Adjust for Volume and Amount
    df['volume_pct_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    df['amount_pct_change'] = (df['amount'] / df['amount'].shift(1)) - 1
    df['volume_adjusted_momentum'] = (df['weighted_momentum'] * 
                                      df['volume_pct_change'] * 
                                      df['amount_pct_change'])
    
    # Incorporate High-Low Differential and Volume Volatility
    df['rolling_high'] = df['high'].rolling(window=20).max()
    df['rolling_low'] = df['low'].rolling(window=20).min()
    df['high_low_diff'] = df['rolling_high'] - df['rolling_low']
    df['volume_volatility'] = df['volume'].rolling(window=10).std()
    df['hl_volatility_impact'] = df['high_low_diff'] * df['volume_volatility']
    
    # Combine Momentum and Volatility Components
    df['aggregated_signal'] = (df['volume_adjusted_momentum'] + 
                               df['hl_volatility_impact'])
    
    # Final Factor Calculation
    df['final_factor'] = (df['aggregated_signal'] / df['volume']) * df['amount_pct_change']
    
    return df['final_factor']
```
