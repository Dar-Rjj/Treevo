```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_to_low_range * 125) + price_change
    
    # Smoothing and Initial Momentum Calculation
    ema_5 = combined_value.ewm(span=5).mean()
    initial_momentum = ema_5 - ema_5.shift(6)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    momentum_2_day = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    momentum_7_day = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    momentum_14_day = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    momentum_30_day = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Weighted Average the Momentum
    weighted_momentum = (2 * momentum_2_day + momentum_7_day + momentum_14_day + momentum_30_day) / 12
    
    # Adjust for Volume Changes
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = weighted_momentum * (1 + 1.5 * volume_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + volume_adjusted_momentum
    
    # Calculate Volume-Weighted Price Momentum
    wma_close_10 = df['close'].rolling(window=10).mean()
    volume_weighted_momentum = (df['close'] - wma_close_10) * daily_log_return
    
    # Introduce High-Frequency Volatility
    true_range = np.maximum(df['high'] - df['low'], 
                            np.abs(df['high'] - df['close'].shift(1)), 
                            np.abs(df['low'] - df['close'].shift(1)))
    sma_true_range_10 = true_range.rolling(window=10).mean()
    
    # Adjust for High-Frequency Volatility
    volatility_adjusted_volume_momentum = volume_weighted_momentum / sma_true_range_10
    
    # Final Factor Construction
    final_factor = intermediate_factor + volatility_adjusted_volume_momentum
    
    return final_factor
```
