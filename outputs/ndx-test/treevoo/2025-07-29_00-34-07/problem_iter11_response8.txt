```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate 5-day Moving Average of Volumes
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    
    # Compute Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Multiply 5-day Moving Average by Volume Growth
    volume_trend = volume_5day_ma * volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Determine 20-day Rolling High and Low
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    
    # Calculate Rolling High-Low Differential
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    
    # Multiply Rolling High-Low Differential by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Subtract Price Difference
    price_diff = df['close'] - df['close'].shift(10)
    
    # Aggregate the Signals
    aggregate_signal = weighted_high_low_diff - price_diff
    
    # Calculate Price Momentum
    momentum_1day = df['close'] / df['close'].shift(1) - 1
    momentum_5day = df['close'] / df['close'].shift(5) - 1
    momentum_20day = df['close'] / df['close'].shift(20) - 1
    
    # Average the Momentum
    averaged_momentum = (momentum_1day + momentum_5day + momentum_20day) / 3
    
    # Calculate Daily Volume Percentage Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Adjust for Volume
    adjusted_aggregate_signal = aggregate_signal * volume_change
    
    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps 
                    + adjusted_aggregate_signal 
                    - averaged_momentum)
    
    return alpha_factor
```
