```python
import pandas as pd

def heuristics_v2(df, n=10, lookback=30):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Adjusted Relative Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    filtered_days = df[df['volume'] > 1.5 * df['avg_volume']]
    
    # Calculate High-Low Range Ratio
    df['range_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['vw_return'] = ((df['close'] - df['open']) * df['volume']).rolling(window=lookback).sum()
    
    # Calculate Cumulative High-Low Momentum for Filtered Days
    filtered_df = df.loc[filtered_days.index]
    cumulative_momentum = filtered_df['price_range'].rolling(window=lookback).sum()
    
    # Generate Alpha Factor based on Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    alpha_factor = (cumulative_momentum + df['vw_return']) / 2
    
    return alpha_factor
```
