```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Average of Price Gaps
    df['price_gap'] = df['high'] - df['low']
    df['volume_trend'] = df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))
    df['weighted_price_gap'] = df['price_gap'] * df['volume_trend']
    weighted_price_gap_sum = df['weighted_price_gap'].sum()

    # Rolling High-Low Differential
    df['rolling_high'] = df['high'].rolling(window=20).max()
    df['rolling_low'] = df['low'].rolling(window=20).min()
    df['rolling_diff'] = df['rolling_high'] - df['rolling_low']

    # Weight by Volume Volatility
    df['vol_volatility'] = df['volume'].rolling(window=10).std()
    df['weighted_rolling_diff'] = df['rolling_diff'] * df['vol_volatility']

    # Aggregate the Signals
    df['signal_aggregate'] = df['weighted_rolling_diff'] - (df['close'] - df['close'].shift(10))

    # High-to-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close'].shift(1)

    # Combine High-to-Low Range with Price Change
    df['1d_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['combined_value'] = df['high_low_range'] * 100 + df['1d_return']

    # Smoothing and Initial Momentum Calculation
    df['ema_5'] = df['combined_value'].ewm(span=5).mean()
    df['momentum_smooth'] = df['ema_5'] - df['ema_5'].shift(5)

    # Multi-Day Price Momentum
    df['7d_return'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['14d_return'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    df['30d_return'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Weighted Average Momentum
    df['weighted_momentum'] = (0.4 * df['1d_return']) + (0.3 * df['7d_return']) + (0.3 * df['30d_return'])

    # Final Volume Adjustment
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['weighted_momentum'] * (1 + df['daily_volume_change'] / df['volume'].shift(1))

    # Intermediate Factor Combination
    df['intermediate_factor'] = df['momentum_smooth'] + df['volume_adjusted_momentum']

    # Final Alpha Factor
    df['alpha_factor'] = weighted_price_gap_sum * df['intermediate_factor'] + df['signal_aggregate'] + df['volume_adjusted_momentum']

    return df['alpha_factor']
```
