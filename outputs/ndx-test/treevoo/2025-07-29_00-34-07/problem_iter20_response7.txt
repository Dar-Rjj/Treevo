```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(15)
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    vol_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_diff = high_low_diff * vol_volatility
    
    # Adjust for Volume Changes
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_vol_adjustment = weighted_high_low_diff * (1 + 2 * daily_vol_change)
    
    # Aggregate the Signals
    aggregate_signal = final_vol_adjustment - price_diff
    
    # Calculate Averaged Momentum
    one_day_return = df['close'] / df['close'].shift(1) - 1
    five_day_return = df['close'] / df['close'].shift(5) - 1
    twenty_day_return = df['close'] / df['close'].shift(20) - 1
    averaged_momentum = (one_day_return + five_day_return + twenty_day_return) / 3
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = aggregate_signal.rolling(window=7).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(7)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    sum_daily_price_range = daily_price_range.sum()
    weighted_price_gaps = (daily_price_range * daily_vol_change).sum()
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + averaged_momentum + aggregate_signal
    
    # Calculate Weighted Price Momentum
    log_return = np.log(df['close'] / df['close'].shift(1))
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=15).mean()) * log_return
    
    # Adjust for Volume
    volume_adjusted_weighted_price_gaps = weighted_price_gaps * daily_vol_change
    
    # High-to-Low Range and Momentum
    high_low_range = (df['high'] - df['low']) * 100
    combined_high_low_weighted_gaps = high_low_range + volume_adjusted_weighted_price_gaps
    smoothed_momentum = combined_high_low_weighted_gaps.rolling(window=5).mean()
    high_low_momentum = smoothed_momentum - smoothed_momentum.shift(5)
    
    # Final Alpha Factor Construction
    alpha_factor = (sum_daily_price_range + volume_adjusted_weighted_price_gaps) * averaged_momentum + \
                   intermediate_factor + weighted_price_momentum + high_low_momentum + \
                   final_vol_adjustment - averaged_momentum
    
    return alpha_factor
```
