```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(15)
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    rolling_high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std()
    weighted_rolling_high_low_diff = rolling_high_low_diff * volume_volatility
    
    # Adjust for Volume Changes
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjusted = weighted_rolling_high_low_diff * (1 + 2 * daily_volume_change)
    
    # Aggregate the Signals
    aggregated_signals = final_volume_adjusted - price_diff
    
    # Calculate Averaged Momentum
    one_day_return = (df['close'] / df['close'].shift(1)) - 1
    five_day_return = (df['close'] / df['close'].shift(5)) - 1
    thirty_day_return = (df['close'] / df['close'].shift(30)) - 1
    averaged_momentum = (one_day_return + five_day_return + thirty_day_return) / 3
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined = aggregated_signals.rolling(window=7).mean()
    initial_momentum = smoothed_combined - smoothed_combined.shift(7)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    sum_daily_price_ranges = daily_price_range.rolling(window=20).sum()
    weighted_price_gaps = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1))
    total_weighted_price_gaps = weighted_price_gaps.rolling(window=20).sum()
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + (averaged_momentum * total_weighted_price_gaps)
    
    # Calculate Weighted Price Momentum
    log_return = np.log(df['close']) - np.log(df['close'].rolling(window=20).mean())
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=20).mean()) * log_return
    
    # Adjust for Volume
    adjusted_weighted_price_gaps = total_weighted_price_gaps * daily_volume_change
    
    # High-to-Low Range and Momentum
    high_to_low_range = (df['high'] - df['low']) * 100
    combined_high_low_range = high_to_low_range + adjusted_weighted_price_gaps
    smoothed_high_low = combined_high_low_range.rolling(window=5).mean()
    high_low_momentum = smoothed_high_low - smoothed_high_low.shift(5)
    
    # Final Alpha Factor Construction
    alpha_factor = (
        sum_daily_price_ranges + total_weighted_price_gaps +
        intermediate_factor + weighted_price_momentum +
        high_low_momentum + final_volume_adjusted -
        averaged_momentum
    )
    
    return alpha_factor
```
