```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['1_day_return'] = df['close'].pct_change(1)
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Average the Momentum
    df['average_momentum'] = (df['1_day_return'] + df['5_day_return'] + df['20_day_return']) / 3
    
    # Filter by Relative Volume
    n = 20  # Define the lookback period for average volume
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_filtered'] = df['volume'] > df['average_volume']
    
    # Calculate Adjusted High-Low Cumulative Return
    df['daily_price_range'] = df['high'] - df['low']
    df['filtered_daily_price_range'] = df['daily_price_range'] * df['volume_filtered']
    df['cumulative_price_range'] = df['filtered_daily_price_range'].rolling(window=n, min_periods=1).sum()
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['average_momentum'] * df['cumulative_price_range']
    
    return df['alpha_factor']
```
