```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['1d_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5d_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['20d_return'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['avg_momentum'] = (df['1d_return'] + df['5d_return'] + df['20d_return']) / 3
    
    # Adjust for Volume
    df['daily_vol_chg'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['vol_adj_mom'] = df['avg_momentum'] * (1 + df['daily_vol_chg'])

    # Filter by Relative Volume
    n = 50  # Number of days to calculate average volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df_filtered = df[df['volume'] > df['avg_volume']]
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    df_filtered['price_range'] = df_filtered['high'] - df_filtered['low']
    df_filtered['weighted_price_gaps'] = df_filtered['price_range'] * (df_filtered['volume'] - df_filtered['volume'].shift(1))
    sum_weighted_price_gaps = df_filtered['weighted_price_gaps'].rolling(window=n).sum()

    # Calculate Rolling High-Low Differential
    df['rolling_high'] = df['high'].rolling(window=20).max()
    df['rolling_low'] = df['low'].rolling(window=20).min()
    df['hl_diff'] = df['rolling_high'] - df['rolling_low']

    # Weight by Volume Volatility
    df['vol_volatility'] = df['volume'].rolling(window=10).std()
    df['weighted_hl_diff'] = df['hl_diff'] * df['vol_volatility']

    # Aggregate the Signals
    df['aggregated_signal'] = sum_weighted_price_gaps + df['weighted_hl_diff']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['vol_adj_mom'] * df['aggregated_signal']
    
    return df['alpha_factor']
```
