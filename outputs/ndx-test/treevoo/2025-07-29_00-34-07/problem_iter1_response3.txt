```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(10)
    
    # Calculate 20-day Rolling High and Low
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    
    # Calculate Rolling High-Low Differential
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility (standard deviation of volume over the last 10 days)
    volume_volatility = df['volume'].rolling(window=10).std()
    
    # Weight the Rolling High-Low Differential by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the signals
    factor = weighted_high_low_diff - price_diff
    
    return factor
```
