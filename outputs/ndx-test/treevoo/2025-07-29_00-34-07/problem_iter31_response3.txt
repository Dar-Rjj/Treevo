```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum Factors
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    df['ROC_1'] = df['close'].pct_change(periods=1)
    df['ROC_5'] = df['close'].pct_change(periods=5)
    df['ROC_14'] = df['close'].pct_change(periods=14)
    
    # Volatility Factors
    df['ATR_14'] = df['high'].combine(df['low'], max) - df['low'].combine(df['high'], min)
    df['ATR_14'] = df['ATR_14'].rolling(window=14).mean()
    
    log_returns = np.log(df['close'] / df['close'].shift(1))
    df['StdDev_5'] = log_returns.rolling(window=5).std()
    df['StdDev_10'] = log_returns.rolling(window=10).std()
    df['StdDev_20'] = log_returns.rolling(window=20).std()
    
    # Volume and Amount Factors
    df['Volume_Trend'] = df['volume'] > df['volume'].rolling(window=5).mean()
    
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['MFI_14'] = 100 - (100 / (1 + money_ratio))
    
    # Price Action Factors
    df['Bullish_Engulfing'] = (df['close'] > df['open']) & (df['close'].shift(1) < df['open'].shift(1)) & \
                              (df['close'] > df['open'].shift(1)) & (df['open'] < df['close'].shift(1))
    df['Bearish_Engulfing'] = (df['close'] < df['open']) & (df['close'].shift(1) > df['open'].shift(1)) & \
                              (df['close'] < df['open'].shift(1)) & (df['open'] > df['close'].shift(1))
    df['Hammer'] = (df['close'] - df['low']) > 2 * (df['high'] - df['close'])
    df['Hanging_Man'] = (df['open'] - df['low']) > 2 * (df['high'] - df['open'])
    
    df['RS'] = df['high'] - df['low']
    
    # Combined Factors
    df['Composite_Factor'] = (df['SMA_5'] + df['SMA_10'] + df['SMA_20'] + 
                              df['ROC_1'] + df['ROC_5'] + df['ROC_14'] +
                              df['ATR_14'] + df['StdDev_5'] + df['StdDev_10'] + df['StdDev_20'] +
                              df['Volume_Trend'] + df['MFI_14'] + 
                              df['Bullish_Engulfing'] - df['Bearish_Engulfing'] + 
                              df['Hammer'] - df['Hanging_Man'] + df['RS'])
    
    return df['Composite_Factor']
```
