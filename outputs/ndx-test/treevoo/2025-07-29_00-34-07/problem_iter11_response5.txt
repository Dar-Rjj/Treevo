```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the percentage change in volume
    volume_pct_change = df['volume'].pct_change()
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between open and close prices
    open_close_diff = df['open'] - df['close']
    
    # Calculate the factor as a weighted sum of the above metrics
    factor = (0.4 * close_pct_change + 
              0.3 * volume_pct_change + 
              0.2 * avg_price.pct_change() + 
              0.1 * open_close_diff.pct_change())
    
    return factor
```
