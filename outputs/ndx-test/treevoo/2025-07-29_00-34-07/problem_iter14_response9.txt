```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(15)
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = weighted_high_low_diff - price_diff
    
    # Calculate Price Momentum
    two_day_momentum = (df['close'] / df['close'].shift(2)) - 1
    seven_day_momentum = (df['close'] / df['close'].shift(7)) - 1
    thirty_day_momentum = (df['close'] / df['close'].shift(30)) - 1
    
    # Average the Momentum
    avg_momentum = (two_day_momentum + seven_day_momentum + thirty_day_momentum) / 3
    
    # Adjust for Volume
    volume_change = (df['volume'] / df['volume'].shift(1)) - 1
    adjusted_aggregated_signal = aggregated_signal * volume_change
    
    # Final Aggregate Signal
    final_signal = adjusted_aggregated_signal - avg_momentum
    
    return final_signal
```
