```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate 5-day Moving Average of Volumes
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    
    # Compute Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Multiply 5-day Moving Average by Volume Growth
    volume_trend = volume_5day_ma * volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Determine 20-day Rolling High
    rolling_high_20 = df['high'].rolling(window=20).max()
    
    # Subtract 20-day Rolling Low
    rolling_low_20 = df['low'].rolling(window=20).min()
    rolling_high_low_diff = rolling_high_20 - rolling_low_20
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    
    # Multiply Rolling High-Low Differential by Volume Volatility
    weighted_rolling_high_low_diff = rolling_high_low_diff * volume_volatility
    
    # Subtract 10-Day Price Difference
    price_diff_10day = df['close'] - df['close'].shift(10)
    aggregated_signal = weighted_rolling_high_low_diff - price_diff_10day
    
    # High-to-Low Range
    high_low_range = (df['high'] - df['low']) / df['open']
    
    # 1-day Price Return
    price_return_1day = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combined Value
    combined_value = high_low_range * 100 + price_return_1day
    
    # Apply Exponential Moving Average on Combined Value over 7 days
    ema_combined_7 = combined_value.ewm(span=7).mean()
    
    # Momentum Difference
    momentum_difference = ema_combined_7 - ema_combined_7.shift(7)
    
    # Multi-Day Price Momentum
    price_return_7day = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    price_return_30day = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Weighted Average Momentum
    weighted_avg_momentum = 0.4 * price_return_1day + 0.3 * price_return_7day + 0.3 * price_return_30day
    
    # Final Volume Adjustment
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_adjusted_momentum = weighted_avg_momentum * (1 + daily_volume_change / df['volume'].shift(1))
    
    # Sum Smoothing and Initial Momentum
    intermediate_factor_combination = momentum_difference + volume_adjusted_momentum
    
    # Volume-Weighted Price Momentum
    close_10day_ema = df['close'].ewm(span=10).mean()
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    volume_weighted_price_momentum = (df['close'] - close_10day_ema) * log_return
    
    # Final Alpha Factor
    final_alpha_factor = weighted_price_gaps * intermediate_factor_combination + volume_weighted_price_momentum
    
    return final_alpha_factor
```
