import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['open']
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_to_low_range * 100) + price_change
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = combined_value.ewm(span=5, adjust=False).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(5)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    two_day_return = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    seven_day_return = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    fourteen_day_return = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Weighted Average the Momentum
    weighted_momentum = (2 * two_day_return + seven_day_return + fourteen_day_return) / 10
    
    # Adjust for Volume Changes
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = weighted_momentum * (1 + daily_volume_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + volume_adjusted_momentum
    
    # Calculate Volume-Weighted Price Momentum
    ten_day_wma_close = df['close'].rolling(window=10).mean()
    volume_weighted_momentum = (df['close'] - ten_day_wma_close) * daily_log_return
    
    # Introduce High-Frequency Volatility
    true_range = np.maximum(
        df['high'] - df['low'],
        np.abs(df['high'] - df['close'].shift(1)),
        np.abs(df['low'] - df['close'].shift(1))
    )
    ten_day_sma_true_range = true_range.rolling(window=10).mean()
    
    # Adjust for High-Frequency Volatility
    hf_volatility_adjusted_momentum = volume_weighted_momentum / ten_day_sma_true_range
    
    # Final Factor Construction
    final_factor = intermediate_factor + hf_volatility_adjusted_momentum
    
    return final_factor.dropna()
