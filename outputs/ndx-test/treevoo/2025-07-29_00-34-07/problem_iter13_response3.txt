```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_low_range * 150) + price_change
    
    # Smoothing and Momentum Calculation
    sma_7 = combined_value.rolling(window=7).mean()
    momentum_7 = sma_7 - sma_7.shift(7)
    
    # Calculate Daily Log Return
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Averaged Price Momentum
    momentum_3 = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    averaged_momentum = (momentum_3 + momentum_10 + momentum_30) / 3
    
    # Adjust for Volume Changes
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = averaged_momentum * (1 + 2 * volume_change)
    
    # Intermediate Factor Combination
    intermediate_factor = momentum_7 + volume_adjusted_momentum
    
    # Calculate Weighted Price Momentum
    close_15_sma = df['close'].rolling(window=15).mean()
    weighted_price_momentum = (df['close'] - close_15_sma) * log_return
    
    # Final Factor Construction
    final_factor = intermediate_factor + weighted_price_momentum
    
    return final_factor
```
