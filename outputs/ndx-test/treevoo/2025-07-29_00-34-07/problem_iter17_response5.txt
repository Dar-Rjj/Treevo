```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative close price within the day's range
    df['range'] = df['high'] - df['low']
    df['close_position'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the change in close position from the previous day
    df['close_position_change'] = df['close_position'].diff()
    
    # Calculate the volume weighted by the close position
    df['volume_weighted'] = df['volume'] * df['close_position']
    
    # Define the factor as the smoothed difference between the volume weighted and the close position change
    alpha_factor = df['volume_weighted'] - df['close_position_change'].rolling(window=3).mean()
    
    return alpha_factor.dropna()
```
