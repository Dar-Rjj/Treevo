```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Adjusted Relative Volume
    average_volume = df['volume'].rolling(window=n).mean()
    volume_filter = df['volume'] > 1.5 * average_volume
    
    # Filtered DataFrame based on volume condition
    filtered_df = df[volume_filter]
    
    # Calculate High-Low Range Ratio
    high_low_ratio = (filtered_df['high'] - filtered_df['low']) / filtered_df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_open_return = (filtered_df['close'] - filtered_df['open']) * filtered_df['volume']
    volume_weighted_return = close_open_return.rolling(window=n).sum()
    
    # Calculate Cumulative High-Low Momentum
    cumulative_high_low_momentum = daily_price_range.rolling(window=n).sum()
    
    # Combine Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    alpha_factor = cumulative_high_low_momentum + volume_weighted_return
    
    return alpha_factor
```
