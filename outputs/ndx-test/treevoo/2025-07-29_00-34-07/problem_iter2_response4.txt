```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['1d_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Average the Momentum
    df['avg_momentum'] = (df['1d_return'] + df['5d_return'] + df['20d_return']) / 3
    
    # Adjust for Volume
    df['volume_pct_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['avg_momentum'] * df['volume_pct_change']
    
    # Incorporate High-Low Momentum with Volume Filter
    df['daily_price_range'] = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    df['filtered_days'] = (df['volume'] > avg_volume) * 1  # Binary filter, 1 if volume is higher, else 0
    
    # Calculate Cumulative High-Low Momentum
    df['cum_high_low_momentum'] = df['daily_price_range'].rolling(window=20).sum() * df['filtered_days']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_momentum'] + df['cum_high_low_momentum']
    
    return df['alpha_factor']
```
