```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(10)
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregate_signal = weighted_high_low_diff - price_diff
    
    # Calculate Average Price Momentum
    price_return_1d = (df['close'] / df['close'].shift(1) - 1)
    price_return_5d = (df['close'] / df['close'].shift(5) - 1)
    price_return_20d = (df['close'] / df['close'].shift(20) - 1)
    average_price_momentum = (price_return_1d + price_return_5d + price_return_20d) / 3
    
    # Adjust for Volume
    volume_change = (df['volume'] / df['volume'].shift(1) - 1)
    adjusted_signal = aggregate_signal * volume_change
    
    # Final alpha factor
    alpha_factor = adjusted_signal + average_price_momentum
    
    return alpha_factor
```
