```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Adjusted Relative Volume
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_filter = df['volume'] > 1.5 * avg_volume
    filtered_days = df[volume_filter]
    
    # Calculate Weighted Average of Price Gaps
    price_gaps = (df['high'] - df['low'])
    volume_growth = df['volume'] / df['volume'].shift(1)
    weighted_price_gaps = (price_gaps * volume_growth).sum()

    # Combine High-to-Low Range and Weighted Average of Price Gaps
    combined_value = (high_low_range * 150) + weighted_price_gaps
    
    # Smoothing and Initial Momentum Calculation
    ema_combined = combined_value.ewm(span=7).mean()
    initial_momentum = ema_combined - ema_combined.shift(7)
    
    # Calculate Multi-Day Price Momentum
    momentum_3d = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10d = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_20d = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    averaged_momentum = (momentum_3d + momentum_10d + momentum_20d) / 3
    
    # Adjust for Volume Changes
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = averaged_momentum * (1 + 2 * daily_vol_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + volume_adjusted_momentum
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    rolling_high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    weighted_rolling_high_low_diff = rolling_high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregate_signal = weighted_rolling_high_low_diff + (df['close'] - df['close'].shift(10))
    
    # Adjust for Volume
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_aggregate_signal = aggregate_signal * daily_vol_change
    
    # Final Factor Construction
    final_factor = intermediate_factor + adjusted_aggregate_signal - averaged_momentum
    
    return final_factor
```
