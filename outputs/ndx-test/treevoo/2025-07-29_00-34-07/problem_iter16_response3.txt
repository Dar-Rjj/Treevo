import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Average of Price Gaps
    df['price_gap'] = df['high'] - df['low']
    df['volume_growth'] = df['volume'] / df['volume'].shift(1)
    df['weighted_price_gap'] = df['price_gap'] * df['volume_growth']
    weighted_average_price_gaps = df['weighted_price_gap'].rolling(window=20).sum() / 20

    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_differential = rolling_high - rolling_low

    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    weighted_high_low_diff = high_low_differential * volume_volatility

    # Aggregate the Signals
    price_difference = df['close'] - df['close'].shift(10)
    aggregate_signal = weighted_high_low_diff + (df['close'] - df['close'].shift(10))

    # Calculate Price Momentum
    one_day_return = df['close'].pct_change(1)
    five_day_return = df['close'].pct_change(5)
    twenty_day_return = df['close'].pct_change(20)
    average_momentum = (one_day_return + five_day_return + twenty_day_return) / 3

    # Adjust for Volume
    volume_percentage_change = df['volume'].pct_change(1)
    adjusted_aggregate_signal = aggregate_signal * volume_percentage_change

    # Final Alpha Factor
    alpha_factor = (weighted_average_price_gaps 
                    + adjusted_aggregate_signal 
                    - average_momentum)

    return alpha_factor.dropna()
