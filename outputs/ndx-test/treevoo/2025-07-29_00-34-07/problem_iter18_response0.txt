```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between Open and Close
    price_gap = df['open'] - df['close']
    
    # Calculate Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Weight the Price Gaps by Volume Growth
    weighted_price_gaps = (price_gap * volume_growth).sum()
    
    # Calculate 20-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_differential = rolling_high - rolling_low
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    
    # Weight the Rolling High-Low Differential by Volume Volatility
    weighted_high_low_diff = high_low_differential * volume_volatility
    
    # Aggregate the Signals
    signal = weighted_high_low_diff + (df['close'] - df['close'].shift(15))
    
    # Calculate Price Momentum
    momentum_1d = df['close'] / df['close'].shift(1) - 1
    momentum_7d = df['close'] / df['close'].shift(7) - 1
    momentum_30d = df['close'] / df['close'].shift(30) - 1
    
    # Average the Momentum
    avg_momentum = (momentum_1d + momentum_7d + momentum_30d) / 3
    
    # Adjust for Volume
    volume_percentage_change = df['volume'].pct_change()
    adjusted_signal = signal * volume_percentage_change
    
    # Final Alpha Factor
    alpha_factor = weighted_price_gaps + adjusted_signal - avg_momentum
    
    return alpha_factor
```
