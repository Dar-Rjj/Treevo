import pandas as pd

def heuristics_v2(df):
    # Calculate 14-day Moving Average (MA)
    df['14_day_MA'] = df['close'].rolling(window=14).mean()
    
    # Trend-following signal based on MA
    df['MA_signal'] = (df['close'] > df['14_day_MA']).astype(int) - (df['close'] < df['14_day_MA']).astype(int)
    
    # Calculate 20-day Exponential Moving Average (EMA)
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Smoother trend indicator based on EMA
    df['EMA_signal'] = (df['close'] > df['20_day_EMA']).astype(int) - (df['close'] < df['20_day_EMA']).astype(int)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    # VWAP benchmark for trading decisions
    df['VWAP_signal'] = (df['close'] > df['VWAP']).astype(int) - (df['close'] < df['VWAP']).astype(int)
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = ((df['close'] > df['close'].shift(1)) - (df['close'] < df['close'].shift(1))) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # OBV to identify buying or selling pressure
    df['OBV_signal'] = (df['OBV'] > df['OBV'].shift(1)).astype(int) - (df['OBV'] < df['OBV'].shift(1)).astype(int)
    
    # Identify Bullish and Bearish Engulfing Patterns
    df['prev_high'] = df['high'].shift(1)
    df['prev_low'] = df['low'].shift(1)
    df['prev_close'] = df['close'].shift(1)
    df['engulfing_bullish'] = (df['close'] > df['open']) & (df['open'] <= df['prev_close']) & (df['close'] >= df['prev_high'])
    df['engulfing_bearish'] = (df['close'] < df['open']) & (df['open'] >= df['prev_close']) & (df['close'] <= df['prev_low'])
    df['engulfing_signal'] = df['engulfing_bullish'].astype(int) - df['engulfing_bearish'].astype(int)
    
    # Identify Inside and Outside Days
    df['inside_day'] = (df['high'] < df['prev_high']) & (df['low'] > df['prev_low'])
    df['outside_day'] = (df['high'] > df['prev_high']) & (df['low'] < df['prev_low'])
    df['inside_outside_signal'] = (df['outside_day'] & (df['close'] > df['close'].shift(1))).astype(int) - (df['inside_day'] & (df['close'] < df['close'].shift(1))).astype(int)
    
    # Calculate Average True Range (ATR)
    df['TR'] = df[['high' - 'low', 'high' - 'close'].shift(1), abs('low' - 'close').shift(1)].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Bollinger Bands
    df['middle_band'] = df['close'].rolling(window=20).mean()
    df['upper_band'] = df['middle_band'] + 2 * df['close'].rolling(window=20).std()
    df['lower_band'] = df['middle_band'] - 2 * df['close'].rolling(window=20).std()
    df['bollinger_signal'] = (df['close'] <= df['lower_band']).astype(int) - (df['close'] >= df['upper_band']).astype(int)
