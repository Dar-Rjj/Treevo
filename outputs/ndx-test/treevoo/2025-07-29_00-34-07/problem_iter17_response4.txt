```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Adjusted Relative Volume
    n = 20  # Number of days to calculate average volume
    avg_volume = df['volume'].rolling(window=n).mean()
    relative_volume = df['volume'] / avg_volume
    volume_filter = (relative_volume > 1.5)
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = 150 * high_low_range + price_change
    
    # Smoothing and Initial Momentum Calculation
    sma_7 = combined_value.rolling(window=7).mean()
    initial_momentum = sma_7 - sma_7.shift(7)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    momentum_3_day = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10_day = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_30_day = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Average the Momentum
    avg_momentum = (momentum_3_day + momentum_10_day + momentum_30_day) / 3
    
    # Adjust for Volume Changes
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_vol_adjusted_momentum = avg_momentum * (1 + 2 * daily_vol_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + final_vol_adjusted_momentum
    
    # Calculate High-Low Range Ratio
    high_low_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    weighted_close_open_return = (df['close'] - df['open']) * df['volume']
    sum_weighted_close_open_return = weighted_close_open_return.rolling(window=n).sum()
    
    # Calculate Cumulative High-Low Momentum for Filtered Days
    filtered_high_low_range = high_low_range * volume_filter
    cumulative_high_low_momentum = filtered_high_low_range.rolling(window=n).sum()
    
    # Generate Alpha Factor based on Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    alpha_factor_cumulative = cumulative_high_low_momentum + sum_weighted_close_open_return
    
    # Calculate Weighted Price Momentum
    close_ma_15 = df['close'].rolling(window=15).mean()
    weighted_price_momentum = (df['close'] - close_ma_15) * daily_log_return
    
    # Final Factor Construction
    final_factor = intermediate_factor + weighted_price_momentum + alpha_factor_cumulative
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    rolling_high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    weighted_rolling_high_low = rolling_high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = weighted_rolling_high_low + (df['close'] - df['close'].shift(1)) * volume_filter
    
    # Generate Alpha Factor
    alpha_factor = final_factor * aggregated_signal
    
    return alpha_factor
```
