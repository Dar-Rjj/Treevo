```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date
    """
    
    # Calculate the average price
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the change in average price
    delta_avg_price = avg_price.diff(1)
    
    # Calculate the relative volume change
    rel_vol_change = df['volume'].pct_change(1)
    
    # Heuristic: Alpha factor is the product of the change in average price and the relative volume change
    alpha_factor = delta_avg_price * rel_vol_change
    
    return alpha_factor
```
