```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-day, 5-day, and 20-day Price Returns
    df['1d_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['5d_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20d_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted Average the Momentum
    df['weighted_momentum'] = (0.6 * df['1d_return'] +
                               0.3 * df['5d_return'] +
                               0.1 * df['20d_return'])
    
    # Adjust for Volume and Amount
    df['volume_pct_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    df['amount_pct_change'] = (df['amount'] / df['amount'].shift(1)) - 1
    df['volume_adjusted_momentum'] = df['weighted_momentum'] * df['volume_pct_change']
    df['amount_adjusted_momentum'] = df['volume_adjusted_momentum'] * df['amount_pct_change']
    
    # Incorporate High-Low Differential and Volume Volatility
    df['20d_high'] = df['high'].rolling(window=20).max()
    df['20d_low'] = df['low'].rolling(window=20).min()
    df['high_low_diff'] = df['20d_high'] - df['20d_low']
    df['volatility_vol'] = df['volume'].rolling(window=10).std()
    df['combined_volatility'] = (df['high_low_diff'] / df['volatility_vol']).rolling(window=5).mean()
    
    # Integrate Open-Close Spread Impact
    df['open_close_spread'] = (df['open'] - df['close']).rolling(window=20).mean()
    df['enhanced_volatility'] = df['open_close_spread'] * df['volatility_vol'] + df['combined_volatility']
    
    # Combine Momentum and Enhanced Volatility Components
    df['aggregated_signal'] = df['amount_adjusted_momentum'] + df['enhanced_volatility']
    
    # Final Factor Calculation
    df['final_factor'] = (df['aggregated_signal'] / df['volume']) * df['amount_pct_change']
    
    return df['final_factor']
```
