```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = high_to_low_range * 100 + price_change
    
    # Smoothing and Initial Momentum Calculation
    sma_5 = combined_value.rolling(window=5).mean()
    initial_momentum = sma_5 - sma_5.shift(5)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Difference
    volume_diff = df['volume'] - df['volume'].shift(1)
    
    # Calculate Weighted Price Momentum
    close_ma_10 = df['close'].rolling(window=10).mean()
    weighted_price_momentum = (df['close'] - close_ma_10) * daily_log_return
    
    # Calculate Weighted Volume Momentum
    weighted_volume_momentum = volume_diff * df['volume']
    
    # Calculate Price Gap
    price_gap = df['high'] - df['low']
    
    # Calculate Volume Trend
    volume_sma_5 = df['volume'].rolling(window=5).mean()
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_growth).sum()
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + weighted_price_momentum
    
    # Extended Factor Construction
    extended_factor = intermediate_factor + weighted_volume_momentum
    
    # Final Factor Construction
    final_factor = extended_factor + weighted_price_gaps
    
    # Calculate Price Momentum
    one_day_price_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    five_day_price_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    twenty_day_price_return = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Weighted Average Momentum
    short_term_weight = 0.4
    medium_term_weight = 0.3
    long_term_weight = 0.3
    weighted_average_momentum = (short_term_weight * one_day_price_return +
                                 medium_term_weight * five_day_price_return +
                                 long_term_weight * twenty_day_price_return)
    
    # Adjust for Volume
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Integrate Transaction Amount
    daily_amount_change = df['amount'] - df['amount'].shift(1)
    final_integrated_factor = final_factor * (1 + (daily_volume_change + daily_amount_change) / df['volume'].shift(1))
    
    return final_integrated_factor
```
