```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(15)
    
    # Calculate 25-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregate_signal = weighted_high_low_diff - price_diff
    
    # Calculate Price Momentum
    price_momentum_2d = (df['close'] / df['close'].shift(2) - 1) * 100
    price_momentum_7d = (df['close'] / df['close'].shift(7) - 1) * 100
    price_momentum_30d = (df['close'] / df['close'].shift(30) - 1) * 100
    
    # Average the Momentum
    avg_momentum = (price_momentum_2d + price_momentum_7d + price_momentum_30d) / 3
    
    # Adjust for Volume
    volume_change = (df['volume'] / df['volume'].shift(1) - 1) * 100
    adjusted_aggregate_signal = aggregate_signal * (1 + volume_change / 100)
    
    # Final Aggregate Signal
    final_signal = adjusted_aggregate_signal - avg_momentum
    
    return final_signal
```
