```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Compute 5-day Moving Average of Volumes
    volume_trend = df['volume'].rolling(window=5).mean()
    
    # Compute Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Multiply 5-day Moving Average by Volume Growth
    volume_trend *= volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-to-Low Range with Weighted Price Gaps
    combined_high_low = (high_low_range * 100) + weighted_price_gaps
    
    # Smoothing and Momentum Calculation
    smoothed_combined = combined_high_low.rolling(window=5).mean()
    momentum_combined = smoothed_combined - smoothed_combined.shift(5)
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_price_change = (high_low_range * 100) + price_change
    
    # Intermediate Smoothing and Momentum
    smoothed_price_change = combined_price_change.rolling(window=5).mean()
    intermediate_momentum = smoothed_price_change - smoothed_price_change.shift(5)
    
    # Calculate Multi-Day Price Momentum
    one_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    five_day_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    twenty_day_return = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Average the Momentum
    averaged_momentum = (one_day_return + five_day_return + twenty_day_return) / 3
    
    # Adjust for Volume Changes
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = averaged_momentum * (1 + daily_vol_change)
    
    # Intermediate Factor Combination
    intermediate_factor = momentum_combined + final_volume_adjustment
    
    # Calculate Weighted Price Momentum
    log_return = np.log(df['close'] / df['close'].shift(1))
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=10).mean()) * log_return
    
    # Final Alpha Factor
    alpha_factor = weighted_price_gaps + intermediate_factor + intermediate_momentum + weighted_price_momentum
    
    return alpha_factor
```
