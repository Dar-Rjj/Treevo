```python
def heuristics_v2(df):
    # Calculate the difference between today's close and yesterday's close, then multiply by today's volume
    df['close_vol_diff'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate the range (High - Low) and divide by the previous day's Close
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Determine if the open is higher or lower than the previous day's close and assign a value (+1 or -1)
    df['open_change'] = (df['open'] > df['close'].shift(1)).astype(int) * 2 - 1
    
    # Compute the sum of the last 5 days' closing price differences
    df['momentum_5d'] = df['close'].diff().rolling(window=5).sum()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['close_vol_diff'] +
        df['intraday_volatility'] +
        df['open_change'] +
        df['momentum_5d']
    )
    
    return alpha_factor
```
