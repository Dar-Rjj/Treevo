```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Weighted Price Momentum
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['weighted_price_momentum'] = (df['close'] - df['ma_10']) * df['log_return']
    
    # Calculate Weighted Volume Momentum
    df['weighted_volume_momentum'] = df['volume_diff'] * df['volume']
    
    # Combine Factors
    df['combined_factor'] = (df['weighted_price_momentum'] + df['weighted_volume_momentum']) / df['volume']
    
    return df['combined_factor']
```
