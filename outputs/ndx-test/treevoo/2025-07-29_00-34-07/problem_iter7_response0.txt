```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Price Momentum
    df['1_day_return'] = df['close'].pct_change(1)
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)

    # Average the Momentum
    df['average_momentum'] = (df['1_day_return'] + df['5_day_return'] + df['20_day_return']) / 3

    # Adjust for Volume
    df['volume_change'] = df['volume'].pct_change(1)
    df['adjusted_momentum'] = df['average_momentum'] * df['volume_change']

    # Incorporate High-Low Momentum with Volume Filter
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Average Volume over a period
    n_days = 20
    df['average_volume'] = df['volume'].rolling(window=n_days).mean()
    
    # Apply Volume Filter Condition
    df_filtered = df[df['volume'] > df['average_volume']]
    
    # Combine with Adjusted Momentum
    df['alpha_factor'] = df_filtered['adjusted_momentum'] + df_filtered['daily_price_range']
    
    return df['alpha_factor']
```
