```python
import pandas as pd

def heuristics(df):
    # Calculate Price Difference
    df['price_diff'] = df['close'] - df['close'].shift(15)
    
    # Calculate 25-day Rolling High-Low Differential
    df['rolling_high_25'] = df['high'].rolling(window=25).max()
    df['rolling_low_25'] = df['low'].rolling(window=25).min()
    df['high_low_diff_25'] = df['rolling_high_25'] - df['rolling_low_25']
    
    # Volume Volatility
    df['volume_volatility'] = df['volume'].rolling(window=15).std()
    
    # Weighted 25-day Rolling High-Low Differential
    df['weighted_high_low_diff_25'] = df['high_low_diff_25'] * df['volume_volatility']
    
    # Aggregate the Signals
    df['aggregated_signal'] = df['weighted_high_low_diff_25'] - df['price_diff']
    
    # Adjust for Volume
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_adjusted_signal'] = df['aggregated_signal'] * (1 + 2 * df['volume_change_pct'])
    
    # Multi-Day Price Momentum
    df['3_day_return'] = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    df['10_day_return'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['30_day_return'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    df['avg_momentum'] = (df['3_day_return'] + df['10_day_return'] + df['30_day_return']) / 3
    
    # Final Volume Adjustment
    df['final_volume_adjusted_momentum'] = df['avg_momentum'] * (1 + 2 * df['volume_change_pct'])
    
    # Volume-Weighted Close-to-Open Return
    df['volume_weighted_close_open'] = (df['close'] - df['open']) * df['volume']
    df['cumulative_volume_weighted_close_open'] = df['volume_weighted_close_open'].rolling(window=25).sum()
    
    # Combine Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    df['alpha_factor'] = df['high_low_diff_25'] + df['cumulative_volume_weighted_close_open']
    
    # Weighted Price Momentum
    df['weighted_price_momentum'] = (df['close'] - df['close'].rolling(window=15).mean()) * (df['close'] / df['close'].shift(1) - 1)
    
    # Final Factor Construction
    df['final_factor'] = df['final_volume_adjusted_momentum'] + df['weighted_price_momentum'] + df['alpha_factor']
    
    return df['final_factor'].dropna()
```
