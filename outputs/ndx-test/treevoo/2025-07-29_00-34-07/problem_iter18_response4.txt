```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of close price from the previous day
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the range between high and low for the day
    df['day_range'] = df['high'] - df['low']
    
    # Calculate the ratio of close to open price
    df['close_open_ratio'] = df['close'] / df['open']

    # Combine the indicators into a single factor
    df['factor_value'] = (df['close_pct_change'] * 0.5) + (df['day_range'] / df['close']) + (df['close_open_ratio'] - 1)
    
    return df['factor_value'].dropna()
```
