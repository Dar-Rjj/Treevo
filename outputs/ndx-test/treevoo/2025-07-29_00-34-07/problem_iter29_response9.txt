```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Factors
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Momentum_Signal'] = df['SMA_10'] - df['SMA_50']
    
    # Calculate Volume Factors
    df['AVG_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['AVG_Volume_50'] = df['volume'].rolling(window=50).mean()
    df['Volume_Signal'] = df['AVG_Volume_10'] - df['AVG_Volume_50']
    
    # Generate Price and Volume Combination Factor
    df['Price_Volume_Strength'] = df['Momentum_Signal'] * df['Volume_Signal']
    
    # Calculate Open-High-Low-Close Factors
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Directional_Bias'] = (df['close'] > df['open']).astype(int) * 2 - 1
    df['Intraday_Volatility_Weighted_Direction'] = df['Intraday_Range'] * df['Intraday_Directional_Bias']
    
    # Combine All Factors
    df['Composite_Alpha_Factor'] = df['Price_Volume_Strength'] + df['Intraday_Volatility_Weighted_Direction']
    
    return df['Composite_Alpha_Factor']
```
