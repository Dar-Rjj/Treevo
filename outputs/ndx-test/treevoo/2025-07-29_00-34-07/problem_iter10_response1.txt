```python
import pandas as pd

def heuristics_v2(data):
    # Calculate Price Difference
    price_diff = data['close'] - data['close'].shift(10)
    
    # Calculate 20-day Rolling High-Low Differential
    rolling_high = data['high'].rolling(window=20).max()
    rolling_low = data['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility (Standard Deviation of Volume over the last 10 days)
    volume_volatility = data['volume'].rolling(window=10).std()
    
    # Weight the High-Low Differential by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = weighted_high_low_diff - price_diff
    
    # Calculate Price Momentum
    one_day_return = (data['close'] - data['close'].shift(1)) / data['close'].shift(1)
    five_day_return = (data['close'] - data['close'].shift(5)) / data['close'].shift(5)
    twenty_day_return = (data['close'] - data['close'].shift(20)) / data['close'].shift(20)
    
    # Average the Momentum
    avg_momentum = (one_day_return + five_day_return + twenty_day_return) / 3
    
    # Calculate Daily Volume Percentage Change
    volume_pct_change = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)
    
    # Adjust for Volume
    adjusted_signal = aggregated_signal * volume_pct_change
    
    # Final Aggregate Signal
    final_signal = adjusted_signal - avg_momentum
    
    return final_signal
```
