```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']

    # Compute 5-day Moving Average of Volumes
    vol_5d_ma = df['volume'].rolling(window=5).mean()
    
    # Compute Current Day Volume Growth
    vol_growth = df['volume'] / df['volume'].shift(1)
    
    # Multiply 5-day Moving Average by Volume Growth
    volume_trend = vol_5d_ma * vol_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()

    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-to-Low Range with Weighted Price Gaps
    combined_value = high_low_range * 100 + weighted_price_gaps
    
    # Smoothing and Momentum Calculation
    smoothed_combined_value = combined_value.rolling(window=5).mean()
    high_low_momentum = smoothed_combined_value - smoothed_combined_value.shift(5)

    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)

    # Calculate Rolling High-Low Range
    rolling_high = df['high'].rolling(window=30).max()
    rolling_low = df['low'].rolling(window=30).min()
    rolling_high_low_range = rolling_high - rolling_low

    # Calculate Volume Trend for Weighting
    vol_10d_ma = df['volume'].rolling(window=10).mean()
    volume_flow = df['volume'] - vol_10d_ma

    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps 
                    + high_low_momentum 
                    + (rolling_high_low_range * volume_flow) 
                    - price_momentum)
    
    return alpha_factor
```
