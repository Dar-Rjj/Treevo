```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Difference
    adj_price_diff = (df['close'] - df['close'].shift(10)) * np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Rolling High-Low Differential
    rolling_high_20 = df['high'].rolling(window=20).max()
    rolling_low_20 = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high_20 - rolling_low_20
    
    # Weight by Volume Volatility
    vol_volatility = df['volume'].rolling(window=10).std()
    weighted_high_low_diff = high_low_diff * vol_volatility
    
    # Calculate Volume Adjusted Momentum
    volume_momentum = (df['volume'] - df['volume'].shift(1)) * (df['close'] - df['close'].rolling(window=10).mean())
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Rolling High-Low Range
    rolling_high_30 = df['high'].rolling(window=30).max()
    rolling_low_30 = df['low'].rolling(window=30).min()
    high_low_range = rolling_high_30 - rolling_low_30
    
    # Weight by Volume Trend
    vol_trend = df['volume'] - df['volume'].rolling(window=10).mean()
    weighted_high_low_range = high_low_range * vol_trend
    
    # Aggregate the Signals
    aggregated_signal = (weighted_high_low_diff - adj_price_diff + 
                         volume_momentum + weighted_high_low_range - 
                         price_momentum)
    
    # Final Factor
    final_factor = aggregated_signal / df['volume']
    
    return final_factor
```
