```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 15-Day Price Difference
    df['15d_price_diff'] = df['close'].shift(0) - df['close'].shift(15)
    
    # Calculate 25-Day Rolling High-Low Differential
    df['25d_rolling_high'] = df['high'].rolling(window=25).max()
    df['25d_rolling_low'] = df['low'].rolling(window=25).min()
    df['25d_high_low_diff'] = df['25d_rolling_high'] - df['25d_rolling_low']
    
    # Daily Volume Percentage Change
    df['daily_vol_perc_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Final Volume Adjustment
    df['final_vol_adjusted_high_low'] = df['25d_high_low_diff'] * (1 + 2 * df['daily_vol_perc_change'])
    
    # Aggregate the Signals
    df['aggregated_signal'] = df['final_vol_adjusted_high_low'] - df['15d_price_diff']
    
    # Calculate Average Price Momentum
    df['2d_return'] = (df['close'] / df['close'].shift(2)) - 1
    df['7d_return'] = (df['close'] / df['close'].shift(7)) - 1
    df['30d_return'] = (df['close'] / df['close'].shift(30)) - 1
    df['average_momentum'] = (df['2d_return'] + df['7d_return'] + df['30d_return']) / 3
    
    # Smoothing and Initial Momentum
    df['smoothed_combined_value'] = df['aggregated_signal'].rolling(window=7).mean()
    df['initial_momentum'] = df['smoothed_combined_value'] - df['smoothed_combined_value'].shift(7)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    df['daily_price_range'] = df['high'] - df['low']
    df['sum_daily_price_ranges'] = df['daily_price_range'].rolling(window=7).sum()
    df['weighted_price_gap'] = (df['close'] - df['close'].shift(1)) * df['daily_vol_perc_change']
    df['sum_weighted_price_gaps'] = df['weighted_price_gap'].rolling(window=7).sum()
    
    # Intermediate Factor Combination
    df['intermediate_factor'] = df['initial_momentum'] + df['average_momentum']
    
    # Calculate Weighted Price Momentum
    df['15d_ma_close'] = df['close'].rolling(window=15).mean()
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['weighted_price_momentum'] = (df['close'] - df['15d_ma_close']) * df['log_return']
    
    # Final Alpha Factor Construction
    df['final_alpha_factor'] = (df['sum_daily_price_ranges'] + df['sum_weighted_price_gaps']) * df['average_momentum'] + df['intermediate_factor'] + df['weighted_price_momentum']
    
    return df['final_alpha_factor'].dropna()
```
