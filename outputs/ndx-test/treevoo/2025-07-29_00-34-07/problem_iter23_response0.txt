```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Average of Price Gaps
    price_gap = df['high'] - df['low']
    volume_trend = df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))
    weighted_price_gap = (price_gap * volume_trend).sum()

    # Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low

    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    weighted_high_low_diff = high_low_diff * volume_volatility

    # Aggregate the Signals
    aggregated_signals = weighted_high_low_diff + (df['close'] - df['close'].shift(10))

    # High-to-Low Range
    high_low_range = (df['high'] - df['low']) / df['close'].shift(1)

    # Combine High-to-Low Range with Price Change
    one_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_value = high_low_range * 100 + one_day_return

    # Smoothing and Initial Momentum Calculation
    ema_5 = combined_value.ewm(span=5, adjust=False).mean()
    initial_momentum = ema_5 - ema_5.shift(5)

    # Multi-Day Price Momentum
    one_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    seven_day_return = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    fourteen_day_return = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    thirty_day_return = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Weighted Average Momentum
    short_term_weight = 0.4
    medium_term_weight = 0.3
    long_term_weight = 0.3
    weighted_average_momentum = (short_term_weight * one_day_return) + \
                               (medium_term_weight * seven_day_return) + \
                               (long_term_weight * thirty_day_return)

    # Final Volume Adjustment
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    final_volume_adjustment = weighted_average_momentum * (1 + (daily_volume_change / df['volume'].shift(1)))

    # Integrate Transaction Amount
    daily_amount_change = df['amount'] - df['amount'].shift(1)
    intermediate_volume_adjustment = final_volume_adjustment * (1 + (daily_amount_change / df['amount'].shift(1)))

    # Intermediate Factor Combination
    intermediate_factor_combination = initial_momentum + intermediate_volume_adjustment

    # Volume-Weighted Price Momentum
    ten_day_ema_close = df['close'].ewm(span=10, adjust=False).mean()
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    volume_weighted_price_momentum = df['close'] - ten_day_ema_close * log_return

    # Final Alpha Factor
    final_alpha_factor = weighted_price_gap + intermediate_factor_combination + volume_weighted_price_momentum

    return final_alpha_factor
```
