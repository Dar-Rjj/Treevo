```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Filter by Relative Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_filter'] = df['volume'] > df['avg_volume_20']
    
    # Calculate Adjusted Price Difference
    df['adj_price_diff'] = (df['close'] - df['close'].shift(10)) * np.log(df['close'] / df['close'].shift(10))
    
    # Calculate Rolling High-Low Differential
    df['rolling_high_20'] = df['high'].rolling(window=20).max()
    df['rolling_low_20'] = df['low'].rolling(window=20).min()
    df['roll_hi_lo_diff'] = df['rolling_high_20'] - df['rolling_low_20']
    
    # Weight by Volume Volatility
    df['vol_volatility'] = df['volume'].rolling(window=10).std()
    df['weighted_hi_lo_diff'] = df['roll_hi_lo_diff'] * df['vol_volatility']
    
    # Calculate Price Momentum
    df['mom_1d'] = df['close'].pct_change(1)
    df['mom_5d'] = df['close'].pct_change(5)
    df['mom_20d'] = df['close'].pct_change(20)
    
    # Average the Momentum
    df['avg_mom'] = (df['mom_1d'] + df['mom_5d'] + df['mom_20d']) / 3
    
    # Adjust for Volume
    df['vol_pct_change'] = df['volume'].pct_change(1)
    df['adj_avg_mom'] = df['avg_mom'] * df['vol_pct_change']
    
    # Aggregate the Signals
    df['sum_price_range_filtered'] = df[df['volume_filter']]['price_range'].sum()
    df['cumulative_hi_lo_mom'] = df['weighted_hi_lo_diff'].cumsum()
    df['aggregated_signal'] = (0.5 * df['adj_avg_mom'] + 0.5 * df['cumulative_hi_lo_mom'] - df['adj_price_diff'])
    
    # Final Factor
    df['final_factor'] = df['aggregated_signal'] / df['volume']
    
    return df['final_factor'].dropna()
```
