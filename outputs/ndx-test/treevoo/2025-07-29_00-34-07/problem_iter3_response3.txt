```python
import pandas as pd

def heuristics_v2(df):
    # Compute 1-day, 5-day, and 20-day price returns
    df['1d_return'] = df['close'].pct_change(1)
    df['5d_return'] = df['close'].pct_change(5)
    df['20d_return'] = df['close'].pct_change(20)
    
    # Weighted average of the momentum
    df['weighted_momentum'] = (0.6 * df['1d_return'] + 
                               0.3 * df['5d_return'] + 
                               0.1 * df['20d_return'])
    
    # Adjust for volume
    df['volume_pct_change'] = df['volume'].pct_change(1)
    df['vol_adj_momentum'] = df['weighted_momentum'] * df['volume_pct_change']
    
    # Incorporate amount impact
    df['amount_pct_change'] = df['amount'].pct_change(1)
    df['final_factor'] = df['vol_adj_momentum'] * df['amount_pct_change']
    
    return df['final_factor'].dropna()
```
