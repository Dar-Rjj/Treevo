import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Percentage Change
    intraday_pct_change = (df['close'] - df['open']) / df['open'] * 100
    
    # Calculate Price Gaps
    price_gaps = df['high'] - df['low']
    
    # Calculate Volume Trend
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    volume_growth = df['volume'] / df['volume'].shift(1)
    volume_trend = volume_5day_ma * volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gaps * volume_trend).sum()
    
    # Combine High-to-Low Range with Weighted Price Gaps
    combined_value = intraday_range * 100 + weighted_price_gaps
    
    # Smooth and Calculate Momentum
    smoothed_value = combined_value.rolling(window=5).mean()
    momentum = smoothed_value - smoothed_value.shift(5)
    
    return momentum
