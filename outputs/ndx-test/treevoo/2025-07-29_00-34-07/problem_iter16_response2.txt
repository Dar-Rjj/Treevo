def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Average Price Momentum
    df['1_day_return'] = df['close'] / df['close'].shift(1) - 1
    df['5_day_return'] = df['close'] / df['close'].shift(5) - 1
    df['20_day_return'] = df['close'] / df['close'].shift(20) - 1
    df['avg_momentum'] = (df['1_day_return'] + df['5_day_return'] + df['20_day_return']) / 3

    # Filter by Relative Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] > avg_volume
    filtered_df = df[df['relative_volume']]

    # Adjusted High-Low Cumulative Return and Price Gaps
    if not filtered_df.empty:
        daily_price_range = filtered_df['high'] - filtered_df['low']
        price_gap = (filtered_df['open'] - filtered_df['close']).abs()
        
        volume_growth_factor = filtered_df['volume'] / avg_volume
        adjusted_price_gap = price_gap * volume_growth_factor
        
        # Generate Alpha Factor
        sum_daily_price_range = daily_price_range.sum()
        sum_adjusted_price_gaps = adjusted_price_gap.sum()
        alpha_factor = (sum_daily_price_range + sum_adjusted_price_gaps) * filtered_df['avg_momentum']
        
        return alpha_factor
    else:
        return pd.Series(index=df.index, data=0)
