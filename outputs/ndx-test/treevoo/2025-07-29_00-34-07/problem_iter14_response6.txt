```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Improved High-to-Low Range
    high_low_range = (df['high'] - df['low'])
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_low_range * 150) + price_change
    
    # Smoothing and Initial Momentum Calculation
    sma_7 = combined_value.rolling(window=7).mean()
    initial_momentum = sma_7 - sma_7.shift(7)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    momentum_3 = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_30 = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Average the Momentum
    average_momentum = (momentum_3 + momentum_10 + momentum_30) / 3
    
    # Adjust for Volume Changes
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = average_momentum * (1 + 2 * volume_percentage_change)
    
    # Filter by Relative Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    filtered_days = (df['volume'] > 1.5 * avg_volume)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = (df['high'] - df['low'])
    sum_daily_price_ranges = daily_price_range[filtered_days].sum()
    weighted_price_gaps = (daily_price_range * volume_percentage_change)[filtered_days].sum()
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + final_volume_adjustment
    
    # Calculate High-Low Range Ratio
    high_low_ratio = high_low_range / df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) * df['volume']
    volume_weighted_c_o_return = close_to_open_return.rolling(window=20).sum()
    
    # Combine Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    alpha_factor = sum_daily_price_ranges + weighted_price_gaps + intermediate_factor + volume_weighted_c_o_return
    
    # Calculate Weighted Price Momentum
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=15).mean()) * daily_log_return
    
    # Final Alpha Factor Construction
    final_alpha_factor = (sum_daily_price_ranges + weighted_price_gaps) * final_volume_adjustment + intermediate_factor + weighted_price_momentum + alpha_factor
    
    return final_alpha_factor
```
