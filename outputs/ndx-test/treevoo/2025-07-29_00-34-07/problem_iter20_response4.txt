```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Price Difference
    df['close_ema_3'] = df['close'].ewm(span=3).mean()
    df['enhanced_price_diff'] = df['close'] - df['close'].shift(20) - (df['close_ema_3'] - df['close_ema_3'].shift(20))
    
    # Calculate 50-day Rolling High-Low Differential
    df['rolling_high_50'] = df['high'].rolling(window=50).max()
    df['rolling_low_50'] = df['low'].rolling(window=50).min()
    df['high_low_diff_50'] = df['rolling_high_50'] - df['rolling_low_50']
    
    # Weight by Volume Volatility
    df['volume_volatility_20'] = df['volume'].rolling(window=20).std()
    df['weighted_high_low_diff_50'] = df['high_low_diff_50'] * df['volume_volatility_20']
    
    # Aggregate the Signals
    df['aggregated_signal'] = df['weighted_high_low_diff_50'] - df['enhanced_price_diff']
    
    # Calculate Averaged Price Momentum
    df['price_return_3d'] = (df['close'] / df['close'].shift(3)) - 1
    df['price_return_10d'] = (df['close'] / df['close'].shift(10)) - 1
    df['price_return_40d'] = (df['close'] / df['close'].shift(40)) - 1
    df['averaged_price_momentum'] = (df['price_return_3d'] + df['price_return_10d'] + df['price_return_40d']) / 3
    
    # Adjust for Volume
    df['volume_pct_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    df['adjusted_aggregated_signal'] = df['aggregated_signal'] * df['volume_pct_change']
    
    return df['adjusted_aggregated_signal']
```
