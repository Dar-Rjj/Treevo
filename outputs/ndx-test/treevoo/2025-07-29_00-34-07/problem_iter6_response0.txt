```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap
    df['price_gap'] = df['high'] - df['low']
    
    # Calculate Volume Trend
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_growth'] = df['volume'] / df['volume'].shift(1)
    
    # Incorporate Amount Influence
    df['amount_ma_5'] = df['amount'].rolling(window=5).mean()
    df['amount_growth'] = df['amount'] / df['amount'].shift(1)
    
    # Weight the Price Gaps
    df['weighted_price_gap'] = df['price_gap'] * df['volume_growth'] * df['amount_growth']
    factor_values = df['weighted_price_gap'].sum(skipna=True)
    
    return pd.Series(factor_values, index=df.index)
```
