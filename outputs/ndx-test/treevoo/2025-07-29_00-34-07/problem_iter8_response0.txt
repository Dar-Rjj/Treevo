```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['High-Low_Range'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Price Momentum
    df['1D_Price_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5D_Price_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['20D_Price_Return'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Combine High-to-Low Range with 1-day Price Return
    df['Combined_Value'] = df['High-Low_Range'] * 100 + df['1D_Price_Return']
    
    # Smoothing and Initial Momentum Calculation
    df['SMA_Combined_5D'] = df['Combined_Value'].rolling(window=5).mean()
    df['Momentum_Difference'] = df['SMA_Combined_5D'] - df['SMA_Combined_5D'].shift(5)
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Weighted Average Momentum
    df['Weighted_Avg_Momentum'] = 0.4 * df['1D_Price_Return'] + 0.3 * df['5D_Price_Return'] + 0.3 * df['20D_Price_Return']
    
    # Adjust for Volume
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Integrate Transaction Amount
    df['Daily_Amount_Change'] = df['amount'] - df['amount'].shift(1)
    df['Final_Volume_Adjustment'] = df['Weighted_Avg_Momentum'] * (1 + (df['Daily_Volume_Change'] + df['Daily_Amount_Change']) / df['volume'].shift(1))
    
    # Final Factor
    df['Final_Factor'] = df['Momentum_Difference'] + df['Final_Volume_Adjustment']

    return df['Final_Factor']
```
