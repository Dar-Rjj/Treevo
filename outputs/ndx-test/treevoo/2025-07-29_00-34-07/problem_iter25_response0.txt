import pandas as pd

def heuristics_v2(df):
    # Calculate daily log return
    df['log_return'] = (df['close'].apply(np.log) - df['close'].shift(1).apply(np.log))
    
    # Momentum: Cumulative return over 10 days
    df['momentum_10d'] = df['log_return'].rolling(window=10).sum()
    
    # Overnight return: Difference between today's open and yesterday's close
    df['overnight_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Intraday price movement: High, Low, Close difference
    df['intraday_movement'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    
    # True range
    df['true_range'] = df[['high' - 'low', 
                           abs(df['high'] - df['close'].shift(1)), 
                           abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    
    # Average true range (ATR) over 14 days
    df['atr_14d'] = df['true_range'].rolling(window=14).mean()
    
    # Volume change from the previous day
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    
    # Ratio of today's volume to the average volume over 20 days
    df['volume_ratio_20d'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Money flow index (MFI) using typical price and volume
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df['money_flow'].where(df['close'] > df['close'].shift(1), 0)
    df['negative_money_flow'] = df['money_flow'].where(df['close'] < df['close'].shift(1), 0)
    df['mfi_positive'] = df['positive_money_flow'].rolling(window=14).sum()
    df['mfi_negative'] = df['negative_money_flow'].rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + (df['mfi_positive'] / df['mfi_negative'])))
    
    # Volume-weighted moving average (VWMA)
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Total transaction amount for the day
    df['total_transaction_amount'] = df['close'] * df['volume']
    
    # Ratio of today's transaction amount to the average amount over 30 days
    df['transaction_amount_ratio_30d'] = df['total_transaction_amount'] / df['total_transaction_amount'].rolling(window=30).mean()
    
    # Select the final alpha factor
    df['alpha_factor'] = (df['momentum_10d'] + 
                          df['overnight_return'] + 
                          df['intraday_movement'] / df['close'] + 
                          df['atr_14d'] + 
                          df['volume_change'] + 
                          df['volume_ratio_20d'] + 
                          df['mfi'] + 
                          df['vwap'] + 
                          df['transaction_amount_ratio_30d'])
    
    return df['alpha_factor']
