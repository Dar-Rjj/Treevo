```python
import pandas as pd

def heuristics_v2(df, n=30, lookback_period=10):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Filter by Relative Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df_filtered = df[df['volume'] > df['avg_volume']]
    
    # Calculate High-Low Range Ratio
    df_filtered['high_low_range_ratio'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df_filtered['vol_weighted_return'] = ((df_filtered['close'] - df_filtered['open']) * df_filtered['volume']).rolling(window=lookback_period).sum() / df_filtered['volume'].rolling(window=lookback_period).sum()
    
    # Calculate Cumulative High-Low Momentum
    df_filtered['cum_high_low_momentum'] = df_filtered['price_range'].rolling(window=lookback_period).sum()
    
    # Generate Alpha Factor based on Cumulative High-Low Momentum
    df_filtered['alpha_factor'] = df_filtered['cum_high_low_momentum'] + df_filtered['vol_weighted_return']
    
    return df_filtered['alpha_factor']
```
