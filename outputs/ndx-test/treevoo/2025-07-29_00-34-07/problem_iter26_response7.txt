import pandas as pd

def heuristics_v2(df):
    # Calculate the money flow multiplier
    mfm = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mfm * df['volume']
    
    # Calculate the 14-day money flow ratio
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    positive_mf = mf_volume.where(mf_volume > 0, 0)
    negative_mf = mf_volume.where(mf_volume < 0, 0).abs()
    positive_sum = positive_mf.rolling(window=14).sum()
    negative_sum = negative_mf.rolling(window=14).sum()
    money_flow_ratio = positive_sum / negative_sum
    
    # Calculate the 14-day money flow index
    money_flow_index = 100 - (100 / (1 + money_flow_ratio))
    
    # Generate the alpha factor using the money flow index
    alpha_factor = (money_flow_index - money_flow_index.shift(1)) / money_flow_index.shift(1)
    
    return alpha_factor
