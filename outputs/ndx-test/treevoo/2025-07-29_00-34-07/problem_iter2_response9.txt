import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Intraday Percentage Change
    df['intraday_percentage_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Average Intraday Volatility over the last 10 days
    df['avg_intraday_volatility'] = df['intraday_volatility'].rolling(window=10).mean()
    
    # Calculate Intraday Volatility Momentum
    df['intraday_volatility_momentum'] = df['intraday_volatility'] - df['avg_intraday_volatility']
    
    # Combine with Intraday Percentage Change and sum over the last 5 days
    df['combined_volatility_change'] = (df['intraday_volatility_momentum'] * df['intraday_percentage_change']).rolling(window=5).sum()
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Volume Difference
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Weighted Price Momentum
    df['weighted_price_momentum'] = (df['close'] - df['close'].rolling(window=10).mean()) * df['log_return']
    
    # Calculate Weighted Volume Momentum
    df['weighted_volume_momentum'] = df['volume_difference'] * df['volume']
    
    # Synthesize Factors
    df['synthesized_factors'] = (
        df['weighted_price_momentum'].sum() + 
        df['weighted_volume_momentum'].sum()
    ) / df['volume'] * df['combined_volatility_change']
    
    # Return the synthesized factor values
    return df['synthesized_factors'].dropna()
