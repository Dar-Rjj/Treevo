```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['open']
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_low_range * 150) + price_change
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = combined_value.rolling(window=7).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(7)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    three_day_return = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    thirty_day_return = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Average the Momentum
    average_momentum = (three_day_return + ten_day_return + thirty_day_return) / 3
    
    # Adjust for Volume Changes
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjusted_momentum = average_momentum * (1 + 2 * volume_percentage_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + final_volume_adjusted_momentum
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open = (df['close'] - df['open']) * df['volume']
    volume_weighted_close_to_open_sum = volume_weighted_close_to_open.rolling(window=7).sum()
    
    # Combine Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    cumulative_high_low_momentum = high_low_range_ratio.rolling(window=7).sum()
    alpha_factor = cumulative_high_low_momentum + volume_weighted_close_to_open_sum
    
    # Calculate Weighted Price Momentum
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=15).mean()) * daily_log_return
    
    # Final Factor Construction
    final_factor = intermediate_factor + weighted_price_momentum + alpha_factor
    
    return final_factor
```
