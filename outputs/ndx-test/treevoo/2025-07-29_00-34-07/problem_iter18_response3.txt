```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_growth).sum()
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=10).max()
    rolling_low = df['low'].rolling(window=10).min()
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std()
    
    # Weight by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    close_t = df['close']
    close_t_20 = df['close'].shift(20)
    aggregate_signal = weighted_high_low_diff + (close_t - close_t_20)
    
    # Calculate Price Momentum
    mom_1d = close_t / close_t.shift(1) - 1
    mom_5d = close_t / close_t.shift(5) - 1
    mom_20d = close_t / close_t.shift(20) - 1
    
    # Average the Momentum
    avg_momentum = (mom_1d + mom_5d + mom_20d) / 3
    
    # Adjust for Volume
    volume_change = (df['volume'] / df['volume'].shift(1)) - 1
    adjusted_aggregate_signal = aggregate_signal * volume_change
    
    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps + adjusted_aggregate_signal) - avg_momentum
    
    return alpha_factor
```
