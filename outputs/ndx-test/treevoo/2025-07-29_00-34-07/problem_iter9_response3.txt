```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Gap
    price_gap = df['open'] - df['close']
    
    # Calculate Volume Momentum
    volume_ema_5 = df['volume'].ewm(span=5, adjust=False).mean()
    volume_ratio = df['volume'] / volume_ema_5
    
    # Incorporate Amount Momentum
    amount_ema_5 = df['amount'].ewm(span=5, adjust=False).mean()
    amount_ratio = df['amount'] / amount_ema_5
    
    # Weight the Price Gaps
    weighted_price_gap = price_gap * volume_ratio * amount_ratio
    
    return weighted_price_gap
```
