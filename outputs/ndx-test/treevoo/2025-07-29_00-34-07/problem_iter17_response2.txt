```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the high and low prices, representing the intraday price range
    intraday_range = df['high'] - df['low']
    
    # Calculate the difference between the close and open prices, representing the daily price movement
    daily_movement = df['close'] - df['open']
    
    # Calculate the percentage of the daily movement relative to the intraday range
    daily_movement_percentage = daily_movement / intraday_range
    
    # Calculate the ratio of the volume to the average volume over the past 10 days
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume_10d
    
    # Combine the daily movement percentage and the volume ratio to form the alpha factor
    alpha_factor = daily_movement_percentage * volume_ratio
    
    return alpha_factor
```
