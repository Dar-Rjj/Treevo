```python
import pandas as pd

def heuristics_v2(df):
    # Compute 1-day, 5-day, and 20-day price returns
    df['1_day_return'] = df['close'].pct_change(1)
    df['5_day_return'] = df['close'].pct_change(5)
    df['20_day_return'] = df['close'].pct_change(20)
    
    # Average the momentum
    df['average_momentum'] = (df['1_day_return'] + df['5_day_return'] + df['20_day_return']) / 3
    
    # Calculate daily volume percentage change
    df['volume_change'] = df['volume'].pct_change(1)
    
    # Adjust for volume
    df['adjusted_momentum'] = df['average_momentum'] * df['volume_change']
    
    # Calculate daily price range
    df['price_range'] = df['high'] - df['low']
    
    # Filter by relative volume
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['filtered_days'] = df['volume'] > df['avg_volume']
    
    # Calculate cumulative high-low momentum for filtered days
    df['cumulative_high_low_momentum'] = df.loc[df['filtered_days'], 'price_range'].transform(lambda x: x.rolling(window=20, min_periods=1).sum())
    
    # Combine with adjusted momentum
    df['alpha_factor'] = df['adjusted_momentum'] + df['cumulative_high_low_momentum'].fillna(0)
    
    return df['alpha_factor']
```
