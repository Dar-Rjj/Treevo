```python
def heuristics_v2(df):
    # Calculate High-Low Range Ratio
    high_low_range = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return and multiply by Volume
    close_to_open_return = (df['close'] / df['open'] - 1) * df['volume']
    
    # Sum over a lookback period, for example, the last 5 days
    lookback_period = 5
    cumulative_return = close_to_open_return.rolling(window=lookback_period).sum()
    
    # Combine the two factors to create the final alpha factor
    alpha_factor = high_low_range * cumulative_return
    
    return alpha_factor
```
