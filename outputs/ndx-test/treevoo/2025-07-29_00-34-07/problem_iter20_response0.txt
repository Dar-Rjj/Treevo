```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Average of Price Gaps
    df['price_gap'] = df['high'] - df['low']
    df['volume_trend'] = df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))
    df['weighted_price_gap'] = (df['price_gap'] * df['volume_trend']).sum()

    # Rolling High-Low Differential
    df['rolling_high_20'] = df['high'].rolling(window=20).max()
    df['rolling_low_20'] = df['low'].rolling(window=20).min()
    df['high_low_diff'] = df['rolling_high_20'] - df['rolling_low_20']

    # Volume Volatility
    df['volume_volatility'] = df['volume'].rolling(window=10).std()
    df['weighted_high_low_diff'] = df['high_low_diff'] * df['volume_volatility']

    # Aggregate the Signals
    df['signal_aggregate'] = df['weighted_high_low_diff'] - (df['close'] - df['close'].shift(10))

    # High-to-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close'].shift(1)

    # Combine High-to-Low Range with Price Change
    df['price_return_1d'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['combined_value'] = df['high_low_range'] * 100 + df['price_return_1d']

    # Smoothing and Initial Momentum Calculation
    df['ema_5'] = df['combined_value'].ewm(span=5, adjust=False).mean()
    df['initial_momentum'] = df['ema_5'] - df['ema_5'].shift(5)

    # Multi-Day Price Momentum
    df['price_return_7d'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['price_return_14d'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)

    # Weighted Average Momentum
    df['weighted_avg_momentum'] = (0.4 * df['price_return_1d']) + (0.3 * df['price_return_7d']) + (0.3 * df['price_return_14d'])

    # Final Volume Adjustment
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['weighted_avg_momentum'] * (1 + (df['volume_change'] / df['volume'].shift(1)))

    # Intermediate Factor Combination
    df['intermediate_factor'] = df['initial_momentum'] + df['volume_adjusted_momentum']

    # Volume-Weighted Price Momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volume_weighted_momentum'] = df['close'] - (df['close'].ewm(span=10, adjust=False).mean() * df['log_return'])

    # Final Alpha Factor
    df['alpha_factor'] = df['weighted_price_gap'] + df['intermediate_factor'] + df['volume_weighted_momentum']

    return df['alpha_factor']
```
