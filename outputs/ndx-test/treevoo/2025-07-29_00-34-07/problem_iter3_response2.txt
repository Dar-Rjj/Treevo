```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Rolling High-Low Range
    rolling_high = df['high'].rolling(window=30).max()
    rolling_low = df['low'].rolling(window=30).min()
    df['rolling_hl_range'] = rolling_high - rolling_low
    
    # Weight by Volume Trend
    vol_10dma = df['volume'].rolling(window=10).mean()
    df['volume_trend'] = df['volume'] - vol_10dma
    
    # Apply volume trend as a weight to the rolling high-low range
    df['weighted_hl_range'] = df['rolling_hl_range'] * (df['volume_trend'] / df['volume'])
    
    # Aggregate the signals
    df['factor'] = df['weighted_hl_range'] - df['price_momentum']
    
    return df['factor'].dropna()
```
