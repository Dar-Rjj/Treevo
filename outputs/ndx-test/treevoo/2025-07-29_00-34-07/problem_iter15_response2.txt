```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Adjusted Relative Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adjusted_relative_volume = np.where(df['volume'] > 1.5 * avg_volume, df['volume'], 0)
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_low_range * 150) + price_change
    
    # Smoothing and Initial Momentum Calculation
    ema_combined = combined_value.ewm(span=7).mean()
    smoothed_momentum = ema_combined - ema_combined.shift(7)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    momentum_3d = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10d = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_20d = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Average the Momentum
    avg_momentum = (momentum_3d + momentum_10d + momentum_20d) / 3
    
    # Adjust for Volume Changes
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = avg_momentum * (1 + 2 * volume_percentage_change)
    
    # Intermediate Factor Combination
    intermediate_factor = smoothed_momentum + final_volume_adjustment
    
    # Calculate High-Low Range Ratio
    high_low_range_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) * df['volume']
    volume_weighted_close_to_open = close_to_open_return.rolling(window=20).sum()
    
    # Calculate Cumulative High-Low Momentum
    filtered_days = df[adjusted_relative_volume > 0]
    cumulative_high_low_momentum = (filtered_days['high'] - filtered_days['low']).sum()
    
    # Generate Alpha Factor based on Cumulative High-Low Momentum
    alpha_factor = cumulative_high_low_momentum + volume_weighted_close_to_open
    
    # Calculate Weighted Price Momentum
    weighted_price_momentum = (df['close'] - df['close'].ewm(span=20).mean()) * daily_log_return
    
    # Final Factor Construction
    final_factor = intermediate_factor + weighted_price_momentum + alpha_factor
    
    return final_factor
```
