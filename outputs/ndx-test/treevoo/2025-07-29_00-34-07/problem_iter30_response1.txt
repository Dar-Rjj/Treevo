```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['open']
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_to_low_range * 150) + price_change
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = combined_value.rolling(window=7).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(7)
    
    # Calculate Multi-Day Price Momentum
    price_return_3d = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    price_return_10d = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    price_return_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Average the Momentum
    average_momentum = (price_return_3d + price_return_10d + price_return_30d) / 3
    
    # Adjust for Volume Changes
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = average_momentum * (1 + 2 * daily_volume_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + volume_adjusted_momentum
    
    # Calculate 25-day Rolling High-Low Differential
    rolling_high_25d = df['high'].rolling(window=25).max()
    rolling_low_25d = df['low'].rolling(window=25).min()
    rolling_high_low_diff = rolling_high_25d - rolling_low_25d
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_diff = rolling_high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = weighted_high_low_diff - initial_momentum
    
    # Adjust for Volume
    aggregated_signal *= daily_volume_change
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) * df['volume']
    cumulative_close_to_open_return = close_to_open_return.rolling(window=25).sum()
    
    # Generate Alpha Factor
    alpha_factor = intermediate_factor + aggregated_signal + cumulative_close_to_open_return
    
    return alpha_factor
```
