```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Price Momentum
    df['1d_return'] = (df['close'].shift(0) - df['close'].shift(1)) / df['close'].shift(1)
    df['5d_return'] = (df['close'].shift(0) - df['close'].shift(5)) / df['close'].shift(5)
    df['20d_return'] = (df['close'].shift(0) - df['close'].shift(20)) / df['close'].shift(20)
    df['avg_momentum'] = (df['1d_return'] + df['5d_return'] + df['20d_return']) / 3
    
    # Filter by Relative Volume
    df['avg_vol'] = df['volume'].rolling(window=n).mean()
    df['volume_filtered'] = df['volume'] > df['avg_vol']
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    df['daily_price_range'] = df['high'] - df['low']
    df['volume_growth'] = df['volume'] / df['avg_vol']
    df['weighted_gap'] = df['daily_price_range'] * df['volume_growth']
    df['weighted_gap'] = df['weighted_gap'].where(df['volume_filtered'], 0)
    df['sum_weighted_gaps'] = df['weighted_gap'].rolling(window=n).sum()
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['sum_weighted_gaps'] * df['avg_momentum']
    
    return df['alpha_factor']
```
