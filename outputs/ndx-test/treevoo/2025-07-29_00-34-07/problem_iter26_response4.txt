def heuristics_v2(df):
    # Calculate the relative position of the closing price within the day's range
    df['range'] = df['high'] - df['low']
    df['close_range_pos'] = (df['close'] - df['low']) / df['range']
    
    # Volume-weighted average price
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['cum_volume'] = df['volume'].rolling(window=10).sum()
    df['cum_vwap'] = df['vwap'].rolling(window=10).sum()
    df['avg_price'] = df['cum_vwap'] / df['cum_volume']
    
    # Factor: difference between close and VWAP, capturing if the stock closed above or below its volume-weighted average
    factor = df['close'] - df['avg_price']
    
    return factor
