```python
import pandas as pd

def heuristics_v2(df):
    # Compute 1, 5, and 20-day Price Returns
    df['1d_return'] = df['close'].pct_change()
    df['5d_return'] = df['close'].pct_change(5)
    df['20d_return'] = df['close'].pct_change(20)
    
    # Average the Momentum
    df['avg_momentum'] = (df['1d_return'] + df['5d_return'] + df['20d_return']) / 3
    
    # Calculate Average Volume
    n_days = 20
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    
    # Apply Volume Filter Condition
    df['filtered_days'] = df['volume'] > df['avg_volume']
    
    # For Each Filtered Day, Calculate Daily Price Range and Adjusted Price Gaps
    df['daily_price_range'] = df['high'] - df['low']
    df['price_gap'] = abs(df['close'] - df['open'])
    df['volume_growth'] = df['volume'] / df['avg_volume']
    df['adjusted_price_gap'] = df['price_gap'] * df['volume_growth']
    
    # Sum of Daily Price Ranges for Filtered Days
    filtered_daily_price_range = (df['daily_price_range'] * df['filtered_days']).sum()
    
    # Sum All Adjusted Price Gaps
    adjusted_price_gaps_sum = (df['adjusted_price_gap'] * df['filtered_days']).sum()
    
    # Generate Alpha Factor
    df['alpha_factor'] = (filtered_daily_price_range + adjusted_price_gaps_sum) * df['avg_momentum']
    
    return df['alpha_factor']
```
