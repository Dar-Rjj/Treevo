```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = high_low_range * 100 + price_change
    
    # Smoothing and Initial Momentum Calculation
    ema_5 = combined_value.ewm(span=5, adjust=False).mean()
    initial_momentum = ema_5 - ema_5.shift(5)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    price_momentum_2d = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    price_momentum_7d = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    price_momentum_14d = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Weighted Average the Momentum
    weighted_momentum = (2 * price_momentum_2d + price_momentum_7d + price_momentum_14d) / 10
    
    # Adjust for Volume Changes
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_momentum = weighted_momentum * (1 + volume_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + adjusted_momentum
    
    # Calculate Volume-Weighted Price Momentum
    close_wma_10 = df['close'].rolling(window=10).mean()
    volume_weighted_momentum = (df['close'] - close_wma_10) * daily_log_return
    
    # Final Factor Construction
    final_factor = intermediate_factor + volume_weighted_momentum
    
    return final_factor
```
