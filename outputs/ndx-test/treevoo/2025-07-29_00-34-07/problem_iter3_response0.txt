```python
import pandas as pd

def heuristics_v2(df, N=14):
    # Calculate Price Momentum
    df['Close_Change'] = df['close'].diff()
    df['Price_Momentum'] = df['Close_Change'].rolling(window=N).sum()

    # Calculate Volume Adjusted Momentum
    avg_volume = df['volume'].rolling(window=N).mean()
    df['Volume_Adjusted_Close_Change'] = (df['close'].diff()) * (df['volume'] / avg_volume)
    df['Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Close_Change'].rolling(window=N).sum()

    # Calculate True Range
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], 
                                                                        abs(x['high'] - df['close'].shift(1)), 
                                                                        abs(x['low'] - df['close'].shift(1))), axis=1)
    df['Sum_True_Range'] = df['True_Range'].rolling(window=N).sum()

    # Integrate Momentum with Volatility
    df['Momentum_Volatility_Adjusted'] = df['Volume_Adjusted_Momentum'] / df['Sum_True_Range']

    # Subtract Baseline
    baseline = df['Volume_Adjusted_Momentum'].rolling(window=N).median()
    df['Alpha_Factor'] = df['Momentum_Volatility_Adjusted'] - baseline

    return df['Alpha_Factor']
```
