```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    enhanced_range = (df['high'] - df['low'])

    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)

    # Combine Enhanced High-to-Low Range with Price Change
    combined_value = enhanced_range * 100 + price_change

    # Smoothing and Initial Momentum Calculation
    ema_7 = combined_value.ewm(span=7).mean()
    initial_momentum = ema_7 - ema_7.shift(7)

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))

    # Calculate Multi-Day Price Momentum
    price_momentum_3 = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    price_momentum_7 = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    price_momentum_21 = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)

    # Weighted Average the Momentum
    weighted_momentum = (3 * price_momentum_3 + price_momentum_7 + price_momentum_21) / 31

    # Adjust for Volume Changes
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = weighted_momentum * (1 + volume_percentage_change)

    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + final_volume_adjustment

    # Calculate Volume-Weighted Price Momentum
    close_5d_wma = df['close'].rolling(window=5).mean()
    volume_weighted_price_momentum = (df['close'] - close_5d_wma) * daily_log_return

    # Introduce High-Frequency Volatility
    true_range = np.maximum(
        df['high'] - df['low'],
        np.abs(df['high'] - df['close'].shift(1)),
        np.abs(df['low'] - df['close'].shift(1))
    )
    sma_true_range_15 = true_range.rolling(window=15).mean()

    # Adjust for High-Frequency Volatility
    volatility_adjusted_momentum = volume_weighted_price_momentum / sma_true_range_15

    # Final Factor Construction
    final_factor = intermediate_factor + volatility_adjusted_momentum

    return final_factor.dropna()
```
