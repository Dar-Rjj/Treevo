```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Gap and Volume-Weighted Gaps
    price_gap = df['open'] - df['close']
    volume_growth = df['volume'] / df['volume'].shift(1)
    weighted_price_gaps = (price_gap * volume_growth).rolling(window=2).sum()
    
    # Combine High-to-Low Range with Volume-Weighted Gaps
    combined_value = high_to_low_range * 150 + weighted_price_gaps
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = combined_value.ewm(span=7).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(7)
    
    # Calculate Multi-Day Price Momentum
    four_day_return = (df['close'] - df['close'].shift(4)) / df['close'].shift(4)
    eight_day_return = (df['close'] - df['close'].shift(8)) / df['close'].shift(8)
    twenty_day_return = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    average_momentum = (four_day_return + eight_day_return + twenty_day_return) / 3
    
    # Adjust for Volume Changes
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_momentum = average_momentum * (1 + 3 * daily_vol_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + adjusted_momentum
    
    # Calculate 20-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_differential = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].shift(1).rolling(window=15).std()
    weighted_high_low_diff = high_low_differential * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = intermediate_factor + weighted_high_low_diff - (df['close'] - df['open'])
    
    # Adjust for Volume
    final_aggregated_signal = aggregated_signal * daily_vol_change
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_return = ((df['close'] - df['open']) * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Combine Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    alpha_factor = final_aggregated_signal + volume_weighted_return
    
    # Calculate Price Momentum
    one_day_return = df['close'] / df['close'].shift(1) - 1
    seven_day_return = df['close'] / df['close'].shift(7) - 1
    thirty_five_day_return = df['close'] / df['close'].shift(35) - 1
    average_price_momentum = (one_day_return + seven_day_return + thirty_five_day_return) / 3
    
    # Final Factor Construction
    final_factor = intermediate_factor + adjusted_momentum + average_price_momentum + alpha_factor
    
    return final_factor
```
