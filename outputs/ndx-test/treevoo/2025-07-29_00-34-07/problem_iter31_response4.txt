```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Average of Price Gaps
    df['price_gap'] = df['high'] - df['low']
    df['volume_trend'] = (df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1)))
    df['weighted_price_gap'] = (df['price_gap'] * df['volume_trend']).rolling(window=5).sum()

    # Rolling High-Low Differential
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['high_low_diff'] = df['20_day_high'] - df['20_day_low']

    # Aggregate the Signals
    df['signal'] = df['high_low_diff'] + (df['close'] - df['close'].shift(10))

    # High-to-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['close'].shift(1)

    # Combine High-to-Low Range with Price Change
    df['1_day_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['combined_value'] = (df['high_low_range'] * 100) + df['1_day_return']

    # Smoothing and Initial Momentum Calculation
    df['ema_5'] = df['combined_value'].ewm(span=5, adjust=False).mean()
    df['initial_momentum'] = df['ema_5'] - df['ema_5'].shift(5)

    # Multi-Day Price Momentum
    df['7_day_return'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['14_day_return'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    df['30_day_return'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Weighted Average Momentum
    df['weighted_momentum'] = (0.4 * df['1_day_return']) + (0.3 * df['7_day_return']) + (0.3 * df['30_day_return'])

    # Final Volume Adjustment
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_adjusted_momentum'] = df['weighted_momentum'] * (1 + df['daily_volume_change'] / df['volume'].shift(1))

    # Intermediate Factor Combination
    df['intermediate_factor'] = df['initial_momentum'] + df['volume_adjusted_momentum']

    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_price_gap'] * df['intermediate_factor']) + df['volume_adjusted_momentum']

    return df['alpha_factor']
```
