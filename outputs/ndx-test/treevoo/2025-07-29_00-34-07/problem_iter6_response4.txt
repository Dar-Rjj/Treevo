import pandas as pd

def heuristics_v2(df, n_days_avg_vol=10, lookback_period=20):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Filter by Relative Volume
    df['avg_volume'] = df['volume'].rolling(window=n_days_avg_vol).mean()
    df_filtered = df[df['volume'] > df['avg_volume']]
    
    # Calculate High-Low Range Ratio
    df_filtered['high_low_ratio'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['low']
    
    # Calculate Cumulative Open-to-Close Return
    df_filtered['open_close_return'] = (df_filtered['close'] - df_filtered['open']) * df_filtered['volume']
    cumulative_open_close_return = df_filtered['open_close_return'].rolling(window=lookback_period).sum()
    
    # Calculate Cumulative High-Low Momentum
    cumulative_high_low_momentum = df_filtered['daily_price_range'].rolling(window=lookback_period).sum()
    
    # Combine Cumulative High-Low Momentum and Cumulative Open-to-Close Return
    df_filtered['alpha_factor'] = (cumulative_high_low_momentum + cumulative_open_close_return) / (cumulative_high_low_momentum - cumulative_open_close_return)
    
    return df_filtered['alpha_factor']
