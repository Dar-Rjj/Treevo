```python
def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Intraday Percentage Change
    intraday_pct_change = (df['close'] - df['open']) / df['open']
    
    # Calculate Average Intraday Volatility (Last 10 Days)
    avg_intraday_volatility = intraday_volatility.rolling(window=10).mean()
    
    # Calculate Intraday Volatility Momentum
    intraday_volatility_momentum = intraday_volatility - avg_intraday_volatility
    
    # Combine with Intraday Percentage Change
    combined_factor = intraday_volatility_momentum * intraday_pct_change
    
    # Sum over Last 5 Days
    alpha_factor = combined_factor.rolling(window=5).sum()
    
    return alpha_factor
```
