import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Weighted Average of Price Gaps
    price_gap = df['high'] - df['low']
    volume_trend = (df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))).fillna(0)
    weighted_sum = (price_gap * volume_trend).fillna(0)

    # Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    rolling_diff = (rolling_high - rolling_low).fillna(0)

    # Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std().fillna(0)

    # Aggregate Signals
    aggregate_signals = (weighted_sum * rolling_diff) - (df['close'] - df['close'].shift(10)).fillna(0)

    # High-to-Low Range
    high_low_range = (df['high'] - df['low']) / df['close'].shift(1)

    # Combined Value
    one_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_value = high_low_range * 100 + one_day_return

    # Smoothing and Initial Momentum
    ema_combined = combined_value.ewm(span=5, adjust=False).mean()
    momentum_diff = ema_combined - ema_combined.shift(5).fillna(0)

    # Multi-Day Price Momentum
    one_day_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    seven_day_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    thirty_day_momentum = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Weighted Average Momentum
    weighted_momentum = 0.4 * one_day_momentum + 0.3 * seven_day_momentum + 0.3 * thirty_day_momentum

    # Final Volume Adjustment
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_momentum = weighted_momentum * (1 + (daily_volume_change / df['volume'].shift(1))).fillna(0)

    # Intermediate Factor Combination
    intermediate_factor = momentum_diff + adjusted_momentum

    # Volume-Weighted Price Momentum
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    vw_momentum = (df['close'] - ema_close) * log_return.fillna(0)

    # Final Alpha Factor
    alpha_factor = weighted_sum * (intermediate_factor + vw_momentum)

    return alpha_factor
