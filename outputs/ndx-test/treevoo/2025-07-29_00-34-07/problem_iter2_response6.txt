```python
import pandas as pd

def heuristics_v2(df, n=50, lookback_period=30):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Filter by Relative Volume
    df['avg_vol'] = df['volume'].rolling(window=n).mean()
    df_filtered = df[df['volume'] > df['avg_vol']]
    
    # Calculate High-Low Range Ratio
    df_filtered['hl_ratio'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['low']
    
    # Calculate Cumulative Close-to-Open Return
    close_to_open_return = (df_filtered['close'] / df_filtered['open'] - 1) * df_filtered['volume']
    cumulative_close_to_open_return = close_to_open_return.rolling(window=lookback_period).sum()
    
    # Calculate Cumulative High-Low Momentum
    cumulative_hl_momentum = df_filtered['price_range'].rolling(window=lookback_period).sum()
    
    # Generate Alpha Factor based on Cumulative High-Low Momentum
    alpha_factor_high_low = cumulative_hl_momentum
    
    # Combine Cumulative High-Low Momentum and Cumulative Close-to-Open Return
    final_alpha_factor = alpha_factor_high_low + cumulative_close_to_open_return
    
    return final_alpha_factor
```
