import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series with index (date) representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the average volume over the last 10 days
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the money flow ratio
    money_flow_ratio = (typical_price * df['volume']) / (avg_volume_10d * df['close'])
    
    # Calculate the 5-day moving average of the money flow ratio
    ma_money_flow_ratio = money_flow_ratio.rolling(window=5).mean()
    
    # Calculate the factor value
    factor_value = (ma_money_flow_ratio - ma_money_flow_ratio.shift(1)) / ma_money_flow_ratio.shift(1)
    
    return factor_value
