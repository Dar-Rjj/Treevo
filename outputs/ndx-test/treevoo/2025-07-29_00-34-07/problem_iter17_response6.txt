```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    delta = df['close'].diff()
    positive_money_flow = money_flow.where(delta > 0, 0)
    negative_money_flow = money_flow.where(delta < 0, 0)
    
    # Sum the positive and negative money flow over a period
    period = 14
    positive_money_flow_sum = positive_money_flow.rolling(window=period).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=period).sum()
    
    # Calculate the money flow ratio
    money_flow_ratio = positive_money_flow_sum / negative_money_flow_sum
    
    # Calculate the money flow index
    money_flow_index = 100 - (100 / (1 + money_flow_ratio))
    
    return money_flow_index
```
