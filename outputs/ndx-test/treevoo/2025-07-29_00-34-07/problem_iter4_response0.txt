```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']

    # Calculate Average Volume over the period of n_days
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()

    # Apply Volume Filter Condition
    df_filtered = df[df['volume'] > df['avg_volume']]
    
    # Calculate Cumulative High-Low Momentum on filtered days
    df_filtered['cum_price_range'] = df_filtered['price_range'].rolling(window=n_days, min_periods=1).sum()
    
    # Generate Alpha Factor
    alpha_factor = df_filtered['cum_price_range']
    
    return alpha_factor
```
