import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow, which is the product of the typical price and the volume
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-period sum of positive and negative money flows
    positive_money_flow = money_flow[df['close'] > df['close'].shift(1)].rolling(window=14).sum()
    negative_money_flow = money_flow[df['close'] < df['close'].shift(1)].rolling(window=14).sum()
    
    # Calculate the Money Flow Ratio
    money_flow_ratio = positive_money_flow / negative_money_flow
    
    # Calculate the Money Flow Index (MFI)
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Generate a custom alpha factor
    alpha_factor = (mfi - 50) * (typical_price - df['open'])
    
    return alpha_factor
