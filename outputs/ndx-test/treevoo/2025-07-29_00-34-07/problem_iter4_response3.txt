import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap
    price_gap = df['high'] - df['low']
    
    # Calculate Volume Trend: 5-day Moving Average of Volumes and Current Day Volume Growth
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Weight the Price Gaps
    weighted_price_gaps = price_gap * volume_growth
    factor_value = weighted_price_gaps
    
    return factor_value
