```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between Open and Close
    price_gap = df['open'] - df['close']
    
    # Calculate Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_growth).sum()
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=10).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std()
    
    # Weight by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregate_signal = weighted_high_low_diff + (df['close'] - df['close'].shift(20))
    
    # Calculate Price Momentum
    momentum_1d = (df['close'] / df['close'].shift(1)) - 1
    momentum_5d = (df['close'] / df['close'].shift(5)) - 1
    momentum_30d = (df['close'] / df['close'].shift(30)) - 1
    
    # Average the Momentum
    avg_momentum = (momentum_1d + momentum_5d + momentum_30d) / 3
    
    # Adjust for Volume
    volume_change = (df['volume'] / df['volume'].shift(1)) - 1
    adjusted_aggregate_signal = aggregate_signal * volume_change
    
    # Calculate the Relationship Between Closing Prices and Trading Volume
    close_diff_volume = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Examine the Behavior of High and Low Prices for Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Analyze the Pattern of Opening Prices Relative to the Previous Day's Close
    open_pattern = (df['open'] > df['close'].shift(1)).astype(int) * 2 - 1
    
    # Investigate the Impact of the Amount Traded on Stock Performance
    amount_ratio = df['amount'] / df['amount'].rolling(window=5).mean()
    
    # Explore the Momentum in Closing Prices Over Short Periods
    short_term_momentum = (df['close'] - df['close'].shift(5)).rolling(window=5).sum()
    
    # Final Alpha Factor
    alpha_factor = (
        weighted_price_gaps +
        adjusted_aggregate_signal -
        avg_momentum +
        close_diff_volume +
        intraday_volatility +
        open_pattern +
        amount_ratio +
        short_term_momentum
    )
    
    return alpha_factor
```
