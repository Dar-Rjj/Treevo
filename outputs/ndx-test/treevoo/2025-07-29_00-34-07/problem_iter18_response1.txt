```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['open']
    
    # Combine High-to-Low Range with Price Change
    combined_value = high_low_range * 150 + price_change
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = combined_value.ewm(span=7).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(7)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    momentum_3_day = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10_day = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_30_day = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Weighted Average the Momentum
    weighted_momentum = 0.2 * momentum_3_day + 0.4 * momentum_10_day + 0.4 * momentum_30_day
    
    # Adjust for Volume Changes
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = weighted_momentum * (1 + 2.5 * volume_change)
    
    # Filter by Relative Volume
    median_volume = df['volume'].rolling(window=20).median()
    filtered_days = (df['volume'] > median_volume)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    cumulative_price_ranges = daily_price_range[filtered_days].sum()
    weighted_price_gaps = (df['open'] - df['close']).shift(1) * (1 + 2.5 * volume_change)
    sum_weighted_price_gaps = weighted_price_gaps[filtered_days].sum()
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + final_volume_adjustment
    
    # Calculate Dynamic Weighted Price Momentum
    dynamic_weighted_momentum = (df['close'] - df['close'].ewm(span=20).mean()) * daily_log_return
    
    # Final Alpha Factor Construction
    alpha_factor = (cumulative_price_ranges + sum_weighted_price_gaps) * weighted_momentum + intermediate_factor + dynamic_weighted_momentum
    
    return alpha_factor
```
