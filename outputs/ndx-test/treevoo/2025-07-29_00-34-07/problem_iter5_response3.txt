```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(10)

    # Calculate 20-day Rolling High-Low Differential
    rolling_high_20 = df['high'].rolling(window=20).max()
    rolling_low_20 = df['low'].rolling(window=20).min()
    high_low_diff_20 = rolling_high_20 - rolling_low_20

    # Calculate Volume Volatility (std of volume over the past 10 days)
    volume_volatility = df['volume'].rolling(window=10).std()

    # Weight by Volume Volatility
    weighted_high_low_diff_20 = high_low_diff_20 * volume_volatility

    # Aggregate the Signals
    aggregate_signal = weighted_high_low_diff_20 - price_diff

    # Calculate Price Momentum
    momentum_1d = (df['close'] / df['close'].shift(1) - 1)
    momentum_5d = (df['close'] / df['close'].shift(5) - 1)
    momentum_20d = (df['close'] / df['close'].shift(20) - 1)
    average_momentum = (momentum_1d + momentum_5d + momentum_20d) / 3

    # Adjust for Volume
    volume_percentage_change = df['volume'] / df['volume'].shift(1) - 1
    adjusted_aggregate_signal = aggregate_signal * volume_percentage_change

    # Calculate 30-day Rolling High-Low Range
    rolling_high_30 = df['high'].rolling(window=30).max()
    rolling_low_30 = df['low'].rolling(window=30).min()
    high_low_range_30 = rolling_high_30 - rolling_low_30

    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=10).mean()

    # Weight by Volume Trend
    weighted_high_low_range_30 = high_low_range_30 * volume_trend

    # Final Aggregate Signal
    final_signal = adjusted_aggregate_signal + weighted_high_low_range_30 - average_momentum

    return final_signal
```
