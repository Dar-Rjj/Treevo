```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=14):
    # Compute Close Price Change
    close_change = df['close'].diff()
    
    # Calculate Price Momentum
    price_momentum = close_change.rolling(window=N).sum()
    
    # Calculate Average Volume over N periods
    avg_volume = df['volume'].rolling(window=N).mean()
    
    # Compute Weighted Close Price Change
    weighted_close_change = close_change * (df['volume'] / avg_volume)
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = weighted_close_change.rolling(window=N).sum()
    
    # Calculate True Range
    true_range = pd.Series(index=df.index)
    true_range['High-Low'] = df['high'] - df['low']
    true_range['High-PrevClose'] = (df['high'] - df['close'].shift()).abs()
    true_range['Low-PrevClose'] = (df['low'] - df['close'].shift()).abs()
    true_range = true_range.max(axis=1)
    
    # Normalize Momentum by Volatility
    normalized_momentum = volume_adjusted_momentum / true_range
    
    return normalized_momentum
```
