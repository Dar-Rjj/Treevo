```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between Open and Close
    price_gap = df['open'] - df['close']
    
    # Calculate Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Weight the Price Gaps
    weighted_price_gap = (price_gap * volume_growth).sum()
    
    # Calculate Rolling High-Low Differential
    rolling_high_15 = df['high'].rolling(window=15).max()
    rolling_low_25 = df['low'].rolling(window=25).min()
    high_low_diff = rolling_high_15 - rolling_low_25
    
    # Aggregate the Signals
    long_term_price_diff = df['close'].shift(25) - df['close']
    aggregated_signal = high_low_diff + long_term_price_diff
    
    # Calculate Price Momentum
    one_day_return = (df['close'] / df['close'].shift(1)) - 1
    seven_day_return = (df['close'] / df['close'].shift(7)) - 1
    thirty_five_day_return = (df['close'] / df['close'].shift(35)) - 1
    
    # Average the Momentum
    avg_momentum = (one_day_return + seven_day_return + thirty_five_day_return) / 3
    
    # Final Alpha Factor
    alpha_factor = weighted_price_gap + aggregated_signal - avg_momentum
    
    return alpha_factor
```
