import pandas as pd

def heuristics_v2(df):
    # Weighted Average of Price Gaps
    price_gap = df['high'] - df['low']
    volume_trend = df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))
    weighted_price_gap = (price_gap * volume_trend).sum()

    # Volume-Adjusted Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    volume_volatility = df['volume'].shift(1).rolling(window=10).std()
    volume_adjusted_diff = (rolling_high - rolling_low) * volume_volatility

    # Aggregate the Signals
    aggregate_signals = volume_adjusted_diff - (df['close'] - df['close'].shift(10))

    # Enhanced High-to-Low Range and Price Change
    high_low_range = (df['high'] - df['low']) / df['close'].shift(1)
    one_day_price_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_value = high_low_range * 100 + one_day_price_return

    # Multi-Day Price Momentum
    one_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    seven_day_return = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    fourteen_day_return = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    thirty_day_return = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Weighted Average Momentum
    short_term_weight = 0.4
    medium_term_weight = 0.3
    long_term_weight = 0.3
    weighted_momentum = (short_term_weight * one_day_return) + \
                        (medium_term_weight * seven_day_return) + \
                        (long_term_weight * thirty_day_return)

    # Volume-Adjusted Momentum
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_adjusted_momentum = weighted_momentum * (1 + (daily_volume_change / df['volume'].shift(1)))

    # Final Alpha Factor
    final_alpha_factor = (weighted_price_gap * aggregate_signals) + volume_adjusted_momentum

    return final_alpha_factor
