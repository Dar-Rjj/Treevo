```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate 30-day Rolling High and Low
    rolling_high = df['high'].rolling(window=30).max()
    rolling_low = df['low'].rolling(window=30).min()
    
    # Calculate High-Low Range Differential
    high_low_differential = rolling_high - rolling_low
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Adjust High-Low Range Differential by Volume Change
    adjusted_high_low = high_low_differential * volume_change
    
    # Aggregate the Signals
    factor = adjusted_high_low - price_momentum
    
    return factor
```
