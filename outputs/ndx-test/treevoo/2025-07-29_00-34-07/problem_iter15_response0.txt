```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, lookback=5):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Adjusted Relative Volume
    df['vol_ema'] = df['volume'].ewm(span=n, adjust=False).mean()
    df['rel_vol'] = df['volume'] / df['vol_ema']
    
    # Apply Volume Filter Condition
    df_filtered = df[df['rel_vol'] > 2.5]
    
    # Calculate High-Low Range Ratio
    df_filtered['high_low_ratio'] = (df_filtered['high'] - df_filtered['low']) / df_filtered['low']
    
    # Calculate Volume-Adjusted Close-to-Open Return
    df_filtered['vol_adj_ret'] = (df_filtered['close'] - df_filtered['open']) * df_filtered['volume']
    df_filtered['cum_vol_adj_ret'] = df_filtered['vol_adj_ret'].rolling(window=lookback).sum()
    
    # Calculate Cumulative High-Low Momentum
    df_filtered['cum_price_range'] = df_filtered['price_range'].rolling(window=lookback).sum()
    
    # Generate Final Alpha Factor
    df_filtered['alpha_factor'] = df_filtered['cum_price_range'] + df_filtered['cum_vol_adj_ret']
    
    return df_filtered['alpha_factor']
```
