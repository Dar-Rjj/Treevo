import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(15)
    
    # Calculate 25-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    high_low_differential = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_differential = high_low_differential * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = weighted_high_low_differential - price_diff
    
    # Calculate Averaged Price Momentum
    price_return_2day = (df['close'] / df['close'].shift(2)) - 1
    price_return_7day = (df['close'] / df['close'].shift(7)) - 1
    price_return_30day = (df['close'] / df['close'].shift(30)) - 1
    avg_price_momentum = (price_return_2day + price_return_7day + price_return_30day) / 3
    
    # Adjust for Volume
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_factor = aggregated_signal * daily_volume_change
    
    # Return the factor values
    return final_factor
