```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Filter by Relative Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df_filtered = df[df['volume'] > df['avg_volume_20']]
    
    # Calculate Price Momentum
    df_filtered['1d_return'] = (df_filtered['close'] / df_filtered['close'].shift(1)) - 1
    df_filtered['5d_return'] = (df_filtered['close'] / df_filtered['close'].shift(5)) - 1
    df_filtered['20d_return'] = (df_filtered['close'] / df_filtered['close'].shift(20)) - 1
    
    # Average the Momentum
    df_filtered['avg_momentum'] = (df_filtered['1d_return'] + df_filtered['5d_return'] + df_filtered['20d_return']) / 3
    
    # Adjust for Volume
    df_filtered['volume_pct_change'] = (df_filtered['volume'] / df_filtered['volume'].shift(1)) - 1
    df_filtered['adjusted_momentum'] = df_filtered['avg_momentum'] * df_filtered['volume_pct_change']
    
    # Generate Alpha Factor
    cumulative_high_low_momentum = df_filtered['price_range'].sum()
    alpha_factor = 0.7 * df_filtered['adjusted_momentum'] + 0.3 * (cumulative_high_low_momentum / len(df_filtered))
    
    return alpha_factor
```
