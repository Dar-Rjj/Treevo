```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate Volume Trend
    volume_5d_ma = df['volume'].rolling(window=5).mean()
    volume_growth = df['volume'] / df['volume'].shift(1)
    volume_trend = volume_5d_ma * volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-to-Low Range with Weighted Price Gaps
    combined_value = high_low_range * 100 + weighted_price_gaps
    
    # Smoothing and Momentum Calculation
    smoothed_value = combined_value.rolling(window=5).mean()
    momentum = smoothed_value - smoothed_value.shift(5)
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Final Alpha Factor
    alpha_factor = weighted_price_gaps + momentum + close_to_open_return
    
    return alpha_factor
```
