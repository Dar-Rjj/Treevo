```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate Volume Trend
    vol_5day_ma = df['volume'].rolling(window=5).mean()
    vol_growth = df['volume'] / df['volume'].shift(1)
    volume_trend = vol_5day_ma * vol_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Calculate High-to-Low Range
    high_low_range = (df['high'] - df['low']) * 100
    
    # Combine High-to-Low Range with Weighted Price Gaps
    combined_value = high_low_range + weighted_price_gaps
    
    # Smoothing and Momentum Calculation
    smoothed_combined_value = combined_value.rolling(window=5).mean()
    hl_momentum = smoothed_combined_value - smoothed_combined_value.shift(5)
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps * hl_momentum) - price_momentum
    
    return alpha_factor
```
