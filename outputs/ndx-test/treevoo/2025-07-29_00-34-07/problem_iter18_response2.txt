```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(20)
    
    # Calculate 30-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=30).max()
    rolling_low = df['low'].rolling(window=30).min()
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility (Standard Deviation of volume over the past 20 days)
    volume_volatility = df['volume'].rolling(window=20).std()
    
    # Weight 30-day Rolling High-Low Differential by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = weighted_high_low_diff - price_diff
    
    # Calculate Averaged Price Momentum
    price_return_3d = (df['close'] / df['close'].shift(3)) - 1
    price_return_10d = (df['close'] / df['close'].shift(10)) - 1
    price_return_40d = (df['close'] / df['close'].shift(40)) - 1
    averaged_price_momentum = (price_return_3d + price_return_10d + price_return_40d) / 3
    
    # Calculate Daily Volume Percentage Change
    volume_change = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Adjust for Volume
    final_factor = (aggregated_signal + averaged_price_momentum) * volume_change
    
    return final_factor
```
