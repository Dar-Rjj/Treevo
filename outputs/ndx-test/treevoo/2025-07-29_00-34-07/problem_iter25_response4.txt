    # Estimate true range
    df['true_range'] = df.apply(
        lambda row: max(row['high'] - row['low'], abs(row['high'] - row['close'].shift(1)), abs(row['low'] - row['close'].shift(1))),
        axis=1
    )

    # Compute average true range (ATR) over a rolling window of 14 days
    df['atr_14d'] = df['true_range'].rolling(window=14).mean()

    # Determine the volume change from the previous day
    df['volume_change'] = df['volume'].pct_change()

    # Analyze the ratio of today's volume to the average volume over 20 days
    df['volume_ratio_20d'] = df['volume'] / df['volume'].rolling(window=20).mean()

    # Calculate the total transaction amount for the day
    df['daily_transaction_amount'] = df['amount']

    # Compute the ratio of today's transaction amount to the average amount over 30 days
    df['transaction_amount_ratio_30d'] = df['daily_transaction_amount'] / df['daily_transaction_amount'].rolling(window=30).mean()

    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (
        df['momentum_10d'] + 
        df['overnight_return'] + 
        df['atr_14d'] + 
        df['volume_ratio_20d'] + 
        df['transaction_amount_ratio_30d']
    ) / 5  # Normalize by the number of factors

    return df['alpha_factor']
