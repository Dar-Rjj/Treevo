```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Percentage Change
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Weighted Average of Price Gaps
    df['price_gap'] = df['high'] - df['low']
    df['volume_trend'] = df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))
    df['weighted_price_gap'] = (df['price_gap'] * df['volume_trend']).rolling(window=5).sum()
    
    # Calculate Average Intraday Range
    df['avg_intraday_range_10d'] = df['intraday_range'].rolling(window=10).mean()
    df['intraday_range_momentum'] = df['intraday_range'] - df['avg_intraday_range_10d']
    
    # Calculate Volume-Weighted Intraday Percentage Change
    df['vol_weighted_intraday_pct_change'] = (df['intraday_pct_change'] * df['volume']).rolling(window=5).mean()
    
    # Combine with Intraday Range Momentum
    df['combined_momentum'] = df['intraday_range_momentum'] * df['vol_weighted_intraday_pct_change']
    df['sum_combined_momentum_5d'] = df['combined_momentum'].rolling(window=5).sum()
    
    # Combine High-to-Low Range with Weighted Price Gaps
    df['combined_high_low_range'] = 100 * df['intraday_range'] + df['weighted_price_gap']
    
    # Smoothing and Momentum Calculation
    df['sma_5d_combined'] = df['combined_high_low_range'].rolling(window=5).mean()
    df['smoothed_momentum'] = df['sma_5d_combined'] - df['sma_5d_combined'].shift(5)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_high_low_range'] + df['sum_combined_momentum_5d']
    
    return df['alpha_factor'].dropna()
```
