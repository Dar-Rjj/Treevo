```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate Volume Trend
    volume_ma5 = df['volume'].rolling(window=5).mean()
    volume_growth = df['volume'] / df['volume'].shift(1)
    volume_trend = volume_ma5 * volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Calculate Rolling High-Low Differential
    rolling_high_20 = df['high'].rolling(window=20).max()
    rolling_low_20 = df['low'].rolling(window=20).min()
    high_low_differential = rolling_high_20 - rolling_low_20
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std().shift(1)
    
    # Weight by Volume Volatility
    weighted_high_low_diff = high_low_differential * volume_volatility
    
    # Aggregate the Signals
    close_t_minus_10 = df['close'].shift(10)
    price_difference = df['close'] - close_t_minus_10
    aggregated_signal = weighted_high_low_diff + price_difference
    
    # Calculate High-to-Low Range
    high_low_range = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    one_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_value = high_low_range * 100 + one_day_return
    
    # Smoothing and Initial Momentum Calculation
    ema_combined_value = combined_value.ewm(span=5, adjust=False).mean()
    momentum_difference = ema_combined_value - ema_combined_value.shift(5)
    
    # Calculate Multi-Day Price Momentum
    one_day_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    seven_day_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    thirty_day_momentum = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Weighted Average Momentum
    short_term_weight = 0.4
    medium_term_weight = 0.3
    long_term_weight = 0.3
    weighted_momentum = (short_term_weight * one_day_momentum + 
                         medium_term_weight * seven_day_momentum + 
                         long_term_weight * thirty_day_momentum)
    
    # Adjust for Volume Changes
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    final_volume_adjustment = weighted_momentum * (1 + (daily_volume_change / df['volume'].shift(1)))
    
    # Integrate Transaction Amount
    daily_amount_change = df['amount'] - df['amount'].shift(1)
    intermediate_volume_adjustment = final_volume_adjustment * (1 + (daily_amount_change / df['amount'].shift(1)))
    
    # Intermediate Factor Combination
    intermediate_factor = momentum_difference + intermediate_volume_adjustment
    
    # Calculate Volume-Weighted Price Momentum
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    volume_weighted_momentum = (df['close'] - ema_close_10) * log_return
    
    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps + 
                    intermediate_factor + 
                    volume_weighted_momentum)
    
    return alpha_factor
```
