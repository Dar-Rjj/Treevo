```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Weighted Average of Price Gaps
    df['PriceGap'] = df['high'] - df['low']
    df['VolumeGrowth'] = df['volume'] / df['volume'].shift(1)
    df['WeightedPriceGap'] = df['PriceGap'] * df['VolumeGrowth']
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=10).min()
    df['HighLowDifferential'] = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std()
    df['WeightedHighLowDifferential'] = df['HighLowDifferential'] * volume_volatility
    
    # Aggregate the Signals
    df['AggregateSignal'] = df['WeightedHighLowDifferential'] - (df['close'] - df['close'].shift(20))
    
    # Calculate High-to-Low Range
    df['HighLowRange'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    price_return_1d = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['CombinedValue'] = df['HighLowRange'] * 100 + price_return_1d
    
    # Smoothing and Initial Momentum Calculation
    ema_5 = df['CombinedValue'].ewm(span=5, adjust=False).mean()
    df['MomentumDifference'] = ema_5 - ema_5.shift(5)
    
    # Calculate Multi-Day Price Momentum
    price_return_1d = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_return_7d = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    price_return_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Weighted Average Momentum
    weighted_momentum = 0.4 * price_return_1d + 0.3 * price_return_7d + 0.3 * price_return_30d
    
    # Adjust for Volume Changes
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    final_volume_adjustment = weighted_momentum * (1 + (daily_volume_change / df['volume'].shift(1)))
    
    # Integrate Transaction Amount
    daily_amount_change = df['amount'] - df['amount'].shift(1)
    intermediate_volume_adjustment = final_volume_adjustment * (1 + (daily_amount_change / df['amount'].shift(1)))
    
    # Intermediate Factor Combination
    intermediate_factor = df['MomentumDifference'] + intermediate_volume_adjustment
    
    # Calculate Volume-Weighted Price Momentum
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    volume_weighted_momentum = (df['close'] - ema_close_10) * (np.log(df['close']) - np.log(df['close'].shift(1)))
    
    # Final Alpha Factor
    alpha_factor = df['WeightedPriceGap'] + intermediate_factor + volume_weighted_momentum
    
    return alpha_factor
```
