def heuristics_v2(df):
    import pandas as pd
    
    # Calculate price change from Open to Close
    price_change = df['close'] - df['open']
    
    # Determine the range of daily price movement
    price_range = df['high'] - df['low']
    
    # Evaluate the upper shadow length
    upper_shadow = df['high'] - df[['open', 'close']].max(axis=1)
    
    # Evaluate the lower shadow length
    lower_shadow = df[['open', 'close']].min(axis=1) - df['low']
    
    # Calculate the ratio of current day's volume to the previous day's volume
    volume_ratio_prev = df['volume'] / df['volume'].shift(1)
    
    # Calculate the ratio of current day's volume to the average volume over the last 5 days
    volume_ratio_avg5 = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of amount to volume, representing the average transaction size
    avg_transaction_size = df['amount'] / df['volume']
    
    # Calculate the ratio of current day's amount to the previous day's amount
    amount_ratio_prev = df['amount'] / df['amount'].shift(1)
    
    # Create a weighted sum of price change, range, and volume ratio
    composite_factor_1 = (df['close'] - df['open']) + (df['high'] - df['low']) + (df['volume'] / df['volume'].shift(1))
    
    # Create an interaction term between price range and volume ratio
    composite_factor_2 = (df['high'] - df['low']) * (df['volume'] / df['volume'].rolling(window=5).mean())
    
    # Calculate the percentage change in Close price from the previous day
    momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate the simple moving average of the Close price over the last 10 days
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Combine all factors into a single DataFrame
    factors = pd.DataFrame({
        'price_change': price_change,
        'price_range': price_range,
        'upper_shadow': upper_shadow,
        'lower_shadow': lower_shadow,
        'volume_ratio_prev': volume_ratio_prev,
        'volume_ratio_avg5': volume_ratio_avg5,
        'avg_transaction_size': avg_transaction_size,
        'amount_ratio_prev': amount_ratio_prev,
        'composite_factor_1': composite_factor_1,
        'composite_factor_2': composite_factor_2,
        'momentum': momentum,
        'sma_10': sma_10
    }, index=df.index)
    
    # Return the combined alpha factor
    return factors.sum(axis=1)
