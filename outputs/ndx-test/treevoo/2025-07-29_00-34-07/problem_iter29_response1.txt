```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'].diff(15)
    
    # Calculate 20-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low
    
    # Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std()
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Daily Volume Percentage Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = weighted_high_low_diff * (1 + 2 * daily_volume_change)
    
    # Aggregate the Signals
    aggregated_signals = final_volume_adjustment - price_diff
    
    # Averaged Momentum
    momentum_1d = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    momentum_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    averaged_momentum = (momentum_1d + momentum_5d + momentum_30d) / 3
    
    # Smoothing and Initial Momentum Calculation
    combined_value = aggregated_signals.rolling(window=7).mean()
    smoothed_initial_momentum = combined_value - combined_value.shift(7)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    summed_daily_ranges = daily_price_range.rolling(window=7).sum()
    weighted_price_gaps = (daily_price_range * (df['volume'] / df['volume'].shift(1))).rolling(window=7).sum()
    
    # Intermediate Factor Combination
    intermediate_factor = smoothed_initial_momentum + averaged_momentum
    
    # Weighted Price Momentum
    log_return = np.log(df['close']) - np.log(df['close'].rolling(window=20).mean())
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=20).mean()) * log_return
    
    # Adjust for Volume
    adjusted_weighted_price_gaps = weighted_price_gaps * (df['volume'] / df['volume'].shift(1))
    
    # High-to-Low Range and Momentum
    high_low_range = df['high'] - df['low']
    combined_value = 100 * high_low_range + adjusted_weighted_price_gaps
    smoothed_high_low_momentum = combined_value.rolling(window=5).mean() - combined_value.shift(5)
    
    # Final Alpha Factor Construction
    alpha_factor = (
        summed_daily_ranges + adjusted_weighted_price_gaps +
        intermediate_factor + weighted_price_momentum + smoothed_high_low_momentum +
        final_volume_adjustment - averaged_momentum
    )

    return alpha_factor.dropna()
```
