```python
def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Intraday Percentage Change
    intraday_percentage_change = (data['close'] - data['open']) / data['open']
    
    # Calculate Average Intraday Range for the last 10 days
    avg_intraday_range = intraday_range.rolling(window=10).mean()
    
    # Calculate Intraday Range Momentum
    intraday_range_momentum = intraday_range - avg_intraday_range
    
    # Calculate Volume-Weighted Intraday Percentage Change
    volume_weighted_intraday_change = (intraday_percentage_change * data['volume']).rolling(window=10).mean()
    
    # Combine with Intraday Range Momentum
    combined_factor = (intraday_range_momentum * volume_weighted_intraday_change).rolling(window=5).sum()
    
    return combined_factor
```
