```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    enhanced_range = df['high'] - df['low']

    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)

    # Combine Enhanced High-to-Low Range with Price Change
    combined_value = enhanced_range * 125 + price_change

    # Smoothing and Initial Momentum Calculation
    smoothed_momentum = combined_value.ewm(span=7).mean()
    initial_momentum = smoothed_momentum - smoothed_momentum.shift(7)

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))

    # Calculate Multi-Day Price Momentum
    momentum_3_day = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10_day = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_21_day = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)

    # Weighted Average the Momentum
    weighted_momentum = (3 * momentum_3_day + momentum_10_day + momentum_21_day) / 34

    # Adjust for Volume Changes
    daily_vol_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = weighted_momentum * (1 + 1.5 * daily_vol_change)

    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + final_volume_adjustment

    # Calculate 25-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    high_low_differential = rolling_high - rolling_low

    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_diff = high_low_differential * volume_volatility

    # Aggregate the Signals
    aggregated_signal = intermediate_factor + weighted_high_low_diff - df['close'].diff()

    # Adjust for Volume
    adjusted_aggregated_signal = aggregated_signal * daily_vol_change

    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) * df['volume']
    volume_weighted_close_to_open = close_to_open_return.rolling(window=25).sum()

    # Combine Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    cumulative_high_low_momentum = weighted_high_low_diff.rolling(window=25).sum()
    alpha_factor = cumulative_high_low_momentum + volume_weighted_close_to_open

    # Introduce High-Frequency Volatility
    true_range = np.max([df['high'] - df['low'], abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))], axis=0)
    true_range_sma = true_range.rolling(window=15).mean()

    # Adjust for High-Frequency Volatility
    adjusted_volume_weighted_momentum = volume_weighted_close_to_open / true_range_sma

    # Final Factor Construction
    final_factor = (intermediate_factor 
                    + adjusted_volume_weighted_momentum 
                    + alpha_factor 
                    + initial_momentum / true_range_sma)

    return final_factor
```
