```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Difference
    adjusted_price_diff = (df['close'] - df['close'].shift(10)) * np.log(df['close'] / df['close'].shift(10))
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_differential = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    weighted_high_low_diff = high_low_differential * volume_volatility
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = (df['volume'] - df['volume'].shift(1)) * (df['close'] - df['close'].rolling(window=10).mean())
    
    # Aggregate the Signals
    aggregated_signal = weighted_high_low_diff - adjusted_price_diff + volume_adjusted_momentum
    
    # Final Factor
    final_factor = aggregated_signal / df['volume']
    
    return final_factor
```
