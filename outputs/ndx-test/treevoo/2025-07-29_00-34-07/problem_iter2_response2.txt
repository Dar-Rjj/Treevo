```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Trend
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_growth'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend'] = df['volume_ma_5'] * df['volume_growth']
    
    # Filter by Relative Volume
    n_days = 10  # Number of days to calculate average volume
    df['average_volume'] = df['volume'].rolling(window=n_days).mean()
    df_filtered = df[df['volume'] > df['average_volume']]
    
    # Calculate Cumulative Weighted High-Low Momentum
    df_filtered['weighted_price_range'] = df_filtered['price_range'] * df_filtered['volume_trend']
    df['cumulative_weighted_momentum'] = df_filtered['weighted_price_range'].expanding().sum()
    
    return df['cumulative_weighted_momentum']
```
