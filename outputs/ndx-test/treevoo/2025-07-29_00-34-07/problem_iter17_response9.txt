```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Averaged Price Momentum
    one_day_return = df['close'].pct_change(1)
    five_day_return = df['close'].pct_change(5)
    twenty_day_return = df['close'].pct_change(20)
    avg_momentum = (one_day_return + five_day_return + twenty_day_return) / 3

    # Determine if Today's Volume > Average Volume
    avg_vol = df['volume'].rolling(window=20).mean()
    vol_filter = df['volume'] > avg_vol
    
    # Adjusted High-Low Cumulative Return and Price Gaps
    daily_price_range = (df['high'] - df['low']).loc[vol_filter]
    sum_daily_price_ranges = daily_price_range.sum()

    price_gap = (df['open'] - df['close']).abs().loc[vol_filter]
    volume_growth_factor = df['volume'][vol_filter] / avg_vol[vol_filter]
    adjusted_price_gaps = (price_gap * volume_growth_factor).sum()

    # Calculate 25-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    high_low_diff = rolling_high - rolling_low

    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_diff = high_low_diff * volume_volatility

    # Aggregate the Signals
    close_t_minus_15 = df['close'].shift(15)
    signal = (weighted_high_low_diff + sum_daily_price_ranges + adjusted_price_gaps 
              - (df['close'] - close_t_minus_15))

    # Adjust for Volume
    daily_vol_change = df['volume'].pct_change(1)
    final_signal = signal * daily_vol_change

    return final_signal
```
