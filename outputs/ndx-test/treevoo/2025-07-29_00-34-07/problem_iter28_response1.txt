```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 15-Day Price Momentum
    df['15DayMomentum'] = df['close'] - df['close'].shift(15)
    
    # Calculate 20-Day Rolling High-Low Differential
    df['20DayHigh'] = df['high'].rolling(window=20).max()
    df['20DayLow'] = df['low'].rolling(window=20).min()
    df['HighLowDifferential'] = df['20DayHigh'] - df['20DayLow']
    
    # Volume-Weight the High-Low Differential
    df['VolumeVolatility'] = df['volume'].rolling(window=10).std()
    df['VolumeWeightedDiff'] = df['HighLowDifferential'] * df['VolumeVolatility']
    
    # Adjust for Volume Changes
    df['DailyVolumeChange'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['VolumeAdjustedDiff'] = df['VolumeWeightedDiff'] * (1 + 2 * df['DailyVolumeChange'])
    
    # Calculate Average Price Momentum
    df['2DayReturn'] = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    df['7DayReturn'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['30DayReturn'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    df['AveragePriceMomentum'] = (df['2DayReturn'] + df['7DayReturn'] + df['30DayReturn']) / 3
    
    # Smooth and Adjust Initial Momentum
    df['SMA7'] = df['15DayMomentum'].rolling(window=7).mean()
    df['SmoothedAdjustedMomentum'] = df['SMA7'] - df['SMA7'].shift(7)
    
    # Calculate Weighted Price Gaps
    df['PriceGap'] = df['high'] - df['low']
    df['VolumeTrend'] = df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))
    df['WeightedPriceGaps'] = (df['PriceGap'] * df['VolumeTrend']).sum()
    
    # Combine Intermediate Factors
    df['CombinedFactors'] = df['SmoothedAdjustedMomentum'] + df['AveragePriceMomentum']
    
    # Final Alpha Factor Construction
    df['WeightedPriceMomentum'] = (df['close'] - df['close'].rolling(window=15).mean()) * np.log(df['close'] / df['close'].shift(1))
    df['AlphaFactor'] = df['CombinedFactors'] + df['WeightedPriceGaps'] * df['AveragePriceMomentum'] + df['WeightedPriceMomentum']
    
    return df['AlphaFactor']
```
