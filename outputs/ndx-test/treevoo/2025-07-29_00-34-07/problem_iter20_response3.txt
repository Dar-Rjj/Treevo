```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(15)
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    high_low_diff = rolling_high - rolling_low
    
    # Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Daily Volume Percentage Change
    daily_volume_pct_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = weighted_high_low_diff * (1 + 2 * daily_volume_pct_change)
    
    # Aggregate the Signals
    aggregated_signal = final_volume_adjustment - price_diff
    
    # Calculate Price Momentum
    momentum_2d = df['close'] / df['close'].shift(2) - 1
    momentum_7d = df['close'] / df['close'].shift(7) - 1
    momentum_30d = df['close'] / df['close'].shift(30) - 1
    average_momentum = (momentum_2d + momentum_7d + momentum_30d) / 3
    
    # Smoothing and Initial Momentum Calculation
    smoothed_momentum = average_momentum.rolling(window=7).mean()
    initial_momentum_diff = smoothed_momentum - smoothed_momentum.shift(7)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    filtered_days = (df['volume'] > df['volume'].mean()).astype(int)
    sum_daily_price_ranges = (daily_price_range * filtered_days).sum()
    weighted_price_gaps = ((df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1))).sum()
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum_diff + average_momentum
    
    # Calculate Weighted Price Momentum
    log_return = np.log(df['close'] / df['close'].shift(1))
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=15).mean()) * log_return
    
    # Enhanced Volume-Filtered High-Low Range
    high_low_range = rolling_high - rolling_low
    filtered_high_low_range = (high_low_range * (df['volume'] > df['volume'].mean())).sum()
    
    # Final Alpha Factor Construction
    alpha_factor = (
        sum_daily_price_ranges + weighted_price_gaps
    ) * average_momentum + intermediate_factor + weighted_price_momentum + filtered_high_low_range
    
    return alpha_factor
```
