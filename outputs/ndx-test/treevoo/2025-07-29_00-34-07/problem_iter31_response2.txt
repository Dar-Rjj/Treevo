```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-to-Low Range
    df['EnhancedHighLowRange'] = (df['high'] - df['low']) * 100
    
    # Combine with Price Change
    df['PriceChange'] = df['close'] - df['open'].shift(1)
    df['CombinedValue'] = df['PriceChange'] + df['EnhancedHighLowRange']
    
    # Smoothing and Initial Momentum
    df['EMA7D'] = df['CombinedValue'].ewm(span=7).mean()
    df['InitialMomentum'] = df['EMA7D'] - df['EMA7D'].shift(7)
    
    # Multi-Day Price Momentum
    df['3DReturn'] = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    df['7DReturn'] = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    df['21DReturn'] = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    
    # Weighted Average Momentum
    df['WeightedMomentum'] = (3 * df['3DReturn'] + df['7DReturn'] + df['21DReturn']) / 31
    
    # Adjust for Volume Changes
    df['DailyVolumePctChange'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['AdjustedWeightedMomentum'] = df['WeightedMomentum'] * (1 + df['DailyVolumePctChange'])
    
    # Volume-Weighted Price Momentum
    df['5DWMA'] = df['close'].rolling(window=5).mean()
    df['LogReturn'] = np.log(df['close'] / df['close'].shift(1))
    df['VolumeWeightedPM'] = (df['close'] - df['5DWMA']) * df['LogReturn']
    
    # Introduce High-Frequency Volatility
    df['TrueRange'] = df[['high' - 'low', (df['high'] - df['close'].shift(1)).abs(), (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    df['HFVolatility'] = df['TrueRange'].rolling(window=15).mean()
    
    # Adjust for HF Volatility
    df['HFVolatilityAdjVWPM'] = df['VolumeWeightedPM'] / df['HFVolatility']
    
    # Final Factor
    df['FinalFactor'] = df['InitialMomentum'] + df['AdjustedWeightedMomentum'] + df['HFVolatilityAdjVWPM']
    
    return df['FinalFactor']
```
