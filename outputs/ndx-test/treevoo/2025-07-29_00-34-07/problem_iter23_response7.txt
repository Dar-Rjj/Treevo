```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # High-to-Low Range and Rolling High-Low Differential
    df['Rolling_High_25'] = df['high'].rolling(window=25).max()
    df['Rolling_Low_25'] = df['low'].rolling(window=25).min()
    df['High_Low_Range'] = (df['high'] - df['low']) / df['low']
    df['Rolling_High_Low_Diff'] = df['Rolling_High_25'] - df['Rolling_Low_25']

    # Price Difference and Momentum Calculation
    df['Price_Change'] = df['close'] - df['open'].shift(1)
    df['2d_Return'] = df['close'].pct_change(periods=2)
    df['7d_Return'] = df['close'].pct_change(periods=7)
    df['30d_Return'] = df['close'].pct_change(periods=30)

    # Volume Adjustments and Volume Volatility
    df['Volume_Pct_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Volatility'] = df['volume'].rolling(window=15).std()
    df['Final_Volume_Adjustment'] = (df['High_Low_Range'] + df['Rolling_High_Low_Diff']) * (1 + 2 * df['Volume_Pct_Change']) * df['Volume_Volatility']

    # Smoothing and Initial Momentum Calculation
    df['Combined_Value'] = 150 * df['High_Low_Range'] + df['Price_Change'] + df['Rolling_High_Low_Diff']
    df['SMA_Combined_Value_7'] = df['Combined_Value'].rolling(window=7).mean()
    df['SMA_Combined_Value_Diff'] = df['SMA_Combined_Value_7'] - df['SMA_Combined_Value_7'].shift(7)

    # Averaged Price Momentum and High-Low Range Sum
    df['Averaged_Momentum'] = (df['2d_Return'] + df['7d_Return'] + df['30d_Return']) / 3
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['Sum_Daily_Price_Ranges'] = df['Daily_Price_Range'].rolling(window=15).sum()
    df['Weighted_Price_Gaps'] = (df['high'] - df['low']) * df['Volume_Pct_Change']
    df['Sum_Weighted_Price_Gaps'] = df['Weighted_Price_Gaps'].rolling(window=15).sum()

    # Intermediate Factor Combination
    df['Intermediate_Factor'] = df['SMA_Combined_Value_Diff'] + df['Averaged_Momentum'] + df['Sum_Weighted_Price_Gaps']

    # Weighted Price Momentum
    df['15d_MA_Close'] = df['close'].rolling(window=15).mean()
    df['Weighted_Price_Momentum'] = (df['close'] - df['15d_MA_Close']) * np.log(df['close'])

    # Final Alpha Factor Construction
    df['Final_Factor'] = (df['Sum_Daily_Price_Ranges'] + df['Sum_Weighted_Price_Gaps']) * df['Averaged_Momentum'] + df['Intermediate_Factor'] + df['Weighted_Price_Momentum'] + df['Final_Volume_Adjustment']
    
    return df['Final_Factor']
```
