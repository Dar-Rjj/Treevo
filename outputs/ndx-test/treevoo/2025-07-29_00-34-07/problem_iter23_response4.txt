```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low for the day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the close and open price
    price_diff = df['close'] - df['open']
    
    # Calculate the volume change compared to the previous day
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine the average price, price difference, and volume change into a single factor
    alpha_factor = (avg_price * price_diff) + volume_change
    
    return alpha_factor
```
