```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Price Momentum
    one_day_return = df['close'].pct_change(1)
    five_day_return = df['close'].pct_change(5)
    twenty_day_return = df['close'].pct_change(20)
    weighted_momentum = (one_day_return * 0.6) + (five_day_return * 0.3) + (twenty_day_return * 0.1)

    # Filter by Relative Volume
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    volume_filter = df['volume'] > avg_vol_20
    filtered_days = df[volume_filter]

    # Adjust for Volume and Amount
    daily_vol_change = df['volume'].pct_change()
    adjusted_momentum = weighted_momentum * daily_vol_change
    daily_amount_change = df['amount'].pct_change()
    adjusted_momentum *= daily_amount_change

    # Incorporate High-Low Differential and Volume Volatility
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low
    vol_volatility = df['volume'].rolling(window=10).std()
    adjusted_high_low = high_low_diff * vol_volatility

    # Calculate Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    sum_filtered_price_ranges = daily_price_range[volume_filter].sum()
    weighted_price_gaps = (daily_price_range * daily_vol_change).sum()

    # Combine Momentum and Volatility Components
    combined_signal = adjusted_momentum + adjusted_high_low

    # Generate Alpha Factor
    final_alpha = (combined_signal + sum_filtered_price_ranges + weighted_price_gaps) / df['volume']
    final_alpha *= daily_amount_change
    
    return final_alpha
```
