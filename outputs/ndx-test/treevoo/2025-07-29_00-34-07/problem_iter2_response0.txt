```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(10)
    
    # Calculate 20-day Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    
    # Weight by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregated_signal = weighted_high_low_diff - price_diff
    
    # Calculate Price Momentum
    price_momentum_1d = (df['close'] / df['close'].shift(1)) - 1
    price_momentum_5d = (df['close'] / df['close'].shift(5)) - 1
    price_momentum_20d = (df['close'] / df['close'].shift(20)) - 1
    
    # Average the Momentum
    average_momentum = (price_momentum_1d + price_momentum_5d + price_momentum_20d) / 3
    
    # Calculate Daily Volume Percentage Change
    volume_change = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Adjust for Volume
    final_factor = aggregated_signal * volume_change + average_momentum
    
    return final_factor
```
