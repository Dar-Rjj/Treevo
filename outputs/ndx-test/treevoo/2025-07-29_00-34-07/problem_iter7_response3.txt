```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate 5-day Moving Average of Volumes
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Current Day's Volume Growth
    df['volume_growth'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume_ma_5'] * df['volume_growth']
    
    # Calculate Average Volume over a period (e.g., 30 days)
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    
    # Apply Volume Filter Condition
    df_filtered = df[df['volume'] > df['avg_volume_30']].copy()
    
    # Multiply Each Price Range by Corresponding Volume Trend
    df_filtered['weighted_price_range'] = df_filtered['price_range'] * df_filtered['volume_trend']
    
    # Sum of Weighted Price Ranges for Filtered Days
    alpha_factor = df_filtered.groupby(level=0)['weighted_price_range'].sum()
    
    return alpha_factor
```
