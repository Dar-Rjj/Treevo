```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']

    # Calculate Price Change
    price_change = df['close'].shift(-1) - df['open']

    # Combine High-to-Low Range with Price Change
    combined_value = (high_to_low_range * 150) + price_change

    # Smoothing and Initial Momentum Calculation
    smoothed_value = combined_value.rolling(window=7).mean()
    initial_momentum = smoothed_value - smoothed_value.shift(7)

    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))

    # Calculate Multi-Day Price Momentum
    momentum_3d = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10d = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Average the Momentum
    average_momentum = (momentum_3d + momentum_10d + momentum_30d) / 3

    # Adjust for Volume Changes
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_momentum = average_momentum * (1 + 2 * daily_volume_change)

    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + volume_adjusted_momentum

    # Calculate 25-day Rolling High-Low Differential
    rolling_high_25 = df['high'].rolling(window=25).max()
    rolling_low_25 = df['low'].rolling(window=25).min()
    high_low_differential = rolling_high_25 - rolling_low_25

    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=15).std()
    weighted_high_low_differential = high_low_differential * volume_volatility

    # Aggregate the Signals
    aggregated_signal = weighted_high_low_differential - (df['close'] - df['close'].shift(1))

    # Adjust for Volume
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_signal = aggregated_signal * volume_percentage_change

    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_c2o = (df['close'] - df['open']) * df['volume']

    # Combine Cumulative High-Low Momentum and Volume-Weighted Close-to-Open Return
    cumulative_hl_momentum = high_low_differential.rolling(window=25).sum()
    alpha_factor = cumulative_hl_momentum + volume_weighted_c2o.rolling(window=25).sum()

    # Calculate Weighted Price Momentum
    close_ma_15 = df['close'].rolling(window=15).mean()
    weighted_price_momentum = (df['close'] - close_ma_15) * daily_log_return

    # Final Factor Construction
    final_factor = intermediate_factor + volume_adjusted_signal + weighted_price_momentum + alpha_factor

    return final_factor
```
