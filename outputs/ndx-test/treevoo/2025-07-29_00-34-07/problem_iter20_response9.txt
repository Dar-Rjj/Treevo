import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the close price over 10 days
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of the close price over 10 days
    df['close_10d_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the Z-score for the close price
    df['close_z_score'] = (df['close'] - df['close_10d_mean']) / df['close_10d_std']
    
    # Calculate the percentage change in volume compared to 5 days ago
    df['volume_change_5d'] = (df['volume'] / df['volume'].shift(5)) - 1
    
    # Calculate the rolling sum of the volume change over 5 days
    df['volume_change_5d_sum'] = df['volume_change_5d'].rolling(window=5).sum()
    
    # Heuristic: If the close price Z-score is negative and the 5-day volume change sum is positive, 
    # it may indicate a potential recovery in the stock.
    df['heuristic_factor'] = (df['close_z_score'] < 0) & (df['volume_change_5d_sum'] > 0)
    
    # Convert the boolean series to a numeric series where True is 1 and False is 0
    df['heuristic_factor'] = df['heuristic_factor'].astype(int)
    
    return df['heuristic_factor']
