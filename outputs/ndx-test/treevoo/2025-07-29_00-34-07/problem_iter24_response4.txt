```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between Open and Close
    df['price_gap'] = df['open'] - df['close']

    # Calculate Current Day Volume Growth
    df['volume_growth'] = df['volume'] / df['volume'].shift(1)

    # Weight the Price Gaps
    df['weighted_price_gap'] = df['price_gap'] * df['volume_growth']
    weighted_price_gap = df['weighted_price_gap'].sum()

    # Calculate 15-day Rolling High and 25-day Rolling Low
    df['rolling_high_15'] = df['high'].rolling(window=15).max()
    df['rolling_low_25'] = df['low'].rolling(window=25).min()

    # Calculate Rolling High-Low Differential
    df['high_low_diff'] = df['rolling_high_15'] - df['rolling_low_25']

    # Calculate Volume Volatility (Standard Deviation of Volume over 25 days)
    df['volume_volatility'] = df['volume'].rolling(window=25).std()

    # Weight by Volume Volatility
    df['weighted_high_low_diff'] = df['high_low_diff'] * df['volume_volatility']

    # Aggregate the Signals
    df['aggregate_signal'] = df['weighted_high_low_diff'] - (df['close'] - df['close'].shift(25))

    # Calculate Price Momentum
    df['price_return_1d'] = (df['close'] / df['close'].shift(1)) - 1
    df['price_return_7d'] = (df['close'] / df['close'].shift(7)) - 1
    df['price_return_35d'] = (df['close'] / df['close'].shift(35)) - 1

    # Average the Momentum
    df['average_momentum'] = (df['price_return_1d'] + df['price_return_7d'] + df['price_return_35d']) / 3

    # Adjust for Volume
    df['volume_percentage_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    df['adjusted_aggregate_signal'] = df['aggregate_signal'] * df['volume_percentage_change']

    # Final Alpha Factor
    df['alpha_factor'] = weighted_price_gap + df['adjusted_aggregate_signal'] - df['average_momentum']

    return df['alpha_factor']
```
