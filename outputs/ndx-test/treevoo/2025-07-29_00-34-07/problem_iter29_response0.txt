```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 15-Day Price Momentum
    df['15d_Price_Momentum'] = df['close'] - df['close'].shift(15)
    
    # Calculate 20-Day Rolling High-Low Differential
    df['20d_High'] = df['high'].rolling(window=20).max()
    df['20d_Low'] = df['low'].rolling(window=20).min()
    df['20d_High_Low_Diff'] = df['20d_High'] - df['20d_Low']
    
    # Volume-Weight the High-Low Differential
    df['Volume_Volatility'] = df['volume'].rolling(window=10).std()
    df['Vol_Weighted_High_Low_Diff'] = df['20d_High_Low_Diff'] * df['Volume_Volatility']
    
    # Adjust for Volume Changes
    df['Daily_Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Final_Adjustment'] = df['Vol_Weighted_High_Low_Diff'] * (1 + 2 * df['Daily_Volume_Change'])
    
    # Calculate Average Price Momentum
    df['2d_Return'] = df['close'] / df['close'].shift(2) - 1
    df['7d_Return'] = df['close'] / df['close'].shift(7) - 1
    df['30d_Return'] = df['close'] / df['close'].shift(30) - 1
    df['Average_Price_Momentum'] = (df['2d_Return'] + df['7d_Return'] + df['30d_Return']) / 3
    
    # Smooth and Adjust Initial Momentum
    df['Combined_Value'] = df['15d_Price_Momentum'] + df['Final_Adjustment']
    df['Smoothed_Combined_Value'] = df['Combined_Value'].rolling(window=7).mean()
    df['Adjusted_Smoothed_Momentum'] = df['Smoothed_Combined_Value'] - df['Smoothed_Combined_Value'].shift(7)
    
    # Calculate Weighted Price Gaps
    df['Daily_Price_Gap'] = df['high'] - df['low']
    df['Volume_Trend'] = df['volume'].rolling(window=5).mean() * (df['volume'] / df['volume'].shift(1))
    df['Weighted_Price_Gaps'] = (df['Daily_Price_Gap'] * df['Volume_Trend']).cumsum()
    
    # Combine Intermediate Factors
    df['Intermediate_Factors'] = df['Adjusted_Smoothed_Momentum'] + df['Average_Price_Momentum']
    
    # Final Alpha Factor Construction
    df['Final_Alpha'] = df['Intermediate_Factors'] + df['Weighted_Price_Gaps']
    df['Final_Alpha'] *= df['Average_Price_Momentum']
    df['Price_Momentum_15d'] = df['close'] - df['close'].rolling(window=15).mean()
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    df['Final_Alpha'] += df['Price_Momentum_15d'] * df['Log_Return']
    
    # Add 20-Day Rolling Range Indicator
    df['20d_Rolling_Range'] = df['20d_High'] - df['20d_Low']
    df['Normalized_Alpha'] = df['Final_Alpha'] / df['20d_Rolling_Range']
    
    # Incorporate Volume-Weighted Momentum
    df['10d_Price_Momentum'] = df['close'] - df['close'].shift(10)
    df['10d_Volume_Weighted_Momentum'] = df['10d_Price_Momentum'] * df['Volume_Volatility']
    df['Final_Alpha'] += df['10d_Volume_Weighted_Momentum']
    
    return df['Final_Alpha']
```
