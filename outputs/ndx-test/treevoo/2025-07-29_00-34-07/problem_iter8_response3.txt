```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Difference
    df['Adjusted_Price_Diff'] = (df['close'] - df['close'].shift(10)) * np.log(df['close'] / df['close'].shift(10))
    
    # Calculate Rolling High-Low Differential
    df['Rolling_High_20'] = df['high'].rolling(window=20).max()
    df['Rolling_Low_20'] = df['low'].rolling(window=20).min()
    df['Rolling_High_Low_Diff_20'] = df['Rolling_High_20'] - df['Rolling_Low_20']
    
    # Weight by Volume Volatility
    df['Volume_Volatility'] = df['volume'].rolling(window=10).std()
    df['Weighted_Rolling_High_Low_Diff_20'] = df['Rolling_High_Low_Diff_20'] * df['Volume_Volatility']
    
    # Calculate Volume Adjusted Momentum
    df['Volume_Adjusted_Momentum'] = (df['volume'] - df['volume'].shift(1)) * (df['close'] - df['close'].rolling(window=10).mean())
    
    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Rolling High-Low Range
    df['Rolling_High_30'] = df['high'].rolling(window=30).max()
    df['Rolling_Low_30'] = df['low'].rolling(window=30).min()
    df['Rolling_High_Low_Range_30'] = df['Rolling_High_30'] - df['Rolling_Low_30']
    
    # Weight by Volume Trend
    df['Volume_Trend'] = df['volume'] - df['volume'].rolling(window=10).mean()
    df['Weighted_Rolling_High_Low_Range_30'] = df['Rolling_High_Low_Range_30'] * df['Volume_Trend']
    
    # Calculate Weighted Average Momentum
    df['1_day_Price_Return'] = df['close'].pct_change()
    df['5_day_Price_Return'] = df['close'].pct_change(periods=5)
    df['20_day_Price_Return'] = df['close'].pct_change(periods=20)
    df['Weighted_Average_Momentum'] = 0.4 * df['1_day_Price_Return'] + 0.3 * df['5_day_Price_Return'] + 0.3 * df['20_day_Price_Return']
    
    # Adjust for Volume and Amount
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Daily_Amount_Change'] = df['amount'] - df['amount'].shift(1)
    df['Weighted_Average_Momentum_Adj'] = df['Weighted_Average_Momentum'] * (1 + (df['Daily_Volume_Change'] + df['Daily_Amount_Change']) / df['volume'].shift(1))
    
    # Aggregate the Signals
    df['Aggregated_Signal'] = (df['Weighted_Rolling_High_Low_Diff_20'] 
                              - df['Adjusted_Price_Diff'] 
                              + df['Volume_Adjusted_Momentum'] 
                              + df['Weighted_Rolling_High_Low_Range_30'] 
                              - df['Price_Momentum'] 
                              + df['Weighted_Average_Momentum_Adj'])
    
    # Final Factor
    df['Final_Factor'] = df['Aggregated_Signal'] / df['volume']
    
    return df['Final_Factor']
```
