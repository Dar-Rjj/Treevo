```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate Volume Trend
    volume_5day_avg = df['volume'].rolling(window=5).mean()
    volume_growth = df['volume'] / df['volume'].shift(1)
    volume_trend = volume_5day_avg * volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_diff = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].shift(1).rolling(window=10).std()
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Aggregate the Signals
    close_diff = df['close'] - df['close'].shift(10)
    aggregated_signal = weighted_high_low_diff - close_diff
    
    # Calculate High-to-Low Range
    high_low_range = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    one_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    combined_value = high_low_range * 100 + one_day_return
    
    # Smoothing and Initial Momentum Calculation
    ema_5 = combined_value.ewm(span=5, adjust=False).mean()
    momentum_diff = ema_5 - ema_5.shift(5)
    
    # Calculate Multi-Day Price Momentum
    one_day_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    seven_day_momentum = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    fourteen_day_momentum = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Weighted Average Momentum
    weights = [0.4, 0.3, 0.3]
    weighted_momentum = (weights[0] * one_day_momentum +
                        weights[1] * seven_day_momentum +
                        weights[2] * fourteen_day_momentum)
    
    # Adjust for Volume Changes
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    final_volume_adjustment = weighted_momentum * (1 + daily_volume_change / df['volume'].shift(1))
    
    # Integrate Transaction Amount
    daily_amount_change = df['amount'] - df['amount'].shift(1)
    intermediate_volume_adjustment = final_volume_adjustment * (1 + daily_amount_change / df['volume'].shift(1))
    
    # Intermediate Factor Combination
    intermediate_factor = momentum_diff + intermediate_volume_adjustment
    
    # Calculate Volume-Weighted Price Momentum
    log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    volume_weighted_price_momentum = (df['close'] - ema_10_close) * log_return
    
    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps + 
                    intermediate_factor + 
                    volume_weighted_price_momentum)
    
    return alpha_factor
```
