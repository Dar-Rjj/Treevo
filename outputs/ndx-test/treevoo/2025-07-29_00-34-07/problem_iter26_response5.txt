```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative close price compared to the open, high, and low
    rel_close_to_open = (df['close'] - df['open']) / df['open']
    rel_close_to_high = (df['close'] - df['high']) / df['high']
    rel_close_to_low = (df['close'] - df['low']) / df['low']
    
    # Volume weighted price change
    vol_weighted_change = (df['close'] - df['open']) * df['volume']
    
    # Heuristic factor: sum of relative close prices and volume weighted change
    factor = rel_close_to_open + rel_close_to_high + rel_close_to_low + vol_weighted_change
    
    return factor
```
