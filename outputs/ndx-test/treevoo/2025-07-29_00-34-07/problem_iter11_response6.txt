```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Compute 5-day Moving Average of Volumes
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    
    # Compute Current Day Volume Growth
    volume_growth = df['volume'] / df['volume'].shift(1)
    
    # Multiply 5-day Moving Average by Volume Growth
    volume_trend = volume_5day_ma * volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-to-Low Range with Weighted Price Gaps
    combined_value = high_low_range * 100 + weighted_price_gaps
    
    # Smoothing and Momentum Calculation
    sma_combined = combined_value.rolling(window=5).mean()
    momentum = sma_combined - sma_combined.shift(5)
    
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(10)
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    rolling_high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    
    # Multiply Rolling High-Low Differential by Volume Volatility
    weighted_rolling_high_low_diff = rolling_high_low_diff * volume_volatility
    
    # Aggregate the Signals
    aggregate_signal = weighted_rolling_high_low_diff - price_diff
    
    # Calculate Price Momentum
    price_return_1d = df['close'] / df['close'].shift(1) - 1
    price_return_5d = df['close'] / df['close'].shift(5) - 1
    price_return_20d = df['close'] / df['close'].shift(20) - 1
    
    # Average the Momentum
    average_momentum = (price_return_1d + price_return_5d + price_return_20d) / 3
    
    # Adjust for Volume
    volume_pct_change = df['volume'].pct_change()
    adjusted_aggregate_signal = aggregate_signal * volume_pct_change
    
    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps 
                    + momentum 
                    + adjusted_aggregate_signal 
                    - average_momentum)
    
    return alpha_factor
```
