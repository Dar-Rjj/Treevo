```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['High'] - df['Low']

    # Calculate Price Change (Close at t - Open at t-1)
    price_change = df['Close'] - df['Open'].shift(1)

    # Combine High-to-Low Range with Price Change
    combined_value = (high_low_range * 150) + price_change

    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = combined_value.rolling(window=7).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(7)

    # Calculate Multi-Day Price Momentum
    three_day_return = (df['Close'] - df['Close'].shift(3)) / df['Close'].shift(3)
    ten_day_return = (df['Close'] - df['Close'].shift(10)) / df['Close'].shift(10)
    thirty_day_return = (df['Close'] - df['Close'].shift(30)) / df['Close'].shift(30)

    # Average the Momentum
    average_momentum = (three_day_return + ten_day_return + thirty_day_return) / 3

    # Adjust for Volume Changes
    daily_vol_change = (df['Volume'] - df['Volume'].shift(1)) / df['Volume'].shift(1)
    volume_adjusted_momentum = average_momentum * (1 + 2 * daily_vol_change)

    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + volume_adjusted_momentum

    # Calculate Weighted Price Momentum
    log_return = np.log(df['Close'] / df['Close'].shift(1))
    weighted_price_momentum = (df['Close'] - df['Close'].rolling(window=15).mean()) * log_return

    # Final Factor Construction
    final_factor = intermediate_factor + weighted_price_momentum

    return final_factor
```
