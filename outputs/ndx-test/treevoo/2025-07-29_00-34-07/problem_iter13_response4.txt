```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Volume Trend
    df['vol_5d_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_growth'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend'] = df['vol_5d_ma'] * df['volume_growth']
    
    # Filter by Relative Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    df_filtered = df[df['volume'] > avg_volume]
    
    # Calculate Adjusted Price Difference
    df['adj_price_diff'] = (df['close'] - df['close'].shift(10)) * np.log(df['close'] / df['close'].shift(10))
    
    # Calculate Rolling High-Low Differential
    df['roll_high'] = df['high'].rolling(window=20).max()
    df['roll_low'] = df['low'].rolling(window=20).min()
    df['roll_high_low_diff'] = df['roll_high'] - df['roll_low']
    
    # Weight by Volume Volatility
    df['vol_volatility'] = df['volume'].rolling(window=10).std()
    df['weighted_high_low_diff'] = df['roll_high_low_diff'] * df['vol_volatility']
    
    # Calculate Volume Adjusted Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) * (df['close'] - df['close'].rolling(window=10).mean())
    
    # Aggregate Price and Volume Signals
    df['aggregated_signal'] = df['weighted_high_low_diff'] - df['adj_price_diff'] + df['volume_momentum']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['aggregated_signal'] / df['volume']
    
    return df['alpha_factor']
```
