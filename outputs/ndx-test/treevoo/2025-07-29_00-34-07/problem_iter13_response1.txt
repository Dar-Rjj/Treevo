```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_low_range * 150) + price_change
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined = combined_value.rolling(window=7).mean()
    initial_momentum = smoothed_combined - smoothed_combined.shift(7)
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Multi-Day Price Momentum
    momentum_3day = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    momentum_10day = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    momentum_30day = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Average the Momentum
    average_momentum = (momentum_3day + momentum_10day + momentum_30day) / 3
    
    # Adjust for Volume Changes
    volume_pct_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjusted_momentum = average_momentum * (1 + 2 * volume_pct_change)
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + final_volume_adjusted_momentum
    
    # Calculate Weighted Price Momentum
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=15).mean()) * daily_log_return
    
    # Final Factor Construction
    final_factor = intermediate_factor + weighted_price_momentum
    
    return final_factor
```
