```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between Open and Close
    df['PriceGap'] = df['open'] - df['close']
    
    # Calculate Current Day Volume Growth
    df['VolumeGrowth'] = df['volume'] / df['volume'].shift(1)
    
    # Weight the Price Gaps
    df['WeightedPriceGap'] = df['PriceGap'] * df['VolumeGrowth']
    
    # Sum of Weighted Price Gaps
    df['SumWeightedPriceGap'] = df['WeightedPriceGap'].rolling(window=5).sum()
    
    # Calculate 15-day Rolling High and 25-day Rolling Low
    df['RollingHigh15'] = df['high'].rolling(window=15).max()
    df['RollingLow25'] = df['low'].rolling(window=25).min()
    
    # Calculate High-Low Differential
    df['HighLowDifferential'] = df['RollingHigh15'] - df['RollingLow25']
    
    # Subtract Long-Term Price Difference
    df['LongTermPriceDifference'] = df['close'] - df['close'].shift(25)
    
    # Aggregate the Signals
    df['AggregatedSignal'] = (df['HighLowDifferential'] + df['SumWeightedPriceGap']) - df['LongTermPriceDifference']
    
    # Calculate Price Momentum
    df['Momentum1d'] = df['close'] / df['close'].shift(1) - 1
    df['Momentum7d'] = df['close'] / df['close'].shift(7) - 1
    df['Momentum35d'] = df['close'] / df['close'].shift(35) - 1
    
    # Average the Momentum
    df['AverageMomentum'] = (df['Momentum1d'] + df['Momentum7d'] + df['Momentum35d']) / 3
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['SumWeightedPriceGap'] + df['AggregatedSignal'] - df['AverageMomentum']
    
    return df['AlphaFactor'].dropna()
```
