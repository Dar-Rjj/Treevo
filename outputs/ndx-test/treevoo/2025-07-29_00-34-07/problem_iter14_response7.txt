```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'] - df['close'].shift(15)
    
    # Calculate 25-day Rolling High and Low
    rolling_high = df['high'].rolling(window=25).max()
    rolling_low = df['low'].rolling(window=25).min()
    
    # Calculate Rolling High-Low Differential
    high_low_diff = rolling_high - rolling_low
    
    # Calculate Volume Volatility (standard deviation of volume over the last 15 days)
    volume_volatility = df['volume'].rolling(window=15).std()
    
    # Weight by Volume Volatility
    weighted_high_low_diff = high_low_diff * volume_volatility
    
    # Daily Volume Percentage Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Final Volume Adjustment
    adjusted_high_low_diff = weighted_high_low_diff * (1 + 2 * daily_volume_change)
    
    # Aggregate the Signals
    aggregated_signals = adjusted_high_low_diff - price_diff
    
    # Calculate Price Momentum
    price_momentum_2d = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    price_momentum_7d = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    price_momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Average the Momentum
    average_momentum = (price_momentum_2d + price_momentum_7d + price_momentum_30d) / 3
    
    # Smoothing and Initial Momentum Calculation
    smoothed_momentum = average_momentum.rolling(window=7).mean()
    initial_momentum = smoothed_momentum - smoothed_momentum.shift(7)
    
    # Calculate Multi-Day Price Momentum
    multi_day_momentum_3d = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    multi_day_momentum_10d = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    multi_day_momentum_30d = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Average the Multi-Day Momentum
    averaged_multi_day_momentum = (multi_day_momentum_3d + multi_day_momentum_10d + multi_day_momentum_30d) / 3
    
    # Filter by Relative Volume
    average_volume = df['volume'].rolling(window=30).mean()
    filtered_days = (df['volume'] > average_volume)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    sum_daily_price_range = daily_price_range.where(filtered_days).rolling(window=30).sum()
    weighted_price_gaps = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1)).where(filtered_days)
    sum_weighted_price_gaps = weighted_price_gaps.rolling(window=30).sum()
    
    # Intermediate Factor Combination
    intermediate_factors = initial_momentum + averaged_multi_day_momentum
    
    # Calculate Weighted Price Momentum
    log_return = (df['close'] - df['close'].rolling(window=15).mean()) / df['close']
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=15).mean()) * log_return
    
    # Final Alpha Factor Construction
    final_alpha_factor = (sum_daily_price_range + sum_weighted_price_gaps) * average_momentum + intermediate_factors + weighted_price_momentum
    
    return final_alpha_factor
```
