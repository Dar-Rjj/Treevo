```python
def heuristics_v2(df):
    # Calculate Price Momentum and Gaps
    df['1_day_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['5_day_return'] = (df['close'] / df['close'].shift(5)) - 1
    df['20_day_return'] = (df['close'] / df['close'].shift(20)) - 1
    df['price_gap'] = df['high'] - df['low']
    
    # Average the Momentum
    df['average_momentum'] = (df['1_day_return'] + df['5_day_return'] + df['20_day_return']) / 3
    
    # Filter by Relative Volume
    n = 20  # period for average volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_filter'] = df['volume'] > df['avg_volume']
    
    # Calculate Adjusted High-Low Cumulative Return and Weighted Price Gaps
    df['daily_price_range'] = df['high'] - df['low']
    df_filtered = df[df['volume_filter']]
    
    # Sum of Daily Price Ranges for Filtered Days
    daily_price_range_sum = df_filtered['daily_price_range'].sum()
    
    # Weight the Price Gaps
    df_filtered['volume_growth'] = df_filtered['volume'] / df_filtered['avg_volume']
    weighted_price_gaps = (df_filtered['price_gap'] * df_filtered['volume_growth']).sum()
    
    # Combine Adjusted High-Low Cumulative Return and Weighted Price Gaps
    combined_value = daily_price_range_sum + weighted_price_gaps
    
    # Generate Alpha Factor
    alpha_factor = combined_value * df['average_momentum']
    
    return alpha_factor
```
