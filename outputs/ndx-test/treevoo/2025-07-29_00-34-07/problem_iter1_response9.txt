```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Average Volume over a period (e.g., 30 days)
    n_days = 30
    df['avg_volume'] = df['volume'].rolling(window=n_days).mean()
    
    # Apply Volume Filter: If Today's Volume is greater than Average Volume
    df_filtered = df[df['volume'] > df['avg_volume']]
    
    # Calculate Cumulative High-Low Momentum
    df_filtered['cumulative_momentum'] = df_filtered['price_range'].cumsum()
    
    # Generate Alpha Factor
    alpha_factor = df_filtered['cumulative_momentum']
    
    return alpha_factor
```
