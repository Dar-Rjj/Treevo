def heuristics_v2(df):
    # Calculate the relative position of close price within the high-low range
    df['close_pos'] = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the ratio of close to open
    df['close_open_ratio'] = df['close'] / df['open']

    # Calculate the volume weighted average price
    df['vwap'] = (df['close'] + df['high'] + df['low']) / 3 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()

    # Calculate the normalized volume change
    df['vol_change'] = df['volume'].pct_change()

    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_pos'] * df['close_open_ratio'] * (df['vwap'] - df['close']) * df['vol_change'])

    return df['alpha_factor']
