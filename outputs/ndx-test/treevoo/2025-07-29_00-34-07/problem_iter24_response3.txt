```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['close'].diff(15)
    
    # Calculate Rolling High-Low Differential
    rolling_high = df['high'].rolling(window=20).max()
    rolling_low = df['low'].rolling(window=20).min()
    high_low_differential = rolling_high - rolling_low
    
    # Weight by Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std()
    weighted_high_low_differential = high_low_differential * volume_volatility
    
    # Adjust for Volume Changes
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_volume_adjustment = weighted_high_low_differential * (1 + 2 * daily_volume_change)
    
    # Aggregate the Signals
    aggregated_signals = final_volume_adjustment - price_diff
    
    # Calculate Averaged Momentum
    one_day_return = (df['close'] / df['close'].shift(1) - 1)
    five_day_return = (df['close'] / df['close'].shift(5) - 1)
    thirty_day_return = (df['close'] / df['close'].shift(30) - 1)
    averaged_momentum = (one_day_return + five_day_return + thirty_day_return) / 3
    
    # Smoothing and Initial Momentum Calculation
    smoothed_combined_value = aggregated_signals.rolling(window=7).mean()
    initial_momentum = smoothed_combined_value - smoothed_combined_value.shift(7)
    
    # Adjusted High-Low Cumulative Return and Weighted Price Gaps
    daily_price_range = df['high'] - df['low']
    weighted_price_gaps = (daily_price_range * (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)).sum()
    
    # Intermediate Factor Combination
    intermediate_factor = initial_momentum + averaged_momentum
    
    # Calculate Weighted Price Momentum
    log_return = (df['close'] / df['close'].rolling(window=20).mean()).apply(lambda x: 2 if x > 1 else -2)
    weighted_price_momentum = (df['close'] - df['close'].rolling(window=20).mean()) * log_return
    
    # Adjust for Volume
    adjusted_weighted_price_gaps = weighted_price_gaps * daily_volume_change
    
    # High-to-Low Range and Momentum
    high_to_low_range = (df['high'] - df['low']) * 100
    combined_value = high_to_low_range + adjusted_weighted_price_gaps
    smoothed_combined_value_high_low = combined_value.rolling(window=5).mean()
    momentum_high_low = smoothed_combined_value_high_low - smoothed_combined_value_high_low.shift(5)
    
    # Final Alpha Factor Construction
    alpha_factor = (
        daily_price_range.sum() + 
        weighted_price_gaps * averaged_momentum + 
        intermediate_factor + 
        weighted_price_momentum + 
        momentum_high_low + 
        final_volume_adjustment - 
        averaged_momentum
    )
    
    return alpha_factor
```
