```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Price Change
    price_change = df['close'] - df['open'].shift(1)
    
    # Combine High-to-Low Range with Price Change
    combined_value = (high_to_low_range * 100) + price_change
    
    # Apply Simple Moving Average on Combined Value over 5 days
    sma_5 = combined_value.rolling(window=5).mean()
    
    # Calculate the difference between day t and day t-5
    momentum = sma_5 - sma_5.shift(5)
    
    return momentum
```
