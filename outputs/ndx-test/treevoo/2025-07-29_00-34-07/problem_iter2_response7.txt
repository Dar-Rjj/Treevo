```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Gap Between High and Low
    price_gap = df['high'] - df['low']
    
    # Calculate Volume Trend
    volume_5d_ma = df['volume'].rolling(window=5).mean()
    current_day_volume_growth = df['volume'] / df['volume'].shift(1)
    volume_trend = volume_5d_ma * current_day_volume_growth
    
    # Weight the Price Gaps
    weighted_price_gaps = (price_gap * volume_trend).sum()
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Combine High-to-Low Range with Weighted Price Gaps
    combined_value = (high_low_range * 100) + weighted_price_gaps
    
    # Smoothing and Momentum Calculation
    smoothed_combined_value = combined_value.rolling(window=5).mean()
    momentum = smoothed_combined_value - smoothed_combined_value.shift(5)
    
    # Final Alpha Factor
    alpha_factor = (weighted_price_gaps * momentum)
    
    return alpha_factor
```
