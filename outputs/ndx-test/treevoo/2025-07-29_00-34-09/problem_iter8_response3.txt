```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low'] * (df['volume'] / (df['high'] + df['low']))
    df['sustained_5day_momentum'] = (df['close'].diff() / df['close'].shift(1)).rolling(window=5).mean()
    df['combined_momentum'] = (df['intraday_momentum'] * df['amount'].rolling(window=5).mean()) + (df['sustained_5day_momentum'] * df['volume'])
    
    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']
    
    # Compute 5-Day Exponential Moving Average of Price Change
    df['5d_ema'] = df['daily_change'].ewm(span=5, adjust=False).mean()
    
    # Compute 10-Day Exponential Moving Average of Price Change
    df['10d_ema'] = df['daily_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['5d_ema'] > df['10d_ema'], 'reversal_signal'] = 1
    df.loc[df['5d_ema'] < df['10d_ema'], 'reversal_signal'] = -1
    df['14d_rolling_median_range'] = df['daily_range'].rolling(window=14).median()
    df['reversal_signal'] = df['reversal_signal'] * (1 + (df['14d_rolling_median_range'] / df['14d_rolling_median_range'].mean()))
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df.loc[df['volume'] < volume_threshold, 'reversal_signal'] = 0
    
    # Calculate Weighted Daily Return
    df['weighted_return'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Cumulative Weighted Momentum
    df['cumulative_weighted_momentum'] = df['weighted_return'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] + df['cumulative_weighted_momentum'] + df['reversal_signal']
    
    return df['alpha_factor']
```
