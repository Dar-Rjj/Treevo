```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Rolling Average of High-to-Low Range (14-day window)
    rolling_avg_high_low_range = high_low_range.rolling(window=14).mean()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Compute 5-Day Moving Average of Price Change
    ma_5_day = daily_price_change.rolling(window=5).mean()
    
    # Compute 10-Day Moving Average of Price Change
    ma_10_day = daily_price_change.rolling(window=10).mean()
    
    # Determine Reversal Signal
    reversal_signal = pd.Series(index=df.index, dtype=float)
    for i in range(1, len(df)):
        if ma_5_day.iloc[i] > ma_10_day.iloc[i] and ma_5_day.iloc[i-1] <= ma_10_day.iloc[i-1]:
            # Bullish trend
            reversal_signal.iloc[i] = 1 * rolling_avg_high_low_range.iloc[i]
        elif ma_5_day.iloc[i] < ma_10_day.iloc[i] and ma_5_day.iloc[i-1] >= ma_10_day.iloc[i-1]:
            # Bearish trend
            reversal_signal.iloc[i] = -1 * rolling_avg_high_low_range.iloc[i]
        else:
            reversal_signal.iloc[i] = 0
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)  # 75th percentile as a threshold
    reversal_signal[df['volume'] < volume_threshold] = 0
    
    return reversal_signal
```
