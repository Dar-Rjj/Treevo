```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the intraday range
    intraday_range = df['high'] - df['low']
    
    # Calculate the close to open change
    close_to_open_change = df['close'] - df['open']
    
    # Calculate the volume weighted price
    volume_weighted_price = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the factor
    factor = (intraday_range / df['close']) + (close_to_open_change / df['open']) + (volume_weighted_price - df['close'])
    
    return factor
```
