```python
import pandas as pd

def heuristics_v2(df):
    # Rate of change in closing prices over different time windows
    df['ROC_5'] = (df['close'] / df['close'].shift(5)) - 1
    df['ROC_20'] = (df['close'] / df['close'].shift(20)) - 1
    df['ROC_60'] = (df['close'] / df['close'].shift(60)) - 1
    
    # Williams %R over different time windows
    df['Williams_R_14'] = ((df['high'].rolling(window=14).max() - df['close']) / 
                           (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min())) * -100
    df['Williams_R_28'] = ((df['high'].rolling(window=28).max() - df['close']) / 
                           (df['high'].rolling(window=28).max() - df['low'].rolling(window=28).min())) * -100
    
    # Volatility based on the standard deviation of price changes
    df['Volatility_30'] = df['close'].rolling(window=30).std()
    df['Volatility_60'] = df['close'].rolling(window=60).std()
    
    # Volume analysis
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['VRS'] = df['Avg_Volume_10'] / df['volume'].rolling(window=20).mean()
    df['VM'] = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Price Breakout Indicator and Price Consolidation Indicator
    min_low = df['low'].rolling(window=20).min()
    max_high = df['high'].rolling(window=20).max()
    df['PBI'] = ((df['close'] - min_low) / min_low * 100).where(df['close'] > max_high, other=0)
    df['PCI'] = (max_high - min_low) / df['close']
    
    # Open-Close Ratio and counts
    df['OCR'] = df['close'] / df['open']
    df['Positive_OCR_Count'] = (df['OCR'] > 1).rolling(window=20).sum()
    df['Negative_OCR_Count'] = (df['OCR'] < 1).rolling(window=20).sum()
    
    # High-Low Ratio and count
    df['HLR'] = df['high'] / df['low']
    df['HLR_1_05_Count'] = (df['HLR'] > 1.05).rolling(window=20).sum()

    # Combining all factors into a single alpha factor
    df['Alpha_Factor'] = (df['ROC_5'] + df['ROC_20'] + df['ROC_60'] +
                          df['Williams_R_14'] + df['Williams_R_28'] +
                          df['Volatility_30'] + df['Volatility_60'] +
                          df['VRS'] + df['VM'] +
                          df['PBI'] + df['PCI'] +
                          df['Positive_OCR_Count'] - df['Negative_OCR_Count'] +
                          df['HLR_1_05_Count'])
    
    return df['Alpha_Factor']
```
