```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate 5-Day Average Amount
    df['5_day_avg_amount'] = df['amount'].rolling(window=5).mean()
    
    # Calculate Weighted Momentum
    df['weighted_momentum'] = (df['daily_return'] * df['volume']).rolling(window=5).sum() / df['5_day_avg_amount']
    
    # Drop NA values
    df.dropna(inplace=True)
    
    # Return the factor values
    return df['weighted_momentum']
```
