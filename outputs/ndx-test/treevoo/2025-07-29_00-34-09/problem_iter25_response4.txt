```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = (df['high'] - df['low']) / df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low'] * df['volume'] / (df['high'] + df['low'])
    sequential_momentum_5d = (df['close'].diff() / df['close'].shift(1)).rolling(window=5).mean()
    combined_momentum = (intraday_momentum * df['amount'].rolling(window=5).mean()) + (sequential_momentum_5d * df['volume'])

    # Daily Price Change
    daily_price_change = df['close'] - df['open']

    # Compute 5-Day and 10-Day EMAs of Price Change
    ema_5d = daily_price_change.ewm(span=5, adjust=False).mean()
    ema_10d = daily_price_change.ewm(span=10, adjust=False).mean()

    # Determine Reversal Signal
    reversal_signal = np.where(ema_5d > ema_10d, 1, np.where(ema_5d < ema_10d, -1, 0))
    high_low_median_14d = high_to_low_range.rolling(window=14).median()
    reversal_signal_amplified = np.where(high_low_median_14d > high_low_median_14d.mean(), reversal_signal * 2, 
                                         np.where(high_low_median_14d < high_low_median_14d.mean(), reversal_signal / 2, reversal_signal))

    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    filtered_reversal_signal = np.where(df['volume'] > volume_threshold, reversal_signal_amplified, 0)

    # Cumulative Weighted Momentum
    weighted_daily_return = (df['close'].pct_change() * df['amount']).rolling(window=5).sum() / df['amount'].rolling(window=5).mean()

    # Incorporate Volatility into Alpha Factor
    historical_volatility_21d = df['close'].pct_change().rolling(window=21).std()
    volatility_adjusted_factor = 1 / historical_volatility_21d

    # Final Alpha Factor
    final_alpha_factor = (combined_momentum + weighted_daily_return + filtered_reversal_signal) * volatility_adjusted_factor

    return final_alpha_factor
```
