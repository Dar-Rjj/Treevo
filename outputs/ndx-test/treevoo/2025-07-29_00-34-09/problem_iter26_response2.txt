```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weigh by Volume
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['ema_vol_adj_log_return'] = df['volume_weighted_log_return'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_sd'] = df['high_low_range'].rolling(window=20).std()
    df['vol_adj_ema'] = df['ema_vol_adj_log_return'] / df['high_low_range_sd']
    
    # 10-day Simple Moving Average (SMA) of the Close Price
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    
    # Momentum Factor
    df['momentum'] = df['close'] - df['sma_close_10']
    df['momentum_sign'] = np.sign(df['momentum'])
    
    # High-Activity Days
    df['volume_30_sma'] = df['volume'].rolling(window=30).mean()
    df['high_activity'] = df['volume'] > df['volume_30_sma']
    
    # Percentage Change in Close Price on High-Activity Days
    df['close_pct_change'] = df['close'].pct_change()
    df['momentum_score_high_activity'] = df['momentum'] * df['close_pct_change'] * df['high_activity']
    
    # Intraday Market Sentiment
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['sentiment'] = np.where(df['close'] > df['midpoint'], 1, -1)
    
    # Composite Signal
    df['composite_signal'] = df['momentum'] * df['sentiment']
    
    # Filter by Volume
    df['composite_signal'] = df['composite_signal'] * (df['volume'] > df['volume_30_sma'])
    
    # Price Reversals Using High, Low, and Open Prices
    df['reversal'] = (df['high'] - df['low']) / df['open']
    df['reversal_sum_20'] = df['reversal'].rolling(window=20).sum()
    
    # Volume-Weighted Close
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean()
    
    # Opening Gap Volatility
    df['open_gap'] = df['open'] - df['close'].shift(1)
    df['open_gap_volatility'] = df['open_gap'].rolling(window=20).std()
    
    # Integrate All Factors for Final Alpha
    df['final_alpha'] = (
        0.4 * df['vol_adj_ema'] +
        0.1 * df['reversal_sum_20'] +
        0.3 * df['reversal_sum_20'] +
        0.05 * df['volume_weighted_close'] +
        0.05 * df['composite_signal'] +
        0.05 * df['high_low_range_sd'] +
        0.05 * df['open_gap_volatility']
    )
    
    return df['final_alpha']
```
