```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    high_low_ratio = (df['high'] - df['low']) / df['low']
    intraday_activity = df['volume'] / (df['high'] + df['low'])
    intraday_momentum = high_low_ratio * intraday_activity
    
    # Measure Sustained 5-Day Momentum
    close_prices = df['close']
    previous_close_prices = close_prices.shift(1)
    sequential_momentum = (close_prices - previous_close_prices) / previous_close_prices
    sustained_5_day_momentum = sequential_momentum.rolling(window=5).mean()
    
    # Combine Intraday and 5-Day Momentum
    average_5_day_amount = df['amount'].rolling(window=5).mean()
    combined_intraday_momentum = intraday_momentum * average_5_day_amount
    combined_sustained_momentum = sustained_5_day_momentum * df['volume']
    combined_momentum = combined_intraday_momentum + combined_sustained_momentum
    
    # Calculate Weighted Daily Return
    weighted_daily_return = (close_prices - previous_close_prices) * df['volume']
    
    # Calculate Cumulative Weighted Momentum
    cumulative_weighted_momentum = weighted_daily_return.rolling(window=5).sum() / average_5_day_amount
    
    # Final Alpha Factor
    alpha_factor = combined_momentum + cumulative_weighted_momentum
    
    return alpha_factor
```
