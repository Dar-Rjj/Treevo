```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['daily_range'] = df['high'] - df['low']

    # Calculate Rolling Average of High-to-Low Range (14-day window)
    df['range_14d_avg'] = df['daily_range'].rolling(window=14).mean()

    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']

    # Compute 5-Day and 10-Day Moving Averages of Volume-Weighted Price Change
    df['volume_weighted_change_5d'] = (df['daily_change'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['volume_weighted_change_10d'] = (df['daily_change'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['volume_weighted_change_5d'] > df['volume_weighted_change_10d'], 'reversal_signal'] = 1
    df.loc[df['volume_weighted_change_5d'] < df['volume_weighted_change_10d'], 'reversal_signal'] = -1

    # Amplify or Dampen the Reversal Signal based on 14-day Rolling Average of High-to-Low Range
    df['amplified_signal'] = df['reversal_signal'] * (df['range_14d_avg'] / df['range_14d_avg'].median())

    # Filter by Volume (only consider if volume is above a certain threshold, e.g., 90th percentile)
    volume_threshold = df['volume'].quantile(0.9)
    df.loc[df['volume'] < volume_threshold, 'amplified_signal'] = 0

    return df['amplified_signal']
```
