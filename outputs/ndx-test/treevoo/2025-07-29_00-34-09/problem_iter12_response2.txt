```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weighted by Volume
    df['volume_weighted_returns'] = df['log_returns'] * df['volume']
    df['ema_volume_adjusted'] = df['volume_weighted_returns'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['hl_volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Volume-Adjusted Price Movement
    df['vol_adj_momentum'] = df['ema_volume_adjusted'] / df['hl_volatility']
    
    # Calculate Rolling Average of High-to-Low Range (14-day window)
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=14).mean()
    
    # Daily Price Change (Close - Open)
    df['daily_change'] = df['close'] - df['open']
    
    # 5-Day and 10-Day Moving Averages of Price Change
    df['ma_5'] = df['daily_change'].rolling(window=5).mean()
    df['ma_10'] = df['daily_change'].rolling(window=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ma_5'] > df['ma_10'], 'reversal_signal'] = 1
    df.loc[df['ma_5'] < df['ma_10'], 'reversal_signal'] = -1
    df['reversal_signal'] *= df['avg_high_low_range'].rank(pct=True) * 2 - 1
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['filtered_signal'] = df['reversal_signal']
    df.loc[df['volume'] <= volume_threshold, 'filtered_signal'] = 0
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversals'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversals'] = df['price_reversals'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['alpha_factor'] = (0.4 * df['vol_adj_momentum'] 
                         + 0.3 * df['filtered_signal'] 
                         + 0.2 * df['sum_price_reversals'] 
                         + 0.1 * df['volume_weighted_close'])
    
    return df['alpha_factor']
```
