```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns
    df['ema_log_returns'] = df['log_returns'].ewm(span=20, adjust=False).mean()
    
    # Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = df['ema_log_returns'] * df['volume']

    # Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # 20-Day Moving SD of High-Low Ranges
    df['volatility'] = df['high_low_range'].rolling(window=20).std()

    # Compute Price Slope
    def compute_price_slope(group):
        x = np.arange(20)
        y = group.values
        slope, _ = np.polyfit(x, y, 1)
        return slope
    
    df['price_slope'] = df['close'].rolling(window=20).apply(compute_price_slope, raw=True)
    
    # 20-Day EMA of Price Slope
    df['ema_price_slope'] = df['price_slope'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors
    df['alpha_factor'] = (df['volume_adjusted_momentum'] + 
                          df['volatility'] + 
                          df['ema_price_slope'])
    
    return df['alpha_factor']
```
