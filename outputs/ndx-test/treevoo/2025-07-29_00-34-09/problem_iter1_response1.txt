import pandas as pd

def heuristics_v2(df):
    # Weight High and Low Prices by Volume
    weighted_price = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Calculate Absolute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Average Volume over Period
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Adjust for Average Volume
    range_volatility = intraday_range / avg_volume
    
    # Difference between Consecutive Volume-Weighted Prices
    price_diff = weighted_price.diff()
    
    # Combine Volume-Weighted Price with Intraday Range Volatility
    alpha_factor = price_diff * range_volatility
    
    return alpha_factor
