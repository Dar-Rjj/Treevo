import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price within the high-low range
    close_position = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2

    # Compute the difference between the close price and the average price
    price_diff = df['close'] - avg_price

    # Calculate the factor value as a weighted sum of the two components
    factor_value = 0.5 * close_position + 0.5 * price_diff

    return factor_value
