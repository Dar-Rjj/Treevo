import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Compute Intraday Range Volatility
    intraday_range = df['high'] - df['low']
    average_volume = df['volume'].rolling(window=20).mean()
    intraday_volatility = intraday_range / average_volume
    
    # Assess Momentum Based on Close Price
    close_momentum = df['close'].rolling(window=20).mean()
    
    # Analyze Price Reversals Using High and Low Prices
    price_reversal = (df['high'] - df['low']) / df['open']
    
    # Evaluate Volume Impact on Price Movement
    vwap_ema = vwap.ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Intermediate Alpha
    vwap_diff = vwap.diff()
    intermediate_alpha = vwap_diff * intraday_volatility + close_momentum + price_reversal + vwap_ema
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Weighted Price
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Smoothed Momentum
    short_ema = weighted_price.ewm(span=5, adjust=False).mean()
    long_ema = weighted_price.ewm(span=20, adjust=False).mean()
    smoothed_momentum = short_ema - long_ema
    
    # Calculate Intraday Price Change Ratio
    high_low_ratio = (df['high'] - df['low']) / df['low']
    intraday_trading_activity = df['volume'] / (df['high'] + df['low'])
    intraday_price_change_ratio = high_low_ratio * intraday_trading_activity
    
    # Measure Sustained Momentum
    sustained_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = df['volume'] * sustained_momentum
    
    # Combine All Factors
    alpha_factor = (
        intermediate_alpha +
        smoothed_momentum +
        intraday_price_change_ratio +
        sustained_momentum +
        volume_weighted_momentum
    )
    
    return alpha_factor
