```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weigh by Volume
    df['vol_adj_log_return'] = df['log_return'] * df['volume']
    df['ema_vol_adj_log_return'] = df['vol_adj_log_return'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['std_high_low_range'] = df['high_low_range'].rolling(window=20).std()
    df['weight_high_low_vol'] = df['high_low_range'] / df['std_high_low_range']
    
    # Volume-Adjusted Price Movement
    df['vol_adj_price_mov'] = df['ema_vol_adj_log_return'] * df['weight_high_low_vol']
    
    # Calculate Rolling Average of High-to-Low Range
    df['avg_high_low_range_14'] = df['high_low_range'].rolling(window=14).mean()
    
    # Intraday Price Change
    df['intraday_change'] = df['close'] - df['open']
    
    # 5-Day and 10-Day Moving Averages of Intraday Price Change
    df['ma_intraday_change_5'] = df['intraday_change'].rolling(window=5).mean()
    df['ma_intraday_change_10'] = df['intraday_change'].rolling(window=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ma_intraday_change_5'] > df['ma_intraday_change_10'], 'reversal_signal'] = 1
    df.loc[df['ma_intraday_change_5'] < df['ma_intraday_change_10'], 'reversal_signal'] = -1
    df['reversal_signal'] *= (df['avg_high_low_range_14'] / df['avg_high_low_range_14'].mean())
    
    # Filter by Volume
    volume_threshold = df['volume'].mean()
    df.loc[df['volume'] < volume_threshold, 'reversal_signal'] = 0
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversal_20'] = df['price_reversal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['vol_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['alpha'] = (0.3 * df['vol_adj_price_mov'] + 
                   0.2 * df['reversal_signal'] + 
                   0.3 * df['sum_price_reversal_20'] + 
                   0.2 * df['vol_weighted_close'])
    
    return df['alpha']
```
