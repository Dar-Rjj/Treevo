```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Based on Close Price
    df['close_momentum'] = df['close'].pct_change(periods=20)
    df['close_20_ma'] = df['close'].rolling(window=20).mean()
    df['momentum_factor'] = df['close_momentum'] / df['close_20_ma']
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['daily_range_ratio'] = (df['high'] - df['low']) / df['open']
    df['price_reversal'] = df['daily_range_ratio'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['volume_weighted_close_ema'] = df['volume_weighted_close'].ewm(span=20, adjust=False).mean()
    df['volume_impact'] = df['volume_weighted_close_ema'] / df['close'].rolling(window=20).mean()
    
    # Combine Factors for Final Alpha
    df['alpha_factor'] = (0.4 * df['momentum_factor'] 
                          + 0.3 * df['price_reversal'] 
                          + 0.3 * df['volume_impact'])
    
    return df['alpha_factor'].dropna()
```
