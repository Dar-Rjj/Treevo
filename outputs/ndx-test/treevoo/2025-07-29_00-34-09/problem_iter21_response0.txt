```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3
    
    # Generate Intraday Volatility Metric
    intraday_volatility = (df['high'] - df['low']) / df['volume']
    
    # Rolling 14-Day High-to-Low Range Sum
    high_low_range_14d_sum = df['high'] - df['low']
    high_low_range_14d_sum = high_low_range_14d_sum.rolling(window=14).sum()
    
    # Difference between Consecutive VWAP
    vwap_diff = vwap.diff()
    
    # Adjust by Intraday Volatility and 14-Day High-to-Low Range Sum
    range_momentum = vwap_diff / (intraday_volatility * high_low_range_14d_sum)
    
    # Calculate Volume-Adjusted Momentum
    vwap_ema_20 = vwap.ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = vwap - vwap_ema_20
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=10).std()
    vol_adjusted_momentum = volume_adjusted_momentum / recent_volatility
    
    # 20-Day Moving Average of Daily High-Low Ranges
    high_low_range_20d_ma = (df['high'] - df['low']).rolling(window=20).mean()
    
    # 20-Day Moving Average of Daily Open-Close Ranges
    open_close_range_20d_ma = (df['open'] - df['close']).rolling(window=20).mean()
    
    # Subtract Average of 20-Day Moving Averages
    avg_20d_ma = (high_low_range_20d_ma + open_close_range_20d_ma) / 2
    final_factor = vol_adjusted_momentum - avg_20d_ma
    
    return final_factor
```
