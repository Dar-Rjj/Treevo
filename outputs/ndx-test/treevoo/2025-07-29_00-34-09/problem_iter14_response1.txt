```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Exponential Moving Average of Price Change
    df['ema_5'] = df['daily_change'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['daily_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = (df['ema_5'] > df['ema_10']).astype(int) - (df['ema_5'] < df['ema_10']).astype(int)
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)  # Example threshold: 75th percentile
    df['filtered_signal'] = df['reversal_signal'] * (df['volume'] > volume_threshold)
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    df['vap_ema_20'] = df['vap'].ewm(span=20, adjust=False).mean()
    df['vam'] = df['vap'] - df['vap_ema_20']
    
    # Incorporate Recent Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility_10'] = df['daily_return'].rolling(window=10).std()
    df['inv_volatility'] = 1 / df['volatility_10']
    df['vam_adjusted'] = df['vam'] * df['inv_volatility']
    
    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['hl_volatility_20'] = df['high_low_range'].rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    df['oc_range'] = abs(df['open'] - df['close'])
    df['oc_volatility_20'] = df['oc_range'].rolling(window=20).std()
    
    # Compute 30-Day Rolling Window of High-Low Range and its Volatility
    df['hl_volatility_30'] = df['high_low_range'].rolling(window=30).std()
    df['avg_hl_volatility'] = (df['hl_volatility_20'] + df['hl_volatility_30']) / 2
    
    # Combine Crossover Momentum and Composite Volatility
    df['composite_volatility'] = (df['hl_volatility_20'] + df['oc_volatility_20'] + df['avg_hl_volatility']) / 3
    df['alpha_factor'] = df['vam_adjusted'] - df['composite_volatility']
    
    return df['alpha_factor'].dropna()
```
