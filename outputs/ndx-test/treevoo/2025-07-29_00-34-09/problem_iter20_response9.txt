```python
import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the daily trading range
    df['trading_range'] = df['high'] - df['low']
    
    # Calculate the volume weighted average price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the factor value
    df['alpha_factor'] = (df['price_change'] / df['trading_range']) * (df['VWAP'] / df['close'])
    
    return df['alpha_factor']
```
