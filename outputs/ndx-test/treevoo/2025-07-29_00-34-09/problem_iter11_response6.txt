```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low'] * (df['volume'] / (df['high'] + df['low']))
    df['sustained_5day_momentum'] = (df['close'].diff(1) / df['close'].shift(1)).rolling(window=5).mean()
    df['intraday_5day_combined'] = (df['intraday_momentum'] * df['amount'] + df['sustained_5day_momentum'] * df['volume'])
    
    # Calculate Rolling Median of High-to-Low Range
    df['high_low_median_14d'] = df['high_low_range'].rolling(window=14).median()
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Exponential Moving Average of Price Change
    df['ema_5d'] = df['daily_price_change'].ewm(span=5, adjust=False).mean()
    df['ema_10d'] = df['daily_price_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ema_5d'] > df['ema_10d'], 'reversal_signal'] = 1
    df.loc[df['ema_5d'] < df['ema_10d'], 'reversal_signal'] = -1
    df['reversal_signal'] = df['reversal_signal'] * (df['high_low_median_14d'] / df['high_low_median_14d'].mean())
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['reversal_signal'] = df['reversal_signal'] * (df['volume'] > volume_threshold).astype(int)
    
    # Calculate Upward and Downward Volume-Weighted Returns
    df['positive_return'] = df['daily_price_change'].clip(lower=0) * df['volume']
    df['negative_return'] = df['daily_price_change'].clip(upper=0).abs() * df['volume']
    df['sum_positive_return_14d'] = df['positive_return'].rolling(window=14).sum()
    df['sum_negative_return_14d'] = df['negative_return'].rolling(window=14).sum()
    
    # Compute RSI
    df['relative_strength'] = df['sum_positive_return_14d'] / df['sum_negative_return_14d']
    df['rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Calculate Weighted Daily Return
    df['weighted_daily_return'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Cumulative Weighted Momentum
    df['cumulative_weighted_momentum'] = df['weighted_daily_return'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['intraday_5day_combined'] 
                          + df['cumulative_weighted_momentum'] 
                          + df['rsi'] 
                          + df['reversal_signal'])
    
    return df['alpha_factor']
```
