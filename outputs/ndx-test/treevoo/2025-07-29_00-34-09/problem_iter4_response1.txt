```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Rolling Average of High-to-Low Range with a 14-day window
    high_to_low_momentum = high_to_low_range.rolling(window=14).mean()
    
    return high_to_low_momentum
```
