```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return using close price
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate weighted momentum
    df['weighted_momentum'] = (df['daily_return'] * df['volume']).rolling(window=5).sum()
    
    # Return the factor values as a pandas Series
    return df['weighted_momentum']
```
