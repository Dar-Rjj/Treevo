import pandas as pd

def heuristics_v2(df):
    # Compute Logarithmic Return using Close price
    df['log_return'] = df['close'].pct_change().apply(lambda x: np.log(1 + x))
    
    # Multiply by Volume to get Volume-Weighted Return
    df['volume_weighted_return'] = df['log_return'] * df['volume']
    
    # Identify Upward and Downward Volume-Weighted Returns
    df['positive_vwr'] = df['volume_weighted_return'].where(df['volume_weighted_return'] > 0, 0)
    df['negative_vwr'] = df['volume_weighted_return'].where(df['volume_weighted_return'] < 0, 0).abs()
    
    # Sum Positive and Negative Volume-Weighted Returns over 14 days
    df['sum_positive_vwr_14'] = df['positive_vwr'].rolling(window=14).sum()
    df['sum_negative_vwr_14'] = df['negative_vwr'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['sum_positive_vwr_14'] / df['sum_negative_vwr_14']
    
    # Apply RSI Formula
    df['vwr_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Return the factor values
    return df['vwr_rsi']
