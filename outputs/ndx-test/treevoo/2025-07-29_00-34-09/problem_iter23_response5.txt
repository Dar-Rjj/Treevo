```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3
    
    # Generate Intraday Volatility Metric
    intraday_volatility = (df['high'] - df['low']) / df['volume']
    
    # Calculate 14-Day High-to-Low Range Sum
    high_low_range_14d_sum = (df['high'] - df['low']).rolling(window=14).sum()
    
    # Calculate Difference between Consecutive VWAP
    vwap_diff = vwap.diff()
    
    # Adjust by Intraday Volatility and 14-Day High-to-Low Range Sum
    high_to_low_momentum = vwap_diff / (intraday_volatility * high_low_range_14d_sum)
    
    # Calculate 20-day EMA of VWAP
    vwap_20ema = vwap.ewm(span=20, adjust=False).mean()
    
    # Subtract 20-day EMA from current VWAP for Volume-Adjusted Momentum
    volume_adjusted_momentum = vwap - vwap_20ema
    
    # Calculate 10-day Standard Deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_10d = daily_returns.rolling(window=10).std()
    
    # Incorporate Recent Volatility: Multiply Momentum by Inverse of 10-day Standard Deviation
    recent_volatility_adjusted_momentum = volume_adjusted_momentum * (1 / std_dev_10d)
    
    # 20-Day Moving Average of Daily High-Low Ranges
    ma_high_low_20d = (df['high'] - df['low']).rolling(window=20).mean()
    
    # 20-Day Moving Average of Daily Open-Close Ranges
    ma_open_close_20d = (df['open'] - df['close']).rolling(window=20).mean()
    
    # Subtract Average of 20-Day Moving Averages
    average_ma_20d = (ma_high_low_20d + ma_open_close_20d) / 2
    
    # Calculate 14-Day Momentum
    momentum_14d = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # Calculate 60-Day Momentum
    momentum_60d = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # Combine Volume-Adjusted Momentum and Enhanced Composite Volatility
    alpha_factor = recent_volatility_adjusted_momentum - average_ma_20d + momentum_14d + momentum_60d
    
    return alpha_factor
```
