```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Weighted Price
    weighted_price = (df['high'] + df['low'] + df['close']) * df['volume']
    
    # Short-term EMA of Weighted Price (e.g., 5 days)
    short_ema = weighted_price.ewm(span=5, adjust=False).mean()
    
    # Long-term EMA of Weighted Price (e.g., 20 days)
    long_ema = weighted_price.ewm(span=20, adjust=False).mean()
    
    # Smoothed Momentum
    smoothed_momentum = short_ema - long_ema
    
    return smoothed_momentum
```
