import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the average volume over the last 10 days
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the range of the day (high - low)
    df['range'] = df['high'] - df['low']
    
    # Calculate the percentage of the close price within the day's range
    df['close_pct_range'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the weighted average of the daily return and the percentage of the close price within the range
    # We use the average volume as a weight to give more importance to days with higher volume
    df['weighted_return'] = (df['return'] * df['avg_volume_10d']) + (df['close_pct_range'] * df['avg_volume_10d'])
    
    # The final alpha factor is the rolling sum of the weighted returns
    df['alpha_factor'] = df['weighted_return'].rolling(window=20).sum()
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
