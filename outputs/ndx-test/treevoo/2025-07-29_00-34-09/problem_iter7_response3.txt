```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Typical Price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Weight Typical Price by Volume to get Volume Adjusted Price (VAP)
    vap = typical_price * df['volume']
    
    # Calculate 20-day Exponential Moving Average (EMA) of VAP
    ema_20 = vap.ewm(span=20, adjust=False).mean()
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = vap - ema_20
    
    # Calculate 10-day Standard Deviation of daily returns for volatility
    daily_returns = df['close'].pct_change()
    volatility_10 = daily_returns.rolling(window=10).std()
    inverse_volatility = 1 / volatility_10
    
    # Incorporate Recent Volatility
    momentum_volatility_adjusted = volume_adjusted_momentum * inverse_volatility
    
    # Compute Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute 20-Day Moving Standard Deviation of High-Low Ranges
    high_low_volatility_20 = high_low_range.rolling(window=20).std()
    
    # Combine Volume Adjusted Momentum and High-Low Range Volatility
    alpha_factor = momentum_volatility_adjusted - high_low_volatility_20
    
    return alpha_factor
```
