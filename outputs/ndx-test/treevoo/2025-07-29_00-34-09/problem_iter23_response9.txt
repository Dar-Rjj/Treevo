```python
def heuristics_v2(data):
    """
    Generate an alpha factor based on the given market data.
    
    Parameters:
        data (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
        
    Returns:
        pd.Series: A Series representing the alpha factor values indexed by date.
    """
    # Calculate the average of the high and low prices
    avg_price = (data['high'] + data['low']) / 2
    
    # Calculate the price change from the previous day's close to the current day's average price
    price_change = avg_price - data['close'].shift(1)
    
    # Calculate the volume change from the previous day's volume to the current day's volume
    volume_change = data['volume'] - data['volume'].shift(1)
    
    # Heuristic: The factor is the product of the price change and the volume change
    factor = price_change * volume_change
    
    return factor
```
