```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of high and low prices, using volume as weights
    weighted_avg = (df['high'] + df['low']) / 2 * df['volume']
    
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    
    # Generate a factor based on the interaction between the weighted average and price difference
    factor_values = (weighted_avg * price_diff).rolling(window=5).mean()
    
    return factor_values
```
