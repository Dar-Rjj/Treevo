import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['Range'] = df['high'] - df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    df['HighLowRatio'] = (df['high'] - df['low']) / df['low']
    df['IntradayActivity'] = df['volume'] / (df['high'] + df['low'])
    df['IntradayMomentum'] = df['HighLowRatio'] * df['IntradayActivity']
    
    df['5DayMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5DayMomentum'] = df['5DayMomentum'].rolling(window=5).mean()
    df['CombinedMomentum'] = df['IntradayMomentum'] * df['amount'].rolling(window=5).mean() + df['5DayMomentum'] * df['volume']
    
    # Calculate Daily Price Change
    df['DailyChange'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Exponential Moving Averages of Price Change
    df['5DayEMA'] = df['DailyChange'].ewm(span=5, adjust=False).mean()
    df['10DayEMA'] = df['DailyChange'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['ReversalSignal'] = 0
    df.loc[df['5DayEMA'] > df['10DayEMA'], 'ReversalSignal'] = 1
    df.loc[df['5DayEMA'] < df['10DayEMA'], 'ReversalSignal'] = -1
    
    df['14DayRollingMedianRange'] = df['Range'].rolling(window=14).median()
    df['AmplifiedReversalSignal'] = df['ReversalSignal'] * df['14DayRollingMedianRange']
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['FilteredSignal'] = df['AmplifiedReversalSignal'] * (df['volume'] > volume_threshold)
    
    # Calculate Cumulative Weighted Momentum
    df['WeightedDailyReturn'] = df['DailyChange'] / df['amount']
    df['CumulativeWeightedMomentum'] = df['WeightedDailyReturn'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()
    
    # Incorporate Volatility into Alpha Factor
    df['21DayVolatility'] = df['close'].rolling(window=21).std()
    df['VolatilityAdjustedFactor'] = df['CombinedMomentum'] / df['21DayVolatility']
    
    # Enhance Momentum with Volume-Weighted Moving Averages
    df['5DayVWMA'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['10DayVWMA'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['5DaySMA'] = df['close'].rolling(window=5).mean()
    df['10DaySMA'] = df['close'].rolling(window=10).mean()
    
    df['VWMASignal'] = 0
    df.loc[df['5DayVWMA'] > df['5DaySMA'], 'VWMASignal'] = 1
    df.loc[df['10DayVWMA'] < df['10DaySMA'], 'VWMASignal'] = -1
    df['VWMASignal'] = df['VWMASignal'] * df['CombinedMomentum']
    
    # Incorporate Trend Strength Indicator
    df['RSI'] = 100 - (100 / (1 + (df['close'].diff().rolling(window=14).mean() / abs(df['close'].diff()).rolling(window=14).mean())))
    
    df['TrendStrength'] = 0
    df.loc[df['RSI'] > 70, 'TrendStrength'] = -1
    df.loc[df['RSI'] < 30, 'TrendStrength'] = 1
    df['TrendStrength'] = df['TrendStrength'] * df['VolatilityAdjustedFactor']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['CombinedMomentum'] +
        df['CumulativeWeightedMomentum']
