```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, weighted by Volume
    df['volume_weighted_log_returns'] = df['log_returns'] * df['volume']
    df['ema_volume_weighted_returns'] = df['volume_weighted_log_returns'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=20).std()
    df['volume_adjusted_price_movement'] = df['ema_volume_weighted_returns'] / df['high_low_range_volatility']
    
    # Rolling Average of High-to-Low Range
    df['average_high_low_range_14d'] = df['high_low_range'].rolling(window=14).mean()
    
    # Determine Reversal Signal
    df['close_open_diff'] = df['close'] - df['open']
    df['5_day_ma_close_open'] = df['close_open_diff'].rolling(window=5).mean()
    df['10_day_ma_close_open'] = df['close_open_diff'].rolling(window=10).mean()
    df['reversal_signal'] = np.where(df['5_day_ma_close_open'] > df['10_day_ma_close_open'], 1,
                                     np.where(df['5_day_ma_close_open'] < df['10_day_ma_close_open'], -1, 0))
    df['reversal_signal'] *= df['average_high_low_range_14d']
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversals_20d'] = df['price_reversal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['ema_volume_weighted_close_20d'] = df['volume_weighted_close'].ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['alpha_factor'] = (
        0.3 * df['volume_adjusted_price_movement'] +
        0.2 * df['reversal_signal'] +
        0.3 * df['sum_price_reversals_20d'] +
        0.2 * df['ema_volume_weighted_close_20d']
    )
    
    return df['alpha_factor']
```
