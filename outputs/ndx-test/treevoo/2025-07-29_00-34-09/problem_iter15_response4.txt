```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['low'] * (df['volume'] / (df['high'] + df['low']))
    df['sustained_5d_momentum'] = (df['close'].diff() / df['close'].shift(1)).rolling(window=5).mean()
    df['combined_momentum'] = (df['intraday_momentum'] * df['amount'].rolling(window=5).mean()) + (df['sustained_5d_momentum'] * df['volume'])
    
    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Exponential Moving Average of Price Change
    df['ema_5d'] = df['daily_change'].ewm(span=5, adjust=False).mean()
    df['ema_10d'] = df['daily_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = np.where(df['ema_5d'] > df['ema_10d'], 1, np.where(df['ema_5d'] < df['ema_10d'], -1, 0))
    df['rolling_median_range'] = df['daily_range'].rolling(window=14).median()
    df['reversal_signal'] = df['reversal_signal'] * (1 + (df['rolling_median_range'] - df['rolling_median_range'].mean()) / df['rolling_median_range'].std())
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['filtered_reversal_signal'] = np.where(df['volume'] > volume_threshold, df['reversal_signal'], 0)
    
    # Calculate Weighted Daily Return
    df['weighted_daily_return'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Calculate Cumulative Weighted Momentum
    df['cumulative_weighted_momentum'] = df['weighted_daily_return'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()
    
    # Incorporate Volatility into Alpha Factor
    df['historical_volatility'] = df['close'].rolling(window=21).std()
    df['volatility_adjusted_factor'] = df['combined_momentum'] / df['historical_volatility']
    
    # Incorporate Market Sentiment
    df['ma_5d_open'] = df['open'].rolling(window=5).mean()
    df['ema_21d_close'] = df['close'].ewm(span=21, adjust=False).mean()
    sentiment_adjustment = np.where(df['ma_5d_open'] > df['ema_21d_close'], 1.2, 0.8)
    df['sentiment_adjusted_factor'] = df['volatility_adjusted_factor'] * sentiment_adjustment
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_momentum'] + 
                                df['cumulative_weighted_momentum'] + 
                                df['filtered_reversal_signal'] + 
                                df['sentiment_adjusted_factor'])
    
    return df['final_alpha_factor']
```
