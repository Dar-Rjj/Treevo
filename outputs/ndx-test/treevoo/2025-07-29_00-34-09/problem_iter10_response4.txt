```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Adjusted Return
    df['adjusted_return'] = (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1)) * 100
    
    # Apply Volume Smoothing
    df['volume_ewma'] = df['volume'].ewm(span=7, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['volume_ewma']
    
    # Weighted Returns Calculation
    df['weighted_return'] = df['adjusted_return'] * df['normalized_volume']
    
    # Separate Positive and Negative Weights
    positive_weighted_returns = df['weighted_return'].where(df['weighted_return'] > 0, 0)
    negative_weighted_returns = df['weighted_return'].where(df['weighted_return'] < 0, 0).abs()
    
    # Sum Over 14 Days
    positive_sum = positive_weighted_returns.rolling(window=14).sum()
    negative_sum = negative_weighted_returns.rolling(window=14).sum()
    
    # Compute Dynamic RSI
    relative_strength = positive_sum / negative_sum
    dynamic_rsi = 100 - (100 / (1 + relative_strength))
    
    return dynamic_rsi
```
