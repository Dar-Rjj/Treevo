```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume Adjusted Price (VAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vap = typical_price * df['volume']
    
    # Calculate Volume Adjusted Momentum
    vap_ema20 = vap.ewm(span=20, adjust=False).mean()
    momentum = vap - vap_ema20
    
    # Incorporate Recent Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_10day = daily_returns.rolling(window=10).std()
    inverse_volatility = 1 / std_10day
    adjusted_momentum = momentum * inverse_volatility
    
    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_std_20day = high_low_range.rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    open_close_range = (df['open'] - df['close']).abs()
    open_close_std_20day = open_close_range.rolling(window=20).std()
    
    # Combine Adjusted Momentum and Composite Volatility
    average_volatility = (high_low_std_20day + open_close_std_20day) / 2
    alpha_factor = adjusted_momentum - average_volatility
    
    return alpha_factor
```
