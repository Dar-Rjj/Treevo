    df['High_to_Low_Range'] = df['high'] - df['low']

    df['Intraday_Trading_Activity'] = df['volume'] / (df['high'] + df['low'])
    df['Intraday_Momentum'] = df['High_Low_Ratio'] * df['Intraday_Trading_Activity']

    df['Sequential_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_Day_Avg_Momentum'] = df['Sequential_Momentum'].rolling(window=5).mean()

    df['Daily_Price_Change'] = df['close'] - df['open']

    df['5_Day_EMA'] = df['Daily_Price_Change'].ewm(span=5, adjust=False).mean()
    df['10_Day_EMA'] = df['Daily_Price_Change'].ewm(span=10, adjust=False).mean()

    df['Reversal_Signal'] = 0
    df.loc[df['5_Day_EMA'].shift(1) < df['10_Day_EMA'].shift(1), 'Reversal_Signal'] = 1
    df.loc[df['5_Day_EMA'].shift(1) > df['10_Day_EMA'].shift(1), 'Reversal_Signal'] = -1
