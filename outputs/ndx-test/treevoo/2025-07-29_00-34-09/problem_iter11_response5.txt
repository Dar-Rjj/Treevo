```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['daily_range'] = df['high'] - df['low']

    # Calculate Rolling Median of High-to-Low Range
    df['rolling_median_range'] = df['daily_range'].rolling(window=14).median()

    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']

    # Compute 5-Day and 10-Day Exponential Moving Average of Price Change
    df['ema_5_price_change'] = df['price_change'].ewm(span=5, adjust=False).mean()
    df['ema_10_price_change'] = df['price_change'].ewm(span=10, adjust=False).mean()

    # Determine Reversal Signal
    df['reversal_signal'] = (df['ema_5_price_change'] > df['ema_10_price_change']) * 1
    df['reversal_signal'] *= (df['rolling_median_range'] / df['rolling_median_range'].max())

    # Enhanced Volume Filter
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_filter'] = (df['volume'] > df['avg_volume_20']) * 1

    # Calculate Volume-Weighted Price (VWP)
    df['vwp'] = (df['high'] + df['low']) / 2  # Simplified VWP for this example

    # Compute Momentum from VWP
    df['vwp_ema_20'] = df['vwp'].ewm(span=20, adjust=False).mean()
    df['momentum_vwp'] = df['vwp'] - df['vwp_ema_20']

    # Incorporate Intraday Range Volatility
    df['intraday_range_volatility'] = (df['high'] - df['low']) / df['avg_volume_20']

    # Assess Momentum Based on Close Price
    df['close_momentum'] = df['close'] - df['close'].rolling(window=20).mean()

    # Analyze Price Reversals Using High and Low Prices
    df['price_reversals'] = (df['high'] - df['low']) / df['open']

    # Evaluate Volume Impact on Price Movement
    df['vwap_20_ema'] = df['vwp'].ewm(span=20, adjust=False).mean()

    # Combine Factors for Final Alpha
    df['alpha_factor'] = (
        (df['momentum_vwp'] / df['intraday_range_volatility']) +
        df['close_momentum'] +
        df['price_reversals'] +
        df['vwap_20_ema']
    ) * df['volume_filter'] * df['reversal_signal']

    return df['alpha_factor']
```
