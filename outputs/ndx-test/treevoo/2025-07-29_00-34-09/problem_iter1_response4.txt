```python
import pandas as pd

def heuristics_v2(df, volume_threshold=1e6):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day Moving Average of Price Change
    df['5_day_ma'] = df['price_change'].rolling(window=5).mean()
    
    # Compute 10-Day Moving Average of Price Change
    df['10_day_ma'] = df['price_change'].rolling(window=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['5_day_ma'] > df['10_day_ma'], 'reversal_signal'] = 1
    df.loc[df['5_day_ma'] < df['10_day_ma'], 'reversal_signal'] = -1
    
    # Filter by Volume
    df['reversal_signal'] = df.apply(lambda row: row['reversal_signal'] if row['volume'] > volume_threshold else 0, axis=1)
    
    return df['reversal_signal']
```
