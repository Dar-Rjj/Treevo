import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # 20-Day EMA of Log Returns
    df['ema_log_returns'] = df['log_returns'].ewm(span=20, adjust=False).mean()
    
    # Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = df['ema_log_returns'] * (df['volume'] / df['volume'].rolling(window=20).mean())
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # 20-Day Moving SD of High-Low Ranges
    df['high_low_volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Daily Open to Previous Close Difference
    df['open_to_prev_close_diff'] = df['open'] - df['close'].shift(1)
    
    # 20-Day Moving SD of Open to Previous Close Differences
    df['opening_gap_volatility'] = df['open_to_prev_close_diff'].rolling(window=20).std()
    
    # Integrate Price Action and Volume
    df['alpha_factor'] = (df['volume_adjusted_momentum'] 
                          + df['high_low_volatility'] 
                          + df['opening_gap_volatility'])
    
    return df['alpha_factor'].dropna()
