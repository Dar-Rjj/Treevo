```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily return using close price
    df['daily_return'] = df['close'].diff()
    
    # Calculate weighted momentum with volume
    df['volume_weighted_momentum'] = (df['daily_return'] * df['volume']).rolling(window=5).sum()
    
    # Calculate weighted momentum with amount
    df['amount_weighted_momentum'] = (df['daily_return'] * df['amount']).rolling(window=5).sum()
    
    # Combine both weighted momentums
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['amount_weighted_momentum']
    
    # Return the combined momentum as the factor
    return df['combined_momentum'].dropna()
```
