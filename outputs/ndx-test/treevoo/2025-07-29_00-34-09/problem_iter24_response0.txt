import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    intraday_range = df['high'] - df['low']
    median_volume = df['volume'].rolling(window=14).median()
    intraday_volatility = intraday_range / median_volume
    
    # Integrate High-to-Low Range Momentum
    high_low_range_sum_20 = (df['high'] - df['low']).rolling(window=20).sum()
    vwp_diff = vwp.diff()
    adjusted_momentum = vwp_diff / (intraday_volatility * high_low_range_sum_20)
    
    # Calculate Volume-Adjusted Momentum
    vwp_ema_25 = vwp.ewm(span=25, adjust=False).mean()
    volume_adjusted_momentum = vwp - vwp_ema_25
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=15).std()
    momentum_adjusted = volume_adjusted_momentum / (1 / recent_volatility)
    
    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=30).mean()
    
    # Calculate Open-Close Range Volatility
    open_close_range = df['open'] - df['close']
    open_close_volatility = open_close_range.rolling(window=30).mean()
    
    # Combine Volume-Adjusted Momentum and Enhanced Composite Volatility
    weighted_average_volatility = 0.7 * high_low_volatility + 0.3 * open_close_volatility
    combined_momentum = momentum_adjusted - weighted_average_volatility
    
    # Explore Momentum and Mean Reversion through Price Trends
    sma_7 = df['close'].rolling(window=7).mean()
    sma_28 = df['close'].rolling(window=28).mean()
    
    # Incorporate Volume-Weighted Price Actions
    total_volume_7 = df['volume'].rolling(window=7).sum()
    total_volume_28 = df['volume'].rolling(window=28).sum()
    vwap_7 = (df['close'] * df['volume']).rolling(window=7).sum() / total_volume_7
    vwap_28 = (df['close'] * df['volume']).rolling(window=28).sum() / total_volume_28
    
    # Investigate Recent Price Volatility
    atr_7 = (df['high'] - df['low']).rolling(window=7).mean()
    atr_28 = (df['high'] - df['low']).rolling(window=28).mean()
    
    # Analyze Trading Volume Patterns
    roc_volume_7 = df['volume'].pct_change(periods=7)
    roc_volume_28 = df['volume'].pct_change(periods=28)
    above_avg_volume_7 = df['volume'] > df['volume'].rolling(window=7).mean()
    above_avg_volume_28 = df['volume'] > df['volume'].rolling(window=28).mean()
    
    # Evaluate Price Momentum
    momentum_14 = df['close'] - df['close'].shift(14)
    momentum_42 = df['close'] - df['close'].shift(42)
    roc_close_14 = df['close'].pct_change(periods=14)
    roc_close_42 = df['close'].pct_change(periods=42)
    
    # Assess Relative Strength
    rsi_21 = 100 - (100 / (1 + (df['close'].diff().clip(lower=0).rolling(window=21).mean() / df['close'].diff().clip(upper=0).rolling(window=21).mean())))
    
    # Final alpha factor
    alpha_factor = combined_momentum + 0.5 * (momentum_14 + momentum_42) + 0.5 * (roc_close_14 + roc_close_42) + rsi_21
    return alpha_factor
