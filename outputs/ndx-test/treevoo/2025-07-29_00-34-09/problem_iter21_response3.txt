```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    df['HighLowRatio'] = (df['high'] - df['low']) / df['low']
    df['IntradayActivity'] = df['volume'] / (df['high'] + df['low'])
    df['IntradayMomentum'] = df['HighLowRatio'] * df['IntradayActivity']
    
    # Measure Sustained 5-Day Momentum
    df['SequentialMomentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5DayMomentum'] = df['SequentialMomentum'].rolling(window=5).sum() / 5
    
    # Combine Intraday and 5-Day Momentum
    df['CombinedMomentum'] = (df['IntradayMomentum'] * df['5DayMomentum'] * df['volume']) + (df['5DayMomentum'] * df['volume'])
    
    # Calculate Daily Price Change
    df['DailyChange'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Exponential Moving Averages of Price Change
    df['5DayEMA'] = df['DailyChange'].ewm(span=5, adjust=False).mean()
    df['10DayEMA'] = df['DailyChange'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['ReversalSignal'] = np.where(df['5DayEMA'] > df['10DayEMA'], 1, -1)
    df['14DayRollingMedianRange'] = df['DailyRange'].rolling(window=14).median()
    df['ReversalSignal'] = df['ReversalSignal'] * (df['14DayRollingMedianRange'] / df['14DayRollingMedianRange'].std())
    
    # Filter by Volume
    df['VolumeThreshold'] = df['volume'].quantile(0.75)
    df['FilteredReversalSignal'] = np.where(df['volume'] > df['VolumeThreshold'], df['ReversalSignal'], 0)
    
    # Calculate Cumulative Weighted Momentum
    df['WeightedDailyReturn'] = df['SequentialMomentum'] * df['amount']
    df['CumulativeWeightedMomentum'] = df['WeightedDailyReturn'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()
    
    # Incorporate Volatility into Alpha Factor
    df['21DayVolatility'] = df['close'].rolling(window=21).std()
    df['VolatilityAdjustedFactor'] = df['CumulativeWeightedMomentum'] / df['21DayVolatility']
    
    # Enhance Momentum with Volume-Weighted Moving Averages
    df['5DayVWMA'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['10DayVWMA'] = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['5DaySMA'] = df['close'].rolling(window=5).mean()
    df['10DaySMA'] = df['close'].rolling(window=10).mean()
    df['VWMAEnhancedMomentum'] = np.where(df['5DayVWMA'] > df['5DaySMA'], 1, -1) * np.where(df['10DayVWMA'] < df['10DaySMA'], -1, 1)
    
    # Final Alpha Factor
    df['AlphaFactor'] = (df['CombinedMomentum'] + 
                         df['CumulativeWeightedMomentum'] + 
                         df['FilteredReversalSignal'] + 
                         df['VolatilityAdjustedFactor'] + 
                         df['VWMAEnhancedMomentum'])
    
    return df['AlphaFactor']
```
