```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range
    df['range'] = df['high'] - df['low']
    
    # Compute the relative position of close to the daily range
    df['close_position'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Combine the relative position and the volume change for the alpha factor
    df['alpha_factor'] = df['close_position'] * df['volume_change']
    
    return df['alpha_factor']
```
