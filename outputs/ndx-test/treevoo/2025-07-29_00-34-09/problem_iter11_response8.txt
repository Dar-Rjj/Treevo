def heuristics_v2(data):
    """
    Generate an alpha factor based on a set of heuristics.
    This function computes the average of the positive and negative differences between
    the close and open prices, and then divides this by the volume to create a novel
    factor. The idea is to capture the momentum and volatility relative to the trading
    volume.

    Parameters:
    data (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.

    Returns:
    pd.Series: A Series representing the computed factor values, indexed by date.
    """
    # Calculate the difference between close and open prices
    diff_close_open = data['close'] - data['open']
    
    # Separate positive and negative differences
    pos_diff = diff_close_open[diff_close_open > 0]
    neg_diff = diff_close_open[diff_close_open < 0]
    
    # Compute the average of the positive and negative differences
    avg_pos_diff = pos_diff.mean()
    avg_neg_diff = neg_diff.mean()
    
    # Combine the averages
    combined_avg = (avg_pos_diff + avg_neg_diff) / 2
    
    # Normalize the combined average by the volume
    factor_values = combined_avg / data['volume']
    
    return factor_values
