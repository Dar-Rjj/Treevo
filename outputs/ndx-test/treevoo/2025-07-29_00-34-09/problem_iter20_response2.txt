```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, weighted by Volume
    df['volume_weighted_log_return'] = df['log_return'] * df['volume']
    df['ema_20_vol_adjusted'] = df['volume_weighted_log_return'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_volatility'] = df['high_low_range'].rolling(window=20).std()
    df['volatility_weighted_ema'] = df['ema_20_vol_adjusted'] / df['high_low_volatility']
    
    # 10-day Simple Moving Average (SMA) of the Close Price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Momentum Factor
    df['momentum_factor'] = df['close'] - df['sma_10']
    
    # Identify High-Activity Days
    df['volume_30_sma'] = df['volume'].rolling(window=30).mean()
    df['is_high_activity_day'] = df['volume'] > df['volume_30_sma']
    
    # Calculate Percentage Change in Close Price on High-Activity Days
    df['percentage_change'] = df['close'].pct_change()
    df['adjusted_momentum_score'] = df['momentum_factor'] * df['percentage_change']
    df['adjusted_momentum_score'] = df['adjusted_momentum_score'].where(df['is_high_activity_day'], 0)
    
    # Intraday Market Sentiment
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['directional_bias'] = np.where(df['close'] > df['midpoint'], 1, -1)
    
    # Composite Sentiment-Adjusted Momentum
    df['composite_sentiment_momentum'] = df['momentum_factor'] * df['directional_bias']
    
    # Filter by Volume
    df['composite_sentiment_momentum'] = df['composite_sentiment_momentum'].where(df['volume'] > df['volume_30_sma'], 0)
    
    # Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversal_20'] = df['price_reversal'].rolling(window=20).sum()
    
    # Volume-Weighted Close
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['final_alpha'] = (
        0.4 * df['volatility_weighted_ema'] +
        0.1 * df['sum_price_reversal_20'] +
        0.3 * df['adjusted_momentum_score'] +
        0.1 * df['volume_weighted_close'] +
        0.1 * df['composite_sentiment_momentum']
    )
    
    return df['final_alpha']
```
