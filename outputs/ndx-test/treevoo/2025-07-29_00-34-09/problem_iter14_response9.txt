```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day EMA of Price Change
    df['ema_5'] = df['daily_change'].ewm(span=5).mean()
    df['ema_10'] = df['daily_change'].ewm(span=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = np.where(df['ema_5'] > df['ema_10'], 1, np.where(df['ema_5'] < df['ema_10'], -1, 0))
    
    # Filter by Volume (above a certain threshold)
    volume_threshold = df['volume'].quantile(0.75)  # Example threshold: 75th percentile
    df['filtered_reversal_signal'] = np.where(df['volume'] > volume_threshold, df['reversal_signal'], 0)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['intraday_range'] / df['volume'].rolling(window=14).mean()
    
    # Integrate High-to-Low Range Momentum
    df['high_low_avg_14'] = df['high'] - df['low']
    df['high_low_avg_14'] = df['high_low_avg_14'].rolling(window=14).mean()
    df['vwap_adjusted'] = df['vwap'].diff() / df['intraday_volatility'] / df['high_low_avg_14']
    
    # Volume-Adjusted Smoothed Momentum
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['ema_log_returns_20'] = df['log_returns'].ewm(span=20).mean()
    df['volume_adjusted_momentum'] = df['volume'] * df['ema_log_returns_20']
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    df['vap_ema_20'] = df['vap'].ewm(span=20).mean()
    df['volume_adjusted_momentum'] = df['vap'] - df['vap_ema_20']
    
    # Incorporate Recent Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['std_10'] = df['daily_returns'].rolling(window=10).std()
    df['inverse_volatility'] = 1 / df['std_10']
    df['adjusted_momentum'] = df['volume_adjusted_momentum'] * df['inverse_volatility']
    
    # Calculate High-Low Range Volatility
    df['high_low_range_volatility'] = df['high'] - df['low']
    df['high_low_range_volatility_20'] = df['high_low_range_volatility'].rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    df['open_close_range_volatility'] = df['open'] - df['close']
    df['open_close_range_volatility_20'] = df['open_close_range_volatility'].rolling(window=20).std()
    
    # Combine Crossover Momentum and Composite Volatility
    df['composite_volatility'] = (df['high_low_range_volatility_20'] + df['open_close_range_volatility_20']) / 2
    df['final_alpha_factor'] = df['adjusted_momentum'] - df['composite_volatility'] + df['vwap_adjusted']
    
    return df['final_alpha_factor']
```
