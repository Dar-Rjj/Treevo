```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # 20-Day EMA of Log Returns Weighed by Volume
    df['vol_weighted_log_returns'] = df['log_returns'] * df['volume']
    df['vol_weighted_ema'] = df['vol_weighted_log_returns'].ewm(span=20).mean()
    
    # Weight by High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_volatility'] = df['high_low_range'].rolling(window=20).std()
    df['vol_adjusted_price_movement'] = df['vol_weighted_ema'] / df['high_low_volatility']
    
    # Calculate Rolling Average of High-to-Low Range (14-day window)
    df['rolling_high_low_avg'] = df['high_low_range'].rolling(window=14).mean()
    
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Moving Averages of Price Change
    df['ma_5d_price_change'] = df['price_change'].rolling(window=5).mean()
    df['ma_10d_price_change'] = df['price_change'].rolling(window=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ma_5d_price_change'] > df['ma_10d_price_change'], 'reversal_signal'] = 1
    df.loc[df['ma_5d_price_change'] < df['ma_10d_price_change'], 'reversal_signal'] = -1
    df['reversal_signal'] = df['reversal_signal'] * (df['rolling_high_low_avg'] / df['rolling_high_low_avg'].max())
    
    # Filter by Volume (Volume above a certain threshold, e.g., 90th percentile)
    volume_threshold = df['volume'].quantile(0.90)
    df['reversal_signal_filtered'] = df['reversal_signal'] * (df['volume'] > volume_threshold)
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversals'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversals'] = df['price_reversals'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['final_alpha'] = (0.3 * df['vol_adjusted_price_movement'] + 
                        0.2 * df['reversal_signal_filtered'] + 
                        0.3 * df['sum_price_reversals'] + 
                        0.2 * df['volume_weighted_close'])
    
    return df['final_alpha']
```
