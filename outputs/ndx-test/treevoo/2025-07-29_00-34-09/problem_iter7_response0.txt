```python
import pandas as pd

def heuristics_v2(df):
    # Compute Adjusted Volume-Weighted Price
    df['volume_weighted_price'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Calculate Absolute Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust for Average Volume over Period (last 14 days)
    df['avg_volume_14_days'] = df['volume'].rolling(window=14).mean()
    df['intraday_volatility'] = df['intraday_range'] / df['avg_volume_14_days']
    
    # Calculate Rolling 14-Day High-to-Low Range Average
    df['high_low_range_14_day_avg'] = df['intraday_range'].rolling(window=14).mean()
    
    # Difference between Consecutive Volume-Weighted Prices
    df['diff_volume_weighted_price'] = df['volume_weighted_price'].diff()
    
    # Combine Metrics: Volume-Adjusted Price and Momentum
    df['alpha_factor'] = df['diff_volume_weighted_price'] * df['intraday_volatility'] * df['high_low_range_14_day_avg']
    
    return df['alpha_factor']
```
