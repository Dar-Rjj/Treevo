```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_ret'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weigh by Volume
    df['vol_weighted_log_ret'] = df['log_ret'] * df['volume']
    df['ema_vol_weighted_log_ret'] = df['vol_weighted_log_ret'].ewm(span=20, adjust=False).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Volume-Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['ema_vol_weighted_log_ret'] / df['high_low_range_volatility']
    
    # Rolling Average of High-to-Low Range (14-day window)
    df['avg_high_low_range_14d'] = df['high_low_range'].rolling(window=14).mean()
    
    # Daily Price Change
    df['daily_price_change'] = df['close'] - df['open']
    
    # 5-Day Moving Average of Price Change
    df['ma_price_change_5d'] = df['daily_price_change'].rolling(window=5).mean()
    
    # 10-Day Moving Average of Price Change
    df['ma_price_change_10d'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = np.where(df['ma_price_change_5d'] > df['ma_price_change_10d'], 1, 
                                     np.where(df['ma_price_change_5d'] < df['ma_price_change_10d'], -1, 0))
    
    # Amplify or dampen the reversal signal based on 14-day rolling average of high-low range
    df['reversal_signal_strength'] = df['reversal_signal'] * (df['avg_high_low_range_14d'] / df['avg_high_low_range_14d'].median())
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['filtered_reversal_signal'] = np.where(df['volume'] > volume_threshold, df['reversal_signal_strength'], 0)
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversal_20d'] = df['price_reversal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['ema_volume_weighted_close'] = df['volume_weighted_close'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Final Alpha
    df['final_alpha'] = (df['volume_adjusted_price_movement'] * 0.4 +
                        df['filtered_reversal_signal'] * 0.2 +
                        df['sum_price_reversal_20d'] * 0.3 +
                        df['ema_volume_weighted_close'] * 0.1)
    
    return df['final_alpha']
```
