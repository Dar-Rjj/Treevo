def heuristics_v2(data):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    data (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the price range
    price_range = data['high'] - data['low']
    
    # Calculate the closing price relative to the price range
    close_to_range = (data['close'] - data['low']) / price_range
    
    # Calculate the volume-weighted price
    vwap = (data['close'] + data['high'] + data['low']) / 3 * data['volume']
    
    # Calculate the moving average of the volume-weighted price
    vwap_ma = vwap.rolling(window=5).mean()
    
    # Calculate the difference between the current volume-weighted price and its moving average
    vwap_diff = vwap - vwap_ma
    
    # Combine the factors
    factor = close_to_range * vwap_diff
    
    return factor
