```python
import pandas as pd

def heuristics(df):
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Weighted Price
    weighted_price = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Smoothed Momentum using EMAs
    short_term_ema = weighted_price.ewm(span=5, adjust=False).mean()
    long_term_ema = weighted_price.ewm(span=20, adjust=False).mean()
    smoothed_momentum = long_term_ema - short_term_ema
    
    return smoothed_momentum
```
