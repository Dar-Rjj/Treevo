```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    df['high_low_ratio'] = (df['high'] - df['low']) / df['low']
    df['intraday_activity'] = df['volume'] / (df['high'] + df['low'])
    df['intraday_momentum'] = df['high_low_ratio'] * df['intraday_activity']
    
    # Measure Sustained 5-Day Momentum
    df['close_lag1'] = df['close'].shift(1)
    df['day_momentum'] = (df['close'] - df['close_lag1']) / df['close_lag1']
    df['sustained_5_day_momentum'] = df['day_momentum'].rolling(window=5).mean()
    
    # Combine Intraday and 5-Day Momentum
    df['intraday_momentum_adjusted'] = df['intraday_momentum'] * df['amount'].rolling(window=5).mean()
    df['sustained_5_day_momentum_adjusted'] = df['sustained_5_day_momentum'] * df['volume']
    df['combined_momentum'] = df['intraday_momentum_adjusted'] + df['sustained_5_day_momentum_adjusted']
    
    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Exponential Moving Averages of Price Change
    df['ema_5'] = df['daily_change'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['daily_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ema_5'] > df['ema_10'], 'reversal_signal'] = 1
    df.loc[df['ema_5'] < df['ema_10'], 'reversal_signal'] = -1
    
    # Consider High-to-Low Range
    df['high_low_range_median'] = df['daily_range'].rolling(window=14).median()
    df['reversal_signal'] = df['reversal_signal'] * (1 + (df['high_low_range_median'] - df['high_low_range_median'].mean()) / df['high_low_range_median'].std())
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['filtered_reversal_signal'] = df['reversal_signal'].where(df['volume'] > volume_threshold, 0)
    
    # Calculate Cumulative Weighted Momentum
    df['weighted_daily_return'] = df['daily_change'] / df['amount']
    df['cumulative_weighted_momentum'] = df['weighted_daily_return'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()
    
    # Incorporate Volatility into Alpha Factor
    df['volatility'] = df['close'].rolling(window=21).std()
    df['volatility_adjusted_factor'] = df['combined_momentum'] + df['cumulative_weighted_momentum'] + df['filtered_reversal_signal']
    df['final_alpha_factor'] = df['volatility_adjusted_factor'] * (1 / df['volatility'])
    
    return df['final_alpha_factor'].dropna()
```
