```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])

    # Compute 20-day Exponential Moving Average (EMA) of VWP
    vwp_ema_20 = vwp.ewm(span=20).mean()
    
    # Momentum from VWP
    vwp_momentum = vwp - vwp_ema_20
    
    # Calculate Absolute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Average Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    intraday_volatility = intraday_range / avg_volume
    
    # Assess Momentum Based on Close Price
    close_ma_20 = df['close'].rolling(window=20).mean()
    close_momentum = df['close'] - close_ma_20
    
    # Analyze Price Reversals Using High and Low Prices
    price_reversal = (df['high'] - df['low']) / df['open']
    
    # Evaluate Volume Impact on Price Movement
    vwap = (df['close'] * df['volume']).ewm(span=20).mean() / df['volume'].ewm(span=20).mean()

    # Combine Factors for Final Alpha
    intraday_volatility_inv = 1 / intraday_volatility
    momentum_factor = vwp_momentum * intraday_volatility_inv
    final_alpha = momentum_factor + close_momentum + price_reversal + vwap
    
    return final_alpha
```
