```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weigh by Volume
    df['weighted_log_returns'] = df['log_returns'] * df['volume']
    df['ema_log_returns'] = df['weighted_log_returns'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['range_volatility'] = df['high_low_range'].rolling(window=20).std()
    df['volatility_weighted_momentum'] = df['ema_log_returns'] / df['range_volatility']
    
    # 10-day Simple Moving Average (SMA) of the Close Price
    df['sma_close'] = df['close'].rolling(window=10).mean()
    
    # Momentum Factor
    df['momentum_factor'] = df['close'] - df['sma_close']
    df['momentum_sign'] = np.sign(df['momentum_factor'])
    
    # High-Activity Days
    df['volume_sma'] = df['volume'].rolling(window=30).mean()
    df['high_activity_day'] = df['volume'] > df['volume_sma']
    
    # Percentage Change in Close Price on High-Activity Days
    df['close_pct_change'] = df['close'].pct_change()
    df['high_activity_momentum'] = df['close_pct_change'] * df['high_activity_day']
    
    # Intraday Market Sentiment
    df['daily_midpoint'] = (df['high'] + df['low']) / 2
    df['sentiment'] = np.sign(df['close'] - df['daily_midpoint'])
    
    # Composite Signal
    df['composite_signal'] = df['momentum_factor'] * df['sentiment']
    
    # Opening Gap Volatility
    df['open_to_prev_close_diff'] = df['open'] - df['close'].shift(1)
    df['opening_gap_volatility'] = df['open_to_prev_close_diff'].rolling(window=20).std()
    
    # Volume-Weighted Close
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['ewa_volume_weighted_close'] = df['volume_weighted_close'].ewm(span=20).mean()
    
    # Final Alpha
    df['alpha_factor'] = (
        0.4 * df['volatility_weighted_momentum'] +
        0.1 * df['momentum_factor'] +
        0.1 * df['composite_signal'] +
        0.15 * df['range_volatility'] +
        0.15 * df['opening_gap_volatility'] +
        0.1 * df['ewa_volume_weighted_close']
    )
    
    return df['alpha_factor']
```
