```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns
    df['ema_log_return'] = df['log_return'].ewm(span=20, adjust=False).mean()
    
    # Weight by Volume to get Volume Adjusted Momentum
    df['vol_adj_mom'] = df['ema_log_return'] * df['volume']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # 20-Day SD of High-Low Ranges for Volatility
    df['hl_volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Combine Factors: Volume Adjusted Momentum and High-Low Range Volatility
    df['alpha_factor'] = (df['vol_adj_mom'] - df['hl_volatility'])
    
    return df['alpha_factor']
```
