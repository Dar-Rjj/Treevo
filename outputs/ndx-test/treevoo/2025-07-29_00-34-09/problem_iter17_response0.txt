```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, weighted by Volume
    df['volume_weighted_log_returns'] = df['log_returns'] * df['volume']
    df['volume_weighted_ema_20'] = df['volume_weighted_log_returns'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['hl_range_sd_20'] = df['high_low_range'].rolling(window=20).std()
    df['hl_volatility_adjusted_ema'] = df['volume_weighted_ema_20'] / df['hl_range_sd_20']
    
    # 10-day Simple Moving Average (SMA) of the Close Price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Momentum Factor
    df['momentum'] = df['close'] - df['sma_10']
    
    # Identify High-Activity Days
    df['volume_sma_30'] = df['volume'].rolling(window=30).mean()
    df['is_high_activity'] = df['volume'] > df['volume_sma_30']
    
    # Percentage Change in Close Price on High-Activity Days
    df['close_pct_change'] = df['close'].pct_change()
    df['adjusted_momentum'] = df['momentum'] * df['close_pct_change'] * df['is_high_activity']
    
    # Intraday Market Sentiment
    df['daily_midpoint'] = (df['high'] + df['low']) / 2
    df['directional_bias'] = np.where(df['close'] > df['daily_midpoint'], 1, -1)
    
    # Composite Sentiment-Adjusted Momentum
    df['composite_momentum'] = df['adjusted_momentum'] * df['directional_bias']
    
    # Filter by Volume
    volume_threshold = df['volume_sma_30'].mean()
    df['filtered_signal'] = np.where(df['volume'] > volume_threshold, df['composite_momentum'], 0)
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['reversal_signal'] = (df['high'] - df['low']) / df['open']
    df['reversal_sum_20'] = df['reversal_signal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['final_alpha'] = (
        0.4 * df['hl_volatility_adjusted_ema'] +
        0.1 * df['reversal_sum_20'] +
        0.3 * df['reversal_sum_20'] +
        0.1 * df['volume_weighted_close'] +
        0.1 * df['filtered_signal']
    )
    
    return df['final_alpha']
```
