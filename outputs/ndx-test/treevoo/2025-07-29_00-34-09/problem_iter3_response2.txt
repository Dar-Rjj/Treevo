```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 20-Day Exponential Moving Average of Log Returns
    df['ema_log_return'] = df['log_return'].ewm(span=20, adjust=False).mean()
    
    # Weight by Volume
    df['volume_adjusted_momentum'] = df['ema_log_return'] * df['volume']
    
    # Compute Daily Open-Close Range
    df['open_close_range'] = df['close'] - df['open']
    
    # Compute 20-Day Moving Standard Deviation of Open-Close Ranges
    df['open_close_volatility'] = df['open_close_range'].rolling(window=20).std()
    
    # Combine Volume Adjusted Momentum and Open-Close Range Volatility
    df['alpha_factor'] = df['volume_adjusted_momentum'] - df['open_close_volatility']
    
    return df['alpha_factor']
```
