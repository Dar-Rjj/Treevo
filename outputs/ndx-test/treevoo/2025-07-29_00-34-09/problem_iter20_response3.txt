import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    df['VWP'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    df['Absolute_Intraday_Range'] = df['high'] - df['low']
    df['Median_Volume_14'] = df['volume'].rolling(window=14).median()
    df['Intraday_Volatility'] = df['Absolute_Intraday_Range'] / df['Median_Volume_14']
    
    # Integrate High-to-Low Range Momentum
    df['High_Low_Range_Sum_20'] = df['high'] - df['low']
    df['High_Low_Range_Sum_20'] = df['High_Low_Range_Sum_20'].rolling(window=20).sum()
    df['VWP_Diff'] = df['VWP'].diff()
    df['Adjusted_Momentum'] = df['VWP_Diff'] / (df['Intraday_Volatility'] * df['High_Low_Range_Sum_20'])
    
    # Calculate Volume-Adjusted Momentum
    df['VWP_EMA_25'] = df['VWP'].ewm(span=25, adjust=False).mean()
    df['Volume_Adjusted_Momentum'] = df['VWP'] - df['VWP_EMA_25']
    
    # Incorporate Recent Volatility
    df['Daily_Returns'] = df['close'].pct_change()
    df['Volatility_15'] = df['Daily_Returns'].rolling(window=15).std()
    df['Inverse_Volatility'] = 1 / df['Volatility_15']
    df['Momentum_Vol_Adj'] = df['Volume_Adjusted_Momentum'] * df['Inverse_Volatility']
    
    # Calculate High-Low Range Volatility
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_Avg_30'] = df['High_Low_Range'].rolling(window=30).mean()
    
    # Calculate Open-Close Range Volatility
    df['Open_Close_Range'] = df['open'] - df['close']
    df['Open_Close_Range_Avg_30'] = df['Open_Close_Range'].rolling(window=30).mean()
    
    # Combine Volume-Adjusted Momentum and Enhanced Composite Volatility
    weighted_high_low_range = df['High_Low_Range_Avg_30'] * 0.6
    weighted_open_close_range = df['Open_Close_Range_Avg_30'] * 0.4
    df['Composite_Volatility'] = weighted_high_low_range + weighted_open_close_range
    df['Alpha_Factor'] = df['Momentum_Vol_Adj'] - df['Composite_Volatility']
    
    # Explore Momentum and Mean Reversion through Price Trends
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Total_Volume_5'] = df['volume'].rolling(window=5).sum()
    df['Total_Volume_20'] = df['volume'].rolling(window=20).sum()
    df['VWAP_5'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['Total_Volume_5']
    df['VWAP_20'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['Total_Volume_20']
    df['ATR_5'] = df['close'].rolling(window=5).std()
    df['ATR_20'] = df['close'].rolling(window=20).std()
    df['Volume_ROC_5'] = df['volume'].pct_change(5)
    df['Volume_ROC_20'] = df['volume'].pct_change(20)
    df['Volume_Above_5_Avg'] = df['volume'] > df['volume'].rolling(window=5).mean()
    df['Volume_Above_20_Avg'] = df['volume'] > df['volume'].rolling(window=20).mean()
    df['Price_Momentum_10'] = df['close'].pct_change(10)
    df['Price_Momentum_30'] = df['close'].pct_change(30
