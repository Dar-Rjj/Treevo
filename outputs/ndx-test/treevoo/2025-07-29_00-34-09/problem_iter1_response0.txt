```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum Based on Close Price
    close_momentum = df['close'].rolling(window=20).mean()
    
    # Analyze Price Reversals Using High and Low Prices
    price_reversal = (df['high'] - df['low']) / df['open']
    
    # Assess Volume Impact on Price Movement
    volume_weighted_close = (df['close'] * df['volume']).ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Final Alpha
    alpha_factor = (close_momentum + price_reversal + volume_weighted_close)
    
    return alpha_factor
```
