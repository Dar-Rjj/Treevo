```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate 20-Day Exponential Moving Average of Log Returns
    df['ema_log_returns'] = df['log_returns'].ewm(span=20, adjust=False).mean()
    
    # Weight by Volume
    df['volume_adjusted_momentum'] = df['ema_log_returns'] * df['volume']
    
    # Compute Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute 20-Day Moving Standard Deviation of High-Low Ranges
    df['high_low_volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Combine Volume Adjusted Momentum and High-Low Range Volatility
    df['alpha_factor'] = df['volume_adjusted_momentum'] - df['high_low_volatility']
    
    return df['alpha_factor']
```
