```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weigh by Volume
    df['vol_weighted_log_return'] = df['log_return'] * df['volume']
    df['ema_20_vol_weighted_log_return'] = df['vol_weighted_log_return'].ewm(span=20).mean()
    
    # Weight by High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['std_20_high_low_range'] = df['high_low_range'].rolling(window=20).std()
    df['vol_weighted_price_movement'] = (df['ema_20_vol_weighted_log_return'] / df['std_20_high_low_range']).fillna(0)
    
    # Calculate Rolling Average of High-to-Low Range
    df['avg_20_high_low_range'] = df['high_low_range'].rolling(window=20).mean()
    
    # Calculate Intraday Price Change
    df['intraday_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day Moving Averages of Intraday Price Change
    df['ma_5_intraday_change'] = df['intraday_change'].rolling(window=5).mean()
    df['ma_10_intraday_change'] = df['intraday_change'].rolling(window=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ma_5_intraday_change'] > df['ma_10_intraday_change'], 'reversal_signal'] = 1
    df.loc[df['ma_5_intraday_change'] < df['ma_10_intraday_change'], 'reversal_signal'] = -1
    
    # Consider High-to-Low Range
    df['reversal_signal'] = df['reversal_signal'] * (df['avg_20_high_low_range'] / df['avg_20_high_low_range'].max())
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['reversal_signal'] = df['reversal_signal'] * (df['volume'] > volume_threshold)
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_20_price_reversal'] = df['price_reversal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['ema_20_volume_weighted_close'] = df['volume_weighted_close'].ewm(span=20).mean()
    
    # Integrate Enhanced Momentum
    df['momentum_20'] = df['close'] / df['close'].shift(20)
    df['enhanced_momentum'] = df['momentum_20'] * df['volume']
    
    # Evaluate Trend Continuation
    df['momentum_50'] = df['close'] / df['close'].shift(50)
    df['trend_continuation'] = np.where(df['momentum_20'] > df['momentum_50'], 1, -1)
    
    # Combine Factors for Final Alpha
    df['alpha_factor'] = (0.4 * df['vol_weighted_price_movement'] +
                          0.1 * df['reversal_signal'] +
                          0.3 * df['sum_20_price_reversal'] +
                          0.1 * df['ema_20_volume_weighted_close'] +
                          0.1 * df['enhanced_momentum'])
    
    return df['alpha_factor']
```
