```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    intraday_activity = df['volume'] / (df['high'] + df['low'])
    intraday_factor = intraday_momentum * intraday_activity

    # Measure Sustained 5-Day Momentum
    close_prices = df['close'].shift(1).rolling(window=5, min_periods=1).mean()
    previous_close_prices = df['close'].shift(2)
    sustained_momentum = ((df['close'] - previous_close_prices) / previous_close_prices).rolling(window=5, min_periods=1).mean()

    # Calculate 5-Day Average Amount
    average_amount = df['amount'].rolling(window=5, min_periods=1).mean()

    # Combine Intraday and 5-Day Momentum
    combined_intraday_5day = (intraday_factor * average_amount) + (sustained_momentum * df['volume'])

    # Calculate Weighted Daily Return
    weighted_daily_return = (df['close'] - df['close'].shift(1)) * df['volume']

    # Calculate Cumulative Weighted Momentum
    cumulative_weighted_momentum = weighted_daily_return.rolling(window=5, min_periods=1).sum() / average_amount

    # Final Alpha Factor
    alpha_factor = combined_intraday_5day + cumulative_weighted_momentum

    return alpha_factor
```
