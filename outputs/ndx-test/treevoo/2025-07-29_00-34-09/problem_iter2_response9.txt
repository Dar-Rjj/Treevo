```python
import pandas as pd

def heuristics_v2(df):
    # Weight High and Low Prices by Volume
    high_weighted = df['high'] * df['volume']
    low_weighted = df['low'] * df['volume']
    
    # Compute Adjusted Volume-Weighted Price
    adj_vol_price = (high_weighted + low_weighted) / (2 * df['volume'])
    
    # Calculate Absolute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Adjust for Average Volume over Period
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    intraday_volatility = intraday_range / avg_volume_14
    
    # Calculate Rolling 14-Day High-to-Low Range Average
    daily_high_low_range = df['high'] - df['low']
    rolling_high_low_avg = daily_high_low_range.rolling(window=14).mean()
    
    # Combine Metrics: Volume-Adjusted Price and Momentum
    vol_adj_price_diff = adj_vol_price.diff()
    
    # Final Alpha Factor Adjustment
    alpha_factor = vol_adj_price_diff * intraday_volatility * rolling_high_low_avg
    
    return alpha_factor
```
