def heuristics_v2(data):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    data (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series with the generated alpha factor values.
    """
    # Calculate the percentage change of the close price
    data['close_pct_change'] = data['close'].pct_change()
    
    # Calculate the average true range (ATR) over the last 14 days
    high_low = data['high'] - data['low']
    high_close = (data['high'] - data['close'].shift(1)).abs()
    low_close = (data['low'] - data['close'].shift(1)).abs()
    ranges = pd.concat([high_low, high_close, low_close], axis=1)
    data['atr_14'] = ranges.max(axis=1).rolling(window=14).mean()
    
    # Calculate the money flow multiplier
    data['mf_multiplier'] = (2 * data['close'] - data['high'] - data['low']) / (data['high'] - data['low'])
    
    # Calculate the money flow volume
    data['mf_volume'] = data['mf_multiplier'] * data['volume']
    
    # Calculate the money flow ratio
    positive_mf = data[data['close'] > data['close'].shift(1)]['mf_volume']
    negative_mf = data[data['close'] < data['close'].shift(1)]['mf_volume']
    data['money_flow_ratio'] = positive_mf.rolling(window=14).sum() / negative_mf.rolling(window=14).sum()
    
    # Combine the factors into a single alpha factor
    data['alpha_factor'] = data['close_pct_change'] * data['atr_14'] * data['money_flow_ratio']
    
    # Return the alpha factor as a pandas Series
    return data['alpha_factor']
