```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weigh by Volume
    df['volume_weighted_log_returns'] = df['log_returns'] * df['volume']
    df['ema_log_returns'] = df['volume_weighted_log_returns'].ewm(span=20).mean()
    
    # Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['sd_high_low_range'] = df['high_low_range'].rolling(window=20).std()
    df['weighted_ema_log_returns'] = df['ema_log_returns'] / df['sd_high_low_range']
    
    # 10-day Simple Moving Average (SMA) of the Close Price
    df['sma_close'] = df['close'].rolling(window=10).mean()
    
    # Momentum Factor
    df['momentum'] = df['close'] - df['sma_close']
    df['momentum_sign'] = np.sign(df['momentum'])
    
    # Identify High-Activity Days
    df['sma_volume'] = df['volume'].rolling(window=30).mean()
    df['high_activity'] = df['volume'] > df['sma_volume']
    
    # Calculate Percentage Change in Close Price on High-Activity Days
    df['pct_change_high_activity'] = df['close'].pct_change().where(df['high_activity'], 0)
    df['adjusted_momentum'] = df['momentum'] + df['pct_change_high_activity']
    
    # Intraday Market Sentiment
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['directional_bias'] = np.where(df['close'] > df['midpoint'], 1, -1)
    
    # Composite Signal
    df['composite_signal'] = df['momentum'] * df['directional_bias']
    
    # Filter by Volume
    df['filtered_composite_signal'] = df['composite_signal'].where(df['volume'] > df['sma_volume'], 0)
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_reversal'] = df['reversal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean() / df['volume'].ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['alpha_factor'] = (0.4 * df['weighted_ema_log_returns'] 
                          + 0.1 * df['sum_reversal'] 
                          + 0.3 * df['reversal'] 
                          + 0.1 * df['volume_weighted_close'] 
                          + 0.1 * df['filtered_composite_signal'])
    
    return df['alpha_factor']
```
