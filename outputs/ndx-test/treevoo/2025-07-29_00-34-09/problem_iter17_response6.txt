```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market features.
    
    Parameters:
    df (pandas.DataFrame): DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
    pandas.Series: Factor values indexed by date.
    """
    # Calculate price change
    price_change = df['close'].diff()
    
    # Calculate the average true range
    atr = (df[['high', 'low']].diff().abs().max(axis=1) + 
           (df['close'] - df['open']).abs()).rolling(window=14).mean()
    
    # Calculate a simple moving average of the close price
    sma_close = df['close'].rolling(window=50).mean()
    
    # Calculate the percentage change of the close price
    pct_change_close = df['close'].pct_change()
    
    # Combine the factors to create the final heuristic
    alpha_factor = (price_change / atr) * (sma_close.pct_change() + pct_change_close)
    
    return alpha_factor
```
