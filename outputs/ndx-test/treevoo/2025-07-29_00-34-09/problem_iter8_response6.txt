import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Ratio'] = (df['high'] - df['low']) / df['low']
    df['Intraday_Trading_Activity'] = df['volume'] / (df['high'] + df['low'])
    df['Intraday_Momentum'] = df['High_Low_Ratio'] * df['Intraday_Trading_Activity']

    # Measure Sustained 5-Day Momentum
    df['Daily_Return'] = df['close'].pct_change()
    df['5Day_Momentum'] = df['Daily_Return'].rolling(window=5).mean()

    # Combine Intraday and 5-Day Momentum
    df['Combined_Momentum'] = (df['Intraday_Momentum'] * df['amount'].rolling(window=5).mean()) + \
                              (df['5Day_Momentum'] * df['volume'])

    # Calculate Weighted Daily Return
    df['Weighted_Daily_Return'] = (df['close'] - df['close'].shift(1)) * df['volume']

    # Calculate Cumulative Weighted Momentum
    df['Cumulative_Weighted_Momentum'] = df['Weighted_Daily_Return'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()

    # Calculate Volume-Weighted Average Price with High and Low Prices
    df['Volume_Weighted_Price'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])

    # Assess Intraday Volatility through Range and Volume
    df['Intraday_Range'] = df['high'] - df['low']
    df['Average_Volume_14D'] = df['volume'].rolling(window=14).mean()
    df['Intraday_Volatility'] = df['Intraday_Range'] / df['Average_Volume_14D']

    # Integrate High-to-Low Range Momentum into Alpha Factor
    df['High_Low_Range'] = df['high'] - df['low']
