```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average price
    avg_price_change = avg_price.pct_change()
    
    # Calculate the volume weighted close price
    vol_weighted_close = df['close'] * df['volume']
    
    # Calculate the smoothed volume weighted close price
    smoothed_vol_weighted_close = vol_weighted_close.rolling(window=5).mean()
    
    # Calculate the alpha factor as the ratio of smoothed volume weighted close to average price change
    alpha_factor = smoothed_vol_weighted_close / avg_price_change
    
    return alpha_factor
```
