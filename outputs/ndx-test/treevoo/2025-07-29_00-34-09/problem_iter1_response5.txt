```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Typical Price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Weight Typical Price by Volume to get Volume Adjusted Price (VAP)
    vap = typical_price * df['volume']
    
    # Compute 20-day Exponential Moving Average (EMA) of VAP
    ema_20 = vap.ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum from VAP
    momentum = vap - ema_20
    
    # Daily Returns Calculation using Adjacent Close Prices
    daily_returns = df['close'].pct_change()
    
    # Compute 10-day Standard Deviation of daily returns
    vol_10 = daily_returns.rolling(window=10).std()
    
    # Invert the 10-day Volatility
    inv_vol = 1 / vol_10
    
    # Multiply Momentum by Inverse of Volatility for final factor
    factor_values = momentum * inv_vol
    
    return factor_values
```
