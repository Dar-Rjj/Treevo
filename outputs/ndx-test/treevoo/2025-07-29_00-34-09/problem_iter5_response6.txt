```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Volume-Weighted Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['volume_weighted_return'] = df['log_return'] * df['volume']

    # Aggregate Upward and Downward Volume-Weighted Returns
    df['positive_returns'] = df['volume_weighted_return'].apply(lambda x: x if x > 0 else 0)
    df['negative_returns'] = -df['volume_weighted_return'].apply(lambda x: x if x < 0 else 0)

    positive_sum = df['positive_returns'].rolling(window=14).sum()
    negative_sum = df['negative_returns'].rolling(window=14).sum()

    # Compute RSI
    relative_strength = positive_sum / negative_sum
    df['volume_weighted_rsi'] = 100 - (100 / (1 + relative_strength))

    # Calculate Volume Adjusted Momentum
    df['ema_log_return'] = df['log_return'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['ema_log_return'] * df['volume']

    # Combine Volume-Weighted RSI and Volume Adjusted Momentum
    df['intermediate_alpha'] = df['volume_weighted_rsi'] + df['volume_adjusted_momentum']

    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['range_volatility'] = df['high_low_range'].rolling(window=20).std()

    # Adjust Intermediate Alpha Factor with Volatility
    df['adjusted_intermediate_alpha'] = df['intermediate_alpha'] - df['range_volatility']

    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']

    # Compute 5-Day and 10-Day Moving Averages of Price Change
    df['5_day_ma_price_change'] = df['price_change'].rolling(window=5).mean()
    df['10_day_ma_price_change'] = df['price_change'].rolling(window=10).mean()

    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['5_day_ma_price_change'] > df['10_day_ma_price_change'], 'reversal_signal'] = 1
    df.loc[df['5_day_ma_price_change'] < df['10_day_ma_price_change'], 'reversal_signal'] = -1

    # Integrate Reversal Signal with Adjusted Intermediate Alpha Factor
    df['final_alpha'] = df['adjusted_intermediate_alpha'] * (1 + df['reversal_signal'])

    return df['final_alpha']
```
