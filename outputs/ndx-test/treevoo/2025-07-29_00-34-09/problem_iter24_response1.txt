import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day EMAs of Price Change
    df['ema_5'] = df['price_change'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ema_5'] > df['ema_10'], 'reversal_signal'] = 1
    df.loc[df['ema_5'] < df['ema_10'], 'reversal_signal'] = -1
    
    # Filter by Volume
    volume_threshold = df['volume'].mean()
    df['filtered_signal'] = df['reversal_signal'] * (df['volume'] > volume_threshold)
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    df['vap_ema_20'] = df['vap'].ewm(span=20, adjust=False).mean()
    df['momentum'] = df['vap'] - df['vap_ema_20']
    
    # Incorporate Recent Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['volatility_10'] = df['daily_returns'].rolling(window=10).std()
    df['adjusted_momentum'] = df['momentum'] / df['volatility_10']
    
    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_volatility_20'] = df['high_low_range'].rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    df['open_close_range'] = df['close'] - df['open']
    df['open_close_volatility_20'] = df['open_close_range'].rolling(window=20).std()
    
    # Combine Crossover Momentum and Composite Volatility
    df['composite_volatility'] = (df['high_low_volatility_20'] + df['open_close_volatility_20']) / 2
    
    # Enhance with Additional Volatility Measure
    df['high_low_volatility_30'] = df['high_low_range'].rolling(window=30).std()
    df['enhanced_volatility'] = (df['high_low_volatility_20'] + df['high_low_volatility_30']) / 2
    
    # Calculate 5-Day Price Range
    df['5_day_high'] = df['high'].rolling(window=5).max()
    df['5_day_low'] = df['low'].rolling(window=5).min()
    df['5_day_price_range'] = df['5_day_high'] - df['5_day_low']
    df['normalized_5_day_price_range'] = df['5_day_price_range'] / ((df['5_day_high'] + df['5_day_low']) / 2)
    
    # Incorporate 5-Day Price Range into Momentum
    df['5_day_range_adjusted_momentum'] = df['momentum'] * df['normalized_5_day_price_range']
    
    # Calculate 10-Day Price Range
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    df['10_day_price_range'] = df['10_day_high'] - df['10_day_low']
    df['normalized_10_day_price_range'] = df['10_day_price_range'] / ((df['10_day_high'] + df['10_day_low']) / 2)
    
    # Incorporate 10-Day Price Range into Momentum
    df['10_day_range_adjusted_momentum'] = df['momentum'] * df['normalized_10_day_price_range']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum'] + df['5_day_range_adjusted_momentum'] + df['10_day_range_adjusted_momentum']) - df['enhanced_volatility']
    
    return df['final_alpha_factor']
