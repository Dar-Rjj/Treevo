```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages (SMA)
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    df['20_day_SMA'] = df['close'].rolling(window=20).mean()

    # Calculate cumulative volume over a period
    df['5_day_Total_Volume'] = df['volume'].rolling(window=5).sum()
    df['20_day_Total_Volume'] = df['volume'].rolling(window=20).sum()

    # Calculate volume-weighted average price (VWAP)
    df['5_day_VWAP'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['5_day_Total_Volume']
    df['20_day_VWAP'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['20_day_Total_Volume']

    # Calculate true range and average true range (ATR)
    df['True_Range'] = df[['high', 'low', 'close']].diff(axis=1).iloc[:, 1:].abs().max(axis=1)
    df['5_day_ATR'] = df['True_Range'].rolling(window=5).mean()
    df['20_day_ATR'] = df['True_Range'].rolling(window=20).mean()

    # Calculate rate of change (ROC) in volume
    df['5_day_ROC_Volume'] = df['volume'].pct_change(periods=5)
    df['20_day_ROC_Volume'] = df['volume'].pct_change(periods=20)

    # Identify high volume days
    df['Volume_Higher_Than_5_day_Avg'] = df['volume'] > df['volume'].rolling(window=5).mean()
    df['Volume_Higher_Than_20_day_Avg'] = df['volume'] > df['volume'].rolling(window=20).mean()

    # Calculate price momentum
    df['10_day_Price_Momentum'] = df['close'] - df['close'].shift(10)
    df['30_day_Price_Momentum'] = df['close'] - df['close'].shift(30)

    # Calculate rate of change (ROC) for close prices
    df['10_day_ROC_Close'] = df['close'].pct_change(periods=10)
    df['30_day_ROC_Close'] = df['close'].pct_change(periods=30)

    # Calculate relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['14_day_RSI'] = 100 - (100 / (1 + rs))

    # Alpha factor: combine the indicators into a single score
    df['alpha_factor'] = (
        df['5_day_SMA'] + 
        df['20_day_SMA'] + 
        df['5_day_VWAP'] + 
        df['20_day_VWAP'] + 
        df['5_day_ATR'] + 
        df['20_day_ATR'] + 
        df['5_day_ROC_Volume'] + 
        df['20_day_ROC_Volume'] + 
        df['Volume_Higher_Than_5_day_Avg'] + 
        df['Volume_Higher_Than_20_day_Avg'] + 
        df['10_day_Price_Momentum'] + 
        df['30_day_Price_Momentum'] + 
        df['10_day_ROC_Close'] + 
        df['30_day_ROC_Close'] + 
        df['14_day_RSI']
    )

    return df['alpha_factor']
```
