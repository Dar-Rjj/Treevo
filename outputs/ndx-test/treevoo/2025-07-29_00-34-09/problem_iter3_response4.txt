    # Apply Volume Smoothing
    df['volume_7d_ma'] = df['volume'].rolling(window=7).mean()
    df['normalized_volume'] = df['volume'] / df['volume_7d_ma']
    
    # Weighted Returns Calculation
    df['weighted_returns'] = df['adjusted_return'] * df['normalized_volume']
    df['positive_weighted_returns'] = df['weighted_returns'].where(df['weighted_returns'] > 0, 0)
    df['negative_weighted_returns'] = df['weighted_returns'].where(df['weighted_returns'] < 0, 0).abs()
    
    # Sum Over 14 Days
    df['sum_positive_weighted_returns'] = df['positive_weighted_returns'].rolling(window=14).sum()
    df['sum_negative_weighted_returns'] = df['negative_weighted_returns'].rolling(window=14).sum()
    
    # Compute Dynamic RSI
    df['relative_strength'] = df['sum_positive_weighted_returns'] / df['sum_negative_weighted_returns']
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    return df['dynamic_rsi']
