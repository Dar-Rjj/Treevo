```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the body range
    body_range = df['close'] - df['open']
    
    # Calculate the full day range
    day_range = df['high'] - df['low']
    
    # Calculate the ratio of body range to day range
    body_to_day_ratio = body_range / day_range
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Integrate the two factors
    factor = body_to_day_ratio * buying_pressure
    
    return factor
```
