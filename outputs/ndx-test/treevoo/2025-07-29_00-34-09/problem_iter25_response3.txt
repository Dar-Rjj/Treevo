```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day EMA of Price Change
    df['ema_5_price_change'] = df['price_change'].ewm(span=5, adjust=False).mean()
    df['ema_10_price_change'] = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = (df['ema_5_price_change'] > df['ema_10_price_change']).astype(int)
    
    # Filter by Volume
    volume_threshold = df['volume'].mean()
    df['reversal_signal_filtered'] = df['reversal_signal'] * (df['volume'] > volume_threshold)
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    df['vap_ema_20'] = df['vap'].ewm(span=20, adjust=False).mean()
    df['vap_momentum'] = df['vap'] - df['vap_ema_20']
    
    # Incorporate Recent Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility_10'] = df['daily_return'].rolling(window=10).std()
    df['vap_momentum_vol_adjusted'] = df['vap_momentum'] / df['volatility_10']
    
    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['volatility_20_high_low'] = df['high_low_range'].rolling(window=20).std()
    df['volatility_30_high_low'] = df['high_low_range'].rolling(window=30).std()
    df['composite_volatility'] = (df['volatility_20_high_low'] + df['volatility_30_high_low']) / 2
    
    # Calculate 5-Day Price Range
    df['5_day_high'] = df['high'].rolling(window=5).max()
    df['5_day_low'] = df['low'].rolling(window=5).min()
    df['5_day_range'] = df['5_day_high'] - df['5_day_low']
    df['5_day_avg_price'] = (df['5_day_high'] + df['5_day_low']) / 2
    df['normalized_5_day_range'] = df['5_day_range'] / df['5_day_avg_price']
    
    # Incorporate 5-Day Price Range into Momentum
    df['5_day_range_momentum'] = df['vap_momentum'] * df['normalized_5_day_range']
    
    # Calculate 10-Day Price Range
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    df['10_day_range'] = df['10_day_high'] - df['10_day_low']
    df['10_day_avg_price'] = (df['10_day_high'] + df['10_day_low']) / 2
    df['normalized_10_day_range'] = df['10_day_range'] / df['10_day_avg_price']
    
    # Incorporate 10-Day Price Range into Momentum
    df['10_day_range_momentum'] = df['vap_momentum'] * df['normalized_10_day_range']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['vap_momentum_vol_adjusted'] +
        df['5_day_range_momentum'] +
        df['10_day_range_momentum']
    ) - df['composite_volatility']
    
    return df['alpha_factor']
```
