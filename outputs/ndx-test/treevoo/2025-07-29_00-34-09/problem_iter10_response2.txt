```python
import pandas as pd

def heuristics(df):
    # Compute Typical Price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Weight Typical Price by Volume to get Volume Adjusted Price (VAP)
    vap = typical_price * df['volume']
    
    # Compute 20-day EMA of VAP
    ema_20 = vap.ewm(span=20, adjust=False).mean()
    
    # Calculate Volume Adjusted Momentum
    momentum = vap - ema_20
    
    # Compute Daily Returns
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Incorporate Recent Volatility (10-day Standard Deviation of daily returns)
    recent_volatility = daily_returns.rolling(window=10).std()
    
    # Multiply Momentum by Inverse of Volatility
    adjusted_momentum = momentum / (recent_volatility + 1e-8)  # small constant to avoid division by zero
    
    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    open_close_range = df['open'] - df['close']
    open_close_volatility = open_close_range.rolling(window=20).std()
    
    # Combine Volume Adjusted Momentum and Composite Volatility
    composite_volatility = (high_low_volatility + open_close_volatility) / 2
    alpha_factor = adjusted_momentum - composite_volatility
    
    return alpha_factor
```
