```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    df['VWP'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Compute 20-day Exponential Moving Average (EMA) of VWP
    df['VWP_20_EMA'] = df['VWP'].ewm(span=20).mean()
    
    # Momentum from VWP
    df['Momentum_VWP'] = df['VWP'] - df['VWP_20_EMA']
    
    # Calculate Absolute Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Divide Absolute Intraday Range by Average Volume over Period
    df['Intraday_Range_Volatility'] = df['Intraday_Range'] / df['volume'].rolling(window=20).mean()
    
    # Assess Momentum Based on Close Price
    df['Close_20_MA'] = df['close'].rolling(window=20).mean()
    df['Momentum_Close'] = df['close'] - df['Close_20_MA']
    
    # Analyze Price Reversals Using High and Low Prices
    df['Price_Reversal'] = (df['high'] - df['low']) / df['open']
    
    # Evaluate Volume Impact on Price Movement
    df['VWAP'] = (df['close'] * df['volume']).ewm(span=20).mean() / df['volume'].ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['Alpha_Factor'] = (
        df['Momentum_VWP'] / df['Intraday_Range_Volatility'] +
        df['Momentum_Close'] +
        df['Price_Reversal'] +
        df['VWAP']
    )
    
    return df['Alpha_Factor']
```
