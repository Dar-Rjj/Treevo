```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day EMA of Price Change
    df['ema_5'] = df['price_change'].ewm(span=5, adjust=False).mean()
    
    # Compute 10-Day EMA of Price Change
    df['ema_10'] = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = (df['ema_5'] > df['ema_10']).astype(int)
    
    # Filter by Volume
    volume_threshold = df['volume'].mean()
    df['filtered_signal'] = df['reversal_signal'] * (df['volume'] > volume_threshold)
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    df['vap_20_ema'] = df['vap'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['vap'] - df['vap_20_ema']
    
    # Incorporate Recent Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility_10'] = df['daily_return'].rolling(window=10).std()
    df['adjusted_momentum'] = df['volume_adjusted_momentum'] / df['volatility_10']
    
    # Calculate 5-Day Price Range
    df['5_day_high'] = df['high'].rolling(window=5).max()
    df['5_day_low'] = df['low'].rolling(window=5).min()
    df['5_day_range'] = df['5_day_high'] - df['5_day_low']
    df['normalized_5_day_range'] = df['5_day_range'] / ((df['5_day_high'] + df['5_day_low']) / 2)
    
    # Incorporate 5-Day Price Range into Momentum
    df['5_day_range_momentum'] = df['normalized_5_day_range'] * df['volume_adjusted_momentum']
    
    # Calculate 10-Day Price Range
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    df['10_day_range'] = df['10_day_high'] - df['10_day_low']
    df['normalized_10_day_range'] = df['10_day_range'] / ((df['10_day_high'] + df['10_day_low']) / 2)
    
    # Incorporate 10-Day Price Range into Momentum
    df['10_day_range_momentum'] = df['normalized_10_day_range'] * df['volume_adjusted_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_momentum'] + df['5_day_range_momentum'] + df['10_day_range_momentum'] - df['volatility_10']
    
    # Calculate Volume Adjusted Log Returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['log_return_ema_20'] = df['log_return'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_log_returns'] = df['log_return'] * df['volume']
    
    # Combine Enhanced Factors
    df['final_alpha_factor'] = df['alpha_factor'] + df['volume_adjusted_log_returns']
    
    return df['final_alpha_factor']
```
