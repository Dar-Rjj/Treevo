```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Compute 20-day Exponential Moving Average (EMA) of VWP
    ema_vwp = vwp.ewm(span=20, adjust=False).mean()
    
    # Momentum from VWP
    momentum_vwp = vwp - ema_vwp
    
    # Incorporate Intraday Range Volatility
    intraday_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    intraday_volatility = intraday_range / avg_volume
    
    # Assess Momentum Based on Close Price
    close_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Analyze Price Reversals Using High and Low Prices
    price_reversal = (df['high'] - df['low']) / df['open']
    
    # Evaluate Volume Impact on Price Movement
    ema_vwap = vwp.ewm(span=20, adjust=False).mean()
    
    # Calculate Volume-Adjusted Momentum
    log_returns = np.log(df['close']) - np.log(df.shift(1)['close'])
    ema_log_returns = log_returns.ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = ema_log_returns * df['volume']
    
    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=20).std()
    
    # Combine Factors for Final Alpha
    alpha = (
        momentum_vwp * (1 / intraday_volatility) +
        close_momentum +
        price_reversal +
        ema_vwap +
        volume_adjusted_momentum -
        high_low_volatility
    )
    
    return alpha
```
