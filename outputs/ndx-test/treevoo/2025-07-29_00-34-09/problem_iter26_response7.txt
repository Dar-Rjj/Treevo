import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    df['range'] = df['high'] - df['low']
    
    # Calculate the percentage change of close price from the previous day
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the weighted sum of range and percentage change
    df['factor'] = (df['range'] * 0.5) + (df['close_pct_change'].abs() * 0.5)
    
    # Return the factor as a pandas Series
    return df['factor']
