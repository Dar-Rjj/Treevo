```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Adjusted Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['adjusted_return'] = 100 * (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))
    
    # Apply Volume Smoothing
    span = 7
    df['volume_ewma'] = df['volume'].ewm(span=span, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['volume_ewma']
    
    # Weighted Returns Calculation
    df['weighted_returns'] = df['adjusted_return'] * df['normalized_volume']
    df['weighted_returns_14d_sum'] = df['weighted_returns'].rolling(window=14).sum()
    
    # Separate Positive and Negative Weights
    df['positive_weighted_returns'] = df['weighted_returns'].where(df['weighted_returns'] > 0, 0)
    df['negative_weighted_returns'] = df['weighted_returns'].where(df['weighted_returns'] < 0, 0).abs()
    df['positive_14d_sum'] = df['positive_weighted_returns'].rolling(window=14).sum()
    df['negative_14d_sum'] = df['negative_weighted_returns'].rolling(window=14).sum()
    
    # Compute Dynamic RSI
    df['relative_strength'] = df['positive_14d_sum'] / df['negative_14d_sum']
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    span = 20
    df['vap_ema'] = df['vap'].ewm(span=span, adjust=False).mean()
    df['vap_momentum'] = df['vap'] - df['vap_ema']
    
    # Incorporate Recent Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility_10d'] = df['daily_return'].rolling(window=10).std()
    df['inverse_volatility'] = 1 / df['volatility_10d']
    df['vap_momentum_adjusted'] = df['vap_momentum'] * df['inverse_volatility']
    
    # Combine Volume Adjusted Momentum and High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=20).std()
    df['alpha_factor'] = df['vap_momentum_adjusted'] - df['high_low_range_volatility']
    
    return df['alpha_factor']
```
