def heuristics_v2(df):
    # Calculate Intraday Price Change Ratio
    high_low_ratio = (df['high'] - df['low']) / df['low']
    volume_price_ratio = df['volume'] / (df['high'] + df['low'])
    intraday_price_change_ratio = high_low_ratio * volume_price_ratio
    
    # Measure Sustained Momentum
    df['previous_close'] = df['close'].shift(1)
    sustained_momentum = (df['close'] - df['previous_close']) / df['previous_close']
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = df['volume'] * sustained_momentum
    
    # Combine Factors
    factor = intraday_price_change_ratio + sustained_momentum + volume_weighted_momentum
    
    return factor
