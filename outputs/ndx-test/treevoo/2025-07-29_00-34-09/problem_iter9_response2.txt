```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the closing price within the high-low range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the relative change in the volume
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the weighted sum of the two metrics
    df['alpha_factor'] = 0.6 * df['close_position'] + 0.4 * df['volume_change']

    return df['alpha_factor']
```
