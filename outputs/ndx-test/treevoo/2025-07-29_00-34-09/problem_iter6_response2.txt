```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Typical Price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Volume Adjusted Price (VAP)
    vap = typical_price * df['volume']
    
    # Calculate 20-day Exponential Moving Average (EMA) of VAP
    ema_20 = vap.ewm(span=20, adjust=False).mean()
    
    # Calculate Volume Adjusted Momentum
    momentum = vap - ema_20
    
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate 10-day Standard Deviation of daily returns
    std_10 = daily_returns.rolling(window=10).std()
    
    # Calculate Inverse of Volatility
    inverse_volatility = 1 / std_10
    
    # Multiply Momentum by Inverse of Volatility
    adjusted_momentum = momentum * inverse_volatility
    
    # Calculate Daily High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 20-Day Moving Standard Deviation of High-Low Ranges
    hl_std_20 = high_low_range.rolling(window=20).std()
    
    # Calculate Daily Open-Close Range
    open_close_range = df['open'] - df['close']
    
    # Calculate 20-Day Moving Standard Deviation of Open-Close Ranges
    oc_std_20 = open_close_range.rolling(window=20).std()
    
    # Calculate Composite Volatility
    composite_volatility = (hl_std_20 + oc_std_20) / 2
    
    # Combine Volume Adjusted Momentum and Composite Volatility
    alpha_factor = adjusted_momentum - composite_volatility
    
    return alpha_factor
```
