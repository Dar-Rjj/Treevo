```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low']
    volume_to_price_ratio = df['volume'] / (df['high'] + df['low'])
    intraday_activity = intraday_momentum * volume_to_price_ratio

    # Measure Sustained 20-Day Volume Adjusted Momentum
    daily_log_returns = np.log(df['close']) - np.log(df['close'].shift(1))
    ema_log_return = daily_log_returns.ewm(span=20).mean()
    sustained_20_day_momentum = ema_log_return * df['volume']

    # Combine Intraday and 20-Day Volume Adjusted Momentum
    avg_amount_20_day = df['amount'].rolling(window=20).mean()
    combined_momentum = (intraday_activity * avg_amount_20_day) + (sustained_20_day_momentum * df['volume'])

    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_range_volatility = high_low_range.rolling(window=20).std()

    # Calculate Cumulative Weighted Momentum
    weighted_daily_return = (df['close'] - df['close'].shift(1)) * df['volume']
    cumulative_weighted_momentum = weighted_daily_return.rolling(window=5).sum() / df['amount'].rolling(window=5).mean()

    # Final Alpha Factor
    alpha_factor = combined_momentum - high_low_range_volatility + cumulative_weighted_momentum
    
    return alpha_factor
```
