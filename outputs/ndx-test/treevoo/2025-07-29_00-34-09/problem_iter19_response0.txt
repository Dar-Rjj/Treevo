```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    df['VWP'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['average_volume_14d'] = df['volume'].rolling(window=14).mean()
    df['volatility_adjusted_intraday_range'] = df['intraday_range'] / df['average_volume_14d']
    
    # Integrate High-to-Low Range Momentum
    df['high_low_range_sum_14d'] = df['intraday_range'].rolling(window=14).sum()
    df['vwp_diff'] = df['VWP'].diff()
    df['high_low_range_momentum'] = df['vwp_diff'] / df['high_low_range_sum_14d']
    
    # Calculate Volume-Adjusted Momentum
    df['vwp_ema_20'] = df['VWP'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['VWP'] - df['vwp_ema_20']
    
    # Incorporate Recent Volatility
    df['returns'] = df['close'].pct_change()
    df['volatility_10d'] = df['returns'].rolling(window=10).std()
    df['recent_volatility_adjusted_momentum'] = df['volume_adjusted_momentum'] / df['volatility_10d']
    
    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_avg_20d'] = df['high_low_range'].rolling(window=20).mean()
    
    # Combine Volume-Adjusted Momentum and Composite Volatility
    df['composite_alpha_factor'] = df['recent_volatility_adjusted_momentum'] - df['high_low_range_avg_20d']
    
    return df['composite_alpha_factor']
```
