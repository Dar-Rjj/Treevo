```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the difference between high and low
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the difference between close and open
    df['close_open_diff'] = df['close'] - df['open']
    
    # Calculate a simple moving average of the daily return over 5 days
    df['sma_5'] = df['return'].rolling(window=5).mean()
    
    # Calculate the ratio of volume to the 5-day average volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Alpha factor: (High-Low Difference) * (Close-Open Difference) * (SMA 5) * (Volume Ratio)
    alpha_factor = df['high_low_diff'] * df['close_open_diff'] * df['sma_5'] * df['volume_ratio']
    
    return alpha_factor
```
