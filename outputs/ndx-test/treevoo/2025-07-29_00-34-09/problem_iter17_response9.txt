```python
def heuristics_v2(df):
    # Calculate the price range for the day
    df['range'] = df['high'] - df['low']
    
    # Calculate the difference between close and open
    df['close_open_diff'] = df['close'] - df['open']
    
    # Calculate the normalized volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Create a factor that combines the range, close-open difference, and volume change
    df['factor'] = (df['close_open_diff'] / df['range']) * df['volume_change']
    
    # Return the factor as a pandas Series
    return df['factor']
```
