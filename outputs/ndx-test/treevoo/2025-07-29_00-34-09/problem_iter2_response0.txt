```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Weighted Price
    df['Weighted_Price'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Smoothed Momentum
    df['Short_Term_EMA'] = df['Weighted_Price'].ewm(span=5, adjust=False).mean()
    df['Long_Term_EMA'] = df['Weighted_Price'].ewm(span=20, adjust=False).mean()
    df['Smoothed_Momentum'] = df['Long_Term_EMA'] - df['Short_Term_EMA']
    
    # Calculate Intraday Price Change Ratio
    df['High_Low_Ratio'] = (df['high'] - df['low']) / df['low']
    df['Volume_to_Price_Ratio'] = df['volume'] / (df['high'] + df['low'])
    df['Intraday_Price_Change_Ratio'] = df['High_Low_Ratio'] * df['Volume_to_Price_Ratio']
    
    # Measure Sustained Momentum
    df['Sequential_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Momentum
    df['Volume_Weighted_Momentum'] = df['volume'] * df['Sequential_Momentum']
    
    # Combine Factors
    df['Alpha_Factor'] = (df['Smoothed_Momentum'] 
                          + df['Intraday_Price_Change_Ratio'] 
                          + df['Sequential_Momentum'] 
                          + df['Volume_Weighted_Momentum'])
    
    return df['Alpha_Factor']
```
