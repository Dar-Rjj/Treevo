```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the relative position of the close price within the daily range
    df['close_position'] = (df['close'] - df['low']) / df['range']
    
    # Calculate the price change from previous day
    df['price_change'] = df['close'].pct_change()
    
    # Calculate the volume change from previous day
    df['volume_change'] = df['volume'].pct_change()
    
    # Compute a simple moving average of the close price over 5 days
    df['sma_5'] = df['close'].rolling(window=5).mean()
    
    # Heuristic: If the current close is above its 5-day SMA and the close is in the upper half of the daily range, and there is positive price change and negative volume change
    alpha_factor = (df['close'] > df['sma_5']) & (df['close_position'] > 0.5) & (df['price_change'] > 0) & (df['volume_change'] < 0)
    
    # Convert the boolean series to integer (1 for True, 0 for False)
    alpha_factor = alpha_factor.astype(int)
    
    return alpha_factor
```
