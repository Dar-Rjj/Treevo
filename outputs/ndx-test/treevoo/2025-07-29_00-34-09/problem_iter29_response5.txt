```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Weigh by Volume and calculate 20-Day EMA of Log Returns
    df['volume_weighted_log_returns'] = df['log_returns'] * df['volume']
    df['ema_vol_weighted_log_returns'] = df['volume_weighted_log_returns'].ewm(span=20).mean()
    
    # Compute Daily High-Low Range and 20-Day Moving SD
    df['high_low_range'] = df['high'] - df['low']
    df['sma_high_low_range_sd'] = df['high_low_range'].rolling(window=20).std()
    
    # Adjust by High-Low Range Volatility
    df['vol_adjusted_price_movement'] = df['ema_vol_weighted_log_returns'] / df['sma_high_low_range_sd']
    
    # Determine 10-Day Simple Moving Average (SMA) of the Close Price
    df['sma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Define a Momentum Factor
    df['momentum_factor'] = df['close'] - df['sma_10_close']
    df['momentum_sign'] = np.sign(df['momentum_factor'])
    
    # Identify High-Activity Days
    df['sma_30_volume'] = df['volume'].rolling(window=30).mean()
    df['is_high_activity_day'] = df['volume'] > df['sma_30_volume']
    
    # Calculate Percentage Change in Close Price on High-Activity Days
    df['pct_change_close'] = df['close'].pct_change()
    df['pct_change_close_high_activity'] = df['pct_change_close'] * df['is_high_activity_day']
    
    # Adjust the momentum score based on the percentage change
    df['adjusted_momentum_score'] = df['momentum_sign'] * df['pct_change_close_high_activity']
    
    # Analyze Intraday Market Sentiment
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['directional_bias'] = np.sign(df['close'] - df['midpoint'])
    
    # Combine Momentum and Sentiment for Composite Signal
    df['composite_signal'] = df['momentum_sign'] * df['directional_bias']
    
    # Filter by Volume
    df['significant_signal'] = df['composite_signal'] * (df['volume'] > df['sma_30_volume'])
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['reversal_signal'] = (df['high'] - df['low']) / df['open']
    df['sum_reversal_signal'] = df['reversal_signal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['ema_volume_weighted_close'] = df['volume_weighted_close'].ewm(span=20).mean()
    
    # Combine Factors for Final Alpha
    df['final_alpha'] = (
        0.4 * df['vol_adjusted_price_movement'] +
        0.1 * df['sum_reversal_signal'] +
        0.3 * df['momentum_factor'] +
        0.1 * df['ema_volume_weighted_close'] +
        0.1 * df['significant_signal']
    )
    
    return df['final_alpha']
```
