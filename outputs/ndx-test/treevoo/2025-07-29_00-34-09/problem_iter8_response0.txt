import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    intraday_range = df['high'] - df['low']
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    intraday_volatility = intraday_range / avg_volume_14
    
    # Integrate High-to-Low Range Momentum into Alpha Factor
    high_low_range = df['high'] - df['low']
    rolling_high_low_range_avg = high_low_range.rolling(window=14).mean()
    vwp_diff = vwp.diff()
    momentum_adjusted = vwp_diff * intraday_volatility * rolling_high_low_range_avg
    
    # Calculate Volume Adjusted Momentum
    vwp_ema_20 = vwp.ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = vwp - vwp_ema_20
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=10).std()
    inverse_volatility = 1 / recent_volatility
    volatility_adjusted_momentum = volume_adjusted_momentum * inverse_volatility
    
    # Calculate High-Low Range Volatility
    daily_high_low_range = df['high'] - df['low']
    high_low_volatility = daily_high_low_range.rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    daily_open_close_range = df['open'] - df['close']
    open_close_volatility = daily_open_close_range.rolling(window=20).std()
    
    # Combine Volume-Adjusted Momentum and Composite Volatility
    composite_volatility = (high_low_volatility + open_close_volatility) / 2
    final_alpha_factor = volatility_adjusted_momentum - composite_volatility
    
    return final_alpha_factor
