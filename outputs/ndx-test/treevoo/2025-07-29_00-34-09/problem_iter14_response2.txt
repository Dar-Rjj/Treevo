import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate High-to-Low Range
    data['High_to_Low_Range'] = data['high'] - data['low']
    
    # Calculate Rolling Median of High-to-Low Range (14-day window)
    data['Rolling_Median_Range'] = data['High_to_Low_Range'].rolling(window=14).median()
    
    # Calculate Daily Price Change
    data['Daily_Price_Change'] = data['close'] - data['open']
    
    # Compute 5-Day and 10-Day Exponential Moving Average of Price Change
    data['EMA_5'] = data['Daily_Price_Change'].ewm(span=5, adjust=False).mean()
    data['EMA_10'] = data['Daily_Price_Change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    data['Reversal_Signal'] = np.where(data['EMA_5'] > data['EMA_10'], 1, -1)
    data['Reversal_Signal'] *= np.where(data['Rolling_Median_Range'] > data['Rolling_Median_Range'].median(), 1.5, 0.5)
    
    # Enhanced Volume Filter
    data['20_Day_Avg_Volume'] = data['volume'].rolling(window=20).mean()
    data['Enhanced_Volume_Filter'] = np.where(data['volume'] > data['20_Day_Avg_Volume'], 1, 0)
    
    # Calculate Volume-Weighted Price (VWP)
    data['VWP'] = (data['high'] * data['volume'] + data['low'] * data['volume']) / (2 * data['volume'])
    
    # Compute Momentum from VWP
    data['Momentum_VWP'] = data['VWP'] - data['VWP'].ewm(span=20, adjust=False).mean()
    
    # Incorporate Intraday Range Volatility
    data['Intraday_Range_Volatility'] = (data['high'] - data['low']) / data['volume'].rolling(window=20).mean()
    
    # Assess Momentum Based on Close Price
    data['Momentum_Close'] = data['close'] - data['close'].rolling(window=20).mean()
    
    # Analyze Price Reversals Using High and Low Prices
    data['Price_Reversal'] = (data['high'] - data['low']) / data['open']
    
    # Evaluate Volume Impact on Price Movement
    data['VWAP_20_EMA'] = (data['high'] * data['volume'] + data['low'] * data['volume']) / (2 * data['volume']).ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Final Alpha
    data['Final_Alpha'] = (
        data['Momentum_VWP'] / data['Intraday_Range_Volatility'] +
        data['Momentum_Close'] +
        data['Price_Reversal'] +
        data['VWAP_20_EMA']
    ) * data['Enhanced_Volume_Filter'] * data['Reversal_Signal']
    
    # Enhance with Additional Factors
    # Calculate Relative Strength Index (RSI)
    delta = data['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    data['RSI'] = 100 - (100 / (1 + RS))
    data['RSI_Normalized'] = (data['RSI'] - 30) / (70 - 30)
    
    # Incorporate RSI into Final Alpha
    data['Final_Alpha'] += data['RSI_Normalized'] * 0.5
    
    # Include Moving Average Convergence Divergence (MACD)
    data['MACD_Line'] = data['close'].ewm(span=12, adjust=False).mean() - data['close'].ewm(span=26, adjust=False).mean()
    data['Signal_Line'] = data['MACD_Line'].ewm(span=9, adjust=False).mean()
    data['MACD_Cross'] = np.where(data['MACD_Line'] > data['Signal_Line'], 0.1, -0.1)
    
    # Integrate MACD into Final Alpha
    data['Final_Alpha'] += data['MACD_Cross']
    
    return data['Final_Alpha']
