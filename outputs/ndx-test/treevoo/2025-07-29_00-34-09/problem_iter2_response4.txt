import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])

    # Compute 20-day Exponential Moving Average (EMA) of VWP
    ema_20 = vwp.ewm(span=20, adjust=False).mean()
    
    # Momentum from VWP
    momentum = vwp - ema_20
    
    # Calculate Absolute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Average Volume over the period
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Adjust for Average Volume
    intraday_volatility = intraday_range / avg_volume
    
    # Invert the Intraday Range Volatility
    inverse_intraday_volatility = 1 / intraday_volatility
    
    # Combine Momentum with Inverse Intraday Range Volatility
    final_factor = momentum * inverse_intraday_volatility
    
    return final_factor
