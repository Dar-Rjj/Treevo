```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to the average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the difference between high and low prices
    price_range = df['high'] - df['low']
    
    # Calculate the ratio of volume to the price range
    volume_to_price_range = df['volume'] / price_range
    
    # Combine the two ratios to create the alpha factor
    alpha_factor = close_to_avg_ratio * volume_to_price_range
    
    return alpha_factor
```
