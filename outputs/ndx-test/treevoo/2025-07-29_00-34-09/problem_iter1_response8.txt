```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Rolling Average of High-to-Low Range
    df['range_momentum'] = df['high_to_low_range'].rolling(window=14).mean()
    
    return df['range_momentum']
```
