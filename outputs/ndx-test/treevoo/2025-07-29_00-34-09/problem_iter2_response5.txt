```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Compute Intraday Range Volatility
    intraday_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    df['intraday_volatility'] = intraday_range / avg_volume
    
    # Assess Momentum Based on Close Price
    df['momentum_20day'] = df['close'].rolling(window=20).mean()
    
    # Analyze Price Reversals Using High and Low Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    
    # Evaluate Volume Impact on Price Movement
    df['vwap_20day_ema'] = df['VWAP'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Final Alpha
    df['vwap_diff'] = df['VWAP'].diff()
    df['vwap_diff_adjusted'] = df['vwap_diff'] * df['intraday_volatility']
    
    # Sum all sub-factors
    df['final_alpha'] = (
        df['vwap_diff_adjusted'] +
        (df['close'] - df['momentum_20day']) +
        df['price_reversal'] +
        (df['VWAP'] - df['vwap_20day_ema'])
    )
    
    return df['final_alpha']
```
