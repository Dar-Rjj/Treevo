```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series indexed by date, representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the money flow multiplier
    mfm = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    mf_volume = mfm * df['volume']
    
    # Calculate the 14-day money flow ratio
    mf_ratio = mf_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Calculate the 14-day exponential moving average of the close price
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-day standard deviation of the daily returns
    std_14 = df['daily_return'].rolling(window=14).std()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (mf_ratio - 0.5) * (ema_14 - df['close']) / std_14
    
    return alpha_factor
```
