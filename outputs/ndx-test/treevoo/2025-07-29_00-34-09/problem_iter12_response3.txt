```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume Adjusted Price (VAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vap = typical_price * df['volume']

    # Calculate Volume Adjusted Momentum
    ema_20_vap = vap.ewm(span=20, adjust=False).mean()
    momentum = vap - ema_20_vap

    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    std_15_daily_returns = daily_returns.rolling(window=15).std()
    inverse_volatility = 1 / std_15_daily_returns
    adjusted_momentum = momentum * inverse_volatility

    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=25).std()

    # Calculate Open-Close Range Volatility
    open_close_range = np.abs(df['open'] - df['close'])
    open_close_volatility = open_close_range.rolling(window=25).std()

    # Combine Adjusted Momentum and Enhanced Composite Volatility
    weighted_average_volatility = 0.6 * high_low_volatility + 0.4 * open_close_volatility
    alpha_factor = adjusted_momentum - weighted_average_volatility

    return alpha_factor
```
