```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day simple moving average (SMA) of the close price
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Define a momentum factor as the difference between the current close price and the 10-day SMA
    df['momentum_factor'] = df['close'] - df['SMA_10']
    
    # Use the sign of the momentum factor to predict continuation or reversal
    df['momentum_sign'] = df['momentum_factor'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate 30-day simple moving average of volume to define high-activity days
    df['SMA_vol_30'] = df['volume'].rolling(window=30).mean()
    df['high_activity_day'] = (df['volume'] > df['SMA_vol_30']).astype(int)
    
    # Calculate the percentage change in the closing price from the previous day on high-activity days
    df['prev_close'] = df['close'].shift(1)
    df['pct_change_high_activity'] = (df['close'] / df['prev_close'] - 1) * 100
    df['pct_change_adjustment'] = df['high_activity_day'] * df['pct_change_high_activity']
    
    # Adjust the momentum score based on the percentage change on high-activity days
    df['adjusted_momentum'] = df['momentum_sign'] + df['pct_change_adjustment'] / 100
    
    # Calculate the daily price range (high - low)
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the midpoint of the daily range
    df['midpoint'] = (df['high'] + df['low']) / 2
    
    # Determine directional bias based on the close price relative to the midpoint
    df['directional_bias'] = df.apply(lambda row: 1 if row['close'] > row['midpoint'] else -1, axis=1)
    
    # Combine the directional bias with the volume-based momentum adjustment
    df['composite_momentum'] = df['adjusted_momentum'] * df['directional_bias']
    
    # Return the final composite sentiment-adjusted momentum score
    return df['composite_momentum']
```
