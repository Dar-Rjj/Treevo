```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day EMA of Price Change
    df['ema_5'] = df['price_change'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = (df['ema_5'] > df['ema_10']) * 1.0
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['significant_volume'] = (df['volume'] > volume_threshold) * 1.0
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    df['vap_ema_20'] = df['vap'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['vap'] - df['vap_ema_20']
    
    # Incorporate Recent Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['volatility_10'] = df['daily_returns'].rolling(window=10).std()
    df['inverse_volatility_10'] = 1 / df['volatility_10']
    df['adjusted_momentum'] = df['volume_adjusted_momentum'] * df['inverse_volatility_10']
    
    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_volatility_20'] = df['high_low_range'].rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    df['open_close_range'] = df['close'] - df['open']
    df['open_close_volatility_20'] = df['open_close_range'].rolling(window=20).std()
    
    # Enhance with Additional Volatility Measure
    df['high_low_volatility_30'] = df['high_low_range'].rolling(window=30).std()
    df['composite_volatility'] = (df['high_low_volatility_20'] + df['high_low_volatility_30'] + df['open_close_volatility_20']) / 3
    
    # Calculate 5-Day Price Range
    df['5_day_high'] = df['high'].rolling(window=5).max()
    df['5_day_low'] = df['low'].rolling(window=5).min()
    df['5_day_price_range'] = df['5_day_high'] - df['5_day_low']
    df['avg_5_day_price'] = (df['5_day_high'] + df['5_day_low']) / 2
    df['normalized_5_day_price_range'] = df['5_day_price_range'] / df['avg_5_day_price']
    df['5_day_adjusted_momentum'] = df['volume_adjusted_momentum'] * df['normalized_5_day_price_range']
    
    # Calculate 10-Day Price Range
    df['10_day_high'] = df['high'].rolling(window=10).max()
    df['10_day_low'] = df['low'].rolling(window=10).min()
    df['10_day_price_range'] = df['10_day_high'] - df['10_day_low']
    df['avg_10_day_price'] = (df['10_day_high'] + df['10_day_low']) / 2
    df['normalized_10_day_price_range'] = df['10_day_price_range'] / df['avg_10_day_price']
    df['10_day_adjusted_momentum'] = df['volume_adjusted_momentum'] * df['normalized_10_day_price_range']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['volume_adjusted_momentum'] +
        df['5_day_adjusted_momentum'] +
        df['10_day_adjusted_momentum'] -
        df['composite_volatility']
    )
    
    return df['final_alpha_factor']
```
