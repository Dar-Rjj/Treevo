```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Adjusted Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['adj_return'] = 100 * (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))

    # Apply Volume Smoothing
    ewm_volume = df['volume'].ewm(span=7, adjust=False).mean()
    df['norm_volume'] = df['volume'] / ewm_volume

    # Weighted Returns Calculation
    df['weighted_returns'] = df['adj_return'] * df['norm_volume']
    positive_weighted_returns = df['weighted_returns'].where(df['weighted_returns'] > 0, 0).rolling(window=14).sum()
    negative_weighted_returns = df['weighted_returns'].where(df['weighted_returns'] < 0, 0).abs().rolling(window=14).sum()

    # Compute Dynamic RSI
    relative_strength = positive_weighted_returns / negative_weighted_returns
    dynamic_rsi = 100 - (100 / (1 + relative_strength))

    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']

    # Calculate Volume Adjusted Momentum
    ema_vap_20 = df['vap'].ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = df['vap'] - ema_vap_20

    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=10).std()
    adjusted_momentum = volume_adjusted_momentum / volatility

    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=20).std()

    # Calculate Open-Close Range Volatility
    open_close_range = df['open'] - df['close']
    open_close_volatility = open_close_range.rolling(window=20).std()

    # Combine Volume Adjusted Momentum, Dynamic RSI, and Composite Volatility
    composite_volatility = (high_low_volatility + open_close_volatility) / 2
    alpha_factor = adjusted_momentum + dynamic_rsi - composite_volatility

    return alpha_factor
```
