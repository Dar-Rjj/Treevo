```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    df['VWP'] = ((df['high'] + df['low']) / 2 * df['volume']) / df['volume']
    
    # Assess Intraday Volatility
    avg_volume_14d = df['volume'].rolling(window=14).mean()
    df['intraday_volatility'] = (df['high'] - df['low']) / avg_volume_14d
    
    # Integrate High-to-Low Range Momentum
    h_l_range_sum_14d = (df['high'] - df['low']).rolling(window=14).sum()
    vwp_diff = df['VWP'].diff()
    df['h_l_momentum'] = vwp_diff / h_l_range_sum_14d
    
    # Calculate Volume-Adjusted Momentum
    df['vwp_ema_20'] = df['VWP'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['VWP'] - df['vwp_ema_20']
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=10).std()
    df['volatility_adjusted_momentum'] = df['volume_adjusted_momentum'] / recent_volatility
    
    # Combine Volume-Adjusted Momentum and Composite Volatility
    hl_range_ma_20 = (df['high'] - df['low']).rolling(window=20).mean()
    oc_range_ma_20 = (df['open'] - df['close']).rolling(window=20).mean()
    composite_volatility = (hl_range_ma_20 + oc_range_ma_20) / 2
    df['alpha_factor'] = df['volatility_adjusted_momentum'] - composite_volatility
    
    return df['alpha_factor']
```
