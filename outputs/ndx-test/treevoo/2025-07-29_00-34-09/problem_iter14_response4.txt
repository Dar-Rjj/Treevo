```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Volume-Adjusted Price Movement
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['vol_weighted_log_returns'] = df['log_returns'] * df['volume']
    df['ema_20_vol_weighted_returns'] = df['vol_weighted_log_returns'].ewm(span=20).mean()
    df['daily_high_low_range'] = df['high'] - df['low']
    df['range_volatility'] = df['daily_high_low_range'].rolling(window=20).std()
    df['volume_adjusted_price_movement'] = (df['ema_20_vol_weighted_returns'] / df['range_volatility']).fillna(0)

    # Calculate Rolling Average of High-to-Low Range
    df['avg_high_low_range_14'] = df['daily_high_low_range'].rolling(window=14).mean()

    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['open']

    # Compute 5-Day Moving Average of Price Change
    df['ma_5_price_change'] = df['daily_price_change'].rolling(window=5).mean()

    # Compute 10-Day Moving Average of Price Change
    df['ma_10_price_change'] = df['daily_price_change'].rolling(window=10).mean()

    # Determine Reversal Signal
    df['reversal_signal'] = np.where(df['ma_5_price_change'] > df['ma_10_price_change'], 1, 
                                     np.where(df['ma_5_price_change'] < df['ma_10_price_change'], -1, 0))

    # Amplify or Dampen the Reversal Signal Based on High-to-Low Range
    df['reversal_signal_adjusted'] = df['reversal_signal'] * (df['avg_high_low_range_14'] / df['avg_high_low_range_14'].max())

    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    df['reversal_signal_filtered'] = df['reversal_signal_adjusted'] * (df['volume'] > volume_threshold)

    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = ((df['high'] - df['low']) / df['open']).rolling(window=20).sum()

    # Assess Volume Impact on Price Movement
    df['vwp'] = ((df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])).fillna(0)
    df['vwp_ema_20'] = df['vwp'].ewm(span=20).mean()
    df['vwp_momentum'] = df['vwp'] - df['vwp_ema_20']

    # Incorporate Intraday Range Volatility
    df['intraday_range_volatility'] = (df['high'] - df['low']) / df['volume'].rolling(window=20).mean()

    # Assess Momentum Based on Close Price
    df['momentum_close'] = df['close'] - df['close'].rolling(window=20).mean()

    # Evaluate Volume Impact on Price Movement
    df['vwap_20'] = (df['close'] * df['volume']).ewm(span=20).mean() / df['volume'].ewm(span=20).mean()

    # Combine Factors for Final Alpha
    df['alpha_factor'] = (0.4 * df['volume_adjusted_price_movement'] +
                          0.3 * df['reversal_signal_filtered'] +
                          0.2 * df['price_reversal'] +
                          0.1 * df['vwap_20'])
    
    return df['alpha_factor']
```
