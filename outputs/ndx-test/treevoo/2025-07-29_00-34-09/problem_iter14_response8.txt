```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = (df['High'] * df['Volume'] + df['Low'] * df['Volume']) / (2 * df['Volume'])
    
    # Assess Intraday Range
    df['IntradayRange'] = df['High'] - df['Low']
    df['14DayAvgIntradayRange'] = df['IntradayRange'].rolling(window=14).mean()
    
    # Integrate High-to-Low Range Momentum
    df['14DayAvgHighToLow'] = (df['High'] - df['Low']).rolling(window=14).mean()
    df['VWAPDiff'] = (df['VWAP'] - df['VWAP'].shift(1)) / df['14DayAvgHighToLow']
    
    # Volume-Adjusted Smoothed Momentum
    df['LogReturns'] = np.log(df['Close'] / df['Close'].shift(1))
    df['20DayEMALogReturns'] = df['LogReturns'].ewm(span=20, adjust=False).mean()
    df['VolumeAdjMomentum'] = df['Volume'] * df['20DayEMALogReturns']
    
    # Open-Close Range Volatility
    df['OpenCloseRange'] = df['Open'] - df['Close']
    df['20DayStdOpenCloseRange'] = df['OpenCloseRange'].rolling(window=20).std()
    
    # Compute Daily Price Change
    df['DailyPriceChange'] = df['Close'] - df['Open']
    
    # Compute 5-Day and 10-Day Exponential Moving Average of Price Change
    df['5DayEMA_PChange'] = df['DailyPriceChange'].ewm(span=5, adjust=False).mean()
    df['10DayEMA_PChange'] = df['DailyPriceChange'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['ReversalSignal'] = 0
    df.loc[df['5DayEMA_PChange'] > df['10DayEMA_PChange'], 'ReversalSignal'] = 1
    df.loc[df['5DayEMA_PChange'] < df['10DayEMA_PChange'], 'ReversalSignal'] = -1
    df['ReversalSignal'] *= df['14DayAvgIntradayRange'] / df['14DayAvgIntradayRange'].rolling(window=14).median()
    
    # Enhanced Volume Filter
    df['20DayAvgVolume'] = df['Volume'].rolling(window=20).mean()
    df['EnhancedVolumeFilter'] = df['Volume'] > df['20DayAvgVolume']
    
    # Incorporate Intraday Range Volatility
    df['AbsIntradayRange'] = df['High'] - df['Low']
    df['IntradayRangeVolatility'] = df['AbsIntradayRange'] / df['20DayAvgVolume']
    
    # Assess Momentum Based on Close Price
    df['20DayMA_Close'] = df['Close'].rolling(window=20).mean()
    df['MomentumBasedOnClose'] = df['Close'] - df['20DayMA_Close']
    
    # Analyze Price Reversals Using High and Low Prices
    df['PriceReversals'] = (df['High'] - df['Low']) / df['Open']
    
    # Evaluate Volume Impact on Price Movement
    df['20DayEMA_VWAP'] = df['VWAP'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Final Alpha
    df['FinalAlpha'] = (
        df['VolumeAdjMomentum'] -
        df['20DayStdOpenCloseRange'] +
        df['MomentumBasedOnClose'] * (1 / df['IntradayRangeVolatility']) +
        df['MomentumBasedOnClose'] +
        df['PriceReversals'] +
        df['20DayEMA_VWAP']
    )
    
    return df['FinalAlpha']
```
