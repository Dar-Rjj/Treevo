import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute VWAP
    df['VWAP'] = (df['close'] * df['volume']) / df['volume']
    
    # Compute Daily Log Returns for VWAP
    df['VWAP_Log_Returns'] = np.log(df['VWAP']).diff()
    
    # 20-Day EMA of VWAP Log Returns
    df['VWAP_EMA_20'] = df['VWAP_Log_Returns'].ewm(span=20).mean()
    
    # Volume-Adjusted Momentum with Enhanced VWAP
    df['Volume_Adjusted_Momentum'] = df['VWAP_EMA_20'] * df['volume']
    
    # Daily High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Intraday Volatility Metric
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['volume']
    
    # Rolling 14-Day High-to-Low Range Sum
    df['14_Day_HL_Sum'] = df['High_Low_Range'].rolling(window=14).sum()
    
    # Difference between Consecutive VWAP
    df['Consecutive_VWAP_Diff'] = df['VWAP'].diff()
    
    # Adjust by Inverse of 14-Day High-to-Low Range Sum and Intraday Volatility
    df['HL_Range_Momentum_Adjusted'] = df['Consecutive_VWAP_Diff'] / (df['14_Day_HL_Sum'] + df['Intraday_Volatility'])
    
    # 14-Day Momentum
    df['14_Day_Momentum'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # 60-Day Momentum
    df['60_Day_Momentum'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # 20-Day Moving Average of Daily High-Low Ranges
    df['20_Day_MA_High_Low_Range'] = df['High_Low_Range'].rolling(window=20).mean()
    
    # 20-Day Moving Average of Daily Open-Close Ranges
    df['20_Day_MA_Open_Close_Range'] = (df['open'] - df['close']).abs().rolling(window=20).mean()
    
    # Subtract Average of 20-Day Moving Averages of Ranges
    df['Average_20_Day_MA_Range'] = (df['20_Day_MA_High_Low_Range'] + df['20_Day_MA_Open_Close_Range']) / 2
    
    # 10-day Standard Deviation of VWAP Log Returns
    df['VWAP_Log_Returns_STD_10'] = df['VWAP_Log_Returns'].rolling(window=10).std()
    
    # Multiply by Inverse of 10-day Standard Deviation of VWAP Log Returns
    df['Composite_Volatility_Adjustments'] = df['Average_20_Day_MA_Range'] / df['VWAP_Log_Returns_STD_10']
    
    # Final Alpha Factor Combination
    df['Final_Alpha_Factor'] = (
        df['Volume_Adjusted_Momentum'] +
        df['HL_Range_Momentum_Adjusted'] +
        df['14_Day_Momentum'] +
        df['60_Day_Momentum'] -
        df['Composite_Volatility_Adjustments']
    )
    
    return df['Final_Alpha_Factor'].dropna()
