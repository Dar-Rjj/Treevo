```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Rolling Median of High-to-Low Range (14-day window)
    df['high_low_range_median'] = df['high_low_range'].rolling(window=14).median()
    
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day Exponential Moving Average of Price Change
    df['ema_5'] = df['price_change'].ewm(span=5, adjust=False).mean()
    
    # Compute 10-Day Exponential Moving Average of Price Change
    df['ema_10'] = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ema_5'] > df['ema_10'], 'reversal_signal'] = 1  # Bullish trend
    df.loc[df['ema_5'] < df['ema_10'], 'reversal_signal'] = -1  # Bearish trend
    
    # Amplify or dampen the reversal signal based on high-to-low range median
    df['reversal_signal'] *= df['high_low_range_median'].apply(lambda x: 2 if x > df['high_low_range_median'].median() else 0.5)
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)  # Example threshold: 75th percentile
    df.loc[df['volume'] <= volume_threshold, 'reversal_signal'] = 0  # Only consider significant signals
    
    return df['reversal_signal']
```
