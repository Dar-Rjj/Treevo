```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    abs_intraday_range = df['high'] - df['low']
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    adjusted_intraday_volatility = abs_intraday_range / avg_volume_14
    
    # Integrate High-to-Low Range Momentum
    rolling_high_low_range = (df['high'] - df['low']).rolling(window=14).mean()
    vwp_diff = vwp.diff(1)
    high_low_momentum = vwp_diff / (adjusted_intraday_volatility * rolling_high_low_range)
    
    # Calculate Volume-Adjusted Momentum
    vwp_ema_20 = vwp.ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = vwp - vwp_ema_20
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=10).std()
    momentum_adj_volatility = volume_adjusted_momentum * (1 / recent_volatility)
    
    # Combine Volume-Adjusted Momentum and Simplified Volatility
    high_low_std_20 = (df['high'] - df['low']).rolling(window=20).std()
    final_factor = momentum_adj_volatility - high_low_std_20
    
    return final_factor
```
