```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day simple moving average (SMA) of the close price
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Define a momentum factor as the difference between the current close price and the 10-day SMA
    df['momentum_factor'] = df['close'] - df['SMA_10']
    
    # Use the sign of the momentum factor to predict continuation or reversion
    df['momentum_sign'] = df['momentum_factor'].apply(lambda x: 1 if x > 0 else -1)
    
    # Define high-activity days as those where the volume is above its 30-day simple moving average
    df['SMA_vol_30'] = df['volume'].rolling(window=30).mean()
    df['high_activity'] = df['volume'] > df['SMA_vol_30']
    
    # On high-activity days, calculate the percentage change in the closing price from the previous day
    df['prev_close'] = df['close'].shift(1)
    df['price_change_pct'] = (df['close'] - df['prev_close']) / df['prev_close']
    
    # Adjust the momentum score based on high-activity days and percentage change
    df['adjusted_momentum_score'] = df['momentum_sign']
    df.loc[df['high_activity'], 'adjusted_momentum_score'] += df['price_change_pct'].apply(lambda x: 1 if x > 0 else -1)
    
    # Incorporate the open, high, low, and close prices to measure intraday market sentiment
    df['daily_range'] = df['high'] - df['low']
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['directional_bias'] = df.apply(lambda row: 1 if row['close'] > row['midpoint'] else -1, axis=1)
    
    # Combine the directional bias indicator with the volume-based momentum adjustment
    df['composite_signal'] = df['adjusted_momentum_score'] * df['directional_bias']
    
    # Introduce a volatility component by calculating the average true range (ATR) over the last 10 days
    df['tr'] = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-pc': abs(df['high'] - df['prev_close']),
        'l-pc': abs(df['low'] - df['prev_close'])
    }).max(axis=1)
    df['ATR_10'] = df['tr'].rolling(window=10).mean()
    
    # Adjust the composite sentiment-adjusted momentum score by multiplying it with the ATR
    df['enhanced_momentum_score'] = df['composite_signal'] * df['ATR_10']
    
    # Return the final enhanced momentum score
    return df['enhanced_momentum_score']
```
