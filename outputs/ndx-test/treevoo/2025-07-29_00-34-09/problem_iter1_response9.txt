import pandas as pd

def heuristics_v2(df):
    # Calculate Logarithmic Return using Close price
    df['log_return'] = df['close'].apply(lambda x: np.log(x))
    df['log_return'] = df['log_return'].diff()
    
    # Weight by Volume
    df['volume_weighted_return'] = df['log_return'] * df['volume']
    
    # Calculate Upward and Downward Volume-Weighted Returns
    df['positive_vwr'] = df['volume_weighted_return'].apply(lambda x: x if x > 0 else 0)
    df['negative_vwr'] = df['volume_weighted_return'].apply(lambda x: -x if x < 0 else 0)
    
    # Sum over 14 days
    df['sum_positive_vwr'] = df['positive_vwr'].rolling(window=14).sum()
    df['sum_negative_vwr'] = df['negative_vwr'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['sum_positive_vwr'] / df['sum_negative_vwr']
    
    # Transform to RSI
    df['rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    return df['rsi']
