```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Volume-Weighted Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['volume_weighted_return'] = df['log_return'] * df['volume']
    
    # Aggregate Upward and Downward Volume-Weighted Returns over 14 days
    df['positive_returns'] = df['volume_weighted_return'].apply(lambda x: x if x > 0 else 0)
    df['negative_returns'] = df['volume_weighted_return'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['positive_returns'].rolling(window=14).sum()
    df['avg_loss'] = df['negative_returns'].rolling(window=14).sum()
    
    # Compute RSI
    df['relative_strength'] = df['avg_gain'] / df['avg_loss']
    df['vrsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Calculate Volume Adjusted Momentum
    df['ema_log_return'] = df['log_return'].ewm(span=20, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['ema_log_return'] * df['volume']
    
    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Combine Volume-Adjusted RSI, Momentum, and Volatility
    df['intermediate_alpha'] = df['vrsi'] + df['volume_adjusted_momentum'] - df['volatility']
    
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['open']
    
    # Compute 5-Day Moving Average of Price Change
    df['ma_5_price_change'] = df['price_change'].rolling(window=5).mean()
    
    # Compute 10-Day Moving Average of Price Change
    df['ma_10_price_change'] = df['price_change'].rolling(window=10).mean()
    
    # Determine Reversal Signal
    df['reversal_signal'] = 0
    df.loc[df['ma_5_price_change'] > df['ma_10_price_change'], 'reversal_signal'] = 1
    df.loc[df['ma_5_price_change'] < df['ma_10_price_change'], 'reversal_signal'] = -1
    
    # Integrate Reversal Signal with Intermediate Alpha Factor
    df['final_alpha'] = df['intermediate_alpha'] + df['reversal_signal'] * 0.5 * df['intermediate_alpha']
    
    return df['final_alpha'].dropna()
```
