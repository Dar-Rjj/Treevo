```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Compute Volume-Adjusted Momentum
    ema_20_vwp = vwp.ewm(span=20, adjust=False).mean()
    momentum = vwp - ema_20_vwp
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    std_10 = daily_returns.rolling(window=10).std()
    inverse_volatility = 1 / std_10
    adjusted_momentum = momentum * inverse_volatility
    
    # Incorporate Intraday Range Volatility
    intraday_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    intraday_volatility = intraday_range / avg_volume
    inverse_intraday_volatility = 1 / intraday_volatility
    
    # Assess Momentum Based on Close Price
    ma_20_close = df['close'].rolling(window=20).mean()
    close_momentum = df['close'] - ma_20_close
    
    # Analyze Price Reversals Using High and Low Prices
    price_reversals = (df['high'] - df['low']) / df['open']
    
    # Evaluate Volume Impact on Price Movement
    ema_20_vwp_close = (df['close'] * df['volume']).ewm(span=20, adjust=False).mean() / df['volume'].ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Final Alpha
    final_alpha = (adjusted_momentum * inverse_intraday_volatility +
                   close_momentum +
                   price_reversals +
                   ema_20_vwp_close)
    
    return final_alpha
```
