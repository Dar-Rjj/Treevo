```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Typical Price
    df['Typical_Price'] = (df['High'] + df['Low'] + df['Close']) / 3
    
    # Calculate Volume Adjusted Price (VAP)
    df['VAP'] = df['Typical_Price'] * df['Volume']
    
    # Calculate 20-day Exponential Moving Average (EMA) of VAP
    df['VAP_20_EMA'] = df['VAP'].ewm(span=20, adjust=False).mean()
    
    # Calculate Volume Adjusted Momentum
    df['Vol_Adjusted_Momentum'] = df['VAP'] - df['VAP_20_EMA']
    
    # Calculate daily returns
    df['Daily_Returns'] = df['Close'].pct_change()
    
    # Calculate 10-day Standard Deviation of daily returns
    df['10_Day_Volatility'] = df['Daily_Returns'].rolling(window=10).std()
    
    # Invert the 10-day Volatility and multiply by momentum
    df['Vol_Adjusted_Momentum'] *= 1 / df['10_Day_Volatility']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['High'] - df['Low']
    
    # Compute 20-Day Moving Standard Deviation of High-Low Ranges
    df['HL_Range_20_STD'] = df['High_Low_Range'].rolling(window=20).std()
    
    # Combine Volume Adjusted Momentum and High-Low Range Volatility
    df['Alpha_Factor'] = df['Vol_Adjusted_Momentum'] - df['HL_Range_20_STD']
    
    return df['Alpha_Factor']
```
