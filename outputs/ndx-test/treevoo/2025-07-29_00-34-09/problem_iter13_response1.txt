import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])

    # Assess Intraday Volatility
    intraday_range = df['high'] - df['low']
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    intraday_volatility = intraday_range / avg_volume_14

    # Integrate High-to-Low Range Momentum
    high_low_range_avg_14 = (df['high'] - df['low']).rolling(window=14).mean()
    vwp_diff = vwp - vwp.shift(1)
    range_momentum = vwp_diff / (intraday_volatility * high_low_range_avg_14)

    # Calculate Volume-Adjusted Momentum
    vwp_ema_20 = vwp.ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = vwp - vwp_ema_20

    # Incorporate Recent Volatility
    daily_returns = vwp.pct_change()
    recent_volatility = daily_returns.rolling(window=10).std()
    adjusted_momentum = volume_adjusted_momentum / recent_volatility

    # Calculate High-Low Range Volatility
    high_low_volatility = (df['high'] - df['low']).rolling(window=20).std()

    # Calculate Open-Close Range Volatility
    open_close_volatility = (df['open'] - df['close']).rolling(window=20).std()

    # Combine Volume-Adjusted Momentum and Composite Volatility
    composite_volatility = (high_low_volatility + open_close_volatility) / 2
    alpha_factor = adjusted_momentum - composite_volatility

    return alpha_factor
