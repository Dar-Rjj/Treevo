```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short-term momentum (5-day return)
    df['short_term_momentum'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate long-term reversal (20-day return)
    df['long_term_reversal'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Generate a combined factor
    df['combined_factor'] = df['short_term_momentum'] + df['long_term_reversal']
    
    # Consider the impact of volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_impact'] = df['5_day_avg_volume'] > df['20_day_avg_volume']
    df['amplified_factor'] = df['combined_factor'] * (1.5 if df['volume_impact'] else 1.0)
    
    # Consider the impact of daily price range
    df['daily_range'] = df['high'] - df['low']
    df['5_day_avg_range'] = df['daily_range'].rolling(window=5).mean()
    df['20_day_avg_range'] = df['daily_range'].rolling(window=20).mean()
    df['range_impact'] = df['5_day_avg_range'] > df['20_day_avg_range']
    df['final_alpha_factor'] = df['amplified_factor'] * (1.2 if df['range_impact'] else 1.0)
    
    # Return the final alpha factor
    return df['final_alpha_factor']
```
