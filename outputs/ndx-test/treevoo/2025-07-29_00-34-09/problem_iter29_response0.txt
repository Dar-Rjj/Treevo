```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns weighted by Volume
    df['volume_weighted_log_returns'] = df['log_returns'] * df['volume']
    df['ema_log_returns'] = df['volume_weighted_log_returns'].ewm(span=20, min_periods=20).mean()
    
    # Daily High-Low Range and 20-Day Moving SD of High-Low Ranges
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=20, min_periods=20).std()
    
    # Volume-Adjusted Price Movement
    df['volume_adjusted_price_movement'] = df['ema_log_returns'] / df['high_low_range_volatility']
    
    # 10-day Simple Moving Average (SMA) of the Close Price
    df['sma_10_close'] = df['close'].rolling(window=10, min_periods=10).mean()
    
    # Momentum Factor
    df['momentum_factor'] = df['close'] - df['sma_10_close']
    
    # Identify High-Activity Days
    df['volume_30_sma'] = df['volume'].rolling(window=30, min_periods=30).mean()
    df['is_high_activity'] = df['volume'] > df['volume_30_sma']
    
    # Percentage Change in Close Price on High-Activity Days
    df['percentage_change_close'] = df['close'].pct_change()
    df['adjusted_momentum_score'] = df['momentum_factor'] * df['is_high_activity'] * df['percentage_change_close']
    
    # Analyze Intraday Market Sentiment
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['sentiment'] = np.where(df['close'] > df['midpoint'], 1, -1)
    
    # Composite Sentiment-Adjusted Momentum
    df['composite_momentum'] = df['momentum_factor'] * df['sentiment']
    
    # Filter by Volume
    df['significant_signal'] = df['composite_momentum'] * (df['volume'] > df['volume_30_sma'])
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversal'] = df['price_reversal'].rolling(window=20, min_periods=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20, min_periods=20).mean()
    
    # Calculate Opening Gap Volatility
    df['open_to_prev_close_diff'] = df['open'] - df['close'].shift(1)
    df['opening_gap_volatility'] = df['open_to_prev_close_diff'].rolling(window=20, min_periods=20).std()
    
    # Integrate All Factors for Final Alpha
    df['alpha'] = (
        0.4 * df['volume_adjusted_price_movement'] +
        0.1 * df['sum_price_reversal'] +
        0.3 * df['price_reversal'] +
        0.05 * df['volume_weighted_close'] +
        0.05 * df['composite_momentum'] +
        0.05 * df['high_low_range_volatility'] +
        0.05 * df['opening_gap_volatility']
    )
    
    return df['alpha']
```
