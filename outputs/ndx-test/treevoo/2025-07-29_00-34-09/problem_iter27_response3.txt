import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns
    df['ema_log_returns'] = df['log_returns'].ewm(span=20, adjust=False).mean()
    
    # Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = df['ema_log_returns'] * df['volume']
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Opening Gap Volatility
    df['open_gap'] = df['open'] - df['close'].shift(1)
    df['opening_gap_volatility'] = df['open_gap'].rolling(window=20).std()
    
    # Volume Adjusted Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    
    # Integrate Price Action and Volume
    df['integrated_factor'] = (df['volume_adjusted_momentum'] +
                               df['high_low_volatility'] +
                               df['opening_gap_volatility'] +
                               df['volume_adjusted_price_change'])
    
    # 50-Day Simple Moving Average
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # 200-Day Simple Moving Average
    df['sma_200'] = df['close'].rolling(window=200).mean()
    
    # SMA Crossover Signal
    df['sma_crossover_signal'] = df['sma_50'] - df['sma_200']
    
    # Incorporate Price Trend Strength
    df['alpha_factor'] = df['integrated_factor'] * np.sign(df['sma_crossover_signal'])
    
    return df['alpha_factor'].dropna()
