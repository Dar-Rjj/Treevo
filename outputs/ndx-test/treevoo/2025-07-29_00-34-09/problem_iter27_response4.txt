```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = (df['high'] - df['low']) / df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low'] * df['volume'] / (df['high'] + df['low'])
    five_day_sequential_momentum = df['close'].diff().shift(-1).rolling(window=5).sum() / 5
    combined_momentum = (intraday_momentum * df['amount'].rolling(window=5).mean()) + (five_day_sequential_momentum * df['volume'])

    # Daily Price Change
    daily_price_change = df['close'] - df['open']

    # Compute 5-Day and 10-Day EMAs of Price Change
    ema_5 = daily_price_change.ewm(span=5, adjust=False).mean()
    ema_10 = daily_price_change.ewm(span=10, adjust=False).mean()

    # Determine Reversal Signal
    reversal_signal = pd.Series(0, index=df.index)
    reversal_signal[(ema_5 > ema_10) & (ema_5.shift(1) < ema_10.shift(1))] = 1  # Bullish trend
    reversal_signal[(ema_5 < ema_10) & (ema_5.shift(1) > ema_10.shift(1))] = -1  # Bearish trend
    reversal_signal *= high_low_range.rolling(window=14).median().rank(pct=True)

    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    reversal_signal[df['volume'] < volume_threshold] = 0

    # Cumulative Weighted Momentum
    weighted_return = daily_price_change * df['volume']
    cumulative_weighted_momentum = weighted_return.rolling(window=5).sum() / df['volume'].rolling(window=5).mean()

    # Incorporate Volatility into Alpha Factor
    historical_volatility = df['close'].rolling(window=21).std()
    volatility_adjusted_factor = 1 / historical_volatility

    # Final Alpha Factor
    final_alpha_factor = (combined_momentum + cumulative_weighted_momentum + reversal_signal) * volatility_adjusted_factor

    return final_alpha_factor
```
