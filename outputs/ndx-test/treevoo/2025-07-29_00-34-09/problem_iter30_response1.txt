```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns for VWAP
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['VWAP_log_returns'] = np.log(df['VWAP'] / df['VWAP'].shift(1))
    
    # 20-Day EMA of VWAP Log Returns
    df['VWAP_ema20'] = df['VWAP_log_returns'].ewm(span=20).mean()
    
    # Volume-Adjusted Momentum with Enhanced VWAP
    df['Volume_Adjusted_Momentum'] = df['VWAP_ema20'] * df['volume']
    
    # High-Low Range and Intraday Volatility Synthesis
    df['High_Low_Range'] = df['high'] - df['low']
    df['Intraday_Volatility'] = (df['high'] - df['low']) / (df['volume'] * df['close'])
    
    # Adjust by Inverse of 14-Day High-to-Low Range Sum and Intraday Volatility
    df['Rolling_14D_HL_Range_Sum'] = df['High_Low_Range'].rolling(window=14).sum()
    df['Adjusted_High_to_Low_Range'] = (df['High_Low_Range'] - df['Intraday_Volatility']) / df['Rolling_14D_HL_Range_Sum']
    
    # Recent Price Momentum
    df['Momentum_14'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    df['Momentum_60'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)
    
    # 20-Day Moving Averages of Ranges
    df['MA20_High_Low_Range'] = df['High_Low_Range'].rolling(window=20).mean()
    df['MA20_Open_Close_Range'] = (df['open'] - df['close']).abs().rolling(window=20).mean()
    
    # Subtract Average of 20-Day Moving Averages of Ranges
    df['Composite_Volatility_Adjustment'] = (df['MA20_High_Low_Range'] + df['MA20_Open_Close_Range']) / 2
    
    # 10-day Standard Deviation of VWAP Log Returns
    df['VWAP_Log_Returns_Std_10'] = df['VWAP_log_returns'].rolling(window=10).std()
    
    # Combine Volume-Adjusted Momentum and Composite Volatility Adjustments
    df['Final_Alpha_Factor'] = (
        df['Volume_Adjusted_Momentum'] +
        df['Intraday_Volatility'] +
        df['Adjusted_High_to_Low_Range'] +
        (df['Momentum_14'] + df['Momentum_60']) / 2 -
        df['Composite_Volatility_Adjustment']
    ) * (1 / df['VWAP_Log_Returns_Std_10'])
    
    return df['Final_Alpha_Factor'].dropna()
```
