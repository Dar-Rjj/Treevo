```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Change Ratio
    high_low_ratio = (df['high'] - df['low']) / df['low']
    intraday_activity = df['volume'] / (df['high'] + df['low'])
    intraday_price_change_ratio = high_low_ratio * intraday_activity
    
    # Measure Sustained Momentum
    sustained_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Weighted Momentum
    volume_weighted_momentum = sustained_momentum * df['volume']
    
    # Calculate Upward and Downward Volume-Weighted Returns
    positive_returns = sustained_momentum.where(sustained_momentum > 0, 0) * df['volume']
    negative_returns = sustained_momentum.where(sustained_momentum < 0, 0).abs() * df['volume']
    
    # Sum over 14 days
    sum_positive_returns = positive_returns.rolling(window=14).sum()
    sum_negative_returns = negative_returns.rolling(window=14).sum()
    
    # Compute RSI
    relative_strength = sum_positive_returns / sum_negative_returns
    rsi = 100 - (100 / (1 + relative_strength))
    
    # Combine Factors
    alpha_factor = (
        intraday_price_change_ratio +
        sustained_momentum +
        volume_weighted_momentum +
        rsi
    )
    
    return alpha_factor
```
