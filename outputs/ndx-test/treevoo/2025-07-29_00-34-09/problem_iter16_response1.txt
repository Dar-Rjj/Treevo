```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    df['daily_range'] = df['high'] - df['low']

    # Compute High-Low Ratio and Intraday Trading Activity
    df['high_low_ratio'] = (df['high'] - df['low']) / df['low']
    df['intraday_trading_activity'] = df['volume'] / (df['high'] + df['low'])

    # Measure Sustained 5-Day Momentum
    df['sequential_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_day_momentum'] = df['sequential_momentum'].rolling(window=5).sum() / 5

    # Combine Intraday and 5-Day Momentum
    df['intraday_momentum'] = df['high_low_ratio'] * df['intraday_trading_activity']
    df['sustained_5_day_momentum'] = df['5_day_momentum'] * df['volume']
    df['combined_momentum'] = df['intraday_momentum'] + df['sustained_5_day_momentum']

    # Calculate Daily Price Change
    df['daily_change'] = df['close'] - df['open']

    # Compute 5-Day and 10-Day Exponential Moving Averages of Price Change
    df['ema_5'] = df['daily_change'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['daily_change'].ewm(span=10, adjust=False).mean()

    # Determine Reversal Signal
    df['reversal_signal'] = np.where(df['ema_5'] > df['ema_10'], 1, -1)
    df['range_14_median'] = df['daily_range'].rolling(window=14).median()
    df['reversal_signal_adjusted'] = df['reversal_signal'] * (1 + (df['range_14_median'] / df['daily_range']))

    # Filter by Volume
    volume_threshold = df['volume'].mean()
    df['filtered_reversal_signal'] = np.where(df['volume'] > volume_threshold, df['reversal_signal_adjusted'], 0)

    # Calculate Cumulative Weighted Momentum
    df['cumulative_weighted_momentum'] = df['daily_change'].rolling(window=5).sum() / df['amount'].rolling(window=5).mean()

    # Incorporate Volatility into Alpha Factor
    df['historical_volatility'] = df['close'].rolling(window=21).std()
    df['volatility_adjusted_factor'] = df['combined_momentum'] + df['cumulative_weighted_momentum'] + df['filtered_reversal_signal']
    df['final_alpha_factor'] = df['volatility_adjusted_factor'] / df['historical_volatility']

    return df['final_alpha_factor']
```
