```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Adjusted Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['adj_return'] = ((df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1))) * 100
    
    # Apply Volume Smoothing
    df['volume_ewma'] = df['volume'].ewm(span=7).mean()
    df['norm_volume'] = df['volume'] / df['volume_ewma']
    
    # Weighted Returns Calculation
    df['weighted_returns'] = df['adj_return'] * df['norm_volume']
    df['sum_weighted_returns'] = df['weighted_returns'].rolling(window=14).sum()
    
    # Separate Positive and Negative Weights
    df['positive_weighted_returns'] = df['weighted_returns'].apply(lambda x: x if x > 0 else 0)
    df['negative_weighted_returns'] = df['weighted_returns'].apply(lambda x: -x if x < 0 else 0)
    df['sum_positive_weighted_returns'] = df['positive_weighted_returns'].rolling(window=14).sum()
    df['sum_negative_weighted_returns'] = df['negative_weighted_returns'].rolling(window=14).sum()
    
    # Compute Dynamic RSI
    df['relative_strength'] = df['sum_positive_weighted_returns'] / df['sum_negative_weighted_returns']
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Calculate Volume Adjusted Price (VAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vap'] = df['typical_price'] * df['volume']
    
    # Calculate Volume Adjusted Momentum
    df['vap_ema_20'] = df['vap'].ewm(span=20).mean()
    df['vap_momentum'] = df['vap'] - df['vap_ema_20']
    
    # Incorporate Recent Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility'] = df['daily_return'].rolling(window=10).std()
    df['volatility_inverse'] = 1 / df['volatility']
    df['momentum_adjusted'] = df['vap_momentum'] * df['volatility_inverse']
    
    # Combine Volume Adjusted Momentum, Dynamic RSI, and Composite Volatility
    df['alpha_factor'] = (df['vap_momentum'] + df['dynamic_rsi'] - df['volatility'].mean())
    
    return df['alpha_factor']
```
