```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Price (VAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vap = typical_price * df['volume']
    
    # Calculate Volume-Adjusted Momentum
    vap_ema_20 = vap.ewm(span=20, min_periods=20).mean()
    momentum = vap - vap_ema_20
    
    # Incorporate Recent Volatility
    daily_returns = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    recent_volatility = daily_returns.rolling(window=10, min_periods=10).std()
    adjusted_momentum = momentum / (recent_volatility + 1e-6)  # Add small constant to avoid division by zero
    
    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=20, min_periods=20).std()
    
    # Calculate Open-Close Range Volatility
    open_close_range = df['open'] - df['close']
    open_close_volatility = open_close_range.rolling(window=20, min_periods=20).std()
    
    # Combine Volume-Adjusted Momentum and Composite Volatility
    composite_volatility = (high_low_volatility + open_close_volatility) / 2
    combined_factor = adjusted_momentum - composite_volatility
    
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Rolling Average of High-to-Low Range
    rolling_high_to_low_avg = high_to_low_range.rolling(window=20, min_periods=20).mean()
    
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Compute 5-Day Moving Average of Volume-Weighted Price Change
    volume_weighted_price_change_5 = (daily_price_change * df['volume']).rolling(window=5, min_periods=5).sum() / df['volume'].rolling(window=5, min_periods=5).sum()
    
    # Compute 15-Day Moving Average of Volume-Weighted Price Change
    volume_weighted_price_change_15 = (daily_price_change * df['volume']).rolling(window=15, min_periods=15).sum() / df['volume'].rolling(window=15, min_periods=15).sum()
    
    # Determine Reversal Signal
    reversal_signal = pd.Series(0, index=df.index)
    reversal_signal[volume_weighted_price_change_5 > volume_weighted_price_change_15] = 1
    reversal_signal[volume_weighted_price_change_5 < volume_weighted_price_change_15] = -1
    reversal_signal = reversal_signal * (rolling_high_to_low_avg / rolling_high_to_low_avg.mean())
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    combined_factor_filtered = combined_factor[df['volume'] > volume_threshold]
    reversal_signal_filtered = reversal_signal[df['volume'] > volume_threshold]
    
    # Final factor
    final_factor = combined_factor_filtered + reversal_signal_filtered
    return final_factor
```
