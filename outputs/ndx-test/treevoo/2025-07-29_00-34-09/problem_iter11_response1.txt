```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Price (VAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vap = typical_price * df['volume']
    
    # Calculate Volume-Adjusted Momentum
    ema_20_vap = vap.ewm(span=20, adjust=False).mean()
    momentum = vap - ema_20_vap
    
    # Incorporate Recent Volatility
    daily_returns = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    volatility_10 = daily_returns.rolling(window=10).std()
    inv_vol_adj_momentum = momentum * (1 / volatility_10)
    
    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_vol = high_low_range.rolling(window=20).std()
    
    # Calculate Open-Close Range Volatility
    open_close_range = abs(df['open'] - df['close'])
    open_close_vol = open_close_range.rolling(window=20).std()
    
    # Calculate Rolling Average of High-to-Low Range
    hl_rolling_avg = (df['high'] - df['low']).rolling(window=14).mean()
    
    # Calculate Daily Price Change
    daily_change = df['close'] - df['open']
    
    # Compute 5-Day Moving Average of Price Change
    ma_5_change = daily_change.rolling(window=5).mean()
    
    # Compute 10-Day Moving Average of Price Change
    ma_10_change = daily_change.rolling(window=10).mean()
    
    # Determine Reversal Signal
    reversal_signal = np.where(ma_5_change > ma_10_change, 1, np.where(ma_5_change < ma_10_change, -1, 0))
    reversal_signal = reversal_signal * (hl_rolling_avg / hl_rolling_avg.max())
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    reversal_signal = np.where(df['volume'] >= volume_threshold, reversal_signal, 0)
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    price_reversals = (df['high'] - df['low']) / df['open']
    price_reversals = price_reversals.rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    volume_weighted_close = (df['close'] * df['volume']).ewm(span=20, adjust=False).mean()
    
    # Combine Factors for Final Alpha
    alpha = (inv_vol_adj_momentum * 0.4) + (reversal_signal * 0.2) + (price_reversals * 0.3) + (volume_weighted_close * 0.1)
    
    return alpha
```
