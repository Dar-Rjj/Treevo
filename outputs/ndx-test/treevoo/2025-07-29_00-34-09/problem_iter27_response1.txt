import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # 20-Day EMA of Log Returns, Weigh by Volume
    df['weighted_log_returns'] = df['log_returns'] * df['volume']
    df['ema_weighted_log_returns'] = df['weighted_log_returns'].ewm(span=20).mean()
    
    # High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_volatility'] = df['high_low_range'].rolling(window=20).std()
    
    # Weight by High-Low Range Volatility
    df['volume_adjusted_price_movement'] = (df['ema_weighted_log_returns'] / df['high_low_volatility']).fillna(0)
    
    # Calculate 10-day Simple Moving Average (SMA) of the Close Price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Define a Momentum Factor
    df['momentum'] = df['close'] - df['sma_10']
    
    # Identify High-Activity Days
    df['volume_sma_30'] = df['volume'].rolling(window=30).mean()
    df['high_activity_day'] = (df['volume'] > df['volume_sma_30']).astype(int)
    
    # Calculate Percentage Change in Close Price on High-Activity Days
    df['percentage_change_high_activity'] = df['close'].pct_change().where(df['high_activity_day'] == 1, 0)
    df['adjusted_momentum'] = df['momentum'] * df['percentage_change_high_activity']
    
    # Analyze Intraday Market Sentiment
    df['daily_midpoint'] = (df['high'] + df['low']) / 2
    df['directional_bias'] = np.where(df['close'] > df['daily_midpoint'], 1, -1)
    
    # Combine Momentum and Sentiment for Composite Signal
    df['composite_sentiment_adjusted_momentum'] = df['adjusted_momentum'] * df['directional_bias']
    
    # Filter by Volume
    df['significant_signal'] = df['composite_sentiment_adjusted_momentum'].where(df['volume'] > df['volume_sma_30'], 0)
    
    # Analyze Price Reversals Using High, Low, and Open Prices
    df['price_reversal'] = (df['high'] - df['low']) / df['open']
    df['sum_price_reversal'] = df['price_reversal'].rolling(window=20).sum()
    
    # Assess Volume Impact on Price Movement
    df['volume_weighted_close'] = (df['close'] * df['volume']).ewm(span=20).mean()
    
    # Integrate Volume-Price Correlation
    df['volume_price_corr'] = df['close'].rolling(window=20).corr(df['volume'])
    
    # Combine Factors for Final Alpha
    df['final_alpha'] = (
        0.4 * df['volume_adjusted_price_movement'] +
        0.1 * df['significant_signal'] +
        0.3 * df['sum_price_reversal'] +
        0.1 * df['volume_weighted_close'] +
        0.1 * df['composite_sentiment_adjusted_momentum'] +
        0.05 * df['volume_price_corr']
    ).fillna(0)
    
    return df['final_alpha']
