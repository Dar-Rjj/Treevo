```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Adjusted Return
    df['adj_return'] = (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1)) * 100
    
    # Apply Volume Smoothing
    df['volume_ewm'] = df['volume'].ewm(span=7).mean()
    df['normalized_volume'] = df['volume'] / df['volume_ewm']
    
    # Weighted Returns Calculation
    df['weighted_returns'] = df['adj_return'] * df['normalized_volume']
    df['sum_weighted_returns'] = df['weighted_returns'].rolling(window=14).sum()
    
    # Introduce Momentum Component
    df['adj_return_ewm'] = df['adj_return'].ewm(span=14).mean()
    df['combined_weighted_returns'] = df['sum_weighted_returns'] + df['adj_return_ewm']
    
    # Separate Positive and Negative Weights
    positive_weights = df['combined_weighted_returns'].clip(lower=0)
    negative_weights = df['combined_weighted_returns'].clip(upper=0).abs()
    
    # Compute Dynamic RSI
    df['positive_sum'] = positive_weights.rolling(window=14).sum()
    df['negative_sum'] = negative_weights.rolling(window=14).sum()
    df['relative_strength'] = df['positive_sum'] / df['negative_sum']
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    return df['dynamic_rsi']
```
