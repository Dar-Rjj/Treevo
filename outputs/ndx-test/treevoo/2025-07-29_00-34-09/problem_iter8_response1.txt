```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust by Previous Day's High and Low
    df['adjusted_return'] = (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1)) * 100
    
    # Calculate 7-Day Exponentially Weighted Moving Average of Volumes
    df['ewm_volume'] = df['volume'].ewm(span=7, adjust=False).mean()
    # Calculate 7-Day Exponentially Weighted Moving Average of Amounts
    df['ewm_amount'] = df['amount'].ewm(span=7, adjust=False).mean()
    
    # Normalize Today's Volume and Amount by 7-Day Exponentially Weighted Averages
    df['normalized_volume'] = df['volume'] / df['ewm_volume']
    df['normalized_amount'] = df['amount'] / df['ewm_amount']
    
    # Weighted Returns Calculation
    df['weighted_return'] = df['adjusted_return'] * df['normalized_volume'] * df['normalized_amount']
    
    # Sum Over 14 Days
    df['sum_positive_weighted_returns'] = df[df['weighted_return'] > 0]['weighted_return'].rolling(window=14, min_periods=1).sum()
    df['sum_negative_weighted_returns'] = df[df['weighted_return'] < 0]['weighted_return'].abs().rolling(window=14, min_periods=1).sum()
    
    # Compute Relative Strength
    df['relative_strength'] = df['sum_positive_weighted_returns'] / df['sum_negative_weighted_returns']
    
    # Transform to Dynamic RSI
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    return df['dynamic_rsi']
```
