```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    abs_intraday_range = df['high'] - df['low']
    median_volume_14 = df['volume'].rolling(window=14).median()
    intraday_volatility_metric = abs_intraday_range / median_volume_14
    
    # Integrate High-to-Low Range Momentum
    rolling_high_low_range_sum = (df['high'] - df['low']).rolling(window=20).sum()
    consecutive_vwp_diff = vwp.diff()
    high_low_momentum = consecutive_vwp_diff / (intraday_volatility_metric * rolling_high_low_range_sum)
    
    # Calculate Volume-Adjusted Momentum
    vwp_ema_25 = vwp.ewm(span=25, adjust=False).mean()
    volume_adjusted_momentum = vwp - vwp_ema_25
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility_15 = daily_returns.rolling(window=15).std()
    volatility_adjusted_momentum = volume_adjusted_momentum / (1 + recent_volatility_15)
    
    # Calculate High-Low Range Volatility
    daily_high_low_range = df['high'] - df['low']
    ma_high_low_30 = daily_high_low_range.rolling(window=30).mean()
    
    # Calculate Open-Close Range Volatility
    daily_open_close_range = df['open'] - df['close']
    ma_open_close_30 = daily_open_close_range.rolling(window=30).mean()
    
    # Combine Volume-Adjusted Momentum and Enhanced Composite Volatility
    weighted_average_volatility = (ma_high_low_30 * 0.6) + (ma_open_close_30 * 0.4)
    alpha_factor = volatility_adjusted_momentum - weighted_average_volatility
    
    return alpha_factor
```
