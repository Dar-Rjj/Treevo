```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3
    
    # Assess Intraday Volatility
    intraday_range = df['high'] - df['low']
    median_volume = df['volume'].rolling(window=14).median()
    intraday_volatility = intraday_range / median_volume
    
    # Integrate High-to-Low Range Momentum
    high_low_sum_14d = (df['high'] - df['low']).rolling(window=14).sum()
    vwap_diff = vwap.diff()
    adjusted_momentum = vwap_diff / (intraday_volatility * high_low_sum_14d)
    
    # Calculate Volume-Adjusted Momentum
    ema_25 = vwap.ewm(span=25, adjust=False).mean()
    volume_adjusted_momentum = vwap - ema_25
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=15).std()
    volatility_adjusted_momentum = volume_adjusted_momentum / (recent_volatility + 1e-8)  # Avoid division by zero
    
    # Calculate High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    high_low_volatility = high_low_range.rolling(window=30).mean()
    
    # Calculate Open-Close Range Volatility
    open_close_range = df['open'] - df['close']
    open_close_volatility = open_close_range.rolling(window=30).mean()
    
    # Combine Volume-Adjusted Momentum and Enhanced Composite Volatility
    weighted_high_low = high_low_volatility * 0.6
    weighted_open_close = open_close_volatility * 0.4
    composite_volatility = weighted_high_low + weighted_open_close
    final_factor = volatility_adjusted_momentum - composite_volatility
    
    return final_factor
```
