import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = ((df['high'] + df['low']) / 2 * df['volume']) / df['volume']
    
    # Assess Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Integrate High-to-Low Range Momentum
    high_low_range_20d_sum = intraday_volatility.rolling(window=20).sum()
    vwp_diff = vwp.diff()
    adjusted_momentum = vwp_diff / high_low_range_20d_sum
    
    # Calculate Volume-Adjusted Momentum
    vwp_ema_25 = vwp.ewm(span=25, adjust=False).mean()
    volume_adjusted_momentum = vwp - vwp_ema_25
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=15).std()
    adjusted_momentum_volatility = volume_adjusted_momentum / (recent_volatility + 1e-8)  # Avoid division by zero
    
    # Combine Volume-Adjusted Momentum and High-Low Range Volatility
    high_low_range_30d_ma = (df['high'] - df['low']).rolling(window=30).mean()
    final_factor = adjusted_momentum_volatility - high_low_range_30d_ma
    
    return final_factor
