```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Adjusted Return
    df['adjusted_return'] = ((df['close'] - df['low'].shift(1)) / 
                             (df['high'].shift(1) - df['low'].shift(1))) * 100
    
    # Apply Volume Smoothing
    df['volume_ewma'] = df['volume'].ewm(span=7, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['volume_ewma']
    
    # Weighted Returns Calculation
    df['weighted_returns'] = df['adjusted_return'] * df['normalized_volume']
    df['sum_weighted_returns'] = df['weighted_returns'].rolling(window=14).sum()
    
    # Introduce Price Momentum Component
    df['price_momentum'] = df['close'].ewm(span=14, adjust=False).mean()
    df['combined_weighted_returns'] = df['sum_weighted_returns'] + df['price_momentum']
    
    # Separate Positive and Negative Weights
    df['positive_weighted_returns'] = df['combined_weighted_returns'].apply(lambda x: x if x > 0 else 0)
    df['negative_weighted_returns'] = df['combined_weighted_returns'].apply(lambda x: abs(x) if x < 0 else 0)
    
    # Compute Dynamic RSI
    df['sum_positive_returns'] = df['positive_weighted_returns'].rolling(window=14).sum()
    df['sum_negative_returns'] = df['negative_weighted_returns'].rolling(window=14).sum()
    df['relative_strength'] = df['sum_positive_returns'] / df['sum_negative_returns']
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Drop NaN values
    df.dropna(inplace=True)
    
    return df['dynamic_rsi']
```
