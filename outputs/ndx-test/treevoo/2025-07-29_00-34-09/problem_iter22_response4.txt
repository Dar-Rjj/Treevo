import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    vwp = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    abs_intraday_range = df['high'] - df['low']
    avg_volume_14 = df['volume'].rolling(window=14).mean()
    intraday_volatility = abs_intraday_range / avg_volume_14
    
    # Integrate High-to-Low Range Momentum
    high_low_range_sum_14 = (df['high'] - df['low']).rolling(window=14).sum()
    vwp_diff = vwp.diff(1)
    adjusted_momentum = vwp_diff / high_low_range_sum_14
    
    # Calculate Volume-Adjusted Momentum
    vwp_ema_20 = vwp.ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = vwp - vwp_ema_20
    
    # Incorporate Recent Volatility
    daily_returns = df['close'].pct_change()
    recent_volatility = daily_returns.rolling(window=10).std()
    momentum_adjusted_volatility = volume_adjusted_momentum / recent_volatility
    
    # Combine Volume-Adjusted Momentum and Composite Volatility
    high_low_range_ma_20 = (df['high'] - df['low']).rolling(window=20).mean()
    combined_factor = momentum_adjusted_volatility - high_low_range_ma_20
    
    return combined_factor
