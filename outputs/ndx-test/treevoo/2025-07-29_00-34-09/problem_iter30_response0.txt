```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = (df['high'] - df['low']) / df['low']
    
    # Integrate Intraday and Multi-Day Momentum
    intraday_momentum = (df['high'] - df['low']) / df['low'] * df['volume'] / (df['high'] + df['low'])
    five_day_avg_seq_momentum = df['close'].pct_change().rolling(window=5).sum() / 5
    combined_momentum = (intraday_momentum * df['amount'].rolling(window=5).mean()) + (five_day_avg_seq_momentum * df['volume'])

    # Daily Price Change
    daily_price_change = df['close'] - df['open']

    # Compute 5-Day and 10-Day EMAs of Price Change
    ema_5 = daily_price_change.ewm(span=5, adjust=False).mean()
    ema_10 = daily_price_change.ewm(span=10, adjust=False).mean()

    # Determine Enhanced Reversal Signal
    reversal_signal = np.where(ema_5 > ema_10, 1, -1)
    std_14 = high_to_low_range.rolling(window=14).std()
    enhanced_reversal_signal = reversal_signal * (1 + (std_14 - std_14.mean()) / std_14.std())

    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    filtered_reversal_signal = np.where(df['volume'] > volume_threshold, enhanced_reversal_signal, 0)

    # Cumulative Weighted Momentum
    weighted_daily_return = (df['close'] - df['open']).shift(1) * df['volume'].shift(1)
    cumulative_weighted_momentum = weighted_daily_return.rolling(window=5).sum() / df['volume'].rolling(window=5).mean()

    # Incorporate Volatility into Alpha Factor
    historical_volatility = df['close'].rolling(window=21).std()
    volatility_adjusted_factor = (combined_momentum + cumulative_weighted_momentum + filtered_reversal_signal) * (1 / historical_volatility)

    # Final Alpha Factor
    final_alpha_factor = volatility_adjusted_factor

    return final_alpha_factor
```
