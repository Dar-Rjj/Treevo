```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as market features (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the daily return
    daily_return = df['close'] / df['open'] - 1
    
    # Calculate the body of the candle
    body = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Calculate the closing price relative to the daily range
    close_to_range = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the Heuristic Alpha Factor
    heuristic_alpha_factor = daily_range * daily_return * body * close_to_range
    
    return heuristic_alpha_factor
```
