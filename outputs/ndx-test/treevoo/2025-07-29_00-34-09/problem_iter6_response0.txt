import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Adjust by Previous Day's High and Low
    df['adjusted_return'] = (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1)) * 100
    
    # Calculate 7-Day Exponentially Weighted Moving Average of Volumes
    df['volume_ewma'] = df['volume'].ewm(span=7).mean()
    
    # Normalize Today's Volume by 7-Day Exponentially Weighted Average
    df['normalized_volume'] = df['volume'] / df['volume_ewma']
    
    # Weighted Returns Calculation
    df['weighted_return'] = df['adjusted_return'] * df['normalized_volume']
    
    # Separate Positive and Negative Weights
    df['positive_weighted_return'] = df['weighted_return'].apply(lambda x: x if x > 0 else 0)
    df['negative_weighted_return'] = df['weighted_return'].apply(lambda x: abs(x) if x < 0 else 0)
    
    # Sum Over 14 Days
    df['sum_positive'] = df['positive_weighted_return'].rolling(window=14).sum()
    df['sum_negative'] = df['negative_weighted_return'].rolling(window=14).sum()
    
    # Compute Relative Strength
    df['relative_strength'] = df['sum_positive'] / df['sum_negative']
    
    # Transform to Dynamic RSI
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    # Drop unnecessary columns and return the result
    factor_values = df['dynamic_rsi'].dropna()
    return factor_values
