```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the change in average price compared to the previous day
    avg_price_change = avg_price - avg_price.shift(1)
    
    # Calculate the change in volume compared to the previous day
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Heuristic: If the change in average price is positive, and the volume increases, it's a good sign
    # Otherwise, it's a bad sign. We can use this to create a binary factor.
    factor = (avg_price_change > 0) & (volume_change > 0)
    factor = factor.astype(int)  # Convert boolean to integer (1 for True, 0 for False)
    
    return factor
```
