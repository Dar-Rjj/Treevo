import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Adjusted Return
    df['adjusted_return'] = (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1)) * 100
    
    # Apply Volume Smoothing
    df['volume_ewma'] = df['volume'].ewm(span=7, adjust=False).mean()
    df['normalized_volume'] = df['volume'] / df['volume_ewma']
    
    # Weighted Returns Calculation
    df['weighted_returns'] = df['adjusted_return'] * df['normalized_volume']
    
    # Separate Positive and Negative Weights
    df['positive_weighted_returns'] = df['weighted_returns'].where(df['weighted_returns'] > 0, 0)
    df['negative_weighted_returns'] = df['weighted_returns'].where(df['weighted_returns'] < 0, 0).abs()
    
    # Sum Over 14 Days
    df['sum_positive_14d'] = df['positive_weighted_returns'].rolling(window=14).sum()
    df['sum_negative_14d'] = df['negative_weighted_returns'].rolling(window=14).sum()
    
    # Compute Dynamic RSI
    df['relative_strength'] = df['sum_positive_14d'] / df['sum_negative_14d']
    df['dynamic_rsi'] = 100 - (100 / (1 + df['relative_strength']))
    
    return df['dynamic_rsi']
