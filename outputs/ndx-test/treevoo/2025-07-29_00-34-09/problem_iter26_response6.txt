```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_low_range = (df['high'] - df['low']) / df['low']
    
    # Intraday Momentum: (High - Low) / Low * Volume / (High + Low)
    intraday_momentum = (df['high'] - df['low']) / df['low'] * df['volume'] / (df['high'] + df['low'])
    
    # 5-Day Average Sequential Momentum: Sum((Close - Previous Close) / Previous Close) / 5
    sequential_momentum_5d = (df['close'].pct_change().rolling(window=5).sum() / 5)
    
    # Combined Momentum: (Intraday Momentum * 5-Day Average Amount) + (5-Day Momentum * Volume)
    combined_momentum = (intraday_momentum * df['amount'].rolling(window=5).mean()) + (sequential_momentum_5d * df['volume'])
    
    # Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Compute 5-Day and 10-Day EMAs of Price Change
    ema_5d = daily_price_change.ewm(span=5, adjust=False).mean()
    ema_10d = daily_price_change.ewm(span=10, adjust=False).mean()
    
    # Determine Reversal Signal
    reversal_signal = np.where(ema_5d > ema_10d, 1, np.where(ema_5d < ema_10d, -1, 0))
    
    # Consider 14-day Rolling Median of High-to-Low Range
    rolling_median_14d = high_low_range.rolling(window=14).median()
    reversal_signal_adjusted = np.where(rolling_median_14d > rolling_median_14d.mean(), reversal_signal * 1.5, reversal_signal * 0.5)
    
    # Filter by Volume
    volume_threshold = df['volume'].quantile(0.75)
    reversal_signal_filtered = np.where(df['volume'] > volume_threshold, reversal_signal_adjusted, 0)
    
    # Cumulative Weighted Momentum
    weighted_daily_return = (df['close'].pct_change() * df['amount']).rolling(window=5).sum() / df['amount'].rolling(window=5).mean()
    
    # 21-Day Historical Volatility: Standard Deviation of last 21 days' Close prices
    historical_volatility_21d = df['close'].pct_change().rolling(window=21).std()
    
    # Final Alpha Factor * (1 / 21-Day Historical Volatility)
    final_alpha_factor_vol_adjusted = (combined_momentum + weighted_daily_return) / (1 + historical_volatility_21d)
    
    # Volume Adjusted Momentum
    daily_log_returns = np.log(df['close']) - np.log(df['close'].shift(1))
    ema_20d_log_returns = daily_log_returns.ewm(span=20, adjust=False).mean()
    volume_adjusted_momentum = ema_20d_log_returns * df['volume']
    
    # Calculate High-Low Range Volatility
    high_low_range_volatility = (df['high'] - df['low']).rolling(window=20).std()
    
    # Combine Factors
    alpha_factor = (final_alpha_factor_vol_adjusted 
                    + reversal_signal_filtered 
                    + volume_adjusted_momentum 
                    + high_low_range_volatility)
    
    return pd.Series(alpha_factor, index=df.index, name='alpha_factor')
```
