```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    df['VWP'] = (df['high'] * df['volume'] + df['low'] * df['volume']) / (2 * df['volume'])
    
    # Assess Intraday Volatility
    df['intraday_range'] = df['high'] - df['low']
    df['median_volume_14d'] = df['volume'].rolling(window=14).median()
    df['intraday_volatility'] = df['intraday_range'] / df['median_volume_14d']
    
    # Integrate High-to-Low Range Momentum
    df['h_l_range_sum_20d'] = df['intraday_range'].rolling(window=20).sum()
    df['vwp_diff'] = df['VWP'].diff()
    df['momentum_adjusted'] = df['vwp_diff'] / df['intraday_volatility'] / df['h_l_range_sum_20d']
    
    # Calculate Volume-Adjusted Momentum
    df['vwp_ema_25d'] = df['VWP'].ewm(span=25, adjust=False).mean()
    df['volume_adjusted_momentum'] = df['VWP'] - df['vwp_ema_25d']
    
    # Incorporate Recent Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['volatility_15d'] = df['daily_returns'].rolling(window=15).std()
    df['adjusted_momentum'] = df['volume_adjusted_momentum'] / (df['volatility_15d'] + 1e-6)  # Add small constant to avoid division by zero
    
    # Calculate High-Low Range Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['high_low_range_volatility'] = df['high_low_range'].rolling(window=30).mean()
    
    # Calculate Open-Close Range Volatility
    df['open_close_range'] = df['open'] - df['close']
    df['open_close_range_volatility'] = df['open_close_range'].rolling(window=30).mean()
    
    # Combine Volume-Adjusted Momentum and Enhanced Composite Volatility
    df['weighted_average_volatility'] = (df['high_low_range_volatility'] * 0.6) + (df['open_close_range_volatility'] * 0.4)
    df['alpha_factor'] = df['adjusted_momentum'] - df['weighted_average_volatility']
    
    return df['alpha_factor']
```
