```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Weighted Close Price
    df['weighted_close'] = df['volume'] * df['close']
    
    # Sum of Weighted Close Prices
    volume_weighted_price = df['weighted_close'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    sma_volume_weighted_price = volume_weighted_price.rolling(window=20).mean()
    
    # Calculate Realized Volatility using High and Low Prices
    df['log_ret'] = np.log(df['high'] / df['low'])
    realized_volatility = df['log_ret'].rolling(window=20).std()
    
    # Smooth Realized Volatility with Exponential Moving Average (EMA)
    ema_realized_volatility = realized_volatility.ewm(span=20, adjust=False).mean()
    
    # Adjust SMA with Smoothed Volatility
    adjusted_sma = sma_volume_weighted_price - ema_realized_volatility
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume Surge Indicator
    df['volume_surge'] = (df['volume_change'] > 0) & (df['volume_change'] > df['volume_change'].quantile(0.9))
    
    # Adjust Final Alpha Factor
    alpha_factor = adjusted_sma.copy()
    alpha_factor[df['volume_surge']] += 1.0  # Increase Alpha Factor if Volume Surge is detected
    
    return alpha_factor.dropna()
```
