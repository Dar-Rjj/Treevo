```python
import pandas as pd

def heuristics_v2(df, window=21, vol_window=20, vol_momentum_lookback=5, integration_weight=0.5):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=window).mean()
    
    # Calculate Rolling Volatility of Close Prices
    df['rolling_volatility'] = df['close'].pct_change().rolling(window=vol_window).std() * (252**0.5)
    
    # Adjust Momentum by Volatility
    df['momentum_adjusted'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Weight by Volume
    df['volume_weighted_momentum'] = df['momentum_adjusted'] * df['volume']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Rolling Average Daily Range
    df['rolling_avg_daily_range'] = df['daily_range'].rolling(window=window).mean()
    
    # Adjust Range by Volatility
    df['range_adjusted'] = df['rolling_avg_daily_range'] / df['rolling_volatility']
    
    # Integrate Adjusted Range into Combined Factors
    df['integrated_factors'] = df['volume_weighted_momentum'] + df['range_adjusted']
    
    # Combine Factors
    df['combined_factors'] = df['integrated_factors'] + df['volume_change']
    
    # Calculate Volume Momentum
    df['volume_momentum'] = df['volume'] / df['volume'].shift(vol_momentum_lookback) - 1
    
    # Adjust Volume Momentum by Volatility
    df['adjusted_volume_momentum'] = df['volume_momentum'] / df['rolling_volatility']
    
    # Add Adjusted Volume Momentum to Combined Factors
    df['final_factor'] = df['combined_factors'] + integration_weight * df['adjusted_volume_momentum']
    
    return df['final_factor'].dropna()
```
