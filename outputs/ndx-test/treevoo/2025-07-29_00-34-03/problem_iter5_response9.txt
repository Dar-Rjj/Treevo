```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Daily Volume Averages for the last 20 days
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean()
    
    # Scale by Volume
    df['scaled_volume'] = df['volume'] / df['avg_vol_20']
    
    # Previous Day's Return
    df['prev_day_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Open-to-Close Momentum
    df['open_to_close_return'] = (df['close'] / df['open']) - 1
    df['otc_momentum'] = df['open_to_close_return'].rolling(window=5).sum()
    
    # Realized Volatility
    df['hl_range_sq'] = (df['high'] - df['low']) ** 2
    df['realized_volatility'] = df['hl_range_sq'].rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['prev_day_return'] / df['realized_volatility']
    
    # Combine Factors
    df['combined_factor'] = (df['intraday_range'] * df['scaled_volume'] +
                              df['otc_momentum'] + 
                              df['adjusted_momentum'])
    
    # Weight by Volume
    df['final_factor'] = df['combined_factor'] * df['volume']
    
    return df['final_factor'].dropna()
```
