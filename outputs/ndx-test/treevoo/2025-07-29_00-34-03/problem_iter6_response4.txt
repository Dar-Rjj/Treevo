```python
def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume and Amount
    volume_weighted_return = close_to_open_return * df['volume']
    amount_weighted_return = volume_weighted_return * df['amount']
    
    return amount_weighted_return
```
