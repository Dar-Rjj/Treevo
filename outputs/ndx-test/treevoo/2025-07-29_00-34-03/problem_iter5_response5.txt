```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale by Volume, divided by the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Open-to-Close Momentum using previous 5 days' open-to-close returns
    open_to_close_returns = (df['close'] - df['open']) / df['open']
    otc_momentum = open_to_close_returns.rolling(window=5).sum()
    
    # Realized Volatility using last 10 days' high-low ranges
    realized_volatility = (high_low_range**2).rolling(window=10).sum()
    
    # Combine components
    combined_factor = (high_low_range * scaled_volume) + otc_momentum + realized_volatility
    
    # Apply price filter based on average of open and close of day t vs t-1
    avg_price_t = (df['open'] + df['close']) / 2
    avg_price_t_minus_1 = avg_price_t.shift(1)
    price_trend = (avg_price_t > avg_price_t_minus_1).astype(int) * 2 - 1
    final_factor = combined_factor * price_trend
    
    # Apply volume spike filter
    volume_spike = (df['volume'] > 2 * df['volume'].shift(1)).astype(int)
    final_factor += volume_spike

    return final_factor
```
