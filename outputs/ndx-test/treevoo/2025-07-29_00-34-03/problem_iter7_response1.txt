```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale by Volume, divide volume by average of previous 20 days volume
    avg_vol = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_vol
    
    adjusted_high_low_range = high_low_range * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Inverse of Intraday Range
    inversed_intraday_range = 1 / intraday_range
    
    # Combine Factors
    combined_factor = (adjusted_high_low_range * inversed_intraday_range) + close_to_open_return
    
    return combined_factor
```
