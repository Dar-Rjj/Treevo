```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume over the last 20 days
    average_volume = df['volume'].rolling(window=20).mean()
    
    # Scale Volume by dividing by the 20-day average volume
    scaled_volume = df['volume'] / average_volume
    
    # Multiply High-Low Range by Scaled Volume
    high_low_scaled = high_low_range * scaled_volume
    
    # Calculate Close-to-Previous Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Use Previous 5 Days Close-to-Previous Close Returns for Momentum
    momentum = close_to_close_return.rolling(window=5).sum()
    
    # Combine High-Low Range and Close-to-Previous Close Momentum
    factor_values = high_low_scaled + momentum
    
    return factor_values
```
