```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Weighted Close Price
    df['weighted_close'] = df['close'] * df['volume']
    
    # Calculate Adjusted SMA of Weighted Close Prices
    sma_weighted_close = df['weighted_close'].rolling(window=20).mean()
    
    # Smooth Realized Volatility
    log_returns = np.log(df['high'] / df['low'])
    realized_volatility = log_returns.rolling(window=20).std()
    smoothed_volatility = realized_volatility.ewm(span=20, adjust=False).mean()
    
    # Subtract Smoothed Volatility from SMA
    adjusted_sma = sma_weighted_close - smoothed_volatility
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_trend = avg_volume_5_days - avg_volume_20_days
    
    # Combine Momentum and Volume-Weighted Return
    combined_factor = adjusted_sma + momentum_difference + volume_trend
    
    return combined_factor
```
