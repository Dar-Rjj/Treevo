```python
def heuristics_v2(data):
    # Calculate High-Low Range
    high_low_range = data['high'] - data['low']
    
    # Scale by Volume
    scaled_by_volume = high_low_range * data['volume']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (data['close'] - data['open']) / data['open']
    
    # Calculate Previous 5 Days Close-to-Open Returns
    previous_5_days_returns = close_to_open_return.rolling(window=5).mean()
    
    # Combine High-Low Range and Close-to-Open Momentum
    combined_factor = (scaled_by_volume) + previous_5_days_returns
    
    return combined_factor
```
