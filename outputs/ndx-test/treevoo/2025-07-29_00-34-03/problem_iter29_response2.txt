```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = df['close'] * df['volume']
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Prices
    df['VW_EMA'] = df['Volume_Weighted_Price'].ewm(span=20, adjust=False).mean()
    
    # Intraday Volatility Adjustment Factor
    df['Intraday_Volatility'] = df['high'] - df['low']
    df['Vol_Adjusted_EMA'] = df['VW_EMA'] + df['Intraday_Volatility']
    
    # Volume Spike Indicator
    df['Volume_Spike'] = df['volume'].diff().apply(lambda x: 1 if x > 5 * df['volume'].std() else 0)
    
    # Short-Term Momentum
    df['Short_Term_Momentum'] = df['close'].rolling(window=5).sum()
    
    # Long-Term Momentum
    df['Long_Term_Momentum'] = df['close'].rolling(window=20).sum()
    
    # Difference between Short and Long-Term Momentum
    df['Momentum_Diff'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Incorporate Volume Trend
    df['Short_Term_Volume'] = df['volume'].rolling(window=5).sum()
    df['Long_Term_Volume'] = df['volume'].rolling(window=20).sum()
    df['Volume_Diff'] = df['Short_Term_Volume'] - df['Long_Term_Volume']
    
    # Adjusted High-Low Range
    df['Scaled_Volume'] = df['volume'] / df['volume'].rolling(window=20).sum()
    df['Adjusted_High_Low_Range'] = (df['high'] - df['low']) * df['Scaled_Volume']
    
    # Previous Day's Return
    df['Prev_Day_Return'] = df['close'].pct_change()
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['Open_to_Close_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Weight by Volume
    df['Volume_Weighted_Open_Close_Return'] = df['volume'] * df['Open_to_Close_Return']
    
    # Scale by Volume
    df['Daily_Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['Scaled_Volume_Factor'] = df['volume'] / df['Daily_Average_Volume']
    
    # Calculate Short-Term Price Momentum
    df['Short_Term_Price_Momentum'] = df['close'].rolling(window=5).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    df['Price_Momentum_Diff'] = df['Short_Term_Price_Momentum'] - df['close'].rolling(window=20).mean()
    
    # Calculate Adjusted Intraday Range
    df['Adjusted_Intraday_Range'] = df['Intraday_Range'] * df['Scaled_Volume']
    
    # Calculate Realized Volatility
    df['Realized_Volatility'] = df['Intraday_Range'].rolling(window=10).apply(lambda x: np.sqrt(np.sum(x**2) / len(x)))
    
    # Adjust Short-Term Momentum by Volatility
    df['Adjusted_Short_Term_Momentum'] = df['Short_Term_Price_Momentum'] / df['Realized_Volatility']
    
    # Combine Factors
    df['Combined_Factors'] = (
        df['Adjusted_High_Low_Range'] * (1 / df['Intraday_Range']) +
        df['Prev_Day_Return'] +
        df['Adjusted_Short_Term_Momentum'] +
        df['Volume_Weighted_Open_Close_Return'] +
        df['Momentum_Diff']
    )
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Momentum_Diff'] / df['Volume_Diff'] +
        df['Adjusted_High_Low_Range'] +
        df['Combined_Factors'] * df['Scaled_Volume_Factor']
    )
    
    return df['Alpha_Factor']
```
