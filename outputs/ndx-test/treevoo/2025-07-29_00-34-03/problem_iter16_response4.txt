```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume_avg = df['volume'].rolling(window=5).mean()
    long_term_volume_avg = df['volume'].rolling(window=20).mean()
    volume_trend = short_term_volume_avg - long_term_volume_avg
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    avg_volume_20 = long_term_volume_avg
    scaled_high_low_range = high_low_range * (df['volume'] / avg_volume_20)
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    inversed_intraday_range = 1 / intraday_range
    
    # Calculate Previous Day's Return
    prev_day_return = df['close'] / df['open'].shift(1) - 1
    
    # Adjust Momentum by Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std() * 252**0.5
    adjusted_momentum = momentum_diff / realized_volatility
    
    # Combine Factors
    combined_factors = (scaled_high_low_range * inversed_intraday_range) + prev_day_return + adjusted_momentum
    
    # Final Alpha Factor
    final_alpha = (momentum_diff / volume_trend) + scaled_high_low_range + combined_factors
    final_alpha = final_alpha * df['volume']
    
    return final_alpha
```
