import pandas as pd

def heuristics_v2(df):
    # Calculate the adjusted high-low range
    df['high_low_range'] = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / avg_volume
    df['adjusted_high_low_range'] = df['high_low_range'] * df['scaled_volume']

    # Calculate close-to-open return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)

    # Calculate intraday range and its inverse
    df['intraday_range'] = df['high'] - df['low']
    df['inverse_intraday_range'] = 1 / df['intraday_range']

    # Combine factors
    df['factor'] = (df['adjusted_high_low_range'] * df['inverse_intraday_range']) + df['close_to_open_return']

    # Return the factor values
    return df['factor']
