```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Scale by Volume
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_scaled = df['volume'] / avg_volume_20_days
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.rolling(window=5).sum()
    
    # Calculate Realized Volatility
    high_low_range = df['high'] - df['low']
    realized_volatility = (high_low_range.rolling(window=10).apply(lambda x: (x ** 2).sum(), raw=True)).pow(0.5)
    
    # Adjust Momentum by Volatility
    adjusted_momentum = open_to_close_momentum / realized_volatility
    
    # Calculate Volume-Scaled Intraday Range
    volume_scaled_intraday_range = intraday_range * volume_scaled
    
    # Combine Factors
    combined_factor = (volume_scaled_intraday_range + 
                       volume_weighted_close_to_open + 
                       adjusted_momentum)
    
    # Final Weighting
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
