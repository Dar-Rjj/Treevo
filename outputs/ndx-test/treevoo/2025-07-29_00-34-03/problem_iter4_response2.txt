```python
def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Short-Term Momentum
    # Use Close prices from t-1, t-2, t-3, t-4
    df['close_t_minus_1'] = df['close'].shift(1)
    df['close_t_minus_2'] = df['close'].shift(2)
    df['close_t_minus_3'] = df['close'].shift(3)
    df['close_t_minus_4'] = df['close'].shift(4)
    
    # Calculate Average Close Price of last 4 days
    df['average_close'] = (df['close_t_minus_1'] + df['close_t_minus_2'] + 
                           df['close_t_minus_3'] + df['close_t_minus_4']) / 4
    
    # Calculate Current Momentum
    df['current_momentum'] = df['close'] - df['average_close']
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['current_momentum'] / df['intraday_volatility']
    
    return df['adjusted_momentum']
```
