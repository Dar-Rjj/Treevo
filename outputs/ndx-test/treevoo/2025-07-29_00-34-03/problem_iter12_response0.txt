```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    rolling_window_avg = 21
    df['Rolling_Avg_Return'] = df['Daily_Return'].rolling(window=rolling_window_avg).mean()
    
    # Calculate Rolling Volatility of Close Prices
    rolling_window_vol = 20
    df['Rolling_Volatility'] = df['close'].rolling(window=rolling_window_vol).std()
    
    # Adjust Momentum by Volatility
    df['Adjusted_Momentum'] = df['Rolling_Avg_Return'] / df['Rolling_Volatility']
    
    # Weight by Volume
    df['Momentum_by_Volume'] = df['Adjusted_Momentum'] * df['volume']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].diff()
    
    # Calculate Volume Momentum
    n = 5
    df['Volume_Momentum'] = df['volume'] / df['volume'].shift(n)
    
    # Adjust Volume Momentum by Volatility
    df['Adjusted_Volume_Momentum'] = df['Volume_Momentum'] / df['Rolling_Volatility']
    
    # Combine Factors
    combined_factors = df['Momentum_by_Volume'] + df['Volume_Change']
    
    # Integrate Volume Momentum
    integration_weight = 0.5
    df['Final_Factor'] = combined_factors + (df['Adjusted_Volume_Momentum'] * integration_weight)
    
    return df['Final_Factor']
```
