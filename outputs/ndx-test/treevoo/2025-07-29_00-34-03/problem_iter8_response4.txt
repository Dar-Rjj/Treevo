```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    df['rolling_avg_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (e.g., 20 days)
    df['rolling_volatility'] = df['close'].pct_change().rolling(window=20).std() * (252**0.5)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Scale by Volume
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20_days']
    
    # Calculate Realized Volatility
    df['squared_high_low_range'] = df['high_low_range'].pow(2)
    df['realized_volatility'] = df['squared_high_low_range'].rolling(window=10).sum()
    
    # Combine High-Low Range, Scaled Volume, and Realized Volatility
    df['combined_factors'] = df['high_low_range'] * df['scaled_volume'] + df['realized_volatility']
    
    # Adjust Momentum by Combined Factors
    df['adjusted_momentum'] = df['rolling_avg_return'] * df['combined_factors']
    
    # Weight by Volume
    df['weighted_adjusted_momentum'] = df['adjusted_momentum'] * df['volume']
    
    # Calculate Open-to-Close Momentum
    df['open_close_return'] = df['close'] / df['open'] - 1
    df['open_close_momentum'] = df['open_close_return'].rolling(window=5).mean()
    
    # Combine Adjusted Momentum, Open-to-Close Momentum, and Volume
    df['preliminary_alpha_factor'] = df['weighted_adjusted_momentum'] + df['open_close_momentum']
    
    # Apply Price Filter
    df['avg_high_low_t'] = (df['high'] + df['low']) / 2
    df['avg_high_low_t-1'] = df['avg_high_low_t'].shift(1)
    df['avg_open_close_t'] = (df['open'] + df['close']) / 2
    df['avg_open_close_t-1'] = df['avg_open_close_t'].shift(1)
    
    def price_filter(row):
        if row['avg_high_low_t'] > row['avg_high_low_t-1'] or row['avg_open_close_t'] > row['avg_open_close_t-1']:
            return row['preliminary_alpha_factor'] * 1.1
        else:
            return row['preliminary_alpha_factor'] * 0.9
    
    df['final_alpha_factor'] = df.apply(price_filter, axis=1)
    
    # Apply Volume Spike Filter
    df['volume_spike'] = (df['volume'] > 2 * df['volume'].shift(1)).astype(int)
    df['final_alpha_factor'] = df['final_alpha_factor'] * (1 + df['volume_spike'] * 0.1)
    
    return df['final_alpha_factor']
```
