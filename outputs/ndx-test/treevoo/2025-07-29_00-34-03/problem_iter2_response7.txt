```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    # Calculate Average Volume over the last 5 days
    short_term_volume_avg = df['volume'].rolling(window=5).mean()
    
    # Calculate Average Volume over the last 20 days
    long_term_volume_avg = df['volume'].rolling(window=20).mean()
    
    # Compute the difference between short and long-term volume averages
    volume_diff = short_term_volume_avg - long_term_volume_avg
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale High-Low Range by Volume
    # Calculate Daily Volume Averages
    daily_volume_avg = df['volume'].rolling(window=20).mean()
    
    # Divide Volume by Average Volume
    scaled_volume = df['volume'] / daily_volume_avg
    
    # Multiply High-Low Range by Scaled Volume
    scaled_high_low_range = high_low_range * scaled_volume
    
    # Calculate Close-to-Open Momentum
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Use Previous 5 Days Close-to-Open Returns
    close_to_open_momentum = close_to_open_return.rolling(window=5).mean()
    
    # Combine Price Momentum and High-Low Range
    combined_momentum = momentum_diff / volume_diff + scaled_high_low_range
    
    # Combine Final Alpha Factor
    final_alpha_factor = combined_momentum + close_to_open_momentum
    
    return final_alpha_factor
```
