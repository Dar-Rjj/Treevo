```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Weight by Volume and Open-to-Close Return
    weighted_open_to_close_return = df['volume'] * open_to_close_return
    
    # Scale by Volume
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20_days
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Adjusted Intraday Range
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Realized Volatility
    high_low_ranges = df['high'] - df['low']
    realized_volatility = (high_low_ranges.rolling(window=10).sum()) ** 2
    
    # Adjust Short-Term Momentum by Volatility
    adjusted_short_term_momentum = short_term_momentum / realized_volatility
    
    # Combine Factors
    combined_factor = (adjusted_intraday_range + 
                       weighted_open_to_close_return + 
                       adjusted_short_term_momentum + 
                       momentum_difference)
    
    # Final Weighting
    final_factor = df['volume'] * combined_factor
    
    return final_factor
```
