```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(-1) / df['open'] - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Adjust Weighted Return Using High-Low Spread
    high_low_spread = df['high'] - df['low']
    adjusted_weighted_return = (volume_weighted_return / high_low_spread).fillna(0)
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Adjusted High-Low Range
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20_days
    adjusted_high_low_range = (df['high'] - df['low']) * scaled_volume
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Intraday Volatility
    intraday_volatility = intraday_range
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'] / df['open'].shift(1) - 1
    
    # Short-Term Momentum
    avg_close_4_days = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - avg_close_4_days
    adjusted_short_term_momentum = current_momentum / intraday_volatility
    
    # Combine Factors
    inverse_intraday_range = 1 / intraday_range
    combined_factors = adjusted_high_low_range * inverse_intraday_range + previous_day_return + adjusted_short_term_momentum
    
    # Final Alpha Factor
    price_momentum_and_high_low = (momentum_difference / (avg_volume_20_days.diff().fillna(0))) + adjusted_high_low_range
    final_alpha_factor = close_to_open_return + combined_factors + price_momentum_and_high_low
    
    # Weight by Volume
    final_alpha_factor *= df['volume']
    
    return final_alpha_factor
```
