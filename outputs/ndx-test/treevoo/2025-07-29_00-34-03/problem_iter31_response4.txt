import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['prev_close'] = df['close'].shift(1)
    df['open_to_close_return'] = (df['close'] - df['prev_close']) / df['prev_close']
    
    # Weight by Volume
    df['weighted_return'] = df['volume'] * df['open_to_close_return']
    
    # Scale by Volume
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20d']
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    df['momentum_diff'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Adjusted Intraday Range
    df['adjusted_intraday_range'] = df['intraday_range'] * df['scaled_volume']
    
    # Calculate Realized Volatility
    df['hl_range_sq'] = (df['high'] - df['low']) ** 2
    df['realized_volatility'] = df['hl_range_sq'].rolling(window=10).sum()
    
    # Adjust Short-Term Momentum by Volatility
    df['adjusted_short_term_momentum'] = df['short_term_momentum'] / df['realized_volatility']
    
    # Combine Factors
    df['combined_factor'] = (
        df['adjusted_intraday_range'] +
        df['weighted_return'] +
        df['adjusted_short_term_momentum'] +
        df['momentum_diff']
    )
    
    # Final Weighting
    df['final_factor'] = df['combined_factor'] * df['volume']
    
    return df['final_factor'].dropna()
