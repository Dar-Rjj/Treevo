```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'].pct_change()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (21 days)
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (20 days)
    df['rolling_volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Inverse of Intraday Range
    df['inv_intraday_range'] = 1 / df['intraday_range']
    
    # Combine Factors
    df['combined_factor'] = df['prev_day_return'] * df['inv_intraday_range']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Final Factor Calculation
    df['final_factor'] = df['combined_factor'] * df['volume_change'] * df['close_to_open_return'] * df['amount']
    
    # Volume-Weighted Price
    df['volume_weighted_price'] = df['volume'] * df['close']
    
    # Scale by Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20']
    
    # Volume Momentum
    df['volume_momentum'] = df['volume'] / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['adjusted_volume_momentum'] = df['volume_momentum'] / df['rolling_volatility']
    
    # Integrate Volume Momentum
    integration_weight = 0.5
    df['integrated_factor'] = df['final_factor'] + integration_weight * df['adjusted_volume_momentum']
    
    # Incorporate High and Low Prices
    df['high_low_spread'] = df['high'] - df['low']
    df['weighted_return'] = df['integrated_factor'] * df['high_low_spread']
    
    # Integrate Momentum
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    df['momentum_adjustment'] = df['close'] - df['sma_close_10']
    df['finalized_factor'] = df['weighted_return'] + df['momentum_adjustment']
    
    # Incorporate Volatility
    df['std_close_20'] = df['close'].rolling(window=20).std()
    df['volatility_adjusted_factor'] = df['finalized_factor'] / df['std_close_20']
    
    # Final Volume and Volatility Adjustment
    df['final_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['final_combined_factor'] = df['volatility_adjusted_factor'] * df['final_volume_change']
    
    # Detect Significant Volume Surge
    surge_threshold = 1.5
    significant_surge = df['final_volume_change'] > surge_threshold
    df['final_alpha_factor'] = df['final_combined_factor']
    df.loc[significant_surge, 'final_alpha_factor'] += df['final_combined_factor'] * 0.5
    
    return df['final_alpha_factor']
```
