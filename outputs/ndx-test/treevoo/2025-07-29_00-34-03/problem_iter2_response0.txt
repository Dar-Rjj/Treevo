```python
import pandas as pd
import numpy as np

def heuristics_v2(df, sma_period=20, ema_span=20):
    # Calculate Volume-Weighted Close Price
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Close Prices
    df['sma_volume_weighted_close'] = df['volume_weighted_close'].rolling(window=sma_period).mean()
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Inverse of Intraday Range
    df['inverse_intraday_range'] = 1 / df['intraday_range']
    
    # Momentum Adjustment
    df['momentum_adjustment'] = df['prev_day_return'] * df['inverse_intraday_range']
    
    # Calculate Realized Volatility
    df['log_high_low'] = np.log(df['high']) - np.log(df['low'])
    df['realized_volatility'] = df['log_high_low'].rolling(window=sma_period).std()
    
    # Smooth Realized Volatility with EMA
    df['smoothed_volatility'] = df['realized_volatility'].ewm(span=ema_span).mean()
    
    # Adjust Combined Momentum with Smoothed Volatility
    df['volatility_adjusted_momentum'] = df['momentum_adjustment'] - df['smoothed_volatility']
    
    return df['volatility_adjusted_momentum']
```
