```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Intraday Range
    intraday_range = (df['high'] - df['low']) * df['volume']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'].shift(1) / df['open'] - 1
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Calculate Realized Volatility
    realized_volatility = (df['high'] - df['low']).rolling(window=10).apply(lambda x: (x**2).sum())
    
    # Adjust Short-Term Momentum by Volatility
    adjusted_short_term_momentum = short_term_momentum / realized_volatility
    
    # Combine Factors
    combined_factor = (
        intraday_range + 
        open_to_close_return + 
        adjusted_short_term_momentum + 
        momentum_diff
    )
    
    # Final Weighting
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
