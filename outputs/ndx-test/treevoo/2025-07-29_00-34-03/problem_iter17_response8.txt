```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume
    adjusted_high_low_range = high_low_range * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Intraday Volatility
    intraday_volatility = intraday_range
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'] / df['open'].shift(1) - 1
    
    # Short-Term Momentum
    avg_close_4_days = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - avg_close_4_days
    adjusted_momentum = current_momentum / intraday_volatility
    
    # Combine Factors
    inversed_intraday_range = 1 / intraday_range
    combined_factors = (adjusted_high_low_range * inversed_intraday_range + previous_day_return)
    
    # Final Alpha Factor
    alpha_factor = (momentum_diff + adjusted_high_low_range + close_to_open_return + combined_factors) * df['volume']
    
    return alpha_factor
```
