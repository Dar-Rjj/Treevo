import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume-Weighted Close-to-Open Return
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust Weighted Return with High-Low Spread
    adjusted_weighted_return = volume_weighted_return - high_low_spread
    
    # Integrate Short-Term Momentum
    average_close_price = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - average_close_price
    momentum_adjusted_return = adjusted_weighted_return + current_momentum
    
    # Incorporate Volume Trend
    short_term_average_volume = df['volume'].rolling(window=5).mean()
    long_term_average_volume = df['volume'].rolling(window=20).mean()
    volume_trend = short_term_average_volume - long_term_average_volume
    
    # Combine Momentum and Volume Trends
    combined_momentum_and_volume = momentum_adjusted_return + volume_trend
    
    # Adjusted True Average Price
    true_average_price = (df['high'] + df['low'] + df['close']) / 3
    price_range_volatility = (df['high'] - df['low']).rolling(window=20).std()
    inverse_price_range_volatility = 1 / price_range_volatility
    adjusted_true_average_price = true_average_price * inverse_price_range_volatility
    
    # Final Momentum Indicator
    final_momentum_indicator = adjusted_true_average_price * volume_trend
    
    # Integrate with Combined Momentum and Volume Trends
    integrated_momentum = combined_momentum_and_volume * final_momentum_indicator
    
    # Calculate Rolling Average Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    rolling_average_daily_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices
    rolling_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_average_daily_return / rolling_volatility
    
    # Final Alpha Factor
    final_alpha_factor = integrated_momentum * adjusted_momentum
    
    return final_alpha_factor
