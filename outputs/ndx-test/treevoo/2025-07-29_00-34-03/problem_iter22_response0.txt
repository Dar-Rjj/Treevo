```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Scale by Volume
    average_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / average_volume_20_days
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Adjusted High-Low Range
    adjusted_high_low_range = (df['high'] - df['low']) * scaled_volume
    
    # Calculate Realized Volatility
    high_low_ranges = (df['high'] - df['low']).rolling(window=10).sum()
    realized_volatility = (high_low_ranges ** 2).sum()
    
    # Adjust Short-Term Momentum by Volatility
    adjusted_short_term_momentum = short_term_momentum / realized_volatility
    
    # Combine Factors
    combined_factor = (
        adjusted_high_low_range +
        volume_weighted_return +
        adjusted_short_term_momentum +
        momentum_difference
    )
    
    # Final Weighting
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
