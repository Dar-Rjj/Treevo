```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Daily Return
    daily_return = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (e.g., 20 days)
    rolling_volatility_daily_return = daily_return.rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_avg_daily_return / rolling_volatility_daily_return
    
    # Combine Factors
    inversed_intraday_range = 1 / intraday_range
    combined_factor = previous_day_return * inversed_intraday_range * adjusted_momentum
    
    # Weight by Volume
    volume_weighted_factor = combined_factor * df['volume']
    
    return volume_weighted_factor
```
