import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_c2o_return = close_to_open_return * df['volume']
    
    # Scale by Volume
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20_days
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Adjusted High-Low Range
    adjusted_high_low_range = intraday_range * scaled_volume
    
    # Calculate Realized Volatility
    high_low_ranges_last_10_days = (df['high'] - df['low']).rolling(window=10).apply(lambda x: (x**2).sum())
    realized_volatility = (high_low_ranges_last_10_days / 10)**0.5
    
    # Adjust Short-Term Momentum by Volatility
    adjusted_short_term_momentum = short_term_momentum / realized_volatility
    
    # Calculate Previous Day's Return
    previous_day_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Combine Factors
    combined_factor = (
        adjusted_high_low_range + 
        volume_weighted_c2o_return + 
        adjusted_short_term_momentum + 
        momentum_difference
    )
    
    # Final Weighting
    final_factor = combined_factor * df['volume']
    
    return final_factor
