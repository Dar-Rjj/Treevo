```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'] / df['close'].shift(1) - 1
    
    # Scale by Volume
    df['Volume_Scaled_Return'] = df['Close_to_Open_Return'] * df['volume']
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Scaled_Factor'] = df['Volume_Scaled_Return'] * (df['volume'] / df['Avg_Volume_20'])
    
    # Adjust by Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) / df['close']
    df['Adjusted_Factor'] = df['Volume_Scaled_Factor'] * df['Intraday_Range']
    
    # Compute Weighted Close Prices
    df['Weighted_Close'] = df['volume'] * df['close']
    
    # Calculate Adjusted SMA of Weighted Close Prices
    df['SMA_Weighted_Close'] = df['Weighted_Close'].rolling(window=20).mean()
    
    # Smooth Realized Volatility
    df['Log_Returns'] = np.log(df['high'] / df['low'])
    df['Realized_Volatility'] = df['Log_Returns'].rolling(window=10).std()
    df['Smoothed_Volatility'] = df['Realized_Volatility'].ewm(span=20, adjust=False).mean()
    df['Adjusted_SMA'] = df['SMA_Weighted_Close'] - df['Smoothed_Volatility']
    
    # Calculate Short-Term Price Momentum
    df['Short_Term_Momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Medium-Term Price Momentum
    df['Medium_Term_Momentum'] = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Price Momentum
    df['Long_Term_Momentum'] = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short, Medium, and Long-Term Momentum
    df['Momentum_Diff_Short_Medium'] = df['Short_Term_Momentum'] - df['Medium_Term_Momentum']
    df['Momentum_Diff_Medium_Long'] = df['Medium_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Incorporate Volume Trend
    df['Avg_Volume_5'] = df['volume'].rolling(window=5).mean()
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Trend'] = df['Avg_Volume_5'] - df['Avg_Volume_20']
    
    # Combine Adjusted SMA, Momentum, and Volatility-Adjusted Return
    df['Combined_Factor'] = (df['Adjusted_SMA'] 
                             + df['Momentum_Diff_Short_Medium'] 
                             + df['Momentum_Diff_Medium_Long'] 
                             + df['Volume_Scaled_Factor'] 
                             + df['Volume_Trend'])
    
    # Final Alpha Factor
    alpha_factor = df['Combined_Factor']
    
    return alpha_factor
```
