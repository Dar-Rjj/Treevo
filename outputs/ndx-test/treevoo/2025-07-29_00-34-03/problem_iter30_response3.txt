import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Mid-Price
    df['volume_weighted_mid_price'] = (df['high'] + df['low']) / 2 * df['volume']
    
    # Calculate Future Log Return
    df['future_log_return'] = np.log(df['close']).shift(-1) - np.log(df['close'])
    
    # Compute the Moving Average of Volume-Weighted Mid-Price
    window_size = 10
    df['moving_avg_volume_weighted_mid_price'] = df['volume_weighted_mid_price'].rolling(window=window_size).mean()
    
    # Correlate the Moving Average with Future Log Return
    correlation = df['moving_avg_volume_weighted_mid_price'].corr(df['future_log_return'], method='pearson')
    
    # The factor is the moving average of volume-weighted mid-price
    alpha_factor = df['moving_avg_volume_weighted_mid_price']
    
    return alpha_factor
