import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = (df['close'] / df['open'] - 1) * df['volume']
    
    # Adjust Weighted Return Using High-Low Spread
    high_low_spread = df['high'] - df['low']
    adjusted_weighted_return = volume_weighted_return * high_low_spread
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'] / df['open'].shift(1) - 1
    
    # Combine Factors
    average_volume = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / average_volume
    adjusted_high_low_range = high_low_spread * scaled_volume
    inverse_intraday_range = 1 / intraday_range
    combined_factors = (adjusted_high_low_range * inverse_intraday_range) + previous_day_return + short_term_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_difference / (df['volume'].rolling(window=5).mean() - df['volume'].rolling(window=20).mean())) + adjusted_high_low_range + close_to_open_return + combined_factors
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
