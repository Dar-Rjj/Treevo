```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'].pct_change()
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Rolling Average Daily Return
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns
    df['rolling_volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['momentum_adjustment'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Combine Factors
    df['inv_intraday_range'] = 1 / df['intraday_range']
    df['momentum_adj_by_inv_range'] = df['prev_day_return'] * df['inv_intraday_range']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_weighted_close_to_open'] = df['close_to_open_return'] * df['volume']
    
    # Incorporate Amount
    df['amount_weighted_close_to_open'] = df['volume_weighted_close_to_open'] * df['amount']
    
    # Incorporate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['adjusted_weighted_return'] = df['amount_weighted_close_to_open'] * (1 + df['high_low_spread'])
    
    # Integrate Momentum
    df['sma_10_close'] = df['close'].rolling(window=10).mean()
    df['mom_adjusted_return'] = (df['close'] - df['sma_10_close']) + df['momentum_adj_by_inv_range']
    
    # Incorporate Volatility
    df['std_20_close'] = df['close'].rolling(window=20).std()
    df['intermediate_alpha_factor'] = df['mom_adjusted_return'] / df['std_20_close']
    
    # Weight by Volume, Amount, and Close-to-Open Return
    df['volume_change'] = df['volume'].pct_change()
    df['intermediate_factor_calculation'] = df['intermediate_alpha_factor'] * df['volume_change'] * df['amount_weighted_close_to_open']
    
    # Calculate Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['vol_adj_volume_momentum'] = df['volume_momentum'] / df['rolling_volatility']
    
    # Integrate Volume Momentum
    df['integrated_factor'] = df['intermediate_factor_calculation'] + 0.5 * df['vol_adj_volume_momentum']
    
    # Calculate Volume-Weighted Intraday Range
    df['volume_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    
    # Incorporate Volume-Weighted Intraday Range
    df['final_factor'] = df['integrated_factor'] + 0.3 * df['volume_weighted_intraday_range']
    
    return df['final_factor']
```
