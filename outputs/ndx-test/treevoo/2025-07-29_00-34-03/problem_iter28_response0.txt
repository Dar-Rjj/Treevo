```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Open Return (Close - Open)
    close_to_open_return = df['close'] - df['open']
    
    # Multiply by Volume
    volume_weighted_return = close_to_open_return * df['volume']

    # Calculate 10-Day Exponential Moving Average (EMA) of Close Price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Subtract 10-Day EMA from Current Close Price
    momentum_adjustment = df['close'] - ema_10
    
    # Add Momentum Adjustment to Adjusted Return
    adjusted_return = volume_weighted_return + momentum_adjustment

    # Calculate 20-Day Standard Deviation of Log Returns
    log_returns = np.log(df['close']).diff()
    std_20 = log_returns.rolling(window=20).std()

    # Divide Adjusted Return by 20-Day Standard Deviation
    final_alpha_factor = adjusted_return / std_20

    return final_alpha_factor
```
