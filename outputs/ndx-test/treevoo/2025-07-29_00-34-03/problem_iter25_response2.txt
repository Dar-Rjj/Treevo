```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Open Return (Close - Open)
    close_to_open_return = df['close'] - df['open']
    
    # Multiply by Volume to get Volume-Weighted Close-to-Open Return
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate High-Low Spread (High - Low)
    high_low_spread = df['high'] - df['low']
    
    # Adjust Weighted Return with High-Low Spread
    adjusted_weighted_return = volume_weighted_return / high_low_spread
    
    # Calculate 10-Day Exponential Moving Average (EMA) of Close Price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Subtract 10-Day EMA from Current Close Price to get Momentum Adjustment
    momentum_adjustment = df['close'] - ema_10
    
    # Add Momentum Adjustment to Adjusted Return
    momentum_adjusted_return = adjusted_weighted_return + momentum_adjustment
    
    # Calculate 20-Day Standard Deviation of Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_dev_20 = log_returns.rolling(window=20).std()
    
    # Divide Adjusted Return by 20-Day Standard Deviation to get Final Alpha Factor
    final_alpha_factor = momentum_adjusted_return / std_dev_20
    
    return final_alpha_factor
```
