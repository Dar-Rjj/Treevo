```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_open'] = df['open'].shift(1)
    prev_day_return = (df['close'] / df['prev_open']) - 1
    
    # Calculate Daily Return
    df['prev_close'] = df['close'].shift(1)
    daily_return = (df['close'] / df['prev_close']) - 1
    
    # Calculate Rolling Average Daily Return (21 days)
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (20 days)
    rolling_volatility = daily_return.rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_avg_daily_return / rolling_volatility
    
    # Inverse of Intraday Range
    inversed_intraday_range = 1 / intraday_range
    
    # Adjusted Momentum combined with Inversed Intraday Range
    combined_factor = prev_day_return * inversed_intraday_range * adjusted_momentum
    
    # Weight by Volume
    weighted_factor = combined_factor * df['volume']
    
    return weighted_factor
```
