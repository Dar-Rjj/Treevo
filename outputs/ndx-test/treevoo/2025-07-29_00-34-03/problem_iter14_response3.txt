import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Adjust Weighted Return
    high_low_spread = df['high'] - df['low']
    adjusted_return = volume_weighted_return / high_low_spread
    
    # Incorporate Daily Log Return
    daily_log_return = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Momentum and Volume
    preliminary_factor = daily_log_return * volume_change
    
    # Apply Price Filter
    avg_price_t = (df['open'] + df['close']) / 2
    avg_price_t_minus_1 = (df['open'].shift(1) + df['close'].shift(1)) / 2
    
    price_filter = np.where(avg_price_t > avg_price_t_minus_1, 1, -1)
    final_factor = preliminary_factor * price_filter
    
    # Incorporate Intraday Dynamics
    intraday_momentum = df['high'] - df['low']
    volume_weighted_intraday_momentum = intraday_momentum * df['volume']
    
    # Identify Volume Shock
    volume_shock = np.where(df['volume'] > 1.5 * df['volume'].shift(1), 1, 0)
    
    # Adjust Combined Momentum Factor
    combined_momentum_factor = final_factor + volume_weighted_intraday_momentum * volume_shock
    
    return combined_momentum_factor
