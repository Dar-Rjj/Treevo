import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    weighted_close = df['close'] * df['volume']
    sum_weighted_close = weighted_close.rolling(window=5).sum()
    sum_volume = df['volume'].rolling(window=5).sum()
    vwp = sum_weighted_close / sum_volume

    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    sma_vwp = vwp.rolling(window=20).mean()

    # Intraday Volatility Adjustment Factor
    intraday_volatility = df['high'] - df['low']
    adjusted_sma = sma_vwp - intraday_volatility

    # Volume Surge Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_surge = (volume_change > 0) & (volume_change > 0.1 * df['volume'].shift(1))

    # Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].rolling(window=4).mean()

    # True Average Price
    true_avg_price = (df['high'] + df['low'] + df['close']) / 3

    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()

    # Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum

    # Calculate Price Range Volatility
    price_range_volatility = (df['high'] - df['low']).rolling(window=20).std()

    # Adjusted True Average Price
    adjusted_true_avg_price = true_avg_price / price_range_volatility

    # Final Momentum Indicator
    final_momentum_indicator = adjusted_true_avg_price * (df['volume'] * (df['high'] - df['open']))

    # Integrate with Combined Momentum and Volume Trends
    integrated_alpha = final_momentum_indicator * momentum_diff

    # Enhanced Momentum-Volatility Interaction
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    rolling_volatility = df['close'].rolling(window=20).std()
    enhanced_momentum = ema_close / rolling_volatility

    # Volume-Adjusted Momentum
    cumulative_vwp = vwp.rolling(window=20).sum()
    vwp_momentum = cumulative_vwp - cumulative_vwp.shift(1)
    vwp_momentum[volume_surge] = vwp_momentum[volume_surge] * 1.5

    # Integrated Alpha Factor
    integrated_alpha_factor = enhanced_momentum * vwp_momentum

    # Final Alpha Factor
    final_alpha_factor = integrated_alpha_factor * adjusted_sma

    return final_alpha_factor
