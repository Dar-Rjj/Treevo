def heuristics_v2(df):

    df['intraday_volatility'] = df['high'] - df['low']

    df['current_momentum'] = df['close'] - df['avg_close_4_days']
    df['volatility_adjusted_momentum'] = df['current_momentum'] / df['intraday_volatility']
    
    # Calculate Volume-Weighted Prices
    df['weighted_close'] = df['volume'] * df['close']
    df['sma_weighted_close_10_days'] = df['weighted_close'].rolling(window=10).mean()

    df['short_term_avg_close'] = df['close'].rolling(window=10).mean()
    df['long_term_avg_close'] = df['close'].rolling(window=60).mean()
    df['price_momentum'] = df['short_term_avg_close'] - df['long_term_avg_close']

    df['short_term_avg_volume'] = df['volume'].rolling(window=10).mean()

    # Final Alpha Factor
    df['alpha_factor'] = (df['volatility_adjusted_momentum'] + df['price_momentum']) * (df['volume_surge'] > 0)
