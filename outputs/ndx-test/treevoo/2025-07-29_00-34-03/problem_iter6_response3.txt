```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale by Volume
    volume_scaled = df['volume'] / df['volume'].shift(1)
    
    # Calculate Close-to-Previous Close Momentum
    close_to_close_momentum = df['close'].pct_change()
    close_to_close_momentum_5d = close_to_close_momentum.rolling(window=5).sum()
    
    # Adjust for Open Price
    open_to_close_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    open_to_close_momentum_5d = open_to_close_return.rolling(window=5).sum()
    
    # Combine High-Low Range, Close-to-Previous Close Momentum, and Open-Price Adjustment
    factor_values = (high_low_range * volume_scaled) + close_to_close_momentum_5d - open_to_close_momentum_5d
    
    return factor_values
```
