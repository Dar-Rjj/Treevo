```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'].pct_change()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (21 days)
    df['rolling_avg_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (20 days)
    df['rolling_volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['momentum_adjusted'] = df['rolling_avg_return'] / df['rolling_volatility']
    
    # Inverse of Intraday Range
    df['inverse_intraday_range'] = 1 / df['intraday_range']
    
    # Combine Factors: Multiply Previous Day's Return by Inversed Intraday Range
    df['combined_factor'] = df['prev_day_return'] * df['inverse_intraday_range']
    
    # Calculate Volume-Weighted Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_weighted_close_to_open'] = df['close_to_open_return'] * df['volume']
    
    # Incorporate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['adjusted_volume_weighted_close_to_open'] = df['volume_weighted_close_to_open'] * df['high_low_spread']
    
    # Integrate Momentum
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['momentum_integrated'] = df['close'] - df['sma_10'] + df['momentum_adjusted']
    
    # Incorporate Volatility
    df['volatility_20'] = df['close'].rolling(window=20).std()
    df['final_momentum_factor'] = df['momentum_integrated'] / df['volatility_20']
    
    # Weight by Volume, Amount, and Close-to-Open Return
    df['volume_change'] = df['volume'].pct_change()
    df['amount'] = df['close'] * df['volume']
    df['weighted_factor'] = df['final_momentum_factor'] * df['volume_change'] * df['close_to_open_return'] * df['amount']
    
    # Calculate Volume Momentum
    df['volume_momentum'] = df['volume'] - df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['volume_momentum_adjusted'] = df['volume_momentum'] / df['rolling_volatility']
    
    # Integrate Volume Momentum
    df['integrated_volume_momentum'] = 0.5 * df['volume_momentum_adjusted']
    
    # Calculate Volume-Weighted Intraday Range
    df['volume_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['weighted_factor'] + 0.3 * df['volume_weighted_intraday_range'] + df['integrated_volume_momentum']
    
    return df['final_alpha_factor']
```
