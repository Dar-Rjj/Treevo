```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close
    volume_weighted_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Volume-Weighted Open
    volume_weighted_open = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Close-to-Open Return
    close_to_open_return = (volume_weighted_open - volume_weighted_close) / volume_weighted_close
    
    # Weight by Volume
    factor_values = close_to_open_return * df['volume']
    
    return factor_values
```
