```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'volume'].
    
    Returns:
    pd.Series: A Series with index as date and values as the computed alpha factor.
    """
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of volume to the average price
    volume_ratio = df['volume'] / avg_price
    
    # Calculate the weighted momentum factor
    momentum_factor = close_change * volume_ratio.shift(1)
    
    # Calculate the price volatility
    price_volatility = df['close'].rolling(window=5).std()
    
    # Combine the factors into an alpha factor
    alpha_factor = momentum_factor - price_volatility
    
    return alpha_factor
```
