```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Calculate Close-to-Open Momentum
    close_open_return = (df['close'] - df['open']) / df['open']
    avg_close_open_return_5 = close_open_return.rolling(window=5).mean()
    
    # Combine High-Low Range and Close-to-Open Momentum
    factor = (high_low_range * scaled_volume) + avg_close_open_return_5
    
    return factor
```
