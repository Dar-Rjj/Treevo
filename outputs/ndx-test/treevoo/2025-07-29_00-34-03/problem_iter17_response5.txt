```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the upper and lower Keltner Channel
    keltner_upper = df['close'].rolling(window=10).mean() + 2 * price_range.rolling(window=10).mean()
    keltner_lower = df['close'].rolling(window=10).mean() - 2 * price_range.rolling(window=10).mean()
    
    # Calculate the relative position of the closing price within the Keltner Channel
    keltner_relative_position = (df['close'] - keltner_lower) / (keltner_upper - keltner_lower)
    
    # Calculate the relative volume
    relative_volume = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Combine the relative position and relative volume to create the heuristic factor
    heuristic_factor = (keltner_relative_position + relative_volume) / 2
    
    return heuristic_factor
```
