import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    df['SMA_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].rolling(window=20).mean()
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['Scaled_Volume'] = df['volume'] / avg_volume_20
    
    # Calculate Previous Day's Return
    df['Prev_Day_Return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['Rolling_Avg_Daily_Return'] = df['Daily_Return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns
    df['Rolling_Volatility'] = df['Daily_Return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['Momentum_Adjustment'] = df['Rolling_Avg_Daily_Return'] / df['Rolling_Volatility']
    
    # Combine Factors
    df['Inverse_Intraday_Range'] = 1 / df['Intraday_Range']
    df['Momentum_Adjusted_Return'] = df['Prev_Day_Return'] * df['Inverse_Intraday_Range']
    
    # Calculate Volume-Weighted Return
    df['Volume_Weighted_Return'] = df['Volume_Weighted_Price'] / df['Volume_Weighted_Price'].shift(1) - 1
    
    # Combine with Momentum Adjustment
    df['Combined_Factor'] = df['Volume_Weighted_Return'] * df['Momentum_Adjusted_Return']
    
    # Final Volume and Volatility Adjustment
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Final_Factor'] = df['Combined_Factor'] * df['Volume_Change']
    
    # Enhanced Volume Surge Indicator
    df['Volume_Surge'] = df['Volume_Change'] > 0
    df['Significant_Volume_Surge'] = df['Volume_Change'] > df['Volume_Change'].quantile(0.95)
    
    # Adjust Final Alpha Factor
    df['Final_Factor'] = df['Final_Factor'] * (1 + df['Volume_Surge']) * (1 + 2 * df['Significant_Volume_Surge'])
    
    return df['Final_Factor']
