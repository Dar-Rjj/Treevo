```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].rolling(window=5).mean().shift(1)
    
    # Normalize Volume, Amount, and Price Momentum to [0, 1] range
    # Note: Normalization is not allowed as per the instruction, so we will skip this step
    
    # Combine Weights
    df['combined_weight'] = df['volume'] + df['amount'] + df['price_momentum']
    
    # Multiply Close-to-Open Return by Combined Weights
    df['factor'] = df['close_to_open_return'] * df['combined_weight']
    
    return df['factor'].dropna()
```
