```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (21 days)
    df['Rolling_Avg_Return'] = df['Daily_Return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (20 days)
    df['Rolling_Volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['Adjusted_Momentum'] = df['Rolling_Avg_Return'] / df['Rolling_Volatility']
    
    # Weight by Volume
    df['Momentum_Volume'] = df['Adjusted_Momentum'] * df['volume']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Combine Momentum and Volume
    df['Combined_Factor'] = df['Momentum_Volume'] + df['Volume_Change']
    
    # Apply Price Filter
    df['Price_Filter'] = (df['high'] + df['low']) / 2
    df['Prev_Price_Filter'] = df['Price_Filter'].shift(1)
    df['Final_Factor'] = df['Combined_Factor'].where(df['Price_Filter'] > df['Prev_Price_Filter'],
                                                      -df['Combined_Factor'])
    
    return df['Final_Factor'].dropna()
```
