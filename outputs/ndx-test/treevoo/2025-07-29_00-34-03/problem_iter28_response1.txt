```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = df['volume'] * df['close']
    
    # EMA of Volume-Weighted Prices
    df['EMA_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].ewm(span=20).mean()
    
    # Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjusted EMA
    df['Adjusted_EMA'] = df['EMA_Volume_Weighted_Price'] + df['Intraday_Volatility']
    
    # Volume Spike Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Spike'] = df['Volume_Change'].apply(lambda x: 1 if x > df['Volume_Change'].quantile(0.9) else 0)
    
    # Short-Term Momentum
    df['Avg_Close_4_days'] = df['close'].rolling(window=4).mean()
    df['Short_Term_Momentum'] = df['close'] - df['Avg_Close_4_days']
    
    # True Average Price
    df['True_Avg_Price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Long-Term Price Momentum (EMA over 20 days)
    df['Long_Term_Momentum'] = df['close'].ewm(span=20).mean()
    
    # Momentum Difference
    df['Momentum_Difference'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Daily Return
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Rolling Average Daily Return (21-day window)
    df['Rolling_Avg_Return'] = df['Daily_Return'].rolling(window=21).mean()
    
    # Rolling Volatility of Close Prices (20-day window)
    df['Rolling_Volatility'] = df['close'].rolling(window=20).std()
    
    # Adjusted Momentum by Volatility
    df['Adjusted_Momentum'] = df['Rolling_Avg_Return'] / df['Rolling_Volatility']
    
    # Volume-Price Movement Score
    df['Volume_Price_Movement'] = df['volume'] * (df['high'] - df['open'])
    
    # Volume Trend
    df['Short_Term_Vol_EMA'] = df['volume'].ewm(span=5).mean()
    df['Long_Term_Vol_EMA'] = df['volume'].ewm(span=20).mean()
    df['Volume_Trend'] = df['Short_Term_Vol_EMA'] - df['Long_Term_Vol_EMA']
    
    # Combined Momentum and Volume Trends
    df['Combined_Momentum_Volume'] = df['Momentum_Difference'] + df['Volume_Trend']
    
    # Adjusted True Average Price
    df['Moving_Std_Price_Range'] = df['high'] - df['low']
    df['Moving_Std_Price_Range'] = df['Moving_Std_Price_Range'].rolling(window=20).std()
    df['Adjusted_True_Avg_Price'] = df['True_Avg_Price'] * (1 / df['Moving_Std_Price_Range'])
    
    # Final Momentum Indicator
    df['Final_Momentum_Indicator'] = df['Adjusted_True_Avg_Price'] * df['Volume_Price_Movement']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Adjusted_Momentum'] + df['Final_Momentum_Indicator'] + df['Volume_Trend'] / df['Rolling_Volatility']) * df['Volume_Spike']
    
    return df['Final_Alpha_Factor']
```
