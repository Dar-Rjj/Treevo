```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Open Return
    close_open_return = df['close'] / df['open'] - 1
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_return = close_open_return * df['volume']
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust Weighted Return with High-Low Spread
    adjusted_weighted_return = volume_weighted_return / high_low_spread
    
    # Integrate Momentum
    # Calculate 10-Day SMA of Close Price
    sma_10 = df['close'].rolling(window=10).mean()
    momentum_adjustment = df['close'] - sma_10
    momentum_adjusted_return = adjusted_weighted_return + momentum_adjustment
    
    # Incorporate Volatility
    # Calculate 20-Day Standard Deviation of Close Price
    std_20 = df['close'].rolling(window=20).std()
    final_alpha_factor = momentum_adjusted_return / std_20
    
    return final_alpha_factor
```
