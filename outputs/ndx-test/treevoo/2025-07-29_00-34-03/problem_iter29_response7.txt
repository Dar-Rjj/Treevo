```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = df['close'] * df['volume']
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Prices
    df['EMA_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].ewm(span=10, adjust=False).mean()
    
    # Intraday Volatility Adjustment Factor
    df['Intraday_Volatility'] = df['high'] - df['low']
    df['Adjusted_EMA'] = df['EMA_Volume_Weighted_Price'] + df['Intraday_Volatility']
    
    # Volume Spike Indicator
    df['Volume_Change'] = df['volume'].diff().fillna(0)
    volume_spike_threshold = df['volume'].quantile(0.95)
    df['Volume_Spike'] = (df['Volume_Change'] > volume_spike_threshold).astype(int)
    
    # Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['close'].rolling(window=5).sum()
    df['Long_Term_Momentum'] = df['close'].rolling(window=20).sum()
    df['Momentum_Difference'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Incorporate Volume Trend
    df['Short_Term_Volume'] = df['volume'].rolling(window=5).sum()
    df['Long_Term_Volume'] = df['volume'].rolling(window=20).sum()
    df['Volume_Difference'] = df['Short_Term_Volume'] - df['Long_Term_Volume']
    
    # Adjusted High-Low Range
    df['Scaled_Volume'] = df['volume'] / df['volume'].rolling(window=20).sum()
    df['Adjusted_High_Low_Range'] = (df['high'] - df['low']) * df['Scaled_Volume']
    
    # Previous Day's Return
    df['Prev_Close'] = df['close'].shift(1)
    df['Prev_Open'] = df['open'].shift(1)
    df['Prev_Return'] = (df['close'] - df['Prev_Open']) / df['Prev_Open']
    
    # Realized Volatility
    df['Realized_Volatility'] = np.sqrt(((df['high'] - df['low']) ** 2).rolling(window=10).sum())
    
    # Adjust Short-Term Momentum by Volatility
    df['Adjusted_Short_Term_Momentum'] = df['Short_Term_Momentum'] / df['Realized_Volatility']
    
    # Combine Factors
    df['Combined_Factor'] = (df['Adjusted_High_Low_Range'] * (1 / df['Intraday_Volatility']) + df['Prev_Return'] + 
                             df['Adjusted_Short_Term_Momentum'] + (df['Short_Term_Momentum'] / df['Realized_Volatility']))
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Momentum_Difference'] / df['Volume_Difference'] + df['Adjusted_High_Low_Range'] + 
                          df['Combined_Factor']) * df['volume']
    
    return df['Alpha_Factor']
```
