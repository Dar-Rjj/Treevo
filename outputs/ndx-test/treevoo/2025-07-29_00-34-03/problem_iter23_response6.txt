```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).sum()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).sum()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume_sum = df['volume'].rolling(window=5).sum()
    long_term_volume_sum = df['volume'].rolling(window=20).sum()
    volume_diff = short_term_volume_sum - long_term_volume_sum
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    scaled_volume = df['volume'] / long_term_volume_sum
    adjusted_high_low_range = high_low_range * scaled_volume
    
    # Calculate Previous Day's Return
    previous_day_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Realized Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum_volatility = previous_day_return / realized_volatility
    
    # Combine Factors
    inversed_intraday_range = 1 / (df['high'] - df['low'])
    combined_factors = (adjusted_high_low_range * inversed_intraday_range) + previous_day_return
    combined_factors += short_term_momentum / long_term_momentum
    combined_factors += adjusted_momentum_volatility
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_diff / volume_diff) + adjusted_high_low_range + combined_factors
    final_alpha_factor *= df['volume']
    
    return final_alpha_factor
```
