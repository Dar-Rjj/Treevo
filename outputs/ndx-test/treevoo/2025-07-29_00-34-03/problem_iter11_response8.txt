```python
def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (data['open'] - data['close'].shift(1)) / data['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_close_to_open = data['volume'] * close_to_open_return
    
    # Scale by Volume
    average_volume_20_days = data['volume'].rolling(window=20).mean()
    scaled_volume = data['volume'] / average_volume_20_days
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = (data['close'] - data['open']) / data['open']
    momentum = open_to_close_return.rolling(window=5).sum()
    
    # Calculate Realized Volatility
    high_low_range_squared = (data['high'] - data['low']).pow(2)
    realized_volatility = high_low_range_squared.rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = momentum / realized_volatility
    
    # Combine Factors
    combined_factor = (intraday_range * scaled_volume) + volume_weighted_close_to_open + momentum + adjusted_momentum
    
    # Final Weighting
    final_factor = data['volume'] * combined_factor
    
    return final_factor
```
