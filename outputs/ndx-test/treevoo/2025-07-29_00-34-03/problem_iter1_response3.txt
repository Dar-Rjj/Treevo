```python
import pandas as pd

def heuristics(df):
    # Compute Short-Term Average Close (10-day MA)
    short_term_close = df['close'].rolling(window=10).mean()
    
    # Compute Long-Term Average Close (60-day MA)
    long_term_close = df['close'].rolling(window=60).mean()
    
    # Calculate the Difference Between Short-Term and Long-Term Averages
    price_momentum = short_term_close - long_term_close
    
    # Compute Short-Term Average Volume (10-day MA)
    short_term_volume = df['volume'].rolling(window=10).mean()
    
    # Compute Long-Term Average Volume (60-day MA)
    long_term_volume = df['volume'].rolling(window=60).mean()
    
    # Calculate the Difference Between Short-Term and Long-Term Volumes
    volume_surge = short_term_volume - long_term_volume
    
    # Combine Price Momentum and Volume Surge
    factor = price_momentum + volume_surge

    return factor
```
