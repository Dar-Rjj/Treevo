```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum with Extended Insight
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Identify Advanced Volume Shock
    df['volume_shock'] = (df['volume'] > 1.5 * df['volume'].shift(1)).astype(int)
    
    # Calculate Price Momentum with Multi-Term Averages
    df['short_term_close_avg'] = df['close'].rolling(window=10).mean()
    df['long_term_close_avg'] = df['close'].rolling(window=60).mean()
    df['price_momentum_diff'] = df['short_term_close_avg'] - df['long_term_close_avg']
    
    # Calculate Volume Surge with Multi-Term Averages
    df['short_term_volume_avg'] = df['volume'].rolling(window=10).mean()
    df['long_term_volume_avg'] = df['volume'].rolling(window=60).mean()
    df['volume_surge_diff'] = df['short_term_volume_avg'] - df['long_term_volume_avg']
    
    # Synthesize Indicators for Comprehensive Analysis
    df['alpha_factor'] = (df['intraday_momentum'] * df['volume_shock'] + 
                          df['price_momentum_diff'] + 
                          df['volume_surge_diff'])
    
    return df['alpha_factor']
```
