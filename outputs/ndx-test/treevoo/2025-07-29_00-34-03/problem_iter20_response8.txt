```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the high and low for the day
    daily_range = df['high'] - df['low']
    
    # Calculate the average true range over the last 14 days
    def average_true_range(high, low, close, n=14):
        tr = pd.concat([high - low, abs(high - close.shift(1)), abs(low - close.shift(1))], axis=1).max(axis=1)
        atr = tr.rolling(window=n).mean()
        return atr
    
    atr = average_true_range(df['high'], df['low'], df['close'])
    
    # Calculate a normalized version of the daily range compared to the ATR
    range_factor = (daily_range / atr).fillna(0)
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=5).fillna(0)
    
    # Combine the range and momentum factors in a heuristic way
    alpha_factor = range_factor * momentum
    return alpha_factor
```
