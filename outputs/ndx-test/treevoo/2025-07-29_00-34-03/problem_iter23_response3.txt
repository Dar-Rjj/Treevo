```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the price change from open to close
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the normalized volume
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Heuristic: Price range multiplied by the price change and the volume change
    df['alpha_factor'] = df['price_range'] * df['price_change'] * df['volume_change']
    
    return df['alpha_factor']
```
