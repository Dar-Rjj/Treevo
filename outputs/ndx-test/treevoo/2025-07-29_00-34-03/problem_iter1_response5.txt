```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Weighted Close Price
    weighted_close = df['close'] * df['volume']
    
    # Compute SMA of Volume-Weighted Prices
    sma_weighted_close = weighted_close.rolling(window=20).mean()
    
    # Calculate Realized Volatility using High and Low Prices
    log_returns = np.log(df['high'] / df['low'])
    realized_volatility = log_returns.rolling(window=20).std()
    
    # Smooth Realized Volatility with EMA
    smoothed_volatility = realized_volatility.ewm(span=20, adjust=False).mean()
    
    # Adjust SMA with Smoothed Volatility
    adjusted_sma = sma_weighted_close - smoothed_volatility
    
    return adjusted_sma
```
