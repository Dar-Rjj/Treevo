```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Adjust Weighted Return Using High-Low Spread
    high_low_spread = df['high'] - df['low']
    adjusted_return = (volume_weighted_return / high_low_spread).fillna(0)
    
    # Integrate Volume Momentum
    volume_change = df['volume'] - df['volume'].shift(1)
    final_alpha_factor = adjusted_return + volume_change
    
    return final_alpha_factor
```
