import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Volume-Weighted Price
    volume_weighted_price = df['close'] * df['volume']
    sum_volume_weighted_price = volume_weighted_price.rolling(window=20).sum()
    
    # Exponential Moving Average (EMA) of Volume-Weighted Prices
    ema_volume_weighted_prices = sum_volume_weighted_price.ewm(span=20, adjust=False).mean()
    
    # Intraday Volatility Adjustment Factor
    intraday_volatility = df['high'] - df['low']
    adjusted_ema = ema_volume_weighted_prices + intraday_volatility
    
    # Volume Spike Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike_filter = (volume_change > 0) & (volume_change > df['volume'].shift(1) * 0.5)
    adjusted_alpha_factor = (volume_spike_filter * 1.5) + 1
    
    # Short-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).sum()
    
    # Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=20).sum()
    
    # Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume_sum = df['volume'].rolling(window=5).sum()
    long_term_volume_sum = df['volume'].rolling(window=20).sum()
    volume_difference = short_term_volume_sum - long_term_volume_sum
    
    # Adjusted High-Low Range
    sum_volume_last_20_days = df['volume'].rolling(window=20).sum()
    scaled_volume = df['volume'] / sum_volume_last_20_days
    adjusted_high_low_range = (df['high'] - df['low']) * scaled_volume
    
    # Previous Day's Return
    previous_day_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Adjust Momentum by Volatility
    adjusted_momentum_by_volatility = previous_day_return * intraday_volatility
    
    # Combine Factors
    combined_factors = (adjusted_high_low_range * (1 / intraday_volatility) + previous_day_return + 
                        adjusted_short_term_momentum + adjusted_momentum_by_volatility)
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_difference / volume_difference) + adjusted_high_low_range + combined_factors
    final_alpha_factor *= df['volume']
    
    return final_alpha_factor
