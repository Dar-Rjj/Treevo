```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = (df['close'].shift(0) / df['close'].shift(1)).apply(lambda x: 0 if x == 0 else (x - 1))
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].shift(0) - df['volume'].shift(1)
    
    # Combine Momentum and Volume
    df['preliminary_factor'] = df['log_return'] * df['volume_change']
    
    # Apply Price Filter
    df['avg_price_t'] = (df['high'].shift(0) + df['low'].shift(0)) / 2
    df['avg_price_t_minus_1'] = (df['high'].shift(1) + df['low'].shift(1)) / 2
    
    df['price_filter'] = df['avg_price_t'] > df['avg_price_t_minus_1']
    
    df['final_factor_value'] = df.apply(lambda row: row['preliminary_factor'] * 1.5 if row['price_filter'] else row['preliminary_factor'] * 0.5, axis=1)
    
    return df['final_factor_value'].dropna()
```
