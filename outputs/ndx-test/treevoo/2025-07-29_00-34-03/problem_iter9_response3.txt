```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'].pct_change()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (e.g., 20 days)
    rolling_volatility = daily_return.rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_avg_daily_return / rolling_volatility
    
    # Inverse of Intraday Range
    inversed_intraday_range = 1 / intraday_range
    
    # Combine Factors
    combined_factor = previous_day_return * inversed_intraday_range
    
    # Weight by Volume
    volume_change = df['volume'].pct_change().fillna(0)  # Handle first day with no previous volume
    weighted_factor = combined_factor * volume_change
    
    return weighted_factor
```
