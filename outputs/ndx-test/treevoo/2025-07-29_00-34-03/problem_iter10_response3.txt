```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    df['Rolling_Avg_Return'] = df['Daily_Return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (e.g., 20 days)
    df['Rolling_Volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['Adjusted_Momentum'] = df['Rolling_Avg_Return'] / df['Rolling_Volatility']
    
    # Weight by Volume
    df['Volume_Weighted_Momentum'] = df['volume'] * df['Adjusted_Momentum']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Combine Factors
    df['Factor'] = df['Volume_Weighted_Momentum'] + df['Volume_Change']
    
    return df['Factor']
```
