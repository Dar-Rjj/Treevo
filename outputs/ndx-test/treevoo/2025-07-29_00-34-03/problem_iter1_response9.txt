```python
def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    rolling_window_momentum = 21
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=rolling_window_momentum).mean()
    
    # Calculate Rolling Volatility of Close Prices
    rolling_window_volatility = 20
    df['rolling_volatility'] = df['close'].rolling(window=rolling_window_volatility).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Weight by Volume
    df['volume_weighted_alpha'] = df['volume'] * df['adjusted_momentum']
    
    return df['volume_weighted_alpha'].dropna()
```
