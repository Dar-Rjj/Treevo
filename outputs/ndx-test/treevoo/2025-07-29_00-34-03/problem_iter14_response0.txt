```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate the average price for the current day
    df['avg_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the volume weighted price change
    df['volume_weighted_price_change'] = df['price_change'] * df['volume']
    
    # Calculate the momentum factor as the sum of the last 5 days' volume weighted price changes
    df['momentum_factor'] = df['volume_weighted_price_change'].rolling(window=5).sum()
    
    # Generate the alpha factor as the ratio of the current day's average price to the momentum factor
    alpha_factor = df['avg_price'] / df['momentum_factor']
    
    return alpha_factor
```
