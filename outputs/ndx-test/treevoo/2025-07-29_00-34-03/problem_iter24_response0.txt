```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum (5-day EMA)
    short_momentum = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum (20-day EMA)
    long_momentum = df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    diff_momentum = short_momentum - long_momentum
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    scaled_high_low_range = high_low_range * (df['volume'] / ema_volume_20)
    
    # Calculate Close-to-Open Return
    close_open_return = df['close'] - df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'].diff()
    
    # Adjust Short-Term Momentum
    adjusted_short_momentum = (short_momentum - long_momentum) / intraday_volatility
    
    # Inverse of Intraday Volatility
    inversed_intraday_volatility = 1 / intraday_volatility
    
    # Combine Factors
    combined_factor = (scaled_high_low_range * inversed_intraday_volatility) + previous_day_return + adjusted_short_momentum
    
    # Final Alpha Factor
    final_alpha = (diff_momentum + scaled_high_low_range + close_open_return) * df['volume']
    
    return final_alpha
```
