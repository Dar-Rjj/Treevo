def heuristics_v2(df):
    # Calculate Close-to-Open return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Multiply by volume to get the volume weighted close-to-open return
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Return the factor as a pandas Series
    return df['volume_weighted_close_to_open_return']
