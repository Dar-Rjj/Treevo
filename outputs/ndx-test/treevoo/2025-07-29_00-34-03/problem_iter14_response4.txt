```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(1) / df['open'] - 1
    
    # Weight by Volume and Amount
    weighted_close_to_open = (df['volume'] * df['amount'] * close_to_open_return)
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    open_to_close_momentum = open_to_close_return.rolling(window=5).sum()
    
    # Calculate Realized Volatility
    high_low_range = df['high'] - df['low']
    realized_volatility = (high_low_range ** 2).rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    daily_return = df['close'].pct_change()
    avg_daily_return = daily_return.rolling(window=21).mean()
    rolling_volatility = daily_return.rolling(window=20).std()
    adjusted_momentum = (avg_daily_return / rolling_volatility) / realized_volatility
    
    # Combine Factors
    combined_factor = (intraday_range * scaled_volume +
                       weighted_close_to_open +
                       open_to_close_momentum +
                       adjusted_momentum)
    
    # Final Weighting
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    vol_relative_to_ma = df['volume'] / avg_volume_20
    final_weighting = (df['volume'] * combined_factor * volume_change * vol_relative_to_ma)
    
    # Final Factor
    final_factor = final_weighting
    
    return final_factor
```
