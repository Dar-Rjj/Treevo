```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['weighted_close'] = df['close'] * df['volume']
    df['sum_weighted_close'] = df['weighted_close'].rolling(window=20).sum()
    df['volume_sum'] = df['volume'].rolling(window=20).sum()
    df['volume_weighted_price'] = df['sum_weighted_close'] / df['volume_sum']

    # Calculate EMA of Volume-Weighted Prices
    df['ema_volume_weighted_price'] = df['volume_weighted_price'].ewm(span=20, adjust=False).mean()

    # Intraday Volatility Adjustment Factor
    df['intraday_volatility'] = df['high'] - df['low']
    df['adjusted_ema'] = df['ema_volume_weighted_price'] + df['intraday_volatility']

    # Volume Spike Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > 3 * df['volume'].rolling(window=20).std()).astype(int)
    df['final_alpha_factor'] = df['adjusted_ema'] + df['volume_spike'] * 5  # Increase Alpha if spike detected

    # Short-Term Momentum
    df['avg_close_4_days'] = df['close'].rolling(window=4).mean()
    df['short_term_momentum'] = df['close'] - df['avg_close_4_days']
    df['final_alpha_factor'] += df['short_term_momentum']

    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].ewm(span=20, adjust=False).mean()
    df['difference_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    df['final_alpha_factor'] += df['difference_momentum']

    # Adjusted High-Low Range
    df['scaled_high_low_range'] = (df['high'] - df['low']) * df['volume'] / df['volume'].rolling(window=20).mean()

    # Previous Day's Return
    df['prev_day_return'] = df['close'] / df['open'].shift(1) - 1

    # Adjust Momentum by Volatility
    df['volatility'] = df['close'].pct_change().rolling(window=20).std()
    df['momentum_adjusted_by_volatility'] = df['prev_day_return'] / df['volatility']
    df['final_alpha_factor'] += df['scaled_high_low_range'] * (1 / df['intraday_volatility']) + df['prev_day_return'] + df['momentum_adjusted_by_volatility']

    # True Average Price
    df['true_avg_price'] = (df['high'] + df['low'] + df['close']) / 3

    # Adjusted True Average Price
    df['inv_price_range_volatility'] = 1 / df['intraday_volatility']
    df['adjusted_true_avg_price'] = df['true_avg_price'] * df['inv_price_range_volatility']

    # Final Momentum Indicator
    df['volume_price_movement_score'] = df['volume'] / df['volume'].shift(5)
    df['final_momentum_indicator'] = df['adjusted_true_avg_price'] * df['volume_price_movement_score']

    # Integrate with Combined Momentum and Volume Trends
    df['volume_momentum'] = df['volume'] / df['volume'].shift(5)
    df['volume_momentum_adjusted'] = df['volume_momentum'] / df['volatility']
    df['integrated_alpha'] = 0.5 * df['final_alpha_factor'] + 0.5 * df['final_momentum_indicator'] + df['volume_momentum_adjusted']

    return df['integrated_alpha']
```
