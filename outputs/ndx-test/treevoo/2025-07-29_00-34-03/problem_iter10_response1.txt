```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Momentum and Volume
    df['prelim_factor'] = df['log_return'] * df['volume_change']
    
    # Apply Price Filter
    df['avg_price_today'] = (df['open'] + df['close']) / 2
    df['avg_price_yesterday'] = (df['open'].shift(1) + df['close'].shift(1)) / 2
    df['price_filter'] = df['avg_price_today'] > df['avg_price_yesterday']
    df['final_factor'] = df['prelim_factor']
    df.loc[df['price_filter'], 'final_factor'] += df['log_return']
    df.loc[~df['price_filter'], 'final_factor'] -= df['log_return']
    
    # Incorporate Intraday Dynamics
    df['intraday_momentum'] = df['high'] - df['low']
    df['weighted_intraday_momentum'] = df['intraday_momentum'] * df['volume']
    
    # Identify Volume Shock
    df['volume_shock'] = (df['volume'] > 1.5 * df['volume'].shift(1)).astype(int)
    
    # Smooth Realized Volatility
    df['realized_volatility'] = np.sqrt(((np.log(df['high']) - np.log(df['low'])) ** 2).rolling(window=20).mean())
    df['smoothed_volatility'] = df['realized_volatility'].ewm(span=20, adjust=False).mean()
    
    # Adjust Combined Momentum Factor
    df['adjusted_factor'] = (df['final_factor'] * df['volume_shock']) - df['smoothed_volatility']
    
    return df['adjusted_factor']
```
