```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale by Volume
    average_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / average_volume_20_days
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'].pct_change(1)
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.rolling(window=5).sum()
    
    # Calculate Realized Volatility
    high_low_range = df['high'] - df['low']
    realized_volatility = (high_low_range ** 2).rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    adjusted_momentum_by_volatility = previous_day_return / realized_volatility
    
    # Combine Factors
    combined_factor = (intraday_range * scaled_volume) + open_to_close_momentum + adjusted_momentum_by_volatility
    
    # Weight by Volume
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
