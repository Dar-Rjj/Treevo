```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the closing price relative to the intraday price range
    df['close_to_range_ratio'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate the volume-weighted close-to-range ratio
    df['volume_weighted_ctr'] = df['close_to_range_ratio'] * df['volume']
    
    # Calculate the factor as a smoothed moving average of the volume-weighted close-to-range ratio
    df['factor'] = df['volume_weighted_ctr'].rolling(window=10, min_periods=1).mean()
    
    return df['factor']
```
