```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume = df['volume'].rolling(window=5).mean()
    long_term_volume = df['volume'].rolling(window=20).mean()
    volume_trend = short_term_volume - long_term_volume
    
    # Calculate Adjusted High-Low Range
    high_low_range = (df['high'] - df['low']) * (df['volume'] / long_term_volume)
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df.shift(1)['open']) / df.shift(1)['open']
    
    # Weight Close-to-Open Return by Volume and Amount
    weighted_close_to_open_return = close_to_open_return * df['volume'] * df['amount']
    
    # Adjust Close-to-Open Return by Price Volatility
    price_range = df['high'] - df['low']
    adjusted_close_to_open_return = weighted_close_to_open_return / price_range
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Intraday Volatility
    intraday_volatility = intraday_range
    
    # Calculate Previous Day's Return
    previous_day_return = (df['close'] - df.shift(1)['open']) / df.shift(1)['open']
    
    # Short-Term Momentum
    short_term_avg_close = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - short_term_avg_close
    adjusted_momentum = current_momentum / intraday_volatility
    
    # Calculate Realized Volatility
    realized_volatility = ((df['high'] - df['low']) ** 2).rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    volatility_adjusted_momentum = previous_day_return / realized_volatility
    
    # Combine Factors
    adjusted_high_low_range = high_low_range * (1 / intraday_range)
    combined_factors = adjusted_high_low_range + previous_day_return + adjusted_momentum + volatility_adjusted_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_diff / volume_trend) + high_low_range + adjusted_close_to_open_return + combined_factors
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
```
