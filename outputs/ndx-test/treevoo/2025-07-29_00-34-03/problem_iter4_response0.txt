```python
def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale by Volume
    scaled_high_low = high_low_range * df['volume']
    
    # Calculate Close-to-Open Momentum
    close_open_momentum = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Range and Close-to-Open Momentum
    factor_values = scaled_high_low + close_open_momentum
    
    return factor_values
```
