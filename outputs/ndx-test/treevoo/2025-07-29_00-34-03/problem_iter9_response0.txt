```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    weighted_return = close_to_open_return * df['volume'] * df['amount']
    
    # Adjust by Price Volatility
    price_range = df['high'] - df['low']
    price_volatility = price_range / df['close']
    
    # Incorporate into Final Factor
    final_factor = weighted_return / price_volatility
    
    return final_factor
```
