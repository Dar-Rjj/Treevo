```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = df['close'] * df['volume']
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    df['SMA_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].rolling(window=21).mean()
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Scale by Volume
    df['Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume_Scaled'] = df['volume'] / df['Average_Volume']
    
    # Calculate Previous Day's Return
    df['Prev_Day_Return'] = df['close'].pct_change(1)
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['Rolling_Avg_Daily_Return'] = df['Daily_Return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns
    df['Rolling_Volatility'] = df['Daily_Return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['Momentum_Adjusted'] = df['Rolling_Avg_Daily_Return'] / df['Rolling_Volatility']
    
    # Combine Factors
    df['Inverted_Intraday_Range'] = 1 / df['Intraday_Range']
    df['Combined_Factor'] = df['Prev_Day_Return'] * df['Inverted_Intraday_Range'] * df['Momentum_Adjusted']
    
    # Final Volume and Volatility Adjustment
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Surge'] = (df['Volume_Change'] > 0) & (df['Volume_Change'] > df['Volume_Change'].quantile(0.9))
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['Volume_Scaled'] * df['Volume_Surge'].astype(int)
    
    return df['Final_Alpha_Factor']
```
