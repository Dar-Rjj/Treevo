```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    rolling_window = 21
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=rolling_window).mean()
    
    # Calculate Rolling Volatility of Close Prices (e.g., 20 days)
    volatility_window = 20
    df['rolling_volatility_close'] = df['close'].pct_change().rolling(window=volatility_window).std()
    
    # Calculate Rolling Volatility of Volumes (e.g., 20 days)
    df['rolling_volatility_volume'] = df['volume'].pct_change().rolling(window=volatility_window).std()
    
    # Adjust Momentum by Price and Volume Volatility
    df['adjusted_momentum'] = df['rolling_avg_daily_return'] / (df['rolling_volatility_close'] + df['rolling_volatility_volume'])
    
    # Weight by Volume
    df['heuristics_factor'] = df['volume'] * df['adjusted_momentum']
    
    return df['heuristics_factor'].dropna()
```
