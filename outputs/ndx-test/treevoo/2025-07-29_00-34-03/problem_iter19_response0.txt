```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (e.g., 20 days)
    df['rolling_volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Weight by Volume
    df['weighted_momentum'] = df['volume'] * df['adjusted_momentum']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Combine Factors
    df['combined_factors'] = df['weighted_momentum'] + df['volume_change']
    
    # Calculate Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['adjusted_volume_momentum'] = df['volume_momentum'] / df['rolling_volatility']
    
    # Integrate Volume Momentum with Combined Factors
    weight = 0.5
    df['final_factor'] = df['combined_factors'] + weight * df['adjusted_volume_momentum']
    
    return df['final_factor']
```
