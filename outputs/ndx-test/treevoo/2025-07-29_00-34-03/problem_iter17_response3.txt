```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'].pct_change()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (21 days)
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (20 days)
    df['rolling_volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['momentum_adjusted'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Combine Factors
    df['inv_intraday_range'] = 1 / df['intraday_range']
    df['momentum_adj_by_inv_range'] = df['prev_day_return'] * df['inv_intraday_range']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Final Factor Calculation
    df['final_factor'] = df['momentum_adj_by_inv_range'] * df['volume_change'] * df['close_to_open_return'] * df['amount']
    
    # Calculate Volume Momentum
    df['volume_momentum'] = df['volume'] / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['adj_volume_momentum'] = df['volume_momentum'] / df['rolling_volatility']
    
    # Integrate Volume Momentum
    df['final_factor'] += 0.5 * df['adj_volume_momentum']
    
    # Incorporate High and Low Prices
    df['high_low_spread'] = (df['high'] - df['low']) / df['close']
    
    # Adjust Weighted Return
    df['final_factor'] = df['final_factor'] * (1 + df['high_low_spread'])
    
    # Integrate Momentum
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['momentum'] = df['close'] - df['sma_10']
    df['final_factor'] += df['momentum']
    
    # Incorporate Volatility
    df['volatility_20'] = df['close'].rolling(window=20).std()
    df['final_factor'] /= df['volatility_20']
    
    return df['final_factor'].dropna()
```
