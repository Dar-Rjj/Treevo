```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['weighted_close'] = df['volume'] * df['close']
    volume_weighted_price = df['weighted_close'].sum() / df['volume'].sum()
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    sma_volume_weighted = df['weighted_close'].rolling(window=20).mean() / df['volume'].rolling(window=20).mean()
    
    # Intraday Volatility Adjustment Factor
    intraday_volatility = df['high'] - df['low']
    sma_volume_weighted_adjusted = sma_volume_weighted - intraday_volatility
    
    # Volume Surge Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_surge = df['volume_change'] > 0.2 * df['volume'].rolling(window=5).mean()
    volume_surge_factor = sma_volume_weighted_adjusted + (sma_volume_weighted_adjusted * volume_surge * 0.1)
    
    # Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].rolling(window=4).mean()
    
    # True Average Price
    true_average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Price Range Volatility
    daily_price_range = df['high'] - df['low']
    price_range_volatility = daily_price_range.rolling(window=21).std()
    
    # Daily Return
    daily_return = df['close'].pct_change()
    
    # Rolling Average Daily Return
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()
    
    # Rolling Volatility of Close Prices
    rolling_volatility = df['close'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_avg_daily_return / rolling_volatility
    
    # Volume-Price Movement Score
    volume_price_movement_score = df['volume'] * (df['high'] - df['open'])
    weighted_score = volume_price_movement_score * daily_price_range
    
    # Incorporate Volume Trend
    short_term_volume_avg = df['volume'].rolling(window=5).mean()
    long_term_volume_avg = df['volume'].rolling(window=20).mean()
    volume_trend = short_term_volume_avg - long_term_volume_avg
    
    # Combine Momentum and Volume Trends
    combined_trend = momentum_difference + volume_trend
    
    # Adjusted True Average Price
    adjusted_true_avg_price = true_average_price / price_range_volatility
    
    # Final Momentum Indicator
    final_momentum_indicator = adjusted_true_avg_price * weighted_score
    
    # Integrate with Combined Momentum and Volume Trends
    adjusted_momentum_final = final_momentum_indicator * combined_trend
    
    # Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    weighted_return = close_to_open_return * df['volume']
    
    # Incorporate High-Low Spread
    high_low_spread = df['high'] - df['low']
    adjusted_weighted_return = weighted_return * high_low_spread
    
    # Integrate Momentum
    sma_10_day = df['close'].rolling(window=10).mean()
    momentum_adjustment = df['close'] - sma_10_day
    integrated_return = adjusted_weighted_return + momentum_adjustment
    
    # Incorporate Volatility
    volatility_adjusted_return = integrated_return / df['close'].rolling(window=20).std()
    
    # Final Alpha Factor
    alpha_factor = volatility_adjusted_return * adjusted_momentum_final
    
    return alpha_factor
```
