```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['weighted_close'] = df['volume'] * df['close']
    df['volume_weighted_price'] = df['weighted_close'] / df['volume'].rolling(window=20).mean()
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    df['sma_volume_weighted_price'] = df['volume_weighted_price'].rolling(window=20).mean()
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Scale by Volume
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20_days']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'].pct_change()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns
    df['rolling_volatility_daily_return'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['momentum_adjustment'] = df['rolling_avg_daily_return'] / df['rolling_volatility_daily_return']
    
    # Combine Factors
    df['inv_intraday_range'] = 1 / df['intraday_range']
    df['combined_factor'] = df['prev_day_return'] * df['inv_intraday_range']
    
    # Final Volume and Volatility Adjustment
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['final_factor'] = df['combined_factor'] * df['scaled_volume'] * df['volume_change']
    
    # Volume Surge Indicator
    volume_surge_threshold = 1.5
    df['volume_surge'] = (df['volume_change'] > 0) & (df['volume_change'] > volume_surge_threshold)
    df['final_factor'] += df['volume_surge'] * df['final_factor']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume, Amount, and Close-to-Open Return
    df['final_factor'] = df['final_factor'] * df['volume_change'] * df['amount'] * df['close_to_open_return']
    
    return df['final_factor']
```
