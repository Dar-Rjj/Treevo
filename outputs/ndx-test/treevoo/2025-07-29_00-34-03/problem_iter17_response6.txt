```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    df['avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (e.g., 20 days)
    df['volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['avg_daily_return'] / df['volatility']
    
    # Weight by Volume
    df['weighted_by_volume'] = df['adjusted_momentum'] * df['volume']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Volume Momentum (e.g., n=5)
    df['volume_momentum'] = df['volume'] / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['adjusted_volume_momentum'] = df['volume_momentum'] / df['volatility']
    
    # Combine Factors
    df['combined_factors'] = df['weighted_by_volume'] + df['volume_change']
    
    # Integrate Volume Momentum
    df['integrated_vol_mom'] = df['combined_factors'] + 0.5 * df['adjusted_volume_momentum']
    
    # Calculate Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Rolling Average Price Range (e.g., 21 days)
    df['avg_price_range'] = df['price_range'].rolling(window=21).mean()
    
    # Adjust Price Range by Rolling Average
    df['adjusted_price_range'] = df['price_range'] / df['avg_price_range']
    
    # Integrate Price Range into Combined Factors
    df['integrated_price_range'] = df['integrated_vol_mom'] + (df['adjusted_price_range'] * df['volume'])
    
    # Calculate Price-Volume Divergence
    df['price_volume_divergence'] = df['daily_return'] - df['volume_change']
    
    # Calculate Rolling Average Price-Volume Divergence (e.g., 21 days)
    df['avg_price_volume_divergence'] = df['price_volume_divergence'].rolling(window=21).mean()
    
    # Adjust Price-Volume Divergence by Volatility
    df['adjusted_price_volume_divergence'] = df['avg_price_volume_divergence'] / df['volatility']
    
    # Integrate Adjusted Price-Volume Divergence into Combined Factors
    df['final_factor'] = df['integrated_price_range'] + 0.3 * df['adjusted_price_volume_divergence']
    
    return df['final_factor'].dropna()
```
