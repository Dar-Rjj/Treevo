```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume = df['volume'].rolling(window=5).mean()
    long_term_volume = df['volume'].rolling(window=20).mean()
    volume_diff = short_term_volume - long_term_volume
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale High-Low Range by Volume
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20_days
    scaled_high_low_range = high_low_range * scaled_volume
    
    # Calculate Adjusted High-Low Range
    adjusted_high_low_range = high_low_range * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Combine Factors
    inversed_intraday_range = 1 / intraday_range
    previous_day_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    combined_factors = adjusted_high_low_range * inversed_intraday_range + previous_day_return
    
    # Final Alpha Factor
    price_momentum_factor = momentum_diff / volume_diff
    final_alpha_factor = price_momentum_factor + scaled_high_low_range + close_to_open_return + combined_factors
    
    return final_alpha_factor
```
