```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Daily Volume Averages
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20']

    # Previous Day's Return
    df['prev_open'] = df['open'].shift(1)
    df['prev_close'] = df['close'].shift(1)
    df['prev_return'] = (df['close'] - df['prev_open']) / df['prev_open']

    # Short-Term Momentum
    df['avg_close_4'] = df['close'].rolling(window=4).mean()
    df['current_momentum'] = df['close'] - df['avg_close_4']
    df['adjusted_momentum'] = df['current_momentum'] / df['intraday_volatility']

    # Realized Volatility
    df['realized_volatility'] = (df['high'] - df['low']).pow(2).rolling(window=10).sum()

    # Adjust Momentum by Volatility
    df['momentum_by_volatility'] = df['prev_return'] / df['realized_volatility']

    # Combine Factors
    df['combined_factor'] = (df['intraday_range'] * df['scaled_volume'] +
                              df['adjusted_momentum'] +
                              df['momentum_by_volatility'])

    # Weight by Volume
    df['heuristics_factor'] = df['combined_factor'] * df['volume']

    return df['heuristics_factor']
```
