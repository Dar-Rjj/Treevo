```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted price change
    weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Calculate the normalized difference between the closing price and the average of high and low
    close_to_avg_diff = df['close'] - avg_price
    
    # Generate the alpha factor by combining the weighted price change and the close to average difference
    alpha_factor = weighted_price_change + close_to_avg_diff
    
    return alpha_factor
```
