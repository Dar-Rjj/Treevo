```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = open_to_close_return * df['volume']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Calculate Short-Term Price Momentum
    short_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_momentum - long_momentum
    
    # Calculate Adjusted Intraday Range
    adjusted_intraday_range = intraday_range * scaled_volume
    
    # Calculate Realized Volatility
    high_low_range = df['high'] - df['low']
    realized_volatility = ((high_low_range.rolling(window=10).sum()) ** 2).sum() ** 0.5
    
    # Adjust Short-Term Momentum by Volatility
    adjusted_short_momentum = short_momentum / realized_volatility
    
    # Combine Factors
    combined_factor = (
        adjusted_intraday_range +
        volume_weighted_return +
        adjusted_short_momentum +
        momentum_diff
    )
    
    # Final Weighting
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
