```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'].pct_change()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    df['rolling_avg_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (e.g., 20 days)
    df['rolling_volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['momentum_adjusted'] = df['rolling_avg_return'] / df['rolling_volatility']
    
    # Inverse of Intraday Range
    df['inv_intraday_range'] = 1 / df['intraday_range']
    
    # Momentum Adjustment
    df['momentum_adjustment'] = df['prev_day_return'] * df['inv_intraday_range']
    
    # Combine Factors
    df['combined_factor'] = df['momentum_adjusted'] * df['momentum_adjustment']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Final Factor Calculation
    df['final_factor'] = df['combined_factor'] * df['volume_change'] * df['close_to_open_return'] * df['amount']
    
    return df['final_factor']
```
