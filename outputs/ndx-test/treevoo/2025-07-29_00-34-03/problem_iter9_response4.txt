```python
def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Weight by Volume, Amount, and Volatility
    df['volume_weight'] = df['volume'] / df['volume'].sum()
    df['amount_weight'] = df['amount'] / df['amount'].sum()
    df['volatility_weight'] = df['daily_volatility'] / df['daily_volatility'].sum()
    
    # Multiply Weights by Return
    df['factor_value'] = df['close_to_open_return'] * df['volume_weight'] * df['amount_weight'] * df['volatility_weight']
    
    return df['factor_value']
```
