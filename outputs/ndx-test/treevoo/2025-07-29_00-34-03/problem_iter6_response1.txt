```python
import pandas as pd

def heuristics_v2(df):
    """
    Computes a weighted close-to-open return, where the weights are based on volume and amount.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume' and index as 'date'.
        
    Returns:
        pd.Series: A Series with the calculated factor values, indexed by 'date'.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    weight = df['volume'] * df['amount']
    weighted_close_to_open_return = close_to_open_return * weight
    
    return weighted_close_to_open_return
```
