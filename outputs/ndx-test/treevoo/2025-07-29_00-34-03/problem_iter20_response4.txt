import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Weight by Volume and Amount
    weighted_close_to_open = close_to_open_return * df['volume'] * df['amount']
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    avg_open_to_close_5 = open_to_close_return.rolling(window=5).mean()
    
    # Calculate Realized Volatility
    high_low_range = df['high'] - df['low']
    realized_volatility = (high_low_range ** 2).rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    rolling_avg_daily_return = df['close'].pct_change().rolling(window=21).mean()
    rolling_volatility = df['close'].pct_change().rolling(window=20).std()
    adjusted_momentum = (rolling_avg_daily_return / rolling_volatility) * realized_volatility
    
    # Combine Factors
    combined_factor = (intraday_range * scaled_volume) + weighted_close_to_open + avg_open_to_close_5 + adjusted_momentum
    
    # Final Factor
    final_factor = combined_factor * (df['volume'] / avg_volume_20)
    
    # Adjust Weighted Return Using High-Low Spread
    high_low_spread = df['high'] - df['low']
    adjusted_weighted_return = weighted_close_to_open + high_low_spread
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Adjusted High-Low Range
    adjusted_high_low_range = (df['high'] - df['low']) * (df['volume'] / avg_volume_20)
    
    # Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    prev_day_return = (df['close'] / df['open'].shift(1)) - 1
    
    # Short-Term Momentum
    short_term_avg_close = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - short_term_avg_close
    adjusted_momentum = current_momentum / intraday_volatility
    
    # Combine Factors
    combined_factors = (adjusted_high_low_range * (1 / intraday_range)) + prev_day_return + adjusted_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_difference / (df['volume'].rolling(window=5).mean() - df['volume'].rolling(window=20).mean())) + adjusted_high_low_range + close_to_open_return + combined_factors
    
    # Weight by Volume
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
