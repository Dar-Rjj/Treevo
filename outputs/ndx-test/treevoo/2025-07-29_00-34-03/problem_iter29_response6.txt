```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'].shift(1) - df['open']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Weight by Volume
    df['volume_weighted_return'] = df['volume'] * df['open_to_close_return'] * df['close_to_open_return']
    
    # Scale by Volume
    df['average_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['average_volume_20_days']
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    df['momentum_difference'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Adjusted Intraday Range
    df['adjusted_intraday_range'] = df['intraday_range'] * df['scaled_volume']
    
    # Calculate Realized Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['realized_volatility'] = df['high_low_range'].rolling(window=10).apply(lambda x: (x**2).sum())
    
    # Adjust Short-Term Momentum by Volatility
    df['volatility_adjusted_momentum'] = df['short_term_momentum'] / df['realized_volatility']
    
    # Combine Factors
    df['combined_factor'] = (
        df['adjusted_intraday_range'] +
        df['volume_weighted_return'] +
        df['close_to_open_return'] +
        df['volatility_adjusted_momentum'] +
        df['momentum_difference']
    )
    
    # Final Weighting
    df['final_factor'] = df['volume'] * df['combined_factor']
    
    return df['final_factor']
```
