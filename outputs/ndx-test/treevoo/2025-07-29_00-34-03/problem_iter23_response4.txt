```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['weighted_close'] = df['volume'] * df['close']
    df['vwap'] = df['weighted_close'].rolling(window=20, min_periods=1).sum() / df['volume'].rolling(window=20, min_periods=1).sum()

    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Prices
    df['ema_vwap'] = df['vwap'].ewm(span=20, adjust=False).mean()

    # Intraday Volatility Adjustment Factor
    df['intraday_volatility'] = df['high'] - df['low']
    df['adjusted_ema'] = df['ema_vwap'] + df['intraday_volatility']

    # Volume Spike Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_spike_threshold = 3.0  # Example threshold
    df['volume_spike'] = df['volume_change'].apply(lambda x: 1 if x > volume_spike_threshold else 0)

    # Adjust Final Alpha Factor
    df['adjusted_ema_with_spike'] = df['adjusted_ema'] + df['volume_spike'] * df['intraday_volatility']

    # Short-Term Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5, min_periods=1).sum()

    # Long-Term Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20, min_periods=1).sum()

    # Determine the Difference between Short and Long-Term Momentum
    df['momentum_diff'] = df['short_term_momentum'] - df['long_term_momentum']

    # Incorporate Volume Trend
    df['short_term_volume_sum'] = df['volume'].rolling(window=5, min_periods=1).sum()
    df['long_term_volume_sum'] = df['volume'].rolling(window=20, min_periods=1).sum()
    df['volume_diff'] = df['short_term_volume_sum'] - df['long_term_volume_sum']

    # Calculate Adjusted High-Low Range
    df['adjusted_high_low_range'] = (df['high'] - df['low']) * df['volume'] / df['long_term_volume_sum']

    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'] / df['open'].shift(1) - 1

    # Adjust Momentum by Volatility
    df['realized_volatility'] = df['prev_day_return'].rolling(window=20, min_periods=1).std() * np.sqrt(252)
    df['momentum_adjusted_by_volatility'] = df['momentum_diff'] / df['realized_volatility']

    # Combine Factors
    df['combined_factors'] = df['adjusted_high_low_range'] * (1 / df['intraday_volatility']) + df['prev_day_return'] + df['momentum_adjusted_by_volatility']

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum_diff'] / df['volume_diff']) + df['adjusted_high_low_range'] + df['combined_factors'] * df['volume']

    return df['final_alpha_factor']
```
