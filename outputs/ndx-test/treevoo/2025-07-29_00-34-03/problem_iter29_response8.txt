import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Calculate Volume-Weighted Open
    df['volume_weighted_open'] = df['open'] * df['volume']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['volume_weighted_close'] - df['volume_weighted_open']) / df['volume_weighted_open']
    
    # Weight by Volume
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Enhance Weighting by Recent Volume
    df['recent_volume'] = df['volume'].rolling(window=5).sum()
    df['enhanced_weight'] = df['weighted_close_to_open_return'] * df['recent_volume']
    
    # Return the alpha factor
    return df['enhanced_weight']
