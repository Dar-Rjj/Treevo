import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = df['volume'] * df['close']
    sum_volume_weighted_price = df['Volume_Weighted_Price'].rolling(window=5).sum()
    sum_volume = df['volume'].rolling(window=5).sum()
    df['Volume_Weighted_Price'] = sum_volume_weighted_price / sum_volume
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    df['SMA_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].rolling(window=5).mean()
    
    # Intraday Volatility Adjustment
    df['Intraday_Volatility'] = df['high'] - df['low']
    df['Adjusted_SMA'] = df['SMA_Volume_Weighted_Price'] - df['Intraday_Volatility']
    
    # Volume Surge Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Surge'] = df['Volume_Change'] > 0.1 * df['volume'].shift(1)
    df['Volume_Surge_Adjustment'] = df['Volume_Surge'] * df['Volume_Change']
    
    # Short-Term Momentum
    df['Avg_Close_4_Days'] = df['close'].rolling(window=4).mean()
    df['Short_Term_Momentum'] = df['close'] - df['Avg_Close_4_Days']
    
    # True Average Price
    df['True_Average_Price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Long-Term Price Momentum
    df['Avg_Close_20_Days'] = df['close'].rolling(window=20).mean()
    df['Long_Term_Momentum'] = df['close'] - df['Avg_Close_20_Days']
    
    # Difference between Short and Long-Term Momentum
    df['Momentum_Difference'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Rolling Average Daily Return
    df['Rolling_Avg_Return'] = df['Daily_Return'].rolling(window=21).mean()
    
    # Rolling Volatility of Close Prices
    df['Rolling_Volatility'] = df['close'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['Adjusted_Momentum'] = df['Rolling_Avg_Return'] / df['Rolling_Volatility']
    
    # Volume-Price Movement Score
    df['Price_Range'] = df['high'] - df['open']
    df['Volume_Price_Movement_Score'] = df['volume'] * df['Price_Range']
    
    # Volume Trend
    df['Avg_Volume_5_Days'] = df['volume'].rolling(window=5).mean()
    df['Avg_Volume_20_Days'] = df['volume'].rolling(window=20).mean()
    df['Volume_Trend'] = df['Avg_Volume_5_Days'] - df['Avg_Volume_20_Days']
    
    # Combined Momentum and Volume Trends
    df['Combined_Momentum_Volume'] = df['Momentum_Difference'] + df['Volume_Trend']
    
    # Adjusted True Average Price
    df['Adjusted_True_Average_Price'] = df['True_Average_Price'] / df['Intraday_Volatility']
    
    # Final Momentum Indicator
    df['Final_Momentum_Indicator'] = df['Adjusted_True_Average_Price'] * df['Volume_Price_Movement_Score']
    
    # Enhanced Momentum-Volatility Interaction
    df['EMA_Close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Enhanced_Momentum_Volatility'] = df['EMA_Close'] / df['Rolling_Volatility']
    
    # Volume-Adjusted Momentum
    df['Cumulative_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].cumsum()
    df['Volume_Adjusted_Momentum'] = df['Cumulative_Volume_Weighted_Price'].diff()
    df['Volume_Adjusted_Momentum'] += df['Volume_Surge_Adjustment']
    
    # Integrated Alpha Factor
    df['Integrated_Alpha_Factor'] = df['Enhanced_Momentum_Volatility'] * df['Volume_Adjusted_Momentum']
    
    # Price Range Momentum
    df['Average_Price_Range_5_Days'] = df['Price_Range'].rolling(window=5).mean()
