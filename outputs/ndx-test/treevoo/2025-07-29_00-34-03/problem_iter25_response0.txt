```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Multiply by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust Weighted Return with High-Low Spread
    adjusted_weighted_return = volume_weighted_return + high_low_spread
    
    # Calculate 10-Day Exponential Moving Average (EMA) of Close Price
    ema_10_day = df['close'].ewm(span=10, adjust=False).mean()
    
    # Subtract 10-Day EMA from Current Close Price
    enhanced_momentum = df['close'] - ema_10_day
    
    # Add Enhanced Momentum to the Adjusted Return
    adjusted_return_with_momentum = adjusted_weighted_return + enhanced_momentum
    
    # Calculate 20-Day Standard Deviation of Log Returns
    log_returns = np.log(df['close']) - np.log(df['close'].shift(1))
    std_20_day = log_returns.rolling(window=20).std()
    
    # Divide Adjusted Return by 20-Day Standard Deviation
    alpha_factor = adjusted_return_with_momentum / std_20_day
    
    return alpha_factor
```
