import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Weighted Close Price
    df['weighted_close'] = df['volume'] * df['close']
    
    # Calculate Simple Moving Average (SMA) of Weighted Close Prices
    sma_window = 20
    df['sma_weighted_close'] = df['weighted_close'].rolling(window=sma_window).mean()
    
    # Calculate Realized Volatility using High and Low Prices
    df['log_high_low_returns'] = np.log(df['high'] / df['low'])
    realized_volatility = df['log_high_low_returns'].rolling(window=sma_window).std()
    
    # Calculate Exponential Moving Average (EMA) of Realized Volatility
    ema_volatility_window = 10
    df['ema_realized_volatility'] = df['log_high_low_returns'].ewm(span=ema_volatility_window, min_periods=sma_window).std()
    
    # Subtract Smoothed Volatility from SMA
    df['adjusted_sma_weighted_close'] = df['sma_weighted_close'] - df['ema_realized_volatility']
    
    # Return the factor values
    return df['adjusted_sma_weighted_close']
