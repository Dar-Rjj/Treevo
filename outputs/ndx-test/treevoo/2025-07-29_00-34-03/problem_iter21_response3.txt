```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_momentum = df['close'].rolling(window=5).sum()
    
    # Calculate Long-Term Price Momentum
    long_momentum = df['close'].rolling(window=20).sum()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_momentum - long_momentum
    
    # Incorporate Volume Trend
    short_volume_sum = df['volume'].rolling(window=5).sum()
    long_volume_sum = df['volume'].rolling(window=20).sum()
    volume_diff = short_volume_sum - long_volume_sum
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    volume_scale = df['volume'] / long_volume_sum
    adjusted_high_low_range = high_low_range * volume_scale
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'].pct_change()
    
    # Adjust Momentum by Volatility
    realized_volatility = previous_day_return.rolling(window=20).std()
    adjusted_momentum_by_volatility = previous_day_return / realized_volatility
    
    # Combine Factors
    inversed_intraday_range = 1 / (df['high'] - df['low'])
    combined_factor = (adjusted_high_low_range * inversed_intraday_range) + previous_day_return + short_momentum + adjusted_momentum_by_volatility
    
    # Final Alpha Factor
    alpha_factor = (momentum_diff / volume_diff) + adjusted_high_low_range + combined_factor
    final_alpha_factor = alpha_factor * df['volume']
    
    return final_alpha_factor
```
