```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate True Average Price
    df['true_avg_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    df['momentum_diff'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Price Range Volatility
    df['price_range_volatility'] = df['daily_price_range'].rolling(window=21).std()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices
    df['rolling_volatility_close'] = df['close'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['avg_daily_return'] / df['rolling_volatility_close']
    
    # Weighting by Volume and Price Movement
    df['volume_price_movement_score'] = df['volume'] * (df['high'] - df['open'])
    df['weighted_price_range'] = df['volume_price_movement_score'] * df['daily_price_range']
    
    # Incorporate Volume Trend
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_trend_diff'] = df['avg_volume_5d'] - df['avg_volume_20d']
    
    # Combine Momentum and Volume Trends
    df['combined_momentum_and_volume'] = df['momentum_diff'] + df['volume_trend_diff']
    
    # Adjusted True Average Price
    df['adjusted_true_avg_price'] = df['true_avg_price'] / df['price_range_volatility']
    
    # Final Momentum Indicator
    df['final_momentum_indicator'] = df['adjusted_true_avg_price'] * df['weighted_price_range']
    
    # Integrate with Combined Momentum and Volume Trends
    df['integrated_factor'] = df['adjusted_momentum'] * df['final_momentum_indicator'] * df['combined_momentum_and_volume']
    
    # Return the integrated factor
    return df['integrated_factor'].dropna()
```
