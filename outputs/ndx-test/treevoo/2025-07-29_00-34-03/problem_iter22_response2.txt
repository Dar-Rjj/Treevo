```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_high_low_range = (df['high'] - df['low']) * (df['volume'] / avg_volume_20)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'] - 1
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'] / df['open'].shift(1) - 1
    
    # Adjust Short-Term Momentum
    adjusted_short_term_momentum = (short_term_momentum - long_term_momentum) / intraday_volatility
    
    # Combine Factors
    inversed_intraday_range = 1 / (df['high'] - df['low'])
    combined_factor = (scaled_high_low_range * inversed_intraday_range) + previous_day_return
    
    # Add Adjusted Short-Term Momentum
    combined_factor += adjusted_short_term_momentum
    
    # Final Alpha Factor
    alpha_factor = (momentum_diff + scaled_high_low_range + close_to_open_return) * df['volume']
    
    return alpha_factor
```
