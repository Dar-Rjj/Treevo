```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Multiply by Volume and Amount to get Weighted Close-to-Open Return
    weighted_close_to_open_return = close_to_open_return * df['volume'] * df['amount']
    
    # Calculate Price Range
    price_range = df['high'] - df['low']
    
    # Adjust by Price Volatility
    volatility_adjusted_return = weighted_close_to_open_return / price_range
    
    # Compute Weighted Close Prices
    weighted_close_prices = df['volume'] * df['close']
    
    # Calculate Adjusted SMA of Weighted Close Prices
    sma_weighted_close_prices = weighted_close_prices.rolling(window=10).mean()
    
    # Smooth Realized Volatility
    log_returns = np.log(df['high']) - np.log(df['low'])
    realized_volatility = log_returns.rolling(window=10).std()
    smoothed_volatility = realized_volatility.ewm(span=20).mean()
    
    # Subtract Smoothed Volatility from SMA
    adjusted_sma = sma_weighted_close_prices - smoothed_volatility
    
    # Calculate Short-Term, Medium-Term, and Long-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    medium_term_momentum = df['close'].rolling(window=10).mean()
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short, Medium, and Long-Term Momentum
    diff_short_medium = short_term_momentum - medium_term_momentum
    diff_medium_long = medium_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_trend = avg_volume_5d - avg_volume_20d
    
    # Combine Adjusted SMA, Momentum, and Volatility-Adjusted Return
    combined_factor = adjusted_sma + diff_short_medium + diff_medium_long + volatility_adjusted_return + volume_trend
    
    return combined_factor
```
