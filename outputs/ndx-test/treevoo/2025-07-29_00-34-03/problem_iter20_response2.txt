```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Weight by Volume
    df['weighted_return'] = df['daily_return'] * df['volume']

    # Adjust Weighted Return Using High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['adjusted_weighted_return'] = df['weighted_return'] / df['high_low_spread']

    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()

    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()

    # Determine the Difference between Short and Long-Term Momentum
    df['momentum_difference'] = df['short_term_momentum'] - df['long_term_momentum']

    # Calculate Adjusted High-Low Range
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / avg_volume_20_days
    df['adjusted_high_low_range'] = (df['high'] - df['low']) * df['scaled_volume']

    # Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Previous Day's Return
    df['prev_day_open'] = df['open'].shift(1)
    df['prev_day_close'] = df['close'].shift(1)
    df['prev_day_return'] = (df['close'] - df['prev_day_open']) / df['prev_day_open']

    # Short-Term Momentum
    df['avg_close_price'] = df['close'].rolling(window=4).mean()
    df['current_momentum'] = df['close'] - df['avg_close_price']
    df['adjusted_momentum'] = df['current_momentum'] / df['intraday_volatility']

    # Combine Factors
    df['inverse_intraday_range'] = 1 / df['intraday_volatility']
    df['combined_factors'] = (df['adjusted_high_low_range'] * df['inverse_intraday_range']) + df['prev_day_return']
    df['final_combined_factors'] = df['combined_factors'] + df['adjusted_momentum']

    # Final Alpha Factor
    df['final_alpha_factor'] = df['momentum_difference'] + df['adjusted_high_low_range'] + df['final_combined_factors']

    # Integrate Volume Momentum
    df['volume_momentum'] = df['volume'].pct_change(periods=5)
    df['rolling_volatility'] = df['close'].rolling(window=20).std()
    df['adjusted_volume_momentum'] = df['volume_momentum'] / df['rolling_volatility']
    df['final_alpha_factor'] += 0.5 * df['adjusted_volume_momentum']

    return df['final_alpha_factor']
```
