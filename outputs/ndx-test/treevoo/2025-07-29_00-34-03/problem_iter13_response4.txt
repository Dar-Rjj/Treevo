```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale by Volume
    avg_vol_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_vol_20_days
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume and Amount
    weighted_close_to_open_return = (df['volume'] * df['amount'] * close_to_open_return)
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = df['close'] / df['open'] - 1
    open_to_close_momentum = open_to_close_return.rolling(window=5).sum()
    
    # Calculate Realized Volatility
    high_low_range = df['high'] - df['low']
    realized_volatility = (high_low_range**2).rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    adjusted_momentum_by_volatility = close_to_open_return / realized_volatility
    
    # Combine Factors
    combined_factor = (intraday_range * scaled_volume +
                       weighted_close_to_open_return +
                       open_to_close_momentum +
                       adjusted_momentum_by_volatility)
    
    # Final Weighting
    final_factor = df['volume'] * combined_factor
    
    return final_factor
```
