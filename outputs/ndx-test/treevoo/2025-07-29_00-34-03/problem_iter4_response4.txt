```python
def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate the average close price over the last 4 days
    avg_close_price = df['close'].rolling(window=4).mean()
    
    # Calculate the current momentum
    current_momentum = df['close'] - avg_close_price
    
    # Adjust the momentum by intraday volatility
    adjusted_momentum = current_momentum / intraday_volatility
    
    return adjusted_momentum
```
