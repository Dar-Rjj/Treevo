```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum (5 days)
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum (20 days)
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume = df['volume'].rolling(window=5).mean()
    long_term_volume = df['volume'].rolling(window=20).mean()
    volume_trend = short_term_volume - long_term_volume
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    average_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_high_low_range = high_low_range * (df['volume'] / average_volume_20_days)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Combine Factors
    adjusted_high_low_range = scaled_high_low_range
    inverse_intraday_range = 1 / intraday_range
    previous_day_return = df['close'] / df['open'].shift(1) - 1
    combined_factors = (adjusted_high_low_range * inverse_intraday_range) + previous_day_return
    
    # Final Alpha Factor
    alpha_factor = (momentum_diff / volume_trend) + scaled_high_low_range + close_to_open_return + combined_factors
    
    return alpha_factor
```
