```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Identify Volume Shock
    df['volume_t-1'] = df['volume'].shift(1)
    volume_shock = (df['volume'] > 1.5 * df['volume_t-1']).astype(int)
    
    # Combine Intraday Momentum and Volume Shock
    factor_values = intraday_momentum * volume_shock
    
    return factor_values
```
