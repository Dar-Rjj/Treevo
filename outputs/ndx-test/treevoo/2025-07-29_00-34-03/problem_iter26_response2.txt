```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume information.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'volume' and index as 'date'.
        
    Returns:
        pd.Series: A Series representing the alpha factor values, indexed by 'date'.
    """
    # Calculate the average price for the day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the daily return
    daily_return = df['close'] / df['open'] - 1
    
    # Calculate the price change from open to close
    price_change = df['close'] - df['open']
    
    # Calculate the weighted average price change
    weighted_avg_price_change = (avg_price * df['volume']).diff(1)
    
    # Create an alpha factor
    alpha_factor = daily_return * weighted_avg_price_change + price_change
    
    return alpha_factor
```
