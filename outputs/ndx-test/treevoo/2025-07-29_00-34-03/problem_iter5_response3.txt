```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = df['volume'] * df['close']
    df['Volume_Weighted_Price_Sum'] = df['Volume_Weighted_Price'].rolling(window=20).sum()
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    df['SMA_Volume_Weighted_Price'] = df['Volume_Weighted_Price_Sum'].rolling(window=20).mean()
    
    # Volatility Adjustment Factor
    df['Log_Return'] = np.log(df['high']) - np.log(df['low'])
    df['Realized_Volatility'] = df['Log_Return'].rolling(window=20).std()
    df['Smoothed_Volatility'] = df['Realized_Volatility'].ewm(span=20, adjust=False).mean()
    df['SMA_Adjusted'] = df['SMA_Volume_Weighted_Price'] - df['Smoothed_Volatility']
    
    # Volume Surge Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Surge'] = (df['Volume_Change'] > 0) & (df['Volume_Change'] > df['volume'].quantile(0.9))
    df['SMA_Adjusted'] = df['SMA_Adjusted'] + df['SMA_Adjusted'] * df['Volume_Surge'] * 0.1
    
    # Adjusted High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['Scaled_Volume'] = df['volume'] / df['Average_Volume']
    df['Adjusted_High_Low_Range'] = df['High_Low_Range'] * df['Scaled_Volume']
    
    # Close-to-Open Return
    df['Close_to_Open_Return'] = df['close'] / df['open'].shift(1) - 1
    
    # Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) / df['low']
    df['Inverse_Intraday_Range'] = 1 / df['Intraday_Range']
    
    # Combine Factors
    df['Alpha_Factor'] = df['Adjusted_High_Low_Range'] * df['Inverse_Intraday_Range'] + df['Close_to_Open_Return']
    
    return df['Alpha_Factor']
```
