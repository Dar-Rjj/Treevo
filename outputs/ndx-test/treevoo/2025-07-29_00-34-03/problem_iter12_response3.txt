```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (21 days)
    df['avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (20 days)
    df['volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Scale by Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20']
    
    # Calculate Realized Volatility (sum of squared high-low ranges over last 10 days)
    df['realized_volatility'] = (df['high_low_range'].rolling(window=10).apply(lambda x: (x**2).sum())) ** 0.5
    
    # Combine High-Low Range, Scaled Volume, and Realized Volatility
    df['combined_factors'] = df['high_low_range'] * df['scaled_volume'] + df['realized_volatility']
    
    # Adjust Momentum by Combined Factors
    df['adjusted_momentum'] = df['avg_daily_return'] * df['combined_factors']
    
    # Weight by Volume
    df['weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    
    # Calculate Open-to-Close Momentum
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_close_momentum'] = df['open_close_return'].rolling(window=5).mean()
    
    # Combine Adjusted Momentum, Open-to-Close Momentum, and Volume
    df['preliminary_alpha'] = df['weighted_momentum'] + df['open_close_momentum']
    
    # Apply Price Filter
    df['avg_high_low_t'] = (df['high'] + df['low']) / 2
    df['avg_high_low_t-1'] = df['avg_high_low_t'].shift(1)
    df['avg_open_close_t'] = (df['open'] + df['close']) / 2
    df['avg_open_close_t-1'] = df['avg_open_close_t'].shift(1)
    
    price_filter = (df['avg_high_low_t'] > df['avg_high_low_t-1']) | (df['avg_open_close_t'] > df['avg_open_close_t-1'])
    df.loc[price_filter, 'final_alpha'] = df['preliminary_alpha'] * 1.1
    df.loc[~price_filter, 'final_alpha'] = df['preliminary_alpha'] * 0.9
    
    # Apply Volume Spike Filter
    volume_spike = df['volume'] > 2 * df['volume'].shift(1)
    df.loc[volume_spike, 'final_alpha'] = df['final_alpha'] * 1.1
    
    # Calculate Accumulated Volume Over Last 5 Days
    df['accumulated_volume_5'] = df['volume'].rolling(window=5).sum() / 5
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume'] - df['accumulated_volume_5']
    
    # Adjust Final Alpha Factor
    df.loc[df['volume_acceleration'] > 0, 'final_alpha'] = df['final_alpha'] * 1.1
    
    return df['final_alpha'].dropna()
```
