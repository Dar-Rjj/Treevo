```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    weighted_close_to_open_return = close_to_open_return * df['volume'] * df['amount']
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.rolling(window=5).sum()
    
    # Calculate Realized Volatility
    high_low_range = df['high'] - df['low']
    realized_volatility = (high_low_range ** 2).rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = close_to_open_return / realized_volatility
    
    # Combine Factors
    combined_factor = (
        intraday_range * scaled_volume +
        weighted_close_to_open_return +
        open_to_close_momentum +
        adjusted_momentum
    )
    
    # Final Weighting
    final_factor = combined_factor * df['volume']
    
    return final_factor
```
