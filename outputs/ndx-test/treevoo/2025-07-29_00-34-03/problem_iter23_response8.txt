```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in average price from the previous day
    price_momentum = avg_price.pct_change()
    
    # Calculate the ratio of volume to average price
    volume_ratio = df['volume'] / avg_price
    
    # Calculate the percentage change in volume from the previous day
    volume_momentum = df['volume'].pct_change()
    
    # Calculate the factor as a combination of price momentum, volume ratio, and volume momentum
    factor = price_momentum * volume_ratio + volume_momentum
    
    return factor
```
