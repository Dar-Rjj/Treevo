```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_trend = avg_volume_5 - avg_volume_20
    
    # Calculate Adjusted High-Low Range
    high_low_range = (df['high'] - df['low']) * (df['volume'] / avg_volume_20)
    
    # Calculate Previous Day's Return
    prev_open = df['open'].shift(1)
    current_close = df['close']
    prev_day_return = (current_close - prev_open) / prev_open
    
    # Adjust Momentum by Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std() * (252 ** 0.5)
    adjusted_momentum = prev_day_return / realized_volatility
    
    # Combine Factors
    inversed_intraday_range = 1 / (df['high'] - df['low'])
    combined_factors = (high_low_range * inversed_intraday_range) + prev_day_return + short_term_momentum + adjusted_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_diff / volume_trend) + high_low_range + combined_factors
    final_alpha_factor = final_alpha_factor * df['volume']
    
    return final_alpha_factor
```
