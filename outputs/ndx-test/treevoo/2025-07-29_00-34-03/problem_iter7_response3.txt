```python
def heuristics_v2(df):
    # Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Short-Term Momentum
    average_close_price = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - average_close_price
    
    # Adjust Momentum
    adjusted_momentum = current_momentum / intraday_volatility
    
    return adjusted_momentum
```
