import pandas as pd

def heuristics_v2(df, n_days=21, volatility_window=20):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate True Average Price
    df['true_avg_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Price Range Volatility
    df['price_range_volatility'] = df['daily_price_range'].rolling(window=n_days).std()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=n_days).mean()
    
    # Calculate Rolling Volatility of Close Prices
    df['rolling_close_volatility'] = df['close'].pct_change().rolling(window=volatility_window).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['rolling_avg_daily_return'] / df['rolling_close_volatility']
    
    # Weighting by Volume and Price Movement
    df['volume_price_movement_score'] = df['volume'] * (df['high'] - df['open'])
    df['weighted_daily_price_range'] = df['volume_price_movement_score'] * df['daily_price_range']
    
    # Combine Momentum and Intraday Dynamics
    df['adjusted_true_avg_price'] = df['true_avg_price'] / df['price_range_volatility']
    df['final_momentum_indicator'] = df['adjusted_true_avg_price'] * df['weighted_daily_price_range']
    
    # Integrate with Adjusted Momentum
    df['integrated_alpha_factor'] = df['adjusted_momentum'] * df['final_momentum_indicator']
    
    return df['integrated_alpha_factor']
