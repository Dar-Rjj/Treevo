```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'].rolling(window=4).mean().shift(1) - df['close']) / intraday_volatility
    
    # Smooth Realized Volatility
    log_returns_high_low = np.log(df['high']) - np.log(df['low'])
    realized_volatility = log_returns_high_low.rolling(window=20).std()
    smoothed_volatility = realized_volatility.ewm(span=10).mean()
    
    # Adjust Short-Term Momentum with Smoothed Volatility
    adjusted_momentum = short_term_momentum - smoothed_volatility
    
    # Volume Surge Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_surge = (volume_change > 1.5 * volume_change.rolling(window=20).std()).astype(int)
    
    # Combine Adjusted Momentum, Volume-Weighted Close-to-Open Return, and Volume Surge
    factor_value = adjusted_momentum + volume_weighted_return + volume_surge * 0.5
    
    return factor_value
```
