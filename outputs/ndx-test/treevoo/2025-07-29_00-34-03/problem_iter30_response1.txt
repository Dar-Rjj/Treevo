```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['weighted_close'] = df['close'] * df['volume']
    df['volume_weighted_price'] = df['weighted_close'].cumsum() / df['volume'].cumsum()
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Prices
    df['ema_volume_weighted_price'] = df['volume_weighted_price'].ewm(span=5, adjust=False).mean()
    
    # Intraday Volatility Adjustment Factor
    df['intraday_volatility'] = df['high'] - df['low']
    df['adjusted_ema'] = df['ema_volume_weighted_price'] + df['intraday_volatility']
    
    # Volume Spike Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_spike_threshold = 2.0  # Define a threshold for volume spike
    df['volume_spike'] = (df['volume_change'] > volume_spike_threshold).astype(int)
    
    # Adjust Final Alpha Factor
    df['alpha_factor'] = 0
    df.loc[df['volume_spike'] == 1, 'alpha_factor'] += 1
    
    # Short-Term and Long-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).sum()
    long_term_momentum = df['close'].rolling(window=20).sum()
    df['momentum_diff'] = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume_sum = df['volume'].rolling(window=5).sum()
    long_term_volume_sum = df['volume'].rolling(window=20).sum()
    df['volume_trend_diff'] = short_term_volume_sum - long_term_volume_sum
    
    # Calculate Adjusted High-Low Range
    total_volume_20_days = df['volume'].rolling(window=20).sum()
    df['scaled_high_low_range'] = (df['high'] - df['low']) * (df['volume'] / total_volume_20_days)
    
    # Calculate Previous Day's Return
    df['previous_day_return'] = df['close'] - df['open'].shift(1)
    
    # Adjust Momentum by Volatility
    df['realized_volatility'] = df['previous_day_return'].rolling(window=20).std()
    df['momentum_adjusted_by_volatility'] = df['previous_day_return'] / df['realized_volatility']
    
    # Combine Factors
    df['inverse_intraday_range'] = 1 / df['intraday_volatility']
    df['combined_factors'] = (df['scaled_high_low_range'] * df['inverse_intraday_range']) + df['previous_day_return'] + df['momentum_diff'] + df['momentum_adjusted_by_volatility']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum_diff'] / df['volume_trend_diff']) + df['scaled_high_low_range'] + df['combined_factors']
    
    # Introduce a Reversal Component
    df['reversal'] = df['high'].shift(1) - df['low'].shift(1)
    df['scaled_reversal'] = df['reversal'] * (df['volume'] / total_volume_20_days)
    df['final_alpha_factor'] -= df['scaled_reversal']
    
    return df['final_alpha_factor']
```
