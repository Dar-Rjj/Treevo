```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate True Average Price
    true_avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Price Range Volatility
    price_range_volatility = daily_price_range.rolling(window=N).std()
    
    # Volume-Price Movement Score
    volume_price_score = df['volume'] * (df['high'] - df['low'])
    
    # Adjusted True Average Price
    adjusted_true_avg_price = true_avg_price / price_range_volatility
    
    # Final Momentum Indicator
    final_momentum_indicator = adjusted_true_avg_price * volume_price_score
    
    return final_momentum_indicator
```
