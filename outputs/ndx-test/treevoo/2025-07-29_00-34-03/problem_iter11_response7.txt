```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Short-Term Momentum
    df['avg_close_4_days'] = df['close'].rolling(window=4).mean()
    df['current_momentum'] = df['close'] - df['avg_close_4_days']
    df['volatility_adjusted_momentum'] = df['current_momentum'] / df['intraday_volatility']
    
    # Calculate Volume-Weighted Prices
    df['weighted_close'] = df['volume'] * df['close']
    df['sma_weighted_close_10'] = df['weighted_close'].rolling(window=10).mean()
    
    # Calculate Realized Volatility
    df['log_returns'] = np.log(df['high'] / df['low'])
    df['realized_volatility'] = df['log_returns'].rolling(window=10).std()
    
    # Smooth Realized Volatility
    df['smoothed_volatility'] = df['realized_volatility'].ewm(span=10, adjust=False).mean()
    
    # Adjust SMA with Smoothed Volatility
    df['adjusted_sma'] = df['sma_weighted_close_10'] - df['smoothed_volatility']
    
    # Calculate Price Momentum
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    df['sma_close_60'] = df['close'].rolling(window=60).mean()
    df['price_momentum'] = df['sma_close_10'] - df['sma_close_60']
    
    # Volume Surge Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_surge_threshold = df['volume_change'].quantile(0.95)
    df['volume_surge'] = (df['volume_change'] > volume_surge_threshold).astype(int)
    
    # Adjust Final Alpha Factor
    df['alpha_factor'] = df['volatility_adjusted_momentum'] + df['adjusted_sma'] + df['price_momentum']
    df['alpha_factor'] += df['volume_surge'] * 0.5  # Increase alpha factor if volume surge is detected
    
    return df['alpha_factor']
```
