```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume-Weighted Close-to-Open Return
    df['volume_weighted_close_to_open'] = df['close_to_open_return'] * df['volume']
    
    # Volume-Weighted Price
    df['volume_weighted_price'] = df['close'] * df['volume']
    df['sum_volume_weighted_price'] = df['volume_weighted_price'].rolling(window=20).sum()
    
    # Simple Moving Average (SMA) of Volume-Weighted Prices
    df['sma_volume_weighted_price'] = df['sum_volume_weighted_price'] / df['volume'].rolling(window=20).sum()
    
    # Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust SMA with Intraday Volatility
    df['adjusted_sma'] = df['sma_volume_weighted_price'] - df['intraday_volatility']
    
    # Volume Surge Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    threshold = df['volume_change'].quantile(0.75)
    df['volume_surge'] = (df['volume_change'] > 0) & (df['volume_change'] > threshold)
    df['volume_surge_factor'] = df['volume_surge'].apply(lambda x: 1 if x else 0)
    
    # Short-Term Momentum
    df['short_term_momentum'] = df['close'] - df['close'].rolling(window=4).mean()
    
    # True Average Price
    df['true_average_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Difference between Short and Long-Term Momentum
    df['momentum_difference'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Price Range Volatility
    df['daily_price_range'] = df['high'] - df['low']
    df['price_range_volatility'] = df['daily_price_range'].rolling(window=20).std()
    
    # Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Rolling Average Daily Return
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Rolling Volatility of Close Prices
    df['rolling_volatility'] = df['close'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Volume-Price Movement Score
    df['volume_price_movement_score'] = df['volume'] * (df['high'] - df['open'])
    df['weighted_price_movement'] = df['volume_price_movement_score'] * df['daily_price_range']
    
    # Incorporate Volume Trend
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_trend'] = df['avg_volume_5d'] - df['avg_volume_20d']
    
    # Combine Momentum and Volume Trends
    df['combined_momentum'] = df['momentum_difference'] + df['volume_trend']
    
    # Adjusted True Average Price
    df['adjusted_true_average_price'] = df['true_average_price'] / (df['price_range_volatility'] + 1e-6)
    
    # Final Momentum Indicator
    df['final_momentum_indicator'] = df['adjusted_true_average_price'] * df['weighted_price_movement']
    
    # Integrate with Combined Momentum and Volume Trends
    df['integrated_alpha_factor'] = df['adjusted_momentum'] * df['final_momentum_indicator'] * df['volume_surge_factor']
    
    return df['integrated_alpha_factor']
```
