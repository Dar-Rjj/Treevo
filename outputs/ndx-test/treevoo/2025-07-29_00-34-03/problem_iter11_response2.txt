```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_diff = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    volume_trend = avg_vol_5d - avg_vol_20d
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    scaled_volume = df['volume'] / avg_vol_20d
    adjusted_high_low_range = high_low_range * scaled_volume
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Intraday Volatility
    intraday_volatility = intraday_range
    
    # Calculate Previous Day's Return
    previous_day_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Short-Term Momentum
    avg_close_4d = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - avg_close_4d
    adjusted_short_term_momentum = current_momentum / intraday_volatility
    
    # Calculate Realized Volatility
    high_low_ranges_10d = df['high'] - df['low']
    realized_volatility = high_low_ranges_10d.rolling(window=10).apply(lambda x: (x**2).sum(), raw=True)
    
    # Adjust Momentum by Volatility
    adjusted_momentum_by_volatility = previous_day_return / realized_volatility
    
    # Combine Factors
    combined_factors = (adjusted_high_low_range * (1 / intraday_range) + previous_day_return +
                        adjusted_short_term_momentum + adjusted_momentum_by_volatility)
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_diff / volume_trend + adjusted_high_low_range +
                          close_to_open_return + combined_factors) * df['volume']
    
    return final_alpha_factor
```
