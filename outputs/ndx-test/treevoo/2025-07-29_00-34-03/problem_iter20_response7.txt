import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    prev_close = df['close'].shift(1)
    prev_day_return = (df['close'] - prev_close) / prev_close
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Rolling Average Daily Return (21 days)
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (20 days)
    rolling_volatility = daily_return.rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    momentum_adjustment = rolling_avg_daily_return / rolling_volatility
    
    # Inverse of Intraday Range
    inv_intraday_range = 1 / intraday_range
    
    # Combine Factors
    combined_factor = prev_day_return * inv_intraday_range * momentum_adjustment
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Weight by Volume, Amount, and Close-to-Open Return
    final_factor = combined_factor * volume_change * close_to_open_return * df['amount']
    
    # Volume-Weighted Price
    volume_weighted_price = (df['volume'] * df['close']).sum() / df['volume'].sum()
    
    # Scale by Volume
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20_days
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    adjusted_volume_momentum = volume_momentum / rolling_volatility
    
    # Integrate Volume Momentum
    final_factor += 0.5 * adjusted_volume_momentum
    
    # Incorporate High and Low Prices
    high_low_spread = df['high'] - df['low']
    weighted_return = final_factor * high_low_spread
    
    # Integrate Momentum
    sma_10 = df['close'].rolling(window=10).mean()
    momentum = df['close'] - sma_10
    final_factor += momentum
    
    # Incorporate Volatility
    std_20 = df['close'].rolling(window=20).std()
    final_factor /= std_20
    
    # Final Volume and Volatility Adjustment
    final_factor *= volume_change * combined_factor
    significant_volume_surge = (volume_change > 0) & (volume_change > df['volume'].shift(1) * 0.1)
    final_factor[significant_volume_surge] *= 1.5
    
    return final_factor
