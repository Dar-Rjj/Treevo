```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    prev_day_close = df['close'].shift(1)
    prev_day_return = df['close'] / prev_day_close - 1
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (21 days)
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (20 days)
    rolling_volatility = daily_return.rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    momentum_adjusted = rolling_avg_daily_return / rolling_volatility
    
    # Inverse of Intraday Range
    inv_intraday_range = 1 / intraday_range
    
    # Momentum Adjustment
    momentum_adj_by_range = prev_day_return * inv_intraday_range
    
    # Calculate Volume Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Combine Factors
    combined_factor = momentum_adj_by_range * volume_change
    
    # Calculate Volume Relative to Moving Average
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    volume_relative_to_ma = df['volume'] / rolling_avg_volume
    
    # Final Factor
    final_factor = combined_factor * volume_relative_to_ma
    
    return final_factor
```
