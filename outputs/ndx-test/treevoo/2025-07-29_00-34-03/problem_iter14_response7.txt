```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Rolling Average Daily Return (21 days)
    df['rolling_avg_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (20 days)
    df['rolling_volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Scale by Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20']
    
    # Calculate Realized Volatility (sum of squared high-low ranges over last 10 days)
    df['realized_volatility'] = (df['high'] - df['low']).pow(2).rolling(window=10).sum()
    
    # Combine Intraday Range, Scaled Volume, and Realized Volatility
    df['combined_factor'] = (1 / df['intraday_range']) * df['scaled_volume'] + df['realized_volatility']
    
    # Adjust Momentum by Combined Factors
    df['adjusted_momentum'] = df['rolling_avg_return'] * df['combined_factor']
    
    # Weight by Volume
    df['weighted_momentum'] = df['adjusted_momentum'] * df['volume']
    
    # Calculate Open-to-Close Momentum
    df['open_to_close_return'] = (df['close'] / df['open']) - 1
    df['open_to_close_momentum'] = df['open_to_close_return'].rolling(window=5).sum()
    
    # Combine Adjusted Momentum, Open-to-Close Momentum, and Volume
    df['prelim_alpha'] = df['weighted_momentum'] + df['open_to_close_momentum']
    
    # Apply Price Filter
    df['price_filter_t'] = (df['high'] + df['low']) / 2
    df['price_filter_t-1'] = df['price_filter_t'].shift(1)
    df['price_filter_openclose_t'] = (df['open'] + df['close']) / 2
    df['price_filter_openclose_t-1'] = df['price_filter_openclose_t'].shift(1)
    df['price_filter'] = (df['price_filter_t'] > df['price_filter_t-1']) | (df['price_filter_openclose_t'] > df['price_filter_openclose_t-1'])
    df['prelim_alpha'] = df['prelim_alpha'] * (1 + df['price_filter'].astype(int) * 0.01)
    
    # Apply Volume Spike Filter
    df['volume_spike'] = (df['volume'] > 2 * df['volume'].shift(1))
    df['prelim_alpha'] = df['prelim_alpha'] * (1 + df['volume_spike'].astype(int) * 0.01)
    
    # Calculate Accumulated Volume Over Last 5 Days
    df['acc_volume_5'] = df['volume'].rolling(window=5).sum() / 5
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume'] - df['acc_volume_5']
    
    # Adjust Final Alpha Factor
    df['final_alpha'] = df['prelim_alpha'] * (1 + (df['volume_acceleration'] > 0).astype(int) * 0.01)
    
    return df['final_alpha']
```
