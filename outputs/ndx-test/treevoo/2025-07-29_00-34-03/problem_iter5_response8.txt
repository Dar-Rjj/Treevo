```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    daily_return = df['close'].pct_change()

    # Calculate Rolling Average Daily Return (e.g., 21 days)
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()

    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']

    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = rolling_avg_daily_return / intraday_volatility

    # Weight by Volume
    volume_weighted_momentum = adjusted_momentum * df['volume']

    # Calculate Short-Term Momentum
    average_close_price = df['close'].rolling(window=4).mean()
    current_momentum = df['close'] - average_close_price

    # Finalize Alpha Factor
    final_factor = volume_weighted_momentum + current_momentum

    # Apply Price Filter
    avg_high_low_current = (df['high'] + df['low']) / 2
    avg_high_low_previous = (df['high'].shift(1) + df['low'].shift(1)) / 2
    price_filter = (avg_high_low_current > avg_high_low_previous).astype(int)

    final_factor = final_factor * (1 + price_filter * 0.1)  # Increase factor value if condition is met, else decrease

    return final_factor
```
