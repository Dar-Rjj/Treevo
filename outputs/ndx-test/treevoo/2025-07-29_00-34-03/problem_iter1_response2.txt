```python
def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Short-Term Momentum
    df['avg_close_4_days'] = (df['close'].shift(1) + df['close'].shift(2) + df['close'].shift(3) + df['close'].shift(4)) / 4
    df['current_momentum'] = df['close'] - df['avg_close_4_days']
    
    # Adjust Momentum by Intraday Volatility
    df['adjusted_momentum'] = df['current_momentum'] / df['intraday_volatility']
    
    return df['adjusted_momentum']
```
