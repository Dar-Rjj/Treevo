```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    sma_volume_weighted = df['Volume_Weighted_Price'].rolling(window=20).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Calculate Previous Day's Return
    previous_day_return = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Open-to-Close Momentum
    open_to_close_return = df['close'] / df['open'] - 1
    momentum_5_days = open_to_close_return.rolling(window=5).sum()
    
    # Volatility Adjustment Factor
    log_returns = (df['high'] - df['low']) / df['low']
    realized_volatility = log_returns.rolling(window=20).std()
    smoothed_volatility = realized_volatility.ewm(span=20).mean()
    adjusted_sma = sma_volume_weighted - smoothed_volatility
    
    # Volume Surge Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_surge = (volume_change > 0) & (volume_change > volume_change.rolling(window=20).mean())
    alpha_factor = adjusted_sma
    alpha_factor[volume_surge] += 1  # Increase Alpha Factor if Volume Surge is detected
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume and Amount
    weighted_close_to_open = close_to_open_return * df['volume'] * df['amount']
    
    # Adjust Momentum by Volatility
    adjusted_momentum = close_to_open_return / (realized_volatility + 1e-6)
    
    # Combine Factors
    combined_factor = (intraday_range * scaled_volume + 
                       weighted_close_to_open + 
                       momentum_5_days + 
                       adjusted_momentum)
    
    # Final Weighting
    final_alpha = combined_factor * df['volume']
    
    return final_alpha
```
