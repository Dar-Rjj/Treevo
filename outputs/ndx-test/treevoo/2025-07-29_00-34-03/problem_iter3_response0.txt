```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / avg_volume_20
    
    # Calculate Open-to-Close Momentum
    open_close_return = (df['close'] - df['open']) / df['open']
    open_close_momentum = open_close_return.rolling(window=5).sum()
    
    # Calculate Realized Volatility
    hl_range_sq = (df['high'] - df['low']) ** 2
    realized_volatility = hl_range_sq.rolling(window=10).sum()
    
    # Combine High-Low Range, Open-to-Close Momentum, and Realized Volatility
    factor = (high_low_range * scaled_volume) + open_close_momentum + realized_volatility
    
    return factor
```
