```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Weighted Close Price
    df['weighted_close'] = df['close'] * df['volume']
    
    # Calculate Adjusted SMA of Weighted Close Prices
    sma_window = 10
    df['sma_weighted_close'] = df['weighted_close'].rolling(window=sma_window).mean()
    
    # Smooth Realized Volatility
    log_returns = np.log(df['high'] / df['low'])
    realized_volatility = log_returns.rolling(window=sma_window).std()
    df['ema_realized_volatility'] = realized_volatility.ewm(span=20, adjust=False).mean()
    
    # Subtract Smoothed Volatility from SMA
    df['adjusted_sma'] = df['sma_weighted_close'] - df['ema_realized_volatility']
    
    # Calculate Short-Term Price Momentum
    short_momentum_window = 5
    df['short_term_momentum'] = df['close'].rolling(window=short_momentum_window).mean()
    
    # Calculate Medium-Term Price Momentum
    medium_momentum_window = 10
    df['medium_term_momentum'] = df['close'].rolling(window=medium_momentum_window).mean()
    
    # Calculate Long-Term Price Momentum
    long_momentum_window = 20
    df['long_term_momentum'] = df['close'].rolling(window=long_momentum_window).mean()
    
    # Determine the Difference between Short, Medium, and Long-Term Momentum
    df['short_medium_diff'] = df['short_term_momentum'] - df['medium_term_momentum']
    df['medium_long_diff'] = df['medium_term_momentum'] - df['long_term_momentum']
    
    # Incorporate Volume Trend
    volume_short_window = 5
    volume_long_window = 20
    df['avg_volume_short'] = df['volume'].rolling(window=volume_short_window).mean()
    df['avg_volume_long'] = df['volume'].rolling(window=volume_long_window).mean()
    df['volume_trend'] = df['avg_volume_short'] - df['avg_volume_long']
    
    # Combine Momentum and Volume-Weighted Return
    df['combined_momentum'] = df['short_medium_diff'] + df['medium_long_diff']
    df['final_alpha_factor'] = df['adjusted_sma'] + df['combined_momentum'] + df['volume_trend']
    
    return df['final_alpha_factor']
```
