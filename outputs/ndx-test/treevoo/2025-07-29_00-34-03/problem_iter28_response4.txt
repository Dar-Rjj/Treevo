```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['Volume_Weighted_Price'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Intraday Volatility Adjustment
    intraday_volatility = df['high'] - df['low']
    df['SMA_Volatility_Adjusted'] = df['close'].rolling(window=5).mean() - intraday_volatility

    # Volume Surge Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    volume_surge_threshold = df['volume'].quantile(0.9)
    df['Volume_Surge'] = (df['Volume_Change'] > 0) & (df['Volume_Change'] > volume_surge_threshold)

    # Short-Term Momentum
    df['Short_Term_Momentum'] = df['close'] - df['close'].rolling(window=4).mean()

    # True Average Price
    df['True_Average_Price'] = (df['high'] + df['low'] + df['close']) / 3

    # Long-Term Price Momentum
    df['Long_Term_Momentum'] = df['close'].rolling(window=20).mean()
    df['Momentum_Diff'] = df['Short_Term_Momentum'] - df['Long_Term_Momentum']

    # Rolling Average Daily Return and Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Rolling_Avg_Return'] = df['Daily_Return'].rolling(window=21).mean()
    df['Rolling_Volatility'] = df['close'].rolling(window=20).std()

    # Adjust Momentum by Volatility
    df['Momentum_Vol_Adjusted'] = df['Rolling_Avg_Return'] / df['Rolling_Volatility']

    # Volume-Price Movement Score
    df['Price_Range'] = df['high'] - df['open']
    df['Volume_Price_Score'] = df['volume'] * df['Price_Range']

    # Volume Trend
    df['Volume_5_Day_Avg'] = df['volume'].rolling(window=5).mean()
    df['Volume_20_Day_Avg'] = df['volume'].rolling(window=20).mean()
    df['Volume_Trend'] = df['Volume_5_Day_Avg'] - df['Volume_20_Day_Avg']

    # Combined Momentum and Volume Trends
    df['Combined_Trend'] = df['Momentum_Diff'] + df['Volume_Trend']

    # Adjusted True Average Price
    df['Adjusted_True_Average_Price'] = df['True_Average_Price'] / df['Price_Range']

    # Final Momentum Indicator
    df['Final_Momentum'] = df['Adjusted_True_Average_Price'] * df['Volume_Price_Score']

    # Enhanced Momentum-Volatility Interaction
    df['EMA_Close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Enhanced_EMA'] = df['EMA_Close'] / df['Rolling_Volatility']

    # Volume-Adjusted Momentum
    df['Cumulative_Volume_Weighted_Price'] = df['Volume_Weighted_Price'].rolling(window=20).sum()
    df['Volume_Adjusted_Momentum'] = df['Cumulative_Volume_Weighted_Price'].diff()
    df['Volume_Adjusted_Momentum'] += df['Volume_Surge'] * df['Volume_Adjusted_Momentum'].std()

    # Integrated Alpha Factor
    df['Integrated_Alpha'] = df['Enhanced_EMA'] * df['Volume_Adjusted_Momentum']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Integrated_Alpha'] * df['SMA_Volatility_Adjusted']

    return df['Final_Alpha_Factor']
```
