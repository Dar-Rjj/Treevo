```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    df['rolling_avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (e.g., 20 days)
    df['rolling_volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['momentum_adjusted'] = df['rolling_avg_daily_return'] / df['rolling_volatility']
    
    # Inverse of Intraday Range
    df['inv_intraday_range'] = 1 / df['intraday_range']
    
    # Combine Factors
    df['combined_factor'] = df['prev_day_return'] * df['inv_intraday_range'] * df['momentum_adjusted']
    
    # Weight by Volume
    df['alpha_factor'] = df['combined_factor'] * df['volume']
    
    return df['alpha_factor'].dropna()
```
