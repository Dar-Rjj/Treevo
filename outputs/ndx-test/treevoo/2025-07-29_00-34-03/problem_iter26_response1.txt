import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    prev_close = df['close'].shift(1)
    prev_return = (df['close'] - prev_close) / prev_close
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Rolling Average Daily Return (21 days)
    rolling_avg_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns (20 days)
    rolling_volatility = daily_return.rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_avg_return / rolling_volatility
    
    # Combine Factors
    inv_intraday_range = 1 / intraday_range
    momentum_adjustment = prev_return * inv_intraday_range
    
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_c2o_return = close_to_open_return * df['volume']
    
    # Incorporate High-Low Spread
    high_low_spread = df['high'] - df['low']
    adjusted_weighted_return = volume_weighted_c2o_return / high_low_spread
    
    # Integrate Momentum
    sma_10 = df['close'].rolling(window=10).mean()
    close_sma_diff = df['close'] - sma_10
    integrated_momentum = close_sma_diff + adjusted_momentum
    
    # Incorporate Volatility
    std_20 = df['close'].rolling(window=20).std()
    intermediate_alpha_factor = integrated_momentum / std_20
    
    # Weight by Volume, Amount, and Close-to-Open Return
    volume_change = df['volume'] - df['volume'].shift(1)
    weighted_intermediate_factor = intermediate_alpha_factor * volume_change * df['amount'] * close_to_open_return
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    adjusted_volume_momentum = volume_momentum / rolling_volatility
    
    # Integrate Volume Momentum
    weighted_intermediate_factor += 0.5 * adjusted_volume_momentum
    
    # Calculate Volume-Weighted Intraday Range
    volume_weighted_intraday_range = intraday_range * df['volume']
    
    # Incorporate Volume-Weighted Intraday Range
    final_alpha_factor = weighted_intermediate_factor + 0.3 * volume_weighted_intraday_range
    
    return final_alpha_factor
