```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    weighted_close = df['close'] * df['volume']
    volume_weighted_price = weighted_close.sum() / df['volume'].sum()
    
    # Calculate Simple Moving Average (SMA) of Volume-Weighted Prices
    sma_volume_weighted_price = df['close'].rolling(window=20).mean() * (df['volume'] / df['volume'].rolling(window=20).sum())
    
    # Intraday Volatility Adjustment Factor
    intraday_volatility = df['high'] - df['low']
    adjusted_sma = sma_volume_weighted_price - intraday_volatility
    
    # Volume Surge Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_surge_threshold = 1.5 * df['volume_change'].std()
    volume_surge_filter = (df['volume_change'] > volume_surge_threshold).astype(int)
    adjusted_alpha_factor = adjusted_sma + (volume_surge_filter * df['volume_change'])
    
    # Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].rolling(window=4).mean()
    
    # True Average Price
    true_avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).mean()
    
    # Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Price Range Volatility
    price_range_volatility = df[['high', 'low', 'close']].rolling(window=20).std().mean(axis=1)
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices
    rolling_volatility = df['close'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_avg_daily_return / rolling_volatility
    
    # Weighting by Volume and Price Movement
    volume_price_movement_score = (df['volume'] * (df['high'] - df['open'])) / df['volume'].rolling(window=20).mean()
    daily_price_range = (df['high'] - df['low']).rolling(window=20).std()
    weighted_score = volume_price_movement_score * daily_price_range
    
    # Incorporate Volume Trend
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_trend = avg_volume_5d - avg_volume_20d
    
    # Combine Momentum and Volume Trends
    combined_trend = momentum_difference + volume_trend
    
    # Adjusted True Average Price
    adjusted_true_avg_price = true_avg_price / price_range_volatility
    
    # Final Momentum Indicator
    final_momentum_indicator = adjusted_true_avg_price * weighted_score
    
    # Integrate with Combined Momentum and Volume Trends
    alpha_factor = adjusted_alpha_factor * final_momentum_indicator * combined_trend
    
    return alpha_factor
```
