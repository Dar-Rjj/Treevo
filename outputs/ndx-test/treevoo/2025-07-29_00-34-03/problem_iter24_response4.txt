```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'].shift(1) - df['open']

    # Weight by Volume
    df['volume_weighted_return'] = df['volume'] * df['open_to_close_return']

    # Scale by Volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20']

    # Adjust Intraday Range by Scaled Volume
    df['adjusted_intraday_range'] = df['intraday_range'] * df['scaled_volume']

    # Short-Term Price Momentum (last 5 days)
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()

    # Long-Term Price Momentum (last 20 days)
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()

    # Difference between Short and Long-Term Momentum
    df['momentum_diff'] = df['short_term_momentum'] - df['long_term_momentum']

    # Realized Volatility (sum of squared high-low ranges over last 10 days)
    df['realized_volatility'] = (df['high'] - df['low']).pow(2).rolling(window=10).sum()

    # Adjust Short-Term Momentum by Volatility
    df['adjusted_short_term_momentum'] = df['short_term_momentum'] / df['realized_volatility']

    # Combine Factors
    df['combined_factor'] = (df['adjusted_intraday_range'] 
                             + df['volume_weighted_return'] 
                             + df['adjusted_short_term_momentum'] 
                             + df['momentum_diff'])

    # Final Weighting
    df['final_factor'] = df['volume'] * df['combined_factor']

    return df['final_factor']
```
