```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume and Amount
    weighted_return = close_to_open_return * df['volume'] * df['amount']
    
    # Calculate Price Range
    price_range = df['high'] - df['low']
    
    # Adjust by Price Volatility
    final_factor = weighted_return / price_range
    
    return final_factor
```
