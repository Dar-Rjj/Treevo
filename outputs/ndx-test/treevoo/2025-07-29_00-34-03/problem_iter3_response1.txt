```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['DailyLogReturn'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Volume Change
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Momentum and Volume
    df['PreliminaryFactor'] = df['DailyLogReturn'] * df['VolumeChange']
    
    # Apply Price Filter
    df['AverageOpenClose_t'] = (df['open'] + df['close']) / 2
    df['AverageOpenClose_t-1'] = df['AverageOpenClose_t'].shift(1)
    
    # Finalize Alpha Factor
    df['FinalFactor'] = df['PreliminaryFactor'].copy()
    df.loc[df['AverageOpenClose_t'] > df['AverageOpenClose_t-1'], 'FinalFactor'] += 1
    df.loc[df['AverageOpenClose_t'] <= df['AverageOpenClose_t-1'], 'FinalFactor'] -= 1
    
    # Apply Volume Spike Filter
    df.loc[df['volume'] > 2 * df['volume'].shift(1), 'FinalFactor'] += 1
    
    return df['FinalFactor']
```
