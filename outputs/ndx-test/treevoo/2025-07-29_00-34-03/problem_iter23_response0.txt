```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the average of high and low prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the average of high and low and the close price
    df['diff_avg_close'] = df['avg_high_low'] - df['close']
    
    # Calculate the rolling sum of the volume over the last 5 days
    df['rolling_volume_sum_5d'] = df['volume'].rolling(window=5).sum()
    
    # Calculate the factor as a combination of the above features
    df['factor'] = (df['close_pct_change'] * df['diff_avg_close']) / df['rolling_volume_sum_5d']
    
    # Return the factor as a pandas Series
    return df['factor']
```
