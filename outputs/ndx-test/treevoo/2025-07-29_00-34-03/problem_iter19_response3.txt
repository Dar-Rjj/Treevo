```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Scale by Volume
    average_volume_20_days = df['volume'].rolling(window=20).mean()
    scaled_volume = df['volume'] / average_volume_20_days
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    weighted_close_to_open_return = (scaled_volume * df['amount'] * close_to_open_return)
    
    # Calculate Open-to-Close Momentum
    open_to_close_momentum = (df['close'] - df['open']) / df['open']
    previous_5_days_open_to_close = df['close'].shift(1) - df['open'].shift(1)
    for i in range(2, 6):
        previous_5_days_open_to_close += (df['close'].shift(i) - df['open'].shift(i))
    open_to_close_momentum += previous_5_days_open_to_close / 5
    
    # Calculate Realized Volatility
    high_low_range_squared = (df['high'] - df['low']).pow(2)
    realized_volatility = high_low_range_squared.rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    rolling_average_daily_return = df['close'].pct_change().rolling(window=21).mean()
    rolling_volatility_daily_returns = df['close'].pct_change().rolling(window=20).std()
    adjusted_momentum_by_volatility = (rolling_average_daily_return / rolling_volatility_daily_returns) + realized_volatility
    
    # Combine Factors
    combined_factor = (intraday_range * scaled_volume) + weighted_close_to_open_return + open_to_close_momentum + adjusted_momentum_by_volatility
    
    # Final Factor
    final_factor = combined_factor * (df['volume'] / average_volume_20_days)
    
    return final_factor
```
