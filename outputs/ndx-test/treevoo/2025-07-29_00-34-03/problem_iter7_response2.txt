```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_price_momentum = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_price_momentum = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Price Momentum
    price_momentum_diff = short_term_price_momentum - long_term_price_momentum
    
    # Incorporate Volume Trend
    # Calculate Average Volume over the last 5 days
    short_term_volume_avg = df['volume'].rolling(window=5).mean()
    
    # Calculate Average Volume over the last 20 days
    long_term_volume_avg = df['volume'].rolling(window=20).mean()
    
    # Compute the difference between short and long-term volume averages
    volume_trend_diff = short_term_volume_avg - long_term_volume_avg
    
    # Combine Momentum and Volume Trends
    combined_trends = price_momentum_diff + volume_trend_diff
    
    # Final Alpha Factor
    alpha_factor = combined_trends
    
    return alpha_factor
```
