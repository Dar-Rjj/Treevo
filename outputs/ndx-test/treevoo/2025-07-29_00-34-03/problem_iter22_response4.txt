```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Scale by Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / avg_volume_20
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    df['weighted_close_to_open_return'] = df['volume'] * df['amount'] * df['close_to_open_return']
    
    # Calculate Open-to-Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_momentum'] = df['open_to_close_return'].rolling(window=5).sum()
    
    # Calculate Realized Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['realized_volatility'] = df['high_low_range'].rolling(window=10).apply(lambda x: (x**2).sum())
    
    # Adjust Momentum by Volatility
    daily_return = df['close'].pct_change()
    rolling_volatility = daily_return.rolling(window=20).std()
    df['adjusted_momentum'] = daily_return.rolling(window=21).sum() / (rolling_volatility + 1e-8) * df['realized_volatility']
    
    # Combine Factors
    df['combined_factor'] = (
        df['intraday_range'] * df['scaled_volume'] +
        df['weighted_close_to_open_return'] +
        df['open_to_close_momentum'] +
        df['adjusted_momentum']
    )
    
    # Final Factor
    df['final_factor'] = df['combined_factor'] * (df['volume'] / avg_volume_20)
    
    return df['final_factor']
```
