```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Incorporate High and Low Prices
    high_low_spread = df['high'] - df['low']
    adjusted_volume_weighted_return = (volume_weighted_return + high_low_spread) / 2  # Simple averaging for combination
    
    # Integrate Momentum
    momentum = df['close'] - df['close'].rolling(window=10).mean()
    momentum_adjusted_factor = adjusted_volume_weighted_return + momentum
    
    # Incorporate Volatility
    volatility = df['close'].rolling(window=20).std()
    final_alpha_factor = momentum_adjusted_factor / volatility
    
    return final_alpha_factor
```
