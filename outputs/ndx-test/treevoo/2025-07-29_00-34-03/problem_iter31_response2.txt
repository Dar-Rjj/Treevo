```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Volume-Weighted Close-to-Open Return
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Adjust Weighted Return with High-Low Spread
    adjusted_return = volume_weighted_return / (1 + high_low_spread)
    
    # Calculate 10-Day Exponential Moving Average of Close Price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Integrate Momentum: Subtract 10-Day EMA from Current Close Price
    momentum_adjustment = df['close'] - ema_10
    momentum_adjusted_return = adjusted_return + momentum_adjustment
    
    # Calculate 20-Day Standard Deviation of Log Returns
    log_returns = np.log(df['close']).diff()
    std_dev_20 = log_returns.rolling(window=20).std()
    
    # Incorporate Volatility: Divide Adjusted Return by 20-Day Standard Deviation
    final_alpha_factor = momentum_adjusted_return / std_dev_20
    
    return final_alpha_factor
```
