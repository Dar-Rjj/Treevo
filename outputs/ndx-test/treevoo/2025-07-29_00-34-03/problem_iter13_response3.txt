```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate True Average Price
    df['true_avg_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    df['momentum_diff'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Price Range Volatility
    df['price_range_volatility'] = df['price_range'].rolling(window=20).std()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['avg_daily_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices
    df['rolling_volatility'] = df['close'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['avg_daily_return'] / df['rolling_volatility']
    
    # Weighting by Volume and Price Movement
    df['volume_price_movement_score'] = df['volume'] * (df['high'] - df['open'])
    
    # Incorporate Volume Trend
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['volume_trend_diff'] = df['avg_volume_5_days'] - df['avg_volume_20_days']
    
    # Combine Momentum and Volume Trends
    df['combined_momentum_volume'] = df['momentum_diff'] + df['volume_trend_diff']
    
    # Adjusted True Average Price
    df['adjusted_true_avg_price'] = df['true_avg_price'] / (df['price_range_volatility'] + 1e-6)
    
    # Final Momentum Indicator
    df['final_momentum_indicator'] = df['adjusted_true_avg_price'] * df['volume_price_movement_score']
    
    # Integrate with Combined Momentum and Volume Trends
    df['alpha_factor'] = df['combined_momentum_volume'] * df['final_momentum_indicator']
    
    return df['alpha_factor']
```
