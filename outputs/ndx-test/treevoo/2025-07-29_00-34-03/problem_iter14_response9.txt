```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Return
    df['prev_day_return'] = df['close'].pct_change()
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return
    df['rolling_avg_return'] = df['daily_return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Daily Returns
    df['rolling_volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['momentum_adjusted'] = df['rolling_avg_return'] / df['rolling_volatility']
    
    # Combine Factors: Inverse of Intraday Range and Momentum Adjustment
    df['combined_factor'] = (1 / df['intraday_range']) * df['prev_day_return']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Volume Momentum
    df['volume_momentum'] = df['volume'] / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['vol_adj_volume_momentum'] = df['volume_momentum'] / df['rolling_volatility']
    
    # Integrate Volume Momentum
    weight = 0.5
    df['final_factor'] = df['combined_factor'] * df['volume_change'] * df['close_to_open_return'] * df['amount']
    df['final_factor'] += weight * df['vol_adj_volume_momentum']
    
    return df['final_factor']
```
