import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume and Amount
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume'] * df['amount']
    
    # Adjust Weighted Return Using High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['adjusted_weighted_return'] = df['weighted_close_to_open_return'] + df['high_low_spread']
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'].rolling(window=20).mean()
    
    # Determine the Difference between Short and Long-Term Momentum
    df['momentum_diff'] = df['short_term_momentum'] - df['long_term_momentm']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Momentum
    df['open_to_close_momentum'] = df['close'] - df['open']
    df['avg_open_to_close_momentum'] = df['open_to_close_momentum'].rolling(window=5).mean()
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['high'] - df['low']
    df['realized_volatility_10_days'] = df['realized_volatility'].rolling(window=10).sum()
    
    # Adjust Momentum by Volatility
    df['daily_return_21_days'] = df['close'].pct_change().rolling(window=21).mean()
    df['rolling_volatility_20_days'] = df['close'].pct_change().rolling(window=20).std() * 252**0.5
    df['adjusted_momentum'] = df['daily_return_21_days'] / (df['rolling_volatility_20_days'] + df['realized_volatility_10_days'])
    
    # Scale by Volume
    df['avg_volume_20_days'] = df['volume'].rolling(window=20).mean()
    df['scaled_volume'] = df['volume'] / df['avg_volume_20_days']
    
    # Combine Factors
    df['adjusted_high_low_range'] = df['high_low_spread'] * df['scaled_volume']
    df['inverse_intraday_range'] = 1 / df['intraday_range']
    df['momentum_adjustment'] = df['previous_day_return'] * (df['adjusted_high_low_range'] * df['inverse_intraday_range'] + df['previous_day_return'])
    df['combined_factors'] = df['adjusted_high_low_range'] + df['inverse_intraday_range'] + df['short_term_momentum'] * df['scaled_volume'] + df['weighted_close_to_open_return'] + df['avg_open_to_close_momentum'] + df['adjusted_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum_diff'] / df['momentum_diff_volume_avg']) + df['scaled_high_low_range'] + df['weighted_close_to_open_return'] + df['combined_factors']
    
    # Weight by Volume
    df['alpha_factor'] = df['final_alpha_factor'] * df['volume']
    
    return df['alpha_factor']
