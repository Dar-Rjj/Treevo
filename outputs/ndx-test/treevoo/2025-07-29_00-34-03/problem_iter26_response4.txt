import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price
    df['weighted_close'] = df['volume'] * df['close']
    volume_weighted_price = df['weighted_close'].sum() / df['volume'].sum()
    
    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Prices
    ema_volume_weighted_price = df['weighted_close'].ewm(span=20, adjust=False).mean()
    
    # Intraday Volatility Adjustment Factor
    df['intraday_volatility'] = df['high'] - df['low']
    adjusted_ema = ema_volume_weighted_price + df['intraday_volatility']
    
    # Volume Spike Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    volume_spike_threshold = 2.0  # Example threshold
    df['volume_spike'] = (df['volume_change'] > volume_spike_threshold).astype(int)
    adjusted_ema = adjusted_ema + df['volume_spike']
    
    # Short-Term Momentum
    short_term_momentum = df['close'].rolling(window=5).sum()
    
    # Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=20).sum()
    
    # Determine the Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume_sum = df['volume'].rolling(window=5).sum()
    long_term_volume_sum = df['volume'].rolling(window=20).sum()
    volume_trend_difference = short_term_volume_sum - long_term_volume_sum
    
    # Calculate Adjusted High-Low Range
    volume_scaled_high_low_range = (df['high'] - df['low']) * (df['volume'] / long_term_volume_sum)
    
    # Calculate Previous Day's Return
    df['previous_day_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Adjust Momentum by Volatility
    adjusted_momentum_by_volatility = df['previous_day_return'] / df['intraday_volatility']
    
    # Combine Factors
    combined_factors = (
        (volume_scaled_high_low_range * (1 / df['intraday_volatility'])) 
        + df['previous_day_return'] 
        + short_term_momentum 
        + adjusted_momentum_by_volatility
    )
    
    # Final Alpha Factor
    final_alpha_factor = (
        momentum_difference / volume_trend_difference 
        + volume_scaled_high_low_range 
        + combined_factors 
        * df['volume']
    )
    
    return final_alpha_factor
