```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_weighted_return = close_to_open_return * df['volume']

    # Calculate Volume-Weighted Price
    weighted_close_price = df['close'] * df['volume']
    sum_weighted_close_prices = weighted_close_price.rolling(window=20).sum()
    sum_volume = df['volume'].rolling(window=20).sum()
    volume_weighted_price = sum_weighted_close_prices / sum_volume

    # Calculate Exponential Moving Average (EMA) of Volume-Weighted Prices
    ema_volume_weighted_price = volume_weighted_price.ewm(span=20, adjust=False).mean()

    # Intraday Volatility Adjustment Factor
    intraday_volatility = df['high'] - df['low']
    adjusted_ema = ema_volume_weighted_price + intraday_volatility

    # Volume Spike Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0) & (volume_change > 3 * df['volume'].rolling(window=20).std())
    adjusted_alpha_factor = adjusted_ema + volume_spike * intraday_volatility

    # Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].rolling(window=4).mean()

    # True Average Price
    true_avg_price = (df['high'] + df['low'] + df['close']) / 3

    # Long-Term Price Momentum
    long_term_momentum = df['close'].ewm(span=20, adjust=False).mean()

    # Difference between Short and Long-Term Momentum
    momentum_difference = short_term_momentum - long_term_momentum

    # Price Range Volatility
    daily_price_range = df['high'] - df['low']
    price_range_volatility = daily_price_range.rolling(window=20).std()

    # Daily Return
    daily_return = df['close'].pct_change()

    # Rolling Average Daily Return
    rolling_avg_daily_return = daily_return.rolling(window=21).mean()

    # Rolling Volatility of Close Prices
    rolling_volatility = df['close'].rolling(window=20).std()

    # Adjust Momentum by Volatility
    adjusted_momentum = rolling_avg_daily_return / rolling_volatility

    # Weighting by Volume and Price Movement
    volume_price_movement_score = df['volume'] * (df['high'] - df['open'])
    weighted_true_avg_price = true_avg_price * (1 / price_range_volatility)

    # Incorporate Volume Trend
    ema_volume_short = df['volume'].ewm(span=5, adjust=False).mean()
    ema_volume_long = df['volume'].ewm(span=20, adjust=False).mean()
    volume_trend_difference = ema_volume_short - ema_volume_long

    # Combine Momentum and Volume Trends
    combined_momentum_volume = momentum_difference + volume_trend_difference

    # Final Momentum Indicator
    final_momentum_indicator = weighted_true_avg_price * volume_price_movement_score

    # Integrate with Combined Momentum and Volume Trends
    integrated_factor = (adjusted_momentum * final_momentum_indicator + combined_momentum_volume) * 0.5

    return integrated_factor
```
