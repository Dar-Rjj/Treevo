```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].rolling(window=5).sum()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].rolling(window=20).sum()
    
    # Determine the Difference in Price Momentum
    diff_price_momentum = short_term_momentum - long_term_momentum
    
    # Incorporate Volume Trend
    short_term_volume = df['volume'].rolling(window=5).sum()
    long_term_volume = df['volume'].rolling(window=20).sum()
    diff_volume = short_term_volume - long_term_volume
    
    # Calculate Adjusted High-Low Range
    high_low_range = df['high'] - df['low']
    scaled_volume = df['volume'] / long_term_volume
    adjusted_high_low_range = high_low_range * scaled_volume
    
    # Calculate Previous Day's Return
    previous_day_return = (df['close'] / df['open'].shift(1)) - 1
    
    # Adjust Momentum by Volatility
    realized_volatility = df['close'].pct_change().rolling(window=20).std()
    adjusted_momentum_by_volatility = previous_day_return / realized_volatility
    
    # Inverse of Intraday Range
    intraday_range_inverse = 1 / (df['high'] - df['low'])
    
    # Combine Factors
    combined_factors = (adjusted_high_low_range * intraday_range_inverse) + previous_day_return + diff_price_momentum + adjusted_momentum_by_volatility
    
    # Final Alpha Factor
    alpha_factor = (diff_price_momentum / diff_volume) + adjusted_high_low_range + combined_factors
    final_alpha_factor = alpha_factor * df['volume']
    
    return final_alpha_factor
```
