```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Calculate Rolling Average Daily Return (e.g., 21 days)
    df['Rolling_Avg_Daily_Return'] = df['Daily_Return'].rolling(window=21).mean()
    
    # Calculate Rolling Volatility of Close Prices (e.g., 20 days)
    df['Rolling_Volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Adjust Momentum by Volatility
    df['Adjusted_Momentum'] = df['Rolling_Avg_Daily_Return'] / df['Rolling_Volatility']
    
    # Weight by Volume
    df['Weighted_Adjusted_Momentum'] = df['Adjusted_Momentum'] * df['volume']
    
    # Calculate Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Calculate Volume Momentum (e.g., 5 days)
    df['Volume_Momentum'] = df['volume'] / df['volume'].shift(5)
    
    # Adjust Volume Momentum by Volatility
    df['Adjusted_Volume_Momentum'] = df['Volume_Momentum'] / df['Rolling_Volatility']
    
    # Combine Factors
    df['Combined_Factors'] = df['Weighted_Adjusted_Momentum'] + df['Volume_Change']
    
    # Integrate Volume Momentum
    df['Integrated_Volume_Momentum'] = df['Combined_Factors'] + 0.5 * df['Adjusted_Volume_Momentum']
    
    # Calculate Price Range
    df['Price_Range'] = df['high'] - df['low']
    
    # Calculate Rolling Average Price Range (e.g., 21 days)
    df['Rolling_Avg_Price_Range'] = df['Price_Range'].rolling(window=21).mean()
    
    # Adjust Price Range by Rolling Average
    df['Adjusted_Price_Range'] = df['Price_Range'] / df['Rolling_Avg_Price_Range']
    
    # Integrate Price Range into Combined Factors
    df['Final_Factor'] = df['Integrated_Volume_Momentum'] + df['Adjusted_Price_Range'] * df['volume']
    
    return df['Final_Factor']
```
