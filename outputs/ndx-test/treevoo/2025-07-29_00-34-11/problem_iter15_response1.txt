```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']

    # Determine Gain and Loss
    df['gain'] = 0
    df['loss'] = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            df['gain'].iloc[i] = (df['close'].iloc[i] - df['close'].iloc[i-1]) * df['volume'].iloc[i]
        else:
            df['loss'].iloc[i] = (df['close'].iloc[i-1] - df['close'].iloc[i]) * df['volume'].iloc[i]

    # Aggregate Gains and Losses over a period (e.g., 14 days)
    df['avg_gain'] = df['gain'].rolling(window=14).sum()
    df['avg_loss'] = df['loss'].rolling(window=14).sum()

    # Calculate Relative Strength
    df['rs'] = df['avg_gain'] / df['avg_loss']

    # Convert to ARSI
    df['arsi'] = 100 - (100 / (1 + df['rs']))

    # Incorporate Average Volume
    df['avg_volume'] = df['volume'].rolling(window=14).mean()
    df['arsi_vol_adjusted'] = df['arsi'] * df['avg_volume']

    # Calculate Intraday Price Movement Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    df['open_close_ratio'] = df['close'] / df['open']
    df['intraday_movement_ratio'] = (df['high_low_ratio'] + df['open_close_ratio']) / 2

    # Measure Price-Volatility Alignment
    df['intraday_volatility'] = df['volume'] * (df['high'] - df['low'])
    df['alignment'] = df['intraday_movement_ratio'] / df['intraday_volatility']
    df['sentiment'] = df['alignment'].apply(lambda x: 1 if x > 0.5 else -1)

    # Calculate Short-Term and Long-Term Returns
    df['short_term_return'] = df['close'].pct_change(periods=5)
    df['long_term_return'] = df['close'].pct_change(periods=20)

    # Calculate Volume-Weighted Returns
    df['vol_weighted_short_term_return'] = df['volume'] * df['short_term_return']
    df['vol_weighted_long_term_return'] = df['volume'] * df['long_term_return']

    # Calculate Short-Term Volatility
    df['atr'] = (df['high'] - df['low']).rolling(window=5).mean()

    # Calculate High-Low Spread and VWMA
    df['high_low_spread'] = df['high'] - df['low']
    df['vwm_high_low_spread'] = (df['high_low_spread'] * df['volume']).rolling(window=14).mean() / df['avg_volume']

    # Adjust for Volatility and Sentiment
    df['adjusted_short_term_return'] = df['vol_weighted_short_term_return'] / df['atr'] - df['vol_weighted_long_term_return']
    df['adjusted_short_term_return'] = df['adjusted_short_term_return'] * df['sentiment']

    # Combine ARSI, Adjusted Returns, and VWMA
    df['alpha_factor'] = (df['arsi_vol_adjusted'] * df['adjusted_short_term_return'] * df['vwm_high_low_spread']) - df['vol_weighted_long_term_return']

    return df['alpha_factor']
```
