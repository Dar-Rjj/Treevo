```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the money flow as the product of the average price and volume
    money_flow = avg_price * df['volume']
    
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Calculate the factor as the ratio of money flow to the high-low range
    factor = money_flow / high_low_range
    
    return factor
```
