import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    alignment = intraday_movement_ratio / intraday_volatility
    sentiment = (alignment > 0).astype(int) * 2 - 1  # Positive if high, negative if low
    
    # Calculate Gain and Loss
    gain = df['close'].diff().clip(lower=0)
    loss = df['close'].diff().clip(upper=0).abs()
    
    # Aggregate Gains and Losses
    window = 14
    sum_gains = gain.rolling(window=window).sum()
    sum_losses = loss.rolling(window=window).sum()
    
    # Calculate Relative Strength
    rs = sum_gains / sum_losses
    
    # Convert to ARSI
    rsi = 100 - (100 / (1 + rs))
    arsi = rsi * df['volume'].rolling(window=window).mean() * (df['close'] / df['open'])
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'].pct_change(periods=5)
    long_term_return = df['close'].pct_change(periods=20)
    
    # Calculate Volume-Weighted Returns
    vol_weighted_short_term_return = short_term_return * df['volume']
    vol_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    atr = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()).mean()
    
    # Adjust for Volatility
    adjusted_short_term_return = vol_weighted_short_term_return / atr
    
    # Calculate Price Oscillator
    price_oscillator = df['close'].rolling(window=5).mean() - df['close'].rolling(window=20).mean()
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    combined_factor = arsi * adjusted_short_term_return + price_oscillator - vol_weighted_long_term_return
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=window).mean()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'].diff()
    multiplied_vwma = vwma_high_low_spread * close_to_close_return
    
    # Adjust with Volume Change
    volume_change = (df['volume'].diff() / df['volume'].shift(1)).fillna(0)
    final_factor = multiplied_vwma * volume_change
    
    return final_factor
