```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_factor = intraday_volatility * close_to_close_return
    
    # Adjust for Volume
    volume_adjusted_factor = combined_factor / (df['volume'] + 1)
    
    # Calculate Short-Term Return (t-1 to t-5)
    short_term_return = df['close'] / df['close'].shift(5) - 1
    
    # Calculate Long-Term Return (t-1 to t-20)
    long_term_return = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = df['volume'] * short_term_return
    
    # Calculate Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = df['volume'] * long_term_return
    
    # Calculate Short-Term Volatility
    high_low_range = df['high'] - df['low']
    short_term_volatility = high_low_range.rolling(window=5).mean()
    
    # Enhance Volume Impact
    volume_enhanced_impact = np.log(df['volume'])
    
    # Final Adjustment
    final_alpha = (volume_adjusted_factor / (short_term_volatility + 1)) + volume_enhanced_impact - volume_weighted_long_term_return
    
    return final_alpha
```
