```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple Moving Average (SMA) of closing prices over 50 days
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Exponential Moving Average (EMA) of closing prices over 20 days
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Rate of Change (ROC) of closing prices over 12 days
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    # Relative Strength Index (RSI) using a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # On-Balance Volume (OBV)
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    
    # Price-Volume Trend (PVT)
    df['PVT'] = ((df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']).cumsum()
    
    # Average True Range (ATR) over a 14-day window
    df['TR'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Bollinger Bands, 20-day SMA of closing prices Â± 2 standard deviations
    df['SMA_20_BB'] = df['close'].rolling(window=20).mean()
    df['std_20'] = df['close'].rolling(window=20).std()
    df['Upper_BB'] = df['SMA_20_BB'] + 2 * df['std_20']
    df['Lower_BB'] = df['SMA_20_BB'] - 2 * df['std_20']
    
    # Custom Momentum Indicator (CMI)
    df['CMI'] = (df['close'] - df['low'].shift(1)) / (df['high'].shift(1) - df['low'].shift(1)) * 100
    
    # Volatility-Adjusted Momentum (VAM)
    df['VAM'] = df['ROC_12'] / df['ATR_14']
    
    # Bullish and Bearish Engulfing Patterns
    df['Bullish_Engulfing'] = (df['open'] < df['close']) & (df['open'].shift(-1) > df['close'].shift(-1)) & (df['close'].shift(-1) > df['open'])
    df['Bearish_Engulfing'] = (df['open'] > df['close']) & (df['open'].shift(-1) < df['close'].shift(-1)) & (df['close'].shift(-1) < df['open'])
    
    # Head and Shoulders formations
    df['Head_and_Shoulders'] = (df['close'] > df['high'].shift(1)) & (df['close'].shift(1) < df['low'].shift(2)) & (df['close'].shift(2) > df['high'].shift(3)) & (df['close'].shift(3) < df['low'].shift(4))
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = df['SMA_50'] + df['EMA_20'] + df['ROC_12'] + df['RSI_14'] + df['OBV'] + df['PVT'] + df['ATR_14'] + df['SMA_20_BB'] + df['CMI'] + df['VAM'] + df['Bullish_Engulfing'] + df['Bearish_Engulfing'] + df['Head_and_Shoulders']
    
    return df['alpha_factor']
```
