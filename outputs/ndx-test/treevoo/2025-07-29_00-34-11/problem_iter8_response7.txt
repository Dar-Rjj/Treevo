```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['loss'] = (df['close'] <= df['close'].shift(1)) * (df['close'].shift(1) - df['close'])
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    gain_sum = df['gain'].rolling(window=14).sum()
    loss_sum = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = gain_sum / loss_sum
    
    # Convert to ARSI
    arsi = 100 - (100 / (1 + rs))
    arsi = arsi * df['volume'].rolling(window=14).mean()  # Incorporate Volume
    arsi = arsi * (df['close'] / df['open'])  # Add Momentum
    
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    sentiment = (intraday_movement_ratio - intraday_volatility) / intraday_volatility
    
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma = (high_low_spread * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'] - df['close'].shift(1)
    vwma_momentum = vwma * close_to_close_return
    
    # Combine Intraday Volatility and Close-to-Close Return
    intraday_volatility_return = (df['high'] - df['low']) * close_to_close_return
    
    # Synthesize Combined Factors
    combined_factors = (vwma_momentum + intraday_volatility_return) / df['volume']
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'].shift(1) / df['close'].shift(5) - 1
    long_term_return = df['close'].shift(1) / df['close'].shift(20) - 1
    
    # Calculate Volume-Weighted Short-Term and Long-Term Returns
    volume_weighted_short_term_return = df['volume'] * short_term_return
    volume_weighted_long_term_return = df['volume'] * long_term_return
    
    # Calculate Short-Term Volatility
    short_term_volatility = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()).mean()
    
    # Adjust for Volatility and Sentiment
    factor = (volume_weighted_short_term_return / short_term_volatility) - volume_weighted_long_term_return
    factor = factor + combined_factors * (1 if sentiment.mean() > 0 else -1)
    factor = factor / df['volume']
    
    return factor
```
