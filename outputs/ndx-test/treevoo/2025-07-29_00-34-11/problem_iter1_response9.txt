```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(4)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(19)
    
    # Calculate Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Calculate Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    high_low_range = df['high'] - df['low']
    true_range = high_low_range.rolling(window=5).mean()
    short_term_volatility = true_range.rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_factor = (volume_weighted_short_term_return / short_term_volatility) - volume_weighted_long_term_return
    
    return adjusted_factor
```
