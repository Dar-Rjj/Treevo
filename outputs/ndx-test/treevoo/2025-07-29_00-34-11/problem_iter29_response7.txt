def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    avg_ratios = (high_low_ratio + open_close_ratio) / 2
    volume_weighted_ratios = avg_ratios * df['volume']
    
    # Measure Price-Volatility Alignment
    intraday_volatility = df['high'] - df['low']
    sentiment = (avg_ratios > intraday_volatility).astype(int)
    
    # Incorporate Momentum
    close_to_close_return = df['close'].pct_change()
    momentum_adjusted_ratios = volume_weighted_ratios * close_to_close_return
    
    # Analyze Price Action Patterns
    relative_closing_price_change = (df['close'].diff() / df['close'].shift(1)).fillna(0)
    true_range = (df['high'] - df['low']).abs()
    upper_shadow_length = (df['high'] - df[['open', 'close']].max(axis=1))
    lower_shadow_length = (df[['open', 'close']].min(axis=1) - df['low'])
    
    # Investigate Volume-Related Factors
    volume_ratio = df['volume'] / df['volume'].shift(1).fillna(df['volume'].mean())
    volume_weighted_close = df['close'] * df['volume']
    
    # Derive Momentum Indicators from Price Data
    n = 5
    sma = df['close'].rolling(window=n).mean()
    roc = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    directional_movement = df['close'].diff()
    
    # Integrate Amount into Analysis
    amount_to_volume_ratio = df['amount'] / df['volume']
    amount_price_relationship = (df['amount'] / df['volume']) * df['close']
    
    # Combine Volume-Weighted Ratios and Momentum
    sma_adjusted_ratios = momentum_adjusted_ratios * sma
    roc_adjusted_return = close_to_close_return * roc
    
    # Synthesize Price Action and Volume-Related Factors
    price_action_synthesis = relative_closing_price_change * true_range
    shadow_length_synthesis = (upper_shadow_length + lower_shadow_length) * volume_ratio
    
    # Enhance Momentum Indicators with Volume-Weighted Close
    sma_with_volume_weighted_close = sma * volume_weighted_close
    roc_with_volume_weighted_close = roc * volume_weighted_close
    directional_movement_with_volume = directional_movement * volume_weighted_close
    
    # Final Alpha Factor
    alpha_factor = (
        sma_adjusted_ratios * 
        roc_adjusted_return * 
        price_action_synthesis * 
        shadow_length_synthesis * 
        sma_with_volume_weighted_close * 
        roc_with_volume_weighted_close * 
        directional_movement_with_volume
    )
    
    return alpha_factor
