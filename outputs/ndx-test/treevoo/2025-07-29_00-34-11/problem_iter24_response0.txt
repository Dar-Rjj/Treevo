```python
def heuristics_v2(df):
    """
    Generate an alpha factor using a combination of price and volume data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'], indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the percentage change in closing price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of volume to the 30-day average volume
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30
    
    # Calculate the range of the high-low prices
    price_range = df['high'] - df['low']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (close_pct_change * volume_ratio) / price_range
    
    return alpha_factor
```
