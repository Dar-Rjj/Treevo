```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()

    # Aggregate Daily Return
    df['Volume_Adjusted_Return'] = df['Daily_Return'] * df['volume']
    df['Aggregated_Return'] = df['Volume_Adjusted_Return'].rolling(window=20).sum()

    # Calculate Long-Term Trend Strength
    df['Trend_Strength'] = df['close'] - df['close'].shift(60)

    # Combine Momentum and Trend Factors
    df['Momentum_Trend_Score'] = df['Aggregated_Return'] + df['Trend_Strength']

    # Adjust for Volume Trend
    df['Volume_Change'] = df['volume'] - df['volume'].shift(20)
    df['Adjusted_Momentum_Trend_Score'] = df['Momentum_Trend_Score'] * (1.5 if df['Volume_Change'].iloc[-1] > 0 else 1/1.5)

    # Calculate Intraday Price Movement Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    df['Open_Close_Ratio'] = df['open'] / df['close']
    df['Intraday_Movement_Ratio'] = (df['High_Low_Ratio'] + df['Open_Close_Ratio']) / 2

    # Measure Price-Volatility Alignment
    df['Intraday_Volatility'] = (df['high'] - df['low']) * df['volume']
    df['Alignment_Score'] = df['Intraday_Movement_Ratio'] / df['Intraday_Volatility']
    df['Sentiment_Score'] = df['Alignment_Score'].apply(lambda x: 1 if x > 1 else -1 if x < 1 else 0)

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Momentum_Trend_Score'] + df['Sentiment_Score']
    
    return df['Final_Alpha_Factor']
```
