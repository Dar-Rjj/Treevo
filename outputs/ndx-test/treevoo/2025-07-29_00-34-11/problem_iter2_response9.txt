import pandas as pd

    # Calculate Short-Term Logarithmic Return
    df['short_term_log_return'] = (df['close'] / df['close'].shift(5)).apply(lambda x: 0 if x <= 0 else np.log(x))
    
    # Calculate Long-Term Logarithmic Return
    df['long_term_log_return'] = (df['close'] / df['close'].shift(20)).apply(lambda x: 0 if x <= 0 else np.log(x))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    df['volume_weighted_short_term_log_return'] = df['short_term_log_return'] * df['volume']
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    df['volume_weighted_long_term_log_return'] = df['long_term_log_return'] * df['volume']
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Aggregate Volume Changes over the last M days, M = 5
    df['aggregated_volume_changes'] = df['daily_volume_change'].rolling(window=5).sum()
    
    # Calculate Short-Term Volatility using High and Low prices of last 5 days
    df['high_low_diff'] = df['high'] - df['low']
    df['short_term_volatility'] = df['high_low_diff'].rolling(window=5).mean()

    # Combine Indicators with Adjustments
    df['factor_value'] = (df['volume_weighted_short_term_log_return'] / df['short_term_volatility']) - \
                         df['volume_weighted_long_term_log_return'] * df['aggregated_volume_changes']

    return df['factor_value'].dropna()
