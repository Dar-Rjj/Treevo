```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Intensity
    high_low_spread = df['high'] - df['low']
    open_close_change = df['close'] - df['open']
    intraday_movement_intensity = high_low_spread + open_close_change

    # Measure Momentum-Volatility Correlation
    intraday_volatility = (df['high'] - df['low']) / df['volume']
    correlation = intraday_movement_intensity * intraday_volatility
    momentum_sentiment = correlation.apply(lambda x: 1 if x > 0 else -1)

    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2

    # Measure Price-Volatility Alignment
    intraday_volatility_weighted = (df['high'] - df['low']) * df['volume']
    alignment = intraday_movement_ratio * intraday_volatility_weighted
    price_volatility_alignment = alignment.apply(lambda x: 1 if x > 0 else -1)

    # Combine Momentum Sentiment and Price-Volatility Alignment
    alpha_factor = momentum_sentiment * price_volatility_alignment

    return alpha_factor
```
