```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame containing at least 'open', 'close', and 'volume' columns.
    
    Returns:
    pd.Series: Series of the calculated factor values, indexed by date.
    """
    # Calculate the close-to-open return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight the return by volume
    weighted_return = close_to_open_return * df['volume']
    
    return weighted_return
```
