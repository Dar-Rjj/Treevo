```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price movement
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the relative strength index (RSI)-like heuristic
    df['up_movement'] = df['price_movement'].apply(lambda x: x if x > 0 else 0)
    df['down_movement'] = df['price_movement'].apply(lambda x: -x if x < 0 else 0)
    df['avg_up_movement'] = df['up_movement'].rolling(window=14).mean()
    df['avg_down_movement'] = df['down_movement'].rolling(window=14).mean()
    df['rsi_like'] = 100 - (100 / (1 + (df['avg_up_movement'] / df['avg_down_movement'])))
    
    # Calculate the volume weighted price movement
    df['vwp_movement'] = df['price_movement'] * (df['volume'] / df['volume'].rolling(window=14).mean())
    
    # Combine the RSI-like with the volume weighted price movement
    df['alpha_factor'] = df['rsi_like'] + df['vwp_movement']
    
    return df['alpha_factor']
```
