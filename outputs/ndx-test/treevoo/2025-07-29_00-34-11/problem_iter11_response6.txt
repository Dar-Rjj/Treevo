import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Average High-Low and Open-Close Ratios
    avg_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Weight by Volume
    volume_weighted_ratios = avg_ratio * df['volume']
    
    # Calculate Intraday Volatility
    intraday_range = df['high'] - df['low']
    weighted_volatility = intraday_range * df['volume']
    
    # Evaluate Alignment
    alignment = volume_weighted_ratios / weighted_volatility
    sentiment = alignment.apply(lambda x: 1 if x > 1 else -1 if x < 1 else 0)
    
    # Incorporate Enhanced Momentum into Adjusted Ratios
    close_to_close_return = df['close'].pct_change()
    enhanced_momentum_ratios = volume_weighted_ratios * close_to_close_return
    
    # Final Volume Adjustment
    final_factor = enhanced_momentum_ratios / df['volume']
    
    return final_factor
