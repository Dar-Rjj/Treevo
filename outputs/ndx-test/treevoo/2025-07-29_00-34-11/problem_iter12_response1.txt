```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['gain'] = 0
    df['loss'] = 0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            df.at[df.index[i], 'gain'] = df['close'].iloc[i] - df['close'].iloc[i-1]
        else:
            df.at[df.index[i], 'loss'] = df['close'].iloc[i-1] - df['close'].iloc[i]
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    
    # Calculate Relative Strength
    df['RS'] = df['avg_gain'] / df['avg_loss']
    
    # Convert to ARSI
    df['ARSI'] = 100 - (100 / (1 + df['RS']))
    
    # Adjust for Advanced Features
    df['ARSI'] = df['ARSI'] * (df['volume'].rolling(window=14).mean())
    df['momentum'] = (df['close'] / df['open']) - 1
    df['ARSI'] = df['ARSI'] * df['momentum']
    
    # Calculate Short-Term Return
    df['short_term_return'] = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = df['close'].pct_change(periods=20)
    
    # Calculate Volume-Weighted Short-Term Return
    df['vol_weighted_short_term_return'] = df['volume'] * df['short_term_return']
    
    # Calculate Volume-Weighted Long-Term Return
    df['vol_weighted_long_term_return'] = df['volume'] * df['long_term_return']
    
    # Calculate Short-Term Volatility
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['short_term_volatility'] = df['true_range'].rolling(window=5).mean()
    
    # Synthesize ARSI with Volume-Weighted Returns and Volatility
    df['combined_arsi'] = (df['ARSI'] * df['vol_weighted_short_term_return']) / df['short_term_volatility']
    df['combined_arsi'] -= df['vol_weighted_long_term_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change(periods=1)
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['intraday_adj'] = df['intraday_volatility'] * df['close_to_close_return']
    
    # Final Adjustment
    df['final_factor'] = df['combined_arsi'] + df['intraday_adj']
    df['vol_weighted_intraday_volatility'] = df['volume'] * df['intraday_volatility']
    df['vol_weighted_momentum'] = df['volume'] * df['momentum']
    df['final_factor'] += df['vol_weighted_intraday_volatility'] + df['vol_weighted_momentum']
    
    return df['final_factor']
```
