```python
def heuristics_v2(data):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
        data (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, volume].
        
    Returns:
        pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily price range
    daily_range = data['high'] - data['low']
    
    # Calculate the daily absolute price change
    daily_abs_change = (data['close'] - data['open']).abs()
    
    # Calculate the factor as the ratio of daily absolute price change to daily price range
    factor = daily_abs_change / daily_range
    
    return factor
```
