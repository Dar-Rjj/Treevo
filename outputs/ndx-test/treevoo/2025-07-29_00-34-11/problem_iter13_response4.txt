```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Intensity
    high_low_spread = df['high'] - df['low']
    open_close_change = df['close'] - df['open']
    movement_intensity = high_low_spread + open_close_change
    
    # Measure Momentum-Volatility Correlation
    intraday_volatility = (df['high'] - df['low']) / df['volume']
    momentum_volatility_correlation = movement_intensity * intraday_volatility
    momentum_volatility_sentiment = momentum_volatility_correlation.apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    average_ratio = (high_low_ratio + open_close_ratio) / 2
    intraday_volatility_ratio = (df['high'] - df['low']) * df['volume']
    sentiment = average_ratio * intraday_volatility_ratio
    movement_ratio_sentiment = sentiment.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine the two sentiment indicators
    combined_sentiment = (momentum_volatility_sentiment + movement_ratio_sentiment) / 2
    
    return combined_sentiment
```
