```python
import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    volume_weighted_ratios = (high_low_ratio + open_close_ratio) / 2 * df['volume']

    # Measure Price-Volatility Alignment
    intraday_volatility = df['high'] - df['low']
    price_volatility_alignment = (high_low_ratio + open_close_ratio) / 2 - intraday_volatility
    price_volatility_sentiment = price_volatility_alignment.apply(lambda x: 1 if x > 0 else -1)

    # Incorporate Momentum
    close_to_close_return = df['close'].pct_change()
    adjusted_volume_weighted_ratios = volume_weighted_ratios * close_to_close_return

    # Analyze Price Action Patterns
    relative_closing_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    true_range = pd.concat([df['high'], df['close'].shift(1)], axis=1).max(axis=1) - pd.concat([df['low'], df['close'].shift(1)], axis=1).min(axis=1)
    upper_shadow_length = df['high'] - pd.concat([df['open'], df['close']], axis=1).max(axis=1)
    lower_shadow_length = pd.concat([df['open'], df['close']], axis=1).min(axis=1) - df['low']

    # Investigate Volume-Related Factors
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_weighted_close = df['close'] * df['volume']

    # Derive Momentum Indicators from Price Data
    sma_5 = df['close'].rolling(window=5).mean()
    roc_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    directional_movement = df['close'] - df['close'].shift(1)

    # Integrate Accumulation/Distribution Line
    ad_line = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    ad_line = ad_line.cumsum()

    # Include Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where((typical_price > typical_price.shift(1)), 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where((typical_price < typical_price.shift(1)), 0).rolling(window=14).sum()
    money_ratio = positive_money_flow / negative_money_flow
    mfi = 100 - (100 / (1 + money_ratio))

    # Combine all factors into a single alpha factor
    alpha_factor = (
        adjusted_volume_weighted_ratios +
        price_volatility_sentiment +
        relative_closing_price_change +
        true_range +
        upper_shadow_length +
        lower_shadow_length +
        volume_ratio +
        volume_weighted_close +
        sma_5 +
        roc_10 +
        directional_movement +
        ad_line +
        mfi
    )

    return alpha_factor
```
