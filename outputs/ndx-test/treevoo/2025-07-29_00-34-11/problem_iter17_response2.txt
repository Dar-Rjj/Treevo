```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Logarithmic Returns
    df['short_term_return'] = np.log(df['close'] / df['close'].shift(5))
    df['long_term_return'] = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term and Long-Term Logarithmic Returns
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Calculate Short-Term Volatility (Average True Range)
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    df['short_term_volatility'] = df['true_range'].rolling(window=5).mean()
    
    # Adjust for Volatility
    df['adjusted_return'] = (df['volume_weighted_short_term_return'] / df['short_term_volatility']) - df['volume_weighted_long_term_return']
    
    # Determine Gain and Loss
    df['gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['loss'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    
    # Aggregate Gains and Losses
    df['sum_gain'] = df['gain'].rolling(window=14).sum()
    df['sum_loss'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['sum_gain'] / df['sum_loss']
    
    # Convert to ARSI
    df['arsi'] = 100 - (100 / (1 + df['rs']))
    df['arsi'] = df['arsi'] * (df['volume'].rolling(window=14).mean()) * ((df['close'] - df['open']) / df['open'])
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['combined_volatility_return'] = df['intraday_volatility'] * df['close_to_close_return']
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Aggregate Volume Changes
    df['aggregated_volume_changes'] = df['daily_volume_change'].rolling(window=5).sum()
    
    # Calculate Price Oscillator
    df['price_oscillator'] = df['short_term_return'] - df['long_term_return']
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    df['factor'] = (df['arsi'] * df['adjusted_return'] + df['price_oscillator'] - df['volume_weighted_long_term_return']) \
                    + df['adjusted_return'] + df['combined_volatility_return']
    df['factor'] *= df['aggregated_volume_changes']
    
    return df['factor']
```
