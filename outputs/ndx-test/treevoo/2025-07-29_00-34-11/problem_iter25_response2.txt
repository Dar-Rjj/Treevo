import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Logarithmic Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)).apply(np.log)
    df['long_term_return'] = (df['close'] / df['close'].shift(20)).apply(np.log)
    
    # Volume-Weighted Short-Term and Long-Term Logarithmic Return
    df['volume_weighted_short_term_return'] = df['short_term_return'] * df['volume']
    df['volume_weighted_long_term_return'] = df['long_term_return'] * df['volume']
    
    # Calculate Short-Term Volatility (Average True Range)
    df['tr'] = df[['high', 'low']].apply(lambda x: x.max() - x.min(), axis=1)
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Adjust for Volatility
    df['adjusted_short_term_return'] = (df['volume_weighted_short_term_return'] / df['atr']) - df['volume_weighted_long_term_return']
    
    # Determine Gain and Loss
    df['gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['loss'] = -df['close'].diff().apply(lambda x: x if x < 0 else 0)
    
    # Aggregate Gains and Losses
    df['sum_gains'] = df['gain'].rolling(window=14).sum()
    df['sum_losses'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['sum_gains'] / df['sum_losses']
    
    # Convert to ARSI
    df['arsi'] = 100 - (100 / (1 + df['rs']))
    df['arsi'] *= df['volume'].rolling(window=14).mean()
    
    # Calculate Intraday Price Movement Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    df['open_close_ratio'] = df['open'] / df['close']
    df['intraday_movement_ratio'] = (df['high_low_ratio'] + df['open_close_ratio']) / 2
    
    # Measure Price-Volatility Alignment
    df['price_range'] = df['high'] - df['low']
    df['volatility_weighted_price_range'] = df['price_range'] * df['volume']
    df['alignment'] = df['intraday_movement_ratio'] - df['volatility_weighted_price_range']
    df['sentiment'] = df['alignment'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Price Oscillator
    df['price_oscillator'] = df['short_term_return'] - df['long_term_return']
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    df['combined_factor'] = df['arsi'] * df['adjusted_short_term_return'] + df['price_oscillator'] - df['volume_weighted_long_term_return']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    df['vwm_high_low_spread'] = (df['high_low_spread'] * df['volume']).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Incorporate Close-to-Close Return
    df['close_to_close_return'] = df['close'].diff()
    df['multiplied_vwma'] = df['vwm_high_low_spread'] * df['close_to_close_return']
    
    # Final Adjustment
    df['final_adjustment'] = (
        df['adjusted_short_term_return'] +
        df['combined_factor'] +
        df['multiplied_vwma'] * (df['volume'] / df['volume'].shift(1) - 1)
    )
    
    return df['final_adjustment']
