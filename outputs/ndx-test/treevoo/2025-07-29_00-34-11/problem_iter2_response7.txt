```python
import pandas as pd

def heuristics_v2(df, window=10):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low = (high_low_spread * df['volume']).rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
    
    # Adjust VWMA by Volume
    adjusted_vwma = vwma_high_low / df['volume']
    
    # Calculate Price Momentum
    price_momentum = (df['close'].shift(1) - df['close']) / df['close'].shift(1)
    
    # Combine Adjusted VWMA and Momentum
    factor_values = adjusted_vwma * price_momentum
    
    return factor_values
```
