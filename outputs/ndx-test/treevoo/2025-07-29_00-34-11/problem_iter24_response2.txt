```python
def heuristics_v2(df):
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the close price to the average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the difference between the open and close prices
    open_close_diff = df['open'] - df['close']
    
    # Calculate the ratio of the volume to the close price
    volume_to_close_ratio = df['volume'] / df['close']
    
    # Combine the factors into a single heuristic
    heuristic = close_to_avg_ratio * open_close_diff * volume_to_close_ratio
    
    return heuristic
```
