```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_price_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    sentiment = intraday_price_movement_ratio > intraday_volatility
    sentiment = sentiment.astype(int) * 2 - 1  # Convert to +1 for positive, -1 for negative
    
    # Calculate Gain and Loss
    gain = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    loss = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    
    # Aggregate Gains and Losses
    sum_gains = gain.rolling(window=14).sum()
    sum_losses = loss.rolling(window=14).sum()
    
    # Calculate Relative Strength
    relative_strength = sum_gains / sum_losses.replace(0, 1)
    
    # Convert to ARSI
    arsi = 100 - (100 / (1 + relative_strength))
    arsi = arsi * df['volume'].rolling(window=14).mean() * (df['close'] / df['open']).shift(1)
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = (df['close'].shift(1) - df['close'].shift(5)) / df['close'].shift(5)
    long_term_return = (df['close'].shift(1) - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume-Weighted Returns
    volume_weighted_short_term_return = short_term_return * df['volume']
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    high_low_5_days = df[['high', 'low']].rolling(window=5).max() - df[['high', 'low']].rolling(window=5).min()
    short_term_volatility = high_low_5_days.mean(axis=1)
    
    # Adjust for Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / short_term_volatility
    
    # Calculate Price Oscillator
    price_oscillator = df['close'].shift(5) - df['close'].shift(20)
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    combined_factor = (arsi * adjusted_short_term_return) + price_oscillator - volume_weighted_long_term_return
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    vwma_high_low_spread *= close_to_close_return
    
    # Adjust with Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_vwma = vwma_high_low_spread * (1 + volume_change)
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / (df['high'] + df['low'])
    weighted_intraday_momentum = intraday_momentum * df['volume']
    
    # Calculate Extended Momentum
    avg_5_day_intraday_momentum = weighted_intraday_momentum.rolling(window=5).mean()
    avg_20_day_intraday_momentum = weighted_intraday_momentum.rolling(window=20).mean()
    
    # Enhance Momentum Signal
    momentum_difference = avg_5_day_intraday_momentum - avg_20_day_intraday_momentum
    adjusted_momentum = momentum_difference * (arsi / 100)
    
    # Integrate All Factors
    integrated_alpha_factor = combined_factor + adjusted_momentum
    
    return integrated_alpha_factor
```
