```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of close price in the day's range
    df['day_range'] = df['high'] - df['low']
    df['close_position'] = (df['close'] - df['low']) / df['day_range']

    # Calculate the change in close position from the previous day
    df['close_position_change'] = df['close_position'].diff()

    # Calculate the volume weighted by the close position
    df['volume_weighted'] = df['volume'] * df['close_position']

    # Calculate the factor value based on the combination of the above
    df['factor_value'] = (df['close_position_change'] * df['volume_weighted']).rolling(window=5).mean()

    return df['factor_value']
```
