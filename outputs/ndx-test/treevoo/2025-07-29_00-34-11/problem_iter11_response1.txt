```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Logarithmic Returns
    short_term_return = np.log(df['close'] / df['close'].shift(5))
    long_term_return = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term and Long-Term Logarithmic Returns
    volume_weighted_short_term_return = short_term_return * df['volume']
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility (Average True Range)
    high_low = df['high'] - df['low']
    high_close = abs(df['high'] - df['close'].shift(1))
    low_close = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
    short_term_volatility = true_range.rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_returns = (volume_weighted_short_term_return / short_term_volatility) - volume_weighted_long_term_return
    
    # Determine Gain and Loss
    gain = (df['close'] - df['close'].shift(1)).clip(lower=0)
    loss = (df['close'].shift(1) - df['close']).clip(lower=0)
    
    # Aggregate Gains and Losses over 14 days
    sum_gains = gain.rolling(window=14).sum()
    sum_losses = loss.rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = sum_gains / sum_losses
    
    # Convert to ARSI
    rsi = 100 - (100 / (1 + rs))
    arsi = rsi * df['volume'].rolling(window=14).mean() * (df['close'] / df['open'])
    
    # Calculate Intraday Price Movement Intensity
    intraday_movement_intensity = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Measure Momentum-Volatility Correlation
    intraday_volatility = (df['high'] - df['low']) / df['volume']
    momentum_sentiment = intraday_movement_intensity.corr(intraday_volatility, method='pearson')
    
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    price_volatility_alignment = (df['high'] - df['low']) * df['volume']
    sentiment = intraday_movement_ratio.corr(price_volatility_alignment, method='pearson')
    
    # Combine Intraday Price Movement Intensity and Momentum Sentiment
    intensity_sentiment = intraday_movement_intensity * np.sign(momentum_sentiment)
    
    # Combine Intraday Price Movement Ratio and Sentiment
    ratio_sentiment = intraday_movement_ratio * np.sign(sentiment)
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    price_oscillator = df['close'] - df['close'].shift(14)
    combined_factor = arsi * adjusted_returns + price_oscillator - volume_weighted_long_term_return
    
    # Final Adjustment
    final_factor = adjusted_returns + intensity_sentiment + ratio_sentiment + combined_factor * df['volume'].pct_change().rolling(window=14).sum()
    
    return final_factor
```
