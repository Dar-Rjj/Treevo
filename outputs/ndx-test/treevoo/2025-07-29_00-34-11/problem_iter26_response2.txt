```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Difference
    df['Price_Diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['Gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['Loss'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    gain_sum = df['Gain'].rolling(window=14).sum()
    loss_sum = df['Loss'].rolling(window=14).sum().replace(0, 1)  # Avoid division by zero
    
    # Calculate Relative Strength
    rs = gain_sum / loss_sum
    
    # Convert to ARSI
    arsi = 100 - (100 / (1 + rs))
    arsi_vol = arsi * df['volume'].rolling(window=14).mean()
    
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    price_range = df['high'] - df['low']
    weighted_volatility = df['volume'] * price_range
    alignment = intraday_movement_ratio / weighted_volatility
    sentiment = np.where(alignment > 1, 1, -1)
    
    # Calculate Short-Term Return
    short_term_return = df['close'] / df['close'].shift(5) - 1
    long_term_return = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Short-Term and Long-Term Logarithmic Returns
    short_term_log_return = np.log(df['close'] / df['close'].shift(5))
    long_term_log_return = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Logarithmic Returns
    vol_weighted_short_term_log_return = df['volume'] * short_term_log_return
    vol_weighted_long_term_log_return = df['volume'] * long_term_log_return
    
    # Calculate Short-Term Volatility
    atr = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_short_term_return = vol_weighted_short_term_log_return / atr
    adjusted_long_term_return = vol_weighted_long_term_log_return
    
    # Adjust based on Intraday Sentiment
    adjusted_ratios = adjusted_short_term_return - adjusted_long_term_return
    adjusted_ratios_sentiment = adjusted_ratios * sentiment
    
    # Incorporate Momentum
    close_to_close_return = df['close'] - df['close'].shift(1)
    momentum_adjusted_ratios = adjusted_ratios_sentiment * close_to_close_return
    
    # Final Volume Adjustment
    final_adjusted_ratios = momentum_adjusted_ratios / df['volume']
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Incorporate Close-to-Close Return
    multiplied_vwma = vwma_high_low_spread * close_to_close_return
    
    # Adjust with Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    combined_factors = multiplied_vwma * volume_change
    
    return combined_factors
```
