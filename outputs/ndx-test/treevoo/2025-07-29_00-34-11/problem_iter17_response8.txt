```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on (open, high, low, close) prices and volume.
    
    The function calculates the average of the relative strength index (RSI) over the last 14 days 
    and the ratio of the current day's closing price to the 20-day moving average of the closing price.
    The idea is to combine momentum and mean-reversion signals.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and indexed by date.
    
    Returns:
    - pandas Series indexed by date with the calculated alpha factor values.
    """
    
    # Calculate daily price changes
    delta = df['close'].diff()
    
    # Gain and loss from previous period
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    # Calculate the 14-period RSI
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 20-day moving average of the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the factor: average RSI over 14 days and the ratio of the current close to the 20-day MA
    factor = (rsi.rolling(window=14).mean() + df['close'] / ma_20) / 2
    
    return factor
```
