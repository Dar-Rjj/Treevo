```python
def heuristics_v2(df, window=20):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'].pct_change()
    
    # Multiply VWMA by Close-to-Close Return
    factor_values = vwma_high_low_spread * close_to_close_return
    
    return factor_values
```
