```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    df['Open_Close_Ratio'] = df['close'] / df['open']
    df['Intraday_Movement_Ratio'] = (df['High_Low_Ratio'] + df['Open_Close_Ratio']) / 2
    
    # Measure Price-Volatility Alignment
    df['Intraday_Volatility'] = (df['high'] - df['low']) * df['volume']
    df['Alignment'] = df['Intraday_Movement_Ratio'] / df['Intraday_Volatility']
    
    # Determine Sentiment
    df['Sentiment'] = df['Alignment'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Gain and Loss
    df['Gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['Loss'] = -df['close'].diff().apply(lambda x: x if x < 0 else 0)
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    window = 14
    df['Sum_Gains'] = df['Gain'].rolling(window=window).sum()
    df['Sum_Losses'] = df['Loss'].rolling(window=window).sum()
    
    # Calculate Relative Strength
    df['RS'] = df['Sum_Gains'] / df['Sum_Losses']
    
    # Convert to ARSI
    df['ARSI'] = 100 - (100 / (1 + df['RS']))
    df['ARSI'] *= df['volume'].rolling(window=window).mean()  # Incorporate Volume
    df['ARSI'] *= (df['close'] / df['open'])  # Add Momentum
    
    # Calculate Short-Term Return
    df['Short_Term_Return'] = df['close'] / df['close'].shift(5) - 1
    df['Volume_Weighted_Short_Term_Return'] = df['Short_Term_Return'] * df['volume']
    
    # Calculate Long-Term Return
    df['Long_Term_Return'] = df['close'] / df['close'].shift(20) - 1
    df['Volume_Weighted_Long_Term_Return'] = df['Long_Term_Return'] * df['volume']
    
    # Calculate Short-Term Volatility
    df['ATR'] = df[['high', 'low', 'close']].rolling(window=5).apply(lambda x: max(x) - min(x), raw=True).mean()
    
    # Adjust for Volatility
    df['Adjusted_Short_Term_Return'] = df['Volume_Weighted_Short_Term_Return'] / df['ATR']
    
    # Calculate Price Oscillator
    df['Price_Oscillator'] = df['close'].shift(5) - df['close'].shift(20)
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    df['Combined_Factor'] = df['ARSI'] * df['Adjusted_Short_Term_Return'] + df['Price_Oscillator'] - df['Volume_Weighted_Long_Term_Return']
    
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    df['VWMA_High_Low_Spread'] = (df['High_Low_Spread'] * df['volume']).rolling(window=window).mean() / df['volume'].rolling(window=window).mean()
    
    # Incorporate Close-to-Close Return
    df['Close_to_Close_Return'] = df['close'] - df['close'].shift(1)
    df['Multiplied_VWMA'] = df['VWMA_High_Low_Spread'] * df['Close_to_Close_Return']
    
    # Adjust with Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Final_Factor'] = df['Multiplied_VWMA'] * df['Volume_Change']
    
    return df['Final_Factor']
```
