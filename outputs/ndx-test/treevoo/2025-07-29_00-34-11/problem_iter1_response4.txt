```python
import pandas as pd

def heuristics_v2(df):
    period = 14
    
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['gain'] = 0.0
    df['loss'] = 0.0
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            df['gain'].iloc[i] = df['close'].iloc[i] - df['close'].iloc[i-1]
        else:
            df['loss'].iloc[i] = df['close'].iloc[i-1] - df['close'].iloc[i]
    
    # Aggregate Gains and Losses
    df['sum_gain'] = df['gain'].rolling(window=period).sum()
    df['sum_loss'] = df['loss'].rolling(window=period).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['sum_gain'] / df['sum_loss']
    
    # Convert to RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Adjust for Advanced Features
    df['avg_volume'] = df['volume'].rolling(window=period).mean()
    df['momentum'] = (df['close'] - df['open']) / df['open']
    
    # Advanced RSI
    df['arsi'] = df['rsi'] * df['avg_volume'] * df['momentum']
    
    return df['arsi']
```
