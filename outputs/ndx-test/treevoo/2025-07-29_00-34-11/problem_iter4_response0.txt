import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['close'] / df['open']
    
    # Average High-Low and Open-Close Ratios
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Compute Intraday Volatility
    price_range = df['high'] - df['low']
    intraday_volatility = df['volume'] * price_range
    
    # Compare Intraday Movement Ratio with Volatility
    alignment = intraday_movement_ratio - (intraday_volatility / df['volume'])
    
    # Determine Sentiment
    sentiment = alignment.apply(lambda x: 1 if x > 0 else -1)
    
    return sentiment
