import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Logarithmic Return
    short_term_log_return = np.log(df['close'] / df['close'].shift(5))
    
    # Calculate Long-Term Logarithmic Return
    long_term_log_return = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    volume_weighted_short_term_log_return = short_term_log_return * df['volume']
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    volume_weighted_long_term_log_return = long_term_log_return * df['volume']
    
    # Calculate Short-Term Volatility (Average True Range over last 5 days)
    true_range = df[['high', 'low', 'close']].join(df[['high', 'low', 'close']].shift(1), rsuffix='_prev')
    true_range['tr'] = np.maximum(np.maximum(true_range['high'] - true_range['low'], 
                                             np.abs(true_range['high'] - true_range['close_prev'])), 
                                  np.abs(true_range['low'] - true_range['close_prev']))
    short_term_volatility = true_range['tr'].rolling(window=5).mean()
    
    # Adjust for Volatility
    volatility_adjusted_return = (volume_weighted_short_term_log_return / short_term_volatility) - volume_weighted_long_term_log_return
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_intraday_volatility_return = intraday_volatility * close_to_close_return
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Aggregate Volume Changes (Sum over the last M days, M = 5)
    aggregated_volume_changes = daily_volume_change.rolling(window=5).sum()
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] / df['open'] - 1
    
    # Adjust for Intraday Momentum
    adjusted_intraday_momentum = (volatility_adjusted_return + 
                                  combined_intraday_volatility_return +
                                  aggregated_volume_changes * intraday_momentum)
    
    # Final Adjustment
    final_adjustment = adjusted_intraday_momentum * aggregated_volume_changes
    
    # Introduce a new factor: Volume-Weighted Intraday Volatility
    volume_weighted_intraday_volatility = intraday_volatility * df['volume']
    
    # Final Alpha Factor
    alpha_factor = final_adjustment + volume_weighted_intraday_volatility
    
    return alpha_factor
