```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Gain and Loss
    df['gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['loss'] = (df['close'] <= df['close'].shift(1)) * (df['close'].shift(1) - df['close'])
    
    # Aggregate Gains and Losses over a 14-day period
    gain_sum = df['gain'].rolling(window=14).sum()
    loss_sum = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = gain_sum / loss_sum
    
    # Convert to ARSI
    rsi = 100 - (100 / (1 + rs))
    arsi = rsi * df['volume'].rolling(window=14).mean() * (df['close'] / df['open'])
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'] / df['close'].shift(5) - 1
    long_term_return = df['close'] / df['close'].shift(20) - 1
    
    # Volume-Weighted Returns
    vol_weighted_short_term_return = short_term_return * df['volume']
    vol_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    tr = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    short_term_volatility = tr.rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_short_term_return = vol_weighted_short_term_return / short_term_volatility
    
    # Calculate Price Oscillator
    price_oscillator = df['close'].shift(5) - df['close'].shift(20)
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    combined_factor = arsi * adjusted_short_term_return + price_oscillator - vol_weighted_long_term_return
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    multiplied_vwma = vwma_high_low_spread * close_to_close_return
    
    # Adjust with Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    factor = multiplied_vwma * volume_change
    
    return combined_factor + factor
```
