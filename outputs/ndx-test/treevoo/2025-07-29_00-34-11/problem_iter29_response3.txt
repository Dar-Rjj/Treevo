import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = (df['high'] - df['low']) / df['low']
    open_close_ratio = (df['close'] - df['open']) / df['open']
    intraday_price_movement = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    alignment = intraday_price_movement / intraday_volatility
    sentiment = alignment.apply(lambda x: 1 if x > 0 else -1)
    
    # Incorporate Momentum into Sentiment
    momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    adjusted_sentiment = sentiment * (momentum + 1)
    
    # Calculate Gain and Loss
    gain = (df['close'] - df['close'].shift(1)).apply(lambda x: x if x > 0 else 0)
    loss = (df['close'] - df['close'].shift(1)).apply(lambda x: -x if x < 0 else 0)
    
    # Aggregate Gains and Losses
    sum_gains = gain.rolling(window=14).sum()
    sum_losses = loss.rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = sum_gains / sum_losses.replace(0, 1)
    
    # Convert to ARSI
    arsi = 100 - (100 / (1 + rs))
    arsi = arsi * df['volume'].rolling(window=14).mean() * ((df['close'] - df['open']) / df['open'])
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = (df['close'].shift(1) - df['close'].shift(5)) / df['close'].shift(5)
    long_term_return = (df['close'].shift(1) - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume-Weighted Returns
    volume_weighted_short_term_return = short_term_return * df['volume']
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    true_range = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / true_range
    
    # Calculate Price Oscillator
    price_oscillator = df['close'].shift(5) - df['close'].shift(20)
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    combined_factor = (arsi * adjusted_short_term_return) + price_oscillator - volume_weighted_long_term_return
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    weighted_high_low_spread = high_low_spread * df['volume']
    vwma = weighted_high_low_spread.rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    adjusted_vwma = vwma * close_to_close_return
    
    # Adjust with Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_vwma = adjusted_vwma * (1 + volume_change)
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / (df['high'] + df['low'])
    weighted_intraday_momentum = intraday_momentum * df['volume']
    
    # Calculate Extended Momentum
    avg_5_day_momentum = weighted_intraday_momentum.rolling(window=5).mean()
    avg_20_day_momentum = weighted_intraday_momentum.rolling(window=20).mean()
    
    # Enhance Momentum Signal
    momentum_difference = avg_5_day_momentum - avg_20_day_momentum
    adjusted_momentum = momentum_difference * (arsi / 100)
    
    # Integrate All Factors
    integrated_alpha_factor = combined_factor + adjusted_momentum
    
    return integrated_alpha_factor
