```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['Price_Diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['Gain'] = 0
    df['Loss'] = 0
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'Gain'] = df.loc[df.index[i], 'close'] - df.loc[df.index[i-1], 'close']
        else:
            df.loc[df.index[i], 'Loss'] = df.loc[df.index[i-1], 'close'] - df.loc[df.index[i], 'close']
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    df['Sum_Gain'] = df['Gain'].rolling(window=14).sum()
    df['Sum_Loss'] = df['Loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['RS'] = df['Sum_Gain'] / df['Sum_Loss'].replace(0, 1)  # Avoid division by zero
    
    # Convert to ARSI
    df['ARSI'] = 100 - (100 / (1 + df['RS']))
    df['ARSI_VW'] = df['ARSI'] * df['volume'].rolling(window=14).mean()
    
    # Calculate Intraday Price Movement Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    df['Open_Close_Ratio'] = df['close'] / df['open']
    df['Intraday_Movement_Ratio'] = (df['High_Low_Ratio'] + df['Open_Close_Ratio']) / 2
    
    # Measure Price-Volatility Alignment
    df['Intraday_Volatility'] = (df['high'] - df['low']) * df['volume']
    df['Alignment_Sentiment'] = df['Intraday_Movement_Ratio'] - df['Intraday_Volatility']
    
    # Calculate Short-Term and Long-Term Returns
    df['Short_Term_Return'] = df['close'] / df['close'].shift(5) - 1
    df['Long_Term_Return'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume-Weighted Returns
    df['Vol_Weighted_Short_Return'] = df['volume'] * df['Short_Term_Return']
    df['Vol_Weighted_Long_Return'] = df['volume'] * df['Long_Term_Return']
    
    # Adjust for Volatility and Sentiment
    df['Adjusted_Short_Return'] = df['Vol_Weighted_Short_Return'] / df['Intraday_Volatility']
    df['Adjusted_Long_Return'] = df['Vol_Weighted_Long_Return']
    df['Sentiment_Adjusted_Return'] = df['Adjusted_Short_Return'] - df['Adjusted_Long_Return'] + df['Alignment_Sentiment']
    
    # Incorporate Momentum
    df['Close_to_Close_Return'] = df['close'].pct_change()
    df['Momentum_Adjusted_Return'] = df['Sentiment_Adjusted_Return'] * df['Close_to_Close_Return'].fillna(0)
    
    # Final Volume Adjustment
    df['Final_Factor'] = df['Momentum_Adjusted_Return'] / df['volume']
    
    return df['Final_Factor']
```
