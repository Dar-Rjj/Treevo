```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on price and volume information.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
                       and indexed by date.
    
    Returns:
    pd.Series: A series of the calculated alpha factor, indexed by date.
    """
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the weighted average of open and close prices
    avg_price = (df['open'] + 2 * df['close']) / 3
    
    # Calculate the percentage change in volume
    vol_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the alpha factor
    alpha_factor = (price_diff / avg_price) * vol_change
    
    return alpha_factor
```
