import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the money flow, which is a measure of buying and selling pressure
    df['money_flow'] = df['close'] * df['volume']
    
    # Calculate the difference between high and low to capture the daily volatility
    df['daily_volatility'] = df['high'] - df['low']
    
    # Calculate a simple moving average of the close price over 10 days
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Create a heuristic factor by combining the above calculated features
    df['heuristic_factor'] = (df['close_change'] + (df['money_flow'] / df['sma_10'])) * df['daily_volatility']
    
    # Return the factor as a pandas Series
    return df['heuristic_factor']
