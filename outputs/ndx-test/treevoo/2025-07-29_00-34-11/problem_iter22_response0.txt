```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    avg_ratios = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    price_range = df['high'] - df['low']
    intraday_volatility = price_range * df['volume']
    
    # Evaluate Alignment
    alignment_diff = avg_ratios - intraday_volatility
    sentiment_indicator = pd.Series(0, index=df.index)
    sentiment_indicator[alignment_diff > 0] = 1  # Positive if High Alignment
    sentiment_indicator[alignment_diff < 0] = -1  # Negative if Low Alignment
    
    return sentiment_indicator
```
