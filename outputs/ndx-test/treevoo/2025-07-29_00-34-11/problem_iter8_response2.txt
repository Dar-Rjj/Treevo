import pandas as pd

def heuristics_v2(df, window=14):
    # Calculate Daily High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    df['Volume_Weighted_HL'] = df['High_Low_Spread'] * df['volume']
    df['VWMA_HL'] = df['Volume_Weighted_HL'].rolling(window=window).mean()
    
    # Incorporate Price Momentum
    df['Close_to_Close_Return'] = df['close'] / df['close'].shift(1) - 1
    df['Momentum_Adjusted_VWMA_HL'] = df['VWMA_HL'] * df['Close_to_Close_Return']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Adjust Intraday Volatility with Price Momentum
    df['Momentum_Adjusted_Intraday_Volatility'] = df['Intraday_Volatility'] * df['Close_to_Close_Return']
    
    # Synthesize Volume-Weighted and Intraday Volatility Factors
    df['Synthesized_Factor'] = (df['Momentum_Adjusted_VWMA_HL'] + df['Momentum_Adjusted_Intraday_Volatility']) / df['volume']
    
    # Enhance with Additional Price Dynamics
    df['Open_to_Close_Return'] = df['open'] / df['close'].shift(1) - 1
    df['Enhanced_Factor'] = df['Synthesized_Factor'] * df['Open_to_Close_Return']
    
    # Determine Gain and Loss
    df['Gain'] = df.apply(lambda x: x['close'] - x['close'].shift(1) if x['close'] > x['close'].shift(1) else 0, axis=1)
    df['Loss'] = df.apply(lambda x: x['close'].shift(1) - x['close'] if x['close'] < x['close'].shift(1) else 0, axis=1)
    
    # Aggregate Gains and Losses
    df['Sum_Gain'] = df['Gain'].rolling(window=window).sum()
    df['Sum_Loss'] = df['Loss'].rolling(window=window).sum()
    
    # Calculate Relative Strength
    df['Relative_Strength'] = df['Sum_Gain'] / df['Sum_Loss'].replace(0, 1)
    
    # Convert to ARSI
    df['ARSI'] = 100 - (100 / (1 + df['Relative_Strength']))
    df['ARSI'] = df['ARSI'] * df['volume'].rolling(window=window).mean() * df['Close_to_Open_Return']
    
    # Compute VWMA of High-Low Spread
    df['VWMA_HL_Spread'] = df['Volume_Weighted_HL'].rolling(window=window).mean()
    
    # Incorporate Close-to-Close Return
    df['Momentum_Adjusted_VWMA_HL'] = df['VWMA_HL_Spread'] * df['Close_to_Close_Return']
    
    # Calculate Short-Term Return
    df['Short_Term_Return'] = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    df['Long_Term_Return'] = df['close'].pct_change(periods=20)
    
    # Calculate Volume-Weighted Short-Term Return
    df['Volume_Weighted_Short_Term_Return'] = df['Short_Term_Return'] * df['volume']
    
    # Calculate Volume-Weighted Long-Term Return
    df['Volume_Weighted_Long_Term_Return'] = df['Long_Term_Return'] * df['volume']
    
    # Calculate Short-Term Volatility
    df['Short_Term_Volatility'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    df['Short_Term_Volatility'] = df['Short_Term_Volatility'].fillna(1)  # Avoid division by zero
    
    # Adjust for Volatility
    df['Adjusted_Short_Term_Return'] = df['Volume_Weighted_Short_Term_Return'] / df['Short_Term_Volatility']
