import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    avg_ratio = (high_low_ratio + open_close_ratio) / 2
    volume_weighted_ratios = avg_ratio * df['volume']

    # Measure Price-Volatility Alignment
    intraday_volatility = df['high'] - df['low']
    sentiment = (avg_ratio - intraday_volatility).apply(lambda x: 1 if x > 0 else -1)

    # Incorporate Momentum
    close_to_close_return = df['close'].pct_change()
    volume_weighted_momentum = volume_weighted_ratios * close_to_close_return

    # Analyze Price Action Patterns
    relative_closing_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    true_range = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    upper_shadow_length = df['high'] - df[['open', 'close']].max(axis=1)
    lower_shadow_length = df[['open', 'close']].min(axis=1) - df['low']

    # Investigate Volume-Related Factors
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_weighted_close = df['close'] * df['volume']

    # Derive Momentum Indicators from Price Data
    sma_close = df['close'].rolling(window=5).mean()
    roc = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    directional_movement = df['close'].diff()

    # Integrate Amount into Analysis
    amount_volume_ratio = df['amount'] / df['volume']
    amount_price_relationship = (df['amount'] / df['volume']) * df['close']

    # Combine Volume-Weighted Ratios and Momentum
    adjusted_volume_weighted_ratios = volume_weighted_ratios * sma_close
    adjusted_close_to_close_return = close_to_close_return * roc

    # Synthesize Price Action and Volume-Related Factors
    integrated_price_action = relative_closing_price_change + true_range
    combined_shadows_volume_ratio = (upper_shadow_length + lower_shadow_length) * volume_ratio

    # Enhance Momentum Indicators with Volume-Weighted Close
    adjusted_sma = sma_close * (volume_weighted_close / df['close'])
    adjusted_roc = roc * (volume_weighted_close / df['close'])
    adjusted_directional_movement = directional_movement * (volume_weighted_close / df['close'])

    # Final Alpha Factor
    alpha_factor = (
        volume_weighted_momentum * 
        sentiment * 
        integrated_price_action * 
        combined_shadows_volume_ratio * 
        adjusted_sma * 
        adjusted_roc * 
        adjusted_directional_movement
    )

    return alpha_factor
