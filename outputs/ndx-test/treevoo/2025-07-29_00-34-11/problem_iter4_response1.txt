```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    weighted_high_low_spread = high_low_spread * df['volume']
    vwma_high_low_spread = (weighted_high_low_spread.rolling(window=20).sum() /
                            df['volume'].rolling(window=20).sum())
    
    # Compute Close-to-Close Return
    close_to_close_return = df['close'].diff()
    
    # Multiply VWMA by Close-to-Close Return
    alpha_factor = vwma_high_low_spread * close_to_close_return
    
    return alpha_factor
```
