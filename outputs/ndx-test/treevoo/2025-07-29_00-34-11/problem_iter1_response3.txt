```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Combine Ratios: Average High-Low and Open-Close Ratios
    intraday_price_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Calculate Intraday Volatility
    price_range = df['high'] - df['low']
    weighted_price_range = price_range * df['volume']
    
    # Evaluate Alignment: Compare Intraday Movement Ratio with Volatility
    alignment = intraday_price_movement_ratio - (weighted_price_range / df['close'])
    
    # Determine Sentiment
    sentiment = alignment.apply(lambda x: 1 if x > 0 else -1)
    
    return sentiment
```
