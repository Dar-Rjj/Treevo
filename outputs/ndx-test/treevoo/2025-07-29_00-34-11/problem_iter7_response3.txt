```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Calculate Intraday Volatility
    price_range = df['high'] - df['low']
    intraday_volatility = price_range * df['volume']
    
    # Compare Movement and Volatility to Determine Sentiment
    def determine_sentiment(row):
        if row['high_low_ratio'] > 1 and row['open_close_ratio'] > 1:
            return 1  # Positive sentiment
        elif row['high_low_ratio'] < 1 and row['open_close_ratio'] < 1:
            return -1  # Negative sentiment
        else:
            return 0  # Neutral or mixed sentiment
    
    # Create a DataFrame with the computed ratios
    ratios_df = pd.DataFrame({
        'high_low_ratio': high_low_ratio,
        'open_close_ratio': open_close_ratio,
        'intraday_volatility': intraday_volatility
    })
    
    # Apply the sentiment determination function
    sentiment_factor = ratios_df.apply(determine_sentiment, axis=1)
    
    return sentiment_factor
```
