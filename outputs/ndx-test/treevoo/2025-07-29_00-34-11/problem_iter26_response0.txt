```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_ratio = (df['high'] - df['low']) / df['low']
    open_close_ratio = (df['close'] - df['open']) / df['open']
    intraday_price_movement = (high_low_ratio + open_close_ratio) / 2

    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    
    # Evaluate Alignment
    alignment = intraday_price_movement / intraday_volatility
    
    # Determine Sentiment
    sentiment = alignment.copy()
    sentiment[alignment > 0] = 1
    sentiment[alignment <= 0] = -1
    
    # Incorporate Momentum
    momentum = (df['close'].diff() / df['close'].shift(1)).fillna(0)
    
    # Adjust Sentiment based on Momentum
    adjusted_sentiment = sentiment + momentum
    adjusted_sentiment[adjusted_sentiment > 0] = 1
    adjusted_sentiment[adjusted_sentiment <= 0] = -1
    
    return adjusted_sentiment
```
