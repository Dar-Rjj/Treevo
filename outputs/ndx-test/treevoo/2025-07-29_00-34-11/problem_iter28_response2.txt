```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    df['Open_Close_Ratio'] = df['open'] / df['close']
    df['Intraday_Price_Movement_Ratio'] = (df['High_Low_Ratio'] + df['Open_Close_Ratio']) / 2

    # Measure Price-Volatility Alignment
    df['Intraday_Volatility'] = (df['high'] - df['low']) * df['volume']
    df['Alignment'] = df['Intraday_Price_Movement_Ratio'] / df['Intraday_Volatility']
    df['Sentiment'] = np.where(df['Alignment'] > 0, 1, -1)

    # Calculate Gain and Loss
    df['Gain'] = np.maximum(df['close'].diff(), 0)
    df['Loss'] = np.abs(np.minimum(df['close'].diff(), 0))

    # Aggregate Gains and Losses
    n_days = 14
    df['Sum_Gains'] = df['Gain'].rolling(window=n_days).sum()
    df['Sum_Losses'] = df['Loss'].rolling(window=n_days).sum()

    # Calculate Relative Strength
    df['RS'] = df['Sum_Gains'] / df['Sum_Losses']

    # Convert to RSI
    df['ARSI'] = 100 - (100 / (1 + df['RS']))
    df['ARSI_Volume_Adjusted'] = df['ARSI'] * df['volume'].rolling(window=n_days).mean()

    # Calculate Short-Term Logarithmic Return
    df['Short_Term_Log_Return'] = np.log(df['close'] / df['close'].shift(5))

    # Calculate Long-Term Logarithmic Return
    df['Long_Term_Log_Return'] = np.log(df['close'] / df['close'].shift(20))

    # Calculate Volume-Weighted Short-Term Logarithmic Return
    df['Volume_Weighted_Short_Term_Log_Return'] = df['Short_Term_Log_Return'] * df['volume']

    # Calculate Volume-Weighted Long-Term Logarithmic Return
    df['Volume_Weighted_Long_Term_Log_Return'] = df['Long_Term_Log_Return'] * df['volume']

    # Adjust for Volatility
    df['Short_Term_Volatility'] = df[['high', 'low']].rolling(window=5).apply(lambda x: (x.max() - x.min()).mean(), raw=True)
    df['Adjusted_Volume_Weighted_Short_Term_Log_Return'] = df['Volume_Weighted_Short_Term_Log_Return'] / df['Short_Term_Volatility']

    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['VWMA_High_Low_Spread'] = (df['High_Low_Spread'] * df['volume']).rolling(window=14).mean()

    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['Sentiment'] 
        + df['ARSI_Volume_Adjusted'] 
        + df['Volume_Weighted_Short_Term_Log_Return'] 
        + df['Volume_Weighted_Long_Term_Log_Return'] 
        + df['Adjusted_Volume_Weighted_Short_Term_Log_Return'] 
        + df['VWMA_High_Low_Spread']
    )

    return alpha_factor
```
