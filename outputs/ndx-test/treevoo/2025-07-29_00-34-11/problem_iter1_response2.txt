```python
def heuristics_v2(data, n=20, m=10, momentum_weight=1.0, volume_acceleration_weight=1.0):
    # Calculate Price Momentum
    data['price_momentum'] = data['close'].diff(n)
    
    # Calculate Volume Acceleration
    data['volume_acceleration'] = data['volume'].pct_change(m).add(1)  # Convert to ratio and handle division by zero
    
    # Combine Momentum and Acceleration
    data['factor_value'] = (data['price_momentum'] * momentum_weight + 
                            data['volume_acceleration'] * volume_acceleration_weight)
    
    return data['factor_value']
```
