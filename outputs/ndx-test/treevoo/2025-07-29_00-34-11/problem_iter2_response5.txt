```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_vol'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['combined'] = df['intraday_vol'] * df['close_to_close_return']
    
    # Calculate Short-Term Return (5 days)
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1
    
    # Calculate Long-Term Return (20 days)
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['volume'] * df['short_term_return']
    
    # Calculate Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['volume'] * df['long_term_return']
    
    # Adjust for Volume: Divide the result of Intraday Volatility and Close-to-Close Return combination by Volume
    df['volume_adjusted_combined'] = df['combined'] / df['volume']
    
    # Calculate Short-Term Volatility (using High and Low, average true range over last 5 days)
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    short_term_volatility = df['true_range'].rolling(window=5).mean()
    
    # Final Adjustment
    df['final_factor'] = (df['volume_adjusted_combined'] / short_term_volatility) - df['volume_weighted_long_term_return']
    
    return df['final_factor']
```
