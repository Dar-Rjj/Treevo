```python
def heuristics_v2(df):
    # Calculate the weighted average of open, high, low, and close prices
    df['weighted_avg'] = (df['open'] + 2*df['high'] + 2*df['low'] + df['close']) / 6
    
    # Calculate the percentage change in the weighted average price
    df['pct_change_weighted_avg'] = df['weighted_avg'].pct_change()
    
    # Calculate the volume change
    df['volume_change'] = df['volume'].diff()
    
    # Heuristic factor: product of the percentage change in weighted average and the volume change
    heuristic_factor = df['pct_change_weighted_avg'] * df['volume_change']
    
    return heuristic_factor.dropna()
```
