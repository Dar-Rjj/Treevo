```python
import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    volume_weighted_ratios = (high_low_ratio + open_close_ratio) / 2 * df['volume']

    # Measure Price-Volatility Alignment
    intraday_volatility = df['high'] - df['low']
    price_volatility_alignment = volume_weighted_ratios / intraday_volatility
    sentiment = price_volatility_alignment.apply(lambda x: 1 if x > 0 else -1)

    # Incorporate Momentum
    close_to_close_return = df['close'].pct_change()
    momentum_adjusted_ratios = volume_weighted_ratios * close_to_close_return

    # Analyze Price Action Patterns
    relative_closing_price_change = (df['close'].diff() / df['close'].shift(1)).fillna(0)
    true_range = (df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1).shift(1)).fillna(0)
    upper_shadow_length = df['high'] - df[['open', 'close']].max(axis=1)
    lower_shadow_length = df[['open', 'close']].min(axis=1) - df['low']

    # Investigate Volume-Related Factors
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_weighted_close_price = df['close'] * df['volume']

    # Derive Momentum Indicators from Price Data
    n = 5  # Period for SMA and ROC
    sma_close = df['close'].rolling(window=n).mean()
    roc = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    directional_movement = df['close'].diff().fillna(0)

    # Combine all factors into a single alpha factor
    alpha_factor = (
        sentiment * momentum_adjusted_ratios +
        relative_closing_price_change +
        (upper_shadow_length - lower_shadow_length) / true_range +
        volume_ratio * volume_weighted_close_price +
        sma_close * roc * directional_movement
    )

    return alpha_factor
```
