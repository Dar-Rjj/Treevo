```python
def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Compute Intraday Volatility (Price Range weighted by Volume)
    price_range = df['high'] - df['low']
    intraday_volatility = price_range * df['volume']
    
    # Evaluate Alignment
    alignment_score = (high_low_ratio + open_close_ratio) / intraday_volatility
    
    # Determine Sentiment
    sentiment = alignment_score.apply(lambda x: 1 if x > 0 else -1)
    
    return sentiment
```
