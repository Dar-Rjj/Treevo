```python
import pandas as pd

def heuristics_v2(df, window=10):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low = (high_low_spread * df['volume']).rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'] - df['close'].shift(1)
    vwma_momentum = vwma_high_low * close_to_close_return
    
    # Combine Intraday Volatility and Close-to-Close Return
    intraday_volatility = df['high'] - df['low']
    volatility_adjusted_return = intraday_volatility * close_to_close_return
    
    # Synthesize Combined Factors
    combined_factor = (vwma_momentum + volatility_adjusted_return) / df['volume']
    
    return combined_factor
```
