import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']

    # Determine Gain and Loss
    df['gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['loss'] = (df['close'] <= df['close'].shift(1)) * (df['close'].shift(1) - df['close'])

    # Aggregate Gains and Losses over a 14-day period
    gain_sum = df['gain'].rolling(window=14).sum()
    loss_sum = df['loss'].rolling(window=14).sum()

    # Calculate Relative Strength
    rs = gain_sum / loss_sum

    # Convert to ARSI
    rsi = 100 - (100 / (1 + rs))
    arsi = rsi * df['volume'].rolling(window=14).mean()

    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2

    # Measure Price-Volatility Alignment
    intraday_volatility = df['volume'] * df['price_diff']
    alignment = intraday_movement_ratio / intraday_volatility

    # Determine Sentiment
    sentiment = alignment.apply(lambda x: 'positive' if x > 0 else 'negative')

    # Calculate Short-Term Return
    short_term_return = df['close'] / df['close'].shift(5) - 1

    # Calculate Long-Term Return
    long_term_return = df['close'] / df['close'].shift(20) - 1

    # Calculate Volume-Weighted Returns
    volume_weighted_short_term = df['volume'] * short_term_return
    volume_weighted_long_term = df['volume'] * long_term_return

    # Adjust for Volatility and Sentiment
    short_term_volatility = df['price_diff'].rolling(window=5).std()
    adjusted_ratio = (volume_weighted_short_term / short_term_volatility) - volume_weighted_long_term

    # Incorporate Momentum
    close_to_close_return = df['close'] - df['close'].shift(1)
    momentum_adjusted = adjusted_ratio * close_to_close_return

    # Final Volume Adjustment
    final_factor = momentum_adjusted / df['volume']

    return final_factor
