```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    df['High_Low_Ratio'] = df['high'] / df['low']
    df['Open_Close_Ratio'] = df['open'] / df['close']
    df['Intraday_Movement_Ratio'] = (df['High_Low_Ratio'] + df['Open_Close_Ratio']) / 2

    # Measure Price-Volatility Alignment
    df['Intraday_Volatility'] = (df['high'] - df['low']) * df['volume']
    df['Alignment'] = df['Intraday_Movement_Ratio'] / df['Intraday_Volatility']

    # Calculate Gain and Loss
    df['Gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['Loss'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)

    # Aggregate Gains and Losses
    df['Sum_Gains'] = df['Gain'].rolling(window=14).sum()
    df['Sum_Losses'] = df['Loss'].rolling(window=14).sum()

    # Calculate Relative Strength
    df['RS'] = df['Sum_Gains'] / df['Sum_Losses']

    # Convert to ARSI
    df['ARSI'] = 100 - (100 / (1 + df['RS']))
    df['ARSI_Adjusted'] = df['ARSI'] * df['volume'].rolling(window=14).mean() * (df['close'] / df['open'])

    # Calculate Short-Term and Long-Term Logarithmic Returns
    df['Log_Return_5D'] = np.log(df['close'] / df['close'].shift(5))
    df['Log_Return_20D'] = np.log(df['close'] / df['close'].shift(20))

    # Volume-Weighted Logarithmic Returns
    df['VW_Log_Return_5D'] = df['volume'] * df['Log_Return_5D']
    df['VW_Log_Return_20D'] = df['volume'] * df['Log_Return_20D']

    # Short-Term Volatility
    df['True_Range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['Short_Term_Volatility'] = df['True_Range'].rolling(window=5).mean()

    # Adjust for Volatility
    df['Adjusted_VW_Log_Return_5D'] = df['VW_Log_Return_5D'] / df['Short_Term_Volatility']

    # Price Oscillator
    df['Price_Oscillator'] = df['Log_Return_5D'] - df['Log_Return_20D']

    # Combine ARSI, Adjusted Returns, and Price Oscillator
    df['Combined_Factor'] = df['ARSI_Adjusted'] * df['Adjusted_VW_Log_Return_5D'] + df['Price_Oscillator'] - df['VW_Log_Return_20D']

    # High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']

    # VWMA of High-Low Spread
    df['VW_High_Low_Spread'] = df['High_Low_Spread'] * df['volume']
    df['VWMA_High_Low_Spread'] = df['VW_High_Low_Spread'].rolling(window=14).mean() / df['volume'].rolling(window=14).mean()

    # Incorporate Close-to-Close Return
    df['Close_to_Close_Return'] = df['close'].pct_change()
    df['Factor'] = df['VWMA_High_Low_Spread'] * df['Close_to_Close_Return']

    # Introduce Volume-Weighted Price Change
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['VW_Price_Change'] = df['Price_Change'] * df['volume']
    df['SMA_VW_Price_Change'] = df['VW_Price_Change'].rolling(window=14).mean() / df['volume'].rolling(window=14).mean()

    # Final Alpha Factor
    df['Final_Factor'] = df['Combined_Factor'] + df['Factor'] + df['SMA_VW_Price_Change']

    return df['Final_Factor']
```
