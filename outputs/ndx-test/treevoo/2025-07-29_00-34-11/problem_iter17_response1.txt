```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', and 'volume'.
    
    Returns:
    pd.Series: Series with the computed alpha factor.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change from open to close
    pct_change_open_close = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume weighted price
    vwap = (df['close'] + df['high'] + df['low']) / 3 * df['volume']
    vwap = vwap / df['volume'].rolling(window=20).mean()
    
    # Combine the factors
    factor = price_range * pct_change_open_close * vwap
    
    return factor
```
