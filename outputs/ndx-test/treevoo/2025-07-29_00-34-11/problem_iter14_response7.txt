```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement Intensity
    high_low_spread = df['high'] - df['low']
    open_close_change = df['close'] - df['open']
    intraday_movement_intensity = high_low_spread + open_close_change

    # Measure Momentum-Volatility Correlation
    intraday_volatility = (df['high'] - df['low']) / df['volume']
    momentum_volatility_corr = np.where(intraday_movement_intensity > intraday_volatility, 1, -1)

    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    intraday_volatility_weighted = (df['high'] - df['low']) * df['volume']
    sentiment = np.where(intraday_movement_ratio > intraday_volatility_weighted, 1, -1)

    # Calculate Short-Term Logarithmic Return
    short_term_log_return = np.log(df['close'] / df['close'].shift(5))

    # Calculate Long-Term Logarithmic Return
    long_term_log_return = np.log(df['close'] / df['close'].shift(20))

    # Calculate Volume-Weighted Short-Term Logarithmic Return
    volume_weighted_short_term_return = df['volume'] * short_term_log_return

    # Calculate Volume-Weighted Long-Term Logarithmic Return
    volume_weighted_long_term_return = df['volume'] * long_term_log_return

    # Adjust for Volatility
    short_term_volatility = (df['high'].rolling(window=5).std() - df['low'].rolling(window=5).std()) / df['volume'].rolling(window=5).mean()
    adjusted_volatility = (volume_weighted_short_term_return / short_term_volatility) - volume_weighted_long_term_return

    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    price_volatility_alignment = np.where(intraday_movement_ratio > intraday_volatility, 1, -1)

    # Calculate Close-to-Close Return
    close_to_close_return = df['close'] / df['close'].shift(1) - 1

    # Combine Intraday Volatility and Close-to-Close Return
    combined_intraday_volatility_return = intraday_volatility * close_to_close_return

    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)

    # Aggregate Volume Changes
    aggregated_volume_changes = daily_volume_change.rolling(window=5).sum()

    # Calculate Intraday Momentum
    intraday_momentum = df['close'] / df['open'] - 1

    # Adjust for Intraday Momentum and Sentiment
    adjusted_factor = (
        adjusted_volatility + 
        combined_intraday_volatility_return * aggregated_volume_changes +
        intraday_momentum * sentiment
    )

    # Final Adjustment
    final_factor = adjusted_factor * aggregated_volume_changes

    return final_factor
```
