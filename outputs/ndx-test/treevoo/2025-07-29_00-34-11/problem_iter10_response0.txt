```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    price_diff = df['high'] - df['low']
    
    # Determine Gain and Loss
    gain = (df['close'].shift(-1) > df['close']).astype(int) * (df['close'].shift(-1) - df['close'])
    loss = (df['close'].shift(-1) <= df['close']).astype(int) * (df['close'] - df['close'].shift(-1))
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    period = 14
    sum_gains = gain.rolling(window=period).sum()
    sum_losses = loss.rolling(window=period).sum()
    
    # Calculate Relative Strength
    rs = sum_gains / sum_losses.replace(0, 1)  # Avoid division by zero
    
    # Convert to ARSI
    arsi = 100 - (100 / (1 + rs))
    arsi *= df['volume'].rolling(window=period).mean()  # Incorporate volume
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=period).sum() / df['volume'].rolling(window=period).sum()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'] - df['close'].shift(1)
    adjusted_vwma = vwma_high_low_spread * close_to_close_return
    
    # Combine ARSI, Adjusted Returns, and VWMA
    combined_factor = arsi * close_to_close_return * adjusted_vwma - df['close'].pct_change(periods=period)
    
    return combined_factor
```
