```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Momentum
    high_low_momentum = df['high'] - df['low']
    
    # Compute Open-Close Momentum
    open_close_momentum = df['close'] - df['open']
    
    # Aggregate Momentum Scores
    momentum_score = (high_low_momentum + open_close_momentum) / 2
    
    # Calculate Intraday Volatility
    price_range = df['high'] - df['low']
    weighted_volatility = price_range * df['volume']
    
    # Evaluate Alignment
    alignment = momentum_score / (weighted_volatility + 1e-6)  # add small constant to avoid division by zero
    
    return alignment
```
