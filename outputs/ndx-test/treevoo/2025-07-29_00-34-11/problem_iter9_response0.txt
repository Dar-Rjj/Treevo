```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(periods=20)
    
    # Calculate Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Calculate Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility (Average True Range over 5 days)
    tr = df[['high', 'low', 'close']].diff().abs().max(axis=1).rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_volatility = (volume_weighted_short_term_return / tr) - volume_weighted_long_term_return
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change(periods=1)
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_intraday_volatility = intraday_volatility * close_to_close_return
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] / df['open']) - 1
    
    # Calculate Open-to-High Ratio
    open_to_high_ratio = (df['high'] / df['open'])
    
    # Combine Open-to-Close Return and Open-to-High Ratio
    combined_open_to_close = open_to_close_return * open_to_high_ratio
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Final Adjustment
    final_factor = (adjusted_volatility + 
                    combined_intraday_volatility + 
                    combined_open_to_close + 
                    intraday_momentum)
    
    return final_factor
```
