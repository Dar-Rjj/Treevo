```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Logarithmic Return
    short_term_log_return = np.log(df['close'] / df['close'].shift(5))
    
    # Calculate Long-Term Logarithmic Return
    long_term_log_return = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    volume_weighted_short_term = df['volume'] * short_term_log_return
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    volume_weighted_long_term = df['volume'] * long_term_log_return
    
    # Calculate Short-Term Volatility (Average True Range over 5 days)
    tr = df[['high', 'low']].rolling(window=5).apply(lambda x: max(x) - min(x), raw=True)
    short_term_volatility = tr.rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_volume_weighted_short_term = volume_weighted_short_term / short_term_volatility['high']
    adjusted_for_volatility = adjusted_volume_weighted_short_term - volume_weighted_long_term
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_intraday_close_to_close = intraday_volatility * close_to_close_return
    
    # Aggregate Volume Changes
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    aggregated_volume_changes = daily_volume_change.rolling(window=5).sum()
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] / df['open'] - 1
    
    # Final Adjustment
    final_factor = (
        adjusted_for_volatility +
        combined_intraday_close_to_close *
        aggregated_volume_changes +
        intraday_momentum
    )
    
    return final_factor
```
