```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low = (high_low_spread * df['volume']).ewm(span=20).mean() / df['volume'].ewm(span=20).mean()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'].diff()
    vwma_momentum = vwma_high_low * close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Intraday Volatility for Open and Close
    open_close_spread = df['open'] - df['close']
    adjusted_intraday_volatility = intraday_volatility + open_close_spread
    
    # Combine Adjusted Intraday Volatility and Close-to-Close Return
    volatility_adjusted_momentum = adjusted_intraday_volatility * close_to_close_return
    
    # Synthesize Volume-Weighted and Adjusted Intraday Volatility Factors
    factor = (vwma_momentum + volatility_adjusted_momentum) / df['volume']
    
    return factor
```
