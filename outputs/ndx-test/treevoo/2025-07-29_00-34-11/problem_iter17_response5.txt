import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement Intensity
    high_low_spread = df['high'] - df['low']
    open_close_change = df['close'] - df['open']
    intraday_movement_intensity = high_low_spread + open_close_change

    # Measure Momentum-Volatility Correlation
    intraday_volatility = (df['high'] - df['low']) / df['volume']
    momentum_volatility_correlation = np.sign(intraday_movement_intensity * intraday_volatility)

    # Calculate Short-Term Logarithmic Return
    short_term_log_return = np.log(df['close'] / df['close'].shift(5))

    # Calculate Long-Term Logarithmic Return
    long_term_log_return = np.log(df['close'] / df['close'].shift(20))

    # Calculate Volume-Weighted Short-Term Logarithmic Return
    volume_weighted_short_term_log_return = df['volume'] * short_term_log_return

    # Calculate Volume-Weighted Long-Term Logarithmic Return
    volume_weighted_long_term_log_return = df['volume'] * long_term_log_return

    # Adjust for Volatility
    short_term_volatility = df['close'].pct_change().rolling(window=5).std()
    adjusted_ratio = (volume_weighted_short_term_log_return / short_term_volatility) - volume_weighted_long_term_log_return

    # Measure Price-Volatility Alignment
    price_range = df['high'] - df['low']
    intraday_volatility = price_range * df['volume']
    intraday_price_movement_ratio = (df['high'] / df['low'] + df['close'] / df['open']) / 2
    alignment = np.sign(intraday_price_movement_ratio - intraday_volatility)
    sentiment = np.where(alignment > 0, 1, -1)

    # Calculate Close-to-Close Return
    close_to_close_return = df['close'] / df['close'].shift(1) - 1

    # Combine Intraday Volatility and Close-to-Close Return
    combined_volatility_return = intraday_volatility * close_to_close_return

    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)

    # Aggregate Volume Changes
    aggregated_volume_changes = daily_volume_change.rolling(window=5).sum()

    # Incorporate Enhanced Momentum into Adjusted Ratios
    enhanced_momentum = volume_weighted_short_term_log_return * close_to_close_return
    final_volume_adjustment = enhanced_momentum / df['volume']

    # Calculate Intraday Momentum
    intraday_momentum = df['close'] / df['open'] - 1

    # Adjust for Intraday Momentum and Sentiment
    final_adjustment = (adjusted_ratio + combined_volatility_return) * aggregated_volume_changes + intraday_momentum * sentiment

    return final_adjustment
