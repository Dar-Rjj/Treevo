```python
def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pandas.DataFrame): DataFrame with index as date and columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
        pandas.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the price range for the day
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['low']) / price_range
    
    # Calculate the selling pressure
    selling_pressure = (df['high'] - df['close']) / price_range
    
    # Calculate the volume-weighted average of buying and selling pressures
    volume_weighted_average = (buying_pressure * selling_pressure) * df['volume']
    
    # Calculate the factor value
    factor_value = (buying_pressure - selling_pressure) * volume_weighted_average
    
    return factor_value
```
