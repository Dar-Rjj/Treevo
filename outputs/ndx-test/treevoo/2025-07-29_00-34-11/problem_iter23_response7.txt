```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between the close and open prices
    df['rel_diff'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the normalized volume, which is the volume divided by the mean volume
    df['norm_volume'] = df['volume'] / df['volume'].mean()
    
    # Calculate the high-low spread
    df['hl_spread'] = (df['high'] - df['low']) / df['low']
    
    # Combine the above heuristics into a single factor
    # This factor captures the momentum, volatility, and volume anomalies
    df['factor'] = df['rel_diff'] * df['norm_volume'] * df['hl_spread']
    
    return df['factor']
```
