import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Logarithmic Return
    short_term_log_return = df['close'].apply(lambda x: (x / df['close'].shift(5)).apply(np.log))
    
    # Calculate Long-Term Logarithmic Return
    long_term_log_return = df['close'].apply(lambda x: (x / df['close'].shift(20)).apply(np.log))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    volume_weighted_short_term_log_return = short_term_log_return * df['volume']
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    volume_weighted_long_term_log_return = long_term_log_return * df['volume']
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=5).mean()
    
    # Adjust VWMA by Volume
    adjusted_vwma = vwma_high_low_spread / df['volume']
    
    # Incorporate Price Momentum into Adjusted VWMA
    close_to_close_return = df['close'] - df['close'].shift(1)
    momentum_adjusted_vwma = adjusted_vwma * close_to_close_return
    
    # Combine Intraday Volatility and Close-to-Close Return
    intraday_volatility = df['high'] - df['low']
    combined_factor = (intraday_volatility * close_to_close_return) / df['volume']
    
    # Synthesize Combined Factors
    synthesized_factor = (momentum_adjusted_vwma + combined_factor) / df['volume']
    
    # Final Volume Adjustment
    final_volume_adjusted_factor = synthesized_factor / df['volume']
    
    # Calculate Short-Term Volatility
    short_term_volatility = (df[['high', 'low']].rolling(window=5).std()).mean(axis=1)
    
    # Adjust for Volatility
    volatility_adjusted_factor = (volume_weighted_short_term_log_return / short_term_volatility) - volume_weighted_long_term_log_return
    
    # Calculate Close-to-Close Return
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_intraday_volatility = (intraday_volatility * close_to_close_return)
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Aggregate Volume Changes
    aggregated_volume_changes = daily_volume_change.rolling(window=5).sum()
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] / df['open']) - 1
    
    # Adjust for Intraday Momentum
    final_adjusted_factor = (volatility_adjusted_factor + combined_intraday_volatility) * aggregated_volume_changes + intraday_momentum
    
    return final_adjusted_factor
