```python
import pandas as pd

def heuristics_v2(df, window=10):
    # Calculate Daily High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    df['vwma_high_low_spread'] = (df['high_low_spread'] * df['volume']).rolling(window=window).mean() / df['volume'].rolling(window=window).mean()
    
    # Incorporate Price Momentum: Close-to-Close Return
    df['close_to_close_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['volatility_adjusted_momentum'] = df['intraday_volatility'] * df['close_to_close_return']
    
    # Synthesize Factors
    df['alpha_factor'] = df['vwma_high_low_spread'] * df['close_to_close_return'] + df['volatility_adjusted_momentum']
    
    return df['alpha_factor']
```
