```python
def heuristics_v2(df):
    # Calculate High-Llow Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Combine Spread and Momentum
    combined_factor = high_low_spread * price_momentum
    
    # Volume and Amount Adjust Factor
    volume_amount_adjusted_factor = combined_factor / (df['volume'] * df['amount'])
    
    return volume_amount_adjusted_factor
```
