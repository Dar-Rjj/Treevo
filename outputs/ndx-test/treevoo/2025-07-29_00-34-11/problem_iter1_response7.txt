```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Compute Logarithmic Returns
    data['log_returns'] = np.log(data['close'] / data['close'].shift(1))
    
    # Aggregate Logarithmic Returns over the last 5 days
    data['price_momentum'] = data['log_returns'].rolling(window=5).sum()
    
    # Compute Daily Volume Change
    data['volume_change'] = data['volume'] - data['volume'].shift(1)
    
    # Aggregate Volume Changes over the last 5 days
    data['volume_surge'] = data['volume_change'].rolling(window=5).sum()
    
    # Combine Indicators by multiplying Price Momentum and Volume Surge
    data['alpha_factor'] = data['price_momentum'] * data['volume_surge']
    
    return data['alpha_factor']
```
