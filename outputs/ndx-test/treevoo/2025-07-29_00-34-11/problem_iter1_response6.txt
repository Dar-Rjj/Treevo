```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].diff()
    
    # Aggregate Daily Return (sum of daily return * volume over the past 20 days)
    df['volume_weighted_momentum'] = (df['daily_return'] * df['volume']).rolling(window=20).sum()
    
    # Calculate Long-Term Trend Strength (close price change over 60 days)
    df['long_term_trend'] = df['close'] - df['close'].shift(60)
    
    # Combine Factors
    df['combined_factor'] = df['volume_weighted_momentum'] + df['long_term_trend']
    
    # Adjust for Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(20)
    df['adjusted_factor'] = df.apply(lambda row: row['combined_factor'] * 1.5 if row['volume_change'] > 0 else row['combined_factor'] / 1.5, axis=1)
    
    return df['adjusted_factor'].dropna()
```
