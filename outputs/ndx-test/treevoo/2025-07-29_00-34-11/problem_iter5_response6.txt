```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['Price_Diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['Gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['Loss'] = (df['close'] <= df['close'].shift(1)) * (df['close'].shift(1) - df['close'])
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    df['Sum_Gain'] = df['Gain'].rolling(window=14).sum()
    df['Sum_Loss'] = df['Loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['RS'] = df['Sum_Gain'] / df['Sum_Loss'].replace(0, 1e-6)  # Avoid division by zero
    
    # Convert to ARSI
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    average_volume = df['volume'].rolling(window=14).mean()
    close_to_open_return = (df['close'] / df['open']) - 1
    df['ARSI'] = df['RSI'] * average_volume * close_to_open_return
    
    # Calculate Short-Term and Long-Term Returns
    df['Short_Term_Return'] = (df['close'] / df['close'].shift(5)) - 1
    df['Long_Term_Return'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Calculate Volume-Weighted Short-Term and Long-Term Returns
    df['Volume_Weighted_Short_Term_Return'] = df['volume'] * df['Short_Term_Return']
    df['Volume_Weighted_Long_Term_Return'] = df['volume'] * df['Long_Term_Return']
    
    # Calculate Short-Term Volatility
    df['True_Range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()  # |High - Low|
    df['Short_Term_Volatility'] = df['True_Range'].rolling(window=5).mean()
    
    # Synthesize ARSI with Volume-Weighted Returns and Volatility
    df['Combined_ARSI'] = df['ARSI'] * df['Volume_Weighted_Short_Term_Return'] / df['Short_Term_Volatility']
    df['Adjusted_ARSI'] = df['Combined_ARSI'] - df['Volume_Weighted_Long_Term_Return']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['Close_to_Close_Return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['Combined_Volatility_Return'] = df['Intraday_Volatility'] * df['Close_to_Close_Return']
    
    # Final Adjustment
    df['Final_Factor'] = df['Adjusted_ARSI'] + df['Combined_Volatility_Return']
    
    return df['Final_Factor']
```
