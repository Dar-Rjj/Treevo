import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    alignment = intraday_movement_ratio / intraday_volatility
    sentiment = np.where(alignment > 1, 1, -1)
    
    # Calculate Gain and Loss
    gain = np.maximum(df['close'].diff(), 0)
    loss = np.abs(np.minimum(df['close'].diff(), 0))
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    sum_gains = gain.rolling(window=14).sum()
    sum_losses = loss.rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = sum_gains / sum_losses
    arsi = 100 - (100 / (1 + rs))
    arsi = arsi * df['volume'].rolling(window=14).mean() * (df['close'] / df['open']).shift(1)
    
    # Calculate Short-Term Logarithmic Return
    short_term_log_return = np.log(df['close'] / df['close'].shift(5))
    
    # Calculate Long-Term Logarithmic Return
    long_term_log_return = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    volume_weighted_short_term_return = df['volume'] * short_term_log_return
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    volume_weighted_long_term_return = df['volume'] * long_term_log_return
    
    # Calculate Short-Term Volatility (Average True Range)
    atr_short_term = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / atr_short_term
    
    # Calculate Price Oscillator
    price_oscillator = short_term_log_return - long_term_log_return
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    combined_factor = arsi * adjusted_short_term_return + price_oscillator - volume_weighted_long_term_return
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    vwma_close_to_close = vwma_high_low_spread * close_to_close_return
    
    # Incorporate Volume-Weighted Momentum
    volume_weighted_momentum = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Integrate into Combined Factor
    final_factor = combined_factor + vwma_close_to_close + volume_weighted_momentum
    
    return final_factor.dropna()
