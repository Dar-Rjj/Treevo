```python
def heuristics_v2(data):
    # Calculate High-Low Spread
    high_low_spread = data['high'] - data['low']
    
    # Calculate Price Momentum
    price_momentum = (data['close'].shift(1) - data['close']) / data['close'].shift(1)
    
    # Combine Spread and Momentum
    combined_factor = high_low_spread * price_momentum
    
    # Volume Trend Adjustment
    volume_change = (data['volume'].shift(1) - data['volume']) / data['volume'].shift(1)
    volume_trend_adjusted_factor = combined_factor * volume_change
    
    # Final Volume Adjust Factor
    final_factor = volume_trend_adjusted_factor / data['volume']
    
    return final_factor
```
