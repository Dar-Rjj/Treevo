def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratios
    high_low_ratio = (df['high'] - df['low']) / df['low']
    open_close_ratio = (df['close'] - df['open']) / df['open']
    combined_ratio = (high_low_ratio + open_close_ratio) / 2

    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']

    # Evaluate Alignment
    alignment = (combined_ratio > intraday_volatility).astype(int) - (combined_ratio < intraday_volatility).astype(int)

    return alignment
