import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    volume_weighted_high_low_spread = high_low_spread * df['volume']
    vwma = volume_weighted_high_low_spread.rolling(window=20).mean()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'].pct_change()
    vwma_momentum = vwma * close_to_close_return
    
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Evaluate Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    
    # Determine Sentiment
    sentiment = (high_low_ratio - 1) * (intraday_volatility - intraday_volatility.shift(1))
    sentiment = sentiment.apply(lambda x: 1 if x > 0 else -1)
    
    # Adjust Intraday Volatility with Price Momentum
    intraday_volatility_momentum = intraday_volatility * close_to_close_return
    
    # Synthesize Volume-Weighted, Intraday Sentiment, and Intraday Volatility Factors
    synthesized_factor = (vwma_momentum + intraday_volatility_momentum * sentiment) / df['volume']
    
    # Enhance with Additional Price Dynamics
    open_to_close_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    final_factor = synthesized_factor * open_to_close_return
    
    return final_factor
