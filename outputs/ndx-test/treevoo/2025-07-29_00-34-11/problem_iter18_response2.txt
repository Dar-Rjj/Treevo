import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()
    
    # Moving average of daily returns over n days
    df['ma_daily_return'] = df['daily_return'].rolling(window=n).mean()
    
    # Difference between current daily return and its moving average
    df['return_diff_ma'] = df['daily_return'] - df['ma_daily_return']
    
    # Calculate volume change
    df['volume_change'] = df['volume'].diff()
    
    # Compute ratio of volume change to average volume over n days
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_change_ratio'] = df['volume_change'] / df['avg_volume']
    
    # High minus Low
    df['high_low_range'] = df['high'] - df['low']
    
    # Range as percentage of closing price
    df['high_low_range_pct'] = df['high_low_range'] / df['close']
    
    # Moving average of High-Low over n days
    df['ma_high_low_range'] = df['high_low_range'].rolling(window=n).mean()
    
    # Open to Close difference
    df['open_close_diff'] = df['close'] - df['open']
    
    # Measure the direction of price move
    df['price_move_direction'] = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Difference between today's and yesterday's close
    df['close_diff_yesterday'] = df['close'].diff()
    
    # Combine factors using a weighted sum
    weights = {'return_diff_ma': 0.4, 'volume_change_ratio': 0.3, 'open_close_diff': 0.3}
    df['combined_factor'] = (df['return_diff_ma'] * weights['return_diff_ma'] +
                             df['volume_change_ratio'] * weights['volume_change_ratio'] +
                             df['open_close_diff'] * weights['open_close_diff'])
    
    # Create a composite score by averaging selected individual factors
    selected_factors = ['return_diff_ma', 'volume_change_ratio', 'high_low_range_pct', 'open_close_diff']
    df['composite_score'] = df[selected_factors].mean(axis=1)
    
    # Return the combined factor as the alpha factor
    return df['combined_factor']

# Example usage:
