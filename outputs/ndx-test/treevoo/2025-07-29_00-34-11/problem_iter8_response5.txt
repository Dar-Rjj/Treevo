import pandas as pd

def heuristics_v2(df, window=20):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    volume_weighted_high_low = high_low_spread * df['volume']
    vwma_high_low = volume_weighted_high_low.rolling(window=window).mean()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'].pct_change()
    momentum_adjusted_vwma = vwma_high_low * close_to_close_return
    
    # Compute High-Low Ratio and Open-Close Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    
    # Evaluate Price-Volatility Alignment
    price_range = df['high'] - df['low']
    intraday_volatility = price_range * df['volume']
    sentiment = (high_low_ratio + open_close_ratio) * intraday_volatility
    
    # Adjust Intraday Volatility with Price Momentum
    adjusted_intraday_volatility = intraday_volatility * close_to_close_return
    
    # Synthesize Volume-Weighted, Intraday Sentiment, and Intraday Volatility Factors
    combined_factor = (momentum_adjusted_vwma + adjusted_intraday_volatility) / df['volume']
    
    # Enhance with Additional Price Dynamics
    open_to_close_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    final_factor = combined_factor * open_to_close_return
    
    return final_factor
