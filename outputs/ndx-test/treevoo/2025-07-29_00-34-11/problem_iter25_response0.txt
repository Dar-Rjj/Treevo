    volume_weighted_ratios = avg_ratios * df['volume']

    # Adjust for Volatility with Intraday Sentiment
    intraday_volatility = df['high'] - df['low']
    intraday_movement_ratio = (df['close'] - df['open']) / (df['high'] - df['low'])
    sentiment_adjusted_short_term_return = short_term_log_return / intraday_volatility - long_term_log_return

    # Combine Intraday Volatility, Close-to-Close Return, and Momentum
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    intraday_momentum = (df['close'] / df['open']) - 1
    combined_volatility_return_momentum = (df['high'] - df['low']) * close_to_close_return + intraday_momentum

    # Aggregate Volume Changes
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    aggregated_volume_changes = daily_volume_change.rolling(window=5).sum()

    # Final Adjustment
    final_factor = (sentiment_adjusted_short_term_return 
                    + combined_volatility_return_momentum 
                    * aggregated_volume_changes 
                    + volume_weighted_ratios)
