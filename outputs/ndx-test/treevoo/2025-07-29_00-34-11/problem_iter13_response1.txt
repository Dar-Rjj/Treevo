import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_volatility_return = intraday_volatility * close_to_close_return
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    high_low_spread = df['high'] - df['low']
    weighted_high_low_spread = high_low_spread * df['volume']
    vwma_high_low_spread = weighted_high_low_spread.rolling(window=20).mean() / df['volume'].rolling(window=20).mean()
    
    # Incorporate Price Momentum
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    volume_weighted_short_term_return = short_term_return * df['volume']
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Volume-Enhanced Impact: Multiply VWMA by Close-to-Close Return
    volume_enhanced_impact = vwma_high_low_spread * close_to_close_return
    
    # Final Adjustment
    final_adjustment = (combined_volatility_return / (intraday_volatility + 1)) + volume_enhanced_impact - volume_weighted_long_term_return
    final_factor_synthesis = final_adjustment / df['volume']
    
    # Enhance with Additional Price Dynamics
    open_to_close_return = (df['open'] - df['close'].shift(1))
    final_factor = final_factor_synthesis * open_to_close_return
    
    return final_factor
