```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['loss'] = (df['close'] <= df['close'].shift(1)) * (df['close'].shift(1) - df['close'])
    
    # Aggregate Gains and Losses over a period of 14 days
    gain_sum = df['gain'].rolling(window=14).sum()
    loss_sum = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = gain_sum / loss_sum
    
    # Convert to ARSI
    arsi = 100 - (100 / (1 + rs))
    arsi_vw = arsi * df['volume'].rolling(window=14).mean()
    
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    intraday_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    intraday_volatility = df['volume'] * (df['high'] - df['low'])
    alignment = intraday_ratio - intraday_volatility
    sentiment = (alignment > 0).astype(int) * 2 - 1  # Positive: 1, Negative: -1
    
    # Calculate Short-Term Return
    short_term_return = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Return
    long_term_return = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume-Weighted Short-Term and Long-Term Returns
    vw_short_term_return = short_term_return * df['volume']
    vw_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    short_term_high = df['high'].rolling(window=5).max()
    short_term_low = df['low'].rolling(window=5).min()
    short_term_volatility = (short_term_high - short_term_low).rolling(window=5).mean()
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Adjust for Volatility and Sentiment
    adjusted_return = (vw_short_term_return / short_term_volatility) - vw_long_term_return
    adjusted_return = adjusted_return * sentiment
    
    # Incorporate Enhanced Momentum into Adjusted Ratios
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    enhanced_momentum_ratios = adjusted_return * close_to_close_return
    
    # Final Volume Adjustment
    final_factor = enhanced_momentum_ratios / df['volume']
    
    return final_factor
```
