import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = df['high'] - df['low']
    open_close_ratio = df['open'] - df['close']
    weighted_ratios = (high_low_ratio + open_close_ratio) * df['volume']

    # Measure Price-Volatility Alignment
    intraday_volatility = high_low_ratio
    sentiment = weighted_ratios > 0

    # Incorporate Momentum
    close_to_close_return = (df['close'].diff() / df['close'].shift(1)).fillna(0)
    final_factor = weighted_ratios * close_to_close_return

    # Analyze Price Action Patterns
    relative_closing_price_change = close_to_close_return
    true_range = pd.Series(0, index=df.index)
    upper_shadow_length = df['high'] - df[['open', 'close']].max(axis=1)
    lower_shadow_length = df[['open', 'close']].min(axis=1) - df['low']

    for i in range(1, len(df)):
        true_range[i] = max(df['high'][i], df['close'][i-1]) - min(df['low'][i], df['close'][i-1])

    # Investigate Volume-Related Factors
    volume_ratio = df['volume'] / df['volume'].shift(1).fillna(1)
    volume_weighted_close_price = df['close'] * df['volume']

    # Combine all factors into a single alpha factor
    alpha_factor = final_factor * true_range * (upper_shadow_length - lower_shadow_length) * volume_ratio * volume_weighted_close_price

    return alpha_factor
