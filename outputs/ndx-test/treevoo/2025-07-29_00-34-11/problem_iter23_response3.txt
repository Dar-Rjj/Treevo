```python
def heuristics_v2(df):
    # Input: df is a pandas DataFrame with columns: open, high, low, close, volume
    # Output: A pandas Series representing the factor values
    
    # Calculate buying and selling pressure based on the ratio of volume traded at prices above and below the current close
    buying_pressure = (df['high'] - df['close']).clip(lower=0) * df['volume']
    selling_pressure = (df['low'] - df['close']).clip(upper=0) * df['volume']

    # Calculate the net pressure
    net_pressure = (buying_pressure + selling_pressure)

    # Calculate the difference between the high and low price
    range_price = df['high'] - df['low']

    # Create the alpha factor as a combination of net pressure and range
    alpha_factor = (net_pressure / range_price).rolling(window=5).mean()

    return alpha_factor
```
