```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of open, high, low, and close prices
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the difference between high and low price
    range_price = df['high'] - df['low']
    
    # Calculate the trading intensity based on volume and price range
    trading_intensity = df['volume'] / range_price
    
    # Generate a simple alpha factor: normalized average price * trading intensity
    alpha_factor = avg_price * trading_intensity
    
    return alpha_factor
```
