```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Gain and Loss
    df['Gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['Loss'] = np.where(df['close'] <= df['close'].shift(1), df['close'].shift(1) - df['close'], 0)

    # Aggregate Gains and Losses over a period (e.g., 14 days)
    df['Sum_Gain_14'] = df['Gain'].rolling(window=14).sum()
    df['Sum_Loss_14'] = df['Loss'].rolling(window=14).sum()

    # Calculate Relative Strength
    df['RS'] = df['Sum_Gain_14'] / df['Sum_Loss_14']

    # Convert to ARSI
    df['ARSI'] = 100 - (100 / (1 + df['RS']))

    # Calculate Short-Term Logarithmic Return (5 days)
    df['Short_Term_Return'] = np.log(df['close'] / df['close'].shift(5))

    # Calculate Long-Term Logarithmic Return (20 days)
    df['Long_Term_Return'] = np.log(df['close'] / df['close'].shift(20))

    # Calculate Short-Term Volatility (Average True Range, 5 days)
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['Short_Term_Volatility'] = df['True_Range'].rolling(window=5).mean()

    # Adjust for Volatility
    df['Adjusted_Short_Term_Return'] = df['Short_Term_Return'] / df['Short_Term_Volatility']

    # Calculate Intraday Price Movement Intensity
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Open_Close_Change'] = df['close'] - df['open']
    df['Intraday_Movement_Intensity'] = df['High_Low_Spread'] + df['Open_Close_Change']

    # Measure Momentum-Volatility Correlation
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['volume']
    df['Momentum_Sentiment'] = np.where(df['Intraday_Volatility'] > df['Intraday_Volatility'].mean(), 1, -1)

    # Combine Intraday Price Movement Intensity and Momentum Sentiment
    df['Combined_Intraday_Movement'] = df['Intraday_Movement_Intensity'] * df['Momentum_Sentiment']

    # Combine ARSI, Adjusted Returns, and Correlation
    df['Final_Factor'] = (df['ARSI'] * df['Adjusted_Short_Term_Return']) + df['Combined_Intraday_Movement'] - df['Long_Term_Return']

    return df['Final_Factor']
```
