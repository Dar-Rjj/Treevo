```python
import pandas as pd

def heuristics_v2(df):
    # Calculate SMAs
    short_sma = df['close'].rolling(window=5).mean()
    long_sma = df['close'].rolling(window=20).mean()
    momentum_signal = short_sma - long_sma
    
    # Calculate VWAP
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_change = vwap.pct_change()
    
    # Calculate True Range and ATR
    true_range = pd.Series(0, index=df.index)
    for i in range(1, len(df)):
        tr1 = df['high'].iloc[i] - df['low'].iloc[i]
        tr2 = abs(df['high'].iloc[i] - df['close'].iloc[i-1])
        tr3 = abs(df['low'].iloc[i] - df['close'].iloc[i-1])
        true_range.iloc[i] = max(tr1, tr2, tr3)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate daily return and cumulative return
    daily_return = (df['close'] - df['open']) / df['open']
    cumulative_return = (1 + daily_return).rolling(window=5).apply(lambda x: x.prod()) - 1
    
    # Calculate EMAs
    ema_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_long = df['close'].ewm(span=20, adjust=False).mean()
    ema_trend = (ema_short - ema_short.shift(1)) - (ema_long - ema_long.shift(1))
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        momentum_signal +
        vwap_change +
        atr +
        cumulative_return +
        ema_trend
    )
    
    return alpha_factor
```
