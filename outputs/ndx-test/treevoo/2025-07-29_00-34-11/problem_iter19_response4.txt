```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Average High-Low and Open-Close Ratios
    avg_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Combine and Weight by Volume
    volume_weighted_ratio = avg_ratio * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Determine Sentiment
    sentiment = (avg_ratio - 1) / intraday_volatility
    sentiment[sentiment > 0] = 1
    sentiment[sentiment < 0] = -1
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Final Adjustment
    final_adjustment = volume_weighted_ratio * close_to_close_return / df['volume']
    
    return final_adjustment
```
