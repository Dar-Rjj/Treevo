```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Return
    short_term_return = df['close'].pct_change(periods=4)
    
    # Calculate Long-Term Return
    long_term_return = df['close'].pct_change(periods=19)
    
    # Calculate Volume-Weighted Short-Term Return
    volume_weighted_short_term_return = short_term_return * df['volume']
    
    # Calculate Volume-Weighted Long-Term Return
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    true_range = df[['high', 'low']].rolling(window=5).apply(lambda x: max(x) - min(x), raw=True)
    short_term_volatility = true_range.mean(axis=1)
    
    # Adjust for Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / short_term_volatility - volume_weighted_long_term_return
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change(periods=1)
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_intraday_close = intraday_volatility * close_to_close_return
    
    # Final Adjustment
    final_factor = adjusted_short_term_return + combined_intraday_close
    
    return final_factor
```
