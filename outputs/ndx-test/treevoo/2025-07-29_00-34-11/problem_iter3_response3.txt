```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Open-Close Change
    open_close_change = df['close'] - df['open']
    
    # Combine Changes
    intraday_price_movement_intensity = high_low_spread + open_close_change
    
    # Calculate Intraday Volatility
    price_range = df['high'] - df['low']
    intraday_volatility = price_range / df['volume']
    
    # Evaluate Correlation
    momentum_sentiment = (intraday_price_movement_intensity * intraday_volatility).rolling(window=5).mean()
    
    # Determine Momentum Sentiment
    alpha_factor = momentum_sentiment
    
    return alpha_factor
```
