```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    df['HighLowRatio'] = df['high'] / df['low']
    df['OpenCloseRatio'] = df['open'] / df['close']
    df['IntradayMovementRatio'] = (df['HighLowRatio'] + df['OpenCloseRatio']) / 2
    
    # Measure Price-Volatility Alignment
    df['PriceRange'] = df['high'] - df['low']
    df['IntradayVolatility'] = df['PriceRange'] * df['volume']
    
    # Evaluate Alignment
    df['Sentiment'] = df.apply(lambda x: 1 if x['IntradayMovementRatio'] > x['IntradayVolatility'] else -1, axis=1)
    
    # Calculate Gain and Loss
    df['Gain'] = df['close'].diff().apply(lambda x: x if x > 0 else 0)
    df['Loss'] = df['close'].diff().apply(lambda x: -x if x < 0 else 0)
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    gain_sum = df['Gain'].rolling(window=14).sum()
    loss_sum = df['Loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = gain_sum / loss_sum
    
    # Convert to ARSI
    rsi = 100 - (100 / (1 + rs))
    arsi = rsi * df['volume'].rolling(window=14).mean() * (df['close'] / df['open'])
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'].pct_change(periods=5)
    long_term_return = df['close'].pct_change(periods=20)
    
    # Volume-Weighted Returns
    vwr_short_term = short_term_return * df['volume']
    vwr_long_term = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    df['ATR_5'] = (df['high'] - df['low']).rolling(window=5).mean()
    adjusted_vwr_short_term = vwr_short_term / df['ATR_5']
    
    # Calculate Price Oscillator
    price_oscillator = df['close'].shift(5) - df['close'].shift(20)
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    combined_factor = arsi * adjusted_vwr_short_term + price_oscillator - vwr_long_term
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwm_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'].diff()
    multiplied_vwma = vwm_high_low_spread * close_to_close_return
    
    # Adjust with Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_factor = multiplied_vwma * volume_change
    
    return final_factor
```
