```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the price range for each day
    price_range = df['high'] - df['low']
    
    # Calculate the closing price relative to the daily range
    close_to_range = (df['close'] - df['low']) / price_range
    
    # Calculate the ratio of closing price to opening price
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the weighted average of the above two ratios
    factor = 0.6 * close_to_range + 0.4 * close_open_ratio
    
    return factor
```
