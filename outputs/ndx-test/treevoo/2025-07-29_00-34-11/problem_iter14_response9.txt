import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    sentiment = intraday_movement_ratio - intraday_volatility
    sentiment[sentiment > 0] = 1
    sentiment[sentiment < 0] = -1
    
    # Calculate Gain and Loss
    gain = df['close'].diff().clip(lower=0)
    loss = -df['close'].diff().clip(upper=0)
    
    # Aggregate Gains and Losses
    sum_gains = gain.rolling(window=14).sum()
    sum_losses = loss.rolling(window=14).sum()
    
    # Calculate Relative Strength
    rs = sum_gains / sum_losses.replace(0, 1e-6)
    
    # Convert to ARSI
    rsi = 100 - (100 / (1 + rs))
    arsi = rsi * df['volume'].rolling(window=14).mean() * (df['close'] / df['open'])
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'].pct_change(5)
    long_term_return = df['close'].pct_change(20)
    
    # Calculate Volume-Weighted Returns
    volume_weighted_short_term_return = short_term_return * df['volume']
    volume_weighted_long_term_return = long_term_return * df['volume']
    
    # Calculate Short-Term Volatility
    short_term_volatility = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()).mean()
    
    # Adjust for Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / short_term_volatility
    
    # Calculate Price Oscillator
    price_oscillator = df['close'].shift(5) - df['close'].shift(20)
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    combined_factor = arsi * adjusted_short_term_return + price_oscillator - volume_weighted_long_term_return
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).mean()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'].diff()
    multiplied_vwma = vwma_high_low_spread * close_to_close_return
    
    # Adjust with Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_factor = multiplied_vwma * volume_change
    
    return final_factor
