import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices (Intraday Price Range)
    df['Intraday_Price_Range'] = df['high'] - df['low']
    
    # Develop a factor representing the ratio of (close - low) / (high - low) (Closing Price Position)
    df['Closing_Price_Position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Formulate a factor indicating days with unusually high volume compared to a moving average
    df['Volume_MA'] = df['volume'].rolling(window=20).mean()
    df['Volume_Spike'] = (df['volume'] > 1.5 * df['Volume_MA']).astype(int)
    
    # Create a factor that integrates significant price changes with large volume (Price-Volume Momentum)
    df['Price_Change'] = df['close'].pct_change()
    df['Price_Volume_Momentum'] = df['Price_Change'] * df['volume'] * df['Volume_Spike']
    
    # Construct a factor based on the percentage change in close price over the last 5 days (Momentum Factor)
    df['Momentum_Factor'] = df['close'].pct_change(periods=5)
