def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Assign Weights
    weight_high_low = 0.7
    weight_open_close = 0.3
    
    # Weighted Average of High-Low and Open-Close Ratios
    intraday_price_movement_ratio = (weight_high_low * high_low_ratio) + (weight_open_close * open_close_ratio)
    
    # Calculate Intraday Volatility
    price_range = df['high'] - df['low']
    
    # Adjust by Volume
    intraday_volatility_volume = price_range * df['volume']
    
    # Evaluate Alignment
    alignment = intraday_price_movement_ratio - intraday_volatility_volume
    
    # Determine Sentiment
    sentiment = alignment.apply(lambda x: 1 if x > 0 else -1)
    
    return sentiment
