```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    df['weighted_spread'] = df['high_low_spread'] * df['volume']
    df['vwma_high_low_spread'] = df['weighted_spread'].rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Incorporate Price Momentum
    df['close_to_close_return'] = df['close'] - df['close'].shift(1)
    df['momentum_adjusted_vwma'] = df['vwma_high_low_spread'] * df['close_to_close_return']
    
    # Determine Gain and Loss
    df['gain'] = df.apply(lambda x: x['close'] - x['close'].shift(1) if x['close'] > x['close'].shift(1) else 0, axis=1)
    df['loss'] = df.apply(lambda x: x['close'].shift(1) - x['close'] if x['close'] < x['close'].shift(1) else 0, axis=1)
    
    # Aggregate Gains and Losses
    df['sum_gains'] = df['gain'].rolling(window=14).sum()
    df['sum_losses'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['sum_gains'] / df['sum_losses']
    
    # Convert to ARSI
    df['arsi'] = 100 - (100 / (1 + df['rs']))
    df['arsi'] = df['arsi'] * df['volume'].rolling(window=14).mean() * (df['close'] - df['open'])
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Adjust Intraday Volatility with Price Momentum
    df['volatility_momentum'] = df['intraday_volatility'] * df['close_to_close_return']
    
    # Synthesize Volume-Weighted and Intraday Volatility Factors
    df['synthesized_factor'] = (df['momentum_adjusted_vwma'] + df['volatility_momentum']) / df['volume']
    
    # Enhance with Additional Price Dynamics
    df['open_to_close_return'] = df['open'] - df['close'].shift(1)
    df['enhanced_factor'] = df['synthesized_factor'] * df['open_to_close_return']
    
    # Introduce Reversal Component
    df['open_to_open_return'] = df['open'] - df['open'].shift(1)
    df['reversal_enhanced_factor'] = df['enhanced_factor'] * df['open_to_open_return']
    
    # Calculate Short-Term Return
    df['short_term_return'] = df['close'].shift(1) - df['close'].shift(5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = df['close'].shift(1) - df['close'].shift(20)
    
    # Calculate Volume-Weighted Short-Term Return
    df['volume_weighted_short_term_return'] = df['volume'] * df['short_term_return']
    
    # Calculate Volume-Weighted Long-Term Return
    df['volume_weighted_long_term_return'] = df['volume'] * df['long_term_return']
    
    # Calculate Short-Term Volatility
    df['atr_5'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Adjust for Volatility and Sentiment
    df['adjusted_short_term_return'] = (df['volume_weighted_short_term_return'] / df['atr_5']) - df['volume_weighted_long_term_return']
    
    # Combine ARSI, Adjusted Returns, and VWMA
    df['factor'] = df['arsi'] * df['adjusted_short_term_return'] * df['vwma_high_low_spread'] - df['volume_weighted_long_term_return']
    
    return df['factor']
```
