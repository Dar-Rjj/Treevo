```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = (df['high'] - df['low']) / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = (df['close'] - df['open']) / df['open']
    
    # Average Ratios
    intraday_price_movement = (high_low_ratio + open_close_ratio) / 2
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    
    # Evaluate Alignment
    alignment = intraday_price_movement / intraday_volatility
    
    # Determine Sentiment
    sentiment = alignment.apply(lambda x: 1 if x > 0 else -1)
    
    return sentiment
```
