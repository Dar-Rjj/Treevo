```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    close_to_close_return = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    vol_adj_momentum = intraday_volatility * close_to_close_return
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    high_low_spread = df['high'] - df['low']
    volume_weighted_high_low = (high_low_spread * df['volume']).rolling(window=20).mean()
    
    # Incorporate Price Momentum
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    long_term_return = (df['close'] / df['close'].shift(20)) - 1
    volume_weighted_short_term_return = df['volume'] * short_term_return
    volume_weighted_long_term_return = df['volume'] * long_term_return
    
    # Final Adjustment
    short_term_volatility = df['close'].rolling(window=5).std()
    final_factor = (vol_adj_momentum / (short_term_volatility + 1))
    final_factor += volume_weighted_high_low * close_to_close_return
    final_factor -= volume_weighted_long_term_return
    final_factor /= df['volume']
    
    # Enhance with Additional Price Dynamics
    open_to_close_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    final_factor *= open_to_close_return
    
    return final_factor
```
