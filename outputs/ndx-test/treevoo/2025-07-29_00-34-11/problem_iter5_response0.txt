```python
import pandas as pd

def heuristics_v2(df, window=20):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low = (high_low_spread * df['volume']).rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
    
    # Adjust VWMA by Volume
    adjusted_vwma = vwma_high_low / df['volume']
    
    # Incorporate Price Momentum into Adjusted VWMA
    close_to_close_return = df['close'].pct_change()
    momentum_adjusted_vwma = adjusted_vwma * close_to_close_return
    
    # Combine Momentum-Adjusted VWMA
    combined_factor = momentum_adjusted_vwma.sum()
    
    # Final Volume Adjustment
    final_factor = combined_factor / df['volume']
    
    return final_factor
```
