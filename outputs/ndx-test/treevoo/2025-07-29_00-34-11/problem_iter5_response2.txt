```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    weighted_high_low_spread = high_low_spread * df['volume']
    vwma_high_low = weighted_high_low_spread.rolling(window=20).mean() / df['volume'].rolling(window=20).mean()
    
    # Incorporate Close-to-Close Return
    close_to_close_return = df['close'].diff(periods=1)
    multiplied_vwma = vwma_high_low * close_to_close_return
    
    # Combine with Volume Adjusted Price Momentum
    price_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    combined_factor = multiplied_vwma * volume_change
    
    # Final Volume Adjusted Momentum Factor
    final_factor = combined_factor / df['volume']
    
    return final_factor
```
