import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as [open, high, low, close, volume].
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the percentage change in closing price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the relative range as (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted by the percentage change in closing price
    df['volume_weighted_close_change'] = df['volume'] * df['close_change']
    
    # Calculate the alpha factor as the product of the relative range and the volume weighted close change
    df['alpha_factor'] = df['relative_range'] * df['volume_weighted_close_change']
    
    # Return the alpha factor
    return df['alpha_factor']
