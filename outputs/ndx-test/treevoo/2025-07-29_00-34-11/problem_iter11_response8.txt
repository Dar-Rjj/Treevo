import pandas as pd

def heuristics_v2(df, window=20):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window).mean()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'].pct_change()
    vwma_price_momentum = vwma_high_low_spread * close_to_close_return
    
    # Compute Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Adjust Intraday Volatility with Price Momentum
    adjusted_intraday_volatility = intraday_volatility * close_to_close_return
    
    # Synthesize Volume-Weighted, Intraday Volatility, and Price Momentum Factors
    synthesized_factor = (vwma_price_momentum + adjusted_intraday_volatility) / df['volume']
    
    # Enhance with Additional Price Dynamics
    open_to_close_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    enhanced_synthesized_factor = synthesized_factor * open_to_close_return
    
    # Incorporate Short-Term and Long-Term Price Momentum
    short_term_return = (df['close'].shift(1) - df['close'].shift(5)) / df['close'].shift(5)
    long_term_return = (df['close'].shift(1) - df['close'].shift(20)) / df['close'].shift(20)
    vw_short_term_return = short_term_return * df['volume']
    vw_long_term_return = long_term_return * df['volume']
    
    # Final Adjustment
    final_adjustment = (enhanced_synthesized_factor / (intraday_volatility.rolling(window).std() + 1) 
                        + vw_short_term_return - vw_long_term_return) / df['volume']
    
    return final_adjustment
