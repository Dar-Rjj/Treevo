import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    high_low_ratio = (df['high'] - df['low']) / df['low']
    open_close_ratio = (df['close'] - df['open']) / df['open']
    intraday_price_movement = (high_low_ratio + open_close_ratio) / 2
    
    # Measure Price-Volatility-Volume Alignment
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    price_volatility_alignment = intraday_price_movement / intraday_volatility
    
    # Determine Sentiment based on alignment
    sentiment = price_volatility_alignment.apply(lambda x: 'Positive' if x > 0 else 'Negative')
    
    # Incorporate Volume Impact
    volume_adjusted_intraday_volatility = (df['high'] - df['low']) * df['volume'] ** 0.5
    volume_alignment = intraday_price_movement / volume_adjusted_intraday_volatility
    
    # Determine Volume Sentiment based on alignment
    volume_sentiment = volume_alignment.apply(lambda x: 'Positive' if x > 0 else 'Negative')
    
    # Combine both sentiments
    combined_sentiment = (price_volatility_alignment + volume_alignment) / 2
    alpha_factor = combined_sentiment.apply(lambda x: 1 if x > 0 else -1)
    
    return alpha_factor
