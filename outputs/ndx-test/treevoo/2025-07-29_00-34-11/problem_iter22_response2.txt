```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low and Open-Close Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    
    # Average the ratios and weight by volume
    avg_ratios = (high_low_ratio + open_close_ratio) / 2
    volume_weighted_ratios = avg_ratios * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Determine Sentiment
    sentiment = (avg_ratios - 1) / intraday_volatility
    
    # Incorporate Momentum: Calculate Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Final Adjustment
    factor_values = volume_weighted_ratios * close_to_close_return * sentiment
    
    return factor_values
```
