import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    avg_ratio = (high_low_ratio + open_close_ratio) / 2
    volume_weighted_ratio = avg_ratio * df['volume']
    
    # Measure Price-Volatility Alignment
    intraday_volatility = df['high'] - df['low']
    sentiment = (avg_ratio > intraday_volatility).astype(int)
    
    # Incorporate Improved VW-ARSI
    price_diff = df['high'] - df['low']
    gain = (df['close'].shift(-1) > df['close']).astype(float) * (df['close'].shift(-1) - df['close'])
    loss = (df['close'].shift(-1) < df['close']).astype(float) * (df['close'] - df['close'].shift(-1))
    
    sum_gains = gain.rolling(window=14).sum()
    sum_losses = loss.rolling(window=14).sum()
    
    relative_strength = sum_gains / sum_losses
    arsi = 100 - (100 / (1 + relative_strength))
    v_arsi = arsi * df['volume'].rolling(window=14).mean()
    
    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    combined_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'] / df['close'].shift(5) - 1
    long_term_return = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume-Weighted Returns
    short_term_volume_weighted = short_term_return * df['volume']
    long_term_volume_weighted = long_term_return * df['volume']
    
    # Adjust for Volatility and Sentiment
    adjusted_short_term = short_term_volume_weighted / intraday_volatility
    adjusted_long_term = long_term_volume_weighted / intraday_volatility
    sentiment_adjusted = adjusted_short_term - adjusted_long_term + (sentiment * 0.1 - 0.05)
    
    # Incorporate Momentum
    close_to_close_return = df['close'] / df['close'].shift(1) - 1
    momentum_adjusted = sentiment_adjusted * close_to_close_return
    final_factor = momentum_adjusted / df['volume']
    
    return final_factor
