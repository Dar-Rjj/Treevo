```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Logarithmic Return
    short_term_log_return = np.log(df['close'] / df['close'].shift(5))
    
    # Calculate Long-Term Logarithmic Return
    long_term_log_return = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    volume_weighted_short_term_log_return = df['volume'] * short_term_log_return
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    volume_weighted_long_term_log_return = df['volume'] * long_term_log_return
    
    # Calculate Short-Term Volatility (Average True Range)
    true_range = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs()
    short_term_volatility = true_range.rolling(window=5).mean()
    
    # Adjust for Volatility
    adjusted_volatility = (volume_weighted_short_term_log_return / short_term_volatility) - volume_weighted_long_term_log_return
    
    # Calculate Volume-Weighted Intraday Movement Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['open'] / df['close']
    intraday_movement_ratio = (high_low_ratio + open_close_ratio) / 2
    volume_weighted_intraday_movement_ratio = df['volume'] * intraday_movement_ratio
    
    # Measure Price-Volatility Alignment
    intraday_price_range = df['high'] - df['low']
    intraday_volatility = df['volume'] * intraday_price_range
    alignment = intraday_movement_ratio / intraday_volatility
    sentiment = np.where(alignment > 0, 1, -1)
    
    # Incorporate Enhanced Momentum into Adjusted Ratios
    close_to_close_return = df['close'] - df['close'].shift(1)
    enhanced_momentum_ratios = volume_weighted_intraday_movement_ratio * close_to_close_return
    
    # Final Volume Adjustment
    final_volume_adjustment = enhanced_momentum_ratios / df['volume']
    
    # Combine Intraday Volatility and Close-to-Close Return
    combined_intraday_volatility = intraday_volatility * close_to_close_return
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Aggregate Volume Changes
    aggregated_volume_changes = daily_volume_change.rolling(window=5).sum()
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] / df['open']) - 1
    
    # Adjust for Intraday Momentum and Sentiment
    adjusted_factor = (adjusted_volatility + combined_intraday_volatility) * aggregated_volume_changes + intraday_momentum * sentiment
    
    # Final Adjustment
    final_factor = adjusted_factor * aggregated_volume_changes
    
    return final_factor
```
