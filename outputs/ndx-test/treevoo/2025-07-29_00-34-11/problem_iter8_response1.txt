```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['loss'] = (df['close'] <= df['close'].shift(1)) * (df['close'].shift(1) - df['close'])
    
    # Aggregate Gains and Losses over a 14-day period
    df['sum_gain_14'] = df['gain'].rolling(window=14).sum()
    df['sum_loss_14'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['sum_gain_14'] / df['sum_loss_14']
    
    # Convert to ARSI
    df['arsi'] = 100 - (100 / (1 + df['rs']))
    df['arsi'] *= df['volume'].rolling(window=14).mean()  # Incorporate Volume
    df['arsi'] *= (df['close'] / df['open'])  # Add Momentum
    
    # Calculate Intraday Price Movement Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    df['open_close_ratio'] = df['close'] / df['open']
    df['intraday_movement_ratio'] = (df['high_low_ratio'] + df['open_close_ratio']) / 2
    
    # Measure Price-Volatility Alignment
    df['intraday_volatility'] = df['volume'] * (df['high'] - df['low'])
    df['sentiment'] = (df['intraday_movement_ratio'] > df['intraday_volatility']).astype(int)
    
    # Calculate Short-Term Return
    df['short_term_return'] = df['close'] / df['close'].shift(5) - 1
    
    # Calculate Long-Term Return
    df['long_term_return'] = df['close'] / df['close'].shift(20) - 1
    
    # Calculate Volume-Weighted Short-Term Return
    df['vwsr'] = df['volume'] * df['short_term_return']
    
    # Calculate Volume-Weighted Long-Term Return
    df['vwlr'] = df['volume'] * df['long_term_return']
    
    # Calculate Short-Term Volatility
    df['atr_5'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1).rolling(window=5).mean()
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute VWMA of High-Low Spread
    df['vwmhs'] = (df['high_low_spread'] * df['volume']).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    
    # Incorporate Close-to-Close Return
    df['ccr'] = df['close'] - df['close'].shift(1)
    df['vwmhs'] *= df['ccr']
    
    # Adjust for Volatility and Sentiment
    df['adjusted_vwsr'] = df['vwsr'] / df['atr_5'] - df['vwlr']
    df['adjusted_vwsr'] += df['sentiment'] * 0.1  # Simple adjustment based on sentiment
    
    # Combine ARSI, Adjusted Returns, and VWMA
    df['alpha_factor'] = df['arsi'] * df['adjusted_vwsr'] * df['vwmhs'] - df['vwlr']
    
    return df['alpha_factor']
```
