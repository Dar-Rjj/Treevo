```python
def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as open, high, low, close, volume.
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the average of the high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the close price to the average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the ratio of the volume to the average volume over the last 10 days
    rolling_avg_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / rolling_avg_volume
    
    # Combine the two ratios to create the final factor
    factor = close_to_avg_ratio * volume_ratio
    
    return factor
```
