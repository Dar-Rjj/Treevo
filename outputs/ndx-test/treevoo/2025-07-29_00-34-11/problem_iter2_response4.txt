```python
import pandas as pd

def heuristics_v2(df, window=20):
    # Calculate Daily High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Volume-Weighted Moving Average (VWMA) of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
    
    # Incorporate Price Momentum
    close_to_close_return = df['close'] - df['close'].shift(1)
    vwma_price_momentum = vwma_high_low_spread * close_to_close_return
    
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Combine Intraday Volatility and Close-to-Close Return
    intraday_volatility_momentum = intraday_volatility * close_to_close_return
    
    # Synthesize Volume-Weighted and Intraday Volatility Factors
    synthesized_factor = (vwma_price_momentum + intraday_volatility_momentum) / df['volume']
    
    return synthesized_factor
```
