```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative closing price change
    df['rel_close_change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute the true range
    df['true_range'] = (df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1).shift(1))
    
    # Measure the upper shadow length
    df['upper_shadow'] = df['high'] - df[['open', 'close']].max(axis=1)
    
    # Measure the lower shadow length
    df['lower_shadow'] = df[['open', 'close']].min(axis=1) - df['low']
    
    # Evaluate the ratio of current to previous day's volume
    df['vol_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Assess the volume-weighted close price
    df['vol_weighted_close'] = (df['close'] * df['volume'])
    
    # Calculate the simple moving average (SMA) of close prices over 5 days
    df['sma_5'] = df['close'].rolling(window=5).mean()
    
    # Determine the rate of change (ROC) over a period of 5 days
    df['roc_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Identify the directional movement using the difference between consecutive closes
    df['directional_movement'] = df['close'] - df['close'].shift(1)
    
    # Consider the ratio of daily amount to volume
    df['amount_vol_ratio'] = df['amount'] / df['volume']
    
    # Examine the relationship between amount and price
    df['amount_price_rel'] = (df['amount'] / df['volume']) * df['close']
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['rel_close_change'] + 
        df['true_range'] +
        df['upper_shadow'] +
        df['lower_shadow'] +
        df['vol_ratio'] + 
        df['vol_weighted_close'] +
        df['sma_5'] + 
        df['roc_5'] + 
        df['directional_movement'] + 
        df['amount_vol_ratio'] + 
        df['amount_price_rel']
    )
    
    return alpha_factor
```
