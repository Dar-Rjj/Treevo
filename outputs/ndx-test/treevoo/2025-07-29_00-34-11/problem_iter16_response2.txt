```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Gain or Loss
    df['gain'] = (df['close'] - df['close'].shift(1)).apply(lambda x: x if x > 0 else 0)
    df['loss'] = (df['close'] - df['close'].shift(1)).apply(lambda x: -x if x < 0 else 0)

    # Aggregate Gains and Losses
    sum_gains = df['gain'].rolling(window=14).sum()
    sum_losses = df['loss'].rolling(window=14).sum()

    # Calculate Relative Strength (RS)
    rs = sum_gains / sum_losses

    # Convert to ARSI
    rsi = 100 - (100 / (1 + rs))
    avg_volume = df['volume'].rolling(window=14).mean()
    arsi = rsi * avg_volume * ((df['close'] / df['close'].shift(1)) - 1)

    # Calculate Short-Term and Long-Term Return
    short_term_return = (df['close'] / df['close'].shift(5)) - 1
    long_term_return = (df['close'] / df['close'].shift(20)) - 1

    # Calculate Volume-Weighted Returns
    volume_weighted_short_term_return = short_term_return * df['volume']
    volume_weighted_long_term_return = long_term_return * df['volume']

    # Calculate Short-Term Volatility
    tr = df[['high', 'low', 'close']].join(
        df[['high', 'low', 'close']].shift(1).rename(columns={'high': 'high_prev', 'low': 'low_prev', 'close': 'close_prev'})
    )
    tr['true_range'] = tr.apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close_prev']), abs(x['low'] - x['close_prev'])), axis=1)
    short_term_volatility = tr['true_range'].rolling(window=5).mean()

    # Adjust for Volatility
    adjusted_short_term_return = volume_weighted_short_term_return / short_term_volatility

    # Calculate Price Oscillator
    price_oscillator = df['close'].shift(5) - df['close'].shift(20)

    # Combine Factors
    combined_factors = arsi * adjusted_short_term_return + price_oscillator - volume_weighted_long_term_return

    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Compute VWMA of High-Low Spread
    vwma_high_low_spread = (high_low_spread * df['volume']).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()

    # Incorporate Close-to-Close Return
    close_to_close_return = vwma_high_low_spread * (df['close'] - df['close'].shift(1))

    # Adjust with Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_factor = (vwma_high_low_spread * close_to_close_return) * volume_change

    return final_factor
```
