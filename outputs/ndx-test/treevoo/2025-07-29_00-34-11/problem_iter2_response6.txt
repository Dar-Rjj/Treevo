```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']

    # Determine Gain and Loss
    df['gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['loss'] = (df['close'] <= df['close'].shift(1)) * (df['close'].shift(1) - df['close'])

    # Aggregate Gains and Losses over a period (e.g., 14 days)
    df['sum_gain'] = df['gain'].rolling(window=14).sum()
    df['sum_loss'] = df['loss'].rolling(window=14).sum()

    # Calculate Relative Strength
    df['rs'] = df['sum_gain'] / df['sum_loss']

    # Convert to ARSI
    df['ars'] = 100 - (100 / (1 + df['rs']))
    df['arsi'] = df['ars'] * df['volume'].rolling(window=14).mean() * (df['close'] / df['open'])

    # Calculate Short-Term Return
    df['short_term_return'] = (df['close'] / df['close'].shift(5)) - 1

    # Calculate Long-Term Return
    df['long_term_return'] = (df['close'] / df['close'].shift(20)) - 1

    # Calculate Volume-Weighted Short-Term Return
    df['vw_short_term_return'] = df['volume'] * df['short_term_return']

    # Calculate Volume-Weighted Long-Term Return
    df['vw_long_term_return'] = df['volume'] * df['long_term_return']

    # Calculate Short-Term Volatility
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['short_term_volatility'] = df['true_range'].rolling(window=5).mean()

    # Synthesize ARSI with Volume-Weighted Returns and Volatility
    df['combined_arsi'] = df['arsi'] * df['vw_short_term_return'] / df['short_term_volatility']
    df['alpha_factor'] = df['combined_arsi'] - df['vw_long_term_return']

    return df['alpha_factor']
```
