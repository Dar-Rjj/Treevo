```python
import pandas as pd

def heuristics_v2(df):
    # Obtain Last 5 Days' Close Prices
    close_prices = df['close']

    # Calculate Moving Averages
    short_term_ma = close_prices.rolling(window=3).mean()
    long_term_ma = close_prices.rolling(window=10).mean()

    # Compute Momentum Signal
    momentum_signal = long_term_ma - short_term_ma
    momentum_direction = (momentum_signal > 0).astype(int) - (momentum_signal < 0).astype(int)

    # Confirm Momentum with Volume
    volume_sum_last_5_days = df['volume'].rolling(window=5).sum()
    avg_daily_volume = df['volume'].expanding().mean()
    volume_ratio = volume_sum_last_5_days / avg_daily_volume
    volume_filter = (volume_ratio > 1.5).astype(int)

    # Confirm Momentum with Amount
    amount_sum_last_5_days = df['amount'].rolling(window=5).sum()
    avg_daily_amount = df['amount'].expanding().mean()
    amount_ratio = amount_sum_last_5_days / avg_daily_amount
    amount_filter = (amount_ratio > 1.5).astype(int)

    # Combine Momentum, Volume, and Amount
    alpha_factor = momentum_direction * volume_filter * amount_filter
    alpha_factor[momentum_direction == 1] = 1  # If momentum is positive, maintain positive factor
    alpha_factor[(momentum_direction == 1) & ((volume_filter == 1) | (amount_filter == 1))] = 2  # If momentum and either volume or amount align, increase positive factor
    alpha_factor[(momentum_direction == 1) & (volume_filter == 1) & (amount_filter == 1)] = 3  # If all three align, assign higher positive factor
    alpha_factor[momentum_direction == -1] = -1  # If momentum is negative, assign negative factor

    return alpha_factor
```
