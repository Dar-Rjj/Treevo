import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    avg_ratio = (high_low_ratio + open_close_ratio) / 2
    volume_weighted_ratios = avg_ratio * df['volume']

    # Measure Price-Volatility Alignment
    intraday_volatility = df['high'] - df['low']
    sentiment = (avg_ratio > intraday_volatility).astype(int) * 2 - 1

    # Incorporate Improved VW-ARSI
    price_diff = df['high'] - df['low']
    gain = (df['close'].shift(-1) > df['close']).astype(float) * (df['close'].shift(-1) - df['close'])
    loss = (df['close'].shift(-1) < df['close']).astype(float) * (df['close'] - df['close'].shift(-1))
    
    sum_gains = gain.rolling(window=14).sum()
    sum_losses = loss.rolling(window=14).sum()
    relative_strength = sum_gains / sum_losses
    rsi = 100 - (100 / (1 + relative_strength))
    vw_arsi = rsi * df['volume'].rolling(window=14).mean()

    # Calculate Intraday Price Movement Ratio
    high_low_ratio = df['high'] / df['low']
    open_close_ratio = df['close'] / df['open']
    combined_ratio = (high_low_ratio + open_close_ratio) / 2

    # Calculate Short-Term and Long-Term Returns
    short_term_return = df['close'].pct_change(periods=5)
    long_term_return = df['close'].pct_change(periods=20)

    # Calculate Volume-Weighted Returns
    volume_weighted_short_term_return = short_term_return * df['volume']
    volume_weighted_long_term_return = long_term_return * df['volume']

    # Adjust for Volatility and Sentiment
    adjusted_short_term = volume_weighted_short_term_return / intraday_volatility
    adjusted_long_term = volume_weighted_long_term_return / intraday_volatility
    adjusted_returns = adjusted_short_term - adjusted_long_term
    adjusted_returns = adjusted_returns + (sentiment * adjusted_returns.abs())

    # Incorporate Momentum
    close_to_close_return = df['close'].pct_change()
    final_adjustment = adjusted_returns * close_to_close_return
    final_factor = final_adjustment / df['volume']

    return final_factor
