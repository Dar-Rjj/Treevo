```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Evaluate Price-Volatility Alignment
    price_range = df['high'] - df['low']
    intraday_volatility = price_range * df['volume']
    
    # Compare Movement and Volatility for Sentiment
    sentiment = (high_low_ratio + open_close_ratio) / 2 - intraday_volatility
    
    # Incorporate Momentum Factor
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd = short_ema - long_ema
    signal_line = macd.ewm(span=9, adjust=False).mean()
    momentum = macd - signal_line
    
    # Combine Sentiment and Momentum
    alpha_factor = (sentiment + momentum) / 2
    
    return alpha_factor
```
