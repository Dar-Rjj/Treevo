```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Logarithmic Return
    df['log_return_short'] = np.log(df['close'] / df['close'].shift(5))
    
    # Calculate Long-Term Logarithmic Return
    df['log_return_long'] = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    df['volume_weighted_short'] = df['log_return_short'] * df['volume']
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    df['volume_weighted_long'] = df['log_return_long'] * df['volume']
    
    # Calculate Short-Term Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['short_term_volatility'] = df['true_range'].rolling(window=5).mean()
    
    # Adjust for Volatility
    df['adjusted_short_term'] = (df['volume_weighted_short'] / df['short_term_volatility']) - df['volume_weighted_long']
    
    # Determine Gain and Loss
    df['gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['loss'] = np.where(df['close'] <= df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    
    # Aggregate Gains and Losses
    df['sum_gain'] = df['gain'].rolling(window=14).sum()
    df['sum_loss'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['relative_strength'] = df['sum_gain'] / df['sum_loss']
    
    # Convert to ARSI
    df['ars'] = 100 - (100 / (1 + df['relative_strength']))
    df['ars_i'] = df['ars'] * df['volume'].rolling(window=14).mean() * (df['close'] / df['open'])
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['close_to_close_return'] = df['close'] / df['close'].shift(1) - 1
    df['combined_volatility_return'] = df['intraday_volatility'] * df['close_to_close_return']
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Aggregate Volume Changes
    df['aggregated_volume_changes'] = df['volume_change'].rolling(window=5).sum()
    
    # Calculate Price Oscillator
    df['price_oscillator'] = df['log_return_short'] - df['log_return_long']
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    df['combined_factor'] = (df['ars_i'] * df['log_return_short'] + df['price_oscillator'] - df['volume_weighted_long']) * df['aggregated_volume_changes']
    
    # Final Adjustment
    df['final_factor'] = df['adjusted_short_term'] + df['combined_volatility_return'] + df['combined_factor']
    
    return df['final_factor']
```
