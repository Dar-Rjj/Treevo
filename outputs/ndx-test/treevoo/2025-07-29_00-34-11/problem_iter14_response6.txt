```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Logarithmic Return
    df['short_term_log_return'] = np.log(df['close'] / df['close'].shift(5))
    
    # Calculate Long-Term Logarithmic Return
    df['long_term_log_return'] = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    df['volume_weighted_short_term_log_return'] = df['volume'] * df['short_term_log_return']
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    df['volume_weighted_long_term_log_return'] = df['volume'] * df['long_term_log_return']
    
    # Calculate Short-Term Volatility (Average True Range)
    df['high_low_range'] = df['high'] - df['low']
    df['short_term_volatility'] = df['high_low_range'].rolling(window=5).mean()
    
    # Adjust for Volatility
    df['adjusted_short_term_return'] = df['volume_weighted_short_term_log_return'] / df['short_term_volatility'] - df['volume_weighted_long_term_log_return']
    
    # Determine Gain and Loss
    df['gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['loss'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    
    # Aggregate Gains and Losses over a period (e.g., 14 days)
    df['sum_gain'] = df['gain'].rolling(window=14).sum()
    df['sum_loss'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['sum_gain'] / df['sum_loss']
    
    # Convert to RSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    
    # Incorporate Volume
    df['avg_volume'] = df['volume'].rolling(window=14).mean()
    df['arsi'] = df['rsi'] * df['avg_volume']
    
    # Add Momentum
    df['close_to_open_return'] = df['close'] / df['open'] - 1
    df['advanced_arsi'] = df['arsi'] * df['close_to_open_return']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['combined_intraday_volatility'] = df['intraday_volatility'] * df['close_to_close_return']
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Aggregate Volume Changes over the last 5 days
    df['aggregated_volume_changes'] = df['daily_volume_change'].rolling(window=5).sum()
    
    # Calculate Price Oscillator
    df['price_oscillator'] = df['short_term_log_return'] - df['long_term_log_return']
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    df['combined_factor'] = (df['advanced_arsi'] * df['adjusted_short_term_return'] +
                             df['price_oscillator'] - df['volume_weighted_long_term_log_return'])
    
    # Final Adjustment
    df['final_factor'] = (df['adjusted_short_term_return'] +
                          df['combined_intraday_volatility'] *
                          df['aggregated_volume_changes'] +
                          df['combined_factor'])
    
    return df['final_factor']
```
