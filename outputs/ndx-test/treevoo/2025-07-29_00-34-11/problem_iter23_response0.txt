import pandas as pd

def heuristics_v2(data):
    # Extract the required series from the DataFrame
    open_price = data['open']
    high_price = data['high']
    low_price = data['low']
    close_price = data['close']
    volume = data['volume']

    # Calculate the True Range
    true_range = (high_price - low_price).abs()

    # Calculate the Average True Range over a 14-period window
    avg_true_range = true_range.rolling(window=14).mean()

    # Calculate the Money Flow Multiplier
    money_flow_multiplier = (close_price - low_price - high_price + close_price) / (high_price - low_price)

    # Calculate the Money Flow Volume
    money_flow_volume = money_flow_multiplier * volume

    # Calculate the 14-period Positive and Negative Money Flow
    pos_money_flow = money_flow_volume.apply(lambda x: x if x > 0 else 0).rolling(window=14).sum()
    neg_money_flow = money_flow_volume.apply(lambda x: x if x < 0 else 0).rolling(window=14).sum().abs()

    # Calculate the Money Flow Ratio
    money_flow_ratio = pos_money_flow / neg_money_flow

    # Calculate the Money Flow Index (MFI)
    mfi = 100 - (100 / (1 + money_flow_ratio))

    # Calculate the alpha factor
    alpha_factor = (mfi - avg_true_range) / (mfi + avg_true_range)

    return alpha_factor
