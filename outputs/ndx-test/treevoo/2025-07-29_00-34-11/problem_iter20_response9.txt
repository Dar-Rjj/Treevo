```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Logarithmic Return
    df['short_term_log_return'] = np.log(df['close'] / df['close'].shift(5))
    
    # Calculate Long-Term Logarithmic Return
    df['long_term_log_return'] = np.log(df['close'] / df['close'].shift(20))
    
    # Calculate Volume-Weighted Short-Term Logarithmic Return
    df['volume_weighted_short_term_log_return'] = df['short_term_log_return'] * df['volume']
    
    # Calculate Volume-Weighted Long-Term Logarithmic Return
    df['volume_weighted_long_term_log_return'] = df['long_term_log_return'] * df['volume']
    
    # Calculate Short-Term Volatility (Average True Range)
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    df['avg_true_range'] = df['true_range'].rolling(window=5).mean()
    
    # Adjust for Volatility
    df['adjusted_volatility_return'] = (df['volume_weighted_short_term_log_return'] / df['avg_true_range']) - df['volume_weighted_long_term_log_return']
    
    # Determine Gain and Loss
    df['gain'] = np.where(df['close'] > df['close'].shift(1), df['close'] - df['close'].shift(1), 0)
    df['loss'] = np.where(df['close'] < df['close'].shift(1), df['close'].shift(1) - df['close'], 0)
    
    # Aggregate Gains and Losses
    df['sum_gains'] = df['gain'].rolling(window=14).sum()
    df['sum_losses'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['sum_gains'] / df['sum_losses']
    
    # Convert to ARSI
    df['ars'] = 100 - (100 / (1 + df['rs']))
    df['ars'] *= df['volume'].rolling(window=14).mean()  # Incorporate Volume
    df['ars'] *= (df['close'] / df['open'])  # Add Momentum
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['combined_volatility_return'] = df['intraday_volatility'] * df['close_to_close_return']
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Aggregate Volume Changes
    df['aggregated_volume_changes'] = df['daily_volume_change'].rolling(window=5).sum()
    
    # Calculate Price Oscillator
    df['price_oscillator'] = df['short_term_log_return'] - df['long_term_log_return']
    
    # Combine ARSI, Adjusted Returns, and Price Oscillator
    df['combined_factor'] = (df['ars'] * df['adjusted_volatility_return']) + df['price_oscillator'] - df['volume_weighted_long_term_log_return']
    
    # Final Adjustment
    df['final_factor'] = (df['combined_factor'] + 
                          df['adjusted_volatility_return'] + 
                          df['combined_volatility_return']) * df['aggregated_volume_changes']
    
    return df['final_factor']
```
