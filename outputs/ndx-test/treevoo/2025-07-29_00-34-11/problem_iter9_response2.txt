```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Difference
    df['price_diff'] = df['high'] - df['low']
    
    # Determine Gain and Loss
    df['gain'] = (df['close'] > df['close'].shift(1)) * (df['close'] - df['close'].shift(1))
    df['loss'] = (df['close'] < df['close'].shift(1)) * (df['close'].shift(1) - df['close'])
    
    # Aggregate Gains and Losses over a period of 14 days
    df['avg_gain'] = df['gain'].rolling(window=14).sum()
    df['avg_loss'] = df['loss'].rolling(window=14).sum()
    
    # Calculate Relative Strength
    df['rs'] = df['avg_gain'] / df['avg_loss']
    
    # Convert to ARSI
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    df['ars'] = df['rsi'] * df['volume'].rolling(window=14).mean() * (df['close'] - df['open']) / df['open']
    
    # Calculate Short-Term Return
    df['short_term_return'] = df['close'].pct_change(periods=5)
    
    # Calculate Long-Term Return
    df['long_term_return'] = df['close'].pct_change(periods=20)
    
    # Calculate Volume-Weighted Short-Term Return
    df['vol_weighted_short_term_return'] = df['volume'] * df['short_term_return']
    
    # Calculate Volume-Weighted Long-Term Return
    df['vol_weighted_long_term_return'] = df['volume'] * df['long_term_return']
    
    # Calculate Short-Term Volatility using ATR
    df['atr_5'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Synthesize ARSI with Volume-Weighted Returns and Volatility
    df['combined_ars'] = df['ars'] * df['vol_weighted_short_term_return'] / df['atr_5'] - df['vol_weighted_long_term_return']
    
    # Calculate Intraday Volatility
    df['intraday_vol'] = df['high'] - df['low']
    
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change(periods=1)
    
    # Combine Intraday Volatility and Close-to-Close Return
    df['combined_vol_return'] = df['intraday_vol'] * df['close_to_close_return']
    
    # Introduce a new factor: Volume-Weighted Intraday Volatility
    df['vol_weighted_intraday_vol'] = df['volume'] * df['intraday_vol']
    
    # Final Adjustment
    df['alpha_factor'] = df['combined_ars'] + df['combined_vol_return'] + df['vol_weighted_intraday_vol']
    
    return df['alpha_factor']
```
