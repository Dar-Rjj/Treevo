def heuristics_v2(df):
    # Compute High-Low Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Compute Open-Close Ratio
    open_close_ratio = df['open'] / df['close']
    
    # Average High-Low and Open-Close Ratios
    avg_ratio = (high_low_ratio + open_close_ratio) / 2
    
    # Weight by Volume
    volume_weighted_ratio = avg_ratio * df['volume']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) * df['volume']
    
    # Evaluate Alignment
    sentiment = volume_weighted_ratio / intraday_volatility
    sentiment[sentiment > 1] = 1
    sentiment[sentiment < -1] = -1
    
    # Incorporate Enhanced Momentum into Adjusted Ratios
    close_to_close_return = df['close'].pct_change()
    enhanced_momentum_ratios = volume_weighted_ratio * close_to_close_return
    
    # Final Volume Adjustment
    final_factor = enhanced_momentum_ratios / df['volume']
    
    return final_factor
