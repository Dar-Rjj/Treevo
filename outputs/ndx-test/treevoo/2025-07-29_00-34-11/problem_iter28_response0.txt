```python
import pandas as pd

def heuristics_v2(df):
    # Compute Volume-Weighted Ratios
    high_low_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    open_close_ratio = (df['open'] - df['close']) / (df['open'] + df['close'])
    weighted_ratios = (high_low_ratio + open_close_ratio) * df['volume']
    
    # Measure Price-Volatility Alignment
    intraday_volatility = df['high'] - df['low']
    avg_ratios = (high_low_ratio + open_close_ratio) / 2
    sentiment = (avg_ratios > intraday_volatility).astype(int)
    
    # Incorporate Momentum
    close_to_close_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    final_adjustment = weighted_ratios * close_to_close_return
    
    # Analyze Price Action Patterns
    relative_closing_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    true_range = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    upper_shadow_length = df['high'] - df[['open', 'close']].max(axis=1)
    lower_shadow_length = df[['open', 'close']].min(axis=1) - df['low']
    
    # Investigate Volume-Related Factors
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_weighted_close = df['close'] * df['volume']
    
    # Derive Momentum Indicators from Price Data
    sma = df['close'].rolling(window=5).mean()
    roc = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    directional_movement = df['close'] - df['close'].shift(1)
    
    # Integrate Amount into Analysis
    amount_to_volume_ratio = df['amount'] / df['volume']
    adjusted_price = (df['amount'] / df['volume']) * df['close']
    
    # Combine all factors into a single alpha factor
    alpha_factor = (final_adjustment * sentiment +
                    relative_closing_price_change * true_range *
                    (upper_shadow_length - lower_shadow_length) +
                    volume_ratio * volume_weighted_close +
                    sma * roc * directional_movement +
                    adjusted_price)
    
    return alpha_factor
```
