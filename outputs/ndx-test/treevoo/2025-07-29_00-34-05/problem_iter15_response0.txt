```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['average_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume-Weighted Range
    df['volume_weighted_range'] = (df['daily_range'] / df['average_price']).replace([np.inf, -np.inf], 0) * np.sqrt(df['volume'])
    
    # Calculate Short-Term Rolling Mean of Volume-Weighted Range
    df['short_term_vol_wt_range_mean'] = df['volume_weighted_range'].rolling(window=7).mean()
    
    # Calculate Long-Term Rolling Mean of Volume-Weighted Range
    df['long_term_vol_wt_range_mean'] = df['volume_weighted_range'].rolling(window=21).mean()
    
    # Calculate Short-Term Price Momentum
    df['5_day_ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['price_momentum'] = df['5_day_ema_close'] - df['10_day_ema_close']
    
    # Assess Volatility Adjustment
    df['volatility_score'] = df['daily_range'].rolling(window=20).sum() / 20
    df['adjusted_momentum'] = df['price_momentum'] / df['volatility_score'].replace(0, np.nan)
    
    # Incorporate Volume Influence
    df['5_day_ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_deviation'] = df['5_day_ema_volume'] - df['10_day_ema_volume']
    
    # Combine Momentum, Volatility, and Volume
    df['combined_adjustment'] = df['adjusted_momentum'] * df['volume_deviation']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['short_term_vol_wt_range_mean'] - df['long_term_vol_wt_range_mean']) * df['combined_adjustment']
    
    return df['alpha_factor'].dropna()
```
