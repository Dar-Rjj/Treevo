import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    close_open_diff = df['close'] - df['open']
    enhanced_intraday_range = (intraday_range + close_open_diff) / 2
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine Intraday Measures
    combined_intraday_measure = (high_low_spread + close_open_diff) * df['volume']
    
    # Adjust Combined Measure by Price Momentum
    short_term_momentum = df['close'].ewm(span=5).mean() - df['close'].ewm(span=10).mean()
    daily_price_change = df['close'].diff()
    moving_avg_price_change = daily_price_change.rolling(window=20).mean()
    final_factor_adjustment = (short_term_momentum * moving_avg_price_change) / combined_intraday_measure
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / np.where(average_price == 0, 1, average_price)) * np.sqrt(df['volume'])
    
    # Calculate Short-Term Rolling Average
    short_term_rolling_avg = volume_adjusted_range.rolling(window=10).mean()
    
    # Calculate Long-Term Rolling Average
    long_term_rolling_avg = volume_adjusted_range.rolling(window=30).mean()
    
    # Adjust by Short-Term Momentum and Volume Trend
    short_term_price_momentum = df['high'].ewm(span=5).mean() - df['low'].ewm(span=10).mean()
    volume_momentum = df['volume'].rolling(window=5).sum() - df['volume'].rolling(window=10).sum()
    final_factor = (short_term_price_momentum * volume_momentum) / (enhanced_intraday_range * df['amount'])
    
    # Combine Indicators
    combined_momentum_indicator = high_low_spread * np.sign(volume_momentum)
    rolling_avg_diff = short_term_rolling_avg - long_term_rolling_avg
    alpha_factor = combined_momentum_indicator - rolling_avg_diff
    
    # Combine Price Momentum and Volume Adjusted Range
    short_term_indicator = moving_avg_price_change * short_term_rolling_avg
    long_term_indicator = moving_avg_price_change / long_term_rolling_avg
    final_composite_factor = (short_term_indicator + long_term_indicator) / 2
    
    # Final Alpha Factor
    final_alpha_factor = (final_factor_adjustment + final_composite_factor) / 2
    
    return final_alpha_factor
