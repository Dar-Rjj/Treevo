```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 10-day EMA for close price
    df['ema_close_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()

    # Subtract 10-day EMA from 5-day EMA to get short-term price momentum
    df['price_momentum'] = df['ema_close_5'] - df['ema_close_10']

    # Calculate 5-day and 10-day EMA for volume
    df['ema_volume_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['ema_volume_10'] = df['volume'].ewm(span=10, adjust=False).mean()

    # Subtract 10-day EMA from 5-day EMA to get volume momentum
    df['volume_momentum'] = df['ema_volume_5'] - df['ema_volume_10']

    # Combine price momentum and volume momentum
    df['combined_momentum'] = df['price_momentum'] * df['volume_momentum']

    # Normalize the combined momentum by dividing it with close price
    df['factor_value'] = df['combined_momentum'] / df['close']

    return df['factor_value']
```
