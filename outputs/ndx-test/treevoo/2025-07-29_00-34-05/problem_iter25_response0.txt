```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-Based Factors
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['SMA_50'] - df['SMA_200']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    # Reversal-Based Factors
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    daily_price_range = df['high'] - df['low']
    df['EMA_7_DPR'] = daily_price_range.ewm(span=7, adjust=False).mean()
    df['Daily_Change_EMA'] = df['EMA_7_DPR'].diff()
    
    # Volume-Price Interaction
    obv = np.where(df['close'].diff() >= 0, df['volume'], -df['volume'])
    df['OBV'] = obv.cumsum()
    
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mf_volume = mf_multiplier * df['volume']
    df['CMF_20'] = mf_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    df['VAR'] = (daily_price_range / df['close']).replace([np.inf, -np.inf], np.nan).fillna(0) * df['volume']
    df['ST_Rolling_Avg_VAR'] = df['VAR'].rolling(window=15).mean()
    df['LT_Rolling_Avg_VAR'] = df['VAR'].rolling(window=25).mean()
    
    # Hybrid Factors
    df['Momentum_Volume'] = df['ROC_12'] * df['CMF_20']
    df['Reversal_Volume'] = df['RSI_14'] * df['OBV']
    df['HLR_Vol_Shock'] = df['Daily_Change_EMA'] * np.sign(df['volume'].pct_change().apply(lambda x: 1 if x > 0.6 else 0))
    
    # Finalize Alpha Factor
    price_volatility = df['close'].rolling(window=30).std()
    alpha_factor = (df['ST_Rolling_Avg_VAR'] - df['LT_Rolling_Avg_VAR']) * df['Momentum_Volume'] * df['Reversal_Volume'] * df['HLR_Vol_Shock'] / price_volatility * df['amount']
    
    return alpha_factor.dropna()
```
