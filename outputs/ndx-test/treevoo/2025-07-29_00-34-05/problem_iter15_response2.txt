```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Short-Term (5-day) and Long-Term (20-day) Price Momentum
    df['short_term_momentum'] = df['price_change'].rolling(window=5).mean()
    df['long_term_momentum'] = df['price_change'].rolling(window=20).mean()
    
    # Volume Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    
    # Combine Momentum and Volume Indicators
    df['combined_short_term'] = df['short_term_momentum'] * df['positive_volume_days']
    df['combined_long_term'] = df['long_term_momentum'] * df['negative_volume_days']
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Range
    df['volume_adjusted_range'] = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Short-Term and Long-Term Rolling Sum of Volume Adjusted Range
    df['short_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=10).sum()
    df['long_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=20).sum()
    
    # Compute Momentum Difference
    df['momentum_difference'] = df['long_term_momentum'] - df['short_term_momentum']
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) * df['momentum_difference']
    
    # Combine Combined Momentum and Volume Factors
    df['combined_factors'] = df['combined_short_term'] + df['combined_long_term']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['combined_factors'] 
        + df['short_term_vol_adj_range'] 
        - df['long_term_vol_adj_range'] 
        + df['volume_acceleration']
    )
    
    return df['alpha_factor']
```
