```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price).replace([np.inf, -np.inf], np.nan) * df['volume']
    
    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    short_term_avg = volume_adjusted_range.rolling(window=10).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    long_term_avg = volume_adjusted_range.rolling(window=20).mean()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine Intraday Measures
    combined_intraday_measures = (intraday_high_low_spread + close_open_diff) * df['volume']
    
    # Calculate Short-Term Price Momentum
    ema_5_close = df['close'].ewm(span=5, adjust=False).mean()
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_5_close - ema_10_close
    
    # Calculate Volume Momentum
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    volume_momentum = ema_5_volume - ema_10_volume
    
    # Final Factor Adjustment
    final_factor_adjustment = (short_term_momentum * volume_momentum) / combined_intraday_measures
    
    # Introduce High-Low Range Factor
    high_low_range_factor = (df['high'] - df['low']) / df['close']
    
    # Combine Momentum, Volume, and High-Low Range
    combined_momentum_volatility = (short_term_momentum * volume_momentum * high_low_range_factor) / df['close']
    
    # Combine Factors
    combined_factors = (short_term_avg - long_term_avg + final_factor_adjustment + combined_momentum_volatility)
    
    return combined_factors
```
