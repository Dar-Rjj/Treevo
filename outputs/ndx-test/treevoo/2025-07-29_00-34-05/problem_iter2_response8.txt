import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = short_term_ema - long_term_ema
    
    # Calculate Long-Term Price Momentum
    long_term_period = 20
    long_term_ema = df['close'].ewm(span=long_term_period, adjust=False).mean()
    long_term_momentum = short_term_ema - long_term_ema
    
    # Compute Momentum Difference
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Calculate Volume Acceleration
    volume_change = df['volume'].pct_change(periods=5)
    momentum_by_volume_acceleration = momentum_difference * volume_change
    
    # Calculate Volume Momentum
    volume_short_term_ema = df['volume'].ewm(span=5, adjust=False).mean()
    volume_long_term_ema = df['volume'].ewm(span=10, adjust=False).mean()
    volume_momentum = volume_short_term_ema - volume_long_term_ema
    
    # Combine Momentum and Volume
    combined_factor = (short_term_momentum * volume_momentum) / df['close']
    
    return combined_factor
