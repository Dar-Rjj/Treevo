```python
def heuristics_v2(data):
    # Input:
    # data: A pandas DataFrame with index as 'date' and columns ['open', 'high', 'low', 'close', 'volume']
    # Output:
    # factor_values: A pandas Series indexed by 'date' representing the factor values
    
    # Calculate the high-low ratio
    high_low_ratio = data['high'] / data['low']
    
    # Calculate the close-open return
    close_open_return = (data['close'] - data['open']) / data['open']
    
    # Calculate the high-close distance
    high_close_distance = (data['high'] - data['close']) / data['high']
    
    # Calculate the low-open distance
    low_open_distance = (data['open'] - data['low']) / data['open']
    
    # Combine the four components into a single factor value
    factor_values = (high_low_ratio * close_open_return * high_close_distance * low_open_distance) * data['volume']
    
    return factor_values
```
