```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change of close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the difference between high and low prices
    df['hl_diff'] = df['high'] - df['low']
    
    # Calculate the ratio of volume to its moving average
    df['volume_ma_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_pct_change'] * 0.5) + (df['hl_diff'] * 0.3) + (df['volume_ma_ratio'] * 0.2)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
