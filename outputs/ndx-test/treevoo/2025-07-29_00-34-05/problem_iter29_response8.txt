```python
import pandas as pd

def heuristics_v2(df):
    # Simple Moving Average (SMA) for 14 days
    df['SMA_14'] = df['close'].rolling(window=14).mean()
    
    # Exponential Moving Average (EMA) for 28 days
    df['EMA_28'] = df['close'].ewm(span=28, adjust=False).mean()
    
    # Momentum Factor
    df['Momentum_Factor'] = df['SMA_14'] - df['EMA_28']
    
    # True Range (TR)
    df['Previous_Close'] = df['close'].shift(1)
    df['High_Low'] = df['high'] - df['low']
    df['High_PrevClose'] = (df['high'] - df['Previous_Close']).abs()
    df['Low_PrevClose'] = (df['low'] - df['Previous_Close']).abs()
    df['TR'] = df[['High_Low', 'High_PrevClose', 'Low_PrevClose']].max(axis=1)
    
    # Average True Range (ATR) for 14 days
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Volatility Factor
    df['Volatility_Factor'] = df['ATR_14'] / df['close']
    
    # Volume Moving Average (VMA) for 14 days
    df['VMA_14'] = df['volume'].rolling(window=14).mean()
    
    # Volume Strength Factor
    df['Volume_Strength_Factor'] = (df['volume'] - df['VMA_14']) / df['VMA_14']
    
    # Combined Momentum and Volatility
    df['Combined_Mom_Vol'] = (df['Momentum_Factor'] * df['Volatility_Factor']) / 2
    
    # Combined Momentum and Volume Strength
    df['Combined_Mom_Volume'] = (df['Momentum_Factor'] + df['Volume_Strength_Factor']) / 2
    
    # Integrated Alpha
    df['Integrated_Alpha'] = (df['Combined_Mom_Vol'] + df['Combined_Mom_Volume']) / 2
    
    return df['Integrated_Alpha']
```
