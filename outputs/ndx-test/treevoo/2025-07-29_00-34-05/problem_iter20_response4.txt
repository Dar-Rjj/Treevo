import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-Based Factors
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['50_day_SMA'] - df['200_day_SMA']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Long_Term_Momentum'] = df['5_day_EMA'] - df['20_day_EMA']
    
    # Reversal-Based Factors
    delta = df['close'].diff(1)
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    avg_gain = up.rolling(window=14).mean()
    avg_loss = down.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    df['Close_to_Open_Ratio'] = df['close'] / df['open']
    df['Avg_Close_to_Open_Ratio'] = df['Close_to_Open_Ratio'].rolling(window=30).mean()
    df['Extreme_Price_Movements'] = np.where(df['Close_to_Open_Ratio'] > df['Avg_Close_to_Open_Ratio'] * 1.5, 1,
                                             np.where(df['Close_to_Open_Ratio'] < df['Avg_Close_to_Open_Ratio'] * 0.67, -1, 0))
    
    # Volume-Price Interaction
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    df['5_day_MA_Volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_MA_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Deviation_Score'] = df['5_day_MA_Volume'] - df['10_day_MA_Volume']
    df['Volume_Percentage_Change'] = df['volume'].pct_change(periods=5)
    
    # Volume and Amount-Based Factors
    df['Avg_30_Volume'] = df['volume'].rolling(window=30).mean()
    df['Volume_Breakout'] = np.where(df['volume'] > df['Avg_30_Volume'] * 2, 1, 0)
    
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    df['Volume_Price_Correlation'] = df['close'].pct_change().rolling(window=30).corr(df['volume'].pct_change())
    
    # Hybrid Factors
    df['Momentum_Volume_Hybrid'] = df['ROC_12'] * df['CMF_20']
    df['Reversal_Volume_Hybrid'] = df['RSI_14'] * df['OBV']
    
    # Enhanced Intraday Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Open_Close_Range'] = df['open'] - df['close']
    df['Enhanced_Intraday_Range'] = (df['High_Low_Range'] + df['Open_Close_Range']) / 2
    
    # Volatility Adjustment
    df['Daily_Range'] = df['high'] - df['low']
    df['30_Day_Agg_Range'] = df['Daily_Range'].rolling(window=30).sum()
    df
