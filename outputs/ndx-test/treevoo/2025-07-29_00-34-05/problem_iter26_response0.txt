```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Adjustment
    df['Volume_Impact_Adjustment'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Enhanced Dynamic Price Range
    df['Daily_Enhanced_Dynamic_Price_Range'] = (df['high'] - df['low']) * (df['close'] / df['open'])
    
    # Average Trading Price
    df['Average_Trading_Price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Adjusted Enhanced Range
    df['Volume_Adjusted_Enhanced_Range'] = (df['Daily_Enhanced_Dynamic_Price_Range'] / 
                                            df['Average_Trading_Price'].replace(0, np.nan)) * df['volume']
    
    # Short-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Short_Term_Rolling_Avg_VAER'] = df['Volume_Adjusted_Enhanced_Range'].rolling(window=15).sum() / 15
    
    # Long-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Long_Term_Rolling_Avg_VAER'] = df['Volume_Adjusted_Enhanced_Range'].rolling(window=25).sum() / 25
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=30).std()
    
    # Trade Activity Influence
    df['Trade_Activity_Influence'] = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    df['Combined_Momentum_Volume'] = df['Momentum'] * df['Volume_Impact_Adjustment']
    
    # Adjust for Price Volatility
    df['Adjusted_Momentum_Volume'] = df['Combined_Momentum_Volume'] / df['Price_Volatility']
    
    # High-Low Range Momentum
    def ema(series, n):
        alpha = 2 / (n + 1)
        return series.ewm(alpha=alpha, adjust=False).mean()
    
    df['High_Low_Range'] = df['high'] - df['low']
    df['EMA_High_Low_Range'] = ema(df['High_Low_Range'], 7)
    df['Daily_Change_in_EMA'] = df['EMA_High_Low_Range'].diff()
    
    # Volume Shock
    df['Volume_Percentage_Change'] = df['volume'].pct_change()
    significant_volume_change_threshold = 0.60
    df['Volume_Shock_Flag'] = (df['Volume_Percentage_Change'].abs() > significant_volume_change_threshold).astype(int)
    
    # Price Efficiency Ratio
    df['Price_Efficiency'] = (df['close'] - df['close'].shift(30)) / (df['close'] - df['close'].shift(30)).abs()
    
    # Combine Indicators
    df['Combined_Indicator'] = df['Daily_Change_in_EMA'] * df['Volume_Shock_Flag'].apply(lambda x: 1 if x == 1 else 0)
    
    # Finalize Alpha Factor
    df['Alpha_Factor'] = (df['Short_Term_Rolling_Avg_VAER'] 
                          - df['Long_Term_Rolling_Avg_VAER'] 
                          + df['Combined_Indicator'] 
                          + df['Adjusted_Momentum_Volume'] 
                          * df['Trade_Activity_Influence'] 
                          * df['Price_Efficiency'])
    
    return df['Alpha_Factor']
```
