import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']

    # Combine Intraday Measures and Multiply by Volume
    combined_intraday = (intraday_high_low_spread + close_open_diff) * df['volume']

    # Adjust Combined Measure by Momentum
    short_term_price_momentum = df['close'].ewm(span=5).mean() - df['close'].ewm(span=10).mean()
    volume_momentum = df['volume'].ewm(span=5).mean() - df['volume'].ewm(span=10).mean()
    final_factor_adjustment = (short_term_price_momentum * volume_momentum) / combined_intraday

    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']

    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3

    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price.replace(0, 1)) * df['volume']

    # Calculate Short-Term Rolling Average
    short_term_rolling_avg = volume_adjusted_range.rolling(window=10).mean()

    # Calculate Long-Term Rolling Average
    long_term_rolling_avg = volume_adjusted_range.rolling(window=20).mean()

    # Calculate High-Low Range Momentum
    high_low_range_ema = daily_price_range.ewm(span=5, adjust=False).mean()
    daily_change_in_ema = high_low_range_ema.diff()

    # Calculate Volume Shock
    volume_percentage_change = (df['volume'] / df['volume'].shift(1)).fillna(1) - 1
    significant_volume_change = (volume_percentage_change > 0.5).astype(int)

    # Combine Indicators
    momentum_and_volume_shock = daily_change_in_ema * significant_volume_change
    short_long_difference = short_term_rolling_avg - long_term_rolling_avg
    generated_alpha_factor = momentum_and_volume_shock + short_long_difference

    # Final Alpha Factor
    final_alpha_factor = final_factor_adjustment + generated_alpha_factor

    return final_alpha_factor
