import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    open_close_range = df['close'] - df['open']
    enhanced_intraday_range = (intraday_range + open_close_range) / 2
    
    # Adjust by Short-Term and Long-Term Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    medium_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    short_momentum = short_term_ema - medium_term_ema
    long_momentum = medium_term_ema - long_term_ema
    momentum_diff = short_momentum - long_momentum
    
    # Assess Volatility Adjustment
    daily_range = df['high'] - df['low']
    avg_volatility = daily_range.rolling(window=30).mean()
    
    # Incorporate Volume Influence
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    volume_deviation_score = volume_5day_ma - volume_10day_ma
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine Factors
    adjusted_momentum = momentum_diff * (1 / avg_volatility)
    combined_factor = (adjusted_momentum * volume_deviation_score) + volume_acceleration
    final_alpha_factor = combined_factor / (enhanced_intraday_range * df['close'])
    
    return final_alpha_factor
