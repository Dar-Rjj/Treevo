```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the ratio of high to low prices
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the difference between the high and low prices
    high_low_diff = df['high'] - df['low']
    
    # Calculate the factor value by multiplying the average price, high-low ratio, and high-low difference
    factor = avg_price * high_low_ratio * high_low_diff
    
    return factor
```
