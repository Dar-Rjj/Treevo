```python
import pandas as pd

def heuristics_v2(df):
    # Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Moving Average of Daily Price Changes
    df['ma_20'] = df['daily_price_change'].rolling(window=20).mean()
    df['ma_50'] = df['daily_price_change'].rolling(window=50).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Positive and Negative Volume Days
    df['positive_volume_days'] = df['volume_change'].apply(lambda x: 1 if x > 0 else 0).rolling(window=20).sum()
    df['negative_volume_days'] = df['volume_change'].apply(lambda x: 1 if x < 0 else 0).rolling(window=20).sum()
    
    # Long-Term Momentum and Volume
    df['long_term_factor'] = df['ma_50'] * df['negative_volume_days']
    
    # Short-Term Momentum and Volume
    df['short_term_factor'] = df['ma_20'] * df['positive_volume_days']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['long_term_factor'] - df['short_term_factor']
    
    return df['alpha_factor']
```
