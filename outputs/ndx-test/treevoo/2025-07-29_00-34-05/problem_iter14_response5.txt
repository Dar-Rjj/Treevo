```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['Average_Price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    df['Volume_Adjusted_Range'] = (df['Daily_Price_Range'] / df['Average_Price']).replace([np.inf, -np.inf], np.nan).fillna(0) * df['volume']
    
    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    df['Short_Term_Avg_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=10).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    df['Long_Term_Avg_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=20).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    df['Close_Open_Difference'] = df['close'] - df['open']
    
    # Combine Intraday Measures
    df['Combined_Intraday_Measure'] = (df['Intraday_High_Low_Spread'] + df['Close_Open_Difference']) * df['volume']
    
    # Calculate Short-Term Price Momentum
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    df['5_day_EMA'] = df['close'].ewm(span=5).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10).mean()
    df['Short_Term_Price_Momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    # Calculate Volume Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Positive_Volume_Days'] = (df['Volume_Change'] > 0).rolling(window=20).sum()
    df['Negative_Volume_Days'] = (df['Volume_Change'] < 0).rolling(window=20).sum()
    
    # Calculate Average True Range (ATR)
    df['True_Range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs()
    df['True_Range_1'] = (df['high'] - df['close'].shift(1)).abs()
    df['True_Range_2'] = (df['low'] - df['close'].shift(1)).abs()
    df['True_Range'] = pd.concat([df['True_Range'], df['True_Range_1'], df['True_Range_2']], axis=1).max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Combine Short-Term Price Momentum and Volume Indicator
    df['Short_Term_Volume_Adjusted_Momentum'] = (df['Short_Term_Price_Momentum'] * df['Positive_Volume_Days']) / df['ATR']
    
    # Combine Long-Term Price Momentum and Volume Indicator
    df['Long_Term_Volume_Adjusted_Momentum'] = (df['Short_Term_Price_Momentum'] * df['Negative_Volume_Days']) / df['ATR']
    
    # Final Factor Adjustment
    df['Final_Factor_Adjustment'] = (df['Short_Term_Price_Momentum'] * df['Volume_Change']) / df['Combined_Intraday_Measure']
    
    # Introduce High-Low Range Factor
    df['High_Low_Range_Factor'] = (df['high'] - df['low']) / df['close']
    
    # Combine Momentum, Volume, and High-Low Range
    df['Combined_Momentum_Volume_HighLow'] = (df['Short_Term_Price_Momentum'] * df['Volume_Change'] * df['High_Low_Range_Factor']) / df['close']
    
    # Combine Factors
    df['Combined_Factors'] = (
        df['Short_Term_Avg_Volume_Adjusted_Range'] 
        - df['Long_Term_Avg_Volume_Adjusted_Range'] 
        + df['Final_Factor_Adjustment']
    )
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factors']
    
    return df['Final_Alpha_Factor']
```
