```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    open_close_diff = df['close'] - df['open']
    enhanced_intraday_range = (intraday_range + open_close_diff) / 2
    
    # Calculate Short-Term Price Momentum
    ema_high_5 = df['high'].ewm(span=5, adjust=False).mean()
    ema_low_10 = df['low'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_high_5 - ema_low_10
    
    # Calculate Volume Momentum
    sum_volume_5 = df['volume'].rolling(window=5).sum()
    sum_volume_10 = df['volume'].rolling(window=10).sum()
    volume_momentum = sum_volume_5 - sum_volume_10
    
    # Final Factor Calculation
    momentum_effect = short_term_momentum * volume_momentum
    final_factor = momentum_effect / (enhanced_intraday_range * df['amount'])
    
    return final_factor
```
