```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the normalized volume as a ratio of the current day's volume to the rolling 30-day mean volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Define the Heuristic Alpha Factor
    # The idea is to combine the momentum (close_pct_change) with the volume activity (volume_ratio)
    # Here, we use a simple weighted sum; you can adjust the weights as needed
    heuristic_factor = 0.7 * close_pct_change + 0.3 * volume_ratio
    
    return heuristic_factor
```
