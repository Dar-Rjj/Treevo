```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(15)
    
    # Volume Impact Factor
    volume_impact = df['volume'] / df['volume'].rolling(window=25).mean()
    
    # Calculate Daily Price Range
    daily_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_range / average_price).replace([0], 0.0001) * df['volume']
    
    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    short_term_avg = volume_adjusted_range.rolling(window=10).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    long_term_avg = volume_adjusted_range.rolling(window=20).mean()
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=25).std()
    
    # Trade Amount Influence
    amount_influence = df['amount'] / df['amount'].rolling(window=25).mean()
    
    # Combine Momentum and Volume Impact
    combined_momentum_volume = momentum / volume_impact
    
    # Adjust for Price Volatility
    adjusted_momentum_volume = combined_momentum_volume / price_volatility
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum using EMA
    alpha = 2 / (5 + 1)
    ema_high_low = high_low_range.ewm(alpha=alpha, adjust=False).mean()
    high_low_momentum = ema_high_low - ema_high_low.shift(1)
    
    # Calculate Volume Shock
    volume_change = df['volume'].pct_change()
    volume_shock = (volume_change.abs() > 0.5).astype(int)
    
    # Combine Indicators
    combined_indicator = high_low_momentum * volume_shock
    
    # Finalize Alpha Factor
    alpha_factor = (short_term_avg - long_term_avg) * combined_indicator * adjusted_momentum_volume * amount_influence
    
    return alpha_factor
```
