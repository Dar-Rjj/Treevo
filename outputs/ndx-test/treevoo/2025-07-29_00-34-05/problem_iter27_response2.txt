import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['Momentum_SMA_Crossover'] = df['SMA_50'] - df['SMA_200']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Momentum'] = df['EMA_5'] - df['EMA_10']
    
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Long_Term_Momentum'] = df['EMA_5'] - df['EMA_20']

    # Reversal-Based Factors
    def rsi(series, period=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['RSI'] = rsi(df['close'])

    df['Close_to_Open_Ratio'] = df['close'] / df['open']
    df['Average_Close_to_Open_Ratio'] = df['Close_to_Open_Ratio'].rolling(window=30).mean()
    df['Extreme_Price_Movements'] = (df['Close_to_Open_Ratio'] - df['Average_Close_to_Open_Ratio']) / df['Average_Close_to_Open_Ratio']

    # Volume-Price Interaction
    df['OBV'] = ((df['close'] - df['close'].shift(1)) / abs(df['close'] - df['close'].shift(1))).fillna(0) * df['volume']
    df['OBV'] = df['OBV'].cumsum()

    df['ADL'] = ((2 * df['close'] - df['low'] - df['high']) / (df['high'] - df['low'])) * df['volume']
    df['CMF_20'] = df['ADL'].rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    df['MA_Volume_5'] = df['volume'].rolling(window=5).mean()
    df['MA_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Deviation_Score'] = df['MA_Volume_5'] - df['MA_Volume_10']
    df['Volume_Change_5'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)

    # Volume and Amount-Based Factors
    df['Volume_Spike'] = df['volume'] > df['volume'].rolling(window=30).mean() * 1.5
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    df['Volume_Price_Correlation'] = df['close'].pct_change().rolling(window=5).corr(df['volume'].pct_change())

    # Hybrid Factors
    df['Momentum_Volume_Combined'] = df['ROC_12'] * df['CMF_20']
    df['Reversal_Volume_Combined'] = df['RSI'] * df['OBV']

    # Enhanced Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    df['Enhanced_Intraday_Range'] = df['Intraday_Range'] / 2

    # Volatility Adjustment
    df['Daily_Range'] = df['high'] - df['low']
    df['Aggregate_30_Day_Ranges'] = df['Daily_Range'].rolling(window=30).sum()
    df['Average_Range_Volatility'] = df['Daily_Range'].rolling(window=30).mean()
    df['Price_Volatility_Ratio'] = df['close'] / df['Average_Range_Volatility']
