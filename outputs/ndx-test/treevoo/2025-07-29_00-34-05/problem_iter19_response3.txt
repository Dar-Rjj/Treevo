```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Compute Volume-Weighted Range
    avg_price = (df['high'] + df['low']) / 2
    avg_price[avg_price == 0] = 1  # Ensure non-zero division
    volume_weighted_range = (daily_price_range / avg_price) * np.sqrt(df['volume'])
    
    # Short-Term and Long-Term Rolling Mean of Volume-Weighted Range
    short_term_mean = volume_weighted_range.rolling(window=7).mean()
    long_term_mean = volume_weighted_range.rolling(window=21).mean()
    
    # Short-Term Price Momentum
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_5 - ema_10
    
    # Assess Volatility Adjustment
    avg_volatility = daily_price_range.rolling(window=20).mean()
    
    # Incorporate Volume Influence
    vol_ema_5 = df['volume'].ewm(span=5, adjust=False).mean()
    vol_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_deviation = vol_ema_5 - vol_ema_10
    
    # Adjust Momentum by inverse of Volatility Score and Multiply by Volume Deviation
    combined_adjustment = short_term_momentum / avg_volatility * volume_deviation
    
    # Final Alpha Factor
    alpha_factor = (short_term_mean - long_term_mean) * combined_adjustment
    
    return alpha_factor
```
