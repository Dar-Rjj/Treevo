```python
import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['50_day_SMA'] - df['200_day_SMA']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    # Reversal-Based Factors
    def rsi(series, period):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['RSI_14'] = rsi(df['close'], 14)
    
    df['Close_to_Open_Ratio'] = df['close'] / df['open'].shift(1)
    avg_ratio = df['Close_to_Open_Ratio'].mean()
    df['Extreme_Price_Movements'] = (df['Close_to_Open_Ratio'] - avg_ratio).abs()
    
    # Volume-Price Interaction
    df['OBV'] = ((df['close'] - df['close'].shift(1)) > 0).astype(int) * df['volume'] - ((df['close'] - df['close'].shift(1)) < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    adl = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])
    df['ADL'] = adl * df['volume'] / df['volume'].rolling(window=20).sum()
    df['CMF_20'] = df['ADL'].rolling(window=20).sum()
    
    # Alpha Factor Generation Based on Volume and Amount
    df['30_day_Avg_Volume'] = df['volume'].rolling(window=30).mean()
    df['Volume_Breakout'] = df['volume'] / df['30_day_Avg_Volume']
    
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    df['Volume_Price_Correlation'] = df['close'].pct_change().rolling(window=20).corr(df['volume'].pct_change())
    
    # Hybrid Factors
    df['Momentum_Volume_Hybrid'] = df['ROC_12'] * df['CMF_20']
    
    df['Reversal_Volume_Hybrid'] = df['RSI_14'] * df['OBV']
    
    # Select the final factor to return
    df['Alpha_Factor'] = (df['SMA_Crossover'] + df['ROC_12'] + df['RSI_14'] + df['Extreme_Price_Movements'] + df['CMF_20'] + df['Volume_Breakout'] + df['VWAP'] + df['Volume_Price_Correlation'] + df['Momentum_Volume_Hybrid'] + df['Reversal_Volume_Hybrid']).dropna()
    
    return df['Alpha_Factor']
```
