```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 5 and 10 day EMA for the close price
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = close_ema_5 - close_ema_10
    
    # Calculate the 5 and 10 day EMA for the volume
    volume_ema_5 = df['volume'].ewm(span=5, adjust=False).mean()
    volume_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Volume Momentum
    volume_momentum = volume_ema_5 - volume_ema_10
    
    # Combine momentum and volume
    combined_momentum = short_term_momentum * volume_momentum
    
    # Calculate the 20 day EMA of the close price for normalization
    close_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Final factor: Divide the combined momentum by the 20-day EMA of close price
    factor = combined_momentum / close_ema_20
    
    return factor
```
