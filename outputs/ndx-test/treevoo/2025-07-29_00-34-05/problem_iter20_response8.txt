import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['Momentum_SMA_Crossover'] = df['SMA_50'] - df['SMA_200']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    # Reversal-Based Factors
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    df['Close_to_Open_Ratio'] = df['close'] / df.shift(1)['open']
    df['Avg_Close_to_Open_Ratio'] = df['Close_to_Open_Ratio'].rolling(window=30).mean()
    df['Extreme_Price_Movement'] = (df['Close_to_Open_Ratio'] - df['Avg_Close_to_Open_Ratio']) / df['Avg_Close_to_Open_Ratio'].std()
    
    # Volume-Price Interaction
    df['OBV'] = (df['close'].diff() / df['close']).apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0).cumsum() * df['volume']
    
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Volume and Amount-Based Factors
    df['Volume_Spike'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    df['Volume_Price_Correlation'] = df['close'].pct_change().rolling(window=30).corr(df['volume'].pct_change())
    
    # Hybrid Factors
    df['Momentum_Volume_Hybrid'] = df['ROC_12'] * df['CMF_20']
    
    df['Reversal_Volume_Hybrid'] = df['RSI_14'] * df['OBV']
    
    # Enhanced Hybrid Factors
    df['Momentum_Reversal_Volume_Hybrid'] = df['ROC_12'] * df['RSI_14'] * df['CMF_20']
    
    df['Volume_Price_Correlation_SMA_Crossover'] = df['Volume_Price_Correlation'] * df['Momentum_SMA_Crossover']
    
    # Advanced Volume-Based Indicators
    df['Volume_Spike_60'] = df['volume'] / df['volume'].rolling(window=60).mean()
    
    df['Volume_Weighted_Close'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    
    # Advanced Momentum and Reversal Indicators
    df['Long_Term_ROC_252'] = df['close'].pct_change(periods=252)
    
    # Adaptive RSI (not implemented, placeholder for future work)
    
    # Advanced Volume-Price Interaction
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['Accumulation_Distribution_Line'] = (adl * df['volume']).cumsum()
    
    df['Volume_Efficiency_Ratio'] = df['close'].diff().abs().rolling(window=30).sum() / (df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min())
    
    # Advanced Hybrid Indicators
    df['Integrated_Momentum_Reversal_Volume_Efficiency'] = df['ROC_12'] * df['RSI_14'] * df['CMF_20'] * df['Volume_Efficiency_Ratio']
    
    df['Dynamic_Momentum_Reversal_Volume_Weighted_Close'] = df
