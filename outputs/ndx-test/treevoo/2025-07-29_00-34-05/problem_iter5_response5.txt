    df['price_range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['avg_price'] = (df['high'] + df['low'] + df['close']) / 3

    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    df['short_term_avg_var'] = df['volume_adjusted_range'].rolling(window=10).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    df['long_term_avg_var'] = df['volume_adjusted_range'].rolling(window=20).mean()
    
    # Calculate Price Momentum
    df['daily_price_change'] = df['close'].diff()
    df['20_day_moving_avg'] = df['daily_price_change'].rolling(window=20).mean()
    df['50_day_moving_avg'] = df['daily_price_change'].rolling(window=50).mean()
    
    # Calculate Volume Indicator
    df['volume_change'] = df['volume'].diff()
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    
    # Combine Price Momentum and Volume Indicator
    df['long_term_momentum_volume'] = df['50_day_moving_avg'] * df['positive_volume_days']
    df['short_term_momentum_volume'] = df['20_day_moving_avg'] * df['negative_volume_days']
    
    # Calculate Short-Term Price Momentum
    df['short_term_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum Difference
    df['momentum_difference'] = df['short_term_ema'] - df['long_term_ema']
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Combine Factors
    df['combined_factors'] = (
        df['short_term_avg_var'] - 
        df['long_term_avg_var'] + 
        df['momentum_difference'] * df['volume_acceleration'] + 
        df['long_term_momentum_volume'] - 
        df['short_term_momentum_volume']
    )
    
    # Final Alpha Factor
    alpha_factor = df['combined_factors']
    
    return alpha_factor
