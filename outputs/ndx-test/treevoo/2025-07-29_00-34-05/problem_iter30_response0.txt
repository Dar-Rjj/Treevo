```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the volume weighted close price
    vw_close = df['close'] * df['volume']
    
    # Calculate the momentum factor using the average price
    momentum = (avg_price - avg_price.shift(10)) / avg_price.shift(10)
    
    # Calculate the volume-weighted return
    vw_return = (vw_close - vw_close.shift(1)) / vw_close.shift(1)
    
    # Combine the momentum and volume-weighted return to create the alpha factor
    alpha_factor = momentum * vw_return
    
    return alpha_factor
```
