```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-Low Spread
    intraday_high_low = df['high'] - df['low']
    
    # Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine Intraday Measures
    combined_intraday = (intraday_high_low + close_open_diff) * df['volume']
    
    # Short-Term Price Momentum (5-day SMA)
    price_change = df['close'].diff()
    short_term_momentum = price_change.rolling(window=5).mean()
    
    # Long-Term Price Momentum (20-day SMA)
    long_term_momentum = price_change.rolling(window=20).mean()
    
    # Volume Indicator
    volume_change = df['volume'].diff()
    positive_volume_days = (df['volume'].diff() > 0).rolling(window=20).sum()
    negative_volume_days = (df['volume'].diff() < 0).rolling(window=20).sum()
    
    # Combine Short-Term Price Momentum and Volume Indicator
    combined_short_term = short_term_momentum * positive_volume_days
    
    # Combine Long-Term Price Momentum and Volume Indicator
    combined_long_term = long_term_momentum * negative_volume_days
    
    # Volume Adjusted Range
    volume_adjusted_range = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Short-Term Rolling Sum of Volume Adjusted Range
    short_term_volume_range = volume_adjusted_range.rolling(window=10).sum()
    
    # Long-Term Rolling Sum of Volume Adjusted Range
    long_term_volume_range = volume_adjusted_range.rolling(window=20).sum()
    
    # Momentum Difference
    momentum_difference = long_term_momentum - short_term_momentum
    
    # Volume Acceleration
    volume_acceleration = (df['volume'].pct_change()) * momentum_difference
    
    # Volume Momentum
    volume_ema_5 = df['volume'].ewm(span=5, adjust=False).mean()
    volume_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_momentum = volume_ema_5 - volume_ema_10
    
    # Final Factor Adjustment
    final_factor = (short_term_momentum * volume_momentum) / combined_intraday
    
    # Combine Combined Momentum and Volume Factors
    combined_factors = combined_short_term + combined_long_term
    
    # Final Alpha Factor
    alpha_factor = combined_factors + short_term_volume_range - long_term_volume_range + volume_acceleration
    
    return alpha_factor
```
