```python
import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    # Simple Moving Average Crossover
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['SMA_50'] - df['SMA_200']
    
    # Rate of Change (ROC)
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    # Reversal-Based Factors
    # Short-Term Overbought/Oversold Indicator (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Extreme Price Movements
    df['Close_to_Open_Ratio'] = df['close'] / df.shift(1)['open']
    df['Avg_Close_to_Open_Ratio'] = df['Close_to_Open_Ratio'].rolling(window=30).mean()
    df['Extreme_Price_Movements'] = (df['Close_to_Open_Ratio'] - df['Avg_Close_to_Open_Ratio']) / df['Avg_Close_to_Open_Ratio']
    
    # Volume-Price Interaction
    # On-Balance Volume (OBV)
    df['OBV'] = (df['close'] > df.shift(1)['close']).astype(int) * df['volume'] - (df['close'] < df.shift(1)['close']).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Chaikin Money Flow (CMF)
    mfm = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mfvol = mfm * df['volume']
    df['CMF'] = mfvol.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Volume and Amount-Based Factors
    # Volume Breakout
    df['Avg_Volume_30'] = df['volume'].rolling(window=30).mean()
    df['Volume_Breakout'] = df['volume'] / df['Avg_Volume_30']
    
    # Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['VWAP'] = (df['Typical_Price'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Volume-Price Correlation
    df['Price_Change'] = df['close'].pct_change()
    df['Volume_Price_Correlation'] = df['Price_Change'].rolling(window=20).corr(df['volume'])
    
    # Hybrid Factors
    # Momentum and Volume Combined
    df['Momentum_Volume_Hybrid'] = df['ROC_12'] * df['CMF']
    
    # Reversal and Volume Combined
    df['Reversal_Volume_Hybrid'] = df['RSI'] * df['OBV']
    
    # Select the final factor to return
    return df['Momentum_Volume_Hybrid']
```
