```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum Factor
    df['momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Assess Volatility Adjustment using High-Low Range
    df['daily_range'] = df['high'] - df['low']
    df['avg_range_volatility'] = df['daily_range'].rolling(window=30).mean()
    
    # Incorporate High-Low Range Momentum
    alpha = 2 / (5 + 1)  # Smoothing factor for EMA, N=5
    df['ema_high_low_range'] = df['daily_range'].ewm(alpha=alpha, adjust=False).mean()
    df['ema_change'] = df['ema_high_low_range'] - df['ema_high_low_range'].shift(1)
    
    # Calculate Volume Shock
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_shock'] = np.sign(df['volume_change'].apply(lambda x: x if abs(x) > 0.5 else 0))
    
    # Combine Factors
    df['adjusted_momentum'] = df['momentum'] / df['avg_range_volatility']
    df['final_alpha'] = df['adjusted_momentum'] * df['ema_change'] * df['volume_shock']
    
    return df['final_alpha']
```
