import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['daily_price_change'] = df['close'].diff()
    df['20_day_moving_avg'] = df['daily_price_change'].rolling(window=20).mean()
    df['50_day_moving_avg'] = df['daily_price_change'].rolling(window=50).mean()

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate High-Low Range Momentum
    alpha = 2 / (5 + 1)
    df['hl_ema'] = df['high_low_range'].ewm(alpha=alpha, adjust=False).mean()
    df['daily_hl_ema_change'] = df['hl_ema'].diff()

    # Calculate Volume Indicator
    df['volume_change'] = df['volume'].diff()
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    df['significant_volume_change'] = (df['volume_change'].abs() > 0.5 * df['volume'].shift(1)).astype(int)

    # Combine Indicators
    df['long_term_factor'] = df['50_day_moving_avg'] * df['negative_volume_days']
    df['short_term_factor'] = df['20_day_moving_avg'] * df['positive_volume_days']
    df['combined_factors'] = df['long_term_factor'] + df['short_term_factor']

    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] * df['daily_hl_ema_change'].apply(lambda x: 1 if x > 0 else -1) * df['significant_volume_change']

    return df['alpha_factor']
