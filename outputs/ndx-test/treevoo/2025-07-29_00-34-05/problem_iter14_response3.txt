```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Short-Term Momentum
    df['5_day_ema_price_change'] = df['daily_price_change'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_price_change'] = df['daily_price_change'].ewm(span=10, adjust=False).mean()
    df['short_term_momentum'] = df['5_day_ema_price_change'] - df['10_day_ema_price_change']
    
    # Long-Term Momentum
    df['20_day_sma_price_change'] = df['daily_price_change'].rolling(window=20).mean()
    df['50_day_sma_price_change'] = df['daily_price_change'].rolling(window=50).mean()
    df['long_term_momentum'] = df['20_day_sma_price_change'] - df['50_day_sma_price_change']
    
    # Volume Indicator
    df['volume_change'] = df['volume'].diff()
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    
    # Volume Momentum
    df['5_day_ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_momentum'] = df['5_day_ema_volume'] - df['10_day_ema_volume']
    
    # Combine Short-Term and Volume
    df['short_term_factor'] = df['short_term_momentum'] * df['positive_volume_days'] * df['volume_momentum']
    
    # Combine Long-Term and Volume
    df['long_term_factor'] = df['long_term_momentum'] * df['negative_volume_days'] * df['volume_momentum']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['short_term_factor'] + (df['long_term_factor'] / df['short_term_factor'])
    
    return df['alpha_factor']
```
