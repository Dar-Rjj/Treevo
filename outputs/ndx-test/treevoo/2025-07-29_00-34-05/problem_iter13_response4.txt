import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range_1 = df['high'] - df['low']
    intraday_range_2 = (df['close'] - df['open']).abs()
    enhanced_intraday_range = (intraday_range_1 + intraday_range_2) / 2
    
    # Short-Term Price Momentum
    ema_high_5 = df['high'].ewm(span=5, adjust=False).mean()
    ema_low_10 = df['low'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_high_5 - ema_low_10
    
    # Volume Momentum
    sum_volume_5 = df['volume'].rolling(window=5).sum()
    sum_volume_10 = df['volume'].rolling(window=10).sum()
    volume_momentum = sum_volume_5 - sum_volume_10
    
    # Final Factor
    final_factor = (short_term_momentum * volume_momentum) / (enhanced_intraday_range * df['amount'])
    
    return final_factor
