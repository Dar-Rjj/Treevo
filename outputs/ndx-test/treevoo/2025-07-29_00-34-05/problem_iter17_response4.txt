```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    intraday_diff = df['close'] - df['open']
    enhanced_intraday_range = (intraday_range + intraday_diff) / 2
    
    # Short-Term and Long-Term Momentum
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    short_term_momentum = ema_5 - ema_10
    long_term_momentum = ema_10 - ema_20
    
    # Volume Momentum
    vol_5 = df['volume'].rolling(window=5).mean()
    vol_10 = df['volume'].rolling(window=10).mean()
    volume_momentum = vol_5 - vol_10
    
    # Daily Range and Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range_volatility = daily_range.rolling(window=30).mean()
    
    # Volume Deviation Score
    avg_30_day_vol = df['volume'].rolling(window=30).mean()
    volume_deviation_score = df['volume'] - avg_30_day_vol
    volume_acceleration = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Final Alpha Factor
    momentum_diff = short_term_momentum - long_term_momentum
    inverse_volatility = 1 / avg_daily_range_volatility
    adjusted_momentum = momentum_diff * inverse_volatility * volume_acceleration
    final_alpha_factor = adjusted_momentum / (enhanced_intraday_range * df['close'])
    final_alpha_factor += volume_deviation_score
    
    return final_alpha_factor
```
