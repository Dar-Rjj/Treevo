```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5 and 10 day EMAs for the close price
    df['close_ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['close_ema_10'] = df['close'].ewm(span=10, adjust=False).mean()

    # Calculate short-term price momentum
    df['short_term_price_momentum'] = df['close_ema_5'] - df['close_ema_10']

    # Calculate 5 and 10 day EMAs for the volume
    df['volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate volume momentum
    df['volume_momentum'] = df['volume_ema_5'] - df['volume_ema_10']

    # Combine momentum and volume: Multiply Short-Term Price Momentum by Volume Momentum
    df['combined_momentum'] = df['short_term_price_momentum'] * df['volume_momentum']

    # Calculate 20-day EMA of close price
    df['close_ema_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Divide the result by 20-day EMA to get the final factor
    df['factor'] = df['combined_momentum'] / df['close_ema_20']

    return df['factor']
```
