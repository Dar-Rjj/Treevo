```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate short-term and long-term price momentum
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    # Calculate long-term momentum
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute momentum difference
    df['Momentum_Difference'] = df['Short_Term_Momentum'] - df['20_day_EMA']
    
    # Calculate daily price range
    df['Daily_Price_Range'] = (df['high'] - df['low']) / df['close']
    
    # Compute volume-weighted range
    df['Volume_Weighted_Range'] = df['Daily_Price_Range'] * (np.sqrt(df['volume']))
    
    # Short-term rolling average of volume-weighted range
    df['Short_Term_Volume_Weighted_Range'] = df['Volume_Weighted_Range'].rolling(window=10).mean()
    
    # Long-term rolling average of volume-weighted range
    df['Long_Term_Volume_Weighted_Range'] = df['Volume_Weighted_Range'].rolling(window=30).mean()
    
    # Combine price momentum and volume-weighted range
    df['Short_Term_Indicator'] = df['20_day_EMA'] * df['Short_Term_Volume_Weighted_Range']
    df['Long_Term_Indicator'] = df['20_day_EMA'] / df['Long_Term_Volume_Weighted_Range']
    df['Composite_Factor'] = (df['Short_Term_Indicator'] + df['Long_Term_Indicator']) / 2
    
    # Adjust momentum by volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['20_Day_Range_Sum'] = df['Daily_Range'].rolling(window=20).sum()
    df['Average_Range_Volatility'] = df['20_Day_Range_Sum'] / 20
    df['Adjusted_Momentum'] = df['Short_Term_Momentum'] / df['Average_Range_Volatility']
    
    # Incorporate volume influence
    df['5_day_EMA_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['Volume_Deviation'] = df['5_day_EMA_volume'] - df['10_day_EMA_volume']
    
    # Combine momentum, volatility, and volume
    df['Adjusted_Momentum_Volatility'] = df['Adjusted_Momentum'] * (1 / df['Volume_Deviation'])
    
    # Crossover integration
    df['Composite_Indicator'] = (df['Momentum_Difference'] * df['Short_Term_Volume_Weighted_Range']) / (df['Long_Term_Volume_Weighted_Range'] * df['Volume_Weighted_Range'])
    
    # Final integrated alpha factor
    df['Final_Alpha_Factor'] = (df['Composite_Factor'] + df['Adjusted_Momentum_Volatility'] + df['Composite_Indicator']) / 3
    
    return df['Final_Alpha_Factor'].dropna()
```
