```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(25)
    
    # Volume Impact Adjustment
    df['Volume_Impact_Adjustment'] = df['volume'] / df['volume'].rolling(window=30).median()
    
    # Daily Enhanced Dynamic Price Range
    df['Daily_Enhanced_Dynamic_Price_Range'] = (df['high'] - df['low']) * (df['close'] - df['open'])
    
    # Average Trading Price
    df['Average_Trading_Price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Adjusted Enhanced Range
    df['Volume_Adjusted_Enhanced_Range'] = (df['Daily_Enhanced_Dynamic_Price_Range'] / df['Average_Trading_Price']).replace([np.inf, -np.inf], np.nan) * np.log(df['volume'])
    
    # Short-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Short_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range'] = df['Volume_Adjusted_Enhanced_Range'].rolling(window=10).mean()
    
    # Long-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Long_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range'] = df['Volume_Adjusted_Enhanced_Range'].rolling(window=30).mean()
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=20).std()
    
    # Trade Activity Influence
    df['Trade_Activity_Influence'] = df['amount'] / df['amount'].rolling(window=30).median()
    
    # Combine Momentum and Volume Impact
    df['Combined_Momentum_Volume_Impact'] = df['Momentum'] * df['Volume_Impact_Adjustment']
    
    # Adjust for Price Volatility
    df['Adjusted_Combined_Momentum_Volume_Impact'] = df['Combined_Momentum_Volume_Impact'] / df['Price_Volatility']
    
    # High-Low Range Momentum
    N = 10
    alpha = 2 / (N + 1)
    df['High_Low_Range'] = df['high'] - df['low']
    df['EMA_High_Low_Range'] = df['High_Low_Range'].ewm(alpha=alpha, adjust=False).mean()
    df['Daily_Change_in_EMA_High_Low_Range'] = df['EMA_High_Low_Range'] - df['EMA_High_Low_Range'].shift(1)
    
    # Volume Shock
    df['Volume_Percentage_Change'] = df['volume'].pct_change()
    significant_volume_change_threshold = 0.5
    df['Volume_Shock_Flag'] = (df['Volume_Percentage_Change'].abs() > significant_volume_change_threshold).astype(int)
    
    # Price Efficiency Ratio
    df['Absolute_Price_Change'] = df['close'] - df['close'].shift(30)
    df['Price_Efficiency_Ratio'] = (df['close'] - df['close'].shift(30)) / df['Absolute_Price_Change'].replace(0, np.nan)
    
    # Combined Indicator
    df['Combined_Indicator'] = df['Daily_Change_in_EMA_High_Low_Range'] * df['Volume_Shock_Flag']
    
    # Finalize Alpha Factor
    df['Alpha_Factor'] = (
        df['Short_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range'] +
        df['Long_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range'] +
        df['Adjusted_Combined_Momentum_Volume_Impact'] +
        df['Combined_Indicator'] * df['Trade_Activity_Influence'] * df['Price_Efficiency_Ratio']
    )
    
    return df['Alpha_Factor']
```
