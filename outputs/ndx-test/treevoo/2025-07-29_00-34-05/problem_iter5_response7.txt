import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['average_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    df['volume_adjusted_range'] = (df['daily_price_range'] / df['average_price']).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Calculate Short-Term Rolling Average
    df['short_term_rolling_avg'] = df['volume_adjusted_range'].rolling(window=10).mean()
    
    # Calculate Long-Term Rolling Average
    df['long_term_rolling_avg'] = df['volume_adjusted_range'].rolling(window=20).mean()
    
    # Calculate High-Low Range Momentum
    def ema(values, alpha):
        ema_values = [values[0]]
        for val in values[1:]:
            ema_values.append(alpha * val + (1 - alpha) * ema_values[-1])
        return ema_values
    
    smoothing_factor = 2 / (5 + 1)
    df['ema_high_low_range'] = ema(df['daily_price_range'], smoothing_factor)
    df['daily_change_ema'] = df['ema_high_low_range'].diff().fillna(0)
    
    # Calculate Volume Shock
    df['volume_percentage_change'] = df['volume'].pct_change() * 100
    df['volume_shock_flag'] = (df['volume_percentage_change'] > 50).astype(int)
    
    # Combine Indicators
    df['combined_indicator'] = df['daily_change_ema'] * df['volume_shock_flag']
    df['rolling_difference'] = df['short_term_rolling_avg'] - df['long_term_rolling_avg']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['combined_indicator'] + df['rolling_difference']
    
    # Output Final Alpha Factor
    return df['alpha_factor']
