import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate Exponential Moving Averages (EMA)
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Compute Moving Average Convergence Divergence (MACD)
    df['MACD_Line'] = df['EMA_12'] - df['EMA_26']
    df['Signal_Line'] = df['MACD_Line'].ewm(span=9, adjust=False).mean()
    df['MACD_Histogram'] = df['MACD_Line'] - df['Signal_Line']

    # Evaluate Momentum Indicators
    df['ROC_10'] = df['close'].pct_change(periods=10) * 100
    df['ROC_20'] = df['close'].pct_change(periods=20) * 100

    # Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))

    # Analyze Volatility
    df['Historical_Volatility_20'] = df['close'].pct_change().rolling(window=20).std() * (252 ** 0.5)
    df['Historical_Volatility_50'] = df['close'].pct_change().rolling(window=50).std() * (252 ** 0.5)

    # True Range and Average True Range (ATR)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    df['ATR_14'] = true_range.rolling(window=14).mean()

    # Investigate Volume-Price Relationship
    df['OBV'] = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(), 0)
    positive_money_flow_14 = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_14 = negative_money_flow.rolling(window=14).sum()
    mfi_ratio = positive_money_flow_14 / negative_money_flow_14
    df['MFI'] = 100 - (100 / (1 + mfi_ratio))
