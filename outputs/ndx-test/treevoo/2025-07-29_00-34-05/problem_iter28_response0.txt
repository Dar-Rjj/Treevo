```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(20)
    
    # Volume Impact Adjustment
    volume_impact = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Enhanced Dynamic Price Range
    enhanced_dynamic_range = (df['high'] - df['low']) * (df['close'] / df['open'])
    
    # Average Trading Price
    avg_trading_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Adjusted Enhanced Range
    volume_adjusted_enhanced_range = (enhanced_dynamic_range / avg_trading_price.replace(0, float('inf'))) * df['volume']
    
    # Short-Term Rolling Avg of Volume Adjusted Enhanced Range
    short_term_avg = volume_adjusted_enhanced_range.rolling(window=15).sum() / 15
    
    # Long-Term Rolling Avg of Volume Adjusted Enhanced Range
    long_term_avg = volume_adjusted_enhanced_range.rolling(window=25).sum() / 25
    
    # Combine Momentum and Volume Impact
    combined_momentum_volume = momentum * volume_impact
    
    # Trade Activity Influence
    trade_activity = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Finalize Alpha Factor
    alpha_factor = (short_term_avg - long_term_avg) * combined_momentum_volume * trade_activity
    
    return alpha_factor
```
