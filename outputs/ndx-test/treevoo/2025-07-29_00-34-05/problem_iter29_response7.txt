```python
def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the daily return based on close price
    daily_return = df['close'].pct_change()
    
    # Calculate the moving average of the daily return over a 5-day window
    ma_daily_return = daily_return.rolling(window=5).mean()
    
    # Calculate the factor value as the product of the average price and the moving average of the daily return
    factor_value = avg_price * ma_daily_return
    
    return factor_value
```
