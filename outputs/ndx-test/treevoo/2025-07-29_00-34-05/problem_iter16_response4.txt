```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day and 10-day EMA
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    # Short-term momentum (difference between 5-day and 10-day EMA)
    df['short_term_momentum'] = df['ema_5'] - df['ema_10']
    
    # Calculate 20-day EMA for long-term momentum
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    # Long-term momentum
    df['long_term_momentum'] = df['close'] - df['ema_20']
    
    # Daily price range
    df['daily_price_range'] = (df['high'] - df['low']) / df['close']
    
    # Volume adjusted range
    df['volume_adjusted_range'] = (df['high'] - df['low']) / ((df['high'] + df['low']) / 2) * np.sqrt(df['volume'])
    
    # Short-term rolling average of volume adjusted range
    df['vol_adj_range_10d_avg'] = df['volume_adjusted_range'].rolling(window=10).mean()
    
    # Long-term rolling average of volume adjusted range
    df['vol_adj_range_30d_avg'] = df['volume_adjusted_range'].rolling(window=30).mean()
    
    # Combine price momentum and volume adjusted range
    df['short_term_indicator'] = df['ema_20'] * df['vol_adj_range_10d_avg']
    df['long_term_indicator'] = df['ema_20'] / df['vol_adj_range_30d_avg']
    
    # Final composite factor
    df['composite_factor'] = (df['short_term_indicator'] + df['long_term_indicator']) / 2
    
    # High-Low Range Factor
    df['high_low_range_factor'] = (df['high'] - df['low']) / df['close']
    
    # Combine Momentum, Volume, and High-Low Range
    df['combined_factor'] = (df['short_term_momentum'] * 
                             df['volume_adjusted_range'] * 
                             df['high_low_range_factor']) / df['close']
    
    # Adjust Momentum
    df['average_range_30d'] = df['high_low_range_factor'].rolling(window=30).mean()
    df['adjusted_momentum'] = df['short_term_momentum'] * (1 / df['average_range_30d'])
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_momentum'] + df['vol_adj_range_10d_avg'] - df['vol_adj_range_30d_avg']
    
    return df['final_alpha_factor']
```
