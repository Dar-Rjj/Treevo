import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range_1 = df['high'] - df['low']
    intraday_range_2 = df['close'] - df['open']
    enhanced_intraday_range = (intraday_range_1 + intraday_range_2) / 2
    
    # Adjust by Short-Term and Long-Term Momentum
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    short_term_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    short_term_momentum = short_term_ema_5 - short_term_ema_10
    long_term_momentum = short_term_ema_5 - long_term_ema_20
    momentum_difference = short_term_momentum - long_term_momentum
    
    # Assess Volatility Adjustment
    daily_range = df['high'] - df['low']
    thirty_day_range = daily_range.rolling(window=30).sum() / 30
    volatility_score = 1 / thirty_day_range  # inverse of average range volatility
    
    # Incorporate Volume Influence
    volume_5_day_ma = df['volume'].rolling(window=5).mean()
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_deviation_score = (volume_5_day_ma - volume_10_day_ma) / (df['volume'].shift(5) - df['volume'])
    
    # Combine Factors
    adjusted_momentum = momentum_difference * volatility_score
    final_alpha_factor = (adjusted_momentum * volume_deviation_score + volume_deviation_score) / (enhanced_intraday_range * df['close'])
    
    return final_alpha_factor
