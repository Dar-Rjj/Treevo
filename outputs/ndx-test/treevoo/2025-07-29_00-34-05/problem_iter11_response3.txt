```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['short_term_momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    # Calculate Long-Term Momentum
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['long_term_momentum'] = df['20_day_EMA'] - df['10_day_EMA']

    # Compute Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']

    # Compute Volume Adjusted Range
    mid_price = (df['high'] + df['low']) / 2
    df['volume_adjusted_range'] = df['daily_price_range'] / mid_price * np.sqrt(df['volume'])

    # Short-Term Rolling Average of Volume Adjusted Range
    df['short_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=10).mean()

    # Long-Term Rolling Average of Volume Adjusted Range
    df['long_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=30).mean()

    # Introduce High-Low Range Factor
    df['high_low_range_factor'] = (df['high'] - df['low']) / df['close']

    # Combine Price Momentum and Volume Adjusted Range
    df['short_term_indicator'] = df['20_day_EMA'] * df['short_term_vol_adj_range']
    df['long_term_indicator'] = df['20_day_EMA'] / df['long_term_vol_adj_range']
    df['composite_factor'] = (df['short_term_indicator'] + df['long_term_indicator']) / 2

    # Combine Momentum, Volume, and High-Low Range
    df['momentum_volume_high_low'] = df['short_term_momentum'] * df['short_term_vol_adj_range'] * df['high_low_range_factor'] / df['close']

    # Adjust Momentum
    df['30_day_avg_range'] = df['daily_price_range'].rolling(window=30).mean()
    df['adjusted_momentum'] = df['momentum_volume_high_low'] * (1 / df['30_day_avg_range'])

    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_momentum'] + df['volume_adjusted_range']

    return df['final_alpha_factor']
```
