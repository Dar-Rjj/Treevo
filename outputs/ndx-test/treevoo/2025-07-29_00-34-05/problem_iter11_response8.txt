def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the average of the open and close prices
    df['avg_open_close'] = (df['open'] + df['close']) / 2
    
    # Calculate the range of the high and low prices
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate the ratio of the volume to the high-low range
    df['volume_to_range_ratio'] = df['volume'] / df['high_low_range']
    
    # Calculate the novel alpha factor
    df['alpha_factor'] = (df['avg_open_close'] * df['close_pct_change']) + df['volume_to_range_ratio']
    
    return df['alpha_factor'].dropna()
