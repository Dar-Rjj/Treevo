import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_200'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['SMA_50'] - df['SMA_200']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['EMA_Crossover'] = df['EMA_10'] - df['EMA_30']
    
    # Reversal-Based Factors
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    df['Close_to_Open_Ratio'] = df['close'] / df['open']
    mean_ratio = df['Close_to_Open_Ratio'].rolling(window=30).mean()
    std_dev_ratio = df['Close_to_Open_Ratio'].rolling(window=30).std()
    df['Extreme_Price_Movements'] = (df['Close_to_Open_Ratio'] - mean_ratio) / std_dev_ratio
    
    df['Stochastic_Oscillator_K'] = 100 * (df['close'] - df['low'].rolling(window=14).min()) / (df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min())
    df['Stochastic_Oscillator_D'] = df['Stochastic_Oscillator_K'].rolling(window=3).mean()
    
    # Volume-Price Interaction
    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['CMF'] = adl.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfr = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['MFI_14'] = 100 - (100 / (1 + mfr))
    
    # Hybrid Factors
    df['Hybrid_Momentum_Volume'] = df['ROC_12'] + df['CMF']
    df['Hybrid_Reversal_Volume'] = df['RSI_14'] + df['OBV']
    
    df['Volume_Spike'] = df['volume'] > df['volume'].rolling(window=30).mean() * 2
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    df['Volume_Price_Correlation'] = df['volume'].rolling(window=30).corr(df['close'])
    df['SMA_Crossover_VPC'] = df['SMA_Crossover'] * df['Volume_Price_Correlation']
    
    # Advanced Volume-Based Indicators
    df['Volume_Spike_Detection'] = df['volume'] > df['volume'].rolling(window=60).mean() * 2
    df['Volume_Weighted_Close'] = (df['volume'] * df['close']).cumsum() / df['volume'].cumsum()
    df['VROC_14'] = df['volume'].pct_change(periods=14)
