```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term EMA
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    
    # Long-Term EMA
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Momentum Difference
    momentum_diff = long_ema - short_ema
    
    # Daily Price Range
    daily_range = df['high'] - df['low']
    
    # Average Price
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Volume Adjusted Range
    vol_adj_range = (daily_range / avg_price.replace(0, 1e-9)) * df['volume']
    
    # Short-Term Rolling Average
    short_roll_avg = vol_adj_range.rolling(window=14).sum() / 14
    
    # Long-Term Rolling Average
    long_roll_avg = vol_adj_range.rolling(window=28).sum() / 28
    
    # Combined Momentum and Volume Adjusted Range
    combined_momentum_vol = momentum_diff * vol_adj_range
    
    # Adjust with Rolling Averages
    rolling_diff = short_roll_avg - long_roll_avg
    final_factor = combined_momentum_vol * rolling_diff
    
    return final_factor
```
