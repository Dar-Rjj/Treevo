def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 20-day Moving Average of Daily Price Changes
    df['price_momentum'] = df['price_change'].rolling(window=20).mean()
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Range
    df['volume_adjusted_range'] = (df['daily_price_range'] / ((df['high'] + df['low']) / 2 + 1e-9)) * (df['volume'] ** 0.5)
    
    # Short-Term Rolling Average of Volume Adjusted Range
    df['short_term_vrange_avg'] = df['volume_adjusted_range'].rolling(window=10).mean()
    
    # Long-Term Rolling Average of Volume Adjusted Range
    df['long_term_vrange_avg'] = df['volume_adjusted_range'].rolling(window=30).mean()
    
    # Combine Price Momentum and Volume Adjusted Range
    df['short_term_indicator'] = df['price_momentum'] * df['short_term_vrange_avg']
    df['long_term_indicator'] = df['price_momentum'] / (df['long_term_vrange_avg'] + 1e-9)
    df['composite_factor'] = (df['short_term_indicator'] + df['long_term_indicator']) / 2
    
    # Final Alpha Factor
    alpha_factor = df['composite_factor']
    
    return alpha_factor
