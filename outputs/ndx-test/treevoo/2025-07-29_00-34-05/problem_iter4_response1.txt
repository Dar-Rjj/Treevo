```python
def heuristics_v2(data):
    # Calculate Intraday High-Low Spread
    high_low_spread = data['high'] - data['low']
    
    # Calculate Close-Open Difference
    close_open_diff = data['close'] - data['open']
    
    # Combine Intraday Measures
    combined_intraday = high_low_spread + close_open_diff
    
    # Weight by Volume-to-Price Ratio
    volume_price_ratio = data['volume'] / data['close']
    weighted_factor = combined_intraday * volume_price_ratio
    
    return weighted_factor
```
