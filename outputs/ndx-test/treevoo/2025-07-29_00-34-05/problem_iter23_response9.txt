```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Calculate Short-Term and Long-Term Rolling Averages of Volume Adjusted Range
    short_term_avg = volume_adjusted_range.rolling(window=10).mean()
    long_term_avg = volume_adjusted_range.rolling(window=20).mean()
    
    # Combine Intraday Measures
    intraday_measures = (df['high'] - df['low'] + df['close'] - df['open']) * df['volume']
    
    # Calculate Short-Term and Long-Term Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum_difference = short_term_ema - long_term_ema
    
    # Assess Volatility Adjustment
    daily_ranges = df['high'] - df['low']
    avg_volatility = daily_ranges.rolling(window=30).mean()
    
    # Incorporate Volume Influence
    positive_volume_days = (df['volume'] > df['volume'].shift(1)).rolling(window=20).sum()
    negative_volume_days = (df['volume'] < df['volume'].shift(1)).rolling(window=20).sum()
    avg_30_day_volume = df['volume'].rolling(window=30).mean()
    volume_deviation_score = df['volume'] - avg_30_day_volume
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine Momentum and Volume Indicators
    adjusted_momentum = momentum_difference * (1 / avg_volatility)
    final_momentum = (short_term_ema * positive_volume_days + long_term_ema * negative_volume_days) * adjusted_momentum * volume_acceleration
    
    # Final Factor Adjustment
    factor_adjustment = (short_term_ema * volume_acceleration) / intraday_measures
    integrated_factor = (final_momentum + short_term_avg - long_term_avg + volume_deviation_score) * factor_adjustment
    
    # Open to Close Price Change
    open_to_close_change = (df['close'] - df['open']) / df['open']
    
    # Final Alpha Factor
    alpha_factor = integrated_factor + open_to_close_change
    
    return alpha_factor
```
