```python
def heuristics_v2(data):
    # Calculate Daily Price Change
    data['daily_price_change'] = data['close'].diff()
    
    # Calculate 20-day and 50-day Moving Averages of Daily Price Changes
    data['20_day_ma'] = data['daily_price_change'].rolling(window=20).mean()
    data['50_day_ma'] = data['daily_price_change'].rolling(window=50).mean()
    
    # Calculate Volume Change
    data['volume_change'] = data['volume'].diff()
    
    # Count Positive and Negative Volume Days over the Last 20 Days
    data['positive_volume_days'] = (data['volume_change'] > 0).rolling(window=20).sum()
    data['negative_volume_days'] = (data['volume_change'] < 0).rolling(window=20).sum()
    
    # Combine Price Momentum and Volume Indicator
    data['long_term_factor'] = data['50_day_ma'] * data['positive_volume_days']
    data['short_term_factor'] = data['20_day_ma'] * data['negative_volume_days']
    
    # Final Alpha Factor
    data['alpha_factor'] = data['long_term_factor'] - data['short_term_factor']
    
    return data['alpha_factor']
```
