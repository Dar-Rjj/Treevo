def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price).replace([np.inf, -np.inf], np.nan) * df['volume']
    
    # Calculate Short-Term Rolling Average
    short_term_rolling_avg = volume_adjusted_range.rolling(window=14).sum() / 14
    
    # Calculate Long-Term Rolling Average
    long_term_rolling_avg = volume_adjusted_range.rolling(window=28).sum() / 28
    
    # Generate the factor value by subtracting the long-term rolling average from the short-term rolling average
    factor_value = short_term_rolling_avg - long_term_rolling_avg
    
    return factor_value
