```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day and 10-day Exponential Moving Averages (EMA)
    df['5_day_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['5_day_ema'] - df['10_day_ema']
    
    # Calculate 20-day EMA for Long-Term Momentum
    df['20_day_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum Difference
    df['momentum_difference'] = df['5_day_ema'] - df['20_day_ema']
    
    # Calculate 30-Day Average Range
    df['daily_range'] = df['high'] - df['low']
    df['30_day_avg_range'] = df['daily_range'].rolling(window=30).mean()
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = df['volume'].pct_change(periods=5)
    
    # Calculate Volume Deviation
    df['30_day_avg_volume'] = df['volume'].rolling(window=30).mean()
    df['volume_deviation'] = df['volume'] - df['30_day_avg_volume']
    
    # Introduce High-Low Range Factor
    df['high_low_range_factor'] = (df['high'] - df['low']) / df['close']
    
    # Combine Momentum, Volume, and High-Low Range
    df['combined_factor'] = df['short_term_momentum'] * df['volume_acceleration'] * df['high_low_range_factor'] / df['close']
    
    # Adjust Momentum
    df['adjusted_momentum'] = df['momentum_difference'] * (1 / df['30_day_avg_range'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_momentum'] + df['volume_deviation']
    
    return df['alpha_factor']
```
