```python
def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Momentum
    df['momentum'] = df['close'].pct_change(30)
    
    # Calculate 30-Day Average Range
    df['daily_range'] = df['high'] - df['low']
    df['avg_30_day_range'] = df['daily_range'].rolling(window=30).mean()
    
    # Calculate Volume Deviation
    df['avg_30_day_volume'] = df['volume'].rolling(window=30).mean()
    df['volume_deviation'] = df['volume'] - df['avg_30_day_volume']
    
    # Adjust Momentum
    df['adjusted_momentum'] = df['momentum'] / df['avg_30_day_range']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_momentum'] + df['volume_deviation']
    
    return df['alpha_factor'].dropna()
```
