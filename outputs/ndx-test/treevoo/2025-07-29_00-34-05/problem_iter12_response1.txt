```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum Factor
    df['Momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Assess Volatility Adjustment using High-Low Range
    df['DailyRange'] = df['high'] - df['low']
    df['AverageRangeVolatility'] = df['DailyRange'].rolling(window=30).mean()
    
    # Incorporate High-Low Range Momentum
    alpha = 2 / (5 + 1)
    df['EMA_HighLow'] = df['DailyRange'].ewm(alpha=alpha, adjust=False).mean()
    df['EMA_Change'] = df['EMA_HighLow'] - df['EMA_HighLow'].shift(1)
    
    # Calculate Price Momentum
    df['DailyPriceChange'] = df['close'] - df['close'].shift(1)
    df['MA20_Price'] = df['DailyPriceChange'].rolling(window=20).mean()
    df['MA50_Price'] = df['DailyPriceChange'].rolling(window=50).mean()
    
    # Calculate Volume Indicator
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['PositiveVolumeDays'] = (df['VolumeChange'] > 0).rolling(window=20).sum()
    df['NegativeVolumeDays'] = (df['VolumeChange'] < 0).rolling(window=20).sum()
    df['SignificantVolumeChange'] = (df['VolumeChange'] / df['volume'].shift(1)).abs() > 0.5
    df['VolumeShockSign'] = np.sign(df['VolumeChange'])
    
    # Combine Factors
    df['AdjustedMomentum'] = df['Momentum'] / df['AverageRangeVolatility']
    df['HighLowRangeMomentum'] = df['AdjustedMomentum'] * df['EMA_Change']
    df['LongTermFactor'] = df['MA50_Price'] * df['PositiveVolumeDays']
    df['ShortTermFactor'] = df['MA20_Price'] * df['NegativeVolumeDays']
    
    # Final Alpha Factor
    long_term_weight = 0.6
    short_term_weight = 0.4
    df['WeightedSum'] = (long_term_weight * df['LongTermFactor']) + (short_term_weight * df['ShortTermFactor'])
    df['FinalAlphaFactor'] = df['WeightedSum'] * df['EMA_Change'] * df['VolumeShockSign']
    
    return df['FinalAlphaFactor']
```
