```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_period = 5
    short_ema = df['close'].ewm(span=short_term_period, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_period = 20
    long_ema = df['close'].ewm(span=long_term_period, adjust=False).mean()
    
    # Compute Momentum Difference
    momentum_difference = long_ema - short_ema
    
    # Calculate Volume Acceleration
    volume_change_period = 5
    volume_acceleration = (df['volume'] / df['volume'].shift(volume_change_period) - 1)
    
    # Combine Momentum Difference with Volume Acceleration
    factor = momentum_difference * volume_acceleration
    
    return factor
```
