import pandas as pd

def heuristics_v2(df):
    # Calculate Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    df['SMA_5_minus_EMA_5'] = df['SMA_5'] - df['EMA_5']
    df['SMA_10_minus_EMA_10'] = df['SMA_10'] - df['EMA_10']
    df['SMA_20_minus_EMA_20'] = df['SMA_20'] - df['EMA_20']
    
    # Identify Crossovers
    df['SMA_5_cross_SMA_10'] = (df['SMA_5'] > df['SMA_10']) & (df['SMA_5'].shift(1) <= df['SMA_10'].shift(1))
    df['SMA_5_cross_below_SMA_10'] = (df['SMA_5'] < df['SMA_10']) & (df['SMA_5'].shift(1) >= df['SMA_10'].shift(1))
    
    df['EMA_5_cross_EMA_10'] = (df['EMA_5'] > df['EMA_10']) & (df['EMA_5'].shift(1) <= df['EMA_10'].shift(1))
    df['EMA_5_cross_below_EMA_10'] = (df['EMA_5'] < df['EMA_10']) & (df['EMA_5'].shift(1) >= df['EMA_10'].shift(1))
    
    # Calculate Price Derivatives
    df['price_first_derivative'] = df['close'].diff()
    df['price_second_derivative'] = df['price_first_derivative'].diff()
    
    # Explore Volume and Amount Relationships
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    df['SMA_volume_ratio_5'] = df['volume_to_amount_ratio'].rolling(window=5).mean()
    df['SMA_volume_ratio_10'] = df['volume_to_amount_ratio'].rolling(window=10).mean()
    df['SMA_volume_ratio_20'] = df['volume_to_amount_ratio'].rolling(window=20).mean()
    
    df['volume_spike'] = (df['volume'] > df['volume'].rolling(window=10).mean() * 2)
    
    # Leverage Intraday Price Information
    df['daily_range'] = df['high'] - df['low']
    df['SMA_daily_range_5'] = df['daily_range'].rolling(window=5).mean()
    df['SMA_daily_range_10'] = df['daily_range'].rolling(window=10).mean()
    df['SMA_daily_range_20'] = df['daily_range'].rolling(window=20).mean()
    
    df['gap_opening'] = df['open'] - df['close'].shift(1)
    df['significant_gap'] = (df['gap_opening'].abs() > 0.01 * df['close'].shift(1))
    
    df['intraday_net_change'] = df['close'] - df['open']
    df['SMA_intraday_net_change_5'] = df['intraday_net_change'].rolling(window=5).mean()
    df['SMA_intraday_net_change_10'] = df['intraday_net_change'].rolling(window=10).mean()
    df['SMA_intraday_net_change_20'] = df['intraday_net_change'].rolling(window=20).mean()
    
    # Utilize Technical Indicators
    def rsi(series, period=14):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        avg_gain = gain.rolling(window=period).mean()
        avg_loss = loss.rolling(window=period).mean()
        rs = avg_gain / avg_loss
