```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    short_ema = df['close'].ewm(span=7, adjust=False).mean()
    long_ema = df['close'].ewm(span=25, adjust=False).mean()
    momentum_diff = long_ema - short_ema  # Subtract Short-Term EMA from Long-Term EMA

    # Calculate True Range
    true_range1 = df['high'] - df['low']
    true_range2 = (df['high'] - df['close'].shift(1)).abs()
    true_range3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([true_range1, true_range2, true_range3], axis=1).max(axis=1)

    # Compute Volume Adjusted True Range
    close = df['close']
    volume_adjusted_true_range = (true_range / close.replace(0, 1)) * df['volume']

    # Calculate Short-Term and Long-Term Rolling Sums of Volume Adjusted True Range
    short_sum = volume_adjusted_true_range.rolling(window=14).sum()
    long_sum = volume_adjusted_true_range.rolling(window=28).sum()

    # Combine Momentum and Volume Adjusted True Range
    combined_momentum = momentum_diff * volume_adjusted_true_range

    # Adjust with Rolling Sums
    rolling_sum_difference = short_sum - long_sum
    final_factor = combined_momentum * rolling_sum_difference

    return final_factor
```
