```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 5-day Exponential Moving Average (EMA) of High-Low Range
    N = 5
    alpha = 2 / (N + 1)
    high_low_ema = high_low_range.ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Daily Change in EMA
    daily_change_ema = high_low_ema.diff()
    
    # Calculate Volume Percentage Change
    volume_change = df['volume'].pct_change()
    
    # Identify Significant Volume Changes
    threshold = 0.5
    volume_shock = (volume_change > threshold).astype(int) * 2 - 1  # Converts to 1 or -1
    
    # Combine Indicators
    combined_indicator = daily_change_ema * volume_shock
    
    return combined_indicator
```
