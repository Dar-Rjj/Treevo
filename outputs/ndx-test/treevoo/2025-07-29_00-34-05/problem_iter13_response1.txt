```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Intraday Spread: High - Low
    intraday_spread = data['high'] - data['low']
    
    # Combined Intraday: (Close - Open) + Intraday Spread
    combined_intraday = (data['close'] - data['open']) + intraday_spread
    
    # Adjust by Volume
    volume_adjusted_intraday = combined_intraday * data['volume']
    
    # Short-Term Price Momentum: 5-day EMA of Close - 10-day EMA of Close
    ema_5_close = data['close'].ewm(span=5, adjust=False).mean()
    ema_10_close = data['close'].ewm(span=10, adjust=False).mean()
    short_term_price_momentum = ema_5_close - ema_10_close
    
    # Volume Momentum: 5-day EMA of Volume - 10-day EMA of Volume
    ema_5_volume = data['volume'].ewm(span=5, adjust=False).mean()
    ema_10_volume = data['volume'].ewm(span=10, adjust=False).mean()
    volume_momentum = ema_5_volume - ema_10_volume
    
    # Final Factor: (Short-Term Price Momentum * Volume Momentum) / Volume-Adjusted Intraday
    final_factor = (short_term_price_momentum * volume_momentum) / volume_adjusted_intraday

    return final_factor
```
