```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range and Daily Range
    df['TrueRange'] = df[['High', 'Low']].apply(lambda x: max(x['High'], x['Close'].shift(1)) - min(x['Low'], x['Close'].shift(1)), axis=1)
    df['DailyRange'] = df['High'] - df['Low']

    # Compute Average True Range (ATR) and Aggregate 30-day Daily Ranges
    n = 30
    df['ATR'] = df['TrueRange'].rolling(window=n).mean()
    df['AvgDailyRange'] = df['DailyRange'].rolling(window=n).mean()

    # Determine Momentum
    df['PriceChange'] = df['Close'] - df['Close'].shift(n)
    df['MomentumRatio'] = df['PriceChange'] / df['ATR']

    # Adjust for Volume Impact
    m = 10
    df['VolumeSmoothing'] = df['Volume'].rolling(window=m).mean() / df['Volume'].rolling(window=m).std()
    df['MomentumAdjByVolume'] = df['MomentumRatio'] * df['VolumeSmoothing']

    # Incorporate Volume Influence
    df['Avg30DayVolume'] = df['Volume'].rolling(window=n).mean()
    df['VolumeDeviationScore'] = df['Volume'] - df['Avg30DayVolume']

    # Combine Factors
    df['VolatilityScore'] = 1 / df['AvgDailyRange']
    df['FinalAlphaFactor'] = df['MomentumAdjByVolume'] * df['VolatilityScore'] + df['VolumeDeviationScore']

    return df['FinalAlphaFactor']
```
