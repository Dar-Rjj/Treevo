import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['High'] - df['Low']
    
    # Calculate Average Price
    df['Average_Price'] = (df['High'] + df['Low'] + df['Close']) / 3
    
    # Compute Volume Adjusted Range
    df['Volume_Adjusted_Range'] = (df['Daily_Price_Range'] / df['Average_Price'].replace(0, np.nan)) * df['Volume']
    
    # Calculate Short-Term and Long-Term Rolling Averages of Volume Adjusted Range
    df['Short_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=10).mean()
    df['Long_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=20).mean()
    
    # Combine Intraday Measures
    df['Intraday_Measure'] = (df['Close'] - df['Open']) * (df['High'] - df['Low']) * df['Volume']
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['Short_Term_EMA'] = df['Close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_EMA'] = df['Close'].ewm(span=20, adjust=False).mean()
    df['Momentum_Difference'] = df['Short_Term_EMA'] - df['Long_Term_EMA']
    
    # Assess Volatility Adjustment
    df['Daily_Range'] = df['High'] - df['Low']
    df['Average_Volatility'] = df['Daily_Range'].rolling(window=30).mean()
    
    # Incorporate Volume Influence
    df['Volume_Acceleration'] = df['Volume'].pct_change(periods=5)
    df['Average_Volume'] = df['Volume'].rolling(window=30).mean()
    
    # Final Factor Adjustment
    df['Adjusted_Momentum'] = df['Momentum_Difference'] / df['Average_Volatility']
    df['Final_Factor'] = (df['Adjusted_Momentum'] * df['Volume_Acceleration']) / df['Intraday_Measure']
    
    # Calculate 50-Day and 200-Day Moving Averages
    df['50_MA'] = df['Close'].rolling(window=50).mean()
    df['200_MA'] = df['Close'].rolling(window=200).mean()
    df['Crossover_Signal'] = (df['50_MA'] > df['200_MA']).astype(int)
    
    return df['Final_Factor']
