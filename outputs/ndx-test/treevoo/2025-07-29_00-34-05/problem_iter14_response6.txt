```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    open_close_diff = abs(df['close'] - df['open'])
    enhanced_intraday_range = (intraday_range + open_close_diff) / 2
    
    # Adjust by Short-Term Momentum and Volume Trend
    short_term_price_momentum = df['high'].ewm(span=5).mean() - df['low'].ewm(span=10).mean()
    volume_momentum = df['volume'].rolling(window=5).sum() - df['volume'].rolling(window=10).sum()
    final_factor = (short_term_price_momentum * volume_momentum) / (enhanced_intraday_range * df['amount'])
    
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(20)
    
    # Volume Impact Factor
    volume_impact_factor = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price).replace([np.inf, -np.inf], np.nan) * df['volume']
    
    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    short_term_volume_adjusted_avg = volume_adjusted_range.rolling(window=15).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    long_term_volume_adjusted_avg = volume_adjusted_range.rolling(window=25).mean()
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=30).std()
    
    # Trade Amount Influence
    trade_amount_influence = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    combined_momentum_vol = momentum / volume_impact_factor
    
    # Adjust for Price Volatility
    adjusted_momentum_vol = combined_momentum_vol / price_volatility
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    alpha = 2 / (7 + 1)
    high_low_ema = high_low_range.ewm(alpha=alpha, adjust=False).mean()
    daily_high_low_ema_change = high_low_ema - high_low_ema.shift(1)
    
    # Calculate Volume Shock
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    significant_volume_change = (volume_percentage_change > 0.60).astype(int)
    
    # Combine Indicators
    combined_indicator = daily_high_low_ema_change * significant_volume_change
    
    # Finalize Alpha Factor
    alpha_factor = (short_term_volume_adjusted_avg 
                    - long_term_volume_adjusted_avg 
                    + combined_indicator 
                    - adjusted_momentum_vol 
                    * trade_amount_influence)
    
    return alpha_factor
```
