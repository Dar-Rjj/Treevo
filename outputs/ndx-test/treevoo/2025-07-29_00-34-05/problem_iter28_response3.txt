```python
import pandas as pd

def heuristics_v2(df):
    # Calculate price differences
    df['open_diff'] = df['open'].diff()
    df['close_diff'] = df['close'].diff()
    df['intra_day_move'] = df['high'] - df['low']
    
    # Detect trend reversals
    df['bullish_reversal'] = (df['low'].shift(1) > df['low']) & (df['close'] > df['open'])
    df['bearish_reversal'] = (df['high'].shift(1) < df['high']) & (df['close'] < df['open'])
    
    # Calculate volume differences
    df['volume_diff'] = df['volume'].diff()
    
    # Identify volume spikes relative to price
    df['vol_spike_min_price_change'] = ((df['close_diff'].abs() < 0.01) & (df['volume'] > df['volume'].expanding().mean()))
    df['vol_low_price_spike'] = ((df['close_diff'].abs() > 0.1) & (df['volume'] < df['volume'].expanding().mean()))
    
    # On-Balance Volume (OBV)
    df['obv'] = 0
    df.loc[df['close_diff'] > 0, 'obv'] = df['volume']
    df.loc[df['close_diff'] < 0, 'obv'] = -df['volume']
    df['obv'] = df['obv'].cumsum()
    
    # Money Flow Index (MFI)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df['money_flow'].where(df['typical_price'] > df['typical_price'].shift(1), 0)
    df['negative_money_flow'] = df['money_flow'].where(df['typical_price'] < df['typical_price'].shift(1), 0)
    df['positive_money_flow'] = df['positive_money_flow'].rolling(window=14).sum()
    df['negative_money_flow'] = df['negative_money_flow'].rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + (df['positive_money_flow'] / df['negative_money_flow'])))
    
    # Evaluate trend strength
    df['strong_upward_trend'] = (df['close'] > df['open']) & (df['obv'] > df['obv'].shift(1))
    df['strong_downward_trend'] = (df['close'] < df['open']) & (df['obv'] < df['obv'].shift(1))
    
    # Assess market sentiment
    df['bullish_sentiment'] = df['mfi'] > 80
    df['bearish_sentiment'] = df['mfi'] < 20
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = (
        df['open_diff'] + 
        df['close_diff'] + 
        df['intra_day_move'] + 
        df['volume_diff'] + 
        100 * df['obv'].pct_change() + 
        100 * (df['mfi'] - 50)
    )
    
    return df['alpha_factor']
```
