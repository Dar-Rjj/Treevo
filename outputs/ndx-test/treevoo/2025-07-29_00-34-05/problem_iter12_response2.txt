```python
import pandas as pd

def heuristics_v2(data):
    # Daily Price Change
    data['price_change'] = data['close'] - data['close'].shift(1)
    
    # Short-Term Price Momentum: 5-day EMA and 10-day EMA
    data['5_ema_price_change'] = data['price_change'].ewm(span=5, adjust=False).mean()
    data['10_ema_price_change'] = data['price_change'].ewm(span=10, adjust=False).mean()
    data['short_term_momentum'] = data['5_ema_price_change'] - data['10_ema_price_change']
    
    # Long-Term Price Momentum: 20-day SMA and 50-day SMA
    data['20_sma_price_change'] = data['price_change'].rolling(window=20).mean()
    data['50_sma_price_change'] = data['price_change'].rolling(window=50).mean()
    data['long_term_momentum'] = data['20_sma_price_change'] - data['50_sma_price_change']
    
    # Volume Change
    data['volume_change'] = data['volume'] - data['volume'].shift(1)
    
    # Positive and Negative Volume Days
    data['positive_volume_days'] = (data['volume_change'] > 0).rolling(window=20).sum()
    data['negative_volume_days'] = (data['volume_change'] < 0).rolling(window=20).sum()
    
    # Volume Momentum: 5-day EMA and 10-day EMA
    data['5_ema_volume'] = data['volume'].ewm(span=5, adjust=False).mean()
    data['10_ema_volume'] = data['volume'].ewm(span=10, adjust=False).mean()
    data['volume_momentum'] = data['5_ema_volume'] - data['10_ema_volume']
    
    # Adjusted Short-Term Price Momentum by Volume Indicator
    data['adjusted_short_momentum'] = (data['short_term_momentum'] * data['positive_volume_days']) + \
                                      (data['short_term_momentum'] * data['volume_momentum'])
    
    # Adjusted Long-Term Price Momentum by Volume Indicator
    data['adjusted_long_momentum'] = (data['long_term_momentum'] * data['negative_volume_days']) + \
                                     (data['long_term_momentum'] * data['volume_momentum'])
    
    # Final Alpha Factor
    data['alpha_factor'] = (data['adjusted_short_momentum'] + data['adjusted_long_momentum']) / data['adjusted_short_momentum']
    
    return data['alpha_factor']
```
