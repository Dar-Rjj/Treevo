import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-Based Factors
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['50_day_SMA'] - df['200_day_SMA']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Long_Term_Momentum'] = df['5_day_EMA'] - df['20_day_EMA']
    
    # Reversal-Based Factors
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    avg_gain = up.rolling(window=14).mean()
    avg_loss = abs(down.rolling(window=14).mean())
    df['RSI'] = 100 - (100 / (1 + (avg_gain / avg_loss)))
    
    df['Close_to_Open_Ratio'] = df['close'] / df.shift(1)['open']
    df['CO_Ratio_Avg'] = df['Close_to_Open_Ratio'].rolling(window=30).mean()
    df['Extreme_Price_Movements'] = df['Close_to_Open_Ratio'] - df['CO_Ratio_Avg']
    
    # Volume-Price Interaction
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    df['OBV'] = obv
    
    money_flow_volume = df['volume'] * ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['CMF'] = cmf
    
    df['Vol_5_day_MA'] = df['volume'].rolling(window=5).mean()
    df['Vol_10_day_MA'] = df['volume'].rolling(window=10).mean()
    df['Volume_Deviation_Score'] = df['Vol_5_day_MA'] - df['Vol_10_day_MA']
    df['Vol_Percentage_Change_5_days'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Volume and Amount-Based Factors
    df['Volume_Spike'] = (df['volume'] - df['volume'].rolling(window=30).mean()) / df['volume'].rolling(window=30).std()
    
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    
    price_change = df['close'].diff()
    df['Vol_Price_Correlation'] = price_change.rolling(window=30).corr(df['volume'])
    
    # Hybrid Factors
    df['Momentum_Volume_Hybrid'] = df['ROC_12'] * df['CMF']
    df['Reversal_Volume_Hybrid'] = df['RSI'] * df['OBV']
    
    # Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    open_close_diff = df['close'] - df['open']
    df['Enhanced_Intraday_Range'] = (intraday_range + open_close_diff) / 2
    
    # Volatility Adjustment
    daily_range = df['high'] - df['low']
    df['30_day_Agg_Range'] = daily_range.rolling(window=30).sum()
    df['Avg_Range_Volatility'] = df['30_day_Agg_Range'] / 30
    
    # New Factor: Price-Volatility Ratio
    df['Price_Volatility_Ratio'] = df['close'] / df['Avg_Range_Volatility']
