```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Enhanced Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine Intraday Measures
    combined_intraday_measure = (high_low_spread + close_open_diff) * df['volume']
    
    # Calculate Short-Term Price Momentum
    ema_5_price = df['close'].ewm(span=5, adjust=False).mean()
    ema_10_price = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_5_price - ema_10_price
    
    # Calculate Volume Momentum
    ema_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    ema_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    volume_momentum = ema_5_volume - ema_10_volume
    
    # Final Factor Adjustment
    final_factor = (short_term_momentum * volume_momentum) / combined_intraday_measure
    
    return final_factor
```
