```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum Factor
    momentum = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Assess Volatility Adjustment using High-Low Range
    daily_range = df['high'] - df['low']
    range_volatility = daily_range.rolling(window=30).sum() / 30
    
    # Incorporate High-Low Range Momentum
    alpha = 2 / (5 + 1)
    ema_high_low = daily_range.ewm(alpha=alpha, adjust=False).mean()
    daily_change_ema = ema_high_low - ema_high_low.shift(1)
    
    # Calculate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    ma_20 = price_change.rolling(window=20).mean()
    ma_50 = price_change.rolling(window=50).mean()
    
    # Calculate Volume Indicator
    volume_change = df['volume'] - df['volume'].shift(1)
    positive_volume_days = (volume_change > 0).rolling(window=20).sum()
    negative_volume_days = (volume_change < 0).rolling(window=20).sum()
    significant_volume_change = (abs(volume_change) > 0.5 * df['volume'].shift(1)).astype(int)
    
    # Combine Factors
    adjusted_momentum = momentum / range_volatility
    adjusted_momentum = adjusted_momentum * daily_change_ema
    
    long_term_factor = ma_50 * negative_volume_days
    short_term_factor = ma_20 * positive_volume_days
    
    final_alpha = (long_term_factor + short_term_factor) * np.sign(significant_volume_change)
    
    return final_alpha
```
