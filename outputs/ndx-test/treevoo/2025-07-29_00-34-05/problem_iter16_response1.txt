```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Factor
    df['volume_impact'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Price Range
    df['daily_range'] = df['high'] - df['low']
    
    # Average Price
    df['average_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Volume Adjusted Range
    df['volume_adjusted_range'] = (df['daily_range'] / df['average_price']).fillna(0) * df['volume']
    
    # Short-Term Rolling Avg of Volume Adjusted Range
    df['short_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=15).mean()
    
    # Long-Term Rolling Avg of Volume Adjusted Range
    df['long_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=25).mean()
    
    # Price Volatility
    df['price_volatility'] = df['close'].rolling(window=30).std()
    
    # Trade Amount Influence
    df['amount_influence'] = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    df['combined_momentum_volume'] = df['momentum'] / df['volume_impact']
    
    # Adjust for Price Volatility
    df['adjusted_momentum_volatility'] = df['combined_momentum_volume'] / df['price_volatility']
    
    # High-Low Range Momentum
    alpha = 2 / (7 + 1)
    df['ema_high_low'] = df['daily_range'].ewm(alpha=alpha, adjust=False).mean()
    df['daily_change_ema'] = df['ema_high_low'] - df['ema_high_low'].shift(1)
    
    # Volume Shock
    df['volume_percentage_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_shock'] = df['volume_percentage_change'].apply(lambda x: 1 if x > 0.6 else -1 if x < -0.6 else 0)
    
    # Combine Indicators
    df['combined_indicator'] = df['daily_change_ema'] * df['volume_shock']
    
    # Finalize Alpha Factor
    df['alpha_factor'] = (df['short_term_vol_adj_range'] / df['long_term_vol_adj_range']) * df['combined_indicator'] * df['adjusted_momentum_volatility'] * df['amount_influence']
    
    return df['alpha_factor']
```
