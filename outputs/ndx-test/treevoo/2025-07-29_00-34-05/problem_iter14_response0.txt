```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Factor
    df['Momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)

    # Assess Volatility Adjustment using High-Low Range
    df['DailyRange'] = df['high'] - df['low']
    df['VolatilityScore'] = df['DailyRange'].rolling(window=30).mean()
    
    # Incorporate High-Low Range Momentum
    alpha = 2 / (5 + 1)
    df['HL_Range_EMA'] = df['DailyRange'].ewm(alpha=alpha, adjust=False).mean()
    df['EMA_Change'] = df['HL_Range_EMA'] - df['HL_Range_EMA'].shift(1)
    
    # Calculate Price Momentum
    df['DailyPriceChange'] = df['close'] - df['close'].shift(1)
    df['20DMA'] = df['DailyPriceChange'].rolling(window=20).mean()
    df['50DMA'] = df['DailyPriceChange'].rolling(window=50).mean()
    
    # Calculate Volume Indicator
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['PositiveVolumeDays'] = (df['VolumeChange'] > 0).rolling(window=20).sum()
    df['NegativeVolumeDays'] = (df['VolumeChange'] < 0).rolling(window=20).sum()
    significant_volume_change_threshold = 0.5
    df['VolumeShock'] = (df['VolumeChange'].abs() > df['volume'].shift(1) * significant_volume_change_threshold).astype(int) * df['VolumeChange'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Combine Factors
    df['AdjustedMomentum'] = df['Momentum'] / df['VolatilityScore']
    df['HL_Range_Momentum_Adjusted'] = df['AdjustedMomentum'] * df['EMA_Change']
    df['LongTermFactor'] = df['50DMA'] * df['PositiveVolumeDays']
    df['ShortTermFactor'] = df['20DMA'] * df['NegativeVolumeDays']
    
    # Final Alpha Factor
    long_term_weight = 0.6
    short_term_weight = 0.4
    df['CombinedFactor'] = (long_term_weight * df['LongTermFactor']) + (short_term_weight * df['ShortTermFactor'])
    df['FinalAlphaFactor'] = df['CombinedFactor'] * df['EMA_Change'] * df['VolumeShock']
    
    return df['FinalAlphaFactor']
```
