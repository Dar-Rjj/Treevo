```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    gap = df['close'] - df['open']
    enhanced_intraday_range = (intraday_range + gap) / 2
    
    # Short-Term Price Momentum
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = ema_5 - ema_10
    
    # Volume Momentum
    vol_5 = df['volume'].rolling(window=5).mean()
    vol_10 = df['volume'].rolling(window=10).mean()
    volume_momentum = vol_5 - vol_10
    
    # Final Factor
    factor = (short_term_momentum * volume_momentum) / (enhanced_intraday_range * df['close'])
    
    return factor
```
