import pandas as pd

def heuristics_v2(df):
    # Momentum-Based Factors
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['50_day_SMA'] - df['200_day_SMA']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Long_Term_Momentum'] = df['5_day_EMA'] - df['20_day_EMA']
    
    # Reversal-Based Factors
    delta = df['close'].diff()
    up, down = delta.clip(lower=0), delta.clip(upper=0).abs()
    rs_up = up.rolling(window=14).mean()
    rs_down = down.rolling(window=14).mean()
    df['RSI'] = 100 - 100 / (1 + rs_up / rs_down)
    
    df['Close_to_Open_Ratio'] = df['close'] / df['open']
    df['Avg_Close_to_Open_Ratio'] = df['Close_to_Open_Ratio'].rolling(window=30).mean()
    df['Extreme_Price_Movement'] = (df['Close_to_Open_Ratio'] - df['Avg_Close_to_Open_Ratio']).abs()
    
    # Volume-Price Interaction
    df['OBV'] = ((df['close'] > df['close'].shift()).astype(int) * df['volume'] - 
                 (df['close'] < df['close'].shift()).astype(int) * df['volume']).cumsum()
    
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    df['Volume_5_day_MA'] = df['volume'].rolling(window=5).mean()
    df['Volume_10_day_MA'] = df['volume'].rolling(window=10).mean()
    df['Volume_Deviation_Score'] = df['Volume_5_day_MA'] - df['Volume_10_day_MA']
    df['Volume_Percentage_Change_5_days'] = df['volume'].pct_change(periods=5)
    
    # Volume and Amount-Based Factors
    df['Avg_Volume_30_days'] = df['volume'].rolling(window=30).mean()
    df['Volume_Breakout'] = (df['volume'] / df['Avg_Volume_30_days']) > 2
    
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    df['Volume_Price_Correlation'] = df['close'].pct_change().corr(df['volume'].pct_change())
    
    # Hybrid Factors
    df['Momentum_Volume_Hybrid'] = df['ROC_12'] * df['CMF_20']
    df['Reversal_Volume_Hybrid'] = df['RSI'] * df['OBV']
    
    # Enhanced Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    df['Enhanced_Intraday_Range'] = df['Intraday_Range'] / 2
    
    # Volatility Adjustment
    df['Daily_Range'] = df['high'] - df['low']
    df['Aggregate_30_Day_Ranges'] = df['Daily_Range'].rolling(window=30).sum()
    df['Average_Range_Volatility'] = df['Aggregate_30_Day_Ranges'] / 30
    df['Price_Volatility_Ratio'] = df['close'] / df['Average_Range_Volatility']
    
    # Final Alpha Factor
