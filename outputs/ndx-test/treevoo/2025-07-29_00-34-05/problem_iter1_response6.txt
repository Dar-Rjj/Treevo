import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price).replace([float('inf'), float('-inf')], 0) * df['volume']
    
    # Calculate Short-Term Rolling Average
    short_term_rolling_avg = volume_adjusted_range.rolling(window=10).sum() / 10
    
    # Calculate Long-Term Rolling Average
    long_term_rolling_avg = volume_adjusted_range.rolling(window=20).sum() / 20
    
    # Generate the factor values
    factor_values = short_term_rolling_avg - long_term_rolling_avg
    
    return factor_values
