```python
def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_range = df['high'] - df['low']
    
    # Compute Volume Adjusted Range
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    volume_adjusted_range = (daily_range / avg_price).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Calculate Short-Term Rolling Average
    short_term_avg = volume_adjusted_range.rolling(window=10).sum() / 10
    
    # Calculate Long-Term Rolling Average
    long_term_avg = volume_adjusted_range.rolling(window=20).sum() / 20
    
    # Compute the factor
    factor = short_term_avg - long_term_avg
    
    return factor
```
