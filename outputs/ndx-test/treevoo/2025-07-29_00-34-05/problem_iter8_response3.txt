```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Calculate Short-Term and Long-Term Rolling Averages of Volume Adjusted Range
    short_term_var_avg = volume_adjusted_range.rolling(window=10).mean()
    long_term_var_avg = volume_adjusted_range.rolling(window=20).mean()
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine Intraday Measures
    combined_intraday_measure = (intraday_high_low_spread + close_open_diff) * df['volume']
    
    # Calculate Short-Term Price Momentum
    price_momentum_5 = df['close'].ewm(span=5, adjust=False).mean()
    price_momentum_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_momentum = price_momentum_5 - price_momentum_10
    
    # Calculate Volume Momentum
    volume_momentum_5 = df['volume'].ewm(span=5, adjust=False).mean()
    volume_momentum_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_momentum = volume_momentum_5 - volume_momentum_10
    
    # Final Factor Adjustment
    final_factor_adjustment = (short_term_price_momentum * volume_momentum) / combined_intraday_measure.replace(0, np.nan)
    
    # Combine Factors
    combined_factors = (
        short_term_var_avg 
        - long_term_var_avg 
        + final_factor_adjustment
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors
    
    return final_alpha_factor
```
