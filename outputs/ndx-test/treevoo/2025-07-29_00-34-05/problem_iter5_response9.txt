```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(15)
    
    # Volume Impact Factor
    df['avg_volume_25'] = df['volume'].rolling(window=25).mean()
    df['volume_impact_factor'] = df['volume'] / df['avg_volume_25']
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['avg_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    df['volume_adjusted_range'] = (df['daily_price_range'] / df['avg_price']).fillna(0) * df['volume']
    
    # Short-Term Rolling Average of Volume Adjusted Range
    df['short_term_avg_var'] = df['volume_adjusted_range'].rolling(window=10).mean()
    
    # Long-Term Rolling Average of Volume Adjusted Range
    df['long_term_avg_var'] = df['volume_adjusted_range'].rolling(window=20).mean()
    
    # Price Volatility
    df['volatility'] = df['close'].rolling(window=25).std()
    
    # Trade Amount Influence
    df['avg_amount_25'] = df['amount'].rolling(window=25).mean()
    df['trade_amount_influence'] = df['amount'] / df['avg_amount_25']
    
    # Combine Momentum and Volume Impact
    df['combined_momentum_volume'] = df['momentum'] / df['volume_impact_factor']
    
    # Adjust for Price Volatility
    df['adjusted_momentum_volatility'] = df['combined_momentum_volume'] / df['volatility']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum using 5-day EMA
    df['ema_high_low'] = df['high_low_range'].ewm(span=5, adjust=False).mean()
    df['ema_change'] = df['ema_high_low'] - df['ema_high_low'].shift(1)
    
    # Calculate Volume Shock
    df['volume_change'] = df['volume'].pct_change()
    df['volume_shock'] = (df['volume_change'].abs() > 0.5).astype(int)
    
    # Combine Indicators
    df['combined_indicator'] = df['ema_change'] * df['volume_shock'].map({0: 1, 1: -1})
    
    # Finalize Alpha Factor
    df['alpha_factor'] = (df['short_term_avg_var'] - df['long_term_avg_var']) * df['combined_indicator'] * df['adjusted_momentum_volatility'] * df['trade_amount_influence']
    
    return df['alpha_factor']
```
