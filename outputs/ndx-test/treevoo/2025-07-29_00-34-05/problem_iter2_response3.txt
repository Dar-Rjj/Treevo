```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 10-day EMA of Close price
    df['EMA_5_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Subtract 10-day EMA from 5-day EMA
    df['short_term_momentum'] = df['EMA_5_close'] - df['EMA_10_close']
    
    # Calculate daily range (High - Low)
    df['daily_range'] = df['high'] - df['low']
    
    # Aggregate 30-day Daily Ranges
    df['volatility_score'] = df['daily_range'].rolling(window=30).sum() / 30
    
    # Calculate 5-day and 10-day EMA of Volume
    df['EMA_5_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['EMA_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Subtract 10-day EMA of Volume from 5-day EMA of Volume
    df['volume_deviation'] = df['EMA_5_volume'] - df['EMA_10_volume']
    
    # Adjust Momentum by Volatility
    df['momentum_adjusted_volatility'] = df['short_term_momentum'] * (1 / df['volatility_score'])
    
    # Adjust by Volume Deviation
    df['final_alpha_factor'] = df['momentum_adjusted_volatility'] + df['volume_deviation']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['final_alpha_factor'] / df['close']
    
    return df['final_alpha_factor']
```
