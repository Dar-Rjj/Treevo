```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the average true range over 14 days
    atr = (df['high'] - df['low']).rolling(window=14).mean()
    
    # Calculate the buying pressure
    buying_pressure = (df['close'] - df['open']) / daily_range
    
    # Calculate a simple moving average of the closing price over 50 days
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the percentage change in volume compared to the previous day
    volume_change = df['volume'].pct_change()
    
    # Generate the alpha factor by combining the different components
    alpha_factor = (buying_pressure * atr * volume_change) / sma_50
    
    return alpha_factor
```
