```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum_diff = long_term_ema - short_term_ema

    # Assess Volatility Adjustment
    daily_range = df['high'] - df['low']
    avg_range_volatility = daily_range.rolling(window=30).mean()

    # Incorporate Volume Influence
    volume_positive_days = (df['volume'].diff() > 0).rolling(window=20).sum()
    volume_negative_days = (df['volume'].diff() < 0).rolling(window=20).sum()
    avg_volume = df['volume'].rolling(window=30).mean()
    current_volume = df['volume']
    volume_deviation_score = current_volume - avg_volume
    volume_acceleration = (current_volume.diff() / current_volume.shift(1)).fillna(0)

    # Combine Momentum and Volume Indicators
    adjusted_momentum = momentum_diff * (1 / avg_range_volatility)
    momentum_by_volume = (short_term_ema * volume_positive_days) + (long_term_ema * volume_negative_days)
    adjusted_momentum_by_volume = adjusted_momentum * volume_acceleration

    # Calculate Enhanced Intraday Range
    enhanced_intraday_range = ((df['high'] - df['low']) + (df['close'] - df['open'])) / 2

    # Calculate Volume Adjusted Range
    volume_adjusted_range = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3) * df['volume']

    # Calculate Short-Term and Long-Term Rolling Average of Volume Adjusted Range
    short_term_avg_var = volume_adjusted_range.rolling(window=10).mean()
    long_term_avg_var = volume_adjusted_range.rolling(window=20).mean()

    # Integrate Volume and Volatility Adjusted Momentum
    integrated_momentum = momentum_by_volume + short_term_avg_var - long_term_avg_var + volume_deviation_score

    # Final Alpha Factor
    open_to_close_change = (df['close'] - df['open']) / df['open']
    final_factor = integrated_momentum + open_to_close_change + enhanced_intraday_range

    return final_factor
```
