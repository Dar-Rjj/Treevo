```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily price change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate short-term price momentum
    df['5_day_ema_price'] = df['daily_price_change'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_price'] = df['daily_price_change'].ewm(span=10, adjust=False).mean()
    df['short_term_momentum'] = df['5_day_ema_price'] - df['10_day_ema_price']
    
    # Calculate long-term price momentum
    df['20_day_sma_price'] = df['daily_price_change'].rolling(window=20).mean()
    df['50_day_sma_price'] = df['daily_price_change'].rolling(window=50).mean()
    df['long_term_momentum'] = df['20_day_sma_price'] - df['50_day_sma_price']
    
    # Calculate volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Count positive and negative volume days over the last 20 days
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    
    # Calculate volume momentum
    df['5_day_ema_vol'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_ema_vol'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_momentum'] = df['5_day_ema_vol'] - df['10_day_ema_vol']
    
    # Combine short-term price momentum with volume indicators
    df['combined_short_term'] = df['short_term_momentum'] * df['positive_volume_days'] * df['volume_momentum']
    
    # Combine long-term price momentum with volume indicators
    df['combined_long_term'] = df['long_term_momentum'] * df['negative_volume_days'] * df['volume_momentum']
    
    # Final alpha factor
    df['alpha_factor'] = df['combined_short_term'] - df['combined_long_term']
    
    return df['alpha_factor']
```
