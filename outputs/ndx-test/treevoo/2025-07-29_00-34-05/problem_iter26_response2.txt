```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum-Volume Integration
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Adjustment
    df['volume_impact'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Dynamic Price Range
    df['daily_dynamic_range'] = (df['high'] - df['low']) * (df['close'] / df['low'])
    
    # Average Trading Price
    df['average_trading_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Adjusted Dynamic Range
    df['volume_adjusted_dynamic_range'] = (df['daily_dynamic_range'] / df['average_trading_price']).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Short-Term and Long-Term Rolling Avg of Volume Adjusted Dynamic Range
    df['short_term_rolling_avg'] = df['volume_adjusted_dynamic_range'].rolling(window=15).sum() / 15
    df['long_term_rolling_avg'] = df['volume_adjusted_dynamic_range'].rolling(window=25).sum() / 25
    
    # Price Volatility
    df['price_volatility'] = df['close'].rolling(window=30).std()
    
    # Trade Activity Influence
    df['trade_activity_influence'] = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    df['combined_momentum_volume'] = df['momentum'] * df['volume_impact']
    
    # Adjust for Price Volatility
    df['volatility_adjusted_momentum'] = df['combined_momentum_volume'] / df['price_volatility']
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['short_term_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['long_term_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['momentum_difference'] = df['long_term_ema'] - df['short_term_ema']
    
    # Assess Volatility Adjustment
    df['daily_range'] = df['high'] - df['low']
    df['avg_range_volatility'] = df['daily_range'].rolling(window=30).mean()
    
    # Incorporate Volume Influence
    df['positive_volume_days'] = (df['volume'] > df['volume'].shift()).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume'] < df['volume'].shift()).rolling(window=20).sum()
    df['volume_deviation_score'] = df['volume'] - df['volume'].rolling(window=30).mean()
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift()) / df['volume'].shift()
    
    # Combine Momentum and Volume Indicators
    df['short_term_momentum_adj'] = df['short_term_ema'] * df['positive_volume_days']
    df['long_term_momentum_adj'] = df['long_term_ema'] * df['negative_volume_days']
    df['momentum_volatility_adjusted'] = df['momentum_difference'] * (1 / df['avg_range_volatility'])
    df['adjusted_by_volume_acceleration'] = df['momentum_volatility_adjusted'] * df['volume_acceleration']
    
    # Volume Adjusted Range
    df['volume_adjusted_range'] = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Short-Term and Long-Term Rolling Average of Volume Adjusted Range
    df['short_term_rolling_avg_range'] = df['volume_adjusted_range'].rolling(window=10).sum() / 10
    df['long_term_rolling_avg_range'] = df['volume_adjusted_range'].rolling(window=20).sum() / 20
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['short_term_rolling_avg'] 
        - df['long_term_rolling_avg'] 
        + df['short_term_rolling_avg_range'] 
        - df['long_term_rolling_avg_range'] 
        + df['volume_deviation_score'] 
        * (1 / df['price_volatility']) 
        * df['trade_activity_influence']
    )
    
    return df['alpha_factor']
```
