```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Price Momentum
    short_term_period = 5
    df['EMA_short'] = df['close'].ewm(span=short_term_period, adjust=False).mean()
    
    # Long-Term Price Momentum
    long_term_period = 20
    df['EMA_long'] = df['close'].ewm(span=long_term_period, adjust=False).mean()
    
    # Momentum Difference
    df['momentum_diff'] = df['EMA_short'] - df['EMA_long']
    
    # Volume Acceleration
    volume_change_period = 5
    df['volume_acceleration'] = df['volume'].pct_change(periods=volume_change_period)
    
    # Relative Strength Index (RSI)
    rsi_period = 14
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    
    avg_gain = gain.rolling(window=rsi_period).mean()
    avg_loss = loss.rolling(window=rsi_period).mean()
    
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Combine Momentum, Volume, and RSI
    df['alpha_factor'] = (df['momentum_diff'] * df['volume_acceleration']) + df['RSI']
    
    return df['alpha_factor'].dropna()
```
