```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum_diff = long_term_ema - short_term_ema

    # Assess Volatility Adjustment
    daily_range = df['high'] - df['low']
    avg_volatility = daily_range.rolling(window=30).mean()

    # Incorporate Volume Influence
    positive_volume_days = (df['volume'] > df['volume'].shift(1)).rolling(window=20).sum()
    negative_volume_days = (df['volume'] < df['volume'].shift(1)).rolling(window=20).sum()
    avg_30_day_volume = df['volume'].rolling(window=30).mean()
    volume_deviation_score = df['volume'] - avg_30_day_volume
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Combine Momentum and Volume Indicators
    adjusted_momentum = momentum_diff * (1 / avg_volatility)
    adjusted_momentum *= volume_acceleration

    # Calculate Enhanced Intraday Range
    intraday_range = (df['high'] - df['low']) + (df['close'] - df['open'])
    enhanced_intraday_range = intraday_range / 2

    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    volume_adjusted_range = ((df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3)) * df['volume']
    short_term_avg_var = volume_adjusted_range.rolling(window=10).mean()

    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    long_term_avg_var = volume_adjusted_range.rolling(window=20).mean()

    # Final Alpha Factor
    alpha_factor = (
        adjusted_momentum +
        short_term_avg_var -
        long_term_avg_var
    ) / (enhanced_intraday_range * df['close'])
    alpha_factor += volume_deviation_score

    return alpha_factor
```
