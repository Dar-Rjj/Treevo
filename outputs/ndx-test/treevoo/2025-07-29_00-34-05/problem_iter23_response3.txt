import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum_diff = short_term_ema - long_term_ema

    # Assess Volatility Adjustment
    daily_range = df['high'] - df['low']
    avg_daily_range_volatility = daily_range.rolling(window=30).mean()

    # Incorporate Volume Influence
    positive_volume_days = (df['volume'] > df['volume'].shift()).rolling(window=20).sum()
    negative_volume_days = (df['volume'] < df['volume'].shift()).rolling(window=20).sum()
    avg_30_day_volume = df['volume'].rolling(window=30).mean()
    volume_deviation_score = df['volume'] - avg_30_day_volume
    volume_acceleration = (df['volume'] - df['volume'].shift()) / df['volume'].shift()

    # Combine Momentum and Volume Indicators
    combined_momentum = (short_term_ema * positive_volume_days) + (long_term_ema * negative_volume_days)
    adjusted_momentum_by_volatility = combined_momentum / avg_daily_range_volatility
    adjusted_momentum_by_volume_acceleration = adjusted_momentum_by_volatility * volume_acceleration

    # Calculate Volume Adjusted Range
    volume_adjusted_range = ((df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3)) * df['volume']

    # Calculate Short-Term and Long-Term Rolling Average of Volume Adjusted Range
    short_term_avg_var = volume_adjusted_range.rolling(window=10).mean()
    long_term_avg_var = volume_adjusted_range.rolling(window=20).mean()

    # Integrate Volume and Volatility Adjusted Momentum
    integrated_factor = (adjusted_momentum_by_volume_acceleration 
                         + short_term_avg_var 
                         - long_term_avg_var 
                         + volume_deviation_score)

    # Include Open to Close Price Change
    open_close_price_change = (df['close'] - df['open']) / df['open']
    final_alpha_factor = integrated_factor + open_close_price_change

    return final_alpha_factor
