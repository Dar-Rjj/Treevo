```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Range
    intraday_range = df['high'] - df['low']
    open_close_diff = df['close'] - df['open']
    enhanced_intraday_range = (intraday_range + open_close_diff) / 2
    
    # Adjust by Short-Term Momentum and Volume Trend
    short_term_momentum = df['high'].ewm(span=5).mean() - df['low'].ewm(span=10).mean()
    volume_momentum = df['volume'].rolling(window=5).sum() - df['volume'].rolling(window=10).sum()
    final_factor = (short_term_momentum * volume_momentum) / (enhanced_intraday_range * df['amount'])
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Average Price
    average_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / average_price.replace(0, np.nan)) * df['volume']
    
    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    short_term_avg_volume_adjusted_range = volume_adjusted_range.rolling(window=15).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    long_term_avg_volume_adjusted_range = volume_adjusted_range.rolling(window=25).mean()
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=30).std()
    
    # Trade Amount Influence
    avg_amount = df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    combined_momentum = final_factor / volume_adjusted_range
    
    # Adjust for Price Volatility
    adjusted_combined_momentum = combined_momentum / price_volatility
    
    # Calculate High-Low Range Momentum
    high_low_range = df['high'] - df['low']
    ema_high_low_range = high_low_range.ewm(span=7, adjust=False).mean()
    daily_change_ema = ema_high_low_range.diff()
    
    # Calculate Volume Shock
    volume_percentage_change = df['volume'].pct_change()
    significant_volume_change = (np.abs(volume_percentage_change) > 0.6).astype(int)
    
    # Combine Indicators
    combined_indicator = daily_change_ema * significant_volume_change
    
    # Introduce Lagged Volume Shock
    lagged_volume_shock = volume_percentage_change.shift(2)
    combined_indicator *= lagged_volume_shock
    
    # Integrate High-Frequency Momentum
    ema_3_high = df['high'].ewm(span=3, adjust=False).mean()
    ema_3_low = df['low'].ewm(span=3, adjust=False).mean()
    high_freq_momentum = ema_3_high - ema_3_low
    
    # Incorporate Trade Amount Momentum
    ema_5_amount = df['amount'].ewm(span=5, adjust=False).mean()
    ema_10_amount = df['amount'].ewm(span=10, adjust=False).mean()
    amount_momentum = ema_5_amount - ema_10_amount
    
    # Finalize Alpha Factor
    alpha_factor = (short_term_avg_volume_adjusted_range - long_term_avg_volume_adjusted_range) * combined_indicator
    alpha_factor = alpha_factor * high_freq_momentum * amount_momentum / avg_amount
    alpha_factor = alpha_factor / price_volatility
    
    return alpha_factor
```
