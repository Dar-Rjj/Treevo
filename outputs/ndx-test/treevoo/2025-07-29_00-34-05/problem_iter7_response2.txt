import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Spread
    intraday_spread = df['high'] - df['low']
    gap_spread = df['close'] - df['open']
    enhanced_intraday_spread = intraday_spread + gap_spread
    
    # Calculate Short-Term Price Momentum
    short_term_ema_close_5 = df['close'].ewm(span=5, adjust=False).mean()
    short_term_ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = short_term_ema_close_5 - short_term_ema_close_10
    
    # Calculate Volume Momentum
    ema_volume_5 = df['volume'].ewm(span=5, adjust=False).mean()
    ema_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_momentum = ema_volume_5 - ema_volume_10
    
    # Final Factor
    factor_value = (price_momentum * volume_momentum) / (enhanced_intraday_spread * df['volume'])
    
    return factor_value
