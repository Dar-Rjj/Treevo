```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate 20-day Moving Average of Daily Price Changes (Price Momentum)
    df['price_momentum'] = df['daily_price_change'].rolling(window=20).mean()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Count Positive Volume Days over the Last 20 Days
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    
    # Short-Term Momentum and Volume
    df['short_term_factor'] = df['price_momentum'] * df['positive_volume_days']
    
    # Final Alpha Factor
    alpha_factor = df['short_term_factor']
    
    return alpha_factor
```
