```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the close price and the average price
    diff_close_avg = df['close'] - avg_price
    
    # Calculate the ratio of volume to the moving average of volume over a 10-day window
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Heuristic factor: The product of the price difference and the volume ratio
    heuristic_factor = diff_close_avg * volume_ratio
    
    return heuristic_factor
```
