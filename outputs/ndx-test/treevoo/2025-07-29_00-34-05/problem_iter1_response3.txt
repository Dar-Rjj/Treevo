```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Factor
    df['momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Assess Volatility Adjustment
    df['daily_range'] = df['high'] - df['low']
    df['average_range_volatility'] = df['daily_range'].rolling(window=30).mean()
    
    # Incorporate Volume Influence
    df['average_volume_30'] = df['volume'].rolling(window=30).mean()
    df['volume_deviation_score'] = df['volume'] - df['average_volume_30']
    
    # Adjust Momentum by Volatility
    df['adjusted_momentum'] = df['momentum'] / df['average_range_volatility']
    
    # Combine Factors
    df['alpha_factor'] = df['adjusted_momentum'] + df['volume_deviation_score']
    
    return df['alpha_factor'].dropna()
```
