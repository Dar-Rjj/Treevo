```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'].diff().ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'].diff().ewm(span=20, adjust=False).mean()
    
    # Volume Indicator: Count Positive and Negative Volume Days Over the Last 20 Days
    positive_volume_days = (df['volume'] > df['volume'].shift()).rolling(window=20).sum()
    negative_volume_days = (df['volume'] < df['volume'].shift()).rolling(window=20).sum()
    
    # Combine Short-Term Price Momentum and Volume Indicator
    combined_short_term = short_term_momentum * positive_volume_days
    
    # Combine Long-Term Price Momentum and Volume Indicator
    combined_long_term = long_term_momentum * negative_volume_days
    
    # Calculate Volume Adjusted Range
    volume_adjusted_range = ((df['high'] - df['low']) / 
                             ((df['high'] + df['low'] + df['close']) / 3) * 
                             df['volume'])
    
    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    short_term_vol_adj_range_avg = volume_adjusted_range.rolling(window=10).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    long_term_vol_adj_range_avg = volume_adjusted_range.rolling(window=20).mean()
    
    # Compute Momentum Difference
    momentum_difference = long_term_momentum - short_term_momentum
    
    # Calculate Volume Acceleration
    volume_acceleration = ((df['volume'] - df['volume'].shift()) / 
                           df['volume'].shift() * 
                           momentum_difference)
    
    # Final Alpha Factor
    final_factor = (combined_short_term + combined_long_term + 
                    short_term_vol_adj_range_avg - long_term_vol_adj_range_avg + 
                    volume_acceleration)
    
    return final_factor
```
