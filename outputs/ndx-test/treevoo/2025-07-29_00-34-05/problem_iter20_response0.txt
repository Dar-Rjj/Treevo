import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Factor
    df['Volume_Impact_Factor'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Volume Adjusted Range
    df['Volume_Adjusted_Range'] = (df['Daily_Price_Range'] / df['close']).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Short-Term Rolling Avg of Volume Adjusted Range
    df['Short_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=15).mean()
    
    # Long-Term Rolling Avg of Volume Adjusted Range
    df['Long_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=25).mean()
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=30).std()
    
    # Trade Amount Influence
    df['Trade_Amount_Influence'] = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    df['Combined_Momentum_Volume'] = df['Momentum'] / df['Volume_Impact_Factor']
    
    # Adjust for Price Volatility
    df['Adjusted_Momentum_Volume'] = df['Combined_Momentum_Volume'] / df['Price_Volatility']
    
    # High-Low Range Momentum
    smoothing_factor = 2 / (7 + 1)
    df['EMA_Daily_Price_Range'] = df['Daily_Price_Range'].ewm(alpha=smoothing_factor, adjust=False).mean()
    df['High_Low_Range_Momentum'] = df['EMA_Daily_Price_Range'].diff()
    
    # Volume Shock
    df['Volume_Shock'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Significant_Volume_Change'] = df['Volume_Shock'].apply(lambda x: 1 if abs(x) > 0.6 else 0)
    
    # Combine Indicators
    df['Combined_Indicators'] = df['High_Low_Range_Momentum'] * np.sign(df['Significant_Volume_Change'])
    
    # Finalize Alpha Factor
    df['Alpha_Factor'] = (
        (df['Short_Term_Volume_Adjusted_Range'] - df['Long_Term_Volume_Adjusted_Range']) 
        * df['Combined_Indicators'] 
        * df['Adjusted_Momentum_Volume'] 
        * df['Trade_Amount_Influence']
    )
    
    return df['Alpha_Factor']
