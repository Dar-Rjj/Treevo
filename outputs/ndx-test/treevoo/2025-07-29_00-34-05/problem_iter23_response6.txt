```python
def heuristics_v2(df):
    # Calculate the change in close price from previous day
    df['close_change'] = df['close'].diff()
    
    # Calculate the amplitude of the price movement (high - low)
    df['amplitude'] = df['high'] - df['low']
    
    # Calculate the ratio of close change to amplitude
    df['close_change_to_amplitude_ratio'] = df['close_change'] / df['amplitude']
    
    # Calculate the signed volume, positive if close is higher than open, negative otherwise
    df['signed_volume'] = df['volume'] * ((df['close'] > df['open']).astype(int) * 2 - 1)
    
    # Calculate the factor as a weighted sum of the close change to amplitude ratio and the signed volume
    df['factor'] = df['close_change_to_amplitude_ratio'] * df['signed_volume']
    
    return df['factor']
```
