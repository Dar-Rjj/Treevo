```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
        df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
    
    Returns:
        pd.Series: Alpha factor values
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the closing price relative to the price range
    close_to_range = (df['close'] - df['low']) / price_range
    
    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine the two signals
    alpha_factor = close_to_range * (df['close'] - vwap)
    
    return alpha_factor
```
