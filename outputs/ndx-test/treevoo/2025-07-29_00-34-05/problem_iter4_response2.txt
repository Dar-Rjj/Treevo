```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum (5-day EMA of Close price)
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum (20-day EMA of Close price)
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum Difference
    momentum_diff = long_term_ema - short_term_ema
    
    # Calculate Volume Acceleration (Percentage Change in Volume for 5 days)
    volume_acceleration = df['volume'].pct_change(periods=5)
    
    # Multiply Momentum Difference by Volume Acceleration to get the final factor
    factor_values = momentum_diff * volume_acceleration
    
    return factor_values
```
