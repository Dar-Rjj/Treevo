```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum-Volume Integration
    momentum = df['close'] - df['close'].shift(20)
    
    # Volume Impact Adjustment
    volume_impact = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Dynamic Price Range
    daily_dynamic_range = (df['high'] - df['low']) * (df['close'] / df['low'])
    
    # Average Trading Price
    avg_trading_price = (df['high'] + df['low'] + df['close']) / 3
    avg_trading_price[avg_trading_price == 0] = 1  # Ensure non-zero division
    
    # Volume Adjusted Dynamic Range
    volume_adjusted_range = (daily_dynamic_range / avg_trading_price) * df['volume']
    
    # Short-Term and Long-Term Rolling Avg of Volume Adjusted Dynamic Range
    short_term_volume_adjusted = volume_adjusted_range.rolling(window=15).sum() / 15
    long_term_volume_adjusted = volume_adjusted_range.rolling(window=25).sum() / 25
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=30).std()
    
    # Combine Momentum and Volume Impact
    combined_momentum_volume = momentum * volume_impact
    
    # Adjust for Price Volatility
    adjusted_momentum_volume = combined_momentum_volume / price_volatility
    
    # Calculate Short-Term and Long-Term EMA of Close
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    ema_diff = short_ema - long_ema
    
    # Assess Volatility Adjustment
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=30).mean()
    
    # Incorporate Volume Influence
    positive_volume_days = (df['volume'].diff() > 0).rolling(window=20).sum()
    negative_volume_days = (df['volume'].diff() < 0).rolling(window=20).sum()
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine Momentum and Volume Indicators
    short_term_momentum = ema_diff * positive_volume_days
    long_term_momentum = ema_diff * negative_volume_days
    adjusted_momentum_volatility = (short_term_momentum - long_term_momentum) / price_volatility
    adjusted_momentum_volatility *= volume_acceleration
    
    # Volume Adjusted Range
    vol_adjusted_range = (df['high'] - df['low']) / avg_trading_price * df['volume']
    
    # Short-Term and Long-Term Rolling Avg of Volume Adjusted Range
    short_term_vol_adjusted = vol_adjusted_range.rolling(window=10).sum() / 10
    long_term_vol_adjusted = vol_adjusted_range.rolling(window=20).sum() / 20
    
    # Final Alpha Factor
    final_alpha_factor = (
        (short_term_volume_adjusted - long_term_volume_adjusted) +
        (short_term_vol_adjusted - long_term_vol_adjusted) +
        adjusted_momentum_volatility * df['volume']
    )
    
    return final_alpha_factor
```
