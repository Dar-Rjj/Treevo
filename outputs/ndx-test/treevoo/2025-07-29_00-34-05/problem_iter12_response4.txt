```python
import pandas as pd

def heuristics(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Factor
    df['Volume_Impact_Factor'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['Average_Price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    df['Volume_Adjusted_Range'] = (df['Daily_Price_Range'] / df['Average_Price']).fillna(0) * df['volume']
    
    # Calculate Short-Term Rolling Average of Volume Adjusted Range
    df['Short_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=15).mean()
    
    # Calculate Long-Term Rolling Average of Volume Adjusted Range
    df['Long_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=25).mean()
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=30).std()
    
    # Trade Amount Influence
    df['Trade_Amount_Influence'] = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    df['Combined_Momentum_Volume'] = df['Momentum'] / df['Volume_Impact_Factor']
    
    # Adjust for Price Volatility
    df['Adjusted_Combined_Momentum_Volume'] = df['Combined_Momentum_Volume'] / df['Price_Volatility']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate High-Low Range Momentum
    N = 7
    alpha = 2 / (N + 1)
    df['EMA_High_Low_Range'] = df['High_Low_Range'].ewm(alpha=alpha, adjust=False).mean()
    df['Daily_Change_in_EMA'] = df['EMA_High_Low_Range'] - df['EMA_High_Low_Range'].shift(1)
    
    # Calculate Volume Shock
    df['Volume_Percentage_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) * 100
    df['Volume_Shock_Flag'] = (df['Volume_Percentage_Change'].abs() > 60).astype(int)
    
    # Combine Indicators
    df['Combined_Indicator'] = df['Daily_Change_in_EMA'] * df['Volume_Shock_Flag']
    
    # Finalize Alpha Factor
    df['Alpha_Factor'] = (
        df['Short_Term_Volume_Adjusted_Range'] +
        df['Long_Term_Volume_Adjusted_Range'] +
        df['Combined_Indicator'] +
        df['Adjusted_Combined_Momentum_Volume']
    ) * df['Trade_Amount_Influence']
    
    return df['Alpha_Factor']
```
