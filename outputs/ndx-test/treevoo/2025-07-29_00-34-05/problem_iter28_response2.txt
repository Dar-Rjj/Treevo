```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-Based Factors
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['200_day_SMA'] = df['close'].rolling(window=200).mean()
    df['SMA_Crossover'] = df['50_day_SMA'] - df['200_day_SMA']
    
    df['ROC_12'] = df['close'].pct_change(periods=12)
    
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Long_Term_Momentum'] = df['5_day_EMA'] - df['20_day_EMA']
    
    # Reversal-Based Factors
    df['RSI_14'] = 100 - (100 / (1 + df['close'].diff().where(lambda x: x > 0, 0).rolling(window=14).mean() / df['close'].diff().where(lambda x: x < 0, 0).abs().rolling(window=14).mean()))
    
    df['Close_Open_Ratio'] = df['close'] / df['open'].shift(1)
    mean_ratio = df['Close_Open_Ratio'].mean()
    df['Extreme_Price_Movements'] = (df['Close_Open_Ratio'] - mean_ratio).abs()
    
    # Volume-Price Interaction
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['CMF_20'] = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    df['Volume_MA_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Deviation_Score'] = df['Volume_MA_5'] - df['Volume_MA_10']
    
    # Hybrid Factors
    df['Momentum_Volume_Combined'] = df['ROC_12'] * df['CMF_20']
    df['Reversal_Volume_Combined'] = df['RSI_14'] * df['OBV']
    
    # Enhanced Intraday Range
    df['Intraday_High_Low'] = df['high'] - df['low']
    df['Intraday_Open_Close'] = df['close'] - df['open']
    df['Enhanced_Intraday_Range'] = (df['Intraday_High_Low'] + df['Intraday_Open_Close']) / 2
    
    # Volatility Adjustment
    df['Daily_Range'] = df['high'] - df['low']
    df['30_Day_Range_Sum'] = df['Daily_Range'].rolling(window=30).sum()
    df['Volatility_Score'] = df['30_Day_Range_Sum'] / 30
    
    # New Factor: Price-Volatility Ratio
    df['Price_Volatility_Ratio'] = df['close'] / df['Volatility_Score']
    
    # Combine Factors
    df['Adjusted_Momentum'] = df['SMA_Crossover'] * (1 / df['Volatility_Score'])
    df['Volume_Adjusted_Momentum'] = df['Adjusted_Momentum'] * df['Volume_Deviation_Score']
    df['Final_Alpha_Factor'] = (df['Volume_Adjusted_Momentum'] + df['Volume_Deviation_Score']) / (df['Enhanced_Intraday_Range'] * df['close']) * df['Price_Volatility_Ratio']
    
    return df['Final_Alpha_Factor'].dropna()
```
