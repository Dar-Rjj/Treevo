```python
import pandas as pd

def heuristics_v2(df, n=14, m=10):
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = (df['high'] - df['prev_close']).abs()
    df['low_prev_close'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # Compute Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Determine Momentum
    df['price_change'] = df['close'] - df['close'].shift(n)
    df['momentum_ratio'] = df['price_change'] / df['atr']
    
    # Adjust for Volume Impact
    df['volume_smoothing'] = df['volume'].rolling(window=m).mean()
    df['adjusted_momentum'] = df['momentum_ratio'] * df['volume_smoothing']
    
    # Final Alpha Factor
    alpha_factor = df['adjusted_momentum']
    
    return alpha_factor
```
