```python
def heuristics_v2(df):
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the average of the high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the weighted average of open, high, and low prices with weights 0.2, 0.3, and 0.5 respectively
    weighted_avg_ohl = 0.2 * df['open'] + 0.3 * df['high'] + 0.5 * df['low']
    
    # Calculate the difference between the close price and the weighted average of open, high, and low
    diff_close_weighted_ohl = df['close'] - weighted_avg_ohl
    
    # Calculate the factor as the product of the percentage change in close price, the average of high and low, and the difference calculated above
    factor = pct_change_close * avg_high_low * diff_close_weighted_ohl
    
    return factor
```
