```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 5-day and 20-day EMA for Short-Term and Long-Term Momentum
    df['short_term_momentum'] = df['price_change'].ewm(span=5, adjust=False).mean()
    df['long_term_momentum'] = df['price_change'].ewm(span=20, adjust=False).mean()
    
    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Count Positive and Negative Volume Days over the Last 20 Days
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    
    # Combine Short-Term and Long-Term Momentum with Volume Indicator
    df['combined_short_term_momentum'] = df['short_term_momentum'] * df['positive_volume_days']
    df['combined_long_term_momentum'] = df['long_term_momentum'] * df['negative_volume_days']
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Range
    df['volume_adjusted_range'] = (df['daily_price_range'] / ((df['high'] + df['low'] + df['close']) / 3)) * df['volume']
    
    # Calculate Short-Term and Long-Term Rolling Average of Volume Adjusted Range
    df['short_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=10).mean()
    df['long_term_vol_adj_range'] = df['volume_adjusted_range'].rolling(window=20).mean()
    
    # Compute Momentum Difference
    df['momentum_difference'] = df['long_term_momentum'] - df['short_term_momentum']
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) * df['momentum_difference']
    
    # Combine Combined Momentum and Volume Factors
    df['combined_momentum_factors'] = df['combined_short_term_momentum'] + df['combined_long_term_momentum']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['combined_momentum_factors'] +
        df['short_term_vol_adj_range'] -
        df['long_term_vol_adj_range'] +
        df['volume_acceleration']
    )
    
    return df['final_alpha_factor']
```
