```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum_difference = long_term_ema - short_term_ema
    
    # Assess Volatility Adjustment
    daily_range = df['high'] - df['low']
    avg_daily_range_volatility = daily_range.rolling(window=30).mean()
    
    # Incorporate Volume Influence
    avg_30day_volume = df['volume'].rolling(window=30).mean()
    volume_deviation_score = (df['volume'] - avg_30day_volume) / avg_30day_volume
    volume_acceleration = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_acceleration = volume_acceleration.rolling(window=5).mean()  # 5-day average of volume acceleration

    # Combine Factors
    adjusted_momentum = momentum_difference / avg_daily_range_volatility
    adjusted_momentum_by_volume = adjusted_momentum * volume_acceleration
    final_alpha_factor = adjusted_momentum_by_volume + volume_deviation_score

    return final_alpha_factor
```
