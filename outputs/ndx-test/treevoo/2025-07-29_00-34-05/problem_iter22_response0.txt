```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['Average_Price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    df['Volume_Adjusted_Range'] = (df['Daily_Price_Range'] / df['Average_Price']).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Short-Term Rolling Average of Volume Adjusted Range
    df['Short_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=10).mean()
    
    # Long-Term Rolling Average of Volume Adjusted Range
    df['Long_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=20).mean()
    
    # Combine Intraday Measures
    df['Intraday_Measure'] = ((df['high'] - df['low']) + (df['close'] - df['open'])) * df['volume']
    
    # Calculate Short-Term Price Momentum
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['5_day_EMA'] = df['Price_Change'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA'] = df['Price_Change'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Momentum'] = df['5_day_EMA'] - df['10_day_EMA']
    
    # Calculate Long-Term Price Momentum
    df['20_day_SMA'] = df['Price_Change'].rolling(window=20).mean()
    df['50_day_SMA'] = df['Price_Change'].rolling(window=50).mean()
    df['Long_Term_Price_Momentum'] = df['20_day_SMA'] - df['50_day_SMA']
    
    # Calculate Volume Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Positive_Volume_Days'] = df['Volume_Change'].apply(lambda x: 1 if x > 0 else 0).rolling(window=20).sum()
    df['Negative_Volume_Days'] = df['Volume_Change'].apply(lambda x: 1 if x < 0 else 0).rolling(window=20).sum()
    
    # Volume Momentum
    df['5_day_Volume_EMA'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_Volume_EMA'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['Volume_Momentum'] = df['5_day_Volume_EMA'] - df['10_day_Volume_EMA']
    
    # Final Factor Adjustment
    df['Final_Factor_Adjustment'] = (df['Short_Term_Price_Momentum'] * df['Volume_Momentum']) / df['Intraday_Measure']
    
    # Combine Momentum, Volume, and High-Low Range
    df['Combined_Factor'] = (df['Short_Term_Price_Momentum'] * df['Volume_Momentum'] * df['Daily_Price_Range']) / df['close']
    
    # Combine Factors
    df['Alpha_Factor'] = (df['Short_Term_Volume_Adjusted_Range'] - df['Long_Term_Volume_Adjusted_Range'] + df['Final_Factor_Adjustment'])
    
    return df['Alpha_Factor']
```
