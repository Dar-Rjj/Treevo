```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = short_ema - long_ema
    
    # Calculate Long-Term Price Momentum
    long_term_period = 20
    long_term_ema = df['close'].ewm(span=long_term_period, adjust=False).mean()
    momentum_difference = short_ema - long_term_ema
    
    # Calculate Volume Acceleration
    volume_percentage_change = df['volume'].pct_change(periods=5)
    volume_acceleration = momentum_difference * volume_percentage_change
    
    # Adjust Momentum by Volume Trend
    volume_momentum = df['volume'].ewm(span=5, adjust=False).mean() - df['volume'].ewm(span=10, adjust=False).mean()
    
    # Introduce High-Low Range Factor
    high_low_range_factor = (df['high'] - df['low']) / df['close']
    
    # Combine Momentum, Volume, and High-Low Range
    factor = (short_term_momentum * volume_momentum * high_low_range_factor) / df['close']
    
    return factor
```
