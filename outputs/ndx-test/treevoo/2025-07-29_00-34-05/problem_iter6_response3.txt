```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Factor
    df['Volume_Impact'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Price Volatility
    df['Price_Volatility'] = df['high'].rolling(window=30).std()
    
    # Trade Amount Influence
    df['Trade_Amount_Influence'] = df['amount'] / df['amount'].rolling(window=30).median()
    
    # Combine Momentum and Volume Impact
    df['Momentum_Volume'] = df['Momentum'] / df['Volume_Impact']
    
    # Adjust for Price Volatility
    df['Adjusted_MV'] = df['Momentum_Volume'] / df['Price_Volatility']
    
    # Final Adjustment with Trade Amount
    df['Factor_Value'] = df['Adjusted_MV'] * df['Trade_Amount_Influence']
    
    return df['Factor_Value']
```
