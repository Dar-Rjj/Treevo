```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Range
    df['volume_adjusted_range'] = (df['high'] - df['low']) / ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Short-Term (10 days) Rolling Average of Volume Adjusted Range
    df['short_term_avg_var'] = df['volume_adjusted_range'].rolling(window=10).sum() / 10
    
    # Calculate Long-Term (20 days) Rolling Average of Volume Adjusted Range
    df['long_term_avg_var'] = df['volume_adjusted_range'].rolling(window=20).sum() / 20
    
    # Calculate Short-Term (5 days) Price Momentum using EMA
    df['short_term_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term (20 days) Price Momentum using EMA
    df['long_term_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum Difference
    df['momentum_diff'] = df['long_term_ema'] - df['short_term_ema']
    
    # Calculate Volume Acceleration
    df['volume_acceleration'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1) * df['momentum_diff']
    
    # Combine Factors
    df['alpha_factor'] = (df['short_term_avg_var'] 
                          - df['long_term_avg_var'] 
                          + df['volume_acceleration'])
    
    return df['alpha_factor']
```
