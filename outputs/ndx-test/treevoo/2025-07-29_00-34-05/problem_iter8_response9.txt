```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Factor
    df['momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Assess Volatility Adjustment using High-Low Range
    df['daily_range'] = df['high'] - df['low']
    df['avg_volatility'] = df['daily_range'].rolling(window=30).mean()
    
    # Incorporate High-Low Range Momentum
    alpha = 2 / (5 + 1)  # Smoothing factor for EMA
    df['ema_5'] = df['daily_range'].ewm(alpha=alpha, adjust=False).mean()
    df['ema_daily_change'] = df['ema_5'] - df['ema_5'].shift(1)
    
    # Calculate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['ma_20'] = df['price_change'].rolling(window=20).mean()
    df['ma_50'] = df['price_change'].rolling(window=50).mean()
    
    # Calculate Volume Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    df['significant_volume_change'] = (df['volume_change'].abs() > 0.5 * df['volume'].shift(1)).astype(int)
    
    # Combine Factors
    df['adjusted_momentum'] = df['momentum'] / df['avg_volatility']
    df['high_low_range_adjusted'] = df['adjusted_momentum'] * df['ema_daily_change']
    df['long_term_factor'] = df['ma_50'] * df['positive_volume_days']
    df['short_term_factor'] = df['ma_20'] * df['negative_volume_days']
    
    # Final Alpha Factor
    df['final_alpha'] = (df['long_term_factor'] + df['short_term_factor']) * df['ema_daily_change'] * df['volume_change'].apply(lambda x: 1 if x > 0 else -1)
    
    return df['final_alpha']
```
