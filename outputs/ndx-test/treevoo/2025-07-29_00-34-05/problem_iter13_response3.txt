    # Calculate Average Price
    df['Average_Price'] = (df['high'] + df['low'] + df['close']) / 3

    # Calculate Short-Term and Long-Term Rolling Averages of Volume Adjusted Range
    df['Short_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=10).mean()
    df['Long_Term_Volume_Adjusted_Range'] = df['Volume_Adjusted_Range'].rolling(window=28).mean()

    # Calculate Short-Term Price Momentum
    df['5_day_EMA_Close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA_Close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Momentum'] = df['5_day_EMA_Close'] - df['10_day_EMA_Close']

    # Assess Volatility Adjustment
    df['Daily_Range'] = df['high'] - df['low']
    df['30_day_Avg_Range'] = df['Daily_Range'].rolling(window=30).mean()
    df['Volatility_Score'] = 1 / (df['30_day_Avg_Range'] + 1e-9)  # Add small value to avoid division by zero

    # Incorporate Volume Influence
    df['5_day_EMA_Volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['10_day_EMA_Volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['Volume_Deviation_Score'] = df['5_day_EMA_Volume'] - df['10_day_EMA_Volume']

    # Combine Momentum, Volatility, and Volume
    df['Adjusted_Momentum'] = df['Short_Term_Price_Momentum'] * df['Volatility_Score']
    df['Combined_Factor'] = df['Adjusted_Momentum'] * df['Volume_Deviation_Score']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Short_Term_Volume_Adjusted_Range'] - df['Long_Term_Volume_Adjusted_Range']) * df['Combined_Factor']

    return df['Final_Alpha_Factor']
