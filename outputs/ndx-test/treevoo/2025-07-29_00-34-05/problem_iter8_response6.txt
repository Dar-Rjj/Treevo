```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    df['momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Factor
    df['volume_impact'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Price Volatility
    df['price_volatility'] = df['high'].rolling(window=30).std()
    
    # Trade Amount Influence
    df['trade_amount_influence'] = df['amount'] / df['amount'].rolling(window=30).median()
    
    # Combine Momentum and Volume Impact
    df['combined_momentum_volume'] = df['momentum'] / df['volume_impact']
    
    # Adjust for Price Volatility
    df['adjusted_momentum_volatility'] = df['combined_momentum_volume'] / df['price_volatility']
    
    # Final Adjustment with Trade Amount
    df['final_adjusted_factor'] = df['adjusted_momentum_volatility'] * df['trade_amount_influence']
    
    # Calculate Short-Term Price Momentum (EMA 5 days)
    df['ema_short'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum (EMA 20 days)
    df['ema_long'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum Difference
    df['momentum_difference'] = df['ema_long'] - df['ema_short']
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Calculate Average Price
    df['average_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    df['volume_adjusted_range'] = (df['daily_price_range'] / df['average_price'].replace(0, 1e-6)) * df['volume']
    
    # Calculate Short-Term Rolling Average (last 14 days)
    df['short_term_rolling_avg'] = df['volume_adjusted_range'].rolling(window=14).sum() / 14
    
    # Calculate Long-Term Rolling Average (last 28 days)
    df['long_term_rolling_avg'] = df['volume_adjusted_range'].rolling(window=28).sum() / 28
    
    # Adjust with Rolling Averages
    df['rolling_avg_difference'] = df['short_term_rolling_avg'] - df['long_term_rolling_avg']
    
    # Combine Momentum and Volume Adjusted Range
    df['combined_factor'] = df['final_adjusted_factor'] * df['volume_adjusted_range']
    
    # Final adjustment with rolling averages
    df['alpha_factor'] = df['combined_factor'] * df['rolling_avg_difference']
    
    return df['alpha_factor']
```
