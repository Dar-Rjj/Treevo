import pandas as pd

def heuristics_v2(df):
    # Daily Volume Delta Factor
    df['Daily_Volume_Delta'] = df['volume'].diff()
    
    # Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # True Range Calculation
    df['True_Range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1).shift(1)
    
    # Simple Moving Average (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Exponential Moving Average (EMA)
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # New High or Low
    df['New_High'] = df['high'].rolling(window=20).max().eq(df['high'])
    
    # Price Reversal
    df['Price_Reversal'] = df['close'].gt(df['close'].rolling(window=50).mean())
    
    # Plus Directional Indicator (+DI)
    df['DM_plus'] = df['high'].diff().apply(lambda x: max(x, 0))
    df['+DI'] = df['DM_plus'].rolling(window=14).mean() / df['ATR_14']
    
    # Minus Directional Indicator (-DI)
    df['DM_minus'] = df['low'].diff().apply(lambda x: -min(x, 0))
    df['-DI'] = df['DM_minus'].rolling(window=14).mean() / df['ATR_14']
    
    # Moving Average Crossover
    df['Fast_MA'] = df['close'].rolling(window=5).mean()
    df['Slow_MA'] = df['close'].rolling(window=20).mean()
    df['MA_Crossover'] = df['Fast_MA'].gt(df['Slow_MA'])
    
    # On-Balance Volume (OBV)
    df['OBV'] = (df['close'].diff() > 0).astype(int) * df['volume'] - (df['close'].diff() < 0).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()
    
    # Volume Spike
    df['Volume_Spike'] = df['volume'].gt(df['volume'].rolling(window=20).mean() * 2)
    
    # VWAP Momentum
    df['VWAP_Momentum'] = (df['VWAP'] - df['VWAP'].shift(1)) / df['VWAP'].shift(1)
    
    # VWPO
    df['VWAP_10'] = df['VWAP'].rolling(window=10).mean()
    df['VWAP_30'] = df['VWAP'].rolling(window=30).mean()
    df['VWPO'] = df['VWAP_10'] - df['VWAP_30']
    
    # Price Channel Breakout
    df['High_Channel'] = df['high'].rolling(window=20).max()
    df['Low_Channel'] = df['low'].rolling(window=20).min()
    df['Channel_Breakout'] = (df['high'] > df['High_Channel']) | (df['low'] < df['Low_Channel'])
