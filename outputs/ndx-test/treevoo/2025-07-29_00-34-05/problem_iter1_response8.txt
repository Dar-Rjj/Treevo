```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine Intraday Measures
    combined_measures = (high_low_spread + close_open_diff) * df['volume']
    
    # Calculate Volume-to-Price Ratio
    volume_to_price_ratio = df['volume'] / df['close']
    
    # Weight by Volume-to-Price Ratio
    factor_values = combined_measures * volume_to_price_ratio
    
    return factor_values
```
