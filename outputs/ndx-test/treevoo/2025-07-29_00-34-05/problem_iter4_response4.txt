def heuristics_v2(df):
    # Calculate Momentum Factor
    df['momentum'] = (df['close'] - df['close'].shift(30)) / df['close'].shift(30)
    
    # Assess Volatility Adjustment
    df['daily_range'] = df['high'] - df['low']
    df['range_volatility_30d'] = df['daily_range'].rolling(window=30).sum() / 30
    
    # Incorporate Volume Influence
    df['avg_volume_30d'] = df['volume'].rolling(window=30).mean()
    df['volume_deviation'] = df['volume'] - df['avg_volume_30d']
    
    # Combine Factors
    df['adjusted_momentum'] = df['momentum'] / df['range_volatility_30d']
    df['alpha_factor'] = df['adjusted_momentum'] + df['volume_deviation']
    
    return df['alpha_factor']
