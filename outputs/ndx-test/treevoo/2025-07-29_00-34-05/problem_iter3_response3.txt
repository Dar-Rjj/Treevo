```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(15)
    
    # Volume Impact Factor
    avg_volume_25 = df['volume'].rolling(window=25).mean()
    volume_impact_factor = df['volume'] / avg_volume_25
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=25).std()
    
    # Trade Amount Influence
    avg_amount_25 = df['amount'].rolling(window=25).mean()
    trade_amount_influence = df['amount'] / avg_amount_25
    
    # Combine Momentum and Volume Impact
    combined_momentum_volume = momentum / volume_impact_factor
    
    # Adjust for Price Volatility
    adjusted_momentum_volatility = combined_momentum_volume / price_volatility
    
    # Final Adjustment with Trade Amount
    final_factor = adjusted_momentum_volatility * trade_amount_influence
    
    return final_factor
```
