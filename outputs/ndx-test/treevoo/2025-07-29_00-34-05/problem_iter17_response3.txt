import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_momentum = short_ema - long_ema
    
    # Calculate Long-Term Momentum
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum Difference
    momentum_difference = long_ema - long_term_ema
    
    # Calculate Daily Price Range
    daily_price_range = (df['high'] - df['low']) / df['close']
    
    # Compute Volume-Weighted Range
    volume_weighted_range = (daily_price_range / ((df['high'] + df['low']) / 2)) * np.sqrt(df['volume'])
    
    # Short-Term and Long-Term Rolling Average of Volume-Weighted Range
    short_term_volume_avg = volume_weighted_range.rolling(window=10).mean()
    long_term_volume_avg = volume_weighted_range.rolling(window=30).mean()
    
    # Combine Price Momentum and Volume-Weighted Range
    short_term_indicator = long_term_ema * short_term_volume_avg
    long_term_indicator = long_term_ema / long_term_volume_avg
    composite_factor = (short_term_indicator + long_term_indicator) / 2
    
    # Adjust Momentum by Volatility
    daily_range = df['high'] - df['low']
    avg_volatility = daily_range.rolling(window=20).mean()
    
    # Incorporate Volume Influence
    vol_ema_5 = df['volume'].ewm(span=5, adjust=False).mean()
    vol_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_deviation = vol_ema_5 - vol_ema_10
    
    # Combine Momentum, Volatility, and Volume
    adjusted_momentum = short_term_momentum / avg_volatility
    final_alpha_factor = adjusted_momentum * volume_deviation
    
    return final_alpha_factor
