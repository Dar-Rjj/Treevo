```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates an alpha factor based on a combination of price and volume indicators.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values, indexed by date.
    """
    # Calculate the relative position of the close price within the high-low range
    relative_close = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the change in volume compared to the previous day
    volume_change = df['volume'].diff()
    
    # Calculate the weighted moving average of the close price
    wma_close = df['close'].rolling(window=10).apply(lambda x: (x * range(1, len(x) + 1)).sum() / sum(range(1, len(x) + 1)), raw=True)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (relative_close * volume_change * wma_close)
    
    return alpha_factor
```
