```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day EMA of Close
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum Difference
    df['momentum_diff'] = df['EMA_20'] - df['EMA_5']
    
    # Calculate Daily Range and 30-day Average Daily Range for Volatility Adjustment
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range_30'] = df['daily_range'].rolling(window=30).mean()
    
    # Incorporate Volume Influence
    df['avg_vol_30'] = df['volume'].rolling(window=30).mean()
    df['vol_deviation_score'] = ((df['volume'] - df['avg_vol_30']) / df['avg_vol_30']) * 100
    
    # Adjust Momentum by Inverse Volatility
    df['adjusted_momentum'] = df['momentum_diff'] / df['avg_daily_range_30']
    
    # Final Alpha Factor: Adjusted Momentum + Volume Deviation Score
    df['alpha_factor'] = df['adjusted_momentum'] + df['vol_deviation_score']

    return df['alpha_factor']
```
