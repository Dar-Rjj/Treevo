```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(20)
    
    # Volume Impact Factor
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_impact_factor = df['volume'] / avg_volume_30
    
    # Calculate Daily Price Range
    daily_price_range = df['high'] - df['low']
    
    # Calculate Weighted Average Price
    weighted_avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Compute Volume Adjusted Range
    volume_adjusted_range = (daily_price_range / weighted_avg_price.replace(0, 1e-9)) * df['volume']
    
    # Short-Term Rolling Average of Volume Adjusted Range
    short_term_avg_var = volume_adjusted_range.rolling(window=15).mean()
    
    # Long-Term Rolling Average of Volume Adjusted Range
    long_term_avg_var = volume_adjusted_range.rolling(window=30).mean()
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=30).std()
    
    # Trade Amount Influence
    avg_amount_30 = df['amount'].rolling(window=30).mean()
    trade_amount_influence = df['amount'] / avg_amount_30
    
    # Combine Momentum and Volume Impact
    combined_momentum_volume = momentum / volume_impact_factor
    
    # Adjust for Price Volatility
    adjusted_momentum_volume = combined_momentum_volume / price_volatility
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate 7-day EMA of High-Low Range
    alpha = 2 / (7 + 1)
    ema_high_low_range = high_low_range.ewm(alpha=alpha, adjust=False).mean()
    
    # Calculate Daily Change in EMA
    daily_change_ema = ema_high_low_range - ema_high_low_range.shift(1)
    
    # Calculate Volume Shock
    volume_pct_change = df['volume'].pct_change()
    volume_shock = (volume_pct_change > 0.60).astype(int) * 2 - 1  # Flag days with significant volume changes
    
    # Combine Indicators
    combined_indicator = daily_change_ema * volume_shock
    
    # Finalize Alpha Factor
    final_alpha = (
        (short_term_avg_var - long_term_avg_var) 
        * combined_indicator 
        / price_volatility 
        * trade_amount_influence
    )
    
    return final_alpha
```
