```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the following thought tree:
    1. Calculate the difference between the high and low price (price range).
    2. Calculate the percentage change in close price from the previous day.
    3. Compute the weighted sum of the price range and the percentage change, with weights
       assigned according to the volume traded on that day.
    
    Parameters:
    df (pd.DataFrame): DataFrame containing the columns 'open', 'high', 'low', 'close', 'volume'.
    
    Returns:
    pd.Series: A Series containing the calculated alpha factor values.
    """
    # Step 1: Calculate the price range for each day
    price_range = df['high'] - df['low']
    
    # Step 2: Calculate the percentage change in close price from the previous day
    pct_change = df['close'].pct_change()
    
    # Step 3: Compute the weighted sum of the price range and the percentage change
    # Weights are determined by the normalized volume for that day
    factor = (price_range + pct_change) * df['volume']
    
    return factor
```
