    df['Long_Term_Vol_Adjusted_Range_Avg'] = df['Volume_Adjusted_Range'].rolling(window=20).mean()

    df['Short_Term_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Price Momentum (EMA, 20 days)
    df['Long_Term_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute Momentum Difference
    df['Momentum_Difference'] = df['Long_Term_EMA'] - df['Short_Term_EMA']
    
    # Calculate Volume Acceleration
    df['Volume_Acceleration'] = df['volume'].pct_change(periods=5)
    
    # Combine Factors
    alpha_factor = (df['Short_Term_Vol_Adjusted_Range_Avg'] 
                    - df['Long_Term_Vol_Adjusted_Range_Avg'] 
                    + df['Volume_Acceleration'])
    
    return alpha_factor.dropna()
