```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine Intraday Measures
    combined_intraday_measure = (intraday_high_low_spread + close_open_diff) * df['volume']
    
    # Calculate Short-Term Price Momentum
    short_term_price_momentum = df['close'].ewm(span=5).mean() - df['close'].ewm(span=10).mean()
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'].ewm(span=5).mean() - df['volume'].ewm(span=10).mean()
    
    # Final Factor Adjustment
    final_factor = (short_term_price_momentum * volume_momentum) / combined_intraday_measure
    
    return final_factor
```
