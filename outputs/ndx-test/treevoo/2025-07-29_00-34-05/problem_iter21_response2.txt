```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Adjustment
    df['Volume_Impact_Adjustment'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Enhanced Dynamic Price Range
    df['Daily_Enhanced_Dynamic_Price_Range'] = (df['high'] - df['low']) * (df['close'] / df['open'])
    
    # Average Trading Price
    df['Average_Trading_Price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Adjusted Enhanced Range
    df['Volume_Adjusted_Enhanced_Range'] = (df['Daily_Enhanced_Dynamic_Price_Range'] / df['Average_Trading_Price']).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Short-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Short_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range'] = df['Volume_Adjusted_Enhanced_Range'].rolling(window=15).mean()
    
    # Long-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Long_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range'] = df['Volume_Adjusted_Enhanced_Range'].rolling(window=25).mean()
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=30).std()
    
    # Trade Activity Influence
    df['Trade_Activity_Influence'] = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    df['Combined_Momentum_and_Volume_Impact'] = df['Momentum'] * df['Volume_Impact_Adjustment']
    
    # Adjust for Price Volatility
    df['Adjusted_Combined_Momentum_and_Volume_Impact'] = df['Combined_Momentum_and_Volume_Impact'] / df['Price_Volatility']
    
    # High-Low Range Momentum (7-day EMA)
    alpha = 2 / (7 + 1)
    df['High_Low_Range'] = df['high'] - df['low']
    df['High_Low_Range_EMA'] = df['High_Low_Range'].ewm(alpha=alpha, adjust=False).mean()
    df['Daily_Change_in_EMA'] = df['High_Low_Range_EMA'] - df['High_Low_Range_EMA'].shift(1)
    
    # Volume Shock
    df['Volume_Shock'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_shock_threshold = 0.6
    df['Significant_Volume_Shock'] = (df['Volume_Shock'].abs() > volume_shock_threshold).astype(int) * np.sign(df['Volume_Shock'])
    
    # Combine Indicators
    df['Combined_Indicator'] = df['Daily_Change_in_EMA'] * df['Significant_Volume_Shock']
    
    # Finalize Alpha Factor
    df['Alpha_Factor'] = (
        (df['Short_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range'] - df['Long_Term_Rolling_Avg_Volume_Adjusted_Enhanced_Range']) 
        * df['Combined_Indicator']
        * df['Adjusted_Combined_Momentum_and_Volume_Impact']
        * df['Trade_Activity_Influence']
    )
    
    return df['Alpha_Factor']
```
