```python
import pandas as pd

def heuristics_v2(df):
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Short-Term Price Momentum
    df['5_day_ema'] = df['daily_price_change'].ewm(span=5, adjust=False).mean()
    df['10_day_ema'] = df['daily_price_change'].ewm(span=10, adjust=False).mean()
    df['short_term_momentum'] = df['5_day_ema'] - df['10_day_ema']
    
    # Long-Term Price Momentum
    df['20_day_sma'] = df['daily_price_change'].rolling(window=20).mean()
    df['50_day_sma'] = df['daily_price_change'].rolling(window=50).mean()
    df['long_term_momentum'] = df['20_day_sma'] - df['50_day_sma']
    
    # Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Volume Indicator
    df['positive_volume_days'] = (df['volume_change'] > 0).rolling(window=20).sum()
    df['negative_volume_days'] = (df['volume_change'] < 0).rolling(window=20).sum()
    
    # Combine Short-Term Price Momentum and Volume Indicator
    df['short_term_factor'] = df['short_term_momentum'] * df['positive_volume_days']
    
    # Combine Long-Term Price Momentum and Volume Indicator
    df['long_term_factor'] = df['long_term_momentum'] * df['negative_volume_days']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['short_term_factor'] - df['long_term_factor']
    
    return df['alpha_factor']
```
