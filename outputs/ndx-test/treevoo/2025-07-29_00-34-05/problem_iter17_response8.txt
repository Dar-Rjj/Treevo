def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()

    # Calculate the true range
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)

    # Calculate the average true range over a 14-day period
    df['avg_true_range_14'] = df['true_range'].rolling(window=14).mean()

    # Calculate the mean of the close price over a 14-day period
    df['avg_close_14'] = df['close'].rolling(window=14).mean()

    # Alpha factor: (Average True Range / Average Close Price) * Daily Return
    alpha_factor = (df['avg_true_range_14'] / df['avg_close_14']) * df['daily_return']

    return alpha_factor.dropna()
