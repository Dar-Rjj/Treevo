```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    df['Momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Impact Adjustment
    df['Volume_Impact'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Daily Enhanced Dynamic Price Range
    df['Daily_Enhanced_Range'] = (df['high'] - df['low']) * (df['close'] / df['open'])
    
    # Average Trading Price
    df['Average_Trading_Price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Adjusted Enhanced Range
    df['Volume_Adjusted_Range'] = (df['Daily_Enhanced_Range'] / df['Average_Trading_Price']).replace([np.inf, -np.inf], 0) * df['volume']
    
    # Short-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Short_Term_Avg'] = df['Volume_Adjusted_Range'].rolling(window=15).sum() / 15
    
    # Long-Term Rolling Avg of Volume Adjusted Enhanced Range
    df['Long_Term_Avg'] = df['Volume_Adjusted_Range'].rolling(window=25).sum() / 25
    
    # Price Volatility
    df['Price_Volatility'] = df['close'].rolling(window=30).std()
    
    # Trade Activity Influence
    df['Trade_Activity'] = df['amount'] / df['amount'].rolling(window=30).mean()
    
    # Combine Momentum and Volume Impact
    df['Combined_Momentum'] = df['Momentum'] * df['Volume_Impact']
    
    # Adjust for Price Volatility
    df['Adjusted_Momentum'] = df['Combined_Momentum'] / df['Price_Volatility']
    
    # High-Low Range Momentum
    alpha = 2 / (7 + 1)
    df['EMA_High_Low'] = df['high'] - df['low']
    df['EMA_High_Low'] = df['EMA_High_Low'].ewm(alpha=alpha, adjust=False).mean()
    df['High_Low_Momentum'] = df['EMA_High_Low'] - df['EMA_High_Low'].shift(1)
    
    # Volume Shock
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Shock'] = (df['Volume_Change'] > 0.6).astype(int) * 2 - 1  # 1 for significant increase, -1 for significant decrease
    
    # Combine Indicators
    df['Combined_Indicator'] = df['High_Low_Momentum'] * df['Volume_Shock']
    
    # Finalize Alpha Factor
    df['Alpha_Factor'] = (df['Short_Term_Avg'] - df['Long_Term_Avg']) * df['Combined_Indicator'] / df['Price_Volatility'] * df['Trade_Activity']
    
    return df['Alpha_Factor']
```
