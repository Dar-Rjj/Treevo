```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term EMA (5 days)
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    
    # Long-Term EMA (20 days)
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Momentum Difference
    momentum_diff = short_term_ema - long_term_ema
    
    # Volume Acceleration (5 days percentage change)
    volume_acceleration = df['volume'].pct_change(periods=5)
    
    # Relative Strength Index (RSI) (14 days)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine Momentum, Volume, and RSI
    combined_factor = momentum_diff * volume_acceleration + rsi
    
    # Volume Impact Factor (Average Volume over last 25 days)
    avg_volume = df['volume'].rolling(window=25).mean()
    volume_impact = df['volume'] / avg_volume
    
    # Adjust for Volume Impact
    adjusted_factor = combined_factor / volume_impact
    
    # Price Volatility (Standard Deviation of Close prices over last 25 days)
    price_volatility = df['close'].rolling(window=25).std()
    
    # Further Adjust for Price Volatility
    further_adjusted_factor = adjusted_factor / price_volatility
    
    # Trade Amount Influence (Current Amount / Average Amount over last 25 days)
    avg_amount = df['amount'].rolling(window=25).mean()
    trade_amount_influence = df['amount'] / avg_amount
    
    # Final Adjustment with Trade Amount
    final_factor = further_adjusted_factor * trade_amount_influence
    
    return final_factor
```
