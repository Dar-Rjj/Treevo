```python
def heuristics_v2(data):
    import pandas as pd

    # Calculate Daily Price Range
    daily_price_range = data['high'] - data['low']
    
    # Compute Volume Adjusted Range
    avg_price = (data['high'] + data['low'] + data['close']) / 3
    volume_adjusted_range = (daily_price_range / avg_price).replace([float('inf'), -float('inf')], 0) * data['volume']
    
    # Calculate Short-Term (10 days) Rolling Average of Volume Adjusted Range
    short_term_avg = volume_adjusted_range.rolling(window=10).mean()
    
    # Calculate Long-Term (20 days) Rolling Average of Volume Adjusted Range
    long_term_avg = volume_adjusted_range.rolling(window=20).mean()
    
    # The factor is the difference between short-term and long-term averages
    factor = short_term_avg - long_term_avg
    
    return factor
```
