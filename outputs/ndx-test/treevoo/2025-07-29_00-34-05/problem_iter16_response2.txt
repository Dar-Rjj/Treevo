```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day EMA of Close
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate Momentum Difference
    df['momentum_diff'] = df['ema_20'] - df['ema_5']
    
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 30-day Average Daily Range
    df['avg_daily_range'] = df['daily_range'].rolling(window=30).mean()
    
    # Calculate 30-day Average Volume
    df['avg_vol_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate Volume Deviation Score
    df['vol_dev_score'] = (df['volume'] - df['avg_vol_30']) / df['avg_vol_30'] * 100
    
    # Adjusted Momentum
    df['adjusted_momentum'] = df['momentum_diff'] / df['avg_daily_range']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['adjusted_momentum'] * df['vol_dev_score']
    
    return df['final_alpha_factor']
```
