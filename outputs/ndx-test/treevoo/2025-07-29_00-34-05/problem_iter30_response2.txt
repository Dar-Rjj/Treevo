```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates an alpha factor based on a combination of price and volume changes.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, volume).
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the percentage change in closing price
    close_change = df['close'].pct_change()
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Create a factor based on the interaction of close price change and volume change
    factor = (close_change * volume_change) / (df['high'] - df['low'])
    
    return factor
```
