```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'].pct_change(30)
    
    # Calculate 30-Day Average Range
    daily_range = df['high'] - df['low']
    average_range_30 = daily_range.rolling(window=30).mean()
    
    # Calculate 50-Day Moving Average
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate 200-Day Moving Average
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate Moving Average Crossover Signal
    crossover_signal = (ma_50 > ma_200).astype(int)
    
    # Adjust Momentum
    adjusted_momentum = momentum * (1 / average_range_30)
    
    # Final Alpha Factor
    alpha_factor = adjusted_momentum + crossover_signal
    
    return alpha_factor
```
