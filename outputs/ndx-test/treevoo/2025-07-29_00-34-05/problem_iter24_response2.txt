```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum
    momentum = df['close'] - df['close'].shift(20)
    
    # Volume Impact Adjustment
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_impact = df['volume'] / avg_volume_30
    
    # Daily Enhanced Dynamic Price Range
    dynamic_price_range = (df['high'] - df['low']) * (df['close'] / df['open'])
    
    # Average Trading Price
    avg_trading_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Adjusted Enhanced Range
    volume_adjusted_range = (dynamic_price_range / avg_trading_price) * df['volume']
    
    # Short-Term Rolling Avg of Volume Adjusted Enhanced Range
    short_term_avg = volume_adjusted_range.rolling(window=15).mean()
    
    # Long-Term Rolling Avg of Volume Adjusted Enhanced Range
    long_term_avg = volume_adjusted_range.rolling(window=25).mean()
    
    # Price Volatility
    price_volatility = df['close'].rolling(window=30).std()
    
    # Trade Activity Influence
    avg_amount_30 = df['amount'].rolling(window=30).mean()
    trade_activity_influence = df['amount'] / avg_amount_30
    
    # Combine Momentum and Volume Impact
    combined_momentum_volume = momentum * volume_impact
    
    # Adjust for Price Volatility
    adjusted_momentum_volume = combined_momentum_volume / price_volatility
    
    # High-Low Range Momentum
    ema_high_low_range = df['high'] - df['low']
    ema_high_low_range = ema_high_low_range.ewm(span=7, adjust=False).mean()
    daily_change_ema_high_low = ema_high_low_range.diff()
    
    # Volume Shock
    volume_percentage_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    significant_volume_change = (volume_percentage_change > 0.60).astype(int)
    
    # High-Frequency Price Momentum
    ema_close_prices = df['close'].ewm(span=5, adjust=False).mean()
    daily_change_ema_close = ema_close_prices.diff()
    
    # Combine Indicators
    high_low_range_momentum = daily_change_ema_high_low * significant_volume_change
    combined_indicator = high_low_range_momentum * daily_change_ema_close
    
    # Finalize Alpha Factor
    alpha_factor = (short_term_avg - long_term_avg) * combined_indicator * adjusted_momentum_volume * trade_activity_influence
    
    return alpha_factor
```
