```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_day_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4 * df['volume']
    
    # Combine Intraday Momentum and VWAP
    combined_value = (vwap - intraday_high_low_spread) * df['volume']
    
    # Apply Exponential Moving Average (EMA) over the last 5 days
    ema_combined_value = combined_value.ewm(span=5).mean()
    
    return ema_combined_value
```
