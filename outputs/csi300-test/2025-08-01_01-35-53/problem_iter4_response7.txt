```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum = vwap - intraday_high_low_spread
    
    # Weight by Intraday Volume
    volume_weighted_momentum = combined_momentum * df['volume']
    
    # Incorporate Previous Day's Close-to-Open Return
    factor = volume_weighted_momentum + prev_close_to_open_return
    
    # Smooth the Factor using EMA over the last 5 days
    smoothed_factor = factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
