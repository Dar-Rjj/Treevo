import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = high_low_spread + open_to_close_return
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_volume = df['volume']
    vwap = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    vwap = vwap.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_factor = vwap - high_low_spread
    combined_factor = combined_factor * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA) over the last 7 days
    smoothed_factor = combined_factor.ewm(span=7).mean()
    
    return smoothed_factor
