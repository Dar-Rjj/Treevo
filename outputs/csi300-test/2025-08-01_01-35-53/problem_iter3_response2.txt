```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = (intraday_high_low_spread + close_to_open_return) / 2
    
    # Compute Daily VWAP
    price_sum = (df['high'] + df['low'] + df['close']) / 3
    vwap = (price_sum * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    
    # Weight by Intraday Volume
    volume_weighted_factor = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    smoothed_factor = volume_weighted_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
