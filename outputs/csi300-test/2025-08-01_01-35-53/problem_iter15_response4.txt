import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_today_close_return'] = df['close'].shift(-1) - df['open']

    # Calculate Volume-Weighted Average Price (VWAP)
    df['daily_vwap'] = ((df['high'] + df['low'] + df['close'] + df['open']) / 4 * df['volume']).cumsum() / df['volume'].cumsum()

    # Integrate Intraday and VWAP Metrics
    df['integrated_metric'] = df['daily_vwap'] - df['intraday_high_low_spread']
    df['integrated_metric_weighted_by_volume'] = df['integrated_metric'] * df['volume']

    # Combine Short and Long-Term EMAs
    df['short_term_ema'] = df['integrated_metric_weighted_by_volume'].ewm(span=5, adjust=False).mean()
    df['long_term_ema'] = df['integrated_metric_weighted_by_volume'].ewm(span=20, adjust=False).mean()

    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['long_term_ema'] - df['short_term_ema']
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=7, adjust=False).mean()

    return df['final_alpha_factor']
