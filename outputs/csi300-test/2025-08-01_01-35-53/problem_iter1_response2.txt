```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Combine signals into the alpha factor
    alpha_factor = (high_low_spread * df['volume'] - close_open_spread) / daily_range
    
    return alpha_factor
```
