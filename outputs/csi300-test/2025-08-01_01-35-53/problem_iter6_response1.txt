```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close = df['close'].shift(1)
    close_to_open_return = df['open'] - prev_close
    
    # Calculate Intraday Momentum
    intraday_momentum = high_low_spread + close_to_open_return
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    total_volume = df['volume']
    vwap = (typical_price * df['volume']).cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - high_low_spread
    volume_weighted_value = combined_value * df['volume']
    
    # Smooth the Factor
    ema_smoothed = volume_weighted_value.ewm(span=3).mean()
    
    return ema_smoothed
```
