```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['Prev_Open_to_Close_Return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Price_Sum'] = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    df['Total_Volume'] = df['volume'].rolling(window=1).sum()
    df['VWAP'] = df['Price_Sum'].rolling(window=1).sum() / df['Total_Volume']
    
    # Combine Intraday Momentum and VWAP
    df['Combined_Value'] = df['VWAP'] - df['Intraday_High_Low_Spread']
    df['Weighted_Combined_Value'] = df['Combined_Value'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over 7 days
    df['Alpha_Factor'] = df['Weighted_Combined_Value'].ewm(span=7, adjust=False).mean()
    
    # Return the Alpha Factor as a pandas Series
    return df['Alpha_Factor']
```
