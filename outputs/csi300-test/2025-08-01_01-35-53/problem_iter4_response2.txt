```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['open'] / df['close'].shift(1)) - 1
    
    # Compute Daily VWAP
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    
    # Compute Average Volume over Last 5 Days and Last 20 Days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Ratio
    df['volume_ratio'] = df['avg_volume_5d'] / df['avg_volume_20d']
    
    # Compute 10-Day Standard Deviation of Log Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['short_term_volatility'] = df['log_returns'].rolling(window=10).std()
    
    # Combine Intraday Momentum, VWAP, and Volume Trend
    df['intraday_momentum'] = df['vwap'] - df['intraday_high_low_spread']
    df['combined_factor'] = df['intraday_momentum'] * df['volume_ratio']
    df['adjusted_factor'] = df['combined_factor'] * df['volume'] - df['short_term_volatility']
    
    # Ensure the final alpha factor is interpretable and robust
    alpha_factor = df['adjusted_factor'].dropna()
    
    return alpha_factor
```
