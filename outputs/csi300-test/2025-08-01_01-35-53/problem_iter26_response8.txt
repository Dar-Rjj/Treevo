import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = (vwap - intraday_high_low_spread) * df['volume']
    
    # Apply Exponential Moving Average (EMA) Over the Last 3 Days
    ema_3d = combined_momentum_vwap.ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = ema_3d
    
    return alpha_factor
