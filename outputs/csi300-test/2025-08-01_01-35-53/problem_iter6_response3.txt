import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily VWAP
    df['Price'] = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    df['Daily_VWAP'] = (df['Price'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Apply Exponential Moving Average (EMA) over the last 3 days for Refined VWAP
    df['Refined_VWAP'] = df['Daily_VWAP'].ewm(span=3, adjust=False).mean()

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # Compute Previous Day's Close-to-Open Return
    df['Prev_Day_Close_to_Open_Return'] = df['open'] - df['close'].shift(1)

    # Compute Volume Trend
    df['Volume_Trend'] = df['volume'] - df['volume'].shift(1)

    # Compute Price Change Momentum
    df['Recent_Price_Changes'] = df['close'] - df['close'].shift(1)
    df['Price_Change_Momentum'] = df['Recent_Price_Changes'].ewm(span=3, adjust=False).mean()

    # Combine Intraday Momentum, VWAP, and Sentiment
    df['Combined_Value'] = df['Refined_VWAP'] - df['Intraday_High_Low_Spread']
    df['Weighted_Combined_Value'] = df['Combined_Value'] * df['volume']

    # Final Factor Smoothing
    df['Final_Factor'] = df['Weighted_Combined_Value'].ewm(span=3, adjust=False).mean()

    return df['Final_Factor']
