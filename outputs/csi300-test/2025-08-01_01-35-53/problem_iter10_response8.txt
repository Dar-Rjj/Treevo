import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['price_sum'] = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume'].rolling(window=1).sum()
    df['vwap'] = df['price_sum'].rolling(window=1).sum() / total_volume
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum'] = df['intraday_high_low_spread'] + df['prev_day_open_to_close_return']
    df['weighted_intraday_momentum'] = (df['vwap'] - df['intraday_momentum']) * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    df['alpha_factor'] = df['weighted_intraday_momentum'].ewm(span=7, adjust=False).mean()
    
    return df['alpha_factor']
