import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_open_to_close_return'] = df['close'].shift(-1) - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    df['vwap'] = (price_sum * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['volume_weighted_combined'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    df['alpha_factor'] = df['volume_weighted_combined'].ewm(span=7).mean()
    
    return df['alpha_factor']
