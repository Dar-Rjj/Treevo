```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Daily VWAP
    prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (prices * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['volume_weighted_combined_value'] = df['combined_value'] * df['volume']
    
    # Incorporate Price Dynamics
    df['price_dynamics'] = df['prev_close_to_open_return']
    df['factor'] = df['volume_weighted_combined_value'] + df['price_dynamics']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    df['factor_ema'] = df['factor'].ewm(span=7, adjust=False).mean()
    
    return df['factor_ema']
```
