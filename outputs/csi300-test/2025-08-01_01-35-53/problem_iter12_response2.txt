```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume'])
    vwap = prices / df['volume']
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    weighted_by_volume = combined_value * df['volume']
    
    # Integrate Additional Momentum Indicator: 2-Day Price Change
    two_day_price_change = df['close'] - df['close'].shift(2)
    
    # Incorporate Longer-Term Price Trends: 7-Day Price Change
    seven_day_price_change = df['close'] - df['close'].shift(7)
    
    # Combine with Additional Momentum Indicators
    combined_factor = weighted_by_volume + two_day_price_change + seven_day_price_change
    
    # Smooth the Factor: Apply Exponential Moving Average (EMA) over the last 7 days
    smoothed_factor = combined_factor.ewm(span=7).mean()
    
    return smoothed_factor
```
