```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['High'] - df['Low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_open_return'] = df['Close'].shift(1) - df['Open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['Open'] * df['Volume'] + df['High'] * df['Volume'] + 
                  df['Low'] * df['Volume'] + df['Close'] * df['Volume']) / (4 * df['Volume'])
    
    # Combine Intraday Momentum and VWAP
    df['combined_momentum_vwap'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_momentum_vwap'] = df['combined_momentum_vwap'] * df['Volume']
    
    # Incorporate Longer-Term Trends
    df['3_day_momentum'] = df['Close'] - df['Close'].shift(3)
    df['5_day_momentum'] = df['Close'] - df['Close'].shift(5)
    
    # Refine Volume Weighting
    df['3_day_avg_volume'] = df['Volume'].rolling(window=3).mean()
    df['5_day_avg_volume'] = df['Volume'].rolling(window=5).mean()
    
    # Smooth the Factor
    df['factor_ema_5'] = df['weighted_combined_momentum_vwap'].ewm(span=5, adjust=False).mean()
    df['factor_ema_10'] = df['weighted_combined_momentum_vwap'].ewm(span=10, adjust=False).mean()
    
    # Final alpha factor
    df['alpha_factor'] = df['factor_ema_5'] + df['factor_ema_10'] + df['3_day_momentum'] + df['5_day_momentum']
    
    return df['alpha_factor']
```
