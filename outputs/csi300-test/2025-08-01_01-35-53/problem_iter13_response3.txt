```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Combine Intraday Momentum and VWAP
    combined_value = (vwap - intraday_high_low_spread) * df['volume']
    
    # Apply Exponential Moving Averages (EMAs)
    ema_5d = combined_value.ewm(span=5, adjust=False).mean()
    ema_20d = combined_value.ewm(span=20, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = ema_5d - ema_20d
    
    return alpha_factor
```
