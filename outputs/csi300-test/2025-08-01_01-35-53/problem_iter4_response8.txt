```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    df['intraday_momentum'] = df['intraday_high_low_spread'] + df['prev_close_to_open_return']

    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap'] = df['daily_vwap'].cumsum() / df['volume'].cumsum()

    # Integrate Liquidity
    df['dollar_volume'] = df['close'] * df['volume']
    df['smoothed_dollar_volume'] = df['dollar_volume'].ewm(span=20).mean()

    # Combine Intraday Momentum, VWAP, and Liquidity
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_value'] = df['combined_value'] * df['volume'] * df['smoothed_dollar_volume']

    # Smooth the Factor
    df['factor'] = df['weighted_combined_value'].ewm(span=5).mean()

    return df['factor']
```
