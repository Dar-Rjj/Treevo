import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_open_to_close_return'] = df['close'] - df['open'].shift(1)
    df['intraday_momentum'] = df['intraday_high_low_spread'] + df['prev_open_to_close_return']

    # Calculate Volume Weighted Average Price (VWAP)
    total_volume = df['volume']
    df['vwap'] = (df[['high', 'low', 'close', 'open']].sum(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()

    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_value'] = df['combined_value'] * df['volume']

    # Smooth the Factor
    df['smoothed_factor'] = df['weighted_combined_value'].ewm(span=7, adjust=False).mean()
    
    return df['smoothed_factor']
