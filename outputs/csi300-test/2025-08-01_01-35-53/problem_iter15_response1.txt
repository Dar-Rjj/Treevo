```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open'] = df['open'].shift(1)
    df['open_to_close_return'] = df['close'] - df['prev_day_open']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_high_low_spread'] + df['open_to_close_return']
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    df['vwap'] = prices_sum.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    
    # Weight by Intraday Volume
    df['weighted_value'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 14 days
    df['smoothed_factor'] = df['weighted_value'].ewm(span=14, adjust=False).mean()
    
    return df['smoothed_factor']
```
