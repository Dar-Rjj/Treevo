import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_close_to_open_return = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = (intraday_high_low_spread + previous_close_to_open_return) / 2
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices = (df[['open', 'high', 'low', 'close']].sum(axis=1) / 4)
    vwap = (prices * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Integrate Intraday Momentum with VWAP
    integrated_factor = vwap - intraday_high_low_spread
    volume_weighted_factor = integrated_factor * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 3 days
    smoothed_factor = volume_weighted_factor.ewm(span=3, adjust=False).mean()
    
    return smoothed_factor
