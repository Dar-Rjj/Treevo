import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_close_to_open_return'] = df['close'].shift(1) - df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()

    # Combine Intraday Momentum and VWAP
    df['combined_momentum_vwap'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_momentum_vwap'] = df['combined_momentum_vwap'] * df['volume']

    # Smooth the Factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['ema_period'] = np.where(df['volume_change'] > 0, 5, 10)
    df['smoothed_factor'] = df.groupby('ema_period')['weighted_combined_momentum_vwap'].transform(lambda x: x.ewm(span=x.name).mean())

    return df['smoothed_factor']
