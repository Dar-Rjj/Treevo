```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    df['vwap'] = price_sum.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_factor'] = df['vwap'] - df['intraday_high_low_spread'] + df['prev_close_to_open_return']
    
    # Weight by Intraday Volume
    df['weighted_factor'] = df['combined_factor'] * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA) over the last 3 days
    df['smoothed_factor'] = df['weighted_factor'].ewm(span=3).mean()
    
    return df['smoothed_factor']
```
