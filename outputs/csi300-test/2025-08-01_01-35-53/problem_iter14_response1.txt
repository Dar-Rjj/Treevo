import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_intraday_momentum_vwap'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Integrate Short-Term Momentum
    df['2_day_price_change'] = df['close'] - df['close'].shift(2)
    
    # Incorporate Longer-Term Momentum
    df['7_day_price_change'] = df['close'] - df['close'].shift(7)
    
    # Combine with Momentum Indicators
    df['combined_factor'] = (df['combined_intraday_momentum_vwap'] + 
                             df['2_day_price_change'] + 
                             df['7_day_price_change'])
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    df['smoothed_factor'] = df['combined_factor'].ewm(span=7).mean()
    
    # Return the alpha factor
    return df['smoothed_factor'].dropna()
