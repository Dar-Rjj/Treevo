```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'].shift(-1) - df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    df['price_sum'] = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    df['vwap'] = (df['price_sum'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_value'] = df['combined_value'] * df['volume']

    # Incorporate Trading Volume Trends
    df['5_day_vol_ma'] = df['volume'].rolling(window=5).mean()
    df['volume_trend'] = df['volume'] / df['5_day_vol_ma']

    # Final alpha factor
    df['alpha_factor'] = df['weighted_combined_value'] * df['volume_trend']

    # Smooth the Factor using Exponential Moving Average (EMA)
    df['smoothed_alpha_factor'] = df['alpha_factor'].ewm(span=7, adjust=False).mean()

    return df['smoothed_alpha_factor']
```
