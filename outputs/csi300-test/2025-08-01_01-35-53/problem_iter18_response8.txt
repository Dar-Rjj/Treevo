import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_day_open_to_close_return = df['close'].shift(-1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_volume = (df['open'] * df['volume'] + df['high'] * df['volume'] + 
                    df['low'] * df['volume'] + df['close'] * df['volume']) / 4
    vwap = (df['open'] * df['volume'] + df['high'] * df['volume'] + 
            df['low'] * df['volume'] + df['close'] * df['volume']) / total_volume
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    weighted_value = combined_value * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA)
    ema_7d = weighted_value.ewm(span=7, adjust=False).mean()
    
    return ema_7d
