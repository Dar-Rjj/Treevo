```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()
    
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_close_to_open_return = df['open'] - df['close'].shift(1)
    
    # Compute Enhanced Intraday Momentum
    enhanced_intraday_momentum = (intraday_high_low_spread + previous_close_to_open_return).fillna(0)
    
    # Compute Volume-Weighted Average Price (VWAP) with More Price Points
    vwap = ((df['high'] + df['low'] + df['close'] + df['open']) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Integrate Sentiment and Volume Trends
    # For simplicity, we'll use a combination of recent price changes and volume trends
    # Assuming 'sentiment' is not available, we'll skip this part
    recent_price_changes = (df['close'] - df['close'].shift(1)).fillna(0)
    volume_trends = (df['volume'] - df['volume'].shift(1)).fillna(0)
    
    # Dynamic Weighting to Intraday Prices
    dynamic_weighting = (recent_price_changes + volume_trends).fillna(0)
    
    # Combine Intraday Momentum and VWAP
    combined_factor = vwap - (intraday_high_low_spread * dynamic_weighting)
    
    # Weight by Intraday Volume
    weighted_factor = combined_factor * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 3 days
    smooth_factor = weighted_factor.ewm(span=3, adjust=False).mean()
    
    return smooth_factor
```
