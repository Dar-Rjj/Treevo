```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['prev_open_to_close_return'] / 
                               df['intraday_high_low_spread'])
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] * df['volume'] + df['low'] * df['volume'] + 
                   df['close'] * df['volume'] + df['open'] * df['volume']) /
                  (4 * df['volume']))
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    
    # Weight by Intraday Volume
    df['weighted_value'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 14 days
    df['alpha_factor'] = df['weighted_value'].ewm(span=14, adjust=False).mean()
    
    return df['alpha_factor']
```
