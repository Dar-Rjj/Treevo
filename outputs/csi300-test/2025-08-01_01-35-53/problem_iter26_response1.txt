import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['High'] - df['Low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['Close'].shift(1) - df['Open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['Open'] + df['High'] + df['Low'] + df['Close']) / 4
    
    # Combine Intraday Momentum and VWAP
    intraday_momentum_vwap = (vwap - intraday_high_low_spread) * df['Volume']
    
    # Apply Exponential Moving Average (EMA) over the Last 3 Days
    ema_3d = intraday_momentum_vwap.ewm(span=3, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = ema_3d
    
    return alpha_factor
