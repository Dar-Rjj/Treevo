```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close', 'open']].sum(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_value'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor
    df['short_term_ema'] = df['weighted_combined_value'].ewm(span=7, adjust=False).mean()
    df['long_term_ema'] = df['weighted_combined_value'].ewm(span=28, adjust=False).mean()
    
    # Compute Recent Volatility
    df['recent_volatility'] = df['close'].rolling(window=14).std()
    df['vol_normalized'] = df['recent_volatility'] / 10
    
    # Adjust Weights
    short_weight = 0.7 + 0.3 * (1 - df['vol_normalized'])
    long_weight = 0.3 + 0.3 * df['vol_normalized']
    
    # Apply Weights
    df['final_ema'] = (df['short_term_ema'] * short_weight) + (df['long_term_ema'] * long_weight)
    
    # Finalize the Alpha Factor
    alpha_factor = df['final_ema']
    
    return alpha_factor
```
