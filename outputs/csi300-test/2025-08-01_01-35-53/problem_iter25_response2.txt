import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Prev_Open_to_Close_Return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['VWAP'] = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['Intraday_Momentum_Factor'] = df['VWAP'] - df['High_Low_Spread']
    df['Volume_Weighted_Factor'] = df['Intraday_Momentum_Factor'] * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA)
    df['Smoothed_Factor'] = df['Volume_Weighted_Factor'].ewm(span=7, adjust=False).mean()
    
    return df['Smoothed_Factor']
