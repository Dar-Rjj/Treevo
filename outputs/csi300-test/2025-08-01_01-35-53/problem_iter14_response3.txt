```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum = vwap - intraday_high_low_spread
    volume_weighted_combined_momentum = combined_momentum * df['volume']
    
    # Integrate Additional Momentum Indicators
    two_day_price_change = df['close'] - df['close'].shift(2)
    seven_day_price_change = df['close'] - df['close'].shift(7)
    
    # Incorporate Longer-Term Price Trends
    thirty_day_price_trend = df['close'] - df['close'].shift(30)
    
    # Combine with Additional Momentum Indicators and Longer-Term Trends
    alpha_factor = (volume_weighted_combined_momentum +
                    two_day_price_change + 
                    seven_day_price_change + 
                    thirty_day_price_trend)
    
    # Smooth the Factor
    alpha_factor_smoothed = alpha_factor.ewm(span=7, adjust=False).mean()
    
    return alpha_factor_smoothed
```
