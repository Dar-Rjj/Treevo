```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_close = df['close'].shift(1)
    open_return = (df['open'] - previous_close) / previous_close
    
    # Compute Intraday Momentum
    intraday_momentum = (intraday_high_low_spread + open_return) / 2
    
    # Compute Daily VWAP
    daily_vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    factor = (daily_vwap - intraday_high_low_spread) * df['volume']
    
    return factor
```
