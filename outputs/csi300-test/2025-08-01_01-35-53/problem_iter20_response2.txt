```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'].shift(-1) - df['open'].shift(-1)
    
    # Compute Daily VWAP
    df['price_sum'] = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume'].rolling(window=1, min_periods=1).sum()
    df['vwap'] = df['price_sum'].rolling(window=1, min_periods=1).sum() / total_volume
    
    # Combine Dynamic Intraday Momentum and Precise VWAP
    df['combined_value'] = (df['vwap'] - df['high_low_spread']) + df['prev_day_open_to_close_return']
    df['weighted_combined_value'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor for Responsiveness using EMA
    df['alpha_factor'] = df['weighted_combined_value'].ewm(span=7, adjust=False).mean()
    
    return df['alpha_factor']
```
