import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum'] = df['vwap'] - df['intraday_high_low_spread']
    df['volume_weighted_factor'] = df['intraday_momentum'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 5 days
    df['factor_ema'] = df['volume_weighted_factor'].ewm(span=5, adjust=False).mean()
    
    return df['factor_ema']
