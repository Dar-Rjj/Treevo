import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = (intraday_high_low_spread + prev_close_to_open_return) / 2
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['high'] * df['volume'] + df['low'] * df['volume'] + 
            df['close'] * df['volume'] + df['open'] * df['volume']) / (4 * df['volume'])
    
    # Combine Intraday Momentum and VWAP, weighted by Intraday Volume
    combined_value = (vwap - intraday_high_low_spread) * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 3 days
    smoothed_factor = combined_value.ewm(span=3, adjust=False).mean()
    
    return smoothed_factor
