```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df[['high', 'low', 'close', 'open']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    weighted_value = combined_value * df['volume']
    
    # Apply Exponential Moving Average (EMA) for smoothing
    ema_7 = weighted_value.ewm(span=7, adjust=False).mean()
    
    # Adjust smoothing based on volume
    adjusted_ema = np.where(df['volume'] > df['volume'].shift(1),
                            weighted_value.ewm(span=5, adjust=False).mean(),
                            weighted_value.ewm(span=14, adjust=False).mean())
    
    # Final factor value
    factor_value = pd.Series(adjusted_ema, index=df.index, name='alpha_factor')
    
    return factor_value
```
