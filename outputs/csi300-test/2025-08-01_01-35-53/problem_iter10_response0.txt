```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_day_close = df['close'].shift(1)
    close_to_open_return = df['open'] - previous_day_close
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (daily_prices * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    weighted_combined_value = combined_value * df['volume']
    
    # Incorporate Price Dynamics
    factor = weighted_combined_value + close_to_open_return
    
    # Smooth the Factor with Exponential Moving Average (EMA)
    ema_factor = factor.ewm(span=7, adjust=False).mean()
    
    return ema_factor
```
