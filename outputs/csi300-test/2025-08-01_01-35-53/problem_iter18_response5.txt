```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['price_sum'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (df['price_sum'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum_vwap'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Incorporate Long-Term Momentum
    df['20_day_momentum'] = df['close'] - df['close'].shift(20)
    
    # Incorporate Price Dynamics
    df['price_dynamics'] = df['prev_close_to_open_return']
    
    # Enhance with Long-Term Price Trend
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    
    # Integrate Short-Term Volatility
    df['true_range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['14_day_atr'] = df['true_range'].rolling(window=14).mean()
    
    # Final Factor Combination
    df['alpha_factor'] = (
        df['intraday_momentum_vwap'] +
        df['20_day_momentum'] +
        df['price_dynamics'] +
        df['50_day_ma'] +
        df['14_day_atr']
    )
    
    # Smooth the Factor
    df['smoothed_alpha_factor'] = df['alpha_factor'].ewm(span=20, adjust=False).mean()
    
    return df['smoothed_alpha_factor']
```
