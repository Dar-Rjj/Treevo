import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Compute Volume Weighted Average Price (VWAP)
    vwap_open = (df['open'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_high = (df['high'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_low = (df['low'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_close = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_vwap_open = vwap_open - high_low_spread
    combined_vwap_high = vwap_high - high_low_spread
    combined_vwap_low = vwap_low - high_low_spread
    combined_vwap_close = vwap_close - high_low_spread
    
    # Weight by Intraday Volume
    weighted_vwap_open = combined_vwap_open * df['volume']
    weighted_vwap_high = combined_vwap_high * df['volume']
    weighted_vwap_low = combined_vwap_low * df['volume']
    weighted_vwap_close = combined_vwap_close * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) Over 5 Days
    ema_weighted_vwap_open = weighted_vwap_open.ewm(span=5).mean()
    ema_weighted_vwap_high = weighted_vwap_high.ewm(span=5).mean()
    ema_weighted_vwap_low = weighted_vwap_low.ewm(span=5).mean()
    ema_weighted_vwap_close = weighted_vwap_close.ewm(span=5).mean()
    
    # Combine the smoothed VWAPs
    alpha_factor = (ema_weighted_vwap_open + ema_weighted_vwap_high + ema_weighted_vwap_low + ema_weighted_vwap_close) / 4
    
    return alpha_factor
