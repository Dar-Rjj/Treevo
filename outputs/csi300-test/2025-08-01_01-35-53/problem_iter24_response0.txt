import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday High-Low Spread and VWAP
    combined_value = vwap - intraday_high_low_spread
    
    # Weight by Intraday Volume
    weighted_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 14 days
    smoothed_factor = weighted_value.ewm(span=14, adjust=False).mean()
    
    return smoothed_factor
