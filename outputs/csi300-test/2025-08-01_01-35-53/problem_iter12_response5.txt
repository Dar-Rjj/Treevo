```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'].shift(-1) - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = ((df['high'] + df['low'] + df['close'] + df['open']) / 4 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['daily_vwap'] - df['intraday_high_low_spread']
    df['volume_weighted_factor'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    df['smoothed_factor'] = df['volume_weighted_factor'].ewm(span=10, adjust=False).mean()
    
    return df['smoothed_factor']
```
