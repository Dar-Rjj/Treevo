```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['average_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['daily_vwap'] = (df['average_price'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_intraday_momentum_vwap'] = df['daily_vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_value'] = df['combined_intraday_momentum_vwap'] * df['volume']
    
    # Incorporate Longer-Term Momentum
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    # Incorporate Price Dynamics
    df['price_dynamics'] = df['weighted_combined_value'] + df['prev_close_to_open_return']
    
    # Add Longer-Term Momentum
    df['factor'] = df['price_dynamics'] + df['10_day_momentum']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 10 days
    df['factor_ema'] = df['factor'].ewm(span=10, adjust=False).mean()
    
    return df['factor_ema']
```
