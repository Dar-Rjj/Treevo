```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['High_Low_Spread'] = df['High'] - df['Low']
    
    # Compute Previous Day's Close-to-Open Return
    df['Prev_Close_Open_Return'] = (df['Close'].shift(1) - df['Open']) / df['Open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Average_Price'] = (df['Open'] + df['High'] + df['Low'] + df['Close']) / 4
    df['VWAP'] = (df['Average_Price'] * df['Volume']).cumsum() / df['Volume'].cumsum()
    
    # Incorporate Volatility
    df['Intraday_Volatility'] = df[['Open', 'High', 'Low', 'Close']].std(axis=1)
    
    # Combine Intraday Momentum, VWAP, and Volatility
    df['Combined_Value'] = (df['VWAP'] - df['High_Low_Spread']) / df['Intraday_Volatility']
    df['Weighted_Combined_Value'] = df['Combined_Value'] * df['Volume']
    
    # Smooth the Factor
    df['Smoothed_Factor'] = df['Weighted_Combined_Value'].ewm(span=7).mean()
    
    return df['Smoothed_Factor']
```
