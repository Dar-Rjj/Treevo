```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['High'] - df['Low']

    # Compute Previous Day's Close-to-Open Return
    df['prev_day_close_to_open_return'] = df['Open'] - df['Close'].shift(1)

    # Calculate Volume Weighted Average Price (VWAP)
    prices = (df['Open'] + df['High'] + df['Low'] + df['Close']) / 4
    df['VWAP'] = (prices * df['Volume']).cumsum() / df['Volume'].cumsum()

    # Compute Volume Trend
    df['volume_trend'] = df['Volume'] - df['Volume'].shift(1)

    # Compute Price Change Momentum using EMA over the last 3 days
    df['price_change_momentum'] = df['Close'] - df['Close'].shift(1)
    df['price_change_momentum_ema'] = df['price_change_momentum'].ewm(span=3).mean()

    # Combine Intraday Momentum, VWAP, and Price-Volume Trends
    df['combined_factor'] = df['VWAP'] - df['intraday_high_low_spread'] + df['price_change_momentum_ema']
    df['weighted_combined_factor'] = df['combined_factor'] * df['Volume']

    # Smooth the Factor using EMA over the last 3 days
    df['smoothed_factor'] = df['weighted_combined_factor'].ewm(span=3).mean()

    return df['smoothed_factor']
```
