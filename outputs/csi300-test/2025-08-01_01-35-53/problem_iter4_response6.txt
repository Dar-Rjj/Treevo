```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['Prev_Close_to_Open_Return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Daily VWAP
    price_avg = (df['high'] + df['low'] + df['close']) / 3
    df['VWAP'] = (price_avg * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['Intraday_Momentum_VWAP'] = df['VWAP'] - df['High_Low_Spread']
    df['Weighted_by_Volume'] = df['Intraday_Momentum_VWAP'] * df['volume']
    
    # Adjust for Previous Day's Close-to-Open Return
    df['Adjusted_Factor'] = df['Weighted_by_Volume'] + df['Prev_Close_to_Open_Return']
    
    # Smooth the Factor with a 5-day EMA
    df['Smoothed_Factor'] = df['Adjusted_Factor'].ewm(span=5, adjust=False).mean()
    
    return df['Smoothed_Factor']
```
