```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Compute Precise VWAP
    daily_vwap = (df[['high', 'low', 'close', 'open']].mean(axis=1) * df['volume']) / df['volume']
    
    # Combine Dynamic Intraday Momentum and Precise VWAP
    combined_value = daily_vwap - intraday_high_low_spread + prev_open_to_close_return
    
    # Weight by Intraday Volume
    weighted_value = combined_value * df['volume']
    
    # Smooth the Factor for Responsiveness using EMA
    smoothed_factor = weighted_value.ewm(span=7, adjust=False).mean()
    
    return smoothed_factor
```
