```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_spread + prev_close_to_open_return
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close'] + df['open']) * df['volume']).cumsum() / (4 * df['volume']).cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_factor = vwap - intraday_spread
    
    # Weight by Intraday Volume
    weighted_factor = combined_factor * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 10 days
    smoothed_factor = weighted_factor.ewm(span=10).mean()
    
    return smoothed_factor
```
