```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['high'] + df['low']) / 2
    df['TP_Volume'] = df['TypicalPrice'] * df['volume']
    vwap = df['TP_Volume'].rolling(window=1, min_periods=1).sum() / df['volume'].rolling(window=1, min_periods=1).sum()
    
    # Compute VWAP Deviation from Close
    vwap_deviation = vwap - df['close']
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']).diff().fillna(0)
    
    # Combine VWAP Deviation and Log Return
    factor_values = -vwap_deviation + daily_log_return
    
    return factor_values
```
