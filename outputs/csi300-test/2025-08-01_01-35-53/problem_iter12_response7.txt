```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df[['open', 'high', 'low', 'close']].sum(axis=1) * df['volume']) / (4 * df['volume']))
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Incorporate Longer-Term Momentum
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    
    # Incorporate Price Dynamics
    df['price_dynamics'] = df['combined_value'] + df['prev_close_to_open_return']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 10 days
    df['factor'] = df['price_dynamics'].ewm(span=10, adjust=False).mean()
    
    return df['factor']
```
