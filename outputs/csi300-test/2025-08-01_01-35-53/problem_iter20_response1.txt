```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = ((df['high'] + df['low'] + df['close'] + df['open']) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum'] = df['prev_day_open_to_close_return'] - df['intraday_high_low_spread']
    df['vwap_weighted_intraday_momentum'] = (df['daily_vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 5 days
    df['alpha_factor'] = df['vwap_weighted_intraday_momentum'].ewm(span=5, adjust=False).mean()
    
    return df['alpha_factor']
```
