```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the dataframe is sorted by date
    df = df.sort_index()
    
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close = df['close'].shift(1)
    close_to_open_return = df['open'] - prev_close
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices_sum = (df['high'] + df['low'] + df['close']) / 3
    daily_vwap = (prices_sum * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = (daily_vwap - intraday_high_low_spread) * df['volume']
    
    # Smooth the factor using Exponential Moving Average (EMA) over the last 5 days
    smoothed_factor = combined_value.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
