```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term Return
    short_term_return = df['close'].pct_change(periods=5)
    
    # Compute Long-Term Return
    long_term_return = df['close'].pct_change(periods=20)
    
    # Calculate Average Volume over Last 5 Days
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate Average Volume over Last 20 Days
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Ratio
    volume_ratio = avg_vol_5d / avg_vol_20d
    
    # Combine Momentum and Volume Trend
    combined_factor = (short_term_return - long_term_return) * volume_ratio
    
    # Apply Exponential Moving Average (EMA) to Combined Factor with a span of 10 days
    smoothed_factor = combined_factor.ewm(span=10, adjust=False).mean()
    
    return smoothed_factor
```
