import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['Prev_Close_To_Open_Return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['VWAP'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume'] / df['volume'].sum()
    
    # Combine Intraday Momentum and VWAP
    df['Intraday_Momentum_VWAP'] = df['VWAP'] - df['High_Low_Spread']
    
    # Weight by Intraday Volume
    df['Volume_Weighted'] = df['Intraday_Momentum_VWAP'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 5 days
    df['Alpha_Factor'] = df['Volume_Weighted'].ewm(span=5, adjust=False).mean()
    
    return df['Alpha_Fctor']
