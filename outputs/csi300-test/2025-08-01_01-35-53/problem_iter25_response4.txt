```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Combine Intraday High-Low Spread and Open-to-Close Return
    intraday_momentum = intraday_high_low_spread + open_to_close_return
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    vwap = price_sum.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_momentum
    weighted_value = combined_value * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA) over the last 7 days
    alpha_factor = weighted_value.ewm(span=7, adjust=False).mean()
    
    return alpha_factor
```
