```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']

    # Compute Previous Day's Open-to-Close Return
    df['Prev_Open_to_Close_Return'] = df['close'].shift(-1) - df['open'].shift(1)

    # Calculate Volume Weighted Average Price (VWAP)
    df['Daily_VWAP'] = ((df[['high', 'low', 'open', 'close']].mean(axis=1) * df['volume']).cumsum()) / df['volume'].cumsum()

    # Combine Intraday Momentum and VWAP
    df['Intraday_Momentum'] = df['Daily_VWAP'] - df['High_Low_Spread'] + df['Prev_Open_to_Close_Return']
    df['Volume_Weighted_Intraday_Momentum'] = df['Intraday_Momentum'] * df['volume']

    # Smooth the Factor with Exponential Moving Averages (EMA)
    df['Short_Term_EMA'] = df['Volume_Weighted_Intraday_Momentum'].ewm(span=5, adjust=False).mean()
    df['Long_Term_EMA'] = df['Volume_Weighted_Intraday_Momentum'].ewm(span=14, adjust=False).mean()

    # Combine Short and Long-Term EMAs
    df['Alpha_Factor'] = 0.7 * df['Short_Term_EMA'] + 0.3 * df['Long_Term_EMA']

    return df['Alpha_Factor']
```
