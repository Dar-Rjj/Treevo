import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (prices * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = vwap - intraday_high_low_spread
    weighted_combined_momentum_vwap = combined_momentum_vwap * df['volume']
    
    # Integrate Additional Momentum Indicator
    two_day_price_change = df['close'] - df['close'].shift(2)
    
    # Combine with Additional Momentum Indicators
    alpha_factor = weighted_combined_momentum_vwap + two_day_price_change
    
    # Smooth the Factor
    alpha_factor_smoothed = alpha_factor.ewm(span=7).mean()
    
    return alpha_factor_smoothed
