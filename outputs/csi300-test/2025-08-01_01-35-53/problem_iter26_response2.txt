```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Combine Intraday Momentum and VWAP
    intraday_momentum_vwap = (vwap - intraday_high_low_spread) * df['volume']
    
    # Apply Exponential Moving Average (EMA) over the last 5 days
    ema_5d = intraday_momentum_vwap.ewm(span=5, adjust=False).mean()
    
    return ema_5d
```
