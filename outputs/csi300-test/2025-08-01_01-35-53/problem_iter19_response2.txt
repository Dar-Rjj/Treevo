import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Compute Intraday Momentum
    intraday_momentum = (intraday_high_low_spread + prev_day_open_to_close_return) / 2
    
    # Compute Daily VWAP
    price = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    daily_vwap = (price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    factor = (daily_vwap - intraday_high_low_spread) * df['volume']
    
    # Smooth the Factor using EMA
    smoothed_factor = factor.ewm(span=7, adjust=False).mean()
    
    return smoothed_factor
