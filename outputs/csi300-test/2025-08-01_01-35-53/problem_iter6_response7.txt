import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['intraday_high_low_spread'] + df['prev_close_to_open_return']) / 2
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_volume = df['volume']
    price_sum = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (price_sum * df['volume']).cumsum() / total_volume.cumsum()
    
    # Integrate Intraday Momentum with VWAP
    df['integrated_factor'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 2 days
    df['smoothed_factor'] = df['integrated_factor'].ewm(span=2).mean()
    
    return df['smoothed_factor']
