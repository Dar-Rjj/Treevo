```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = vwap - intraday_high_low_spread
    
    # Integrate Volume
    integrated_value = combined_momentum_vwap * df['volume']
    
    # Apply Shorter Exponential Moving Average (EMA) over the last 5 days
    ema_5_days = integrated_value.ewm(span=5, adjust=False).mean()
    
    return ema_5_days
```
