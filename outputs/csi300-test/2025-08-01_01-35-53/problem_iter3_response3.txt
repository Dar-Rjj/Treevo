import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close = df['close'].shift(1)
    close_to_open_return = df['open'] - prev_close
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['high'] + df['low'] + df['close']) / 3
    vwap = (price_sum * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = vwap - intraday_high_low_spread
    weighted_factor = combined_momentum_vwap * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA) over the last 5 days
    factor_ema = weighted_factor.ewm(span=5, adjust=False).mean()
    
    return factor_ema
