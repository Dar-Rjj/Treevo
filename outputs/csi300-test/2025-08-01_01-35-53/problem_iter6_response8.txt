```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Previous_Close_to_Open_Return'] = df['open'] - df['close'].shift(1)
    df['Intraday_Momentum'] = df['Intraday_High_Low_Spread'] + df['Previous_Close_to_Open_Return']

    # Calculate Volume Weighted Average Price (VWAP)
    df['Daily_VWAP'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()

    # Combine Intraday Momentum and VWAP
    df['Combined_Value'] = df['Daily_VWAP'] - df['Intraday_High_Low_Spread']
    df['Volume_Weighted_Combined'] = df['Combined_Value'] * df['volume']

    # Smooth the Factor with Exponential Moving Average (EMA) over the last 3 days
    df['Alpha_Factor'] = df['Volume_Weighted_Combined'].ewm(span=3, adjust=False).mean()

    return df['Alpha_Factor']
```
