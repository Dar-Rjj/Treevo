```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['High'] - df['Low']
    
    # Calculate Previous Day's Open-to-Close Return
    open_to_close_return = df['Close'].shift(-1) - df['Open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = ((df['High'] + df['Low'] + df['Close']) / 3 * df['Volume']).cumsum() / df['Volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum = vwap - intraday_high_low_spread
    weighted_combined_momentum = combined_momentum * df['Volume']
    
    # Smooth the Factor using Exponential Moving Averages (EMA)
    short_term_ema = weighted_combined_momentum.ewm(span=5, adjust=False).mean()
    long_term_ema = weighted_combined_momentum.ewm(span=20, adjust=False).mean()
    
    # Combine Short and Long-Term EMAs
    factor = 0.7 * short_term_ema + 0.3 * long_term_ema
    
    return factor
```
