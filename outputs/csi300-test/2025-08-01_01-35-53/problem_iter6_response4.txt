```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily VWAP
    df['VWAP'] = (df[['high', 'low', 'close', 'open']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Apply EMA over the last 3 days
    df['VWAP_EMA'] = df['VWAP'].ewm(span=3).mean()
    
    # Compute Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['Prev_Close_To_Open_Return'] = df['open'] - df['close'].shift(1)
    
    # Compute Volume Trend
    df['Volume_Trend'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Price Change Momentum
    df['Price_Change_Momentum'] = (df['close'] - df['close'].shift(1)).ewm(span=3).mean()
    
    # Combine Intraday Momentum, VWAP, and Sentiment
    df['Combined_Value'] = (df['VWAP_EMA'] - df['Intraday_Spread'] + df['Prev_Close_To_Open_Return']) * df['volume']
    
    # Final Factor Smoothing
    df['Final_Factor'] = df['Combined_Value'].ewm(span=3).mean()
    
    return df['Final_Factor']
```
