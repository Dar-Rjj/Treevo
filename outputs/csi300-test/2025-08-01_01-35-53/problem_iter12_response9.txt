import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = ((df['high'] + df['low'] + df['close'] + df['open']) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = daily_vwap - intraday_high_low_spread
    volume_weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor
    smoothed_factor = volume_weighted_combined_value.ewm(span=10, adjust=False).mean()
    
    return smoothed_factor
