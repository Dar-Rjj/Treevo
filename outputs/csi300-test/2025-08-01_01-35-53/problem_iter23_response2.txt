```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['price_sum'] = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume'].cumsum()
    df['vwap'] = df['price_sum'].cumsum() / total_volume
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_value'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    df['alpha_factor'] = df['weighted_combined_value'].ewm(span=7).mean()
    
    return df['alpha_factor']
```
