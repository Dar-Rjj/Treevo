```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    previous_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Compute Volume Weighted Average Price (VWAP)
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = (vwap - intraday_high_low_spread) * df['volume']
    
    # Compute 20-Day Momentum
    twenty_day_momentum = df['close'] - df['close'].shift(20)
    
    # Compute 50-Day Moving Average
    fifty_day_moving_average = df['close'].rolling(window=50).mean()
    
    # Compute 14-Day Average True Range (ATR)
    df['tr'] = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), abs(df['low'] - df['close'].shift(1)))
    fourteen_day_atr = df['tr'].rolling(window=14).mean()
    
    # Compute 7-Day Moving Average
    seven_day_moving_average = df['close'].rolling(window=7).mean()
    
    # Final Factor Combination
    factor = (
        combined_value +
        twenty_day_momentum +
        previous_close_to_open_return +
        fifty_day_moving_average +
        fourteen_day_atr +
        seven_day_moving_average
    )
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    final_factor = factor.ewm(span=20, adjust=False).mean()
    
    return final_factor
```
