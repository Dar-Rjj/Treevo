import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_volume = df['volume']
    price_volume_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    vwap = price_volume_sum.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    alpha_factor = weighted_combined_value.ewm(span=7).mean()
    
    return alpha_factor
