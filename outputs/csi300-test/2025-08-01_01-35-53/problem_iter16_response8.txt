import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_high_low_spread + previous_day_open_to_close_return
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    vwap = price_sum.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    volume_weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 14 days
    ema_factor = volume_weighted_combined_value.ewm(span=14, adjust=False).mean()
    
    return ema_factor
