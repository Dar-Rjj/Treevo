```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Compute Intraday High-Low Spread
    high_low_spread = data['high'] - data['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_open_to_close_return = data['close'] - data['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap_numerator = (data['high'] + data['low'] + data['close'] + data['open']) * data['volume']
    total_volume = data['volume']
    vwap = vwap_numerator / total_volume
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - high_low_spread
    weighted_combined_value = combined_value * data['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    alpha_factor = weighted_combined_value.ewm(span=7, adjust=False).mean()
    
    return alpha_factor
```
