import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    vwap = (price_sum * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday High-Low Spread and VWAP
    combined_value = vwap - intraday_high_low_spread
    
    # Weight by Intraday Volume
    weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    smoothed_factor = weighted_combined_value.ewm(span=7).mean()
    
    return smoothed_factor
