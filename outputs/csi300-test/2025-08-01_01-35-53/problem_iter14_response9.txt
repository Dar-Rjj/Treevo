```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Compute Daily VWAP
    daily_price_volume = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume'])
    total_volume = df['volume']
    vwap = daily_price_volume / total_volume
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = (vwap - intraday_high_low_spread) * df['volume']
    
    # Compute 2-Day Price Change
    two_day_price_change = df['close'] - df['close'].shift(2)
    
    # Compute 20-Day Moving Average and 20-Day Price Change
    twenty_day_ma = df['close'].rolling(window=20).mean()
    twenty_day_price_change = df['close'] - twenty_day_ma
    
    # Combine with Additional Momentum Indicators
    combined_factor = combined_momentum_vwap + two_day_price_change + twenty_day_price_change
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the Last 7 Days
    ema_7d = combined_factor.ewm(span=7, adjust=False).mean()
    
    return ema_7d
```
