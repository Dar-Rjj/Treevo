import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Volume-Weighted Average Price (VWAP)
    cumulative_volume = df['volume'].cumsum()
    cumulative_price_volume = (df['close'] * df['volume']).cumsum()
    vwap = cumulative_price_volume / cumulative_volume
    
    # Calculate Close-to-Open Spread
    close_to_open_spread = df['close'] - df['open']
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Combine Signals into Alpha Factor
    combined_signal = (intraday_range * df['volume'] - close_to_open_spread) / daily_range
    
    # Incorporate Relative Strength
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    relative_strength = ma_5 - ma_20
    
    # Final Alpha Factor
    alpha_factor = relative_strength * combined_signal
    
    return alpha_factor
