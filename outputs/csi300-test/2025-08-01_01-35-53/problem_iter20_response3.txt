```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_volume_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    df['vwap'] = price_volume_sum / total_volume
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Integrate Relative Strength Measure
    df['5_day_ma'] = df['close'].rolling(window=5).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    df['relative_strength'] = (df['5_day_ma'] - df['20_day_ma']) / df['20_day_ma']
    
    # Smooth the Factor
    df['smoothed_factor'] = df['combined_value'].ewm(span=7, adjust=False).mean()
    
    # Combine All Factors
    df['alpha_factor'] = df['smoothed_factor'] + df['relative_strength']
    
    return df['alpha_factor']
```
