```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Open-to-Close Return
    previous_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices = df[['open', 'high', 'low', 'close']].mean(axis=1)
    vwap = (prices * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    volume_weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor
    factor = volume_weighted_combined_value.ewm(span=7).mean()
    
    return factor
```
