import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Open-to-Close Return
    prev_day_open = df['open'].shift(1)
    open_to_close_return = df['close'] - prev_day_open
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_volume = df['volume']
    vwap = (df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume'] + df['open'] * df['volume']) / (4 * total_volume)
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    volume_weighted_factor = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    smoothed_factor = volume_weighted_factor.ewm(span=7, adjust=False).mean()
    
    return smoothed_factor
