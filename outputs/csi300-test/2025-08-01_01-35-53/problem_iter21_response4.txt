```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA)
    def adaptive_ema(data, short_span=7, long_span=28):
        # Calculate the standard deviation of the last 7 days to determine recent volatility
        recent_volatility = data.rolling(window=short_span).std()
        
        # Determine the EMA span based on recent volatility
        ema_span = np.where(recent_volatility > recent_volatility.mean(), short_span, long_span)
        
        # Apply EMA with the determined span
        return data.ewm(span=ema_span, adjust=False).mean()
    
    df['smoothed_factor'] = adaptive_ema(df['combined_value'])
    
    return df['smoothed_factor']
```
