```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['close'].shift(1) - df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4

    # Combine Intraday Momentum and VWAP
    combined_value = (vwap - high_low_spread) * df['volume']

    # Apply Exponential Moving Average (EMA) over the last 5 days
    ema_5d = combined_value.ewm(span=5, adjust=False).mean()

    return ema_5d
```
