import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_momentum_vwap'] = df['vwap'] - df['intraday_high_low_spread']
    df['volume_weighted_combined'] = df['combined_momentum_vwap'] * df['volume']
    
    # Integrate Adaptive EMA
    df['short_ema'] = df['volume_weighted_combined'].ewm(span=7).mean()
    df['long_ema'] = df['volume_weighted_combined'].ewm(span=21).mean()
    
    # Blend Momentum and Trend
    df['weighted_ema'] = df['short_ema'] * 0.7 + df['long_ema'] * 0.3
    df['final_factor'] = df['weighted_ema'] + df['volume_weighted_combined']
    
    return df['final_factor']
