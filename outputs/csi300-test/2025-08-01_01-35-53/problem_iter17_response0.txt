```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_day_open = df['open'].shift(1)
    todays_close = df['close']
    open_to_close_return = todays_close - prev_day_open
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = (df[['high', 'low', 'close', 'open']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = daily_vwap - intraday_high_low_spread
    weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Averages (EMA)
    short_term_ema = weighted_combined_value.ewm(span=7, adjust=False).mean()
    long_term_ema = weighted_combined_value.ewm(span=28, adjust=False).mean()
    
    # Combine Short-Term and Long-Term EMAs
    weighted_average = (short_term_ema * 0.7) + (long_term_ema * 0.3)
    
    return weighted_average
```
