```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = (vwap - intraday_high_low_spread) * df['volume']
    
    # Incorporate Multi-Period Momentum
    momentum_20 = df['close'] - df['close'].shift(20)
    momentum_60 = df['close'] - df['close'].shift(60)
    
    # Incorporate Price Dynamics
    price_dynamics = close_to_open_return
    
    # Enhance with Long-Term Price Trend
    long_term_trend = df['close'].rolling(window=50).mean()
    
    # Integrate Short-Term Volatility
    tr = np.maximum((df['high'] - df['low']), np.abs(df['high'] - df['close'].shift(1)), np.abs(df['low'] - df['close'].shift(1)))
    atr_14 = tr.rolling(window=14).mean()
    
    # Incorporate Short-Term Price Trend
    short_term_trend = df['close'].rolling(window=7).mean()
    
    # Final Factor Combination
    final_factor = (
        combined_momentum_vwap +
        momentum_20 +
        momentum_60 +
        price_dynamics +
        long_term_trend +
        atr_14 +
        short_term_trend
    )
    
    # Smooth the Factor
    smoothed_factor = final_factor.ewm(span=20, adjust=False).mean()
    
    return smoothed_factor
```
