import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['Intraday_High_Low_Spread'] = df['High'] - df['Low']
    df['Prev_Open'] = df['Open'].shift(1)
    df['Prev_Open_to_Close_Return'] = (df['Close'] - df['Prev_Open']) / df['Prev_Open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['High'] + df['Low'] + df['Close']) / 3
    df['Price_Volume'] = df['Typical_Price'] * df['Volume']
    df['Cumulative_PV'] = df['Price_Volume'].rolling(window=1, min_periods=1).sum()
    df['Cumulative_Volume'] = df['Volume'].rolling(window=1, min_periods=1).sum()
    df['VWAP'] = df['Cumulative_PV'] / df['Cumulative_Volume']
    
    # Combine Intraday Momentum and VWAP
    df['Intraday_Momentum'] = df['Prev_Open_to_Close_Return'] - df['Intraday_High_Low_Spread'] / df['VWAP']
    df['Volume_Weighted_Intraday_Momentum'] = df['Intraday_Momentum'] * df['Volume']
    
    # Smooth the Factor
    df['Smoothed_Factor'] = df['Volume_Weighted_Intraday_Momentum'].ewm(span=7, adjust=False).mean()
    
    return df['Smoothed_Factor']
