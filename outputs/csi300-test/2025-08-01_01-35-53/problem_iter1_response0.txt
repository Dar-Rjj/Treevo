```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close Prices for last 14 days
    sma_14 = df['close'].rolling(window=14).mean()
    
    # Calculate Exponential Moving Average (EMA) of Close Prices for last 12 days
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    
    # Compute Momentum
    momentum = np.abs(sma_14 - ema_12)
    
    # Calculate Volume Weighted Moving Average (VWMA)
    vwma = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Combine Momentum with VWMA
    final_factor = vwma * momentum
    
    return final_factor
```
