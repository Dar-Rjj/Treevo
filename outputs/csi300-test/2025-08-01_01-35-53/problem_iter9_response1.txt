import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] * df['volume'] + df['high'] * df['volume'] + df['low'] * df['volume'] + df['close'] * df['volume']) / (4 * df['volume'])
    
    # Combine Intraday Momentum and VWAP
    combined_value = (vwap - intraday_high_low_spread) * df['volume']
    
    # Apply Exponential Moving Average (EMA) over the last 5 days
    ema_5 = combined_value.ewm(span=5).mean()
    
    # Final Alpha Factor
    return ema_5
