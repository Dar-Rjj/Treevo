import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (prices * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_value'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor
    df['smoothed_factor'] = df['weighted_value'].ewm(span=5, adjust=False).mean()
    
    return df['smoothed_factor']
