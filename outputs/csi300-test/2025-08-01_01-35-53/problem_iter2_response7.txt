```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Spread
    close_open_spread = df['close'] - df['open']
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Volume-Weighted High-Low Spread
    volume_weighted_high_low_spread = high_low_spread * df['volume']
    
    # Adjust for Daily Range
    adjusted_volume_weighted_high_low_spread = volume_weighted_high_low_spread / daily_range
    
    # Combine with Close-to-Open Spread
    final_factor = adjusted_volume_weighted_high_low_spread - close_open_spread
    
    return final_factor
```
