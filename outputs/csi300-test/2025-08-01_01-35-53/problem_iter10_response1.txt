import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].sum(axis=1) * df['volume']) / (4 * df['volume'])
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum_vwap'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Compute 2-Day Price Change
    df['two_day_price_change'] = df['close'] - df['close'].shift(2)
    
    # Combine with Additional Momentum Indicator
    df['combined_factor'] = df['intraday_momentum_vwap'] + df['two_day_price_change']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    df['smoothed_factor'] = df['combined_factor'].ewm(span=7, adjust=False).mean()
    
    return df['smoothed_factor']
