```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close'] + df['open']) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum'] = (df['intraday_high_low_spread'] + df['prev_close_to_open_return'])
    df['weighted_momentum_vwap'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Smooth the Factor using EMA
    df['alpha_factor'] = df['weighted_momentum_vwap'].ewm(span=7, adjust=False).mean()
    
    return df['alpha_factor']
```
