```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close = df['close'].shift(1)
    close_to_open_return = df['open'] - prev_close
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    
    # Weight by Intraday Volume
    weighted_value = combined_value * df['volume']
    
    # Incorporate Previous Day's Close-to-Open Return
    alpha_factor = weighted_value + close_to_open_return
    
    # Smooth the Factor
    alpha_factor_smoothed = alpha_factor.ewm(span=5, adjust=False).mean()
    
    return alpha_factor_smoothed
```
