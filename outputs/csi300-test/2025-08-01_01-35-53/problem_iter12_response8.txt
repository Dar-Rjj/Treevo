```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum_vwap'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Integrate Additional Momentum Indicator
    df['2_day_price_change'] = df['close'] - df['close'].shift(2)
    
    # Incorporate Longer-Term Price Trends
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    df['20_day_price_change'] = df['close'] - df['20_day_ma']
    
    # Combine with Additional Momentum Indicators
    df['combined_factor'] = df['intraday_momentum_vwap'] + df['2_day_price_change'] + df['20_day_price_change']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    df['factor_ema_7'] = df['combined_factor'].ewm(span=7, adjust=False).mean()
    
    return df['factor_ema_7']
```
