```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum (High - Low)
    intraday_momentum = df['high'] - df['low']
    
    # Calculate Volume Reversal Indicator
    # Calculate the average volume of the past 5 days (excluding the current day)
    avg_volume_5d = df['volume'].rolling(window=6).apply(lambda x: x[:-1].mean(), raw=False)
    
    # Compare today's volume with the average volume
    volume_reversal = df['volume'] / avg_volume_5d
    
    # Set threshold for significant increase in volume
    volume_threshold = 1.5  # Example threshold, can be adjusted
    
    # Determine if volume reversal is significant
    significant_volume_reversal = (volume_reversal > volume_threshold).astype(int)
    
    # Combine Intraday Momentum and Volume Reversal
    # If volume reversal is significant, apply intraday momentum
    # If no significant volume reversal, do not apply any intraday momentum effect
    factor = intraday_momentum * significant_volume_reversal
    
    return factor
```
