```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Intraday Returns
    high_low_diff = df['high'] - df['low']
    intraday_return = high_low_diff / df['open']

    # Incorporate Volume Adjusted Momentum
    close_to_open_return = (df['close'] - df['open']) / df['open']
    volume_adjusted_momentum = close_to_open_return * df['volume']

    # Integrate Volatility Measure
    daily_range = df['high'] - df['low']
    smoothed_daily_range = daily_range.ewm(span=10, adjust=False).mean()
    volatility_measure = smoothed_daily_range / df['close']

    # Assess Trade Direction
    up_movement = (df['close'] - df['close'].shift(1)).apply(lambda x: max(x, 0))
    down_movement = (df['close'].shift(1) - df['close']).apply(lambda x: max(x, 0))
    up_sum = up_movement.rolling(window=14).sum()
    down_sum = down_movement.rolling(window=14).sum()
    relative_strength = up_sum / down_sum

    # Combine Factors into Final Alpha
    final_alpha = (intraday_return + 
                   volume_adjusted_momentum + 
                   volatility_measure + 
                   relative_strength)

    return final_alpha
```
