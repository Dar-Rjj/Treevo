```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_open = df['open'].shift(1)
    open_to_close_return = df['close'] - prev_open
    
    # Calculate Volume Weighted Average Price (VWAP)
    avg_price = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    vwap = (avg_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    intraday_momentum = open_to_close_return - intraday_spread
    combined_value = vwap - intraday_spread
    weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA)
    factor = weighted_combined_value.ewm(span=7, adjust=False).mean()
    
    return factor
```
