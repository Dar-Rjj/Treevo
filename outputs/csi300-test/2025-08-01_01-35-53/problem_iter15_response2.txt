import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Open-to-Close Return
    prev_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    vwap = price_sum.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    intraday_momentum_vwap = (vwap - intraday_high_low_spread) * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 14 days
    ema_factor = intraday_momentum_vwap.ewm(span=14, adjust=False).mean()
    
    return ema_factor
