import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Combine Intraday Momentum and VWAP
    combined_factor = (vwap - intraday_high_low_spread) * df['volume']
    
    # Apply Shorter Exponential Moving Average (EMA)
    shorter_ema = combined_factor.ewm(span=5, adjust=False).mean()
    
    # Apply Longer Exponential Moving Average (EMA)
    longer_ema = combined_factor.ewm(span=20, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = shorter_ema - longer_ema
    
    return alpha_factor
