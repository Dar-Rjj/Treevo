```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on intraday momentum with volume influence.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    
    # Calculate Daily Intraday Returns
    high_low_diff = df['high'] - df['low']
    normalized_high_low_diff = high_low_diff / df['open']
    
    # Incorporate Volume Adjusted Momentum
    close_open_return = (df['close'] - df['open']) / df['open']
    volume_adjusted_momentum = close_open_return * df['volume']
    
    # Combine Factors into Final Alpha
    final_alpha = normalized_high_low_diff + volume_adjusted_momentum
    
    return final_alpha
```
