import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['High'] - df['Low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['Open'] - df['Close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = ((df['Open'] + df['High'] + df['Low'] + df['Close']) / 4 * df['Volume']).cumsum() / df['Volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = (daily_vwap - intraday_high_low_spread) * df['Volume']
    
    # Incorporate Price Trends
    def compute_3_day_price_trend(close_prices):
        x = np.arange(3)
        y = close_prices[-3:]
        slope, _ = np.polyfit(x, y, 1)
        return slope
    
    price_trend = df['Close'].rolling(window=3).apply(compute_3_day_price_trend, raw=True)
    
    # Integrate Price Trend with Momentum and VWAP
    integrated_factor = combined_momentum_vwap + price_trend
    
    # Smooth the Factor
    smoothed_factor = integrated_factor.ewm(span=3).mean()
    
    return smoothed_factor
