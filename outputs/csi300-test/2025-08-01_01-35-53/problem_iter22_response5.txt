```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_day_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Compute Intraday Momentum
    df['intraday_momentum'] = df['prev_day_open_to_close_return'] / df['intraday_high_low_spread']
    
    # Compute Daily VWAP
    mid_price = (df['high'] + df['low']) / 2
    df['vwap'] = (mid_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['combined_factor'] = df['vwap'] - df['intraday_high_low_spread']
    df['volume_weighted_factor'] = df['combined_factor'] * df['volume']
    
    # Smooth the Factor using EMA
    df['smoothed_factor'] = df['volume_weighted_factor'].ewm(span=3).mean()
    
    return df['smoothed_factor']
```
