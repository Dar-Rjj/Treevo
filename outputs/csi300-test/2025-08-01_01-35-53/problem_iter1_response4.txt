```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['open'] / df['close'].shift(1)) - 1
    
    # Compute Daily VWAP
    df['vwap'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum'] = df['intraday_high_low_spread'] * df['prev_close_to_open_return']
    df['combined_value'] = df['vwap'] - df['intraday_high_low_spread']
    df['weighted_combined_value'] = df['combined_value'] * df['volume']
    
    return df['weighted_combined_value']
```
