import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    prev_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_volume_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    vwap = price_volume_sum.cumsum() / total_volume.cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    volume_weighted_value = combined_value * df['volume']
    
    # Smooth the Factor with Exponential Moving Average (EMA) over 7 days
    ema_7_days = volume_weighted_value.ewm(span=7, adjust=False).mean()
    
    return ema_7_days
