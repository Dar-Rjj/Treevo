import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Combine Intraday Momentum and VWAP
    combined_momentum = (vwap - intraday_spread) * df['volume']
    
    # Apply Shorter Exponential Moving Average (EMA) for Responsiveness (5 days)
    ema_5d = combined_momentum.ewm(span=5, adjust=False).mean()
    
    # Apply Longer Exponential Moving Average (EMA) for Stability (20 days)
    ema_20d = combined_momentum.ewm(span=20, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = ema_5d - ema_20d
    
    return alpha_factor
