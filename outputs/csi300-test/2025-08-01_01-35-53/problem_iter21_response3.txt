```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['High'] - df['Low']
    
    # Compute Previous Day's Open-to-Close Return
    open_to_close_return = df['Close'] - df['Open'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = (intraday_high_low_spread + open_to_close_return) / 2
    
    # Compute Daily VWAP
    daily_vwap = (df['High'] + df['Low'] + df['Close'] + df['Open']) / 4
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (daily_vwap * df['Volume']).cumsum() / df['Volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    
    # Weight by Intraday Volume
    weighted_value = combined_value * df['Volume']
    
    # Apply Exponential Moving Average (EMA) over the last 7 days
    smoothed_factor = weighted_value.ewm(span=7, adjust=False).mean()
    
    return smoothed_factor
```
