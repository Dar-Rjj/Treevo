```python
import pandas as pd

def heuristics_v2(df):
    # Ensure the dataframe is sorted by date
    df = df.sort_index()
    
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    
    # Calculate Intraday Volume Trend
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Ratio
    volume_ratio = avg_volume_5d / avg_volume_20d
    
    # Combine Intraday Momentum, VWAP, and Volume Trend
    combined_value = (vwap - intraday_high_low_spread) * volume_ratio
    weighted_combined_value = combined_value * df['volume']
    
    return weighted_combined_value
```
