```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    vwap = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Combine Intraday Momentum and VWAP
    intraday_momentum = (vwap - intraday_high_low_spread) * df['volume']
    
    # Incorporate Multi-Period Momentum
    multi_period_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Combine Multi-Period Momentum with Volume-Weighted Intraday Momentum
    combined_momentum = intraday_momentum * multi_period_momentum
    
    # Apply Exponential Moving Average (EMA) over the last 5 days
    ema_5d = combined_momentum.ewm(span=5, adjust=False).mean()
    
    return ema_5d
```
