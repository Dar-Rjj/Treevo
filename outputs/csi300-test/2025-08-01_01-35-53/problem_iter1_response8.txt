```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume Change
    volume_change = df['volume'].diff(1)
    
    # Normalize Volume Change by Average
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    normalized_volume_change = volume_change / avg_volume_5d
    
    # Combine Components
    factor = intraday_range * normalized_volume_change
    
    # Smooth Factor using a 5-day moving average
    smoothed_factor = factor.rolling(window=5).mean()
    
    return smoothed_factor
```
