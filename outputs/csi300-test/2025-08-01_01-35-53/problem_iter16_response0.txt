import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Compute Previous Day's Open-to-Close Return
    df['prev_open_to_today_close_return'] = df['close'].shift(-1) - df['open']

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['vwap'] = df['vwap'].cumsum() / df['volume'].cumsum()

    # Combine Intraday Momentum and VWAP
    df['intraday_momentum'] = df['prev_open_to_today_close_return'] - df['intraday_high_low_spread']
    df['weighted_intraday_momentum'] = df['intraday_momentum'] * df['volume']

    # Smooth the Factor using Exponential Moving Average (EMA)
    df['smoothed_factor'] = df['weighted_intraday_momentum'].ewm(span=7, adjust=False).mean()
    
    return df['smoothed_factor']
