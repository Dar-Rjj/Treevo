import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_day_open = df['open'].shift(1)
    open_to_close_return = df['close'] - previous_day_open
    
    # Calculate Volume-Weighted Average Price (VWAP)
    price_volume_sum = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    total_volume = df['volume']
    vwap = price_volume_sum / total_volume
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    ema_7_days = weighted_combined_value.ewm(span=7, adjust=False).mean()
    
    return ema_7_days
