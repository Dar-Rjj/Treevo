```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['Prev_Close_to_Open_Return'] = df['open'] - df['close'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['Price_Volume'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['Cumulative_Price_Volume'] = df['Price_Volume'].cumsum()
    df['Cumulative_Volume'] = df['volume'].cumsum()
    df['VWAP'] = df['Cumulative_Price_Volume'] / df['Cumulative_Volume']
    
    # Combine Intraday Momentum and VWAP
    df['Combined_Value'] = df['VWAP'] - df['Intraday_High_Low_Spread']
    df['Volume_Weighted_Combined'] = df['Combined_Value'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    df['Alpha_Factor'] = df['Volume_Weighted_Combined'].ewm(span=5, adjust=False).mean()
    
    return df['Alpha_Factor']
```
