import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['prev_open_to_close_return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['daily_vwap'] = ((df['high'] + df['low'] + df['close'] + df['open']) / 4) * df['volume']
    total_volume = df['volume'].rolling(window=1).sum()
    df['vwap'] = df['daily_vwap'].rolling(window=1).sum() / total_volume
    
    # Combine Intraday Momentum and VWAP
    df['combined_value'] = df['vwap'] - df['high_low_spread']
    df['weighted_combined_value'] = df['combined_value'] * df['volume']
    
    # Smooth the Factor
    df['smoothed_factor'] = df['weighted_combined_value'].ewm(span=7, adjust=False).mean()
    
    return df['smoothed_factor']
