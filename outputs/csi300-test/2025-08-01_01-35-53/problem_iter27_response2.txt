import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = previous_open_to_close_return / intraday_high_low_spread
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = (df[['high', 'low', 'close', 'open']].sum(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = daily_vwap - intraday_high_low_spread
    weighted_combined_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    smoothed_factor = weighted_combined_value.ewm(span=7).mean()
    
    return smoothed_factor
