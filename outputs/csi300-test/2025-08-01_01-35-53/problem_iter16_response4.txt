```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_day_close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Compute Daily VWAP
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP, Weight by Intraday Volume
    df['combined_value'] = (df['vwap'] - df['intraday_high_low_spread']) * df['volume']
    
    # Compute 20-Day Momentum
    df['20_day_momentum'] = df['close'] - df['close'].shift(20)
    
    # Incorporate Price Dynamics by Adding Previous Day's Close-to-Open Return
    df['factor_raw'] = df['combined_value'] + df['prev_day_close_to_open_return'] + df['20_day_momentum']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 20 days
    df['alpha_factor'] = df['factor_raw'].ewm(span=20).mean()
    
    return df['alpha_factor']
```
