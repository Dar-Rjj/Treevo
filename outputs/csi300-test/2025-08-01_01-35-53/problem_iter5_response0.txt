import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['close_prev'] = df['close'].shift(1)
    close_to_open_return = df['open'] - df['close_prev']
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_price_volume = (df['open'] + df['high'] + df['low'] + df['close']) * df['volume']
    total_volume = df['volume']
    vwap = total_price_volume / (4 * total_volume)
    
    # Combine Intraday Momentum and VWAP
    combined_value = vwap - intraday_high_low_spread
    volume_weighted_factor = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 3 days
    smoothed_factor = volume_weighted_factor.ewm(span=3).mean()
    
    return smoothed_factor
