```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Intraday Momentum
    high_low_diff = df['high'] - df['low']
    intraday_momentum = high_low_diff / df['open']

    # Incorporate Volume Adjusted Close-to-Open Momentum
    close_open_return = (df['close'] - df['open']) / df['open']
    volume_weighted_close_open_momentum = close_open_return * df['volume']

    # Determine Relative Strength
    twenty_day_ma = df['close'].rolling(window=20).mean()
    relative_strength = df['close'] - twenty_day_ma

    # Measure Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    ten_day_volatility = log_returns.rolling(window=10).std()

    # Incorporate Liquidity Measure
    five_day_avg_volume = df['volume'].rolling(window=5).mean()

    # Combine Factors into Final Alpha
    alpha_factor = (
        intraday_momentum +
        volume_weighted_close_open_momentum +
        relative_strength -
        ten_day_volatility +
        five_day_avg_volume
    )

    return alpha_factor
```
