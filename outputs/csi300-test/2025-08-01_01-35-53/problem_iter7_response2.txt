```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    prices = df[['open', 'high', 'low', 'close']]
    volumes = df['volume'].values.reshape(-1, 1)
    vwap = (prices * volumes).sum(axis=1) / df['volume']
    df['vwap'] = vwap
    
    # Combine Intraday Momentum and VWAP
    df['intraday_momentum'] = df['intraday_high_low_spread'] + df['prev_close_to_open_return']
    df['vwap_intraday_momentum'] = vwap - df['intraday_momentum']
    
    # Weight by Intraday Volume
    df['vwap_intraday_volume_weighted'] = df['vwap_intraday_momentum'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 5 days
    df['alpha_factor'] = df['vwap_intraday_volume_weighted'].ewm(span=5, adjust=False).mean()
    
    return df['alpha_factor']
```
