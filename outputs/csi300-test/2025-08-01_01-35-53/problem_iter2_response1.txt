import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weight by Volume
    volume_weighted_spread = high_low_spread * df['volume']
    
    # Sum of Volumes over the Last 5 Days
    volume_sum_5d = df['volume'].rolling(window=5).sum()
    
    # Volume-Weighted Intraday Range
    volume_weighted_intraday_range = (volume_weighted_spread / volume_sum_5d) * df['volume']

    # Volume Momentum Component
    volume_change = df['volume'].diff()

    # Smooth Volume Change with a 5-day Moving Average
    smoothed_volume_momentum = volume_change.rolling(window=5).mean()

    # Combine Components
    combined_factor = volume_weighted_intraday_range * smoothed_volume_momentum

    # Final Factor: Apply Moving Average Over the Last 5 Days
    final_factor = combined_factor.rolling(window=5).mean()

    return final_factor
