```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Moving Average
    short_window = 5
    df['short_mavg'] = df['close'].rolling(window=short_window).mean()

    # Calculate Long-Term Moving Average
    long_window = 20
    df['long_mavg'] = df['close'].rolling(window=long_window).mean()
    
    # Determine Trend Direction
    df['trend_direction'] = df['short_mavg'] - df['long_mavg']
    df['trend_direction'] = df['trend_direction'].apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff(1)
    
    # Identify Volume Shock Days
    volume_shock_threshold = 2  # Define a threshold for significant volume change
    df['volume_shock'] = (df['volume_change'] / df['volume'].shift(1)).abs() > volume_shock_threshold
    
    # Combine Trend and Volume Shock
    df['alpha_factor'] = df['trend_direction'] * df['volume_shock'].astype(int)
    
    return df['alpha_factor']
```
