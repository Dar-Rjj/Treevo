```python
import pandas as pd

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = df['open'] - df['close'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = prev_close_to_open_return + high_low_spread
    
    # Calculate Volume-Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate 5-Day and 20-Day Moving Average Volume
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Ratio
    volume_ratio = avg_vol_5 / avg_vol_20
    
    # Combine Intraday Momentum, VWAP, and Volume Ratio
    combined_factor = (vwap - high_low_spread) * volume_ratio
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
