```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    previous_day_open_to_close_return = df['close'] - df['open'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = intraday_high_low_spread + previous_day_open_to_close_return
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = ((df['high'] + df['low'] + df['close'] + df['open']) / 4 * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_value = daily_vwap - intraday_momentum
    volume_weighted_value = combined_value * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    smoothed_factor = volume_weighted_value.ewm(span=14, adjust=False).mean()
    
    return smoothed_factor
```
