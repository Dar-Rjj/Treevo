```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Compute Previous Day's Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    daily_vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Momentum and VWAP
    combined_momentum_vwap = daily_vwap - intraday_high_low_spread
    weighted_combined_momentum_vwap = combined_momentum_vwap * df['volume']
    
    # Integrate Additional Momentum Indicators
    two_day_price_change = df['close'] - df['close'].shift(2)
    seven_day_price_change = df['close'] - df['close'].shift(7)
    
    # Incorporate Longer-Term Price Trends
    thirty_day_price_trend = df['close'] - df['close'].shift(30)
    
    # Combine with Additional Momentum Indicators and Longer-Term Trends
    combined_factor = weighted_combined_momentum_vwap + two_day_price_change + seven_day_price_change + thirty_day_price_trend
    
    # Smooth the Factor using Exponential Moving Average (EMA)
    smooth_factor = combined_factor.ewm(span=7).mean()
    
    return smooth_factor
```
