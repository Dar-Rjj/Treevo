```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Compute Previous Day's Open-to-Close Return
    df['Prev_Open_to_Close_Return'] = df['close'] - df['open'].shift(1)
    
    # Calculate Volume Weighted Average Price (VWAP)
    total_price_volume = (df['high'] + df['low'] + df['close'] + df['open']) * df['volume']
    df['VWAP'] = total_price_volume.rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Combine Intraday Momentum and VWAP
    df['Combined_Factor'] = df['VWAP'] - df['Intraday_High_Low_Spread']
    df['Volume_Weighted_Factor'] = df['Combined_Factor'] * df['volume']
    
    # Smooth the Factor using Exponential Moving Average (EMA) over the last 7 days
    df['Smoothed_Factor'] = df['Volume_Weighted_Factor'].ewm(span=7, adjust=False).mean()
    
    return df['Smoothed_Factor']
```
