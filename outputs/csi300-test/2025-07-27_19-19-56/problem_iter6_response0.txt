```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day momentum (difference between today's close and the close 10 days ago)
    momentum_10 = df['close'] - df['close'].shift(10)
    
    # Calculate the 5-day rate of change for the closing price
    roc_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate the 21-day standard deviation of the log returns to measure volatility
    log_return = np.log(df['close'] / df['close'].shift(1))
    volatility_21 = log_return.rolling(window=21).std()
    
    # Calculate the 10-day moving average of volume to measure liquidity
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    
    # Calculate the 50-day simple moving average of the close price
    sma_close_50 = df['close'].rolling(window=50).mean()
    
    # Alpha factor: combine momentum, rate of change, volatility, and liquidity, adjusted by the 50-day SMA
    factor_values = (momentum_10 * roc_5) / (volatility_21 + 1e-7) * volume_ma_10 * (df['close'] / sma_close_50)
    
    return factor_values
```
