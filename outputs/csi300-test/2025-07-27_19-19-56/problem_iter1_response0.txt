```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then multiply by the 5-day moving average of volume
    price_change = df['close'].diff()
    volume_ma_5 = df['volume'].rolling(window=5).mean()
    factor_values = price_change * volume_ma_5
    return factor_values
```
