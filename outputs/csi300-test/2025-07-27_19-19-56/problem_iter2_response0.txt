```txt
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    price_change = df['close'].diff()
    
    # Calculate the 5-day moving average of volume
    volume_ma_5 = df['volume'].rolling(window=5).mean()
    
    # Calculate the 3-day rate of change for the closing price
    roc_3 = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    
    # Alpha factor: combine price change with volume and rate of change
    factor_values = price_change * volume_ma_5 * roc_3
    
    return factor_values
```
