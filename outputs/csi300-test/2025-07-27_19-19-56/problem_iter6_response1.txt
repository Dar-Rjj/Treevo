```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the 5-day simple moving average of the close price for responsiveness
    df['sma_close_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day simple moving average of the close price as a medium-term trend
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 10-day standard deviation of the log returns for volatility
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return_10'] = df['log_return'].rolling(window=10).std()
    
    # Calculate the 3-day exponential moving average of volume for smoothing
    df['volume_ema_3'] = df['volume'].ewm(span=3, adjust=False).mean()
    
    # Introduce a new factor: the ratio of 5-day SMA to 20-day SMA
    df['sma_ratio'] = df['sma_close_5'] / df['sma_close_20']
    
    # Calculate the factor value with a focus on shorter term trends, smoothed volume, and the SMA ratio
    factor_value = (df['close_diff'] - (df['sma_close_5'] - df['sma_close_20'])) * df['volume_ema_3'] * df['sma_ratio'] / (df['std_log_return_10'] + 1e-7)
    
    return factor_value
```
