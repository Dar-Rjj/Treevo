```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the 20-day and 50-day simple moving average of the close price
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    df['sma_close_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the 20-day standard deviation of the log returns to measure volatility
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Calculate the 5-day moving average of volume for weighting
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the adaptive window for the moving averages based on recent volatility
    df['adaptive_window'] = (df['std_log_return_20'] * 20).clip(lower=10, upper=60).astype(int)
    
    # Recalculate the simple moving averages with the adaptive window
    df['sma_close_adaptive'] = df['close'].rolling(window=df['adaptive_window']).mean()
    
    # Calculate the factor value, incorporating more diverse features and using adaptive windows
    factor_value = (
        (df['close_diff'] - (df['sma_close_20'] - df['sma_close_50'])) 
        + (df['sma_close_adaptive'] - df['sma_close_50']) 
        + (df['open'] - df['close'])
        + (df['high'] - df['low'])
    ) * df['volume_ma_5'] / (df['std_log_return_20'] + 1e-7)
    
    return factor_value
```
