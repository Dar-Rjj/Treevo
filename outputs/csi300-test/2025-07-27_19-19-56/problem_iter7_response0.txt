```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    # EMA of log returns with a 10-day window
    df['ema_log_return_10'] = df['log_return'].ewm(span=10, adjust=False).mean()
    # Normalize EMA-smoothed log returns by their standard deviation over the last 20 days
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    normalized_ema_log_return = df['ema_log_return_10'] / (df['std_log_return_20'] + 1e-7)
    
    # Short-term SMA and EMA for trend strength
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    trend_strength = df['sma_close_10'] / df['ema_close_10']
    
    # Integrate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Momentum
    df['momentum_10'] = df['close'] / df['close'].shift(10) - 1
    
    # Volatility
    df['volatility_20'] = df['log_return'].rolling(window=20).std()
    
    # Liquidity
    liquidity = df['amount'] / df['volume']
    
    # Combine all factors
    factor_value = (
        normalized_ema_log_return * trend_strength * 
        rsi * df['momentum_10'] / (df['volatility_20'] + 1e-7) * 
        liquidity
    )
    
    return factor_value
```
