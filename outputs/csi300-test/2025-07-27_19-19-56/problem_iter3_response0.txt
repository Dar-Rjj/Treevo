```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the 10-day simple moving average of the close price for responsiveness
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    # Calculate the 30-day simple moving average of the close price as a medium-term trend
    df['sma_close_30'] = df['close'].rolling(window=30).mean()
    # Calculate the 20-day standard deviation of the log returns for volatility
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    # Calculate the 5-day exponential moving average of volume for smoothing
    df['volume_ema_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    # Calculate the factor value with a focus on shorter term trends and smoothed volume
    factor_value = (df['close_diff'] - (df['sma_close_10'] - df['sma_close_30'])) * df['volume_ema_5'] / (df['std_log_return_20'] + 1e-7)
    return factor_value
```
