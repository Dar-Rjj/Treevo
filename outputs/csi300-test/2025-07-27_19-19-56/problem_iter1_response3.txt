```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the change in close price from one day to the next
    df['close_change'] = df['close'].diff()
    
    # Calculate the average true range over a 14-day period
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Alpha factor: the ratio of the daily close change to the 14-day ATR, normalized by the volume
    alpha_factor = (df['close_change'] / (df['atr_14'] + 1e-7)) * (df['volume'] / df['volume'].rolling(window=10).mean())
    
    return alpha_factor
```
