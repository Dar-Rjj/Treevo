```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the 10-day EMA of the close price for short-term trend
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 60-day EMA of the close price for long-term trend
    df['ema_close_60'] = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the RSI (Relative Strength Index) over 14 days
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the 20-day standard deviation of the log returns
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Calculate the 5-day moving average of volume
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the factor value
    factor_value = (df['log_return'] * df['volume_ma_5']) * (df['ema_close_10'] - df['ema_close_60']) / (df['std_log_return_20'] + 1e-7) * (1 - df['rsi'] / 100)
    return factor_value
```
