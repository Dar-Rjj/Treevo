```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the 20-day smoothed moving average of the close price
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation of the log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Calculate the exponential moving average (EMA) of the volume over 10 days
    df['ema_volume_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the factor value
    factor_value = (
        (df['close_diff'] - df['sma_close_20']) / (df['std_log_return_20'] + 1e-7) +
        (df['volume'] - df['ema_volume_10']) / (df['ema_volume_10'] + 1e-7) +
        (50 - df['rsi_14']) / 50
    ) / 3
    
    return factor_value
```
