```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate a 10-day Exponential Moving Average (EMA) of the close price for responsiveness
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate a 30-day EMA of the close price
    df['ema_close_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 20-day standard deviation of the log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Calculate the 5-day moving average of volume
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the factor value
    factor_value = (df['close_pct_change'] - (df['ema_close_10'] - df['ema_close_30'])) * df['volume_ma_5'] / (df['std_log_return_20'] + 1e-7)
    
    return factor_value
```
