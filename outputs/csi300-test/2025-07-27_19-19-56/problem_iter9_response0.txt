```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 20-day EMA of the log returns
    df['ema_log_return_20'] = df['log_return'].ewm(span=20, adjust=False).mean()
    
    # Calculate 20-day rolling standard deviation of the log returns for normalization
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Normalize the EMA of log returns by its standard deviation
    df['normalized_ema_lr_20'] = df['ema_log_return_20'] / (df['std_log_return_20'] + 1e-7)
    
    # Calculate volume-weighted RSI
    df['gain'] = df['close_diff'] * (df['close_diff'] > 0)
    df['loss'] = -df['close_diff'] * (df['close_diff'] < 0)
    df['avg_gain_14'] = df['gain'].rolling(window=14).mean() * df['volume']
    df['avg_loss_14'] = df['loss'].rolling(window=14).mean() * df['volume']
    df['rs_14'] = df['avg_gain_14'] / (df['avg_loss_14'] + 1e-7)
    df['vrsi_14'] = 100 - (100 / (1 + df['rs_14']))
    
    # Calculate momentum using 10-day return
    df['momentum_10'] = df['close'] / df['close'].shift(10) - 1
    
    # Calculate SMA/EMA ratios for trend strength
    df['sma_close_50'] = df['close'].rolling(window=50).mean()
    df['ema_close_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['sma_ema_ratio_50'] = df['sma_close_50'] / (df['ema_close_50'] + 1e-7)
    
    # Combine all factors into a single alpha factor
    factor_value = (
        df['normalized_ema_lr_20'] +
        df['vrsi_14'] +
        df['momentum_10'] +
        df['sma_ema_ratio_50']
    )
    
    return factor_value
```
