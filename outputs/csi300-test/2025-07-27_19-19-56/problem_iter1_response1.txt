```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the smoothed moving average of the close price
    df['sma_close'] = df['close'].rolling(window=20).mean()
    # Calculate the 20-day standard deviation of the log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return'] = df['log_return'].rolling(window=20).std()
    # Calculate the factor value
    factor_value = (df['close_diff'] - df['sma_close']) / (df['std_log_return'] + 1e-7)
    return factor_value
```
