```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a percentage of yesterday's close
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate the exponential moving average (EMA) of the daily return with a span of 7 days, to smooth out short-term fluctuations
    ema_return = daily_return.ewm(span=7, adjust=False).mean()
    
    # Calculate the ratio of amount and volume as a proxy for the average transaction price
    avg_price_proxy = df['amount'] / df['volume']
    
    # Calculate the difference between the average price proxy and the close price, to capture the deviation from the actual close
    price_deviation = avg_price_proxy - df['close']
    
    # Final alpha factor is the smoothed daily return multiplied by the negative of the price deviation,
    # implying if the price deviation is positive (proxy > close), it negatively impacts the expected future return, and vice versa.
    alpha_factor = ema_return * (-price_deviation)
    
    return alpha_factor
```
