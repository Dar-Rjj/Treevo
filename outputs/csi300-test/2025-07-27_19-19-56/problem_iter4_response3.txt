```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum (difference between today's close and the close 5 days ago)
    momentum_5 = df['close'] - df['close'].shift(5)
    
    # Calculate the 3-day rate of change for the closing price
    roc_3 = (df['close'] - df['close'].shift(3)) / df['close'].shift(3)
    
    # Calculate the 10-day standard deviation of the log returns to measure volatility
    log_return = np.log(df['close'] / df['close'].shift(1))
    volatility_10 = log_return.rolling(window=10).std()
    
    # Calculate the 5-day moving average of volume to measure liquidity
    volume_ma_5 = df['volume'].rolling(window=5).mean()
    
    # Alpha factor: combine momentum, rate of change, volatility, and liquidity
    factor_values = (momentum_5 * roc_3) / (volatility_10 + 1e-7) * volume_ma_5
    
    return factor_values
```
