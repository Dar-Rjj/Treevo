```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 10-day and 30-day EMA of log returns
    df['ema_log_return_10'] = df['log_return'].ewm(span=10, adjust=False).mean()
    df['ema_log_return_30'] = df['log_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate the standard deviation of the last 20 days log returns for normalization
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Calculate 5-day simple moving average of volume
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate 10-day and 30-day simple moving average of close price
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    df['sma_close_30'] = df['close'].rolling(window=30).mean()
    
    # Trend strength as the ratio between short-term and long-term SMA
    df['trend_strength'] = (df['sma_close_10'] - df['sma_close_30']) / df['sma_close_30']
    
    # RSI calculation
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    df['rsi'] = rsi
    
    # Momentum as the percentage change in closing price over the past 20 days
    df['momentum_20'] = df['close'].pct_change(20)
    
    # Volatility as the standard deviation of log returns
    df['volatility_20'] = df['log_return'].rolling(window=20).std()
    
    # Factor value integrating multiple components
    factor_value = (
        (df['ema_log_return_10'] - df['ema_log_return_30']) * df['volume_ma_5'] / (df['std_log_return_20'] + 1e-7) +
        df['trend_strength'] +
        (df['rsi'] - 50) / 50 +
        df['momentum_20'] -
        df['volatility_20']
    )
    
    return factor_value
```
