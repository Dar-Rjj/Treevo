```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close for momentum
    df['close_diff'] = df['close'].diff()
    
    # Calculate the 20-day and 50-day simple moving averages of the close price for short and long-term trends
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    df['sma_close_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the 20-day standard deviation of the log returns to measure volatility
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Calculate the 5-day moving average of volume for weighting
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    
    # Incorporate the high-low range as a measure of intraday volatility
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate the factor value, focusing on key trends, momentum, and using volume for weighting
    factor_value = (df['close_diff'] - (df['sma_close_20'] - df['sma_close_50'])) * df['volume_ma_5'] / (df['std_log_return_20'] + 1e-7) * df['high_low_range']
    
    return factor_value
```
