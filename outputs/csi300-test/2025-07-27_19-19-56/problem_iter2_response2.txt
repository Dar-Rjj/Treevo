```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day simple moving averages for close prices to assess short-term vs. long-term trends
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Compute the difference between 5-day and 20-day moving averages, which can signal a potential trend reversal
    trend_difference = sma_5 - sma_20
    
    # Calculate the 14-day Average True Range (ATR) to measure market volatility
    true_range = df[['high', 'low']].diff(axis=1).max(axis=1)
    atr_14 = true_range.rolling(window=14).mean()
    
    # Calculate the 30-day moving average of volume to incorporate longer-term volume trends
    volume_ma_30 = df['volume'].rolling(window=30).mean()
    
    # Final alpha factor combines the trend difference, scaled by ATR and volume MA, to capture both trend strength and investor interest
    alpha_factor = trend_difference * (atr_14 / volume_ma_30)
    
    return alpha_factor
```
