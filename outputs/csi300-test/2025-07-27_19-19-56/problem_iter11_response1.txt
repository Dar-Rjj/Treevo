```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the EMA of the close price with a 10-day period
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    # Calculate the RSI of the close price with a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(com=13, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(com=13, adjust=False).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the 60-day momentum of the close price
    df['momentum_60'] = df['close'] / df['close'].shift(60)
    
    # Volume-weighted close price
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Calculate the 20-day standard deviation of the log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_return_20'] = df['log_return'].rolling(window=20).std()
    
    # Combine the factors into a single alpha factor
    factor_value = (df['momentum_60'] - df['rsi_14']) * (df['volume_weighted_close'] / (df['std_log_return_20'] + 1e-7))
    return factor_value
```
