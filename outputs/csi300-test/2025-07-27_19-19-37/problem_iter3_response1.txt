```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the close price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the percentage change in the close price over the last 5 days
    pct_change_5 = df['close'].pct_change(periods=5)
    
    # Generate the alpha factor by scaling the close-open difference with the price factor, ATR, and including momentum and trend
    alpha_factor = (close_open_diff * price_factor * (ma_5 - ma_20) * pct_change_5) / (atr + 1e-7)
    
    return alpha_factor
```
