```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day exponential moving average of close
    long_term_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the relative strength index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (delta.where(delta < 0, 0).abs()).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust the alpha factor based on the RSI
    rsi_adjusted_factor = rsi / 50
    
    # Calculate the money flow index (MFI) with a 14-day window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Adjust the alpha factor based on the MFI
    mfi_adjusted_factor = mfi / 50
    
    # Calculate the Stochastic Oscillator with a 14-day window
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    d_percent = k_percent.rolling(window=3).mean()
    
    # Adjust the alpha factor based on the Stochastic Oscillator
    stochastic_adjusted_factor = (k_percent + d_percent) / 100
    
    # Calculate the Williams %R with a 14-day window
    williams_r = -100 * (highest_high - df['close']) / (highest_high - lowest_low)
    
    # Adjust the alpha factor based on the Williams %R
    williams_r_adjusted_factor = (williams_r + 100) / 100
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, and dynamic weighted moving average, adjusted for RSI, MFI, Stochastic Oscillator, and Williams %R
    alpha_factor = ((long_term_momentum + short_term_volatility) / (weighted_close + 1e-7)) * rsi_adjusted_factor * mfi_adjusted_factor * stochastic_adjusted_factor * williams_r_adjusted_factor
    
    return alpha_factor
```
