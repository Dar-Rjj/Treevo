def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum using 20-day exponential moving average
    long_term_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (delta.where(delta < 0, 0).abs()).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust the alpha factor based on the RSI
    rsi_adjusted_factor = rsi / 50
    
    # Calculate the Money Flow Index (MFI) with a 14-day window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Adjust the alpha factor based on the MFI
    mfi_adjusted_factor = mfi / 50
    
    # Calculate the Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = adl * df['volume']
    adl = adl.cumsum()
    
    # Normalize ADL for use in the alpha factor
    adl_normalized = (adl - adl.rolling(window=20).min()) / (adl.rolling(window=20).max() - adl.rolling(window=20).min())
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, and dynamic weighted moving average, adjusted for RSI, MFI, and ADL
    alpha_factor = ((long_term_momentum + short_term_volatility) / (weighted_close + 1e-7)) * rsi_adjusted_factor * mfi_adjusted_factor * adl_normalized
    
    return alpha_factor
