def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a percentage of yesterday's close
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate the ratio of today's volume to yesterday's volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Create an alpha factor by combining daily return and volume ratio
    alpha_factor = daily_return * volume_ratio
    
    # Adjust the alpha factor with momentum signal: if the price is increasing, give more weight to upward momentum
    trend_signal = df['close'] > df['close'].shift(5)
    alpha_factor_adjusted = alpha_factor * (1 + 0.5 * trend_signal)
    
    return alpha_factor_adjusted
