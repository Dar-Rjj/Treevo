```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day moving average of close
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Generate the alpha factor by combining the long-term momentum and short-term volatility
    alpha_factor = (long_term_momentum + short_term_volatility) / (df['close'] + 1e-7)
    
    return alpha_factor
```
