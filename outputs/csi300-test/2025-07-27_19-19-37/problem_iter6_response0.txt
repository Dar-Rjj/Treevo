```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a liquidity factor
    liquidity_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days for volatility
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the momentum as the 10-day and 30-day rate of change in closing prices
    momentum_10d = df['close'].pct_change(periods=10).fillna(0)
    momentum_30d = df['close'].pct_change(periods=30).fillna(0)
    
    # Calculate the 10-day and 30-day moving averages
    ma_10d = df['close'].rolling(window=10).mean()
    ma_30d = df['close'].rolling(window=30).mean()
    
    # Calculate the 10-day and 30-day moving average convergence divergence (MACD)
    macd_line = ma_10d - ma_30d
    signal_line = macd_line.rolling(window=9).mean()
    
    # Calculate the MACD histogram
    macd_histogram = macd_line - signal_line
    
    # Generate the alpha factor by scaling the close-open difference with the liquidity factor, ATR, and momentum, and MACD histogram
    alpha_factor = (close_open_diff * liquidity_factor * (momentum_10d + momentum_30d) * macd_histogram) / (atr + 1e-7)
    
    return alpha_factor
```
