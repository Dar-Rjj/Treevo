```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Dynamic weighted moving averages for close and open prices
    short_window = 5
    long_window = 20
    close_ma_short = df['close'].rolling(window=short_window).mean()
    close_ma_long = df['close'].rolling(window=long_window).mean()
    open_ma_short = df['open'].rolling(window=short_window).mean()
    open_ma_long = df['open'].rolling(window=long_window).mean()
    
    # Calculate the dynamic weighted moving average difference
    ma_diff_close = (close_ma_short - close_ma_long) / (close_ma_long + 1e-7)
    ma_diff_open = (open_ma_short - open_ma_long) / (open_ma_long + 1e-7)
    
    # Momentum calculation
    momentum = df['close'] - df['close'].shift(long_window)
    
    # Volatility calculation using standard deviation
    volatility = df['close'].rolling(window=long_window).std()
    
    # Relative strength index (RSI) calculation
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Generate the alpha factor by incorporating momentum, volatility, and RSI
    alpha_factor = (close_open_diff * price_factor * (ma_diff_close - ma_diff_open) * momentum / (volatility + 1e-7) * rsi) / (atr + 1e-7)
    
    return alpha_factor
```
