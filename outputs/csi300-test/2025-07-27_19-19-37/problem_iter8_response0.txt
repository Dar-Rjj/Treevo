```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Generate the alpha factor by scaling the close-open difference with the price factor
    alpha_factor = close_open_diff * price_factor
    
    return alpha_factor
```
