```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factors over different timeframes
    momentum_5 = df['close'] / df['close'].shift(5) - 1
    momentum_10 = df['close'] / df['close'].shift(10) - 1
    momentum_20 = df['close'] / df['close'].shift(20) - 1
    momentum_60 = df['close'] / df['close'].shift(60) - 1

    # Exponential Moving Averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_60 = df['close'].ewm(span=60, adjust=False).mean()

    # Volatility as standard deviation of returns over different timeframes
    vol_5 = df['close'].pct_change().rolling(window=5).std()
    vol_10 = df['close'].pct_change().rolling(window=10).std()
    vol_20 = df['close'].pct_change().rolling(window=20).std()
    vol_60 = df['close'].pct_change().rolling(window=60).std()

    # Volume trends
    volume_trend_5 = df['volume'] / df['volume'].rolling(window=5).mean() - 1
    volume_trend_10 = df['volume'] / df['volume'].rolling(window=10).mean() - 1

    # Ratio of amount to volume
    price_factor = df['amount'] / (df['volume'] + 1e-7)

    # Combining factors
    alpha_factor = (
        (momentum_5 + momentum_10 + momentum_20 + momentum_60) +
        (ema_5 + ema_10 + ema_20 + ema_60) +
        (volume_trend_5 + volume_trend_10)
    ) / (vol_5 + vol_10 + vol_20 + vol_60 + 1e-7) * price_factor

    return alpha_factor
```
