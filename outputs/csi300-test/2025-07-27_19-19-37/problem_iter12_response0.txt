```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 50-day moving average of close
    long_term_momentum = df['close'] - df['close'].rolling(window=50).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 10 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=10).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the average true range (ATR) over the last 14 days for additional volatility factor
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr_14 = true_range.rolling(window=14).mean()
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, ATR, and dynamic weighted moving average
    alpha_factor = (long_term_momentum + short_term_volatility + atr_14) / (weighted_close + 1e-7)
    
    return alpha_factor
```
