```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr_14 = true_range.rolling(window=14).mean()
    
    # Calculate the ATR over the last 5 days for short-term volatility
    atr_5 = true_range.rolling(window=5).mean()
    
    # Combine the ATRs with a weighted average
    combined_atr = (0.7 * atr_14) + (0.3 * atr_5)
    
    # Generate the alpha factor by scaling the close-open difference with the price factor and combined ATR
    alpha_factor = (close_open_diff * price_factor) / (combined_atr + 1e-7)
    
    return alpha_factor
```
