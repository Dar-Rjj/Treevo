```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 30-day moving average of close
    long_term_momentum = df['close'] - df['close'].rolling(window=30).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 7 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=7).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the relative strength index (RSI) over the last 14 days for an additional technical indicator
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rsi_14 = 100 - (100 / (1 + (gain / loss)))
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, RSI, and dynamic weighted moving average
    alpha_factor = (long_term_momentum + short_term_volatility + rsi_14) / (weighted_close + 1e-7)
    
    return alpha_factor
```
