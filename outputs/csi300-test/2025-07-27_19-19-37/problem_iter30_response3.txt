```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day exponential moving average of close
    long_term_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the relative strength index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (delta.where(delta < 0, 0).abs()).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust the alpha factor based on the RSI
    rsi_adjusted_factor = (rsi - 50) / 50  # Center RSI around 0
    
    # Calculate the volume-weighted close
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate the amount-weighted close
    amount_weighted_close = (df['close'] * df['amount']).rolling(window=20).sum() / df['amount'].rolling(window=20).sum()
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, dynamic weighted moving average,
    # volume-weighted close, and amount-weighted close, adjusted for RSI
    alpha_factor = ((long_term_momentum + short_term_volatility) / (weighted_close + 1e-7)) * rsi_adjusted_factor * (volume_weighted_close / amount_weighted_close)
    
    return alpha_factor
```
