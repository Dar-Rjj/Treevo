```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change in close price from 10 days ago to capture a longer trend
    close_change_10d = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate the ratio of today's volume to the average volume over the last 10 days
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio_10d = df['volume'] / avg_volume_10d
    
    # Create an alpha factor by combining the 10-day close change and the volume ratio
    alpha_factor = close_change_10d * volume_ratio_10d
    
    # Incorporate a trend signal: if the 30-day moving average is increasing, give more weight to the alpha factor
    ma_30 = df['close'].rolling(window=30).mean()
    trend_signal = ma_30 > ma_30.shift(1)
    alpha_factor_adjusted = alpha_factor * (1 + 0.5 * trend_signal)
    
    # Add a momentum component: if the 10-day return is positive, increase the factor
    momentum_signal = (df['close'] - df['close'].shift(10)) > 0
    alpha_factor_final = alpha_factor_adjusted * (1 + 0.3 * momentum_signal)
    
    return alpha_factor_final
```
