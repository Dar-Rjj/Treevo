```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change in close price from 5 days ago
    close_change_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate the ratio of today's volume to the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_ratio_5d = df['volume'] / avg_volume_5d
    
    # Create an alpha factor by combining the 5-day close change and the volume ratio
    alpha_factor = close_change_5d * volume_ratio_5d
    
    # Incorporate a trend signal: if the 20-day moving average is increasing, give more weight to the alpha factor
    ma_20 = df['close'].rolling(window=20).mean()
    trend_signal = ma_20 > ma_20.shift(1)
    alpha_factor_adjusted = alpha_factor * (1 + 0.5 * trend_signal)
    
    return alpha_factor_adjusted
```
