```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the difference to the average
    ratio = diff_close_open / (avg_high_low + 1e-7)
    
    # Calculate the change in volume
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the 5-day moving average of the closing price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 50-day moving average of the closing price
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the volatility as the standard deviation of the last 5 days' closing prices
    volatility = df['close'].rolling(window=5).std()
    
    # Integrate external economic indicator (as a placeholder, using 'amount' here)
    economic_indicator = df['amount'].pct_change().fillna(0)
    
    # Combine the ratio, volume change, moving averages, volatility, and economic indicator
    factor = (ratio * volume_change * (ma_5 - ma_20) / (ma_50 + 1e-7)) / (volatility + 1e-7) * economic_indicator
    
    return factor
```
