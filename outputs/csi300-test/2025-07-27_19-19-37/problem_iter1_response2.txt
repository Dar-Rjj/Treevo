```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a simple momentum
    momentum = df['close'].diff(1)
    
    # Calculate the average of the open, high, low, and close for each day
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the ratio of volume to amount as a measure of price per unit of volume
    price_per_volume = df['amount'] / (df['volume'] + 1e-7)
    
    # Combine the features into a single factor
    factor = (momentum * avg_price) / (price_per_volume + 1e-7)
    
    return factor
```
