```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day exponential moving average of close
    long_term_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the relative strength index (RSI) with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(com=13, adjust=False).mean()
    loss = (delta.where(delta < 0, 0).abs()).ewm(com=13, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust the RSI for linear normalization
    rsi_normalized = (rsi - 50) / 50
    
    # Incorporate volume trends for robustness
    volume_trend = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, dynamic weighted moving average, and volume trend, adjusted for RSI
    alpha_factor = ((long_term_momentum + short_term_volatility) / (weighted_close + 1e-7)) * rsi_normalized * volume_trend
    
    return alpha_factor
```
