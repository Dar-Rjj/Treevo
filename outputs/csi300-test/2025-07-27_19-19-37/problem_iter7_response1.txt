```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Dynamic weighted moving averages for close and open prices
    close_ma_5 = df['close'].rolling(window=5).mean()
    close_ma_10 = df['close'].rolling(window=10).mean()
    close_ma_20 = df['close'].rolling(window=20).mean()
    dynamic_close_ma = (close_ma_5 * 0.6 + close_ma_10 * 0.3 + close_ma_20 * 0.1)
    
    open_ma_5 = df['open'].rolling(window=5).mean()
    open_ma_10 = df['open'].rolling(window=10).mean()
    open_ma_20 = df['open'].rolling(window=20).mean()
    dynamic_open_ma = (open_ma_5 * 0.6 + open_ma_10 * 0.3 + open_ma_20 * 0.1)
    
    # Calculate the difference between the dynamic moving averages of close and open prices
    dynamic_diff = dynamic_close_ma - dynamic_open_ma
    
    # Generate the alpha factor by scaling the dynamic difference with the price factor and ATR
    alpha_factor = (dynamic_diff * price_factor) / (atr + 1e-7)
    
    return alpha_factor
```
