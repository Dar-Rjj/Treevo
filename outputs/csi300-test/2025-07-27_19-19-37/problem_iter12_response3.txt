```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day moving average of close
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the ratio of amount to volume as a liquidity factor
    liquidity_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the price range as the difference between high and low
    price_range = df['high'] - df['low']
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, dynamic weighted moving average, and liquidity factor
    alpha_factor = (long_term_momentum + short_term_volatility + price_range) / (weighted_close + liquidity_factor + 1e-7)
    
    return alpha_factor
```
