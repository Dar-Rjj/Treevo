def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 10-day moving average of close
    long_term_momentum = df['close'] - df['close'].rolling(window=10).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 3 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=3).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 11)
    weighted_close = (df['close'].rolling(window=10).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, and dynamic weighted moving average
    alpha_factor = (long_term_momentum + short_term_volatility) / (weighted_close + 1e-7)
    
    return alpha_factor
