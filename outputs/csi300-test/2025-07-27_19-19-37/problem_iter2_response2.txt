```txt
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change().fillna(0)
    
    # Calculate the percentage change in open price
    pct_change_open = df['open'].pct_change().fillna(0)
    
    # Calculate the difference between the percentage changes of close and open prices
    diff_pct_change = pct_change_close - pct_change_open
    
    # Calculate the average of high and low prices
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the difference to the average
    ratio = diff_pct_change / (avg_high_low + 1e-7)
    
    # Calculate the change in volume
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Calculate the change in amount
    amount_change = df['amount'].pct_change().fillna(0)
    
    # Combine the ratio with the volume and amount changes
    factor = ratio * volume_change * amount_change
    
    return factor
```
