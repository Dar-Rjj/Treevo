```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr_14 = true_range.rolling(window=14).mean()
    
    # Calculate momentum over the last 5 days
    momentum_5d = df['close'].pct_change(periods=5)
    
    # Calculate dynamic weighted moving averages over the last 10 and 20 days
    wma_10d = df['close'].ewm(span=10, adjust=False).mean()
    wma_20d = df['close'].ewm(span=20, adjust=False).mean()
    dynamic_wma = (wma_10d * 0.6 + wma_20d * 0.4)
    
    # Calculate a long-term weighted moving average for normalization
    wma_30d = df['close'].ewm(span=30, adjust=False).mean()
    
    # Generate the alpha factor by combining the close-open difference, price factor, ATR, momentum, and dynamic weighted moving average
    alpha_factor = (close_open_diff * price_factor) / (atr_14 + 1e-7) * (momentum_5d + 1) * (dynamic_wma / wma_30d)
    
    return alpha_factor
```
