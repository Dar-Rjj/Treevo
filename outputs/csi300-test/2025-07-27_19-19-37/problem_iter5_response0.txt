```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr_14 = true_range.rolling(window=14).mean()
    atr_28 = true_range.rolling(window=28).mean()
    
    # Generate the alpha factor by scaling the close-open difference with the price factor and ATR
    alpha_factor = (close_open_diff * price_factor) / (atr_14 + 1e-7)
    
    # Incorporate a momentum factor using the 28-day ATR
    momentum_factor = (df['close'] - df['close'].shift(28)) / (atr_28 + 1e-7)
    
    # Combine the factors with dynamic weighting
    combined_factor = 0.5 * alpha_factor + 0.5 * momentum_factor
    
    return combined_factor
```
