```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor over different timeframes
    momentum_10 = df['close'] / df['close'].shift(10) - 1
    momentum_20 = df['close'] / df['close'].shift(20) - 1
    momentum_60 = df['close'] / df['close'].shift(60) - 1
    
    # Weighted moving averages
    wma_5 = (df['close'] * 5 + df['close'].shift(1) * 4 + df['close'].shift(2) * 3 + df['close'].shift(3) * 2 + df['close'].shift(4)) / 15
    wma_10 = (df['close'] * 10 + df['close'].shift(1) * 9 + df['close'].shift(2) * 8 + df['close'].shift(3) * 7 + df['close'].shift(4) * 6 + 
              df['close'].shift(5) * 5 + df['close'].shift(6) * 4 + df['close'].shift(7) * 3 + df['close'].shift(8) * 2 + df['close'].shift(9)) / 55

    # Volatility as standard deviation of returns
    vol_10 = df['close'].pct_change().rolling(window=10).std()
    vol_20 = df['close'].pct_change().rolling(window=20).std()

    # Ratio of amount to volume
    price_factor = df['amount'] / (df['volume'] + 1e-7)

    # Combining factors
    alpha_factor = (momentum_10 + momentum_20 + momentum_60 + wma_5 + wma_10) / (vol_10 + vol_20 + 1e-7) * price_factor

    return alpha_factor
```
