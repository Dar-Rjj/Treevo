```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the true range over the last 5 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr_5 = true_range.rolling(window=5).mean()
    
    # Calculate the 5-day and 10-day moving averages of the close price
    ma5 = df['close'].rolling(window=5).mean()
    ma10 = df['close'].rolling(window=10).mean()
    
    # Generate the alpha factor by scaling the close-open difference with the price factor and ATR
    alpha_factor = (close_open_diff * price_factor) / (atr_5 + 1e-7)
    
    # Incorporate relative strength by adding the difference between the 5-day and 10-day moving averages
    alpha_factor += (ma5 - ma10)
    
    return alpha_factor
```
