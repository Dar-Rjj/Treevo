```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day moving average of close
    long_term_momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the volume trend as the 10-day moving average of volume
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Calculate the RSI (Relative Strength Index) with a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, dynamic weighted moving average, volume trend, and RSI
    alpha_factor = (long_term_momentum + short_term_volatility + volume_trend + rsi) / (weighted_close + 1e-7)
    
    return alpha_factor
```
