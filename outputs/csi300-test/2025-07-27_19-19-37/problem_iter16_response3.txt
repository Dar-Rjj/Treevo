```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day exponential moving average of close
    long_term_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Calculate the relative strength index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (delta.where(delta < 0, 0).abs()).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust the alpha factor based on the RSI
    rsi_adjusted_factor = rsi / 50
    
    # Incorporate volume by calculating the volume-weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).cumsum() / (df['volume']).cumsum()
    
    # Calculate the adaptive window for the moving average based on the recent volatility
    adaptive_window = 5 + (short_term_volatility * 10).astype(int)
    adaptive_window = adaptive_window.clip(lower=5, upper=20)
    
    # Calculate the adaptive moving average
    adaptive_ma = df['close'].rolling(window=adaptive_window, min_periods=1).mean()
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, dynamic weighted moving average, VWAP, and adaptive moving average, adjusted for RSI
    alpha_factor = ((long_term_momentum + short_term_volatility) / (weighted_close + 1e-7)) * rsi_adjusted_factor * (vwap / adaptive_ma)
    
    return alpha_factor
```
