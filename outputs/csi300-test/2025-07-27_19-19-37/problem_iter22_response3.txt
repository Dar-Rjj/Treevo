def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the difference between current close and 20-day exponential moving average of close
    long_term_momentum = df['close'] - df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the short-term volatility as the standard deviation of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).std()
    
    # Calculate a dynamic weighted moving average, giving more weight to recent data
    weights = np.arange(1, 21)
    weighted_close = (df['close'].rolling(window=20).apply(lambda x: (x * weights).sum() / weights.sum(), raw=False))
    
    # Incorporate trading volume by calculating the 20-day moving average of volume
    volume_20_day_ma = df['volume'].rolling(window=20).mean()
    volume_factor = df['volume'] / volume_20_day_ma
    
    # Calculate market breadth indicators: number of advancing issues minus declining issues
    advancing_issues = (df['close'] > df['open']).astype(int)
    declining_issues = (df['close'] < df['open']).astype(int)
    market_breadth = advancing_issues - declining_issues
    market_breadth_ma = market_breadth.rolling(window=20).mean()
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, dynamic weighted moving average, and market breadth
    alpha_factor = ((long_term_momentum + short_term_volatility + volume_factor) / (weighted_close + 1e-7)) * market_breadth_ma
    
    return alpha_factor
