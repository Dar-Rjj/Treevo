```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the long-term momentum as the 20-day rate of change (ROC)
    long_term_momentum = df['close'] / df['close'].shift(20) - 1
    
    # Calculate the short-term volatility as the absolute value of log returns over the last 5 days
    log_returns = np.log(df['close'] / df['close'].shift(1))
    short_term_volatility = log_returns.rolling(window=5).apply(lambda x: np.abs(x).sum(), raw=True)
    
    # Calculate a simple moving average (SMA) for the close price over the last 20 days
    sma_close = df['close'].rolling(window=20).mean()
    
    # Calculate the relative strength index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (delta.where(delta < 0, 0).abs()).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust the alpha factor based on the RSI
    rsi_adjusted_factor = (rsi - 50) / 50  # Center RSI around 0
    
    # Generate the alpha factor by combining the long-term momentum, short-term volatility, and simple moving average, adjusted for RSI
    alpha_factor = ((long_term_momentum + short_term_volatility) / (sma_close + 1e-7)) * rsi_adjusted_factor
    
    return alpha_factor
```
