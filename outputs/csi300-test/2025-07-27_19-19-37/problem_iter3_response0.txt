```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the ratio of amount to volume as a price factor
    price_factor = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate the average true range (ATR) over the last 14 days
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Calculate the 50-day moving average of the closing price
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the percentage change in the 50-day moving average
    ma_50_pct_change = ma_50.pct_change()
    
    # Generate the alpha factor by combining the close-open difference, price factor, ATR, and 50-day MA percentage change
    alpha_factor = (close_open_diff * price_factor * ma_50_pct_change) / (atr + 1e-7)
    
    return alpha_factor
```
