def heuristics_v2(df):
    # Rate of change in adjusted close price over the last 10 days
    roc_close = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Volume relative to the 30-day average volume
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    rel_vol = df['volume'] / avg_vol_30d
    
    # Standard deviation of the closing price over the last 50 days
    std_50d = df['close'].rolling(window=50).std()
    
    # Combine factors
    heuristics_matrix = roc_close + 0.4 * rel_vol + 0.6 * std_50d
    
    return heuristics_matrix
