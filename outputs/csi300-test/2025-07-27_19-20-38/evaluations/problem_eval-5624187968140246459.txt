def heuristics_v2(df):
    # Momentum: The rate of change in price over the last 10 days
    momentum = df['close'].pct_change(10)
    
    # Adjusted Relative Strength: The ratio of the current close price to the 50-day simple moving average, adjusted by the 20-day standard deviation
    sma_50 = df['close'].rolling(window=50).mean()
    std_20 = df['close'].rolling(window=20).std()
    rel_strength_adj = (df['close'] / sma_50) - std_20
    
    # Volume Burst: Current volume divided by the 30-day moving average volume
    ma_volume_30 = df['volume'].rolling(window=30).mean()
    volume_burst = df['volume'] / ma_volume_30
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.6 * momentum + 0.3 * rel_strength_adj + 0.1 * volume_burst
    
    return heuristics_matrix
