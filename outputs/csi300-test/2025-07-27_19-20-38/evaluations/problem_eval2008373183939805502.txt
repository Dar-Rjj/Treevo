def heuristics_v2(df):
    # Momentum: Rate of change in closing price over the last 20 days
    momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Amount relative to the 60-day average amount
    avg_amount_60d = df['amount'].rolling(window=60).mean()
    rel_amount = df['amount'] / avg_amount_60d
    
    # RSI calculation over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine factors
    heuristics_matrix = momentum + 0.5 * rel_amount + 0.5 * rsi
    
    return heuristics_matrix
