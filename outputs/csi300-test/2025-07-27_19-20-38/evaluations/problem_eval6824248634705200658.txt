def heuristics_v2(df):
    # Rate of change in price over the last 20 days
    roc_price = df['close'].pct_change(20)
    
    # Relative Strength: The ratio of the current close price to the 50-day simple moving average
    sma_50 = df['close'].rolling(window=50).mean()
    rel_strength = df['close'] / sma_50
    
    # Momentum: Difference between the current close price and the 50-day high
    high_50 = df['high'].rolling(window=50).max()
    momentum = df['close'] - high_50
    
    # Volume Ratio: Current volume divided by the 50-day moving average volume
    ma_volume_50 = df['volume'].rolling(window=50).mean()
    volume_ratio = df['volume'] / ma_volume_50
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * roc_price + 0.4 * rel_strength + 0.1 * momentum + 0.2 * volume_ratio
    
    return heuristics_matrix
