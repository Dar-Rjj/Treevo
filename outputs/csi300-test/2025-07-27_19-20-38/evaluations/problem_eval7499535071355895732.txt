import pandas as pd

def heuristics_v2(df):
    # Relative Strength: (Close - 20D Low) / (20D High - 20D Low)
    high_20d = df['high'].rolling(window=20).max()
    low_20d = df['low'].rolling(window=20).min()
    relative_strength = (df['close'] - low_20d) / (high_20d - low_20d)
    
    # Volatility: Standard deviation of daily returns over the last 20 days
    volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Volume Change: Percentage change in volume
    vol_change = df['volume'].pct_change()
    
    # Combine factors
    heuristics_matrix = relative_strength - 0.5 * volatility + 0.3 * vol_change
    
    return heuristics_matrix
