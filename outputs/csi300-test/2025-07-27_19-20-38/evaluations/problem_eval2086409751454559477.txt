def heuristics_v2(df):
    # Difference between the highest and lowest price over 5 days
    diff_high_low = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Simple Moving Average of closing prices over 10 days
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Rate of Change of Volume over 20 days
    roc_volume = (df['volume'] - df['volume'].shift(20)) / df['volume'].shift(20)
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * diff_high_low + 0.4 * (df['close'] / sma_10) + 0.3 * roc_volume
    
    return heuristics_matrix
