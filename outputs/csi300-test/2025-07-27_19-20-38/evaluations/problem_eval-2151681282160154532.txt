def heuristics_v2(df):
    # True Range over 10 days with weights
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    weights = [w / sum(range(1, 11)) for w in range(1, 11)]
    atr_10_weighted = tr.rolling(window=10).apply(lambda x: sum(w * y for w, y in zip(weights, x)), raw=True)
    
    # Exponential Moving Average of closing prices over 30 days
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # RSI of Volume over 14 days
    delta_volume = df['volume'].diff(1)
    gain = delta_volume.where(delta_volume > 0, 0)
    loss = -delta_volume.where(delta_volume < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * atr_10_weighted + 0.4 * (df['close'] / ema_30) + 0.2 * rsi_volume
    
    return heuristics_matrix
