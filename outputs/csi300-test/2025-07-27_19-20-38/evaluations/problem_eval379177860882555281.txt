import pandas as pd

def heuristics_v2(df):
    high = df['high']
    low = df['low']
    close = df['close']
    
    price_range = (high.rolling(window=10).max() - low.rolling(window=10).min()) / close
    trend_persistence = close.diff().cumsum()
    
    heuristics_matrix = (price_range + trend_persistence) / 2
    
    return heuristics_matrix
