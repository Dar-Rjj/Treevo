import pandas as pd

def heuristics_v2(df):
    def calculate_weighted_moving_average(series, weights):
        return (series * weights).sum() / weights.sum()
    
    heuristics_matrix = pd.Series(index=df.index)
    for i in range(len(df)):
        if i < 10:
            heuristics_matrix.iloc[i] = 0
            continue
        price_changes = df['close'].iloc[i-10:i] - df['close'].iloc[i-11:i-1]
        volume_changes = df['volume'].iloc[i-10:i] - df['volume'].iloc[i-11:i-1]
        weights = pd.Series([1/(j+1) for j in range(10)], index=price_changes.index)
        wma_price = calculate_weighted_moving_average(price_changes, weights)
        wma_volume = calculate_weighted_moving_average(volume_changes, weights)
        heuristics_matrix.iloc[i] = wma_price + wma_volume
    return heuristics_matrix
