def heuristics_v2(df):
    # Normalized Average True Range over 14 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_14 = (tr.rolling(window=14).mean() / df['close'].rolling(window=14).mean()).fillna(0)
    
    # Simple Moving Average of closing prices over 20 days
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Rate of Change in Volume over 10 days
    roc_volume = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * atr_14 + 0.5 * (df['close'] / sma_20) + 0.2 * roc_volume
    
    return heuristics_matrix
