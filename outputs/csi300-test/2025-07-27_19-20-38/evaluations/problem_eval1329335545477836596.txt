def heuristics_v2(df):
    # Rate of change in adjusted close price over the last 10 days
    roc_close = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Volume relative to the 30-day average volume
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    rel_vol = df['volume'] / avg_vol_30d
    
    # Divergence: Difference between closing price and 50D EMA
    ema_50d = df['close'].ewm(span=50, adjust=False).mean()
    divergence = df['close'] - ema_50d
    
    # Combine factors
    heuristics_matrix = roc_close + 0.4 * rel_vol + 0.6 * divergence
    
    return heuristics_matrix
