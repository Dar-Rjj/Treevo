def heuristics_v2(df):
    # Price Volatility: Standard deviation of closing prices over 10 days
    price_volatility = df['close'].rolling(window=10).std()
    
    # Volume relative to the 7-day average volume
    avg_vol_7d = df['volume'].rolling(window=7).mean()
    rel_vol = df['volume'] / avg_vol_7d
    
    # Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine factors
    heuristics_matrix = 0.4 * price_volatility + 0.3 * rel_vol + 0.3 * rsi
    
    return heuristics_matrix
