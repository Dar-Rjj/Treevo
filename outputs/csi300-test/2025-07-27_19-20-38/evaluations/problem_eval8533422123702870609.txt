def heuristics_v2(df):
    # Rate of Change over 10 days
    roc_10 = df['close'].pct_change(10)
    
    # Simple Moving Average of (High + Low) / 2 over 25 days
    sma_high_low = ((df['high'] + df['low']) / 2).rolling(window=25).mean()
    
    # Relative Strength Index of Volume over 14 days
    delta_vol = df['volume'].diff(1)
    gain_vol = delta_vol.where(delta_vol > 0, 0)
    loss_vol = -delta_vol.where(delta_vol < 0, 0)
    avg_gain_vol = gain_vol.rolling(window=14).mean()
    avg_loss_vol = loss_vol.rolling(window=14).mean()
    rs_vol = avg_gain_vol / avg_loss_vol
    rsi_vol = 100 - (100 / (1 + rs_vol))
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * roc_10 + 0.4 * (df['close'] / sma_high_low) + 0.3 * rsi_vol
    
    return heuristics_matrix
