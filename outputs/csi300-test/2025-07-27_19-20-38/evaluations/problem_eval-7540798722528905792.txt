def heuristics_v2(df):
    # Average True Range over 10 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_10 = tr.rolling(window=10).mean()
    
    # Simple Moving Average of closing prices over 30 days
    sma_30 = df['close'].rolling(window=30).mean()
    
    # Rate of Change of Volume over 5 days
    roc_volume = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * atr_10 + 0.4 * (df['close'] / sma_30) + 0.2 * roc_volume
    
    return heuristics_matrix
