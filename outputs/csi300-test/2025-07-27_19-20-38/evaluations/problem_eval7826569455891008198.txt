def heuristics_v2(df):
    # Relative Strength: (Close - 50D Low) / (50D High - 50D Low)
    high_50d = df['high'].rolling(window=50).max()
    low_50d = df['low'].rolling(window=50).min()
    relative_strength = (df['close'] - low_50d) / (high_50d - low_50d)
    
    # Volatility: Standard deviation of daily returns over the last 50 days
    volatility = df['close'].pct_change().rolling(window=50).std()
    
    # Momentum: Rate of change in closing price over the last 10 days
    momentum = df['close'].pct_change(periods=10)
    
    # Volume Change: Percentage change in volume
    vol_change = df['volume'].pct_change()
    
    # Combine factors
    heuristics_matrix = relative_strength - 0.7 * volatility + 0.3 * vol_change + 0.5 * momentum
    
    return heuristics_matrix
