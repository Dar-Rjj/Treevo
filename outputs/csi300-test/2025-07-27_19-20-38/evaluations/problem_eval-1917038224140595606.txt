def heuristics_v2(df):
    # Rate of change in adjusted close price over the last 5 days
    roc_close = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Volume relative to the 20-day average volume
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    rel_vol = df['volume'] / avg_vol_20d
    
    # Percentage difference: Difference between closing price and 30D SMA
    sma_30d = df['close'].rolling(window=30).mean()
    pct_diff_sma = (df['close'] - sma_30d) / sma_30d
    
    # Combine factors
    heuristics_matrix = roc_close + 0.5 * rel_vol + 1.0 * pct_diff_sma
    
    return heuristics_matrix
