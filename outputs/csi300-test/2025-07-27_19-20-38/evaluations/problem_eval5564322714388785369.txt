def heuristics_v2(df):
    # Volatility as standard deviation of daily returns over the last 20 days
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Ratio of current amount to the 10-day moving average amount
    avg_amount_10d = df['amount'].rolling(window=10).mean()
    rel_amount = df['amount'] / avg_amount_10d
    
    # Percentage of days in the last 60 where the close is above the 200D MA
    ma_200d = df['close'].rolling(window=200).mean()
    above_ma = (df['close'] > ma_200d).rolling(window=60).mean()
    
    # Combine factors
    heuristics_matrix = volatility + rel_amount + above_ma
    return heuristics_matrix
