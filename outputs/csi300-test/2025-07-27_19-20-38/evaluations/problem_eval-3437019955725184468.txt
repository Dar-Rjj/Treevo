def heuristics_v2(df):
    # Rate of change in adjusted close price over the last 20 days
    roc_close = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volume relative to the 60-day average volume
    avg_vol_60d = df['volume'].rolling(window=60).mean()
    rel_vol = df['volume'] / avg_vol_60d
    
    # Divergence: Difference between closing price and 100D EMA
    ema_100d = df['close'].ewm(span=100, adjust=False).mean()
    divergence = df['close'] - ema_100d
    
    # Combine factors with different weights
    heuristics_matrix = 0.5 * roc_close + 0.3 * rel_vol + 0.2 * divergence
    
    return heuristics_matrix
