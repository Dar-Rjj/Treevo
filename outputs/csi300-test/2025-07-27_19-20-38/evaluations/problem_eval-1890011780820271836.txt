def heuristics_v2(df):
    # Average True Range over 15 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_15 = tr.rolling(window=15).mean()
    
    # Exponential Moving Average of closing prices over 20 days
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * atr_15 + 0.5 * (df['close'] / ema_20) + 0.2 * rsi_14
    
    return heuristics_matrix
