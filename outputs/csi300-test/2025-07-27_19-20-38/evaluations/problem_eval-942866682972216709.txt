def heuristics_v2(df):
    def volatility(column):
        return column.rolling(window=5).std()
    
    def moving_average(column):
        return column.rolling(window=5).mean()
        
    close_volatility = volatility(df['close'])
    volume_moving_average = moving_average(df['volume'])
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    modified_money_flow_index = raw_money_flow / df['volume'].rolling(window=14).sum()
    
    heuristics_matrix = 0.5*close_volatility + 0.3*volume_moving_average + 0.2*modified_money_flow_index
    return heuristics_matrix
