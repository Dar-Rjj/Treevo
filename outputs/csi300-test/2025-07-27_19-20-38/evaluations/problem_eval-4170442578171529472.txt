def heuristics_v2(df):
    # Rate of change in the average of high and low prices over the last 10 days
    avg_high_low = (df['high'] + df['low']) / 2
    roc_avg_high_low = (avg_high_low - avg_high_low.shift(10)) / avg_high_low.shift(10)
    
    # Difference between the current close price and its 200-day simple moving average
    sma_200d = df['close'].rolling(window=200).mean()
    diff_close_sma = df['close'] - sma_200d
    
    # Volume relative to the 60-day average volume
    avg_vol_60d = df['volume'].rolling(window=60).mean()
    rel_vol = df['volume'] / avg_vol_60d
    
    # Combine factors
    heuristics_matrix = 0.5 * roc_avg_high_low + 0.3 * diff_close_sma + 0.2 * rel_vol
    
    return heuristics_matrix
