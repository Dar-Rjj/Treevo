def heuristics_v2(df):
    # Price-to-High Ratio: The current close price divided by the highest price in the last 50 days
    high_50 = df['high'].rolling(window=50).max()
    price_to_high = df['close'] / high_50

    # Volume Growth: The rate of change in volume over the last 20 days
    volume_growth = df['volume'].pct_change(20)

    # Accumulation/Distribution Line: A cumulative measure of buying and selling pressure
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line = money_flow_volume.cumsum()

    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * price_to_high + 0.3 * volume_growth + 0.3 * ad_line

    return heuristics_matrix
