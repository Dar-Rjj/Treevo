def heuristics_v2(df):
    # Relative Strength Index over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Simple Moving Average of closing prices over 50 days
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Rate of Change of Volume over 10 days
    roc_volume = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10) * 100
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * rsi_14 + 0.5 * (df['close'] / sma_50) + 0.2 * roc_volume
    
    return heuristics_matrix
