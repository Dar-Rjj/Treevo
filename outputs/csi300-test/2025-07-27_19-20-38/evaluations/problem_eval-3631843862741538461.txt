def heuristics_v2(df):
    # Logarithmic Rate of Change in price over the last 20 days
    log_roc_price = (df['close'].shift(-20) / df['close']).apply(np.log)
    
    # Volatility: 20-day standard deviation of closing prices
    std_20 = df['close'].rolling(window=20).std()
    
    # Volume Ratio: Current volume divided by the 50-day moving average volume
    ma_volume_50 = df['volume'].rolling(window=50).mean()
    volume_ratio = df['volume'] / ma_volume_50
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.5 * log_roc_price + 0.3 * std_20 + 0.2 * volume_ratio
    
    return heuristics_matrix
