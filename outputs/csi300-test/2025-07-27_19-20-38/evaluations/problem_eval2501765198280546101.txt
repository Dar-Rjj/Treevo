import pandas as pd

def heuristics_v3(df):
    close = df['close']
    change = close.pct_change()
    positive_changes = change.apply(lambda x: x if x > 0 else 0)
    negative_changes = change.apply(lambda x: x if x < 0 else 0)
    avg_gain = positive_changes.rolling(window=14).mean()
    avg_loss = negative_changes.abs().rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    pos_neg_ratio = positive_changes.sum() / negative_changes.abs().sum()
    heuristics_matrix = (rsi * pos_neg_ratio).fillna(0)
    return heuristics_matrix
