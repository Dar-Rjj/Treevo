def heuristics_v2(df):
    # Exponential Moving Average Crossover: Difference between 10-day EMA and 30-day EMA
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_crossover = ema_10 - ema_30
    
    # Volatility: Standard deviation of log returns over a 30-day window
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=30).std()
    
    # Accumulation/Distribution Line
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl *= df['volume']
    adl = adl.cumsum()
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * ema_crossover + 0.4 * volatility + 0.2 * adl
    
    return heuristics_matrix
