def heuristics_v2(df):
    # Relative Strength: (Close - 20D Low) / (20D High - 20D Low)
    high_20d = df['high'].rolling(window=20).max()
    low_20d = df['low'].rolling(window=20).min()
    relative_strength = (df['close'] - low_20d) / (high_20d - low_20d)
    
    # Adjusted Volatility: Standard deviation of daily returns over the last 20 days, adjusted by the relative strength
    volatility = df['close'].pct_change().rolling(window=20).std() * (1 + relative_strength)
    
    # Volume Change: Logarithmic percentage change in volume
    vol_change_log = df['volume'].pct_change().apply(lambda x: np.log(1 + x))
    
    # Combine factors
    heuristics_matrix = relative_strength - 0.4 * volatility + 0.6 * vol_change_log
    
    return heuristics_matrix
