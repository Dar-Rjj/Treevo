def heuristics_v2(df):
    # Momentum of the adjusted close price over 20 days
    mom_close = df['close'] / df['close'].shift(20)
    
    # Amount relative to the 60-day average amount
    avg_amount_60d = df['amount'].rolling(window=60).mean()
    rel_amount = df['amount'] / avg_amount_60d
    
    # Distance: Difference between the highest and lowest prices over the last 10 days
    high_low_distance = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    
    # Combine factors
    heuristics_matrix = mom_close + 0.5 * rel_amount + 0.5 * high_low_distance
    
    return heuristics_matrix
