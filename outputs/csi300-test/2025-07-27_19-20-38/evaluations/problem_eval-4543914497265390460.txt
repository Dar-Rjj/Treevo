def heuristics_v2(df):
    # Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Standard Deviation of closing prices over 20 days
    std_20 = df['close'].rolling(window=20).std()
    
    # Ratio of today's volume to the 20-day moving average of volume
    vol_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * rsi_14 + 0.3 * std_20 + 0.3 * vol_ratio
    
    return heuristics_matrix
