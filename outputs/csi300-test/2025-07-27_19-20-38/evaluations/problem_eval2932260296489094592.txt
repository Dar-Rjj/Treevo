def heuristics_v2(df):
    # Momentum: Rate of change in adjusted close price over the last 20 days
    momentum_close = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volume relative to the 30-day max volume
    max_vol_30d = df['volume'].rolling(window=30).max()
    rel_vol_max = df['volume'] / max_vol_30d
    
    # RSI: Relative Strength Index over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine factors
    heuristics_matrix = momentum_close + 0.5 * rel_vol_max + 0.5 * rsi
    
    return heuristics_matrix
