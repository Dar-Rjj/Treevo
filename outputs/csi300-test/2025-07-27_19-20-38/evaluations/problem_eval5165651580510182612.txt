def heuristics_v2(df):
    # Relative Strength Index over 14 days
    delta = df['close'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Simple Moving Average of closing prices over 20 days
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Rate of change of Volume over 7 days
    roc_volume = df['volume'] / df['volume'].shift(7) - 1
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.5 * rsi_14 + 0.3 * (df['close'] / sma_20) + 0.2 * roc_volume
    
    return heuristics_matrix
