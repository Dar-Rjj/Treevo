def heuristics_v2(df):
    # Rate of change in adjusted close price over the last 15 days
    roc_close = (df['close'] - df['close'].shift(15)) / df['close'].shift(15)
    
    # Volume relative to the 20-day average volume
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    rel_vol = df['volume'] / avg_vol_20d
    
    # Distance: Difference between 20D SMA and 50D SMA
    sma_20d = df['close'].rolling(window=20).mean()
    sma_50d = df['close'].rolling(window=50).mean()
    distance_sma = sma_20d - sma_50d
    
    # Combine factors
    heuristics_matrix = roc_close + 0.5 * rel_vol + 0.5 * distance_sma
    
    return heuristics_matrix
