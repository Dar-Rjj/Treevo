def heuristics_v2(df):
    # Average True Range over 10 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_10 = tr.rolling(window=10).mean()
    
    # Exponential Moving Average of closing prices over 30 days
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Momentum of Volume over 5 days
    momentum_volume = df['volume'] / df['volume'].shift(5)
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * atr_10 + 0.6 * (df['close'] / ema_30) + 0.1 * momentum_volume
    
    return heuristics_matrix
