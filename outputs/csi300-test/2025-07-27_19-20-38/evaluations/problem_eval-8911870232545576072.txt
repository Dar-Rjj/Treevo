def heuristics_v2(df):
    # Rate of Change (ROC) of closing prices over 14 days
    roc_14 = df['close'].pct_change(periods=14)
    
    # Bollinger Bands: Calculate the 20-day moving average and 20-day standard deviation
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    bb_width = (sma_20 + 2*std_20) - (sma_20 - 2*std_20)
    
    # Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * roc_14 + 0.5 * bb_width + 0.2 * rsi_14
    
    return heuristics_matrix
