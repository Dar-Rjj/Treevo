def heuristics_v2(df):
    # Stochastic Oscillator %K over 14 days
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    stoch_k = 100 * ((df['close'] - lowest_low) / (highest_high - lowest_low))
    
    # Percentage Price Oscillator using 12 and 26 day EMA
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    ppo = ((ema_12 - ema_26) / ema_26) * 100
    
    # On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'] > df['close'].shift())).cumsum()
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * stoch_k + 0.4 * ppo + 0.3 * obv
    
    return heuristics_matrix
