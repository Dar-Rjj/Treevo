def heuristics_v2(df):
    # Relative Strength Index over 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Weighted Moving Average of closing prices over 20 days
    wma_20 = df['close'].rolling(window=20).apply(lambda prices: (prices * range(1, 21)).sum() / 210, raw=True)
    
    # Rate of Change of Volume over 7 days
    roc_volume = (df['volume'] - df['volume'].shift(7)) / df['volume'].shift(7)
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.5 * rsi_14 + 0.3 * (df['close'] / wma_20) + 0.2 * roc_volume
    
    return heuristics_matrix
