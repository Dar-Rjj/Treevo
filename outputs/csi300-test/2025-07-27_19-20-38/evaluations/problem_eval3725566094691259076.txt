def heuristics_v2(df):
    # Simple Moving Average of closing prices over 30 days
    sma_30 = df['close'].rolling(window=30).mean()
    
    # Rate of Change of the highest price over 5 days
    roc_high_5 = (df['high'] / df['high'].shift(5)) - 1
    
    # Logarithmic Return of Volume
    log_return_volume = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * sma_30 + 0.4 * roc_high_5 + 0.2 * log_return_volume
    
    return heuristics_matrix
