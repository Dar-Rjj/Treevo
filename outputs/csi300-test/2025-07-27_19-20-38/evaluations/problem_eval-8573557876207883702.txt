def heuristics_v2(df):
    # Rate of change in adjusted close price over the last 20 days
    roc_close = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volume relative to the 50-day average volume
    avg_vol_50d = df['volume'].rolling(window=50).mean()
    rel_vol = df['volume'] / avg_vol_50d
    
    # Relative strength: Difference between closing price and 100D SMA
    sma_100d = df['close'].rolling(window=100).mean()
    relative_strength = (df['close'] - sma_100d) / sma_100d
    
    # Combine factors
    heuristics_matrix = roc_close + 0.5 * rel_vol + 0.5 * relative_strength
    
    return heuristics_matrix
