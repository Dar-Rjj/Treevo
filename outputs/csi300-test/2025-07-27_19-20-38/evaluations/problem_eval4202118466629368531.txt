def heuristics_v2(df):
    # Rate of change in high price over the last 10 days
    roc_high = (df['high'] - df['high'].shift(10)) / df['high'].shift(10)
    
    # Volume relative to the 5-day average volume
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    rel_vol = df['volume'] / avg_vol_5d
    
    # Divergence: Difference between closing price and 200D SMA
    sma_200d = df['close'].rolling(window=200).mean()
    divergence = df['close'] - sma_200d
    
    # Combine factors
    heuristics_matrix = 0.5 * roc_high + 0.3 * rel_vol + 0.2 * divergence
    
    return heuristics_matrix
