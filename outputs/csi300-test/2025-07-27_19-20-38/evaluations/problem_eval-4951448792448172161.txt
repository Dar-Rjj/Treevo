def heuristics_v2(df):
    # Calculate RSI for the closing price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume relative to the 20-day average volume
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    rel_vol = df['volume'] / avg_vol_20d
    
    # Momentum: Difference between closing price and 20D SMA
    sma_20d = df['close'].rolling(window=20).mean()
    momentum = df['close'] - sma_20d
    
    # Combine factors
    heuristics_matrix = rsi + 0.3 * rel_vol + 0.7 * momentum
    
    return heuristics_matrix
