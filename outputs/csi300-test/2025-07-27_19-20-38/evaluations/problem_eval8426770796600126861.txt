def heuristics_v2(df):
    # Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_14 = 100 - (100 / (1 + rs))
    
    # Percentage change in volume over 7 days
    pct_change_volume = df['volume'].pct_change(periods=7)
    
    # Standard deviation of closing prices over 20 days
    std_close_20 = df['close'].rolling(window=20).std()
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.5 * rsi_14 + 0.3 * pct_change_volume + 0.2 * std_close_20
    
    return heuristics_matrix
