import pandas as pd

def heuristics_v2(df):
    # Normalized Average True Range over 14 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_14 = tr.rolling(window=14).mean() / df['close']
    
    # Difference between the closing price and its 20-day Exponential Moving Average
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    diff_close_ema = df['close'] - ema_20
    
    # Relative Strength Index (RSI) of the Volume over 14 days
    delta_volume = df['volume'].diff(1)
    gain = (delta_volume.where(delta_volume > 0, 0)).rolling(window=14).mean()
    loss = (-delta_volume.where(delta_volume < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.3 * atr_14 + 0.5 * diff_close_ema + 0.2 * rsi_volume
    
    return heuristics_matrix
