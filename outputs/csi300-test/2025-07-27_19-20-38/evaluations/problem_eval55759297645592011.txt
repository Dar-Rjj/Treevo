def heuristics_v2(df):
    df['price_change'] = df['close'].pct_change()
    df['volume_momentum'] = df['volume'].pct_change()
    df['heuristic_factor'] = df['price_change'] * df['volume_momentum']
    df['wma_returns'] = df['price_change'].rolling(window=5).apply(lambda x: (x*list(range(1,6))).sum()/15)
    heuristics_matrix = df['heuristic_factor'] + df['wma_returns']
    return heuristics_matrix
