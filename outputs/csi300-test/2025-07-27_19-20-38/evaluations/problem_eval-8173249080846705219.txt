def heuristics_v2(df):
    # Momentum in adjusted close price over the last 20 days
    mom_close = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volume relative to the 60-day average volume
    avg_vol_60d = df['volume'].rolling(window=60).mean()
    rel_vol = df['volume'] / avg_vol_60d
    
    # Modified Divergence: Relative difference between closing price and 100D EMA
    ema_100d = df['close'].ewm(span=100, adjust=False).mean()
    div_rel = (df['close'] - ema_100d) / ema_100d
    
    # Combine factors with unique weights
    heuristics_matrix = mom_close + 0.5 * rel_vol + 1.2 * div_rel
    
    return heuristics_matrix
