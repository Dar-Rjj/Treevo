import pandas as pd

def heuristics_v2(df):
    # Relative Strength: (Close - 30D Low) / (30D High - 30D Low)
    high_30d = df['high'].rolling(window=30).max()
    low_30d = df['low'].rolling(window=30).min()
    relative_strength = (df['close'] - low_30d) / (high_30d - low_30d)
    
    # Moving Average: 30-day moving average of the closing price
    ma_30d = df['close'].rolling(window=30).mean()
    
    # Volatility: Standard deviation of daily returns over the last 30 days
    volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Volume Change: Percentage change in volume
    vol_change = df['volume'].pct_change()
    
    # Combine factors
    heuristics_matrix = relative_strength - 0.7 * volatility + 0.5 * vol_change + 0.2 * (df['close'] - ma_30d) / ma_30d
    
    return heuristics_matrix
