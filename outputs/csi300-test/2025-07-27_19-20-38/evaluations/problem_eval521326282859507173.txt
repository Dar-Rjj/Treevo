def heuristics_v2(df):
    # 10-day Standard Deviation of Returns
    returns = df['close'].pct_change()
    std_10d = returns.rolling(window=10).std()
    
    # Volume relative to the 50-day minimum volume
    min_vol_50d = df['volume'].rolling(window=50).min()
    rel_min_vol = df['volume'] / min_vol_50d
    
    # Percentage difference between the current close and the highest high over the last 20 days
    max_high_20d = df['high'].rolling(window=20).max()
    pct_diff_max_high = (df['close'] - max_high_20d) / max_high_20d
    
    # Heuristic matrix combining the factors
    heuristics_matrix = 0.5 * std_10d + 0.3 * rel_min_vol + 0.2 * pct_diff_max_high
    
    return heuristics_matrix
