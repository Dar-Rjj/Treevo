def heuristics_v2(df):
    # Rate of change in adjusted close price over the last 20 days
    roc_close = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Amount relative to the 40-day average amount
    avg_amount_40d = df['amount'].rolling(window=40).mean()
    rel_amount = df['amount'] / avg_amount_40d
    
    # Difference between the current close and the minimum close over the past 60 days
    min_close_60d = df['close'].rolling(window=60).min()
    diff_min_close = df['close'] - min_close_60d
    
    # Combine factors
    heuristics_matrix = 0.5 * roc_close + 0.3 * rel_amount + 0.2 * diff_min_close
    
    return heuristics_matrix
