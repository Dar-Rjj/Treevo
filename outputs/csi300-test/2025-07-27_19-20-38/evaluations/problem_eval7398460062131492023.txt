def heuristics_v2(df):
    # Average Directional Movement Index over 14 days
    up_move = df['high'].diff(1)
    down_move = -df['low'].diff(1)
    pos_dm = up_move.where((up_move > down_move) & (up_move > 0), 0)
    neg_dm = down_move.where((down_move > up_move) & (down_move > 0), 0)
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_14 = tr.rolling(window=14).mean()
    pos_di = 100 * (pos_dm.rolling(window=14).sum() / atr_14)
    neg_di = 100 * (neg_dm.rolling(window=14).sum() / atr_14)
    adx = 100 * ((pos_di - neg_di).abs() / (pos_di + neg_di)).rolling(window=14).mean()

    # Simple Moving Average of closing prices over 30 days
    sma_30 = df['close'].rolling(window=30).mean()

    # Rate of Change of Volume over 5 days
    roc_volume = df['volume'] / df['volume'].shift(5) - 1

    # Heuristic matrix combining the factors
    heuristics_matrix = 0.4 * adx + 0.4 * (df['close'] / sma_30) + 0.2 * roc_volume

    return heuristics_matrix
