def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 30 days for a more extended and stable period
    df['momentum_30d'] = df['close'].pct_change(periods=30)
    
    # Adaptive average true range over 40 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr40'] = df['tr'].ewm(span=40, adjust=False).mean()

    # Volume trend over 120 days for a more stable long-term trend
    df['volume_ratio_120d'] = df['volume'] / df['volume'].rolling(window=120).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Exponential moving average of the weighted price over a 90-day window for enhanced stability
    df['ema_weighted_price_90d'] = df['weighted_price'].ewm(span=90, adjust=False).mean()
    
    # Relative strength index (RSI) over 20 days for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.ewm(span=20, adjust=False).mean()
    avg_loss = loss.ewm(span=20, adjust=False).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_30d'] * rsi
    df['volatility_volume_interaction'] = df['atr40'] * df['volume_ratio_120d']

    # Enhanced interaction terms
    df['price_volatility_interaction'] = df['close'] * df['atr40']
    df['momentum_volume_interaction'] = df['momentum_30d'] * df['volume_ratio_120d']

    # Alpha factor: Combine momentum, RSI, ATR (as volatility), volume trend, and weighted price with interaction terms
    alpha_factor = (df['momentum_30d'] + df['atr40'] + df['volume_ratio_120d'] + 
                    (df['weighted_price'] / df['ema_weighted_price_90d']) + 
                    rsi + df['momentum_rsi_interaction'] + df['volatility_volume_interaction'] +
                    df['price_volatility_interaction'] + df['momentum_volume_interaction']) / 8

    return alpha_factor.dropna()
