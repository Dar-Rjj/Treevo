def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 60 days for a more extended and stable period
    df['momentum_60d'] = df['close'].pct_change(periods=60)
    
    # Average true range over 70 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr70'] = df['tr'].rolling(window=70).mean()

    # Volume trend over 200 days for a more stable long-term trend
    df['volume_ratio_200d'] = df['volume'] / df['volume'].rolling(window=200).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 180-day window for enhanced stability
    df['rolling_mean_weighted_price_180d'] = df['weighted_price'].rolling(window=180).mean()
    
    # Relative strength index (RSI) over 30 days for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=30).mean()
    avg_loss = loss.rolling(window=30).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Adaptive multi-period momentum
    df['momentum_40d'] = df['close'].pct_change(periods=40)
    df['momentum_avg'] = (df['momentum_60d'] + df['momentum_40d']) / 2

    # Alpha factor: Combine adaptive multi-period momentum, RSI, ATR (as volatility), volume trend, and weighted price
    alpha_factor = (
        (df['momentum_avg'] + rsi + (df['atr70'] * df['volume_ratio_200d']) + 
         (df['weighted_price'] / df['rolling_mean_weighted_price_180d'])) / 4
    )

    return alpha_factor.dropna()
