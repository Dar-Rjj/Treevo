```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 60 days for a more extended and stable period
    df['momentum_60d'] = df['close'].pct_change(periods=60)
    
    # Average true range over 50 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr50'] = df['tr'].rolling(window=50).mean()

    # Volume trend over 150 days for a more stable long-term trend
    df['volume_ratio_150d'] = df['volume'] / df['volume'].rolling(window=150).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 180-day window for enhanced stability
    df['rolling_mean_weighted_price_180d'] = df['weighted_price'].rolling(window=180).mean()
    
    # Relative strength index (RSI) over 30 days for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=30).mean()
    avg_loss = loss.rolling(window=30).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_60d'] * rsi
    df['volatility_volume_interaction'] = df['atr50'] * df['volume_ratio_150d']

    # Non-linear interaction terms
    df['momentum_squared'] = df['momentum_60d'] ** 2
    df['rsi_squared'] = rsi ** 2

    # Alpha factor: Combine momentum, ATR (as volatility), volume trend, weighted price, and interaction terms
    alpha_factor = (df['momentum_60d'] + 
                    df['atr50'] + 
                    df['volume_ratio_150d'] + 
                    df['weighted_price'] / df['rolling_mean_weighted_price_180d'] + 
                    df['momentum_rsi_interaction'] + 
                    df['volatility_volume_interaction'] +
                    df['momentum_squared'] + 
                    df['rsi_squared']) / 8

    return alpha_factor.dropna()
```
