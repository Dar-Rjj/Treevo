```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Short-term momentum over 10 days for a more responsive signal
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Average true range over 15 days for short-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr15'] = df['tr'].rolling(window=15).mean()

    # Volume trend over 50 days for a more stable medium-term trend
    df['volume_ratio_50d'] = df['volume'] / df['volume'].rolling(window=50).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 40-day window for enhanced stability
    df['rolling_mean_weighted_price_40d'] = df['weighted_price'].rolling(window=40).mean()
    
    # Relative strength index (RSI) over 14 days for more responsive momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_10d'] * rsi
    df['volatility_volume_interaction'] = df['atr15'] * df['volume_ratio_50d']

    # MACD (Moving Average Convergence Divergence) for additional momentum
    df['ema12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = df['ema12'] - df['ema26']
    df['signal'] = df['macd'].ewm(span=9, adjust=False).mean()
    df['histogram'] = df['macd'] - df['signal']

    # Combine factors with different weights
    alpha_factor = (0.2 * df['momentum_10d'] + 
                    0.2 * df['atr15'] + 
                    0.2 * df['volume_ratio_50d'] + 
                    0.1 * (df['weighted_price'] / df['rolling_mean_weighted_price_40d']) + 
                    0.1 * rsi + 
                    0.1 * df['momentum_rsi_interaction'] + 
                    0.1 * df['volatility_volume_interaction'] +
                    0.1 * df['histogram'])

    return alpha_factor.dropna()
```
