def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 20 days for a balanced stable and responsive period
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Average true range over 30 days for medium-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr30'] = df['tr'].rolling(window=30).mean()

    # Volume trend over 100 days for a more reactive long-term trend
    df['volume_ratio_100d'] = df['volume'] / df['volume'].rolling(window=100).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 70-day window for enhanced stability
    df['rolling_mean_weighted_price_70d'] = df['weighted_price'].rolling(window=70).mean()
    
    # Relative strength index (RSI) over 14 days for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_20d'] * rsi
    df['volatility_volume_interaction'] = df['atr30'] * df['volume_ratio_100d']

    # Alpha factor: Combine momentum, RSI, ATR (as volatility), volume trend, and weighted price with interaction terms
    alpha_factor = (df['momentum_20d'] + 
                    df['atr30'] + 
                    df['volume_ratio_100d'] + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_70d']) + 
                    rsi + 
                    df['momentum_rsi_interaction'] + 
                    df['volatility_volume_interaction']) / 7

    return alpha_factor.dropna()
