def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, to get the range of the day
    daily_range = df['high'] - df['low']
    
    # Calculate the weighted average price by considering both amount and volume
    weighted_price = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Calculate the moving average of the weighted price over a 5-day window
    rolling_mean_weighted_price = weighted_price.rolling(window=5).mean()
    
    # Alpha factor is the ratio of the current weighted price to the moving average of the weighted price
    alpha_factor = weighted_price / rolling_mean_weighted_price
    alpha_factor.fillna(0, inplace=True)
    
    return alpha_factor
