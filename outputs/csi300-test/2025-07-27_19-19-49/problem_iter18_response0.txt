```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 14 days for a balanced responsive period
    df['momentum_14d'] = df['close'].pct_change(periods=14)
    
    # Average true range over 25 days for mid-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr25'] = df['tr'].rolling(window=25).mean()

    # Volume trend over 70 days for a more stable medium-term trend
    df['volume_ratio_70d'] = df['volume'] / df['volume'].rolling(window=70).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 50-day window for enhanced stability
    df['rolling_mean_weighted_price_50d'] = df['weighted_price'].rolling(window=50).mean()
    
    # Relative strength index (RSI) over 12 days for more sensitive momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=12).mean()
    avg_loss = loss.rolling(window=12).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_14d'] * rsi
    df['volatility_volume_interaction'] = df['atr25'] * df['volume_ratio_70d']

    # Combine factors without normalization
    alpha_factor = (df['momentum_14d'] + df['atr25'] + df['volume_ratio_70d'] + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_50d']) + 
                    rsi + df['momentum_rsi_interaction'] + df['volatility_volume_interaction']) / 7

    return alpha_factor.dropna()
```
