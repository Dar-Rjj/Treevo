```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, capturing the momentum
    df['momentum'] = df['close'].diff()
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=3).mean()
    
    # Calculate the factor by combining the momentum and the change in VWAP
    factor = df['momentum'] + (df['vwap'] - df['close'].shift(1))
    
    return factor
```
