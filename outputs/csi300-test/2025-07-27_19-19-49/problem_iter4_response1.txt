```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 200-day momentum
    rolling_mean_50 = df['close'].rolling(window=50).mean()
    rolling_mean_200 = df['close'].rolling(window=200).mean()
    momentum_factor = (rolling_mean_50 - rolling_mean_200) / df['close']

    # Calculate the 20-day volatility using standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    volatility_factor = daily_returns.rolling(window=20).std()

    # Calculate the volume-weighted average price (VWAP) over a 3-day window
    vwap = (df['amount'] / df['volume']).rolling(window=3).mean()

    # Volume ratio compared to the 30-day mean
    volume_ratio = df['volume'] / df['volume'].rolling(window=30).mean()

    # Positive and negative volume index changes
    df['volume_change'] = df['volume'].pct_change()
    df['positive_volume_index'] = (df['close'].diff() > 0).astype(int) * df['volume_change']
    df['negative_volume_index'] = (df['close'].diff() < 0).astype(int) * df['volume_change']

    # Sum of positive and negative volume index over a rolling window of 14 days
    pos_vol_sum = df['positive_volume_index'].rolling(window=14).sum()
    neg_vol_sum = df['negative_volume_index'].rolling(window=14).sum()

    # Volume sentiment factor
    volume_sentiment_factor = (pos_vol_sum - neg_vol_sum) / (pos_vol_sum + neg_vol_sum + 1e-7)

    # Calculate the close-to-VWAP ratio
    close_vwap_ratio = df['close'] / vwap

    # Combine factors into an alpha factor
    alpha_factor = (
        momentum_factor * 
        (1 / (volatility_factor + 1e-7)) * 
        volume_ratio * 
        volume_sentiment_factor * 
        (close_vwap_ratio - 1)
    )

    return alpha_factor.dropna()
```
