def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 20 days for a more sensitive and responsive period
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Average true range over 30 days for shorter-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr30'] = df['tr'].rolling(window=30).mean()

    # Volume trend over 90 days for a more stable long-term trend
    df['volume_ratio_90d'] = df['volume'] / df['volume'].rolling(window=90).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 60-day window for enhanced stability
    df['rolling_mean_weighted_price_60d'] = df['weighted_price'].rolling(window=60).mean()
    
    # Relative strength index (RSI) over 15 days for more sensitive momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=15).mean()
    avg_loss = loss.rolling(window=15).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Normalize factors before combining
    momentum_20d_norm = (df['momentum_20d'] - df['momentum_20d'].mean()) / (df['momentum_20d'].std() + 1e-7)
    atr30_volume_ratio_90d_norm = ((df['atr30'] * df['volume_ratio_90d']) - (df['atr30'] * df['volume_ratio_90d']).mean()) / ((df['atr30'] * df['volume_ratio_90d']).std() + 1e-7)
    weighted_price_norm = ((df['weighted_price'] / df['rolling_mean_weighted_price_60d']) - (df['weighted_price'] / df['rolling_mean_weighted_price_60d']).mean()) / ((df['weighted_price'] / df['rolling_mean_weighted_price_60d']).std() + 1e-7)
    rsi_norm = (rsi - rsi.mean()) / (rsi.std() + 1e-7)

    # Alpha factor: Combine normalized momentum, RSI, ATR (as volatility), volume trend, and weighted price
    alpha_factor = (momentum_20d_norm + atr30_volume_ratio_90d_norm + weighted_price_norm + rsi_norm) / 4

    return alpha_factor.dropna()
