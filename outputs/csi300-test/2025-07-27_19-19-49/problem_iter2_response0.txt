```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 5 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Average true range over 14 days for volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr14'] = df['tr'].rolling(window=14).mean()

    # Volume ratio over 30 days
    df['volume_ratio_30d'] = df['volume'] / df['volume'].rolling(window=30).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 20-day window
    df['rolling_mean_weighted_price_20d'] = df['weighted_price'].rolling(window=20).mean()
    
    # Alpha factor: Combine momentum, average true range (as volatility), volume ratio, and weighted price
    alpha_factor = (df['momentum_5d'] + (df['atr14'] * df['volume_ratio_30d']) + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_20d'])) / 3

    return alpha_factor.dropna()
```
