```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a momentum indicator
    df['momentum'] = df['close'].diff()
    
    # Average true range over 14 days for volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr14'] = df['tr'].rolling(window=14).mean()

    # Volume weighted average price (VWAP) over 3 days
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=3).mean()

    # Calculate the ratio of the volume to the mean of past 30 days' volume as a flow indicator
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=30).mean()

    # Calculate the ratio of the close price to the VWAP to capture the relative pricing
    df['close_vwap_ratio'] = df['close'] / df['vwap']

    # Final factor: Combine momentum, average true range (as volatility), volume ratio, and close to VWAP ratio
    alpha_factor = (df['momentum'] + (df['atr14'] * df['volume_ratio']) + (df['close_vwap_ratio'] - 1)) / 3

    return alpha_factor.dropna()
```
