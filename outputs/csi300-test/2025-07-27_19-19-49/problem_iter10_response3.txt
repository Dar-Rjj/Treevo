def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 50 days for a more extended and stable period
    df['momentum_50d'] = df['close'].pct_change(periods=50)
    
    # Average true range over 60 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr60'] = df['tr'].rolling(window=60).mean()

    # Volume trend over 180 days for a more stable long-term trend
    df['volume_ratio_180d'] = df['volume'] / df['volume'].rolling(window=180).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 150-day window for enhanced stability
    df['rolling_mean_weighted_price_150d'] = df['weighted_price'].rolling(window=150).mean()
    
    # Relative strength index (RSI) over 25 days for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=25).mean()
    avg_loss = loss.rolling(window=25).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Adaptive multi-period momentum
    df['momentum_30d'] = df['close'].pct_change(periods=30)
    df['momentum_avg'] = (df['momentum_50d'] + df['momentum_30d']) / 2

    # Alpha factor: Combine adaptive multi-period momentum, RSI, ATR (as volatility), volume trend, and weighted price
    alpha_factor = (
        (df['momentum_avg'] + rsi + (df['atr60'] * df['volume_ratio_180d']) + 
         (df['weighted_price'] / df['rolling_mean_weighted_price_150d'])) / 4
    )

    return alpha_factor.dropna()
