```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative volume index changes
    df['volume_change'] = df['volume'].pct_change()
    df['positive_volume_index'] = (df['close'].diff() > 0).astype(int) * df['volume_change']
    df['negative_volume_index'] = (df['close'].diff() < 0).astype(int) * df['volume_change']

    # Sum of positive and negative volume index over a rolling window of 10 days
    pos_vol_sum = df['positive_volume_index'].rolling(window=10).sum()
    neg_vol_sum = df['negative_volume_index'].rolling(window=10).sum()

    # Generate the alpha factor
    alpha_factor = (pos_vol_sum - neg_vol_sum) / (pos_vol_sum + neg_vol_sum + 1e-7)

    return alpha_factor
```
