```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 10 days for a longer momentum signal
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Average true range over 20 days for volatility, extending the window for more stability
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr20'] = df['tr'].rolling(window=20).mean()

    # Volume ratio over 40 days to capture a longer volume trend
    df['volume_ratio_40d'] = df['volume'] / df['volume'].rolling(window=40).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 30-day window for a smoother trend
    df['rolling_mean_weighted_price_30d'] = df['weighted_price'].rolling(window=30).mean()
    
    # Alpha factor: Combine 10-day momentum, 20-day ATR (as volatility), 40-day volume ratio, and 30-day weighted price
    alpha_factor = (df['momentum_10d'] + (df['atr20'] * df['volume_ratio_40d']) + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_30d'])) / 3

    return alpha_factor.dropna()
```
