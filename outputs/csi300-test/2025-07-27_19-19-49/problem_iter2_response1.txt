```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative volume index changes
    df['volume_change'] = df['volume'].pct_change()
    df['positive_volume_index'] = (df['close'].diff() > 0).astype(int) * df['volume_change']
    df['negative_volume_index'] = (df['close'].diff() < 0).astype(int) * df['volume_change']

    # Sum of positive and negative volume index over a rolling window of 14 days
    pos_vol_sum = df['positive_volume_index'].rolling(window=14).sum()
    neg_vol_sum = df['negative_volume_index'].rolling(window=14).sum()

    # Calculate the momentum factor using the 50-day and 200-day moving averages of closing prices
    rolling_mean_50 = df['close'].rolling(window=50).mean()
    rolling_mean_200 = df['close'].rolling(window=200).mean()
    momentum_factor = (rolling_mean_50 - rolling_mean_200) / df['close']

    # Calculate the volatility factor using the 20-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    volatility_factor = daily_returns.rolling(window=20).std()

    # Generate the alpha factor
    alpha_factor = (pos_vol_sum - neg_vol_sum) / (pos_vol_sum + neg_vol_sum + 1e-7)
    alpha_factor = alpha_factor * momentum_factor / (volatility_factor + 1e-7)

    return alpha_factor
```
