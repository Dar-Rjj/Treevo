```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Short-term momentum over 5 days for a highly responsive signal
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Average true range over 10 days for short-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr10'] = df['tr'].rolling(window=10).mean()

    # Volume trend over 30 days for a more dynamic volume trend
    df['volume_ratio_30d'] = df['volume'] / df['volume'].rolling(window=30).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 45-day window for enhanced stability
    df['rolling_mean_weighted_price_45d'] = df['weighted_price'].rolling(window=45).mean()
    
    # Relative strength index (RSI) over 14 days for more responsive momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_5d'] * rsi
    df['volatility_volume_interaction'] = df['atr10'] * df['volume_ratio_30d']

    # Non-linear interaction terms
    df['momentum_squared'] = df['momentum_5d'] ** 2
    df['rsi_squared'] = rsi ** 2

    # Alpha factor: Combine momentum, ATR (as volatility), volume trend, weighted price, and interaction terms
    alpha_factor = (df['momentum_5d'] + 
                    df['atr10'] + 
                    df['volume_ratio_30d'] + 
                    df['weighted_price'] / df['rolling_mean_weighted_price_45d'] + 
                    df['momentum_rsi_interaction'] + 
                    df['volatility_volume_interaction'] +
                    df['momentum_squared'] + 
                    df['rsi_squared']) / 8

    return alpha_factor.dropna()
```
