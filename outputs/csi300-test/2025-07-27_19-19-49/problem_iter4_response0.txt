```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 7 days
    df['momentum_7d'] = df['close'].pct_change(periods=7)
    
    # Average true range over 10 days for volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr10'] = df['tr'].rolling(window=10).mean()

    # Volume ratio over 25 days
    df['volume_ratio_25d'] = df['volume'] / df['volume'].rolling(window=25).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 25-day window
    df['rolling_mean_weighted_price_25d'] = df['weighted_price'].rolling(window=25).mean()
    
    # Alpha factor: Combine momentum, average true range (as volatility), volume ratio, and weighted price
    alpha_factor = (df['momentum_7d'] + (df['atr10'] * df['volume_ratio_25d']) + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_25d'])) / 3

    return alpha_factor.dropna()
```
