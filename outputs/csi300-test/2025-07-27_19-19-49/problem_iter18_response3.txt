```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 40 days for a more extended and stable period
    df['momentum_40d'] = df['close'].pct_change(periods=40)
    
    # Average true range over 50 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr50'] = df['tr'].rolling(window=50).mean()

    # Volume trend over 150 days for a very stable long-term trend
    df['volume_ratio_150d'] = df['volume'] / df['volume'].rolling(window=150).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 120-day window for enhanced stability
    df['rolling_mean_weighted_price_120d'] = df['weighted_price'].rolling(window=120).mean()
    
    # Relative strength index (RSI) over 25 days for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=25).mean()
    avg_loss = loss.rolling(window=25).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_40d'] * rsi
    df['volatility_volume_interaction'] = df['atr50'] * df['volume_ratio_150d']

    # Additional interaction terms
    df['momentum_volatility_interaction'] = df['momentum_40d'] * df['atr50']
    df['rsi_volume_interaction'] = rsi * df['volume_ratio_150d']
    
    # Alpha factor: Combine all factors with their interaction terms
    alpha_factor = (df['momentum_40d'] + rsi + df['atr50'] + df['volume_ratio_150d'] + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_120d']) + 
                    df['momentum_rsi_interaction'] + df['volatility_volume_interaction'] +
                    df['momentum_volatility_interaction'] + df['rsi_volume_interaction']) / 10

    return alpha_factor.dropna()
```
