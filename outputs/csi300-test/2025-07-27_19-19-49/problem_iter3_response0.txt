```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 10 days for a more extended period
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    
    # Average true range over 21 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr21'] = df['tr'].rolling(window=21).mean()

    # Volume trend over 60 days
    df['volume_ratio_60d'] = df['volume'] / df['volume'].rolling(window=60).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 50-day window for enhanced stability
    df['rolling_mean_weighted_price_50d'] = df['weighted_price'].rolling(window=50).mean()
    
    # Alpha factor: Combine momentum, average true range (as volatility), volume trend, and weighted price
    alpha_factor = (df['momentum_10d'] + (df['atr21'] * df['volume_ratio_60d']) + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_50d'])) / 3

    return alpha_factor.dropna()
```
