```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive multi-period momentum, use a range of periods to adapt to different market conditions
    momentum_30d = df['close'].pct_change(periods=30)
    momentum_60d = df['close'].pct_change(periods=60)
    df['adaptive_momentum'] = 0.5 * (momentum_30d + momentum_60d)

    # Average true range over 40 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr40'] = df['tr'].rolling(window=40).mean()

    # Volume trend over 120 days for a more stable long-term trend
    df['volume_ratio_120d'] = df['volume'] / df['volume'].rolling(window=120).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 90-day window for enhanced stability
    df['rolling_mean_weighted_price_90d'] = df['weighted_price'].rolling(window=90).mean()
    
    # Alpha factor: Combine adaptive momentum, average true range (as volatility), volume trend, and weighted price
    alpha_factor = (df['adaptive_momentum'] + (df['atr40'] * df['volume_ratio_120d']) + 
                    (df['weighted_price'] / df['rolling_mean_weighted_price_90d'])) / 3

    return alpha_factor.dropna()
```
