```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 30 days for a more extended and stable period
    df['momentum_30d'] = df['close'].pct_change(periods=30)
    
    # Average true range over 40 days for longer-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr40'] = df['tr'].rolling(window=40).mean()

    # Volume trend over 120 days for a more stable long-term trend
    df['volume_ratio_120d'] = df['volume'] / df['volume'].rolling(window=120).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 90-day window for enhanced stability
    df['rolling_mean_weighted_price_90d'] = df['weighted_price'].rolling(window=90).mean()
    
    # Relative strength index (RSI) over 20 days for momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Interaction terms
    df['momentum_rsi_interaction'] = df['momentum_30d'] * rsi
    df['volatility_volume_interaction'] = df['atr40'] * df['volume_ratio_120d']

    # Normalize factors before combining
    momentum_30d_norm = (df['momentum_30d'] - df['momentum_30d'].mean()) / (df['momentum_30d'].std() + 1e-7)
    atr40_volume_ratio_120d_norm = ((df['atr40'] * df['volume_ratio_120d']) - (df['atr40'] * df['volume_ratio_120d']).mean()) / ((df['atr40'] * df['volume_ratio_120d']).std() + 1e-7)
    weighted_price_norm = ((df['weighted_price'] / df['rolling_mean_weighted_price_90d']) - (df['weighted_price'] / df['rolling_mean_weighted_price_90d']).mean()) / ((df['weighted_price'] / df['rolling_mean_weighted_price_90d']).std() + 1e-7)
    rsi_norm = (rsi - rsi.mean()) / (rsi.std() + 1e-7)
    momentum_rsi_interaction_norm = (df['momentum_rsi_interaction'] - df['momentum_rsi_interaction'].mean()) / (df['momentum_rsi_interaction'].std() + 1e-7)
    volatility_volume_interaction_norm = (df['volatility_volume_interaction'] - df['volatility_volume_interaction'].mean()) / (df['volatility_volume_interaction'].std() + 1e-7)

    # Alpha factor: Combine normalized momentum, RSI, ATR (as volatility), volume trend, and weighted price with interaction terms
    alpha_factor = (momentum_30d_norm + atr40_volume_ratio_120d_norm + weighted_price_norm + rsi_norm + 
                    momentum_rsi_interaction_norm + volatility_volume_interaction_norm) / 6

    return alpha_factor.dropna()
```
