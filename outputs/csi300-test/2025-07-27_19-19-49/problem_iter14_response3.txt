```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum over 15 days for a more responsive momentum signal
    df['momentum_15d'] = df['close'].pct_change(periods=15)
    
    # Average true range over 20 days for mid-term volatility
    df['h-l'] = df['high'] - df['low']
    df['h-cp'] = abs(df['high'] - df['close'].shift(1))
    df['l-cp'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['h-l', 'h-cp', 'l-cp']].max(axis=1)
    df['atr20'] = df['tr'].rolling(window=20).mean()

    # Volume trend over 60 days for a more stable medium-term trend
    df['volume_ratio_60d'] = df['volume'] / df['volume'].rolling(window=60).mean()

    # Calculate the weighted average price by considering both amount and volume
    df['weighted_price'] = (df['amount'] * df['close']) / (df['volume'] + 1e-7)
    
    # Moving average of the weighted price over a 45-day window for enhanced stability
    df['rolling_mean_weighted_price_45d'] = df['weighted_price'].rolling(window=45).mean()
    
    # Relative strength index (RSI) over 10 days for short-term momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))

    # Adaptive multi-period momentum
    df['momentum_10d'] = df['close'].pct_change(periods=10)
    df['momentum_avg'] = (df['momentum_15d'] + df['momentum_10d']) / 2

    # Alpha factor: Combine adaptive multi-period momentum, RSI, ATR (as volatility), volume trend, and weighted price
    alpha_factor = (
        (df['momentum_avg'] + rsi + (df['atr20'] * df['volume_ratio_60d']) + 
         (df['weighted_price'] / df['rolling_mean_weighted_price_45d'])) / 4
    )

    return alpha_factor.dropna()
```
