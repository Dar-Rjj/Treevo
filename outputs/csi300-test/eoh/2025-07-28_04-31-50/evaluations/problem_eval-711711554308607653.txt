def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    open_20_ma = df['open'].rolling(window=20).mean()
    volume_5_max = df['volume'].rolling(window=5).max()
    volume_5_min = df['volume'].rolling(window=5).min()
    heuristics_matrix = (close_10_ema / open_20_ma) - (volume_5_max - volume_5_min)
    return heuristics_matrix
