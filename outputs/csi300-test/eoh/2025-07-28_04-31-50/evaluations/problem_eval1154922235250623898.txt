import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day moving average of the close-to-volume ratio
    close_volume_ratio = df['close'] / df['volume']
    mavol_10 = close_volume_ratio.rolling(window=10).mean()
    
    # Calculate the 5-day percentage change in volume
    pct_change_vol = df['volume'].pct_change(periods=5)
    
    # Define the heuristic factor as a combination of mavol_10 and pct_change_vol
    heuristics_matrix = (mavol_10 * 0.6) + (pct_change_vol * 0.4)
    return heuristics_matrix
