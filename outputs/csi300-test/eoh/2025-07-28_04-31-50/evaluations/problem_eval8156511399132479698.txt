def heuristics_v2(df):
    # Calculate the 20-day high-low range
    high_low_range = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 20-day average trading amount
    avg_amount = df['amount'].rolling(window=20).mean()
    
    # Combine the factors into a single score
    heuristics_matrix = high_low_range / avg_amount
    
    return heuristics_matrix
