def heuristics_v2(df):
    close_10_sma = df['close'].rolling(window=10).mean()
    open_25_ema = df['open'].ewm(span=25, adjust=False).mean()
    high_3_sma = df['high'].rolling(window=3).mean()
    low_7_sma = df['low'].rolling(window=7).mean()
    sma_diff = high_3_sma - low_7_sma
    heuristics_matrix = (close_10_sma / open_25_ema) * sma_diff
    return heuristics_matrix
