def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    open_20_ma = df['open'].rolling(window=20).mean()
    volume_30_ma = df['volume'].rolling(window=30).mean()
    amount_40_ma = df['amount'].rolling(window=40).mean()
    heuristics_matrix = (close_10_ma / open_20_ma) + (volume_30_ma + amount_40_ma)
    return heuristics_matrix
