def heuristics_v2(df):
    amount_10_ma = df['amount'].rolling(window=10).mean()
    volume_20_ma = df['volume'].rolling(window=20).mean()
    high_5_ma = df['high'].rolling(window=5).mean()
    low_5_ma = df['low'].rolling(window=5).mean()
    heuristics_matrix = (amount_10_ma / volume_20_ma) + (high_5_ma - low_5_ma)
    return heuristics_matrix
