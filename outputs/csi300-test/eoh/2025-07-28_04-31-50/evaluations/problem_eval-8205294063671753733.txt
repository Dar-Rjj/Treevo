import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-period Exponential Moving Average (EMA) of closing prices
    ema = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the difference between the close price and EMA
    diff_ema_close = df['close'] - ema
    
    # Calculate the Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a single heuristic
    heuristics_matrix = diff_ema_close + 0.5 * rsi
    
    return heuristics_matrix
