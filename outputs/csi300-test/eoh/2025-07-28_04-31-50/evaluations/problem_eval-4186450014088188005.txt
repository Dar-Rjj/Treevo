def heuristics_v2(df):
    close_7_ema = df['close'].ewm(span=7, adjust=False).mean()
    low_14_ema = df['low'].ewm(span=14, adjust=False).mean()
    high_20_ema = df['high'].ewm(span=20, adjust=False).mean()
    open_30_ema = df['open'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (close_7_ema - low_14_ema) + (high_20_ema / open_30_ema)
    return heuristics_matrix
