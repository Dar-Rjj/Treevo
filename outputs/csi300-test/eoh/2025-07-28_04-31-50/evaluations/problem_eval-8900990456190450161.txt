def heuristics_v2(df):
    close_30_ma = df['close'].rolling(window=30).mean()
    volume_5_ma = df['volume'].rolling(window=5).mean()
    high_10_max = df['high'].rolling(window=10).max()
    low_10_min = df['low'].rolling(window=10).min()
    heuristics_matrix = (close_30_ma / volume_5_ma) * (high_10_max - low_10_min)
    return heuristics_matrix
