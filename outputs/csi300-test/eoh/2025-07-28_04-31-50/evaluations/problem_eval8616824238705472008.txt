def heuristics_v2(df):
    close_10_wma = df['close'].rolling(window=10).apply(lambda x: (x * pd.Series(range(1,11))).sum() / 55, raw=True)
    high_15_ma = df['high'].rolling(window=15).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    amount_25_ma = df['amount'].rolling(window=25).mean()
    heuristics_matrix = (close_10_wma / high_15_ma) - (amount_25_ma - volume_20_ema)
    return heuristics_matrix
