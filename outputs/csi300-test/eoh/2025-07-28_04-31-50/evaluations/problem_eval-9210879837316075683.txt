def heuristics_v2(df):
    high_10_ema = df['high'].ewm(span=10, adjust=False).mean()
    low_30_ema = df['low'].ewm(span=30, adjust=False).mean()
    amount_5_std = df['amount'].rolling(window=5).std()
    volume_20_std = df['volume'].rolling(window=20).std()
    heuristics_matrix = (high_10_ema - low_30_ema) + (amount_5_std / volume_20_std)
    return heuristics_matrix
