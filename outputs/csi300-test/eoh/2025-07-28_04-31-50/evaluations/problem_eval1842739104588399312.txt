def heuristics_v2(df):
    # Calculate the RSI for the closing prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the cumulative sum of volume changes
    volume_change = df['volume'].diff().cumsum()
    
    # Combine the factors into a single score
    heuristics_matrix = rsi + volume_change
    
    return heuristics_matrix
