def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    open_20_ma = df['open'].rolling(window=20).mean()
    high_5_ema = df['high'].ewm(span=5, adjust=False).mean()
    low_10_ema = df['low'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (close_10_ma / open_20_ma) - (high_5_ema - low_10_ema)
    return heuristics_matrix
