import pandas as pd

def heuristics_v2(df):
    # Momentum factor: 10-day close price difference
    momentum = df['close'].diff(10)
    
    # Mean reversion factor: 10-day simple moving average deviation
    sma_10 = df['close'].rolling(window=10).mean()
    mean_reversion = (df['close'] - sma_10) / sma_10
    
    # Volatility factor: 10-day standard deviation of close prices
    volatility = df['close'].rolling(window=10).std()
    
    # Volume change factor: 10-day volume change
    volume_change = df['volume'].pct_change(10)
    
    # Aggregate factors into a heuristics matrix
    heuristics_matrix = pd.concat([momentum, mean_reversion, volatility, volume_change], axis=1)
    heuristics_matrix.columns = ['Momentum', 'Mean_Reversion', 'Volatility', 'Volume_Change']
    
    return heuristics_matrix
