def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    open_20_ma = df['open'].rolling(window=20).mean()
    volume_5_ma = df['volume'].rolling(window=5).mean()
    amount_15_ma = df['amount'].rolling(window=15).mean()
    heuristics_matrix = (close_10_ma - open_20_ma) * (volume_5_ma / amount_15_ma)
    return heuristics_matrix
