def heuristics_v2(df):
    close_20_ema = df['close'].ewm(span=20, adjust=False).mean()
    volume_10_ema = df['volume'].ewm(span=10, adjust=False).mean()
    amount_30_sum = df['amount'].rolling(window=30).sum()
    heuristics_matrix = (close_20_ema / volume_10_ema) - amount_30_sum
    return heuristics_matrix
