def heuristics_v2(df):
    # Calculate the 5-day and 10-day moving averages of 'close'
    close_5_ma = df['close'].rolling(window=5).mean()
    close_10_ma = df['close'].rolling(window=10).mean()

    # Calculate the 5-day and 10-day moving averages of 'volume'
    volume_5_ma = df['volume'].rolling(window=5).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()

    # Calculate the difference in moving averages of 'close' and 'volume'
    close_diff = close_5_ma - close_10_ma
    volume_diff = volume_5_ma - volume_10_ma

    # Calculate the Relative Strength Index (RSI) with a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Combine the factors into a single heuristic
    heuristics_matrix = (close_diff * volume_diff) + rsi

    return heuristics_matrix
