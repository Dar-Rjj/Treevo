def heuristics_v2(df):
    high_15_ema = df['high'].ewm(span=15, adjust=False).mean()
    low_30_ema = df['low'].ewm(span=30, adjust=False).mean()
    volume_20_prod = df['volume'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (high_15_ema - low_30_ema) / volume_20_prod
    return heuristics_matrix
