def heuristics_v2(df):
    close_7_ma = df['close'].rolling(window=7).mean()
    open_15_ma = df['open'].rolling(window=15).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()
    amount_20_ma = df['amount'].rolling(window=20).mean()
    heuristics_matrix = (close_7_ma - open_15_ma) * (volume_10_ma / amount_20_ma)
    return heuristics_matrix
