def heuristics_v2(df):
    # Calculate the Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Calculate the Standard Deviation of the closing price
    std_dev = df['close'].rolling(window=14).std()
    
    # Combine the factors into a single heuristic
    heuristics_matrix = roc + 0.5 * std_dev
    
    return heuristics_matrix
