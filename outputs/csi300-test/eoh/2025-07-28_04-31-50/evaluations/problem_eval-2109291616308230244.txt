def heuristics_v2(df):
    open_5_ma = df['open'].rolling(window=5).mean()
    close_10_ma = df['close'].rolling(window=10).mean()
    high_20_max = df['high'].rolling(window=20).max()
    low_30_min = df['low'].rolling(window=30).min()
    heuristics_matrix = (open_5_ma / close_10_ma) - (high_20_max - low_30_min)
    return heuristics_matrix
