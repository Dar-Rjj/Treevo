def heuristics_v2(df):
    close_5_ema = df['close'].ewm(span=5, adjust=False).mean()
    high_20_ema = df['high'].ewm(span=20, adjust=False).mean()
    volume_10_std = df['volume'].rolling(window=10).std()
    heuristics_matrix = (close_5_ema / high_20_ema) - volume_10_std
    return heuristics_matrix
