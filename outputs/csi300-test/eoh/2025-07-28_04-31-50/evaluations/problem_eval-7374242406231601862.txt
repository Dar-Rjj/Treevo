def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    open_20_ema = df['open'].ewm(span=20, adjust=False).mean()
    volume_40_ma = df['volume'].rolling(window=40).mean()
    amount_50_ema = df['amount'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = (close_10_ema / open_20_ema) * (volume_40_ma - amount_50_ema)
    return heuristics_matrix
