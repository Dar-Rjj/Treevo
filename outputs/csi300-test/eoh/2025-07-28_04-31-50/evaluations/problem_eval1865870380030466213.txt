def heuristics_v2(df):
    # Calculate the 20-day rate of change (ROC)
    roc_20 = df['close'].pct_change(periods=20)

    # Calculate the daily returns
    daily_returns = df['close'].pct_change()

    # Calculate the 10-day standard deviation of the daily returns
    std_10 = daily_returns.rolling(window=10).std()

    # Combine the factors into a single heuristic
    heuristics_matrix = roc_20 + std_10

    return heuristics_matrix
