def heuristics_v2(df):
    close_20_ma = df['close'].rolling(window=20).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()
    amount_5_prod = df['amount'].rolling(window=5).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (close_20_ma / volume_10_ma) - amount_5_prod
    return heuristics_matrix
