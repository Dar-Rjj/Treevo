def heuristics_v2(df):
    high_5_ma = df['high'].rolling(window=5).mean()
    low_10_ma = df['low'].rolling(window=10).mean()
    amount_20_ma = df['amount'].rolling(window=20).mean()
    volume_30_ma = df['volume'].rolling(window=30).mean()
    heuristics_matrix = (high_5_ma - low_10_ma) * (amount_20_ma / volume_30_ma)
    return heuristics_matrix
