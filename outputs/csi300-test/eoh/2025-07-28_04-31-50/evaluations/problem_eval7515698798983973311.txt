import pandas as pd

def heuristics_v2(df):
    open_30_ema = df['open'].ewm(span=30, adjust=False).mean()
    low_20_std = df['low'].rolling(window=20).std()
    high_40_std = df['high'].rolling(window=40).std()
    std_ratio = low_20_std / high_40_std
    heuristics_matrix = open_30_ema * std_ratio
    return heuristics_matrix
