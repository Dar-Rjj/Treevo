def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10).mean()
    volume_5_ema = df['volume'].ewm(span=5).mean()
    amount_20_product = df['amount'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (close_10_ema / volume_5_ema) - amount_20_product
    return heuristics_matrix
