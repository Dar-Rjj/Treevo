import pandas as pd

def heuristics_v2(df):
    # Assuming df is a DataFrame with at least 'close' and 'volume' columns
    n = 5  # Define the number of days for WMA
    weights = list(range(1, n + 1))
    
    def wma(series, weights):
        return (series * weights).sum() / sum(weights)
    
    close_wma = df['close'].rolling(window=n).apply(lambda x: wma(x, weights), raw=True)
    volume_wma = df['volume'].rolling(window=n).apply(lambda x: wma(x, weights), raw=True)
    
    heuristics_matrix = close_wma / volume_wma
    return heuristics_matrix
