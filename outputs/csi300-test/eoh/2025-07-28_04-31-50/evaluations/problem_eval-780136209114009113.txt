import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day simple moving average of closing prices
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 50-day exponential moving average of closing prices
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Combine the factors into a single score
    heuristics_matrix = sma_20 / ema_50
    
    return heuristics_matrix
