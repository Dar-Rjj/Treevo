def heuristics_v2(df):
    close_open_ratio_ema = (df['close'] / df['open']).ewm(span=7, adjust=False).mean()
    volume_15_ma = df['volume'].rolling(window=15).mean()
    amount_25_ema = df['amount'].ewm(span=25, adjust=False).mean()
    heuristics_matrix = close_open_ratio_ema * (volume_15_ma - amount_25_ema)
    return heuristics_matrix
