def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    open_30_ma = df['open'].rolling(window=30).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    amount_40_ma = df['amount'].rolling(window=40).mean()
    heuristics_matrix = (close_10_ma / open_30_ma) - (amount_40_ma - volume_20_ema)
    return heuristics_matrix
