def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    open_30_ema = df['open'].ewm(span=30, adjust=False).mean()
    volume_20_std = df['volume'].rolling(window=20).std()
    heuristics_matrix = (close_10_ema - open_30_ema) + volume_20_std
    return heuristics_matrix
