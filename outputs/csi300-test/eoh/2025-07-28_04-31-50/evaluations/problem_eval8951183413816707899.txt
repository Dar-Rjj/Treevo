def heuristics_v2(df):
    # Calculate the 14-day Rate of Change (ROC)
    roc = ((df['close'] / df['close'].shift(14)) - 1) * 100
    
    # Calculate the 20-day simple moving average and standard deviation
    ma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    
    # Calculate the upper and lower Bollinger Bands
    bb_upper = ma_20 + 2 * std_20
    bb_lower = ma_20 - 2 * std_20
    
    # Create the heuristics matrix combining ROC and Bollinger Band width
    heuristics_matrix = roc + (bb_upper - bb_lower)

    return heuristics_matrix
