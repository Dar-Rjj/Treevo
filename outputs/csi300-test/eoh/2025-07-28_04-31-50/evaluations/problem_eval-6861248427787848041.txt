def heuristics_v2(df):
    close_14_ema = df['close'].ewm(span=14, adjust=False).mean()
    open_28_ema = df['open'].ewm(span=28, adjust=False).mean()
    volume_30_ema = df['volume'].ewm(span=30, adjust=False).mean()
    amount_40_ema = df['amount'].ewm(span=40, adjust=False).mean()
    heuristics_matrix = (close_14_ema / open_28_ema) - (volume_30_ema - amount_40_ema)
    return heuristics_matrix
