def heuristics_v2(df):
    close_5_ma = df['close'].rolling(window=5).mean()
    open_10_ma = df['open'].rolling(window=10).mean()
    amount_20_ma = df['amount'].rolling(window=20).mean()
    volume_30_ma = df['volume'].rolling(window=30).mean()
    heuristics_matrix = (close_5_ma * open_10_ma) + (amount_20_ma - volume_30_ma)
    return heuristics_matrix
