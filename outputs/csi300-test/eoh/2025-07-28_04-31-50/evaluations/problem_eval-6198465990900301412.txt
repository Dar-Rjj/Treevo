def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    open_5_ma = df['open'].rolling(window=5).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    amount_30_ema = df['amount'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (close_10_ma - open_5_ma) / (volume_20_ema - amount_30_ema)
    return heuristics_matrix
