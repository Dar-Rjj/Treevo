import pandas as pd

def heuristics_v2(df):
    # Calculate the moving averages
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()

    # Calculate the difference between high and low
    diff_high_low = df['high'] - df['low']

    # Combine the factors into a single heuristic
    heuristics_matrix = (ma_5 - ma_10) + 0.5 * diff_high_low

    return heuristics_matrix
