def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    high_15_ma = df['high'].rolling(window=15).mean()
    volume_5_ema = df['volume'].ewm(span=5, adjust=False).mean()
    amount_20_ma = df['amount'].rolling(window=20).mean()
    open_30_ma = df['open'].rolling(window=30).mean()
    heuristics_matrix = (close_10_ma / high_15_ma) + (amount_20_ma - open_30_ma + volume_5_ema)
    return heuristics_matrix
