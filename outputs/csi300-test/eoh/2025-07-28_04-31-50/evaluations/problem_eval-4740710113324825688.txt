def heuristics_v2(df):
    amount_5_ma = df['amount'].rolling(window=5).mean()
    volume_20_ma = df['volume'].rolling(window=20).mean()
    close_10_ma = df['close'].rolling(window=10).mean()
    open_30_ma = df['open'].rolling(window=30).mean()
    heuristics_matrix = (amount_5_ma - volume_20_ma) / (close_10_ma / open_30_ma)
    return heuristics_matrix
