import numpy as np

def heuristics_v2(df):
    high_low_diff_5_ema = (df['high'] - df['low']).ewm(span=5, adjust=False).mean()
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    volume_20_ma = df['volume'].rolling(window=20).mean()
    amount_30_ma = df['amount'].rolling(window=30).mean()
    heuristics_matrix = (high_low_diff_5_ema / close_10_ema) + np.log(volume_20_ma) - np.log(amount_30_ma)
    return heuristics_matrix
