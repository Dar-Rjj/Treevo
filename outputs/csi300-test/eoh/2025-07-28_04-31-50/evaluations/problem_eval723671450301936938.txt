import pandas as pd

def heuristics_v2(df):
    amount_3_ma = df['amount'].rolling(window=3).mean()
    pct_change_amount_3_ma = amount_3_ma.pct_change(periods=1)
    
    close_5_std = df['close'].rolling(window=5).std()
    volume_10_mean = df['volume'].rolling(window=10).mean()
    std_close_to_mean_volume = close_5_std / volume_10_mean
    
    heuristics_matrix = pct_change_amount_3_ma + std_close_to_mean_volume
    return heuristics_matrix
