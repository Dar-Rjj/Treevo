def heuristics_v2(df):
    # Calculate the 20-day rate of change (ROC) for the close price
    roc_20 = df['close'].pct_change(periods=20)
    # Calculate the 10-day rate of change (ROC) for the volume
    vol_roc_10 = df['volume'].pct_change(periods=10)
    # Combine the factors into a single score
    heuristics_matrix = roc_20 * vol_roc_10
    return heuristics_matrix
