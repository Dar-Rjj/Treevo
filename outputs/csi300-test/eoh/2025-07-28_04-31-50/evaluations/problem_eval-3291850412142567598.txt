def heuristics_v2(df):
    open_15_cp = df['open'].rolling(window=15).apply(lambda x: x.prod(), raw=True)
    low_25_cp = df['low'].rolling(window=25).apply(lambda x: x.prod(), raw=True)
    amount_50_ema = df['amount'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = (open_15_cp / low_25_cp) - amount_50_ema
    return heuristics_matrix
