def heuristics_v2(df):
    high_15_ema = df['high'].ewm(span=15, adjust=False).mean()
    low_25_ema = df['low'].ewm(span=25, adjust=False).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()
    amount_30_ma = df['amount'].rolling(window=30).mean()
    heuristics_matrix = (high_15_ema - low_25_ema) * (volume_10_ma / amount_30_ma)
    return heuristics_matrix
