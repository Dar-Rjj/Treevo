def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    high_5_ema = df['high'].ewm(span=5, adjust=False).mean()
    low_5_ema = df['low'].ewm(span=5, adjust=False).mean()
    heuristics_matrix = close_10_ema + (high_5_ema - low_5_ema)
    return heuristics_matrix
