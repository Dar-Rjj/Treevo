import pandas as pd

def heuristics_v2(df):
    # Calculate the daily percentage change of closing prices
    price_change = df['close'].pct_change()
    # Calculate the daily change in volume
    volume_change = df['volume'].diff()
    # Generate the heuristic factor
    heuristics_matrix = (volume_change / price_change).fillna(0)
    return heuristics_matrix
