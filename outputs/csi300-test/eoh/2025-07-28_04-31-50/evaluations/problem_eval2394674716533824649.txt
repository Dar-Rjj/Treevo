def heuristics_v2(df):
    close_20_mean = df['close'].rolling(window=20).mean()
    volume_10_mean = df['volume'].rolling(window=10).mean()
    amount_30_log_prod = df['amount'].apply(np.log).rolling(window=30).agg(lambda x: np.prod(x))
    heuristics_matrix = (close_20_mean / volume_10_mean) - amount_30_log_prod
    return heuristics_matrix
