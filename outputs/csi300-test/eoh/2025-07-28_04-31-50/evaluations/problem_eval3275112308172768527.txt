def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    volume_20_ma = df['volume'].rolling(window=20).mean()
    high_5_mean = df['high'].rolling(window=5).mean()
    low_5_mean = df['low'].rolling(window=5).mean()
    heuristics_matrix = (close_10_ema / volume_20_ma) + (high_5_mean - low_5_mean)
    return heuristics_matrix
