def heuristics_v2(df):
    close_5_ema = df['close'].ewm(span=5, adjust=False).mean()
    open_10_ema = df['open'].ewm(span=10, adjust=False).mean()
    amount_30_ema = df['amount'].ewm(span=30, adjust=False).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (close_5_ema / open_10_ema) + (amount_30_ema - volume_20_ema)
    return heuristics_matrix
