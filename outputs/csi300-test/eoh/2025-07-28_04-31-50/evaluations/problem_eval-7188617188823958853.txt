import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC)
    roc_10 = df['close'].pct_change(periods=10)

    # Calculate the 5-day and 20-day average true range
    tr = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    atr_5 = tr.rolling(window=5).mean()
    atr_20 = tr.rolling(window=20).mean()

    # Calculate the ratio of the 5-day ATR to the 20-day ATR
    atr_ratio = atr_5 / atr_20

    # Combine the factors into a single heuristic
    heuristics_matrix = roc_10 * atr_ratio

    return heuristics_matrix
