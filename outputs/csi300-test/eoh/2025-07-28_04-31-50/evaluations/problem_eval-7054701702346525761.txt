def heuristics_v2(df):
    # Calculate 10-day rate of change of the closing price
    close_roc = df['close'].pct_change(periods=10)
    
    # Calculate 20-day coefficient of variation of the trading volume
    volume_mean = df['volume'].rolling(window=20).mean()
    volume_std = df['volume'].rolling(window=20).std()
    volume_cv = volume_std / volume_mean
    
    # Adjust the rate of change by the coefficient of variation
    heuristics_matrix = close_roc / volume_cv
    return heuristics_matrix
