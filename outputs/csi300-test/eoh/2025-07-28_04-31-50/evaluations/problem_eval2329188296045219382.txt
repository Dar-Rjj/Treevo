def heuristics_v2(df):
    close_5_ma = df['close'].rolling(window=5).mean()
    open_10_ma = df['open'].rolling(window=10).mean()
    volume_20_ma = df['volume'].rolling(window=20).mean()
    amount_30_ma = df['amount'].rolling(window=30).mean()
    heuristics_matrix = (close_5_ma / open_10_ma) + (amount_30_ma - volume_20_ma)
    return heuristics_matrix
