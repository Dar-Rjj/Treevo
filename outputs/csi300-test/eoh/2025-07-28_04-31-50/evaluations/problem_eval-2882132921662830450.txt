def heuristics_v2(df):
    close_10_mean = df['close'].rolling(window=10).mean()
    volume_5_mean = df['volume'].rolling(window=5).mean()
    amount_20_prod = df['amount'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (close_10_mean / volume_5_mean) * amount_20_prod
    return heuristics_matrix
