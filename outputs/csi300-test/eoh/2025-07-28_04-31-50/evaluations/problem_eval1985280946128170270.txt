def heuristics_v2(df):
    close_20_ma = df['close'].rolling(window=20).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()
    high_10_max = df['high'].rolling(window=10).max()
    low_10_min = df['low'].rolling(window=10).min()
    heuristics_matrix = (close_20_ma * volume_10_ma) - (high_10_max - low_10_min)
    return heuristics_matrix
