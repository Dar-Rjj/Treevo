def heuristics_v2(df):
    close_7_ma = df['close'].rolling(window=7).mean()
    open_15_ma = df['open'].rolling(window=15).mean()
    amount_25_ma = df['amount'].rolling(window=25).mean()
    volume_35_ma = df['volume'].rolling(window=35).mean()
    heuristics_matrix = (close_7_ma / open_15_ma) + (volume_35_ma - amount_25_ma)
    return heuristics_matrix
