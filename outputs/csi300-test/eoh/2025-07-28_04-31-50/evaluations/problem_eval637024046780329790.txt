def heuristics_v2(df):
    close_20_ema = df['close'].ewm(span=20, adjust=False).mean()
    volume_10_ema = df['volume'].ewm(span=10, adjust=False).mean()
    high_10_max = df['high'].rolling(window=10).max()
    low_10_min = df['low'].rolling(window=10).min()
    heuristics_matrix = (close_20_ema * volume_10_ema) - (high_10_max - low_10_min)
    return heuristics_matrix
