def heuristics_v2(df):
    close_30_ema = df['close'].ewm(span=30, adjust=False).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    amount_40_sum = df['amount'].rolling(window=40).sum()
    heuristics_matrix = (close_30_ema / volume_20_ema) - amount_40_sum
    return heuristics_matrix
