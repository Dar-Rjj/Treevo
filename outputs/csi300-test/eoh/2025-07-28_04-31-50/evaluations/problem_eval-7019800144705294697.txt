def heuristics_v2(df):
    close_5_ema = df['close'].ewm(span=5, adjust=False).mean()
    open_10_ema = df['open'].ewm(span=10, adjust=False).mean()
    volume_20_ma = df['volume'].rolling(window=20).mean()
    amount_30_ema = df['amount'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (close_5_ema / open_10_ema) * (volume_20_ma + amount_30_ema)
    return heuristics_matrix
