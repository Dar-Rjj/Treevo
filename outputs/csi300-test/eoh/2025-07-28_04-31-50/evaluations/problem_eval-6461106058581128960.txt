import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 50-day moving average of close price
    ma_50 = df['close'].rolling(window=50).mean()

    # Calculate the 21-day rate of change (ROC)
    roc = ((df['close'] - df['close'].shift(21)) / df['close'].shift(21)) * 100

    # Define the heuristic factor as a combination of RSI, 50-day moving average, and ROC
    heuristics_matrix = (ma_50 * 0.5) + (rsi * 0.3) + (roc * 0.2)
    return heuristics_matrix
