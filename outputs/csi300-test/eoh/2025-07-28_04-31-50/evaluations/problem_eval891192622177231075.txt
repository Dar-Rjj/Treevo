def heuristics_v2(df):
    close_5_ma = df['close'].rolling(window=5).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()
    high_20_max = df['high'].rolling(window=20).max()
    low_20_min = df['low'].rolling(window=20).min()
    heuristics_matrix = (close_5_ma * volume_10_ma) + (high_20_max - low_20_min)
    return heuristics_matrix
