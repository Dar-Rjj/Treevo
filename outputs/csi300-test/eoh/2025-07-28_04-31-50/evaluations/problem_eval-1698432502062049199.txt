def heuristics_v2(df):
    amount_5_ma = df['amount'].rolling(window=5).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()
    high_10_ma = df['high'].rolling(window=10).mean()
    close_20_ma = df['close'].rolling(window=20).mean()
    heuristics_matrix = (amount_5_ma / volume_10_ma) + (high_10_ma - close_20_ma)
    return heuristics_matrix
