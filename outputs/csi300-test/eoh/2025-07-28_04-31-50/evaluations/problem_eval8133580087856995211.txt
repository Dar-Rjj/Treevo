def heuristics_v2(df):
    amount_7_ma = df['amount'].rolling(window=7).mean()
    volume_14_ma = df['volume'].rolling(window=14).mean()
    close_5_ma = df['close'].rolling(window=5).mean()
    open_5_ma = df['open'].rolling(window=5).mean()
    heuristics_matrix = (amount_7_ma / volume_14_ma) + (close_5_ma - open_5_ma)
    return heuristics_matrix
