def heuristics_v2(df):
    high_7_ema = df['high'].ewm(span=7, adjust=False).mean()
    low_15_ema = df['low'].ewm(span=15, adjust=False).mean()
    amount_25_ema = df['amount'].ewm(span=25, adjust=False).mean()
    volume_35_ema = df['volume'].ewm(span=35, adjust=False).mean()
    heuristics_matrix = (high_7_ema / low_15_ema) + (amount_25_ema - volume_35_ema)
    return heuristics_matrix
