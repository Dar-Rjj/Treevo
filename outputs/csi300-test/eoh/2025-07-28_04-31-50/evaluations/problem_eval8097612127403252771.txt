import pandas as pd

def heuristics_v2(df):
    amount_10_ma = df['amount'].rolling(window=10).mean()
    pct_change_amount_10_ma = amount_10_ma.pct_change(periods=1)
    
    close_5_std = df['close'].rolling(window=5).std()
    volume_15_mean = df['volume'].rolling(window=15).mean()
    std_close_to_mean_volume = close_5_std / volume_15_mean
    
    heuristics_matrix = pct_change_amount_10_ma + std_close_to_mean_volume
    return heuristics_matrix
