def heuristics_v2(df):
    high_7_ma = df['high'].rolling(window=7).mean()
    low_15_ma = df['low'].rolling(window=15).mean()
    volume_25_ma = df['volume'].rolling(window=25).mean()
    amount_35_ma = df['amount'].rolling(window=35).mean()
    heuristics_matrix = (high_7_ma / low_15_ma) - (volume_25_ma - amount_35_ma)
    return heuristics_matrix
