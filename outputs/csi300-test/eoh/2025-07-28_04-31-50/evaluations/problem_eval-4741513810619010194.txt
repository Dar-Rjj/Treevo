def heuristics_v2(df):
    high_7_ema = df['high'].ewm(span=7, adjust=False).mean()
    low_14_ema = df['low'].ewm(span=14, adjust=False).mean()
    volume_25_ma = df['volume'].rolling(window=25).mean()
    amount_35_ema = df['amount'].ewm(span=35, adjust=False).mean()
    heuristics_matrix = (high_7_ema / low_14_ema) * (volume_25_ma + amount_35_ema)
    return heuristics_matrix
