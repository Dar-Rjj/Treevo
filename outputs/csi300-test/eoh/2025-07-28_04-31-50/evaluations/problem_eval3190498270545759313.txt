def heuristics_v2(df):
    close_15_ema = df['close'].ewm(span=15, adjust=False).mean()
    open_30_ema = df['open'].ewm(span=30, adjust=False).mean()
    volume_60_std = df['volume'].rolling(window=60).std()
    heuristics_matrix = (close_15_ema - open_30_ema) / volume_60_std
    return heuristics_matrix
