def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    volume_20_ma = df['volume'].rolling(window=20).mean()
    high_5_max = df['high'].rolling(window=5).max()
    low_5_min = df['low'].rolling(window=5).min()
    heuristics_matrix = (close_10_ma * volume_20_ma) - (high_5_max - low_5_min)
    return heuristics_matrix
