import numpy as np

def heuristics_v2(df):
    # Calculate the 20-day high-low range
    high_low_range = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    
    # Calculate the 20-day average trading volume
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Combine the factors into a single score
    heuristics_matrix = high_low_range * np.log(avg_volume)
    
    return heuristics_matrix
