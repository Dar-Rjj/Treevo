def heuristics_v2(df):
    close_15_ma = df['close'].rolling(window=15).mean()
    returns_7 = df['close'].pct_change().rolling(window=7).std()
    ewm_volume_30 = df['volume'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (close_15_ma + returns_7) / ewm_volume_30
    return heuristics_matrix
