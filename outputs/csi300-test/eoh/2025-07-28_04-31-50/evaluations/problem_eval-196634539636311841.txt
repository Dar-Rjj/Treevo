def heuristics_v2(df):
    high_10_ma = df['high'].rolling(window=10).mean()
    low_20_ma = df['low'].rolling(window=20).mean()
    amount_15_ma = df['amount'].rolling(window=15).mean()
    volume_30_ma = df['volume'].rolling(window=30).mean()
    heuristics_matrix = (high_10_ma - low_20_ma) * (amount_15_ma - volume_30_ma)
    return heuristics_matrix
