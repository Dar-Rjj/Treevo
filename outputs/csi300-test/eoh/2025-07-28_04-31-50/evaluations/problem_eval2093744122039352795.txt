import pandas as pd

def heuristics_v2(df):
    short_ema = df['close'].ewm(span=10, adjust=False).mean()
    long_ema = df['close'].ewm(span=30, adjust=False).mean()
    ema_diff = short_ema - long_ema
    avg_volume = df['volume'].rolling(window=30).mean()
    heuristics_matrix = ema_diff / avg_volume
    return heuristics_matrix
