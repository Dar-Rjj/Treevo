def heuristics_v2(df):
    close_5_ma = df['close'].rolling(window=5).mean()
    open_10_ma = df['open'].rolling(window=10).mean()
    volume_20_sum = df['volume'].rolling(window=20).sum()
    amount_30_sum = df['amount'].rolling(window=30).sum()
    heuristics_matrix = (close_5_ma - open_10_ma) * (volume_20_sum / amount_30_sum)
    return heuristics_matrix
