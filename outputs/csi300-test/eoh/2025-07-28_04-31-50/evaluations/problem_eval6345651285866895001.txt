def heuristics_v2(df):
    close_20_ema = df['close'].ewm(span=20, adjust=False).mean()
    open_10_ema = df['open'].ewm(span=10, adjust=False).mean()
    volume_50_ma = df['volume'].rolling(window=50).mean()
    amount_60_ma = df['amount'].rolling(window=60).mean()
    heuristics_matrix = (close_20_ema * open_10_ema) - (volume_50_ma - amount_60_ma)
    return heuristics_matrix
