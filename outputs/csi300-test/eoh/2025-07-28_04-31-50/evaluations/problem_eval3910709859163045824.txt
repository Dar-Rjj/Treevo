def heuristics_v2(df):
    open_10_ma = df['open'].rolling(window=10).mean()
    close_5_ema = df['close'].ewm(span=5, adjust=False).mean()
    volume_20_prod = df['volume'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    amount_15_ma = df['amount'].rolling(window=15).mean()
    heuristics_matrix = (open_10_ma / close_5_ema) + (volume_20_prod - amount_15_ma)
    return heuristics_matrix
