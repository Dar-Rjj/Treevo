def heuristics_v2(df):
    amount_20_ma = df['amount'].rolling(window=20).mean()
    volume_5_ma = df['volume'].rolling(window=5).mean()
    high_low_10_ma = (df['high'] - df['low']).rolling(window=10).mean()
    heuristics_matrix = (amount_20_ma / volume_5_ma) - high_low_10_ma
    return heuristics_matrix
