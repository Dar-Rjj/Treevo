def heuristics_v2(df):
    # Calculate the 50-day moving average of the close price
    ma_50 = df['close'].rolling(window=50).mean()
    # Calculate the difference between the current close and the 50-day moving average
    trend_factor = df['close'] - ma_50
    # Calculate the 20-day moving average of the volume
    vol_ma_20 = df['volume'].rolling(window=20).mean()
    # Calculate the ratio of the current volume to the 20-day moving average
    liquidity_factor = df['volume'] / vol_ma_20
    # Combine the factors into a single score
    heuristics_matrix = trend_factor + liquidity_factor
    return heuristics_matrix
