import pandas as pd

def heuristics_v2(df):
    amount_5_ma = df['amount'].rolling(window=5).mean()
    pct_change_amount_5_ma = amount_5_ma.pct_change(periods=1)
    
    close_10_std = df['close'].rolling(window=10).std()
    volume_20_mean = df['volume'].rolling(window=20).mean()
    std_close_to_mean_volume = close_10_std / volume_20_mean
    
    heuristics_matrix = pct_change_amount_5_ma + std_close_to_mean_volume
    return heuristics_matrix
