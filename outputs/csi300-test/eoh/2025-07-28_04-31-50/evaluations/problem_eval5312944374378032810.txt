def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    open_20_ema = df['open'].ewm(span=20, adjust=False).mean()
    volume_5_ma = df['volume'].rolling(window=5).mean()
    amount_30_ma = df['amount'].rolling(window=30).mean()
    heuristics_matrix = (close_10_ema - open_20_ema) * (volume_5_ma / amount_30_ma)
    return heuristics_matrix
