def heuristics_v2(df):
    close_15_ma = df['close'].rolling(window=15).mean()
    open_20_ma = df['open'].rolling(window=20).mean()
    volume_10_ema = df['volume'].ewm(span=10, adjust=False).mean()
    amount_30_ma = df['amount'].rolling(window=30).mean()
    heuristics_matrix = (close_15_ma / open_20_ma) - (amount_30_ma - volume_10_ema)
    return heuristics_matrix
