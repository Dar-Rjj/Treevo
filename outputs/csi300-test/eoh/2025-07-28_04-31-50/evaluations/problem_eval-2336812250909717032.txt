def heuristics_v2(df):
    close_15_ema = df['close'].ewm(span=15, adjust=False).mean()
    open_25_ema = df['open'].ewm(span=25, adjust=False).mean()
    volume_10_cp = df['volume'].rolling(window=10).apply(lambda x: x.prod(), raw=True)
    amount_30_cp = df['amount'].rolling(window=30).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (close_15_ema - open_25_ema) + (volume_10_cp / amount_30_cp)
    return heuristics_matrix
