def heuristics_v2(df):
    close_10_sma = df['close'].rolling(window=10).mean()
    open_15_ema = df['open'].ewm(span=15, adjust=False).mean()
    high_3_sma = df['high'].rolling(window=3).mean()
    low_3_sma = df['low'].rolling(window=3).mean()
    sma_diff = high_3_sma - low_3_sma
    heuristics_matrix = (close_10_sma / open_15_ema) * sma_diff
    return heuristics_matrix
