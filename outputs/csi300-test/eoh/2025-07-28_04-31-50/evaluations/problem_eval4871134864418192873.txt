import pandas as pd

def heuristics_v2(df):
    # Calculate the EMA of the close-to-volume ratio
    ema_close_to_volume = df['close'] / df['volume']
    ema_close_to_volume = ema_close_to_volume.ewm(span=5, adjust=False).mean()
    
    # Calculate the Rate of Change (ROC) of the close prices
    roc_close = df['close'].pct_change(periods=5)
    
    # Combine the factors into a single heuristic
    heuristics_matrix = ema_close_to_volume * roc_close
    
    return heuristics_matrix
