def heuristics_v2(df):
    high_15_ema = df['high'].ewm(span=15, adjust=False).mean()
    low_25_ema = df['low'].ewm(span=25, adjust=False).mean()
    volume_40_ma = df['volume'].rolling(window=40).mean()
    amount_50_ma = df['amount'].rolling(window=50).mean()
    heuristics_matrix = (high_15_ema / low_25_ema) + (volume_40_ma / amount_50_ma)
    return heuristics_matrix
