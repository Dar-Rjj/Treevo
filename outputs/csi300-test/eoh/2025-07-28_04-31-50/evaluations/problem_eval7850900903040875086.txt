def heuristics_v2(df):
    high_30_ema = df['high'].ewm(span=30, adjust=False).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    amount_10_sum = df['amount'].rolling(window=10).sum()
    heuristics_matrix = (high_30_ema - volume_20_ema) / amount_10_sum
    return heuristics_matrix
