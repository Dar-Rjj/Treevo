def heuristics_v2(df):
    # Calculate the change in close price
    price_change = df['close'].diff()
    # Calculate the 5-day moving average of price change
    avg_price_change_5 = price_change.rolling(window=5).mean()
    # Calculate the daily volume change
    volume_change = df['volume'].diff()
    # Calculate the 5-day moving average of volume change
    avg_volume_change_5 = volume_change.rolling(window=5).mean()
    # Define the heuristic factor as a combination of average price change and volume change
    heuristics_matrix = (avg_price_change_5 * 0.6) + (avg_volume_change_5 * 0.4)
    return heuristics_matrix
