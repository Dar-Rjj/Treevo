def heuristics_v2(df):
    high_5_sum = df['high'].rolling(window=5).sum()
    low_10_sum = df['low'].rolling(window=10).sum()
    volume_20_ewm = df['volume'].ewm(span=20, adjust=False).mean()
    amount_30_ewm = df['amount'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (high_5_sum / low_10_sum) - (volume_20_ewm - amount_30_ewm)
    return heuristics_matrix
