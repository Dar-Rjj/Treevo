def heuristics_v2(df):
    high_10_sma = df['high'].rolling(window=10).mean()
    low_30_ema = df['low'].ewm(span=30, adjust=False).mean()
    close_7_sma = df['close'].rolling(window=7).mean()
    open_7_sma = df['open'].rolling(window=7).mean()
    sma_diff = close_7_sma - open_7_sma
    heuristics_matrix = (high_10_sma / low_30_ema) * sma_diff
    return heuristics_matrix
