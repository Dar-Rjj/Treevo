def heuristics_v2(df):
    open_15_ma = df['open'].rolling(window=15).mean()
    close_25_ma = df['close'].rolling(window=25).mean()
    volume_10_prod = df['volume'].rolling(window=10).apply(lambda x: x.prod(), raw=True)
    amount_5_prod = df['amount'].rolling(window=5).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (open_15_ma - close_25_ma) + (volume_10_prod / amount_5_prod)
    return heuristics_matrix
