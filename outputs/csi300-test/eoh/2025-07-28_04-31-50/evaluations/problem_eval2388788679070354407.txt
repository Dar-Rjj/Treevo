def heuristics_v2(df):
    close_5_ema = df['close'].ewm(span=5, adjust=False).mean()
    low_10_ema = df['low'].ewm(span=10, adjust=False).mean()
    volume_15_ema = df['volume'].ewm(span=15, adjust=False).mean()
    amount_20_ema = df['amount'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (close_5_ema - low_10_ema) * (volume_15_ema - amount_20_ema)
    return heuristics_matrix
