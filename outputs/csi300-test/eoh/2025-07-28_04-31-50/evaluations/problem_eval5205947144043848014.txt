def heuristics_v2(df):
    close_14_ema = df['close'].ewm(span=14, adjust=False).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    amount_50_sum = df['amount'].rolling(window=50).sum()
    heuristics_matrix = (close_14_ema / volume_20_ema) + amount_50_sum
    return heuristics_matrix
