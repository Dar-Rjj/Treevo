def heuristics_v2(df):
    # Calculate the momentum of closing prices over 20 days
    momentum = df['close'].pct_change(periods=20)
    
    # Calculate the True Range
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate the Average True Range (ATR) over 7 days
    atr = true_range.rolling(window=7).mean()
    
    # Combine the factors into a single heuristic
    heuristics_matrix = momentum + 0.3 * atr
    
    return heuristics_matrix
