def heuristics_v2(df):
    close_5_ema = df['close'].ewm(span=5, adjust=False).mean()
    high_10_ema = df['high'].ewm(span=10, adjust=False).mean()
    amount_20_ema = df['amount'].ewm(span=20, adjust=False).mean()
    volume_30_ema = df['volume'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (close_5_ema * (1 / high_10_ema)) + (amount_20_ema - volume_30_ema)
    return heuristics_matrix
