import pandas as pd

def heuristics_v2(df):
    # Calculate the 21-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=21).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=21).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 30-day moving average of close price
    ma_30 = df['close'].rolling(window=30).mean()

    # Define the heuristic factor as a combination of RSI and 30-day moving average
    heuristics_matrix = (ma_30 * 0.5) + (rsi * 0.5)
    return heuristics_matrix
