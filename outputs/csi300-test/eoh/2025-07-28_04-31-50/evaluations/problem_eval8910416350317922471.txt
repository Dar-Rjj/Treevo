def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    sma_10_daily_return = daily_return.rolling(window=10).mean()
    std_30_amount = df['amount'].rolling(window=30).std()
    max_20_volume = df['volume'].rolling(window=20).max()
    heuristics_matrix = (sma_10_daily_return * std_30_amount) + max_20_volume
    return heuristics_matrix
