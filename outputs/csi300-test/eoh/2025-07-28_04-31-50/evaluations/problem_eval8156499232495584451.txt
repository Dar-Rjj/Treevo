def heuristics_v2(df):
    close_15_ema = df['close'].ewm(span=15, adjust=False).mean()
    volume_30_ema = df['volume'].ewm(span=30, adjust=False).mean()
    amount_10_sum_log = df['amount'].apply(np.log).rolling(window=10).sum()
    heuristics_matrix = (close_15_ema / volume_30_ema) - amount_10_sum_log
    return heuristics_matrix
