def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    open_20_ema = df['open'].ewm(span=20, adjust=False).mean()
    volume_5_std = df['volume'].rolling(window=5).std()
    amount_15_std = df['amount'].rolling(window=15).std()
    heuristics_matrix = (close_10_ema - open_20_ema) * (volume_5_std / amount_15_std)
    return heuristics_matrix
