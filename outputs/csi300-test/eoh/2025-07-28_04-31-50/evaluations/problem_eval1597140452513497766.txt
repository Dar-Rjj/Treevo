def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    volume_change_5_ma = (df['volume'].pct_change().rolling(window=5).mean())
    high_low_20_ma = (df['high'] - df['low']).rolling(window=20).mean()
    heuristics_matrix = (close_10_ema * volume_change_5_ma) - high_low_20_ma
    return heuristics_matrix
