def heuristics_v2(df):
    close_20_ma = df['close'].rolling(window=20).mean()
    volume_15_ma = df['volume'].rolling(window=15).mean()
    high_7_max = df['high'].rolling(window=7).max()
    low_7_min = df['low'].rolling(window=7).min()
    heuristics_matrix = (close_20_ma * volume_15_ma) + (high_7_max - low_7_min)
    return heuristics_matrix
