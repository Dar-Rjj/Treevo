def heuristics_v2(df):
    high_20_ema = df['high'].ewm(span=20, adjust=False).mean()
    low_15_ema = df['low'].ewm(span=15, adjust=False).mean()
    volume_log = np.log(df['volume'])
    volume_30_cump = volume_log.rolling(window=30).apply(lambda x: np.prod(x), raw=True)
    heuristics_matrix = (high_20_ema / low_15_ema) + volume_30_cump
    return heuristics_matrix
