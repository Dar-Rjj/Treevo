def heuristics_v2(df):
    close_15_mean = df['close'].rolling(window=15).mean()
    volume_7_mean = df['volume'].rolling(window=7).mean()
    amount_25_prod = df['amount'].rolling(window=25).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (close_15_mean / volume_7_mean) * amount_25_prod
    return heuristics_matrix
