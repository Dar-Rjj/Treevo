def heuristics_v2(df):
    close_10_std = df['close'].rolling(window=10).std()
    volume_5_mean = df['volume'].rolling(window=5).mean()
    amount_log_20_product = df['amount'].apply(lambda x: np.log10(x+1)).rolling(window=20).apply(np.prod, raw=True)
    heuristics_matrix = (close_10_std / volume_5_mean) - amount_log_20_product
    return heuristics_matrix
