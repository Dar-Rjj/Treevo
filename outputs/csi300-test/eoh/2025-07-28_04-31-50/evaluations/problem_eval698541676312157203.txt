def heuristics_v2(df):
    close_20_ema = df['close'].ewm(span=20, adjust=False).mean()
    open_15_ema = df['open'].ewm(span=15, adjust=False).mean()
    volume_14_std = df['volume'].rolling(window=14).std()
    amount_28_std = df['amount'].rolling(window=28).std()
    std_ratio = volume_14_std / amount_28_std
    heuristics_matrix = (close_20_ema - open_15_ema) * std_ratio
    return heuristics_matrix
