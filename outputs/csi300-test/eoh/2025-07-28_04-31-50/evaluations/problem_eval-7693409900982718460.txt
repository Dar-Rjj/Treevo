import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day SMA of close prices
    sma_50 = df['close'].rolling(window=50).mean()
    # Rate of change in 50-day SMA
    sma_change_rate = sma_50.pct_change()
    
    # Ratio of today's volume to 60-day average volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=60).mean()
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine into a single heuristics matrix
    heuristics_matrix = sma_change_rate + volume_ratio + rsi
    return heuristics_matrix
