import pandas as pd

def heuristics_v2(df):
    close_20_ema = df['close'].ewm(span=20, adjust=False).mean()
    high_10_std = df['high'].rolling(window=10).std()
    low_30_std = df['low'].rolling(window=30).std()
    std_ratio = high_10_std / low_30_std
    heuristics_matrix = close_20_ema * std_ratio
    return heuristics_matrix
