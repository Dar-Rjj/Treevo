def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    open_20_ema = df['open'].ewm(span=20, adjust=False).mean()
    volume_30_ma = df['volume'].rolling(window=30).mean()
    amount_40_ma = df['amount'].rolling(window=40).mean()
    heuristics_matrix = (close_10_ema * open_20_ema) - (volume_30_ma - amount_40_ma)
    return heuristics_matrix
