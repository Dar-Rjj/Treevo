def heuristics_v2(df):
    open_7_ema = df['open'].ewm(span=7, adjust=False).mean()
    close_15_ema = df['close'].ewm(span=15, adjust=False).mean()
    volume_25_sma = df['volume'].rolling(window=25).mean()
    heuristics_matrix = (open_7_ema * close_15_ema) - volume_25_sma
    return heuristics_matrix
