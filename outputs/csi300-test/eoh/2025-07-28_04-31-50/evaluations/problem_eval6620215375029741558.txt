def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    volume_5_ema = df['volume'].ewm(span=5, adjust=False).mean()
    amount_20_logprod = (np.log(df['amount'] + 1)).rolling(window=20).apply(np.prod, raw=True)
    heuristics_matrix = (close_10_ema / volume_5_ema) - amount_20_logprod
    return heuristics_matrix
