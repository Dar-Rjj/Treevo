def heuristics_v2(df):
    # Calculate the 20-day simple moving average of the closing price
    sma_20 = df['close'].rolling(window=20).mean()
    # Calculate the ratio of the current closing price to its 20-day SMA
    price_strength = df['close'] / sma_20
    # Calculate the change in the close price
    price_change = df['close'].diff()
    # Calculate the 5-day moving average of the price change
    avg_price_change_5 = price_change.rolling(window=5).mean()
    # Combine the price strength with the 5-day average price change for the alpha factor
    heuristics_matrix = price_strength * avg_price_change_5
    return heuristics_matrix
