def heuristics_v2(df):
    open_10_ma = df['open'].rolling(window=10).mean()
    close_20_ma = df['close'].rolling(window=20).mean()
    high_5_ema = df['high'].ewm(span=5, adjust=False).mean()
    low_15_ema = df['low'].ewm(span=15, adjust=False).mean()
    heuristics_matrix = (open_10_ma * close_20_ma) - (high_5_ema - low_15_ema)
    return heuristics_matrix
