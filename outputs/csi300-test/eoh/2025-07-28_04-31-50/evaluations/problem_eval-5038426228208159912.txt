    # Short-term moving average
    short_ma = df['close'].rolling(window=10).mean()
    # Long-term moving average
    long_ma = df['close'].rolling(window=50).mean()
    # Momentum factor
    momentum = (df['close'] - df['close'].shift(12)) / df['close'].shift(12)
    # Volatility, represented through the standard deviation of returns
    volatility = df['close'].pct_change().rolling(window=30).std()
    # Composite heuristic combining momentum, and relative position between short and long MAs
    heuristics_matrix = 0.5 * ((short_ma - long_ma) / long_ma) + 0.3 * momentum - 0.2 * volatility
    return heuristics_matrix
