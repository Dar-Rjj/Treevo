def heuristics_v2(df):
    high_10_ema = df['high'].ewm(span=10, adjust=False).mean()
    low_20_ema = df['low'].ewm(span=20, adjust=False).mean()
    amount_5_ema = df['amount'].ewm(span=5, adjust=False).mean()
    volume_30_ema = df['volume'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (high_10_ema / low_20_ema) + (amount_5_ema / volume_30_ema)
    return heuristics_matrix
