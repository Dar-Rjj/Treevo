import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 20-day moving average of close price
    ma_20 = df['close'].rolling(window=20).mean()

    # Define the heuristic factor as a combination of RSI and 20-day moving average
    heuristics_matrix = (ma_20 * 0.6) + (rsi * 0.4)
    return heuristics_matrix
