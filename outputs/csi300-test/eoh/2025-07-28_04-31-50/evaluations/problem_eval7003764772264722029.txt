def heuristics_v2(df):
    close_7_ema = df['close'].ewm(span=7, adjust=False).mean()
    open_15_ema = df['open'].ewm(span=15, adjust=False).mean()
    volume_30_ma = df['volume'].rolling(window=30).mean()
    amount_40_ema = df['amount'].ewm(span=40, adjust=False).mean()
    heuristics_matrix = (close_7_ema / open_15_ema) * (volume_30_ma + amount_40_ema)
    return heuristics_matrix
