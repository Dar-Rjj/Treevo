def heuristics_v2(df):
    high_20_ma = df['high'].rolling(window=20).mean()
    low_10_ma = df['low'].rolling(window=10).mean()
    amount_15_sum = df['amount'].rolling(window=15).sum()
    volume_30_sum = df['volume'].rolling(window=30).sum()
    heuristics_matrix = (high_20_ma / low_10_ma) + (amount_15_sum - volume_30_sum)
    return heuristics_matrix
