def heuristics_v2(df):
    high_low_diff = df['high'] - df['low']
    high_low_5_std = high_low_diff.rolling(window=5).std()
    close_10_mom = df['close'] - df['close'].shift(10)
    volume_20_roc = (df['volume'] / df['volume'].shift(20) - 1) * 100
    heuristics_matrix = high_low_5_std + close_10_mom - volume_20_roc
    return heuristics_matrix
