def heuristics_v2(df):
    open_10_ma = df['open'].rolling(window=10).mean()
    close_20_ma = df['close'].rolling(window=20).mean()
    volume_5_std = df['volume'].rolling(window=5).std()
    amount_15_sum = df['amount'].rolling(window=15).sum()
    heuristics_matrix = (open_10_ma - close_20_ma) + (volume_5_std + amount_15_sum)
    return heuristics_matrix
