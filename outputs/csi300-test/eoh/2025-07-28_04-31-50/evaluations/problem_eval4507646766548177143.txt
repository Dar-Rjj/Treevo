def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    low_15_ma = df['low'].rolling(window=15).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    amount_5_ma = df['amount'].rolling(window=5).mean()
    heuristics_matrix = (close_10_ma / low_15_ma) + (amount_5_ma + volume_20_ema)
    return heuristics_matrix
