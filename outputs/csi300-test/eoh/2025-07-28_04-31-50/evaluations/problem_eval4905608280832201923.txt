def heuristics_v2(df):
    amount_5_ema = df['amount'].ewm(span=5, adjust=False).mean()
    volume_10_sma = df['volume'].rolling(window=10).mean()
    close_20_ema = df['close'].ewm(span=20, adjust=False).mean()
    high_3_sma = df['high'].rolling(window=3).mean()
    low_7_sma = df['low'].rolling(window=7).mean()
    sma_diff = high_3_sma - low_7_sma
    heuristics_matrix = ((amount_5_ema * volume_10_sma) / close_20_ema) + sma_diff
    return heuristics_matrix
