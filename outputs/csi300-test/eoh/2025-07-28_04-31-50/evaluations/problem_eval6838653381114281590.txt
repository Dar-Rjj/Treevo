def heuristics_v2(df):
    # Calculate 10-day SMA of close prices
    sma_10 = df['close'].rolling(window=10).mean()
    # Rate of change in 10-day SMA
    sma_change_rate = sma_10.pct_change()
    
    # Ratio of today's volume to 20-day max volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).max()
    
    # Distance between the current close and the 10-day high
    high_10 = df['high'].rolling(window=10).max()
    high_distance = df['close'] - high_10
    
    # Combine into a single heuristics matrix
    heuristics_matrix = sma_change_rate + volume_ratio + high_distance
    return heuristics_matrix
