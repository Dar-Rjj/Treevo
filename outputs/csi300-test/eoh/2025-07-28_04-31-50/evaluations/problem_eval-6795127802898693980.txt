def heuristics_v2(df):
    close_20_ma = df['close'].rolling(window=20).mean()
    open_15_ma = df['open'].rolling(window=15).mean()
    volume_10_ema = df['volume'].ewm(span=10, adjust=False).mean()
    amount_30_ma = df['amount'].rolling(window=30).mean()
    heuristics_matrix = (close_20_ma / open_15_ma) + volume_10_ema - amount_30_ma
    return heuristics_matrix
