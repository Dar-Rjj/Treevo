def heuristics_v2(df):
    amount_15_ma = df['amount'].rolling(window=15).mean()
    volume_25_ma = df['volume'].rolling(window=25).mean()
    high_7_ma = df['high'].rolling(window=7).mean()
    low_7_ma = df['low'].rolling(window=7).mean()
    heuristics_matrix = (amount_15_ma / volume_25_ma) + (high_7_ma - low_7_ma)
    return heuristics_matrix
