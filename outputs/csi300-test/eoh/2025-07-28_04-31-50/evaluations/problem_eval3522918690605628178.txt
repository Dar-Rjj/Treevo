def heuristics_v2(df):
    high_7_ma = df['high'].rolling(window=7).mean()
    low_14_ma = df['low'].rolling(window=14).mean()
    amount_25_ma = df['amount'].rolling(window=25).mean()
    volume_35_ma = df['volume'].rolling(window=35).mean()
    heuristics_matrix = (high_7_ma / low_14_ma) + (amount_25_ma - volume_35_ma)
    return heuristics_matrix
