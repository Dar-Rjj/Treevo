def heuristics_v2(df):
    # Calculate the 30-day high-low range
    high_low_range = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    
    # Calculate the 30-day sum of trading volume
    sum_volume = df['volume'].rolling(window=30).sum()
    
    # Combine the factors into a single score
    heuristics_matrix = high_low_range / sum_volume
    
    return heuristics_matrix
