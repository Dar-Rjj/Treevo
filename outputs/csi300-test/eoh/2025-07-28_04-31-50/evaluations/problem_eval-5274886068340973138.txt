def heuristics_v2(df):
    close_15_ema = df['close'].ewm(span=15, adjust=False).mean()
    volume_5_ema = df['volume'].ewm(span=5, adjust=False).mean()
    amount_20_sum = df['amount'].rolling(window=20).sum()
    heuristics_matrix = (close_15_ema / volume_5_ema) - amount_20_sum
    return heuristics_matrix
