def heuristics_v2(df):
    close_5_ema = df['close'].ewm(span=5, adjust=False).mean()
    volume_10_ema = df['volume'].ewm(span=10, adjust=False).mean()
    high_3_max = df['high'].rolling(window=3).max()
    low_3_min = df['low'].rolling(window=3).min()
    heuristics_matrix = (close_5_ema * volume_10_ema) + (high_3_max - low_3_min)
    return heuristics_matrix
