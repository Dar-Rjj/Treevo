def heuristics_v2(df):
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    volume_20_ema = df['volume'].ewm(span=20, adjust=False).mean()
    high_5_std = df['high'].rolling(window=5).std()
    heuristics_matrix = (close_10_ema * volume_20_ema) - high_5_std
    return heuristics_matrix
