import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 20-day moving average of close price
    ma_20 = df['close'].rolling(window=20).mean()

    # Calculate the 5-day moving average of volume
    vol_5 = df['volume'].rolling(window=5).mean()

    # Define the heuristic factor as a combination of RSI, 20-day moving average, and 5-day volume moving average
    heuristics_matrix = (ma_20 * 0.5) + (rsi * 0.3) + (vol_5 * 0.2)
    return heuristics_matrix
