import pandas as pd

def heuristics_v2(df):
    # Calculate the high-low price spread
    spread = df['high'] - df['low']
    
    # Calculate the daily change in the spread
    spread_change = spread.diff()
    
    # Calculate the RSI of the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the spread change with the RSI for the alpha factor
    heuristics_matrix = spread_change * rsi
    
    return heuristics_matrix
