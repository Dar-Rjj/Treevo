def heuristics_v2(df):
    # Calculate the 10-day momentum in close price
    momentum_10 = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day moving average of volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate the current volume relative to its 20-day moving average
    rel_strength_volume = df['volume'] / avg_volume_20
    
    # Define the heuristic factor as a combination of momentum and relative volume strength
    heuristics_matrix = (momentum_10 * 0.7) + (rel_strength_volume * 0.3)
    
    return heuristics_matrix
