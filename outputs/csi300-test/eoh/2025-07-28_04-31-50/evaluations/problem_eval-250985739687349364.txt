def heuristics_v2(df):
    close_15_ma = df['close'].rolling(window=15).mean()
    volume_30_ma = df['volume'].rolling(window=30).mean()
    high_10_max = df['high'].rolling(window=10).max()
    low_10_min = df['low'].rolling(window=10).min()
    heuristics_matrix = (close_15_ma + volume_30_ma) / (high_10_max - low_10_min)
    return heuristics_matrix
