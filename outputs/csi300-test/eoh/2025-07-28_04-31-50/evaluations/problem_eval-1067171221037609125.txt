def heuristics_v2(df):
    close_10_momentum = df['close'].pct_change(periods=10)
    amount_5_mean = df['amount'].rolling(window=5).mean()
    volume_20_std = df['volume'].rolling(window=20).std()
    amount_to_volume_std = amount_5_mean / volume_20_std
    heuristics_matrix = close_10_momentum * amount_to_volume_std
    return heuristics_matrix
