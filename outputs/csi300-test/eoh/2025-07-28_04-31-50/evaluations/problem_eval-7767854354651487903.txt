def heuristics_v2(df):
    close_15_sma = df['close'].rolling(window=15).mean()
    open_20_ema = df['open'].ewm(span=20, adjust=False).mean()
    high_5_sma = df['high'].rolling(window=5).mean()
    low_5_sma = df['low'].rolling(window=5).mean()
    sma_diff = high_5_sma - low_5_sma
    heuristics_matrix = (close_15_sma / open_20_ema) * sma_diff
    return heuristics_matrix
