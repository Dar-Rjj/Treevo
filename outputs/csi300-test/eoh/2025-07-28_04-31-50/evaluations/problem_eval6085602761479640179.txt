import pandas as pd

def heuristics_v2(df):
    close_10_ma = df['close'].rolling(window=10).mean()
    close_50_ma = df['close'].rolling(window=50).mean()
    high_low_std = (df['high'] - df['low']).rolling(window=20).std()
    heuristics_matrix = (close_10_ma - close_50_ma) / high_low_std
    return heuristics_matrix
