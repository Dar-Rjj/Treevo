def heuristics_v2(df):
    close_open_ratio = df['close'] / df['open']
    ema_close_open_14 = close_open_ratio.ewm(span=14, adjust=False).mean()
    ema_log_volume_28 = np.log(df['volume']).ewm(span=28, adjust=False).mean()
    amount_5_std = df['amount'].rolling(window=5).std()
    heuristics_matrix = (ema_close_open_14 / ema_log_volume_28) + amount_5_std
    return heuristics_matrix
