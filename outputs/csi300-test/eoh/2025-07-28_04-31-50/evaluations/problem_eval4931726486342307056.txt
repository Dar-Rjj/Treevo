import pandas as pd

def heuristics_v2(df):
    close_5_wma = df['close'].rolling(window=5).apply(lambda x: (x * pd.Series([1, 2, 3, 4, 5], index=x.index)).sum() / 15, raw=False)
    open_20_ema = df['open'].ewm(span=20, adjust=False).mean()
    high_14_ma = df['high'].rolling(window=14).mean()
    low_28_ma = df['low'].rolling(window=28).mean()
    heuristics_matrix = (close_5_wma - open_20_ema) + (high_14_ma - low_28_ma)
    return heuristics_matrix
