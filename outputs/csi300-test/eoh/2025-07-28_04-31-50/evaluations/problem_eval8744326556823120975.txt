def heuristics_v2(df):
    close_15_mean = df['close'].rolling(window=15).mean()
    volume_7_ema = df['volume'].ewm(span=7).mean()
    amount_10_logcumprod = np.log(df['amount']).rolling(window=10).apply(np.prod, raw=True)
    heuristics_matrix = (close_15_mean / volume_7_ema) - amount_10_logcumprod
    return heuristics_matrix
