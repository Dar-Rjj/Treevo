def heuristics_v2(df):
    high_15_ema = df['high'].ewm(span=15, adjust=False).mean()
    low_30_ema = df['low'].ewm(span=30, adjust=False).mean()
    close_20_ma = df['close'].rolling(window=20).mean()
    open_50_ma = df['open'].rolling(window=50).mean()
    heuristics_matrix = (high_15_ema - low_30_ema) * (close_20_ma / open_50_ma)
    return heuristics_matrix
