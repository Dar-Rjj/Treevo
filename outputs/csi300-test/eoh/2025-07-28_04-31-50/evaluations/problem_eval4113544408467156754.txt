def heuristics_v2(df):
    amount_5_ma = df['amount'].rolling(window=5).mean()
    volume_10_ma = df['volume'].rolling(window=10).mean()
    high_3_ma = df['high'].rolling(window=3).mean()
    low_3_ma = df['low'].rolling(window=3).mean()
    heuristics_matrix = (amount_5_ma / volume_10_ma) - (high_3_ma - low_3_ma)
    return heuristics_matrix
