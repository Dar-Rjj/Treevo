def heuristics_v2(df):
    close_open_diff = df['close'] - df['open']
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    vol_change = df['volume'] / avg_vol_10
    weighted_average = (close_open_diff * vol_change).rolling(window=10).sum()
    heuristics_matrix = weighted_average
    return heuristics_matrix
