def heuristics_v2(df):
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    avg_vol_100 = df['volume'].rolling(window=100).mean()
    close_change_10_avg = df['close'].pct_change(periods=10).rolling(window=10).mean()
    heuristics_matrix = (avg_vol_20 / avg_vol_100) * (1 + close_change_10_avg)
    return heuristics_matrix
