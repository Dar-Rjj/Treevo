def heuristics_v2(df):
    close_avg_50 = df['close'].rolling(window=50).mean()
    close_avg_200 = df['close'].rolling(window=200).mean()
    vol_change_10_std = df['volume'].pct_change().rolling(window=10).std()
    heuristics_matrix = (close_avg_50 / close_avg_200) * (1 + vol_change_10_std)
    return heuristics_matrix
