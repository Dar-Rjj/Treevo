def heuristics_v2(df):
    # Calculate the rate of change of closing prices
    roc_close = df['close'].pct_change()
    # Calculate the relative strength of volume compared to its 10-day average
    avg_volume = df['volume'].rolling(window=10).mean()
    rel_strength_volume = df['volume'] / avg_volume
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (roc_close * rel_strength_volume).dropna()
    return heuristics_matrix
