import pandas as pd

def heuristics_v2(df):
    def calculate_daily_returns(prices):
        return prices.pct_change().dropna()
    
    daily_returns = calculate_daily_returns(df['close'])
    ma_10 = daily_returns.rolling(window=10).mean()
    ewma_10 = daily_returns.ewm(span=10).mean()
    divergence = ma_10 - ewma_10
    heuristics_matrix = pd.Series(divergence, index=df.index[10:])
    return heuristics_matrix
