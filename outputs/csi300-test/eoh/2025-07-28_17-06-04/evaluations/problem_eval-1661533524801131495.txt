import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change of volume
    roc_volume = df['volume'].pct_change()
    # Calculate the average true range over a 14-day window
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (roc_volume + atr).dropna()
    return heuristics_matrix
