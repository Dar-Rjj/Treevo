import pandas as pd

def heuristics_v2(df):
    positive_diff = (df['close'] - df['open']).apply(lambda x: max(x, 0))
    positive_sum = positive_diff.rolling(window=10).sum()
    range_diff = df['high'] - df['low']
    range_sum = range_diff.rolling(window=10).sum()
    heuristics_matrix = positive_sum / range_sum
    return heuristics_matrix
