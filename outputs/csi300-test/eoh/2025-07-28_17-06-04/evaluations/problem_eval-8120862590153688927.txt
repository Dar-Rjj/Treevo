def heuristics_v2(df):
    avg_price_range_60 = (df['high'] - df['low']).rolling(window=60).mean()
    avg_price_range_200 = (df['high'] - df['low']).rolling(window=200).mean()
    close_momentum_30 = df['close'].pct_change(periods=30)
    heuristics_matrix = (avg_price_range_60 / avg_price_range_200) * (1 + close_momentum_30)
    return heuristics_matrix
