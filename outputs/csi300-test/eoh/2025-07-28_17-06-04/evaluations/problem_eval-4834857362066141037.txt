def heuristics_v2(df):
    daily_high_low_diff = df['high'] - df['low']
    sum_daily_high_low_diff_60 = daily_high_low_diff.rolling(window=60).sum()
    avg_close_300 = df['close'].rolling(window=300).mean()
    heuristics_matrix = sum_daily_high_low_diff_60 / avg_close_300
    return heuristics_matrix
