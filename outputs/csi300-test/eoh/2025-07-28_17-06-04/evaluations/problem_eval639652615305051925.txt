import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages
    short_window = 5
    long_window = 20
    sma_short = df['close'].rolling(window=short_window).mean()
    sma_long = df['close'].rolling(window=long_window).mean()
    
    # Momentum factor
    momentum = (df['close'] / df['close'].shift(long_window)) - 1
    
    # Volatility (standard deviation of daily returns)
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=long_window).std()
    
    # Combining factors into a single heuristic score
    heuristics_matrix = ((sma_short - sma_long) / sma_long) + momentum - volatility
    
    return heuristics_matrix
