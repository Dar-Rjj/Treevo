import pandas as pd

def heuristics_v2(df):
    # Calculate the RSI for the close price
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the coefficient of variation for the volume
    vol_mean = df['volume'].rolling(window=14).mean()
    vol_std = df['volume'].rolling(window=14).std()
    cv = vol_std / vol_mean
    
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (rsi + cv).dropna()
    return heuristics_matrix
