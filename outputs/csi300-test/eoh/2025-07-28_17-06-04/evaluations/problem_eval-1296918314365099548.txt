def heuristics_v2(df):
    daily_returns = df['close'].pct_change()
    std_20 = daily_returns.rolling(window=20).std()
    avg_return_20 = daily_returns.rolling(window=20).mean()
    med_return_20 = daily_returns.rolling(window=20).median()
    heuristics_matrix = std_20 * (avg_return_20 / med_return_20)
    return heuristics_matrix
