import pandas as pd

def heuristics_v2(df):
    daily_range = df['high'] - df['low']
    daily_return = df['close'].pct_change().fillna(0)
    ewma_daily_range = (daily_range * (1 + abs(daily_return))).ewm(span=20, adjust=False).mean()
    heuristics_matrix = ewma_daily_range
    return heuristics_matrix
