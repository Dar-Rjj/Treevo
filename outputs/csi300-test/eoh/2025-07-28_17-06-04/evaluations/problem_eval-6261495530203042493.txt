import pandas as pd

def heuristics_v2(df):
    price_change = df['close'] - df['open']
    ewma_price_change = price_change.ewm(span=10, adjust=False).mean()
    ewma_volume = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (ewma_price_change / ewma_volume).rename('heuristics_factor')
    return heuristics_matrix
