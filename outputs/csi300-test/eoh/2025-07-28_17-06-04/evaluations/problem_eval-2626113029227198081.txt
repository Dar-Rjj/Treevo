import pandas as pd

def heuristics_v2(df):
    ema_close_short = df['close'].ewm(span=5, adjust=False).mean()
    ema_close_long = df['close'].ewm(span=20, adjust=False).mean()
    ema_volume_short = df['volume'].ewm(span=5, adjust=False).mean()
    ema_volume_long = df['volume'].ewm(span=20, adjust=False).mean()
    price_range = df['high'] - df['low']
    heuristics_matrix = (ema_close_short / ema_close_long) * (ema_volume_short / ema_volume_long) * price_range
    return heuristics_matrix
