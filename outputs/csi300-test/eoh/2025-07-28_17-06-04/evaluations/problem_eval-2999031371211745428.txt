def heuristics_v2(df):
    avg_vol_60 = df['volume'].rolling(window=60).mean()
    avg_vol_180 = df['volume'].rolling(window=180).mean()
    close_change_30_ema = df['close'].pct_change(periods=30).ewm(span=30, adjust=False).mean()
    heuristics_matrix = (avg_vol_60 / avg_vol_180) * (1 + close_change_30_ema)
    return heuristics_matrix
