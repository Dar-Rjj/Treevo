def heuristics_v2(df):
    ema_vol_50 = df['volume'].ewm(span=50, adjust=False).mean()
    ema_vol_100 = df['volume'].ewm(span=100, adjust=False).mean()
    std_close_60 = df['close'].rolling(window=60).std()
    avg_close_60 = df['close'].rolling(window=60).mean()
    cv_close_60 = std_close_60 / avg_close_60
    heuristics_matrix = (ema_vol_50 / ema_vol_100) * cv_close_60
    return heuristics_matrix
