def heuristics_v2(df):
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    avg_vol_200 = df['volume'].rolling(window=200).mean()
    vol_weighted_close_20 = (df['close'] * df['volume']).rolling(window=20).mean() / avg_vol_20
    heuristics_matrix = vol_weighted_close_20 / avg_vol_200
    return heuristics_matrix
