def heuristics_v2(df):
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    avg_vol_100 = df['volume'].rolling(window=100).mean()
    close_change_60 = df['close'].pct_change(periods=60)
    adjustment_factor = (1 + close_change_60.abs().pow(1/3))
    heuristics_matrix = (avg_vol_20 / avg_vol_100) * adjustment_factor
    return heuristics_matrix
