def heuristics_v2(df):
    close_30 = df['close'].rolling(window=30).mean()
    avg_vol_30 = df['volume'].rolling(window=30).mean()
    avg_vol_150 = df['volume'].rolling(window=150).mean()
    volat_30 = df['close'].pct_change().rolling(window=30).std()
    heuristics_matrix = (close_30 * (avg_vol_30 / avg_vol_150)) * (1 + volat_30)
    return heuristics_matrix
