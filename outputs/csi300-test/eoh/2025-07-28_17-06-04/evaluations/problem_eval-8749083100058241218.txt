import pandas as pd

def heuristics_v2(df):
    # Compute rolling mean and std for closing prices
    roll_mean_close = df['close'].rolling(window=20).mean()
    roll_std_close = df['close'].rolling(window=20).std()
    
    # Calculate average volume differences over short (10 days) and long (40 days) windows
    avg_vol_short = df['volume'].rolling(window=10).mean()
    avg_vol_long = df['volume'].rolling(window=40).mean()
    vol_trend = avg_vol_short - avg_vol_long
    
    # Generate the heuristics by combining the calculated metrics
    heuristics_matrix = 0.6 * roll_mean_close + 0.4 * roll_std_close + vol_trend
    
    return heuristics_matrix
