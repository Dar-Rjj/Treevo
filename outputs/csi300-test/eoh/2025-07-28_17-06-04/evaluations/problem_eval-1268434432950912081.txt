def heuristics_v2(df):
    close_50 = df['close'].rolling(window=50).mean()
    close_250 = df['close'].rolling(window=250).mean()
    vol_ratio = df['volume'].rolling(window=10).mean() / df['volume'].rolling(window=100).mean()
    heuristics_matrix = (close_50 / close_250) * vol_ratio
    return heuristics_matrix
