def heuristics_v2(df):
    # Calculate the percentage change in close prices
    pct_change_close = df['close'].pct_change()
    # Calculate the standard deviation of daily returns over a 30-day window
    std_daily_returns = pct_change_close.rolling(window=30).std()
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (pct_change_close + std_daily_returns).dropna()
    return heuristics_matrix
