def heuristics_v2(df):
    std_close_30 = df['close'].rolling(window=30).std()
    std_close_150 = df['close'].rolling(window=150).std()
    avg_turnover_30 = (df['volume'] / df['amount']).rolling(window=30).mean()
    heuristics_matrix = (std_close_30 / std_close_150) * avg_turnover_30
    return heuristics_matrix
