def heuristics_v2(df):
    def price_momentum(close, period=10):
        return close.pct_change(period)

    def relative_strength(volume, period_short=20, period_long=50):
        avg_vol_short = volume.rolling(window=period_short).mean()
        avg_vol_long = volume.rolling(window=period_long).mean()
        return avg_vol_short / avg_vol_long

    momentum = price_momentum(df['close'])
    rs = relative_strength(df['volume'])
    
    heuristics_matrix = momentum + rs
    return heuristics_matrix
