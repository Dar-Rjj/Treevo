def heuristics_v2(df):
    std_daily_returns_60 = df['close'].pct_change().rolling(window=60).std()
    sma_close_10 = df['close'].rolling(window=10).mean()
    sma_close_50 = df['close'].rolling(window=50).mean()
    heuristics_matrix = std_daily_returns_60 * (sma_close_10 / sma_close_50)
    return heuristics_matrix
