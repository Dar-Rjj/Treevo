def heuristics_v2(df):
    avg_amount_30 = df['amount'].rolling(window=30).mean()
    avg_vol_150 = df['volume'].rolling(window=150).mean()
    high_change_20 = df['high'].pct_change(periods=20)
    heuristics_matrix = (avg_amount_30 / avg_vol_150) * (1 + high_change_20)
    return heuristics_matrix
