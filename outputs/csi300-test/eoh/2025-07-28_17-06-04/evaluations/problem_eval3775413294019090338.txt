import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    # Calculate the log of the volume
    log_volume = np.log(df['volume'])
    # Compute the exponential moving average of the daily return, weighted by the log of the volume
    ema_weighted_return = (daily_return * log_volume).ewm(span=10, adjust=False).mean()
    heuristics_matrix = ema_weighted_return.dropna()
    return heuristics_matrix
