def heuristics_v2(df):
    avg_vol_50 = df['volume'].rolling(window=50).mean()
    avg_vol_200 = df['volume'].rolling(window=200).mean()
    close_change_ema_10 = df['close'].pct_change().ewm(span=10, adjust=False).mean()
    heuristics_matrix = (avg_vol_50 / avg_vol_200) * (1 + close_change_ema_10)
    return heuristics_matrix
