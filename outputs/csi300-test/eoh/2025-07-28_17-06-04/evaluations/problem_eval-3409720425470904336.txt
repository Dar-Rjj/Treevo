def heuristics_v2(df):
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    high_low_diff_90 = df['high'].rolling(window=90).max() - df['low'].rolling(window=90).min()
    std_dev_60 = df['close'].pct_change().rolling(window=60).std()
    heuristics_matrix = (ema_20 + high_low_diff_90) * (1 + std_dev_60)
    return heuristics_matrix
