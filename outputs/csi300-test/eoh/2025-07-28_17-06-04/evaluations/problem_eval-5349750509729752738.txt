def heuristics_v2(df):
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the standard deviation of daily returns over a 20-day period
    std_dev_returns = daily_returns.rolling(window=20).std()
    
    # Calculate the highest high and the lowest low over a 20-day period
    highest_high = df['high'].rolling(window=20).max()
    lowest_low = df['low'].rolling(window=20).min()
    
    # Calculate the ratio of the highest high to the lowest low
    high_low_ratio = highest_high / lowest_low
    
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (std_dev_returns + high_low_ratio).dropna()
    
    return heuristics_matrix
