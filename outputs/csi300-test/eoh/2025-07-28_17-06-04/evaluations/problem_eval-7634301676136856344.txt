def heuristics_v2(df):
    ema_close_short = df['close'].ewm(span=10, adjust=False).mean()
    ema_close_long = df['close'].ewm(span=30, adjust=False).mean()
    ema_volume_short = df['volume'].ewm(span=10, adjust=False).mean()
    ema_volume_long = df['volume'].ewm(span=30, adjust=False).mean()
    price_range_normalized = (df['high'] - df['low']) / df['open']
    heuristics_matrix = (ema_close_short / ema_close_long) * (ema_volume_short / ema_volume_long) * price_range_normalized
    return heuristics_matrix
