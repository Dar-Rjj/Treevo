def heuristics_v2(df):
    # Calculate the rate of change of volume
    roc_volume = df['volume'].pct_change()
    # Apply an exponential moving average to the rate of change of volume
    ema_roc_volume = roc_volume.ewm(span=21, adjust=False).mean()
    # Calculate the true range
    tr = df['high'] - df['low']
    # Calculate the simple moving average of the true range over a 7-day window
    sma_tr = tr.rolling(window=7).mean()
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (ema_roc_volume + sma_tr).dropna()
    return heuristics_matrix
