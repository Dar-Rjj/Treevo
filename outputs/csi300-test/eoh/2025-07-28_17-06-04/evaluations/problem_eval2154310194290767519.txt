def heuristics_v2(df):
    avg_close_30 = df['close'].rolling(window=30).mean()
    avg_close_150 = df['close'].rolling(window=150).mean()
    vol_change_30 = df['volume'].pct_change(periods=30)
    heuristics_matrix = (avg_close_30 / avg_close_150) * (1 + vol_change_30)
    return heuristics_matrix
