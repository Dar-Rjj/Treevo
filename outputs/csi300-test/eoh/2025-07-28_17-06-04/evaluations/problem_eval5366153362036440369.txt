def heuristics_v2(df):
    avg_close_50 = df['close'].rolling(window=50).mean()
    avg_close_200 = df['close'].rolling(window=200).mean()
    daily_returns_30 = df['close'].pct_change().rolling(window=30).std()
    heuristics_matrix = (avg_close_50 / avg_close_200) * (1 + daily_returns_30)
    return heuristics_matrix
