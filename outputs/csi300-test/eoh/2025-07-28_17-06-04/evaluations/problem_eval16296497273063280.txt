import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change of the closing price
    roc_close = df['close'].pct_change()
    # Calculate the true range
    tr = df['high'] - df['low']
    # Calculate the exponentially weighted moving average of the true range over a 14-day window
    ewma_tr = tr.ewm(span=14).mean()
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (roc_close + ewma_tr).dropna()
    return heuristics_matrix
