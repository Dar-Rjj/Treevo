import pandas as pd

def heuristics_v2(df):
    daily_range = df['high'] - df['low']
    pct_change_close = df['close'].pct_change().fillna(0)
    weighted_factor = df['volume'] * (1 + pct_change_close)
    weighted_average = (daily_range * weighted_factor).rolling(window=20).sum() / weighted_factor.rolling(window=20).sum()
    heuristics_matrix = weighted_average
    return heuristics_matrix
