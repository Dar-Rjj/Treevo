def heuristics_v2(df):
    close_ma_50 = df['close'].rolling(window=50).mean()
    close_ma_200 = df['close'].rolling(window=200).mean()
    avg_vol_50 = df['volume'].rolling(window=50).mean()
    avg_vol_200 = df['volume'].rolling(window=200).mean()
    vol_ratio = avg_vol_50 / avg_vol_200
    price_diff = close_ma_50 - close_ma_200
    heuristics_matrix = price_diff * vol_ratio
    return heuristics_matrix
