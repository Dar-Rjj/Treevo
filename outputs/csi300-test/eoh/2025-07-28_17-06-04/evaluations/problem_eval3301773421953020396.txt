import pandas as pd

def heuristics_v2(df):
    # Calculate the 7-day rate of change of volume
    roc_volume = df['volume'].pct_change(periods=7)
    # Calculate the 21-day average true range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=21).mean()
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (roc_volume + atr).dropna()
    return heuristics_matrix
