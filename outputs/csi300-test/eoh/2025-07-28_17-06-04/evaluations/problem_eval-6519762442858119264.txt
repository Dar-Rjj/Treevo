def heuristics_v2(df):
    # Calculate the 14-day rate of change of the closing price
    roc_close = df['close'].pct_change(periods=14)
    # Calculate the 14-day standard deviation of the high-low spread
    std_hl = (df['high'] - df['low']).rolling(window=14).std()
    # Combine the two factors to create a new heuristics matrix
    heuristics_matrix = (roc_close + std_hl).dropna()
    return heuristics_matrix
