def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling mean of the daily range
    rolling_mean_range = daily_range.rolling(window=20).mean()
    
    # Compute the simple moving average of the volume
    sma_volume = df['volume'].rolling(window=20).mean()
    
    # Generate the heuristic factor by dividing the rolling mean with SMA volume
    heuristics_matrix = rolling_mean_range / sma_volume
    
    return heuristics_matrix
