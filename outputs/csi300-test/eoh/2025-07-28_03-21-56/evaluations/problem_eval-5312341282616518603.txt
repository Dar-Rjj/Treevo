def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling standard deviation of the daily range
    rolling_std_range = daily_range.rolling(window=20).std()
    
    # Generate the heuristic factor by incorporating the square root of the volume
    heuristics_matrix = rolling_std_range * df['volume'].pow(0.5)
    
    return heuristics_matrix
