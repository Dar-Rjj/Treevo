def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling median of the daily range
    rolling_median_range = daily_range.rolling(window=20).median()
    
    # Generate the heuristic factor by incorporating the logarithm of the volume
    heuristics_matrix = rolling_median_range * np.log(df['volume'])
    
    return heuristics_matrix
