def heuristics_v2(df):
    def compute_ppo(series, short_window=12, long_window=26, signal_window=9):
        ema_short = series.ewm(span=short_window, adjust=False).mean()
        ema_long = series.ewm(span=long_window, adjust=False).mean()
        ppo_line = (ema_short - ema_long) / ema_long * 100
        return ppo_line

    def compute_vwap(high, low, close, volume):
        typical_price = (high + low + close) / 3
        vwap = (typical_price * volume).cumsum() / volume.cumsum()
        return vwap

    df['ppo'] = compute_ppo(df['close'])
    df['vwap'] = compute_vwap(df['high'], df['low'], df['close'], df['volume'])
    heuristics_matrix = df['ppo'] + df['vwap']
    return heuristics_matrix
