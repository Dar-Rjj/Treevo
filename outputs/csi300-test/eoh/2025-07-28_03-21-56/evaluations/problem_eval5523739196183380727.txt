import pandas as pd

def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    roc_high_low = (high_low_range / high_low_range.shift(2) - 1).fillna(0)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])) * df['volume']
    cum_adl_change = adl.diff().cumsum().fillna(0)
    heuristics_matrix = roc_high_low + cum_adl_change
    return heuristics_matrix
