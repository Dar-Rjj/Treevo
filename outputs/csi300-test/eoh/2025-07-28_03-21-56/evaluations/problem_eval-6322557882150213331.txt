def heuristics_v2(df):
    df['close_open_diff'] = df['close'] - df['open']
    df['positive_diff'] = df['close_open_diff'].where(df['close_open_diff'] > 0, 0)
    df['negative_diff'] = df['close_open_diff'].where(df['close_open_diff'] < 0, 0).abs()
    df['sum_positive_diff'] = df['positive_diff'].rolling(window=14).sum()
    df['sum_negative_diff'] = df['negative_diff'].rolling(window=14).sum()
    df['close_std'] = df['close'].rolling(window=14).std()
    heuristics_matrix = (df['sum_positive_diff'] + df['sum_negative_diff']) / df['close_std']
    return heuristics_matrix
