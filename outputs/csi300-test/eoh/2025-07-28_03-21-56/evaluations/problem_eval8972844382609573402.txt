def heuristics_v2(df):
    # Calculate the rolling standard deviation of the high and low prices
    rolling_std_high = df['high'].rolling(window=10).std()
    rolling_std_low = df['low'].rolling(window=10).std()
    
    # Average the two standard deviations
    avg_rolling_std = (rolling_std_high + rolling_std_low) / 2
    
    # Generate the heuristic factor by incorporating the square root of volume
    heuristics_matrix = avg_rolling_std * np.sqrt(df['volume'])
    
    return heuristics_matrix
