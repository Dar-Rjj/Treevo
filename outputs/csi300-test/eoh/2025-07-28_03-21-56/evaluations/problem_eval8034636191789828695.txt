import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling mean of the daily range
    rolling_mean_range = daily_range.rolling(window=30).mean()
    
    # Apply a moving average to the log of volume
    log_volume_rolling = np.log(df['volume']).rolling(window=30).mean()
    
    # Generate the heuristic factor by incorporating the smoothed log of volume
    heuristics_matrix = rolling_mean_range * log_volume_rolling
    
    return heuristics_matrix
