import pandas as pd

def heuristics_v2(df):
    avg_price = (df['high'] + df['low']) / 2
    rel_strength = df['close'] / avg_price.rolling(window=10).mean()
    volatility = df['close'].pct_change().rolling(window=10).std()
    heuristics_matrix = rel_strength * volatility
    return heuristics_matrix
