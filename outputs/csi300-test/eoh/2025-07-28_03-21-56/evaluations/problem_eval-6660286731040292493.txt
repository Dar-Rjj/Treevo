import pandas as pd

def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    ema_high_low = high_low_range.ewm(span=10, adjust=False).mean()
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    money_flow_index = 100 - (100 / (1 + money_ratio))
    heuristics_matrix = ema_high_low + money_flow_index
    return heuristics_matrix
