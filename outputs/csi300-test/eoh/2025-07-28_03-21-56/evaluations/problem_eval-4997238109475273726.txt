def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the exponentially weighted moving average of the daily range
    ewma_range = daily_range.ewm(span=20, adjust=False).mean()
    
    # Generate the heuristic factor by incorporating the square root of the volume
    heuristics_matrix = ewma_range * df['volume'].pow(0.5)
    
    return heuristics_matrix
