def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) of the close price
    roc_close = df['close'].pct_change(periods=10)
    
    # Compute the 30-day average volume
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of current volume to the 30-day average volume
    vol_ratio = df['volume'] / avg_vol_30d
    
    # Calculate the natural logarithm of the volume ratio
    log_vol_ratio = np.log(vol_ratio)
    
    # Apply an EMA on the ROC with a span of 10 days
    ema_roc_close = roc_close.ewm(span=10, adjust=False).mean()
    
    # Generate the heuristic factor by combining EMA of ROC and logarithmic volume ratio
    heuristics_matrix = ema_roc_close * log_vol_ratio
    
    return heuristics_matrix
