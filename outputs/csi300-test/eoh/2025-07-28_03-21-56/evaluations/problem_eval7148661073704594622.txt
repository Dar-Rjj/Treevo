import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for close prices over a 10-day period
    roc = df['close'].pct_change(periods=10)
    # Compute the rolling standard deviation of daily returns over a 10-day window
    vol = df['close'].pct_change().rolling(window=10).std()
    # Generate the heuristics matrix by combining ROC and volatility
    heuristics_matrix = roc * vol
    return heuristics_matrix
