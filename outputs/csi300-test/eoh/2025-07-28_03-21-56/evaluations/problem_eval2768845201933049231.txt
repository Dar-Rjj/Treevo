def heuristics_v2(df):
    # Calculate short-term and long-term moving averages of the close price
    ma_short = df['close'].rolling(window=10).mean()
    ma_long = df['close'].rolling(window=30).mean()
    
    # Calculate the weighted sum of the moving averages
    weighted_ma = 0.7 * ma_short + 0.3 * ma_long
    
    # Calculate the change in volume
    vol_change = df['volume'].diff()
    
    # Generate the heuristic factor by combining the weighted MA and volume change
    heuristics_matrix = weighted_ma + vol_change
    
    return heuristics_matrix
