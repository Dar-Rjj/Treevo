import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    ma_short = close.rolling(window=10).mean()
    ma_long = close.rolling(window=50).mean()
    std_dev = close.rolling(window=30).std()
    momentum = close / close.shift(1) - 1
    rsi = 100 - (100 / (1 + (close.diff().where(lambda x: x > 0).fillna(0).rolling(window=14).mean() / 
                                                -close.diff().where(lambda x: x < 0).fillna(0).rolling(window=14).mean())))
    vol_change = volume.pct_change()
    heuristics_matrix = (ma_short / ma_long) * (rsi / 100) + (momentum / std_dev) * vol_change
    return heuristics_matrix
