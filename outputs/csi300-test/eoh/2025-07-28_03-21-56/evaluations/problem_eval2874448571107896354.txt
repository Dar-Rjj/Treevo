def heuristics_v2(df):
    # Calculate the 10-day moving average of the close price
    ma_close_10d = df['close'].rolling(window=10).mean()
    
    # Compute the difference between the 10-day moving average and the current close price
    diff_ma_close = ma_close_10d - df['close']
    
    # Calculate the 30-day average volume
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of current volume to the 30-day average volume
    vol_ratio_avg = df['volume'] / avg_vol_30d
    
    # Generate the heuristic factor by combining the difference and volume ratio
    heuristics_matrix = diff_ma_close * vol_ratio_avg
    
    return heuristics_matrix
