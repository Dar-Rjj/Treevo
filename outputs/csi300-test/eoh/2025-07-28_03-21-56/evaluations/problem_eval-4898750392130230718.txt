def heuristics_v2(df):
    close = df['close']
    high = df['high']
    low = df['low']
    volume = df['volume']

    def calculate_rsi(series, window=14):
        delta = series.diff()
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window).mean()
        roll_down = down.abs().rolling(window).mean()
        rs = roll_up / roll_down
        rsi = 100.0 - (100.0 / (1.0 + rs))
        return rsi

    rsi = calculate_rsi(close)
    daily_range_change = (high - low).pct_change()
    avg_volume = volume.rolling(window=30).mean()
    volume_signal = (volume - avg_volume) / avg_volume

    heuristics_matrix = rsi + daily_range_change + volume_signal
    return heuristics_matrix
