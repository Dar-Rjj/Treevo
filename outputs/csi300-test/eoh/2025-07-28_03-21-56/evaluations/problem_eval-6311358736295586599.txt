def heuristics_v2(df):
    df['open_close_diff'] = df['open'] - df['close']
    df['ema_open_close_diff'] = df['open_close_diff'].ewm(span=7, adjust=False).mean()
    df['volume_rolling_std'] = df['volume'].rolling(window=7).std()
    heuristics_matrix = df['ema_open_close_diff'] / df['volume_rolling_std']
    return heuristics_matrix
