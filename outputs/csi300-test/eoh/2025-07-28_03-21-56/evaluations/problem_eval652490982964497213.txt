def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling standard deviation of the daily range
    rolling_std_range = daily_range.rolling(window=20).std()
    
    # Compute the exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Generate the heuristic factor by multiplying the rolling std with EMA volume
    heuristics_matrix = rolling_std_range * ema_volume
    
    return heuristics_matrix
