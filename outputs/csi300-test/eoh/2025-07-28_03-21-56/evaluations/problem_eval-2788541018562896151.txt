def heuristics_v2(df):
    def simple_moving_average(series, window=10):
        return series.rolling(window=window).mean()

    df['momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['volume_change'] = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    df['volatility'] = df['close'].pct_change().rolling(window=10).std()
    heuristics_matrix = 0.5 * df['momentum'] + 0.3 * df['volume_change'] - 0.2 * df['volatility']
    return heuristics_matrix
