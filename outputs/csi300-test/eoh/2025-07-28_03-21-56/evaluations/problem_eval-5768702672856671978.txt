def heuristics_v2(df):
    def exponential_moving_average(series, span):
        return series.ewm(span=span, adjust=False).mean()

    df['ema_close'] = exponential_moving_average(df['close'], 5)
    df['ema_volume'] = exponential_moving_average(df['volume'], 20)
    heuristics_matrix = df['ema_close'] - df['ema_volume']
    return heuristics_matrix
