def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    ma = close.rolling(window=30).mean()
    rs = (close - ma) / ma
    vol_pressure = (volume - volume.rolling(window=30).mean()) / volume.rolling(window=30).std()
    heuristics_matrix = rs + vol_pressure
    return heuristics_matrix
