import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day average volume
    avg_vol_30 = df['volume'].rolling(window=30).mean()
    
    # Calculate the volume ratio
    vol_ratio = df['volume'] / avg_vol_30
    
    # Calculate the high-low price difference
    hl_diff = df['high'] - df['low']
    
    # Adjust high-low difference by volume ratio
    weighted_hl_diff = hl_diff * vol_ratio
    
    # Calculate the 10-day moving average of close prices
    ma_close_10 = df['close'].rolling(window=10).mean()
    
    # Combine the weighted high-low difference with the 10-day moving average of close prices
    heuristics_matrix = (weighted_hl_diff + ma_close_10).dropna()
    
    return heuristics_matrix
