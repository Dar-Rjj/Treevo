import numpy as np

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling median of the daily range
    rolling_median_range = daily_range.rolling(window=30).median()
    
    # Generate the heuristic factor by incorporating the log of volume
    heuristics_matrix = rolling_median_range * np.log(df['volume'])
    
    return heuristics_matrix
