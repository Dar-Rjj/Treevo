def heuristics_v2(df):
    # Calculate the 10-day exponential moving average (EMA) of the close price
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute the daily return
    daily_return = df['close'].pct_change(periods=1)
    
    # Generate the heuristic factor by combining EMA and daily return
    heuristics_matrix = (df['close'] - ema_close) * daily_return
    
    return heuristics_matrix
