import pandas as pd
import numpy as np
from scipy import stats

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling mean and standard deviation of the daily range
    rolling_mean_range = daily_range.rolling(window=10).mean()
    rolling_std_range = daily_range.rolling(window=10).std()
    
    # Compute the rolling z-score of the daily range
    rolling_zscore_range = (daily_range - rolling_mean_range) / rolling_std_range
    
    # Generate the heuristic factor by incorporating the square root of volume
    heuristics_matrix = rolling_zscore_range * np.sqrt(df['volume'])
    
    return heuristics_matrix
