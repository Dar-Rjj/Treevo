def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    ma_short = close.rolling(window=15).mean()
    ma_long = close.rolling(window=60).mean()
    std_dev = close.rolling(window=40).std()
    volume_momentum = (volume - volume.shift(1)) / volume.shift(1)
    heuristics_matrix = (ma_short - ma_long) / std_dev + volume_momentum
    return heuristics_matrix
