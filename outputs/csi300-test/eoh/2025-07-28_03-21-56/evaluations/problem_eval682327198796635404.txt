def heuristics_v2(df):
    df['high_rolling_max'] = df['high'].rolling(window=7).max()
    df['low_rolling_min'] = df['low'].rolling(window=7).min()
    df['volume_rolling_std'] = df['volume'].rolling(window=7).std()
    heuristics_matrix = (df['high_rolling_max'] - df['low_rolling_min']) / df['volume_rolling_std']
    return heuristics_matrix
