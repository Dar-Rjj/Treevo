def heuristics_v2(df):
    # Calculate the True Range
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - x['close'].shift(1)), abs(x['low'] - x['close'].shift(1))), axis=1)
    
    # Compute the 14-day Average True Range (ATR)
    atr_14d = tr.rolling(window=14).mean()
    
    # Calculate the 30-day maximum amount
    max_amount_30d = df['amount'].rolling(window=30).max()
    
    # Calculate the ratio of current amount to the 30-day maximum amount
    amount_ratio = df['amount'] / max_amount_30d
    
    # Generate the heuristic factor by combining ATR and amount ratio
    heuristics_matrix = atr_14d * amount_ratio
    
    return heuristics_matrix
