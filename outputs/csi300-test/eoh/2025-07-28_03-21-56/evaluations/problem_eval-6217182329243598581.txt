import pandas as pd

def heuristics_v2(df):
    short_period = 5
    long_period = 20
    short_returns = df['close'].pct_change(short_period).rolling(window=short_period).mean()
    long_returns = df['close'].pct_change(long_period).rolling(window=long_period).mean()
    heuristics_matrix = short_returns - long_returns
    return heuristics_matrix
