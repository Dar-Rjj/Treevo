import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day EMAs for close prices
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Compute the difference between 20-day EMA and 5-day EMA
    ema_diff = ema_20 - ema_5
    
    # Calculate 10-day EMA for close prices
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()

    # Compute the ratio of the day's close to its 10-day EMA
    close_to_ema_10_ratio = df['close'] / ema_10
    
    # Generate the heuristic factor
    heuristics_matrix = ema_diff * close_to_ema_10_ratio
    
    return heuristics_matrix
