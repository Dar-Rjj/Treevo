def heuristics_v2(df):
    close = df['close']
    ma_short = close.rolling(window=7).mean()
    ma_long = close.rolling(window=21).mean()
    std_dev = close.rolling(window=14).std()
    adjusted_momentum = (close / close.shift(5)) - 1
    heuristics_matrix = (ma_short - ma_long) / std_dev + adjusted_momentum
    return heuristics_matrix
