import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling standard deviation of the daily range
    rolling_std_range = daily_range.rolling(window=10).std()
    
    # Generate the heuristic factor by incorporating the log of volume
    heuristics_matrix = rolling_std_range * np.log(df['volume'])
    
    return heuristics_matrix
