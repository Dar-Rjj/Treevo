def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) of the close price
    roc_close = df['close'].pct_change(periods=10)
    
    # Compute the 30-day minimum volume
    min_vol_30d = df['volume'].rolling(window=30).min()
    
    # Calculate the ratio of current volume to the 30-day minimum volume
    vol_ratio = df['volume'] / min_vol_30d
    
    # Generate the heuristic factor by combining ROC and volume ratio
    heuristics_matrix = roc_close * vol_ratio
    
    return heuristics_matrix
