import pandas as pd

def heuristics_v2(df):
    def wma(series, window=10):
        weights = pd.Series(range(1, window + 1))
        return series.rolling(window=window).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    close_momentum = df['close'].pct_change(periods=5)
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    vol_strength = df['volume'] / avg_vol_20
    combined_signal = close_momentum * vol_strength
    heuristics_matrix = wma(combined_signal)
    return heuristics_matrix
