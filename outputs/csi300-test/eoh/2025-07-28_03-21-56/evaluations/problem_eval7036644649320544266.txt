def heuristics_v2(df):
    # Calculate the 10-day and 20-day rate of change (ROC) of the close price
    roc_close_10d = df['close'].pct_change(periods=10)
    roc_close_20d = df['close'].pct_change(periods=20)
    
    # Average the 10-day and 20-day ROCs
    avg_roc_close = (roc_close_10d + roc_close_20d) / 2
    
    # Compute the 30-day minimum volume
    min_vol_30d = df['volume'].rolling(window=30).min()
    
    # Calculate the logarithm of the difference between current volume and 30-day minimum volume
    log_vol_diff = (df['volume'] - min_vol_30d).apply(lambda x: max(0, x)).transform('log1p')
    
    # Generate the heuristic factor by combining average ROC and the logarithmic volume difference
    heuristics_matrix = avg_roc_close * log_vol_diff
    
    return heuristics_matrix
