import pandas as pd

def heuristics_v2(df):
    close = df['close']
    ewma_short = close.ewm(span=10, adjust=False).mean()
    ewma_long = close.ewm(span=50, adjust=False).mean()
    rsi = 100 - (100 / (1 + (close.diff(1).clip(lower=0).rolling(window=14).mean() / -close.diff(1).clip(upper=0).rolling(window=14).mean())))
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    heuristics_matrix = (ewma_short - ewma_long) * rsi + volume_change
    return heuristics_matrix
