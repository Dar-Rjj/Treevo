def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling skewness of the daily range
    rolling_skew_range = daily_range.rolling(window=20).skew()
    
    # Compute the simple moving average of the volume
    sma_volume = df['volume'].rolling(window=20).mean()
    
    # Generate the heuristic factor by dividing the rolling skew with SMA volume
    heuristics_matrix = rolling_skew_range / sma_volume
    
    return heuristics_matrix
