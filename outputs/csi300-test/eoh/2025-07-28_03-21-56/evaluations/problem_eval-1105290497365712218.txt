import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling standard deviation of the daily range
    rolling_std_range = daily_range.rolling(window=10).std()
    
    # Generate the heuristic factor by incorporating the square root of volume
    volume_component = np.sqrt(df['volume'])
    intermediate_factor = rolling_std_range * volume_component
    
    # Incorporate the rate of change (ROC) of the closing price
    roc_close = df['close'].pct_change(periods=5)
    heuristics_matrix = intermediate_factor + roc_close
    
    return heuristics_matrix
