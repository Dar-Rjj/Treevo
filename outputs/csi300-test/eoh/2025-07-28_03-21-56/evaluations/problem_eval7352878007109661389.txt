def heuristics_v2(df):
    def compute_ppo(series, short_window=12, long_window=26):
        ema_short = series.ewm(span=short_window, adjust=False).mean()
        ema_long = series.ewm(span=long_window, adjust=False).mean()
        ppo_line = (ema_short - ema_long) / ema_long * 100
        return ppo_line

    def compute_cmf(high, low, close, volume, period=20):
        money_flow_multiplier = ((close - low) - (high - close)) / (high - low)
        money_flow_volume = money_flow_multiplier * volume
        cmf = money_flow_volume.rolling(window=period).sum() / volume.rolling(window=period).sum()
        return cmf

    df['ppo'] = compute_ppo(df['close'])
    df['cmf'] = compute_cmf(df['high'], df['low'], df['close'], df['volume'])
    heuristics_matrix = df['ppo'] + df['cmf']
    return heuristics_matrix
