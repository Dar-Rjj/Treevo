def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the exponentially weighted moving average of the daily range
    ewma_range = daily_range.ewm(span=20, adjust=False).mean()
    
    # Generate the heuristic factor by incorporating the logarithm of the volume
    log_volume = df['volume'].apply(lambda x: max(1, x)).apply(np.log)
    heuristics_matrix = ewma_range * log_volume
    
    # Add a momentum component based on the percentage change in close prices
    momentum = df['close'].pct_change().fillna(0)
    heuristics_matrix += momentum

    return heuristics_matrix
