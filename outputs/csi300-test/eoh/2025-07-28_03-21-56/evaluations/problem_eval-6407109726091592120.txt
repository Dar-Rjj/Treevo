def heuristics_v2(df):
    close = df['close']
    volume_change = df['volume'].diff()
    max_10d = close.rolling(window=10).max()
    min_10d = close.rolling(window=10).min()
    relative_strength = (close - min_10d) / (max_10d - min_10d)
    heuristics_matrix = relative_strength * volume_change
    return heuristics_matrix
