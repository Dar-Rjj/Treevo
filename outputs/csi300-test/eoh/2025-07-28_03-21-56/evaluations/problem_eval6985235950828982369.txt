import pandas as pd

def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    roc_high_low = (high_low_range / high_low_range.shift(1) - 1).rolling(window=3).mean()
    adl = ((df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])) * df['volume']
    adl_change = adl.rolling(window=5).mean().diff()
    heuristics_matrix = roc_high_low + adl_change
    return heuristics_matrix
