def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['absolute_returns'] = df['daily_return'].abs()
    df['positive_sum_rolling'] = df['positive_returns'].rolling(window=7).sum()
    df['absolute_sum_rolling'] = df['absolute_returns'].rolling(window=7).sum()
    heuristics_matrix = df['positive_sum_rolling'] / df['absolute_sum_rolling']
    return heuristics_matrix
