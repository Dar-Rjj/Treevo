def heuristics_v2(df):
    # Calculate the 20-day EMA of the close price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the 30-day average trading amount
    avg_amount_30d = df['amount'].rolling(window=30).mean()
    
    # Calculate the ratio of current trading amount to the 30-day average trading amount
    amount_ratio = df['amount'] / avg_amount_30d
    
    # Generate the heuristic factor by combining EMA and amount ratio
    heuristics_matrix = ema_close * amount_ratio
    
    return heuristics_matrix
