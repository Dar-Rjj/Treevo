def heuristics_v2(df):
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    ma_close = df['close'].rolling(window=20).mean()
    directional_movement = df['close'] - ma_close
    heuristics_matrix = (volatility * 0.5) + (directional_movement * 0.5)
    return heuristics_matrix
