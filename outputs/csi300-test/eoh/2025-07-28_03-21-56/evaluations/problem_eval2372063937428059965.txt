def heuristics_v2(df):
    close = df['close']
    ma_short = close.rolling(window=20).mean()
    ma_long = close.rolling(window=100).mean()
    std_dev = close.rolling(window=60).std()
    momentum = (close / close.shift(5)) - 1
    heuristics_matrix = (ma_short - ma_long) / std_dev + momentum
    return heuristics_matrix
