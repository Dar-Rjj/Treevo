def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the EMA of the daily range
    ema_range = daily_range.ewm(span=30).mean()
    
    # Generate the heuristic factor by incorporating the natural logarithm of the volume
    heuristics_matrix = ema_range * np.log(df['volume'])
    
    return heuristics_matrix
