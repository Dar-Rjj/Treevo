def heuristics_v2(df):
    close_sum_5 = df['close'].rolling(window=5).sum()
    high_sum_20 = df['high'].rolling(window=20).sum()
    volume_change = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    heuristics_matrix = (close_sum_5 / high_sum_20) + volume_change
    return heuristics_matrix
