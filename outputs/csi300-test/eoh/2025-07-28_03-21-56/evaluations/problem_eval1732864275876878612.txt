import pandas as pd

def heuristics_v2(df):
    def compute_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    def compute_mfi(high, low, close, volume, window=14):
        typical_price = (high + low + close) / 3
        raw_money_flow = typical_price * volume
        positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
        negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
        money_flow_ratio = positive_flow.rolling(window=window).sum() / negative_flow.rolling(window=window).sum()
        mfi = 100 - (100 / (1 + money_flow_ratio))
        return mfi

    df['rsi'] = compute_rsi(df['close'])
    df['mfi'] = compute_mfi(df['high'], df['low'], df['close'], df['volume'])
    heuristics_matrix = df['rsi'] + df['mfi']
    return heuristics_matrix
