def heuristics_v2(df):
    close = df['close']
    ema_short = close.ewm(span=10, adjust=False).mean()
    ema_long = close.ewm(span=50, adjust=False).mean()
    aroon_up = 100 * (df['high'].rolling(window=25).apply(lambda x: (x.argmax() + 1) / 25))
    aroon_down = 100 * (df['low'].rolling(window=25).apply(lambda x: (x.argmin() + 1) / 25))
    aroon_oscillator = aroon_up - aroon_down
    momentum = close / close.shift(1) - 1
    heuristics_matrix = (ema_short - ema_long) / aroon_oscillator + momentum
    return heuristics_matrix
