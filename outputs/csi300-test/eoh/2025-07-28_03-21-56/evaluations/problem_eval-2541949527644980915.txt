def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) of the close price
    roc_close = df['close'].pct_change(periods=10)
    
    # Compute the 30-day average volume
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    
    # Calculate the logarithmic difference between the current volume and the 30-day average volume
    log_diff_vol = (df['volume'] - avg_vol_30d).apply(lambda x: np.log(1 + abs(x)) if x != 0 else 0)
    
    # Generate the heuristic factor by combining ROC and log-difference of volume
    heuristics_matrix = roc_close * log_diff_vol
    
    return heuristics_matrix
