def heuristics_v2(df):
    def compute_ppo(series, fast_period=12, slow_period=26):
        ema_fast = series.ewm(span=fast_period, adjust=False).mean()
        ema_slow = series.ewm(span=slow_period, adjust=False).mean()
        ppo = 100 * (ema_fast - ema_slow) / ema_slow
        return ppo

    def compute_returns_volatility(series, period=20):
        returns = series.pct_change()
        volatility = returns.rolling(window=period).std()
        return volatility

    df['ppo'] = compute_ppo(df['close'])
    df['volatility'] = compute_returns_volatility(df['close'])
    heuristics_matrix = df['ppo'] + df['volatility']
    return heuristics_matrix
