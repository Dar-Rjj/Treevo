def heuristics_v2(df):
    def exponential_moving_average(series, span):
        return series.ewm(span=span, adjust=False).mean()
    
    df['ema_close'] = exponential_moving_average(df['close'], 10)
    df['volume_rolling_std'] = df['volume'].rolling(window=20).std()
    heuristics_matrix = df['ema_close'] - df['volume_rolling_std']
    return heuristics_matrix
