import pandas as pd

def heuristics_v2(df):
    price_change = df['close'] - df['close'].shift(1)
    ema_price_change = price_change.ewm(span=20, adjust=False).mean()
    volume_ratio = df['volume'] / df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = ema_price_change * volume_ratio
    return heuristics_matrix
