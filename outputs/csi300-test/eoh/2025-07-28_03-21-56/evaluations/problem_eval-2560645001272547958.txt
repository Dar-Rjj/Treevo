def heuristics_v2(df):
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volume_change = df['volume'].pct_change().rolling(window=5).mean()
    close_high_ratio = df['close'] / df['high']
    weighted_sum = (log_returns * 0.6) + (volume_change * 0.2) + (close_high_ratio * 0.2)
    heuristics_matrix = (weighted_sum - weighted_sum.rolling(window=20).mean()) / weighted_sum.rolling(window=20).std()
    return heuristics_matrix
