def heuristics_v2(df):
    def compute_roc(series, period=12):
        return ((series / series.shift(period)) - 1) * 100

    roc_close = compute_roc(df['close'])
    rolling_std_volume = df['volume'].rolling(window=30).std()
    heuristics_matrix = roc_close + rolling_std_volume
    return heuristics_matrix
