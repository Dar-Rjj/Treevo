def heuristics_v2(df):
    momentum = df['close'].pct_change(periods=10)
    volatility = df['close'].pct_change().rolling(window=20).std()
    trend_following = (df['close'] - df['close'].rolling(window=50).mean()) / df['close'].rolling(window=50).std()
    heuristics_matrix = (momentum + 1/(1+volatility) + trend_following) / 3
    return heuristics_matrix
