def heuristics_v2(df):
    df['high_rolling_max'] = df['high'].rolling(window=14).max()
    df['low_rolling_min'] = df['low'].rolling(window=14).min()
    price_ratio = df['high_rolling_max'] / df['low_rolling_min']
    close_change = df['close'].pct_change()
    heuristics_matrix = (price_ratio + close_change) / 2
    return heuristics_matrix
