import pandas as pd

def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    roc_high_low = high_low_range / high_low_range.shift(1) - 1
    mfi = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    mfi_change = mfi.diff()
    heuristics_matrix = roc_high_low + mfi_change
    return heuristics_matrix
