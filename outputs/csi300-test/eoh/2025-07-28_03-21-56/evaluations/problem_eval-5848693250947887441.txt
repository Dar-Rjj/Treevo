import pandas as pd

def heuristics_v2(df):
    high_low_range = df['high'] - df['low']
    roc_high_low = high_low_range / high_low_range.shift(1) - 1
    ma_roc_high_low = roc_high_low.rolling(window=5).mean()
    adl = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low']) * df['volume']
    ma_adl = adl.rolling(window=10).sum()
    heuristics_matrix = ma_roc_high_low + ma_adl
    return heuristics_matrix
