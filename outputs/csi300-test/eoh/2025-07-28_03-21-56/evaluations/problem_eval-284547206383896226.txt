def heuristics_v2(df):
    # Calculate the 20-day percentage change in closing prices
    pct_change_close = df['close'].pct_change(periods=20)
    
    # Calculate the 50-day moving average of the volume
    volume_50_day_ma = df['volume'].rolling(window=50).mean()
    
    # Compute the ratio of the current volume to the 50-day moving average of the volume
    volume_ratio = df['volume'] / volume_50_day_ma
    
    # Generate the heuristic factor
    heuristics_matrix = pct_change_close * volume_ratio
    
    return heuristics_matrix
