import pandas as pd

def heuristics_v2(df):
    close = df['close']
    
    # Calculate RSI
    delta = close.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Bollinger Bands
    ma = close.rolling(window=20).mean()
    std_dev = close.rolling(window=20).std()
    upper_band = ma + 2 * std_dev
    lower_band = ma - 2 * std_dev
    
    # Heuristic adjustment based on price momentum
    momentum = close / close.shift(1) - 1
    heuristics_matrix = (rsi - 50) / (upper_band - lower_band) + momentum
    
    return heuristics_matrix
