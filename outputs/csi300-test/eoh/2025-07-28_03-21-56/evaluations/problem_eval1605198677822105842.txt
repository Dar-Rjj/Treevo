def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the rolling mean of the daily range
    rolling_mean_range = daily_range.rolling(window=30).mean()
    
    # Generate the heuristic factor by incorporating the natural logarithm of the volume
    heuristics_matrix = rolling_mean_range * np.log(df['volume'])
    
    return heuristics_matrix
