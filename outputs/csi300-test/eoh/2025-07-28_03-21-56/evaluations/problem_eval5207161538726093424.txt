def heuristics_v2(df):
    df['high_rolling_max'] = df['high'].rolling(window=14).max()
    df['low_rolling_min'] = df['low'].rolling(window=14).min()
    df['volume_rolling_sum'] = df['volume'].rolling(window=14).sum()
    heuristics_matrix = (df['high_rolling_max'] - df['low_rolling_min']) / df['volume_rolling_sum']
    return heuristics_matrix
