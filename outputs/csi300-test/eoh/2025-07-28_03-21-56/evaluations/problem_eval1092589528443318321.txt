def heuristics_v2(df):
    # Calculate the 20-day exponential moving average (EMA) of the close price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the 20-day average trading range
    avg_range_20d = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Calculate the ratio of the current day's trading range to the 20-day average trading range
    range_ratio = (df['high'] - df['low']) / avg_range_20d
    
    # Generate the heuristic factor by combining EMA and range ratio
    heuristics_matrix = ema_close * range_ratio
    
    return heuristics_matrix
