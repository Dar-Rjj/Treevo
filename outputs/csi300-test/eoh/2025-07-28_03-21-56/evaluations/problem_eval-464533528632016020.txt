import pandas as pd

def heuristics_v2(df):
    # Calculate daily price change
    df['price_change'] = df['close'].diff()
    
    # Compute the exponential moving average of daily price change
    ema_price_change = df['price_change'].ewm(span=7, adjust=False).mean()
    
    # Calculate the 7-day rolling mean of volume
    df['volume_rolling_mean'] = df['volume'].rolling(window=7).mean()
    
    # Avoid division by zero
    df['volume_rolling_mean'] = df['volume_rolling_mean'].replace(0, 1)
    
    # Generate the heuristic factor
    heuristics_matrix = ema_price_change * (df['volume'] / df['volume_rolling_mean'])
    
    return heuristics_matrix
