import pandas as pd

def heuristics_v2(df):
    # Momentum indicator: 10-day return
    momentum = df['close'].pct_change(10)
    
    # Volatility indicator: 30-day standard deviation of daily returns
    volatility = df['close'].pct_change().rolling(window=30).std()
    
    # Simple heuristic combination (for demonstration, weights are chosen arbitrarily)
    factor = (momentum * 0.6) - (volatility * 0.4)
    
    # Ensure the output is a Series with the appropriate name
    heuristics_matrix = factor.rename('heuristics_factor')
    
    return heuristics_matrix
