def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Compute the rolling mean of the daily return
    rolling_mean_return = daily_return.rolling(window=30).mean()
    
    # Generate the heuristic factor by incorporating the logarithm of the volume
    heuristics_matrix = rolling_mean_return * np.log(df['volume'])
    
    return heuristics_matrix
