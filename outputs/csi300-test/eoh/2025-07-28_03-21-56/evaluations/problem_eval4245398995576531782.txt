def heuristics_v2(df):
    # Calculate the close-to-open price ratio
    close_open_ratio = df['close'] / df['open']
    
    # Compute the rolling mean of the close-to-open price ratio
    rolling_mean_close_open = close_open_ratio.rolling(window=20).mean()
    
    # Calculate the rate of change in volume
    volume_roc = df['volume'].pct_change(periods=1)
    
    # Generate the heuristic factor
    heuristics_matrix = rolling_mean_close_open * volume_roc
    
    return heuristics_matrix
