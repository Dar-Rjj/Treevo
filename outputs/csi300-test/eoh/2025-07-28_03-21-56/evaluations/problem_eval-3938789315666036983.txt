import pandas as pd

def heuristics_v2(df):
    close_prices = df['close']
    # Calculate 12-month percentage change
    pct_change_12m = close_prices.pct_change(periods=240)
    # Calculate daily returns
    daily_returns = close_prices.pct_change()
    # Calculate 12-month rolling standard deviation (volatility)
    vol_12m = daily_returns.rolling(window=240).std()
    # Adjust momentum signal with volatility
    heuristics_matrix = pct_change_12m / vol_12m
    return heuristics_matrix
