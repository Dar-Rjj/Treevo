def heuristics_v2(df):
    def macd_heuristic(series, short_window=12, long_window=26, signal_smoothing=9):
        short_ema = series.ewm(span=short_window, adjust=False).mean()
        long_ema = series.ewm(span=long_window, adjust=False).mean()
        macd_line = short_ema - long_ema
        signal_line = macd_line.ewm(span=signal_smoothing, adjust=False).mean()
        return macd_line - signal_line
    
    close_macd = macd_heuristic(df['close'])
    volume_macd = macd_heuristic(df['volume'])
    
    heuristics_matrix = close_macd + 2 * volume_macd # Giving more weight to volume changes
    return heuristics_matrix
