def heuristics_v2(df):
    # Calculate the EMA of the daily price change
    daily_change = df['close'].diff()
    ema_daily_change = daily_change.ewm(span=5, adjust=False).mean()
    
    # Compute the 10-day and 30-day EMAs of the close prices
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the momentum indicator as the difference between 10-day and 30-day EMAs
    momentum_indicator = ema_10 - ema_30
    
    # Combine the two components into a single heuristic matrix
    heuristics_matrix = ema_daily_change + momentum_indicator
    
    return heuristics_matrix
