def heuristics_v2(df):
    # Calculate the 20-day rate of change (ROC) of the close price
    roc_close = df['close'].pct_change(periods=20)
    
    # Compute the 60-day minimum volume
    min_vol_60d = df['volume'].rolling(window=60).min()
    
    # Calculate the ratio of current volume to the 60-day minimum volume
    vol_ratio = df['volume'] / min_vol_60d
    
    # Generate the heuristic factor by combining ROC and volume ratio
    heuristics_matrix = roc_close * vol_ratio
    
    return heuristics_matrix
