def heuristics_v2(df):
    close = df['close']
    ma_short = close.rolling(window=10).mean()
    ma_long = close.rolling(window=50).mean()
    volume = df['volume']
    avg_volume = volume.rolling(window=30).mean()
    price_change_velocity = (close - close.shift(30)) / 30
    heuristics_matrix = (ma_short - ma_long) + (volume / avg_volume) + price_change_velocity
    return heuristics_matrix
