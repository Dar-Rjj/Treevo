def heuristics_v2(df):
    # Calculate the EMA of daily close prices
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute daily price change
    price_change = df['close'] - df['close'].shift(1)
    
    # Calculate the EMA of daily price changes
    ema_price_change = price_change.ewm(span=20, adjust=False).mean()
    
    # Determine the rolling max and min of the volume over the last 20 days
    rolling_max_volume = df['volume'].rolling(window=20).max()
    rolling_min_volume = df['volume'].rolling(window=20).min()
    
    # Generate a signal based on the difference between the highest and lowest volume
    volume_signal = rolling_max_volume - rolling_min_volume
    
    # Combine the EMA of price changes with the volume signal
    heuristics_matrix = ema_price_change * volume_signal
    
    return heuristics_matrix
