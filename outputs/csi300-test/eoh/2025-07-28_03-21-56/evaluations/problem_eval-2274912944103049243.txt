def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the 20-day standard deviation of the closing price
    std_dev = df['close'].rolling(window=20).std()
    
    # Calculate the closing price to 20-day standard deviation ratio
    close_to_std_ratio = df['close'] / std_dev
    
    # Generate the heuristic factor
    heuristics_matrix = daily_return + close_to_std_ratio
    
    return heuristics_matrix
