import pandas as pd

def heuristics_v2(df):
    short_term_ma = df['close'].rolling(window=10).mean()
    long_term_ma = df['close'].rolling(window=60).mean()
    price_momentum = df['close'] / df['close'].shift(5) - 1
    heuristics_matrix = (short_term_ma - long_term_ma) * price_momentum
    return heuristics_matrix
