def heuristics_v2(df):
    # Calculate the 10-day and 20-day rate of change (ROC) of the close price
    roc_close_10d = df['close'].pct_change(periods=10)
    roc_close_20d = df['close'].pct_change(periods=20)
    
    # Average the 10-day and 20-day ROC
    avg_roc_close = (roc_close_10d + roc_close_20d) / 2
    
    # Compute the 30-day average volume
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of current volume to the 30-day average volume
    vol_ratio = df['volume'] / avg_vol_30d
    
    # Generate the heuristic factor by combining the average ROC and volume ratio
    heuristics_matrix = avg_roc_close * vol_ratio
    
    return heuristics_matrix
