def heuristics_v2(df):
    # Calculate daily close price changes
    delta_close = df['close'].diff()
    
    # Separate positive and negative changes
    pos_changes = delta_close.where(delta_close > 0, 0)
    neg_changes = delta_close.where(delta_close < 0, 0).abs()
    
    # Compute rolling sum of positive and negative changes over a 14-day window
    rolling_sum_pos = pos_changes.rolling(window=14).sum()
    rolling_sum_neg = neg_changes.rolling(window=14).sum()
    
    # Generate sentiment score by dividing positive sum by absolute value of negative sum
    sentiment_score = rolling_sum_pos / rolling_sum_neg
    
    # Calculate 5-day rate of change in volume
    roc_volume = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Final heuristic factor as the product of sentiment score and ROC volume
    heuristics_matrix = sentiment_score * roc_volume
    
    return heuristics_matrix
