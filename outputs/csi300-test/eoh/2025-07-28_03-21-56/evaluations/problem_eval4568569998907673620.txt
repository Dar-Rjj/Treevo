def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['volume_rolling_std'] = df['volume'].rolling(window=7).std()
    df['return_rolling_sum'] = df['daily_return'].rolling(window=7).sum()
    heuristics_matrix = df['return_rolling_sum'] / df['volume_rolling_std']
    return heuristics_matrix
