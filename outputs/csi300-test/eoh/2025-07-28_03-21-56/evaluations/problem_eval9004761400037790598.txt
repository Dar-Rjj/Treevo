def heuristics_v2(df):
    # Calculate the 20-day exponential moving average (EMA) of the close price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the 5-day standard deviation of the close price
    std_close_5d = df['close'].rolling(window=5).std()
    
    # Generate the heuristic factor by combining EMA and standard deviation
    heuristics_matrix = ema_close * std_close_5d
    
    return heuristics_matrix
