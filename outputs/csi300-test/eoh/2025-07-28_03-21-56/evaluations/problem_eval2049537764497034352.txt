def heuristics_v2(df):
    # Calculate the 10-day EMA of the close price
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 20-day EMA of the close price
    ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    
    # Compute the 50-day SMA of the close price
    sma_50d = df['close'].rolling(window=50).mean()
    
    # Generate the heuristic factor by dividing the difference of EMAs by the SMA
    heuristics_matrix = (ema_10d - ema_20d) / sma_50d
    
    return heuristics_matrix
