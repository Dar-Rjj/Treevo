def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the exponential moving average of the daily range
    ema_range = daily_range.ewm(span=20, adjust=False).mean()
    
    # Generate the heuristic factor by incorporating the log of volume
    heuristics_matrix = ema_range * np.log1p(df['volume'])
    
    return heuristics_matrix
