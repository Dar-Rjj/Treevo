def heuristics_v2(df):
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volume_change = df['volume'].pct_change()
    close_open_ratio = df['close'] / df['open']
    weighted_sum = (log_returns * 0.5) + (volume_change * 0.3) + (close_open_ratio * 0.2)
    heuristics_matrix = (weighted_sum - weighted_sum.rolling(window=20).mean()) / weighted_sum.rolling(window=20).std()
    return heuristics_matrix
