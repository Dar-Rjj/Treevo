def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Compute the 20-day moving average of the daily return
    ma_20d_return = daily_return.rolling(window=20).mean()
    
    # Compute the 10-day exponential moving average of the volume
    ema_10d_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Generate the heuristic factor by combining the 20-day MA of the daily return and the 10-day EMA of the volume
    heuristics_matrix = ma_20d_return * ema_10d_volume
    
    return heuristics_matrix
