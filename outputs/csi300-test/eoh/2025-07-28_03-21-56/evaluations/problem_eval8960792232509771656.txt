def heuristics_v2(df):
    log_returns = np.log(df['close'] / df['close'].shift(1))
    high_low_diff = df['high'] - df['low']
    close_open_ratio = df['close'] / df['open']
    weighted_sum = (log_returns * 0.4) + (high_low_diff * 0.3) + (close_open_ratio * 0.3)
    short_ma = weighted_sum.rolling(window=5).mean()
    long_ma = weighted_sum.rolling(window=20).mean()
    heuristics_matrix = short_ma - long_ma
    return heuristics_matrix
