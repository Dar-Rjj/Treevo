def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the ratio of the difference between close and open prices to the daily range
    close_open_ratio = (df['close'] - df['open']) / daily_range
    
    # Compute the 20-day simple moving average of the volume
    sma_volume = df['volume'].rolling(window=20).mean()
    
    # Take the natural logarithm of the SMA volume
    log_sma_volume = np.log(sma_volume)
    
    # Generate the heuristic factor by multiplying the close-open ratio with log of SMA volume
    heuristics_matrix = close_open_ratio * log_sma_volume
    
    return heuristics_matrix
