def heuristics_v2(df):
    def compute_ppo(series, short_window=9, long_window=21, signal_window=10):
        ema_short = series.ewm(span=short_window, adjust=False).mean()
        ema_long = series.ewm(span=long_window, adjust=False).mean()
        ppo_line = (ema_short - ema_long) / ema_long * 100
        return ppo_line

    def compute_adl(high, low, close, volume):
        money_flow_multiplier = ((close - low) - (high - close)) / (high - low)
        money_flow_volume = money_flow_multiplier * volume
        adl = money_flow_volume.cumsum()
        return adl

    df['ppo'] = compute_ppo(df['close'])
    df['adl'] = compute_adl(df['high'], df['low'], df['close'], df['volume'])
    heuristics_matrix = df['ppo'] * 0.6 + df['adl'] * 0.4
    return heuristics_matrix
