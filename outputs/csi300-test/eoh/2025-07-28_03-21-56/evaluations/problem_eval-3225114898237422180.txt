import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Compute the rolling standard deviation of the daily return
    rolling_std_return = daily_return.rolling(window=10).std()
    
    # Generate the heuristic factor by incorporating the square root of volume
    heuristics_matrix = rolling_std_return * np.sqrt(df['volume'])
    
    return heuristics_matrix
