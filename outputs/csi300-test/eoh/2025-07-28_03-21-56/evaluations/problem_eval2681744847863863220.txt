import numpy as np

def heuristics_v2(df):
    # Calculate the 10-day rolling mean of the close prices
    close_rolling_mean = df['close'].rolling(window=10).mean()
    
    # Calculate the absolute difference between each day's close and the 10-day rolling mean of the close
    abs_diff_close = (df['close'] - close_rolling_mean).abs()
    
    # Calculate the 10-day rolling mean of the amount
    amount_rolling_mean = df['amount'].rolling(window=10).mean()
    
    # Generate the heuristic factor
    heuristics_matrix = abs_diff_close * np.log(amount_rolling_mean)
    
    return heuristics_matrix
