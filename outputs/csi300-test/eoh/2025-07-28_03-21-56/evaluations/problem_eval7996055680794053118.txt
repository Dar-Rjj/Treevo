def heuristics_v2(df):
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volume_change = df['volume'].pct_change()
    hl_change = (df['high'] - df['low']) / df['low']
    weighted_sum = (hl_change * 0.4) + (log_returns * 0.3) + (volume_change * 0.3)
    heuristics_matrix = weighted_sum.ewm(span=20).mean()
    return heuristics_matrix
