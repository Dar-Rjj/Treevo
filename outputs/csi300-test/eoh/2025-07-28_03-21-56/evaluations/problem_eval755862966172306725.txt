def heuristics_v2(df):
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Compute the EMA of daily returns over a 10-day window
    ema_daily_return = daily_return.ewm(span=10).mean()
    
    # Compute the EMA of volume over a 10-day window
    ema_volume = df['volume'].ewm(span=10).mean()
    
    # Generate the heuristic factor by combining the EMAs
    heuristics_matrix = ema_daily_return * ema_volume
    
    return heuristics_matrix
