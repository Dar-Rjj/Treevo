def heuristics_v2(df):
    # Calculate the 10-day average return
    avg_return_10d = df['close'].pct_change().rolling(window=10).mean()
    
    # Compute the 30-day mean volume
    mean_vol_30d = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of current volume to the 30-day mean volume
    vol_ratio = df['volume'] / mean_vol_30d
    
    # Generate the heuristic factor by combining average return and volume ratio
    heuristics_matrix = avg_return_10d * vol_ratio
    
    return heuristics_matrix
