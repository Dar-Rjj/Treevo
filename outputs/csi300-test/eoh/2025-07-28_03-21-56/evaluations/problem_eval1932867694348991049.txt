def heuristics_v2(df):
    def compute_ppo(series, short_window=12, long_window=26, signal_window=9):
        ema_short = series.ewm(span=short_window, adjust=False).mean()
        ema_long = series.ewm(span=long_window, adjust=False).mean()
        ppo_line = (ema_short - ema_long) / ema_long * 100
        return ppo_line

    def compute_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    df['ppo'] = compute_ppo(df['close'])
    df['rsi'] = compute_rsi(df['close'])
    heuristics_matrix = df['ppo'] + df['rsi']
    return heuristics_matrix
