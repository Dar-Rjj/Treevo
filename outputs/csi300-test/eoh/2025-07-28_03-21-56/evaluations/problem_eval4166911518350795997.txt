def heuristics_v2(df):
    # Calculate the 20-day rolling minimum low price
    min_low_20d = df['low'].rolling(window=20).min()
    
    # Compute the price-to-low ratio
    price_to_low_ratio = df['close'] / min_low_20d
    
    # Calculate the 10-day and 30-day moving averages of the volume
    ma_volume_10d = df['volume'].rolling(window=10).mean()
    ma_volume_30d = df['volume'].rolling(window=30).mean()
    
    # Compute the average of the 10-day and 30-day moving averages of the volume
    avg_ma_volume = (ma_volume_10d + ma_volume_30d) / 2
    
    # Generate the heuristic factor by combining the price-to-low ratio and the average of the moving averages of the volume
    heuristics_matrix = price_to_low_ratio * avg_ma_volume
    
    return heuristics_matrix
