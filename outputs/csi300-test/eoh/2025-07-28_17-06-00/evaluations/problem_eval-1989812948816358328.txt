def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute exponentially weighted moving average
    span = 60
    df['ewma'] = df['pct_change_close'].ewm(span=span, adjust=False).mean()
    
    # Add momentum indicator: difference of rolling mean of high and low
    df['momentum'] = df['high'].rolling(window=span).mean() - df['low'].rolling(window=span).mean()
    
    # Scale the result by the 20-day standard deviation of close prices
    scale_factor = df['close'].rolling(window=20).std()
    df['scaled_momentum'] = df['momentum'] / scale_factor
    
    # Heuristic formula: Exponentially Weighted Moving Average + Scaled Momentum
    heuristics_matrix = df['ewma'] + df['scaled_momentum']
    
    return heuristics_matrix
