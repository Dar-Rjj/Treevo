import pandas as pd

def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute exponentially weighted moving average of the percentage change
    window_size = 30
    df['ewma'] = df['pct_change_close'].ewm(span=window_size, adjust=False).mean()
    
    # Add momentum indicator: Stochastic Oscillator
    k_period = 14
    d_period = 3
    df['k'] = ((df['close'] - df['low'].rolling(window=k_period).min()) / 
               (df['high'].rolling(window=k_period).max() - df['low'].rolling(window=k_period).min()))
    df['d'] = df['k'].rolling(window=d_period).mean()
    
    # Heuristic formula: Exponentially Weighted Moving Average + %K of Stochastic Oscillator
    heuristics_matrix = df['ewma'] + df['k']
    
    return heuristics_matrix
