def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    
    # Calculate the RSI for the closing prices over a 14-day period
    delta = close.diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    
    r_up = up.ewm(alpha=1/14, adjust=False).mean()
    r_down = down.abs().ewm(alpha=1/14, adjust=False).mean()
    
    rs = r_up / r_down
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the ratio of the sum of the last 10 days' trading volume to the maximum daily volume within that period
    vol_ratio = volume.rolling(window=10).sum() / volume.rolling(window=10).max()
    
    # Heuristic formula: RSI + Volume Ratio
    heuristics_matrix = rsi + vol_ratio
    
    return heuristics_matrix
