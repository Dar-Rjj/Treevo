def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['sma_amount'] = df['amount'].rolling(window=window_size).mean()
    adjustment_factor = (df['amount'] - df['sma_amount']) / df['amount']
    heuristics_matrix = df['ema_volume'] * (df['close'] / df['sma_close']) * (1 + adjustment_factor)
    return heuristics_matrix
