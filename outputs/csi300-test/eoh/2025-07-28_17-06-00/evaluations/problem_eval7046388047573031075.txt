def heuristics_v2(df):
    window_size = 50
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['std_close'] = df['close'].rolling(window=window_size).std()
    volatility_factor = df['std_close'] / df['sma_close']
    trend_deviation = (df['close'] - df['sma_close']) / df['close']
    heuristics_matrix = volatility_factor * (1 + trend_deviation)
    return heuristics_matrix
