def heuristics_v2(df):
    window_size = 20
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['rsi_close'] = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=window_size).mean() / -df['close'].diff(1).clip(upper=0).rolling(window=window_size).mean())))
    heuristics_matrix = df['ema_volume'] * df['rsi_close']
    return heuristics_matrix
