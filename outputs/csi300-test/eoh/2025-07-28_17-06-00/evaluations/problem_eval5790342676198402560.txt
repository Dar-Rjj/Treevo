def heuristics_v2(df):
    # Calculate daily price change
    price_change = df['close'].diff(1).fillna(0)
    
    # Calculate the difference between high and low as a measure of intraday volatility, with a 20-day rolling mean
    intraday_volatility = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Compute the cumulative difference between up and down days in volume
    volume_diff = (df['volume'] * (price_change > 0) - df['volume'] * (price_change < 0)).cumsum()
    
    # Apply an exponential moving average to the intraday volatility for emphasis on recent data
    ema_intraday_volatility = intraday_volatility.ewm(span=20, adjust=False).mean()
    
    # Combine the EMA-adjusted intraday volatility with the volume difference
    heuristics_matrix = ema_intraday_volatility * volume_diff
    
    return heuristics_matrix
