def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    high = df['high']
    low = df['low']
    typical_price = (high + low + close) / 3
    
    # Bollinger Bandwidth as a measure of volatility
    ma_typical_price = typical_price.rolling(window=20).mean()
    std_typical_price = typical_price.rolling(window=20).std()
    bollinger_band_width = (ma_typical_price + 2 * std_typical_price - (ma_typical_price - 2 * std_typical_price)) / ma_typical_price
    
    # Money Flow Index (MFI) for assessing buying or selling pressure
    money_flow = typical_price * volume
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(), 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(), 0).rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Simple Moving Average Crossover Strategy to identify trend
    sma_50 = close.rolling(window=50).mean()
    sma_200 = close.rolling(window=200).mean()
    sma_crossover_signal = sma_50 - sma_200

    # Alpha factor combining Bollinger Bandwidth, MFI, and SMA Crossover Signal
    heuristics_matrix = bollinger_band_width * mfi * (sma_crossover_signal > 0)
    
    return heuristics_matrix
