def heuristics_v2(df):
    window_size = 30
    df['ema_open'] = df['open'].ewm(span=window_size, adjust=False).mean()
    df['sma_amount'] = df['amount'].rolling(window=window_size).mean()
    adjustment_factor = (df['amount'] - df['sma_amount']) / df['amount']
    heuristics_matrix = (df['open'] / df['ema_open']) * (1 + adjustment_factor)
    return heuristics_matrix
