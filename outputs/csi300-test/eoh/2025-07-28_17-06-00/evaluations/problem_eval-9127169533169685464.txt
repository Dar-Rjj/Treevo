def heuristics_v2(df):
    window_size = 20
    # Calculate the 20-day rate of change (ROC) for the closing price
    roc_close = df['close'].pct_change(periods=window_size)
    # Calculate the standard deviation of daily returns over the lookback window
    std_daily_return = df['close'].pct_change().rolling(window=window_size).std()
    # Inverse of daily return's standard deviation as weights
    weight_volatility = 1 / std_daily_return
    # Weighted moving average of the trading volume
    wma_volume = (df['volume'] * weight_volatility).rolling(window=window_size).mean() / weight_volatility.rolling(window=window_size).mean()
    # Alpha factor as a combination of ROC close and WMA volume
    heuristics_matrix = roc_close + wma_volume
    return heuristics_matrix
