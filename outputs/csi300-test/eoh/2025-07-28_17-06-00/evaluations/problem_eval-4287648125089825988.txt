import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of the close prices
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the average volume over a 20-day window
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Compute the ratio of the current day's volume to the average volume
    volume_ratio = df['volume'] / avg_volume
    
    # Combine the EMA of close prices with the volume ratio
    heuristics_matrix = ema_close * volume_ratio
    
    return heuristics_matrix
