def heuristics_v2(df):
    window_size = 20
    df['ema_high'] = df['high'].ewm(span=window_size, adjust=False).mean()
    df['ema_low'] = df['low'].ewm(span=window_size, adjust=False).mean()
    df['roc_volume'] = df['volume'].pct_change()
    heuristics_matrix = (df['ema_high'] - df['ema_low']) * df['roc_volume']
    return heuristics_matrix
