def heuristics_v2(df):
    window_size = 30
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean()
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    adjustment_factor = (df['volume'] - df['sma_volume']) / df['volume']
    heuristics_matrix = df['ema_amount'] * (df['close'] / df['ema_close']) * (1 + adjustment_factor)
    return heuristics_matrix
