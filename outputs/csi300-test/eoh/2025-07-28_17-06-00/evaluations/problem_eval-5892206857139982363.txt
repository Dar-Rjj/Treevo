def heuristics_v2(df):
    # Calculate the True Range
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    # Calculate the Average True Range over 14 days
    df['ATR_14'] = df['true_range'].rolling(window=14).mean()
    # Determine the direction of the price movement
    df['direction'] = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    # Create the heuristics matrix as ATR multiplied by the direction
    heuristics_matrix = df['ATR_14'] * df['direction']
    return heuristics_matrix
