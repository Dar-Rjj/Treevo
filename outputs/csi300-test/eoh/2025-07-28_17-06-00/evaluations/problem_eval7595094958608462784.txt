def heuristics_v2(df):
    window_size = 20
    df['ema_high'] = df['high'].ewm(span=window_size, adjust=False).mean()
    df['ema_low'] = df['low'].ewm(span=window_size, adjust=False).mean()
    df['roc_close'] = df['close'].pct_change()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    heuristics_matrix = df['roc_close'] + (df['ema_high'] / df['ema_low']) * (df['ema_volume'] - df['sma_volume'])
    return heuristics_matrix
