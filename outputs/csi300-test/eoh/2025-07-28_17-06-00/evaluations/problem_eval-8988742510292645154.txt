def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    daily_change_factor = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    heuristics_matrix = df['ema_volume'] * (df['close'] / df['ema_close']) * (1 + daily_change_factor)
    return heuristics_matrix
