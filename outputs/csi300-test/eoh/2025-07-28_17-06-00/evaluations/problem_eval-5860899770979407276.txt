import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    high = df['high']
    low = df['low']
    
    # Rate of Change (ROC) for closing price over a 10-day period
    roc_close = close.pct_change(periods=10)
    
    # Relative Strength Index (RSI) for volume over a 14-day period
    delta_volume = volume.diff()
    up_days = delta_volume.clip(lower=0)
    down_days = -delta_volume.clip(upper=0)
    ema_up = up_days.ewm(span=14, adjust=False).mean()
    ema_down = down_days.ewm(span=14, adjust=False).mean()
    rs = ema_up / ema_down
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Average True Range (ATR) over a 14-day period
    tr = pd.concat([high - low, abs(high - close.shift()), abs(low - close.shift())], axis=1).max(axis=1)
    atr = tr.ewm(span=14, adjust=False).mean()
    
    # Alpha factor as a combination of ROC close, RSI volume, and ATR
    heuristics_matrix = (roc_close + rsi_volume) * atr
    
    return heuristics_matrix
