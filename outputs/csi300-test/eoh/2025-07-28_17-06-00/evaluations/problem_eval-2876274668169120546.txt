def heuristics_v2(df):
    window_size = 10
    df['price_diff'] = df['high'] - df['low']
    df['momentum_price_diff'] = df['price_diff'].pct_change(window_size)
    df['volume_sum'] = df['volume'].rolling(window=window_size).sum()
    adjustment_factor = df['volume'] / df['volume_sum']
    heuristics_matrix = df['momentum_price_diff'] * adjustment_factor
    return heuristics_matrix
