def heuristics_v2(df):
    window_size = 20
    df['sma_amount'] = df['amount'].rolling(window=window_size).mean()
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    df['roc_close'] = df['close'].pct_change()
    heuristics_matrix = df['roc_close'] + (df['sma_amount'] / df['sma_volume']) * (df['close'] - df['open'])
    return heuristics_matrix
