def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean()
    std_close = df['close'].rolling(window=window_size).std()
    weight_close_ratio = std_close / (std_close + df['amount'].rolling(window=window_size).std())
    heuristics_matrix = (df['close'] / df['ema_close']) * weight_close_ratio + df['ema_amount'] * (1 - weight_close_ratio)
    return heuristics_matrix
