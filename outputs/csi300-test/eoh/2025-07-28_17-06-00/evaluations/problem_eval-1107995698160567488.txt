def heuristics_v2(df):
    window_size = 20
    df['daily_return'] = df['close'].pct_change()
    df['std_daily_return'] = df['daily_return'].rolling(window=window_size).std()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['volume_to_ema_ratio'] = df['volume'] / df['ema_volume']
    heuristics_matrix = (df['close'] - df['close'].shift(window_size)) / df['close'].shift(window_size) * df['std_daily_return'] * df['volume_to_ema_ratio']
    return heuristics_matrix
