def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    amount = df['amount']
    
    # Weighted moving average for closing price and volume
    wma_close = close.rolling(window=10).mean()
    wma_volume = volume.rolling(window=10).mean()
    
    # Rate of change in amount
    roc_amount = (amount - amount.shift(10)) / amount.shift(10)
    
    # Alpha factor as a combination of WMA close, WMA volume, and ROC amount
    heuristics_matrix = (wma_close + wma_volume) * roc_amount
    
    return heuristics_matrix
