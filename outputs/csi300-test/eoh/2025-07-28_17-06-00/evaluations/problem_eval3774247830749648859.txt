def heuristics_v2(df):
    window_size = 20
    df['ema_high'] = df['high'].ewm(span=window_size, adjust=False).mean()
    df['ema_low'] = df['low'].ewm(span=window_size, adjust=False).mean()
    df['roc_ema_high'] = df['ema_high'].pct_change()
    df['roc_ema_low'] = df['ema_low'].pct_change()
    df['roc_ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean().pct_change()
    heuristics_matrix = (df['roc_ema_high'] - df['roc_ema_low']) * (df['roc_ema_amount'] * (df['close'] / df['high'].rolling(window=window_size).max()))
    return heuristics_matrix
