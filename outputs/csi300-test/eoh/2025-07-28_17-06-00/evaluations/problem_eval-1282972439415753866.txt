def heuristics_v2(df):
    window_size = 30
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    df['daily_return'] = df['close'].pct_change()
    df['volatility'] = df['daily_return'].rolling(window=window_size).std()
    adjustment_factor = df['volume'] / df['sma_volume']
    heuristics_matrix = (df['close'] / df['sma_close']) * df['volatility'] * adjustment_factor
    return heuristics_matrix
