def heuristics_v2(df):
    window_size = 10
    df['ema_open'] = df['open'].ewm(span=window_size, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['sma_high'] = df['high'].rolling(window=window_size).mean()
    adjustment_factor = (df['high'] - df['sma_high']) / df['high']
    heuristics_matrix = df['ema_volume'] * (df['open'] / df['ema_open']) * (1 + adjustment_factor)
    return heuristics_matrix
