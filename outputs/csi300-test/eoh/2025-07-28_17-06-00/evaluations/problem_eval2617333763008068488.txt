def heuristics_v2(df):
    # Calculate daily price change
    price_change = df['close'].diff(1).fillna(0)
    
    # Calculate the 10-day moving average of the close price as a measure of momentum
    momentum = df['close'].rolling(window=10).mean()
    
    # Compute the weighted cumulative sum of volume based on price change
    volume_weighted = (df['volume'] * price_change.abs()).cumsum()
    
    # Combine the momentum with the weighted volume
    heuristics_matrix = momentum * volume_weighted
    
    return heuristics_matrix
