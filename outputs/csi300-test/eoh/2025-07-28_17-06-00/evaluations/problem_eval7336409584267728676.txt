def heuristics_v2(df):
    window_size_ema_close = 10
    window_size_ema_volume = 30
    
    df['ema_close'] = df['close'].ewm(span=window_size_ema_close, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size_ema_volume, adjust=False).mean()
    
    heuristics_matrix = df['ema_volume'] * (df['close'] / df['ema_close'])
    
    return heuristics_matrix
