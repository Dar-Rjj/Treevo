def heuristics_v2(df):
    short_window = 10
    df['ema_close'] = df['close'].ewm(span=short_window, adjust=False).mean()
    df['ema_amount'] = df['amount'].ewm(span=short_window, adjust=False).mean()
    df['sma_high'] = df['high'].rolling(window=short_window).mean()
    adjustment_factor = (df['high'] - df['sma_high']) / df['sma_high']
    sma_amount = df['amount'].rolling(window=short_window).mean()
    heuristics_matrix = sma_amount * (df['close'] / df['ema_close']) * (1 + adjustment_factor)
    return heuristics_matrix
