def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute simple moving average
    window_size = 60
    df['sma'] = df['pct_change_close'].rolling(window=window_size).mean()
    
    # Add momentum indicator: standard deviation of high/low over a period
    df['momentum'] = (df['high'].rolling(window=window_size).std() + df['low'].rolling(window=window_size).std()) / 2
    
    # Heuristic formula: Simple Moving Average + Momentum
    heuristics_matrix = df['sma'] + df['momentum']
    
    return heuristics_matrix
