def heuristics_v2(df):
    window_size = 30
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    df['volatility'] = df['volume'].rolling(window=window_size).std()
    heuristics_matrix = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * (df['close'] / df['ema_close']) * (df['volatility'] / df['sma_volume'])
    return heuristics_matrix
