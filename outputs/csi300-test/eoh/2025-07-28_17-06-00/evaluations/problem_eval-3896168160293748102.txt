def heuristics_v2(df):
    ema_volume_window = 10
    ema_close_window = 30
    intraday_volatility_window = 5
    
    df['ema_close'] = df['close'].ewm(span=ema_close_window, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=ema_volume_window, adjust=False).mean()
    price_trend_ratio = df['close'] / df['ema_close']
    
    # Calculating the moving average of the high-low range
    ma_high_low_diff = (df['high'] - df['low']).rolling(window=intraday_volatility_window).mean()
    
    heuristics_matrix = df['ema_volume'] * price_trend_ratio * ma_high_low_diff
    return heuristics_matrix
