def heuristics_v2(df):
    window_size = 15
    df['std_high_low'] = (df['high'].rolling(window=window_size).std() + df['low'].rolling(window=window_size).std()) / 2
    df['max_close'] = df['close'].rolling(window=window_size).max()
    df['min_close'] = df['close'].rolling(window=window_size).min()
    df['median_close'] = df['close'].rolling(window=window_size).median()
    heuristics_matrix = df['std_high_low'] * (df['max_close'] / df['min_close']) * (df['close'] - df['median_close'])
    return heuristics_matrix
