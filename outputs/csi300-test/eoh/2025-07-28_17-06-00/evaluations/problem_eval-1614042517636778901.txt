def heuristics_v2(df):
    # Define the lookback window for calculations
    window_size = 20
    
    # Calculate the exponential moving average (EMA) of the closing price
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    
    # Calculate the EMA of the trading volume
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    
    # Construct the heuristics matrix by combining the difference between the closing price and its EMA
    # with the ratio of the trading volume to its EMA
    heuristics_matrix = (df['close'] - df['ema_close']) * (df['volume'] / df['ema_volume'])
    
    return heuristics_matrix
