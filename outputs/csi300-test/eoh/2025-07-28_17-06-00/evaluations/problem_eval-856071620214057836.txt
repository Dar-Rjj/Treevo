import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    high = df['high']
    low = df['low']
    
    # Momentum of the closing price over a 10-day period
    momentum_close = close / close.shift(10) - 1
    
    # Volume-Weighted Average Price (VWAP)
    typical_price = (high + low + close) / 3
    vwap = (typical_price * volume).cumsum() / volume.cumsum()
    
    # Bollinger Bands: Middle Band (20-day moving average), Upper and Lower Bands
    ma20 = close.rolling(window=20).mean()
    std20 = close.rolling(window=20).std()
    upper_band = ma20 + 2*std20
    lower_band = ma20 - 2*std20
    bollinger_width = (upper_band - lower_band) / ma20
    
    # Alpha factor as a combination of momentum close, VWAP, and Bollinger Bands width
    heuristics_matrix = (momentum_close + (vwap - close) / close) * bollinger_width
    
    return heuristics_matrix
