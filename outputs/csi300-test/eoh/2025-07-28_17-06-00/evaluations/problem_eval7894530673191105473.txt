import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define the lookback window for calculations
    window_size = 14
    
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=window_size).mean()
    avg_loss = loss.rolling(window=window_size).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Average True Range (ATR)
    tr = pd.DataFrame({
        'high-low': df['high'] - df['low'],
        'high-close': np.abs(df['high'] - df['close'].shift()),
        'low-close': np.abs(df['low'] - df['close'].shift())
    }).max(axis=1)
    atr = tr.rolling(window=window_size).mean()
    
    # Calculate the Mean Absolute Deviation (MAD) of the closing price
    mad = df['close'].rolling(window=window_size).apply(lambda x: np.mean(np.abs(x - np.mean(x))), raw=True)
    
    # Construct the heuristics matrix by combining the RSI and the ratio of ATR to MAD
    heuristics_matrix = rsi + (atr / mad)
    
    return heuristics_matrix
