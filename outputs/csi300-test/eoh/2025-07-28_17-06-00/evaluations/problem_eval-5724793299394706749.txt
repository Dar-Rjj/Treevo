def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_high'] = df['high'].ewm(span=window_size, adjust=False).mean()
    df['ema_low'] = df['low'].ewm(span=window_size, adjust=False).mean()
    heuristics_matrix = (df['ema_high'] + df['ema_low']) * (df['close'] / df['ema_close'])
    return heuristics_matrix
