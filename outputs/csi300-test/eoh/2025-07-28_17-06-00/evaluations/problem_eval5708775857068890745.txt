def heuristics_v2(df):
    # Calculate short and long term moving averages for closing prices to determine trend
    short_mavg = df['close'].rolling(window=10).mean()
    long_mavg = df['close'].rolling(window=50).mean()
    
    # Determine the trend strength by the difference between the short and long term moving averages
    trend_strength = short_mavg - long_mavg
    
    # Daily price and volume changes
    price_change = df['close'].diff(1).fillna(0)
    volume_change = df['volume'].diff(1).fillna(0)
    
    # Calculate a combined signal, weighted by the trend strength
    weighted_signal = (price_change + volume_change) * trend_strength
    
    # Generate the final heuristics matrix as a cumulative sum of the weighted signals
    heuristics_matrix = weighted_signal.cumsum()
    
    return heuristics_matrix
