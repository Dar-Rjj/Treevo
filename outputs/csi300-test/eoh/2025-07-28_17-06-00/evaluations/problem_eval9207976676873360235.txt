def heuristics_v2(df):
    # Calculate daily returns
    daily_return = df['close'].pct_change()
    
    # Standard deviation of daily returns as a proxy for volatility
    std_dev_daily_returns = daily_return.rolling(window=10).std()
    
    # Relative Strength Index (RSI) to gauge recent price momentum
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Cumulative sum of volume over a trailing 10 day window to capture liquidity trends
    cum_sum_volume = df['volume'].rolling(window=10).sum()
    
    # Combine the RSI-adjusted standard deviation (volatility) with the cumulative sum of volume
    heuristics_matrix = (std_dev_daily_returns * rsi) * cum_sum_volume
    
    return heuristics_matrix
