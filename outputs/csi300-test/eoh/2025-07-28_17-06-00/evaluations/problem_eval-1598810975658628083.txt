def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    adjustment_factor = (df['close'] - df['ema_close']) / df['close']
    daily_range_ratio = df['high'] / df['low']
    heuristics_matrix = df['ema_volume'] * adjustment_factor * daily_range_ratio
    return heuristics_matrix
