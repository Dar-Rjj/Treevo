def heuristics_v2(df):
    window_size = 20
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    roc_close = df['close'].pct_change(periods=window_size)
    cv_close = (df['close'].rolling(window=window_size).std()) / df['close'].rolling(window=window_size).mean()
    weight_roc = cv_close / (cv_close + (df['volume'].rolling(window=window_size).std() / df['volume'].rolling(window=window_size).mean()))
    heuristics_matrix = roc_close * weight_roc + (df['volume'] / df['ema_volume']) * (1 - weight_roc)
    return heuristics_matrix
