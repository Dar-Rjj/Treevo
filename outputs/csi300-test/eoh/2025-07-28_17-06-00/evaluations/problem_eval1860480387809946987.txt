def heuristics_v2(df):
    window_size = 14
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['liquidity_ratio'] = df['amount'] / df['volume']
    heuristics_matrix = (df['ema_close'] - df['sma_close']) * df['liquidity_ratio']
    return heuristics_matrix
