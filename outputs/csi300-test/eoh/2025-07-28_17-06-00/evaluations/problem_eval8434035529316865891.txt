def heuristics_v2(df):
    window_size = 20
    df['wma_close'] = (df['close'] * df['volume']).rolling(window=window_size).sum() / df['volume'].rolling(window=window_size).sum()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    adjustment_factor = 1 - ((df['close'] - df['sma_close']) / df['close']).abs()
    heuristics_matrix = df['ema_volume'] * (df['close'] / df['wma_close']) * adjustment_factor
    return heuristics_matrix
