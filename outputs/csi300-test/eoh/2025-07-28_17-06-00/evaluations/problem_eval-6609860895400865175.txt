def heuristics_v2(df):
    # Define the lookback window for calculations and the weight parameter
    window_size = 10
    weight_volume = 0.7
    
    # Calculate the exponential moving average (EMA) of the closing price
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    
    # Calculate the EMA of the trading volume
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    
    # Construct the heuristics matrix
    heuristics_matrix = weight_volume * df['ema_volume'] + (1 - weight_volume) * (df['close'] / df['ema_close'])
    
    return heuristics_matrix
