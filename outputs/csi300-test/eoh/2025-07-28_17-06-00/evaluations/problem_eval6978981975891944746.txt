def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    wma_close = close.rolling(window=10).apply(lambda x: (x * range(1, 11)).sum() / sum(range(1, 11)), raw=True)
    wma_volume = volume.rolling(window=10).apply(lambda x: (x * range(1, 11)).sum() / sum(range(1, 11)), raw=True)
    momentum = (df['close'] / df['close'].shift(5)) - 1
    heuristics_matrix = (wma_close + wma_volume) * momentum
    return heuristics_matrix
