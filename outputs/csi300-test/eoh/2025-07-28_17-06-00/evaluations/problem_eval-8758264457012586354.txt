def heuristics_v2(df):
    # Calculate daily returns
    daily_return = df['close'].pct_change()
    
    # Exponentially weighted moving average of daily returns as a proxy for volatility
    ewma_daily_returns = daily_return.ewm(span=20, adjust=False).std()
    
    # Average volume over a trailing 20 day window to capture steady liquidity trends
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Combine the EWMA (volatility) with the average volume
    heuristics_matrix = ewma_daily_returns * avg_volume
    
    return heuristics_matrix
