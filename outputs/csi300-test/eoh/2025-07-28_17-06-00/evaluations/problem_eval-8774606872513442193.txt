def heuristics_v2(df):
    window_size = 20
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['momentum'] = df['close'].pct_change(periods=window_size)
    df['volatility'] = df['close'].rolling(window=window_size).std()
    adjustment_factor = df['volatility'] / df['close']
    heuristics_matrix = (df['volume'] / df['ema_volume']) * df['momentum'] * (1 + adjustment_factor)
    return heuristics_matrix
