def heuristics_v2(df):
    # Rate of Change (ROC) for trading amount over a 10-day period
    roc_amount = df['amount'].pct_change(periods=10)
    
    # Ratio of the highest price to the lowest price over a 14-day period
    high_low_ratio = df['high'].rolling(window=14).max() / df['low'].rolling(window=14).min()
    
    # Exponential Moving Average (EMA) of the closing price over a 20-day period
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Alpha factor as a combination of ROC amount, high-low ratio, and EMA close
    heuristics_matrix = (roc_amount + high_low_ratio) * ema_close
    
    return heuristics_matrix
