def heuristics_v2(df):
    window_size = 14
    df['roc_volume'] = df['volume'].pct_change(window_size)
    gain = df['close'].diff().clip(lower=0).rolling(window=window_size).mean()
    loss = -df['close'].diff().clip(upper=0).rolling(window=window_size).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    heuristics_matrix = df['roc_volume'] * rsi
    return heuristics_matrix
