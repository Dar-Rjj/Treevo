def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    high = df['high']
    low = df['low']
    
    # Exponential Moving Average for closing price with a 20-day span
    ema_close = close.ewm(span=20, adjust=False).mean()
    
    # Percentage change in volume over a 10-day period
    pct_change_volume = volume.pct_change(periods=10)
    
    # Ratio of highest to lowest price within a 20-day lookback window
    high_low_ratio = high.rolling(window=20).max() / low.rolling(window=20).min()
    
    # Alpha factor as a combination of EMA close, % change in volume, and high-low ratio
    heuristics_matrix = (ema_close + pct_change_volume) * high_low_ratio
    
    return heuristics_matrix
