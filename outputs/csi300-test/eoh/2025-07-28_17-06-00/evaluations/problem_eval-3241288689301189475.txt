def heuristics_v2(df):
    # Calculate rate of change for close price
    roc_close = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate the standard deviation of (high - low) over a 14-day window
    std_high_low = (df['high'] - df['low']).rolling(window=14).std()
    
    # Calculate the ratio of (high - low) to the standard deviation of (high - low) over a 14-day window
    ratio_high_low_std = (df['high'] - df['low']) / std_high_low
    
    # Calculate the rate of change for volume
    roc_volume = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10)
    
    # Heuristic formula: ROC of Close * Ratio of (High - Low) to STD of (High - Low) * ROC of Volume
    heuristics_matrix = roc_close * ratio_high_low_std * roc_volume
    
    return heuristics_matrix
