def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/window_size, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/window_size, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    heuristics_matrix = (df['ema_amount'] - df['ema_volume']) * rsi
    return heuristics_matrix
