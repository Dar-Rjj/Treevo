def heuristics_v2(df):
    # Calculate daily price change
    price_change = df['close'].diff(1).fillna(0)
    
    # Calculate the difference between high and low as a measure of intraday volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Compute the weighted cumulative difference between up and down days in volume
    volume_weighted_diff = ((df['volume'] * (price_change > 0)) - (df['volume'] * (price_change < 0))) * (price_change.abs() / df['close']).cumsum()
    
    # Combine the rolling sum of the daily price change with the weighted volume difference
    heuristics_matrix = price_change.rolling(window=10).sum() + (intraday_volatility * volume_weighted_diff)
    
    return heuristics_matrix
