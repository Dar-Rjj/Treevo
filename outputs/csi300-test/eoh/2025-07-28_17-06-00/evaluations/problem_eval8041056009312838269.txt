def heuristics_v2(df):
    window_size = 50
    df['price_momentum'] = df['close'].pct_change(periods=window_size)
    delta = df['volume'].diff(1)
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=window_size).mean()
    avg_loss = loss.rolling(window=window_size).mean()
    rs = avg_gain / avg_loss
    rsi_volume = 100 - (100 / (1 + rs))
    heuristics_matrix = df['price_momentum'] * rsi_volume
    return heuristics_matrix
