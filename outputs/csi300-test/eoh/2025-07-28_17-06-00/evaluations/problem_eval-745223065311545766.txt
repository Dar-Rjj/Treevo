def heuristics_v2(df):
    window_size = 20
    df['ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean()
    df['sma_amount'] = df['amount'].rolling(window=window_size).mean()
    df['roc_volume'] = df['volume'].pct_change()
    df['std_close'] = df['close'].rolling(window=window_size).std()
    heuristics_matrix = df['roc_volume'] * df['std_close'] * (df['ema_amount'] - df['sma_amount'])
    return heuristics_matrix
