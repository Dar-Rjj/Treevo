def heuristics_v2(df):
    window_size = 20
    df['std_close'] = df['close'].rolling(window=window_size).std()
    df['avg_volume'] = df['volume'].rolling(window=window_size).mean()
    df['high_low_diff'] = df['high'].rolling(window=window_size).max() - df['low'].rolling(window=window_size).min()
    heuristics_matrix = (df['std_close'] * (df['volume'] / df['avg_volume'])) + df['high_low_diff']
    return heuristics_matrix
