def heuristics_v2(df):
    window_size = 20
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['highest_close'] = df['close'].rolling(window=window_size).max()
    close_strength = (df['close'] - df['close'].shift(window_size)) / df['close'].shift(window_size)
    weight_close_ratio = close_strength / (close_strength + (df['volume'] - df['volume'].shift(window_size)) / df['volume'].shift(window_size))
    heuristics_matrix = (df['close'] / df['highest_close']) * weight_close_ratio + df['ema_volume'] * (1 - weight_close_ratio)
    return heuristics_matrix
