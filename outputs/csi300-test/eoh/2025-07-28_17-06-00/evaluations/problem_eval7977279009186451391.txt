def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    high = df['high']
    low = df['low']
    
    # Momentum of closing price over a 20-day period
    momentum_close = close / close.shift(20) - 1
    
    # Relative strength index (RSI) over a 14-day period
    delta = close.diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volatility-adjusted trading volume
    volatility = (high - low).rolling(window=20).std()
    adj_volume = volume / volatility
    
    # Alpha factor as a combination of momentum, RSI, and adjusted volume
    heuristics_matrix = (momentum_close + rsi) * adj_volume
    
    return heuristics_matrix
