def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['rsi_close'] = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=window_size).mean() / df['close'].diff(1).clip(upper=0).abs().rolling(window=window_size).mean())))
    adjustment_factor = (df['close'] - df['ema_close']) / df['close']
    heuristics_matrix = df['rsi_close'] * (df['ema_volume'] / df['sma_volume']) * (1 + adjustment_factor)
    return heuristics_matrix
