def heuristics_v2(df):
    window_size = 30
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    price_trend_ratio = df['close'] / df['ema_close']
    intraday_volatility = (df['high'] - df['low']) / df['close']
    heuristics_matrix = df['ema_volume'] * (0.7 * price_trend_ratio + 0.3 * intraday_volatility)
    return heuristics_matrix
