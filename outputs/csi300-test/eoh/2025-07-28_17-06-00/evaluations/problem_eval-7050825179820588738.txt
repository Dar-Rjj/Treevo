def heuristics_v2(df):
    window_size = 30
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['roc_sma_close'] = df['sma_close'].pct_change()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean()
    heuristics_matrix = df['roc_sma_close'] + (df['ema_volume'] / df['ema_amount']) * (df['sma_close'] / df['close'])
    return heuristics_matrix
