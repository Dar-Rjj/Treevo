def heuristics_v2(df):
    window_size = 10
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['roc_sma_close'] = df['sma_close'].pct_change()
    df['roc_ema_close'] = df['ema_close'].pct_change()
    df['sma_amount'] = df['amount'].rolling(window=window_size).mean()
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    df['std_volume'] = df['volume'].rolling(window=window_size).std()
    heuristics_matrix = df['roc_sma_close'] + df['roc_ema_close'] + (df['sma_amount'] - df['sma_volume']) * (df['std_volume'] / df['close'])
    return heuristics_matrix
