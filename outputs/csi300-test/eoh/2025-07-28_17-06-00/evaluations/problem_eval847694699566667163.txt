def heuristics_v2(df):
    window_size = 20
    df['sma_volume'] = df['volume'].rolling(window=window_size).mean()
    df['price_diff'] = df['high'] - df['low']
    df['roc_close'] = df['close'].pct_change()
    heuristics_matrix = df['price_diff'] * (df['volume'] / df['sma_volume']) * df['roc_close']
    return heuristics_matrix
