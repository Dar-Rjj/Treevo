def heuristics_v2(df):
    window_size = 10
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['sma_close'] = df['close'].rolling(window=window_size * 2).mean()
    adjustment_factor = (df['close'] - df['ema_close']) / df['close']
    heuristics_matrix = df['ema_amount'] * (df['close'] / df['ema_close']) * (1 + adjustment_factor) * df['ema_volume']
    return heuristics_matrix
