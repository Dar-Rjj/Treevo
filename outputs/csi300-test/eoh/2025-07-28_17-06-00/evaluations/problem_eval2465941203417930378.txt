def heuristics_v2(df):
    # Calculate daily price change
    price_change = df['close'].diff(1).fillna(0)
    
    # Calculate the difference between high and low as a measure of intraday volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=10).mean()
    
    # Compute the cumulative difference between up and down days in volume
    volume_diff = (df['volume'] * (price_change > 0) - df['volume'] * (price_change < 0)).cumsum()
    
    # Combine the intraday volatility with the volume difference
    heuristics_matrix = intraday_volatility * volume_diff
    
    return heuristics_matrix
