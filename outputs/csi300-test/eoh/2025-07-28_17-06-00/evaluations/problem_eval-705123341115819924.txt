def heuristics_v2(df):
    window_size = 10
    df['avg_close'] = df['close'].rolling(window=window_size).mean()
    df['high_low_diff'] = df['high'].rolling(window=window_size).max() - df['low'].rolling(window=window_size).min()
    heuristics_matrix = (df['high_low_diff'] + df['close'] / df['avg_close'])
    return heuristics_matrix
