def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['high_low_ratio'] = df['high'].rolling(window=window_size).max() / df['low'].rolling(window=window_size).min()
    heuristics_matrix = (df['ema_close'] - df['sma_close']) * df['high_low_ratio']
    return heuristics_matrix
