def heuristics_v2(df):
    # Calculate the high-low price range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate the standard deviation of the closing prices over a 30-day window as a measure of volatility
    std_window = 30
    df['volatility'] = df['close'].rolling(window=std_window).std()
    
    # Momentum is calculated as the current close relative to the close 30 days ago
    df['momentum'] = df['close'] / df['close'].shift(std_window) - 1
    
    # Adjusting the momentum by the volatility to get a more stable signal
    heuristics_matrix = (df['high_low_range'] * df['momentum']) / df['volatility']

    return heuristics_matrix
