import pandas as pd

def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute exponentially weighted moving average
    window_size = 30
    df['ewma'] = df['pct_change_close'].ewm(span=window_size, adjust=False).mean()
    
    # Add momentum indicator: standard deviation of high/low over a period
    df['momentum'] = df['high'].rolling(window=window_size).std() / df['low'].rolling(window=window_size).std()
    
    # Heuristic formula: Exponentially Weighted Moving Average + Momentum
    heuristics_matrix = df['ewma'] + df['momentum']
    
    return heuristics_matrix
