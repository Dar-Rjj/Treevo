def heuristics_v2(df):
    window_size = 15
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['roc_volume'] = df['volume'].pct_change()
    std_close = df['close'].rolling(window=window_size).std()
    weight_volatility = (df['close'] - df['sma_close']) / std_close
    heuristics_matrix = df['roc_volume'] + weight_volatility * (df['close'] / df['sma_close'])
    return heuristics_matrix
