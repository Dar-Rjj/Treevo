def heuristics_v2(df):
    # Calculate daily price change
    price_change = df['close'].diff(1).fillna(0)
    
    # Calculate the difference between high and low as a measure of intraday volatility
    intraday_volatility = df['high'] - df['low']
    
    # Compute the 10-day moving average of the absolute daily return
    abs_return_ma = price_change.abs().rolling(window=10).mean()
    
    # Calculate the True Range
    tr = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs() + (df['high'] - df['low']).abs() + (df['high'] - df['close'].shift()).abs() + (df['low'] - df['close'].shift()).abs()
    tr = tr / 4
    
    # Calculate the Average True Range (ATR) over a 14-day period
    atr = tr.rolling(window=14).mean()
    
    # Calculate the Relative Strength Index (RSI) for the ATR
    gain = atr.diff(1).apply(lambda x: max(x, 0))
    loss = atr.diff(1).apply(lambda x: abs(min(x, 0)))
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_atr = 100 - (100 / (1 + rs))
    
    # Combine the intraday volatility, 10-day moving average of the absolute daily return, and RSI-ATR
    heuristics_matrix = intraday_volatility * abs_return_ma * rsi_atr
    
    return heuristics_matrix
