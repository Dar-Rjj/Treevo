def heuristics_v2(df):
    def calculate_strength(price, ma):
        return (price - ma) / ma

    short_ma = df['close'].rolling(window=5).mean()
    medium_ma = df['close'].rolling(window=20).mean()
    long_ma = df['close'].rolling(window=60).mean()

    short_strength = calculate_strength(df['close'], short_ma)
    medium_strength = calculate_strength(df['close'], medium_ma)
    long_strength = calculate_strength(df['close'], long_ma)

    heuristics_matrix = 0.3 * short_strength + 0.4 * medium_strength + 0.3 * long_strength
    return heuristics_matrix
