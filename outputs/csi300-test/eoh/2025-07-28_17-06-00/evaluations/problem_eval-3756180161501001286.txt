def heuristics_v2(df):
    # Calculate the 10-day moving average of the closing price
    close_ma = df['close'].rolling(window=10).mean()
    
    # Compute the 10-day rate of change (ROC) of the closing price
    roc_close = df['close'].pct_change(periods=10).fillna(0)
    
    # Calculate the 10-day ratio of the current volume to the average volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Combine the ROC of the closing price with the volume ratio
    heuristics_matrix = roc_close * volume_ratio
    
    return heuristics_matrix
