def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    high = df['high']
    low = df['low']

    # Simple Moving Average for closing price
    sma_close = close.rolling(window=10).mean()
    
    # Exponential Moving Average for closing price
    ema_close = close.ewm(span=10, adjust=False).mean()
    
    # Rate of Change in volume over a 5-day period
    roc_volume = (volume / volume.shift(5)) - 1
    
    # Normalized ratio of highest to lowest price within a 30-day lookback window
    high_low_ratio = (high.rolling(window=30).max() - low.rolling(window=30).min()) / high.rolling(window=30).max()
    
    # Alpha factor as a combination of SMA-EMA difference, ROC in volume, and high-low ratio
    heuristics_matrix = (sma_close - ema_close + roc_volume) * high_low_ratio

    return heuristics_matrix
