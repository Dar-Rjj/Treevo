def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute exponentially weighted moving average
    span = 60
    df['ewma'] = df['pct_change_close'].ewm(span=span, adjust=False).mean()
    
    # Add momentum indicator: difference between the rolling mean of high and low
    df['momentum_shift'] = df['high'].rolling(window=span).mean() - df['low'].rolling(window=span).mean()
    
    # Heuristic formula: Exponentially Weighted Moving Average + Momentum Shift
    heuristics_matrix = df['ewma'] + df['momentum_shift']
    
    return heuristics_matrix
