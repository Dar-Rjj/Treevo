def heuristics_v2(df):
    # Define the lookback window for calculations
    window_size = 20
    
    # Calculate the exponential moving average (EMA) of the closing price
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    
    # Calculate the EMA of the trading amount
    df['ema_amount'] = df['amount'].ewm(span=window_size, adjust=False).mean()
    
    # Construct the heuristics matrix by combining the EMA of the trading amount and the ratio of the closing price to its EMA
    heuristics_matrix = df['ema_amount'] * (df['close'] / df['ema_close'])
    
    return heuristics_matrix
