def heuristics_v2(df):
    window_size = 10
    df['high_low_ratio'] = (df['high'].rolling(window=window_size).max() / df['low'].rolling(window=window_size).min())
    df['amount_change'] = df['amount'].pct_change(periods=window_size)
    heuristics_matrix = df['high_low_ratio'] * df['amount_change']
    return heuristics_matrix
