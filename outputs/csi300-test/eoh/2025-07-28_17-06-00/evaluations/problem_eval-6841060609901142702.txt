def heuristics_v2(df):
    close = df['close']
    amount = df['amount']
    
    # Calculate the rate of change in closing price over a 10-day period
    roc_close = close.pct_change(periods=10).fillna(0)
    
    # Calculate the ratio of the amount traded to the average amount over a 10-day window
    avg_amount = amount.rolling(window=10).mean()
    liquidity_ratio = amount / avg_amount
    
    # Alpha factor as a combination of ROC close and liquidity ratio
    heuristics_matrix = roc_close * liquidity_ratio
    
    return heuristics_matrix
