def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    adjustment_factor = (df['close'] - df['sma_close']) / df['close']
    momentum = df['close'].pct_change(periods=window_size)  # Momentum over the window size
    heuristics_matrix = df['ema_volume'] * (df['close'] / df['ema_close']) * (1 + adjustment_factor) * (1 + momentum)
    return heuristics_matrix
