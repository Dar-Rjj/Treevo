def heuristics_v2(df):
    window_size = 14
    df['roc_volume'] = df['volume'].pct_change()
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=window_size).mean()
    avg_loss = loss.rolling(window=window_size).mean()
    rs = avg_gain / avg_loss
    df['rsi_close'] = 100 - (100 / (1 + rs))
    df['hl_diff_ratio'] = (df['high'] - df['low']) / df['close']
    heuristics_matrix = df['roc_volume'] * df['rsi_close'] * df['hl_diff_ratio']
    return heuristics_matrix
