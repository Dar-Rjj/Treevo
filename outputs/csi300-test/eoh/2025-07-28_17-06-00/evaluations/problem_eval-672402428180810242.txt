def heuristics_v2(df):
    window_size = 20
    long_window_size = 60
    df['ema_volume'] = df['volume'].ewm(span=long_window_size, adjust=False).mean()
    df['z_score_close'] = (df['close'] - df['close'].rolling(window=window_size).mean()) / df['close'].rolling(window=window_size).std()
    heuristics_matrix = df['z_score_close'] + (df['volume'] / df['ema_volume'])
    return heuristics_matrix
