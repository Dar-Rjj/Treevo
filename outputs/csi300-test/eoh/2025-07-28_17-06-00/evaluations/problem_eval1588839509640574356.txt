def heuristics_v2(df):
    window_size = 20
    
    # Calculate the RSI of the closing price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=window_size).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window_size).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the EMA of the trading volume
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    
    # Calculate the adjustment factor based on the deviation of the closing price from its highest value over the past 20 days
    df['highest_close_20'] = df['close'].rolling(window=window_size).max()
    adjustment_factor = (df['close'] - df['highest_close_20']) / df['close']
    
    # Construct the heuristics matrix
    heuristics_matrix = rsi * (df['volume'] / df['ema_volume']) * (1 + adjustment_factor)
    
    return heuristics_matrix
