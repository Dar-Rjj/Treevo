import pandas as pd

def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute exponentially weighted moving average with an adjusted span
    span = 20
    df['ewma'] = df['pct_change_close'].ewm(span=span, adjust=False).mean()
    
    # Add momentum indicator: ratio of the sum of high to the sum of low over a fixed period
    df['momentum'] = df['high'].rolling(window=span).sum() / df['low'].rolling(window=span).sum()
    
    # Heuristic formula: Exponentially Weighted Moving Average + Momentum
    heuristics_matrix = df['ewma'] + df['momentum']
    
    return heuristics_matrix
