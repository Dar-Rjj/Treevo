def heuristics_v2(df):
    window_size = 20
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    momentum_close = df['close'].pct_change(window_size)
    weight_momentum = (momentum_close + abs(momentum_close.min())) / (momentum_close + abs(momentum_close.min()) + df['volume'].pct_change(window_size) + abs(df['volume'].pct_change(window_size).min()))
    heuristics_matrix = (df['close'] / df['ema_close']) * weight_momentum + df['ema_volume'] * (1 - weight_momentum)
    return heuristics_matrix
