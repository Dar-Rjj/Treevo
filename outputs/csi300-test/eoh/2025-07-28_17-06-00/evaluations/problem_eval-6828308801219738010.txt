import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']

    # Calculate MACD for closing prices
    ema_12 = close.ewm(span=12, adjust=False).mean()
    ema_26 = close.ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    
    # Calculate ROC for trading volume over a 5-day period
    roc_volume = volume.pct_change(periods=5)
    
    # Alpha factor as a combination of MACD and ROC of volume
    heuristics_matrix = macd_line + roc_volume
    
    return heuristics_matrix
