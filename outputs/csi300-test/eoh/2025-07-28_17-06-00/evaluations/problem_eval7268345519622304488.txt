import pandas as pd

def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute weighted moving average with 'volume' as weights
    window_size = 30
    df['wma'] = (df['pct_change_close'] * df['volume']).rolling(window=window_size).sum() / df['volume'].rolling(window=window_size).sum()
    
    # Add momentum indicator: ratio of high/low over a period
    df['momentum'] = df['high'].rolling(window=window_size).mean() / df['low'].rolling(window=window_size).mean()
    
    # Heuristic formula: Weighted Moving Average + Momentum
    heuristics_matrix = df['wma'] + df['momentum']
    
    return heuristics_matrix
