def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    df['MA_short'] = df['close'].rolling(window=10).mean()
    df['MA_long'] = df['close'].rolling(window=50).mean()
    # Compute the relative strength indicator (RSI) without normalization
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    RS = avg_gain / avg_loss
    RSI = 100 - (100 / (1 + RS))
    # Combine MA crossover signal with RSI for the heuristics matrix
    df['Signal'] = (df['MA_short'] > df['MA_long']).astype(int)
    heuristics_matrix = df['Signal'] * (100 - RSI)
    return heuristics_matrix
