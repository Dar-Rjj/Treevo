def heuristics_v2(df):
    # Calculate percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Compute exponentially weighted moving average
    span = 60
    df['ewma'] = df['pct_change_close'].ewm(span=span, adjust=False).mean()
    
    # Add momentum indicator: ratio of rolling mean of high/low adjusted by their standard deviations
    rolling_high = df['high'].rolling(window=span).mean()
    rolling_low = df['low'].rolling(window=span).mean()
    std_high = df['high'].rolling(window=span).std()
    std_low = df['low'].rolling(window=span).std()
    df['momentum'] = (rolling_high / rolling_low) * (std_high / std_low)
    
    # Heuristic formula: Exponentially Weighted Moving Average + Volatility-Adjusted Momentum
    heuristics_matrix = df['ewma'] + df['momentum']
    
    return heuristics_matrix
