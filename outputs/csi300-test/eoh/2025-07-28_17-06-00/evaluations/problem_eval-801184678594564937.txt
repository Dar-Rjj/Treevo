def heuristics_v2(df):
    momentum_window = 20
    volatility_window = 20
    df['daily_return'] = df['close'].pct_change()
    df['momentum'] = df['close'].pct_change(periods=momentum_window)
    df['volatility'] = df['daily_return'].rolling(window=volatility_window).std()
    df['cumulative_volume'] = df['volume'].rolling(window=momentum_window).sum()
    heuristics_matrix = df['momentum'] * df['volatility'] * df['cumulative_volume']
    return heuristics_matrix
