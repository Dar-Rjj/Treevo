def heuristics_v2(df):
    window_size = 30
    df['ema_volume'] = df['volume'].ewm(span=window_size, adjust=False).mean()
    df['sma_amount'] = df['amount'].rolling(window=window_size).mean()
    df['roc_close'] = df['close'].pct_change()
    df['momentum_close'] = df['close'] - df['close'].shift(window_size)
    heuristics_matrix = df['roc_close'] + (df['ema_volume'] / df['sma_amount']) * df['momentum_close']
    return heuristics_matrix
