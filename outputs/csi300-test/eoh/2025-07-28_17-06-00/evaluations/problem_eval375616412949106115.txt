def heuristics_v2(df):
    window_size = 20
    df['price_momentum'] = df['close'].pct_change(periods=window_size).fillna(0)
    df['wma_volume'] = df['volume'].rolling(window=window_size).apply(lambda x: (x * range(1, window_size + 1)).sum() / sum(range(1, window_size + 1)), raw=True).fillna(0)
    recent_volatility = df['close'].rolling(window=window_size).std().fillna(0)
    heuristics_matrix = df['price_momentum'] * df['wma_volume'] / recent_volatility
    return heuristics_matrix
