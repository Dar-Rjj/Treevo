def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    
    # Simple Moving Averages for closing price with 20-day and 50-day spans
    sma_close_20 = close.rolling(window=20).mean()
    sma_close_50 = close.rolling(window=50).mean()
    
    # Difference between 20-day and 50-day SMAs
    sma_diff = sma_close_20 - sma_close_50
    
    # Absolute value of the percentage change in volume over a 10-day period
    abs_pct_change_volume = volume.pct_change(periods=10).abs()
    
    # Standard deviation of the closing prices over a 20-day lookback window
    std_close_20 = close.rolling(window=20).std()
    
    # Alpha factor as a combination of SMA difference, absolute % change in volume, and standard deviation
    heuristics_matrix = (sma_diff + abs_pct_change_volume) * std_close_20
    
    return heuristics_matrix
