def heuristics_v2(df):
    window_short = 5
    window_long = 20
    window_ema = 30
    df['ema_close'] = df['close'].ewm(span=window_ema, adjust=False).mean()
    df['high_low_ratio'] = df['high'] / df['low']
    df['ema_volume_5'] = df['volume'].ewm(span=window_short, adjust=False).mean()
    df['ema_volume_20'] = df['volume'].ewm(span=window_long, adjust=False).mean()
    adjustment_factor = (df['ema_volume_5'] - df['ema_volume_20']) / df['volume']
    heuristics_matrix = df['ema_close'] * df['high_low_ratio'] * (1 + adjustment_factor)
    return heuristics_matrix
