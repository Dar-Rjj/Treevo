def heuristics_v2(df):
    # Calculate rate of change for close price
    roc_close = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate the ratio of (high - low) to the average true range over a 14-day window
    atr = df['high'] - df['low']
    avg_true_range = atr.rolling(window=14).mean()
    ratio_high_low_atr = (df['high'] - df['low']) / avg_true_range
    
    # Heuristic formula: ROC of Close * Ratio of (High - Low) to ATR
    heuristics_matrix = roc_close * ratio_high_low_atr
    
    return heuristics_matrix
