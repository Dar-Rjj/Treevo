def heuristics_v2(df):
    # Define the lookback windows for calculations
    short_window = 10
    long_window = 50
    
    # Calculate the Exponential Moving Averages (EMAs)
    ema_short = df['close'].ewm(span=short_window, adjust=False).mean()
    ema_long = df['close'].ewm(span=long_window, adjust=False).mean()
    
    # Calculate the EMA difference
    ema_diff = ema_short - ema_long
    
    # Calculate the volume-to-amount ratio
    volume_to_amount_ratio = df['volume'] / df['amount']
    
    # Construct the heuristics matrix by combining the EMA difference and the volume-to-amount ratio
    heuristics_matrix = ema_diff * volume_to_amount_ratio
    
    return heuristics_matrix
