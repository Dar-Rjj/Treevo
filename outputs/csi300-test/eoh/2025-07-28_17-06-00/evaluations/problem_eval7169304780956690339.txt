def heuristics_v2(df):
    # Calculate daily returns
    daily_return = df['close'].pct_change()
    
    # Z-score of daily returns
    z_score_returns = (daily_return - daily_return.mean()) / daily_return.std()
    
    # Calculate the long-term moving average of volume over 50 days
    long_term_avg_volume = df['volume'].rolling(window=50).mean()
    
    # Percentage change in volume compared to the long-term average
    pct_change_volume = (df['volume'] - long_term_avg_volume) / long_term_avg_volume * 100
    
    # Coefficient of variation (CV) of high and low prices over a 10-day period
    cv_high_low = (df['high'] - df['low']).rolling(window=10).std() / (df['high'] + df['low']) * 2
    
    # Alpha factor as a combination of Z-score, percentage change in volume, and CV
    heuristics_matrix = z_score_returns + pct_change_volume + cv_high_low
    
    return heuristics_matrix
