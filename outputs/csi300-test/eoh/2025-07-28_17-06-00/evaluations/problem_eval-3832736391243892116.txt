def heuristics_v2(df):
    # Define the lookback window for calculations
    window_size = 20
    
    # Calculate the average trading amount over the past 20 days
    avg_amount = df['amount'].rolling(window=window_size).mean()
    
    # Calculate the standard deviation of the closing prices over the past 20 days
    std_close = df['close'].rolling(window=window_size).std()
    
    # Construct the heuristics matrix by combining the average trading amount and the standard deviation of the closing prices
    heuristics_matrix = avg_amount / std_close
    
    return heuristics_matrix
