def heuristics_v2(df):
    window_size = 30
    df['sma_close'] = df['close'].rolling(window=window_size).mean()
    df['ema_close'] = df['close'].ewm(span=window_size, adjust=False).mean()
    df['volatility'] = df['close'].pct_change().rolling(window=window_size).std()
    heuristics_matrix = (df['ema_close'] - df['sma_close']) / df['volatility']
    return heuristics_matrix
