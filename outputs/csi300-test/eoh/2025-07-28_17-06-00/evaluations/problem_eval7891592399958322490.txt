def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    # Shift volume to align with next day's return
    df['shifted_volume'] = df['volume'].shift(-1)
    # Define a heuristic factor as the product of daily return and shifted volume
    heuristics_matrix = df['daily_return'] * df['shifted_volume']
    return heuristics_matrix
