import pandas as pd

def heuristics_v2(df):
    window_size = 30
    # Calculate daily returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=window_size).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window_size).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate Volatility
    df['volatility'] = df['daily_returns'].rolling(window=window_size).std()
    
    # Combine RSI and Volatility, scaled by average volume
    avg_volume = df['volume'].rolling(window=window_size).mean()
    heuristics_matrix = (df['rsi'] * df['volatility']) / avg_volume
    
    return heuristics_matrix
