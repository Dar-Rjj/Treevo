import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_ratio = (ema_20 / ema_50) - 1
    
    # Adjusted Stochastic Oscillator
    low_21 = df['low'].rolling(window=21).min()
    high_21 = df['high'].rolling(window=21).max()
    k_percent = 100 * ((df['close'] - low_21) / (high_21 - low_21))
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + k_percent + rsi + volume_weighted) / 4
    return heuristics_matrix
