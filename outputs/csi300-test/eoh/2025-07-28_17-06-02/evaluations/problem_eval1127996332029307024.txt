def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    ema_ratio = (ema_7 / ema_14) - 1
    
    # Modified Relative Strength Index (RSI) with a 14-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Standard Deviation of Returns
    std_dev = df['close'].pct_change().rolling(window=30).std()
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + rsi - std_dev + volume_weighted) / 4
    return heuristics_matrix
