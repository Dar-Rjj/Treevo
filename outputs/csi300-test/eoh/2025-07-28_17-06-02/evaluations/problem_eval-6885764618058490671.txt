def heuristics_v2(df):
    # Momentum factor: 7-day percentage change in close price
    momentum = df['close'].pct_change(7)
    
    # Modified Volatility Clustering: Incorporate both ATR and close-to-previous close price difference
    hl_range = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    cpc_diff = (df['close'] - df['close'].shift()).abs()
    atr_mod = (hl_range + cpc_diff).rolling(window=14).mean()
    
    # Liquidity Measure: Close price divided by the exponential moving average of volume
    ema_volume = df['volume'].ewm(span=50, adjust=False).mean()
    liquidity = df['close'] / ema_volume
    
    # Composite heuristic
    heuristics_matrix = (momentum + atr_mod + liquidity) / 3
    return heuristics_matrix
