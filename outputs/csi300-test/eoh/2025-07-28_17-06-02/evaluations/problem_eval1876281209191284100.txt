def heuristics_v2(df):
    # Calculate exponentially weighted moving averages
    ewma_10 = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30 = df['close'].ewm(span=30, adjust=False).mean()
    ewma_ratio = (ewma_10 / ewma_30) - 1
    
    # Relative momentum indicator
    momentum = df['close'] / df['close'].shift(15) - 1
    
    # Volatility measure
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Aroon Oscillator
    aroon_up = 100 * (df['high'].rolling(window=25).apply(lambda x: (x.argmax() + 1) / 25, raw=True))
    aroon_down = 100 * (df['low'].rolling(window=25).apply(lambda x: (x.argmin() + 1) / 25, raw=True))
    aroon_oscillator = aroon_up - aroon_down
    
    # Composite heuristic
    heuristics_matrix = (ewma_ratio + momentum - volatility + price_range + volume_weighted + aroon_oscillator) / 6
    return heuristics_matrix
