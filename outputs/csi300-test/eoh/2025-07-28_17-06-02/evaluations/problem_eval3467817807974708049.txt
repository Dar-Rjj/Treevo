import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_ratio = (ema_5 / ema_10) - 1
    
    # Modified Relative Strength Index (RSI) with a 7-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=7).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=7).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Standard Deviation of Returns
    std_dev = df['close'].pct_change().rolling(window=21).std()
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + rsi - std_dev + volume_weighted) / 4
    return heuristics_matrix
