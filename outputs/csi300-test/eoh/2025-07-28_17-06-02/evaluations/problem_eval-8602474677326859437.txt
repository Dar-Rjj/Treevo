import pandas as pd

def heuristics_v2(df):
    # On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    obv -= (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Rate of Change (ROC)
    roc = df['close'] / df['close'].shift(5) - 1
    
    # Composite heuristic
    heuristics_matrix = (obv + adl + roc) / 3
    return heuristics_matrix
