import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving average
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Bollinger Bands
    bb_upper = sma_20 + 2 * df['close'].rolling(window=20).std()
    bb_lower = sma_20 - 2 * df['close'].rolling(window=20).std()
    bb_width = (bb_upper - bb_lower) / sma_20
    
    # Chaikin Money Flow (CMF)
    mfv = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    mfv = mfv.fillna(0)
    cmf = (mfv * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (sma_20 + bb_width + cmf + volume_weighted) / 4
    return heuristics_matrix
