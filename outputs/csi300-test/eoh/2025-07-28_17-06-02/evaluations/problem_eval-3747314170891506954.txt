def heuristics_v2(df):
    # Simple Moving Averages
    sma_7 = df['close'].rolling(window=7).mean()
    sma_14 = df['close'].rolling(window=14).mean()
    sma_ratio = (sma_7 / sma_14) - 1
    
    # Modified Relative Strength Index (RSI) with 7-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=7).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=7).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Custom Volatility Measure
    volatility = df['close'].rolling(window=7).std()
    
    # Volume-Adjusted Momentum Indicator
    mom = df['close'] - df['close'].shift(7)
    volume_adjusted_mom = mom * df['volume']
    
    # Composite heuristic
    heuristics_matrix = (sma_ratio + rsi + volatility + volume_adjusted_mom) / 4
    return heuristics_matrix
