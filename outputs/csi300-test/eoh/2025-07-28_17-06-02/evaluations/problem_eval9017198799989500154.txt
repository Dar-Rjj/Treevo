import pandas as pd

def heuristics_v2(df):
    # Momentum signal
    momentum = df['close'] / df['close'].rolling(window=20).mean()
    
    # Aroon Oscillator for trend strength and direction
    aroon_up = 100 * (25 - df['high'].rolling(window=25).apply(lambda x: list(x).index(max(x)), raw=True))
    aroon_down = 100 * (df['low'].rolling(window=25).apply(lambda x: list(x).index(min(x)), raw=True) - 25)
    aroon_oscillator = aroon_up - aroon_down
    
    # Volume-Price Relationship Indicator
    volume_price_indicator = (df['volume'] * (df['close'] - df['close'].shift(1)))
    
    # Combine the factors
    heuristics_matrix = (momentum + aroon_oscillator + volume_price_indicator) / 3
    
    return heuristics_matrix
