import pandas as pd

def heuristics_v2(df):
    # 5-day Rate of Change (ROC) of Closing Price
    roc_5 = df['close'].pct_change(periods=5)
    
    # Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Heuristics matrix combining ROC and ADL
    heuristics_matrix = roc_5 + adl
    return heuristics_matrix
