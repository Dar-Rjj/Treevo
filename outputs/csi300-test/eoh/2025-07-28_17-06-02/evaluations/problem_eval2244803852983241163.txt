import pandas as pd

def heuristics_v2(df):
    # Relative Strength (RS) Ratio
    sma_5 = df['close'].rolling(window=5).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    rs_ratio = sma_5 / sma_30
    
    # Chaikin Money Flow (CMF) with 14-day window
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Average True Range (ATR) with 14-day window
    true_range = df[['high', 'low']].diff(axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Simple Moving Averages Crossover System
    sma_diff = sma_5 - sma_30
    
    # Composite heuristic
    heuristics_matrix = (rs_ratio + cmf + atr + sma_diff) / 4
    return heuristics_matrix
