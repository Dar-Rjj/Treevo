import pandas as pd

def heuristics_v2(df):
    # Kaufman's Adaptive Moving Average (KAMA)
    def kama(data, er, fast=2, slow=30):
        fast_ma = data.ewm(span=fast).mean()
        slow_ma = data.ewm(span=slow).mean()
        sc = (abs(data - data.shift(10)) / 10).ewm(span=er).mean()
        er = (sc / (data.rolling(window=10).std())).ewm(span=er).mean()
        return (er * (fast_ma - slow_ma) + slow_ma)

    er = 10
    kama_10 = kama(df['close'], er)
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Standard Deviation of Returns
    std_returns = df['close'].pct_change().rolling(window=15).std()
    
    # Volume weighted by typical price
    volume_weighted = df['volume'] * typical_price
    
    # Composite heuristic
    heuristics_matrix = (kama_10 + mfi - std_returns + volume_weighted) / 4
    return heuristics_matrix
