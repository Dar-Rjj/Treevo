import pandas as pd

def heuristics_v2(df):
    # Volatility Component using the Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Price Momentum Component using the 20-day Rate of Change
    momentum = df['close'].pct_change(periods=20)
    
    # Trading Activity Indicator based on Average Volume over the Past 10 Days
    trading_activity = df['volume'].rolling(window=10).mean()
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (volatility + momentum + trading_activity) / 3
    
    return heuristics_matrix
