import pandas as pd

def heuristics_v2(df):
    # Rate of Change (ROC) over 10 days
    roc = df['close'].pct_change(periods=10)
    
    # Chaikin Money Flow (CMF) over 20 days
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Heuristics matrix combining ROC and CMF
    heuristics_matrix = roc + cmf
    return heuristics_matrix
