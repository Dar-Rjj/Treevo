import pandas as pd

def heuristics_v2(df):
    # Exponential Moving Averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_ratio = (ema_5 / ema_10) - 1
    
    # Modified Relative Strength Index (RSI) with 20-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Money Flow Index (MFI) with 28-day lookback
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=28).sum() / negative_flow.rolling(window=28).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Volume weighted by typical price
    volume_weighted = df['volume'] * typical_price
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + rsi + mfi + volume_weighted) / 4
    return heuristics_matrix
