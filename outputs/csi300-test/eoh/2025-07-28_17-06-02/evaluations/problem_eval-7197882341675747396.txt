import pandas as pd

def heuristics_v2(df):
    # Calculate On-Balance Volume (OBV)
    obv = (df['volume'] * ((df['close'] > df['close'].shift(1)).astype(int) - (df['close'] < df['close'].shift(1)).astype(int))).cumsum()
    
    # Exponential Moving Average (EMA) with a 10-day span
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Money Flow Index (MFI) with a 14-period lookback
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_ratio = positive_money_flow.rolling(window=14).sum() / (positive_money_flow.rolling(window=14).sum() + negative_money_flow.rolling(window=14).sum())
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Composite heuristic
    heuristics_matrix = (obv + ema_10 + mfi) / 3
    return heuristics_matrix
