import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum as the difference between the current close and the 20-day average close
    momentum = df['close'] - df['close'].rolling(window=20).mean()
    
    # Calculate the volume change as the percentage change in volume from 5 days ago
    volume_change = df['volume'].pct_change(periods=5)
    
    # Calculate the average true range (a measure of volatility) over the last 10 days
    tr1 = abs(df["high"] - df["low"])
    tr2 = abs(df["high"] - df["close"].shift())
    tr3 = abs(df["low"] - df["close"].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    avg_true_range = true_range.rolling(window=10).mean()
    
    # Combine the factors into a heuristics matrix, assigning equal weights
    heuristics_matrix = (momentum + volume_change + avg_true_range) / 3
    
    return heuristics_matrix
