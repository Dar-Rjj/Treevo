import pandas as pd

def heuristics_v2(df):
    # Calculate the 50-day moving average and the ratio of close price to this average
    ma_50 = df['close'].rolling(window=50).mean()
    ma_ratio = df['close'] / ma_50
    
    # Exponential Moving Average (EMA) for RSI with a 7-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(span=7, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=7, adjust=False).mean()
    rs = gain / loss
    rsi_ema = 100 - (100 / (1 + rs))
    
    # Standard deviation of daily returns over a 28-day period
    std_returns = df['close'].pct_change().rolling(window=28).std()
    
    # Volume weighted by the difference between high and low prices
    price_diff_weighted = (df['high'] - df['low']) * df['volume']
    
    # Composite heuristic
    heuristics_matrix = (ma_ratio + rsi_ema + std_returns + price_diff_weighted) / 4
    return heuristics_matrix
