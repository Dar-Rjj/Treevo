import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    sma_ratio = (sma_5 / sma_10) - 1
    
    # Rate of Change (ROC) indicator
    roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Volatility measure
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (sma_ratio * 2 + roc * 3 - volatility + price_range + volume_weighted) / 7
    return heuristics_matrix
