import pandas as pd

def heuristics_v2(df):
    # Calculate relative strength index (RSI) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate trend in trading volume over the last 5 days
    volume_trend = df['volume'].pct_change(periods=5)
    
    # Composite heuristic
    heuristics_matrix = (rsi + volume_trend) / 2
    return heuristics_matrix
