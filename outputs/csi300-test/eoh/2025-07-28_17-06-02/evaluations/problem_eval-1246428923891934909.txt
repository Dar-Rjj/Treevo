import pandas as pd

def heuristics_v2(df):
    # Smoothed Directional Movement Index (DMI) using a 5-day moving average
    dmi = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Momentum Component: Rate of change in closing prices over the last 10 days
    momentum = df['close'].pct_change(periods=10)
    
    # Adjusted Liquidity Indicator: Cumulative volume adjusted by the 20-day average volume
    avg_volume = df['volume'].rolling(window=20).mean()
    liquidity = (df['close'] * (df['volume'] / avg_volume)).cumsum()
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (dmi + momentum + liquidity) / 3
    
    return heuristics_matrix
