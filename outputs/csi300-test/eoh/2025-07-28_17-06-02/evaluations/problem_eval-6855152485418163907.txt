import pandas as pd

def heuristics_v2(df):
    def calculate_squared_return(series, window):
        return (series.pct_change().fillna(0) ** 2).rolling(window=window).sum()
    
    def calculate_absolute_return(series, window):
        return series.pct_change().fillna(0).abs().rolling(window=window).sum()

    squared_return_5 = calculate_squared_return(df['close'], 5)
    abs_return_20 = calculate_absolute_return(df['close'], 20)

    heuristics_matrix = squared_return_5 / abs_return_20
    heuristics_matrix.name = 'heuristic_factor'
    return heuristics_matrix
