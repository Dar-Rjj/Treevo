import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day Moving Average
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate 20-day Average True Range (ATR) for volatility
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr_20 = tr.rolling(window=20).mean()
    
    # Relative Strength Index (RSI) with a 7-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=7).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=7).mean()
    rs = gain / loss
    rsi_7 = 100 - (100 / (1 + rs))
    
    # 5-day Volume Weighted Average Price
    vwap_5 = (df['volume'] * df['close']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Composite heuristic
    heuristics_matrix = (ma_10 + atr_20 + rsi_7 + vwap_5) / 4
    return heuristics_matrix
