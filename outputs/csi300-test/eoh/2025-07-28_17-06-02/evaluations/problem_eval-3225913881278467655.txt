def heuristics_v2(df):
    # Mean Reversion: 5-day change in close price
    mean_reversion = df['close'].pct_change(5)
    
    # Volatility Measure: Bollinger Band width (20-day SMA Â± 2 standard deviations)
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    bollinger_band_width = (sma_20 + 2 * std_20) - (sma_20 - 2 * std_20)
    
    # Accumulation/Distribution Line
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Composite heuristic
    heuristics_matrix = (mean_reversion + bollinger_band_width + adl) / 3
    return heuristics_matrix
