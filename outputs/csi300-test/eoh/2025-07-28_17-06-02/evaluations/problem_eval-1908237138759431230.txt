import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10) * 100
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift(1)).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Combine ROC and OBV into a heuristics matrix
    heuristics_matrix = (roc + obv) / 2
    
    return heuristics_matrix
