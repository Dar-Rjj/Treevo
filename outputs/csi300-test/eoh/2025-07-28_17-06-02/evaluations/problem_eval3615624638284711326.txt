def heuristics_v2(df):
    # Volatility-Adjusted Directional Movement Index with a 20-day window
    dmi = (df['high'] - df['low']) / df['close'].rolling(window=20).std()
    
    # Momentum Component with a 10-day period
    momentum = df['close'].pct_change(periods=10)
    
    # Liquidity Adjustment Factor
    liquidity = (df['volume'] * df['amount']).cumsum() / df['volume'].cumsum()
    
    # Relative Strength Index (RSI) with a 14-day window
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (dmi + momentum + liquidity + rsi) / 4
    
    return heuristics_matrix
