import pandas as pd

def heuristics_v2(df):
    # 5-day Exponential Moving Average (EMA) of Closing Price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # 10-day Simple Moving Average (SMA) of Closing Price
    sma_10 = df['close'].rolling(window=10).mean()
    
    # 30-day Simple Moving Average (SMA) of Closing Price
    sma_30 = df['close'].rolling(window=30).mean()
    
    # Difference between 10-day and 30-day SMAs
    sma_diff = sma_10 - sma_30
    
    # Heuristics matrix combining EMA and SMA difference
    heuristics_matrix = ema_5 + sma_diff
    return heuristics_matrix
