def heuristics_v2(df):
    # Momentum factor: 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volatility Clustering: Bollinger Bands Width with a 20-day window and 2 standard deviations
    sma = df['close'].rolling(window=20).mean()
    std_dev = df['close'].rolling(window=20).std()
    upper_band = sma + 2 * std_dev
    lower_band = sma - 2 * std_dev
    bollinger_width = (upper_band - lower_band) / sma
    
    # Liquidity Measure: Close price divided by the 30-day Exponential Moving Average (EMA) of volume
    ema_volume = df['volume'].ewm(span=30, adjust=False).mean()
    liquidity = df['close'] / ema_volume
    
    # Composite heuristic
    heuristics_matrix = (rsi + bollinger_width + liquidity) / 3
    return heuristics_matrix
