import pandas as pd

def heuristics_v2(df):
    # Calculate weighted moving averages
    wma_5 = df['close'].ewm(span=5, adjust=False).mean()
    wma_20 = df['close'].ewm(span=20, adjust=False).mean()
    wma_ratio = (wma_5 / wma_20) - 1
    
    # Rate of Change (ROC) indicator
    roc = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Average True Range (ATR) for volatility
    tr = df['high'] - df['low']
    atr = tr.rolling(window=20).mean()
    
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    
    # Volume momentum
    volume_momentum = df['volume'] / df['volume'].shift(5) - 1
    
    # Simple Moving Average (SMA)
    sma_100 = df['close'].rolling(window=100).mean()
    sma_signal = (df['close'] - sma_100) / df['close']
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Composite heuristic
    heuristics_matrix = (wma_ratio + roc - atr + price_range + volume_momentum + sma_signal + rsi) / 7
    return heuristics_matrix
