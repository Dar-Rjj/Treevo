def heuristics_v2(df):
    # Calculate simple moving averages
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    sma_ratio = (sma_5 / sma_10) - 1
    
    # Momentum indicator
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Volatility measure
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Volume to price ratio
    volume_to_price = df['volume'] / df['close']
    
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    
    # Composite heuristic
    heuristics_matrix = (2 * sma_ratio + 3 * momentum - 1.5 * volatility + 0.5 * price_range + volume_to_price)
    return heuristics_matrix
