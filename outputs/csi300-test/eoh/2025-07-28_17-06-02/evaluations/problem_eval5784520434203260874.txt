import pandas as pd

def heuristics_v2(df):
    # Exponential Moving Averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_ratio = (ema_5 / ema_10) - 1
    
    # Traditional Relative Strength Index (RSI) with 9-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=9).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=9).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Modified Money Flow Index (MFI) with 9-day window
    typical_price = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=9).sum() / negative_flow.rolling(window=9).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Volume weighted by the median of open, high, low, and close prices
    med_price = (df[['open', 'high', 'low', 'close']].median(axis=1))
    volume_weighted = df['volume'] * med_price
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + rsi + mfi + volume_weighted) / 4
    return heuristics_matrix
