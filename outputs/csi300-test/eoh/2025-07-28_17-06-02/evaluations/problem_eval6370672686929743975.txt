import pandas as pd

def heuristics_v2(df):
    # Calculate RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volatility measure using standard deviation
    vol = df['close'].rolling(window=20).std()
    
    # Liquidity measure using the ratio of volume to ATR
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    liquidity = df['volume'] / atr
    
    # Composite heuristic
    heuristics_matrix = (rsi + vol + liquidity) / 3
    return heuristics_matrix
