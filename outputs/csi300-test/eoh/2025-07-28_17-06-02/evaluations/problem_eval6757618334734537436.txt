def heuristics_v2(df):
    # Volatility-Adjusted Directional Movement Index
    dmi = (df['high'] - df['low']) / df['close'].rolling(window=10).std()
    
    # Momentum Component
    momentum = df['close'].pct_change(periods=20)
    
    # Liquidity Adjustment Factor
    liquidity = (df['volume'] * df['amount']).cumsum() / df['volume'].cumsum()
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (dmi + momentum + liquidity) / 3
    
    return heuristics_matrix
