import pandas as pd

def heuristics_v2(df):
    # Rate of Change (ROC) over 10 days
    roc_10 = df['close'].pct_change(periods=10)
    
    # Stochastic Oscillator with 14-day lookback
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - lowest_low) / (highest_high - lowest_low)
    
    # Modified Accumulation/Distribution Line (ADL)
    adl = ((2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'])) * df['volume']
    adl = adl.cumsum()
    
    # Composite heuristic
    heuristics_matrix = (roc_10 + k_percent + adl) / 3
    return heuristics_matrix
