def heuristics_v2(df):
    # Exponential Moving Averages
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    ema_28 = df['close'].ewm(span=28, adjust=False).mean()
    ema_ratio = (ema_14 / ema_28) - 1
    
    # Traditional Relative Strength Index (RSI) with 21-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=21).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=21).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Modified Money Flow Index (MFI) with 21-day window
    typical_price = (df['high'] + df['low'] + df['open'] + df['close']) / 4
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=21).sum() / negative_flow.rolling(window=21).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Volume weighted by the weighted average of open, high, low, and close prices
    volume_weighted = df['volume'] * (0.2*df['open'] + 0.3*df['high'] + 0.3*df['low'] + 0.2*df['close'])
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + rsi + mfi + volume_weighted) / 4
    return heuristics_matrix
