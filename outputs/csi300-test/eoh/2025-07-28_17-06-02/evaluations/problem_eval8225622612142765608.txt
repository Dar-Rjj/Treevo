import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Modified volatility measure: standard deviation of close prices over the last 14 days
    volatility = df['close'].rolling(window=14).std()
    
    # Percentage change in trading amount
    amount_change = df['amount'].pct_change()
    
    # Composite heuristic
    heuristics_matrix = (roc + volatility + amount_change) / 3
    return heuristics_matrix
