import pandas as pd

def heuristics_v2(df):
    # Calculate MACD and Signal Line
    ema_short = df['close'].ewm(span=12, adjust=False).mean()
    ema_long = df['close'].ewm(span=26, adjust=False).mean()
    macd = ema_short - ema_long
    signal_line = macd.ewm(span=9, adjust=False).mean()
    
    # MACD Signal
    macd_signal = macd - signal_line
    
    # Volatility Indicator
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Liquidity Measure
    true_range = df['high'] - df['low']
    avg_true_range = true_range.rolling(window=14).mean()
    liquidity = df['volume'] / avg_true_range
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (macd_signal + volatility + liquidity) / 3
    
    return heuristics_matrix
