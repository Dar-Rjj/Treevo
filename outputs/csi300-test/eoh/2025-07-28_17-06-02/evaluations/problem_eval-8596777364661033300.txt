import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    ema_ratio = (ema_7 / ema_14) - 1
    
    # Modified Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).ewm(span=21, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=21, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume-Weighted Average True Range (ATR)
    tr = df['high'] - df['low']
    atr = (tr * df['volume']).rolling(window=21).mean() / df['volume'].rolling(window=21).mean()
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + rsi - atr) / 3
    return heuristics_matrix
