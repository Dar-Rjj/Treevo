import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages with adjusted spans
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    ema_7 = df['close'].ewm(span=7, adjust=False).mean()
    ema_ratio = (ema_3 / ema_7) - 1
    
    # Momentum indicator with a shorter lookback
    momentum = df['close'] / df['close'].shift(5) - 1
    
    # Percentage Price Oscillator for trend strength
    ppo = ((df['close'].ewm(span=9, adjust=False).mean()) / (df['close'].ewm(span=26, adjust=False).mean())) - 1
    
    # Volatility measure using average true range
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    
    # Composite heuristic without normalization
    heuristics_matrix = (ema_ratio + momentum + ppo - atr)
    return heuristics_matrix
