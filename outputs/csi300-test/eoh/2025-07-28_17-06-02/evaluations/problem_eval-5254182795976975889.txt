import pandas as pd

def heuristics_v2(df):
    # Calculate MACD line
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    
    # Calculate the difference between high and low prices
    price_diff = df['high'] - df['low']
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the rate of change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Composite heuristic
    heuristics_matrix = (macd_line + price_diff + volume_change + roc) / 4
    return heuristics_matrix
