import pandas as pd

def heuristics_v2(df):
    # Percentage Change in Closing Price
    pct_change = df['close'].pct_change()
    
    # True Range (TR)
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Exponential Moving Average of True Range (EMA-TR) over 5 days
    ema_tr = true_range.ewm(span=5, adjust=False).mean()
    
    # 5-Day Simple Moving Average of Volume
    sma_volume = df['volume'].rolling(window=5).mean()
    
    # Volume Momentum: Ratio of today's volume to the 5-day SMA of volume
    volume_momentum = df['volume'] / sma_volume
    
    # Heuristics matrix combining percentage change, EMA-TR, and volume momentum
    heuristics_matrix = (pct_change + ema_tr) * volume_momentum
    return heuristics_matrix
