import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day Exponential Moving Average (EMA)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate 30-day Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=30).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=30).sum()
    mfi_ratio = positive_money_flow_sum / (positive_money_flow_sum + negative_money_flow_sum)
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Calculate 20-day Standard Deviation (SD) of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    sd_20 = log_returns.rolling(window=20).std()
    
    # Volume ratio: today's volume / 30-day average volume
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_ratio = df['volume'] / avg_volume_30
    
    # Composite heuristic
    heuristics_matrix = (ema_10 + mfi - sd_20 + volume_ratio) / 4
    return heuristics_matrix
