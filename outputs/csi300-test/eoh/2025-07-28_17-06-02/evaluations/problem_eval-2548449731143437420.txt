def heuristics_v2(df):
    # Rate of Change (ROC) over 14 periods
    roc = df['close'].pct_change(periods=14)
    
    # Standard Deviation of closing prices over 14 periods
    std_dev = df['close'].rolling(window=14).std()
    
    # Turnover Rate (volume / amount)
    turnover_rate = df['volume'] / df['amount']
    
    # Composite heuristic
    heuristics_matrix = (roc + std_dev + turnover_rate) / 3
    return heuristics_matrix
