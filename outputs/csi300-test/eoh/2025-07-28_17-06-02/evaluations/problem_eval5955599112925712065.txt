import pandas as pd

def heuristics_v2(df):
    # Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Skewness of Returns
    skew_returns = df['close'].pct_change().rolling(window=21).skew()
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (roc + mfi - skew_returns + volume_weighted) / 4
    return heuristics_matrix
