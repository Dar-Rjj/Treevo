def heuristics_v2(df):
    # Calculate Momentum Indicator (10-day lookback)
    momentum = df['close'].pct_change(periods=10)
    
    # Volume Rate of Change (10-day lookback)
    volume_roc = (df['volume'] - df['volume'].shift(10)) / df['volume'].shift(10) * 100
    
    # Accumulation/Distribution Line
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Composite heuristic
    heuristics_matrix = (momentum + volume_roc + adl) / 3
    return heuristics_matrix
