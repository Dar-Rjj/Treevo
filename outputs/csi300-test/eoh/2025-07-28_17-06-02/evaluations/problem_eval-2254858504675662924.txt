def heuristics_v2(df):
    # Calculate the 5-day and 60-day moving averages of the trading volume
    ma_5_vol = df['volume'].rolling(window=5).mean()
    ma_60_vol = df['volume'].rolling(window=60).mean()
    
    # Compute the ratio of the 5-day moving average volume to the 60-day moving average volume
    vol_ratio = ma_5_vol / ma_60_vol
    
    # Calculate the 10-day rate of change (ROC) of the closing prices
    roc_10 = df['close'].pct_change(periods=10)
    
    # Generate the heuristics matrix combining the volume ratio and ROC
    heuristics_matrix = vol_ratio - roc_10
    
    return heuristics_matrix
