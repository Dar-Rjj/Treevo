import pandas as pd

def heuristics_v2(df):
    # Calculate Bollinger Bands
    middle_band = df['close'].rolling(window=10).mean()
    std_dev = df['close'].rolling(window=10).std()
    upper_band = middle_band + 2 * std_dev
    lower_band = middle_band - 2 * std_dev
    bollinger_ratio = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Modified Relative Strength Index (RSI) with a 7-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=7).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=7).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Coefficient of Variation of Returns
    mean_returns = df['close'].pct_change().rolling(window=30).mean()
    std_returns = df['close'].pct_change().rolling(window=30).std()
    cv = std_returns / mean_returns
    
    # Composite heuristic
    heuristics_matrix = (bollinger_ratio + rsi - cv) / 3
    return heuristics_matrix
