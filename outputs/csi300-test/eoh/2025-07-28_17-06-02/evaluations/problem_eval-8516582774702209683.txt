import pandas as pd

def heuristics_v2(df):
    def compute_heuristic(df):
        avg_high_low = (df['high'].rolling(window=20).mean() + df['low'].rolling(window=20).mean()) / 2
        mv_avg_close = df['close'].rolling(window=20).mean()
        heuristic = avg_high_low - mv_avg_close
        return heuristic
    
    heuristics_matrix = compute_heuristic(df)
    return heuristics_matrix
