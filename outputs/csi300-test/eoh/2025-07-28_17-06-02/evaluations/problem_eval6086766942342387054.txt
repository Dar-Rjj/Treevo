import pandas as pd

def heuristics_v2(df):
    # Momentum signal
    mom_10 = df['close'].pct_change(10)
    
    # Volatility (Average True Range)
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    
    # Market sentiment (Volume to ATR ratio)
    volume_atr_ratio = df['volume'] / atr
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (mom_10 + volume_atr_ratio) / 2
    
    return heuristics_matrix
