import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day moving average of the close price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the cumulative difference between the close price and its 20-day moving average
    cum_diff_ma = (df['close'] - ma_20).cumsum()
    
    # Calculate the Relative Strength Index (RSI) over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (cum_diff_ma + rsi) / 2
    
    return heuristics_matrix
