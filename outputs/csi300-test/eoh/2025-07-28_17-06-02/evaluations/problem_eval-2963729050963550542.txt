import pandas as pd

def heuristics_v2(df):
    # Calculate momentum
    momentum_5 = df['close'] / df['close'].shift(5) - 1
    
    # Calculate volatility
    volatility_14 = df['close'].rolling(window=14).std()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (momentum_5 + volatility_14 + rsi + volume_weighted) / 4
    return heuristics_matrix
