import pandas as pd

def heuristics_v2(df):
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Volatility Component based on Standard Deviation
    volatility = df['close'].rolling(window=20).std()
    
    # Enhanced Liquidity Indicator: Ratio of Current Volume to Average Volume
    avg_volume = df['volume'].rolling(window=50).mean()
    liquidity = df['volume'] / avg_volume
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (rsi + volatility + liquidity) / 3
    
    return heuristics_matrix
