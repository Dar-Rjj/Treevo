import pandas as pd

def heuristics_v2(df):
    # Calculate directional movement
    positive_dm = (df['high'] - df['high'].shift(1)).where(lambda x: x > 0, 0)
    negative_dm = (df['low'].shift(1) - df['low']).where(lambda x: x > 0, 0)
    
    # Calculate average directional movement
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    plus_di = 100 * (positive_dm.rolling(window=14).sum() / atr)
    minus_di = 100 * (negative_dm.rolling(window=14).sum() / atr)
    
    # Calculate money flow index
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(df['close'] > df['close'].shift(1), 0)
    negative_money_flow = raw_money_flow.where(df['close'] < df['close'].shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate realized volatility
    daily_returns = df['close'].pct_change()
    realized_volatility = daily_returns.rolling(window=20).std()
    
    # Composite heuristic
    heuristics_matrix = (plus_di - minus_di + mfi - realized_volatility) / 4
    return heuristics_matrix
