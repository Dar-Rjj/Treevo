import pandas as pd

def heuristics_v2(df):
    # Moving Average Convergence Divergence (MACD)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    
    # Average True Range (ATR) with a 14-day lookback
    tr = pd.DataFrame({'high-low': df['high'] - df['low'],
                       'high-close_prev': abs(df['high'] - df['close'].shift(1)),
                       'low-close_prev': abs(df['low'] - df['close'].shift(1))}).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Chaikin Money Flow (CMF) with a 20-day window
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (macd_line + atr + cmf + volume_weighted) / 4
    return heuristics_matrix
