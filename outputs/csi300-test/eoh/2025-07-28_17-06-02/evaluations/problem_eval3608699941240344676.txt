import pandas as pd

def heuristics_v2(df):
    # Percentage Change in Closing Price
    pct_change = df['close'].pct_change()
    
    # True Range
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Exponential Moving Average of True Range (EMA)
    ema_true_range = true_range.ewm(span=10, adjust=False).mean()
    
    # Heuristics matrix combining percentage change and EMA of true range
    heuristics_matrix = pct_change + ema_true_range
    return heuristics_matrix
