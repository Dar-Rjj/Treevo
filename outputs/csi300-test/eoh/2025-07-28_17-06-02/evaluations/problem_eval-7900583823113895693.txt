import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day and 50-day simple moving averages
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    sma_diff = (sma_20 - sma_50) / sma_50
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi = 100 - (100 / (1 + (positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum())))
    
    # Relative Momentum Index (RMI)
    close_change = df['close'].pct_change()
    up_change = close_change.where(close_change > 0, 0).rolling(window=14).mean()
    down_change = abs(close_change.where(close_change < 0, 0)).rolling(window=14).mean()
    rmi = 100 * (up_change / (up_change + down_change))
    
    # Composite heuristic
    heuristics_matrix = (sma_diff + mfi + rmi) / 3
    return heuristics_matrix
