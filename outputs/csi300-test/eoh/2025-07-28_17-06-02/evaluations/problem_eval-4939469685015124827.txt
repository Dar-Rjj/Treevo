import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_5 = df['close'].pct_change().fillna(0).ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].pct_change().fillna(0).ewm(span=20, adjust=False).mean()
    ema_ratio = ema_5 / ema_20
    
    # Average True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift()).abs()
    tr3 = (df['low'] - df['close'].shift()).abs()
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio - atr)
    heuristics_matrix.name = 'heuristic_factor'
    return heuristics_matrix
