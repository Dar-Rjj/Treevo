import pandas as pd

def heuristics_v2(df):
    # Directional Movement Index (DMI) with Exponential Smoothing
    dmi = (df['high'] - df['low']).ewm(span=10).mean()
    
    # Mean Reversion Component with Exponential Weighted Moving Average
    mean_reversion = df['close'] - df['close'].ewm(span=50).mean()
    
    # Liquidity Indicator based on Exponentially Smoothed Cumulative Volume Adjusted Close
    liquidity = ((df['close'] * df['volume']).cumsum()).ewm(span=20).mean()
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (dmi + mean_reversion + liquidity) / 3
    
    return heuristics_matrix
