import pandas as pd

def heuristics_v2(df):
    # Chande Momentum Oscillator (CMO)
    cmo_window = 14
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=cmo_window).sum()
    loss = (-delta.where(delta < 0, 0)).rolling(window=cmo_window).sum()
    cmo = ((gain - loss) / (gain + loss)) * 100
    
    # Bollinger Bands
    bb_window = 20
    std_dev = df['close'].rolling(window=bb_window).std()
    middle_band = df['close'].rolling(window=bb_window).mean()
    upper_band = middle_band + (std_dev * 2)
    lower_band = middle_band - (std_dev * 2)
    bollinger_percentage = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Price-to-Volume Ratio
    price_to_volume = df['close'] / df['volume']
    
    # Composite heuristic
    heuristics_matrix = (cmo + bollinger_percentage + price_to_volume) / 3
    return heuristics_matrix
