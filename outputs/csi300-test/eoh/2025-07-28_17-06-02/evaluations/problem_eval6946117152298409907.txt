import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_ratio = (ema_10 / ema_30) - 1
    
    # Volatility-adjusted momentum indicator
    daily_returns = df['close'].pct_change()
    momentum = (df['close'] / df['close'].shift(20) - 1) / daily_returns.rolling(window=20).std()
    
    # Volatility measure
    volatility = daily_returns.rolling(window=20).std()
    
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=20).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=20).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + momentum - volatility + price_range + volume_weighted + rsi) / 6
    return heuristics_matrix
