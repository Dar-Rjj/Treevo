import pandas as pd

def heuristics_v2(df):
    # Calculate weighted moving averages
    wma_7 = df['close'].ewm(span=7, adjust=False).mean()
    wma_14 = df['close'].ewm(span=14, adjust=False).mean()
    wma_ratio = (wma_7 / wma_14) - 1
    
    # Rate of Change (ROC) indicator
    roc = (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    
    # Average True Range (ATR) for volatility
    tr = df['high'] - df['low']
    atr = tr.rolling(window=10).mean()
    
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    
    # Volume momentum
    volume_momentum = df['volume'] / df['volume'].shift(3) - 1
    
    # Simple Moving Average (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    sma_signal = (df['close'] - sma_50) / df['close']
    
    # Composite heuristic
    heuristics_matrix = (wma_ratio + roc - atr + price_range + volume_momentum + sma_signal) / 6
    return heuristics_matrix
