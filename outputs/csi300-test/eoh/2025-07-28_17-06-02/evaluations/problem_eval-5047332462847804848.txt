import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc_5 = df['close'].pct_change(periods=5)
    roc_10 = df['close'].pct_change(periods=10)
    
    # Accumulation/Distribution Line (ADL)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    adl = money_flow_volume.cumsum()
    
    # Chaikin Oscillator
    ema_3 = df['volume'] * (df['close'] - df['low']) / (df['high'] - df['low']).ewm(span=3, adjust=False).mean()
    ema_10 = df['volume'] * (df['close'] - df['low']) / (df['high'] - df['low']).ewm(span=10, adjust=False).mean()
    chaikin_oscillator = ema_3 - ema_10
    
    # Composite heuristic
    heuristics_matrix = (roc_5 + roc_10 + adl + chaikin_oscillator) / 4
    return heuristics_matrix
