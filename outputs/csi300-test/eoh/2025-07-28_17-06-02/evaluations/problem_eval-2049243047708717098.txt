def heuristics_v2(df):
    # Simple Moving Averages
    sma_7 = df['close'].rolling(window=7).mean()
    sma_21 = df['close'].rolling(window=21).mean()
    sma_ratio = (sma_7 / sma_21) - 1
    
    # Stochastic Oscillator with 14-day lookback
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_percent = 100 * ((df['close'] - lowest_low) / (highest_high - lowest_low))
    
    # Accumulation/Distribution Line (ADL) adjusted for volume
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (money_flow_multiplier * df['volume']).cumsum()
    
    # Price Change Momentum Factor
    price_change_momentum = df['close'].pct_change(periods=1)
    
    # Composite heuristic
    heuristics_matrix = (sma_ratio + k_percent + adl + price_change_momentum) / 4
    return heuristics_matrix
