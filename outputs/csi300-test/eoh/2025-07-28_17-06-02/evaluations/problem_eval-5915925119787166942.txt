def heuristics_v2(df):
    # Calculate the 50-day and 200-day moving averages of the closing price
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()

    # Compute the difference between the current closing price and the 50-day moving average
    diff_50 = df['close'] - ma_50

    # Compute the difference between the 50-day moving average and the 200-day moving average
    diff_200 = ma_50 - ma_200

    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (diff_50 + diff_200) / 2

    return heuristics_matrix
