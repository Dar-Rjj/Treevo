import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_ratio = (ema_5 / ema_10) - 1
    
    # Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_percent = 100 * ((df['close'] - low_14) / (high_14 - low_14))
    
    # Average Directional Index (ADX)
    tr = df['high'] - df['low']
    up_move = df['high'] - df['close'].shift(1)
    down_move = df['low'] - df['close'].shift(1)
    plus_dm = up_move.where((up_move > down_move) & (up_move > 0), 0)
    minus_dm = down_move.where((down_move > up_move) & (down_move > 0), 0)
    tr_14 = tr.rolling(window=14).sum()
    plus_di = 100 * (plus_dm.rolling(window=14).sum() / tr_14)
    minus_di = 100 * (minus_dm.rolling(window=14).sum() / tr_14)
    dx = 100 * (abs(plus_di - minus_di) / (plus_di + minus_di))
    adx = dx.rolling(window=14).mean()
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + k_percent - adx + volume_weighted) / 4
    return heuristics_matrix
