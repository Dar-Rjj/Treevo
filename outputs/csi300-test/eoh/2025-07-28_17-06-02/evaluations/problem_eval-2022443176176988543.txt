import pandas as pd

def heuristics_v2(df):
    # 14-day Stochastic Oscillator
    lowest_low = df['low'].rolling(window=14).min()
    highest_high = df['high'].rolling(window=14).max()
    k_percent = (df['close'] - lowest_low) / (highest_high - lowest_low) * 100
    
    # 20-day Bollinger Bands %B
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    b_percent = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # 20-day Chaikin Money Flow (CMF)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    cmf = (positive_flow.rolling(window=20).sum() - negative_flow.rolling(window=20).sum()) / df['volume'].rolling(window=20).sum()
    
    # 20-day Volume-weighted close price
    volume_weighted = df['volume'] * df['close'].rolling(window=20).mean()
    
    # Composite heuristic
    heuristics_matrix = (k_percent + b_percent + cmf + volume_weighted) / 4
    return heuristics_matrix
