import pandas as pd

def heuristics_v2(df):
    # On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int)).cumsum()
    
    # Bollinger Bands
    sma_20 = df['close'].rolling(window=20).mean()
    std_20 = df['close'].rolling(window=20).std()
    upper_band = sma_20 + 2 * std_20
    lower_band = sma_20 - 2 * std_20
    bb_ratio = (df['close'] - lower_band) / (upper_band - lower_band)
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi_raw = positive_money_flow.rolling(window=14).sum() / (positive_money_flow.rolling(window=14).sum() + negative_money_flow.rolling(window=14).sum())
    mfi = 100 - (100 / (1 + mfi_raw))
    
    # Composite heuristic
    heuristics_matrix = (obv + bb_ratio + mfi) / 3
    return heuristics_matrix
