import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages with adjusted window sizes
    sma_7 = df['close'].rolling(window=7).mean()
    sma_21 = df['close'].rolling(window=21).mean()
    sma_ratio = (sma_7 / sma_21) - 1
    
    # Momentum indicator with an adjusted lookback period
    momentum = df['close'] / df['close'].shift(10) - 1
    
    # Volatility measure
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Price range
    price_range = (df['high'] - df['low']) / df['close']
    
    # Volume-to-volume-moving-average ratio
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / volume_ma
    
    # Composite heuristic
    heuristics_matrix = (sma_ratio + momentum - volatility + price_range + volume_ratio) / 5
    return heuristics_matrix
