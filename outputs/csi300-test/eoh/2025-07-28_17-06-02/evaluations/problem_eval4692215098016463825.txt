import pandas as pd

def heuristics_v2(df):
    # Calculate 14-day Rate of Change (ROC)
    roc_14 = df['close'].pct_change(periods=14)
    
    # Calculate 20-day Volatility (standard deviation of daily returns)
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Calculate 20-day Average Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Liquidity Indicator: Volume divided by the 20-day average volume
    liquidity_indicator = df['volume'] / avg_volume_20
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (roc_14 + volatility + liquidity_indicator) / 3
    
    return heuristics_matrix
