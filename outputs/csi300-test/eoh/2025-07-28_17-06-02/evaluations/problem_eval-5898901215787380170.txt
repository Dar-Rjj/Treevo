import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day Exponential Moving Average (EMA)
    ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 14-day Stochastic Oscillator
    low_14 = df['low'].rolling(window=14).min()
    high_14 = df['high'].rolling(window=14).max()
    k_percent = 100 * (df['close'] - low_14) / (high_14 - low_14)
    
    # Modified Relative Strength Index (RSI) with a 14-day lookback
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Coefficient of Variation of Returns
    mean_returns = df['close'].pct_change().rolling(window=21).mean()
    std_returns = df['close'].pct_change().rolling(window=21).std()
    cv = std_returns / mean_returns
    
    # Volume weighted by close price
    volume_weighted = df['volume'] * df['close']
    
    # Composite heuristic
    heuristics_matrix = (ema + k_percent + rsi - cv + volume_weighted) / 5
    return heuristics_matrix
