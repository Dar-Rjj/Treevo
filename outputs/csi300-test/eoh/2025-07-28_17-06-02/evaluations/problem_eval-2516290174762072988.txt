def heuristics_v2(df):
    # Volatility-Adjusted Directional Movement Index with adjusted window
    dmi = (df['high'] - df['low']) / df['close'].rolling(window=5).std()
    
    # Momentum Component with a shorter period
    momentum = df['close'].pct_change(periods=10)
    
    # Liquidity Adjustment Factor, now as a ratio for more sensitivity to changes
    liquidity = (df['volume'] / (df['amount'] / df['volume']))
    
    # Combine the factors into a heuristics matrix, assigning equal weights
    heuristics_matrix = (dmi + momentum + liquidity) / 3
    
    return heuristics_matrix
