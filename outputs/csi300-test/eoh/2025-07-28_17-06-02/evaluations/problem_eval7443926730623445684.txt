import pandas as pd

def heuristics_v2(df):
    # Rate of Change (ROC) of Closing Price
    roc_close = df['close'].pct_change(periods=5)
    
    # On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Heuristics matrix combining ROC and OBV
    heuristics_matrix = roc_close + obv
    return heuristics_matrix
