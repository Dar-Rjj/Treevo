import pandas as pd

def heuristics_v2(df):
    # Average Price over 10 days
    avg_price = df['close'].rolling(window=10).mean()
    
    # Ratio of Closing Price to Average Price
    price_ratio = df['close'] / avg_price
    
    # Calculate RSI over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Heuristics matrix combining price ratio and RSI
    heuristics_matrix = price_ratio * rsi
    return heuristics_matrix
