def heuristics_v2(df):
    # Momentum factor: 14-day percentage change in close price
    momentum = df['close'].pct_change(14)
    
    # Volatility Clustering: Average True Range (ATR) with a 14-day window
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Liquidity Measure: Close price divided by the moving average of volume
    ma_volume = df['volume'].rolling(window=50).mean()
    liquidity = df['close'] / ma_volume
    
    # Composite heuristic
    heuristics_matrix = (momentum + atr + liquidity) / 3
    return heuristics_matrix
