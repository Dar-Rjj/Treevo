import pandas as pd

def heuristics_v2(df):
    # Calculate exponential moving averages
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    ema_ratio = (ema_10 / ema_30) - 1
    
    # Rate of Change (ROC) indicator
    roc = df['close'] / df['close'].shift(20) - 1
    
    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(), 0)
    mfi = positive_money_flow.rolling(window=14).sum() / (positive_money_flow.rolling(window=14).sum() + negative_money_flow.rolling(window=14).sum())
    
    # Standard deviation of trading volume
    volume_std = df['volume'].rolling(window=20).std()
    
    # Composite heuristic
    heuristics_matrix = (ema_ratio + roc + mfi - volume_std) / 4
    return heuristics_matrix
