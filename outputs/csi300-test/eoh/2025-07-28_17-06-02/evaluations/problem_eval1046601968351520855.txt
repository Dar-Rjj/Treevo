import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change over 14 days for the closing price
    roc = df['close'].pct_change(periods=14)
    
    # Calculate the average daily trading volume over 20 days
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate the Relative Strength Index (RSI) over 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (roc + avg_volume + rsi) / 3
    
    return heuristics_matrix
