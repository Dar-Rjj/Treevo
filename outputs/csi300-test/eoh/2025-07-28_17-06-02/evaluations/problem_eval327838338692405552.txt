import pandas as pd

def heuristics_v2(df):
    # Percentage Price Oscillator (PPO)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    ppo_line = ((ema_12 - ema_26) / ema_26) * 100
    
    # Modified Average True Range (ATR) as a measure of volatility
    true_range = df[['high', 'low']].apply(lambda x: x.max() - x.min(), axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Chaikin Money Flow (CMF)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow_multiplier = (typical_price - df['low']) - (df['high'] - typical_price) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Composite heuristic
    heuristics_matrix = (ppo_line + atr + cmf) / 3
    return heuristics_matrix
