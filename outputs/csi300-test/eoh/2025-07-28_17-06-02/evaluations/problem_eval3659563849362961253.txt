import pandas as pd

def heuristics_v2(df):
    # Directional Movement Index (DMI)
    dmi = df['high'] - df['low']
    
    # Mean Reversion Component
    mean_reversion = df['close'] - df['close'].rolling(window=50).mean()
    
    # Liquidity Indicator based on Cumulative Volume Adjusted Close
    liquidity = (df['close'] * df['volume']).cumsum()
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (dmi + mean_reversion + liquidity) / 3
    
    return heuristics_matrix
