def heuristics_v2(df):
    # Calculate Exponential Moving Average (EMA) of the close price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    close_ema_ratio = df['close'] / ema_close
    
    # Percentage change in volume compared to its 50-day average
    avg_volume_50 = df['volume'].rolling(window=50).mean()
    volume_change = (df['volume'] - avg_volume_50) / avg_volume_50 * 100
    
    # Ratio of positive to negative trading days over the past 21 days
    pos_days = (df['close'] > df['open']).rolling(window=21).sum()
    neg_days = (df['close'] < df['open']).rolling(window=21).sum()
    day_ratio = (pos_days + 1) / (neg_days + 1)
    
    # Composite heuristic
    heuristics_matrix = (close_ema_ratio + volume_change + day_ratio) / 3
    return heuristics_matrix
