import pandas as pd

def heuristics_v2(df):
    # Calculate Aroon Oscillator
    aroon_up = df['close'].rolling(window=14).apply(lambda x: (x.argmax() + 1) / 14 * 100, raw=True)
    aroon_down = df['close'].rolling(window=14).apply(lambda x: (x.argmin() + 1) / 14 * 100, raw=True)
    aroon_oscillator = aroon_up - aroon_down
    
    # Calculate Chaikin Money Flow
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Percentage Price Oscillator
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    ppo = (ema_12 - ema_26) / ema_26 * 100
    
    # Composite heuristic
    heuristics_matrix = (aroon_oscillator + cmf + ppo) / 3
    return heuristics_matrix
