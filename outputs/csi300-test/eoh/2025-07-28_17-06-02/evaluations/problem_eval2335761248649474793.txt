import pandas as pd

def heuristics_v2(df):
    # Momentum Signal: Rate of Change in Closing Prices
    roc = df['close'].pct_change(periods=14)
    
    # Volatility Measure: Standard Deviation of Price Changes
    volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Market Sentiment Indicator: Ratio of Positive to Total Daily Returns
    positive_returns = (df['close'] > df['close'].shift(1)).astype(int)
    total_days = len(df)
    market_sentiment = positive_returns.cumsum() / total_days
    
    # Combine the factors into a heuristics matrix, assigning equal weights for simplicity
    heuristics_matrix = (roc + volatility + market_sentiment) / 3
    
    return heuristics_matrix
