import pandas as pd

def heuristics_v2(df):
    # Momentum of closing prices
    momentum = df['close'] / df['close'].shift(60) - 1
    
    # Price-to-Volume Ratio
    price_to_volume_ratio = df['close'] / df['volume']
    
    # Distance from 200-day Moving Average
    ma_200 = df['close'].rolling(window=200).mean()
    distance_from_ma_200 = (df['close'] - ma_200) / ma_200
    
    # Composite heuristic
    heuristics_matrix = (momentum + price_to_volume_ratio + distance_from_ma_200) / 3
    return heuristics_matrix
