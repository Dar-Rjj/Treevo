def heuristics_v2(df):
    df['close_change'] = df['close'].pct_change()
    df['weighted_volume'] = df['volume'] * df['close_change'].abs()
    df['cumulative_impact'] = df['weighted_volume'].rolling(window=10).sum()
    heuristics_matrix = df['cumulative_impact'].dropna()
    return heuristics_matrix
