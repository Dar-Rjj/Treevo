def heuristics_v2(df):
    df['RollingHigh'] = df['high'].rolling(window=10).max()
    df['RollingLow'] = df['low'].rolling(window=10).min()
    df['AverageVolume'] = df['volume'].rolling(window=10).mean()
    heuristics_matrix = (df['RollingHigh'] - df['RollingLow']) / df['AverageVolume']
    return heuristics_matrix
