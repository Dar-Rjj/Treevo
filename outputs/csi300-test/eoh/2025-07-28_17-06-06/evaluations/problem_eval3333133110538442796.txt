def heuristics_v2(df):
    # Calculate the 5-day price range (high - low)
    price_range = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate the 10-day cumulative volume
    cum_volume = df['volume'].rolling(window=10).sum()
    
    # Create a heuristic score by combining the 5-day price range and 10-day cumulative volume
    heuristics_matrix = (price_range + cum_volume).dropna()
    
    return heuristics_matrix
