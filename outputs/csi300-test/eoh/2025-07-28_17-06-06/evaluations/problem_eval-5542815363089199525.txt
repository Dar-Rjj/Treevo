def heuristics_v2(df):
    # Calculate the 21-day Exponential Moving Average (EMA) of closing prices
    ema = df['close'].ewm(span=21, adjust=False).mean()
    
    # Calculate the 7-day Standard Deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev = daily_returns.rolling(window=7).std()
    
    # Create a heuristic score by combining EMA and the standard deviation
    heuristics_matrix = (ema / std_dev).dropna()
    
    return heuristics_matrix
