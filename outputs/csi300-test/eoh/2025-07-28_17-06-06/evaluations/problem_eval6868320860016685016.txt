def heuristics_v2(df):
    df['5d_high'] = df['high'].rolling(window=5).max()
    df['5d_low'] = df['low'].rolling(window=5).min()
    df['volume_5d_avg'] = df['volume'].rolling(window=5).mean()
    df['weight'] = df['volume'] / df['volume_5d_avg']
    df['rel_dist_to_5d_max'] = (df['5d_high'] - df['close']) / (df['5d_high'] - df['5d_low'])
    df['rel_dist_to_5d_min'] = (df['close'] - df['5d_low']) / (df['5d_high'] - df['5d_low'])
    df['weighted_avg_rel_dist'] = (df['rel_dist_to_5d_max']*df['weight'] + df['rel_dist_to_5d_min']*df['weight']) / 2
    heuristics_matrix = df['weighted_avg_rel_dist'].dropna()
    return heuristics_matrix
