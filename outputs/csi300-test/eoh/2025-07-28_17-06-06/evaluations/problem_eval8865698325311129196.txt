def heuristics_v2(df):
    df['10d_mean_close'] = df['close'].rolling(window=10).mean()
    df['abs_diff_to_10d_mean'] = (df['close'] - df['10d_mean_close']).abs()
    df['weighted_volume'] = df['volume'] * df['abs_diff_to_10d_mean']
    df['weighted_amount'] = df['amount'] * df['abs_diff_to_10d_mean']
    df['ewma_vol'] = df['weighted_volume'].ewm(span=20).mean()
    df['ewma_amt'] = df['weighted_amount'].ewm(span=20).mean()
    df['ewma_avg'] = (df['ewma_vol'] + df['ewma_amt']) / 2
    heuristics_matrix = df['ewma_avg'].dropna()
    return heuristics_matrix
