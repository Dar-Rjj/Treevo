def heuristics_v2(df):
    df['5d_high'] = df['high'].rolling(window=5).max()
    df['5d_low'] = df['low'].rolling(window=5).min()
    df['volatility_factor'] = (df['close'] - df[['5d_high', '5d_low']].mean(axis=1)).abs() / (df['5d_high'] - df['5d_low'])
    df['extreme_movement'] = df['volatility_factor'].apply(lambda x: 1 if x > 0.5 else 0)
    heuristics_matrix = df['extreme_movement'].dropna()
    return heuristics_matrix
