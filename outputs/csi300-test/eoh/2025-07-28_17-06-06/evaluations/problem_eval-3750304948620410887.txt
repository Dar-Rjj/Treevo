import pandas as pd
import numpy as np

def heuristics_v2(df):
    short_ema_high = df['high'].ewm(span=10, adjust=False).mean()
    long_ema_high = df['high'].ewm(span=30, adjust=False).mean()
    short_ema_low = df['low'].ewm(span=10, adjust=False).mean()
    long_ema_low = df['low'].ewm(span=30, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    crossover_high = short_ema_high - long_ema_high
    crossover_low = short_ema_low - long_ema_low
    heuristics_matrix = (crossover_high - crossover_low) - ema_volume
    return heuristics_matrix
