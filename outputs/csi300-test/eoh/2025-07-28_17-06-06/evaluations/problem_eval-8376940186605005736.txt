import pandas as pd

def heuristics_v2(df):
    df['VolumeCloseRatio'] = df['volume'] / df['close']
    short_ema = df['VolumeCloseRatio'].ewm(span=10, adjust=False).mean()
    long_ema = df['VolumeCloseRatio'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = short_ema - long_ema
    return heuristics_matrix
