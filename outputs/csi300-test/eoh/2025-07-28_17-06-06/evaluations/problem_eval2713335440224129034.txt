def heuristics_v2(df):
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 14-day Moving Average (MA) of closing prices
    ma = df['close'].rolling(window=14).mean()

    # Create a heuristic score by combining RSI and MA
    heuristics_matrix = (rsi / ma).dropna()
    
    return heuristics_matrix
