def heuristics_v2(df):
    df['10d_high'] = df['high'].rolling(window=10).max()
    df['10d_low'] = df['low'].rolling(window=10).min()
    df['rel_dist_to_10d_max'] = (df['10d_high'] - df['close']) / (df['10d_high'] - df['10d_low'])
    df['rel_dist_to_10d_min'] = (df['close'] - df['10d_low']) / (df['10d_high'] - df['10d_low'])
    df['geo_mean_rel_dist'] = (df['rel_dist_to_10d_max'] * df['rel_dist_to_10d_min']) ** 0.5
    heuristics_matrix = df['geo_mean_rel_dist'].dropna()
    return heuristics_matrix
