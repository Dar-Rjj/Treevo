import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day average true range (ATR)
    tr = df['high'] - df['low']
    atr_14 = tr.rolling(window=14).mean()
    
    # Calculate the 14-day percentage change in the closing price
    close_pct_change = df['close'].pct_change(14)
    
    # Calculate the momentum of the price change over 14 days
    momentum = df['close'] - df['close'].shift(14)
    
    # Combine the ATR, percentage change in closing price, and momentum
    heuristics_matrix = (momentum * atr_14) * close_pct_change
    
    return heuristics_matrix
