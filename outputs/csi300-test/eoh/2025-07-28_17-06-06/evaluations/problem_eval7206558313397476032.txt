import pandas as pd

def heuristics_v2(df):
    ma_10_close = df['close'].rolling(window=10).mean()
    std_20_close = df['close'].rolling(window=20).std()
    ma_5_volume = df['volume'].rolling(window=5).mean()
    heuristics_matrix = (ma_10_close / std_20_close) * ma_5_volume
    return heuristics_matrix
