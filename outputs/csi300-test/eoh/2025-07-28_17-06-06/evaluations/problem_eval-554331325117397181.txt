def heuristics_v2(df):
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 14-day Average True Range (ATR)
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=14).mean()
    
    # Calculate the 7-day moving average of the 14-day ATR
    atr_ma7 = atr.rolling(window=7).mean()
    
    # Create a heuristic score by combining RSI and the difference between ATR and its 7-day MA
    heuristics_matrix = (rsi * (atr - atr_ma7)).dropna()
    
    return heuristics_matrix
