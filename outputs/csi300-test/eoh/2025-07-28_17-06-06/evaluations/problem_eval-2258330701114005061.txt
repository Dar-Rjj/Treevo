def heuristics_v2(df):
    # Calculate the 21-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=21).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=21).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 7-day Average True Range (ATR)
    tr = df[['high', 'low']].diff().abs().max(axis=1)
    atr = tr.rolling(window=7).mean()

    # Create a heuristic score by combining RSI and ATR
    heuristics_matrix = (rsi / atr).dropna()
    
    return heuristics_matrix
