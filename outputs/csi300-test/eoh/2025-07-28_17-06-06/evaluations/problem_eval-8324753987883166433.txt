import pandas as pd

def heuristics_v2(df):
    # Calculate the 60-day average volume
    avg_volume_60 = df['volume'].rolling(window=60).mean()
    
    # Calculate the volume ratio
    volume_ratio = df['volume'] / avg_volume_60
    
    # Calculate the change in closing price from the previous day
    close_change = df['close'].pct_change()
    
    # Adjust the volume ratio by the change in closing price
    heuristics_matrix = (volume_ratio * close_change).dropna()
    
    return heuristics_matrix
