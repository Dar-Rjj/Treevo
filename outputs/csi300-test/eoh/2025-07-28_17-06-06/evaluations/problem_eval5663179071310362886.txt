def heuristics_v2(df):
    # Calculate the 5-day Rate of Change (ROC) of closing prices
    roc_close = df['close'].pct_change(periods=5)
    
    # Calculate the 30-day cumulative sum of trading volume
    cumsum_volume = df['volume'].rolling(window=30).sum()
    
    # Create a heuristic score by combining ROC of closing prices and cumulative sum of volume
    heuristics_matrix = (roc_close * cumsum_volume).dropna()
    
    return heuristics_matrix
