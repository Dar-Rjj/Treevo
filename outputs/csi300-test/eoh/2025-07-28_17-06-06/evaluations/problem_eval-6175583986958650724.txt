def heuristics_v2(df):
    high_rolling_max = df['high'].rolling(window=20).max()
    low_rolling_min = df['low'].rolling(window=20).min()
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    daily_return = df['close'] / df['open']
    avg_daily_return = daily_return.rolling(window=20).mean()
    return_ratio = daily_return / avg_daily_return
    heuristics_matrix = (high_rolling_max - low_rolling_min) * volume_ratio * return_ratio
    return heuristics_matrix
