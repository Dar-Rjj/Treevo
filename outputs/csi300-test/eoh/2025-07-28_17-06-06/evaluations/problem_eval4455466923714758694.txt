def heuristics_v2(df):
    # Calculate the 5-day Moving Average (MA) of low prices
    ma_low = df['low'].rolling(window=5).mean()
    
    # Calculate the 30-day Exponential Moving Average (EMA) of volume
    ema_volume = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Create a heuristic score by combining MA of low prices and EMA of volume
    heuristics_matrix = (ma_low * ema_volume).dropna()
    
    return heuristics_matrix
