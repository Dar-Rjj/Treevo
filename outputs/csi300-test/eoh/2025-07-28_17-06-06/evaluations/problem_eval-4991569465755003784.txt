def heuristics_v2(df):
    df['10d_mean_close'] = df['close'].rolling(window=10).mean()
    df['10d_std_close'] = df['close'].rolling(window=10).std()
    df['abs_diff_to_10d_mean'] = (df['close'] - df['10d_mean_close']).abs()
    df['norm_diff'] = df['abs_diff_to_10d_mean'] / df['10d_std_close']
    heuristics_matrix = df['norm_diff'].dropna()
    return heuristics_matrix
