def heuristics_v2(df):
    df['10d_mean_close'] = df['close'].rolling(window=10).mean()
    df['momentum'] = (df['close'] - df['10d_mean_close']) / df['10d_mean_close']
    df['10d_mean_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['10d_mean_volume']
    df['adjusted_momentum'] = df['momentum'] * df['volume_ratio']
    heuristics_matrix = df['adjusted_momentum'].dropna()
    return heuristics_matrix
