def heuristics_v2(df):
    df['20d_mean_close'] = df['close'].rolling(window=20).mean()
    df['sqr_dist_to_20d_mean'] = (df['close'] - df['20d_mean_close'])**2
    total_weight = df['sqr_dist_to_20d_mean'].rolling(window=15).sum()
    df['weighted_volume'] = df['volume'] * df['sqr_dist_to_20d_mean']
    df['weighted_amount'] = df['amount'] * df['sqr_dist_to_20d_mean']
    df['weighted_avg'] = (df['weighted_volume'].rolling(window=15).sum() + df['weighted_amount'].rolling(window=15).sum()) / total_weight
    heuristics_matrix = df['weighted_avg'].dropna()
    return heuristics_matrix
