import pandas as pd

def heuristics_v2(df):
    # Calculate the cumulative daily return and volume change over a 30-day window
    df['cumulative_return'] = (df['close'] / df['close'].shift(30)).fillna(1)
    df['volume_change'] = df['volume'] - df['volume'].shift(30)
    
    # Define a factor that combines the cumulative return and volume change
    df['combined_factor'] = df['cumulative_return'] * df['volume_change']
    
    # Create the heuristics matrix based on the combined factor
    heuristics_matrix = df['combined_factor']
    
    return heuristics_matrix
