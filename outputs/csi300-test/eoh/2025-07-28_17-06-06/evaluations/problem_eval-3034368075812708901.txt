def heuristics_v2(df):
    df['10d_mean_close'] = df['close'].rolling(window=10).mean()
    df['abs_dist_to_10d_mean'] = abs(df['close'] - df['10d_mean_close'])
    df['exp_weight'] = df['abs_dist_to_10d_mean'].ewm(span=20, adjust=False).mean()
    df['weighted_volume'] = df['volume'] * df['exp_weight']
    df['weighted_amount'] = df['amount'] * df['exp_weight']
    df['exp_weighted_avg'] = (df['weighted_volume'].ewm(span=20, adjust=False).mean() + df['weighted_amount'].ewm(span=20, adjust=False).mean()) / df['exp_weight']
    heuristics_matrix = df['exp_weighted_avg'].dropna()
    return heuristics_matrix
