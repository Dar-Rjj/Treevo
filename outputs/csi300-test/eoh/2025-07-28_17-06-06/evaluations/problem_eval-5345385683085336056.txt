def heuristics_v2(df):
    # Calculate the 14-day Exponentially Weighted Moving Average (EWMA) of closing prices
    ewma = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 5-day Volatility (standard deviation of daily returns)
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=5).std()
    
    # Create a heuristic score by combining EWMA and Volatility
    heuristics_matrix = (ewma / volatility).dropna()
    
    return heuristics_matrix
