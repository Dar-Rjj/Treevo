def heuristics_v2(df):
    # Calculate the 10-day Exponential Moving Average (EMA) of closing prices
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 5-day mean of volume
    mean_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the squared difference between the volume and its 5-day mean
    sq_diff_volume = (df['volume'] - mean_volume) ** 2
    
    # Calculate the 30-day cumulative sum of the squared differences
    cum_sq_diff_volume = sq_diff_volume.rolling(window=30).sum()
    
    # Create a heuristic score by combining EMA of closing prices and the 30-day cumulative sum of squared differences
    heuristics_matrix = (ema_close * cum_sq_diff_volume).dropna()
    
    return heuristics_matrix
