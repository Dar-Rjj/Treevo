import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day and 100-day moving averages for the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    ma_100 = df['close'].rolling(window=100).mean()
    
    # Calculate the standard deviation of the closing price over a 20-day period
    std_20 = df['close'].rolling(window=20).std()
    
    # Create a signal based on the difference between the two moving averages
    signal = ma_20 - ma_100
    
    # Create an additional feature: ratio of close price to 20-day moving average
    price_to_ma_ratio = df['close'] / ma_20
    
    # Create a heuristic score by combining the signal and the price-to-MA ratio, then dividing by the standard deviation
    heuristics_matrix = ((signal + price_to_ma_ratio) / std_20).dropna()
    
    return heuristics_matrix
