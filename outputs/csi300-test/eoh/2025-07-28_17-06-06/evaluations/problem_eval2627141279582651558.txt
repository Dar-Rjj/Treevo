def heuristics_v2(df):
    # Calculate daily price change
    df['price_change'] = df['close'].diff()
    
    # Sum positive price changes over the last 10 days
    pos_sum = df['price_change'].apply(lambda x: x if x > 0 else 0).rolling(window=10).sum()
    
    # Sum absolute values of negative price changes over the last 10 days
    neg_sum = df['price_change'].apply(lambda x: abs(x) if x < 0 else 0).rolling(window=10).sum()
    
    # Calculate the heuristic score
    df['heuristic_score'] = pos_sum / neg_sum
    heuristics_matrix = df['heuristic_score'].dropna()
    
    return heuristics_matrix
