import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day and 200-day moving averages for the closing price
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the standard deviation of the closing price over a 50-day period
    std_50 = df['close'].rolling(window=50).std()
    
    # Create a signal based on the difference between the two moving averages
    signal = ma_50 - ma_200
    
    # Create a heuristic score by dividing the signal by the standard deviation
    heuristics_matrix = (signal / std_50).dropna()
    
    return heuristics_matrix
