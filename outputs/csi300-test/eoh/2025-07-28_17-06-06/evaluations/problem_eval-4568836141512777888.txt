def heuristics_v2(df):
    df['RollingAvgClose'] = df['close'].rolling(window=10).mean()
    df['RollingAvgLow'] = df['low'].rolling(window=5).mean()
    df['RollingStdClose'] = df['close'].rolling(window=20).std()
    heuristics_matrix = (df['RollingAvgClose'] - df['RollingAvgLow']) / df['RollingStdClose']
    return heuristics_matrix
