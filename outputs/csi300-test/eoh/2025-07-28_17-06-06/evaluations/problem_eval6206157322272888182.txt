def heuristics_v2(df):
    # Calculate 10-day momentum of the close price
    df['close_momentum'] = df['close'].pct_change(periods=10)
    
    # Calculate the ratio of the current day's trading amount to the 10-day average trading amount
    df['amount_ratio'] = df['amount'] / df['amount'].rolling(window=10).mean()
    
    # Create the heuristics matrix
    heuristics_matrix = df['close_momentum'] * df['amount_ratio']
    
    return heuristics_matrix
