def heuristics_v2(df):
    # Calculate the 14-day Volume-Weighted Moving Average (VWMA)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Calculate the 14-day Standard Deviation of Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    std_dev = df['daily_returns'].rolling(window=14).std()
    
    # Create a heuristic score by combining VWMA and Standard Deviation
    heuristics_matrix = (df['vwap'] / std_dev).dropna()
    
    return heuristics_matrix
