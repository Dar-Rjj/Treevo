def heuristics_v2(df):
    df['5d_high'] = df['high'].rolling(window=5).max()
    df['5d_low'] = df['low'].rolling(window=5).min()
    df['10d_high'] = df['high'].rolling(window=10).max()
    df['10d_low'] = df['low'].rolling(window=10).min()
    df['weight'] = (df['5d_high'] - df['5d_low']) / (df['10d_high'] - df['10d_low'])
    df['rel_dist_to_5d_max'] = (df['5d_high'] - df['close']) / (df['5d_high'] - df['5d_low'])
    df['rel_dist_to_5d_min'] = (df['close'] - df['5d_low']) / (df['5d_high'] - df['5d_low'])
    df['weighted_avg_rel_dist'] = (df['rel_dist_to_5d_max'] * df['weight'] + df['rel_dist_to_5d_min'] * (1 - df['weight']))
    heuristics_matrix = df['weighted_avg_rel_dist'].dropna()
    return heuristics_matrix
