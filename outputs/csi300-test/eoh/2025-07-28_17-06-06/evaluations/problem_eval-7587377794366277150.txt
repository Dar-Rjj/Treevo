def heuristics_v2(df):
    # Calculate the 21-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=21).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=21).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the 21-day Moving Average (MA) of closing prices
    ma = df['close'].rolling(window=21).mean()

    # Calculate the 7-day Rate of Change (ROC) in closing prices
    roc = df['close'].pct_change(7)

    # Create a heuristic score by combining RSI, MA, and ROC
    heuristics_matrix = ((rsi / ma) * roc).dropna()
    
    return heuristics_matrix
