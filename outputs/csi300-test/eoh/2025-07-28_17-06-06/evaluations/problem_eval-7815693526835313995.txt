import pandas as pd

def heuristics_v2(df):
    def macd_series(price, fast=12, slow=26, signal=9):
        exp1 = price.ewm(span=fast, adjust=False).mean()
        exp2 = price.ewm(span=slow, adjust=False).mean()
        macd = exp1 - exp2
        signal_line = macd.ewm(span=signal, adjust=False).mean()
        return macd - signal_line

    def rsi_series(price, periods=14):
        delta = price.diff(1)
        gain = delta.where(delta > 0, 0)
        loss = -delta.where(delta < 0, 0)
        avg_gain = gain.rolling(window=periods, min_periods=1).mean()
        avg_loss = loss.rolling(window=periods, min_periods=1).mean()
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))

    macd = macd_series(df['close'])
    rsi = rsi_series(df['close'])

    heuristics_matrix = 0.5 * macd + 0.5 * rsi
    return heuristics_matrix
