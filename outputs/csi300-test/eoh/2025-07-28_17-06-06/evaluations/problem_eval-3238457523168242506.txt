def heuristics_v2(df):
    # Calculate the 20-day price change
    df['price_change_20d'] = df['close'].pct_change(periods=20).fillna(0)
    
    # Calculate the average volume over the same 20-day period
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean().fillna(0)
    
    # Adjust the price change by the average volume
    df['adjusted_price_change'] = df['price_change_20d'] / (df['avg_volume_20d'] + 1)  # +1 to avoid division by zero
    
    # Create the heuristics matrix
    heuristics_matrix = df['adjusted_price_change']
    
    return heuristics_matrix
