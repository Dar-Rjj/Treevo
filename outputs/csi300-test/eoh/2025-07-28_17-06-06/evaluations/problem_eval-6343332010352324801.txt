def heuristics_v2(df):
    df['5d_mean_close'] = df['close'].rolling(window=5).mean()
    df['std_dev_close'] = df['close'].rolling(window=5).std()
    total_weight = df['std_dev_close'].rolling(window=10).sum()
    df['weighted_high'] = df['high'] * df['std_dev_close']
    df['weighted_low'] = df['low'] * df['std_dev_close']
    df['weighted_avg'] = (df['weighted_high'].rolling(window=10).sum() + df['weighted_low'].rolling(window=10).sum()) / total_weight
    heuristics_matrix = df['weighted_avg'].dropna()
    return heuristics_matrix
