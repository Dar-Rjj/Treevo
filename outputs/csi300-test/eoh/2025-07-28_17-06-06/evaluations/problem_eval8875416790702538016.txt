import pandas as pd

def heuristics_v2(df):
    # Calculate the highest high and lowest low over the last 20 days
    high_20 = df['high'].rolling(window=20).max()
    low_20 = df['low'].rolling(window=20).min()
    
    # Calculate the ratio of the highest high to the lowest low
    hl_ratio = high_20 / low_20
    
    # Calculate the rate of change in volume over the last 20 days
    volume_roc = df['volume'].pct_change(20)
    
    # Calculate the RSI over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the HL ratio, volume ROC, and RSI
    heuristics_matrix = (hl_ratio * volume_roc) * rsi
    
    return heuristics_matrix
