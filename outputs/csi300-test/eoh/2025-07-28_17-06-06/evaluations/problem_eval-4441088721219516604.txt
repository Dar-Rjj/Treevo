import pandas as pd

def heuristics_v2(df):
    # Compute the daily return
    daily_return = df['close'].pct_change()
    # Compute the difference in volume
    volume_diff = df['volume'].diff()
    # Heuristic factor: daily return multiplied by the change in volume
    heuristics_matrix = daily_return * volume_diff
    return heuristics_matrix
