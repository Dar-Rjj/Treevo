def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['ema_daily_return_30d'] = df['daily_return'].ewm(span=30, adjust=False).mean()
    df['std_volume_30d'] = df['volume'].rolling(window=30).std()
    df['factor_value'] = df['ema_daily_return_30d'] * df['std_volume_30d']
    heuristics_matrix = df['factor_value'].dropna()
    return heuristics_matrix
