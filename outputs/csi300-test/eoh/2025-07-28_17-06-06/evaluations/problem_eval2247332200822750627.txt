import pandas as pd

def heuristics_v2(df):
    # Calculate 14-day Rate of Change (ROC) for the closing price
    roc_close = df['close'].pct_change(periods=14)
    
    # Calculate 14-day Rate of Change (ROC) for the volume
    roc_volume = df['volume'].pct_change(periods=14)
    
    # Function to calculate RSI
    def rsi(series, n=14):
        delta = series.diff(1)
        up_days = delta.copy()
        up_days[delta<=0] = 0.0
        down_days = abs(delta.copy())
        down_days[delta>0] = 0.0
        RS_up = up_days.rolling(n).mean()
        RS_down = down_days.rolling(n).mean()
        RSI = 100 - 100 / (1 + RS_up / RS_down)
        return RSI
    
    # Calculate 14-day RSI for the closing price
    rsi_close = rsi(df['close'])
    
    # Create a heuristics matrix by combining ROC and RSI
    heuristics_matrix = (roc_close + roc_volume) * rsi_close
    return heuristics_matrix
