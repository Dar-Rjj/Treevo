def heuristics_v2(df):
    df['5d_high'] = df['high'].rolling(window=5).max()
    df['5d_low'] = df['low'].rolling(window=5).min()
    df['rel_dist_to_5d_high'] = (df['5d_high'] - df['close']) / (df['5d_high'] - df['5d_low'])
    df['rel_dist_to_5d_low'] = (df['close'] - df['5d_low']) / (df['5d_high'] - df['5d_low'])
    df['ema_rel_dist_high'] = df['rel_dist_to_5d_high'].ewm(span=5, adjust=False).mean()
    df['ema_rel_dist_low'] = df['rel_dist_to_5d_low'].ewm(span=5, adjust=False).mean()
    df['weighted_sum'] = df['ema_rel_dist_high'] * 0.6 + df['ema_rel_dist_low'] * 0.4  # Adjust weights as needed
    heuristics_matrix = df['weighted_sum'].dropna()
    return heuristics_matrix
