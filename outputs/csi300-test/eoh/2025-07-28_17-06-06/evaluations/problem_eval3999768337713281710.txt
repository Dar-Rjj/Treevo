def heuristics_v2(df):
    # Calculate the 30-day momentum of the closing price
    df['momentum'] = df['close'] / df['close'].shift(30) - 1
    
    # Calculate the 30-day RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=30).mean()
    avg_loss = loss.rolling(window=30).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Integrate momentum and RSI
    heuristics_matrix = df['momentum'] * df['rsi']
    
    return heuristics_matrix
