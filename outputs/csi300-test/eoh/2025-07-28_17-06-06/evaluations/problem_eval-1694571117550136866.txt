def heuristics_v2(df):
    high_rolling_max = df['high'].rolling(window=30).max()
    low_rolling_min = df['low'].rolling(window=30).min()
    average_volume = df['volume'].rolling(window=30).mean()
    heuristics_matrix = (high_rolling_max - low_rolling_min) / average_volume
    return heuristics_matrix
