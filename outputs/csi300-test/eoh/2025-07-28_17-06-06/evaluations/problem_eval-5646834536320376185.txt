def heuristics_v2(df):
    df['20d_mean_close'] = df['close'].rolling(window=20).mean()
    df['20d_std_close'] = df['close'].rolling(window=20).std()
    df['rel_dist_to_20d_mean'] = (df['close'] - df['20d_mean_close']) / df['20d_std_close']
    total_weight = df['rel_dist_to_20d_mean'].rolling(window=30).sum()
    df['weighted_volume'] = df['volume'] * df['rel_dist_to_20d_mean']
    df['weighted_amount'] = df['amount'] * df['rel_dist_to_20d_mean']
    df['weighted_avg'] = (df['weighted_volume'].rolling(window=30).sum() + df['weighted_amount'].rolling(window=30).sum()) / total_weight
    heuristics_matrix = df['weighted_avg'].dropna()
    return heuristics_matrix
