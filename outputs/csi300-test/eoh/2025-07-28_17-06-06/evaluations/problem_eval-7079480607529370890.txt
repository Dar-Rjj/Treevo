def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    ema_daily_return = daily_return.ewm(span=20, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    std_5 = df['close'].rolling(window=5).std()
    std_20 = df['close'].rolling(window=20).std()
    heuristics_matrix = (ema_daily_return - ema_volume) * (std_5 / std_20)
    return heuristics_matrix
