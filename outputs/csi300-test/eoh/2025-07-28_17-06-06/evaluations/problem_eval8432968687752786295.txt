def heuristics_v2(df):
    # Calculate the 10-day Rate of Change (ROC)
    roc = ((df['close'] - df['close'].shift(10)) / df['close'].shift(10)) * 100
    
    # Calculate the 30-day Moving Average (MA) of volume
    ma_volume = df['volume'].rolling(window=30).mean()
    
    # Normalize the current volume with the 30-day MA volume
    volume_ratio = df['volume'] / ma_volume
    
    # Create a heuristic score by combining ROC and volume ratio
    heuristics_matrix = (roc * volume_ratio).dropna()
    
    return heuristics_matrix
