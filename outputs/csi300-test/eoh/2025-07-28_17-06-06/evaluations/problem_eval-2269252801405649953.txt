def heuristics_v2(df):
    df['5d_high'] = df['high'].rolling(window=5).max()
    df['5d_low'] = df['low'].rolling(window=5).min()
    df['avg_5d_vol'] = df['volume'].rolling(window=5).mean()
    df['rel_dist_to_5d_max'] = (df['5d_high'] - df['close']) / (df['5d_high'] - df['5d_low']) * (df['volume'] / df['avg_5d_vol'])
    df['rel_dist_to_5d_min'] = (df['close'] - df['5d_low']) / (df['5d_high'] - df['5d_low']) * (df['volume'] / df['avg_5d_vol'])
    df['avg_rel_dist'] = (df['rel_dist_to_5d_max'] + df['rel_dist_to_5d_min']) / 2
    heuristics_matrix = df['avg_rel_dist'].dropna()
    return heuristics_matrix
