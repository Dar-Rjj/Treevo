def heuristics_v2(df):
    df['10d_mean_close'] = df['close'].rolling(window=10).mean()
    df['rel_dist_to_10d_mean'] = (df['close'] - df['10d_mean_close']) / df['10d_mean_close']
    total_weight = df['rel_dist_to_10d_mean'].rolling(window=30).sum()
    df['weighted_volume'] = df['volume'] * df['rel_dist_to_10d_mean']
    df['weighted_amount'] = df['amount'] * df['rel_dist_to_10d_mean']
    df['weighted_sum'] = (df['weighted_volume'].rolling(window=30).sum() + df['weighted_amount'].rolling(window=30).sum()) / total_weight
    heuristics_matrix = df['weighted_sum'].dropna()
    return heuristics_matrix
