def heuristics_v2(df):
    df['5d_mean_close'] = df['close'].rolling(window=5).mean()
    df['abs_dist_to_5d_mean'] = abs(df['close'] - df['5d_mean_close'])
    df['ema_weighted_volume'] = df['volume'] * df['abs_dist_to_5d_mean']
    df['ema_weighted_amount'] = df['amount'] * df['abs_dist_to_5d_mean']
    df['ema_weighted_avg'] = (df['ema_weighted_volume'].ewm(span=30).mean() + df['ema_weighted_amount'].ewm(span=30).mean()) / 2
    heuristics_matrix = df['ema_weighted_avg'].dropna()
    return heuristics_matrix
