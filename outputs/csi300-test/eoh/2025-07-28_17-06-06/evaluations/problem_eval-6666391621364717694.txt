def heuristics_v2(df):
    df['5d_mean_close'] = df['close'].rolling(window=5).mean()
    df['sq_dist_to_5d_mean'] = (df['close'] - df['5d_mean_close'])**2
    total_weight = df['sq_dist_to_5d_mean'].rolling(window=30).sum()
    df['weighted_volume'] = df['volume'] * df['sq_dist_to_5d_mean']
    df['weighted_amount'] = df['amount'] * df['sq_dist_to_5d_mean']
    df['weighted_avg'] = (df['weighted_volume'].rolling(window=30).sum() + df['weighted_amount'].rolling(window=30).sum()) / total_weight
    heuristics_matrix = df['weighted_avg'].dropna()
    return heuristics_matrix
