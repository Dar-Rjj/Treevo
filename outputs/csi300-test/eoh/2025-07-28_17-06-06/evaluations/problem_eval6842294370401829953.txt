def heuristics_v2(df):
    df['abs_pct_change'] = df['close'].pct_change().abs()
    df['ema_smoothing_factor'] = 2 / (1 + 1/df['abs_pct_change'].rolling(window=20).mean())
    df['volume_ema'] = df['volume'].ewm(span=30, adjust=False).mean()
    df['amount_ema'] = df['amount'].ewm(span=30, adjust=False).mean()
    df['dynamic_volume_ema'] = df['volume_ema'] * df['ema_smoothing_factor']
    df['dynamic_amount_ema'] = df['amount_ema'] * df['ema_smoothing_factor']
    df['heuristics_matrix'] = (df['dynamic_volume_ema'] + df['dynamic_amount_ema']) / 2
    heuristics_matrix = df['heuristics_matrix'].dropna()
    return heuristics_matrix
