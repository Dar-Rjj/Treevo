def heuristics_v2(df):
    # Calculate the 15-day momentum
    momentum_15d = df['close'] - df['close'].shift(15)
    
    # Calculate the 5-day rate of change in volume
    roc_volume_5d = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Create a heuristics matrix by multiplying 15-day momentum with 5-day ROC of volume
    heuristics_matrix = (momentum_15d * roc_volume_5d).dropna()
    
    return heuristics_matrix
