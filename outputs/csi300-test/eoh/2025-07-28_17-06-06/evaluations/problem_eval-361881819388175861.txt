import pandas as pd

def heuristics_v2(df):
    def roc_series(price, periods=14):
        return (price / price.shift(periods) - 1) * 100

    def cmf_series(df, periods=20):
        money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
        money_flow_volume = money_flow_multiplier * df['volume']
        cmf = money_flow_volume.rolling(window=periods).sum() / df['volume'].rolling(window=periods).sum()
        return cmf

    roc = roc_series(df['close'])
    cmf = cmf_series(df)

    heuristics_matrix = 0.5 * roc + 0.5 * cmf
    return heuristics_matrix
