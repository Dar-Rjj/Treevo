def heuristics_v2(df):
    df['15d_mean_close'] = df['close'].rolling(window=15).mean()
    df['squared_dist_to_15d_mean'] = (df['close'] - df['15d_mean_close'])**2
    df['exp_weight'] = df['squared_dist_to_15d_mean'].ewm(span=40, adjust=False).mean()
    df['weighted_volume'] = df['volume'] * df['exp_weight']
    df['weighted_amount'] = df['amount'] * df['exp_weight']
    df['exp_weighted_avg'] = (df['weighted_volume'].ewm(span=40, adjust=False).mean() + df['weighted_amount'].ewm(span=40, adjust=False).mean()) / df['exp_weight']
    heuristics_matrix = df['exp_weighted_avg'].dropna()
    return heuristics_matrix
