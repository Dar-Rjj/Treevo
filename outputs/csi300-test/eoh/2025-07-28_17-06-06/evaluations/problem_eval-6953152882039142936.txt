def heuristics_v2(df):
    df['sum_high_low'] = df[['high', 'low']].sum(axis=1).rolling(window=10).sum()
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['heuristics_factor'] = df['sum_high_low'] / df['avg_volume']
    heuristics_matrix = df['heuristics_factor'].dropna()
    return heuristics_matrix
