def heuristics_v2(df):
    df['20d_mean_close'] = df['close'].rolling(window=20).mean()
    df['abs_diff_to_20d_mean'] = abs(df['close'] - df['20d_mean_close'])
    df['weighted_abs_diff'] = df['abs_diff_to_20d_mean'] * df['volume']
    df['sum_weighted_abs_diff'] = df['weighted_abs_diff'].rolling(window=30).sum()
    heuristics_matrix = df['sum_weighted_abs_diff'].dropna()
    return heuristics_matrix
