def heuristics_v2(df):
    df['10d_mean_volume'] = df['volume'].rolling(window=10).mean()
    df['20d_min_close'] = df['close'].rolling(window=20).min()
    df['abs_dist_to_20d_min'] = abs(df['close'] - df['20d_min_close'])
    df['volume_ratio'] = df['volume'] / df['10d_mean_volume']
    df['weighted_volume_ratio'] = df['volume_ratio'] * df['abs_dist_to_20d_min']
    df['ema_weighted_volume_ratio'] = df['weighted_volume_ratio'].ewm(span=5, adjust=False).mean()
    heuristics_matrix = df['ema_weighted_volume_ratio'].dropna()
    return heuristics_matrix
