def heuristics_v2(df):
    # Calculate the 14-day Exponential Moving Average (EMA) for the close price
    df['ema_close'] = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the mean of daily return
    df['mean_return'] = df['daily_return'].rolling(window=30).mean().fillna(0)
    
    # Calculate the deviation of daily return from its mean
    df['return_deviation'] = df['daily_return'] - df['mean_return']
    
    # Calculate the cumulative sum of daily return deviations
    df['cumulative_return_deviation'] = df['return_deviation'].cumsum()
    
    # Create the heuristic score by combining EMA of close price and cumulative return deviation
    heuristics_matrix = (df['ema_close'] + df['cumulative_return_deviation']).dropna()
    
    return heuristics_matrix
