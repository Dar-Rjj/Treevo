def heuristics_v2(df):
    # Calculate the 14-day Exponential Moving Average (EMA) of the close price
    ema = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Create a heuristic score by combining EMA and RSI
    heuristics_matrix = (ema * rsi).dropna()

    return heuristics_matrix
