def heuristics_v2(df):
    ema_high = df['high'].ewm(span=10, adjust=False).mean()
    ema_low = df['low'].ewm(span=10, adjust=False).mean()
    std_close = df['close'].rolling(window=10).std()
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    heuristics_matrix = ((ema_high - ema_low) / std_close) * volume_ratio
    return heuristics_matrix
