def heuristics_v2(df):
    # Calculate the 10-day Relative Strength Index (RSI) of closing prices
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 60-day cumulative sum of volume
    cum_vol = df['volume'].rolling(window=60).sum()
    
    # Create a momentum indicator by combining RSI and cumulative volume
    heuristics_matrix = (rsi * cum_vol).dropna()
    
    return heuristics_matrix
