def heuristics_v2(df):
    # Calculate the 5-day Rate of Change (ROC) of closing prices
    roc_close = df['close'].pct_change(periods=5)
    
    # Calculate the 30-day Standard Deviation (STD) of volume
    std_volume = df['volume'].rolling(window=30).std()
    
    # Create a heuristic score by combining ROC of closing prices and STD of volume
    heuristics_matrix = (roc_close * std_volume).dropna()
    
    return heuristics_matrix
