def heuristics_v2(df):
    df['20d_mean_close'] = df['close'].rolling(window=20).mean()
    df['abs_dist_to_20d_mean'] = abs(df['close'] - df['20d_mean_close'])
    total_weight = (df['abs_dist_to_20d_mean'] ** 2).rolling(window=20).sum()
    df['weighted_volume'] = df['volume'] * (df['abs_dist_to_20d_mean'] ** 2)
    df['weighted_amount'] = df['amount'] * (df['abs_dist_to_20d_mean'] ** 2)
    df['weighted_sum'] = (df['weighted_volume'].rolling(window=20).sum() + df['weighted_amount'].rolling(window=20).sum()) / total_weight
    heuristics_matrix = df['weighted_sum'].dropna()
    return heuristics_matrix
