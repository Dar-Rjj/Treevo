def heuristics_v2(df):
    df['10d_mean_close'] = df['close'].rolling(window=10).mean()
    df['20d_mean_volume'] = df['volume'].rolling(window=20).mean()
    df['rel_strength_to_10d_mean'] = df['close'] / df['10d_mean_close']
    df['vol_ratio'] = df['volume'] / df['20d_mean_volume']
    df['composite_score'] = df['rel_strength_to_10d_mean'] * df['vol_ratio']
    heuristics_matrix = df['composite_score'].dropna()
    return heuristics_matrix
