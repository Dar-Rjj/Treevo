def heuristics_v2(df):
    df['5d_mean_close'] = df['close'].rolling(window=5).mean()
    df['close_change'] = df['close'] - df['5d_mean_close']
    df['30d_volatility'] = df['close'].rolling(window=30).std()
    df['relative_strength'] = df['close_change'] / df['30d_volatility']
    heuristics_matrix = df['relative_strength'].dropna()
    return heuristics_matrix
