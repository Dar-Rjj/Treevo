def heuristics_v2(df):
    high_rolling_max = df['high'].rolling(window=20).max()
    low_rolling_min = df['low'].rolling(window=20).min()
    avg_close = df['close'].rolling(window=20).mean()
    volume_ratio = df['volume'] / df['volume'].rolling(window=20).median()
    heuristics_matrix = ((high_rolling_max - low_rolling_min) / avg_close) * volume_ratio
    return heuristics_matrix
