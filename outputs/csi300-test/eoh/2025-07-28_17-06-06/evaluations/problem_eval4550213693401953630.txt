def heuristics_v2(df):
    df['return'] = df['close'].pct_change()
    df['positive_return'] = df['return'].apply(lambda x: x if x > 0 else 0)
    df['negative_return'] = df['return'].apply(lambda x: x if x < 0 else 0)
    pos_sum = df['positive_return'].rolling(window=60).sum()
    neg_sum = df['negative_return'].rolling(window=60).sum().abs()
    heuristics_matrix = (pos_sum / neg_sum).dropna()
    return heuristics_matrix
