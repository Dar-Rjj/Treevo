import pandas as pd

def heuristics_v2(df):
    df['return_5d'] = df['close'].pct_change(5)
    df['return_10d'] = df['close'].pct_change(10)
    df['return_20d'] = df['close'].pct_change(20)
    
    df['volatility_5d'] = df['return_5d'].rolling(window=5).std()
    df['volatility_10d'] = df['return_10d'].rolling(window=10).std()
    df['volatility_20d'] = df['return_20d'].rolling(window=20).std()
    
    # Avoid division by zero
    df['volatility_5d'] = df['volatility_5d'].replace(0, 1e-6)
    df['volatility_10d'] = df['volatility_10d'].replace(0, 1e-6)
    df['volatility_20d'] = df['volatility_20d'].replace(0, 1e-6)
    
    # Inverse of volatilities (weights)
    df['weight_5d'] = 1 / df['volatility_5d']
    df['weight_10d'] = 1 / df['volatility_10d']
    df['weight_20d'] = 1 / df['volatility_20d']
    
    # Normalize the weights
    total_weight = df['weight_5d'] + df['weight_10d'] + df['weight_20d']
    df['weight_5d'] /= total_weight
    df['weight_10d'] /= total_weight
    df['weight_20d'] /= total_weight
    
    # Weighted momentum
    df['weighted_momentum'] = (df['return_5d'] * df['weight_5d'] + 
                               df['return_10d'] * df['weight_10d'] + 
                               df['return_20d'] * df['weight_20d'])
    
    heuristics_matrix = df['weighted_momentum']
    
    return heuristics_matrix
