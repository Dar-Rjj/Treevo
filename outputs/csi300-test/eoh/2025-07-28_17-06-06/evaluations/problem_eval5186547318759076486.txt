def heuristics_v2(df):
    # Calculate the 10-day Moving Average (MA) of high prices
    ma_high = df['high'].rolling(window=10).mean()
    
    # Calculate the 20-day Exponential Moving Average (EMA) of volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Create a heuristic score by combining MA of high prices and EMA of volume
    heuristics_matrix = (ma_high * ema_volume).dropna()
    
    return heuristics_matrix
