def heuristics_v2(df):
    df['5d_high'] = df['high'].rolling(window=5).max()
    df['5d_low'] = df['low'].rolling(window=5).min()
    df['dist_to_5d_extreme'] = (df[['close', '5d_high', '5d_low']]
                                .apply(lambda x: min(abs(x[0] - x[1]), abs(x[0] - x[2])), axis=1))
    heuristics_matrix = df['dist_to_5d_extreme'].dropna()
    return heuristics_matrix
