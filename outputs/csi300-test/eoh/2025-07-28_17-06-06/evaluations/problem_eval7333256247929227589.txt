def heuristics_v2(df):
    # Calculate the 10-day Exponential Moving Average (EMA) of low prices
    ema_low = df['low'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 20-day Moving Average (MA) of volume
    ma_volume = df['volume'].rolling(window=20).mean()
    
    # Create a heuristic score by combining EMA of low prices and MA of volume
    heuristics_matrix = (ema_low * ma_volume).dropna()
    
    return heuristics_matrix
