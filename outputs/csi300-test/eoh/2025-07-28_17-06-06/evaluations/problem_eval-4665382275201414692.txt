def heuristics_v2(df):
    # Calculate the 50-day Simple Moving Average (SMA)
    sma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the ratio of the current closing price to the 50-day SMA
    close_to_sma_ratio = df['close'] / sma_50
    
    # Calculate the 20-day standard deviation of daily returns
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    
    # Create a heuristic score by combining the ratio and the standard deviation
    heuristics_matrix = (close_to_sma_ratio / std_dev_20).dropna()
    
    return heuristics_matrix
