def heuristics_v2(df):
    df['7d_mean_close'] = df['close'].rolling(window=7).mean()
    df['abs_dist_to_7d_mean'] = abs(df['close'] - df['7d_mean_close'])
    total_weight = df['abs_dist_to_7d_mean'].rolling(window=15).sum()
    df['weighted_volume'] = df['volume'] * df['abs_dist_to_7d_mean']
    df['weighted_amount'] = df['amount'] * df['abs_dist_to_7d_mean']
    df['weighted_avg'] = (df['weighted_volume'].rolling(window=15).sum() + df['weighted_amount'].rolling(window=15).sum()) / total_weight
    heuristics_matrix = df['weighted_avg'].dropna()
    return heuristics_matrix
