def heuristics_v2(df):
    # Calculate the 5-day rolling range (high - low)
    df['RollingRange'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Calculate the 20-day exponentially weighted moving average (EWMA) of the closing price
    ewma = df['close'].ewm(span=20, adjust=False).mean()
    
    # Create a heuristic score by combining the rolling range and the EWMA
    heuristics_matrix = df['RollingRange'] / ewma
    
    return heuristics_matrix
