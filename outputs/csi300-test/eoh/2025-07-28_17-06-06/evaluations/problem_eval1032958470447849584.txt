def heuristics_v2(df):
    df['20d_high'] = df['high'].rolling(window=20).max()
    df['20d_low'] = df['low'].rolling(window=20).min()
    df['rel_dist_to_20d_max'] = (df['20d_high'] - df['close']) / (df['20d_high'] - df['20d_low'])
    df['rel_dist_to_20d_min'] = (df['close'] - df['20d_low']) / (df['20d_high'] - df['20d_low'])
    df['avg_rel_dist'] = (df['rel_dist_to_20d_max'] + df['rel_dist_to_20d_min']) / 2
    heuristics_matrix = df['avg_rel_dist'].dropna()
    return heuristics_matrix
