def heuristics_v2(df):
    # Calculate the 14-day Directional Movement Index (DMI)
    df['high_low_diff'] = df['high'] - df['low']
    df['high_close_diff'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_close_diff'] = (df['low'] - df['close'].shift(1)).abs()
    df['tr'] = df[['high_low_diff', 'high_close_diff', 'low_close_diff']].max(axis=1)
    
    df['dm_pos'] = 0
    df['dm_neg'] = 0
    df.loc[(df['high'] - df['high'].shift(1)) > (df['low'].shift(1) - df['low']), 'dm_pos'] = df['high'] - df['high'].shift(1)
    df.loc[(df['low'].shift(1) - df['low']) > (df['high'] - df['high'].shift(1)), 'dm_neg'] = df['low'].shift(1) - df['low']
    
    di_pos = 100 * (df['dm_pos'].rolling(window=14).sum() / df['tr'].rolling(window=14).sum())
    di_neg = 100 * (df['dm_neg'].rolling(window=14).sum() / df['tr'].rolling(window=14).sum())
    
    # Calculate the 14-day Average Volume
    avg_volume = df['volume'].rolling(window=14).mean()
    
    # Create a heuristic score by combining DMI and Average Volume
    heuristics_matrix = (di_pos - di_neg) / avg_volume
    heuristics_matrix = heuristics_matrix.dropna()
    
    return heuristics_matrix
