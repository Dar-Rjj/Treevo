def heuristics_v2(df):
    df['volatility_ratio'] = df['volume'] / (df['high'] - df['low'])
    df['ema_volatility_ratio_10d'] = df['volatility_ratio'].ewm(span=10).mean()
    df['ema_close_30d'] = df['close'].ewm(span=30).mean()
    df['heuristic_score'] = df['ema_volatility_ratio_10d'] - df['ema_close_30d']
    heuristics_matrix = df['heuristic_score'].dropna()
    return heuristics_matrix
