import pandas as pd

def heuristics_v2(df):
    # Calculate 50-day and 200-day exponential moving averages for the closing price
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the average true range over a 50-day period
    tr = df['high'] - df['low']
    atr_50 = tr.rolling(window=50).mean()
    
    # Create a signal based on the difference between the two exponential moving averages
    signal = ema_50 - ema_200
    
    # Create a heuristic score by dividing the signal by the average true range
    heuristics_matrix = (signal / atr_50).dropna()
    
    return heuristics_matrix
