def heuristics_v2(df):
    df['20d_mean_close'] = df['close'].rolling(window=20).mean()
    df['abs_dev_to_20d_mean'] = abs(df['close'] - df['20d_mean_close'])
    total_weight = df['abs_dev_to_20d_mean'].rolling(window=30).sum()
    df['weighted_volume'] = df['volume'] * df['abs_dev_to_20d_mean']
    df['weighted_amount'] = df['amount'] * df['abs_dev_to_20d_mean']
    df['weighted_avg'] = (df['weighted_volume'].rolling(window=30).sum() + df['weighted_amount'].rolling(window=30).sum()) / total_weight
    heuristics_matrix = df['weighted_avg'].dropna()
    return heuristics_matrix
