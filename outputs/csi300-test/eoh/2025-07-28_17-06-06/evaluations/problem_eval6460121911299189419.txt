def heuristics_v2(df):
    df['5d_high'] = df['high'].rolling(window=5).max()
    df['5d_low'] = df['low'].rolling(window=5).min()
    df['dist_to_5d_max'] = (df['5d_high'] - df['close'])
    df['dist_to_5d_min'] = (df['close'] - df['5d_low'])
    df['weight_max'] = 1 / df['dist_to_5d_max']
    df['weight_min'] = 1 / df['dist_to_5d_min']
    df['weighted_avg_rel_dist'] = (df['weight_max']*df['dist_to_5d_max'] + df['weight_min']*df['dist_to_5d_min']) / (df['weight_max'] + df['weight_min'])
    heuristics_matrix = df['weighted_avg_rel_dist'].dropna()
    return heuristics_matrix
