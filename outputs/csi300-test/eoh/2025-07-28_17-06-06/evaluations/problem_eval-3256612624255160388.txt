def heuristics_v2(df):
    # Calculate the 10-day median of low prices
    median_low = df['low'].rolling(window=10).median()
    
    # Calculate the 30-day standard deviation of high prices
    std_high = df['high'].rolling(window=30).std()
    
    # Create a heuristic score by combining the difference of close to 10-day median low and 30-day std of high
    heuristics_matrix = ((df['close'] - median_low) / std_high).dropna()
    
    return heuristics_matrix
