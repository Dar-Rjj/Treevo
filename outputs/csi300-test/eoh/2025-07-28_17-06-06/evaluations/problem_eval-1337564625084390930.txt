import pandas as pd

def heuristics_v2(df):
    # Calculate the cumulative daily return over a 30-day window
    df['cumulative_return'] = (df['close'] / df['close'].shift(30)).fillna(1)
    
    # Calculate a weighted moving average of volume change over a 30-day window
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)).ewm(span=30, adjust=False).mean()
    
    # Calculate the RSI over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Define a factor that combines the cumulative return, volume change, and RSI
    df['combined_factor'] = df['cumulative_return'] * df['volume_change'] * (50 - df['rsi'])
    
    # Create the heuristics matrix based on the combined factor
    heuristics_matrix = df['combined_factor']
    
    return heuristics_matrix
