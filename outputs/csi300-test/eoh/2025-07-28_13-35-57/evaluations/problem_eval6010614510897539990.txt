def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_volume_5d'] = df['volume'].ewm(span=5, adjust=False).mean()  # Calculating 5-day EMA of volume
    df['volume_ratio'] = df['volume'] / df['ema_volume_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_ratio']).dropna()
    return heuristics_matrix
