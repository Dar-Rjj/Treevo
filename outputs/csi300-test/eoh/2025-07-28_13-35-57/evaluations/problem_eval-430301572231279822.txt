def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['ma_20d'] = df['close'].rolling(window=20).mean()  # Calculating 20-day MA
    df['ma_50d'] = df['close'].rolling(window=50).mean()  # Calculating 50-day MA
    df['avg_daily_return_10d'] = df['daily_return'].rolling(window=10).mean()  # Calculating 10-day average daily return
    df['heuristics_score'] = (df['ma_20d'] - df['ma_50d']) * df['avg_daily_return_10d']
    heuristics_matrix = df['heuristics_score'].dropna()
    return heuristics_matrix
