def heuristics_v2(df):
    df['range'] = df['high'] - df['low']
    df['true_range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs().combine((df['close'] - df['open']).abs(), max)
    df['avg_true_range_20d'] = df['true_range'].rolling(window=20).mean()
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['range_ratio'] = df['range'] / df['avg_true_range_20d']
    heuristics_matrix = (df['log_return'].shift(-1) * df['range_ratio']).dropna()
    return heuristics_matrix
