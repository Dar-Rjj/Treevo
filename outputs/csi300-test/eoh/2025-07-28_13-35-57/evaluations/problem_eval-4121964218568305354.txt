def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['cumulative_return_10d'] = (1 + df['daily_return']).rolling(window=10).apply(lambda x: x.prod(), raw=True) - 1
    df['mean_close_20d'] = df['close'].rolling(window=20).mean()
    df['price_deviation_20d'] = df['close'] / df['mean_close_20d']
    df['momentum_5d'] = df['close'] / df['close'].shift(5) - 1
    df['shifted_daily_return_2d'] = df['daily_return'].shift(-2)
    heuristics_matrix = (df['shifted_daily_return_2d'] * (df['price_deviation_20d'] + df['momentum_5d'])).dropna()
    return heuristics_matrix
