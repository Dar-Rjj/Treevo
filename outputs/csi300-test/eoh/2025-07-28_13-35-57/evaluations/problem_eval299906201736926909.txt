def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['high_20d'] = df['high'].rolling(window=20).max()
    df['low_20d'] = df['low'].rolling(window=20).min()
    df['price_ratio_high'] = df['close'] / df['high_20d']
    df['price_ratio_low'] = df['close'] / df['low_20d']
    df['average_price_ratio'] = (df['price_ratio_high'] + df['price_ratio_low']) / 2
    heuristics_matrix = (df['shifted_daily_return'] * df['average_price_ratio']).dropna()
    return heuristics_matrix
