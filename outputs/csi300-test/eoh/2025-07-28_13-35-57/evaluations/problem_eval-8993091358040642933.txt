import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day moving average of volume
    df['Volume_Moving_Avg'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the ratio of current volume to its 20-day moving average
    df['Volume_Ratio'] = df['volume'] / df['Volume_Moving_Avg']
    
    # Calculate the standard deviation of returns over 5, 20, and 60 days
    std_5 = df['close'].pct_change().rolling(window=5).std().fillna(0)
    std_20 = df['close'].pct_change().rolling(window=20).std().fillna(0)
    std_60 = df['close'].pct_change().rolling(window=60).std().fillna(0)
    
    # Generate the heuristic score
    heuristics_matrix = (std_5 + std_20 + std_60) * df['Volume_Ratio']
    
    return heuristics_matrix
