def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_close_30d'] = df['close'].rolling(window=30).mean()
    df['price_deviation'] = df['close'] - df['avg_close_30d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_deviation']).dropna()
    return heuristics_matrix
