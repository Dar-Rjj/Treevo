def heuristics_v3(df):
    # Calculate the RSI for 'close' prices
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    roll_up14 = up.rolling(window=14).mean()
    roll_down14_abs = down.abs().rolling(window=14).mean()
    RS = roll_up14 / roll_down14_abs
    RSI = 100.0 - (100.0 / (1.0 + RS))
    
    # Calculate the rate of change in volume
    df['Volume_ROC'] = df['volume'].pct_change(periods=10)
    
    # Generate a heuristic score combining RSI and Volume ROC
    df['Heuristic_Score'] = RSI + df['Volume_ROC']
    
    # Extracting the heuristic score series
    heuristics_matrix = df['Heuristic_Score'].dropna()
    return heuristics_matrix
