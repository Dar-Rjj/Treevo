def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    weights = np.arange(1, 11)
    df['wma_close_10d'] = df['close'].rolling(window=10).apply(lambda x: np.average(x, weights=weights), raw=True)
    df['price_ratio'] = df['close'] / df['wma_close_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio']).dropna()
    return heuristics_matrix
