def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['sma_5d'] = df['close'].rolling(window=5).mean()  # Calculating 5-day SMA
    df['price_ratio'] = df['close'] / df['sma_5d']
    df['high_low_diff'] = df['high'] - df['low']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio'] * df['high_low_diff']).dropna()
    return heuristics_matrix
