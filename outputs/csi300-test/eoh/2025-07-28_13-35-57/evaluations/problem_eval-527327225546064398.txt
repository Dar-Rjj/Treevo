def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    weights = [1, 2, 3, 4, 5]
    df['weighted_avg_volume_5d'] = df['volume'].rolling(window=5).apply(lambda x: sum(w * v for w, v in zip(weights, x)) / sum(weights), raw=True)
    df['volume_ratio'] = df['volume'] / df['weighted_avg_volume_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_ratio']).dropna()
    return heuristics_matrix
