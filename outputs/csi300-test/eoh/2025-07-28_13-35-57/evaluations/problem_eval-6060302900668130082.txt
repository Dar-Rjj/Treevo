import pandas as pd

def heuristics_v2(df):
    def calculate_percentage_change(series, periods):
        return series.pct_change(periods=periods).fillna(0)

    momentum_5 = calculate_percentage_change(df['close'], 5)
    momentum_20 = calculate_percentage_change(df['close'], 20)
    momentum_60 = calculate_percentage_change(df['close'], 60)
    
    heuristics_matrix = (momentum_5 + momentum_20 + momentum_60) / 3
    return heuristics_matrix
