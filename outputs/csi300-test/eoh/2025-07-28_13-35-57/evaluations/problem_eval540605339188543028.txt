def heuristics_v2(df):
    # Calculate EMAs for close price and volume
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate 30-day volatility as a measure of price movement
    volatility = df['close'].rolling(window=30).std().fillna(0)

    # Dynamically adjust the weights based on the 30-day volatility
    weight_price = 1 / (1 + volatility)
    weight_volume = 1 - weight_price

    # Compute the heuristic factor
    heuristics_matrix = (ema_close * weight_price) + (ema_volume * weight_volume)

    return heuristics_matrix
