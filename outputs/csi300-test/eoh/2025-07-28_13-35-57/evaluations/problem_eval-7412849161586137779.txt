def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['vwap_5d'] = (df['amount'] / df['volume']).rolling(window=5).mean()  # Calculating 5-day VWAP
    df['price_ratio'] = df['close'] / df['vwap_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio']).dropna()
    return heuristics_matrix
