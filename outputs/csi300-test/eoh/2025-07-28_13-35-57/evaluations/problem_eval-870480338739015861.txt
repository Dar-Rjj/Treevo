import pandas as pd

def heuristics_v2(df):
    # Calculate the 12-period EMA of 'close' prices
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    
    # Calculate the 10-period simple moving average of 'volume'
    sma_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Ratio of current 'volume' to its 10-period simple moving average
    volume_ratio = df['volume'] / sma_volume_10
    
    # Calculate the Accumulation/Distribution Oscillator (AD)
    ad = (df['close'] - df['low']) - (df['high'] - df['close'])
    ad *= df['volume'] / (df['high'] - df['low'])
    ad.cumsum()
    
    # Determine dynamic weights based on the volume ratio
    weight_ema = 1 / (1 + abs(volume_ratio.fillna(0)))
    weight_ad = 1 - weight_ema
    
    # Compute the heuristic factor
    heuristics_matrix = (ema_12 * weight_ema) + (ad * weight_ad)
    
    return heuristics_matrix
