import pandas as pd

def heuristics_v2(df):
    atr_window = 14
    rsi_window = 14
    close_change = df['close'].pct_change()
    
    # Calculate ATR
    tr = pd.DataFrame({'high-low': df['high'] - df['low'],
                       'high-close': abs(df['high'] - df['close'].shift(1)),
                       'low-close': abs(df['low'] - df['close'].shift(1))})
    tr = tr.max(axis=1)
    atr = tr.rolling(window=atr_window).mean()
    
    # Calculate RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=rsi_window).mean()
    avg_loss = loss.rolling(window=rsi_window).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    heuristics_matrix = (atr * close_change) + rsi
    return heuristics_matrix
