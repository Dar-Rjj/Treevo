def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_high_10d'] = df['high'].rolling(window=10).mean()
    df['high_ratio'] = df['high'] / df['avg_high_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['high_ratio']).dropna()
    return heuristics_matrix
