import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-period standard deviation of close price
    std_10 = df['close'].rolling(window=10).std()
    
    # Calculate the 5-period RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=5).mean()
    avg_loss = loss.rolling(window=5).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the rate of change (ROC) of 'amount'
    roc_amount = df['amount'].pct_change(periods=10).fillna(0)
    
    # Determine dynamic weights based on the standard deviation
    weight_rsi = 1 / (1 + std_10.fillna(0))
    weight_roc = 1 - weight_rsi
    
    # Compute the heuristic factor
    heuristics_matrix = (rsi * weight_rsi) + (roc_amount * weight_roc)
    
    return heuristics_matrix
