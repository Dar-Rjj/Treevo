def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_close_10d'] = df['close'].ewm(span=10).mean()
    df['price_ratio'] = df['close'] / df['ema_close_10d']
    df['volume_ratio'] = df['volume'] / df['volume'].ewm(span=10).mean()
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio'] * df['volume_ratio']).dropna()
    return heuristics_matrix
