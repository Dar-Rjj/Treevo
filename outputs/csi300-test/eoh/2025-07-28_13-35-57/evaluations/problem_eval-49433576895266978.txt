def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_close_5d'] = df['close'].rolling(window=5).mean()
    df['close_ratio'] = df['close'] / df['avg_close_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['close_ratio']).dropna()
    return heuristics_matrix
