def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for different periods
    df['ROC_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    df['ROC_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['ROC_60'] = (df['close'] - df['close'].shift(60)) / df['close'].shift(60)

    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    df['OBV'] = df['OBV'].cumsum()

    # Calculate a custom volatility measure
    df['Volatility'] = df['close'].rolling(window=20).std()

    # Integrate ROC, OBV, and Volatility
    heuristics_matrix = (df['ROC_5'] + df['ROC_20'] + df['ROC_60']) * df['OBV'] / df['Volatility']

    return heuristics_matrix
