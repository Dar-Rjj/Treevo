def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_5d'] = df['daily_return'].shift(-5)
    df['sma_10d'] = df['close'].rolling(window=10).mean()  # Calculating 10-day SMA
    df['ema_20d'] = df['close'].ewm(span=20, adjust=False).mean()  # Calculating 20-day EMA
    df['price_diff_sma_ema'] = df['sma_10d'] - df['ema_20d']
    heuristics_matrix = (df['shifted_daily_return_5d'] * df['price_diff_sma_ema']).dropna()
    return heuristics_matrix
