def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_5d'] = df['daily_return'].shift(-5)
    df['vwap_14d'] = (df['amount'] / df['volume']).rolling(window=14).mean()  # Calculating VWAP
    df['ema_7d'] = df['close'].ewm(span=7, adjust=False).mean()  # Calculating 7-day EMA
    df['price_ratio_vwap'] = df['close'] / df['vwap_14d']
    df['price_ratio_ema'] = df['close'] / df['ema_7d']
    heuristics_matrix = (df['shifted_daily_return_5d'] * (df['price_ratio_vwap'] + df['price_ratio_ema'])).dropna()
    return heuristics_matrix
