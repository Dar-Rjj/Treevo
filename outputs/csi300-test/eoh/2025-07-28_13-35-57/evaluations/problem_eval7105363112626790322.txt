def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change(5)
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_ratio']).dropna()
    return heuristics_matrix
