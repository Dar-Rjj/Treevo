import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-period moving average
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Distance of close price from its 20-period moving average
    dist_from_ma20 = df['close'] - ma_20
    
    # Calculate the RSI for 'close' prices
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the rate of change (ROC) of 'volume'
    roc_volume = df['volume'].pct_change(periods=10).fillna(0)
    
    # Determine dynamic weights based on the distance from MA20
    weight_rsi = 1 / (1 + abs(dist_from_ma20.fillna(0)))
    weight_roc = 1 - weight_rsi
    
    # Compute the heuristic factor
    heuristics_matrix = (rsi * weight_rsi) + (roc_volume * weight_roc)
    
    return heuristics_matrix
