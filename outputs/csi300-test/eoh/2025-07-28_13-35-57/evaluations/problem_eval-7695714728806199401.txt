def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.ewm(span=28, adjust=False).mean()
    avg_loss = loss.ewm(span=28, adjust=False).mean()
    rs = avg_gain / avg_loss
    rsi_28d_ema = 100 - (100 / (1 + rs))
    df['rsi_28d_ema'] = rsi_28d_ema
    heuristics_matrix = (df['shifted_daily_return'] * df['rsi_28d_ema']).dropna()
    return heuristics_matrix
