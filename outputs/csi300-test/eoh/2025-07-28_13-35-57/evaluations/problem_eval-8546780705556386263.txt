import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-period moving average and standard deviation of close price
    ma_10 = df['close'].rolling(window=10).mean()
    std_10 = df['close'].rolling(window=10).std()
    
    # Distance of close price from its 10-period moving average
    dist_from_ma10 = df['close'] - ma_10
    
    # Calculate the RSI
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the rate of change (ROC) of 'volume'
    roc_volume = df['volume'].pct_change(periods=10).fillna(0)
    
    # Determine dynamic weights based on the distance from MA10 divided by the standard deviation
    weight_rsi = 1 / (1 + abs(dist_from_ma10.fillna(0)) / (std_10.fillna(1)))
    weight_roc = 1 - weight_rsi
    
    # Compute the heuristic factor
    heuristics_matrix = (rsi * weight_rsi) + (roc_volume * weight_roc)
    
    return heuristics_matrix
