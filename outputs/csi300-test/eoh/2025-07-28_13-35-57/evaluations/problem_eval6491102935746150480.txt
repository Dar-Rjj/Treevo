def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_volume_5d'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['ema_daily_return_5d'] = df['daily_return'].ewm(span=5, adjust=False).mean()
    heuristics_matrix = (df['shifted_daily_return'] * df['ema_daily_return_5d'] * df['ema_volume_5d']).dropna()
    return heuristics_matrix
