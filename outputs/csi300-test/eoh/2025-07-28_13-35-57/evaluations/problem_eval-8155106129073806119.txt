def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_30d'] = df['close'].ewm(span=30, adjust=False).mean()
    df['price_ratio_ema'] = df['open'] / df['ema_30d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio_ema']).dropna()
    return heuristics_matrix
