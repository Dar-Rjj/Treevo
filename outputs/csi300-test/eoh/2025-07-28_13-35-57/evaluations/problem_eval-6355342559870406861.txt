def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_volume_20d'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_ratio'] = df['volume'] / df['ema_volume_20d']
    df['price_range'] = (df['high'] - df['low']) / df['close']
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_ratio'] * df['price_range']).dropna()
    return heuristics_matrix
