def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_3d'] = df['daily_return'].shift(-3)
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    df['tr_high_low'] = df['high'] - df['low']
    df['tr_high_close_prev'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr_low_close_prev'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr_high_low', 'tr_high_close_prev', 'tr_low_close_prev']].max(axis=1)
    df['atr_10d'] = df['true_range'].rolling(window=10).mean()
    df['price_ratio_atr'] = df['atr_10d'] / df['close']
    heuristics_matrix = (df['shifted_daily_return_3d'] * (df['momentum_5d'] + df['price_ratio_atr'])).dropna()
    return heuristics_matrix
