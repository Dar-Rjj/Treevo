def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_low_5d'] = df['low'].rolling(window=5).mean()
    df['low_ratio'] = df['low'] / df['avg_low_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['low_ratio']).dropna()
    return heuristics_matrix
