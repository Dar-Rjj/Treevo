def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['signed_volume'] = df['volume'] * (df['close'] - df['open']) / df['close']
    df['cumulative_signed_volume_20d'] = df['signed_volume'].rolling(window=20).sum()
    heuristics_matrix = (df['shifted_daily_return'] * df['cumulative_signed_volume_20d']).dropna()
    return heuristics_matrix
