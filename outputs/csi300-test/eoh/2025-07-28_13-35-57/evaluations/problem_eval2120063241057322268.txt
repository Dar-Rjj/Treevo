def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_3d'] = df['daily_return'].shift(-3)
    df['ema_7d'] = df['close'].ewm(span=7, adjust=False).mean()  # Calculating 7-day EMA
    df['ema_14d'] = df['close'].ewm(span=14, adjust=False).mean()  # Calculating 14-day EMA
    df['ema_ratio'] = df['ema_7d'] / df['ema_14d']
    heuristics_matrix = (df['shifted_daily_return_3d'] * df['ema_ratio']).dropna()
    return heuristics_matrix
