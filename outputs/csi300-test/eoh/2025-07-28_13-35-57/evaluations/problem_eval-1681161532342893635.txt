def heuristics_v2(df):
    # Calculate the daily high-low price difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute the rolling sum of the high-low differences
    high_low_sum = df['high_low_diff'].rolling(window=20).sum()
    
    # Calculate the total trading amount over the same window
    amount_sum = df['amount'].rolling(window=20).sum()
    
    # Compute the heuristic factor as the ratio of the two sums
    heuristics_matrix = high_low_sum / amount_sum
    
    return heuristics_matrix
