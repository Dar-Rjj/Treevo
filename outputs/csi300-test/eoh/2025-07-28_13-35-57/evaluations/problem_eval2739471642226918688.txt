import pandas as pd

def heuristics_v2(df):
    # Calculate 10-period RSI from 'close' price
    delta = df['close'].diff(1).dropna()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=10).mean()
    avg_loss = loss.rolling(window=10).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate MACD for 'volume'
    exp1 = df['volume'].ewm(span=12, adjust=False).mean()
    exp2 = df['volume'].ewm(span=26, adjust=False).mean()
    macd_line = exp1 - exp2
    signal_line = macd_line.rolling(window=9).mean()
    macd_diff = macd_line - signal_line
    
    # Volatility (Standard Deviation) of 'close' price over 10 periods
    vol_10 = df['close'].rolling(window=10).std().fillna(0)
    
    # Determine dynamic weights based on the volatility
    weight_rsi = 1 / (1 + vol_10)
    weight_macd = 1 - weight_rsi
    
    # Compute the heuristic factor
    heuristics_matrix = (rsi * weight_rsi) + (macd_diff * weight_macd)
    
    return heuristics_matrix
