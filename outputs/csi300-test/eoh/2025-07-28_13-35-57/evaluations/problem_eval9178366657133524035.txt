def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['amount_ratio']).dropna()
    return heuristics_matrix
