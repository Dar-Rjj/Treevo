def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['sum_amount_20d'] = df['amount'].rolling(window=20).sum()
    df['amount_ratio'] = df['amount'] / df['sum_amount_20d']
    heuristics_matrix = (df['shifted_daily_return'] * df['amount_ratio']).dropna()
    return heuristics_matrix
