import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-period Average True Range (ATR)
    tr = df['high'] - df['low']
    atr = tr.rolling(window=14).mean()
    
    # Calculate the momentum of price changes
    momentum = df['close'].pct_change(periods=14).fillna(0)
    
    # Calculate the cumulative volume over the last 14 periods
    cum_volume = df['volume'].rolling(window=14).sum()
    
    # Compute the heuristic factor
    heuristics_matrix = (momentum * cum_volume) / atr
    
    return heuristics_matrix
