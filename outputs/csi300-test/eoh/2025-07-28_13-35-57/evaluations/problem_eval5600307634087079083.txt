def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Shift the daily return to align it with the volume data as a forward-looking indicator
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    
    # Calculate the difference between today's and yesterday's volume
    df['volume_change'] = df['volume'].diff()
    
    # Calculate the price change
    df['price_change'] = df['close'].diff()
    
    # Generate a trend signal by multiplying the shifted daily return with the sign of the price change
    trend_signal = (df['shifted_daily_return'] * df['price_change'].apply(lambda x: 1 if x > 0 else -1)).dropna()
    
    # Generate a volume trend signal by multiplying the volume change with the sign of the price change
    volume_trend_signal = (df['volume_change'] * df['price_change'].apply(lambda x: 1 if x > 0 else -1)).dropna()
    
    # Combine the trend and volume trend signals
    heuristics_matrix = (trend_signal + volume_trend_signal) / 2
    
    return heuristics_matrix
