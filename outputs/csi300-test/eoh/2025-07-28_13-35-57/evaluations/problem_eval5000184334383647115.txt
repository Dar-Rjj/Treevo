def heuristics_v2(df):
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['shifted_log_return'] = df['log_return'].shift(-1)
    df['vwap_10d'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    df['ma_5d'] = df['close'].rolling(window=5).mean()
    df['price_ratio_vwap'] = df['close'] / df['vwap_10d']
    df['price_ratio_ma'] = df['close'] / df['ma_5d']
    heuristics_matrix = (df['shifted_log_return'] * (df['price_ratio_vwap'] + df['price_ratio_ma'])).dropna()
    return heuristics_matrix
