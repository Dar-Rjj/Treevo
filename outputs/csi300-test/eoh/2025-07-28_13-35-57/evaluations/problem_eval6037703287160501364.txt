def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['vwap_10d'] = (df['amount'] / df['volume']).rolling(window=10).mean()  # Calculating VWAP
    df['price_diff'] = df['close'] - df['vwap_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_diff']).dropna()
    return heuristics_matrix
