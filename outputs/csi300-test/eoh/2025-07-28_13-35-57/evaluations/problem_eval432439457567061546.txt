def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_high_low_14d'] = (df['high'].rolling(window=14).mean() + df['low'].rolling(window=14).mean()) / 2
    df['price_avg_ratio'] = df['close'] / df['avg_high_low_14d']
    df['std_7d'] = df['daily_return'].rolling(window=7).std()
    heuristics_matrix = (df['price_avg_ratio'] * df['std_7d']).dropna()
    return heuristics_matrix
