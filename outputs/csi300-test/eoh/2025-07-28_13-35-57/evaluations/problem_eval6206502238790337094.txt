def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_5d'] = df['daily_return'].shift(-5)
    df['vwap_15d'] = (df['amount'] / df['volume']).rolling(window=15).mean()  # Calculating VWAP
    df['sma_20d'] = df['close'].rolling(window=20).mean()  # Calculating 20-day SMA
    df['price_diff_vwap'] = df['close'] - df['vwap_15d']
    df['price_diff_sma'] = df['close'] - df['sma_20d']
    heuristics_matrix = (df['shifted_daily_return_5d'] * (df['price_diff_vwap'] + df['price_diff_sma'])).dropna()
    return heuristics_matrix
