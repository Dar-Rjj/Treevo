def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['volatility_10d'] = df['daily_return'].rolling(window=10).std()
    df['avg_volume_30d'] = df['volume'].rolling(window=30).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_30d']
    heuristics_matrix = (df['volatility_10d'] * df['volume_ratio']).dropna()
    return heuristics_matrix
