def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['volatility_10d'] = df['daily_return'].rolling(window=10).std()
    df['min_volume_20d'] = df['volume'].rolling(window=20).min()
    df['volume_ratio'] = df['volume'] / df['min_volume_20d']
    heuristics_matrix = (df['volatility_10d'] * df['volume_ratio']).dropna()
    return heuristics_matrix
