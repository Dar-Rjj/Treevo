def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['high_10d_ema'] = df['high'].ewm(span=10, adjust=False).mean()
    df['low_10d_ema'] = df['low'].ewm(span=10, adjust=False).mean()
    df['price_ratio_high_ema'] = df['close'] / df['high_10d_ema']
    df['price_ratio_low_ema'] = df['close'] / df['low_10d_ema']
    heuristics_matrix = (df['shifted_daily_return'] * (df['price_ratio_high_ema'] + df['price_ratio_low_ema'])).dropna()
    return heuristics_matrix
