import pandas as pd

def heuristics_v2(df):
    short_window = 10
    long_window = 50
    df['SMA'] = df['close'].rolling(window=short_window).mean()
    df['LMA'] = df['close'].rolling(window=long_window).mean()
    df['Volume_Change'] = df['volume'].pct_change()
    df['Momentum'] = df['close'].pct_change(periods=short_window)
    heuristics_matrix = (df['SMA'] - df['LMA']) + df['Volume_Change'] * df['Momentum']
    return heuristics_matrix
