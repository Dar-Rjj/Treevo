import pandas as pd

def heuristics_v2(df):
    n = 20  # Example lookback period for the moving average
    # Calculate the simple moving average of the 'volume' column
    df['sma_volume'] = df['volume'].rolling(window=n).mean()
    # Calculate the ratio between the 'close' price and the SMA of volumes
    df['price_volume_ratio'] = df['close'] / df['sma_volume']
    # Calculate the momentum of the price-volume ratio by taking the difference over a lookback period
    heuristics_matrix = df['price_volume_ratio'].diff(periods=n)
    return heuristics_matrix
