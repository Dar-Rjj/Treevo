def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_3d'] = df['daily_return'].shift(-3)
    df['ema_7d'] = df['close'].ewm(span=7, adjust=False).mean()  # Calculating 7-day EMA
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))  # Calculating 14-day RSI
    df['tr'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['atr_5d'] = df['tr'].rolling(window=5).mean()  # Calculating 5-day ATR
    heuristics_matrix = (df['shifted_daily_return_3d'] * (df['ema_7d'] + df['rsi_14d'] + df['atr_5d'])).dropna()
    return heuristics_matrix
