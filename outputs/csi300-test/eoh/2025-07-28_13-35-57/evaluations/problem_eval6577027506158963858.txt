def heuristics_v2(df):
    df['sma_10d'] = df['close'].rolling(window=10).mean()
    df['sma_10d_change'] = df['sma_10d'].pct_change(5)
    df['volatility_20d'] = df['close'].pct_change().rolling(window=20).std()
    heuristics_matrix = (df['sma_10d_change'] * df['volatility_20d']).dropna()
    return heuristics_matrix
