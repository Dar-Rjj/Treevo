    df['Momentum_5'] = df['close'].pct_change(periods=5)
    
    # Volatility factor
    df['Volatility'] = df['close'].pct_change().rolling(window=5).std()
    
    # Volume growth
    df['VolumeGrowth'] = df['volume'].pct_change()
    
    # Heuristics matrix construction
    heuristics_matrix = df[['PriceChange_SMA_5', 'PriceChange_SMA_10', 'Momentum_5', 'Volatility', 'VolumeGrowth']].dropna()
    
    return heuristics_matrix
