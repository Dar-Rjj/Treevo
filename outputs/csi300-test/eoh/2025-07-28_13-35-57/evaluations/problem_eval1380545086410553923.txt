import pandas as pd

def heuristics_v2(df):
    n = 10
    df['Weighted_Close_Volume'] = (df['close'] * df['volume']).ewm(span=n, adjust=False).mean()
    volume_ema = df['volume'].ewm(span=n, adjust=False).mean()
    heuristics_matrix = df['Weighted_Close_Volume'] / volume_ema
    return heuristics_matrix
