def heuristics_v2(df):
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    # Shift the daily return to align it with the volume data as a forward-looking indicator
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    # Calculate the difference between today's and yesterday's volume
    df['volume_change'] = df['volume'].diff()
    # Create the heuristic factor by multiplying the shifted daily return by the sign of the volume change
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_change'].apply(lambda x: 1 if x > 0 else -1)).dropna()
    return heuristics_matrix
