def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['volume_10d_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['volume_10d_avg']
    df['price_change_10d'] = df['close'] / df['close'].shift(9) - 1
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_ratio'] * df['price_change_10d']).dropna()
    return heuristics_matrix
