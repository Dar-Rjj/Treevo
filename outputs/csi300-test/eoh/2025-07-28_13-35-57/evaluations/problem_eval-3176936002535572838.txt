def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_close_5d'] = df['close'].ewm(span=5).mean()
    df['ema_close_20d'] = df['close'].ewm(span=20).mean()
    df['price_trend_diff'] = df['ema_close_5d'] - df['ema_close_20d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_trend_diff']).dropna()
    return heuristics_matrix
