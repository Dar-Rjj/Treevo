def heuristics_v2(df):
    df['avg_daily_return_3d'] = df['close'].pct_change().rolling(window=3).mean()
    df['max_volume_10d'] = df['volume'].rolling(window=10).max()
    df['volume_ratio'] = df['volume'] / df['max_volume_10d']
    heuristics_matrix = (df['avg_daily_return_3d'] * df['volume_ratio']).dropna()
    return heuristics_matrix
