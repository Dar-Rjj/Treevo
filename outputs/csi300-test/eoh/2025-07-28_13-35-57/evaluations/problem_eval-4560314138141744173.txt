def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()  # Calculating 5-day EMA
    df['price_ratio'] = df['close'] / df['ema_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio']).dropna()
    return heuristics_matrix
