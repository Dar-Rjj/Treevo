def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['volatility_5d'] = df['close'].rolling(window=5).std()
    df['max_volume_5d'] = df['volume'].rolling(window=5).max()
    df['volume_ratio'] = df['volume'] / df['max_volume_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['volatility_5d'] * df['volume_ratio']).dropna()
    return heuristics_matrix
