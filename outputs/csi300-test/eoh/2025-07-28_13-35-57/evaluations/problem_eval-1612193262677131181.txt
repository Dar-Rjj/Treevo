def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_3d'] = df['daily_return'].shift(-3)
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()  # Calculating 10-day EMA
    df['price_ratio_ema'] = df['close'] / df['ema_10d']
    heuristics_matrix = (df['shifted_daily_return_3d'] * df['price_ratio_ema']).dropna()
    return heuristics_matrix
