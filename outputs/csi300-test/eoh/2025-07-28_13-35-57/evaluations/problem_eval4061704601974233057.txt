def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_volume_10d'] = df['volume'].ewm(span=10).mean()
    df['price_change'] = (df['close'] - df['open']) / df['open']
    df['score'] = (df['shifted_daily_return'] * df['price_change'] * (df['volume'] / df['ema_volume_10d']))
    heuristics_matrix = df['score'].dropna()
    return heuristics_matrix
