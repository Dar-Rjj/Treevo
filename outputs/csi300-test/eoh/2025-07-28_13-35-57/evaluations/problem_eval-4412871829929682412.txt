import pandas as pd

def heuristics_v2(df):
    # Calculate 14-day EMA for closing prices
    df['EMA'] = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate RSI
    delta = df['close'].diff(1)
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=13, adjust=False).mean()
    ema_down = down.ewm(com=13, adjust=False).mean()
    rs = ema_up/ema_down
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Calculate the adjusted buying pressure ratio
    df['Volume_Diff'] = df['volume'].diff(1).fillna(0)
    df['Adj_Volume_Pos_Ratio'] = (df['volume'] * (df['close'] > df['open']).astype(int) + df['Volume_Diff']) / df['volume']
    
    # Integrate EMA, RSI, and Adjusted Volume Pos Ratio
    heuristics_matrix = df['EMA'] * df['RSI'] * df['Adj_Volume_Pos_Ratio']
    
    return heuristics_matrix
