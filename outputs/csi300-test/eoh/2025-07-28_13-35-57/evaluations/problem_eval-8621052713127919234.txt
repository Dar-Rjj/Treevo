def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['volume_log_diff'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: np.log1p(x) if x > 0 else -np.log1p(-x))
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_log_diff']).dropna()
    return heuristics_matrix
