def heuristics_v2(df):
    n = 10
    df['PriceDiff'] = (df['close'] - df['open']) * df['volume']
    df['Weighted_PriceDiff_EMA'] = df['PriceDiff'].ewm(span=n, adjust=False).mean() / df['volume'].ewm(span=n, adjust=False).mean()
    heuristics_matrix = df['close'] / df['Weighted_PriceDiff_EMA']
    return heuristics_matrix
