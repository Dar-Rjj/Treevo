def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['high_20d'] = df['high'].rolling(window=20).max()
    df['low_20d'] = df['low'].rolling(window=20).min()
    df['ema_close_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    df['price_ratio_high'] = df['ema_close_5d'] / df['high_20d']
    df['price_ratio_low'] = df['ema_close_5d'] / df['low_20d']
    heuristics_matrix = (df['shifted_daily_return'] * (df['price_ratio_high'] + df['price_ratio_low'])).dropna()
    return heuristics_matrix
