def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_5d'] = df['daily_return'].shift(-5)
    df['vwap_20d'] = (df['amount'] / df['volume']).rolling(window=20).mean()  # Calculating VWAP
    df['sma_10d'] = df['close'].rolling(window=10).mean()  # Calculating 10-day SMA
    df['price_ratio_vwap'] = df['close'] / df['vwap_20d']
    df['price_ratio_sma'] = df['close'] / df['sma_10d']
    heuristics_matrix = (df['shifted_daily_return_5d'] * (df['price_ratio_vwap'] + df['price_ratio_sma'])).dropna()
    return heuristics_matrix
