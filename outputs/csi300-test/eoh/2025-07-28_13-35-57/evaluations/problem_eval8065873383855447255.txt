def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['volatility_volume_5d'] = df['volume'].rolling(window=5).std()
    df['volume_volatility'] = df['volume'] / df['volatility_volume_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_volatility']).dropna()
    return heuristics_matrix
