import pandas as pd

def heuristics_v2(df):
    # Calculate the price change
    df['price_change'] = df['close'].pct_change()
    # Weighted moving average of volume
    df['wma_volume'] = (df['volume'] * df['close']).rolling(window=10).mean() / df['close'].rolling(window=10).mean()
    # Alpha factor: difference between today's volume and WMA volume, scaled by price change
    heuristics_matrix = (df['volume'] - df['wma_volume']) * df['price_change']
    return heuristics_matrix
