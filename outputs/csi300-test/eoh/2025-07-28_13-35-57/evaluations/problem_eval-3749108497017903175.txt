def heuristics_v2(df):
    def compute_heuristic(row):
        price_diff = row['high'] - row['low']
        volume_change = (row['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
        return (price_diff * volume_change).fillna(0)
    
    heuristics_matrix = df.apply(compute_heuristic, axis=1)
    return heuristics_matrix
