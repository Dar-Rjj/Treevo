def heuristics_v2(df):
    df['delta'] = df['close'].diff()
    df['gain'] = (df['delta'] > 0) * df['delta']
    df['loss'] = (df['delta'] < 0) * -df['delta']
    avg_gain = df['gain'].rolling(window=3).mean()
    avg_loss = df['loss'].rolling(window=3).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    df['avg_amount_10d'] = df['amount'].rolling(window=10).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_10d']
    heuristics_matrix = (rsi * df['amount_ratio']).dropna()
    return heuristics_matrix
