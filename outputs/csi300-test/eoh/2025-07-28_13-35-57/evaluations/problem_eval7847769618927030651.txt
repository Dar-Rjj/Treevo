def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['ma_10d_daily_return'] = df['daily_return'].rolling(window=10).mean()
    df['vwap_20d'] = (df['amount'] / df['volume']).rolling(window=20).mean()  # Calculating 20-day VWAP
    df['price_ratio'] = df['close'] / df['vwap_20d']
    heuristics_matrix = (df['ma_10d_daily_return'] * df['price_ratio']).dropna()
    return heuristics_matrix
