import pandas as pd

def heuristics_v2(df):
    # Calculate daily price change
    df['price_change'] = df['close'].diff()
    
    # Calculate standard deviation as a measure of volatility
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Compute weight for each day, inversely proportional to the volatility
    df['weight'] = 1 / (df['volatility'] + 1e-6)  # Adding small value to avoid division by zero
    
    # Calculate the price change per unit volume
    df['price_change_per_volume'] = df['price_change'] / df['volume']
    
    # Calculate the weighted moving average of price change per unit volume
    heuristics_matrix = (df['price_change_per_volume'] * df['weight']).rolling(window=20).sum()
    
    return heuristics_matrix
