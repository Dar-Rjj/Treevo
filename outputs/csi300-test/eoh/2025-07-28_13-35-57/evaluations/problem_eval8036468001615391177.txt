def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_high_10d'] = df['high'].ewm(span=10, adjust=False).mean()
    df['ema_low_10d'] = df['low'].ewm(span=10, adjust=False).mean()
    df['high_low_ema_diff'] = df['ema_high_10d'] - df['ema_low_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['high_low_ema_diff']).dropna()
    return heuristics_matrix
