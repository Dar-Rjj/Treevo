def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['avg_high_low_10d'] = (df['high'].rolling(window=10).mean() + df['low'].rolling(window=10).mean()) / 2
    df['price_ratio'] = df['close'] / df['avg_high_low_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio']).dropna()
    return heuristics_matrix
