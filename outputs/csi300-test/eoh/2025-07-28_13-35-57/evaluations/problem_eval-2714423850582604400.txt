def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['momentum_5d'] = df['daily_return'].rolling(window=5).sum()
    df['volume_ma_20d'] = df['volume'].rolling(window=20).mean()
    df['shifted_daily_return_1d'] = df['daily_return'].shift(-1)
    heuristics_matrix = (df['momentum_5d'] * df['volume_ma_20d']).mul(df['shifted_daily_return_1d'], axis=0).dropna()
    return heuristics_matrix
