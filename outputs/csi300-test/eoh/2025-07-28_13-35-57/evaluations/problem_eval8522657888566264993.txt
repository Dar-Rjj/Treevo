import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day EMA for closing prices
    df['EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 14-period RSI
    delta = df['close'].diff(1)
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=13, adjust=False).mean()
    ema_down = down.ewm(com=13, adjust=False).mean()
    rs = ema_up/ema_down
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Calculate the ratio of positive volume to total volume
    df['Volume_Pos_Ratio'] = (df['volume'] * (df['close'] > df['open']).astype(int)) / df['volume']
    
    # Integrate EMA, RSI, and Volume_Pos_Ratio
    heuristics_matrix = (df['EMA'] + df['RSI']) * df['Volume_Pos_Ratio']
    
    return heuristics_matrix
