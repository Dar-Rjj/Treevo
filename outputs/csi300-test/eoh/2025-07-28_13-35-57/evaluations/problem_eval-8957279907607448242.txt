def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_3d'] = df['daily_return'].shift(-3)
    df['vwap_20d'] = (df['amount'] / df['volume']).rolling(window=20).mean()  # Calculating VWAP
    df['ma_10d'] = df['close'].rolling(window=10).mean()  # Calculating 10-day MA
    df['price_ratio_vwap'] = df['close'] / df['vwap_20d']
    df['price_ratio_ma'] = df['close'] / df['ma_10d']
    heuristics_matrix = (df['shifted_daily_return_3d'] * (df['price_ratio_vwap'] + df['price_ratio_ma'])).dropna()
    return heuristics_matrix
