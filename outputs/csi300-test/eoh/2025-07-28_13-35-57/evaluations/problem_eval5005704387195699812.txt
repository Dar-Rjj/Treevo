import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['Return'] = df['close'].pct_change().fillna(0)
    
    # Calculate 14-period EMA for daily return
    df['EMA_Return'] = df['Return'].ewm(span=14, adjust=False).mean()
    
    # Calculate 14-period EMA for volume
    df['EMA_Volume'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate the relative strength of trading volume
    df['RS_Volume'] = df['volume'] / df['EMA_Volume']
    
    # Combine the two factors
    heuristics_matrix = df['EMA_Return'] * df['RS_Volume']
    
    return heuristics_matrix
