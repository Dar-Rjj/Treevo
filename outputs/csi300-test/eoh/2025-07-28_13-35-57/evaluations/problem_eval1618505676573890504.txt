def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['high_low_ratio_5d'] = df['high'].rolling(window=5).max() / df['low'].rolling(window=5).min()
    df['cum_volume_20d'] = df['volume'].rolling(window=20).sum()
    heuristics_matrix = (df['shifted_daily_return'] * (df['high_low_ratio_5d'] + df['cum_volume_20d'])).dropna()
    return heuristics_matrix
