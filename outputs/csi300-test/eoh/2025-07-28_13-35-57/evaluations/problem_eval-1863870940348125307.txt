def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['roc_20d'] = df['close'].pct_change(periods=20)
    df['std_30d'] = df['daily_return'].rolling(window=30).std()
    heuristics_matrix = (df['roc_20d'] * df['std_30d']).dropna()
    return heuristics_matrix
