def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_10d'] = df['daily_return'].shift(-10)
    df['ema_15d'] = df['close'].ewm(span=15, adjust=False).mean()
    df['ema_30d'] = df['close'].ewm(span=30, adjust=False).mean()
    df['ema_ratio'] = df['ema_15d'] / df['ema_30d']
    heuristics_matrix = (df['shifted_daily_return_10d'] * df['ema_ratio']).dropna()
    return heuristics_matrix
