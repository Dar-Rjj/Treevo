def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['vwap_10d'] = (df['amount'] / df['volume']).rolling(window=10).mean()  # Calculating 10-day VWAP
    df['price_ratio'] = df['close'] / df['vwap_10d']
    df['future_std'] = df['daily_return'].shift(-1).rolling(window=5).std()  # Standard deviation of the next 5 days' returns
    heuristics_matrix = (df['price_ratio'] * df['future_std']).dropna()
    return heuristics_matrix
