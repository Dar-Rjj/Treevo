def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ewma_volume_10d'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_ratio'] = df['volume'] / df['ewma_volume_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_ratio']).dropna()
    return heuristics_matrix
