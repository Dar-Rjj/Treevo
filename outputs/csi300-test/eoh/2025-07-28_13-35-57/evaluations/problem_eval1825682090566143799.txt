import pandas as pd

def heuristics_v2(df):
    # Calculate the price range as a percentage of the closing price
    df['Price_Range_Pct'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the 20-day moving average of volume
    df['Volume_Moving_Avg'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the ratio of current volume to its 20-day moving average
    df['Volume_Ratio'] = df['volume'] / df['Volume_Moving_Avg']
    
    # Calculate the Relative Strength Index (RSI) over a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Generate the heuristic score
    heuristics_matrix = df['Price_Range_Pct'] * df['Volume_Ratio'] * df['RSI']
    
    return heuristics_matrix
