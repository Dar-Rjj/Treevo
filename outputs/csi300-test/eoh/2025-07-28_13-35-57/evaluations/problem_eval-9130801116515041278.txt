def heuristics_v2(df):
    df['ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_price_ratio'] = df['ema_volume'] / df['close']
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_price_ratio']).dropna()
    return heuristics_matrix
