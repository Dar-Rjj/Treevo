def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_5d'] = df['daily_return'].shift(-5)
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()  # Calculating 10-day EMA
    df['ema_20d'] = df['close'].ewm(span=20, adjust=False).mean()  # Calculating 20-day EMA
    df['ema_ratio'] = df['ema_10d'] / df['ema_20d']
    heuristics_matrix = (df['shifted_daily_return_5d'] * df['ema_ratio']).dropna()
    return heuristics_matrix
