def heuristics_v2(df):
    # Calculate the 30-day standard deviation of daily returns for volatility
    df['daily_return'] = df['close'].pct_change()
    volatility = df['daily_return'].rolling(window=30).std().fillna(0)
    
    # Compute the momentum factor using the 10-day and 30-day moving averages of close prices
    mom_10d = df['close'].rolling(window=10).mean().fillna(0)
    mom_30d = df['close'].rolling(window=30).mean().fillna(0)
    momentum = (mom_10d - mom_30d) / mom_30d
    
    # Combine the momentum and volatility to form the heuristic matrix
    heuristics_matrix = (momentum * volatility)
    
    return heuristics_matrix
