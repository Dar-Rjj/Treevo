def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_low_10d'] = df['low'].ewm(span=10, adjust=False).mean()
    df['low_ema_ratio'] = df['low'] / df['ema_low_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['low_ema_ratio']).dropna()
    return heuristics_matrix
