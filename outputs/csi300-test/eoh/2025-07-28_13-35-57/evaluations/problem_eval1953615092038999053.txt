import pandas as pd

def heuristics_v2(df):
    n = 14
    # Calculate the 14-period EMA of the 'close' price
    ema_close = df['close'].ewm(span=n, adjust=False).mean()
    
    # Calculate the momentum of the 'close' price
    momentum = (df['close'] / ema_close) - 1
    
    # Calculate the standard deviation of 'volume' over 14 periods
    vol_std = df['volume'].rolling(window=n).std().fillna(0)
    
    # Compute the heuristic factor
    heuristics_matrix = momentum * vol_std
    
    return heuristics_matrix
