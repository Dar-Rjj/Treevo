import pandas as pd

def heuristics_v2(df):
    n = 10
    df['PriceRange'] = df['high'] - df['low']
    df['SqrtVolume'] = df['volume'].apply(lambda x: x**0.5)
    df['Weighted_PriceRange'] = (df['PriceRange'] * df['SqrtVolume']).rolling(window=n).sum() / df['SqrtVolume'].rolling(window=n).sum()
    avg_close = df['close'].rolling(window=n).mean()
    heuristics_matrix = avg_close / df['Weighted_PriceRange']
    return heuristics_matrix
