def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ma_10d'] = df['close'].rolling(window=10).mean()  # Calculating 10-day MA
    df['price_ma_ratio'] = df['close'] / df['ma_10d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ma_ratio']).dropna()
    return heuristics_matrix
