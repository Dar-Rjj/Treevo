def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_2d'] = df['daily_return'].shift(-2)
    df['vwap_15d'] = (df['amount'] / df['volume']).rolling(window=15).mean()  # Calculating VWAP
    df['ma_7d'] = df['close'].rolling(window=7).mean()  # Calculating 7-day MA
    df['price_ratio_vwap'] = df['close'] / df['vwap_15d']
    df['price_ratio_ma'] = df['close'] / df['ma_7d']
    heuristics_matrix = (df['shifted_daily_return_2d'] * (df['price_ratio_vwap'] + df['price_ratio_ma'])).dropna()
    return heuristics_matrix
