def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['vwap_10d'] = (df['amount'] / df['volume']).rolling(window=10).mean()  # Calculating VWAP
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()  # Calculating 5-day EMA
    df['std_5d'] = df['daily_return'].rolling(window=5).std()  # Calculating 5-day Standard Deviation of Returns
    df['price_ratio_vwap'] = df['close'] / df['vwap_10d']
    df['price_ratio_ema'] = df['close'] / df['ema_5d']
    heuristics_matrix = (df['shifted_daily_return'] * ((df['price_ratio_vwap'] + df['price_ratio_ema']) / df['std_5d'])).dropna()
    return heuristics_matrix
