def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['gain'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['daily_return'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain_14d'] = df['gain'].rolling(window=14).mean()
    df['avg_loss_14d'] = df['loss'].rolling(window=14).mean()
    df['rs'] = df['avg_gain_14d'] / df['avg_loss_14d']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    df['max_volume_10d'] = df['volume'].rolling(window=10).max()
    df['volume_ratio'] = df['volume'] / df['max_volume_10d']
    heuristics_matrix = (df['rsi'] * df['volume_ratio']).dropna()
    return heuristics_matrix
