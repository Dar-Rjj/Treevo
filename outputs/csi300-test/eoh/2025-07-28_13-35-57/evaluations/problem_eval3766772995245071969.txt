def heuristics_v2(df):
    # Calculate EMAs for close price and volume
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the standard deviation of close price as a measure of volatility
    price_volatility = df['close'].rolling(window=20).std().fillna(0)
    
    # Dynamically adjust the weights based on the price volatility
    weight_price = 1 / (1 + price_volatility)
    weight_volume = 1 - weight_price
    
    # Compute the heuristic factor
    heuristics_matrix = (ema_close * weight_price) + (ema_volume * weight_volume)
    
    return heuristics_matrix
