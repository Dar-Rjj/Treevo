def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ema_high_20d'] = df['high'].ewm(span=20, adjust=False).mean()
    df['high_ema_ratio'] = df['high'] / df['ema_high_20d']
    heuristics_matrix = (df['shifted_daily_return'] * df['high_ema_ratio']).dropna()
    return heuristics_matrix
