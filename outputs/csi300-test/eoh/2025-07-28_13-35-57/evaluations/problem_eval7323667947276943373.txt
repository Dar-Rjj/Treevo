def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['gain'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['daily_return'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain_14d'] = df['gain'].rolling(window=14).mean()
    df['avg_loss_14d'] = df['loss'].rolling(window=14).mean()
    df['rsi_14d'] = 100 - (100 / (1 + df['avg_gain_14d'] / df['avg_loss_14d']))
    heuristics_matrix = (df['shifted_daily_return'] * df['rsi_14d']).dropna()
    return heuristics_matrix
