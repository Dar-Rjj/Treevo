def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    weights = np.arange(1, 6)
    df['wma_volume_5d'] = df['volume'].rolling(window=5).apply(lambda x: np.average(x, weights=weights), raw=True)
    df['volume_ratio'] = df['volume'] / df['wma_volume_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['volume_ratio']).dropna()
    return heuristics_matrix
