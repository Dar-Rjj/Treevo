def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ma_5d'] = df['close'].rolling(window=5).mean()  # Calculating 5-day MA
    df['price_ma_ratio'] = df['close'] / df['ma_5d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ma_ratio']).dropna()
    return heuristics_matrix
