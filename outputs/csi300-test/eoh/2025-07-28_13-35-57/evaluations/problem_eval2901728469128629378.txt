def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['vwap_10d'] = (df['amount'] / df['volume']).rolling(window=10).mean()  # Calculating VWAP
    df['price_ratio'] = df['close'] / df['vwap_10d']
    df['price_ratio_ma_20d'] = df['price_ratio'].rolling(window=20).mean()  # 20-day moving average of price-to-VWAP ratio
    heuristics_matrix = (df['shifted_daily_return'] * df['price_ratio_ma_20d']).dropna()
    return heuristics_matrix
