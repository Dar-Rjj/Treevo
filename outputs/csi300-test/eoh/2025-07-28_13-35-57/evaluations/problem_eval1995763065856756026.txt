def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['sma_20d'] = df['close'].rolling(window=20).mean()  # 20-day Simple Moving Average
    df['price_deviation_ratio'] = (df['close'] - df['sma_20d']) / df['sma_20d']
    heuristics_matrix = (df['shifted_daily_return'] * df['price_deviation_ratio']).dropna()
    return heuristics_matrix
