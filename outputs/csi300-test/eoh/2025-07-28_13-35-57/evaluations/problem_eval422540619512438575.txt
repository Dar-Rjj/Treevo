def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return_3d'] = df['daily_return'].shift(-3)
    df['vwap_10d'] = (df['amount'] / df['volume']).rolling(window=10).mean()  # Calculating VWAP
    df['ma_5d'] = df['close'].rolling(window=5).mean()  # Calculating 5-day MA
    df['price_ratio_vwap'] = df['close'] / df['vwap_10d']
    df['price_ratio_ma'] = df['close'] / df['ma_5d']
    heuristics_matrix = (df['shifted_daily_return_3d'] * (df['price_ratio_vwap'] + df['price_ratio_ma'])).dropna()
    return heuristics_matrix
