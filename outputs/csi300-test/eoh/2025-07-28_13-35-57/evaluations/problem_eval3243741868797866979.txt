import pandas as pd

def heuristics_v2(df):
    n = 20
    df['PriceRange'] = df['high'] - df['low']
    df['Weighted_PriceRange'] = (df['PriceRange'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    heuristics_matrix = df['close'] / df['Weighted_PriceRange']
    return heuristics_matrix
