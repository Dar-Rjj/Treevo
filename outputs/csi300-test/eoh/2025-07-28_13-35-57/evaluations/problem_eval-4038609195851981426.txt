def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['rolling_mean_20d'] = df['close'].rolling(window=20).mean()
    df['rolling_std_20d'] = df['close'].rolling(window=20).std()
    df['z_score'] = (df['close'] - df['rolling_mean_20d']) / df['rolling_std_20d']
    heuristics_matrix = (df['shifted_daily_return'] * df['z_score']).dropna()
    return heuristics_matrix
