import pandas as pd

def heuristics_v2(df):
    n = 20  # Example lookback period for the moving average
    # Calculate the weighted moving average of the 'low' column using 'volume' as weights
    df['wma_low'] = (df['low'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    # Calculate the difference between the 'close' price and the weighted moving average of lows
    heuristics_matrix = df['close'] - df['wma_low']
    return heuristics_matrix
