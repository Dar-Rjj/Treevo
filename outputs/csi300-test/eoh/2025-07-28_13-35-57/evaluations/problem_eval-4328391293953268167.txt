def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ewma_5d_vol'] = df['volume'].ewm(span=5).mean()  # Calculating 5-day EWMA of volume
    df['ma_10d'] = df['close'].rolling(window=10).mean()  # Calculating 10-day MA
    df['price_ratio_ewma_vol'] = df['close'] / df['ewma_5d_vol']
    df['price_ratio_ma'] = df['close'] / df['ma_10d']
    heuristics_matrix = (df['shifted_daily_return'] * (df['price_ratio_ewma_vol'] + df['price_ratio_ma'])).dropna()
    return heuristics_matrix
