import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day Rate of Change (ROC) for closing prices
    df['ROC'] = df['close'].pct_change(periods=14)
    
    # Calculate the 7-day Weighted Moving Average (WMA) of the volume
    weights = pd.Series(range(1,8))
    df['WMA_Volume'] = df['volume'].rolling(window=7).apply(lambda x: (x*weights).sum() / weights.sum(), raw=True)
    
    # Integrate ROC and WMA_Volume
    heuristics_matrix = df['ROC'] * df['WMA_Volume']
    
    return heuristics_matrix
