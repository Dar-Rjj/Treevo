def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['shifted_daily_return'] = df['daily_return'].shift(-1)
    df['ewma_10d_vol'] = df['volume'].ewm(span=10).mean()  # Calculating 10-day EWMA of volume
    df['ma_5d'] = df['close'].rolling(window=5).mean()  # Calculating 5-day MA
    df['price_ratio_ewma_vol'] = df['close'] / df['ewma_10d_vol']
    df['price_ratio_ma'] = df['close'] / df['ma_5d']
    heuristics_matrix = (df['shifted_daily_return'] * (df['price_ratio_ewma_vol'] + df['price_ratio_ma'])).dropna()
    return heuristics_matrix
