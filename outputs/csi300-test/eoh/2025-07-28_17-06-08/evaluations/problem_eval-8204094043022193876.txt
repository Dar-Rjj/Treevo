def heuristics_v2(df):
    df['price_ratio'] = df['close'] / df['open']
    df['cum_product_20'] = df['price_ratio'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    df['ema_volume_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = df['cum_product_20'] * df['ema_volume_10']
    return heuristics_matrix
