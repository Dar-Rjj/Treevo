def heuristics_v2(df):
    df['avg_5_close'] = df['close'].rolling(window=5).mean()
    df['avg_15_close'] = df['close'].rolling(window=15).mean()
    df['cum_volume_30'] = df['volume'].rolling(window=30).sum()
    heuristics_matrix = ((df['avg_5_close'] + df['avg_15_close']) / 2 * df['cum_volume_30']).dropna()
    return heuristics_matrix
