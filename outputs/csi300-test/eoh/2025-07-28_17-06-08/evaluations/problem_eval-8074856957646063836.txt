def heuristics_v2(df):
    df['high_low_diff'] = df['high'] - df['low']
    df['volume_weighted_high_low_wma'] = (df['high_low_diff'] * df['volume']).rolling(window=15).sum() / df['volume'].rolling(window=15).sum()
    df['volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (df['volume_weighted_high_low_wma'] / df['volume_ema']).dropna()
    return heuristics_matrix
