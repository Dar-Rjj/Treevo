def heuristics_v2(df):
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    df['ema_volume_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['std_high_20'] = df['high'].rolling(window=20).std()
    heuristics_matrix = (df['sma_close_10'] / df['ema_volume_5']) * df['std_high_20']
    return heuristics_matrix
