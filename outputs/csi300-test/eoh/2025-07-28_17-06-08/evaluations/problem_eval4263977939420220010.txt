import pandas as pd

def heuristics_v3(df):
    close_sma = df['close'].rolling(window=20).mean()
    close_momentum_sma = (df['close'] / close_sma) - 1
    volume_change = df['volume'].pct_change()
    atr = (df['high'] - df['low']).rolling(window=14).mean()  # ATR over 14 days
    
    # Compute historical correlation for weighting (simplified using static values here for example)
    corr_close_return = 0.5  # Example value, to be computed dynamically
    corr_vol_change = 0.3   # Example value, to be computed dynamically
    corr_atr = 0.2          # Example value, to be computed dynamically
    
    heuristics_matrix = (corr_close_return * close_momentum_sma + 
                         corr_vol_change * volume_change + 
                         corr_atr * atr)
    return heuristics_matrix
