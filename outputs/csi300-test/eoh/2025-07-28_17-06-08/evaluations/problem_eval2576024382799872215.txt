def heuristics_v2(df):
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['ma_typical_10'] = df['typical_price'].rolling(window=10).mean()
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['roc_typical_5'] = df['typical_price'].pct_change(periods=5)
    heuristics_matrix = df['ma_typical_10'] * df['sma_volume_20'] * df['roc_typical_5']
    return heuristics_matrix
