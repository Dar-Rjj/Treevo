def heuristics_v2(df):
    df['open_close_diff'] = df['open'] - df['close']
    df['open_close_ema'] = df['open_close_diff'].ewm(span=10, adjust=False).mean()
    df['amount_std'] = df['amount'].rolling(window=5).std()
    heuristics_matrix = (df['open_close_ema'] * (1 + df['amount_std'])).dropna()
    return heuristics_matrix
