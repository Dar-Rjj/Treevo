import pandas as pd

def heuristics_v3(df):
    close_sma = df['close'].rolling(window=20).mean()
    close_diff_sma = df['close'] - close_sma
    volume_ema = df['volume'].ewm(span=20, adjust=False).mean()
    volume_deviation = df['volume'] - volume_ema
    tr = df['high'] - df['low']
    
    # Dynamically calculate correlations for weighting
    corr_close_return = df['close'].pct_change().rolling(60).corr(close_diff_sma)
    corr_vol_deviation = df['close'].pct_change().rolling(60).corr(volume_deviation)
    corr_tr = df['close'].pct_change().rolling(60).corr(tr)
    
    # Fill NaN values in correlation series with a forward fill
    corr_close_return = corr_close_return.fillna(method='ffill').fillna(0)
    corr_vol_deviation = corr_vol_deviation.fillna(method='ffill').fillna(0)
    corr_tr = corr_tr.fillna(method='ffill').fillna(0)
    
    heuristics_matrix = (corr_close_return * close_diff_sma + 
                         corr_vol_deviation * volume_deviation + 
                         corr_tr * tr)
    return heuristics_matrix
