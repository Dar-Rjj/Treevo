def heuristics_v2(df):
    df['avg_20_close'] = df['close'].rolling(window=20).mean()
    df['daily_return'] = df['close'].pct_change()
    df['cum_return_10'] = df['daily_return'].rolling(window=10).sum()
    heuristics_matrix = df['avg_20_close'] * df['cum_return_10']
    return heuristics_matrix
