import pandas as pd

def heuristics_v2(df):
    # Calculate short-term and long-term moving averages
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['LMA_50'] = df['close'].rolling(window=50).mean()
    
    # Compute daily return and volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility'] = df['daily_return'].rolling(window=20).std() * (252 ** 0.5)
    
    # Heuristic factor calculation: difference between short and long MA, scaled by inverse of volatility
    heuristics_matrix = (df['SMA_10'] - df['LMA_50']) / df['volatility']
    
    return heuristics_matrix
