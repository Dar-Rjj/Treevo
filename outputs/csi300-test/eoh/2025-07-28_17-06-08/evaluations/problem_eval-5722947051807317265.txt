def heuristics_v2(df):
    close_momentum_30 = df['close'].pct_change(30)
    close_momentum_60 = df['close'].pct_change(60)
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_ratio = df['volume'] / avg_volume
    heuristics_matrix = ((close_momentum_30 + close_momentum_60) / 2) * volume_ratio
    return heuristics_matrix
