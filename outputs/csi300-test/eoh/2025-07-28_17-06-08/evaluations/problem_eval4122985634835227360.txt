def heuristics_v2(df):
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['volume_inverse'] = 1 / df['volume']
    df['weighted_return'] = (df['daily_return'] * df['volume_inverse']).rolling(window=20).sum() / df['volume_inverse'].rolling(window=20).sum()
    df['high_median'] = df['high'].rolling(window=10).median()
    heuristics_matrix = (df['weighted_return'] * df['high_median']).dropna()
    return heuristics_matrix
