def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    std_dev = df['daily_return'].rolling(window=20).std()
    avg_return = df['daily_return'].rolling(window=20).mean()
    adjustment_factor = df['daily_return'] / avg_return
    heuristics_matrix = std_dev * adjustment_factor
    return heuristics_matrix
