def heuristics_v2(df):
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    df['sma_open_5'] = df['open'].rolling(window=5).mean()
    df['ema_amount_10'] = df['amount'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (df['sma_close_20'] / df['sma_open_5']) * df['ema_amount_10']
    return heuristics_matrix
