def heuristics_v2(df):
    df['weighted_mean_price'] = (df['open'].rolling(window=15).mean() * 0.6 + df['close'].rolling(window=15).mean() * 0.4)
    df['volume_sma'] = df['volume'].rolling(window=7).mean()
    heuristics_matrix = (df['weighted_mean_price'] * df['volume_sma']).dropna()
    return heuristics_matrix
