import pandas as pd

def heuristics_v2(df):
    # Calculate short and long term moving averages for open and close
    short_window = 5
    long_window = 20
    df['Open_Short_MA'] = df['open'].rolling(window=short_window).mean()
    df['Close_Short_MA'] = df['close'].rolling(window=short_window).mean()
    df['Open_Long_MA'] = df['open'].rolling(window=long_window).mean()
    df['Close_Long_MA'] = df['close'].rolling(window=long_window).mean()
    
    # Momentum calculation based on volume
    df['Volume_Momentum'] = df['volume'].pct_change(periods=long_window)
    
    # Combine factors into a single heuristic
    heuristics_matrix = (df['Open_Short_MA'] - df['Open_Long_MA']) * 0.5 + (df['Close_Short_MA'] - df['Close_Long_MA']) * 0.5 + df['Volume_Momentum']
    
    return heuristics_matrix
