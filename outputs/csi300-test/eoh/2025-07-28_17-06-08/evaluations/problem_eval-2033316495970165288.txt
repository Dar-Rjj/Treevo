import pandas as pd

def heuristics_v2(df):
    df['avg_close_10'] = df['close'].rolling(window=10).mean()
    df['min_low_10'] = df['low'].rolling(window=10).min()
    df['price_diff'] = df['avg_close_10'] - df['min_low_10']
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['price_diff'] * df['ema_volume_20']
    return heuristics_matrix
