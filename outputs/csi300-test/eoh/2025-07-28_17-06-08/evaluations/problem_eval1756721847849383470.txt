def heuristics_v2(df):
    df['ema_high'] = df['high'].ewm(span=15, adjust=False).mean()
    df['ema_low'] = df['low'].ewm(span=15, adjust=False).mean()
    df['sma_volume'] = df['volume'].rolling(window=5).mean()
    heuristics_matrix = (df['ema_high'] - df['ema_low']) * df['sma_volume']
    return heuristics_matrix
