import pandas as pd

def heuristics_v2(df):
    close_ema = df['close'].ewm(span=20, adjust=False).mean()
    delta = df['volume'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=28).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=28).mean()
    rs = gain / loss
    rsi_vol = 100 - (100 / (1 + rs))
    
    corr_close_ema_return = 0.6  # Updated value
    corr_rsi_vol_return = 0.4    # Updated value
    
    heuristics_matrix = (corr_close_ema_return * close_ema + 
                         corr_rsi_vol_return * rsi_vol).dropna()
    return heuristics_matrix
