def heuristics_v2(df):
    df['return'] = df['close'].pct_change()
    df['volatility_20'] = df['return'].rolling(window=20).std()
    df['adj_close_20'] = df['close'] / df['volatility_20']
    df['sma_adj_close_20'] = df['adj_close_20'].rolling(window=20).mean()
    df['cum_vol_10'] = df['volume'].rolling(window=10).sum()
    heuristics_matrix = df['sma_adj_close_20'] / df['cum_vol_10']
    return heuristics_matrix
