def heuristics_v2(df):
    def compute_intraday_volatility(df, window=10):
        return (df['high'] - df['low']).rolling(window).mean()
    
    def compute_volume_change(df, window=10):
        return df['volume'].pct_change(window)
    
    intraday_volatility = compute_intraday_volatility(df)
    volume_change = compute_volume_change(df)
    
    heuristics_matrix = intraday_volatility * volume_change
    return heuristics_matrix
