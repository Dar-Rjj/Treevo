import pandas as pd

def heuristics_v2(df):
    close_roc = df['close'].pct_change()
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / avg_volume_10d
    
    corr_close_roc_return = 0.7  # Example value
    corr_volume_ratio_return = 0.3  # Example value
    
    heuristics_matrix = (corr_close_roc_return * close_roc + 
                         corr_volume_ratio_return * volume_ratio)
    return heuristics_matrix
