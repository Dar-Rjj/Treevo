def heuristics_v2(df):
    df['high_10'] = df['high'].rolling(window=10).mean()
    df['low_10'] = df['low'].rolling(window=10).mean()
    df['price_diff_10'] = df['high_10'] - df['low_10']
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['price_diff_10'] * df['ema_volume_20']
    return heuristics_matrix
