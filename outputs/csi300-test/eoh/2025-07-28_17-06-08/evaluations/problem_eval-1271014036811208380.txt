def heuristics_v2(df):
    df['cumulative_return_30'] = (df['close'] / df['close'].shift(30)) - 1
    df['sma_volume_5'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['sma_volume_5']
    heuristics_matrix = df['cumulative_return_30'] * df['volume_ratio']
    return heuristics_matrix
