def heuristics_v2(df):
    df['weighted_price'] = ((df['high'] * df['volume']) + (df['low'] * df['volume'])).rolling(window=14).mean() / df['volume'].rolling(window=14).mean()
    df['volume_std'] = df['volume'].rolling(window=5).std()
    heuristics_matrix = (df['weighted_price'] * df['volume_std']).dropna()
    return heuristics_matrix
