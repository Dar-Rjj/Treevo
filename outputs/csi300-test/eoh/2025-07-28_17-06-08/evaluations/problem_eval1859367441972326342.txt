import pandas as pd

def heuristics_v2(df):
    close_ema = df['close'].ewm(span=20, adjust=False).mean()
    close_momentum_ema = (df['close'] / close_ema) - 1
    volume_change = df['volume'].pct_change()
    atr = df['high'] - df['low']
    
    # Compute historical correlation for weighting (simplified using static values here for example)
    corr_close_return = 0.5  # Example value
    corr_vol_change = 0.3   # Example value
    corr_atr = 0.2          # Example value
    
    heuristics_matrix = (corr_close_return * close_momentum_ema + 
                         corr_vol_change * volume_change + 
                         corr_atr * atr)
    return heuristics_matrix
