def heuristics_v2(df):
    # Calculate the daily return based on open and close prices
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute the relative change in volume compared to its 10-day moving average
    volume_ma = df['volume'].rolling(window=10).mean()
    relative_volume_change = df['volume'] / volume_ma
    
    # Combine factors into a single interpretable alpha factor
    heuristics_matrix = daily_return * relative_volume_change
    
    return heuristics_matrix
