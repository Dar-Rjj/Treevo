def heuristics_v2(df):
    df['sma_close_10'] = df['close'].rolling(window=10).mean()
    df['sma_open_20'] = df['open'].rolling(window=20).mean()
    df['ema_volume_5'] = df['volume'].ewm(span=5, adjust=False).mean()
    heuristics_matrix = (df['sma_close_10'] / df['sma_open_20']) * df['ema_volume_5']
    return heuristics_matrix
