def heuristics_v2(df):
    df['ewma_5_close'] = df['close'].ewm(span=5).mean()
    df['ewma_10_close'] = df['close'].ewm(span=10).mean()
    df['ewma_30_volume'] = df['volume'].ewm(span=30).mean()
    heuristics_matrix = ((df['ewma_5_close'] + df['ewma_10_close']) / 2 * df['ewma_30_volume']).dropna()
    return heuristics_matrix
