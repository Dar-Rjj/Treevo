import pandas as pd

def heuristics_v2(df):
    df['avg_high'] = df['high'].rolling(window=10).mean()
    df['avg_low'] = df['low'].rolling(window=10).mean()
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['price_strength'] = (df['avg_high'] + df['avg_low']) / 2
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    gain = df['close'].diff(1).clip(lower=0)
    loss = -df['close'].diff(1).clip(upper=0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    heuristics_matrix = (rsi * df['price_strength'] * df['volume_ratio']).dropna()
    return heuristics_matrix
