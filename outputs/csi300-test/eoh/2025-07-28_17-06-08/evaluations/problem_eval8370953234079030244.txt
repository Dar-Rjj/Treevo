def heuristics_v2(df):
    df['avg_10_close'] = df['close'].rolling(window=10).mean()
    df['avg_20_close'] = df['close'].rolling(window=20).mean()
    df['cum_volume_60'] = df['volume'].rolling(window=60).sum()
    heuristics_matrix = ((df['avg_10_close'] + df['avg_20_close']) / 2 * df['cum_volume_60']).dropna()
    return heuristics_matrix
