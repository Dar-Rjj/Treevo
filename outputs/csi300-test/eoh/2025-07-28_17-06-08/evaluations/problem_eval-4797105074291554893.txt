import pandas as pd

def heuristics_v2(df):
    df['highest_high'] = df['high'].rolling(window=20).max()
    close_to_highest_high = df['close'] / df['highest_high']
    volume_change_std = df['volume'].pct_change().rolling(window=10).std()
    heuristics_matrix = (close_to_highest_high * (1 + volume_change_std)).dropna()
    return heuristics_matrix
