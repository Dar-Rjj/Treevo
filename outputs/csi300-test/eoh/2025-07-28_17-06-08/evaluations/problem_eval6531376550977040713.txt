import pandas as pd

def heuristics_v2(df):
    close_to_open = df['close'] / df['open']
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    std_low = df['low'].rolling(window=5).std()
    
    corr_close_open_return = 0.6  # Example value
    corr_avg_vol_return = 0.3     # Example value
    corr_std_low_return = 0.1     # Example value
    
    heuristics_matrix = (corr_close_open_return * close_to_open + 
                         corr_avg_vol_return * avg_volume_5d + 
                         corr_std_low_return * std_low)
    return heuristics_matrix
