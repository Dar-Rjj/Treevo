import pandas as pd

def heuristics_v2(df):
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['ma_typical_20'] = df['typical_price'].rolling(window=20).mean()
    df['ema_volume_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    combined_metric = df['ma_typical_20'] * df['ema_volume_10']
    
    # Calculate RSI
    delta = combined_metric.diff(1)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    avg_gain = up.rolling(window=14, min_periods=1).mean()
    avg_loss = down.abs().rolling(window=14, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    heuristics_matrix = rsi
    return heuristics_matrix
