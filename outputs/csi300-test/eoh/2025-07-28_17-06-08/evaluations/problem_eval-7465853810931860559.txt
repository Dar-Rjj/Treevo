def heuristics_v2(df):
    df['price_ratio'] = df['close'] / df['open']
    df['cum_product_20'] = df['price_ratio'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    df['sma_amount_10'] = df['amount'].rolling(window=10).mean()
    heuristics_matrix = df['cum_product_20'] * df['sma_amount_10']
    return heuristics_matrix
