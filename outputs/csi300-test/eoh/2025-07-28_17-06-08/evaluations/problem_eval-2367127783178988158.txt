def heuristics_v2(df):
    df['price_change'] = df['close'].diff(1)
    df['gain'] = df['price_change'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['price_change'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain_14'] = df['gain'].rolling(window=14).mean()
    df['avg_loss_14'] = df['loss'].rolling(window=14).mean()
    df['rs'] = df['avg_gain_14'] / df['avg_loss_14']
    df['rsi_14'] = 100 - (100 / (1 + df['rs']))
    df['high_low_range_20'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    heuristics_matrix = df['rsi_14'] * df['high_low_range_20']
    return heuristics_matrix
