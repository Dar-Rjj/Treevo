def heuristics_v2(df):
    df['price_ratio'] = df['close'] / df['open']
    df['cum_product_30'] = df['price_ratio'].rolling(window=30).apply(lambda x: x.prod(), raw=True)
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    heuristics_matrix = df['cum_product_30'] * df['sma_volume_20']
    return heuristics_matrix
