def heuristics_v2(df):
    df['weighted_avg_price'] = ((df['high'] * df['volume']).rolling(window=20).mean() + (df['low'] * df['volume']).rolling(window=20).mean()) / (df['volume'].rolling(window=20).sum() * 2)
    df['volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (df['weighted_avg_price'] / df['volume_ema']).dropna()
    return heuristics_matrix
