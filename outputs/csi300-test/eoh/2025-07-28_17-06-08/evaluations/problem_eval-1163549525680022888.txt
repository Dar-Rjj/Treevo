import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 10-day price returns
    df['5_day_return'] = df['close'].pct_change(5)
    df['10_day_return'] = df['close'].pct_change(10)
    
    # Calculate the average of the highest and lowest returns
    df['avg_high_low_return'] = 0.5 * (df['5_day_return'].rolling(window=5).max() + df['10_day_return'].rolling(window=10).min())
    
    # Calculate the cumulative return over 20 days
    df['cumulative_20_day_return'] = (1 + df['close'].pct_change()).rolling(window=20).apply(lambda x: x.prod(), raw=True) - 1
    
    # Calculate the RSI over 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Generate the heuristics matrix using a weighted sum of the calculated factors
    df['heuristics'] = 0.3 * df['avg_high_low_return'] + 0.4 * df['cumulative_20_day_return'] + 0.3 * df['rsi']
    
    heuristics_matrix = df['heuristics']
    
    return heuristics_matrix
