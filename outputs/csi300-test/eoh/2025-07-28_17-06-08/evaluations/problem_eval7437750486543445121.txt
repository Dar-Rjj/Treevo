def heuristics_v2(df):
    short_window = 10
    long_window = 30
    short_mavg = df['close'].rolling(window=short_window, min_periods=short_window).mean()
    long_mavg = df['close'].rolling(window=long_window, min_periods=long_window).mean()
    heuristics_matrix = short_mavg - long_mavg
    return heuristics_matrix
