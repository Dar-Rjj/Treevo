def heuristics_v2(df):
    df['weighted_avg_price'] = (df['open'].ewm(span=5).mean() + df['close'].ewm(span=5).mean()) / 2
    df['cum_volume_10'] = df['volume'].rolling(window=10).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (df['weighted_avg_price'] * df['cum_volume_10']).dropna()
    return heuristics_matrix
