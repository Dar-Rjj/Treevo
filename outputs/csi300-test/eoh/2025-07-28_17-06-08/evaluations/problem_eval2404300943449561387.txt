def heuristics_v2(df):
    df['weighted_price'] = (df['high'].rolling(window=20).mean() * 0.6 + df['low'].rolling(window=20).mean() * 0.4)
    df['volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (df['weighted_price'] * df['volume_ema']).dropna()
    return heuristics_matrix
