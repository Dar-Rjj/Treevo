def heuristics_v2(df):
    def momentum(df):
        return df['close'].pct_change(10)
    
    def volatility(df):
        return df['close'].pct_change().rolling(window=20).std()
    
    def volume_trend(df):
        return df['volume'].pct_change(5)
    
    heuristics_matrix = (momentum(df) + volatility(df) + volume_trend(df)) / 3
    return heuristics_matrix
