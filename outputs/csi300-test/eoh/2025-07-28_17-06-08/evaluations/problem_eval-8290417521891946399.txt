def heuristics_v2(df):
    df['avg_high_10'] = df['high'].rolling(window=10).mean()
    df['avg_low_10'] = df['low'].rolling(window=10).mean()
    df['ratio_price_10'] = df['avg_high_10'] / df['avg_low_10']
    df['ema_volume_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['ratio_price_10'] * df['ema_volume_20']
    return heuristics_matrix
