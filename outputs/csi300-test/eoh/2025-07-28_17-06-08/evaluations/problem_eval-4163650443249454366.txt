def heuristics_v2(df):
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['close_ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['daily_return'] = df['close'].pct_change()
    df['daily_return_std'] = df['daily_return'].rolling(window=20).std()
    heuristics_matrix = ((df['close_ema_30'] - df['close_ema_10']) / df['daily_return_std']).dropna()
    return heuristics_matrix
