def heuristics_v2(df):
    df['ewma_7_close'] = df['close'].ewm(span=7).mean()
    df['ewma_14_close'] = df['close'].ewm(span=14).mean()
    df['ewma_20_volume'] = df['volume'].ewm(span=20).mean()
    heuristics_matrix = ((df['ewma_7_close'] + df['ewma_14_close']) / 2 * df['ewma_20_volume']).dropna()
    return heuristics_matrix
