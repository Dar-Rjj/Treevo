def heuristics_v2(df):
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['cum_volume_30'] = df['volume'].rolling(window=30).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (df['ema_10_close'] / df['ema_20_close'] * df['cum_volume_30']).dropna()
    return heuristics_matrix
