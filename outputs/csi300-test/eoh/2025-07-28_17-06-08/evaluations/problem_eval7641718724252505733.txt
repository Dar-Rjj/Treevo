def heuristics_v2(df):
    df['close_20'] = df['close'].rolling(window=20).mean()
    df['open_10'] = df['open'].rolling(window=10).mean()
    df['price_ratio'] = df['close_20'] / df['open_10']
    df['ema_amount_15'] = df['amount'].ewm(span=15, adjust=False).mean()
    heuristics_matrix = (df['price_ratio'] * df['ema_amount_15']).dropna()
    return heuristics_matrix
