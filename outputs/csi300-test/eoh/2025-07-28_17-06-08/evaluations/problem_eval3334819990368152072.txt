def heuristics_v2(df):
    df['avg_price'] = (df['open'].rolling(window=5).mean() + df['close'].rolling(window=5).mean()) / 2
    df['ema_cum_volume_10'] = df['volume'].rolling(window=10).sum().ewm(span=10, adjust=False).mean()
    heuristics_matrix = (df['avg_price'] * df['ema_cum_volume_10']).dropna()
    return heuristics_matrix
