import pandas as pd

def heuristics_v2(df):
    close_skew = df['close'].rolling(window=30).skew()
    volume_rsi = 100 - (100 / (1 + (df['volume'].diff(1) > 0).rolling(window=14).mean() / (df['volume'].diff(1) < 0).rolling(window=14).mean()))
    heuristics_matrix = close_skew * volume_rsi
    return heuristics_matrix
