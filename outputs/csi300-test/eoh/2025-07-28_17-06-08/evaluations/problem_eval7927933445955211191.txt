import pandas as pd

def heuristics_v2(df):
    close_momentum = df['close'].pct_change(20)
    volume_change = df['volume'].pct_change()
    volatility = df['high'] - df['low']
    
    # Compute historical correlation for weighting (simplified using static values here for example)
    corr_close_return = 0.5  # Example value
    corr_vol_change = 0.3   # Example value
    corr_volatility = 0.2   # Example value
    
    heuristics_matrix = (corr_close_return * close_momentum + 
                         corr_vol_change * volume_change + 
                         corr_volatility * volatility)
    return heuristics_matrix
