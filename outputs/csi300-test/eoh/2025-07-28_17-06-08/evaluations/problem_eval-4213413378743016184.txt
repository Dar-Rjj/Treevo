def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['return_sum'] = df['daily_return'].rolling(window=7).sum()
    df['volume_cumsum'] = df['volume'].rolling(window=3).sum()
    heuristics_matrix = (df['return_sum'] * df['volume_cumsum']).dropna()
    return heuristics_matrix
