def heuristics_v2(df):
    df['close_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    df['volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['returns'] = df['close'].pct_change()
    df['std_dev_returns'] = df['returns'].rolling(window=20).std()
    heuristics_matrix = ((df['close_ema'] - df['volume_ema']) * df['std_dev_returns']).dropna()
    return heuristics_matrix
