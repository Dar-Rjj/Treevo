def heuristics_v2(df):
    df['close_5'] = df['close'].rolling(window=5).mean()
    df['low_10_min'] = df['low'].rolling(window=10).min()
    df['volume_30_sum'] = df['volume'].rolling(window=30).sum()
    heuristics_matrix = (df['close_5'] / df['low_10_min']) * df['volume_30_sum']
    return heuristics_matrix
