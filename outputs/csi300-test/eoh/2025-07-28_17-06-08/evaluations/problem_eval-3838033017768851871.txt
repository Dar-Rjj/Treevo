def heuristics_v2(df):
    def vpt(df):
        df['vpt'] = (df['volume'] * ((df['close'] - df['close'].shift(1)) / df['close'].shift(1))).cumsum()
        return df['vpt']
    
    momentum = df['close'] - df['close'].shift(20)
    vpt_values = vpt(df)
    heuristics_matrix = (momentum + vpt_values).dropna()
    return heuristics_matrix
