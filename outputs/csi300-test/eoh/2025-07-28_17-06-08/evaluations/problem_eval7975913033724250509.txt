def heuristics_v2(df):
    df['mean_price'] = (df['high'] + df['low']) / 2
    df['mean_price_5'] = df['mean_price'].rolling(window=5).mean()
    df['ema_volume_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    heuristics_matrix = df['mean_price_5'] * df['ema_volume_15']
    return heuristics_matrix
