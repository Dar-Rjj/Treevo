import pandas as pd

def heuristics_v2(df):
    df['high_10'] = df['high'].rolling(window=10).max()
    df['low_10'] = df['low'].rolling(window=10).min()
    df['volatility'] = df['high_10'] - df['low_10']
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    heuristics_matrix = df['volatility'] * df['sma_volume_20']
    return heuristics_matrix
