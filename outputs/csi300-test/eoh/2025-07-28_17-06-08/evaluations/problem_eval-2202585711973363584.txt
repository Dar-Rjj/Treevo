def heuristics_v2(df):
    df['ema_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['cum_product_volume_20'] = df['volume'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (df['ema_close'] * df['cum_product_volume_20']).dropna()
    return heuristics_matrix
