def heuristics_v2(df):
    df['ema_price'] = ((df['open'].ewm(span=10).mean() + df['close'].ewm(span=10).mean()) / 2)
    df['cum_volume_20_prod'] = df['volume'].rolling(window=20).apply(lambda x: x.prod(), raw=True)
    heuristics_matrix = (df['ema_price'] * df['cum_volume_20_prod']).dropna()
    return heuristics_matrix
