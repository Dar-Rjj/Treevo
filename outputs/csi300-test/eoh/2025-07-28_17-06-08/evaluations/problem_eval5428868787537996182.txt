import pandas as pd

def heuristics_v2(df):
    close_momentum = df['close'].pct_change(30)
    volume_ma_change = df['volume'].pct_change().rolling(window=5).mean()
    atr = (df['high'] - df['low']).rolling(window=14).mean()  # Average True Range simplified
    
    # Compute historical correlation for weighting (simplified using static values here for example)
    corr_close_return = 0.6  # Example value
    corr_vol_change = 0.2   # Example value
    corr_volatility = 0.2   # Example value
    
    heuristics_matrix = (corr_close_return * close_momentum + 
                         corr_vol_change * volume_ma_change + 
                         corr_volatility * atr)
    return heuristics_matrix
