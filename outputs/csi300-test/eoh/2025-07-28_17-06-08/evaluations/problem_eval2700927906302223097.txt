def heuristics_v2(df):
    # Compute the 10-day change in closing price
    close_change = df['close'].pct_change(periods=10)
    
    # Calculate the price-to-volume ratio
    price_volume_ratio = df['close'] / df['volume']
    
    # Combine the close change and price-to-volume ratio
    heuristics_matrix = (price_volume_ratio * (1 + close_change)).dropna()
    
    return heuristics_matrix
