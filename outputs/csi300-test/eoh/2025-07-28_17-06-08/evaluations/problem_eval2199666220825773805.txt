def heuristics_v2(df):
    df['high_low_diff'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    df['volume_trend'] = df['volume'].pct_change().rolling(window=5).mean()
    heuristics_matrix = (df['high_low_diff'] * (1 + df['volume_trend'])).dropna()
    return heuristics_matrix
