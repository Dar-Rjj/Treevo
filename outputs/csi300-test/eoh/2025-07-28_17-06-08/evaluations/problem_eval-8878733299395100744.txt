def heuristics_v2(df):
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_20_close'] = df['close'].ewm(span=20, adjust=False).mean()
    df['daily_returns'] = df['close'].pct_change()
    df['std_30_daily_returns'] = df['daily_returns'].rolling(window=30).std()
    heuristics_matrix = (df['ema_10_close'] / df['ema_20_close'] * df['std_30_daily_returns']).dropna()
    return heuristics_matrix
