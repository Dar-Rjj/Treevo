import pandas as pd

def heuristics_v2(df):
    df['range'] = df['high'] - df['low']
    df['range_to_close'] = df['range'] / df['close']
    df['sma_volume'] = df['volume'].rolling(window=10, min_periods=1).mean()
    heuristics_matrix = df['range_to_close'] * df['sma_volume']
    return heuristics_matrix
