import pandas as pd

def heuristics_v2(df):
    df['high_5'] = df['high'].rolling(window=5).mean()
    df['low_5'] = df['low'].rolling(window=5).mean()
    df['ema_volume_30'] = df['volume'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = (df['high_5'] - df['low_5']) * df['ema_volume_30']
    return heuristics_matrix
