def heuristics_v2(df):
    df['max_high'] = df['high'].rolling(window=30).max()
    df['min_low'] = df['low'].rolling(window=30).min()
    df['volume_sma_14'] = df['volume'].rolling(window=14).mean()
    df['volume_sma_28'] = df['volume'].rolling(window=28).mean()
    heuristics_matrix = ((df['max_high'] - df['min_low']) * (df['volume_sma_14'] / df['volume_sma_28'])).dropna()
    return heuristics_matrix
