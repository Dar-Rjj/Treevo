def heuristics_v2(df):
    # Compute momentum factor
    momentum = df['close'].pct_change(periods=126)
    
    # Compute volatility factor
    volatility = df['close'].pct_change().rolling(window=252).std()
    
    # Compute volume factor
    volume_change = df['volume'].pct_change()
    
    # Combine factors into a matrix
    heuristics_matrix = (momentum + volatility + volume_change) / 3
    
    return heuristics_matrix
