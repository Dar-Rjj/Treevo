def heuristics_v2(df):
    vol_ema = df['volume'].ewm(span=10, adjust=False).mean()
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_close = 100 - (100 / (1 + rs))
    
    corr_vol_ema_return = 0.6  # Example value
    corr_rsi_close_return = 0.4  # Example value
    
    heuristics_matrix = (corr_vol_ema_return * vol_ema + 
                         corr_rsi_close_return * rsi_close).dropna()
    return heuristics_matrix
