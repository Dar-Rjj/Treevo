def heuristics_v2(df):
    df['median_price'] = (df['high'].rolling(window=30).median() + df['low'].rolling(window=30).median()) / 2
    df['volume_ema'] = df['volume'].ewm(span=15, adjust=False).mean()
    heuristics_matrix = (df['median_price'] * df['volume_ema']).dropna()
    return heuristics_matrix
