def heuristics_v2(df):
    # Calculate price change
    price_change = df['close'].pct_change()
    
    # Calculate the momentum of price change over a 10-day period
    momentum = price_change.rolling(window=10).sum()
    
    # Calculate the relative strength of trading volume compared to its 30-day average
    avg_volume = df['volume'].rolling(window=30).mean()
    rel_strength_vol = df['volume'] / avg_volume
    
    # Combine momentum and relative strength of volume for the heuristic
    heuristics_matrix = (momentum + rel_strength_vol) / 2
    
    return heuristics_matrix
