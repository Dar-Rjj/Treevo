def heuristics_v2(df):
    df['avg_price'] = (df['high'].rolling(window=20).mean() + df['low'].rolling(window=20).mean()) / 2
    df['price_ratio'] = df['close'] / df['avg_price']
    df['volume_std'] = df['volume'].rolling(window=10).std()
    heuristics_matrix = (df['price_ratio'] * df['volume_std']).dropna()
    return heuristics_matrix
