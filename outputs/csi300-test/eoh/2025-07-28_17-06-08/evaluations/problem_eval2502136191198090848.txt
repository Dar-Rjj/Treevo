def heuristics_v2(df):
    df['weighted_price'] = (df['open'] * 0.2 + df['high'] * 0.3 + df['low'] * 0.3 + df['close'] * 0.2)
    df['sma_weighted_20'] = df['weighted_price'].rolling(window=20).mean()
    df['ema_amount_15'] = df['amount'].ewm(span=15, adjust=False).mean()
    heuristics_matrix = df['sma_weighted_20'] * df['ema_amount_15']
    return heuristics_matrix
