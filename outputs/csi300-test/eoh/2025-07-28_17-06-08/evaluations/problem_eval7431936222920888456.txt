def heuristics_v2(df):
    df['sma_10_close'] = df['close'].rolling(window=10).mean()
    df['sma_50_close'] = df['close'].rolling(window=50).mean()
    df['avg_volume_60'] = df['volume'].rolling(window=60).mean()
    heuristics_matrix = (df['sma_10_close'] - df['sma_50_close']) / df['avg_volume_60']
    return heuristics_matrix
