def heuristics_v2(df):
    df['close_30'] = df['close'].rolling(window=30).mean()
    df['open_20'] = df['open'].rolling(window=20).mean()
    df['price_ratio'] = df['close_30'] / df['open_20']
    df['ema_volume_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = (df['price_ratio'] * df['ema_volume_10']).dropna()
    return heuristics_matrix
