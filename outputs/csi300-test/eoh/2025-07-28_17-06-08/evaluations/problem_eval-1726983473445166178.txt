def heuristics_v2(df):
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['wma_close'] = (df['close'] * df['volume_ratio']).rolling(window=20).sum() / df['volume_ratio'].rolling(window=20).sum()
    df['high_low_std'] = df['high_low_diff'] = df['high'] - df['low']
    df['high_low_std'] = df['high_low_diff'].rolling(window=20).std()
    heuristics_matrix = df['wma_close'] * df['high_low_std']
    return heuristics_matrix
