def heuristics_v2(df):
    # Calculate the 30-day mean of the daily trading volume
    volume_mean = df['volume'].rolling(window=30).mean()
    
    # Calculate the highest price over the past 60 days
    high_60 = df['high'].rolling(window=60).max()
    
    # Calculate the ratio of the closing price to its highest price over the past 60 days
    close_to_high_ratio = df['close'] / high_60
    
    # Final heuristic matrix
    heuristics_matrix = volume_mean * close_to_high_ratio
    
    return heuristics_matrix
