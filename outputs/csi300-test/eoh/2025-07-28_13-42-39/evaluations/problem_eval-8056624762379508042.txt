import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    close_ema = close.ewm(span=20, adjust=False).mean()
    vol_change_rate = volume.diff() / volume.shift(1)
    factor1 = close / close_ema
    factor2 = vol_change_rate + 1  # Adding 1 to avoid negative values
    heuristics_matrix = (factor1 * factor2).cumsum()
    return heuristics_matrix
