def heuristics_v2(df):
    # Calculate the 10-day rate of change (ROC) of the closing price
    df['ROC_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day moving average of the trading volume
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the ratio of the current volume to its 20-day moving average
    volume_ratio = df['volume'] / df['volume_ma_20']
    
    # Final heuristic matrix
    heuristics_matrix = df['ROC_10'] * volume_ratio
    
    return heuristics_matrix
