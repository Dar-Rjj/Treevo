def heuristics_v2(df):
    def custom_alpha(row):
        return row['adj_close'] / ((row['high'] + row['low']) / 2)
    
    df['custom_factor'] = df.apply(custom_alpha, axis=1)
    df['custom_factor_rolling_mean'] = df['custom_factor'].rolling(window=10).mean()
    df['adj_close_ema'] = df['adj_close'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = df['adj_close_ema'] - df['custom_factor_rolling_mean']
    return heuristics_matrix
