def heuristics_v2(df):
    df['vwap_14'] = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    df['vwap_50'] = (df['close'] * df['volume']).rolling(window=50).sum() / df['volume'].rolling(window=50).sum()
    heuristics_matrix = df['vwap_14'] - df['vwap_50']
    return heuristics_matrix
