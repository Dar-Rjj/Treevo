def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['std_daily_return'] = df['daily_return'].rolling(window=20).std()
    df['volume_ewma_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    heuristics_matrix = df['std_daily_return'] / df['volume_ewma_10']
    return heuristics_matrix
