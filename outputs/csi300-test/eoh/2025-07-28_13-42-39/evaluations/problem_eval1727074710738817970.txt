def heuristics_v2(df):
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    df['range'] = df['high'] - df['low']
    df['range_sma_70'] = df['range'].rolling(window=70).mean()
    heuristics_matrix = (df['close_ema_30'] / df['volume_ema_15']) - df['range_sma_70']
    return heuristics_matrix
