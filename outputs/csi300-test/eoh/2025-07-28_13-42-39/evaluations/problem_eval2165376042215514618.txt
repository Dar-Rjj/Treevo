import pandas as pd

def heuristics_v2(df):
    df['close_avg'] = df['close'].rolling(window=10).mean()
    df['volume_avg'] = df['volume'].rolling(window=10).mean()
    close_volume_ratio = df['close_avg'] / df['volume_avg']
    heuristics_matrix = close_volume_ratio.ewm(span=30, adjust=False).mean()
    return heuristics_matrix
