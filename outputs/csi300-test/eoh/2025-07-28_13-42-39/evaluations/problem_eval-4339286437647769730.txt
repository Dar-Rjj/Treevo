import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 25-day exponential moving averages of the closing price
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_25'] = df['close'].ewm(span=25, adjust=False).mean()
    
    # Calculate the difference between the 10-day and 25-day EMA
    ema_diff = df['EMA_10'] - df['EMA_25']
    
    # Calculate the 14-day mean of the daily trading volume
    volume_mean = df['volume'].rolling(window=14).mean()
    
    # Final heuristic matrix
    heuristics_matrix = ema_diff + volume_mean
    
    return heuristics_matrix
