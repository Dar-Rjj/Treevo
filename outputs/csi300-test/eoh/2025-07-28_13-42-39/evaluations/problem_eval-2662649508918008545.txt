import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day and 10-day simple moving averages of the closing price
    df['MA_5'] = df['close'].rolling(window=5).mean()
    df['MA_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the difference between the 5-day and 10-day moving averages
    ma_diff = df['MA_5'] - df['MA_10']
    
    # Calculate the 12-day and 26-day exponential moving averages of the closing price
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    
    # Calculate the EMA crossover signal
    ema_crossover = df['EMA_12'] - df['EMA_26']
    
    # Final heuristic matrix
    heuristics_matrix = ema_crossover + ma_diff
    
    return heuristics_matrix
