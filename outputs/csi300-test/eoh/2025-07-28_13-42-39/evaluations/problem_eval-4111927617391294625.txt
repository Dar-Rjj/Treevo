import pandas as pd
import numpy as np

def heuristics_v2(df):
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['ema_log_returns'] = df['log_returns'].ewm(span=20, adjust=False).mean()
    df['volume_sum_20'] = df['volume'].rolling(window=20).sum()
    df['high_min_50'] = df['high'].rolling(window=50).min()
    heuristics_matrix = (df['ema_log_returns'] * df['volume_sum_20']) - df['high_min_50']
    return heuristics_matrix
