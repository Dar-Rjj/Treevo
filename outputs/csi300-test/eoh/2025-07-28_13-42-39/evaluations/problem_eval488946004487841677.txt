def heuristics_v2(df):
    df['close_ewma_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['close_ewma_20'] = df['close'].ewm(span=20, adjust=False).mean()
    ewma_diff = df['close_ewma_5'] - df['close_ewma_20']
    df['volume_sma_14'] = df['volume'].rolling(window=14).mean()
    heuristics_matrix = ewma_diff / df['volume_sma_14']
    return heuristics_matrix
