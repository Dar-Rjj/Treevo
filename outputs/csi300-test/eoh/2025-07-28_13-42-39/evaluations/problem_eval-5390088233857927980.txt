def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    short_term_ema = df['daily_return'].ewm(span=10, adjust=False).mean()
    long_term_ema = df['daily_return'].ewm(span=30, adjust=False).mean()
    diff_ema = short_term_ema - long_term_ema
    positive_returns = df['daily_return'].where(df['daily_return'] > 0, 0)
    cum_positive_returns = positive_returns.rolling(window=60).sum()
    heuristics_matrix = diff_ema * cum_positive_returns
    return heuristics_matrix
