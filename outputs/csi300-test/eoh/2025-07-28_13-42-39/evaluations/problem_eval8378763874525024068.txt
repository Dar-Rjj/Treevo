import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    
    # Calculate RSI for closing price
    delta = close.diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate EMA for volume and generate crossover signals
    ema_short = volume.ewm(span=5, adjust=False).mean()
    ema_long = volume.ewm(span=20, adjust=False).mean()
    ema_crossover = ema_short - ema_long
    
    # Combine RSI and EMA crossover signal into a single heuristics score
    heuristics_matrix = (rsi * ema_crossover)
    
    return heuristics_matrix
