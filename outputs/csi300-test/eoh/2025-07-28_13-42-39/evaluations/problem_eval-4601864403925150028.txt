def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['amount']
    df['rolling_median'] = df['ratio'].rolling(window=20).median()
    df['close_sma_100'] = df['close'].rolling(window=100).mean()
    heuristics_matrix = df['close_sma_100'] - df['rolling_median']
    return heuristics_matrix
