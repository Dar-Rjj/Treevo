def heuristics_v2(df):
    df['close_ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['range'] = df['high'] - df['low']
    df['range_sma_60'] = df['range'].rolling(window=60).mean()
    heuristics_matrix = (df['close_ema_20'] / df['volume_ema_10']) - df['range_sma_60']
    return heuristics_matrix
