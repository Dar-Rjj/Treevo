import pandas as pd
import numpy as np

def heuristics_v2(df):
    df['high_low_diff'] = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    heuristics_matrix = df['high_low_diff'] + df['log_returns'].cumsum()
    return heuristics_matrix
