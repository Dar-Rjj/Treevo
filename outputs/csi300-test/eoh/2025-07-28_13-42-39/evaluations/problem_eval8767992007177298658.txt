def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_mean'] = df['ratio'].rolling(window=20).mean()
    df['close_ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['rolling_mean'] * df['close_ema_20']
    return heuristics_matrix
