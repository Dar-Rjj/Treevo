import pandas as pd

def heuristics_v2(df):
    # Momentum factor: 5-day return
    momentum_5d = df['close'].pct_change(5)
    
    # Mean reversion factor: Difference between 10-day moving average and current price
    mean_reversion = df['close'].rolling(window=10).mean() - df['close']
    
    # Volume-to-Price ratio: Volume divided by close price
    volume_price_ratio = df['volume'] / df['close']
    
    # Aggregating the factors into a heuristics matrix
    heuristics_matrix = pd.DataFrame({
        'momentum_5d': momentum_5d,
        'mean_reversion': mean_reversion,
        'volume_price_ratio': volume_price_ratio
    })
    
    return heuristics_matrix
