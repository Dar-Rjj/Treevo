def heuristics_v2(df):
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the 20-day standard deviation (volatility)
    volatility = daily_returns.rolling(window=20).std()
    
    # Calculate the 60-day exponential moving average of the closing price
    ema_60 = df['close'].ewm(span=60, adjust=False).mean()
    
    # Calculate the ratio of the closing price to its 60-day EMA
    close_to_ema_ratio = df['close'] / ema_60
    
    # Final heuristic matrix
    heuristics_matrix = volatility * close_to_ema_ratio
    
    return heuristics_matrix
