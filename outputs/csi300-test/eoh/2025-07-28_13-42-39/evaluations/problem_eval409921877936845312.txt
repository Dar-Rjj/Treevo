import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day high and low for the stochastic oscillator
    df['high_14'] = df['high'].rolling(window=14).max()
    df['low_14'] = df['low'].rolling(window=14).min()
    
    # Calculate the %K line of the stochastic oscillator
    k_line = 100 * (df['close'] - df['low_14']) / (df['high_14'] - df['low_14'])
    
    # Calculate the 10-day momentum
    momentum_10 = df['close'] - df['close'].shift(10)
    
    # Final heuristic matrix
    heuristics_matrix = k_line + momentum_10
    
    return heuristics_matrix
