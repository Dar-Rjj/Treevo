def heuristics_v2(df):
    df['volume_ema_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    ema_ratio = df['volume_ema_10'] / df['close_ema_30']
    sma_200_ema_ratio = ema_ratio.rolling(window=200).mean()
    heuristics_matrix = ema_ratio - sma_200_ema_ratio
    return heuristics_matrix
