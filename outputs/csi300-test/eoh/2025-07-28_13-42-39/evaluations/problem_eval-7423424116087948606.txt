def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_mean'] = df['ratio'].rolling(window=10).mean()
    df['close_sma_20'] = df['close'].rolling(window=20).mean()
    heuristics_matrix = df['rolling_mean'] / df['close_sma_20']
    return heuristics_matrix
