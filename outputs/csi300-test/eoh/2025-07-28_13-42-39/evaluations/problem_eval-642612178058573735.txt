def heuristics_v2(df):
    # Calculate the 14-day average of the daily trading volume
    avg_volume = df['volume'].rolling(window=14).mean()
    
    # Calculate the 30-day momentum of the closing prices
    momentum = df['close'] / df['close'].shift(30) - 1
    
    # Final heuristic matrix
    heuristics_matrix = avg_volume * momentum
    
    return heuristics_matrix
