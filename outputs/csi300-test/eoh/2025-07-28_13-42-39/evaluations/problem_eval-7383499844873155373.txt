def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_std'] = df['ratio'].rolling(window=20).std()
    df['close_ewma_50'] = df['close'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = df['close_ewma_50'] - df['rolling_std']
    return heuristics_matrix
