def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_mean'] = df['ratio'].rolling(window=20).mean()
    df['close_ewma_20'] = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['close_ewma_20'] - df['rolling_mean']
    return heuristics_matrix
