def heuristics_v2(df):
    df['ratio'] = df['volume'] / (df['high'] - df['low'])
    df['rolling_median'] = df['ratio'].rolling(window=20).median()
    df['close_ewma_50'] = df['close'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = df['rolling_median'] * df['close_ewma_50']
    return heuristics_matrix
