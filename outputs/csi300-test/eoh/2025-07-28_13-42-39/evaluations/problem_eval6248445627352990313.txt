import pandas as pd
import numpy as np

def heuristics_v2(df):
    df['high_ema_10'] = df['high'].ewm(span=10, adjust=False).mean()
    df['low_ema_10'] = df['low'].ewm(span=10, adjust=False).mean()
    ratio_ema = df['high_ema_10'] / df['low_ema_10']
    df['close_sma_30'] = df['close'].rolling(window=30).mean()
    heuristics_matrix = ratio_ema / np.log(df['close_sma_30'])
    return heuristics_matrix
