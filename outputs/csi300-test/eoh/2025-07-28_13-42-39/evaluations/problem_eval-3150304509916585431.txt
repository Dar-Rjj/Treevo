def heuristics_v2(df):
    # Calculate the 10-day momentum of the closing price
    df['momentum_10'] = df['close'].pct_change(periods=10)
    
    # Calculate the 20-day and 50-day exponentially weighted moving averages of the trading volume
    df['volume_ewma_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_ewma_50'] = df['volume'].ewm(span=50, adjust=False).mean()
    
    # Calculate the ratio of the 20-day to 50-day exponentially weighted moving averages of the trading volume
    volume_ratio = df['volume_ewma_20'] / df['volume_ewma_50']
    
    # Final heuristic matrix
    heuristics_matrix = df['momentum_10'] * volume_ratio
    
    return heuristics_matrix
