import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day and 20-day simple moving averages of the closing price
    df['MA_10'] = df['close'].rolling(window=10).mean()
    df['MA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the 10-day and 20-day moving averages
    ma_diff = df['MA_10'] - df['MA_20']
    
    # Calculate the daily price change
    price_change = df['close'].diff(1)
    
    # Separate gains and losses
    gain = price_change.where(price_change > 0, 0)
    loss = -price_change.where(price_change < 0, 0)
    
    # Calculate the average gain and loss over a 20-day period
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    
    # Calculate the relative strength (RS)
    rs = avg_gain / avg_loss
    
    # Calculate the RSI
    rsi = 100 - (100 / (1 + rs))
    
    # Final heuristic matrix
    heuristics_matrix = rsi + ma_diff
    
    return heuristics_matrix
