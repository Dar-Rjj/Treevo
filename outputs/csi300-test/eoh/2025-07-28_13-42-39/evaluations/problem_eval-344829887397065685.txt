def heuristics_v2(df):
    # Calculate the 20-day Rate of Change (ROC)
    df['ROC_20'] = df['close'].pct_change(periods=20)
    
    # Calculate the 5-day standard deviation of the closing prices for measuring volatility
    df['STD_5'] = df['close'].rolling(window=5).std()
    
    # Calculate the daily price change
    price_change = df['close'].diff(1)
    
    # Separate gains and losses
    gain = price_change.where(price_change > 0, 0)
    loss = -price_change.where(price_change < 0, 0)
    
    # Calculate the average gain and loss over a 14-day period
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    
    # Calculate the relative strength (RS)
    rs = avg_gain / avg_loss
    
    # Calculate the RSI
    rsi = 100 - (100 / (1 + rs))
    
    # Final heuristic matrix combining RSI, ROC, and STD
    heuristics_matrix = rsi + df['ROC_20'] + df['STD_5']
    
    return heuristics_matrix
