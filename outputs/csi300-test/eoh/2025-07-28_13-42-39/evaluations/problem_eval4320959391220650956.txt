def heuristics_v2(df):
    df['high_avg'] = df['high'].rolling(window=10).mean()
    df['low_avg'] = df['low'].rolling(window=10).mean()
    df['std_close'] = df['close'].rolling(window=10).std()
    adjusted_high = df['high_avg'] / df['std_close']
    adjusted_low = df['low_avg'] / df['std_close']
    direction_movement = adjusted_high / adjusted_low
    heuristics_matrix = direction_movement.rolling(window=30).mean()
    return heuristics_matrix
