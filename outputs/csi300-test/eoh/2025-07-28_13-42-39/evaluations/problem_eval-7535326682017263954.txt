def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'].diff()
    
    # Calculate the 10-day exponential moving average of the price change
    ema_price_change = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day standard deviation of the volume
    std_volume = df['volume'].rolling(window=30).std()
    
    # Final heuristic matrix
    heuristics_matrix = ema_price_change + std_volume
    
    return heuristics_matrix
