def heuristics_v2(df):
    # Calculate the 14-day SMA of the closing price
    close_sma = df['close'].rolling(window=14).mean()
    
    # Compute the difference between consecutive days' 14-day SMA to capture momentum
    sma_momentum = close_sma.diff()
    
    # Calculate the 14-day average of the trading amount
    avg_amount = df['amount'].rolling(window=14).mean()
    
    # Calculate the 14-day average of the trading volume
    avg_volume = df['volume'].rolling(window=14).mean()
    
    # Ratio of average trading amount to average trading volume
    amount_volume_ratio = avg_amount / avg_volume
    
    # Final heuristic matrix
    heuristics_matrix = sma_momentum * amount_volume_ratio
    
    return heuristics_matrix
