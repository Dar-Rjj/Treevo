def heuristics_v2(df):
    # Calculate momentum factor using the difference between today's close and the 10-day moving average of close prices
    momentum = df['close'] - df['close'].rolling(window=10).mean()
    
    # Calculate volatility factor using the standard deviation of the last 5 days' closing prices
    volatility = df['close'].rolling(window=5).std()
    
    # Calculate activity factor using the 5-day moving average of trading volume
    activity = df['volume'].rolling(window=5).mean()
    
    # Combine factors into a single heuristics score
    heuristics_matrix = (momentum / volatility) * activity
    
    return heuristics_matrix
