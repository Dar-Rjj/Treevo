def heuristics_v2(df):
    # Calculate the 14-day rate of change (ROC)
    df['close_14_days_ago'] = df['close'].shift(14)
    roc = (df['close'] - df['close_14_days_ago']) / df['close_14_days_ago']
    
    # Calculate daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the rolling standard deviation of daily returns over a 20-day window
    std_dev = daily_returns.rolling(window=20).std()
    
    # Calculate the 20-day exponential moving average of the closing price
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Final heuristic matrix
    heuristics_matrix = (roc + std_dev) * ema_20
    
    return heuristics_matrix
