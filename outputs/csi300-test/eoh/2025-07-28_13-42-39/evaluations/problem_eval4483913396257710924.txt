def heuristics_v2(df):
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['rolling_std_returns'] = df['log_return'].rolling(window=20).std()
    df['vol_adj_return'] = df['log_return'].rolling(window=10).sum() / df['rolling_std_returns']
    heuristics_matrix = df['vol_adj_return']
    return heuristics_matrix
