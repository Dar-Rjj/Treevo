def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_std'] = df['ratio'].rolling(window=20).std()
    df['high_ewma_50'] = df['high'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = df['rolling_std'] / df['high_ewma_50']
    return heuristics_matrix
