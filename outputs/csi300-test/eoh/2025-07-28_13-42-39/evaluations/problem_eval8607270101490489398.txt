import pandas as pd

def heuristics_v2(df):
    N = 5  # Define the lookback period for volume
    df['VolumeAvg'] = df['volume'].rolling(window=N).mean()
    df['MomentumShift'] = (df['close'] - df['open'].shift(1)) / df['VolumeAvg']
    heuristics_matrix = df['MomentumShift'].dropna()
    return heuristics_matrix
