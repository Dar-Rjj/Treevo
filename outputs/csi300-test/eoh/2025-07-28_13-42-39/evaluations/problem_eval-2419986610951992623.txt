def heuristics_v2(df):
    df['price_diff'] = df['high'] - df['low']
    df['close_rolling_std_30'] = df['close'].rolling(window=30).std()
    df['volume_ewma_120'] = df['volume'].ewm(span=120, adjust=False).mean()
    heuristics_matrix = (df['price_diff'] + df['close_rolling_std_30']) / df['volume_ewma_120']
    return heuristics_matrix
