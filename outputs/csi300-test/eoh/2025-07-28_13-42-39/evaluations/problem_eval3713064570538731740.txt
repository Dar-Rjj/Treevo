import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day simple moving average and the 20-day exponential moving average of the closing price
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the ratio between EMA_20 and SMA_10
    ema_sma_ratio = df['EMA_20'] / df['SMA_10']
    
    # Calculate the daily price change
    price_change = df['close'].diff(1)
    
    # Calculate the standard deviation of price changes over a 14-day period
    std_price_change = price_change.rolling(window=14).std()
    
    # Calculate the 14-day mean of the absolute value of price changes as a measure of average magnitude
    mean_abs_price_change = price_change.abs().rolling(window=14).mean()
    
    # Compute the coefficient of variation (CV) for the 14-day period
    cv = std_price_change / mean_abs_price_change
    
    # Adjust the EMA_SMA ratio using the CV
    adjusted_ratio = ema_sma_ratio * (1 + cv)
    
    # Final heuristic matrix
    heuristics_matrix = adjusted_ratio
    
    return heuristics_matrix
