def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_mean'] = df['ratio'].rolling(window=20).mean()
    df['close_ewma_100'] = df['close'].ewm(span=100, adjust=False).mean()
    heuristics_matrix = df['rolling_mean'] / df['close_ewma_100']
    return heuristics_matrix
