def heuristics_v2(df):
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['volume_ema_15'] = df['volume'].ewm(span=15, adjust=False).mean()
    df['avg_price'] = (df['high'] + df['low']) / 2
    df['avg_price_sma_50'] = df['avg_price'].rolling(window=50).mean()
    heuristics_matrix = (df['close_ema_30'] / df['volume_ema_15']) - df['avg_price_sma_50']
    return heuristics_matrix
