import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    close_ma = close.rolling(window=50).mean()
    vol_change = volume.pct_change(periods=10)
    factor1 = close / close_ma
    factor2 = (vol_change + 1).cumprod() - 1
    heuristics_matrix = (factor1 * factor2)
    return heuristics_matrix
