import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    close_ma = close.rolling(window=50).mean()
    vol_change = (volume / volume.rolling(window=10).mean()) - 1
    factor1 = close / close_ma
    factor2 = vol_change
    heuristics_matrix = (factor1 * factor2)
    return heuristics_matrix
