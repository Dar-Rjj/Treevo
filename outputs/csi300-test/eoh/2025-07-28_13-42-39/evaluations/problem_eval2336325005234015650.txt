def heuristics_v2(df):
    def custom_alpha(row):
        return (row['high'] - row['low']) / row['close']
    
    df['custom_factor'] = df.apply(custom_alpha, axis=1)
    df['custom_factor_rolling_avg'] = df['custom_factor'].rolling(window=10).mean()
    df['close_sma'] = df['close'].rolling(window=30).mean()
    heuristics_matrix = df['custom_factor_rolling_avg'] - df['close_sma']
    return heuristics_matrix
