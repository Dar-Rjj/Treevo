def heuristics_v2(df):
    intraday_range = df['high'] - df['low']
    daily_range = df['close'] - df['open']
    average_range = (intraday_range + daily_range) / 2
    volume_change_5_days = df['volume'].pct_change(periods=5)
    heuristics_matrix = average_range * volume_change_5_days
    return heuristics_matrix
