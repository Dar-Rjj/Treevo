def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['amount']
    df['rolling_skew'] = df['ratio'].rolling(window=20).skew()
    df['return_ewma_50'] = df['close'].pct_change().ewm(span=50, adjust=False).mean()
    heuristics_matrix = df['return_ewma_50'] - df['rolling_skew']
    return heuristics_matrix
