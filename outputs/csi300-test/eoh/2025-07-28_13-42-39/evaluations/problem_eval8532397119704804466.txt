def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_median'] = df['ratio'].rolling(window=20).median()
    df['open_sma_50'] = df['open'].rolling(window=50).mean()
    heuristics_matrix = df['rolling_median'] / df['open_sma_50']
    return heuristics_matrix
