def heuristics_v2(df):
    df['high_avg'] = df['high'].rolling(window=20).mean()
    df['low_avg'] = df['low'].rolling(window=20).mean()
    direction_movement = df['high_avg'] - df['low_avg']
    smoothed_direction = direction_movement.ewm(span=14, adjust=False).mean()
    roc_5 = df['close'].pct_change(periods=5)
    heuristics_matrix = smoothed_direction * roc_5
    return heuristics_matrix
