def heuristics_v2(df):
    df['ewma_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['ewma_60'] = df['close'].ewm(span=60, adjust=False).mean()
    ewma_diff = df['ewma_30'] - df['ewma_60']
    df['volume_rolling_std'] = df['volume'].rolling(window=20).std()
    heuristics_matrix = ewma_diff + df['volume_rolling_std']
    return heuristics_matrix
