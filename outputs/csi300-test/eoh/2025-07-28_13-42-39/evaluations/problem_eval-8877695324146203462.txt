def heuristics_v2(df):
    df['price_ratio'] = df['close'] / df['open']
    df['rolling_mean_price_ratio'] = df['price_ratio'].rolling(window=20).mean()
    vol_momentum = df['volume'].pct_change(periods=20)
    heuristics_matrix = df['rolling_mean_price_ratio'] * vol_momentum
    return heuristics_matrix
