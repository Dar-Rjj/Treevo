def heuristics_v2(df):
    # Compute percentage changes
    pct_change_open = df['open'].pct_change()
    pct_change_high = df['high'].pct_change()
    pct_change_low = df['low'].pct_change()
    pct_change_close = df['close'].pct_change()
    volume_change = df['volume'].diff()

    # Create a weighted combination
    heuristics_matrix = 0.25 * pct_change_open + 0.25 * pct_change_high + 0.25 * pct_change_low + 0.25 * pct_change_close + volume_change

    return heuristics_matrix
