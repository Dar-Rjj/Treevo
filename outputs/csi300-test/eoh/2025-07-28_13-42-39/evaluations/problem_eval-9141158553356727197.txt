import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    close_ema = close.ewm(span=20, adjust=False).mean()
    vol_sum = volume.rolling(window=20).sum()
    vol_change = vol_sum.diff()
    factor1 = close / close_ema
    factor2 = vol_change / vol_sum
    heuristics_matrix = (factor1 * factor2)
    return heuristics_matrix
