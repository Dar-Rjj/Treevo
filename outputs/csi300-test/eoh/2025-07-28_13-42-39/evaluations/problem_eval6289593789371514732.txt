def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['mean_return_20'] = df['daily_return'].rolling(window=20).mean()
    weighted_deviation = (df['daily_return'] - df['mean_return_20']) * df['volume']
    heuristics_matrix = weighted_deviation
    return heuristics_matrix
