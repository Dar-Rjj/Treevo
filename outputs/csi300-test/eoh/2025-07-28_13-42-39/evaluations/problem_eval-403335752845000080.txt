def heuristics_v2(df):
    df['ratio'] = df['volume'] / (df['high'] - df['low'])
    df['rolling_median'] = df['ratio'].rolling(window=20).median()
    df['close_sma_50'] = df['close'].rolling(window=50).mean()
    heuristics_matrix = df['close_sma_50'] - df['rolling_median']
    return heuristics_matrix
