import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    close_ma = close.rolling(window=20).mean()
    vol_roc = volume.diff(1) / volume.shift(1)
    factor1 = close / close_ma
    factor2 = vol_roc
    heuristics_matrix = 0.7 * factor1 + 0.3 * factor2
    return heuristics_matrix
