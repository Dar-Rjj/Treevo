def heuristics_v2(df):
    df['high_avg'] = df['high'].rolling(window=20).mean()
    df['low_avg'] = df['low'].rolling(window=20).mean()
    df['std_close'] = df['close'].rolling(window=20).std()
    adjusted_high = df['high_avg'] / df['std_close']
    adjusted_low = df['low_avg'] / df['std_close']
    direction_movement = adjusted_high / adjusted_low
    heuristics_matrix = direction_movement.rolling(window=60).mean()
    return heuristics_matrix
