import pandas as pd

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    close_ema = close.ewm(span=20, adjust=False).mean()
    vol_roc = volume.diff(periods=1) / volume.shift(1)
    factor1 = close / close_ema
    factor2 = vol_roc
    heuristics_matrix = (factor1 * factor2)
    return heuristics_matrix
