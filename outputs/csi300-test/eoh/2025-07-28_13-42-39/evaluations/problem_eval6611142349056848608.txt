def heuristics_v2(df):
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['volume_sma_15'] = df['volume'].rolling(window=15).mean()
    df['range_ema_45'] = (df['high'] - df['low']).ewm(span=45, adjust=False).mean()
    heuristics_matrix = (df['close_ema_30'] / df['volume_sma_15']) - df['range_ema_45']
    return heuristics_matrix
