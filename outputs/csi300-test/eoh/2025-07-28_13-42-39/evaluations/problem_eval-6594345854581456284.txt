import pandas as pd
import numpy as np

def heuristics_v2(df):
    df['price_range_ratio'] = (df['high'] - df['low']) / df['close']
    df['sma_10'] = df['price_range_ratio'].rolling(window=10).mean()
    df['log_ret_vol'] = np.log(df['volume']).diff()
    heuristics_matrix = df['sma_10'] * df['log_ret_vol']
    return heuristics_matrix
