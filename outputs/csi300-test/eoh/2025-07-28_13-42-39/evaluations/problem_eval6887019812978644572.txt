def heuristics_v2(df):
    df['high_avg'] = df['high'].rolling(window=15).mean()
    df['low_avg'] = df['low'].rolling(window=15).mean()
    directional_movement = df['high_avg'] - df['low_avg']
    heuristics_matrix = directional_movement.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
