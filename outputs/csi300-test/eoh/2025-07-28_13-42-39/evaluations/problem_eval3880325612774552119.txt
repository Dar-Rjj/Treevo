import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the inverse of the volume for weighting
    inv_volume = 1 / df['volume']
    
    # Calculate the 14-day rolling weighted average of the daily range with weights inversely proportional to the volume
    weighted_avg_range = (daily_range * inv_volume).rolling(window=14).mean() / inv_volume.rolling(window=14).mean()
    
    # Calculate the 20-day EMA of the closing price
    close_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Final heuristic matrix
    heuristics_matrix = weighted_avg_range - close_ema
    
    return heuristics_matrix
