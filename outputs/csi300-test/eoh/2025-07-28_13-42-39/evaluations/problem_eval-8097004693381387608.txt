def heuristics_v2(df):
    short_window = 10
    long_window = 50
    vol_change_period = 20
    
    # Calculate moving averages
    ma_short = df['close'].rolling(window=short_window).mean()
    ma_long = df['close'].rolling(window=long_window).mean()
    
    # Calculate volume change
    vol_change = df['volume'].pct_change(periods=vol_change_period)
    
    # Alpha factor calculation
    heuristics_matrix = (ma_short - ma_long) * vol_change
    
    return heuristics_matrix
