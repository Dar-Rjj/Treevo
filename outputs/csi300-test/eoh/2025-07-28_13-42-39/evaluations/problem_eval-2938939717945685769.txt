def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_sum'] = df['ratio'].rolling(window=20).sum()
    df['close_ewma_20'] = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['rolling_sum'] / df['close_ewma_20']
    return heuristics_matrix
