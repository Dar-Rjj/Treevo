def heuristics_v2(df):
    close = df['close']
    high = df['high']
    low = df['low']
    amount = df['amount']
    
    roc_close = close.pct_change(periods=10)
    daily_range = high - low
    sma_daily_range = daily_range.rolling(window=20).mean()
    mom_amount = amount.pct_change(periods=5)
    
    factor_roc = roc_close / sma_daily_range
    heuristics_matrix = factor_roc * mom_amount
    return heuristics_matrix
