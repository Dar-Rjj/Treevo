def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['rolling_skew'] = df['daily_return'].rolling(window=10).skew()
    df['volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['close_sma_30'] = df['close'].rolling(window=30).mean()
    heuristics_matrix = (df['rolling_skew'] + df['volume_ema_20']) / df['close_sma_30']
    return heuristics_matrix
