def heuristics_v2(df):
    df['close_ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['low_sma_50'] = df['low'].rolling(window=50).mean()
    diff_price = df['close_ema_20'] - df['low_sma_50']
    df['volume_avg_10'] = df['volume'].rolling(window=10).mean()
    heuristics_matrix = diff_price / df['volume_avg_10']
    return heuristics_matrix
