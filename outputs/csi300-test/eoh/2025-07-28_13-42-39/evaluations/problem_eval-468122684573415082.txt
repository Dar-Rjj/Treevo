import pandas as pd

def heuristics_v2(df):
    # Calculate rolling mean for 10 days
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    # Calculate rolling standard deviation for 10 days
    df['std_10'] = df['daily_return'].rolling(window=10).std()
    # Calculate simple moving average difference (today's close - SMA_10)
    df['SMA_diff'] = df['close'] - df['SMA_10']
    # Calculate the ratio of today's volume to the 10-day average volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    # Construct the heuristics matrix
    heuristics_matrix = df[['SMA_diff', 'std_10', 'volume_ratio']].dropna()
    return heuristics_matrix
