def heuristics_v2(df):
    df['volume_ema'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['high_low_diff'] = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    df['adj_close_ema'] = df['adj_close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (df['volume_ema'] + df['high_low_diff']) / df['adj_close_ema']
    return heuristics_matrix
