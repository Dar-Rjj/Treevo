def heuristics_v2(df):
    def custom_alpha(row):
        return (row['close'] * row['volume']) / row['open']
    
    df['custom_factor'] = df.apply(custom_alpha, axis=1)
    df['custom_factor_rolling_avg'] = df['custom_factor'].rolling(window=15).mean()
    df['high_sma'] = df['high'].rolling(window=20).mean()
    heuristics_matrix = df['custom_factor_rolling_avg'] - df['high_sma']
    return heuristics_matrix
