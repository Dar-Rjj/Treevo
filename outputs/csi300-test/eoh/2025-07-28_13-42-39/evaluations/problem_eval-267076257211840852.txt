def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['sma_20_daily_return'] = df['daily_return'].rolling(window=20).mean()
    df['volume_ewma_50'] = df['volume'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = df['sma_20_daily_return'] * df['volume_ewma_50']
    return heuristics_matrix
