def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_std'] = df['ratio'].rolling(window=20).std()
    df['close_sma_100'] = df['close'].rolling(window=100).mean()
    heuristics_matrix = df['rolling_std'] / df['close_sma_100']
    return heuristics_matrix
