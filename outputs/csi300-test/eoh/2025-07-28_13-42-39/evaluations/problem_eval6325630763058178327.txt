import pandas as pd

def heuristics_v2(df):
    close = df['close']
    open_price = df['open']
    volume = df['volume']
    close_ema = close.ewm(span=20, adjust=False).mean()
    price_diff = close - open_price
    vol_change = volume.pct_change()
    factor1 = (close / close_ema) + price_diff
    factor2 = (vol_change + 1).cumprod() - 1
    heuristics_matrix = (factor1 * factor2)
    return heuristics_matrix
