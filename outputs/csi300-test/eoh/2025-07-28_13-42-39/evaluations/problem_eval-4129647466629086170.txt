def heuristics_v2(df):
    # Calculate the median of high and low prices
    price_median = df[['high', 'low']].median(axis=1)
    
    # Calculate the ratio of closing price to the median of high and low prices
    close_to_median_ratio = df['close'] / price_median
    
    # Compute the 10-day moving average of the close to median ratio
    close_to_median_ratio_avg = close_to_median_ratio.rolling(window=10).mean()
    
    # Calculate the 20-day EMA of the trading volume
    volume_ema = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Final heuristic matrix
    heuristics_matrix = close_to_median_ratio_avg - volume_ema
    
    return heuristics_matrix
