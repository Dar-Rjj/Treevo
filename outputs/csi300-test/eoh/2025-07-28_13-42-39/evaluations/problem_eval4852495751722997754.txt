import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day and 10-day exponential moving averages of the closing price
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio between the 5-day and 10-day exponential moving averages
    ema_ratio = df['EMA_5'] / df['EMA_10']
    
    # Calculate the daily price change
    price_change = df['close'].diff(1)
    
    # Separate gains and losses
    gain = price_change.where(price_change > 0, 0)
    loss = -price_change.where(price_change < 0, 0)
    
    # Calculate the average gain and loss over a 14-day period
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    
    # Calculate the relative strength (RS)
    rs = avg_gain / avg_loss
    
    # Calculate the RSI
    rsi = 100 - (100 / (1 + rs))
    
    # Final heuristic matrix
    heuristics_matrix = rsi * ema_ratio
    
    return heuristics_matrix
