def heuristics_v2(df):
    def custom_alpha(row):
        return (row['high'] - row['low']) / row['open']
    
    df['custom_factor'] = df.apply(custom_alpha, axis=1)
    df['custom_factor_rolling_median'] = df['custom_factor'].rolling(window=10).median()
    df['close_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['custom_factor_rolling_median'] - df['close_ema']
    return heuristics_matrix
