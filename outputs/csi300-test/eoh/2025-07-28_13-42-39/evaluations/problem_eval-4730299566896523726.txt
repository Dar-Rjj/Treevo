import pandas as pd

def heuristics_v2(df):
    short_window = 5
    long_window = 20
    short_avg = (df['high'].rolling(window=short_window).mean() + df['low'].rolling(window=short_window).mean()) / 2
    long_avg = (df['high'].rolling(window=long_window).mean() + df['low'].rolling(window=long_window).mean()) / 2
    heuristics_matrix = (short_avg - long_avg) * 1.5
    return heuristics_matrix
