def heuristics_v2(df):
    df['daily_return'] = df['close'].pct_change()
    df['rolling_std'] = df['daily_return'].rolling(window=10).std()
    df['volume_ewma_30'] = df['volume'].ewm(span=30, adjust=False).mean()
    heuristics_matrix = df['rolling_std'] * df['volume_ewma_30']
    return heuristics_matrix
