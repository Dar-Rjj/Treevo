def heuristics_v2(df):
    df['ratio'] = (df['high'] - df['low']) / df['volume']
    df['rolling_mean'] = df['ratio'].rolling(window=20).mean()
    df['rolling_std'] = df['ratio'].rolling(window=20).std()
    df['z_score'] = (df['ratio'] - df['rolling_mean']) / df['rolling_std']
    df['open_sma_50'] = df['open'].rolling(window=50).mean()
    heuristics_matrix = df['z_score'] / df['open_sma_50']
    return heuristics_matrix
