import pandas as pd

def heuristics_v2(df):
    # Calculate the 7-day and 14-day simple moving averages of the closing price
    df['MA_7'] = df['close'].rolling(window=7).mean()
    df['MA_14'] = df['close'].rolling(window=14).mean()
    
    # Calculate the difference between the 7-day and 14-day moving averages
    ma_diff = df['MA_7'] - df['MA_14']
    
    # Calculate the daily price change
    price_change = df['close'].diff(1)
    
    # Separate gains and losses
    gain = price_change.where(price_change > 0, 0)
    loss = -price_change.where(price_change < 0, 0)
    
    # Calculate the average gain and loss over a 21-day period
    avg_gain = gain.rolling(window=21).mean()
    avg_loss = loss.rolling(window=21).mean()
    
    # Calculate the relative strength (RS)
    rs = avg_gain / avg_loss
    
    # Calculate the RSI
    rsi = 100 - (100 / (1 + rs))
    
    # Final heuristic matrix
    heuristics_matrix = rsi + ma_diff
    
    return heuristics_matrix
