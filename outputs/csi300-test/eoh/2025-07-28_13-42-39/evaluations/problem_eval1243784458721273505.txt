def heuristics_v2(df):
    df['high_ema_30'] = df['high'].ewm(span=30, adjust=False).mean()
    df['low_ema_15'] = df['low'].ewm(span=15, adjust=False).mean()
    diff_ema = df['high_ema_30'] - df['low_ema_15']
    df['close_sma_60'] = df['close'].rolling(window=60).mean()
    heuristics_matrix = diff_ema / df['close_sma_60']
    return heuristics_matrix
