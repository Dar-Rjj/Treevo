def heuristics_v2(df):
    df['high_ema_30'] = df['high'].ewm(span=30, adjust=False).mean()
    df['low_ema_30'] = df['low'].ewm(span=30, adjust=False).mean()
    diff_ema = df['high_ema_30'] - df['low_ema_30']
    df['close_ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    heuristics_matrix = diff_ema / df['close_ema_50']
    return heuristics_matrix
