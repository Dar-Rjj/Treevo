def heuristics_v2(df):
    def custom_alpha(row):
        return (row['high'] - row['low']) / row['adj_close']
    
    df['custom_factor'] = df.apply(custom_alpha, axis=1)
    df['custom_factor_rolling_median'] = df['custom_factor'].rolling(window=10).median()
    df['adj_close_sma'] = df['adj_close'].rolling(window=20).mean()
    heuristics_matrix = df['custom_factor_rolling_median'] - df['adj_close_sma']
    return heuristics_matrix
