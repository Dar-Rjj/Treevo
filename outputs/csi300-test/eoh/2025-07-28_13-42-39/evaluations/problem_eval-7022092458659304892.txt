def heuristics_v2(df):
    df['high_avg'] = df['high'].rolling(window=15).mean()
    df['low_avg'] = df['low'].rolling(window=15).mean()
    direction_movement = df['high_avg'] / df['low_avg']
    sma_direction_movement = direction_movement.rolling(window=20).mean()
    heuristics_matrix = sma_direction_movement - direction_movement
    return heuristics_matrix
