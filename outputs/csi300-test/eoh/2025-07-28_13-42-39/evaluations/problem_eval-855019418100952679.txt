def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    
    # Calculate percentage change in closing prices
    pct_change_close = close.pct_change()
    
    # Calculate 20-day and 5-day moving averages for volume
    ma_volume_20 = volume.rolling(window=20).mean()
    ma_volume_5 = volume.rolling(window=5).mean()
    
    # Compute the ratio of 20-day moving average volume to 5-day moving average volume
    volume_ratio = ma_volume_20 / ma_volume_5
    
    # Combine the percentage change in closing prices with the volume ratio
    heuristics_matrix = pct_change_close * volume_ratio
    
    return heuristics_matrix
