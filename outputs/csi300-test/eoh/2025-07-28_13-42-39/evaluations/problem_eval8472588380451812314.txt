def heuristics_v2(df):
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['ema_typical_30'] = df['typical_price'].ewm(span=30, adjust=False).mean()
    df['low_ema_30'] = df['low'].ewm(span=30, adjust=False).mean()
    diff_ema = df['ema_typical_30'] - df['low_ema_30']
    df['volume_sma_50'] = df['volume'].rolling(window=50).mean()
    heuristics_matrix = diff_ema / df['volume_sma_50']
    return heuristics_matrix
