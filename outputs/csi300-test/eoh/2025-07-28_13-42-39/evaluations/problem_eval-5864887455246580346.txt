def heuristics_v2(df):
    df['log_diff'] = (df['high'] - df['low']).apply(np.log)
    df['rolling_mean_log_diff'] = df['log_diff'].rolling(window=20).mean()
    df['volume_ewma_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['rolling_mean_log_diff'] - df['volume_ewma_20']
    return heuristics_matrix
