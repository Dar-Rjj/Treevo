import pandas as pd

def heuristics_v2(df):
    close = df['close']
    # Calculate the 30-day exponential moving average of the closing price
    ema_30 = close.ewm(span=30, adjust=False).mean()
    # Calculate the daily price change
    price_change = close.diff(1)
    # Calculate the 30-day standard deviation of the daily price change
    vol_30 = price_change.rolling(window=30).std()
    # Final heuristic matrix
    heuristics_matrix = ema_30 + vol_30
    return heuristics_matrix
