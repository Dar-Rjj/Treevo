def heuristics_v2(df):
    df['daily_range'] = df['high'] - df['low']
    avg_daily_range = df['daily_range'].rolling(window=10).mean()
    std_close = df['close'].rolling(window=10).std()
    ratio = avg_daily_range / std_close
    heuristics_matrix = ratio.ewm(span=20, adjust=False).mean()
    return heuristics_matrix
