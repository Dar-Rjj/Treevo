def heuristics_v2(df):
    df['volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['close_sma_50'] = df['close'].rolling(window=50).mean()
    df['close_sma_200'] = df['close'].rolling(window=200).mean()
    sma_difference = df['close_sma_50'] - df['close_sma_200']
    heuristics_matrix = df['volume_ema'] * sma_difference
    return heuristics_matrix
