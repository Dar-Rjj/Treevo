def heuristics_v2(df):
    df['price_vol_ratio'] = df['close'] / df['volume']
    df['rolling_median'] = df['price_vol_ratio'].rolling(window=20).median()
    df['close_sma_10'] = df['close'].rolling(window=10).mean()
    df['high_momentum_30'] = df['high'] - df['high'].shift(30)
    heuristics_matrix = (df['rolling_median'] - df['close_sma_10']) * df['high_momentum_30']
    return heuristics_matrix
