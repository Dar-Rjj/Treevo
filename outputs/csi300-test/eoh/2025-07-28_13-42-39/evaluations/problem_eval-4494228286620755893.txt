def heuristics_v2(df):
    close = df['close']
    volume = df['volume']
    
    # Calculate the standard deviation of closing prices over a 20-day window
    std_close = close.rolling(window=20).std()
    
    # Calculate the rate of change (ROC) of volume
    roc_volume = volume.pct_change(periods=10)
    
    # Combine the standard deviation of closing prices with the ROC of volume
    heuristics_matrix = std_close * roc_volume
    
    return heuristics_matrix
