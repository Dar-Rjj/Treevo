import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day Rate of Change (ROC) of the closing price
    df['ROC_14'] = df['close'].pct_change(periods=14)
    
    # Calculate the 20-day standard deviation of the volume
    vol_std = df['volume'].rolling(window=20).std()
    
    # Final heuristic matrix
    heuristics_matrix = df['ROC_14'] + vol_std
    
    return heuristics_matrix
