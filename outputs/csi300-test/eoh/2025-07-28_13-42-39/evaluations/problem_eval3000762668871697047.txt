def heuristics_v2(df):
    close = df['close']
    amount = df['amount']
    ewma_close = close.ewm(span=20, adjust=False).mean()
    pct_change_amount = amount.pct_change()
    factor1 = close / ewma_close
    factor2 = pct_change_amount.shift(1)
    heuristics_matrix = (factor1 * factor2).dropna()
    return heuristics_matrix
