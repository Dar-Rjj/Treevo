def heuristics_v2(df):
    close = df['close']
    high = df['high']
    low = df['low']
    
    # Calculate Rate of Change (ROC) for closing price
    roc = close.pct_change(periods=14)
    
    # Calculate the difference between the 20-day high and low
    range_diff = high.rolling(window=20).max() - low.rolling(window=20).min()
    
    # Combine ROC and range difference into a single heuristics score
    heuristics_matrix = (roc * range_diff)
    
    return heuristics_matrix
