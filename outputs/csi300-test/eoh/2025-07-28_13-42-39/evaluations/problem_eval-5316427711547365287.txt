def heuristics_v2(df):
    df['volume_change'] = df['volume'].pct_change(periods=10)
    df['ewma_volume_change'] = df['volume_change'].ewm(span=20, adjust=False).mean()
    df['close_sma_50'] = df['close'].rolling(window=50).mean()
    heuristics_matrix = df['ewma_volume_change'] / df['close_sma_50']
    return heuristics_matrix
