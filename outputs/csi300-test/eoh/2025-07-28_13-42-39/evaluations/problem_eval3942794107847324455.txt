import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in volume
    df['volume_change'] = df['volume'].pct_change()
    # Calculate the price momentum over a 10-day period
    df['price_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    # Generate the alpha factor as a combination of volume change and price momentum
    heuristics_matrix = df['volume_change'] * df['price_momentum']
    return heuristics_matrix
