def heuristics_v2(df):
    df['volume_sum_10'] = df['volume'].rolling(window=10).sum()
    df['close_sma_60'] = df['close'].rolling(window=60).mean()
    df['open_ewma_20'] = df['open'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = (df['volume_sum_10'] / df['close_sma_60']) - df['open_ewma_20']
    return heuristics_matrix
