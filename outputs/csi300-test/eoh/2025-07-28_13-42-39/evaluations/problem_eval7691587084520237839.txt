def heuristics_v2(df):
    # Calculate the 20-day average volume
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate the 20-day average closing price
    avg_close_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the ratio of average volume to average closing price
    vol_price_ratio = avg_vol_20 / avg_close_20
    
    # Calculate the 60-day cumulative volume
    cum_vol_60 = df['volume'].rolling(window=60).sum()
    
    # Calculate the percentage change in the 60-day cumulative volume
    vol_change_pct = cum_vol_60.pct_change() * 100
    
    # Final heuristic matrix
    heuristics_matrix = vol_price_ratio + vol_change_pct
    
    return heuristics_matrix
