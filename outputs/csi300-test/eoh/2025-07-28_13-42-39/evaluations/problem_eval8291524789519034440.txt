def heuristics_v2(df):
    # Calculate the 20-day rolling standard deviation of the close price
    df['std_close_20'] = df['close'].rolling(window=20).std()
    
    # Calculate the 60-day EMA of the volume
    df['ema_volume_60'] = df['volume'].ewm(span=60, adjust=False).mean()
    
    # Calculate the 10-day Rate of Change (ROC) for the open price
    df['roc_open_10'] = df['open'].pct_change(periods=10)
    
    # Combine the factors into the heuristics matrix
    heuristics_matrix = (df['std_close_20'] / df['ema_volume_60']) + df['roc_open_10']
    
    return heuristics_matrix
