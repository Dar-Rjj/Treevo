def heuristics_v2(df):
    # Calculate the 20-day simple moving average of the closing price
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the momentum as the difference between the current close and the 20-day SMA
    momentum = df['close'] - df['SMA_20']
    
    # Calculate the daily returns
    daily_returns = df['close'].pct_change()
    
    # Calculate the 20-day rolling standard deviation of daily returns for volatility
    volatility = daily_returns.rolling(window=20).std()
    
    # Final heuristic matrix
    heuristics_matrix = momentum + volatility
    
    return heuristics_matrix
