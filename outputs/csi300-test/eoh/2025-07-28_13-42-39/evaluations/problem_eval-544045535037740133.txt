def heuristics_v2(df):
    df['return'] = df['close'].pct_change()
    df['price_vol_ratio'] = df['close'] / df['volume']
    df['rolling_std'] = df['price_vol_ratio'].rolling(window=20).std()
    df['return_ewma_20'] = df['return'].ewm(span=20, adjust=False).mean()
    heuristics_matrix = df['rolling_std'] * df['return_ewma_20']
    return heuristics_matrix
