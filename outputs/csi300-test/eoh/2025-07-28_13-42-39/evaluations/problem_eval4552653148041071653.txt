def heuristics_v2(df):
    df['momentum_30'] = df['close'].pct_change(periods=30)
    df['volume_std_10'] = df['volume'].rolling(window=10).std()
    df['low_ewma_14'] = df['low'].ewm(span=14, adjust=False).mean()
    heuristics_matrix = df['momentum_30'] * (df['volume_std_10'] / df['low_ewma_14'])
    return heuristics_matrix
