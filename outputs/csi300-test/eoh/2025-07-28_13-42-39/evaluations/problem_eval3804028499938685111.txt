import pandas as pd

def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the inverse of the square root of the volume for weighting
    inv_sqrt_volume = 1 / df['volume'] ** 0.5
    
    # Calculate the 21-day rolling weighted average of the daily range with weights inversely proportional to the square root of volume
    weighted_avg_range = (daily_range * inv_sqrt_volume).rolling(window=21).mean() / inv_sqrt_volume.rolling(window=21).mean()
    
    # Calculate the 30-day SMA of the closing price
    close_sma = df['close'].rolling(window=30).mean()
    
    # Final heuristic matrix
    heuristics_matrix = weighted_avg_range - close_sma
    
    return heuristics_matrix
