def heuristics_v2(df):
    df['price_vol_ratio'] = df['close'] / df['volume']
    momentum = df['price_vol_ratio'].pct_change(periods=20)
    rolling_mean = df['price_vol_ratio'].rolling(window=20).mean()
    rolling_std = df['price_vol_ratio'].rolling(window=20).std()
    coeff_of_var = rolling_std / rolling_mean
    heuristics_matrix = momentum * coeff_of_var
    return heuristics_matrix
