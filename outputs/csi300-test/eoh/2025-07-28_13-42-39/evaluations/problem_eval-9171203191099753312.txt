def heuristics_v2(df):
    df['price_diff'] = df['close'] - df['open']
    df['weighted_volume'] = df['price_diff'] * df['volume']
    df['ema_weighted_volume'] = df['weighted_volume'].ewm(span=20, adjust=False).mean()
    df['high_sma_60'] = df['high'].rolling(window=60).mean()
    heuristics_matrix = df['ema_weighted_volume'] / df['high_sma_60']
    return heuristics_matrix
