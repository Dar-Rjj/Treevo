def heuristics_v2(df):
    df['close_ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['volume_avg_10'] = df['volume'].rolling(window=10).mean()
    df['volume_avg_50'] = df['volume'].rolling(window=50).mean()
    volume_ratio = df['volume_avg_10'] / df['volume_avg_50']
    heuristics_matrix = df['close_ema_30'] * volume_ratio
    return heuristics_matrix
