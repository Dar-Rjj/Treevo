def heuristics_v2(df):
    high_low_band = df['high'].rolling(window=15).max() - df['low'].rolling(window=15).min()
    price_to_band = df['close'] / high_low_band
    vol_change = (df['volume'].pct_change() + 1).cumprod() - 1
    heuristics_matrix = price_to_band * vol_change.cumsum()
    return heuristics_matrix
