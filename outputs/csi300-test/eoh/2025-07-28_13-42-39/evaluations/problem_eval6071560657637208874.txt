def heuristics_v2(df):
    # Calculate the 20-day EMA of the trading volume
    df['EMA_20_vol'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 20-day simple moving average (SMA) and standard deviation of the closing price
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['STD_20'] = df['close'].rolling(window=20).std()
    
    # Calculate the Bollinger Bands width
    bollinger_width = df['STD_20'] / df['SMA_20']
    
    # Final heuristic matrix
    heuristics_matrix = df['EMA_20_vol'] * bollinger_width
    
    return heuristics_matrix
