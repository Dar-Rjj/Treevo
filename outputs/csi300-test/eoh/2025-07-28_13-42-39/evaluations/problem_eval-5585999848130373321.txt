import pandas as pd

def heuristics_v2(df):
    # Calculate the 20-day and 50-day simple moving averages of the closing price
    df['MA_20'] = df['close'].rolling(window=20).mean()
    df['MA_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the difference between the 20-day and 50-day moving averages
    ma_diff = df['MA_20'] - df['MA_50']
    
    # Calculate the True Range
    df['true_range'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    
    # Calculate the Average True Range (ATR) over a 14-day period
    atr = df['true_range'].rolling(window=14).mean()
    
    # Final heuristic matrix
    heuristics_matrix = ma_diff + atr
    
    return heuristics_matrix
