def heuristics_v2(df):
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Compute the inverse of the cube root of the volume for weighting
    inv_cbrt_volume = 1 / df['volume'] ** (1/3)
    
    # Calculate the 14-day rolling weighted average of the daily range with weights inversely proportional to the cube root of volume
    weighted_avg_range = (daily_range * inv_cbrt_volume).rolling(window=14).mean() / inv_cbrt_volume.rolling(window=14).mean()
    
    # Calculate the 20-day SMA of the closing price
    close_sma = df['close'].rolling(window=20).mean()
    
    # Final heuristic matrix
    heuristics_matrix = weighted_avg_range - close_sma
    
    return heuristics_matrix
