def heuristics_v2(df):
    df['volume_close_ratio'] = df['volume'] / df['close']
    df['rolling_std'] = df['volume_close_ratio'].rolling(window=20).std()
    df['high_ewma_60'] = df['high'].ewm(span=60, adjust=False).mean()
    heuristics_matrix = df['high_ewma_60'] - df['rolling_std']
    return heuristics_matrix
