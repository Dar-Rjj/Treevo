```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['price_momentum'] = df['close'].diff(10)
    
    # Calculate High-Low Price Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Intraday Close-Open Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Spread and Close-Open Return
    df['intraday_momentum'] = df['high_low_spread'] * df['intraday_return']
    
    # Calculate Volume Surprise
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] - df['volume_ma_10']
    
    # Combine Price Momentum and Volume Surprise
    df['intermediate_alpha_factor'] = df['price_momentum'] * df['volume_surprise']
    
    # Calculate Volume Surge
    df['volume_surge'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df[df['close'] > df['open']]['volume']
    df['downward_volume'] = df[df['close'] < df['open']]['volume']
    df['upward_volume'] = df['upward_volume'].fillna(0).cumsum()
    df['downward_volume'] = df['downward_volume'].fillna(0).cumsum()
    df['volume_influence_ratio'] = df['upward_volume'] / df['downward_volume']
    
    # Multiply High-Low Spread with Volume Influence Ratio
    df['volume_influence_high_low'] = df['high_low_spread'] * df['volume_influence_ratio']
    
    # Combine All Alpha Factors
    df['final_alpha_factor'] = (
        (df['intermediate_alpha_factor'] * df['volume_influence_high_low']) +
        df['intraday_momentum'] * df['volume_influence_ratio'] +
        df['volume_surge']
    )
    
    return df['final_alpha_factor']
```
