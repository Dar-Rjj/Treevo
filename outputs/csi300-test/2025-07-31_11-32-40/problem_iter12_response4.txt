```python
import pandas as pd

def heuristics(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    volume_diff = df['volume'].diff()
    open_interest_adjustment = 1 + volume_diff / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Alpha Factor Base
    alpha_factor_base = adjusted_intraday_reversal * close_to_midpoint_deviation + day_to_day_open_change
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Surge
    volume_surge = volume_diff
    
    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    ma5_intraday_return = intraday_close_open_return.rolling(window=5).mean()
    
    # Calculate 5-Day Moving Average of Intraday High-Low Spread
    ma5_high_low_spread = intraday_high_low_spread.rolling(window=5).mean()
    
    # Calculate Weighted Intraday Return
    weighted_intraday_return = 0.6 * intraday_close_open_return + 0.4 * intraday_high_low_spread
    
    # Combine Weighted Intraday Return, Moving Averages, and Volume Surge
    intraday_momentum = (weighted_intraday_return * ma5_intraday_return +
                         intraday_high_low_spread * ma5_high_low_spread) * volume_surge
    
    # Final Alpha Factor
    combined_intraday_reversal_momentum = alpha_factor_base + intraday_momentum + volume_surge
    
    return combined_intraday_reversal_momentum
```
