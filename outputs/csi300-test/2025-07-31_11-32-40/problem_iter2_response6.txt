def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    close_to_mid_dev = df['close'] - intraday_midpoint
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Return
    prev_day_close = df['close'].shift(1)
    prev_day_return = (df['open'] - prev_day_close) / prev_day_close
    
    # Combine Metrics
    combined_metrics = (intraday_range * close_to_mid_dev) + (high_low_spread * close_open_return)
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Generate Alpha Factor
    alpha_factor = combined_metrics - (prev_day_return * combined_metrics) + volume_surge
    
    return alpha_factor.rename("Intraday Reversal Momentum and Volume Surge Indicator")
