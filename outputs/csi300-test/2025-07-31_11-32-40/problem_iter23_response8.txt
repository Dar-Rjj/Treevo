```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume-Weighted Price Movement
    upward_days = (df['close'] > df['open'])
    downward_days = (df['close'] < df['open'])
    
    # Accumulate Volume-Price Product on Upward Days
    upward_product = (df['volume'] * (df['close'] - df['open'])) * upward_days
    upward_sum = upward_product.rolling(window=20, min_periods=1).sum()
    
    # Accumulate Volume-Price Product on Downward Days
    downward_product = (df['volume'] * (df['close'] - df['open'])) * downward_days
    downward_sum = downward_product.rolling(window=20, min_periods=1).sum()
    
    # Divide Upward Volume-Price Product by Downward Volume-Price Product
    volume_weighted_movement = upward_sum / downward_sum.abs()
    
    # Combine High-Low Difference with Volume-Weighted Price Movement
    alpha_factor = high_low_diff * volume_weighted_movement
    
    return alpha_factor
```
