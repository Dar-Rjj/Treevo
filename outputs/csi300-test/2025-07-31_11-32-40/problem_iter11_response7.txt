```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    price_momentum = df['close'].diff(10)
    
    # Calculate Volume Surprise
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_ma_10
    
    # Combine Price Momentum and Volume Surprise
    combined_momentum_volume = price_momentum * volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = (upward_volume / downward_volume) if downward_volume != 0 else 0
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha = combined_momentum_volume * high_low_diff
    
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    open_interest_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment
    
    # Calculate Intraday High-Low Difference
    intraday_high_low_diff = df['high'] - df['low']
    
    # Volume Weighted Close-Open Change
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open = close_open_diff * df['volume']
    
    # Combine Intraday Metrics
    combined_intraday_metrics = intraday_high_low_diff + volume_weighted_close_open
    
    # Moving Average Comparison
    moving_avg_5 = combined_intraday_metrics.rolling(window=5).mean()
    moving_avg_score = (combined_intraday_metrics > moving_avg_5).astype(int) * 2 - 1
    
    # Consider Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Combine Metrics
    combined_metrics = intraday_range + close_to_midpoint_deviation + day_to_day_open_change
    
    # Final Alpha Factor
    final_alpha = (
        intermediate_alpha * volume_influence_ratio +
        adjusted_intraday_reversal * combined_metrics +
        combined_momentum_volume * intraday_range * volume_influence_ratio
    )
    
    return final_alpha
```
