```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Day-to-Day Open Price Change
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Preliminary Alpha Factor
    previous_day_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation +
                                day_to_day_open_price_change - previous_day_return)
    
    # Calculate Daily Volume Surprise
    volume_10_ma = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_10_ma
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(10)
    combined_price_momentum_volume_surprise = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume']
    downward_volume = df[df['close'] < df['open']]['volume']
    accumulated_upward_volume = upward_volume.rolling(window=10, min_periods=1).sum()
    accumulated_downward_volume = downward_volume.rolling(window=10, min_periods=1).sum()
    volume_influence_ratio = accumulated_upward_volume / accumulated_downward_volume.fillna(1)  # Avoid division by zero
    
    # Intermediate Alpha Factor Synthesis
    high_low_price_difference = df['high'] - df['low']
    intermediate_alpha_factor = combined_price_momentum_volume_surprise * high_low_price_difference
    
    # Final Alpha Factor
    final_alpha_factor = (preliminary_alpha_factor * intermediate_alpha_factor *
                          volume_influence_ratio).fillna(0)  # Fill NaN values with 0
    
    return final_alpha_factor
```
