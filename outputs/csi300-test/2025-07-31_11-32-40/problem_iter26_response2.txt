```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = (1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)).fillna(1)
    volume_change_adjustment = (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)).fillna(1)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment * volume_change_adjustment

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Generate Alpha Factor Base
    alpha_factor_base = adjusted_intraday_reversal * close_to_midpoint_deviation
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    alpha_factor_base += day_to_day_open_price_change

    # Calculate Daily Volume Surprise
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    daily_volume_surprise = df['volume'] - avg_volume_5d

    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']

    # Calculate 5-Day Moving Averages
    avg_intraday_close_open_return_5d = intraday_close_open_return.rolling(window=5).mean()
    avg_intraday_high_low_spread_5d = intraday_high_low_spread.rolling(window=5).mean()

    # Calculate Weighted Intraday Return
    weighted_intraday_return = 0.6 * intraday_close_open_return + 0.4 * intraday_high_low_spread

    # Combine Weighted Intraday Return and Moving Averages
    intraday_momentum = (weighted_intraday_return * avg_intraday_close_open_return_5d) + \
                        (intraday_high_low_spread * avg_intraday_high_low_spread_5d)

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume']
    downward_volume = df[df['close'] < df['open']]['volume']
    volume_influence_ratio = upward_volume.cumsum() / downward_volume.cumsum().replace(0, 1)

    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(5)
    combined_price_momentum_volume_surprise = price_momentum * daily_volume_surprise

    # Smoothed Price Momentum
    smoothed_price_momentum = (df['close'] - df['close'].shift(1)).ewm(span=5).mean()

    # Adjust by Smoothed Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1)).ewm(span=5).mean()
    smoothed_combined_momentum = smoothed_price_momentum * volume_ratio

    # Final Alpha Factor Components
    final_alpha_factor_components = intraday_momentum * combined_price_momentum_volume_surprise * volume_influence_ratio + \
                                   alpha_factor_base + (df['volume'] - df['volume'].shift(1))

    return final_alpha_factor_components
```
