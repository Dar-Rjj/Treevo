import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] - df['open']) / df['open'] > 0.005
    downward_volume = (df['open'] - df['close']) / df['open'] > 0.005
    
    accumulated_upward_volume = df.loc[upward_volume, 'volume'].rolling(window=21).sum()
    accumulated_downward_volume = df.loc[downward_volume, 'volume'].rolling(window=21).sum()
    
    volume_influence_ratio = (accumulated_upward_volume + 1) / (accumulated_downward_volume + 1)
    
    # Adjusted Cumulative Directional Volume Impact
    adjusted_cumulative_impact = high_low_diff * volume_influence_ratio
    
    # Apply Weighted Moving Average
    sma_high_low_diff = high_low_diff.rolling(window=21).mean()
    weighted_moving_average = sma_high_low_diff * volume_influence_ratio
    
    return weighted_moving_average
