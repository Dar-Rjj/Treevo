import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']).astype(int) * df['volume']
    downward_volume = (df['close'] < df['open']).astype(int) * df['volume']
    volume_influence_ratio = upward_volume.rolling(window=10).sum() / downward_volume.rolling(window=10).sum()
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_volume_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    daily_volume_change = df['volume'] - volume_moving_avg
    volume_momentum_strength = (daily_volume_change > 0).astype(int) * 2 - 1  # +1 for positive, -1 for negative
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor_1 = high_low_volume_influence * intraday_return * volume_momentum_strength
    
    # Adjust for Intraday Reversal with Momentum
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Adjust for Open Interest
    open_interest_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal *= open_interest_adjustment
    
    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - volume_moving_avg
    
    # Calculate Daily Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Price Momentum and Volume Surprise
    combined_price_momentum = price_momentum * daily_volume_surprise
    
    # Integrate Intraday Range and Volume Dynamics
    intraday_range = df['high'] - df['low']
    integrated_intraday_dynamics = intraday_range * df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = (combined_factor_1 * 
                          adjusted_intraday_reversal * 
                          integrated_intraday_dynamics * 
                          combined_price_momentum)
    
    return final_alpha_factor
