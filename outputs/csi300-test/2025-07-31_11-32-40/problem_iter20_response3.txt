```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Cumulative Directional Price Impact Adjusted by Volume.
    
    Parameters:
    - df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    - pd.Series: A series with the calculated factor values indexed by date.
    """
    
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Directional Adjusted Volume
    upward_volume = ((df['close'] > df['open']) * (df['close'] - df['open']) * df['volume']).cumsum()
    downward_volume = ((df['close'] < df['open']) * (df['open'] - df['close']) * df['volume']).cumsum()
    net_directional_volume = upward_volume - downward_volume
    
    # Multiply High-Low Difference with Net Directional Adjusted Volume
    factor = high_low_diff * net_directional_volume
    
    return factor
```
