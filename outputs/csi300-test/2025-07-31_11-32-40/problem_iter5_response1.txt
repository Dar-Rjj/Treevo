import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Difference
    intraday_high_low_diff = df['high'] - df['low']
    
    # Volume Weighted Close-Open Change
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open_change = close_open_diff * df['volume']
    
    # Combine Factors
    combined_factor = intraday_high_low_diff + volume_weighted_close_open_change
    
    # 5-day Moving Average of Combined Factor
    moving_avg_5d = combined_factor.rolling(window=5).mean()
    
    # Compare with Current Day Factor
    current_day_factor = combined_factor
    score = (current_day_factor > moving_avg_5d).astype(int) - (current_day_factor < moving_avg_5d).astype(int)
    
    # Calculate Daily Price Momentum
    price_momentum = df['close'].diff(periods=10)
    
    # Calculate Volume Surprise
    volume_moving_avg_10d = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_moving_avg_10d
    
    # Combine Price Momentum and Volume Surprise
    price_vol_combined = price_momentum * volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume if downward_volume != 0 else 0
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (price_vol_combined * high_low_diff) * volume_influence_ratio
    
    return final_alpha_factor
