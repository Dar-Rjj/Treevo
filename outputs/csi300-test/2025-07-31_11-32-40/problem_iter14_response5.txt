```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - midpoint
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Calculate Previous Day Return
    previous_day_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (intraday_range * close_to_midpoint_deviation) - previous_day_return
    
    # Calculate Daily Volume Surprise
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_10day_ma
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(10)
    combined_momentum_volume_surprise = price_momentum * daily_volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_difference = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['volume'] * (df['close'] > df['open']).astype(int)).rolling(window=10).sum()
    downward_volume = (df['volume'] * (df['close'] < df['open']).astype(int)).rolling(window=10).sum()
    volume_influence_ratio = upward_volume / downward_volume
    
    # Adjust Momentum by Volume and Amount
    weighted_high_low_return = (high_low_difference * df['volume']) * volume_influence_ratio
    volume_adjusted_momentum = weighted_high_low_return.rolling(window=10).sum()
    
    # Introduce Time Decay
    decay_rate = 0.95
    time_decay = (decay_rate ** (10 - 1))
    finalized_volume_adjusted_momentum = volume_adjusted_momentum * time_decay
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_synthesis = combined_momentum_volume_surprise * high_low_difference
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (df['close'] * df['volume']) / df['volume'].sum()
    
    # Generate Alpha Factor
    alpha_factor = (intraday_range * close_to_midpoint_deviation) + volume_weighted_price
    alpha_factor += (high_low_difference * volume_influence_ratio) * (combined_momentum_volume_surprise + finalized_volume_adjusted_momentum)
    alpha_factor -= previous_day_return
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor * volume_influence_ratio * alpha_factor
    
    return final_alpha_factor
```
