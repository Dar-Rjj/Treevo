```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    open_interest_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment
    
    # Calculate Daily Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Surprise
    volume_moving_average = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_moving_average
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Consider Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Metrics
    combined_metrics = (intraday_range * close_to_midpoint_deviation * open_price_change).fillna(0)
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].rolling(window=10).sum()
    downward_volume = df[df['close'] < df['open']]['volume'].rolling(window=10).sum()
    volume_influence_ratio = (upward_volume / downward_volume).fillna(1)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (adjusted_intraday_reversal * combined_metrics) * (price_momentum * volume_surprise * intraday_range * volume_influence_ratio)
    
    return final_alpha_factor
```
