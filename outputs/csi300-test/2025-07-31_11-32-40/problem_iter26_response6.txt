```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Net Directional Adjusted Volume
    upward_volume = (df['close'] > df['open']) * (df['close'] - df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * (df['open'] - df['close']) * df['volume']
    net_directional_volume = upward_volume - downward_volume
    
    # Multiply High-Low Difference with Net Directional Adjusted Volume
    directional_price_impact = high_low_diff * net_directional_volume
    
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * (1 + volume_change)
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + open_price_change
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Moving Averages
    ma_close_open_return = intraday_close_open_return.rolling(window=5).mean()
    ma_high_low_spread = intraday_high_low_spread.rolling(window=5).mean()
    
    # Calculate Weighted Intraday Return
    weighted_intraday_return = (intraday_close_open_return * 0.7) + (intraday_high_low_spread * 0.3)
    
    # Combine Weighted Intraday Return and Moving Averages
    intraday_momentum = (weighted_intraday_return * ma_close_open_return) + (intraday_high_low_spread * ma_high_low_spread)
    
    # Calculate Price Momentum with EMA
    lookback_period = 20
    price_momentum = df['close'].ewm(span=lookback_period, adjust=False).mean()
    
    # Adjust by Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1)).rolling(window=lookback_period).mean()
    adjusted_momentum = price_momentum * volume_ratio
    
    # Compute Volume Influence Ratio
    upward_volume_sum = (df['close'] > df['open']) * df['volume']
    downward_volume_sum = (df['close'] < df['open']) * df['volume']
    volume_influence_ratio = upward_volume_sum.rolling(window=lookback_period).sum() / downward_volume_sum.rolling(window=lookback_period).sum()
    
    # Integrate Core Components into Intermediate Alpha Factor
    intermediate_alpha_factor = adjusted_momentum * (intraday_high_low_spread * volume_influence_ratio * intraday_close_open_return)
    
    # Final Alpha Factor Components
    final_alpha_components = intraday_momentum * volume_influence_ratio + adjusted_intraday_reversal + open_price_change
    
    # Final Alpha Factor
    final_alpha_factor = final_alpha_components + (df['volume'] - df['volume'].shift(1))
    
    # Integrate Directional Price Impact
    final_alpha_factor += directional_price_impact
    
    return final_alpha_factor
```
