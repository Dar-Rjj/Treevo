```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - midpoint
    
    # Calculate Previous Day Return
    previous_day_return = df['open'] - df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (intraday_range * close_to_midpoint_deviation) - previous_day_return
    
    # Calculate High-Low Price Difference
    high_low_price_difference = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * df['volume']
    volume_influence_ratio = upward_volume.rolling(window=20).sum() / downward_volume.rolling(window=20).sum()
    
    # Adjust Momentum by Volume and Amount
    weighted_high_low_return = high_low_price_difference * df['volume'] * volume_influence_ratio
    volume_adjusted_momentum = weighted_high_low_return.rolling(window=20).sum()
    
    # Introduce Time Decay
    decay_rate = 0.95
    volume_adjusted_momentum_decay = volume_adjusted_momentum * (decay_rate ** (pd.Series(range(len(df)), index=df.index)))
    
    # Generate Final Alpha Factor
    final_alpha_factor = intermediate_alpha + volume_adjusted_momentum_decay
    
    return final_alpha_factor
```
