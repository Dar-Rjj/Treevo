```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    lookback_period = 10
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    df['smoothed_momentum'] = df['price_momentum'].ewm(span=lookback_period, adjust=False).mean()

    # Adjust by Volume Change
    volume_lookback_period = 10
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=volume_lookback_period, min_periods=1).mean()
    df['combined_momentum_volume'] = df['smoothed_momentum'] * df['smoothed_volume_ratio']

    # Calculate High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']

    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    total_upward_volume = df['upward_volume'].sum()
    total_downward_volume = df['downward_volume'].sum()
    df['volume_influence_ratio'] = total_upward_volume / total_downward_volume

    # Multiply High-Low Difference with Volume Influence Ratio
    df['hl_volume_influence'] = df['high_low_diff'] * df['volume_influence_ratio']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Close to Midpoint Deviation
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['close_midpoint_deviation'] = df['close'] - df['midpoint']

    # Calculate Volume Weighted Price
    df['volume_weighted_price'] = (df['close'] * df['volume']) / df['volume'].sum()

    # Generate Alpha Factor
    df['alpha_factor'] = (
        df['intraday_range'] * df['close_midpoint_deviation'] +
        df['volume_weighted_price'] +
        df['hl_volume_influence'] * df['combined_momentum_volume'] -
        (df['open'] - df['close'].shift(1))
    )

    return df['alpha_factor']
```
