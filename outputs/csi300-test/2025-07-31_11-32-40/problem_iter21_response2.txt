```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = (upward_volume / downward_volume).fillna(1)
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    df['volume_10_day_ma'] = df['volume'].rolling(window=10).mean()
    daily_volume_change = df['volume'] - df['volume_10_day_ma']
    volume_momentum_strength = daily_volume_change.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor = high_low_vol_influence * intraday_return * volume_momentum_strength
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment.fillna(1)
    
    # Adjust for Open Interest
    open_interest_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal *= open_interest_adjustment.fillna(1)
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + day_to_day_open_change
    
    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - df['volume_10_day_ma']
    
    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Price Momentum and Volume Surprise
    combined_price_vol = daily_price_momentum * daily_volume_surprise
    
    # Integrate Intraday Range and Volume Dynamics
    intraday_range_vol_dynamics = (df['high'] - df['low']) * df['volume']
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_price_vol * intraday_range_vol_dynamics
    
    # Final Alpha Factor
    final_alpha_factor = (preliminary_alpha_factor * intermediate_alpha_factor * volume_influence_ratio).fillna(0)
    
    return final_alpha_factor
```
