```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    volume_change = (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1))
    adjusted_intraday_reversal = intraday_reversal * volume_change

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Calculate Day-to-Day Open Price Change
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + day_to_day_open_price_change

    # Calculate Price Momentum
    lookback_period = 20
    price_momentum = df['close'].pct_change().ewm(span=lookback_period, adjust=False).mean()

    # Adjust by Volume Change
    volume_lookback_period = 20
    volume_ratio = df['volume'].rolling(window=volume_lookback_period).mean() / df['volume'].shift(1).rolling(window=volume_lookback_period).mean()
    combined_momentum_volume = price_momentum * volume_ratio

    # Compute High-Low Price Difference
    high_low_price_difference = df['high'] - df['low']

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume']
    downward_volume = df[df['close'] < df['open']]['volume']
    upward_volume_sum = upward_volume.rolling(window=lookback_period).sum()
    downward_volume_sum = downward_volume.rolling(window=lookback_period).sum()
    volume_influence_ratio = upward_volume_sum / downward_volume_sum

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Integrate Core Components into Intermediate Alpha Factor
    intermediate_alpha_factor = combined_momentum_volume * (high_low_price_difference * volume_influence_ratio * intraday_return)

    # Combine High-Low Spread and Close-Open Return
    high_low_price_difference = df['high'] - df['low']
    close_open_return = (df['close'] - df['open']) / df['open']
    intraday_momentum = high_low_price_difference * close_open_return

    # Final Alpha Factor
    final_alpha_factor = (
        intermediate_alpha_factor +
        intraday_momentum * volume_influence_ratio +
        adjusted_intraday_reversal +
        day_to_day_open_price_change +
        (df['volume'] - df['volume'].shift(1))
    )

    return final_alpha_factor
```
