```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['price_diff'] = df['close'] - df['low'].shift(1)
    ema_smooth_factor = 10
    df['smoothed_momentum'] = df['price_diff'].ewm(span=ema_smooth_factor, adjust=False).mean()
    
    # Combine Adjusted Momentum and Simple Price Momentum
    lookback_period = 10
    df['simple_momentum'] = df['close'] - df['close'].shift(lookback_period)
    df['combined_momentum'] = df['smoothed_momentum'] + df['simple_momentum']
    
    # Adjust by Volume Change
    df['vol_ma_5'] = df['volume'].rolling(window=5).mean()
    df['vol_ratio'] = df['volume'] / df['vol_ma_5']
    sma_length = 10
    df['smoothed_vol_ratio'] = df['vol_ratio'].rolling(window=sma_length).mean()
    df['adjusted_momentum'] = df['combined_momentum'] * df['smoothed_vol_ratio']
    
    # Compute Volume Influence Ratio
    df['up_volume'] = df.apply(lambda x: x['volume'] if x['close'] > x['open'] else 0, axis=1).rolling(window=lookback_period).sum()
    df['down_volume'] = df.apply(lambda x: x['volume'] if x['close'] < x['open'] else 0, axis=1).rolling(window=lookback_period).sum()
    df['vol_influence_ratio'] = (df['up_volume'] / df['down_volume']).fillna(1)
    df['cbrt_vol_influence_ratio'] = df['vol_influence_ratio'] ** (1/3)
    
    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    df['integrated_momentum'] = df['adjusted_momentum'] * df['cbrt_vol_influence_ratio']
    
    # Calculate Adjusted Intraday Reversal
    df['intraday_reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    df['momentum_adjustment'] = 1 + (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    df['adjusted_intraday_reversal'] = df['intraday_reversal'] * df['momentum_adjustment']
    
    # Generate Preliminary Alpha Factor
    df['close_to_midpoint_deviation'] = (df['close'] - (df['high'] + df['low']) / 2) / ((df['high'] + df['low']) / 2)
    df['day_to_day_open_change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['preliminary_alpha'] = df['adjusted_intraday_reversal'] * df['close_to_midpoint_deviation'] + df['day_to_day_open_change']
    
    # Calculate Daily Volume Surprise
    df['vol_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] - df['vol_ma_10']
    
    # Combine Price Momentum and Volume Surprise
    df['price_momentum'] = df['close'] - df['close'].shift(lookback_period)
    df['combined_momentum_vol_surprise'] = df['price_momentum'] * df['volume_surprise']
    
    # Final Alpha Factor
    df['final_alpha'] = df['preliminary_alpha'] * df['combined_momentum_vol_surprise'] * df['cbrt_vol_influence_ratio']
    
    # Integrate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    df['final_alpha'] = df['final_alpha'] * df['high_low_diff']
    
    return df['final_alpha']
```
