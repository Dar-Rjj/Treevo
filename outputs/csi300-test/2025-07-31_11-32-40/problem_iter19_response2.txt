```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df[df['close'] > df['open']]['volume']).sum()
    downward_volume = (df[df['close'] < df['open']]['volume']).sum()
    volume_influence_ratio = upward_volume / (downward_volume + 1e-6)  # Avoid division by zero
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    df['vol_10day_ma'] = df['volume'].rolling(window=10).mean()
    daily_vol_change = df['volume'] - df['vol_10day_ma']
    volume_momentum_strength = daily_vol_change.apply(lambda x: 1 if x > 0 else 0.5)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_hl_vol_ret = high_low_vol_influence * intraday_return * volume_momentum_strength
    
    # Calculate Daily Price Reversal
    close_diff = df['close'].diff()
    prev_close_diff = df['close'].shift(1).diff()
    price_reversal = -close_diff * (prev_close_diff > 0).apply(lambda x: -1 if x else 1)
    
    # Calculate Volume Surprise
    vol_surprise = df['volume'] - df['vol_10day_ma']
    combined_price_vol = price_reversal * vol_surprise
    
    # Integrate Short-Term Price Volatility
    close_5day_std = df['close'].rolling(window=5).std()
    final_factor = (combined_hl_vol_ret * combined_price_vol) / (close_5day_std + 1e-6)  # Avoid division by zero
    
    return final_factor
```
