```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    price_momentum = df['close'].diff(periods=10)
    
    # Calculate Volume Surprise
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_ma_10
    
    # Combine Price Momentum and Volume Surprise
    combined_price_volume = price_momentum * volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = (upward_volume / downward_volume) if downward_volume != 0 else 0
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha = combined_price_volume * high_low_diff
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_change = df['volume'] - volume_ma_10
    volume_momentum_strength = daily_volume_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_high_low_volume = (high_low_diff * volume_influence_ratio) * intraday_return * volume_momentum_strength
    
    # Calculate Daily Price Reversal
    daily_price_reversal = df['close'].diff(periods=1)
    daily_price_reversal = daily_price_reversal * (df['close'].shift(1) < df['close'].shift(2)).map({True: -1, False: 1})
    
    # Adjust for Open Interest
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_intraday_reversal = daily_price_reversal * (1 + volume_change / df['volume'].shift(1))
    
    # Calculate Intraday High-Low Difference
    intraday_high_low_diff = df['high'] - df['low']
    
    # Volume Weighted Close-Open Change
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open = close_open_diff * df['volume']
    
    # Combine Intraday Metrics
    combined_intraday_metrics = intraday_high_low_diff + volume_weighted_close_open
    
    # Moving Average Comparison
    ma_5_combined_intraday = combined_intraday_metrics.rolling(window=5).mean()
    ma_comparison_score = (combined_intraday_metrics > ma_5_combined_intraday).apply(lambda x: 1 if x else -1)
    
    # Consider Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    close_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Combine Metrics
    combined_metrics = intraday_range * close_midpoint_deviation * day_to_day_open_change
    
    # Final Alpha Factor
    final_alpha_factor = (
        (intermediate_alpha * volume_influence_ratio) 
        * (adjusted_intraday_reversal * combined_metrics) 
        * (price_momentum * volume_surprise) 
        * (intraday_range * volume_influence_ratio)
    )
    
    return final_alpha_factor
```
