import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'].pct_change(10))
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha = adjusted_intraday_reversal * close_to_midpoint_deviation
    preliminary_alpha += (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Adjusted Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(10)
    
    # Adjust by High-Low Spread
    high_low_spread = df['high'] - df['low']
    ema_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()
    adjusted_daily_momentum = daily_momentum * ema_high_low_spread
    
    # Incorporate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    combined_adjusted_momentum = adjusted_daily_momentum + intraday_return
    
    # Calculate Daily Volume Surprise
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_10_day_ma
    
    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    smoothed_volume_ratio = volume_ratio.ewm(span=10, adjust=False).mean()
    combined_volume_impact = daily_volume_surprise * smoothed_volume_ratio
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(10)
    combined_price_momentum_volume = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = df.loc[df['close'] > df['open'], 'volume'].sum()
    downward_volume = df.loc[df['close'] < df['open'], 'volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha = (combined_price_momentum_volume * (df['high'] - df['low'])) * (combined_adjusted_momentum * combined_volume_impact)
    
    # Final Alpha Factor
    final_alpha = (preliminary_alpha * intermediate_alpha * volume_influence_ratio) + intraday_return
    final_alpha *= volume_influence_ratio
    final_alpha += df['volume'].pct_change()
    
    return final_alpha.dropna()
