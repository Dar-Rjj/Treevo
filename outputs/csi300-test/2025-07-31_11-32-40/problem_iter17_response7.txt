```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Reversal: 2 * (High - Low) / (Close + Open)
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])

    # Adjust for Momentum and Volume Change
    momentum_change = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * (1 + momentum_change) * (1 + volume_change)

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Generate Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday Reversal, Close to Midpoint Deviation, and Open Price Change
    combined_reversal_midpoint_open = (adjusted_intraday_reversal * close_to_midpoint_deviation) + open_price_change

    # Calculate Daily Volume Surprise
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_moving_avg

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume']
    downward_volume = df[df['close'] < df['open']]['volume']
    cumulative_upward_volume = upward_volume.rolling(window=len(df), min_periods=1).sum()
    cumulative_downward_volume = downward_volume.rolling(window=len(df), min_periods=1).sum()
    volume_influence_ratio = cumulative_upward_volume / cumulative_downward_volume

    # Calculate Smoothed Price Momentum
    lookback_period = 10
    price_momentum = (df['close'] - df['close'].shift(lookback_period)).ewm(span=lookback_period).mean()

    # Adjust by Smoothed Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1)).rolling(window=lookback_period, min_periods=1).mean()
    smoothed_momentum_volume = price_momentum * volume_ratio

    # Compute Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']

    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    intraday_return_5d_ma = intraday_close_open_return.rolling(window=5, min_periods=1).mean()

    # Combine High-Low Spread and Close-Open Return
    high_low_spread = df['high'] - df['low']
    intraday_momentum = high_low_spread * intraday_close_open_return

    # Integrate Intraday Components
    integrated_intraday_factor = (intraday_momentum * intraday_return_5d_ma * volume_surprise)

    # Final Alpha Factor
    previous_day_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    alpha_factor = (combined_reversal_midpoint_open * integrated_intraday_factor * volume_influence_ratio +
                    adjusted_intraday_reversal + smoothed_momentum_volume - previous_day_return)

    return alpha_factor
```
