```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    open_interest_adjustment = 1 + volume_change
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment
    
    # Consider Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Reversal and Open Interest Adjustments
    combined_reversal = adjusted_intraday_reversal + day_to_day_open_change
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    ema_smoothing_factor = 0.9
    smoothed_price_momentum = price_momentum.ewm(alpha=ema_smoothing_factor).mean()
    
    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    sma_length = 5
    smoothed_volume_ratio = volume_ratio.rolling(window=sma_length).mean()
    adjusted_price_momentum = smoothed_price_momentum * smoothed_volume_ratio
    
    # Final Alpha Factor
    weight_reversal = 0.6
    weight_momentum = 0.4
    alpha_factor = (weight_reversal * combined_reversal) + (weight_momentum * adjusted_price_momentum)
    
    return alpha_factor
```
