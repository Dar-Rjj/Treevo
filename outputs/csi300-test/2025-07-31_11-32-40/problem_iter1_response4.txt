import pandas as pd

def heuristics_v2(df, price_lookback=10, volume_lookback=10, ema_span=5, sma_window=5):
    # Calculate Price Momentum
    df['price_diff'] = df['close'] - df['close'].shift(1)
    df['price_momentum'] = df['price_diff'].ewm(span=ema_span, adjust=False).mean()
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=sma_window).mean()
    
    # Combine Momentum and Volume Ratio
    df['alpha_factor'] = df['price_momentum'] * df['smoothed_volume_ratio']
    
    return df['alpha_factor']
