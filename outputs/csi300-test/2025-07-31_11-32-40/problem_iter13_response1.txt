```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Close-Open Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['upward_volume_cum'] = df['upward_volume'].rolling(window=len(df), min_periods=1).sum()
    df['downward_volume_cum'] = df['downward_volume'].rolling(window=len(df), min_periods=1).sum()
    df['volume_influence_ratio'] = df['upward_volume_cum'] / df['downward_volume_cum']
    
    # Combine High-Low Spread and Close-Open Return
    df['high_low_spread'] = df['high'] - df['low']
    df['intraday_momentum'] = df['high_low_spread'] * df['intraday_return']
    
    # Calculate Intraday Reversal
    df['intraday_reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['adjusted_intraday_reversal'] = df['intraday_reversal'] * (1 + df['volume_change'])
    
    # Consider Day-to-Day Open Price Change
    df['day_to_day_open_price_change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor Components
    df['alpha_factor'] = (df['intraday_momentum'] * df['volume_influence_ratio'] 
                          + df['adjusted_intraday_reversal'] 
                          + df['day_to_day_open_price_change'])
    
    # Calculate Moving Average
    df['5_day_ma'] = df['alpha_factor'].rolling(window=5, min_periods=1).mean()
    
    # Compare with Current Day Factor
    df['alpha_score'] = df.apply(lambda row: 1 if row['alpha_factor'] > row['5_day_ma'] else -1, axis=1)
    
    return df['alpha_score']
```
