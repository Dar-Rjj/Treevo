```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    df['close_to_midpoint_deviation'] = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    df['adjusted_intraday_reversal'] = (2 * (df['high'] - df['low']) / (df['close'] + df['open'])) * (1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10))
    
    # Calculate Previous Day Return
    df['previous_day_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor 1
    df['intermediate_alpha_factor_1'] = (df['intraday_range'] * df['close_to_midpoint_deviation']) - df['previous_day_return']
    
    # Calculate Daily Volume Surprise
    df['daily_volume_surprise'] = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surprise
    df['price_momentum_and_volume_surprise'] = (df['close'] - df['close'].shift(10)) * df['daily_volume_surprise']
    
    # Calculate High-Low Price Difference
    df['high_low_price_difference'] = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    volume_up = df[df['close'] > df['open']]['volume'].sum()
    volume_down = df[df['close'] < df['open']]['volume'].sum()
    df['volume_influence_ratio'] = volume_up / volume_down if volume_down != 0 else np.nan
    
    # Adjust Momentum by Volume and Amount
    df['adjusted_momentum'] = (df['high'] - df['low']) * df['volume'] * df['volume_influence_ratio']
    
    # Introduce Time Decay
    decay_rate = 0.1
    time = np.arange(len(df))
    df['adjusted_momentum_with_decay'] = df['adjusted_momentum'] * np.exp(-decay_rate * time)
    
    # Incorporate Volume Volatility
    df['volume_volatility'] = df['volume'].rolling(window=10).std()
    
    # Adjust Volume Influence Ratio by Volume Volatility
    df['adjusted_volume_influence_ratio'] = df['volume_influence_ratio'] * (1 + df['volume_volatility'])
    
    # Intermediate Alpha Factor Synthesis
    df['intermediate_alpha_factor_synthesis'] = (df['price_momentum_and_volume_surprise'] * df['high_low_price_difference']) * df['adjusted_volume_influence_ratio']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['intermediate_alpha_factor_1'] * df['intermediate_alpha_factor_synthesis'])
    
    return df['final_alpha_factor']
```
