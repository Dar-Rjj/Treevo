```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha = (intraday_reversal * momentum_adjustment) + ((df['high'] - df['close'].shift(1)) / df['close'].shift(1))
    
    # Calculate Daily Volume Surprise
    volume_surprise = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # Combine Price Momentum and Volume Surprise
    price_momentum_volume = momentum_adjustment * volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']).astype(int) * df['volume']
    downward_volume = (df['close'] < df['open']).astype(int) * df['volume']
    volume_influence_ratio = upward_volume.rolling(window=7).sum() / downward_volume.rolling(window=7).sum()
    volume_influence_ratio.fillna(1, inplace=True)  # Fill NaNs with 1 to avoid division by zero
    
    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha * price_momentum_volume * volume_influence_ratio
    
    return final_alpha_factor
```
