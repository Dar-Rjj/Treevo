import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].diff()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Smooth the Price Momentum with Exponential Moving Average (EMA)
    ema_smoothing_factor = 0.1
    df['smoothed_price_momentum'] = df['price_momentum'].ewm(alpha=ema_smoothing_factor).mean()
    
    # Smooth Intraday Close-Open Return
    df['intraday_close_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Adjust by Volume Change through Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Accumulate Volume on Upward and Downward Days
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1).cumsum()
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1).cumsum()
    
    # Compute Volume Surge
    df['volume_surge'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    df['close_to_midpoint_deviation'] = df['close'] - (df['high'] + df['low']) / 2
    
    # Combine Price Momentum and Intraday High-Low Spread
    df['combined_momentum_intraday_pattern'] = df['smoothed_price_momentum'] * df['intraday_high_low_spread']
    
    # Integrate Volume Ratio and Volume Surge
    df['integrated_volume_adjustment'] = df['volume_ratio'] + df['volume_surge']
    
    # Incorporate Intraday Range and Midpoint Deviation
    df['intraday_range_midpoint_deviation'] = df['intraday_range'] * df['close_to_midpoint_deviation']
    
    # Incorporate Volume Influence into Momentum and Intraday Pattern
    df['volume_influence'] = df['upward_volume'] / (df['downward_volume'] + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Synthesize Components
    df['alpha_factor'] = (df['combined_momentum_intraday_pattern'] * df['volume_influence'] +
                          df['integrated_volume_adjustment'] +
                          df['intraday_range_midpoint_deviation'])
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['alpha_factor']
