```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['10_day_ma'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['30_day_ma'] = df['close'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day moving average to the 30-day moving average
    df['ma_ratio'] = df['10_day_ma'] / df['30_day_ma']
    
    # Calculate the 10-day rolling standard deviation of the daily return
    df['10_day_std'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 30-day rolling standard deviation of the daily return
    df['30_day_std'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the ratio of the 10-day rolling standard deviation to the 30-day rolling standard deviation
    df['std_ratio'] = df['10_day_std'] / df['30_day_std']
    
    # Calculate the daily trading range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the 10-day rolling mean of the daily trading range
    df['10_day_range_mean'] = df['daily_range'].rolling(window=10).mean()
    
    # Calculate the 30-day rolling mean of the daily trading range
    df['30_day_range_mean'] = df['daily_range'].rolling(window=30).mean()
    
    # Calculate the ratio of the 10-day rolling mean of the daily trading range to the 30-day rolling mean of the daily trading range
    df['range_ratio'] = df['10_day_range_mean'] / df['30_day_range_mean']
    
    # Combine the ratios into a single factor
    df['alpha_factor'] = (df['ma_ratio'] + df['std_ratio'] + df['range_ratio']) / 3
    
    return df['alpha_factor']
```
