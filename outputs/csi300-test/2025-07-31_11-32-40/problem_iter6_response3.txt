```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Average
    high_low_avg = (df['high'] + df['low']) / 2
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Calculate Weighted Average Price
    weighted_price = high_low_avg * close_open_diff
    
    # Compute Volume Momentum Score
    # Accumulate Volume on Positive Days
    positive_volume = (df['close'] > df['close'].shift(1)) * df['volume']
    positive_volume_sum = positive_volume.rolling(window=20).sum()
    
    # Accumulate Volume on Negative Days
    negative_volume = (df['close'] < df['close'].shift(1)) * df['volume']
    negative_volume_sum = negative_volume.rolling(window=20).sum()
    
    # Avoid division by zero
    negative_volume_sum = negative_volume_sum.replace(0, 1)
    
    # Divide Positive Volume by Negative Volume
    volume_momentum = positive_volume_sum / negative_volume_sum
    
    # Combine Weighted Price with Volume Momentum
    factor = weighted_price * volume_momentum
    
    return factor
```
