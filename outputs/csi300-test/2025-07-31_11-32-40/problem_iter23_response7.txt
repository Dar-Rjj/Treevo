```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum and Volume Change
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    volume_change = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment * volume_change
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Generate Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday Reversal, Close to Midpoint Deviation, and Open Price Change
    combined_intraday_factor = adjusted_intraday_reversal * close_to_midpoint_deviation + open_price_change
    
    # Calculate Daily Volume Surprise
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_10_day_ma
    
    # Compute Directional Adjusted Volume
    upward_days = (df['close'] > df['open']).astype(int)
    downward_days = (df['close'] < df['open']).astype(int)
    adjusted_volume_up = ((df['close'] - df['open']) * df['volume'] * upward_days).cumsum()
    adjusted_volume_down = ((df['open'] - df['close']) * df['volume'] * downward_days).cumsum()
    net_directional_adjusted_volume = adjusted_volume_up - adjusted_volume_down
    
    # Calculate Cumulative Directional Price Impact
    high_low_diff = df['high'] - df['low']
    cumulative_directional_price_impact = high_low_diff * net_directional_adjusted_volume
    
    # Calculate Smoothed Price Momentum
    lookback_period = 10
    price_momentum = (df['close'] - df['close'].shift(lookback_period)).ewm(span=lookback_period).mean()
    
    # Adjust by Smoothed Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1)).rolling(window=5).mean()
    smoothed_momentum = price_momentum * volume_ratio
    
    # Compute Intraday Close-Open Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    intraday_return_5d_ma = intraday_return.rolling(window=5).mean()
    
    # Integrate Intraday Components
    integrated_intraday_factor = cumulative_directional_price_impact * intraday_return_5d_ma * daily_volume_surprise
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_intraday_factor * integrated_intraday_factor * net_directional_adjusted_volume +
        adjusted_intraday_reversal +
        smoothed_momentum -
        (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    )
    
    return final_alpha_factor
```
