```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['price_diff'] = df['close'] - df['low'].shift(1)
    smoothing_factor = 0.6  # Define EMA smoothing factor
    df['adjusted_momentum_ema'] = df['price_diff'].ewm(alpha=smoothing_factor, adjust=False).mean()

    # Adjust by Volume Change
    sma_length = 5  # Define SMA length
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=sma_length).mean()
    df['volume_ratio_sma'] = df['volume_ratio'].rolling(window=sma_length).mean()

    # Combine Adjusted Momentum and Volume Ratio
    df['combined_adjusted_momentum'] = df['adjusted_momentum_ema'] * df['volume_ratio_sma']

    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1).cumsum()
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1).cumsum()
    df['volume_influence_ratio'] = df['upward_volume'] / (df['downward_volume'] + 1e-6)  # Adding small value to avoid division by zero

    # Integrate Combined Adjusted Momentum, Volume Ratio, and Directional Volume Impact
    df['alpha_factor'] = df['combined_adjusted_momentum'] * df['volume_influence_ratio'] ** 0.5

    return df['alpha_factor']
```
