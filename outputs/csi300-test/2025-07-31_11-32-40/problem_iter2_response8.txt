```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    df['up_volume'] = (df['close'] > df['open']) * df['volume']
    df['down_volume'] = (df['close'] < df['open']) * df['volume']
    up_vol_sum = df['up_volume'].rolling(window=5, min_periods=1).sum()
    down_vol_sum = df['down_volume'].rolling(window=5, min_periods=1).sum()
    volume_influence_ratio = up_vol_sum / down_vol_sum.replace(0, 1)  # Avoid division by zero
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_factor = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Trend
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_trend_strength = volume_change.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine All Factors
    final_alpha_factor = high_low_vol_factor * intraday_return * volume_trend_strength
    
    return final_alpha_factor
```
