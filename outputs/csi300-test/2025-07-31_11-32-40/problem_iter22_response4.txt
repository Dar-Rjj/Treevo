```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    up_volume = df.loc[df['close'] > df['open'], 'volume'].sum()
    down_volume = df.loc[df['close'] < df['open'], 'volume'].sum()
    volume_influence_ratio = (up_volume / down_volume).fillna(1)
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    df['volume_10d_ma'] = df['volume'].rolling(window=10).mean()
    daily_vol_change = df['volume'] - df['volume_10d_ma']
    volume_momentum_strength = daily_vol_change.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor_1 = (high_low_vol_influence * intraday_return) * volume_momentum_strength
    
    # Calculate Daily Price Reversal
    daily_price_reversal = df['close'].diff().mul(-1 * (df['close'].shift(1) > df['close'].shift(2)))
    
    # Calculate Volume Surprise
    volume_surprise = df['volume'] - df['volume_10d_ma']
    price_reversal_vol_surprise = daily_price_reversal * volume_surprise
    
    # Integrate Short-Term Price Volatility
    close_5d_std = df['close'].rolling(window=5).std()
    adjusted_factor_1 = (combined_factor_1 * price_reversal_vol_surprise) / close_5d_std
    
    # Introduce New Factor: Momentum of Intraday Range
    intraday_range_5d_ma = high_low_diff.rolling(window=5).mean()
    daily_intraday_range_change = high_low_diff - intraday_range_5d_ma
    intraday_range_momentum_strength = daily_intraday_range_change.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = adjusted_factor_1 * intraday_range_momentum_strength
    
    return final_alpha_factor
```
