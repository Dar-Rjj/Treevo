```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * (1 + volume_change)
    
    # Daily Price Momentum
    daily_price_momentum = df['close'] - df['close'].shift(10)
    
    # Volume Surprise
    volume_surprise = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Metrics
    combined_metrics = intraday_range * close_to_midpoint_deviation * open_price_change
    
    # Volume Influence Ratio
    upward_volume = df.loc[df['close'] > df['open'], 'volume'].fillna(0)
    downward_volume = df.loc[df['close'] < df['open'], 'volume'].fillna(0)
    volume_influence_ratio = upward_volume.rolling(window=10).sum() / downward_volume.rolling(window=10).sum().replace(0, 1)  # Avoid division by zero
    
    # Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # 5-Day Moving Average of Intraday Close-Open Return
    ma_5_intraday_close_open_return = intraday_close_open_return.rolling(window=5).mean()
    
    # 5-Day Moving Average of Intraday High-Low Spread
    ma_5_intraday_high_low_spread = intraday_high_low_spread.rolling(window=5).mean()
    
    # Weighted Intraday Return
    weighted_intraday_return = 0.5 * intraday_close_open_return + 0.5 * intraday_high_low_spread
    
    # Intraday Momentum
    intraday_momentum = (weighted_intraday_return * ma_5_intraday_close_open_return + 
                         intraday_high_low_spread * ma_5_intraday_high_low_spread) * volume_surge
    
    # Synthesize Final Alpha Factor
    alpha_factor = (adjusted_intraday_reversal * combined_metrics + 
                    daily_price_momentum * volume_surprise * intraday_range * volume_influence_ratio + 
                    intraday_momentum + volume_surge)
    
    return alpha_factor
```
