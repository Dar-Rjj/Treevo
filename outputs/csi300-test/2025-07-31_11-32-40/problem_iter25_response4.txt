```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns using close prices
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute rolling window mean of log returns with 20-day and 50-day windows
    df['rolling_mean_20_log_returns'] = df['log_returns'].rolling(window=20).mean()
    df['rolling_mean_50_log_returns'] = df['log_returns'].rolling(window=50).mean()
    
    # Calculate daily range (High - Low)
    df['daily_range'] = df['high'] - df['low']
    
    # Compute moving average of daily ranges with 30-day and 100-day windows
    df['ma_30_daily_range'] = df['daily_range'].rolling(window=30).mean()
    df['ma_100_daily_range'] = df['daily_range'].rolling(window=100).mean()
    
    # Calculate percentage change in daily volume
    df['pct_change_volume'] = df['volume'].pct_change()
    
    # Calculate Pearson correlation between volume and close prices with a 60-day rolling window
    df['volume_close_corr'] = df['volume'].rolling(window=60).corr(df['close'])
    
    # Create a composite factor using log returns, range, and volume
    # Assign weights based on historical performance (using 90-day historical data for weight calculation)
    df['composite_factor'] = (
        0.4 * df['rolling_mean_20_log_returns'] +
        0.3 * (df['daily_range'] - df['ma_30_daily_range']) / df['ma_30_daily_range'] +
        0.3 * df['pct_change_volume']
    )
    
    return df['composite_factor']
```
