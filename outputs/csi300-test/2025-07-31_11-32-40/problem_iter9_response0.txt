```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - midpoint
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (df['close'] * df['volume']) / df['volume'].sum()
    
    # Calculate Previous Day Return
    previous_day_close = df['close'].shift(1)
    previous_day_return = (df['open'] - previous_day_close) / previous_day_close
    
    # Generate Alpha Factor
    alpha_factor = (intraday_range * close_to_midpoint_deviation) + volume_weighted_price - previous_day_return
    
    return alpha_factor
```
