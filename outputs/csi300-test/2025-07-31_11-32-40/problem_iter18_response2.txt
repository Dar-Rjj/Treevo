import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    close_t_minus_10 = df['close'].shift(10)
    momentum_adjustment = (df['close'] - close_t_minus_10) / close_t_minus_10
    adjusted_intraday_reversal = (2 * (df['high'] - df['low']) / (df['close'] + df['open'])) * (1 + momentum_adjustment)
    
    # Calculate Previous Day Return
    previous_day_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (intraday_range * midpoint_deviation) - previous_day_return
    
    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surprise
    price_momentum_and_volume_surprise = (df['close'] - close_t_minus_10) * daily_volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_difference = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    volume_up = df[df['close'] > df['open']]['volume'].sum()
    volume_down = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = volume_up / volume_down
    
    # Adjust Momentum by Volume and Amount
    adjusted_momentum = high_low_difference * df['volume'] * volume_influence_ratio
    
    # Introduce Time Decay
    decay_rate = 0.05  # Example decay rate
    time = np.arange(len(df))
    time_decay = np.exp(-decay_rate * time)
    adjusted_momentum_with_decay = adjusted_momentum * time_decay
    
    # Incorporate Volume Volatility
    volume_volatility = df['volume'].rolling(window=10).std()
    
    # Adjust Volume Influence Ratio by Volume Volatility
    adjusted_volume_influence_ratio = volume_influence_ratio * (1 + volume_volatility)
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor_synthesis = (price_momentum_and_volume_surprise * high_low_difference) * adjusted_volume_influence_ratio
    
    # Final Alpha Factor
    final_alpha_factor = (intermediate_alpha_factor * adjusted_volume_influence_ratio) + adjusted_momentum_with_decay
    
    return final_alpha_factor
