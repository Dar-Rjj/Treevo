```python
import pandas as pd

def heuristics(df):
    # Calculate High-Low Price Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute Intraday Close-Open Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume if downward_volume != 0 else 0
    
    # Combine High-Low Spread and Close-Open Return
    intraday_momentum = high_low_spread * intraday_return
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1).fillna(0)
    
    # Final Alpha Factor
    alpha_factor = intraday_momentum * volume_influence_ratio + volume_surge
    
    return alpha_factor
```
