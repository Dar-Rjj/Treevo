def heuristics_v2(df):
    import pandas as pd
    
    # Calculate High-Low Percentage Change
    high_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Compute Weighted Directional Volume
    upward_volume = ((df['close'] > df['open']) * (df['close'] - df['open']) * df['volume'] * (df['close'] / df['open'])).rolling(window=10).sum()
    downward_volume = ((df['close'] < df['open']) * (df['open'] - df['close']) * df['volume'] * (df['open'] / df['close'])).rolling(window=10).sum()
    
    net_weighted_directional_volume = upward_volume - downward_volume
    
    # Multiply High-Low Percentage Change with Net Weighted Directional Volume
    factor = high_low_pct_change * net_weighted_directional_volume
    
    return factor
