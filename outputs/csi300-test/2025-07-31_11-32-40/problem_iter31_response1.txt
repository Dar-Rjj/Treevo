```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha = (intraday_reversal * momentum_adjustment) + \
                        ((df['high'] - df['close'].shift(1)) / df['close'].shift(1))
    
    # Calculate Daily Volume Surprise
    volume_surprise = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # Volume Influence Ratio
    upward_volume = df.loc[df['close'] > df['open'], 'volume'].sum()
    downward_volume = df.loc[df['close'] < df['open'], 'volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume
    
    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha * volume_surprise * volume_influence_ratio
    
    return final_alpha_factor
```
