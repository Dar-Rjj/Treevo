```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['PriceMomentum'] = df['close'].diff(10)
    
    # Calculate Volume Surprise
    df['Volume10DMA'] = df['volume'].rolling(window=10).mean()
    df['VolumeSurprise'] = df['volume'] - df['Volume10DMA']
    
    # Combine Price Momentum and Volume Surprise
    df['CombinedMomentumSurprise'] = df['PriceMomentum'] * df['VolumeSurprise']
    
    # Calculate High-Low Price Difference
    df['HighLowDiff'] = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    df['UpVolume'] = (df['close'] > df['open']) * df['volume']
    df['DownVolume'] = (df['close'] < df['open']) * df['volume']
    df['AccumulatedUpVolume'] = df['UpVolume'].cumsum()
    df['AccumulatedDownVolume'] = df['DownVolume'].cumsum()
    df['VolumeInfluenceRatio'] = df['AccumulatedUpVolume'] / df['AccumulatedDownVolume'].replace(0, 1)  # Avoid division by zero
    
    # Synthesize Final Alpha Factor
    df['AlphaFactor'] = (df['CombinedMomentumSurprise'] * df['HighLowDiff'] * df['VolumeInfluenceRatio'])
    
    return df['AlphaFactor']
```
