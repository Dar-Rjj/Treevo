```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Percentage Change
    high_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Compute Weighted Directional Volume
    weighted_up_volume = ((df['close'] - df['open']) * df['volume'] * (df['close'] / df['open'])).where(df['close'] > df['open'], 0).rolling(window=20, min_periods=1).sum()
    weighted_down_volume = ((df['open'] - df['close']) * df['volume'] * (df['open'] / df['close'])).where(df['close'] < df['open'], 0).rolling(window=20, min_periods=1).sum()
    net_weighted_directional_volume = weighted_up_volume - weighted_down_volume
    
    # Compute Weighted Directional Amount
    weighted_up_amount = ((df['close'] - df['open']) * df['amount'] * (df['close'] / df['open'])).where(df['close'] > df['open'], 0).rolling(window=20, min_periods=1).sum()
    weighted_down_amount = ((df['open'] - df['close']) * df['amount'] * (df['open'] / df['close'])).where(df['close'] < df['open'], 0).rolling(window=20, min_periods=1).sum()
    net_weighted_directional_amount = weighted_up_amount - weighted_down_amount
    
    # Combine Net Weighted Directional Volumes
    combined_net_weighted_directional = net_weighted_directional_volume + net_weighted_directional_amount
    
    # Multiply High-Low Percentage Change with Combined Net Weighted Directional Volumes
    factor = high_low_pct_change * combined_net_weighted_directional
    
    return factor
```
