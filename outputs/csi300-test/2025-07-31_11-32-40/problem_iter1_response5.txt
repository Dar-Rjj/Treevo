import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    df['IntradayReversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    df['VolumeChange'] = df['volume'] - df['volume'].shift(1)
    df['OpenInterestAdjustment'] = 1 + (df['VolumeChange'] / df['volume'].shift(1))
    df['AdjustedIntradayReversal'] = df['IntradayReversal'] * df['OpenInterestAdjustment']
    
    # Consider Day-to-Day Open Price Change
    df['DayToDayOpenPriceChange'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Adjustments
    df['AlphaFactor'] = df['AdjustedIntradayReversal'] + df['DayToDayOpenPriceChange']
    
    return df['AlphaFactor']
