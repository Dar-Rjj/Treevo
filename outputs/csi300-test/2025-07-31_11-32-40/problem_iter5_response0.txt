```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Surprise
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_ma_10
    
    # Combine Price Momentum and Volume Surprise
    combined_momentum_surprise = price_momentum * volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / (downward_volume + 1e-6)  # Add small constant to avoid division by zero
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_momentum_surprise * high_low_diff
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor * volume_influence_ratio
    
    return final_alpha_factor
```
