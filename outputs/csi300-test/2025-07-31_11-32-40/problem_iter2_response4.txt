```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    price_momentum = df['close'].diff(periods=10)
    
    # Calculate Volume Surprise
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_moving_avg
    
    # Combine Price Momentum and Volume Surprise
    intermediate_alpha_factor = price_momentum * volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_difference = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * df['volume']
    upward_volume_sum = upward_volume.rolling(window=10).sum()
    downward_volume_sum = downward_volume.rolling(window=10).sum()
    volume_influence_ratio = upward_volume_sum / downward_volume_sum
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_volume_influence = high_low_difference * volume_influence_ratio
    
    # Combine Intermediate Alpha Factors
    final_alpha_factor = intermediate_alpha_factor * high_low_volume_influence
    
    return final_alpha_factor
```
