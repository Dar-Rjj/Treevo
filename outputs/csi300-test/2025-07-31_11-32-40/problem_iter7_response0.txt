```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - midpoint
    
    # Generate Alpha Factor
    alpha_factor = intraday_range * close_to_midpoint_deviation
    
    # Calculate Previous Day Return
    previous_day_close = df['close'].shift(1)
    today_open = df['open']
    previous_day_return = (today_open - previous_day_close) / previous_day_close
    
    # Final Alpha Factor
    final_alpha_factor = alpha_factor - previous_day_return
    
    return final_alpha_factor
```
