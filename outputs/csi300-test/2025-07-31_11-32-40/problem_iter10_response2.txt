```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Accumulate Volume on Upward Days
    upward_volume = (df['close'] > df['open']) * df['volume']
    
    # Accumulate Volume on Downward Days
    downward_volume = (df['close'] < df['open']) * df['volume']
    
    # Calculate cumulative upward and downward volumes
    cum_upward_volume = upward_volume.cumsum()
    cum_downward_volume = downward_volume.cumsum()
    
    # Avoid division by zero
    cum_downward_volume = cum_downward_volume.replace(0, 1)
    
    # Calculate Volume Influence Ratio
    volume_influence_ratio = cum_upward_volume / cum_downward_volume
    
    # Multiply with High-Low Difference
    factor_values = high_low_diff * volume_influence_ratio
    
    return factor_values
```
