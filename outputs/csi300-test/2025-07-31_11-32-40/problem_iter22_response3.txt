```python
import pandas as pd

def heuristics(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Generate Preliminary Alpha Factor
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + day_to_day_open_change

    # Calculate Adjusted Daily Momentum
    adjusted_daily_momentum = df['close'] - df['close'].shift(10)

    # Incorporate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Combine Adjusted Momentum and Intraday Return
    combined_momentum_intraday = adjusted_daily_momentum + intraday_return

    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - df['volume'].rolling(window=10).mean()

    # Adjust by Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1))
    smoothed_volume_ratio = volume_ratio.ewm(span=10, adjust=False).mean()

    # Combine Volume Impact Factors
    combined_volume_impact = daily_volume_surprise * smoothed_volume_ratio

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].rolling(window=10).sum()
    downward_volume = df[df['close'] < df['open']]['volume'].rolling(window=10).sum()
    volume_influence_ratio = (upward_volume / downward_volume).fillna(0)

    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_momentum_intraday * combined_volume_impact

    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha_factor * intermediate_alpha_factor * volume_influence_ratio

    return final_alpha_factor
```
