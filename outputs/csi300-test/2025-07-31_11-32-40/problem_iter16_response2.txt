```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Close-Open Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Influence Ratio
    df['up_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['down_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['up_volume_sum'] = df['up_volume'].rolling(window=5).sum()
    df['down_volume_sum'] = df['down_volume'].rolling(window=5).sum()
    df['volume_influence_ratio'] = df['up_volume_sum'] / (df['down_volume_sum'] + 1e-6)  # Avoid division by zero
    
    # Combine High-Low Spread and Close-Open Return
    df['high_low_spread'] = df['high'] - df['low']
    df['intraday_momentum'] = df['high_low_spread'] * df['intraday_return']
    
    # Calculate Adjusted Intraday Reversal
    df['adjusted_intraday_reversal'] = 2 * df['high_low_spread'] / (df['close'] + df['open'])
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / (df['volume'].shift(1) + 1e-6)  # Avoid division by zero
    df['adjusted_intraday_reversal'] *= (1 + df['volume_change'])
    
    # Consider Day-to-Day Open Price Change
    df['day_to_day_open_change'] = (df['open'] - df['close'].shift(1)) / (df['close'].shift(1) + 1e-6)  # Avoid division by zero
    
    # Final Alpha Factor Components
    df['alpha_factor'] = (df['intraday_momentum'] * df['volume_influence_ratio'] 
                          + df['adjusted_intraday_reversal'] 
                          + df['day_to_day_open_change'])
    
    # Evaluate with Moving Average
    df['alpha_factor_ma_5'] = df['alpha_factor'].rolling(window=5).mean()
    df['score'] = df.apply(lambda row: 1 if row['alpha_factor'] > row['alpha_factor_ma_5'] else -1, axis=1)
    
    return df['score']
```
