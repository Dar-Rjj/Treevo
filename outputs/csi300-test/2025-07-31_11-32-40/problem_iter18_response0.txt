```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * df['volume']
    volume_influence_ratio = (upward_volume.rolling(window=10).sum() / downward_volume.rolling(window=10).sum()).fillna(0)
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    daily_volume_change = df['volume'] - avg_volume_10
    volume_momentum_strength = daily_volume_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor_1 = high_low_vol_influence * intraday_return * volume_momentum_strength
    
    # Calculate Daily Price Reversal
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_reversal = -daily_return * (daily_return.shift(1) > 0)
    
    # Calculate Volume Surprise
    volume_surprise = df['volume'] - avg_volume_10
    price_vol_surprise = price_reversal * volume_surprise
    
    # Integrate Short-Term Price Volatility
    close_std_5 = df['close'].rolling(window=5).std()
    adjusted_combined_factors = (combined_factor_1 + price_vol_surprise) / close_std_5
    
    # Introduce New Factor: Momentum of Intraday Range
    avg_high_low_diff_5 = high_low_diff.rolling(window=5).mean()
    intraday_range_change = high_low_diff - avg_high_low_diff_5
    intraday_range_momentum_strength = intraday_range_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = adjusted_combined_factors * intraday_range_momentum_strength
    
    return final_alpha_factor
```
