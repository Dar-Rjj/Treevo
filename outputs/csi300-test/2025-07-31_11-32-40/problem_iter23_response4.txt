```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    volume_change = (1 + (df['volume'].diff() / df['volume'].shift(1)))
    adjusted_intraday_reversal = intraday_reversal * volume_change

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Calculate Day-to-Day Open Price Change
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + day_to_day_open_price_change

    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']

    # Calculate 5-Day Moving Averages
    ma_close_open_return = intraday_close_open_return.rolling(window=5).mean()
    ma_high_low_spread = intraday_high_low_spread.rolling(window=5).mean()

    # Calculate Weighted Intraday Return
    weighted_close_open_return = 0.7 * intraday_close_open_return
    weighted_high_low_spread = 0.3 * intraday_high_low_spread
    weighted_intraday_return = weighted_close_open_return + weighted_high_low_spread

    # Combine Weighted Intraday Return and Moving Averages
    intraday_momentum = (weighted_intraday_return * ma_close_open_return) + (intraday_high_low_spread * ma_high_low_spread)

    # Calculate Price Momentum with EMA
    price_momentum = df['close'].pct_change().ewm(span=20, adjust=False).mean()

    # Adjust by Volume Change
    volume_ratio = df['volume'].rolling(window=5).mean() / df['volume'].shift(5).rolling(window=5).mean()
    combined_momentum = price_momentum * volume_ratio

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume']
    downward_volume = df[df['close'] < df['open']]['volume']
    volume_influence_ratio = upward_volume.sum() / downward_volume.sum()

    # Integrate Core Components into Intermediate Alpha Factor
    intermediate_alpha_factor = combined_momentum * (intraday_high_low_spread * volume_influence_ratio * intraday_close_open_return)

    # Final Alpha Factor Components
    final_alpha_factor_components = (intraday_momentum * volume_influence_ratio) + adjusted_intraday_reversal + day_to_day_open_price_change

    # Final Alpha Factor
    final_alpha_factor = final_alpha_factor_components + (df['volume'] - df['volume'].shift(1))

    return final_alpha_factor
```
