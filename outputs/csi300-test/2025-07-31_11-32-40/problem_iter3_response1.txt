```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-to-Low Return
    high_to_low_return = df['high'] / df['low'] - 1
    
    # Calculate Price Movement Indicator
    price_movement = df['close'] - df['open']
    direction = np.where(price_movement > 0, 1.0, -1.0)
    
    # Adjust Momentum by Volume and Amount
    weighted_high_to_low_return = high_to_low_return * df['volume'] * direction
    
    # Normalize by Total Volume and Amount
    total_volume = df['volume'].sum()
    total_amount = df['amount'].sum()
    normalized_momentum = weighted_high_to_low_return / (total_volume + total_amount)
    
    # Introduce Time Decay
    decay_rate = 0.95  # Example decay rate, can be adjusted
    days = np.arange(len(df))
    decayed_momentum = normalized_momentum * (decay_rate ** days)
    
    # Finalize Alpha Factor
    alpha_factor = decayed_momentum.cumsum()
    
    return alpha_factor
```
