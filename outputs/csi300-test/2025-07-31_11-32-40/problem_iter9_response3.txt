```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['PriceMomentum'] = df['close'].diff(10)
    
    # Calculate High-Low Price Spread
    df['HighLowSpread'] = df['high'] - df['low']
    
    # Compute Intraday Close-Open Return
    df['CloseOpenReturn'] = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Spread and Close-Open Return
    df['IntradayMomentum'] = df['HighLowSpread'] * df['CloseOpenReturn']
    
    # Calculate Volume Surprise
    df['VolumeMA'] = df['volume'].rolling(window=10).mean()
    df['VolumeSurprise'] = df['volume'] - df['VolumeMA']
    
    # Combine Price Momentum and Volume Surprise
    df['IntermediateAlphaFactor'] = df['PriceMomentum'] + df['VolumeSurprise']
    
    # Calculate Volume Surge
    df['VolumeSurge'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Volume Influence Ratio
    df['UpwardVolume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['DownwardVolume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['UpwardAccumulatedVolume'] = df['UpwardVolume'].cumsum()
    df['DownwardAccumulatedVolume'] = df['DownwardVolume'].cumsum()
    df['VolumeInfluenceRatio'] = df['UpwardAccumulatedVolume'] / df['DownwardAccumulatedVolume'].replace(0, 1)  # Avoid division by zero
    
    # Multiply High-Low Spread with Volume Influence Ratio
    df['HighLowVolumeInfluence'] = df['HighLowSpread'] * df['VolumeInfluenceRatio']
    
    # Combine All Alpha Factors
    df['FinalAlphaFactor'] = (
        df['IntermediateAlphaFactor'] * df['HighLowVolumeInfluence'] +
        df['IntradayMomentum'] +
        df['VolumeSurge']
    )
    
    return df['FinalAlphaFactor']
```
