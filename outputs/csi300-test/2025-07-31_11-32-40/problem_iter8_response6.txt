```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['PriceMomentum'] = df['close'].pct_change(10)
    
    # Calculate High-Low Price Spread
    df['HighLowSpread'] = df['high'] - df['low']
    
    # Compute Intraday Close-Open Return
    df['CloseOpenReturn'] = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Spread and Close-Open Return
    df['IntradayMomentum'] = df['HighLowSpread'] * df['CloseOpenReturn']
    
    # Calculate Volume Surprise
    df['VolumeMA10'] = df['volume'].rolling(window=10).mean()
    df['VolumeSurprise'] = df['volume'] - df['VolumeMA10']
    
    # Combine Price Momentum and Volume Surprise
    df['CombinedAlphaFactor'] = df['PriceMomentum'] * df['VolumeSurprise']
    
    # Calculate Volume Surge
    df['VolumeSurge'] = df['volume'].diff()
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume if downward_volume != 0 else 0
    
    # Intermediate Alpha Factor Synthesis
    df['IntermediateAlphaFactor'] = df['CombinedAlphaFactor'] * df['HighLowSpread']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = (
        df['IntermediateAlphaFactor'] + 
        df['IntradayMomentum'] * volume_influence_ratio + 
        df['VolumeSurge']
    )
    
    return df['FinalAlphaFactor']
```
