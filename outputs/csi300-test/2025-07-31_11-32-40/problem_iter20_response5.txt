import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day rolling standard deviation of the close price
    df['std_10'] = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day mean volume
    df['vol_ma_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the volume
    df['vol_std_10'] = df['volume'].rolling(window=10).std()
    
    # Calculate the 10-day mean amount
    df['amt_ma_10'] = df['amount'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the amount
    df['amt_std_10'] = df['amount'].rolling(window=10).std()
    
    # Calculate the 10-day momentum
    df['momentum_10'] = df['close'] / df['close'].shift(10) - 1
    
    # Calculate the 10-day volatility
    df['volatility_10'] = df['std_10'] / df['ma_10']
    
    # Calculate the 10-day volume-to-amount ratio
    df['vol_to_amt_ratio_10'] = df['vol_ma_10'] / df['amt_ma_10']
    
    # Calculate the 10-day normalized volume
    df['norm_vol_10'] = (df['volume'] - df['vol_ma_10']) / df['vol_std_10']
    
    # Calculate the 10-day normalized amount
    df['norm_amt_10'] = (df['amount'] - df['amt_ma_10']) / df['amt_std_10']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (
        df['momentum_10'] * 0.4 +
        df['volatility_10'] * 0.3 +
        df['vol_to_amt_ratio_10'] * 0.2 +
        df['norm_vol_10'] * 0.05 +
        df['norm_amt_10'] * 0.05
    )
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
