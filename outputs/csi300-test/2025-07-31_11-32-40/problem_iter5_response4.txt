```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods and smoothing factors
    momentum_lookback = 10
    ema_smoothing_factor = 2 / (momentum_lookback + 1)
    volume_lookback = 5
    sma_length = 5
    trend_strength_positive = 1.5
    trend_strength_negative = 0.5

    # Calculate Price Momentum
    df['price_diff'] = df['close'].diff()
    df['momentum_ema'] = df['price_diff'].ewm(alpha=ema_smoothing_factor, adjust=False).mean()

    # Adjust by Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_sma'] = df['volume_ratio'].rolling(window=sma_length).mean()
    df['momentum_volume_adjusted'] = df['momentum_ema'] * df['volume_sma']

    # Calculate High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']

    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['total_upward_volume'] = df['upward_volume'].rolling(window=volume_lookback).sum()
    df['total_downward_volume'] = df['downward_volume'].rolling(window=volume_lookback).sum()
    df['volume_influence_ratio'] = df['total_upward_volume'] / df['total_downward_volume']
    df['volume_influence_ratio'] = df['volume_influence_ratio'].fillna(1)

    # Multiply High-Low Difference with Volume Influence Ratio
    df['high_low_volume_influence'] = df['high_low_diff'] * df['volume_influence_ratio']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Identify Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_trend_strength'] = df['volume_change'].apply(lambda x: trend_strength_positive if x > 0 else trend_strength_negative)

    # Integrate All Components into Final Alpha Factor
    df['alpha_factor'] = (df['momentum_volume_adjusted'] *
                          df['high_low_volume_influence'] *
                          df['intraday_return'] *
                          df['volume_trend_strength'])

    return df['alpha_factor']
```
