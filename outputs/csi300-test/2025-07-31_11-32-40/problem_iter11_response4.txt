import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint = (df['high'] + df['low']) / 2
    close_to_mid_dev = df['close'] - midpoint
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Calculate Previous Day Return
    prev_day_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (intraday_range * close_to_mid_dev) - prev_day_return
    
    # Calculate Daily Volume Surprise
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_ma_10
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(10)
    combined_momentum_volume = price_momentum * volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume
    
    # Adjust Momentum by Volume and Amount
    weighted_high_low = high_low_diff * df['volume'] * volume_influence_ratio
    volume_adjusted_momentum = weighted_high_low.rolling(window=10).sum()
    
    # Introduce Time Decay
    decay_rate = 0.95
    decayed_momentum = volume_adjusted_momentum * (decay_rate ** (10 - 1))
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_momentum_volume * high_low_diff
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha * intermediate_alpha_factor * volume_influence_ratio
    
    return final_alpha_factor
