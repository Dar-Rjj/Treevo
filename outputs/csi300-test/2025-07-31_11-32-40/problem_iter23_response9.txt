import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    df['close_to_midpoint_deviation'] = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    df['intraday_reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    df['momentum_adjustment'] = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['adjusted_intraday_reversal'] = df['intraday_reversal'] * df['momentum_adjustment']
    
    # Calculate Previous Day Return
    df['previous_day_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor 1
    df['intermediate_alpha_factor_1'] = (df['intraday_range'] * df['close_to_midpoint_deviation']) - df['previous_day_return']
    
    # Calculate Daily Volume Surprise
    df['volume_10d_ma'] = df['volume'].rolling(window=10).mean()
    df['daily_volume_surprise'] = df['volume'] - df['volume_10d_ma']
    
    # Combine Price Momentum and Volume Surprise
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    df['combined_price_momentum_volume_surprise'] = df['price_momentum'] * df['daily_volume_surprise']
    
    # Calculate High-Low Price Difference
    df['high_low_difference'] = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1).rolling(window=10).sum()
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1).rolling(window=10).sum()
    df['volume_influence_ratio'] = df['upward_volume'] / (df['downward_volume'] + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Adjust Momentum by Volume and Amount
    df['volume_adjusted_momentum'] = df['high_low_difference'] * df['volume'] * df['volume_influence_ratio']
    
    # Introduce Time Decay
    decay_rate = 0.95
    df['time_decayed_volume_adjusted_momentum'] = df['volume_adjusted_momentum'] * (decay_rate ** (len(df) - df.index))
    
    # Intermediate Alpha Factor Synthesis 1
    df['intermediate_alpha_factor_synthesis_1'] = df['combined_price_momentum_volume_surprise'] * df['high_low_difference']
    
    # Compute Price Momentum
    lookback_period = 10
    df['price_momentum_ema'] = df['close'] - df['close'].shift(lookback_period)
    df['price_momentum_ema'] = df['price_momentum_ema'].ewm(span=lookback_period).mean()
    
    # Adjust by Volume Change
    df['volume_ratio'] = (df['volume'] / df['volume'].shift(1)).rolling(window=lookback_period).mean()
    df['combined_momentum_volume_ratio'] = df['price_momentum_ema'] * df['volume_ratio']
    
    # Compute Intraday Close-Open Return
    df['intraday_close_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Spread and Close-Open Return
    df['intraday_momentum'] = df['high_low_difference'] * df['intraday_close_open_return']
    
    # Consider Day-to-Day Open Price Change
    df['day_to_day_open_price_change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Adjust for Open Interest
    df['adjusted_intraday_reversal'] *= (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1))

    df['intermediate_alpha_factor_synthesis_2'] = df['combined_momentum_volume_ratio'] * df
