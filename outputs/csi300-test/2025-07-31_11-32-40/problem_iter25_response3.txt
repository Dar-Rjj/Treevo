```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Reversal
    df['intraday_reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum and Volume Change
    df['momentum_adjustment'] = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['volume_change'] = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['adjusted_intraday_reversal'] = df['intraday_reversal'] * df['momentum_adjustment'] * df['volume_change']
    
    # Calculate Close to Midpoint Deviation
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['close_to_midpoint_deviation'] = df['close'] - df['midpoint']
    
    # Generate Day-to-Day Open Price Change
    df['day_to_day_open_price_change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday Reversal, Close to Midpoint Deviation, and Open Price Change
    df['combined_factor'] = df['adjusted_intraday_reversal'] * df['close_to_midpoint_deviation'] + df['day_to_day_open_price_change']
    
    # Calculate Daily Volume Surprise
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['daily_volume_surprise'] = df['volume'] - df['volume_ma_10']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df[df['close'] > df['open']]['volume']
    df['downward_volume'] = df[df['close'] < df['open']]['volume']
    df['upward_volume'] = df['upward_volume'].fillna(0).cumsum()
    df['downward_volume'] = df['downward_volume'].fillna(0).cumsum()
    df['volume_influence_ratio'] = df['upward_volume'] / df['downward_volume'].replace(0, np.nan)
    
    # Calculate Smoothed Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    df['smoothed_price_momentum'] = df['price_momentum'].ewm(span=10, adjust=False).mean()
    
    # Adjust by Smoothed Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=10).mean()
    df['integrated_intraday_factor'] = df['smoothed_price_momentum'] * df['smoothed_volume_ratio']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_factor'] * df['integrated_intraday_factor'] * df['volume_influence_ratio']
                                + df['adjusted_intraday_reversal'] + df['integrated_intraday_factor'])
    
    # Calculate Previous Day Return
    df['previous_day_return'] = (df['close'].shift(1) - df['open']) / df['open']
    
    return df['final_alpha_factor']
```
