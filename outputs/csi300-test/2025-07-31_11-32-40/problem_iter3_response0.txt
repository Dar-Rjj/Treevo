```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Surge
    volume_surge = df['volume'].diff().fillna(0)
    
    # Calculate Weighted Intraday Return
    weighted_intraday_return = 0.5 * intraday_close_open_return + 0.5 * intraday_high_low_spread
    
    # Combine Weighted Intraday Return and Volume Surge
    intraday_momentum = weighted_intraday_return * volume_surge
    
    # Final Alpha Factor
    intraday_momentum_volume_surge_indicator = intraday_momentum + volume_surge
    
    return intraday_momentum_volume_surge_indicator
```
