```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    adjusted_momentum = df['close'] - df['low'].shift(1)
    ema_adjusted_momentum = adjusted_momentum.ewm(span=5, adjust=False).mean()

    # Simple Price Momentum
    simple_momentum = df['close'] - df['close'].shift(1)
    ema_simple_momentum = simple_momentum.ewm(span=5, adjust=False).mean()

    # Combine Adjusted Momentum and Simple Price Momentum
    combined_momentum = ema_adjusted_momentum + ema_simple_momentum

    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    sma_volume_ratio = volume_ratio.rolling(window=5).mean()
    combined_momentum *= sma_volume_ratio

    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * df['volume']
    volume_influence_ratio = upward_volume.cumsum() / downward_volume.cumsum()

    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    combined_momentum *= np.sqrt(volume_influence_ratio)

    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']

    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_diff_volume = high_low_diff * volume_influence_ratio

    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Close to Midpoint Deviation
    close_mid_dev = df['close'] - (df['high'] + df['low']) / 2

    # Calculate Volume Weighted Price
    volume_weighted_price = (df['close'] * df['volume']) / df['volume']

    # Generate Final Alpha Factor
    alpha_factor = (intraday_range * close_mid_dev) + volume_weighted_price
    alpha_factor *= high_low_diff_volume
    alpha_factor += combined_momentum
    alpha_factor -= (df['open'] - df['close'].shift(1))

    return alpha_factor
```
