```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Weighted Price Momentum
    close_prices = df['close']
    price_changes = close_prices - close_prices.shift(1)
    
    # Compute the weighted sum of close price change over the last 10 days
    weights = pd.Series(range(1, 11), index=price_changes.index[-10:])
    weighted_price_momentum = (price_changes.rolling(window=10).apply(lambda x: (x * weights).sum(), raw=True))
    
    # Adjust by average volume over the period
    avg_volume = df['volume'].rolling(window=10).mean()
    weighted_price_momentum /= avg_volume
    
    # Revisiting Volume Surprise
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    volume_surprise = (df['volume'] - volume_ema) ** 2
    
    # Integrating Price and Volume Signals
    alpha_factor = weighted_price_momentum / volume_surprise
    
    return alpha_factor
```
