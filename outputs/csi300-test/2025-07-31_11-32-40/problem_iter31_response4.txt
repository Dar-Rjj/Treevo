```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Percentage Change
    high_low_pct_change = (df['high'] - df['low']) / df['low']
    
    # Compute Weighted Directional Volume
    upward_vol = ((df['close'] - df['open']) * df['volume'] * (df['close'] / df['open'])).where(df['close'] > df['open'], 0).rolling(window=10, min_periods=1).sum()
    downward_vol = ((df['open'] - df['close']) * df['volume'] * (df['open'] / df['close'])).where(df['close'] < df['open'], 0).rolling(window=10, min_periods=1).sum()
    net_weighted_directional_vol = upward_vol - downward_vol
    
    # Multiply High-Low Percentage Change with Net Weighted Directional Volume
    factor = high_low_pct_change * net_weighted_directional_vol
    
    return factor
```
