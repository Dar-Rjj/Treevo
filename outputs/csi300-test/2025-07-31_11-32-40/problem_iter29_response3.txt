import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Short-Term Momentum and Volume Change
    momentum_adjustment = 1 + (df['close'].shift(5) - df['close']) / df['close'].shift(5)
    volume_change = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment * volume_change
    
    # Calculate Close to Midpoint Deviation
    midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - midpoint
    
    # Generate Day-to-Day Price Change
    high_price_change = (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday Reversal, Close to Midpoint Deviation, and Price Changes
    combined_factor = adjusted_intraday_reversal * close_to_midpoint_deviation + high_price_change + open_price_change
    
    # Calculate Daily Volume Surprise
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    daily_volume_surprise = df['volume'] - avg_volume_5d
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume if downward_volume != 0 else 0
    
    # Calculate Smoothed Price Momentum
    lookback_period = 5
    price_momentum = (df['close'] - df['close'].shift(lookback_period)).ewm(span=lookback_period).mean()
    
    # Adjust by Smoothed Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(lookback_period)).rolling(window=lookback_period).mean()
    combined_momentum_volume = price_momentum * volume_ratio
    
    # Integrate Intraday Range and Volume Dynamics
    intraday_range = df['high'] - df['low']
    integrated_intraday_volume = intraday_range * df['volume']
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_factor * integrated_intraday_volume * combined_momentum_volume
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor * volume_influence_ratio + adjusted_intraday_reversal
    
    return final_alpha_factor
