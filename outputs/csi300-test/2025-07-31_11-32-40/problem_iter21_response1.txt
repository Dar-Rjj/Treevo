```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume' and index 'date'.
    
    Returns:
    pd.Series: A pandas Series indexed by 'date' representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted price
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=5, min_periods=1).mean()
    
    # Calculate the momentum of the volume weighted price
    df['vwap_momentum'] = df['vwap'].pct_change(periods=5)
    
    # Calculate the normalized volume
    df['normalized_volume'] = df['volume'] / df['volume'].rolling(window=20, min_periods=1).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (
        df['daily_return'].shift(1) * 0.3 +
        df['relative_range'].shift(1) * 0.2 +
        df['vwap_momentum'].shift(1) * 0.3 +
        df['normalized_volume'].shift(1) * 0.2
    )
    
    return df['alpha_factor']
```
