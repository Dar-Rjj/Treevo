```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    lookback_momentum = 20
    df['Price_Momentum'] = df['close'].pct_change(lookback_momentum)
    df['Smoothed_Momentum'] = df['Price_Momentum'].ewm(span=lookback_momentum, adjust=False).mean()

    # Adjust by Volume Change
    lookback_volume = 10
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=lookback_volume).mean()
    df['Smoothed_Volume_Ratio'] = df['Volume_Ratio'].rolling(window=lookback_volume).mean()
    df['Momentum_Volume_Adjusted'] = df['Smoothed_Momentum'] * df['Smoothed_Volume_Ratio']

    # Compute High-Low Price Difference
    df['High_Low_Diff'] = df['high'] - df['low']

    # Compute Volume Influence Ratio
    df['Upward_Volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['Downward_Volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['Upward_Volume_Sum'] = df['Upward_Volume'].rolling(window=lookback_volume).sum()
    df['Downward_Volume_Sum'] = df['Downward_Volume'].rolling(window=lookback_volume).sum()
    df['Volume_Influence_Ratio'] = df['Upward_Volume_Sum'] / (df['Downward_Volume_Sum'] + 1e-6)

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Identify Volume Trend
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Trend_Strength'] = np.where(df['Volume_Change'] > 0, 1.5, 0.5)

    # Integrate Core Components into Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = (
        df['Momentum_Volume_Adjusted'] *
        df['High_Low_Diff'] *
        df['Volume_Influence_Ratio'] *
        df['Intraday_Return'] *
        df['Volume_Trend_Strength']
    )

    # Combine High-Low Spread and Close-Open Return for Intraday Momentum
    df['Intraday_Momentum'] = df['High_Low_Diff'] * df['Intraday_Return']

    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])

    # Adjust for Open Interest
    df['Volume_Change_Normalized'] = (df['volume'] - df['volume'].shift(1)) / (df['volume'].shift(1) + 1e-6)
    df['Adjusted_Intraday_Reversal'] = df['Intraday_Reversal'] * (1 + df['Volume_Change_Normalized'])

    # Consider Day-to-Day Open Price Change
    df['Day_to_Day_Open_Price_Change'] = (df['open'] - df['close'].shift(1)) / (df['close'].shift(1) + 1e-6)

    # Final Alpha Factor Components
    df['Final_Alpha_Factor'] = (
        df['Intermediate_Alpha_Factor'] +
        df['Intraday_Momentum'] +
        df['Adjusted_Intraday_Reversal'] +
        df['Day_to_Day_Open_Price_Change']
    )

    # Calculate Moving Average
    df['MA_5'] = df['Final_Alpha_Factor'].rolling(window=5).mean()
    df['Score'] = np.where(df['Final_Alpha_Factor'] > df['MA_5'], 1, -1)

    return df['Score'].dropna()
```
