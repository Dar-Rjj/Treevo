```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    momentum_lookback = 20
    volume_lookback = 10

    # Compute Price Momentum
    df['Price_Momentum'] = df['close'].diff()
    df['Smoothed_Price_Momentum'] = df['Price_Momentum'].ewm(span=momentum_lookback).mean()

    # Adjust by Volume Change
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Smoothed_Volume_Ratio'] = df['Volume_Ratio'].rolling(window=volume_lookback).mean()
    df['Combined_Momentum_and_Volume'] = df['Smoothed_Price_Momentum'] * df['Smoothed_Volume_Ratio']

    # Compute Intraday Close-Open Return
    df['Intraday_Close_Open_Return'] = (df['close'] - df['open']) / df['open']

    # Compute Volume Influence Ratio
    df['Upward_Volume'] = df.apply(lambda x: x['volume'] if x['close'] > x['open'] else 0, axis=1).rolling(window=volume_lookback).sum()
    df['Downward_Volume'] = df.apply(lambda x: x['volume'] if x['close'] < x['open'] else 0, axis=1).rolling(window=volume_lookback).sum()
    df['Volume_Influence_Ratio'] = df['Upward_Volume'] / df['Downward_Volume'].replace(0, 1)  # Avoid division by zero

    # Combine High-Low Spread and Close-Open Return
    df['High_Low_Difference'] = df['high'] - df['low']
    df['Intraday_Momentum'] = df['High_Low_Difference'] * df['Intraday_Close_Open_Return']

    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])

    # Adjust for Open Interest
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Intraday_Reversal'] = df['Intraday_Reversal'] * (1 + df['Volume_Change'])

    # Consider Day-to-Day Open Price Change
    df['Day_to_Day_Open_Price_Change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Final Alpha Factor Components
    df['Final_Alpha_Factor'] = (
        df['Intraday_Momentum'] * df['Volume_Influence_Ratio'] +
        df['Adjusted_Intraday_Reversal'] +
        df['Day_to_Day_Open_Price_Change'] +
        df['High_Low_Difference'] * df['Volume_Influence_Ratio'] +
        df['Combined_Momentum_and_Volume'] -
        df['close'].pct_change().shift(1)
    )

    return df['Final_Alpha_Factor']
```
