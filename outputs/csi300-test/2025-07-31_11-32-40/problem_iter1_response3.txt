```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Price Movement Indicator
    price_movement_indicator = df['close'] - df['open']
    direction = (price_movement_indicator > 0).astype(int) - (price_movement_indicator < 0).astype(int)
    
    # Adjust Momentum by Volume
    weighted_return = high_to_low_return * df['volume'] * direction
    
    # Normalize by Total Volume
    total_volume = df['volume'].sum()
    momentum_factor = weighted_return.sum() / total_volume
    
    return pd.Series(momentum_factor, index=df.index)
```
