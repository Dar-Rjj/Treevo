```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * df['volume']
    total_upward_volume = upward_volume.rolling(window=10).sum()
    total_downward_volume = downward_volume.rolling(window=10).sum()
    volume_influence_ratio = total_upward_volume / total_downward_volume
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_with_volume_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    vol_moving_avg = df['volume'].rolling(window=10).mean()
    daily_vol_change = df['volume'] - vol_moving_avg
    volume_momentum_strength = daily_vol_change.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor = (high_low_with_volume_influence * intraday_return) * volume_momentum_strength
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Adjust for Open Interest
    adjusted_intraday_reversal *= (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1))
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + open_price_change
    
    # Calculate Daily Volume Surprise
    daily_vol_surprise = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Price Momentum and Volume Surprise
    combined_momentum_vol_surprise = daily_price_momentum * daily_vol_surprise
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_momentum_vol_surprise * high_low_diff
    
    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha_factor * intermediate_alpha_factor * volume_influence_ratio
    
    return final_alpha_factor
```
