```python
import pandas as pd

def heuristics(df):
    # Compute Intraday Reversal
    df['intraday_reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum
    df['momentum_adjustment'] = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['adjusted_intraday_reversal'] = df['intraday_reversal'] * df['momentum_adjustment']
    
    # Calculate Close to Midpoint Deviation
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['close_to_midpoint_deviation'] = df['close'] - df['midpoint']
    
    # Calculate Day-to-Day Open Price Change
    df['day_to_day_open_price_change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Preliminary Alpha Factor
    df['preliminary_alpha_factor'] = (
        df['adjusted_intraday_reversal'] * df['close_to_midpoint_deviation'] +
        df['day_to_day_open_price_change'] - 
        df['close'].pct_change()
    )
    
    # Calculate Daily Volume Surprise
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['daily_volume_surprise'] = df['volume'] - df['volume_ma_10']
    
    # Combine Price Momentum and Volume Surprise
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    df['combined_price_momentum_volume_surprise'] = df['price_momentum'] * df['daily_volume_surprise']
    
    # Calculate High-Low Price Difference
    df['high_low_difference'] = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['upward_volume_sum'] = df['upward_volume'].rolling(window=10).sum()
    df['downward_volume_sum'] = df['downward_volume'].rolling(window=10).sum()
    df['volume_influence_ratio'] = df['upward_volume_sum'] / df['downward_volume_sum']
    
    # Intermediate Alpha Factor Synthesis
    df['intermediate_alpha_factor'] = df['combined_price_momentum_volume_surprise'] * df['high_low_difference']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['preliminary_alpha_factor'] * 
        df['intermediate_alpha_factor'] * 
        df['volume_influence_ratio']
    )
    
    return df['final_alpha_factor']
```
