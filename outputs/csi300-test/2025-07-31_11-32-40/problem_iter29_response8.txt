```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['adjusted_momentum'] = df['close'] - df['low'].shift(1)
    ema_smooth_factor = 10
    df['smoothed_adjusted_momentum'] = df['adjusted_momentum'].ewm(span=ema_smooth_factor, adjust=False).mean()
    
    # Calculate Simple Price Momentum
    lookback_period = 10
    df['simple_momentum'] = df['close'] - df['close'].shift(1)
    df['smoothed_simple_momentum'] = df['simple_momentum'].ewm(span=lookback_period, adjust=False).mean()
    
    # Combine Adjusted and Simple Momentum
    df['combined_momentum'] = df['smoothed_adjusted_momentum'] + df['smoothed_simple_momentum']
    
    # Adjust by Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    sma_length = 10
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=sma_length).mean()
    df['momentum_volume_adjusted'] = df['combined_momentum'] * df['smoothed_volume_ratio']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda x: x['volume'] if x['close'] > x['open'] else 0, axis=1).rolling(window=10).sum()
    df['downward_volume'] = df.apply(lambda x: x['volume'] if x['close'] < x['open'] else 0, axis=1).rolling(window=10).sum()
    df['volume_influence_ratio'] = np.sqrt(df['upward_volume'] / df['downward_volume'])
    
    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    df['integrated_momentum'] = df['momentum_volume_adjusted'] * df['volume_influence_ratio']
    
    # Calculate Enhanced Intraday Reversal
    df['intraday_reversion'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    df['momentum_adjustment'] = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['enhanced_intraday_reversal'] = df['intraday_reversion'] * df['momentum_adjustment'] * df['volume'].rolling(window=5).mean()
    
    # Generate Preliminary Alpha Factor
    df['preliminary_alpha'] = df['enhanced_intraday_reversal'] * df['integrated_momentum'] + (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volume Surprise
    df['volume_surprise'] = df['volume'] - df['volume'].rolling(window=7).mean()
    
    # Combine Price Momentum and Volume Surprise
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    df['momentum_volume_surprise'] = df['price_momentum'] * df['volume_surprise']
    
    # Final Alpha Factor
    df['final_alpha'] = df['preliminary_alpha'] * df['momentum_volume_surprise'] * df['volume_influence_ratio']
    
    return df['final_alpha']
```
