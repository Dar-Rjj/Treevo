```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['price_diff_adj'] = df['close'] - df['high'].shift(1)
    df['momentum_adj_ema'] = df['price_diff_adj'].ewm(span=10, adjust=False).mean()

    # Calculate Simple Price Momentum
    lookback_period = 10
    df['price_diff_simple'] = df['close'] - df['close'].shift(1)
    df['momentum_simple_ema'] = df['price_diff_simple'].ewm(span=lookback_period, adjust=False).mean()

    # Weighted sum of the smoothed adjusted momentum and simple price momentum
    weight_adj = 0.7
    weight_simple = 0.3
    df['combined_momentum'] = weight_adj * df['momentum_adj_ema'] + weight_simple * df['momentum_simple_ema']

    # Adjust by Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['volume_ratio_ema'] = df['volume_ratio'].ewm(span=10, adjust=False).mean()
    df['adjusted_momentum'] = df['combined_momentum'] * df['volume_ratio_ema']

    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['upward_volume_sum'] = df['upward_volume'].rolling(window=lookback_period).sum()
    df['downward_volume_sum'] = df['downward_volume'].rolling(window=lookback_period).sum()
    df['volume_influence_ratio'] = (df['upward_volume_sum'] + 1e-6) / (df['downward_volume_sum'] + 1e-6)

    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    df['integrated_momentum'] = df['adjusted_momentum'] * np.sqrt(df['volume_influence_ratio'])

    # Calculate Enhanced Intraday Reversal
    df['intraday_reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    df['momentum_adjustment'] = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['enhanced_intraday_reversal'] = df['intraday_reversal'] * df['momentum_adjustment'] * df['volume'].rolling(window=5).mean()

    # Generate Preliminary Alpha Factor
    df['prelim_alpha_factor'] = df['enhanced_intraday_reversal'] * df['integrated_momentum']
    df['day_to_day_high_change'] = (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    df['prelim_alpha_factor'] += df['day_to_day_high_change']

    # Calculate Daily Volume Surprise
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] - df['volume_ma_10']

    # Combine Price Momentum and Volume Surprise
    df['high_price_diff'] = df['high'] - df['high'].shift(10)
    df['price_momentum_volume_surprise'] = df['high_price_diff'] * df['volume_surprise']

    # Final Alpha Factor
    df['final_alpha_factor'] = df['prelim_alpha_factor'] * df['price_momentum_volume_surprise'] * df['volume_influence_ratio']

    return df['final_alpha_factor']
```
