```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'].pct_change(5))
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + (df['high'] - df['close'].shift(1)).pct_change().fillna(0)
    
    # Calculate Daily Volume Surprise
    volume_5day_ma = df['volume'].rolling(window=5).mean()
    daily_volume_surprise = df['volume'] - volume_5day_ma
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(5)
    combined_price_momentum_volume_surprise = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = (df['volume'] * (df['close'] > df['open']).astype(int)).rolling(window=5).sum()
    downward_volume = (df['volume'] * (df['close'] < df['open']).astype(int)).rolling(window=5).sum()
    volume_influence_ratio = upward_volume / (downward_volume + 1e-6)  # Add small constant to avoid division by zero
    
    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha_factor * combined_price_momentum_volume_surprise * volume_influence_ratio
    
    return final_alpha_factor
```
