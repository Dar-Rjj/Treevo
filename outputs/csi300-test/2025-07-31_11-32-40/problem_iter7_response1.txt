```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * (1 + volume_change)
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Day-to-Day Open Price Change
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Previous Day Return
    previous_day_return = df['close'].pct_change()
    
    # Generate Alpha Factor
    alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation 
                    + day_to_day_open_price_change - previous_day_return)
    
    return alpha_factor
```
