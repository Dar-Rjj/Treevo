import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(7)) / df['close'].shift(7)
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha = (intraday_reversal * momentum_adjustment) + \
                        ((df['high'] - df['close'].shift(1)) / df['close'].shift(1))
    
    # Calculate Daily Volume Surprise
    volume_ma_7 = df['volume'].rolling(window=7).mean()
    daily_volume_surprise = df['volume'] - volume_ma_7
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(7)
    combined_price_volume = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = df.loc[df['close'] > df['open'], 'volume'].sum()
    downward_volume = df.loc[df['close'] < df['open'], 'volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume if downward_volume != 0 else 0
    
    # Final Alpha Factor
    final_alpha = preliminary_alpha * combined_price_volume * volume_influence_ratio
    
    return final_alpha
