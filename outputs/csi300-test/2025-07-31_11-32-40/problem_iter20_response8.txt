```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['price_diff'] = df['close'] - df['low'].shift(1)
    ema_smooth_factor = 5
    df['adjusted_momentum_ema'] = df['price_diff'].ewm(span=ema_smooth_factor, adjust=False).mean()
    
    # Combine Adjusted Momentum and Simple Price Momentum
    lookback_period = 10
    df['simple_price_momentum'] = df['close'] - df['close'].shift(lookback_period)
    df['simple_momentum_ema'] = df['simple_price_momentum'].ewm(span=ema_smooth_factor, adjust=False).mean()
    df['combined_momentum'] = df['adjusted_momentum_ema'] + df['simple_momentum_ema']
    
    # Adjust by Volume Change
    df['vol_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['vol_ma_5']
    sma_length = 5
    df['volume_ratio_sma'] = df['volume_ratio'].rolling(window=sma_length).mean()
    df['momentum_adjusted_by_volume'] = df['combined_momentum'] * df['volume_ratio_sma']
    
    # Compute Volume Influence Ratio
    df['upward_vol'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1).rolling(window=lookback_period).sum()
    df['downward_vol'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1).rolling(window=lookback_period).sum()
    df['volume_influence_ratio'] = df['upward_vol'] / df['downward_vol']
    
    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    df['integrated_momentum'] = df['momentum_adjusted_by_volume'] * np.sqrt(df['volume_influence_ratio'])
    
    # Calculate Adjusted Intraday Reversal
    df['intraday_reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    df['momentum_adjustment'] = 1 + (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    df['adjusted_intraday_reversal'] = df['intraday_reversal'] * df['momentum_adjustment']
    
    # Calculate Close to Midpoint Deviation
    df['midpoint'] = (df['high'] + df['low']) / 2
    df['close_to_mid_deviation'] = df['close'] - df['midpoint']
    
    # Generate Preliminary Alpha Factor
    df['prelim_alpha_factor'] = (df['adjusted_intraday_reversal'] * df['close_to_mid_deviation']) + ((df['open'] - df['close'].shift(1)) / df['close'].shift(1))
    
    # Calculate Daily Volume Surprise
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] - df['volume_ma_10']
    
    # Combine Price Momentum and Volume Surprise
    df['price_momentum_t_to_t_minus_10'] = df['close'] - df['close'].shift(10)
    df['combined_momentum_and_volume_surprise'] = df['price_momentum_t_to_t_minus_10'] * df['volume_surprise']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['prelim_alpha_factor'] * df['combined_momentum_and_volume_surprise'] * df['volume_influence_ratio']
    
    # Integrate High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    df['final_alpha_factor_with_high_low'] = df['final_alpha_factor'] * (df['high_low_diff'] * df['volume_influence_ratio'])
    
    return df['final_alpha_factor_with_high_low']
```
