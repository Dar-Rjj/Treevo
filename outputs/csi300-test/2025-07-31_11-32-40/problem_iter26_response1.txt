```python
import pandas as pd

def heuristics_v2(df):
    # Compute Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adj = (1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)).fillna(1)
    volume_change_adj = (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)).fillna(1)
    adjusted_intraday_reversal = intraday_reversal * momentum_adj * volume_change_adj

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_mid_deviation = df['close'] - intraday_midpoint

    # Generate Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday Reversal, Close to Midpoint Deviation, and Open Price Change
    combined_intraday_factor = adjusted_intraday_reversal * close_to_mid_deviation + open_price_change

    # Calculate Daily Volume Surprise
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_ma_10

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume']
    downward_volume = df[df['close'] < df['open']]['volume']
    upward_accumulated = upward_volume.rolling(window=10).sum()
    downward_accumulated = downward_volume.rolling(window=10).sum()
    volume_influence_ratio = (upward_accumulated / downward_accumulated).fillna(1)

    # Calculate Smoothed Price Momentum
    price_momentum = (df['close'] - df['close'].shift(1))
    smoothed_price_momentum = price_momentum.ewm(span=10).mean()

    # Adjust by Smoothed Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1)).fillna(1)
    smoothed_volume_ratio = volume_ratio.rolling(window=10).mean()
    combined_momentum_vol_ratio = smoothed_price_momentum * smoothed_volume_ratio

    # Integrate Intraday Range and Volume Dynamics
    intraday_range = df['high'] - df['low']
    integrated_intraday_factor = intraday_range * df['volume']

    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_momentum_vol_ratio * daily_volume_surprise * integrated_intraday_factor

    # Final Alpha Factor
    final_alpha_factor = (
        combined_intraday_factor * intermediate_alpha_factor +
        adjusted_intraday_reversal +
        combined_momentum_vol_ratio
    )

    return final_alpha_factor
```
