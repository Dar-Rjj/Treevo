```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df[df['close'] > df['open']]['volume']).sum()
    downward_volume = (df[df['close'] < df['open']]['volume']).sum()
    volume_influence_ratio = upward_volume / downward_volume if downward_volume != 0 else 0
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    df['volume_10_day_ma'] = df['volume'].rolling(window=10).mean()
    daily_volume_change = df['volume'] - df['volume_10_day_ma']
    volume_momentum_strength = daily_volume_change.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor_1 = high_low_vol_influence * intraday_return * volume_momentum_strength
    
    # Calculate Daily Price Reversal
    df['prev_close'] = df['close'].shift(1)
    price_reversal = df['close'] - df['prev_close']
    price_reversal = price_reversal * df['prev_close'].diff().apply(lambda x: -1 if x > 0 else 1)
    
    # Calculate Volume Surprise
    volume_surprise = df['volume'] - df['volume_10_day_ma']
    
    # Combine Price Reversal and Volume Surprise
    combined_factor_2 = price_reversal * volume_surprise
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = combined_factor_1 * combined_factor_2
    
    return final_alpha_factor
```
