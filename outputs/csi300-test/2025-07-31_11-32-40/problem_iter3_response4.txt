```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Weight by Volume
    volume_weighted_change = close_to_open_change * df['volume']
    
    # Combine Intraday Momentum and Volume Weighted Close-to-Open Change
    combined_factor = intraday_momentum + volume_weighted_change
    
    # Calculate 5-day Moving Average
    moving_average = combined_factor.rolling(window=5).mean()
    
    # Adjust Current Day Factor
    adjusted_factor = combined_factor - moving_average
    
    # Compare with Zero and Assign Score
    score = adjusted_factor.apply(lambda x: 1 if x > 0 else -1)
    
    return score
```
