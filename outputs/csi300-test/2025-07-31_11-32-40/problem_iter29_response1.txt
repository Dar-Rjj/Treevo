```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal: 2 * (High - Low) / (Close + Open)
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Volume Change: (1 + (Volume_t - Volume_{t-1}) / Volume_{t-1})
    volume_change_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    intraday_reversal *= volume_change_adjustment
    
    # Calculate Close to Midpoint Deviation: Close - (High + Low) / 2
    close_to_midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Generate Preliminary Alpha Factor
    short_term_momentum_adjustment = 1 + (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    preliminary_alpha_factor = intraday_reversal * short_term_momentum_adjustment + (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - df['volume'].rolling(window=5).mean()
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(5)
    combined_momentum_and_volume = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume
    
    # Integrate Core Components into Intermediate Alpha Factor
    intermediate_alpha_factor = combined_momentum_and_volume * (intraday_reversal * volume_influence_ratio * (df['close'] - df['open']) / (df['close'] + df['open']))
    
    # Calculate Adjusted Intraday Reversal
    adjusted_intraday_reversal = intraday_reversal * volume_change_adjustment
    
    # Calculate Net Directional Adjusted Volume
    upward_adjusted_volume = ((df['close'] - df['open']) * df['volume']).where(df['close'] > df['open'], 0).cumsum()
    downward_adjusted_volume = ((df['open'] - df['close']) * df['volume']).where(df['close'] < df['open'], 0).cumsum()
    net_directional_adjusted_volume = upward_adjusted_volume - downward_adjusted_volume
    
    # Multiply High-Low Difference with Net Directional Adjusted Volume
    high_low_difference_with_net_volume = (df['high'] - df['low']) * net_directional_adjusted_volume
    
    # Calculate Day-to-Day Open Price Change
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor Components
    final_alpha_factor_components = (intraday_reversal * volume_influence_ratio + adjusted_intraday_reversal + day_to_day_open_price_change)
    
    # Final Alpha Factor
    final_alpha_factor = final_alpha_factor_components + high_low_difference_with_net_volume
    
    return final_alpha_factor
```
