import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    open_interest_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Alpha Factor Base
    alpha_factor_base = (adjusted_intraday_reversal * close_to_midpoint_deviation) + day_to_day_open_change
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    close_open_return_5d_ma = close_open_return.rolling(window=5).mean()
    
    # Calculate 5-Day Moving Average of Intraday High-Low Spread
    high_low_spread_5d_ma = high_low_spread.rolling(window=5).mean()
    
    # Calculate Weighted Intraday Return
    weighted_intraday_return = (close_open_return * 0.6) + (high_low_spread * 0.4)
    
    # Combine Weighted Intraday Return, Moving Averages, and Volume Surge
    intraday_momentum = (weighted_intraday_return * close_open_return_5d_ma) + (high_low_spread * high_low_spread_5d_ma)
    intraday_momentum = intraday_momentum * volume_surge
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / downward_volume
    
    # Final Alpha Factor Components
    final_alpha_components = (intraday_momentum * volume_influence_ratio) + adjusted_intraday_reversal + day_to_day_open_change
    
    # Final Alpha Factor
    combined_intraday_reversal_momentum = final_alpha_components + volume_surge
    
    return combined_intraday_reversal_momentum
