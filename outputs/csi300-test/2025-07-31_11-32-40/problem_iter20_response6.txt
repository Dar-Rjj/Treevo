```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Calculate Previous Day Return
    previous_day_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor 1
    intermediate_alpha_factor_1 = (intraday_range * close_to_midpoint_deviation) - previous_day_return
    
    # Calculate Daily Volume Surprise
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_ma_10
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = (df['close'] - df['close'].shift(10))
    combined_price_vol_momentum = price_momentum * daily_volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_difference = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * df['volume']
    cumulative_upward_volume = upward_volume.rolling(window=10).sum()
    cumulative_downward_volume = downward_volume.rolling(window=10).sum()
    volume_influence_ratio = cumulative_upward_volume / cumulative_downward_volume.replace(0, 1)
    
    # Adjust Momentum by Volume and Amount
    weighted_high_low_return = high_low_difference * df['volume'] * volume_influence_ratio
    volume_adjusted_momentum = weighted_high_low_return.rolling(window=10).sum()
    
    # Introduce Time Decay
    decay_rate = 0.95
    time_decay = (decay_rate ** np.arange(10, 0, -1)).reshape(-1, 1)
    volume_adjusted_momentum = volume_adjusted_momentum.rolling(window=10).apply(lambda x: np.sum(x * time_decay), raw=True)
    
    # Intermediate Alpha Factor Synthesis 1
    intermediate_alpha_factor_synthesis_1 = combined_price_vol_momentum * high_low_difference
    
    # Compute Price Momentum
    price_momentum_ema = df['close'].diff().ewm(span=10).mean()
    
    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    smoothed_volume_ratio = volume_ratio.rolling(window=10).mean()
    combined_momentum_vol_ratio = price_momentum_ema * smoothed_volume_ratio
    
    # Compute Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Spread and Close-Open Return
    intraday_momentum = high_low_difference * intraday_close_open_return
    
    # Consider Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Adjust for Open Interest
    adjusted_intraday_reversal_with_open_interest = adjusted_intraday_reversal * (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1))
    
    # Intermediate Alpha Factor Synthesis 2
    intermediate_alpha_factor_synthesis_2 = combined_momentum_vol_ratio * intraday_momentum * adjusted_intraday_reversal_with_open_interest
    
    # Cumulative Directional Volume Impact
    cumulative_directional_volume_impact = high_low_difference * volume_influence_ratio
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor_1 * intermediate_alpha_factor_synthesis_2 * cumulative_directional_volume_impact + daily_volume_surprise
    
    return final_alpha_factor
```
