```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    open_interest_adjustment = 1 + volume_change
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment
    
    # Calculate Volume Weighted Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    volume_weighted_close_to_open = close_to_open_change * df['volume']
    
    # Consider Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Adjustments
    combined_adjusted_reversal = adjusted_intraday_reversal + day_to_day_open_change
    
    # Combine Intraday Momentum, Volume-Weighted Close-to-Open Change, and Adjusted Intraday Reversal
    combined_factor = intraday_momentum + volume_weighted_close_to_open + combined_adjusted_reversal
    
    # Calculate 5-day Moving Average of the Combined Factor
    moving_average_5d = combined_factor.rolling(window=5).mean()
    
    # Compare with Current Day Factor
    score = (combined_factor > moving_average_5d).astype(int) - (combined_factor < moving_average_5d).astype(int)
    
    return score
```
