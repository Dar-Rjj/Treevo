```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal: 2 * (High - Low) / (Close + Open)
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum: (1 + (Close_t - Close_{t-10}) / Close_{t-10})
    momentum_adjustment = 1 + (df['close'].diff(10) / df['close'].shift(10))
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Adjust for Open Interest using Volume
    volume_adjustment = 1 + (df['volume'].pct_change())
    adjusted_intraday_reversal *= volume_adjustment
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Day-to-Day Open Price Change
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (adjusted_intraday_reversal * close_to_midpoint_deviation) + day_to_day_open_price_change
    
    # Calculate Daily Volume Surprise
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_10d_ma
    
    # Calculate Daily Price Momentum
    daily_price_momentum = df['close'].diff(10)
    
    # Combine Price Momentum and Volume Surprise
    combined_price_momentum_volume_surprise = daily_price_momentum * daily_volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_price_difference = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].rolling(window=10).sum()
    downward_volume = df[df['close'] < df['open']]['volume'].rolling(window=10).sum()
    volume_influence_ratio = upward_volume / downward_volume.fillna(1)  # Fill NA with 1 to avoid division by zero
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_price_momentum_volume_surprise * high_low_price_difference
    
    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha_factor * intermediate_alpha_factor * volume_influence_ratio
    
    return final_alpha_factor
```
