import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative volume (volume / average volume)
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    df['relative_volume'] = df['volume'] / df['avg_volume_30']
    
    # Calculate the money flow ratio (amount / volume)
    df['money_flow_ratio'] = df['amount'] / df['volume']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['relative_range'] * df['relative_volume'] * df['money_flow_ratio']).rolling(window=5).mean()
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
