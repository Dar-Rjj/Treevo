```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns using close prices
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute rolling window mean of log returns
    df['log_returns_20d_mean'] = df['log_returns'].rolling(window=20).mean()
    df['log_returns_50d_mean'] = df['log_returns'].rolling(window=50).mean()
    
    # Calculate daily range (High - Low)
    df['daily_range'] = df['high'] - df['low']
    df['daily_range_30d_ma'] = df['daily_range'].rolling(window=30).mean()
    df['daily_range_100d_ma'] = df['daily_range'].rolling(window=100).mean()
    
    # Calculate intraday reversal adjusted for momentum
    df['intraday_reversal'] = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate daily volume changes
    df['volume_change_pct'] = df['volume'].pct_change()
    
    # Calculate Pearson correlation between volume and close prices
    df['volume_close_corr_60d'] = df['close'].rolling(window=60).corr(df['volume'])
    
    # Generate intermediate alpha factor 1
    df['midpoint_deviation'] = (df['close'] - (df['high'] + df['low']) / 2) / df['close'].shift(1)
    df['intermediate_alpha_factor_1'] = (df['daily_range'] * df['close']) * df['midpoint_deviation'] - df['log_returns'].shift(1)
    
    # Combine price momentum and volume surprise
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    df['volume_surprise'] = df['volume'] - df['volume'].rolling(window=10).mean()
    df['combined_price_volume'] = df['price_momentum'] * df['volume_surprise']
    
    # Calculate daily volume surprise
    df['volume_ma_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] - df['volume_ma_10d']
    
    # Calculate high-low price difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute volume influence ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['volume_influence_ratio'] = df['upward_volume'].cumsum() / df['downward_volume'].cumsum()
    
    # Adjust momentum by volume and amount
    df['volume_weighted_high_low'] = df['high_low_diff'] * df['volume'] * df['volume_influence_ratio']
    
    # Introduce time decay
    decay_rate = 0.95
    df['decay_adjusted_momentum'] = df['volume_weighted_high_low'] * (decay_rate ** (df.index.to_series().diff().dt.days - 1))
    
    # Intermediate Alpha Factor Synthesis 1
    df['intermediate_alpha_factor_1'] = df['combined_price_volume'] * df['high_low_diff']
    
    # Adjust for open interest
    df['intraday_reversal_adjusted'] = df['intraday_reversal'] * (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1))
    
    # Intermediate Alpha Factor Synthesis 2
    df['intermediate_alpha_factor_2'] = df['intraday_reversal_adjusted'] * df['high_low_diff'] * df['log_returns']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['intermediate_alpha_factor_1'] * df['volume_influence_ratio'] + df['volume_surprise']
    
    return df['final_alpha_factor'].dropna()
```
