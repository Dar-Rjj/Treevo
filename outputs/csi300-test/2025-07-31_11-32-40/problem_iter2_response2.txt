```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    open_interest_adj = 1 + (df['volume'].diff() / df['volume'].shift(1))
    intraday_reversal_adjusted = intraday_reversal * open_interest_adj
    
    # Consider Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Reversal and Open Interest Adjustments
    intraday_combined = intraday_reversal_adjusted + open_price_change
    
    # Calculate Price Momentum
    price_diff = df['close'] - df['close'].shift(1)
    momentum_ema = price_diff.ewm(span=5, adjust=False).mean()
    
    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1)
    volume_ratio_sma = volume_ratio.rolling(window=5).mean()
    momentum_adjusted = momentum_ema * volume_ratio_sma
    
    # Final Alpha Factor
    alpha_factor = 0.6 * intraday_combined + 0.4 * momentum_adjusted
    
    return alpha_factor
```
