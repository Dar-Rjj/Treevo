import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Generate Preliminary Alpha Factor
    preliminary_alpha = (adjusted_intraday_reversal * close_to_midpoint_deviation) + \
                        (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Calculate Daily Volume Surprise
    volume_ma_20 = df['volume'].rolling(window=20).mean()
    daily_volume_surprise = df['volume'] - volume_ma_20

    # Combine Price Momentum and Volume Surprise
    price_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    combined_price_momentum_volume_surprise = price_momentum * daily_volume_surprise

    # Compute Volume Influence Ratio
    upward_volume = (df['volume'] * (df['close'] > df['open'])).rolling(window=20).sum()
    downward_volume = (df['volume'] * (df['close'] < df['open'])).rolling(window=20).sum()
    volume_influence_ratio = upward_volume / downward_volume

    # Intermediate Alpha Factor Synthesis
    high_low_difference = df['high'] - df['low']
    intermediate_alpha_factor = combined_price_momentum_volume_surprise * high_low_difference

    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha * intermediate_alpha_factor * volume_influence_ratio

    return final_alpha_factor
