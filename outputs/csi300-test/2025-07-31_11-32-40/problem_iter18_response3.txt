```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].rolling(window=10, min_periods=1).sum()
    downward_volume = df[df['close'] < df['open']]['volume'].rolling(window=10, min_periods=1).sum()
    volume_influence_ratio = (upward_volume / downward_volume).fillna(1)
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    volume_10_day_ma = df['volume'].rolling(window=10, min_periods=1).mean()
    daily_volume_change = df['volume'] - volume_10_day_ma
    volume_momentum_strength = np.where(daily_volume_change > 0, 1, -1)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor_1 = (high_low_diff * volume_influence_ratio * intraday_return) * volume_momentum_strength
    
    # Calculate Adjusted Daily Momentum
    adjusted_daily_momentum = df['close'] - df['close'].shift(10)
    
    # Adjust by High-Low Spread
    high_low_spread = df['high'] - df['low']
    ema_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Incorporate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Adjusted Momentum, High-Low Spread, and Intraday Return
    combined_factor_2 = adjusted_daily_momentum * ema_high_low_spread + intraday_return
    
    # Calculate Volume Surprise
    volume_surprise = df['volume'] - df['volume'].rolling(window=10, min_periods=1).mean()
    
    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].shift(1).fillna(1)
    sma_volume_ratio = volume_ratio.rolling(window=10, min_periods=1).mean()
    
    # Combine Volume Impact Factors
    combined_volume_impact = volume_surprise * sma_volume_ratio
    
    # Incorporate High-Frequency Price Movements
    high_open_diff = df['high'] - df['open']
    low_open_diff = df['low'] - df['open']
    high_low_open_imbalance = high_open_diff * low_open_diff
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = (combined_factor_1 * combined_factor_2 + high_low_open_imbalance) * volume_influence_ratio
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor + intraday_return * volume_influence_ratio + combined_volume_impact
    
    return final_alpha_factor
```
