import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * (1 + volume_change)

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Calculate Price Momentum
    lookback_period = 10
    price_momentum = df['close'].pct_change().ewm(span=lookback_period, adjust=False).mean()

    # Adjust by Volume Change
    volume_lookback_period = 10
    volume_ratio = df['volume'] / df['volume'].rolling(window=volume_lookback_period).mean()
    smoothed_volume_ratio = volume_ratio.ewm(span=volume_lookback_period, adjust=False).mean()
    momentum_and_volume = price_momentum * smoothed_volume_ratio

    # Compute High-Low Price Difference
    high_low_price_difference = df['high'] - df['low']

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / (downward_volume + 1e-6)  # Add small constant to avoid division by zero

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Identify Volume Trend
    volume_change_trend = volume_change.apply(lambda x: 1.5 if x > 0 else 0.5)

    # Integrate Core Components into Intermediate Alpha Factor
    intermediate_alpha_factor = momentum_and_volume * (high_low_price_difference * volume_influence_ratio * intraday_return * volume_change_trend)

    # Generate Alpha Factor Base
    alpha_factor_base = adjusted_intraday_reversal * close_to_midpoint_deviation + (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']

    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)

    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    ma_5d_close_open_return = intraday_close_open_return.rolling(window=5).mean()

    # Calculate 5-Day Moving Average of Intraday High-Low Spread
    ma_5d_high_low_spread = intraday_high_low_spread.rolling(window=5).mean()

    # Calculate Weighted Intraday Return
    weighted_intraday_return = 0.6 * intraday_close_open_return + 0.4 * intraday_high_low_spread

    # Combine Weighted Intraday Return, Moving Averages, and Volume Surge
    intraday_momentum = (weighted_intraday_return * ma_5d_close_open_return + intraday_high_low_spread * ma_5d_high_low_spread) * volume_surge

    # Final Alpha Factor Components
    final_alpha_factor = intermediate_alpha_factor + intraday_momentum * volume_influence_ratio + adjusted_intraday_reversal + (df['open'] - df['close'].shift(1)) / df['close'].shift(1) + volume_surge

    return final_alpha_factor
