import pandas as pd

def heuristics(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    open_interest_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Calculate Day-to-Day Open Price Change
    day_to_day_open_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Generate Alpha Factor Base
    alpha_factor_base = adjusted_intraday_reversal * close_to_midpoint_deviation + day_to_day_open_change

    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']

    # Calculate 5-Day Moving Averages
    moving_avg_close_open_return = intraday_close_open_return.rolling(window=5).mean()
    moving_avg_high_low_spread = intraday_high_low_spread.rolling(window=5).mean()

    # Calculate Weighted Intraday Return
    weighted_intraday_return = (intraday_close_open_return * 0.6) + (intraday_high_low_spread * 0.4)

    # Combine Weighted Intraday Return and Moving Averages
    intraday_momentum = (weighted_intraday_return * moving_avg_close_open_return) + (intraday_high_low_spread * moving_avg_high_low_spread)

    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].rolling(window=5).sum()
    downward_volume = df[df['close'] < df['open']]['volume'].rolling(window=5).sum()
    volume_influence_ratio = upward_volume / downward_volume

    # Final Alpha Factor Components
    final_alpha_factor_components = (intraday_momentum * volume_influence_ratio) + adjusted_intraday_reversal + day_to_day_open_change

    # Final Alpha Factor
    final_alpha_factor = final_alpha_factor_components + (df['volume'] - df['volume'].shift(1))

    return final_alpha_factor
