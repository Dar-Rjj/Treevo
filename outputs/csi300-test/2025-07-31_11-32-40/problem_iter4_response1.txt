import pandas as pd

def heuristics_v2(df):
    # Define lookback periods and smoothing factors
    price_lookback = 20
    ema_smoothing_factor = 0.1
    volume_lookback = 5
    sma_length = 5

    # Calculate Price Momentum
    df['price_momentum'] = df['close'].diff()

    # Smooth the momentum with Exponential Moving Average (EMA)
    df['smoothed_price_momentum'] = df['price_momentum'].ewm(alpha=ema_smoothing_factor).mean()

    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)

    # Smooth the volume ratio with Simple Moving Average (SMA)
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=sma_length).mean()

    # Combine Momentum and Volume Ratio
    df['alpha_factor'] = df['smoothed_price_momentum'] * df['smoothed_volume_ratio']

    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
