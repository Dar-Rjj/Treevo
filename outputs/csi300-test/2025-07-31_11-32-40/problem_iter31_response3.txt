```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment

    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint

    # Generate Alpha Factor Base
    day_to_day_high_price_change = (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    alpha_factor_base = (adjusted_intraday_reversal * close_to_midpoint_deviation) + day_to_day_high_price_change

    # Calculate Daily Volume Surprise
    volume_5d_ma = df['volume'].rolling(window=5).mean()
    daily_volume_surprise = df['volume'] - volume_5d_ma

    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']

    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']

    # Calculate 5-Day Moving Averages
    intraday_close_open_return_5d_ma = intraday_close_open_return.rolling(window=5).mean()
    intraday_high_low_spread_5d_ma = intraday_high_low_spread.rolling(window=5).mean()

    # Calculate Weighted Intraday Return
    weighted_intraday_return = (intraday_close_open_return * 0.6) + (intraday_high_low_spread * 0.4)

    # Combine Weighted Intraday Return and Moving Averages
    combined_weighted_return = (weighted_intraday_return * intraday_close_open_return_5d_ma) + \
                               (intraday_high_low_spread * intraday_high_low_spread_5d_ma)

    # Final Alpha Factor
    volume_influence_ratio = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_alpha_factor = (combined_weighted_return * daily_volume_surprise) * volume_influence_ratio + \
                         alpha_factor_base + volume_influence_ratio ** 0.5 + (df['volume'] - df['volume'].shift(1))

    return final_alpha_factor
```
