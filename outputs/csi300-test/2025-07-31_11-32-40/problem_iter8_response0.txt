```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Smooth the Price Momentum with Exponential Moving Average (EMA)
    ema_smoothing_factor = 0.9
    smoothed_price_momentum = price_momentum.ewm(alpha=ema_smoothing_factor, adjust=False).mean()
    
    # Smooth Intraday Close-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Adjust by Volume Change through Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Accumulate Volume on Upward Days
    upward_volume = (df['close'] > df['open']).astype(int) * df['volume']
    
    # Accumulate Volume on Downward Days
    downward_volume = (df['close'] < df['open']).astype(int) * df['volume']
    
    # Compute Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Combine Price Momentum and Intraday High-Low Spread
    combined_momentum_intraday = smoothed_price_momentum * intraday_high_low_spread
    
    # Integrate Volume Ratio and Volume Surge
    integrated_volume_adjustment = volume_ratio + volume_surge
    
    # Incorporate Volume Influence into Momentum and Intraday Pattern
    volume_influence = upward_volume.sum() / downward_volume.sum()
    
    # Synthesize Components
    final_factor = (combined_momentum_intraday * volume_influence) + integrated_volume_adjustment
    
    return final_factor
```
