```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Adjusted Intraday Reversal
    df['Intraday_Reversal'] = 2 * (df['High'] - df['Low']) / (df['Close'] + df['Open'])
    df['Momentum_Adjustment'] = (df['Close'] - df['Close'].shift(10)) / df['Close'].shift(10)
    df['Volume_Change_Adjustment'] = (df['Volume'] - df['Volume'].shift(5)) / df['Volume'].shift(5)
    df['Adjusted_Intraday_Reversal'] = df['Intraday_Reversal'] * (1 + df['Momentum_Adjustment']) * (1 + df['Volume_Change_Adjustment'])

    # Calculate Close to Midpoint Deviation
    df['Midpoint'] = (df['High'] + df['Low']) / 2
    df['Close_to_Midpoint_Deviation'] = df['Close'] - df['Midpoint']

    # Generate Day-to-Day Open Price Change
    df['Open_Price_Change'] = (df['Open'] - df['Close'].shift(1)) / df['Close'].shift(1)

    # Combine Intraday Reversal, Close to Midpoint Deviation, and Open Price Change
    df['Combined_Factor'] = df['Adjusted_Intraday_Reversal'] * df['Close_to_Midpoint_Deviation'] + df['Open_Price_Change']

    # Calculate Daily Volume Surprise
    df['Volume_MA_5'] = df['Volume'].rolling(window=5).mean()
    df['Daily_Volume_Surprise'] = df['Volume'] - df['Volume_MA_5']

    # Compute Volume Influence Ratio
    df['Upward_Volume'] = df.apply(lambda row: row['Volume'] if row['Close'] > row['Open'] else 0, axis=1).rolling(window=20).sum()
    df['Downward_Volume'] = df.apply(lambda row: row['Volume'] if row['Close'] < row['Open'] else 0, axis=1).rolling(window=20).sum()
    df['Volume_Influence_Ratio'] = df['Upward_Volume'] / df['Downward_Volume']

    # Calculate Smoothed Price Momentum
    df['Price_Momentum'] = df['Close'] - df['Close'].shift(20)
    df['Smoothed_Price_Momentum'] = df['Price_Momentum'].ewm(span=20).mean()

    # Adjust by Smoothed Volume Change
    df['Volume_Ratio'] = df['Volume'] / df['Volume'].shift(5)
    df['Smoothed_Volume_Ratio'] = df['Volume_Ratio'].rolling(window=5).mean()
    df['Combined_Momentum_and_Volume_Ratio'] = df['Smoothed_Price_Momentum'] * df['Smoothed_Volume_Ratio']

    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Combined_Factor'] * df['Volume_Influence_Ratio'] +
        df['Adjusted_Intraday_Reversal'] +
        df['Combined_Momentum_and_Volume_Ratio']
    )

    # Calculate Previous Day Return
    df['Prev_Day_Return'] = (df['Open'] - df['Close'].shift(1)) / df['Close'].shift(1)

    return df['Alpha_Factor']
```
