```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Final Alpha Factor
    alpha_factor = (intraday_high_low_spread * intraday_close_open_return) + volume_surge
    
    return alpha_factor
```
