```python
import pandas as pd

def heuristics(df):
    # Calculate Adjusted Intraday Reversal
    df['Intraday_Reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Intraday_Reversal'] = (1 + momentum) * (1 + volume_change)

    # Calculate Close to Midpoint Deviation
    df['Intraday_Midpoint'] = (df['high'] + df['low']) / 2
    df['Close_to_Midpoint_Deviation'] = df['close'] - df['Intraday_Midpoint']

    # Calculate Day-to-Day Open Price Change
    df['Day_to_Day_Open_Price_Change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Generate Preliminary Alpha Factor
    df['Preliminary_Alpha_Factor'] = (df['Adjusted_Intraday_Reversal'] * df['Close_to_Midpoint_Deviation']
                                      + df['Day_to_Day_Open_Price_Change'] - df['close'].pct_change())

    # Calculate Daily Volume Surprise
    df['Volume_10d_MA'] = df['volume'].rolling(window=10).mean()
    df['Daily_Volume_Surprise'] = df['volume'] - df['Volume_10d_MA']

    # Combine Price Momentum and Volume Surprise
    price_momentum = (df['close'] - df['close'].shift(10))
    df['Price_Momentum_and_Volume_Surprise'] = price_momentum * df['Daily_Volume_Surprise']

    # Compute Volume Influence Ratio
    df['Up_Volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['Down_Volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['Volume_Influence_Ratio'] = df['Up_Volume'].cumsum() / df['Down_Volume'].cumsum()

    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']

    # Calculate Intraday Close-Open Return
    df['Intraday_Close_Open_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    df['5d_MA_Intraday_Close_Open_Return'] = df['Intraday_Close_Open_Return'].rolling(window=5).mean()

    # Calculate 5-Day Moving Average of Intraday High-Low Spread
    df['5d_MA_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'].rolling(window=5).mean()

    # Calculate Weighted Intraday Return
    df['Weighted_Intraday_Return'] = (df['Intraday_Close_Open_Return'] * 0.6 
                                      + df['Intraday_High_Low_Spread'] * 0.4)

    # Combine Weighted Intraday Return, Moving Averages, and Volume Surge
    df['Weighted_Intraday_Return_Component'] = (df['Weighted_Intraday_Return'] 
                                                * df['5d_MA_Intraday_Close_Open_Return'])
    df['High_Low_Spread_Component'] = (df['Intraday_High_Low_Spread'] 
                                       * df['5d_MA_Intraday_High_Low_Spread'])
    df['Intraday_Momentum'] = (df['Weighted_Intraday_Return_Component'] + df['High_Low_Spread_Component']
                               ) * df['Daily_Volume_Surprise']

    # Final Alpha Factor
    df['Intermediate_Alpha_Factor'] = df['Preliminary_Alpha_Factor'] * df['Price_Momentum_and_Volume_Surprise'] * df['Volume_Influence_Ratio']
    df['Final_Alpha_Factor'] = (df['Intermediate_Alpha_Factor'] + df['Intraday_Momentum'] + df['Daily_Volume_Surprise'])

    return df['Final_Alpha_Factor']
```
