```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = (upward_volume / downward_volume).fillna(1)
    
    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    ma_5d_close_open_return = intraday_close_open_return.rolling(window=5).mean()
    
    # Calculate 5-Day Moving Average of Intraday High-Low Spread
    ma_5d_high_low_spread = intraday_high_low_spread.rolling(window=5).mean()
    
    # Calculate Weighted Intraday Return
    weighted_intraday_return = (intraday_close_open_return * 0.6) + (intraday_high_low_spread * 0.4)
    
    # Combine Weighted Intraday Return, Moving Averages, and Volume Surge
    intraday_momentum = (weighted_intraday_return * ma_5d_close_open_return) + (intraday_high_low_spread * ma_5d_high_low_spread)
    intraday_momentum = intraday_momentum * volume_surge
    
    # Adjust for Open Interest
    open_interest_adjustment = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    adjusted_intraday_reversal = intraday_reversal * (1 + open_interest_adjustment)
    
    # Calculate Volume Weighted Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    volume_weighted_close_to_open_change = close_to_open_change * df['volume']
    
    # Consider Day-to-Day Open Price Change
    day_to_day_open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Final Alpha Factor Components
    final_alpha_factor = (
        intraday_momentum * volume_influence_ratio + 
        volume_weighted_close_to_open_change + 
        adjusted_intraday_reversal + 
        day_to_day_open_price_change
    )
    
    # Calculate 5-day Moving Average of the Final Alpha Factor
    ma_5d_final_alpha_factor = final_alpha_factor.rolling(window=5).mean()
    
    # Compare with Current Day Factor
    score = final_alpha_factor.gt(ma_5d_final_alpha_factor).astype(int).replace({True: 1, False: -1})
    
    return score
```
