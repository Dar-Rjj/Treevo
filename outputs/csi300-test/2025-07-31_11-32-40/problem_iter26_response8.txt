```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['PriceDiff_Adjusted'] = df['close'] - df['low'].shift(1)
    ema_smooth_factor = 5
    df['Momentum_Adjusted'] = df['PriceDiff_Adjusted'].ewm(span=ema_smooth_factor).mean()

    # Combine Adjusted Momentum and Simple Price Momentum
    lookback_period = 10
    df['Simple_Price_Momentum'] = df['close'] - df['close'].shift(1)
    df['Smoothed_Simple_Momentum'] = df['Simple_Price_Momentum'].ewm(span=lookback_period).mean()
    df['Combined_Momentum'] = df['Momentum_Adjusted'] + df['Smoothed_Simple_Momentum']

    # Adjust by Volume Change
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    sma_length = 5
    df['Smoothed_Volume_Ratio'] = df['Volume_Ratio'].rolling(window=sma_length).mean()
    df['Adjusted_Momentum'] = df['Combined_Momentum'] * df['Smoothed_Volume_Ratio']

    # Compute Volume Influence Ratio
    df['Upward_Volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['Downward_Volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['Volume_Influence_Ratio'] = df['Upward_Volume'].rolling(window=lookback_period).sum() / df['Downward_Volume'].rolling(window=lookback_period).sum()

    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    df['Integrated_Momentum'] = df['Adjusted_Momentum'] * np.sqrt(df['Volume_Influence_Ratio'])

    # Calculate Enhanced Intraday Reversal
    df['Intraday_Reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    df['Momentum_Adjustment'] = 1 + (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    df['Enhanced_Intraday_Reversal'] = df['Intraday_Reversal'] * df['Momentum_Adjustment'] * df['volume'].rolling(window=5).mean()

    # Generate Preliminary Alpha Factor
    df['Preliminary_Alpha_Factor'] = df['Enhanced_Intraday_Reversal'] * df['Integrated_Momentum']
    df['Day_to_Day_Open_Price_Change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Preliminary_Alpha_Factor'] += df['Day_to_Day_Open_Price_Change']

    # Calculate Daily Volume Surprise
    df['Volume_10d_MA'] = df['volume'].rolling(window=10).mean()
    df['Daily_Volume_Surprise'] = df['volume'] - df['Volume_10d_MA']

    # Combine Price Momentum and Volume Surprise
    df['Price_Momentum_Trend'] = df['close'] - df['close'].shift(lookback_period)
    df['Combined_PV_Factor'] = df['Price_Momentum_Trend'] * df['Daily_Volume_Surprise']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Preliminary_Alpha_Factor'] * df['Combined_PV_Factor'] * df['Volume_Influence_Ratio']

    return df['Final_Alpha_Factor']
```
