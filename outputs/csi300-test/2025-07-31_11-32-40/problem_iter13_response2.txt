```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    momentum_lookback = 20
    volume_lookback = 10
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'].pct_change(periods=momentum_lookback)
    df['smoothed_momentum'] = df['price_momentum'].ewm(span=momentum_lookback, adjust=False).mean()
    
    # Adjust by Volume Change
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=volume_lookback).mean()
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=volume_lookback).mean()
    df['momentum_adjusted'] = df['smoothed_momentum'] * df['smoothed_volume_ratio']
    
    # Compute High-Low Price Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['upward_volume_sum'] = df['upward_volume'].rolling(window=volume_lookback).sum()
    df['downward_volume_sum'] = df['downward_volume'].rolling(window=volume_lookback).sum()
    df['volume_influence_ratio'] = df['upward_volume_sum'] / (df['downward_volume_sum'] + 1e-6)  # Add small constant to avoid division by zero
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Trend
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_trend_strength'] = df['volume_change'].apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Integrate All Components into Final Alpha Factor
    df['alpha_factor'] = (
        df['momentum_adjusted'] * 
        df['high_low_diff'] * 
        df['volume_influence_ratio'] * 
        df['intraday_return'] * 
        df['volume_trend_strength']
    )
    
    return df['alpha_factor']
```
