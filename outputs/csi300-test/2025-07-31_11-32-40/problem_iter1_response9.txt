```python
def heuristics_v2(df):
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(20)
    
    # Combine Price Momentum and Volume Change
    alpha_factor = price_momentum * volume_change
    
    return alpha_factor
```
