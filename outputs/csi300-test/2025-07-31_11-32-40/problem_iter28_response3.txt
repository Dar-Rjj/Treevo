import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Weighted Cumulative Directional Price Movement Adjusted by Volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume)
    
    Returns:
    pd.Series: Alpha factor values indexed by date
    """
    # Calculate High-Low Percentage Change
    high_low_pct_change = (df['high'] - df['low']) / df['low']

    # Compute Weighted Directional Volume
    upward_volume = ((df['close'] - df['open']) * df['volume'] * (df['close'] / df['open'])).where(df['close'] > df['open'], 0).cumsum()
    downward_volume = ((df['open'] - df['close']) * df['volume'] * (df['open'] / df['close'])).where(df['close'] < df['open'], 0).cumsum()
    net_weighted_directional_volume = upward_volume - downward_volume

    # Multiply High-Low Percentage Change with Net Weighted Directional Volume
    alpha_factor = high_low_pct_change * net_weighted_directional_volume

    return alpha_factor
