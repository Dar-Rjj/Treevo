```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['PriceMomentum'] = df['close'].diff(10)

    # Calculate High-Low Price Spread
    df['HighLowSpread'] = df['high'] - df['low']

    # Compute Intraday Close-Open Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']

    # Combine High-Low Spread and Close-Open Return
    df['IntradayMomentum'] = df['HighLowSpread'] * df['IntradayReturn']

    # Calculate Volume Surprise
    df['VolumeMA'] = df['volume'].rolling(window=10).mean()
    df['VolumeSurprise'] = df['volume'] - df['VolumeMA']
    df['IntermediateAlphaFactor'] = df['PriceMomentum'] * df['VolumeSurprise']

    # Calculate Volume Surge
    df['VolumeSurge'] = df['volume'] - df['volume'].shift(1)

    # Compute Volume Influence Ratio
    df['UpwardVolume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1).cumsum()
    df['DownwardVolume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1).cumsum()
    df['VolumeInfluenceRatio'] = df['UpwardVolume'] / df['DownwardVolume'].replace(0, 1)  # Avoid division by zero

    # Multiply High-Low Spread with Volume Influence Ratio
    df['MultipliedHighLowSpread'] = df['HighLowSpread'] * df['VolumeInfluenceRatio']

    # Combine All Alpha Factors
    df['FinalAlphaFactor'] = (
        df['IntermediateAlphaFactor'] * df['MultipliedHighLowSpread'] +
        df['IntradayMomentum'] * df['VolumeInfluenceRatio'] +
        df['VolumeSurge']
    )

    return df['FinalAlphaFactor'].dropna()
```
