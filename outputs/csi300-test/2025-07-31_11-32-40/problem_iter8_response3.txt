```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open']).fillna(0)
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1).fillna(0)
    
    # Weigh Metrics
    weighted_high_low_spread = intraday_high_low_spread * 0.5
    weighted_close_open_return = intraday_close_open_return * 0.3
    weighted_midpoint_deviation = close_to_midpoint_deviation * 0.2
    
    # Combine Weighted Metrics
    combined_weighted_metrics = (weighted_high_low_spread +
                                 weighted_close_open_return +
                                 weighted_midpoint_deviation)
    
    # Adjust for Previous Day Return
    adjusted_combined_metrics = combined_weighted_metrics * (1 - previous_day_return)
    
    # Calculate 5-Day Moving Average of Intraday Close-Open Return
    moving_avg_close_open_return = intraday_close_open_return.rolling(window=5).mean().fillna(0)
    
    # Calculate 5-Day Moving Average of Intraday High-Low Spread
    moving_avg_high_low_spread = intraday_high_low_spread.rolling(window=5).mean().fillna(0)
    
    # Combine Weighted Metrics, Moving Averages, and Volume Surge
    intraday_momentum_and_reversal = (adjusted_combined_metrics * moving_avg_close_open_return +
                                      intraday_high_low_spread * moving_avg_high_low_spread) * volume_surge
    
    # Final Alpha Factor
    final_alpha_factor = intraday_momentum_and_reversal + volume_surge
    
    return final_alpha_factor
```
