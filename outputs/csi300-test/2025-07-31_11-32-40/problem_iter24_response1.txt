```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple Moving Average of closing prices over a 10-day window
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Difference between the current day's close and its 10-day moving average
    df['Close_SMA_Diff'] = df['close'] - df['SMA_10']
    
    # Ratio of the highest high to the lowest low within the last 20 trading days
    df['High_Low_Ratio'] = df['high'].rolling(window=20).max() / df['low'].rolling(window=20).min()
    
    # Correlation coefficient between daily volume changes and price percentage changes for the last 30 days
    df['Volume_Change'] = df['volume'].pct_change()
    df['Price_Change'] = df['close'].pct_change()
    df['Vol_Price_Corr'] = df[['Volume_Change', 'Price_Change']].rolling(window=30).corr().dropna(how='all').unstack().iloc[::2].values
    
    # Count the number of days, over the last 50, where the open was higher than the close
    df['Bearish_Count'] = (df['open'] > df['close']).rolling(window=50).sum()
    
    # Count the number of days, over the last 50, where the open was lower than the close
    df['Bullish_Count'] = (df['open'] < df['close']).rolling(window=50).sum()
    
    # Days where the volume traded is at least 2 standard deviations above the mean volume calculated over the past year
    df['Volume_ZScore'] = (df['volume'] - df['volume'].rolling(window=252).mean()) / df['volume'].rolling(window=252).std()
    high_volume_days = df[df['Volume_ZScore'] > 2].index
    
    # Calculate the return from the next day's open to the close 10 days later for each high volume day
    def calculate_return_on_high_volume_day(date):
        if date + pd.Timedelta(days=10) in df.index:
            start_price = df.at[date + pd.Timedelta(days=1), 'open']
            end_price = df.at[date + pd.Timedelta(days=10), 'close']
            return (end_price - start_price) / start_price
        return np.nan
    
    high_volume_returns = [calculate_return_on_high_volume_day(date) for date in high_volume_days]
    df['High_Volume_Returns'] = df.index.map(dict(zip(high_volume_days, high_volume_returns)))
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Close_SMA_Diff'] +
        df['High_Low_Ratio'] +
        df['Vol_Price_Corr'] +
        df['Bearish_Count'] -
        df['Bullish_Count'] +
        df['High_Volume_Returns'].fillna(0)
    )
    
    return df['Alpha_Factor']
```
