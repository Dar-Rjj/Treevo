```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Daily Momentum
    df['adjusted_momentum'] = df['close'].diff(10)
    
    # Adjust by High-Low Spread and smooth with EMA
    df['high_low_spread'] = df['high'] - df['low']
    ema_smooth_factor = 0.9
    df['smoothed_high_low_spread'] = df['high_low_spread'].ewm(alpha=ema_smooth_factor).mean()
    
    # Incorporate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Adjusted Momentum, High-Low Spread, and Intraday Return
    df['combined_price_factors'] = df['adjusted_momentum'] * df['smoothed_high_low_spread'] + df['intraday_return']
    
    # Calculate Volume Surprise
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] - df['volume_ma_10']
    
    # Adjust by Volume Change and smooth with SMA
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    sma_length = 5
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=sma_length).mean()
    
    # Combine Volume Impact Factors
    df['combined_volume_impact'] = df['volume_surprise'] * df['smoothed_volume_ratio']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    df['accumulated_upward_volume'] = df['upward_volume'].cumsum()
    df['accumulated_downward_volume'] = df['downward_volume'].cumsum()
    df['volume_influence_ratio'] = df['accumulated_upward_volume'] / df['accumulated_downward_volume']
    
    # Intermediate Alpha Factor Synthesis
    df['intermediate_alpha_factor'] = df['combined_price_factors'] * df['combined_volume_impact']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['intermediate_alpha_factor'] + 
                                df['intraday_return'] * 
                                df['volume_influence_ratio'] + 
                                df['volume_surprise'])
    
    return df['final_alpha_factor']
```
