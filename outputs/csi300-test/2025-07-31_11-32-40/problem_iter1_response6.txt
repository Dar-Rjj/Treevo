```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    # Accumulate Volume on Upward Days
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    
    # Accumulate Volume on Downward Days
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    
    # Divide Upward Volume by Downward Volume, add a small constant to avoid division by zero
    volume_influence_ratio = (upward_volume + 1e-6) / (downward_volume + 1e-6)
    
    # Multiply High-Low Difference with Volume Influence Ratio
    factor_values = high_low_diff * volume_influence_ratio
    
    return factor_values
```
