```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - midpoint
    
    # Calculate Previous Day Return
    previous_day_close = df['close'].shift(1)
    previous_day_return = (df['open'] - previous_day_close) / previous_day_close
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha = (intraday_range * close_to_midpoint_deviation) - previous_day_return
    
    # Calculate High-to-Low Return
    high_to_low_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Price Movement Indicator
    price_movement = df['close'] - df['open']
    price_movement_indicator = price_movement.apply(lambda x: 1.0 if x > 0 else -1.0)
    
    # Adjust Momentum by Volume and Amount
    weighted_high_to_low_return = high_to_low_return * df['volume'] * price_movement_indicator
    
    # Introduce Time Decay
    decay_rate = 0.95  # Example decay rate, can be adjusted
    decayed_momentum = weighted_high_to_low_return.rolling(window=2).apply(lambda x: x[0] * decay_rate ** (len(x) - 1))
    
    # Generate Final Alpha Factor
    final_alpha = intermediate_alpha + decayed_momentum
    
    return final_alpha
```
