```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Calculate Close-to-Open Change
    close_to_open_change = df['close'] - df['open']
    
    # Weight by Volume
    volume_weighted_change = close_to_open_change * df['volume']
    
    # Combine Intraday Momentum and Volume Weighted Close-to-Open Change
    combined_factor = intraday_momentum + volume_weighted_change
    
    # Calculate 5-day Moving Average of the Combined Factor
    moving_average_5d = combined_factor.rolling(window=5).mean()
    
    # Compare with Current Day Factor
    current_day_factor = combined_factor
    score = (current_day_factor > moving_average_5d).astype(int) - (current_day_factor < moving_average_5d).astype(int)
    
    return score
```
