```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using open, high, low, close, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the price momentum
    momentum = typical_price.pct_change(periods=5)
    
    # Calculate the volume weighted typical price
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the volatility of the typical price
    volatility = typical_price.rolling(window=20).std()
    
    # Calculate the money flow ratio
    positive_flow = df['amount'].where(typical_price > typical_price.shift(1), 0)
    negative_flow = df['amount'].where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    
    # Combine the factors into a single alpha factor
    alpha_factor = momentum + (vwap - typical_price) / volatility + (money_flow_ratio - 1)
    
    return alpha_factor
```
