import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Short-Term Momentum
    short_term_momentum = 1 + (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Generate Preliminary Alpha Factor
    day_to_day_high_price_change = (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    preliminary_alpha_factor = (intraday_reversal * short_term_momentum) + day_to_day_high_price_change
    
    # Calculate Daily Volume Surprise
    volume_5_day_ma = df['volume'].rolling(window=5).mean()
    daily_volume_surprise = df['volume'] - volume_5_day_ma
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(5)
    combined_price_vol_momentum = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = (df['volume'] * (df['close'] > df['open']).astype(int)).rolling(window=5).sum()
    downward_volume = (df['volume'] * (df['close'] < df['open']).astype(int)).rolling(window=5).sum()
    volume_influence_ratio = upward_volume / downward_volume
    
    # Incorporate Trading Amount Trends
    amount_5_day_ma = df['amount'].rolling(window=5).mean()
    trading_amount_trend = df['amount'] - amount_5_day_ma
    
    # Integrate Trading Amount and Price Movements
    integrated_trading_amount_price = (df['close'] - df['close'].shift(5)) * trading_amount_trend
    
    # Final Alpha Factor
    final_alpha_factor = (
        preliminary_alpha_factor * 
        combined_price_vol_momentum * 
        volume_influence_ratio * 
        integrated_trading_amount_price
    )
    
    return final_alpha_factor
