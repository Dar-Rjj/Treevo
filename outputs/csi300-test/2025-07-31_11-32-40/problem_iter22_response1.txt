```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute rolling window mean of log returns
    df['log_return_20d_mean'] = df['log_return'].rolling(window=20).mean()
    df['log_return_50d_mean'] = df['log_return'].rolling(window=50).mean()
    
    # Calculate daily range (High - Low)
    df['daily_range'] = df['high'] - df['low']
    
    # Compute moving average of daily ranges
    df['range_30d_mean'] = df['daily_range'].rolling(window=30).mean()
    df['range_100d_mean'] = df['daily_range'].rolling(window=100).mean()
    
    # Calculate percentage change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate Pearson correlation between volume and close prices
    df['vol_close_corr_60d'] = df['close'].rolling(window=60).corr(df['volume'])
    
    # Create a composite factor using log returns, range, and volume
    df['composite_factor'] = (
        0.4 * df['log_return_50d_mean'] +
        0.3 * (df['daily_range'] / df['range_100d_mean']) +
        0.3 * df['volume_change']
    )
    
    # Normalize the composite factor
    df['composite_factor'] = (df['composite_factor'] - df['composite_factor'].rolling(window=90).mean()) / df['composite_factor'].rolling(window=90).std()
    
    return df['composite_factor']
```
