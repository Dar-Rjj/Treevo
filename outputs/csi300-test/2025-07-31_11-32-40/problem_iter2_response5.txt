```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    df['IntradayReversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    df['VolumeChange'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['IntradayReversalAdjusted'] = df['IntradayReversal'] * (1 + df['VolumeChange'])
    
    # Calculate Intraday Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['IntradayMidpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    df['CloseToMidDeviation'] = df['close'] - df['IntradayMidpoint']
    
    # Consider Day-to-Day Open Price Change
    df['DayToDayOpenChange'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Metrics
    df['AlphaFactor'] = (
        df['IntradayReversalAdjusted'] * df['IntradayRange'] * 
        df['CloseToMidDeviation'] + df['DayToDayOpenChange'] - 
        df['close'].pct_change()
    )
    
    return df['AlphaFactor']
```
