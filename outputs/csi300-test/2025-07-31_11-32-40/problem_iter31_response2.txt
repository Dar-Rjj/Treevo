```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = upward_volume / (downward_volume + 1e-6)  # Add a small constant to avoid division by zero
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_volume_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor = high_low_volume_influence * intraday_return
    
    # Calculate Daily Price Reversal
    daily_price_reversal = df['close'].diff().shift(1)
    daily_price_reversal = daily_price_reversal.where(df['close'].shift(1) > df['close'].shift(2), -daily_price_reversal)
    
    # Integrate Short-Term Price Volatility
    short_term_volatility = df['close'].rolling(window=5).std()
    adjusted_factor = combined_factor * daily_price_reversal / (short_term_volatility + 1e-6)  # Add a small constant to avoid division by zero
    
    # Incorporate Trend Strength
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    trend_adjustment = (sma_10 > sma_30).astype(int) - (sma_10 < sma_30).astype(int)
    final_alpha_factor = adjusted_factor * trend_adjustment
    
    return final_alpha_factor
```
