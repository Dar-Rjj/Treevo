```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    close_midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Calculate Previous Day Return
    previous_day_return = df['open'] / df['close'].shift(1) - 1
    
    # Generate Intermediate Alpha Factor 1
    intermediate_alpha_factor_1 = (intraday_range * close_midpoint_deviation - previous_day_return)
    
    # Calculate Daily Volume Surprise
    volume_moving_average = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_moving_average
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(10)
    combined_price_momentum_vol_surprise = price_momentum * daily_volume_surprise
    
    # Calculate High-Low Price Difference
    high_low_difference = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df['close'] > df['open']) * df['volume']
    downward_volume = (df['close'] < df['open']) * df['volume']
    volume_influence_ratio = upward_volume.rolling(window=10).sum() / downward_volume.rolling(window=10).sum()
    
    # Adjust Momentum by Volume and Amount
    weighted_high_low_return = high_low_difference * df['volume'] * volume_influence_ratio
    volume_adjusted_momentum = weighted_high_low_return.rolling(window=10).sum()
    
    # Introduce Time Decay
    decay_rate = 0.95
    finalized_volume_adjusted_momentum = volume_adjusted_momentum * decay_rate ** (pd.Series(range(len(df)), index=df.index))
    
    # Intermediate Alpha Factor Synthesis 1
    intermediate_alpha_synthesis_1 = combined_price_momentum_vol_surprise * high_low_difference
    
    # Compute Price Momentum
    smoothed_price_momentum = (df['close'] - df['close'].shift(1)).ewm(span=10).mean()
    
    # Adjust by Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1)).rolling(window=10).mean()
    combined_momentum_volume_ratio = smoothed_price_momentum * volume_ratio
    
    # Compute Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Spread and Close-Open Return
    intraday_momentum = high_low_difference * intraday_close_open_return
    
    # Consider Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Adjust for Open Interest
    adjusted_intraday_reversal_open_interest = adjusted_intraday_reversal * (1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1))
    
    # Intermediate Alpha Factor Synthesis 2
    intermediate_alpha_synthesis_2 = combined_momentum_volume_ratio * intraday_momentum * adjusted_intraday_reversal_open_interest
    
    # Final Alpha Factor
    final_alpha_factor = (intermediate_alpha_synthesis_1 * intermediate_alpha_factor_1 * volume_influence_ratio) + daily_volume_surprise
    
    return final_alpha_factor
```
