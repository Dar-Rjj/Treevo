```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    df['adjusted_momentum'] = df['close'] - df['low'].shift(1)
    ema_smoothing_factor = 12
    df['smoothed_adjusted_momentum'] = df['adjusted_momentum'].ewm(span=ema_smoothing_factor, adjust=False).mean()
    
    # Calculate Simple Price Momentum
    lookback_period = 10
    df['simple_momentum'] = df['close'] - df['close'].shift(lookback_period)
    df['smoothed_simple_momentum'] = df['simple_momentum'].ewm(span=ema_smoothing_factor, adjust=False).mean()
    
    # Combine Adjusted Momentum and Simple Price Momentum
    df['combined_momentum'] = df['smoothed_adjusted_momentum'] + df['smoothed_simple_momentum']
    
    # Adjust by Volume Change
    df['volume_ma_5'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ma_5']
    sma_length = 5
    df['smoothed_volume_ratio'] = df['volume_ratio'].rolling(window=sma_length).mean()
    df['momentum_with_volume'] = df['combined_momentum'] * df['smoothed_volume_ratio']
    
    # Compute Volume Influence Ratio
    df['upward_volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1)
    df['downward_volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1)
    upward_cumulative_volume = df['upward_volume'].cumsum()
    downward_cumulative_volume = df['downward_volume'].cumsum()
    df['volume_influence_ratio'] = (upward_cumulative_volume + 1) / (downward_cumulative_volume + 1)
    
    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    df['integrated_factor'] = df['momentum_with_volume'] * df['volume_influence_ratio'].pow(0.5)
    
    # Calculate High-Low Price Difference
    df['high_low_difference'] = df['high'] - df['low']
    
    # Multiply High-Low Difference with Volume Influence Ratio
    df['final_alpha'] = df['high_low_difference'] * df['volume_influence_ratio']
    
    return df['final_alpha']
```
