```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = (df[df['close'] > df['open']]['volume']).sum()
    downward_volume = (df[df['close'] < df['open']]['volume']).sum()
    volume_influence_ratio = upward_volume / downward_volume if downward_volume != 0 else 0
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    df['vol_10_day_ma'] = df['volume'].rolling(window=10).mean()
    daily_vol_change = df['volume'] - df['vol_10_day_ma']
    volume_momentum_strength = (daily_vol_change > 0).astype(int) * 2 - 1
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_factor = high_low_vol_influence * intraday_return * volume_momentum_strength
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    close_t_minus_10 = df['close'].shift(10)
    momentum_adjustment = 1 + (df['close'] - close_t_minus_10) / close_t_minus_10
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment
    
    # Adjust for Open Interest
    volume_t_minus_1 = df['volume'].shift(1)
    adjusted_intraday_reversal = adjusted_intraday_reversal * (1 + (df['volume'] - volume_t_minus_1) / volume_t_minus_1)
    
    # Generate Preliminary Alpha Factor
    close_to_midpoint_deviation = (df['close'] - (df['high'] + df['low']) / 2) / (df['high'] - df['low'])
    day_to_day_open_price_change = df['open'].diff()
    preliminary_alpha_factor = adjusted_intraday_reversal * close_to_midpoint_deviation + day_to_day_open_price_change
    
    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - df['vol_10_day_ma']
    
    # Calculate Daily Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Price Momentum and Volume Surprise
    combined_price_momentum_vol_surprise = price_momentum * daily_volume_surprise
    
    # Integrate Intraday Range and Volume Dynamics
    intraday_range = df['high'] - df['low']
    intraday_range_vol_dynamics = intraday_range * df['volume']
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_price_momentum_vol_surprise * intraday_range_vol_dynamics
    
    # Final Alpha Factor
    final_alpha_factor = preliminary_alpha_factor * intermediate_alpha_factor * volume_influence_ratio
    
    return final_alpha_factor
```
