```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Calculate Volume Surprise
    df['volume_moving_avg'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] - df['volume_moving_avg']
    
    # Combine Price Momentum and Volume Surprise
    df['alpha_factor'] = df['price_momentum'] * df['volume_surprise']
    
    return df['alpha_factor']
```
