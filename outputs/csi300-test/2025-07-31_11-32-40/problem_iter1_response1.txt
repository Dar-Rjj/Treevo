```python
def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Combine High-Low Spread and Close-Open Return
    intraday_momentum = intraday_high_low_spread * intraday_close_open_return
    
    # Final Alpha Factor
    final_alpha_factor = intraday_momentum + volume_surge
    
    return final_alpha_factor
```
