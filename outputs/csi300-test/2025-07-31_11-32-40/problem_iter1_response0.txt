import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    df['close_to_midpoint_deviation'] = df['close'] - df['intraday_midpoint']
    
    # Calculate Previous Day Return
    df['previous_day_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Combine Metrics
    df['intraday_range_deviation'] = df['intraday_range'] * df['close_to_midpoint_deviation']
    
    # Generate Alpha Factor
    df['alpha_factor'] = df['intraday_range_deviation'] - df['previous_day_return']
    
    # Return the alpha factor
    return df['alpha_factor']
