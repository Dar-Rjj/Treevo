```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Difference
    high_low_diff = df['high'] - df['low']
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].rolling(window=10).sum()
    downward_volume = df[df['close'] < df['open']]['volume'].rolling(window=10).sum()
    volume_influence_ratio = (upward_volume / downward_volume).fillna(1)
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_influence = high_low_diff * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    daily_volume_change = df['volume'] - avg_volume_10d
    volume_momentum_strength = daily_volume_change.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_high_low_vol_influence = high_low_influence * intraday_return * volume_momentum_strength
    
    # Calculate Daily Price Reversal
    price_reversal = (df['close'] - df['close'].shift(1)).apply(
        lambda x: -x if (df['close'].shift(1) - df['close'].shift(2)) > 0 else x
    )
    
    # Calculate Volume Surprise
    volume_surprise = df['volume'] - avg_volume_10d
    combined_price_reversal_vol_surprise = price_reversal * volume_surprise
    
    # Integrate Short-Term Price Volatility
    close_std_5d = df['close'].rolling(window=5).std()
    final_alpha_factor = (combined_high_low_vol_influence * combined_price_reversal_vol_surprise) / close_std_5d
    
    # Incorporate Trend Strength
    sma_10d = df['close'].rolling(window=10).mean()
    sma_30d = df['close'].rolling(window=30).mean()
    trend_adjustment = sma_10d.apply(lambda x: 1.2 if x > sma_30d else 0.8)
    enhanced_final_alpha_factor = final_alpha_factor * trend_adjustment
    
    return enhanced_final_alpha_factor
```
