import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Midpoint Deviation
    midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Calculate Adjusted Intraday Reversal with Momentum Adjustment
    momentum_adjustment = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    adjusted_intraday_reversal = (2 * (df['high'] - df['low']) / (df['close'] + df['open'])) * (1 + momentum_adjustment)
    
    # Calculate Previous Day Return
    previous_day_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Intermediate Alpha Factor
    intermediate_alpha_factor = (intraday_range * midpoint_deviation) - previous_day_return
    
    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(10)
    combined_price_vol = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    volume_up = df['volume'][df['close'] > df['open']].sum()
    volume_down = df['volume'][df['close'] < df['open']].sum()
    volume_influence_ratio = volume_up / volume_down if volume_down != 0 else 0
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_vol_influence = (df['high'] - df['low']) * volume_influence_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Identify Volume Momentum Strength
    volume_momentum_strength = daily_volume_surprise.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine High-Low Difference, Volume Influence, and Intraday Return
    combined_high_low_vol_influence_intraday = (df['high'] - df['low']) * volume_influence_ratio * intraday_return * volume_momentum_strength
    
    # Calculate Daily Price Reversal
    daily_price_reversal = df['close'] - df['close'].shift(1)
    daily_price_reversal = daily_price_reversal * (-1 if (df['close'].shift(1) > df['close'].shift(2)).any() else 1)
    
    # Combine Price Reversal and Volume Surprise
    combined_price_reversal_vol_surprise = daily_price_reversal * daily_volume_surprise
    
    # Integrate Short-Term Price Volatility
    short_term_volatility = df['close'].rolling(window=5).std()
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (intermediate_alpha_factor * combined_high_low_vol_influence_intraday * combined_price_reversal_vol_surprise) / short_term_volatility
    
    return final_alpha_factor
