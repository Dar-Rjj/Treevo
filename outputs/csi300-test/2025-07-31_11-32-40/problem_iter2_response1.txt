import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine High-Low Spread and Close-Open Return
    intraday_momentum = intraday_high_low_spread * intraday_close_open_return
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Final Intraday Component
    intraday_momentum_and_volume_surge = intraday_momentum + volume_surge
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(20)
    
    # Combine Price Momentum and Volume Change
    historical_momentum_and_volume_dynamics = price_momentum * volume_change
    
    # Final Alpha Factor
    alpha_factor = intraday_momentum_and_volume_surge + historical_momentum_and_volume_dynamics
    
    return alpha_factor
