```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Price Momentum
    adjusted_momentum = df['close'] - df['low'].shift(1)
    ema_smoothed_adjusted_momentum = adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Calculate Simple Price Momentum
    simple_momentum = df['close'] - df['close'].shift(10)
    
    # Combine Adjusted and Simple Momentum
    combined_momentum = ema_smoothed_adjusted_momentum + simple_momentum
    
    # Calculate Volume Surprise
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_surprise = df['volume'] - volume_ma_10
    
    # Adjust by Volume Change
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    sma_smoothed_volume_ratio = volume_ratio.rolling(window=5).mean()
    adjusted_combined_momentum = combined_momentum * sma_smoothed_volume_ratio
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].sum()
    downward_volume = df[df['close'] < df['open']]['volume'].sum()
    volume_influence_ratio = (upward_volume / downward_volume) if downward_volume != 0 else 0
    
    # Integrate Combined Momentum, Volume Ratio, and Directional Volume Impact
    integrated_momentum = adjusted_combined_momentum * np.sqrt(volume_influence_ratio)
    
    # Calculate High-Low Price Difference
    high_low_difference = df['high'] - df['low']
    
    # Multiply High-Low Difference with Volume Influence Ratio
    high_low_diff_with_volume_influence = high_low_difference * volume_influence_ratio
    
    # Calculate Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Open Interest
    open_interest_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * open_interest_adjustment
    
    # Combine Intraday Metrics
    combined_intraday_metrics = high_low_diff_with_volume_influence + adjusted_intraday_reversal
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = integrated_momentum * high_low_difference
    
    # Final Alpha Factor
    final_alpha_factor = intermediate_alpha_factor * volume_influence_ratio * combined_intraday_metrics + integrated_momentum
    
    return final_alpha_factor
```
