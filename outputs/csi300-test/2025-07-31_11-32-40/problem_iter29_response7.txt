import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum and Volume Change
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    volume_adjustment = 1 + (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_intraday_reversal = intraday_reversal * momentum_adjustment * volume_adjustment
    
    # Calculate Close to Midpoint Deviation
    intraday_midpoint = (df['high'] + df['low']) / 2
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Generate Alpha Factor Base
    alpha_factor_base = (adjusted_intraday_reversal * close_to_midpoint_deviation) + (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Generate Day-to-Day Open Price Change
    open_price_change = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday Reversal, Close to Midpoint Deviation, and Open Price Change
    combined_intraday_factors = adjusted_intraday_reversal * close_to_midpoint_deviation + open_price_change
    
    # Calculate Daily Volume Surprise
    daily_volume_surprise = df['volume'] - df['volume'].rolling(window=10).mean()
    
    # Compute Directional Adjusted Volume
    upward_days = (df['close'] > df['open']).astype(int)
    downward_days = (df['close'] < df['open']).astype(int)
    adjusted_volume_upward = (df['close'] - df['open']) * df['volume'] * upward_days
    adjusted_volume_downward = (df['open'] - df['close']) * df['volume'] * downward_days
    net_directional_adjusted_volume = adjusted_volume_upward.sum() - adjusted_volume_downward.sum()
    
    # Calculate Cumulative Directional Price Impact
    high_low_difference = df['high'] - df['low']
    cumulative_directional_price_impact = high_low_difference * net_directional_adjusted_volume
    
    # Calculate Smoothed Price Momentum
    smoothed_momentum = (df['close'] - df['close'].shift(1)).ewm(span=10).mean()
    
    # Adjust by Smoothed Volume Change
    volume_ratio = (df['volume'] / df['volume'].shift(1)).rolling(window=10).mean()
    combined_momentum_volume = smoothed_momentum * volume_ratio
    
    # Integrate Intraday Range and Volume Dynamics
    intraday_range = df['high'] - df['low']
    integrated_intraday_factor = intraday_range * df['volume']
    
    # Intermediate Alpha Factor Synthesis
    intermediate_alpha_factor = combined_intraday_factors * integrated_intraday_factor * volume_ratio + adjusted_intraday_reversal
    
    # Final Alpha Factor Components
    final_alpha_factor_components = (
        intermediate_alpha_factor * integrated_intraday_factor * net_directional_adjusted_volume
        + alpha_factor_base + (df['volume'] - df['volume'].shift(1)) * combined_momentum_volume
    )
    
    # Final Alpha Factor
    final_alpha_factor = final_alpha_factor_components
    
    return final_alpha_factor
