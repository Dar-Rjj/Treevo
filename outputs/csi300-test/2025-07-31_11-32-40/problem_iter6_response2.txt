import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - intraday_midpoint
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Close-Open Return
    intraday_close_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Return
    previous_day_return = df['close'].shift(1) - df['open']
    
    # Calculate Volume Surge
    volume_surge = df['volume'] - df['volume'].shift(1)
    
    # Weigh Metrics
    weighted_intraday_range = intraday_range * 0.5
    weighted_intraday_high_low_spread = intraday_high_low_spread * 0.3
    weighted_intraday_close_open_return = intraday_close_open_return * 0.2
    
    # Combine Weighted Metrics
    combined_weighted_metrics = (
        weighted_intraday_range +
        weighted_intraday_high_low_spread +
        weighted_intraday_close_open_return
    )
    
    # Adjust for Previous Day Return
    adjusted_combined_weighted_metrics = combined_weighted_metrics * (1 - previous_day_return)
    
    # Generate Alpha Factor
    alpha_factor = adjusted_combined_weighted_metrics + volume_surge
    
    # Store as Intraday Reversal Momentum and Volume Surge Indicator
    df['intraday_reversal_momentum_and_volume_surge'] = alpha_factor
    
    return df['intraday_reversal_momentum_and_volume_surge']
