```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    momentum_lookback = 20
    volume_lookback = 10

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].pct_change(momentum_lookback)
    df['Smoothed_Price_Momentum'] = df['Price_Momentum'].ewm(span=momentum_lookback, adjust=False).mean()

    # Adjust by Volume Change
    df['Volume_Ratio'] = df['volume'].rolling(window=volume_lookback).mean() / df['volume'].shift(volume_lookback)
    df['Smoothed_Volume_Ratio'] = df['Volume_Ratio'].rolling(window=volume_lookback).mean()
    df['Combined_Momentum_Volume'] = df['Smoothed_Price_Momentum'] * df['Smoothed_Volume_Ratio']

    # Compute High-Low Price Difference
    df['High_Low_Diff'] = df['high'] - df['low']

    # Compute Volume Influence Ratio
    df['Upward_Volume'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else 0, axis=1).rolling(window=volume_lookback).sum()
    df['Downward_Volume'] = df.apply(lambda row: row['volume'] if row['close'] < row['open'] else 0, axis=1).rolling(window=volume_lookback).sum()
    df['Volume_Influence_Ratio'] = df['Upward_Volume'] / (df['Downward_Volume'] + 1e-6)

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Integrate Core Components into Intermediate Alpha Factor
    df['Intermediate_Alpha_Factor'] = df['Combined_Momentum_Volume'] * (df['High_Low_Diff'] * df['Volume_Influence_Ratio'] * df['Intraday_Return'])

    # Combine High-Low Spread and Close-Open Return
    df['Intraday_Momentum'] = df['High_Low_Diff'] * df['Intraday_Return']

    # Calculate Intraday Reversal
    df['Intraday_Reversal'] = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])

    # Adjust for Open Interest
    df['Adjusted_Intraday_Reversal'] = df['Intraday_Reversal'] * (1 + (df['volume'] - df['volume'].shift(1)) / (df['volume'].shift(1) + 1e-6))

    # Consider Day-to-Day Open Price Change
    df['Day_to_Day_Open_Price_Change'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Intermediate_Alpha_Factor'] + 
                                df['Intraday_Momentum'] * df['Volume_Influence_Ratio'] + 
                                df['Adjusted_Intraday_Reversal'] + 
                                df['Day_to_Day_Open_Price_Change'])

    # Generate Preliminary Alpha Factor
    df['Preliminary_Alpha_Factor'] = (df['Adjusted_Intraday_Reversal'] * (df['close'] - (df['high'] + df['low']) / 2) + 
                                      df['Day_to_Day_Open_Price_Change'])

    return df['Final_Alpha_Factor']
```
