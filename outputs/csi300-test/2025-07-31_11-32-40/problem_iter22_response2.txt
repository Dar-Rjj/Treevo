import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Reversal
    intraday_reversal = 2 * (df['high'] - df['low']) / (df['close'] + df['open'])
    
    # Adjust for Momentum
    momentum_adjustment = 1 + (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Calculate Close to Midpoint Deviation
    close_to_midpoint_deviation = df['close'] - (df['high'] + df['low']) / 2
    
    # Generate Preliminary Alpha Factor
    preliminary_alpha_factor = (intraday_reversal * momentum_adjustment) + \
                               ((df['open'] - df['close'].shift(1)) / df['close'].shift(1))
    
    # Calculate Daily Volume Surprise
    volume_moving_average = df['volume'].rolling(window=10).mean()
    daily_volume_surprise = df['volume'] - volume_moving_average
    
    # Combine Price Momentum and Volume Surprise
    price_momentum = df['close'] - df['close'].shift(10)
    combined_price_momentum_volume_surprise = price_momentum * daily_volume_surprise
    
    # Compute Volume Influence Ratio
    upward_volume = df[df['close'] > df['open']]['volume'].rolling(window=10).sum()
    downward_volume = df[df['close'] < df['open']]['volume'].rolling(window=10).sum()
    volume_influence_ratio = upward_volume / downward_volume.fillna(1)
    
    # Final Alpha Factor
    final_alpha_factor = (preliminary_alpha_factor * 
                          combined_price_momentum_volume_surprise * 
                          volume_influence_ratio)
    
    return final_alpha_factor
