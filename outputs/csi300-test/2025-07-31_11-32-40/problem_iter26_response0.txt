    # Compute rolling window mean of log returns for 20-day and 50-day windows
    df['log_returns_mean_20'] = df['log_returns'].rolling(window=20).mean()
    df['log_returns_mean_50'] = df['log_returns'].rolling(window=50).mean()
    
    # Calculate daily range (High - Low)
    df['daily_range'] = df['high'] - df['low']
    
    # Compute moving average of daily ranges for 30-day and 100-day windows
    df['daily_range_ma_30'] = df['daily_range'].rolling(window=30).mean()
    df['daily_range_ma_100'] = df['daily_range'].rolling(window=100).mean()
    
    # Calculate percentage change in daily volume
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # Calculate Pearson correlation between volume and close prices using 60-day rolling window
    df['volume_close_corr'] = df['close'].rolling(window=60).corr(df['volume'])
    
    # Calculate high-low price difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Accumulate adjusted volume on upward days
    df['upward_volume'] = (df['close'] > df['open']) * (df['close'] - df['open']) * df['volume']
    df['upward_volume'] = df['upward_volume'].rolling(window=60).sum()
    
    # Accumulate adjusted volume on downward days
    df['downward_volume'] = (df['close'] < df['open']) * (df['open'] - df['close']) * df['volume']
    df['downward_volume'] = df['downward_volume'].rolling(window=60).sum()
    
    # Calculate net directional adjusted volume
    df['net_directional_volume'] = df['upward_volume'] - df['downward_volume']
    
    # Multiply high-low difference with net directional adjusted volume
    df['directional_price_impact'] = df['high_low_diff'] * df['net_directional_volume']
    
    # Create a composite factor using log returns, range, volume, and directional adjusted volume
    # Assign weights based on historical performance (for simplicity, equal weights are used here)
    df['composite_factor'] = (df['log_returns_mean_20'] + 
                              df['log_returns_mean_50'] + 
                              df['daily_range_ma_30'] + 
                              df['daily_range_ma_100'] + 
                              df['volume_close_corr'] + 
                              df['directional_price_impact']) / 6
