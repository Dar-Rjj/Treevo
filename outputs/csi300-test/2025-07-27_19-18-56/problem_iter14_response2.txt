import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Composite Momentum Indicator
    n = 5
    close_to_close_return = (df['close'] / df['close'].shift(n)) - 1
    composite_momentum = close_to_close_return.rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_indicator = (volume_change > 0).astype(int)
    momentum_with_volume_confirmation = composite_momentum * volume_indicator
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_term_ma_period = 5
    long_term_ma_period = 20
    daily_avg_high_low = (df['high'] + df['low']) / 2
    short_term_ma_high_low = daily_avg_high_low.rolling(window=short_term_ma_period).mean()
    long_term_ma_high_low = daily_avg_high_low.rolling(window=long_term_ma_period).mean()
    convergence_divergence = short_term_ma_high_low - long_term_ma_high_low
    
    # Filter by Volume Change
    integrated_momentum = momentum_with_volume_confirmation * (volume_change > 0).astype(int)
    
    # Combine Momentum and Volume Indicators
    final_alpha_factor = integrated_momentum + convergence_divergence
    
    return final_alpha_factor
