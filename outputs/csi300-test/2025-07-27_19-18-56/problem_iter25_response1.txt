```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return and sum over n periods (e.g., 10 days)
    n = 10
    close_to_close_return = df['close'].diff(periods=n)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_term_ma = (df['high'] + df['low']) / 2.0
    long_term_ma = (df['high'] + df['low']) / 2.0
    short_term_ma = short_term_ma.rolling(window=5).mean()
    long_term_ma = long_term_ma.rolling(window=20).mean()
    convergence_divergence = short_term_ma - long_term_ma
    
    # Volume-Adjusted High-Low Range
    high_low_range = (df['high'] - df['low']).rolling(window=10).apply(
        lambda x: (x * df['volume']).sum() / df['volume'].sum(), raw=False)
    
    # Smoothed High-Low Spread
    ema_period = 10
    high_low_spread = df['high'] - df['low']
    smoothed_high_low_spread = high_low_spread - high_low_spread.ewm(span=ema_period, adjust=False).mean()
    
    # Trend Confirmation
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    trend_confirmation = (sma_5 > sma_20).astype(int)
    
    # Filter by Volume Change
    volume_change = df['volume'].diff()
    volume_filter = (volume_change > 0).astype(int)
    
    # Combine Indicators
    composite_momentum = close_to_close_return * volume_filter
    enhanced_convergence_divergence = convergence_divergence * volume_filter
    integrated_momentum = composite_momentum + enhanced_convergence_divergence + high_low_range + smoothed_high_low_spread
    final_alpha_factor = integrated_momentum * trend_confirmation
    
    return final_alpha_factor
```
