```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, sma_period=5):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    df['volume_indicator'] = (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Multiply Momentum by Volume Indicator
    df['confirmed_price_momentum'] = df['price_momentum'] * df['volume_indicator']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    df['sma_high_low_spread'] = df['high_low_spread'].rolling(window=sma_period).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['sma_high_low_spread']
    
    # Calculate Volume Relative Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Filter by Volume Change
    df['filtered_high_low_momentum'] = np.where(df['volume_change'] > 0, 
                                                df['smoothed_high_low_spread'], 
                                                0)
    
    # Integrate Both Momentum Indicators
    df['integrated_momentum'] = df['confirmed_price_momentum'] + df['filtered_high_low_momentum']
    
    # Calculate Final Alpha Factor
    alpha_factor = df['integrated_momentum'].dropna()
    
    return alpha_factor
```
