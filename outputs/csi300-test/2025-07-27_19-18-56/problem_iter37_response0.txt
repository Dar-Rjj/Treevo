```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, ema_period=5, short_ma_period=5, long_ma_period=20):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].diff()
    
    # Sum over n periods to get momentum
    df['momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['volume_adjusted_momentum'] = df['momentum'] * df['volume_increase']
    
    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) * df['volume']
    df['volume_weighted_high_low'] = df['high_low_range'].rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    df['close_open_return'] = df['close'] - df['open']
    sentiment_factor = np.where(df['close_open_return'] > 0, 1, -1)
    df['adjusted_high_low_range'] = df['volume_weighted_high_low'] * sentiment_factor
    
    # Smooth High-Low Spread
    df['ema_high_low'] = df['high_low_range'].ewm(span=ema_period, adjust=False).mean()
    df['smoothed_high_low'] = df['high_low_range'] - df['ema_high_low']
    
    # Integrate with Convergence-Divergence of High-Low Prices
    df['short_term_ma'] = (df['high'] + df['low']) / 2.0
    df['long_term_ma'] = (df['high'] + df['low']) / 2.0
    df['short_term_ma'] = df['short_term_ma'].rolling(window=short_ma_period).mean()
    df['long_term_ma'] = df['long_term_ma'].rolling(window=long_ma_period).mean()
    df['con_divergence'] = df['short_term_ma'] - df['long_term_ma']
    
    # Filter by Volume Change
    df['integrated_factors'] = df['volume_adjusted_momentum'] + df['adjusted_high_low_range'] + df['smoothed_high_low'] + df['con_divergence']
    df['integrated_factors'] = df['integrated_factors'].where(df['volume_increase'] == 1, 0)
    
    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=short_ma_period).mean()
    df['sma_20'] = df['close'].rolling(window=long_ma_period).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['integrated_factors'] * df['trend_confirmation']
    
    return df['alpha_factor']
```
