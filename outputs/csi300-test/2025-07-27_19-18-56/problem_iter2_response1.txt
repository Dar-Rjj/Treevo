```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    n = 10  # number of periods for momentum
    df['close_return'] = df['close'].pct_change(n)
    
    # Calculate Composite Momentum Indicator (sum over n periods)
    df['momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'].pct_change(n)
    df['volume_indicator'] = (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Multiply Momentum by Volume Indicator
    df['composite_momentum'] = df['momentum'] * df['volume_indicator']
    
    # Calculate Daily Average of High and Low
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate Short Term Moving Average of High and Low
    short_window = 5
    df['short_ma'] = df['avg_high_low'].rolling(window=short_window).mean()
    
    # Calculate Long Term Moving Average of High and Low
    long_window = 20
    df['long_ma'] = df['avg_high_low'].rolling(window=long_window).mean()
    
    # Subtract Long Term MA from Short Term MA
    df['convergence_divergence'] = df['short_ma'] - df['long_ma']
    
    # Combine Indicators
    df['alpha_factor'] = df['composite_momentum'] + df['convergence_divergence']
    
    return df['alpha_factor']
```
