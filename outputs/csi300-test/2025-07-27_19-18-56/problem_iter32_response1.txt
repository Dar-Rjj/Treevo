```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return and sum over n periods (n=10)
    df['close_return'] = df['close'].pct_change()
    df['momentum'] = df['close_return'].rolling(window=10).sum()

    # Confirm with Trading Volume Increase
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['momentum'] *= df['volume_increase']

    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) * df['volume']
    df['total_high_low_volume'] = df['high_low_range'].rolling(window=10).sum()
    df['total_volume'] = df['volume'].rolling(window=10).sum()
    df['volume_weighted_high_low'] = df['total_high_low_volume'] / df['total_volume']

    # Adjust for Market Sentiment
    df['close_open_return'] = (df['close'] - df['open']).pct_change()
    sentiment_modifier = np.where(df['close_open_return'] > 0, 1.2, 0.8)
    df['volume_weighted_high_low'] *= sentiment_modifier

    # Smoothing High-Low Spread using EMA
    df['high_low_ema'] = df['high_low_range'].ewm(span=10, adjust=False).mean()
    df['smoothed_high_low_spread'] = df['high_low_range'] - df['high_low_ema']

    # Integrate with Convergence-Divergence of High-Low Prices
    df['short_ma_hl'] = (df['high'] + df['low']) / 2.0
    df['long_ma_hl'] = (df['high'] + df['low']) / 2.0
    df['short_ma_hl'] = df['short_ma_hl'].rolling(window=5).mean()
    df['long_ma_hl'] = df['long_ma_hl'].rolling(window=20).mean()
    df['converge_diverge'] = df['short_ma_hl'] - df['long_ma_hl']

    # Filter by Volume Change
    df['converge_diverge'] = df['converge_diverge'] * df['volume_increase']

    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)

    # Final Alpha Factor Construction
    df['combined_factors'] = (df['volume_weighted_high_low'] + 
                              df['smoothed_high_low_spread'] + 
                              df['momentum'] +
                              df['converge_diverge'])
    df['final_alpha_factor'] = df['combined_factors'] * df['trend_confirmation'] * df['volume_increase']

    return df['final_alpha_factor']
```
