```python
import pandas as pd

def heuristics_v2(df, n=10, sma_period=5, short_ma_period=5, long_ma_period=20):
    # Calculate High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    df['SMA_High-Low_Spread'] = df['High-Low_Spread'].rolling(window=sma_period).mean()
    df['SMA_Adjusted_High-Low_Spread'] = df['High-Low_Spread'] - df['SMA_High-Low_Spread']
    
    # Calculate Price Momentum
    df['Close_to_Close_Return'] = df['close'].pct_change(periods=n)
    df['Price_Momentum'] = df['Close_to_Close_Return'].rolling(window=n).sum()
    
    # Calculate Volume Relative Change
    df['Volume_Relative_Change'] = (df['volume'] / df['volume'].shift(1)) - 1
    
    # Incorporate Convergence Divergence of High-Low Prices
    df['Short_MA_High'] = df['high'].rolling(window=short_ma_period).mean()
    df['Short_MA_Low'] = df['low'].rolling(window=short_ma_period).mean()
    df['Long_MA_High'] = df['high'].rolling(window=long_ma_period).mean()
    df['Long_MA_Low'] = df['low'].rolling(window=long_ma_period).mean()
    df['Convergence_Divergence'] = (df['Short_MA_High'] + df['Short_MA_Low']) - (df['Long_MA_High'] + df['Long_MA_Low'])
    
    # Final Alpha Factor
    volume_filter = df['volume'] > df['volume'].shift(1)
    df['Volume_Filter_Indicator'] = volume_filter.astype(int)
    df['Composite_Price-Volume_Momentum'] = df['Price_Momentum'] * df['Volume_Filter_Indicator']
    
    df['Alpha_Factor'] = df['Composite_Price-Volume_Momentum'] + df['Convergence_Divergence'] + df['SMA_Adjusted_High-Low_Spread']
    
    return df['Alpha_Factor']
```
