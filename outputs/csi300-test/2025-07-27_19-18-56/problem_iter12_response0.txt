```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, ema_period=5):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    ema_high_low_spread = high_low_spread.ewm(span=ema_period, min_periods=ema_period).mean()
    smoothed_high_low_spread = high_low_spread - ema_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change(periods=n)
    price_momentum = close_to_close_return.sum()
    
    # Calculate Volume Relative Change
    daily_volume_change = df['volume'].pct_change()
    
    # Filter by Volume Change
    volume_filter = (daily_volume_change > 0).astype(int)
    
    # Confirm with Trading Volume Increase
    volume_increase = (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Combine Momentum and Filters
    combined_momentum = smoothed_high_low_spread + price_momentum
    final_alpha_factor = combined_momentum * volume_filter * volume_increase
    
    return final_alpha_factor
```
