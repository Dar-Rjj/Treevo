```python
import pandas as pd

def heuristics_v2(df, sma_period=20, momentum_periods=10):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread with SMA
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    high_low_momentum = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change(periods=momentum_periods)
    price_momentum = close_to_close_return.rolling(window=momentum_periods).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'].diff()
    volume_indicator = (volume_change > 0).astype(int)
    
    # Final Alpha Factor Construction
    high_low_factor = high_low_momentum * volume_indicator
    price_momentum_factor = price_momentum * volume_indicator
    alpha_factor = (high_low_factor + price_momentum_factor) / 2.0
    
    return alpha_factor
```
