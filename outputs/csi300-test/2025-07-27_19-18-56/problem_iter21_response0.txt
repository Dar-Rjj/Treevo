import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    ema_period = 10
    ema_high_low = high_low_spread.ewm(span=ema_period, adjust=False).mean()
    smoothed_high_low_spread = high_low_spread - ema_high_low
    
    # Calculate Composite Momentum Indicator
    n_periods = 5
    close_to_close_return = (df['close'] / df['close'].shift(n_periods)) - 1
    composite_momentum = close_to_close_return.rolling(window=n_periods).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_increase = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_term_ema_period = 5
    long_term_ema_period = 20
    daily_avg_high_low = (df['high'] + df['low']) / 2
    short_term_ema = daily_avg_high_low.ewm(span=short_term_ema_period, adjust=False).mean()
    long_term_ema = daily_avg_high_low.ewm(span=long_term_ema_period, adjust=False).mean()
    convergence_divergence = short_term_ema - long_term_ema
    
    # Filter by Volume Change
    integrated_momentum = composite_momentum * volume_increase + convergence_divergence
    filtered_momentum = integrated_momentum.where(volume_change > 0, 0)
    
    # Combine Indicators
    final_alpha_factor = filtered_momentum
    
    return final_alpha_factor
