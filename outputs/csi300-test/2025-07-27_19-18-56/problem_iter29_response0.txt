```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Close Return
    df['close_to_close_return'] = df['close'].pct_change()

    # Compute Composite Momentum
    n_periods = 10
    df['composite_momentum'] = df['close_to_close_return'].rolling(window=n_periods).sum() * (df['volume'] > df['volume'].shift(1)).astype(int)

    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['volume_weighted_high_low_range'] = (df['high_low_range'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    df['close_to_open_return'] = df['close'] - df['open']
    df['adjusted_volume_weighted_high_low_range'] = df['volume_weighted_high_low_range'] + df['close_to_open_return'].apply(lambda x: 1 if x > 0 else -1)

    # Smoothed High-Low Spread
    ema_period = 10
    df['high_low_spread'] = df['high'] - df['low']
    df['ema_high_low_spread'] = df['high_low_spread'].ewm(span=ema_period, adjust=False).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['ema_high_low_spread']

    # Integrate with Convergence-Divergence
    df['short_term_ma'] = (df['high'] + df['low']) / 2.0
    df['long_term_ma'] = df['short_term_ma'].rolling(window=20).mean()
    df['convergence_divergence'] = df['short_term_ma'].rolling(window=5).mean() - df['long_term_ma']

    # Filter by Volume Change
    df['volume_change'] = df['volume'].diff()
    df['filtered_composite_momentum'] = df['composite_momentum'] * (df['volume_change'] > 0).astype(int)
    df['filtered_smoothed_high_low_spread'] = df['smoothed_high_low_spread'] * (df['volume_change'] > 0).astype(int)
    df['filtered_convergence_divergence'] = df['convergence_divergence'] * (df['volume_change'] > 0).astype(int)

    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)

    # Final Alpha Factor Construction
    df['final_alpha_factor'] = (
        df['adjusted_volume_weighted_high_low_range'] * (df['volume_change'] > 0).astype(int) +
        df['filtered_smoothed_high_low_spread'] +
        df['filtered_composite_momentum'] +
        df['filtered_convergence_divergence']
    ) * df['trend_confirmation']

    return df['final_alpha_factor']
```
