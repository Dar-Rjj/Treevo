import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted High-Low Range
    df['VolWeightedHL'] = 0.0
    for i in range(1, 11):
        df['VolWeightedHL'] += (df['high'].shift(i) - df['low'].shift(i)) * df['volume'].shift(i)
    df['VolWeightedHL'] /= df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    df['CloseToOpenRet'] = df['close'] - df['open']
    df['AdjustedVolWeightedHL'] = df['VolWeightedHL'] + df['CloseToOpenRet']
    
    # Smoothed High-Low Spread
    sma_window = 10
    df['HighLowSpread'] = df['high'] - df['low']
    df['SMA_HighLowSpread'] = df['HighLowSpread'].rolling(window=sma_window).mean()
    df['SmoothedHighLowSpread'] = df['HighLowSpread'] - df['SMA_HighLowSpread']
    
    # Price Momentum
    momentum_window = 10
    df['PriceMomentum'] = df['close'] - df['close'].shift(momentum_window)
    
    # Trading Volume Increase Indicator
    df['VolumeIncrease'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Final Alpha Factor Construction
    df['CombinedFactor'] = (
        df['AdjustedVolWeightedHL'] * df['VolumeIncrease'] +
        df['SmoothedHighLowSpread'] * df['VolumeIncrease'] +
        df['PriceMomentum'] * df['VolumeIncrease']
    )
    
    return df['CombinedFactor'].dropna()
