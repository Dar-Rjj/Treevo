```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted High-Low Range over the past 10 days
    df['Volume_Weighted_High_Low_Range'] = (df['High_Low_Range'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Initialize the score column
    df['Score'] = 0
    
    # Assign scores based on the comparison of current and past Volume-Weighted High-Low Range
    for i in range(10, len(df)):
        if df.loc[df.index[i], 'Volume_Weighted_High_Low_Range'] > df.loc[df.index[i-1], 'Volume_Weighted_High_Low_Range']:
            df.loc[df.index[i], 'Score'] += 1
        else:
            df.loc[df.index[i], 'Score'] -= 1
    
    # Modify the score based on Market Sentiment
    df.loc[df['Close_to_Open_Return'] > 0, 'Score'] += 1
    df.loc[df['Close_to_Open_Return'] < 0, 'Score'] -= 1
    
    return df['Score']
```
