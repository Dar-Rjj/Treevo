```python
import pandas as pd

def heuristics_v2(df):
    # Volume-Weighted High-Low Range
    df['VW_HL_Range'] = (df['high'] - df['low']).rolling(window=10).apply(
        lambda x: (x * df.loc[x.index, 'volume']).sum() / df.loc[x.index, 'volume'].sum(), raw=False)
    
    # Adjust for Market Sentiment
    sentiment_factor = (df['close'] - df['open']) / df['open']
    df['Adjusted_VW_HL_Range'] = df['VW_HL_Range'] * (1 + sentiment_factor)

    # Smoothing High-Low Spread with SMA
    SMA_period = 10
    df['HL_Spread'] = df['high'] - df['low']
    df['SMA_HL_Spread'] = df['HL_Spread'].rolling(window=SMA_period).mean()
    df['Smoothed_HL_Spread'] = df['HL_Spread'] - df['SMA_HL_Spread']

    # Calculate Composite Momentum Indicator
    momentum_period = 10
    df['Close_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Composite_Momentum'] = df['Close_Return'].rolling(window=momentum_period).sum()

    # Confirm with Trading Volume Increase
    df['Volume_Increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['Confirmed_Composite_Momentum'] = df['Composite_Momentum'] * df['Volume_Increase']

    # Enhanced Moving Average Convergence Divergence of High-Low Prices
    short_term = 7
    long_term = 25
    signal_line = 12
    df['Avg_High_Low'] = (df['high'] + df['low']) / 2
    df['Short_MA_HL'] = df['Avg_High_Low'].rolling(window=short_term).mean()
    df['Long_MA_HL'] = df['Avg_High_Low'].rolling(window=long_term).mean()
    df['MACD_Line'] = df['Short_MA_HL'] - df['Long_MA_HL']
    df['Signal_Line'] = df['MACD_Line'].ewm(span=signal_line, adjust=False).mean()
    df['Enhanced_MACD'] = df['MACD_Line'] - df['Signal_Line']

    # Final Alpha Factor Construction
    alpha_factor = (
        df['Adjusted_VW_HL_Range'] * df['Volume_Increase'] +
        df['Smoothed_HL_Spread'] * df['Volume_Increase'] +
        df['Confirmed_Composite_Momentum'] +
        df['Enhanced_MACD']
    )
    return alpha_factor
```
