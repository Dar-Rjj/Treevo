import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    def compute_close_to_close_return(df, n=5):
        return (df['close'] / df['close'].shift(n)) - 1
    
    # Sum over n periods
    def sum_over_n_periods(returns, n=5):
        return returns.rolling(window=n).sum()
    
    # Compute Volume Change
    def compute_volume_change(df, n=5):
        return df['volume'] - df['volume'].shift(n)
    
    # Check if Volume is Increasing
    def volume_indicator(df, n=5):
        return (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Calculate Short Term and Long Term Moving Average of High and Low
    def calculate_moving_averages(df, short_window=5, long_window=20):
        hl_avg = (df['high'] + df['low']) / 2
        short_ma = hl_avg.rolling(window=short_window).mean()
        long_ma = hl_avg.rolling(window=long_window).mean()
        return short_ma, long_ma
    
    # Subtract Long Term MA from Short Term MA
    def enhanced_convergence_divergence(short_ma, long_ma):
        return short_ma - long_ma
    
    # Calculate High-Low Spread
    def high_low_spread(df):
        return df['high'] - df['low']
    
    # Smoothing High-Low Spread
    def smooth_high_low_spread(spread, sma_period=5):
        sma_spread = spread.rolling(window=sma_period).mean()
        return spread - sma_spread
    
    # Filter by Volume Change
    def filter_by_volume_change(integrated_momentum, volume_change):
        return integrated_momentum.where(volume_change > 0, 0)
    
    # Compute the composite momentum indicator
    close_to_close_returns = compute_close_to_close_return(df)
    composite_momentum = sum_over_n_periods(close_to_close_returns)
    
    # Confirm with trading volume increase
    volume_change = compute_volume_change(df)
    vol_indicator = volume_indicator(df)
    confirmed_momentum = composite_momentum * vol_indicator
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_ma, long_ma = calculate_moving_averages(df)
    ecd = enhanced_convergence_divergence(short_ma, long_ma)
    integrated_momentum = confirmed_momentum + ecd
    
    # Calculate and smooth High-Low Spread
    hl_spread = high_low_spread(df)
    smoothed_hl_spread = smooth_high_low_spread(hl_spread)
    
    # Combine Indicators
    final_factor = integrated_momentum + smoothed_hl_spread
    
    # Filter by Volume Change
    final_factor = filter_by_volume_change(final_factor, volume_change)
    
    return final_factor
