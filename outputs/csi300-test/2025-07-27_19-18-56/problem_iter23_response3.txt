```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Composite Momentum Indicator
    n_periods = 5
    close_to_close_return = df['close'].pct_change(periods=n_periods)
    composite_momentum = close_to_close_return.rolling(window=n_periods).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'].diff()
    volume_increase = (volume_change > 0).astype(int)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_term_ma = (df[['high', 'low']].mean(axis=1)).rolling(window=5).mean()
    long_term_ma = (df[['high', 'low']].mean(axis=1)).rolling(window=20).mean()
    convergence_divergence = short_term_ma - long_term_ma
    
    # Volume-Adjusted High-Low Range
    past_days = 10
    volume_adjusted_high_low = (high_low_spread * df['volume']).rolling(window=past_days).sum() / df['volume'].rolling(window=past_days).sum()
    
    # Adjust for Market Sentiment
    close_to_open_return = df['close'] - df['open']
    market_sentiment = np.where(close_to_open_return > 0, 1, -1)
    adjusted_volume_high_low = volume_adjusted_high_low * market_sentiment
    
    # Smoothed High-Low Spread using EMA
    ema_period = 10
    ema_high_low_spread = high_low_spread.ewm(span=ema_period, adjust=False).mean()
    smoothed_ema_high_low_spread = high_low_spread - ema_high_low_spread
    
    # Price Momentum
    price_momentum = (df['close'] - df['close'].shift(n_periods)).rolling(window=n_periods).sum()
    
    # Trend Confirmation
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    trend_confirmation = (sma_5 > sma_20).astype(int)
    
    # Filter by Volume Change
    integrated_momentum = composite_momentum * volume_increase
    filtered_integrated_momentum = np.where(volume_change > 0, integrated_momentum, 0)
    
    # Combine Indicators
    combined_factors = (
        filtered_integrated_momentum +
        convergence_divergence +
        adjusted_volume_high_low +
        smoothed_ema_high_low_spread
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * trend_confirmation
    
    return final_alpha_factor
```
