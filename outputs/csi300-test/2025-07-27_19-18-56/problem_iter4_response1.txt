```python
import pandas as pd

def heuristics_v2(df, sma_period=10):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Filter and Combine
    high_low_momentum = smoothed_high_low_spread
    high_low_momentum[volume_change <= 0] = 0
    
    return high_low_momentum
```
