```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return and sum over n periods
    n = 5
    df['Close_Return'] = df['close'].pct_change()
    df['Momentum'] = df['Close_Return'].rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    df['Volume_Change'] = df['volume'] > df['volume'].shift(1)
    df['Volume_Indicator'] = df['Volume_Change'].astype(int)
    df['Momentum_Adjusted'] = df['Momentum'] * df['Volume_Indicator']

    # Volume-Adjusted High-Low Range
    df['HL_Range'] = (df['high'] - df['low']) * df['volume']
    df['HL_Range_Adjusted'] = df['HL_Range'].rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    df['CO_Return'] = df['close'] - df['open']
    df['Sentiment_Adjustment'] = df['HL_Range_Adjusted'] + df['CO_Return'] / df['HL_Range_Adjusted']
    df['Sentiment_Adjustment'] = df['Sentiment_Adjustment'].fillna(0)

    # Smoothed High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    ema_period = 10
    df['EMA_HL_Spread'] = df['High_Low_Spread'].ewm(span=ema_period, adjust=False).mean()
    df['Smoothed_HL_Spread'] = df['High_Low_Spread'] - df['EMA_HL_Spread']

    # Integrate with Convergence-Divergence of High-Low Prices
    df['Avg_HL'] = (df['high'] + df['low']) / 2
    df['Short_MA'] = df['Avg_HL'].rolling(window=5).mean()
    df['Long_MA'] = df['Avg_HL'].rolling(window=20).mean()
    df['Convergence_Divergence'] = df['Short_MA'] - df['Long_MA']

    # Filter by Volume Change
    df['Filtered_Momentum'] = df['Momentum_Adjusted'] * df['Volume_Indicator']
    df['Filtered_HL_Spread'] = df['Smoothed_HL_Spread'] * df['Volume_Indicator']

    # Trend Confirmation
    df['5_day_EMA'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_day_EMA'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Trend_Confirmation'] = (df['5_day_EMA'] > df['20_day_EMA']).astype(int)

    # Final Alpha Factor Construction
    df['Combined_Factor'] = (df['Sentiment_Adjustment'] * df['Volume_Indicator'] + 
                             df['Filtered_HL_Spread'] * df['Volume_Indicator'] + 
                             df['Filtered_Momentum'])
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['Trend_Confirmation']

    # Confirm with Trading Volume Increase
    df['Volume_Change_n'] = df['volume'] > df['volume'].shift(n)
    df['Volume_Increase_Indicator'] = df['Volume_Change_n'].astype(int)
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['Volume_Increase_Indicator']

    return df['Final_Alpha_Factor']
```
