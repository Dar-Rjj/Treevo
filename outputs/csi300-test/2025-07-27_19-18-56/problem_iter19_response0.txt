```python
import pandas as pd

def heuristics_v2(df, n=10, sma_period=20, short_term_ma=5, long_term_ma=20):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change(periods=n)
    price_momentum = close_to_close_return.rolling(window=n).sum()
    
    # Calculate Volume Relative Change
    daily_volume_change = (df['volume'] / df['volume'].shift(1)).fillna(1)  # handle first day
    
    # Incorporate Convergence Divergence of High-Low Prices
    short_term_ma_high = df['high'].rolling(window=short_term_ma).mean()
    short_term_ma_low = df['low'].rolling(window=short_term_ma).mean()
    long_term_ma_high = df['high'].rolling(window=long_term_ma).mean()
    long_term_ma_low = df['low'].rolling(window=long_term_ma).mean()
    convergence_divergence_high_low = (short_term_ma_high - long_term_ma_high) - (short_term_ma_low - long_term_ma_low)
    
    # Combine Momentum and Volume Indicators
    volume_filter_indicator = (daily_volume_change > 1).astype(int)
    composite_price_volume_momentum = price_momentum * volume_filter_indicator
    
    # Final Alpha Factor
    final_alpha_factor = composite_price_volume_momentum + convergence_divergence_high_low + smoothed_high_low_spread
    
    return final_alpha_factor
```
