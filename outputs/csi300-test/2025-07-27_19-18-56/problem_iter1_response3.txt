```python
import pandas as pd

def heuristics_v2(df):
    """
    High-Low Range Momentum Indicator.
    
    Args:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: Factor values indexed by date.
    """
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate the 10-day average of the High-Low Range
    avg_high_low_range = high_low_range.rolling(window=10).mean()
    
    # Compare the current High-Low Range with the 10-day average
    def assign_score(current, average):
        if current > average:
            return 1  # Positive Score
        else:
            return -1  # Negative Score
    
    scores = high_low_range.apply(assign_score, args=(avg_high_low_range,))
    
    return scores
```
