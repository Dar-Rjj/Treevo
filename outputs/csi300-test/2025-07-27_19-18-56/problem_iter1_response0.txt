```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Average of High and Low
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate Short Term Moving Average of High and Low over 5 days
    short_term_ma = df['avg_high_low'].rolling(window=5).mean()
    
    # Calculate Long Term Moving Average of High and Low over 20 days
    long_term_ma = df['avg_high_low'].rolling(window=20).mean()
    
    # Subtract Long Term MA from Short Term MA to get the MACD
    macd = short_term_ma - long_term_ma
    
    return macd
```
