```python
import pandas as pd

def heuristics_v2(df, sma_period=20):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread with SMA
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Volume Relative Change
    volume_change = df['volume'].diff().fillna(0)
    
    # Filter by Volume Change
    volume_filter = (volume_change > 0).astype(int)
    
    # Calculate Final Alpha Factor
    alpha_factor = smoothed_high_low_spread * volume_filter
    
    return alpha_factor
```
