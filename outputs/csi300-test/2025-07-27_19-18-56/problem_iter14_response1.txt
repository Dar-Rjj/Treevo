```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()

    # Sum over n periods (e.g., 10 days)
    n = 10
    df['composite_momentum'] = df['close_return'].rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'].diff()
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Multiply Momentum by Volume Indicator
    df['momentum_volume_confirmed'] = df['composite_momentum'] * df['volume_increase']

    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    df['hl_avg'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['hl_avg'].rolling(window=5).mean()
    df['long_term_ma'] = df['hl_avg'].rolling(window=20).mean()
    df['enhanced_convergence_divergence'] = df['short_term_ma'] - df['long_term_ma']

    # Combine Indicators
    df['integrated_momentum'] = df['momentum_volume_confirmed'] + df['enhanced_convergence_divergence']

    # Calculate Enhanced High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    sma_period = 5
    df['sma_high_low_spread'] = df['high_low_spread'].rolling(window=sma_period).mean()
    df['enhanced_high_low_spread'] = df['high_low_spread'] - df['sma_high_low_spread']

    # Filter by Volume Change
    df['integrated_momentum_filtered'] = df['integrated_momentum'] * (df['volume_change'] > 0).astype(int)

    # Incorporate Convergence Divergence of High-Low Prices
    df['final_alpha_factor'] = df['integrated_momentum_filtered'] + df['enhanced_high_low_spread'] + df['enhanced_convergence_divergence']

    return df['final_alpha_factor']
```
