```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, short_window=5, long_window=20, signal_window=9):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change(periods=n)
    
    # Sum over n periods to get the Composite Momentum Indicator
    df['composite_momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Compute Volume Change and Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'].pct_change(periods=n)
    df['volume_increase'] = (df['volume'] > df.shift(n)['volume']).astype(int)
    
    # Multiply Momentum by Volume Indicator
    df['confirmed_momentum'] = df['composite_momentum'] * df['volume_increase']
    
    # Calculate Daily Average of High and Low
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate Short Term Moving Average of High and Low
    df['short_ma'] = df['avg_price'].rolling(window=short_window).mean()
    
    # Calculate Long Term Moving Average of High and Low
    df['long_ma'] = df['avg_price'].rolling(window=long_window).mean()
    
    # Calculate the MACD Line
    df['macd_line'] = df['short_ma'] - df['long_ma']
    
    # Calculate Signal Line
    df['signal_line'] = df['macd_line'].ewm(span=signal_window, adjust=False).mean()
    
    # Adjusted Convergence Divergence
    df['adjusted_macd'] = df['macd_line'] - df['signal_line']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['confirmed_momentum'] + df['adjusted_macd']
    
    return df['alpha_factor']
```
