```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Composite Momentum Indicator
    n_periods = 5
    close_to_close_return = (df['close'] / df['close'].shift(n_periods)) - 1
    composite_momentum = close_to_close_return.rolling(window=n_periods).sum()
    
    # Calculate Volume Relative Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Filter by Volume Change
    volume_filter = (daily_volume_change > 0).astype(int)
    high_low_momentum = smoothed_high_low_spread * volume_filter
    
    # Confirm with Trading Volume Increase
    volume_increase = (df['volume'] > df['volume'].shift(1)).astype(int)
    filtered_high_low_momentum = high_low_momentum * volume_increase
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    avg_high_low = (df['high'] + df['low']) / 2
    short_term_ma = avg_high_low.rolling(window=5).mean()
    long_term_ma = avg_high_low.rolling(window=20).mean()
    diff_short_long = short_term_ma - long_term_ma
    signal_line = diff_short_long.ewm(span=12, adjust=False).mean()
    convergence_divergence = diff_short_long - signal_line
    
    # Combine Indicators
    combined_momentum = composite_momentum + convergence_divergence
    final_combined_momentum = filtered_high_low_momentum * volume_increase * combined_momentum
    
    return final_combined_momentum
```
