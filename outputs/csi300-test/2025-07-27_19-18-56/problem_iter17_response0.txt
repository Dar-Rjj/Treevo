```python
import pandas as pd

def heuristics_v2(df, n=5, sma_period=20, short_term_ma=5, long_term_ma=20):
    # Calculate Enhanced High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    df['SMA_High-Low_Spread'] = df['High-Low_Spread'].rolling(window=sma_period).mean()
    df['Smoothed_High-Low_Spread'] = df['High-Low_Spread'] - df['SMA_High-Low_Spread']

    # Calculate Price Momentum
    df['Close_to_Close_Return'] = (df['close'] / df['close'].shift(n)) - 1
    df['Price_Momentum'] = df['Close_to_Close_Return'].rolling(window=n).sum()

    # Calculate Volume Relative Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)

    # Filter by Volume Change
    df['Volume_Filter'] = df.apply(lambda row: row['Smoothed_High-Low_Spread'] if row['Volume_Change'] > 0 else 0, axis=1)

    # Incorporate Convergence Divergence of High-Low Prices
    df['Short_Term_MA_High'] = df['high'].rolling(window=short_term_ma).mean()
    df['Short_Term_MA_Low'] = df['low'].rolling(window=short_term_ma).mean()
    df['Long_Term_MA_High'] = df['high'].rolling(window=long_term_ma).mean()
    df['Long_Term_MA_Low'] = df['low'].rolling(window=long_term_ma).mean()
    df['Convergence_Divergence'] = (df['Short_Term_MA_High'] + df['Short_Term_MA_Low']) - (df['Long_Term_MA_High'] + df['Long_Term_MA_Low'])

    # Combine Momentum and Volume Indicators
    df['Combined_Momentum'] = df['Price_Momentum'] * df['Volume_Filter']
    df['Volume_Increase'] = (df['volume'] > df['volume'].shift(n)).astype(int)
    df['Interim_Combined_Momentum'] = df['Combined_Momentum'] * df['Volume_Increase']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Interim_Combined_Momentum'] + df['Convergence_Divergence']

    # Confirm with Trading Volume Increase
    df['Final_Volume_Increase'] = (df['volume'] > df['volume'].shift(n)).astype(int)
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * df['Final_Volume_Increase']

    return df['Final_Alpha_Factor']
```
