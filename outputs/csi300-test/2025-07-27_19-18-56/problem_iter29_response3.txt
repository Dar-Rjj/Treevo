```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, short_term=5, long_term=20, ema_period=10):
    # Compute Close-to-Close Return and Sum over n periods
    df['close_return'] = df['close'].pct_change()
    df['price_momentum'] = df['close_return'].rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'].diff()
    df['volume_increase'] = (df['volume_change'] > 0).astype(int)

    # Calculate Volume-Weighted High-Low Range
    df['hl_range'] = df['high'] - df['low']
    df['volume_weighted_hl'] = (df['hl_range'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()

    # Smoothing High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['ema_hl_spread'] = df['high_low_spread'].ewm(span=ema_period, adjust=False).mean()
    df['smoothed_hl_spread'] = df['high_low_spread'] - df['ema_hl_spread']

    # Integrate with Convergence-Divergence of High-Low Prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['avg_high_low'].rolling(window=short_term).mean()
    df['long_term_ma'] = df['avg_high_low'].rolling(window=long_term).mean()
    df['high_low_converge_diverge'] = df['short_term_ma'] - df['long_term_ma']

    # Filter by Volume Change
    df['filtered_high_low_cd'] = df['high_low_converge_diverge'] * df['volume_increase']

    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=short_term).mean()
    df['sma_20'] = df['close'].rolling(window=long_term).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)

    # Include Open-Price Volatility
    df['open_to_high'] = df['high'] - df['open']
    df['open_to_low'] = df['open'] - df['low']
    df['open_price_volatility'] = df['open_to_high'] + df['open_to_low']
    df['open_price_volatility_sum'] = df['open_price_volatility'].rolling(window=n).sum()

    # Integrate Intraday Price Movement
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_return_sum'] = df['intraday_return'].rolling(window=n).sum()

    # Adjust for Market Sentiment
    df['close_to_open_return'] = df['close'] - df['open']
    df['sentiment_adjusted_hl'] = df['volume_weighted_hl'] + df['close_to_open_return'] * 0.1  # Simple adjustment factor

    # Final Alpha Factor Construction
    df['alpha_factor'] = (
        df['sentiment_adjusted_hl'] * df['volume_increase'] +
        df['smoothed_hl_spread'] * df['volume_increase'] +
        df['price_momentum'] * df['volume_increase'] +
        df['open_price_volatility_sum'] * df['volume_increase'] +
        df['intraday_return_sum'] * df['volume_increase']
    ) * df['trend_confirmation']

    return df['alpha_factor']
```
