```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, ema_period=10):
    # Compute Close-to-Close Return and sum over n periods
    close_returns = df['close'].pct_change()
    momentum = close_returns.rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    volume_indicator = (df['volume'] > df['volume'].shift(1)).astype(int)
    confirmed_momentum = momentum * volume_indicator

    # Calculate Volume-Weighted High-Low Range
    high_low_range = (df['high'] - df['low']) * df['volume']
    volume_sum = df['volume'].rolling(window=10).sum()
    weighted_high_low = high_low_range.rolling(window=10).sum() / volume_sum

    # Adjust for Market Sentiment
    close_open_return = (df['close'] - df['open']) / df['open']
    sentiment_adjustment = np.where(close_open_return > 0, 1, -1)
    adjusted_high_low = weighted_high_low * sentiment_adjustment

    # Smoothing High-Low Spread using EMA
    high_low_spread = df['high'] - df['low']
    ema_high_low = high_low_spread.ewm(span=ema_period, adjust=False).mean()
    smoothed_high_low = high_low_spread - ema_high_low

    # Integrate with Convergence-Divergence of High-Low Prices
    short_term_ma = (df['high'] + df['low']) / 2.0
    long_term_ma = (df['high'] + df['low']) / 2.0
    short_term_ma = short_term_ma.rolling(window=5).mean()
    long_term_ma = long_term_ma.rolling(window=20).mean()
    convergence_divergence = short_term_ma - long_term_ma

    # Filter by Volume Change
    integrated_factors = convergence_divergence * volume_indicator

    # Trend Confirmation
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    trend_confirmation = (sma_5 > sma_20).astype(int)

    # Integrate Price-Volatility Factor
    price_volatility = df['close'].rolling(window=20).std()
    price_volatility_factor = price_volatility * volume_indicator

    # Final Alpha Factor Construction
    combined_high_low = adjusted_high_low + smoothed_high_low
    combined_momentum_volatility = confirmed_momentum + price_volatility_factor
    final_alpha_factor = combined_high_low * combined_momentum_volatility * trend_confirmation

    return final_alpha_factor
```
