```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    n = 10  # Number of periods for the momentum sum
    df['close_return'] = df['close'].pct_change()
    df['momentum'] = df['close_return'].rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Calculate Short Term and Long Term Moving Average of High and Low
    df['high_low_avg'] = (df['high'] + df['low']) / 2
    short_term_ma = df['high_low_avg'].rolling(window=5).mean()
    long_term_ma = df['high_low_avg'].rolling(window=20).mean()

    # Subtract Long Term MA from Short Term MA
    df['enhanced_convergence_divergence'] = short_term_ma - long_term_ma

    # Combine Indicators
    df['integrated_momentum'] = df['momentum'] + df['enhanced_convergence_divergence']

    # Filter by Volume Change
    df['alpha_factor'] = df.apply(lambda row: row['integrated_momentum'] if row['volume_change'] == 1 else 0, axis=1)

    # Output the Resulting Alpha Factor
    return df['alpha_factor']
```
