```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    n = 10  # Number of periods for momentum calculation
    ema_period = 10  # EMA period for high-low spread smoothing
    short_term_ma_period = 5
    long_term_ma_period = 20

    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change(periods=n)
    
    # Sum over n periods
    df['momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    df['volume_change'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['momentum'] *= df['volume_change']
    
    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) * df['volume']
    df['volume_weighted_high_low'] = df['high_low_range'].rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    df['close_open_return'] = df['close'] - df['open']
    df['volume_weighted_high_low'] += df['close_open_return'] * df['close_open_return'].apply(lambda x: 1 if x > 0 else -1)
    
    # Smoothed High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['ema_high_low_spread'] = df['high_low_spread'].ewm(span=ema_period, adjust=False).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['ema_high_low_spread']
    
    # Integrate with Convergence-Divergence of High-Low Prices
    df['daily_avg_high_low'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['daily_avg_high_low'].rolling(window=short_term_ma_period).mean()
    df['long_term_ma'] = df['daily_avg_high_low'].rolling(window=long_term_ma_period).mean()
    df['convergence_divergence'] = df['short_term_ma'] - df['long_term_ma']
    
    # Filter by Volume Change
    df['filtered_convergence_divergence'] = df['convergence_divergence'] * df['volume_change']
    
    # Trend Confirmation
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['trend_confirmation'] = (df['5_day_sma'] > df['20_day_sma']).astype(int)
    
    # Final Alpha Factor Construction
    df['alpha_factor'] = (
        df['volume_weighted_high_low'] * df['volume_change'] +
        df['smoothed_high_low_spread'] * df['volume_change'] +
        df['momentum'] * df['volume_change'] +
        df['filtered_convergence_divergence']
    ) * df['trend_confirmation']
    
    return df['alpha_factor']
```
