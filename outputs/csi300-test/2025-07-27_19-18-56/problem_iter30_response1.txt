import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Close-to-Close Return
    df['Close_Return'] = df['close'].pct_change()
    
    # Sum of Close-to-Close Returns over n periods
    df['Price_Momentum'] = df['Close_Return'].rolling(window=n).sum()
    
    # Volume Change
    df['Volume_Change'] = df['volume'].diff()
    df['Volume_Indicator'] = (df['Volume_Change'] > 0).astype(int)
    
    # Volume-Weighted High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    df['Volume_Weighted_High_Low'] = (df['High_Low_Range'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Smoothed High-Low Spread
    df['Smoothed_High_Low_Spread'] = df['High_Low_Range'] - df['High_Low_Range'].ewm(span=5, adjust=False).mean()
    
    # Short Term and Long Term Moving Average of High and Low
    df['Daily_Avg_High_Low'] = (df['high'] + df['low']) / 2
    df['Short_Term_MA'] = df['Daily_Avg_High_Low'].rolling(window=5).mean()
    df['Long_Term_MA'] = df['Daily_Avg_High_Low'].rolling(window=20).mean()
    df['Convergence_Divergence'] = df['Short_Term_MA'] - df['Long_Term_MA']
    
    # Filter by Volume Change
    df['Integrated_Momentum'] = df['Convergence_Divergence'] * df['Volume_Indicator']
    df['Integrated_High_Low_Spread'] = df['Smoothed_High_Low_Spread'] * df['Volume_Indicator']
    
    # Trend Confirmation
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Trend_Confirmation'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Include Open-Price Volatility
    df['Open_Price_Volatility'] = ((df['high'] - df['open']).abs() + (df['low'] - df['open']).abs()).rolling(window=n).sum()
    
    # Integrate Intraday Price Movement
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    df['Intraday_Movement'] = df['Intraday_Return'].rolling(window=n).sum()
    
    # Accumulated Volume
    df['Accumulated_Volume'] = df['volume'].rolling(window=n).sum()
    
    # Final Alpha Factor Construction
    df['Alpha_Factor'] = (
        df['Integrated_Momentum'] +
        df['Integrated_High_Low_Spread'] +
        (df['Price_Momentum'] * df['Volume_Indicator']) +
        (df['Open_Price_Volatility'] * df['Volume_Indicator']) +
        (df['Intraday_Movement'] * df['Volume_Indicator']) +
        (df['Accumulated_Volume'] * df['Volume_Indicator'])
    ) * df['Trend_Confirmation']
    
    return df['Alpha_Factor']
