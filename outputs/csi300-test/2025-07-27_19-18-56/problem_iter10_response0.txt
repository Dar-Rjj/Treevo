```python
import pandas as pd

def heuristics_v2(df, sma_period=20, momentum_periods=10):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    high_low_momentum = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = (df['close'].shift(-momentum_periods) / df['close']) - 1
    price_momentum = close_to_close_return.rolling(window=momentum_periods).sum()
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Combine Factors Based on Volume Change
    combined_factors = []
    for idx, row in df.iterrows():
        if volume_change.loc[idx] > 0:
            factor = high_low_momentum.loc[idx] + price_momentum.loc[idx]
        else:
            factor = 0
        combined_factors.append(factor)
    
    # Final Alpha Factor Construction
    alpha_factor = pd.Series(combined_factors, index=df.index)
    
    return alpha_factor
```
