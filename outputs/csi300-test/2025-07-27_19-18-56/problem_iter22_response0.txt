```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return and Sum over n periods (n=5)
    df['close_return'] = df['close'].pct_change()
    df['momentum'] = df['close_return'].rolling(window=5).sum()

    # Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'].diff()
    df['volume_indicator'] = (df['volume_change'] > 0).astype(int)
    df['confirmed_momentum'] = df['momentum'] * df['volume_indicator']

    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['volume_weighted_high_low'] = (df['high_low_range'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    df['close_open_return'] = df['close'] - df['open']
    df['adjusted_volume_weighted_high_low'] = df['volume_weighted_high_low'] + df['close_open_return']

    # Smoothed High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['ema_high_low_spread'] = df['high_low_spread'].ewm(span=10, adjust=False).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['ema_high_low_spread']

    # Integrate with Convergence-Divergence of High-Low Prices
    df['short_term_ma'] = (df['high'] + df['low']) / 2
    df['long_term_ma'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['short_term_ma'].rolling(window=5).mean()
    df['long_term_ma'] = df['long_term_ma'].rolling(window=20).mean()
    df['convergence_divergence'] = df['short_term_ma'] - df['long_term_ma']

    # Filter by Volume Change
    df['filtered_convergence_divergence'] = df['convergence_divergence'] * df['volume_indicator']

    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)

    # Final Alpha Factor Construction
    df['final_alpha'] = (df['adjusted_volume_weighted_high_low'] * df['volume_indicator'] +
                         df['smoothed_high_low_spread'] * df['volume_indicator'] +
                         df['confirmed_momentum'] * df['volume_indicator']) * df['trend_confirmation']

    return df['final_alpha']
```
