```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates the Volume-Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: A series representing the factor values, indexed by date.
    """
    # Calculate Close-to-Open return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
