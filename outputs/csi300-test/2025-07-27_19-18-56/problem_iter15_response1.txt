```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['High_Low_Range'] = (df['high'] - df['low']) * df['volume']
    df['Volume_Weighted_HL'] = df['High_Low_Range'].rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Volume_Adjusted_HL'] = df['Volume_Weighted_HL'] + df['Close_to_Open_Return']
    
    # Smoothing High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    df['EMA_High_Low_Spread'] = df['High_Low_Spread'].ewm(span=10, adjust=False).mean()
    df['Smoothed_High_Low_Spread'] = df['High_Low_Spread'] - df['EMA_High_Low_Spread']
    
    # Calculate Composite Momentum Indicator
    n = 5
    df['Close_Return'] = (df['close'] / df['close'].shift(n)) - 1
    df['Composite_Momentum'] = df['Close_Return'].rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    df['Volume_Increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['Confirmed_Composite_Momentum'] = df['Composite_Momentum'] * df['Volume_Increase']
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    df['Avg_High_Low'] = (df['high'] + df['low']) / 2
    df['Short_Term_MA'] = df['Avg_High_Low'].rolling(window=7).mean()
    df['Long_Term_MA'] = df['Avg_High_Low'].rolling(window=25).mean()
    df['MACD'] = df['Short_Term_MA'] - df['Long_Term_MA']
    df['Signal_Line'] = df['MACD'].ewm(span=9, adjust=False).mean()
    df['Enhanced_MACD'] = df['MACD'] - df['Signal_Line']
    
    # Final Alpha Factor Construction
    df['Adjusted_Volume_Weighted_HL'] = df['Volume_Adjusted_HL'] * df['Volume_Increase']
    df['Adjusted_Smoothed_High_Low_Spread'] = df['Smoothed_High_Low_Spread'] * df['Volume_Increase']
    df['Alpha_Factor'] = (df['Adjusted_Volume_Weighted_HL'] + 
                          df['Adjusted_Smoothed_High_Low_Spread'] + 
                          df['Confirmed_Composite_Momentum'] + 
                          df['Enhanced_MACD'])
    
    return df['Alpha_Factor']
```
