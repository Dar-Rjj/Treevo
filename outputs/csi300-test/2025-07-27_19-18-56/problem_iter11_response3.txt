import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Sum over n periods (e.g., 10 days)
    df['momentum'] = df['close_return'].rolling(window=10).sum()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Check if Volume is Increasing
    df['volume_increase'] = (df['volume_change'] > 0).astype(int)
    
    # Multiply Momentum by Volume Indicator
    df['momentum_vol_confirmed'] = df['momentum'] * df['volume_increase']
    
    # Calculate Daily Average of High and Low
    df['high_low_avg'] = (df['high'] + df['low']) / 2
    
    # Calculate Short Term Moving Average of High and Low
    df['short_term_ma'] = df['high_low_avg'].rolling(window=5).mean()
    
    # Calculate Long Term Moving Average of High and Low
    df['long_term_ma'] = df['high_low_avg'].rolling(window=20).mean()
    
    # Subtract Long Term MA from Short Term MA
    df['convergence_divergence'] = df['short_term_ma'] - df['long_term_ma']
    
    # Combine Indicators
    df['composite_momentum'] = df['momentum_vol_confirmed'] + df['convergence_divergence']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10
    df['sma_high_low_spread'] = df['high_low_spread'].rolling(window=sma_period).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['sma_high_low_spread']
    
    # Filter by Volume Change
    df['alpha_factor'] = df.apply(lambda row: row['composite_momentum'] if row['volume_change'] > 0 else 0, axis=1)
    
    # Final Alpha Factor Construction
    df['alpha_factor'] = df['alpha_factor'] + df['smoothed_high_low_spread']
    
    # Output the Resulting Alpha Factor
    return df['alpha_factor']
