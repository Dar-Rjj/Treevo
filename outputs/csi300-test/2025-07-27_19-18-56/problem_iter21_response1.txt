```python
import pandas as pd
import numpy as np

def heuristics_v2(df, ema_period=10, momentum_periods=5, volatility_periods=10):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    ema_high_low_spread = high_low_spread.ewm(span=ema_period, min_periods=ema_period).mean()
    smoothed_high_low_spread = high_low_spread - ema_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change()
    price_momentum = close_to_close_return.rolling(window=momentum_periods).sum()
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=volatility_periods).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = price_momentum / volatility
    
    # Calculate Volume Relative Change
    volume_change = df['volume'].pct_change()
    
    # Filter by Volume Change
    combined_momentum = adjusted_momentum.where(volume_change > 0, 0)
    
    # Confirm with Trading Volume Increase
    volume_increase_indicator = (df['volume'] > df['volume'].shift(momentum_periods)).astype(int)
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * volume_increase_indicator
    
    return alpha_factor
```
