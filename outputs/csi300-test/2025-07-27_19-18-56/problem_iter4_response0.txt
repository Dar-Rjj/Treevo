```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Short term moving average (5 days)
    short_term_ma = avg_price.rolling(window=5).mean()
    
    # Long term moving average (20 days)
    long_term_ma = avg_price.rolling(window=20).mean()
    
    # Subtract long term MA from short term MA
    macd_signal = short_term_ma - long_term_ma
    
    return macd_signal
```
