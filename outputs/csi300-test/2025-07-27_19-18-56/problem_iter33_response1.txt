```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Sum over n periods
    df['momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'].diff()
    df['volume_indicator'] = (df['volume_change'] > 0).astype(int)
    df['momentum_confirmed'] = df['momentum'] * df['volume_indicator']
    
    # Volume-Adjusted High-Low Range
    df['high_low_range'] = (df['high'] - df['low'])
    df['volume_adjusted_high_low'] = (df['high_low_range'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Adjust for Market Sentiment
    df['close_open_return'] = df['close'] - df['open']
    sentiment_factor = np.where(df['close_open_return'] > 0, 1.1, 0.9)
    df['adjusted_volume_high_low'] = df['volume_adjusted_high_low'] * sentiment_factor
    
    # Smoothed High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['ema_high_low_spread'] = df['high_low_spread'].ewm(span=n, adjust=False).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['ema_high_low_spread']
    
    # Integrate with Convergence-Divergence of High-Low Prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['avg_high_low'].rolling(window=5).mean()
    df['long_term_ma'] = df['avg_high_low'].rolling(window=20).mean()
    df['high_low_convergence'] = df['short_term_ma'] - df['long_term_ma']
    
    # Filter by Volume Change
    df['integrated_momentum'] = df['momentum_confirmed'] + df['smoothed_high_low_spread'] + df['high_low_convergence']
    df['integrated_momentum'] = df['integrated_momentum'] * df['volume_indicator']
    
    # Trend Confirmation
    df['5_ema'] = df['close'].ewm(span=5, adjust=False).mean()
    df['20_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['trend_confirmation'] = (df['5_ema'] > df['20_ema']).astype(int)
    
    # Final Alpha Factor Construction
    df['alpha_factor'] = (df['adjusted_volume_high_low'] + df['smoothed_high_low_spread'] + df['momentum_confirmed']) * df['trend_confirmation']
    
    # Enhance with Additional Volume Check
    df['volume_increase'] = (df['volume'] > df['volume'].shift(n)).astype(int)
    df['final_alpha_factor'] = df['alpha_factor'] * df['volume_increase']
    
    return df['final_alpha_factor']
```
