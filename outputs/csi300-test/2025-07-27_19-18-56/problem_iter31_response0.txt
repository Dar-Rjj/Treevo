```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return and sum over n periods (n=5)
    n = 5
    df['close_return'] = df['close'].pct_change()
    df['momentum'] = df['close_return'].rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['confirmed_momentum'] = df['momentum'] * df['volume_increase']

    # Calculate Volume-Weighted High-Low Range
    df['high_low_range_vol_weighted'] = ((df['high'] - df['low']) * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    df['close_open_return'] = (df['close'] - df['open']) / df['open']
    df['market_sentiment'] = df['close_open_return'].apply(lambda x: 1 if x > 0 else -1)
    df['adjusted_high_low_range'] = df['high_low_range_vol_weighted'] * df['market_sentiment']

    # Smoothing High-Low Spread
    df['high_low_sma'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['smoothed_high_low_spread'] = (df['high'] - df['low']) - df['high_low_sma']

    # Integrate with Convergence-Divergence of High-Low Prices
    df['short_term_high_ma'] = df['high'].rolling(window=5).mean()
    df['short_term_low_ma'] = df['low'].rolling(window=5).mean()
    df['long_term_high_ma'] = df['high'].rolling(window=20).mean()
    df['long_term_low_ma'] = df['low'].rolling(window=20).mean()
    df['high_low_converge_diverge'] = (df['short_term_high_ma'] - df['long_term_high_ma']) - (df['short_term_low_ma'] - df['long_term_low_ma'])

    # Filter by Volume Change
    df['filtered_high_low_cd'] = df['high_low_converge_diverge'] * df['volume_increase']

    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)

    # Final Alpha Factor Construction
    df['alpha_factor'] = (df['adjusted_high_low_range'] + df['smoothed_high_low_spread'] + df['confirmed_momentum']) * df['trend_confirmation']

    return df['alpha_factor']
```
