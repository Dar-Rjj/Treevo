import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    n = 5  # Number of periods
    df['close_return'] = df['close'].pct_change(n)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Composite Price-Volume Momentum
    df['composite_momentum'] = df['close_return'] * df['volume_increase']
    
    # Calculate Enhanced High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10
    df['sma_high_low_spread'] = df['high_low_spread'].rolling(window=sma_period).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['sma_high_low_spread']
    
    # Integrate with Convergence-Divergence of High-Low Prices
    short_term_ma = 5
    long_term_ma = 20
    df['daily_avg_high_low'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['daily_avg_high_low'].rolling(window=short_term_ma).mean()
    df['long_term_ma'] = df['daily_avg_high_low'].rolling(window=long_term_ma).mean()
    df['convergence_divergence'] = df['short_term_ma'] - df['long_term_ma']
    
    # Combine Indicators
    df['combined_indicator'] = df['composite_momentum'] + df['convergence_divergence']
    
    # Filter by Volume Change
    df['final_alpha_factor'] = df['combined_indicator'] * (df['volume_change'] > 0)
    
    # Drop NaN values
    df.dropna(inplace=True)
    
    return df['final_alpha_factor']
