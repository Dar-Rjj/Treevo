```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return over n periods
    n = 5
    df['close_return'] = (df['close'].pct_change(n)).fillna(0)
    
    # Confirm with Trading Volume Increase
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['momentum_with_volume'] = df['close_return'] * df['volume_increase']
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    short_term_ma = df['avg_high_low'].rolling(window=5).mean()
    long_term_ma = df['avg_high_low'].rolling(window=20).mean()
    df['enhanced_macd'] = short_term_ma - long_term_ma
    
    # Combine Indicators
    df['composite_momentum'] = df['momentum_with_volume'] + df['enhanced_macd']
    
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10
    df['sma_high_low_spread'] = df['high_low_spread'].rolling(window=sma_period).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['sma_high_low_spread']
    
    # Filter by Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['filtered_momentum'] = df.apply(lambda row: row['composite_momentum'] if row['volume_change'] > 0 else 0, axis=1)
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['filtered_momentum'] + df['smoothed_high_low_spread']
    
    return df['alpha_factor']
```
