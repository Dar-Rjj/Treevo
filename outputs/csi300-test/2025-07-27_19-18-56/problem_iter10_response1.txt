import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['High-Low'] = df['high'] - df['low']
    df['Volume-Weighted-High-Low'] = (df['High-Low'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    df['Close-to-Open'] = df['close'] - df['open']
    sentiment_adjustment = df['Close-to-Open'].apply(lambda x: 1 if x > 0 else -1)
    df['Adjusted-VWHL'] = df['Volume-Weighted-High-Low'] * sentiment_adjustment
    
    # Smoothing High-Low Spread
    SMA_period = 10
    df['SMA-High-Low'] = df['High-Low'].rolling(window=SMA_period).mean()
    df['Smoothed-High-Low'] = df['High-Low'] - df['SMA-High-Low']
    
    # Calculate Price Momentum
    n_periods = 10
    df['Close-to-Close-Return'] = df['close'].pct_change(periods=n_periods)
    df['Price-Momentum'] = df['Close-to-Close-Return'].rolling(window=n_periods).sum()
    
    # Confirm with Trading Volume Increase
    df['Volume-Indicator'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Final Alpha Factor Construction
    df['Combined-Factor'] = (df['Adjusted-VWHL'] + df['Smoothed-High-Low']) * df['Volume-Indicator'] + df['Price-Momentum'] * df['Volume-Indicator']
    
    # Output the Resulting Alpha Factor
    return df['Combined-Factor']
