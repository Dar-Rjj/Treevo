```python
import pandas as pd

def heuristics_v2(df, n=10, short_term_ma=5, long_term_ma=20, sma_period=5):
    # Compute Close-to-Close Return
    close_to_close_return = df['close'].pct_change()
    
    # Sum over n periods for Composite Momentum Indicator
    composite_momentum = close_to_close_return.rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'].pct_change()
    volume_indicator = (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Multiply Momentum by Volume Indicator
    momentum_with_volume = composite_momentum * volume_indicator
    
    # Integrate with Convergence Divergence of High-Low Prices
    daily_avg_high_low = (df['high'] + df['low']) / 2
    short_term_ma_high_low = daily_avg_high_low.rolling(window=short_term_ma).mean()
    long_term_ma_high_low = daily_avg_high_low.rolling(window=long_term_ma).mean()
    high_low_macd = short_term_ma_high_low - long_term_ma_high_low
    
    # Combine Indicators
    integrated_momentum = momentum_with_volume + high_low_macd
    
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    high_low_spread_smoothed = high_low_spread - sma_high_low_spread
    
    # Filter by Volume Change
    final_factor = integrated_momentum.copy()
    final_factor[volume_change <= 0] = 0
    
    # Calculate Final Alpha Factor
    final_alpha_factor = final_factor + high_low_spread_smoothed
    
    return final_alpha_factor
```
