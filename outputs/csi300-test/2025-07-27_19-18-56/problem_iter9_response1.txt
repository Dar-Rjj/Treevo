```python
import pandas as pd

def heuristics_v2(df):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Sum over n periods (e.g., 14 days)
    df['momentum'] = df['close_return'].rolling(window=14).sum()
    
    # Compute Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Check if Volume is Increasing
    df['volume_increase'] = (df['volume'] > df['volume'].shift()).astype(int)
    
    # Multiply Momentum by Volume Indicator
    df['composite_momentum'] = df['momentum'] * df['volume_increase']
    
    # Calculate Daily Average of High and Low
    df['high_low_avg'] = (df['high'] + df['low']) / 2
    
    # Calculate Short Term Moving Average of High and Low
    df['short_term_ma'] = df['high_low_avg'].rolling(window=7).mean()
    
    # Calculate Long Term Moving Average of High and Low
    df['long_term_ma'] = df['high_low_avg'].rolling(window=25).mean()
    
    # Calculate the Difference between Short and Long Term MAs
    df['macd_line'] = df['short_term_ma'] - df['long_term_ma']
    
    # Calculate Signal Line
    df['signal_line'] = df['macd_line'].ewm(span=12, adjust=False).mean()
    
    # Combine Indicators
    df['alpha_factor'] = df['composite_momentum'] + (df['macd_line'] - df['signal_line'])
    
    return df['alpha_factor']
```
