```python
import pandas as pd

def heuristics_v2(df, n=10, sma_period=20, short_term_ma=5, long_term_ma=20):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothed High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change(periods=n)
    price_momentum = close_to_close_return.rolling(window=n).sum()
    
    # Calculate Daily Volume Change
    volume_change = df['volume'].pct_change()
    
    # Incorporate Convergence-Divergence of High-Low Prices
    short_term_high_ma = df['high'].rolling(window=short_term_ma).mean()
    long_term_high_ma = df['high'].rolling(window=long_term_ma).mean()
    short_term_low_ma = df['low'].rolling(window=short_term_ma).mean()
    long_term_low_ma = df['low'].rolling(window=long_term_ma).mean()
    convergence_divergence = (short_term_high_ma - long_term_high_ma) - (short_term_low_ma - long_term_low_ma)
    
    # Composite Price-Volume Momentum
    composite_price_volume_momentum = price_momentum + volume_change
    
    # Final Alpha Factor
    alpha_factor = smoothed_high_low_spread + composite_price_volume_momentum + convergence_divergence
    
    return alpha_factor
```
