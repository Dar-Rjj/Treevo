```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, sma_period=5, short_term=5, long_term=20, signal_line=9):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low
    
    # Calculate Price Momentum
    close_returns = df['close'].pct_change()
    price_momentum = close_returns.rolling(window=n).sum()
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    daily_avg_high_low = (df['high'] + df['low']) / 2
    short_term_sma = daily_avg_high_low.rolling(window=short_term).mean()
    long_term_sma = daily_avg_high_low.rolling(window=long_term).mean()
    
    macd_line = short_term_sma - long_term_sma
    signal_line = macd_line.ewm(span=signal_line, adjust=False).mean()
    adjusted_macd = macd_line - signal_line
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'].pct_change()
    volume_increase = (volume_change > 0).astype(int)
    
    # Final Alpha Factor
    combined_momentum = smoothed_high_low_spread + price_momentum
    final_alpha_factor = (combined_momentum * volume_increase) + adjusted_macd
    
    return final_alpha_factor
```
