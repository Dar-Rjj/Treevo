```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    ema_period = 10
    n_periods = 10
    short_term_ma = 5
    long_term_ma = 20

    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Smooth High-Low Spread with EMA
    ema_high_low = high_low_spread.ewm(span=ema_period, adjust=False).mean()
    smoothed_high_low = high_low_spread - ema_high_low

    # Volume-Adjusted High-Low Range
    volume_adjusted_high_low = (df['high'] - df['low']).rolling(window=n_periods).apply(
        lambda x: np.sum(x * df.loc[x.index, 'volume']) / df['volume'].rolling(window=n_periods).sum(),
        raw=False
    )

    # Adjust for Market Sentiment
    close_to_open_return = df['close'] - df['open']
    volume_adjusted_high_low = volume_adjusted_high_low + close_to_open_return

    # Calculate Composite Momentum Indicator
    close_to_close_return = df['close'].diff(periods=n_periods)
    composite_momentum = close_to_close_return.rolling(window=n_periods).sum()

    # Confirm with Trading Volume Increase
    volume_change = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Integrate with Enhanced Convergence Divergence
    daily_avg_high_low = (df['high'] + df['low']) / 2
    short_term_ma_daily_avg = daily_avg_high_low.rolling(window=short_term_ma).mean()
    long_term_ma_daily_avg = daily_avg_high_low.rolling(window=long_term_ma).mean()
    enhanced_convergence_divergence = short_term_ma_daily_avg - long_term_ma_daily_avg

    # Trend Confirmation
    ema_5 = df['close'].ewm(span=short_term_ma, adjust=False).mean()
    ema_20 = df['close'].ewm(span=long_term_ma, adjust=False).mean()
    trend_confirmation = (ema_5 > ema_20).astype(int)

    # Combine Indicators
    combined_factors = (
        (composite_momentum * volume_change) +
        enhanced_convergence_divergence +
        (volume_adjusted_high_low + smoothed_high_low)
    ) * trend_confirmation

    return combined_factors
```
