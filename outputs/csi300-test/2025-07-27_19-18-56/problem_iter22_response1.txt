```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, ema_period=9, short_term=5, long_term=20):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Sum over n periods
    df['price_momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'].diff()
    df['volume_indicator'] = (df['volume_change'] > 0).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['volume_weighted_high_low'] = (df['high_low_range'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Smoothed High-Low Spread
    df['high_low_spread'] = df['high_low_range']
    df['ema_high_low_spread'] = df['high_low_spread'].ewm(span=ema_period, adjust=False).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['ema_high_low_spread']
    
    # Integrate with Convergence-Divergence of High-Low Prices
    df['daily_avg_high_low'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['daily_avg_high_low'].rolling(window=short_term).mean()
    df['long_term_ma'] = df['daily_avg_high_low'].rolling(window=long_term).mean()
    df['high_low_convergence_divergence'] = df['short_term_ma'] - df['long_term_ma']
    
    # Filter by Volume Change
    df['integrated_momentum_high_low'] = df['price_momentum'] + df['high_low_convergence_divergence']
    df.loc[df['volume_indicator'] == 0, 'integrated_momentum_high_low'] = 0
    
    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=short_term).mean()
    df['sma_20'] = df['close'].rolling(window=long_term).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)
    
    # Final Alpha Factor Construction
    df['adjusted_volume_weighted_high_low'] = df['volume_weighted_high_low'] * df['volume_indicator']
    df['adjusted_smoothed_high_low_spread'] = df['smoothed_high_low_spread'] * df['volume_indicator']
    df['price_momentum_adjusted'] = df['price_momentum'] * df['volume_indicator']
    df['final_alpha_factor'] = (df['adjusted_volume_weighted_high_low'] + 
                                df['adjusted_smoothed_high_low_spread'] + 
                                df['price_momentum_adjusted']) * df['trend_confirmation']
    
    return df['final_alpha_factor']
```
