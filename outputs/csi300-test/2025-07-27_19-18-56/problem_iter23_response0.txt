```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return over n periods
    n = 10
    df['close_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['composite_momentum'] = df['close_return'].rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'] > df['volume'].shift(1)
    df['volume_indicator'] = df['volume_change'].apply(lambda x: 1 if x else 0)
    df['confirmed_momentum'] = df['composite_momentum'] * df['volume_indicator']

    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['volume_weighted_high_low'] = (df['high_low_range'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    df['close_open_return'] = df['close'] - df['open']
    df['adjusted_volume_weighted_high_low'] = df['volume_weighted_high_low'] + df['close_open_return']

    # Smoothed High-Low Spread
    ema_period = 14
    df['high_low_spread'] = df['high'] - df['low']
    df['ema_high_low_spread'] = df['high_low_spread'].ewm(span=ema_period, adjust=False).mean()
    df['smoothed_high_low_spread'] = df['high_low_spread'] - df['ema_high_low_spread']

    # Integrate with Convergence-Divergence of High-Low Prices
    df['daily_avg_high_low'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['daily_avg_high_low'].rolling(window=5).mean()
    df['long_term_ma'] = df['daily_avg_high_low'].rolling(window=20).mean()
    df['convergence_divergence'] = df['short_term_ma'] - df['long_term_ma']

    # Filter by Volume Change
    df['filtered_convergence_divergence'] = df['convergence_divergence'] * df['volume_indicator']

    # Trend Confirmation
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['trend_confirmation'] = (df['5_day_sma'] > df['20_day_sma']).astype(int)

    # Final Alpha Factor Construction
    df['combined_factors'] = (
        df['adjusted_volume_weighted_high_low'] * df['volume_indicator'] +
        df['smoothed_high_low_spread'] * df['volume_indicator'] +
        df['confirmed_momentum'] * df['volume_indicator']
    )
    df['alpha_factor'] = df['combined_factors'] * df['trend_confirmation']

    return df['alpha_factor']
```
