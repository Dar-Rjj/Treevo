```python
import pandas as pd

def heuristics_v2(df, n=10, sma_period=5, short_term=5, long_term=20):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change(periods=n)
    price_momentum = close_to_close_return.rolling(window=n).sum()
    
    # Calculate Volume Relative Change
    daily_volume_change = df['volume'].pct_change()
    
    # Filter by Volume Change
    volume_filter = (daily_volume_change > 0).astype(int)
    filtered_high_low_momentum = smoothed_high_low_spread * volume_filter
    
    # Incorporate Convergence Divergence of High-Low Prices
    short_term_ma_high = df['high'].rolling(window=short_term).mean()
    short_term_ma_low = df['low'].rolling(window=short_term).mean()
    long_term_ma_high = df['high'].rolling(window=long_term).mean()
    long_term_ma_low = df['low'].rolling(window=long_term).mean()
    convergence_divergence = (short_term_ma_high - long_term_ma_high) - (short_term_ma_low - long_term_ma_low)
    
    # Combine Momentum and Volume Indicators
    combined_momentum = (price_momentum + filtered_high_low_momentum) * (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Final Alpha Factor
    alpha_factor = combined_momentum + convergence_divergence
    
    return alpha_factor
```
