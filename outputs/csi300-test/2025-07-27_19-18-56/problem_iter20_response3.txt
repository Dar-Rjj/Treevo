import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the average true range over a 14-day period
    df['ATR'] = (df['high'] - df['low']).rolling(window=14).mean()
    
    # Calculate the relative strength compared to the average true range
    rs = (df['close'] - df['open']) / df['ATR']
    
    # Calculate the volume weighted close price
    df['VWAP'] = (df['amount'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the momentum of the volume weighted close price
    df['MOM_VWAP'] = df['VWAP'].pct_change(periods=10)
    
    # Calculate the final alpha factor
    alpha_factor = rs * df['MOM_VWAP']
    
    return alpha_factor
