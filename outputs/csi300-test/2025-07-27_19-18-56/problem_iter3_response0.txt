```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average High-Low Range over the past 10 days
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=10).mean()
    
    # Assign Positive or Negative Score based on current vs average high-low range
    df['range_score'] = df.apply(lambda row: 1 if row['high_low_range'] > row['avg_high_low_range'] else -1, axis=1)
    
    # Calculate Average Volume over the past 10 days
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Adjust the score based on current vs average volume
    df['volume_score'] = df.apply(lambda row: 1 if row['volume'] > row['avg_volume'] else -1, axis=1)
    
    # Combine the scores
    df['final_factor'] = df['range_score'] + df['volume_score']
    
    return df['final_factor']
```
