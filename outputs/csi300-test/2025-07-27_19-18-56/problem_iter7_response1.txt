```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, sma_period=20):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    close_returns = df['close'].pct_change(n)
    price_momentum = close_returns.rolling(window=n).sum()
    
    # Calculate Volume Relative Change
    volume_change = df['volume'].pct_change()
    
    # Filter by Volume Change
    high_low_momentum = smoothed_high_low_spread * price_momentum
    high_low_momentum = high_low_momentum.where(volume_change > 0, 0)
    
    # Confirm with Trading Volume Increase
    volume_increase = (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Final Alpha Factor
    alpha_factor = high_low_momentum * volume_increase
    
    return alpha_factor
```
