```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Volume-Adjusted High-Low Range
    volume_adjusted_high_low_range = (df['high'] - df['low']) * df['volume']
    volume_adjusted_high_low_range = volume_adjusted_high_low_range.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    close_to_open_return = df['close'] - df['open']
    sentiment_adjusted_range = volume_adjusted_high_low_range + close_to_open_return
    
    # Calculate Composite Momentum Indicator
    n_periods = 5
    close_to_close_return = df['close'].pct_change(n_periods)
    composite_momentum = close_to_close_return.rolling(window=n_periods).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'] > df['volume'].shift(1)
    volume_increase_indicator = volume_change.astype(int)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_term_ma_high_low = (df['high'] + df['low']) / 2
    short_term_ma_high_low = short_term_ma_high_low.rolling(window=5).mean()
    
    long_term_ma_high_low = (df['high'] + df['low']) / 2
    long_term_ma_high_low = long_term_ma_high_low.rolling(window=20).mean()
    
    convergence_divergence = short_term_ma_high_low - long_term_ma_high_low
    
    # Smoothed High-Low Spread with EMA
    ema_period = 10
    ema_high_low_spread = high_low_spread.ewm(span=ema_period, adjust=False).mean()
    smoothed_ema_high_low_spread = high_low_spread - ema_high_low_spread
    
    # Price Momentum
    price_momentum = df['close'].diff(n_periods).rolling(window=n_periods).sum()
    
    # Trend Confirmation
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    trend_confirmation = (sma_5 > sma_20).astype(int)
    
    # Filter by Volume Change
    integrated_momentum = composite_momentum * volume_increase_indicator
    integrated_momentum = integrated_momentum.add(convergence_divergence, fill_value=0)
    
    # Combine Indicators
    combined_factors = (sentiment_adjusted_range + smoothed_ema_high_low_spread).add(price_momentum, fill_value=0)
    combined_factors = combined_factors * trend_confirmation
    
    # Final Alpha Factor
    final_alpha_factor = integrated_momentum + combined_factors
    return final_alpha_factor.dropna()
```
