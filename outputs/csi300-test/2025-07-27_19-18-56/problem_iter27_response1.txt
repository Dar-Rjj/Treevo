```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, ema_period=10, short_term_ma=5, long_term_ma=20):
    # Compute Close-to-Close Return
    df['close_return'] = df['close'].pct_change()
    
    # Sum over n periods
    df['price_momentum'] = df['close_return'].rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    df['volume_change'] = df['volume'].pct_change()
    df['volume_increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Volume-Weighted High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['volume_weighted_high_low'] = (df['high_low_range'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Smoothed High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    df['ema_high_low'] = df['high_low_spread'].ewm(span=ema_period, adjust=False).mean()
    df['smoothed_high_low'] = df['high_low_spread'] - df['ema_high_low']
    
    # Integrate with Convergence-Divergence of High-Low Prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    df['short_term_ma'] = df['avg_high_low'].rolling(window=short_term_ma).mean()
    df['long_term_ma'] = df['avg_high_low'].rolling(window=long_term_ma).mean()
    df['macd'] = df['short_term_ma'] - df['long_term_ma']
    
    # Filter by Volume Change
    df['filtered_macd'] = df['macd'] * df['volume_increase']
    df['filtered_volume_weighted_high_low'] = df['volume_weighted_high_low'] * df['volume_increase']
    df['filtered_smoothed_high_low'] = df['smoothed_high_low'] * df['volume_increase']
    
    # Trend Confirmation
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['trend_confirmation'] = (df['sma_5'] > df['sma_20']).astype(int)
    
    # Include Open-Price Volatility
    df['open_high_diff'] = df['high'] - df['open']
    df['open_low_diff'] = df['open'] - df['low']
    df['open_price_volatility'] = (df['open_high_diff'] + df['open_low_diff']).rolling(window=n).sum()
    
    # Integrate Intraday Price Movement
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['intraday_return_sum'] = df['intraday_return'].rolling(window=n).sum()
    
    # Final Alpha Factor Construction
    df['alpha_factor'] = (
        df['filtered_volume_weighted_high_low'] +
        df['filtered_smoothed_high_low'] +
        df['price_momentum'] * df['volume_increase'] +
        df['open_price_volatility'] * df['volume_increase'] +
        df['intraday_return_sum'] * df['volume_increase']
    ) * df['trend_confirmation']
    
    return df['alpha_factor']
```
