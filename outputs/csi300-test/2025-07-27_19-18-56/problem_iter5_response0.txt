```python
import pandas as pd

def heuristics_v2(df, sma_period=20, momentum_periods=5):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread using SMA
    sma_high_low = high_low_spread.rolling(window=sma_period).mean()
    high_low_momentum = high_low_spread - sma_high_low
    
    # Calculate Price Momentum
    close_return = df['close'].pct_change()
    price_momentum = close_return.rolling(window=momentum_periods).sum()
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Initialize the combined factors
    combined_factors = pd.Series(index=df.index)
    
    # Filter and Combine
    for date in df.index:
        if volume_change.loc[date] > 0:
            combined_factors.loc[date] = high_low_momentum.loc[date] + price_momentum.loc[date]
        else:
            combined_factors.loc[date] = 0
    
    # Final Alpha Factor Construction
    alpha_factor = combined_factors
    
    return alpha_factor
```
