```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    sum_high_low_volume = (df['High_Low_Range'] * df['volume']).rolling(window=10).sum()
    sum_volume = df['volume'].rolling(window=10).sum()
    volume_weighted_high_low_range = sum_high_low_volume / sum_volume
    
    # Adjust for Market Sentiment
    df['Close_to_Open_Return'] = df['close'] - df['open']
    market_sentiment_adjustment = volume_weighted_high_low_range + df['Close_to_Open_Return'] / 100.0
    
    # Calculate High-Low Spread and Smoothing
    df['High_Low_Spread'] = df['high'] - df['low']
    sma_high_low_spread = df['High_Low_Spread'].rolling(window=10).mean()
    smoothed_high_low_spread = df['High_Low_Spread'] - sma_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change(periods=10)
    price_momentum = close_to_close_return.sum()
    
    # Confirm with Trading Volume Increase
    df['Volume_Change'] = df['volume'].diff()
    volume_increase_indicator = (df['Volume_Change'] > 0).astype(int)
    
    # Final Alpha Factor Construction
    adjusted_volume_weighted_high_low_range = market_sentiment_adjustment * volume_increase_indicator
    combined_high_low_spread = smoothed_high_low_spread * volume_increase_indicator
    combined_price_momentum = price_momentum * volume_increase_indicator
    alpha_factor = adjusted_volume_weighted_high_low_range + combined_high_low_spread + combined_price_momentum
    
    return alpha_factor
```
