import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return
    n = 10
    df['Close_Return'] = df['close'].pct_change(periods=n)
    
    # Confirm with Trading Volume Increase
    df['Volume_Change'] = df['volume'].diff()
    df['Volume_Indicator'] = (df['Volume_Change'] > 0).astype(int)
    df['Momentum_with_Volume'] = df['Close_Return'] * df['Volume_Indicator']
    
    # Calculate Volume-Weighted High-Low Range
    df['HL_Range'] = df['high'] - df['low']
    df['Volume_Weighted_HL_Range'] = (df['HL_Range'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    df['CO_Return'] = df['close'] - df['open']
    sentiment_adjustment = df['CO_Return'].apply(lambda x: 1 if x > 0 else -1)
    df['Adjusted_Volume_Weighted_HL_Range'] = df['Volume_Weighted_HL_Range'] * sentiment_adjustment
    
    # Smoothed High-Low Spread
    ema_period = 10
    df['HL_Spread'] = df['high'] - df['low']
    df['EMA_HL_Spread'] = df['HL_Spread'].ewm(span=ema_period, adjust=False).mean()
    df['Smoothed_HL_Spread'] = df['HL_Spread'] - df['EMA_HL_Spread']
    
    # Integrate with Convergence-Divergence of High-Low Prices
    df['Avg_HL'] = (df['high'] + df['low']) / 2
    short_term_ma = df['Avg_HL'].rolling(window=5).mean()
    long_term_ma = df['Avg_HL'].rolling(window=20).mean()
    df['HL_Convergence_Divergence'] = short_term_ma - long_term_ma
    
    # Filter by Volume Change
    df['Filtered_HL_CD'] = df['HL_Convergence_Divergence'] * df['Volume_Indicator']
    
    # Trend Confirmation
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['Trend_Confirmation'] = (df['SMA_5'] > df['SMA_20']).astype(int)
    
    # Final Alpha Factor Construction
    df['Combined_Factor'] = (
        (df['Adjusted_Volume_Weighted_HL_Range'] + df['Smoothed_HL_Spread'] + df['Momentum_with_Volume']) 
        * df['Trend_Confirmation'] * df['Volume_Indicator']
    )
    
    return df['Combined_Factor']
