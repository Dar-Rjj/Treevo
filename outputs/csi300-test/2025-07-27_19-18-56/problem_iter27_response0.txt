import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define EMA periods
    short_ema_period = 5
    long_ema_period = 20
    high_low_ema_period = 10
    
    # Calculate Enhanced High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Smoothing High-Low Spread with EMA
    df['High_Low_Spread_EMA'] = df['High_Low_Spread'].ewm(span=high_low_ema_period, adjust=False).mean()
    df['Smoothed_High_Low_Spread'] = df['High_Low_Spread'] - df['High_Low_Spread_EMA']
    
    # Volume-Adjusted High-Low Range
    df['Volume_Adjusted_High_Low_Range'] = 0.0
    for i in range(1, 11):
        df['Volume_Adjusted_High_Low_Range'] += (df['high'].shift(i) - df['low'].shift(i)) * df['volume'].shift(i)
    df['Volume_Adjusted_High_Low_Range'] /= df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Market_Sentiment_Adjusted_Range'] = df['Volume_Adjusted_High_Low_Range'] + df['Close_to_Open_Return']
    
    # Calculate Composite Momentum Indicator
    df['Close_to_Close_Return'] = df['close'] - df['close'].shift(1)
    df['Composite_Momentum'] = df['Close_to_Close_Return'].rolling(window=10).sum()
    
    # Confirm with Trading Volume Increase
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Increase_Indicator'] = (df['Volume_Change'] > 0).astype(int)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    df['Average_High_Low'] = (df['high'] + df['low']) / 2
    df['Short_Term_MA_HL'] = df['Average_High_Low'].rolling(window=short_ema_period).mean()
    df['Long_Term_MA_HL'] = df['Average_High_Low'].rolling(window=long_ema_period).mean()
    df['Enhanced_Convergence_Divergence'] = df['Short_Term_MA_HL'] - df['Long_Term_MA_HL']
    
    # Price Momentum
    df['Price_Momentum'] = (df['close'] - df['close'].shift(10)).rolling(window=10).sum()
    
    # Trend Confirmation
    df['EMA_5'] = df['close'].ewm(span=short_ema_period, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=long_ema_period, adjust=False).mean()
    df['Trend_Confirmation'] = (df['EMA_5'] > df['EMA_20']).astype(int)
    
    # Filter by Volume Change
    df['Filtered_Composite_Momentum'] = df['Composite_Momentum'] * df['Volume_Increase_Indicator']
    
    # Combine Indicators
    df['Combined_Factor'] = (
        df['Filtered_Composite_Momentum'] +
        df['Enhanced_Convergence_Divergence'] +
        df['Market_Sentiment_Adjusted_Range'] +
        df['Smoothed_High_Low_Spread'] +
        df['Price_Momentum']
    )
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['Trend_Confirmation']
    
    return df['Final_Alpha_Factor']
