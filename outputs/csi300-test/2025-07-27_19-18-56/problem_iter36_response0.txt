```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Close-to-Close Return and sum over n periods
    n = 5
    close_to_close_return = df['close'].diff(n)
    momentum = close_to_close_return.rolling(window=n).sum()

    # Confirm with Trading Volume Increase
    volume_increase = (df['volume'] > df['volume'].shift(1)).astype(int)
    momentum *= volume_increase

    # Calculate Volume-Weighted High-Low Range
    high_low_range = (df['high'] - df['low']) * df['volume']
    volume_weighted_high_low = high_low_range.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    close_to_open_return = df['close'] - df['open']
    market_sentiment_adjustment = np.where(close_to_open_return > 0, 1, -1)
    volume_weighted_high_low_adjusted = volume_weighted_high_low * (1 + market_sentiment_adjustment)

    # Smoothing High-Low Spread
    ema_period = 10
    high_low_spread_ema = df['high'] - df['low']
    high_low_spread_smoothed = high_low_spread_ema.ewm(span=ema_period, adjust=False).mean()
    high_low_spread_adjusted = (df['high'] - df['low']) - high_low_spread_smoothed

    # Integrate with Convergence-Divergence of High-Low Prices
    short_term_ma_high_low = df[['high', 'low']].rolling(window=5).mean()
    long_term_ma_high_low = df[['high', 'low']].rolling(window=20).mean()
    convergence_divergence = short_term_ma_high_low.mean(axis=1) - long_term_ma_high_low.mean(axis=1)

    # Filter by Volume Change
    integrated_factors = np.where(volume_increase > 0, convergence_divergence, 0)

    # Trend Confirmation
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    trend_confirmation = (sma_5 > sma_20).astype(int)

    # Incorporate Price Oscillator
    short_term_ema_close = df['close'].ewm(span=12, adjust=False).mean()
    long_term_ema_close = df['close'].ewm(span=26, adjust=False).mean()
    price_oscillator = short_term_ema_close - long_term_ema_close

    # Final Alpha Factor Construction
    alpha_factor = (
        volume_weighted_high_low_adjusted +
        high_low_spread_adjusted +
        momentum +
        integrated_factors +
        price_oscillator
    ) * trend_confirmation

    return alpha_factor
```
