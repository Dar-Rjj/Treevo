```python
import pandas as pd

def heuristics_v2(df):
    # Calculate short term moving average (5 days)
    short_term_ma = df[['high', 'low']].mean(axis=1).rolling(window=5).mean()
    
    # Calculate long term moving average (20 days)
    long_term_ma = df[['high', 'low']].mean(axis=1).rolling(window=20).mean()
    
    # Subtract long term MA from short term MA
    macd = short_term_ma - long_term_ma
    
    return macd
```
