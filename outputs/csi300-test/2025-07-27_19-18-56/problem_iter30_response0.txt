```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread with EMA (EMA period: 10)
    ema_high_low_spread = high_low_spread.ewm(span=10, adjust=False).mean()
    smoothed_high_low_spread = high_low_spread - ema_high_low_spread
    
    # Volume-Adjusted High-Low Range
    volume_adjusted_high_low_range = (df['high'] - df['low']) * df['volume']
    volume_adjusted_high_low_range = volume_adjusted_high_low_range.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Adjust for Market Sentiment
    close_to_open_return = df['close'] - df['open']
    market_sentiment_adjustment = np.where(close_to_open_return > 0, 1, -1)
    adjusted_volume_high_low_range = volume_adjusted_high_low_range * market_sentiment_adjustment
    
    # Calculate Composite Momentum Indicator
    n = 10
    close_to_close_return = df['close'] - df['close'].shift(n)
    composite_momentum = close_to_close_return.rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_increase_indicator = np.where(volume_change > 0, 1, 0)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_term_ma_high_low = (df['high'] + df['low']) / 2
    short_term_ma_high_low = short_term_ma_high_low.rolling(window=5).mean()
    long_term_ma_high_low = (df['high'] + df['low']) / 2
    long_term_ma_high_low = long_term_ma_high_low.rolling(window=20).mean()
    enhanced_convergence_divergence = short_term_ma_high_low - long_term_ma_high_low
    
    # Smoothed High-Low Spread
    smoothed_high_low_spread = high_low_spread - high_low_spread.ewm(span=10, adjust=False).mean()
    
    # Price Momentum
    price_momentum = df['close'] - df['close'].shift(n)
    price_momentum = price_momentum.rolling(window=n).sum()
    
    # Trend Confirmation
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    trend_confirmation = np.where(ema_5 > ema_20, 1, 0)
    
    # Filter by Volume Change
    integrated_momentum = composite_momentum * volume_increase_indicator
    
    # Combine Indicators
    combined_factors = (integrated_momentum + 
                        enhanced_convergence_divergence + 
                        adjusted_volume_high_low_range + 
                        smoothed_high_low_spread + 
                        price_momentum) * trend_confirmation
    
    # Incorporate Volume-Weighted Return
    volume_weighted_return = ((df['close'] - df['open']) * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    final_alpha_factor = combined_factors + volume_weighted_return
    
    return final_alpha_factor
```
