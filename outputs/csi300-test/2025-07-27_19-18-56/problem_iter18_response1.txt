```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted High-Low Range
    df['High_Low_Range'] = (df['high'] - df['low']) * df['volume']
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'].rolling(window=10).sum() / df['volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    df['Close_Open_Return'] = df['close'] - df['open']
    df['Market_Sentiment_Adjustment'] = df['Close_Open_Return'].apply(lambda x: 1 if x > 0 else -1)
    df['Adjusted_Volume_Weighted_High_Low_Range'] = df['Volume_Weighted_High_Low_Range'] * df['Market_Sentiment_Adjustment']

    # Smoothing High-Low Spread with EMA
    df['High_Low_Spread'] = df['high'] - df['low']
    df['EMA_High_Low_Spread'] = df['High_Low_Spread'].ewm(span=7, adjust=False).mean()
    df['Smoothed_High_Low_Spread'] = df['High_Low_Spread'] - df['EMA_High_Low_Spread']

    # Calculate Composite Momentum Indicator
    df['Close_To_Close_Return'] = df['close'].pct_change(periods=1)
    df['Composite_Momentum'] = df['Close_To_Close_Return'].rolling(window=10).sum()

    # Confirm with Trading Volume Increase
    df['Volume_Increase'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['Confirmed_Composite_Momentum'] = df['Composite_Momentum'] * df['Volume_Increase']

    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    df['Avg_High_Low'] = (df['high'] + df['low']) / 2
    df['Short_Term_MA'] = df['Avg_High_Low'].rolling(window=7).mean()
    df['Long_Term_MA'] = df['Avg_High_Low'].rolling(window=25).mean()
    df['MACD'] = df['Short_Term_MA'] - df['Long_Term_MA']
    df['Signal_Line'] = df['MACD'].ewm(span=9, adjust=False).mean()
    df['Enhanced_MACD'] = df['MACD'] - df['Signal_Line']

    # Final Alpha Factor Construction
    df['Volume_Weighted_Factor'] = df['Adjusted_Volume_Weighted_High_Low_Range'] * df['Volume_Increase']
    df['Smoothed_Spread_Factor'] = df['Smoothed_High_Low_Spread'] * df['Volume_Increase']
    df['Momentum_Factor'] = df['Confirmed_Composite_Momentum'] * df['Volume_Increase']
    df['Alpha_Factor'] = df['Volume_Weighted_Factor'] + df['Smoothed_Spread_Factor'] + df['Momentum_Factor'] + df['Enhanced_MACD']

    # Incorporate Price Volatility
    df['Volatility'] = df['close'].rolling(window=14).std()
    df['Final_Alpha_Factor'] = df['Alpha_Factor'] / (1 + df['Volatility'])

    return df['Final_Alpha_Factor'].dropna()
```
