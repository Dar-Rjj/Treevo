```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define periods for moving averages
    sma_period = 10
    ema_short_period = 5
    ema_long_period = 20
    momentum_period = 10
    volume_adjustment_period = 10

    # Calculate Enhanced High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']

    # Smoothing High-Low Spread
    df['SMA_High-Low_Spread'] = df['High-Low_Spread'].rolling(window=sma_period).mean()
    df['EMA_High-Low_Spread'] = df['High-Low_Spread'].ewm(span=ema_short_period, adjust=False).mean()
    df['Smoothed_High-Low_Spread'] = df['High-Low_Spread'] - df['EMA_High-Low_Spread']

    # Volume-Adjusted High-Low Range
    df['Volume_Adjusted_High-Low_Range'] = 0
    for i in range(1, volume_adjustment_period + 1):
        df['Volume_Adjusted_High-Low_Range'] += (df['high'].shift(i) - df['low'].shift(i)) * df['volume'].shift(i)
    df['Volume_Adjusted_High-Low_Range'] /= df['volume'].rolling(window=volume_adjustment_period).sum()

    # Adjust for Market Sentiment
    df['Close-to-Open_Return'] = df['close'] - df['open']
    df['Volume_Adjusted_High-Low_Range'] = np.where(df['Close-to-Open_Return'] > 0,
                                                    df['Volume_Adjusted_High-Low_Range'] * 1.1,
                                                    df['Volume_Adjusted_High-Low_Range'] * 0.9)

    # Calculate Price Momentum
    df['Price_Momentum'] = df['close'].diff(momentum_period).rolling(window=momentum_period).sum()

    # Incorporate Convergence Divergence of High-Low Prices
    df['Short_MA_High'] = df['high'].rolling(window=ema_short_period).mean()
    df['Long_MA_High'] = df['high'].rolling(window=ema_long_period).mean()
    df['Short_MA_Low'] = df['low'].rolling(window=ema_short_period).mean()
    df['Long_MA_Low'] = df['low'].rolling(window=ema_long_period).mean()
    df['Convergence_Divergence'] = (df['Short_MA_High'] - df['Long_MA_High']) - (df['Short_MA_Low'] - df['Long_MA_Low'])

    # Combine Momentum and Volume Indicators
    df['Volume_Filter'] = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    df['Combined_Momentum'] = df['Price_Momentum'] * df['Volume_Filter']

    # Smoothed High-Low Spread
    df['Smoothed_High-Low_Spread'] = df['High-Low_Spread'] - df['EMA_High-Low_Spread']

    # Trend Confirmation
    df['EMA_5'] = df['close'].ewm(span=ema_short_period, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=ema_long_period, adjust=False).mean()
    df['Trend_Confirmation'] = np.where(df['EMA_5'] > df['EMA_20'], 1, 0)

    # Filter by Volume Change
    df['Integrated_Momentum'] = df['Combined_Momentum'] * df['Trend_Confirmation']

    # Combine Indicators
    df['Enhanced_Convergence_Divergence'] = df['Convergence_Divergence'] * df['Volume_Filter']
    df['Combined_Factor'] = df['Volume_Adjusted_High-Low_Range'] + df['Smoothed_High-Low_Spread'] + \
                            df['Enhanced_Convergence_Divergence'] + df['Integrated_Momentum']

    # Incorporate Volume-Weighted Return
    df['Volume_Weighted_Return'] = 0
    for i in range(1, volume_adjustment_period + 1):
        df['Volume_Weighted_Return'] += (df['close'].shift(i) - df['open'].shift(i)) * df['volume'].shift(i)
    df['Volume_Weighted_Return'] /= df['volume'].rolling(window=volume_adjustment_period).sum()
    df['Final_Alpha_Factor'] = df['Combined_Factor'] + df['Volume_Weighted_Return']

    return df['Final_Alpha_Factor']
```
