```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Smoothing High-Low Spread with EMA
    ema_period = 10
    high_low_ema = high_low_spread.ewm(span=ema_period, adjust=False).mean()
    smoothed_high_low_spread = high_low_spread - high_low_ema

    # Calculate Price Momentum
    n = 10
    close_to_close_return = (df['close'].shift(-n) / df['close']) - 1
    price_momentum = close_to_close_return.rolling(window=n).sum()

    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    daily_avg_high_low = (df['high'] + df['low']) / 2

    short_term_ema = daily_avg_high_low.ewm(span=5, adjust=False).mean()
    long_term_ema = daily_avg_high_low.ewm(span=20, adjust=False).mean()

    macd_line = short_term_ema - long_term_ema
    signal_line = macd_line.rolling(window=9).mean()
    adjusted_convergence_divergence = macd_line - signal_line

    # Confirm with Trading Volume Increase
    volume_change = df['volume'].pct_change()
    volume_increase_indicator = (volume_change > 0).astype(int)

    # Final Alpha Factor
    combined_momentum = smoothed_high_low_spread + price_momentum
    final_alpha_factor = combined_momentum * volume_increase_indicator + adjusted_convergence_divergence

    return final_alpha_factor
```
