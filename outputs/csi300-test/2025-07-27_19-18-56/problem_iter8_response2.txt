import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_period = 10  # Define SMA period
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    n = 5  # Define the number of periods for momentum
    close_return = df['close'].pct_change(periods=n)
    price_momentum = close_return.sum()
    
    # Calculate Volume Relative Change
    volume_change = df['volume'] / df['volume'].shift(1) - 1
    
    # Filter by Volume Change
    high_low_momentum = smoothed_high_low_spread * (volume_change > 0)
    
    # Confirm with Trading Volume Increase
    volume_increase = (df['volume'] > df['volume'].shift(n)).astype(int)
    
    # Incorporate Convergence Divergence of High-Low Prices
    short_term_ma = (df['high'] + df['low']) / 2
    short_term_ma = short_term_ma.rolling(window=5).mean()
    long_term_ma = (df['high'] + df['low']) / 2
    long_term_ma = long_term_ma.rolling(window=20).mean()
    convergence_divergence = short_term_ma - long_term_ma
    
    # Combine Momentum and Volume Indicators
    combined_momentum = high_low_momentum * volume_increase
    final_factor = combined_momentum + convergence_divergence
    
    return final_factor
