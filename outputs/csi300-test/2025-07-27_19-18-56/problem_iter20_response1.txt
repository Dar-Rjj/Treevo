```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread with SMA
    sma_period = 10
    hl_sma = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - hl_sma
    
    # Calculate Composite Momentum Indicator
    n_periods = 5
    close_to_close_return = df['close'].pct_change(periods=n_periods)
    composite_momentum = close_to_close_return.rolling(window=n_periods).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'].pct_change()
    volume_increase = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    short_term_ma = (df[['high', 'low']].mean(axis=1)).rolling(window=5).mean()
    long_term_ma = (df[['high', 'low']].mean(axis=1)).rolling(window=20).mean()
    enhanced_convergence_divergence = short_term_ma - long_term_ma
    
    # Volume-Adjusted High-Low Range
    past_10_days = 10
    adjusted_hl_range = (df['high'] - df['low']).mul(df['volume'], axis=0).rolling(window=past_10_days).sum() / df['volume'].rolling(window=past_10_days).sum()
    
    # Adjust for Market Sentiment
    close_to_open_return = df['close'] - df['open']
    adjusted_hl_range += close_to_open_return * (1 if close_to_open_return.mean() > 0 else -1)
    
    # Smoothed High-Low Spread with EMA
    ema_period = 10
    hl_ema = high_low_spread.ewm(span=ema_period, adjust=False).mean()
    smoothed_high_low_ema = high_low_spread - hl_ema
    
    # Price Momentum
    price_momentum = df['close'].pct_change(periods=n_periods)
    
    # Trend Confirmation
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    trend_confirmation = (sma_5 > sma_20).astype(int)
    
    # Filter by Volume Change
    integrated_momentum = composite_momentum * volume_increase
    integrated_momentum = integrated_momentum.where(volume_change > 0, 0)
    
    # Combine Indicators
    combined_factors = (integrated_momentum + 
                        enhanced_convergence_divergence + 
                        adjusted_hl_range + 
                        smoothed_high_low_ema)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * trend_confirmation
    
    return final_alpha_factor
```
