```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, sma_period=20, volume_lookback=5):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low = high_low_spread - sma_high_low
    
    # Calculate Composite Momentum Indicator
    close_to_close_return = (df['close'].pct_change(periods=n)).sum()
    
    # Calculate Volume Relative Change
    daily_volume_change = df['volume'].pct_change()
    
    # Filter by Volume Change
    high_low_momentum = smoothed_high_low.where(daily_volume_change > 0, 0)
    
    # Confirm with Trading Volume Increase
    volume_increase = (df['volume'] > df['volume'].shift(volume_lookback)).astype(int)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    avg_high_low = (df['high'] + df['low']) / 2
    short_term_ma = avg_high_low.rolling(window=7).mean()
    long_term_ma = avg_high_low.rolling(window=25).mean()
    convergence_divergence = short_term_ma - long_term_ma
    signal_line = convergence_divergence.ewm(span=12, adjust=False).mean()
    divergence_signal = convergence_divergence - signal_line
    
    # Combine Indicators
    combined_momentum = close_to_close_return + divergence_signal
    
    # Combine Momentum and Volume Indicators
    interim_combined_momentum = combined_momentum * high_low_momentum * volume_increase
    
    # Final Alpha Factor
    final_alpha_factor = interim_combined_momentum + close_to_close_return + divergence_signal
    
    return final_alpha_factor
```
