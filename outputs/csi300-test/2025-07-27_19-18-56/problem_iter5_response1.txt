```python
import pandas as pd

def heuristics_v2(df, sma_period=5, n_periods=5):
    # Compute High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Apply Smoothing to High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_momentum = high_low_spread - sma_high_low_spread
    
    # Determine Volume Change
    volume_change = df['volume'].diff()
    
    # Apply Volume Filter
    high_low_momentum = smoothed_high_low_momentum * (volume_change > 0)
    
    # Integrate Composite Price-Volume Momentum
    close_to_close_return = df['close'].pct_change(periods=n_periods).sum()
    volume_indicator = (df['volume'] > df['volume'].shift(n_periods)).astype(int)
    composite_price_volume_momentum = close_to_close_return * volume_indicator
    
    # Incorporate Convergence Divergence of High-Low Prices
    daily_avg_high_low = (df['high'] + df['low']) / 2
    short_term_ma = daily_avg_high_low.rolling(window=5).mean()
    long_term_ma = daily_avg_high_low.rolling(window=20).mean()
    convergence_divergence = short_term_ma - long_term_ma
    
    # Combine All Indicators
    combined_indicator = high_low_momentum + composite_price_volume_momentum + convergence_divergence
    
    return combined_indicator
```
