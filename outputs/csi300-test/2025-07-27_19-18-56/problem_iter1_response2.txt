import pandas as pd

def heuristics_v2(df, n=20):
    # Compute Close-to-Close Return
    close = df['close']
    returns = close.pct_change()

    # Calculate Price Momentum over n periods
    momentum = returns.rolling(window=n).sum()

    # Compute Volume Change
    volume = df['volume']
    volume_change = volume - volume.shift(n)

    # Check if Volume is Increasing
    volume_indicator = (volume > volume.shift(n)).astype(int)

    # Multiply Momentum by Volume Indicator
    factor_values = momentum * volume_indicator

    return factor_values
