import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['range'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate the daily volatility
    df['volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1) / df['close'].shift(1)
    
    # Calculate the volume-to-amount ratio
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Calculate the weighted average of these metrics to form the alpha factor
    df['alpha_factor'] = 0.4 * df['return'] + 0.3 * df['range'] + 0.2 * df['volatility'] + 0.1 * df['volume_to_amount_ratio']
    
    return df['alpha_factor'].dropna()
