```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted High-Low Range
    df['High_Low_Range'] = df['High'] - df['Low']
    df['Volume_Weighted_High_Low'] = (df['High_Low_Range'] * df['Volume']).rolling(window=10).sum() / df['Volume'].rolling(window=10).sum()

    # Adjust for Market Sentiment
    df['Close_to_Open_Return'] = df['Close'] - df['Open']
    df['Adjusted_Volume_Weighted_High_Low'] = df['Volume_Weighted_High_Low'] + df['Close_to_Open_Return']

    # Smoothed High-Low Spread
    df['High_Low_Spread'] = df['High'] - df['Low']
    df['EMA_High_Low_Spread'] = df['High_Low_Spread'].ewm(span=10, adjust=False).mean()
    df['Smoothed_High_Low_Spread'] = df['High_Low_Spread'] - df['EMA_High_Low_Spread']

    # Price Momentum
    df['Close_to_Close_Return'] = df['Close'] - df['Close'].shift(1)
    df['Price_Momentum'] = df['Close_to_Close_Return'].rolling(window=10).sum()

    # Trading Volume Increase Indicator
    df['Volume_Increase_Indicator'] = (df['Volume'] > df['Volume'].shift(1)).astype(int)

    # Trend Confirmation
    df['5D_SMA'] = df['Close'].rolling(window=5).mean()
    df['20D_SMA'] = df['Close'].rolling(window=20).mean()
    df['Trend_Confirmation'] = (df['5D_SMA'] > df['20D_SMA']).astype(int)

    # Final Alpha Factor Construction
    df['Combined_Factor'] = (
        (df['Adjusted_Volume_Weighted_High_Low'] * df['Volume_Increase_Indicator']) +
        (df['Smoothed_High_Low_Spread'] * df['Volume_Increase_Indicator']) +
        (df['Price_Momentum'] * df['Volume_Increase_Indicator'])
    ) * df['Trend_Confirmation']

    return df['Combined_Factor']
```
