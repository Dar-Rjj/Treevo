```python
import pandas as pd

def heuristics_v2(df, sma_period=10, n=5):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread

    # Calculate Price Momentum
    close_to_close_return = df['close'].pct_change(periods=n)
    price_momentum = close_to_close_return.rolling(window=n).sum()

    # Calculate Volume Relative Change
    volume_change = df['volume'].pct_change()

    # Filter by Volume Change
    volume_filter = (volume_change > 0).astype(int)

    # Confirm with Trading Volume Increase
    volume_increase = (df['volume'] > df['volume'].shift(n)).astype(int)

    # Combine High-Low Momentum and Price Momentum
    combined_momentum = smoothed_high_low_spread * price_momentum

    # Final Alpha Factor
    alpha_factor = combined_momentum * volume_filter * volume_increase

    return alpha_factor
```
