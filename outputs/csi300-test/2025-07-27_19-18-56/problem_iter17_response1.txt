import pandas as pd

def heuristics_v2(df, n=10, sma_period=5, short_term_ma=5, long_term_ma=20):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    high_low_sma = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - high_low_sma
    
    # Calculate Composite Momentum Indicator
    close_returns = df['close'].pct_change(periods=n)
    composite_momentum = close_returns.rolling(window=n).sum()
    
    # Confirm with Trading Volume Increase
    volume_change = df['volume'].pct_change()
    volume_increase = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Integrate with Enhanced Convergence Divergence of High-Low Prices
    avg_price = (df['high'] + df['low']) / 2
    short_term_ma_avg = avg_price.rolling(window=short_term_ma).mean()
    long_term_ma_avg = avg_price.rolling(window=long_term_ma).mean()
    convergence_divergence = short_term_ma_avg - long_term_ma_avg
    
    # Filter by Volume Change
    filtered_convergence_divergence = convergence_divergence * volume_increase
    
    # Combine Indicators
    combined_indicator = (composite_momentum * volume_increase) + filtered_convergence_divergence
    
    return combined_indicator
