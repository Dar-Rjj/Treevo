```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Short term EMA calculation with span 3 over 5 days
    short_ema = avg_price.ewm(span=3, adjust=False).mean()
    
    # Long term EMA calculation with span 7 over 20 days
    long_ema = avg_price.ewm(span=7, adjust=False).mean()
    
    # Subtract long term EMA from short term EMA
    factor_values = short_ema - long_ema
    
    return factor_values
```
