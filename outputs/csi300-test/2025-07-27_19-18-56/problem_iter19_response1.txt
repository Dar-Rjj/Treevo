```python
import pandas as pd

def heuristics_v2(df, sma_period=10, momentum_periods=5, volume_increase_periods=5):
    # Calculate Enhanced High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Smoothing High-Low Spread
    sma_high_low_spread = high_low_spread.rolling(window=sma_period).mean()
    smoothed_high_low_spread = high_low_spread - sma_high_low_spread
    
    # Calculate Price Momentum
    close_to_close_return = (df['close'].shift(-momentum_periods) / df['close']) - 1
    price_momentum = close_to_close_return.rolling(window=momentum_periods).sum()
    
    # Calculate Volume Relative Change
    daily_volume_change = df['volume'].pct_change()
    
    # Filter by Volume Change
    combined_momentum = smoothed_high_low_spread + price_momentum
    combined_momentum = combined_momentum.where(daily_volume_change > 0, 0)
    
    # Confirm with Trading Volume Increase
    volume_increase_indicator = (df['volume'] > df['volume'].shift(volume_increase_periods)).astype(int)
    
    # Final Alpha Factor
    alpha_factor = combined_momentum * volume_increase_indicator
    
    return alpha_factor
```
