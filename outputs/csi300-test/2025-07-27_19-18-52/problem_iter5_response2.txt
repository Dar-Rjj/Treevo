```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute VWMA of Close Prices
    close_prices = df['close']
    volumes = df['volume']
    vwma = (close_prices * volumes).rolling(window=lookback_period).sum() / volumes.rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Change and VWMA
    integrated_factor = intraday_price_change * vwma
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Trading Range
    intraday_trading_range = df['high'] - df['low']
    
    # Volume Intensity Score
    average_high_low = (df['high'] - df['low']).rolling(window=lookback_period).mean()
    volume_intensity_score = volumes / average_high_low
    
    # Composite Factor
    composite_factor = (intraday_return * intraday_trading_range) * volume_intensity_score
    composite_factor = composite_factor + integrated_factor
    
    # Final Alpha Factor
    final_alpha_factor = composite_factor + integrated_factor
    
    return final_alpha_factor
```
