```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity_score = df['volume'] / (df['high'] - df['low']).replace(0, 1)
    
    # Composite Momentum Factor
    composite_momentum_factor = (intraday_return * intraday_range * volume_intensity_score) + 1e-6
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum_factor = (intraday_return + intraday_range) * volume_intensity_score
    
    # Enhanced Composite Momentum Factor
    enhanced_composite_momentum_factor = (intraday_return + intraday_range) * np.sqrt(volume_intensity_score)
    
    # Integrate Enhanced Composite Momentum Factor with VWMA
    integrated_enhanced_momentum = enhanced_composite_momentum_factor * vwma - vwma
    
    # Incorporate Short-Term Price Trend
    short_term_ma = df['close'].rolling(window=5).mean()
    short_term_trend_adjusted = integrated_enhanced_momentum - short_term_ma
    
    # Combine Short-Term Trend and VWMA Adjusted Momentum
    combined_trend_and_momentum = short_term_trend_adjusted * integrated_enhanced_momentum
    
    # Calculate Daily Return
    daily_return = df['close'].diff().fillna(0)
    
    # Integrate Daily Return with Adjusted Composite Momentum Factor
    weighted_daily_return = daily_return * 0.5
    integrated_daily_return_momentum = combined_trend_and_momentum + weighted_daily_return
    
    # Integrate Long-Term Price Trend
    long_term_ma = df['close'].rolling(window=20).mean()
    final_factor = integrated_daily_return_momentum - long_term_ma
    
    return final_factor
```
