```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range over a 10-day window
    df['High-Low'] = df['high'] - df['low']
    df['High-Close_Prev'] = abs(df['high'] - df['close'].shift(1))
    df['Low-Close_Prev'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['High-Low', 'High-Close_Prev', 'Low-Close_Prev']].max(axis=1)
    
    # Compute 10-Day Average True Range (ATR)
    df['ATR_10'] = df['True_Range'].rolling(window=10).mean()
    
    # Calculate Intraday Price Change
    df['Intraday_Price_Change'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['Close_Volume'] = df['close'] * df['volume']
    df['Cumulative_Volume'] = df['volume'].rolling(window=lookback_period).sum()
    df['Cumulative_Close_Volume'] = df['Close_Volume'].rolling(window=lookback_period).sum()
    df['VWMA'] = df['Cumulative_Close_Volume'] / df['Cumulative_Volume']
    
    # Integrate Intraday Price Change with VWMA and ATR
    df['Alpha_Factor'] = (df['Intraday_Price_Change'] * df['VWMA']) / df['ATR_10']
    
    return df['Alpha_Factor'].dropna()
```
