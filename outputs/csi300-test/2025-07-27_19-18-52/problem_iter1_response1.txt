```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    avg_high_low = (df['high'] - df['low']).rolling(window=20).mean()
    volume_intensity = df['volume'] / avg_high_low
    
    # Composite Factor
    composite_factor = intraday_return * intraday_range * volume_intensity
    
    return composite_factor
```
