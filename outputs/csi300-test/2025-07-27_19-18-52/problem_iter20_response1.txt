import pandas as pd

def heuristics_v2(df):
    """
    This function generates a novel and interpretable alpha factor based on the given DataFrame.
    The alpha factor is a combination of momentum, volatility, and trading volume.
    
    Parameters:
    df (pd.DataFrame): A pandas DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    # Momentum: Calculate the 10-day return
    df['momentum'] = df['close'].pct_change(10)
    
    # Volatility: Calculate the 30-day standard deviation of daily returns
    df['daily_return'] = df['close'].pct_change()
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Trading Volume: Calculate the 5-day average volume
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Normalize the factors
    df['momentum_normalized'] = (df['momentum'] - df['momentum'].mean()) / df['momentum'].std()
    df['volatility_normalized'] = (df['volatility'] - df['volatility'].mean()) / df['volatility'].std()
    df['avg_volume_normalized'] = (df['avg_volume'] - df['avg_volume'].mean()) / df['avg_volume'].std()
    
    # Combine the normalized factors to create the final alpha factor
    df['alpha_factor'] = df['momentum_normalized'] + df['volatility_normalized'] + df['avg_volume_normalized']
    
    return df['alpha_factor']
