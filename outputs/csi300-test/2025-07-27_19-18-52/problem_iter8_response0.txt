```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute VWMA of Close Prices
    lookback_period = 20
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / total_volume
    
    # Integrate Intraday Price Change and VWMA
    integrated_intraday_vwma = intraday_price_change * vwma
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Trading Range
    intraday_trading_range = df['high'] - df['low']
    
    # Volume Intensity Score
    avg_high_low = (df['high'] - df['low']).rolling(window=lookback_period).mean()
    volume_intensity_score = df['volume'] / avg_high_low
    
    # Composite Factor
    composite_factor = (intraday_return * intraday_trading_range) * volume_intensity_score + integrated_intraday_vwma
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume and Price Shocks
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_shock = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    avg_close_price = df['close'].rolling(window=lookback_period).mean()
    price_shock = (df['close'] > 1.5 * avg_close_price).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = composite_factor * daily_momentum * volume_shock * price_shock
    
    return final_alpha_factor
```
