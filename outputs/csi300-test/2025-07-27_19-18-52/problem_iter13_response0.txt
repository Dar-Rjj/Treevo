import pandas as pd

def heuristics_v2(df, lookback=20):
    # Calculate Intraday Price Range
    df['Intraday_Price_Range'] = df['High'] - df['Low']
    
    # Compute Volume-Weighted Average Price (VWAP)
    df['Typical_Price'] = (df['High'] + df['Low'] + df['Close']) / 3
    df['VWAP'] = (df['Typical_Price'] * df['Volume']).rolling(window=lookback).sum() / df['Volume'].rolling(window=lookback).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['Integrated_Intraday_Dynamics'] = df['Intraday_Price_Range'] * df['VWAP']
    
    # Calculate Daily Momentum
    df['Daily_Momentum'] = (df['Close'] - df['Close'].shift(lookback)) / df['Close'].shift(lookback)
    
    # Calculate Volume Surge
    df['Median_Volume'] = df['Volume'].rolling(window=lookback).median()
    df['Volume_Surge'] = (df['Volume'] > 2 * df['Median_Volume']).astype(int)
    
    # Calculate Price Breakout
    df['Highest_High'] = df['High'].rolling(window=lookback).max()
    df['Price_Breakout'] = (df['High'] > df['Highest_High']).astype(int)
    
    # Combine Integrated Intraday Dynamics, Momentum, Volume Surge, and Price Breakout
    df['Alpha_Factor'] = (df['Integrated_Intraday_Dynamics'] 
                          * df['Daily_Momentum'] 
                          * df['Volume_Surge'] 
                          * df['Price_Breakout'])
    
    return df['Alpha_Factor']
