```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    close_prices = df['close']
    volume = df['volume']
    vwma = (close_prices * volume).rolling(window=lookback_period).sum() / volume.rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Change and VWMA
    integrated_momentum = intraday_price_change * vwma
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Trading Range
    trading_range = df['high'] - df['low']
    
    # Volume Intensity Score
    avg_high_low_range = (df['high'] - df['low']).rolling(window=lookback_period).mean()
    volume_intensity = volume / avg_high_low_range
    
    # Composite Factor
    composite_factor = (intraday_return * trading_range * volume_intensity) + integrated_momentum
    
    return composite_factor
```
