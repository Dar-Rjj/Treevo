```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP) over a 20-day lookback period
    vwap_prices = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    df['vwap'] = vwap_prices / total_volume
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_intraday_dynamics'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Integrate Intraday and Daily Momentum with VWAP
    df['integrated_momentum'] = (df['intraday_range'] + df['daily_momentum']) * df['vwap']
    
    # Combine Integrated Intraday Dynamics and Momentum
    df['combined_factors'] = df['integrated_intraday_dynamics'] * df['integrated_momentum']
    
    # Calculate Volume and Price Shocks
    average_volume = df['volume'].rolling(window=20).mean()
    average_close_price = df['close'].rolling(window=20).mean()
    
    df['volume_shock'] = (df['volume'] > 1.5 * average_volume).astype(int)
    df['price_shock'] = (df['close'] > 1.5 * average_close_price).astype(int)
    
    # Combine Integrated Factors and Shocks
    df['alpha_factor'] = df['combined_factors'] * df['volume_shock'] * df['price_shock']
    
    return df['alpha_factor']
```
