```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume Intensity Score
    df['volume_intensity'] = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    df['composite_momentum'] = df['intraday_return'] * df['volume_intensity']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['vwap'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Adjust Composite Momentum Factor with VWMA
    df['adjusted_composite_momentum'] = df['composite_momentum'] - df['vwap']
    
    # Analyze price patterns
    short_term_sma_window = 10
    long_term_sma_window = 50
    
    df['short_term_sma'] = df['close'].rolling(window=short_term_sma_window).mean()
    df['long_term_sma'] = df['close'].rolling(window=long_term_sma_window).mean()
    
    # Compare to current close price and short-term SMA
    df['price_trend'] = (df['close'] > df['short_term_sma']).astype(int) + (df['short_term_sma'] > df['long_term_sma']).astype(int)
    
    # Incorporate volume analysis
    df['avg_volume_short_term'] = df['volume'].rolling(window=short_term_sma_window).mean()
    df['volume_trend'] = (df['volume'] > df['avg_volume_short_term']).astype(int)
    
    # Synthesize alpha factor
    df['alpha_factor'] = (df['adjusted_composite_momentum'] + df['price_trend'] + df['volume_trend']) / 3.0
    
    return df['alpha_factor']
```
