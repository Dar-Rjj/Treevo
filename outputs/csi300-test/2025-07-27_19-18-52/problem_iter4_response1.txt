```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    df['close_volume'] = df['close'] * df['volume']
    close_volume_sum = df['close_volume'].rolling(window=lookback_period).sum()
    volume_sum = df['volume'].rolling(window=lookback_period).sum()
    vwma = close_volume_sum / volume_sum
    
    # Integrate Intraday Price Change and VWMA
    factor_values = intraday_price_change * vwma
    
    return factor_values
```
