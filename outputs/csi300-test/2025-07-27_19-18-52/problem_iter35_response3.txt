import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity_score = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    composite_momentum_factor = (intraday_return * intraday_range) * volume_intensity_score
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum_factor = (intraday_return + intraday_range) * volume_intensity_score
    
    # Enhanced Composite Momentum Factor
    enhanced_composite_momentum_factor = (intraday_return + intraday_range) * np.sqrt(volume_intensity_score)
    
    # Integrated Enhanced Composite Momentum Factor with VWMA
    integrated_enhanced_factor = (enhanced_composite_momentum_factor * vwma) - vwma
    
    # Adjusted Composite Momentum Factor Integrated with VWMA
    adjusted_integrated_factor = (adjusted_composite_momentum_factor * vwma) - vwma
    
    return adjusted_integrated_factor
