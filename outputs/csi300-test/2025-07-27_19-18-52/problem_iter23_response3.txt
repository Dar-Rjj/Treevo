```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    trading_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / trading_range
    
    # Composite Momentum Factor
    composite_momentum = intraday_return * volume_intensity
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period_vwma = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period_vwma).sum() / df['volume'].rolling(window=lookback_period_vwma).sum()
    
    # Trend Strength Indicator
    lookback_period_sma = 50
    sma = df['close'].rolling(window=lookback_period_sma).mean()
    trend_strength = sma - vwma
    
    # Weighted Momentum-VWMA Difference
    weighting_factor = 1.0  # This can be tuned based on further analysis
    weighted_diff = (vwma - composite_momentum) * weighting_factor
    
    # Integrated Alpha Factor
    alpha_factor = trend_strength + weighted_diff
    
    return alpha_factor
```
