```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (df['high'] - df['low']).replace(0, np.nan)
    
    # Composite Momentum Factor
    composite_momentum = (intraday_return * intraday_range * volume_intensity).fillna(0)
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['close_vol'] = df['close'] * df['volume']
    df['cum_close_vol'] = df['close_vol'].rolling(window=lookback_period).sum()
    df['cum_volume'] = df['volume'].rolling(window=lookback_period).sum()
    vwma = (df['cum_close_vol'] / df['cum_volume']).fillna(0)
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum = ((intraday_return + intraday_range) * volume_intensity).fillna(0)
    
    # Enhanced Composite Momentum Factor
    enhanced_composite_momentum = ((intraday_return + intraday_range) * volume_intensity.pow(1/3)).fillna(0)
    
    # Integrate Enhanced Composite Momentum Factor with VWMA
    integrated_factor = (enhanced_composite_momentum * vwma) / enhanced_composite_momentum
    
    return integrated_factor
```
