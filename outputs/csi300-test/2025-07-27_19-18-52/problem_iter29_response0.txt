```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Return'] = df['close'] - df['open']
    
    # Intraday Trading Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Volume Intensity Score
    df['Volume_Intensity_Score'] = df['volume'] / df['Intraday_Range']
    
    # Composite Momentum Factor
    df['Composite_Momentum_Factor'] = df['Intraday_Return'] * df['Intraday_Range'] * df['Volume_Intensity_Score']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period_vwma = 20
    df['VWMA'] = (df['close'] * df['volume']).rolling(window=lookback_period_vwma).sum() / df['volume'].rolling(window=lookback_period_vwma).sum()
    
    # Trend Strength Indicator
    lookback_period_sma = 50
    df['SMA'] = df['close'].rolling(window=lookback_period_sma).mean()
    df['Trend_Strength_Indicator'] = df['SMA'] - df['VWMA']
    
    # Adjusted Composite Momentum Factor
    df['Adjusted_Composite_Momentum_Factor'] = df['Composite_Momentum_Factor'] - df['VWMA']
    
    # Weighted Momentum-VWMA Difference
    df['Weighted_Momentum_VWMA_Difference'] = df['Adjusted_Composite_Momentum_Factor'] - df['VWMA']
    
    # Analyze price patterns
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    df['Short_Term_Trend'] = short_term_sma > df['close']
    df['Long_Term_Trend'] = short_term_sma > long_term_sma
    
    # Incorporate volume analysis
    avg_volume_short = df['volume'].rolling(window=10).mean()
    df['Volume_Strength'] = df['volume'] > avg_volume_short
    df['Volume_Spike'] = (df['volume'] / avg_volume_short) > 2.0
    
    # Combine price and volume insights
    df['Price_Volume_Confirmation'] = (df['Short_Term_Trend'] & df['Volume_Strength']) | (df['Long_Term_Trend'] & df['Volume_Strength'])
    
    # Explore advanced price and volume features
    def rsi(series, periods=14):
        delta = series.diff().dropna()
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(periods).mean()
        roll_down = down.abs().rolling(periods).mean()
        rs = roll_up / roll_down
        return 100.0 - (100.0 / (1.0 + rs))
    
    df['RSI'] = rsi(df['close'])
    df['Overbought'] = df['RSI'] > 70
    df['Oversold'] = df['RSI'] < 30
    
    # On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'][i] > df['close'][i-1]:
            obv.append(obv[-1] + df['volume'][i])
        elif df['close'][i] < df['close'][i-1]:
            obv.append(obv[-1] - df['volume'][i])
        else:
            obv.append(obv[-1])
    df['OBV'] = obv
    
    # Synthesize alpha factor
    df['Alpha_Factor'] = (
        0.3 * df['Weighted_Momentum_VWMA_Difference'] +
        0.3 * df['Price_Volume_Confirmation'] +
        0.2 * (df['Overbought'] * -1 + df['Oversold']) +
        0.2 * df['OBV']
    )
    
    return df['Alpha_Factor']
```
