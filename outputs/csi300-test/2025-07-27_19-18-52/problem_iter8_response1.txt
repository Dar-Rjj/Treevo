import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Change and Daily Momentum
    df['Intraday_Price_Change'] = df['high'] - df['low']
    df['Daily_Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute VWMA of Close Prices with a lookback period of 20 days
    df['VWMA'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Price Change, Daily Momentum, and VWMA
    df['Integrated_Factor'] = (df['Intraday_Price_Change'] * df['VWMA']) + df['Daily_Momentum']
    
    # Calculate Volume and Price Shocks
    avg_volume = df['volume'].rolling(window=20).mean()
    df['Volume_Shock'] = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    avg_close_price = df['close'].rolling(window=20).mean()
    df['Price_Shock'] = (df['close'] > 1.5 * avg_close_price).astype(int)
    
    # Combine Integrated Factors and Shocks
    df['Combined_Factors_and_Shocks'] = df['Integrated_Fctor'] * df['Volume_Shock'] * df['Price_Shock']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = df['close'] - df['open']
    
    # Intraday Trading Range
    df['Intraday_Trading_Range'] = df['high'] - df['low']
    
    # Volume Intensity Score
    avg_high_low_range = (df['high'] - df['low']).rolling(window=20).mean()
    df['Volume_Intensity_Score'] = df['volume'] / avg_high_low_range
    
    # Composite Factor
    df['Composite_Factor'] = df['Intraday_Return'] * df['Intraday_Trading_Range'] * df['Volume_Intensity_Score']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factors_and_Shocks'] + df['Composite_Factor']
    
    return df['Final_Alpha_Factor']
