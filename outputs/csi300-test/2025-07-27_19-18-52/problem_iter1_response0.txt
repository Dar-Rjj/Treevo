```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on intraday momentum with volume confirmation.

    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.

    Returns:
    pd.Series: A Series with the alpha factor values indexed by date.
    """
    
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Change and VWMA
    alpha_factor = intraday_price_change * vwma
    
    return alpha_factor
```
