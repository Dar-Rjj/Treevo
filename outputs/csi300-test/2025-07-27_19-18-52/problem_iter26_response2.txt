import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify trend with moving average crossover
    df['50_day_MA'] = df['close'].rolling(window=50).mean()
    df['200_day_MA'] = df['close'].rolling(window=200).mean()
    df['trend'] = (df['50_day_MA'] > df['200_day_MA']).astype(int)
    
    # Compute high-low spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Assess range-to-close ratio
    df['range_to_close'] = df['high_low_spread'] / df['close']
    
    # Measure volume change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Evaluate volume-price relationship
    df['volume_price_relationship'] = ((df['close'] > df['close'].shift(1)) & (df['volume'] > df['volume'].shift(1))).astype(int)
    
    # Identify potential breakout signals
    df['breakout_signal'] = ((df['close'] > df['high'].shift(1)) | (df['close'] < df['low'].shift(1))).astype(int)
    
    # Detect gap-up or gap-down openings
    df['gap_opening'] = ((df['open'] > df['close'].shift(1) * 1.01) | (df['open'] < df['close'].shift(1) * 0.99)).astype(int)
    
    # Calculate simple moving average (SMA) of close prices over a short period
    df['short_sma'] = df['close'].rolling(window=20).mean()
    df['short_sma_trend'] = (df['close'] > df['short_sma']).astype(int)
    
    # Calculate simple moving average (SMA) of close prices over a long period
    df['long_sma'] = df['close'].rolling(window=200).mean()
    df['sma_crossover'] = (df['short_sma'] > df['long_sma']).astype(int)
    
    # Calculate average volume over a short period
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_strength'] = (df['volume'] > df['avg_volume_20']).astype(int)
    
    # Identify volume spikes
    df['volume_spike'] = (df['volume'] > df['avg_volume_20'] * 2).astype(int)
    
    # Calculate relative strength index (RSI) using close prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    df['rsi_signal'] = ((df['rsi'] < 30) | (df['rsi'] > 70)).astype(int)
    
    # Calculate on-balance volume (OBV)
    df['obv'] = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    df['obv_signal'] = (df['obv'] > df['obv'].shift(1)).astype(int)
    
    # Synthesize alpha factor
    # Assign weights to each component based on historical performance
    weights = {
        'daily_return': 0.1,
        'trend': 0.1,
        'range_to_close': 0.05,
        'volume_change': 0.05,
        'volume_price_relationship': 0.1,
        'breakout_signal': 0.1,
        'gap_opening': 0.1,
        'short_sma_trend': 0.1,
        'sma_crossover': 0.1,
        'volume_strength': 0.05,
        'volume_spike': 0.05,
        'rsi_signal': 0.1,
        'obv_signal': 0.1
    }
    
    # Generate a composite alpha factor that combines all signals
    df['alpha_factor'] = sum(df[col] * weight for col, weight in weights.items())
    
    # Return the alpha factor as a pandas Series
    return
