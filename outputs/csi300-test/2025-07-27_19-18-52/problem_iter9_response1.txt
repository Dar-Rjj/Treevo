```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (df['high'] - df['low']).replace(0, 1)  # Avoid division by zero
    
    # Composite Momentum Factor
    composite_momentum = (intraday_return * intraday_range * volume_intensity)
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices with lookback period of 20 days
    vwma = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum = ((intraday_return + intraday_range) * volume_intensity)
    
    # Integrate Adjusted Composite Momentum Factor with VWMA
    alpha_factor = adjusted_composite_momentum * vwma
    
    return alpha_factor
```
