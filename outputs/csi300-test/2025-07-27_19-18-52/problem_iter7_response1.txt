```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute Volume-Weighted Close Price (VWCP)
    vwcp = df['close'] * df['volume']
    
    # Integrate Intraday Price Change and VWCP
    integrated_factor = intraday_price_change * vwcp
    
    return integrated_factor
```
