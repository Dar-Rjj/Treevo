```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP)
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Price Range and VWAP
    intraday_vwap = intraday_range * vwap
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Surge
    median_volume = df['volume'].rolling(window=20).median()
    volume_surge = (df['volume'] > 2 * median_volume).astype(int)
    
    # Calculate Price Breakout
    highest_high = df['high'].rolling(window=20).max()
    price_breakout = (df['high'] > highest_high).astype(int)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=20).std()
    
    # Compute Intraday Momentum
    intraday_price_change = df['high'] - df['low']
    intraday_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute VWMA of Close Prices
    vwma_close = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday and Daily Momentum with VWMA
    integrated_momentum = (intraday_price_change * vwma_close) + daily_momentum
    
    # Calculate Volume and Price Shocks
    average_volume = df['volume'].rolling(window=20).mean()
    volume_shock = (df['volume'] > 1.5 * average_volume).astype(int)
    
    average_close = df['close'].rolling(window=20).mean()
    close_shock = (df['close'] > 1.5 * average_close).astype(int)
    
    average_open = df['open'].rolling(window=20).mean()
    open_shock = (df['open'] > 1.5 * average_open).astype(int)
    
    # Combine All Integrated Factors and Shocks
    alpha_factor = (intraday_vwap * daily_momentum * volume_surge * price_breakout * intraday_volatility * 
                    volume_shock * close_shock * open_shock)
    
    return alpha_factor
```
