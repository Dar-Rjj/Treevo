```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP)
    vwap = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Price Range and VWAP
    integrated_intraday = intraday_range * vwap
    
    # Calculate Daily Momentum
    daily_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Integrate Intraday and Daily Momentum with VWAP
    integrated_momentum = (intraday_momentum + daily_momentum) * vwap
    
    # Combine Integrated Intraday Dynamics and Momentum
    combined_factors = integrated_intraday * integrated_momentum
    
    # Calculate Volume Surge
    median_volume = df['volume'].rolling(window=20).median()
    volume_surge = (df['volume'] > 2 * median_volume).astype(int)
    
    # Calculate Volume and Price Shocks
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_close = df['close'].rolling(window=20).mean()
    
    volume_shock = (df['volume'] > 1.5 * avg_volume).astype(int)
    price_shock = (df['close'] > 1.5 * avg_close).astype(int)
    
    # Combine Integrated Factors, Surges, and Shocks
    alpha_factor = combined_factors * volume_surge * volume_shock * price_shock
    
    return alpha_factor
```
