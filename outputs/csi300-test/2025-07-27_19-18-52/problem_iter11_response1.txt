```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP) over a Lookback Period (e.g., 20 days)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_intraday_vwap'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Median Volume over last 20 days
    df['median_volume'] = df['volume'].rolling(window=20).median()
    
    # Determine if Today's Volume is a Surge
    df['volume_surge'] = (df['volume'] > 2 * df['median_volume']).astype(int)
    
    # Calculate Highest High over last 20 days
    df['highest_high_20'] = df['high'].rolling(window=20).max()
    
    # Determine if Today's High Price is a Breakout
    df['price_breakout'] = (df['high'] > df['highest_high_20']).astype(int)
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Intraday Trading Range
    df['intraday_trading_range'] = df['high'] - df['low']
    
    # Volume Intensity Score
    df['avg_high_low'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['volume_intensity_score'] = df['volume'] / df['avg_high_low']
    
    # Composite Factor
    df['composite_factor'] = (df['intraday_return'] * df['intraday_trading_range'] * df['volume_intensity_score']) + df['integrated_intraday_vwap']
    
    # Calculate Average Volume over last 20 days
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Determine if Today's Volume is a Shock
    df['volume_shock'] = (df['volume'] > 1.5 * df['avg_volume_20']).astype(int)
    
    # Calculate Average Close Price over last 20 days
    df['avg_close_20'] = df['close'].rolling(window=20).mean()
    
    # Determine if Today's Close Price is a Shock
    df['price_shock'] = (df['close'] > 1.5 * df['avg_close_20']).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['composite_factor'] * df['daily_momentum'] * df['volume_surge'] * 
                           df['price_breakout'] * df['volume_shock'] * df['price_shock'])
    
    return df['alpha_factor']
```
