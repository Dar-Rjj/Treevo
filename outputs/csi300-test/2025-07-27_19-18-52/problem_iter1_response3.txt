import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Average Volume over last 20 days
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Determine if Today's Volume is a Shock
    df['Volume_Shock'] = (df['volume'] > 1.5 * df['Avg_Volume_20']).astype(int)
    
    # Combine Momentum and Volume Shock
    df['Momentum_Volume_Shock'] = df['Momentum'] * df['Volume_Shock']
    
    return df['Momentum_Volume_Shock'].dropna()
