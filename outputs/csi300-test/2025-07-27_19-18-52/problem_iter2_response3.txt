```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range
    df['High_Low'] = df['high'] - df['low']
    df['High_Close_Previous'] = abs(df['high'] - df['close'].shift(1))
    df['Low_Close_Previous'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['High_Low', 'High_Close_Previous', 'Low_Close_Previous']].max(axis=1)
    
    # Intraday Return Calculation
    df['Intraday_Return'] = df['close'] - df['open']

    # 10-Day Average True Range
    df['10_Day_ATR'] = df['True_Range'].rolling(window=10).mean()
    
    # Intraday Trading Range
    df['Intraday_Trading_Range'] = df['high'] - df['low']
    
    # Price Momentum
    df['Price_Momentum'] = (df['close'] - df['close'].shift(10)) / 10
    
    # Volume Intensity Score
    df['Volume_Intensity_Score'] = df['volume'] / (df['high'] - df['low']).rolling(window=10).mean()
    
    # Composite Alpha Factor
    df['Inverse_ATR'] = 1 / df['10_Day_ATR']
    df['Momentum_Adjusted'] = df['Price_Momentum'] * df['Inverse_ATR']
    df['Intraday_Adjusted'] = df['Intraday_Return'] * df['Intraday_Trading_Range']
    df['Composite_Alpha_Factor'] = df['Momentum_Adjusted'] * df['Intraday_Adjusted'] * df['Volume_Intensity_Score']
    
    return df['Composite_Alpha_Factor']
```
