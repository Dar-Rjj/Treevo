```python
import pandas as pd

def heuristics_v2(df, lookback=20):
    # Compute Intraday Dynamics
    df['intraday_price_range'] = df['high'] - df['low']
    df['intraday_return'] = df['close'] - df['open']
    
    # Compute Volume-Weighted Average Metrics
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=lookback).sum() / df['volume'].rolling(window=lookback).sum()
    df['vwma_close'] = (df['close'] * df['volume']).rolling(window=lookback).sum() / df['volume'].rolling(window=lookback).sum()
    
    # Integrate Intraday and Daily Momentum with Volume-Weighted Metrics
    df['composite_factor'] = (df['intraday_price_range'] * df['vwap']) + (df['intraday_return'] * df['intraday_price_range'])
    df['volume_intensity_score'] = df['volume'] / (df['high'] - df['low']).rolling(window=lookback).mean()
    df['composite_factor'] *= df['volume_intensity_score']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['composite_factor'] *= df['daily_momentum']
    
    # Determine if Today's Volume is a Shock
    avg_volume = df['volume'].rolling(window=lookback).mean()
    df['volume_shock'] = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    # Determine if Today's Close Price is a Shock
    avg_close = df['close'].rolling(window=lookback).mean()
    df['price_shock'] = (df['close'] > 1.5 * avg_close).astype(int)
    
    # Determine if Today's Volume is a Surge
    median_volume = df['volume'].rolling(window=lookback).median()
    df['volume_surge'] = (df['volume'] > 2 * median_volume).astype(int)
    
    # Determine if Today's High Price is a Breakout
    highest_high = df['high'].rolling(window=lookback).max()
    df['price_breakout'] = (df['high'] > highest_high).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['composite_factor'] * df['volume_shock'] * df['price_shock'] * df['volume_surge'] * df['price_breakout']
    
    return df['alpha_factor']
```
