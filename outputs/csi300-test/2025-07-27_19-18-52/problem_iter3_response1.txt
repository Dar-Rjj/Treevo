```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Change
    intraday_change = df['high'] - df['low']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Compute Amount-Weighted Moving Average (AWMA) of Close Prices
    awma = (df['close'] * df['amount']).rolling(window=lookback_period).sum() / df['amount'].rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Change, VWMA, and AWMA
    alpha_factor = intraday_change * vwma * awma
    
    return alpha_factor
```
