```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (intraday_range + 1e-9)
    
    # Composite Momentum Factor
    composite_momentum = (intraday_return * intraday_range * volume_intensity) + 1e-9
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum = (intraday_return + intraday_range) * volume_intensity
    
    # Enhanced Composite Momentum Factor
    enhanced_composite_momentum = (intraday_return + intraday_range) * np.sqrt(volume_intensity)
    
    # Integrate Enhanced Composite Momentum Factor with VWMA
    integrated_enhanced_momentum = enhanced_composite_momentum * vwma - vwma
    
    # Incorporate Short-Term Price Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    trend_adjusted_momentum = adjusted_composite_momentum - short_term_trend
    
    # Combine Short-Term Trend and VWMA Adjusted Momentum
    combined_momentum = integrated_enhanced_momentum * trend_adjusted_momentum
    
    # Calculate Daily Return
    daily_return = df['close'].diff()
    
    # Integrate Daily Return with Composite Momentum Factor
    weighting_factor = 0.5
    final_factor = combined_momentum + (daily_return * weighting_factor)
    
    return final_factor
```
