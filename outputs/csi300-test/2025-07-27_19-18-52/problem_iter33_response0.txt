```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return: (Close - Open)
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range: (High - Low)
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score: Volume / (High - Low)
    volume_intensity = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    composite_momentum = (intraday_return * intraday_range) * volume_intensity
    
    # Compute VWMA of Close Prices with a 20-day lookback period
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum = (intraday_return + intraday_range) * volume_intensity
    
    # Integrate Adjusted Composite Momentum Factor with VWMA
    integrated_factor = (adjusted_composite_momentum * vwma) - vwma
    
    return integrated_factor
```
