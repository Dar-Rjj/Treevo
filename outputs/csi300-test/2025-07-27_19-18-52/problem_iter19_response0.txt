import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['DailyMomentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Surge
    df['MedianVolume'] = df['volume'].rolling(window=20, center=True).median()
    df['VolumeSurge'] = (df['volume'] > 2 * df['MedianVolume']).astype(int)
    
    # Calculate Price Breakout
    df['HighestHigh'] = df['high'].rolling(window=20, center=False).max()
    df['PriceBreakout'] = (df['high'] > df['HighestHigh']).astype(int)
    
    # Calculate Intraday Price Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['VWAP'] = (typical_price * df['volume']).rolling(window=20, center=False).sum() / df['volume'].rolling(window=20, center=False).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['IntradayDynamics'] = df['IntradayRange'] * df['VWAP']
    
    # Calculate Intraday Price Change
    df['IntradayPriceChange'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Compute Volume-Weighted Close Price (VWCP)
    df['VWCP'] = df['close'] * df['volume'] / df['volume']
    
    # Compute Volume-Weighted Moving Average (VWMA) of High and Low Prices
    df['VWMA_High'] = (df['high'] * df['volume']).rolling(window=20, center=False).sum() / df['volume'].rolling(window=20, center=False).sum()
    df['VWMA_Low'] = (df['low'] * df['volume']).rolling(window=20, center=False).sum() / df['volume'].rolling(window=20, center=False).sum()
    df['VWMA'] = (df['VWMA_High'] + df['VWMA_Low']) / 2
    
    # Integrate All Factors
    df['AlphaFactor'] = (
        df['IntradayDynamics'] * 
        df['DailyMomentum'] * 
        df['VolumeSurge'] * 
        df['PriceBreakout'] * 
        df['IntradayPriceChange'] * 
        df['VWCP'] +
        df['IntradayPriceChange'] * 
        df['VWMA']
    )
    
    return df['AlphaFactor'].dropna()
