```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20, short_term_sma=5, long_term_sma=20):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Trading Range
    intraday_trading_range = (df['high'] - df['low'])
    
    # Volume Intensity Score
    volume_intensity_score = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    composite_momentum_factor = (intraday_return * intraday_trading_range * volume_intensity_score)
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum_factor = (intraday_return + intraday_trading_range) * volume_intensity_score
    
    # Integrate Adjusted Composite Momentum Factor with VWMA
    integrated_factor = adjusted_composite_momentum_factor * vwma - vwma
    
    # Analyze price patterns
    short_term_sma_series = df['close'].rolling(window=short_term_sma).mean()
    long_term_sma_series = df['close'].rolling(window=long_term_sma).mean()
    sma_diff = short_term_sma_series - long_term_sma_series
    price_above_short_sma = df['close'] > short_term_sma_series
    
    # Incorporate volume analysis
    average_volume = df['volume'].rolling(window=short_term_sma).mean()
    volume_signal = df['volume'] / average_volume
    
    # Synthesize alpha factor
    alpha_factor = (
        0.4 * integrated_factor +  # Weight for Integrated Factor
        0.3 * sma_diff +           # Weight for SMA Difference
        0.3 * volume_signal         # Weight for Volume Signal
    )
    
    return alpha_factor
```
