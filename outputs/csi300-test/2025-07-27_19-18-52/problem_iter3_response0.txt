```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Shock
    df['average_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_shock'] = (df['volume'] > 1.5 * df['average_volume_20']).astype(int)
    
    # Calculate Price Shock
    df['average_close_20'] = df['close'].rolling(window=20).mean()
    df['price_shock'] = (df['close'] > 1.5 * df['average_close_20']).astype(int)
    
    # Combine Momentum, Volume Shock, and Price Shock
    df['alpha_factor'] = df['daily_momentum'] * df['volume_shock'] * df['price_shock']
    
    return df['alpha_factor']
```
