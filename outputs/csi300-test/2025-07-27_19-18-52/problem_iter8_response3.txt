import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP) over the lookback period
    price_avg = (df[['open', 'high', 'low', 'close']].mean(axis=1))
    df['vwap'] = (price_avg * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_dynamics'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    
    # Calculate Volume Surge
    df['median_volume'] = df['volume'].rolling(window=lookback_period).median()
    df['volume_surge'] = (df['volume'] > 2 * df['median_volume']).astype(int)
    
    # Calculate Price Breakout
    df['highest_high'] = df['high'].rolling(window=lookback_period).max()
    df['price_breakout'] = (df['high'] > df['highest_high']).astype(int)
    
    # Combine Integrated Intraday Dynamics, Momentum, Volume Surge, and Price Breakout
    df['alpha_factor'] = df['integrated_dynamics'] * df['daily_momentum'] * df['volume_surge'] * df['price_breakout']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
