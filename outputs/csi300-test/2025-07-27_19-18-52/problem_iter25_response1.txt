```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Identify trend with moving average crossover (50-day and 200-day MA)
    df['50_MA'] = df['close'].rolling(window=50).mean()
    df['200_MA'] = df['close'].rolling(window=200).mean()
    df['trend'] = (df['50_MA'] > df['200_MA']).astype(int)
    
    # Compute high-low spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Assess range-to-close ratio
    df['range_to_close_ratio'] = (df['high'] - df['low']) / df['close']
    
    # Measure volume change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Evaluate volume-price relationship
    df['volume_price_relationship'] = ((df['close'] > df['close'].shift(1)) & 
                                      (df['volume'] > df['volume'].shift(1))).astype(int)
    
    # Identify potential breakout signals
    df['breakout_signal'] = ((df['close'] > df['high'].shift(1)) | 
                             (df['close'] < df['low'].shift(1))).astype(int)
    
    # Detect gap-up or gap-down openings
    df['gap_opening'] = ((df['open'] > df['close'].shift(1) * 1.01) | 
                         (df['open'] < df['close'].shift(1) * 0.99)).astype(int)
    
    # Combine all the factors into a single alpha factor
    df['alpha_factor'] = (df['daily_return'] + 
                          df['trend'] + 
                          df['high_low_spread'] + 
                          df['range_to_close_ratio'] + 
                          df['volume_change'] + 
                          df['volume_price_relationship'] + 
                          df['breakout_signal'] + 
                          df['gap_opening'])
    
    # Return the alpha factor
    return df['alpha_factor']
```
