```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)

    # Calculate Volume Surge
    df['VolumeMedian'] = df['volume'].rolling(window=20).median()
    df['VolumeSurge'] = (df['volume'] > 2 * df['VolumeMedian']).astype(int)

    # Calculate Price Breakout
    df['HighestHigh'] = df['high'].rolling(window=20).max()
    df['PriceBreakout'] = (df['high'] > df['HighestHigh']).astype(int)

    # Calculate Intraday Price Change
    df['IntradayChange'] = df['high'] - df['low']
    df['IntradayCloseOpen'] = df['close'] - df['open']

    # Compute Volume-Weighted Close Price (VWCP)
    df['VWCP'] = df['close'] * df['volume']

    # Compute Volume-Weighted Moving Average (VWMA) of High and Low Prices
    df['VWMA_High'] = (df['high'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['VWMA_Low'] = (df['low'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Compute Amount-Weighted Moving Average (AWMA) of High and Low Prices
    df['AWMA_High'] = (df['high'] * df['amount']).rolling(window=20).sum() / df['amount'].rolling(window=20).sum()
    df['AWMA_Low'] = (df['low'] * df['amount']).rolling(window=20).sum() / df['amount'].rolling(window=20).sum()

    # Integrate All Factors
    df['Factor'] = (
        df['Momentum'] * 
        df['VolumeSurge'] * 
        df['PriceBreakout'] * 
        (df['IntradayChange'] + df['IntradayCloseOpen']) * 
        df['VWCP'] + 
        (df['IntradayChange'] + df['IntradayCloseOpen']) * df['VWMA_High'] + 
        (df['IntradayChange'] + df['IntradayCloseOpen']) * df['AWMA_High']
    )

    return df['Factor']
```
