```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Change and VWMA
    integrated_factor = intraday_price_change * vwma
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_trading_range = df['high'] - df['low']
    
    # Volume Intensity Score
    avg_high_low_diff = (df['high'] - df['low']).rolling(window=lookback_period).mean()
    volume_intensity_score = df['volume'] / avg_high_low_diff
    
    # Composite Factor
    composite_factor = (intraday_return * intraday_trading_range * volume_intensity_score) + integrated_factor
    
    return composite_factor
```
