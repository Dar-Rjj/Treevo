```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    df['daily_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute VWMA of Close Prices
    lookback_period = 20
    df['vwma'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Intraday and Daily Momentum with VWMA
    df['integrated_momentum'] = (df['intraday_momentum'] * df['vwma']) + df['daily_momentum']
    
    # Calculate Volume and Price Shocks
    df['avg_volume'] = df['volume'].rolling(window=lookback_period).mean()
    df['volume_shock'] = (df['volume'] > 1.5 * df['avg_volume']).astype(int)
    
    df['avg_close'] = df['close'].rolling(window=lookback_period).mean()
    df['price_shock'] = (df['close'] > 1.5 * df['avg_close']).astype(int)
    
    df['avg_open'] = df['open'].rolling(window=lookback_period).mean()
    df['open_price_shock'] = (df['open'] > 1.5 * df['avg_open']).astype(int)
    
    # Combine Integrated Factors and Shocks
    df['alpha_factor'] = df['integrated_momentum'] * df['volume_shock'] * df['price_shock'] * df['open_price_shock']
    
    return df['alpha_factor']
```
