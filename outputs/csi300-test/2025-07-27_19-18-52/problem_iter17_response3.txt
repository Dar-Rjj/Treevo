```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Intraday Trading Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Volume Intensity Score
    df['volume_intensity'] = df['volume'] / (df['high'] - df['low'])
    
    # Compute Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Composite Momentum Factor
    df['composite_momentum'] = df['intraday_return'] * df['intraday_range'] * df['volume_intensity']
    
    # Adjusted Composite Momentum Factor
    df['adjusted_composite_momentum'] = (df['intraday_return'] + df['intraday_range']) * df['volume_intensity']
    
    # Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(21)
    
    # Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Integrate Intraday and Daily Momentum with VWAP
    df['integrated_momentum'] = (df['intraday_momentum'] + df['daily_momentum']) * df['vwap']
    
    # Combine Integrated Intraday Dynamics and Momentum
    df['combined_momentum'] = df['integrated_momentum'] * df['integrated_intraday_dynamics']
    
    # Calculate Volume Surge
    median_volume = df['volume'].rolling(window=20).median()
    df['volume_surge'] = (df['volume'] > 2 * median_volume).astype(int)
    
    # Calculate Volume and Price Shocks
    avg_volume = df['volume'].rolling(window=20).mean()
    avg_close = df['close'].rolling(window=20).mean()
    df['volume_shock'] = (df['volume'] > 1.5 * avg_volume).astype(int)
    df['price_shock'] = (df['close'] > 1.5 * avg_close).astype(int)
    
    # Introduce Trend Reversal Indicator
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['trend_reversal'] = np.where(df['close'] < df['ma_20'], -1, np.where(df['close'] > df['ma_20'], 1, 0))
    
    # Final Alpha Factor Integration
    df['alpha_factor'] = (
        df['combined_momentum'] * 
        df['volume_surge'] * 
        df['volume_shock'] * 
        df['price_shock'] * 
        df['trend_reversal'] * 
        df['vwap']
    )
    
    return df['alpha_factor']
```
