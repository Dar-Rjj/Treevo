```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (intraday_range + 1e-6)  # Add a small constant to avoid division by zero
    
    # Composite Momentum Factor
    composite_momentum = intraday_return * volume_intensity + 1e-6  # Add a small constant to avoid division by zero
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period_vwma = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period_vwma).sum() / df['volume'].rolling(window=lookback_period_vwma).sum()
    
    # Trend Strength Indicator
    lookback_period_sma = 50
    sma_close = df['close'].rolling(window=lookback_period_sma).mean()
    trend_strength = sma_close - vwma
    
    # Weighted Momentum-VWMA Difference
    momentum_vwma_diff = composite_momentum - vwma
    
    # Integrated Short-Term Price Trend
    ma_5_days = df['close'].rolling(window=5).mean()
    integrated_short_term_trend = composite_momentum - ma_5_days
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Final Integrated Alpha Factor
    final_alpha_factor = (daily_return * 0.5 + composite_momentum) + trend_strength + momentum_vwma_diff + integrated_short_term_trend
    
    return final_alpha_factor
```
