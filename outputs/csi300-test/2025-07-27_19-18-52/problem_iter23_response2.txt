```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate simple moving averages
    short_sma_window = 10
    long_sma_window = 50
    df['SMA_short'] = df['close'].rolling(window=short_sma_window).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_window).mean()
    
    # Compare SMA to current close price
    df['SMA_trend'] = (df['close'] > df['SMA_short']).astype(int)
    
    # Crossover signals
    df['SMA_crossover'] = (df['SMA_short'] > df['SMA_long']).astype(int)
    
    # Volume analysis
    volume_window = 10
    df['avg_volume_short'] = df['volume'].rolling(window=volume_window).mean()
    df['volume_strength'] = (df['volume'] > df['avg_volume_short']).astype(int)
    
    # Volume spikes
    df['volume_spike'] = (df['volume'] > 2 * df['avg_volume_short']).astype(int)
    
    # Relative Strength Index (RSI)
    def rsi(series, period=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['RSI'] = rsi(df['close'])
    df['RSI_signal'] = ((df['RSI'] < 30) | (df['RSI'] > 70)).astype(int)
    
    # On-Balance Volume (OBV)
    def obv(close, volume):
        obv = np.zeros(volume.shape[0])
        obv[0] = 0
        for i in range(1, volume.shape[0]):
            if close[i] > close[i-1]:
                obv[i] = obv[i-1] + volume[i]
            elif close[i] < close[i-1]:
                obv[i] = obv[i-1] - volume[i]
            else:
                obv[i] = obv[i-1]
        return obv
    
    df['OBV'] = obv(df['close'], df['volume'])
    df['OBV_signal'] = (df['OBV'].diff() > 0).astype(int)
    
    # Combine signals
    df['alpha_factor'] = (df['SMA_trend'] * 0.3 + 
                          df['SMA_crossover'] * 0.3 + 
                          df['volume_strength'] * 0.2 + 
                          df['RSI_signal'] * 0.1 + 
                          df['OBV_signal'] * 0.1)
    
    return df['alpha_factor']
```
