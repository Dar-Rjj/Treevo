```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate Intraday Trading Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Volume Intensity Score
    avg_intraday_range = (df['high'] - df['low']).rolling(window=20).mean()
    df['volume_intensity'] = df['volume'] / avg_intraday_range
    
    # Composite Momentum Factor
    df['composite_momentum'] = df['intraday_return'] * df['intraday_range'] * df['volume_intensity']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['vwma'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Composite Momentum Factor with VWMA
    df['alpha_factor'] = df['composite_momentum'] * df['vwma']
    
    return df['alpha_factor']
```
