```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (intraday_range + 1e-9)  # Adding a small constant to avoid division by zero
    
    # Composite Momentum Factor
    composite_momentum = (intraday_return * intraday_range * volume_intensity) + 1e-9  # Avoid division by zero
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Adjust Composite Momentum Factor with VWMA
    adjusted_composite_momentum = composite_momentum - vwma
    
    # Incorporate Short-Term Price Trend
    short_term_trend = df['close'].rolling(window=5).mean()
    trend_adjusted_factor = adjusted_composite_momentum - short_term_trend
    
    # Combine Short-Term Trend and VWMA Adjusted Momentum
    final_factor = trend_adjusted_factor * adjusted_composite_momentum
    
    return final_factor
```
