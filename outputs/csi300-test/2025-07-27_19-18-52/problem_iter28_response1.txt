import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume Intensity Score
    df['volume_intensity'] = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    df['composite_momentum'] = df['intraday_return'] * df['volume_intensity']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_vwma = 20
    df['vwma'] = (df['close'] * df['volume']).rolling(window=lookback_vwma).sum() / df['volume'].rolling(window=lookback_vwma).sum()
    
    # Trend Strength Indicator
    lookback_sma = 50
    df['sma'] = df['close'].rolling(window=lookback_sma).mean()
    df['trend_strength'] = df['sma'] - df['vwma']
    
    # Weighted Momentum-VWMA Difference
    df['weighted_momentum_vwma_diff'] = df['vwma'] - df['composite_momentum']
    
    # Integrated Alpha Factor
    df['integrated_alpha_factor'] = df['trend_strength'] + df['weighted_momentum_vwma_diff']
    
    return df['integrated_alpha_factor']
