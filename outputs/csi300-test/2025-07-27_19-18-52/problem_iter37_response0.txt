```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (df['high'] - df['low'])
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['close_volume'] = df['close'] * df['volume']
    vwma = df['close_volume'].rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Enhanced Composite Momentum Factor
    intraday_trading_range = df['high'] - df['low']
    composite_momentum = (intraday_return + intraday_trading_range) * np.sqrt(volume_intensity)
    
    # Integrated Enhanced Composite Momentum Factor with VWMA
    integrated_factor = composite_momentum * vwma - vwma
    
    return integrated_factor
```
