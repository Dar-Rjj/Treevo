```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (intraday_range + 1e-6)  # Add a small constant to avoid division by zero
    
    # Composite Momentum Factor
    composite_momentum = intraday_return * volume_intensity
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Composite Momentum Factor with VWMA
    composite_with_vwma = composite_momentum - vwma
    
    # Incorporate Short-Term Price Trend
    short_term_ma = df['close'].rolling(window=5).mean()
    composite_with_trend = composite_with_vwma - short_term_ma
    
    # Calculate Daily Return
    daily_return = df['close'].diff()
    
    # Integrate Daily Return with Composite Momentum Factor
    weighting_factor = 0.5
    final_factor = composite_with_trend + (daily_return * weighting_factor)
    
    return final_factor
```
