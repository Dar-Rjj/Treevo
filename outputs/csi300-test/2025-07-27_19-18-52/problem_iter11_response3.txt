import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (typical_price * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_intraday_dynamics'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    
    # Calculate Volume Surge
    median_volume = df['volume'].rolling(window=lookback_period).median()
    df['volume_surge'] = (df['volume'] > 2 * median_volume).astype(int)
    
    # Calculate Price Breakout
    highest_high = df['high'].rolling(window=lookback_period).max()
    df['price_breakout'] = (df['high'] > highest_high).astype(int)
    
    # Calculate Intraday Price Change
    df['intraday_price_change'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Close Price (VWCP)
    df['vwcp'] = (df['close'] * df['volume']) / df['volume']
    
    # Combine Integrated Intraday Dynamics, Momentum, Volume Surge, and Price Breakout
    df['alpha_factor'] = (
        df['integrated_intraday_dynamics'] *
        df['daily_momentum'] *
        df['volume_surge'] *
        df['price_breakout'] *
        df['intraday_price_change'] *
        df['vwcp']
    )
    
    return df['alpha_factor']
