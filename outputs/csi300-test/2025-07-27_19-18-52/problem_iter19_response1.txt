```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (intraday_range + 1e-9)  # Add small constant to avoid division by zero
    
    # Composite Momentum Factor
    composite_momentum = intraday_return * volume_intensity
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Composite Momentum Factor with VWMA
    factor_values = composite_momentum - vwma
    
    return factor_values
```
