```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Intraday Trading Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Volume Intensity Score
    df['volume_intensity'] = df['volume'] / df['intraday_range']
    
    # Composite Momentum Factor
    df['composite_momentum'] = df['intraday_return'] * df['volume_intensity']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['vwma'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Trend Strength Indicator
    sma_lookback_period = 50
    df['sma'] = df['close'].rolling(window=sma_lookback_period).mean()
    df['trend_strength'] = df['sma'] - df['vwma']
    
    # Integrate Composite Momentum Factor, VWMA, and Trend Strength
    df['integrated_momentum'] = df['trend_strength'] + (df['vwma'] - df['composite_momentum'])
    
    # Compute Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['close'] + df['high'] + df['low']) / 3 * df['volume']
    df['vwap'] = df['vwap'].rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_intraday_dynamics'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Integrate Intraday and Daily Momentum with VWAP
    df['integrated_momentum'] = (df['intraday_momentum'] + df['daily_momentum']) * df['vwap']
    
    # Combine Integrated Intraday Dynamics and Momentum
    df['integrated_factors'] = df['integrated_intraday_dynamics'] * df['integrated_momentum']
    
    # Calculate Volume Surge
    median_volume = df['volume'].rolling(window=lookback_period).median()
    df['volume_surge'] = (df['volume'] > 2 * median_volume).astype(int)
    
    # Determine if Today's Volume is a Shock
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    df['volume_shock'] = (df['volume'] > 1.5 * avg_volume).astype(int)
    
    # Determine if Today's Close Price is a Shock
    avg_close = df['close'].rolling(window=lookback_period).mean()
    df['price_shock'] = (df['close'] > 1.5 * avg_close).astype(int)
    
    # Introduce Trend Reversal Indicator
    ma_20 = df['close'].rolling(window=lookback_period).mean()
    df['trend_reversal'] = np.where(df['close'] < ma_20, -1, np.where(df['close'] > ma_20, 1, 0))
    
    # Calculate Price Breakout
    highest_high = df['high'].rolling(window=lookback_period).max()
    df['price_breakout'] = (df['high'] > highest_high).astype(int)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_range'].rolling(window=lookback_period).std()
    
    # Combine All Integrated Factors and Shocks
    df['alpha_factor'] = (
        df['integrated_factors'] * 
        df['volume_surge'] * 
        df['volume_shock'] * 
        df['price_shock'] * 
        df['trend_reversal'] * 
        df['price_breakout'] * 
        df['intraday_volatility']
    )
    
    return df['alpha_factor']
```
