```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    composite_momentum = (intraday_return * intraday_range) * volume_intensity
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['vwap'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Trend Strength Indicator
    sma_lookback_period = 50
    df['sma_close'] = df['close'].rolling(window=sma_lookback_period).mean()
    trend_strength = df['sma_close'] - df['vwap']
    
    # Weighted Momentum-VWMA Difference
    weighted_momentum_vwma_diff = composite_momentum - df['vwap']
    
    # Integrated Short-Term Price Trend
    short_term_ma = df['close'].rolling(window=5).mean()
    integrated_short_term_trend = composite_momentum - short_term_ma
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum = (intraday_return + intraday_range) * volume_intensity
    
    # Integrate Adjusted Composite Momentum Factor with VWMA
    integrated_adjusted_cm = (adjusted_composite_momentum * df['vwap']) - df['vwap']
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Final Integrated Alpha Factor
    final_alpha_factor = (daily_return * 0.5 + composite_momentum) + (trend_strength + weighted_momentum_vwma_diff) + (integrated_short_term_trend + integrated_adjusted_cm)
    
    return final_alpha_factor
```
