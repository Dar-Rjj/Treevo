```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP) over a lookback period (e.g., 20 days)
    def vwap(df, lookback=20):
        typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
        vwap = (typical_price * df['volume']).rolling(window=lookback).sum() / df['volume'].rolling(window=lookback).sum()
        return vwap
    df['vwap'] = vwap(df)
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_intraday_dynamics'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Surge
    df['median_volume'] = df['volume'].rolling(window=20).median()
    df['volume_surge'] = (df['volume'] > 2 * df['median_volume']).astype(int)
    
    # Calculate Price Breakout
    df['highest_high_20'] = df['high'].rolling(window=20).max()
    df['price_breakout'] = (df['high'] > df['highest_high_20']).astype(int)
    
    # Combine Integrated Intraday Dynamics, Momentum, Volume Surge, and Price Breakout
    df['alpha_factor'] = (df['integrated_intraday_dynamics'] 
                          * df['momentum'] 
                          * df['volume_surge'] 
                          * df['price_breakout'])
    
    return df['alpha_factor']
```
