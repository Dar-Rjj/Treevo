```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Volume-Weighted Close Price (VWCP)
    vwcp = (df['close'] * df['volume']) / df['volume'].sum()
    
    # Integrated Alpha Factor
    integrated_alpha_factor = intraday_price_change * vwcp
    
    return integrated_alpha_factor
```
