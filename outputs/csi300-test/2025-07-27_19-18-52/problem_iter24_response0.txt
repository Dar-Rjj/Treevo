```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    trading_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / (trading_range + 1e-6)  # Adding a small constant to avoid division by zero
    
    # Composite Momentum Factor
    composite_momentum = intraday_return * volume_intensity + 1e-6  # Adding a small constant to avoid division by zero
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Composite Momentum Factor with VWMA
    composite_momentum_vwma = composite_momentum - vwma
    
    # Incorporate Short-Term Price Trend
    short_term_ma = df['close'].rolling(window=5).mean()
    composite_momentum_trend = composite_momentum_vwma - short_term_ma
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Integrate Daily Return with Composite Momentum Factor
    final_factor = composite_momentum_trend + daily_return
    
    return final_factor
```
