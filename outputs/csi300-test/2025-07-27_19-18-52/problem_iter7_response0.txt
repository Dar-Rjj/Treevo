```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return: (Close - Open)
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range: (High - Low)
    intraday_trading_range = df['high'] - df['low']
    
    # Volume Intensity Score: Volume / (High - Low)
    volume_intensity_score = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor: (Intraday Return * Intraday Trading Range * Volume Intensity Score)
    composite_momentum_factor = intraday_return * intraday_trading_range * volume_intensity_score
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['vwap'] = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Integrate Composite Momentum Factor with VWMA
    factor_values = composite_momentum_factor * df['vwap']
    
    return factor_values
```
