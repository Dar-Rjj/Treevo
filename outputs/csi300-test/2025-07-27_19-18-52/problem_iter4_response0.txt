```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute Volume Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    typical_price = df['close'] * df['volume']
    total_volume = df['volume'].rolling(window=lookback_period).sum()
    vwma = (typical_price.rolling(window=lookback_period).sum()) / total_volume
    
    # Integrate Intraday Price Change and VWMA
    alpha_factor = intraday_price_change * vwma
    
    return alpha_factor
```
