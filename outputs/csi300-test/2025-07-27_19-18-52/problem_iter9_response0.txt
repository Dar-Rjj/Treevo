```python
import pandas as pd

def heuristics_v2(data, lookback_period=20):
    # Calculate Intraday Price Change
    intraday_price_change = data['close'] - data['open']

    # Compute Volume-Weighted Moving Average (VWMA) of High and Low Prices
    high_low_volume_weighted = (data['high'] + data['low']) * data['volume']
    volume_sum = data['volume'].rolling(window=lookback_period).sum()
    vwma = high_low_volume_weighted.rolling(window=lookback_period).sum() / volume_sum

    # Compute Amount-Weighted Moving Average (AWMA) of High and Low Prices
    high_low_amount_weighted = (data['high'] + data['low']) * data['amount']
    amount_sum = data['amount'].rolling(window=lookback_period).sum()
    awma = high_low_amount_weighted.rolling(window=lookback_period).sum() / amount_sum

    # Integrate Intraday Price Change, VWMA, and AWMA
    factor_value = (intraday_price_change * vwma) + (intraday_price_change * awma)
    
    return factor_value
```
