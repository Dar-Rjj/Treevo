```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Intraday Trading Range
    intraday_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity = df['volume'] / intraday_range
    
    # Composite Momentum Factor
    composite_momentum = intraday_return * volume_intensity
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices with a 20-day lookback
    lookback_period_vwma = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period_vwma).sum() / df['volume'].rolling(window=lookback_period_vwma).sum()
    
    # Trend Strength Indicator using a 50-day lookback for SMA
    lookback_period_sma = 50
    sma = df['close'].rolling(window=lookback_period_sma).mean()
    trend_strength = sma - vwma
    
    # Integrate Composite Momentum Factor, VWMA, and Trend Strength
    alpha_factor = trend_strength + (vwma - composite_momentum)
    
    return alpha_factor
```
