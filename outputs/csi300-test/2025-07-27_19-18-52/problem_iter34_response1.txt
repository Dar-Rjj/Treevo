```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Intraday Trading Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Volume Intensity Score
    df['volume_intensity'] = df['volume'] / df['intraday_range']
    
    # Composite Momentum Factor
    df['composite_momentum'] = df['intraday_return'] * df['volume_intensity']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period_vwma = 20
    df['pv'] = df['close'] * df['volume']
    df['cum_pv'] = df['pv'].rolling(window=lookback_period_vwma).sum()
    df['cum_vol'] = df['volume'].rolling(window=lookback_period_vwma).sum()
    df['vwma'] = df['cum_pv'] / df['cum_vol']
    
    # Trend Strength Indicator
    lookback_period_sma = 50
    df['sma'] = df['close'].rolling(window=lookback_period_sma).mean()
    df['trend_strength'] = df['sma'] - df['vwma']
    
    # Integrated Crossover Momentum and Trend
    df['integrated_crossover'] = (df['vwma'] - df['composite_momentum']) + df['trend_strength']
    
    # Combine Integrated Crossover Dynamics and Momentum
    df['alpha_factor'] = df['integrated_crossover'] * (df['vwma'] - df['composite_momentum'])
    
    return df['alpha_factor']
```
