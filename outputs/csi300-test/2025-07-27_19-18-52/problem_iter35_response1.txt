import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Intraday Trading Range
    intraday_range = df['high'] - df['low']

    # Volume Intensity Score
    volume_intensity = df['volume'] / (df['high'] - df['low'])

    # Composite Momentum Factor
    composite_momentum = (intraday_return * intraday_range) * volume_intensity

    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    vwma = (df['close'] * df['volume']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()

    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum = (intraday_return + intraday_range) * volume_intensity

    # Enhanced Composite Momentum Factor
    enhanced_composite_momentum = (intraday_return + intraday_range) * np.sqrt(volume_intensity)

    # Integrated Enhanced Composite Momentum Factor with VWMA
    integrated_enhanced_factor = (enhanced_composite_momentum * vwma) - vwma

    # Integrate Adjusted Composite Momentum Factor with VWMA
    integrated_adjusted_factor = (adjusted_composite_momentum * vwma) - vwma

    # Choose the final alpha factor (here we choose the integrated enhanced factor)
    alpha_factor = integrated_enhanced_factor
    
    return alpha_factor
