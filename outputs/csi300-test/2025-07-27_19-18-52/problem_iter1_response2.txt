import pandas as pd

def heuristics_v2(df):
    # Calculate the True Range
    df['High-Low'] = df['high'] - df['low']
    df['High-Close_Prev'] = abs(df['high'] - df['close'].shift(1))
    df['Low-Close_Prev'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['High-Low', 'High-Close_Prev', 'Low-Close_Prev']].max(axis=1)
    
    # Compute 10-Day Average True Range
    df['10D_ATR'] = df['True_Range'].rolling(window=10).mean()
    
    # Calculate Price Momentum
    df['Price_Momentum'] = (df['close'] - df['close'].shift(10)) / 10
    
    # Multiply Momentum by Inverse of ATR
    df['ATR_Momentum_Factor'] = df['Price_Momentum'] * (1 / df['10D_ATR'])
    
    # Return the factor values as a pandas Series
    return df['ATR_Momentum_Factor'].dropna()
