```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Intraday Trading Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Volume Intensity Score
    df['volume_intensity'] = df['volume'] / df['intraday_range']
    
    # Composite Momentum Factor
    df['composite_momentum'] = df['intraday_return'] * df['volume_intensity']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    lookback_period = 20
    df['vwap'] = (df['volume'] * (df['open'] + df['high'] + df['low'] + df['close']) / 4).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    df['vwma'] = (df['volume'] * df['close']).rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Trend Strength Indicator
    sma_period = 50
    df['sma'] = df['close'].rolling(window=sma_period).mean()
    df['trend_strength'] = df['sma'] - df['vwma']
    
    # Integrate Composite Momentum Factor, VWMA, and Trend Strength
    df['integrated_momentum'] = df['trend_strength'] + (df['vwma'] - df['composite_momentum'])
    
    # Calculate Intraday Price Range
    df['intraday_price_range'] = df['high'] - df['low']
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_intraday_dynamics'] = df['intraday_price_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Integrate Intraday and Daily Momentum with VWAP
    df['integrated_momentum_vwap'] = (df['intraday_momentum'] + df['daily_momentum']) * df['vwap']
    
    # Combine Integrated Intraday Dynamics and Momentum
    df['combined_integrated_factors'] = df['integrated_intraday_dynamics'] * df['integrated_momentum_vwap']
    
    # Calculate Volume Surge
    df['volume_surge'] = (df['volume'] > 2 * df['volume'].median()).astype(int)
    
    # Determine if Today's Volume is a Shock
    df['volume_shock'] = (df['volume'] > 1.5 * df['volume'].rolling(window=lookback_period).mean()).astype(int)
    
    # Determine if Today's Close Price is a Shock
    df['price_shock'] = (df['close'] > 1.5 * df['close'].rolling(window=lookback_period).mean()).astype(int)
    
    # Introduce Trend Reversal Indicator
    df['trend_reversal'] = np.where(df['close'] < df['close'].rolling(window=lookback_period).mean(), -1, 1)
    
    # Calculate Price Breakout
    df['price_breakout'] = (df['high'] > df['high'].rolling(window=lookback_period).max()).astype(int)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_range'].rolling(window=lookback_period).std()
    
    # Introduce Price Efficiency
    df['price_efficiency'] = (df['close'] - df['open']) / df['intraday_range']
    
    # Combine All Integrated Factors and Shocks
    df['alpha_factor'] = (
        df['combined_integrated_factors'] * 
        df['volume_surge'] * 
        df['volume_shock'] * 
        df['price_shock'] * 
        df['trend_reversal'] * 
        df['price_breakout'] * 
        df['intraday_volatility'] * 
        df['price_efficiency']
    )
    
    return df['alpha_factor']
```
