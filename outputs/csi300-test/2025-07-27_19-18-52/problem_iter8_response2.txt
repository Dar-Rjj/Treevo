```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Surge
    df['MedianVolume'] = df['volume'].rolling(window=20, min_periods=1).median()
    df['VolumeSurge'] = (df['volume'] > 2 * df['MedianVolume']).astype(int)
    
    # Calculate Price Breakout
    df['HighestHigh'] = df['high'].rolling(window=20, min_periods=1).max()
    df['PriceBreakout'] = (df['high'] > df['HighestHigh']).astype(int)
    
    # Calculate Intraday Price Change
    df['IntradayChange'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Close Price (VWCP)
    df['VWCP'] = df['close'] * df['volume']
    
    # Combine factors
    df['Factor'] = (df['Momentum'] * 
                    df['VolumeSurge'] * 
                    df['PriceBreakout'] * 
                    df['IntradayChange'] * 
                    df['VWCP'])
    
    return df['Factor']
```
