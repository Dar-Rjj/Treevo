```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Surge
    df['MedianVolume'] = df['volume'].rolling(window=20).median()
    df['VolumeSurge'] = (df['volume'] > 2 * df['MedianVolume']).astype(int)
    
    # Calculate Price Breakout
    df['HighestHigh'] = df['high'].rolling(window=20).max()
    df['PriceBreakout'] = (df['high'] > df['HighestHigh']).astype(int)
    
    # Calculate Intraday Return
    df['IntradayReturn'] = df['close'] - df['open']
    
    # Calculate Intraday Trading Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Volume Intensity Score
    df['VolumeIntensityScore'] = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    df['CompositeMomentumFactor'] = df['IntradayReturn'] * df['IntradayRange'] * df['VolumeIntensityScore']
    
    # Compute VWAP
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['VWAP'] = (typical_price * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Dynamics with VWAP
    df['IntradayDynamics'] = df['IntradayRange'] * df['VWAP']
    
    # Compute VWMA of Close Prices
    df['VWMA'] = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjusted Composite Momentum Factor
    df['AdjustedCompositeMomentumFactor'] = (df['IntradayReturn'] + df['IntradayRange']) * df['VolumeIntensityScore']
    
    # Integrated Momentum and VWMA Factor
    df['IntegratedMomentumVWMAFactor'] = df['AdjustedCompositeMomentumFactor'] - df['VWMA']
    
    # Enhance Integrated Factor with Volume Shock
    df['VolumeShock'] = df['volume'] - df['volume'].shift(1)
    df['EnhancedIntegratedFactor'] = df['IntegratedMomentumVWMAFactor'] + df['VolumeShock']
    
    # Integrate All Factors
    integrated_factor = (
        df['IntradayDynamics'] * 
        df['Momentum'] * 
        df['VolumeSurge'] * 
        df['PriceBreakout'] * 
        df['IntradayReturn'] * 
        df['VWAP'] + 
        df['IntradayReturn'] * df['VWMA'] + 
        df['EnhancedIntegratedFactor']
    )
    
    return integrated_factor
```
