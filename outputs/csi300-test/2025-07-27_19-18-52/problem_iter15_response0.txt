```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP) over a Lookback Period (e.g., 20 days)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_intraday'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['daily_momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Integrate Intraday and Daily Momentum with VWAP
    df['integrated_momentum'] = (df['intraday_momentum'] + df['daily_momentum']) * df['vwap']
    
    # Combine Integrated Intraday Dynamics and Momentum
    df['combined_dynamics_momentum'] = df['integrated_intraday'] * df['integrated_momentum']
    
    # Calculate Volume Surge
    df['median_volume'] = df['volume'].rolling(window=20).median()
    df['volume_surge'] = (df['volume'] > 2 * df['median_volume']).astype(int)
    
    # Calculate Volume and Price Shocks
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_shock'] = (df['volume'] > 1.5 * df['avg_volume']).astype(int)
    
    df['avg_close'] = df['close'].rolling(window=20).mean()
    df['price_shock'] = (df['close'] > 1.5 * df['avg_close']).astype(int)
    
    # Introduce Trend Reversal Indicator
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['trend_reversal'] = (df['close'] > df['ma_20']).astype(int) * 2 - 1
    
    # Combine Integrated Factors, Surges, Shocks, and Trend Reversal
    df['alpha_factor'] = df['combined_dynamics_momentum'] * df['volume_surge'] * df['volume_shock'] * df['price_shock'] * df['trend_reversal']
    
    return df['alpha_factor']
```
