```python
import pandas as pd

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Price Change
    intraday_price_change = df['high'] - df['low']
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices
    close_prices = df['close']
    volumes = df['volume']
    vwma = (close_prices * volumes).rolling(window=lookback_period).sum() / volumes.rolling(window=lookback_period).sum()
    
    # Compute Amount-Weighted Moving Average (AWMA) of Close Prices
    amounts = df['amount']
    awma = (close_prices * amounts).rolling(window=lookback_period).sum() / amounts.rolling(window=lookback_period).sum()
    
    # Integrate Intraday Price Change, VWMA, and AWMA
    alpha_factor = (intraday_price_change * vwma) + (intraday_price_change * awma)
    
    return alpha_factor
```
