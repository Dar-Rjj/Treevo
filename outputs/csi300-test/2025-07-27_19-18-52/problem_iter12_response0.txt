```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP)
    vwap_prices = (df[['open', 'high', 'low', 'close']].sum(axis=1) / 4)
    df['vwap'] = (vwap_prices * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['integrated_dynamics'] = df['intraday_range'] * df['vwap']
    
    # Calculate Daily Momentum
    df['momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Surge
    df['median_volume'] = df['volume'].rolling(window=20).median()
    df['volume_surge'] = (df['volume'] > 2 * df['median_volume']).astype(int)
    
    # Calculate Price Breakout
    df['highest_high'] = df['high'].rolling(window=20).max()
    df['price_breakout'] = (df['high'] > df['highest_high']).astype(int)
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=20).std()
    
    # Combine factors
    df['alpha_factor'] = (
        df['integrated_dynamics'] * 
        df['momentum'] * 
        df['volume_surge'] * 
        df['price_breakout'] * 
        df['intraday_volatility']
    )
    
    return df['alpha_factor']
```
