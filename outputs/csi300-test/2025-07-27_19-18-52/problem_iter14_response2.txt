```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Volume Surge
    df['MedianVolume'] = df['volume'].rolling(window=20).median()
    df['VolumeSurge'] = np.where(df['volume'] > 2 * df['MedianVolume'], 1, 0)
    
    # Calculate Price Breakout
    df['HighestHigh'] = df['high'].rolling(window=20).max()
    df['PriceBreakout'] = np.where(df['high'] > df['HighestHigh'], 1, 0)
    
    # Calculate Intraday Price Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Compute Volume-Weighted Average Price (VWAP)
    df['TypicalPrice'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['VWAP'] = (df['TypicalPrice'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Integrate Intraday Price Range and VWAP
    df['IntegratedIntradayDynamics'] = df['IntradayRange'] * df['VWAP']
    
    # Calculate Intraday Price Change
    df['IntradayPriceChange'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Compute Volume-Weighted Close Price (VWCP)
    df['VWCP'] = df['close'] * df['volume']
    
    # Compute Volume-Weighted Moving Average (VWMA) of High and Low Prices
    df['VWMAHigh'] = (df['high'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['VWMALow'] = (df['low'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    df['VWMA'] = (df['VWMAHigh'] + df['VWMALow']) / 2
    
    # Compute Amount-Weighted Moving Average (AWMA) of High and Low Prices
    df['AWMAHigh'] = (df['high'] * df['amount']).rolling(window=20).sum() / df['amount'].rolling(window=20).sum()
    df['AWMALow'] = (df['low'] * df['amount']).rolling(window=20).sum() / df['amount'].rolling(window=20).sum()
    df['AWMA'] = (df['AWMAHigh'] + df['AWMALow']) / 2
    
    # Integrate All Factors
    df['AlphaFactor'] = (
        df['IntegratedIntradayDynamics'] * 
        df['Momentum'] * 
        df['VolumeSurge'] * 
        df['PriceBreakout'] * 
        df['IntradayPriceChange'] * 
        df['VWCP'] + 
        df['IntradayPriceChange'] * df['VWMA'] + 
        df['IntradayPriceChange'] * df['AWMA']
    )
    
    return df['AlphaFactor']
```
