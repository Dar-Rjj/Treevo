import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Intraday Trading Range
    intraday_trading_range = df['high'] - df['low']
    
    # Volume Intensity Score
    volume_intensity_score = df['volume'] / (df['high'] - df['low'])
    
    # Composite Momentum Factor
    composite_momentum_factor = (intraday_return * intraday_trading_range * volume_intensity_score)
    
    # Compute Volume-Weighted Moving Average (VWMA) of Close Prices with a 20-day lookback period
    lookback_period = 20
    df['vwmultiplier'] = df['volume'] * df['close']
    vwmavol = df['volume'].rolling(window=lookback_period).sum()
    vwma = (df['vwmultiplier'].rolling(window=lookback_period).sum()) / vwmavol
    
    # Adjusted Composite Momentum Factor
    adjusted_composite_momentum_factor = (intraday_return + intraday_trading_range) * volume_intensity_score
    
    # Enhanced Composite Momentum Factor
    enhanced_composite_momentum_factor = (intraday_return + intraday_trading_range) * np.sqrt(volume_intensity_score)
    
    # Integrate Enhanced Composite Momentum Factor with VWMA
    integrated_factor = enhanced_composite_momentum_factor * vwma - vwma
    
    return pd.Series(integrated_factor, index=df.index)
