def heuristics_v2(df):
    df['avg_price'] = (df['high'] + df['low']) / 2
    df['price_diff'] = df['close'] - df['avg_price'].rolling(window=10).mean()
    df['sum_volume'] = df['volume'].rolling(window=10).sum()
    heuristics_matrix = (df['price_diff'] / df['sum_volume']).fillna(0)
    return heuristics_matrix
