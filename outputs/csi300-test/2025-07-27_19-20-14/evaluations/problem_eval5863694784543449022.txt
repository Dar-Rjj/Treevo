def heuristics_v2(df):
    df['returns_20'] = df['close'].pct_change(periods=20)
    df['volume_ma_60'] = df['volume'].rolling(window=60).mean()
    df['vol_ratio'] = df['volume'] / df['volume_ma_60']
    df['z_score'] = (df['returns_20'] - df['returns_20'].rolling(window=60).mean()) / df['returns_20'].rolling(window=60).std()
    heuristics_matrix = df['z_score'] * df['vol_ratio']
    return heuristics_matrix
