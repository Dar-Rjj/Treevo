def heuristics_v2(df):
    df['avg_price'] = (df['high'] + df['low']) / 2
    df['price_diff'] = df['close'] - df['avg_price'].rolling(window=30).mean()
    df['std_volume'] = df['volume'].rolling(window=30).std()
    heuristics_matrix = (df['price_diff'] / df['std_volume']).fillna(0)
    return heuristics_matrix
