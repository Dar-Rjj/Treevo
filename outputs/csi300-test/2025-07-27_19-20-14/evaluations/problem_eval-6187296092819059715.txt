def heuristics_v2(df):
    df['mean_price'] = (df['open'] + df['close']) / 2
    df['price_diff'] = df['close'] - df['mean_price'].rolling(window=15).mean()
    df['max_volume'] = df['volume'].rolling(window=15).max()
    heuristics_matrix = (df['price_diff'] / df['max_volume']).fillna(0)
    return heuristics_matrix
