def heuristics_v2(df):
    df['avg_hl'] = (df['high'] + df['low']) / 2
    df['close_avg_hl_ratio'] = df['close'] / df['avg_hl']
    df['ma_20_close_avg_hl_ratio'] = df['close_avg_hl_ratio'].rolling(window=20).mean()
    heuristics_matrix = df['ma_20_close_avg_hl_ratio']
    return heuristics_matrix
