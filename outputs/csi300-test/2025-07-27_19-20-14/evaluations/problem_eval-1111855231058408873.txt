def heuristics_v2(df):
    df['gain'] = df['close'].diff().clip(lower=0)
    df['loss'] = -df['close'].diff().clip(upper=0)
    df['avg_gain'] = df['gain'].rolling(window=14).mean()
    df['avg_loss'] = df['loss'].rolling(window=14).mean()
    df['rs'] = df['avg_gain'] / df['avg_loss']
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    df['volume_change'] = df['volume'].pct_change(periods=20)
    df['true_range_10'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['true_range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    heuristics_matrix = (df['rsi'] * df['volume_change'] * (df['true_range_10'] / df['true_range_20'])).fillna(0)
    return heuristics_matrix
