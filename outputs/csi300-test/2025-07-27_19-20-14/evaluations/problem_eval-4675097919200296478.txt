def heuristics_v2(df):
    df['weighted_avg_price'] = (df['high'] * 0.6 + df['low'] * 0.4)
    df['price_diff'] = df['close'] - df['weighted_avg_price'].rolling(window=10).mean()
    df['max_volume'] = df['volume'].rolling(window=10).max()
    heuristics_matrix = (df['price_diff'] / df['max_volume']).fillna(0)
    return heuristics_matrix
