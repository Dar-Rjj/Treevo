def heuristics_v2(df):
    df['TR_14'] = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    df['ATR_14'] = df['TR_14'].ewm(span=14, adjust=False).mean()
    df['volume_roc_20'] = df['volume'].pct_change(periods=20)
    heuristics_matrix = df['volume_roc_20'] * df['ATR_14']
    return heuristics_matrix
