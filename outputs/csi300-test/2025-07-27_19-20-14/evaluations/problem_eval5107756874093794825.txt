def heuristics_v2(df):
    df['RSI_30'] = 100 - (100 / (1 + (df['close'].diff().fillna(0).clip(lower=0).rolling(window=30).mean() / 
                                     df['close'].diff().fillna(0).clip(upper=0).abs().rolling(window=30).mean())))
    df['std_10'] = df['close'].rolling(window=10).std()
    heuristics_matrix = df['RSI_30'] * df['std_10']
    return heuristics_matrix
