def heuristics_v2(df):
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR_14_EMA'] = df['TR'].ewm(span=14, adjust=False).mean()
    df['volume_roc_20'] = df['volume'].pct_change(periods=20)
    heuristics_matrix = df['volume_roc_20'] * df['ATR_14_EMA']
    return heuristics_matrix
