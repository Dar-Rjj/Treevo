def heuristics_v2(df):
    df['avg_open_close'] = (df['open'] + df['close']) / 2
    df['price_diff'] = df['close'] - df['avg_open_close'].rolling(window=20).mean()
    df['std_volume'] = df['volume'].rolling(window=20).std()
    heuristics_matrix = (df['price_diff'] / df['std_volume']).fillna(0)
    return heuristics_matrix
