def heuristics_v2(df):
    df['high_low_diff'] = df['high'].rolling(window=30).max() - df['low'].rolling(window=30).min()
    df['avg_amount_30'] = df['amount'].rolling(window=30).mean()
    df['close_pct_change'] = df['close'].pct_change()
    heuristics_matrix = (df['high_low_diff'] / df['avg_amount_30']) * df['close_pct_change'].fillna(0)
    return heuristics_matrix
