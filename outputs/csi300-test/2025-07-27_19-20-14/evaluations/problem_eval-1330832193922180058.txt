def heuristics_v2(df):
    df['weighted_avg_price'] = 0.6 * df['high'] + 0.4 * df['low']
    df['price_diff'] = df['close'] - df['weighted_avg_price'].rolling(window=30).mean()
    df['ema_volume'] = df['volume'].ewm(span=15, adjust=False).mean()
    heuristics_matrix = (df['price_diff'] / df['ema_volume']).fillna(0)
    return heuristics_matrix
