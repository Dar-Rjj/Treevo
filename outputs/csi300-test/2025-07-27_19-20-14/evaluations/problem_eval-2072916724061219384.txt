def heuristics_v2(df):
    df['delta'] = df['close'].diff()
    gain = df['delta'].clip(lower=0)
    loss = -df['delta'].clip(upper=0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = np.abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = np.abs(df['low'] - df['close'].shift(1))
    tr = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    atr_20 = tr.rolling(window=20).mean()
    heuristics_matrix = rsi * atr_20
    return heuristics_matrix
