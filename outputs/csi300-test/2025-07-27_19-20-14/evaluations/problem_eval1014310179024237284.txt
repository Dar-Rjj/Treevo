def heuristics_v2(df):
    df['close_mom_100'] = df['close'].pct_change(periods=100)
    df['volume_mean_60'] = df['volume'].rolling(window=60).mean()
    df['volume_std_60'] = df['volume'].rolling(window=60).std()
    df['volume_cv_60'] = df['volume_std_60'] / df['volume_mean_60']
    heuristics_matrix = df['close_mom_100'] / df['volume_cv_60']
    return heuristics_matrix
