def heuristics_v2(df):
    df['open_roc_10'] = df['open'].pct_change(periods=10)
    df['ma50_close'] = df['close'].rolling(window=50).mean()
    df['ma20_close'] = df['close'].rolling(window=20).mean()
    df['ma_diff'] = df['ma50_close'] - df['ma20_close']
    heuristics_matrix = df['open_roc_10'] * df['ma_diff']
    return heuristics_matrix
