def heuristics_v2(df):
    df['ATR_14'] = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    df['volume_ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['atr_roc_14'] = df['ATR_14'].pct_change(periods=1)
    heuristics_matrix = df['volume_ema_20'] * df['atr_roc_14']
    return heuristics_matrix
