def heuristics_v2(df):
    df['close_sma_10'] = df['close'].rolling(window=10).mean()
    df['daily_return'] = df['close'].pct_change()
    df['return_cv_50'] = df['daily_return'].rolling(window=50).std() / df['daily_return'].rolling(window=50).mean()
    heuristics_matrix = df['close_sma_10'] * df['return_cv_50']
    return heuristics_matrix
