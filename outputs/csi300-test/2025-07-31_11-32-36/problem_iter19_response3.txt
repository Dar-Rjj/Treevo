```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=5, M=5):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Average Intraday Range
    avg_intraday_range = intraday_range.rolling(window=N).mean()
    
    # Compute Close Price Returns
    close_returns = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = close_returns.rolling(window=M).sum()
    
    # Combine Intraday Range and Momentum
    combined_intraday_momentum = avg_intraday_range * cumulative_return
    
    # Adjust for Volume
    ema_volume = df['volume'].ewm(span=N, adjust=False).mean()
    adjusted_factor = combined_intraday_momentum / ema_volume
    
    # Calculate 5-Day Price Momentum
    price_momentum = df['close'].diff().rolling(window=5).sum()
    
    # Calculate Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(2)).astype(int)
    
    # Calculate Amplitude Confirmation
    daily_amplitude = df['high'] - df['low']
    avg_amplitude = daily_amplitude.rolling(window=5).mean()
    amplitude_confirmation = (daily_amplitude > avg_amplitude).astype(int)
    
    # Combined Volume and Amplitude Momentum Factor
    combined_vol_amp_momentum = price_momentum * (volume_confirmation + amplitude_confirmation) / 2
    
    # Calculate Daily Return Confirmation
    avg_close_return = close_returns.rolling(window=5).mean()
    return_confirmation = (close_returns > avg_close_return).astype(int)
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_factor + combined_vol_amp_momentum + return_confirmation) / 3
    
    return final_alpha_factor
```
