```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily close-to-close return
    df['return'] = df['close'].pct_change()
    
    # Apply Exponential Moving Average (EMA) to returns
    lookback_period = 20
    df['ema_return'] = df['return'].ewm(span=lookback_period, adjust=False).mean()
    
    # Compute EMA of squared returns for volatility
    df['squared_return'] = df['return'] ** 2
    df['ema_volatility'] = df['squared_return'].ewm(span=lookback_period, adjust=False).mean()
    
    # Calculate divergence between EMA of returns and volatility
    df['divergence'] = np.abs(df['ema_return'] - df['ema_volatility'])
    
    # Weight divergence by volume and high-low range
    df['high_low_range'] = df['high'] - df['low']
    df['weighted_divergence'] = (df['divergence'] * df['volume']) / df['high_low_range']
    
    # Calculate intraday movement factor
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_day_true_range'] = (df['high'].shift(1) - df['low'].shift(1)).combine_first(df['high'].shift(1) - df['close'].shift(1)).combine_first(df['close'].shift(1) - df['low'].shift(1))
    df['intraday_movement_factor'] = df['intraday_high_low_spread'] / df['prev_day_true_range']
    
    # Calculate close price direction
    df['close_price_direction'] = np.sign(df['close'] - df['open'])
    
    # Integrate volume impact
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['integrated_factor'] = df['weighted_divergence'] + df['intraday_movement_factor']
    
    # Calculate price momentum
    df['price_momentum'] = df['return'].rolling(window=lookback_period).mean()
    
    # Volume trend
    df['volume_trend'] = df['volume_change'].rolling(window=lookback_period).mean()
    
    # Weighted combination
    df['weighted_combination'] = df['intraday_movement_factor'] + df['close_price_direction']
    
    # Synthesize initial alpha factor
    df['synthesized_alpha_factor'] = (df['price_momentum'] * df['volume_trend'] * df['weighted_combination'] * df['volume_change'])
    
    # Final alpha factor
    df['final_alpha_factor'] = df['integrated_factor'] * df['volume_change'] * df['synthesized_alpha_factor']
    
    return df['final_alpha_factor']
```
