import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index (date).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the daily average price
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of volume to average price
    df['vol_to_avg_price_ratio'] = df['volume'] / df['avg_price']
    
    # Calculate the cumulative sum of the daily return
    df['cumulative_return'] = df['daily_return'].cumsum()
    
    # Calculate the rolling 10-day mean of the daily range
    df['rolling_daily_range_mean'] = df['daily_range'].rolling(window=10).mean()
    
    # Calculate the rolling 10-day mean of the volume to average price ratio
    df['rolling_vol_to_avg_price_ratio_mean'] = df['vol_to_avg_price_ratio'].rolling(window=10).mean()
    
    # Calculate the factor as a weighted combination of the above metrics
    df['factor'] = (df['cumulative_return'] * 0.4) + \
                   (df['rolling_daily_range_mean'] * 0.3) + \
                   (df['rolling_vol_to_avg_price_ratio_mean'] * 0.3)
    
    return df['factor']
