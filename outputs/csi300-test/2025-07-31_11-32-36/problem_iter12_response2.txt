```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    close_to_open_diff = df['close'] - df['open']
    
    # Compute Intraday Momentum Score
    intraday_momentum_score = (intraday_high_low_range + close_to_open_diff) * df['volume']
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] * 2)
    
    # Calculate Volume Weighted Price Change
    volume_weighted_price_change = volume_weighted_price - volume_weighted_price.shift(1)
    
    # Compute Combined Intraday and Daily Momentum
    combined_momentum = intraday_momentum_score + volume_weighted_price_change
    
    # Apply Exponential Moving Average (EMA) to Combined Momentum
    ema_combined_momentum = combined_momentum.ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    average_deltas = (high_close_delta + low_close_delta) / 2
    intraday_volatility_score = average_deltas * df['amount']
    
    # Compute EMA of Squared Returns for Volatility
    daily_returns = df['close'].pct_change()
    squared_returns = daily_returns ** 2
    ema_squared_returns = squared_returns.ewm(span=20, adjust=False).mean()
    
    # Calculate Divergence between EMA of Returns and Volatility
    divergence = (ema_squared_returns - ema_combined_momentum).abs()
    
    # Enhanced Volatility Measure
    true_range = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    atr = true_range.ewm(span=14, adjust=False).mean()
    
    # Volume Trend
    volume_difference = df['volume'] - df['volume'].shift(1)
    rolling_avg_volume_difference = volume_difference.rolling(window=20).mean()
    
    # Integrate Composite Momentum, Volatility, and Volume Trends
    ema_30_combined_momentum = combined_momentum.ewm(span=30, adjust=False).mean()
    integrated_factor = (combined_momentum - ema_30_combined_momentum - intraday_volatility_score + atr + (divergence * df['volume'])) / intraday_high_low_range
    
    # Introduce New Factor: Intraday Price Reversal Score
    intraday_midpoint = (df['high'] + df['low']) / 2
    abs_distance_from_midpoint = (df['close'] - intraday_midpoint).abs()
    intraday_price_reversal_score = abs_distance_from_midpoint * df['volume']
    
    # Integrate into Final Alpha Factor
    final_alpha_factor = integrated_factor + intraday_price_reversal_score
    final_alpha_factor = final_alpha_factor * 0.9  # Smoothing constant
    
    return final_alpha_factor
```
