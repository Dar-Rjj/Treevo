```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['daily_return'] = df['close'].pct_change()
    lookback_window = 30
    df['average_daily_return'] = df['daily_return'].rolling(window=lookback_window).mean()

    # Calculate Volume Trend
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    df['rolling_avg_volume_difference'] = df['volume_difference'].rolling(window=lookback_window).mean()
    df['price_momentum_with_volume_confirmation'] = df['average_daily_return'] * df['rolling_avg_volume_difference']

    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']

    # Calculate Close to Open Difference
    df['close_open_difference'] = df['close'] - df['open']

    # Compute Intraday Momentum Score
    df['sum_of_deltas'] = df['intraday_high_low_range'] + df['close_open_difference']
    df['intraday_momentum_score'] = df['sum_of_deltas'] * df['volume']

    # Calculate Volume Weighted Price
    df['volume_weighted_price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] * 2)

    # Calculate Volume Weighted Price Change
    df['volume_weighted_price_change'] = df['volume_weighted_price'] - df['volume_weighted_price'].shift(1)

    # Compute Combined Intraday and Daily Momentum
    df['combined_intraday_daily_momentum'] = df['intraday_momentum_score'] + df['volume_weighted_price_change']

    # Calculate Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['close'] - df['low']
    df['average_of_deltas'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['intraday_volatility_score'] = df['average_of_deltas'] * df['amount']

    # Integrate Composite Momentum and Volatility
    df['ema_combined_intraday_daily_momentum'] = df['combined_intraday_daily_momentum'].ewm(span=30, adjust=False).mean()
    df['composite_momentum_and_volatility'] = df['combined_intraday_daily_momentum'] - df['ema_combined_intraday_daily_momentum'] - df['intraday_volatility_score']

    # Combine Overall Alpha Factor
    df['overall_alpha_factor'] = (df['intraday_momentum_score'] - df['intraday_volatility_score']) * df['price_momentum_with_volume_confirmation']

    return df['overall_alpha_factor']
```
