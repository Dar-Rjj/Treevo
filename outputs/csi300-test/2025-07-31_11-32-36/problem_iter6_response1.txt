import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum Score
    high_low_delta = df['high'] - df['low']
    close_open_delta = df['close'] - df['open']
    sum_of_deltas = high_low_delta + close_open_delta
    intraday_momentum_score = sum_of_deltas * df['volume']
    
    # Calculate Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    average_of_deltas = (high_close_delta + low_close_delta) / 2
    intraday_volatility_score = average_of_deltas * df['amount']
    
    # Calculate Daily Close-to-Close Return
    daily_return = df['close'].pct_change()
    
    # Apply Exponential Moving Average (EMA) to Returns
    ema_returns = daily_return.ewm(span=20, adjust=False).mean()
    
    # Compute EMA of Squared Returns for Volatility
    squared_returns = daily_return ** 2
    ema_squared_returns = squared_returns.ewm(span=20, adjust=False).mean()
    
    # Combine Scores
    combined_scores = intraday_momentum_score - intraday_volatility_score
    
    # Calculate Divergence and Weight Factor
    divergence = np.abs(ema_returns - np.sqrt(ema_squared_returns))
    high_low_range = df['high'] - df['low']
    weight_factor = (divergence * df['volume']) / high_low_range
    
    # Introduce Time-Varying Volatility Adjustment
    long_term_volatility = squared_returns.ewm(span=50, adjust=False).mean()
    short_term_volatility = squared_returns.ewm(span=10, adjust=False).mean()
    volatility_ratio = short_term_volatility / long_term_volatility
    
    # Adjust Combined Scores
    adjusted_combined_scores = combined_scores * volatility_ratio
    
    return adjusted_combined_scores
