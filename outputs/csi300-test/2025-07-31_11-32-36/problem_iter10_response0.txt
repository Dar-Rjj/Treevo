```python
import pandas as pd
import numpy as np

def heuristics_v2(df, mav_window=20, maa_window=20, ema_span=10):
    # Calculate High-Low Delta
    high_low_delta = df['high'] - df['low']
    
    # Calculate Close-Open Delta
    close_open_delta = df['close'] - df['open']
    
    # Sum of Deltas
    sum_of_deltas = high_low_delta + close_open_delta
    
    # Calculate Moving Average of Volume (MAV)
    mav = df['volume'].rolling(window=mav_window).mean()
    
    # Adjusted Volume
    adjusted_volume = df['volume'] / mav
    
    # Intraday Momentum Score
    intraday_momentum_score = sum_of_deltas * adjusted_volume
    
    # Calculate High-Close Delta
    high_close_delta = df['high'] - df['close']
    
    # Calculate Low-Close Delta
    low_close_delta = df['close'] - df['low']
    
    # Average of Deltas
    avg_of_deltas = (high_close_delta + low_close_delta) / 2
    
    # Calculate Moving Average of Amount (MAA)
    maa = df['amount'].rolling(window=maa_window).mean()
    
    # Adjusted Amount
    adjusted_amount = df['amount'] / maa
    
    # Intraday Volatility Score
    intraday_volatility_score = avg_of_deltas * adjusted_amount
    
    # Calculate Close-Open Ratio
    close_open_ratio = df['close'] / df['open']
    
    # Smooth Close-Open Ratio using Exponential Moving Average (EMA)
    smooth_close_open_ratio = close_open_ratio.ewm(span=ema_span, adjust=False).mean()
    
    # Price Trend Factor
    price_trend_factor = smooth_close_open_ratio
    
    # Combine Scores
    alpha_factor = (intraday_momentum_score - intraday_volatility_score) + price_trend_factor
    
    return alpha_factor
```
