```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Average Daily Return over 5-day lookback window
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()

    # Rolling Average Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['rolling_avg_volume_diff'] = df['volume_diff'].rolling(window=5).mean()

    # Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Amplitude Confirmation
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)

    # Combine Volume and Amplitude Confirmation
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) * 0.5

    # Price Momentum Adjusted by Volume and Amplitude
    df['price_momentum_adjusted'] = df['avg_daily_return'] * df['rolling_avg_volume_diff'] * df['combined_confirmation']

    # Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']

    # Close to Open Difference
    df['close_open_diff'] = df['close'] - df['open']

    # Sum of Deltas
    df['sum_of_deltas'] = df['intraday_high_low_range'] + df['close_open_diff']

    # Weight by Volume: Volume of day t * Sum of Deltas
    df['intraday_momentum_score'] = df['volume'] * df['sum_of_deltas']

    # Volume Weighted Price Change
    df['volume_weighted_price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / df['volume']
    df['volume_weighted_price_change'] = df['volume_weighted_price'] - df['volume_weighted_price'].shift(1)

    # Combined Intraday and Daily Momentum
    df['combined_intraday_daily_momentum'] = df['intraday_momentum_score'] + df['volume_weighted_price_change']

    # Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['average_delta'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['intraday_volatility_score'] = df['amount'] * df['intraday_high_low_range'].rolling(window=30).std() * df['average_delta']

    # Integrate Composite Momentum and Volatility
    df['ema_30_combined_momentum'] = df['combined_intraday_daily_momentum'].ewm(span=30, adjust=False).mean()
    df['integrated_momentum_volatility'] = df['combined_intraday_daily_momentum'] - df['ema_30_combined_momentum'] - df['intraday_volatility_score']

    # Adjusted by Volume and Price Trend
    df['volume_price_trend'] = df['volume'] * df['close']

    # Combine Factors
    df['alpha_factor'] = (
        df['price_momentum_adjusted'] +
        df['intraday_momentum_score'] +
        df['combined_intraday_daily_momentum'] -
        df['intraday_volatility_score'] +
        df['volume_weighted_price_change'] +
        df['volume_price_trend']
    )

    return df['alpha_factor']
```
