```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Momentum
    momentum = df['close'] / df['close'].shift(1) - 1
    
    # Combine Intraday Range and Momentum
    combined_factor = intraday_range * momentum
    
    # Adjust for Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_factor = combined_factor * volume_ratio
    
    return adjusted_factor
```
