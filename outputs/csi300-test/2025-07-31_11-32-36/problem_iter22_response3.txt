```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Average Daily Return
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()

    # Calculate Volume Difference
    df['volume_diff'] = df['volume'].diff()

    # Rolling Average Volume Difference
    df['rolling_avg_volume_diff'] = df['volume_diff'].rolling(window=5).mean()

    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Calculate Amplitude Confirmation
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)

    # Combine Volume and Amplitude Confirmation
    df['combined_confirmation'] = 0.5 * (df['volume_confirmation'] + df['amplitude_confirmation'])

    # Multiply Price Momentum by Volume and Amplitude Trend
    df['price_momentum_with_trend'] = df['avg_daily_return'] * df['rolling_avg_volume_diff'] * df['combined_confirmation']

    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']

    # Calculate Close to Open Difference
    df['close_open_diff'] = df['close'] - df['open']

    # Compute Intraday Momentum Score
    df['intraday_momentum_score'] = (df['intraday_high_low_range'] + df['close_open_diff']) * df['volume']

    # Calculate Volume Weighted Price
    df['volume_weighted_price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / df['volume']

    # Calculate Volume Weighted Price Change
    df['volume_weighted_price_change'] = df['volume_weighted_price'] - df['volume_weighted_price'].shift(1)

    # Compute Combined Intraday and Daily Momentum
    df['combined_intraday_daily_momentum'] = df['intraday_momentum_score'] + df['volume_weighted_price_change']

    # Calculate Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['average_deltas'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['intraday_volatility_score'] = df['average_deltas'] * df['amount'] * df['intraday_high_low_range'].rolling(window=30).std()

    # Integrate Composite Momentum and Volatility
    df['ema_combined_momentum'] = df['combined_intraday_daily_momentum'].ewm(span=30, adjust=False).mean()
    df['integrated_momentum_volatility'] = df['combined_intraday_daily_momentum'] - df['ema_combined_momentum'] - df['intraday_volatility_score']

    # Combine Factors
    df['alpha_factor'] = (
        df['price_momentum_with_trend'] +
        df['intraday_high_low_range'] +
        df['intraday_momentum_score'] +
        df['combined_intraday_daily_momentum'] +
        df['volume_weighted_price_change'] -
        df['intraday_volatility_score']
    ) * df['volume'] * df['close']

    return df['alpha_factor']
```
