```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close Price Direction
    close_price_direction = df['close'] - df['open']
    
    # Compute Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Integrate Volume Impact
    integrated_factor = (intraday_high_low_spread * close_price_direction) * volume_change
    
    return integrated_factor
```
