import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['daily_close_change'] = df['close'].diff()
    df['5_day_momentum'] = df['daily_close_change'].rolling(window=5).sum()
    
    # Ensure sum is non-zero
    df['5_day_momentum'] = df['5_day_momentum'].apply(lambda x: x if x != 0 else 1e-6)
    
    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Amplitude Confirmation
    df['daily_amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['daily_amplitude'] > df['daily_amplitude'].shift(1)).astype(int)
    
    # Calculate 5-Day Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x[0] - x[1], abs(x[0] - df['close'].shift(1)), abs(df['close'].shift(1) - x[1])), axis=1)
    df['5_day_atr'] = df['true_range'].rolling(window=5).mean()
    
    # Adjust for Volatility
    df['volatility_adjustment'] = (df['volume_confirmation'] + df['amplitude_confirmation']) / df['5_day_atr']
    
    # Introduce Trend Strength Confirmation
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['trend_strength_confirmation'] = (df['close'] > df['20_day_sma']).astype(int)
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = df['5_day_momentum'] * df['volatility_adjustment'] * df['trend_strength_confirmation']
    
    return df['alpha_factor'].dropna()
