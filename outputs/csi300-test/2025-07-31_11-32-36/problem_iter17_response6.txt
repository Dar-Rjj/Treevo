```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['Price_Momentum'] = df['close'].diff().rolling(window=5).sum()
    
    # Enhance Volume Confirmation
    df['Volume_Confirmation'] = (df['volume'] > (df['volume'].shift(1) + df['volume'].shift(2)) / 2).astype(int)
    
    # Incorporate Amplitude Analysis for Confirmation
    df['Amplitude'] = df['high'] - df['low']
    avg_amplitude_5_days = df['Amplitude'].rolling(window=5).mean()
    df['Amplitude_Analysis'] = ((df['Amplitude'] > df['Amplitude'].shift(1)) & 
                                (df['Amplitude'] > avg_amplitude_5_days)).astype(int)
    
    # Adjust for Volatility
    df['True_Range'] = df[['high' - 'low', 'high' - df['close'].shift(1), 'low' - df['close'].shift(1)]].max(axis=1)
    df['ATR_5_Days'] = df['True_Range'].rolling(window=5).mean()
    df['Volatility_Adjustment'] = (df['Volume_Confirmation'] + df['Amplitude_Analysis']) * df['ATR_5_Days']
    
    # Introduce Advanced Trend Strength Confirmation
    df['SMA_20_Days'] = df['close'].rolling(window=20).mean()
    df['Trend_Strength'] = (df['close'] > df['SMA_20_Days']).rolling(window=3).sum() == 3
    
    # Combine all components to form the final alpha factor
    df['Alpha_Factor'] = df['Price_Momentum'] + df['Volatility_Adjustment'] + df['Trend_Strength'].astype(int)
    
    return df['Alpha_Factor']
```
