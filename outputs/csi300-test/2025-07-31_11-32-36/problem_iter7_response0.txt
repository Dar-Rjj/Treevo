```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['returns'] = df['close'].pct_change()
    
    # Apply Exponential Moving Average (EMA) to Returns
    lookback_period = 20
    ema_alpha = 2 / (lookback_period + 1)
    df['ema_returns'] = df['returns'].ewm(alpha=ema_alpha, adjust=False).mean()
    
    # Compute EMA of Squared Returns for Volatility
    df['squared_returns'] = df['returns'] ** 2
    df['ema_squared_returns'] = df['squared_returns'].ewm(alpha=ema_alpha, adjust=False).mean()
    df['volatility'] = np.sqrt(df['ema_squared_returns'])
    
    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = (df['volatility'] - df['ema_returns']).abs()
    
    # Weight Divergence by Volume and High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['weighted_divergence'] = (df['divergence'] * df['volume']) / df['high_low_range']
    
    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Compute Intraday Momentum
    df['intraday_momentum'] = df['intraday_high_low_range'] / df['close_open_diff']
    
    # Calculate Volume Weighted Price
    df['volume_weighted_price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])
    
    # Calculate Volume Weighted Price Change
    df['volume_weighted_price_change'] = df['volume_weighted_price'].diff()
    
    # Compute Combined Intraday and Daily Momentum
    df['combined_momentum'] = df['intraday_momentum'] + df['volume_weighted_price_change']
    
    # Integrate Momentum and Reversal
    long_lookback_period = 30
    long_ema_alpha = 2 / (long_lookback_period + 1)
    df['long_ema_combined_momentum'] = df['combined_momentum'].ewm(alpha=long_ema_alpha, adjust=False).mean()
    df['integrated_momentum'] = df['combined_momentum'] - df['long_ema_combined_momentum']
    
    # Combine Weighted Divergence and Integrated Momentum
    df['alpha_factor'] = df['weighted_divergence'] + df['integrated_momentum']
    
    return df['alpha_factor']
```
