```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_window=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Average Daily Return
    df['avg_daily_return'] = df['daily_return'].rolling(window=lookback_window).mean()
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'].diff()
    
    # Rolling Average Volume Difference
    df['avg_volume_diff'] = df['volume_diff'].rolling(window=lookback_window).mean()
    
    # Multiply Price Momentum by Volume Trend
    df['price_momentum_volume_trend'] = df['avg_daily_return'] * df['avg_volume_diff']
    
    # Calculate Intraday Price Range Ratio
    df['intraday_price_range_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Close
    df['volume_adjusted_close'] = df['close'] * np.sqrt(df['volume'])
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    df['alpha_factor'] = (df['price_momentum_volume_trend'] + 
                           df['intraday_price_range_ratio'] + 
                           df['volume_adjusted_close'] + 
                           df['intraday_return'])
    
    return df['alpha_factor']
```
