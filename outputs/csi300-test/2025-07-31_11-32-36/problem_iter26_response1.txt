```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Score
    high_low_delta = df['high'] - df['low']
    close_open_delta = df['close'] - df['open']
    sum_of_deltas = high_low_delta + close_open_delta
    mav = df['volume'].rolling(window=10).mean()
    weighted_momentum = sum_of_deltas * (df['volume'] / mav)

    # Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    average_of_deltas = (high_close_delta + low_close_delta) / 2
    maa = df['amount'].rolling(window=10).mean()
    std_intraday_high_low = df['high'] - df['low']
    std_intraday_high_low_30 = std_intraday_high_low.rolling(window=30).std()
    enhanced_volatility = average_of_deltas * std_intraday_high_low_30 * (df['amount'] / maa)

    # Price Trend Factor
    close_open_ratio = df['close'] / df['open']
    smoothed_ratio = close_open_ratio.ewm(span=10, adjust=False).mean()

    # Daily Momentum and Volatility Components
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ema_returns = daily_returns.ewm(span=10, adjust=False).mean()
    ema_squared_returns = (daily_returns**2).ewm(span=10, adjust=False).mean()
    divergence = abs(ema_returns - ema_squared_returns)
    high_low_range = df['high'] - df['low']
    weighted_divergence = divergence * (df['volume'] / high_low_range)

    # Combined Intraday and Daily Scores
    total_momentum = weighted_momentum + smoothed_ratio
    net_score = total_momentum - enhanced_volatility + weighted_divergence

    # Volume and Price Adjusted Momentum
    volume_weighted_price = (df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] + df['volume'].shift(1))
    volume_weighted_price_change = volume_weighted_price - volume_weighted_price.shift(1)
    integrated_momentum = (weighted_momentum + volume_weighted_price_change).ewm(span=30, adjust=False).mean()
    momentum_reversal = (weighted_momentum + volume_weighted_price_change) - integrated_momentum

    # Long-Term and Short-Term Volatility
    long_term_volatility = ema_squared_returns.ewm(span=50, adjust=False).mean()
    short_term_volatility = ema_squared_returns.ewm(span=10, adjust=False).mean()
    volatility_ratio = short_term_volatility / long_term_volatility

    # Volume and Amplitude Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    amplitude = df['high'] - df['low']
    amplitude_confirmation = (amplitude > amplitude.shift(1)).astype(int)
    combined_confirmation = (volume_confirmation + amplitude_confirmation) * 0.5

    # Final Alpha Factor
    core_factor = (net_score + integrated_momentum) * volatility_ratio
    volume_trend = (df['volume'] - df['volume'].shift(1)) / 30
    price_momentum_over_volume_trend = daily_returns / (df['volume'] - df['volume'].shift(1)).rolling(window=30).mean()
    final_alpha = core_factor + volume_trend + (price_momentum_over_volume_trend * combined_confirmation)

    return final_alpha
```
