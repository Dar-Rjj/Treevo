import pandas as pd

def heuristics_v2(df, N=10, M=5, K=20):
    """
    Calculate a combined factor using intraday volatility and momentum, adjusted for trading volume.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as (date).
    - N: Number of days to compute the average range for intraday volatility.
    - M: Number of days to compute the cumulative return for momentum.
    - K: Number of days to compute the average trading volume.
    
    Returns:
    - pandas Series with index as (date) representing the factor values.
    """
    # Calculate Intraday Volatility
    df['range'] = df['high'] - df['low']
    df['avg_range'] = df['range'].rolling(window=N).mean()
    
    # Calculate Momentum from Close Price
    df['daily_return'] = df['close'].pct_change()
    df['cumulative_return'] = (df['close'] / df['close'].shift(M) - 1).fillna(0)
    
    # Combine Factors
    df['raw_combined_factor'] = df['avg_range'] * df['cumulative_return']
    
    # Adjust for Trading Volume
    df['avg_volume'] = df['volume'].rolling(window=K).mean()
    df['factor'] = df['raw_combined_factor'] / df['avg_volume']
    
    return df['factor']
