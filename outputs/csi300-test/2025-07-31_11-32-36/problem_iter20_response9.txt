```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on (open, high, low, close, amount, volume) prices.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: The alpha factor values indexed by date.
    """
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the money flow
    ma_money_flow_10 = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the typical price
    ma_typical_price_10 = typical_price.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the volume
    ma_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Alpha factor: (money flow / average money flow) - (typical price / average typical price) * (volume / average volume)
    alpha_factor = (money_flow / ma_money_flow_10) - (typical_price / ma_typical_price_10) * (df['volume'] / ma_volume_10)
    
    return alpha_factor
```
