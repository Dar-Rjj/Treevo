```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Score
    high_low_delta = df['high'] - df['low']
    close_open_delta = df['close'] - df['open']
    sum_of_deltas = (high_low_delta + close_open_delta) / 2
    
    # Dynamic Volume Adjustment
    vroc = df['volume'].pct_change()
    intraday_momentum_score = sum_of_deltas * vroc
    
    # Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    max_of_deltas = pd.concat([high_close_delta, low_close_delta.abs()], axis=1).max(axis=1)
    
    # Dynamic Amount Adjustment
    aroc = df['amount'].pct_change()
    intraday_volatility_score = max_of_deltas * aroc
    
    # Price Trend Factor
    close_open_ratio = df['close'] / df['open']
    ema_close_open_ratio = close_open_ratio.ewm(span=10, adjust=False).mean()
    
    # Combine Scores
    combined_factor = (intraday_momentum_score 
                       - intraday_volatility_score 
                       + ema_close_open_ratio)
    
    return combined_factor
```
