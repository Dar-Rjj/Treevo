import pandas as pd

def heuristics_v2(df, N=10, M=10, K=10):
    # Calculate Intraday Volatility
    df['range'] = df['high'] - df['low']
    df['avg_range'] = df['range'].rolling(window=N).mean()
    
    # Calculate Momentum from Close Price
    df['daily_returns'] = df['close'] / df['close'].shift(1) - 1
    df['cumulative_return'] = df['daily_returns'].rolling(window=M).sum()
    
    # Combine Factors
    df['raw_combined_factor'] = df['avg_range'] * df['cumulative_return']
    
    # Adjust for Trading Volume
    df['volume_ema'] = df['volume'].ewm(span=K, adjust=False).mean()
    df['adjusted_factor'] = df['raw_combined_factor'] / df['volume_ema']
    
    return df['adjusted_factor'].dropna()
