```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Daily Returns
    df['volume_adjusted_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Calculate Weighted Momentum
    df['10_day_moving_avg'] = df['volume_adjusted_return'].rolling(window=9).mean().shift(1)
    df['20_day_ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['weighted_momentum'] = df['10_day_moving_avg'] / df['20_day_ema_volume']
    
    # Calculate Volatility
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['10_day_std'] = df['daily_return'].rolling(window=9).std().shift(1)
    df['inverse_volatility'] = 1 / (df['10_day_std'] + 1e-8)
    
    # Sum Weighted Momentum and Inverse of Volatility
    df['factor_value'] = df['weighted_momentum'] + df['inverse_volatility']
    
    return df['factor_value'].copy()
```
