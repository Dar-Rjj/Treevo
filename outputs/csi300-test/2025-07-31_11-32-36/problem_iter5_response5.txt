```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    close_to_open_diff = df['close'] - df['open']
    
    # Compute Intraday Momentum Score
    sum_of_deltas = intraday_high_low_range + close_to_open_diff
    intraday_momentum_score = sum_of_deltas * df['volume']
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] * 2)
    
    # Calculate Volume Weighted Price Change
    volume_weighted_price_change = volume_weighted_price - volume_weighted_price.shift(1)
    
    # Compute Combined Intraday and Daily Momentum
    combined_intraday_daily_momentum = intraday_momentum_score + volume_weighted_price_change
    
    # Calculate Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    average_of_deltas = (high_close_delta + low_close_delta) / 2
    intraday_volatility_score = average_of_deltas * df['amount']
    
    # Integrate Composite Momentum and Volatility
    combined_momentum_30d_ema = combined_intraday_daily_momentum.ewm(span=30, adjust=False).mean()
    integrated_composite_momentum = combined_intraday_daily_momentum - combined_momentum_30d_ema - intraday_volatility_score
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Average Daily Return
    lookback_window = 30
    average_daily_return = daily_return.rolling(window=lookback_window).mean()
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Rolling Average Volume Difference
    rolling_average_volume_difference = volume_difference.rolling(window=lookback_window).mean()
    
    # Multiply Price Momentum by Volume Trend
    price_momentum_by_volume_trend = average_daily_return * rolling_average_volume_difference
    
    # Calculate Intraday Price Range Ratio
    intraday_price_range_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Close
    volume_adjusted_close = df['close'] * np.sqrt(df['volume'])
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine All Factors
    alpha_factor = (price_momentum_by_volume_trend 
                    + intraday_price_range_ratio 
                    + volume_adjusted_close 
                    + intraday_return 
                    + integrated_composite_momentum)
    
    return alpha_factor
```
