import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns: open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A Series with (date) as index and the generated alpha factor as values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the daily return
    df['std_10_daily_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day Exponential Moving Average (EMA) of the close price
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day EMA of the volume
    df['ema_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the volatility factor: (EMA of close - EMA of volume) / (Standard Deviation of returns)
    df['volatility_factor'] = (df['ema_10_close'] - df['ema_10_volume']) / df['std_10_daily_return']
    
    # Normalize the volatility factor using the 10-day moving average
    df['alpha_factor'] = df['volatility_factor'] / df['ma_10_daily_return']
    
    # Drop NaN values
    df.dropna(inplace=True)
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
