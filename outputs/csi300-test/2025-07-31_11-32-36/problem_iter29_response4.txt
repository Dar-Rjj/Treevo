```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index (date).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the relative range
    relative_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative return
    relative_return = df['close'].pct_change()
    
    # Calculate the volume weighted price
    volume_weighted_price = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(21)
    
    # Combine the factors
    factor = (relative_range * relative_return) + volume_weighted_price + momentum
    
    return factor.dropna()
```
