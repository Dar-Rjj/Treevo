```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Average Daily Return over 5-day lookback window
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Rolling Average Volume Difference over 5-day lookback window
    df['avg_volume_diff'] = df['volume_diff'].rolling(window=5).mean()
    
    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Amplitude Confirmation
    df['amplitude_t'] = df['high'] - df['low']
    df['amplitude_t-1'] = df['amplitude_t'].shift(1)
    df['amplitude_confirmation'] = (df['amplitude_t'] > df['amplitude_t-1']).astype(int)
    
    # Combine Volume and Amplitude Confirmation
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) * 0.5
    
    # Multiply Price Momentum by Volume and Amplitude Trend
    df['price_momentum_with_trend'] = df['avg_daily_return'] * df['avg_volume_diff'] * df['combined_confirmation']
    
    # Calculate Intraday Price Range Ratio
    df['intraday_price_range_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Close
    df['volume_adjusted_close'] = df['close'] * df['volume'].apply(lambda x: x**0.5)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    df['alpha_factor'] = (df['price_momentum_with_trend'] 
                          + df['intraday_price_range_ratio'] 
                          + df['volume_adjusted_close'] 
                          + df['intraday_return'])
    
    return df['alpha_factor']
```
