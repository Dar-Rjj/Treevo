```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].diff()

    # Average Daily Return over the 5-day lookback window
    df['Average_Daily_Return'] = df['Daily_Return'].rolling(window=5).mean()

    # Calculate Volume Difference
    df['Volume_Diff'] = df['volume'].diff()

    # Rolling Average Volume Difference
    df['Rolling_Avg_Volume_Diff'] = df['Volume_Diff'].rolling(window=5).mean()

    # Volume Confirmation
    df['Volume_Confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Amplitude Confirmation
    df['Amplitude'] = df['high'] - df['low']
    df['Amplitude_Confirmation'] = (df['Amplitude'] > df['Amplitude'].shift(1)).astype(int)
    df['Combined_Confirmation'] = 0.5 * (df['Volume_Confirmation'] + df['Amplitude_Confirmation'])

    # Multiply Price Momentum by Volume and Amplitude Trend
    df['Price_Momentum_Volume_Amplitude_Trend'] = df['Average_Daily_Return'] * df['Rolling_Avg_Volume_Diff'] * df['Combined_Confirmation']

    # Intraday High-Low Range
    df['Intraday_High_Low_Range'] = df['high'] - df['low']

    # Close to Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']

    # Intraday Momentum Score
    df['Intraday_Momentum_Score'] = df['Intraday_High_Low_Range'] + df['Close_Open_Diff']

    # Volume Weighted Price
    df['Volume_Weighted_Price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])

    # Volume Weighted Price Change
    df['Volume_Weighted_Price_Change'] = df['Volume_Weighted_Price'].diff()

    # Combined Intraday and Daily Momentum
    df['Combined_Momentum'] = df['Intraday_Momentum_Score'] + df['Volume_Weighted_Price_Change']

    # Intraday Volatility Score
    df['High_Close_Delta'] = df['high'] - df['close']
    df['Low_Close_Delta'] = df['low'] - df['close']
    df['Average_Deltas'] = (df['High_Close_Delta'] + df['Low_Close_Delta']) / 2
    df['Enhanced_Volatility_Score'] = df['Average_Deltas'] * df['Intraday_High_Low_Range'].rolling(window=30).std()

    # Integrate Composite Momentum and Volatility
    df['Exp_Moving_Avg_Combined_Momentum'] = df['Combined_Momentum'].ewm(span=30).mean()
    df['Composite_Momentum_Volatility'] = df['Combined_Momentum'] - df['Exp_Moving_Avg_Combined_Momentum'] - df['Enhanced_Volatility_Score']

    # Price Momentum Over Volume Trend
    df['Price_Momentum_Over_Volume_Trend'] = df['Average_Daily_Return'] / df['Rolling_Avg_Volume_Diff']

    # Combine Factors
    df['Alpha_Factor'] = (
        df['Price_Momentum_Volume_Amplitude_Trend'] +
        df['Intraday_High_Low_Range'] +
        df['Intraday_Momentum_Score'] +
        df['Combined_Momentum'] +
        df['Enhanced_Volatility_Score'] +
        df['Volume_Weighted_Price_Change'] +
        df['Price_Momentum_Over_Volume_Trend']
    )

    return df['Alpha_Factor'].dropna()
```
