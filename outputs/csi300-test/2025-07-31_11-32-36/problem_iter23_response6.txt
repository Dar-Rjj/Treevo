```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: Factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day EMA of the money flow
    ema_10 = money_flow.ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day EMA of the money flow
    ema_30 = money_flow.ewm(span=30, adjust=False).mean()
    
    # Calculate the difference between the 10-day and 30-day EMA
    ema_diff = ema_10 - ema_30
    
    # Normalize the EMA difference by the 30-day standard deviation of the money flow
    std_30 = money_flow.rolling(window=30).std()
    normalized_ema_diff = ema_diff / std_30
    
    # Return the final factor
    return normalized_ema_diff
```
