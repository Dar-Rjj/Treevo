import pandas as pd

def heuristics_v2(df):
    """
    This function generates a novel and interpretable alpha factor based on the provided
    (open, high, low, close, amount, volume) data. The factor aims to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] 
                       and an index of dates.
    
    Returns:
    pd.Series: A Series with the index as dates and the values as the calculated alpha factor.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the 10-day and 30-day moving averages
    df['ma_diff'] = df['ma_10'] - df['ma_30']
    
    # Calculate the 10-day standard deviation of the daily return
    df['std_10'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day sum of the volume
    df['vol_sum_10'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the factor as a combination of the moving average difference and the standardized volume
    df['factor'] = (df['ma_diff'] / df['std_10']) * df['vol_sum_10']
    
    # Return the factor as a Series
    return df['factor']
