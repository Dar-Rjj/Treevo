```python
import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the provided DataFrame.
    The factor is a combination of intraday price movement, volume, and amount traded.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series representing the alpha factor values.
    """
    # Calculate the intraday price range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the closing price relative to the day's range
    df['close_rel_to_range'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Calculate the volume-weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume'])
    
    # Calculate the difference between the closing price and VWAP
    df['close_vwap_diff'] = df['close'] - df['vwap']
    
    # Calculate the 10-day rolling standard deviation of the intraday range
    df['intraday_range_volatility'] = df['intraday_range'].rolling(window=10).std()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_rel_to_range'] * df['close_vwap_diff']) / df['intraday_range_volatility']
    
    return df['alpha_factor']
```
