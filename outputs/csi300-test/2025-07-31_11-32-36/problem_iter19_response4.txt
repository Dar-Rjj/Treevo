```python
import pandas as pd

def heuristics_v2(df, n=10):
    """
    Compute a novel alpha factor that combines intraday volatility and momentum, adjusted for volume.

    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    - n: Number of days to compute the EMA of volume

    Returns:
    - A pandas Series with the computed alpha factor
    """
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Momentum
    momentum = df['close'].pct_change(1).fillna(0)
    
    # Combine Intraday Range and Momentum
    combined_factor = intraday_range * momentum
    
    # Adjust for Volume
    ema_volume = df['volume'].ewm(span=n, min_periods=n).mean()
    adjusted_factor = combined_factor / ema_volume
    
    return adjusted_factor
```
