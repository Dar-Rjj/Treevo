```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Score
    high_low_delta = df['high'] - df['low']
    close_open_delta = df['close'] - df['open']
    sum_of_deltas = high_low_delta + close_open_delta
    mav = df['volume'].rolling(window=20).mean()
    intraday_momentum_score = (sum_of_deltas / mav) * df['volume']

    # Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    average_of_deltas = (high_close_delta + low_close_delta) / 2
    maa = df['amount'].rolling(window=20).mean()
    intraday_volatility_score = (average_of_deltas / maa) * df['amount']

    # Price Trend Factor
    close_open_ratio = df['close'] / df['open']
    price_trend_factor = close_open_ratio.ewm(span=10, adjust=False).mean()

    # Volume Thrust Indicator
    volume_ratio = df['volume'] / df['volume'].shift(1)
    intraday_range = df['high'] - df['low']
    volume_thrust_indicator = volume_ratio * (intraday_range / (df['close'] - df['open']))
    smoothed_volume_thrust = volume_thrust_indicator.ewm(span=5, adjust=False).mean()

    # Combine Scores
    composite_alpha = (
        intraday_momentum_score 
        - intraday_volatility_score 
        + price_trend_factor 
        * smoothed_volume_thrust
    )

    return composite_alpha
```
