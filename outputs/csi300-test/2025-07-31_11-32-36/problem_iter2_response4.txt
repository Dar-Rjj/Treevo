```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Calculate Weighted Momentum
    df['10_day_moving_avg'] = df['daily_return'].shift(1).rolling(window=10).mean()
    df['weighted_momentum'] = df['10_day_moving_avg'] * df['volume']

    # Calculate Volatility
    df['10_day_std'] = df['daily_return'].shift(1).rolling(window=10).std() + 1e-8  # Add a small constant to avoid division by zero
    df['inverse_volatility'] = 1 / df['10_day_std']

    # Calculate Intraday Price Range Ratio
    df['intraday_price_range_ratio'] = (df['high'] - df['low']) / df['low']

    # Calculate Volume Adjusted Close
    df['volume_adjusted_close'] = df['close'] * np.sqrt(df['volume'])

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Combine All Factors
    df['alpha_factor'] = (
        df['weighted_momentum'] +
        df['inverse_volatility'] +
        df['intraday_price_range_ratio'] +
        df['volume_adjusted_close'] +
        df['intraday_return']
    )

    return df['alpha_factor']
```
