```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Short-Term and Long-Term Momentum
    df['short_term_momentum'] = (df['close'] / df['close'].shift(7)) - 1
    df['long_term_momentum'] = (df['close'] / df['close'].shift(25)) - 1
    
    # Calculate Relative Momentum
    df['relative_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Integrate Intraday Volatility and Momentum
    df['integrated_factor'] = df['intraday_volatility'] * df['relative_momentum']
    
    # Calculate Daily Volume Growth
    df['daily_volume_growth'] = (df['volume'] / df['volume'].shift(2)) - 1
    
    # Factor Adjusted for Volume Changes
    df['factor_adjusted_for_volume'] = df['integrated_factor'] * df['daily_volume_growth']
    
    # Calculate Average Trade Size
    df['average_trade_size'] = df['amount'] / df['volume']
    
    # Calculate Price Trend
    df['price_trend'] = (df['close'] - df['open']) / df['open']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['factor_adjusted_for_volume'] * 
                          df['price_trend'] * 
                          df['average_trade_size'])
    
    return df['alpha_factor'].dropna()
```
