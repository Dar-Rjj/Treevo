```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()

    # Calculate 5-Day Price Momentum
    df['5_day_momentum'] = df['price_change'].rolling(window=5).sum()
    df['5_day_momentum'] = df['5_day_momentum'].replace(0, np.nan)

    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Adjust for Volatility using 5-Day Average True Range (ATR)
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['5_day_atr'] = df['true_range'].rolling(window=5).mean()
    df['momentum_adjusted_volatility'] = (df['5_day_momentum'] * df['volume_confirmation']) / df['5_day_atr']

    # Apply Exponential Moving Average (EMA) to Returns
    lookback_period = 20
    df['returns'] = df['close'].pct_change()
    df['ema_returns'] = df['returns'].ewm(span=lookback_period, adjust=False).mean()

    # Compute EMA of Squared Returns for Volatility
    df['squared_returns'] = df['returns'] ** 2
    df['ema_squared_returns'] = df['squared_returns'].ewm(span=lookback_period, adjust=False).mean()

    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = np.abs(df['ema_returns'] - df['ema_squared_returns'])

    # Volume Trend
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    df['rolling_avg_volume_difference'] = df['volume_difference'].rolling(window=lookback_period).mean()

    # Integrate and Weight by Volume and High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['weighted_divergence'] = (df['divergence'] * df['rolling_avg_volume_difference']) / df['high_low_range']

    # Final factor
    alpha_factor = df['weighted_divergence'].fillna(0)

    return alpha_factor
```
