```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Compute Intraday Momentum
    intraday_momentum = intraday_range / close_open_diff
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] + df['volume'].shift(1))
    
    # Calculate Volume Weighted Price Change
    volume_weighted_price_change = volume_weighted_price - volume_weighted_price.shift(1)
    
    # Compute Combined Intraday and Daily Momentum
    combined_momentum = intraday_momentum + volume_weighted_price_change
    
    # Integrate Momentum and Reversal
    ema_30 = combined_momentum.ewm(span=30, adjust=False).mean()
    factor = combined_momentum - ema_30
    
    return factor
```
