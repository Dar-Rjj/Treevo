import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['range'] = df['high'] - df['low']
    intraday_volatility = df['range'].rolling(window=20).mean()

    # Calculate Daily Close-to-Close Return
    df['returns'] = df['close'].pct_change()

    # Apply Exponential Moving Average (EMA) to Returns
    ema_returns = df['returns'].ewm(span=20, adjust=False).mean()

    # Compute EMA of Squared Returns for Volatility
    squared_returns = df['returns'] ** 2
    ema_squared_returns = squared_returns.ewm(span=20, adjust=False).mean()
    ema_volatility = np.sqrt(ema_squared_returns)

    # Calculate Divergence between EMA of Returns and Volatility
    divergence = np.abs(ema_returns - ema_volatility)

    # Weight Divergence by Volume and High-Low Range
    weighted_divergence = (divergence * df['volume']) / df['range']

    # Calculate Intraday Momentum
    intraday_return = (df['high'] - df['low']) / df['low']
    average_volume = df['volume'].rolling(window=20).mean()
    intraday_momentum = intraday_return * (df['volume'] / average_volume)

    # Calculate Intraday Movement Factor
    high_low_spread = df['high'] - df['low']
    prev_true_range = (df['high'].shift(1) - df['low'].shift(1)).rolling(window=20).max()
    intraday_movement_factor = high_low_spread / prev_true_range

    # Integrate Volume Impact
    volume_change = df['volume'].pct_change()
    integrated_factor = weighted_divergence + intraday_movement_factor

    # Calculate Price Momentum
    daily_return = df['returns']
    price_momentum = daily_return.rolling(window=20).mean()

    # Volume Trend
    volume_difference = df['volume'].diff()
    volume_trend = volume_difference.rolling(window=20).mean()

    # Weighted Combination
    weighted_combination = (intraday_movement_factor + intraday_momentum) / 2

    # Synthesize Initial Alpha Factor
    initial_alpha_factor = price_momentum * volume_trend * weighted_combination * (1 + volume_change)

    # Final Alpha Factor
    final_alpha_factor = (weighted_divergence + intraday_movement_factor) * (1 + volume_change) * initial_alpha_factor

    return final_alpha_factor
