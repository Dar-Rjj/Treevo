```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum Score
    high_low_delta = df['high'] - df['low']
    close_open_delta = df['close'] - df['open']
    sum_of_deltas = high_low_delta + close_open_delta
    
    # Calculate Moving Average of Volume (MAV)
    mav = df['volume'].rolling(window=5).mean()
    adjusted_volume = df['volume'] / mav
    intraday_momentum_score = sum_of_deltas * adjusted_volume

    # Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    average_of_deltas = (high_close_delta + low_close_delta) / 2
    
    # Calculate Moving Average of Amount (MAA)
    maa = df['amount'].rolling(window=5).mean()
    adjusted_amount = df['amount'] / maa
    intraday_volatility_score = average_of_deltas * adjusted_amount

    # Price Trend Factor
    close_open_ratio = df['close'] / df['open']
    smoothed_close_open_ratio = close_open_ratio.ewm(span=5, adjust=False).mean()
    
    # Combine Scores
    combined_score = (intraday_momentum_score - intraday_volatility_score + smoothed_close_open_ratio)

    return combined_score
```
