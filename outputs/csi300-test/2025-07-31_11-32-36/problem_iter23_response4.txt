```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the 5-day moving average of the close price
    df['close_5d_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 10-day moving average of the close price
    df['close_10d_ma'] = df['close'].rolling(window=10).mean()
    
    # Calculate the ratio of the 5-day moving average to the 10-day moving average
    df['ma_ratio'] = df['close_5d_ma'] / df['close_10d_ma']
    
    # Calculate the 5-day rolling standard deviation of the close price
    df['close_5d_std'] = df['close'].rolling(window=5).std()
    
    # Calculate the volatility-adjusted close price
    df['close_vol_adj'] = df['close'] / df['close_5d_std']
    
    # Calculate the 5-day rolling sum of volume
    df['volume_5d_sum'] = df['volume'].rolling(window=5).sum()
    
    # Calculate the 5-day rolling sum of amount
    df['amount_5d_sum'] = df['amount'].rolling(window=5).sum()
    
    # Calculate the liquidity measure (volume/amount)
    df['liquidity_measure'] = df['volume_5d_sum'] / df['amount_5d_sum']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['ma_ratio'] * df['close_vol_adj'] * df['liquidity_measure'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
