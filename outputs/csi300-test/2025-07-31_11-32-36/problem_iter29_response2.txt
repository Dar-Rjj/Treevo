```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] / df['close'].shift(7) - 1).fillna(0)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] / df['close'].shift(25) - 1).fillna(0)
    
    # Weighted Momentum Combination
    weighted_momentum = (short_term_momentum * 0.5 + long_term_momentum * 0.5)
    
    # Combine Intraday Range and Momentum
    combined_factor = intraday_range * weighted_momentum
    
    # Adjust for Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_factor = combined_factor * volume_ratio.fillna(1)
    
    # Simple Moving Average of Volume Over 20 Days
    sma_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Volume Ratio
    volume_sma_ratio = df['volume'] / sma_volume_20
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_factor * volume_sma_ratio.fillna(1)
    
    return final_alpha_factor
```
