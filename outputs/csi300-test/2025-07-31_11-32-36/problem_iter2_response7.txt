```python
import pandas as pd

def heuristics_v2(df, lookback_window=20):
    # Calculate Price Momentum
    df['Daily_Return'] = df['close'].pct_change()
    df['Average_Daily_Return'] = df['Daily_Return'].rolling(window=lookback_window).mean()

    # Calculate Volume Trend
    df['Volume_Difference'] = df['volume'].diff()
    df['Rolling_Average_Volume_Difference'] = df['Volume_Difference'].rolling(window=lookback_window).mean()
    df['Price_Momentum_with_Volume'] = df['Average_Daily_Return'] * df['Rolling_Average_Volume_Difference']

    # Calculate Intraday Momentum Score
    df['High_Low_Delta'] = df['high'] - df['low']
    df['Close_Open_Delta'] = df['close'] - df['open']
    df['Sum_of_Deltas'] = df['High_Low_Delta'] + df['Close_Open_Delta']
    df['Intraday_Momentum_Score'] = df['Sum_of_Deltas'] * df['volume']

    # Calculate Intraday Volatility Score
    df['High_Close_Delta'] = df['high'] - df['close']
    df['Low_Close_Delta'] = df['low'] - df['close']
    df['Average_of_Deltas'] = (df['High_Close_Delta'] + df['Low_Close_Delta']) / 2
    df['Intraday_Volatility_Score'] = df['Average_of_Deltas'] * df['amount']

    # Combine Scores
    df['Combined_Score'] = df['Intraday_Momentum_Score'] - df['Intraday_Volatility_Score']
    df['Final_Alpha_Factor'] = df['Combined_Score'] * df['Price_Momentum_with_Volume']

    return df['Final_Alpha_Factor']
```
