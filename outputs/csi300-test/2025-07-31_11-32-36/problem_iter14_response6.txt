```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the money flow
    money_flow_ma = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the volume
    volume_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate the ratio of the money flow to its moving average
    money_flow_ratio = money_flow / money_flow_ma
    
    # Calculate the ratio of the volume to its moving average
    volume_ratio = df['volume'] / volume_ma
    
    # Calculate the final factor as the product of the two ratios
    factor = money_flow_ratio * volume_ratio
    
    return factor
```
