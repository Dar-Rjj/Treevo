```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['return'] = df['close'].pct_change()
    
    # Apply Exponential Moving Average (EMA) to Returns
    ema_window = 20
    df['ema_return'] = df['return'].ewm(span=ema_window, adjust=False).mean()
    
    # Compute EMA of Squared Returns for Volatility
    df['squared_return'] = df['return'] ** 2
    df['ema_volatility'] = df['squared_return'].ewm(span=ema_window, adjust=False).mean()
    
    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = (df['ema_volatility'] - df['ema_return']).abs()
    
    # Weight Divergence by Volume and High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['weighted_divergence'] = (df['divergence'] * df['volume']) / df['high_low_range']
    
    # Calculate Intraday Movement Factor
    df['intraday_spread'] = df['high'] - df['low']
    df['prev_true_range'] = df[['high', 'low', 'close']].shift().max(axis=1) - df[['high', 'low', 'close']].shift().min(axis=1)
    df['intraday_factor'] = df['intraday_spread'] / df['prev_true_range']
    
    # Integrate Volume Impact
    df['volume_change'] = df['volume'] - df['volume'].shift()
    df['combined_factor'] = df['weighted_divergence'] * df['intraday_factor'] * df['volume_change']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['return'].rolling(window=ema_window).mean()
    
    # Volume Trend
    df['volume_diff'] = df['volume'] - df['volume'].shift()
    df['volume_trend'] = df['volume_diff'].rolling(window=ema_window).mean()
    
    # Synthesize Initial Alpha Factor
    df['initial_alpha'] = df['price_momentum'] * df['volume_trend'] * df['intraday_factor'] * df['volume_change']
    
    # Final Alpha Factor
    df['final_alpha'] = df['combined_factor'] * df['volume_change'] * df['initial_alpha']
    
    return df['final_alpha']
```
