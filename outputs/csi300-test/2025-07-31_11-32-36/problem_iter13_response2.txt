```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['range'] = df['high'] - df['low']
    df['avg_range_20'] = df['range'].rolling(window=20).mean()

    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Apply Exponential Moving Average (EMA) to Returns
    df['ema_return_20'] = df['daily_return'].ewm(span=20, adjust=False).mean()

    # Compute EMA of Squared Returns for Volatility
    df['squared_returns'] = df['daily_return'] ** 2
    df['ema_volatility_20'] = df['squared_returns'].ewm(span=20, adjust=False).mean()

    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = abs(df['ema_return_20'] - df['ema_volatility_20'])

    # Weight Divergence by Volume and High-Low Range
    df['weighted_divergence'] = (df['divergence'] * df['volume']) / df['range']

    # Calculate Intraday Movement Factor
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_true_range'] = df['high'].shift(1) - df['low'].shift(1)
    df['intraday_movement_factor'] = df['intraday_high_low_spread'] / df['prev_true_range']

    # Integrate Volume Impact
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Combine Weighted Divergence and Intraday Factors
    df['weighted_intraday_factor'] = (df['weighted_divergence'] + df['intraday_movement_factor']) * df['volume_change']

    # Calculate Price Momentum
    df['avg_daily_return_20'] = df['daily_return'].rolling(window=20).mean()

    # Volume Trend
    df['volume_diff_20'] = df['volume_change'].rolling(window=20).mean()

    # Synthesize Initial Alpha Factor
    df['initial_alpha_factor'] = df['avg_daily_return_20'] * df['volume_diff_20'] * df['intraday_movement_factor'] * df['volume_change']

    # Final Alpha Factor
    df['final_alpha_factor'] = df['weighted_intraday_factor'] * df['initial_alpha_factor']

    return df['final_alpha_factor'].dropna()
```
