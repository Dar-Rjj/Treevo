```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Average Intraday Range Over N Days
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Compute Close Price Returns
    close_returns = df['close'].pct_change()
    
    # Combine Intraday Range and Momentum
    combined_factor = avg_intraday_range * close_returns
    
    # Adjust for Volume
    ema_volume = df['volume'].ewm(span=n, adjust=False).mean()
    adjusted_factor = combined_factor / ema_volume
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Average Daily Return Over the Lookback Window
    avg_daily_return = daily_return.rolling(window=m).mean()
    
    # Calculate Cumulative Return Over M Days
    cumulative_return = daily_return.rolling(window=m).sum()
    
    # Adjust Combined Intraday Range and Momentum
    adjusted_combined_factor = avg_intraday_range * cumulative_return
    
    # Calculate Volume Difference
    volume_diff = df['volume'].diff()
    
    # Rolling Average Volume Difference
    rolling_avg_volume_diff = volume_diff.rolling(window=m).mean()
    
    # Multiply Price Momentum by Volume Trend
    price_momentum_volume_trend = avg_daily_return * rolling_avg_volume_diff
    
    # Calculate Intraday Price Range Ratio
    intraday_price_range_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Close
    volume_adjusted_close = df['close'] * np.sqrt(df['volume'])
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-Day Price Momentum
    five_day_momentum = df['close'].diff().rolling(window=5).sum()
    
    # Calculate Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Amplitude Confirmation
    amplitude = df['high'] - df['low']
    amplitude_confirmation = (amplitude > amplitude.shift(1)).astype(int)
    
    # Adjust for Volatility
    true_range = pd.DataFrame({
        'h-l': df['high'] - df['low'],
        'h-c_prev': abs(df['high'] - df['close'].shift(1)),
        'l-c_prev': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    atr_5 = true_range.rolling(window=5).mean()
    volatility_adjustment = (volume_confirmation + amplitude_confirmation) / atr_5
    
    # Introduce Trend Strength Confirmation
    sma_20 = df['close'].rolling(window=20).mean()
    trend_strength_confirmation = (df['close'] > sma_20).astype(int)
    
    # Final Combined Alpha Factor
    final_factor = (price_momentum_volume_trend 
                    + adjusted_combined_factor 
                    + intraday_price_range_ratio 
                    + volume_adjusted_close 
                    + intraday_return 
                    + five_day_momentum 
                    + volume_confirmation 
                    + amplitude_confirmation 
                    + volatility_adjustment 
                    + trend_strength_confirmation)
    
    return final_factor
```
