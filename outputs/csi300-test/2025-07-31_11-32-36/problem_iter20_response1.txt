import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor from the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as (date).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the intraday return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate the daily volatility
    daily_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted average price
    vwap = (df['amount'] / df['volume']).rolling(window=20).mean()
    
    # Calculate the 20-day moving average of the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day standard deviation of the closing price
    std_20 = df['close'].rolling(window=20).std()
    
    # Calculate the normalized intraday return
    norm_intraday_return = (df['close'] - ma_20) / std_20
    
    # Calculate the alpha factor
    alpha_factor = intraday_return * daily_volatility * (vwap - ma_20) / std_20
    
    return alpha_factor
