```python
import pandas as pd

def heuristics_v2(df, n_days=20):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Momentum
    momentum = df['close'] / df['close'].shift(1) - 1
    
    # Combine Intraday Range and Momentum
    combined_factor = intraday_range * momentum
    
    # Adjust for Volume
    ema_volume = df['volume'].ewm(span=n_days, adjust=False).mean()
    adjusted_factor = combined_factor / ema_volume
    
    return adjusted_factor
```
