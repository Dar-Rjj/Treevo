```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily close-to-close return
    df['daily_return'] = df['close'].pct_change()

    # Apply Exponential Moving Average (EMA) to returns with a lookback period of 20 days
    df['ema_returns'] = df['daily_return'].ewm(span=20, adjust=False).mean()

    # Compute EMA of squared returns for volatility
    df['squared_returns'] = df['daily_return'] ** 2
    df['ema_volatility'] = df['squared_returns'].ewm(span=20, adjust=False).mean()

    # Calculate divergence between EMA of returns and EMA of volatility
    df['divergence'] = np.abs(df['ema_returns'] - df['ema_volatility'])

    # Calculate high-low range
    df['high_low_range'] = df['high'] - df['low']

    # Weight divergence by volume and high-low range
    df['weighted_divergence'] = (df['divergence'] * df['volume']) / df['high_low_range']

    # Calculate intraday movement factor
    df['intraday_spread'] = df['high'] - df['low']
    df['prev_true_range'] = (df['high'].shift(1) - df['low'].shift(1)).fillna(0)
    df['intraday_movement_factor'] = df['intraday_spread'] / df['prev_true_range']

    # Calculate close price direction
    df['close_direction'] = df['close'] - df['open']

    # Integrate volume impact
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Combine weighted divergence and intraday factors
    df['adjusted_factor'] = df['weighted_divergence'] + df['intraday_movement_factor']

    # Final alpha factor
    df['alpha_factor'] = df['adjusted_factor'] * df['volume_change']

    return df['alpha_factor']
```
