```python
import pandas as pd

def heuristics(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Average Daily Return over 5-day lookback window
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Rolling Average Volume Difference
    df['volume_diff'] = df['volume'].diff()
    df['avg_volume_diff'] = df['volume_diff'].rolling(window=5).mean()
    
    # Volume and Amplitude Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) * 0.5
    
    # Price Momentum by Volume and Amplitude Trend
    df['price_momentum_vol_amp'] = df['avg_daily_return'] * df['avg_volume_diff'] * df['combined_confirmation']
    
    # Compute Intraday Momentum Score
    df['intraday_high_low_range'] = df['high'] - df['low']
    df['close_to_open_diff'] = df['close'] - df['open']
    df['sum_of_deltas'] = df['intraday_high_low_range'] + df['close_to_open_diff']
    df['intraday_momentum_score'] = df['volume'] * df['sum_of_deltas']
    
    # Calculate Volume Weighted Price Change
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['volume_weighted_open'] = df['open'] * df['volume']
    df['volume_weighted_price'] = (df['volume_weighted_close'] + df['volume_weighted_open']) / df['volume']
    df['volume_weighted_price_change'] = df['volume_weighted_price'] - df['volume_weighted_price'].shift(1)
    
    # Combined Intraday and Daily Momentum
    df['combined_intraday_daily_momentum'] = df['intraday_momentum_score'] + df['volume_weighted_price_change']
    
    # Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Price Momentum
    df['short_term_momentum'] = (df['close'] / df['close'].shift(7) - 1)
    df['long_term_momentum'] = (df['close'] / df['close'].shift(25) - 1)
    df['weighted_momentum_combination'] = (df['short_term_momentum'] * 0.5 + df['long_term_momentum'] * 0.5)
    
    # Combine Intraday Volatility and Momentum
    df['combined_vol_mom'] = df['intraday_volatility'] * df['weighted_momentum_combination']
    
    # Adjust for Volume
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['combined_vol_mom_adj_volume'] = df['combined_vol_mom'] / df['sma_volume_20']
    
    # Incorporate Trading Volume Impact
    df['volume_ratio'] = df['volume'] / df['sma_volume_20']
    df['final_alpha_factor'] = df['combined_vol_mom_adj_volume'] * df['volume_ratio']
    
    # Integrate Composite Momentum and Volatility
    df['ema_30_combined_momentum'] = df['combined_intraday_daily_momentum'].ewm(span=30, adjust=False).mean()
    df['composite_mom_vol'] = df['combined_intraday_daily_momentum'] - df['ema_30_combined_momentum'] - df['intraday_volatility']
    
    return df['composite_mom_vol']
```
