import pandas as pd

def heuristics_v2(df, n_days=10, m_days=30, k_days=20):
    """
    Calculate a novel and interpretable alpha factor that combines intraday volatility and momentum.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    - n_days: Number of days to calculate the average intraday range.
    - m_days: Number of days to calculate the cumulative return for momentum.
    - k_days: Number of days to calculate the average trading volume.
    
    Returns:
    - pandas Series indexed by (date) representing the factor values.
    """
    # Calculate Intraday Volatility
    df['range'] = df['high'] - df['low']
    df['avg_range'] = df['range'].rolling(window=n_days).mean()
    
    # Calculate Momentum from Close Price
    df['daily_return'] = df['close'].pct_change()
    df['cumulative_return'] = (df['daily_return'] + 1).rolling(window=m_days).apply(lambda x: x.prod() - 1)
    
    # Combine Factors
    df['raw_combined_factor'] = df['avg_range'] * df['cumulative_return']
    
    # Adjust for Trading Volume
    df['avg_volume'] = df['volume'].rolling(window=k_days).mean()
    df['factor'] = df['raw_combined_factor'] / df['avg_volume']
    
    return df['factor'].dropna()
