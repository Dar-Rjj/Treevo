```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the provided market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A pandas Series indexed by (date) representing the factor values.
    """
    
    # Calculate the relative range (high - low) / open
    relative_range = (df['high'] - df['low']) / df['open']
    
    # Calculate the relative close (close - open) / open
    relative_close = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume weighted close
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate the 10-day moving average of the volume weighted close
    volume_weighted_close_10d_ma = volume_weighted_close.rolling(window=10).mean()
    
    # Calculate the 5-day rolling sum of the relative range
    relative_range_5d_sum = relative_range.rolling(window=5).sum()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (relative_close + relative_range_5d_sum) / volume_weighted_close_10d_ma
    
    return alpha_factor
```
