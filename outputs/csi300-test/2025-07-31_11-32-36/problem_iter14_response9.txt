```python
import pandas as pd
import numpy as np

def heuristics(df, N=20, M=5):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Average Intraday Range
    avg_intraday_range = intraday_range.rolling(window=N).mean()
    
    # Compute Close Price Returns
    close_returns = df['close'].pct_change()
    
    # Calculate Cumulative Return
    cumulative_return = close_returns.rolling(window=M).sum()
    
    # Combine Intraday Range and Momentum
    combined_factor = avg_intraday_range * cumulative_return
    
    # Adjust for Volume
    ema_volume = df['volume'].ewm(span=N, adjust=False).mean()
    adjusted_factor = combined_factor / ema_volume
    
    # Calculate Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Calculate 5-Day Price Momentum
    price_momentum = daily_price_change.rolling(window=5).sum()
    
    # Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(2)).astype(int)
    
    # Amplitude Confirmation
    daily_amplitude = df['high'] - df['low']
    avg_amplitude_5_days = daily_amplitude.rolling(window=5).mean()
    amplitude_confirmation = (daily_amplitude > avg_amplitude_5_days).astype(int)
    
    # Combined Volume and Amplitude Momentum Factor
    combined_volume_amplitude_momentum = price_momentum * (volume_confirmation + amplitude_confirmation) / 2
    
    # Daily Return Confirmation
    avg_close_return_5_days = close_returns.rolling(window=5).mean()
    return_confirmation = (close_returns > avg_close_return_5_days).astype(int)
    
    # Final Alpha Factor
    alpha_factor = (adjusted_factor + combined_volume_amplitude_momentum + return_confirmation) / 3
    
    return alpha_factor
```
