import pandas as pd

def heuristics(df, lookback_window=20):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate average daily return over the lookback window
    df['average_daily_return'] = df['daily_return'].rolling(window=lookback_window).mean()
    
    # Calculate volume difference
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate rolling average volume difference over the lookback window
    df['rolling_avg_volume_difference'] = df['volume_difference'].rolling(window=lookback_window).mean()
    
    # Multiply price momentum by volume trend
    df['factor'] = df['average_daily_return'] * df['rolling_avg_volume_difference']
    
    # Return the factor as a Series
    return df['factor']
