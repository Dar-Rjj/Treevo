```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Calculate 10-Day Moving Average of Daily Returns (excluding the most recent return)
    df['10_day_moving_avg'] = df['daily_returns'].shift(1).rolling(window=10, min_periods=1).mean()
    
    # Compute 20-Day Exponential Moving Average of Volume
    df['20_day_ema_volume'] = df['volume'].ewm(span=20, min_periods=1).mean()
    
    # Adjust for Volume Trend and calculate Weighted Momentum
    df['weighted_momentum'] = (df['10_day_moving_avg'] * df['volume']) / df['20_day_ema_volume']
    
    # Calculate 10-Day Standard Deviation of Daily Returns (excluding the most recent return)
    df['10_day_std_dev'] = df['daily_returns'].shift(1).rolling(window=10, min_periods=1).std()
    
    # Inverse of Volatility
    small_constant = 1e-8  # To avoid division by zero
    df['inverse_volatility'] = 1 / (df['10_day_std_dev'] + small_constant)
    
    # Sum Weighted Momentum and Inverse of Volatility
    df['factor'] = df['weighted_momentum'] + df['inverse_volatility']
    
    return df['factor']
```
