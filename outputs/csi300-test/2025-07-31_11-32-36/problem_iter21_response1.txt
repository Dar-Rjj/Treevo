```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    
    # Calculate Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(5)) - 1
    long_term_momentum = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted Momentum Combination
    weighted_momentum = (short_term_momentum * 0.7) + (long_term_momentum * 0.3)
    
    # Combine Intraday Volatility and Momentum
    combined_factor = intraday_volatility * weighted_momentum
    
    # Adjust for Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_factor = combined_factor / ema_volume
    
    return adjusted_factor
```
