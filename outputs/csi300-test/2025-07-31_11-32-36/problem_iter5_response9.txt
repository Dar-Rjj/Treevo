```python
import pandas as pd

def heuristics_v2(df, N=10, M=5, K=10):
    # Calculate Intraday High-Low Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=N).mean()
    
    # Calculate Close to Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Compute Intraday Momentum Score
    df['intraday_momentum_score'] = (df['high_low_range'] + df['close_open_diff']) * df['volume']
    
    # Calculate Momentum from Close Price
    df['daily_return'] = df['close'].pct_change()
    df['cumulative_return'] = df['daily_return'].rolling(window=M).sum()
    
    # Combine Intraday and Daily Momentum
    df['raw_combined_momentum'] = df['intraday_momentum_score'] + df['cumulative_return']
    
    # Adjust for Trading Volume
    df['ema_volume'] = df['volume'].ewm(span=K, adjust=False).mean()
    df['adjusted_combined_momentum'] = df['raw_combined_momentum'] / df['ema_volume']
    
    # Calculate Volume Weighted Price
    df['vwp_close'] = df['close'] * df['volume']
    df['vwp_open'] = df['open'] * df['volume']
    df['total_volume'] = df['volume'].rolling(window=2).sum()
    df['vwp'] = (df['vwp_close'] + df['vwp_open']) / df['total_volume']
    
    # Calculate Volume Weighted Price Change
    df['vwp_change'] = df['vwp'].diff()
    
    # Compute Enhanced Combined Momentum
    df['enhanced_combined_momentum'] = df['adjusted_combined_momentum'] + df['vwp_change']
    
    # Calculate Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['avg_deltas'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['intraday_volatility_score'] = df['avg_deltas'] * df['amount']
    
    # Integrate Composite Momentum and Volatility
    df['ema_enhanced_momentum'] = df['enhanced_combined_momentum'].ewm(span=30, adjust=False).mean()
    df['alpha_factor'] = df['enhanced_combined_momentum'] - df['ema_enhanced_momentum'] - df['intraday_volatility_score']
    
    return df['alpha_factor']
```
