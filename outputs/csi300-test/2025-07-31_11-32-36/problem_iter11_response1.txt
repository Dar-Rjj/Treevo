```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_window=20, n_days=10, m_days=5):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Average Daily Return
    df['avg_daily_return'] = df['daily_return'].rolling(window=lookback_window).mean()

    # Compute Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Average Intraday Range Over N Days
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n_days).mean()

    # Calculate Cumulative Return Over M Days
    df['cumulative_return'] = df['daily_return'].rolling(window=m_days).sum()

    # Combine Intraday Range and Momentum
    df['combined_factor'] = df['avg_intraday_range'] * df['cumulative_return']

    # Adjust for Volume (EMA of Volume)
    df['ema_volume'] = df['volume'].ewm(span=lookback_window).mean()
    df['adjusted_combined_factor'] = df['combined_factor'] / df['ema_volume']

    # Calculate Volume Difference
    df['volume_diff'] = df['volume'].diff()

    # Rolling Average Volume Difference
    df['rolling_avg_volume_diff'] = df['volume_diff'].rolling(window=lookback_window).mean()

    # Multiply Price Momentum by Volume Trend
    df['price_momentum_volume_trend'] = df['avg_daily_return'] * df['rolling_avg_volume_diff']

    # Calculate Intraday Price Range Ratio
    df['intraday_price_range_ratio'] = (df['high'] - df['low']) / df['low']

    # Calculate Volume Adjusted Close
    df['volume_adjusted_close'] = df['close'] * np.sqrt(df['volume'])

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Final Combined Alpha Factor
    df['alpha_factor'] = (
        df['price_momentum_volume_trend'] +
        df['adjusted_combined_factor'] +
        df['intraday_price_range_ratio'] +
        df['volume_adjusted_close'] +
        df['intraday_return']
    )

    # Drop NaN values
    df.dropna(inplace=True)

    return df['alpha_factor']
```
