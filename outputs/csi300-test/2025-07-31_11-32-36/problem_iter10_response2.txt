```python
import pandas as pd

def heuristics_v2(df, N=10, M=5, volume_window=20):
    # Compute Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Average Intraday Range Over N Days
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=N).mean()
    
    # Compute Close Price Returns
    df['close_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Cumulative Return Over M Days
    df['cumulative_return'] = df['close_return'].rolling(window=M).sum()
    
    # Combine Intraday Range and Momentum
    df['combined_factor'] = df['avg_intraday_range'] * df['cumulative_return']
    
    # EMA of Volume
    df['ema_volume'] = df['volume'].ewm(span=volume_window, adjust=False).mean()
    
    # Adjust for Volume
    df['adjusted_factor'] = df['combined_factor'] / df['ema_volume']
    
    return df['adjusted_factor']
```
