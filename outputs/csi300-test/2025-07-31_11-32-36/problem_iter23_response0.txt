import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on historical and current market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A series of the generated alpha factor values indexed by date.
    """
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily returns
    df['ma_10_day_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day exponential moving average of the closing prices
    df['ema_10_day_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day volatility (standard deviation) of the daily returns
    df['volatility_10_day'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the money flow index (MFI) over 14 days
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    df['mfi_14_day'] = 100 - (100 / (1 + (positive_money_flow_sum / negative_money_flow_sum)))
    
    # Generate the alpha factor
    df['alpha_factor'] = (df['ema_10_day_close'] - df['close']) / df['volatility_10_day'] + df['mfi_14_day'] - df['ma_10_day_return']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
