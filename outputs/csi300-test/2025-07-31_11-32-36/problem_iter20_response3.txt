import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Average Daily Return over 7-day lookback window
    df['avg_daily_return'] = df['daily_return'].rolling(window=7).mean()
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Rolling Average Volume Difference over 7-day lookback window
    df['rolling_avg_volume_diff'] = df['volume_diff'].rolling(window=7).mean()
    
    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Amplitude Confirmation
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)
    
    # Combine Volume and Amplitude Confirmation
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) * 0.5
    
    # Multiply Price Momentum by Volume and Amplitude Trend
    df['price_momentum_with_trend'] = df['avg_daily_return'] * df['rolling_avg_volume_diff'] * df['combined_confirmation']
    
    # Compute Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Price Momentum
    df['short_term_momentum'] = (df['close'] / df['close'].shift(5) - 1)
    df['long_term_momentum'] = (df['close'] / df['close'].shift(20) - 1)
    
    # Weighted Momentum Combination
    df['weighted_momentum'] = df['short_term_momentum'] * 0.6 + df['long_term_momentum'] * 0.4
    
    # Combine Intraday Volatility and Momentum
    df['combined_intraday_daily_momentum'] = df['intraday_volatility'] * df['weighted_momentum']
    
    # Compute 30-day Exponential Moving Average of Combined Intraday and Daily Momentum
    df['ema_30'] = df['combined_intraday_daily_momentum'].ewm(span=30, adjust=False).mean()
    
    # Subtract from Current Combined Intraday and Daily Momentum
    df['enhanced_momentum'] = df['combined_intraday_daily_momentum'] - df['ema_30']
    
    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    df['close_to_open_diff'] = df['close'] - df['open']
    
    # Compute Intraday Momentum Score
    df['intraday_momentum_score'] = df['intraday_high_low_range'] + df['close_to_open_diff']
    
    # Calculate Volume Weighted Price
    df['vwp_close'] = df['close'] * df['volume']
    df['vwp_open'] = df['open'] * df['volume']
    df['vwp'] = (df['vwp_close'] + df['vwp_open']) / df['volume']
    
    # Calculate Volume Weighted Price Change
    df['vwp_change'] = df['vwp'] - df['vwp'].shift(1)
    
    # Compute Combined Intraday and Daily Momentum
    df['combined_intraday_daily_momentum'] += df['vwp_change']

    df['high_close_delta'] = df['high'] - df['close']

    df['std_intraday_high_low'] = df['intraday_high_low_range'].rolling(window=30).std()
