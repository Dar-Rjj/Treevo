import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Average Daily Return over 5-day lookback window
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()

    # Calculate Volume Difference
    df['volume_diff'] = df['volume'].diff()

    # Rolling Average Volume Difference over 5-day lookback window
    df['avg_volume_diff'] = df['volume_diff'].rolling(window=5).mean()

    # Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Amplitude Confirmation
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)

    # Combined Volume and Amplitude Confirmation
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) * 0.5

    # Multiply Price Momentum by Volume and Amplitude Trend
    df['price_momentum_trend'] = df['avg_daily_return'] * df['avg_volume_diff'] * df['combined_confirmation']

    # Intraday Momentum Score
    df['high_low_delta'] = df['high'] - df['low']
    df['close_open_delta'] = df['close'] - df['open']
    df['sum_of_deltas'] = df['high_low_delta'] + df['close_open_delta']
    df['MAV'] = df['volume'].rolling(window=10).mean()
    df['weighted_momentum'] = df['sum_of_deltas'] * (df['volume'] / df['MAV'])

    # Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['avg_of_deltas'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['MAA'] = df['amount'].rolling(window=10).mean()
    df['weighted_volatility'] = df['avg_of_deltas'] * (df['amount'] / df['MAA'])

    # Price Trend Factor
    df['close_open_ratio'] = df['close'] / df['open']
    df['smoothed_ratio'] = df['close_open_ratio'].ewm(span=10, adjust=False).mean()

    # Daily Returns
    df['daily_returns'] = df['daily_return']

    # EMA Returns and EMA Squared Returns
    df['EMA_returns'] = df['daily_returns'].ewm(span=10, adjust=False).mean()
    df['EMA_squared_returns'] = (df['daily_returns'] ** 2).ewm(span=10, adjust=False).mean()

    # Divergence
    df['divergence'] = np.abs(df['EMA_returns'] - df['EMA_squared_returns'])

    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Weighted Divergence
    df['weighted_divergence'] = df['divergence'] * (df['volume'] / df['high_low_range'])

    # Total Momentum
    df['total_momentum'] = df['weighted_momentum'] + df['smoothed_ratio']

    # Net Score
    df['net_score'] = df['total_momentum'] - df['weighted_volatility'] + df['weighted_divergence']

    df['volume_weighted_price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / df['volume'].rolling(window=10).sum()

    # Volume Weighted Price Change
    df['volume_weighted_price_change'] = df['volume_weighted_price'].diff()
