```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A series of the computed alpha factor indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the average amount over the past 5 days
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    
    # Calculate the volume weighted typical price
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate the momentum of the vwap
    momentum_vwap = vwap.pct_change(periods=14)
    
    # Calculate the standard deviation of the typical price over the past 20 days
    std_typical_price_20d = typical_price.rolling(window=20).std()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (momentum_vwap * std_typical_price_20d) / (avg_amount_5d + 1e-6)
    
    return alpha_factor
```
