```python
import pandas as pd

def heuristics_v2(df, lookback_window=10):
    # Calculate Price Momentum
    df['daily_return'] = df['close'].pct_change()
    df['average_daily_return'] = df['daily_return'].rolling(window=lookback_window).mean()

    # Volume Trend
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    df['rolling_average_volume_difference'] = df['volume_difference'].rolling(window=lookback_window).mean()

    # Intraday Movement Factor
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['previous_day_true_range'] = df['high'].shift(1) - df['low'].shift(1)
    df['intraday_movement_factor'] = df['intraday_high_low_spread'] / df['previous_day_true_range']

    # Close Price Direction
    df['close_price_direction'] = df['close'] - df['open']

    # Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Weighted Combination
    intraday_weight = 0.6
    close_direction_weight = 0.4
    df['weighted_combination'] = (intraday_weight * df['intraday_movement_factor']) + (close_direction_weight * df['close_price_direction'])

    # Synthesized Alpha Factor
    df['synthesized_alpha_factor'] = (df['average_daily_return'] * df['rolling_average_volume_difference'] * df['weighted_combination'] * df['volume_change'])

    return df['synthesized_alpha_factor']
```
