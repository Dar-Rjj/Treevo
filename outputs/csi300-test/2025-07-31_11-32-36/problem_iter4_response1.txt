import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Weight by Average Volume
    df['avg_volume'] = (df['volume'] + df['volume'].shift(1)) / 2
    df['intraday_momentum'] = df['intraday_return'] * df['avg_volume']
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust for Intraday Volatility
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    df['adjusted_volume_thrust'] = df['volume_ratio'] * df['intraday_range']
    
    # Combine Intraday Momentum and Volume Thrust
    df['indicator'] = df['intraday_momentum'] * df['adjusted_volume_thrust']
    
    # Smooth the Indicator
    window_size = 5
    df['smoothed_indicator'] = df['indicator'].rolling(window=window_size).mean()
    
    return df['smoothed_indicator']
