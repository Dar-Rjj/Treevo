import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The alpha factor is a weighted combination of:
    - Normalized volume change
    - Normalized amount change
    - Normalized high-low range
    - Normalized close-open range
    - Momentum (10-day moving average of returns)
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    
    Returns:
    pd.Series: Alpha factor values
    """
    # Calculate normalized volume change
    df['volume_change'] = df['volume'].pct_change()
    df['normalized_volume_change'] = (df['volume_change'] - df['volume_change'].rolling(window=20).mean()) / df['volume_change'].rolling(window=20).std()
    
    # Calculate normalized amount change
    df['amount_change'] = df['amount'].pct_change()
    df['normalized_amount_change'] = (df['amount_change'] - df['amount_change'].rolling(window=20).mean()) / df['amount_change'].rolling(window=20).std()
    
    # Calculate normalized high-low range
    df['high_low_range'] = (df['high'] - df['low']) / df['close']
    df['normalized_high_low_range'] = (df['high_low_range'] - df['high_low_range'].rolling(window=20).mean()) / df['high_low_range'].rolling(window=20).std()
    
    # Calculate normalized close-open range
    df['close_open_range'] = (df['close'] - df['open']) / df['close']
    df['normalized_close_open_range'] = (df['close_open_range'] - df['close_open_range'].rolling(window=20).mean()) / df['close_open_range'].rolling(window=20).std()
    
    # Calculate momentum (10-day moving average of returns)
    df['return'] = df['close'].pct_change()
    df['momentum'] = df['return'].rolling(window=10).mean()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        0.3 * df['normalized_volume_change'] +
        0.2 * df['normalized_amount_change'] +
        0.2 * df['normalized_high_low_range'] +
        0.2 * df['normalized_close_open_range'] +
        0.1 * df['momentum']
    )
    
    return df['alpha_factor']
