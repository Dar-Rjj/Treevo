```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate 5-Day Price Momentum
    df['5_day_momentum'] = df['daily_price_change'].rolling(window=5).sum()
    
    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(2)).astype(int)
    
    # Calculate Amplitude Confirmation
    df['daily_amplitude'] = df['high'] - df['low']
    df['average_amplitude_5_days'] = df['daily_amplitude'].rolling(window=5).mean()
    df['amplitude_confirmation'] = (df['daily_amplitude'] > df['average_amplitude_5_days']).astype(int)
    
    # Combine Volume and Amplitude Confirmation
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) / 2
    
    # Final Alpha Factor
    df['alpha_factor'] = df['5_day_momentum'] * df['combined_confirmation']
    
    return df['alpha_factor']
```
