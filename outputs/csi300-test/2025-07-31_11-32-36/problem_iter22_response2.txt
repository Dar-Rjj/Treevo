import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate daily return
    df['daily_return'] = df['close'] / df['open']
    
    # Analyze the trend of closing over opening price for the past n days
    df['trend_close_over_open'] = df['daily_return'].rolling(window=n).mean()
    
    # Compute the average true range (ATR) over the last n days
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x[0] - x[1], abs(x[0] - df['prev_close']), abs(x[1] - df['prev_close'])), axis=1)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Determine the daily volatility as (high - low)
    df['daily_volatility'] = df['high'] - df['low']
    
    # Calculate the on-balance volume (OBV) incrementally
    df['obv'] = 0
    obv_values = []
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            obv_values.append(obv_values[-1] + df.loc[df.index[i], 'volume'])
        elif df.loc[df.index[i], 'close'] < df.loc[df.index[i-1], 'close']:
            obv_values.append(obv_values[-1] - df.loc[df.index[i], 'volume'])
        else:
            obv_values.append(obv_values[-1])
    df['obv'][1:] = obv_values
    
    # Examine the ratio of current day's volume to the n-day moving average volume
    df['vol_ratio'] = df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Identify significant volume changes compared to the prior day
    df['vol_change'] = df['volume'].pct_change()
    
    # Create a factor that combines price movement and volume surge
    df['price_vol_factor'] = df['close'].pct_change() * df['vol_change']
    
    # Develop a factor that measures the strength of a move
    df['price_strength'] = (df['close'] - df['open']).abs() / df['atr']
    df['price_strength'] *= df['close'].pct_change().apply(lambda x: 1 if x > 0 else -1)
    
    # Construct a factor using the relative position of the close price within the high-low range
    df['close_range'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    df['close_range_trend'] = df['close_range'].rolling(window=n).mean()
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['trend_close_over_open'] +
        df['atr'] +
        df['daily_volatility'] +
        df['obv'] +
        df['vol_ratio'] +
        df['vol_change'] +
        df['price_vol_factor'] +
        df['price_strength'] +
        df['close_range_trend']
    )
    
    return df['alpha_factor']
