```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Average Daily Return over the last 5 days
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()

    # Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)

    # Amplitude Confirmation
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)

    # Combined Volume and Amplitude Confirmation
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) * 0.5

    # Price Momentum by Volume and Amplitude Trend
    df['price_momentum_trend'] = df['avg_daily_return'] * df['combined_confirmation']

    # Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']

    # Close to Open Difference
    df['close_to_open_diff'] = df['close'] - df['open']

    # Sum of Deltas
    df['sum_of_deltas'] = df['intraday_high_low_range'] + df['close_to_open_diff']

    # Weight by Volume
    df['intraday_momentum_score'] = df['sum_of_deltas'] * df['volume']

    # Volume Weighted Price
    df['volume_weighted_price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / df['volume']

    # Volume Weighted Price Change
    df['volume_weighted_price_change'] = df['volume_weighted_price'] - df['volume_weighted_price'].shift(1)

    # Combined Intraday and Daily Momentum
    df['combined_intraday_daily_momentum'] = df['intraday_momentum_score'] + df['volume_weighted_price_change']

    # Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['avg_of_deltas'] = (df['high_close_delta'] + df['low_close_delta'].abs()) / 2
    df['std_intraday_high_low_range'] = df['intraday_high_low_range'].rolling(window=30).std()
    df['enhanced_intraday_volatility_score'] = df['avg_of_deltas'] * df['std_intraday_high_low_range']

    # 30-day Exponential Moving Average of Combined Intraday and Daily Momentum
    df['ema_30_intraday_daily_momentum'] = df['combined_intraday_daily_momentum'].ewm(span=30).mean()
    df['adjusted_combined_momentum'] = df['combined_intraday_daily_momentum'] - df['ema_30_intraday_daily_momentum'] - df['enhanced_intraday_volatility_score']

    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['price_momentum_trend'] +
        df['intraday_high_low_range'] +
        df['intraday_momentum_score'] +
        df['combined_intraday_daily_momentum'] +
        df['enhanced_intraday_volatility_score'] +
        df['volume_weighted_price_change']
    ) * (df['volume'] * df['close'])

    return df['final_alpha_factor']
```
