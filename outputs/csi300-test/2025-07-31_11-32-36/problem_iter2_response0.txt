```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback=20):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Average Daily Return over the lookback window
    df['avg_daily_return'] = df['daily_return'].rolling(window=lookback).mean()
    
    # Apply Exponential Moving Average (EMA) to Returns
    df['ema_returns'] = df['daily_return'].ewm(span=lookback, adjust=False).mean()
    
    # Compute EMA of Squared Returns for Volatility
    df['squared_returns'] = df['daily_return'] ** 2
    df['ema_volatility'] = df['squared_returns'].ewm(span=lookback, adjust=False).mean()
    
    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = (df['ema_volatility'] - df['ema_returns']).abs()
    
    # Volume Trend
    df['volume_diff'] = df['volume'].diff()
    df['avg_volume_diff'] = df['volume_diff'].rolling(window=lookback).mean()
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Weight by Volume and High-Low Range
    df['factor'] = df['divergence'] * df['avg_volume_diff'] / df['high_low_range']
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    return df['factor']
```
