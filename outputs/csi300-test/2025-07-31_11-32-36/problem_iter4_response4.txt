```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate 5-Day Price Momentum
    df['5_day_momentum'] = df['daily_price_change'].rolling(window=5).sum()
    df['5_day_momentum'] = df['5_day_momentum'].replace(0, 1)  # Ensure sum is non-zero
    
    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate 5-Day Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['5_day_atr'] = df['true_range'].rolling(window=5).mean()
    
    # Adjust for Volatility
    df['factor'] = (df['5_day_momentum'] * df['volume_confirmation']) / df['5_day_atr']
    
    return df['factor']
```
