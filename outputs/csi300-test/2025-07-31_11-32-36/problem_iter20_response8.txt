import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    
    # Calculate Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(5)) - 1
    long_term_momentum = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted Momentum Combination
    weighted_momentum = short_term_momentum * 0.6 + long_term_momentum * 0.4
    
    # Combine Intraday Volatility and Momentum
    combined_factor = intraday_volatility * weighted_momentum
    
    # Adjust for Volume
    sma_volume = df['volume'].rolling(window=20).mean()
    combined_factor_adjusted = combined_factor / sma_volume
    
    # Incorporate Trading Volume Impact
    volume_ratio = df['volume'] / sma_volume
    final_alpha_factor = combined_factor * volume_ratio
    
    return final_alpha_factor
