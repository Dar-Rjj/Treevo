import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Apply Exponential Moving Average (EMA) to Returns
    ema_window_returns = 20
    df['ema_returns'] = df['daily_return'].ewm(span=ema_window_returns, min_periods=ema_window_returns).mean()

    # Compute EMA of Squared Returns for Volatility
    df['squared_returns'] = df['daily_return'] ** 2
    df['ema_volatility'] = df['squared_returns'].ewm(span=ema_window_returns, min_periods=ema_window_returns).mean()

    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = (df['ema_volatility'] - df['ema_returns']).abs()

    # Weight Divergence by Volume and High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['weighted_divergence'] = (df['divergence'] * df['volume']) / df['high_low_range']

    # Calculate Intraday Volatility
    intraday_volatility_window = 10
    df['intraday_volatility'] = (df['high'] - df['low']).rolling(window=intraday_volatility_window, min_periods=intraday_volatility_window).mean()

    # Calculate Intraday Momentum
    df['intraday_return'] = (df['high'] - df['low']) / (df['open'] - df['close'])
    df['intraday_momentum'] = df['intraday_return'] + (df['close'] - df['open']) / (df['close'] - df['open'])

    # Calculate Volume Weighted Price
    df['vwp'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] + df['volume'])

    # Calculate Volume Weighted Price Change
    df['vwp_change'] = df['vwp'].diff()

    # Compute Combined Intraday and Daily Momentum
    df['combined_momentum'] = df['intraday_momentum'] + df['vwp_change']

    # Integrate Momentum and Reversal
    ema_window_momentum = 30
    df['integrated_momentum'] = df['combined_momentum'] - df['combined_momentum'].ewm(span=ema_window_momentum, min_periods=ema_window_momentum).mean()

    # Combine Initial Factors
    df['raw_combined_factor'] = df['intraday_volatility'] * df['integrated_momentum']

    # Adjust for Trading Volume
    avg_volume_window = 20
    df['avg_volume'] = df['volume'].rolling(window=avg_volume_window, min_periods=avg_volume_window).mean()
    df['adjusted_factor'] = df['raw_combined_factor'] / df['avg_volume']

    # Calculate Volume Thrust
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['intraday_range'] = df['high'] - df['low']
    df['volume_thrust'] = df['volume_ratio'] * (df['intraday_range'] / df['close'])

    # Introduce Trend Filter
    sma_window_trend = 50
    df['sma_close'] = df['close'].rolling(window=sma_window_trend, min_periods=sma_window_trend).mean()
    df['trend'] = np.where(df['close'] > df['sma_close'], 1.2, 0.8)

    # Final Combination and Smoothing
    df['final_factor'] = df['adjusted_factor'] * df['weighted_divergence'] * df['volume_thrust'] * df['trend']
    smoothing_window = 5
    df['smoothed_factor'] = df['final_factor'].rolling(window=smoothing_window, min_periods=smoothing_window).mean()

    # Additional Intraday Momentum Component
    df['intraday_high_low_return'] = (df['high'] - df['low']) / (df['high'] - df['low'])
    df['intraday_momentum'] += df['intraday_high_low_return']

    # Introduce Volume Thrust Adjusted by Close-Open Gap
    df['close_open_gap'] = df['close'] - df['open']
    df['adjusted_volume_thrust'] = df['volume_thrust'] / df['close_open_gap']

    # Update the final factor with the adjusted volume thrust
    df['final_factor'] = df['adjusted_factor'] * df['weighted_divergence']
