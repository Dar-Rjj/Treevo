```python
import pandas as pd
import numpy as np

def heuristics_v2(df, ema_span=10, lookback_window=20):
    # Intraday Momentum Score
    high_low_delta = df['high'] - df['low']
    close_open_delta = df['close'] - df['open']
    sum_of_deltas = high_low_delta + close_open_delta
    mav = df['volume'].rolling(window=lookback_window).mean()
    weighted_momentum = sum_of_deltas * (df['volume'] / mav)

    # Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    average_of_deltas = (high_close_delta + low_close_delta) / 2
    maa = df['amount'].rolling(window=lookback_window).mean()
    weighted_volatility = average_of_deltas * (df['amount'] / maa)

    # Price Trend Factor
    close_open_ratio = df['close'] / df['open']
    smoothed_ratio = close_open_ratio.ewm(span=ema_span, adjust=False).mean()

    # Daily Momentum and Volatility Components
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ema_returns = daily_returns.ewm(span=ema_span, adjust=False).mean()
    ema_squared_returns = (daily_returns ** 2).ewm(span=ema_span, adjust=False).mean()
    divergence = np.abs(ema_returns - ema_squared_returns)
    high_low_range = df['high'] - df['low']
    weighted_divergence = divergence * (df['volume'] / high_low_range)

    # Combined Scores
    total_momentum = weighted_momentum + smoothed_ratio
    net_score = total_momentum - weighted_volatility + weighted_divergence

    # Integrated Momentum
    volume_weighted_price = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])
    volume_weighted_price_change = volume_weighted_price - volume_weighted_price.shift(1)
    integrated_momentum = (weighted_momentum + volume_weighted_price_change).ewm(span=30, adjust=False).mean()

    # Volatility Ratio
    long_term_volatility = ema_squared_returns.ewm(span=50, adjust=False).mean()
    short_term_volatility = ema_squared_returns.ewm(span=10, adjust=False).mean()
    volatility_ratio = short_term_volatility / long_term_volatility

    # Core Factor
    core_factor = (net_score + integrated_momentum) * volatility_ratio

    # Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(1)) / (lookback_window * 2)

    # Final Alpha Factor
    intraday_volatility = (df['high'] - df['low']) / df['open']
    short_term_momentum = (df['close'] / df['close'].shift(5) - 1)
    long_term_momentum = (df['close'] / df['close'].shift(20) - 1)
    weighted_momentum_combination = short_term_momentum * 0.6 + long_term_momentum * 0.4
    combined_intraday_volatility_and_momentum = intraday_volatility * weighted_momentum_combination
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    adjusted_combined_factor = combined_intraday_volatility_and_momentum / ema_volume
    volume_ratio = df['volume'] / ema_volume
    final_alpha = core_factor + volume_trend + (combined_intraday_volatility_and_momentum * volume_ratio)

    return final_alpha
```
