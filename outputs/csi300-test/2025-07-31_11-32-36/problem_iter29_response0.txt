```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor from the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and indexed by (date).
    
    Returns:
    pd.Series: A Series of factor values indexed by (date).
    """
    # Calculate the relative position of the closing price within the day's range
    close_position = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the 10-day rolling average of the close position
    close_position_rolling_avg = close_position.rolling(window=10).mean()
    
    # Calculate the 5-day rolling sum of the volume
    volume_rolling_sum = df['volume'].rolling(window=5).sum()
    
    # Normalize the volume rolling sum by its 30-day rolling standard deviation
    volume_normalized = (volume_rolling_sum - volume_rolling_sum.rolling(window=30).mean()) / volume_rolling_sum.rolling(window=30).std()
    
    # Calculate the alpha factor as the product of the normalized close position and the normalized volume
    alpha_factor = close_position_rolling_avg * volume_normalized
    
    return alpha_factor
```
