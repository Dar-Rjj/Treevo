```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum Score
    high_low_range = df['high'] - df['low']
    close_open_change = df['close'] - df['open']
    sum_of_ranges_and_changes = high_low_range + close_open_change
    intraday_momentum_score = sum_of_ranges_and_changes * df['volume']

    # Calculate Intraday Volatility Score
    high_close_deviation = abs(df['high'] - df['close'])
    low_close_deviation = abs(df['low'] - df['close'])
    max_of_deviations = pd.concat([high_close_deviation, low_close_deviation], axis=1).max(axis=1)
    intraday_volatility_score = max_of_deviations * df['amount']

    # Combine Scores
    alpha_factor = intraday_momentum_score - intraday_volatility_score

    return alpha_factor
```
