```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Compute Close to Open Difference
    close_to_open_diff = df['close'] - df['open']
    
    # Calculate Intraday Momentum Score
    intraday_momentum_score = (intraday_high_low_range + close_to_open_diff) * df['volume']
    
    # Calculate Volume Weighted Price
    volume_weighted_price = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])
    
    # Calculate Volume Weighted Price Change
    volume_weighted_price_change = volume_weighted_price - volume_weighted_price.shift(1)
    
    # Compute Combined Intraday and Daily Momentum
    combined_intraday_daily_momentum = intraday_momentum_score + volume_weighted_price_change
    
    # Calculate Intraday Volatility Score
    high_close_delta = df['high'] - df['close']
    low_close_delta = df['low'] - df['close']
    avg_deltas = (high_close_delta + low_close_delta) / 2
    intraday_volatility_score = avg_deltas * df['amount']
    
    # Integrate Composite Momentum and Volatility
    combined_momentum_ema30 = combined_intraday_daily_momentum.ewm(span=30).mean()
    composite_momentum_volatility = combined_intraday_daily_momentum - combined_momentum_ema30 - intraday_volatility_score
    
    # Calculate Volume-Adjusted Daily Returns
    volume_adjusted_daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Calculate Weighted Momentum
    vol_adj_ret_10d_ma = volume_adjusted_daily_returns.rolling(window=10).mean().shift(1)
    vol_20d_ema = df['volume'].ewm(span=20).mean()
    weighted_momentum = vol_adj_ret_10d_ma / vol_20d_ema
    
    # Calculate Volatility
    daily_returns = df['close'] / df['close'].shift(1) - 1
    volatility_10d_std = daily_returns.rolling(window=10).std().shift(1)
    inverse_volatility = 1 / (volatility_10d_std + 1e-6)
    
    # Sum Weighted Momentum and Inverse of Volatility
    weighted_momentum_volatility = weighted_momentum + inverse_volatility
    
    # Calculate Daily Return
    daily_return = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Volume Difference
    volume_difference = df['volume'] - df['volume'].shift(1)
    
    # Rolling Average Volume Difference
    rolling_avg_volume_diff = volume_difference.rolling(window=10).mean()
    
    # Multiply Price Momentum by Volume Trend
    price_momentum_volume_trend = daily_return * rolling_avg_volume_diff
    
    # Calculate Intraday Price Range Ratio
    intraday_price_range_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Close
    volume_adjusted_close = df['close'] * (df['volume'] ** 0.5)
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine All Factors
    alpha_factor = (
        price_momentum_volume_trend +
        intraday_price_range_ratio +
        volume_adjusted_close +
        intraday_return +
        composite_momentum_volatility +
        weighted_momentum_volatility
    )
    
    return alpha_factor
```
