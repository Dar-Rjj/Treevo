import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']

    # Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(5) - 1)
    long_term_momentum = (df['close'] / df['close'].shift(20) - 1)

    # Combined Momentum
    combined_momentum = (short_term_momentum * 0.7 + long_term_momentum * 0.3)

    # Combined Factor
    combined_factor = (intraday_volatility * combined_momentum)

    # Volume Ratio
    volume_ratio = (df['volume'] / df['volume'].ewm(span=10, adjust=False).mean())

    # Final Alpha Factor
    final_alpha_factor = (combined_factor * volume_ratio)

    return final_alpha_factor
