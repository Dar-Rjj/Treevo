```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: Volume weighted close-to-open return for each date.
    """
    # Calculate the close-to-open return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
