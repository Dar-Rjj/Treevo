import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the calculated alpha factor values indexed by (date).
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume weighted price
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10, min_periods=1).mean()
    
    # Calculate the momentum factor
    df['momentum'] = df['close'].pct_change(periods=21)
    
    # Calculate the volatility factor
    df['volatility'] = df['daily_return'].rolling(window=21, min_periods=1).std()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['momentum'] / df['volatility']) * df['intraday_range'] * (df['close'] / df['vwap'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
