```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Price Momentum
    df['short_term_momentum'] = df['close'].pct_change(periods=5)
    df['long_term_momentum'] = df['close'].pct_change(periods=20)
    
    # Weighted Momentum Combination
    df['weighted_momentum'] = (df['short_term_momentum'] * 0.6 + df['long_term_momentum'] * 0.4)
    
    # Combine Intraday Volatility and Momentum
    df['combined_factor'] = df['intraday_volatility'] * df['weighted_momentum']
    
    # Adjust for Volume
    df['sma_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_factor'] = df['combined_factor'] / df['sma_volume']
    
    return df['adjusted_factor'].dropna()
```
