```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].diff()
    
    # Average Daily Return
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'].diff()
    
    # Rolling Average Volume Difference
    df['avg_volume_diff'] = df['volume_diff'].rolling(window=5).mean()
    
    # Calculate Volume and Amplitude Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['amplitude_confirmation'] = (df['high'] - df['low']) > (df['high'].shift(1) - df['low'].shift(1))
    df['combined_confirmation'] = 0.5 * (df['volume_confirmation'] + df['amplitude_confirmation'])
    
    # Multiply Price Momentum by Volume and Amplitude Trend
    df['price_momentum_vol_amp_trend'] = df['avg_daily_return'] * df['avg_volume_diff'] * df['combined_confirmation']
    
    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    
    # Compute Intraday Momentum Score
    df['sum_of_deltas'] = df['intraday_high_low_range'] + df['close_open_diff']
    df['intraday_momentum_score'] = df['volume'] * df['sum_of_deltas']
    
    # Calculate Volume Weighted Price
    df['vol_weighted_close'] = df['close'] * df['volume']
    df['vol_weighted_open'] = df['open'] * df['volume']
    df['total_vol'] = df['volume'].rolling(window=2).sum()
    df['vol_weighted_price'] = (df['vol_weighted_close'] + df['vol_weighted_open']) / df['total_vol']
    
    # Calculate Volume Weighted Price Change
    df['vol_weighted_price_change'] = df['vol_weighted_price'] - df['vol_weighted_price'].shift(1)
    
    # Compute Combined Intraday and Daily Momentum
    df['combined_intraday_daily_momentum'] = df['intraday_momentum_score'] + df['vol_weighted_price_change']
    
    # Calculate Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['avg_deltas'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['intraday_vol_std'] = df['intraday_high_low_range'].rolling(window=30).std()
    df['enhanced_intraday_vol_score'] = df['avg_deltas'] * df['intraday_vol_std']
    
    # Integrate Composite Momentum and Volatility
    df['ema_combined_momentum'] = df['combined_intraday_daily_momentum'].ewm(span=30, adjust=False).mean()
    df['composite_momentum_volatility'] = df['combined_intraday_daily_momentum'] - df['ema_combined_momentum'] - df['enhanced_intraday_vol_score']
    
    # Combine Factors
    df['alpha_factor'] = (
        df['price_momentum_vol_amp_trend'] +
        df['intraday_high_low_range'] +
        df['intraday_momentum_score'] +
        df['combined_intraday_daily_momentum'] +
        df['enhanced_intraday_vol_score'] +
        df['vol_weighted_price_change']
    )
    
    return df['alpha_factor']
```
