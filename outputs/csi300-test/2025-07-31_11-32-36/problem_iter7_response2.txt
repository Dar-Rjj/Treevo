```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=10, ema_span=10):
    # Compute Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Average Intraday Range Over N Days
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n_days).mean()
    
    # Compute Close Price Returns
    df['close_returns'] = df['close'].pct_change()
    
    # Calculate Cumulative Return Over M Days
    df['cumulative_return'] = df['close_returns'].rolling(window=m_days).sum()
    
    # Combine Intraday Range and Momentum
    df['combined_factor'] = df['avg_intraday_range'] * df['cumulative_return']
    
    # Compute EMA of Volume
    df['ema_volume'] = df['volume'].ewm(span=ema_span, adjust=False).mean()
    
    # Adjust for Volume
    df['final_factor'] = df['combined_factor'] / df['ema_volume']
    
    return df['final_factor'].dropna()
```
