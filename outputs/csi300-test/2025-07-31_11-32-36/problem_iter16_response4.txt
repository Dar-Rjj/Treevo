```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-Day Price Momentum
    df['price_momentum'] = df['close'].diff().rolling(window=5).sum()
    
    # Calculate Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Amplitude Confirmation
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)
    
    # Calculate 5-Day Average True Range (ATR)
    df['true_range'] = df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)
    df['atr_5'] = df['true_range'].rolling(window=5).mean()
    
    # Adjust for Volatility
    df['volatility_adjustment'] = (df['volume_confirmation'] + df['amplitude_confirmation']) / df['atr_5']
    
    # Calculate 20-Day Simple Moving Average (SMA)
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Introduce Trend Strength Confirmation
    df['trend_strength_confirmation'] = (df['close'] > df['sma_20']).astype(int)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['price_momentum'] * df['volatility_adjustment'] * df['trend_strength_confirmation']
    
    return df['alpha_factor']
```
