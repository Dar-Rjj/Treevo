```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: Series with the calculated alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 30-day volatility (standard deviation) of the daily return
    df['vol_30_daily_return'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the 10-day moving average of the volume
    df['ma_10_volume'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 30-day volatility (standard deviation) of the volume
    df['vol_30_volume'] = df['volume'].rolling(window=30).std()
    
    # Calculate the 10-day price momentum
    df['price_momentum_10'] = df['close'] / df['close'].shift(10) - 1
    
    # Calculate the 10-day volume momentum
    df['volume_momentum_10'] = df['volume'] / df['volume'].shift(10) - 1
    
    # Alpha factor: (10-day price momentum / 30-day volatility of daily return) * (10-day volume momentum / 30-day volatility of volume)
    df['alpha_factor'] = (df['price_momentum_10'] / df['vol_30_daily_return']) * (df['volume_momentum_10'] / df['vol_30_volume'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
