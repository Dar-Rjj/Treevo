```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['range'] = df['high'] - df['low']
    n_days = 20
    df['avg_range'] = df['range'].rolling(window=n_days).mean()

    # Calculate Daily Close-to-Close Return
    df['daily_return'] = df['close'].pct_change()

    # Apply Exponential Moving Average (EMA) to Returns
    ema_span = 20
    df['ema_returns'] = df['daily_return'].ewm(span=ema_span, adjust=False).mean()

    # Compute EMA of Squared Returns for Volatility
    df['squared_returns'] = df['daily_return'] ** 2
    df['ema_volatility'] = df['squared_returns'].ewm(span=ema_span, adjust=False).mean()

    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = np.abs(df['ema_returns'] - df['ema_volatility'])

    # Weight Divergence by Volume and High-Low Range
    df['divergence_weighted'] = (df['divergence'] * df['volume']) / df['range']

    # Calculate Intraday Momentum
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    df['intraday_momentum'] = df['intraday_return'] * avg_volume

    # Calculate Intraday Movement Factor
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['prev_true_range'] = df[['high', 'low', 'close']].shift(1).max(axis=1) - df[['high', 'low', 'close']].shift(1).min(axis=1)
    df['intraday_movement_factor'] = df['intraday_high_low_spread'] / df['prev_true_range']

    # Integrate Volume Impact
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['weighted_divergence'] = df['divergence_weighted'] + df['intraday_movement_factor']

    # Calculate Price Momentum
    df['avg_daily_return'] = df['daily_return'].rolling(window=20).mean()

    # Volume Trend
    df['volume_difference'] = df['volume'] - df['volume'].shift(1)
    df['rolling_avg_volume_diff'] = df['volume_difference'].rolling(window=20).mean()

    # Weighted Combination
    df['weighted_combination'] = df['intraday_movement_factor'] + df['intraday_momentum']

    # Synthesize Initial Alpha Factor
    df['synthesized_alpha'] = df['avg_daily_return'] * df['rolling_avg_volume_diff'] * df['weighted_combination'] * df['volume_change']

    # Final Alpha Factor
    df['final_alpha_factor'] = df['weighted_divergence'] * df['volume_change'] * df['synthesized_alpha']

    return df['final_alpha_factor'].dropna()
```
