import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Price Momentum
    df['short_term_momentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['long_term_momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Weighted Momentum Combination
    df['weighted_momentum'] = df['short_term_momentum'] * 0.6 + df['long_term_momentum'] * 0.4
    
    # Combine Intraday Volatility and Momentum
    df['combined_factor'] = df['intraday_volatility'] * df['weighted_momentum']
    
    # Adjust for Volume
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['adjusted_factor'] = df['combined_factor'] / df['sma_volume_20']
    
    # Incorporate Trading Volume Impact
    df['volume_ratio'] = df['volume'] / df['sma_volume_20']
    df['final_alpha_factor'] = df['combined_factor'] * df['volume_ratio']
    
    return df['final_alpha_factor'].dropna()
