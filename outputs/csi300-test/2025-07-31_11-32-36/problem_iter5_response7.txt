```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Close-to-Close Return
    df['return'] = df['close'] - df['close'].shift(1)
    
    # Apply Exponential Moving Average (EMA) to Returns
    lookback_period = 20
    df['ema_return'] = df['return'].ewm(span=lookback_period, adjust=False).mean()
    
    # Compute EMA of Squared Returns for Volatility
    df['squared_return'] = df['return'] ** 2
    df['ema_volatility'] = df['squared_return'].ewm(span=lookback_period, adjust=False).mean().sqrt()
    
    # Calculate Weighted Momentum
    df['10_day_moving_avg_return'] = df['return'].rolling(window=10).mean().shift(1)
    df['weighted_momentum'] = df['10_day_moving_avg_return'] * df['volume']
    
    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = (df['ema_volatility'] - df['ema_return']).abs()
    
    # Weight Divergence by Volume and High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['weighted_divergence'] = (df['divergence'] * df['volume']) / df['high_low_range']
    
    # Calculate Intraday Price Range Ratio
    df['intraday_price_range_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Movement Factor
    df['prev_true_range'] = df[['high', 'low', 'close']].shift(1).apply(lambda x: max(x) - min(x), axis=1)
    df['intraday_movement_factor'] = (df['high'] - df['low']) / df['prev_true_range']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Impact
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['integrated_volume_impact'] = (df['weighted_divergence'] + df['intraday_movement_factor']) * df['volume_change']
    
    # Combine All Factors
    df['alpha_factor'] = (
        df['weighted_momentum'] +
        df['divergence'] +
        df['weighted_divergence'] +
        df['intraday_price_range_ratio'] +
        df['intraday_movement_factor'] +
        df['intraday_return']
    ) * df['volume_change']
    
    return df['alpha_factor']
```
