```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, ema_span=5):
    # Compute Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Average Intraday Range Over N Days
    avg_intraday_range = intraday_range.rolling(window=n_days).mean()
    
    # Compute Close Price Returns
    close_returns = df['close'].pct_change()
    
    # Combine Intraday Range and Momentum
    combined_factor = avg_intraday_range * close_returns
    
    # Adjust for Volume
    ema_volume = df['volume'].ewm(span=ema_span, adjust=False).mean()
    adjusted_combined_factor = combined_factor / ema_volume
    
    # Incorporate Open to Close Returns
    open_to_close_returns = (df['close'] / df['open']) - 1
    final_combined_factor = adjusted_combined_factor + open_to_close_returns
    
    # Threshold Adjustment
    final_combined_factor = final_combined_factor.apply(lambda x: max(x, 0))
    
    return final_combined_factor
```
