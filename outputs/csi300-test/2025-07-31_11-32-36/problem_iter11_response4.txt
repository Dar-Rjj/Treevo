```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A series representing the alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3.0
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day EMA of the typical price
    ema_14_typical = typical_price.ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-day EMA of the money flow
    ema_14_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the ratio of the two EMAs
    alpha_factor = ema_14_money_flow / ema_14_typical
    
    return alpha_factor
```
