```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range Ratio
    intraday_price_range_ratio = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Close
    volume_adjusted_close = df['close'] * (df['volume'] ** 0.5)
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Combine Factors
    factor = intraday_price_range_ratio + volume_adjusted_close + intraday_return
    
    return factor
```
