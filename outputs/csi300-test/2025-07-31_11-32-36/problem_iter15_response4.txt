import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using (open, high, low, close, amount, volume) data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume' and index 'date'.
    
    Returns:
    pd.Series: A Series with the calculated alpha factor indexed by 'date'.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the distance between the current close and the 10-day moving average
    df['dist_ma_10'] = df['close'] - df['ma_10_close']
    
    # Calculate the distance between the 10-day moving average and the 30-day moving average
    df['dist_ma_10_30'] = df['ma_10_close'] - df['ma_30_close']
    
    # Calculate the 10-day rolling standard deviation of the daily returns
    df['std_10_daily_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day rolling sum of volume
    df['sum_10_volume'] = df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day rolling sum of amount
    df['sum_10_amount'] = df['amount'].rolling(window=10).sum()
    
    # Calculate the ratio of the 10-day rolling sum of amount to the 10-day rolling sum of volume
    df['amount_to_volume_ratio'] = df['sum_10_amount'] / df['sum_10_volume']
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['dist_ma_10'] / df['std_10_daily_return']) * df['amount_to_volume_ratio'] + df['dist_ma_10_30']
    
    # Drop the intermediate columns
    df.drop(['daily_return', 'ma_10_close', 'ma_30_close', 'dist_ma_10', 'dist_ma_10_30', 'std_10_daily_return', 'sum_10_volume', 'sum_10_amount', 'amount_to_volume_ratio'], axis=1, inplace=True)
    
    return df['alpha_factor']
