```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Previous Day's True Range
    prev_day_high = df['high'].shift(1)
    prev_day_low = df['low'].shift(1)
    prev_day_close = df['close'].shift(1)
    prev_day_true_range = (prev_day_high - prev_day_low).fillna(0)  # Fill NaN with 0 for the first day
    
    # Normalize Intraday High-Low Spread by Previous Day's True Range
    intraday_movement_factor = intraday_high_low_spread / prev_day_true_range.replace(0, 1)  # Avoid division by zero
    
    # Calculate Close Price Direction
    close_price_direction = df['close'] - df['open']
    
    # Compute Volume Change
    volume_change = (df['volume'] - df['volume'].shift(1)).fillna(0)  # Fill NaN with 0 for the first day
    
    # Weighted Combination of Intraday Movement Factor and Close Price Direction
    weighted_combination = 0.5 * intraday_movement_factor + 0.5 * close_price_direction
    
    # Multiply by Volume Change
    factor_values = weighted_combination * volume_change
    
    return factor_values
```
