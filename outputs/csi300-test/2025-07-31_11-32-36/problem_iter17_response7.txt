```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close to Open Difference
    df['close_to_open'] = df['close'] - df['open']
    
    # Compute Intraday Momentum Score
    df['sum_of_deltas'] = df['intraday_range'] + df['close_to_open']
    df['intraday_momentum_score'] = df['sum_of_deltas'] * df['volume']
    
    # Calculate Volume Weighted Price
    df['vwp'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])
    
    # Calculate Volume Weighted Price Change
    df['vwp_change'] = df['vwp'].diff()
    
    # Compute Combined Intraday and Daily Momentum
    df['combined_momentum'] = df['intraday_momentum_score'] + df['vwp_change']
    
    # Apply Exponential Moving Average (EMA) to Combined Momentum
    df['ema_momentum_20'] = df['combined_momentum'].ewm(span=20, adjust=False).mean()
    
    # Calculate Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['avg_delta'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['intraday_volatility_score'] = df['avg_delta'] * df['amount']
    
    # Compute EMA of Squared Returns for Volatility
    df['daily_returns'] = df['close'].pct_change()
    df['squared_returns'] = df['daily_returns'] ** 2
    df['ema_volatility_20'] = df['squared_returns'].ewm(span=20, adjust=False).mean()
    
    # Calculate Divergence between EMA of Returns and Volatility
    df['divergence'] = np.abs(df['ema_momentum_20'] - df['ema_volatility_20'])
    
    # Calculate Divergence and Weight Factor
    df['divergence_weighted'] = (df['divergence'] * df['volume']) / df['intraday_range']
    
    # Integrate Composite Momentum, Volatility, and Volume Trends
    df['ema_combined_momentum_30'] = df['combined_momentum'].ewm(span=30, adjust=False).mean()
    df['integrated_scores'] = df['combined_momentum'] - df['ema_combined_momentum_30'] - df['intraday_volatility_score'] + df['divergence_weighted']
    
    # Introduce Time-Varying Volatility Adjustment
    df['long_term_volatility'] = df['squared_returns'].ewm(span=50, adjust=False).mean()
    df['short_term_volatility'] = df['squared_returns'].ewm(span=10, adjust=False).mean()
    df['volatility_ratio'] = df['short_term_volatility'] / df['long_term_volatility']
    df['adjusted_scores'] = df['integrated_scores'] * df['volatility_ratio']
    
    return df['adjusted_scores']
```
