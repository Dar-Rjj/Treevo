import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Average Daily Return over 5-day lookback window
    df['avg_daily_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Rolling Average Volume Difference over 5-day lookback window
    df['avg_volume_diff'] = df['volume_diff'].rolling(window=5).mean()
    
    # Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Amplitude Confirmation
    df['amplitude'] = df['high'] - df['low']
    df['amplitude_confirmation'] = (df['amplitude'] > df['amplitude'].shift(1)).astype(int)
    
    # Combined Confirmation
    df['combined_confirmation'] = (df['volume_confirmation'] + df['amplitude_confirmation']) * 0.5
    
    # Multiply Price Momentum by Volume and Amplitude Trend
    df['price_momentum'] = df['avg_daily_return'] * df['avg_volume_diff'] * df['combined_confirmation']
    
    # Intraday Momentum Score
    df['high_low_delta'] = df['high'] - df['low']
    df['close_open_delta'] = df['close'] - df['open']
    df['sum_of_deltas'] = df['high_low_delta'] + df['close_open_delta']
    df['mav'] = df['volume'].rolling(window=5).mean()
    df['weighted_momentum'] = df['sum_of_deltas'] * (df['volume'] / df['mav'])
    
    # Intraday Volatility Score
    df['high_close_delta'] = df['high'] - df['close']
    df['low_close_delta'] = df['low'] - df['close']
    df['average_of_deltas'] = (df['high_close_delta'] + df['low_close_delta']) / 2
    df['maa'] = df['amount'].rolling(window=5).mean()
    df['weighted_volatility'] = df['average_of_deltas'] * (df['amount'] / df['maa'])
    
    # Price Trend Factor
    df['close_open_ratio'] = df['close'] / df['open']
    df['smoothed_ratio'] = df['close_open_ratio'].ewm(span=30, adjust=False).mean()
    
    # Daily Momentum and Volatility Components
    df['daily_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['ema_returns'] = df['daily_returns'].ewm(span=30, adjust=False).mean()
    df['ema_squared_returns'] = (df['daily_returns'] ** 2).ewm(span=30, adjust=False).mean()
    df['divergence'] = abs(df['ema_returns'] - df['ema_squared_returns'])
    df['high_low_range'] = df['high'] - df['low']
    df['weighted_divergence'] = df['divergence'] * (df['volume'] / df['high_low_range'])
    
    # Combined Intraday and Daily Scores
    df['total_momentum'] = df['weighted_momentum'] + df['smoothed_ratio']
    df['net_score'] = df['total_momentum'] - df['weighted_volatility'] + df['weighted_divergence']
    
    # Volume and Price Adjusted Momentum
    df['volume_weighted_price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (2 * df['volume'])
    df['volume_weighted_price_change'] = df['volume_weighted_price'] - df['volume_weighted_price'].shift(1)
    df['integrated_momentum'] = (df['weighted_momentum'] + df['volume_weighted_price_change']).ewm(span=30, adjust=False).mean()
    df['momentum_reversal'] = (df['weighted_momentum'] + df['volume_weighted_price_change']) - df['integrated_momentum']
    
    # Long-Term and Short-Term Volatility
    df['long_term_volatility'] = df['ema_squared_returns'].ewm(span=50, adjust=False).mean()
    df['short_term_volatility'] = df['ema_squared_returns'].ewm(span=10, adjust=False).mean()
    df['volatility_ratio'] = df['short_term_volatility'] / df['long_term_volatility']
