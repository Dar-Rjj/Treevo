```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility and Momentum
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['open']
    df['High_Low_Delta'] = df['high'] - df['low']
    df['Close_Open_Delta'] = df['close'] - df['open']
    df['Sum_of_Deltas'] = df['High_Low_Delta'] + df['Close_Open_Delta']
    df['SMA_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Weighted_Intraday_Momentum'] = df['Sum_of_Deltas'] * (df['volume'] / df['SMA_Volume_20'])
    df['Combined_Intraday_Volatility_and_Momentum'] = df['Intraday_Volatility'] * df['Weighted_Intraday_Momentum']

    # Daily Momentum and Volatility
    df['Short_Term_Momentum'] = df['close'] / df['close'].shift(7) - 1
    df['Long_Term_Momentum'] = df['close'] / df['close'].shift(25) - 1
    df['Weighted_Momentum_Combination'] = df['Short_Term_Momentum'] * 0.5 + df['Long_Term_Momentum'] * 0.5
    df['Daily_Returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['EMA_Returns'] = df['Daily_Returns'].ewm(span=10, adjust=False).mean()
    df['EMA_Squared_Returns'] = (df['Daily_Returns'] ** 2).ewm(span=10, adjust=False).mean()
    df['Divergence'] = (df['EMA_Returns'] - df['EMA_Squared_Returns']).abs()
    df['High_Low_Range'] = df['high'] - df['low']
    df['Weighted_Divergence'] = df['Divergence'] * (df['volume'] / df['High_Low_Range'])

    # Price Trend and Smoothed Ratio
    df['Close_Open_Ratio'] = df['close'] / df['open']
    df['Smoothed_Ratio'] = df['Close_Open_Ratio'].ewm(span=10, adjust=False).mean()

    # Combined Intraday and Daily Scores
    df['Total_Momentum'] = df['Weighted_Intraday_Momentum'] + df['Weighted_Momentum_Combination']
    df['Net_Score'] = df['Total_Momentum'] - df['Combined_Intraday_Volatility_and_Momentum'] + df['Weighted_Divergence']
    
    df['Volume_Weighted_Price'] = (df['close'] * df['volume'] + df['open'] * df['volume']) / (df['volume'] * 2)
    df['Volume_Weighted_Price_Change'] = df['Volume_Weighted_Price'] - df['Volume_Weighted_Price'].shift(1)
    df['Integrated_Momentum'] = (df['Weighted_Intraday_Momentum'] + df['Volume_Weighted_Price_Change']).ewm(span=30, adjust=False).mean()

    # Final Alpha Factor
    df['Core_Factor'] = (df['Net_Score'] + df['Integrated_Momentum']) * (df['Short_Term_Momentum'] / df['Long_Term_Momentum'])
    df['Volume_Trend'] = (df['volume'] - df['volume'].shift(1)) / 20
    df['Average_Daily_Return'] = df['Daily_Returns'].rolling(window=20).mean()
    df['Rolling_Average_Volume_Difference'] = df['volume'].diff().rolling(window=20).mean()
    df['Price_Momentum_Over_Volume_Trend'] = df['Average_Daily_Return'] / df['Rolling_Average_Volume_Difference']
    df['Volume_Confirmation'] = (df['volume'] - df['volume'].shift(1)) > 0
    df['Amplitude_Confirmation'] = (df['high'] - df['low']) > (df['high'].shift(1) - df['low'].shift(1))
    df['Final_Alpha'] = df['Core_Factor'] + df['Volume_Trend'] + (df['Price_Momentum_Over_Volume_Trend'] * (df['Volume_Confirmation'] + df['Amplitude_Confirmation']) * 0.5)

    return df['Final_Alpha']
```
