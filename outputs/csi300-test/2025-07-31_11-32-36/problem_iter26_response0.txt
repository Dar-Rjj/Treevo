```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    
    # Calculate Relative Momentum
    short_term_momentum = (df['close'] / df['close'].shift(5)) - 1
    long_term_momentum = (df['close'] / df['close'].shift(20)) - 1
    relative_momentum = short_term_momentum - long_term_momentum
    
    # Integrate Intraday Volatility and Momentum
    integrated_factor = intraday_volatility * relative_momentum
    
    # Factor Adjusted for Volume Changes
    daily_volume_growth = (df['volume'] / df['volume'].shift(1)) - 1
    factor_adjusted_for_volume = integrated_factor * daily_volume_growth
    
    # Incorporate Amount and Price Trend
    average_trade_size = df['amount'] / df['volume']
    price_trend = (df['close'] - df['open']) / df['open']
    final_factor = factor_adjusted_for_volume * price_trend * average_trade_size
    
    return final_factor
```
