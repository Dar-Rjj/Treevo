```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate 10-Day Moving Average of Daily Returns (excluding the most recent return for look-back calculation)
    df['10_day_moving_avg'] = df['daily_return'].shift(1).rolling(window=9).mean()
    
    # Calculate Weighted Momentum
    df['weighted_momentum'] = df['10_day_moving_avg'] * df['volume']
    
    # Calculate 10-Day Standard Deviation of Daily Returns (excluding the most recent return for look-back calculation)
    df['10_day_std_dev'] = df['daily_return'].shift(1).rolling(window=9).std()
    
    # Compute Inverse of Volatility with a small constant to avoid division by zero
    epsilon = 1e-6
    df['inverse_volatility'] = 1 / (df['10_day_std_dev'] + epsilon)
    
    # Sum Weighted Momentum and Inverse of Volatility
    df['factor'] = df['weighted_momentum'] + df['inverse_volatility']
    
    # Return the factor as a pandas Series
    return df['factor'].dropna()
```
