```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5):
    # Compute Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Average Intraday Range Over N Days
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=N).mean()
    
    # Compute Close Price Returns
    df['close_returns'] = df['close'].pct_change()
    
    # Combine Intraday Range and Momentum
    df['combined_factor'] = df['avg_intraday_range'] * df['close_returns']
    
    # Adjust for Volume
    df['ema_volume'] = df['volume'].ewm(span=N, adjust=False).mean()
    df['adjusted_combined_factor'] = df['combined_factor'] / df['ema_volume']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Average Daily Return
    df['avg_daily_return'] = df['daily_return'].rolling(window=N).mean()
    
    # Calculate Cumulative Return Over M Days
    df['cumulative_return'] = df['close_returns'].rolling(window=M).sum()
    
    # Adjust Combined Intraday Range and Momentum
    df['adjusted_intraday_momentum'] = df['avg_intraday_range'] * df['cumulative_return']
    
    # Calculate Volume Difference
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Rolling Average Volume Difference
    df['rolling_avg_volume_diff'] = df['volume_diff'].rolling(window=N).mean()
    
    # Multiply Price Momentum by Volume Trend
    df['price_momentum_volume_trend'] = df['avg_daily_return'] * df['rolling_avg_volume_diff']
    
    # Calculate Intraday Price Range Ratio
    df['intraday_range_ratio'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume Adjusted Close
    df['volume_adjusted_close'] = df['close'] * np.sqrt(df['volume'])
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = (
        df['price_momentum_volume_trend'] + 
        df['adjusted_intraday_momentum'] + 
        df['intraday_range_ratio'] + 
        df['volume_adjusted_close'] + 
        df['intraday_return']
    )
    
    return df['final_alpha_factor']
```
