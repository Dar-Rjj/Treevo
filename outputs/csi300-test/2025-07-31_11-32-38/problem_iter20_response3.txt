def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Weighted Sum of Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'].abs() * df['intraday_return']
    df['weighted_close_to_open_return'] = df['close_to_open_return'].abs() * df['close_to_open_return']
    df['preliminary_factor'] = df['weighted_intraday_return'] + df['weighted_close_to_open_return']
    
    # Calculate True Range (TR)
    df['previous_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['previous_close']).abs()
    df['tr3'] = (df['low'] - df['previous_close']).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Positive Directional Movement (+DM)
    df['high_diff'] = df['high'] - df['high'].shift(1)
    df['low_diff'] = df['low'] - df['low'].shift(1)
    df['positive_dm'] = df.apply(lambda x: x['high_diff'] if (x['high_diff'] > 0) and (x['high_diff'] > x['low_diff']) else 0, axis=1)
    
    # Calculate Negative Directional Movement (-DM)
    df['negative_dm'] = df.apply(lambda x: x['low_diff'] if (x['low_diff'] > 0) and (x['low_diff'] > x['high_diff']) else 0, axis=1)

    smoothing_period = 14

    df['smoothed_negative_dm'] = df['negative_dm'].rolling(window=smoothing_period).mean()
    df['smoothed_true_range'] = df['true_range'].rolling(window=smoothing_period).mean()
    df['+di'] = df['smoothed_positive_dm'] / df['smoothed_true_range']
    df['-di'] = df['smoothed_negative_dm'] / df['smoothed_true_range']

    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
