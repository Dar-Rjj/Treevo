```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Volume-Weighted Daily Returns
    df['VWDR'] = (df['close'] - df['close'].shift(1)) / df['volume']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t-1'] = df['log_volume_t'].shift(1)
    df['log_vol_diff'] = df['log_volume_t'] - df['log_volume_t-1']
    df['smoothed_vol_diff'] = df['log_vol_diff'].ewm(alpha=0.2).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['20_day_vol_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['20_day_vol_ma']
    df['weighted_intraday_spread'] = df['intraday_high_low_spread'] * df['volume_ratio']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    
    # Integrate Volume Change and Intraday Volatility
    df['adjusted_weights'] = df['weighted_intraday_return'] + df['weighted_close_to_open_return']
    df['adjusted_weights'] *= df['smoothed_vol_diff'] * df['weighted_intraday_spread']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['raw_momentum'] = (df['close'] - df['close'].shift(n))
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    df['price_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['price_range']
    
    # Weigh Combined Momentum by Intraday Volume Activity
    df['final_momentum'] = df['combined_momentum'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['final_factor'] = df['final_momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['final_factor'] *= df['VWDR']
    df['final_factor'] = df['final_factor'].rolling(window=n).sum()
    
    # Incorporate Intraday Range Breakouts
    df['intraday_range_breakout'] = (df['high'] - df['low'].shift(1)) + (df['high'].shift(1) - df['low'])
    df['weighted_breakout'] = df['intraday_range_breakout'] * df['volume_ratio']
    df['final_factor'] += df['weighted_breakout']
    
    # Calculate Intraday Trading Imbalance
    df['trading_imbalance'] = (df['high'] + df['low'] - df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Integrate Intraday Trading Imbalance with Final Factor
    df['imbalance_factor'] = df['trading_imbalance'] * df['volume']
    df['final_factor'] += df['imbalance_factor']
    
    return df['final_factor']
```
