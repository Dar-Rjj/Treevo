import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on price action patterns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Short-term momentum (sum of daily price changes over 10 days)
    df['momentum_10d'] = df['daily_price_change'].rolling(window=10).sum()
    
    # Daily range (High - Low) as a measure of volatility
    df['daily_range'] = df['high'] - df['low']
    
    # Average daily range over the last 20 days for smoother volatility
    df['avg_daily_range_20d'] = df['daily_range'].rolling(window=20).mean()
    
    # Directional Move (1 if close > open, -1 if close < open, 0 if close == open)
    df['directional_move'] = (df['close'] - df['open']).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    # Net number of upward and downward moves over the last 30 days
    df['net_directional_moves_30d'] = df['directional_move'].rolling(window=30).sum()
    
    # Volume Ratio (Volume on up days / Volume on down days over the last 50 days)
    df['volume_up'] = df[df['directional_move'] == 1]['volume']
    df['volume_down'] = df[df['directional_move'] == -1]['volume']
    df['volume_ratio_50d'] = df['volume_up'].rolling(window=50).sum() / df['volume_down'].rolling(window=50).sum().fillna(1)
    
    # Total volume over the last 50 days
    df['total_volume_50d'] = df['volume'].rolling(window=50).sum()
    
    # Amount change from one day to the next
    df['amount_change'] = df['amount'].diff()
    
    # Cumulative amount change over the last 60 days
    df['cum_amount_change_60d'] = df['amount_change'].rolling(window=60).sum()
    
    # Alpha factor: Combine all the components
    df['alpha_factor'] = (
        df['momentum_10d'] +
        df['avg_daily_range_20d'] * df['net_directional_moves_30d'] +
        df['volume_ratio_50d'] * df['total_volume_50d'] +
        df['cum_amount_change_60d']
    )
    
    return df['alpha_factor']
