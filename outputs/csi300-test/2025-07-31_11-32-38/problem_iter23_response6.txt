```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate percentage change in close price
    df['Close_Change'] = df['close'].pct_change()
    
    # Calculate daily volume change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Calculate average true range (ATR)
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Combine price and volume
    df['SMA_10_Volume'] = df['SMA_10'] * df['volume']
    df['SMA_30_Volume'] = df['SMA_30'] * df['volume']
    
    # Incorporate high and low prices
    df['High_Low_Diff'] = df['high'] - df['low']
    df['Avg_High_Low_10'] = df['High_Low_Diff'].rolling(window=10).mean()
    df['Avg_High_Low_30'] = df['High_Low_Diff'].rolling(window=30).mean()
    
    # Create a composite alpha factor
    df['Alpha_Factor'] = (
        df['SMA_10_Volume'] + 
        df['SMA_30_Volume'] + 
        df['Close_Change'] + 
        df['Volume_Change'] + 
        df['ATR_14'] + 
        df['Avg_High_Low_10'] + 
        df['Avg_High_Low_30']
    )
    
    return df['Alpha_Factor']
```
