```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['open']
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Intraday and Close-to-Open Returns
    intraday_weight = df['intraday_return'].abs()
    close_to_open_weight = df['close_to_open_return'].abs()
    df['combined_returns'] = (intraday_weight * df['intraday_return'] + 
                              close_to_open_weight * df['close_to_open_return'])
    
    # Integrate Volume Change
    df['preliminary_factor'] = df['combined_returns'] * df['volume_change']
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['positive_dm'] = 0
    df.loc[(df['high'] > df['high'].shift(1)) & (df['high'] - df['high'].shift(1) > df['low'].shift(1) - df['low']), 'positive_dm'] = df['high'] - df['high'].shift(1)
    
    # Calculate Negative Directional Movement (-DM)
    df['negative_dm'] = 0
    df.loc[(df['low'].shift(1) > df['low']) & (df['low'].shift(1) - df['low'] > df['high'] - df['high'].shift(1)), 'negative_dm'] = df['low'].shift(1) - df['low']
    
    # Calculate +DM Smoothed Over n Periods
    df['smoothed_positive_dm'] = df['positive_dm'].rolling(window=n).sum() / n
    
    # Calculate -DM Smoothed Over n Periods
    df['smoothed_negative_dm'] = df['negative_dm'].rolling(window=n).sum() / n
    
    # Calculate +DI and -DI
    df['positive_di'] = 100 * (df['smoothed_positive_dm'] / df['atr'])
    df['negative_di'] = 100 * (df['smoothed_negative_dm'] / df['atr'])
    
    # Calculate ADMI
    df['admi'] = (df['positive_di'] - df['negative_di']) / (df['positive_di'] + df['negative_di'])
    
    # Calculate Volume-Weighted Momentum
    raw_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = raw_momentum / cumulative_volume
    
    # Incorporate Price Range Factor
    df['price_range'] = df['high'] - df['low']
    df['volume_weighted_momentum_adjusted'] = df['volume_weighted_momentum'] * df['price_range']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = (df['preliminary_factor'] * df['admi'] + 
                          df['volume_weighted_momentum_adjusted'])
    
    return df['alpha_factor']
```
