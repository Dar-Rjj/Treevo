```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * df['intraday_return'].abs()
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['close_to_open_return'].abs()
    
    # Integrate Volume Change
    df['adjusted_weighted_intraday_return'] = df['weighted_intraday_return'] * df['volume_change']
    df['adjusted_weighted_close_to_open_return'] = df['weighted_close_to_open_return'] * df['volume_change']
    df['preliminary_factor'] = df['adjusted_weighted_intraday_return'] + df['adjusted_weighted_close_to_open_return']
    
    # Calculate True Range (TR)
    df['previous_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['previous_close']).abs()
    df['tr3'] = (df['low'] - df['previous_close']).abs()
    df['tr'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR) over n periods
    n = 14
    df['atr'] = df['tr'].rolling(window=n, min_periods=1).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+dm'] = 0
    mask = (df['high'] > df['high'].shift(1)) & ((df['high'] - df['high'].shift(1)) > (df['low'].shift(1) - df['low']))
    df.loc[mask, '+dm'] = df['high'] - df['high'].shift(1)
    
    # Calculate Negative Directional Movement (-DM)
    df['-dm'] = 0
    mask = (df['low'].shift(1) > df['low']) & ((df['low'].shift(1) - df['low']) > (df['high'] - df['high'].shift(1)))
    df.loc[mask, '-dm'] = df['low'].shift(1) - df['low']
    
    # Calculate +DM Smoothed Over n Periods
    df['+dm_smoothed'] = df['+dm'].rolling(window=n, min_periods=1).sum() / n
    
    # Calculate -DM Smoothed Over n Periods
    df['-dm_smoothed'] = df['-dm'].rolling(window=n, min_periods=1).sum() / n
    
    # Calculate +DI
    df['+di'] = df['+dm_smoothed'] / df['atr']
    
    # Calculate -DI
    df['-di'] = df['-dm_smoothed'] / df['atr']
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['preliminary_factor'] * df['admi']
    
    return df['alpha_factor']
```
