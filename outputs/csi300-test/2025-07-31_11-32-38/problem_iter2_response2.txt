```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Compute Raw Momentum
    raw_momentum = df['close'] / df['close'].shift(n) - 1
    
    # Calculate Cumulative Volume
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    
    # Weight by Cumulative Volume
    volume_weighted_momentum = raw_momentum / cumulative_volume
    
    # Incorporate Price Range Factor
    price_range = df['high'] - df['low']
    combined_momentum = volume_weighted_momentum * price_range + volume_weighted_momentum
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    volume_20_day_ma = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / volume_20_day_ma
    weighted_intraday_spread = intraday_spread * volume_ratio
    
    # Combine Momentum and Intraday Volatility
    combined_factor = combined_momentum + weighted_intraday_spread
    
    # Smooth Combined Momentum and Intraday Volatility
    smoothed_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
