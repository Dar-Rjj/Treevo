```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Calculate 10-Day Moving Average of Daily Price Changes
    df['10D_MA_Price_Change'] = df['Daily_Price_Change'].rolling(window=10).mean()
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'].diff()
    
    # Combine Intraday and Close-to-Open Returns
    intraday_weight = df['Intraday_Return'].abs()
    close_open_weight = df['Close_to_Open_Return'].abs()
    df['Preliminary_Factor'] = (intraday_weight * df['Intraday_Return'] + 
                                close_open_weight * df['Close_to_Open_Return'])
    
    # Integrate Volume Change
    df['Preliminary_Factor'] *= df['Daily_Volume_Change']
    
    # Calculate True Range (TR)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['close'].shift(1) - df['high']).abs()
    df['TR3'] = (df['close'].shift(1) - df['low']).abs()
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+DM'] = (df['high'] - df['high'].shift(1)).apply(lambda x: max(x, 0))
    
    # Calculate Negative Directional Movement (-DM)
    df['-DM'] = (df['low'].shift(1) - df['low']).apply(lambda x: max(x, 0))
    
    # Smooth +DM and -DM over 14 periods
    df['+DM_Smoothed'] = df['+DM'].rolling(window=14).sum() / 14
    df['-DM_Smoothed'] = df['-DM'].rolling(window=14).sum() / 14
    
    # Calculate +DI and -DI
    df['+DI'] = 100 * (df['+DM_Smoothed'] / df['ATR'])
    df['-DI'] = 100 * (df['-DM_Smoothed'] / df['ATR'])
    
    # Calculate ADMI
    df['ADMI'] = (df['+DI'] - df['-DI']) / (df['+DI'] + df['-DI'])
    
    # Compute 20-Day Moving Average of Volume and Amount Weighted Returns
    df['Volume_Amount_Weighted_Return'] = (df['close'] / df['open']) * df['volume'] * df['amount']
    df['20D_MA_Volume_Amount_Weighted_Return'] = df['Volume_Amount_Weighted_Return'].rolling(window=20).mean()
    
    # Combine Signals
    df['Diff_10D_20D_MA'] = df['10D_MA_Price_Change'] - df['20D_MA_Volume_Amount_Weighted_Return']
    df['Smoothed_Diff'] = df['Diff_10D_20D_MA'].rolling(window=5).mean()
    
    # Calculate Volume-Weighted Momentum
    raw_momentum = df['close'] / df['close'].shift(20) - 1
    cumulative_volume = df['volume'].rolling(window=20).sum().shift(1)
    df['Volume_Weighted_Momentum'] = raw_momentum / cumulative_volume
    
    # Incorporate Price Range Factor
    price_range = df['high'] - df['low']
    df['Price_Range_Adjusted_Vol_Mom'] = df['Volume_Weighted_Momentum'] * price_range
    
    # Synthesize Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Preliminary_Factor'] * df['ADMI'] + 
                                df['Price_Range_Adjusted_Vol_Mom'] + 
                                df['Smoothed_Diff'])
    
    return df['Final_Alpha_Factor']
```
