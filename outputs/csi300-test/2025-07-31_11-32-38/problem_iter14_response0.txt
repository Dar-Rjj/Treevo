import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor from the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day and 30-day rolling average of returns
    df['10_day_avg_return'] = df['return'].rolling(window=10).mean()
    df['30_day_avg_return'] = df['return'].rolling(window=30).mean()
    
    # Calculate the 10-day and 30-day rolling standard deviation of returns
    df['10_day_std_return'] = df['return'].rolling(window=10).std()
    df['30_day_std_return'] = df['return'].rolling(window=30).std()
    
    # Calculate the 10-day and 30-day rolling volume
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['30_day_avg_volume'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the 10-day and 30-day rolling high and low
    df['10_day_max_high'] = df['high'].rolling(window=10).max()
    df['30_day_max_high'] = df['high'].rolling(window=30).max()
    
    df['10_day_min_low'] = df['low'].rolling(window=10).min()
    df['30_day_min_low'] = df['low'].rolling(window=30).min()
    
    # Calculate the volatility-adjusted return
    df['vol_adj_return'] = (df['10_day_avg_return'] - df['30_day_avg_return']) / (df['30_day_std_return'] + 1e-6)
    
    # Calculate the volume momentum
    df['volume_momentum'] = (df['10_day_avg_volume'] - df['30_day_avg_volume']) / (df['30_day_avg_volume'] + 1e-6)
    
    # Calculate the price range factor
    df['price_range_factor'] = (df['10_day_max_high'] - df['10_day_min_low']) / (df['30_day_max_high'] - df['30_day_min_low'] + 1e-6)
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['vol_adj_return'] * df['volume_momentum'] * df['price_range_factor']
    
    # Drop rows with NaN values
    df = df.dropna()
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
