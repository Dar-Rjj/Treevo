    df['price_change'] = df['close'] - df['close'].shift(1)

    df['rolling_avg_price_change'] = df['price_change'].rolling(window=15).mean()

    df['weighted_daily_return'] = df['daily_return'] * df['volume'] * df['amount']

    df['weighted_daily_return_ma'] = df['weighted_daily_return'].rolling(window=25).mean()
