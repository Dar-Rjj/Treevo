import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Momentum Adjustment
    df['ema_10'] = df['daily_returns'].ewm(span=10, adjust=False).mean()
    df['ema_5'] = df['daily_returns'].ewm(span=5, adjust=False).mean()
    df['momentum_adjustment'] = df['ema_10'] - df['ema_5']
    
    # Reversal Adjustment
    weights = np.arange(1, 22)
    df['wma_21'] = df['daily_returns'].rolling(window=21).apply(lambda x: np.average(x, weights=weights), raw=True)
    df['reversal_adjustment'] = -df['wma_21']
    
    # Composite Factor
    df['composite_factor'] = (df['momentum_adjustment'] + df['reversal_adjustment']) * np.sqrt(df['volume'])
    
    return df['composite_factor'].dropna()
