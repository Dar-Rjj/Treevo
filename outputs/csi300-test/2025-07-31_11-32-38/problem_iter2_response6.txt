```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Logarithmic Volumes
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_shift'] = df['log_volume'].shift(1)
    
    # Subtract and Smooth Volume Difference
    df['volume_diff'] = df['log_volume'] - df['log_volume_shift']
    df['smoothed_volume_diff'] = df['volume_diff'].ewm(span=n, adjust=False).mean()
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    
    # Adjust Weights by Volume
    df['adjusted_weighted_intraday_return'] = df['weighted_intraday_return'] * df['smoothed_volume_diff']
    df['adjusted_weighted_close_to_open_return'] = df['weighted_close_to_open_return'] * df['smoothed_volume_diff']
    
    # Final Factor Calculation
    df['weighted_daily_return'] = df['daily_return'] * df['smoothed_volume_diff']
    df['combined_returns'] = df['adjusted_weighted_intraday_return'] + df['adjusted_weighted_close_to_open_return']
    df['factor'] = df['weighted_daily_return'] + df['combined_returns']
    
    # Accumulate Over n Days
    df['accumulated_factor'] = df['factor'].rolling(window=n).sum()
    
    return df['accumulated_factor']
```
