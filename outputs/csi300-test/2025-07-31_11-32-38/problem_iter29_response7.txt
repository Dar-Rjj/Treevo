```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily price changes
    df['daily_price_change'] = df['close'].diff()
    
    # Short-term momentum: sum of daily price changes over a 10-day window
    short_term_momentum = df['daily_price_change'].rolling(window=10).sum()
    
    # Medium-term trend: cumulative sum of daily price changes over a 30-day window
    medium_term_trend = df['daily_price_change'].rolling(window=30).sum().cumsum()
    
    # Volatility: average daily range (High - Low) over a 20-day period
    avg_daily_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Volatility consistency: standard deviation of the daily range (High - Low) over a 20-day period
    std_daily_range = (df['high'] - df['low']).rolling(window=20).std()
    
    # Net number of days where Close > Open versus Close < Open over a 30-day period
    up_days = (df['close'] > df['open']).rolling(window=30).sum()
    down_days = (df['close'] < df['open']).rolling(window=30).sum()
    net_sentiment = up_days - down_days
    
    # Ratio of the sum of Close - Open on up days to down days over a 30-day period
    close_open_up = (df['close'] - df['open']).where(df['close'] > df['open'], 0).rolling(window=30).sum()
    close_open_down = (df['close'] - df['open']).where(df['close'] < df['open'], 0).rolling(window=30).sum()
    directional_bias = close_open_up / close_open_down.abs()
    
    # Volume ratio on up days to down days over a 50-day period
    volume_up = df['volume'].where(df['close'] > df['open'], 0).rolling(window=50).sum()
    volume_down = df['volume'].where(df['close'] < df['open'], 0).rolling(window=50).sum()
    volume_ratio = volume_up / volume_down.abs()
    
    # Sum of volume on up days minus the sum of volume on down days over a 50-day period
    net_volume_pressure = volume_up - volume_down.abs()
    
    # Cumulative amount change over a 60-day lookback period
    cumulative_amount_change = df['amount'].diff().rolling(window=60).sum()
    
    # Ratio of the sum of amount on up days to down days over a 60-day period
    amount_up = df['amount'].where(df['close'] > df['open'], 0).rolling(window=60).sum()
    amount_down = df['amount'].where(df['close'] < df['open'], 0).rolling(window=60).sum()
    amount_ratio = amount_up / amount_down.abs()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        short_term_momentum + 
        medium_term_trend + 
        avg_daily_range - 
        std_daily_range + 
        net_sentiment + 
        directional_bias + 
        volume_ratio + 
        net_volume_pressure + 
        cumulative_amount_change + 
        amount_ratio
    )
    
    return alpha_factor
```
