```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_amount_weighted_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns
    df['20_day_ma_volume_amount_weighted_return'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_difference'] = df['10_day_ma_price_change'] - df['20_day_ma_volume_amount_weighted_return']
    df['5_day_ma_signal_difference'] = df['signal_difference'].rolling(window=5).mean()
    
    # Calculate Intraday and Close-to-Open Returns
    df['intraday_return'] = df['high'] - df['low']
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Weight by Logarithmic Volume Difference
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t_1'] = np.log(df['volume']).shift(1)
    df['log_volume_diff'] = df['log_volume_t'] - df['log_volume_t_1']
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(span=5, adjust=False).mean()
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = (df['intraday_return'] / df['intraday_return'].abs()) * df['smoothed_log_volume_diff']
    df['weighted_close_to_open_return'] = (df['close_to_open_return'] / df['close_to_open_return'].abs()) * df['smoothed_log_volume_diff']
    df['combined_returns'] = df['weighted_intraday_return'] + df['weighted_close_to_open_return']
    
    # Integrate Volume Change
    df['daily_volume_change'] = df['volume'] / df['volume'].shift(1)
    df['preliminary_factor'] = (df['5_day_ma_signal_difference'] + df['combined_returns']) * df['daily_volume_change']
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Positive and Negative Directional Movement (+DM, -DM)
    df['+dm'] = 0
    df['-dm'] = 0
    df['+dm'] = np.where((df['high'] > df['high'].shift(1)) & (df['high'] - df['high'].shift(1) > df['low'].shift(1) - df['low']), df['high'] - df['high'].shift(1), 0)
    df['-dm'] = np.where((df['low'].shift(1) > df['low']) & (df['low'].shift(1) - df['low'] > df['high'] - df['high'].shift(1)), df['low'].shift(1) - df['low'], 0)
    
    # Calculate +DI and -DI
    df['+di'] = 100 * (df['+dm'].rolling(window=14).sum() / df['atr'])
    df['-di'] = 100 * (df['-dm'].rolling(window=14).sum() / df['atr'])
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['preliminary_factor'] * df['admi']) + (df['combined_returns'] * df['admi'])
    
    return df['final_alpha_factor']
```
