```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor that could help predict future stock returns.
    
    The function calculates a weighted combination of the following:
    - Momentum: 12-day moving average return
    - Reversal: 1-day return
    - Volatility: Standard deviation of the last 12 days' returns
    - Volume: 12-day moving average of volume
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate daily returns
    df['returns'] = df['close'].pct_change()
    
    # Calculate 12-day momentum
    df['momentum'] = df['returns'].rolling(window=12).mean()
    
    # Calculate 1-day reversal
    df['reversal'] = -df['returns']
    
    # Calculate 12-day volatility
    df['volatility'] = df['returns'].rolling(window=12).std()
    
    # Calculate 12-day moving average of volume
    df['volume_avg'] = df['volume'].rolling(window=12).mean()
    
    # Normalize the factors
    df['norm_momentum'] = (df['momentum'] - df['momentum'].mean()) / df['momentum'].std()
    df['norm_reversal'] = (df['reversal'] - df['reversal'].mean()) / df['reversal'].std()
    df['norm_volatility'] = (df['volatility'] - df['volatility'].mean()) / df['volatility'].std()
    df['norm_volume_avg'] = (df['volume_avg'] - df['volume_avg'].mean()) / df['volume_avg'].std()
    
    # Weighted combination of the factors
    alpha_factor = 0.4 * df['norm_momentum'] + 0.3 * df['norm_reversal'] + 0.2 * df['norm_volatility'] + 0.1 * df['norm_volume_avg']
    
    return alpha_factor
```
