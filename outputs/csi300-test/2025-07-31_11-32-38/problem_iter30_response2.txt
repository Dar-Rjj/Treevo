import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 30-day standard deviation of the daily return
    df['std_30_daily_return'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the 10-day exponential moving average of the close price
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the close price
    df['ema_30_close'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the momentum factor using the difference between the 10-day and 30-day EMA
    df['momentum_factor'] = df['ema_10_close'] - df['ema_30_close']
    
    # Calculate the money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(df['close'] > df['close'].shift(1), 0)
    negative_flow = raw_money_flow.where(df['close'] < df['close'].shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + money_flow_ratio))
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['ma_10_daily_return'] 
                          + df['momentum_factor'] 
                          - df['std_30_daily_return'] 
                          + df['mfi'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor'].dropna()
