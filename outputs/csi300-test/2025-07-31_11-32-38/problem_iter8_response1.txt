```python
import pandas as pd
import numpy as np

def heuristics(df, n=20):
    # Calculate Volume-Weighted Daily Returns
    df['Volume-Weighted Daily Return'] = (df['close'] - df['close'].shift(1)) / df['volume']
    
    # Compute Intraday Return
    df['Intraday Return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['Close-to-Open Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['Log Volume t'] = np.log(df['volume'])
    df['Log Volume t-1'] = df['Log Volume t'].shift(1)
    df['Log Volume Diff'] = df['Log Volume t'] - df['Log Volume t-1']
    df['Smoothed Volume Diff'] = df['Log Volume Diff'].ewm(alpha=0.2).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday High-Low Spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['20-Day Vol MA'] = df['volume'].rolling(window=20).mean()
    df['Volume Ratio'] = df['volume'] / df['20-Day Vol MA']
    df['Intraday High-Low Spread Weight'] = df['Intraday High-Low Spread'] * df['Volume Ratio']
    
    # Combine Intraday and Close-to-Open Returns
    df['Intraday Weight'] = np.abs(df['Intraday Return'])
    df['Close-to-Open Weight'] = np.abs(df['Close-to-Open Return'])
    df['Combined Returns'] = (df['Intraday Return'] * df['Intraday Weight']) + (df['Close-to-Open Return'] * df['Close-to-Open Weight'])
    
    # Integrate Volume Change and Intraday Volatility
    df['Adjusted Weights'] = df['Combined Returns'] * df['Smoothed Volume Diff'] * df['Intraday High-Low Spread Weight']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['Raw Momentum'] = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    df['Cumulative Volume'] = df['volume'].rolling(window=n).sum()
    df['Volume-Weighted Momentum'] = df['Raw Momentum'] / df['Cumulative Volume']
    df['Price Range'] = df['high'] - df['low']
    df['Momentum with Price Range'] = df['Volume-Weighted Momentum'] + df['Price Range']
    
    # Weigh Combined Momentum by Intraday Volume Activity
    df['Weighted Momentum'] = df['Momentum with Price Range'] * df['Volume Ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['EMA Factor'] = df['Weighted Momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['Final Factor'] = (df['Volume-Weighted Daily Return'] * df['Adjusted Weights']).rolling(window=n).sum()
    
    return df['Final Factor']
```
