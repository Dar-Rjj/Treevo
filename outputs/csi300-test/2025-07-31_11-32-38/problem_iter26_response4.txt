import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is designed to be predictive of future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and indexed by date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday range (high - low)
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate the average true range over a 10-day window
    df['atr_10'] = df['intraday_range'].rolling(window=10).mean()
    
    # Calculate the volume-weighted price
    df['vwp'] = (df['amount'] / df['volume']).shift(1)  # Shift to avoid lookahead bias
    
    # Calculate the change in volume-weighted price
    df['vwp_change'] = df['vwp'].pct_change()
    
    # Calculate the factor as a combination of daily return, ATR, and VWP change
    df['alpha_factor'] = (df['daily_return'] * 0.5 + 
                          (df['intraday_range'] / df['atr_10']) * 0.3 + 
                          df['vwp_change'] * 0.2)
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
