```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Volume-Weighted Daily Returns
    df['Volume-Weighted Returns'] = (df['close'].pct_change() * df['volume']).shift(1)
    
    # Compute Intraday Return
    df['Intraday Return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['Close-to-Open Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['Log Volume'] = np.log(df['volume'])
    df['Log Volume Diff'] = df['Log Volume'] - df['Log Volume'].shift(1)
    df['Smoothed Log Volume Diff'] = df['Log Volume Diff'].ewm(alpha=0.2).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday Spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['Volume Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Weighted Intraday Spread'] = df['Intraday Spread'] * df['Volume Ratio']
    
    # Combine Intraday and Close-to-Open Returns
    intraday_weight = np.abs(df['Intraday Return'])
    close_to_open_weight = np.abs(df['Close-to-Open Return'])
    total_weight = intraday_weight + close_to_open_weight
    df['Combined Return'] = (intraday_weight * df['Intraday Return'] + close_to_open_weight * df['Close-to-Open Return']) / total_weight
    
    # Integrate Volume Change and Intraday Volatility
    df['Adjusted Combined Return'] = df['Combined Return'] * df['Smoothed Log Volume Diff'] * df['Weighted Intraday Spread']
    
    # Combine Volume-Weighted Momentum with Price Range
    raw_momentum = (df['close'] / df['close'].shift(n) - 1)
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    volume_weighted_momentum = raw_momentum / cumulative_volume
    price_range = df['high'] - df['low']
    df['Combined Momentum'] = volume_weighted_momentum + price_range
    
    # Weigh Combined Momentum by Intraday Volume Activity
    df['Weighted Combined Momentum'] = df['Combined Momentum'] * df['Volume Ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['EMA Weighted Combined Momentum'] = df['Weighted Combined Momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['Final Factor'] = (df['Volume-Weighted Returns'] * df['Adjusted Combined Return']).rolling(window=n).sum()
    
    # Incorporate Intraday Range Breakouts
    df['Intraday Range Breakout'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    df['Intraday Breakout Weight'] = df['Intraday Range Breakout'] * df['Volume Ratio']
    df['Final Factor'] += df['Intraday Breakout Weight']
    
    return df['Final Factor'].dropna()
```
