```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Rolling Average of Daily Price Changes (10-day moving average)
    df['rolling_price_change_avg'] = df['price_change'].rolling(window=10).mean()
    
    # Calculate Volume Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Summarize Volume Weighted Daily Returns (20-day moving average)
    df['rolling_volume_weighted_return_avg'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_difference'] = df['rolling_price_change_avg'] - df['rolling_volume_weighted_return_avg']
    
    # Smooth the Result (5-day moving average)
    df['smoothed_signal'] = df['signal_difference'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = df['smoothed_signal']
    
    return alpha_factor
```
