```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()

    # Calculate Rolling Average of Daily Price Changes (10-day MA)
    df['price_change_10_day_ma'] = df['daily_price_change'].rolling(window=10).mean()

    # Calculate Volume and Amount Weighted Daily Return
    df['volume_amount_weighted_return'] = (df['close'] / df['open']) * df['volume'] * df['amount']

    # Summarize Volume and Amount Weighted Daily Returns (20-day MA)
    df['volume_amount_weighted_20_day_ma'] = df['volume_amount_weighted_return'].rolling(window=20).mean()

    # Combine Signals
    df['signal_difference'] = df['price_change_10_day_ma'] - df['volume_amount_weighted_20_day_ma']
    df['combined_signal'] = df['signal_difference'].rolling(window=5).mean()

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']

    # Weighted Sum of Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * df['intraday_return'].abs()
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['close_to_open_return'].abs()
    df['preliminary_factor'] = df['weighted_intraday_return'] + df['weighted_close_to_open_return']

    # Integrate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_change'].abs()
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] * df['volume_change'].abs()
    df['intraday_preliminary_factor'] = df['adjusted_intraday_return'] + df['adjusted_close_to_open_return']

    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['tr'] = df[['high', 'low', 'prev_close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)

    # Calculate Positive Directional Movement (+DM) and Negative Directional Movement (-DM)
    df['+dm'] = np.where((df['high'] > df['high'].shift(1)) & ((df['high'] - df['high'].shift(1)) > (df['low'].shift(1) - df['low'])), 
                         df['high'] - df['high'].shift(1), 0)
    df['-dm'] = np.where((df['low'].shift(1) > df['low']) & ((df['low'].shift(1) - df['low']) > (df['high'] - df['high'].shift(1))), 
                         df['low'].shift(1) - df['low'], 0)

    # Smooth +DM and -DM, then calculate +DI and -DI
    df['+di'] = df['+dm'].rolling(window=14).sum() / df['tr'].rolling(window=14).sum()
    df['-di'] = df['-dm'].rolling(window=14).sum() / df['tr'].rolling(window=14).sum()

    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])

    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_return'].rolling(window=10).mean()

    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_signal'] * df['admi'] +
                                df['intraday_preliminary_factor'] * df['admi'] +
                                df['intraday_momentum'] * df['admi'])

    return df['final_alpha_factor']
```
