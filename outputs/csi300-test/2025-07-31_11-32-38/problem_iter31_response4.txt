```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate 10-Day Moving Average of Daily Price Changes
    df['10_day_ma'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate 20-Day Moving Average of Daily Price Changes
    df['20_day_ma'] = df['daily_price_change'].rolling(window=20).mean()
    
    # Combine Moving Averages
    df['ma_diff'] = df['10_day_ma'] - df['20_day_ma']
    df['5_day_ma_of_diff'] = df['ma_diff'].rolling(window=5).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['weighted_daily_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns
    df['20_day_avg_weighted_return'] = df['weighted_daily_return'].rolling(window=20).mean()
    
    # Integrate Volume Change
    df['daily_volume_change'] = df['volume'].diff()
    df['preliminary_factor'] = df['20_day_avg_weighted_return'] * df['daily_volume_change']
    
    # Calculate True Range (TR)
    df['high_low'] = df['high'] - df['low']
    df['prev_close_high'] = (df['close'].shift() - df['high']).abs()
    df['prev_close_low'] = (df['close'].shift() - df['low']).abs()
    df['true_range'] = df[['high_low', 'prev_close_high', 'prev_close_low']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    n = 14
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Calculate Positive and Negative Directional Movement (+DM, -DM)
    df['+dm'] = (df['high'] - df['high'].shift()).apply(lambda x: max(x, 0))
    df['-dm'] = (df['low'].shift() - df['low']).apply(lambda x: max(x, 0))
    
    # Calculate +DI and -DI
    df['+di'] = (df['+dm'] / df['atr']) * 100
    df['-di'] = (df['-dm'] / df['atr']) * 100
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['5_day_ma_of_diff'] * df['admi']) + (df['preliminary_factor'] * df['admi'])
    
    return df['final_alpha_factor']
```
