```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Returns
    df['raw_return'] = df['close'].pct_change()

    # Momentum Component
    df['momentum_10_day'] = df['raw_return'].rolling(window=10).sum()
    df['momentum_5_day'] = df['raw_return'].rolling(window=5).sum()
    df['momentum_component'] = df['momentum_10_day'] - df['momentum_5_day']

    # Reversal Component
    df['reversal_21_day'] = df['raw_return'].rolling(window=21).mean()
    df['reversal_component'] = -df['reversal_21_day']

    # Price Momentum with Volume Weight
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['rolling_avg_daily_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    df['volume_weighted_daily_return'] = (df['close'] / df['open']) * df['volume']
    df['volume_weighted_ma_20_day'] = df['volume_weighted_daily_return'].rolling(window=20).mean()

    # Adjusted Momentum and Reversal
    momentum_std = df['momentum_component'].rolling(window=20).std()
    reversal_std = df['reversal_component'].rolling(window=20).std()
    
    df['normalized_momentum'] = df['momentum_component'] / momentum_std
    df['normalized_reversal'] = df['reversal_component'] / reversal_std
    
    df['combined_normalized'] = df['normalized_momentum'] + df['normalized_reversal']

    # Final Alpha Factor
    df['signal_difference'] = df['rolling_avg_daily_price_change'] - df['volume_weighted_ma_20_day']
    df['smoothed_signal'] = df['signal_difference'].rolling(window=5).mean()

    # Incorporate Volume Shock Factor
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_shock'] = df['volume_change'].rolling(window=5).mean()
    
    df['final_alpha_factor'] = df['combined_normalized'] * df['volume_shock']

    return df['final_alpha_factor'].dropna()
```
