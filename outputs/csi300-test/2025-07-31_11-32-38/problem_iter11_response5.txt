```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Returns
    df['return'] = df['close'].pct_change()
    
    # Momentum Component
    ema_5 = df['return'].ewm(span=5, adjust=False).mean()
    ema_10 = df['return'].ewm(span=10, adjust=False).mean()
    momentum = ema_5 - ema_10
    
    # Reversal Component
    wma_21 = df['return'].rolling(window=21).apply(lambda x: np.sum(x * np.arange(1, 22)) / np.sum(np.arange(1, 22)), raw=True)
    reversal = -wma_21
    
    # Combine Momentum and Reversal Components
    combined_mr = (momentum + reversal) / 2
    
    # Calculate Volume-Weighted Momentum
    n = 5
    raw_momentum = df['close'].pct_change(n)
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    volume_weighted_momentum = raw_momentum / cumulative_volume
    
    # Incorporate Price Range Factor
    price_range = df['high'] - df['low']
    volume_price_adjusted = volume_weighted_momentum * price_range
    combined_momentum = combined_mr + volume_price_adjusted
    
    # Integrate Open to Close Price Movement
    open_to_close = df['close'] - df['open']
    combined_momentum += volume_weighted_momentum * open_to_close
    
    # Integrate Combined Momentum and Reversal
    final_momentum = combined_momentum * np.sqrt(df['volume'])
    
    # Calculate True Range (TR)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    df['tr'] = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR)
    atr = df['tr'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+dm'] = np.where((df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1)), 
                         df['high'] - df['high'].shift(1), 0)
    
    # Calculate Negative Directional Movement (-DM)
    df['-dm'] = np.where((df['low'] < df['low'].shift(1)) & (df['high'] > df['high'].shift(1)), 
                         df['low'].shift(1) - df['low'], 0)
    
    # Smooth +DM and -DM over 14 periods
    smooth_period = 14
    df['+dm_smooth'] = df['+dm'].rolling(window=smooth_period).mean()
    df['-dm_smooth'] = df['-dm'].rolling(window=smooth_period).mean()
    
    # Calculate +DI and -DI
    df['+di'] = (df['+dm_smooth'] / atr) * 100
    df['-di'] = (df['-dm_smooth'] / atr) * 100
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    ma_10_price = df['close'].rolling(window=10).mean()
    ma_20_vol_weighted = (df['return'] * df['volume']).rolling(window=20).mean() / df['volume'].rolling(window=20).mean()
    diff_ma = ma_10_price - ma_20_vol_weighted
    final_factor = diff_ma.rolling(window=5).mean() * df['admi']
    
    return final_factor
```
