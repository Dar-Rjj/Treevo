import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=20, p=5, q=5):
    # Calculate Price Momentum Factor
    df['Close_t_minus_n'] = df['close'].shift(n)
    df['Price_Momentum'] = (df['close'] - df['Close_t_minus_n']) / df['Close_t_minus_n']
    
    # Incorporate Trading Volume into the Analysis
    df['Average_Volume'] = df['volume'].rolling(window=m).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Explore the Relationship Between Open and Close Prices
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Moving_Sum_Open_Close_Diff'] = df['Open_Close_Diff'].rolling(window=p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    df['Daily_Range'] = df['high'] - df['low']
    df['Moving_Avg_Daily_Range'] = df['Daily_Range'].rolling(window=q).mean()
    
    # Combine daily price changes with volume to identify momentum
    df['Volume_Weighted_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Examine intraday returns and their significance
    df['Intraday_Return'] = (df['high'] - df['low']) / df['open']
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Assess the impact of volume on daily and intraday returns
    df['Log_Volume_Diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    df['Volume_Activity'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrate volume-adjusted momentum with price range
    df['Combined_Return'] = (abs(df['Intraday_Return']) + abs(df['Close_to_Open_Return'])) * df['Log_Volume_Diff'] * (df['Daily_Range'] / df['Moving_Avg_Daily_Range'])
    
    # Raw Momentum
    df['Raw_Momentum'] = (df['close'] - df['Close_t_minus_n']) / df['Close_t_minus_n']
    df['Cumulative_Volume'] = df['volume'].rolling(window=n-1).sum().shift(1)
    df['Weighted_Momentum'] = df['Raw_Momentum'] * df['Cumulative_Volume']
    
    # Apply smoothing and finalize the factor
    df['Smoothed_Factor'] = df['Combined_Return'].ewm(span=5, adjust=False).mean()
    df['Final_Factor'] = df['Smoothed_Factor'] * df['Volume_Weighted_Return'].rolling(window=n).sum()
    
    return df['Final_Factor']
