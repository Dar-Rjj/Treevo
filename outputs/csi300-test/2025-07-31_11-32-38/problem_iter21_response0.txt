import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 10-Day Exponential Moving Average of Daily Price Changes
    df['ema_10_price_change'] = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    df['intraday_return_mean_5'] = df['intraday_return'].rolling(window=5).mean()
    df['intraday_return_adj'] = df['intraday_return'] - df['intraday_return_mean_5']
    
    # Calculate Volume Weighted Daily Return
    df['volume_weighted_return'] = (df['close'] - df['open']) * df['volume']
    
    # Compute 20-Day Exponential Moving Average of Volume Weighted Returns
    df['ema_20_vol_weighted_return'] = df['volume_weighted_return'].ewm(span=20, adjust=False).mean()
    
    # Weigh by Intraday Volume Activity
    df['ma_20_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['ma_20_volume']
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['volume_weighted_intraday_spread'] = df['intraday_high_low_spread'] * df['volume_ratio']
    
    # Calculate True Range (TR)
    df['tr_high_low'] = abs(df['high'] - df['low'])
    df['tr_close_high'] = abs(df['close'].shift(1) - df['high'])
    df['tr_close_low'] = abs(df['close'].shift(1) - df['low'])
    df['true_range'] = df[['tr_high_low', 'tr_close_high', 'tr_close_low']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    n = 14
    df['atr'] = df['true_range'].rolling(window=n).sum() / n
    
    # Calculate Positive Directional Movement (+DM)
    df['pdm'] = (df['high'] - df['high'].shift(1)).where((df['high'] > df['high'].shift(1)) & (df['low'] >= df['low'].shift(1)), 0)
    
    # Calculate Negative Directional Movement (-DM)
    df['ndm'] = (df['low'].shift(1) - df['low']).where((df['low'] < df['low'].shift(1)) & (df['high'] <= df['high'].shift(1)), 0)
    
    # Smooth +DM and -DM over n periods
    df['smooth_pdm'] = df['pdm'].rolling(window=n).sum() / n
    df['smooth_ndm'] = df['ndm'].rolling(window=n).sum() / n
    
    # Calculate +DI and -DI
    df['pos_di'] = df['smooth_pdm'] / df['atr']
    df['neg_di'] = df['smooth_ndm'] / df['atr']
    
    # Calculate ADMI
    df['admi'] = (df['pos_di'] - df['neg_di']) / (df['pos_di'] + df['neg_di'])
    
    # Combine Intraday and Interday Momentum
    df['combined_momentum'] = df['ema_10_price_change'] + df['intraday_return_adj'] - df['ema_20_vol_weighted_return']
    
    # Adjust for Price Volatility
    df['daily_price_range'] = df['high'] - df['low']
    df['ma_30_daily_price_range'] = df['daily_price_range'].rolling(window=30).mean()
    df['adjusted_momentum'] = df['combined_momentum'] / df['ma_30_daily_price_range']
    
    # Smooth the Result
    df['smoothed_factor'] = df['adjusted_momentum'].ewm(span=5, adjust=False).mean()
    
    # Synthesize Final Alpha Factor
    df['final_alpha'] = df['smoothed_factor'] * df['admi'] + df['volume'].ewm(span=5, adjust=False).mean()
    
    return df['final_alpha']
