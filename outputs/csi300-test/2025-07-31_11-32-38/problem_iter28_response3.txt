import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages (SMA)
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate percentage change in close price from one day to the next
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate daily volume change as a percentage of the previous day's volume
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # Calculate the 10-day moving average of volume
    df['volume_MA_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the average true range (ATR) over a 20-day period
    df['true_range'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    df['ATR_20'] = df['true_range'].rolling(window=20).mean()
    
    # Combine price and volume for a more comprehensive factor
    df['SMA_20_volume_MA_10'] = df['SMA_20'] * df['volume_MA_10']
    
    # Calculate the average high-low difference over a 20-day period
    df['avg_high_low_20'] = df['high'].sub(df['low']).rolling(window=20).mean()
    
    # Calculate the percentage change from open to close
    df['open_close_pct_change'] = (df['close'] - df['open']) / df['open']
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['SMA_20'] + 
        df['SMA_50'] + 
        df['close_pct_change'] + 
        df['volume_pct_change'] + 
        df['volume_MA_10'] + 
        df['ATR_20'] + 
        df['SMA_20_volume_MA_10'] + 
        df['avg_high_low_20'] + 
        df['open_close_pct_change']
    )
    
    return alpha_factor
