```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Rolling Average of Daily Price Changes (10-day MA)
    df['price_momentum_10d'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['volume_amount_weighted_return'] = (df['close'] / df['open']) * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns (20-day MA)
    df['weighted_return_20d'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_difference'] = df['price_momentum_10d'] - df['weighted_return_20d']
    df['preliminary_alpha_factor'] = df['signal_difference'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['intraday_preliminary_factor'] = (df['intraday_return'] + df['close_to_open_return']) * df['volume_change']
    
    # Calculate True Range (TR)
    df['tr_high_low'] = abs(df['high'] - df['low'])
    df['tr_close_high'] = abs(df['close'].shift(1) - df['high'])
    df['tr_close_low'] = abs(df['close'].shift(1) - df['low'])
    df['true_range'] = df[['tr_high_low', 'tr_close_high', 'tr_close_low']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['plus_dm'] = df['high'] - df['high'].shift(1)
    df['plus_dm'] = df['plus_dm'].where((df['plus_dm'] > 0) & (df['plus_dm'] > (df['low'] - df['low'].shift(1))), 0)
    
    # Calculate Negative Directional Movement (-DM)
    df['minus_dm'] = df['low'].shift(1) - df['low']
    df['minus_dm'] = df['minus_dm'].where((df['minus_dm'] > 0) & (df['minus_dm'] > (df['high'] - df['high'].shift(1))), 0)
    
    # Smooth +DM and -DM Over 14 Periods
    df['plus_dm_smoothed'] = df['plus_dm'].rolling(window=14).sum() / 14
    df['minus_dm_smoothed'] = df['minus_dm'].rolling(window=14).sum() / 14
    
    # Calculate +DI and -DI
    df['plus_di'] = df['plus_dm_smoothed'] / df['atr']
    df['minus_di'] = df['minus_dm_smoothed'] / df['atr']
    
    # Calculate ADMI
    df['admi'] = (df['plus_di'] - df['minus_di']) / (df['plus_di'] + df['minus_di'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['preliminary_alpha_factor'] * df['admi']) + (df['intraday_preliminary_factor'] * df['admi'])
    
    return df['final_alpha_factor']
```
