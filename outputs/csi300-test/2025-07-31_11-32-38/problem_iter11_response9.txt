```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the average price
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the percentage change in volume
    volume_change = df['volume'].pct_change()
    
    # Calculate the percentage change in average price
    avg_price_change = avg_price.pct_change()
    
    # Calculate the factor
    factor = (price_range / avg_price) * volume_change * avg_price_change
    
    return factor
```
