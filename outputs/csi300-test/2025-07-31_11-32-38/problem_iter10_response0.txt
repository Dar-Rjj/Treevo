```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Compute Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(n)
    
    # Calculate Cumulative Volume from day (t-n) to day (t-1)
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    
    # Weight by Cumulative Volume
    volume_weighted_momentum = raw_momentum / cumulative_volume
    
    # Calculate Price Range
    price_range = df['high'] - df['low']
    
    # Combine with Volume-Weighted Momentum
    price_range_adjusted_momentum = volume_weighted_momentum * price_range + volume_weighted_momentum
    
    # Calculate Daily Volume Relative to 20-Day Moving Average
    volume_20_day_ma = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / volume_20_day_ma
    
    # Weigh by Intraday Volume Activity
    volume_activity_adjusted_momentum = price_range_adjusted_momentum * volume_ratio
    
    # Calculate Open to Close Change
    open_to_close_change = df['close'] - df['open']
    
    # Integrate Open to Close Price Movement
    final_momentum = volume_activity_adjusted_momentum * open_to_close_change + volume_activity_adjusted_momentum
    
    return final_momentum.dropna()
```
