```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()

    # Calculate Logarithmic Volume Difference
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_diff'] = df['log_volume'].diff()
    df['smoothed_volume_diff'] = df['log_volume_diff'].ewm(span=n, adjust=False).mean()

    # Calculate Momentum
    df['momentum'] = df['daily_returns'] * df['smoothed_volume_diff']

    # Integrate Price Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=n).mean()
    
    # Final Factor Calculation
    df['vol_adjusted_momentum'] = df['momentum'] / df['avg_high_low_range']
    df['accumulated_vol_adjusted_momentum'] = df['vol_adjusted_momentum'].rolling(window=n).sum()

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']

    # Combine Intraday and Close-to-Open Returns
    intraday_weight = np.abs(df['intraday_return'])
    close_to_open_weight = np.abs(df['close_to_open_return'])
    total_weight = intraday_weight + close_to_open_weight
    df['combined_return'] = (intraday_weight * df['intraday_return'] + close_to_open_weight * df['close_to_open_return']) / total_weight

    # Integrate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['interim_factor'] = df['combined_return'] * df['volume_change']

    # Calculate Volume-Weighted Momentum
    raw_momentum = (df['close'] / df['close'].shift(n)) - 1
    cumulative_volume = df['volume'].rolling(window=n).sum()
    df['volume_weighted_momentum'] = raw_momentum / cumulative_volume

    # Incorporate Price Range Factor
    df['price_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['volume_weighted_momentum'] * df['price_range'] + df['volume_weighted_momentum']

    # Final Factor Composition
    df['final_factor'] = df['accumulated_vol_adjusted_momentum'] + df['combined_momentum'] + df['interim_factor']

    return df['final_factor'].dropna()
```
