```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate simple moving averages (SMA)
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Measure short-term and medium-term momentum
    df['momentum_1d'] = df['close'].pct_change()
    df['momentum_n_days'] = df['close'] - df['close'].shift(10)
    
    # Analyze trading volume
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['vol_10d_ma'] = df['volume'].rolling(window=10).mean()
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Use high, low, and range for additional volatility insights
    df['high_low_diff'] = df['high'] - df['low']
    df['atr_20'] = df['high_low_diff'].rolling(window=20).mean()
    
    # Integrate open and close prices with volume
    df['open_close_pct_change'] = (df['close'] - df['open']) / df['open']
    df['open_close_diff_sum'] = df['close'] - df['open'].rolling(window=10).sum()
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['log_volume_diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Develop a volume-weighted daily returns factor
    df['daily_returns'] = df['close'].pct_change()
    df['volume_weighted_returns'] = df['daily_returns'] * df['log_volume_diff']
    df['ema_5d'] = df['volume_weighted_returns'].ewm(span=5, adjust=False).mean()
    
    # Incorporate intraday range breakouts
    df['intraday_breakout'] = (df['high'] - df['high'].shift(1)) + (df['low'].shift(1) - df['low'])
    df['intraday_breakout_weighted'] = df['intraday_breakout'] * df['volume_weight']
    
    # Combine multiple factors for a more robust signal
    df['combined_momentum'] = (df['close_to_open_return'] + df['open_close_pct_change']) * abs(df['open_close_pct_change']) * df['high_low_diff'] * df['volume_ratio']
    
    # Final alpha factor
    df['alpha_factor'] = df['combined_momentum'] + df['volume_weighted_returns'] + df['intraday_breakout_weighted']
    
    return df['alpha_factor'].dropna()
```
