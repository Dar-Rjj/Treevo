```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Compute Raw Momentum
    df['RawMomentum'] = df['close'] - df['close'].shift(n)
    
    # Weight by Cumulative Volume
    df['CumulativeVolume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['VolumeWeightedMomentum'] = df['RawMomentum'] / df['CumulativeVolume']
    
    # Calculate True Range (TR)
    df['HighLow'] = df['high'] - df['low']
    df['PrevCloseHigh'] = (df['close'].shift(1) - df['high']).abs()
    df['PrevCloseLow'] = (df['close'].shift(1) - df['low']).abs()
    df['TrueRange'] = df[['HighLow', 'PrevCloseHigh', 'PrevCloseLow']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['ATR'] = df['TrueRange'].rolling(window=n).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+DM'] = 0
    df.loc[(df['high'] > df['high'].shift(1)) & (df['high'].shift(1) != df['low'].shift(1)), '+DM'] = df['high'] - df['high'].shift(1)
    
    # Calculate Negative Directional Movement (-DM)
    df['-DM'] = 0
    df.loc[(df['low'] < df['low'].shift(1)) & (df['low'].shift(1) != df['high'].shift(1)), '-DM'] = df['low'].shift(1) - df['low']
    
    # Smooth +DM and -DM over n periods
    df['+DMSmoothed'] = df['+DM'].rolling(window=n).sum() / n
    df['-DMSmoothed'] = df['-DM'].rolling(window=n).sum() / n
    
    # Calculate +DI and -DI
    df['+DI'] = df['+DMSmoothed'] / df['ATR']
    df['-DI'] = df['-DMSmoothed'] / df['ATR']
    
    # Calculate ADMI
    df['ADMI'] = (df['+DI'] - df['-DI']) / (df['+DI'] + df['-DI'])
    
    # Combine Volume-Weighted Momentum with ADMI
    df['VolumeAdjustedMomentumWithADMI'] = df['VolumeWeightedMomentum'] * df['ADMI'] + df['VolumeWeightedMomentum']
    
    return df['VolumeAdjustedMomentumWithADMI']
```
