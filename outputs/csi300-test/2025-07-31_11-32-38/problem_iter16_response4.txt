```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-day and 30-day simple moving average (SMA) of close prices
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_30'] = df['close'].rolling(window=30).mean()

    # Calculate the percentage change in close price from one day to the next
    df['close_pct_change'] = df['close'].pct_change()

    # Calculate daily volume change as a percentage of the previous day's volume
    df['volume_pct_change'] = df['volume'].pct_change()

    # Calculate the true range
    df['true_range'] = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1)

    # Calculate the average true range (ATR) over a 14-day period
    df['ATR_14'] = df['true_range'].rolling(window=14).mean()

    # Multiplying the 10-day SMA by the daily volume
    df['SMA_10_volume'] = df['SMA_10'] * df['volume']

    # Multiplying the 30-day SMA by the daily volume
    df['SMA_30_volume'] = df['SMA_30'] * df['volume']

    # Calculate the difference between the high and low of the day
    df['high_low_diff'] = df['high'] - df['low']

    # Average the high-low difference over a 10-day period
    df['high_low_avg_10'] = df['high_low_diff'].rolling(window=10).mean()

    # Average the high-low difference over a 30-day period
    df['high_low_avg_30'] = df['high_low_diff'].rolling(window=30).mean()

    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (df['SMA_10'] + df['SMA_30'] + df['SMA_10_volume'] + df['SMA_30_volume'] + 
                          df['high_low_avg_10'] + df['high_low_avg_30'] + df['ATR_14']) / 7

    return df['alpha_factor']
```
