```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Daily Returns
    df['Volume_Weighted_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Logarithmic Volume Difference
    df['Log_Volume_Diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # 20-Day Moving Average of Volume
    df['Vol_20DMA'] = df['volume'].rolling(window=20).mean()
    
    # Weigh by Intraday Volume Activity
    df['Volume_Ratio'] = df['volume'] / df['Vol_20DMA']
    df['Weighted_Intraday_High_Low_Spread'] = df['Intraday_High_Low_Spread'] * df['Volume_Ratio']
    
    # Combine Intraday and Close-to-Open Returns
    df['Combined_Returns'] = abs(df['Intraday_Return']) + abs(df['Close_to_Open_Return'])
    
    # Integrate Volume Change and Intraday Volatility
    df['Adjusted_Weights'] = df['Combined_Returns'] * df['Log_Volume_Diff'] * df['Weighted_Intraday_High_Low_Spread']
    
    # Adjusted Momentum with Price Range and Volume
    n = 20  # Number of days for momentum
    df['Raw_Momentum'] = df['close'] - df['close'].shift(n)
    df['Cumulative_Volume'] = df['volume'].rolling(window=n).sum()
    df['Volume_Adjusted_Momentum'] = df['Raw_Momentum'] / df['Cumulative_Volume']
    df['Price_Range'] = df['high'] - df['low']
    df['Momentum_with_Price_Range'] = df['Volume_Adjusted_Momentum'] + df['Price_Range']
    
    # Weigh Combined Momentum by Intraday Volume Activity
    df['Weighted_Momentum'] = df['Momentum_with_Price_Range'] * df['Volume_Ratio']
    
    # Apply 5-Day Exponential Moving Average
    alpha = 0.2
    df['EMA_5D'] = df['Weighted_Momentum'].ewm(span=5, adjust=False).mean()
    
    # Calculate Price Momentum Factor
    df['Price_Momentum_Factor'] = df['close'] - df['close'].shift(n)
    
    # Incorporate Trading Volume into the Analysis
    m = 20  # Number of days for average volume
    df['Volume_Ratio_t_m'] = df['volume'] / df['volume'].rolling(window=m).mean()
    
    # Explore the Relationship Between Open and Close Prices
    p = 5  # Window for moving sum
    df['Open_Close_Diff'] = df['close'] - df['open']
    df['Moving_Sum_Open_Close'] = df['Open_Close_Diff'].rolling(window=p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    q = 5  # Window for moving average of ranges
    df['Daily_Range'] = df['high'] - df['low']
    df['MA_Daily_Range'] = df['Daily_Range'].rolling(window=q).mean()
    
    # Final Factor Calculation
    df['Final_Factor'] = df['Volume_Weighted_Return'] * df['Adjusted_Weights'] * df['EMA_5D']
    
    # Incorporate Intraday Range Breakouts
    df['Prev_High'] = df['high'].shift(1)
    df['Prev_Low'] = df['low'].shift(1)
    df['Breakout'] = (df['high'] > df['Prev_High']) | (df['low'] < df['Prev_Low'])
    df['Weighted_Breakout'] = df['Breakout'] * df['Volume_Ratio']
    df['Final_Factor'] = df['Final_Factor'] + df['Weighted_Breakout']
    
    return df['Final_Factor']
```
