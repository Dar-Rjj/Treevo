import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=20, p=10, q=10):
    # Calculate Price Momentum Factor
    df['momentum'] = df['close'] - df['close'].shift(n)
    
    # Incorporate Trading Volume into the Analysis
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(m).mean()
    
    # Explore the Relationship Between Open and Close Prices
    df['open_close_diff'] = df['open'] - df['close']
    df['sum_open_close_diff'] = df['open_close_diff'].rolling(p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    df['range'] = df['high'] - df['low']
    df['avg_range'] = df['range'].rolling(q).mean()
    
    # Calculate Volume-Weighted Daily Returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume'] / df['volume'].shift(1))
    
    # Weigh by Intraday Volume Activity
    df['vol_20_day_avg'] = df['volume'].rolling(20).mean()
    df['intraday_vol_activity'] = df['volume'] / df['vol_20_day_avg']
    
    # Integrate Components
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    df['integrated_returns'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['log_volume_diff'] * df['intraday_vol_activity']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['cumulative_volume'] = df['volume'].rolling(n).sum()
    df['volume_weighted_momentum'] = (df['close'] - df['close'].shift(n)) / df['cumulative_volume']
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['range']
    df['combined_momentum'] *= df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_combined_momentum'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    df['final_factor'] = df['volume_weighted_return'] * df['integrated_returns'] * df['ema_combined_momentum']
    
    # Incorporate Intraday Range Breakouts
    df['prev_high'] = df['high'].shift(1)
    df['prev_low'] = df['low'].shift(1)
    df['intraday_breakout'] = (df['high'] > df['prev_high']) & (df['low'] < df['prev_low'])
    df['breakout_weight'] = df['intraday_breakout'] * df['intraday_vol_activity']
    df['final_factor'] += df['breakout_weight']
    
    return df['final_factor'].dropna()
