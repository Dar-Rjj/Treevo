import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Compute 20-Day Moving Average of Volume Weighted Returns
    df['20_day_ma_vol_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Calculate Daily Price Range
    df['daily_price_range'] = df['high'] - df['low']
    
    # Compute 30-Day Moving Average of Daily Price Ranges
    df['30_day_ma_price_range'] = df['daily_price_range'].rolling(window=30).mean()
    
    # Final Alpha Factor
    df['difference'] = df['10_day_ma_price_change'] - df['20_day_ma_vol_weighted_return']
    df['adjusted_difference'] = df['difference'] / df['30_day_ma_price_range']
    df['final_alpha_factor'] = df['adjusted_difference'].rolling(window=5).mean()
    
    return df['final_alpha_factor'].dropna()
