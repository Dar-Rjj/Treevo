import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum Using Close Prices
    n = 20
    momentum = df['close'] - df['close'].shift(n)
    
    # Incorporate Trading Volume into the Analysis
    m = 20
    avg_volume = df['volume'].rolling(window=m).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Explore the Relationship Between Open and Close Prices
    p = 5
    open_close_diff = df['open'] - df['close']
    open_close_sum = open_close_diff.rolling(window=p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    q = 10
    daily_range = df['high'] - df['low']
    range_moving_avg = daily_range.rolling(window=q).mean()
    
    # Calculate Volume-Weighted Daily Returns
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    volume_weighted_return = daily_return * df['volume']
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    log_volume_diff = np.log(df['volume'] / df['volume'].shift(1))
    
    # Weigh by Intraday Volume Activity
    volume_activity = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine Intraday and Close-to-Open Returns
    intraday_weight = np.abs(intraday_return)
    close_to_open_weight = np.abs(close_to_open_return)
    combined_return = (intraday_weight * intraday_return + close_to_open_weight * close_to_open_return) / (intraday_weight + close_to_open_weight)
    
    # Adjust weights by multiplying them with the Logarithmic Volume Difference and Intraday High-Low Spread Weight
    adjusted_return = combined_return * log_volume_diff * (daily_range / df['open'])
    
    # Combine Volume-Weighted Momentum with Price Range
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    volume_weighted_momentum = momentum / cumulative_volume
    combined_momentum = volume_weighted_momentum + daily_range
    
    # Multiply Combined Momentum by Volume Ratio
    momentum_factor = combined_momentum * volume_ratio
    
    # Apply 5-Day Exponential Moving Average
    ema_factor = momentum_factor.ewm(span=5, adjust=False).mean()
    
    # Incorporate Intraday Range Breakouts
    intraday_breakout = (df['high'] - df['low'].shift(1)) / df['low'].shift(1)
    weighted_breakout = intraday_breakout * volume_activity
    
    # Evaluate Additional Momentum Indicators
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()
    close_change = df['close'].pct_change()
    volume_change = df['volume'].pct_change()
    
    # Enhance Volatility Analysis
    true_range = np.maximum(np.maximum(df['high'] - df['low'], np.abs(df['high'] - df['close'].shift(1))), np.abs(df['low'] - df['close'].shift(1)))
    atr_14 = true_range.rolling(window=14).mean()
    
    # Combine Price and Volume for Comprehensive Factors
    sma_10_vol = sma_10 * df['volume']
    sma_30_vol = sma_30 * df['volume']
    
    # Incorporate High and Low Prices for Additional Volatility Insights
    high_low_diff = df['high'] - df['low']
    high_low_10 = high_low_diff.rolling(window=10).mean()
    high_low_30 = high_low_diff.rolling(window=30).mean()
    
    # Final Alpha Factor
    alpha_factor = (ema_factor + weighted_breakout + sma_10_vol + sma_30_vol + high_low_10 + high_low_30 + atr_14) / 7
    return alpha_factor
