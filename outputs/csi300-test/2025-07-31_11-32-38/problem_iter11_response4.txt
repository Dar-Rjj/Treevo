import numpy as np

    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # Compute 10-Day Moving Average of Daily Price Changes
    df['MA_10_Daily_Price_Change'] = df['Daily_Price_Change'].rolling(window=10).mean()

    # Compute Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['open']

    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']

    df['Log_Volume'] = np.log(df['volume'])
    df['Log_Volume_Diff'] = df['Log_Volume'].diff()
    df['Smoothed_Volume_Diff'] = df['Log_Volume_Diff'].ewm(alpha=0.2).mean()
    df['Preliminary_Factor'] = df['Combined_Return'] * df['Smoothed_Volume_Diff']
