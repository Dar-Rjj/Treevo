```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Compute Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(n)
    
    # Weight by Cumulative Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    
    # Incorporate Price Range Factor
    df['price_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['price_range']
    
    # Weigh by Intraday Volume Activity
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_20_day_ma']
    df['weighted_combined_momentum'] = df['combined_momentum'] * df['volume_ratio']
    
    # Smooth Combined Momentum
    df['smoothed_momentum'] = df['weighted_combined_momentum'].ewm(span=5, adjust=False).mean()
    
    return df['smoothed_momentum']
```
