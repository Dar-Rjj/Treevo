import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Returns
    df['raw_returns'] = df['close'].pct_change()
    
    # Momentum Component
    df['ema_10'] = df['raw_returns'].ewm(span=10, adjust=False).mean()
    df['ema_5'] = df['raw_returns'].ewm(span=5, adjust=False).mean()
    df['momentum'] = df['ema_10'] - df['ema_5']
    
    # Reversal Component
    weights = np.arange(1, 22)
    df['wma_21'] = df['raw_returns'].rolling(window=21).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    df['reversal'] = -df['wma_21']
    
    # Scale Components
    momentum_scaled = df['momentum'] / df['momentum'].abs().max()
    reversal_scaled = df['reversal'] / df['reversal'].abs().max()
    
    # Combine Momentum and Reversal
    df['factor'] = (momentum_scaled + reversal_scaled) * np.sqrt(df['volume'])
    
    return df['factor']
