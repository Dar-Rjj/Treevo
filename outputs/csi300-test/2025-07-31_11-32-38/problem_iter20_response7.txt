```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Daily Returns
    df['volume_weighted_return'] = (df['close'] / df['close'].shift(1) - 1) * df['volume'].shift(1)
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume'] / df['volume'].shift(1))
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=22).mean()
    
    # Integrate Components
    df['intraday_close_to_open_combined'] = (abs(df['intraday_return']) * df['intraday_return'] + 
                                             abs(df['close_to_open_return']) * df['close_to_open_return']) / 2
    
    df['adjusted_weights'] = df['intraday_close_to_open_combined'] * df['log_volume_diff'] * df['intraday_high_low_spread']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['raw_momentum'] = df['close'] / df['close'].shift(20) - 1
    df['cumulative_volume'] = df['volume'].rolling(window=20).sum()
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    df['price_range'] = df['high'] - df['low']
    df['volume_weighted_momentum_with_price_range'] = df['volume_weighted_momentum'] + df['price_range']
    
    # Multiply Combined Momentum by Volume Ratio
    df['combined_momentum'] = df['volume_weighted_momentum_with_price_range'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_5'] = df['combined_momentum'].ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    df['weighted_returns'] = df['volume_weighted_return'] * df['adjusted_weights']
    
    # Incorporate Intraday Range Breakouts
    df['intraday_range_breakout'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    df['breakout_weighted'] = df['intraday_range_breakout'] * df['volume_ratio']
    
    # Analyze Close Price Trends
    df['close_price_trend'] = df['close'].pct_change()
    
    # Examine the Relationship Between Close Price and Volume
    df['close_to_avg_volume_ratio'] = df['close'] / df['volume'].rolling(window=20).mean()
    
    # Look at Intra-day Price Action
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_range_above_prev_close'] = (df['high'] - df['close'].shift(1)) / df['intraday_range']
    
    # Consider the Open-to-Close Behavior
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_direction'] = np.sign(df['open_close_diff'])
    
    # Evaluate the High and Low Prices
    df['consecutive_higher_highs'] = (df['high'] > df['high'].shift(1)).astype(int).cumsum()
    df['consecutive_lower_lows'] = (df['low'] < df['low'].shift(1)).astype(int).cumsum()
    
    # Assess the Volatility
    df['daily_return_volatility'] = df['close'].pct_change().rolling(window=20).std()
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['ema_5'] * df['close_to_avg_volume_ratio'] * 
                          df['intraday_range_above_prev_close'] * 
                          df['daily_return_volatility'])
    
    return df['alpha_factor']
```
