```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * df['intraday_return'].abs()
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['close_to_open_return'].abs()
    df['combined_return'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return'])
    
    # Integrate Volume Change
    df['daily_volume_change'] = df['volume'].pct_change()
    df['preliminary_factor'] = df['combined_return'] * df['daily_volume_change']
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift()).abs()
    df['tr3'] = (df['low'] - df['close'].shift()).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR) over 14 periods
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+dm'] = 0
    condition = (df['high'] > df['high'].shift()) & (df['high'] - df['high'].shift() > df['low'].shift() - df['low'])
    df.loc[condition, '+dm'] = df['high'] - df['high'].shift()
    
    # Calculate Negative Directional Movement (-DM)
    df['-dm'] = 0
    condition = (df['low'].shift() > df['low']) & (df['low'].shift() - df['low'] > df['high'] - df['high'].shift())
    df.loc[condition, '-dm'] = df['low'].shift() - df['low']
    
    # Smooth +DM and -DM over 14 periods
    df['+dm_smoothed'] = df['+dm'].rolling(window=14).mean()
    df['-dm_smoothed'] = df['-dm'].rolling(window=14).mean()
    
    # Calculate +DI and -DI
    df['+di'] = 100 * (df['+dm_smoothed'] / df['atr'])
    df['-di'] = 100 * (df['-dm_smoothed'] / df['atr'])
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['20_day_ma_volume_weighted_return'] = (df['daily_price_change'] * df['volume']).rolling(window=20).mean() / df['volume'].rolling(window=20).mean()
    df['difference'] = df['10_day_ma_price_change'] - df['20_day_ma_volume_weighted_return']
    df['5_day_ma_difference'] = df['difference'].rolling(window=5).mean()
    df['final_alpha_factor'] = df['5_day_ma_difference'] * df['preliminary_factor'] * df['admi']
    
    return df['final_alpha_factor']
```
