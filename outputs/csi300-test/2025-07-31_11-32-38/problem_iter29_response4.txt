import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor from the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series with the calculated alpha factor.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price
    vwap = (typical_price * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Calculate the 10-day moving average of the closing price
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the factor as the difference between the VWAP and the 10-day moving average
    factor = vwap - ma_10
    
    return factor
