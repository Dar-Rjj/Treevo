```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['price_change'].rolling(window=10).mean()
    
    # Calculate 20-Day Moving Average of Daily Price Changes
    df['20_day_ma_price_change'] = df['price_change'].rolling(window=20).mean()
    
    # Combine Moving Averages
    df['ma_diff'] = df['10_day_ma_price_change'] - df['20_day_ma_price_change']
    df['5_day_ma_diff'] = df['ma_diff'].rolling(window=5).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns
    df['20_day_ma_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Integrate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['preliminary_factor'] = df['volume_change'] * df['volume_weighted_return']
    
    # Calculate Intraday and Close-to-Open Returns
    df['intraday_return'] = df['high'] - df['low']
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    
    # Adjust Weights by Logarithmic Volume Difference
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_diff'] = df['log_volume'] - df['log_volume'].shift(1)
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(span=20, adjust=False).mean()
    
    df['combined_returns'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['smoothed_log_volume_diff']
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['tr'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Calculate Positive and Negative Directional Movement (+DM, -DM)
    df['+dm'] = (df['high'] - df['high'].shift(1)).apply(lambda x: max(x, 0))
    df['-dm'] = (df['low'].shift(1) - df['low']).apply(lambda x: max(x, 0))
    
    # Calculate +DI and -DI
    df['+di'] = (df['+dm'] / df['atr']) * 100
    df['-di'] = (df['-dm'] / df['atr']) * 100
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['5_day_ma_diff'] * df['admi'] +
        df['20_day_ma_volume_weighted_return'] * df['admi'] +
        df['combined_returns'] * df['admi']
    )
    
    return df['final_alpha_factor']
```
