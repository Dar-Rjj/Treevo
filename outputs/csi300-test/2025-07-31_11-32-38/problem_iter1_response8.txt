```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Returns
    df['raw_returns'] = df['close'].pct_change()
    
    # Momentum Component
    df['10_day_moving_sum'] = df['raw_returns'].rolling(window=10).sum()
    df['5_day_moving_sum'] = df['raw_returns'].rolling(window=5).sum()
    df['momentum_component'] = df['10_day_moving_sum'] - df['5_day_moving_sum']
    
    # Reversal Component
    df['21_day_moving_avg'] = df['raw_returns'].rolling(window=21).mean()
    df['reversal_component'] = -df['21_day_moving_avg']
    
    # Normalize Momentum and Reversal Components
    momentum_std = df['momentum_component'].std()
    reversal_std = df['reversal_component'].std()
    
    df['normalized_momentum'] = df['momentum_component'] / momentum_std
    df['normalized_reversal'] = df['reversal_component'] / reversal_std
    
    # Combine Normalized Components
    df['alpha_factor'] = (df['normalized_momentum'] + df['normalized_reversal']) * df['volume']
    
    return df['alpha_factor'].dropna()
```
