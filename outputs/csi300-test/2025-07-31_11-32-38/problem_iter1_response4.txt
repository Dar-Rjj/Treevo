```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Compute Raw Momentum
    df['raw_momentum'] = df['close'] - df['close'].shift(n)
    
    # Calculate Cumulative Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    
    # Weight by Cumulative Volume to get Volume-Weighted Momentum
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    
    # Calculate Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Combine with Volume-Weighted Momentum
    df['adjusted_momentum'] = df['volume_weighted_momentum'] * df['price_range'] + df['volume_weighted_momentum']
    
    return df['adjusted_momentum']
```
