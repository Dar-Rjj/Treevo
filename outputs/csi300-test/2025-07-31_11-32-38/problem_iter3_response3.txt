```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Rolling Average of Daily Price Changes (10-Day Moving Average)
    df['rolling_avg_daily_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_amount_weighted_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns (20-Day Moving Average)
    df['rolling_avg_volume_amount_weighted_return'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_difference'] = df['rolling_avg_daily_price_change'] - df['rolling_avg_volume_amount_weighted_return']
    
    # Smooth the Result (5-Day Moving Average)
    df['smoothed_signal'] = df['signal_difference'].rolling(window=5).mean()
    
    # Final Alpha Factor
    alpha_factor = df['smoothed_signal']
    
    return alpha_factor
```
