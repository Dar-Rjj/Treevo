```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Daily Returns
    df['Volume-Weighted Return'] = (df['close'].pct_change() * df['volume']).shift(1)
    
    # Compute Intraday Return
    df['Intraday Return'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Return
    df['Close-to-Open Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Logarithmic Volume Difference
    df['Log Volume'] = np.log(df['volume'])
    df['Log Volume Diff'] = df['Log Volume'] - df['Log Volume'].shift(1)
    df['Smoothed Log Volume Diff'] = df['Log Volume Diff'].ewm(alpha=0.2).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday High-Low Spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['Volume Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Intraday High-Low Weight'] = df['Intraday High-Low Spread'] * df['Volume Ratio']
    
    # Combine Intraday and Close-to-Open Returns
    df['Combined Return'] = (df['Intraday Return'] * abs(df['Intraday Return'])) + (df['Close-to-Open Return'] * abs(df['Close-to-Open Return']))
    
    # Integrate Volume Change and Intraday Volatility
    df['Adjusted Combined Return'] = df['Combined Return'] * df['Smoothed Log Volume Diff'] * df['Intraday High-Low Weight']
    
    # Combine Volume-Weighted Momentum with Price Range
    n = 5  # Number of days for momentum
    df['Raw Momentum'] = df['close'].pct_change(periods=n)
    df['Cumulative Volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['Volume-Weighted Momentum'] = df['Raw Momentum'] / df['Cumulative Volume']
    df['Price Range'] = df['high'] - df['low']
    df['Combined Momentum'] = df['Volume-Weighted Momentum'] + df['Price Range']
    
    # Weigh Combined Momentum by Intraday Volume Activity
    df['Weighted Combined Momentum'] = df['Combined Momentum'] * df['Volume Ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['EMA Factor'] = df['Weighted Combined Momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Final Factor Calculation
    df['Final Factor'] = df['Volume-Weighted Return'] * df['Adjusted Combined Return'].shift(1) * df['EMA Factor']
    
    # Incorporate Intraday Range Breakouts
    df['Intraday Range Breakout'] = (df['high'] > df['high'].shift(1)) | (df['low'] < df['low'].shift(1))
    df['Breakout Weight'] = df['Intraday Range Breakout'] * df['Volume Ratio']
    df['Final Factor'] += df['Intraday High-Low Spread'] * df['Breakout Weight']
    
    return df['Final Factor']
```
