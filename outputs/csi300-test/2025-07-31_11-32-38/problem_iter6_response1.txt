```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    volume_20_day_ma = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / volume_20_day_ma
    weighted_high_low_spread = high_low_spread * volume_ratio
    
    # Incorporate Intraday Price Trend
    intraday_return = (df['close'] - df['open']) / df['open']
    normalized_intraday_return = intraday_return - intraday_return.rolling(window=5).mean()
    
    # Combine the weighted high-low spread with the normalized intraday return
    combined_factor = weighted_high_low_spread + normalized_intraday_return
    
    # Apply 5-Day Exponential Moving Average
    ema_factor = combined_factor.ewm(span=5, adjust=False).mean()
    
    return ema_factor
```
