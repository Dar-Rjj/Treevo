```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, alpha=0.3):
    """
    Calculate the Volume-Adjusted Momentum Indicator.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    - n: Number of days to accumulate the factor
    - alpha: Smoothing factor for exponential smoothing
    
    Returns:
    - pandas Series representing the factor values indexed by date
    """
    
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate Logarithmic Volume
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t-1'] = df['log_volume_t'].shift(1)
    
    # Subtract and Smooth Volume Difference
    df['log_vol_diff'] = df['log_volume_t'] - df['log_volume_t-1']
    df['smoothed_log_vol_diff'] = df['log_vol_diff'].ewm(alpha=alpha).mean()
    
    # Calculate Momentum
    df['weighted_daily_return'] = df['daily_return'] * df['smoothed_log_vol_diff']
    
    # Final Factor Calculation
    df['factor'] = df['weighted_daily_return'].rolling(window=n).sum()
    
    return df['factor']
```
