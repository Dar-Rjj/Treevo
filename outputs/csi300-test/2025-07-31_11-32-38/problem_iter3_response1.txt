```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Daily Volume Change
    daily_volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Daily Amount Change
    daily_amount_change = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    
    # Weight Intraday and Close-to-Open Returns by their absolute values
    weighted_intraday_return = intraday_return * intraday_return.abs()
    weighted_close_to_open_return = close_to_open_return * close_to_open_return.abs()
    
    # Adjust weights by multiplying them with the Daily Volume and Amount Changes
    adjusted_intraday_weight = weighted_intraday_return * daily_volume_change * daily_amount_change
    adjusted_close_to_open_weight = weighted_close_to_open_return * daily_volume_change * daily_amount_change
    
    # Sum the adjusted weights to get the final factor
    final_factor = adjusted_intraday_weight + adjusted_close_to_open_weight
    
    # Smooth the Final Factor with a 3-day moving average
    smoothed_factor = final_factor.rolling(window=3).mean()
    
    return smoothed_factor
```
