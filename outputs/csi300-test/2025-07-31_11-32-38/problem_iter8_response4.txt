```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Volume-Weighted Momentum
    raw_momentum = df['close'] - df['close'].shift(n)
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    volume_weighted_momentum = raw_momentum / cumulative_volume

    # Incorporate Price Range Factor
    price_range = df['high'] - df['low']
    combined_momentum = volume_weighted_momentum * price_range + volume_weighted_momentum

    # Integrate Open to Close Price Movement
    open_to_close_change = df['close'] - df['open']
    combined_momentum += combined_momentum * open_to_close_change

    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(1) - df['open']) / df['open']

    # Combine Intraday and Close-to-Open Returns
    weighted_intraday_return = intraday_return * abs(intraday_return)
    weighted_close_to_open_return = close_to_open_return * abs(close_to_open_return)

    # Integrate Volume Change
    daily_volume_change = df['volume'] / df['volume'].shift(1)
    preliminary_factor = (weighted_intraday_return + weighted_close_to_open_return) * daily_volume_change

    # Calculate True Range (TR)
    tr_high_low = df['high'] - df['low']
    tr_close_high = abs(df['close'].shift(1) - df['high'])
    tr_close_low = abs(df['close'].shift(1) - df['low'])
    true_range = pd.concat([tr_high_low, tr_close_high, tr_close_low], axis=1).max(axis=1)

    # Calculate Average True Range (ATR)
    atr = true_range.rolling(window=n).mean()

    # Calculate Positive Directional Movement (+DM)
    positive_dm = (df['high'] - df['high'].shift(1)).apply(lambda x: x if x > 0 else 0)
    negative_dm = (df['low'].shift(1) - df['low']).apply(lambda x: x if x > 0 else 0)

    # Smooth +DM and -DM over n periods
    smooth_positive_dm = positive_dm.rolling(window=n).sum() / n
    smooth_negative_dm = negative_dm.rolling(window=n).sum() / n

    # Calculate +DI and -DI
    positive_di = smooth_positive_dm / atr
    negative_di = smooth_negative_dm / atr

    # Calculate ADMI
    admi = (positive_di - negative_di) / (positive_di + negative_di)

    # Synthesize Final Alpha Factor
    daily_price_changes = df['close'] - df['close'].shift(1)
    ma_10 = daily_price_changes.rolling(window=10).mean()
    ma_20 = volume_weighted_momentum.rolling(window=20).mean()
    final_factor = (ma_10 - ma_20).rolling(window=5).mean() * preliminary_factor * admi

    return final_factor
```
