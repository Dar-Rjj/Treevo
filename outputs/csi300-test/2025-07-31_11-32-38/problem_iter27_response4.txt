```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, p=5, q=10):
    # Calculate Price Momentum Factor
    df['price_momentum'] = df['close'].diff(n)
    
    # Incorporate Trading Volume into the Analysis
    df['avg_volume_m'] = df['volume'].rolling(window=m).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_m']
    
    # Explore the Relationship Between Open and Close Prices
    df['open_close_diff'] = df['close'] - df['open']
    df['sum_open_close_diff_p'] = df['open_close_diff'].rolling(window=p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range_q'] = df['daily_range'].rolling(window=q).mean()
    
    # Calculate Volume-Weighted Daily Returns
    df['prev_close'] = df['close'].shift(1)
    df['volume_weighted_return'] = (df['close'] / df['prev_close'] - 1) * df['volume']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Weigh by Intraday Volume Activity
    df['vol_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['intraday_vol_activity'] = df['volume'] / df['vol_20_day_ma']
    
    # Integrate Components
    df['weighted_intraday_return'] = df['intraday_return'] * np.abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * np.abs(df['close_to_open_return'])
    df['adjusted_weights'] = df['log_volume_diff'] * df['intraday_vol_activity']
    df['combined_returns'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['adjusted_weights']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = (df['price_momentum'] / df['cumulative_volume']) + df['avg_daily_range_q']
    df['momentum_with_volume_ratio'] = df['volume_weighted_momentum'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_5_day'] = df['momentum_with_volume_ratio'].ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    df['final_factor'] = df['volume_weighted_return'] * df['adjusted_weights']
    
    # Incorporate Intraday Range Breakouts
    df['intraday_range_breakout'] = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))
    df['weighted_breakout'] = df['intraday_range_breakout'] * df['intraday_vol_activity']
    
    # Final factor
    df['alpha_factor'] = df['final_factor'] * df['weighted_breakout']
    
    return df['alpha_factor']
```
