```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Raw Returns
    df['raw_returns'] = df['close'].pct_change()

    # Calculate Volume-Weighted Momentum
    n = 10
    df['raw_momentum'] = df['close'] / df['close'].shift(n) - 1
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']

    # Incorporate Price Range Factor
    df['price_range'] = (df['high'] - df['low']) / df['close']
    df['price_range_momentum'] = df['volume_weighted_momentum'] * df['price_range']

    # Momentum Component
    df['10_day_moving_sum'] = df['raw_returns'].rolling(window=10).sum()
    df['5_day_moving_sum'] = df['raw_returns'].rolling(window=5).sum()
    df['momentum_component'] = df['10_day_moving_sum'] - df['5_day_moving_sum']

    # Reversal Component
    df['21_day_moving_avg'] = df['raw_returns'].rolling(window=21).mean()
    df['reversal_component'] = df['21_day_moving_avg'] * -1

    # Adjusted Momentum and Reversal
    df['adjusted_momentum_reversal'] = df['momentum_component'] + df['reversal_component']

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['raw_returns'].rolling(window=5).mean() - df['volume_weighted_momentum']).rolling(window=5).mean() * df['adjusted_momentum_reversal']

    return df['final_alpha_factor'].dropna()
```
