```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the Volume-Weighted Daily Return
    df['volume_weighted_return'] = (df['close'].pct_change() * df['volume']).fillna(0)
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Track Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine the Logarithmic Volume Difference
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_diff'] = df['log_volume'].diff().fillna(0)
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(span=5, adjust=False).mean()
    
    # Calculate Weigh by Intraday Volume Activity
    df['vol_ma_20'] = df['volume'].rolling(window=20).mean()
    df['intraday_vol_activity'] = df['volume'] / df['vol_ma_20']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'].abs() * df['intraday_vol_activity']
    df['weighted_close_to_open_return'] = df['close_to_open_return'].abs() * df['intraday_vol_activity']
    
    # Adjust weights by integrating volume and intraday volatility
    df['adjusted_weights'] = df['smoothed_log_volume_diff'] * (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Raw Momentum
    n = 20
    df['raw_momentum'] = df['close'] / df['close'].shift(n) - 1
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum()
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    
    # Add Price Range to Volume-Weighted Momentum
    df['price_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['volume_weighted_momentum_with_range'] = df['volume_weighted_momentum'] + df['price_range']
    
    # Apply 5-Day Exponential Moving Average
    df['factor'] = (df['volume_weighted_return'] * df['adjusted_weights']).ewm(span=5, adjust=False).mean()
    
    # Incorporate Intraday Range Breakouts
    df['intraday_range_breakout'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    df['breakout_weight'] = df['intraday_range_breakout'].astype(int) * df['intraday_vol_activity']
    df['factor_with_breakout'] = df['factor'] + df['breakout_weight']
    
    # Calculate the 5-Day Standard Deviation of the Factor
    df['factor_std_5d'] = df['factor_with_breakout'].rolling(window=5).std()
    
    # Determine the Signal-to-Noise Ratio
    df['factor_mean_5d'] = df['factor_with_breakout'].rolling(window=5).mean()
    df['signal_to_noise_ratio'] = df['factor_mean_5d'] / df['factor_std_5d']
    
    # Calculate the 5-Day Price Trend
    df['price_trend_5d'] = df['close'].rolling(window=5).mean() / df['close'].shift(5) - 1
    
    # Adjust the factor based on the trend
    df['factor_adjusted_by_trend'] = df['factor_with_breakout'] * (1 + df['price_trend_5d'])
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Adjust the factor by intraday volatility
    df['final_factor'] = df['factor_adjusted_by_trend'] * (1 + df['intraday_volatility'])

    return df['final_factor']
```
