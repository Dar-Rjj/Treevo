```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute 10-Day Exponential Moving Average of Daily Price Changes
    df['ema_10_daily_price_change'] = df['daily_price_change'].ewm(span=10).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Close-to-Open Returns
    df['combined_return'] = (df['intraday_return'].abs() * df['intraday_return'] + 
                             df['close_to_open_return'].abs() * df['close_to_open_return'])
    
    # Integrate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['preliminary_factor'] = df['combined_return'] * df['volume_change']
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR) over 14 periods
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['positive_dm'] = df['high'].diff().apply(lambda x: max(x, 0)) - df['low'].diff().apply(lambda x: max(-x, 0))
    
    # Calculate Negative Directional Movement (-DM)
    df['negative_dm'] = df['low'].diff().apply(lambda x: max(-x, 0)) - df['high'].diff().apply(lambda x: max(x, 0))
    
    # Smooth +DM and -DM over 14 periods
    df['smoothed_positive_dm'] = df['positive_dm'].rolling(window=14).sum() / 14
    df['smoothed_negative_dm'] = df['negative_dm'].rolling(window=14).sum() / 14
    
    # Calculate +DI and -DI
    df['plus_di'] = 100 * (df['smoothed_positive_dm'] / df['atr'])
    df['minus_di'] = 100 * (df['smoothed_negative_dm'] / df['atr'])
    
    # Calculate ADMI
    df['admi'] = (df['plus_di'] - df['minus_di']) / (df['plus_di'] + df['minus_di'])
    
    # Incorporate Volume Weighted Returns
    df['volume_weighted_intraday_return'] = df['intraday_return'] * df['volume']
    df['volume_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    df['volume_weighted_return'] = df['volume_weighted_intraday_return'] + df['volume_weighted_close_to_open_return']
    df['ema_20_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=20).mean()
    
    # Synthesize Final Alpha Factor
    df['final_difference'] = df['ema_10_daily_price_change'] - df['ema_20_volume_weighted_return']
    df['final_alpha_factor'] = df['final_difference'].ewm(span=5).mean() * df['preliminary_factor'] * df['admi']
    
    return df['final_alpha_factor']
```
