```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Weighted Sum of Intraday and Close-to-Open Returns
    weighted_intraday_volatility = intraday_volatility.abs() * intraday_volatility
    weighted_close_to_open_return = close_to_open_return.abs() * close_to_open_return
    preliminary_factor = weighted_intraday_volatility + weighted_close_to_open_return
    
    # Calculate True Range (TR)
    prev_close = df['close'].shift(1)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - prev_close).abs()
    tr3 = (df['low'] - prev_close).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Positive Directional Movement (+DM)
    prev_high = df['high'].shift(1)
    prev_low = df['low'].shift(1)
    plus_dm = (df['high'] - prev_high).clip(lower=0) - (prev_low - df['low']).clip(lower=0)
    plus_dm = plus_dm.where(plus_dm > 0, 0)
    
    # Calculate Negative Directional Movement (-DM)
    minus_dm = (prev_low - df['low']).clip(lower=0) - (df['high'] - prev_high).clip(lower=0)
    minus_dm = minus_dm.where(minus_dm > 0, 0)
    
    # Smooth +DM and -DM over 14 periods
    smooth_plus_dm = plus_dm.rolling(window=14).sum()
    smooth_minus_dm = minus_dm.rolling(window=14).sum()
    
    # Calculate +DI and -DI
    tr14 = true_range.rolling(window=14).sum()
    plus_di = 100 * (smooth_plus_dm / tr14)
    minus_di = 100 * (smooth_minus_dm / tr14)
    
    # Calculate ADMI
    admi = (plus_di - minus_di) / (plus_di + minus_di)
    
    # Multiply Preliminary Factor by ADMI
    alpha_factor = preliminary_factor * admi
    
    return alpha_factor
```
