```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    
    # Combine Intraday and Close-to-Open Returns
    df['Intraday_Return_Weighted'] = df['Intraday_Return'] * df['Intraday_Return'].abs()
    df['Close_to_Open_Return_Weighted'] = df['Close_to_Open_Return'] * df['Close_to_Open_Return'].abs()
    
    # Integrate Volume Change
    df['Preliminary_Factor'] = (df['Intraday_Return_Weighted'] + df['Close_to_Open_Return_Weighted']) * df['Volume_Change'].fillna(0)
    
    # Calculate True Range (TR)
    df['High_Low'] = df['high'] - df['low']
    df['High_PrevClose'] = (df['high'] - df['close'].shift(1)).abs()
    df['Low_PrevClose'] = (df['low'] - df['close'].shift(1)).abs()
    df['TR'] = df[['High_Low', 'High_PrevClose', 'Low_PrevClose']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['ATR'] = df['TR'].rolling(window=n).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+DM'] = 0.0
    df.loc[(df['high'] > df['high'].shift(1)) & (df['low'] <= df['low'].shift(1)), '+DM'] = df['high'] - df['high'].shift(1)
    
    # Calculate Negative Directional Movement (-DM)
    df['-DM'] = 0.0
    df.loc[(df['low'] < df['low'].shift(1)) & (df['high'] >= df['high'].shift(1)), '-DM'] = df['low'].shift(1) - df['low']
    
    # Calculate +DM Smoothed Over n Periods
    df['+DM_Smoothed'] = df['+DM'].rolling(window=n).sum() / n
    
    # Calculate -DM Smoothed Over n Periods
    df['-DM_Smoothed'] = df['-DM'].rolling(window=n).sum() / n
    
    # Calculate +DI
    df['+DI'] = df['+DM_Smoothed'] / df['ATR']
    
    # Calculate -DI
    df['-DI'] = df['-DM_Smoothed'] / df['ATR']
    
    # Calculate ADMI
    df['ADMI'] = (df['+DI'] - df['-DI']) / (df['+DI'] + df['-DI'])
    
    # Synthesize Final Alpha Factor
    df['Alpha_Factor'] = df['Preliminary_Factor'] * df['ADMI']
    
    return df['Alpha_Factor']
```
