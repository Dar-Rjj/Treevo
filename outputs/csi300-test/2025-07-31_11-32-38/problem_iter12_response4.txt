```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Rolling Average of Daily Price Changes (10-Day Moving Average)
    df['price_momentum_10d'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['volume_amount_weighted_return'] = (df['close'] / df['open']) * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns (20-Day Moving Average)
    df['volume_amount_weighted_ma_20d'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_difference'] = df['price_momentum_10d'] - df['volume_amount_weighted_ma_20d']
    df['preliminary_alpha_factor'] = df['signal_difference'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'] - 1
    
    # Integrate Volume Change
    df['volume_change'] = df['volume'].diff()
    df['intraday_adjusted'] = df['intraday_return'] * df['volume_change']
    df['close_to_open_adjusted'] = df['close_to_open_return'] * df['volume_change']
    df['intraday_preliminary_factor'] = df['intraday_adjusted'] + df['close_to_open_adjusted']
    
    # Calculate True Range (TR)
    df['tr1'] = abs(df['high'] - df['low'])
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR) (14-Day Moving Average)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['positive_dm'] = (df['high'] - df['high'].shift(1)).apply(lambda x: max(x, 0))
    df['negative_dm'] = (df['low'].shift(1) - df['low']).apply(lambda x: max(x, 0))
    
    # Smooth +DM and -DM over 14 periods
    df['smoothed_positive_dm'] = df['positive_dm'].rolling(window=14).sum() / 14
    df['smoothed_negative_dm'] = df['negative_dm'].rolling(window=14).sum() / 14
    
    # Calculate +DI and -DI
    df['di_positive'] = df['smoothed_positive_dm'] / df['atr']
    df['di_negative'] = df['smoothed_negative_dm'] / df['atr']
    
    # Calculate ADMI
    df['admi'] = (df['di_positive'] - df['di_negative']) / (df['di_positive'] + df['di_negative'])
    
    # Calculate Intraday Momentum
    df['intraday_momentum_10d'] = df['intraday_return'].rolling(window=10).mean()
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (
        df['preliminary_alpha_factor'] * df['admi'] +
        df['intraday_preliminary_factor'] * df['admi'] +
        df['intraday_momentum_10d'] * df['admi']
    )
    
    return df['final_alpha_factor']
```
