```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Calculate the price range for each day
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the average price
    df['avg_price'] = (df['open'] + df['close']) / 2
    
    # Calculate the return for the day
    df['return'] = df['close'] / df['open'] - 1
    
    # Calculate the cumulative volume over the past 5 days
    df['cumulative_volume_5d'] = df['volume'].rolling(window=5).sum()
    
    # Calculate the relative strength index (RSI) over the past 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the factor using a combination of the above metrics
    df['factor'] = (df['price_range'] / df['avg_price']) * df['return'] * (df['cumulative_volume_5d'] / df['volume']) * (100 - df['rsi'])
    
    return df['factor']
```
