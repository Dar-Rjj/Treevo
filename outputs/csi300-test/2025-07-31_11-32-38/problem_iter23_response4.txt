```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=20, p=10, q=10):
    # Calculate Price Momentum Factor
    df['price_momentum'] = df['close'] - df['close'].shift(n)
    
    # Incorporate Trading Volume into the Analysis
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=m).mean()
    
    # Explore the Relationship Between Open and Close Prices
    df['open_close_diff'] = df['close'] - df['open']
    df['moving_sum_open_close'] = df['open_close_diff'].rolling(window=p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    df['range'] = df['high'] - df['low']
    df['moving_avg_range'] = df['range'].rolling(window=q).mean()
    
    # Calculate Volume-Weighted Daily Returns
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volume_weighted_return'] = df['daily_return'] * df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Weigh by Intraday Volume Activity
    df['intraday_volume_activity'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrate Components
    # Combine Intraday and Close-to-Open Returns
    df['combined_returns'] = (df['intraday_return'].abs() * df['intraday_return'] + 
                              df['close_to_open_return'].abs() * df['close_to_open_return']) / 2
    
    # Adjust weights by multiplying them with the Logarithmic Volume Difference
    df['adjusted_weights'] = df['combined_returns'] * df['log_volume_diff'] * df['intraday_volume_activity']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['raw_momentum'] = df['close'] - df['close'].shift(n)
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    df['combined_momentum_range'] = df['volume_weighted_momentum'] + df['range']
    
    # Multiply Combined Momentum by Volume Ratio
    df['weighted_momentum'] = df['combined_momentum_range'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_factor'] = df['weighted_momentum'].ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    df['final_factor'] = df['volume_weighted_return'] * df['adjusted_weights']
    
    # Incorporate Intraday Range Breakouts
    df['intraday_breakout'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    df['breakout_weighted'] = df['intraday_breakout'] * df['volume_ratio']
    
    # Final factor
    df['alpha_factor'] = df['ema_factor'] + df['final_factor'] + df['breakout_weighted']
    
    return df['alpha_factor']
```
