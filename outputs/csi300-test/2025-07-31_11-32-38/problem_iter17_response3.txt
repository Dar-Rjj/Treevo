```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()

    # Calculate Rolling Average of Daily Price Changes (10-day MA)
    df['price_change_ma_10'] = df['price_change'].rolling(window=10).mean()

    # Calculate Volume and Amount Weighted Daily Return
    df['volume_amount_weighted_return'] = (df['close'] / df['open']) * df['volume'] * df['amount']

    # Summarize Volume and Amount Weighted Daily Returns (20-day MA)
    df['volume_amount_weighted_return_ma_20'] = df['volume_amount_weighted_return'].rolling(window=20).mean()

    # Combine Signals
    df['signal_difference'] = df['price_change_ma_10'] - df['volume_amount_weighted_return_ma_20']
    df['combined_signal'] = df['signal_difference'].rolling(window=5).mean()

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']

    # Calculate Close-to-Open Return
    df['close_open_return'] = (df['close'] - df['open']) / df['open']

    # Integrate Volume Change
    df['volume_change'] = df['volume'].pct_change()
    df['intraday_preliminary_factor'] = (df['intraday_return'] + df['close_open_return']) * df['volume_change']

    # Calculate True Range (TR)
    df['high_low'] = abs(df['high'] - df['low'])
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)

    # Calculate Average True Range (ATR) (14-day ATR)
    df['atr'] = df['true_range'].rolling(window=14).mean()

    # Calculate Positive Directional Movement (+DM)
    df['pos_dm'] = np.where((df['high'] > df['high'].shift(1)) & (df['low'] >= df['low'].shift(1)), 
                            df['high'] - df['high'].shift(1), 0)

    # Calculate Negative Directional Movement (-DM)
    df['neg_dm'] = np.where((df['low'] < df['low'].shift(1)) & (df['high'] <= df['high'].shift(1)), 
                            df['low'].shift(1) - df['low'], 0)

    # Smooth +DM and -DM over 14 periods
    df['pos_dm_smoothed'] = df['pos_dm'].rolling(window=14).sum() / 14
    df['neg_dm_smoothed'] = df['neg_dm'].rolling(window=14).sum() / 14

    # Calculate +DI and -DI
    df['pos_di'] = (df['pos_dm_smoothed'] / df['atr']) * 100
    df['neg_di'] = (df['neg_dm_smoothed'] / df['atr']) * 100

    # Calculate ADX
    df['adx'] = (abs(df['pos_di'] - df['neg_di']) / (df['pos_di'] + df['neg_di'])) * 100

    # Calculate Intraday Momentum (10-day MA)
    df['intraday_momentum'] = df['intraday_return'].rolling(window=10).mean()

    # Synthesize Final Alpha Factor
    df['alpha_factor'] = (df['combined_signal'] * df['adx'] + 
                          df['intraday_preliminary_factor'] * df['adx'] + 
                          df['intraday_momentum'] * df['adx'])

    return df['alpha_factor']
```
