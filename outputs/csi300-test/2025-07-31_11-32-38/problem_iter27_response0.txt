import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate 10-Day and 20-Day Moving Averages of Daily Price Changes
    df['10_day_ma'] = df['daily_price_change'].rolling(window=10).mean()
    df['20_day_ma'] = df['daily_price_change'].rolling(window=20).mean()
    
    # Combine Moving Averages
    df['ma_diff'] = df['10_day_ma'] - df['20_day_ma']
    df['combined_ma'] = df['ma_diff'].rolling(window=5).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns
    df['20_day_vol_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Integrate Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['preliminary_factor'] = df['20_day_vol_weighted_return'] * df['daily_volume_change']
    
    # Calculate Intraday and Close-to-Open Returns
    df['intraday_return'] = df['high'] - df['low']
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    
    # Adjust Weights by Logarithmic Volume Difference
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t_1'] = np.log(df['volume'].shift(1))
    df['log_volume_diff'] = df['log_volume_t'] - df['log_volume_t_1']
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(span=10, adjust=False).mean()
    
    df['adjusted_weighted_returns'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['smoothed_log_volume_diff']
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['close'].shift(1))
    df['tr3'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    n = 14
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Calculate Positive and Negative Directional Movement (+DM, -DM)
    df['+dm'] = df['high'] - df['high'].shift(1)
    df['-dm'] = df['low'].shift(1) - df['low']
    
    df['+dm'] = df['+dm'].apply(lambda x: max(x, 0))
    df['-dm'] = df['-dm'].apply(lambda x: max(x, 0))
    
    # Calculate +DI and -DI
    df['+di'] = 100 * (df['+dm'].rolling(window=n).sum() / df['atr'])
    df['-di'] = 100 * (df['-dm'].rolling(window=n).sum() / df['atr'])
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Calculate 10-Day Volatility
    df['10_day_volatility'] = df['daily_price_change'].rolling(window=10).std()
    
    # Combine Volatility and ADMI
    df['vol_admi'] = df['10_day_volatility'] * df['admi']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['combined_ma'] + df['preliminary_factor'] + 
                                df['adjusted_weighted_returns'] + df['vol_admi']) * df['admi']
    
    return df
