```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the Daily Price Changes
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Sum of Daily Price Changes over a 20-day lookback window
    df['sum_daily_price_change_20d'] = df['daily_price_change'].rolling(window=20).sum()
    
    # Average of the last 5 Daily Price Changes
    df['avg_daily_price_change_5d'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Average daily range (High - Low) over a 20-day period
    df['avg_daily_range_20d'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Standard deviation of the daily range (High - Low) over a 20-day period
    df['std_daily_range_20d'] = (df['high'] - df['low']).rolling(window=20).std()
    
    # Net number of days where Close > Open versus Close < Open over a 30-day period
    df['net_days_close_above_open_30d'] = ((df['close'] > df['open']) - (df['close'] < df['open'])).rolling(window=30).sum()
    
    # Ratio of days where Close > Open to days where Close < Open over a 30-day period
    df['ratio_close_above_to_below_open_30d'] = (
        (df['close'] > df['open']).rolling(window=30).sum() / 
        (df['close'] < df['open']).rolling(window=30).sum()
    ).fillna(0)
    
    # Calculate the ratio of total volume on up days to down days over a 50-day period
    df['volume_up'] = df['volume'] * (df['close'] > df['close'].shift(1))
    df['volume_down'] = df['volume'] * (df['close'] < df['close'].shift(1))
    df['ratio_volume_up_to_down_50d'] = (
        df['volume_up'].rolling(window=50).sum() / 
        df['volume_down'].rolling(window=50).sum()
    ).fillna(0)
    
    # Compute the average volume on up days and down days separately over a 50-day period
    df['avg_volume_up_50d'] = df['volume_up'].rolling(window=50).mean()
    df['avg_volume_down_50d'] = df['volume_down'].rolling(window=50).mean()
    
    # Cumulative amount change over a 60-day lookback period
    df['cumulative_amount_change_60d'] = df['amount'].rolling(window=60).sum() - df['amount'].rolling(window=60).sum().shift(60)
    
    # Average daily amount over a 60-day lookback period
    df['avg_daily_amount_60d'] = df['amount'].rolling(window=60).mean()
    
    # Combine all alpha factors into a single factor
    df['alpha_factor'] = (
        df['sum_daily_price_change_20d'] + 
        df['avg_daily_price_change_5d'] + 
        df['avg_daily_range_20d'] + 
        df['std_daily_range_20d'] + 
        df['net_days_close_above_open_30d'] + 
        df['ratio_close_above_to_below_open_30d'] + 
        df['ratio_volume_up_to_down_50d'] + 
        df['avg_volume_up_50d'] - 
        df['avg_volume_down_50d'] + 
        df['cumulative_amount_change_60d'] + 
        df['avg_daily_amount_60d']
    )
    
    return df['alpha_factor']
```
