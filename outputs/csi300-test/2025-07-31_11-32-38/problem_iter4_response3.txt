```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum Indicators
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    delta_close = df['close'].diff()
    gain = (delta_close.where(delta_close > 0, 0)).fillna(0)
    loss = (-delta_close.where(delta_close < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Volatility Measures
    df['TR'] = np.maximum(
        np.maximum(df['high'] - df['low'], np.abs(df['high'] - df['close'].shift(1))),
        np.abs(df['low'] - df['close'].shift(1))
    )
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    df['Close_to_Close_Change'] = df['close'].pct_change()
    df['Volume_Weighted_Change'] = (df['Close_to_Close_Change'].abs() * df['volume']).rolling(window=14).mean()
    
    # Composite Alpha Factor
    momentum_factor = (
        (df['SMA_5'] > df['SMA_20']).astype(int) +
        (df['EMA_10'] > df['EMA_30']).astype(int) +
        (df['RSI_14'] > 50).astype(int)
    )
    
    volatility_factor = (
        (df['ATR_14'] - df['ATR_14'].rolling(window=14).min()) / 
        (df['ATR_14'].rolling(window=14).max() - df['ATR_14'].rolling(window=14).min()) +
        (df['Volume_Weighted_Change'] - df['Volume_Weighted_Change'].rolling(window=14).min()) / 
        (df['Volume_Weighted_Change'].rolling(window=14).max() - df['Volume_Weighted_Change'].rolling(window=14).min())
    )
    
    composite_alpha_factor = (momentum_factor * 2 + volatility_factor) / 3
    return composite_alpha_factor.dropna()
```
