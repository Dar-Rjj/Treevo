```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t_minus_1'] = df['log_volume_t'].shift(1)
    df['log_volume_diff'] = df['log_volume_t'] - df['log_volume_t_minus_1']
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(alpha=0.2).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['weighted_intraday_high_low_spread'] = df['intraday_high_low_spread'] * df['volume_ratio']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * np.abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * np.abs(df['close_to_open_return'])
    
    # Integrate Volume Change and Intraday Volatility
    df['adjusted_weighted_intraday_return'] = df['weighted_intraday_return'] * df['smoothed_log_volume_diff'] * df['weighted_intraday_high_low_spread']
    df['adjusted_weighted_close_to_open_return'] = df['weighted_close_to_open_return'] * df['smoothed_log_volume_diff'] * df['weighted_intraday_high_low_spread']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_adjusted_weighted_intraday_return'] = df['adjusted_weighted_intraday_return'].ewm(span=5, adjust=False).mean()
    df['ema_adjusted_weighted_close_to_open_return'] = df['adjusted_weighted_close_to_open_return'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['final_factor'] = df['daily_returns'] * (df['ema_adjusted_weighted_intraday_return'] + df['ema_adjusted_weighted_close_to_open_return'])
    
    # Calculate Rolling Average of Daily Price Changes
    df['rolling_daily_price_change'] = df['daily_returns'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['volume_amount_weighted_return'] = (df['close'] / df['open'] - 1) * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns
    df['volume_amount_weighted_return_ma'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Integrate Price Volatility
    df['daily_high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['daily_high_low_range'].rolling(window=10).mean()
    
    # Combine Signals
    df['signal_difference'] = df['rolling_daily_price_change'] - df['volume_amount_weighted_return_ma']
    df['adjusted_signal'] = df['signal_difference'] * df['smoothed_log_volume_diff'] / df['avg_high_low_range']
    
    # Smooth the Result
    df['final_alpha_factor'] = df['adjusted_signal'].rolling(window=5).mean()
    
    return df['final_alpha_factor']
```
