```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of money flow
    money_flow_10d = money_flow.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of volume
    volume_10d = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of close price
    close_10d = df['close'].rolling(window=10).mean()
    
    # Calculate the normalized money flow
    normalized_money_flow = (money_flow - money_flow_10d) / money_flow_10d
    
    # Calculate the normalized volume
    normalized_volume = (df['volume'] - volume_10d) / volume_10d
    
    # Calculate the normalized close price
    normalized_close = (df['close'] - close_10d) / close_10d
    
    # Combine the normalized factors to create the final alpha factor
    alpha_factor = normalized_money_flow * normalized_volume * normalized_close
    
    return alpha_factor
```
