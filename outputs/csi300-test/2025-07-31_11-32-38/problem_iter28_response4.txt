```python
import pandas as pd

def heuristics_v2(df):
    # Sum the Daily Price Changes (Close t - Close t-1) over a 10-day lookback window
    df['price_change'] = df['close'].diff()
    df['momentum_10_day'] = df['price_change'].rolling(window=10).sum()

    # Compute the average daily range (High - Low) over a 20-day period
    df['daily_range'] = df['high'] - df['low']
    df['avg_range_20_day'] = df['daily_range'].rolling(window=20).mean()

    # Track the net number of days where Close > Open versus Close < Open over a 30-day period
    df['close_open_diff'] = df['close'] - df['open']
    df['up_days'] = (df['close_open_diff'] > 0).astype(int)
    df['down_days'] = (df['close_open_diff'] < 0).astype(int)
    df['net_sentiment_30_day'] = df['up_days'].rolling(window=30).sum() - df['down_days'].rolling(window=30).sum()

    # Calculate the ratio of total volume on up days to down days over a 50-day period
    df['volume_up'] = df['volume'] * (df['close_open_diff'] > 0).astype(int)
    df['volume_down'] = df['volume'] * (df['close_open_diff'] < 0).astype(int)
    df['volume_up_total'] = df['volume_up'].rolling(window=50).sum()
    df['volume_down_total'] = df['volume_down'].rolling(window=50).sum()
    df['volume_ratio_50_day'] = df['volume_up_total'] / df['volume_down_total'].replace(0, 1)

    # Calculate the cumulative amount change over a 60-day lookback period
    df['amount_change'] = df['amount'].diff().fillna(0)
    df['cumulative_amount_60_day'] = df['amount_change'].rolling(window=60).sum()

    # Combine all the alpha factors into a single factor
    df['alpha_factor'] = (
        df['momentum_10_day'] 
        + df['avg_range_20_day'] 
        + df['net_sentiment_30_day'] 
        + df['volume_ratio_50_day'] 
        + df['cumulative_amount_60_day']
    )

    return df['alpha_factor']
```
