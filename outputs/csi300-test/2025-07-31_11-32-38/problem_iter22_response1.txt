```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # Sum of Daily Price Changes over a 10-day lookback window (short-term momentum)
    df['Momentum_10d'] = df['Daily_Price_Change'].rolling(window=10).sum()

    # Calculate daily range (High - Low) to measure volatility
    df['Daily_Range'] = df['high'] - df['low']

    # Compute the average daily range over a 20-day period for a smoother volatility metric
    df['Average_Daily_Range_20d'] = df['Daily_Range'].rolling(window=20).mean()

    # Determine if the close is higher or lower than the open (Directional Move)
    df['Directional_Move'] = (df['close'] > df['open']).astype(int)

    # Track the net number of upward and downward moves over a 30-day period as a sentiment indicator
    df['Net_Directional_Moves_30d'] = df['Directional_Move'].rolling(window=30).sum() - (30 - df['Directional_Move'].rolling(window=30).sum())

    # Evaluate the ratio of volume on up days to down days (Volume Ratio)
    df['Volume_Up_Days'] = df['volume'] * (df['close'] > df['open']).astype(int)
    df['Volume_Down_Days'] = df['volume'] * (df['close'] < df['open']).astype(int)
    df['Volume_Ratio_30d'] = df['Volume_Up_Days'].rolling(window=30).sum() / df['Volume_Down_Days'].rolling(window=30).sum()

    # Calculate the total volume over a 50-day time frame to gauge interest and liquidity
    df['Total_Volume_50d'] = df['volume'].rolling(window=50).sum()

    # Measure the amount change from one day to the next (Amount Change)
    df['Amount_Change'] = df['amount'].diff()

    # Calculate the cumulative amount change over a 60-day lookback period to assess capital flow trends
    df['Cumulative_Amount_Change_60d'] = df['Amount_Change'].rolling(window=60).sum()

    # Combine factors into a single alpha factor
    df['Alpha_Factor'] = (
        df['Momentum_10d'] + 
        df['Average_Daily_Range_20d'] + 
        df['Net_Directional_Moves_30d'] + 
        df['Volume_Ratio_30d'] + 
        df['Total_Volume_50d'] + 
        df['Cumulative_Amount_Change_60d']
    )

    return df['Alpha_Factor']
```
