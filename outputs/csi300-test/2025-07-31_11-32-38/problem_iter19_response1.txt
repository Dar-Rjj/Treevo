```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Daily Returns
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1) * df['volume']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine Intraday and Close-to-Open Returns
    intraday_weight = np.abs(df['intraday_return'])
    close_to_open_weight = np.abs(df['close_to_open_return'])
    combined_returns = (intraday_weight * df['intraday_return'] + close_to_open_weight * df['close_to_open_return']) / (intraday_weight + close_to_open_weight)
    
    # Adjust weights by multiplying them with the Logarithmic Volume Difference and Intraday High-Low Spread Weight
    adjusted_weights = combined_returns * df['log_volume_diff'] * df['intraday_high_low_spread']
    
    # Combine Volume-Weighted Momentum with Price Range
    n = 5
    raw_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    volume_weighted_momentum = raw_momentum / cumulative_volume
    price_range = df['high'] - df['low']
    combined_momentum = volume_weighted_momentum + price_range
    
    # Multiply Combined Momentum by Volume Ratio
    combined_momentum *= df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    smoothed_momentum = combined_momentum.ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    final_factor = df['volume_weighted_returns'] * adjusted_weights * smoothed_momentum
    
    # Incorporate Intraday Range Breakouts
    df['intraday_range_breakout'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    intraday_breakout_weight = df['intraday_range_breakout'] * df['volume_ratio']
    
    # Final factor
    final_factor += intraday_breakout_weight * df['intraday_high_low_spread']
    
    return final_factor
```
