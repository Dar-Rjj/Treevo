```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['daily_return'] = df['close'].pct_change()

    # Calculate Volume-Weighted Average Price (VWAP) over a 10-day period
    df['price_times_volume'] = df['close'] * df['volume']
    df['cumulative_price_times_volume'] = df['price_times_volume'].rolling(window=10).sum()
    df['cumulative_volume'] = df['volume'].rolling(window=10).sum()
    df['vwap'] = df['cumulative_price_times_volume'] / df['cumulative_volume']

    # Calculate volume-to-price change ratio
    df['price_change'] = df['close'].diff()
    df['volume_to_price_change_ratio'] = df['volume'] / df['price_change']

    # Calculate high-low spread
    df['high_low_spread'] = df['high'] - df['low']

    # Calculate trend reversal indicator
    df['trend_reversal'] = (df['high'] < df['low'].shift(1)).astype(int)

    # Calculate intraday price range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate close to intraday range ratio
    df['close_to_range_ratio'] = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Combine all the sub-factors into a single alpha factor
    df['alpha_factor'] = (
        df['daily_return'] +
        (df['close'] - df['vwap']) / df['close'] +
        df['volume_to_price_change_ratio'] +
        df['high_low_spread'] +
        df['trend_reversal'] +
        df['close_to_range_ratio']
    )

    return df['alpha_factor']
```
