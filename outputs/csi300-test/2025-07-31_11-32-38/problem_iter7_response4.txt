```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_amount_weighted_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Compute 20-Day Moving Average of Volume and Amount Weighted Returns
    df['20_day_ma_weighted_return'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_difference'] = df['10_day_ma_price_change'] - df['20_day_ma_weighted_return']
    
    # Smooth the Result with 5-Day Moving Average
    df['smoothed_signal'] = df['signal_difference'].rolling(window=5).mean()
    
    return df['smoothed_signal'].dropna()
```
