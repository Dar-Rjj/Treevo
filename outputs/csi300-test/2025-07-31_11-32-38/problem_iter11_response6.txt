```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume-Weighted Daily Returns
    df['daily_return'] = df['close'] / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Compute 20-Day Moving Average of Volume-Weighted Returns
    df['20_day_ma_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Calculate Volume-Weighted Momentum
    n = 20
    df['raw_momentum'] = df['close'] / df['close'].shift(n)
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    
    # Incorporate Price Range Factor
    df['price_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['price_range']
    
    # Weigh by Intraday Volume Activity
    df['20_day_ma_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['20_day_ma_volume']
    df['weighted_combined_momentum'] = df['combined_momentum'] * df['volume_ratio']
    
    # Combine Signals
    df['difference_10_20_ma'] = df['10_day_ma_price_change'] - df['20_day_ma_volume_weighted_return']
    df['smoothed_difference'] = df['difference_10_20_ma'].rolling(window=5).mean()
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'].shift(1) - 1
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t_minus_1'] = np.log(df['volume'].shift(1))
    df['log_volume_difference'] = df['log_volume_t'] - df['log_volume_t_minus_1']
    df['smoothed_log_volume_difference'] = df['log_volume_difference'].ewm(span=5, adjust=False).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Integrate Volume Change and Intraday Volatility
    df['adjusted_weights'] = df['smoothed_log_volume_difference'] * df['intraday_high_low_spread']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    
    # Weigh Combined Momentum by Intraday Volume Activity
    df['final_weighted_momentum'] = df['weighted_combined_momentum'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_final_factor'] = df['final_weighted_momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['final_factor'] = df['volume_weighted_return'] * df['adjusted_weights']
    
    return df['final_factor']
```
