import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A pandas Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the percentage difference between the 10-day and 30-day moving averages
    df['ma_diff_percent'] = (df['ma_10'] - df['ma_30']) / df['ma_30']
    
    # Calculate the 10-day standard deviation of the daily returns
    df['std_10'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the daily returns
    df['std_30'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the ratio of the 10-day to 30-day standard deviations
    df['std_ratio'] = df['std_10'] / df['std_30']
    
    # Combine the moving average difference and standard deviation ratio
    df['factor'] = df['ma_diff_percent'] * df['std_ratio']
    
    return df['factor']
