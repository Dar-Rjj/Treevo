```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=20, p=5, q=5, r=5):
    # Calculate Price Momentum Factor
    df['momentum'] = df['close'].diff(n)
    
    # Incorporate Trading Volume into the Analysis
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(m).mean()
    
    # Explore the Relationship Between Open and Close Prices
    df['open_close_diff'] = df['open'] - df['close']
    df['open_close_sum'] = df['open_close_diff'].rolling(p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    df['range'] = df['high'] - df['low']
    df['range_ma'] = df['range'].rolling(q).mean()
    
    # Calculate Volume-Weighted Daily Returns
    df['volume_weighted_return'] = (df['close'] - df['close'].shift(1)) * df['volume']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume'] / df['volume'].shift(1))
    
    # Weigh by Intraday Volume Activity
    df['volume_activity'] = df['volume'] / df['volume'].rolling(20).mean()
    
    # Integrate Components
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    df['adjusted_weights'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['log_volume_diff']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['raw_momentum'] = df['close'] - df['close'].shift(n)
    df['cumulative_volume'] = df['volume'].rolling(n).sum().shift(1)
    df['combined_momentum'] = df['raw_momentum'] * (df['cumulative_volume'] / df['volume'].rolling(n).sum()) + df['range']
    
    # Multiply Combined Momentum by Volume Ratio
    df['final_momentum'] = df['combined_momentum'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_final_momentum'] = df['final_momentum'].ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    df['final_volume_weighted_return'] = df['volume_weighted_return'] * df['adjusted_weights']
    
    # Incorporate Intraday Range Breakouts
    df['intraday_range_breakout'] = (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1))
    df['weighted_breakout'] = df['intraday_range_breakout'] * df['volume_activity']
    
    # Analyze Intraday Volatility
    df['intraday_volatility'] = df['intraday_return'].rolling(r).std()
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['log_volume_diff']
    
    # Enhance Volume-Weighted Indicators
    df['volume_weighted_price_change'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1))
    df['combined_intraday_range_volatility'] = df['range'] * df['intraday_volatility']
    df['weighted_combined_intraday_range_volatility'] = df['combined_intraday_range_volatility'] * df['volume_activity']
    
    # Apply 10-Day Simple Moving Average to the Volume-Weighted Price Change
    df['sma_volume_weighted_price_change'] = df['volume_weighted_price_change'].rolling(10).mean()
    
    # Multiply the smoothed Volume-Weighted Price Change by the combined Intraday Range and Volatility
    df['final_factor'] = df['sma_volume_weighted_price_change'] * df['weighted_combined_intraday_range_volatility']
    
    return df['final_factor']
```
