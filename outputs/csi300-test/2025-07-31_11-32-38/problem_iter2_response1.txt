```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Returns
    df['daily_return'] = df['close'].pct_change()

    # Momentum Component
    df['10_day_moving_sum'] = df['daily_return'].rolling(window=10).sum()
    df['5_day_moving_sum'] = df['daily_return'].rolling(window=5).sum()
    df['momentum_component'] = df['10_day_moving_sum'] - df['5_day_moving_sum']

    # Reversal Component
    df['21_day_moving_avg'] = df['daily_return'].rolling(window=21).mean()
    df['reversal_component'] = -df['21_day_moving_avg']

    # Price Momentum with Volume Weight
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['10_day_price_change_ma'] = df['price_change'].rolling(window=10).mean()
    
    df['volume_weighted_return'] = (df['close'] / df['open']) * df['volume']
    df['20_day_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()

    # Adjusted Momentum and Reversal
    df['momentum_component_std'] = df['momentum_component'].rolling(window=21).std()
    df['normalized_momentum'] = df['momentum_component'] / df['momentum_component_std']

    df['reversal_component_std'] = df['reversal_component'].rolling(window=21).std()
    df['normalized_reversal'] = df['reversal_component'] / df['reversal_component_std']

    df['combined_momentum_reversal'] = df['normalized_momentum'] + df['normalized_reversal']

    # Final Alpha Factor
    df['signal_difference'] = df['10_day_price_change_ma'] - df['20_day_volume_weighted_return']
    df['smoothed_signal'] = df['signal_difference'].rolling(window=5).mean()
    df['final_alpha_factor'] = df['smoothed_signal'] * df['combined_momentum_reversal']

    return df['final_alpha_factor']
```
