```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Volume-Weighted Daily Returns
    df['Volume-Weighted Daily Return'] = (df['close'].pct_change() * df['volume']).fillna(0)
    
    # Compute Intraday Return
    df['Intraday Return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['Close-to-Open Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['Log Volume'] = np.log(df['volume'])
    df['Log Volume Diff'] = df['Log Volume'].diff().fillna(0)
    df['Smoothed Log Volume Diff'] = df['Log Volume Diff'].ewm(alpha=0.2).mean()
    
    # Calculate Intraday High-Low Spread
    df['High-Low Spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['Volume Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Weighted High-Low Spread'] = df['High-Low Spread'] * df['Volume Ratio']
    
    # Combine Intraday and Close-to-Open Returns
    df['Combined Return'] = (df['Intraday Return'] * abs(df['Intraday Return'])) + (df['Close-to-Open Return'] * abs(df['Close-to-Open Return']))
    
    # Integrate Volume Change and Intraday Volatility
    df['Adjusted Combined Return'] = df['Combined Return'] * df['Smoothed Log Volume Diff'] * df['Weighted High-Low Spread']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['Raw Momentum'] = df['close'].pct_change(periods=n).fillna(0)
    df['Cumulative Volume'] = df['volume'].rolling(window=n).sum()
    df['Volume-Weighted Momentum'] = df['Raw Momentum'] / df['Cumulative Volume']
    df['Price Range'] = df['high'] - df['low']
    df['Combined Momentum'] = df['Volume-Weighted Momentum'] + df['Price Range']
    
    # Weigh Combined Momentum by Intraday Volume Activity
    df['Weighted Combined Momentum'] = df['Combined Momentum'] * df['Volume Ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['EMA Weighted Combined Momentum'] = df['Weighted Combined Momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['Final Factor'] = df['Volume-Weighted Daily Return'] * df['Adjusted Combined Return']
    df['Final Factor'] = df['Final Factor'].rolling(window=n).sum()
    
    # Incorporate Intraday Range Breakouts
    df['Intraday Range Breakout'] = (df['high'] > df['high'].shift()) & (df['low'] < df['low'].shift())
    df['Breakout Weighted'] = df['Intraday Range Breakout'] * df['Volume Ratio']
    df['Final Factor'] += df['Breakout Weighted']
    
    # Calculate Intraday Trading Imbalance
    df['Intraday Trading Imbalance'] = (df['high'] - df['low']) / (df['close'] - df['open'])
    
    # Integrate Intraday Trading Imbalance with Final Factor
    df['Trading Imbalance Factor'] = df['Intraday Trading Imbalance'] * df['volume']
    df['Final Factor'] += df['Trading Imbalance Factor']
    
    return df['Final Factor']
```
