```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume Adjusted High-Low Spread
    volume_adjusted_high_low_spread = high_low_spread * df['volume']
    
    # Calculate Momentum of Volume Adjusted High-Low Spread
    momentum_5_days = volume_adjusted_high_low_spread.rolling(window=5).mean()
    
    # Subtract Previous Day's Average
    previous_day_average = momentum_5_days.shift(1)
    alpha_factor = momentum_5_days - previous_day_average
    
    return alpha_factor
```
