import pandas as pd
import numpy as np

def heuristics_v2(df, n=14):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Weight by Logarithmic Volume Difference
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t-1'] = df['log_volume_t'].shift(1)
    df['log_vol_diff'] = df['log_volume_t'] - df['log_volume_t-1']
    df['smoothed_log_vol_diff'] = df['log_vol_diff'].ewm(span=n, adjust=False).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    df['combined_return'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) / 2
    
    # Integrate Volume Change
    df['preliminary_factor'] = df['combined_return'] * df['smoothed_log_vol_diff']
    
    # Integrate Price Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=n).mean()
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=n).mean()
    
    # Calculate Positive and Negative Directional Movement (+DM, -DM)
    df['+dm'] = 0.0
    df['-dm'] = 0.0
    mask_plus = (df['high'] > df['high'].shift(1)) & (df['high'] - df['high'].shift(1) > df['low'].shift(1) - df['low'])
    mask_minus = (df['low'].shift(1) > df['low']) & (df['low'].shift(1) - df['low'] > df['high'] - df['high'].shift(1))
    df.loc[mask_plus, '+dm'] = df['high'] - df['high'].shift(1)
    df.loc[mask_minus, '-dm'] = df['low'].shift(1) - df['low']
    
    # Smooth +DM and -DM over n periods
    df['+dm_smoothed'] = df['+dm'].rolling(window=n).sum() / n
    df['-dm_smoothed'] = df['-dm'].rolling(window=n).sum() / n
    
    # Calculate +DI and -DI
    df['+di'] = 100 * (df['+dm_smoothed'] / df['atr'])
    df['-di'] = 100 * (df['-dm_smoothed'] / df['atr'])
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Final Factor Calculation
    df['final_factor'] = df['preliminary_factor'] * df['admi'] * (df['high_low_range'] / df['avg_high_low_range'])
    
    return df['final_factor']
