```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Weight by Logarithmic Volume Difference
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_diff'] = df['log_volume'].diff().fillna(0)
    df['smoothed_volume_diff'] = df['log_volume_diff'].ewm(alpha=0.2).mean()
    
    # Weigh by Intraday Volume Activity
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_20_day_ma']
    
    # Combine Intraday High-Low Spread and Volume Activity
    df['weighted_intraday_spread'] = df['intraday_high_low_spread'] * df['volume_ratio']
    
    # Final Factor Calculation
    df['weighted_daily_returns'] = df['daily_returns'] * df['smoothed_volume_diff']
    df['accumulated_returns'] = df['weighted_daily_returns'].rolling(window=5).sum()
    df['final_factor'] = df['accumulated_returns'] + df['weighted_intraday_spread']
    df['smoothed_final_factor'] = df['final_factor'].ewm(span=5, adjust=False).mean()
    
    return df['smoothed_final_factor']
```
