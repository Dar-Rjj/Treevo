import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns 'open', 'high', 'low', 'close', 'amount', 'volume' and index 'date'.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the intraday range
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted close price
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10).mean()
    
    # Calculate the momentum factor
    df['momentum_10'] = df['close'].pct_change(10)
    
    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Combine factors into a single alpha factor
    df['alpha_factor'] = (
        0.3 * df['daily_return'].fillna(0) +
        0.2 * df['intraday_range'].fillna(0) +
        0.2 * (df['close'] - df['vwap']).fillna(0) / df['close'] +
        0.2 * df['momentum_10'].fillna(0) +
        0.1 * (df['rsi'] - 50).fillna(0)
    )
    
    return df['alpha_factor']
