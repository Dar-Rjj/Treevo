```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 5-day and 20-day Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate 10-day and 30-day Exponential Moving Averages
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Calculate True Range
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    
    # Calculate 14-day Average True Range (ATR)
    df['ATR_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate Volume Weighted Price Change
    df['price_change_abs'] = df['close'].diff().abs()
    df['volume_weighted_price_change'] = (df['price_change_abs'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Integrate Momentum Indicators into a Single Score
    df['momentum_score'] = (df['SMA_5'] > df['SMA_20']) + (df['EMA_10'] > df['EMA_30']) + (df['RSI_14'] > 50)
    
    # Incorporate Volatility Measures
    df['volatility_adjustment'] = df['ATR_14'] + df['volume_weighted_price_change']
    
    # Final Composite Alpha Factor
    df['composite_alpha_factor'] = df['momentum_score'] / df['volatility_adjustment']
    
    return df['composite_alpha_factor'].dropna()
```
