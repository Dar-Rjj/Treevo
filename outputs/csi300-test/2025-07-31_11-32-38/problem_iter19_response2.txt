```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Calculate 10-Day Moving Average of Daily Price Changes
    df['price_change_ma_10'] = df['price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['weighted_daily_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns
    df['weighted_return_ma_20'] = df['weighted_daily_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_diff'] = df['price_change_ma_10'] - df['weighted_return_ma_20']
    df['signal_smoothed'] = df['signal_diff'].rolling(window=5).mean()
    
    # Calculate Intraday and Close-to-Open Returns
    df['intraday_return'] = df['high'] - df['low']
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Weight by Logarithmic Volume Difference
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t_minus_1'] = df['log_volume_t'].shift(1)
    df['log_volume_diff'] = df['log_volume_t'] - df['log_volume_t_minus_1']
    df['log_volume_diff_smoothed'] = df['log_volume_diff'].ewm(span=5, adjust=False).mean()
    
    # Combine Intraday and Close-to-Open Returns
    intraday_weight = df['intraday_return'].abs() * df['log_volume_diff_smoothed']
    close_to_open_weight = df['close_to_open_return'].abs() * df['log_volume_diff_smoothed']
    df['combined_intraday_close_to_open'] = (intraday_weight + close_to_open_weight) / 2
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['close'].shift(1) - df['high']).abs()
    df['tr3'] = (df['close'].shift(1) - df['low']).abs()
    df['tr'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    n = 14
    df['atr'] = df['tr'].rolling(window=n).sum() / n
    
    # Calculate Positive and Negative Directional Movement (+DM, -DM)
    df['+dm'] = np.where((df['high'] > df['high'].shift(1)) & 
                         (df['high'] - df['high'].shift(1) > df['low'].shift(1) - df['low']), 
                         df['high'] - df['high'].shift(1), 0)
    df['-dm'] = np.where((df['low'].shift(1) > df['low']) & 
                         (df['low'].shift(1) - df['low'] > df['high'] - df['high'].shift(1)), 
                         df['low'].shift(1) - df['low'], 0)
    
    # Calculate +DI and -DI
    df['+di'] = 100 * (df['+dm'].rolling(window=n).sum() / df['atr'])
    df['-di'] = 100 * (df['-dm'].rolling(window=n).sum() / df['atr'])
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['preliminary_alpha_factor'] = df['signal_smoothed'] * df['admi']
    df['intraday_close_to_open_factor'] = df['combined_intraday_close_to_open'] * df['admi']
    df['final_alpha_factor'] = df['preliminary_alpha_factor'] + df['intraday_close_to_open_factor']
    
    return df['final_alpha_factor']
```
