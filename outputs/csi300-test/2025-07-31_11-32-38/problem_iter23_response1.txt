```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Daily Return
    df['prev_close'] = df['close'].shift(1)
    df['volume_weighted_daily_return'] = (df['close'] - df['prev_close']) * df['volume']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Track Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Determine the Logarithmic Volume Difference
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_diff'] = df['log_volume'] - df['log_volume'].shift(1)
    df['log_volume_diff_smoothed'] = df['log_volume_diff'].ewm(span=5, adjust=False).mean()
    
    # Calculate Weigh by Intraday Volume Activity
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['volume_activity'] = df['volume'] / df['volume_ma_20']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'].abs() * df['volume_activity']
    df['weighted_close_to_open_return'] = df['close_to_open_return'].abs() * df['volume_activity']
    
    # Adjust weights by integrating volume and intraday volatility
    df['adjusted_weights'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['log_volume_diff_smoothed']
    
    # Calculate Raw Momentum
    n = 5
    df['raw_momentum'] = df['close'] - df['close'].shift(n)
    
    # Weigh by Cumulative Volume
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    
    # Add Price Range to Volume-Weighted Momentum
    df['price_range'] = df['high'] - df['low']
    df['volume_weighted_momentum_with_range'] = df['volume_weighted_momentum'] + df['price_range']
    
    # Apply 5-Day Exponential Moving Average
    df['factor'] = df['volume_weighted_daily_return'] * df['adjusted_weights']
    df['factor'] = df['factor'].ewm(span=5, adjust=False).mean()
    
    # Incorporate Intraday Range Breakouts
    df['intraday_breakout'] = (df['high'] > df['high'].shift(1)) | (df['low'] < df['low'].shift(1))
    df['intraday_breakout_weighted'] = df['intraday_breakout'] * (df['volume'] / df['volume_ma_20'])
    
    # Final Factor Calculation
    df['final_factor'] = df['factor'] + df['intraday_breakout_weighted']
    
    return df['final_factor']
```
