```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Range
    high_low_range = (df['high'] - df['low']) / df['low']
    
    # Integrate Intraday Return and High-Low Range
    combined_metric = 0.6 * intraday_return + 0.4 * high_low_range
    
    # Apply Exponential Moving Average (EMA) to the Combined Metric with span 12
    ema_combined = combined_metric.ewm(span=12, adjust=False).mean()
    
    # Adjust for Volume
    # Calculate Rolling Standard Deviation of Volume with window 20
    volume_std = df['volume'].rolling(window=20).std()
    
    # Invert the Rolling Standard Deviation
    inverted_volume_std = 1 / volume_std
    
    # Scale the EMA by the Inverse of the Rolling Standard Deviation
    adjusted_ema = ema_combined * inverted_volume_std
    
    return adjusted_ema
```
