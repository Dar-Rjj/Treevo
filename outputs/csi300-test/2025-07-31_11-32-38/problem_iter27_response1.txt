import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate the 20-day moving average of the closing price
    df['ma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 50-day moving average of the closing price
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the price-to-moving-average ratio
    df['price_to_ma_20_ratio'] = df['close'] / df['ma_20']
    
    # Calculate the price-to-volume ratio
    df['price_to_volume_ratio'] = df['close'] / df['volume']
    
    # Calculate the 20-day volatility
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['price_to_ma_20_ratio'] * df['price_to_volume_ratio']) / df['volatility_20']
    
    return df['alpha_factor']
