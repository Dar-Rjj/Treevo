import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       indexed by date.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the relative range (high - low) / close
    df['relative_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=10, min_periods=1).mean()
    
    # Calculate the relative distance from VWAP
    df['rel_dist_vwap'] = (df['close'] - df['vwap']) / df['vwap']
    
    # Calculate the standard deviation of the daily returns over a 20-day window
    df['std_daily_return'] = df['daily_return'].rolling(window=20, min_periods=1).std()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (
        0.4 * df['relative_range'] +
        0.3 * df['rel_dist_vwap'] +
        0.3 * df['std_daily_return']
    )
    
    return df['alpha_factor']
