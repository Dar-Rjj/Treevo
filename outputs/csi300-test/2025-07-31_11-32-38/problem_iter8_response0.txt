import numpy as np

    df['Daily_Price_Change'] = df['close'].diff()

    df['Rolling_Avg_Price_Change'] = df['Daily_Price_Change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['Daily_Return'] = df['close'] / df['open'] - 1
    df['Volume_Weighted_Return'] = df['Daily_Return'] * df['volume']
    df['Amount_Weighted_Return'] = df['Volume_Weighted_Return'] * df['amount']

    df['Intraday_Return'] = (df['high'] / df['low']) - 1
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] / df['open']) - 1
    
    # Combine Intraday and Close-to-Open Returns
    df['Intraday_Weight'] = np.abs(df['Intraday_Return'])
    df['Close_to_Open_Weight'] = np.abs(df['Close_to_Open_Return'])

    df['ATR'] = df['TR'].rolling(window=14).mean()
