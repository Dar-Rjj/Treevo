```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume-Weighted Daily Returns
    df['daily_return'] = df['close'].pct_change()
    df['volume_weighted_return'] = df['daily_return'] * df['volume']

    # Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']

    # Logarithmic Volume Difference
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_diff'] = df['log_volume'].diff().fillna(0)
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(alpha=0.2).mean()

    # Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']

    # Weigh by Intraday Volume Activity
    df['vol_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['vol_20_day_ma']
    df['weighted_intraday_spread'] = df['intraday_high_low_spread'] * df['volume_ratio']

    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])

    # Integrate Volume Change and Intraday Volatility
    df['adjusted_weights'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['smoothed_log_volume_diff'] * df['weighted_intraday_spread']

    # Combine Volume-Weighted Momentum with Price Range
    n = 20
    df['raw_momentum'] = df['close'] - df['close'].shift(n)
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum()
    df['volume_weighted_momentum'] = df['raw_momentum'] / df['cumulative_volume']
    df['price_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['price_range']

    # Weigh Combined Momentum by Intraday Volume Activity
    df['weighted_combined_momentum'] = df['combined_momentum'] * df['volume_ratio']

    # Apply 5-Day Exponential Moving Average
    df['ema_combined_momentum'] = df['weighted_combined_momentum'].ewm(span=5, adjust=False).mean()

    # Final Factor Calculation
    df['alpha_factor'] = df['volume_weighted_return'] * df['adjusted_weights'] * df['ema_combined_momentum']

    # Incorporate Intraday Range Breakouts
    df['prev_high'] = df['high'].shift(1)
    df['prev_low'] = df['low'].shift(1)
    df['range_breakout'] = (df['high'] > df['prev_high']) | (df['low'] < df['prev_low'])
    df['range_breakout_value'] = np.where(df['range_breakout'], 1, 0) * df['volume_ratio']

    # Final Alpha Factor
    df['final_alpha_factor'] = df['alpha_factor'] + df['range_breakout_value']

    return df['final_alpha_factor'].iloc[-1]
```
