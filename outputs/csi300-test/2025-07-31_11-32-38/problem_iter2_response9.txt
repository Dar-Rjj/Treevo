```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted High-Low Spread
    df['volume_adjusted_high_low'] = (df['high'] - df['low']) * df['volume']
    
    # Calculate Momentum of Volume-Adjusted High-Low Spread
    df['vol_adj_momentum'] = df['volume_adjusted_high_low'].rolling(window=5).mean() - df['volume_adjusted_high_low'].shift(1)
    
    # Integrate Simple Moving Average Crossover
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['sma_crossover'] = df['sma_5'] - df['sma_20']
    
    # Calculate Exponential Weighted Moving Average Growth
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['ema_growth'] = df['ema_10'] - df['ema_30']
    
    # Compute Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Measure True Range Averaging
    df['true_range'] = df[['high', 'low', 'close']].join(df['close'].shift(1), rsuffix='_prev').apply(
        lambda row: max(row['high'] - row['low'], abs(row['high'] - row['close_prev']), abs(row['low'] - row['close_prev'])), axis=1
    )
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Volume-Weighted Price Change
    df['abs_close_change'] = np.abs(df['close'].diff())
    df['volume_weighted_change'] = (df['abs_close_change'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Combine Indicators into Final Composite Alpha Factor
    df['composite_alpha'] = (
        df['vol_adj_momentum'] + 
        df['sma_crossover'] + 
        df['ema_growth'] + 
        (df['rsi'] - 50) / 50 + 
        df['volume_weighted_change'] / df['atr_14']
    )
    
    return df['composite_alpha'].dropna()
```
