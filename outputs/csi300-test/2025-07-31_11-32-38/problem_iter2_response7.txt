```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Rolling Average of Daily Price Changes (10-Day Moving Average)
    df['Price_Change_MA_10'] = df['Daily_Price_Change'].rolling(window=10).mean()
    
    # Calculate Volume Weighted Daily Return
    df['Daily_Return'] = df['close'] / df['open']
    df['Volume_Weighted_Return'] = df['Daily_Return'] * df['volume']
    
    # Summarize Volume Weighted Daily Returns (20-Day Moving Average)
    df['Volume_Weighted_Return_MA_20'] = df['Volume_Weighted_Return'].rolling(window=20).mean()
    
    # Combine Signals
    df['Signal'] = df['Price_Change_MA_10'] - df['Volume_Weighted_Return_MA_20']
    df['Smoothed_Signal'] = df['Signal'].rolling(window=5).mean()
    
    # Calculate True Range (TR)
    df['High_Low'] = df['high'] - df['low']
    df['High_PrevClose'] = (df['high'] - df['close'].shift(1)).abs()
    df['Low_PrevClose'] = (df['low'] - df['close'].shift(1)).abs()
    df['True_Range'] = df[['High_Low', 'High_PrevClose', 'Low_PrevClose']].max(axis=1)
    
    # Calculate Average True Range (ATR) (14-day ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['Positive_DM'] = df['high'] - df['high'].shift(1)
    df['Positive_DM'] = df['Positive_DM'].where((df['Positive_DM'] > 0) & (df['high'] > df['high'].shift(1)) & (df['low'] < df['low'].shift(1)), 0)
    
    # Calculate Negative Directional Movement (-DM)
    df['Negative_DM'] = df['low'].shift(1) - df['low']
    df['Negative_DM'] = df['Negative_DM'].where((df['Negative_DM'] > 0) & (df['low'] < df['low'].shift(1)) & (df['high'] > df['high'].shift(1)), 0)
    
    # Calculate +DM and -DM Smoothed Over 14 Periods
    df['Positive_DM_Smoothed'] = df['Positive_DM'].rolling(window=14).sum() / 14
    df['Negative_DM_Smoothed'] = df['Negative_DM'].rolling(window=14).sum() / 14
    
    # Calculate +DI and -DI
    df['Positive_DI'] = df['Positive_DM_Smoothed'] / df['ATR_14']
    df['Negative_DI'] = df['Negative_DM_Smoothed'] / df['ATR_14']
    
    # Calculate ADMI
    df['ADMI'] = (df['Positive_DI'] - df['Negative_DI']) / (df['Positive_DI'] + df['Negative_DI'])
    
    # Combine Final Alpha Factors
    df['Final_Alpha_Factor'] = df['ADMI'] + df['Smoothed_Signal']
    
    return df['Final_Alpha_Factor']
```
