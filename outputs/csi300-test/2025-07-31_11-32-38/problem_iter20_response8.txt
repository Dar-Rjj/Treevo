```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor from the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: Series representing the factor values indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the 20-day simple moving average of the typical price
    sma_20 = typical_price.rolling(window=20).mean()
    
    # Calculate the 5-day exponential moving average of the typical price
    ema_5 = typical_price.ewm(span=5, adjust=False).mean()
    
    # Calculate the money flow ratio
    money_flow = df['amount'] / df['volume']
    
    # Calculate the 10-day rolling mean of the money flow
    money_flow_10 = money_flow.rolling(window=10).mean()
    
    # Calculate the 50-day rolling standard deviation of the close price
    std_50 = df['close'].rolling(window=50).std()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (ema_5 - sma_20) / std_50 * money_flow_10
    
    return alpha_factor
```
