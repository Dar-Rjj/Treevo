```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) indexed by date.
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by date.
    """
    # Calculate the price range for each day
    price_range = df['high'] - df['low']
    
    # Calculate the buying pressure (volume * (close - low))
    buying_pressure = (df['close'] - df['low']) * df['volume']
    
    # Calculate the selling pressure (volume * (high - close))
    selling_pressure = (df['high'] - df['close']) * df['volume']
    
    # Calculate the net pressure (buying - selling)
    net_pressure = buying_pressure - selling_pressure
    
    # Calculate the normalized net pressure
    normalized_net_pressure = net_pressure / price_range
    
    # Calculate the 10-day moving average of the normalized net pressure
    avg_normalized_net_pressure = normalized_net_pressure.rolling(window=10).mean()
    
    # Calculate the 10-day moving standard deviation of the normalized net pressure
    std_normalized_net_pressure = normalized_net_pressure.rolling(window=10).std()
    
    # Calculate the Z-score of the normalized net pressure
    z_score = (normalized_net_pressure - avg_normalized_net_pressure) / std_normalized_net_pressure
    
    # Return the Z-score as the alpha factor
    return z_score
```
