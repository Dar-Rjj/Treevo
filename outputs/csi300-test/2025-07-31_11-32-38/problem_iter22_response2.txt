```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 30-day simple moving average (SMA) of close prices
    sma_30 = df['close'].rolling(window=30).mean()
    
    # Calculate the percentage change in close price from one day to the next
    close_pct_change = df['close'].pct_change()
    
    # Calculate daily volume change as a percentage of the previous day's volume
    volume_pct_change = df['volume'].pct_change()
    
    # Calculate the true range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate the average true range (ATR) over a 14-day period
    atr_14 = tr.rolling(window=14).mean()
    
    # Calculate the average high-low difference over a 30-day period
    avg_high_low_30 = (df['high'] - df['low']).rolling(window=30).mean()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (sma_30 + close_pct_change + volume_pct_change + atr_14 + avg_high_low_30) / 5
    
    return alpha_factor
```
