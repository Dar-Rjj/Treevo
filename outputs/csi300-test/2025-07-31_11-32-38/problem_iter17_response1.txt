```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['volume_amount_weighted_return'] = (df['close'] / df['open']) * (df['volume'] * df['amount'])
    
    # Summarize Volume and Amount Weighted Daily Returns
    df['20_day_ma_volume_amount_return'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['20_day_ma_volume'] = df['volume'].rolling(window=20).mean()
    df['weighted_intraday_high_low_spread'] = df['intraday_high_low_spread'] * (df['volume'] / df['20_day_ma_volume'])
    
    # Incorporate Intraday Price Trend
    df['intraday_return'] = df['close'] - df['open']
    df['5_day_mean_intraday_return'] = df['intraday_return'].rolling(window=5).mean()
    df['intraday_trend'] = df['intraday_return'] - df['5_day_mean_intraday_return']
    
    # Combine Intraday and Interday Momentum
    df['combined_momentum'] = df['10_day_ma_price_change'] + df['intraday_trend'] - df['20_day_ma_volume_amount_return']
    
    # Adjust for Price Volatility
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(), axis=0).abs().max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    df['volatility_adjusted_momentum'] = df['combined_momentum'] / df['atr_14']
    
    # Integrate Volume Change
    df['daily_volume_change'] = df['volume'].diff()
    min_vol = df['daily_volume_change'].min()
    max_vol = df['daily_volume_change'].max()
    df['normalized_volume_change'] = (df['daily_volume_change'] - min_vol) / (max_vol - min_vol)
    df['momentum_with_volume'] = df['volatility_adjusted_momentum'] * df['normalized_volume_change']
    
    # Calculate Positive and Negative Directional Movements
    df['+dm'] = (df['high'] - df['high'].shift()).where((df['high'] > df['high'].shift()) & (df['low'] >= df['low'].shift()), 0)
    df['-dm'] = (df['low'].shift() - df['low']).where((df['low'] < df['low'].shift()) & (df['high'] <= df['high'].shift()), 0)
    
    # Smooth Directional Movements
    df['+dm_smoothed'] = df['+dm'].rolling(window=14).mean()
    df['-dm_smoothed'] = df['-dm'].rolling(window=14).mean()
    
    # Calculate +DI and -DI
    df['+di'] = df['+dm_smoothed'] / df['atr_14']
    df['-di'] = df['-dm_smoothed'] / df['atr_14']
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum_with_volume'] * df['admi']) + (df['weighted_intraday_high_low_spread'] * df['admi'])
    
    # Smooth the Result
    df['alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(span=5, adjust=False).mean()
    
    return df['alpha_factor_smoothed']
```
