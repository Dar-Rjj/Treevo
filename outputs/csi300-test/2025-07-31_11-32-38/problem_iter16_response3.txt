```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Close-to-Open Returns
    weighted_intraday = intraday_return * abs(intraday_return)
    weighted_close_to_open = close_to_open_return * abs(close_to_open_return)
    preliminary_factor = weighted_intraday + weighted_close_to_open
    
    # Calculate True Range (TR)
    prev_close = df['close'].shift(1)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - prev_close).abs()
    tr3 = (df['low'] - prev_close).abs()
    tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Positive Directional Movement (+DM)
    high_diff = df['high'] - df['high'].shift(1)
    low_diff = df['low'].shift(1) - df['low']
    positive_dm = (high_diff > low_diff) & (high_diff > 0) * high_diff
    positive_dm[positive_dm < 0] = 0
    
    # Calculate Negative Directional Movement (-DM)
    negative_dm = (low_diff > high_diff) & (low_diff > 0) * low_diff
    negative_dm[negative_dm < 0] = 0
    
    # Smooth +DM and -DM over 14 periods
    smooth_positive_dm = positive_dm.rolling(window=14).sum()
    smooth_negative_dm = negative_dm.rolling(window=14).sum()
    tr_14 = tr.rolling(window=14).sum()
    
    # Calculate +DI and -DI
    plus_di = 100 * (smooth_positive_dm / tr_14)
    minus_di = 100 * (smooth_negative_dm / tr_14)
    
    # Calculate ADMI
    admi = (plus_di - minus_di) / (plus_di + minus_di)
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = preliminary_factor * admi
    
    return final_alpha_factor
```
