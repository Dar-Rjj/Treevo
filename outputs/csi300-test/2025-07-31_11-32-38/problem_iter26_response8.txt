import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=20, p=5, q=14):
    # Calculate Price Momentum Factor
    df['price_momentum'] = df['close'].diff(n)
    
    # Incorporate Trading Volume into the Analysis
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=m).mean()
    
    # Explore the Relationship Between Open and Close Prices
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_sum'] = df['open_close_diff'].rolling(window=p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range'] = df['daily_range'].rolling(window=q).mean()
    
    # Calculate Volume-Weighted Daily Returns
    df['vol_weighted_returns'] = (df['close'] / df['close'].shift(1) - 1) * df['volume']
    
    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Weigh by Intraday Volume Activity
    df['intraday_volume_activity'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Integrate Components
    df['weighted_intraday_return'] = df['intraday_return'] * df['intraday_return'].abs()
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['close_to_open_return'].abs()
    df['adjusted_weights'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['log_volume_diff']
    
    # Combine Volume-Weighted Momentum with Price Range
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    df['raw_momentum'] = df['close'] - df['close'].shift(n)
    df['momentum_with_volume'] = df['raw_momentum'] * cumulative_volume
    df['combined_momentum'] = df['momentum_with_volume'] + df['daily_range']
    df['combined_momentum_adjusted'] = df['combined_momentum'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_5'] = df['combined_momentum_adjusted'].ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    df['final_factor'] = df['vol_weighted_returns'] * df['adjusted_weights'] * df['ema_5']
    
    # Calculate Simple Moving Average (SMA) of Close Prices
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Measure Price Changes for Short-Term Momentum
    df['short_term_momentum'] = df['close'].pct_change()
    
    # Analyze Trading Volume for Liquidity and Activity
    df['volume_change'] = df['volume'].pct_change()
    
    # Evaluate Volatility Using True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Combine Price and Volume for a Comprehensive Factor
    df['comprehensive_factor'] = df['sma_10'] * df['volume'] + df['sma_30'] * df['volume']
    
    # Incorporate High and Low Prices for Additional Volatility Insights
    df['high_low_diff_10'] = (df['high'] - df['low']).rolling(window=10).mean()
    df['high_low_diff_30'] = (df['high'] - df['low']).rolling(window=30).mean()
    
    return df['final_factor']
