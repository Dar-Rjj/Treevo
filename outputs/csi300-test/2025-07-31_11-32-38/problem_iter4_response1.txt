```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate 20-Day Moving Average of Volume
    volume_20_day_moving_average = df['volume'].rolling(window=20).mean()
    
    # Weigh by Intraday Volume Activity
    volume_ratio = df['volume'] / volume_20_day_moving_average
    weighted_intraday_spread = intraday_high_low_spread * volume_ratio
    
    # Apply 5-Day Exponential Moving Average with smoothing factor 0.2
    ema_factor = weighted_intraday_spread.ewm(span=5, adjust=False).mean()
    
    return ema_factor
```
