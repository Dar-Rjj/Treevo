```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Enhanced Momentum Indicators
    df['5_SMA'] = df['close'].rolling(window=5).mean()
    df['20_SMA'] = df['close'].rolling(window=20).mean()
    df['SMA_Crossover'] = (df['5_SMA'] > df['20_SMA']).astype(int)

    df['10_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['30_EMA'] = df['close'].ewm(span=30, adjust=False).mean()
    df['EMA_Growth'] = (df['10_EMA'] - df['30_EMA']) / df['30_EMA']

    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    df['RSI'] = 100 - (100 / (1 + RS))

    # Enhanced Volatility Measures
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    df['ATR'] = true_range.rolling(window=20).mean()

    df['Abs_Close_Change'] = abs(df['close'].diff())
    df['Volume_Weighted_Price_Change'] = df['Abs_Close_Change'] * df['volume']

    df['Volume_Change_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Volume_Spike'] = (df['Volume_Change_Ratio'] > 2).astype(int)

    # On-Balance Volume (OBV)
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).cumsum()

    # Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    MFR = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    df['MFI'] = 100 - (100 / (1 + MFR))

    # Combine Momentum, Volatility, and Volume
    df['Momentum_Score'] = 0.5 * df['SMA_Crossover'] + 0.3 * df['EMA_Growth'] + 0.2 * (100 - df['RSI'])
    df['Volatility_Score'] = 1 / df['ATR']
    df['Volume_Score'] = 0.5 * df['OBV'] + 0.5 * (100 - df['MFI'])

    # Final Composite Alpha Factor
    df['Composite_Alpha'] = 0.4 * df['Momentum_Score'] + 0.3 * df['Volatility_Score'] + 0.3 * df['Volume_Score']

    return df['Composite_Alpha']
```
