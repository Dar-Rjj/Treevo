```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate Rolling Average of Daily Price Changes (10-day MA)
    df['rolling_avg_daily_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume and Amount Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_amount_weighted_return'] = df['daily_return'] * df['volume'] * df['amount']
    
    # Summarize Volume and Amount Weighted Daily Returns (20-day MA)
    df['weighted_return_ma'] = df['volume_amount_weighted_return'].rolling(window=20).mean()
    
    # Combine Signals
    df['signal_difference'] = df['rolling_avg_daily_change'] - df['weighted_return_ma']
    df['prelim_alpha_factor'] = df['signal_difference'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Weight by Logarithmic Volume Difference
    df['log_volume'] = np.log(df['volume'])
    df['log_volume_shift'] = df['log_volume'].shift(1)
    df['log_volume_diff'] = df['log_volume'] - df['log_volume_shift']
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(span=5, adjust=False).mean()
    
    # Combine Intraday and Close-to-Open Returns
    df['intraday_weight'] = np.abs(df['intraday_return'])
    df['close_to_open_weight'] = np.abs(df['close_to_open_return'])
    
    # Adjust Weights by Volume
    df['intraday_weight_adj'] = df['intraday_weight'] * df['smoothed_log_volume_diff']
    df['close_to_open_weight_adj'] = df['close_to_open_weight'] * df['smoothed_log_volume_diff']
    
    # Integrate Volume Change
    df['intraday_prelim_factor'] = (df['intraday_weight_adj'] + df['close_to_open_weight_adj']) * df['volume_change']
    
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Average True Range (ATR) (14-day ATR)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+dm'] = (df['high'] - df['high'].shift(1)).apply(lambda x: max(x, 0))
    df['-dm'] = (df['low'].shift(1) - df['low']).apply(lambda x: max(x, 0))
    
    # Smooth +DM and -DM
    df['smoothed_dm_plus'] = df['+dm'].rolling(window=14).sum() / 14
    df['smoothed_dm_minus'] = df['-dm'].rolling(window=14).sum() / 14
    
    # Calculate +DI and -DI
    df['+di'] = df['smoothed_dm_plus'] / df['atr']
    df['-di'] = df['smoothed_dm_minus'] / df['atr']
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['prelim_alpha_factor'] * df['admi']) + (df['intraday_prelim_factor'] * df['admi'])
    
    return df['final_alpha_factor']
```
