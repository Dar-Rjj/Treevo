    # Calculate 20-day Average True Range (ATR)
    df['TR'] = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1)
    df['ATR_20'] = df['TR'].rolling(window=20).mean()
    
    # Calculate standard deviation of log returns over a 20-day period
    df['Log_Return_STD_20'] = df['Log_Return'].rolling(window=20).std()
    
    # Combine price and volume for a more comprehensive factor
    df['SMA_Vol_20'] = df['SMA_20'] * df['Volume_MA_10']
    df['SMA_Vol_50'] = df['SMA_50'] * df['Volume_MA_10']
    
    # Calculate the difference between the high and low of the day
    df['High_Low_Diff'] = df['high'] - df['low']
    
    # Average the high-low difference over a 20-day period
    df['High_Low_MA_20'] = df['High_Low_Diff'].rolling(window=20).mean()
    
    # Average the high-low difference over a 50-day period
    df['High_Low_MA_50'] = df['High_Low_Diff'].rolling(window=50).mean()
    
    # Calculate the percentage change from open to close
    df['Open_Close_Change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the ratio of close to open prices
    df['Close_Open_Ratio'] = df['close'] / df['open']
    
    # Generate the alpha factor
    df['Alpha_Factor'] = (
        df['SMA_20'] * df['Volume_MA_10'] +
        df['SMA_50'] * df['Volume_MA_10'] +
        df['Close_Change'] +
        df['Log_Return'] +
        df['Volume_Change'] +
        df['ATR_20'] +
        df['Log_Return_STD_20'] +
        df['High_Low_MA_20'] +
        df['High_Low_MA_50'] +
        df['Open_Close_Change'] +
        df['Close_Open_Ratio']
    ) / 11  # Normalize by the number of factors
    
    return df['Alpha_Factor']
