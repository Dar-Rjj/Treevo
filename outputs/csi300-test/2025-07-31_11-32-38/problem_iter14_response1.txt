import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume Weighted Daily Return
    df['volume_weighted_return'] = (df['close'] / df['open']) * df['volume']
    
    # Compute 20-Day Moving Average of Volume Weighted Returns
    df['20_day_ma_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    df['20_day_ma_volume'] = df['volume'].rolling(window=20).mean()
    df['weighted_intraday_high_low_spread'] = df['intraday_high_low_spread'] * (df['volume'] / df['20_day_ma_volume'])
    
    # Incorporate Intraday Price Trend
    df['intraday_return'] = df['close'] - df['open']
    df['5_day_mean_intraday_return'] = df['intraday_return'].rolling(window=5).mean()
    df['intraday_price_trend'] = df['intraday_return'] - df['5_day_mean_intraday_return']
    
    # Combine Intraday and Interday Momentum
    df['combined_momentum'] = (df['10_day_ma_price_change'] + df['intraday_price_trend']) - df['20_day_ma_volume_weighted_return']
    
    # Adjust for Price Volatility
    df['daily_price_range'] = df['high'] - df['low']
    df['30_day_ma_daily_price_range'] = df['daily_price_range'].rolling(window=30).mean()
    df['volatility_adjusted_momentum'] = df['combined_momentum'] / df['30_day_ma_daily_price_range']
    
    # Smooth the Result
    df['smoothed_volatility_adjusted_momentum'] = df['volatility_adjusted_momentum'].ewm(span=5, adjust=False).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(1)
    
    # Weight by Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))
    df['smoothed_log_volume_diff'] = df['log_volume_diff'].ewm(span=5, adjust=False).mean()
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * np.abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * np.abs(df['close_to_open_return'])
    
    # Adjust Weights by Volume
    df['weighted_intraday_return'] *= df['smoothed_log_volume_diff']
    df['weighted_close_to_open_return'] *= df['smoothed_log_volume_diff']
    
    # Final Factor Calculation
    df['final_factor'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']).cumsum()
    
    # Integrate Smoothed Result with Volume-Adjusted Returns
    df['alpha_factor'] = df['smoothed_volatility_adjusted_momentum'] + df['final_factor']
    
    return df['alpha_factor'].dropna()
