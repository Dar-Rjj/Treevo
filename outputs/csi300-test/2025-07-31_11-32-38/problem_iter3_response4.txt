```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Calculate Logarithmic Volume
    df['log_volume_t'] = np.log(df['volume'])
    df['log_volume_t-1'] = df['log_volume_t'].shift(1)
    
    # Subtract and Smooth Volume Difference
    df['log_volume_diff'] = df['log_volume_t'] - df['log_volume_t-1']
    df['smoothed_volume_diff'] = df['log_volume_diff'].ewm(span=n, adjust=False).mean()
    
    # Calculate Momentum
    df['momentum'] = df['daily_returns'] * df['smoothed_volume_diff']
    
    # Calculate Daily High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average High-Low Range Over n Days
    df['avg_high_low_range'] = df['high_low_range'].rolling(window=n).mean()
    
    # Final Factor Calculation
    df['price_volatility_adjustment'] = df['high_low_range'] / df['avg_high_low_range']
    df['factor_value'] = (df['daily_returns'] * df['smoothed_volume_diff']) / df['price_volatility_adjustment']
    df['final_factor'] = df['factor_value'].rolling(window=n).sum()
    
    return df['final_factor']
```
