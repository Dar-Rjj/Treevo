```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 20-Day Moving Average of Daily Price Changes
    df['20_day_ma_price_change'] = df['daily_price_change'].rolling(window=20).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday and Close-to-Open Returns
    df['weighted_intraday_return'] = df['intraday_return'] * df['intraday_return'].abs()
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['close_to_open_return'].abs()
    
    # Integrate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['preliminary_factor'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['volume_change']
    
    # Calculate True Range (TR)
    df['tr_1'] = df['high'] - df['low']
    df['tr_2'] = (df['close'].shift(1) - df['high']).abs()
    df['tr_3'] = (df['close'].shift(1) - df['low']).abs()
    df['true_range'] = df[['tr_1', 'tr_2', 'tr_3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Positive Directional Movement (+DM)
    df['+dm'] = np.where((df['high'] > df['high'].shift(1)) & (df['low'] >= df['low'].shift(1)), 
                         df['high'] - df['high'].shift(1), 0)
    
    # Calculate Negative Directional Movement (-DM)
    df['-dm'] = np.where((df['low'] < df['low'].shift(1)) & (df['high'] <= df['high'].shift(1)), 
                         df['low'].shift(1) - df['low'], 0)
    
    # Calculate +DM Smoothed Over 14 Periods
    df['+dm_smoothed'] = df['+dm'].rolling(window=14).mean()
    
    # Calculate -DM Smoothed Over 14 Periods
    df['-dm_smoothed'] = df['-dm'].rolling(window=14).mean()
    
    # Calculate +DI
    df['+di'] = df['+dm_smoothed'] / df['atr']
    
    # Calculate -DI
    df['-di'] = df['-dm_smoothed'] / df['atr']
    
    # Calculate ADMI
    df['admi'] = (df['+di'] - df['-di']) / (df['+di'] + df['-di'])
    
    # Calculate 5-Day Moving Average of Preliminary Factor
    df['5_day_ma_preliminary_factor'] = df['preliminary_factor'].rolling(window=5).mean()
    
    # Synthesize Final Alpha Factor
    df['difference'] = df['20_day_ma_price_change'] - df['5_day_ma_preliminary_factor']
    df['final_alpha_factor'] = df['difference'].rolling(window=7).mean() * df['admi']
    
    return df['final_alpha_factor']
```
