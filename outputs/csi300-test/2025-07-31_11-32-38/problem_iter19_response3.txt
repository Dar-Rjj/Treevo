```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, p=5, q=10):
    """
    Calculate a composite alpha factor using the provided DataFrame.
    
    Parameters:
    - df: pandas DataFrame with (date) index and columns: open, high, low, close, amount, volume
    - n: Number of days to look back for price momentum
    - m: Number of days to look back for average volume
    - p: Number of days to look back for moving sum of open-close differences
    - q: Number of days to look back for moving average of daily price range
    
    Returns:
    - pandas Series with the calculated alpha factor values indexed by (date)
    """
    
    # Price Momentum Factor
    price_momentum = df['close'] - df['close'].shift(n)
    
    # Volume Ratio Factor
    avg_volume = df['volume'].rolling(window=m).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Open-Close Difference Moving Sum
    open_close_diff = df['close'] - df['open']
    open_close_sum = open_close_diff.rolling(window=p).sum()
    
    # Daily Price Range and Moving Average
    daily_range = df['high'] - df['low']
    range_moving_avg = daily_range.rolling(window=q).mean()
    
    # Composite Alpha Factor
    alpha_factor = (price_momentum + volume_ratio + open_close_sum + range_moving_avg) / 4
    
    return alpha_factor
```
