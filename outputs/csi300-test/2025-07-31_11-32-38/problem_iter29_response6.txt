import pandas as pd
import numpy as np

def heuristics_v2(df, n=30, m=10, p=5, q=10, r=20):
    # Calculate Price Momentum Factor
    momentum = df['close'] - df['close'].shift(n)
    
    # Incorporate Trading Volume into the Analysis
    volume_ratio = df['volume'] / df['volume'].rolling(m).mean()
    
    # Explore the Relationship Between Open and Close Prices
    open_close_diff = df['open'] - df['close']
    open_close_sum = open_close_diff.rolling(p).sum()
    
    # Analyze the Range of Daily Price Fluctuations
    price_range = df['high'] - df['low']
    avg_price_range = price_range.rolling(q).mean()
    
    # Calculate Volume-Weighted Daily Returns
    daily_returns = (df['close'] / df['close'].shift(1)) - 1
    volume_weighted_returns = daily_returns * volume_ratio
    
    # Compute Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Logarithmic Volume Difference
    log_volume_diff = np.log(df['volume']) - np.log(df['volume'].shift(1))
    
    # Weigh by Intraday Volume Activity
    volume_20_day_avg = df['volume'].rolling(20).mean()
    volume_activity_ratio = df['volume'] / volume_20_day_avg
    
    # Integrate Components
    # Combine Intraday and Close-to-Open Returns
    weighted_intraday_return = intraday_return * np.abs(intraday_return)
    weighted_close_to_open_return = close_to_open_return * np.abs(close_to_open_return)
    combined_returns = weighted_intraday_return + weighted_close_to_open_return
    
    # Adjust weights by multiplying them with the Logarithmic Volume Difference
    adjusted_weights = combined_returns * log_volume_diff * volume_activity_ratio
    
    # Combine Volume-Weighted Momentum with Price Range
    raw_momentum = df['close'] - df['close'].shift(n)
    cumulative_volume = df['volume'].rolling(n).sum().shift(1)
    volume_weighted_momentum = raw_momentum / cumulative_volume
    combined_momentum = volume_weighted_momentum + avg_price_range
    
    # Multiply Combined Momentum by Volume Ratio
    combined_momentum_adjusted = combined_momentum * volume_ratio
    
    # Apply 5-Day Exponential Moving Average
    ema_factor = combined_momentum_adjusted.ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    final_returns = volume_weighted_returns * adjusted_weights * ema_factor
    
    # Incorporate Intraday Range Breakouts
    intraday_breakout = (df['high'] - df['low']) / (df['high'].shift(1) - df['low'].shift(1))
    intraday_breakout_weighted = intraday_breakout * volume_activity_ratio
    
    # Analyze Intraday Volatility
    intraday_volatility = intraday_return.rolling(r).std()
    intraday_volatility_weighted = intraday_volatility * log_volume_diff
    
    # Enhance Volume-Weighted Indicators
    volume_weighted_price_change = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].shift(1))
    smoothed_volume_weighted_price_change = volume_weighted_price_change.rolling(10).mean()
    enhanced_factor = smoothed_volume_weighted_price_change * (intraday_breakout * intraday_volatility_weighted)
    
    # Final Alpha Factor
    alpha_factor = final_returns * enhanced_factor
    
    return alpha_factor
