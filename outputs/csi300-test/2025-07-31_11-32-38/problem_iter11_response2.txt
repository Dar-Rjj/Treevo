import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel and interpretable alpha factor using the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_10_daily_return'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the daily return
    df['std_10_daily_return'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the 10-day exponential moving average of the close price
    df['ema_10_close'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the close price
    df['ema_30_close'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 10-day exponential moving average of the volume
    df['ema_10_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the volume
    df['ema_30_volume'] = df['volume'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 10-day rolling Z-score of the daily return
    df['z_score_10_daily_return'] = (df['daily_return'] - df['ma_10_daily_return']) / df['std_10_daily_return']
    
    # Calculate the 10-day rolling Z-score of the close price
    df['z_score_10_close'] = (df['close'] - df['close'].rolling(window=10).mean()) / df['close'].rolling(window=10).std()
    
    # Calculate the 10-day rolling Z-score of the volume
    df['z_score_10_volume'] = (df['volume'] - df['volume'].rolling(window=10).mean()) / df['volume'].rolling(window=10).std()
    
    # Calculate the alpha factor
    df['alpha_factor'] = (
        df['z_score_10_daily_return'] * 0.3 +
        df['z_score_10_close'] * 0.4 +
        (df['ema_10_close'] - df['ema_30_close']) * 0.2 +
        (df['ema_10_volume'] - df['ema_30_volume']) * 0.1
    )
    
    return df['alpha_factor']
