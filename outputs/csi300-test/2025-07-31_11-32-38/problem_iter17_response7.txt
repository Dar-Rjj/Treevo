import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is a combination of price and volume data, designed to predict future stock returns.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by date.
    """
    
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the price momentum
    price_momentum = typical_price.pct_change(periods=5)
    
    # Calculate the volume momentum
    volume_momentum = df['volume'].pct_change(periods=5)
    
    # Calculate the weighted average of the price and volume momentum
    weight_price = 0.7
    weight_volume = 0.3
    combined_momentum = (weight_price * price_momentum + weight_volume * volume_momentum)
    
    # Calculate the moving average of the combined momentum over a 10-day window
    ma_combined_momentum = combined_momentum.rolling(window=10).mean()
    
    # Calculate the final alpha factor
    alpha_factor = ma_combined_momentum * df['amount']
    
    return alpha_factor
