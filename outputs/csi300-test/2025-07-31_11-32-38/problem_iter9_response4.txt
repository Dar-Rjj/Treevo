```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # Compute 10-Day Moving Average of Daily Price Changes
    df['10D_MA_Daily_Price_Change'] = df['Daily_Price_Change'].rolling(window=10).mean()

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['open']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']

    # Combine Intraday and Close-to-Open Returns
    df['Weighted_Intraday_Return'] = df['Intraday_Return'] * abs(df['Intraday_Return'])
    df['Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * abs(df['Close_to_Open_Return'])

    # Integrate Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    df['Preliminary_Factor'] = (df['Weighted_Intraday_Return'] + df['Weighted_Close_to_Open_Return']) * df['Volume_Change']

    # Calculate True Range (TR)
    df['TR_High_Low'] = df['high'] - df['low']
    df['TR_PrevClose_High'] = (df['high'] - df['close'].shift()).abs()
    df['TR_PrevClose_Low'] = (df['low'] - df['close'].shift()).abs()
    df['True_Range'] = df[['TR_High_Low', 'TR_PrevClose_High', 'TR_PrevClose_Low']].max(axis=1)

    # Calculate Average True Range (ATR) over n periods (e.g., 14 days)
    n = 14
    df['ATR'] = df['True_Range'].rolling(window=n).mean()

    # Calculate Positive Directional Movement (+DM)
    df['+DM'] = np.where((df['high'] > df['high'].shift()) & (df['low'] >= df['low'].shift()), df['high'] - df['high'].shift(), 0)

    # Calculate Negative Directional Movement (-DM)
    df['-DM'] = np.where((df['low'] < df['low'].shift()) & (df['high'] <= df['high'].shift()), df['low'].shift() - df['low'], 0)

    # Smooth +DM and -DM over n periods
    df['+DM_Smoothed'] = df['+DM'].rolling(window=n).sum() / n
    df['-DM_Smoothed'] = df['-DM'].rolling(window=n).sum() / n

    # Calculate +DI and -DI
    df['+DI'] = (df['+DM_Smoothed'] / df['ATR']) * 100
    df['-DI'] = (df['-DM_Smoothed'] / df['ATR']) * 100

    # Calculate ADMI
    df['ADMI'] = (df['+DI'] - df['-DI']) / (df['+DI'] + df['-DI'])

    # Incorporate Volume Weighted Returns
    df['Volume_Weighted_Intraday_Return'] = df['Intraday_Return'] * df['volume']
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['volume']
    df['20D_MA_Volume_Weighted_Return'] = df[['Volume_Weighted_Intraday_Return', 'Volume_Weighted_Close_to_Open_Return']].sum(axis=1).rolling(window=20).mean()

    # Synthesize Final Alpha Factor
    df['Final_Factor'] = (df['10D_MA_Daily_Price_Change'] - df['20D_MA_Volume_Weighted_Return']).rolling(window=5).mean() * df['Preliminary_Factor'] * df['ADMI']

    return df['Final_Factor']
```
