```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Compute Raw Momentum
    raw_momentum = df['close'] - df['close'].shift(n)
    
    # Weight by Cumulative Volume
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    volume_weighted_momentum = raw_momentum / cumulative_volume
    
    # Incorporate Price Range Factor
    price_range = df['high'] - df['low']
    combined_momentum = volume_weighted_momentum * price_range + volume_weighted_momentum
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    daily_volume_ma = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / daily_volume_ma
    weighted_spread = intraday_spread * volume_ratio
    
    # Integrate Open to Close Price Movement
    open_to_close_change = df['close'] - df['open']
    combined_momentum += volume_weighted_momentum * open_to_close_change
    
    # Smooth Combined Momentum and Intraday Volatility
    smooth_factor = 0.3
    smoothed_data = combined_momentum.ewm(span=7, adjust=False).mean() + weighted_spread.ewm(span=7, adjust=False).mean()
    
    # Enhanced Momentum and Reversal
    raw_returns = df['close'].pct_change()
    momentum_component = raw_returns.rolling(window=10).sum() - raw_returns.rolling(window=5).sum()
    reversal_component = raw_returns.rolling(window=21).mean() * -1
    
    # Normalize Momentum and Reversal Components
    normalized_momentum = (momentum_component - momentum_component.mean()) / momentum_component.std()
    normalized_reversal = (reversal_component - reversal_component.mean()) / reversal_component.std()
    
    # Combine Normalized Components
    combined_signal = normalized_momentum + normalized_reversal
    
    # Final Alpha Factor
    final_alpha = (smoothed_data - df['close'].rolling(window=5).mean()).rolling(window=5).mean() * combined_signal
    
    return final_alpha
```
