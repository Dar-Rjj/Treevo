```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple moving averages (SMA) of close prices
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_30'] = df['close'].rolling(window=30).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Measure price changes to capture short-term momentum
    df['pct_change_1'] = df['close'].pct_change()
    df['pct_change_2'] = df['close'].pct_change(periods=2)
    
    # Analyze trading volume for liquidity and activity signals
    df['volume_pct_change'] = df['volume'].pct_change()
    df['vol_3d_ma'] = df['volume'].rolling(window=3).mean()
    
    # Evaluate volatility using true range
    df['true_range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['ATR_14'] = df['true_range'].rolling(window=14).mean()
    df['ATR_21'] = df['true_range'].rolling(window=21).mean()
    
    # Combine price and volume for a more comprehensive factor
    df['SMA_10_vol'] = df['SMA_10'] * df['volume']
    df['SMA_30_vol_3d_ma'] = df['SMA_30'] * df['vol_3d_ma']
    df['SMA_50_vol_3d_ma'] = df['SMA_50'] * df['vol_3d_ma']
    
    # Incorporate high and low prices for additional volatility insights
    df['high_low_diff'] = df['high'] - df['low']
    df['avg_high_low_10'] = df['high_low_diff'].rolling(window=10).mean()
    df['avg_high_low_30'] = df['high_low_diff'].rolling(window=30).mean()
    df['avg_high_low_50'] = df['high_low_diff'].rolling(window=50).mean()
    
    # Integrate open and close prices for price gap analysis
    df['open_close_diff'] = df['open'] - df['close'].shift(1)
    df['avg_open_close_5'] = df['open_close_diff'].rolling(window=5).mean()
    
    # Examine the relationship between high, low, and close prices
    df['close_high_ratio'] = df['close'] / df['high']
    df['close_low_ratio'] = df['close'] / df['low']
    df['avg_close_high_10'] = df['close_high_ratio'].rolling(window=10).mean()
    df['avg_close_low_10'] = df['close_low_ratio'].rolling(window=10).mean()
    
    # Investigate the impact of large price movements
    df['large_movement'] = (df['high'] - df['low']) / df['close'] > 0.05
    df['large_movement_count_30'] = df['large_movement'].rolling(window=30).sum()
    df['avg_return_after_large_move'] = df['pct_change_1'].where(df['large_movement'].shift(1)).rolling(window=30).mean()
    
    # Final alpha factor: a combination of various factors
    df['alpha_factor'] = (
        df['SMA_30_vol_3d_ma'] + 
        df['avg_high_low_30'] + 
        df['avg_close_high_10'] + 
        df['avg_return_after_large_move']
    )
    
    return df['alpha_factor']
```
