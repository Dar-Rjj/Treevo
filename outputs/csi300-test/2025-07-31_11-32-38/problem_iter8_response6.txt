```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Raw Returns
    df['raw_returns'] = df['close'].pct_change()
    
    # Momentum Component
    df['ema_10'] = df['raw_returns'].ewm(span=10, adjust=False).mean()
    df['ema_5'] = df['raw_returns'].ewm(span=5, adjust=False).mean()
    df['momentum'] = df['ema_10'] - df['ema_5']
    
    # Reversal Component
    weights = np.arange(1, 22)
    df['wma_21'] = (df['raw_returns'].rolling(window=21).apply(lambda x: np.sum(weights * x) / np.sum(weights), raw=True))
    df['reversal'] = -df['wma_21']
    
    # Combine Momentum and Reversal
    df['scaled_momentum'] = df['momentum'] / df['momentum'].abs().max()
    df['scaled_reversal'] = df['reversal'] / df['reversal'].abs().max()
    df['combined'] = df['scaled_momentum'] + df['scaled_reversal']
    
    # Volume-Weighted Combined Momentum
    df['cumulative_volume'] = df['volume'].rolling(window=20).sum()
    df['volume_weighted_combined'] = df['combined'] / df['cumulative_volume']
    
    # Incorporate Price Range Factor
    df['price_range'] = df['high'] - df['low']
    df['price_range_factor'] = df['volume_weighted_combined'] + df['price_range']
    
    # Weigh by Intraday Volume Activity
    df['volume_20_day_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['volume_20_day_ma']
    df['weighted_momentum'] = df['price_range_factor'] * df['volume_ratio']
    
    # Smooth Final Momentum
    df['smoothed_momentum'] = df['weighted_momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    return df['smoothed_momentum']
```
