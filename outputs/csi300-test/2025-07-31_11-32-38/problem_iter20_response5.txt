```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=30, m=20, p=10, q=10):
    # Calculate Price Momentum Factor
    df['price_momentum'] = df['close'].diff(n)

    # Incorporate Trading Volume into the Analysis
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=m).mean()

    # Explore the Relationship Between Open and Close Prices
    df['open_close_diff'] = df['close'] - df['open']
    df['open_close_sum'] = df['open_close_diff'].rolling(window=p).sum()

    # Analyze the Range of Daily Price Fluctuations
    df['daily_range'] = df['high'] - df['low']
    df['range_moving_avg'] = df['daily_range'].rolling(window=q).mean()

    # Calculate Volume-Weighted Daily Returns
    df['prev_close'] = df['close'].shift(1)
    df['volume_weighted_return'] = (df['close'] - df['prev_close']) * df['volume']

    # Compute Intraday Return
    df['intraday_return'] = (df['high'] - df['low'])

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'])

    # Calculate Logarithmic Volume Difference
    df['log_volume_diff'] = np.log(df['volume']) - np.log(df['volume'].shift(1))

    # Weigh by Intraday Volume Activity
    df['volume_20_day_avg'] = df['volume'].rolling(window=20).mean()
    df['intraday_volume_weight'] = df['volume'] / df['volume_20_day_avg']

    # Integrate Components
    df['weighted_intraday_return'] = df['intraday_return'] * abs(df['intraday_return'])
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * abs(df['close_to_open_return'])
    
    df['adjusted_weights'] = df['log_volume_diff'] * (df['high'] - df['low'])
    df['combined_returns'] = (df['weighted_intraday_return'] + df['weighted_close_to_open_return']) * df['adjusted_weights']
    
    # Combine Volume-Weighted Momentum with Price Range
    df['cumulative_volume'] = df['volume'].rolling(window=n).sum().shift(1)
    df['volume_weighted_momentum'] = (df['close'] - df['close'].shift(n)) / df['cumulative_volume']
    df['price_range'] = df['high'] - df['low']
    df['combined_momentum'] = df['volume_weighted_momentum'] + df['price_range']
    
    df['momentum_with_volume_ratio'] = df['combined_momentum'] * df['volume_ratio']
    
    # Apply 5-Day Exponential Moving Average
    df['ema_factor'] = df['momentum_with_volume_ratio'].ewm(span=5, adjust=False).mean()
    
    # Multiply Volume-Weighted Daily Returns by Adjusted Weights
    df['final_factor'] = df['volume_weighted_return'] * df['adjusted_weights'] * df['ema_factor']
    
    # Incorporate Intraday Range Breakouts
    df['prev_high'] = df['high'].shift(1)
    df['prev_low'] = df['low'].shift(1)
    df['intraday_breakout'] = (df['high'] > df['prev_high']) & (df['low'] < df['prev_low'])
    df['breakout_weight'] = df['intraday_breakout'] * df['intraday_volume_weight']
    
    df['final_factor'] = df['final_factor'] * df['breakout_weight']
    
    return df['final_factor']
```
