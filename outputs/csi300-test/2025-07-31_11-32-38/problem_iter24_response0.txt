import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the provided DataFrame.
    The factor is based on the interaction between the close price, volume, and the relative strength of the close price compared to the high and low prices.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the alpha factor values indexed by date.
    """
    # Calculate the relative strength of the close price compared to the high and low prices
    df['close_strength'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the 10-day moving average of the close price
    df['close_ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the volume
    df['volume_ma_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the 10-day standard deviation of the close price
    df['close_std_10'] = df['close'].rolling(window=10).std()
    
    # Calculate the z-score of the close price
    df['close_z_score'] = (df['close'] - df['close_ma_10']) / df['close_std_10']
    
    # Calculate the alpha factor
    df['alpha_factor'] = df['close_strength'] * df['volume'] / df['volume_ma_10'] * df['close_z_score']
    
    return df['alpha_factor']
