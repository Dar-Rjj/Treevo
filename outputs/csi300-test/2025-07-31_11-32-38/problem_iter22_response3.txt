```python
import pandas as pd

def heuristics_v2(df, n=30, m=10, p=5, q=10):
    """
    Calculate a novel alpha factor based on the given thoughts.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    - n: Number of days to look back for price momentum
    - m: Number of days to calculate the average volume
    - p: Number of days for the moving sum of open-close differences
    - q: Number of days for the moving average of daily price ranges
    
    Returns:
    - pandas Series indexed by date representing the factor values
    """
    
    # Price Momentum Factor
    price_momentum = df['close'] - df['close'].shift(n)
    
    # Volume Ratio
    avg_volume_m_days = df['volume'].rolling(window=m).mean()
    volume_ratio = df['volume'] / avg_volume_m_days
    
    # Open-Close Difference Moving Sum
    open_close_diff = df['close'] - df['open']
    oc_moving_sum = open_close_diff.rolling(window=p).sum()
    
    # Daily Price Range Moving Average
    daily_range = df['high'] - df['low']
    range_moving_avg = daily_range.rolling(window=q).mean()
    
    # Combine all factors into a single alpha factor
    alpha_factor = (price_momentum * volume_ratio + oc_moving_sum) / range_moving_avg
    
    return alpha_factor
```
