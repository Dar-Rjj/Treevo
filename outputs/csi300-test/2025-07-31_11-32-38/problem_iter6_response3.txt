```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Compute Raw Momentum
    raw_momentum = df['close'].pct_change(periods=n)
    
    # Weight by Cumulative Volume
    cumulative_volume = df['volume'].rolling(window=n).sum().shift(1)
    volume_weighted_momentum = raw_momentum / cumulative_volume
    
    # Incorporate Price Range Factor
    price_range = df['high'] - df['low']
    combined_momentum = volume_weighted_momentum * price_range + volume_weighted_momentum
    
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Weigh by Intraday Volume Activity
    daily_volume_ma_20 = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / daily_volume_ma_20
    weighted_intraday_spread = intraday_high_low_spread * volume_ratio
    
    # Integrate Open to Close Price Movement
    open_to_close_change = df['close'] - df['open']
    combined_momentum = combined_momentum * open_to_close_change + combined_momentum
    
    # Smooth Combined Momentum and Intraday Volatility
    smoothed_factor = (combined_momentum + weighted_intraday_spread).ewm(span=7, adjust=False).mean()
    
    return smoothed_factor
```
