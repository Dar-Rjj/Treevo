```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 10-Day Moving Average of Daily Price Changes
    df['10_day_ma_price_change'] = df['daily_price_change'].rolling(window=10).mean()
    
    # Calculate Volume Weighted Daily Return
    df['daily_return'] = df['close'] / df['open']
    df['volume_weighted_return'] = df['daily_return'] * df['volume']
    
    # Compute 20-Day Moving Average of Volume Weighted Returns
    df['20_day_ma_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=20).mean()
    
    # Take Difference between 10-Day MA of Daily Price Changes and 20-Day MA of Volume Weighted Returns
    df['difference'] = df['10_day_ma_price_change'] - df['20_day_ma_volume_weighted_return']
    
    # Smooth the Result with 5-Day Moving Average
    df['smoothed_difference'] = df['difference'].rolling(window=5).mean()
    
    # Return the final alpha factor
    return df['smoothed_difference']
```
