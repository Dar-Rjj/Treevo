def heuristics_v2(df):
    # Calculate the exponential moving average of price returns with a span of 21
    ema_price_return = df['close'].pct_change().ewm(span=21, adjust=False).mean()
    
    # Calculate the modified accumulation/distribution line with a weighted volume
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line = (money_flow_volume.rolling(window=5).sum()).cumsum()
    
    # Heuristic factor
    heuristics_matrix = ema_price_return * ad_line
    
    return heuristics_matrix
