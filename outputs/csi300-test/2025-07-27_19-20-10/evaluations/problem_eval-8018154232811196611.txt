import pandas as pd

def heuristics_v2(df):
    # Calculate the mean absolute price change over a 21-day window
    mean_abs_price_change = df['close'].diff().abs().rolling(window=21).mean()
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'].diff() > 0).astype(int)).cumsum()
    
    # Heuristic factor
    heuristics_matrix = mean_abs_price_change * obv
    
    return heuristics_matrix
