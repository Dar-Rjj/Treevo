import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted moving average of price returns
    wma_price_return = df['close'].pct_change().rolling(window=14).apply(lambda x: (x * range(1, 15)).sum() / sum(range(1, 15)), raw=True)
    
    # Calculate the modified money flow index
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi = 100 - (100 / (1 + (positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum())))
    
    # Heuristic factor
    heuristics_matrix = wma_price_return * mfi
    
    return heuristics_matrix
