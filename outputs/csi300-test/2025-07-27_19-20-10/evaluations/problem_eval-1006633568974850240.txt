import pandas as pd

def heuristics_v2(df):
    # Calculate the mean absolute price change over a 30-day window
    mean_abs_price_change = df['close'].diff().abs().rolling(window=30).mean()
    
    # Calculate the Exponential Moving Average (EMA) of the volume
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Heuristic factor
    heuristics_matrix = mean_abs_price_change * ema_volume
    
    return heuristics_matrix
