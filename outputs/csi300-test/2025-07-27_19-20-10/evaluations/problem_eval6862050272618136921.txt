import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change (ROC) of the closing price over a 14-day period
    roc_price = df['close'].pct_change(periods=14)
    
    # Calculate the Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Heuristic factor
    heuristics_matrix = roc_price * cmf
    
    return heuristics_matrix
