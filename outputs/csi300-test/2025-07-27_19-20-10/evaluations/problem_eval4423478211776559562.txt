import pandas as pd

def heuristics_v2(df):
    def calculate_momentum(df, window=20):
        return df['close'].pct_change(window)
    
    def calculate_volatility(df, window=20):
        return df['close'].pct_change().rolling(window=window).std()
    
    def calculate_volume_trend(df, window=20):
        return df['volume'].rolling(window=window).mean().pct_change()
    
    momentum = calculate_momentum(df)
    volatility = calculate_volatility(df)
    volume_trend = calculate_volume_trend(df)

    # Hypothetical historical performance weights; in practice, these would be derived from backtesting
    weights = {'momentum': 0.5, 'volatility': -0.3, 'volume_trend': 0.2}
    
    # Calculate the heuristic score
    heuristic_score = (momentum * weights['momentum'] + 
                       volatility * weights['volatility'] +
                       volume_trend * weights['volume_trend'])
    
    heuristics_matrix = pd.Series(heuristic_score, index=df.index, name='Alpha Factor')
    return heuristics_matrix
