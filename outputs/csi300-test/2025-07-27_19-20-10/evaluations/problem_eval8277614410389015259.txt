def heuristics_v2(df):
    # Calculate the exponential moving average of price returns
    ema_price_return = df['close'].pct_change().ewm(span=14, adjust=False).mean()
    
    # Calculate the volume-weighted average price (VWAP)
    cum_volume = df['volume'].cumsum()
    cum_volume_price = (df['close'] * df['volume']).cumsum()
    vwap = cum_volume_price / cum_volume
    
    # Heuristic factor
    heuristics_matrix = ema_price_return * vwap
    
    return heuristics_matrix
