import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential moving average of price changes
    ema_price_change = df['close'].diff().ewm(span=14, adjust=False).mean()
    
    # Calculate the RSI of trading volume with an emphasis on recent data
    delta_volume = df['volume'].diff()
    gain = (delta_volume.where(delta_volume > 0, 0)).fillna(0)
    loss = (-delta_volume.where(delta_volume < 0, 0)).fillna(0)
    avg_gain = gain.ewm(com=6, adjust=False).mean()
    avg_loss = loss.ewm(com=6, adjust=False).mean()
    rs = avg_gain / avg_loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * rsi_volume
    
    return heuristics_matrix
