import pandas as pd

def heuristics_v2(df):
    # Calculate the difference in closing prices
    close_diff = df['close'].diff().fillna(0)
    
    # Calculate the Exponential Moving Average (EMA) of the close price difference
    ema_close_diff = close_diff.ewm(span=14, adjust=False).mean()
    
    # Calculate the accumulation/distribution line
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Heuristic factor
    heuristics_matrix = ema_close_diff + adl
    
    return heuristics_matrix
