import pandas as pd

def heuristics_v2(df):
    # Calculate the exponentially weighted moving average of price changes
    ewma_price_change = df['close'].pct_change().ewm(span=14, adjust=False).mean()
    
    # Calculate the rate of change (ROC) of trading volume
    roc_volume = df['volume'].pct_change(periods=14)
    
    # Heuristic factor
    heuristics_matrix = ewma_price_change * roc_volume
    
    return heuristics_matrix
