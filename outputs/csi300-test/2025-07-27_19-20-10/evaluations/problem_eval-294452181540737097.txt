import pandas as pd

def heuristics_v2(df):
    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(delta > 0, 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(delta < 0, 0).rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Calculate Weighted Moving Average (WMA) of the closing price
    weights = pd.Series(range(1, 21))
    wma_close = df['close'].rolling(window=20).apply(lambda prices: (prices * weights).sum() / weights.sum(), raw=True)
    
    # Heuristic factor
    heuristics_matrix = (rsi * mfi) / wma_close
    
    return heuristics_matrix
