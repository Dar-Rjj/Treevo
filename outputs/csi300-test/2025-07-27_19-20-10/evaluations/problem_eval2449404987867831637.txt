import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted moving average of price returns
    wma_price_return = df['close'].pct_change().rolling(window=14).apply(lambda x: (x * pd.Series(range(1, len(x) + 1))).sum() / x.sum(), raw=False)
    
    # Adjusted accumulation/distribution line calculation
    high_low_diff = df['high'] - df['low']
    close_low_diff = df['close'] - df['low']
    close_high_diff = df['high'] - df['close']
    money_flow_multiplier = (close_low_diff - close_high_diff) / high_low_diff
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line_adjusted = (money_flow_volume - money_flow_volume.mean()) / money_flow_volume.std()
    
    # Heuristic factor
    heuristics_matrix = wma_price_return * ad_line_adjusted
    
    return heuristics_matrix
