import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price changes
    ema_price_change = df['close'].diff().ewm(span=14, adjust=False).mean()
    
    # Calculate the Accumulation/Distribution Line (A/D)
    ad_line = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    ad_line = ad_line.cumsum()
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * ad_line
    
    return heuristics_matrix
