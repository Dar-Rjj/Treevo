import pandas as pd

def heuristics_v2(df):
    # Calculate the average of price changes over a 14-day window
    avg_price_change = df['close'].diff().rolling(window=14).mean()
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'].diff() > 0).astype(int)).cumsum()
    
    # Heuristic factor
    heuristics_matrix = avg_price_change * obv
    
    return heuristics_matrix
