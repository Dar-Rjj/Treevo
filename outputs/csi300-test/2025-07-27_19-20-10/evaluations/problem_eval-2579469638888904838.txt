import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the 50-day moving average of close price
    ma_close = df['close'].rolling(window=50).mean()
    
    # Calculate the RSI of the close price
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the logarithm of the trading volume
    log_volume = np.log(df['volume'])
    
    # Heuristic factor
    heuristics_matrix = (ma_close * rsi) / (1 + log_volume)
    
    return heuristics_matrix
