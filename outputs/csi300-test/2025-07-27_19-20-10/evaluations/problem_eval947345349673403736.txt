def heuristics_v2(df):
    # Calculate the 20-day moving average of close prices
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between today's close and the 20-day MA
    diff_close_ma = df['close'] - ma_20
    
    # Calculate the 20-day mean of volumes
    vol_20 = df['volume'].rolling(window=20).mean()
    
    # Determine if today's volume is higher or lower than the 20-day mean
    vol_diff = df['volume'] - vol_20
    vol_adjustment = vol_diff / vol_20  # Relative change in volume
    
    # Adjust the price difference by the volume adjustment
    heuristics_matrix = (diff_close_ma * vol_adjustment)
    
    return heuristics_matrix
