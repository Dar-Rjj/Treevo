import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices
    diff_high_low = df['high'] - df['low']
    
    # Calculate the ratio of current volume to its moving average
    volume_ma_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the average of daily returns
    daily_returns = df['close'].pct_change().fillna(0)
    avg_daily_returns = daily_returns.rolling(window=20).mean()
    
    # Heuristic factor
    heuristics_matrix = (diff_high_low * volume_ma_ratio) / (1 + avg_daily_returns)
    
    return heuristics_matrix
