import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price changes over a 7-day window
    ema_price_change = df['close'].diff().ewm(span=7, adjust=False).mean()
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'].diff() > 0).astype(int)).cumsum()
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * obv
    
    return heuristics_matrix
