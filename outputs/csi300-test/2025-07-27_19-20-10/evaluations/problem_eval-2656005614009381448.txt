import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change (ROC)
    roc = df['close'].pct_change(periods=10)
    
    # Calculate Chaikin Money Flow (CMF)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Calculate Exponentially Weighted Moving Average (EWMA) of the closing price
    ewma_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Heuristic factor
    heuristics_matrix = (roc * cmf) * ewma_close
    
    return heuristics_matrix
