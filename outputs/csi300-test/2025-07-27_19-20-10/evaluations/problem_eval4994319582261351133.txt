def heuristics_v2(df):
    # Calculate daily price change
    df['price_change'] = df['close'] - df['open']
    # Compute the weight as the ratio of the day's volume to the rolling sum of the last 5 days' volumes
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=5).sum()
    # The heuristic is the product of the price change and the volume weight
    heuristics_matrix = (df['price_change'] * df['volume_weight']).dropna()
    return heuristics_matrix
