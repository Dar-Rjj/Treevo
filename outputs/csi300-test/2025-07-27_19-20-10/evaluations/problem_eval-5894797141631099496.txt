import pandas as pd

def heuristics_v2(df):
    # Calculate the EMA of price changes
    ema_price_change = df['close'].diff().ewm(span=21, adjust=False).mean()
    
    # Calculate the RSI of trading volume
    delta_volume = df['volume'].diff()
    gain = (delta_volume.where(delta_volume > 0, 0)).fillna(0)
    loss = (-delta_volume.where(delta_volume < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=21).mean()
    avg_loss = loss.rolling(window=21).mean()
    rs = avg_gain / avg_loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * rsi_volume
    
    return heuristics_matrix
