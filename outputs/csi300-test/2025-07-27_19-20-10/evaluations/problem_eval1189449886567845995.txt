import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price-to-volume ratio changes
    price_volume_ratio = df['close'] / df['volume']
    ema_pv_change = price_volume_ratio.diff().ewm(span=14, adjust=False).mean()
    
    # Calculate the Relative Strength Index (RSI) of closing prices
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_close = 100 - (100 / (1 + rs))
    
    # Heuristic factor
    heuristics_matrix = ema_pv_change * rsi_close
    
    return heuristics_matrix
