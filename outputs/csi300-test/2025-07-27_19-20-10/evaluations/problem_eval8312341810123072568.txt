import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling standard deviation of price returns as a measure of volatility
    volatility = df['close'].pct_change().rolling(window=14).std()
    
    # Calculate the rate of change (ROC) of trading volume
    roc_volume = df['volume'].pct_change(periods=14)
    
    # Heuristic factor
    heuristics_matrix = volatility * roc_volume
    
    return heuristics_matrix
