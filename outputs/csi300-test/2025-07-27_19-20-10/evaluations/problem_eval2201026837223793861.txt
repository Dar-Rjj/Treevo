import pandas as pd

def heuristics_v2(df):
    # Calculate the standard deviation of high and low prices
    std_high_low = df[['high', 'low']].rolling(window=14).std().mean(axis=1)
    
    # Calculate the accumulation/distribution line
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Heuristic factor
    heuristics_matrix = std_high_low / adl
    
    return heuristics_matrix
