import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) of closing prices
    roc_price = df['close'].pct_change(periods=14) * 100
    
    # Calculate the Volume-Price Trend (VPT)
    vpt = ((df['close'] - df['close'].shift(1)) / df['close'].shift(1)) * df['volume']
    vpt = vpt.cumsum()
    
    # Heuristic factor
    heuristics_matrix = roc_price * vpt
    
    return heuristics_matrix
