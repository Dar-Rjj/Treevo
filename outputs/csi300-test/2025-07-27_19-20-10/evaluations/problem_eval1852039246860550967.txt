import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the highest and lowest prices over a 14-day window
    price_diff = df['high'].rolling(window=14).max() - df['low'].rolling(window=14).min()
    
    # Calculate the ratio of current volume to its moving average
    volume_ma_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Heuristic factor
    heuristics_matrix = price_diff * volume_ma_ratio
    
    return heuristics_matrix
