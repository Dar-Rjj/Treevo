import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price returns with a shorter span
    ema_price_return = df['close'].pct_change().ewm(span=7, adjust=False).mean()
    
    # Calculate the weighted accumulation/distribution line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line = (money_flow_volume * 2 + money_flow_volume.shift(1)).cumsum()
    
    # Heuristic factor
    heuristics_matrix = ema_price_return * ad_line
    
    return heuristics_matrix
