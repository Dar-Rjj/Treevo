import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of volume changes
    ema_volume_change = df['volume'].diff().ewm(span=14, adjust=False).mean()
    
    # Calculate the Relative Strength Index (RSI) of price
    price_gain = df['close'].diff().clip(lower=0)
    price_loss = -df['close'].diff().clip(upper=0)
    avg_gain = price_gain.rolling(window=14).mean()
    avg_loss = price_loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_price = 100 - (100 / (1 + rs))
    
    # Heuristic factor
    heuristics_matrix = ema_volume_change * rsi_price
    
    return heuristics_matrix
