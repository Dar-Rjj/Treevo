import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change of the closing price
    roc_price = df['close'].pct_change(periods=14)
    
    # Calculate the Accumulation/Distribution Line (ADL)
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    adl = (money_flow_multiplier * df['volume']).cumsum()
    
    # Heuristic factor
    heuristics_matrix = roc_price * adl
    
    return heuristics_matrix
