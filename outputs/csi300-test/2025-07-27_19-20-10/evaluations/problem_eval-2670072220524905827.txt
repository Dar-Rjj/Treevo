import pandas as pd

def heuristics_v2(df):
    # Calculate the rate of change in daily high price
    roc_high = df['high'].pct_change().fillna(0)
    
    # Calculate the ratio of current volume to its 10-day moving average
    volume_ma_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the mean absolute deviation of daily returns
    daily_returns = df['close'].pct_change().fillna(0)
    mad_daily_returns = daily_returns.rolling(window=20).apply(lambda x: (x - x.mean()).abs().mean())
    
    # Heuristic factor
    heuristics_matrix = (roc_high * volume_ma_ratio) / (1 + mad_daily_returns)
    
    return heuristics_matrix
