import pandas as pd

def heuristics_v2(df):
    # Calculate the adjusted exponential moving average of price returns
    ema_price_return = df['close'].pct_change().ewm(span=21, adjust=False).mean()
    
    # Adjusted calculation for the accumulation/distribution line with a weighted money flow multiplier
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * 0.5
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line = money_flow_volume.cumsum()
    
    # Heuristic factor
    heuristics_matrix = ema_price_return * ad_line
    
    return heuristics_matrix
