import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the exponential moving average of price changes
    ema_price_change = df['close'].diff().ewm(span=14, adjust=False).mean()
    
    # Calculate the Money Flow Index
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = np.where(typical_price > typical_price.shift(1), money_flow, 0)
    negative_money_flow = np.where(typical_price < typical_price.shift(1), money_flow, 0)
    mfi_ratio = np.sum(positive_money_flow, axis=0, keepdims=True) / np.sum(negative_money_flow, axis=0, keepdims=True)
    mfi = 100 - (100 / (1 + mfi_ratio))
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * mfi.squeeze()
    
    return heuristics_matrix
