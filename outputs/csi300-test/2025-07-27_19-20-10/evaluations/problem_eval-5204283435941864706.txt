import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Money Flow Index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    pmf_sum = positive_money_flow.rolling(window=14).sum()
    nmf_sum = negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (pmf_sum / nmf_sum)))
    
    # Calculate Rate of Change (ROC) of the closing price
    roc = df['close'].pct_change(periods=14) * 100
    
    # Calculate Weighted Moving Average (WMA) of the trading amount
    weights = np.arange(1, 15)
    wma_amount = df['amount'].rolling(window=14).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
    
    # Heuristic factor
    heuristics_matrix = (mfi * roc) / wma_amount
    
    return heuristics_matrix
