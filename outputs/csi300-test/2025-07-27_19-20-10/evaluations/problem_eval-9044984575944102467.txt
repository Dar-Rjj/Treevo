import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) of price changes
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_price_change = 100 - (100 / (1 + rs))
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * ((df['close'] > df['close'].shift(1)) - (df['close'] < df['close'].shift(1)))).cumsum()
    
    # Heuristic factor
    heuristics_matrix = rsi_price_change * obv
    
    return heuristics_matrix
