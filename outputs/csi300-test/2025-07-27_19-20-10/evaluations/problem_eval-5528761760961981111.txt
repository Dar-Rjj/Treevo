import pandas as pd

def heuristics_v2(df):
    # Calculate the Exponential Moving Average (EMA) for the closing price
    EMA = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the percentage change in volume
    vol_change = df['volume'].pct_change(periods=1)
    
    # Calculate the On-Balance Volume (OBV)
    OBV = (df['close'] > df['close'].shift()).astype(int) * df['volume']
    OBV = OBV.cumsum()
    
    # Construct the heuristic matrix
    heuristics_matrix = EMA + vol_change.fillna(0) + OBV
    return heuristics_matrix
