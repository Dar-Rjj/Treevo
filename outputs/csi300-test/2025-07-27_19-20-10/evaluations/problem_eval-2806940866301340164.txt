def heuristics_v2(df):
    # Calculate Rate of Change (ROC) for close prices over 14 days
    roc_close = df['close'].pct_change(periods=14).fillna(0)
    
    # Calculate 20-day moving average of volume
    ma_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate difference between current volume and its 20-day MA
    vol_diff = df['volume'] - ma_volume_20
    
    # Heuristic factor
    heuristics_matrix = roc_close * vol_diff
    
    return heuristics_matrix
