import pandas as pd

def heuristics_v2(df):
    # Calculate logarithmic difference in close price
    log_diff_close = (df['close'] - df['close'].shift(1)).fillna(0)
    
    # Calculate the ratio of current volume to its exponential moving average
    volume_ema_ratio = df['volume'] / df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the mean absolute deviation of daily returns
    daily_returns = df['close'].pct_change().fillna(0)
    mad_daily_returns = daily_returns.rolling(window=20).apply(lambda x: (x - x.mean()).abs().mean(), raw=True)
    
    # Heuristic factor
    heuristics_matrix = (log_diff_close * volume_ema_ratio) / (1 + mad_daily_returns)
    
    return heuristics_matrix
