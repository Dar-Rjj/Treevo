import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted moving average of price changes
    wma_price_change = df['close'].diff().rolling(window=14).apply(lambda x: (x * range(1, 15)).sum() / sum(range(1, 15)), raw=False)
    
    # Calculate the Relative Strength Index (RSI) of trading volume
    delta = df['volume'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Heuristic factor
    heuristics_matrix = wma_price_change * rsi_volume
    
    return heuristics_matrix
