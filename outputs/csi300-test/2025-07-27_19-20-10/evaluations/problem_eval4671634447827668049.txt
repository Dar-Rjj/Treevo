import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price changes
    ema_price_change = df['close'].diff().ewm(span=14, adjust=False).mean()
    
    # Calculate the RSI of trading volume
    delta_volume = df['volume'].diff()
    gain = (delta_volume.where(delta_volume > 0, 0)).fillna(0)
    loss = (-delta_volume.where(delta_volume < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi_volume = 100 - (100 / (1 + rs))
    
    # Normalize RSI of trading volume
    rsi_volume_normalized = (rsi_volume - rsi_volume.min()) / (rsi_volume.max() - rsi_volume.min())
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * rsi_volume_normalized
    
    return heuristics_matrix
