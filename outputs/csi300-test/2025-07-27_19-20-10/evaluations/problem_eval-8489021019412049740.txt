def heuristics_v2(df):
    # Calculate RSI with a different smoothing (SMA over 14 periods)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean().replace(0, np.nan)
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate EMA of volume
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Heuristic factor with a weighting factor
    weight = 0.6  # Adjusting the influence of RSI vs Volume EMA
    heuristics_matrix = (rsi * weight) + (ema_volume * (1 - weight))
    
    return heuristics_matrix
