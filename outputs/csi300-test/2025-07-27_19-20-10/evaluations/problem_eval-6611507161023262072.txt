import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price changes
    ema_price_change = df['close'].diff().ewm(span=14, adjust=False).mean()
    
    # Adjusted Money Flow Index (AMFI) incorporating trading amount
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    adjusted_money_flow = raw_money_flow * (df['amount'] / df['amount'].mean())  # Weighted by average trading amount
    positive_flow = adjusted_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = adjusted_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    amfi_volume = 100 - (100 / (1 + money_flow_ratio))
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * amfi_volume
    
    return heuristics_matrix
