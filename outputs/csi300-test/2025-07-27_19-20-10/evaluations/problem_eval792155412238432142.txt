import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) of closing prices
    roc_price = df['close'].pct_change(periods=14)
    
    # Calculate the Exponential Moving Average (EMA) of trading volumes
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Heuristic factor
    heuristics_matrix = roc_price * ema_volume
    
    return heuristics_matrix
