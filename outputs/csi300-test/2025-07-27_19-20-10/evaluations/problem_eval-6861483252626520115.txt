import pandas as pd

def heuristics_v2(df):
    # Calculate the 14-day EMA of closing prices
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the 50-day SMA of closing prices
    sma_close = df['close'].rolling(window=50).mean()
    
    # Compute the difference between EMA and SMA
    diff_ema_sma = ema_close - sma_close
    
    # Calculate the positive and negative price changes
    delta_price = df['close'].diff().fillna(0)
    positive_changes = (delta_price > 0).astype(int)
    negative_changes = (delta_price < 0).astype(int)
    
    # Calculate the ratio of positive to negative price changes
    ratio_pos_neg = positive_changes.rolling(window=14).sum() / negative_changes.rolling(window=14).sum().replace(0, 1)
    
    # Heuristic factor
    heuristics_matrix = diff_ema_sma * ratio_pos_neg
    
    return heuristics_matrix
