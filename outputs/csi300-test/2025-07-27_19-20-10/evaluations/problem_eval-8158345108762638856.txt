import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted moving average of price returns
    wma_price_return = df['close'].pct_change().rolling(window=14).apply(lambda x: (x * range(1, 15)).sum() / sum(range(1, 15)), raw=True)
    
    # Calculate the Chaikin Oscillator
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line = money_flow_volume.cumsum()
    chaikin_oscillator = ad_line.ewm(span=3, adjust=False).mean() - ad_line.ewm(span=10, adjust=False).mean()
    
    # Heuristic factor
    heuristics_matrix = wma_price_return * chaikin_oscillator
    
    return heuristics_matrix
