import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price changes
    ema_price_change = df['close'].diff().ewm(span=14, adjust=False).mean()
    
    # Calculate the cumulative sum of trading volume
    cum_volume = df['volume'].cumsum()
    
    # Heuristic factor
    heuristics_matrix = ema_price_change * cum_volume
    
    return heuristics_matrix
