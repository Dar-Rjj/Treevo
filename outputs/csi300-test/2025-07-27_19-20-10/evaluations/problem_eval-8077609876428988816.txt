def heuristics_v2(df):
    # Calculate the standard deviation of daily returns over a 21-day window
    std_daily_returns = df['close'].pct_change().rolling(window=21).std()
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'].diff() > 0).astype(int)).cumsum()
    
    # Calculate the rate of change in OBV over a 21-day window
    roc_obv = obv.pct_change(periods=21)
    
    # Heuristic factor
    heuristics_matrix = std_daily_returns * roc_obv
    
    return heuristics_matrix
