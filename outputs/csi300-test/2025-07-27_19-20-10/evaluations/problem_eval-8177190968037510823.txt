import pandas as pd

def heuristics_v2(df):
    # Calculate the Relative Strength Index (RSI) for the closing price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    RS = gain / loss
    RSI = 100 - (100 / (1 + RS))
    
    # Calculate the Rate of Change (ROC) for the trading amount
    ROC = df['amount'].pct_change(periods=12)
    
    # Calculate the Accumulation/Distribution (A/D) Line
    ad_line = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    ad_line *= df['volume']
    AD = ad_line.cumsum()
    
    # Construct the heuristic matrix
    heuristics_matrix = RSI + ROC.fillna(0) + AD
    return heuristics_matrix
