import pandas as pd

def heuristics_v2(df):
    # Calculate the Exponential Moving Average (EMA) of the close prices over a 30-day window
    ema_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the Rate of Change (ROC) of volume
    roc_volume = df['volume'].pct_change(periods=1)
    
    # Heuristic factor
    heuristics_matrix = ema_close * roc_volume
    
    return heuristics_matrix
