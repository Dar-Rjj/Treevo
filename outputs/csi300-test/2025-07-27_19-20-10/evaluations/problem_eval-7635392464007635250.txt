import pandas as pd

def heuristics_v2(df):
    # Calculate the mean absolute deviation (MAD) of price changes over a 14-day window
    mad_price_change = df['close'].diff().rolling(window=14).apply(lambda x: x.abs().mean(), raw=True)
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'].diff() > 0).astype(int)).cumsum()
    
    # Heuristic factor
    heuristics_matrix = mad_price_change * obv
    
    return heuristics_matrix
