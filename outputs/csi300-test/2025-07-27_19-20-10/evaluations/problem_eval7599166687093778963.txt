import pandas as pd

def heuristics_v2(df):
    # Calculate the exponential moving average of price returns
    ema_price_return = df['close'].pct_change().ewm(span=14, adjust=False).mean()
    
    # Calculate the modified accumulation/distribution line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line_modified = money_flow_volume.rolling(window=10).sum()
    
    # Heuristic factor
    heuristics_matrix = ema_price_return * ad_line_modified
    
    return heuristics_matrix
