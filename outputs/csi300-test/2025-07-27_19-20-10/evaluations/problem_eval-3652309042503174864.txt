import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between the current close price and its 20-day moving average
    close_ma_diff = df['close'] - df['close'].rolling(window=20).mean()
    
    # Calculate the ratio of current volume to its cumulative sum
    cum_volume_ratio = df['volume'] / df['volume'].cumsum()
    
    # Heuristic factor
    heuristics_matrix = close_ma_diff * cum_volume_ratio
    
    return heuristics_matrix
