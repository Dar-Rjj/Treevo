import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate ROC
    roc = df['close'].pct_change(periods=14)
    
    # Calculate EMA of volume
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Heuristic factor
    heuristics_matrix = roc * ema_volume
    
    return heuristics_matrix
