import pandas as pd

def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) of the closing price
    roc = df['close'].pct_change(periods=14) * 100
    
    # Calculate the Negative Volume Index (NVI)
    nvi = pd.Series(1, index=df.index)
    nvi[df['volume'] > df['volume'].shift()] = 1
    nvi[df['volume'] <= df['volume'].shift()] = (df['close'] / df['close'].shift()) * nvi.shift()
    nvi.ffill(inplace=True)
    
    # Heuristic factor
    heuristics_matrix = roc * nvi
    
    return heuristics_matrix
