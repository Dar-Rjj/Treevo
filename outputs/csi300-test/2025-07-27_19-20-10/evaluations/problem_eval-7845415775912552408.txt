import pandas as pd

def heuristics_v2(df):
    momentum = df['close'].pct_change()
    volatility = df['close'].pct_change().rolling(window=5).std()
    volume_change = df['volume'].pct_change()
    combined_factor = (momentum * 0.4) + (volatility * 0.3) + (volume_change * 0.3)
    heuristics_matrix = combined_factor
    return heuristics_matrix
