def heuristics_v2(df):
    # Calculate the mean absolute price change over a 14-day window
    mean_abs_price_change = df['close'].diff().abs().rolling(window=14).mean()
    
    # Calculate the average volume over a 14-day window
    avg_volume = df['volume'].rolling(window=14).mean()
    
    # Calculate the On-Balance Volume (OBV)
    obv = (df['volume'] * (df['close'].diff() > 0).astype(int)).cumsum()
    
    # Adjust OBV by the average volume
    obv_adjusted = obv / avg_volume
    
    # Heuristic factor
    heuristics_matrix = mean_abs_price_change * obv_adjusted
    
    return heuristics_matrix
