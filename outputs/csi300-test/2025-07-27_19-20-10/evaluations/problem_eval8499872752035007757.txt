import pandas as pd

def heuristics_v2(df):
    # Calculate Rate of Change (ROC) of the closing price
    roc = df['close'].pct_change(periods=14)
    
    # Calculate On-Balance Volume (OBV)
    obv = (df['close'] > df['close'].shift()).astype(int) * df['volume'] - (df['close'] < df['close'].shift()).astype(int) * df['volume']
    obv = obv.cumsum()
    
    # Heuristic factor
    heuristics_matrix = roc * obv
    
    return heuristics_matrix
