import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted moving average of price returns
    wma_price_return = df['close'].pct_change().rolling(window=14).apply(lambda x: (x * pd.Series(range(1, len(x) + 1))).sum() / pd.Series(range(1, len(x) + 1)).sum(), raw=False)
    
    # Calculate the Chaikin Money Flow
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Heuristic factor
    heuristics_matrix = wma_price_return * cmf
    
    return heuristics_matrix
