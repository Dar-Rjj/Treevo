```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Price Volatility
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Volume Volatility
    volume_volatility = df['volume'].rolling(window=20).std()
    
    # Corrected Intraday Momentum
    corrected_intraday_momentum = volume_weighted_intraday_return - price_volatility - volume_volatility
    
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Composite Alpha Factor
    volume_weighted_close_open_difference = (df['close'] - df['open']) * df['volume']
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    subtract_ema_from_volume_weighted_close = df['close'] * df['volume'] - ema_close
    medium_term_reversal = df['close'].pct_change().rolling(window=10).sum()
    composite_alpha_factor = volume_weighted_close_open_difference + subtract_ema_from_volume_weighted_close + medium_term_reversal
    
    # Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = np.maximum(df['high'], df.shift(1)['close']) - np.minimum(df['low'], df.shift(1)['close'])
    adjusted_intraday_return = intraday_log_return / true_range
    volume_change = df['volume'] - df.shift(1)['volume']
    adjusted_return = adjusted_intraday_return + volume_change
    
    # Final Integration
    combined_factors = high_low_spread_ratio + corrected_intraday_momentum + composite_alpha_factor
    final_factor = combined_factors + adjusted_return
    
    # Trend Adjustment
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    trend_difference = short_term_trend - long_term_trend
    final_factor = final_factor.where(trend_difference > 0, final_factor * 1.5)
    final_factor = final_factor.where(trend_difference <= 0, final_factor * 0.5)
    
    return final_factor
```
