```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Volume-Weighted Close-Open Difference
    vol_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + vol_weighted_close_open_diff
    
    # Composite Alpha Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    vol_weighted_close = df['close'] * df['volume']
    composite_alpha_factor = vol_weighted_close - ema_close
    
    # Medium-term Reversal
    daily_returns = df['close'].pct_change()
    medium_term_reversal = daily_returns.rolling(window=10).sum()
    
    # Incorporate into Composite Alpha Factor
    composite_alpha_factor += medium_term_reversal
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    true_range = np.max([df['high'] - df['low'], np.abs(df['high'] - df.shift(1)['close']), np.abs(df['low'] - df.shift(1)['close'])], axis=0)
    adjusted_return = intraday_log_return / true_range
    
    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=21).mean()
    volume_change = df['volume'].pct_change()
    volume_impact = volume_change * (df['volume'] / avg_volume)
    
    # Add Volume Impact to the Adjusted Return
    adjusted_return += volume_impact
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha_factor + adjusted_return
    
    return final_factor
```
