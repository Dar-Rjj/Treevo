```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_weighted_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])

    # Incorporate Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread

    # Adjust for Price and Volume Volatility
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].rolling(window=20).std()
    corrected_intraday_momentum = volume_weighted_spreads - price_volatility - volume_volatility

    # Apply Trend Adjustment
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend

    trend_adjustment = np.where(trend_difference > 0, 1.5, 0.5)
    trend_adjusted_factor = corrected_intraday_momentum * trend_adjustment

    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    sqrt_volume_ma = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = (log_return / high_low_spread) + (volume_weighted_price_change / sqrt_volume_ma)

    # Intraday Momentum Component
    twenty_day_momentum = (df['close'] - df['close'].shift(20)) / df['close'].rolling(window=20).std()
    five_day_momentum = (df['close'] - df['close'].shift(5)) / df['close'].rolling(window=5).std()

    # Introduce Intraday Volume-Ratio Factor
    intraday_volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    adjusted_intraday_return = intraday_return * intraday_volume_ratio

    # Incorporate Intraday High-Low Momentum
    high_low_momentum = (df['high'] - df['low']) / df['open']
    scaled_high_low_momentum = high_low_momentum / (df['open'] * (df['high'] - df['low']).rolling(window=10).std())

    # Final Alpha Combination
    final_alpha = (volume_weighted_spreads + 
                   scaled_log_return + 
                   trend_adjusted_factor + 
                   vami + 
                   twenty_day_momentum + 
                   five_day_momentum + 
                   adjusted_intraday_return + 
                   scaled_high_low_momentum)

    return final_alpha
```
