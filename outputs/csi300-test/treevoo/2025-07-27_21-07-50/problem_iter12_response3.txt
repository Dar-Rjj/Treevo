```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Momentum
    short_term_period = 5
    df['Return'] = df['close'].pct_change(short_term_period)
    df['AvgVolume'] = df['volume'].rolling(window=short_term_period).mean()
    df['VolumeAdjustedMomentum'] = df['Return'] * df['AvgVolume']

    # Calculate Intraday Factors
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    df['IntradayRange'] = (df['high'] - df['low']) / df['low']
    df['DailyReturn'] = df['close'].pct_change()
    df['AbsDailyReturn'] = df['DailyReturn'].abs()
    df['IntradayVolatility'] = df['AbsDailyReturn'].rolling(window=20).sum() / 20

    # Calculate Price Trend Factors
    long_term_period = 20
    df['LongTermMA'] = df['close'].rolling(window=long_term_period).mean()
    df['ShortTermMA'] = df['close'].rolling(window=short_term_period).mean()

    # Combine Adjusted Momentum, Intraday Factors, and Price Trends
    df['Factor1'] = df['VolumeAdjustedMomentum'] - df['IntradayVolatility']
    df['Factor2'] = df['IntradayReturn'] + df['IntradayRange']
    df['Factor3'] = df['ShortTermMA'] - df['LongTermMA']
    df['CombinedFactor'] = df['Factor1'] + df['Factor2'] + df['Factor3']

    # Adjust by Volume Thrust
    df['VolumeThrust'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['FinalFactor'] = df['CombinedFactor'] * df['VolumeThrust']

    return df['FinalFactor'].dropna()
```
