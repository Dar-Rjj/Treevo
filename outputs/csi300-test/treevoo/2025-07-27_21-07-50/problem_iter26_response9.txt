```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average (SMA) of the close price over 10 and 30 days
    sma_10 = df['close'].rolling(window=10).mean()
    sma_30 = df['close'].rolling(window=30).mean()

    # Calculate the exponential moving average (EMA) of the close price over 10 and 30 days
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()

    # Calculate the ratio of SMA to EMA for the 10-day and 30-day period
    sma_ema_ratio_10 = sma_10 / ema_10
    sma_ema_ratio_30 = sma_30 / ema_30

    # Calculate the money flow index (MFI) over a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    money_ratio = positive_money_flow_sum / negative_money_flow_sum
    mfi = 100 - (100 / (1 + money_ratio))

    # Combine the factors into a single alpha factor
    alpha_factor = (sma_ema_ratio_10 - sma_ema_ratio_30) + mfi

    return alpha_factor
```
