```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Average (SMA) of Close prices
    sma_period = 21
    df['SMA'] = df['close'].rolling(window=sma_period).mean()
    
    # Calculate Volume-Weighted Close Price
    df['volume_weighted_close'] = df['close'] * df['volume']
    volume_weighted_close = df['volume_weighted_close'].rolling(window=sma_period).sum() / df['volume'].rolling(window=sma_period).sum()
    
    # Calculate Short-Term Reversal
    n_days = 5
    df['daily_returns'] = (df['open'] - df['close']) / df['open']
    short_term_reversal = df['daily_returns'].rolling(window=n_days).sum()
    
    # Composite Alpha Factor
    df['alpha_factor'] = (volume_weighted_close - df['SMA']) + short_term_reversal
    
    return df['alpha_factor']
```
