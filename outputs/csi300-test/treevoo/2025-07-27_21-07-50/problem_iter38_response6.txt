import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spreads = (high_low_spread + open_close_spread) * df['volume']

    # Volume-Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / ema_volume_20

    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']

    # Short-Term Reversal (5-day sum of returns)
    short_term_reversal = (df['close'] / df['close'].shift(1) - 1).rolling(window=5).sum()

    # Long-Term Trend (60-day SMA of Volume-Weighted Close)
    sma_volume_weighted_close_60 = volume_weighted_close.rolling(window=60).mean()

    # Composite Alpha Factor
    composite_alpha_factor = (volume_weighted_close - sma_volume_weighted_close_60) + vami + short_term_reversal + combined_spreads

    # Additional Factors
    high_low_spread_ratio = (high_low_spread) / (df['high'] + df['low'])
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    integrated_both_factors = high_low_spread_ratio + volume_weighted_close_open_diff

    # EMA of Close Prices (21-day EMA)
    ema_close_21 = df['close'].ewm(span=21, adjust=False).mean()

    # Medium-Term Reversal (aggregate past 10-day returns)
    medium_term_reversal = (df['close'] / df['open'] - 1).rolling(window=10).sum()

    # Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = np.maximum(df['high'], df['close'].shift(1)) - np.minimum(df['low'], df['close'].shift(1))
    scaled_log_return = intraday_log_return * true_range
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_return = scaled_log_return * volume_change

    # Final Integration
    final_integration = (integrated_both_factors + composite_alpha_factor) + adjusted_return

    return final_integration
