def heuristics_v2(df):
    # Calculate High-Low Spread Intensity
    high_low_spread_intensity = (df['high'] - df['low']) / df['close']
    
    # Calculate Volume-Weighted Average Price Change
    volume_weighted_avg_price_change = ((df['close'] + df['open']) / 2) * df['volume']
    
    # Evaluate Price Trend Reversal Indicator
    price_trend_reversal = (df['close'] - df['open']) / df['open']
    price_trend_reversal_indicator = price_trend_reversal.apply(lambda x: 1 if x > 0 else -1)
    
    # Integrate All Factors
    integrated_factor = (high_low_spread_intensity * volume_weighted_avg_price_change) + \
                        (price_trend_reversal_indicator * volume_weighted_avg_price_change)
    
    return integrated_factor
