```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the momentum factor
    sma_14 = df['close'].rolling(window=14).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    momentum = sma_14 - sma_50
    
    # Assess the volatility using historical price data
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Incorporate trading volume into the analysis
    vol_7 = df['volume'].rolling(window=7).mean()
    vol_21 = df['volume'].rolling(window=21).mean()
    volume_ratio = vol_7 / vol_21
    
    # Evaluate the money flow index
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum()
    mfr = positive_money_flow / negative_money_flow
    mfi = 100 - (100 / (1 + mfr))
    
    # Analyze the relative strength
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    relative_strength = ema_14 / ema_200
    
    # Consider the open interest in futures and options
    open_interest = df['amount'].diff().rolling(window=30).sum()
    
    # Evaluate the market sentiment
    market_sentiment = (df['close'] - df['open']).rolling(window=10).sum()
    
    # Incorporate the market breadth
    advance_decline_line = (df['close'] > df['close'].shift(1)).astype(int) - (df['close'] < df['close'].shift(1)).astype(int)
    market_breadth = advance_decline_line.rolling(window=20).sum()
    
    # Integrate intraday price movement
    intraday_return = (df['high'] - df['low']) / df['low']
    intraday_ma_10 = intraday_return.rolling(window=10).mean()
    
    # Assess the directional movement
    positive_directional_movement = (df['high'] - df['high'].shift(1)).where(lambda x: x > (df['low'].shift(1) - df['low']), 0)
    negative_directional_movement = (df['low'].shift(1) - df['low']).where(lambda x: x > (df['high'] - df['high'].shift(1)), 0)
    dmi = positive_directional_movement.ewm(span=14, adjust=False).mean() - negative_directional_movement.ewm(span=14, adjust=False).mean()
    
    # Evaluate the price action
    close_pct_change_5 = df['close'].pct_change(periods=5)
    close_ma_20 = close_pct_change_5.rolling(window=20).mean()
    
    # Combine all the factors
    combined_factors = (
        0.3 * momentum +
        0.2 * (-volatility) +
        0.1 * volume_ratio +
        0.1 * mfi +
        0.05 * relative_strength +
        0.05 * open_interest +
        0.05 * market_sentiment +
        0.05 * market_breadth +
        0.05 * intraday_ma_10 +
        0.025 * dmi +
        0.025 * close_ma_20
    )
    
    # Apply a logistic function to the weighted sum to create a bounded output
    alpha_factor = 1 / (1 + np.exp(-combined_factors))
    
    return alpha_factor
```
