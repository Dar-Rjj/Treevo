```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A series of factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3.0
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the EMA of the typical price
    ema_typical_price = typical_price.ewm(span=14, adjust=False).mean()
    
    # Calculate the EMA of the money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the heuristic factor
    factor = (ema_typical_price - df['close']) / df['close'] * (money_flow / ema_money_flow)
    
    return factor
```
