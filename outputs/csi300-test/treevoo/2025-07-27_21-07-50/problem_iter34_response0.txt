import pandas as pd

def heuristics_v2(df):
    # Calculate the 5-day rate of change (momentum)
    df['roc_5'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate the 20-day rate of change (longer-term trend)
    df['roc_20'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate the 20-day average daily range (volatility and mean reversion potential)
    df['avg_range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Calculate the 20-day Volume-Weighted Average Price (VWAP)
    vwap_sum_close_volume = (df['close'] * df['volume']).rolling(window=20).sum()
    vwap_total_volume = df['volume'].rolling(window=20).sum()
    df['vwap_20'] = vwap_sum_close_volume / vwap_total_volume
    
    # Calculate the 20-day average daily amount traded (liquidity)
    df['avg_amount_20'] = df['amount'].rolling(window=20).mean()
    
    # Combine the factors into a single alpha factor
    # This is a simple combination, you can adjust the weights or use more sophisticated methods
    df['alpha_factor'] = (
        df['roc_5'] + 
        df['roc_20'] - 
        0.1 * df['avg_range_20'] + 
        0.2 * (df['vwap_20'] - df['close']) / df['close'] + 
        0.1 * df['avg_amount_20']
    )
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
