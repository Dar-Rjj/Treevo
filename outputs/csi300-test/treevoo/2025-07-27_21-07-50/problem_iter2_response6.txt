```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_lookback = 5
    df['short_term_momentum'] = df['close'].pct_change(periods=short_lookback)
    
    # Integrate Volume into Short-Term Momentum
    df['volume_adjusted_short_momentum'] = df['short_term_momentum'] * df['volume']
    
    # Calculate Long-Term Momentum
    long_lookback = 20
    df['long_term_momentum'] = df['close'].pct_change(periods=long_lookback)
    
    # Compute Price Momentum
    df['price_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Price Volatility
    df['price_volatility'] = df['close'].pct_change().rolling(window=short_lookback).std()
    
    # Combine Adjusted Momentum and Volatility
    df['corrected_momentum'] = df['volume_adjusted_short_momentum'] - df['price_volatility']
    
    # Volume Weighted Average Price (VWAP) Oscillator
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['sma_vwap_20'] = df['vwap'].rolling(window=long_lookback).mean()
    df['vwap_oscillator'] = df['vwap'] - df['sma_vwap_20']
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['corrected_momentum'] * df['vwap_oscillator']
    
    return df['alpha_factor']
```
