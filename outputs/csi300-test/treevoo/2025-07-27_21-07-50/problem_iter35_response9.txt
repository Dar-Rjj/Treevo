```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A series indexed by (date) representing the factor values.
    """
    # Calculate the midpoint of the high and low prices
    mid_price = (df['high'] + df['low']) / 2
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change in the mid-price
    mid_price_change = mid_price.pct_change()
    
    # Calculate the volume weighted average price
    vwap = (df['amount'] / df['volume']).fillna(0)
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=5)
    
    # Alpha factor: combination of mid-price change, price range, and momentum
    alpha_factor = mid_price_change * price_range * momentum + vwap
    
    return alpha_factor
```
