```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Intraday Open-Close Difference
    open_close_diff = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio * open_close_diff
    
    # Calculate Integrated Momentum
    lookback_period = 5
    close_returns = df['close'].pct_change(periods=lookback_period)
    volume_adjusted_momentum = close_returns * df['volume'].rolling(window=lookback_period).mean()
    
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    daily_returns = df['close'].pct_change()
    absolute_daily_returns = abs(daily_returns)
    intraday_volatility = absolute_daily_returns.rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    volume_thrust = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Adjusted Momentum and Intraday Factors
    combined_factors = (volume_adjusted_momentum - intraday_volatility + 
                        intraday_return + intraday_range)
    
    # Final Factor
    final_factor = (integrated_factor + combined_factors) * integrated_factor
    
    # Final Adjusted Factor
    final_adjusted_factor = final_factor * volume_thrust
    
    return final_adjusted_factor
```
