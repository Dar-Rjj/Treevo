```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5):
    # Calculate the difference between the close price of day t and the open price of day t
    df['close_open_diff'] = df['close'] - df['open']
    
    # Calculate the percentage change from the open price to the close price of day t
    df['close_open_pct_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the range (high - low) of day t
    df['high_low_range'] = df['high'] - df['low']
    
    # Determine the position of the close price relative to the high and low of day t
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Compute the ratio of the amount to the volume for day t
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Evaluate the change in volume from day t-1 to day t
    df['volume_change'] = df['volume'].diff()
    
    # Consider the trend in volume over the past n days (e.g., a moving average of volume)
    df['volume_trend'] = df['volume'].rolling(window=n).mean()
    
    # Look at the cumulative sum of daily returns over the past n days
    df['cumulative_return'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=n).sum()
    
    # Analyze the momentum based on the highest and lowest prices over the past n days
    df['momentum_high'] = df['high'].rolling(window=n).max() - df['close']
    df['momentum_low'] = df['close'] - df['low'].rolling(window=n).min()
    
    # Create a factor that reflects the volatility, such as the standard deviation of returns over the past n days
    df['volatility'] = (df['close'] / df['close'].shift(1) - 1).rolling(window=n).std()
    
    # Combine the open-close difference with the high-low range to create a composite price movement indicator
    df['price_movement'] = df['close_open_diff'] + df['high_low_range']
    
    # Integrate the volume changes with the price movements to form a more comprehensive market activity gauge
    df['market_activity'] = df['close_open_diff'] * df['volume_change']
    
    # Create a novel alpha factor
    df['alpha_factor'] = (
        df['close_open_pct_change'] +
        df['close_position'] +
        df['amount_volume_ratio'] +
        df['volume_trend'] +
        df['cumulative_return'] +
        df['momentum_high'] -
        df['momentum_low'] +
        df['volatility'] +
        df['price_movement'] +
        df['market_activity']
    )
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
