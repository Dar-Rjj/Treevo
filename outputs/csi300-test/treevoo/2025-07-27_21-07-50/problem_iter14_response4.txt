```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])

    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_log_return = intraday_log_return / (df['high'] - df['low'])

    # Volume Impact and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    sum_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=20).sum()
    ma_sqrt_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = sum_volume_adjusted_price_change / ma_sqrt_volume

    # Combined Alpha
    combined_alpha = scaled_log_return + vami

    # Intraday Momentum Component
    momentum_20d = df['close'] - df['close'].shift(20)
    volatility_20d = df['close'].rolling(window=20).std()
    scaled_momentum_20d = momentum_20d / volatility_20d

    # Final Alpha Combination
    final_alpha = combined_alpha + scaled_momentum_20d

    return final_alpha
```
