```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['Close'] - df['Open']
    
    # Calculate Volume Adjusted Price Change
    df['Volume_Adjusted_Price_Change'] = df['Daily_Price_Change'] * df['Volume']
    df['Volume_Adjusted_Price_Change_Sum'] = df['Volume_Adjusted_Price_Change'].rolling(window=20).sum()
    
    # Calculate Moving Average of Volume
    df['Avg_Volume'] = df['Volume'].rolling(window=20).mean()
    
    # Calculate True Range
    df['True_Range'] = df[['High', 'Low', 'Close']].join(df['Close'].shift(1)).apply(
        lambda x: max(x['High'] - x['Low'], x['High'] - x[2], x[2] - x['Low']), axis=1
    )
    
    # Calculate Average True Range
    df['Avg_True_Range'] = df['True_Range'].rolling(window=20).mean()
    
    # Adjust for Volatility
    df['Adjusted_VAMI'] = df['Volume_Adjusted_Price_Change_Sum'] / df['Avg_True_Range']
    
    return df['Adjusted_VAMI'].dropna()
```
