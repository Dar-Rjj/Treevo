import pandas as pd

def heuristics_v2(df, lookback_period=5):
    # Calculate Integrated Momentum
    df['Momentum'] = df['close'].pct_change(lookback_period)
    
    # Integrate Volume into Momentum
    df['Volume_Adjusted_Momentum'] = df['Momentum'] * df['volume']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Abs_Daily_Return'] = df['Daily_Return'].abs()
    df['Intraday_Volatility'] = df['Abs_Daily_Return'].rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    df['Volume_Thrust'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Adjusted Momentum and Intraday Factors
    df['Adjusted_Momentum_Intraday'] = (df['Volume_Adjusted_Momentum'] - df['Intraday_Volatility'] 
                                        + df['Intraday_Return'] + df['Intraday_Range'])
    
    # Final Factor
    df['Final_Factor'] = df['Adjusted_Momentum_Intraday'] * df['Volume_Thrust']
    
    return df['Final_Factor']
