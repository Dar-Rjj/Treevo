```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day exponential moving average of the money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-day exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate the Money Flow Index
    mfi = 100 - (100 / (1 + (ema_money_flow / ema_volume)))
    
    # Calculate the 14-day exponential moving average of the close price
    ema_close = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the final factor value
    factor_value = (mfi - 50) * (df['close'] - ema_close)
    
    return factor_value
```
