```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return_high'] = (df['high'] - df['open']) / df['open']
    df['intraday_return_low'] = (df['low'] - df['open']) / df['open']
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Momentum Component
    df['lagged_intraday_return'] = df['intraday_return'].shift(1)
    df['current_intraday_return'] = df['intraday_return']
    df['momentum_component'] = df['lagged_intraday_return'] * df['current_intraday_return']
    
    # Reversal Component
    df['lagged_close_to_open_return'] = (df['open'].shift(-1) - df['close'].shift(1)) / df['close'].shift(1)
    df['current_close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['reversal_component'] = df['lagged_close_to_open_return'] * df['current_close_to_open_return']
    
    # Momentum and Reversal Interaction
    df['interaction'] = df['momentum_component'] * df['reversal_component']
    
    # Volume Adjusted Factor
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_factor'] = df['interaction'] / df['avg_volume_20']
    
    return df['volume_adjusted_factor']
```
