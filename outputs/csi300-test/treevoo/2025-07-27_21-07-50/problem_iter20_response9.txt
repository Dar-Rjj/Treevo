```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Return
    volume_adjusted_intraday_return = (df['close'] - df['open']) * df['volume']
    
    # Enhance Volatility Correction
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].rolling(window=20).std()
    corrected_intraday_momentum = (
        volume_adjusted_intraday_return - 
        price_volatility - 
        volume_volatility
    )
    
    # Incorporate Trend Adjustment
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjustment = np.where(trend_difference > 0, 1.5, 0.5)
    corrected_intraday_momentum = corrected_intraday_momentum * trend_adjustment
    
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spreads = (high_low_spread + open_close_spread) * df['volume']
    
    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']
    
    # Simple Moving Average (SMA) of Close
    sma = df['close'].rolling(window=21).mean()
    
    # Volume-Weighted Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_intraday_log_return = intraday_log_return / (df['high'] - df['low'])
    volume_weighted_intraday_log_return = scaled_intraday_log_return * df['volume']
    
    # Short-Term Reversal
    past_returns = df['close'].pct_change().rolling(window=5).sum()
    
    # Composite Alpha Factor
    composite_alpha_factor = (
        (volume_weighted_close - sma) +
        volume_weighted_intraday_log_return +
        corrected_intraday_momentum +
        past_returns
    )
    
    return composite_alpha_factor
```
