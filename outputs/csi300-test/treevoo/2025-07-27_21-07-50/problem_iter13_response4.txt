```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Momentum
    short_term_period = 5
    df['Return'] = df['close'].pct_change(short_term_period)
    df['AvgVolume'] = df['volume'].rolling(window=short_term_period).mean()
    df['Volume_Adjusted_Momentum'] = df['Return'] * df['AvgVolume']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] / df['open']) - 1
    
    # Calculate Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Abs_Daily_Return'] = df['Daily_Return'].abs()
    df['Intraday_Volatility'] = df['Abs_Daily_Return'].rolling(window=20).sum() / 20
    
    # Subtract Intraday Volatility from Volume-Adjusted Momentum
    df['Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] - df['Intraday_Volatility']
    
    # Add Intraday Return and Range
    df['Intraday_Factor'] = df['Intraday_Return'] + df['Intraday_Range']
    
    # Combine the adjusted momentum and intraday factors
    df['Factor'] = df['Adjusted_Momentum'] + df['Intraday_Factor']
    
    # Adjust by Volume Thrust
    df['Volume_Thrust'] = df['volume'].diff().apply(lambda x: 1 if x > 0 else 0)
    df['Final_Factor'] = df['Factor'] * df['Volume_Thrust']
    
    return df['Final_Factor']
```
