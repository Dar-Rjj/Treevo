```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    spread_sum = high_low_spread + open_close_spread
    intraday_volatility = spread_sum * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_intraday_log_return = intraday_log_return / (df['high'] - df['low'])
    
    # Volume Impact and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    moving_average_volume = df['volume'].rolling(window=20).mean()
    volume_impact_adjustment = df['volume'] - moving_average_volume
    
    # Generate Combined Alpha
    vami = volume_adjusted_price_change
    combined_alpha = (scaled_intraday_log_return + vami) - volume_impact_adjustment
    
    return combined_alpha
```
