```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Adjusted Momentum
    short_term_period = 5
    df['Return'] = df['close'].pct_change()
    df['Volume_Avg'] = df['volume'].rolling(window=short_term_period).mean()
    df['Volume_Adjusted_Momentum'] = df['Return'] * df['Volume_Avg']

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Range
    df['Intraday_Range'] = (df['high'] - df['low']) / df['low']

    # Calculate Intraday Volatility
    df['Daily_Return'] = df['close'].pct_change()
    df['Absolute_Daily_Return'] = np.abs(df['Daily_Return'])
    df['Intraday_Volatility'] = df['Absolute_Daily_Return'].rolling(window=20).sum() / 20

    # Combine Adjusted Momentum and Intraday Factors
    df['Combined_Factor'] = df['Volume_Adjusted_Momentum'] - df['Intraday_Volatility'] + df['Intraday_Return'] + df['Intraday_Range']

    # Introduce Volume Weighting
    df['Prev_Volume'] = df['volume'].shift(1)
    df['Volume_Weight'] = (df['volume'] > df['Prev_Volume']).astype(int)

    # Final Factor
    df['Final_Factor'] = df['Combined_Factor'] * df['Volume_Weight']

    return df['Final_Factor']
```
