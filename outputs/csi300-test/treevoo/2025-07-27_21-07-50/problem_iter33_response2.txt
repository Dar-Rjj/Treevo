```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_volatility=20, lookback_trend_long=60, lookback_trend_short=10, adjustment_factor_positive=1.5, adjustment_factor_negative=0.5):
    # Compute Base Intraday Components
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['close'] - df['open']
    df['volume_adjusted_spread'] = (df['high_low_spread'] + df['open_close_spread']) * np.sqrt(df['volume'])
    
    # Incorporate Log-Return Scaled by Volatility
    df['log_return'] = np.log(df['close'] / df['open'])
    df['scaled_log_return'] = df['log_return'] / df['high_low_spread']
    
    # Adjust for Price and Volume Volatility
    close_returns = df['close'].pct_change()
    df['price_volatility'] = close_returns.rolling(window=lookback_volatility).std()
    df['volume_volatility'] = df['volume'].rolling(window=lookback_volatility).std()
    df['corrected_intraday_momentum'] = df['scaled_log_return'] - df['price_volatility'] - df['volume_volatility']
    
    # Apply Trend Adjustment
    df['long_term_trend'] = df['close'].rolling(window=lookback_trend_long).mean()
    df['short_term_trend'] = df['close'].ewm(span=lookback_trend_short, adjust=False).mean()
    df['trend_difference'] = df['long_term_trend'] - df['short_term_trend']
    df['trend_adjusted_factor'] = df['corrected_intraday_momentum'] * np.where(df['trend_difference'] > 0, adjustment_factor_positive, adjustment_factor_negative)
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    df['daily_price_change'] = df['close'] - df['open']
    df['volume_weighted_price_change'] = df['daily_price_change'] * np.sqrt(df['volume'])
    df['sqrt_volume_ma'] = np.sqrt(df['volume']).rolling(window=20).mean()
    df['vami'] = (df['scaled_log_return'] + df['volume_weighted_price_change'] / df['sqrt_volume_ma'])
    
    # Intraday Momentum Component
    df['momentum_20'] = (df['close'] - df['close'].shift(20))
    df['volatility_20'] = df['close'].rolling(window=20).std()
    df['scaled_momentum_20'] = df['momentum_20'] / df['volatility_20']
    df['momentum_5'] = (df['close'] - df['close'].shift(5))
    df['volatility_5'] = df['close'].rolling(window=5).std()
    df['scaled_momentum_5'] = df['momentum_5'] / df['volatility_5']
    
    # Introduce Intraday Volume-Ratio Factor
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['adjusted_intraday_return'] = df['intraday_return'] * df['volume_ratio']
    
    # Final Alpha Combination
    df['alpha_factor'] = (df['volume_adjusted_spread'] +
                          df['scaled_log_return'] +
                          df['trend_adjusted_factor'] +
                          df['vami'] +
                          df['scaled_momentum_20'] +
                          df['scaled_momentum_5'] +
                          df['adjusted_intraday_return'])
    
    return df['alpha_factor']
```
