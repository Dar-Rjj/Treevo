import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=5):
    # Calculate Momentum Factor
    df['return'] = df['close'].pct_change(lookback_period)
    
    # Integrate Volume into Momentum
    df['volume_adjusted_momentum'] = df['return'] * df['volume']
    
    # Calculate Price Volatility
    df['volatility'] = df['close'].pct_change().rolling(window=lookback_period).std()
    
    # Combine Adjusted Momentum and Volatility
    df['corrected_momentum_factor'] = df['volume_adjusted_momentum'] - df['volatility']
    
    # Return the factor values as a Series
    return df['corrected_momentum_factor']
