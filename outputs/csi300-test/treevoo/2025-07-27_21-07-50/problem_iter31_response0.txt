```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']

    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    ema_20_volume = df['volume'].ewm(span=20).mean()
    vami = volume_adjusted_price_change / ema_20_volume

    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=5).sum()

    # Composite Alpha Factor
    volume_weighted_close = df['close'] * df['volume']
    sma_21_close = df['close'].rolling(window=21).mean()
    volume_weighted_close_minus_sma = volume_weighted_close - sma_21_close

    composite_alpha = (volume_weighted_close_minus_sma
                      + vami
                      + short_term_reversal
                      + intraday_volatility)

    return composite_alpha
```
