```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame, lookback_period: int = 5) -> pd.Series:
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open_diff = close_open_diff * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio * volume_weighted_close_open_diff
    
    # Calculate Integrated Momentum
    momentum = df['close'].pct_change(lookback_period)
    
    # Integrate Volume into Momentum
    avg_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_adjusted_momentum = momentum * avg_volume
    
    # Calculate Intraday Return
    intraday_return = (df['close'] / df['open']) - 1
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    daily_return = df['close'].pct_change().abs()
    intraday_volatility = daily_return.rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    volume_thrust = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Adjusted Momentum and Intraday Factors
    adjusted_momentum_intraday_factors = (volume_adjusted_momentum - intraday_volatility + 
                                          intraday_return + intraday_range)
    
    # Final Factor
    final_factor = (integrated_factors + adjusted_momentum_intraday_factors) * integrated_factors
    
    # Final Adjusted Factor
    final_adjusted_factor = final_factor * volume_thrust
    
    return final_adjusted_factor
```
