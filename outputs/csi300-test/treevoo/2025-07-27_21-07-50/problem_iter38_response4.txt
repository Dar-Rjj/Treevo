import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    df['High_Low_Spread_Ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    df['Volume_Weighted_Close_Open_Diff'] = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    df['Integrated_Factor'] = df['High_Low_Spread_Ratio'] + df['Volume_Weighted_Close_Open_Diff']
    
    # Composite Alpha Factor
    # Calculate EMA of close prices
    ema_period = 21
    df['EMA'] = df['close'].ewm(span=ema_period, adjust=False).mean()
    
    # Calculate Volume-Weighted Close Price
    df['Volume_Weighted_Close'] = df['close'] * df['volume']
    
    # Subtract EMA from volume-weighted close
    df['Composite_Alpha'] = df['Volume_Weighted_Close'] - df['EMA']
    
    # Add medium-term reversal
    m_days = 10
    df['Daily_Return'] = df['close'].pct_change()
    df['Medium_Term_Reversal'] = df['Daily_Return'].rolling(window=m_days).sum()
    
    # Generate Adjusted Return
    df['Intraday_Log_Return'] = np.log(df['close'] / df['open'])
    
    # Incorporate Intraday Volatility
    df['True_Range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    df['Adjusted_Return'] = df['Intraday_Log_Return'] / df['True_Range']
    
    # Incorporate Volume Impact
    avg_volume_window = 10
    df['Average_Volume'] = df['volume'].rolling(window=avg_volume_window).mean()
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Return'] += df['Volume_Change']
    
    # Final Integration
    df['Final_Integration'] = df['Integrated_Factor'] + df['Composite_Alpha'] + df['Medium_Term_Reversal']
    df['Final_Integration'] += df['Adjusted_Return']
    
    # Analyze Price Patterns
    df['Daily_Return'] = (df['close'] - df['open']) / df['open']
    
    # Examine Momentum
    short_sma_period = 7
    long_sma_period = 21
    df['SMA_short'] = df['close'].rolling(window=short_sma_period).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_period).mean()
    df['SMA_diff'] = df['SMA_short'] - df['SMA_long']
    
    # Evaluate Trading Volume
    df['Volume_Price_Signal'] = (df['close'] - df['close'].shift(1)) * df['Volume_Change']
    
    # Assess Volatility
    volatility_window = 20
    df['Volatility'] = df['close'].rolling(window=volatility_window).std()
    
    # Consider Market Depth
    df['Bid_Ask_Spread'] = df['high'] - df['low']
    
    # Integrate Multiple Factors
    df['Composite_Score'] = (df['SMA_diff'] * 0.3) + (df['Volume_Price_Signal'] * 0.2) + (df['Volatility'] * 0.2) + (df['Bid_Ask_Spread'] * 0.3)
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Final_Integration'] + df['Composite_Score']
    
    return df['Final_Alpha_Factor']
