import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Log-Return
    df['intraday_log_return'] = np.log(df['close']) - np.log(df['open'])
    
    # Calculate Intraday Volatility
    df['high_low_range'] = (df['high'] - df['low']) / df['open']
    
    # Measure Volume Impact
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Generate Adjusted Return
    df['adjusted_return'] = df['intraday_log_return'] * df['high_low_range'] + df['volume_change']
    
    return df['adjusted_return']
