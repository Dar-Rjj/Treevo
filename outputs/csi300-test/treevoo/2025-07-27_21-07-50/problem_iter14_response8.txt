```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Intraday Open-Close Difference
    open_close_diff = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio * open_close_diff
    
    # Composite Alpha Factor
    ema_period = 21
    volume_weighted_close = df['close'] * df['volume']
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()
    composite_alpha = volume_weighted_close - ema_close
    
    # Add short-term reversal
    daily_returns = df['close'].pct_change()
    n_day_returns = daily_returns.rolling(window=5).sum()
    composite_alpha += n_day_returns
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    high_low_range = df['high'] - df['low']
    adjusted_return = intraday_log_return * (high_low_range / df['close'])
    
    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return += volume_change / avg_volume
    
    # Calculate Integrated Momentum
    lookback_period = 5
    momentum = df['close'].pct_change(periods=lookback_period)
    integrated_momentum = momentum * avg_volume
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    abs_daily_return = np.abs(np.log(df['close'] / df['close'].shift(1)))
    intraday_volatility = abs_daily_return.rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    volume_thrust = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Adjusted Momentum and Intraday Factors
    combined_factors = integrated_momentum - intraday_volatility + intraday_return + intraday_range
    
    # Final Integration
    final_factor = integrated_factor + composite_alpha + combined_factors + adjusted_return
    
    # Final Adjusted Factor
    final_adjusted_factor = final_factor * volume_thrust
    
    return final_adjusted_factor
```
