```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    ema_period = 21
    df['ema_close'] = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha_factor = volume_weighted_close - df['ema_close']
    
    # Add medium-term reversal
    m_days = 10
    daily_returns = df['close'].pct_change()
    medium_term_reversal = daily_returns.rolling(window=m_days).sum()
    composite_alpha_factor += medium_term_reversal
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Incorporate Intraday Volatility
    true_range = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs()  # High - Low
    true_range = true_range.combine_first((df['high'] - df['close'].shift(1)).abs())  # High - Previous Close
    true_range = true_range.combine_first((df['close'].shift(1) - df['low']).abs())  # Previous Close - Low
    volatility_adjusted_return = intraday_log_return / true_range
    
    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=21).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return = volatility_adjusted_return + volume_change / avg_volume
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha_factor + adjusted_return
    
    return final_factor
```
