import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum and Trend Analysis
    df['weighted_close'] = df['close'].rolling(window=10).apply(lambda x: (x * np.arange(1, 11)).sum() / np.arange(1, 11).sum(), raw=True)
    df['roc_vol_adjusted'] = (df['close'] - df['close'].shift(5)) / df['volume']
    
    # Volume Anomaly and Surge Detection
    df['volume_30_ma'] = df['volume'].rolling(window=30).mean()
    df['volume_anomaly'] = (df['volume'] - df['volume_30_ma']) / df['volume_30_ma']
    df['vol_ratio_40_max'] = df['volume'] / df['volume'].rolling(window=40).max()
    
    # Price Volatility and Stability
    df['close_std_10'] = df['close'].rolling(window=10).std()
    df['atr_14'] = np.maximum.reduce([df['high'] - df['low'], abs(df['high'] - df['close'].shift()), abs(df['low'] - df['close'].shift())]).rolling(window=14).mean()
    df['atr_14_shift_14'] = df['atr_14'].shift(14)
    df['atr_ratio'] = df['atr_14'] / df['atr_14_shift_14']
    
    # Reversal and Continuation Patterns
    df['bearish_engulfing'] = ((df['open'] > df['close'].shift()) & (df['close'] < df['open'].shift()) & (df['close'] < df['open']) & (df['open'].shift() < df['close'].shift()))
    df['bullish_engulfing'] = ((df['open'] < df['close'].shift()) & (df['close'] > df['open'].shift()) & (df['close'] > df['open']) & (df['open'].shift() > df['close'].shift()))
    df['gap_up_bullish'] = (df['bearish_engulfing'].shift(-1) & (df['open'].shift(-1) < df['close']))
    df['gap_down_bearish'] = (df['bullish_engulfing'].shift(-1) & (df['open'].shift(-1) > df['close']))
    
    # Liquidity and Trading Activity
    df['volume_amount_ratio'] = df['volume'] / df['amount']
    df['volume_amount_ratio_20_ma'] = df['volume_amount_ratio'].rolling(window=20).mean()
    df['up_day_volume'] = df[df['close'] > df['close'].shift()]['volume']
    df['down_day_volume'] = df[df['close'] < df['close'].shift()]['volume']
    df['up_down_vol_ratio'] = df['up_day_volume'].fillna(0) / df['down_day_volume'].fillna(0)
    df['up_down_vol_ratio_20_ma'] = df['up_down_vol_ratio'].rolling(window=20).mean()
    
    # Intraday Volatility
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['close'] - df['open']
    df['intraday_volatility'] = (df['high_low_spread'] + df['open_close_spread']) * df['volume']
    
    # Volume-Weighted Close
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['vwc_sma_21'] = df['volume_weighted_close'].rolling(window=21).mean()
    
    # Volume Adjusted Momentum (VAMI)
    df['price_change'] = df['close'] - df['open']
    df['volume_adjusted_price_change'] = df['price_change'] * df['volume']
    df['ema_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['vami'] = df['volume_adjusted_price_change'] / df['ema_20']
    
    # Short-Term Reversal
    df['returns'] = df['close'].pct_change()
    df['short_term_reversal'] = df['returns'].rolling(window=5).sum()
