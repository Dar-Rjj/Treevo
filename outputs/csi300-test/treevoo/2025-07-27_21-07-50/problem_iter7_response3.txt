```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Momentum
    lookback_period = 5
    momentum = df['close'].pct_change(lookback_period)
    
    # Integrate Volume
    volume_adjusted_momentum = momentum * df['volume']
    
    # Calculate Price Volatility
    returns = df['close'].pct_change()
    price_volatility = returns.rolling(window=lookback_period).std()
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].pct_change().rolling(window=lookback_period).std()
    
    # Combine Adjusted Momentum and Volatility
    corrected_momentum = (volume_adjusted_momentum - price_volatility) - volume_volatility
    
    return corrected_momentum
```
