import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=5):
    # Calculate Momentum Factor
    df['returns'] = df['close'].pct_change()
    momentum = df['returns'].rolling(window=lookback_period).sum()

    # Integrate Volume into Momentum
    volume_adjusted_momentum = momentum * df['volume']

    # Calculate Price Volatility
    price_volatility = df['returns'].rolling(window=lookback_period).std()

    # Calculate Volume Volatility
    volume_returns = df['volume'].pct_change()
    volume_volatility = volume_returns.rolling(window=lookback_period).std()

    # Combine Adjusted Momentum and Volatility
    corrected_momentum = (volume_adjusted_momentum 
                          - price_volatility 
                          - volume_volatility)

    return corrected_momentum.dropna()
