import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Component
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_log_return = intraday_log_return / (df['high'] - df['low'])
    
    # Volume Impact and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    
    # Moving Average of Square Root of Volume
    ma_sqrt_volume = df['volume'].apply(np.sqrt).rolling(window=20).mean()
    current_sqrt_volume = np.sqrt(df['volume'])
    prev_sqrt_volume = current_sqrt_volume.shift(1)
    
    # Measure and Adjust Volume Impact
    volume_impact_adjustment = current_sqrt_volume - ma_sqrt_volume
    
    # Generate Combined Alpha
    combined_alpha = scaled_log_return + (volume_adjusted_price_change / ma_sqrt_volume) - volume_impact_adjustment
    
    return combined_alpha
