```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Short-Term Momentum
    short_term_mom = (data['close'] - data['close'].shift(5)) / data['close'].shift(5)
    
    # Medium-Term Trend
    medium_term_trend = (data['close'] - data['close'].shift(20)) / data['close'].shift(20)
    
    # Volatility and Mean Reversion Indicator
    avg_daily_range = (data['high'] - data['low']).rolling(window=20).mean()
    
    # Institutional Activity Indicator (20-day VWAP)
    vwap = (data['close'] * data['volume']).rolling(window=20).sum() / data['volume'].rolling(window=20).sum()
    
    # Liquidity Proxy
    avg_amount_traded = data['amount'].rolling(window=20).mean()
    
    # Corrected Intraday Momentum with Adjustments
    intraday_return_vol_weighted = (data['close'] - data['open']) * np.sqrt(data['volume'])
    close_returns_std = data['close'].pct_change().rolling(window=20).std()
    volume_std = data['volume'].rolling(window=20).std()
    adjusted_intraday_momentum = intraday_return_vol_weighted - close_returns_std - volume_std
    
    # High-Low Spread Ratio
    high_low_spread_ratio = (data['high'] - data['low']) / (data['high'] + data['low'])
    
    # Composite Alpha Factor Components
    vol_weighted_close_open_diff = (data['close'] - data['open']) * data['volume']
    ema_21 = data['close'].ewm(span=21, adjust=False).mean()
    subtract_ema_from_volume_weighted_close = (data['close'] - ema_21) * data['volume']
    medium_term_reversal = data['close'].pct_change().rolling(window=10).sum()
    
    composite_alpha_factor = vol_weighted_close_open_diff + subtract_ema_from_volume_weighted_close + medium_term_reversal
    
    # Combine Core Factors
    core_factors = (
        short_term_mom +
        medium_term_trend +
        avg_daily_range +
        vwap +
        avg_amount_traded +
        adjusted_intraday_momentum +
        high_low_spread_ratio +
        composite_alpha_factor
    )
    
    # Long-Term and Short-Term Trend Adjustment
    long_term_trend = data['close'].rolling(window=100).mean()
    short_term_trend = data['close'].ewm(span=20, adjust=False).mean()
    trend_adjustment = np.where((short_term_trend - long_term_trend) > 0, 1.7, 0.3)
    
    # Final Alpha Factor
    alpha_factor = core_factors * trend_adjustment
    
    return alpha_factor
```
