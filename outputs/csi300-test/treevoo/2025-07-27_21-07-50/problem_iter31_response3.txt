```python
import pandas as pd

def heuristics_v2(df):
    # Compute the daily return using close price
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate On-Balance Volume (OBV) to gauge buying and selling pressure
    df['obv'] = (df['volume'] * (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))).cumsum()
    
    # Measure the range of daily trading as an indicator of volatility
    df['high_low_spread'] = df['high'] - df['low']
    
    # Determine the difference between open and close prices for sentiment analysis
    df['open_close_diff'] = df['close'] - df['open']
    
    # Compare the amount traded with volume to infer average transaction size
    df['avg_transaction_size'] = df['amount'] / df['volume']
    
    # Evaluate the change in average transaction size over time
    df['change_avg_transaction_size'] = df['avg_transaction_size'] - df['avg_transaction_size'].shift(1)
    
    # Simple Moving Average (SMA) of closing prices to identify trends
    df['sma_5'] = df['close'].rolling(window=5).mean()
    
    # Exponential Moving Average (EMA) of closing prices for more recent data weighting
    n = 5
    alpha = 2 / (n + 1)
    df['ema_5'] = df['close'].ewm(alpha=alpha, adjust=False).mean()
    
    # Implement a breakout strategy using high and low prices
    df['is_breakout_high'] = df['high'] > df['high'].rolling(window=20).max().shift(1)
    df['is_breakout_low'] = df['low'] < df['low'].rolling(window=20).min().shift(1)
    
    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (df['daily_return'] +
                          df['obv'].pct_change() +
                          (df['high_low_spread'] / df['close']) +
                          (df['open_close_diff'] / df['close']) +
                          df['change_avg_transaction_size'] +
                          (df['sma_5'] - df['close']) / df['close'] +
                          (df['ema_5'] - df['close']) / df['close'] +
                          df['is_breakout_high'].astype(int) -
                          df['is_breakout_low'].astype(int))
    
    return df['alpha_factor']
```
