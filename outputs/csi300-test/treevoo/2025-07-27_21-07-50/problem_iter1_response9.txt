```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = price_momentum / intraday_volatility
    
    # Weight by Volume
    weighted_momentum = adjusted_momentum * df['volume'].shift(1)
    
    return weighted_momentum
```
