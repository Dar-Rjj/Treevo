def heuristics_v2(df):
    # Calculate the change in closing price from one day to the next
    df['close_change'] = df['close'].diff()
    
    # Determine if the close price is higher or lower than the previous day's close
    df['close_direction'] = df['close_change'].apply(lambda x: 1 if x > 0 else -1)
    
    # Examine the relationship between the opening and closing prices
    df['open_close_diff'] = df['close'] - df['open']
    
    # Determine if the day closed with an upward or downward movement
    df['day_direction'] = df['open_close_diff'].apply(lambda x: 1 if x > 0 else -1)
    
    # Investigate the range of the daily price movement
    df['high_low_spread'] = df['high'] - df['low']
    
    # Evaluate the spread as a measure of volatility
    df['volatility_change'] = df['high_low_spread'].diff().apply(lambda x: 1 if x > 0 else -1)
    
    # Consider the volume in relation to price changes
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Determine if the volume has increased or decreased
    df['volume_direction'] = df['volume_ratio'].apply(lambda x: 1 if x > 1 else -1)
    
    # Combine the volume change with the price change
    def combined_signal(row):
        if row['close_direction'] == 1 and row['volume_direction'] == 1:
            return 2
        elif row['close_direction'] == -1 and row['volume_direction'] == -1:
            return -2
        else:
            return 0
    
    df['combined_signal'] = df.apply(combined_signal, axis=1)
    
    # Final alpha factor
    alpha_factor = df['close_direction'] * df['day_direction'] * df['volatility_change'] * df['combined_signal']
    
    return alpha_factor
