```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = df['daily_price_change'] * df['volume']
    df['volume_adjusted_price_change_sum_20'] = df['volume_adjusted_price_change'].rolling(window=20).sum()
    
    # Calculate Intraday High-Low Difference
    df['intraday_high_low_diff'] = df['high'] - df['low']
    
    # Momentum Component
    df['lagged_intraday_high_low_diff'] = df['intraday_high_low_diff'].shift(1)
    df['momentum_component'] = df['lagged_intraday_high_low_diff'] + df['intraday_high_low_diff']
    
    # Reversal Component
    df['lagged_close_to_open_return'] = (df['close'] - df['open']).shift(1)
    df['current_close_to_open_return'] = df['close'] - df['open']
    df['reversal_component'] = df['lagged_close_to_open_return'] + df['current_close_to_open_return']
    
    # Momentum and Reversal Interaction
    df['momentum_reversal_interaction'] = df['momentum_component'] * df['reversal_component']
    
    # Volume and Price Adjusted Factor
    df['volume_20_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_and_price_adjusted_factor'] = df['volume'] / df['volume_20_ma']
    
    # Final VPA-MRF
    df['vpa_mrf'] = (df['volume_adjusted_price_change_sum_20'] * df['momentum_reversal_interaction']) / df['volume_and_price_adjusted_factor']
    
    return df['vpa_mrf'].dropna()
```
