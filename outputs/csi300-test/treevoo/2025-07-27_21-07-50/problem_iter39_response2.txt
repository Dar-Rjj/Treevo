```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    factor_product = high_low_spread * volume_weighted_diff
    
    # Composite Alpha Factor
    ema_period = 21
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha = volume_weighted_close - ema_close
    
    # Add medium-term reversal
    daily_returns = (df['close'] - df['open']) / df['open']
    m_day = 10
    medium_term_reversal = daily_returns.rolling(window=m_day).sum()
    composite_alpha += medium_term_reversal
    
    # Generate Adjusted Return
    log_return = np.log(df['close'] / df['open'])
    true_range = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), abs(df['low'] - df['close'].shift(1)))
    intraday_volatility = true_range / df['close'].shift(1)
    adjusted_return = log_return * intraday_volatility
    
    # Incorporate Volume Impact
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return *= volume_change
    
    # Final Integration
    final_factor = (factor_product + composite_alpha) + adjusted_return
    return final_factor
```
