import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum and Volume-Weighted Close-Open Difference Integration
    df['SMA_14'] = df['close'].rolling(window=14).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Momentum'] = df['SMA_14'] - df['SMA_50']
    
    df['Volume_Weighted_CO'] = (df['close'] - df['open']) * df['volume']
    
    df['Momentum_Volume_Integration'] = df['Momentum'] * df['Volume_Weighted_CO']
    
    # Composite Alpha Factor with Intraday Volatility and Medium-Term Reversal
    df['EMA_21'] = df['close'].ewm(span=21, adjust=False).mean()
    df['Volume_Weighted_Close'] = df['close'] * df['volume']
    df['EMA_Adjusted_Return'] = df['Volume_Weighted_Close'] - df['EMA_21']
    
    df['Daily_Return'] = df['close'].pct_change()
    df['10_Day_Return'] = df['Daily_Return'].rolling(window=10).sum()
    
    df['Intraday_Log_Return'] = np.log(df['close'] / df['open'])
    
    df['True_Range'] = df[['high', 'low']].diff(axis=1).iloc[:, 1].abs().replace(0, np.nan)
    df['Intraday_Volatility_Adjusted_Return'] = df['Intraday_Log_Return'] * df['True_Range']
    
    df['Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Adjusted_Return_with_Volume'] = df['Intraday_Volatility_Adjusted_Return'] * df['Volume_Change']
    
    # Money Flow Index and Relative Strength Analysis
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Raw_Money_Flow'] = df['Typical_Price'] * df['volume']
    positive_money_flow = df[df['close'] > df['close'].shift(1)]['Raw_Money_Flow'].fillna(0)
    negative_money_flow = -df[df['close'] < df['close'].shift(1)]['Raw_Money_Flow'].fillna(0)
    df['Positive_Money_Flow_14'] = positive_money_flow.rolling(window=14).sum()
    df['Negative_Money_Flow_14'] = negative_money_flow.rolling(window=14).sum()
    df['Money_Flow_Ratio'] = df['Positive_Money_Flow_14'] / df['Negative_Money_Flow_14']
    df['Money_Flow_Index'] = 100 - (100 / (1 + df['Money_Flow_Ratio']))
    
    df['EMA_14'] = df['close'].ewm(span=14, adjust=False).mean()
    df['EMA_200'] = df['close'].ewm(span=200, adjust=False).mean()
    df['Relative_Strength'] = df['EMA_14'] / df['EMA_200']
    
    # Final Integration
    df['Composite_Alpha'] = df['EMA_Adjusted_Return'] + df['10_Day_Return'] + df['Adjusted_Return_with_Volume']
    df['Final_Alpha'] = df['Momentum_Volume_Integration'] + df['Composite_Alpha'] + df['Money_Flow_Index'] + df['Relative_Strength']
    
    return df['Final_Alpha']
