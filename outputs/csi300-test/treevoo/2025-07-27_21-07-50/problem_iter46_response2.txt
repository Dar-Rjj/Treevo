```python
import pandas as pd
import numpy as np
from scipy.stats import linregress

def heuristics_v2(df):
    # Momentum Factors
    df['short_momentum'] = df['close'].pct_change(periods=25)
    df['medium_momentum'] = np.log(df['close']).pct_change(periods=75)
    df['long_momentum'] = (df['close'] / df['close'].shift(150)).apply(np.log)  # Compounded return
    
    # Reversal or Mean Reversion Strategies
    df['high_14'] = df['high'].rolling(window=14).max()
    df['reversal_high'] = df['close'] - df['high_14']
    
    df['ma_200'] = df['close'].rolling(window=200).mean()
    df['reversal_ma200_distance'] = df['close'] - df['ma_200']
    
    df['ema_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['reversal_ema30_ratio'] = df['close'] / df['ema_30']
    
    # Volume Integration with Price
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    df['obv'] = obv.fillna(method='ffill')
    
    df['volume_delta'] = ((df['close'] - df['open']) * df['volume']).cumsum()
    
    df['volume_average_10'] = df['volume'].rolling(window=10).mean()
    df['volume_surprise'] = df['volume'] / df['volume_average_10']
    
    # Volatility and Trend Strength
    df['daily_return'] = df['close'].pct_change()
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    
    tr = df[['high', 'low', 'close']].join(df['close'].shift(-1), rsuffix='_prev').assign(
        TR=lambda x: pd.concat([x.high - x.low, (x.close_prev - x.high).abs(), (x.close_prev - x.low).abs()], axis=1).max(axis=1))
    df['atr_14'] = tr['TR'].rolling(window=14).mean()
    
    x = np.arange(60)
    y = df['close'].rolling(window=60).apply(lambda w: linregress(x, w)[0], raw=False)
    df['trend_strength'] = y
    
    # Composite Scores
    momentum_composite = 0.3 * df['short_momentum'] + 0.5 * df['medium_momentum'] + 0.2 * df['long_momentum']
    
    reversal_composite = (0.4 * (df['reversal_high'] / df['close'])
                         + 0.3 * (df['reversal_ma200_distance'] / df['close'])
                         + 0.3 * (df['reversal_ema30_ratio'] - 1))
    
    volume_composite = 0.4 * df['obv'] + 0.3 * df['volume_delta'] + 0.3 * df['volume_surprise']
    
    alpha_factor = (momentum_composite 
                    - 0.5 * reversal_composite 
                    + 0.3 * (volume_composite - volume_composite.mean()) / volume_composite.std())
    
    return alpha_factor
```
