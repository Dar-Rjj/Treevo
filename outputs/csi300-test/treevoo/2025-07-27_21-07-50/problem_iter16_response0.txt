```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    open_close_spread = df['close'] - df['open']
    
    # Combine Spreads and Weight by Volume
    combined_spread = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Compute Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Divide by High-Low Range
    scaled_intraday_log_return = intraday_log_return / (df['high'] - df['low'])
    
    # Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    
    # 20-day Moving Average of Square Root of Volume
    rolling_avg_sqrt_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    
    # Volume Impact Adjustment
    volume_impact_adjustment = np.sqrt(df['volume']) - rolling_avg_sqrt_volume
    
    # Generate Combined Alpha
    combined_alpha = (scaled_intraday_log_return + combined_spread) - volume_impact_adjustment
    
    return combined_alpha
```
