```python
import pandas as pd
import numpy as np

def heuristics_v2(df, ema_period=21, m_days=10, volume_window=20):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    close_ema = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close_price = df['close'] * df['volume']
    composite_alpha_factor = (volume_weighted_close_price - close_ema) + df['close'].pct_change().rolling(window=m_days).sum()
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    intraday_volatility = true_range.rolling(window=volume_window).mean()
    adjusted_return = intraday_log_return / intraday_volatility
    
    # Incorporate Volume Impact
    average_volume = df['volume'].rolling(window=volume_window).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_impact = volume_change / average_volume
    adjusted_return_with_volume = adjusted_return * volume_impact
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha_factor + adjusted_return_with_volume
    
    return final_factor
```
