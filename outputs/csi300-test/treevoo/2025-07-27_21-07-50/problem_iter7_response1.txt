```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Short-Term Momentum
    data['Momentum'] = data['close'].pct_change(periods=5)
    
    # Integrate Volume with Momentum
    data['Volume_Adjusted_Momentum'] = data['Momentum'] * data['volume']
    
    # Compute Intraday Return
    data['Intraday_Return'] = (data['close'] - data['open']) / data['open']
    
    # Compute Intraday Range
    data['Intraday_Range'] = (data['high'] - data['low']) / data['low']
    
    # Calculate Intraday Volatility
    data['Daily_Return'] = data['close'].pct_change()
    data['Absolute_Daily_Return'] = data['Daily_Return'].abs()
    data['Intraday_Volatility'] = data['Absolute_Daily_Return'].rolling(window=20).mean()
    
    # Volume Thrust Indicator
    data['Volume_Thrust'] = (data['volume'] > data['volume'].shift(1)).astype(int)
    
    # Final Alpha Factor
    data['Alpha_Factor'] = (
        (data['Volume_Adjusted_Momentum'] - data['Intraday_Volatility']) + 
        data['Intraday_Return'] + 
        data['Intraday_Range']
    ) * data['Volume_Thrust']
    
    return data['Alpha_Factor']
```
