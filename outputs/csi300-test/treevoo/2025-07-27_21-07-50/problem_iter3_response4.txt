```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Change
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=20).sum()
    
    # Calculate Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Adjust by Exponential Moving Average of Volume
    vpti = volume_adjusted_price_change / ema_volume
    
    return vpti
```
