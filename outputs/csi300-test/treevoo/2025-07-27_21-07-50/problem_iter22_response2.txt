```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_volatility=20, lookback_trend_long=60, lookback_trend_short=10, positive_adjustment=1.5, negative_adjustment=0.5):
    # Compute Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume
    volume_adjusted_intraday_return = intraday_return * np.sqrt(df['volume'])
    
    # Subtract Price Volatility
    close_returns = df['close'].pct_change()
    price_volatility = close_returns.rolling(window=lookback_volatility).std()
    
    # Subtract Volume Volatility
    volume_volatility = df['volume'].pct_change().rolling(window=lookback_volatility).std()
    
    # Corrected Intraday Momentum Factor
    corrected_intraday_momentum = volume_adjusted_intraday_return - price_volatility - volume_volatility
    
    # Sum High-Low and Open-Close Spreads
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = high_low_spread + open_close_spread
    
    # Weight by Volume
    weighted_intraday_volatility = intraday_volatility * np.sqrt(df['volume'])
    
    # Calculate Long-Term Trend
    long_term_trend = df['close'].rolling(window=lookback_trend_long).mean()
    
    # Calculate Short-Term Trend
    short_term_trend = df['close'].ewm(span=lookback_trend_short, adjust=False).mean()
    
    # Determine Trend Difference
    trend_difference = long_term_trend - short_term_trend
    
    # Apply Trend Adjustment to Corrected Factor
    adjusted_factor = corrected_intraday_momentum + weighted_intraday_volatility
    adjusted_factor = adjusted_factor.where(trend_difference > 0, adjusted_factor * negative_adjustment)
    adjusted_factor = adjusted_factor.where(trend_difference <= 0, adjusted_factor * positive_adjustment)
    
    return adjusted_factor
```
