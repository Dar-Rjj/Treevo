```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['Daily_Return'] = (df['Close'] - df['Open']) / df['Open']
    
    # Examine Momentum
    df['SMA_short'] = df['Close'].rolling(window=7).mean()
    df['SMA_long'] = df['Close'].rolling(window=21).mean()
    df['SMA_diff'] = df['SMA_short'] - df['SMA_long']
    df['Volume_Weighted_Intraday_Return'] = (df['Close'] - df['Open']) * np.sqrt(df['Volume'])
    
    # Identify Trend Strength
    df['ATR'] = (df['High'].rolling(window=14).sum() - df['Low'].rolling(window=14).sum() + 
                 (np.abs(df['High'] - df['Close'].shift(1)).rolling(window=14).sum()) + 
                 (np.abs(df['Low'] - df['Close'].shift(1)).rolling(window=14).sum())) / 42
    df['Price_Volatility'] = df['Close'].rolling(window=22).std()
    
    # Evaluate Trading Volume
    df['Volume_change'] = (df['Volume'] - df['Volume'].shift(1)) / df['Volume'].shift(1)
    df['Volume_price_signal'] = (df['Close'] - df['Close'].shift(1)) * df['Volume_change']
    
    # Assess Volatility
    df['Historical_Volatility'] = df['Close'].rolling(window=20).std()
    df['Corrected_Momentum'] = df['Volume_Weighted_Intraday_Return'] / (df['Price_Volatility'] + df['Volume'].rolling(window=22).std())
    
    # Composite Alpha Factor
    df['Volume_weighted_close_open'] = (df['Close'] - df['Open']) * df['Volume']
    df['EMA_21'] = df['Close'].ewm(span=21, adjust=False).mean()
    df['Volume_weighted_close'] = df['Close'] * df['Volume']
    df['EMA_diff'] = df['Volume_weighted_close'] - df['EMA_21']
    df['VAMI'] = (df['Close'] - df['Open']) * df['Volume'] / df['Volume'].ewm(span=20, adjust=False).mean()
    df['Sum_Past_Returns'] = df['Close'].pct_change().rolling(window=5).sum()
    df['Intraday_Volatility'] = (df['High'] - df['Low'] + df['Close'] - df['Open']) * df['Volume']
    
    # Final Integration
    df['Integrated_Factor'] = (df['Corrected_Momentum'] + df['EMA_diff'] + df['VAMI'] + 
                               df['Sum_Past_Returns'] + df['Intraday_Volatility'])
    df['Long_Term_Trend'] = df['Close'].rolling(window=100).mean()
    df['Short_Term_Trend'] = df['Close'].ewm(span=20, adjust=False).mean()
    
    df.loc[df['Short_Term_Trend'] - df['Long_Term_Trend'] > 0, 'Integrated_Factor'] *= 1.75
    df.loc[df['Short_Term_Trend'] - df['Long_Term_Trend'] < 0, 'Integrated_Factor'] *= 0.75
    
    return df['Integrated_Factor']
```
