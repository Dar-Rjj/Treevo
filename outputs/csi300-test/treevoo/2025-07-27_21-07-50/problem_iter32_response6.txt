```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Price Volatility (Standard deviation of close price returns over 22-day lookback period)
    df['price_volatility'] = df['close'].pct_change().rolling(window=22).std()
    
    # Volume Volatility (Standard deviation of volume over 22-day lookback period)
    df['volume_volatility'] = df['volume'].rolling(window=22).std()
    
    # Corrected Intraday Momentum
    df['corrected_intraday_momentum'] = df['volume_weighted_intraday_return'] / (df['price_volatility'] + df['volume_volatility'])
    
    # Intraday Volatility
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['close'] - df['open']
    df['intraday_volatility'] = (df['high_low_spread'] + df['open_close_spread']) * df['volume']
    
    # Composite Alpha Factor
    df['volume_weighted_close_open_diff'] = (df['close'] - df['open']) * df['volume']
    df['ema_21_close'] = df['close'].ewm(span=21, adjust=False).mean()
    df['volume_weighted_close'] = df['close'] * df['volume']
    df['composite_alpha_factor'] = df['volume_weighted_close'] - df['ema_21_close']
    
    # Add VAMI
    df['vami_price_change'] = (df['close'] - df['open']) * df['volume']
    df['ema_20_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['vami'] = df['vami_price_change'] / df['ema_20_volume']
    df['composite_alpha_factor'] += df['vami']
    
    # Add Short-Term Reversal
    df['short_term_reversal'] = df['intraday_return'].rolling(window=5).sum()
    df['composite_alpha_factor'] += df['short_term_reversal']
    
    # Add Intraday Volatility
    df['composite_alpha_factor'] += df['intraday_volatility']
    
    # Adjusted Return
    df['intraday_log_return'] = np.log(df['close'] / df['open'])
    df['true_range'] = np.maximum(df['high'], df['close'].shift(1)) - np.minimum(df['low'], df['close'].shift(1))
    df['intraday_log_return_scaled'] = df['intraday_log_return'] / df['true_range']
    
    # Incorporate Volume Impact
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_return'] = df['intraday_log_return_scaled'] * (1 + df['volume_change'] / df['volume'].shift(1))
    
    # Final Integration
    df['integrated_factor'] = df['corrected_intraday_momentum'] + df['composite_alpha_factor'] + df['adjusted_return']
    
    # Trend Adjustment
    df['long_term_trend'] = df['close'].rolling(window=100).mean()
    df['short_term_trend'] = df['close'].ewm(span=20, adjust=False).mean()
    df['trend_adjustment'] = (df['short_term_trend'] - df['long_term_trend']).apply(lambda x: 1.75 if x > 0 else 0.75 if x < 0 else 1.0)
    df['final_factor'] = df['integrated_factor'] * df['trend_adjustment']
    
    return df['final_factor']
```
