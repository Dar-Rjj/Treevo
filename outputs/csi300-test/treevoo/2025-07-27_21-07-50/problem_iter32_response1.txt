```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # High-Low Spread Ratio
    high_low_spread = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Volume-Weighted Close-Open Difference
    volume_weighted_co_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread + volume_weighted_co_diff
    
    # Composite Alpha Factor
    ema_close = df['close'].ewm(span=21).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha = volume_weighted_close - ema_close
    
    # Medium-term Reversal
    daily_returns = df['close'].pct_change()
    medium_term_reversal = daily_returns.rolling(window=10).sum()
    composite_alpha += medium_term_reversal
    
    # Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # True Range for Volatility
    true_range = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    adjusted_intraday_volatility = intraday_log_return / true_range
    
    # Volume Impact
    avg_volume = df['volume'].rolling(window=21).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return = adjusted_intraday_volatility + volume_change / avg_volume
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha + adjusted_return
    
    return final_factor
```
