```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Open-Close Spread
    open_close_spread = df['close'] - df['open']
    
    # Combine Intraday Spreads
    combined_spread = high_low_spread + open_close_spread
    
    # Volume Weighted Combined Spread
    volume_weighted_spread = combined_spread * df['volume']
    
    # Momentum Component
    # Lagged Intraday Return
    lagged_intraday_return = (df['high'].shift(1) - df['low'].shift(1)) / df['low'].shift(1)
    
    # Current Intraday Return
    current_intraday_return = (df['high'] - df['low']) / df['low']
    
    # Reversal Component
    # Lagged Close-to-Open Return
    lagged_close_to_open_return = (df['close'].shift(1) - df['open']) / df['open']
    
    # Current Close-to-Open Return
    current_close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Momentum and Reversal Interaction
    momentum_reversal_interaction = (lagged_intraday_return + current_intraday_return) * \
                                    (lagged_close_to_open_return + current_close_to_open_return)
    
    # 20 Day Average Volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Volume Adjusted Factor
    volume_adjusted_factor = (momentum_reversal_interaction + volume_weighted_spread) / \
                              (df['volume'] / avg_volume_20)
    
    return volume_adjusted_factor
```
