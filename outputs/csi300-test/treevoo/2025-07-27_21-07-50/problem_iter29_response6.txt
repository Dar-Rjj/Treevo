import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']

    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / ema_volume_20

    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']

    # Simple Moving Average (SMA) of Volume-Weighted Close
    sma_volume_weighted_close_21 = volume_weighted_close.rolling(window=21).mean()

    # Weighted Moving Average (WMA) of Close
    weights = np.arange(1, 22)
    wma_close_21 = (df['close'].rolling(window=21).apply(lambda prices: np.dot(prices, weights)/weights.sum(), raw=True))

    # Short-Term Reversal
    short_term_reversal_5 = df['close'].pct_change().rolling(window=5).sum()

    # Long-Term Trend
    long_term_trend_60 = volume_weighted_close.rolling(window=60).mean()

    # Composite Alpha Factor
    composite_alpha_factor = (
        volume_weighted_close - sma_volume_weighted_close_21 +
        vami +
        short_term_reversal_5 +
        intraday_volatility -
        long_term_trend_60
    )
    return composite_alpha_factor
