```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']

    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    ema_volume = df['volume'].ewm(span=20).mean()
    vami = volume_adjusted_price_change / ema_volume

    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']

    # Weighted Moving Average (WMA) of Close with period 21
    wma_close = df['close'].rolling(window=21).apply(lambda x: np.average(x, weights=np.arange(1, len(x)+1)), raw=False)

    # Short-Term Reversal (5-day returns)
    short_term_reversal = df['close'].pct_change(5)

    # Composite Alpha Factor
    composite_alpha = (volume_weighted_close - wma_close) + vami + short_term_reversal

    return composite_alpha
```
