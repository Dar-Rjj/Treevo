```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    sqrt_volume_20_day_ma = df['volume'].apply(np.sqrt).rolling(window=20).mean()
    vami = volume_weighted_price_change.rolling(window=20).sum() / sqrt_volume_20_day_ma
    
    # Intraday Momentum Component
    momentum_20_day = df['close'] - df['close'].shift(20)
    volatility_20_day = df['close'].rolling(window=20).std()
    scaled_momentum_20_day = momentum_20_day / volatility_20_day
    
    # Combined Alpha
    combined_alpha = intraday_volatility + scaled_log_return + vami + scaled_momentum_20_day
    
    return combined_alpha
```
