```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility and Volume Adjusted Momentum Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spreads_with_volume = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_log_return = intraday_log_return / (df['high'] - df['low'])
    
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    
    moving_avg_sqrt_volume = df['volume'].apply(np.sqrt).rolling(window=20).mean()
    current_sqrt_volume = np.sqrt(df['volume'])
    previous_sqrt_volume = current_sqrt_volume.shift(1)
    adjusted_volume_impact = current_sqrt_volume - moving_avg_sqrt_volume
    
    # High-Low Spread Ratio and Volume-Weighted Factors
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Composite Alpha Factor Integration
    ema_period = 21
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    subtract_ema_from_volume_weighted_close = volume_weighted_close - ema_close
    
    medium_term_reversal = (df['close'] - df['open']).rolling(window=10).sum()
    
    # Adjusted Return Incorporation
    true_range = np.maximum(
        np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))),
        abs(df['low'] - df['close'].shift(1))
    )
    scaled_by_log_return = intraday_log_return / true_range
    
    # Final Combined Alpha Factor
    integrated_factors = combined_spreads_with_volume + scaled_log_return + volume_adjusted_price_change + adjusted_volume_impact
    composite_alpha_factor = subtract_ema_from_volume_weighted_close + medium_term_reversal
    adjusted_return = scaled_by_log_return
    
    final_factor = (integrated_factors + composite_alpha_factor + adjusted_return +
                    high_low_spread_ratio + volume_weighted_close_open_diff)
    
    return final_factor
```
