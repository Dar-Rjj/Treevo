import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Change
    df['volume_adjusted_price_change'] = df['daily_price_change'] * df['volume']
    df['volume_adjusted_price_change_20d'] = df['volume_adjusted_price_change'].rolling(window=20).sum()
    
    # Calculate Moving Average of Volume
    df['moving_average_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Volume-Adjusted Momentum Factor
    df['volume_adjusted_momentum'] = df['volume_adjusted_price_change_20d'] / df['moving_average_volume_20d']
    
    # Calculate Momentum Factor
    lookback_period = 5
    df['momentum'] = df['close'].pct_change(periods=lookback_period)
    
    # Integrate Volume into Momentum
    df['volume_integrated_momentum'] = df['momentum'] * df['volume']
    
    # Calculate Price Volatility
    df['price_volatility'] = df['close'].pct_change().rolling(window=lookback_period).std()
    
    # Combine Adjusted Momentum and Volatility
    df['corrected_momentum'] = df['volume_adjusted_momentum'] - df['price_volatility']
    
    # Calculate Volume Trend
    df['volume_trend'] = df['moving_average_volume_20d']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['volume_adjusted_momentum'] + 
                          df['corrected_momentum'] + 
                          df['volume_trend'])
    
    return df['alpha_factor']
