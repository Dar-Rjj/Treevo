```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = (df['high'] - df['low']) * df['volume']
    open_close_spread = (df['close'] - df['open']) * df['volume']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']

    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    vami_ema = volume_adjusted_price_change.ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / vami_ema

    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=5).mean()

    # Composite Alpha Factor
    close_sma = df['close'].rolling(window=21).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha_factor = (volume_weighted_close - close_sma) + vami + short_term_reversal + intraday_volatility

    return composite_alpha_factor
```
