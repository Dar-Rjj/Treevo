```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Calculate Price Volatility
    daily_returns = df['close'].pct_change()
    price_volatility = daily_returns.rolling(window=lookback_period).std()
    
    # Calculate Volume Volatility
    volume_volatility = df['volume'].pct_change().rolling(window=lookback_period).std()
    
    # Correct for Volatility
    corrected_momentum = (volume_adjusted_intraday_return 
                          - price_volatility 
                          - volume_volatility)
    
    return corrected_momentum
```
