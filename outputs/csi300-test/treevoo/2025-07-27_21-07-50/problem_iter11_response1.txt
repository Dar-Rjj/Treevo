```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the return from open to close
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the average of the high and low prices
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the relative volume, which is the current volume divided by the moving average of volume
    df['vol_moving_avg'] = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] / df['vol_moving_avg']
    
    # Calculate the amount traded per volume
    df['amount_per_volume'] = df['amount'] / df['volume']
    
    # Combine the factors into a composite alpha factor
    df['alpha_factor'] = (df['intraday_return'] 
                          + (df['close'] - df['avg_price']) / df['avg_price']
                          + (df['relative_volume'] - 1)
                          + (df['amount_per_volume'] - df['amount_per_volume'].rolling(window=20).mean()))
    
    # Normalize the alpha factor
    df['alpha_factor'] = (df['alpha_factor'] - df['alpha_factor'].mean()) / df['alpha_factor'].std()

    return df['alpha_factor']
```
