```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate the relative close position within the day
    df['rel_close_pos'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the volume weighted average price for the day
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=3).mean()
    
    # Calculate the money flow index over a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(df['close'] > df['close'].shift(1), 0)
    negative_flow = raw_money_flow.where(df['close'] < df['close'].shift(1), 0)
    mfi = 100 - (100 / (1 + (positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum())))
    
    # Construct the alpha factor
    df['alpha_factor'] = (df['return'].shift(-1) 
                          * (0.5 * df['range'] + 0.3 * df['rel_close_pos'] + 0.2 * (df['close'] - df['vwap']))
                          + 0.1 * mfi)
    
    # Return the alpha factor, droping NA values
    return df['alpha_factor'].dropna()
```
