```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_log_return = intraday_log_return / high_low_spread
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    sum_volume_weighted_price_change = volume_weighted_price_change.rolling(window=30).sum()
    ma_sqrt_volume = np.sqrt(df['volume']).rolling(window=30).mean()
    vami = sum_volume_weighted_price_change / ma_sqrt_volume
    
    # Combined Alpha
    combined_alpha = scaled_log_return + vami
    
    # Intraday Momentum Component
    thirty_day_momentum = df['close'].diff(30)
    thirty_day_volatility = df['close'].rolling(window=30).std()
    scaled_momentum = thirty_day_momentum / thirty_day_volatility
    
    # Final Alpha Combination
    final_alpha = combined_alpha + scaled_momentum
    
    return final_alpha
```
