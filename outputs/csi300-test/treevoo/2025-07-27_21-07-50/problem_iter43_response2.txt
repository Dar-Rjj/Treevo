```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum and Volume-Weighted Close-Open Difference Integration
    df['14_day_SMA'] = df['close'].rolling(window=14).mean()
    df['50_day_SMA'] = df['close'].rolling(window=50).mean()
    df['momentum'] = df['14_day_SMA'] - df['50_day_SMA']
    df['volume_weighted_close_open'] = (df['close'] - df['open']) * df['volume']
    df['momentum_vol_weighted_diff'] = df['momentum'] + df['volume_weighted_close_open']

    # Composite Alpha Factor with Medium-Term Reversal
    df['21_day_EMA'] = df['close'].ewm(span=21, adjust=False).mean()
    df['adjusted_close'] = df['close'] * df['volume']
    df['10_day_return'] = df['close'].pct_change(10)
    df['intraday_log_return'] = np.log(df['close'] / df['open'])
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['volume_change'] = df['volume'].pct_change(1)
    df['composite_alpha'] = df['volume_change'] * df['adjusted_close'] * df['10_day_return'] * df['intraday_log_return'] / df['true_range']

    # Money Flow Index and Relative Strength
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df['money_flow'].where(df['typical_price'] > df['typical_price'].shift(), 0)
    df['negative_money_flow'] = df['money_flow'].where(df['typical_price'] < df['typical_price'].shift(), 0)
    df['14_day_positive_money_flow'] = df['positive_money_flow'].rolling(window=14).sum()
    df['14_day_negative_money_flow'] = df['negative_money_flow'].rolling(window=14).sum()
    df['money_flow_ratio'] = df['14_day_positive_money_flow'] / df['14_day_negative_money_flow']
    df['money_flow_index'] = 100 - (100 / (1 + df['money_flow_ratio']))
    
    df['14_day_EMA'] = df['close'].ewm(span=14, adjust=False).mean()
    df['200_day_EMA'] = df['close'].ewm(span=200, adjust=False).mean()
    df['relative_strength'] = df['14_day_EMA'] / df['200_day_EMA']

    # Final Integration
    df['alpha_factor'] = (
        df['momentum_vol_weighted_diff'] +
        df['composite_alpha'] +
        df['10_day_return'] +
        df['money_flow_index'] +
        df['relative_strength']
    )

    return df['alpha_factor'].dropna()
```
