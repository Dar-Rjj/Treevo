```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].mean()) / df['volume'].std()
    
    # Calculate the high-low spread
    df['high_low_spread'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the open-close return
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the amount-volume ratio
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Combine the features into a single alpha factor
    df['alpha_factor'] = (df['close_pct_change'] * 0.3 +
                          df['norm_volume'] * 0.2 +
                          df['high_low_spread'] * 0.15 +
                          df['open_close_return'] * 0.25 +
                          df['amount_volume_ratio'] * 0.1)
    
    return df['alpha_factor']
```
