```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Close Price
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Calculate Simple Moving Average (SMA) of Close prices with a period of 21 days
    sma_period = 21
    df['sma_close'] = df['close'].rolling(window=sma_period).mean()
    
    # Calculate Short-Term Reversal using Open and Close prices
    n_days_reversal = 5
    df['daily_returns'] = (df['open'] - df['close']) / df['open']
    df['short_term_reversal'] = df['daily_returns'].rolling(window=n_days_reversal).sum()
    
    # Composite Alpha Factor
    df['alpha_factor'] = (df['volume_weighted_close'] - df['sma_close']) + df['short_term_reversal']
    
    return df['alpha_factor']
```
