```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Intraday Open-Close Spread
    open_close_spread = df['close'] - df['open']

    # Combine Intraday Spreads
    combined_spread = high_low_spread + open_close_spread

    # Volume Weighted Combined Spread
    volume_weighted_spread = combined_spread * (df['volume'] / 1000)

    # Lagged Intraday Return
    lag_high = df['high'].shift(1)
    lag_low = df['low'].shift(1)
    lag_intraday_return = (lag_high - lag_low) / lag_low

    # Current Intraday Return
    current_intraday_return = (df['high'] - df['low']) / df['low']

    # Lagged Close-to-Open Return
    lag_close = df['close'].shift(1)
    close_to_open_return_lag = (lag_close - df['open']) / df['open']

    # Current Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']

    # Momentum and Reversal Interaction
    momentum_reversal_interaction = (
        (lag_intraday_return + current_intraday_return) *
        (close_to_open_return_lag + close_to_open_return)
    )

    # 50 Day Average Volume
    avg_50_day_volume = df['volume'].rolling(window=50).mean()

    # Volume Adjusted Factor
    volume_adjusted_factor = (
        (momentum_reversal_interaction + volume_weighted_spread) /
        (df['volume'] / avg_50_day_volume)
    )

    return volume_adjusted_factor
```
