```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return_high = df['high'] / df['open'] - 1
    intraday_return_low = df['low'] / df['open'] - 1
    
    # Momentum Component
    momentum_high = intraday_return_high.shift(1)
    momentum_low = intraday_return_low.shift(1)
    
    # Reversal Component
    reversal = (df['close'] - df['open']).shift(1) / df['open'].shift(1)
    
    # Momentum and Reversal Interaction
    momentum_reversal_interaction = (momentum_high + momentum_low) * reversal
    
    # Volume Adjusted Factor
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_factor = momentum_reversal_interaction / avg_volume_20
    
    return volume_adjusted_factor
```
