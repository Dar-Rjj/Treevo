```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Intraday Open-Close Spread
    open_close_spread = df['close'] - df['open']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'].diff()
    
    # Combine Intraday Spreads and Momentum
    combined_measure = high_low_spread + open_close_spread + intraday_momentum
    
    # Weight by Volume
    weighted_factor = combined_measure * df['volume']
    
    return weighted_factor
```
