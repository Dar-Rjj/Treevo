```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Combine Initial Factors
    combined_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Compute Adjusted Close Price
    adjusted_close = df['close'] * df['volume']
    
    # Subtract EMA from Adjusted Close
    ema_period = 21
    ema = df['close'].ewm(span=ema_period, adjust=False).mean()
    ema_adjusted_close = adjusted_close - ema
    
    # Add Medium-Term Reversal
    daily_returns = (df['close'] - df['open']) / df['open']
    medium_term_reversal = daily_returns.rolling(window=10).sum()
    
    # Generate Intraday Log-Return
    intraday_log_return = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Impact
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_adjusted_intraday_log_return = intraday_log_return + volume_change
    
    # Integrate Composite Alpha Factor
    composite_alpha_factor = combined_factors + ema_adjusted_close + medium_term_reversal
    
    # Analyze Momentum
    sma_short = df['close'].rolling(window=7).mean()
    sma_long = df['close'].rolling(window=21).mean()
    sma_diff = sma_short - sma_long
    
    # Evaluate Volume and Price Movement
    volume_price_signal = (df['close'] - df['close'].shift(1)) * volume_change
    
    # Assess Volatility
    volatility = df['close'].rolling(window=20).std()
    
    # Create Final Composite Score
    composite_score = (sma_diff * 0.3) + (volume_price_signal * 0.2) + (volatility * 0.2)
    
    # Final Integration
    final_integrated_factor = combined_factors + composite_alpha_factor + composite_score
    
    # Incorporate Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=200).mean()
    final_integrated_factor += long_term_momentum
    
    # Final Alpha Factor
    final_alpha_factor = final_integrated_factor + composite_score
    
    return final_alpha_factor
```
