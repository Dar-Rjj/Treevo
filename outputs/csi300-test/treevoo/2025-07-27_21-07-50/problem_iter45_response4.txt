```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Volume-Weighted Close-Open Difference
    close_open_diff = (df['close'] - df['open']) * df['volume']

    # Average of Both Factors
    avg_factors = (high_low_spread_ratio + close_open_diff) / 2

    # Volume-Adjusted EMA Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    ema_factor = volume_weighted_close - ema_close

    # Medium-Term Reversal Factor
    daily_returns = df['close'].pct_change()
    m_day_reversal = daily_returns.rolling(window=10).mean()

    # Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])

    # True Range
    prev_close = df['close'].shift(1)
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    true_range.fillna((df['high'] - df['low']), inplace=True)

    # Scaled Intraday Log-Return
    scaled_intraday_log_return = intraday_log_return / true_range

    # Volume Impact
    volume_change = df['volume'].pct_change().fillna(0)
    volume_impact = volume_change * scaled_intraday_log_return

    # Weighted Final Integration
    weighted_sum = (0.2 * high_low_spread_ratio + 0.2 * close_open_diff + 
                    0.3 * ema_factor + 0.1 * m_day_reversal + 
                    0.2 * volume_impact)

    return weighted_sum
```
