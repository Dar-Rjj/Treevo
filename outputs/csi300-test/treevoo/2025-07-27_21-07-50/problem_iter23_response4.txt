import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the Multi-period Weighted Momentum
    returns_10d = df['close'].pct_change(10)
    returns_20d = df['close'].pct_change(20)
    returns_50d = df['close'].pct_change(50)
    momentum_score = 0.5 * returns_10d + 0.3 * returns_20d + 0.2 * returns_50d

    # Calculate the Intraday Volatility and Volume Impact
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])

    # Calculate the Intraday Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread

    # Calculate the Volume Adjusted Momentum
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=20).sum()
    volume_20d_ma = df['volume'].rolling(window=20).mean()
    vami = volume_adjusted_price_change / volume_20d_ma

    # Combine Intraday Log-Return and VAMI
    combined_alpha = scaled_log_return + vami

    # Calculate the 20-day Scaled Momentum
    momentum_20d = df['close'].diff(20)
    std_20d = df['close'].rolling(window=20).std()
    scaled_momentum_20d = momentum_20d / std_20d

    # Final Alpha Combination
    final_alpha = combined_alpha + scaled_momentum_20d

    return final_alpha
