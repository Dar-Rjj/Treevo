```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow, which is the product of the typical price and the volume
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day exponential moving average of the money flow
    ema_money_flow = money_flow.ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponential moving average of the volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # The alpha factor is the ratio of the two EMAs
    alpha_factor = ema_money_flow / ema_volume
    
    return alpha_factor
```
