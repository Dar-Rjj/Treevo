```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given price and volume data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with (date) as index, and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: A Series with (date) as index, representing the factor values.
    """
    # Calculate the relative range
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the normalized volume
    df['norm_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the price momentum
    df['momentum'] = df['close'] / df['close'].shift(10) - 1
    
    # Combine the factors into a single score
    df['alpha_factor'] = (df['range'] + df['norm_volume'] + df['momentum']) / 3
    
    return df['alpha_factor']
```
