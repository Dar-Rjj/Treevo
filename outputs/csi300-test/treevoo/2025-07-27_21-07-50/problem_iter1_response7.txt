```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Log-Return
    df['log_return'] = np.log(df['close']) - np.log(df['open'])
    
    # Calculate Intraday Volatility (High-Low Range)
    df['high_low_range'] = df['high'] - df['low']
    # Scale high-low range by log-return to get a volatility measure
    df['intraday_volatility'] = df['high_low_range'] / np.abs(df['log_return'].replace(0, np.nan))
    
    # Generate Adjusted Return
    df['adjusted_return'] = df['log_return'] - (df['log_return'] * df['intraday_volatility'])
    
    return df['adjusted_return']
```
