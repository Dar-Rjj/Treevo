```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha_factor = volume_weighted_close - ema_close
    
    # Add medium-term reversal
    daily_returns = (df['close'] - df['open']) / df['open']
    medium_term_reversal = daily_returns.rolling(window=10).mean()
    composite_alpha_factor += medium_term_reversal
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Intraday Volatility
    true_range = np.maximum(np.maximum(df['high'] - df['low'], np.abs(df['high'] - df['close'].shift(1))),
                            np.abs(df['low'] - df['close'].shift(1)))
    intraday_volatility = (true_range * df['volume'])
    adjusted_return = intraday_log_return * intraday_volatility
    
    # Incorporate Volume Impact
    average_volume = df['volume'].rolling(window=20).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return += volume_change
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    vami_ema = volume_adjusted_price_change.ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / vami_ema
    
    # Short-Term Reversal
    short_term_reversal = daily_returns.rolling(window=5).mean()
    
    # Final Integration
    final_integration = integrated_factors + composite_alpha_factor + adjusted_return + vami + short_term_reversal
    
    # Improved Composite Alpha Factor
    sma_close = df['close'].rolling(window=21).mean()
    improved_composite_alpha_factor = (volume_weighted_close - sma_close) + vami + short_term_reversal + intraday_volatility
    
    return improved_composite_alpha_factor
```
