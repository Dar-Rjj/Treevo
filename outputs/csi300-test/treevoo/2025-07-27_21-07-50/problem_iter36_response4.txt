```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20, short_lookback_period=5, long_lookback_period=60):
    # Compute Base Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_adjusted_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Incorporate Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread
    
    # Adjust for Price and Volume Volatility
    price_volatility = df['close'].pct_change().rolling(window=lookback_period).std()
    volume_volatility = df['volume'].rolling(window=lookback_period).std()
    corrected_factor = scaled_log_return - price_volatility - volume_volatility
    
    # Apply Trend Adjustment
    long_term_trend = df['close'].rolling(window=long_lookback_period).mean()
    short_term_trend = df['close'].ewm(span=short_lookback_period, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjustment = np.where(trend_difference > 0, 1.2, 0.8)
    trend_adjusted_factor = corrected_factor * trend_adjustment
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    sqrt_volume_ma = np.sqrt(df['volume']).rolling(window=lookback_period).mean()
    vami = (log_return / high_low_spread) + (volume_weighted_price_change / sqrt_volume_ma)
    
    # Intraday Momentum Component
    twenty_day_momentum = df['close'] - df['close'].shift(lookback_period)
    twenty_day_volatility = df['close'].rolling(window=lookback_period).std()
    scaled_twenty_day_momentum = twenty_day_momentum / twenty_day_volatility
    five_day_momentum = df['close'] - df['close'].shift(short_lookback_period)
    five_day_volatility = df['close'].rolling(window=short_lookback_period).std()
    scaled_five_day_momentum = five_day_momentum / five_day_volatility
    weighted_momentum = 0.7 * scaled_twenty_day_momentum + 0.3 * scaled_five_day_momentum
    
    # Introduce Intraday Volume-Ratio Factor
    intraday_volume_ratio = df['volume'] / df['volume'].rolling(window=lookback_period).mean()
    adjusted_intraday_return = intraday_return * intraday_volume_ratio
    
    # Incorporate Intraday High-Low Momentum
    high_low_momentum = (df['high'] - df['low']) / df['open']
    high_low_volatility = (df['high'] - df['low']).rolling(window=10).std()
    scaled_high_low_momentum = high_low_momentum / (df['open'] * high_low_volatility)
    
    # Final Alpha Combination
    alpha = (volume_adjusted_spreads 
             + scaled_log_return 
             + trend_adjusted_factor 
             + vami 
             + weighted_momentum 
             + adjusted_intraday_return 
             + scaled_high_low_momentum)
    
    return alpha
```
