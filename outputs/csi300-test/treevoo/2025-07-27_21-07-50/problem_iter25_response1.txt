```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']

    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    moving_average_volume_20 = df['volume'].rolling(window=20).mean()
    vami = volume_adjusted_price_change / moving_average_volume_20

    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']

    # Simple Moving Average (SMA) of Volume-Weighted Close
    sma_volume_weighted_close = volume_weighted_close.rolling(window=21).mean()

    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=5).sum()

    # Composite Alpha Factor
    composite_alpha_factor = (
        (volume_weighted_close - sma_volume_weighted_close)
        + vami
        + short_term_reversal
        + intraday_volatility
    )

    return composite_alpha_factor.dropna()
```
