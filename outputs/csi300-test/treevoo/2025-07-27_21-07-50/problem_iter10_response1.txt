```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Momentum
    lookback_period = 5
    df['Return'] = df['close'].pct_change(lookback_period)
    df['AvgVolume'] = df['volume'].rolling(window=lookback_period).mean()
    df['VolAdjMomentum'] = df['Return'] * df['AvgVolume']

    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Range
    df['IntradayRange'] = (df['high'] - df['low']) / df['open']

    # Calculate Intraday Volatility
    df['AbsDailyReturn'] = (df['close'] - df['close'].shift(1)).abs() / df['close'].shift(1)
    df['IntradayVolatility'] = df['AbsDailyReturn'].rolling(window=20).sum() / 20

    # Combine Adjusted Momentum and Intraday Factors
    df['CombinedFactor'] = df['VolAdjMomentum'] - df['IntradayVolatility'] + df['IntradayReturn'] + df['IntradayRange']

    # Adjust by Volume Thrust
    df['VolumeThrust'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['FinalFactor'] = df['CombinedFactor'] * df['VolumeThrust']

    return df['FinalFactor']
```
