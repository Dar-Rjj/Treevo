```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']

    # Volume-Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    ema_volume = df['volume'].ewm(span=20).mean()
    vami = volume_adjusted_price_change / ema_volume

    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']

    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=5).sum()

    # Long-Term Trend
    sma_volume_weighted_close = volume_weighted_close.rolling(window=60).mean()

    # Composite Alpha Factor
    composite_alpha = (volume_weighted_close - sma_volume_weighted_close +
                       vami + short_term_reversal + intraday_volatility -
                       df['close'].rolling(window=60).mean())

    # Additional Factors
    # Momentum Indicator
    ema_12 = df['close'].ewm(span=12).mean()
    ema_26 = df['close'].ewm(span=26).mean()
    momentum_indicator = ema_12 - ema_26

    # On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()

    # Volume-Adjusted Return
    daily_log_return = np.log(df['close'] / df['open'])
    volume_adjusted_log_return = daily_log_return * df['volume']
    ema_volume_adjusted_log_return = volume_adjusted_log_return.ewm(span=20).mean()

    # Enhanced Composite Alpha Factor
    enhanced_composite_alpha = (composite_alpha +
                                momentum_indicator +
                                obv +
                                ema_volume_adjusted_log_return)

    return pd.Series(enhanced_composite_alpha, index=df.index, name='enhanced_alpha')
```
