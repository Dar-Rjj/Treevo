import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])

    # Intraday Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread

    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    sum_volume_weighted_price_change = volume_weighted_price_change.rolling(window=20).sum()
    
    sqrt_volume_20_day_ma = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = sum_volume_weighted_price_change / sqrt_volume_20_day_ma

    combined_alpha = scaled_log_return + vami

    # Intraday Momentum Component
    twenty_day_momentum = df['close'].diff(20)
    twenty_day_volatility = df['close'].rolling(window=20).std()
    scaled_twenty_day_momentum = twenty_day_momentum / twenty_day_volatility

    # Final Alpha Combination
    final_alpha = combined_alpha + scaled_twenty_day_momentum

    return final_alpha
