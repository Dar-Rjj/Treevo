```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Adjusted Momentum
    lookback_period = 5
    momentum = df['close'].pct_change(lookback_period)
    average_volume = df['volume'].rolling(window=lookback_period).mean()
    volume_adjusted_momentum = momentum * average_volume
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    intraday_range = (df['high'] - df['low']) / df['low']
    
    # Calculate Intraday Volatility
    absolute_daily_return = (df['close'] - df['close'].shift(1)).abs() / df['close'].shift(1)
    intraday_volatility = absolute_daily_return.rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    df['volume_thrust'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Adjusted Momentum and Intraday Factors
    combined_factors = (volume_adjusted_momentum - intraday_volatility + 
                        intraday_return + intraday_range) * df['volume_thrust']
    
    return combined_factors
```
