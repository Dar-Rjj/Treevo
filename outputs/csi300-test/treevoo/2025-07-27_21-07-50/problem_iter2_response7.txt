```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Intraday Volatility
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['absolute_daily_return'] = df['daily_return'].abs()
    df['intraday_volatility'] = df['absolute_daily_return'].rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    df['volume_thrust'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Intraday Open-Close Spread
    df['open_close_spread'] = df['close'] - df['open']
    
    # Combine Intraday Spreads
    df['combined_spread'] = (df['high_low_spread'] + df['open_close_spread']) * df['volume']
    
    # Combine Factors
    df['combined_factor'] = (
        df['intraday_return'] +
        df['intraday_range'] +
        df['intraday_volatility'] +
        df['volume_thrust'] +
        df['combined_spread']
    )
    
    # Weight by Volume
    df['alpha_factor'] = df['combined_factor'] * df['volume']
    
    return df['alpha_factor']
```
