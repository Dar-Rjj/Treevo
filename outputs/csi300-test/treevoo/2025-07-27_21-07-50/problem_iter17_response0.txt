import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of open and close prices using volume as weights
    weighted_avg = (df['open'] * df['volume'] + df['close'] * df['volume']) / (df['volume'] + df['volume'])
    
    # Calculate the price change from the previous day
    prev_close = df['close'].shift(1)
    price_change = df['close'] - prev_close
    
    # Calculate the high-low range
    hl_range = df['high'] - df['low']
    
    # Calculate the amount per volume as a measure of trading intensity
    amt_per_vol = df['amount'] / df['volume']
    
    # Combine the above factors into a single alpha factor
    alpha_factor = (price_change / hl_range) * (amt_per_vol - amt_per_vol.shift(1)) * (weighted_avg - df['close'])
    
    return alpha_factor
