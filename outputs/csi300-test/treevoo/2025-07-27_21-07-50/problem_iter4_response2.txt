```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['open']
    
    # Generate Adjusted Return
    adjusted_return = intraday_log_return - (intraday_log_return * intraday_volatility)
    
    return adjusted_return
```
