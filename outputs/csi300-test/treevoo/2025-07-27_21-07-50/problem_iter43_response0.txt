```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']

    # Calculate Exponential Moving Average (EMA) of Close prices
    ema_period = 21
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()

    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']

    # Subtract EMA from volume-weighted close
    ema_adj_close = volume_weighted_close - ema_close

    # Add medium-term reversal
    daily_returns = df['close'].pct_change()
    m_day = 10
    m_day_reversal = daily_returns.rolling(window=m_day).sum()

    # Composite Alpha Factor
    composite_alpha_factor = ema_adj_close + m_day_reversal

    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])

    # Compute True Range
    tr_high = df['high'] - df['low']
    tr_low = abs(df['high'] - df['close'].shift(1))
    tr_close = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr_high, tr_low, tr_close], axis=1).max(axis=1)

    # Scale by Log-Return
    adjusted_return = intraday_log_return / true_range

    # Incorporate Volume Impact
    volume_change = df['volume'].pct_change()
    adjusted_return_with_volume = adjusted_return * volume_change

    # Final Integration
    combined_factors = high_low_spread + volume_weighted_close_open_diff + composite_alpha_factor + adjusted_return_with_volume

    return combined_factors
```
