```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_volatility=20, long_trend_period=100, short_trend_period=20, reversal_days=5):
    # Base Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_adjusted_spread = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread
    
    # Adjust for Price and Volume Volatility
    close_returns = df['close'].pct_change()
    price_volatility = close_returns.rolling(window=lookback_volatility).std()
    volume_volatility = df['volume'].rolling(window=lookback_volatility).std()
    corrected_momentum = log_return - price_volatility - volume_volatility
    
    # Trend Adjustment
    long_term_trend = df['close'].rolling(window=long_trend_period).mean()
    short_term_trend = df['close'].ewm(span=short_trend_period, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjustment = np.where(trend_difference > 0, 1.2, 0.8)
    corrected_momentum *= trend_adjustment
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    sqrt_volume_20d_ma = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = (log_return / high_low_spread) + (volume_weighted_price_change / sqrt_volume_20d_ma)
    
    # Intraday Momentum Component
    twenty_day_momentum = df['close'] - df['close'].shift(20)
    twenty_day_volatility = df['close'].rolling(window=20).std()
    scaled_twenty_day_momentum = twenty_day_momentum / twenty_day_volatility
    
    # Volume-Adjusted Momentum (VAMI)
    volume_adjusted_price_change = (df['close'] - df['open']) * df['volume']
    volume_ema = df['volume'].ewm(span=20, adjust=False).mean()
    vami_momentum = volume_adjusted_price_change / volume_ema
    
    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=reversal_days).sum()
    
    # Final Alpha Combination
    alpha = (volume_adjusted_spread +
             scaled_log_return +
             corrected_momentum +
             vami +
             scaled_twenty_day_momentum +
             vami_momentum +
             short_term_reversal)
    
    return alpha
```
