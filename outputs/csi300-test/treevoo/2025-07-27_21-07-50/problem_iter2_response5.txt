```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open_diff = close_open_diff * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio * volume_weighted_close_open_diff
    
    # Calculate Simple Moving Average (SMA) of Close prices
    sma_period = 21
    sma_close = df['close'].rolling(window=sma_period).mean()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate Short-Term Reversal
    daily_returns = (df['close'] - df['open']) / df['open']
    short_term_reversal = daily_returns.rolling(window=5).sum()
    
    # Composite Alpha Factor
    composite_alpha = (volume_weighted_close - sma_close) + short_term_reversal
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha
    
    return final_factor
```
