```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']

    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio + volume_weighted_close_open_diff

    # Composite Alpha Factor
    ema_period = 21
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha = (volume_weighted_close - ema_close) / volume_weighted_close

    # Add medium-term reversal
    daily_returns = df['close'].pct_change()
    m_day = 10
    medium_term_reversal = daily_returns.rolling(window=m_day).sum()

    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Compute True Range
    true_range = pd.concat(
        [df['high'] - df['low'], 
         (df['high'] - df['close'].shift(1)).abs(), 
         (df['low'] - df['close'].shift(1)).abs()], 
        axis=1
    ).max(axis=1)

    # Incorporate Intraday Volatility
    scaled_log_return = intraday_log_return / true_range

    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=ema_period).mean()
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_return = scaled_log_return + volume_change

    # Final Integration
    final_factor = integrated_factor + composite_alpha + medium_term_reversal + adjusted_return

    return final_factor
```
