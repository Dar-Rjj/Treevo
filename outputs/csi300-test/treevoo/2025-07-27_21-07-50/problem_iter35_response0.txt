```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the moving average of the close price for a 20-day window
    df['ma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the difference between the close price and its 20-day moving average
    df['diff_close_ma_20'] = df['close'] - df['ma_20']
    
    # Calculate the normalized volume, which is the volume divided by the amount
    # This assumes that 'amount' is the total value traded in the period
    df['norm_volume'] = df['volume'] / df['amount']
    
    # Calculate the factor: 
    # (normalized volume * the difference between the close price and its 20-day MA)
    df['factor'] = df['norm_volume'] * df['diff_close_ma_20']

    # Return the factor as a series
    return df['factor']
```
