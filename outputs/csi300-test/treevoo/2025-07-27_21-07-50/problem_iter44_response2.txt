```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average of the close price for a 10-day window
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the simple moving average of the volume for a 10-day window
    vol_sma_10 = df['volume'].rolling(window=10).mean()
    
    # Compute the difference between the high and low prices
    hl_diff = df['high'] - df['low']
    
    # Calculate the average of the amount over a 5-day period
    amount_avg_5 = df['amount'].rolling(window=5).mean()
    
    # Factor: (Close price - 10-day SMA of Close) / (10-day SMA of Volume)
    alpha_factor = (df['close'] - sma_10) / vol_sma_10
    
    # Normalize the factor by the high-low difference
    alpha_factor_normalized = alpha_factor / hl_diff
    
    # Finally, weight the normalized factor by the 5-day average amount
    final_alpha = alpha_factor_normalized * amount_avg_5
    
    return final_alpha
```
