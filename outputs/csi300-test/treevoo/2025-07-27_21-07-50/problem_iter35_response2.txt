import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with the index as date representing the factor values.
    """
    # Calculate the relative change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the relative change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the volatility of the close price
    df['close_volatility'] = df['close'].rolling(window=5).std()
    
    # Calculate the ratio of the amount to the volume
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    
    # Calculate the average true range
    df['high_low_diff'] = df['high'] - df['low']
    df['high_close_diff'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_close_diff'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['high_low_diff', 'high_close_diff', 'low_close_diff']].max(axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_change'] * df['volume_change'] + 
                          df['close_volatility'] * df['amount_volume_ratio'] + 
                          df['average_true_range'])
    
    return df['alpha_factor'].dropna()
