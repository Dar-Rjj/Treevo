```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Integrated Momentum
    lookback_period = 5
    momentum = df['close'].pct_change(lookback_period)

    # Integrate Volume into Momentum
    volume_adjusted_momentum = momentum * df['volume']

    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Intraday Range
    intraday_range = (df['high'] - df['low'])

    # Calculate Intraday Volatility
    abs_daily_return = np.abs(df['close'].pct_change())
    intraday_volatility = abs_daily_return.rolling(window=20).sum() / 20

    # Calculate Volume Thrust
    volume_thrust = np.where(df['volume'] > df['volume'].shift(1), 1, 0)

    # Combine Adjusted Momentum and Intraday Factors
    combined_factor = (volume_adjusted_momentum - intraday_volatility + intraday_return + intraday_range)

    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']

    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio * volume_weighted_close_open_diff

    # Final Factor Integration
    final_factor = combined_factor * volume_thrust * integrated_factor

    return final_factor
```
