```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_co_diff = (df['close'] - df['open']) * df['volume']

    # Combine Initial Factors
    initial_factors_sum = high_low_spread_ratio + volume_weighted_co_diff

    # Compute Adjusted Close Price
    adjusted_close = df['close'] * df['volume']

    # Subtract EMA from Adjusted Close
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    ema_adjusted_close = adjusted_close - ema_close

    # Add Medium-Term Reversal
    m_day_returns = (df['close'] / df['open']).rolling(window=10).mean()
    
    # Generate Intraday Log-Return
    intraday_log_return = (df['close'] - df['open']) / df['open']

    # Incorporate Volume Impact
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    intraday_volume_impact = intraday_log_return + volume_change

    # Integrate Composite Alpha Factor
    integrated_alpha_factor = initial_factors_sum + ema_adjusted_close + m_day_returns + intraday_volume_impact

    # Analyze Momentum
    sma_short = df['close'].rolling(window=7).mean()
    sma_long = df['close'].rolling(window=21).mean()
    sma_diff = sma_short - sma_long

    # Evaluate Volume and Price Movement
    volume_price_signal = (df['close'] - df['close'].shift(1)) * volume_change

    # Assess Volatility
    volatility = df['close'].rolling(window=20).std()

    # Create Final Composite Score
    composite_score = (sma_diff * 0.3) + (volume_price_signal * 0.2) + (volatility * 0.2)

    # Final Alpha Factor
    final_alpha_factor = integrated_alpha_factor + composite_score

    return final_alpha_factor
```
