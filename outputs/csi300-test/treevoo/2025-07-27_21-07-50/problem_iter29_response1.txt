```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']
    
    # Volume-Weighted Close-Open Difference
    volume_weighted_close_open = (df['close'] - df['open']) * df['volume']
    
    # Momentum
    returns_10d = df['close'].pct_change(10)
    returns_20d = df['close'].pct_change(20)
    returns_50d = df['close'].pct_change(50)
    momentum_score = 0.4 * returns_10d + 0.35 * returns_20d + 0.25 * returns_50d
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / ema_volume
    
    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=5).sum()
    
    # Medium-Term Reversal
    medium_term_reversal = df['close'].pct_change().rolling(window=10).sum()
    
    # Composite Alpha Factor
    close_volume = df['close'] * df['volume']
    wma_close_volume = close_volume.rolling(window=21).mean()
    composite_alpha_factor = (close_volume - wma_close_volume) + vami + short_term_reversal + medium_term_reversal + intraday_volatility
    
    # Final Integration
    integrated_factors = momentum_score * volume_weighted_close_open + composite_alpha_factor
    intraday_log_return = np.log(df['close'] / df['open'])
    average_volume = df['volume'].rolling(window=20).mean()
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_return = intraday_log_return * volume_change
    final_alpha = integrated_factors + adjusted_return
    
    return final_alpha
```
