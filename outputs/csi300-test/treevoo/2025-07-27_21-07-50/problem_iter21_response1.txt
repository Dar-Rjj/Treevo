```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])

    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_diff = (df['close'] - df['open']) * df['volume']

    # Integrate Both Factors
    integrated_factors = high_low_ratio + volume_weighted_diff

    # Composite Alpha Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha = volume_weighted_close - ema_close

    # Medium-term Reversal
    daily_returns = df['close'].pct_change()
    medium_term_reversal = daily_returns.rolling(window=10).sum()

    composite_alpha += medium_term_reversal

    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = np.max([df['high'] - df['low'], df['high'] - df['close'].shift(1), df['close'].shift(1) - df['low']], axis=0)
    adjusted_return = intraday_log_return / true_range

    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=21).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return += volume_change / avg_volume

    # Enhance Volume Impact
    volume_momentum = df['volume'] - df['volume'].rolling(window=5).mean()
    adjusted_return += volume_momentum

    # Final Integration
    final_factor = integrated_factors + composite_alpha + adjusted_return

    return final_factor
```
