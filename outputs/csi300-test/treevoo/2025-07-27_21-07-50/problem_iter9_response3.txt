import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility Component
    intraday_high_low_spread = df['high'] - df['low']
    intraday_open_close_spread = df['close'] - df['open']
    combined_spread = (intraday_high_low_spread + intraday_open_close_spread) * df['volume']

    # Volume Adjusted Momentum Component
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=20).sum()
    moving_average_volume = df['volume'].rolling(window=20).mean()
    adjusted_by_moving_average = volume_adjusted_price_change / moving_average_volume

    # Combined Alpha Factor
    combined_alpha = combined_spread + adjusted_by_moving_average

    return combined_alpha
