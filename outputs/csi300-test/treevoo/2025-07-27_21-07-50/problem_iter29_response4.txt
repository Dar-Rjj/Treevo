import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the average amount traded over the last 5 days
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    
    # Calculate the relative difference between today's high and low price to the close price
    df['rel_diff_high_low_to_close'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the money flow index (MFI) for a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    pos_mf_14 = positive_money_flow.rolling(window=14).sum()
    neg_mf_14 = negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (pos_mf_14 / neg_mf_14)))
    
    # Combine the factors to form an alpha factor
    df['alpha_factor'] = (df['return'] * 0.3) + \
                         (df['avg_amount_5d'] * 0.2) + \
                         (df['rel_diff_high_low_to_close'] * 0.2) + \
                         (mfi * 0.3)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
