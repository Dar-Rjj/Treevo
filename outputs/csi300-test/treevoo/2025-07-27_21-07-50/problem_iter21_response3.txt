```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the momentum factor using closing prices
    df['return_10'] = df['close'].pct_change(10)
    df['return_20'] = df['close'].pct_change(20)
    df['return_60'] = df['close'].pct_change(60)
    df['momentum'] = 0.4 * df['return_10'] + 0.35 * df['return_20'] + 0.25 * df['return_60']

    # Calculate the price-volume trend
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['price_change'] = df['close'].pct_change()
    df['price_volume_indicator'] = df['price_change'] * (df['volume'] / df['avg_volume_20'])
    df['pvt'] = df['price_volume_indicator'].rolling(window=30).mean()

    # Calculate the volatility factor
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=40).std()

    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    df['rsi_smoothed'] = df['rsi'].ewm(span=20).mean()

    # Calculate the money flow index (MFI)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['raw_money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = np.where(df['typical_price'] > df['typical_price'].shift(1), df['raw_money_flow'], 0)
    df['negative_money_flow'] = np.where(df['typical_price'] < df['typical_price'].shift(1), df['raw_money_flow'], 0)
    positive_money_flow_sum = df['positive_money_flow'].rolling(window=20).sum()
    negative_money_flow_sum = df['negative_money_flow'].rolling(window=20).sum()
    money_flow_ratio = positive_money_flow_sum / negative_money_flow_sum
    df['mfi'] = 100 - (100 / (1 + money_flow_ratio))

    # Combine all factors into a single alpha factor
    df['alpha_factor'] = 0.4 * df['momentum'] + 0.3 * df['pvt'] - 0.2 * df['volatility'] + 0.1 * df['rsi_smoothed'] + 0.1 * df['mfi']

    return df['alpha_factor']
```
