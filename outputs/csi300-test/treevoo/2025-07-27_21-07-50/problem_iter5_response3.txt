import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Momentum Component
    df['momentum'] = df['intraday_return'].shift(1)
    
    # Reversal Component
    df['reversal'] = (df['open'] - df['close']).shift(1)
    
    # Momentum and Reversal Interaction
    df['momentum_reversal_interaction'] = df['momentum'] * df['reversal']
    
    # Volume Adjusted Factor
    df['volume_avg_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_factor'] = df['volume'] / df['volume_avg_20']
    
    # Calculate Short-Term Momentum
    lookback_period_short = 5
    df['short_term_momentum'] = df['close'].pct_change(lookback_period_short)
    
    # Integrate Volume into Short-Term Momentum
    df['vol_integrated_short_term_momentum'] = df['short_term_momentum'] * df['volume']
    
    # Calculate Long-Term Momentum
    lookback_period_long = 20
    df['long_term_momentum'] = df['close'].pct_change(lookback_period_long)
    
    # Compute Price Momentum
    df['price_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Price Volatility
    df['price_volatility'] = df['close'].pct_change().rolling(window=lookback_period_short).std()
    
    # Combine Adjusted Momentum and Volatility
    df['corrected_momentum'] = df['vol_integrated_short_term_momentum'] - df['price_volatility']
    
    # Volume Weighted Average Price (VWAP) Oscillator
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['sma_vwap_20'] = df['vwap'].rolling(window=20).mean()
    df['vwap_oscillator'] = df['vwap'] - df['sma_vwap_20']
    
    # Synthesize Final Alpha Factor
    df['final_alpha_factor'] = (df['corrected_momentum'] * df['vwap_oscillator'] * df['volume_adjusted_factor'])
    
    return df['final_alpha_factor']
