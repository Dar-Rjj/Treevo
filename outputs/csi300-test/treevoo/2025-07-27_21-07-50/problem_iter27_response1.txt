```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_adjusted_intraday_return = intraday_return * np.sqrt(df['volume'])

    # Enhanced Volatility Correction
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].rolling(window=20).std()
    corrected_intraday_return = volume_adjusted_intraday_return - price_volatility - volume_volatility

    # High-Low and Open-Close Spreads
    high_low_spread = (df['high'] - df['low'])
    open_close_spread = (df['close'] - df['open'])
    combined_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])

    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_intraday_log_return = intraday_log_return / (df['high'] - df['low'])

    # Volume-Weighted Intraday Log-Return
    volume_weighted_intraday_log_return = scaled_intraday_log_return * np.sqrt(df['volume'])

    # Volume-Weighted Momentum
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    sqrt_volume_ma_20 = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = (volume_adjusted_price_change / sqrt_volume_ma_20).sum()

    # Long-Term and Short-Term Trends
    long_term_trend = df['close'].rolling(window=50).mean()
    short_term_trend = df['close'].ewm(span=10).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjusted_factor = corrected_intraday_return.where(trend_difference > 0, corrected_intraday_return * 1.2).where(trend_difference < 0, corrected_intraday_return * 0.8)

    # Intraday Momentum Component
    twenty_day_momentum = df['close'] - df['close'].shift(20)
    scaled_twenty_day_momentum = twenty_day_momentum / df['close'].rolling(window=20).std()

    # New Intraday Sentiment Indicator
    intraday_sentiment = (df['close'] - df['open']) / (df['high'] - df['low'])
    volume_adjusted_sentiment = intraday_sentiment * np.sqrt(df['volume'])

    # Final Composite Alpha Factor
    final_alpha = (corrected_intraday_return 
                   + vami 
                   + combined_spreads 
                   + scaled_intraday_log_return 
                   + volume_weighted_intraday_log_return 
                   + trend_adjusted_factor 
                   + scaled_twenty_day_momentum 
                   + volume_adjusted_sentiment)

    return final_alpha
```
