```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Volume-Weighted Close-Open Difference
    vol_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio + vol_weighted_close_open_diff
    
    # Composite Alpha Factor
    ema_close_21 = df['close'].ewm(span=21, adjust=False).mean()
    vol_weighted_close = df['close'] * df['volume']
    composite_alpha = (vol_weighted_close - ema_close_21)
    
    # Add Medium-Term Reversal
    daily_returns = (df['close'] / df['open']) - 1
    past_10_day_returns = daily_returns.rolling(window=10).sum()
    medium_term_reversal = past_10_day_returns
    
    # Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Medium-Term Trend
    medium_term_trend = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volatility and Mean Reversion Indicator
    avg_daily_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Institutional Activity Indicator
    vwap_20 = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Liquidity Proxy
    avg_amount_traded = df['amount'].rolling(window=20).mean()
    
    # Corrected Intraday Momentum with Adjustments
    vol_weighted_intraday_return = (df['close'] - df['open']) * np.sqrt(df['volume'])
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].pct_change().rolling(window=20).std()
    adjusted_intraday_momentum = vol_weighted_intraday_return - price_volatility - volume_volatility
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = (df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1)).shift(1)
    scaled_log_return = intraday_log_return / true_range
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_return = scaled_log_return * volume_change
    
    # Combined Alpha Factor
    combined_alpha = (short_term_momentum + 
                      medium_term_trend + 
                      avg_daily_range + 
                      vwap_20 + 
                      avg_amount_traded + 
                      adjusted_intraday_momentum + 
                      integrated_factor + 
                      composite_alpha + 
                      medium_term_reversal)
    
    # Trend-Adjusted Final Alpha Factor
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    trend_adjustment = 1.5 if (short_term_trend - long_term_trend).iloc[-1] > 0 else 0.5
    trend_adjusted_alpha = combined_alpha * trend_adjustment
    
    # Add Adjusted Return
    final_alpha = trend_adjusted_alpha + adjusted_return
    
    return final_alpha
```
