```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted Intraday Return
    df['volume_weighted_intraday_return'] = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Volatility Correction
    df['price_volatility'] = df['close'].pct_change().rolling(window=20).std()
    df['volume_volatility'] = df['volume'].rolling(window=20).std()
    df['corrected_intraday_momentum'] = df['volume_weighted_intraday_return'] / (df['price_volatility'] + df['volume_volatility'])
    
    # High-Low Spread Ratio
    df['high_low_spread_ratio'] = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Composite Alpha Factor
    df['volume_weighted_close_open_diff'] = (df['close'] - df['open']) * df['volume']
    df['ema_close_21'] = df['close'].ewm(span=21, adjust=False).mean()
    df['volume_weighted_close_price'] = df['close'] * df['volume']
    df['subtract_ema'] = df['volume_weighted_close_price'] - df['ema_close_21']
    df['medium_term_reversal'] = df['close'].pct_change(10).sum()
    df['composite_alpha_factor'] = df['volume_weighted_close_open_diff'] + df['subtract_ema'] + df['medium_term_reversal']
    
    # Adjusted Return
    df['intraday_log_return'] = np.log(df['close'] / df['open'])
    df['true_range'] = (df[['high', 'close']].max(axis=1) - df[['low', 'close']].min(axis=1).shift(1))
    df['intraday_log_return_scaled'] = df['intraday_log_return'] / df['true_range']
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['adjusted_return'] = df['intraday_log_return_scaled'] * (1 + df['volume_change'] / df['volume'].shift(1))
    
    # Final Integration
    df['combined_factors'] = df['high_low_spread_ratio'] + df['corrected_intraday_momentum'] + df['composite_alpha_factor']
    df['final_factor'] = df['combined_factors'] + df['adjusted_return']
    
    # Trend Adjustment
    df['long_term_trend'] = df['close'].rolling(window=100).mean()
    df['short_term_trend'] = df['close'].ewm(span=20, adjust=False).mean()
    trend_diff = df['short_term_trend'] - df['long_term_trend']
    df['trend_adjustment'] = np.where(trend_diff > 0, 1.75, 0.75)
    df['final_factor'] = df['final_factor'] * df['trend_adjustment']
    
    return df['final_factor']
```
