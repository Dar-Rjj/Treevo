```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback=20, long_term_lookback=100, short_term_lookback=10, positive_adj=1.5, negative_adj=0.5):
    # Compute Base Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_adjusted_spread = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Incorporate Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread
    
    # Adjust for Price and Volume Volatility
    price_volatility = df['close'].pct_change().rolling(window=lookback).std()
    volume_volatility = df['volume'].rolling(window=lookback).std()
    corrected_intraday_momentum = volume_adjusted_spread - price_volatility - volume_volatility
    
    # Apply Trend Adjustment
    long_term_trend = df['close'].rolling(window=long_term_lookback).mean()
    short_term_trend = df['close'].ewm(span=short_term_lookback, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjusted_factor = np.where(trend_difference > 0, corrected_intraday_momentum * positive_adj,
                                     np.where(trend_difference < 0, corrected_intraday_momentum * negative_adj, corrected_intraday_momentum))
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    vol_20_day_ma = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = scaled_log_return + (volume_weighted_price_change / vol_20_day_ma)
    
    # Intraday Momentum Component
    twenty_day_momentum = df['close'] - df['close'].shift(20)
    volatility_scaled_momentum = twenty_day_momentum / df['close'].rolling(window=20).std()
    
    # Final Alpha Combination
    final_alpha = (volume_adjusted_spread + scaled_log_return + trend_adjusted_factor + vami + volatility_scaled_momentum).fillna(0)
    
    return final_alpha
```
