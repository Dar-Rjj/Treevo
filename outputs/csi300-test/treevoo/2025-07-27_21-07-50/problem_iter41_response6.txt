```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Medium-Term Trend
    medium_term_trend = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volatility and Mean Reversion Indicator
    avg_daily_range = (df['high'] - df['low']).rolling(window=20).mean()
    
    # Institutional Activity Indicator
    vwap_20 = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Liquidity Proxy
    avg_daily_amount = df['amount'].rolling(window=20).mean()
    
    # Corrected Intraday Momentum with Adjustments
    intraday_return_vol_wgt = (df['close'] - df['open']) * np.sqrt(df['volume'])
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].pct_change().rolling(window=20).std()
    adjusted_intraday_momentum = intraday_return_vol_wgt - price_volatility - volume_volatility
    
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Volume-Weighted Close-Open Difference
    vol_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Composite Alpha Factor Components
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    composite_alpha_factor = (df['close'] - ema_21) * df['volume']
    medium_term_reversal = df['close'].pct_change().rolling(window=10).sum()
    composite_alpha_factor += medium_term_reversal
    
    # Integrated Factors
    integrated_factors = high_low_spread_ratio * vol_weighted_close_open_diff
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = df[['high', 'low', df['close'].shift(1)]].max(axis=1) - df[['high', 'low', df['close'].shift(1)]].min(axis=1)
    intraday_volatility = true_range.rolling(window=20).mean()
    adjusted_return = intraday_log_return * intraday_volatility
    volume_change = df['volume'].pct_change()
    adjusted_return *= volume_change
    
    # Final Integration
    final_integration = integrated_factors + composite_alpha_factor
    final_integration += adjusted_return
    
    # Trend-Adjusted Final Alpha Factor
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    trend_adjustment = 1.5 if (short_term_trend - long_term_trend).iloc[-1] > 0 else 0.5
    final_alpha_factor = (short_term_momentum + 
                          medium_term_trend + 
                          avg_daily_range + 
                          vwap_20 + 
                          avg_daily_amount + 
                          adjusted_intraday_momentum + 
                          high_low_spread_ratio + 
                          composite_alpha_factor + 
                          integrated_factors) * trend_adjustment
    
    return final_alpha_factor
```
