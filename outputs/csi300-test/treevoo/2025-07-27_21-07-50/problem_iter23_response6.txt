```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] indexed by date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the average price
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change relative to the average price
    price_change_rel_to_avg = (df['close'] - avg_price) / avg_price
    
    # Calculate the amount per volume as a measure of liquidity
    liquidity_measure = df['amount'] / df['volume']
    
    # Calculate the factor as a weighted sum of the above metrics
    factor = (0.5 * daily_return + 
              0.3 * price_change_rel_to_avg + 
              0.2 * liquidity_measure)
    
    return factor
```
