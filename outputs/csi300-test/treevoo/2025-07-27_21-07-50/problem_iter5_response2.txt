import pandas as pd

def heuristics_v2(df):
    # Short-Term Momentum
    short_term_momentum = df['close'].pct_change(periods=5)
    
    # Integrate Volume into Short-Term Momentum
    volume_adjusted_short_term_momentum = short_term_momentum * df['volume']
    
    # Long-Term Momentum
    long_term_momentum = df['close'].pct_change(periods=20)
    
    # Price Momentum
    price_momentum = short_term_momentum - long_term_momentum
    
    # Price Volatility
    price_volatility = df['close'].pct_change().rolling(window=5).std()
    
    # Corrected Momentum Factor
    corrected_momentum = volume_adjusted_short_term_momentum - price_volatility
    
    # Volume Weighted Average Price (VWAP) Oscillator
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    vwap_sma_20 = vwap.rolling(window=20).mean()
    vwap_oscillator = vwap - vwap_sma_20
    
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_spread = high_low_spread + open_close_spread
    volume_weighted_intraday_spread = intraday_spread * df['volume']
    
    # Volume Adjusted Momentum Indicator (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=20).sum()
    moving_average_volume = df['volume'].rolling(window=20).mean()
    vami = volume_adjusted_price_change / moving_average_volume
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (corrected_momentum * vwap_oscillator) + volume_weighted_intraday_spread + vami
    
    return final_alpha_factor
