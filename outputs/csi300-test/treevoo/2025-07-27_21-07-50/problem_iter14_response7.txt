import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the relative strength
    df['rs'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the average true range
    df['atr'] = (df['high'] - df['low'])
    
    # Calculate the money flow
    df['mf'] = df['amount'] / df['volume']
    
    # Compute the momentum
    df['momentum'] = df['close'] - df['close'].shift(1)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (df['rs'] + df['atr'] + df['mf'] + df['momentum']).rolling(window=5).mean()
    
    return alpha_factor.dropna()
