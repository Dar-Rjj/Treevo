```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Momentum
    short_term_lookback = 5
    df['short_term_momentum'] = df['close'].pct_change(short_term_lookback)
    
    # Integrate Volume into Short-Term Momentum
    df['volume_adjusted_short_term_momentum'] = df['short_term_momentum'] * np.sqrt(df['volume'])
    
    # Long-Term Momentum
    long_term_lookback = 20
    df['long_term_momentum'] = df['close'].pct_change(long_term_lookback)
    
    # Compute Price Momentum
    df['price_momentum'] = df['short_term_momentum'] - df['long_term_momentum']
    
    # Calculate Price Volatility
    df['volatility'] = df['close'].pct_change().rolling(window=short_term_lookback).std()
    
    # Combine Adjusted Momentum and Volatility
    df['corrected_momentum'] = df['volume_adjusted_short_term_momentum'] / df['volatility']
    
    # Volume Weighted Average Price (VWAP) Oscillator
    df['daily_vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap_20_sma'] = df['daily_vwap'].rolling(window=20).mean()
    df['vwap_oscillator'] = df['daily_vwap'] - df['vwap_20_sma']
    
    # Enhance VWAP Oscillator
    df['vwap_10_ema'] = df['daily_vwap'].ewm(span=10, adjust=False).mean()
    df['adjusted_vwap_oscillator'] = df['vwap_oscillator'] - (df['vwap_20_sma'] - df['vwap_10_ema'])
    
    # Synthesize Final Alpha Factor
    df['alpha_factor'] = df['corrected_momentum'] * df['adjusted_vwap_oscillator']
    
    return df['alpha_factor'].dropna()
```
