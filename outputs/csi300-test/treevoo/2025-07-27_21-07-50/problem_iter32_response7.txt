import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Base Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_adjusted_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Incorporate Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / (high_low_spread)
    
    # Adjust for Price and Volume Volatility
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].rolling(window=20).std()
    corrected_factor = scaled_log_return - price_volatility - volume_volatility
    
    # Apply Trend Adjustment
    long_term_trend = df['close'].rolling(window=200).mean()
    short_term_trend = df['close'].ewm(span=20).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjustment = np.where(trend_difference > 0, 1.5, 0.5)
    corrected_factor = corrected_factor * trend_adjustment
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = (df['close'] - df['open'])
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    ma_sqrt_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = scaled_log_return + (volume_weighted_price_change.sum() / ma_sqrt_volume)
    
    # Intraday Momentum Component
    momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].rolling(window=20).std()
    momentum_5 = (df['close'] - df['close'].shift(5)) / df['close'].rolling(window=5).std()
    momentum_component = momentum_20 + 0.5 * momentum_5
    
    # Final Alpha Combination
    final_alpha = volume_adjusted_spreads + scaled_log_return + corrected_factor + vami + momentum_component
    
    return final_alpha
