```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_volatility=20, lookback_trend_long=60, lookback_trend_short=10, sma_period=21, reversal_days=5, pos_adj_factor=1.2, neg_adj_factor=0.8):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Enhance Volatility Correction
    price_volatility = df['close'].pct_change().rolling(window=lookback_volatility).std()
    volume_volatility = df['volume'].rolling(window=lookback_volatility).std()
    corrected_intraday_momentum = volume_adjusted_intraday_return - price_volatility - volume_volatility
    
    # Incorporate Trend Adjustment
    long_term_trend = df['close'].rolling(window=lookback_trend_long).mean()
    short_term_trend = df['close'].ewm(span=lookback_trend_short, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjusted_factor = np.where(trend_difference > 0, corrected_intraday_momentum * pos_adj_factor, corrected_intraday_momentum * neg_adj_factor)
    
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spreads = (high_low_spread + open_close_spread) * df['volume']
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    vol_20d_ma = df['volume'].rolling(window=20).mean()
    vami = volume_adjusted_price_change / vol_20d_ma
    
    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']
    
    # Simple Moving Average (SMA) of Close
    sma_close = df['close'].rolling(window=sma_period).mean()
    
    # Composite Alpha Factor
    composite_alpha = (volume_weighted_close - sma_close) + vami
    short_term_reversal = df['close'].pct_change().rolling(window=reversal_days).sum()
    final_composite_alpha = composite_alpha + short_term_reversal + trend_adjusted_factor
    
    return final_composite_alpha
```
