```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A series of factor values indexed by (date).
    """
    # Calculate the relative difference between high and low prices
    rel_diff_high_low = (df['high'] - df['low']) / df['low']
    
    # Calculate the relative difference between close and open prices
    rel_diff_close_open = (df['close'] - df['open']) / df['open']
    
    # Calculate the relative difference between close and previous close
    rel_diff_close_prev_close = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate the relative difference between volume and previous volume
    rel_diff_volume_prev_volume = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine the factors to create a novel alpha factor
    factor = (rel_diff_high_low + rel_diff_close_open + rel_diff_close_prev_close + rel_diff_volume_prev_volume) / 4
    
    return factor
```
