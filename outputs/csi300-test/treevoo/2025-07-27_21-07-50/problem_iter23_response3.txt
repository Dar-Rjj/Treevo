```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the average price
    avg_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    
    # Calculate the price change
    price_change = df['close'].pct_change()
    
    # Calculate the volume change
    volume_change = df['volume'].pct_change()
    
    # Calculate the amount change
    amount_change = df['amount'].pct_change()
    
    # Calculate the weighted price change
    weighted_price_change = avg_price * price_change
    
    # Calculate the weighted volume change
    weighted_volume_change = df['volume'] * volume_change
    
    # Calculate the weighted amount change
    weighted_amount_change = df['amount'] * amount_change
    
    # Combine the factors to create the final alpha factor
    alpha_factor = weighted_price_change + 0.5 * weighted_volume_change + 0.3 * weighted_amount_change
    
    # Normalize the alpha factor
    alpha_factor = (alpha_factor - alpha_factor.mean()) / alpha_factor.std()
    
    return alpha_factor
```
