```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Volume-Weighted Close-Open Difference
    volume_weighted_diff = close_open_diff * df['volume']
    
    # Integrate Both Factors
    factor = high_low_spread * volume_weighted_diff
    
    return factor
```
