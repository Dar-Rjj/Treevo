import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_vol=20, long_trend=200, short_trend=50, pos_adj=1.2, neg_adj=0.8):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Return
    volume_adjusted_return = intraday_return * df['volume']
    
    # Enhance Volatility Correction
    close_returns = df['close'].pct_change()
    price_volatility = close_returns.rolling(window=lookback_vol).std()
    volume_volatility = df['volume'].rolling(window=lookback_vol).std()
    
    corrected_momentum = (volume_adjusted_return - price_volatility - volume_volatility)
    
    # Incorporate Trend Adjustment
    long_term_trend = df['close'].rolling(window=long_trend).mean()
    short_term_trend = df['close'].ewm(span=short_trend, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    
    # Adjust the corrected intraday momentum factor based on trend difference
    adjusted_factor = corrected_momentum.copy()
    adjusted_factor[trend_difference > 0] *= pos_adj
    adjusted_factor[trend_difference <= 0] *= neg_adj
    
    return adjusted_factor

# Assuming df is a pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
