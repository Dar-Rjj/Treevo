import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price within the high-low range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])

    # Calculate the weighted average of the open and close prices, with weights based on volume
    df['weighted_avg_price'] = (df['open'] * df['volume'].shift(1) + df['close'] * df['volume']) / (df['volume'].shift(1) + df['volume'])

    # Calculate the trend in the weighted average price over a 5-day window
    df['trend_weighted_avg'] = df['weighted_avg_price'].pct_change(periods=5)

    # Combine the two signals: close_position and trend_weighted_avg
    df['alpha_factor'] = df['close_position'] * df['trend_weighted_avg']

    return df['alpha_factor']
