```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Medium-Term Trend
    medium_term_trend = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volatility and Mean Reversion Indicator
    avg_daily_range = (df['high'] - df['low']).rolling(window=20).mean()
    std_close_returns = df['close'].pct_change().rolling(window=20).std()
    mean_reversion = (df['close'] - df['close'].rolling(window=20).mean()) / df['close'].rolling(window=20).std()
    
    # Institutional Activity Indicator
    vwap = (df['close'] * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Liquidity Proxy
    avg_daily_amount = df['amount'].rolling(window=20).mean()
    
    # Corrected Intraday Momentum with Adjustments
    intraday_return_vol_weighted = (df['close'] - df['open']) * np.sqrt(df['volume'])
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].pct_change().rolling(window=20).std()
    adjusted_intraday_momentum = intraday_return_vol_weighted - price_volatility - volume_volatility
    
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Composite Alpha Factor Components
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close_ema_diff = (df['close'] - ema_21) * df['volume']
    medium_term_reversal = df['close'].pct_change().rolling(window=10).sum()
    
    # Combine Core Factors
    core_factors = (
        short_term_momentum +
        medium_term_trend +
        mean_reversion +
        vwap +
        avg_daily_amount +
        adjusted_intraday_momentum +
        high_low_spread_ratio +
        volume_weighted_close_open_diff +
        volume_weighted_close_ema_diff +
        medium_term_reversal
    )
    
    # Apply Long-Term and Short-Term Trend Adjustment
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    trend_adjustment = np.where(
        (short_term_trend - long_term_trend) > 0,
        1.5,
        np.where((short_term_trend - long_term_trend) < 0, 0.5, 1.0)
    )
    final_alpha_factor = core_factors * trend_adjustment
    
    return final_alpha_factor
```
