import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Momentum Component: Lagged Intraday Return * Current Intraday Return
    df['Lagged_Intraday_Return'] = df['Intraday_Return'].shift(1)
    df['Momentum_Component'] = df['Intraday_Return'] * df['Lagged_Intraday_Return']
    
    # Reversal Component: Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(1) - df['open']) / df['open']
    df['Current_Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    df['Reversal_Component'] = df['Close_to_Open_Return'] * df['Current_Close_to_Open_Return']
    
    # Interaction of Momentum and Reversal
    df['Momentum_Reversal_Interaction'] = df['Momentum_Component'] * df['Reversal_Component']
    
    # Volume Adjusted Factor
    df['Volume_Adjusted_Factor'] = (df['Momentum_Reversal_Interaction'] / 
                                    (df['volume'] / df['volume'].rolling(window=20).mean()))
    
    return df['Volume_Adjusted_Factor']

# Example usage:
# Assuming df is a pandas DataFrame with the columns 'open', 'high', 'low', 'close', 'amount', 'volume' and index is 'date'
