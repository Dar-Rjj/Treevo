```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum Factor
    df['10_day_return'] = df['close'].pct_change(10)
    df['20_day_return'] = df['close'].pct_change(20)
    df['50_day_return'] = df['close'].pct_change(50)
    df['momentum_score'] = 0.4 * df['10_day_return'] + 0.35 * df['20_day_return'] + 0.25 * df['50_day_return']

    # Price-Volume Trend
    df['avg_vol_10'] = df['volume'].rolling(window=10).mean()
    df['price_change_pct'] = df['close'].pct_change()
    df['price_vol_indicator'] = df['price_change_pct'] * (df['volume'] / df['avg_vol_10'])

    # Volatility Factor
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=30).std()

    # Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))

    # Money Flow Index (MFI)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['raw_money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df['raw_money_flow'].where(df['typical_price'] > df['typical_price'].shift(1), 0)
    df['negative_money_flow'] = df['raw_money_flow'].where(df['typical_price'] < df['typical_price'].shift(1), 0)
    positive_flow = df['positive_money_flow'].rolling(window=14).sum()
    negative_flow = df['negative_money_flow'].rolling(window=14).sum()
    money_flow_ratio = positive_flow / negative_flow
    df['mfi'] = 100 - (100 / (1 + money_flow_ratio))

    # High-Low Range Factor
    df['high_low_range'] = df['high'] - df['low']
    df['avg_high_low_range_30'] = df['high_low_range'].rolling(window=30).mean()
    df['high_low_range_factor'] = df['high_low_range'] / df['avg_high_low_range_30']

    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        0.3 * df['momentum_score'] 
        + 0.2 * df['price_vol_indicator'] 
        + 0.1 * df['volatility'] 
        + 0.2 * (100 - df['rsi'])  # RSI is inverted for the purpose of the alpha factor
        + 0.1 * (100 - df['mfi'])  # MFI is inverted for the purpose of the alpha factor
        + 0.1 * df['high_low_range_factor']
    )

    return df['alpha_factor']
```
