import pandas as pd
import numpy as np

def heuristics_v2(df, lookback=20, short_term_window=5):
    # Compute Base Intraday Return
    base_intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume-Adjusted Spreads
    volume_adjusted_spread = (df['high'] - df['low'] + df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Incorporate Scaled Log-Return
    scaled_log_return = np.log(df['close'] / df['open']) / (df['high'] - df['low'])
    
    # Adjust for Price and Volume Volatility
    close_std = df['close'].rolling(window=lookback).std()
    volume_std = df['volume'].rolling(window=lookback).std()
    adjusted_factor = scaled_log_return - close_std - volume_std
    
    # Apply Trend Adjustment
    long_term_ma = df['close'].rolling(window=lookback).mean()
    short_term_ema = df['close'].ewm(span=short_term_window, adjust=False).mean()
    trend_difference = short_term_ema - long_term_ma
    trend_adjusted_factor = adjusted_factor + trend_difference
    
    # Combine Volume-Weighted Price Movement
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    volume_ma = np.sqrt(df['volume']).rolling(window=lookback).mean()
    combined_alpha = scaled_log_return + (volume_weighted_price_change.rolling(window=lookback).sum() / volume_ma)
    
    # Intraday Momentum Scaling
    twenty_day_momentum = (df['close'] - df['close'].shift(20))
    twenty_day_volatility = df['close'].rolling(window=20).std()
    scaled_twenty_day_momentum = twenty_day_momentum / twenty_day_volatility
    five_day_momentum = (df['close'] - df['close'].shift(short_term_window))
    five_day_volatility = df['close'].rolling(window=short_term_window).std()
    scaled_five_day_momentum = five_day_momentum / five_day_volatility
    momentum_scaled_factor = scaled_twenty_day_momentum + scaled_five_day_momentum
    
    # Introduce Intraday Volume-Ratio Factor
    intraday_volume_ratio = df['volume'] / df['volume'].rolling(window=lookback).mean()
    final_factor = trend_adjusted_factor * intraday_volume_ratio
    
    return final_factor
