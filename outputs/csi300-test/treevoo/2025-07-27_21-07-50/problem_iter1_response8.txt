```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Moving Average of Close Prices (10-day)
    short_term_ma = df['close'].rolling(window=10).mean()
    
    # Calculate Long-Term Moving Average of Close Prices (50-day)
    long_term_ma = df['close'].rolling(window=50).mean()
    
    # Compute Price Momentum
    price_momentum = long_term_ma - short_term_ma
    
    # Calculate Daily VWAP
    vwap_daily = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate 20-day Simple Moving Average of VWAP
    vwap_20_day_sma = vwap_daily.rolling(window=20).mean()
    
    # Calculate VWAP Oscillator
    vwap_oscillator = vwap_daily - vwap_20_day_sma
    
    # Combine Price Momentum and VWAP Oscillator
    combined_factor = price_momentum * vwap_oscillator
    
    return combined_factor
```
