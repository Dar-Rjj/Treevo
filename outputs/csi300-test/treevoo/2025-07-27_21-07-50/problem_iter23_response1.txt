import pandas as pd

def heuristics_v2(df):
    """
    This function computes a novel alpha factor based on the given price and volume data.
    The factor is calculated as the ratio of the difference between close and open prices to the average true range (ATR) over a 14-day period,
    multiplied by the relative volume (volume divided by its 30-day moving average).
    
    :param df: A pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as (date)
    :return: A pandas Series representing the alpha factor values indexed by (date)
    """
    # Calculate the Average True Range (ATR)
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the relative volume
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    df['relative_volume'] = df['volume'] / df['avg_volume_30']
    
    # Calculate the alpha factor
    df['alpha_factor'] = ((df['close'] - df['open']) / df['atr_14']) * df['relative_volume']
    
    return df['alpha_factor']
