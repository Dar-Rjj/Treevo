```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Momentum
    short_term_lookback = 5
    short_term_momentum = df['close'].pct_change(short_term_lookback)
    
    # Integrate Volume into Short-Term Momentum
    volume_adjusted_short_term_momentum = short_term_momentum * np.sqrt(df['volume'])
    
    # Calculate Long-Term Momentum
    long_term_lookback = 20
    long_term_momentum = df['close'].pct_change(long_term_lookback)
    
    # Compute Price Momentum
    price_momentum = short_term_momentum - long_term_momentum
    
    # Calculate Price Volatility
    price_volatility = df['close'].pct_change().rolling(window=short_term_lookback).std()
    
    # Combine Adjusted Momentum and Volatility
    corrected_momentum = volume_adjusted_short_term_momentum / price_volatility
    
    # Volume Weighted Average Price (VWAP) Oscillator
    daily_vwap = ((df['high'] + df['low'] + df['close']) / 3 * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_20_sma = daily_vwap.rolling(window=long_term_lookback).mean()
    vwap_oscillator = daily_vwap - vwap_20_sma
    
    # Enhance VWAP Oscillator
    vwap_10_ema = daily_vwap.ewm(span=10, adjust=False).mean()
    adjusted_vwap_oscillator = vwap_oscillator - vwap_10_ema
    
    # Intraday Volatility Component
    intraday_high_low_spread = df['high'] - df['low']
    intraday_open_close_spread = df['close'] - df['open']
    combined_intraday_spreads = (intraday_high_low_spread + intraday_open_close_spread) * df['volume']
    
    # Volume Adjusted Momentum Indicator (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=long_term_lookback).sum()
    moving_average_volume = df['volume'].rolling(window=long_term_lookback).mean()
    vami = volume_adjusted_price_change / moving_average_volume
    
    # Synthesize Final Alpha Factor
    final_alpha_factor = (corrected_momentum * adjusted_vwap_oscillator) + combined_intraday_spreads + vami
    
    return final_alpha_factor
```
