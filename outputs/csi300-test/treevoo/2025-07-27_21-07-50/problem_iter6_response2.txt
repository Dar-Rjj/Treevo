```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Change
    df['Volume_Adjusted_Price_Change'] = df['Daily_Price_Change'] * df['volume']
    df['Volume_Adjusted_Price_Change_20D'] = df['Volume_Adjusted_Price_Change'].rolling(window=20).sum()
    
    # Calculate Intraday Return
    df['Intraday_Return'] = df['high'] - df['low']
    
    # Momentum Component
    df['Lagged_Intraday_Return'] = df['Intraday_Return'].shift(1)
    df['Momentum_Component'] = df['Lagged_Intraday_Return'] + df['Intraday_Return']
    
    # Reversal Component
    df['Close_to_Open_Return'] = df['close'] - df['open']
    df['Lagged_Close_to_Open_Return'] = df['Close_to_Open_Return'].shift(1)
    df['Reversal_Component'] = df['Lagged_Close_to_Open_Return'] + df['Close_to_Open_Return']
    
    # Momentum and Reversal Interaction
    df['Momentum_and_Reversal_Interaction'] = df['Momentum_Component'] * df['Reversal_Component']
    
    # Volume Adjusted Factor
    df['Volume_Moving_Average'] = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Factor'] = df['volume'] / df['Volume_Moving_Average']
    
    # Enhanced Momentum Component
    df['Intraday_Return_EMA_20D'] = df['Intraday_Return'].ewm(span=20, adjust=False).mean()
    df['Enhanced_Momentum_Component'] = df['Intraday_Return'] - df['Intraday_Return_EMA_20D']
    
    # Enhanced Reversal Component
    df['Close_to_Open_Return_EMA_20D'] = df['Close_to_Open_Return'].ewm(span=20, adjust=False).mean()
    df['Enhanced_Reversal_Component'] = df['Close_to_Open_Return'] - df['Close_to_Open_Return_EMA_20D']
    
    # Enhanced Momentum and Reversal Interaction
    df['Enhanced_Momentum_and_Reversal_Interaction'] = df['Enhanced_Momentum_Component'] * df['Enhanced_Reversal_Component']
    
    # Final VAIMRF+
    df['VAIMRF+'] = (df['Volume_Adjusted_Price_Change_20D'] * df['Enhanced_Momentum_and_Reversal_Interaction']) / df['Volume_Adjusted_Factor']
    
    return df['VAIMRF+']
```
