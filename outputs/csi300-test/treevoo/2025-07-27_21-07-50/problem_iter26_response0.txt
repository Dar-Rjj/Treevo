```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / ema_volume_20
    
    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']
    
    # Weighted Moving Average (WMA) of Volume-Weighted Close
    wma_period = 21
    weights = np.arange(1, wma_period+1)
    wma_volume_weighted_close = volume_weighted_close.rolling(window=wma_period).apply(lambda x: np.dot(x, weights) / weights.sum(), raw=True)
    
    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=5).sum()
    
    # Composite Alpha Factor
    composite_alpha_factor = (volume_weighted_close - wma_volume_weighted_close
                              + vami
                              + short_term_reversal
                              + intraday_volatility)
    
    return composite_alpha_factor
```
