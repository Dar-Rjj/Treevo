import pandas as pd

def heuristics_v2(df):
    # Calculate the momentum factor using closing prices
    returns_10 = df['close'].pct_change(10)
    returns_20 = df['close'].pct_change(20)
    returns_50 = df['close'].pct_change(50)
    momentum_score = 0.5 * returns_10 + 0.3 * returns_20 + 0.2 * returns_50

    # Calculate the price-volume trend
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    pct_change_close = df['close'].pct_change()
    price_volume_indicator = pct_change_close * (df['volume'] / avg_volume_10)

    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))

    # Combine all factors into a single alpha factor
    alpha_factor = momentum_score + pvt - volatility + rsi + mfi

    return alpha_factor
