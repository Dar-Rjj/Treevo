```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    # Exponential Moving Average (EMA) of Close prices
    ema_period = 21
    ema = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha = volume_weighted_close - ema
    
    # Add medium-term reversal
    daily_returns = df['close'].pct_change()
    m_day_returns = daily_returns.rolling(window=10).sum()
    composite_alpha += m_day_returns
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Intraday Volatility
    true_range = df[['high', 'low', df['close'].shift(1)]].max(axis=1) - df[['high', 'low', df['close'].shift(1)]].min(axis=1)
    scaled_log_return = intraday_log_return / true_range
    
    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=21).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return = scaled_log_return + volume_change
    
    # Final Integration
    final_factor = integrated_factor + composite_alpha + adjusted_return
    
    # Enhance Volatility Correction
    price_volatility = df['close'].pct_change().rolling(window=21).std()
    volume_volatility = df['volume'].rolling(window=21).std()
    corrected_intraday_momentum = (intraday_log_return * np.sqrt(df['volume'])) - price_volatility - volume_volatility
    
    # Incorporate Trend Adjustment
    long_term_trend = df['close'].rolling(window=200).mean()
    short_term_trend = df['close'].ewm(span=50, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    adjustment_factor = np.where(trend_difference > 0, 1.1, 0.9)
    corrected_factor = corrected_intraday_momentum * adjustment_factor
    
    # Combine with the final factor
    final_factor += corrected_factor
    
    return final_factor
```
