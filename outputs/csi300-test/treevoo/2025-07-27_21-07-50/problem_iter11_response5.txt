import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Momentum
    volume_weighted_momentum = intraday_momentum * df['volume']
    
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spreads = high_low_spread + open_close_spread
    weighted_volatility = combined_spreads * np.sqrt(df['volume'])
    
    # Intraday Log-Return Component
    log_return = np.log(df['close'] / df['open'])
    high_low_range = df['high'] - df['low']
    scaled_log_return = log_return / high_low_range
    
    # Volume Impact and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    rolling_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=20).sum()
    moving_avg_sqrt_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = rolling_volume_adjusted_price_change / moving_avg_sqrt_volume
    
    # Composite Alpha Factor
    close_sma = df['close'].rolling(window=20).mean()
    short_term_reversal = df['close'].diff().shift(1)
    adjusted_momentum = volume_weighted_momentum + short_term_reversal
    composite_factor = (df['close'] - close_sma) + adjusted_momentum + scaled_log_return + vami - (np.sqrt(df['volume']) - np.sqrt(df['volume'].shift(1)))
    
    return pd.Series(composite_factor, index=df.index, name='alpha_factor')
