import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20, extended_lookback_period=60, short_lookback_period=5, positive_adjustment_factor=1.2, negative_adjustment_factor=0.8, sentiment_threshold=1.0):
    # Compute Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * np.sqrt(df['volume'])
    
    # Enhance Volatility Correction
    df['price_volatility'] = df['close'].pct_change().rolling(window=lookback_period).std()
    df['volume_volatility'] = df['volume'].rolling(window=lookback_period).std()
    df['corrected_intraday_momentum'] = df['volume_adjusted_intraday_return'] - df['price_volatility'] - df['volume_volatility']
    
    # Incorporate Intraday Volatility Component
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['close'] - df['open']
    df['summed_spreads'] = df['high_low_spread'] + df['open_close_spread']
    df['weighted_by_volume'] = df['summed_spreads'] * np.sqrt(df['volume'])
    
    # Adjust by Trend
    df['long_term_trend'] = df['close'].rolling(window=extended_lookback_period).mean()
    df['short_term_trend'] = df['close'].ewm(span=short_lookback_period).mean()
    df['trend_difference'] = df['long_term_trend'] - df['short_term_trend']
    
    def apply_trend_adjustment(row):
        if row['trend_difference'] > 0:
            return row['corrected_intraday_momentum'] * positive_adjustment_factor
        else:
            return row['corrected_intraday_momentum'] * negative_adjustment_factor
    
    df['trend_adjusted_factor'] = df.apply(apply_trend_adjustment, axis=1)
    
    # Incorporate Sentiment Indicator
    df['daily_range'] = df['high'] - df['low']
    df['atr'] = df['close'].rolling(window=lookback_period).apply(lambda x: np.mean(np.abs(x - x.shift())))
    df['intraday_sentiment'] = df['daily_range'] / df['atr']
    
    def apply_sentiment_adjustment(row):
        if row['intraday_sentiment'] > sentiment_threshold:
            return row['trend_adjusted_factor'] * 1.1
        else:
            return row['trend_adjusted_factor'] * 0.9
    
    df['final_factor'] = df.apply(apply_sentiment_adjustment, axis=1)
    
    return df['final_factor']
