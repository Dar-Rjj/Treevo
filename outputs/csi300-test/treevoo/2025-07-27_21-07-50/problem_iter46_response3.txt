```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Medium-Term Trend
    medium_term_trend = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Volatility and Mean Reversion Indicator
    vol_mean_reversion = (df['high'] - df['low']).rolling(window=20).sum() / 20
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Price Volatility
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Volume Volatility
    volume_volatility = df['volume'].pct_change().rolling(window=20).std()
    
    # Adjusted Intraday Momentum
    adjusted_intraday_momentum = volume_weighted_intraday_return - price_volatility - volume_volatility
    
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Exponential Moving Average (EMA) of Close prices (21 days)
    ema_21 = df['close'].ewm(span=21, adjust=False).mean()
    
    # EMA-Adjusted Volume-Weighted Close
    ema_adjusted_volume_weighted_close = (df['close'] - ema_21) * df['volume']
    
    # Medium-Term Reversal
    medium_term_reversal = df['close'].pct_change().rolling(window=10).sum()
    
    # Integrated Factors
    integrated_factors = high_low_spread_ratio * volume_weighted_close_open_diff + ema_adjusted_volume_weighted_close + medium_term_reversal
    
    # Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # True Range
    true_range = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))], axis=1).max(axis=1)
    
    # Scaled Intraday Volatility
    scaled_intraday_volatility = true_range * intraday_log_return
    
    # Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Adjusted Return
    adjusted_return = intraday_log_return + scaled_intraday_volatility + volume_change
    
    # Core Factors
    core_factors = (short_term_momentum + 
                    medium_term_trend + 
                    vol_mean_reversion + 
                    adjusted_intraday_momentum + 
                    integrated_factors + 
                    adjusted_return)
    
    # Trend-Adjusted Final Alpha Factor
    short_term_trend = df['close'].ewm(span=20, adjust=False).mean()
    long_term_trend = df['close'].rolling(window=100).mean()
    trend_adjustment = (short_term_trend - long_term_trend).apply(lambda x: 1.5 if x > 0 else (0.5 if x < 0 else 1))
    
    final_alpha_factor = core_factors * trend_adjustment
    
    return final_alpha_factor
```
