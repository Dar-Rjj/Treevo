```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Volume-Weighted Close-Open Difference
    volume_weighted_diff = (df['close'] - df['open']) * df['volume']
    
    # Average of Both Factors
    avg_of_both_factors = (high_low_spread_ratio + volume_weighted_diff) / 2
    
    # Volume-Adjusted EMA Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    volume_adjusted_ema_factor = volume_weighted_close - ema_close
    
    # Medium-Term Reversal Factor
    daily_returns = df['close'].pct_change()
    medium_term_reversal = daily_returns.rolling(window=10).mean()
    
    # Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # True Range
    true_range = df['high'] - df['low']
    
    # Scaled Intraday Log-Return
    scaled_intraday_log_return = intraday_log_return / true_range.shift(1)
    
    # Volume Impact
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_impact = volume_change * scaled_intraday_log_return
    
    # Final Integration
    final_factor = (avg_of_both_factors + volume_adjusted_ema_factor + 
                    medium_term_reversal + volume_impact) / 4
    
    return final_factor
```
