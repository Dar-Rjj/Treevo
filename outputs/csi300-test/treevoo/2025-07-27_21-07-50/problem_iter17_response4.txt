import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=20, extended_lookback_period=100, short_lookback_period=10, positive_adjustment_factor=1.5, negative_adjustment_factor=0.5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Return
    volume_adjusted_intraday_return = intraday_return * np.sqrt(df['volume'])
    
    # Enhance Volatility Correction
    price_volatility = df['close'].pct_change().rolling(window=lookback_period).std()
    volume_volatility = df['volume'].pct_change().rolling(window=lookback_period).std()
    corrected_intraday_momentum = volume_adjusted_intraday_return - price_volatility - volume_volatility
    
    # Incorporate Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['open'] - df['close']
    combined_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open']) / (df['high'] - df['low'])
    
    # Incorporate Trend Adjustment
    long_term_trend = df['close'].rolling(window=extended_lookback_period).mean()
    short_term_trend = df['close'].ewm(span=short_lookback_period, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    
    # Adjust the corrected intraday momentum factor based on trend difference
    adjusted_factor = np.where(trend_difference > 0, corrected_intraday_momentum * positive_adjustment_factor, 
                                np.where(trend_difference < 0, corrected_intraday_momentum * negative_adjustment_factor, 
                                         corrected_intraday_momentum))
    
    # Combine all components to form the final alpha factor
    alpha_factor = adjusted_factor + combined_spreads + intraday_log_return
    
    return alpha_factor
