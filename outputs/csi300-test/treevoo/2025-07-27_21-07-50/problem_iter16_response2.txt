```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spread = high_low_spread + open_close_spread
    intraday_volatility = combined_spread * np.sqrt(df['volume'])

    # Intraday Log-Return Component
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_log_return = intraday_log_return / (df['high'] - df['low'])

    # Volume Impact and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change.rolling(window=20).sum()
    
    sqrt_volume = np.sqrt(df['volume'])
    moving_avg_sqrt_volume = sqrt_volume.rolling(window=20).mean()
    
    volume_change = sqrt_volume - sqrt_volume.shift(1)
    volume_impact_adjustment = sqrt_volume - moving_avg_sqrt_volume
    
    normalized_volume_adjusted_price_change = volume_adjusted_price_change / moving_avg_sqrt_volume

    # Combine all components
    vami = normalized_volume_adjusted_price_change + volume_change
    alpha_factor = scaled_log_return + vami - volume_impact_adjustment

    return alpha_factor
```
