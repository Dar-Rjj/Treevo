```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily volume change
    df['volume_change'] = df['volume'].diff()

    # Calculate 20-day moving average of volume
    df['vol_20ma'] = df['volume'].rolling(window=20).mean()
    
    # Determine the ratio of current day's volume to its 20-day moving average
    df['vol_ratio'] = df['volume'] / df['vol_20ma']

    # Calculate 5, 10, and 20-day simple moving averages of close prices
    df['close_5ma'] = df['close'].rolling(window=5).mean()
    df['close_10ma'] = df['close'].rolling(window=10).mean()
    df['close_20ma'] = df['close'].rolling(window=20).mean()
    
    # Compute the difference between 5-day and 20-day SMAs
    df['sma_diff'] = df['close_5ma'] - df['close_20ma']
    
    # Calculate the daily price change
    df['price_change'] = df['close'].diff()
    
    # Consider the percentage change in closing price
    df['price_change_pct'] = df['close'].pct_change()
    
    # Calculate the high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Use the ratio of today's high-low range to yesterday's
    df['high_low_ratio'] = df['high_low_range'] / df['high_low_range'].shift(1)
    
    # Combine the daily price change with the daily volume change
    df['price_vol_trend'] = df['price_change'] * df['volume_change']
    
    # Calculate On-Balance Volume (OBV)
    obv = [0]
    for i in range(1, len(df)):
        if df['close'].iloc[i] > df['close'].iloc[i-1]:
            obv.append(obv[-1] + df['volume'].iloc[i])
        elif df['close'].iloc[i] < df['close'].iloc[i-1]:
            obv.append(obv[-1] - df['volume'].iloc[i])
        else:
            obv.append(obv[-1])
    df['obv'] = obv
    
    # Construct an alpha factor using a combination of the calculated features
    df['alpha_factor'] = (df['vol_ratio'] * df['sma_diff'] * df['price_vol_trend'] * df['obv'])
    
    return df['alpha_factor']
```
