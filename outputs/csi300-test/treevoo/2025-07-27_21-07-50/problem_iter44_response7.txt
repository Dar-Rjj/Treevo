```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread + volume_weighted_diff
    
    # Composite Alpha Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha = volume_weighted_close - ema_close
    
    # Medium-term reversal
    daily_returns = df['close'].pct_change()
    medium_term_reversal = daily_returns.rolling(window=10).mean()
    
    composite_alpha += medium_term_reversal
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    intraday_volatility = true_range * df['volume']
    
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return = intraday_log_return + intraday_volatility + volume_change
    
    # Intraday Volatility
    high_low_spread_vol = (df['high'] - df['low']) * df['volume']
    open_close_spread_vol = (df['close'] - df['open']) * df['volume']
    intraday_volatility = high_low_spread_vol + open_close_spread_vol
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    vami_ema = volume_adjusted_price_change.ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / vami_ema
    
    # Short-Term Reversal
    short_term_reversal = daily_returns.rolling(window=5).mean()
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha + adjusted_return + vami + short_term_reversal + intraday_volatility
    
    # Improved Composite Alpha Factor
    sma_close = df['close'].rolling(window=21).mean()
    improved_composite_alpha = volume_weighted_close - sma_close + vami + short_term_reversal + intraday_volatility
    
    return final_factor
```
