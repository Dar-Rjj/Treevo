```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Price Momentum
    momentum = df['close'] - df['close'].shift(1)
    
    # Adjust Momentum by Intraday Volatility
    adjusted_momentum = momentum / intraday_volatility
    
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open_diff = close_open_diff * df['volume']
    
    # Integrate Adjusted Momentum with High-Low Spread Ratio
    integrated_momentum = adjusted_momentum * high_low_spread_ratio
    
    # Integrate Combined Factors with Volume-Weighted Close-Open Difference
    alpha_factor = integrated_momentum * volume_weighted_close_open_diff
    
    return alpha_factor
```
