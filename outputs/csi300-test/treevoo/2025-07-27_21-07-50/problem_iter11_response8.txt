```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day average money flow
    n = 14
    positive_money_flow = money_flow.rolling(window=n).apply(lambda x: sum(x[x > 0]), raw=True)
    negative_money_flow = abs(money_flow.rolling(window=n).apply(lambda x: sum(x[x < 0]), raw=True))
    
    # Calculate the money flow ratio
    money_ratio = positive_money_flow / negative_money_flow
    
    # Calculate the money flow index
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Calculate the 14-day EMA of the MFI
    mfi_ema = mfi.ewm(span=n, adjust=False).mean()
    
    # Calculate the alpha factor
    alpha_factor = mfi - mfi_ema
    
    return alpha_factor
```
