import pandas as pd

def heuristics_v2(df, sma_period=21, reversal_days=5):
    # Calculate Simple Moving Average (SMA) of Close prices
    df['SMA'] = df['close'].rolling(window=sma_period).mean()
    
    # Calculate Volume-Weighted Close Price
    df['Volume_Weighted_Close'] = df['close'] * df['volume']
    
    # Calculate Short-Term Reversal
    daily_returns = df['close'] / df['open'] - 1
    df['Short_Term_Reversal'] = daily_returns.rolling(window=reversal_days).sum()
    
    # Composite Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Weighted_Close'] - df['SMA']) + df['Short_Term_Reversal']
    
    return df['Alpha_Factor']
