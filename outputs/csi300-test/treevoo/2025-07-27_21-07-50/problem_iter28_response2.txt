```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between today's close and yesterday's close
    df['Close_Diff'] = df['close'].diff()
    
    # Calculate the 5-day moving average of closes
    df['5Day_MA'] = df['close'].rolling(window=5).mean()
    # Calculate the difference between today's close and the 5-day moving average of closes
    df['Close_5Day_MA_Diff'] = df['close'] - df['5Day_MA']
    
    # Determine the range of daily price movement
    df['Daily_Range'] = df['high'] - df['low']
    
    # Calculate the True Range
    df['True_Range'] = df[['high' - 'low', abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    # Calculate the 14-day Average True Range
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Calculate the 20-day average volume
    df['20Day_Avg_Volume'] = df['volume'].rolling(window=20).mean()
    # Calculate the ratio of today's volume to the 20-day average volume
    df['Volume_Ratio'] = df['volume'] / df['20Day_Avg_Volume']
    # Determine if there is a significant increase or decrease in volume compared to the previous day
    df['Volume_Change_Ratio'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate the typical price
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    # Calculate the money flow
    df['Money_Flow'] = df['Typical_Price'] * df['volume']
    # Calculate the positive and negative money flow
    df['Positive_Money_Flow'] = df['Money_Flow'].where(df['Money_Flow'] > 0, 0)
    df['Negative_Money_Flow'] = df['Money_Flow'].where(df['Money_Flow'] < 0, 0)
    # Calculate the 14-day sum of positive and negative money flow
    df['Sum_Positive_Money_Flow_14'] = df['Positive_Money_Flow'].rolling(window=14).sum()
    df['Sum_Negative_Money_Flow_14'] = df['Negative_Money_Flow'].rolling(window=14).sum()
    # Calculate the Money Flow Index
    df['MFI_14'] = df['Sum_Positive_Money_Flow_14'] / df['Sum_Negative_Money_Flow_14']
    
    # Combine the factors into a single alpha factor
    alpha_factor = (
        df['Close_Diff'] +
        df['Close_5Day_MA_Diff'] +
        df['Daily_Range'] +
        df['ATR_14'] +
        df['Volume_Ratio'] +
        df['Volume_Change_Ratio'] +
        df['MFI_14']
    )
    
    return alpha_factor
```
