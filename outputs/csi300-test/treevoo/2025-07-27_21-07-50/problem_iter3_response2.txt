```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=30):
    # Calculate Intraday Log-Return
    df['log_return'] = np.log(df['close']) - np.log(df['open'])
    
    # Calculate Intraday Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['intraday_volatility'] = df['high_low_range'] * df['log_return'].abs()
    
    # Measure Volume Impact
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Generate Adjusted Return
    df['adjusted_return'] = (df['log_return'] * df['intraday_volatility']) + df['volume_change']
    
    return df['adjusted_return']
```
