```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Calculate Exponential Moving Average (EMA) of Close prices (21-day EMA)
    ema_period = 21
    df['ema_close'] = df['close'].ewm(span=ema_period, adjust=False).mean()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']
    
    # Subtract EMA from Volume-Weighted Close
    composite_alpha_factor = volume_weighted_close - df['ema_close']
    
    # Add medium-term reversal (10-day returns)
    m_day_returns = df['close'].pct_change(10)
    composite_alpha_factor += m_day_returns
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Compute True Range
    true_range = np.maximum(
        np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))),
        abs(df['low'] - df['close'].shift(1))
    )
    
    # Scale by Log-Return
    adjusted_return = intraday_log_return * true_range
    
    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_change = (df['volume'] - df['volume'].shift(1)) / avg_volume
    adjusted_return += volume_change
    
    # Final Integration
    final_factor = integrated_factor + composite_alpha_factor + adjusted_return
    
    return final_factor
```
