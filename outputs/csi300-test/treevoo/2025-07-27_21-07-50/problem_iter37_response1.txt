```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Base Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_adjusted_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])

    # Incorporate Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread

    # Adjust for Price and Volume Volatility
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].rolling(window=20).std()
    corrected_factor = log_return / high_low_spread - price_volatility - volume_volatility

    # Apply Trend Adjustment
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].ewm(span=20).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjusted_factor = corrected_factor * np.where(trend_difference > 0, 1.1, 0.9)

    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    ma_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = scaled_log_return + volume_weighted_price_change / ma_volume

    # Intraday Momentum Component
    twenty_day_momentum = df['close'] - df['close'].shift(20)
    twenty_day_volatility = df['close'].rolling(window=20).std()
    scaled_twenty_day_momentum = twenty_day_momentum / twenty_day_volatility
    five_day_momentum = df['close'] - df['close'].shift(5)
    five_day_volatility = df['close'].rolling(window=5).std()
    scaled_five_day_momentum = five_day_momentum / five_day_volatility
    weighted_momentum = scaled_twenty_day_momentum + 0.5 * scaled_five_day_momentum

    # Introduce Intraday Volume-Ratio Factor
    intraday_volume_ratio = df['volume'] / df['volume'].rolling(window=20).mean()
    adjusted_intraday_return = intraday_return * intraday_volume_ratio

    # Final Alpha Combination
    final_alpha = (volume_adjusted_spreads + 
                   scaled_log_return + 
                   trend_adjusted_factor + 
                   vami + 
                   weighted_momentum + 
                   adjusted_intraday_return)

    return final_alpha
```
