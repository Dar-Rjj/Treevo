```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the close price with a 10-day window
    df['close_10d_mean'] = df['close'].rolling(window=10).mean()
    
    # Calculate the rolling standard deviation of the close price with a 10-day window
    df['close_10d_std'] = df['close'].rolling(window=10).std()
    
    # Calculate the normalized close price (z-score)
    df['close_zscore'] = (df['close'] - df['close_10d_mean']) / df['close_10d_std']
    
    # Calculate the price change over the last 5 days
    df['price_change_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the volume change over the last 5 days
    df['volume_change_5d'] = df['volume'].pct_change(periods=5)
    
    # Calculate the amount change over the last 5 days
    df['amount_change_5d'] = df['amount'].pct_change(periods=5)
    
    # Calculate the average true range (ATR) over a 14-day window
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x), axis=1) - df[['open', 'close']].apply(lambda x: min(x), axis=1)
    df['atr_14d'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the factor: (normalized close price * ATR) + (price change * volume change * amount change)
    df['factor'] = (df['close_zscore'] * df['atr_14d']) + (df['price_change_5d'] * df['volume_change_5d'] * df['amount_change_5d'])
    
    return df['factor']
```
