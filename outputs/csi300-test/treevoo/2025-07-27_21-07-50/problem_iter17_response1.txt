import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open_diff = close_open_diff * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Calculate Exponential Moving Average (EMA) of Close prices
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    
    # Calculate Short-Term Reversal
    daily_returns = df['close'].pct_change()
    short_term_reversal = daily_returns.rolling(window=5).sum()
    
    # Composite Alpha Factor
    composite_alpha = (df['close'] - ema_close) + short_term_reversal
    
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / (df['high'] - df['low'])
    
    # Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Volume Adjusted Price Change
    volume_adjusted_price_change = (daily_price_change * np.sqrt(df['volume'])).rolling(window=20).sum()
    
    # 20-day Moving Average of sqrt(Volume)
    ma_sqrt_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    
    # Generate Combined Alpha
    combined_alpha = scaled_log_return + (volume_adjusted_price_change / ma_sqrt_volume)
    
    # Intraday Momentum Component
    momentum = df['close'] - df['close'].shift(20)
    volatility = df['close'].rolling(window=20).std()
    scaled_momentum = momentum / volatility
    
    # Final Integration
    final_alpha = (integrated_factors + composite_alpha + combined_alpha + scaled_momentums)

    return final_alpha
