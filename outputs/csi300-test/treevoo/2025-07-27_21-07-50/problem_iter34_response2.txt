```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate short and long term SMAs
    df['SMA_short'] = df['close'].rolling(window=7).mean()
    df['SMA_long'] = df['close'].rolling(window=21).mean()
    
    # Calculate SMA difference
    df['SMA_diff'] = df['SMA_short'] - df['SMA_long']
    
    # Calculate ATR
    high_low = df['high'] - df['low']
    high_close_prev = abs(df['high'] - df['close'].shift())
    low_close_prev = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([high_low, high_close_prev, low_close_prev], axis=1).max(axis=1)
    df['ATR'] = true_range.rolling(window=14).sum() / 3
    
    # Calculate volume change
    df['Volume_change'] = (df['volume'] - df['volume'].shift()) / df['volume'].shift()
    
    # Calculate volume price signal
    df['Volume_price_signal'] = (df['close'] - df['close'].shift()) * df['Volume_change']
    
    # Calculate historical volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    
    # Calculate bid-ask spread
    df['Bid_ask_spread'] = df['high'] - df['low']
    
    # Create a composite score
    df['Composite_score'] = (df['SMA_diff'] * 0.3) + (df['Volume_price_signal'] * 0.2) + (df['Volatility'] * 0.2) + (df['Bid_ask_spread'] * 0.3)
    
    return df['Composite_score']
```
