```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['open']
    
    # Calculate Volume Adjusted Price Change over a 20-day Window
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=20).sum()
    
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Momentum Component
    lagged_intraday_return = intraday_return.shift(1)
    current_intraday_return = intraday_return
    momentum_component = lagged_intraday_return + current_intraday_return
    
    # Reversal Component
    lagged_close_to_open_return = (df['close'] - df['open']).shift(1)
    current_close_to_open_return = df['close'] - df['open']
    reversal_component = lagged_close_to_open_return + current_close_to_open_return
    
    # Momentum and Reversal Interaction
    interaction = momentum_component * reversal_component
    
    # Volume Adjusted Factor
    moving_average_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_factor = df['volume'] / moving_average_volume
    
    # Final VAIMRF
    vaimrf = (volume_adjusted_price_change * interaction) / volume_adjusted_factor
    
    return vaimrf
```
