```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open_diff = close_open_diff * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Calculate Exponential Moving Average (EMA) of Close prices
    ema_period = 21
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()
    
    # Calculate Short-Term Reversal
    daily_returns = df['close'].pct_change()
    short_term_reversal = daily_returns.rolling(window=5).sum()
    
    # Composite Alpha Factor
    composite_alpha_factor = (volume_weighted_close_open_diff - ema_close) + short_term_reversal
    
    # Calculate Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range * intraday_log_return.abs()
    
    # Measure Volume Impact
    average_volume = df['volume'].rolling(window=20).mean()
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Generate Adjusted Return
    adjusted_return = intraday_log_return * intraday_volatility + volume_change
    
    # Final Integration
    final_alpha_factor = integrated_factors + composite_alpha_factor + adjusted_return
    
    return final_alpha_factor
```
