import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spread = high_low_spread + open_close_spread
    weighted_spread = combined_spread * np.sqrt(df['volume'])
    
    # Intraday Log-Return Component
    intraday_log_return = np.log(df['close'] / df['open'])
    high_low_range = df['high'] - df['low']
    scaled_intraday_log_return = intraday_log_return / high_low_range
    
    # Volume Impact and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = (daily_price_change * np.sqrt(df['volume'])).rolling(window=20).sum()
    ma_sqrt_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    avg_sqrt_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    volume_change = np.sqrt(df['volume']) - np.sqrt(df['volume'].shift(1))
    volume_impact_adjustment = np.sqrt(df['volume']) - avg_sqrt_volume
    vami = volume_adjusted_price_change / ma_sqrt_volume
    
    # Combine all components
    combined_alpha = (scaled_intraday_log_return + vami) - volume_impact_adjustment
    
    return combined_alpha
