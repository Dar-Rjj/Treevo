```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    ema_period = 21
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close_price = df['close'] * df['volume']
    composite_alpha_factor = volume_weighted_close_price - ema_close
    
    # Add medium-term reversal
    daily_returns = df['close'].pct_change()
    m_day = 10
    medium_term_reversal = daily_returns.rolling(window=m_day).sum()
    composite_alpha_factor += medium_term_reversal
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Incorporate Intraday Volatility
    true_range = df[['high', 'low']].diff(axis=1).iloc[:, 0].abs() + \
                 (df['close'].shift(1) - df['low']).abs() + \
                 (df['close'].shift(1) - df['high']).abs()
    true_range = true_range / 3
    adjusted_return = intraday_log_return / true_range
    
    # Incorporate Volume Impact
    avg_volume_window = 20
    average_volume = df['volume'].rolling(window=avg_volume_window).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_return += volume_change / average_volume
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha_factor + adjusted_return
    
    return final_factor
```
