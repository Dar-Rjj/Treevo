```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Momentum
    df['volume_adjusted_momentum'] = df['intraday_momentum'] * df['volume']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['abs_daily_return'] = abs((df['close'] - df['close'].shift(1)) / df['close'].shift(1))
    df['intraday_volatility'] = df['abs_daily_return'].rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    df['volume_thrust'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Factors
    df['combined_factor'] = (
        0.4 * df['volume_adjusted_momentum'] + 
        0.3 * df['intraday_range'] + 
        0.2 * df['intraday_volatility'] + 
        0.1 * df['volume_thrust']
    )
    
    # Adjust for Price Volatility
    short_term_volatility = df['close'].pct_change().rolling(window=5).std()
    df['adjusted_combined_factor'] = df['combined_factor'] - short_term_volatility
    
    # Calculate Simple Moving Average (SMA) of Close prices
    df['sma_21'] = df['close'].rolling(window=21).mean()
    
    # Calculate Volume-Weighted Close Price
    df['volume_weighted_close'] = df['close'] * df['volume']
    
    # Calculate Short-Term Reversal
    df['daily_returns'] = (df['close'] - df['open']) / df['open']
    df['short_term_reversal'] = df['daily_returns'].rolling(window=5).sum()
    
    # Composite Alpha Factor
    df['composite_alpha'] = (
        (df['volume_weighted_close'] - df['sma_21']) + 
        df['short_term_reversal'] + 
        df['adjusted_combined_factor']
    )
    
    return df['composite_alpha']
```
