```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_price_vol=20, lookback_vol=20, long_term_lookback=60, short_term_lookback=10, adjustment_factor_pos=1.5, adjustment_factor_neg=0.5):
    # Compute Base Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_integrated_spread = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])

    # Incorporate Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread

    # Adjust for Price and Volume Volatility
    close_returns = df['close'].pct_change()
    price_volatility = close_returns.rolling(window=lookback_price_vol).std()
    vol_volatility = df['volume'].pct_change().rolling(window=lookback_vol).std()
    corrected_intraday_momentum = (log_return - price_volatility - vol_volatility)

    # Apply Trend Adjustment
    long_term_trend = df['close'].rolling(window=long_term_lookback).mean()
    short_term_trend = df['close'].ewm(span=short_term_lookback, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjusted_factor = corrected_intraday_momentum.where(trend_difference > 0, corrected_intraday_momentum * adjustment_factor_neg).where(trend_difference <= 0, corrected_intraday_momentum * adjustment_factor_pos)

    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    ma_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = (log_return / high_low_spread) + (volume_weighted_price_change / ma_volume)

    # Intraday Momentum Component
    momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].rolling(window=20).std()
    momentum_5 = (df['close'] - df['close'].shift(5)) / df['close'].rolling(window=5).std()
    weighted_momentum = (momentum_20 + 0.5 * momentum_5)

    # Final Alpha Combination
    alpha = (volume_integrated_spread + 
             scaled_log_return + 
             trend_adjusted_factor + 
             vami +
             weighted_momentum)
    
    return alpha
```
