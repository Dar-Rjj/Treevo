```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return Component
    intraday_return = (df['close'] - df['open']) / df['open']
    volume_adjusted_intraday_return = intraday_return * np.sqrt(df['volume'])
    
    # Enhanced Volatility Correction
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    volume_volatility = df['volume'].rolling(window=20).std()
    corrected_intraday_momentum = volume_adjusted_intraday_return - price_volatility - volume_volatility
    
    # High-Low and Open-Close Spreads
    high_low_spread = (df['high'] - df['low'])
    open_close_spread = (df['close'] - df['open'])
    combined_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open'])
    log_return_scaled = intraday_log_return / (df['high'] - df['low'])
    
    # Volume-Weighted Intraday Log-Return
    volume_weighted_log_return = log_return_scaled * np.sqrt(df['volume'])
    
    # Volume-Weighted Momentum
    daily_price_change = (df['close'] - df['open'])
    volume_adjusted_price_change = daily_price_change * np.sqrt(df['volume'])
    ma_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = volume_adjusted_price_change / ma_volume
    
    # Long-Term and Short-Term Trends
    long_term_trend = df['close'].rolling(window=50).mean()
    short_term_trend = df['close'].ewm(span=10, adjust=False).mean()
    trend_difference = short_term_trend - long_term_trend
    trend_adjusted_factor = corrected_intraday_momentum.where(trend_difference > 0, corrected_intraday_momentum * 1.2).where(trend_difference <= 0, corrected_intraday_momentum * 0.8)
    
    # Intraday Momentum Component
    momentum_20 = (df['close'] - df['close'].shift(20)) / df['close'].rolling(window=20).std()
    
    # Final Composite Alpha Factor
    alpha_factor = (corrected_intraday_momentum 
                    + vami 
                    + combined_spreads 
                    + log_return_scaled 
                    + volume_weighted_log_return 
                    + trend_adjusted_factor 
                    + momentum_20)
    
    return alpha_factor
```
