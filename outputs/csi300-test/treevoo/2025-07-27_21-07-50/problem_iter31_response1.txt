```python
import pandas as pd

def heuristics_v2(df):
    # Examine the relationship between current and past prices
    df['Close_diff'] = df['close'].diff()
    df['Close_vs_avg5'] = df['close'] - df['close'].rolling(window=5).mean()

    # Explore intraday price dynamics
    df['Intraday_range'] = df['high'] - df['low']
    df['Avg_intraday_movement'] = (df['high'] - df['low'] + 
                                   abs(df['high'] - df['close'].shift(1)) + 
                                   abs(df['low'] - df['close'].shift(1))) / 3
    df['14d_Avg_intraday_movement'] = df['Avg_intraday_movement'].rolling(window=14).mean()

    # Investigate volume trends
    df['Volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Volume_change_ratio'] = df['volume'].pct_change()

    # Consider the interaction between price and volume
    df['Money_Flow'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    df['Pos_Money_Flow'] = df['Money_Flow'].where(df['close'] > df['close'].shift(), 0)
    df['Neg_Money_Flow'] = df['Money_Flow'].where(df['close'] < df['close'].shift(), 0)
    df['Money_Flow_Index'] = df['Pos_Money_Flow'].rolling(window=14).sum() / df['Neg_Money_Flow'].rolling(window=14).sum()

    # Combine all factors into a final score
    df['Alpha_factor'] = (df['Close_diff'] + df['Close_vs_avg5'] + 
                          df['14d_Avg_intraday_movement'] + 
                          df['Volume_ratio'] + df['Volume_change_ratio'] + 
                          df['Money_Flow_Index'])

    return df['Alpha_factor']
```
