```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_log_return = intraday_log_return / high_low_spread
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * df['volume']
    sum_volume_weighted_price_change = volume_weighted_price_change.rolling(window=20).sum()
    twenty_day_ma_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Combined Alpha
    combined_alpha = scaled_log_return + (sum_volume_weighted_price_change / twenty_day_ma_volume)
    
    return combined_alpha
```
