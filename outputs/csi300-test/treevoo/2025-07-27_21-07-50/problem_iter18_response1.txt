```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) / 2 * df['volume']
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    volume_20_day_ma = df['volume'].rolling(window=20).mean()
    vami = volume_adjusted_price_change / volume_20_day_ma
    
    # Volume-Weighted Close
    volume_weighted_close = df['close'] * df['volume']
    
    # Exponential Moving Average (EMA) of Close
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    
    # Short-Term Mean Reversion
    short_term_mean_reversion = df['close'].pct_change().rolling(window=5).mean()
    
    # Composite Alpha Factor
    composite_alpha_factor = (volume_weighted_close - ema_close) + vami + short_term_mean_reversion
    
    return composite_alpha_factor
```
