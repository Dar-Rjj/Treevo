```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Intraday Log-Return Scaled by Volatility
    intraday_log_return = np.log(df['close'] / df['open'])
    scaled_intraday_log_return = intraday_log_return / high_low_spread
    
    # Volume Adjusted Price Change
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    sum_volume_adjusted_price_change = volume_adjusted_price_change.rolling(window=20).sum()
    
    # 20-day Moving Average of Volume
    ma_20_volume = df['volume'].rolling(window=20).mean()
    
    # Generate Combined Alpha
    vami = sum_volume_adjusted_price_change / ma_20_volume
    combined_alpha = scaled_intraday_log_return + vami
    
    # Intraday Momentum Component
    momentum_20 = df['close'] - df['close'].shift(20)
    std_20 = df['close'].rolling(window=20).std()
    scaled_momentum_20 = momentum_20 / std_20
    
    # Final Alpha Combination
    final_alpha = combined_alpha + scaled_momentum_20
    
    return final_alpha
```
