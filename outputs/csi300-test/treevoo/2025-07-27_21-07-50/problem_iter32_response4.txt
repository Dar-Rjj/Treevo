import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_volatility=20, lookback_trend_long=60, lookback_trend_short=10, trend_adjustment_pos=1.5, trend_adjustment_neg=0.5):
    # Compute Base Intraday Components
    intraday_return = (df['close'] - df['open']) / df['open']
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    volume_adjusted_spreads = (high_low_spread + open_close_spread) * np.sqrt(df['volume'])
    
    # Incorporate Log-Return Scaled by Volatility
    log_return = np.log(df['close'] / df['open'])
    scaled_log_return = log_return / high_low_spread
    
    # Adjust for Price and Volume Volatility
    price_volatility = df['close'].pct_change().rolling(window=lookback_volatility).std()
    volume_volatility = df['volume'].pct_change().rolling(window=lookback_volatility).std()
    corrected_intraday_momentum = volume_adjusted_spreads - price_volatility - volume_volatility
    
    # Apply Trend Adjustment
    long_term_trend = df['close'].rolling(window=lookback_trend_long).mean()
    short_term_trend = df['close'].ewm(span=lookback_trend_short, adjust=False).mean()
    trend_difference = long_term_trend - short_term_trend
    trend_adjusted_factor = corrected_intraday_momentum.where(trend_difference < 0, corrected_intraday_momentum * trend_adjustment_pos).where(trend_difference >= 0, corrected_intraday_momentum * trend_adjustment_neg)
    
    # Volume-Weighted Price Movement and Momentum Adjustment
    daily_price_change = df['close'] - df['open']
    volume_weighted_price_change = daily_price_change * np.sqrt(df['volume'])
    ma_volume = np.sqrt(df['volume']).rolling(window=20).mean()
    vami = (log_return / high_low_spread) + (volume_weighted_price_change / ma_volume)
    
    # Intraday Momentum Component
    momentum_20 = (df['close'] - df['close'].shift(20))
    scaled_momentum_20 = momentum_20 / df['close'].rolling(window=20).std()
    momentum_5 = (df['close'] - df['close'].shift(5)) / df['close'].rolling(window=5).std()
    weighted_momentum = scaled_momentum_20 + momentum_5
    
    # Final Alpha Combination
    alpha_factor = (intraday_return + volume_adjusted_spreads + scaled_log_return + 
                    trend_adjusted_factor + vami + weighted_momentum)
    
    return alpha_factor.dropna()
