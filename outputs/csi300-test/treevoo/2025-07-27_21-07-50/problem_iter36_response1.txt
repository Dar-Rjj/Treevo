import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open, high, low, close) for a smoothed price
    df['smoothed_price'] = df[['open', 'high', 'low', 'close']].mean(axis=1)
    
    # Calculate the return based on the smoothed price
    df['smoothed_return'] = df['smoothed_price'].pct_change()
    
    # Normalize the volume by dividing it by the mean of the last 30 days' volume
    df['norm_volume'] = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Calculate the relative amount - amount divided by the sum of the past 30 days' amount
    df['rel_amount'] = df['amount'] / df['amount'].rolling(window=30).sum()
    
    # Generate the alpha factor
    # This is a combination of the smoothed return and the relative amount, weighted by normalized volume
    df['alpha_factor'] = (df['smoothed_return'] * df['norm_volume']) + (df['rel_amount'] * df['norm_volume'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
