```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio * volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    # Calculate Simple Moving Average (SMA) of Close prices
    sma_period = 21
    sma = df['close'].rolling(window=sma_period).mean()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']
    
    # Subtract SMA from volume-weighted close
    composite_alpha = volume_weighted_close - sma
    
    # Add short-term reversal
    daily_returns = df['close'] / df['open'] - 1
    n_days_reversal = 5
    short_term_reversal = daily_returns.rolling(window=n_days_reversal).sum()
    composite_alpha += short_term_reversal
    
    # Generate Adjusted Return
    # Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Incorporate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility_adjusted_return = intraday_log_return / high_low_range
    
    # Incorporate Volume Impact
    avg_volume_window = 30
    avg_volume = df['volume'].rolling(window=avg_volume_window).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_impact = volume_change / avg_volume
    adjusted_return = intraday_volatility_adjusted_return + volume_impact
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha + adjusted_return
    
    return final_factor
```
