import pandas as pd

def heuristics_v2(df):
    # Calculate the log return of close price
    df['log_return'] = df['close'].apply(lambda x: np.log(x)).diff()
    
    # Calculate the average of (high - low) over 5 days
    df['avg_price_range'] = df['high'] - df['low']
    df['avg_price_range'] = df['avg_price_range'].rolling(window=5).mean()
    
    # Calculate the relative strength index (RSI) on close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Weighted sum to combine the factors
    df['factor'] = 0.4 * df['log_return'] + 0.3 * df['avg_price_range'] + 0.3 * df['rsi']
    
    # Return the alpha factor values
    return df['factor']
