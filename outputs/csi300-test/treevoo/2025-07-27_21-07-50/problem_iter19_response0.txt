```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the momentum factor using closing prices
    df['10_day_return'] = df['close'].pct_change(10)
    df['20_day_return'] = df['close'].pct_change(20)
    df['50_day_return'] = df['close'].pct_change(50)
    df['momentum_score'] = (df['10_day_return'] * 0.5 + 
                            df['20_day_return'] * 0.3 + 
                            df['50_day_return'] * 0.2)

    # Calculate the price-volume trend
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['price_change_pct'] = df['close'].pct_change()
    df['price_vol_indicator'] = df['price_change_pct'] * (df['volume'] / df['avg_volume_10'])
    df['pv_trend'] = df['price_vol_indicator'].rolling(window=20).mean()

    # Calculate the volatility factor
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility_factor'] = df['log_returns'].rolling(window=30).std()

    # Calculate the relative strength index (RSI)
    df['gain'] = df['close'].diff().clip(lower=0)
    df['loss'] = -df['close'].diff().clip(upper=0)
    avg_gain = df['gain'].rolling(window=14).mean()
    avg_loss = df['loss'].rolling(window=14).mean()
    df['rs'] = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + df['rs']))
    df['smoothed_rsi'] = df['rsi'].ewm(span=14, adjust=False).mean()

    # Calculate the money flow index (MFI)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['raw_money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = np.where(df['typical_price'] > df['typical_price'].shift(1), df['raw_money_flow'], 0)
    df['negative_money_flow'] = np.where(df['typical_price'] < df['typical_price'].shift(1), df['raw_money_flow'], 0)
    positive_money_flow_sum = df['positive_money_flow'].rolling(window=14).sum()
    negative_money_flow_sum = df['negative_money_flow'].rolling(window=14).sum()
    df['money_flow_ratio'] = positive_money_flow_sum / negative_money_flow_sum
    df['mfi'] = 100 - (100 / (1 + df['money_flow_ratio']))

    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (df['momentum_score'] + 
                          df['pv_trend'] + 
                          df['volatility_factor'] + 
                          df['smoothed_rsi'] + 
                          df['mfi'])

    return df['alpha_factor']
```
