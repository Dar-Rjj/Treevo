```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    ema_period = 21
    df['ema_close'] = df['close'].ewm(span=ema_period, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha_factor = volume_weighted_close - df['ema_close']
    
    # Add medium-term reversal
    daily_returns = df['close'].pct_change()
    m_day_returns = daily_returns.rolling(window=10).mean()
    composite_alpha_factor += m_day_returns
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Incorporate Intraday Volatility
    true_range = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    scaled_intraday_volatility = true_range * intraday_log_return
    
    # Incorporate Volume Impact
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_change = df['volume'].diff().fillna(0)
    adjusted_return = intraday_log_return + volume_change / avg_volume + scaled_intraday_volatility
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha_factor + adjusted_return
    
    # Intraday Volatility
    high_low_spread = (df['high'] - df['low']) * df['volume']
    open_close_spread = (df['close'] - df['open']) * df['volume']
    combined_spreads = high_low_spread + open_close_spread
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = daily_price_change * df['volume']
    vami_ema = volume_adjusted_price_change.ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / vami_ema
    
    # Short-Term Reversal
    short_term_reversal = daily_returns.rolling(window=5).mean()
    
    # Composite Alpha Factor
    sma_close = df['close'].rolling(window=21).mean()
    composite_alpha_factor_v2 = volume_weighted_close - sma_close + vami + short_term_reversal + combined_spreads
    
    # Combine with Final Factor
    final_factor += composite_alpha_factor_v2
    
    return final_factor
```
