```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Intraday Volatility
    df['daily_return'] = df['close'].pct_change()
    df['abs_daily_return'] = df['daily_return'].abs()
    df['intraday_volatility'] = df['abs_daily_return'].rolling(window=20).mean()
    
    # Calculate Volume Thrust
    df['volume_thrust'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Factors
    df['combined_factor'] = (df['intraday_return'] + 
                             df['intraday_range'] + 
                             df['intraday_volatility'] + 
                             df['volume_thrust'])
    
    # Weight by Volume
    df['weighted_factor'] = df['combined_factor'] * df['volume']
    
    # Return the factor values as a Series
    return df['weighted_factor']
```
