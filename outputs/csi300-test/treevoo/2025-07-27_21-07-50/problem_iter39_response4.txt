import numpy as np

def heuristics_v2(df, lookback_period=20):
    # Compute Base Intraday Components
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    df['high_low_spread'] = df['high'] - df['low']
    df['open_close_spread'] = df['close'] - df['open']
    df['volume_adjusted_spread'] = (df['high_low_spread'] + df['open_close_spread']) * np.sqrt(df['volume'])
    
    # Incorporate Log-Return Scaled by Volatility
    df['log_return'] = np.log(df['close'] / df['open'])
    df['log_return_scaled'] = df['log_return'] / df['high_low_spread']

    df['price_volatility'] = df['close'].pct_change().rolling(window=lookback_period).std()
    df['volume_volatility'] = df['volume'].rolling(window=lookback_period).std()
    df['corrected_intraday_momentum'] = df['log_return_scaled'] - df['price_volatility'] - df['volume_volatility']
    
    # Apply Trend Adjustment
    df['long_term_trend'] = df['close'].rolling(window=lookback_period * 2).mean()
    df['short_term_trend'] = df['close'].ewm(span=lookback_period // 2, adjust=False).mean()
    df['trend_difference'] = df['long_term_trend'] - df['short_term_trend']
    df['trend_adjusted_factor'] = df['corrected_intraday_momentum'] * np.sign(df['trend_difference']) * (1 + np.abs(df['trend_difference']))

    df['daily_price_change'] = df['close'] - df['open']
    df['volume_weighted_price_change'] = df['daily_price_change'] * np.sqrt(df['volume'])
