import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the ratio of today's close to VWAP
    df['close_to_vwap'] = df['close'] / df['vwap']
    
    # Calculate the 5-day standard deviation of the close-to-VWAP ratio
    df['std_close_to_vwap'] = df['close_to_vwap'].rolling(window=5).std()
    
    # Calculate the 5-day momentum of the close price
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the 5-day volatility of the high and low prices
    df['high_low_volatility'] = (df['high'] - df['low']).rolling(window=5).std()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['close_to_vwap'] - 1) * df['std_close_to_vwap'] + df['momentum_5d'] - df['high_low_volatility']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
