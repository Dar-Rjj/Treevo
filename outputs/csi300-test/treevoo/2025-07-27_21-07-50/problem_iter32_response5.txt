```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted Intraday Return
    volume_weighted_intraday_return = (df['close'] - df['open']) * np.sqrt(df['volume'])
    
    # Price Volatility
    price_volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Volume Volatility
    volume_volatility = df['volume'].pct_change().rolling(window=20).std()
    
    # Corrected Intraday Momentum
    corrected_intraday_momentum = volume_weighted_intraday_return - price_volatility - volume_volatility
    
    # High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Composite Alpha Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha_factor = (df['close'] - df['open']) * df['volume'] - (ema_close * df['volume']) + df['close'].pct_change().rolling(window=10).sum()
    
    # Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = np.maximum(df['high'], df['close'].shift(1)) - np.minimum(df['low'], df['close'].shift(1))
    adjusted_return = intraday_log_return / true_range + (df['volume'] - df['volume'].shift(1))
    
    # Intraday Volatility
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    intraday_volatility = (high_low_spread + open_close_spread) * df['volume']
    
    # Volume Adjusted Momentum (VAMI)
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = (df['close'] - df['open']) * df['volume']
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    vami = volume_adjusted_price_change / ema_volume
    
    # Simple Moving Average (SMA) of Volume-Weighted Close
    sma_volume_weighted_close = volume_weighted_close.rolling(window=21).mean()
    
    # Weighted Moving Average (WMA) of Close
    wma_close = df['close'].rolling(window=21).apply(lambda x: np.average(x, weights=np.arange(1, 22)), raw=True)
    
    # Short-Term Reversal
    short_term_reversal = df['close'].pct_change().rolling(window=5).sum()
    
    # Long-Term Trend
    long_term_trend = volume_weighted_close.rolling(window=60).mean()
    
    # Enhanced Composite Alpha Factor
    enhanced_composite_alpha_factor = (volume_weighted_close - sma_volume_weighted_close - (volume_weighted_close - wma_close)) + vami + short_term_reversal + intraday_volatility - long_term_trend
    
    # Final Integration
    integrated_factors = high_low_spread_ratio + corrected_intraday_momentum + composite_alpha_factor
    final_factor = integrated_factors + adjusted_return
    
    # Trend Adjustment
    long_term_trend_100 = df['close'].rolling(window=100).mean()
    short_term_trend_20 = df['close'].ewm(span=20, adjust=False).mean()
    trend_diff = short_term_trend_20 - long_term_trend_100
    final_factor = final_factor.where(trend_diff > 0, final_factor * 1.5).where(trend_diff <= 0, final_factor * 0.5)
    
    return final_factor
```
