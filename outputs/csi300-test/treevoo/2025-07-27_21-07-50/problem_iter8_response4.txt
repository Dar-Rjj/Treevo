```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] - df['open']) / df['open']
    
    # Integrate Volume into Intraday Momentum
    volume_adjusted_momentum = intraday_momentum * df['volume']
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    abs_daily_returns = daily_returns.abs()
    intraday_volatility = abs_daily_returns.rolling(window=20).sum() / 20
    
    # Calculate Volume Thrust
    volume_thrust = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Factors
    combined_factor = (
        0.4 * volume_adjusted_momentum + 
        0.2 * intraday_range + 
        0.2 * intraday_volatility + 
        0.2 * volume_thrust
    )
    
    # Adjust for Price Volatility
    short_term_volatility = df['close'].pct_change().rolling(window=5).std()
    adjusted_combined_factor = combined_factor - short_term_volatility
    
    # Calculate Simple Moving Average (SMA) of Close prices
    sma_21 = df['close'].rolling(window=21).mean()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate Short-Term Reversal
    daily_returns = (df['close'] - df['open']) / df['open']
    short_term_reversal = daily_returns.rolling(window=5).sum()
    
    # Composite Alpha Factor
    composite_alpha_factor = (
        (volume_weighted_close - sma_21) +
        short_term_reversal +
        adjusted_combined_factor
    )
    
    # Intraday Volatility and Momentum Component
    intraday_high_low_spread = df['high'] - df['low']
    intraday_open_close_spread = df['close'] - df['open']
    combined_intraday_spreads = (intraday_high_low_spread + intraday_open_close_spread) * df['volume']
    
    # Momentum Component
    lagged_intraday_return = (df['high'].shift(1) - df['low'].shift(1)) / df['low'].shift(1)
    current_intraday_return = (df['high'] - df['low']) / df['low']
    
    # Reversal Component
    lagged_close_to_open_return = (df['close'].shift(1) - df['open']) / df['open']
    current_close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Momentum and Reversal Interaction
    momentum_reversal_interaction = (
        (lagged_intraday_return + current_intraday_return) *
        (lagged_close_to_open_return + current_close_to_open_return)
    )
    
    # Volume Adjusted Factor
    average_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_factor = (
        (momentum_reversal_interaction + combined_intraday_spreads) / 
        (df['volume'] / average_volume_20)
    )
    
    # Final Alpha Factor
    final_alpha_factor = composite_alpha_factor + volume_adjusted_factor
    
    return final_alpha_factor.dropna()
```
