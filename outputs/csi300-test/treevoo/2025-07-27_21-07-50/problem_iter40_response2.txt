```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    volume_weighted_close_open_diff = (df['close'] - df['open']) * df['volume']
    
    # Integrate Both Factors
    integrated_factor = high_low_spread_ratio + volume_weighted_close_open_diff
    
    # Composite Alpha Factor
    ema_close = df['close'].ewm(span=21, adjust=False).mean()
    volume_weighted_close = df['close'] * df['volume']
    composite_alpha_factor = (volume_weighted_close - ema_close)
    
    # Add Medium-Term Reversal
    daily_returns = df['close'] / df['open'] - 1
    past_10_day_returns = daily_returns.rolling(window=10).sum()
    composite_alpha_factor += past_10_day_returns
    
    # Generate Adjusted Return
    intraday_log_return = np.log(df['close'] / df['open'])
    true_range = np.maximum(df['high'], df['close'].shift(1)) - np.minimum(df['low'], df['close'].shift(1))
    scaled_log_return = intraday_log_return / true_range
    
    # Incorporate Volume Impact
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    adjusted_return = scaled_log_return * volume_change
    
    # Final Integration
    combined_factors = integrated_factor + composite_alpha_factor
    final_alpha = combined_factors + adjusted_return
    
    return final_alpha
```
