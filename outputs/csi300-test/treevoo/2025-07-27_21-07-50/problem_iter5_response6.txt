import pandas as pd

def heuristics_v2(df):
    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spread = high_low_spread + open_close_spread
    intraday_volatility = (combined_spread * df['volume']).rolling(window=20).mean()

    # Volume Adjusted Momentum Component
    daily_price_change = df['close'] - df['open']
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=20).sum()
    moving_average_volume = df['volume'].rolling(window=20).mean()
    vami = volume_adjusted_price_change / moving_average_volume

    # Volume-Weighted Close Price
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=20).mean() / df['volume'].rolling(window=20).mean()

    # Simple Moving Average (SMA) of Close prices
    sma_close = df['close'].rolling(window=21).mean()

    # Short-Term Reversal
    daily_returns = df['close'] / df['open'] - 1
    short_term_reversal = daily_returns.rolling(window=5).sum()

    # Composite Alpha Factor
    composite_alpha_factor = (volume_weighted_close - sma_close) + vami + short_term_reversal

    return composite_alpha_factor
