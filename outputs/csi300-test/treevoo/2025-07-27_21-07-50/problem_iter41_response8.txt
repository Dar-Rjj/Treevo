import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the close price in the range of high and low
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the amount-weighted volume
    df['amount_weighted_volume'] = df['amount'] * df['volume']
    
    # Calculate the average of the close position over 10 days
    df['avg_close_position_10d'] = df['close_position'].rolling(window=10).mean()
    
    # Calculate the total amount-weighted volume over 10 days
    df['total_amount_weighted_volume_10d'] = df['amount_weighted_volume'].rolling(window=10).sum()
    
    # Generate the alpha factor
    df['alpha_factor'] = df['avg_close_position_10d'] * (df['total_amount_weighted_volume_10d'] / df['volume'].rolling(window=10).sum())
    
    return df['alpha_factor']
