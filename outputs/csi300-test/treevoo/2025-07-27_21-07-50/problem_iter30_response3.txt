import pandas as pd
import numpy as np
from talib import abstract

def heuristics_v2(df):
    # Calculate Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate Exponential Moving Averages
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate Relative Strength Index (RSI)
    RSI_14 = abstract.RSI(df['close'], timeperiod=14)
    RSI_28 = abstract.RSI(df['close'], timeperiod=28)
    df['RSI_14'] = RSI_14
    df['RSI_28'] = RSI_28

    # Calculate On Balance Volume (OBV)
    OBV = abstract.OBV(df['close'], df['volume'])
    df['OBV'] = OBV

    # Calculate True Range
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)), abs(x['low'] - df['close'].shift(1))), axis=1)

    # Calculate Percentage Change in Close Prices
    df['Daily_Return'] = df['close'].pct_change()
    df['Weekly_Return'] = df['close'].pct_change(periods=5)
    df['Monthly_Return'] = df['close'].pct_change(periods=20)

    # Calculate Volume Rate of Change (ROC)
    df['Volume_ROC'] = df['volume'].pct_change()

    # Calculate Volume SMA
    df['Volume_SMA_5'] = df['volume'].rolling(window=5).mean()
    df['Volume_SMA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_SMA_20'] = df['volume'].rolling(window=20).mean()

    # Calculate SMA Crossover
    df['SMA_5_over_SMA_10'] = df['SMA_5'] > df['SMA_10']
    df['SMA_10_over_SMA_20'] = df['SMA_10'] > df['SMA_20']

    # Calculate EMA Crossover
    df['EMA_5_over_EMA_10'] = df['EMA_5'] > df['EMA_10']
    df['EMA_10_over_EMA_20'] = df['EMA_10'] > df['EMA_20']

    # Calculate RSI with Moving Averages
    df['RSI_14_over_SMA_10'] = df['RSI_14'] > df['SMA_10']
    df['RSI_28_over_SMA_20'] = df['RSI_28'] > df['SMA_20']

    # Calculate OBV with Volume-SMA
    df['OBV_over_Volume_SMA_5'] = df['OBV'] > df['Volume_SMA_5']
    df['OBV_over_Volume_SMA_10'] = df['OBV'] > df['Volume_SMA_10']

    # Calculate Average Directional Index (ADX)
    ADX_14 = abstract.ADX(df['high'], df['low'], df['close'], timeperiod=14)
    ADX_28 = abstract.ADX(df['high'], df['low'], df['close'], timeperiod=28)
    df['ADX_14'] = ADX_14
    df['ADX_28'] = ADX_28

    # Calculate Plus Directional Indicator (+DI)
    PLUS_DI_14 = abstract.PLUS_DI(df['high'], df['low'], df['close'], timeperiod=14)
    PLUS_DI_28 = abstract.PLUS_DI(df['high'], df['low'], df['close'], timeperiod=28)
