import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    daily_price_change = df['close'] - df['open']

    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']

    # Momentum Component
    lagged_intraday_return = intraday_return.shift(1)
    momentum_component = lagged_intraday_return + intraday_return

    # Reversal Component
    close_to_open_return = df['close'] - df['open']
    lagged_close_to_open_return = close_to_open_return.shift(1)
    reversal_component = lagged_close_to_open_return + close_to_open_return

    # Momentum and Reversal Interaction
    interaction = momentum_component * reversal_component

    # Volume Adjusted Factor
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_adjusted_factor = df['volume'] / volume_ma

    # Enhanced Momentum Component
    ema_intraday_return = intraday_return.ewm(span=20, adjust=False).mean()
    enhanced_momentum = intraday_return - ema_intraday_return

    # Enhanced Reversal Component
    ema_close_to_open_return = close_to_open_return.ewm(span=20, adjust=False).mean()
    enhanced_reversal = close_to_open_return - ema_close_to_open_return

    # Enhanced Momentum and Reversal Interaction
    enhanced_interaction = enhanced_momentum * enhanced_reversal

    # Intraday Volatility Component
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['close'] - df['open']
    combined_spread = (high_low_spread + open_close_spread) * df['volume']

    # Volume Adjusted Price Change
    volume_adjusted_price_change = (daily_price_change * df['volume']).rolling(window=20).sum()

    # Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']

    # Simple Moving Average (SMA) of Close prices
    sma_close = df['close'].rolling(window=21).mean()

    # Short-Term Reversal
    daily_returns = close_to_open_return
    short_term_reversal = daily_returns.rolling(window=5).sum()

    # Composite Alpha Factor
    composite_alpha_factor = (volume_weighted_close - sma_close) + enhanced_interaction + short_term_reversal

    # Final CE-VAIMRF+
    ce_vaimrf_plus = (volume_adjusted_price_change * enhanced_interaction) / volume_adjusted_factor + composite_alpha_factor

    return ce_vaimrf_plus
