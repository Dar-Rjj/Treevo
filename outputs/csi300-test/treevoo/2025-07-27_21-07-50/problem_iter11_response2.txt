```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / (df['high'] + df['low'])
    
    # Calculate Volume-Weighted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open = close_open_diff * df['volume']
    
    # Integrate Both Factors
    integrated_factors = high_low_spread_ratio + volume_weighted_close_open
    
    # Calculate Exponential Moving Average (EMA) of Close prices
    ema_period = 21
    ema_close = df['close'].ewm(span=ema_period, adjust=False).mean()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate Short-Term Reversal
    n_days = 5
    daily_returns = df['close'].pct_change()
    short_term_reversal = daily_returns.rolling(window=n_days).sum()
    
    # Composite Alpha Factor
    composite_alpha = volume_weighted_close - ema_close + short_term_reversal
    
    # Calculate Intraday Log-Return
    intraday_log_return = np.log(df['close'] / df['open'])
    
    # Calculate Intraday Volatility
    high_low_range = df['high'] - df['low']
    intraday_volatility = high_low_range * intraday_log_return
    
    # Measure Volume Impact
    average_volume = df['volume'].rolling(window=ema_period).mean()
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Generate Adjusted Return
    adjusted_return = intraday_log_return * intraday_volatility + volume_change - intraday_log_return
    
    # Final Integration
    final_factor = integrated_factors + composite_alpha + adjusted_return
    
    return final_factor
```
