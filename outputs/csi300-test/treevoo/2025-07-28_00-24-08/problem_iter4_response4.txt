```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].rolling(window=n-1).mean()) / df['close'].rolling(window=n-1).mean()
    
    # Weight by Intraday Volatility
    weighted_momentum = intraday_volatility * momentum
    
    return weighted_momentum
```
