```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Price Momentum
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_filter'] = (df['volume_change'] > 0).astype(int)
    
    # Integrate Amount for Momentum Strengthening
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['positive_amount_filter'] = (df['amount_change'] > 0).astype(int)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    
    # Combine High-Low Range and Volume
    df['volume_weighted_high_low'] = df['high_low_range'] * df['volume']
    df['normalized_volatility'] = df['volume_weighted_high_low'] / df['average_volume']
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['normalized_volatility']
    
    # Introduce Trend Reversal Indicator
    def compute_rsi(series, period=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['rsi'] = compute_rsi(df['close'])
    df['trend_direction'] = (df['close'] > df['close'].rolling(window=50).mean()).astype(int)
    df['trend_reversal'] = ((df['rsi'] < 30) | (df['rsi'] > 70)) & (df['trend_direction'].diff() != 0).astype(int)
    
    # Smoothed Moving Average
    df['ema_adjusted_returns'] = df['adjusted_returns'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['combined_factor'] = df['ema_adjusted_returns'] * (1 + df['trend_reversal'])
    df['alpha_signal'] = df['combined_factor'] * df['price_momentum'] * df['positive_volume_filter'] * df['positive_amount_filter']
    
    return df['alpha_signal']
```
