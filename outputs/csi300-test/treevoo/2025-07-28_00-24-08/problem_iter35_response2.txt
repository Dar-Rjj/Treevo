```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight High-Low Range by Volume
    high_low_range = (df['high'] - df['low'])
    volume_weighted_high_low = high_low_range * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low
    
    # Smoothed Moving Average
    sma_5 = adjusted_returns.rolling(window=5).mean()
    smoothed_returns = sma_5.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factor = (intraday_return * overnight_return) * price_momentum_ratio
    average_volume = df['volume'].rolling(window=5).mean()
    integrated_factor = integrated_factor * np.where(df['volume'] > average_volume, 1.7, 0.3)
    
    # Compute Price Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    price_momentum = log_returns.rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    
    # Generate Final Momentum Factor
    final_momentum_factor = price_momentum * volume_confirmation
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = high_low_range * intraday_return
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = price_momentum * (high_low_range * df['volume'])
    
    # Incorporate Historical Volatility
    historical_volatility = df['close'].pct_change().rolling(window=20).std() * df['volume']
    
    # Integrate All Components
    final_alpha_factor = smoothed_returns + final_momentum_factor + high_low_volatility + combined_momentum_volatility + historical_volatility
    
    return final_alpha_factor
```
