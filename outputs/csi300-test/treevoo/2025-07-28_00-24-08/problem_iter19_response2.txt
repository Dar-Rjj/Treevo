```python
import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()

    # Compute Price Momentum
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()

    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'].diff()
    df['positive_volume'] = (df['volume_change'] > 0).astype(int)

    # Integrate Amount for Momentum Strengthening
    df['amount_change'] = df['amount'].diff()
    df['positive_amount'] = (df['amount_change'] > 0).astype(int)

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] / df['close'].shift(1)) - 1

    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Average Volume Over a Period
    df['average_volume'] = df['volume'].rolling(window=20).mean()

    # Combine High-Low Range and Volume
    df['weighted_high_low_range'] = df['high_low_range'] * (df['volume'] / df['average_volume'])

    # Adjust for Combined Volatility
    df['normalized_volatility'] = zscore(df['weighted_high_low_range'])
    df['adjusted_returns'] = df['aggregate_returns'] * df['normalized_volatility']

    # Introduce Trend Reversal Indicator
    def compute_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        avg_gain = gain.rolling(window=window).mean()
        avg_loss = loss.rolling(window=window).mean()
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    df['rsi'] = compute_rsi(df['close'], window=14)
    df['trend_direction'] = (df['close'] > df['close'].rolling(window=50).mean()).astype(int)
    df['trend_reversal'] = ((df['rsi'] < 30) | (df['rsi'] > 70)) & (df['trend_direction'].diff() != 0).astype(int)

    # Smoothed Moving Average
    df['sma_adjusted_returns'] = df['adjusted_returns'].rolling(window=5).mean()
    df['ema_smoothed_returns'] = df['sma_adjusted_returns'].ewm(span=5, adjust=False).mean()

    # Final Factor Calculation
    df['combined_factor'] = df['ema_smoothed_returns'] * (1 + df['trend_reversal'])
    df['alpha_signal'] = (df['combined_factor'] > 0).astype(int)
    df['final_alpha_factor'] = df['price_momentum'] * df['positive_volume'] * df['positive_amount'] * df['adjusted_returns'] * df['alpha_signal']

    # Incorporate Trading Activity
    df['turnover_ratio'] = df['volume'] / df['share_capital']
    df['adjusted_momentum'] = df['price_momentum'] * df['turnover_ratio']

    # Add Market Sentiment Indicator
    df['market_sentiment'] = (df['news_sentiment_score'] + df['social_media_sentiment_score']) / 2
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['market_sentiment']

    return df['final_alpha_factor']
```
