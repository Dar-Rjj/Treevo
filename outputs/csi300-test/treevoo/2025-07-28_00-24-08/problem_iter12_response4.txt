```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    previous_close = df['close'].shift(1)
    close_to_open_return = (previous_close - df['open']) / previous_close
    
    # Aggregate Returns
    aggregate_returns = intraday_return + close_to_open_return
    
    # Volume Surge Indicator
    avg_5d_volume = df['volume'].rolling(window=5).mean()
    volume_surge = df['volume'] > avg_5d_volume
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Combine High-Low Range and Volume
    volume_weighted_high_low_range = high_low_range * (df['volume'] / avg_volume)
    
    # Adjust for Combined Volatility
    adjusted_volatility = aggregate_returns * (volume_weighted_high_low_range / avg_volume)
    
    # Integrate Factors
    factor = (intraday_return * close_to_open_return) * price_momentum_ratio
    factor = factor * (1.5 if volume_surge else 0.5)
    factor = factor * (short_term_momentum + adjusted_volatility)
    
    return factor
```
