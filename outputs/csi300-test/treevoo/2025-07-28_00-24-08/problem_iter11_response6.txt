```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low = high_low_range * df['volume'] / average_volume
    
    # Integrate Intraday and Previous Day's Returns
    integrated_returns = intraday_return * overnight_return
    
    # Adjust for Combined Volatility
    adjusted_aggregate_returns = aggregate_returns * volume_weighted_high_low
    
    # Volume Surge Indicator
    volume_surge = (df['volume'] > average_volume).astype(int)
    
    # Conditionally Adjust by Volume Surge
    conditional_adjustment = volume_surge.apply(lambda x: 1.75 if x == 1 else 0.75)
    
    # Combine Price Momentum and Intraday Reversal
    combined_momentum_reversal = momentum_ratio * integrated_returns
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum_reversal 
                    * adjusted_aggregate_returns 
                    * conditional_adjustment)
    
    return alpha_factor
```
