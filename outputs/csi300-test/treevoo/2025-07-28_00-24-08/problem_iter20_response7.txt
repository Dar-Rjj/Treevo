```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the relative difference between the high and low prices
    rel_diff = (df['high'] - df['low']) / df['low']
    
    # Calculate a simple moving average of the volume
    sma_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the factor value using the above computed features
    factor_value = (avg_price * rel_diff) / sma_volume
    
    # Normalize the factor values to have a mean of 0 and standard deviation of 1
    normalized_factor = (factor_value - factor_value.mean()) / factor_value.std()
    
    return normalized_factor
```
