import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period (e.g., 5 days)
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Combine High-Low Range and Volume
    df['vol_weighted_high_low'] = df['high_low_range'] * df['volume']
    df['normalized_volatility'] = df['vol_weighted_high_low'] / df['avg_volume']
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['normalized_volatility']
    
    # Smoothed Moving Average
    df['sma_adjusted_returns'] = df['adjusted_returns'].rolling(window=5).mean()
    df['smoothed_factor'] = df['sma_adjusted_returns'].ewm(alpha=0.2, adjust=False).mean()
    
    return df['smoothed_factor'].dropna()
