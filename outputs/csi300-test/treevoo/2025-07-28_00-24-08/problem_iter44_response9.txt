```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return'].shift(1)
    
    # Weight High-Low Range by Volume
    df['high_low_range_vol_weighted'] = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    df['adjusted_return'] = df['aggregate_return'] * df['high_low_range_vol_weighted']
    
    # Smoothed Moving Average
    df['sma'] = df['adjusted_return'].rolling(window=5).mean()
    df['ema'] = df['sma'].ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    df['integrated_factor'] = df['aggregate_return'] * df['momentum_ratio']
    df['integrated_factor'] = np.where(df['volume'] > df['volume'].rolling(window=20).mean(), 
                                       df['integrated_factor'] * 1.7, 
                                       df['integrated_factor'] * 0.3)
    df['integrated_factor'] = np.where(df['volume_change'] > 0, 
                                       df['integrated_factor'] * 1.5, 
                                       df['integrated_factor'] * 0.8)
    
    # Compute Price Momentum
    df['price_momentum'] = df['close'].pct_change().rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['volume_confirmation'] = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    
    # Generate Final Momentum Factor
    df['final_momentum_factor'] = df['price_momentum'] * df['volume_confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['high_low_volatility'] = (df['high'] - df['low']) * df['intraday_return']
    
    # Integrate All Components
    df['combined_factor'] = df['ema'] + df['final_momentum_factor'] + df['high_low_volatility']
    
    # Combine Momentum and Volatility
    df['weighted_high_low_range'] = (df['high'] - df['low']) * df['volume']
    df['combined_mom_vol'] = df['price_momentum'] * df['weighted_high_low_range']
    
    # Incorporate Historical Volatility
    df['historical_volatility'] = df['close'].pct_change().rolling(window=20).std() * df['volume']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['intraday_return'].rolling(window=5).std()
    
    # Combine Intraday and Historical Volatility
    df['combined_volatility'] = 0.5 * df['intraday_volatility'] + 0.5 * df['historical_volatility']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] + df['combined_mom_vol'] + df['combined_volatility']
    
    return df['alpha_factor']
```
