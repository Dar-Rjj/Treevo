import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Return (Enhanced)
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['Prev_Close_to_Open_Return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Calculate Previous Day Overnight Return
    df['Prev_Day_Overnight_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['Aggregate_Returns'] = df['Intraday_Return'] + df['Prev_Day_Overnight_Return']
    
    # Intraday Volatility
    df['Intraday_Volatility'] = df['High_Low_Range'] / df['open']
    
    # Weight by Intraday Volatility
    df['Weighted_Aggregate_Returns'] = df['Aggregate_Returns'] * df['Intraday_Volatility']
    
    # Adjust for Volume
    df['Volume_Adjusted_Returns'] = df['Weighted_Aggregate_Returns'] * df['volume']
    
    # Smoothed Moving Average
    df['SMA_Volume_Adjusted_Returns'] = df['Volume_Adjusted_Returns'].rolling(window=5).mean()
    df['EMA_SMA'] = df['SMA_Volume_Adjusted_Returns'].ewm(span=5, adjust=False).mean()
    
    # Compute Price Momentum
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Momentum'] = df['Log_Returns'].rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['Volume_Confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    df['Final_Momentum_Factor'] = df['Price_Momentum'] * df['Volume_Confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['High_Low_Volatility'] = df['High_Low_Range'] * df['Intraday_Volatility']
    
    # Integrate All Components
    df['Combined_Factor'] = df['EMA_SMA'] + df['Final_Momentum_Factor'] + df['High_Low_Volatility']
    
    # Calculate Short-Term Price Momentum
    df['Short_Term_Price_Momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    df['Long_Term_Price_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['Volume_Change_Indicator'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['Price_Momentum_Ratio'] = df['Short_Term_Price_Momentum'] / df['Long_Term_Price_Momentum']
    
    # Integrate Additional Factors
    df['Additional_Factor'] = (df['Intraday_Return'] * df['Prev_Close_to_Open_Return'] * df['Price_Momentum_Ratio'])
    
    # Average Volume Over a Period
    df['Average_Volume_5D'] = df['volume'].rolling(window=5).mean()
    df['Volume_Above_Average'] = (df['volume'] > df['Average_Volume_5D']).astype(int)
    
    # Conditionally Adjust by Volume Metrics
    df['Volume_Adjustment'] = np.where(df['Volume_Above_Average'] == 1, 1.5, 0.5)
    df['Adjusted_Additional_Factor'] = df['Additional_Factor'] * df['Volume_Adjustment']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factor'] + df['Adjusted_Additional_Factor']
    
    return df['Final_Alpha_Factor'].dropna()
