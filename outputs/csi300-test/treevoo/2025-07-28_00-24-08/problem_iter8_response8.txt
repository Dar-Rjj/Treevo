import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Daily Log Return
    df['Log_Return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Price Momentum (Sum of last 20 days log returns)
    df['Price_Momentum'] = df['Log_Return'].rolling(window=20).sum()
    
    # Confirm Momentum with Volume
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Positive_Volume'] = np.where(df['Volume_Change'] > 0, 1, 0)
    
    # Integrate Amount for Momentum Strengthening
    df['Amount_Change'] = df['amount'] - df['amount'].shift(1)
    df['Positive_Amount'] = np.where(df['Amount_Change'] > 0, 1, 0)
    
    # Calculate Intraday Return Factors
    df['Intraday_High_Low_Ratio'] = df['high'] / df['low']
    df['Open_to_Close_Direction'] = np.where(df['close'] > df['open'], 1, 0)
    
    # Volume Weighted Smoothed Returns
    n = 20
    df['SMA_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Weighted_Returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['SMA_Volume'])
    
    # Calculate Overnight Return Factor
    df['Overnight_Return'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday and Overnight Factors
    df['Combined_Factor'] = df['Intraday_High_Low_Ratio'] + df['Overnight_Return']
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] * (df['volume'] / df['SMA_Volume'])
    
    # Weight High-Low Range by Volume
    df['Volume_Weighted_High_Low_Range'] = df['High_Low_Range'] * (df['volume'] / df['SMA_Volume'])
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Price_Momentum'] + df['Volume_Weighted_High_Low_Range'] + df['Adjusted_Combined_Factor']
    
    # Determine the Sign for Final Factor
    df['Final_Alpha_Sign'] = np.where(df['Final_Alpha_Factor'] > 0, 1, -1)
    
    return df['Final_Alpha_Sign']
