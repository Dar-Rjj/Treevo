```python
import pandas as pd
import numpy as np
from statsmodels.tsa.holtwinters import ExponentialSmoothing

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    df['volume_weighted_high_low'] = df['high_low_range'] * df['volume']
    
    # Calculate Price Momentum
    n = 7  # Number of periods for momentum
    df['price_momentum'] = (df['close'] - df['open'].shift(n)) / df['open'].shift(n)
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['volume_weighted_high_low']
    
    # Smoothed Moving Average
    # Compute Exponential Moving Average of Adjusted Returns
    df['ema_adjusted_returns'] = df['adjusted_returns'].ewm(span=7).mean()
    
    # Apply Triple Exponential Smoothing
    alpha, beta, gamma = 0.5, 0.5, 0.5  # Smoothing factors
    df['tes_ema_adjusted_returns'] = df['ema_adjusted_returns'].apply(lambda x: ExponentialSmoothing(x, trend='add', seasonal='add', seasonal_periods=7).fit(smoothing_level=alpha, smoothing_slope=beta, smoothing_seasonal=gamma).fittedvalues[-1])
    
    # Integrate Amount Data
    df['daily_dollar_volume'] = df['volume'] * df['close']
    
    # Adjust Smoothed Moving Average by Daily Dollar Volume
    df['heuristic_factor'] = df['tes_ema_adjusted_returns'] / df['daily_dollar_volume']
    
    return df['heuristic_factor']
```
