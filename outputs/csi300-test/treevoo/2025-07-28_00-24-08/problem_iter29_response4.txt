```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['is_above_avg_vol'] = df['volume'] > df['avg_volume_5d']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['volume_change_5d'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_open_return'] * df['price_momentum_ratio']
    df['integrated_factor'] = df['integrated_factor'].where(~df['is_above_avg_vol'], df['integrated_factor'] * 1.5)
    df['integrated_factor'] = df['integrated_factor'].where(df['is_above_avg_vol'], df['integrated_factor'] * 0.5)
    
    # Combine Momentum and Volatility
    df['momentum_volatility'] = df['price_momentum_ratio'] + (df['high_low_range'] / df['volume'])
    
    # Additional Volatility Component
    df['volatility_std_5d'] = df['high_low_range'].rolling(window=5).std() / df['avg_volume_5d']
    
    # Enhance Momentum with Volume
    df['enhanced_momentum'] = df['short_term_momentum'] + df['long_term_momentum']
    df['enhanced_momentum'] = df['enhanced_momentum'].where(df['volume_change_5d'] > 0, df['enhanced_momentum'] * 1.2)
    df['enhanced_momentum'] = df['enhanced_momentum'].where(df['volume_change_5d'] <= 0, df['enhanced_momentum'] * 0.8)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['momentum_volatility'] + df['volatility_std_5d'] + df['enhanced_momentum']
    
    return df['final_alpha_factor']
```
