```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']

    # Adjust for Volume
    df['volume_adjusted_returns'] = df['aggregate_returns'] * df['volume']

    # Weight by Intraday Volatility
    df['weighted_adjusted_returns'] = df['volume_adjusted_returns'] * df['intraday_volatility']

    # Smoothed Moving Average
    df['sma_5'] = df['weighted_adjusted_returns'].rolling(window=5).mean()
    df['smoothed_moving_average'] = df['sma_5'].ewm(alpha=0.3).mean()

    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)

    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(25)

    # Compute Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']

    # Integrate Intraday and Previous Day's Returns
    df['integrated_returns'] = df['intraday_return'] * (df['close'].shift(1) - df['open'])

    # Combine Price Momentum and Intraday Reversal
    df['combined_momentum_reversal'] = df['price_momentum_ratio'] * df['integrated_returns']

    # Conditionally Adjust by Volume Surge
    df['avg_volume_7'] = df['volume'].rolling(window=7).mean()
    df['volume_surge_factor'] = np.where(df['volume'] > df['avg_volume_7'], 1.8, 0.7)
    df['adjusted_momentum_reversal'] = df['combined_momentum_reversal'] * df['volume_surge_factor']

    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['smoothed_moving_average'] + df['adjusted_momentum_reversal']

    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)

    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low'] = (df['high_low_range'] * df['volume']).rolling(window=5).sum()

    # Calculate Average Volume Over a Period
    df['average_volume_5'] = df['volume'].rolling(window=5).mean()

    # Adjust for Combined Volatility
    df['adjusted_volatility'] = df['aggregate_returns'] * (df['volume_weighted_high_low'] / df['average_volume_5'])

    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = df['price_momentum_ratio'] + df['adjusted_volatility']

    # Final Factor
    df['final_factor'] = df['final_alpha_factor'] + df['combined_momentum_volatility']

    return df['final_factor']
```
