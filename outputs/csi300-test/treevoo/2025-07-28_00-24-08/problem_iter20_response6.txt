```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)

    # Aggregate Returns
    aggregate_returns = intraday_return + prev_close_to_open_return

    # Volume Surge Indicator
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_surge_indicator = (df['volume'] > avg_volume_5d).astype(int)

    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum

    # Volume-Weighted High-Low Volatility
    high_low_volatility = (df['high'] - df['low']) * df['volume']
    avg_volume = df['volume'].rolling(window=5).mean()
    vol_weighted_high_low_volatility = high_low_volatility / avg_volume

    # Integrate Factors
    integrated_factor = (intraday_return * prev_close_to_open_return * price_momentum_ratio)
    integrated_factor = integrated_factor * (1.5 if volume_surge_indicator else 0.5)
    integrated_factor = integrated_factor * vol_weighted_high_low_volatility

    # Smoothed Moving Average
    sma_5d = integrated_factor.rolling(window=5).mean()
    ema_factor = sma_5d.ewm(span=5, adjust=False).mean()

    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    final_factor = ema_factor * daily_dollar_volume

    return final_factor
```
