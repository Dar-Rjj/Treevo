import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['short_momentum'] = df['close'].diff(periods=5)
    
    # Calculate Long-Term Price Momentum
    df['long_momentum'] = df['close'].diff(periods=20)
    
    # Compute Price Momentum Ratio
    df['momentum_ratio'] = df['short_momentum'] / df['long_momentum']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Volume Surge Indicator
    df['avg_5d_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = df['volume'] > df['avg_5d_volume']
    
    # Integrate Intraday and Previous Day's Returns
    df['integrated_returns'] = df['intraday_return'] * df['prev_close_to_open']
    
    # Combine Price Momentum and Intraday Reversal
    df['combined_factor'] = df['momentum_ratio'] * df['integrated_returns']
    
    # Conditionally Adjust by Volume Surge
    df['adjusted_factor'] = df.apply(lambda row: row['combined_factor'] * 1.75 if row['volume_surge'] else row['combined_factor'] * 0.75, axis=1)
    
    # Final Alpha Factor
    return df['adjusted_factor']
