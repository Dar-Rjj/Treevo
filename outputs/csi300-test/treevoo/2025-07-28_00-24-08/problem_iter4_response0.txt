```python
import pandas as pd

def heuristics_v2(df, n_periods=20):
    """
    Calculate a novel alpha factor based on the High-Low Volatility Adjusted Volume Momentum.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index (date)
    - n_periods: int, number of periods to look back for calculating momentum and average volume
    
    Returns:
    - pandas Series indexed by (date) representing the factor values
    """
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=n_periods).mean()
    
    # Calculate Price Momentum
    price_momentum = (df['close'] / df['open'].shift(n_periods)) - 1
    
    # Weight High-Low Range by Volume
    weighted_volatility = high_low_range * (df['volume'] / avg_volume)
    
    # Combine Momentum and Weighted Volatility
    factor = price_momentum + weighted_volatility
    
    return factor
```
