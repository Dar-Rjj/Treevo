```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Percentage Change
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Volume Momentum
    df['volume_pct_change'] = df['volume'].pct_change()
    
    # Calculate Close-to-Next Open Return
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Calculate Weighted Inverse of Close-to-Next Open Return
    df['weighted_inverse_return'] = 1 / (1 + df['close_to_next_open_return'].abs()) * df['volume_pct_change'].shift(1)
    
    # Combine all the components to create the alpha factor
    df['alpha_factor'] = (df['intraday_pct_change'] * df['daily_range']) + df['weighted_inverse_return']
    
    return df['alpha_factor']
```
