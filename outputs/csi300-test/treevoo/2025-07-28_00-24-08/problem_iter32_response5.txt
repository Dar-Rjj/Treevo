```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']

    # Weight by Intraday Volatility
    df['intraday_volatility'] = df['intraday_return'].rolling(window=5).std()
    df['weighted_aggregate_returns'] = df['aggregate_returns'] * df['intraday_volatility']

    # Adjust for Volume
    df['volume_adjusted_returns'] = df['weighted_aggregate_returns'] * df['volume']

    # Smoothed Moving Average
    df['sma_volume_adjusted_returns'] = df['volume_adjusted_returns'].rolling(window=5).mean()
    df['smoothed_returns'] = df['sma_volume_adjusted_returns'].ewm(span=5, adjust=False).mean()

    # Compute Price Momentum
    df['log_returns'] = np.log(df['close']).diff()
    df['price_momentum'] = df['log_returns'].rolling(window=20).sum()

    # Apply Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['final_momentum_factor'] = df['price_momentum'] * df['volume_confirmation']

    # Incorporate High-Low Volatility into Final Factor
    df['high_low_volatility'] = df['high_low_range'] * df['intraday_volatility']
    df['integrated_factor'] = df['smoothed_returns'] + df['final_momentum_factor'] + df['high_low_volatility']

    # Calculate Short-Term and Long-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)

    # Volume Change Indicator
    df['volume_change_indicator'] = df['volume'] - df['volume'].shift(5)

    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']

    # Integrate Additional Factors
    df['intraday_overnight_interaction'] = df['intraday_return'] * df['overnight_return']
    df['momentum_ratio_factor'] = df['intraday_overnight_interaction'] * df['price_momentum_ratio']
    
    # Conditionally Adjust by Volume Metrics
    df['volume_above_average'] = (df['volume'] > df['volume'].rolling(window=20).mean()).astype(float)
    df['volume_adjusted_factor'] = df['momentum_ratio_factor'] * (df['volume_above_average'] * 1.5 + (1 - df['volume_above_average']) * 0.5)

    # Combine Momentum and Volatility
    df['volatility_weighted_high_low'] = df['high_low_range'] * df['volume']

    # Incorporate Historical Volatility
    df['historical_volatility'] = df['log_returns'].rolling(window=20).std()
    df['volatility_weighted_by_volume'] = df['historical_volatility'] * df['volume']

    # New Intraday-Overnight Return Interaction
    df['new_intraday_overnight'] = df['intraday_return'] * df['overnight_return']

    # New Volume-Adjusted High-Low Volatility
    df['new_vol_adj_high_low_vol'] = df['high_low_range'] * df['volume'] * df['intraday_volatility']

    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['integrated_factor'] 
        + df['volume_adjusted_factor'] 
        + df['volatility_weighted_high_low'] 
        + df['volatility_weighted_by_volume']
        + df['new_intraday_overnight']
        + df['new_vol_adj_high_low_vol']
    )

    return df['final_alpha_factor']
```
