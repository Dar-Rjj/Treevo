```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['is_above_avg_volume'] = df['volume'] > df['avg_volume_5d']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['volume_change_indicator'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio']
    df['integrated_factor'] = df['integrated_factor'].where(~df['is_above_avg_volume'], df['integrated_factor'] * 1.5)
    df['integrated_factor'] = df['integrated_factor'].where(df['is_above_avg_volume'], df['integrated_factor'] * 0.5)
    
    # Combine Momentum and Volatility
    df['momentum_volatility'] = df['short_term_momentum'] + df['high_low_range'] * df['volume']
    
    # Confirm Momentum with Volume and Amount
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_filter'] = (df['volume_change'] > 0).astype(int)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['positive_amount_filter'] = (df['amount_change'] > 0).astype(int)
    
    # Conditional Adjustment Based on Volume and Amount
    conditions = [
        (df['positive_volume_filter'] == 1) & (df['positive_amount_filter'] == 1),
        (df['positive_volume_filter'] == 1) & (df['positive_amount_filter'] == 0),
        (df['positive_volume_filter'] == 0) & (df['positive_amount_filter'] == 1),
        (df['positive_volume_filter'] == 0) & (df['positive_amount_filter'] == 0)
    ]
    choices = [1.75, 1.25, 1.25, 0.75]
    df['conditional_adjustment'] = pd.np.select(conditions, choices, default=1)
    df['integrated_factor'] = df['integrated_factor'] * df['conditional_adjustment']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] * df['momentum_volatility']
    
    return df['final_alpha_factor']
```
