import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the 5-day moving average of the close price
    df['ma_5_close'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the close price
    df['ma_20_close'] = df['close'].rolling(window=20).mean()
    
    # Calculate the relative strength (RS) as the ratio of the 5-day MA to the 20-day MA
    df['rs'] = df['ma_5_close'] / df['ma_20_close']
    
    # Calculate the money flow by multiplying the average of high and low with volume
    df['money_flow'] = (df['high'] + df['low']) / 2 * df['volume']
    
    # Calculate the 5-day exponential moving average of the money flow
    df['ema_money_flow'] = df['money_flow'].ewm(span=5, adjust=False).mean()
    
    # The alpha factor is a combination of the relative strength and the exponential moving average of the money flow
    df['alpha_factor'] = df['rs'] * df['ema_money_flow']
    
    return df['alpha_factor']
