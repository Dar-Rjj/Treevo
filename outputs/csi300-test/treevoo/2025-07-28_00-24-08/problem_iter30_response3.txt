import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight by Intraday Volatility
    intraday_volatility = np.abs(intraday_return).rolling(window=5).std()
    weighted_aggregate_returns = aggregate_returns * intraday_volatility
    
    # Adjust for Volume
    volume_adjusted_returns = weighted_aggregate_returns * df['volume']
    
    # Smoothed Moving Average
    sma = volume_adjusted_returns.rolling(window=5).mean()
    exp_smoothing_factor = 0.2
    smoothed_returns = sma.ewm(alpha=exp_smoothing_factor).mean()
    
    # Compute Price Momentum
    price_momentum = np.log(df['close']).diff().rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    final_momentum_factor = price_momentum * volume_confirmation
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = high_low_range * intraday_volatility
    
    # Integrate All Components
    integrated_factor = smoothed_returns + final_momentum_factor + high_low_volatility
    
    # Calculate Short-Term and Long-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    long_term_momentum = df['close'] - df['close'].shift(20)
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Additional Factors
    adjusted_intraday_return = intraday_return * overnight_return * price_momentum_ratio
    volume_metrics_adjustment = np.where(df['volume'] > df['volume'].rolling(window=20).mean(), 1.5, 0.5)
    additional_factor = adjusted_intraday_return * volume_metrics_adjustment
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = price_momentum + high_low_range * df['volume']
    
    # Incorporate Historical Volatility
    historical_volatility = np.log(df['close']).diff().rolling(window=20).std()
    weighted_historical_volatility = historical_volatility * df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = integrated_factor + additional_factor + combined_momentum_volatility + weighted_historical_volatility
    return final_alpha_factor
