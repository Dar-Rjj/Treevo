```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['open'].shift(1) - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    aggregate_returns = intraday_return * prev_close_to_open_return

    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=20).mean()

    # Weight High-Low Range by Volume
    volume_weighted_high_low_range = high_low_range * df['volume']

    # Adjust by Average Volume
    adjusted_high_low_range = volume_weighted_high_low_range / avg_volume

    # Volume Change Indicator
    volume_change_indicator = df['volume'] - df['volume'].shift(5)

    # Combine Volume Adjusted Volatility and Volume Change
    combined_volatility_and_volume_change = adjusted_high_low_range + volume_change_indicator

    # Integrate Intraday and Previous Day's Returns
    integrated_intraday_prev_day_return = aggregate_returns * prev_close_to_open_return

    # Combine Price Momentum and Intraday Reversal
    combined_price_momentum_intraday_reversal = price_momentum_ratio * integrated_intraday_prev_day_return

    # Conditionally Adjust by Volume Surge
    conditionally_adjusted_volume_surge = 1.5 if df['volume'] > avg_volume else 0.5

    # Final Alpha Factor
    final_alpha_factor = (combined_price_momentum_intraday_reversal * conditionally_adjusted_volume_surge) + combined_volatility_and_volume_change

    return final_alpha_factor
```
