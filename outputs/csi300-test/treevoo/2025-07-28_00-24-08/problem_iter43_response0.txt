```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate change in volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Analyze intra-day volatility
    df['intra_day_volatility'] = df['high'] - df['low']
    
    # Identify trends using moving averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Exponential Moving Average (EMA) of Close prices
    def ema(series, span):
        return series.ewm(span=span, adjust=False).mean()
    
    df['EMA_12'] = ema(df['close'], 12)
    df['EMA_26'] = ema(df['close'], 26)
    
    # Momentum indicators
    df['ROC_14'] = (df['close'] - df['close'].shift(14)) / df['close'].shift(14)
    
    # MACD
    df['MACD_Line'] = df['EMA_12'] - df['EMA_26']
    df['Signal_Line'] = ema(df['MACD_Line'], 9)
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Volume-weighted average price (VWAP)
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Cum_Vol_Price'] = (df['Typical_Price'] * df['volume']).cumsum()
    df['Cum_Volume'] = df['volume'].cumsum()
    df['VWAP'] = df['Cum_Vol_Price'] / df['Cum_Volume']
    
    # Volume and amount ratios
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # On-balance volume (OBV)
    obv = np.where(df['close'] > df['close'].shift(1), df['volume'],
                   np.where(df['close'] < df['close'].shift(1), -df['volume'], 0)).cumsum()
    df['OBV'] = obv
    
    # Bollinger Bands
    df['Middle_Band'] = df['close'].rolling(window=20).mean()
    df['std_dev'] = df['close'].rolling(window=20).std()
    df['Upper_Band'] = df['Middle_Band'] + 2 * df['std_dev']
    df['Lower_Band'] = df['Middle_Band'] - 2 * df['std_dev']
    
    # Stochastic Oscillator
    high14 = df['high'].rolling(window=14).max()
    low14 = df['low'].rolling(window=14).min()
    df['%K'] = (df['close'] - low14) / (high14 - low14) * 100
    df['%D'] = df['%K'].rolling(window=3).mean()
    
    # Combine multiple factors into a single alpha factor
    df['alpha_factor'] = (
        0.3 * df['daily_return'] +
        0.1 * df['volume_change'] +
        0.1 * df['intra_day_volatility'] +
        0.1 * (df['close'] - df['SMA_5']) +
        0.1 * (df['close'] - df['SMA_20']) +
        0.05 * (df['MACD_Line'] - df['Signal_Line']) +
        0.05 * df['RSI_14'] +
        0.05 * df['VWAP'] +
        0.05 * df['OBV'] +
        0.05 * (df['close'] - df['Middle_Band'])
    )
    
    return df['alpha_factor'].dropna()
```
