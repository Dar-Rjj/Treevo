```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Volume Surge Indicator
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_surge_indicator'] = df['volume'] > df['5_day_avg_volume']
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_to_open_return']
    
    # Conditionally Adjust by Volume Surge Indicator
    df['final_alpha_factor'] = df.apply(
        lambda row: row['integrated_factor'] * 1.5 if row['volume_surge_indicator'] else row['integrated_factor'] * 0.5,
        axis=1
    )
    
    return df['final_alpha_factor']
```
