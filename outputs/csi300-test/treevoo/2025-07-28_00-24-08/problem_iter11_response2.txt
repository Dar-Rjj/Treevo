```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with (date) as index and the generated alpha factor as values.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the relative position of the closing price within the price range
    rel_pos_close = (df['close'] - df['low']) / price_range
    
    # Calculate the relative position of the amount within the total volume
    rel_pos_amount = df['amount'] / (df['volume'] * df['close'])
    
    # Combine the two relative positions into a single alpha factor
    alpha_factor = 0.5 * rel_pos_close + 0.5 * rel_pos_amount
    
    return alpha_factor
```
