```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Intraday Trend Strength
    daily_range = df['high'] - df['low']
    intraday_pct_change = (df['close'] - df['open']) / df['open']
    intraday_trend_strength = intraday_pct_change / daily_range
    
    # Volume-Weighted High-Low Volatility
    high_low_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_weighted_volatility = (high_low_range * df['volume']) / avg_volume
    
    # Volume Adjusted Reversal Signal
    prev_day_volume_momentum = df['volume'] / df['volume'].shift(1)
    close_to_next_open_return = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    price_reversal_score = -close_to_next_open_return * prev_day_volume_momentum
    
    # Combine all factors
    composite_factor = (aggregate_returns 
                       + intraday_trend_strength 
                       + volume_weighted_volatility 
                       + price_reversal_score)
    
    # Smoothed Moving Average of Composite Factor
    sma_composite_factor = composite_factor.rolling(window=5).mean()
    smoothed_composite_factor = sma_composite_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_composite_factor
```
