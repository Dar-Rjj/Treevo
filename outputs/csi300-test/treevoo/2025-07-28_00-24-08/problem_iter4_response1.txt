```python
import pandas as pd

def heuristics_v2(df, n_periods=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=n_periods).mean()
    
    # Calculate Price Momentum
    price_momentum = (df['close'] / df['open'].shift(n_periods)) - 1
    
    # Weight High-Low Range by Volume
    volatility_adjusted = high_low_range * (avg_volume / df['volume'])
    
    # Combine Momentum and Volatility
    alpha_factor = price_momentum + volatility_adjusted
    
    return alpha_factor
```
