import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['Previous_Day_Overnight_Return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Aggregate Returns
    df['Aggregate_Returns'] = df['Intraday_Return'] + df['Previous_Day_Overnight_Return']
    
    # Adjust for Volume
    df['Volume_Adjusted_Returns'] = df['Aggregate_Returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['Weighted_Adjusted_Returns'] = df['Volume_Adjusted_Returns'] * df['Intraday_Volatility']
    
    # Smoothed Moving Average
    df['SMA_Weighted_Adjusted_Returns'] = df['Weighted_Adjusted_Returns'].rolling(window=5).mean()
    df['Exponential_Smoothed'] = df['SMA_Weighted_Adjusted_Returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['Short_Term_Price_Momentum'] = df['close'] - df['close'].shift(7)
    df['Long_Term_Price_Momentum'] = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    df['Price_Momentum_Ratio'] = df['Short_Term_Price_Momentum'] / df['Long_Term_Price_Momentum']
    
    # Integrate Intraday and Previous Day's Returns
    df['Integrated_Intraday_Prev_Return'] = df['Intraday_Return'] * (df['close'] / df['open'].shift(1)) - 1
    
    # Combine Price Momentum and Intraday Reversal
    df['Combined_Price_Momentum_Reversal'] = df['Price_Momentum_Ratio'] * df['Integrated_Intraday_Prev_Return']
    
    # Conditionally Adjust by Volume Surge
    df['7Day_Avg_Volume'] = df['volume'].rolling(window=7).mean()
    df['Volume_Above_Average'] = np.where(df['volume'] > df['7Day_Avg_Volume'], 1.8, 0.7)
    df['Conditionally_Adjusted_Factor'] = df['Combined_Price_Momentum_Reversal'] * df['Volume_Above_Average']
    
    # Average Volume Over a Period
    df['10Day_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    
    # Volume Change Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(10)
    
    # Confirm Momentum with Volume and Amount
    df['Volume_Positive_Filter'] = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    df['Amount_Positive_Filter'] = np.where(df['amount'] > df['amount'].shift(1), 1, 0)
    df['Momentum_Confirmation'] = df['Volume_Positive_Filter'] * df['Amount_Positive_Filter']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Exponential_Smoothed'] + df['Conditionally_Adjusted_Factor']) * df['Momentum_Confirmation']
    
    return df['Final_Alpha_Factor']
