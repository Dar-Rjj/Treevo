import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Sum of daily returns over the past 5 days
    df['sum_5d_returns'] = df['daily_return'].rolling(window=5).sum()
    
    # Standard deviation of high and low prices over the last 20 days
    df['std_high_low'] = df[['high', 'low']].rolling(window=20).std().mean(axis=1)
    
    # Average trading volume over the past 10 days
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Ratio of today's volume to the 10-day average volume
    df['volume_ratio'] = df['volume'] / df['avg_volume_10d']
    
    # Price momentum: difference between today's close and the lowest low over the last 30 days
    df['momentum'] = df['close'] - df['low'].rolling(window=30).min()
    
    # Correlation between daily return and volume over the last 60 days
    df['return_volume_corr'] = df['daily_return'].rolling(window=60).corr(df['volume'])
    
    # Number of days in the last 10 where the closing price was higher than the opening price
    df['up_days_count'] = (df['close'] > df['open']).rolling(window=10).sum()
    
    # Check if the closing price is within the top third of the (high, low) range more often than not in the past 7 days
    df['within_top_third'] = (df['close'] > (df['high'] + 2*df['low'])/3).rolling(window=7).mean() > 0.5
    
    # Combine all factors into a single alpha factor
    alpha_factor = (0.2 * df['sum_5d_returns'] 
                    - 0.15 * df['std_high_low'] 
                    + 0.15 * df['volume_ratio'] 
                    + 0.2 * df['momentum'] 
                    + 0.1 * df['return_volume_corr'] 
                    + 0.1 * df['up_days_count'] 
                    + 0.1 * df['within_top_third'])
    
    return alpha_factor.dropna()
