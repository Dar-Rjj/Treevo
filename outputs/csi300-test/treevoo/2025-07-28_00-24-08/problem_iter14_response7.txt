```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day EMA of the typical price
    ema_typical_14 = typical_price.ewm(span=14, adjust=False).mean()
    
    # Calculate the 14-day EMA of the money flow
    ema_money_flow_14 = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the alpha factor
    alpha_factor = (ema_money_flow_14 - ema_typical_14) / ema_typical_14
    
    return alpha_factor
```
