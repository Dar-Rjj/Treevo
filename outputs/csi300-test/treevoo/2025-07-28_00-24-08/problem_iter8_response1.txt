```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['is_above_avg_vol'] = (df['volume'] > df['avg_volume_5d']).astype(int)
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['vol_change_indicator'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['factor'] = df['intraday_return'] * df['close_to_open_return'] * df['price_momentum_ratio']
    df['factor'] *= df['is_above_avg_vol'] * 1.5 + (1 - df['is_above_avg_vol']) * 0.5
    
    # Combine Momentum and Volatility
    df['weighted_high_low_range'] = df['high_low_range'] * df['volume']
    
    # Confirm Momentum with Volume and Amount
    df['vol_change'] = df['volume'] - df['volume'].shift(1)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['positive_vol_filter'] = (df['vol_change'] > 0).astype(int)
    df['positive_amount_filter'] = (df['amount_change'] > 0).astype(int)
    
    # Final Alpha Factor
    df['final_factor'] = (df['factor'] 
                          * df['weighted_high_low_range'] 
                          * df['positive_vol_filter'] 
                          * df['positive_amount_filter'])
    
    return df['final_factor'].dropna()
```
