import pandas as pd

def heuristics_v2(df, n=10, m=20, p=20, close_weight=0.5, volume_weight=0.3, sentiment_weight=0.1, range_weight=0.1):
    # Calculate the rate of change in closing prices over n days
    df['close_change'] = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Compute the average volume over the past m days
    df['avg_volume'] = df['volume'].rolling(window=m).mean()
    
    # Determine the ratio of today's volume to the m-day average volume
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Evaluate if today's close is higher than today's open (bullish signal) or lower (bearish signal)
    df['is_bullish'] = (df['close'] > df['open']).astype(int)
    df['is_bearish'] = (df['close'] < df['open']).astype(int)
    
    # Check if the closing price is also above the high of the previous day for bullish
    df['bullish_strength'] = (df['close'] > df['high'].shift(1)).astype(int)
    
    # Check if the closing price is also below the low of the previous day for bearish
    df['bearish_strength'] = (df['close'] < df['low'].shift(1)).astype(int)
    
    # Calculate the difference between today's high and low
    df['daily_range'] = df['high'] - df['low']
    
    # Compare this with the average daily range over the last p days
    df['avg_daily_range'] = df['daily_range'].rolling(window=p).mean()
    df['range_ratio'] = df['daily_range'] / df['avg_daily_range']
    
    # Combine multiple factors into a composite alpha factor
    df['alpha_factor'] = (
        close_weight * df['close_change'] +
        volume_weight * df['volume_ratio'] +
        sentiment_weight * (df['is_bullish'] * df['bullish_strength'] - df['is_bearish'] * df['bearish_strength']) +
        range_weight * df['range_ratio']
    )
    
    return df['alpha_factor'].dropna()
