```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum.replace(0, np.nan)
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factor = (intraday_return * overnight_return) * price_momentum_ratio
    
    # Conditionally Adjust by Volume Metrics
    avg_volume = df['volume'].mean()
    integrated_factor = np.where(df['volume'] > avg_volume, integrated_factor * 1.7, integrated_factor * 0.3)
    
    # Weight High-Low Range by Volume
    high_low_range = df['high'] - df['low']
    volume_weighted_high_low = high_low_range * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low
    
    # Smoothed Moving Average
    sma_5 = adjusted_returns.rolling(window=5).mean()
    ema_smoothed = sma_5.ewm(span=5, adjust=False).mean()
    
    # Explore Momentum Factors
    sma_5_close = df['close'].rolling(window=5).mean()
    sma_20_close = df['close'].rolling(window=20).mean()
    momentum_sma = sma_5_close - sma_20_close
    
    ema_close = df['close'].ewm(span=5, adjust=False).mean()
    momentum_ema = ema_close - ema_close.shift(5)
    
    # Analyze Volatility
    price_range = df['high'] - df['low']
    atr_5 = price_range.rolling(window=5).mean()
    volatility = atr_5
    
    # Investigate Volume-Based Indicators
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    vwap_deviation = df['close'] - vwap
    
    # Consider Transaction Amounts
    amount_change = df['amount'] - df['amount'].shift(1)
    transaction_intensity = amount_change / volume_change.replace(0, np.nan)
    
    # Combine Multiple Factors
    composite_factor = 0.4 * momentum_sma + 0.3 * volatility + 0.3 * vwap_deviation
    
    # Incorporate Volume Change Indicator
    composite_factor = np.where(volume_change > 0, composite_factor * 1.5, composite_factor * 0.8)
    
    # Final Factor
    final_factor = integrated_factor * ema_smoothed * composite_factor
    
    return final_factor
```
