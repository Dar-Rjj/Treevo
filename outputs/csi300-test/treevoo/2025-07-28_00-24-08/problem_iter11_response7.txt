```python
import pandas as pd
import numpy as np
from scipy.stats import zscore
import talib

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / (df['long_term_momentum'] + 1e-6)  # add small constant to avoid division by zero
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    df['average_volume'] = df['volume'].rolling(window=10).mean()
    
    # Weight High-Low Range by Volume
    df['volume_weighted_high_low'] = df['high_low_range'] * (df['volume'] / df['average_volume'])
    
    # Calculate Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Combine Volume Adjusted Volatility and Volume Change
    df['combined_volatility'] = df['volume_weighted_high_low'] + df['volume_change']
    
    # Adjust for Combined Volatility
    df['adjusted_return'] = df['aggregate_return'] / (df['combined_volatility'] + 1e-6)  # add small constant to avoid division by zero
    
    # Smoothed Moving Average
    df['sma_adjusted_return'] = df['adjusted_return'].rolling(window=10).mean()
    df['ema_smoothed'] = df['sma_adjusted_return'].ewm(span=10, adjust=False).mean()
    
    # Introduce Trend Reversal Indicator
    df['rsi'] = talib.RSI(df['close'], timeperiod=14)
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['trend_direction'] = np.where(df['close'] > df['50_day_ma'], 1, -1)
    df['trend_reversal'] = np.where((df['rsi'] < 30) & (df['trend_direction'].diff() == -2), 1,
                                     np.where((df['rsi'] > 70) & (df['trend_direction'].diff() == 2), -1, 0))
    
    # Final Alpha Factor
    df['alpha_factor'] = df['price_momentum_ratio'] * df['combined_volatility'] * df['ema_smoothed'] * (1 + df['trend_reversal'])
    
    return df['alpha_factor']
```
