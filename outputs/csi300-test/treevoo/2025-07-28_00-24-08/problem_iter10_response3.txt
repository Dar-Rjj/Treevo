```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Average Volume Over a Period
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Weight High-Low Range by Volume
    df['vol_weighted_high_low'] = df['high_low_range'] * (df['volume'] / df['avg_volume'])
    
    # Calculate Price Momentum
    df['price_momentum'] = (df['close'] - df['open'].shift(5)) / df['open'].shift(5)
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_return'] * df['vol_weighted_high_low']
    
    # Smoothed Moving Average
    df['sma_adjusted_returns'] = df['adjusted_returns'].rolling(window=5).mean()
    
    # Apply Exponential Smoothing
    df['alpha_factor'] = df['sma_adjusted_returns'].ewm(span=5, adjust=False).mean()
    
    return df['alpha_factor']
```
