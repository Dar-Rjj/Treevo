```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    aggregate_return = intraday_return + overnight_return

    # Weight High-Low Range by Volume
    high_low_range = (df['high'] - df['low']) * df['volume']

    # Adjust for Combined Volatility
    adjusted_returns = aggregate_return * high_low_range

    # Smoothed Moving Average
    sma = adjusted_returns.rolling(window=5).mean()
    ema = sma.ewm(span=5, adjust=False).mean()

    # Short-Term Price Momentum
    short_momentum = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    long_momentum = df['close'] - df['close'].shift(20)

    # Price Momentum Ratio
    price_momentum_ratio = short_momentum / long_momentum

    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)

    # Integrate Factors
    integrated_factor = (intraday_return * overnight_return) * price_momentum_ratio

    # Conditionally Adjust by Volume Metrics
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_adjustment = np.where(df['volume'] > avg_volume, 1.5, 0.5)
    integrated_factor *= volume_adjustment

    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    adjusted_ema = ema * daily_dollar_volume

    # Volume Surge Indicator
    volume_surge = np.where(df['volume'] > avg_volume, 1, 0)

    # Final Factor
    final_factor = integrated_factor * adjusted_ema * volume_surge

    return final_factor
```
