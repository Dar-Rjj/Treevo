```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(10)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low = high_low_range * df['volume']
    
    # Adjust Volume-Weighted High-Low Range with Exponential Moving Average
    ema_volatility = volume_weighted_high_low.ewm(span=5).mean()
    
    # Combine Volume Adjusted Volatility and Volume Change
    combined_volatility_volume = ema_volatility + volume_change
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum_ratio * combined_volatility_volume
    
    return final_alpha_factor
```
