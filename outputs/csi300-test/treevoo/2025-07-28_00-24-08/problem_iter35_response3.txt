```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the average price
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the relative change in volume
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the return
    df['return'] = df['close'].pct_change()
    
    # Calculate the weighted momentum
    df['momentum'] = (df['close'] - df['open']) / df['avg_price']
    
    # Calculate the amount per volume
    df['amount_per_volume'] = df['amount'] / df['volume']
    
    # Combine factors into a single score
    df['score'] = (df['momentum'] * df['volume_change'] * df['return'] * df['amount_per_volume'])
    
    # Normalize the score
    df['alpha_factor'] = (df['score'] - df['score'].mean()) / df['score'].std()
    
    return df['alpha_factor']
```
