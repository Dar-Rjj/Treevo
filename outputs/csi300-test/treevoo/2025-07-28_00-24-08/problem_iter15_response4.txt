```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = (df['intraday_return'] + df['prev_close_to_open_return'])
    
    # Volume Surge Indicator
    df['avg_5d_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = (df['volume'] > df['avg_5d_volume']).astype(int)
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Integrate Factors
    df['factor'] = df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio']
    
    # Conditionally Adjust by Volume Metrics
    df['factor'] = df['factor'] * (1.5 if df['volume_surge'] == 1 else 0.5)
    
    # Calculate Volume-Weighted High-Low Range
    df['vol_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['vol_weighted_high_low_range_sum'] = df['vol_weighted_high_low_range'].rolling(window=5).sum()
    df['avg_5d_volume_sum'] = df['volume'].rolling(window=5).sum() / 5
    
    # Adjust for Combined Volatility
    df['adjusted_volatility'] = df['aggregate_returns'] * (df['vol_weighted_high_low_range_sum'] / df['avg_5d_volume_sum'])
    
    # Combine Momentum and Volatility
    df['final_factor'] = (df['short_term_momentum'] + df['long_term_momentum']) * df['adjusted_volatility']
    
    return df['final_factor']
```
