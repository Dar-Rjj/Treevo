```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Identify Trends in Daily Returns
    df['5_day_return_sum'] = df['daily_return'].rolling(window=5).sum()

    # Analyze Price Volatility
    df['20_day_volatility'] = df[['high', 'low']].std(axis=1).rolling(window=20).mean()

    # Explore Trading Volume Patterns
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['10_day_avg_volume']

    # Consider Price Momentum
    df['30_day_price_momentum'] = df['close'] - df['low'].rolling(window=30).min()

    # Investigate the Relationship Between Price Change and Volume
    df['60_day_return_volume_corr'] = df['daily_return'].rolling(window=60).corr(df['volume'])

    # Examine Market Strength
    df['10_day_market_strength'] = (df['close'] > df['open']).rolling(window=10).sum()

    # Look for Buying or Selling Pressure
    df['7_day_top_third_range'] = (df['close'] > (df['high'] + 2 * df['low']) / 3).rolling(window=7).mean()

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] * df['prev_close_to_open_return']

    # Short-Term Price Momentum
    df['5_day_price_momentum'] = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    df['20_day_price_momentum'] = df['close'] - df['close'].shift(20)

    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['5_day_price_momentum'] / df['20_day_price_momentum']

    # Volume-Weighted High-Low Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['volume_weighted_high_low_volatility'] = df['high_low_range'] * df['volume'] / df['10_day_avg_volume']

    # Integrate Factors
    df['integrated_factor'] = (df['aggregate_returns'] * df['price_momentum_ratio'])
    df['integrated_factor'] = df['integrated_factor'] * (1.5 if df['volume'] > df['10_day_avg_volume'] else 0.5)

    return df['integrated_factor']
```
