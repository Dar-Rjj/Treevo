```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on price momentum and volume change.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A series indexed by date representing the factor values.
    """
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum_ratio * volume_change
    
    return final_alpha_factor
```
