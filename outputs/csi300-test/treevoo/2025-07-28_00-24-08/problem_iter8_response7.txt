```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)

    # Volume Surge Indicator
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_surge = (df['volume'] > avg_volume_5d).astype(float)

    # Amount-Volume Ratio
    avg_amount_5d = df['amount'].rolling(window=5).sum()
    avg_volume_sum_5d = df['volume'].rolling(window=5).sum()
    avg_amount_volume_ratio_5d = avg_amount_5d / avg_volume_sum_5d
    current_amount_volume_ratio = df['amount'] / df['volume']
    amount_volume_ratio_above_avg = (current_amount_volume_ratio > avg_amount_volume_ratio_5d).astype(float)

    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum

    # High-Low Range
    high_low_range = df['high'] - df['low']

    # Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=20).mean()

    # Combine High-Low Range and Volume
    volume_weighted_high_low_range = high_low_range * df['volume']
    normalized_volatility = volume_weighted_high_low_range / avg_volume

    # Integrate Factors
    integrated_factor = (intraday_return * prev_close_to_open_return * price_momentum_ratio)
    integrated_factor = integrated_factor * volume_surge.apply(lambda x: 1.5 if x == 1 else 0.5)
    integrated_factor = integrated_factor * amount_volume_ratio_above_avg.apply(lambda x: 1.2 if x == 1 else 0.8)
    final_alpha_factor = integrated_factor * (short_term_momentum + normalized_volatility)

    return final_alpha_factor
```
