import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']
    
    # Adjust for Volume
    df['volume_adjusted_return'] = df['aggregate_return'] * df['volume']
    
    # Smoothed Moving Average
    df['sma_5_days'] = df['volume_adjusted_return'].rolling(window=5).mean()
    
    # Apply Exponential Smoothing
    alpha = 0.2  # Exponential smoothing factor
    df['smoothed_factor'] = df['sma_5_days'].ewm(alpha=alpha, adjust=False).mean()
    
    return df['smoothed_factor']
