```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close = df['close'].shift(1)
    close_to_open_return = (prev_close - df['open']) / prev_close
    
    # Aggregate Returns
    aggregate_returns = intraday_return + close_to_open_return
    
    # Volume Surge Indicator
    avg_5d_volume = df['volume'].rolling(window=5).mean()
    volume_surge_indicator = (df['volume'] > avg_5d_volume).astype(int)
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Combine High-Low Range and Volume
    vol_weighted_high_low_range = high_low_range * df['volume']
    normalized_volatility = vol_weighted_high_low_range / avg_volume
    
    # Adjust for Combined Volatility
    adjusted_aggregate_returns = aggregate_returns * normalized_volatility
    
    # Integrate Factors
    factor_values = (intraday_return * close_to_open_return) * price_momentum_ratio
    factor_values = factor_values * (1.5 if volume_surge_indicator else 0.5)
    final_factor = factor_values * (short_term_momentum + normalized_volatility)
    
    return final_factor
```
