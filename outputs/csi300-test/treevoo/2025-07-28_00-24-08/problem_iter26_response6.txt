import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['HighLowRange'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['PrevDayClose'] = df['close'].shift(1)
    df['OvernightReturn'] = (df['close'] - df['PrevDayClose']) / df['PrevDayClose']
    
    # Aggregate Returns
    df['AggregateReturns'] = df['IntradayReturn'] + df['OvernightReturn']
    
    # Weight by Intraday Volatility
    df['IntradayVolatility'] = df['IntradayReturn'].rolling(window=5).std()
    df['WeightedAggregateReturns'] = df['AggregateReturns'] * df['IntradayVolatility']
    
    # Adjust for Volume
    df['VolumeAdjustedReturns'] = df['WeightedAggregateReturns'] * df['volume']
    
    # Smoothed Moving Average
    df['SMA'] = df['VolumeAdjustedReturns'].rolling(window=5).mean()
    df['EMA'] = df['SMA'].ewm(span=5, adjust=False).mean()
    
    # Compute Price Momentum
    df['LogReturns'] = np.log(df['close']).diff()
    df['PriceMomentum'] = df['LogReturns'].rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['VolumeConfirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    df['FinalMomentumFactor'] = df['PriceMomentum'] * df['VolumeConfirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['HighLowVolatility'] = df['HighLowRange'] * df['IntradayVolatility']
    
    # Integrate All Components
    df['CombinedFactor'] = df['EMA'] + df['FinalMomentumFactor'] + df['HighLowVolatility']
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['ShortTermMomentum'] = df['close'] - df['close'].shift(5)
    df['LongTermMomentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['VolumeChangeIndicator'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['PriceMomentumRatio'] = df['ShortTermMomentum'] / df['LongTermMomentum']
    
    # Integrate Additional Factors
    df['IntradayReturnByPrevDayOpenClose'] = df['IntradayReturn'] * (df['close'] - df['open']).shift(1) / df['open'].shift(1)
    df['IntegratedFactor'] = df['IntradayReturnByPrevDayOpenClose'] * df['PriceMomentumRatio']
    
    # Conditionally Adjust by Volume Metrics
    volume_avg = df['volume'].rolling(window=20).mean()
    df['IntegratedFactor'] = np.where(df['volume'] > volume_avg, df['IntegratedFactor'] * 1.5, df['IntegratedFactor'] * 0.5)
    
    # Combine Momentum and Volatility
    df['CombinedMomentumVolatility'] = df['PriceMomentum'] + df['HighLowVolatility'] * df['volume']
    
    # Final Alpha Factor
    df['FinalAlphaFactor'] = df['CombinedFactor'] + df['IntegratedFactor'] + df['CombinedMomentumVolatility']
    
    return df['FinalAlphaFactor']
