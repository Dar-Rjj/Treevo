```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']

    # Intraday Trend Strength
    df['daily_range'] = df['high'] - df['low']
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']

    # Volume Adjusted Reversal Signal
    df['volume_momentum'] = df['volume'].pct_change()
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['price_reversal_score'] = -df['close_to_next_open_return']

    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = df['aggregate_return'] + df['intraday_pct_change'] + df['price_reversal_score']
    df['sma_composite'] = df['composite_factor'].rolling(window=5).mean()
    df['ema_composite'] = df['sma_composite'].ewm(span=5, adjust=False).mean()

    # Calculate Intraday Return Factors
    df['high_low_range_ratio'] = df['high'] / df['low']
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)

    # Volume Weighted Smoothed Returns
    df['sma_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['sma_volume'])

    # Summarize Momentum-Reversal Indicator
    df['momentum_reversal_indicator'] = df['intraday_return'] + df['volume_weighted_returns']
    df['momentum_reversal_sign'] = np.sign(df['momentum_reversal_indicator'])

    # Calculate Overnight Return Factor
    df['overnight_return_factor'] = df['open'] - df['close'].shift(1)

    # Combine Intraday and Overnight Factors
    df['combined_factor'] = df['intraday_return'] + df['overnight_return_factor']
    df['adjusted_combined_factor'] = df['combined_factor'] * (df['volume'] / df['sma_volume'])
    df['final_combined_sign'] = np.sign(df['adjusted_combined_factor'])

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Average Volume Over a Period
    df['average_volume'] = df['volume'].rolling(window=5).mean()

    # Weight High-Low Range by Volume
    df['volume_weighted_high_low_range'] = df['high_low_range'] * df['volume']

    # Normalize by Average Volume
    df['normalized_volatility'] = df['volume_weighted_high_low_range'] / df['average_volume']

    # Adjust for Combined Volatility
    df['adjusted_aggregate_return'] = df['aggregate_return'] * df['normalized_volatility']

    # Smoothed Moving Average
    df['sma_adjusted'] = df['adjusted_aggregate_return'].rolling(window=5).mean()
    df['ema_adjusted'] = df['sma_adjusted'].ewm(span=5, adjust=False).mean()

    return df['ema_adjusted'].dropna()
```
