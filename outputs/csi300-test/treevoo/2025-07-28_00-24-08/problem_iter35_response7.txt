```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change
    df['price_change'] = df['close'].pct_change()
    
    # Calculate the volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the daily range
    df['daily_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the average true range over 14 days
    df['atr_14'] = df['close'].rolling(window=14).apply(lambda x: x.max() - x.min(), raw=True) / df['close']
    
    # Calculate the money flow index
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['raw_money_flow'] = df['typical_price'] * df['volume']
    df['money_flow_index'] = df['raw_money_flow'].rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Combine the factors into a single score
    df['alpha_factor'] = (df['price_change'].rank(ascending=False)
                          + df['volume_change'].rank(ascending=False)
                          + df['daily_range'].rank(ascending=False)
                          + df['atr_14'].rank(ascending=False)
                          + df['money_flow_index'].rank(ascending=False))
    
    # Return the alpha factor
    return df['alpha_factor']
```
