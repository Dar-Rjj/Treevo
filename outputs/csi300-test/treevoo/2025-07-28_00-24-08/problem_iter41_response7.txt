import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return'].shift(1)
    
    # Weight High-Low Range by Volume
    df['weighted_high_low_range'] = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    df['adjusted_return'] = df['aggregate_return'] * df['weighted_high_low_range']
    
    # Smooth Adjusted Returns
    df['smoothed_adjusted_return'] = df['adjusted_return'].rolling(window=5).mean()
    df['smoothed_adjusted_return'] = df['smoothed_adjusted_return'].ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    df['integrated_factor'] = df['aggregate_return'] * df['momentum_ratio']
    df.loc[df['volume'] > df['volume'].mean(), 'integrated_factor'] *= 1.7
    df.loc[df['volume'] <= df['volume'].mean(), 'integrated_factor'] *= 0.3
    df.loc[df['volume_change'] > 0, 'integrated_factor'] *= 1.5
    df.loc[df['volume_change'] < 0, 'integrated_factor'] *= 0.8
    
    # Compute Price Momentum
    df['price_momentum'] = np.log(df['close']).diff().rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    df['final_momentum_factor'] = df['price_momentum'] * df['volume_confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['high_low_volatility'] = (df['high'] - df['low']) * (df['close'] - df['open'])
    
    # Integrate All Components
    df['combined_factor'] = df['smoothed_adjusted_return'] + df['final_momentum_factor'] + df['high_low_volatility']
    
    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = df['price_momentum'] * (df['high'] - df['low']) * df['volume']
    
    # Incorporate Historical Volatility
    df['historical_volatility'] = df['close'].pct_change().rolling(window=20).std() * np.sqrt(252)
    df['weighted_historical_volatility'] = df['historical_volatility'] * df['volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] + df['combined_momentum_volatility'] + df['weighted_historical_volatility']
    
    return df['alpha_factor']
