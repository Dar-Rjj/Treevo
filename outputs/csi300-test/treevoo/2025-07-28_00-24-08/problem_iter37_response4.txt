```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Combine Returns
    combined_return = intraday_return * overnight_return
    
    # Volume-weighted High-Low Range
    high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Volatility
    adjusted_return = combined_return * high_low_range
    
    # Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    long_term_momentum = df['close'] - df['close'].shift(20)
    price_momentum_ratio = short_term_momentum / long_term_momentum
    volume_diff = df['volume'] - df['volume'].shift(5)
    
    # Final Factor
    final_factor = adjusted_return * price_momentum_ratio
    
    # Conditionally Adjust by Volume Metrics
    avg_volume = df['volume'].mean()
    final_factor = final_factor.where(
        df['volume'] > avg_volume, 
        final_factor * 0.3
    ).where(
        df['volume'] <= avg_volume, 
        final_factor * 1.7
    )
    
    return final_factor.dropna()
```
