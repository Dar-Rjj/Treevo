```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period (5 days)
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['open'].shift(1) - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight High-Low Range by Volume and Normalize by Average Volume
    volume_weighted_high_low_range = high_low_range * df['volume'] / avg_volume
    
    # Volume Surge Indicator
    volume_surge_indicator = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    
    # Integrate Factors
    integrated_factors = intraday_return * prev_close_to_open_return
    integrated_factors = integrated_factors * (1.5 if volume_surge_indicator else 0.5)
    
    # Final Alpha Factor
    final_alpha_factor = volume_weighted_high_low_range + integrated_factors
    
    return final_alpha_factor
```
