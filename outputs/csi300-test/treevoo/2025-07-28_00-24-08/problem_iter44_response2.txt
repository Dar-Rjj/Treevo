```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the Exponential Moving Average (EMA) of money flow for a 14-day period
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the positive and negative money flow
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    
    # Calculate the ratio of positive to negative money flow
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    
    # Generate the alpha factor by combining EMA of money flow and money ratio
    alpha_factor = ema_money_flow * money_ratio
    
    return alpha_factor
```
