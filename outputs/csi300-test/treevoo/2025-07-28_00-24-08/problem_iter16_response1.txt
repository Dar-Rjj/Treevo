```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']
    
    # Adjust for Volume
    df['volume_adjusted_return'] = df['aggregate_return'] * df['volume']
    
    # Weight by Intraday Volatility
    df['weighted_adjusted_return'] = df['intraday_volatility'] * df['volume_adjusted_return']
    
    # Smoothed Moving Average
    df['sma_5'] = df['weighted_adjusted_return'].rolling(window=5).mean()
    df['smoothed_sma_5'] = df['sma_5'].ewm(alpha=0.2).mean()
    
    # Generate Final Momentum Factor
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['final_momentum_factor'] = df['price_momentum'] * df['volume_confirmation']
    
    # Combine All Factors
    df['combined_factor'] = df['smoothed_sma_5'] + df['final_momentum_factor']
    df['high_low_volatility'] = (df['high'] - df['low']) / df['close']
    df['final_alpha_factor'] = df['combined_factor'] + df['high_low_volatility']
    
    return df['final_alpha_factor']
```
