```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Volume Surge Indicator
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    volume_surge = df['volume'] > avg_5_day_volume
    
    # Integrate Intraday and Previous Day's Returns
    integrated_intraday_return = intraday_return * prev_close_to_open_return
    
    # Combine Price Momentum and Intraday Reversal
    combined_momentum_reversal = price_momentum_ratio * integrated_intraday_return
    
    # Conditionally Adjust by Volume Surge
    conditionally_adjusted_factor = combined_momentum_reversal.where(~volume_surge, combined_momentum_reversal * 1.5)
    conditionally_adjusted_factor = conditionally_adjusted_factor.where(volume_surge, combined_momentum_reversal * 0.5)
    
    return conditionally_adjusted_factor
```
