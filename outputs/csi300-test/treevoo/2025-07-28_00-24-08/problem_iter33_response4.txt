```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight High-Low Range by Volume
    high_low_range = (df['high'] - df['low'])
    volume_weighted_high_low = high_low_range * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low
    
    # Smoothed Moving Average
    sma_5_days = adjusted_returns.rolling(window=5).mean()
    ema_smoothed = sma_5_days.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factor = (intraday_return * overnight_return) * price_momentum_ratio
    average_volume = df['volume'].rolling(window=5).mean()
    volume_condition = (df['volume'] > average_volume).astype(int)
    integrated_factor *= volume_condition.apply(lambda x: 1.5 if x else 0.5)
    
    # Calculate Daily Dollar Volume
    daily_dollar_volume = df['volume'] * df['close']
    
    # Adjust Smoothed Moving Average by Daily Dollar Volume
    adjusted_sma = ema_smoothed / daily_dollar_volume
    
    # Volume Surge Indicator
    volume_surge = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    
    # Explore Momentum Factors
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    momentum = sma_5 - sma_20
    
    ema_t = df['close'].ewm(span=5, adjust=False).mean()
    ema_t_minus_5 = df['close'].shift(5).ewm(span=5, adjust=False).mean()
    momentum_ema = ema_t - ema_t_minus_5
    
    # Analyze Volatility
    price_range = df['high'] - df['low']
    atr_n = price_range.rolling(window=5).mean()
    volatility = atr_n
    
    # Investigate Volume-Based Indicators
    amount_change = df['amount'] - df['amount'].shift(1)
    transaction_intensity = amount_change / volume_change
    
    # Combine Multiple Factors
    w1, w2, w3 = 0.4, 0.3, 0.3
    composite = (w1 * momentum + w2 * volatility + w3 * transaction_intensity)
    
    # Generate Alpha Signals
    alpha_signal = np.where(composite > 0, 1, -1)
    
    # Final Factor
    final_factor = alpha_signal * integrated_factor * adjusted_sma * volume_surge
    
    return final_factor
```
