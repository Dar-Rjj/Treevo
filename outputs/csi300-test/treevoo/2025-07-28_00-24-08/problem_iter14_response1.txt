```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = np.log(df['close'] / df['open'])

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = np.log(df['open'] / df['close'].shift(1))

    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']

    # Sum over a window of 20 days
    df['momentum_20d'] = df['aggregate_returns'].rolling(window=20).sum()

    # Intraday Trend Strength
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']

    # Volume-Weighted High-Low Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['vol_weighted_high_low_vol'] = df['high_low_range'] * (df['volume'] / df['avg_volume_20d'])

    # Volume Adjusted Reversal Signal
    df['prev_volume_momentum'] = df['volume'] - df['volume'].shift(1)
    df['volume_pct_change'] = df['volume'].pct_change()
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['reversal_signal'] = -df['close_to_next_open_return']

    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = (df['momentum_20d'] + 
                              df['intraday_pct_change'] + 
                              df['vol_weighted_high_low_vol'] + 
                              df['reversal_signal'])
    df['sma_5d'] = df['composite_factor'].rolling(window=5).mean()
    df['ema_smoothed'] = df['sma_5d'].ewm(span=5, adjust=False).mean()

    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_filter'] = (df['volume_change'] > 0).astype(int)
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['ema_volume'])

    # Adjust for Combined Volatility
    df['volatility'] = df['aggregate_returns'].rolling(window=20).std()
    df['adjusted_momentum'] = df['momentum_20d'] * (df['volatility'] / df['volatility'].mean())

    # Final Alpha Factor
    df['price_intraday_momentum'] = (df['momentum_20d'] + 
                                     df['intraday_pct_change'] + 
                                     df['intraday_trend_strength'])
    df['final_alpha_factor'] = (df['price_intraday_momentum'] * 
                                df['volume_weighted_returns'])

    return df['final_alpha_factor']
```
