```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=20):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Sum over a window to get momentum
    df['momentum'] = df['log_return'].rolling(window=window).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Range Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Open-to-Close Direction
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Combine High-Low Range and Volume
    df['volume_weighted_high_low'] = df['high_low_range'] * df['volume']
    avg_volume = df['volume'].rolling(window=window).mean()
    df['normalized_high_low'] = df['volume_weighted_high_low'] / avg_volume
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_change'] = (df['volume_change'] > 0).astype(int)
    ema_volume = df['volume'].ewm(span=window, adjust=False).mean()
    df['volume_weighted_smoothed_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / ema_volume)
    
    # Adjust for Combined Volatility
    df['volatility_adjusted_momentum'] = df['aggregate_returns'] * df['normalized_high_low']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['momentum'] + df['high_low_ratio'] + df['open_close_direction']
    df['alpha_factor'] = df['alpha_factor'] * df['volume_weighted_smoothed_returns']
    
    return df['alpha_factor']
```
