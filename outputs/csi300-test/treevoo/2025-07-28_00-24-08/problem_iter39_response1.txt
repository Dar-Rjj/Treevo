import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple Daily Returns
    df['simple_daily_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Open-Close Spread
    df['open_close_spread'] = df['close'] - df['open']
    
    # Intraday Return (Enhanced)
    df['intraday_return_enhanced'] = (df['close'] - df['open']) / df['open']
    
    # Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return_enhanced'] + df['overnight_return']
    
    # Daily Volume Changes
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Moving Averages Crossover
    df['MA_50'] = df['close'].rolling(window=50).mean()
    df['MA_200'] = df['close'].rolling(window=200).mean()
    df['ma_crossover'] = df['MA_50'] - df['MA_200']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Generate Final Momentum Factor
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['final_momentum_factor'] = df['log_returns'].rolling(window=20).sum() * df['volume_confirmation']
    
    # Standard Deviation of Returns
    df['std_returns'] = df['simple_daily_return'].rolling(window=60).std()
    
    # High-Low Volatility

    # Daily Amount Changes
    df['daily_amount_change'] = df['amount'] - df['amount'].shift(1)

    df['amount_to_volume_ratio'] = df['amount'] / df['volume']

        w9 * df['daily_amount_change'] +

    )
