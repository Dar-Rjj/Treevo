```python
import pandas as pd

def heuristics_v2(df):
    # Price Movement Patterns
    # Simple Moving Average (SMA) Crossover
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    sma_crossover = sma_50 - sma_200
    
    # High-Low Range Volatility
    high_low_range = df['high'] - df['low']
    avg_high_low_21 = high_low_range.rolling(window=21).mean()
    high_low_volatility = high_low_range - avg_high_low_21
    
    # Volume Analysis
    # Volume Momentum
    vol_5 = df['volume'].rolling(window=5).mean()
    vol_20 = df['volume'].rolling(window=20).mean()
    volume_momentum = vol_5 - vol_20
    
    # Volume-Price Correlation
    volume_price_corr = df['close'].rolling(window=60).corr(df['volume'])
    
    # Combine all factors into a single alpha factor
    alpha_factor = 0.4 * sma_crossover + 0.3 * high_low_volatility + 0.2 * volume_momentum + 0.1 * volume_price_corr
    
    return alpha_factor
```
