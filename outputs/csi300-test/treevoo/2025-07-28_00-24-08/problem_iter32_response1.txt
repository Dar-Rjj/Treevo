import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Adjust for Volume
    adjusted_returns = aggregate_returns * df['volume']
    
    # Weight by Intraday Volatility
    weighted_adjusted_returns = intraday_volatility * adjusted_returns
    
    # Smoothed Moving Average
    sma = weighted_adjusted_returns.rolling(window=5).mean()
    exponential_smoothing_factor = 0.2
    ema = sma.ewm(alpha=exponential_smoothing_factor, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    long_term_momentum = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Intraday and Previous Day's Returns
    integrated_returns = intraday_return * overnight_return
    
    # Combine Price Momentum and Intraday Reversal
    combined_price_momentum = price_momentum_ratio * integrated_returns
    
    # Conditionally Adjust by Volume Surge
    avg_7_day_volume = df['volume'].rolling(window=7).mean()
    volume_surge = (df['volume'] > avg_7_day_volume).astype(int)
    conditional_adjustment = combined_price_momentum * (1.8 * volume_surge + 0.7 * (1 - volume_surge))
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = intraday_volatility * (df['high'] - df['low'])
    
    # Generate Final Alpha Factor
    daily_dollar_volume = df['volume'] * df['close']
    final_alpha_factor = ema + conditional_adjustment + high_low_volatility
    final_alpha_factor = final_alpha_factor / daily_dollar_volume
    
    # Additional Enhancements
    rsi = compute_rsi(df['close'], window=14)
    macd_line = compute_macd(df['close'])
    rsi_macd_combination = rsi * macd_line
    vwap = compute_vwap(df)
    final_alpha_factor += rsi_macd_combination + vwap / daily_dollar_volume
    
    return final_alpha_factor

def compute_rsi(close, window=14):
    delta = close.diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def compute_macd(close):
    ema_12 = close.ewm(span=12, adjust=False).mean()
    ema_26 = close.ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    return macd_line

def compute_vwap(df):
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    return vwap
