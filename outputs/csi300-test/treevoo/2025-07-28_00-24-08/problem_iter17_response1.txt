import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low_range
    
    # Smoothed Moving Average
    sma_5 = adjusted_returns.rolling(window=5).mean()
    smoothed_sma = sma_5.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Factors
    integrated_factor = (intraday_return * overnight_return) * price_momentum_ratio
    average_volume = df['volume'].rolling(window=5).mean().shift(1)
    integrated_factor = integrated_factor.where(df['volume'] > average_volume, integrated_factor * 0.5)
    integrated_factor = integrated_factor.where(df['volume'] <= average_volume, integrated_factor * 1.5)
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    final_factor = smoothed_sma * (daily_dollar_volume / daily_dollar_volume.rolling(window=5).mean())
    
    return final_factor
