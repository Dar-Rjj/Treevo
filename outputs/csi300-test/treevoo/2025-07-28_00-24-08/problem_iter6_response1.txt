import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)

    # Volume Surge Indicator
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = df['volume'] > df['5_day_avg_volume']

    # Amount-Volume Ratio
    df['amount_volume_ratio'] = df['amount'] / df['volume']
    df['5_day_avg_amount_volume_ratio'] = (df['amount'].rolling(window=5).sum() / 
                                           df['volume'].rolling(window=5).sum())
    df['amount_volume_ratio_surge'] = df['amount_volume_ratio'] > df['5_day_avg_amount_volume_ratio']

    # Integrate Factors
    df['alpha_factor'] = df['intraday_return'] * df['prev_close_to_open_return']

    # Adjust by Volume Surge Indicator
    df.loc[df['volume_surge'], 'alpha_factor'] *= 1.5
    df.loc[~df['volume_surge'], 'alpha_factor'] *= 0.5

    # Adjust by Amount-Volume Ratio
    df.loc[df['amount_volume_ratio_surge'], 'alpha_factor'] *= 1.2
    df.loc[~df['amount_volume_ratio_surge'], 'alpha_factor'] *= 0.8

    return df['alpha_factor'].dropna()
