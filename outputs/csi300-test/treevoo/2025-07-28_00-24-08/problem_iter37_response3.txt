```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_vol'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Adjust for Volume
    df['volume_adjusted_returns'] = df['aggregate_returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['weighted_adjusted_returns'] = df['intraday_vol'] * df['volume_adjusted_returns']
    
    # Smoothed Moving Average
    df['sma_5'] = df['weighted_adjusted_returns'].rolling(window=5).mean()
    df['ema_smoothed'] = df['sma_5'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)
    df['long_term_momentum'] = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Intraday and Previous Day's Returns
    df['integrated_returns'] = df['intraday_return'] * df['overnight_return']
    
    # Combine Price Momentum and Intraday Reversal
    df['combined_momentum_reversal'] = df['momentum_ratio'] * df['integrated_returns']
    
    # Conditionally Adjust by Volume Surge
    df['avg_volume_7'] = df['volume'].rolling(window=7).mean()
    df['volume_above_avg'] = df['volume'] > df['avg_volume_7']
    df['conditionally_adjusted'] = df['combined_momentum_reversal'] * np.where(df['volume_above_avg'], 1.8, 0.7)
    
    # Calculate Daily Dollar Volume
    df['daily_dollar_volume'] = df['volume'] * df['close']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['ema_smoothed'] + df['conditionally_adjusted'] + df['intraday_vol']
    df['final_alpha_factor'] = df['final_alpha_factor'] / df['daily_dollar_volume']
    
    return df['final_alpha_factor']
```
