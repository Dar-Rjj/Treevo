import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()
    
    # Compute Price Momentum
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'].diff()
    df['positive_volume'] = (df['volume_change'] > 0).astype(int)
    
    # Integrate Amount for Momentum Strengthening
    df['amount_change'] = df['amount'].diff()
    df['positive_amount'] = (df['amount_change'] > 0).astype(int)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] / df['open']) - 1
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Combine High-Low Range and Volume
    df['weighted_high_low'] = df['high_low_range'] * df['volume']
    df['normalized_volatility'] = df['weighted_high_low'] / df['avg_volume']
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['normalized_volatility']
    
    # Introduce Trend Reversal Indicator
    df['rsi'] = compute_rsi(df['close'], 14)
    df['trend_direction'] = (df['close'] > df['close'].rolling(window=50).mean()).astype(int)
    df['trend_reversal'] = ((df['rsi'] < 30) | (df['rsi'] > 70)) & (df['trend_direction'].diff() != 0)
    
    # Smoothed Moving Average
    df['sma_adjusted_returns'] = df['adjusted_returns'].rolling(window=5).mean()
    df['ema_smoothed'] = df['sma_adjusted_returns'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['combined_factor'] = df['ema_smoothed'] * (df['trend_reversal'] + 1)
    df['alpha_signal'] = np.sign(df['combined_factor'])
    df['final_alpha_factor'] = df['price_momentum'] * df['positive_volume'] * df['positive_amount'] * df['adjusted_returns'] * df['alpha_signal']
    
    # Incorporate Trading Activity
    df['turnover_ratio'] = df['volume'] / df['share_capital']
    df['final_alpha_factor'] *= df['turnover_ratio']
    
    # Add Market Sentiment Indicator
    df['market_sentiment'] = compute_market_sentiment(df['news_sentiment_score'], df['social_media_sentiment_score'])
    df['final_alpha_factor'] *= df['market_sentiment']
    
    # Incorporate Breakout Indicator
    df['breakout_level'] = df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()
    df['breakout_indicator'] = (df['close'] > df['breakout_level']).astype(int)
    df['final_alpha_factor'] *= df['breakout_indicator']
    
    return df['final_alpha_factor']

def compute_rsi(series, period):
    delta = series.diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=period).mean()
    avg_loss = loss.rolling(window=period).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def compute_market_sentiment(news_sentiment, social_media_sentiment):
    return (news_sentiment + social_media_sentiment) / 2
