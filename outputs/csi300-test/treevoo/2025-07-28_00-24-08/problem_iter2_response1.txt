```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    previous_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Volume Change Indicator
    volume_change_indicator = df['volume'] - df['volume'].shift(5)
    
    # Volume Surge Indicator
    average_5_day_volume = df['volume'].rolling(window=5).mean()
    volume_surge_indicator = (df['volume'] > average_5_day_volume).astype(int)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Factors
    integrated_factor = (intraday_return * previous_close_to_open_return * price_momentum_ratio)
    
    # Conditionally Adjust by Volume Dynamics
    volume_adjustment = volume_surge_indicator * 1.5 + (1 - volume_surge_indicator) * 0.5
    final_factor = integrated_factor * (volume_change_indicator * volume_adjustment)
    
    return final_factor
```
