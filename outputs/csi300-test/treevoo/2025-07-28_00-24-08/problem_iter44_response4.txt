```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pandas.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pandas.Series: A Series with index as date representing the factor values.
    """
    # Calculate the normalized volume
    df['norm_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Calculate the relative return
    df['rel_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the volatility
    df['volatility'] = df['high'] - df['low']
    
    # Calculate the price change
    df['price_change'] = df['close'].pct_change()
    
    # Alpha factor: weighted combination of normalized volume, relative return, and volatility
    df['alpha_factor'] = (0.5 * df['norm_volume'] + 0.3 * df['rel_return'] - 0.2 * df['volatility']) * df['price_change'].rolling(window=10).mean()
    
    return df['alpha_factor']
```
