```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Combine Volume Adjusted Volatility and Volume Change
    combined_volatility_and_volume_change = (volume_weighted_high_low_range + volume_change) / 2
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum_ratio * combined_volatility_and_volume_change
    
    return final_alpha_factor
```
