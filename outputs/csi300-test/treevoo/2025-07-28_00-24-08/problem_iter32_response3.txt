```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['Overnight_Return'] = df['close'].pct_change()
    
    # Aggregate Returns
    df['Aggregate_Returns'] = df['Intraday_Return'] + df['Overnight_Return'].shift(1)
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['Intraday_Return'].rolling(window=5).std()
    
    # Weight by Intraday Volatility
    df['Weighted_Aggregate_Returns'] = df['Aggregate_Returns'] * df['Intraday_Volatility']
    
    # Adjust for Volume
    df['Volume_Adjusted_Returns'] = df['Weighted_Aggregate_Returns'] * df['volume']
    
    # Smoothed Moving Average
    df['SMA'] = df['Volume_Adjusted_Returns'].rolling(window=5).mean()
    df['Smoothed_SMA'] = df['SMA'].ewm(span=5, adjust=False).mean()
    
    # Compute Price Momentum
    df['Log_Returns'] = np.log(df['close']).diff()
    df['Price_Momentum'] = df['Log_Returns'].rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['Volume_Confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    df['Final_Momentum_Factor'] = df['Price_Momentum'] * df['Volume_Confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['High_Low_Volatility'] = df['High_Low_Range'] * df['Intraday_Volatility']
    
    # Integrate All Components
    df['Integrated_Factor'] = df['Smoothed_SMA'] + df['Final_Momentum_Factor'] + df['High_Low_Volatility']
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['Short_Term_Price_Momentum'] = df['close'] - df['close'].shift(5)
    df['Long_Term_Price_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Price Momentum Ratio
    df['Price_Momentum_Ratio'] = df['Short_Term_Price_Momentum'] / df['Long_Term_Price_Momentum']
    
    # Integrate Additional Factors
    df['Prev_Day_Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Intraday_Return_Prev_Day_Adj'] = df['Intraday_Return'] * df['Prev_Day_Close_to_Open_Return']
    df['Adjusted_Factor'] = df['Intraday_Return_Prev_Day_Adj'] * df['Price_Momentum_Ratio']
    
    # Conditionally Adjust by Volume Metrics
    avg_volume = df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Factor'] = df['Adjusted_Factor'] * np.where(df['volume'] > avg_volume, 1.5, 0.5)
    
    # Combine Momentum and Volatility
    df['Combined_Factor'] = df['Price_Momentum'] + df['High_Low_Range'] * df['volume']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Integrated_Factor'] + df['Volume_Adjusted_Factor'] + df['Combined_Factor']
    
    return df['Final_Alpha_Factor']
```
