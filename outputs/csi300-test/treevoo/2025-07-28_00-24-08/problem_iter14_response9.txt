```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_day_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Volume Surge Indicator
    avg_7d_volume = df['volume'].rolling(window=7).mean()
    volume_surge = df['volume'] > avg_7d_volume
    
    # Integrate Intraday and Previous Day's Returns
    integrated_return = intraday_return * prev_day_close_to_open_return
    
    # Combine Price Momentum and Intraday Reversal
    combined_momentum = price_momentum_ratio * integrated_return
    
    # Conditionally Adjust by Volume Surge
    volume_adjusted_factor = combined_momentum.where(~volume_surge, combined_momentum * 1.8)
    volume_adjusted_factor = volume_adjusted_factor.where(volume_surge, combined_momentum * 0.7)
    
    # Final Alpha Factor
    alpha_factor = volume_adjusted_factor
    
    return alpha_factor
```
