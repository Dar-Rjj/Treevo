```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the relative difference between the close and open price
    df['rel_diff_close_open'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the relative difference between the high and low price
    df['rel_diff_high_low'] = (df['high'] - df['low']) / df['low']
    
    # Calculate the weighted moving average of the amount
    df['wma_amount'] = df['amount'].ewm(span=10, adjust=False).mean()
    
    # Calculate the ratio of volume to wma_amount
    df['vol_to_wma_amount_ratio'] = df['volume'] / df['wma_amount']
    
    # The alpha factor is a combination of these components
    df['alpha_factor'] = (df['rel_diff_close_open'] * 0.4 +
                          df['rel_diff_high_low'] * 0.3 +
                          df['vol_to_wma_amount_ratio'] * 0.3)
    
    return df['alpha_factor']
```
