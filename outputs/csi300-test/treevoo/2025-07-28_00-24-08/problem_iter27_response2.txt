```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the difference between high and low prices, normalized by volume
    vol_normalized_range = (df['high'] - df['low']) / df['volume']
    
    # Calculate the money flow index (MFI) using typical price and volume
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = raw_money_flow.where(typical_price < typical_price.shift(1), 0)
    money_flow_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (close_to_avg_ratio + vol_normalized_range + mfi) / 3
    
    return alpha_factor
```
