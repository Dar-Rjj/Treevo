import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Adjust for Volume
    df['volume_adjusted_returns'] = df['aggregate_returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['weighted_returns'] = df['intraday_volatility'] * df['volume_adjusted_returns']
    
    # Smoothed Moving Average
    df['sma'] = df['weighted_returns'].rolling(window=5).mean()
    df['ema'] = df['sma'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Intraday and Previous Day's Returns
    df['integrated_returns'] = df['intraday_return'] * df['overnight_return']
    
    # Combine Price Momentum and Intraday Reversal
    df['combined_momentum'] = df['momentum_ratio'] * df['integrated_returns']
    
    # Conditionally Adjust by Volume Surge
    df['avg_volume_7d'] = df['volume'].rolling(window=7).mean()
    df['volume_surge'] = np.where(df['volume'] > df['avg_volume_7d'], 1.8, 0.7)
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['volume_surge']
    
    # Incorporate High-Low Volatility into Final Factor
    df['high_low_volatility'] = df['intraday_volatility'] * (df['high'] - df['low'])
    
    # Generate Final Alpha Factor
    df['daily_dollar_volume'] = df['volume'] * df['close']
    df['final_factor'] = df['ema'] + df['adjusted_combined_momentum'] + df['high_low_volatility']
    df['final_factor'] = df['final_factor'] / df['daily_dollar_volume']
    
    return df['final_factor']
