```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']).diff()

    # Compute Price Momentum
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()

    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'].diff()
    df['positive_volume_filter'] = (df['volume_change'] > 0).astype(int)

    # Integrate Amount for Momentum Strengthening
    df['amount_change'] = df['amount'].diff()
    df['positive_amount_filter'] = (df['amount_change'] > 0).astype(int)

    # Calculate Intraday Return
    df['intraday_return'] = np.log(df['close'] / df['open'])

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = np.log(df['close'] / df['close'].shift(1))

    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Average Volume Over a Period
    df['avg_volume'] = df['volume'].rolling(window=20).mean()

    # Combine High-Low Range and Volume
    df['volatility_weighted_high_low'] = df['high_low_range'] * df['volume'] / df['avg_volume']

    # Adjust for Combined Volatility
    df['adjusted_return'] = df['aggregate_return'] * df['volatility_weighted_high_low']

    # Introduce Trend Reversal Indicator
    def rsi(series, n=14):
        delta = series.diff()
        dUp, dDown = delta.copy(), delta.copy()
        dUp[dUp < 0] = 0
        dDown[dDown > 0] = 0
        RolUp = dUp.rolling(window=n).mean()
        RolDown = dDown.rolling(window=n).mean().abs()
        RS = RolUp / RolDown
        RSI = 100.0 - (100.0 / (1.0 + RS))
        return RSI

    df['rsi'] = rsi(df['close'])
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['trend_direction'] = (df['close'] > df['50_day_ma']).astype(int)
    df['trend_reversal_indicator'] = ((df['rsi'] > 70) | (df['rsi'] < 30)) & (df['trend_direction'].diff() != 0).astype(int)

    # Smoothed Moving Average
    df['sma_adjusted_return'] = df['adjusted_return'].rolling(window=5).mean()
    df['ema_smoothed'] = df['sma_adjusted_return'].ewm(span=5, adjust=False).mean()

    # Final Factor Calculation
    df['final_factor'] = (df['price_momentum'] * 
                          df['positive_volume_filter'] * 
                          df['positive_amount_filter'] * 
                          df['ema_smoothed'] * 
                          (1 + df['trend_reversal_indicator']))

    return df['final_factor']
```
