import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    prev_close = df['close'].shift(1)
    overnight_return = (df['open'] - prev_close) / prev_close
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight High-Low Range by Volume
    high_low_range = df['high'] - df['low']
    volume_weighted_high_low = high_low_range * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low
    
    # Smoothed Moving Average
    sma_5 = adjusted_returns.rolling(window=5).mean()
    ema_smoothed = sma_5.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factor = (intraday_return * overnight_return * price_momentum_ratio)
    
    # Conditionally Adjust by Volume Metrics
    average_volume = df['volume'].rolling(window=7).mean()
    integrated_factor = integrated_factor * np.where(df['volume'] > average_volume, 1.5, 0.5)
    
    # Additional Components
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Adjust for Volume
    volume_adjusted_returns = adjusted_returns * df['volume']
    
    # Weight by Intraday Volatility
    weighted_adjusted_returns = intraday_volatility * volume_adjusted_returns
    
    # Smoothed Moving Average with New Components
    sma_5_new = weighted_adjusted_returns.rolling(window=5).mean()
    ema_smoothed_new = sma_5_new.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum (7 days)
    short_term_momentum_7 = df['close'] - df['close'].shift(7)
    
    # Long-Term Price Momentum (25 days)
    long_term_momentum_25 = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio_7_25 = short_term_momentum_7 / long_term_momentum_25
    
    # Integrate Intraday and Previous Day's Returns
    integrated_intraday_overnight = intraday_return * overnight_return
    
    # Combine Price Momentum and Intraday Reversal
    combined_price_momentum_reversal = price_momentum_ratio_7_25 * integrated_intraday_overnight
    
    # Conditionally Adjust by Volume Surge
    volume_surge = np.where(df['volume'] > average_volume, 1.8, 0.7)
    conditionally_adjusted_factor = combined_price_momentum_reversal * volume_surge
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = intraday_volatility * high_low_range
    
    # Generate Final Alpha Factor
    daily_dollar_volume = df['volume'] * df['close']
    final_alpha_factor = (ema_smoothed_new + conditionally_adjusted_factor + high_low_volatility) / daily_dollar_volume
    
    return final_alpha_factor
