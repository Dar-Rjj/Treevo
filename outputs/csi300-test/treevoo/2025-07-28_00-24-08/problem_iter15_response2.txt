import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, alpha=0.2, beta=0.1):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low = high_low_range * (df['volume'] / avg_volume)
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['open'].shift(n)) / df['open'].shift(n)
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low * price_momentum
    
    # Smoothed Moving Average
    ema_adjusted_returns = adjusted_returns.ewm(span=n, adjust=False).mean()
    smoothed_ema = ema_adjusted_returns.ewm(alpha=alpha, beta=beta, adjust=False).mean()
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    
    # Adjust Smoothed Moving Average by Daily Dollar Volume
    final_factor = smoothed_ema * (daily_dollar_volume / daily_dollar_volume.rolling(window=n).mean())
    
    return final_factor.dropna()
