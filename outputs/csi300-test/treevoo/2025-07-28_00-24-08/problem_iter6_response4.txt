```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute daily log return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Sum over a window (last 20 days log returns) to get momentum
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Intraday High-Low Range Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Open-to-Close Direction (1 if close > open, else 0)
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Positive Volume Change Filter
    df['positive_volume_change'] = (df['volume_change'] > 0).astype(int)
    
    # Calculate Exponential Moving Average of Volumes
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Compute Volume Weighted Smoothed Returns
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['ema_volume'])
    
    # Combine Price and Intraday Momentum
    df['combined_momentum'] = df['price_momentum'] + df['high_low_ratio'] + df['open_close_direction']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_momentum'] * df['volume_weighted_returns']
    
    return df['alpha_factor']
```
