```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Adjust for Volume
    volume_adjusted_returns = aggregate_returns * df['volume']
    
    # Weight by Intraday Volatility
    weighted_adjusted_returns = volume_adjusted_returns * intraday_volatility
    
    # Smoothed Moving Average
    sma_5 = weighted_adjusted_returns.rolling(window=5).mean()
    smoothed_factor = sma_5.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
