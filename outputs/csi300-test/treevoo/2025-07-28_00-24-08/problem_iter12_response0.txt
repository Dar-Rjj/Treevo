import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['Prev_Day_Close'] = df['close'].shift(1)
    df['Overnight_Return'] = (df['close'] - df['Prev_Day_Close']) / df['Prev_Day_Close']
    
    # Aggregate Returns
    df['Aggregate_Returns'] = df['Intraday_Return'] + df['Overnight_Return']
    
    # Adjust for Volume
    df['Volume_Adjusted_Returns'] = df['Aggregate_Returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['Weighted_Adjusted_Returns'] = df['Intraday_Volatility'] * df['Volume_Adjusted_Returns']
    
    # Smoothed Moving Average
    df['SMA_5'] = df['Weighted_Adjusted_Returns'].rolling(window=5).mean()
    df['EMA'] = df['SMA_5'].ewm(span=5, adjust=False).mean()
    
    # Calculate Daily Log Return
    df['Daily_Log_Return'] = np.log(df['close']).diff()
    
    # Compute Price Momentum
    df['Price_Momentum'] = df['Daily_Log_Return'].rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['Volume_Confirmation'] = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    
    # Generate Final Momentum Factor
    df['Final_Momentum_Factor'] = df['Price_Momentum'] * df['Volume_Confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['High_Low_Volatility'] = df['High_Low_Range'] * df['Intraday_Volatility']
    
    # Combine All Factors
    df['Final_Factor'] = df['EMA'] + df['Final_Momentum_Factor'] + df['High_Low_Volatility']
    
    return df['Final_Factor'].dropna()
