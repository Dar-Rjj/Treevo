```python
import pandas as pd

def heuristics_v2(df, n=20):
    """
    Generate a hybrid intraday momentum and reversal alpha factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as (date).
    n (int): Number of days to compute the Simple Moving Average of Volumes.
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Intraday High-Low Range Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Open-to-Close Direction
    open_close_direction = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    # Calculate Simple Moving Average of Volumes
    sma_volume = df['volume'].rolling(window=n).mean()
    
    # Compute Volume Weighted Returns
    adj_close_changes = df['close'].diff(1)
    volume_weighted_returns = adj_close_changes * (df['volume'] / sma_volume)
    
    # Summarize Momentum-Reversal Indicator
    momentum_reversal_indicator = high_low_ratio + open_close_direction + volume_weighted_returns
    final_factor = momentum_reversal_indicator.apply(lambda x: 1 if x > 0 else -1)
    
    return final_factor
```
