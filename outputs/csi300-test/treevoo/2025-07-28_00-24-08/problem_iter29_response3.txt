```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open + close) / 2 to get a simple moving average
    typical_price = (df['open'] + df['close']) / 2
    
    # Calculate the volume weighted price using the formula: (typical_price * volume) / amount
    volume_weighted_price = (typical_price * df['volume']) / df['amount']
    
    # Calculate the difference between the high and low prices for each day
    price_range = df['high'] - df['low']
    
    # Define a factor that captures the ratio of volume weighted price to the price range
    factor = volume_weighted_price / price_range
    
    # Smooth the factor with a 5-day simple moving average to reduce noise
    smoothed_factor = factor.rolling(window=5).mean()
    
    return smoothed_factor
```
