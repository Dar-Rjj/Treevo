```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Long-Term Price Momentum
    df['short_term_mom'] = df['close'].diff(5)
    df['long_term_mom'] = df['close'].diff(20)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Surge Indicator
    avg_5d_volume = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = np.where(df['volume'] > avg_5d_volume, 1.5, 0.5)
    
    # Compute Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_mom'] / df['long_term_mom']
    
    # Integrate Factors
    df['integrated_factors'] = df['intraday_return'] * df['prev_close_to_open'] * df['volume_surge']
    
    # Calculate Intraday Return (Close to Open)
    df['intraday_return_co'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return_co'] + df['overnight_return']
    
    # Intraday Trend Strength
    df['daily_range'] = df['high'] - df['low']
    df['intraday_pct_change'] = df['intraday_return_co']
    
    # Volume Adjusted Reversal Signal
    df['volume_momentum'] = df['volume'].pct_change()
    df['close_to_next_open'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['price_reversal_score'] = -df['close_to_next_open']
    
    # Smoothed Moving Average of Composite Factor
    composite_factor = df['aggregate_returns'] + df['intraday_pct_change'] + df['price_reversal_score']
    df['smoothed_ma'] = composite_factor.rolling(window=5).mean()
    df['exp_smoothed_ma'] = df['smoothed_ma'].ewm(span=5, adjust=False).mean()
    
    # Additional Volume-Adjusted Reversal Factor
    df['5d_volume_momentum'] = (df['volume'] - avg_5d_volume) / avg_5d_volume
    df['5d_price_reversal_score'] = (df['open'].rolling(window=5).mean() - df['close'].rolling(window=5).mean()) / df['close'].rolling(window=5).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['price_momentum_ratio'] * df['integrated_factors'] 
                                + df['exp_smoothed_ma'] 
                                + df['5d_volume_momentum'] * df['5d_price_reversal_score'])
    
    return df['final_alpha_factor']
```
