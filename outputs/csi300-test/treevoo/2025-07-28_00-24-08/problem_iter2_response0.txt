```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Surge Indicator
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    volume_surge = df['volume'] > avg_5_day_volume
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Factors
    integrated_factors = intraday_return * prev_close_to_open_return
    integrated_factors = integrated_factors.where(volume_surge, integrated_factors * 0.5) * 1.5
    
    # Final Alpha Factor
    final_alpha_factor = price_momentum_ratio * integrated_factors
    
    return final_alpha_factor
```
