```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)

    # Volume Surge Indicator
    df['avg_5d_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = (df['volume'] > df['avg_5d_volume']).astype(int)

    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)

    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Average Volume Over a Period
    df['avg_20d_volume'] = df['volume'].rolling(window=20).mean()

    # Combine High-Low Range and Volume
    df['vol_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    df['normalized_volatility'] = df['vol_weighted_high_low_range'] / df['avg_20d_volume']

    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio']
    df['integrated_factor'] = df['integrated_factor'] * (1.5 if df['volume_surge'] == 1 else 0.5)
    df['final_alpha_factor'] = df['integrated_factor'] + df['normalized_volatility']

    return df['final_alpha_factor']
```
