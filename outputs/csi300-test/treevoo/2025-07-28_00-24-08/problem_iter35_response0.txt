```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index as date and values as the generated alpha factor.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the positive and negative money flow
    prev_typical_price = typical_price.shift(1)
    positive_money_flow = money_flow.where(typical_price > prev_typical_price, 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(typical_price < prev_typical_price, 0).rolling(window=14).sum()
    
    # Calculate the money flow ratio
    money_flow_ratio = positive_money_flow / negative_money_flow.abs()
    
    # Calculate the money flow index
    alpha_factor = 100 - (100 / (1 + money_flow_ratio))
    
    return alpha_factor
```
