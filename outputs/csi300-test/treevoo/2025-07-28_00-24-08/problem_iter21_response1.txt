import pandas as pd
import numpy as np

def heuristics_v2(df, sma_window=50, vol_lookback=20, avg_vol_period=30, high_low_period=10, bullish_pattern_days=7):
    """
    Generate a composite alpha factor based on price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as date.
    sma_window (int): Number of days to calculate the Simple Moving Average.
    vol_lookback (int): Lookback period for calculating volatility.
    avg_vol_period (int): Period over which to average the volume.
    high_low_period (int): Number of days to sum the high-low spread.
    bullish_pattern_days (int): Number of days to count consecutive bullish patterns.
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    
    # Momentum-based Factor
    df['SMA'] = df['close'].rolling(window=sma_window).mean()
    df['Momentum'] = df['close'] - df['SMA']
    
    # Volatility-based Factor
    df['Daily_Return'] = df['close'].pct_change()
    df['Volatility'] = df['Daily_Return'].rolling(window=vol_lookback).std()
    
    # Volume-related Factor
    df['Avg_Volume'] = df['volume'].rolling(window=avg_vol_period).mean()
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume']
    
    # High-Low Spread Indicator
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Sum_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=high_low_period).sum()
    
    # Price Patterns
    df['Bullish_Pattern'] = (df['close'] > df['open']).astype(int)
    df['Count_Bullish_Pattern'] = df['Bullish_Pattern'].rolling(window=bullish_pattern_days).sum()
    
    # Combined Metrics
    df['Composite_Factor'] = (df['Momentum'] + df['Volatility'] + df['Volume_Ratio'] + 
                              1/df['Sum_High_Low_Spread'] + df['Count_Bullish_Pattern']) / 5
    
    return df['Composite_Factor'].dropna()
