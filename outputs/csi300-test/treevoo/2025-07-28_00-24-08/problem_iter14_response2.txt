import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open'] - 1
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = df['open'] / df['close'].shift(1) - 1
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Intraday Trend Strength
    df['daily_range'] = (df['high'] - df['low']) / df['low']
    df['intraday_trend_strength'] = df['close'] / df['open'] - 1
    
    # Volume Adjusted Reversal Signal
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    df['close_to_next_open_return'] = (df['open'].shift(-1) / df['close']) - 1
    df['reversal_signal'] = -df['close_to_next_open_return']
    
    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = df['aggregate_returns'] + df['intraday_trend_strength'] + df['reversal_signal']
    df['sma_composite_factor'] = df['composite_factor'].rolling(window=5).mean()
    df['ema_composite_factor'] = df['sma_composite_factor'].ewm(span=5, adjust=False).mean()
    
    # Intraday Return Factors
    df['high_low_ratio'] = df['high'] / df['low']
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    df['sma_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['sma_volume'])
    
    # Summarize Momentum-Reversal Indicator
    df['momentum_reversal_indicator'] = df['intraday_return'] + df['volume_weighted_returns']
    df['momentum_reversal_sign'] = np.sign(df['momentum_reversal_indicator'])
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    df['avg_volume'] = df['volume'].rolling(window=5).mean()
    
    # Weight High-Low Range by Volume
    df['weighted_high_low_range'] = df['high_low_range'] * df['volume'] / df['avg_volume']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] / df['open'].shift(5) - 1
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['weighted_high_low_range']
    
    # Smoothed Moving Average
    df['sma_adjusted_returns'] = df['adjusted_returns'].rolling(window=5).mean()
    df['ema_adjusted_returns'] = df['sma_adjusted_returns'].ewm(span=5, adjust=False).mean()
    
    # Integrate Amount Data
    df['daily_dollar_volume'] = df['volume'] * df['close']
    df['final_factor'] = df['ema_adjusted_returns'] * df['daily_dollar_volume']
    
    return df['final_factor'].dropna()
