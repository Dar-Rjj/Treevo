import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Average Volume Over a Period
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    above_avg_volume = (df['volume'] > avg_volume_10d).astype(float)

    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(10)

    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / (long_term_momentum + 1e-6)  # Avoid division by zero

    # Integrate Factors
    integrated_factor = (intraday_return * prev_close_to_open_return * price_momentum_ratio)
    integrated_factor *= (1.5 * above_avg_volume) + (0.5 * (1 - above_avg_volume))

    # Combine Momentum and Volatility
    high_low_weighted = high_low_range * df['volume']
    combined_momentum_volatility = integrated_factor * high_low_weighted

    # Confirm Momentum with Volume and Amount
    volume_change_indicator = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    amount_change_indicator = (df['amount'] - df['amount'].shift(1)).apply(lambda x: 1 if x > 0 else 0)

    # Final Alpha Factor
    final_alpha_factor = combined_momentum_volatility * volume_change_indicator * amount_change_indicator

    return final_alpha_factor
