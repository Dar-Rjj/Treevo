import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] / df['open']) - 1
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['open'] / df['close'].shift(1)) - 1
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return.shift(1)
    
    # Intraday Trend Strength
    daily_range = df['high'] - df['low']
    intraday_pct_change = (df['close'] - df['open']) / df['open']
    
    # Volume Adjusted Reversal Signal
    pct_change_volume = (df['volume'] / df['volume'].shift(1)) - 1
    close_to_next_open_return = (df['open'].shift(-1) / df['close']) - 1
    inverse_close_to_next_open = -close_to_next_open_return
    
    # Smoothed Moving Average of Composite Factor
    composite_factor = (aggregate_returns + intraday_pct_change + inverse_close_to_next_open).rolling(window=5).mean()
    smoothed_moving_average = composite_factor.ewm(span=5, adjust=False).mean()
    
    # Intraday High-Low Range Ratio
    high_low_ratio = df['high'] / df['low']
    
    # Open-to-Close Direction
    open_to_close_direction = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    volume_sma = df['volume'].rolling(window=5).mean()
    volume_weighted_returns = (df['close'] - df['close'].shift(1)) * (df['volume'] / volume_sma)
    
    # Summarize Momentum-Reversal Indicator
    momentum_reversal_indicator = aggregate_returns + volume_weighted_returns
    final_factor_sign = np.sign(momentum_reversal_indicator)
    
    # Weight High-Low Range by Volume
    weighted_high_low_range = high_low_ratio * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * weighted_high_low_range
    
    # Smoothed Moving Average
    sma_adjusted_returns = adjusted_returns.rolling(window=5).mean()
    smoothed_adjusted_returns = sma_adjusted_returns.ewm(span=5, adjust=False).mean()
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    adjusted_smoothed_moving_average = smoothed_adjusted_returns * daily_dollar_volume
    
    return adjusted_smoothed_moving_average
