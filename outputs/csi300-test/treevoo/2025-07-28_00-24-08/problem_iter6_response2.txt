```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] / df['close'].shift(n) - 1
    
    # Calculate Price Change Factor
    price_change_factor = (df['close'] - df['open']) / df['open']
    
    # Weight High-Low Range by Volume
    weighted_high_low = high_low_range * (df['volume'] / avg_volume)
    
    # Combine Momentum, Volatility, and Price Change
    factor = (price_momentum + weighted_high_low + price_change_factor) / 3
    
    return factor
```
