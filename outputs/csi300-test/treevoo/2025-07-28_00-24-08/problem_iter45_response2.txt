import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return.shift(1)
    
    # Weight High-Low Range by Volume
    high_low_range_vol_weighted = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * high_low_range_vol_weighted
    
    # Smoothed Moving Average
    sma = adjusted_returns.rolling(window=5).mean()
    ema = sma.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(25)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factors = (intraday_return * overnight_return) * price_momentum_ratio
    volume_adjusted_factors = np.where(df['volume'] > df['volume'].rolling(window=7).mean(), 
                                       integrated_factors * 1.7, 
                                       integrated_factors * 0.3)
    
    # Final Factor
    final_factor = volume_adjusted_factors + ema
    
    return final_factor
