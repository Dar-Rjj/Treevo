```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight High-Low Range by Volume
    high_low_range = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * high_low_range
    
    # Smoothed Moving Average
    sma_5 = adjusted_returns.rolling(window=5).mean()
    smoothed_moving_average = sma_5.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factor = intraday_return * overnight_return * price_momentum_ratio
    integrated_factor = np.where(df['volume'] > df['volume'].rolling(window=5).mean(), 
                                 integrated_factor * 1.5, 
                                 integrated_factor * 0.5)
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    smoothed_moving_average = smoothed_moving_average * daily_dollar_volume
    
    # Volume Surge Indicator
    volume_surge = np.where(df['volume'] > df['volume'].rolling(window=5).mean(), 1, 0)
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = (df['high'] - df['low'])
    high_low_adjusted = high_low_volatility * intraday_return
    
    # Generate Final Momentum Factor
    final_momentum = (np.log(df['close']) - np.log(df['close'].shift(1))).rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    
    # Combine Price Momentum and Intraday Reversal
    combined_price_momentum = price_momentum_ratio * (intraday_return * overnight_return)
    
    # Conditionally Adjust by Volume Surge
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    volume_surge_condition = np.where(df['volume'] > seven_day_avg_volume, 1.8, 0.7)
    combined_price_momentum = combined_price_momentum * volume_surge_condition
    
    # Integrate All Components
    final_alpha_factor = (smoothed_moving_average + 
                          final_momentum + 
                          high_low_adjusted + 
                          combined_price_momentum)
    
    return final_alpha_factor
```
