```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the relative range
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the money flow
    df['money_flow'] = df['amount'] / df['volume']
    
    # Calculate the weighted moving average of the money flow
    df['wma_money_flow'] = df['money_flow'].rolling(window=10).mean()
    
    # Calculate the alpha factor
    df['alpha_factor'] = df['return'] * df['range'] * df['wma_money_flow']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
```
