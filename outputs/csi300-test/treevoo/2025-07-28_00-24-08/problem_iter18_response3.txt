import pandas as pd

def heuristics_v2(data):
    """
    Generate a novel and interpretable alpha factor that could help predict future stock returns.
    
    Parameters:
    data (pandas.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pandas.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate typical price
    typical_price = (data['high'] + data['low'] + data['close']) / 3
    
    # Calculate average volume over the last 10 days
    avg_volume = data['volume'].rolling(window=10).mean()
    
    # Calculate the percentage change in typical price
    pct_change_typical_price = typical_price.pct_change()
    
    # Calculate the weighted average of percentage change and normalized volume
    weight_typical_price = 0.7
    weight_volume = 0.3
    normalized_volume = (data['volume'] - avg_volume) / avg_volume
    factor_values = (weight_typical_price * pct_change_typical_price) + (weight_volume * normalized_volume)
    
    return factor_values
