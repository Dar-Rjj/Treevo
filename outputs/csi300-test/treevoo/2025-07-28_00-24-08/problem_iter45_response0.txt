    # Momentum-based Factors
    df['ema_close_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['roc_close'] = df['close'].pct_change()
    df['rs_10'] = df['close'] / df['close'].rolling(window=10).mean()

    # Volatility-based Factors
    df['daily_range'] = df['high'] - df['low']
    df['atr_14'] = df['high'].combine(df['close'].shift(1), max) - df['low'].combine(df['close'].shift(1), min)
    df['atr_14'] = df['atr_14'].rolling(window=14).mean()

    # Trend Strength Indicators
    df['dmi_14_pos'] = df['high'].diff().apply(lambda x: max(x, 0))
    df['dmi_14_neg'] = -df['low'].diff().apply(lambda x: min(x, 0))
    df['dmi_14_pos'] = df['dmi_14_pos'].rolling(window=14).sum()
    df['dmi_14_neg'] = df['dmi_14_neg'].rolling(window=14).sum()
    df['adx_14'] = 100 * (abs(df['dmi_14_pos'] - df['dmi_14_neg']) / (df['dmi_14_pos'] + df['dmi_14_neg']))
    df['rsi_14'] = 100 - (100 / (1 + df['close'].diff().apply(lambda x: x if x > 0 else 0).rolling(window=14).mean() / -df['close'].diff().apply(lambda x: x if x < 0 else 0).rolling(window=14).mean()))

    df['vol_oscillator'] = df['volume'].rolling(window=5).mean() - df['volume'].rolling(window=10).mean()
