```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate Price Momentum
    open_n_periods_ago = df['open'].shift(1)
    price_momentum = (df['close'] - open_n_periods_ago) / open_n_periods_ago
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    prev_day_close = df['close'].shift(1)
    overnight_return = (df['close'] - prev_day_close) / prev_day_close
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Adjust for Volume
    volume_adjusted_returns = aggregate_returns * df['volume']
    
    # Weight by Intraday Volatility
    weighted_returns = volume_adjusted_returns * intraday_volatility
    
    # Smoothed Moving Average
    sma_weighted_returns = weighted_returns.rolling(window=5).mean()
    smoothed_returns = sma_weighted_returns.ewm(alpha=0.2).mean()
    
    # Combine Momentum, Volatility, and Smoothed Adjusted Returns
    high_low_volatility_adjusted = high_low_range * avg_volume
    combined_factor = (price_momentum + high_low_volatility_adjusted + smoothed_returns)
    
    return combined_factor
```
