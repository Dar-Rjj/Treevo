```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight by Intraday Volatility
    weighted_aggregate_returns = aggregate_returns * intraday_volatility
    
    # Adjust for Volume
    volume_adjusted_returns = weighted_aggregate_returns * df['volume']
    
    # Smoothed Moving Average
    sma_5 = volume_adjusted_returns.rolling(window=5).mean()
    smoothed_sma = sma_5.ewm(span=5, adjust=False).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close']).diff()
    
    # Compute Price Momentum
    price_momentum = daily_log_return.rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    final_momentum_factor = price_momentum * volume_confirmation
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = high_low_range * intraday_volatility
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    adjusted_smoothed_sma = smoothed_sma * daily_dollar_volume
    
    # Average Volume Over a Period
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_above_average = (df['volume'] > avg_volume_5).astype(int)
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Factors
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    integrated_factor = intraday_return * close_to_open_return * price_momentum_ratio
    integrated_factor = integrated_factor * (1.5 if volume_above_average else 0.5)
    
    # Combine All Factors
    final_alpha_factor = (
        adjusted_smoothed_sma 
        + final_momentum_factor 
        + high_low_volatility 
        + integrated_factor
    )
    
    return final_alpha_factor
```
