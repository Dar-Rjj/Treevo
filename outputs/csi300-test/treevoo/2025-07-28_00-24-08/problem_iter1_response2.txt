```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Determine Direction of Price Movement
    df['trend'] = 'Uptrend'
    df.loc[df['price_change'] <= 0, 'trend'] = 'Downtrend'
    
    # Calculate Volume Delta
    df['volume_delta'] = df['volume'] - df['volume'].shift(1)
    
    # Assess Volume Confirmation
    df['volume_confirmation'] = 'None'
    df.loc[(df['volume_delta'] > 0) & (df['price_change'] > 0), 'volume_confirmation'] = 'Strong Uptrend'
    df.loc[(df['volume_delta'] < 0) & (df['price_change'] < 0), 'volume_confirmation'] = 'Strong Downtrend'
    
    # Factor Calculation
    df['factor_value'] = 0.0
    df.loc[(df['trend'] == 'Uptrend') & (df['volume_confirmation'] == 'Strong Uptrend'), 'factor_value'] = 1.5 * df['price_change']
    df.loc[(df['trend'] == 'Downtrend') & (df['volume_confirmation'] == 'Strong Downtrend'), 'factor_value'] = -1.5 * df['price_change']
    
    return df['factor_value']
```
