import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['integrated_factor'] = (df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio'])
    df['integrated_factor'] = df['integrated_factor'] * 1.5 if df['volume'] > df['avg_volume_5d'] else df['integrated_factor'] * 0.5
    
    # Combine Momentum and Volatility
    df['momentum_volatility'] = df['short_term_momentum'] * (df['high_low_range'] * df['volume'])
    
    # Generate Final Momentum Factor
    df['final_momentum_factor'] = df['momentum_volatility'] * (df['volume_change'] > 0)
    
    # Incorporate High-Low Volatility into Final Factor
    df['high_low_volatility'] = df['high_low_range'] * df['intraday_return']
    
    # Integrate Amount Data
    df['daily_dollar_volume'] = df['volume'] * df['close']
    df['final_momentum_factor'] = df['final_momentum_factor'] * df['daily_dollar_volume']
    
    # Combine All Factors
    df['combined_factor'] = (df['integrated_factor'] + df['final_momentum_factor'] + df['high_low_volatility'])
    
    return df['combined_factor']
