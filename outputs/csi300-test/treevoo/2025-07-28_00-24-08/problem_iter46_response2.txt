import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Short-Term Price Momentum
    short_momentum = df['close'] - df['close'].shift(7)
    
    # Long-Term Price Momentum
    long_momentum = df['close'] - df['close'].shift(25)
    
    # Price Momentum Ratio
    momentum_ratio = short_momentum / long_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factor = (aggregate_returns * momentum_ratio).copy()
    
    # Conditionally Adjust by Volume Metrics
    average_volume = df['volume'].mean()
    integrated_factor[df['volume'] > average_volume] *= 1.5
    integrated_factor[df['volume'] <= average_volume] *= 0.5
    
    # Final Factor
    final_factor = integrated_factor
    
    return final_factor
