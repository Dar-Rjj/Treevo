```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    :param df: A pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
               indexed by (date).
    :return: A pandas Series representing the factor values, indexed by (date).
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the relative range: (High - Low) / Close
    relative_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume-weighted price change
    vwp_change = (df['close'] - df['open']) * df['volume'] / df['volume'].mean()
    
    # Calculate the amount momentum: (Current Amount / Previous Amount) - 1
    df['amount_momentum'] = df['amount'].pct_change()
    
    # Combine the factors into a single alpha factor
    alpha_factor = daily_return + relative_range + vwp_change + df['amount_momentum']
    
    # Normalize the alpha factor
    alpha_factor = (alpha_factor - alpha_factor.mean()) / alpha_factor.std()
    
    return alpha_factor
```
