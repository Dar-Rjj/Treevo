```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Average Volume Over a Period
    df['avg_volume_10_days'] = df['volume'].rolling(window=10).mean()
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio']
    
    # Conditionally Adjust by Volume Metrics
    df['integrated_factor_adjusted'] = df['integrated_factor'] * 1.5 * (df['volume'] > df['avg_volume_10_days'])
    df['integrated_factor_adjusted'] += df['integrated_factor'] * 0.5 * (df['volume'] <= df['avg_volume_10_days'])
    
    # Combine Momentum and Volatility
    df['weighted_high_low_range'] = (df['high'] - df['low']) * df['volume']
    
    # Confirm Momentum with Volume and Amount
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume'] = (df['volume_change'] > 0).astype(int)
    
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['positive_amount'] = (df['amount_change'] > 0).astype(int)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor_adjusted'] * df['weighted_high_low_range'] * df['positive_volume'] * df['positive_amount']
    
    return df['final_alpha_factor']
```
