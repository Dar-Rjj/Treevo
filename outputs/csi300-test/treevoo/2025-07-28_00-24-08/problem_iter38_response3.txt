import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10 day moving average of the close price
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30 day moving average of the close price
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between 10-day and 30-day moving averages
    df['sma_diff'] = df['sma_10'] - df['sma_30']
    
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Factor: (SMA Difference * RSI) / Volume
    df['alpha_factor'] = (df['sma_diff'] * df['rsi']) / df['volume']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
