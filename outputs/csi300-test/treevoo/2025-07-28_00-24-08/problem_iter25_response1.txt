import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Identify trends in daily returns (sum of daily returns over the past 5 days)
    df['return_trend'] = df['daily_return'].rolling(window=5).sum()
    
    # Analyze price volatility (standard deviation of high and low prices over the last 20 days)
    df['price_volatility'] = df[['high', 'low']].rolling(window=20).std().mean(axis=1)
    
    # Explore trading volume patterns
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_10']
    
    # Consider price momentum (difference between today's close and the lowest low over the last 30 days)
    df['price_momentum'] = df['close'] - df['low'].rolling(window=30).min()
    
    # Investigate the relationship between price change and volume (correlation between daily return and volume over the last 60 days)
    df['return_volume_corr'] = df[['daily_return', 'volume']].rolling(window=60).corr().iloc[::2]['daily_return']
    
    # Examine market strength (number of days in the last 10 where the closing price was higher than the opening price)
    df['positive_days'] = (df['close'] > df['open']).rolling(window=10).sum()
    
    # Look for buying or selling pressure (closing price in the top third of the (high, low) range more often than not in the past 7 days)
    df['top_third'] = (df['close'] >= (df['low'] + (df['high'] - df['low']) * 2/3)).rolling(window=7).mean() > 0.5
    
    # Short-term price momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-term price momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price momentum ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Volume-weighted high-low volatility
    df['hl_range'] = df['high'] - df['low']
    df['volume_weighted_hl'] = df['hl_range'] * df['volume']
    df['avg_volume_60'] = df['volume'].rolling(window=60).mean()
    df['vol_adj_volatility'] = df['volume_weighted_hl'] / df['avg_volume_60']
    
    # Integrate factors
    factor = (df['return_trend'] 
              + df['price_momentum'] 
              + df['momentum_ratio'] 
              + df['vol_adj_volatility'] 
              + (df['positive_days'] / 10) 
              + (df['top_third'] * 2))
    
    # Conditionally adjust by volume metrics
    factor = factor.where(df['volume'] <= df['avg_volume_10'], factor * 1.5)
    factor = factor.where(df['volume'] > df['avg_volume_10'], factor * 0.5)
    
    return factor.dropna()
