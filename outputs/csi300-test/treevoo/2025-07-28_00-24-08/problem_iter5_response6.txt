import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_momentum = df['close'] - df['close'].shift(20)
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Volume Surge Indicator
    avg_5d_volume = df['volume'].rolling(window=5).mean()
    volume_surge = (df['volume'] > avg_5d_volume).astype(int)
    
    # Compute Price Momentum Ratio
    momentum_ratio = short_momentum / long_momentum
    
    # Integrate Factors
    integrated_factors = (intraday_return * prev_close_to_open) * (1.5 if volume_surge else 0.5)
    
    # Calculate Intraday Return
    intraday_return_close_open = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)
    
    # Aggregate Returns
    aggregate_returns = intraday_return_close_open + overnight_return
    
    # Intraday Trend Strength
    daily_range = df['high'] - df['low']
    intraday_percentage_change = (df['close'] - df['open']) / df['open']
    
    # Volume Adjusted Reversal Signal
    volume_momentum = df['volume'].pct_change()
    close_to_next_open = (df['close'] - df['open'].shift(-1)) / df['close']
    price_reversal_score = -close_to_next_open
    
    # Smoothed Moving Average of Composite Factor
    composite_factor = aggregate_returns + intraday_percentage_change + price_reversal_score
    sma_composite = composite_factor.rolling(window=5).mean()
    ema_composite = sma_composite.ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_ratio * integrated_factors) + ema_composite
    
    return final_alpha_factor
