```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Sum over a window to get Price Momentum
    momentum_window = 20
    df['price_momentum'] = df['log_return'].rolling(window=momentum_window).sum()
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Positive Volume Change Filter
    df['volume_confirmation'] = (df['volume_change'] > 0).astype(int)
    
    # Final Alpha Factor: Multiply Price Momentum by Volume Confirmation
    df['alpha_factor'] = df['price_momentum'] * df['volume_confirmation']
    
    return df['alpha_factor']
```
