```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = np.log(df['close'] / df['open'])
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = np.log(df['close'].shift(1) / df['close'].shift(2))
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']
    
    # Sum over a window of 20 days
    df['momentum'] = df['aggregate_return'].rolling(window=20).sum()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Range Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Open-to-Close Direction
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Combine High-Low Range and Volume
    df['high_low_volume_weighted'] = df['high_low_range'] * df['volume']
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume_change'] = (df['volume_change'] > 0).astype(int)
    
    # Exponential Moving Average of Volumes
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Compute Volume Weighted Returns
    df['adj_close_diff'] = df['close'] - df['close'].shift(1)
    df['volume_weighted_returns'] = df['adj_close_diff'] * (df['volume'] / df['ema_volume'])
    
    # Adjust for Combined Volatility
    df['volatility'] = df['aggregate_return'].rolling(window=20).std()
    df['normalized_volatility'] = df['volatility'] / df['volatility'].mean()
    df['adjusted_momentum'] = df['momentum'] * df['normalized_volatility']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['momentum'] +
        df['high_low_ratio'] +
        df['open_close_direction']
    ) * df['volume_weighted_returns']
    
    return df['alpha_factor']
```
