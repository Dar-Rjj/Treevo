```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change
    df['price_change'] = df['close'].diff()
    
    # Calculate the relative volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Momentum factor: 5-day return
    df['momentum_5d'] = df['close'].pct_change(5)
    
    # Volatility factor: 10-day standard deviation of returns
    df['volatility_10d'] = df['close'].pct_change().rolling(window=10).std()
    
    # Money flow factor: 5-day average of (typical price * volume)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow_5d'] = (df['typical_price'] * df['volume']).rolling(window=5).mean()
    
    # Calculate the heuristic alpha factor
    df['heuristic_factor'] = (df['momentum_5d'] / df['volatility_10d']) * df['money_flow_5d'] * df['volume_change']
    
    # Return the heuristic factor as a pandas Series
    return df['heuristic_factor'].dropna()
```
