```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']

    # Intraday Trend Strength
    df['daily_range'] = df['high'] - df['low']
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']

    # Volume Adjusted Reversal Signal
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['reversal_score'] = 0 - df['close_to_next_open_return']

    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = df['aggregate_returns'] + df['intraday_pct_change'] + df['reversal_score']
    df['sma_composite_factor'] = df['composite_factor'].rolling(window=5).mean()
    df['exp_smoothed_factor'] = df['sma_composite_factor'].ewm(span=5, adjust=False).mean()

    # Intraday Return Factors
    df['intraday_high_low_ratio'] = df['high'] / df['low']
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)

    # Volume Weighted Smoothed Returns
    df['sma_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['sma_volume'])

    # Summarize Momentum-Reversal Indicator
    df['momentum_reversal_indicator'] = df['intraday_high_low_ratio'] + df['open_close_direction'] + df['volume_weighted_returns']
    df['momentum_reversal_indicator_sign'] = np.sign(df['momentum_reversal_indicator'])

    # Calculate Overnight Return Factor
    df['overnight_return_factor'] = df['open'] - df['close'].shift(1)

    # Combine Intraday and Overnight Factors
    df['combined_factor'] = df['momentum_reversal_indicator'] + df['overnight_return_factor']
    df['final_factor'] = df['combined_factor'] * (df['volume'] / df['sma_volume'])
    df['final_factor_sign'] = np.sign(df['final_factor'])

    return df['final_factor']
```
