import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Intraday High-Low Range Ratio
    df['High_Low_Range_Ratio'] = df['high'] / df['low']
    
    # Open-to-Close Direction
    df['Open_Close_Direction'] = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    df['Volume_EMA'] = df['volume'].ewm(span=n, adjust=False).mean()
    df['Volume_Weighted_Return'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['Volume_EMA'])
    
    # Momentum-Reversal Indicator
    df['Momentum_Reversal'] = df['High_Low_Range_Ratio'] + df['Open_Close_Direction'] + df['Volume_Weighted_Return']
    
    # Additional Intraday Volatility Factor
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['SMA_True_Range'] = df['True_Range'].rolling(window=n).mean()
    df['Normalized_Volatility'] = df['True_Range'] / df['SMA_True_Range']
    
    # Combined Momentum-Volatility Indicator
    df['Combined_Indicator'] = df['Momentum_Reversal'] + df['Normalized_Volatility']
    df['Final_Factor'] = df['Combined_Indicator'].apply(lambda x: 1 if x > 0 else -1)
    
    return df['Final_Factor']
