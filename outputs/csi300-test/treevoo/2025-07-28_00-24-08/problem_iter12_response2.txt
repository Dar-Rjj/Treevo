```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Volume Surge Indicator
    avg_volume_7d = df['volume'].rolling(window=7).mean()
    volume_surge = (df['volume'] > avg_volume_7d).astype(float)
    
    # Integrate Intraday and Previous Day's Returns
    integrated_intraday_returns = intraday_return * prev_close_to_open_return
    
    # Combine Price Momentum and Intraday Reversal
    combined_momentum_reversal = momentum_ratio * integrated_intraday_returns
    
    # Conditionally Adjust by Volume Surge
    adjusted_factor = combined_momentum_reversal * (1.8 * volume_surge + 0.7 * (1 - volume_surge))
    
    return adjusted_factor
```
