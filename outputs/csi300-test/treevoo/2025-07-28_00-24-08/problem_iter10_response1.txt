```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate daily log return
    data['log_return'] = np.log(data['close']) - np.log(data['close'].shift(1))
    
    # Compute price momentum
    data['price_momentum'] = data['log_return'].rolling(window=20).sum()
    
    # Confirm momentum with volume
    data['volume_filter'] = (data['volume'] > data['volume'].shift(1)).astype(int)
    
    # Final alpha factor
    data['alpha_factor'] = data['price_momentum'] * data['volume_filter']
    
    return data['alpha_factor']
```
