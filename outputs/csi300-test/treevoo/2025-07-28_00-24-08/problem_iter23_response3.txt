```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['Prev_Day_Overnight_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['Aggregate_Returns'] = df['Intraday_Return'] + df['Prev_Day_Overnight_Return']
    
    # Adjust for Volume
    df['Volume_Adjusted_Returns'] = df['Aggregate_Returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['Weighted_Adjusted_Returns'] = df['Intraday_Volatility'] * df['Volume_Adjusted_Returns']
    
    # Smoothed Moving Average
    df['SMA_Adjusted_Returns'] = df['Weighted_Adjusted_Returns'].rolling(window=5).mean()
    df['EMA_Adjusted_Returns'] = df['SMA_Adjusted_Returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['Short_Term_Momentum'] = df['close'] - df['close'].shift(7)
    df['Long_Term_Momentum'] = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    df['Price_Momentum_Ratio'] = df['Short_Term_Momentum'] / df['Long_Term_Momentum']
    
    # Integrate Intraday and Previous Day's Returns
    df['Integrated_Intraday_Returns'] = df['Intraday_Return'] * df['Prev_Day_Overnight_Return']
    
    # Combine Price Momentum and Intraday Reversal
    df['Combined_Price_Momentum'] = df['Price_Momentum_Ratio'] * df['Integrated_Intraday_Returns']
    
    # Conditionally Adjust by Volume Surge
    df['7Day_Avg_Volume'] = df['volume'].rolling(window=7).mean()
    df['Volume_Surge_Factor'] = df.apply(lambda x: 1.8 if x['volume'] > x['7Day_Avg_Volume'] else 0.7, axis=1)
    df['Conditionally_Adjusted_Price_Momentum'] = df['Combined_Price_Momentum'] * df['Volume_Surge_Factor']
    
    # Incorporate High-Low Volatility into Final Factor
    df['Final_High_Low_Volatility'] = df['High_Low_Range'] * df['Intraday_Volatility']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['EMA_Adjusted_Returns'] + 
                                df['Conditionally_Adjusted_Price_Momentum'] + 
                                df['Final_High_Low_Volatility'])
    
    return df['Final_Alpha_Factor']
```
