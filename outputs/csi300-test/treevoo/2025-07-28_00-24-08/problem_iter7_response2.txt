```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'].shift(1) - df['close'].shift(2)) / df['close'].shift(2)

    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Average Volume Over a Period
    average_volume = df['volume'].rolling(window=5).mean()

    # Weight High-Low Range by Volume
    volume_weighted_high_low_range = high_low_range * df['volume']

    # Normalize by Average Volume
    normalized_volatility = volume_weighted_high_low_range / average_volume

    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * normalized_volatility

    # Smoothed Moving Average
    sma_adjusted_returns = adjusted_returns.rolling(window=5).mean()
    
    # Apply Exponential Smoothing
    smoothed_returns = sma_adjusted_returns.ewm(span=5, adjust=False).mean()

    return smoothed_returns
```
