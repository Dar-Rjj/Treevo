```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Percentage Change
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Volume Momentum
    df['volume_momentum'] = df['volume'].pct_change()
    
    # Calculate Close-to-Next Open Return
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Calculate Inverse of Close-to-Next Open Return
    df['inverse_close_to_next_open'] = -df['close_to_next_open_return']
    
    # Combine the components into a final alpha factor
    df['alpha_factor'] = (df['intraday_pct_change'] * df['daily_range'] * df['volume_momentum'] 
                          + df['inverse_close_to_next_open'])
    
    return df['alpha_factor']
```
