```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    df['integrated_factor'] = (df['intraday_return'] * df['overnight_return'] * df['momentum_ratio'])
    df['integrated_factor'] *= np.where(df['volume'] > df['volume'].mean(), 1.5, 0.5)
    
    # Weight High-Low Range by Volume
    df['high_low_range_weighted'] = (df['high'] - df['low']) * (df['volume'] / df['volume'].mean())
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['high_low_range_weighted']
    
    # Smoothed Moving Average
    df['sma_5'] = df['adjusted_returns'].rolling(window=5).mean()
    df['ema_smoothed'] = df['sma_5'].ewm(alpha=0.1).mean()
    
    # Explore Momentum Factors
    df['sma_5_close'] = df['close'].rolling(window=5).mean()
    df['sma_20_close'] = df['close'].rolling(window=20).mean()
    df['momentum'] = df['sma_5_close'] - df['sma_20_close']
    
    # Analyze Volatility
    df['price_range'] = df['high'] - df['low']
    df['atr_14'] = df['price_range'].rolling(window=14).mean()
    df['volatility'] = df['atr_14']
    
    # Investigate Volume-Based Indicators
    df['vwap'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Consider Transaction Amounts
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['transaction_intensity'] = df['amount_change'] / df['volume_change']
    
    # Combine Multiple Factors
    df['composite'] = 0.3 * df['momentum'] + 0.4 * df['volatility'] + 0.3 * df['vwap_deviation']
    
    # Generate Alpha Signals
    alpha_signal = np.where(df['composite'] > 0, 1, -1)
    
    # Final Factor
    final_factor = df['integrated_factor'] * df['ema_smoothed'] * alpha_signal
    
    return final_factor
```
