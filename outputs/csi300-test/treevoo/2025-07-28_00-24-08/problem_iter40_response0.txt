```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregated_returns = intraday_return + overnight_return
    
    # Weight High-Low Range by Volume
    high_low_weighted = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregated_returns * high_low_weighted
    
    # Smooth Adjusted Returns
    smoothed_returns = adjusted_returns.rolling(window=5).mean().ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factor = aggregated_returns * momentum_ratio
    # Conditional Volume Adjustment
    volume_above_avg = df['volume'] > df['volume'].mean()
    integrated_factor[volume_above_avg] *= 1.7
    integrated_factor[~volume_above_avg] *= 0.3
    # Incorporate Volume Change Indicator
    integrated_factor[volume_change > 0] *= 1.5
    integrated_factor[volume_change < 0] *= 0.8
    
    # Final Factor
    final_factor = integrated_factor * smoothed_returns
    
    return final_factor
```
