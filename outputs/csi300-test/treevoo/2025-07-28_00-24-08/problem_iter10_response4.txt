```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Price Momentum: Sum Last 20 days log returns
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_filter'] = (df['volume_change'] > 0).astype(int)
    
    # Integrate Amount for Momentum Strengthening
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['amount_filter'] = (df['amount_change'] > 0).astype(int)
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['price_momentum'] * df['volume_filter'] * df['amount_filter']
    
    return df['alpha_factor']
```
