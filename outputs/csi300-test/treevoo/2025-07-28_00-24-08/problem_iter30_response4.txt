```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = np.log(df['close'] / df['open'])
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = np.log(df['open'] / df['close'].shift(1))
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return']
    
    # Sum over a window (last 20 days log returns)
    df['momentum'] = df['aggregate_return'].rolling(window=20).sum()
    
    # Intraday Trend Strength
    df['daily_range'] = df['high'] - df['low']
    df['intraday_pct_change'] = df['close'] / df['open'] - 1
    
    # Volume-Weighted High-Low Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['vol_weighted_volatility'] = df['high_low_range'] * df['volume']
    
    # Volume Adjusted Reversal Signal
    df['volume_pct_change'] = df['volume'] / df['volume'].shift(1) - 1
    df['close_to_next_open_return'] = np.log(df['close']) - np.log(df['open'].shift(-1))
    df['inverse_close_to_next_open_return'] = -df['close_to_next_open_return']
    
    # Combine factors
    df['composite_factor'] = (
        df['momentum'] +
        df['intraday_pct_change'] +
        df['vol_weighted_volatility'] +
        df['inverse_close_to_next_open_return']
    )
    
    # Smoothed Moving Average of Composite Factor
    df['sma_10'] = df['composite_factor'].rolling(window=10).mean()
    df['ema_smooth'] = df['sma_10'].ewm(span=10, adjust=False).mean()
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume'] = (df['volume_change'] > 0).astype(int)
    df['volume_weighted_returns'] = df['close'] - df['close'].shift(1)
    df['volume_ema'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_weighted_returns'] *= df['volume'] / df['volume_ema']
    
    # Add Price-Volume Correlation
    df['price_volume_corr'] = df['close'].rolling(window=30).corr(df['volume']).abs()
    
    # Final Alpha Factor
    df['final_alpha'] = (
        df['momentum'] +
        df['intraday_pct_change'] +
        df['intraday_high_low_ratio'] +  # Placeholder for a ratio or other relevant metric
        df['open_to_close_direction'] +  # Placeholder for direction or other relevant metric
        df['volume_weighted_returns'] +
        df['price_volume_corr']
    )
    
    return df['final_alpha']
```
