```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()
    
    # Identify Trends in Daily Returns
    df['trend_5d'] = df['daily_return'].rolling(window=5).sum()
    
    # Analyze Price Volatility
    df['price_volatility'] = df[['high', 'low']].rolling(window=20).std().mean(axis=1)
    
    # Explore Trading Volume Patterns
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / avg_volume_10d
    
    # Consider Price Momentum
    df['momentum_30d'] = df['close'] - df['low'].rolling(window=30).min()
    
    # Investigate the Relationship Between Price Change and Volume
    df['return_volume_corr_60d'] = df.rolling(window=60).apply(lambda x: x['daily_return'].corr(x['volume']))
    
    # Examine Market Strength
    df['market_strength'] = (df['close'] > df['open']).rolling(window=10).sum()
    
    # Look for Buying or Selling Pressure
    df['top_third'] = (df['close'] > (df['high'] + 2 * df['low']) / 3).rolling(window=7).mean() > 0.5
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Volume-Weighted High-Low Volatility
    df['hl_range'] = df['high'] - df['low']
    df['volume_weighted_hl'] = df['hl_range'] * df['volume']
    avg_volume_weighted_hl = df['volume_weighted_hl'].rolling(window=20).mean()
    df['vol_weighted_hl_volatility'] = df['volume_weighted_hl'] / avg_volume_weighted_hl
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        0.2 * df['trend_5d'] +
        0.1 * df['price_volatility'] +
        0.1 * df['volume_ratio'] +
        0.1 * df['momentum_30d'] +
        0.1 * df['return_volume_corr_60d'] +
        0.1 * df['market_strength'] +
        0.05 * df['top_third'].astype(int) +
        0.05 * df['momentum_ratio'] +
        0.1 * df['vol_weighted_hl_volatility']
    )
    
    return alpha_factor.dropna()
```
