```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = np.log(df['close'] / df['open'])
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = np.log(df['open'].shift(-1) / df['close'])
    
    # Aggregate Returns
    df['total_return'] = df['intraday_return'] + df['overnight_return']
    
    # Sum over a window (last 20 days log returns)
    df['momentum'] = df['total_return'].rolling(window=20).sum()
    
    # Intraday Trend Strength
    df['daily_range'] = df['high'] - df['low']
    df['intraday_trend'] = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted High-Low Volatility
    df['high_low_range'] = df['high'] - df['low']
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['vol_weighted_volatility'] = (df['high_low_range'] * df['volume']) / df['avg_volume']
    
    # Volume Adjusted Reversal Signal
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    df['volume_change'] = df['volume'] / df['volume'].shift(1) - 1
    df['close_to_next_open'] = (df['close'] - df['open'].shift(-1)) / df['open'].shift(-1)
    df['reversal_signal'] = -df['close_to_next_open']
    
    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = df['momentum'] + df['intraday_trend'] + df['vol_weighted_volatility'] + df['reversal_signal']
    df['sma_composite'] = df['composite_factor'].rolling(window=5).mean()
    df['ema_composite'] = df['sma_composite'].ewm(span=5, adjust=False).mean()
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume'] = (df['volume_change'] > 0).astype(int)
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['ema_volume'])
    
    # Add Price-Volume Correlation
    df['price_volume_corr'] = df['close'].rolling(window=20).corr(df['volume']).abs()
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        df['momentum'] + 
        df['intraday_trend'] + 
        df['open_to_close_direction'] + 
        df['volume_weighted_returns'] + 
        df['price_volume_corr']
    )
    
    return df['alpha_factor']
```
