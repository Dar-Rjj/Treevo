```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Weight High-Low Range by Volume
    df['vol_weighted_high_low_range'] = df['high_low_range'] * df['volume']
    
    # Adjust by Average Volume
    df['adjusted_high_low_range'] = df['vol_weighted_high_low_range'] / df['avg_volume']
    
    # Combine Volume Adjusted Volatility and Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    df['combined_volatility'] = df['adjusted_high_low_range'] + df['volume_change']
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] / df['combined_volatility']
    
    # Smoothed Moving Average
    df['sma_adjusted_returns'] = df['adjusted_returns'].rolling(window=10).mean()
    df['ema_adjusted_returns'] = df['sma_adjusted_returns'].ewm(span=10, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['price_momentum_ratio'] * df['ema_adjusted_returns']
    
    return df['final_alpha_factor']
```
