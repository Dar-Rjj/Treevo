```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['integrated_factor'] = (df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio'])
    
    # Conditionally Adjust by Volume Metrics
    df['volume_adjusted_factor'] = df['integrated_factor'] * (1.5 if df['volume'] > df['avg_volume_5d'] else 0.5)
    
    # Combine Momentum and Volatility
    df['combined_factor'] = df['price_momentum_ratio'] + (df['high_low_range'] * df['volume'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_factor'] * df['combined_factor']
    
    return df['alpha_factor']
```
