```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight High-Low Range by Volume
    high_low_range = df['high'] - df['low']
    volume_weighted_high_low = high_low_range * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low
    
    # Smoothed Moving Average
    sma_adjusted_returns = adjusted_returns.rolling(window=5).mean()
    smoothed_returns = sma_adjusted_returns.ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    integrated_factors = intraday_return * overnight_return * price_momentum_ratio
    
    # Conditionally Adjust by Volume Metrics
    volume_above_average = df['volume'] > df['volume'].rolling(window=20).mean()
    integrated_factors[volume_above_average] *= 1.5
    integrated_factors[~volume_above_average] *= 0.5
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    smoothed_returns_adjusted = smoothed_returns * daily_dollar_volume
    
    # Volume Surge Indicator
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    volume_surge = df['volume'] > five_day_avg_volume
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = (df['high'] - df['low']) * intraday_return
    
    # Generate Final Momentum Factor
    log_returns = np.log(df['close'] / df['close'].shift(1))
    final_momentum_factor = log_returns.rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Combine Price Momentum and Intraday Reversal
    combined_price_momentum_intraday_reversal = price_momentum_ratio * (intraday_return * overnight_return)
    
    # Conditionally Adjust by Volume Surge
    seven_day_avg_volume = df['volume'].rolling(window=7).mean()
    volume_surge_7_days = df['volume'] > seven_day_avg_volume
    combined_price_momentum_intraday_reversal[volume_surge_7_days] *= 1.8
    combined_price_momentum_intraday_reversal[~volume_surge_7_days] *= 0.7
    
    # Integrate All Components
    final_alpha_factor = (smoothed_returns_adjusted + final_momentum_factor + high_low_volatility + combined_price_momentum_intraday_reversal) * volume_confirmation
    
    return final_alpha_factor
```
