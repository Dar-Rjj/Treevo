import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Sum over a window
    log_returns = np.log(aggregate_returns + 1)
    momentum = log_returns.rolling(window=20).sum()
    
    # Intraday Trend Strength
    daily_range = df['high'] - df['low']
    intraday_pct_change = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted High-Low Volatility
    high_low_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_high_low_volatility = high_low_range * df['volume'] / avg_volume
    
    # Volume Adjusted Reversal Signal
    prev_day_volume_momentum = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    close_to_next_open_return = (df['open'].shift(-1) - df['close']) / df['close']
    inverse_close_to_next_open_return = -close_to_next_open_return
    
    # Smoothed Moving Average of Composite Factor
    combined_factor = (momentum + intraday_pct_change + volume_weighted_high_low_volatility + inverse_close_to_next_open_return)
    sma_combined_factor = combined_factor.rolling(window=5).mean()
    ema_smoothed_factor = sma_combined_factor.ewm(span=5, adjust=False).mean()
    
    # Confirm Momentum with Volume
    volume_change = df['volume'] - df['volume'].shift(1)
    positive_volume_filter = (volume_change > 0).astype(int)
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    volume_weighted_smoothed_returns = (df['close'] - df['close'].shift(1)) * (df['volume'] / ema_volume)
    
    # Add Price-Volume Correlation
    price_volume_corr = df['close'].rolling(window=20).corr(df['volume'])
    normalized_corr = price_volume_corr.abs()
    
    # Final Alpha Factor
    price_intraday_momentum = ema_smoothed_factor
    final_alpha_factor = (
        price_intraday_momentum * volume_weighted_smoothed_returns
        + normalized_corr
    )
    
    return final_alpha_factor
