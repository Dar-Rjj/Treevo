```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['open'].shift(-1) - df['close']) / df['close'].shift(-1)
    
    # Volume Surge Indicator
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_surge = df['volume'].apply(lambda x: 1.7 if x > avg_volume_5d else 0.3)
    
    # Integrate Factors
    integrated_factors = intraday_return * prev_close_to_open_return * volume_surge
    
    # Intraday High-Low Range Ratio
    high_low_range_ratio = df['high'] / df['low']
    
    # Open-to-Close Direction
    open_close_direction = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    sma_volume = df['volume'].rolling(window=5).mean()
    weighted_returns = (df['close'].diff() * (df['volume'] / sma_volume)).fillna(0)
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    volume_change = df['volume'].diff()
    pvt_indicator = price_change * volume_change
    
    # Hybrid Momentum-Reversal Indicator
    hybrid_momentum_reversal = (intraday_return + weighted_returns)
    final_sign = hybrid_momentum_reversal.apply(lambda x: 1 if x > 0 else -1)
    
    # Final Alpha Factor
    final_alpha_factor = (
        integrated_factors * 
        high_low_range_ratio * 
        open_close_direction * 
        pvt_indicator * 
        final_sign
    )
    
    return final_alpha_factor
```
