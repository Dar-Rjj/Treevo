import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['Prev_Day_Close'] = df['close'].shift(1)
    df['Overnight_Return'] = (df['close'] - df['Prev_Day_Close']) / df['Prev_Day_Close']
    
    # Aggregate Returns
    df['Aggregate_Returns'] = df['Intraday_Return'] + df['Overnight_Return']
    
    # Adjust for Volume
    df['Volume_Adjusted_Returns'] = df['Aggregate_Returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['Weighted_Adjusted_Returns'] = df['Intraday_Volatility'] * df['Volume_Adjusted_Returns']
    
    # Smoothed Moving Average
    df['SMA_5D'] = df['Weighted_Adjusted_Returns'].rolling(window=5).mean()
    df['EMA_Smoothed'] = df['SMA_5D'].ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    df['Short_Term_Momentum'] = df['close'] - df['close'].shift(7)
    
    # Long-Term Price Momentum
    df['Long_Term_Momentum'] = df['close'] - df['close'].shift(25)
    
    # Price Momentum Ratio
    df['Price_Momentum_Ratio'] = df['Short_Term_Momentum'] / (df['Long_Term_Momentum'] + 1e-9)  # avoid division by zero
    
    # Integrate Intraday and Previous Day's Returns
    df['Integrated_Returns'] = df['Intraday_Return'] * df['Overnight_Return']
    
    # Combine Price Momentum and Intraday Reversal
    df['Combined_Price_Momentum'] = df['Price_Momentum_Ratio'] * df['Integrated_Returns']
    
    # Conditionally Adjust by Volume Surge
    df['Avg_Volume_7D'] = df['volume'].rolling(window=7).mean()
    df['Volume_Surge'] = np.where(df['volume'] > df['Avg_Volume_7D'], 1.8, 0.7)
    df['Conditionally_Adjusted'] = df['Combined_Price_Momentum'] * df['Volume_Surge']
    
    # Generate Final Alpha Factor
    df['Final_Alpha'] = df['EMA_Smoothed'] * df['Conditionally_Adjusted']
    
    # Volume Change Indicator
    df['Volume_Change'] = df['volume'] - df['volume'].shift(5)
    
    # Volume-Weighted High-Low Range
    df['Volume_Weighted_HL'] = (df['High_Low_Range'] * df['volume']).rolling(window=5).sum()
    df['Avg_Volume_Period'] = df['volume'].rolling(window=5).mean()
    df['Adjusted_Volatility'] = df['Aggregate_Returns'] * (df['Volume_Weighted_HL'] / (df['Avg_Volume_Period'] + 1e-9))  # avoid division by zero
    
    # Integrate Amount Data
    df['Daily_Dollar_Volume'] = df['volume'] * df['close']
    df['Amount_Adjusted_EMA'] = df['EMA_Smoothed'] * (df['Daily_Dollar_Volume'] / df['Daily_Dollar_Volume'].rolling(window=5).mean())
    
    # Final Factor
    df['Final_Factor'] = df['Final_Alpha'] * df['Amount_Adjusted_EMA'] * df['Adjusted_Volatility']
    
    return df['Final_Factor'].fillna(0)
