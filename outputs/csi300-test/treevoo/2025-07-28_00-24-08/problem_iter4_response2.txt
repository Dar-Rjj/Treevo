```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)

    # Volume Surge Indicator
    # Calculate 5-Day Average Volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    # Determine if Today's Volume is Above 5-Day Average
    volume_surge = df['volume'] > avg_volume_5d

    # Integrate Factors
    integrated_factor = intraday_return * prev_close_to_open_return
    # Conditionally Adjust by Volume Surge Indicator
    integrated_factor = integrated_factor.where(~volume_surge, integrated_factor * 1.5)
    integrated_factor = integrated_factor.where(volume_surge, integrated_factor * 0.5)

    return integrated_factor
```
