```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = np.log(df['close'] / df['open'])
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = np.log(df['open'] / df['close'].shift(1))
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Sum over a window (Last 20 days log returns)
    df['momentum'] = df['aggregate_returns'].rolling(window=20).sum()
    
    # Intraday Trend Strength
    df['daily_range'] = df['high'] - df['low']
    df['intraday_percent_change'] = (df['close'] - df['open']) / df['open']
    
    # Volume-Weighted High-Low Volatility
    df['high_low_volatility'] = df['daily_range'] * df['volume']
    
    # Volume Adjusted Reversal Signal
    df['percent_change_volume'] = df['volume'].pct_change()
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['inverse_close_to_next_open_return'] = -df['close_to_next_open_return']
    
    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = df['momentum'] + df['intraday_percent_change'] + df['high_low_volatility'] + df['inverse_close_to_next_open_return']
    df['sma_composite_factor'] = df['composite_factor'].rolling(window=5).mean()
    df['ema_sma_composite_factor'] = df['sma_composite_factor'].ewm(span=5, adjust=False).mean()
    
    # Confirm Momentum with Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume'] = (df['volume_change'] > 0).astype(int)
    df['ema_volume'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['ema_volume'])
    
    # Add Price-Volume Correlation
    df['price_volume_corr'] = df['close'].rolling(window=20).corr(df['volume']).abs()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['momentum'] +
        df['intraday_percent_change'] +
        (df['high'] - df['low']) / df['open'] +
        (df['close'] - df['open']) / df['open']
    ) * df['volume_weighted_returns'] + df['price_volume_corr']
    
    return df['final_alpha_factor']
```
