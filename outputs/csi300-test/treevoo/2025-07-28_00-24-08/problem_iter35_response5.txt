import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume weighted price
    df['vwp'] = (df['amount'] / df['volume']) * 10  # Assuming amount is in currency and volume is in shares, adjust the multiplier as needed
    
    # Calculate the momentum of the volume weighted price
    df['vwp_mom'] = df['vwp'].pct_change(periods=5)  # 5-day momentum
    
    # Calculate the average true range over 14 days
    df['tr'] = df[['high'-'low', 'high'-'close.shift(1)', 'low'+'close.shift(1)']].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate a simple moving average of the close price
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Alpha factor: normalized momentum of volume weighted price divided by average true range
    df['alpha_factor'] = (df['vwp_mom'] - df['vwp_mom'].mean()) / df['vwp_mom'].std() / df['atr_14']
    
    return df['alpha_factor']
