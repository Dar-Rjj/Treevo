```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Price Momentum
    df['price_momentum'] = df['log_return'].rolling(window=20).sum()
    
    # Intraday and Overnight Momentum with Volume-Weighted Volatility
    df['intraday_return'] = (df['close'] / df['open']) - 1
    df['overnight_return'] = (df['close'] / df['close'].shift(1)) - 1
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    df['daily_range'] = df['high'] - df['low']
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']
    df['high_low_volatility'] = (df['high'] - df['low']) * df['volume']
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['vol_weighted_hl_volatility'] = df['high_low_volatility'] / df['avg_volume']
    
    # Volume Adjusted Reversal Signal
    df['volume_change'] = df['volume'].pct_change()
    df['close_to_next_open'] = (df['close'] / df['open'].shift(-1)) - 1
    df['reversal_score'] = -df['close_to_next_open']
    
    # Combine Factors
    df['composite_factor'] = (
        df['aggregate_returns'] + 
        df['intraday_pct_change'] + 
        df['vol_weighted_hl_volatility'] + 
        df['reversal_score']
    )
    
    # Smoothed Moving Average of Composite Factor
    df['sma_composite'] = df['composite_factor'].rolling(window=5).mean()
    df['smoothed_composite'] = df['sma_composite'].ewm(span=5, adjust=False).mean()
    
    # Confirm Momentum with Volume
    df['volume_filter'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['final_alpha'] = df['price_momentum'] * df['volume_filter'] * df['smoothed_composite']
    
    return df['final_alpha'].dropna()
```
