import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']

    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return.shift(1)

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Weight High-Low Range by Volume
    volume_weighted_high_low_range = high_low_range * df['volume']

    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low_range

    # Smoothed Moving Average
    sma_adjusted_returns = adjusted_returns.rolling(window=5).mean()
    ema_adjusted_returns = sma_adjusted_returns.ewm(span=5, adjust=False).mean()

    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)

    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum

    # Integrate Factors
    integrated_factors = (intraday_return * overnight_return.shift(1)) * price_momentum_ratio

    # Conditionally Adjust by Volume Metrics
    average_volume = df['volume'].rolling(window=5).mean()
    volume_condition = np.where(df['volume'] > average_volume, 1.5, 0.5)
    integrated_factors *= volume_condition

    # Calculate Daily Dollar Volume
    daily_dollar_volume = df['volume'] * df['close']

    # Adjust Smoothed Moving Average by Daily Dollar Volume
    adjusted_ema = ema_adjusted_returns * (daily_dollar_volume / daily_dollar_volume.rolling(window=5).mean())

    # Volume Surge Indicator
    volume_surge = np.where(df['volume'] > average_volume, 1, 0)

    # Final Factor
    final_factor = integrated_factors * adjusted_ema * volume_surge

    return final_factor
