```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Calculate Volume Change Indicator
    volume_change_indicator = df['volume'] - df['volume'].shift(5)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Combine Volume Adjusted Volatility and Volume Change
    combined_volatility_and_volume_change = volume_weighted_high_low_range + volume_change_indicator
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_day_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Integrate Intraday and Previous Day's Returns
    integrated_intraday_return = intraday_return * prev_day_close_to_open_return
    
    # Combine Price Momentum and Intraday Reversal
    combined_price_momentum_and_reversal = price_momentum_ratio * integrated_intraday_return
    
    # Volume Surge Indicator
    avg_5_day_volume = df['volume'].rolling(window=5).mean()
    volume_surge_indicator = (df['volume'] > avg_5_day_volume).astype(float)
    conditionally_adjusted_volume_surge = volume_surge_indicator.apply(lambda x: 1.5 if x == 1 else 0.5)
    
    # Final Alpha Factor
    final_alpha_factor = (combined_price_momentum_and_reversal * conditionally_adjusted_volume_surge +
                          combined_volatility_and_volume_change)
    
    return final_alpha_factor
```
