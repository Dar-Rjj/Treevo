```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Overnight Return
    df['overnight_return'] = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    df['vol_weighted_high_low'] = df['high_low_range'] * df['volume']
    
    # Calculate Price Momentum
    n = 5  # number of periods for momentum
    df['price_momentum'] = (df['close'] - df['open'].shift(n)) / df['open'].shift(n)
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['vol_weighted_high_low']
    
    # Smoothed Moving Average
    sma = df['adjusted_returns'].rolling(window=5).mean()
    alpha_factor = sma.ewm(span=5, adjust=False).mean()
    
    return alpha_factor
```
