```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Adjust for Volume
    df['volume_adjusted_returns'] = df['aggregate_returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['weighted_returns'] = df['intraday_volatility'] * df['volume_adjusted_returns']
    
    # Smoothed Moving Average
    df['sma_weighted_returns'] = df['weighted_returns'].rolling(window=5).mean()
    df['ema_weighted_returns'] = df['sma_weighted_returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)
    df['long_term_momentum'] = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Intraday and Previous Day's Returns
    df['integrated_returns'] = df['intraday_return'] * df['overnight_return'].shift(1)
    
    # Combine Price Momentum and Intraday Reversal
    df['combined_momentum'] = df['price_momentum_ratio'] * df['integrated_returns']
    
    # Conditionally Adjust by Volume Surge
    df['avg_volume_7d'] = df['volume'].rolling(window=7).mean()
    df['volume_surge'] = np.where(df['volume'] > df['avg_volume_7d'], 1.8, 0.7)
    df['adjusted_combined_momentum'] = df['combined_momentum'] * df['volume_surge']
    
    # Average Volume Over a Period
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Volume Change Indicator
    df['volume_change_indicator'] = df['volume'] - df['volume'].shift(10)
    
    # Confirm Momentum with Volume and Amount
    df['positive_volume_filter'] = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['positive_amount_filter'] = np.where(df['amount_change'] > 0, 1, 0)
    df['momentum_confirmation'] = df['positive_volume_filter'] * df['positive_amount_filter']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['ema_weighted_returns'] + df['adjusted_combined_momentum'] * df['momentum_confirmation']
    
    return df['final_alpha_factor']
```
