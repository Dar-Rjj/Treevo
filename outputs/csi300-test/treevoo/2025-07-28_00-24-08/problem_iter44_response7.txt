    df['Combined_Return'] = df['Intraday_Return'] + df['Overnight_Return']

    df['Volume_Adjustment'] = np.where(df['volume'] > df['volume'].rolling(window=5).mean(), 1.5, 0.5)

    df['Weighted_Historical_Volatility'] = df['20_Day_Historical_Volatility'] * df['volume']
