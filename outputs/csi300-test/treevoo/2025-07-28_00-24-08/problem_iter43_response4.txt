```python
import pandas as pd
import numpy as np

def heuristics_v2(df, lookback_period=50):
    """
    Generate a novel and interpretable alpha factor using price and volume data.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    - lookback_period: int, the number of days to look back for moving averages
    
    Returns:
    - factor: pandas Series with the generated alpha factor values indexed by date
    """
    
    # Calculate Simple Moving Average (SMA)
    df['SMA'] = df['close'].rolling(window=lookback_period).mean()
    df['SMA_diff'] = df['close'] - df['SMA']
    
    # Calculate Weighted Moving Average (WMA)
    weights = np.arange(1, lookback_period + 1)
    df['WMA'] = df['close'].rolling(window=lookback_period).apply(lambda prices: np.dot(prices, weights) / weights.sum(), raw=True)
    df['WMA_diff'] = df['close'] - df['WMA']
    
    # Calculate Average True Range (ATR)
    df['TR'] = np.maximum(
        np.maximum(df['high'] - df['low'], np.abs(df['high'] - df['close'].shift())),
        np.abs(df['low'] - df['close'].shift())
    )
    df['ATR'] = df['TR'].rolling(window=lookback_period).mean()
    
    # Calculate Standard Deviation of daily returns
    df['daily_returns'] = df['close'].pct_change()
    df['std_dev'] = df['daily_returns'].rolling(window=lookback_period).std()
    
    # Calculate Chaikin Money Flow (CMF)
    df['MF_multiplier'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['MF_volume'] = df['MF_multiplier'] * df['volume']
    df['CMF'] = df['MF_volume'].rolling(window=lookback_period).sum() / df['volume'].rolling(window=lookback_period).sum()
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    
    # Calculate Relative Strength Index (RSI)
    df['gain'] = df['daily_returns'].apply(lambda x: x if x > 0 else 0)
    df['loss'] = df['daily_returns'].apply(lambda x: -x if x < 0 else 0)
    df['avg_gain'] = df['gain'].rolling(window=lookback_period).mean()
    df['avg_loss'] = df['loss'].rolling(window=lookback_period).mean()
    df['RS'] = df['avg_gain'] / df['avg_loss']
    df['RSI'] = 100 - (100 / (1 + df['RS']))
    
    # Combine all factors into a single alpha factor
    factor = (
        df['SMA_diff'] + 
        df['WMA_diff'] + 
        (df['ATR'] / df['std_dev']) +
        df['CMF'] + 
        df['OBV'] + 
        (df['RSI'] - 50)
    )
    
    return factor
```
