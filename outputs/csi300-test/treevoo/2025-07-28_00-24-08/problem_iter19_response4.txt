```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Overnight Return
    df['overnight_return'] = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    df['volume_weighted_high_low'] = df['high_low_range'] * df['volume']
    
    # Calculate Price Momentum
    df['price_momentum'] = (df['close'] - df['open'].shift(n)) / df['open'].shift(n)
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['volume_weighted_high_low']
    
    # Smoothed Moving Average
    df['ema_adjusted_returns'] = df['adjusted_returns'].ewm(span=5).mean()
    df['smoothed_returns'] = df['ema_adjusted_returns'].ewm(alpha=0.1, beta=0.2).mean()
    
    # Integrate Amount Data
    df['daily_dollar_volume'] = df['volume'] * df['close']
    
    # Adjust Smoothed Moving Average by Daily Dollar Volume
    df['factor'] = df['smoothed_returns'] / df['daily_dollar_volume']
    
    return df['factor']
```
