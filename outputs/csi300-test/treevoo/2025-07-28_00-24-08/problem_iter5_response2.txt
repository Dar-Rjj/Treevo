```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day Overnight Return
    df['prev_close'] = df['close'].shift(1)
    overnight_return = (df['open'] - df['prev_close']) / df['prev_close']
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Intraday Trend Strength
    daily_range = df['high'] - df['low']
    intraday_pct_change = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Volume-Weighted High-Low Volatility
    high_low_range = df['high'] - df['low']
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_weighted_volatility = (high_low_range * df['volume']) / avg_volume
    
    # Volume Adjusted Reversal Signal
    df['prev_volume'] = df['volume'].shift(1)
    volume_momentum = (df['volume'] - df['prev_volume']) / df['prev_volume']
    
    df['next_open'] = df['open'].shift(-1)
    close_to_next_open_return = (df['next_open'] - df['close']) / df['close']
    price_reversal_score = -close_to_next_open_return  # Inverse of close-to-next open return
    
    # Combine all factors
    composite_factor = (aggregate_returns + intraday_pct_change + 
                        volume_weighted_volatility + price_reversal_score)
    
    # Smoothed Moving Average of Composite Factor
    sma_5_days = composite_factor.rolling(window=5).mean()
    smoothed_factor = sma_5_days.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
