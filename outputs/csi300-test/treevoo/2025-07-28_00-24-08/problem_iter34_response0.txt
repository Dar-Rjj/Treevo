import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    previous_close_to_open_return = (df['close'].shift(1) - df['open']) / df['close'].shift(1)

    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)

    # High-Low Range
    high_low_range = df['high'] - df['low']

    # Average Volume Over a Period
    average_volume_5_days = df['volume'].rolling(window=5).mean()

    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    price_momentum_ratio.replace([np.inf, -np.inf], 0, inplace=True)

    # Integrate Factors
    integrated_factors = (
        intraday_return * 
        previous_close_to_open_return * 
        price_momentum_ratio
    )

    # Conditionally Adjust by Volume Metrics
    integrated_factors = integrated_factors.where(
        df['volume'] > average_volume_5_days,
        integrated_factors * 0.5
    ).where(
        df['volume'] <= average_volume_5_days,
        integrated_factors * 1.5
    )

    # Combine Momentum and Volatility
    weighted_high_low_range = high_low_range * df['volume']

    # Additional Volatility Component
    std_high_low_range = df['high_low_range'].rolling(window=5).std()
    normalized_volatility = std_high_low_range / average_volume_5_days

    # Enhance Momentum with Volume
    enhanced_momentum = (short_term_momentum + long_term_momentum)
    enhanced_momentum = enhanced_momentum.where(
        volume_change > 0,
        enhanced_momentum * 1.2
    ).where(
        volume_change < 0,
        enhanced_momentum * 0.8
    )

    # Final Alpha Factor
    final_alpha_factor = (
        integrated_factors +
        weighted_high_low_range +
        normalized_volatility +
        enhanced_momentum
    )

    return final_alpha_factor.dropna()
