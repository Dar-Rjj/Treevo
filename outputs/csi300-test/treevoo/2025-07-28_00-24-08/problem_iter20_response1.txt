```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Identify Trends in Daily Returns
    df['trend_5_days'] = df['daily_return'].rolling(window=5).sum()

    # Analyze Price Volatility
    df['volatility_20_days'] = df[['high', 'low']].rolling(window=20).std().mean(axis=1)

    # Explore Trading Volume Patterns
    df['avg_volume_10_days'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_10_days']

    # Consider Price Momentum
    df['price_momentum_30_days'] = df['close'] - df['low'].rolling(window=30).min()

    # Investigate the Relationship Between Price Change and Volume
    df['return_volume_corr_60_days'] = df['daily_return'].rolling(window=60).corr(df['volume'])

    # Examine the Strength of the Market
    df['positive_days_10'] = (df['close'] > df['open']).rolling(window=10).sum()

    # Look for Buying or Selling Pressure
    df['top_third_range_7_days'] = (df['close'] > (df['high'] + df['low']) * 2/3).rolling(window=7).mean()

    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']

    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open'] = (df['open'].shift(1) - df['close'].shift(1)) / df['close'].shift(1)

    # Aggregate Returns
    df['agg_returns'] = df['intraday_return'] * df['prev_close_to_open']

    # Volume Surge Indicator
    df['avg_volume_5_days'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = df['volume'] > df['avg_volume_5_days']

    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)

    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']

    # Volume-Weighted High-Low Volatility
    df['high_low_volatility'] = (df['high'] - df['low']) * df['volume'] / df['avg_volume_5_days']

    # Integrate Factors
    df['alpha_factor'] = df['agg_returns'] * df['price_momentum_ratio']
    df['alpha_factor'] = df['alpha_factor'] * (1.5 if df['volume_surge'] else 0.5)

    return df['alpha_factor']
```
