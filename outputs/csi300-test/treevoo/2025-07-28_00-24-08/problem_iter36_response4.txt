```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['open']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['volume_change'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['integrated_factor'] = (df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio'])
    df['integrated_factor'] = df['integrated_factor'] * np.where(df['volume'] > df['avg_volume_5d'], 1.5, 0.5)
    
    # Combine Momentum and Volatility
    df['weighted_high_low'] = df['high_low_range'] * df['volume']
    
    # Additional Volatility Component
    df['std_high_low_5d'] = df['high_low_range'].rolling(window=5).std()
    df['volatility_component'] = df['std_high_low_5d'] / df['avg_volume_5d']
    
    # Enhance Momentum with Volume
    df['enhanced_momentum'] = (df['short_term_momentum'] + df['long_term_momentum']) * np.where(df['volume_change'] > 0, 1.2, 0.8)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] + df['weighted_high_low'] + df['volatility_component'] + df['enhanced_momentum']
    
    return df['final_alpha_factor']
```
