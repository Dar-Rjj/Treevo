import pandas as pd
import numpy as np

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the provided thought tree.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Daily Price Changes
    df['price_change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Amount per Unit Volume
    df['amount_per_unit_volume'] = df['amount'] / df['volume']
    
    # Price-Volume Correlation
    df['price_volume_corr'] = df['price_change'].rolling(window=5).corr(df['volume_momentum'])
    
    # Volatility-Adjusted Returns
    df['volatility_adjusted_returns'] = (df['close'] - df['close'].shift(1)) / df['high_low_range']
    
    # Combined Momentum
    df['combined_momentum'] = df['price_change'] + df['volume_momentum']
    
    # Transaction Size Adjusted Volatility
    df['transaction_size_adjusted_volatility'] = df['high_low_range'] * df['amount_per_unit_volume']
    
    # Trend Following Factor
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['trend_following_factor'] = df['sma_5'] - df['close']
    
    # Breakout Signal
    df['breakout_signal'] = (df['close'] > df['high'].rolling(window=5).max()).astype(int)
    
    # Composite Alpha Factor
    df['composite_alpha_factor'] = (
        0.3 * df['combined_momentum'] +
        0.2 * df['volatility_adjusted_returns'] +
        0.1 * df['price_volume_corr'] +
        0.2 * df['trend_following_factor'] +
        0.2 * df['breakout_signal']
    )
    
    return df['composite_alpha_factor']
