```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Cumulative Daily Return
    data['daily_return'] = data['close'].pct_change()

    # Calculate High-Low Price Range
    data['price_range'] = data['high'] - data['low']

    # Calculate Volume Difference
    data['volume_diff'] = data['volume'].diff()

    # Calculate Average Volume
    data['avg_volume'] = data['volume'].rolling(window=2).mean()

    # Integrate Volume and Price Movement
    data['integrated_movement'] = (data['price_range'] * data['volume_diff']) / data['avg_volume']

    # Combine with Cumulative Return
    data['alpha_factor'] = data['daily_return'] - data['integrated_movement']
    
    return data['alpha_factor'].dropna()
```
