import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_above_avg'] = df['volume'] > df['avg_volume_10d']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['volume_change_10d'] = df['volume'] - df['volume'].shift(10)
    
    # Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_to_open_return'] * df['momentum_ratio']
    df['integrated_factor'] = df['integrated_factor'] * df['volume_above_avg'].apply(lambda x: 1.8 if x else 0.7)
    
    # Combine Momentum and Volatility
    df['weighted_high_low_range'] = df['high_low_range'] * df['volume']
    
    # Confirm Momentum with Volume and Amount
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['positive_volume'] = df['volume_change'].apply(lambda x: 1 if x > 0 else 0)
    
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['positive_amount'] = df['amount_change'].apply(lambda x: 1 if x > 0 else 0)
    
    df['confirmed_momentum'] = df['positive_volume'] * df['positive_amount']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['integrated_factor'] * df['weighted_high_low_range'] * df['confirmed_momentum']
    
    return df['final_alpha_factor']
