import pandas as pd
import numpy as np

def heuristics_v2(df, n=20):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return Factors
    df['intraday_high_low_ratio'] = df['high'] / df['low']
    df['open_to_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    df['sma_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['sma_volume'])
    
    # Calculate Overnight Return Factor
    df['overnight_return'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday and Overnight Factors
    df['intraday_overnight_combined'] = df['volume_weighted_returns'] + df['overnight_return']
    df['intraday_overnight_adjusted'] = df['intraday_overnight_combined'] * (df['volume'] / df['sma_volume'])
    
    # Weight High-Low Range by Volume
    df['volume_weighted_high_low_range'] = df['high_low_range'] * (df['volume'] / df['sma_volume'])
    
    # Combine All Factors
    df['final_factor'] = (df['intraday_overnight_adjusted'] + df['volume_weighted_high_low_range']) * df['open_to_close_direction']
    
    return df['final_factor']
