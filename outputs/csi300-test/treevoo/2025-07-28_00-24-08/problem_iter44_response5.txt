import pandas as pd

def heuristics_v2(df):
    # Calculate the log return of close price
    df['log_return'] = (df['close'].apply(np.log) - df['close'].shift(1).apply(np.log)).fillna(0)
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume']).fillna(0)
    
    # Calculate the relative spread between high and low price to vwap
    df['rel_spread_high'] = (df['high'] - df['vwap']) / df['vwap']
    df['rel_spread_low'] = (df['low'] - df['vwap']) / df['vwap']
    
    # Standardize the log returns, rel_spread_high, and rel_spread_low
    df['std_log_return'] = (df['log_return'] - df['log_return'].mean()) / df['log_return'].std()
    df['std_rel_spread_high'] = (df['rel_spread_high'] - df['rel_spread_high'].mean()) / df['rel_spread_high'].std()
    df['std_rel_spread_low'] = (df['rel_spread_low'] - df['rel_spread_low'].mean()) / df['rel_spread_low'].std()
    
    # Combine the standardized factors
    factor = 0.5 * df['std_log_return'] + 0.3 * df['std_rel_spread_high'] - 0.2 * df['std_rel_spread_low']

    return factor
