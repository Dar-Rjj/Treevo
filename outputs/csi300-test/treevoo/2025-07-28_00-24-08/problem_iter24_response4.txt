```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Adjust for Volume
    adjusted_returns = aggregate_returns * df['volume']
    
    # Weight by Intraday Volatility
    weighted_adjusted_returns = intraday_volatility * adjusted_returns
    
    # Smoothed Moving Average
    sma_10 = weighted_adjusted_returns.rolling(window=10).mean()
    smoothed_moving_average = sma_10.ewm(span=10, adjust=False).mean()
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(30)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Intraday and Previous Day's Returns
    integrated_returns = intraday_return * (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Combine Price Momentum and Intraday Reversal
    combined_price_momentum = price_momentum_ratio * integrated_returns
    
    # Conditionally Adjust by Volume Surge
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_surge_factor = np.where(df['volume'] > avg_volume_10, 2.0, 0.5)
    conditionally_adjusted_momentum = combined_price_momentum * volume_surge_factor
    
    # Generate Final Alpha Factor
    alpha_factor = smoothed_moving_average + conditionally_adjusted_momentum
    
    return alpha_factor
```
