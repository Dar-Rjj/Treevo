import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the range of the day (high - low)
    df['day_range'] = df['high'] - df['low']
    
    # Calculate the ratio of the close price to the open price
    df['close_open_ratio'] = df['close'] / df['open']
    
    # Calculate the ratio of the volume to the amount
    df['volume_amount_ratio'] = df['volume'] / df['amount']
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = df['close_pct_change'] * df['day_range'] * df['close_open_ratio'] * df['volume_amount_ratio']
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
