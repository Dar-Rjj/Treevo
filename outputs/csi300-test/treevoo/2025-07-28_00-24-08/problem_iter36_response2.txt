import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between today's close and the previous day's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the ratio of today's volume to the previous day's volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Use the product of the close difference and volume ratio as a factor
    df['close_vol_factor'] = df['close_diff'] * df['volume_ratio']
    
    # Determine the range (high - low) for each day
    df['range'] = df['high'] - df['low']
    
    # Calculate the difference between the high and the close
    df['high_close_diff'] = df['high'] - df['close']
    
    # Derive a factor by dividing the high-close difference by the range
    df['high_close_range_factor'] = df['high_close_diff'] / df['range']
    
    # Calculate the open-to-close return
    df['open_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Use this return as a direct factor
    df['open_close_return_factor'] = df['open_close_return']
    
    # Calculate the absolute difference between open and close, then divide by the range
    df['open_close_abs_diff'] = (df['close'] - df['open']).abs()
    df['normalized_open_close_diff'] = df['open_close_abs_diff'] / df['range']
    
    # Calculate the average trading amount over the past n days (e.g., 5 days)
    n = 5
    df['avg_amount'] = df['amount'].rolling(window=n).mean()
    
    # Compute the ratio of today's amount to the moving average
    df['amount_ratio'] = df['amount'] / df['avg_amount']
    
    # Use this ratio as an indicator of unusual trading activity
    df['amount_ratio_factor'] = df['amount_ratio']
    
    # Multiply the close-to-open return by the ratio to create a composite factor
    df['composite_factor'] = df['open_close_return'] * df['amount_ratio']
    
    # Create a final alpha factor by combining all the individual factors
    df['alpha_factor'] = (
        df['close_vol_factor'] + 
        df['high_close_range_factor'] + 
        df['open_close_return_factor'] + 
        df['normalized_open_close_diff'] + 
        df['amount_ratio_factor'] + 
        df['composite_factor']
    )
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
