```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return'].shift(1)
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    df['volume_adjustment'] = np.where(df['volume'] > df['volume'].mean(), 1.5, 0.5)
    df['integrated_factor'] = df['intraday_return'] * df['overnight_return'].shift(1) * df['momentum_ratio'] * df['volume_adjustment']
    
    # Weight High-Low Range by Volume
    df['volatility_weight'] = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    df['adjusted_volatility'] = df['aggregate_returns'] * df['volatility_weight']
    
    # Smoothed Moving Average
    df['sma_5'] = df['adjusted_volatility'].rolling(window=5).mean()
    df['ema_smoothed'] = df['sma_5'].ewm(span=5, adjust=False).mean()
    
    # Intraday Volume Momentum
    df['intraday_volume_change'] = df['volume'] - df['volume'].shift(5)
    df['normalized_intraday_volume_change'] = df['intraday_volume_change'] / df['volume'].rolling(window=5).mean()
    df['intraday_volume_momentum'] = df['intraday_return'] * df['normalized_intraday_volume_change']
    
    # Combine Multiple Factors
    df['composite'] = (df['intraday_return'] * df['overnight_return'].shift(1) * df['momentum_ratio'] * df['volume_adjustment']) + (df['adjusted_volatility'] * df['intraday_volume_momentum'])
    
    # Generate Alpha Signals
    threshold = df['composite'].quantile(0.5)
    df['alpha_signal'] = np.where(df['composite'] > threshold, 1, -1)
    
    # Final Factor
    df['final_factor'] = df['composite'] * df['alpha_signal']
    
    return df['final_factor']
```
