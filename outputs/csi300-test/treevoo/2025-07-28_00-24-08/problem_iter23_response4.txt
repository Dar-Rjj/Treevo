```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).cumsum() / (1 / df['volume']).cumsum()
    
    # Calculate the simple moving average (SMA) of the VWAP for 5 and 10 days
    df['vwap_5_sma'] = df['vwap'].rolling(window=5).mean()
    df['vwap_10_sma'] = df['vwap'].rolling(window=10).mean()
    
    # Calculate the relative strength index (RSI) for the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Combine factors: use the difference between VWAP SMAs and RSI
    df['alpha_factor'] = (df['vwap_5_sma'] - df['vwap_10_sma']) * df['rsi']
    
    return df['alpha_factor']
```
