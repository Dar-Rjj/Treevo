```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Intraday Volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Adjust for Volume
    df['volume_adjusted_returns'] = df['aggregate_returns'] * df['volume']
    
    # Weight by Intraday Volatility
    df['weighted_returns'] = df['intraday_volatility'] * df['volume_adjusted_returns']
    
    # Smoothed Moving Average
    df['sma_weighted_returns'] = df['weighted_returns'].rolling(window=5).mean()
    df['ema_smoothed_returns'] = df['sma_weighted_returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(7)
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(25)
    
    # Compute Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Integrate Intraday and Previous Day's Returns
    df['integrated_intraday_overnight_returns'] = df['intraday_return'] * df['overnight_return']
    
    # Combine Price Momentum and Intraday Reversal
    df['combined_momentum_reversal'] = df['price_momentum_ratio'] * df['integrated_intraday_overnight_returns']
    
    # Conditionally Adjust by Volume Surge
    df['avg_volume_7d'] = df['volume'].rolling(window=7).mean()
    df['volume_surge'] = np.where(df['volume'] > df['avg_volume_7d'], 1.8, 0.7)
    df['conditionally_adjusted_momentum'] = df['combined_momentum_reversal'] * df['volume_surge']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['ema_smoothed_returns'] * df['conditionally_adjusted_momentum']
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Calculate Volume-Weighted High-Low Range
    df['volume_weighted_high_low_range'] = (df['high_low_range'] * df['volume']).rolling(window=5).sum()
    df['average_volume'] = df['volume'].rolling(window=5).mean()
    df['adjusted_aggregate_returns'] = df['aggregate_returns'] * (df['volume_weighted_high_low_range'] / df['average_volume'])
    
    # Integrate Amount Data
    df['daily_dollar_volume'] = df['volume'] * df['close']
    df['amount_adjusted_ema'] = df['ema_smoothed_returns'] * df['daily_dollar_volume']
    
    # Final Factor
    df['final_factor'] = (df['final_alpha_factor'] + df['adjusted_aggregate_returns'] + df['amount_adjusted_ema']) / 3
    
    return df['final_factor'].dropna()
```
