```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Weight by Intraday Volatility
    intraday_volatility = np.abs(intraday_return.rolling(window=5).std())
    weighted_aggregate_returns = aggregate_returns * intraday_volatility
    
    # Adjust for Volume
    volume_adjusted_returns = weighted_aggregate_returns * df['volume']
    
    # Smoothed Moving Average
    sma = volume_adjusted_returns.rolling(window=5).mean()
    exp_smoothed = sma.ewm(span=5, adjust=False).mean()
    
    # Compute Price Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    price_momentum = log_returns.rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    final_momentum_factor = price_momentum * volume_confirmation
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = high_low_range * intraday_volatility
    combined_factor = exp_smoothed + final_momentum_factor + high_low_volatility
    
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Additional Factors
    intraday_overnight_interaction = intraday_return * overnight_return.shift(1)
    volume_adj_high_low_volatility = high_low_range * df['volume'] * intraday_volatility
    additional_factors = intraday_overnight_interaction * price_momentum_ratio
    
    # Conditionally Adjust by Volume Metrics
    volume_above_average = (df['volume'] > df['volume'].rolling(window=20).mean()).astype(int)
    adjusted_additional_factors = additional_factors * (1.5 * volume_above_average + 0.5 * (1 - volume_above_average))
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor + adjusted_additional_factors
    
    return final_alpha_factor
```
