```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Average Volume Over a Period
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_above_avg'] = df['volume'] > df['avg_volume_5d']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum'].replace(0, np.nan)
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_to_open_return'] * df['price_momentum_ratio']
    df['integrated_factor'] = df['integrated_factor'] * np.where(df['volume_above_avg'], 1.5, 0.5)
    
    # Combine Momentum and Volatility
    df['weighted_high_low_range'] = df['high_low_range'] * df['volume']
    
    # Additional Volatility Component
    df['std_high_low_range_5d'] = df['high_low_range'].rolling(window=5).std()
    df['volatility_component'] = df['std_high_low_range_5d'] / df['avg_volume_5d'].replace(0, np.nan)
    
    # Enhance Momentum with Volume
    df['enhanced_momentum'] = df['short_term_momentum'] + df['long_term_momentum']
    df['enhanced_momentum'] = df['enhanced_momentum'] * np.where(df['volume_change'] > 0, 1.2, 0.8)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['prev_close_to_open_return']
    
    # Weight by Intraday Volatility
    df['intraday_volatility'] = df['high_low_range'] / df['open']
    df['weighted_aggregate_returns'] = df['aggregate_returns'] * df['intraday_volatility']
    
    # Adjust for Volume
    df['volume_adjusted_returns'] = df['weighted_aggregate_returns'] * df['volume']
    
    # Smoothed Moving Average
    df['sma_volume_adjusted_returns_5d'] = df['volume_adjusted_returns'].rolling(window=5).mean()
    df['ema_sma_volume_adjusted_returns_5d'] = df['sma_volume_adjusted_returns_5d'].ewm(span=5, adjust=False).mean()
    
    # Compute Price Momentum
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_returns'].rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    df['final_momentum_factor'] = df['price_momentum'] * df['volume_confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['high_low_volatility'] = df['high_low_range'] * df['intraday_volatility']
    
    # Integrate All Components
    df['final_alpha_factor'] = (df['ema_sma_volume_adjusted_returns_5d'] 
                                 + df['final_momentum_factor'] 
                                 + df['high_low_volatility'])
    
    return df['final_alpha_factor'].dropna()
```
