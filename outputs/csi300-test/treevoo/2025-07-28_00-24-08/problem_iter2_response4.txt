import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Log Return
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Sum over a window of last 20 days log returns to get momentum
    df['momentum_20'] = df['log_return'].rolling(window=20).sum()
    
    # Calculate Intraday High-Low Range Ratio
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Open-to-Close Direction
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Positive Volume Change Filter
    df['positive_volume_change'] = (df['volume_change'] > 0).astype(int)
    
    # Calculate Simple Moving Average of Volumes over N days
    n_days = 20
    df['sma_volume'] = df['volume'].rolling(window=n_days).mean()
    
    # Compute Volume Weighted Returns
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['sma_volume'])
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['momentum_20'] + df['high_low_ratio'] + df['open_close_direction']) * df['volume_weighted_returns']
    
    return df['alpha_factor']
