import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Intraday Return
    data['intraday_return'] = (data['close'] - data['open']) / data['open']
    
    # Calculate Previous Day's Overnight Return
    data['overnight_return'] = (data['open'] - data['close'].shift(1)) / data['close'].shift(1)
    
    # Aggregate Returns
    data['aggregate_return'] = data['intraday_return'] + data['overnight_return']
    
    # Weight High-Low Range by Volume
    data['hl_range_vol_weighted'] = (data['high'] - data['low']) * data['volume']
    
    # Adjust for Combined Volatility
    data['adjusted_return'] = data['aggregate_return'] * data['hl_range_vol_weighted']
    
    # Smoothed Moving Average
    data['sma_adjusted_return'] = data['adjusted_return'].rolling(window=5).mean()
    data['ema_adjusted_return'] = data['sma_adjusted_return'].ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    data['short_term_momentum'] = data['close'] - data['close'].shift(5)
    
    # Long-Term Price Momentum
    data['long_term_momentum'] = data['close'] - data['close'].shift(20)
    
    # Price Momentum Ratio
    data['momentum_ratio'] = data['short_term_momentum'] / data['long_term_momentum']
    
    # Volume Change Indicator
    data['volume_change'] = data['volume'] - data['volume'].shift(5)
    
    # Integrate Factors
    data['factor'] = (data['intraday_return'] * data['overnight_return']) * data['momentum_ratio']
    
    # Conditionally Adjust by Volume Metrics
    average_volume = data['volume'].mean()
    data['factor'] = np.where(data['volume'] > average_volume, data['factor'] * 1.5, data['factor'] * 0.5)
    
    # Final Factor
    data['final_factor'] = data['factor'] * data['ema_adjusted_return']
    
    return data['final_factor']
