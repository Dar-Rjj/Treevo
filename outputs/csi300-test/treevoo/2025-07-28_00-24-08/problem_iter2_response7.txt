```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Intraday Trend Strength
    df['daily_range'] = (df['high'] - df['low']) / df['low']
    df['intraday_trend_strength'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Volume Adjusted Reversal Signal
    df['volume_momentum'] = df['volume'] - df['volume'].shift(1)
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['price_reversal_score'] = -df['close_to_next_open_return']
    
    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = df['aggregate_returns'] + df['intraday_trend_strength'] + df['price_reversal_score']
    df['sma_composite_factor'] = df['composite_factor'].rolling(window=5).mean()
    df['ema_composite_factor'] = df['sma_composite_factor'].ewm(span=5, adjust=False).mean()
    
    return df['ema_composite_factor']
```
