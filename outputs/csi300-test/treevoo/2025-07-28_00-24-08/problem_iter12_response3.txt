```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, alpha=0.2):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate Previous Day Overnight Return
    overnight_return = df['open'] / df['close'].shift(1) - 1
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Weight High-Low Range by Volume
    weighted_high_low_range = high_low_range * (df['volume'] / avg_volume)
    
    # Calculate Price Momentum
    price_momentum = df['close'] / df['open'].shift(n) - 1
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * weighted_high_low_range
    
    # Smoothed Moving Average
    sma = adjusted_returns.rolling(window=n).mean()
    ema = sma.ewm(alpha=alpha, adjust=False).mean()
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    final_factor = ema / daily_dollar_volume
    
    return final_factor
```
