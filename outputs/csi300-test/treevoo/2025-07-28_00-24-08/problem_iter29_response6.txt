```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High-Low_Range'] = df['high'] - df['low']
    
    # Calculate Intraday Return (Enhanced)
    df['Intraday_Return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['Prev_Close_to_Open_Return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Previous Day Overnight Return
    df['Prev_Overnight_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['Aggregate_Returns'] = df['Intraday_Return'] + df['Prev_Overnight_Return']
    
    # Calculate Intraday Volatility
    df['Intraday_Volatility'] = df['High-Low_Range'] / df['open']
    
    # Weight by Intraday Volatility
    df['Weighted_Aggregate_Returns'] = df['Aggregate_Returns'] * df['Intraday_Volatility']
    
    # Adjust for Volume
    df['Volume_Adjusted_Returns'] = df['Weighted_Aggregate_Returns'] * df['volume']
    
    # Smoothed Moving Average
    df['SMA_Volume_Adjusted_Returns'] = df['Volume_Adjusted_Returns'].rolling(window=5).mean()
    df['Smoothed_Returns'] = df['SMA_Volume_Adjusted_Returns'].ewm(span=5, adjust=False).mean()
    
    # Compute Price Momentum
    df['Log_Returns'] = np.log(df['close'] / df['close'].shift(1))
    df['Price_Momentum'] = df['Log_Returns'].rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['Volume_Confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    df['Final_Momentum_Factor'] = df['Price_Momentum'] * df['Volume_Confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['High_Low_Volatility'] = df['High-Low_Range'] * df['Intraday_Volatility']
    
    # Integrate All Components
    df['Combined_Factor'] = df['Smoothed_Returns'] + df['Final_Momentum_Factor'] + df['High_Low_Volatility']
    
    # Calculate Short-Term and Long-Term Price Momentum
    df['Short_Term_Price_Momentum'] = df['close'] - df['close'].shift(5)
    df['Long_Term_Price_Momentum'] = df['close'] - df['close'].shift(20)
    
    # Calculate Volume Change Indicator
    df['Volume_Change_Indicator'] = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    df['Price_Momentum_Ratio'] = df['Short_Term_Price_Momentum'] / df['Long_Term_Price_Momentum']
    
    # Integrate Additional Factors
    df['Intraday_Return_Prev_Close_Open'] = df['Intraday_Return'] * df['Prev_Close_to_Open_Return']
    df['Adjusted_Factor'] = df['Intraday_Return_Prev_Close_Open'] * df['Price_Momentum_Ratio']
    
    # Conditionally Adjust by Volume Metrics
    average_volume = df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Factor'] = df['Adjusted_Factor'] * np.where(df['volume'] > average_volume, 1.5, 0.5)
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factor'] + df['Volume_Adjusted_Factor']
    
    return df['Final_Alpha_Factor']
```
