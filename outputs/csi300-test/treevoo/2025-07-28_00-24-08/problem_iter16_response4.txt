```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    overnight_return.fillna(0, inplace=True)  # Fill first day's overnight return with 0
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * volume_weighted_high_low_range
    
    # Smoothed Moving Average
    sma_5_days = adjusted_returns.rolling(window=5).mean()
    alpha_factor = sma_5_days.ewm(span=5, adjust=False).mean()
    
    # Integrate Amount Data
    daily_dollar_volume = df['volume'] * df['close']
    alpha_factor *= daily_dollar_volume
    
    return alpha_factor
```
