```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    df['aggregate_return'] = df['intraday_return'] + df['overnight_return'].shift(1)
    
    # Weight High-Low Range by Volume
    df['high_low_volatility'] = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    df['adjusted_return'] = df['aggregate_return'] * df['high_low_volatility']
    
    # Smoothed Moving Average
    df['sma_5'] = df['adjusted_return'].rolling(window=5).mean()
    df['ema_smoothed'] = df['sma_5'].ewm(span=5, adjust=False).mean()
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum']
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['overnight_return'].shift(1) * df['price_momentum_ratio']
    df['integrated_factor'] *= np.where(df['volume'] > df['volume'].mean(), 1.7, 0.3)
    
    # Compute Price Momentum
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['price_momentum'] = df['log_returns'].rolling(window=20).sum()
    
    # Apply Volume Confirmation
    df['volume_confirmation'] = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    
    # Generate Final Momentum Factor
    df['final_momentum_factor'] = df['price_momentum'] * df['volume_confirmation']
    
    # Incorporate High-Low Volatility into Final Factor
    df['intraday_volatility'] = df['intraday_return'].rolling(window=5).std()
    df['final_factor_high_low'] = df['high_low_volatility'] * df['intraday_volatility']
    
    # Integrate All Components
    df['final_factor'] = df['ema_smoothed'] + df['final_momentum_factor'] + df['final_factor_high_low']
    
    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = df['price_momentum'] * (df['high_low_volatility'] / df['volume'])
    
    # Incorporate Historical Volatility
    df['historical_volatility'] = df['log_returns'].rolling(window=20).std()
    df['weighted_historical_volatility'] = df['historical_volatility'] * (df['high_low_volatility'] / df['volume'])
    
    # Combine Intraday and Historical Volatility
    df['total_volatility'] = df['intraday_volatility'] * 0.5 + df['historical_volatility'] * 0.5
    
    # Final Alpha Factor
    df['alpha_factor'] = df['final_factor'] + df['combined_momentum_volatility'] + df['weighted_historical_volatility'] * df['total_volatility']
    
    return df['alpha_factor'].dropna()
```
