```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['open'].shift(1) - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Average Volume Over a Period
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum
    
    # Integrate Factors
    integrated_factor = intraday_return * prev_close_to_open_return * price_momentum_ratio
    integrated_factor = integrated_factor * np.where(df['volume'] > avg_volume_10, 1.5, 0.5)
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = short_term_momentum + (high_low_range * df['volume'])
    
    # Confirm Momentum with Volume and Amount
    volume_change = df['volume'] - df['volume'].shift(1)
    positive_volume_filter = np.where(volume_change > 0, 1, 0)
    
    amount_change = df['amount'] - df['amount'].shift(1)
    positive_amount_filter = np.where(amount_change > 0, 1, 0)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + prev_close_to_open_return
    
    # Adjust for Volume
    adjusted_aggregate_returns = aggregate_returns * df['volume']
    
    # Weight by Intraday Volatility
    weighted_adjusted_returns = adjusted_aggregate_returns * intraday_volatility
    
    # Smoothed Moving Average
    sma_5 = weighted_adjusted_returns.rolling(window=5).mean()
    ema_smoothed = sma_5.ewm(span=5, adjust=False).mean()
    
    # Calculate Daily Log Return
    daily_log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Price Momentum
    price_momentum = daily_log_return.rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = np.where(df['volume'] > df['volume'].shift(1), 1, 0)
    
    # Generate Final Momentum Factor
    final_momentum_factor = price_momentum * volume_confirmation
    
    # Combine All Factors
    combined_factor = ema_smoothed + final_momentum_factor + (high_low_range * intraday_volatility)
    
    return combined_factor
```
