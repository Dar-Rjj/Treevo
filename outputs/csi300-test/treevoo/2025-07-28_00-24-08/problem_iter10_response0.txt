import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return'].shift(1)
    
    # Intraday Trend Strength
    df['daily_range'] = (df['high'] - df['low']) / df['low']
    df['intraday_pct_change'] = (df['close'] - df['open']) / df['open']
    
    # Volume Adjusted Reversal Signal
    df['volume_pct_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['close_to_next_open_return'] = (df['open'].shift(-1) - df['close']) / df['close']
    df['inverse_close_to_next_open_return'] = -df['close_to_next_open_return']
    
    # Smoothed Moving Average of Composite Factor
    df['composite_factor'] = df['aggregate_returns'] + df['intraday_pct_change'] + df['inverse_close_to_next_open_return']
    df['sma_5d_composite'] = df['composite_factor'].rolling(window=5).mean()
    df['ema_composite'] = df['sma_5d_composite'].ewm(span=5, adjust=False).mean()
    
    # Intraday Return Factors
    df['intraday_high_low_ratio'] = df['high'] / df['low']
    df['open_to_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    df['sma_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['sma_volume_5d'])
    
    # Summarize Momentum-Reversal Indicator
    df['momentum_reversal_indicator'] = df['intraday_high_low_ratio'] + df['open_to_close_direction'] + df['volume_weighted_returns']
    df['final_factor'] = df['momentum_reversal_indicator'].apply(lambda x: 1 if x > 0 else -1)
    
    return df['final_factor']
