```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength index (RSI) as a component of the alpha factor
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -1 * delta.clip(upper=0)
    ema_up = up.ewm(com=13, adjust=False).mean()
    ema_down = down.ewm(com=13, adjust=False).mean()
    rs = ema_up / ema_down
    rsi = 100 - (100 / (1 + rs))

    # Calculate the average true range (ATR) as another component
    tr = df[['high' - 'low', abs('high' - df['close'].shift()), abs('low' - df['close'].shift())]].max(axis=1)
    atr = tr.rolling(window=14).mean()

    # Use the volume-weighted close price change
    vwap_change = (df['close'] * df['volume']).pct_change()

    # Combine the three components into an alpha factor
    alpha_factor = (rsi - 50) * (vwap_change / atr)

    return alpha_factor
```
