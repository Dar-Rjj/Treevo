```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day's Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Short-Term Price Momentum
    df['short_term_momentum'] = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    df['long_term_momentum'] = df['close'] - df['close'].shift(20)
    
    # Price Momentum Ratio
    df['price_momentum_ratio'] = df['short_term_momentum'] / df['long_term_momentum'].replace(0, np.nan)
    
    # Volume Change Indicator
    df['volume_change'] = df['volume'] - df['volume'].shift(5)
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['overnight_return'] * df['price_momentum_ratio']
    df['volume_adjusted_factor'] = df['integrated_factor'] * np.where(df['volume'] > df['volume'].mean(), 1.5, 0.5)
    
    # Weight High-Low Range by Volume
    df['high_low_range'] = (df['high'] - df['low']) * df['volume']
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['high_low_range']
    
    # Smoothed Moving Average
    df['sma_adjusted_returns'] = df['adjusted_returns'].rolling(window=5).mean()
    df['ema_adjusted_returns'] = df['sma_adjusted_returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate 5-Day Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Adjust Final Alpha Factor by 5-Day EMA of Volume
    df['final_alpha_factor'] = df['ema_adjusted_returns'] * df['ema_volume']
    
    # Incorporate Historical Volatility
    df['historical_volatility'] = df['close'].pct_change().rolling(window=20).std() * np.sqrt(252)
    df['volatility_weighted'] = df['historical_volatility'] * df['volume']
    
    # Combine Multiple Factors
    w1, w2 = 0.5, 0.5  # Example weights
    df['composite_factor'] = w1 * df['final_alpha_factor'] + w2 * df['volatility_weighted']
    
    # Enhance Alpha Factor with VWAP Deviation
    df['vwap'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3).cumsum() / df['volume'].cumsum()
    df['vwap_deviation'] = df['close'] - df['vwap']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['composite_factor'] + df['vwap_deviation']
    
    return df['final_alpha_factor']
```
