```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_mom = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_mom = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_mom / long_term_mom
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Calculate Average Volume Over a Period (last 20 days for example)
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Weight High-Low Range by Volume
    vol_weighted_high_low_range = high_low_range * (df['volume'] / avg_volume)
    
    # Combine Volume Adjusted Volatility and Volume Change
    combined_vol_adj_volatility = vol_weighted_high_low_range + volume_change
    
    # Final Alpha Factor
    alpha_factor = price_momentum_ratio * combined_vol_adj_volatility
    
    return alpha_factor
```
