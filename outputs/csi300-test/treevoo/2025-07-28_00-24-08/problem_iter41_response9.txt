```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily price changes
    df['price_change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate volume momentum
    df['volume_momentum'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate amount per unit volume
    df['amount_per_volume'] = df['amount'] / df['volume']
    
    # Calculate price-volume correlation
    df['price_volume_corr'] = df['close'].diff().rolling(window=10).corr(df['volume'].diff())
    
    # Calculate volatility-adjusted returns
    df['volatility_adjusted_returns'] = (df['close'] - df['close'].shift(1)) / df['high_low_range']
    
    # Calculate combined momentum
    df['combined_momentum'] = df['price_change'] + df['volume_momentum']
    
    # Calculate transaction size adjusted volatility
    df['adjusted_volatility'] = df['high_low_range'] * df['amount_per_volume']
    
    # Calculate trend following factor using 20-day SMA
    df['sma_20'] = df['close'].rolling(window=20).mean()
    df['trend_following'] = df['sma_20'] - df['close']
    
    # Calculate breakout signal
    df['breakout_signal'] = (df['close'] > df['high'].rolling(window=20).max()).astype(int)
    
    # Composite alpha factor
    df['composite_alpha_factor'] = (df['combined_momentum'] 
                                   + df['price_volume_corr'] 
                                   + df['volatility_adjusted_returns'] 
                                   + df['trend_following'] 
                                   + df['breakout_signal'])
    
    return df['composite_alpha_factor']
```
