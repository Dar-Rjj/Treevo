import pandas as pd
import numpy as np
from scipy.signal import find_peaks

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']

    # Calculate Previous Day's Overnight Return
    overnight_return = (df['open'].shift(-1) - df['close']) / df['close']

    # Aggregate Returns
    aggregated_returns = intraday_return + overnight_return

    # Weight High-Low Range by Volume
    high_low_range_weighted = (df['high'] - df['low']) * df['volume']

    # Adjust for Combined Volatility
    adjusted_returns = aggregated_returns * high_low_range_weighted

    # Smoothed Moving Average
    sma_5 = adjusted_returns.rolling(window=5).mean()
    exponential_smoothed = sma_5.ewm(span=5, adjust=False).mean()

    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)

    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)

    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum

    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)

    # Integrate Factors
    integrated_factor = intraday_return * overnight_return * price_momentum_ratio
    average_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_factor = np.where(df['volume'] > average_volume, integrated_factor * 1.7, integrated_factor * 0.3)

    # Final Factor
    final_factor = volume_adjusted_factor * exponential_smoothed

    # Trend Following Factors
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    ema_9 = df['close'].ewm(span=9, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()

    # Momentum Indicators
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    roc_12 = (df['close'] - df['close'].shift(12)) / df['close'].shift(12) * 100

    # Volume Patterns
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    cmf = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])
    cmf = (cmf * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Combine Price and Volume
    pvt = (df['close'].pct_change() * df['volume']).cumsum()
    vwap = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Market Sentiment Analysis
    bullish_candles = ((df['close'] > df['open']) & (df['close'].shift(1) > df['open'].shift(1))).rolling(window=10).sum()
    bearish_candles = ((df['close'] < df['open']) & (df['close'].shift(1) < df['open'].shift(1))).rolling(window=10).sum()
    high_low_range = (df['high'] - df['low']).rolling(window=10).mean()

    # Advanced Pattern Recognition
    def find_head_and_shoulders(prices):
        peaks, _ = find_peaks(prices, prominence=0.05)
        troughs, _ = find_peaks(-prices, prominence=0.05)
        return len(peaks) >= 3 and len(troughs) >= 2

    def find_double_top_bottom(prices):
        peaks, _ = find_peaks(prices, prominence=0.05)
        troughs, _ = find_peaks(-prices, prominence=0.05)
        return len(peaks) == 2 or len(troughs) == 2

    head_and_shoulders = df['close'].rolling(window=30).apply(find_head_and_shoulders, raw=True)
