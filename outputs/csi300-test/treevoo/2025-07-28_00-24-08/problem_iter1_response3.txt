```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['open'].shift(n)
    
    # Weight High-Low Range by Volume
    volume_weighted_high_low = high_low_range * df['volume']
    
    # Normalize by Average Volume
    normalized_volatility = volume_weighted_high_low / avg_volume
    
    # Combine Momentum and Volatility
    factor_value = price_momentum + normalized_volatility
    
    return factor_value
```
