```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of high to low
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the average volume over a 5-day period
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the amount per volume as a measure of price per share
    amount_per_volume = df['amount'] / df['volume']
    
    # Calculate the factor by combining the above indicators
    # This is a heuristic approach where we weigh each signal
    factor = (0.4 * close_pct_change) + (0.3 * high_low_ratio) + (0.2 * (df['volume'] - avg_volume_5d)) + (0.1 * amount_per_volume)
    
    # Normalize the factor values
    factor = (factor - factor.mean()) / factor.std()
    
    return factor
```
