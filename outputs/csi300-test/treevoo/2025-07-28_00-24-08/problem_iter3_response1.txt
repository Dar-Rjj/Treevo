```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday High-Low Range Ratio
    df['High_Low_Range'] = df['high'] / df['low']

    # Open-to-Close Direction (1 if close > open, 0 otherwise)
    df['Open_Close_Direction'] = (df['close'] > df['open']).astype(int)

    # Simple Moving Average of Volumes
    df['SMA_Volume'] = df['volume'].rolling(window=n).mean()

    # Volume Weighted Returns
    df['Volume_Weighted_Returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['SMA_Volume'])

    # Summarize Momentum-Reversal Indicator
    df['Intraday_Factor'] = df['High_Low_Range'] + df['Open_Close_Direction'] + df['Volume_Weighted_Returns']
    df['Momentum_Reversal_Indicator'] = df['Intraday_Factor'].apply(lambda x: 1 if x > 0 else -1)

    # Calculate Overnight Return Factor
    df['Overnight_Return'] = df['open'] - df['close'].shift(1)

    # Combine Intraday and Overnight Factors
    df['Combined_Factor'] = df['Intraday_Factor'] + df['Overnight_Return']
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] * (df['volume'] / df['SMA_Volume'])

    # Final Combined Factor
    df['Final_Combined_Factor'] = df['Adjusted_Combined_Factor'].apply(lambda x: 1 if x > 0 else -1)

    return df['Final_Combined_Factor']
```
