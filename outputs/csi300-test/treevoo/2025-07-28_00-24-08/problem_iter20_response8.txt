```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns for open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate some intermediate values
    df['mid_price'] = (df['high'] + df['low']) / 2
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()
    
    # Heuristic 1: Price change momentum
    df['price_momentum'] = df['price_change'].rolling(window=5).mean()
    
    # Heuristic 2: Volume change momentum
    df['volume_momentum'] = df['volume_change'].rolling(window=5).mean()
    
    # Heuristic 3: Relative strength of price to mid-price
    df['rel_strength'] = (df['close'] - df['mid_price']) / df['mid_price']
    
    # Heuristic 4: Amount-to-volume ratio
    df['amt_vol_ratio'] = df['amount'] / df['volume']
    
    # Combine the heuristics into a single factor
    df['alpha_factor'] = (
        0.4 * df['price_momentum'] +
        0.3 * df['volume_momentum'] +
        0.2 * df['rel_strength'] +
        0.1 * df['amt_vol_ratio']
    )
    
    return df['alpha_factor']
```
