```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    overnight_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return + overnight_return
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    average_volume = df['volume'].rolling(window=5).mean()
    
    # Combine High-Low Range and Volume
    volume_weighted_high_low = high_low_range * df['volume']
    normalized_volatility = volume_weighted_high_low / average_volume
    
    # Calculate Price Momentum
    price_momentum = (df['close'] - df['open'].shift(5)) / df['open'].shift(5)
    
    # Adjust for Combined Volatility
    adjusted_returns = aggregate_returns * normalized_volatility
    
    # Smoothed Moving Average
    simple_moving_average = adjusted_returns.rolling(window=5).mean()
    smoothed_moving_average = simple_moving_average.ewm(span=5, adjust=False).mean()
    
    return smoothed_moving_average
```
