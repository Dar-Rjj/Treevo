```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Volume Surge Indicator
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = (df['volume'] > df['avg_volume_5d']).astype(int)
    
    # Integrate Factors
    df['integrated_factor'] = df['intraday_return'] * df['prev_close_to_open_return']
    df['integrated_factor'] = df['integrated_factor'] * (1.5 if df['volume_surge'] == 1 else 0.5)
    
    # Intraday Return Factors
    df['intraday_high_low_ratio'] = df['high'] / df['low']
    df['open_close_direction'] = (df['close'] > df['open']).astype(int)
    
    # Volume Weighted Smoothed Returns
    df['sma_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_returns'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['sma_volume'])
    
    # Summarize Hybrid Momentum-Reversal Indicator
    df['hybrid_momentum_reversal'] = df['intraday_return'] + df['volume_weighted_returns']
    df['final_factor'] = df['hybrid_momentum_reversal'].apply(lambda x: 1 if x > 0 else -1)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['final_factor'] * df['integrated_factor']
    
    return df['alpha_factor']
```
