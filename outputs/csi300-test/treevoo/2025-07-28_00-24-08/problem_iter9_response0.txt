import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, lookback=14, rsi_overbought=70, rsi_oversold=30):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    df['overnight_return'] = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    df['aggregate_returns'] = df['intraday_return'] + df['overnight_return']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Combine High-Low Range and Volume
    df['volume_weighted_high_low'] = df['high_low_range'] * df['volume']
    df['normalized_volatility'] = df['volume_weighted_high_low'] / df['avg_volume']
    
    # Calculate Price Momentum
    df['momentum'] = (df['close'] - df['open'].shift(n)) / df['open'].shift(n)
    
    # Adjust for Combined Volatility
    df['adjusted_returns'] = df['aggregate_returns'] * df['normalized_volatility']
    
    # Introduce Trend Reversal Indicator
    def compute_rsi(prices, periods=lookback):
        delta = prices.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=periods).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=periods).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    df['rsi'] = compute_rsi(df['close'], lookback)
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['trend_direction'] = np.where(df['close'] > df['50_day_ma'], 1, -1)
    
    # Trend Reversal Logic
    df['trend_reversal'] = np.where((df['rsi'] > rsi_overbought) & (df['trend_direction'] == -1), -1,
                                    np.where((df['rsi'] < rsi_oversold) & (df['trend_direction'] == 1), 1, 0))
    
    # Smoothed Moving Average
    df['smoothed_ma'] = df['adjusted_returns'].rolling(window=5).mean()
    df['exponential_smoothed'] = df['smoothed_ma'].ewm(span=5, adjust=False).mean()
    
    # Final Factor Calculation
    df['final_factor'] = df['exponential_smoothed'] * df['trend_reversal']
    
    return df['final_factor']
