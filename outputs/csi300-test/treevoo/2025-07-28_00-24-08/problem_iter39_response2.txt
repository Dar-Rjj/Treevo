```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum-based alpha factors
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    df['10_day_momentum'] = df['close'] - df['close'].shift(10)
    
    avg_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['20_day_avg_momentum'] = avg_price - avg_price.shift(20)
    df['30_day_avg_momentum'] = avg_price - avg_price.shift(30)
    
    df['5_day_vol_weighted_mom'] = ((df['volume'] * (df['close'] - df['close'].shift(1))).rolling(window=5).sum() 
                                    / df['volume'].rolling(window=5).sum())
    df['10_day_vol_weighted_mom'] = ((df['volume'] * (df['close'] - df['close'].shift(1))).rolling(window=10).sum() 
                                     / df['volume'].rolling(window=10).sum())

    # Volatility-based alpha factors
    df['5_day_ATR'] = (df['high'] - df['low']).rolling(window=5).mean()
    df['10_day_ATR'] = (df['high'] - df['low']).rolling(window=10).mean()
    
    df['20_day_std_dev'] = df['close'].diff().rolling(window=20).std()
    df['30_day_std_dev'] = df['close'].diff().rolling(window=30).std()

    # Liquidity-based alpha factors
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    df['5_day_avg_amount'] = df['amount'].rolling(window=5).mean()
    df['10_day_avg_amount'] = df['amount'].rolling(window=10).mean()

    # Pattern-based alpha factors
    df['bearish_engulfing'] = ((df['open'] < df['close'].shift(1)) & (df['close'] > df['open'].shift(1))).astype(int)
    df['bullish_engulfing'] = ((df['open'] > df['close'].shift(1)) & (df['close'] < df['open'].shift(1))).astype(int)
    
    df['new_5_day_high'] = df['high'].rolling(window=5).max()
    df['new_5_day_low'] = df['low'].rolling(window=5).min()

    # Combine all factors into a single series
    alpha_factors = (df['5_day_momentum'] + df['10_day_momentum'] + 
                     df['20_day_avg_momentum'] + df['30_day_avg_momentum'] + 
                     df['5_day_vol_weighted_mom'] + df['10_day_vol_weighted_mom'] +
                     -df['5_day_ATR'] - df['10_day_ATR'] - 
                     -df['20_day_std_dev'] - df['30_day_std_dev'] + 
                     df['5_day_avg_volume'] + df['10_day_avg_volume'] + 
                     df['5_day_avg_amount'] + df['10_day_avg_amount'] + 
                     df['bearish_engulfing'] + df['bullish_engulfing'] + 
                     df['new_5_day_high'] - df['new_5_day_low'])
    
    return alpha_factors
```
