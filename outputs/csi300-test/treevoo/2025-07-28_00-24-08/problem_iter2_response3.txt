```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / (long_term_momentum + 1e-8)  # Adding small constant to avoid division by zero
    
    # Calculate Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume Over a Period
    average_volume = df['volume'].rolling(window=20).mean()
    
    # Weight High-Low Range by Volume
    weighted_high_low_range = high_low_range * df['volume']
    
    # Normalize by Average Volume
    normalized_volatility = weighted_high_low_range / (average_volume + 1e-8)  # Adding small constant to avoid division by zero
    
    # Combine Volume Adjusted Volatility and Volume Change
    combined_volatility_and_volume = normalized_volatility + volume_change
    
    # Final Alpha Factor
    alpha_factor = price_momentum_ratio * combined_volatility_and_volume
    
    return alpha_factor
```
