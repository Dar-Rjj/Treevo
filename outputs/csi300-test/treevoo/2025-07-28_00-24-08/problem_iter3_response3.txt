```python
import pandas as pd

def heuristics_v2(df, n_period=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Average Volume Over a Period
    avg_volume = df['volume'].rolling(window=n_period).mean()

    # Calculate Price Momentum
    price_momentum = (df['close'] / df['open'].shift(n_period)) - 1
    
    # Combine High-Low Range and Volume
    vol_weighted_high_low = high_low_range * df['volume']
    adjusted_volatility = vol_weighted_high_low / avg_volume

    # Combine Momentum and Adjusted Volatility
    factor_value = price_momentum + adjusted_volatility

    return factor_value
```
