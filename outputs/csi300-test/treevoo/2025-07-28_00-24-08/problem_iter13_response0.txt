import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    df['short_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Momentum
    df['long_momentum'] = df['close'] - df['close'].shift(20)
    
    # Compute Price Momentum Ratio
    df['momentum_ratio'] = df['short_momentum'] / df['long_momentum']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    df['prev_day_close_to_open_return'] = (df['close'].shift(1) - df['open']) / df['close'].shift(1)
    
    # Integrate Intraday and Previous Day's Returns
    df['integrated_returns'] = df['intraday_return'] * df['prev_day_close_to_open_return']
    
    # Combine Price Momentum and Intraday Reversal
    df['combined_factor'] = df['momentum_ratio'] * df['integrated_returns']
    
    # Volume Surge Indicator
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_surge'] = df['volume'] > df['5_day_avg_volume']
    
    # Conditionally Adjust by Volume Surge
    df['final_factor'] = df['combined_factor'] * (1.75 if df['volume_surge'] else 0.75)
    
    return df['final_factor'].dropna()
