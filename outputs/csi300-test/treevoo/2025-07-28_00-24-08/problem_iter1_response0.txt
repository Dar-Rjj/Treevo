```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].rolling(window=n-1).mean().shift(1)) / df['close'].rolling(window=n-1).mean().shift(1)
    
    # Weight by Intraday Volatility
    factor_values = intraday_volatility * momentum
    
    return factor_values
```
