import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['open']
    
    # Calculate Previous Day's Close-to-Open Return
    prev_close_to_open_return = (df['open'].shift(1) - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Average Volume Over a Period
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    is_above_avg_volume = df['volume'] > avg_volume_5d
    
    # Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Long-Term Price Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Volume Change Indicator
    volume_change = df['volume'] - df['volume'].shift(5)
    
    # Price Momentum Ratio
    price_momentum_ratio = short_term_momentum / long_term_momentum.replace(0, np.nan)
    
    # Integrate Factors
    integrated_factor = (intraday_return * prev_close_to_open_return * price_momentum_ratio)
    integrated_factor = integrated_factor.where(is_above_avg_volume, integrated_factor * 0.5)
    integrated_factor = integrated_factor.where(~is_above_avg_volume, integrated_factor * 1.5)
    
    # Combine Momentum and Volatility
    weighted_high_low_range = high_low_range * df['volume']
    
    # Calculate Intraday Return (Enhanced)
    intraday_return_enhanced = (df['close'] - df['open']) / df['open']
    
    # Calculate Previous Day Overnight Return
    prev_day_overnight_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Aggregate Returns
    aggregate_returns = intraday_return_enhanced + prev_day_overnight_return
    
    # Weight by Intraday Volatility
    intraday_volatility = df[['open', 'high', 'low', 'close']].std(axis=1)
    weighted_aggregate_returns = aggregate_returns * intraday_volatility
    
    # Adjust for Volume
    volume_adjusted_returns = weighted_aggregate_returns * df['volume']
    
    # Smoothed Moving Average
    sma_5d = volume_adjusted_returns.rolling(window=5).mean()
    smoothed_returns = sma_5d.ewm(span=5, adjust=False).mean()
    
    # Compute Price Momentum
    log_returns = np.log(df['close'] / df['close'].shift(1))
    price_momentum = log_returns.rolling(window=20).sum()
    
    # Apply Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Generate Final Momentum Factor
    final_momentum_factor = price_momentum * volume_confirmation
    
    # Incorporate High-Low Volatility into Final Factor
    high_low_volatility = high_low_range * intraday_volatility
    
    # Integrate All Components
    final_alpha_factor = (smoothed_returns + final_momentum_factor + high_low_volatility).dropna()
    
    return final_alpha_factor
