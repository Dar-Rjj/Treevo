```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price by Open Price
    adj_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = high_price_indicator * low_price_indicator
    
    # Adjust for Volume Impact on Intraday Momentum
    volume_adjusted_intraday_momentum = df['volume'] * intraday_momentum
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    n = 20
    long_term_momentum = (df['close'].pct_change(n))
    volume_ratio = df['volume'] / df['volume'].shift(n)
    final_long_term_factor = long_term_momentum + (long_term_momentum * volume_ratio)
    
    # Daily Price Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Compute Volume-Based Adjustment Factor
    avg_volume = df['volume'].rolling(window=20).mean()
    volatility_adjustment_factor = (daily_range / avg_daily_range) * (df['volume'] / avg_volume)
    
    # High-Low Price Range Momentum
    close_t_minus_10 = df['close'].shift(10)
    high_low_momentum = df['close'] - close_t_minus_10
    
    # Calculate True Range
    true_range = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    vwatr = (true_range * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Adjust Momentum by VWATR
    adjusted_high_low_momentum = high_low_momentum / vwatr
    
    # Final Alpha Factor
    alpha_factor = (adj_close + volume_adjusted_intraday_momentum + final_long_term_factor) * adjusted_high_low_momentum
    
    return alpha_factor
```
