import pandas as pd

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = 0.5 * high_price_indicator + 0.5 * low_price_indicator

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema = df['volume'].ewm(span=10).mean()
    normalized_volume = df['volume'] / volume_ema
    adjusted_intraday_momentum = intraday_momentum * normalized_volume

    # Calculate Weighted Intraday Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean() * df['volume'].rolling(window=7).mean()

    # Calculate Adjusted High-Low Breakout Potential
    high_low_range = df['high'] - df['low']
    breakout_potential = (high_low_range * df['volume']).ewm(span=25).mean() * df['volume'].rolling(window=15).mean()

    # Calculate Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_intraday_momentum) / breakout_potential

    # Incorporate Longer-Term Volume-Adjusted Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    volume_ratio = df['volume'] / df['volume'].shift(20)
    long_term_factor = long_term_momentum + (long_term_momentum * volume_ratio)

    # Calculate Daily Price Return
    daily_return = df['close'] - df['close'].shift(1)

    # Identify Significant Price Increase
    significant_increase = (daily_return > 0).astype(int)

    # Calculate Volume Spike Indicator
    volume_spike = (df['volume'] > df['volume'].rolling(window=7).mean()).astype(int)

    # Compute Weighted Price Momentum
    weighted_price_momentum = (daily_return * significant_increase).cumsum()

    # Apply Volume Spike Filter
    filtered_momentum = weighted_price_momentum * volume_spike

    # Final Alpha Factor Synthesis
    alpha_factor = (adjusted_intraday_momentum + momentum_reversal + long_term_factor + filtered_momentum) * volume_spike

    # Adjust for Open Price
    final_alpha_factor = alpha_factor / df['open']

    return final_alpha_factor
