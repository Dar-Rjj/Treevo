```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    df['adjusted_close'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    weights = [0.6, 0.4]  # Weights for High and Low
    weighted_intraday_momentum = (weights[0] * high_price_indicator + weights[1] * low_price_indicator)

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    ema_volume_10 = df['volume'].ewm(span=10).mean()
    volume_normalized = df['volume'] / ema_volume_10
    intraday_momentum_adjusted = weighted_intraday_momentum * volume_normalized
    
    # Calculate Weighted Intraday Momentum
    intraday_return = df['close'] - df['open']
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean()
    recent_volume_weight = df['volume'].rolling(window=7).mean()
    weighted_intraday_momentum = weighted_intraday_momentum * recent_volume_weight
    
    # Calculate Adjusted High-Low Breakout Potential
    high_low_range = df['high'] - df['low']
    breakout_potential = (high_low_range * df['volume']).ewm(span=25).mean()
    recent_volume_weight_15 = df['volume'].rolling(window=15).mean()
    adjusted_breakout_potential = breakout_potential * recent_volume_weight_15
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    volume_ratio = df['volume'] / df['volume'].shift(20)
    long_term_factor = long_term_momentum * volume_ratio
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    significant_increase = (daily_price_return > 0).astype(int)
    
    # Calculate Volume Spike Indicator
    avg_weekly_volume = df['volume'].rolling(window=7).mean()
    volume_spike = (df['volume'] > avg_weekly_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_increase
    weighted_price_momentum = weighted_price_momentum.rolling(window=7).sum() * volume_spike
    
    # Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_intraday_momentum) / adjusted_breakout_potential
    
    # Price-Volume Synchronization
    price_change = df['close'] - df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    price_volume_sync = price_change.rolling(window=10).corr(volume_change)
    
    # Consolidate All Factors
    total_factor = (
        df['adjusted_close'] + 
        intraday_momentum_adjusted + 
        weighted_intraday_momentum + 
        adjusted_breakout_potential + 
        long_term_factor + 
        price_volume_sync
    )
    final_factor = total_factor / df['open']
    
    return final_factor
```
