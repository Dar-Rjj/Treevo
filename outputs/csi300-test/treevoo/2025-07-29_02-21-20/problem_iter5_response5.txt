```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Logarithmic Changes
    log_open_close_change = np.log(df['close'] / df['open'])
    log_high_low_change = np.log(df['high'] / df['low'])

    # Aggregate Intraday Logarithmic Changes
    aggregated_log_changes = log_open_close_change + log_high_low_change

    # Volume Synchrony and Impact
    volume_ma = df['volume'].rolling(window=10).mean()
    daily_volume_deviation = df['volume'] - volume_ma
    volume_impact_factor = aggregated_log_changes * daily_volume_deviation

    # Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = high_price_indicator * low_price_indicator

    # Volume Adjusted Intraday Factors
    volume_adjusted_log_change = aggregated_log_changes * df['volume']
    volume_adjusted_high_low_range = (df['high'] - df['low']) * df['volume']

    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_log_change.rolling(window=10).mean()
    recent_volume_weight = df['volume'].rolling(window=5).mean()
    smoothed_intraday_momentum = smoothed_intraday_momentum * recent_volume_weight

    # High-Low Breakout Potential
    breakout_potential = volume_adjusted_high_low_range.rolling(window=20).mean()
    recent_breakout_volume_weight = df['volume'].rolling(window=10).mean()
    breakout_potential = breakout_potential * recent_breakout_volume_weight

    # Momentum Reversal Indicator
    momentum_reversal = (aggregated_log_changes - smoothed_intraday_momentum) / breakout_potential

    # Consolidate Factors
    factor = (volume_impact_factor + intraday_momentum + momentum_reversal) / df['open']

    return factor
```
