import pandas as pd

def heuristics_v2(df):
    n = 5
    m = 10
    p = 20
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust by Close
    high_low_range_adj = high_low_range / df['close']
    
    # Volume and Amount Adjustment
    sum_volume_n_days = df['volume'].rolling(window=n).sum()
    sum_amount_n_days = df['amount'].rolling(window=n).sum()
    current_volume_ratio = df['volume'] / sum_volume_n_days
    current_amount_ratio = df['amount'] / sum_amount_n_days
    
    # Combine Ratios (Weighted Average)
    weighted_ratio = (0.6 * current_volume_ratio) + (0.4 * current_amount_ratio)
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.rolling(window=m).mean()
    recent_volume_avg = df['volume'].rolling(window=5).mean()
    smoothed_intraday_momentum_weighted = smoothed_intraday_momentum * recent_volume_avg
    
    # High-Low Breakout Potential
    breakout_potential = volume_adjusted_high_low_range.rolling(window=p).mean()
    breakout_potential_weighted = breakout_potential * recent_volume_avg
    
    # Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / breakout_potential
    
    # Calculate VWAP
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Range and VWAP
    high_vwap_diff = df['high'] - vwap
    vwap_low_diff = vwap - df['low']
    vwap_based_intraday_range = high_vwap_diff + vwap_low_diff
    
    # Apply a moving average to smooth the factor
    smoothed_factor = vwap_based_intraday_range.rolling(window=m).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (momentum_reversal_indicator + breakout_potential_weighted + smoothed_factor) / 3
    
    return final_alpha_factor
