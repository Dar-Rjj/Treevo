def heuristics_v2(df, n=20):
    close_t = df['close']
    close_tn = df['close'].shift(n)
    volume_t = df['volume']
    volume_tn = df['volume'].shift(n)
    
    # Calculate Momentum
    momentum = (close_t - close_tn) / close_tn
    
    # Calculate Volume Ratio
    volume_ratio = volume_t / volume_tn
    
    # Adjust Momentum for Volume Changes
    final_alpha_factor = momentum * volume_ratio
    
    return final_alpha_factor
