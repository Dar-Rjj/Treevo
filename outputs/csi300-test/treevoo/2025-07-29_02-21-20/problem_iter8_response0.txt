```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Smoothed Intraday Momentum
    smoothed_intraday_momentum = (volume_adjusted_intraday_return.rolling(window=10).mean() 
                                  * df['volume'].rolling(window=5).mean())
    
    # Calculate True Range
    true_range = pd.concat([df['high'] - df['low'],
                            (df['high'] - df['close'].shift(1)).abs(),
                            (df['low'] - df['close'].shift(1)).abs()], axis=1).max(axis=1)
    
    # Calculate Volume Weighted Average True Range (VWATR)
    vwatr = true_range * df['volume']
    
    # Calculate High-Low Breakout Potential
    high_low_breakout_potential = (volume_adjusted_high_low_range.rolling(window=20).mean() 
                                   * df['volume'].rolling(window=10).mean())
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Adjust Price Momentum by VWATR
    adjusted_price_momentum = price_momentum / vwatr
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Range and VWAP
    combined_intraday_range_vwap = (df['high'] - vwap) + (vwap - df['low'])
    
    # Apply a moving average to smooth the factor
    smoothed_factor = combined_intraday_range_vwap.rolling(window=10).mean()
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    volume_ratio = df['volume'] / df['volume'].shift(10)
    combined_momentum_volume_ratio = volume_adjusted_momentum * volume_ratio + volume_adjusted_momentum
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_price_momentum * high_low_breakout_potential 
                          + momentum_reversal_indicator 
                          + combined_momentum_volume_ratio 
                          + smoothed_factor)
    
    return final_alpha_factor
```
