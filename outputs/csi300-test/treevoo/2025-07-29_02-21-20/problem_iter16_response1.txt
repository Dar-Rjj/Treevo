```python
import pandas as pd

def heuristics_v2(df, intraday_lookback=10, longterm_lookback=20):
    # Calculate Adjusted Close Price
    adj_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    avg_intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=intraday_lookback).mean()
    normalized_volume = df['volume'] / volume_ma
    vol_impact_intraday_mom = avg_intraday_momentum * normalized_volume
    
    # Long-Term Momentum Adjusted for Volume
    long_term_mom = (df['close'] - df['close'].shift(longterm_lookback)) / df['close'].shift(longterm_lookback)
    vol_ratio = df['volume'] / df['volume'].shift(longterm_lookback)
    vol_adj_long_term_mom = long_term_mom * vol_ratio
    
    # Consolidate All Factors
    total_factor = adj_close + vol_impact_intraday_mom + vol_adj_long_term_mom
    final_factor = total_factor / df['open']

    return final_factor
```
