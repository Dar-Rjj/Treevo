```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff(10)
    
    # Calculate True Range
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Volume Weighted Average True Range (VWATR)
    vwatr = true_range * df['volume']
    
    # Adjust Price Momentum by VWATR
    adjusted_momentum = price_momentum / vwatr
    
    # Final Alpha Factor
    alpha_factor = high_low_range * adjusted_momentum
    
    return alpha_factor
```
