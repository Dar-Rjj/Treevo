```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    df['adjusted_close'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    weights = [0.5, 0.5]  # Equal weights for simplicity
    intraday_momentum = (weights[0] * high_indicator + weights[1] * low_indicator)
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema = df['volume'].ewm(span=10).mean()
    normalized_volume = df['volume'] / volume_ema
    volume_impacted_intraday_momentum = intraday_momentum * normalized_volume
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    n = 20  # Lookback period
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    adjusted_long_term_momentum = long_term_momentum * volume_ratio + long_term_momentum
    
    # Add Volume Spike Filter
    spike_threshold = 2.0
    volume_spike = df['volume'] / df['volume'].shift(1)
    no_spike = (volume_spike < spike_threshold) | (volume_spike.isna())
    adjusted_long_term_momentum = adjusted_long_term_momentum * no_spike
    
    # Calculate Weighted Intraday Momentum
    intraday_return = df['close'] - df['open']
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean() * df['volume'].rolling(window=7).mean()
    
    # Calculate Adjusted High-Low Breakout Potential
    high_low_range = df['high'] - df['low']
    adjusted_high_low_breakout_potential = (high_low_range * df['volume']).ewm(span=25).mean() * df['volume'].rolling(window=15).mean()
    
    # Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_intraday_momentum) / adjusted_high_low_breakout_potential
    
    # Price-Volume Synchronization
    price_change = df['close'] - df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    price_volume_corr = price_change.rolling(window=20).corr(volume_change)
    
    # Consolidate All Factors
    total_factor = (
        df['adjusted_close'] +
        volume_impacted_intraday_momentum +
        weighted_intraday_momentum +
        adjusted_high_low_breakout_potential +
        adjusted_long_term_momentum +
        price_volume_corr
    ) / df['open']
    
    return total_factor
```
