import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']

    # Calculate Volume-Adjusted Typical Price (VATP)
    typical_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    volume_adjusted_tp = typical_price * np.sqrt(df['volume'])
    vatp = volume_adjusted_tp.sum() / df['volume'].sum()

    # Calculate Short-Term Momentum
    lookback_period_short = 5
    short_term_momentum = (df['close'] - df['close'].shift(lookback_period_short)) / df['close'].shift(lookback_period_short)

    # Adjust for Volume Changes in Short-Term Momentum
    volume_ratio_short = df['volume'] / df['volume'].shift(lookback_period_short)
    adjusted_short_term_momentum = short_term_momentum * volume_ratio_short + short_term_momentum

    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    enhanced_intraday_momentum = (high_price_indicator + low_price_indicator) / 2

    # Adjust for Dynamic Volume Impact on Enhanced Intraday Momentum
    moving_average_volume = df['volume'].rolling(window=lookback_period_short).mean()
    normalized_volume = df['volume'] / moving_average_volume
    adjusted_intraday_momentum = enhanced_intraday_momentum * normalized_volume

    # Longer-Term Momentum Adjusted for Volume
    lookback_period_long = 20
    long_term_momentum = (df['close'] - df['close'].shift(lookback_period_long)) / df['close'].shift(lookback_period_long)
    volume_ratio_long = df['volume'] / df['volume'].shift(lookback_period_long)
    final_long_term_factor = long_term_momentum * (1 + volume_ratio_long)

    # Combine Intraday Volatility, VATP, and Enhanced Intraday Momentum
    combined_intraday_factors = (df['high'] - vatp) + (vatp - df['low']) + enhanced_intraday_momentum

    # Consolidate All Factors
    consolidated_factors = (adjusted_intraday_momentum + final_long_term_factor + combined_intraday_factors) / df['open']

    # Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)

    # Significant Price Increase
    significant_price_increase = (daily_price_return > 0.01).astype(int)

    # Volume Spike Indicator
    average_past_week_volume = df['volume'].rolling(window=5).mean()
    volume_spike_indicator = (df['volume'] > average_past_week_volume).astype(int)

    # Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_price_increase

    # Apply Volume Spike Filter
    filtered_weighted_price_momentum = weighted_price_momentum * volume_spike_indicator

    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + filtered_weighted_price_momentum) * volume_spike_indicator

    # Apply an exponential moving average to smooth the factor
    alpha = 0.2
    ema_smoothed_factor = final_alpha_factor.ewm(alpha=alpha).mean()

    return ema_smoothed_factor
