```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Volume Impact on Intraday Momentum
    moving_avg_volume = df['volume'].rolling(window=n).mean()
    volume_impact_intraday = moving_avg_volume * intraday_momentum
    
    # Long-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_difference = df['volume'] - df['volume'].shift(n)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_difference
    
    # Consolidate All Factors
    total_factor = adjusted_close + volume_impact_intraday + volume_adjusted_long_term_momentum
    factor_value = total_factor / df['open']
    
    return factor_value
```
