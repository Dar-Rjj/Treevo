```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Momentum
    close_lag_10 = df['close'].shift(10)
    price_momentum = df['close'] - close_lag_10
    
    # Calculate Volume Weighted Average True Range (VWATR)
    true_range_1 = df['high'] - df['low']
    true_range_2 = (df['high'] - df['close'].shift(1)).abs()
    true_range_3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([true_range_1, true_range_2, true_range_3], axis=1).max(axis=1)
    vwatr = true_range * df['volume']

    # Adjust Price Momentum by VWATR
    adjusted_momentum = price_momentum / vwatr
    
    # Final Alpha Factor
    alpha_factor = high_low_range * adjusted_momentum
    
    return alpha_factor
```
