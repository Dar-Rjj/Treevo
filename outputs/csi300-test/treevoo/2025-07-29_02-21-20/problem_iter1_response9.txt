```python
import pandas as pd

def heuristics_v2(df, n=20):
    """
    Calculate the Volume Adjusted Momentum Indicator.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    n (int): Lookback period for momentum calculation.
    
    Returns:
    pd.Series: Series with the alpha factor values indexed by date.
    """
    # Calculate Momentum
    momentum = (df['close'].shift(-n) - df['close']) / df['close']
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'].shift(-n) / df['volume']
    
    # Combine Momentum and Volume Ratio
    adjusted_momentum = momentum * volume_ratio + momentum
    
    return adjusted_momentum
```
