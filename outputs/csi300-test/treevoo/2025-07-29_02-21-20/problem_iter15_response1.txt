```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, smoothing_window=20):
    # Calculate Close Price Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Calculate Intraday Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['VWAP'] = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Range and VWAP
    df['High_VWAP_Diff'] = df['high'] - df['VWAP']
    df['VWAP_Low_Diff'] = df['VWAP'] - df['low']
    df['Combined_Intraday_Range_VWAP'] = df['High_VWAP_Diff'] + df['VWAP_Low_Diff']
    
    # Calculate Intraday Price Changes
    df['Log_Open_Close_Change'] = np.log(df['close'] / df['open'])
    df['Log_High_Low_Change'] = np.log(df['high'] / df['low'])
    df['Aggregated_Price_Changes'] = df['Log_Open_Close_Change'] + df['Log_High_Low_Change']
    
    # Calculate Volume Adjusted Synchrony
    df['N_day_Median_Volume'] = df['volume'].rolling(window=n).median()
    df['Daily_Volume_Deviation'] = df['volume'] - df['N_day_Median_Volume']
    
    # Integrate Combined Intraday Range-VWAP and Aggregated Price Changes with Volume Adjusted Synchrony
    df['Integrated_Factor'] = (df['Combined_Intraday_Range_VWAP'] * df['Aggregated_Price_Changes']) * df['Daily_Volume_Deviation']
    
    # Multiply by Volume Ratio and Momentum
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(n+1)
    df['Factor'] = df['Integrated_Factor'] * df['Volume_Ratio'] * df['Momentum']
    
    # Apply a weighted moving average to smooth the factor
    df['Smoothed_Factor'] = df['Factor'].ewm(span=smoothing_window, adjust=False).mean()
    
    return df['Smoothed_Factor'].dropna()
```
