```python
def heuristics_v2(data):
    import pandas as pd
    
    # Calculate Close Price Adjustment by Open Price
    close_price_adjustment = data['close'] - data['open']
    
    # Intraday Momentum
    high_indicator = data['high'] - data['open']
    low_indicator = data['open'] - data['low']
    intraday_momentum = high_indicator * low_indicator
    
    # Adjust for Volume Impact
    volume_impact = data['volume'] * intraday_momentum
    
    # Consolidate Initial Factors
    initial_factors = (close_price_adjustment + volume_impact) / data['open']
    
    # Calculate Daily Price Return
    daily_return = data['close'].diff().fillna(0)
    
    # Identify Significant Price Increase
    threshold = 0.01  # Define a positive threshold
    significant_increase = (daily_return > threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    average_volume = data['volume'].rolling(window=5).mean()
    volume_spike = (data['volume'] > average_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_momentum = (daily_return * significant_increase).cumsum()
    
    # Apply Volume Spike Filter
    filtered_momentum = weighted_momentum * volume_spike
    
    # Final Alpha Factor Synthesis
    alpha_factor = (initial_factors + filtered_momentum) * volume_spike
    
    # Adjust for Open Price
    final_alpha_factor = alpha_factor / data['open']
    
    return final_alpha_factor
```
