import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day SMA of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Momentum indicator: 5-day SMA - 20-day SMA
    df['Momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Compute daily VWAP
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # 10-day SMA of VWAP
    df['SMA_VWAP_10'] = df['VWAP'].rolling(window=10).mean()
    
    # Liquidity signal: Today's VWAP - 10-day SMA of VWAP
    df['Liquidity_Signal'] = df['VWAP'] - df['SMA_VWAP_10']
    
    # Compute the average true range (ATR) over 14 days
    high_low = df['high'] - df['low']
    high_close = (df['high'] - df['close'].shift()).abs()
    low_close = (df['low'] - df['close'].shift()).abs()
    ranges = pd.concat([high_low, high_close, low_close], axis=1)
    df['True_Range'] = ranges.max(axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # A/D Line
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    df['A_D_Line'] = money_flow_volume.cumsum()
    
    # Slope of A/D Line over the past 30 days
    df['A_D_Slope'] = df['A_D_Line'].diff(periods=30)
    
    # Alpha factor: Combine all sub-factors
    df['Alpha_Factor'] = df['Momentum'] + df['Liquidity_Signal'] + df['ATR_14'] + df['A_D_Slope']
    
    return df['Alpha_Factor']
