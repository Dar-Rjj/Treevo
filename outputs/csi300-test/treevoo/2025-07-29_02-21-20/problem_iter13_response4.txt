```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Weighted Intraday Momentum
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean()
    recent_volume_7d_avg = df['volume'].rolling(window=7).mean()
    weighted_intraday_momentum_weighted = weighted_intraday_momentum * recent_volume_7d_avg
    
    # Calculate Adjusted High-Low Breakout Potential
    adjusted_high_low_breakout_potential = (high_low_range * df['volume']).ewm(span=25).mean()
    recent_volume_15d_avg = df['volume'].rolling(window=15).mean()
    adjusted_high_low_breakout_potential_weighted = adjusted_high_low_breakout_potential * recent_volume_15d_avg
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - weighted_intraday_momentum_weighted) / adjusted_high_low_breakout_potential_weighted
    
    # Calculate Volume Impact Factor
    volume_impact_factor = (df['volume'] - df['volume'].rolling(window=10).mean()) * momentum_reversal_indicator
    
    # Final Alpha Factor
    alpha_factor = momentum_reversal_indicator + adjusted_high_low_breakout_potential_weighted + volume_impact_factor
    
    return alpha_factor
```
