```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume_5 = df['volume'].rolling(window=5).mean()
    smoothed_intraday_momentum_weighted = smoothed_intraday_momentum * recent_volume_5
    
    # Calculate High-Low Breakout Potential
    high_low_breakout_potential = volume_adjusted_high_low_range.ewm(span=20).mean()
    recent_volume_10 = df['volume'].rolling(window=10).mean()
    high_low_breakout_potential_weighted = high_low_breakout_potential * recent_volume_10
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential_weighted
    
    # Calculate Trading Activity Intensity
    trading_activity_intensity = (df['volume'].rolling(window=5).mean()) / df['volume']
    
    # Final Alpha Factor
    final_alpha_factor = momentum_reversal_indicator + high_low_breakout_potential_weighted + trading_activity_intensity
    
    return final_alpha_factor
```
