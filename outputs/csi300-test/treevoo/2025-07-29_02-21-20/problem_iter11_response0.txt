```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price by Open Price
    adjusted_close = df['close'] - df['open']

    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=10).mean()
    normalized_volume = df['volume'] / volume_ma
    adjusted_intraday_momentum = intraday_momentum * normalized_volume

    # Incorporate Longer-Term Volume-Adjusted Momentum
    n = 20  # Lookback period for long-term momentum
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    final_long_term_factor = long_term_momentum * (1 + volume_ratio)

    # Consolidate All Factors
    total_factors = adjusted_close + adjusted_intraday_momentum + final_long_term_factor
    consolidated_factors = total_factors / df['open']

    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)

    # Identify Significant Price Increase
    positive_threshold = 0.01  # Define a positive threshold
    significant_increase = (daily_price_return > positive_threshold).astype(int)

    # Calculate Volume Spike Indicator
    average_weekly_volume = df['volume'].rolling(window=5).mean()
    volume_spike = (df['volume'] > average_weekly_volume).astype(int)

    # Compute Weighted Price Momentum
    weighted_momentum = (daily_price_return * significant_increase).cumsum()

    # Apply Volume Spike Filter
    filtered_momentum = weighted_momentum * volume_spike

    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + filtered_momentum) * volume_spike

    # Adjust for Open Price
    final_alpha_factor = final_alpha_factor / df['open']

    return final_alpha_factor
```
