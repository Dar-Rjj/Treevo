```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Adjust for Volume Changes
    volume_ratio = df['volume'] / df['volume'].shift(n)
    adjusted_momentum = momentum * volume_ratio + momentum
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume_weight = df['volume'].rolling(window=5).mean()
    smoothed_intraday_momentum_weighted = smoothed_intraday_momentum * recent_volume_weight
    
    # Calculate High-Low Breakout Potential
    high_low_breakout_potential = volume_adjusted_high_low_range.ewm(span=20).mean()
    recent_volume_weight_10 = df['volume'].rolling(window=10).mean()
    high_low_breakout_potential_weighted = high_low_breakout_potential * recent_volume_weight_10
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential
    
    # Calculate Trading Activity Intensity
    moving_average_volume = df['volume'].rolling(window=5).mean()
    trading_activity_intensity = moving_average_volume / df['volume']
    
    # Final Alpha Factor
    alpha_factor = (
        adjusted_momentum +
        momentum_reversal_indicator +
        high_low_breakout_potential_weighted +
        trading_activity_intensity
    )
    
    return alpha_factor
```
