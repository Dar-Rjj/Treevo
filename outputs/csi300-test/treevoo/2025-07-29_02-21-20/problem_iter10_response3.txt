```python
def heuristics_v2(data, n=5):
    close_t = data['close']
    close_t_n = close_t.shift(n)
    volume_t = data['volume']
    volume_t_n = volume_t.shift(n)
    
    momentum = (close_t - close_t_n) / close_t_n
    volume_adjusted_momentum = momentum * (volume_t / volume_t_n)
    
    return volume_adjusted_momentum
```
