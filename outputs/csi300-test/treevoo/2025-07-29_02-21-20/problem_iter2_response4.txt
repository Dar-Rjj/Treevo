import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Adjusted Close Price by Open Price
    adj_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = high_indicator * low_indicator
    
    # Adjust for Volume Impact on Intraday Momentum
    volume_impact = df['volume'] * intraday_momentum
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    adjusted_long_term_momentum = long_term_momentum * volume_ratio + long_term_momentum
    
    # Consolidate All Factors
    total_factor = adj_close + volume_impact + adjusted_long_term_momentum
    factor_values = total_factor / df['open']
    
    return factor_values
