import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Open-Close Change
    log_open_close_change = np.log(df['close']) - np.log(df['open'])
    
    # Calculate Logarithmic High-Low Change
    log_high_low_change = np.log(df['high']) - np.log(df['low'])
    
    # Aggregate Intraday Price Changes
    aggregated_price_changes = log_open_close_change + log_high_low_change
    
    # Calculate Volume Synchrony
    n_day = 20  # You can set this to any number of days for the moving average
    moving_average_volume = df['volume'].rolling(window=n_day).mean()
    daily_volume_deviation = df['volume'] - moving_average_volume
    
    # Integrate Aggregated Price Changes and Volume Synchrony
    alpha_factor = aggregated_price_changes * daily_volume_deviation
    
    return alpha_factor
