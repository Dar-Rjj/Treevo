```python
import pandas as pd

def heuristics_v2(df):
    n = 5  # Define the number of past days to consider for sum of volume and amount
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust by Close
    high_low_close_ratio = high_low_range / df['close']
    
    # Sum of past n days volume and amount
    sum_volume = df['volume'].rolling(window=n).sum()
    sum_amount = df['amount'].rolling(window=n).sum()
    
    # Current day volume and amount
    current_volume = df['volume']
    current_amount = df['amount']
    
    # Ratio of current day volume to sum of past n days volume
    volume_ratio = current_volume / sum_volume
    
    # Ratio of current day amount to sum of past n days amount
    amount_ratio = current_amount / sum_amount
    
    # Select the more significant ratio (higher ratio)
    significant_ratio = pd.DataFrame({'volume_ratio': volume_ratio, 'amount_ratio': amount_ratio}).max(axis=1)
    
    # Combine with High-Low-Close ratio
    factor = high_low_close_ratio * significant_ratio
    
    return factor
```
