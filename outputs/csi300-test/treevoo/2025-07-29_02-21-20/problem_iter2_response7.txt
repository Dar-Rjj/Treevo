```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close Price Adjustment by Open Price
    close_price_adjustment = df['close'] - df['open']

    # Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = high_price_indicator * low_price_indicator

    # Adjust for Volume Impact
    volume_impact = df['volume'] * intraday_momentum

    # Consolidate Intraday Factors
    consolidated_intraday_factors = (close_price_adjustment + volume_impact) / df['open']

    # Calculate Daily Price Return
    daily_price_return = df['close'].diff().fillna(0)

    # Identify Significant Price Increase
    significant_price_increase = (daily_price_return > 0).astype(int)

    # Calculate Volume Spike Indicator
    avg_past_week_volume = df['volume'].rolling(window=5).mean()
    volume_spike_indicator = (df['volume'] > avg_past_week_volume).astype(int)

    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_price_increase

    # Apply Volume Spike Filter
    volume_spiked_price_momentum = weighted_price_momentum * volume_spike_indicator
    combined_momentum = consolidated_intraday_factors + volume_spiked_price_momentum

    # Final Alpha Factor
    final_alpha_factor = combined_momentum

    return final_alpha_factor
```
