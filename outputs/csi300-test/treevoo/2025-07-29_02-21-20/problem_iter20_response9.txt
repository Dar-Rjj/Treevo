```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume trends.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index (date) representing the alpha factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the daily return
    df['ma_daily_return_10'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the volume
    df['ma_volume_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the volume trend
    df['volume_trend'] = df['volume'] - df['ma_volume_5']
    
    # Calculate the price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate the 10-day moving average of the price range
    df['ma_price_range_10'] = df['price_range'].rolling(window=10).mean()
    
    # Calculate the price range trend
    df['price_range_trend'] = df['price_range'] - df['ma_price_range_10']
    
    # Combine the factors to create the alpha factor
    df['alpha_factor'] = df['ma_daily_return_10'] + df['volume_trend'] + df['price_range_trend']
    
    # Return the alpha factor as a Series
    return df['alpha_factor'].dropna()
```
