import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    weights = [0.6, 0.4]  # Define Weights for High and Low
    enhanced_intraday_momentum = (weights[0] * high_price_indicator + weights[1] * low_price_indicator)
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema = df['volume'].ewm(span=10).mean()
    normalized_volume = df['volume'] / volume_ema
    intraday_momentum_adjusted = enhanced_intraday_momentum * normalized_volume
    
    # Calculate Weighted Intraday Momentum
    weighted_intraday_momentum = ((df['close'] - df['open']) * df['volume']).ewm(span=12).mean()
    recent_volume = df['volume'].rolling(window=7).mean().shift(1)
    weighted_intraday_momentum *= recent_volume
    
    # Calculate Adjusted High-Low Breakout Potential
    breakout_potential = (df['high'] - df['low']) * df['volume']
    adjusted_breakout_potential = breakout_potential.ewm(span=25).mean()
    recent_volume_15 = df['volume'].rolling(window=15).mean().shift(1)
    adjusted_breakout_potential *= recent_volume_15
    
    # Calculate Momentum Reversal Indicator
    intraday_return = df['close'] - df['open']
    momentum_reversal_indicator = (intraday_return - weighted_intraday_momentum) / adjusted_breakout_potential
    
    # Incorporate Longer-Term Volume-Adjusted Momentum
    n = 20
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    final_long_term_factor = long_term_momentum + long_term_momentum * volume_ratio
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01
    significant_price_increase = (daily_price_return > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    past_week_average_volume = df['volume'].rolling(window=7).mean().shift(1)
    volume_spike_indicator = (df['volume'] > past_week_average_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_price_increase
    accumulated_momentum = weighted_price_momentum.cumsum()
    
    # Apply Volume Spike Filter
    filtered_accumulated_momentum = accumulated_momentum * volume_spike_indicator
    
    # Final Alpha Factor Synthesis
    consolidated_initial_factors = momentum_reversal_indicator + final_long_term_factor
    final_alpha_factor = (consolidated_initial_factors + filtered_accumulated_momentum) * volume_spike_indicator
    
    # Adjust for Open Price
    final_alpha_factor /= df['open']
    
    return final_alpha_factor
