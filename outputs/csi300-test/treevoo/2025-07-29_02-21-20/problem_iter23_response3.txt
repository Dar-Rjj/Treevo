```python
import pandas as pd

def heuristics_v2(df, intraday_lookback=10, long_term_lookback=252, spike_threshold=3.0):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=intraday_lookback).mean()
    volume_ratio = df['volume'] / volume_ma
    volume_impacted_intraday_momentum = intraday_momentum * volume_ratio
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(long_term_lookback)) / df['close'].shift(long_term_lookback)
    volume_adjustment_factor = df['volume'] / df['volume'].shift(long_term_lookback)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_adjustment_factor
    
    # Add Volume Spike Filter
    volume_spike = df['volume'] / df['volume'].shift(1)
    filter_spike = (volume_spike < spike_threshold) | (pd.isna(volume_spike))
    
    # Consolidate All Factors
    total_factor = (adjusted_close + volume_impacted_intraday_momentum + volume_adjusted_long_term_momentum) / df['open']
    total_factor[~filter_spike] = 0  # Apply the volume spike filter

    return total_factor
```
