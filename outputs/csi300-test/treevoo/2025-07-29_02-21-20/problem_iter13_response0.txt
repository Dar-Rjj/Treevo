```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    
    # Volume Adjusted Factors
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume_5d = df['volume'].rolling(window=5).mean()
    smoothed_intraday_momentum_weighted = smoothed_intraday_momentum * recent_volume_5d
    
    # High-Low Breakout Potential
    high_low_breakout_potential = volume_adjusted_high_low_range.ewm(span=20).mean()
    recent_volume_10d = df['volume'].rolling(window=10).mean()
    high_low_breakout_potential_weighted = high_low_breakout_potential * recent_volume_10d
    
    # Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential
    
    # Trading Activity Intensity
    moving_average_volume_5d = df['volume'].rolling(window=5).mean()
    trading_activity_intensity = moving_average_volume_5d / df['volume']
    
    # Consolidate Factors
    consolidated_factors = (smoothed_intraday_momentum_weighted + 
                            high_low_breakout_potential_weighted + 
                            trading_activity_intensity)
    factor_values = consolidated_factors / df['open']
    
    return factor_values
```
