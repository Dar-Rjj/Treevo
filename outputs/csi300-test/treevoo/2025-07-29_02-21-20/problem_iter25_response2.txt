```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    weights = [0.6, 0.4]  # Weights for High and Low
    enhanced_intraday_momentum = (high_indicator * weights[0]) + (low_indicator * weights[1])
    
    # Volume-Impacted Intraday Momentum
    volume_ema = df['volume'].ewm(span=10).mean()
    normalized_volume = df['volume'] / volume_ema
    volume_impacted_intraday = enhanced_intraday_momentum * normalized_volume
    
    # Weighted Intraday Momentum
    intraday_return_volume = (df['close'] - df['open']) * df['volume']
    weighted_intraday_momentum = intraday_return_volume.ewm(span=12).mean() * (df['volume'] / df['volume'].rolling(window=7).mean())
    
    # Adjusted High-Low Breakout Potential
    high_low_range_volume = (df['high'] - df['low']) * df['volume']
    adjusted_high_low_breakout = high_low_range_volume.ewm(span=25).mean() * (df['volume'] / df['volume'].rolling(window=15).mean())
    
    # Momentum Reversal
    momentum_reversal = (df['close'] - df['open']) - weighted_intraday_momentum
    momentum_reversal = momentum_reversal / adjusted_high_low_breakout
    
    # Long-Term Momentum Adjusted for Volume
    n = 30  # Lookback period for long-term momentum
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    long_term_momentum_adjusted = long_term_momentum * volume_ratio
    
    # Final Alpha Factor
    total_factors = adjusted_close + volume_impacted_intraday + weighted_intraday_momentum + adjusted_high_low_breakout + long_term_momentum_adjusted
    final_alpha_factor = total_factors / df['open']
    
    return final_alpha_factor
```
