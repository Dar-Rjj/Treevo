def heuristics_v2(data, n=10):
    # Calculate Momentum
    momentum = (data['close'] - data['close'].shift(n)) / data['close'].shift(n)
    
    # Calculate Volume Ratio
    volume_ratio = data['volume'] / data['volume'].shift(n)
    
    # Calculate Volume Impact
    volume_impact = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)
    
    # Combine Momentum, Volume Ratio, and Volume Impact
    adjusted_momentum = (momentum * volume_ratio) * (1 + volume_impact) + momentum
    
    return adjusted_momentum
