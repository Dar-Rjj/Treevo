```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close Price Adjustment by Open Price
    close_price_adjustment = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Adjust for Volume Impact on Intraday Momentum
    volume_impact_intraday = df['volume'] * intraday_momentum
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    n = 20  # Define lookback period
    long_term_momentum = (df['close'].pct_change(n)).fillna(0)
    volume_ratio = df['volume'] / df['volume'].shift(n).fillna(1)
    long_term_factor = long_term_momentum * np.cbrt(volume_ratio)
    
    # Consolidate All Factors
    consolidated_factors = (close_price_adjustment + volume_impact_intraday + long_term_factor) / df['open']
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01  # Define a positive threshold
    significant_price_increase = (daily_price_return > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    past_week_volume_median = df['volume'].rolling(window=5).median()
    volume_spike_indicator = (df['volume'] > past_week_volume_median).astype(int)
    
    # Compute Weighted Price Momentum
    accumulated_momentum = (daily_price_return * significant_price_increase).cumsum()
    
    # Apply Volume Spike Filter
    filtered_momentum = accumulated_momentum * volume_spike_indicator
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + filtered_momentum) * volume_spike_indicator
    
    # Add Price Volatility Factor
    price_volatility = df['close'].pct_change().rolling(window=n).std().fillna(0)
    scaled_volatility = price_volatility / np.cbrt(df['volume'])
    final_alpha_factor += scaled_volatility
    
    return final_alpha_factor
```
