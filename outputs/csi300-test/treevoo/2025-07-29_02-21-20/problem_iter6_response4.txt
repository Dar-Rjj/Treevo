```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Adjusted Close Price by Open Price
    df['adjusted_close'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    df['intraday_momentum'] = high_price_indicator + low_price_indicator
    
    # Adjust for Volume Impact on Intraday Momentum
    df['volume_adjusted_intraday_momentum'] = df['volume'] * df['intraday_momentum']
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    df['long_term_momentum'] = (df['close'].shift(-n) - df['close']) / df['close'].shift(-n)
    df['volume_ratio'] = df['volume'].shift(-n) / df['volume']
    df['volume_adjusted_long_term_momentum'] = df['long_term_momentum'] * df['volume_ratio']
    
    # Consolidate All Factors
    df['total_factor'] = df['adjusted_close'] + df['volume_adjusted_intraday_momentum'] + df['volume_adjusted_long_term_momentum']
    df['final_factor'] = df['total_factor'] / df['open']
    
    return df['final_factor']
```
