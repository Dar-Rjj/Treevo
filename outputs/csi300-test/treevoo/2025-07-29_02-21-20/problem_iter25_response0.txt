```python
import pandas as pd
import numpy as np

def heuristics_v2(data, lookback_period=20):
    # Calculate Adjusted Close Price
    data['adjusted_close'] = data['close'] - data['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = data['high'] - data['open']
    low_price_indicator = data['open'] - data['low']
    intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Volume Impact on Intraday Momentum
    alpha = 0.1  # Smoothing Factor
    smoothed_volume = data['volume'].ewm(alpha=alpha).mean()
    volume_impact_intraday_momentum = smoothed_volume * intraday_momentum
    
    # Long-Term Momentum Adjusted for Volume
    long_term_momentum = (data['close'] - data['close'].shift(lookback_period)) / data['close'].shift(lookback_period)
    volume_ratio = data['volume'] / data['volume'].shift(lookback_period)
    volume_adjusted_long_term_momentum = long_term_momentum * np.cbrt(volume_ratio)
    
    # Consolidate All Factors
    total_momentum = data['adjusted_close'] + volume_impact_intraday_momentum + volume_adjusted_long_term_momentum
    consolidated_factors = total_momentum / data['open']
    
    # Identify Significant Price Increase
    daily_price_return = (data['close'] - data['open']) / data['open']
    significant_price_increase = (daily_price_return > 0.01).astype(int)
    
    # Calculate Volume Spike Indicator
    past_week_volume_median = data['volume'].rolling(window=5).median()
    volume_spike_indicator = (data['volume'] > past_week_volume_median).astype(int)
    
    # Apply Volume Spike Filter
    volume_spike_filtered_momentum = significant_price_increase * volume_spike_indicator
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + volume_spike_filtered_momentum) * volume_spike_indicator
    
    return final_alpha_factor
```
