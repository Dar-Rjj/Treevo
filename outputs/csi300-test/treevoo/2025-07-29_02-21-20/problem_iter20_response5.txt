```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, smoothing_factor=0.2, positive_threshold=0.01):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Volume Impact on Intraday Momentum
    smoothed_volume = df['volume'].ewm(alpha=smoothing_factor).mean()
    volume_impacted_intraday_momentum = smoothed_volume * intraday_momentum
    
    # Long-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term_momentum = long_term_momentum * (volume_ratio ** (1/3))
    
    # Consolidate All Factors
    total_momentum = adjusted_close + volume_impacted_intraday_momentum + volume_adjusted_long_term_momentum
    consolidated_factors = total_momentum / df['open']
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    significant_increase = (daily_price_return > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    volume_spike = (df['volume'] > df['volume'].rolling(window=7).median()).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_increase
    
    # Apply Volume Spike Filter
    filtered_momentum = weighted_price_momentum * volume_spike
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + filtered_momentum) * volume_spike
    final_alpha_factor /= df['open']
    
    return final_alpha_factor
```
