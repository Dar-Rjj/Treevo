```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low for the day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the total value traded during the day
    total_value = df['close'] * df['volume']
    
    # Generate a signal based on whether the closing price is higher than the average
    # and if the total value traded is above the mean, could indicate strong buying pressure
    factor = (df['close'] > avg_price).astype(int) * (total_value > total_value.mean()).astype(int)
    
    return pd.Series(factor, index=df.index, name='factor')
```
