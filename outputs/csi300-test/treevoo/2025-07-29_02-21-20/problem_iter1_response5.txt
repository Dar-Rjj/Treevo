import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).sum() / df['volume'].sum()
    
    # Combine Intraday Range and VWAP
    high_vwap_diff = df['high'] - vwap
    vwap_low_diff = vwap - df['low']
    combined_factor = high_vwap_diff + vwap_low_diff
    
    # Apply a moving average to smooth the factor
    smoothed_factor = combined_factor.rolling(window=n).mean()
    
    return smoothed_factor
