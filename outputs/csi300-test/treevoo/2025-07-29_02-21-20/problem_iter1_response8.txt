```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate 20 Day Average of Daily Range
    df['avg_daily_range_20'] = df['daily_range'].rolling(window=20).mean()
    
    # Calculate 20 Day Moving Average of Volume
    df['vol_20_ma'] = df['volume'].rolling(window=20).mean()
    
    # Calculate Volume Relative to its Moving Average
    df['vol_rel'] = df['volume'] / df['vol_20_ma']
    
    # Compute Volatility Adjustment Factor
    df['vol_adj_factor'] = (df['daily_range'] / df['avg_daily_range_20']) * df['vol_rel']
    
    return df['vol_adj_factor'].dropna()
```
