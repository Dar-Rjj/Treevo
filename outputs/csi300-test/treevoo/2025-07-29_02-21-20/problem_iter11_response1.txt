```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate the simple momentum first
    momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Calculate the volume ratio
    volume_ratio = df['volume'] / df['volume'].shift(n)
    
    # Multiply momentum by the volume ratio for the final factor
    final_factor = momentum * volume_ratio
    
    return final_factor
```
