import pandas as pd

def heuristics_v2(df):
    # Calculate the rolling mean of the closing price with a 5-day window
    df['close_5d_mean'] = df['close'].rolling(window=5).mean()
    
    # Calculate the difference between the current close and the 5-day mean of the close
    df['close_diff_5d_mean'] = df['close'] - df['close_5d_mean']
    
    # Calculate the rolling standard deviation of the amount with a 5-day window
    df['amount_5d_std'] = df['amount'].rolling(window=5).std()
    
    # Calculate the alpha factor as the ratio of the close difference to the amount's standard deviation
    alpha_factor = df['close_diff_5d_mean'] / df['amount_5d_std']
    
    return alpha_factor
