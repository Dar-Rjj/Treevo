```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    df['adjusted_close'] = df['close'] - df['open']

    # Enhanced Intraday Momentum
    df['high_low_spread'] = df['high'] - df['low']
    df['avg_high_low_spread'] = df['high_low_spread'].rolling(window=10).mean()
    
    # Define Weights for High, Low, and High-Low Spread
    high_weight, low_weight, spread_weight = 0.4, 0.3, 0.3
    df['weighted_avg_indicators'] = (df['high'] * high_weight + df['low'] * low_weight + df['high_low_spread'] * spread_weight)

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_lookback = 10
    df['volume_moving_avg'] = df['volume'].rolling(window=volume_lookback).mean()
    df['dynamic_volume_impact'] = (df['volume'] / df['volume_moving_avg']) * df['weighted_avg_indicators']

    # Calculate Weighted Intraday Momentum
    df['intraday_return_vol'] = (df['close'] - df['open']) * df['volume']
    df['weighted_intraday_momentum'] = df['intraday_return_vol'].ewm(span=12).mean() * df['volume'].rolling(window=7).mean()

    # Incorporate Longer-Term Momentum Adjusted for Volume
    n = 60
    df['long_term_momentum'] = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(n)
    df['volume_adjusted_long_term_momentum'] = df['long_term_momentum'] * np.cbrt(df['volume_ratio'])

    # Price-Volume Synchronization
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['price_volume_correlation'] = df['price_change'].rolling(window=20).corr(df['volume_change'])

    # Daily Price Return
    df['daily_price_return'] = df['close'] - df['close'].shift(1)

    # Significant Price Increase
    positive_threshold = 0.01
    df['significant_price_increase'] = (df['daily_price_return'] > positive_threshold).astype(int)

    # Volume Spike Indicator
    df['volume_spike'] = (df['volume'] > df['volume'].rolling(window=7).median()).astype(int)

    # Weighted Price Momentum
    df['weighted_price_momentum'] = df['daily_price_return'] * df['significant_price_increase']

    # Apply Volume Spike Filter
    df['volume_spike_filtered_momentum'] = df['weighted_price_momentum'] * df['volume_spike']

    # Consolidate All Factors
    df['consolidated_factors'] = (df['adjusted_close'] + df['dynamic_volume_impact'] + 
                                  df['weighted_intraday_momentum'] + df['volume_adjusted_long_term_momentum'] + 
                                  df['price_volume_correlation'])
    df['consolidated_factors'] /= df['open']

    # Final Alpha Factor Synthesis
    df['alpha_factor'] = (df['consolidated_factors'] + df['volume_spike_filtered_momentum']) * df['volume_spike']

    # Adjust for Open Price
    df['alpha_factor'] /= df['open']

    return df['alpha_factor']
```
