```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Intraday Logarithmic Changes
    log_open_close_change = np.log(df['close']) - np.log(df['open'])
    log_high_low_change = np.log(df['high']) - np.log(df['low'])
    aggregated_log_changes = log_open_close_change + log_high_low_change
    
    # Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = high_price_indicator * low_price_indicator
    
    # Volume Adjusted Intraday Factors
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    volume_adjusted_log_changes = aggregated_log_changes * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume = df['volume'].rolling(window=5).mean()
    weighted_smoothed_momentum = smoothed_intraday_momentum * recent_volume
    
    # High-Low Breakout Potential
    breakout_potential = volume_adjusted_high_low_range.ewm(span=20).mean()
    weighted_breakout_potential = breakout_potential * recent_volume.rolling(window=10).mean()
    
    # Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_smoothed_momentum) / weighted_breakout_potential
    
    # Trading Activity Intensity
    activity_intensity = df['volume'].rolling(window=5).mean() / df['volume']
    
    # Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    significant_increase = (daily_price_return > 0).astype(int)
    
    # Volume Spike Indicator
    volume_spike = (df['volume'] > df['volume'].rolling(window=7).mean()).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_momentum = (daily_price_return * significant_increase).cumsum()
    
    # Apply Volume Spike Filter
    spike_filtered_momentum = weighted_momentum * volume_spike
    
    # Consolidate Initial Factors
    consolidated_factors = (adjusted_close + weighted_smoothed_momentum) / df['open']
    
    # Final Alpha Factor Synthesis
    final_factor = (consolidated_factors + spike_filtered_momentum + activity_intensity) * volume_spike
    final_factor /= df['open']
    
    return final_factor
```
