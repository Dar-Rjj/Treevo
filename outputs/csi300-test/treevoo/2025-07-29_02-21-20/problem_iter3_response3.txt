```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close Price Adjustment by Open Price
    close_price_adjustment = df['close'] - df['open']
    
    # Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = high_indicator * low_indicator
    
    # Adjust for Volume and Amount Impact
    volume_amount_impact = (df['volume'] + df['amount']) * intraday_momentum
    
    # Consolidate Factors
    factor_value = (close_price_adjustment + volume_amount_impact) / df['open']
    
    return factor_value
```
