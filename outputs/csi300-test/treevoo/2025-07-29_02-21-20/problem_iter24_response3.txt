```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Generate buy/sell signals based on SMA crossover
    df['SMA_Crossover'] = 0
    df.loc[df['SMA_5'] > df['SMA_20'], 'SMA_Crossover'] = 1
    df.loc[df['SMA_5'] < df['SMA_20'], 'SMA_Crossover'] = -1
    
    # Calculate Price Rate of Change (ROC) over 10 days
    df['ROC'] = df['close'].pct_change(periods=10)
    
    # Generate buy/sell signals based on ROC
    df['ROC_Signal'] = 0
    df.loc[(df['ROC'] > 0) & (df['ROC'].diff() > 0), 'ROC_Signal'] = 1
    df.loc[(df['ROC'] < 0) & (df['ROC'].diff() < 0), 'ROC_Signal'] = -1
    
    # Calculate On-Balance Volume (OBV)
    df['OBV'] = ((df['close'] > df['close'].shift(1)) * df['volume']).cumsum() - ((df['close'] < df['close'].shift(1)) * df['volume']).cumsum()
    
    # Generate buy/sell signals based on OBV
    df['OBV_Signal'] = 0
    df.loc[df['OBV'].diff() > 0, 'OBV_Signal'] = 1
    df.loc[df['OBV'].diff() < 0, 'OBV_Signal'] = -1
    
    # Calculate average volume over 30 days
    df['Avg_Volume_30'] = df['volume'].rolling(window=30).mean()
    
    # Define a threshold for volume spike (2x the average volume)
    df['Volume_Spike'] = 0
    df.loc[df['volume'] > 2 * df['Avg_Volume_30'], 'Volume_Spike'] = 1
    
    # Generate buy/sell signals based on volume spike
    df['Volume_Spike_Signal'] = 0
    df.loc[(df['Volume_Spike'] == 1) & (df['close'] > df['close'].shift(1)), 'Volume_Spike_Signal'] = 1
    df.loc[(df['Volume_Spike'] == 1) & (df['close'] < df['close'].shift(1)), 'Volume_Spike_Signal'] = -1
    
    # Calculate Price-Volume Trend (PVT)
    df['PVT'] = (df['close'].pct_change() * df['volume']).cumsum()
    
    # Generate buy/sell signals based on PVT
    df['PVT_Signal'] = 0
    df.loc[df['PVT'].diff() > 0, 'PVT_Signal'] = 1
    df.loc[df['PVT'].diff() < 0, 'PVT_Signal'] = -1
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['Cumulative_PriceVol'] = (df['close'] * df['volume']).cumsum()
    df['Cumulative_Vol'] = df['volume'].cumsum()
    df['VWAP'] = df['Cumulative_PriceVol'] / df['Cumulative_Vol']
    
    # Generate buy/sell signals based on VWAP
    df['VWAP_Signal'] = 0
    df.loc[df['close'] > df['VWAP'], 'VWAP_Signal'] = 1
    df.loc[df['close'] < df['VWAP'], 'VWAP_Signal'] = -1
    
    # Combine all signals into a single alpha factor
    df['Alpha_Factor'] = (df['SMA_Crossover'] + df['ROC_Signal'] + df['OBV_Signal'] + 
                          df['Volume_Spike_Signal'] + df['PVT_Signal'] + df['VWAP_Signal'])
    
    return df['Alpha_Factor']
```
