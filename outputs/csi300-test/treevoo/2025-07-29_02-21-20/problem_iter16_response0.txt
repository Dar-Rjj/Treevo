```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Weighted Intraday Momentum
    ema_intraday_vol = df['volume'].rolling(window=7).mean() * intraday_return.ewm(span=12).mean()
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean()
    
    # Calculate Adjusted High-Low Breakout Potential
    ema_high_low_vol = df['volume'].rolling(window=15).mean() * high_low_range.ewm(span=25).mean()
    adjusted_high_low_breakout = (high_low_range * df['volume']).ewm(span=25).mean()
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - weighted_intraday_momentum) / adjusted_high_low_breakout
    
    # Calculate Volume Impact Factor
    volume_impact_factor = (df['volume'] - df['volume'].rolling(window=10).mean()) * momentum_reversal_indicator
    
    # Final Alpha Factor
    final_alpha_factor = momentum_reversal_indicator + adjusted_high_low_breakout + volume_impact_factor
    
    return final_alpha_factor
```
