```python
def heuristics_v2(df, n=20):
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Calculate Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(n)
    
    # Combine Momentum and Volume Ratio
    alpha_factor = momentum * volume_ratio
    
    return alpha_factor
```
