import pandas as pd

def heuristics_v2(df):
    # Adjusted Price Momentum
    df['adjusted_close'] = df['close'] - df['open']
    df['high_indicator'] = df['high'] - df['open']
    df['low_indicator'] = df['open'] - df['low']
    df['intraday_momentum'] = (df['high_indicator'] + df['low_indicator']) / 2
    df['daily_change_close'] = df['close'].diff()
    df['daily_change_high'] = df['high'].diff()
    df['daily_change_low'] = df['low'].diff()
    df['3d_close_trend'] = df['daily_change_close'].rolling(window=3).mean()
    df['7d_close_trend'] = df['daily_change_close'].rolling(window=7).mean()
    df['21d_close_trend'] = df['daily_change_close'].rolling(window=21).mean()

    # Volume-Adjusted Momentum and Liquidity
    df['volume_ma'] = df['volume'].rolling(window=21).mean()
    df['volume_impact'] = df['volume'] / df['volume_ma']
    df['long_term_momentum'] = (df['close'] - df['close'].shift(21)) / df['close'].shift(21)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(21)
    df['volume_adjusted_momentum'] = df['long_term_momentum'] * df['volume_ratio']
    df['daily_change_volume'] = df['volume'].diff()
    df['3d_volume_trend'] = df['daily_change_volume'].rolling(window=3).mean()
    df['7d_volume_trend'] = df['daily_change_volume'].rolling(window=7).mean()
    df['21d_volume_trend'] = df['daily_change_volume'].rolling(window=21).mean()
    df['avg_amount_21d'] = df['amount'].rolling(window=21).mean()
    df['total_volume_21d'] = df['volume'].rolling(window=21).sum()

    # Gaps and Breaks
    df['up_gap'] = (df['open'] > df['close'].shift(1)).astype(int)
    df['down_gap'] = (df['open'] < df['close'].shift(1)).astype(int)
    df['gap_size_up'] = (df['open'] - df['close'].shift(1))
    df['gap_size_down'] = (df['open'] - df['close'].shift(1))
    df['upper_breakout'] = (df['high'] > df['high'].rolling(window=21).max()).astype(int)
    df['lower_breakout'] = (df['low'] < df['low'].rolling(window=21).min()).astype(int)
    df['breakout_upper_size'] = df['high'] - df['high'].rolling(window=21).max()
    df['breakout_lower_size'] = df['low'].rolling(window=21).min() - df['low']

    # Intraday Patterns
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_momentum_strength'] = (df['close'] - df['open']) / df['intraday_range']
    df['intraday_volatility'] = df[['high', 'low']].rolling(window=21).std().mean(axis=1)

    # Final alpha factor
    df['alpha_factor'] = (
        df['adjusted_close'] + 
        0.5 * df['intraday_momentum'] + 
        0.2 * df['3d_close_trend'] + 
        0.1 * df['7d_close_trend'] + 
        0.05 * df['21d_close_trend'] + 
        0.3 * df['volume_impact'] + 
        0.4 * df['volume_adjusted_momentum'] + 
        0.1 * df['3d_volume_trend'] + 
        0.05 * df['7d_volume_trend'] + 
        0.01 * df['21d_volume_trend'] + 
        0.05 * df['avg_amount_21d'] + 
        0.01 * df['total_volume_21d'] +
