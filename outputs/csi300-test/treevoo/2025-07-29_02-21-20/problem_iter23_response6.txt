```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Close Price
    adj_close = df['close'] - df['open']

    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    vol_lookback = 20  # Define Lookback Period
    vol_ma = df['volume'].rolling(window=vol_lookback).mean()
    normalized_volume = df['volume'] / vol_ma
    volume_impacted_intraday = intraday_momentum * normalized_volume

    # Incorporate Longer-Term Momentum Adjusted for Volume
    long_lookback = 60  # Define n as a Lookback Period
    long_term_momentum = (df['close'] - df['close'].shift(long_lookback)) / df['close'].shift(long_lookback)
    volume_ratio = df['volume'] / df['volume'].shift(long_lookback)
    volume_adjusted_long_term = long_term_momentum * (volume_ratio ** (1/3))

    # Consolidate All Factors
    total_factors = adj_close + volume_impacted_intraday + volume_adjusted_long_term
    consolidated_factors = total_factors / df['open']

    # Daily Price Return
    daily_return = df['close'] - df['close'].shift(1)

    # Significant Price Increase
    significant_increase = (daily_return > 0).astype(int)

    # Volume Spike Indicator
    median_vol_week = df['volume'].rolling(window=5).median()
    volume_spike = (df['volume'] > median_vol_week).astype(int)

    # Weighted Price Momentum
    weighted_momentum = daily_return * significant_increase

    # Apply Volume Spike Filter
    filtered_momentum = weighted_momentum * volume_spike

    # Final Alpha Factor Synthesis
    final_factor = consolidated_factors + filtered_momentum
    final_alpha_factor = final_factor * volume_spike

    # Adjust for Open Price
    final_alpha_factor = final_alpha_factor / df['open']

    return final_alpha_factor
```
