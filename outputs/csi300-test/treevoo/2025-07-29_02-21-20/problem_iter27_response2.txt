```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']

    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    
    # Define Weights for High and Low Indicators
    weight_high = 0.6
    weight_low = 0.4
    
    # Compute Weighted Average
    intraday_momentum = (weight_high * high_indicator) + (weight_low * low_indicator)

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    lookback_period = 10
    ema_volume = df['volume'].ewm(span=lookback_period, adjust=False).mean()
    volume_ratio = df['volume'] / ema_volume
    volume_impacted_intraday_momentum = intraday_momentum * volume_ratio

    # Incorporate Volume-Adjusted Long-Term Momentum
    n = 20
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_adjustment_factor = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_adjustment_factor

    # Add Volume Spike Filter
    spike_threshold = 2.0
    volume_spike = df['volume'] / df['volume'].shift(1)
    no_spike_filter = (volume_spike < spike_threshold).astype(int)
    volume_impacted_intraday_momentum_filtered = volume_impacted_intraday_momentum * no_spike_filter
    volume_adjusted_long_term_momentum_filtered = volume_adjusted_long_term_momentum * no_spike_filter

    # Consolidate All Factors
    total_factor = (adjusted_close + volume_impacted_intraday_momentum_filtered + volume_adjusted_long_term_momentum_filtered) / df['open']
    
    return total_factor
```
