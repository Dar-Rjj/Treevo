```python
def heuristics_v2(df, n=10):
    close = df['close']
    volume = df['volume']
    
    # Calculate Momentum
    momentum = (close - close.shift(n)) / close.shift(n)
    
    # Calculate Volume Ratio
    volume_ratio = volume / volume.shift(n)
    
    # Volume Adjusted Momentum Indicator
    vami = momentum * volume_ratio
    
    return vami
```
