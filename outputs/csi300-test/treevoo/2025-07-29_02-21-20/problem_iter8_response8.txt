import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust by Close
    high_low_range_adj = high_low_range / df['close']
    
    # Volume and Amount Adjustment
    n_days = 5
    sum_volume = df['volume'].rolling(window=n_days).sum()
    sum_amount = df['amount'].rolling(window=n_days).sum()
    
    current_volume_ratio = df['volume'] / sum_volume
    current_amount_ratio = df['amount'] / sum_amount
    
    # Combine Ratios (Weighted average)
    weighted_ratio = 0.6 * current_volume_ratio + 0.4 * current_amount_ratio
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    enhanced_intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    vol_lookback = 5
    vol_moving_avg = df['volume'].rolling(window=vol_lookback).mean()
    normalized_vol = df['volume'] / vol_moving_avg
    dynamic_vol_impact = enhanced_intraday_momentum * normalized_vol
    
    # Calculate Long-Term Momentum
    long_term_n_days = 20
    long_term_momentum = (df['close'] - df['close'].shift(long_term_n_days)) / df['close'].shift(long_term_n_days)
    
    # Adjust for Volume Changes
    volume_ratio = df['volume'] / df['volume'].shift(long_term_n_days)
    adjusted_long_term_momentum = long_term_momentum * volume_ratio
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Range and VWAP
    range_vwap_high = df['high'] - vwap
    range_vwap_low = vwap - df['low']
    intraday_range_vwap = range_vwap_high + range_vwap_low
    
    # Apply a moving average to smooth the factor
    smoothing_window = 3
    smoothed_factor = intraday_range_vwap.rolling(window=smoothing_window).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.3 * dynamic_vol_impact +
        0.3 * adjusted_long_term_momentum +
        0.4 * smoothed_factor
    )
    
    return final_alpha_factor
