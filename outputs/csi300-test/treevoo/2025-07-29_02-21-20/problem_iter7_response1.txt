```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Intraday Return
    vol_adj_intraday_return = intraday_return * df['volume']
    
    # Calculate Volume Adjusted High-Low Range
    vol_adj_high_low_range = high_low_range * df['volume']
    
    # Calculate Smoothed Intraday Momentum
    smoothed_intraday_momentum = vol_adj_intraday_return.rolling(window=10).mean()
    
    # Calculate High-Low Breakout Potential
    high_low_breakout_potential = vol_adj_high_low_range.rolling(window=20).mean()
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate VWAP-based Intraday Range
    vwap_based_intraday_range = (df['high'] - vwap) - (vwap - df['low'])
    
    # Combine Indicators
    combined_indicators = (smoothed_intraday_momentum + 
                           high_low_breakout_potential + 
                           vwap_based_intraday_range)
    
    # Final Alpha Factor
    final_alpha_factor = combined_indicators.rolling(window=n).mean()
    
    return final_alpha_factor
```
