```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Volume Momentum
    df['Volume_Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20) * (df['volume'] / df['volume'].shift(20))
    
    # Calculate Intraday Return
    df['Intraday_Return'] = df['close'] - df['open']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Weighted Intraday Momentum
    weighted_intraday_momentum = (df['Intraday_Return'] * df['volume']).ewm(span=12).mean()
    recent_volume = df['volume'].rolling(window=7).mean()
    df['Weighted_Intraday_Momentum'] = weighted_intraday_momentum * recent_volume
    
    # Calculate Adjusted High-Low Breakout Potential
    adjusted_breakout_potential = (df['High_Low_Range'] * df['volume']).ewm(span=25).mean()
    recent_volume_15 = df['volume'].rolling(window=15).mean()
    df['Adjusted_Breakout_Potential'] = adjusted_breakout_potential * recent_volume_15
    
    # Calculate Momentum Reversal Indicator
    df['Momentum_Reversal_Indicator'] = (df['Intraday_Return'] - df['Weighted_Intraday_Momentum']) / df['Adjusted_Breakout_Potential']
    
    # Calculate Trading Activity Intensity
    moving_average_volume_7 = df['volume'].rolling(window=7).mean()
    df['Trading_Activity_Intensity'] = moving_average_volume_7 / df['volume']
    
    # Calculate Volume Impact Factor
    volume_diff = df['volume'] - df['volume'].rolling(window=10).mean()
    df['Volume_Impact_Factor'] = volume_diff * df['Momentum_Reversal_Indicator']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Volume_Momentum'] +
        df['Momentum_Reversal_Indicator'] +
        df['Adjusted_Breakout_Potential'] +
        df['Trading_Activity_Intensity'] +
        df['Volume_Impact_Factor']
    )
    
    return df['Alpha_Factor']
```
