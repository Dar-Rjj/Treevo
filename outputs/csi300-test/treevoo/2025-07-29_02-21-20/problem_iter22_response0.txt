```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Volume Impact on Intraday Momentum
    volume_ratio = df['volume'] / df['volume'].rolling(window=n).mean()
    volume_impact_intraday_momentum = intraday_momentum * volume_ratio
    
    # Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Adjust Long-Term Momentum by Volume
    volume_adjustment_factor = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_adjustment_factor
    
    # Consolidate Factors
    total_factors = adjusted_close + volume_impact_intraday_momentum + volume_adjusted_long_term_momentum
    factor_values = total_factors / df['open']
    
    return factor_values
```
