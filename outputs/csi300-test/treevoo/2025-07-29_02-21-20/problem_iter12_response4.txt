```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day SMA of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Momentum indicator: 5-day SMA - 20-day SMA
    df['Momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Compute daily VWAP
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']) / df['volume']
    
    # 10-day SMA of VWAP
    df['VWAP_SMA_10'] = df['VWAP'].rolling(window=10).mean()
    
    # Liquidity signal: Today's VWAP - 10-day SMA of VWAP
    df['Liquidity'] = df['VWAP'] - df['VWAP_SMA_10']
    
    # Compute the average true range (ATR) over 14 days
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # ATR-based Volatility Indicator
    df['Volatility'] = df['close'] - df['close'].shift(1)
    df['Volatility'] = df['Volatility'].abs() / df['ATR_14']
    
    # Calculate the A/D Line
    df['A_D_Line'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['A_D_Line'] = df['A_D_Line'].cumsum()
    
    # Slope of the A/D Line over the past 30 days
    df['A_D_Slope'] = df['A_D_Line'].diff(periods=30)
    
    # Combine all factors into a single alpha factor
    df['Alpha_Factor'] = df['Momentum'] + df['Liquidity'] + df['Volatility'] + df['A_D_Slope']
    
    return df['Alpha_Factor']
```
