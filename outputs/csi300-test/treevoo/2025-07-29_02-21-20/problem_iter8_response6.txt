```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price by Open Price
    adjusted_close = df['close'] - df['open']

    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    avg_intraday_momentum = (high_indicator + low_indicator) / 2

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=10).mean()
    normalized_volume = df['volume'] / volume_ma
    volume_adjusted_momentum = avg_intraday_momentum * normalized_volume

    # Incorporate Longer-Term Momentum Adjusted for Volume
    n = 30
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term_momentum = (long_term_momentum * volume_ratio) + long_term_momentum

    # Consolidate All Factors
    total_factor = (adjusted_close + volume_adjusted_momentum + volume_adjusted_long_term_momentum) / df['open']

    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)

    # Identify Significant Price Increase
    positive_threshold = 0.01
    significant_price_increase = (daily_price_return > positive_threshold).astype(int)

    # Calculate Volume Spike Indicator
    average_past_week_volume = df['volume'].rolling(window=5).mean()
    volume_spike_indicator = (df['volume'] > average_past_week_volume).astype(int)

    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_price_increase

    # Apply Volume Spike Filter
    volume_spike_filtered_momentum = weighted_price_momentum * volume_spike_indicator

    # Final Alpha Factor Synthesis
    final_alpha_factor = (total_factor + volume_spike_filtered_momentum) * volume_spike_indicator

    return final_alpha_factor
```
