```python
import pandas as pd

def heuristics_v2(df, n=20, alpha=0.1):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Volume Impact on Intraday Momentum
    smoothed_volume = df['volume'].ewm(alpha=alpha).mean()
    volume_impact_intraday = smoothed_volume * intraday_momentum
    
    # Long-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term = long_term_momentum * volume_ratio
    
    # Incorporate Amount Impact
    average_transaction_size = df['amount'] / df['volume']
    
    # Combine all factors and adjust by Average Transaction Size
    total_factor = (adjusted_close + volume_impact_intraday + volume_adjusted_long_term) / average_transaction_size
    
    return total_factor
```
