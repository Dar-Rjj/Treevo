```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Adjust by Close
    range_over_close = high_low_range / df['close']
    
    # Volume Adjustment
    # Sum of past n days volume
    sum_volume = df['volume'].rolling(window=n).sum()
    
    # Current day volume
    current_volume = df['volume']
    
    # Ratio of current day volume to sum
    volume_ratio = current_volume / sum_volume
    
    # Final Alpha Factor
    alpha_factor = range_over_close * volume_ratio
    
    return alpha_factor
```
