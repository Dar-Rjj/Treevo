```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_low_indicator_high = df['high'] - df['open']
    high_low_indicator_low = df['open'] - df['low']
    
    # Define Weights for High and Low
    weight_high = 0.6
    weight_low = 0.4
    
    # Compute Weighted Average
    weighted_intraday = (weight_high * high_low_indicator_high) + (weight_low * high_low_indicator_low)
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema = df['volume'].ewm(span=10).mean()
    normalized_volume = df['volume'] / volume_ema
    dynamic_intraday = weighted_intraday * normalized_volume
    
    # Calculate Weighted Intraday Momentum
    intraday_momentum_ema = (dynamic_intraday * df['volume']).ewm(span=12).mean()
    recent_volume_weight = df['volume'].rolling(window=7).mean()
    weighted_intraday_momentum = intraday_momentum_ema * recent_volume_weight
    
    # Calculate Adjusted High-Low Breakout Potential
    high_low_range = df['high'] - df['low']
    high_low_ema = (high_low_range * df['volume']).ewm(span=25).mean()
    recent_volume_weight_long = df['volume'].rolling(window=15).mean()
    adjusted_high_low = high_low_ema * recent_volume_weight_long
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    volume_ratio = df['volume'] / df['volume'].shift(20)
    adjusted_long_term_momentum = long_term_momentum + (long_term_momentum * volume_ratio)
    
    # Consolidate All Factors
    total_factor = (adjusted_close + dynamic_intraday + 
                    weighted_intraday_momentum + adjusted_high_low + 
                    adjusted_long_term_momentum)
    
    # Adjust for Open Price
    final_factor = total_factor / df['open']
    
    return final_factor
```
