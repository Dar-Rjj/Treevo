```python
def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the average price
    df['avg_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].diff()
    
    # Calculate the volume change from the previous day
    df['volume_change'] = df['volume'].diff()
    
    # Calculate the amount change from the previous day
    df['amount_change'] = df['amount'].diff()
    
    # Calculate the factor as a weighted sum of price change, volume change, and amount change
    df['factor'] = (df['price_change'] * 0.6) + (df['volume_change'] * 0.3) + (df['amount_change'] * 0.1)
    
    # Return the factor as a pandas Series
    return df['factor']
```
