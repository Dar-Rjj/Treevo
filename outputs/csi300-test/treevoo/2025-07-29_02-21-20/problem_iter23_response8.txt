```python
def heuristics_v2(df, n=20):
    import pandas as pd
    import numpy as np
    
    # Calculate Intraday Movement
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_movement = (high_indicator + low_indicator) / 2

    # Adjust for Volume Changes
    volume_ratio = df['volume'] / df['volume'].shift(n)
    adjusted_intraday_movement = intraday_movement + (intraday_movement * volume_ratio)

    # Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)

    # Adjust Long-Term Momentum by Volume
    volume_adjustment_factor = volume_ratio
    adjusted_long_term_momentum = long_term_momentum * volume_adjustment_factor

    # Incorporate Volatility
    volatility = df['close'].rolling(window=n).std()
    
    # Adjust Combined Alpha Factors with Volatility
    combined_factors = adjusted_intraday_movement + adjusted_long_term_momentum
    adjusted_alpha_factor = combined_factors - volatility

    # Consolidate Factors
    total_factor = adjusted_alpha_factor
    consolidated_factor = total_factor / df['open']

    return consolidated_factor
```
