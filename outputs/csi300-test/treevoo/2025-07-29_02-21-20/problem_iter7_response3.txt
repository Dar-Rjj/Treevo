```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Smoothed Intraday Momentum
    recent_volume_5d = df['volume'].rolling(window=5).mean()
    smoothed_intraday_momentum = (intraday_return * df['volume']).ewm(span=10).mean() / recent_volume_5d
    
    # Calculate High-Low Breakout Potential
    recent_volume_10d = df['volume'].rolling(window=10).mean()
    high_low_breakout_potential = (high_low_range * df['volume']).ewm(span=20).mean() / recent_volume_10d
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential
    
    # Calculate Trading Activity Intensity
    trading_activity_intensity = df['volume'].rolling(window=5).mean() / df['volume']
    
    # Final Alpha Factor
    alpha_factor = (momentum_reversal_indicator + 
                    high_low_breakout_potential + 
                    trading_activity_intensity)
    
    return alpha_factor
```
