```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum_indicator = high_price_indicator * low_price_indicator
    
    # Volume Adjusted Intraday Factors
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume = df['volume'].rolling(window=5).mean()
    weighted_smoothed_intraday_momentum = smoothed_intraday_momentum * recent_volume
    
    # High-Low Breakout Potential
    smoothed_high_low_breakout = volume_adjusted_high_low_range.ewm(span=20).mean()
    recent_volume_10 = df['volume'].rolling(window=10).mean()
    weighted_high_low_breakout = smoothed_high_low_breakout * recent_volume_10
    
    # Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / weighted_high_low_breakout
    
    # Trading Activity Intensity
    trading_activity_intensity = df['volume'].rolling(window=5).mean() / df['volume']
    
    # Consolidate Factors
    consolidated_factors = (intraday_momentum_indicator + 
                            weighted_smoothed_intraday_momentum + 
                            trading_activity_intensity) / df['open']
    
    return consolidated_factors
```
