```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    adj_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Volume Impact on Intraday Momentum
    alpha = 0.3  # Smoothing factor for EMA
    smoothed_volume = df['volume'].ewm(alpha=alpha, adjust=False).mean()
    volume_impact_intraday_momentum = smoothed_volume * intraday_momentum
    
    # Long-Term Momentum Adjusted for Volume
    lookback_period = 20  # Define n as a Lookback Period
    long_term_momentum = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    volume_ratio = df['volume'] / df['volume'].shift(lookback_period)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_ratio
    
    # Consolidate All Factors
    total_factor = adj_close + volume_impact_intraday_momentum + volume_adjusted_long_term_momentum
    factor_values = total_factor / df['open']
    
    return factor_values
```
