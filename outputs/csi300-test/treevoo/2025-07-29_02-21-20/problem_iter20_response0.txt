```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Enhanced Intraday Momentum
    df['high_price_indicator'] = df['high'] - df['open']
    df['low_price_indicator'] = df['open'] - df['low']
    high_weight = 0.6  # Define Weights for High and Low
    low_weight = 0.4
    df['enhanced_intraday_momentum'] = (high_weight * df['high_price_indicator']) + (low_weight * df['low_price_indicator'])
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    ema_volume = df['volume'].ewm(span=10).mean()
    df['normalized_volume'] = df['volume'] / ema_volume
    df['adjusted_intraday_momentum'] = df['enhanced_intraday_momentum'] * df['normalized_volume']
    
    # Calculate Weighted Intraday Momentum
    intraday_vol = (df['intraday_return'] * df['volume']).ewm(span=12).mean()
    recent_vol = df['volume'].rolling(window=7).mean()
    df['weighted_intraday_momentum'] = intraday_vol * recent_vol
    
    # Calculate Adjusted High-Low Breakout Potential
    hl_breakout_potential = (df['high_low_range'] * df['volume']).ewm(span=25).mean()
    recent_vol_15 = df['volume'].rolling(window=15).mean()
    df['adjusted_high_low_breakout_potential'] = hl_breakout_potential * recent_vol_15
    
    # Calculate Momentum Reversal Indicator
    df['momentum_reversal_indicator'] = (df['intraday_return'] - df['weighted_intraday_momentum']) / df['adjusted_high_low_breakout_potential']
    
    # Incorporate Longer-Term Volume-Adjusted Momentum
    n_days = 20
    df['long_term_momentum'] = (df['close'] - df['close'].shift(n_days)) / df['close'].shift(n_days)
    volume_ratio = df['volume'] / df['volume'].shift(n_days)
    df['final_long_term_factor'] = df['long_term_momentum'] * volume_ratio + df['long_term_momentum']
    
    # Calculate Daily Price Return
    df['daily_price_return'] = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01
    df['significant_price_increase'] = (df['daily_price_return'] > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    df['volume_spike_indicator'] = (df['volume'] > df['volume'].rolling(window=7).mean()).astype(int)
    
    # Compute Weighted Price Momentum
    df['accumulated_momentum'] = (df['daily_price_return'] * df['significant_price_increase']).cumsum()
    
    # Apply Volume Spike Filter
    df['filtered_accumulated_momentum'] = df['accumulated_momentum'] * df['volume_spike_indicator']
    
    # Final Alpha Factor Synthesis
    df['final_alpha_factor'] = (df['adjusted_intraday_momentum'] + df['final_long_term_factor'] + df['filtered_accumulated_momentum']) * df['volume_spike_indicator']
    
    # Adjust for Open Price
    df['final_alpha_factor'] = df['final_alpha_factor'] / df['open']
    
    return df['final_alpha_factor']
```
