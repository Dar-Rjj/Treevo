```python
import pandas as pd

def heuristics_v2(df):
    # 1.1. Factor = Close - Open
    df['price_movement'] = df['close'] - df['open']
    
    # 1.2. Factor = High - Low
    df['intraday_range'] = df['high'] - df['low']
    
    # 1.3. Factor = (Close - Low) / (High - Low)
    df['closing_sentiment'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # 1.4. Factor = (Close_t - Close_{t-1}) + (Close_{t-1} - Close_{t-2})
    df['multi_day_trend'] = df['close'].diff() + df['close'].shift(1).diff()
    
    # 2.1. Factor = Volume - Volume_{t-1}
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # 2.2. If Close > Open then Factor = Volume else Factor = -Volume
    df['strength_of_move'] = df.apply(lambda row: row['volume'] if row['close'] > row['open'] else -row['volume'], axis=1)
    
    # 2.3. If (High - Low) > Average_Range then Factor = Volume else Factor = 0
    average_range = df['intraday_range'].mean()
    df['volume_on_extreme_moves'] = df.apply(lambda row: row['volume'] if row['intraday_range'] > average_range else 0, axis=1)
    
    # 3.1. Factor = Amount
    df['amount_traded'] = df['amount']
    
    # 3.2. If Close > Open then Factor = Amount else Factor = -Amount
    df['amount_with_price_direction'] = df.apply(lambda row: row['amount'] if row['close'] > row['open'] else -row['amount'], axis=1)
    
    # Combine all factors into a single DataFrame and return the mean of all factors
    combined_factors = df[['price_movement', 'intraday_range', 'closing_sentiment', 'multi_day_trend', 'volume_change', 'strength_of_move', 'volume_on_extreme_moves', 'amount_traded', 'amount_with_price_direction']].fillna(0)
    overall_factor = combined_factors.mean(axis=1)

    return overall_factor
```
