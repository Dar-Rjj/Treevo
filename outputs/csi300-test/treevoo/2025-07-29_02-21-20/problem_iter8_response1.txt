import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (df['volume'] * typical_price).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Range and VWAP
    high_vwap_diff = df['high'] - vwap
    low_vwap_diff = vwap - df['low']
    combined_intraday_range_vwap = high_vwap_diff + low_vwap_diff
    
    # Calculate Intraday Price Changes
    log_open_close_change = np.log(df['close'] / df['open'])
    log_high_low_change = np.log(df['high'] / df['low'])
    aggregated_intraday_changes = log_open_close_change + log_high_low_change
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    enhanced_intraday_momentum = high_price_indicator * low_price_indicator
    
    # Adjust for Volume Impact on Intraday Momentum
    volume_impact_intraday_momentum = df['volume'] * enhanced_intraday_momentum
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    n = 20  # Lookback Period
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    adjusted_long_term_momentum = long_term_momentum * (1 + volume_ratio)
    
    # Integrate Combined Intraday Range-VWAP and Aggregated Price Changes with Volume Synchrony
    daily_volume_deviation = df['volume'] - df['volume'].rolling(window=20).mean()
    integrated_factor = (combined_intraday_range_vwap * aggregated_intraday_changes) * daily_volume_deviation
    
    # Consolidate All Factors
    consolidated_factor = volume_impact_intraday_momentum + adjusted_long_term_momentum + integrated_factor
    
    # Daily Price Volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=20).mean()
    
    # Compute Volume-Based Adjustment Factor
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_based_adjustment = (daily_range / avg_daily_range) * (df['volume'] / volume_ma)
    
    # High-Low Price Range Momentum
    close_diff = df['close'] - df['close'].shift(10)
    true_range = np.maximum(np.abs(df['high'] - df['low']), np.maximum(np.abs(df['high'] - df['close'].shift(1)), np.abs(df['low'] - df['close'].shift(1))))
    vwatr = (true_range * df['volume']).cumsum() / df['volume'].cumsum()
    high_low_momentum = close_diff / vwatr
    
    # Adjust Momentum by VWATR
    adjusted_momentum = high_low_momentum / vwatr
    
    # Apply a moving average to smooth the factor
    smoothed_factor = consolidated_factor.rolling(window=20).mean()
    
    return smoothed_factor
