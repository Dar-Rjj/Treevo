```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    
    # Volume Adjusted Factors
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume = df['volume'].rolling(window=5).mean()
    weighted_smoothed_intraday_momentum = smoothed_intraday_momentum * (recent_volume / df['volume'])
    
    # High-Low Breakout Potential
    smoothed_high_low_breakout = volume_adjusted_high_low_range.ewm(span=20).mean()
    recent_volume_long = df['volume'].rolling(window=10).mean()
    weighted_smoothed_high_low_breakout = smoothed_high_low_breakout * (recent_volume_long / df['volume'])
    
    # Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_smoothed_intraday_momentum) / weighted_smoothed_high_low_breakout
    
    # Trading Activity Intensity
    moving_avg_volume = df['volume'].rolling(window=5).mean()
    trading_activity_intensity = moving_avg_volume / df['volume']
    
    # Consolidate Factors
    alpha_factor = (weighted_smoothed_intraday_momentum + momentum_reversal + trading_activity_intensity) / df['open']
    
    return alpha_factor
```
