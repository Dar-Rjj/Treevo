```python
def heuristics_v2(df, n=10):
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Adjust for Volume Changes
    volume_ratio = df['volume'] / df['volume'].shift(n)
    momentum_volume_adjusted = momentum * volume_ratio
    
    # Adjust for Amount Changes
    amount_ratio = df['amount'] / df['amount'].shift(n)
    momentum_amount_adjusted = momentum_volume_adjusted * amount_ratio
    
    # Combine Adjustments
    final_factor = momentum + momentum_amount_adjusted

    return final_factor
```
