```python
import pandas as pd

def heuristics_v2(df, n=20, spike_threshold=5):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=n).mean()
    normalized_volume = df['volume'] / volume_ma
    dynamic_volume_intraday_momentum = intraday_momentum * normalized_volume
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    long_term_factor = long_term_momentum + long_term_momentum * volume_ratio
    
    # Add Volume Spike Filter
    volume_spike = df['volume'] / df['volume'].shift(1)
    filtered_long_term_factor = long_term_factor.where(volume_spike < spike_threshold, 0)
    
    # Consolidate All Factors
    total_factor = adjusted_close + dynamic_volume_intraday_momentum + filtered_long_term_factor
    
    # Adjust for Open Price
    final_factor = total_factor / df['open']
    
    return final_factor
```
