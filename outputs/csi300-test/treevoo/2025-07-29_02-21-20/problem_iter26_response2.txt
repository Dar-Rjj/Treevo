```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    high_low_weights = [0.6, 0.4]  # Define Weights for High and Low
    weighted_intraday_momentum = (high_indicator * high_low_weights[0] + low_indicator * high_low_weights[1])
    
    # Volume Impact on Intraday Momentum
    alpha = 0.1  # Define Smoothing Factor
    smoothed_volume = df['volume'].ewm(alpha=alpha).mean()
    volume_impact_intraday_momentum = weighted_intraday_momentum * smoothed_volume
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    lookback_period = 10  # Define Lookback Period
    ema_volume = df['volume'].ewm(span=lookback_period).mean()
    normalized_volume = df['volume'] / ema_volume
    dynamic_volume_adjusted_momentum = normalized_volume * weighted_intraday_momentum
    
    # Long-Term Momentum Adjusted for Volume
    n = 20  # Define Lookback Period
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    final_long_term_factor = long_term_momentum + long_term_momentum * volume_ratio
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01  # Define Positive Threshold
    significant_price_increase = (daily_price_return > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    past_week_median_volume = df['volume'].rolling(window=5).median()
    volume_spike_indicator = (df['volume'] > past_week_median_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_price_increase
    
    # Apply Volume Spike Filter
    accumulated_momentum = weighted_price_momentum * volume_spike_indicator
    
    # Consolidate All Factors
    initial_factors_sum = (adjusted_close + volume_impact_intraday_momentum + 
                           dynamic_volume_adjusted_momentum + final_long_term_factor) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).std()
    
    # Incorporate Intraday Volatility into Alpha Factor
    volatility_impact = daily_price_return * intraday_volatility
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (initial_factors_sum + accumulated_momentum + 
                          volatility_impact) * volume_spike_indicator / df['open']
    
    return final_alpha_factor
```
