```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Define lookback periods
    short_lookback = 5
    long_lookback = 20
    
    # Close Price Adjustment
    df['Close_Adj'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Avg_High_Low_Spread'] = df['High_Low_Spread'].rolling(window=short_lookback).mean()
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    df['Vol_Moving_Avg'] = df['volume'].rolling(window=short_lookback).mean()
    df['Volume_Impact'] = df['volume'] / df['Vol_Moving_Avg']
    
    def non_linear_transform(x):
        return 1 + 0.5 * (x - 1)
    
    df['Transformed_Volume_Impact'] = df['Volume_Impact'].apply(non_linear_transform)
    df['Adjusted_Intraday_Momentum'] = df['Avg_High_Low_Spread'] * df['Transformed_Volume_Impact']
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    df['Long_Term_Momentum'] = (df['close'] - df['close'].shift(long_lookback)) / df['close'].shift(long_lookback)
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(long_lookback)
    df['Volume_Adjusted_Long_Term_Momentum'] = df['Long_Term_Momentum'] * np.cbrt(df['Volume_Ratio'])
    
    # Consolidate All Factors
    df['Consolidated_Factor'] = (df['Close_Adj'] + df['Adjusted_Intraday_Momentum'] + df['Volume_Adjusted_Long_Term_Momentum']) / df['open']
    
    # Daily Price Return
    df['Daily_Price_Return'] = df['close'] - df['close'].shift(1)
    
    # Significant Price Increase
    positive_threshold = 0.01
    df['Significant_Price_Increase'] = (df['Daily_Price_Return'] > positive_threshold).astype(int)
    
    # Volume Spike Indicator
    df['Past_Week_Volumes'] = df['volume'].rolling(window=5).median()
    df['Volume_Spike_Indicator'] = (df['volume'] > df['Past_Week_Volumes']).astype(int)
    
    # Weighted Price Momentum
    df['Weighted_Price_Momentum'] = df['Daily_Price_Return'] * df['Significant_Price_Increase']
    
    # Apply Volume Spike Filter
    df['Filtered_Momentum'] = df['Weighted_Price_Momentum'] * df['Volume_Spike_Indicator']
    
    # Final Alpha Factor Synthesis
    df['Final_Alpha_Factor'] = (df['Consolidated_Factor'] + df['Filtered_Momentum']) * df['Volume_Spike_Indicator']
    
    # Adjust for Open Price
    df['Final_Alpha_Factor'] /= df['open']
    
    return df['Final_Alpha_Factor']
```
