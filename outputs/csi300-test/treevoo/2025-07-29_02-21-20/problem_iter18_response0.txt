```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close Price Adjustment by Open Price
    close_adjustment = df['close'] - df['open']

    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    volume_impacted_intraday_momentum = df['volume'] * intraday_momentum

    # Incorporate Longer-Term Momentum Adjusted for Volume
    lookback_period = 20
    long_term_momentum = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    volume_ratio = df['volume'] / df['volume'].shift(lookback_period)
    adjusted_long_term_momentum = long_term_momentum * np.sqrt(volume_ratio)

    # Consolidate All Factors
    consolidated_factors = (close_adjustment + volume_impacted_intraday_momentum + adjusted_long_term_momentum) / df['open']

    # Calculate Adjusted Volume Momentum
    adjusted_volume_momentum = ((df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)) * (df['volume'] / df['volume'].shift(lookback_period))

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Weighted Intraday Momentum
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=15, min_periods=1).mean() / df['volume'].rolling(window=10, min_periods=1).mean()

    # Calculate Adjusted High-Low Breakout Potential
    breakout_potential = (high_low_range * df['volume']).ewm(span=30, min_periods=1).mean() / df['volume'].rolling(window=20, min_periods=1).mean()

    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - weighted_intraday_momentum) / breakout_potential

    # Calculate Trading Activity Intensity
    trading_activity_intensity = df['volume'].rolling(window=10, min_periods=1).mean() / df['volume']

    # Calculate Volume Impact Factor
    volume_spread = df['volume'] - df['volume'].rolling(window=15, min_periods=1).mean()
    volume_impact_factor = volume_spread * momentum_reversal_indicator

    # Final Alpha Factor Synthesis
    final_alpha_factor = (
        consolidated_factors +
        adjusted_long_term_momentum +
        adjusted_volume_momentum +
        momentum_reversal_indicator +
        breakout_potential +
        trading_activity_intensity +
        volume_impact_factor
    ) * (df['volume'] > df['volume'].shift(lookback_period))

    # Adjust for Open Price
    final_alpha_factor = final_alpha_factor / df['open']

    return final_alpha_factor
```
