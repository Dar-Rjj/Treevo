```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average Daily Range over 20 Days
    df['avg_daily_range_20'] = df['daily_range'].rolling(window=20).mean()
    
    # Calculate Moving Average of Volume over 20 Days
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volatility Adjustment Factor
    df['volatility_adjustment_factor'] = (df['daily_range'] / df['avg_daily_range_20']) * (df['volume'] / df['avg_volume_20'])
    
    return df['volatility_adjustment_factor']
```
