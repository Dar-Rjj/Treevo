```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close Price Adjustment by Open Price
    close_price_adjustment = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    volume_impacted_intraday_momentum = intraday_momentum * df['volume']
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    n = 20  # Lookback Period
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term_momentum = long_term_momentum * np.sqrt(volume_ratio)
    
    # Consolidate All Factors
    consolidated_factors = (close_price_adjustment + 
                            volume_impacted_intraday_momentum + 
                            volume_adjusted_long_term_momentum) / df['open']
    
    # Calculate Adjusted Volume Momentum
    adjusted_volume_momentum = ((df['close'] - df['close'].shift(n)) / df['close'].shift(n)) * (df['volume'] / df['volume'].shift(n))
    
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Weighted Intraday Momentum
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean() / df['volume'].rolling(window=7).mean()
    
    # Calculate Adjusted High-Low Breakout Potential
    adjusted_high_low_breakout = (high_low_range * df['volume']).ewm(span=25).mean() / df['volume'].rolling(window=15).mean()
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - weighted_intraday_momentum) / adjusted_high_low_breakout
    
    # Calculate Trading Activity Intensity
    trading_activity_intensity = df['volume'].rolling(window=7).mean() / df['volume']
    
    # Calculate Volume Impact Factor
    volume_impact_factor = (df['volume'] - df['volume'].rolling(window=10).mean()) * momentum_reversal_indicator
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + 
                          volume_adjusted_long_term_momentum + 
                          adjusted_volume_momentum + 
                          momentum_reversal_indicator + 
                          adjusted_high_low_breakout + 
                          trading_activity_intensity + 
                          volume_impact_factor)
    
    # Adjust for Open Price
    final_alpha_factor = final_alpha_factor / df['open']
    
    return final_alpha_factor
```
