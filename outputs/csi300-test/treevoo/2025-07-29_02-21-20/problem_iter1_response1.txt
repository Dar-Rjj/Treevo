```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01  # Define a positive threshold (e.g., 1%)
    df['significant_increase'] = (df['daily_return'] > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    df['volume_spike'] = (df['volume'] > df['volume'].rolling(window=7).mean()).astype(int)
    
    # Compute Weighted Price Momentum
    df['weighted_momentum'] = df['daily_return'] * df['significant_increase']
    df['accumulated_momentum'] = df['weighted_momentum'].cumsum()
    
    # Apply Volume Spike Filter
    df['alpha_factor'] = df['accumulated_momentum'] * df['volume_spike']
    
    return df['alpha_factor']
```
