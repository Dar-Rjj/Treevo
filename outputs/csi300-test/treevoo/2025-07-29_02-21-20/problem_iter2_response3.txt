```python
import pandas as pd

def heuristics(df):
    # Calculate Daily Price Volatility
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range_20'] = df['daily_range'].rolling(window=20).mean()
    
    # Compute Volume-Based Adjustment Factor
    df['volume_ma_20'] = df['volume'].rolling(window=20).mean()
    df['vol_adj_factor'] = (df['daily_range'] / df['avg_daily_range_20']) * (df['volume'] / df['volume_ma_20'])
    
    # Calculate High-Low Price Range Momentum
    df['close_diff_10'] = df['close'] - df['close'].shift(10)
    
    # Calculate True Range
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)
    
    # Calculate Volume Weighted Average True Range (VWATR)
    df['vwap_true_range'] = df['true_range'] * df['volume']
    df['vwap_true_range_sum'] = df['vwap_true_range'].rolling(window=20).sum()
    df['vwap_true_range_avg'] = df['vwap_true_range_sum'] / df['volume'].rolling(window=20).sum()
    
    # Adjust Momentum by VWATR
    df['momentum_adj'] = df['close_diff_10'] / df['vwap_true_range_avg']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['daily_range'] / df['avg_daily_range_20']) * df['momentum_adj']
    
    return df['alpha_factor']
```
