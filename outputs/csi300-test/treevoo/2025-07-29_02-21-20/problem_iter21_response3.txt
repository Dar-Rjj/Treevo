```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day SMA of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Subtract 20-day SMA from 5-day SMA for a momentum indicator
    df['SMA_Momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Compute daily VWAP using high, low, close, and volume
    df['VWAP'] = (df['volume'] * (df['high'] + df['low'] + df['close']) / 3) / df['volume']
    
    # Calculate 10-day simple moving average of VWAP
    df['SMA_VWAP_10'] = df['VWAP'].rolling(window=10).mean()
    
    # Compare today's VWAP with the 10-day SMA of VWAP for a liquidity signal
    df['VWAP_Signal'] = df['VWAP'] - df['SMA_VWAP_10']
    
    # Compute the average true range (ATR) over 14 days using high, low, and previous close
    df['TR'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)
    df['TR'] = df[['TR', (df['high'] - df['close'].shift(1)).abs(), (df['low'] - df['close'].shift(1)).abs()]].max(axis=1)
    df['ATR_14'] = df['TR'].rolling(window=14).mean()
    
    # Calculate the A/D Line using high, low, close, and volume
    df['MF'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['MF'] = df['MF'] * df['volume']
    df['ADL'] = df['MF'].cumsum()
    
    # Observe the slope of the A/D Line over the past 30 days to determine buying vs. selling pressure
    df['ADL_Slope'] = df['ADL'].diff(30)
    
    # Calculate Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Adjust for Volume Changes
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(20)
    df['Volume_Adjusted_Momentum'] = df['Momentum'] * df['Volume_Ratio'] + df['Momentum']
    
    # Calculate 14-day RSI using closing prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Calculate OBV using close and volume
    df['OBV'] = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int) - df['volume'] * (df['close'] < df['close'].shift(1)).astype(int)).cumsum()
    
    # Calculate 10-day simple moving average of OBV
    df['SMA_OBV_10'] = df['OBV'].rolling(window=10).mean()
    
    # Compare today's OBV with the 10-day SMA of OBV for a volume trend signal
    df['OBV_Signal'] = df['OBV'] - df['SMA_OBV_10']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['SMA_Momentum'] + df['VWAP_Signal'] + df['ATR_14'] + df['ADL_Slope'] + 
                          df['Volume_Adjusted_Momentum'] + df['RSI_14'] + df['OBV_Signal'])
    
    return df['Alpha_Factor']
```
