```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = high_price_indicator * low_price_indicator
    
    # Volume Adjusted Intraday Factors
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = (volume_adjusted_intraday_return.rolling(window=10).mean() * 
                                  df['volume'].rolling(window=5).mean())
    
    # High-Low Breakout Potential
    high_low_breakout_potential = (volume_adjusted_high_low_range.rolling(window=20).mean() * 
                                   df['volume'].rolling(window=10).mean())
    
    # Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential
    
    # Adjust for Volume Impact
    volume_impacted_intraday_momentum = df['volume'] * momentum_reversal_indicator
    
    # Consolidate Initial Factors
    consolidated_factors = (smoothed_intraday_momentum + volume_impacted_intraday_momentum) / df['open']
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    significant_price_increase = (daily_price_return > 0).astype(int)
    
    # Calculate Volume Spike Indicator
    avg_week_volume = df['volume'].rolling(window=7).mean()
    volume_spike_indicator = (df['volume'] > avg_week_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = (daily_price_return * significant_price_increase).cumsum()
    
    # Apply Volume Spike Filter
    volume_spike_filtered_momentum = weighted_price_momentum * volume_spike_indicator
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + volume_spike_filtered_momentum) / df['open']
    
    return final_alpha_factor
```
