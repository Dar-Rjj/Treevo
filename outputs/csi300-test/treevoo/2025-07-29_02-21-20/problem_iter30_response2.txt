```python
def heuristics_v2(data, n):
    # Calculate Basic Momentum
    data['basic_momentum'] = (data['close'] - data['close'].shift(n)) / data['close'].shift(n)
    
    # Adjust for Volume Changes
    data['volume_ratio'] = data['volume'] / data['volume'].shift(n)
    data['interim_alpha_factor'] = data['basic_momentum'] * data['volume_ratio']
    
    # Incorporate Average Daily Return
    data['avg_daily_return'] = data['close'].pct_change().rolling(window=n).mean()
    data['final_alpha_factor'] = data['interim_alpha_factor'] + data['avg_daily_return']
    
    return data['final_alpha_factor']
```
