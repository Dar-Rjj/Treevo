import pandas as pd

def heuristics_v2(df):
    # Adjusted Close Price
    adj_close = df['close'] - df['open']

    # Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2

    # Volume-Impacted Intraday Momentum
    alpha = 0.1  # Smoothing Factor
    ema_volume = df['volume'].ewm(alpha=alpha, adjust=False).mean()
    volume_impact_intraday = intraday_momentum * ema_volume

    # Volume-Adjusted Long-Term Momentum
    n = 20  # Lookback Period
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term = long_term_momentum * (volume_ratio ** (1/3))

    # Consolidate All Factors
    consolidated_factors = (adj_close + volume_impact_intraday + volume_adjusted_long_term) / df['open']

    # Volume Spike Indicator
    volume_spike_indicator = (df['volume'] > df['volume'].rolling(7).median()).astype(int)

    # Apply Volume Spike Filter
    filtered_consolidated_factors = consolidated_factors * volume_spike_indicator

    # Final Alpha Factor Synthesis
    final_alpha_factor = (consolidated_factors + filtered_consolidated_factors) * volume_spike_indicator

    return final_alpha_factor
