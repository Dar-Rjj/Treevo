import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    df['adjusted_close'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    high_low_spread = df['high'] - df['low']
    weights = [0.4, 0.3, 0.3]  # Define Weights for High, Low, and High-Low Spread
    weighted_avg_momentum = (weights[0] * high_indicator + 
                             weights[1] * low_indicator + 
                             weights[2] * high_low_spread)
    
    # Volume-Impacted Intraday Momentum
    ema_volume = df['volume'].ewm(span=10).mean()
    normalized_volume = df['volume'] / ema_volume
    volume_impacted_momentum = weighted_avg_momentum * normalized_volume
    
    # Calculate Weighted Intraday Momentum
    intraday_return = df['close'] / df['open'] - 1
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean() * df['volume'].rolling(window=7).mean()
    
    # Calculate Adjusted High-Low Breakout Potential
    adjusted_high_low_breakout = (df['high'] - df['low']) * df['volume']
    adjusted_high_low_breakout_potential = adjusted_high_low_breakout.ewm(span=25).mean() * df['volume'].rolling(window=15).mean()
    
    # Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_intraday_momentum) / adjusted_high_low_breakout_potential
    
    # Long-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    volume_ratio = df['volume'] / df['volume'].shift(20)
    long_term_momentum_adjusted = long_term_momentum * volume_ratio
    
    # Price-Volume Synchronization
    price_change = df['close'] - df['close'].shift(1)
    volume_change = df['volume'] - df['volume'].shift(1)
    price_volume_sync = price_change.rolling(window=20).corr(volume_change)
    
    # Add Volume Spike Filter
    volume_spike = df['volume'] / df['volume'].shift(1)
    spike_threshold = 2.0  # Define Spike Threshold
    filtered_volume_spike = volume_spike.apply(lambda x: 0 if x > spike_threshold else x)
    
    # Final Alpha Factor Synthesis
    total_alpha = (df['adjusted_close'] + 
                   volume_impacted_momentum + 
                   weighted_intraday_momentum + 
                   adjusted_high_low_breakout_potential + 
                   long_term_momentum_adjusted + 
                   price_volume_sync)
    final_alpha_factor = total_alpha / df['open']
    
    return final_alpha_factor
