```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close
    adjusted_close = df['close'] - df['open']
    
    # Compute Intraday Momentum Indicators
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    
    # Weighted Average for Intraday Momentum
    weights = [0.6, 0.4]  # Define Weights for High and Low Indicators
    weighted_intraday_momentum = (weights[0] * high_indicator + weights[1] * low_indicator)
    
    # Apply Dynamic Volume Impact
    volume_ema_window = 20  # Define Lookback Period for EMA
    ema_volume = df['volume'].ewm(span=volume_ema_window).mean()
    dynamic_volume_impact = df['volume'] / ema_volume
    volume_impacted_intraday_momentum = weighted_intraday_momentum * dynamic_volume_impact
    
    # Incorporate Long-Term Volume-Adjusted Momentum
    long_term_window = 60  # Define Lookback Period for Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(long_term_window)
    volume_adjustment_factor = df['volume'] / df['volume'].shift(long_term_window)
    long_term_volume_adjusted_momentum = long_term_momentum * volume_adjustment_factor
    
    # Implement Volume Spike Filter
    spike_threshold = 3.0  # Define Spike Threshold
    volume_spike = df['volume'] / df['volume'].shift(1)
    volume_spike_filter = (volume_spike < spike_threshold).astype(int)
    filtered_long_term_momentum = long_term_volume_adjusted_momentum * volume_spike_filter
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_close + volume_impacted_intraday_momentum + filtered_long_term_momentum) / df['open']
    
    return final_alpha_factor
```
