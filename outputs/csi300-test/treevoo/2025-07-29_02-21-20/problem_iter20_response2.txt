```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=30, sma_short=5, sma_long=20, vwap_sma=10, atr_lookback=14):
    # Calculate Close Price Adjustment by Open Price
    close_adj_open = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Adjust for Volume Impact on Intraday Momentum
    volume_impact_intraday_mom = df['volume'] * intraday_momentum
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'].diff(periods=n) / df['close'].shift(n))
    volume_ratio = df['volume'] / df['volume'].shift(n)
    long_term_factor = long_term_momentum * np.cbrt(volume_ratio)
    
    # Consolidate All Factors
    total_factors = close_adj_open + volume_impact_intraday_mom + long_term_factor
    consolidated_factors = total_factors / df['open']
    
    # Calculate Daily Price Return
    daily_price_return = df['close'].diff()
    
    # Identify Significant Price Increase
    significant_increase = (daily_price_return > 0).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = (daily_price_return * significant_increase).cumsum()
    
    # Apply Volume Spike Filter
    median_volume = df['volume'].rolling(window=7, min_periods=1).median()
    volume_spike = (df['volume'] > median_volume).astype(int)
    filtered_momentum = weighted_price_momentum * volume_spike
    
    # Simple Moving Average (SMA) Factors
    sma_5 = df['close'].rolling(window=sma_short, min_periods=1).mean()
    sma_20 = df['close'].rolling(window=sma_long, min_periods=1).mean()
    sma_factor = sma_5 - sma_20
    
    # Volume-weighted Average Price (VWAP) as a Factor
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume'] / df['volume'].rolling(window=vwap_sma, min_periods=1).sum()
    vwap_sma_10 = vwap.rolling(window=vwap_sma, min_periods=1).mean()
    vwap_factor = vwap - vwap_sma_10
    
    # Price Range-Based Indicators
    true_range = np.maximum.reduce([df['high'] - df['low'], abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1))])
    atr = true_range.rolling(window=atr_lookback, min_periods=1).mean()
    
    # Accumulation/Distribution Line (A/D Line) for Insights
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    ad_line = money_flow_volume.cumsum()
    ad_slope = ad_line.diff(periods=30)
    
    # Final Alpha Factor Synthesis
    final_factor = (consolidated_factors + filtered_momentum + sma_factor + atr + ad_slope) * volume_spike
    final_factor = final_factor / df['open']
    
    return final_factor
```
