import pandas as pd

def heuristics_v2(df):
    # Close Price Adjustment
    close_price_adjustment = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    avg_intraday_momentum = intraday_momentum.rolling(window=10).mean()
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=20).mean()
    volume_impact = (df['volume'] / volume_ma) * avg_intraday_momentum
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(50)) / df['close'].shift(50)
    volume_ratio = df['volume'] / df['volume'].shift(50)
    long_term_volume_adjusted_momentum = long_term_momentum * (volume_ratio ** (1/3))
    
    # Consolidate All Factors
    consolidated_factors = (close_price_adjustment + volume_impact + long_term_volume_adjusted_momentum) / df['open']
    
    # Daily Price Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Significant Price Increase
    significant_increase = (daily_return > 0.01).astype(int)
    
    # Volume Spike Indicator
    volume_spike = (df['volume'] > df['volume'].rolling(window=7).median()).astype(int)
    
    # Weighted Price Momentum
    weighted_momentum = daily_return * significant_increase
    
    # Apply Volume Spike Filter
    volume_filtered_momentum = weighted_momentum * volume_spike
    
    # Final Alpha Factor Synthesis
    final_factor = (consolidated_factors + volume_filtered_momentum) * volume_spike
    
    # Adjust for Open Price
    final_factor /= df['open']
    
    return final_factor
