import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Volume Adjusted Momentum
    n = 10
    momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    adjusted_momentum = momentum * volume_ratio
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    
    # Volume Adjusted Factors
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume = df['volume'].rolling(window=5).mean()
    weighted_smoothed_momentum = smoothed_intraday_momentum * recent_volume
    
    # High-Low Breakout Potential
    smoothed_high_low_breakout = volume_adjusted_high_low_range.ewm(span=20).mean()
    recent_10_days_volume = df['volume'].rolling(window=10).mean()
    weighted_high_low_breakout = smoothed_high_low_breakout * recent_10_days_volume
    
    # Momentum Reversal Indicator
    momentum_reversal = (intraday_return - smoothed_intraday_momentum) / weighted_high_low_breakout
    
    # Trading Activity Intensity
    moving_average_volume = df['volume'].rolling(window=5).mean()
    trading_activity_intensity = moving_average_volume / df['volume']
    
    # Consolidate and Combine Factors
    alpha_factor = (
        adjusted_momentum +
        weighted_smoothed_momentum +
        weighted_high_low_breakout +
        trading_activity_intensity
    )
    
    return alpha_factor
