```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    weights = [0.6, 0.4]  # Define Weights for High and Low
    enhanced_intraday_momentum = (weights[0] * high_price_indicator + 
                                  weights[1] * low_price_indicator)

    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema = df['volume'].ewm(span=10).mean()
    volume_normalized = df['volume'] / volume_ema
    adjusted_intraday_momentum = enhanced_intraday_momentum * volume_normalized

    # Calculate Weighted Intraday Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean()
    recent_volume_avg = df['volume'].rolling(window=7).mean()
    weighted_intraday_momentum = weighted_intraday_momentum * recent_volume_avg

    # Calculate Adjusted High-Low Breakout Potential
    high_low_range = df['high'] - df['low']
    breakout_potential = (high_low_range * df['volume']).ewm(span=25).mean()
    recent_volume_avg_15 = df['volume'].rolling(window=15).mean()
    adjusted_breakout_potential = breakout_potential * recent_volume_avg_15

    # Calculate Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_intraday_momentum) / adjusted_breakout_potential

    # Incorporate Longer-Term Volume-Adjusted Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    volume_ratio = df['volume'] / df['volume'].shift(20)
    final_long_term_factor = long_term_momentum + long_term_momentum * volume_ratio

    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)

    # Identify Significant Price Increase
    significant_price_increase = (daily_price_return > 0).astype(int)

    # Calculate Volume Spike Indicator
    volume_spike = (df['volume'] > df['volume'].rolling(window=7).mean()).astype(int)

    # Compute Weighted Price Momentum
    weighted_price_momentum = (daily_price_return * significant_price_increase).cumsum()

    # Apply Volume Spike Filter
    filtered_momentum = weighted_price_momentum * volume_spike

    # Final Alpha Factor Synthesis
    initial_factors = (adjusted_intraday_momentum + 
                       weighted_intraday_momentum + 
                       adjusted_breakout_potential + 
                       momentum_reversal + 
                       final_long_term_factor + 
                       filtered_momentum)
    alpha_factor = initial_factors * volume_spike

    # Adjust for Open Price
    alpha_factor = alpha_factor / df['open']

    return alpha_factor
```
