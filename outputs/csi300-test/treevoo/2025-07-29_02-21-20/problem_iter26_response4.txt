```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adjusted Close Price
    df['adjusted_close'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    weighted_intraday_momentum = 0.5 * high_price_indicator + 0.5 * low_price_indicator
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_normalized = df['volume'] / volume_ema_10
    intraday_momentum_vol_adj = weighted_intraday_momentum * volume_normalized
    
    # Calculate Weighted Intraday Momentum
    intraday_price_movement_vol = (df['close'] - df['open']) * df['volume']
    ema_12 = intraday_price_movement_vol.ewm(span=12, adjust=False).mean()
    recent_volume_7 = df['volume'].rolling(window=7).mean()
    weighted_intraday_momentum = ema_12 * recent_volume_7
    
    # Calculate Adjusted High-Low Breakout Potential
    high_low_range_vol = (df['high'] - df['low']) * df['volume']
    ema_25 = high_low_range_vol.ewm(span=25, adjust=False).mean()
    recent_volume_15 = df['volume'].rolling(window=15).mean()
    adjusted_high_low_breakout = ema_25 * recent_volume_15
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    lookback_period = 20
    long_term_momentum = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    volume_ratio = df['volume'] / df['volume'].shift(lookback_period)
    final_long_term_factor = long_term_momentum + long_term_momentum * volume_ratio
    
    # Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01
    significant_price_increase = (daily_price_return > positive_threshold).astype(int)
    
    # Volume Spike Indicator
    past_week_volume_avg = df['volume'].rolling(window=7).mean().shift(1)
    volume_spike_indicator = (df['volume'] > past_week_volume_avg).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_price_increase
    
    # Apply Volume Spike Filter
    filtered_weighted_momentum = weighted_price_momentum * volume_spike_indicator
    
    # Consolidate All Factors
    total_factor = (
        df['adjusted_close'] +
        intraday_momentum_vol_adj +
        weighted_intraday_momentum +
        adjusted_high_low_breakout +
        final_long_term_factor
    )
    
    # Adjust for Open Price
    total_factor_adjusted = total_factor / df['open']
    
    return total_factor_adjusted
```
