```python
def heuristics_v2(df, n=20):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    enhanced_intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Adjust for Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / volume_ma
    volume_impacted_intraday_momentum = enhanced_intraday_momentum * volume_ratio
    
    # Incorporate Longer-Term Volume-Adjusted Momentum
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_change_ratio = df['volume'] / df['volume'].shift(n)
    long_term_factor = long_term_momentum * (1 + volume_change_ratio) + long_term_momentum
    
    # Consolidate All Factors
    total_factor = intraday_price_movement + volume_impacted_intraday_momentum + long_term_factor
    
    # Adjust for Open Price
    adjusted_factor = total_factor / df['open']
    
    return adjusted_factor
```
