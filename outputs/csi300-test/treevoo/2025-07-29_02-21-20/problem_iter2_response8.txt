```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjustment
    n = 5
    sum_past_n_days_volume = df['volume'].rolling(window=n).sum()
    volume_ratio = df['volume'] / sum_past_n_days_volume
    
    # Calculate Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * volume_ratio
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * volume_ratio
    
    # Calculate Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.rolling(window=10).mean()
    recent_volume_average = df['volume'].rolling(window=5).mean()
    smoothed_intraday_momentum_weighted = smoothed_intraday_momentum * (df['volume'] / recent_volume_average)
    
    # Calculate High-Low Breakout Potential
    high_low_breakout_potential = volume_adjusted_high_low_range.rolling(window=20).mean()
    recent_volume_average_10 = df['volume'].rolling(window=10).mean()
    high_low_breakout_potential_weighted = high_low_breakout_potential * (df['volume'] / recent_volume_average_10)
    
    # Adjust High-Low Breakout Potential by Close
    adjusted_high_low_breakout_potential = high_low_breakout_potential_weighted / df['close']
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / adjusted_high_low_breakout_potential
    
    # Final Alpha Factor
    alpha_factor = momentum_reversal_indicator + adjusted_high_low_breakout_potential
    
    return alpha_factor
```
