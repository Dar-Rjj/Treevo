```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Calculate Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Calculate Smoothed Intraday Momentum
    smoothed_intraday_momentum = (volume_adjusted_intraday_return.rolling(window=10).mean() 
                                  * df['volume'].rolling(window=5).mean())
    
    # Calculate True Range
    true_range = pd.concat([df['high'] - df['low'], 
                            abs(df['high'] - df['close'].shift(1)), 
                            abs(df['low'] - df['close'].shift(1))], axis=1).max(axis=1)
    
    # Calculate Volume Weighted Average True Range (VWATR)
    vwatr = true_range * df['volume']
    
    # Calculate High-Low Breakout Potential
    breakout_potential = (volume_adjusted_high_low_range.rolling(window=20).mean() 
                          * df['volume'].rolling(window=10).mean())
    
    # Calculate Price Momentum
    price_momentum = df['close'] - df['close'].shift(10)
    
    # Adjust Price Momentum by VWATR
    adjusted_price_momentum = price_momentum / vwatr
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / breakout_potential
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Range and VWAP
    high_vwap_diff = df['high'] - vwap
    vwap_low_diff = vwap - df['low']
    vwap_based_intraday_range = high_vwap_diff + vwap_low_diff
    
    # Apply a moving average to smooth the factor
    n = 5
    smoothed_factor = (adjusted_price_momentum * breakout_potential + momentum_reversal_indicator 
                       + vwap_based_intraday_range).rolling(window=n).mean()
    
    return smoothed_factor
```
