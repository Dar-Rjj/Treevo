```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price by Open Price
    adj_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    avg_intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_lookback = 10
    moving_avg_volume = df['volume'].rolling(window=volume_lookback).mean()
    normalized_volume = df['volume'] / moving_avg_volume
    dynamic_intraday_momentum = avg_intraday_momentum * normalized_volume
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    long_term_lookback = 20
    long_term_momentum = (df['close'] - df.shift(long_term_lookback)['close']) / df.shift(long_term_lookback)['close']
    volume_ratio = df['volume'] / df.shift(long_term_lookback)['volume']
    vol_adj_long_term_mom = long_term_momentum * volume_ratio + long_term_momentum
    
    # Consolidate All Factors
    total_factors = (adj_close + dynamic_intraday_momentum + vol_adj_long_term_mom) / df['open']
    
    # Calculate Daily Price Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01
    significant_increase = (daily_return > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    avg_past_week_volume = df['volume'].rolling(window=5).mean()
    volume_spike = (df['volume'] > avg_past_week_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = daily_return * significant_increase
    
    # Apply Volume Spike Filter
    filtered_momentum = weighted_price_momentum * volume_spike
    
    # Final Alpha Factor Synthesis
    final_alpha_factor = (total_factors + filtered_momentum) * volume_spike
    
    return final_alpha_factor
```
