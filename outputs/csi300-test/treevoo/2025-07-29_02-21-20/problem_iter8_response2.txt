```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from open to close
    price_change = df['close'] - df['open']
    
    # Calculate the volume relative to the average volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    rel_volume = df['volume'] / avg_volume_5d
    
    # Calculate the dollar amount traded relative to the average amount traded over the last 5 days
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    rel_amount = df['amount'] / avg_amount_5d
    
    # Calculate the high-low range
    high_low_range = df['high'] - df['low']
    
    # Calculate the alpha factor
    alpha_factor = (price_change * rel_volume * rel_amount) / high_low_range
    
    return alpha_factor
```
