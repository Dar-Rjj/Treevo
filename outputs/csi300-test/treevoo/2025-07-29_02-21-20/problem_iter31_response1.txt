```python
import pandas as pd

def heuristics_v2(df, lookback_period=20, ema_span=10, spike_threshold=2.5):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    weighted_average = (0.6 * high_indicator + 0.4 * low_indicator)  # Define Weights for High and Low
    
    # Adjust for Dynamic Volume Impact
    ema_volume = df['volume'].ewm(span=ema_span).mean()
    normalized_volume = df['volume'] / ema_volume
    volume_impacted_intraday_momentum = weighted_average * normalized_volume
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = df['close'].pct_change(lookback_period)
    volume_adjustment_factor = df['volume'] / df['volume'].shift(lookback_period)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_adjustment_factor
    
    # Add Volume Spike Filter
    volume_spike = df['volume'] / df['volume'].shift(1)
    volume_spike_indicator = (volume_spike > spike_threshold).astype(int)
    filtered_volume_adjusted_long_term_momentum = volume_adjusted_long_term_momentum * volume_spike_indicator
    
    # Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Significant Price Increase
    significant_price_increase = (daily_price_return > 0).astype(int)
    
    # Volume Spike Indicator
    avg_past_week_volume = df['volume'].rolling(window=7).mean()
    volume_spike_indicator_2 = (df['volume'] > avg_past_week_volume).astype(int)
    
    # Weighted Price Momentum
    weighted_price_momentum = daily_price_return * significant_price_increase
    accumulated_momentum = weighted_price_momentum.cumsum()  # Accumulate the momentum
    
    # Apply Volume Spike Filter
    filtered_accumulated_momentum = accumulated_momentum * volume_spike_indicator_2
    
    # Consolidate All Factors
    total_factor = (adjusted_close 
                    + volume_impacted_intraday_momentum 
                    + filtered_volume_adjusted_long_term_momentum 
                    + filtered_accumulated_momentum)
    
    # Adjust for Open Price
    final_factor = total_factor / df['open']
    
    return final_factor
```
