```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Changes
    df['daily_close_change'] = df['close'] - df['close'].shift(1)
    df['daily_high_change'] = df['high'] - df['high'].shift(1)
    df['daily_low_change'] = df['low'] - df['low'].shift(1)

    # Identify Multi-Day Price Trends
    df['3d_close_avg'] = df['daily_close_change'].rolling(window=3).mean()
    df['7d_close_avg'] = df['daily_close_change'].rolling(window=7).mean()
    df['21d_close_avg'] = df['daily_close_change'].rolling(window=21).mean()

    # Measure Volatility
    df['21d_close_std'] = df['daily_close_change'].rolling(window=21).std()

    # Calculate Daily Volume Changes
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)

    # Identify Multi-Day Volume Trends
    df['3d_volume_avg'] = df['daily_volume_change'].rolling(window=3).mean()
    df['7d_volume_avg'] = df['daily_volume_change'].rolling(window=7).mean()
    df['21d_volume_avg'] = df['daily_volume_change'].rolling(window=21).mean()

    # Identify Gaps and Breaks
    df['up_gap'] = (df['open'] - df['close'].shift(1)) > 0
    df['down_gap'] = (df['open'] - df['close'].shift(1)) < 0
    df['up_gap_size'] = df['open'] - df['close'].shift(1)
    df['down_gap_size'] = df['open'] - df['close'].shift(1)
    
    df['upper_breakout'] = (df['high'] > df['high'].rolling(window=21).max())
    df['lower_breakout'] = (df['low'] < df['low'].rolling(window=21).min())
    df['upper_breakout_size'] = df['high'] - df['high'].rolling(window=21).max()
    df['lower_breakout_size'] = df['low'].rolling(window=21).min() - df['low']

    # Measure Intraday Patterns
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_momentum'] = df['close'] - df['open']
    df['strength_relative_to_open'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    df['21d_high_low_std'] = df[['high', 'low']].rolling(window=21).std().mean(axis=1)

    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['21d_close_avg'] + 
        df['21d_close_std'] +
        df['21d_volume_avg'] +
        df['up_gap_size'] +
        df['down_gap_size'] +
        df['upper_breakout_size'] +
        df['lower_breakout_size'] +
        df['intraday_momentum'] +
        df['strength_relative_to_open'] +
        df['21d_high_low_std']
    )

    return alpha_factor
```
