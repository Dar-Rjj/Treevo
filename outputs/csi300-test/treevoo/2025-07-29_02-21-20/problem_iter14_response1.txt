```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 1. Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # 2. Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # 3. High-Low Range
    high_low_range = df['high'] - df['low']
    
    # 4. Intraday Logarithmic Changes
    log_open_close_change = np.log(df['close']) - np.log(df['open'])
    log_high_low_change = np.log(df['high']) - np.log(df['low'])
    
    # 5. Aggregate Intraday Logarithmic Changes
    aggregated_log_change = log_open_close_change + log_high_low_change
    
    # 6. Intraday Momentum Indicators
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    intraday_momentum = high_price_indicator * low_price_indicator
    
    # 7. Volume Adjusted Intraday Factors
    volume_adjusted_intraday_return = intraday_return * df['volume']
    volume_adjusted_high_low_range = high_low_range * df['volume']
    volume_adjusted_log_change = aggregated_log_change * df['volume']
    
    # 8. Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    recent_volume = df['volume'].rolling(window=5).mean()
    weighted_smoothed_momentum = smoothed_intraday_momentum * recent_volume
    
    # 9. High-Low Breakout Potential
    breakout_potential = volume_adjusted_high_low_range.ewm(span=20).mean()
    recent_volume_10 = df['volume'].rolling(window=10).mean()
    weighted_breakout_potential = breakout_potential * recent_volume_10
    
    # 10. Momentum Reversal Indicator
    momentum_reversal = (intraday_return - smoothed_intraday_momentum) / weighted_breakout_potential
    
    # 11. Trading Activity Intensity
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    trading_activity_intensity = avg_volume_5 / df['volume']
    
    # 12. Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # 13. Identify Significant Price Increase
    significant_increase = (daily_price_return > 0).astype(int)
    
    # 14. Volume Spike Indicator
    avg_volume_week = df['volume'].rolling(window=5).mean()
    volume_spike = (df['volume'] > avg_volume_week).astype(int)
    
    # 15. Compute Weighted Price Momentum
    weighted_momentum = (daily_price_return * significant_increase).cumsum()
    
    # 16. Apply Volume Spike Filter
    filtered_momentum = weighted_momentum * volume_spike
    
    # 17. Consolidate Initial Factors
    consolidated_factors = (adjusted_close + volume_adjusted_intraday_return) / df['open']
    
    # 18. Final Alpha Factor Synthesis
    alpha_factor = (consolidated_factors + filtered_momentum + trading_activity_intensity) * volume_spike
    
    # 19. Adjust for Open Price
    alpha_factor = alpha_factor / df['open']
    
    return alpha_factor
```
