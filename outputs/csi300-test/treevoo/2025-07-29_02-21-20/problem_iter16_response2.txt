```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Weighted Intraday Momentum
    intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean()
    recent_volume_avg_7 = df['volume'].rolling(window=7).mean()
    weighted_intraday_momentum = intraday_momentum * recent_volume_avg_7
    
    # Calculate Adjusted High-Low Breakout Potential
    high_low_breakout = (high_low_range * df['volume']).ewm(span=25).mean()
    recent_volume_avg_15 = df['volume'].rolling(window=15).mean()
    adjusted_high_low_breakout = high_low_breakout * recent_volume_avg_15
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - weighted_intraday_momentum) / adjusted_high_low_breakout
    
    # Calculate Volume Impact Factor
    volume_impact = (df['volume'] - df['volume'].rolling(window=10).mean()) * momentum_reversal_indicator
    
    # Final Alpha Factor
    alpha_factor = momentum_reversal_indicator + volume_impact
    
    return alpha_factor
```
