def heuristics_v2(df, lookback_period=20):
    # Adjusted Close Price
    adjusted_close = df['close'] - df['open']

    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2

    # Volume-Impacted Intraday Momentum
    volume_ma = df['volume'].rolling(window=lookback_period).mean()
    volume_impacted_intraday = intraday_momentum * (df['volume'] / volume_ma)

    # Long-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(lookback_period)) / df['close'].shift(lookback_period)
    volume_ratio = df['volume'] / df['volume'].shift(lookback_period)
    volume_adjusted_long_term = long_term_momentum * volume_ratio

    # Consolidate All Factors
    total_factor = adjusted_close + volume_impacted_intraday + volume_adjusted_long_term
    final_factor = total_factor / df['open']

    return final_factor
