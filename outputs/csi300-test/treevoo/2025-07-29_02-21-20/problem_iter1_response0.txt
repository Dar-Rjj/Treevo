```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Component
    short_term_sma = df['close'].rolling(window=5).mean()
    long_term_sma = df['close'].rolling(window=20).mean()
    momentum_indicator = short_term_sma - long_term_sma
    
    # Introduce Volume Sensitivity
    cumulative_volume = df['volume'].rolling(window=20).sum()
    average_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = cumulative_volume / average_volume
    
    # Combine Components
    raw_alpha_factor = momentum_indicator * volume_ratio
    
    # Apply Threshold Logic
    positive_threshold = 0.5
    negative_threshold = -0.5
    alpha_factor = pd.Series(index=df.index, dtype='float64')
    
    alpha_factor[raw_alpha_factor > positive_threshold] = 1  # Buy Signal
    alpha_factor[raw_alpha_factor < negative_threshold] = -1 # Sell Signal
    alpha_factor[(raw_alpha_factor <= positive_threshold) & (raw_alpha_factor >= negative_threshold)] = 0  # Neutral Stance

    return alpha_factor
```
