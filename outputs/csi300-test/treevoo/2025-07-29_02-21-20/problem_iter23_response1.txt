```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    avg_intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_ratio = df['volume'] / volume_ma
    volume_impacted_intraday_momentum = avg_intraday_momentum * volume_ratio
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    volume_tn_ratio = df['volume'] / df['volume'].shift(20)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_tn_ratio
    
    # Add Volume Spike Filter
    volume_spike = df['volume'] / df['volume'].shift(1)
    spike_threshold = 5
    volume_spike_filter = (volume_spike < spike_threshold).astype(int)
    
    # Calculate Weighted Intraday Momentum
    intraday_return = (df['close'] - df['open']) / df['open']
    weighted_intraday_momentum = (intraday_return * df['volume']).ewm(span=12).mean()
    recent_volume_weight = df['volume'].rolling(window=7).mean() / df['volume'].mean()
    weighted_intraday_momentum = weighted_intraday_momentum * recent_volume_weight
    
    # Calculate Adjusted High-Low Breakout Potential
    high_low_range = df['high'] - df['low']
    breakout_potential = (high_low_range * df['volume']).ewm(span=25).mean()
    recent_breakout_volume_weight = df['volume'].rolling(window=15).mean() / df['volume'].mean()
    adjusted_breakout_potential = breakout_potential * recent_breakout_volume_weight
    
    # Calculate Momentum Reversal Indicator
    momentum_reversal = (intraday_return - weighted_intraday_momentum) / adjusted_breakout_potential
    
    # Final Alpha Factor Synthesis
    alpha_factor = (adjusted_close + volume_impacted_intraday_momentum + 
                    volume_adjusted_long_term_momentum + momentum_reversal) / df['open']
    
    # Apply Volume Spike Filter
    alpha_factor = alpha_factor * volume_spike_filter
    
    return alpha_factor
```
