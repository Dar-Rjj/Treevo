```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    weight_high, weight_low = 0.6, 0.4
    enhanced_intraday_momentum = (weight_high * high_price_indicator) + (weight_low * low_price_indicator)
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema = df['volume'].ewm(span=10).mean()
    normalized_volume = df['volume'] / volume_ema
    adjusted_intraday_momentum = enhanced_intraday_momentum * normalized_volume
    
    # Calculate Weighted Intraday Momentum
    weighted_intraday_momentum = (intraday_price_movement * df['volume']).ewm(span=12).mean() * df['volume'].rolling(window=7).mean()
    
    # Calculate Adjusted High-Low Breakout Potential
    high_low_range = df['high'] - df['low']
    breakout_potential = (high_low_range * df['volume']).ewm(span=25).mean() * df['volume'].rolling(window=15).mean()
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    n_days = 20
    long_term_momentum = (df['close'] - df['close'].shift(n_days)) / df['close'].shift(n_days)
    volume_ratio = df['volume'] / df['volume'].shift(n_days)
    final_long_term_momentum = long_term_momentum + (long_term_momentum * volume_ratio)
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01  # Define a positive threshold
    significant_price_increase = (daily_price_return > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    avg_week_volume = df['volume'].rolling(window=7).mean()
    volume_spike_indicator = (df['volume'] > avg_week_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = (daily_price_return * significant_price_increase).cumsum()
    
    # Apply Volume Spike Filter
    filtered_weighted_price_momentum = weighted_price_momentum * volume_spike_indicator
    
    # Consolidate All Factors
    total_factor = (intraday_price_movement 
                    + adjusted_intraday_momentum 
                    + final_long_term_momentum 
                    + filtered_weighted_price_momentum)
    
    # Adjust for Open Price
    factor_values = total_factor / df['open']
    
    return factor_values
```
