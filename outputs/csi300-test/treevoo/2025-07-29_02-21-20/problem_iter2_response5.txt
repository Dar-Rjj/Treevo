```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average Daily Range over 20 Days
    df['avg_daily_range'] = df['daily_range'].rolling(window=20).mean()
    
    # Calculate Moving Average of Volume over 20 Days
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volatility Adjustment Factor
    df['vol_adj_factor'] = (df['daily_range'] / df['avg_daily_range']) * (df['volume'] / df['avg_volume'])
    
    # Calculate Daily Price Return
    df['daily_return'] = df['close'].pct_change()
    
    # Identify Significant Price Increase
    positive_threshold = 0.01  # Define a positive threshold for significant price increase
    df['sig_increase'] = (df['daily_return'] > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    df['volume_spike'] = (df['volume'] > df['volume'].rolling(window=7).mean()).astype(int)
    
    # Compute Weighted Price Momentum
    df['weighted_momentum'] = df['daily_return'] * df['sig_increase']
    
    # Integrate Adjusted Volatility and Momentum
    df['integrated_factor'] = df['vol_adj_factor'] * df['weighted_momentum'] * df['volume_spike']
    
    # Output Final Alpha Factor
    alpha_factor = df['integrated_factor']
    
    return alpha_factor
```
