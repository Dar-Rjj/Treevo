import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day and 20-day Simple Moving Average (SMA) of closing prices
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Momentum indicator: 5-day SMA - 20-day SMA
    df['Momentum'] = df['SMA_5'] - df['SMA_20']
    
    # Volume-weighted Average Price (VWAP) calculation
    df['VWAP'] = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # 10-day SMA of VWAP
    df['SMA_VWAP_10'] = df['VWAP'].rolling(window=10).mean()
    
    # Liquidity signal: today's VWAP compared to 10-day SMA of VWAP
    df['Liquidity_Signal'] = df['VWAP'] - df['SMA_VWAP_10']

    # Calculate the True Range
    df['True_Range'] = df.eval('high - low')
    df.loc[df.index[1:], 'True_Range'] = df[['high' - df['close'].shift(1), 'low' - df['close'].shift(1), df['high'] - df['low']]].max(axis=1)
    
    # 14-day Average True Range (ATR)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # A/D Line calculation
    df['AD_Line'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    df['AD_Line'] *= df['volume']
    df['AD_Line'] = df['AD_Line'].cumsum()
    
    # Slope of the A/D Line over the past 30 days
    df['AD_Slope_30'] = df['AD_Line'].diff(periods=30)
    
    # Volume Adjusted Momentum Indicator
    df['Momentum_Raw'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(20)
    df['Volume_Adjusted_Momentum'] = df['Momentum_Raw'] + (df['Momentum_Raw'] * df['Volume_Ratio'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Momentum'] + 
        df['Liquidity_Signal'] + 
        df['ATR_14'] + 
        df['AD_Slope_30'] + 
        df['Volume_Adjusted_Momentum']
    )
    
    return df['Alpha_Factor'].dropna()
