```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    df['adjusted_close'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    high_weight = 0.6
    low_weight = 0.4
    intraday_momentum = (high_weight * high_indicator + low_weight * low_indicator) / (df['high'] - df['low'])
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ema_lookback = 20
    df['volume_ema'] = df['volume'].ewm(span=volume_ema_lookback, adjust=False).mean()
    volume_normalized = df['volume'] / df['volume_ema']
    volume_impacted_intraday_momentum = intraday_momentum * volume_normalized
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_lookback = 252
    long_term_momentum = (df['close'] - df['close'].shift(long_term_lookback)) / df['close'].shift(long_term_lookback)
    volume_adjustment_factor = df['volume'] / df['volume'].shift(long_term_lookback)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_adjustment_factor
    
    # Add Volume Spike Filter
    volume_spike_threshold = 3
    df['volume_spike'] = df['volume'] / df['volume'].shift(1)
    df['volume_spike_filter'] = (df['volume_spike'] < volume_spike_threshold).astype(int)
    
    # Calculate Daily Price Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    positive_threshold = 0.01
    significant_price_increase = (df['daily_return'] > positive_threshold).astype(int)
    
    # Calculate Volume Spike Indicator
    past_week_avg_volume = df['volume'].rolling(window=5).mean()
    volume_spike_indicator = (df['volume'] > past_week_avg_volume).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_price_momentum = df['daily_return'] * significant_price_increase
    
    # Apply Volume Spike Filter
    filtered_momentum = weighted_price_momentum * df['volume_spike_filter']
    
    # Consolidate All Factors
    total_factor = (volume_impacted_intraday_momentum 
                    + volume_adjusted_long_term_momentum 
                    + filtered_momentum)
    
    # Adjust for Open Price
    final_factor = total_factor / df['open']
    
    return final_factor
```
