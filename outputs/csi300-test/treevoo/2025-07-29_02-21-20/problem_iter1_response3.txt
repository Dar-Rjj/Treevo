```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Intraday Return
    df['volume_adjusted_intraday_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Volume Adjusted High-Low Range
    df['volume_adjusted_high_low_range'] = df['high_low_range'] * df['volume']
    
    # Calculate Smoothed Intraday Momentum
    df['smoothed_intraday_momentum'] = df['volume_adjusted_intraday_return'].rolling(window=10).mean()
    recent_volume = df['volume'].rolling(window=5).mean()
    df['smoothed_intraday_momentum_weighted'] = df['smoothed_intraday_momentum'] * recent_volume
    
    # Calculate High-Low Breakout Potential
    df['high_low_breakout_potential'] = df['volume_adjusted_high_low_range'].rolling(window=20).mean()
    recent_volume_10 = df['volume'].rolling(window=10).mean()
    df['high_low_breakout_potential_weighted'] = df['high_low_breakout_potential'] * recent_volume_10
    
    # Calculate Momentum Reversal Indicator
    df['momentum_reversal_indicator'] = (df['intraday_return'] - df['smoothed_intraday_momentum']) / df['high_low_breakout_potential']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['momentum_reversal_indicator'] + df['high_low_breakout_potential_weighted']
    
    return df['final_alpha_factor']
```
