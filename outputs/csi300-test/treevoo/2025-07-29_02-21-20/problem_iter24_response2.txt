```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close Price Adjustment by Open Price
    close_open_adj = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    avg_intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Adjust for Volume Impact on Intraday Momentum
    volume_impact_intraday = df['volume'] * avg_intraday_momentum
    
    # Incorporate Longer-Term Momentum Adjusted for Volume
    n = 20  # Lookback period
    long_term_momentum = (df['close'].diff(n) / df['close'].shift(n))
    volume_ratio = df['volume'] / df['volume'].shift(n)
    final_long_term_factor = long_term_momentum * (volume_ratio ** (1/3))
    
    # Consolidate All Factors
    consolidated_factors = (close_open_adj + volume_impact_intraday + final_long_term_factor) / df['open']
    
    # Calculate Daily Price Return
    daily_price_return = df['close'] - df['close'].shift(1)
    
    # Identify Significant Price Increase
    significant_increase = (daily_price_return > 0).astype(int)
    
    # Compute Weighted Price Momentum
    weighted_momentum_accumulator = 0
    weighted_momentum = daily_price_return * significant_increase
    weighted_momentum_accumulator += weighted_momentum
    
    # Apply Volume Spike Filter
    median_past_week_vol = df['volume'].rolling(window=5).median()
    volume_spike_indicator = (df['volume'] > median_past_week_vol).astype(int)
    filtered_weighted_momentum = weighted_momentum * volume_spike_indicator
    
    # Simple Moving Average (SMA) Factors
    sma_5 = df['close'].rolling(window=5).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    sma_momentum = sma_5 - sma_20
    
    # Volume-weighted Average Price (VWAP) as a Factor
    vwap = (df[['high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    sma_vwap_10 = vwap.rolling(window=10).mean()
    vwap_signal = vwap - sma_vwap_10
    
    # Price Range-Based Indicators: ATR
    tr = pd.DataFrame()
    tr['h-l'] = df['high'] - df['low']
    tr['h-pc'] = abs(df['high'] - df['close'].shift(1))
    tr['l-pc'] = abs(df['low'] - df['close'].shift(1))
    tr['tr'] = tr.max(axis=1)
    atr = tr['tr'].rolling(window=14).mean()
    
    # Accumulation/Distribution Line (A/D Line)
    ad_line = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    ad_line *= df['volume']
    ad_line = ad_line.cumsum()
    ad_slope = ad_line.diff(30)
    
    # New: Volatility-Adjusted Momentum
    std_dev_10 = df['close'].rolling(window=10).std()
    vol_adjusted_momentum = long_term_momentum / std_dev_10
    
    # New: Volume-Weighted Momentum Over Short-Term
    ema_volume_5 = df['volume'].ewm(span=5).mean()
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    volume_weighted_momentum = ema_volume_5 * short_term_momentum
    
    # Final Alpha Factor Synthesis
    alpha_factor = (consolidated_factors + filtered_weighted_momentum + sma_momentum + atr + ad_slope + vol_adjusted_momentum + volume_weighted_momentum) * volume_spike_indicator
    final_alpha_factor = alpha_factor / df['open']
    
    return final_alpha_factor
```
