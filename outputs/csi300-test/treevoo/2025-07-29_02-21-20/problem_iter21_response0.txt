```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    
    # Adjust for Volume Changes
    volume_ratio = df['volume'] / df['volume'].shift(n)
    adjusted_momentum = momentum + (momentum * volume_ratio)
    
    # Incorporate Volatility
    volatility = df['close'].rolling(window=n).std()
    final_alpha_factor = adjusted_momentum - volatility
    
    return final_alpha_factor
```
