```python
import pandas as pd

def heuristics_v2(df, n=20, volume_lookback=5, price_threshold=0.01):
    # Adjusted Close Price
    df['adj_close'] = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_price_indicator = df['high'] - df['open']
    low_price_indicator = df['open'] - df['low']
    enhanced_intraday_momentum = (high_price_indicator + low_price_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    moving_avg_volume = df['volume'].rolling(window=volume_lookback).mean()
    normalized_volume = df['volume'] / moving_avg_volume
    adjusted_intraday_momentum = enhanced_intraday_momentum * normalized_volume
    
    # Longer-Term Momentum Adjusted for Volume
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio = df['volume'] / df['volume'].shift(n)
    final_long_term_factor = long_term_momentum * (1 + volume_ratio)
    
    # Consolidate All Factors
    consolidated_factors = (df['adj_close'] + adjusted_intraday_momentum + final_long_term_factor) / df['open']
    
    # Daily Price Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Significant Price Increase
    significant_increase = (daily_return > price_threshold).astype(int)
    
    # Volume Spike Indicator
    average_weekly_volume = df['volume'].rolling(window=5).mean()
    volume_spike_indicator = (df['volume'] > average_weekly_volume).astype(int)
    
    # Weighted Price Momentum
    weighted_momentum = daily_return * significant_increase
    
    # Apply Volume Spike Filter
    filtered_momentum = weighted_momentum * volume_spike_indicator
    
    # Final Alpha Factor Synthesis
    alpha_factor = (consolidated_factors + filtered_momentum) * volume_spike_indicator
    
    return alpha_factor
```
