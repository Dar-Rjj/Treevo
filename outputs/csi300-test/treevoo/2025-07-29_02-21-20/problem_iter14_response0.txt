import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Calculate High-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Weighted Intraday Momentum
    df['intraday_momentum'] = df['intraday_return'] * df['volume']
    df['intraday_ema_12'] = df['intraday_momentum'].ewm(span=12).mean()
    recent_vol_7 = df['volume'].rolling(window=7).mean()
    df['weighted_intraday_momentum'] = df['intraday_ema_12'] * recent_vol_7
    
    # Calculate Adjusted High-Low Breakout Potential
    df['high_low_momentum'] = df['high_low_range'] * df['volume']
    df['high_low_ema_25'] = df['high_low_momentum'].ewm(span=25).mean()
    recent_vol_15 = df['volume'].rolling(window=15).mean()
    df['adjusted_high_low_breakout'] = df['high_low_ema_25'] * recent_vol_15
    
    # Calculate Smoothed Intraday Momentum
    df['smoothed_intraday_momentum'] = df['weighted_intraday_momentum'].ewm(span=10).mean() * recent_vol_7.rolling(window=5).mean()
    
    # Calculate Volume Weighted Average True Range (VWATR)
    df['true_range'] = np.maximum(np.abs(df['high'] - df['low']),
                                 np.abs(df['high'] - df['close'].shift(1)),
                                 np.abs(df['low'] - df['close'].shift(1)))
    df['vwatr'] = df['true_range'] * df['volume']
    
    # Calculate Momentum Reversal Indicator
    df['momentum_reversal'] = (df['intraday_return'] - df['smoothed_intraday_momentum']) / df['adjusted_high_low_breakout']
    
    # Calculate Volume Impact Factor
    vol_ma_10 = df['volume'].rolling(window=10).mean()
    df['volume_impact'] = (df['volume'] - vol_ma_10) * df['momentum_reversal']
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(10)
    
    # Adjust Price Momentum by VWATR
    df['adjusted_price_momentum'] = df['price_momentum'] / df['vwatr']
    
    # Calculate Volume Weighted Average Price (VWAP)
    df['typical_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (df['typical_price'] * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Combine Intraday Range and VWAP
    df['vwap_high_diff'] = df['high'] - df['vwap']
    df['vwap_low_diff'] = df['vwap'] - df['low']
    df['vwap_intraday_range'] = df['vwap_high_diff'] + df['vwap_low_diff']
    
    # VWAP-based Momentum
    df['vwap_momentum'] = df['vwap'] - df['vwap'].shift(1)
    
    # Combine Indicators
    combined_factor = (df['smoothed_intraday_momentum'] +
                       df['adjusted_high_low_breakout'] +
                       df['vwap_intraday_range'] +
                       df['vwap_momentum'])
    
    # Apply a moving average to smooth the factor
    df['smoothed_factor'] = combined_factor.rolling(window=30).mean()
    
    # Calculate Volume Adjusted Momentum
    df['volume_adjusted_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(10)
