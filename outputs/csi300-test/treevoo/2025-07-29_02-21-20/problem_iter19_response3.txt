```python
import pandas as pd

def heuristics_v2(data, lookback_short=5, lookback_long=20):
    # Calculate Adjusted Close Price
    adjusted_close = data['close'] - data['open']
    
    # Enhanced Intraday Momentum
    high_indicator = data['high'] - data['open']
    low_indicator = data['open'] - data['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    ma_volume = data['volume'].rolling(window=lookback_short).mean()
    volume_normalized = data['volume'] / ma_volume
    volume_impacted_intraday_momentum = intraday_momentum * volume_normalized
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = (data['close'] - data['close'].shift(lookback_long)) / data['close'].shift(lookback_long)
    volume_ratio = data['volume'] / data['volume'].shift(lookback_long)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_ratio
    
    # Consolidate All Factors
    total_factor = adjusted_close + volume_impacted_intraday_momentum + volume_adjusted_long_term_momentum
    alpha_factor = total_factor / data['open']
    
    return alpha_factor
```
