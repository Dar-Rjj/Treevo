```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the volume weighted price
    df['vwp'] = (df['amount'] / df['volume'])
    
    # Calculate the ratio of price change to daily range
    df['change_to_range_ratio'] = df['price_change'] / df['daily_range']
    
    # Calculate the difference between the close and the volume weighted price
    df['close_to_vwp_diff'] = df['close'] - df['vwp']
    
    # Generate the alpha factor by multiplying the change to range ratio with the close to vwp difference
    df['alpha_factor'] = df['change_to_range_ratio'] * df['close_to_vwp_diff']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
