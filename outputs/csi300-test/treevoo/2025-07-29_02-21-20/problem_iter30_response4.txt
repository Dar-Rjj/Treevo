```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low prices, representing the daily price range
    price_range = df['high'] - df['low']
    
    # Calculate the average amount traded over the past 5 days
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    
    # Calculate the ratio of the current day's volume to the 5-day average volume, representing relative activity
    relative_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Heuristic: The factor is a combination of the price range, the 5-day average amount, and the relative volume
    # It reflects the idea that large price ranges, high trading amounts, and high relative volume may indicate strong market interest
    factor_value = (price_range * avg_amount_5d * relative_volume)
    
    return factor_value
```
