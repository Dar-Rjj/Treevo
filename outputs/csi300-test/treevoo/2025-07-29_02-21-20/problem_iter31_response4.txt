```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_low_spread = df['high'] - df['low']
    intraday_momentum_weights = [0.4, 0.1, 0.5]  # Weights for High, Low, and High-Low Spread
    enhanced_intraday_momentum = (df['high'] * intraday_momentum_weights[0] +
                                  df['low'] * intraday_momentum_weights[1] +
                                  high_low_spread * intraday_momentum_weights[2])
    
    # Volume-Impacted Intraday Momentum
    lookback_period = 10
    ema_volume = df['volume'].ewm(span=lookback_period, adjust=False).mean()
    volume_ratio = df['volume'] / ema_volume
    volume_impacted_intraday_momentum = enhanced_intraday_momentum * volume_ratio
    
    # Weighted Intraday Momentum
    weighted_intraday_return = (df['close'] - df['open']) * df['volume']
    recent_7_day_avg_volume = df['volume'].rolling(window=7).mean()
    ema_weighted_intraday_return = weighted_intraday_return.ewm(span=12, adjust=False).mean()
    weighted_intraday_momentum = ema_weighted_intraday_return * recent_7_day_avg_volume
    
    # Adjusted High-Low Breakout Potential
    high_low_range_volume = (df['high'] - df['low']) * df['volume']
    recent_15_day_avg_volume = df['volume'].rolling(window=15).mean()
    ema_high_low_breakout_potential = high_low_range_volume.ewm(span=25, adjust=False).mean()
    adjusted_high_low_breakout = ema_high_low_breakout_potential * recent_15_day_avg_volume
    
    # Long-Term Momentum Adjusted for Volume
    n = 60  # Lookback period for long-term momentum
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_ratio_long_term = df['volume'] / df['volume'].shift(n)
    long_term_momentum_adjusted = long_term_momentum * volume_ratio_long_term
    
    # Add Volume Spike Filter
    volume_spike_threshold = 2.0
    volume_spike = df['volume'] / df['volume'].shift(1)
    volume_spike_filter = (volume_spike < volume_spike_threshold).astype(int)
    
    # Final Alpha Factor Synthesis
    alpha_factor = (adjusted_close + 
                    volume_impacted_intraday_momentum + 
                    weighted_intraday_momentum + 
                    adjusted_high_low_breakout + 
                    long_term_momentum_adjusted) * volume_spike_filter
    final_alpha_factor = alpha_factor / df['open']
    
    return final_alpha_factor
```
