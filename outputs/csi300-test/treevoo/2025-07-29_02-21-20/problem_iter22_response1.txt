```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily volatility
    df['daily_volatility'] = df['high'] - df['low']
    
    # Calculate body length
    df['body_length'] = abs(df['open'] - df['close'])
    
    # Identify doji patterns
    df['doji'] = (df['open'] - df['close']).abs() < (df['high'] - df['low']) * 0.1
    
    # Compute the ratio of today's volume to the average volume over a 5-day lookback period
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Measure the correlation between closing price change and volume change
    df['close_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()
    df['price_volume_corr'] = df[['close_change', 'volume_change']].rolling(window=5).corr().unstack().iloc[::2, :]['volume_change']
    
    # Investigate high volume with minimal price movement
    df['vol_price_movement'] = df['volume'] * (df['high'] - df['low']) / df['close']
    
    # Develop a short-term moving average crossover system (5-day vs 10-day)
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['ma_crossover'] = df['ma_5'] - df['ma_10']
    
    # Utilize the relative position of the closing price compared to the day's range
    df['rel_close_position'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Create a simple oscillator using the difference between a 5-day and 10-day moving average
    df['oscillator'] = df['ma_5'] - df['ma_10']
    
    # Combine all factors into a single alpha factor
    df['alpha_factor'] = (
        df['daily_volatility'] +
        df['body_length'] -
        df['doji'] +
        df['volume_ratio'] +
        df['price_volume_corr'] -
        df['vol_price_movement'] +
        df['ma_crossover'] +
        df['rel_close_position'] +
        df['oscillator']
    )
    
    return df['alpha_factor']
```
