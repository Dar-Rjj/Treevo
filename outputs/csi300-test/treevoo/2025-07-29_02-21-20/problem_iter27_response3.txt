```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Adjusted Close Price
    adj_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = 0.5 * high_indicator + 0.5 * low_indicator
    
    # Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_impact_factor = df['volume'] / volume_ma
    non_linear_transform = lambda x: np.sign(x) * (np.abs(x) ** 0.5)
    transformed_volume_impact = non_linear_transform(volume_impact_factor)
    adjusted_intraday_momentum = intraday_momentum * transformed_volume_impact
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    n = 20
    long_term_momentum = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    volume_adjustment_factor = df['volume'] / df['volume'].shift(n)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_adjustment_factor
    
    # Add Volume Spike Filter
    volume_spike = df['volume'] / df['volume'].shift(1)
    spike_threshold = 5
    spike_filter = (volume_spike < spike_threshold) | (volume_spike.isna())
    
    # Consolidate All Factors
    total_factor = adj_close + adjusted_intraday_momentum + volume_adjusted_long_term_momentum
    total_factor[~spike_filter] = 0
    consolidated_factor = total_factor / df['open']
    
    return consolidated_factor
```
