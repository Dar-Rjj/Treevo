```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Simple Moving Averages (SMA)
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Compute Momentum Indicator
    df['Momentum_Indicator'] = df['SMA_5'] - df['SMA_20']

    # Introduce Volume Sensitivity
    df['Cumulative_Volume'] = df['volume'].rolling(window=20).sum()
    df['Average_Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume_Ratio'] = df['Cumulative_Volume'] / df['Average_Volume']

    # Combine Components
    df['Raw_Alpha_Factor'] = df['Momentum_Indicator'] * df['Volume_Ratio']

    # Apply Threshold Logic
    positive_threshold = 0.1
    negative_threshold = -0.1

    conditions = [
        (df['Raw_Alpha_Factor'] > positive_threshold),
        (df['Raw_Alpha_Factor'] < negative_threshold)
    ]
    choices = [1, -1]  # 1 for buy, -1 for sell
    df['Alpha_Factor'] = pd.np.select(conditions, choices, default=0)  # 0 for neutral stance

    return df['Alpha_Factor']
```
