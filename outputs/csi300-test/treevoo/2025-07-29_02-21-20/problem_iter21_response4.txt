```python
import pandas as pd

def heuristics_v2(df, intraday_lookback=10, long_term_lookback=20, spike_threshold=5):
    # Calculate Adjusted Close Price
    adjusted_close = df['close'] - df['open']
    
    # Enhanced Intraday Momentum
    high_indicator = df['high'] - df['open']
    low_indicator = df['open'] - df['low']
    intraday_momentum = (high_indicator + low_indicator) / 2
    
    # Adjust for Dynamic Volume Impact on Intraday Momentum
    volume_ma = df['volume'].rolling(window=intraday_lookback).mean()
    normalized_volume = df['volume'] / volume_ma
    volume_adjusted_intraday_momentum = intraday_momentum * normalized_volume
    
    # Incorporate Volume-Adjusted Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(long_term_lookback)) / df['close'].shift(long_term_lookback)
    volume_ratio = df['volume'] / df['volume'].shift(long_term_lookback)
    volume_adjusted_long_term_momentum = long_term_momentum * volume_ratio
    
    # Add Volume Spike Filter
    volume_spike = df['volume'] / df['volume'].shift(1)
    volume_spike_filter = (volume_spike < spike_threshold).astype(int)
    
    # Consolidate All Factors
    total_factor = (adjusted_close + volume_adjusted_intraday_momentum + volume_adjusted_long_term_momentum) * volume_spike_filter
    consolidated_factor = total_factor / df['open']
    
    return consolidated_factor
```
