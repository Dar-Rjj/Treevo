```python
import pandas as pd

def heuristics_v2(df):
    # Adjusted Price Momentum
    df['daily_change'] = df['close'] - df['close'].shift(1)
    df['3d_avg_change'] = df['daily_change'].rolling(window=3).mean()
    df['7d_avg_change'] = df['daily_change'].rolling(window=7).mean()
    df['21d_avg_change'] = df['daily_change'].rolling(window=21).mean()

    # Volume-Adjusted Momentum and Liquidity
    n = 21
    df['long_term_momentum'] = (df['close'] - df['close'].shift(n)) / df['close'].shift(n)
    df['volume_ratio'] = df['volume'] / df['volume'].shift(n)
    df['vol_adj_momentum'] = df['long_term_momentum'] * df['volume_ratio']

    # Enhanced Intraday Momentum
    df['high_indicator'] = df['high'] - df['open']
    df['low_indicator'] = df['open'] - df['low']
    df['avg_intraday_ind'] = (df['high_indicator'] + df['low_indicator']) / 2
    df['vol_ma'] = df['volume'].rolling(window=n).mean()
    df['normalized_vol'] = df['volume'] / df['vol_ma']
    df['vol_adj_intraday'] = df['avg_intraday_ind'] * df['normalized_vol']

    # Gaps and Breaks
    df['up_gap'] = (df['open'] - df['close'].shift(1) > 0).astype(int)
    df['down_gap'] = (df['open'] - df['close'].shift(1) < 0).astype(int)
    df['size_up_gap'] = df['open'] - df['close'].shift(1)
    df['size_down_gap'] = df['open'] - df['close'].shift(1)

    df['upper_breakout'] = (df['high'] > df['high'].rolling(window=21).max()).astype(int)
    df['lower_breakout'] = (df['low'] < df['low'].rolling(window=21).min()).astype(int)
    df['size_upper_breakout'] = df['high'] - df['high'].rolling(window=21).max()
    df['size_lower_breakout'] = df['low'].rolling(window=21).min() - df['low']

    # Intraday Patterns
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_momentum'] = df['close'] - df['open']
    df['strength_rel_open'] = df['intraday_momentum'] / df['intraday_range']
    df['std_high_low_21d'] = df[['high', 'low']].rolling(window=21).std().mean(axis=1)

    # Add Volume Spike Filter
    df['volume_spike'] = df['volume'] / df['volume'].shift(1)
    volume_spike_threshold = 5
    df['filtered_volume_spike'] = (df['volume_spike'] < volume_spike_threshold).astype(int)

    # Consolidate All Factors
    df['total_factor'] = (
        df['21d_avg_change'] + 
        df['vol_adj_intraday'] + 
        df['vol_adj_momentum']
    )
    df['final_factor'] = df['total_factor'] / df['open']

    return df['final_factor']
```
