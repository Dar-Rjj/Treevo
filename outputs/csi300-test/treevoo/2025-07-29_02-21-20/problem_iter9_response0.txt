import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10):
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Volume Adjusted Intraday Return
    volume_adjusted_intraday_return = intraday_return * df['volume']
    
    # Volume Adjusted High-Low Range
    volume_adjusted_high_low_range = high_low_range * df['volume']
    
    # Smoothed Intraday Momentum
    smoothed_intraday_momentum = volume_adjusted_intraday_return.ewm(span=10).mean()
    
    # High-Low Breakout Potential
    high_low_breakout_potential = volume_adjusted_high_low_range.ewm(span=20).mean()
    
    # Momentum Reversal Indicator
    momentum_reversal_indicator = (intraday_return - smoothed_intraday_momentum) / high_low_breakout_potential
    
    # Typical Price
    typical_price = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Volume Weighted Average Price (VWAP)
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # VWAP-based Intraday Range
    vwap_based_intraday_range = (df['high'] - vwap) + (vwap - df['low'])
    
    # VWAP-based Momentum
    vwap_based_momentum = vwap - vwap.shift(1)
    
    # Combine Indicators
    combined_indicators = (
        smoothed_intraday_momentum +
        high_low_breakout_potential +
        vwap_based_intraday_range +
        vwap_based_momentum
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_indicators.ewm(span=n_days).mean()
    
    return final_alpha_factor
