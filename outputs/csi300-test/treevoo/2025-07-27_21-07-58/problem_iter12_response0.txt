```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['5_day_price_momentum'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['10_day_volume_change'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['price_vol_momentum'] = df['5_day_price_momentum'] * df['10_day_volume_change']
    
    # Calculate 2-Period Price Momentum
    df['2_period_price_momentum'] = df['close'] - df['close'].shift(2)
    
    # Calculate 3-Period Average Volume
    df['3_period_avg_volume'] = df['volume'].rolling(window=3).mean()
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['3_period_avg_volume']
    
    # Combine Momentum and Volume Adjustment
    df['combined_mom_vol_adjustment'] = df['2_period_price_momentum'] * df['volume_ratio']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['price_vol_momentum'] + df['combined_mom_vol_adjustment']
    
    return df['alpha_factor']
```
