```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    
    # Calculate Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Adjusted Intraday Momentum
    df['volume_adjusted_intraday_momentum'] = df['intraday_momentum'] * df['ema_volume']
    
    # Calculate True Range Volatility
    df['true_range_volatility'] = df['true_range'].rolling(window=10).std()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] / df['open']
    
    # Combine Factors
    df['final_alpha_factor'] = (df['volume_adjusted_intraday_momentum'] * df['intraday_return']) - df['true_range_volatility']
    
    return df['final_alpha_factor']
```
