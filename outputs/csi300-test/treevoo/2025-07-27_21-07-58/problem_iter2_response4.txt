```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate Price Momentum Component
    price_change_percentage = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Surge Component
    volume_change_percentage = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine High-Low Range and Volume
    high_low_volume = (change_high_low_range * df['volume']) / df['close']
    
    # Combine Components
    alpha_factor = (price_change_percentage * volume_change_percentage) + high_low_volume
    
    return alpha_factor
```
