```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()

    # Compute Daily Price Change Percentage
    df['Daily_Price_Change'] = df['close'].pct_change() * 100

    # Compute Average Price Momentum over 5 Days
    df['5_Day_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).mean()

    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'].diff()

    # Compute Average Volume Change over 10 Days
    df['10_Day_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['Price_Volume_Surge'] = df['5_Day_Price_Momentum'] * df['10_Day_Volume_Change']

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()

    # Combine High-Low Range and Volume
    df['High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']

    # Adjust by Average Volume
    df['Average_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_10']

    # Adjust by Average Amount
    df['Average_Amount_10'] = df['amount'].rolling(window=10).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount_10']

    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    df['Combined_Adjusted_Momentum'] = df['Adjusted_Momentum'] + df['Price_Volume_Surge']

    # Incorporate High-Low Range Dynamics
    df['5_Day_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['5_Day_Change_High_Low_Range'] = df['Change_High_Low_Range'].rolling(window=5).mean()
    df['High_Low_Range_Dynamics'] = df['5_Day_High_Low_Range'] * df['5_Day_Change_High_Low_Range']

    # Final Combined Alpha Factor
    df['Alpha_Factor'] = df['Combined_Adjusted_Momentum'] + df['High_Low_Range_Dynamics']

    return df['Alpha_Factor']
```
