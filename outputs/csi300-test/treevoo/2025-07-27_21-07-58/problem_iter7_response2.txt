```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate 1-Period Price Momentum
    df['Momentum'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Average Volume over n periods
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Average Amount over n periods
    df['Average_Amount'] = df['amount'].rolling(window=n).mean()
    
    # Compute Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Compute Amount Ratio
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Combine Momentum, Volume, and Amount Ratios
    df['Adjusted_Momentum'] = df['Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    return df['Adjusted_Momentum']
```
