```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['price_momentum_1p'] = df['close'].pct_change()

    # Compute Daily Price Change Percentage
    df['daily_price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Compute Average Price Momentum over 5 Days
    df['avg_price_momentum_5d'] = df['daily_price_change_pct'].rolling(window=5).mean()

    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)

    # Compute Average Volume Change over 10 Days
    df['avg_volume_change_10d'] = df['daily_volume_change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['combined_momentum_volume'] = df['avg_price_momentum_5d'] * df['avg_volume_change_10d']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Combine High-Low Range and Volume
    df['combined_high_low_volume'] = (df['change_high_low_range'] * df['volume']) / df['close']

    # Adjust by Average Volume
    df['avg_volume_n'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_n']

    # Adjust by Average Amount
    df['avg_amount_m'] = df['amount'].rolling(window=10).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_m']

    # Combine Adjusted Momentum, Volume, and Amount
    df['adjusted_momentum'] = df['price_momentum_1p'] * df['volume_ratio'] * df['amount_ratio'] + df['combined_momentum_volume']

    # Incorporate High-Low Range Dynamics
    df['avg_high_low_range_5d'] = df['high_low_range'].rolling(window=5).mean()
    df['avg_change_high_low_range_5d'] = df['change_high_low_range'].rolling(window=5).mean()
    df['combined_high_low_dynamics'] = df['avg_high_low_range_5d'] * df['avg_change_high_low_range_5d']

    # Introduce Price-Volume Interaction
    df['price_volume_product'] = df['close'] * df['volume']
    df['avg_price_volume_product_3d'] = df['price_volume_product'].rolling(window=3).mean()
    df['change_avg_price_volume_product_3d'] = df['avg_price_volume_product_3d'] - df['avg_price_volume_product_3d'].shift(1)

    # Final Combined Alpha Factor
    df['alpha_factor'] = df['combined_momentum_volume'] + df['adjusted_momentum'] + df['combined_high_low_dynamics'] + df['change_avg_price_volume_product_3d']

    return df['alpha_factor']
```
