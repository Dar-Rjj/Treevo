```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum_1D'] = df['close'].pct_change()

    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)

    # Compute Average Price Momentum over 5 Days
    df['Avg_Price_Momentum_5D'] = df['Daily_Price_Change'].rolling(window=5).mean()

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)

    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)

    # Compute Average Volume Change over 10 Days
    df['Avg_Volume_Change_10D'] = df['Daily_Volume_Change'].rolling(window=10).mean()

    # Calculate Volume Surge Component
    df['Volume_Surge'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Combine High-Low Range and Volume
    df['High_Low_Volume_Adjusted'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']

    # Adjust by Average Volume
    df['Avg_Volume_n'] = df['volume'].rolling(window=10).mean()
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_n']

    # Adjust by Average Amount
    df['Avg_Amount_m'] = df['amount'].rolling(window=10).mean()
    df['Amount_Ratio'] = df['amount'] / df['Avg_Amount_m']

    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.1, 0.2, 0.2

    # Combine Weighted Components
    df['Weighted_High_Low_Volume'] = w1 * df['High_Low_Volume_Adjusted']
    df['Weighted_Price_Momentum'] = w2 * df['Avg_Price_Momentum_5D']
    df['Weighted_Volume_Surge'] = w3 * df['Volume_Surge']
    df['Weighted_Volume_Ratio'] = w4 * df['Volume_Ratio']
    df['Weighted_Amount_Ratio'] = w5 * df['Amount_Ratio']

    # Final Combined Alpha Factor
    df['Alpha_Factor'] = (
        df['Weighted_High_Low_Volume'] +
        df['Weighted_Price_Momentum'] +
        df['Weighted_Volume_Surge'] +
        df['Weighted_Volume_Ratio'] +
        df['Weighted_Amount_Ratio']
    )

    # Adjusted Momentum with Volume and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum_1D'] * df['Volume_Ratio'] * df['Amount_Ratio']

    # Add Adjusted Momentum to Combined Alpha Factor
    df['Final_Alpha_Factor'] = df['Alpha_Factor'] + df['Adjusted_Momentum']

    return df['Final_Alpha_Factor'].dropna()
```
