```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Determine Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Short-Term Momentum
    df['short_term_momentum'] = (df['daily_returns'].rolling(window=5).sum() 
                                 - df['close'].rolling(window=5).mean())
    
    # Compute Long-Term Momentum
    df['long_term_momentum'] = (df['daily_returns'].rolling(window=20).sum() 
                                - df['close'].rolling(window=20).mean())
    
    # Determine True Range
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    
    # Compute 14-day Simple Moving Average of True Range
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Average Volume over n periods (n=20 for this example)
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Compute Volume Ratio
    df['volume_ratio'] = df['volume'] / df['avg_volume_20']
    
    # Combine Momentum, Volatility, and Volume Adjustment
    df['momentum_factor'] = (df['short_term_momentum'] + df['long_term_momentum'])
    df['inverse_atr'] = 1 / df['atr_14']
    df['alpha_factor'] = (df['momentum_factor'] * df['inverse_atr'] * df['volume_ratio'])
    
    return df['alpha_factor']
```
