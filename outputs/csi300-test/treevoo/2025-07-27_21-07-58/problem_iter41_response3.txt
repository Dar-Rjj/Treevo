```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()

    # Calculate the relative range (high - low) / close
    relative_range = (df['high'] - df['low']) / df['close']

    # Calculate the volume-to-amount ratio
    volume_to_amount = df['volume'] / df['amount']

    # Calculate the moving average of the daily return over 10 days
    moving_avg_return = daily_return.rolling(window=10).mean()

    # Calculate the moving standard deviation of the daily return over 10 days
    moving_std_return = daily_return.rolling(window=10).std()

    # Combine the factors into a single heuristic score
    heuristic_score = (moving_avg_return * relative_range * volume_to_amount) / (moving_std_return + 1e-6)

    return heuristic_score
```
