```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a combination of price and volume data.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
        
    Returns:
        pd.Series: A Series with the index (date) representing the factor values.
    """
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the daily volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the daily range
    df['daily_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the amount-to-volume ratio
    df['amt_vol_ratio'] = df['amount'] / df['volume']
    
    # Calculate the 10-day moving average of the daily return
    df['ma_daily_return_10'] = df['daily_return'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the daily range
    df['ma_daily_range_10'] = df['daily_range'].rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the amount-to-volume ratio
    df['ma_amt_vol_ratio_10'] = df['amt_vol_ratio'].rolling(window=10).mean()
    
    # Combine the factors to create the final heuristic
    df['heuristic_factor'] = (
        df['ma_daily_return_10'] * df['ma_daily_range_10'] * df['ma_amt_vol_ratio_10'] * df['volume_change']
    )
    
    # Return the series with the date as the index
    return df['heuristic_factor']
```
