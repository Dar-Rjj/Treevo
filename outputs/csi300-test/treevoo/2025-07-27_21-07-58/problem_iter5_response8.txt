```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['avg_price_momentum_5d'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['avg_volume_change_10d'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_and_volume'] = df['avg_price_momentum_5d'] * df['avg_volume_change_10d']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: max(0, x))
    df['sum_positive_returns_21d'] = df['positive_returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['cumulative_volume_21d'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['sum_positive_returns_21d'] * df['cumulative_volume_21d']) / 21
    
    # Calculate 1-Period Price Momentum
    df['price_momentum_1p'] = df['close'] - df['close'].shift(1)
    
    # Adjust by Average Volume
    n = 21  # Number of periods for average volume calculation
    df['average_volume_n'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume_n']
    
    # Combine Momentum and Volume Adjustment
    df['combined_momentum_vol_adj'] = df['price_momentum_1p'] * df['volume_ratio']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_momentum'] + df['combined_momentum_vol_adj']
    
    return df['alpha_factor']
```
