import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple Moving Averages
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    
    # Crossover signal
    df['Crossover_Signal'] = 0
    df.loc[df['SMA_20'] > df['SMA_50'], 'Crossover_Signal'] = 1
    df.loc[df['SMA_20'] < df['SMA_50'], 'Crossover_Signal'] = -1
    
    # Rate of Change (ROC) over 21 days
    df['ROC_21'] = df['close'].pct_change(periods=21)
    
    # True Range (TR) and Average True Range (ATR) over 21 days
    df['TR'] = df[['high', 'low', 'close']].shift(1).apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    df['ATR_21'] = df['TR'].rolling(window=21).mean()
    
    # On Balance Volume (OBV)
    df['OBV'] = 0
    for i in range(1, len(df)):
        if df.loc[df.index[i], 'close'] > df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'OBV'] = df.loc[df.index[i-1], 'OBV'] + df.loc[df.index[i], 'volume']
        elif df.loc[df.index[i], 'close'] < df.loc[df.index[i-1], 'close']:
            df.loc[df.index[i], 'OBV'] = df.loc[df.index[i-1], 'OBV'] - df.loc[df.index[i], 'volume']
        else:
            df.loc[df.index[i], 'OBV'] = df.loc[df.index[i-1], 'OBV']
    
    # Money Flow Index (MFI) over 14 days
    df['Typical_Price'] = (df['high'] + df['low'] + df['close']) / 3
    df['Raw_Money_Flow'] = df['Typical_Price'] * df['volume']
    df['Positive_Money_Flow'] = df.apply(lambda x: x['Raw_Money_Flow'] if x['Typical_Price'] > df.shift(1)['Typical_Price'] else 0, axis=1)
    df['Negative_Money_Flow'] = df.apply(lambda x: x['Raw_Money_Flow'] if x['Typical_Price'] < df.shift(1)['Typical_Price'] else 0, axis=1)
    positive_flow = df['Positive_Money_Flow'].rolling(window=14).sum()
    negative_flow = df['Negative_Money_Flow'].rolling(window=14).sum()
    df['MFI'] = 100 - (100 / (1 + (positive_flow / negative_flow)))
    
    # Pivot Points (PP)
    df['PP'] = (df['high'] + df['low'] + df['close']) / 3
    df['R1'] = 2 * df['PP'] - df['low']
    df['S1'] = 2 * df['PP'] - df['high']
    df['R2'] = df['PP'] + df['high'] - df['low']
    df['S2'] = df['PP'] - df['high'] + df['low']
    df['R3'] = df['high'] + 2 * (df['PP'] - df['low'])
    df['S3'] = df['low'] - 2 * (df['high'] - df['PP'])
    
    # Volume Price Trend (VPT)
    df['VPT'] = 0
    for i in range(1, len(df)):
        df.loc[df.index[i], 'VPT'] = df.loc[df.index[i-1], 'VPT'] + (df.loc[df.index[i], 'volume'] * (df.loc[df.index[i], 'close'] - df.loc[df.index[i-1], 'close']) / df.loc[df.index[i-1], 'close'])
    
    # Up Gap and Down Gap
    df['Up_Gap'] = (df['open'] > df['high'].shift(1)).astype(int)
    df['Down_Gap'] = (df['open'] < df['low'].shift(1)).astype(int)
    
    # High-Low Range
    df['High_Low_Range'] = df['high'] - df
