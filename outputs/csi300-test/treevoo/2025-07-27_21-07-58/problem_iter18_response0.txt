```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Compute Weighted Average Price Momentum over 7 Days
    weights_price = np.arange(1, 8)
    df['weighted_price_momentum'] = df['price_change'].rolling(window=7).apply(
        lambda x: (x * weights_price).sum() / weights_price.sum(), raw=True
    )
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute Weighted Average Volume Change over 15 Days
    weights_volume = np.arange(1, 16)
    df['weighted_volume_change'] = df['volume_change'].rolling(window=15).apply(
        lambda x: (x * weights_volume).sum() / weights_volume.sum(), raw=True
    )
    
    # Combine Price Momentum and Volume Surge
    df['combined_factor'] = df['weighted_price_momentum'] * df['weighted_volume_change']
    
    # Adjust by Exponential Moving Average of Volume
    n = 10  # Number of periods for EMA
    alpha = 2 / (n + 1)  # Smoothing factor
    df['ema_volume'] = df['volume'].ewm(alpha=alpha, adjust=False).mean()
    
    # Volume Ratio
    df['volume_ratio'] = df['volume'] / df['ema_volume']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = (
        df['combined_factor'] +
        (df['close'].pct_change().shift(-1) * df['volume_ratio'])
    )
    
    return df['final_alpha_factor']
```
