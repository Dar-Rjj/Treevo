```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Short-Term Momentum
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Momentum
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Determine True Range
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = np.abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = np.abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    
    # Compute 14-day Simple Moving Average of True Range for Volatility
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Average Volume over n periods (n=20)
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Calculate Average Amount over m periods (m=20)
    df['average_amount'] = df['amount'].rolling(window=20).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    df['momentum'] = df['short_term_momentum'] + df['long_term_momentum']
    df['alpha_factor'] = (df['momentum'] * (1 / df['average_true_range']) * 
                          df['volume_ratio'] * df['amount_ratio'])
    
    return df['alpha_factor']
```
