```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute Average Price Momentum over 5 Days
    df['price_momentum_5d'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff()
    
    # Compute Average Volume Change over 10 Days
    df['volume_change_10d'] = df['daily_volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['combined_momentum_surge'] = df['price_momentum_5d'] * df['volume_change_10d']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['moving_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['cumulative_volume_21d'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['moving_sum_positive_returns'] * df['cumulative_volume_21d']) / 21
    
    # Calculate 1-Period Price Momentum
    df['price_momentum_1p'] = df['close'] / df['close'].shift(1)
    
    # Calculate Average Volume over n periods
    df['average_volume_21d'] = df['volume'].rolling(window=21).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume_21d']
    
    # Calculate Average Amount over n periods
    df['average_amount_21d'] = df['amount'].rolling(window=21).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount_21d']
    
    # Adjust by Average Volume and Amount
    df['adjusted_momentum'] = df['price_momentum_1p'] * df['volume_ratio'] * df['amount_ratio']
    
    # Final Combined Alpha Factor
    df['improved_synthesized_momentum'] = df['volume_adjusted_momentum'] + df['adjusted_momentum']
    
    return df['improved_synthesized_momentum'].dropna()
```
