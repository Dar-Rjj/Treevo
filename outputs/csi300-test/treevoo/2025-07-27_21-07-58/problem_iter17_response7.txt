import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the average true range (ATR) over 14 days
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the relative strength (RS) over 14 days
    df['up_move'] = df['high'] - df['high'].shift(1)
    df['down_move'] = df['low'].shift(1) - df['low']
    df['positive_dm'] = df['up_move'].apply(lambda x: x if x > 0 else 0)
    df['negative_dm'] = df['down_move'].apply(lambda x: x if x > 0 else 0)
    df['positive_dm_avg'] = df['positive_dm'].rolling(window=14).mean()
    df['negative_dm_avg'] = df['negative_dm'].rolling(window=14).mean()
    df['rs'] = df['positive_dm_avg'] / df['negative_dm_avg']
    
    # Calculate the directional movement index (DMI) over 14 days
    df['dmi'] = 100 * (df['rs'] / (1 + df['rs']))
    
    # Calculate the money flow index (MFI) over 14 days
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    df['positive_money_flow'] = df['money_flow'].where(df['typical_price'] > df['typical_price'].shift(1), 0)
    df['negative_money_flow'] = df['money_flow'].where(df['typical_price'] < df['typical_price'].shift(1), 0)
    df['positive_money_flow_avg'] = df['positive_money_flow'].rolling(window=14).sum()
    df['negative_money_flow_avg'] = df['negative_money_flow'].rolling(window=14).sum()
    df['mfi'] = 100 - (100 / (1 + (df['positive_money_flow_avg'] / df['negative_money_flow_avg'])))
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['dmi'] + df['mfi'] - df['atr_14']) / 3
    
    return df['alpha_factor'].dropna()
