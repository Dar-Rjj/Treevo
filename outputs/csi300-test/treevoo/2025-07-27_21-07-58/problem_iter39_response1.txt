```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term and Medium-Term Price Momentum
    df['short_momentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['medium_momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Compute Average Price Momentum over 5 Days
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['avg_price_momentum_5d'] = df['price_change'].rolling(window=5).mean()
    
    # Identify Volume-Weighted Price Actions
    df['weighted_avg_hl'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['weighted_avg_hl_5d'] = df['weighted_avg_hl'].rolling(window=5).mean()
    df['cumulative_volume_adjusted_return'] = df[['close', 'open', 'volume']].apply(lambda x: sum((x[0] - x[1]) * x[2] for x in df[['close', 'open', 'volume']].iloc[-10:].values), axis=1)
    
    # Analyze Trading Volume Patterns
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_change_10d'] = df['daily_volume_change'].rolling(window=10).mean()
    
    # Evaluate Volatility and Range
    df['daily_price_range'] = df['high'] - df['low']
    df['relative_daily_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['std_close_22d'] = df['close'].rolling(window=22).std()
    
    # Combine Price Momentum and Volume Surge
    df['combined_momentum_volume'] = df['avg_price_momentum_5d'] * df['avg_volume_change_10d']
    
    # Adjust by Average Volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Adjust by Average Amount
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_5d']
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.2, 0.1, 0.2
    
    # Combine Weighted Components
    df['weighted_high_low_vol'] = w1 * df['weighted_avg_hl_5d'] * df['volume_ratio']
    df['weighted_price_momentum'] = w2 * df['combined_momentum_volume']
    df['weighted_volume_surge'] = w3 * df['avg_volume_change_10d']
    df['weighted_volume_ratio'] = w4 * df['volume_ratio']
    df['weighted_amount_ratio'] = w5 * df['amount_ratio']
    
    # Incorporate High-Low Range Dynamics
    df['avg_high_low_range_5d'] = df['daily_price_range'].rolling(window=5).mean()
    df['change_high_low_range'] = df['daily_price_range'] - df['daily_price_range'].shift(1)
    df['avg_change_high_low_range_5d'] = df['change_high_low_range'].rolling(window=5).mean()
    df['combined_high_low_dynamics'] = df['avg_high_low_range_5d'] * df['avg_change_high_low_range_5d']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = (
        df['weighted_high_low_vol'] +
        df['weighted_price_momentum'] +
        df['weighted_volume_surge'] +
        df['weighted_volume_ratio'] +
        df['weighted_amount_ratio'] +
        df['combined_high_low_dynamics'] +
        (df['close'] / df['close'].shift(1) - 1) * df['volume_ratio']
    )
    
    return df['final_alpha_factor'].dropna()
```
