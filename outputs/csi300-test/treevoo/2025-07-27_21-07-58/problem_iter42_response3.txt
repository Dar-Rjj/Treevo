```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate alpha factors based on price and volume data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns [open, high, low, close, amount, volume].
    
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    
    # Momentum factor based on intraday price range
    momentum_factor = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Alternate momentum insight
    alternate_momentum = (df['open'] - df['close']) / (df['high'] - df['low'])
    
    # Volume-weighted price change
    vw_price_change = (df['close'] - df['open']) * df['volume']
    
    # Cumulative volume impact over 5 days as an example
    cumulative_volume_impact = vw_price_change.rolling(window=5).sum()
    
    # Trend consistency factor
    trend_consistency = (df['close'].diff() > 0).astype(int) - (df['close'].diff() < 0).astype(int)
    
    # Transaction amount to price change ratio
    amount_to_price_change_ratio = df['amount'] / (df['close'] - df['open']).replace(0, 1e-6)  # Avoid division by zero
    
    # Significant trading day indicator, comparing to 20-day average
    avg_amount = df['amount'].rolling(window=20).mean()
    significant_trading_day = (df['amount'] / avg_amount).apply(lambda x: 1 if x > 1.5 else 0)
    
    # Combine all factors into a single composite alpha factor using a simple weighted sum
    alpha_factor = (
        0.2 * momentum_factor + 
        0.1 * alternate_momentum +
        0.2 * vw_price_change + 
        0.15 * cumulative_volume_impact +
        0.15 * trend_consistency + 
        0.1 * amount_to_price_change_ratio +
        0.1 * significant_trading_day
    )
    
    return alpha_factor
```
