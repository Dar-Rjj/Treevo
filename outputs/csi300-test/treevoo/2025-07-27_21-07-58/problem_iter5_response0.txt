```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()

    # Adjust by Average Volume
    n = 20  # Number of periods for average volume
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']

    # Adjust by Average Amount
    m = 20  # Number of periods for average amount
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']

    # Combine Momentum, Volume, and Amount Adjustment
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']

    # Calculate Price Momentum Component
    df['Daily_Price_Change_Pct'] = df['close'].pct_change()
    df['Avg_Price_Momentum_5D'] = df['Daily_Price_Change_Pct'].rolling(window=5).mean()

    # Calculate Volume Surge Component
    df['Daily_Volume_Change_Pct'] = df['volume'].pct_change()
    df['Avg_Volume_Change_10D'] = df['Daily_Volume_Change_Pct'].rolling(window=10).mean()

    # Combine Adjusted Momentum and Volume Surge
    df['Alpha_Factor'] = df['Adjusted_Momentum'] * df['Avg_Price_Momentum_5D'] * df['Avg_Volume_Change_10D']

    return df['Alpha_Factor']
```
