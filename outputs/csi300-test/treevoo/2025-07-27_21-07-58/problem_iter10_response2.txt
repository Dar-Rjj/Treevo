```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['avg_price_momentum_5d'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['avg_volume_change_10d'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['combined_momentum_surge'] = df['avg_price_momentum_5d'] * df['avg_volume_change_10d']
    
    # Calculate 1-Period Price Momentum
    df['1_period_price_momentum'] = df['close'] / df['close'].shift(1) - 1
    
    # Adjust by Average Volume
    n = 10  # Number of periods to calculate the average volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    
    # Compare with Recent Volume
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Combine Momentum and Volume Adjustment
    df['combined_momentum_volume_adj'] = df['1_period_price_momentum'] * df['volume_ratio']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['combined_momentum_surge'] + df['combined_momentum_volume_adj']
    
    return df['alpha_factor'].dropna()
```
