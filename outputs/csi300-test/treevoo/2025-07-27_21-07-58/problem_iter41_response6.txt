```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute 5-Day Average Price Momentum
    df['5_day_avg_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff()
    
    # Compute 10-Day Average Volume Change
    df['10_day_avg_volume_change'] = df['daily_volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['price_vol_momentum'] = df['5_day_avg_price_momentum'] * df['10_day_avg_volume_change']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute 21-Day Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['21_day_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['21_day_cumulative_volume'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['21_day_sum_positive_returns'] * df['21_day_cumulative_volume']) / 21
    
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Daily Change in High-Low Range
    df['daily_high_low_range_change'] = df['high_low_range'].diff()
    
    # Combine High-Low Range and Volume
    df['high_low_vol_combo'] = (df['daily_high_low_range_change'] * df['volume']) / df['close']
    
    # Introduce Weighting Factors
    w1, w2, w3 = 0.4, 0.3, 0.3  # Example weights, these can be adjusted
    
    # Combine Weighted Components
    df['weighted_high_low_vol'] = df['high_low_vol_combo'] * w1
    df['weighted_price_momentum'] = df['5_day_avg_price_momentum'] * w2
    df['weighted_volume_change'] = df['10_day_avg_volume_change'] * w3
    df['combined_weighted_factors'] = df['weighted_high_low_vol'] + df['weighted_price_momentum'] + df['weighted_volume_change']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['volume_adjusted_momentum'] + df['combined_weighted_factors']
    
    # Adjust by 21-Day Average Volume
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['recent_volume_ratio'] = df['volume'] / df['21_day_avg_volume']
    df['final_alpha_factor'] += df['recent_volume_ratio']
    
    return df['final_alpha_factor'].dropna()
```
