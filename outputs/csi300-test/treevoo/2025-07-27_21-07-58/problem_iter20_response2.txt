```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the price change from the previous day
    price_change = df['close'].diff()
    
    # Calculate the volume change from the previous day
    volume_change = df['volume'].diff()
    
    # Calculate the amount change from the previous day
    amount_change = df['amount'].diff()
    
    # Calculate a momentum indicator based on the close price
    momentum = df['close'].pct_change(periods=5)
    
    # Calculate a volatility measure using the standard deviation of the last 5 days' closing prices
    volatility = df['close'].rolling(window=5).std()
    
    # Combine the factors into a single alpha factor
    alpha_factor = (avg_price * momentum) / (volatility + 1e-6)  # add small constant to avoid division by zero
    
    return alpha_factor
```
