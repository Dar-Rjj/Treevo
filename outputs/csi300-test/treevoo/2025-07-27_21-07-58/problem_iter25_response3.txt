```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['price_momentum'] = df['close'].pct_change()

    # Compute Daily Price Change Percentage
    df['daily_price_change'] = df['close'].diff()

    # Compute Average Price Momentum over 5 Days
    df['avg_5d_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()

    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff()

    # Compute Average Volume Change over 10 Days
    df['avg_10d_volume_change'] = df['daily_volume_change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['combined_mom_vol'] = df['avg_5d_price_momentum'] * df['avg_10d_volume_change']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'].diff()

    # Combine High-Low Range and Volume
    df['high_low_vol_adj'] = (df['change_high_low_range'] * df['volume']) / df['close']

    # Adjust by Average Volume
    n = 10
    df['avg_volume_n'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_n']

    # Adjust by Average Amount
    m = 5
    df['avg_amount_m'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_m']

    # Combine Adjusted Momentum, Volume, and Amount
    df['combined_adj_mom_vol_amt'] = (df['price_momentum'] * df['volume_ratio']) * df['amount_ratio'] + df['combined_mom_vol']

    # Incorporate High-Low Range Dynamics
    df['avg_5d_high_low_range'] = df['high_low_range'].rolling(window=5).mean()
    df['avg_5d_change_high_low_range'] = df['change_high_low_range'].rolling(window=5).mean()
    df['combined_high_low_dynamics'] = df['avg_5d_high_low_range'] * df['avg_5d_change_high_low_range']

    # Final Combined Alpha Factor
    df['alpha_factor'] = df['combined_adj_mom_vol_amt'] + df['combined_high_low_dynamics']

    return df['alpha_factor']
```
