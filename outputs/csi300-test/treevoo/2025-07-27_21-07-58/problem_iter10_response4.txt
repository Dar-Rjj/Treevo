```python
import pandas as pd

def heuristics_v2(df, window=5, n_periods=10):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Intraday Volume
    df['intraday_volume'] = df['volume']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['intraday_volume']
    
    # Cumulative Sum of Weighted Intraday Return over the defined window
    df['cumulative_weighted_return'] = df['weighted_intraday_return'].rolling(window=window).sum()
    
    # Calculate Average Volume over n periods
    df['average_volume'] = df['intraday_volume'].rolling(window=n_periods).mean()
    
    # Volume Ratio
    df['volume_ratio'] = df['intraday_volume'] / df['average_volume']
    
    # Adjust by Volume Ratio
    df['adjusted_cumulative_weighted_return'] = df['cumulative_weighted_return'] * df['volume_ratio']
    
    # Extract the factor values as a pandas Series
    factor_values = df['adjusted_cumulative_weighted_return']
    
    return factor_values
```
