```python
import pandas as pd

def heuristics_v2(df):
    # Calculate daily returns
    df['returns'] = df['close'].pct_change()
    
    # Compute short-term momentum
    df['short_momentum'] = df['returns'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Compute long-term momentum
    df['long_momentum'] = df['returns'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()

    # Determine True Range
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_close_prev'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    
    # Compute 14-day Simple Moving Average of True Range
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Combine Momentum and Volatility
    df['momentum_factor'] = (df['short_momentum'] + df['long_momentum']) * (1 / df['atr'])
    
    return df['momentum_factor'].dropna()
```
