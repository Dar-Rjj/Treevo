```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Change Percentage
    df['price_change_pct'] = df['close'].pct_change()
    
    # Compute 5-Day Average Price Momentum
    df['5_day_avg_price_momentum'] = df['price_change_pct'].rolling(window=5).mean()
    
    # Compute Daily Volume Change Percentage
    df['volume_change_pct'] = df['volume'].pct_change()
    
    # Compute 10-Day Average Volume Change
    df['10_day_avg_volume_change'] = df['volume_change_pct'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['combined_momentum_surge'] = df['5_day_avg_price_momentum'] * df['10_day_avg_volume_change']
    
    # Calculate 1-Period Price Momentum
    df['1_period_price_momentum'] = df['close'].pct_change(periods=1)
    
    # Calculate 21-Day Average Volume
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    
    # Compute Volume Ratio
    df['volume_ratio'] = df['volume'] / df['21_day_avg_volume']
    
    # Combine Momentum and Volume Adjustment
    df['combined_momentum_vol_adj'] = df['1_period_price_momentum'] * df['volume_ratio'] + df['combined_momentum_surge']
    
    # Final Synthesized Momentum Indicator
    synthesized_momentum_factor = df['combined_momentum_vol_adj']
    
    return synthesized_momentum_factor
```
