```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Positive Returns
    positive_returns = daily_return.where(daily_return > 0, 0)
    moving_avg_positive_returns = positive_returns.rolling(window=21).mean()
    
    # Volume Adjusted Momentum
    cumulative_volume = df['volume'].rolling(window=21).sum()
    volume_adjusted_momentum = (moving_avg_positive_returns * cumulative_volume) / 21
    
    # Calculate 5-Period Price Momentum
    price_momentum_5 = df['close'] - df['close'].shift(5)
    
    # Combine High-Low Range and Volume
    high_low_range_vol = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    average_volume = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / average_volume
    
    # Adjust by Average Amount
    average_amount = df['amount'].rolling(window=21).mean()
    amount_ratio = df['amount'] / average_amount
    
    # Introduce New Component: Price-Volume Divergence
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    price_volume_divergence = daily_return - daily_volume_change / df['volume'].shift(1)
    
    # Introduce New Component: Relative Strength Index (RSI)
    gains = daily_return.where(daily_return > 0, 0)
    losses = -daily_return.where(daily_return < 0, 0)
    avg_gain = gains.rolling(window=14).mean()
    avg_loss = losses.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate 21-Day Price Volatility
    price_volatility = df['close'].rolling(window=21).std()
    
    # Adjust by Price Volatility
    volatility_adjustment = np.where(price_volatility > price_volatility.quantile(0.75), 1.75, 1.25)
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5, w6, w7 = 0.15, 0.2, 0.1, 0.1, 0.1, 0.1, 0.2
    
    # Combine Weighted Components
    weighted_high_low_range_vol = high_low_range_vol * w1
    weighted_price_momentum_5 = price_momentum_5 * w2
    weighted_volume_adjusted_momentum = volume_adjusted_momentum * w3
    weighted_volume_ratio = volume_ratio * w4
    weighted_amount_ratio = amount_ratio * w5
    weighted_price_volume_divergence = price_volume_divergence * w6
    weighted_rsi = rsi * w7
    
    combined_alpha_factor = (
        weighted_high_low_range_vol +
        weighted_price_momentum_5 +
        weighted_volume_adjusted_momentum +
        weighted_volume_ratio +
        weighted_amount_ratio +
        weighted_price_volume_divergence +
        weighted_rsi
    )
    
    # Final Combined Alpha Factor
    final_alpha_factor = combined_alpha_factor * volatility_adjustment * rsi
    
    return final_alpha_factor
```
