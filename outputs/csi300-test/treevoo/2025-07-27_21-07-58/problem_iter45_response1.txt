```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns 
                       representing market features (open, high, low, close, amount, volume).
                       
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the body of the candle
    df['body'] = df['close'] - df['open']
    
    # Calculate the range of the day
    df['range'] = df['high'] - df['low']
    
    # Calculate the percentage of the body relative to the range
    df['body_range_ratio'] = df['body'].abs() / df['range']
    
    # Calculate the money flow
    df['money_flow'] = df['amount'] / df['volume']
    
    # Calculate the smoothed money flow using a 5-day moving average
    df['smoothed_money_flow'] = df['money_flow'].rolling(window=5).mean()
    
    # Combine the two factors: body-range ratio and smoothed money flow
    df['alpha_factor'] = df['body_range_ratio'] * df['smoothed_money_flow']
    
    return df['alpha_factor']
```
