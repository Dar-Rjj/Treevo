```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Compute High-to-Low Range
    df['High_to_Low_Range'] = df['high'] - df['low']
    
    # Compute 5-Day and 20-Day Moving Average of Price Change
    df['5D_MA_Price_Change'] = df['Daily_Price_Change'].rolling(window=5).mean()
    df['20D_MA_Price_Change'] = df['Daily_Price_Change'].rolling(window=20).mean()
    
    # Determine Daily Returns
    df['Daily_Returns'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['Daily_Returns'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    df['Long_Term_Momentum'] = df['Daily_Returns'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate Average True Range for Volatility
    df['True_Range'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Adjust by Volume
    df['10D_MA_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Ratio'] = df['volume'] / df['10D_MA_Volume']
    
    # Adjust by Amount
    df['7D_MA_Amount'] = df['amount'].rolling(window=7).mean()
    df['Amount_Ratio'] = df['amount'] / df['7D_MA_Amount']
    
    # Introduce High-to-Low Range Momentum Adjustment
    df['HL_Range_Mom_Adjustment'] = (df['High_to_Low_Range'] / df['High_to_Low_Range'].shift(1)) * df['10D_MA_Volume']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    combined_factor = (
        df['5D_MA_Price_Change'] * 
        df['HL_Range_Mom_Adjustment'] * 
        df['10D_MA_Volume']
    )
    combined_factor += (df['Short_Term_Momentum'] + df['Long_Term_Momentum']) / df['ATR']
    combined_factor *= df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Enhance Signal with Price Change and Volume Ratio
    combined_factor += df['Daily_Price_Change'] * df['Volume_Ratio']
    
    # Introduce Price Volatility Adjustment
    df['10D_Std_Price_Change'] = df['Daily_Price_Change'].rolling(window=10).std()
    combined_factor /= df['10D_Std_Price_Change']
    
    # Introduce Relative Strength Index (RSI) Adjustment
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['RSI'] = 100 - (100 / (1 + rs))
    combined_factor /= df['RSI']
    
    # Introduce High-to-Low Range Over Last 5 Days
    df['5D_High_to_Low_Range'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    combined_factor /= df['5D_High_to_Low_Range']
    
    return combined_factor
```
