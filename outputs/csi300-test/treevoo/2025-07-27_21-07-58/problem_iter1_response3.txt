```python
import pandas as pd

def heuristics_v2(df, window=5):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume
    df['intraday_volume'] = df['volume'].diff().fillna(df['volume'])
    
    # Multiply Intraday Return by Intraday Volume
    df['intraday_return_volume'] = df['intraday_return'] * df['intraday_volume']
    
    # Cumulative Sum Over a Window
    df['cumulative_intraday_volume_weighted_momentum'] = df['intraday_return_volume'].rolling(window=window).sum()
    
    return df['cumulative_intraday_volume_weighted_momentum']
```
