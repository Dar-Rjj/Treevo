import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Thought 1.1: Daily Price Range
    df['price_range'] = df['high'] - df['low']
    
    # Thought 1.2: Daily Return (Close-to-Close)
    df['daily_return_close'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Thought 1.3: Daily Return (Open-to-Close)
    df['daily_return_open_close'] = (df['close'] - df['open']) / df['open']
    
    # Thought 1.4: Average Volume (Last 5 Days)
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    
    # Thought 2.1: Volume-to-Price Ratio
    df['vol_price_ratio'] = df['volume'] / df['close']
    
    # Thought 2.2: Average Volume-to-Price Ratio (Last 5 Days)
    df['avg_vol_price_ratio_5d'] = df['vol_price_ratio'].rolling(window=5).mean()
    
    # Thought 3.1: Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Thought 3.2: Average Intraday Momentum (Last 5 Days)
    df['avg_intraday_momentum_5d'] = df['intraday_momentum'].rolling(window=5).mean()
    
    # Thought 4.1: Price Gap
    df['price_gap'] = df['open'] - df['close'].shift(1)
    
    # Thought 4.2: Average Price Gap (Last 5 Days)
    df['avg_price_gap_5d'] = df['price_gap'].rolling(window=5).mean()
    
    # Thought 6.1: Relative Strength Index (RSI)
    def compute_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    df['rsi'] = compute_rsi(df['close'])
    
    # Thought 6.2: Moving Average Convergence Divergence (MACD)
    def compute_macd(series, fast=12, slow=26, signal=9):
        ema_fast = series.ewm(span=fast, adjust=False).mean()
        ema_slow = series.ewm(span=slow, adjust=False).mean()
        macd_line = ema_fast - ema_slow
        signal_line = macd_line.ewm(span=signal, adjust=False).mean()
        macd_histogram = macd_line - signal_line
        return macd_line, signal_line, macd_histogram
    macd_line, signal_line, macd_histogram = compute_macd(df['close'])
    df['macd_line'] = macd_line
    df['signal_line'] = signal_line
    df['macd_histogram'] = macd_histogram

    df['std_close_20d'] = df['close'].rolling(window=20).std()
