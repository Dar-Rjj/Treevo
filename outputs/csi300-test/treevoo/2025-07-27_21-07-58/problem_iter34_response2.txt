```python
import pandas as pd

def heuristics_v2(df):
    # Thought 1.1: Daily price range
    df['price_range'] = df['high'] - df['low']
    
    # Thought 1.2: Daily return based on previous close
    df['return_close'] = (df['close'].diff() / df['close'].shift()).fillna(0)
    
    # Thought 1.3: Daily return based on open
    df['return_open'] = (df['close'] - df['open']) / df['open']
    
    # Thought 2.1: Volume-to-price ratio
    df['volume_to_price_ratio'] = df['volume'] / df['close']
    
    # Thought 3.1: Intraday momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Thought 4.1: Price gap
    df['price_gap'] = df['open'] - df['close'].shift().fillna(0)
    
    # Thought 5: Create a composite factor
    # Assign weights to each factor based on their perceived importance
    weight_price_range = 0.2
    weight_return_close = 0.2
    weight_return_open = 0.2
    weight_volume_to_price_ratio = 0.1
    weight_intraday_momentum = 0.2
    weight_price_gap = 0.1
    
    # Sum the weighted factors to create a single composite factor
    df['composite_factor'] = (
        weight_price_range * df['price_range'] +
        weight_return_close * df['return_close'] +
        weight_return_open * df['return_open'] +
        weight_volume_to_price_ratio * df['volume_to_price_ratio'] +
        weight_intraday_momentum * df['intraday_momentum'] +
        weight_price_gap * df['price_gap']
    )
    
    # Return the composite factor as a pandas Series
    return df['composite_factor']
```
