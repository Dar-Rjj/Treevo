```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a heuristic approach.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns [open, high, low, close, amount, volume].
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day exponential moving average of the money flow
    ema_money_flow = money_flow.ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the money flow
    ema30_money_flow = money_flow.ewm(span=30, adjust=False).mean()
    
    # The alpha factor is the difference between the two EMA values
    alpha_factor = ema_money_flow - ema30_money_flow
    
    return alpha_factor
```
