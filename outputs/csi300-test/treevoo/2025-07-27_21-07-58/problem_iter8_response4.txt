```python
import pandas as pd

def heuristics_v2(df, n=5, m=5):
    # Calculate High-Low Range
    df['Range'] = df['High'] - df['Low']
    
    # Calculate Change in High-Low Range
    df['Delta_Range'] = df['Range'] - df['Range'].shift(1)
    
    # Calculate 1-Period Price Momentum
    df['Momentum'] = df['Close'] - df['Close'].shift(1)
    
    # Calculate Volume Surge Component
    df['Volume_Change_Pct'] = (df['Volume'] - df['Volume'].shift(1)) / df['Volume'].shift(1)
    
    # Combine High-Low Range and Volume
    df['Range_Volume'] = df['Delta_Range'] * df['Volume'] / df['Close']
    
    # Calculate Average Volume over n periods
    df['Avg_Volume'] = df['Volume'].rolling(window=n).mean()
    
    # Compare Recent Volume with Average Volume
    df['Volume_Ratio'] = df['Volume'] / df['Avg_Volume']
    
    # Calculate Average Amount over m periods
    df['Avg_Amount'] = df['Amount'].rolling(window=m).mean()
    
    # Compare Recent Amount with Average Amount
    df['Amount_Ratio'] = df['Amount'] / df['Avg_Amount']
    
    # Combine Components
    df['Alpha_Factor'] = (df['Momentum'] * df['Volume_Change_Pct'] + 
                          df['Range_Volume'] + 
                          df['Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio'])
    
    # Return the Alpha Factor
    return df['Alpha_Factor']
```
