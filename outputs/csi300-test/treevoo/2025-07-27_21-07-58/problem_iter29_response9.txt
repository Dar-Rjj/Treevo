import pandas as pd

def heuristics_v2(df):
    """
    The function takes a pandas DataFrame with columns (open, high, low, close, amount, volume)
    and returns a pandas Series representing the novel alpha factor.
    
    :param df: A pandas DataFrame with index (date) and columns (open, high, low, close, amount, volume)
    :return: A pandas Series with the alpha factor values indexed by (date)
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the ratio of money flow to amount
    money_flow_ratio = money_flow / df['amount']
    
    # Calculate the 10-day exponential moving average of the money flow ratio
    ema_money_flow_ratio = money_flow_ratio.ewm(span=10).mean()
    
    # Calculate the 30-day exponential moving average of the money flow ratio
    ema30_money_flow_ratio = money_flow_ratio.ewm(span=30).mean()
    
    # Calculate the alpha factor
    alpha_factor = ema_money_flow_ratio - ema30_money_flow_ratio
    
    return alpha_factor
