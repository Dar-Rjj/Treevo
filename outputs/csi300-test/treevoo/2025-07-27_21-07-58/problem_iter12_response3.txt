```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Component
    df['price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['avg_price_momentum_5'] = df['price_change_pct'].rolling(window=5).mean()

    # Calculate Volume Surge Component
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['avg_volume_change_10'] = df['volume_change_pct'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['combined_momentum_surge'] = df['avg_price_momentum_5'] * df['avg_volume_change_10']

    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['moving_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()

    # Volume Adjusted Momentum
    df['cumulative_volume_21'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['moving_sum_positive_returns'] * df['cumulative_volume_21']) / 21

    # Calculate 1-Period Price Momentum
    df['one_period_momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Adjust by Average Volume
    df['average_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    df['adjusted_one_period_momentum'] = df['one_period_momentum'] * df['volume_ratio']

    # Introduce Volatility Component
    df['std_daily_return'] = df['daily_return'].rolling(window=21).std()
    df['avg_abs_daily_return'] = df['daily_return'].abs().rolling(window=21).mean()
    df['volatility_component'] = df['std_daily_return'] / df['avg_abs_daily_return'].replace(0, np.nan)

    # Combine All Components
    df['final_factor'] = (df['combined_momentum_surge'] * df['volatility_component']) + \
                         (df['adjusted_one_period_momentum'] + df['volume_adjusted_momentum'])

    return df['final_factor']
```
