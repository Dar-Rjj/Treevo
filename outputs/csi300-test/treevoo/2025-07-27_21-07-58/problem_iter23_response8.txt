```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Momentum_1'] = df['close'].pct_change()
    
    # Compute Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Compute Moving Average of Positive Returns
    df['Positive_Returns'] = df['Daily_Return'].apply(lambda x: x if x > 0 else 0)
    df['MA_Positive_Returns'] = df['Positive_Returns'].rolling(window=21).mean()
    
    # Volume Adjusted Momentum
    df['Cumulative_Volume'] = df['volume'].rolling(window=21).sum()
    df['Volume_Adjusted_Momentum'] = (df['MA_Positive_Returns'] * df['Cumulative_Volume']) / 21
    
    # Calculate 5-Period Price Momentum
    df['Momentum_5'] = df['close'].pct_change(5)
    
    # Combine High-Low Range and Volume
    df['High_Low_Range'] = df['high'] - df['low']
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    df['High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=21).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Adjust by Average Amount
    df['Average_Amount'] = df['amount'].rolling(window=21).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Introduce New Component: Price-Volume Divergence
    df['Volume_Change_Percentage'] = df['volume'].pct_change()
    df['Price_Volume_Divergence'] = df['Daily_Return'] - df['Volume_Change_Percentage']
    
    # Introduce New Component: Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))
    
    # Calculate 21-Day Price Volatility
    df['Volatility_21'] = df['close'].rolling(window=21).std()
    
    # Adjust by Price Volatility
    df['Volatility_Adjustment'] = np.where(df['Volatility_21'] > 0.05, 1.75, 1.25)
    
    # Introduce Weighting Factors
    w1 = 0.2
    w2 = 0.15
    w3 = 0.1
    w4 = 0.15
    w5 = 0.1
    w6 = 0.15
    w7 = 0.15
    
    # Combine Weighted Components
    df['Weighted_High_Low_Volume'] = df['High_Low_Volume'] * w1
    df['Weighted_Momentum_1'] = df['Momentum_1'] * w2
    df['Weighted_Momentum_5'] = df['Momentum_5'] * w3
    df['Weighted_Volume_Adjusted_Momentum'] = df['Volume_Adjusted_Momentum'] * w4
    df['Weighted_Price_Volume_Divergence'] = df['Price_Volume_Divergence'] * w5
    df['Weighted_RSI'] = df['RSI'] * w6
    df['Weighted_Volatility_Adjustment'] = df['Volatility_Adjustment'] * w7
    
    # Final Combined Alpha Factor
    df['Alpha_Factor'] = (df['Weighted_High_Low_Volume'] + 
                          df['Weighted_Momentum_1'] + 
                          df['Weighted_Momentum_5'] + 
                          df['Weighted_Volume_Adjusted_Momentum'] + 
                          df['Weighted_Price_Volume_Divergence'] + 
                          df['Weighted_RSI'] + 
                          df['Weighted_Volatility_Adjustment'])
    
    return df['Alpha_Factor']
```
