import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate the price change
    df['price_change'] = df['close'].pct_change()
    
    # Calculate the volume change
    df['volume_change'] = df['volume'].pct_change()
    
    # Calculate the intraday range
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the weighted sum of the factors
    factor = (
        0.5 * df['price_change'] +
        0.3 * df['volume_change'] +
        0.1 * df['intraday_range'] +
        0.1 * rsi
    )
    
    return factor.dropna()
