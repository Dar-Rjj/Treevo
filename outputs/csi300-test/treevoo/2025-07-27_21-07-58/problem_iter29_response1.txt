```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the closing price
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the closing price
    df['ma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the 10-day and 30-day moving averages
    df['ma_diff'] = df['ma_10'] - df['ma_30']
    
    # Calculate the Average True Range (ATR)
    df['tr'] = df[['high', 'low']].diff(axis=1).abs().max(axis=1)  # True Range
    df['atr_14'] = df['tr'].rolling(window=14).mean()  # 14-period ATR
    
    # Calculate the Money Flow Index (MFI)
    tp = (df['high'] + df['low'] + df['close']) / 3  # Typical Price
    mf = tp * df['volume']  # Money Flow
    df['pos_mf'] = mf.where(df['close'] > df['close'].shift(), 0)
    df['neg_mf'] = mf.where(df['close'] < df['close'].shift(), 0)
    mfr = df['pos_mf'].rolling(window=14).sum() / df['neg_mf'].rolling(window=14).sum()
    df['mfi_14'] = 100 - (100 / (1 + mfr))
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['ma_diff'] / df['close']) + (df['atr_14'] / df['close']) + (50 - df['mfi_14'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
