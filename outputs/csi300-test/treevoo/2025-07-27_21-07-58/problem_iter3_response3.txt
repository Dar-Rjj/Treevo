import pandas as pd

def heuristics_v2(df, n=10, m=10):
    """
    Calculate a novel alpha factor: Price Momentum with Volume and Amount Adjustment.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index (date).
    - n: Number of periods to calculate the average volume.
    - m: Number of periods to calculate the average amount.
    
    Returns:
    - pandas Series indexed by (date) representing the factor values.
    """
    
    # Calculate 1-Period Price Momentum
    momentum = df['close'].pct_change(1).fillna(0)
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=m).mean()
    amount_ratio = df['amount'] / avg_amount
    
    # Combine Momentum, Volume, and Amount Adjustment
    adjusted_momentum = momentum * volume_ratio * amount_ratio
    
    return adjusted_momentum
