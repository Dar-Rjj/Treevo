import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute 5-Day Moving Average of Price Momentum
    df['Price_Momentum_5D_MA'] = df['Price_Momentum'].rolling(window=5).mean()
    
    # Compute 20-Day Moving Average Volume Change
    df['Volume_Change'] = df['volume'] / df['volume'].shift(1) - 1
    df['Volume_Change_20D_MA'] = df['Volume_Change'].rolling(window=20).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Price_Volume_Surge'] = df['Price_Momentum_5D_MA'] * df['Volume_Change_20D_MA']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio']
    df['Combined_Adjusted_Momentum_Volume_Amount'] = df['Adjusted_Momentum'] + df['Price_Volume_Surge'] + df['Combined_High_Low_Volume']
    
    # Incorporate High-Low Range Dynamics
    df['High_Low_Range_10D_MA'] = df['High_Low_Range'].rolling(window=10).mean()
    df['Change_High_Low_Range_10D_MA'] = df['Change_High_Low_Range'].rolling(window=10).mean()
    df['Combined_High_Low_Range_Dynamics'] = df['High_Low_Range_10D_MA'] * df['Change_High_Low_Range_10D_MA']
    
    # Integrate High-Low Range with Volume
    df['High_Low_Range_Volume_10D_MA'] = (df['High_Low_Range'] / df['volume']).rolling(window=10).mean()
    df['Volume_10D_MA'] = df['volume'].rolling(window=10).mean()
    df['Integrated_High_Low_Range_Volume'] = df['High_Low_Range_Volume_10D_MA'] * df['Volume_10D_MA']
    
    # Final Combined Alpha Factor
    df['Alpha_Factor'] = (df['Combined_Adjusted_Momentum_Volume_Amount'] +
                          df['Combined_High_Low_Range_Dynamics'] +
                          df['Integrated_High_Low_Range_Volume'])
    
    return df['Alpha_Factor']

# Example usage:
# alpha_factor = heuristics_v2(your_dataframe)
