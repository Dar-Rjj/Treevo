```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    price_momentum = df['close'] / df['close'].shift(1)
    
    # Compute Daily Price Change Percentage
    daily_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    avg_price_momentum_5d = daily_price_change.rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    avg_volume_change_10d = daily_volume_change.rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    price_vol_surge = avg_price_momentum_5d * avg_volume_change_10d
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Combine High-Low Range and Volume
    high_low_vol = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 20
    avg_volume_n = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / avg_volume_n
    
    # Adjust by Average Amount
    m = 20
    avg_amount_m = df['amount'].rolling(window=m).mean()
    amount_ratio = df['amount'] / avg_amount_m
    
    # Combine Adjusted Momentum, Volume, and Amount
    adjusted_momentum = price_momentum * volume_ratio * amount_ratio + price_vol_surge
    
    # Incorporate High-Low Range Dynamics
    high_low_range_5d = high_low_range.rolling(window=5).mean()
    change_high_low_range_5d = change_high_low_range.rolling(window=5).mean()
    high_low_dynamics = high_low_range_5d * change_high_low_range_5d
    
    # Introduce Price-Volume Interaction
    price_vol_product = df['close'] * df['volume']
    avg_price_vol_product_3d = price_vol_product.rolling(window=3).mean()
    change_avg_price_vol_product_3d = avg_price_vol_product_3d - avg_price_vol_product_3d.shift(1)
    
    # Final Combined Alpha Factor
    alpha_factor = (
        adjusted_momentum +
        high_low_dynamics +
        change_avg_price_vol_product_3d
    )
    
    return alpha_factor
```
