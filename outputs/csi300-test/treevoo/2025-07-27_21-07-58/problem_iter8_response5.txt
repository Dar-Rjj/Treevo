```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate Price Momentum Component
    price_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Surge Component
    volume_surge = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine High-Low Range and Volume
    high_low_volume = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=m).mean()
    amount_ratio = df['amount'] / avg_amount
    
    # Combine Components
    combined_price_volume = price_momentum * volume_surge
    adjusted_factor = (high_low_volume + combined_price_volume) * volume_ratio * amount_ratio
    
    return adjusted_factor
```
