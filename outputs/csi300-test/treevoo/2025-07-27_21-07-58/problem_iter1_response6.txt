```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    daily_return = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    positive_returns = daily_return.where(daily_return > 0, 0)
    moving_sum_positive_returns = positive_returns.rolling(window=21).sum()
    
    # Calculate Cumulative Volume
    cumulative_volume = df['volume'].rolling(window=21).sum()
    
    # Multiply Moving Sum of Positive Returns by Cumulative Volume
    volume_adjusted_momentum = moving_sum_positive_returns * cumulative_volume
    
    # Divide by Total Days
    total_days = 21
    volume_adjusted_momentum /= total_days
    
    return volume_adjusted_momentum
```
