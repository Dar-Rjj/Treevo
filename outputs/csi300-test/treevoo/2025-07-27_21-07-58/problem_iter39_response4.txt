```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Daily Price Change Percentage
    df['daily_price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['avg_price_momentum_5d'] = df['daily_price_change_pct'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['daily_vol_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['avg_vol_change_10d'] = df['daily_vol_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_vol_surge'] = df['avg_price_momentum_5d'] * df['avg_vol_change_10d']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['high_low_vol_ratio'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['avg_volume_21d'] = df['volume'].rolling(window=21).mean()
    
    # Adjust by Average Amount
    df['avg_amount_21d'] = df['amount'].rolling(window=21).mean()
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['vol_ratio'] = df['volume'] / df['avg_volume_21d']
    df['amount_ratio'] = df['amount'] / df['avg_amount_21d']
    df['adjusted_momentum_vol_amount'] = df['price_momentum'] * df['vol_ratio'] * df['amount_ratio']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['ma_positive_returns_21d'] = df['positive_returns'].rolling(window=21).mean()
    
    # Volume Adjusted Momentum
    df['cumulative_vol_21d'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['ma_positive_returns_21d'] * df['cumulative_vol_21d']) / 21
    
    # Introduce New Component: Price-Volume Divergence
    df['daily_vol_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['price_vol_divergence'] = df['daily_return'] - df['daily_vol_change_pct']
    
    # Introduce New Component: Price-Amount Divergence
    df['daily_amount_change_pct'] = (df['amount'] - df['amount'].shift(1)) / df['amount'].shift(1)
    df['price_amount_divergence'] = df['daily_return'] - df['daily_amount_change_pct']
    
    # Introduce New Component: Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']
    
    # Introduce New Component: Open-Open Spread
    df['open_open_spread'] = df['open'] - df['open'].shift(1)
    
    # Introduce New Component: Price-Volatility Interaction
    df['price_volatility_21d'] = df['close'].rolling(window=21).std()
    df['volatility_adjusted_momentum'] = df['price_momentum'] * df['price_volatility_21d']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = (
        df['volume_adjusted_momentum'] +
        df['momentum_vol_surge'] +
        df['high_low_vol_ratio'] +
        df['adjusted_momentum_vol_amount'] +
        df['price_vol_divergence'] +
        df['price_amount_divergence'] +
        df['open_close_spread'] +
        df['open_open_spread'] +
        df['volatility_adjusted_momentum']
    )
    
    return df['alpha_factor']
```
