import pandas as pd

def heuristics_v2(df):
    # Factor 1: (Close_t - Open_t) * Volume_t
    df['Factor1'] = (df['close'] - df['open']) * df['volume']
    
    # Factor 2: (High_t - Max(High_{t-5},...,High_{t-1})) / (Max(High_{t-5},...,High_{t-1}) - Min(Low_{t-5},...,Low_{t-1}))
    df['Factor2'] = (df['high'] - df['high'].rolling(window=5).max()) / (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min())
    
    # Factor 3: (Low_t - Min(Low_{t-5},...,Low_{t-1})) / (Max(High_{t-5},...,High_{t-1}) - Min(Low_{t-5},...,Low_{t-1}))
    df['Factor3'] = (df['low'] - df['low'].rolling(window=5).min()) / (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min())
    
    # Factor 4: Amount_t / Average(Amount_{t-9},...,Amount_{t-1})
    df['Factor4'] = df['amount'] / df['amount'].rolling(window=9).mean()
    
    # Factor 5: (Close_t - Min(Low_{t-20},...,Low_{t-1})) / (Max(High_{t-20},...,High_{t-1}) - Min(Low_{t-20},...,Low_{t-1}))
    df['Factor5'] = (df['close'] - df['low'].rolling(window=20).min()) / (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min())
    
    # Factor 6: Sum(Close_{t-4},...,Close_t) * Volume_t
    df['Factor6'] = df['close'].rolling(window=5).sum() * df['volume']
    
    # Define weights for the composite factor
    w1, w2, w3, w4, w5, w6 = 0.1, 0.1, 0.1, 0.2, 0.2, 0.3
    
    # Composite_Factor
    df['Composite_Factor'] = (w1 * df['Factor1'] + 
                              w2 * df['Factor2'] + 
                              w3 * df['Factor3'] + 
                              w4 * df['Factor4'] + 
                              w5 * df['Factor5'] + 
                              w6 * df['Factor6'])
    
    return df['Composite_Factor']
