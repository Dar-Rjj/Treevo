```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_in_high_low_range = high_low_range.diff(1).fillna(0)
    
    # Combine with Volume and Close Price
    factor = (change_in_high_low_range * df['volume']) / df['close']
    
    return factor
```
