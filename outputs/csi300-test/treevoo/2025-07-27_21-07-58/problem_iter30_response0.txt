```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['price_momentum'] = df['close'] / df['close'].shift(1)
    
    # Compute Daily Price Change Percentage
    df['price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['avg_price_momentum_5d'] = df['price_change_pct'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['avg_volume_change_10d'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['combined_price_vol'] = df['avg_price_momentum_5d'] * df['avg_volume_change_10d']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['high_low_vol_adj'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 20  # Number of periods for average volume
    df['avg_volume_n'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_n']
    
    # Adjust by Average Amount
    m = 20  # Number of periods for average amount
    df['avg_amount_m'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_m']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['adjusted_momentum_vol_amt'] = df['price_momentum'] * df['volume_ratio'] * df['amount_ratio'] + df['combined_price_vol']
    
    # Incorporate Open Price Impact
    df['open_price_change'] = df['open'] - df['open'].shift(1)
    df['avg_open_price_change_5d'] = df['open_price_change'].rolling(window=5).mean()
    df['open_price_impact'] = df['avg_open_price_change_5d'] * df['avg_price_momentum_5d']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['adjusted_momentum_vol_amt'] + df['high_low_vol_adj'] + df['open_price_impact']
    
    return df['final_alpha_factor']
```
