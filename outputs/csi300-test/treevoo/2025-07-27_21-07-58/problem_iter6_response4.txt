```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-to-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Compute Previous Day High-to-Low Price Range
    prev_high_low_range = high_low_range.shift(1)
    
    # Calculate High-to-Low Range Momentum
    high_low_momentum = high_low_range / prev_high_low_range
    
    # Adjust by Volume
    volume_adjusted_momentum = high_low_momentum / df['volume']
    
    # Compute Today's Open to Close Movement
    todays_open_close_movement = df['close'] - df['open']
    
    # Compute Previous Day's Open to Close Movement
    prev_open_close_movement = (df['close'].shift(1) - df['open'].shift(1))
    
    # Calculate Open Price Difference Momentum
    open_price_diff_momentum = todays_open_close_movement / prev_open_close_movement
    
    # Combine Adjustments
    final_factor = volume_adjusted_momentum * open_price_diff_momentum
    
    return final_factor
```
