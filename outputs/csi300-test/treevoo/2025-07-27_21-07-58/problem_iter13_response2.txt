import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Calculate High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Short-Term Momentum
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Momentum
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate Average True Range for Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x[0] - x[1], abs(x[0] - df['close'].shift(1)), abs(x[1] - df['close'].shift(1))), axis=1
    )
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Adjust by Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_ma_10'] = df['volume_change'].rolling(window=10).mean()
    df['average_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume_10']
    
    # Adjust by Amount
    df['average_amount_10'] = df['amount'].rolling(window=10).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount_10']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    factor = (
        (df['daily_return'].rolling(window=5).mean() * df['high_low_range'])  # Multiply 5-Day Moving Average of Price Change by High-to-Low Range
        * df['volume_ma_10']  # Further multiply by 10-Day Moving Average of Volume Change
        * (df['short_term_momentum'] + df['long_term_momentum'])  # Sum of Short-Term and Long-Term Momentum
        * (1 / df['average_true_range'])  # Inverse of Average True Range
        * df['volume_ratio']  # Volume Ratio
        * df['amount_ratio']  # Amount Ratio
    )
    
    return factor
