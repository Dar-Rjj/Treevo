import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['price_momentum_5d'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['volume_change_10d'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_volume_surge'] = df['price_momentum_5d'] * df['volume_change_10d']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['positive_returns_sum'] = df[df['daily_return'] > 0]['daily_return'].rolling(window=21).sum().fillna(0)
    
    # Volume Adjusted Momentum
    df['cumulative_volume'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['positive_returns_sum'] * df['cumulative_volume']) / 21
    
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['high_low_range_change'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Volume Surge Component
    df['volume_change_pct'] = df['volume_change'] / df['volume'].shift(1)
    df['volume_surge_5d'] = df['volume_change_pct'].rolling(window=5).mean()
    
    # Combine High-Low Range and Volume
    df['high_low_volume_combo'] = (df['high_low_range_change'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 21
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Adjust by Average Amount
    m = 21
    df['average_amount'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount']
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.1, 0.2, 0.2
    
    # Combine Weighted Components
    df['weighted_high_low_volume_combo'] = w1 * df['high_low_volume_combo']
    df['weighted_price_momentum_5d'] = w2 * df['price_momentum_5d']
    df['weighted_volume_surge_5d'] = w3 * df['volume_surge_5d']
    df['weighted_volume_ratio'] = w4 * df['volume_ratio']
    df['weighted_amount_ratio'] = w5 * df['amount_ratio']
    
    df['combined_weighted_components'] = (
        df['weighted_high_low_volume_combo'] +
        df['weighted_price_momentum_5d'] +
        df['weighted_volume_surge_5d'] +
        df['weighted_volume_ratio'] +
        df['weighted_amount_ratio']
    )
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_momentum'] + df['combined_weighted_components'] + (df['close'] - df['close'].shift(1)) * df['volume_ratio']
    
    return df['alpha_factor']
