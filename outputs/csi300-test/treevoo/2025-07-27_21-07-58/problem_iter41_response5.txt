```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()
    
    # Compute 5-Day Moving Average of Price Momentum
    df['MA5_Price_Momentum'] = df['Price_Momentum'].rolling(window=5).mean()
    
    # Compute 10-Day Moving Average Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    df['MA10_Volume_Change'] = df['Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Combined_Momentum_Volume'] = df['MA5_Price_Momentum'] * df['MA10_Volume_Change']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Combine High-Low Range and Volume
    df['Combined_HL_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['Positive_Returns'] = df['Daily_Return'].apply(lambda x: x if x > 0 else 0)
    df['MA21_Positive_Returns'] = df['Positive_Returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['Cumulative_Volume'] = df['volume'].rolling(window=21).sum()
    df['Volume_Adjusted_Momentum'] = (df['MA21_Positive_Returns'] * df['Cumulative_Volume']) / 21
    
    # Adjust by Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=21).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Adjust by Average Amount
    df['Average_Amount'] = df['amount'].rolling(window=21).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.1, 0.2, 0.2
    
    # Combine Weighted Components
    df['Weighted_Combined_HL_Volume'] = df['Combined_HL_Volume'] * w1
    df['Weighted_Combined_Momentum_Volume'] = df['Combined_Momentum_Volume'] * w2
    df['Weighted_Volume_Surge'] = (df['Volume_Change'] * w3) * df['MA10_Volume_Change']
    df['Weighted_Volume_Ratio'] = df['Volume_Ratio'] * w4
    df['Weighted_Amount_Ratio'] = df['Amount_Ratio'] * w5
    
    # Sum all weighted components
    df['Combined_Weighted_Components'] = (
        df['Weighted_Combined_HL_Volume'] +
        df['Weighted_Combined_Momentum_Volume'] +
        df['Weighted_Volume_Surge'] +
        df['Weighted_Volume_Ratio'] +
        df['Weighted_Amount_Ratio']
    )
    
    # Final Combined Alpha Factor
    df['Integrated_HL_Volume'] = (df['High_Low_Range'] / df['volume']).rolling(window=5).mean() * df['volume'].rolling(window=5).mean()
    df['Final_Alpha_Factor'] = (
        df['Volume_Adjusted_Momentum'] +
        df['Combined_Weighted_Components'] +
        df['Integrated_HL_Volume'] +
        (df['Price_Momentum'] * df['Volume_Ratio'])
    )
    
    return df['Final_Alpha_Factor']
```
