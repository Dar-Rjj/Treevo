```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Price Momentum Component
    df['price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Surge Component
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine High-Low Range and Volume
    df['high_low_range_volume'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Combine Components
    df['price_momentum_vol_surge'] = df['price_change_pct'] * df['volume_change_pct']
    df['alpha_factor'] = df['high_low_range_volume'] + (df['price_momentum_vol_surge'] * df['volume_ratio'])
    
    return df['alpha_factor']
```
