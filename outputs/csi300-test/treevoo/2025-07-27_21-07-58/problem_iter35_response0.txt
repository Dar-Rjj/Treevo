```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term and Medium-Term Price Momentum
    df['short_term_momentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['medium_term_momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # 5-Day Moving Average of Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['5_day_avg_price_change'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Volume-Weighted Price Actions
    df['weighted_avg_hl'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['5_day_avg_weighted_hl'] = df['weighted_avg_hl'].rolling(window=5).mean()
    df['cumulative_vol_adj_return'] = (df['close'] - df['open']) * df['volume']
    df['cumulative_vol_adj_return'] = df['cumulative_vol_adj_return'].rolling(window=10, min_periods=1).sum()
    
    # Trading Volume Patterns
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['10_day_avg_volume_change'] = df['daily_volume_change'].rolling(window=10).mean()
    
    # Volatility and Range
    df['daily_price_range'] = df['high'] - df['low']
    df['relative_daily_range'] = df['daily_price_range'] / df['close'].shift(1)
    df['std_22_days'] = df['close'].rolling(window=22).std()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_vol_surge'] = df['5_day_avg_price_change'] * df['10_day_avg_volume_change']
    
    # Adjust by Average Volume
    n = 10
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_adjustment'] = df['volume'] / df['avg_volume']
    
    # Adjust by Average Amount
    m = 10
    df['avg_amount'] = df['amount'].rolling(window=m).mean()
    df['amount_adjustment'] = df['amount'] / df['avg_amount']
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.2, 0.2, 0.1
    df['high_low_range_vol_combo'] = w1 * (df['5_day_avg_weighted_hl'] * df['5_day_avg_volume_change'])
    df['price_momentum_component'] = w2 * df['momentum_vol_surge']
    df['volume_surge_component'] = w3 * df['10_day_avg_volume_change']
    df['volume_ratio_component'] = w4 * df['volume_ratio']
    df['amount_ratio_component'] = w5 * df['amount_adjustment']
    
    # Combine Weighted Components
    df['combined_factor'] = (df['high_low_range_vol_combo'] + 
                             df['price_momentum_component'] + 
                             df['volume_surge_component'] + 
                             df['volume_ratio_component'] + 
                             df['amount_ratio_component'])
    
    # Incorporate High-Low Range Dynamics
    df['5_day_avg_high_low_range'] = df['daily_price_range'].rolling(window=5).mean()
    df['change_in_high_low_range'] = df['daily_price_range'] - df['daily_price_range'].shift(1)
    df['5_day_avg_change_high_low_range'] = df['change_in_high_low_range'].rolling(window=5).mean()
    df['high_low_range_dynamics'] = df['5_day_avg_high_low_range'] * df['5_day_avg_change_high_low_range']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = (df['combined_factor'] + 
                                df['high_low_range_dynamics'] + 
                                (df['short_term_momentum'] * df['volume_ratio']))
    
    return df['final_alpha_factor']
```
