```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum_1'] = df['close'].pct_change()
    
    # Compute 5-Day Moving Average of Price Momentum
    df['Price_Momentum_5_MA'] = df['Price_Momentum_1'].rolling(window=5).mean()
    
    # Compute 20-Day Moving Average Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Change_20_MA'] = df['Volume_Change'].rolling(window=20).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Momentum_Volume_Surge'] = df['Price_Momentum_5_MA'] * df['Volume_Change_20_MA']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Combine High-Low Range and Volume
    df['High_Low_Volume_Adjusted'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['Combined_Adjusted_Momentum'] = df['Price_Momentum_1'] * df['Volume_Ratio']
    df['Adjusted_Momentum_Volume_Amount'] = df['Combined_Adjusted_Momentum'] + df['Momentum_Volume_Surge'] + df['High_Low_Volume_Adjusted']
    
    # Incorporate High-Low Range Dynamics
    df['High_Low_Range_10_MA'] = df['High_Low_Range'].rolling(window=10).mean()
    df['Change_High_Low_Range_10_MA'] = df['Change_High_Low_Range'].rolling(window=10).mean()
    df['High_Low_Dynamics'] = df['High_Low_Range_10_MA'] * df['Change_High_Low_Range_10_MA']
    
    # Integrate High-Low Range with Volume
    df['High_Low_Range_by_Volume'] = df['High_Low_Range'] / df['volume']
    df['High_Low_Range_by_Volume_10_MA'] = df['High_Low_Range_by_Volume'].rolling(window=10).mean()
    df['Volume_10_MA'] = df['volume'].rolling(window=10).mean()
    df['Integrated_High_Low_Volume'] = df['High_Low_Range_by_Volume_10_MA'] * df['Volume_10_MA']
    
    # Include Long-Term Trend Indicator
    df['EMA_Close_60'] = df['close'].ewm(span=60).mean()
    df['EMA_Volume_60'] = df['volume'].ewm(span=60).mean()
    df['Long_Term_Trend'] = (df['close'] - df['EMA_Close_60']) + (df['volume'] - df['EMA_Volume_60'])
    
    # Final Combined Alpha Factor
    df['Final_Alpha_Factor'] = df['Adjusted_Momentum_Volume_Amount'] + df['High_Low_Dynamics'] + df['Integrated_High_Low_Volume'] + df['Long_Term_Trend']
    
    return df['Final_Alpha_Factor']
```
