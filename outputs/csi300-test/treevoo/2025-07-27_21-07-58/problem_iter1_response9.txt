```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['range'] = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday Price Range
    df['prev_range'] = df['range'].shift(1)
    
    # Calculate Intraday Price Range Change
    df['range_change'] = df['range'] - df['prev_range']
    
    # Volume Momentum Indicator
    df['volume_momentum'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Indicators
    df['alpha_factor'] = df['range_change'] * df['volume_momentum']
    
    return df['alpha_factor']
```
