```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Factor1: (Close_t - Open_t) / Open_t * Volume_t
    factor1 = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Calculate Factor2: (High_t - Max(High_{t-5},...,High_{t-1})) / Max(High_{t-5},...,High_{t-1})
    high_roll_max = df['high'].rolling(window=6).max().shift(1)
    factor2 = (df['high'] - high_roll_max) / high_roll_max
    
    # Calculate Factor3: (Low_t - Min(Low_{t-5},...,Low_{t-1})) / Min(Low_{t-5},...,Low_{t-1})
    low_roll_min = df['low'].rolling(window=6).min().shift(1)
    factor3 = (df['low'] - low_roll_min) / low_roll_min
    
    # Calculate Factor4: Amount_t / Average(Amount_{t-9},...,Amount_{t-1})
    amount_roll_avg = df['amount'].rolling(window=10).mean().shift(1)
    factor4 = df['amount'] / amount_roll_avg
    
    # Calculate Factor5: (Close_t - Min(Low_{t-20},...,Low_{t-1})) / (Max(High_{t-20},...,High_{t-1}) - Min(Low_{t-20},...,Low_{t-1}))
    high_20_max = df['high'].rolling(window=20).max().shift(1)
    low_20_min = df['low'].rolling(window=20).min().shift(1)
    factor5 = (df['close'] - low_20_min) / (high_20_max - low_20_min)
    
    # Calculate Factor6: Std(Close_{t-20},...,Close_{t-1})
    close_roll_std = df['close'].rolling(window=20).std()
    factor6 = close_roll_std
    
    # Combine the factors to create a composite alpha factor
    w1, w2, w3, w4, w5, w6 = 1, 1, 1, 1, 1, 1  # Example weights
    composite_factor = w1 * factor1 + w2 * factor2 + w3 * factor3 + w4 * factor4 + w5 * factor5 + w6 * factor6
    
    return composite_factor
```
