```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_in_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Calculate Price Momentum Component
    df['Daily_Price_Change_Percentage'] = df['close'].pct_change()
    df['Smoothed_Price_Momentum'] = df['Daily_Price_Change_Percentage'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Surge Component
    df['Daily_Volume_Change_Percentage'] = df['volume'].pct_change()
    df['Volume_Surge_Component'] = df['Daily_Volume_Change_Percentage'].rolling(window=10).mean()
    
    # Combine High-Low Range and Volume
    df['High_Low_Volume_Combined'] = (df['Change_in_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 20
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Adjust by Average Amount
    m = 20
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Combine Components
    df['Combined_Factor'] = (df['Smoothed_Price_Momentum'] * df['Volume_Surge_Component']) + df['High_Low_Volume_Combined']
    df['Final_Factor'] = df['Combined_Factor'] * df['Adjusted_Momentum']
    
    return df['Final_Factor']
```
