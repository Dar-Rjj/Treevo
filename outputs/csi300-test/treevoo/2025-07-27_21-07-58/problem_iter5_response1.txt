```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Previous Day High-to-Low Range
    df['prev_high_to_low_range'] = df['high_to_low_range'].shift(1)
    
    # Compute High-to-Low Range Momentum
    df['range_momentum'] = df['high_to_low_range'] / df['prev_high_to_low_range']
    
    # Adjust Price Change by Volume
    df['volume_adjusted_price_change'] = df['daily_price_change'] / df['volume']
    
    # Calculate 5-Day Moving Average of Price Change
    df['5_day_ma_price_change'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate 10-Day Moving Average of Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    df['10_day_ma_volume_change'] = df['daily_volume_change'].rolling(window=10).mean()
    
    # Combine Momentum and Volume Factors
    df['alpha_factor'] = (df['5_day_ma_price_change'] 
                          * df['range_momentum'] 
                          * df['10_day_ma_volume_change'])
    
    return df['alpha_factor']
```
