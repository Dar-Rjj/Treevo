```python
import pandas as pd

def heuristics_v2(df):
    """
    This function generates a novel and interpretable alpha factor based on the given market data.
    The alpha factor is calculated using a combination of price and volume information.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A series representing the factor values indexed by (date).
    """

    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume-weighted typical price
    vwt_price = typical_price * df['volume']
    
    # Calculate the smoothed volume-weighted typical price
    smoothed_vwt_price = vwt_price.ewm(span=14, adjust=False).mean()
    
    # Calculate the volatility of the asset
    volatility = df['close'].rolling(window=20).std()
    
    # Generate the final alpha factor
    alpha_factor = (smoothed_vwt_price - typical_price) / volatility
    
    return alpha_factor
```
