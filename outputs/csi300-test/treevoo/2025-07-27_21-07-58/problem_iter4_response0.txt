```python
import pandas as pd

def heuristics_v2(data):
    # Compute Daily Price Change Percentage
    daily_price_change = (data['close'] - data['close'].shift(1)) / data['close'].shift(1)
    
    # Compute 5-Day Moving Average of Price Change
    price_momentum = daily_price_change.rolling(window=5).mean()
    
    # Compute Daily Volume Change Percentage
    daily_volume_change = (data['volume'] - data['volume'].shift(1)) / data['volume'].shift(1)
    
    # Compute 10-Day Moving Average of Volume Change
    volume_surge = daily_volume_change.rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    alpha_factor = price_momentum * volume_surge
    
    return alpha_factor
```
