```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Price Change Percentage
    df['price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['price_momentum_5d'] = df['price_change_pct'].rolling(window=5).mean()
    
    # Compute Daily Volume Change Percentage
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['volume_surge_10d'] = df['volume_change_pct'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['combined_momentum_surge'] = df['price_momentum_5d'] * df['volume_surge_10d']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['moving_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['cumulative_volume'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['moving_sum_positive_returns'] * df['cumulative_volume']) / 21
    
    # Calculate 1-Period Price Momentum
    df['1_period_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Adjust by Average Volume
    n = 21
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Combine Momentum and Volume Adjustment
    df['adjusted_momentum'] = df['1_period_price_momentum'] * df['volume_ratio'] + df['volume_adjusted_momentum']
    
    # Final Synthesized Momentum Indicator
    df['synthesized_momentum_indicator'] = df['combined_momentum_surge'] + df['adjusted_momentum']
    
    return df['synthesized_momentum_indicator']
```
