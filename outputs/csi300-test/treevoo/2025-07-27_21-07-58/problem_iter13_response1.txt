import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['5_day_price_momentum'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['10_day_volume_change'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_volume_factor'] = (df['5_day_price_momentum'] * df['10_day_volume_change'])
    
    # Calculate Average Volume over n periods
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    
    # Volume Ratio
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = (df['5_day_price_momentum'] * df['10_day_volume_change']) + \
                         (df['price_change'] * df['volume_ratio'])
    
    return df['alpha_factor'].dropna()
