```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['short_term_momentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['medium_term_momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Identify Volume-Weighted Price Actions
    df['weighted_avg_hl'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['5day_moving_avg_weighted_avg_hl'] = df['weighted_avg_hl'].rolling(window=5).mean()
    
    df['cumulative_vol_adj_return'] = (df['close'] - df['open']) * df['volume']
    df['cumulative_vol_adj_return'] = df['cumulative_vol_adj_return'].rolling(window=10).sum()
    
    # Analyze Trading Volume Patterns
    df['ratio_to_5day_avg_volume'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['trend_in_volume'] = (df['volume'] - df['volume'].shift(5)) / 5
    
    # Evaluate Volatility and Range
    df['daily_price_range'] = df['high'] - df['low']
    df['5day_moving_avg_daily_price_range'] = df['daily_price_range'].rolling(window=5).mean()
    
    # Incorporate Additional Price and Volume Dynamics
    df['relative_daily_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['std_dev_close_20_days'] = df['close'].rolling(window=20).std()
    
    df['vol_weighted_close_price'] = df['close'] * df['volume']
    df['5day_moving_avg_vol_weighted_close_price'] = df['vol_weighted_close_price'].rolling(window=5).mean()
    
    avg_5day_volume = df['volume'].rolling(window=5).mean()
    df['price_change_with_volume_influence'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / avg_5day_volume)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['short_term_momentum'] +
        df['medium_term_momentum'] +
        df['5day_moving_avg_weighted_avg_hl'] +
        df['cumulative_vol_adj_return'] +
        df['ratio_to_5day_avg_volume'] +
        df['trend_in_volume'] +
        df['5day_moving_avg_daily_price_range'] +
        df['relative_daily_range'] +
        df['std_dev_close_20_days'] +
        df['5day_moving_avg_vol_weighted_close_price'] +
        df['price_change_with_volume_influence']
    )
    
    return alpha_factor.dropna()
```
