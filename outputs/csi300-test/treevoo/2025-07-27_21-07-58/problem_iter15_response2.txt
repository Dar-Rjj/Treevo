```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute Average Price Momentum over 5 Days
    df['price_momentum_5d'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_in_high_low_range'] = df['high_low_range'].diff()
    
    # Calculate Price Momentum Component
    df['price_momentum_component'] = (df['close'] / df['close'].shift(1) - 1)
    
    # Calculate Volume Surge Component
    df['volume_surge_component'] = (df['volume'] / df['volume'].shift(1) - 1)
    
    # Combine High-Low Range and Volume
    df['hl_volume_combo'] = (df['change_in_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 5  # number of periods for average volume
    df['average_volume_n'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume_n']
    
    # Adjust by Average Amount
    m = 5  # number of periods for average amount
    df['average_amount_m'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount_m']
    
    # Introduce New Component: Price-Volume Divergence
    df['price_volume_divergence'] = df['daily_price_change'] - df['volume_surge_component']
    
    # Combine Components
    df['combined_components'] = (df['price_momentum_component'] * df['volume_surge_component']) + df['hl_volume_combo']
    df['adjusted_combined_components'] = df['combined_components'] * df['volume_ratio'] * df['amount_ratio']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = (df['price_momentum_5d'] * df['volume_surge_component'].rolling(window=10).mean()) + \
                         df['adjusted_combined_components'] + \
                         df['price_volume_divergence']
    
    return df['alpha_factor']
```
