import pandas as pd

def heuristics_v2(df):
    # Thought 1.1: Calculate the daily price range as (High - Low)
    df['price_range'] = df['high'] - df['low']
    
    # Thought 1.2: Calculate the daily return based on Close-to-Close
    df['close_to_close_return'] = df['close'].pct_change()
    
    # Thought 1.3: Calculate the daily return based on Open-to-Close
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Thought 1.4: Calculate the average volume over the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Thought 2.1: Calculate the volume-to-price ratio as (Volume[t] / Close[t])
    df['volume_to_price_ratio'] = df['volume'] / df['close']
    
    # Thought 2.2: Calculate the average volume-to-price ratio over the last 5 days
    df['avg_volume_to_price_ratio_5d'] = df['volume_to_price_ratio'].rolling(window=5).mean()
    
    # Thought 3.1: Calculate the intraday momentum as (Close[t] - Open[t])
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Thought 3.2: Calculate the average intraday momentum over the last 5 days
    df['avg_intraday_momentum_5d'] = df['intraday_momentum'].rolling(window=5).mean()
    
    # Thought 4.1: Calculate the gap between today's Open and yesterday's Close
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Thought 4.2: Calculate the average gap over the last 5 days
    df['avg_gap_5d'] = df['gap'].rolling(window=5).mean()
    
    # Thought 5.1: Create a composite factor using a weighted sum of the above factors
    # Assign weights to each factor based on their perceived importance
    weights = {
        'price_range': 0.1,
        'close_to_close_return': 0.2,
        'open_to_close_return': 0.1,
        'avg_volume_5d': 0.1,
        'volume_to_price_ratio': 0.1,
        'avg_volume_to_price_ratio_5d': 0.1,
        'intraday_momentum': 0.1,
        'avg_intraday_momentum_5d': 0.1,
        'gap': 0.05,
        'avg_gap_5d': 0.05
    }
    
    # Sum the weighted factors to create a single composite factor
    df['composite_factor'] = sum(df[col] * weight for col, weight in weights.items())
    
    return df['composite_factor']
