import pandas as pd

def heuristics_v2(df):
    # Determine Daily Returns
    df['daily_returns'] = df['close'].pct_change()
    
    # Compute Short-Term and Long-Term Momentum
    df['5_day_momentum'] = df['daily_returns'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    df['20_day_momentum'] = df['daily_returns'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate Average True Range
    df['true_range'] = df[['high', 'low']].diff(axis=1).max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Combine Momentum and Volatility
    df['weighted_momentum'] = (df['5_day_momentum'] * 0.7 + df['20_day_momentum'] * 0.3) / df['atr_14']
    
    # Adjust by Volume and Amount
    n = 20  # Number of periods for volume average
    m = 20  # Number of periods for amount average
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['avg_amount'] = df['amount'].rolling(window=m).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    df['amount_ratio'] = df['amount'] / df['avg_amount']
    df['adjusted_momentum'] = df['weighted_momentum'] * df['volume_ratio'] * df['amount_ratio']
    
    # Add Trend Following Component
    df['50_day_sma'] = df['close'].rolling(window=50).mean()
    trend_multiplier = df['close'] > df['50_day_sma']
    df['trend_factor'] = trend_multiplier.map({True: 1.25, False: 1})
    df['trend_adjusted_momentum'] = df['adjusted_momentum'] * df['trend_factor']
    
    # Incorporate Price Oscillator
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['price_oscillator'] = df['ema_12'] - df['ema_26']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['trend_adjusted_momentum'] + df['price_oscillator']
    
    return df['alpha_factor']
