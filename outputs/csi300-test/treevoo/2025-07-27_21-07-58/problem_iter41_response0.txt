import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the ratio of high to low
    df['high_low_ratio'] = df['high'] / df['low']
    
    # Calculate the ratio of close to open
    df['close_open_ratio'] = df['close'] / df['open']
    
    # Calculate the volume weighted average price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=30).mean()
    
    # Calculate the simple moving average (SMA) of the close price for a 30-day window
    df['sma_30'] = df['close'].rolling(window=30).mean()
    
    # Calculate the relative strength index (RSI) for a 14-day window
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Alpha factor: Combining various heuristics
    df['alpha_factor'] = (
        df['close_pct_change'] * 0.2 +
        (df['high_low_ratio'] - 1) * 0.15 +
        (df['close_open_ratio'] - 1) * 0.15 +
        (df['vwap'] - df['close']) / df['close'] * 0.1 +
        (df['sma_30'] - df['close']) / df['close'] * 0.2 +
        (50 - df['rsi_14']) / 50 * 0.2
    )
    
    return df['alpha_factor'].dropna()
