```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()

    # Compute Average Price Momentum over 5 Days
    df['5D_Price_Momentum'] = df['Price_Momentum'].rolling(window=5).mean()

    # Compute Average Volume Change over 10 Days
    df['Volume_Change'] = df['volume'].pct_change()
    df['10D_Volume_Change'] = df['Volume_Change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['Momentum_Volume_Surge'] = df['5D_Price_Momentum'] * df['10D_Volume_Change']

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()

    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']

    # Adjust by Average Volume
    df['Average_Volume_10D'] = df['volume'].rolling(window=10).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_10D']

    # Adjust by Average Amount
    df['Average_Amount_10D'] = df['amount'].rolling(window=10).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount_10D']

    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    df['Combined_Adjusted_Momentum'] = df['Adjusted_Momentum'] + df['Momentum_Volume_Surge']

    # Incorporate High-Low Range Dynamics
    df['5D_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['5D_Change_High_Low_Range'] = df['Change_High_Low_Range'].rolling(window=5).mean()
    df['Combined_High_Low_Dynamics'] = df['5D_High_Low_Range'] * df['5D_Change_High_Low_Range']

    # Final Combined Alpha Factor
    df['Alpha_Factor'] = df['Combined_Adjusted_Momentum'] + df['Combined_High_Low_Dynamics']

    return df['Alpha_Factor']
```
