```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['close'] / df['open'] - 1
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average Volume
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate Volume Weighted Intraday Return
    vol_weighted_intraday_return = intraday_return * (df['volume'] / avg_volume)
    
    # Calculate High-Low Range Volatility
    high_low_volatility = high_low_range.rolling(window=10).std()
    
    # Combine Factors
    alpha_factor = vol_weighted_intraday_return - high_low_volatility
    
    return alpha_factor
```
