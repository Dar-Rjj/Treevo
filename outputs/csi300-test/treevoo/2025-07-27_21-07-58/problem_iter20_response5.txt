import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    high_low_range_change = high_low_range - high_low_range.shift(1)
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Positive Returns
    positive_returns = daily_return[daily_return > 0]
    moving_avg_positive_returns = positive_returns.rolling(window=21, min_periods=1).mean()
    
    # Volume Adjusted Momentum
    cumulative_volume = df['volume'].rolling(window=21, min_periods=1).sum()
    volume_adjusted_momentum = (moving_avg_positive_returns * cumulative_volume) / 21
    
    # Calculate Short-Term and Long-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Average True Range
    true_range = np.maximum(np.maximum(df['high'] - df['low'], 
                                       abs(df['high'] - df['close'].shift(1))), 
                            abs(df['low'] - df['close'].shift(1)))
    average_true_range = true_range.rolling(window=14, min_periods=1).mean()
    
    # Combine High-Low Range and Volume
    high_low_range_vol_adj = (high_low_range_change * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=21, min_periods=1).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=21, min_periods=1).mean()
    amount_ratio = df['amount'] / avg_amount
    
    # Introduce New Component: Price-Volume Divergence
    daily_volume_change_percentage = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    price_volume_divergence = daily_return - daily_volume_change_percentage
    
    # Combine Components
    combined_momentum = (short_term_momentum * volume_ratio * amount_ratio 
                         + volume_adjusted_momentum 
                         + high_low_range_vol_adj)
    
    # Weighted Sum of Momentum
    weighted_sum_momentum = (short_term_momentum * 0.6 + long_term_momentum * 0.4)
    
    # Apply Trend Multiplier
    trend_multiplier = 1.5 if df['close'] > df['close'].ewm(span=50, adjust=False).mean() else 1.0
    combined_momentum *= trend_multiplier
    
    # Final Combined Alpha Factor
    final_alpha_factor = (combined_momentum 
                          + volume_adjusted_momentum 
                          + price_volume_divergence 
                          * weighted_sum_momentum)
    
    return final_alpha_factor
