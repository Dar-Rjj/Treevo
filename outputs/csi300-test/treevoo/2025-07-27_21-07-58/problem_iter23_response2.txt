```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Positive Returns
    positive_returns = daily_return.where(daily_return > 0, 0)
    moving_avg_positive_returns = positive_returns.rolling(window=21).mean()
    
    # Volume Adjusted Momentum
    cumulative_volume = df['volume'].rolling(window=21).sum()
    volume_adjusted_momentum = (moving_avg_positive_returns * cumulative_volume) / 21
    
    # Calculate 5-Period Price Momentum
    price_momentum_5 = df['close'] - df['close'].shift(5)
    
    # Combine High-Low Range and Volume
    high_low_range_volume = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=21).mean()
    amount_ratio = df['amount'] / avg_amount
    
    # Introduce New Component: Price-Volume Divergence
    daily_volume_change_pct = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    price_volume_divergence = daily_return - daily_volume_change_pct
    
    # Assign Weights
    w1 = 0.2
    w2 = 0.3
    w3 = 0.1
    w4 = 0.1
    w5 = 0.1
    
    # Combine Weighted Components
    combined_alpha = (w1 * high_low_range_volume +
                      w2 * price_momentum_5 +
                      w3 * volume_adjusted_momentum +
                      w4 * volume_ratio +
                      w5 * amount_ratio)
    
    # Calculate 21-Day Price Volatility
    price_volatility_21 = df['close'].rolling(window=21).std()
    
    # Adjust by Price Volatility
    volatility_adjustment = np.where(price_volatility_21 > 0.01, 1.75, 1.25)
    
    # Introduce New Component: Relative Strength Index (RSI)
    gain = daily_return.where(daily_return > 0, 0)
    loss = -daily_return.where(daily_return < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rsi = 100 - (100 / (1 + (avg_gain / avg_loss)))
    
    # Final Combined Alpha Factor
    final_alpha_factor = (combined_alpha + price_volume_divergence) * volatility_adjustment * rsi
    
    return final_alpha_factor
```
