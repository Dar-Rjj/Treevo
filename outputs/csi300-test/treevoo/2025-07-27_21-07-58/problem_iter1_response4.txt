```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Incorporate Volume Momentum
    volume_momentum = (df['close'] - df['open']) * df['volume']
    
    # Combine Factors
    combined_factor = (intraday_range + close_to_open_ratio + volume_momentum) / 3
    
    return combined_factor
```
