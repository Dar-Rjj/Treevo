```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 5-Day Average Price Momentum
    df['5_day_momentum'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute 10-Day Average Volume Change
    df['10_day_avg_volume_change'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_volume_surge'] = df['5_day_momentum'] * df['10_day_avg_volume_change']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute 21-Day Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['21_day_positive_returns_sum'] = df['positive_returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['21_day_cumulative_volume'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['21_day_positive_returns_sum'] * df['21_day_cumulative_volume']) / 21
    
    # Compute High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Daily Change in High-Low Range
    df['daily_high_low_change'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['high_low_volume_combo'] = (df['daily_high_low_change'] * df['volume']) / df['close']
    
    # Introduce Weighting Factors
    w1, w2, w3 = 0.4, 0.3, 0.3  # Example weights, can be adjusted
    df['weighted_high_low_volume'] = df['high_low_volume_combo'] * w1
    df['weighted_5_day_momentum'] = df['5_day_momentum'] * w2
    df['weighted_10_day_volume_change'] = df['10_day_avg_volume_change'] * w3
    
    # Combine Weighted Components
    df['combined_weighted_factors'] = (df['weighted_high_low_volume'] + 
                                       df['weighted_5_day_momentum'] + 
                                       df['weighted_10_day_volume_change'])
    
    # Final Combined Alpha Factor
    df['1_period_price_momentum'] = df['close'] - df['close'].shift(1)
    df['21_day_avg_volume'] = df['volume'].rolling(window=21).mean()
    df['adjusted_by_21_day_avg_volume'] = (df['volume'] / df['21_day_avg_volume'])
    df['final_alpha_factor'] = (df['combined_weighted_factors'] + 
                                df['volume_adjusted_momentum'] +
                                df['1_period_price_momentum'] * df['adjusted_by_21_day_avg_volume'])
    
    return df['final_alpha_factor']
```
