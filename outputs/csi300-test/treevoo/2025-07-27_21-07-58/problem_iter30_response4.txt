```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple daily return
    df['simple_daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate cumulative 5-day return
    df['cumulative_5day_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Average True Range over 14 days
    df['true_range'] = df[['high' - 'low', 
                           abs('high' - df['close'].shift(1)), 
                           abs('low' - df['close'].shift(1))]].max(axis=1)
    df['average_true_range_14'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate change in daily trading volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate volume weighted by price movement
    df['volume_weighted_price_movement'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate price momentum with volume
    df['price_momentum_with_volume'] = (df['close'] - df['open']) * (df['volume'] / df['volume'].shift(1))
    
    # Construct the final alpha factor
    # Here, we are using a simple combination of the generated factors. You can adjust the weights or add more sophisticated logic.
    df['alpha_factor'] = (
        0.3 * df['simple_daily_return'] + 
        0.2 * df['cumulative_5day_return'] + 
        0.1 * df['daily_range'] + 
        0.1 * df['average_true_range_14'] + 
        0.1 * df['volume_change'] + 
        0.1 * df['volume_weighted_price_movement'] + 
        0.1 * df['price_momentum_with_volume']
    )
    
    return df['alpha_factor']
```
