```python
def heuristics_v2(df):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']
    
    # Volume Momentum
    volume_momentum = (df['close'] - df['open']) * df['volume']
    
    # Final Factor: Sum the three factors
    final_factor = intraday_range + close_to_open_ratio + volume_momentum
    
    return final_factor
```
