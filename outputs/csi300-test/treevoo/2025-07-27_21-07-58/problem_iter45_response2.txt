```python
import pandas as pd

def heuristics_v2(df):
    # Factor1: (Close_t - Open_t) / Open_t * Volume_t
    factor1 = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Factor2: (High_t - Average(High_{t-5},...,High_{t-1})) / Average(High_{t-5},...,High_{t-1})
    high_avg_5 = df['high'].rolling(window=5).mean().shift(1)
    factor2 = (df['high'] - high_avg_5) / high_avg_5
    
    # Factor3: (Low_t - Average(Low_{t-5},...,Low_{t-1})) / Average(Low_{t-5},...,Low_{t-1})
    low_avg_5 = df['low'].rolling(window=5).mean().shift(1)
    factor3 = (df['low'] - low_avg_5) / low_avg_5
    
    # Factor4: Amount_t / Average(Amount_{t-10},...,Amount_{t-1})
    amount_avg_10 = df['amount'].rolling(window=10).mean()
    factor4 = df['amount'] / amount_avg_10
    
    # Factor5: (Close_t - Min(Low_{t-20},...,Low_{t-1})) / (Max(High_{t-20},...,High_{t-1}) - Min(Low_{t-20},...,Low_{t-1}))
    low_min_20 = df['low'].rolling(window=20).min()
    high_max_20 = df['high'].rolling(window=20).max()
    factor5 = (df['close'] - low_min_20) / (high_max_20 - low_min_20)
    
    # Factor6: StdDev(Close_{t-20},...,Close_{t-1})
    factor6 = df['close'].rolling(window=20).std()
    
    # Composite_Factor: w1 * Factor1 + w2 * Factor2 + w3 * Factor3 + w4 * Factor4 + w5 * Factor5 + w6 * Factor6
    w1, w2, w3, w4, w5, w6 = 1, 1, 1, 1, 1, 1  # Equal weights for simplicity
    composite_factor = (w1 * factor1 + w2 * factor2 + w3 * factor3 + w4 * factor4 + w5 * factor5 + w6 * factor6) / 6
    
    # Normalize the composite factor
    composite_factor = (composite_factor - composite_factor.mean()) / composite_factor.std()
    
    return composite_factor
```
