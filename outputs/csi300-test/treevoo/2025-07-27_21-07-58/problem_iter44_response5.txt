```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Price Momentum
    df['price_diff'] = df['close'] - df['close'].shift(1)
    df['5_day_momentum'] = df['price_diff'].rolling(window=5).mean()
    
    # Trend Strength
    df['positive_change'] = (df['price_diff'] > 0).astype(int)
    df['10_day_positive_count'] = df['positive_change'].rolling(window=10).sum()
    
    # Price Volatility
    df['daily_return'] = df['close'].pct_change()
    df['20_day_volatility'] = df['daily_return'].rolling(window=20).std()
    
    # Volume Trends
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['10_day_avg_volume']
    
    # Amount Metrics
    df['5_day_amount_sum'] = df['amount'].rolling(window=5).sum()
    df['amount_ratio'] = df['amount'] / df['5_day_amount_sum']
    
    # EMA Indicators
    df['10_day_ema_close'] = df['close'].ewm(span=10, adjust=False).mean()
    df['10_day_ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['ema_ratio'] = df['10_day_ema_close'] / df['10_day_ema_volume']
    
    # High and Low Prices
    df['range'] = df['high'] - df['low']
    df['5_day_avg_range'] = df['range'].rolling(window=5).mean()
    df['range_ratio'] = df['range'] / df['5_day_avg_range']
    
    # Combine Momentum and Volatility
    df['momentum_volatility'] = df['5_day_momentum'] * df['20_day_volatility']
    
    # Trading Activity
    df['open_close_diff'] = abs(df['open'] - df['close'])
    df['5_day_open_close_sum'] = df['open_close_diff'].rolling(window=5).sum()
    df['activity_ratio'] = df['open_close_diff'] / df['5_day_open_close_sum']
    
    # Final alpha factor
    df['alpha_factor'] = (
        df['5_day_momentum'] +
        df['10_day_positive_count'] +
        df['20_day_volatility'] +
        df['volume_ratio'] +
        df['amount_ratio'] +
        df['ema_ratio'] +
        df['range_ratio'] +
        df['momentum_volatility'] +
        df['activity_ratio']
    )
    
    return df['alpha_factor'].dropna()
```
