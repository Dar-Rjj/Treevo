```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'].diff()
    
    # Compute High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Previous Day High-to-Low Range
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    
    # Compute High-to-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'] / df['prev_high_low_range']
    
    # Compute 5-Day and 20-Day Moving Average of Price Change
    df['5_day_ma_price_change'] = df['price_change'].rolling(window=5).mean()
    df['20_day_ma_price_change'] = df['price_change'].rolling(window=20).mean()
    
    # Determine Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Short-Term and Long-Term Momentum
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate True Range for Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2].shift(1)), abs(x[1] - x[2].shift(1))), axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Adjust by Volume
    df['10_day_ma_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['10_day_ma_volume']
    
    # Adjust by Amount
    df['7_day_ma_amount'] = df['amount'].rolling(window=7).mean()
    df['amount_ratio'] = df['amount'] / df['7_day_ma_amount']
    
    # Introduce High-to-Low Range Momentum Adjustment
    df['volume_adjusted_high_low_range_momentum'] = df['high_low_range_momentum'] * df['10_day_ma_volume']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    df['alpha_factor'] = (
        df['5_day_ma_price_change'] * 
        df['volume_adjusted_high_low_range_momentum'] * 
        df['10_day_ma_volume'] * 
        (df['short_term_momentum'] + df['long_term_momentum']) * 
        (1 / df['average_true_range']) * 
        df['volume_ratio'] * 
        df['amount_ratio']
    )
    
    # Enhance Signal with Price Change and Volume Ratio
    df['alpha_factor'] *= df['price_change'] * df['volume_ratio']
    
    return df['alpha_factor']
```
