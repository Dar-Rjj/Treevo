import pandas as pd

def heuristics_v2(df):
    # Calculate individual factors
    df['factor1'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['max_high_5'] = df['high'].rolling(window=5).max()
    df['min_low_5'] = df['low'].rolling(window=5).min()
    df['factor2'] = (df['high'] - df['max_high_5']) / df['max_high_5']
    df['factor3'] = (df['low'] - df['min_low_5']) / df['min_low_5']
    df['avg_amount_10'] = df['amount'].rolling(window=10).mean()
    df['factor4'] = df['amount'] / df['avg_amount_10']
    df['max_high_20'] = df['high'].rolling(window=20).max()
    df['min_low_20'] = df['low'].rolling(window=20).min()
    df['factor5'] = (df['close'] - df['min_low_20']) / (df['max_high_20'] - df['min_low_20'])
    df['sum_close_5'] = df['close'].rolling(window=5).sum()
    df['factor6'] = df['sum_close_5'] * df['volume']

    # Weights for each factor
    w1, w2, w3, w4, w5, w6 = 0.1, 0.15, 0.15, 0.15, 0.2, 0.25

    # Composite alpha factor
    df['composite_factor'] = (w1 * df['factor1'] + 
                              w2 * df['factor2'] + 
                              w3 * df['factor3'] + 
                              w4 * df['factor4'] + 
                              w5 * df['factor5'] + 
                              w6 * df['factor6'])

    return df['composite_factor'].dropna()
