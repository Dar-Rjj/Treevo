import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['price_momentum_5d'] = df['price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['volume_surge_10d'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_volume_combined'] = df['price_momentum_5d'] * df['volume_surge_10d']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['moving_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['cumulative_volume_21d'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['moving_sum_positive_returns'] * df['cumulative_volume_21d']) / 21
    
    # Compute High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Previous Day High-to-Low Price Range
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    
    # Calculate High-to-Low Range Momentum
    df['high_low_range_momentum'] = df['high_low_range'] / df['prev_high_low_range']
    
    # Adjust by Volume
    df['high_low_range_vol_adj'] = df['high_low_range_momentum'] / df['volume']
    
    # Incorporate Open Price Difference
    df['today_open_close_movement'] = df['close'] - df['open']
    df['prev_day_open_close_movement'] = df['close'].shift(1) - df['open'].shift(1)
    
    # Calculate Open Price Difference Momentum
    df['open_price_diff_momentum'] = df['today_open_close_movement'] / df['prev_day_open_close_movement']
    
    # Combine Adjustments
    df['combined_adjustments'] = df['high_low_range_vol_adj'] * df['open_price_diff_momentum']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['volume_adjusted_momentum'] + df['combined_adjustments']
    
    # Multiply 1-Period Price Momentum by Volume Ratio
    df['price_momentum_1d'] = df['close'] - df['close'].shift(1)
    df['average_volume'] = df['volume'].rolling(window=21).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    df['price_momentum_vol_adj'] = df['price_momentum_1d'] * df['volume_ratio']
    
    # Final Combined Alpha Factor with Volume-Adjusted 1-Period Price Momentum
    df['final_alpha_factor'] += df['price_momentum_vol_adj']
    
    return df['final_alpha_factor'].dropna()
