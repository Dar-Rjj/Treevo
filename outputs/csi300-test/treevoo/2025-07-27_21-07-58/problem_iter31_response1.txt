```python
import pandas as pd

def heuristics_v2(df):
    # Sub-sub-sub-thought: Factor = (Close_t - Open_t) / Open_t * Volume_t
    price_change_impact = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Sub-sub-sub-thought: Factor = (High_t - Average(High_{t-5},...,High_{t-1})) / Average(High_{t-5},...,High_{t-1})
    high_trend = (df['high'] - df['high'].rolling(window=5).mean().shift(1)) / df['high'].rolling(window=5).mean().shift(1)
    
    # Sub-sub-sub-thought: Factor = (Low_t - Average(Low_{t-5},...,Low_{t-1})) / Average(Low_{t-5},...,Low_{t-1})
    low_trend = (df['low'] - df['low'].rolling(window=5).mean().shift(1)) / df['low'].rolling(window=5).mean().shift(1)
    
    # Sub-sub-sub-thought: Factor = Amount_t / Average(Amount_{t-9},...,Amount_{t-1})
    amount_ratio = df['amount'] / df['amount'].rolling(window=10).mean().shift(1)
    
    # Sub-sub-sub-thought: Factor = (Close_t - Min(Low_{t-20},...,Low_{t-1})) / (Max(High_{t-20},...,High_{t-1}) - Min(Low_{t-20},...,Low_{t-1}))
    relative_position = (df['close'] - df['low'].rolling(window=20).min()) / (df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min())
    
    # Combine all factors into a single series
    combined_factor = (price_change_impact + high_trend.fillna(0) + low_trend.fillna(0) + amount_ratio + relative_position) / 5
    
    return combined_factor
```
