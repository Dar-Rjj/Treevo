import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Calculate Price Momentum Component
    df['Daily_Price_Change_Percentage'] = df['close'].pct_change()
    df['Avg_Price_Momentum_5D'] = df['Daily_Price_Change_Percentage'].rolling(window=5).mean()
    
    # Calculate Volume Surge Component
    df['Daily_Volume_Change_Percentage'] = df['volume'].pct_change()
    df['Avg_Volume_Change_10D'] = df['Daily_Volume_Change_Percentage'].rolling(window=10).mean()
    
    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 20  # Number of periods for average volume
    df['Avg_Volume_n'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_n']
    
    # Adjust by Average Amount
    m = 20  # Number of periods for average amount
    df['Avg_Amount_m'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Avg_Amount_m']
    
    # Introduce Weighting Factors
    w1 = 0.3  # Weight for Combined High-Low Range and Volume
    w2 = 0.2  # Weight for Price Momentum Component
    w3 = 0.2  # Weight for Volume Surge Component
    w4 = 0.15  # Weight for Volume Ratio
    w5 = 0.15  # Weight for Amount Ratio
    
    # Combine Weighted Components
    df['Alpha_Factor'] = (
        w1 * df['Combined_High_Low_Volume'] +
        w2 * df['Avg_Price_Momentum_5D'] +
        w3 * df['Avg_Volume_Change_10D'] +
        w4 * df['Volume_Ratio'] +
        w5 * df['Amount_Ratio']
    )
    
    # Return the alpha factor as a pandas Series
    return df['Alpha_Factor']
