import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow, which is the typical price multiplied by the volume traded
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day average of the money flow
    average_money_flow_14 = money_flow.rolling(window=14).mean()
    
    # Calculate the 14-day EMA of the close price for the market sentiment
    ema_close_14 = df['close'].ewm(span=14, adjust=False).mean()
    
    # Heuristic: The factor will be a combination of the relative strength of money flow and close price
    alpha_factor = (money_flow - average_money_flow_14) / (df['close'] - ema_close_14)
    
    return alpha_factor
