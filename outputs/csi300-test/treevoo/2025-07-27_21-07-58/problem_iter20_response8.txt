```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Compute Average Price Momentum over 5 Days
    df['avg_price_momentum_5d'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'].diff()
    
    # Calculate Price Momentum Component
    df['price_momentum_component'] = (df['close'].pct_change() * 100)
    
    # Calculate Volume Surge Component
    df['volume_surge_component'] = (df['volume'].pct_change() * 100)
    
    # Combine High-Low Range and Volume
    df['high_low_volume_combination'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 10
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Adjust by Average Amount
    m = 10
    df['avg_amount'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount']
    
    # Introduce New Component: Price-Volume Divergence
    df['price_volume_divergence'] = df['daily_price_change'] - df['volume'].diff()
    
    # Combine Components
    df['combined_components'] = (df['price_momentum_component'] * df['volume_surge_component']) + df['high_low_volume_combination']
    
    # Multiply 1-Period Price Momentum by Volume Ratio
    df['price_momentum_volume_ratio'] = df['price_momentum_component'] * df['volume_ratio']
    
    # Multiply Result by Amount Ratio
    df['price_momentum_volume_amount_adjustment'] = df['price_momentum_volume_ratio'] * df['amount_ratio']
    
    # Final Combined Alpha Factor
    df['final_combined_alpha_factor'] = (df['avg_price_momentum_5d'] * df['volume'].rolling(window=10).mean()) + df['combined_components'] + df['price_volume_divergence']
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    
    # Compute 14-Day Simple Moving Average of True Range
    df['sma_true_range_14d'] = df['true_range'].rolling(window=14).mean()
    
    # Combine Momentum, Volatility, and High-Low Range Components
    df['combined_momentum_volatility_high_low'] = (df['avg_price_momentum_5d'] + df['price_momentum_volume_amount_adjustment']) * (1 / df['sma_true_range_14d'])
    
    # Calculate Daily Log Return
    df['daily_log_return'] = np.log(df['close']).diff()
    
    # Compute 5-Day Exponential Moving Average of Log Return
    df['ema_log_return_5d'] = df['daily_log_return'].ewm(span=5, adjust=False).mean()
    
    # Combine Log Momentum and Volume Surge
    df['log_momentum_volume_surge'] = df['ema_log_return_5d'] * df['volume'].rolling(window=10).mean()
    
    # Combine All Components
    df['final_alpha_factor'] = (df['avg_price_momentum_5d'] * df['volume'].rolling(window=10).mean()) + df['log_momentum_volume_surge'] + df['combined_momentum_volatility_high_low'] + df['final_combined_alpha_factor']
    df['final_alpha_factor'] *= df['amount_ratio']
    
    return df['final_alpha_factor'].dropna()
```
