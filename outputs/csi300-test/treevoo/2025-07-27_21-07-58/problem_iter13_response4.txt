```python
import pandas as pd

def heuristics_v2(df, n=5, m=5, w1=0.3, w2=0.2, w3=0.1, w4=0.2, w5=0.2):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate Price Momentum Component
    price_change_pct = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Surge Component
    volume_change_pct = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine High-Low Range and Volume
    high_low_volume = (change_high_low_range * df['volume']) / df['close']
    
    # Compare with Recent Volume
    avg_volume_n = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / avg_volume_n
    
    # Compare with Recent Amount
    avg_amount_m = df['amount'].rolling(window=m).mean()
    amount_ratio = df['amount'] / avg_amount_m
    
    # Introduce Weighting Factors
    alpha_factor = (w1 * high_low_volume) + \
                   (w2 * price_change_pct) + \
                   (w3 * volume_change_pct) + \
                   (w4 * volume_ratio) + \
                   (w5 * amount_ratio)
    
    return alpha_factor
```
