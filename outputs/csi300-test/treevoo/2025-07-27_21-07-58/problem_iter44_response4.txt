import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Initialize the DataFrame
    df = df.copy()
    
    # Factor: (Close_t - Open_t) / Open_t * Volume_t
    df['price_change_volume'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    # Trend in Highs
    df['max_high_5'] = df['high'].rolling(window=5).max().shift(1)
    df['min_low_5'] = df['low'].rolling(window=5).min().shift(1)
    df['trend_high'] = (df['high'] - df['max_high_5']) / (df['max_high_5'] - df['min_low_5'])
    
    # Trend in Lows
    df['trend_low'] = (df['low'] - df['min_low_5']) / (df['max_high_5'] - df['min_low_5'])
    
    # Average amount transacted
    df['avg_amount_9'] = df['amount'].rolling(window=9).mean().shift(1)
    df['amount_factor'] = df['amount'] / df['avg_amount_9']
    
    # Momentum in closing prices relative to recent high and low
    df['min_low_20'] = df['low'].rolling(window=20).min().shift(1)
    df['max_high_20'] = df['high'].rolling(window=20).max().shift(1)
    df['momentum_close'] = (df['close'] - df['min_low_20']) / (df['max_high_20'] - df['min_low_20'])
    
    # Volatility of the stock
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    
    # 10-Day Moving Average of High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['ma_high_low_range_10'] = df['high_low_range'].rolling(window=10).mean()
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    df['ma_change_high_low_range_10'] = df['change_high_low_range'].rolling(window=10).mean()
    df['high_low_dynamics'] = df['ma_high_low_range_10'] * df['ma_change_high_low_range_10']
    
    # 10-Day Moving Average of High-Low Range divided by Volume
    df['ma_high_low_range_div_volume_10'] = (df['high_low_range'] / df['volume']).rolling(window=10).mean()
    df['ma_volume_10'] = df['volume'].rolling(window=10).mean()
    df['high_low_range_vol'] = df['ma_high_low_range_div_volume_10'] * df['ma_volume_10']
    
    # 60-Day Exponential Moving Average of Close Price and Volume
    df['ema_close_60'] = df['close'].ewm(span=60, adjust=False).mean()
    df['ema_volume_60'] = df['volume'].ewm(span=60, adjust=False).mean()
    df['long_term_trend'] = (df['close'] - df['ema_close_60']) + (df['volume'] - df['ema_volume_60'])
    
    # Composite factor
    w1, w2, w3, w4, w5, w6 = 1, 1, 1, 1, 1, 1
    df['composite_factor'] = (
        w1 * df['price_change_volume'] +
        w2 * df['trend_high'] +
        w3 * df['trend_low'] +
        w4 * df['amount_factor'] +
        w5 * df['momentum_close'] +
        w6 * df['volatility'] +
        df['high_low_dynamics'] +
        df['high_low_range_vol'] +
        df['long_term_trend']
    )
    
    # Return the alpha factor
    return df['composite_factor']
