import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum_1D'] = df['close'].pct_change()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Calculate Daily Price Change Percentage
    df['Daily_Price_Change_Pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['Avg_Price_Momentum_5D'] = df['Daily_Price_Change_Pct'].rolling(window=5).mean()
    
    # Compute Daily Volume Change Percentage
    df['Daily_Volume_Change_Pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['Avg_Volume_Change_10D'] = df['Daily_Volume_Change_Pct'].rolling(window=10).mean()
    
    # Combine High-Low Range and Volume
    df['HLV_Combined'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['Avg_Volume_n'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_n']
    
    # Adjust by Average Amount
    df['Avg_Amount_m'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Avg_Amount_m']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum_Factor'] = (
        df['Price_Momentum_1D'] * 
        df['Volume_Ratio'] * 
        df['Amount_Ratio']
    )
    
    # Combine Components
    df['Alpha_Factor'] = (
        df['Avg_Price_Momentum_5D'] * 
        df['Avg_Volume_Change_10D'] + 
        df['HLV_Combined'] +
        df['Adjusted_Momentum_Factor']
    )
    
    return df['Alpha_Factor']
