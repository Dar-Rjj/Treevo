```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Change
    df['price_change_1d'] = df['close'].diff()
    
    # Compute Weighted Average Price Momentum over 5 Days
    weights = np.arange(1, 6)
    df['weighted_momentum_5d'] = df['price_change_1d'].rolling(window=5).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['high_low_range_change'] = df['high_low_range'].diff()
    
    # Calculate Daily Volume Change
    df['volume_change_1d'] = df['volume'].diff()
    
    # Compute Average Volume Change over 10 Days
    df['avg_volume_change_10d'] = df['volume_change_1d'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['momentum_volume'] = df['weighted_momentum_5d'] * df['avg_volume_change_10d']
    
    # Adjust by Exponential Moving Average of Volume
    df['ema_volume'] = df['volume'].ewm(span=10, adjust=False).mean()
    df['volume_ratio'] = df['volume'] / df['ema_volume']
    
    # Calculate Momentum Indicator
    df['daily_returns'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_returns'].rolling(window=5).sum()
    df['long_term_momentum'] = df['daily_returns'].rolling(window=20).sum()
    
    # Calculate Average True Range
    df['true_range'] = df[['high', 'low']].max(axis=1) - df[['high', 'low']].min(axis=1)
    df['true_range'] = df[['true_range', df['high'] - df['close'].shift(1), abs(df['low'] - df['close'].shift(1))]].max(axis=1)
    df['atr_14d'] = df['true_range'].rolling(window=14).mean()
    
    # Combine Momentum and Volatility
    df['combined_momentum'] = df['short_term_momentum'] * 0.6 + df['long_term_momentum'] * 0.4
    df['trend_multiplier'] = np.where(df['close'] > df['close'].ewm(span=50, adjust=False).mean(), 1.5, 1.0)
    df['adjusted_momentum'] = df['combined_momentum'] * df['trend_multiplier']
    
    # Combine High-Low Range and Volume
    df['high_low_volume'] = df['high_low_range_change'] * df['volume'] / df['close']
    
    # Calculate 21-Day Price Volatility
    df['price_volatility_21d'] = df['close'].rolling(window=21).std()
    
    # Introduce New Component: Price-Volume Divergence
    df['daily_volume_change_pct'] = df['volume'].pct_change()
    df['price_volume_divergence'] = df['daily_returns'] - df['daily_volume_change_pct']
    
    # Introduce New Component: Price-Amount Divergence
    df['daily_amount_change_pct'] = df['amount'].pct_change()
    df['price_amount_divergence'] = df['daily_returns'] - df['daily_amount_change_pct']
    
    # Adjust by Price Volatility
    df['volatility_adjustment'] = np.where(df['price_volatility_21d'] > df['price_volatility_21d'].quantile(0.75), 1.5, 1.0)
    
    # Final Combined Alpha Factor
    df['amount_ratio'] = df['amount'] / df['amount'].rolling(window=10).mean()
    df['alpha_factor'] = (df['weighted_momentum_5d'] * df['avg_volume_change_10d']) + (df['price_change_1d'] * df['volume_ratio']) + \
                         df['price_volume_divergence'] + df['price_amount_divergence']
    df['alpha_factor'] = df['alpha_factor'] * df['volatility_adjustment'] * df['amount_ratio']
    
    return df['alpha_factor']
```
