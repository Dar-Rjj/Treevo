```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()

    # Compute 5-Day Moving Average of Price Momentum
    df['MA5_Price_Momentum'] = df['Price_Momentum'].rolling(window=5).mean()

    # Compute 10-Day Moving Average Volume Change
    df['Volume_Change'] = df['volume'].pct_change()
    df['MA10_Volume_Change'] = df['Volume_Change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['Combined_Momentum_Volume'] = df['MA5_Price_Momentum'] * df['MA10_Volume_Change']

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()

    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']

    # Combine Adjusted Momentum, Volume, and Amount
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['Combined_Adjusted_Momentum_Volume_Amount'] = df['Price_Momentum'] * df['Volume_Ratio'] + df['Combined_Momentum_Volume'] + df['Combined_High_Low_Volume']

    # Incorporate High-Low Range Dynamics
    df['MA5_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['MA5_Change_High_Low_Range'] = df['Change_High_Low_Range'].rolling(window=5).mean()
    df['Combined_High_Low_Range_Dynamics'] = df['MA5_High_Low_Range'] * df['MA5_Change_High_Low_Range']

    # Integrate High-Low Range with Volume
    df['MA5_High_Low_Range_Div_Volume'] = (df['High_Low_Range'] / df['volume']).rolling(window=5).mean()
    df['MA5_Volume'] = df['volume'].rolling(window=5).mean()
    df['Integrated_High_Low_Range_Volume'] = df['MA5_High_Low_Range_Div_Volume'] * df['MA5_Volume']

    # Final Combined Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Adjusted_Momentum_Volume_Amount'] + df['Combined_High_Low_Range_Dynamics'] + df['Integrated_High_Low_Range_Volume']

    return df['Final_Alpha_Factor']
```
