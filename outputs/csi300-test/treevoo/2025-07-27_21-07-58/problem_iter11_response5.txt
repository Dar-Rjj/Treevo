```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Close_t-1'] = df['close'].shift(1)
    df['Price_Momentum_1D'] = df['close'] - df['Close_t-1']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['High_Low_Range_t-1'] = df['High_Low_Range'].shift(1)
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range_t-1']
    
    # Calculate Price Momentum Component
    df['Daily_Price_Change_Percentage'] = (df['close'] - df['Close_t-1']) / df['Close_t-1']
    df['Average_Price_Momentum_5D'] = df['Daily_Price_Change_Percentage'].rolling(window=5).mean()
    
    # Calculate Volume Surge Component
    df['Volume_t-1'] = df['volume'].shift(1)
    df['Daily_Volume_Change_Percentage'] = (df['volume'] - df['Volume_t-1']) / df['Volume_t-1']
    df['Average_Volume_Change_10D'] = df['Daily_Volume_Change_Percentage'].rolling(window=10).mean()
    
    # Combine High-Low Range and Volume
    df['HL_Vol_Adjusted'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 30  # Number of periods for average volume
    df['Average_Volume_n'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_n']
    
    # Adjust by Average Amount
    m = 30  # Number of periods for average amount
    df['Average_Amount_m'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount_m']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum_Factor'] = df['Price_Momentum_1D'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Combine Components
    df['Combined_Factor'] = (df['Average_Price_Momentum_5D'] * df['Average_Volume_Change_10D'] + 
                             df['HL_Vol_Adjusted'] + 
                             df['Adjusted_Momentum_Factor'])
    
    # Output as Alpha Factor
    alpha_factor = df['Combined_Factor']
    
    return alpha_factor
```
