import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor using the given (open, high, low, close) prices, amount, and volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series representing the factor values indexed by date.
    """
    # Calculate the daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the rolling window average of the daily range over N days (e.g., 20 days)
    N = 20
    df['rolling_avg_range'] = df['daily_range'].rolling(window=N).mean()
    
    # Calculate the percentage change from open to close each day
    df['pct_change_open_close'] = (df['close'] - df['open']) / df['open'] * 100
    
    # Create a moving average of the daily percentage changes (e.g., 10 days)
    M = 10
    df['ma_pct_change'] = df['pct_change_open_close'].rolling(window=M).mean()
    
    # Compute the ratio of today’s volume to yesterday’s volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Calculate the gap between the previous day's close and the current day's open
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Identify days with significant gaps (both positive and negative)
    significant_gap_threshold = 0.05  # 5% threshold
    df['significant_gap'] = df['gap'].abs() > (df['close'].shift(1) * significant_gap_threshold)
    
    # Combine all factors into a single alpha factor
    # Example: Weighted sum of the factors
    alpha_factor = (
        0.4 * (df['rolling_avg_range'] - df['rolling_avg_range'].mean()) / df['rolling_avg_range'].std() +
        0.3 * (df['ma_pct_change'] - df['ma_pct_change'].mean()) / df['ma_pct_change'].std() +
        0.2 * (df['volume_ratio'] - df['volume_ratio'].mean()) / df['volume_ratio'].std() +
        0.1 * (df['gap'] - df['gap'].mean()) / df['gap'].std()
    )
    
    return alpha_factor.dropna()
