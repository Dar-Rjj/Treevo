```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['1_period_momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['5_day_moving_avg_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['high_low_volume_adjusted'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Introduce Weighting Factors
    w1 = 0.6  # Weight for High-Low Range and Volume Combination
    w2 = 0.4  # Weight for Price Momentum Component
    
    # Combine Weighted Components
    df['weighted_high_low_vol'] = df['high_low_volume_adjusted'] * w1
    df['weighted_momentum'] = df['5_day_moving_avg_momentum'] * w2
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['weighted_high_low_vol'] + df['weighted_momentum']
    
    # Multiply 1-Period Price Momentum by Volume Ratio
    average_volume = df['volume'].rolling(window=5).mean()
    df['volume_ratio'] = df['volume'] / average_volume
    df['adjusted_momentum'] = df['1_period_momentum'] * df['volume_ratio']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['final_alpha_factor'] += df['adjusted_momentum']
    
    return df['final_alpha_factor']
```
