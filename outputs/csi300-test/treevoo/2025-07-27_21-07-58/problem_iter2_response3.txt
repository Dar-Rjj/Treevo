```python
import pandas as pd

def heuristics_v2(df, window=5):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume
    df['intraday_volume'] = df['volume'] - df['volume'].shift(1)
    
    # Multiply Intraday Return by Intraday Volume
    df['intraday_vol_adj_momentum'] = df['intraday_return'] * df['intraday_volume']
    
    # Cumulative Sum Over a Window
    df['cum_intraday_vol_adj_momentum'] = df['intraday_vol_adj_momentum'].rolling(window=window).sum()
    
    # Adjust by Average Volume
    df['average_volume'] = df['volume'].rolling(window=window).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Combine Momentum and Volume Adjustment
    df['adjusted_momentum_factor'] = df['cum_intraday_vol_adj_momentum'] * df['volume_ratio']
    
    return df['adjusted_momentum_factor']
```
