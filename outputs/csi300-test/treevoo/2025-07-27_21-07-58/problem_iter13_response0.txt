import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Calculate Price Momentum Component
    df['Daily_Price_Change_Percentage'] = (df['close'].pct_change()) * 100
    
    # Calculate Volume Surge Component
    df['Daily_Volume_Change_Percentage'] = (df['volume'].pct_change()) * 100
    
    # Combine High-Low Range and Volume
    df['High_Low_Volume_Combined'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Adjust by Average Amount
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Combine Components
    df['Price_Momentum_Volume_Surge'] = df['Daily_Price_Change_Percentage'] * df['Daily_Volume_Change_Percentage']
    df['Alpha_Factor'] = df['High_Low_Volume_Combined'] + df['Price_Momentum_Volume_Surge']
    df['Alpha_Factor'] *= df['Volume_Ratio'] * df['Amount_Ratio']
    
    return df['Alpha_Factor']
