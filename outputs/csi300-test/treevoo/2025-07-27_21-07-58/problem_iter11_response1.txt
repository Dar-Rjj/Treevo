```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term Momentum (5-day sum of daily returns)
    df['returns'] = df['close'].pct_change()
    short_term_momentum = df['returns'].rolling(window=5).sum()

    # Compute Long-Term Momentum (20-day sum of daily returns)
    long_term_momentum = df['returns'].rolling(window=20).sum()

    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)

    # Calculate 14-day Simple Moving Average of True Range
    average_true_range = df['true_range'].rolling(window=14).mean()

    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range.diff()

    # Calculate Daily Price Change Percentage
    price_change_percentage = df['close'].pct_change()

    # Smooth Price Momentum with EMA
    price_momentum = price_change_percentage.ewm(span=10, adjust=False).mean()

    # Calculate Average Volume over n periods
    n = 10
    avg_volume = df['volume'].rolling(window=n).mean()

    # Compare with Recent Volume
    volume_ratio = df['volume'] / avg_volume

    # Calculate Volume Surge Component
    volume_change_percentage = df['volume'].pct_change()
    volume_surge = volume_change_percentage.ewm(span=10, adjust=False).mean()

    # Combine High-Low Range and Volume
    combined_high_low_volume = (change_high_low_range * df['volume']) / df['close']

    # Calculate Average Amount over m periods
    m = 10
    avg_amount = df['amount'].rolling(window=m).mean()

    # Compare with Recent Amount
    amount_ratio = df['amount'] / avg_amount

    # Combine Indicators
    indicator_sum = short_term_momentum + long_term_momentum
    adjusted_indicator = indicator_sum / average_true_range
    final_factor = (adjusted_indicator * volume_ratio * amount_ratio) + (price_momentum * volume_surge) + combined_high_low_volume

    return final_factor
```
