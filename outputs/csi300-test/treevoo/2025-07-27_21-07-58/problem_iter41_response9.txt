```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute 5-Day Moving Average of Price Momentum
    df['MA5_Price_Momentum'] = df['Price_Momentum'].rolling(window=5).mean()
    
    # Compute 10-Day Moving Average Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['MA10_Volume_Change'] = df['Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Price_Vol_Surge'] = df['MA5_Price_Momentum'] * df['MA10_Volume_Change']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['High_Low_Vol'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Final Combined Alpha Factor
    df['Alpha_Factor'] = df['Price_Vol_Surge'] + df['High_Low_Vol']
    
    return df['Alpha_Factor']
```
