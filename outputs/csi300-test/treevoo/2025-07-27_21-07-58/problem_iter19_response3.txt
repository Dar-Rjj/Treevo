```python
import pandas as pd

def heuristics_v2(df, n=5, m=10):
    # Calculate 1-Period Price Momentum
    price_momentum = df['close'] / df['close'].shift(1)
    
    # Compute Daily Price Change Percentage
    daily_price_change_pct = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    avg_price_momentum_5d = daily_price_change_pct.rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    avg_volume_change_10d = daily_volume_change.rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    combined_price_vol = avg_price_momentum_5d * avg_volume_change_10d
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Combine High-Low Range and Volume
    combined_high_low_vol = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=m).mean()
    amount_ratio = df['amount'] / avg_amount
    
    # Combine Adjusted Momentum, Volume, and Amount
    adjusted_momentum = price_momentum * volume_ratio * amount_ratio
    
    # Final Combined Alpha Factor
    final_alpha_factor = combined_price_vol + adjusted_momentum + combined_high_low_vol
    
    return final_alpha_factor
```
