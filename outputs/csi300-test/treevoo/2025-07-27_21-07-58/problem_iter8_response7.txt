```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate High-to-Low Range
    df['hl_range'] = df['high'] - df['low']
    df['prev_hl_range'] = df['hl_range'].shift(1)
    
    # Compute High-to-Low Range Momentum
    df['hl_momentum'] = df['hl_range'] / df['prev_hl_range']
    
    # Adjust Price Change by Volume
    df['volume_adjusted_price_change'] = df['price_change'] / df['volume']
    
    # Determine Daily Returns
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute Short-Term Momentum
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Momentum
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate Average True Range for Volatility
    df['true_range'] = df[['high' - 'low', 
                           abs('high' - df['close'].shift(1)), 
                           abs('low' - df['close'].shift(1))]].max(axis=1)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Adjust by Volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_10d'] = df['volume_change'].rolling(window=10).mean()
    df['avg_volume_n'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_n']
    
    # Adjust by Amount
    df['avg_amount_10d'] = df['amount'].rolling(window=10).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_10d']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    factor = (df['price_change'].rolling(window=5).mean() * df['hl_momentum'] 
              * df['avg_volume_10d'] * (df['short_term_momentum'] + df['long_term_momentum']) 
              * (1 / df['atr']) * df['volume_ratio'] * df['amount_ratio'])
    
    return factor
```
