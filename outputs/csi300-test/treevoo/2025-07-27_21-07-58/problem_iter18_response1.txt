```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['5_Day_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_in_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Calculate Price Momentum Component
    df['Price_Momentum_Component'] = df['Daily_Price_Change'] / df['close'].shift(1)
    
    # Calculate Volume Surge Component
    df['Volume_Surge_Component'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine High-Low Range and Volume
    df['High_Low_Volume_Combination'] = (df['Change_in_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 10  # Number of periods for average volume
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Adjust by Average Amount
    m = 10  # Number of periods for average amount
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Introduce New Component: Price-Volume Divergence
    df['Price_Volume_Divergence'] = df['Daily_Price_Change'] - (df['volume'] - df['volume'].shift(1))
    
    # Combine Components
    df['Combined_Components'] = (df['Price_Momentum_Component'] * df['Volume_Surge_Component']) + df['High_Low_Volume_Combination']
    df['Combined_Components_Adjusted'] = (df['Combined_Components'] * df['Volume_Ratio'] * df['Amount_Ratio'])
    
    # Add 5-Day Moving Average Price Momentum * 10-Day Moving Average Volume Change to Combined Components
    df['Final_Combined_Alpha_Factor'] = df['5_Day_Price_Momentum'] * df['volume'].rolling(window=10).mean() + df['Combined_Components_Adjusted'] + df['Price_Volume_Divergence']
    
    return df['Final_Combined_Alpha_Factor']
```
