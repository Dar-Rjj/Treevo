```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change Percentage
    df['price_change_pct'] = df['close'].pct_change() * 100
    
    # Compute 5-Day Moving Average of Price Change
    df['5_day_mavg_price_change'] = df['price_change_pct'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()
    
    # Compute 10-Day Moving Average of Volume Change
    df['10_day_mavg_volume_change'] = df['volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['price_vol_momentum'] = df['5_day_mavg_price_change'] * df['10_day_mavg_volume_change']
    
    # Calculate 1-Period Price Momentum
    df['1_period_price_momentum'] = df['close'].pct_change() * 100
    
    # Adjust by Average Volume
    n = 20
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Combine Momentum and Volume Adjustment
    df['volume_adjusted_momentum'] = df['1_period_price_momentum'] * df['volume_ratio']
    
    # Calculate Cumulative Volume
    df['cumulative_volume'] = df['volume'].rolling(window=21).sum()
    
    # Calculate Moving Sum of Positive Returns
    df['positive_returns'] = df['close'].pct_change().apply(lambda x: x if x > 0 else 0)
    df['moving_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()
    
    df['volume_adjusted_momentum_2'] = (df['moving_sum_positive_returns'] * df['cumulative_volume']) / 21
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)), abs(x['low'] - df['close'].shift(1))), axis=1)
    
    # Compute 14-Day Simple Moving Average of True Range
    df['14_day_sma_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'].diff()
    
    # Combine High-Low Range and Volume
    df['hl_range_volume'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Amount
    m = 20
    df['avg_amount'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount']
    
    # Combine Momentum, Volatility, and High-Low Range Components
    df['combined_momentum_volatility'] = df['price_vol_momentum'] + df['volume_adjusted_momentum'] + df['volume_adjusted_momentum_2']
    df['combined_momentum_volatility'] = df['combined_momentum_volatility'] * (1 / df['14_day_sma_true_range'])
    df['combined_momentum_volatility'] = df['combined_momentum_volatility'] + df['hl_range_volume']
    df['combined_momentum_volatility'] = df['combined_momentum_volatility'] * df['amount_ratio']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['price_vol_momentum'] + df['volume_adjusted_momentum'] + df['combined_momentum_volatility']
    
    return df['final_alpha_factor']
```
