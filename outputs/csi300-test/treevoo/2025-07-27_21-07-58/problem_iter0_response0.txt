```python
import pandas as pd

def heuristics_v2(df):
    """
    The function calculates a novel alpha factor, which is the Volume Weighted Close-to-Open Return.
    
    Parameters:
    - df: A pandas DataFrame with (date) as index and market features (open, high, low, close, amount, volume) as columns.
    
    Returns:
    - A pandas Series indexed by (date) representing the factor values.
    """
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
