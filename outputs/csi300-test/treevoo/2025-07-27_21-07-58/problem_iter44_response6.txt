```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['price_diff'] = df['close'].diff()
    df['momentum_5d'] = df['price_diff'].rolling(window=5).mean()
    df['pct_change_close'] = df['close'].pct_change()

    # Evaluate Trend Strength
    df['positive_days_10d'] = (df['price_diff'] > 0).rolling(window=10).sum()
    df['pct_positive_10d'] = df['positive_days_10d'] / 10

    # Measure Price Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility_20d'] = df['daily_return'].rolling(window=20).std()
    df['avg_abs_daily_return_20d'] = df['daily_return'].abs().rolling(window=20).mean()

    # Assess Volume Trends
    df['volume_ma_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['volume_ma_10d']
    df['volume_change'] = df['volume'].diff()

    # Incorporate Amount Metrics
    df['amount_5d_sum'] = df['amount'].rolling(window=5).sum()
    df['amount_to_5d_sum_ratio'] = df['amount'] / df['amount_5d_sum']
    df['pct_change_amount'] = df['amount'].pct_change()

    # Analyze High and Low Prices
    df['daily_range'] = df['high'] - df['low']
    df['avg_daily_range_10d'] = df['daily_range'].rolling(window=10).mean()
    df['range_to_avg_ratio'] = df['daily_range'] / df['avg_daily_range_10d']

    # Explore Open and Close Price Relationships
    df['open_close_diff'] = df['open'] - df['close']
    df['open_close_diff_5d_avg'] = df['open_close_diff'].rolling(window=5).mean()
    df['positive_open_close_10d'] = (df['open_close_diff'] > 0).rolling(window=10).sum()

    # Combine Momentum and Volatility
    df['momentum_volatility_ratio'] = df['momentum_5d'] / df['volatility_20d']
    df['momentum_volatility_diff'] = df['momentum_5d'] - df['volatility_20d']

    # Integrate Volume and Amount Metrics
    df['volume_amount_product'] = df['volume'] * df['amount']
    df['volume_amount_product_5d_sum'] = df['volume_amount_product'].rolling(window=5).sum()
    df['volume_amount_ratio'] = df['volume_amount_product'] / df['volume_amount_product_5d_sum']

    # Generate the alpha factor
    alpha_factor = (
        df['momentum_volatility_ratio']
        + df['volume_ratio']
        + df['amount_to_5d_sum_ratio']
        + df['range_to_avg_ratio']
        + df['momentum_5d']
        + df['pct_positive_10d']
    )

    return alpha_factor
```
