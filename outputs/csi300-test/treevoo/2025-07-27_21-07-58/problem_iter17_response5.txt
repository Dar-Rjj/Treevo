```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()

    # Compute Daily Price Change Percentage
    df['Daily_Price_Change'] = df['close'].pct_change()

    # Compute Average Price Momentum over 5 Days
    df['5D_Avg_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).mean()

    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'].diff()

    # Compute Average Volume Change over 10 Days
    df['10D_Avg_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['Combined_Momentum_Volume'] = df['5D_Avg_Price_Momentum'] * df['10D_Avg_Volume_Change']

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()

    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']

    # Adjust by Average Volume
    n = 20  # Number of periods for average volume
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']

    # Adjust by Average Amount
    m = 20  # Number of periods for average amount
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']

    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    df['Combined_Adjusted_Momentum_Volume'] = df['Combined_Momentum_Volume'] + df['Adjusted_Momentum']

    # Incorporate High-Low Range Dynamics
    df['5D_Avg_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['5D_Avg_Change_High_Low_Range'] = df['Change_High_Low_Range'].rolling(window=5).mean()
    df['Combined_High_Low_Dynamics'] = df['5D_Avg_High_Low_Range'] * df['5D_Avg_Change_High_Low_Range']

    # Final Combined Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Momentum_Volume'] + df['Combined_Adjusted_Momentum_Volume'] + df['Combined_High_Low_Dynamics']

    return df['Final_Alpha_Factor']
```
