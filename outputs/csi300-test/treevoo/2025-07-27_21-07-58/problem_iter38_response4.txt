```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index as date representing the factor values.
    """
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the average of high and low for each day
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the difference between the close price and the average of high and low
    df['diff_close_avg_high_low'] = df['close'] - df['avg_high_low']
    
    # Calculate the percentage of volume relative to the rolling mean of volume
    df['rolling_volume_mean'] = df['volume'].rolling(window=30).mean()
    df['volume_ratio'] = df['volume'] / df['rolling_volume_mean']
    
    # Normalize the volume ratio
    df['norm_volume_ratio'] = (df['volume_ratio'] - df['volume_ratio'].mean()) / df['volume_ratio'].std()
    
    # Combine the factors into a single alpha factor
    df['alpha_factor'] = (df['pct_change_close'] * 0.5) + (df['diff_close_avg_high_low'] * 0.3) + (df['norm_volume_ratio'] * 0.2)
    
    return df['alpha_factor']
```
