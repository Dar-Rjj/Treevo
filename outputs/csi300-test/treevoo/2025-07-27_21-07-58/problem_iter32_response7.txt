```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the 5-day and 20-day moving average of the close price
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate the 14-day RSI (Relative Strength Index) of the close price
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate the 14-day ATR (Average True Range)
    true_range = df[['high', 'low']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14'] = true_range.rolling(window=14).mean()
    
    # Calculate the money flow ratio over a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    positive_money_flow_sum = positive_money_flow.rolling(window=14).sum()
    negative_money_flow_sum = negative_money_flow.rolling(window=14).sum()
    df['mfr_14'] = positive_money_flow_sum / negative_money_flow_sum
    
    # Calculate the alpha factor
    df['alpha_factor'] = (df['ma_5'] - df['ma_20']) / df['atr_14'] + (df['rsi_14'] - 50) / 50 + (df['mfr_14'] - 1)
    
    return df['alpha_factor']
```
