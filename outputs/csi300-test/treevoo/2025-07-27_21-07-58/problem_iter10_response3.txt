```python
import pandas as pd

def heuristics_v2(df):
    # Compute Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Compute True Range
    high_low = df['high'] - df['low']
    high_prev_close = df['high'] - df['close'].shift(1)
    prev_close_low = df['close'].shift(1) - df['low']
    true_range = pd.concat([high_low, high_prev_close, prev_close_low], axis=1).max(axis=1)
    
    # Compute Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Volume Adjusted Intraday Momentum
    volume_adjusted_intraday_momentum = intraday_momentum * ema_volume
    
    # True Range Volatility
    true_range_volatility = true_range.rolling(window=20).std()
    
    # Intraday Return
    intraday_return = df['close'] / df['open']
    
    # Final Alpha Factor
    alpha_factor = (volume_adjusted_intraday_momentum * intraday_return) - true_range_volatility
    
    return alpha_factor
```
