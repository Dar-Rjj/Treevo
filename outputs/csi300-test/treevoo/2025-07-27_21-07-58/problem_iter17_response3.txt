```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['1_period_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate 5-Period Price Momentum
    df['5_period_momentum'] = df['close'] - df['close'].shift(5)
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Positive Returns (21-day)
    df['positive_returns'] = df['daily_return'].where(df['daily_return'] > 0, 0)
    df['ma_positive_returns'] = df['positive_returns'].rolling(window=21).mean()
    
    # Volume Adjusted Momentum
    df['cumulative_volume'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['ma_positive_returns'] * df['cumulative_volume']) / 21
    
    # Combine High-Low Range and Volume
    df['combined_high_low_volume'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=21).mean()
    df['volume_ratio'] = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=21).mean()
    df['amount_ratio'] = df['amount'] / avg_amount
    
    # Introduce Weighting Factors
    w1 = 0.3
    w2 = 0.2
    w3 = 0.1
    w4 = 0.1
    w5 = 0.3
    
    # Combine Weighted Components
    df['weighted_combined_high_low_volume'] = w1 * df['combined_high_low_volume']
    df['weighted_price_momentum'] = w2 * (df['1_period_momentum'] + df['5_period_momentum'])
    df['weighted_volume_surge'] = w3 * (df['volume'] - df['volume'].shift(1))
    df['weighted_volume_ratio'] = w4 * df['volume_ratio']
    df['weighted_amount_ratio'] = w5 * df['amount_ratio']
    
    df['combined_components'] = (df['weighted_combined_high_low_volume'] +
                                df['weighted_price_momentum'] +
                                df['weighted_volume_surge'] +
                                df['weighted_volume_ratio'] +
                                df['weighted_amount_ratio'])
    
    # Calculate 21-Day Price Volatility
    df['price_volatility'] = df['close'].rolling(window=21).std()
    
    # Adjust by Price Volatility
    df['volatility_adjustment'] = df['price_volatility'].apply(lambda x: 1.5 if x > 0.01 else 1.0)
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = (df['volume_adjusted_momentum'] + df['combined_components']) * df['volatility_adjustment']
    
    return df['final_alpha_factor'].dropna()
```
