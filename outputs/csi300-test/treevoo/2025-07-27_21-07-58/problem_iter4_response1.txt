```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Returns using t / (t-1) Close price ratio
    df['daily_return'] = df['close'].pct_change()

    # Short-Term Momentum: 5-day sum of daily returns
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()

    # Long-Term Momentum: 20-day sum of daily returns
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum()

    # Determine True Range as Max[High - Low, Abs(High - Previous Close), Abs(Low - Previous Close)]
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], 
                                                                        abs(x['high'] - df['close'].shift(1)), 
                                                                        abs(x['low'] - df['close'].shift(1))), axis=1)

    # 14-day Simple Moving Average of True Range
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()

    # Combine Momentum and Volatility
    # Sum of Short-Term and Long-Term Momentum
    df['momentum_sum'] = df['short_term_momentum'] + df['long_term_momentum']

    # Multiply by Inverse of Average True Range
    df['alpha_factor'] = df['momentum_sum'] * (1 / df['avg_true_range'])

    return df['alpha_factor']
```
