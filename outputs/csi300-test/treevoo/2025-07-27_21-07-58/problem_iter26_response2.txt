import pandas as pd

def heuristics_v2(df):
    """
    Generates an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the 10-day moving average of the close price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between the high and low prices
    df['range'] = df['high'] - df['low']
    
    # Calculate the ratio of volume to amount
    df['vol_to_amt_ratio'] = df['volume'] / df['amount']
    
    # Generate the alpha factor
    df['alpha_factor'] = (df['pct_change_close'] * df['range'] * (df['ma_10_close'] - df['ma_30_close'])) / df['vol_to_amt_ratio']
    
    return df['alpha_factor']
