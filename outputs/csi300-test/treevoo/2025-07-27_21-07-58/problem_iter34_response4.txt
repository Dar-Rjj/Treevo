```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['Short_Term_Price_Momentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['Medium_Term_Price_Momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Identify Volume-Weighted Price Actions
    df['Weighted_Avg_H_L'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['5D_MA_Weighted_Avg_H_L'] = df['Weighted_Avg_H_L'].rolling(window=5).mean()
    df['Cum_Vol_Adj_Return'] = ((df['close'] - df['open']) * df['volume']).rolling(window=10).sum()
    
    # Analyze Trading Volume Patterns
    df['Ratio_Volume_5D_Avg'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['Trend_Volume'] = (df['volume'] - df['volume'].shift(5)) / 5
    
    # Evaluate Volatility and Range
    df['Daily_Price_Range'] = df['high'] - df['low']
    df['5D_MA_Daily_Price_Range'] = df['Daily_Price_Range'].rolling(window=5).mean()
    
    # Incorporate Additional Price and Volume Dynamics
    df['Relative_Daily_Range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['Std_Dev_Close_20D'] = df['close'].rolling(window=20).std()
    df['Vol_Weighted_Close'] = df['close'] * df['volume']
    df['5D_MA_Vol_Weighted_Close'] = df['Vol_Weighted_Close'].rolling(window=5).mean()
    df['Price_Change_Vol_Influence'] = (df['close'] - df['close'].shift(1)) * (df['volume'] / df['volume'].rolling(window=5).mean())
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['Short_Term_Price_Momentum'] 
        + df['Medium_Term_Price_Momentum'] 
        + df['5D_MA_Weighted_Avg_H_L'] 
        + df['Cum_Vol_Adj_Return'] 
        + df['Ratio_Volume_5D_Avg'] 
        + df['Trend_Volume'] 
        + df['5D_MA_Daily_Price_Range'] 
        + df['Relative_Daily_Range'] 
        + df['Std_Dev_Close_20D'] 
        + df['5D_MA_Vol_Weighted_Close'] 
        + df['Price_Change_Vol_Influence']
    )
    
    return alpha_factor
```
