```python
import pandas as pd

def heuristics_v2(df):
    # Analyze Price Patterns and Returns
    df['daily_price_range'] = df['high'] - df['low']
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['gap_open_close'] = df['open'] - df['close'].shift(1)

    # Explore Volume and Its Relationship with Price
    df['volume_to_price_ratio'] = df['volume'] / df['close']

    # Investigate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']

    # Analyze Price Gaps
    df['average_gap_5d'] = df['gap_open_close'].rolling(window=5).mean()

    # Compute 1-Period Price Momentum
    df['price_momentum_1d'] = df['close'] - df['close'].shift(1)

    # Compute 5-Day Moving Average of Price Momentum
    df['price_momentum_5d_avg'] = df['close_to_close_return'].rolling(window=5).mean()

    # Compute 10-Day Moving Average Volume Change
    df['volume_change_1d'] = df['volume'] - df['volume'].shift(1)
    df['volume_change_10d_avg'] = df['volume_change_1d'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['combined_price_vol'] = df['price_momentum_5d_avg'] * df['volume_change_10d_avg']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)

    # Combine High-Low Range and Volume
    df['combined_high_low_vol'] = (df['change_high_low_range'] * df['volume']) / df['close']

    # Combine Adjusted Momentum, Volume, and Amount
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['adjusted_momentum'] = df['price_momentum_1d'] * df['volume_ratio']
    df['combined_factors'] = df['adjusted_momentum'] + df['combined_price_vol'] + df['combined_high_low_vol']

    # Incorporate High-Low Range Dynamics
    df['high_low_range_5d_avg'] = df['high_low_range'].rolling(window=5).mean()
    df['change_high_low_range_5d_avg'] = df['change_high_low_range'].rolling(window=5).mean()
    df['high_low_dynamics'] = df['high_low_range_5d_avg'] * df['change_high_low_range_5d_avg']

    # Create a Composite Factor
    df['composite_factor'] = (
        0.3 * df['combined_factors'] +
        0.4 * df['high_low_dynamics'] +
        0.2 * df['average_gap_5d']
    )

    return df['composite_factor'].dropna()
```
