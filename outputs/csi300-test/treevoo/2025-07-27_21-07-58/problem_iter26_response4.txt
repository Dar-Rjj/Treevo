```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by date representing the factor values.
    """
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    
    # Calculate the money flow ratio
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(daily_return > 0, 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(daily_return < 0, 0).rolling(window=14).sum()
    money_flow_ratio = positive_money_flow / negative_money_flow
    
    # Calculate the price momentum
    price_momentum = df['close'] / df['close'].shift(5) - 1
    
    # Combine the factors into a single alpha factor
    alpha_factor = (money_flow_ratio * price_momentum).rolling(window=10).mean()
    
    return alpha_factor
```
