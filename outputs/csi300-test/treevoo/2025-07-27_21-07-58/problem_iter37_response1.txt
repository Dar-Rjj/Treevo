```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['price_momentum_1'] = df['close'].pct_change()

    # Compute Average Price Momentum over 5 Days
    df['price_momentum_5'] = df['price_momentum_1'].rolling(window=5).mean()

    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()

    # Compute Average Volume Change over 10 Days
    df['volume_change_10'] = df['volume_change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['momentum_volume_surge'] = df['price_momentum_5'] * df['volume_change_10']

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['high_low_range_change'] = df['high_low_range'].diff()

    # Combine High-Low Range and Volume
    df['high_low_volume'] = (df['high_low_range_change'] * df['volume']) / df['close']

    # Calculate Average Volume over 21 periods
    df['avg_volume_21'] = df['volume'].rolling(window=21).mean()

    # Calculate Average Amount over 21 periods
    df['avg_amount_21'] = df['amount'].rolling(window=21).mean()

    # Combine Adjusted Momentum, Volume, and Amount
    df['volume_ratio'] = df['volume'] / df['avg_volume_21']
    df['amount_ratio'] = df['amount'] / df['avg_amount_21']
    df['adjusted_momentum'] = df['price_momentum_1'] * df['volume_ratio'] * df['amount_ratio']

    # Introduce New Component: Price-Volume Divergence
    df['daily_return'] = df['close'].pct_change()
    df['volume_change_percentage'] = df['volume'].pct_change()
    df['price_volume_divergence'] = df['daily_return'] - df['volume_change_percentage']

    # Introduce New Component: Open-Close Spread
    df['open_close_spread'] = df['open'] - df['close']

    # Adjust by Price Volatility
    df['price_volatility_21'] = df['close'].rolling(window=21).std()
    df['volatility_adjustment'] = df['price_volatility_21'].apply(lambda x: 1.5 if x > 0.01 else 1.0)

    # Final Combined Alpha Factor
    df['alpha_factor'] = (
        df['momentum_volume_surge'] + 
        df['adjusted_momentum'] + 
        df['price_volume_divergence'] + 
        df['open_close_spread']
    ) * df['volatility_adjustment']

    return df['alpha_factor']
```
