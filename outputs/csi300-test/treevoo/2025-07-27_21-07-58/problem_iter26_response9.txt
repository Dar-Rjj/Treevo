```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day EMA of the money flow
    ema_14_money_flow = money_flow.ewm(span=14).mean()
    
    # Calculate the 14-day EMA of the volume
    ema_14_volume = df['volume'].ewm(span=14).mean()
    
    # Calculate the Money Flow Index (MFI) like ratio
    mfi_ratio = ema_14_money_flow / ema_14_volume
    
    # Calculate the 14-day EMA of the MFI ratio
    ema_14_mfi_ratio = mfi_ratio.ewm(span=14).mean()
    
    # The alpha factor is the 14-day EMA of the MFI ratio
    alpha_factor = ema_14_mfi_ratio
    
    return alpha_factor
```
