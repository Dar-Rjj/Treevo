```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['ShortTermPriceMomentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['MediumTermPriceMomentum'] = (df['close'] / df['close'].shift(20)) - 1

    # Identify Volume-Weighted Price Actions
    df['WeightedAvgHL'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['5DayMAWeightedAvgHL'] = df['WeightedAvgHL'].rolling(window=5).mean()
    df['CumulativeVolumeAdjustedReturn'] = (
        (df['close'] - df['open']) * df['volume']).rolling(window=10, min_periods=1).sum()

    # Analyze Trading Volume Patterns
    df['RatioTodayTo5DayAvgVolume'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['TrendInVolume'] = (df['volume'] - df['volume'].shift(5)) / 5

    # Evaluate Volatility and Range
    df['DailyPriceRange'] = df['high'] - df['low']
    df['5DayMADailyPriceRange'] = df['DailyPriceRange'].rolling(window=5).mean()
    df['RelativeDailyRange'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['20DayStdDevClose'] = df['close'].rolling(window=20).std()

    # Combine all factors into a single alpha factor
    df['AlphaFactor'] = (
        df['ShortTermPriceMomentum'] +
        df['MediumTermPriceMomentum'] +
        df['5DayMAWeightedAvgHL'] +
        df['CumulativeVolumeAdjustedReturn'] +
        df['RatioTodayTo5DayAvgVolume'] +
        df['TrendInVolume'] +
        df['DailyPriceRange'] +
        df['5DayMADailyPriceRange'] +
        df['RelativeDailyRange'] +
        df['20DayStdDevClose']
    )

    return df['AlphaFactor']
```
