```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate the first factor: (Close_t - Open_t) / Open_t * Volume_t
    factor1 = (data['close'] - data['open']) / data['open'] * data['volume']
    
    # Calculate the second factor: (High_t - Average(High_{t-5},...,High_{t-1})) / Average(High_{t-5},...,High_{t-1})
    rolling_high_avg_5 = data['high'].rolling(window=5).mean()
    factor2 = (data['high'] - rolling_high_avg_5) / rolling_high_avg_5
    
    # Calculate the third factor: (Low_t - Average(Low_{t-5},...,Low_{t-1})) / Average(Low_{t-5},...,Low_{t-1})
    rolling_low_avg_5 = data['low'].rolling(window=5).mean()
    factor3 = (data['low'] - rolling_low_avg_5) / rolling_low_avg_5
    
    # Calculate the fourth factor: Amount_t / Average(Amount_{t-9},...,Amount_{t-1})
    rolling_amount_avg_9 = data['amount'].rolling(window=9).mean()
    factor4 = data['amount'] / rolling_amount_avg_9
    
    # Calculate the fifth factor: (Close_t - Min(Low_{t-20},...,Low_{t-1})) / (Max(High_{t-20},...,High_{t-1}) - Min(Low_{t-20},...,Low_{t-1}))
    min_low_20 = data['low'].rolling(window=20).min()
    max_high_20 = data['high'].rolling(window=20).max()
    factor5 = (data['close'] - min_low_20) / (max_high_20 - min_low_20)
    
    # Combine the factors into a composite alpha factor with equal weights
    composite_factor = (
        0.2 * factor1 + 
        0.2 * factor2.fillna(0) + 
        0.2 * factor3.fillna(0) + 
        0.2 * factor4.fillna(0) + 
        0.2 * factor5.fillna(0)
    )
    
    return composite_factor
```
