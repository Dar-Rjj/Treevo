    # Calculate the 30-day volatility of the log returns
    df['volatility_30d'] = df['log_return'].rolling(window=30).std()
    
    # Compute a momentum factor based on the last 20 days' close prices
    df['momentum_20d'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate the 14-day RSI of the close price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))
    
    # Construct the alpha factor
    df['alpha_factor'] = (
        (df['avg_log_return_5d'] - df['volatility_30d']) * 
        df['momentum_20d'] * 
        (50 - abs(df['rsi_14d'] - 50))
    )
    
    return df['alpha_factor']
