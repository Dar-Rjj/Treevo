```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Incorporate Volume Momentum
    df['volume_momentum'] = (df['close'] - df['open']) * df['volume']
    
    # Calculate Change in High-Low Range
    df['high_low_range_change'] = df['intraday_range'] - df['intraday_range'].shift(1)
    
    # Combine Factors
    df['alpha_factor'] = (
        df['intraday_range'] + 
        df['close_to_open_ratio'] + 
        df['volume_momentum'] + 
        df['high_low_range_change']
    )
    
    return df['alpha_factor']
```
