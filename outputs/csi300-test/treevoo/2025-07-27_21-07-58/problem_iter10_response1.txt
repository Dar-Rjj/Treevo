```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Short-Term Momentum
    df['daily_return'] = df['close'].pct_change()
    df['5_day_mom'] = df['daily_return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Momentum
    df['20_day_mom'] = df['daily_return'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Calculate True Range and ATR
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_close_prev'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)
    df['atr_14'] = df['true_range'].rolling(window=14).mean()
    
    # Adjust by Volume
    n = 10  # Number of periods for average volume
    df['avg_volume_n'] = df['volume'].rolling(window=n).mean()
    df['vol_ratio'] = df['volume'] / df['avg_volume_n']
    
    # Adjust by Amount
    m = 10  # Number of periods for average amount
    df['avg_amount_m'] = df['amount'].rolling(window=m).mean()
    df['amt_ratio'] = df['amount'] / df['avg_amount_m']
    
    # Combine Indicators
    df['momentum'] = df['5_day_mom'] + df['20_day_mom']
    df['volatility_inverse'] = 1 / df['atr_14']
    df['heuristic_factor'] = (df['momentum'] * df['volatility_inverse'] * 
                              df['vol_ratio'] * df['amt_ratio'])
    
    return df['heuristic_factor'].dropna()
```
