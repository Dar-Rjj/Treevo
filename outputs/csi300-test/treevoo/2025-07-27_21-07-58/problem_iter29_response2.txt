```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['close_pct_change'] = df['close'].pct_change()
    
    # Calculate the average of high and low prices
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the weighted average price using volume
    df['wap'] = (df['amount'] / df['volume'])
    
    # Calculate the normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].mean()) / df['volume'].std()
    
    # Calculate the return of the weighted average price
    df['wap_return'] = df['wap'].pct_change()
    
    # Combine multiple features into a single alpha factor
    df['alpha_factor'] = (df['close_pct_change'] + df['norm_volume'] * df['wap_return']) / df['avg_high_low']
    
    # Fill NaN values with 0
    df['alpha_factor'].fillna(0, inplace=True)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
