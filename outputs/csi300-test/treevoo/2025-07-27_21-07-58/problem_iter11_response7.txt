```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate 1-Period Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Calculate Price Momentum Component
    daily_price_change_percentage = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    smoothed_price_momentum = daily_price_change_percentage.ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Surge Component
    daily_volume_change_percentage = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    smoothed_volume_surge = daily_volume_change_percentage.ewm(span=5, adjust=False).mean()
    
    # Adjust by Average Volume
    average_volume = df['volume'].rolling(window=20).mean()
    volume_ratio = df['volume'] / average_volume
    
    # Adjust by Average Amount
    average_amount = df['amount'].rolling(window=20).mean()
    amount_ratio = df['amount'] / average_amount
    
    # Combine High-Low Range and Volume
    combined_high_low_volume = (change_high_low_range * df['volume']) / df['close']
    
    # Combine Momentum, Volume, and Amount Adjustment
    adjusted_momentum = price_momentum * volume_ratio * amount_ratio
    
    # Combine Components
    combined_momentum_volume_surge = smoothed_price_momentum * smoothed_volume_surge
    final_factor = combined_high_low_volume + combined_momentum_volume_surge * adjusted_momentum
    
    return final_factor
```
