```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Component
    df['daily_price_change'] = df['close'].pct_change()
    df['avg_5_day_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate Volume Surge Component
    df['daily_volume_change'] = df['volume'].pct_change()
    df['avg_10_day_volume_change'] = df['daily_volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['combined_momentum_surge'] = df['avg_5_day_momentum'] * df['avg_10_day_volume_change']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Compute Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['moving_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['cumulative_volume'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['moving_sum_positive_returns'] * df['cumulative_volume']) / 21
    
    # Calculate 1-Period Price Momentum
    df['price_momentum_1_period'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Average Volume over n periods (n=21)
    df['average_volume'] = df['volume'].rolling(window=21).mean()
    
    # Calculate Average Amount over n periods (n=21)
    df['average_amount'] = df['amount'].rolling(window=21).mean()
    
    # Compute Volume Ratio
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Compute Amount Ratio
    df['amount_ratio'] = df['amount'] / df['average_amount']
    
    # Adjust 1-Period Price Momentum
    df['adjusted_price_momentum'] = df['price_momentum_1_period'] * df['volume_ratio'] * df['amount_ratio']
    
    # Combine Momentum, Volume, and Amount Ratios
    df['final_momentum_factor'] = df['volume_adjusted_momentum'] + df['adjusted_price_momentum']
    
    # Final Synthesized and Adjusted Momentum Indicator
    df['synthesized_adjusted_momentum'] = df['final_momentum_factor'] + df['combined_momentum_surge']
    
    return df['synthesized_adjusted_momentum']
```
