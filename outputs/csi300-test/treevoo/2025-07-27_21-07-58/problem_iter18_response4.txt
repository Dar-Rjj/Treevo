```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Daily Price Change Percentage
    data['price_change_pct'] = data['close'].pct_change()
    
    # Compute 5-Day EMA of Price Momentum
    data['price_momentum_ema'] = data['price_change_pct'].ewm(span=5, min_periods=5).mean()
    
    # Calculate Daily Volume Change Percentage
    data['volume_change_pct'] = data['volume'].pct_change()
    
    # Compute 10-Day EMA of Volume Change
    data['volume_surge_ema'] = data['volume_change_pct'].ewm(span=10, min_periods=10).mean()
    
    # Combine Price Momentum and Volume Surge
    data['combined_momentum_surge'] = data['price_momentum_ema'] * data['volume_surge_ema']
    
    # Calculate Intraday Return
    data['intraday_return'] = (data['close'] - data['open']) / data['open']
    
    # Filter positive intraday returns
    positive_returns = data[data['intraday_return'] > 0]['intraday_return']
    
    # Apply weights based on volume
    data['weighted_positive_returns'] = positive_returns * data['volume']
    
    # Apply 21-day moving sum
    data['weighted_sum_positive_returns'] = data['weighted_positive_returns'].rolling(window=21, min_periods=21).sum()
    
    # Calculate Cumulative Volume
    data['cumulative_volume'] = data['volume'].rolling(window=21, min_periods=21).sum()
    
    # Volume Adjusted Momentum
    data['volume_adjusted_momentum'] = (data['weighted_sum_positive_returns'] * data['cumulative_volume']) / 21
    
    # Calculate 1-Period Price Momentum
    data['one_period_price_momentum'] = data['close'].pct_change()
    
    # Calculate Average Volume over n periods
    n = 21
    data['average_volume'] = data['volume'].rolling(window=n, min_periods=n).mean()
    
    # Volume Ratio
    data['volume_ratio'] = data['volume'] / data['average_volume']
    
    # Combined Momentum and Volume Adjustment
    data['combined_momentum_volume_adjustment'] = data['one_period_price_momentum'] * data['volume_ratio'] + data['volume_adjusted_momentum']
    
    # Introduce Volatility Component
    data['volatility_component'] = data['intraday_return'].rolling(window=21, min_periods=21).std() / data['intraday_return'].abs().rolling(window=21, min_periods=21).mean()
    
    # Normalize Volatility Component
    data['normalized_volatility'] = data['volatility_component'].replace(0, np.nan).fillna(method='ffill')
    
    # Multiply Combined Momentum-Surge Factor by Volatility Component
    data['combined_momentum_volatility'] = data['combined_momentum_surge'] * data['normalized_volatility']
    
    # Calculate High-Low Range
    data['high_low_range'] = data['high'] - data['low']
    
    # Compute 21-day EMA of High-Low Range
    data['high_low_range_ema'] = data['high_low_range'].ewm(span=21, min_periods=21).mean()
    
    # Final Synthesized Momentum
    data['final_synthesized_momentum'] = (data['combined_momentum_volatility'] + data['combined_momentum_volume_adjustment']) * data['high_low_range_ema']
    
    return data['final_synthesized_momentum']
```
