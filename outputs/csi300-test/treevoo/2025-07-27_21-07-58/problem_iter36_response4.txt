```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series with (date) as index representing the factor values.
    """
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of close to open prices
    close_open_ratio = df['close'] / df['open']
    
    # Calculate the ratio of high to low prices
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the trading intensity as the ratio of volume to amount
    trading_intensity = df['volume'] / df['amount']
    
    # Combine the factors using a weighted sum
    alpha_factor = 0.4 * close_pct_change + 0.3 * close_open_ratio - 0.2 * high_low_ratio + 0.1 * trading_intensity
    
    # Normalize the alpha factor
    alpha_factor = (alpha_factor - alpha_factor.mean()) / alpha_factor.std()
    
    return alpha_factor
```
