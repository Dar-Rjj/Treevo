import pandas as pd

def heuristics_v2(df, n=20, m=20):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'].diff()
    
    # Calculate Price Momentum Component
    df['price_change_percent'] = (df['close'].pct_change()) * 100
    
    # Calculate Volume Surge Component
    df['volume_change_percent'] = (df['volume'].pct_change()) * 100
    
    # Combine High-Low Range and Volume
    df['high_low_range_volume'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    
    # Adjust by Average Amount
    df['average_amount'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount']
    
    # Combine Components
    df['price_momentum_volume_surge'] = df['price_change_percent'] * df['volume_change_percent']
    df['alpha_factor'] = (
        df['high_low_range_volume'] + 
        (df['price_momentum_volume_surge'] * df['volume_ratio']) * df['amount_ratio']
    )
    
    return df['alpha_factor'].dropna()
