import pandas as pd

def heuristics_v2(df):
    # Compute Daily Returns
    df['daily_returns'] = df['close'] / df['close'].shift(1)
    
    # Short-Term Momentum
    df['short_term_momentum'] = df['daily_returns'].rolling(window=5).sum()
    
    # Long-Term Momentum
    df['long_term_momentum'] = df['daily_returns'].rolling(window=20).sum()
    
    # Determine True Range
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = (df['high'] - df['close'].shift(1)).abs()
    df['low_prev_close'] = (df['low'] - df['close'].shift(1)).abs()
    df['true_range'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # 14-day Simple Moving Average of True Range
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Weighted Sum of Momentum
    df['weighted_momentum'] = df['short_term_momentum'] * 0.6 + df['long_term_momentum'] * 0.4
    
    # 50-day Exponential Moving Average of Close Price
    df['ema_50'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Apply Trend Multiplier
    df['trend_multiplier'] = 1.5 if df['close'] > df['ema_50'] else 1
    df['combined_momentum'] = df['weighted_momentum'] * df['trend_multiplier']
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int)).cumsum()
    
    # 30-day Simple Moving Average of OBV
    df['sma_obv_30'] = df['obv'].rolling(window=30).mean()
    
    # Enhance Combined Momentum
    df['enhanced_momentum'] = df['combined_momentum'] * (1.2 if df['obv'] > df['sma_obv_30'] else 1)
    
    # Return the alpha factor
    return df['enhanced_momentum']
