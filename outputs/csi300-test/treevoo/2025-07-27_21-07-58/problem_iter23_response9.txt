```python
import pandas as pd

def heuristics_v2(df):
    # Calculate typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow multiplier
    mf_multiplier = (typical_price - df['typical_price'].shift(1)) / (df['volume'] * typical_price)
    
    # Fill the first row's NA value with 0
    mf_multiplier.fillna(0, inplace=True)
    
    # Calculate the 14-day moving average of money flow multiplier
    mf_multiplier_avg = mf_multiplier.rolling(window=14).mean()
    
    # Calculate the alpha factor
    alpha_factor = typical_price * mf_multiplier_avg
    
    return alpha_factor
```
