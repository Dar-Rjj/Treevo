```python
import pandas as pd

def heuristics_v2(df):
    # Compute High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Previous Day High-to-Low Price Range
    df['prev_high_low_range'] = df['high_low_range'].shift(1)
    
    # Calculate High-to-Low Range Momentum
    df['range_momentum'] = df['high_low_range'] / df['prev_high_low_range']
    
    # Adjust by Volume
    df['volume_adjusted_range_momentum'] = df['range_momentum'] / df['volume']
    
    # Drop NaN values and return the factor
    return df['volume_adjusted_range_momentum'].dropna()
```
