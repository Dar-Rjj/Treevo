```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the 10-day moving average of the close price
    df['ma_10_close'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of the close price
    df['ma_30_close'] = df['close'].rolling(window=30).mean()
    
    # Calculate the difference between 10-day and 30-day moving averages
    df['ma_diff'] = df['ma_10_close'] - df['ma_30_close']
    
    # Calculate the 10-day standard deviation of the close price
    df['std_10_close'] = df['close'].rolling(window=10).std()
    
    # Calculate the 30-day standard deviation of the close price
    df['std_30_close'] = df['close'].rolling(window=30).std()
    
    # Calculate the ratio of 10-day to 30-day standard deviations
    df['std_ratio'] = df['std_10_close'] / df['std_30_close']
    
    # Calculate the amount/volume ratio
    df['amount_vol_ratio'] = df['amount'] / df['volume']
    
    # Alpha factor: (MA Difference * Amount/Volume Ratio) / Standard Deviation Ratio
    alpha_factor = (df['ma_diff'] * df['amount_vol_ratio']) / df['std_ratio']
    
    return alpha_factor
```
