import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Calculate Average Volume over n periods
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Calculate Average Amount over m periods
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    
    # Calculate Amount Ratio
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Combine High-Low Range and Volume
    df['HLV_Factor'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Combine Momentum, Volume, and Amount Adjustment
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Calculate Daily Price Change Percentage
    df['Daily_Price_Change'] = df['close'].pct_change()
    
    # Compute 5-Day Moving Average of Price Momentum
    df['5_Day_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change Percentage
    df['Daily_Volume_Change'] = df['volume'].pct_change()
    
    # Compute 10-Day Moving Average of Volume Surge
    df['10_Day_Volume_Surge'] = df['Daily_Volume_Change'].rolling(window=10).mean()
    
    # Combine Adjusted Momentum and Volume Surge
    df['Combined_Factor'] = df['Adjusted_Momentum'] * df['5_Day_Price_Momentum'] * df['10_Day_Volume_Surge']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['HLV_Factor'] + df['Combined_Factor']
    
    return df['Alpha_Factor']
