```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']

    # Incorporate Volume Momentum
    df['volume_momentum'] = (df['close'] - df['open']) * df['volume']

    # Determine Daily Returns
    df['daily_returns'] = df['close'].pct_change()

    # Compute Short-Term Momentum
    df['short_term_momentum'] = df['daily_returns'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Momentum
    df['long_term_momentum'] = df['daily_returns'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()

    # Calculate True Range
    df['true_range'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)), abs(x['low'] - df['close'].shift(1))), axis=1)

    # Calculate Average True Range
    df['average_true_range'] = df['true_range'].rolling(window=14).mean()
    df['inverse_average_true_range'] = 1 / df['average_true_range']

    # Adjust by Average Volume
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']

    # Adjust by Average Amount
    df['average_amount'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['average_amount']

    # Combine Factors
    df['combined_factor'] = (
        df['intraday_range'] +
        df['close_to_open_ratio'] +
        df['volume_momentum'] +
        df['short_term_momentum'] +
        df['long_term_momentum'] +
        df['inverse_average_true_range']
    )

    # Final Factor
    df['final_factor'] = df['combined_factor'] * df['volume_ratio'] * df['amount_ratio']

    return df['final_factor']
```
