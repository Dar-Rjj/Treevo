```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Daily Price Change Percentage
    df['Daily_Price_Change'] = df['close'].pct_change()
    
    # Compute Average Price Momentum over 5 Days
    df['5D_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['10D_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Momentum_Volume_Surge'] = df['5D_Price_Momentum'] * df['10D_Volume_Change']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['Range_Volume_Adjustment'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['Average_Volume_21D'] = df['volume'].rolling(window=21).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_21D']
    
    # Adjust by Average Amount
    df['Average_Amount_21D'] = df['amount'].rolling(window=21).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount_21D']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()
    
    # Compute Moving Average of Positive Returns
    df['Positive_Returns'] = df['Daily_Return'].apply(lambda x: x if x > 0 else 0)
    df['21D_Positive_Return_MA'] = df['Positive_Returns'].rolling(window=21).mean()
    
    # Volume Adjusted Momentum
    df['Cumulative_Volume_21D'] = df['volume'].rolling(window=21).sum()
    df['Volume_Adjusted_Momentum'] = (df['21D_Positive_Return_MA'] * df['Cumulative_Volume_21D']) / 21
    
    # Introduce New Component: Price-Volume Divergence
    df['Daily_Volume_Change_Percentage'] = df['volume'].pct_change()
    df['Price_Volume_Divergence'] = df['Daily_Return'] - df['Daily_Volume_Change_Percentage']
    
    # Introduce New Component: Price-Amount Divergence
    df['Daily_Amount_Change_Percentage'] = df['amount'].pct_change()
    df['Price_Amount_Divergence'] = df['Daily_Return'] - df['Daily_Amount_Change_Percentage']
    
    # Calculate 21-Day Price Volatility
    df['21D_Price_Volatility'] = df['close'].rolling(window=21).std()
    
    # Adjust by Price Volatility
    volatility_threshold = df['21D_Price_Volatility'].median()
    df['Volatility_Adjustment'] = df['21D_Price_Volatility'].apply(lambda x: 1.5 if x > volatility_threshold else 1.0)
    
    # Final Combined Alpha Factor
    df['Combined_Factor'] = (
        df['Momentum_Volume_Surge'] + 
        df['Adjusted_Momentum'] + 
        df['Volume_Adjusted_Momentum'] + 
        df['Range_Volume_Adjustment'] + 
        df['Price_Volume_Divergence'] + 
        df['Price_Amount_Divergence']
    ) * df['Volatility_Adjustment']
    
    return df['Combined_Factor']
```
