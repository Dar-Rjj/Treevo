import pandas as pd

def heuristics_v2(df):
    # Calculate the individual factors
    factor1 = (df['close'] - df['open']) / df['open'] * df['volume']
    
    high_roll_max = df['high'].rolling(window=5).max().shift(1)
    factor2 = (df['high'] - high_roll_max) / high_roll_max
    
    low_roll_min = df['low'].rolling(window=5).min().shift(1)
    factor3 = (df['low'] - low_roll_min) / low_roll_min
    
    amount_roll_avg = df['amount'].rolling(window=9).mean()
    factor4 = df['amount'] / amount_roll_avg
    
    recent_high = df['high'].rolling(window=20).max().shift(1)
    recent_low = df['low'].rolling(window=20).min().shift(1)
    factor5 = (df['close'] - recent_low) / (recent_high - recent_low)
    
    close_sum = df['close'].rolling(window=5).sum()
    factor6 = close_sum * df['volume']

    # Combine the factors to create a composite alpha factor
    # Weights can be adjusted based on empirical testing or domain knowledge
    w1, w2, w3, w4, w5, w6 = 0.2, 0.15, 0.15, 0.1, 0.2, 0.2
    composite_factor = (w1 * factor1 + w2 * factor2 + w3 * factor3 + 
                       w4 * factor4 + w5 * factor5 + w6 * factor6)

    return composite_factor.dropna()
