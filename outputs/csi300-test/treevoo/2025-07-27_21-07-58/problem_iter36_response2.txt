import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum_1D'] = df['close'].pct_change()

    # Compute Daily Price Change Percentage
    df['Daily_Price_Change_Pct'] = df['close'].pct_change()

    # Compute Average Price Momentum over 5 Days
    df['Price_Momentum_5D'] = df['Daily_Price_Change_Pct'].rolling(window=5).mean()

    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'].diff()

    # Compute Average Volume Change over 10 Days
    df['Volume_Change_10D'] = df['Daily_Volume_Change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['Momentum_Volume_Surge'] = df['Price_Momentum_5D'] * df['Volume_Change_10D']

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()

    # Combine High-Low Range and Volume
    df['High_Low_Volume_Combined'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']

    # Adjust by Average Volume
    df['Average_Volume_21D'] = df['volume'].rolling(window=21).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_21D']

    # Adjust by Average Amount
    df['Average_Amount_21D'] = df['amount'].rolling(window=21).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount_21D']

    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum_1D'] * df['Volume_Ratio'] * df['Amount_Ratio']

    # Calculate Daily Return
    df['Daily_Return'] = df['close'].pct_change()

    # Compute Moving Average of Positive Returns
    df['Positive_Returns'] = df['Daily_Return'].apply(lambda x: x if x > 0 else 0)
    df['MA_Positive_Returns_21D'] = df['Positive_Returns'].rolling(window=21).mean()

    # Volume Adjusted Momentum
    df['Cumulative_Volume_21D'] = df['volume'].rolling(window=21).sum()
    df['Volume_Adjusted_Momentum'] = (df['MA_Positive_Returns_21D'] * df['Cumulative_Volume_21D']) / 21

    # Introduce New Component: Price-Volume Divergence
    df['Daily_Volume_Change_Pct'] = df['volume'].pct_change()
    df['Price_Volume_Divergence'] = df['Daily_Return'] - df['Daily_Volume_Change_Pct']

    # Combine Components
    df['Combined_Components'] = df['Momentum_Volume_Surge'] + df['Adjusted_Momentum'] + df['Volume_Adjusted_Momentum'] + df['High_Low_Volume_Combined']

    # Calculate 21-Day Price Volatility
    df['Price_Volatility_21D'] = df['close'].rolling(window=21).std()

    # Introduce New Component: Price-Amount Divergence
    df['Daily_Amount_Change_Pct'] = df['amount'].pct_change()
    df['Price_Amount_Divergence'] = df['Daily_Return'] - df['Daily_Amount_Change_Pct']

    # Introduce New Component: Open-Close Spread
    df['Open_Close_Spread'] = df['open'] - df['close']

    # Adjust by Price Volatility
    volatility_threshold = 0.05
    df['Volatility_Adjustment'] = df['Price_Volatility_21D'].apply(lambda x: 1.5 if x > volatility_threshold else 1.0)

    # Introduce New Component: Open-Open Spread
    df['Open_Open_Spread'] = df['open'].diff()

    # Final Combined Alpha Factor
    df['Final_Alpha_Factor'] = (df['Volume_Adjusted_Momentum'] + 
                                df['Combined_Components'] + 
                                df['Price_Volume_Divergence'] + 
                                df['Price_Amount_Divergence'] + 
                                df['Open_Close_Spread'] + 
                                df['Open_Open_Spread
