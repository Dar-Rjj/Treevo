```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 5-Day Price Momentum
    df['5_day_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute 10-Day Volume Surge
    df['10_day_volume_surge'] = df['daily_volume_change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['price_vol_momentum'] = df['5_day_price_momentum'] * df['10_day_volume_surge']
    
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['open']
    
    # Filter positive daily returns and compute 21-Day Moving Sum of Positive Returns
    df['positive_returns'] = df['daily_return'].apply(lambda x: x if x > 0 else 0)
    df['21_day_sum_positive_returns'] = df['positive_returns'].rolling(window=21).sum()
    
    # Calculate 21-Day Cumulative Volume
    df['21_day_cum_vol'] = df['volume'].rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    df['vol_adjusted_momentum'] = (df['21_day_sum_positive_returns'] * df['21_day_cum_vol']) / 21
    
    # Calculate 1-Period Price Momentum
    df['1_period_price_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Average Volume over n periods
    n = 21  # Assuming 21 days for the average volume
    df['average_vol'] = df['volume'].rolling(window=n).mean()
    
    # Adjust by Volume Ratio
    df['volume_ratio'] = df['volume'] / df['average_vol']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['vol_adjusted_momentum'] + (df['1_period_price_momentum'] * df['volume_ratio'])
    
    return df['alpha_factor']
```
