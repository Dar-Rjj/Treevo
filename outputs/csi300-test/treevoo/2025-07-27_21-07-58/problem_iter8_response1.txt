```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Calculate True Range
    df['true_range'] = df.apply(lambda x: max(x['high'] - x['low'], 
                                               abs(x['high'] - df['close'].shift(1)), 
                                               abs(x['low'] - df['close'].shift(1))), axis=1)
    
    # Calculate Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    df['ema_volume'] = ema_volume
    
    # Calculate Volume Adjusted Intraday Momentum
    df['vol_adj_intraday_momentum'] = df['intraday_momentum'] * df['ema_volume']
    
    # Calculate True Range Volatility
    df['true_range_volatility'] = df['true_range'].rolling(window=5).std()
    
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Compute 5-Day Average Price Momentum
    df['5d_avg_price_momentum'] = df['daily_price_change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['daily_vol_change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute 10-Day Average Volume Change
    df['10d_avg_vol_change'] = df['daily_vol_change'].rolling(window=10).mean()
    
    # Combine Factors
    df['alpha_factor'] = (df['vol_adj_intraday_momentum'] - df['true_range_volatility']) * \
                         (df['5d_avg_price_momentum'] + df['10d_avg_vol_change'])
    
    return df['alpha_factor']
```
