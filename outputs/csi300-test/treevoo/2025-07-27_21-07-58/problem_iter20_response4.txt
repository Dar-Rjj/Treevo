```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, w1=0.2, w2=0.3, w3=0.1, w4=0.1, w5=0.1, w6=0.2):
    # Calculate 1-Period Price Momentum
    price_momentum = df['close'].pct_change(1)
    
    # Compute Daily Price Change Percentage
    daily_price_change = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    avg_price_momentum = daily_price_change.rolling(window=n).mean()
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    avg_volume_change = daily_volume_change.rolling(window=m).mean()
    
    # Combine Price Momentum and Volume Surge
    price_vol_surge = avg_price_momentum * avg_volume_change
    
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Combine High-Low Range and Volume
    high_low_vol = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=n).mean()
    amount_ratio = df['amount'] / avg_amount
    
    # Combine Adjusted Momentum, Volume, and Amount
    adjusted_momentum = price_momentum * volume_ratio * amount_ratio + price_vol_surge
    
    # Incorporate High-Low Range Dynamics
    ma_high_low_range = high_low_range.rolling(window=n).mean()
    ma_change_high_low_range = change_high_low_range.rolling(window=n).mean()
    high_low_dynamics = ma_high_low_range * ma_change_high_low_range
    
    # Calculate 1-Period Open-Close Spread
    open_close_spread = (df['open'] - df['close']) / df['close']
    
    # Combine Weighted Components
    weighted_components = (
        w1 * high_low_vol +
        w2 * adjusted_momentum +
        w3 * price_vol_surge +
        w4 * volume_ratio +
        w5 * amount_ratio +
        w6 * open_close_spread
    )
    
    # Final Combined Alpha Factor
    alpha_factor = weighted_components + high_low_dynamics
    
    return alpha_factor
```
