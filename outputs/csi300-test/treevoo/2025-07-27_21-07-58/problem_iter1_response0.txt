```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate Previous Day High-to-Low Price Range
    prev_high_to_low_range = high_to_low_range.shift(1)
    
    # Compute High-to-Low Range Momentum
    range_momentum = high_to_low_range / prev_high_to_low_range
    
    # Adjust by Volume
    volume_adjusted_momentum = range_momentum / df['volume']
    
    return volume_adjusted_momentum
```
