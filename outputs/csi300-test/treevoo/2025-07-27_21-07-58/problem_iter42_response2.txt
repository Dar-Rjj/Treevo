import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Compute 5-Day and 20-Day Moving Average of Price Change
    df['5D_MA_Price_Change'] = df['Daily_Price_Change'].rolling(window=5).mean()
    df['20D_MA_Price_Change'] = df['Daily_Price_Change'].rolling(window=20).mean()
    
    # Determine Daily Returns
    df['Daily_Returns'] = df['close'].pct_change()
    
    # Compute Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['Daily_Returns'].rolling(window=5).sum()
    df['Long_Term_Momentum'] = df['Daily_Returns'].rolling(window=20).sum()
    
    # Calculate Average True Range for Volatility
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Adjust by Volume
    df['10D_MA_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Ratio'] = df['volume'] / df['10D_MA_Volume']
    
    # Adjust by Amount
    df['7D_MA_Amount'] = df['amount'].rolling(window=7).mean()
    df['Amount_Ratio'] = df['amount'] / df['7D_MA_Amount']
    
    # Introduce High-to-Low Range Momentum Adjustment
    df['HL_Range_Momentum'] = (df['high'] - df['low']) * df['10D_MA_Volume']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    combined_momentum = (df['5D_MA_Price_Change'] + df['Short_Term_Momentum'] + df['Long_Term_Momentum']) / df['ATR']
    adjusted_combined_factor = combined_momentum * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Adjust Combined Factor by Price Volatility
    df['10D_Std_Daily_Price_Changes'] = df['Daily_Price_Change'].rolling(window=10).std()
    adjusted_combined_factor = adjusted_combined_factor / df['10D_Std_Daily_Price_Changes']
    
    # Add Combined Price Momentum and Volume Surge
    combined_price_vol_surges = (df['5D_MA_Price_Change'] + df['Short_Term_Momentum'] + df['Long_Term_Momentum']) * df['10D_MA_Volume']
    adjusted_combined_factor += combined_price_vol_surges
    
    # Add Combined High-Low Range and Volume
    high_low_range_vol = (df['high'] - df['low']) * df['volume'] / df['close']
    adjusted_combined_factor += high_low_range_vol
    
    # Incorporate High-Low Range Dynamics
    df['5D_MA_High_Low_Range'] = (df['high'] - df['low']).rolling(window=5).mean()
    df['5D_MA_Change_High_Low_Range'] = df['5D_MA_High_Low_Range'].diff().rolling(window=5).mean()
    high_low_dynamics = df['5D_MA_High_Low_Range'] * df['5D_MA_Change_High_Low_Range']
    adjusted_combined_factor += high_low_dynamics
    
    # Integrate High-Low Range with Volume
    df['5D_MA_HL_Range_Vol'] = (df['high'] - df['low']).rolling(window=5).mean() / df['volume']
    df['5D_MA_Volume'] = df['volume'].rolling(window=5).mean()
    high_low_vol_integration = df['5D_MA_HL_Range_Vol'] * df['5D_MA_Volume']
    adjusted_combined_factor += high_low_vol_integration
    
    # Introduce High-Frequency Momentum
    df['3D_EMA_Daily_Returns'] = df['Daily_Returns'].ewm(span=3, adjust=False).mean()
    
    # Incorporate Open-to-Close Momentum
    df['Open_To_Close_Momentum'] = (df['close'] - df['open']) / df['open']
    
    # Combine High-Frequency and Open-to-Close Momentum
    combined_high_freq_open_close = (df['3D_EMA_Daily_Returns'] + df['Open_To_Close_Momentum']) * df['10D_MA_Volume']
    adjusted_combined_factor += combined_high_freq_open_close
