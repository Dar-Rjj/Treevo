```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['price_momentum'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute 5-Day Moving Average of Price Momentum
    df['price_mom_5d_ma'] = df['price_momentum'].rolling(window=5).mean()
    
    # Compute 20-Day Moving Average Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['vol_change_20d_ma'] = df['volume_change'].rolling(window=20).mean()
    
    # Combine Price Momentum and Volume Surge
    df['mom_vol_surge'] = df['price_mom_5d_ma'] * df['vol_change_20d_ma']
    
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'] - df['high_low_range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['high_low_vol_adjusted'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['volume_ratio'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['adj_mom_vol_amount'] = df['price_momentum'] * df['volume_ratio'] + df['mom_vol_surge'] + df['high_low_vol_adjusted']
    
    # Incorporate High-Low Range Dynamics
    df['high_low_range_10d_ma'] = df['high_low_range'].rolling(window=10).mean()
    df['change_high_low_range_10d_ma'] = df['change_high_low_range'].rolling(window=10).mean()
    df['high_low_range_dynamics'] = df['high_low_range_10d_ma'] * df['change_high_low_range_10d_ma']
    
    # Integrate High-Low Range with Volume
    df['high_low_range_vol_ratio_10d_ma'] = (df['high_low_range'] / df['volume']).rolling(window=10).mean()
    df['volume_10d_ma'] = df['volume'].rolling(window=10).mean()
    df['integrated_high_low_vol'] = df['high_low_range_vol_ratio_10d_ma'] * df['volume_10d_ma']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['adj_mom_vol_amount'] + df['high_low_range_dynamics'] + df['integrated_high_low_vol']
    
    return df['alpha_factor']
```
