```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price movement from the previous day's close to the current day's open
    df['overnight_move'] = df['open'] - df['close'].shift(1)
    
    # Calculate the intraday volatility
    df['intraday_volatility'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Compute the momentum factor
    df['momentum'] = df['close'] / df['close'].shift(20) - 1
    
    # Combine all the factors into a single alpha factor
    alpha_factor = (
        0.4 * (df['overnight_move'] / df['close'].shift(1)) +
        0.3 * df['intraday_volatility'] +
        0.2 * (df['close'] / df['vwap'] - 1) +
        0.1 * df['momentum']
    )
    
    return alpha_factor
```
