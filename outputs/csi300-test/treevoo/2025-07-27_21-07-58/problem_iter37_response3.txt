```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(1)

    # Compute 5-Day Moving Average of Price Momentum
    df['Daily_Price_Change_Percentage'] = df['Price_Momentum'] / df['close'].shift(1)
    df['5D_MA_Price_Momentum'] = df['Daily_Price_Change_Percentage'].rolling(window=5).mean()

    # Compute 10-Day Moving Average Volume Change
    df['Daily_Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['10D_MA_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['Combined_Price_Volume'] = df['5D_MA_Price_Momentum'] * df['10D_MA_Volume_Change']

    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Change in High-Low Range
    df['Change_in_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)

    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_in_High_Low_Range'] * df['volume']) / df['close']

    # Calculate Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()

    # Combine Adjusted Momentum, Volume, and Amount
    df['Combined_Adjusted_Momentum'] = (df['Price_Momentum'] * df['Volume_Ratio']) + df['Combined_Price_Volume'] + df['Combined_High_Low_Volume']

    # Calculate 5-Day Moving Average of High-Low Range divided by Volume
    df['5D_MA_High_Low_Range_Vol'] = (df['High_Low_Range'] / df['volume']).rolling(window=5).mean()

    # Calculate 5-Day Moving Average of Volume
    df['5D_MA_Volume'] = df['volume'].rolling(window=5).mean()

    # Multiply 5-Day Moving Average of High-Low Range divided by Volume by 5-Day Moving Average of Volume
    df['Integrated_High_Low_Volume'] = df['5D_MA_High_Low_Range_Vol'] * df['5D_MA_Volume']

    # Final Combined Alpha Factor
    df['Alpha_Factor'] = df['Combined_Adjusted_Momentum'] + df['Integrated_High_Low_Volume']

    return df['Alpha_Factor']
```
