import pandas as pd

def heuristics_v2(df):
    # Calculate each of the factors based on the sub-thoughts provided
    df['Factor1'] = (df['close'] - df['open']) / df['open'] * df['volume']
    
    max_high_rolling = df['high'].rolling(window=5).max()
    df['Factor2'] = (df['high'] - max_high_rolling.shift(1)) / max_high_rolling.shift(1)
    
    min_low_rolling = df['low'].rolling(window=5).min()
    df['Factor3'] = (df['low'] - min_low_rolling.shift(1)) / min_low_rolling.shift(1)
    
    amount_ma = df['amount'].rolling(window=10).mean()
    df['Factor4'] = df['amount'] / amount_ma
    
    max_high_rolling_long = df['high'].rolling(window=20).max()
    min_low_rolling_long = df['low'].rolling(window=20).min()
    df['Factor5'] = (df['close'] - min_low_rolling_long) / (max_high_rolling_long - min_low_rolling_long)

    # Define weights for each factor. These can be adjusted based on their importance or optimization results.
    w1, w2, w3, w4, w5 = 0.3, 0.15, 0.15, 0.2, 0.2  # Example weights, can be tuned
    df['Composite_Factor'] = w1*df['Factor1'] + w2*df['Factor2'] + w3*df['Factor3'] + w4*df['Factor4'] + w5*df['Factor5']

    # Return the composite alpha factor series
    return df['Composite_Factor']
