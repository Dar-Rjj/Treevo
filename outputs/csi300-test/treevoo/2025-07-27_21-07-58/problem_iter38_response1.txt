```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    price_momentum = df['close'].pct_change()
    
    # Compute Average Price Momentum over 5 Days
    avg_price_momentum_5d = price_momentum.rolling(window=5).mean()
    
    # Compute 10-Day Moving Average Volume Change
    volume_change = df['volume'].pct_change()
    avg_volume_change_10d = volume_change.rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    combined_momentum_surge = avg_price_momentum_5d * avg_volume_change_10d
    
    # Calculate Daily Return
    daily_return = (df['close'] - df['open']) / df['open']
    
    # Compute Moving Sum of Positive Returns
    positive_returns = daily_return.where(daily_return > 0, 0)
    moving_sum_positive_returns = positive_returns.rolling(window=21).sum()
    
    # Volume Adjusted Momentum
    cumulative_volume = df['volume'].rolling(window=21).sum()
    vol_adjusted_momentum = (moving_sum_positive_returns * cumulative_volume) / 21
    
    # Compute High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range.diff()
    
    # Combine High-Low Range and Volume
    hl_vol_combination = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume_n = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / avg_volume_n
    
    # Adjust by Average Amount
    avg_amount_m = df['amount'].rolling(window=21).mean()
    amount_ratio = df['amount'] / avg_amount_m
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.2, 0.1, 0.2
    
    # Combine Weighted Components
    weighted_hl_vol = hl_vol_combination * w1
    weighted_price_momentum = avg_price_momentum_5d * w2
    weighted_volume_surge = combined_momentum_surge * w3
    weighted_volume_ratio = volume_ratio * w4
    weighted_amount_ratio = amount_ratio * w5
    
    combined_weighted_components = (
        weighted_hl_vol + 
        weighted_price_momentum + 
        weighted_volume_surge + 
        weighted_volume_ratio + 
        weighted_amount_ratio
    )
    
    # Final Combined Alpha Factor
    final_alpha_factor = (
        vol_adjusted_momentum + 
        combined_weighted_components + 
        (price_momentum * volume_ratio) + 
        (high_low_range.rolling(window=5).mean() / df['volume'].rolling(window=5).mean()) * df['volume'].rolling(window=5).mean()
    )
    
    return final_alpha_factor
```
