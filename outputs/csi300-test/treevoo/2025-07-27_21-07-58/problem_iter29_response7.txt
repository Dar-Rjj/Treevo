```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Simple Moving Averages
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    
    # Momentum Indicators
    df['ROC_10'] = df['close'].pct_change(10)
    
    # Volatility Indicators
    df['True_Range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Volume-Based Indicators
    df['OBV'] = (np.sign(df['close'].diff()) * df['volume']).fillna(0).cumsum()
    df['VPT'] = (df['volume'] * df['close'].pct_change()).fillna(0).cumsum()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Accumulation/Distribution Line (A/D Line)
    df['ADL'] = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    df['ADL'] = df['ADL'].cumsum()
    
    # Bollinger Bands
    df['SMA_20_BB'] = df['close'].rolling(window=20).mean()
    df['STD_20'] = df['close'].rolling(window=20).std()
    df['Upper_BB'] = df['SMA_20_BB'] + 2 * df['STD_20']
    df['Lower_BB'] = df['SMA_20_BB'] - 2 * df['STD_20']
    
    # Moving Average Convergence Divergence (MACD)
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['MACD'] = df['EMA_12'] - df['EMA_26']
    df['Signal_Line'] = df['MACD'].ewm(span=9, adjust=False).mean()
    
    # Custom Composite Factors
    df['Price_Volume_Momentum'] = df['ROC_10'] + df['OBV'].pct_change()
    df['Volatility_Sentiment'] = df['ATR_14'] + (100 - df['RSI_14'])
    
    # Generate the final alpha factor
    df['Alpha_Factor'] = (df['Price_Volume_Momentum'] + df['Volatility_Sentiment']) / 2
    
    return df['Alpha_Factor']
```
