```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum_1D'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['Daily_Price_Change_Pct'] = df['Price_Momentum_1D']
    df['Avg_Price_Momentum_5D'] = df['Daily_Price_Change_Pct'].rolling(window=5).mean()
    
    # Compute Average Volume Change over 10 Days
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    df['Avg_Volume_Change_10D'] = df['Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Combined_Price_Vol'] = df['Avg_Price_Momentum_5D'] * df['Avg_Volume_Change_10D']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['Combined_HL_Vol'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 20
    df['Avg_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume']
    
    # Adjust by Average Amount
    m = 20
    df['Avg_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Avg_Amount']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum_1D'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Incorporate High-Low Range Dynamics
    df['Avg_High_Low_Range_5D'] = df['High_Low_Range'].rolling(window=5).mean()
    df['Avg_Change_High_Low_Range_5D'] = df['Change_High_Low_Range'].rolling(window=5).mean()
    df['Combined_HL_Dynamics'] = df['Avg_High_Low_Range_5D'] * df['Avg_Change_High_Low_Range_5D']
    
    # Final Combined Alpha Factor
    df['Alpha_Factor'] = df['Combined_Price_Vol'] + df['Adjusted_Momentum'] + df['Combined_HL_Dynamics']
    
    return df['Alpha_Factor']
```
