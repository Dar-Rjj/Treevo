```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute daily returns
    df['returns'] = df['close'].pct_change()
    
    # Compute short-term and long-term momentum
    df['short_term_momentum'] = df['returns'].rolling(window=5).sum()
    df['long_term_momentum'] = df['returns'].rolling(window=20).sum()
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    
    # Compute 14-day Simple Moving Average of True Range for ATR
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Incorporate Volume Momentum
    df['volume_momentum'] = (df['close'] - df['open']) * df['volume']
    
    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = (df['short_term_momentum'] + df['long_term_momentum']) / df['atr']
    
    # Combine Intraday Factors
    df['combined_intraday_factors'] = df['intraday_range'] + df['close_to_open_ratio'] + df['volume_momentum']
    
    # Final Combined Alpha Factor
    df['final_alpha_factor'] = df['combined_momentum_volatility'] + df['combined_intraday_factors']
    
    return df['final_alpha_factor'].dropna()
```
