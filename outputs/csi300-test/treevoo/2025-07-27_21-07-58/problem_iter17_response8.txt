```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range.diff()
    
    # Calculate Daily Return
    daily_return = df['close'].pct_change()
    
    # Compute Moving Average of Positive Returns
    positive_returns = daily_return[daily_return > 0]
    moving_avg_positive_returns = positive_returns.rolling(window=21).mean().fillna(0)
    
    # Volume Adjusted Momentum
    cumulative_volume = df['volume'].rolling(window=21).sum()
    volume_adjusted_momentum = (moving_avg_positive_returns * cumulative_volume) / 21
    
    # Calculate 5-Period Price Momentum
    price_momentum_5 = df['close'] / df['close'].shift(5) - 1
    
    # Combine High-Low Range and Volume
    high_low_volume = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    average_volume = df['volume'].rolling(window=21).mean()
    volume_ratio = df['volume'] / average_volume
    
    # Adjust by Average Amount
    average_amount = df['amount'].rolling(window=21).mean()
    amount_ratio = df['amount'] / average_amount
    
    # Introduce New Component: Price-Volume Divergence
    volume_change_percentage = df['volume'].pct_change()
    price_volume_divergence = daily_return - volume_change_percentage
    
    # Combine Components
    combined_component = (price_momentum_5 * volume_ratio * amount_ratio + 
                          volume_adjusted_momentum + 
                          high_low_volume)
    
    # Calculate 21-Day Price Volatility
    price_volatility_21 = df['close'].rolling(window=21).std()
    
    # Adjust by Price Volatility
    volatility_adjustment = price_volatility_21.apply(lambda x: 1.5 if x > 0.01 else 1.0)
    
    # Final Combined Alpha Factor
    final_alpha_factor = (combined_component + price_volume_divergence) * volatility_adjustment
    
    return final_alpha_factor
```
