```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Calculate True Range
    df['prev_close'] = df['close'].shift(1)
    df['true_range'] = df[['high', 'low', 'prev_close']].apply(
        lambda x: max(x['high'] - x['low'], 
                      x['high'] - x['prev_close'], 
                      x['prev_close'] - x['low']),
        axis=1
    )
    
    # Calculate Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=5, adjust=False).mean()
    df['ema_volume'] = ema_volume
    
    # Calculate Volume Adjusted Intraday Momentum
    df['vol_adj_intraday_mom'] = df['intraday_momentum'] * df['ema_volume']
    
    # Calculate True Range Volatility
    tr_volatility = df['true_range'].rolling(window=5).std()
    df['tr_volatility'] = tr_volatility
    
    # Compute Final Alpha Factor
    df['alpha_factor'] = df['vol_adj_intraday_mom'] - df['tr_volatility']
    
    return df['alpha_factor']
```
