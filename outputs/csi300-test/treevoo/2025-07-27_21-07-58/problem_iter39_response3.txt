```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index (date) representing the generated factor values.
    """
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the percentage change from open to close
    pct_change = (df['close'] - df['open']) / df['open']
    
    # Calculate the volume weighted price
    vwp = (df['amount'] / df['volume'])
    
    # Calculate the factor
    factor = (vwp * pct_change) / price_range
    
    return factor
```
