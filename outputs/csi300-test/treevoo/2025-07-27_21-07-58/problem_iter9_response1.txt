```python
import pandas as pd

def heuristics_v2(df, n=5, m=5, w1=0.3, w2=0.2, w3=0.2, w4=0.15, w5=0.15):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['change_high_low_range'] = df['high_low_range'].diff()
    
    # Calculate Price Momentum Component
    df['price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Surge Component
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine High-Low Range and Volume
    df['high_low_vol_adjusted'] = (df['change_high_low_range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume']
    
    # Adjust by Average Amount
    df['avg_amount'] = df['amount'].rolling(window=m).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount']
    
    # Combine Weighted Components
    df['alpha_factor'] = (
        w1 * df['high_low_vol_adjusted'] +
        w2 * df['price_change_pct'] +
        w3 * df['volume_change_pct'] +
        w4 * df['volume_ratio'] +
        w5 * df['amount_ratio']
    )
    
    return df['alpha_factor']
```
