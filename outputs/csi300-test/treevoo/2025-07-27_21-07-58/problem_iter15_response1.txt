```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Component
    daily_price_change = df['close'].pct_change()
    price_momentum_ema = daily_price_change.ewm(span=5, adjust=False).mean()

    # Calculate Volume Surge Component
    daily_volume_change = df['volume'].pct_change()
    volume_surge_ema = daily_volume_change.ewm(span=10, adjust=False).mean()

    # Combine Price Momentum and Volume Surge
    combined_momentum_surge = price_momentum_ema * volume_surge_ema

    # Calculate Intraday Return
    intraday_return = df['close'] - df['open']

    # Compute Weighted Sum of Positive Returns
    positive_returns = intraday_return.where(intraday_return > 0, 0)
    weighted_sum_positive_returns = (positive_returns * df['volume']).rolling(window=21).sum()

    # Volume Adjusted Momentum
    cumulative_volume = df['volume'].rolling(window=21).sum()
    volume_adjusted_momentum = (weighted_sum_positive_returns * cumulative_volume) / 21

    # Calculate 1-Period Price Momentum
    one_period_momentum = df['close'] / df['close'].shift(1)

    # Adjust by Recent Volume
    average_volume = df['volume'].rolling(window=21).mean()
    recent_volume = df['volume']
    volume_ratio = recent_volume / average_volume
    adjusted_momentum = one_period_momentum * volume_ratio

    # Introduce Volatility Component
    std_intraday_returns = intraday_return.rolling(window=21).std()
    avg_abs_intraday_return = intraday_return.abs().rolling(window=21).mean()
    volatility_component = std_intraday_returns / avg_abs_intraday_return
    volatility_component = volatility_component.replace([np.inf, -np.inf], 0)

    # Combine All Components
    synthesized_factor = combined_momentum_surge * volatility_component + adjusted_momentum + volume_adjusted_momentum

    return synthesized_factor
```
