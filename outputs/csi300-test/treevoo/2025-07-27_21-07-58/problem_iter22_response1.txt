import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Close_t-1'] = df['close'].shift(1)
    df['Price_Momentum'] = df['close'] / df['Close_t-1'] - 1
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Calculate Price Momentum Component
    df['Daily_Price_Change_Percentage'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Smoothed_Price_Momentum'] = df['Daily_Price_Change_Percentage'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Surge Component
    df['Daily_Volume_Change_Percentage'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['10_Day_Avg_Volume_Change'] = df['Daily_Volume_Change_Percentage'].rolling(window=10).mean()
    
    # Combine High-Low Range and Volume
    df['HL_Volume_Factor'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    df['21_Day_Avg_Volume'] = df['volume'].rolling(window=21).mean()
    df['Volume_Ratio'] = df['volume'] / df['21_Day_Avg_Volume']
    
    # Calculate Amount Ratio
    df['21_Day_Avg_Amount'] = df['amount'].rolling(window=21).mean()
    df['Amount_Ratio'] = df['amount'] / df['21_Day_Avg_Amount']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum_Factor'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Combine Components
    df['Final_Alpha_Factor'] = (df['Smoothed_Price_Momentum'] * df['10_Day_Avg_Volume_Change']) + df['HL_Volume_Factor']
    
    return df['Final_Alpha_Factor']
