```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate 1-Period Price Momentum
    df['PriceMomentum'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Average Volume over n periods
    df['AverageVolume'] = df['volume'].rolling(window=n).mean()
    # Compare with Recent Volume
    df['VolumeRatio'] = df['volume'] / df['AverageVolume']
    
    # Calculate Average Amount over n periods
    df['AverageAmount'] = df['amount'].rolling(window=n).mean()
    # Compare with Recent Amount
    df['AmountRatio'] = df['amount'] / df['AverageAmount']
    
    # Combine Momentum, Volume, and Amount Adjustment
    df['AdjustedMomentum'] = df['PriceMomentum'] * df['VolumeRatio'] * df['AmountRatio']
    
    return df['AdjustedMomentum']
```
