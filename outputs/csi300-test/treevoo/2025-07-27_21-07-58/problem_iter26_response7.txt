```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    df['pct_change'] = df['close'].pct_change()
    
    # Calculate the average of the high and low for each day
    df['avg_high_low'] = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of volume to amount
    df['volume_to_amount'] = df['volume'] / df['amount']
    
    # Calculate the weighted moving average of the pct_change, giving more weight to recent changes
    df['wma_pct_change'] = df['pct_change'].ewm(span=5, adjust=False).mean()
    
    # Generate a novel alpha factor by combining the above calculated features
    df['alpha_factor'] = (df['avg_high_low'] * df['wma_pct_change']) / df['volume_to_amount']

    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
