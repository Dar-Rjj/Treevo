```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low', 'close']].join(df['close'].shift(1), rsuffix='_prev').apply(
        lambda x: max(x['high'] - x['low'], x['high'] - x['close_prev'], x['close_prev'] - x['low']), axis=1
    )
    
    # Calculate Exponential Moving Average of Volume
    df['volume_ema'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Adjusted Intraday Momentum
    df['volume_adjusted_intraday_momentum'] = df['intraday_momentum'] * (df['volume'] / df['volume_ema'])
    
    # Calculate True Range Volatility
    df['true_range_volatility'] = df['true_range'].rolling(window=10).std()
    
    # Compute Final Alpha Factor
    df['alpha_factor'] = df['volume_adjusted_intraday_momentum'] - df['true_range_volatility']
    
    return df['alpha_factor'].copy()
```
