```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term and Medium-Term Price Momentum
    short_term_momentum = (df['close'] / df['close'].shift(5)) - 1
    medium_term_momentum = (df['close'] / df['close'].shift(20)) - 1
    
    # Compute Average Price Momentum over 5 Days
    daily_price_change = df['close'] - df['close'].shift(1)
    avg_5_day_momentum = daily_price_change.rolling(window=5).mean()
    
    # Identify Volume-Weighted Price Actions
    weighted_avg_hl = ((df['high'] + df['low']) / 2) * df['volume']
    avg_weighted_hl = weighted_avg_hl.rolling(window=5).mean()
    cum_vol_adj_return = (df['close'] - df['open']) * df['volume']
    cum_vol_adj_return = cum_vol_adj_return.rolling(window=10).sum()
    
    # Analyze Trading Volume Patterns
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    avg_10_day_vol_change = daily_volume_change.rolling(window=10).mean()
    
    # Evaluate Volatility and Range
    daily_price_range = df['high'] - df['low']
    relative_daily_range = daily_price_range / df['close'].shift(1)
    std_close_22_days = df['close'].rolling(window=22).std()
    
    # Combine Price Momentum and Volume Surge
    combined_momentum_vol_surge = avg_5_day_momentum * avg_10_day_vol_change
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.2, 0.2, 0.1
    
    # Combine Weighted Components
    high_low_vol_combination = (avg_weighted_hl * volume_ratio) * w1
    price_momentum_component = (short_term_momentum + medium_term_momentum) * w2
    vol_surge_component = combined_momentum_vol_surge * w3
    volume_ratio_component = volume_ratio * w4
    amount_ratio = (df['amount'] / df['amount'].rolling(window=5).mean()) * w5
    
    # Final Combined Alpha Factor
    alpha_factor = (
        combined_momentum_vol_surge +
        high_low_vol_combination +
        price_momentum_component +
        vol_surge_component +
        volume_ratio_component +
        amount_ratio
    )
    
    return alpha_factor
```
