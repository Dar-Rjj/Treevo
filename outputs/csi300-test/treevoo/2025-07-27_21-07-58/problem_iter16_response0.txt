```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Momentum Indicators
    df['daily_return'] = df['close'].pct_change()
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum()

    # Calculate Average True Range for Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)), abs(x['low'] - df['close'].shift(1))), axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()

    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    df['change_high_low_range'] = df['high_low_range'].diff()

    # Adjust by Volume
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_ratio'] = df['volume'] / df['avg_volume_10']

    # Adjust by Amount
    df['avg_amount_10'] = df['amount'].rolling(window=10).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_10']

    # Combine Indicators
    df['momentum_sum'] = df['short_term_momentum'] + df['long_term_momentum']
    df['volatility_adjustment'] = 1 / df['avg_true_range']
    df['combined_factor'] = (df['momentum_sum'] * df['volatility_adjustment'] * df['volume_ratio'] * df['amount_ratio']) + df['change_high_low_range']
    
    return df['combined_factor']
```
