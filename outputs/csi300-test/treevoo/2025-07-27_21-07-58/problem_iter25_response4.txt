```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Returns
    df['daily_return'] = df['close'] / df['close'].shift(1) - 1
    
    # Calculate Short-Term Momentum (5-day sum of daily returns)
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum()
    
    # Calculate Long-Term Momentum (20-day sum of daily returns)
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum()
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    
    # Calculate 14-day Simple Moving Average of True Range (Average True Range)
    df['atr'] = df['true_range'].rolling(window=14).mean()
    
    # Weighted Sum of Momentum
    df['weighted_momentum'] = df['short_term_momentum'] * 0.6 + df['long_term_momentum'] * 0.4
    
    # Apply Trend Multiplier
    df['50_day_ema'] = df['close'].ewm(span=50, adjust=False).mean()
    df['trend_multiplier'] = df.apply(lambda x: 1.5 if x['close'] > x['50_day_ema'] else 1, axis=1)
    df['combined_momentum'] = df['weighted_momentum'] * df['trend_multiplier']
    
    # Calculate On-Balance Volume (OBV)
    df['obv'] = (df['volume'] * (df['close'] > df['close'].shift(1)).astype(int)).cumsum()
    
    # Enhance Combined Momentum with OBV
    df['obv_sma_30'] = df['obv'].rolling(window=30).mean()
    df['enhanced_momentum'] = df.apply(lambda x: x['combined_momentum'] * 1.2 if x['obv'] > x['obv_sma_30'] else x['combined_momentum'], axis=1)
    
    return df['enhanced_momentum'].dropna()
```
