```python
import pandas as pd

def heuristics_v2(df):
    # Determine Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Short-Term Momentum
    short_term_momentum = df['daily_return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Momentum
    long_term_momentum = df['daily_return'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()
    
    # Determine True Range
    df['true_range'] = pd.concat(
        (df['high'] - df['low'], 
         abs(df['high'] - df['close'].shift(1)), 
         abs(df['low'] - df['close'].shift(1))), 
        axis=1).max(axis=1)
    
    # Compute 14-day Simple Moving Average of True Range
    avg_true_range = df['true_range'].rolling(window=14).mean()
    
    # Combine Momentum and Volatility
    weighted_momentum = short_term_momentum * 0.6 + long_term_momentum * 0.4
    combined_factor = weighted_momentum / avg_true_range
    
    # Add a trend-following component
    fifty_day_sma = df['close'].rolling(window=50).mean()
    trend_multiplier = (df['close'] > fifty_day_sma) * 1.2 + 1.0
    final_alpha_factor = combined_factor * trend_multiplier
    
    return final_alpha_factor
```
