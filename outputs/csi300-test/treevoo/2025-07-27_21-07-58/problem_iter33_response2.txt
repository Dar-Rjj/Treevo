import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum_1D'] = df['close'] - df['close'].shift(1)
    
    # Compute 5-Day Moving Average of Price Momentum
    df['Daily_Price_Change_Percentage'] = df['Price_Momentum_1D'] / df['close'].shift(1)
    df['5D_MA_Price_Momentum'] = df['Daily_Price_Change_Percentage'].rolling(window=5).mean()
    
    # Compute 10-Day Moving Average Volume Change
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['10D_MA_Volume_Change'] = df['Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Combined_Price_Momentum_Volume_Surge'] = df['5D_MA_Price_Momentum'] * df['10D_MA_Volume_Change']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['Combined_Adjusted_Momentum_Volume_Amount'] = (df['Price_Momentum_1D'] * df['Volume_Ratio']) + \
                                                      df['Combined_Price_Momentum_Volume_Surge'] + \
                                                      df['Combined_High_Low_Volume']
    
    # Incorporate High-Low Range Dynamics
    df['5D_MA_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['5D_MA_Change_High_Low_Range'] = df['Change_High_Low_Range'].rolling(window=5).mean()
    df['Combined_High_Low_Range_Dynamics'] = df['5D_MA_High_Low_Range'] * df['5D_MA_Change_High_Low_Range']
    
    # Introduce High-Low Range Volatility
    df['5D_STD_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).std()
    df['Normalized_High_Low_Volatility'] = df['5D_STD_High_Low_Range'] / df['5D_MA_High_Low_Range']
    
    # Final Combined Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Adjusted_Momentum_Volume_Amount'] + \
                               df['Combined_High_Low_Range_Dynamics'] + \
                               df['Normalized_High_Low_Volatility']
    
    # Return the final alpha factor as a Series
    return df['Final_Alpha_Factor'].dropna()
