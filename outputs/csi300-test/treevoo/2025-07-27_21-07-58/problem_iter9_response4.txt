```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate Price Momentum Component
    daily_price_change_percent = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_momentum_smoothed = daily_price_change_percent.ewm(span=5, adjust=False).mean()
    
    # Calculate Volume Surge Component
    daily_volume_change_percent = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_surge_smoothed = daily_volume_change_percent.ewm(span=5, adjust=False).mean()
    
    # Combine High-Low Range and Volume
    high_low_volume_combination = (change_high_low_range * df['volume']) / df['close']
    
    # Combine Components
    component_multiplication = price_momentum_smoothed * volume_surge_smoothed
    alpha_factor = high_low_volume_combination + component_multiplication
    
    return alpha_factor
```
