```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'].pct_change()
    
    # Compute Average Price Momentum over 5 Days
    df['5D_Price_Momentum_MA'] = df['Price_Momentum'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['Volume_Change'] = df['volume'].diff()
    
    # Compute Average Volume Change over 10 Days
    df['10D_Volume_Change_MA'] = df['Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Momentum_Volume_Surge'] = df['5D_Price_Momentum_MA'] * df['10D_Volume_Change_MA']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'].diff()
    
    # Combine High-Low Range and Volume
    df['Combined_HLR_Vol'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 21
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    
    # Adjust by Average Amount
    m = 21
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Volume Adjusted Momentum
    df['Cumulative_Volume'] = df['volume'].rolling(window=21).sum()
    df['Positive_Returns'] = df['close'].pct_change().apply(lambda x: max(0, x))
    df['Moving_Average_Positive_Returns'] = df['Positive_Returns'].rolling(window=21).mean()
    df['Volume_Adjusted_Momentum'] = (df['Moving_Average_Positive_Returns'] * df['Cumulative_Volume']) / 21
    
    # Introduce New Component: Price-Volume Divergence
    df['Daily_Return'] = df['close'].pct_change()
    df['Volume_Change_Percentage'] = df['volume'].pct_change()
    df['Price_Volume_Divergence'] = df['Daily_Return'] - df['Volume_Change_Percentage']
    
    # Introduce New Component: Price-Amount Divergence
    df['Amount_Change_Percentage'] = df['amount'].pct_change()
    df['Price_Amount_Divergence'] = df['Daily_Return'] - df['Amount_Change_Percentage']
    
    # Calculate 21-Day Price Volatility
    df['21D_Price_Volatility'] = df['close'].rolling(window=21).std()
    
    # Adjust by Price Volatility
    df['Volatility_Adjustment'] = df['21D_Price_Volatility'].apply(lambda x: 1.5 if x > df['21D_Price_Volatility'].mean() else 1.0)
    
    # Final Combined Alpha Factor
    df['Final_Alpha_Factor'] = (
        df['Volume_Adjusted_Momentum'] + 
        df['Momentum_Volume_Surge'] + 
        df['Price_Volume_Divergence'] + 
        df['Price_Amount_Divergence']
    ) * df['Volatility_Adjustment']
    
    return df['Final_Alpha_Factor'].dropna()
```
