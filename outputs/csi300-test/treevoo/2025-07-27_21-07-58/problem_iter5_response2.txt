```python
import pandas as pd

def heuristics_v2(df, n=5, m=5):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Average Volume over n periods
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Average Amount over m periods
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    
    # Calculate Volume and Amount Ratio
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Combine Momentum, Volume, and Amount Adjustment
    df['Combined_Ratio'] = df['Volume_Ratio'] * df['Amount_Ratio']
    df['Adjusted_Price_Momentum'] = df['Price_Momentum'] * df['Combined_Ratio']
    
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Ratio
    df['Close_to_Open_Ratio'] = df['close'] / df['open']
    
    # Calculate Change in High-Low Range
    df['Change_in_High_Low_Range'] = df['Intraday_Range'] - df['Intraday_Range'].shift(1)
    
    # Combine all factors
    df['Alpha_Factor'] = (df['Adjusted_Price_Momentum'] + 
                          df['Intraday_Range'] + 
                          df['Close_to_Open_Ratio'] + 
                          df['Change_in_High_Low_Range'])
    
    return df['Alpha_Factor']
```
