import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on a custom heuristic.
    
    Parameters:
    -----------
    df : pandas.DataFrame
        A DataFrame with (date) as index and columns for open, high, low, close, amount, and volume.
    
    Returns:
    --------
    pandas.Series
        A Series with (date) as index representing the factor values.
    """
    # Calculate the percentage change in close price
    df['close_change'] = df['close'].pct_change()
    
    # Calculate the typical price
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the weighted average of typical price and amount
    df['weighted_typical_price'] = (df['amount'] * df['typical_price']) / df['volume']
    
    # Calculate the moving average of the weighted typical price
    df['ma_weighted_typical_price'] = df['weighted_typical_price'].rolling(window=10).mean()
    
    # Calculate the momentum factor
    df['momentum_factor'] = df['close_change'] * (df['typical_price'] - df['ma_weighted_typical_price'])
    
    # Drop the intermediate columns
    df.drop(['close_change', 'typical_price', 'weighted_typical_price', 'ma_weighted_typical_price'], axis=1, inplace=True)
    
    return df['momentum_factor']
