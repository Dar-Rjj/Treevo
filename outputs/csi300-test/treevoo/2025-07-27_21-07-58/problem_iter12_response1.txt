```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Price Change
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    
    # Calculate High-to-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Previous Day High-to-Low Range
    df['Prev_High_Low_Range'] = df['High_Low_Range'].shift(1)
    
    # Compute High-to-Low Range Momentum
    df['HL_Range_Momentum'] = df['High_Low_Range'] / df['Prev_High_Low_Range']
    
    # Adjust Price Change by Volume
    df['Vol_Adjusted_Price_Change'] = df['Price_Change'] / df['volume']
    
    # Determine Daily Returns
    df['Daily_Return'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute Short-Term Momentum
    df['5D_Sum_Returns'] = df['Daily_Return'].rolling(window=5).sum()
    df['5D_SMA_Close'] = df['close'].rolling(window=5).mean()
    df['Short_Term_Mom'] = df['5D_Sum_Returns'] - df['5D_SMA_Close']
    
    # Compute Long-Term Momentum
    df['20D_Sum_Returns'] = df['Daily_Return'].rolling(window=20).sum()
    df['20D_SMA_Close'] = df['close'].rolling(window=20).mean()
    df['Long_Term_Mom'] = df['20D_Sum_Returns'] - df['20D_SMA_Close']
    
    # Calculate Average True Range for Volatility
    df['True_Range'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], 
                                                               abs(x['high'] - df['close'].shift(1)), 
                                                               abs(x['low'] - df['close'].shift(1))), axis=1)
    df['Avg_True_Range'] = df['True_Range'].rolling(window=14).mean()
    
    # Adjust by Volume
    df['10D_MA_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Ratio'] = df['volume'] / df['10D_MA_Volume']
    
    # Adjust by Amount
    df['7D_MA_Amount'] = df['amount'].rolling(window=7).mean()
    df['Amount_Ratio'] = df['amount'] / df['7D_MA_Amount']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    df['Momentum_Factor'] = (df['5D_Sum_Returns'] * df['HL_Range_Momentum']).rolling(window=5).mean() * \
                            df['10D_MA_Volume']
    df['Combined_Mom'] = df['Short_Term_Mom'] + df['Long_Term_Mom']
    df['Final_Factor'] = df['Momentum_Factor'] * df['Combined_Mom'] * (1 / df['Avg_True_Range']) * \
                         df['Volume_Ratio'] * df['Amount_Ratio']
    
    return df['Final_Factor']
```
