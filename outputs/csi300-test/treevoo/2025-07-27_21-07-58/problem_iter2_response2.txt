import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume
    df['intraday_volume'] = df['volume'] - df['volume'].shift(1)
    
    # Multiply Intraday Return by Intraday Volume
    df['intraday_vol_weighted_return'] = df['intraday_return'] * df['intraday_volume']
    
    # Cumulative Sum Over a Window
    window = 5
    df['cumulative_intraday_vol_weighted_momentum'] = df['intraday_vol_weighted_return'].rolling(window=window).sum()
    
    # Calculate Price Momentum Component
    df['price_change_pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['price_momentum_5_day_avg'] = df['price_change_pct'].rolling(window=5).mean()
    
    # Calculate Volume Surge Component
    df['volume_change_pct'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['volume_surge_10_day_avg'] = df['volume_change_pct'].rolling(window=10).mean()
    
    # Combine Indicators
    df['alpha_factor'] = (df['cumulative_intraday_vol_weighted_momentum'] 
                          * df['price_momentum_5_day_avg'] 
                          * df['volume_surge_10_day_avg'])
    
    # Drop NaN values and return the alpha factor
    df.dropna(inplace=True)
    
    return df['alpha_factor']
