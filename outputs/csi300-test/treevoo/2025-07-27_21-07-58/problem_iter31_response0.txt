import pandas as pd

def heuristics_v2(df):
    # Calculate daily price range
    df['price_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Close return
    df['close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Open-to-Close return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate 5-day average volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate volume-to-price ratio
    df['volume_to_price_ratio'] = df['volume'] / df['close']
    
    # Calculate 5-day average volume-to-price ratio
    df['avg_volume_to_price_ratio_5d'] = df['volume_to_price_ratio'].rolling(window=5).mean()
    
    # Calculate intraday momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Calculate 5-day average intraday momentum
    df['avg_intraday_momentum_5d'] = df['intraday_momentum'].rolling(window=5).mean()
    
    # Calculate gap between today's Open and yesterday's Close
    df['gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate 5-day average gap
    df['avg_gap_5d'] = df['gap'].rolling(window=5).mean()
    
    # Create a composite factor using weighted sum
    weights = {
        'price_range': 0.1,
        'close_to_close_return': 0.3,
        'open_to_close_return': 0.2,
        'avg_volume_5d': 0.05,
        'avg_volume_to_price_ratio_5d': 0.05,
        'avg_intraday_momentum_5d': 0.2,
        'avg_gap_5d': 0.1
    }
    
    df['composite_factor'] = (
        df['price_range'] * weights['price_range'] +
        df['close_to_close_return'] * weights['close_to_close_return'] +
        df['open_to_close_return'] * weights['open_to_close_return'] +
        df['avg_volume_5d'] * weights['avg_volume_5d'] +
        df['avg_volume_to_price_ratio_5d'] * weights['avg_volume_to_price_ratio_5d'] +
        df['avg_intraday_momentum_5d'] * weights['avg_intraday_momentum_5d'] +
        df['avg_gap_5d'] * weights['avg_gap_5d']
    )
    
    # Return the composite factor as a pandas Series
    return df['composite_factor']
