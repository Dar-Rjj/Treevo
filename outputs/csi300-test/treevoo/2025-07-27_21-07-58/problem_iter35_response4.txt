```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Compute 5-Day and 20-Day Moving Average of Price Change
    df['5D_MA_Price_Change'] = df['Daily_Price_Change'].rolling(window=5).mean()
    df['20D_MA_Price_Change'] = df['Daily_Price_Change'].rolling(window=20).mean()
    
    # Determine Daily Returns
    df['Daily_Returns'] = df['close'].pct_change()
    
    # Compute Short-Term and Long-Term Momentum
    df['Short_Term_Momentum'] = df['Daily_Returns'].rolling(window=5).sum()
    df['Long_Term_Momentum'] = df['Daily_Returns'].rolling(window=20).sum()
    
    # Calculate Average True Range for Volatility
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Adjust by Volume
    df['10D_MA_Volume'] = df['volume'].rolling(window=10).mean()
    df['Volume_Ratio'] = df['volume'] / df['10D_MA_Volume']
    
    # Adjust by Amount
    df['7D_MA_Amount'] = df['amount'].rolling(window=7).mean()
    df['Amount_Ratio'] = df['amount'] / df['7D_MA_Amount']
    
    # Introduce High-to-Low Range Momentum Adjustment
    df['HL_Range_Momentum_Adjustment'] = (df['high'] - df['low']) * df['10D_MA_Volume']
    
    # Combine Momentum, Volatility, and Volume-Amount Adjustment
    df['Combined_Factor'] = (df['5D_MA_Price_Change'] + df['Short_Term_Momentum'] + df['Long_Term_Momentum']) / df['ATR']
    df['Adjusted_Combined_Factor'] = df['Combined_Factor'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Adjust Combined Factor by Price Volatility
    df['Price_Volatility'] = df['Daily_Price_Change'].rolling(window=10).std()
    df['Final_Factor'] = df['Adjusted_Combined_Factor'] / df['Price_Volatility']
    
    # Add Combined Price Momentum and Volume Surge
    df['5D_MA_Price_Momentum'] = df['5D_MA_Price_Change'] * df['10D_MA_Volume']
    df['Final_Factor'] += df['5D_MA_Price_Momentum']
    
    # Add Combined High-Low Range and Volume
    df['Change_HL_Range'] = (df['high'] - df['low']).pct_change()
    df['HL_Volume_Adjustment'] = df['Change_HL_Range'] * df['volume'] / df['close']
    df['Final_Factor'] += df['HL_Volume_Adjustment']
    
    # Incorporate High-Low Range Dynamics
    df['5D_MA_HL_Range'] = (df['high'] - df['low']).rolling(window=5).mean()
    df['5D_MA_Change_HL_Range'] = df['5D_MA_HL_Range'].pct_change().rolling(window=5).mean()
    df['HL_Range_Dynamics'] = df['5D_MA_HL_Range'] * df['5D_MA_Change_HL_Range']
    df['Final_Factor'] += df['HL_Range_Dynamics']
    
    # Integrate High-Low Range with Volume
    df['5D_MA_HL_Range_Vol_Adjusted'] = df['5D_MA_HL_Range'] / df['volume']
    df['5D_MA_Volume'] = df['volume'].rolling(window=5).mean()
    df['Integrated_HL_Vol'] = df['5D_MA_HL_Range_Vol_Adjusted'] * df['5D_MA_Volume']
    df['Final_Factor'] += df['Integrated_HL_Vol']
    
    return df['Final_Factor'].dropna()
```
