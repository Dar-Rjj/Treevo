import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    daily_return = df['close'] - df['open']

    # Compute Price Momentum Component
    daily_price_change_percent = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    price_momentum_5day = daily_price_change_percent.rolling(window=5).mean()

    # Compute Volume Adjusted Momentum
    cumulative_volume = df['volume'].rolling(window=21).sum()
    volume_adjusted_momentum = (price_momentum_5day * cumulative_volume) / 21

    # Compute Volume Surge Component
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    volume_surge_10day = daily_volume_change.rolling(window=10).mean()

    # Combine Adjusted Momentum and Volume Surge
    alpha_factor = (volume_adjusted_momentum * volume_surge_10day) * df['close']

    return alpha_factor
