```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of close price to average price
    close_to_avg_ratio = df['close'] / avg_price
    
    # Calculate the volume weighted amount
    volume_weighted_amount = df['amount'] * df['volume']
    
    # Calculate the moving average of the volume weighted amount over a 10-day window
    ma_volume_weighted_amount = volume_weighted_amount.rolling(window=10).mean()
    
    # Calculate the standard deviation of the volume weighted amount over a 10-day window
    std_volume_weighted_amount = volume_weighted_amount.rolling(window=10).std()
    
    # Calculate the z-score for the volume weighted amount
    z_score_volume_weighted_amount = (volume_weighted_amount - ma_volume_weighted_amount) / std_volume_weighted_amount
    
    # Combine the factors into an alpha factor
    alpha_factor = close_to_avg_ratio * z_score_volume_weighted_amount
    
    return alpha_factor
```
