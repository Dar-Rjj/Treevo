import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'] - df['close'].shift(1)
    
    # Compute Daily Price Change Percentage
    df['Daily_Price_Change_Pct'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['5D_Avg_Price_Momentum'] = df['Daily_Price_Change_Pct'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['10D_Avg_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=n).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Price_Vol_Surge'] = df['5D_Avg_Price_Momentum'] * df['10D_Avg_Volume_Change']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['High_Low_Vol'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Calculate Average Volume over n periods
    df['Avg_Volume'] = df['volume'].rolling(window=n).mean()
    
    # Calculate Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume']
    
    # Calculate Average Amount over m periods
    df['Avg_Amount'] = df['amount'].rolling(window=m).mean()
    
    # Calculate Amount Ratio
    df['Amount_Ratio'] = df['amount'] / df['Avg_Amount']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio'] + df['Price_Vol_Surge']
    
    # Final Combined Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_Momentum'] + df['High_Low_Vol']
    
    return df['Alpha_Factor']
