import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].diff()
    
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the money flow, which is typical price * volume
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['money_flow'] = df['typical_price'] * df['volume']
    
    # Calculate the Chaikin Oscillator
    df['money_flow_3_period_EMA'] = df['money_flow'].ewm(span=3, adjust=False).mean()
    df['money_flow_10_period_EMA'] = df['money_flow'].ewm(span=10, adjust=False).mean()
    df['chaikin_oscillator'] = df['money_flow_3_period_EMA'] - df['money_flow_10_period_EMA']
    
    # Calculate the factor as a combination of the chaikin oscillator and daily return
    df['alpha_factor'] = df['chaikin_oscillator'] * df['daily_return']
    
    # Return the alpha factor
    return df['alpha_factor']
