```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Compute Moving Average of Positive Returns
    positive_returns = df['daily_return'].where(df['daily_return'] > 0, 0)
    df['positive_returns_ma'] = positive_returns.rolling(window=21).mean()

    # Volume Adjusted Momentum
    df['cumulative_volume'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['positive_returns_ma'] * df['cumulative_volume']) / 21

    # Calculate 5-Period Price Momentum
    df['5_period_momentum'] = df['close'] / df['close'].shift(5) - 1

    # Adjust by Average Volume
    n = 5
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']

    # Combine Momentum and Volume Adjustment
    df['combined_momentum'] = df['5_period_momentum'] * df['volume_ratio']

    # Calculate 21-Day Price Volatility
    df['21_day_volatility'] = df['close'].rolling(window=21).std()

    # Adjust by Price Volatility
    volatility_threshold = 0.01  # Example threshold
    df['volatility_adjustment'] = df['21_day_volatility'].apply(lambda x: 1.5 if x > volatility_threshold else 1.0)

    # Final Adjusted Momentum Factor
    df['final_factor'] = (df['volume_adjusted_momentum'] + df['combined_momentum']) * df['volatility_adjustment']

    return df['final_factor']
```
