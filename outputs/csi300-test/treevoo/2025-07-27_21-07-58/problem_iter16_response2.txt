```python
import pandas as pd

def heuristics_v2(df, n=5, m=5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range.diff()
    
    # Calculate Price Momentum Component
    daily_price_change_pct = (df['close'].pct_change()) * 100
    
    # Calculate Volume Surge Component
    daily_volume_change_pct = (df['volume'].pct_change()) * 100
    
    # Combine High-Low Range and Volume
    hl_range_volume_combination = (change_high_low_range * df['volume']) / df['close']
    
    # Adjust by Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_ratio = df['volume'] / avg_volume
    
    # Adjust by Average Amount
    avg_amount = df['amount'].rolling(window=m).mean()
    amount_ratio = df['amount'] / avg_amount
    
    # Combine Components
    price_vol_momentum = daily_price_change_pct * daily_volume_change_pct
    combined_factor = hl_range_volume_combination + price_vol_momentum
    adjusted_factor = (daily_price_change_pct * volume_ratio) * amount_ratio
    
    # Final Alpha Factor
    alpha_factor = combined_factor + adjusted_factor
    
    return alpha_factor
```
