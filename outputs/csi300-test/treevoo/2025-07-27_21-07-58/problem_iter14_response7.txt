import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Close_t-1'] = df['close'].shift(1)
    df['1_Period_Price_Momentum'] = df['close'] - df['Close_t-1']
    
    # Compute Daily Price Change Percentage
    df['Daily_Price_Change'] = df['close'].pct_change()
    
    # Compute Average Price Momentum over 5 Days
    df['5_Day_Avg_Price_Momentum'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['Prev_Volume'] = df['volume'].shift(1)
    df['Daily_Volume_Change'] = df['volume'] - df['Prev_Volume']
    
    # Compute Average Volume Change over 10 Days
    df['10_Day_Avg_Volume_Change'] = df['Daily_Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Price_Volume_Surge'] = df['5_Day_Avg_Price_Momentum'] * df['10_Day_Avg_Volume_Change']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Prev_High_Low_Range'] = df['High_Low_Range'].shift(1)
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['Prev_High_Low_Range']
    
    # Combine High-Low Range and Volume
    df['High_Low_Volume_Adjusted'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 20
    df['Average_Volume_n'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume_n']
    
    # Adjust by Average Amount
    m = 20
    df['Average_Amount_m'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount_m']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['1_Period_Price_Momentum'] * df['Volume_Ratio'] * df['Amount_Ratio']
    df['Combined_Factor'] = df['Price_Volume_Surge'] + df['Adjusted_Momentum']
    
    # Final Combined Alpha Factor
    return df['Combined_Factor'].dropna()
