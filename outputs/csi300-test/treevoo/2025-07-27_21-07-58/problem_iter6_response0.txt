```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'] - df['close'].shift(1)
    
    # Compute Moving Average of Positive Returns
    positive_returns = df['daily_return'].where(df['daily_return'] > 0, 0)
    df['positive_returns_ma_21'] = positive_returns.rolling(window=21).mean()
    
    # Volume Adjusted Momentum
    df['cumulative_volume_21'] = df['volume'].rolling(window=21).sum()
    df['volume_adjusted_momentum'] = (df['positive_returns_ma_21'] * df['cumulative_volume_21']) / 21
    df['volume_adjusted_momentum'] = df['volume_adjusted_momentum'].fillna(0)  # Ensure no division by zero
    
    # Calculate 5-Period Price Momentum
    df['price_momentum_5'] = df['close'] / df['close'].shift(5) - 1
    
    # Adjust by Average Volume
    n = 5
    df['average_volume_5'] = df['volume'].rolling(window=n).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume_5']
    
    # Combine Momentum and Volume Adjustment
    df['combined_momentum'] = df['price_momentum_5'] * df['volume_ratio']
    
    # Final Adjusted Momentum Factor
    df['adjusted_momentum_factor'] = df['volume_adjusted_momentum'] + df['combined_momentum']
    
    return df['adjusted_momentum_factor']
```
