```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    change_high_low_range = high_low_range - high_low_range.shift(1)
    
    # Calculate Price Momentum Component
    daily_price_change_pct = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    avg_5_day_price_momentum = daily_price_change_pct.rolling(window=5).mean()
    
    # Calculate Volume Surge Component
    daily_volume_change_pct = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    avg_10_day_volume_change = daily_volume_change_pct.rolling(window=10).mean()
    
    # Combine High-Low Range and Volume
    high_low_volume = (change_high_low_range * df['volume']) / df['close']
    
    # Compare with Recent Volume
    volume_t = df['volume']
    avg_n_period_volume = df['volume'].rolling(window=10).mean()
    volume_ratio = volume_t / avg_n_period_volume
    
    # Compare with Recent Amount
    amount_t = df['amount']
    avg_m_period_amount = df['amount'].rolling(window=10).mean()
    amount_ratio = amount_t / avg_m_period_amount
    
    # Introduce Weighting Factors
    w1, w2, w3, w4, w5 = 0.2, 0.3, 0.1, 0.2, 0.2
    
    # Combine Weighted Components
    weighted_high_low_volume = high_low_volume * w1
    weighted_price_momentum = avg_5_day_price_momentum * w2
    weighted_volume_change = avg_10_day_volume_change * w3
    weighted_volume_ratio = volume_ratio * w4
    weighted_amount_ratio = amount_ratio * w5
    combined_weighted_factors = (weighted_high_low_volume + 
                                 weighted_price_momentum + 
                                 weighted_volume_change + 
                                 weighted_volume_ratio + 
                                 weighted_amount_ratio)
    
    # Calculate Momentum and Volatility
    daily_returns = df['close'].pct_change()
    short_term_momentum = daily_returns.rolling(window=5).sum() - df['close'].ewm(span=5).mean()
    long_term_momentum = daily_returns.rolling(window=20).sum() - df['close'].ewm(span=20).mean()
    true_range = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), abs(df['low'] - df['close'].shift(1)))
    avg_true_range = true_range.ewm(span=14).mean()
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = (short_term_momentum * 0.7 + long_term_momentum * 0.3) / avg_true_range
    
    # Add Trend-Following Component
    sma_50 = df['close'].rolling(window=50).mean()
    trend_following_multiplier = np.where(df['close'] > sma_50, 1.5, 1.0)
    
    # Final Synthesized Alpha Factor
    alpha_factor = combined_weighted_factors * combined_momentum_volatility * trend_following_multiplier
    return alpha_factor
```
