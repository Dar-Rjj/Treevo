```python
import pandas as pd

def heuristics_v2(df, window=5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume Change
    intraday_volume = df['volume'] - df['volume'].shift(1)
    
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Previous Day's Intraday Price Range
    prev_intraday_range = (df['high'].shift(1) - df['low'].shift(1))
    
    # Calculate Intraday Price Range Change
    range_change = intraday_range - prev_intraday_range
    
    # Calculate Volume Momentum Indicator
    volume_momentum = df['volume'] / df['volume'].shift(1)
    
    # Combine Indicators: Volume Adjusted High-Low Momentum
    volume_adjusted_momentum = range_change * volume_momentum
    
    # Combine Intraday Return with Volume Adjusted High-Low Momentum
    combined_factor = (intraday_return * intraday_volume) + volume_adjusted_momentum
    
    # Cumulative Sum Over a Window
    cumulative_factor = combined_factor.rolling(window=window).sum()
    
    return cumulative_factor
```
