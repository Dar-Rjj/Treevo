```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate 1-Period Price Momentum
    df['Close_t-1'] = df['close'].shift(1)
    df['Price_Momentum_1d'] = (df['close'] - df['Close_t-1']) / df['Close_t-1']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Calculate Daily Price Change Percentage
    df['Daily_Price_Change'] = (df['close'] - df['Close_t-1']) / df['Close_t-1']
    
    # Compute Average Price Momentum over 5 Days
    df['Avg_Price_Momentum_5d'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change Percentage
    df['Volume_t-1'] = df['volume'].shift(1)
    df['Daily_Volume_Change'] = (df['volume'] - df['Volume_t-1']) / df['Volume_t-1']
    
    # Compute Average Volume Change over 10 Days
    df['Avg_Volume_Change_10d'] = df['Daily_Volume_Change'].rolling(window=n).mean()
    
    # Combine High-Low Range and Volume
    df['High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Calculate Average Volume over n periods
    df['Avg_Volume'] = df['volume'].rolling(window=n).mean()
    
    # Compare with Recent Volume
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume']
    
    # Calculate Average Amount over m periods
    df['Avg_Amount'] = df['amount'].rolling(window=m).mean()
    
    # Compare with Recent Amount
    df['Amount_Ratio'] = df['amount'] / df['Avg_Amount']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Alpha_Factor'] = df['Price_Momentum_1d'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    return df['Alpha_Factor']
```
