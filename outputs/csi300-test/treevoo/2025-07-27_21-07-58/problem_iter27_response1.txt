```python
import pandas as pd

def heuristics_v2(df):
    # Calculate simple daily return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate cumulative 5-day return
    df['cumulative_5d_return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Daily range
    df['daily_range'] = df['high'] - df['low']
    
    # Average true range over 14 days
    df['true_range'] = df[['high' - 'low', 
                           abs('high' - df['close'].shift(1)), 
                           abs('low' - df['close'].shift(1))]].max(axis=1)
    df['avg_true_range_14d'] = df['true_range'].rolling(window=14).mean()
    
    # Change in daily trading volume
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Volume weighted by price movement
    df['volume_weighted_price_move'] = (df['close'] - df['open']) * df['volume']
    
    # Construct the factor
    factor_values = (
        df['daily_return'] +
        df['cumulative_5d_return'] +
        df['daily_range'] +
        df['avg_true_range_14d'] +
        df['volume_change'] +
        df['volume_weighted_price_move']
    )
    
    return factor_values
```
