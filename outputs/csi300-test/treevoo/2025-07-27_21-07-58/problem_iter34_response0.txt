import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum Factors
    df['short_term_momentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['medium_term_momentum'] = (df['close'] / df['close'].shift(20)) - 1
    
    # Identify Volume-Weighted Price Actions
    df['weighted_avg_price'] = ((df['high'] + df['low']) / 2) * df['volume']
    df['5d_moving_avg_weighted_price'] = df['weighted_avg_price'].rolling(window=5).mean()
    df['cumulative_vol_adjusted_return'] = (df['close'] - df['open']) * df['volume']
    df['cumulative_vol_adjusted_return'] = df['cumulative_vol_adjusted_return'].rolling(window=10, min_periods=1).sum()
    
    # Analyze Trading Volume Patterns
    df['ratio_to_5d_avg_volume'] = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Evaluate Volatility and Range
    df['daily_price_range'] = df['high'] - df['low']
    df['relative_daily_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    df['std_close_20_days'] = df['close'].rolling(window=20).std()
    
    # Combine all factors into a single composite factor
    df['composite_factor'] = (
        df['short_term_momentum'] +
        df['medium_term_momentum'] +
        df['5d_moving_avg_weighted_price'] +
        df['cumulative_vol_adjusted_return'] +
        df['ratio_to_5d_avg_volume'] +
        df['daily_price_range'] +
        df['relative_daily_range'] +
        df['std_close_20_days']
    )
    
    # Return the composite factor as a pandas Series
    return df['composite_factor']
