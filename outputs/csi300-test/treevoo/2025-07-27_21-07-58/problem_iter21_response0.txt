```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum_1D'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute Daily Price Change Percentage
    df['Daily_Price_Change'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute Average Price Momentum over 5 Days
    df['Avg_Price_Momentum_5D'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Calculate Daily Volume Change
    df['Daily_Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Compute Average Volume Change over 10 Days
    df['Avg_Volume_Change_10D'] = df['Daily_Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Combined_Price_Vol_Surge'] = df['Avg_Price_Momentum_5D'] * df['Avg_Volume_Change_10D']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['Combined_High_Low_Volume'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Adjust by Average Volume
    n = 20  # Number of periods for average volume
    df['Avg_Volume_n'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Avg_Volume_n']
    
    # Adjust by Average Amount
    m = 20  # Number of periods for average amount
    df['Avg_Amount_m'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Avg_Amount_m']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Adjusted_Momentum'] = df['Price_Momentum_1D'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Incorporate High-Low Range Change
    df['Combined_Factor'] = df['Combined_Price_Vol_Surge'] + df['Adjusted_Momentum'] + df['Combined_High_Low_Volume']
    
    # Final Combined Alpha Factor
    alpha_factor = df['Combined_Factor']
    
    return alpha_factor
```
