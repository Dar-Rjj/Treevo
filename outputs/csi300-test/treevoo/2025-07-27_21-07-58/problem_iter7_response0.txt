```python
import pandas as pd

def heuristics_v2(df, window=5):
    # Calculate Intraday Return
    intraday_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volume
    intraday_volume = df['volume']
    
    # Multiply Intraday Return by Intraday Volume
    volume_weighted_return = intraday_return * intraday_volume
    
    # Cumulative Sum Over a Window
    cumulative_momentum = volume_weighted_return.rolling(window=window).sum()
    
    # Adjust by Average Volume
    average_volume = df['volume'].rolling(window=window).mean()
    volume_ratio = df['volume'] / average_volume
    
    # Combine Momentum and Volume Adjustment
    final_factor = cumulative_momentum * volume_ratio
    
    return final_factor
```
