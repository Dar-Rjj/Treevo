```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10):
    # Calculate Momentum Indicator
    df['Daily_Return'] = df['close'].pct_change()
    df['Momentum'] = df['Daily_Return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Calculate Average True Range
    df['True_Range'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], 
                                                                        abs(x['high'] - df['close'].shift(1)), 
                                                                        abs(x['low'] - df['close'].shift(1))), axis=1)
    df['Average_True_Range'] = df['True_Range'].rolling(window=14).mean()
    
    # Combine Momentum and Volatility
    df['Weighted_Momentum'] = 0.7 * df['Momentum']
    df['Inv_ATR'] = 1 / df['Average_True_Range']
    df['Combined_MV'] = df['Weighted_Momentum'] * df['Inv_ATR']

    # Adjust by Volume and Amount
    df['Average_Volume'] = df['volume'].rolling(window=n).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Volume']
    df['Average_Amount'] = df['amount'].rolling(window=m).mean()
    df['Amount_Ratio'] = df['amount'] / df['Average_Amount']
    
    # Price Oscillator
    df['EMA_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['EMA_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['Price_Oscillator'] = df['EMA_12'] - df['EMA_26']
    
    # Adjusted High-Low Range and Volume Combination
    df['Change_High_Low_Range'] = (df['high'] - df['low']) - (df['high'].shift(1) - df['low'].shift(1))
    df['HLRV_Adjustment'] = df['Change_High_Low_Range'] * (df['volume'] / df['close'])
    
    # Trend Following Multiplier
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Trend_Following_Multiplier'] = df.apply(lambda x: 1 if x['close'] > x['SMA_50'] else 0, axis=1)
    
    # Final Alpha Factor
    df['Final_Alpha'] = (df['Combined_MV'] * df['Volume_Ratio'] * df['Amount_Ratio']) + \
                        df['Price_Oscillator'] + df['HLRV_Adjustment'] + \
                        (df['HLRV_Adjustment'] * df['Trend_Following_Multiplier'])

    return df['Final_Alpha']
```
