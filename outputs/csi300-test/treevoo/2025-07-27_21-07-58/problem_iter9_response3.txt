```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute daily returns
    df['returns'] = df['close'].pct_change()
    
    # Momentum Indicators
    df['short_momentum'] = df['returns'].rolling(window=5).sum()
    df['long_momentum'] = df['returns'].rolling(window=20).sum()
    
    # Average True Range
    df['true_range'] = np.maximum(
        df[['high', 'low']].diff(axis=1).iloc[:, 0].abs(),
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ).max(axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Ratio
    df['close_to_open_ratio'] = df['close'] / df['open']
    
    # Volume Momentum
    df['volume_momentum'] = df['volume'] * (df['close'] - df['open'])
    
    # Combine Momentum and Volatility
    df['combined_momentum_volatility'] = (df['short_momentum'] + df['long_momentum']) / df['avg_true_range']
    
    # Intraday Momentum
    df['intraday_momentum'] = ((df['close'] - df['open']) / df['open']).rolling(window=5).sum()
    
    # Combine Intraday Factors
    df['intraday_factors'] = df['intraday_range'] + df['close_to_open_ratio'] + df['volume_momentum'] + df['intraday_momentum']
    
    # Final Combined Alpha Factor
    df['alpha_factor'] = df['combined_momentum_volatility'] + df['intraday_factors']
    
    return df['alpha_factor'].dropna()
```
