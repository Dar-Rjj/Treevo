import pandas as pd

def heuristics_v2(df):
    # Calculate 1-Period Price Momentum
    df['Price_Momentum'] = df['close'] / df['close'].shift(1) - 1
    
    # Compute 5-Day Moving Average of Price Momentum
    df['5DMA_Price_Momentum'] = df['Price_Momentum'].rolling(window=5).mean()
    
    # Compute 10-Day Moving Average Volume Change
    df['Volume_Change'] = df['volume'] / df['volume'].shift(1) - 1
    df['10DMA_Volume_Change'] = df['Volume_Change'].rolling(window=10).mean()
    
    # Combine Price Momentum and Volume Surge
    df['Combined_Price_Vol'] = df['5DMA_Price_Momentum'] * df['10DMA_Volume_Change']
    
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Change in High-Low Range
    df['Change_High_Low_Range'] = df['High_Low_Range'] - df['High_Low_Range'].shift(1)
    
    # Combine High-Low Range and Volume
    df['Combined_High_Low_Vol'] = (df['Change_High_Low_Range'] * df['volume']) / df['close']
    
    # Combine Adjusted Momentum, Volume, and Amount
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['Adjusted_Momentum'] = df['Price_Momentum'] * df['Volume_Ratio']
    df['Combined_Adjusted_Mom_Vol_Amt'] = (df['Price_Momentum'] * df['Volume_Ratio']) + df['Combined_Price_Vol'] + df['Combined_High_Low_Vol']
    
    # Incorporate High-Low Range Dynamics
    df['5DMA_High_Low_Range'] = df['High_Low_Range'].rolling(window=5).mean()
    df['5DMA_Change_High_Low_Range'] = df['Change_High_Low_Range'].rolling(window=5).mean()
    df['High_Low_Dynamics'] = df['5DMA_High_Low_Range'] * df['5DMA_Change_High_Low_Range']
    
    # Final Combined Alpha Factor
    df['Alpha_Factor'] = df['Combined_Adjusted_Mom_Vol_Amt'] + df['High_Low_Dynamics']
    
    return df['Alpha_Factor'].dropna()
