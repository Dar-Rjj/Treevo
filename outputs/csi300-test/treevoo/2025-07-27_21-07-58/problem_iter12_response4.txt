```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Determine Daily Returns
    df['daily_return'] = df['close'].pct_change()
    
    # Compute Short-Term Momentum
    df['short_term_momentum'] = df['daily_return'].rolling(window=5).sum() - df['close'].ewm(span=5, adjust=False).mean()
    
    # Compute Long-Term Momentum
    df['long_term_momentum'] = df['daily_return'].rolling(window=20).sum() - df['close'].ewm(span=20, adjust=False).mean()
    
    # Determine True Range
    df['true_range'] = df[['high', 'low', 'close']].diff().abs().max(axis=1)
    
    # Compute 14-day Exponential Moving Average of True Range
    df['avg_true_range'] = df['true_range'].ewm(span=14, adjust=False).mean()
    
    # Weighted Sum of Short-Term and Long-Term Momentum
    df['weighted_momentum'] = (df['short_term_momentum'] * 0.7) + (df['long_term_momentum'] * 0.3)
    
    # Multiply by Inverse of Average True Range
    df['alpha_factor'] = df['weighted_momentum'] / df['avg_true_range']
    
    # Add a trend-following component
    df['50_day_sma'] = df['close'].rolling(window=50).mean()
    df['trend_multiplier'] = np.where(df['close'] > df['50_day_sma'], 1.5, 1.0)
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['alpha_factor'] * df['trend_multiplier']
    
    return df['final_alpha_factor']
```
