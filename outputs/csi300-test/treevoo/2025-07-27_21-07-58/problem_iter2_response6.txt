```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Determine Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Compute Short-Term Momentum
    short_term_momentum = df['daily_return'].rolling(window=5).sum() - df['close'].rolling(window=5).mean()
    
    # Compute Long-Term Momentum
    long_term_momentum = df['daily_return'].rolling(window=20).sum() - df['close'].rolling(window=20).mean()

    # Calculate Average True Range
    df['true_range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    average_true_range = df['true_range'].rolling(window=14).mean()

    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Close-to-Open Ratio
    close_to_open_ratio = df['close'] / df['open']

    # Incorporate Volume Momentum
    volume_momentum = (df['close'] - df['open']) * df['volume']

    # Combine Momentum and Volatility
    combined_momentum_volatility = (short_term_momentum + long_term_momentum) * (1 / average_true_range)

    # Combine Intraday Factors
    combined_intraday_factors = intraday_range + close_to_open_ratio + volume_momentum

    # Final Combined Alpha Factor
    alpha_factor = combined_momentum_volatility + combined_intraday_factors

    return alpha_factor.dropna()
```
