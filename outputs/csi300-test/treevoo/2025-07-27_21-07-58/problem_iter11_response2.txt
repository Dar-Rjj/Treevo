```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Price Change Percentage
    df['price_change_pct'] = df['close'].pct_change()

    # Compute Average Price Momentum over 5 Days
    df['5_day_price_momentum'] = df['price_change_pct'].rolling(window=5).mean()

    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].diff()

    # Compute Average Volume Change over 10 Days
    df['10_day_volume_change'] = df['volume_change'].rolling(window=10).mean()

    # Combine Price Momentum and Volume Surge
    df['momentum_volume_surge'] = df['5_day_price_momentum'] * df['10_day_volume_change']

    # Calculate 1-Period Price Momentum
    df['1_period_momentum'] = df['close'].pct_change(periods=1)

    # Adjust by Average Volume
    n = 21  # number of periods for average volume
    df['average_volume'] = df['volume'].rolling(window=n, min_periods=1).mean()
    df['volume_ratio'] = df['volume'] / df['average_volume']
    df['volume_adjusted_momentum'] = df['1_period_momentum'] * df['volume_ratio']

    # Calculate Cumulative Volume
    df['cumulative_volume'] = df['volume'].rolling(window=21, min_periods=1).sum()

    # Calculate Moving Sum of Positive Returns
    positive_returns = (df['price_change_pct'] > 0) * 1
    moving_sum_positive_returns = positive_returns.rolling(window=21, min_periods=1).sum()

    # Volume Adjusted Momentum Component
    df['volume_adjusted_momentum_component'] = (moving_sum_positive_returns * df['cumulative_volume']) / 21

    # Final Combined Alpha Factor
    df['alpha_factor'] = df['momentum_volume_surge'] + df['volume_adjusted_momentum'] + df['volume_adjusted_momentum_component']

    # Return the alpha factor
    return df['alpha_factor']
```
