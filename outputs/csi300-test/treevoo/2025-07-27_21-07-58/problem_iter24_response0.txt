```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Returns
    df['daily_returns'] = df['close'] / df['close'].shift(1) - 1
    
    # Short-Term Momentum (5-day sum)
    df['short_term_mom'] = df['daily_returns'].rolling(window=5).sum()
    
    # Long-Term Momentum (20-day sum)
    df['long_term_mom'] = df['daily_returns'].rolling(window=20).sum()
    
    # Calculate True Range
    df['true_range'] = pd.concat([
        (df['high'] - df['low']),
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ], axis=1).max(axis=1)
    
    # Average True Range (14-day SMA)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Weighted Sum of Momentum
    df['weighted_momentum'] = df['short_term_mom'] * 0.6 + df['long_term_mom'] * 0.4
    
    # 50-day Exponential Moving Average of Close Price
    df['50_ema_close'] = df['close'].ewm(span=50, adjust=False).mean()
    
    # Apply Trend Multiplier
    trend_multiplier = df['close'] > df['50_ema_close']
    df['trend_adjusted_mom'] = df['weighted_momentum'] * (1.5 if trend_multiplier.any() else 1)
    
    # Calculate Money Flow Index (MFI)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['raw_money_flow'] = df['typical_price'] * df['volume']
    
    positive_flow = df['typical_price'] > df['typical_price'].shift(1)
    negative_flow = df['typical_price'] < df['typical_price'].shift(1)
    
    df.loc[positive_flow, 'positive_money_flow'] = df['raw_money_flow']
    df.loc[negative_flow, 'negative_money_flow'] = df['raw_money_flow']
    
    df['positive_money_flow'] = df['positive_money_flow'].fillna(0).rolling(window=14).sum()
    df['negative_money_flow'] = df['negative_money_flow'].fillna(0).rolling(window=14).sum()
    
    df['mfi'] = 100 - (100 / (1 + (df['positive_money_flow'] / df['negative_money_flow'])))
    
    # 30-day Simple Moving Average of MFI
    df['sma_mfi'] = df['mfi'].rolling(window=30).mean()
    
    # Enhance Combined Momentum with MFI
    mfi_trend_multiplier = df['mfi'] > df['sma_mfi']
    df['final_factor'] = df['trend_adjusted_mom'] * (1.2 if mfi_trend_multiplier.any() else 1)
    
    return df['final_factor']
```
