import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate daily returns using Close price
    data['Daily_Return'] = data['close'].pct_change()
    
    # Compute short-term momentum (5-day sum of daily returns - 5-day EMA of close)
    data['Short_Term_Momentum'] = data['Daily_Return'].rolling(window=5).sum() - data['close'].ewm(span=5, adjust=False).mean()
    
    # Compute long-term momentum (20-day sum of daily returns - 20-day EMA of close)
    data['Long_Term_Momentum'] = data['Daily_Return'].rolling(window=20).sum() - data['close'].ewm(span=20, adjust=False).mean()
    
    # Determine True Range
    data['True_Range'] = data[['high', 'low', 'close']].join(data['close'].shift()).apply(
        lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1)
    
    # Compute 14-day Exponential Moving Average of True Range
    data['Average_True_Range'] = data['True_Range'].ewm(span=14, adjust=False).mean()
    
    # Combine weighted sum of short-term and long-term momentum
    data['Weighted_Momentum'] = data['Short_Term_Momentum'] * 0.7 + data['Long_Term_Momentum'] * 0.3
    
    # Generate the final alpha factor by multiplying the weighted momentum by the inverse of the Average True Range
    data['Alpha_Factor'] = data['Weighted_Momentum'] / data['Average_True_Range']
    
    return data['Alpha_Factor'].dropna()
