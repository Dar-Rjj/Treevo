import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Spikes
    volume_spikes = (df['volume'] > df['volume'].shift(1)).astype(int)
    price_spikes = (df['close'] > df['close'].shift(1)).astype(int)
    combined_spikes = volume_spikes + price_spikes
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=30).std()
    
    # Scale Intraday Momentum
    scaled_momentum = intraday_range / (combined_spikes + 1e-6)  # Avoid division by zero
    scaled_momentum = np.clip(scaled_momentum, -1, 1)  # Cap at a threshold
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday_feature = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=5).mean()
    
    # Adjust for Volume Spikes and Weight by Volume
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(lower=1, upper=3)  # Cap at threshold
    adjusted_feature = combined_intraday_feature * volume_spike_factor
    average_volume = df['volume'].rolling(window=30).mean()
    relative_volume = df['volume'] / average_volume
    final_factor = adjusted_feature * relative_volume
    
    return final_factor
