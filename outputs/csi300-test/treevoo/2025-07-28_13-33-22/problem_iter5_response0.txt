```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Identify Price Spikes
    price_spike = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Price Spikes
    combined_spike = volume_spike + price_spike
    
    # Scale Intraday Momentum
    scaled_intraday_momentum = intraday_range / combined_spike
    scaled_intraday_momentum = scaled_intraday_momentum.clip(upper=5)  # Cap at a threshold to avoid extreme values
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Intraday and Price Movements
    n = 5  # Define the number of periods for averaging
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = avg_intraday_range * (close_to_open_return + price_change)
    
    # Weight by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    weighted_combined_factor = combined_factor * volume_weight
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['close'].shift(1)
    
    # Adjust for Volume Volatility
    daily_volume_change = df['volume'] - df['volume'].shift(1)
    std_volume_change = daily_volume_change.rolling(window=n).std()
    adjusted_daily_returns = daily_returns / std_volume_change
    
    # Apply Momentum Indicator
    momentum_sum = adjusted_daily_returns.rolling(window=20).sum()
    
    # Finalize the Alpha Factor
    alpha_factor = momentum_sum * weighted_combined_factor
    
    return alpha_factor
```
