```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Weighted Average Log Return
    n = 20  # Number of days for the rolling window
    df['weighted_log_return'] = (df['log_return'] * df['volume']).rolling(window=n).sum() / df['volume'].rolling(window=n).sum()
    
    # Adjust for Short-Term Dynamics
    df['high_low_range'] = df['high'] - df['low']
    df['true_range'] = np.maximum(df['high'], df['close'].shift(1)) - np.minimum(df['low'], df['close'].shift(1))
    df['daily_volatility'] = (df['high'] - df['low']) + (df['high'] - df['close'].shift(1)) + (df['close'].shift(1) - df['low'])
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    df['adjusted_weighted_log_return'] = df['weighted_log_return'] - df['high_low_range'] - df['true_range'] - df['daily_volatility'] - df['volume_change']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1) * 1.5).astype(int)
    df['price_spike'] = (np.abs(df['close'] - df['close'].shift(1)) > (df['close'].shift(1) * 0.03)).astype(int)
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    
    # Introduce Volatility Adjustment
    df['std_log_returns'] = df['log_return'].rolling(window=20).std()
    df['intraday_momentum_adj'] = df['intraday_momentum'] / (df['std_log_returns'] + 1e-6)
    df['intraday_momentum_scaled'] = df['intraday_momentum_adj'] / (df['combined_spike'] + 1e-6)
    df['intraday_momentum_capped'] = np.clip(df['intraday_momentum_scaled'], -3, 3)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['intraday_features'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    
    # Calculate Close-to-Open Return
    df['close_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=20).mean()
    df['combined_factor'] = df['avg_intraday_range'] * (df['close_open_return'] + df['price_change'])
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = np.abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=20).mean()
    df['enhanced_factor'] = df['combined_factor'] * df['avg_abs_price_change']
    
    # Adjust Combined Factor by Volume Spike
    df['enhanced_factor_adj'] = df['enhanced_factor'] / (df['volume_spike'] + 1e-6)
    df['factor_capped'] = np.clip(df['enhanced_factor_adj'], -3, 3)
    
    # Weight by Relative Volume
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] / (df['avg_volume'] + 1e-6)
    df['final_factor'] = df['factor_capped'] * df['relative_volume']
    
    return df['final_factor']
```
