```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, k=10):
    # Intraday Range
    intraday_range = df['high'] - df['low']

    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Price Change
    price_change = df['close'].diff().fillna(0)

    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2

    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1).fillna(0)

    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap

    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = (avg_intraday_range * (close_to_open_return + price_change)) + combined_intraday_features

    # Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs().fillna(0)
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    recent_volatility = enhanced_intraday_factor * avg_abs_price_change

    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=k).mean()
    volume_adjusted_factor = df['volume'] / avg_volume
    volume_adjusted_factor = volume_adjusted_factor.apply(lambda x: 1 if x > 1 else 0.5)
    volume_adjusted_factor = enhanced_intraday_factor * volume_adjusted_factor

    # Weighted Volatility
    weighted_volatility = recent_volatility * volume_adjusted_factor

    return weighted_volatility
```
