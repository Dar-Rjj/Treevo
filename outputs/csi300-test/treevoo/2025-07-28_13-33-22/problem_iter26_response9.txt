```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, k=30, p=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (np.abs(df['close'] - df['close'].shift(1)) > np.abs(df['close'].shift(1) - df['close'].shift(2))).astype(int)
    
    # Introduce Volatility Adjustment
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['std_log_returns'] = df['log_returns'].rolling(window=n).std()
    df['intraday_momentum_vol_adj'] = df['intraday_range'] / df['std_log_returns']
    
    # Scale Intraday Momentum
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    df['scaled_intraday_momentum'] = df['intraday_momentum_vol_adj'] / df['combined_spike']
    df['scaled_intraday_momentum'] = df['scaled_intraday_momentum'].clip(lower=-5, upper=5)
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Combined Intraday Features
    df['combined_intraday_features'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = np.abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=m).mean()
    df['enhanced_intraday_factor'] = df['enhanced_intraday_factor'] * df['avg_abs_price_change']
    
    # Weight by Volume
    df['avg_volume'] = df['volume'].rolling(window=k).mean()
    df['volume_weight'] = np.where(df['volume'] > df['avg_volume'], 1, 0.5)
    df['enhanced_intraday_factor'] = df['enhanced_intraday_factor'] * df['volume_weight']
    
    # Adjust for Combined Volume and Price Spikes
    df['combined_spike_factor'] = (df['volume'] / df['volume'].shift(1)) * (df['price_change'] / df['price_change'].shift(1))
    df['adjusted_enhanced_intraday_factor'] = df['enhanced_intraday_factor'] / df['combined_spike_factor']
    
    # Add Recent Amount Trend
    df['avg_amount'] = df['amount'].rolling(window=p).mean()
    df['recent_amount_trend'] = df['amount'] / df['avg_amount']
    df['final_factor'] = df['adjusted_enhanced_intraday_factor'] * df['recent_amount_trend']
    
    return df['final_factor']
```
