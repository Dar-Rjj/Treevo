import pandas as pd

def heuristics_v2(df, n=14):
    # Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    atr = df['high'] - df['low']
    atr = atr.rolling(window=n).mean()
    
    # Overnight Return
    overnight_return = df['open'].shift(-1) - df['close']
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    vati = df['amount'] / df['volume']
    vati_sum = vati.rolling(window=n).sum()
    
    # Price Trend Reversal
    price_trend_reversal = (df['close'] - df['close'].shift(1)) * (vati / vati.mean())
    
    # Combined Alpha Factor
    combined_alpha_factor = (intraday_volatility * overnight_return + daily_momentum + price_trend_reversal) * (vati / vati.mean())
    
    # Volume Adjustment
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_adjustment = df['volume'] / avg_volume
    
    # Final Alpha Factor with Volume Adjustment
    final_alpha_factor = combined_alpha_factor * volume_adjustment
    
    return final_alpha_factor
