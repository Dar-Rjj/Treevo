```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price range and normalize it with the close price
    price_range = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted price change
    volume_weighted_price_change = (df['close'] - df['open']) * df['volume']
    
    # Calculate the amount to volume ratio
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Combine the factors using a weighted sum
    alpha_factor = 0.4 * price_range + 0.3 * volume_weighted_price_change + 0.3 * amount_volume_ratio
    
    return alpha_factor
```
