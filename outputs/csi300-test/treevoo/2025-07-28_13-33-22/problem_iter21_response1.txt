import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (abs(df['close'] - df['close'].shift(1)) > 0.05 * df['close'].shift(1)).astype(int)
    df['combined_spikes'] = df['volume_spike'] + df['price_spike']
    
    # Introduce Volatility Adjustment
    df['log_returns'] = (df['close'] / df['close'].shift(1)).apply(lambda x: np.log(x))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    
    # Scale Intraday Momentum
    df['scaled_intraday_momentum'] = df['intraday_range'] / (df['combined_spikes'] + 1)
    df['scaled_intraday_momentum'] = df['scaled_intraday_momentum'].clip(upper=3*df['intraday_range'].mean())
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['combined_intraday_feature'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Enhance with Recent Volatility
    df['absolute_price_change'] = abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['absolute_price_change'].rolling(window=10).mean()
    
    # Adjust for Volume Spikes
    df['volume_spike_factor'] = 1 + 0.1 * df['volume_spike']
    df['adjusted_combined_feature'] = df['combined_intraday_feature'] * df['volume_spike_factor']
    df['adjusted_combined_feature'] = df['adjusted_combined_feature'].clip(upper=3*df['combined_intraday_feature'].mean())
    
    # Weight by Volume
    df['weighted_feature'] = df['adjusted_combined_feature'] * df['volume']
    
    # Incorporate Accumulated Volume
    df['accumulated_volume'] = df['volume'].rolling(window=20).sum()
    df['normalized_feature'] = df['weighted_feature'] / df['accumulated_volume']
    
    # Apply Moving Average Smoothing
    df['final_alpha'] = df['normalized_feature'].rolling(window=5).mean()
    
    return df['final_alpha'].dropna()
