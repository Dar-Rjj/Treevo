```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute weighted average log return using volume as weights
    n = 5  # Number of days for the rolling window
    df['weighted_log_return'] = (df['log_return'].rolling(window=n).apply(
        lambda x: np.sum(x * df['volume'][x.index]), raw=False
    ) / df['volume'].rolling(window=n).sum())
    
    # Adjust for short-term dynamics
    # Calculate high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate daily volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Subtract adjustments from weighted average log return
    df['factor'] = df['weighted_log_return'] - df['high_low_range'] - df['daily_volatility'] - df['volume_change']
    
    # Return the factor values
    return df['factor'].dropna()
```
