import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    df['volume_spike'] = (df['volume'] / df['volume'].shift(1)) - 1
    df['price_spike'] = (df['close'] / df['close'].shift(1)) - 1
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    
    # Introduce Volatility Adjustment
    df['log_returns'] = (df['close'] / df['close'].shift(1)).apply(lambda x: np.log(x))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    df['intraday_momentum_vol_adj'] = df['intraday_momentum'] * (1 / df['volatility'])
    
    # Scale Intraday Momentum
    df['scaled_intraday_momentum'] = df['intraday_momentum_vol_adj'] / df['combined_spike']
    df['scaled_intraday_momentum'] = df['scaled_intraday_momentum'].clip(lower=-10, upper=10)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['intraday_features'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    df['average_intraday_range'] = df['intraday_range'].rolling(window=5).mean()
    df['combined_factors'] = df['average_intraday_range'] * (df['close_to_open_return'] + df['price_change'])
    
    # Enhance with Recent Volatility
    df['absolute_price_change'] = (df['close'] - df['close'].shift(1)).abs()
    df['average_absolute_price_change'] = df['absolute_price_change'].rolling(window=10).mean()
    df['enhanced_factor'] = df['combined_factors'] * df['average_absolute_price_change']
    
    # Adjust Combined Factor by Volume Spike
    df['adjusted_enhanced_factor'] = df['enhanced_factor'] / df['volume_spike']
    df['adjusted_enhanced_factor'] = df['adjusted_enhanced_factor'].clip(lower=-10, upper=10)
    
    # Weight by Relative Volume
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['final_factor'] = df['adjusted_enhanced_factor'] * df['relative_volume']
    
    return df['final_factor']
