```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']

    # Calculate High-to-Low Price Range
    df['High_to_Low_Range'] = df['high'] - df['low']

    # Calculate Intraday Momentum
    df['Prev_Close'] = df['close'].shift(1)
    df['Intraday_Momentum'] = df['Intraday_Range'] / df['Prev_Close']

    # Determine Momentum Factor
    df['Prev_High_to_Low_Range'] = df['High_to_Low_Range'].shift(1)
    df['Momentum_Factor'] = df['High_to_Low_Range'] - df['Prev_High_to_Low_Range']

    # Calculate Volume Change
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)

    # Multiply Intraday Momentum by Volume Change
    df['Vol_Adj_Intraday_Momentum'] = df['Intraday_Momentum'] * df['Volume_Change']

    # Add the result to the Intraday Momentum
    df['Intraday_Momentum_Plus_Vol_Change'] = df['Intraday_Momentum'] + df['Vol_Adj_Intraday_Momentum']

    # Weight by Trade Amount
    df['5Day_Avg_Amount'] = df['amount'].rolling(window=5).mean()
    df['Amount_Weight'] = df['amount'] / df['5Day_Avg_Amount']

    # Calculate Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)

    # Multiply by Weighted Momentum
    df['Weighted_Momentum'] = (df['Intraday_Momentum_Plus_Vol_Change'] * df['Amount_Weight']) * df['Volume_Ratio'] * df['Momentum_Factor']

    return df['Weighted_Momentum']
```
