```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume Spikes
    # Compute the volume spike factor by comparing current to previous volume
    df['volume_shifted'] = df['volume'].shift(1)
    volume_spike_factor = df['volume'] / df['volume_shifted']
    
    # Scale intraday momentum by volume spike factor
    adjusted_momentum = intraday_momentum * volume_spike_factor
    
    # Cap at threshold to avoid extreme values
    cap_threshold = 3
    capped_adjusted_momentum = adjusted_momentum.clip(lower=-cap_threshold, upper=cap_threshold)
    
    return capped_adjusted_momentum
```
