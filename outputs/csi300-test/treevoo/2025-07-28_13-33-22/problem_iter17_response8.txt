import pandas as pd
import numpy as np

def heuristics_v2(df, n=10):
    # Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    atr = intraday_volatility.rolling(window=n).mean()
    
    # Overnight Return
    overnight_return = df['open'] - df['close'].shift(1)
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    avg_transaction_intensity = transaction_intensity.rolling(window=n).sum()
    
    # Price Trend Reversal
    price_trend_reversal = (df['close'] - df['close'].shift(1)) * (avg_transaction_intensity / avg_transaction_intensity.rolling(window=n).mean())
    
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Intraday Volatility Standard Deviation
    intraday_volatility_std = intraday_range.rolling(window=n).std()
    
    # Combined Alpha Factor
    combined_factor = (intraday_volatility * overnight_return) + daily_momentum + price_trend_reversal + (intraday_volatility_std * close_to_open_return) + daily_momentum
    
    # Volume Adjustment
    volume_adjustment = df['volume'] / df['volume'].rolling(window=n).mean()
    final_factor = combined_factor * volume_adjustment
    
    return final_factor
