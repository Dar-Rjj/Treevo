```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=5):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Identify Price Spikes
    price_spike = (np.abs(df['close'] - df['close'].shift(1)) > 0.01 * df['close'].shift(1)).astype(int)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=n).std()
    intraday_momentum_adjusted = intraday_momentum * (1 / volatility)
    
    # Scale Intraday Momentum
    combined_spikes = volume_spike + price_spike
    scaled_intraday_momentum = np.clip(intraday_momentum_adjusted, -3, 3) * combined_spikes
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_features = df['open'] - ((df['high'] + df['low']) / 2 - opening_gap)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Factors
    combined_factors = intraday_range * (close_to_open_return + (df['close'] - df['close'].shift(1)))
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    enhanced_combined_factors = combined_factors * avg_abs_price_change
    
    # Adjust Combined Factor by Volume Spike
    adjusted_factor = np.where(volume_spike > 0, enhanced_combined_factors / volume_spike, enhanced_combined_factors)
    adjusted_factor = np.clip(adjusted_factor, -3, 3)
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=n).mean()
    weighted_factor = adjusted_factor * relative_volume
    
    return weighted_factor
```
