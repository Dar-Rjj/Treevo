```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, p=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday = df['open'] - intraday_midpoint - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Average Intraday Range over n periods
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    volume_adjusted_intensity = df['amount'] / df['volume']
    avg_volume_adjusted_intensity = volume_adjusted_intensity.rolling(window=n).mean()
    
    # Price Trend Reversal
    trend_reversal = (df['close'] - df['close'].shift(1)) * (volume_adjusted_intensity / avg_volume_adjusted_intensity)
    
    # Intraday Volatility Standard Deviation
    intraday_volatility = intraday_range.rolling(window=n).std()
    
    # Combined Alpha Factor
    combined_alpha_factor = (avg_intraday_range * (close_to_open_return + price_change) + 
                             daily_momentum + 
                             trend_reversal + 
                             intraday_volatility * close_to_open_return)
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(price_change)
    recent_volatility = abs_price_change.rolling(window=m).mean()
    scaled_combined_feature = combined_alpha_factor * recent_volatility
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(upper=3)
    price_spike_factor = (price_change / price_change.shift(1)).abs().clip(upper=3)
    combined_spike_factor = volume_spike_factor * price_spike_factor
    adjusted_scaled_feature = scaled_combined_feature * combined_spike_factor
    
    # Weight by Volume
    weighted_feature = adjusted_scaled_feature * df['volume']
    
    # Add Recent Amount Trend
    avg_amount = df['amount'].rolling(window=p).mean()
    recent_amount_trend = (df['amount'] / avg_amount)
    weighted_feature_with_amount = weighted_feature * recent_amount_trend
    
    # Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Standard Deviation of Log Returns
    std_log_returns = log_returns.rolling(window=n).std()
    
    # Adjust Intraday Momentum
    adjusted_intraday_momentum = (intraday_range / std_log_returns).fillna(0)
    
    # Final Factor
    final_factor = weighted_feature_with_amount * adjusted_intraday_momentum
    
    return final_factor
```
