```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return and High-to-Low Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_range'] / df['close'].shift(1)
    
    # Determine Momentum Factor
    df['prev_day_high_low_range'] = (df['high'].shift(1) - df['low'].shift(1))
    df['momentum_factor'] = df['intraday_range'] - df['prev_day_high_low_range']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Factors: Multiply Intraday Momentum by Volume Change and add to Intraday Momentum
    df['weighted_momentum'] = df['intraday_momentum'] * df['volume_change'] + df['intraday_momentum']
    
    # Adjust for Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['adj_intraday_momentum'] = df['intraday_momentum'] * df['volume_ratio']
    
    # Introduce Volume Shock Factor
    df['5_day_avg_vol'] = df['volume'].rolling(window=5).mean()
    df['volume_shock'] = df['volume'] - df['5_day_avg_vol']
    
    df['volume_adjusted_momentum'] = df.apply(
        lambda row: row['intraday_momentum'] * (1 + row['volume_shock'] / row['5_day_avg_vol']) 
        if row['volume_shock'] > 0 else row['intraday_momentum'] * (1 - row['volume_shock'] / row['5_day_avg_vol']),
        axis=1
    )
    
    # Weight by Trade Amount
    df['5_day_avg_amount'] = df['amount'].rolling(window=5).mean()
    df['amount_weight'] = df['amount'] / df['5_day_avg_amount']
    
    df['final_factor'] = df['volume_adjusted_momentum'] * df['amount_weight']
    
    return df['final_factor']
```
