```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Identify Price Spikes
    df['price_spike'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Introduce Volatility Adjustment
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_return'].rolling(window=20).std()
    df['intraday_momentum_adjusted'] = df['intraday_momentum'] * (1 / df['volatility'])
    
    # Scale Intraday Momentum
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    df['heuristic_factor'] = df['intraday_momentum_adjusted'] / df['combined_spike']
    
    # Cap at a Threshold to Avoid Extreme Values
    df['heuristic_factor'] = df['heuristic_factor'].apply(lambda x: min(max(x, -3), 3))
    
    return df['heuristic_factor']
```
