```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday_features = df['open'] - intraday_midpoint - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    averaged_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factors = averaged_intraday_range * (close_to_open_return + price_change)
    
    # Enhance with Recent Volatility
    absolute_price_change = (df['close'] - df['close'].shift(1)).abs()
    average_absolute_price_change = absolute_price_change.rolling(window=m).mean()
    enhanced_factors = combined_factors * average_absolute_price_change
    
    # Weight by Volume
    final_factor = enhanced_factors * df['volume']
    
    return final_factor
```
