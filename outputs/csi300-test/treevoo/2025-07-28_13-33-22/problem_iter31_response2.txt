import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1))
    
    # Identify Price Spikes
    price_spike = (df['close'] > df['close'].shift(1))
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Scale Intraday Range
    combined_spikes = (volume_spike & price_spike)
    scaled_intraday_range = intraday_range * combined_spikes
    scaled_intraday_range = np.clip(scaled_intraday_range, 0, 0.5 * intraday_range.mean())
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_feature = (intraday_midpoint - df['open']) - opening_gap
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=5).mean()
    enhanced_feature = combined_feature * (abs_price_change / avg_abs_price_change)
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(upper=2)
    price_spike_factor = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    adjusted_feature = enhanced_feature * (volume_spike_factor + price_spike_factor).clip(upper=2)
    
    # Weight by Volume and Recent Amount Trend
    weighted_feature = adjusted_feature * df['volume']
    weighted_feature = np.clip(weighted_feature, 0, 1.5 * weighted_feature.mean())
    
    recent_amount_trend = df['amount'].rolling(window=10).mean()
    final_feature = weighted_feature * recent_amount_trend
    
    return final_feature
