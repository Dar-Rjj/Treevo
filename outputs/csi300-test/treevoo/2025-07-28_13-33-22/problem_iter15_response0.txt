import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Adjust for Volume and Price Spikes
    df['volume_spike'] = (df['volume'] / df['volume'].shift(1)) - 1
    df['price_spike'] = (df['close'] / df['close'].shift(1)) - 1
    
    # Introduce Volatility Adjustment
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=30).std()
    df['intraday_momentum_vol_adjusted'] = df['intraday_momentum'] / df['volatility']
    
    # Scale Intraday Momentum
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    df['scaled_intraday_momentum'] = df['intraday_momentum_vol_adjusted'] / df['combined_spike']
    
    # Apply Non-Linear Transformation to Smooth Out Extreme Values
    df['softmax_scaled_momentum'] = np.exp(df['scaled_intraday_momentum']) / np.sum(np.exp(df['scaled_intraday_momentum']))
    df['tanh_scaled_momentum'] = np.tanh(df['scaled_intraday_momentum'])
    
    # Return the final factor values
    return df['tanh_scaled_momentum']
