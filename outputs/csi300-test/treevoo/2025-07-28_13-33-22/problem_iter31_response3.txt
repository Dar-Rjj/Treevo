```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, p=15):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    volume_spikes = df['volume'].diff().apply(lambda x: 1 if x > 0 else 0)
    price_spikes = (df['close'].diff() / df['close'].shift(1)).apply(lambda x: 1 if abs(x) > 0.05 else 0)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns = log_returns.rolling(window=n).std()
    intraday_momentum_adjusted = intraday_momentum / std_log_returns
    
    # Scale Intraday Momentum
    spike_combination = volume_spikes + price_spikes
    spike_combination_capped = spike_combination.clip(lower=0, upper=2)
    scaled_intraday_momentum = intraday_momentum_adjusted * spike_combination_capped
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday = df['open'] - ((df['high'] + df['low']) / 2) + opening_gap
    
    # Enhance with Recent Volatility
    absolute_price_change = abs(df['close'].diff())
    avg_abs_price_change = absolute_price_change.rolling(window=m).mean()
    enhanced_intraday = combined_intraday * avg_abs_price_change
    
    # Adjust Combined Factor by Volume Spike
    adjusted_enhanced_intraday = enhanced_intraday / (volume_spikes + 1)
    adjusted_enhanced_intraday_capped = adjusted_enhanced_intraday.clip(lower=-10, upper=10)
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=n).mean()
    weighted_enhanced_intraday = adjusted_enhanced_intraday_capped * relative_volume
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    sum_transaction_intensity = transaction_intensity.rolling(window=n).sum()
    
    # Price Trend Reversal
    price_trend_reversal = (df['close'] - df['close'].shift(1)) * (sum_transaction_intensity / sum_transaction_intensity.rolling(window=n).mean())
    
    # Intraday Volatility Standard Deviation
    intraday_range = df['high'] - df['low']
    intraday_volatility_std = intraday_range.rolling(window=n).std()
    
    # Combined Alpha Factor
    close_to_open_return = (df['close'] - df['open']) / df['open']
    price_change = df['close'] - df['close'].shift(1)
    combined_alpha = (intraday_range.mean() * (close_to_open_return + price_change)) + daily_momentum + price_trend_reversal
    combined_alpha += intraday_volatility_std * close_to_open_return
    
    # Add Recent Amount Trend
    recent_amount_trend = df['amount'].rolling(window=p).mean()
    final_alpha = combined_alpha * recent_amount_trend
    
    return final_alpha
```
