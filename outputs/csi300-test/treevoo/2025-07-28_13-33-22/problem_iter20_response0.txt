```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Price Change
    df['price_change'] = df['close'].diff()
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['combined_intraday_features'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = (df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change'])) + df['combined_intraday_features']
    
    # Recent Volatility
    df['abs_price_change'] = df['price_change'].abs()
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=m).mean()
    df['recent_volatility'] = df['enhanced_intraday_factor'] * df['avg_abs_price_change']
    
    # Adjust for Combined Volume and Price Spikes
    df['volume_spike_factor'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['price_spike_factor'] = (df['price_change'] - df['price_change'].shift(1)) / df['price_change'].shift(1)
    df['combined_spike_factor'] = df['volume_spike_factor'] * df['price_spike_factor']
    df['adjusted_recent_volatility'] = df['recent_volatility'] * df['combined_spike_factor']
    df['adjusted_recent_volatility'] = df['adjusted_recent_volatility'].clip(lower=-3, upper=3)
    
    # Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']
    df['volume_weight'] = df['volume_weight'].apply(lambda x: 1 if x > 1 else 0.5)
    df['adjusted_recent_volatility'] = df['adjusted_recent_volatility'] * df['volume_weight']
    
    # Log Returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Standard Deviation of Log Returns
    df['std_log_returns'] = df['log_returns'].rolling(window=m).std()
    
    # Adjust Intraday Momentum
    df['adjusted_intraday_momentum'] = df['enhanced_intraday_factor'] / df['std_log_returns']
    
    # Final Factor
    df['final_factor'] = df['enhanced_intraday_factor'] * df['adjusted_recent_volatility'] * df['volume_weight'] * df['adjusted_intraday_momentum']
    
    return df['final_factor']
```
