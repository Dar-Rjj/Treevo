def heuristics_v2(data):
    # Compute Intraday Momentum
    intraday_momentum = (data['high'] - data['low']) + (data['close'] - data['open'])
    
    # Volume Spike Factor
    volume_spike_factor = data['volume'] / data['volume'].shift(1)
    
    # Price Spike Factor
    price_change = data['close'] - data['close'].shift(1)
    price_spike_factor = abs(price_change) / abs(price_change.shift(1))
    
    # Final Adjusted Momentum
    adjusted_momentum = intraday_momentum * volume_spike_factor.combine(price_spike_factor, min)
    
    return adjusted_momentum
