```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Intraday Volatility
    intraday_range = df['high'] - df['low']
    std_intraday_range = intraday_range.rolling(window=n).std()

    # Overnight Return
    overnight_return = df['open'] - df['close'].shift(1)

    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)

    # Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    avg_transaction_intensity = transaction_intensity.rolling(window=n).mean()

    # Combined Factor
    combined_factor = (std_intraday_range * overnight_return) + daily_momentum

    # Volume and Transaction Intensity Adjustment
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_adjustment = df['volume'] / avg_volume
    intensity_adjustment = transaction_intensity / avg_transaction_intensity
    
    # Final Alpha Factor
    alpha_factor = combined_factor * (volume_adjustment * intensity_adjustment)
    
    return alpha_factor
```
