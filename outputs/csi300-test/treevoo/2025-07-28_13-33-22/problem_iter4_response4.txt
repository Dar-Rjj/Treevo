def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (data['high'] + data['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = data['open'] - data['close'].shift(1)
    
    # Combine Intraday Features
    combined_feature = (data['open'] - intraday_midpoint) - opening_gap
    
    # Weight by Volume
    factor_values = combined_feature * data['volume']

    return factor_values
