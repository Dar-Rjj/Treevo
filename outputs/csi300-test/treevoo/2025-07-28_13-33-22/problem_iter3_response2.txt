```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    prev_close = df['close'].shift(1)
    intraday_momentum = intraday_range / prev_close
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Calculate Volume Trend
    volume_trend = df['volume'].rolling(window=5).mean()
    
    # Combine Intraday Momentum and Volume Change
    combined_factor = (intraday_momentum * volume_change) + intraday_momentum
    
    # Weight by Trade Amount and Volume Trend
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    amount_weight = df['amount'] / avg_amount_5d
    normalized_volume_trend = volume_trend / volume_trend.rolling(window=5).max()
    final_factor = combined_factor * amount_weight * normalized_volume_trend
    
    return final_factor
```
