```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Intraday Volatility
    intraday_volatility = intraday_range.rolling(window=n).std()
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    recent_volatility_adjusted = enhanced_intraday_factor * avg_abs_price_change
    
    # Volume and Price Spike Adjustment
    volume_spike = (df['volume'] > df['volume'].shift(1)) * 1
    price_spike = (df['close'] > df['close'].shift(1)) * 1
    combined_spike = volume_spike + price_spike
    adjusted_combined_factor = recent_volatility_adjusted / (combined_spike + 1e-6)  # Add small value to avoid division by zero
    adjusted_combined_factor = np.clip(adjusted_combined_factor, -5, 5)  # Cap at a threshold to avoid extreme values
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / (avg_volume + 1e-6)  # Add small value to avoid division by zero
    
    # Transaction Intensity
    transaction_intensity = df['amount'] / (df['volume'] + 1e-6)  # Add small value to avoid division by zero
    avg_transaction_intensity = transaction_intensity.rolling(window=n).mean()
    normalized_transaction_intensity = transaction_intensity / (avg_transaction_intensity + 1e-6)  # Add small value to avoid division by zero
    
    # Volume Adjusted Factor
    volume_adjusted_factor = np.where(df['volume'] > avg_volume, 1, 0.5)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_combined_factor * (volume_adjusted_factor * normalized_transaction_intensity)
    
    return final_alpha_factor
```
