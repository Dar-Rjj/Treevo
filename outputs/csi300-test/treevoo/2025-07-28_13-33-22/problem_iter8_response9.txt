```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'].diff(1)
    
    # Combined Factor
    combined_factor = intraday_momentum + close_to_open_return + price_change
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    weighted_combined_factor = combined_factor * volume_weight
    
    # Adjust by Volume and Price Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1)).fillna(1)
    price_spike = (df['close'] / df['close'].shift(1)).fillna(1)
    combined_spike = volume_spike * price_spike
    adjusted_factor = weighted_combined_factor / combined_spike
    
    # Cap at a Threshold to Avoid Extreme Values
    threshold = 5
    final_alpha_factor = adjusted_factor.clip(lower=-threshold, upper=threshold)
    
    return final_alpha_factor
```
