import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']

    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()

    df['avg_close_5d'] = df['close'].rolling(window=5).mean()
    df['price_spike'] = (df['close'] / df['avg_close_5d']) > 1.5
    
    # Combine Volume and Price Spikes

    df['std_log_return_20d'] = df['log_return'].rolling(window=20).std()
    df['vol_adjusted_momentum'] = df['intraday_range'] / df['std_log_return_20d']
    
    # Scale Intraday Momentum
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    df['scaled_intraday_momentum'] = df['intraday_range'] / df['combined_spike']
    df['scaled_intraday_momentum'] = df['scaled_intraday_momentum'].clip(upper=3) * df['volume_ratio']
    
    # Enhance with Additional Features
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    df['intraday_feature'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    df['close_to_open_return'] = df['close'] - df['open']
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['avg_intraday_range_10d'] = df['intraday_range'].rolling(window=10).mean()
    df['combined_factor'] = df['avg_intraday_range_10d'] * (df['close_to_open_return'] + df['price_change'])
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = (df['close'] - df['close'].shift(1)).abs()
    df['avg_abs_price_change_10d'] = df['abs_price_change'].rolling(window=10).mean()
    df['enhanced_factor'] = df['combined_factor'] * df['avg_abs_price_change_10d'] / df['combined_spike']
    df['enhanced_factor'] = df['enhanced_factor'].clip(upper=3)
    
    # Incorporate Trading Volume Intensity
    df['volume_intensity'] = df['volume'] / df['avg_volume_5d']
    df['adjusted_factor'] = df['enhanced_factor'] * df['volume_intensity']
    df['adjusted_factor'] = df['adjusted_factor'].clip(upper=3)
    
    # Weight by Relative Volume
    df['relative_volume'] = df['volume'] / df['avg_volume_5d']
    df['final_factor'] = df['adjusted_factor'] * df['relative_volume']
    
    return df['final_factor']
