```python
def heuristics_v2(data):
    # Calculate High-to-Low Price Range
    data['price_range'] = data['high'] - data['low']
    
    # Determine Momentum Factor
    data['prev_price_range'] = data['price_range'].shift(1)
    data['momentum_factor'] = data['price_range'] - data['prev_price_range']
    
    # Calculate Volume Ratio
    data['prev_volume'] = data['volume'].shift(1)
    data['volume_ratio'] = data['volume'] / data['prev_volume']
    
    # Calculate Weighted Momentum
    data['weighted_momentum'] = data['momentum_factor'] * data['volume_ratio']

    return data['weighted_momentum']
```
