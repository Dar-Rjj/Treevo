```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Price Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Previous Day's High-to-Low Price Range
    df['Prev_High_Low_Range'] = df['High_Low_Range'].shift(1)
    
    # Determine Momentum Factor
    df['Momentum_Factor'] = df['High_Low_Range'] - df['Prev_High_Low_Range']
    
    # Calculate Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Weight by Volume Ratio
    df['Weighted_Momentum'] = df['Momentum_Factor'] * df['Volume_Ratio']
    
    # Calculate Daily Volume Changes
    df['Volume_Change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate Standard Deviation of Volume Changes
    volume_volatility = df['Volume_Change'].rolling(window=20).std()
    
    # Normalize Weighted Momentum by Volume Volatility
    df['Adjusted_Weighted_Momentum'] = df['Weighted_Momentum'] / volume_volatility
    
    # Apply Momentum Indicator
    momentum = df['Adjusted_Weighted_Momentum'].rolling(window=20).sum()
    
    # Return the Final Alpha Factor
    return momentum
```
