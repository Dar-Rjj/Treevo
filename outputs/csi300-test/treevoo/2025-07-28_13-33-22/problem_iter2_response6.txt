```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Average Intraday Range over n periods
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Combine Factors
    combined_factor = avg_intraday_range * (close_to_open_return + price_change)
    
    # Calculate Intraday Momentum
    prev_close = df['close'].shift(1)
    intraday_momentum = (df['high'] - df['low']) / prev_close
    
    # Calculate Volume Change
    volume_change = df['volume'].diff()
    
    # Combine Intraday Momentum and Volume Change
    combined_intraday_vol = intraday_momentum * volume_change + intraday_momentum
    
    # Weight by Trade Amount and Volume
    avg_amount_5d = df['amount'].rolling(window=5).mean()
    avg_volume_n = df['volume'].rolling(window=n).mean()
    
    weight = (df['amount'] / avg_amount_5d) * (df['volume'] / avg_volume_n)
    final_factor = combined_intraday_vol * weight
    
    return final_factor
```
