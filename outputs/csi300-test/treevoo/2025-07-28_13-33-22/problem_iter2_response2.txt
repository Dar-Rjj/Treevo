```python
import pandas as pd

def heuristics_v2(df, n=10, momentum_window=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Combine Intraday and Price Movements
    combined_factor = (intraday_range.rolling(window=n).mean() * 
                       (close_to_open_return + price_change))
    
    # Weight by Volume
    volume_weight = df['volume'] / df['volume'].rolling(window=n).mean()
    weighted_combined_factor = combined_factor * volume_weight
    
    # Calculate Daily Returns
    daily_returns = df['close'].diff()
    
    # Adjust for Volume Volatility
    volume_changes = df['volume'].diff().fillna(0)
    vol_volatility = volume_changes.rolling(window=n).std()
    adjusted_daily_returns = daily_returns / vol_volatility
    
    # Apply Momentum Indicator
    momentum_sum = adjusted_daily_returns.rolling(window=momentum_window).sum()
    
    # Finalize the Alpha Factor
    alpha_factor = momentum_sum * weighted_combined_factor
    
    return alpha_factor
```
