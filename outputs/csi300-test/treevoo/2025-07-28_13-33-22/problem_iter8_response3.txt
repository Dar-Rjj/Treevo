```python
import pandas as pd

def heuristics_v2(df, n=20, cap_threshold=5):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'].diff()
    
    # Combined Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = avg_intraday_range + (close_to_open_return + price_change)
    
    # Volume and Price Spike Adjustment
    volume_spike = df['volume'].diff().abs()
    price_spike = df['close'].diff().abs()
    combined_spike = volume_spike + price_spike
    adjusted_combined_factor = combined_factor / combined_spike
    adjusted_combined_factor = adjusted_combined_factor.clip(lower=-cap_threshold, upper=cap_threshold)
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    
    # Final Alpha Factor
    alpha_factor = adjusted_combined_factor * volume_weight
    
    return alpha_factor
```
