import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Define the lookback period for the weighted average
    n = 20
    
    # Compute weighted average log return using volume as weights
    df['weighted_log_return'] = (df['log_return'].rolling(window=n).apply(
        lambda x: np.sum(x * df.loc[x.index, 'volume']), raw=False) / 
        df['volume'].rolling(window=n).sum())
    
    # Calculate high-low range over short-term
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate daily volatility using Parkinson's formula
    df['daily_volatility'] = np.sqrt(np.log(df['high'] / df['low'])**2 / (4 * np.log(2)))
    
    # Subtract combined reversal and volatility from weighted average log return
    df['alpha_factor'] = (df['weighted_log_return'] - 
                          (df['high_low_range'] + df['daily_volatility']))
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor'].dropna()
