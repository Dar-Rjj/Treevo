```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the percentage change in close price
    close_pct_change = df['close'].pct_change()
    
    # Calculate the ratio of high to low price
    high_low_ratio = df['high'] / df['low']
    
    # Calculate the amount to volume ratio
    amt_vol_ratio = df['amount'] / df['volume']
    
    # Heuristic factor: (High-Low Ratio) * (Amount-Volume Ratio) * (Close Price % Change)
    # This heuristic aims to capture the momentum, volatility, and liquidity aspects of the stock
    alpha_factor = high_low_ratio * amt_vol_ratio * close_pct_change
    
    return alpha_factor
```
