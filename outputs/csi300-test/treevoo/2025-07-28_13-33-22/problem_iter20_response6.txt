```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Base Intraday Momentum
    base_intraday_momentum = (df['high'] - df['low']) + (df['close'] + df['open'])

    # Volume and Price Spike Adjustments
    volume_spike = df['volume'] / df['volume'].shift(1)
    price_spike = df['close'] / df['close'].shift(1)

    # Cap at threshold to avoid extreme values
    volume_spike = volume_spike.clip(lower=0.5, upper=2.0)
    price_spike = price_spike.clip(lower=0.9, upper=1.1)

    combined_spike = (volume_spike * price_spike).fillna(1)
    adjusted_intraday_momentum = base_intraday_momentum / combined_spike

    # Intraday Range
    intraday_range = df['high'] - df['low']

    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Price Change
    price_change = df['close'] - df['close'].shift(1)

    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2

    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)

    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap

    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=10).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features

    # Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    recent_volatility = abs_price_change.rolling(window=10).mean()

    # Volume Weight
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_weight = df['volume'] / avg_volume

    # Final Alpha Factor
    final_alpha_factor = enhanced_intraday_factor * recent_volatility * volume_weight

    return final_alpha_factor
```
