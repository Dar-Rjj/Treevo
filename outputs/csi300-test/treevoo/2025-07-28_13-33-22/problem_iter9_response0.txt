def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) + (df['close'] + df['open'])
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'].pct_change().fillna(0)
    price_change = df['close'].diff().fillna(0)
    price_spike_factor = price_change.pct_change().fillna(0)
    
    combined_spike_factor = volume_spike_factor * price_spike_factor
    
    # Scale Intraday Momentum
    scaled_momentum = intraday_momentum * (1 + combined_spike_factor)
    
    # Cap at Threshold to Avoid Extreme Values
    threshold = 3
    factor_values = scaled_momentum.clip(-threshold, threshold)
    
    return factor_values
