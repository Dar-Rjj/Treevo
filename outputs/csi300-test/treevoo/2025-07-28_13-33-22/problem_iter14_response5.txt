import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] / df['avg_volume_5d']) > 1.5
    
    df['avg_close_5d'] = df['close'].rolling(window=5).mean()
    df['price_spike'] = (df['close'] / df['avg_close_5d']) > 1.05
    
    # Combine Volume and Price Spikes
    df['combined_spike'] = df['volume_spike'] * df['price_spike']

    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    df['intraday_momentum'] = df['intraday_range'] * (1 / df['volatility'])
    
    # Scale Intraday Momentum
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    df['scaled_intraday_momentum'] = (df['intraday_range'] / df['combined_spike']).apply(lambda x: min(x, 10)) * df['volume_ratio']
    
    # Enhance with Additional Features
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    df['intraday_features'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    
    df['close_to_open_return'] = df['close'] - df['open']
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    df['average_intraday_range'] = df['intraday_range'].rolling(window=10).mean()
    df['combined_feature'] = df['average_intraday_range'] * (df['close_to_open_return'] + df['price_change'])
    
    df['absolute_price_change'] = abs(df['close'] - df['close'].shift(1))
    df['average_absolute_price_change'] = df['absolute_price_change'].rolling(window=10).mean()
    df['enhanced_factor'] = df['combined_feature'] * df['average_absolute_price_change']
    
    df['adjusted_enhanced_factor'] = (df['enhanced_factor'] / df['combined_spike']).apply(lambda x: min(x, 10))
    
    # Weight by Relative Volume
    df['relative_volume'] = df['volume'] / df['avg_volume_5d']
    df['final_factor'] = df['adjusted_enhanced_factor'] * df['relative_volume']
    
    return df['final_factor']
