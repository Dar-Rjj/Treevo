```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Weighted Average Log Return
    n_days = 10  # Number of days to look back for weighted average
    volumes = df['volume'].rolling(window=n_days).apply(lambda x: x[:len(x)-1], raw=False)
    log_returns = df['log_return'].rolling(window=n_days).apply(lambda x: x[:len(x)-1], raw=False)
    weights = volumes / volumes.sum()
    weighted_log_returns = (weights * log_returns).sum(axis=1)
    
    # Adjust for Short-Term Dynamics
    high_low_range = df['high'] - df['low']
    volume_change = df['volume'] - df['volume'].shift(1)
    adjusted_factor = weighted_log_returns - high_low_range + volume_change
    
    return adjusted_factor
```
