def heuristics_v2(df):
    n = 10  # Number of days for weighted average and intraday range

    # Compute Weighted Average Log Return
    df['weighted_log_return'] = (df['log_return'].rolling(window=n).sum() * df['volume'].rolling(window=n).sum()) / (df['volume'].rolling(window=n).sum())

    # Incorporate Short-Term Price Reversal
    df['high_low_range'] = df['high'] - df['low']
    df['reversal_adjusted'] = df['weighted_log_return'] - df['high_low_range']

    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Average Intraday Range over n periods
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] / df['open'] - 1

    # Combine and Enhance Features
    df['combined_feature'] = (df['close_to_open_return'] - df['high_low_range'] + df['weighted_log_return'])

    # Enhance with Recent Volatility
    df['abs_price_change'] = (df['close'] - df['close'].shift(1)).abs()
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=m).mean()
    df['enhanced_feature'] = df['combined_feature'] * df['avg_abs_price_change']

    # Adjust by Volume Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1) * 2).astype(int)
    df['final_factor'] = df['enhanced_feature'] / (df['volume_spike'] + 1)  # Avoid division by zero
    df['final_factor'] = df['final_factor'].clip(lower=-threshold, upper=threshold)  # Cap at threshold

    return df['final_factor']
