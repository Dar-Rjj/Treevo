import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Spikes
    df['volume_spike'] = (df['volume'] / df['volume'].shift(1) > 2).astype(int)
    df['price_spike'] = (np.abs(df['close'] - df['close'].shift(1)) / df['close'].shift(1) > 0.05).astype(int)
    
    # Introduce Volatility Adjustment
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    df['intraday_momentum'] = df['intraday_range'] * (1 / df['volatility'])
    
    # Scale Intraday Momentum
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    df['scaled_intraday_momentum'] = df['intraday_momentum'] / (df['combined_spike'] + 1)  # +1 to avoid division by zero
    df['scaled_intraday_momentum'] = df['scaled_intraday_momentum'].clip(lower=-10, upper=10)  # Cap at a threshold
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Combined Intraday Features
    df['combined_intraday_features'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=20).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = np.abs(df['price_change'])
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=20).mean()
    df['enhanced_volatility'] = df['enhanced_intraday_factor'] * df['avg_abs_price_change']
    
    # Adjust for Combined Volume and Price Spikes
    df['volume_spike_factor'] = (df['volume'] / df['volume'].shift(1)).clip(lower=1, upper=2)
    df['price_spike_factor'] = (np.abs(df['price_change'] / df['price_change'].shift(1))).clip(lower=1, upper=2)
    df['combined_spike_factor'] = df['volume_spike_factor'] * df['price_spike_factor']
    df['adjusted_enhanced_factor'] = df['enhanced_volatility'] / (df['combined_spike_factor'] + 1)  # +1 to avoid division by zero
    df['adjusted_enhanced_factor'] = df['adjusted_enhanced_factor'].clip(lower=-10, upper=10)  # Cap at a threshold
    
    # Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_weight'] = np.where(df['volume'] > df['avg_volume'], 1, 0.5)
    df['final_factor'] = df['adjusted_enhanced_factor'] * df['volume_weight']
    
    return df['final_factor']
