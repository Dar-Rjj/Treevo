import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1) - 1).abs()
    
    # Identify Price Spikes
    price_spike = (df['close'] / df['close'].shift(1) - 1).abs()
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=30).std()
    intraday_momentum = intraday_range / volatility
    
    # Scale Intraday Momentum
    combined_spike = volume_spike + price_spike
    scaled_intraday_momentum = intraday_momentum / (combined_spike + 1e-6)
    scaled_intraday_momentum = np.clip(scaled_intraday_momentum, -3, 3)
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_feature = (df['open'] - intraday_midpoint) - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    average_intraday_range = intraday_range.rolling(window=5).mean()
    combined_factor = average_intraday_range * (close_to_open_return + price_change)
    
    # Enhance with Recent Volatility
    absolute_price_change = (df['close'] - df['close'].shift(1)).abs()
    recent_volatility = absolute_price_change.rolling(window=10).mean()
    enhanced_factor = combined_factor * recent_volatility
    
    # Adjust for Volume Spikes
    volume_spike_factor = 1 / (1 + volume_spike)
    volume_adjusted_factor = enhanced_factor * volume_spike_factor
    volume_adjusted_factor = np.clip(volume_adjusted_factor, -5, 5)
    
    # Weight by Volume
    final_factor = volume_adjusted_factor * df['volume']
    
    return final_factor
