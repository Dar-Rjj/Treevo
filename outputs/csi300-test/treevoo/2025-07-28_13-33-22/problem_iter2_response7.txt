```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Momentum and Volume Change
    df['momentum_volume_factor'] = (df['intraday_momentum'] * df['volume_change']) + df['intraday_momentum']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Weight by Trade Amount
    df['amount_weight'] = df['amount'] / df['amount'].rolling(window=5).mean()
    
    # Weight by Volume
    n_periods = 5  # Define the number of periods for average volume
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=n_periods).mean()
    
    # Combine Weights
    df['combined_weights'] = df['momentum_volume_factor'] * df['amount_weight'] * df['volume_weight']
    
    # Final Combined Factor
    df['final_factor'] = (df['close_to_open_return'] + df['price_change']) * df['combined_weights']
    
    return df['final_factor']
```
