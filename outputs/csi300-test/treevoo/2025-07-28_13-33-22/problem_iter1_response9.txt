```python
def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['intraday_range'] / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Momentum and Volume Change
    df['combined_factor'] = (df['intraday_momentum'] * df['volume_change']) + df['intraday_momentum']
    
    # Weight by Trade Amount
    df['amount_5_day_avg'] = df['amount'].rolling(window=5).mean()
    df['weighted_factor'] = df['combined_factor'] * (df['amount'] / df['amount_5_day_avg'])
    
    return df['weighted_factor']
```
