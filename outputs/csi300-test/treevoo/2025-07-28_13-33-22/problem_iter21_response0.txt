import pandas as pd

def heuristics_v2(df, n=10, m=10, k=10, p=10, q=10):
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['close_to_open'] = df['close'] - df['open'].shift(1)
    
    # Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['combined_intraday'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = (df['avg_intraday_range'] * 
                                      (df['close_to_open'] + df['price_change']) + 
                                      df['combined_intraday'])
    
    # Recent Volatility
    df['abs_price_change'] = df['price_change'].abs()
    df['recent_volatility'] = df['abs_price_change'].rolling(window=m).mean()
    df['volatility_adjusted_factor'] = df['enhanced_intraday_factor'] * df['recent_volatility']
    
    # Volume Adjusted Factor
    df['avg_volume'] = df['volume'].rolling(window=k).mean()
    df['volume_adj_factor'] = df['volume'] / df['avg_volume']
    df['volume_adj_factor'] = df['volume_adj_factor'].apply(lambda x: 1 if x > 1 else 0.5)
    df['volume_adjusted_factor'] = df['volatility_adjusted_factor'] * df['volume_adj_factor']
    
    # Adjust for Combined Volume and Price Spikes
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['price_spike'] = (df['price_change'] - df['price_change'].shift(1)) / df['price_change'].shift(1)
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    
    # Introduce Volatility Adjustment
    df['log_returns'] = (df['close'] / df['close'].shift(1)).apply(lambda x: np.log(x) if x > 0 else 0)
    df['std_log_returns'] = df['log_returns'].rolling(window=10).std()
    df['volatility_adjusted_factor'] = df['volume_adjusted_factor'] / (df['std_log_returns'] + 1e-6)
    
    # Scale Enhanced Intraday Factor
    df['scaled_enhanced_intraday'] = df['volatility_adjusted_factor'] / df['combined_spike']
    df['scaled_enhanced_intraday'] = df['scaled_enhanced_intraday'].clip(lower=-10, upper=10)
    
    # New Momentum Component
    df['momentum'] = df['close'] - df['close'].shift(n)
    df['new_momentum_component'] = df['scaled_enhanced_intraday'] + df['momentum']
    
    # High-Low Volatility
    df['high_low_volatility'] = df['intraday_range'].rolling(window=p).mean()
    
    # Volume Weighted High-Low Volatility
    df['volume_weighted_high_low_volatility'] = df['high_low_volatility'] * df['volume']
    df['avg_volume_weighted_high_low_volatility'] = df['volume_weighted_high_low_volatility'].rolling(window=q).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['scaled_enhanced_intraday'] + 
                                df['new_momentum_component'] + 
                                df['avg_volume_weighted_high_low_volatility'])
    
    return df['final_alpha_factor']
