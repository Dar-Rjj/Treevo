```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, k=30):
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(1)
    
    # Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['combined_intraday_features'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhanced Intraday Factor
    avg_intraday_range = df['intraday_range'].rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Recent Volatility
    abs_price_change = df['close'] - df['close'].shift(1)
    abs_price_change = abs_price_change.abs()
    recent_volatility = abs_price_change.rolling(window=m).mean()
    enhanced_intraday_factor *= recent_volatility
    
    # Adjust for Volume and Price Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1)) * 1
    price_spike = (df['close'] > df['close'].shift(1)) * 1
    combined_spikes = volume_spike + price_spike
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns = log_returns.rolling(window=k).std()
    intraday_momentum = enhanced_intraday_factor / std_log_returns
    
    # Scale Intraday Momentum
    scaled_intraday_momentum = intraday_momentum / combined_spikes
    
    # Apply Non-Linear Transformation to Smooth Out Extreme Values
    softmax = np.exp(scaled_intraday_momentum) / np.sum(np.exp(scaled_intraday_momentum))
    tanh_transformed = np.tanh(scaled_intraday_momentum)
    transformed_momentum = (softmax + tanh_transformed) / 2
    
    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=k).mean()
    volume_adjusted_factor = np.where(df['volume'] > avg_volume, 1, 0.5)
    transformed_momentum *= volume_adjusted_factor
    
    # Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    avg_transaction_intensity = transaction_intensity.rolling(window=n).mean()
    normalized_volume = df['volume'] / avg_volume
    normalized_transaction_intensity = transaction_intensity / avg_transaction_intensity
    volume_intensity_adjustment = normalized_volume * normalized_transaction_intensity
    
    # Final Factor
    final_factor = transformed_momentum * volume_intensity_adjustment
    
    return final_factor
```
