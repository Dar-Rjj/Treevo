import pandas as pd
import numpy as np

def heuristics_v2(df, m=10, n=20):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    volume_spikes = (df['volume'] > df['volume'].shift(1)).astype(int)
    price_spikes = (df['close'] > df['close'].shift(1)).astype(int)

    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=m).std()
    adjusted_momentum = intraday_momentum / volatility

    # Scale Intraday Momentum
    combined_spikes = volume_spikes + price_spikes
    scaled_momentum = adjusted_momentum * combined_spikes
    scaled_momentum = scaled_momentum.clip(lower=-3, upper=3)  # Cap at a threshold

    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)

    # Combine Intraday Features
    combined_intraday = df['open'] - (df['high'] + df['low']) / 2 + opening_gap

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Enhance with Recent Volatility
    abs_price_change = np.abs(log_returns)
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    enhanced_feature = combined_intraday * avg_abs_price_change

    # Adjust Combined Factor by Volume Spike
    adjusted_factor = enhanced_feature / (volume_spikes + 1)  # Add 1 to avoid division by zero
    adjusted_factor = adjusted_factor.clip(lower=-3, upper=3)  # Cap at a threshold

    # Weight by Relative Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    relative_volume = df['volume'] / avg_volume
    weighted_factor = adjusted_factor * relative_volume

    return weighted_factor
