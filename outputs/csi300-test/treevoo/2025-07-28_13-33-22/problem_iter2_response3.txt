```python
import pandas as pd

def heuristics_v2(df, n=10, cap_threshold=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Momentum and Range
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = (close_to_open_return + price_change) * avg_intraday_range
    
    # Adjust by Volume Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1)) > 1.5
    scaled_factor = combined_factor / (volume_spike + 1)
    capped_scaled_factor = scaled_factor.clip(upper=cap_threshold)
    
    # Weight by Relative Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    relative_volume = df['volume'] / avg_volume
    final_factor = capped_scaled_factor * relative_volume
    
    return final_factor
```
