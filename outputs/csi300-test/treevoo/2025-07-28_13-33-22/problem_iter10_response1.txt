```python
def heuristics_v2(df, n=20, threshold=5):
    import pandas as pd
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Combine Momentum and Range
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = avg_intraday_range * (close_to_open_return + price_change)
    
    # Adjust Combined Factor by Volume Spike
    volume_spike = (df['volume'] / df['volume'].shift(1)).fillna(1)
    adjusted_factor = combined_factor / volume_spike
    adjusted_factor = adjusted_factor.clip(lower=-threshold, upper=threshold)
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=n).mean()
    final_factor = adjusted_factor * relative_volume
    
    return final_factor
```
