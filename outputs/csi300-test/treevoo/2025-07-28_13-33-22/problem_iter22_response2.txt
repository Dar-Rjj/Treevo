import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1) * 1.5).astype(int)
    price_spike = (df['close'] > df['close'].shift(1) * 1.05).astype(int)
    combined_spike = volume_spike + price_spike
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns = log_returns.rolling(window=20).std()
    adjusted_intraday_momentum = intraday_range / std_log_returns
    
    # Scale Intraday Momentum
    scaled_momentum = adjusted_intraday_momentum / combined_spike
    scaled_momentum = np.clip(scaled_momentum, -3, 3)
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Combined Intraday Features
    combined_intraday_features = (intraday_midpoint - df['open']) - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=10).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(price_change)
    avg_abs_price_change = abs_price_change.rolling(window=10).mean()
    enhanced_intraday_factor *= avg_abs_price_change
    
    # Weight by Volume
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_weight = np.where(df['volume'] > avg_volume, 1, 0.5)
    enhanced_intraday_factor *= volume_weight
    
    # Adjust for Volume Spikes
    volume_spike_factor = np.where(volume_spike, 0.8, 1)
    final_factor = enhanced_intraday_factor * volume_spike_factor
    final_factor = np.clip(final_factor, -3, 3)
    
    return pd.Series(final_factor, index=df.index, name='factor')
