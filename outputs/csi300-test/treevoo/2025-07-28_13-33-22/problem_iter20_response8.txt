```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Identify Volume and Price Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    price_spike = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    intraday_momentum_adjusted = intraday_momentum * (1 / volatility)
    
    # Scale Intraday Momentum
    combined_spikes = volume_spike + price_spike
    scaled_intraday_momentum = intraday_momentum_adjusted * combined_spikes
    scaled_intraday_momentum = scaled_intraday_momentum.clip(lower=-3, upper=3)
    
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    combined_intraday_features = df['open'] - ((df['high'] + df['low']) / 2 - opening_gap)
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Intraday Features
    combined_feature = intraday_range * (close_to_open_return + df['close'] - df['close'].shift(1))
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=20).mean()
    enhanced_feature = combined_feature * avg_abs_price_change
    
    # Adjust Combined Factor
    adjusted_factor = enhanced_feature / (volume_spike + 1e-6)  # Adding small constant to avoid division by zero
    adjusted_factor = adjusted_factor.clip(lower=-3, upper=3)
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    weighted_factor = adjusted_factor * relative_volume
    
    # Final Volatility Adjustment
    final_volatility = log_returns.rolling(window=20).std()
    final_factor = weighted_factor * (1 / final_volatility)
    
    # Scale Final Factor
    combined_spike = volume_spike + price_spike
    final_scaled_factor = final_factor / (combined_spike + 1e-6)  # Adding small constant to avoid division by zero
    final_scaled_factor = final_scaled_factor.clip(lower=-3, upper=3)
    
    return final_scaled_factor
```
