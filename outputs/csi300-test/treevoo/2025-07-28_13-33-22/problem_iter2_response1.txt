```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    daily_returns = df['close'] - df['close'].shift(1)
    
    # Calculate Simple Moving Average of Daily Returns over 10 days
    sma_daily_returns = daily_returns.rolling(window=10).mean()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Factors
    combined_factor = (intraday_range.rolling(window=10).sum()) * (close_to_open_return + daily_returns)
    
    # Adjust for Volume Volatility
    volume_changes = df['volume'] - df['volume'].shift(1)
    volume_volatility = volume_changes.rolling(window=20).std()
    adjusted_combined_factor = combined_factor / volume_volatility
    
    # Apply Momentum Indicator
    momentum = adjusted_combined_factor.rolling(window=20).sum()
    
    return momentum
```
