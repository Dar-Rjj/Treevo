```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']

    # Adjust by Volume and Price Spikes
    # Identify Volume Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Identify Price Spikes
    price_spike = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Combine Volume and Price Spikes
    combined_spikes = volume_spike + price_spike

    # Scale Intraday Momentum
    intraday_momentum = intraday_range / (combined_spikes + 1)  # Avoid division by zero
    intraday_momentum = intraday_momentum.clip(lower=-3, upper=3)  # Cap at a threshold to avoid extreme values
    
    # Adjust for Volume Impact
    volume_ratio = df['volume'] / df['volume'].shift(1).fillna(1)  # Avoid division by zero
    intraday_momentum = intraday_momentum * volume_ratio
    
    # Weight by Trade Amount
    amount_ratio = df['amount'] / df['amount'].rolling(window=5).mean().fillna(1)  # Avoid division by zero
    intraday_momentum = intraday_momentum * amount_ratio

    return intraday_momentum
```
