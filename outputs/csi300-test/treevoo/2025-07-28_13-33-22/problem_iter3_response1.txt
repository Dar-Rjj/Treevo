```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    prev_close = df['close'].shift(1)
    intraday_momentum = intraday_range / prev_close
    
    # Calculate Volume Change
    prev_volume = df['volume'].shift(1)
    volume_change = df['volume'] - prev_volume
    
    # Analyze Volume Trend
    volume_5d_avg = df['volume'].rolling(window=5).mean()
    
    # Combine Intraday Momentum, Volume Change, and Volume Trend
    combined_factor = (intraday_momentum * volume_change) * volume_5d_avg + intraday_momentum
    
    # Weight by Trade Amount
    amount_5d_avg = df['amount'].rolling(window=5).mean()
    amount_weight = df['amount'] / amount_5d_avg
    final_factor = combined_factor * amount_weight
    
    return final_factor
```
