```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Intraday Volatility
    intraday_volatility = (df['high'] - df['low']).rolling(window=n).std()
    
    # Overnight Return
    overnight_return = df['open'] - df['close'].shift(1)
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    transaction_intensity = (df['amount'] / df['volume']).rolling(window=n).mean()
    
    # Combined Alpha Factor
    combined_factor = (intraday_volatility * overnight_return) + daily_momentum
    combined_factor *= (transaction_intensity / transaction_intensity.rolling(window=n).mean())
    
    # Volume Adjustment
    volume_adjustment = df['volume'] / df['volume'].rolling(window=n).mean()
    combined_factor *= volume_adjustment
    
    return combined_factor
```
