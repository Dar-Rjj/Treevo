```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = avg_intraday_range + (close_to_open_return + price_change)
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    
    # Final Alpha Factor
    alpha_factor = combined_factor * volume_weight
    
    return alpha_factor
```
