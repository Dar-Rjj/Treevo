```python
import pandas as pd

def heuristics_v2(df, n=10, cap_threshold=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Momentum and Range
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    momentum_and_range = (close_to_open_return + price_change) * avg_intraday_range
    
    # Identify Volume Spikes
    volume_spike = df['volume'] / df['volume'].shift(1)
    
    # Adjust Combined Factor by Volume Spike
    adjusted_factor = momentum_and_range / volume_spike
    adjusted_factor = adjusted_factor.clip(lower=-cap_threshold, upper=cap_threshold)
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=n).mean()
    weighted_factor = adjusted_factor * relative_volume
    
    return weighted_factor
```
