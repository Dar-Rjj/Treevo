```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, threshold=3):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factors = avg_intraday_range * (close_to_open_return + price_change)
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    enhanced_factors = combined_factors * avg_abs_price_change
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'] / df['volume'].shift(1)
    price_spike_factor = abs_price_change / abs_price_change.shift(1)
    combined_spike_factor = volume_spike_factor * price_spike_factor
    adjusted_factors = enhanced_factors * combined_spike_factor.clip(upper=threshold)
    
    # Weight by Volume
    final_factor = adjusted_factors * df['volume']
    
    return final_factor
```
