```python
import pandas as pd

def heuristics_v2(df, n=5, m=10, k=20, volume_spike_threshold=2.0):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'].diff()
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    recent_volatility = abs_price_change.rolling(window=m).mean()
    volatility_adjusted_factor = enhanced_intraday_factor * recent_volatility
    
    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=k).mean()
    volume_adjusted_factor = (df['volume'] > avg_volume).astype(float) * 1.0 + ((df['volume'] <= avg_volume).astype(float) * 0.5)
    volume_adjusted_factor = volatility_adjusted_factor * volume_adjusted_factor
    
    # Weighted Volatility
    weighted_volatility = recent_volatility * volume_adjusted_factor
    
    # Adjust for Volume Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(upper=volume_spike_threshold)
    final_factor = weighted_volatility * volume_spike_factor
    
    return final_factor
```
