```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday_features = df['open'] - intraday_midpoint - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Factor
    average_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = average_intraday_range + (close_to_open_return + price_change)
    
    # Volume Weight
    volume_weight = df['volume'] / df['volume'].rolling(window=n).mean()
    
    # Final Factor
    final_factor = combined_factor * volume_weight
    
    return final_factor
```
