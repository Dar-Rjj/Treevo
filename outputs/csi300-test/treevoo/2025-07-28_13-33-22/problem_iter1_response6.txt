```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Moving Average of Intraday Returns
    weighted_intraday_return = intraday_return * df['volume']
    sum_weighted_intraday_return = weighted_intraday_return.rolling(window=5).sum()
    sum_volume = df['volume'].rolling(window=5).sum()
    volume_weighted_moving_avg = sum_weighted_intraday_return / sum_volume
    
    # Calculate Daily Price Change
    daily_price_change = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Score
    momentum_score = daily_price_change + 0.5 * volume_weighted_moving_avg
    
    # Calculate Final Alpha Factor with Hysteresis
    alpha_factor = pd.Series(index=momentum_score.index)
    for i in range(len(momentum_score)):
        if momentum_score.iloc[i] > 0.05:
            alpha_factor.iloc[i] = 1
        elif momentum_score.iloc[i] < -0.05:
            alpha_factor.iloc[i] = -1
        else:
            alpha_factor.iloc[i] = 0
    
    return alpha_factor
```
