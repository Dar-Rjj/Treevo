```python
import pandas as pd

def heuristics_v2(df, n=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Intraday and Price Movements
    combined_movements = intraday_range + close_to_open_return + price_change
    
    # Weight by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    weighted_combined = combined_movements * volume_weight
    
    # Calculate Daily Returns
    daily_returns = df['close'] - df['close'].shift(1)
    
    # Adjust for Volume Volatility
    volume_changes = df['volume'] - df['volume'].shift(1)
    std_vol_changes = volume_changes.rolling(window=n).std()
    adjusted_daily_returns = daily_returns / std_vol_changes
    
    # Apply Momentum Indicator
    momentum_sum = adjusted_daily_returns.rolling(window=n).sum()
    
    # Finalize the Alpha Factor
    alpha_factor = momentum_sum * weighted_combined
    
    return alpha_factor
```
