```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'].diff()
    price_change.iloc[0] = 0  # First value is NaN, set to 0
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    opening_gap.iloc[0] = 0  # First value is NaN, set to 0
    
    # Combined Intraday Features
    combined_intraday = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=5).mean()  # n periods
    enhanced_intraday_factor = (avg_intraday_range * (close_to_open_return + price_change)) + combined_intraday
    
    # Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    abs_price_change.iloc[0] = 0  # First value is NaN, set to 0
    recent_volatility = abs_price_change.rolling(window=10).mean()  # m periods
    
    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=20).mean()  # k periods
    volume_adjusted_factor = df['volume'] / avg_volume
    volume_adjusted_factor = volume_adjusted_factor.apply(lambda x: 1 if x > 1 else 0.5)
    
    # Final Alpha Factor
    final_alpha_factor = enhanced_intraday_factor * recent_volatility * volume_adjusted_factor
    
    return final_alpha_factor
```
