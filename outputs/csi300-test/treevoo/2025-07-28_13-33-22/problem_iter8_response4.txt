```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, volume_spike_threshold=3.0):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    combined_factor = (intraday_range.rolling(window=n).mean() * (close_to_open_return + price_change))
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    recent_volatility = abs_price_change.rolling(window=m).mean()
    enhanced_feature = combined_factor * recent_volatility
    
    # Adjust for Volume Spikes
    volume_spike_factor = df['volume'] / df['volume'].shift(1)
    volume_spike_factor = volume_spike_factor.clip(upper=volume_spike_threshold)
    adjusted_feature = enhanced_feature / volume_spike_factor
    
    # Weight by Volume
    final_factor = adjusted_feature * df['volume']
    
    return final_factor
```
