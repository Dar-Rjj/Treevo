```python
def heuristics_v2(df, n=20):
    # Intraday Volatility
    intraday_range = df['high'] - df['low']
    intraday_volatility = intraday_range.rolling(window=n).std()
    
    # Overnight Return
    overnight_return = df['open'].shift(-1) - df['close']
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Combined Factor
    combined_factor = (intraday_volatility * overnight_return) + daily_momentum
    
    # Volume Adjustment
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_adj = df['volume'] / avg_volume
    
    # Final adjusted factor
    final_factor = combined_factor * volume_adj
    
    return final_factor
```
