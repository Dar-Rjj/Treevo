```python
import pandas as pd

def heuristics_v2(df, n=10, m=5, p=7):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'].shift(-1) - df['close']
    
    # Combine Intraday Features
    combined_intraday = df['open'] - intraday_midpoint - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Average Intraday Range over n periods
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    volume_adjusted_intensity = df['amount'] / df['volume']
    sum_volume_intensity = volume_adjusted_intensity.rolling(window=n).sum()
    
    # Price Trend Reversal
    avg_volume_intensity = volume_adjusted_intensity.rolling(window=n).mean()
    price_trend_reversal = price_change * (volume_adjusted_intensity / avg_volume_intensity)
    
    # Intraday Volatility Standard Deviation
    std_intraday_range = intraday_range.rolling(window=n).std()
    
    # Combined Alpha Factor
    combined_alpha_factor = (avg_intraday_range * (close_to_open_return + price_change)) + daily_momentum + price_trend_reversal
    combined_alpha_factor += (std_intraday_range * close_to_open_return)
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    volatility_factor = abs_price_change.rolling(window=m).mean()
    combined_alpha_factor *= volatility_factor
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1))
    price_spike_factor = (df['close'] - df['close'].shift(1)).abs() / (df['close'].shift(1) - df['close'].shift(2)).abs()
    combined_spike_factor = volume_spike_factor * price_spike_factor
    combined_alpha_factor *= combined_spike_factor
    combined_alpha_factor = combined_alpha_factor.clip(lower=-3, upper=3)  # Cap at Threshold to Avoid Extreme Values
    
    # Weight by Volume
    combined_alpha_factor *= df['volume']
    
    # Add Recent Amount Trend
    avg_amount = df['amount'].rolling(window=p).mean()
    combined_alpha_factor *= avg_amount
    
    return combined_alpha_factor
```
