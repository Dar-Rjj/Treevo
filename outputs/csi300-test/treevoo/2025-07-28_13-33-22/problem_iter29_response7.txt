```python
import pandas as pd

def heuristics_v2(df, n=20, m=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open']
    
    # Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    df['combined_intraday_features'] = df['open'] - df['intraday_midpoint'] + df['opening_gap']
    
    # Enhanced Intraday Factor
    avg_intraday_range = df['intraday_range'].rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Recent Volatility
    abs_price_change = df['close'] - df['close'].shift(1)
    abs_price_change = abs_price_change.abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    recent_volatility = enhanced_intraday_factor * avg_abs_price_change
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'] / df['volume'].shift(1)
    price_spike_factor = abs(df['close'] - df['close'].shift(1)) / abs(df['close'].shift(1) - df['close'].shift(2))
    combined_spike_factor = volume_spike_factor * price_spike_factor
    adjusted_recent_volatility = recent_volatility * combined_spike_factor
    adjusted_recent_volatility = adjusted_recent_volatility.clip(lower=-10, upper=10)
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    volume_weight = volume_weight.apply(lambda x: 1 if x > 1 else 0.5)
    adjusted_recent_volatility_weighted = adjusted_recent_volatility * volume_weight
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Price Trend Reversal
    trend_reversal = (df['close'] - df['close'].shift(1)) * (df['volume'] / avg_volume)
    avg_trend_reversal = trend_reversal.rolling(window=n).mean()
    normalized_trend_reversal = trend_reversal / avg_trend_reversal
    
    # Intraday Volatility Standard Deviation
    intraday_volatility_std = df['intraday_range'].rolling(window=n).std()
    
    # Final Alpha Factor
    final_alpha_factor = (
        avg_intraday_range * (df['close_to_open_return'] + df['price_change']) +
        daily_momentum +
        normalized_trend_reversal +
        intraday_volatility_std * df['close_to_open_return']
    ) * volume_weight
    
    return final_alpha_factor
```
