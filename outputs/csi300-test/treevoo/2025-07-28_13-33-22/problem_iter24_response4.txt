```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Momentum Indicator
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['momentum'] = np.where(df['close'] > df['SMA_20'], 1, 
                              np.where(df['close'] < df['SMA_20'], -1, 0))
    
    # Volatility Indicator
    df['daily_range'] = df['high'] - df['low']
    df['EMA_5_daily_range'] = df['daily_range'].ewm(span=5, adjust=False).mean()
    df['volatility_trend'] = np.where(df['EMA_5_daily_range'] > df['EMA_5_daily_range'].shift(1), 1, 
                                      np.where(df['EMA_5_daily_range'] < df['EMA_5_daily_range'].shift(1), -1, 0))
    
    # Liquidity Indicator
    df['avg_volume_30'] = df['volume'].rolling(window=30).mean()
    df['liquidity_ratio'] = df['volume'] / df['avg_volume_30']
    
    # Market Interest Indicator
    df['total_volume_10'] = df['volume'].rolling(window=10).sum()
    df['market_interest_ratio'] = df['volume'] / df['total_volume_10']
    
    # Average Transaction Size
    df['avg_transaction_size'] = df['amount'] / df['volume']
    df['large_transactions'] = np.where(df['avg_transaction_size'] > df['avg_transaction_size'].quantile(0.75), 1, 
                                        np.where(df['avg_transaction_size'] < df['avg_transaction_size'].quantile(0.25), -1, 0))
    
    # Combine all indicators into a single factor
    df['alpha_factor'] = (df['momentum'] + 
                          df['volatility_trend'] * 0.5 + 
                          df['liquidity_ratio'] * 0.3 + 
                          df['market_interest_ratio'] * 0.2 + 
                          df['large_transactions'] * 0.1)
    
    return df['alpha_factor']
```
