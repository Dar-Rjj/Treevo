```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)).abs()
    price_spike = (df['close'] - df['close'].shift(1)).abs()
    
    # Introduce Dynamic Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    rolling_volatility = log_returns.rolling(window=30).std()
    
    # Scale Intraday Momentum
    combined_spikes = volume_spike + price_spike
    scaled_intraday_momentum = intraday_range / (combined_spikes + 1e-6)  # Avoid division by zero
    scaled_intraday_momentum = np.clip(scaled_intraday_momentum, a_min=-5, a_max=5)
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday_feature = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    avg_abs_price_change = abs_price_change.rolling(window=10).mean()
    combined_feature = combined_intraday_feature * rolling_volatility / avg_abs_price_change
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).fillna(1)
    price_spike_factor = (price_spike / price_spike.shift(1)).fillna(1)
    combined_spike_factor = volume_spike_factor * price_spike_factor
    adjusted_feature = combined_feature / combined_spike_factor
    
    # Weight by Volume and Recent Amount Trend
    recent_amount_trend = df['amount'].rolling(window=10).mean()
    weighted_feature = (adjusted_feature * df['volume'] / volume_spike_factor).clip(lower=-10, upper=10)
    weighted_feature = weighted_feature * recent_amount_trend
    
    # Incorporate Accumulated Volume
    accumulated_volume = df['volume'].rolling(window=30).sum()
    normalized_feature = weighted_feature / (accumulated_volume + 1e-6)  # Avoid division by zero
    
    # Integrate Long-Term Momentum
    long_term_momentum = df['close'].rolling(window=60).mean()
    final_feature = normalized_feature + np.sign(normalized_feature) * long_term_momentum
    
    return final_feature
```
