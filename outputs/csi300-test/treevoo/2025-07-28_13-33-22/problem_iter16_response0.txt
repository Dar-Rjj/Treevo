```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1) * 1.5).astype(int)
    
    # Identify Price Spikes
    price_spike = (np.abs(df['close'] - df['close'].shift(1)) > 0.03 * df['close'].shift(1)).astype(int)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    adjusted_intraday_momentum = intraday_range / (volatility + 1e-8)  # Avoid division by zero
    
    # Scale Intraday Momentum
    combined_spike = volume_spike + price_spike
    scaled_intraday_momentum = adjusted_intraday_momentum / (combined_spike + 1e-8)  # Avoid division by zero
    capped_intraday_momentum = np.clip(scaled_intraday_momentum, -5, 5)
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Factors
    avg_intraday_range = capped_intraday_momentum.rolling(window=5).mean()
    combined_factor = avg_intraday_range * (close_to_open_return + opening_gap)
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=10).mean()
    enhanced_factor = combined_factor * avg_abs_price_change
    
    # Weight by Volume
    weighted_factor = enhanced_factor * df['volume']
    
    return weighted_factor
```
