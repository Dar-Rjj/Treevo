import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Volume and Price Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1)).astype(float)
    price_spike = (np.abs(df['close'] - df['close'].shift(1)) > 0.05 * df['close'].shift(1)).astype(float)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_dev = log_returns.rolling(window=20).std()
    volatility_adjusted_momentum = intraday_range / std_dev
    
    # Scale Intraday Momentum
    combined_spike = volume_spike + price_spike
    scaled_momentum = volatility_adjusted_momentum / combined_spike
    capped_momentum = np.clip(scaled_momentum, -3, 3)
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=10).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=20).mean()
    enhanced_feature = enhanced_intraday_factor * avg_abs_price_change
    
    # Weight by Volume
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_factor = np.where(df['volume'] > avg_volume, 1, 0.5)
    weighted_enhanced_feature = enhanced_feature * volume_factor
    
    # Adjust for Volume Spikes
    volume_spike_factor = np.where(volume_spike > 0, 0.5, 1)
    final_feature = weighted_enhanced_feature * volume_spike_factor
    final_capped_feature = np.clip(final_feature, -3, 3)
    
    return pd.Series(final_capped_feature, index=df.index)
