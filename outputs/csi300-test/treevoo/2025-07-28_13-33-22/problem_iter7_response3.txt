```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['prev_close'] = df['close'].shift(1)
    df['intraday_momentum'] = df['intraday_range'] / df['prev_close']
    
    # Calculate Volume Change
    df['prev_volume'] = df['volume'].shift(1)
    df['volume_change'] = df['volume'] - df['prev_volume']
    
    # Calculate Volume Trend
    df['volume_trend'] = df['volume'].rolling(window=5).mean()
    
    # Combine Intraday Momentum and Volume Change
    df['combined_factor'] = (df['intraday_momentum'] * df['volume_change']) + df['intraday_momentum']
    
    # Weight by Trade Amount and Volume Trend
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    df['amount_weight'] = df['amount'] / df['avg_amount_5d']
    df['final_factor'] = df['combined_factor'] * (df['amount_weight'] * df['volume_trend'])
    
    return df['final_factor'].dropna()
```
