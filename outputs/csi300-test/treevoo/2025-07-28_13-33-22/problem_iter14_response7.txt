import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].shift(-1) - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Adjust by Volume and Price Spikes
    volume_spike = df['volume'] / df['volume'].rolling(window=5).mean()
    price_spike = df['close'] / df['close'].rolling(window=5).mean()
    combined_spike = volume_spike * price_spike
    
    # Scale Intraday Momentum
    intraday_momentum = (intraday_range / combined_spike).clip(upper=3.0)
    
    # Adjust for Volume Impact
    volume_ratio = df['volume'] / df['volume'].rolling(window=5).mean()
    amount_ratio = df['amount'] / df['amount'].rolling(window=10).mean()
    adjusted_intraday_momentum = intraday_momentum * volume_ratio * amount_ratio
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2.0
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Momentum and Range
    combined_feature = intraday_range * (close_to_open_return + price_change)
    
    # Combine Intraday Momentum and Reversal
    combined_momentum_reversal = (adjusted_intraday_momentum - (df['open'] - intraday_midpoint)) - opening_gap
    
    # Adjust Combined Feature by Volume Spike
    if (volume_spike > 1.5).any():
        combined_feature_adjusted = (combined_momentum_reversal / volume_spike).clip(upper=3.0)
    else:
        combined_feature_adjusted = combined_momentum_reversal * df['volume']
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=5).mean()
    final_factor = combined_feature_adjusted * relative_volume
    
    return final_factor
