import pandas as pd

def heuristics_v2(df, n=10, m=5, k=3, threshold=5.0):
    # Calculate daily log returns
    df['log_return'] = df['close'].apply(lambda x: np.log(x) - np.log(df['close'].shift(1)))
    
    # Compute weighted average log return using volume as weights
    volumes = df['volume'].rolling(window=n).sum()
    weighted_log_returns = (df['log_return'] * df['volume']).rolling(window=n).sum() / volumes
    
    # Incorporate short-term price reversal
    high_low_range = df['high'] - df['low']
    weighted_log_returns_with_reversal = weighted_log_returns - high_low_range
    
    # Calculate intraday range and its average over n periods
    intraday_range = df['high'] - df['low']
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Calculate close-to-open return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Combine and enhance features
    combined_feature = close_to_open_return - high_low_range + weighted_log_returns_with_reversal + avg_intraday_range
    
    # Enhance with recent volatility
    absolute_price_change = (df['close'] - df['close'].shift(1)).abs()
    avg_absolute_price_change = absolute_price_change.rolling(window=m).mean()
    enhanced_feature = combined_feature * avg_absolute_price_change
    
    # Adjust by volume changes
    avg_volume = df['volume'].rolling(window=k).mean().shift(1)
    volume_change = (df['volume'] - avg_volume) / avg_volume
    adjusted_factor = enhanced_feature / volume_change
    
    # Cap at a threshold to avoid extreme values
    final_factor = adjusted_factor.clip(lower=-threshold, upper=threshold)
    
    return final_factor
