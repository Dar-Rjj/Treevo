```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Factors
    df['factor'] = (df['intraday_momentum'] * df['volume_change']) + df['intraday_momentum']
    
    # Weight by Trade Amount
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    df['amount_weight'] = df['amount'] / df['avg_amount_5d']
    
    df['factor'] *= df['amount_weight']
    
    # Adjust for Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    df['factor'] *= df['volume_ratio']
    
    return df['factor']
```
