```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (np.abs(df['close'] - df['close'].shift(1)) > 0.05 * df['close'].shift(1)).astype(int)
    df['combined_spikes'] = df['volume_spike'] + df['price_spike']
    
    # Introduce Dynamic Volatility Adjustment
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=30).std()
    
    # Scale Intraday Momentum
    df['scaled_intraday_momentum'] = df['intraday_range'] / (df['combined_spikes'] + 1)  # Avoid division by zero
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['intraday_feature'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = np.abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=5).mean()
    
    # Adjust for Dynamic Volume Spikes
    df['volume_spike_factor'] = df['volume'] / df['volume'].shift(1)
    df['volume_spike_factor'] = np.clip(df['volume_spike_factor'], 0.5, 2.0)  # Apply dynamic threshold to avoid extreme values
    df['adjusted_feature'] = df['intraday_feature'] * df['volume_spike_factor']
    
    # Weight by Volume
    df['weighted_feature'] = df['adjusted_feature'] * df['volume']
    
    # Incorporate Accumulated Volume
    df['accumulated_volume'] = df['volume'].rolling(window=10).sum()
    df['normalized_feature'] = df['weighted_feature'] / (df['accumulated_volume'] + 1)  # Avoid division by zero
    
    # Apply Adaptive Moving Average Smoothing
    n = 10  # Variable number of days for smoothing
    df['smoothed_feature'] = df['normalized_feature'].ewm(span=n, adjust=False).mean()
    
    return df['smoothed_feature']
```
