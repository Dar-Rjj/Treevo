```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate High-to-Low Price Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    df['prev_close'] = df['close'].shift(1)
    df['intraday_momentum'] = df['intraday_range'] / df['prev_close']
    
    # Calculate Previous Day's High-to-Low Price Range
    df['prev_high_low_range'] = (df['high'].shift(1) - df['low'].shift(1))
    
    # Determine Momentum Factor
    df['momentum_factor'] = df['high_low_range'] - df['prev_high_low_range']
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Multiply Intraday Momentum by Volume Change
    df['intraday_momentum_adjusted'] = df['intraday_momentum'] * df['volume_change']
    
    # Add the result to the Intraday Momentum
    df['intraday_momentum_combined'] = df['intraday_momentum'] + df['intraday_momentum_adjusted']
    
    # Weight by Trade Amount
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    df['amount_weight'] = df['amount'] / df['avg_amount_5d']
    
    # Adjust for Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['intraday_momentum_volume_adjusted'] = df['intraday_momentum'] * df['volume_ratio']
    df['weighted_momentum'] = df['momentum_factor'] * df['volume_ratio']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['intraday_momentum_combined'] * df['amount_weight'] * df['weighted_momentum']
    
    return df['alpha_factor']
```
