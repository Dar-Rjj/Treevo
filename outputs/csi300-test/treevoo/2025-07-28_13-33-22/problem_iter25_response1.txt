```python
import pandas as pd

def heuristics_v2(df, n=10, m=20, p=15):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_feature = (df['open'] - (df['high'] + df['low']) / 2) - opening_gap
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = avg_intraday_range * (opening_gap + price_change)
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    volatility_factor = abs_price_change.rolling(window=m).mean()
    enhanced_factor = combined_factor * volatility_factor
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike = df['volume'] / df['volume'].shift(1)
    price_spike = (price_change / price_change.shift(1)).abs()
    combined_spike = volume_spike * price_spike
    adjusted_factor = enhanced_factor * combined_spike
    
    # Weight by Volume
    weighted_factor = adjusted_factor * df['volume']
    
    # Add Recent Amount Trend
    recent_amount_trend = df['amount'].rolling(window=p).mean()
    final_factor = weighted_factor * recent_amount_trend
    
    return final_factor
```
