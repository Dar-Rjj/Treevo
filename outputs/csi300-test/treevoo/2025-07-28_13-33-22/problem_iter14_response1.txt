```python
import pandas as pd

def heuristics_v2(df, n=10, m=10):
    # Calculate Intraday Return
    df['IntradayRange'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['CloseToOpenReturn'] = df['close'] - df['open'].shift(1)
    
    # Price Change
    df['PriceChange'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['IntradayMidpoint'] = (df['high'] + df['low']) / 2
    
    # Opening Gap
    df['OpeningGap'] = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['CombinedIntradayFeatures'] = (df['open'] - df['IntradayMidpoint']) - df['OpeningGap']
    
    # Enhanced Intraday Factor
    df['AvgIntradayRange'] = df['IntradayRange'].rolling(window=n).mean()
    df['EnhancedIntradayFactor'] = df['AvgIntradayRange'] * (df['CloseToOpenReturn'] + df['PriceChange']) + df['CombinedIntradayFeatures']
    
    # Recent Volatility
    df['AbsolutePriceChange'] = (df['close'] - df['close'].shift(1)).abs()
    df['AvgAbsolutePriceChange'] = df['AbsolutePriceChange'].rolling(window=m).mean()
    df['RecentVolatility'] = df['EnhancedIntradayFactor'] * df['AvgAbsolutePriceChange']
    
    # Volume Weight
    df['AvgVolume'] = df['volume'].rolling(window=n).mean()
    df['VolumeWeight'] = df['volume'] / df['AvgVolume']
    df['VolumeAdjustedFactor'] = df['EnhancedIntradayFactor'] * df['VolumeWeight']
    
    # Adjust for Combined Volume and Price Spikes
    df['VolumeSpikeFactor'] = (df['volume'] - df['volume'].shift(1)) / df['AvgVolume']
    df['PriceSpikeFactor'] = (df['PriceChange'] - df['PriceChange'].shift(1)) / df['AvgAbsolutePriceChange']
    df['SpikeAdjustment'] = (df['VolumeSpikeFactor'] + df['PriceSpikeFactor']).clip(lower=-1, upper=1)
    df['AdjustedIntradayMomentum'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Final Factor
    df['FinalFactor'] = df['VolumeAdjustedFactor'] * df['RecentVolatility'] * df['SpikeAdjustment'] + df['AdjustedIntradayMomentum']
    
    return df['FinalFactor']
```
