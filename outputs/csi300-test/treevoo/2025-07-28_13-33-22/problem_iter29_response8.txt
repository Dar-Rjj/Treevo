```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Intraday Features
    intraday_midpoint = (df['high'] + df['low']) / 2
    opening_gap = df['open'] - df['close'].shift(1)
    combined_intraday_features = (intraday_midpoint - df['open']) - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = np.abs(price_change)
    ema_abs_price_change = abs_price_change.ewm(span=m, adjust=False).mean()
    recent_volatility = enhanced_intraday_factor * ema_abs_price_change
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'] / df['volume'].shift(1)
    price_spike_factor = np.abs(price_change) / np.abs(price_change).shift(1)
    combined_spike_factor = volume_spike_factor * price_spike_factor
    adjusted_recent_volatility = np.clip(recent_volatility * combined_spike_factor, -1e6, 1e6)
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = np.where(df['volume'] > avg_volume, 1, 0.5)
    final_integrated_factor = enhanced_intraday_factor * adjusted_recent_volatility * volume_weight
    
    # Adjusted Intraday Momentum
    adjusted_intraday_momentum = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Overnight Return
    overnight_return = df['open'] - df['close'].shift(1)
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    transaction_intensity = (df['amount'] / df['volume']).rolling(window=n).sum()
    
    # Price Trend Reversal
    avg_transaction_intensity = (df['amount'] / df['volume']).rolling(window=n).mean()
    price_trend_reversal = (df['close'] - df['close'].shift(1)) * (transaction_intensity / avg_transaction_intensity)
    
    # Intraday Volatility Standard Deviation
    intraday_volatility_std = intraday_range.rolling(window=n).std()
    
    # Combined Alpha Factor
    combined_alpha_factor = (enhanced_intraday_factor * overnight_return) + daily_momentum + price_trend_reversal + (intraday_volatility_std * close_to_open_return)
    
    # Final Factor
    final_factor = final_integrated_factor + (combined_alpha_factor + transaction_intensity) * (df['close'] - df['open'].shift(1))
    
    return final_factor
```
