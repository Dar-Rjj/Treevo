```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=20, k=30, p=15, q=20):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'].diff()
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    combined_intraday_features = df['open'] - intraday_midpoint - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = price_change.abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    enhanced_intraday_factor *= avg_abs_price_change
    
    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=k).mean()
    volume_adjusted_factor = np.where(df['volume'] > avg_volume, 1, 0.5)
    enhanced_intraday_factor *= volume_adjusted_factor
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1)).fillna(1)
    price_spike = (price_change / price_change.shift(1)).abs().fillna(1)
    combined_spike = volume_spike + price_spike
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns = log_returns.rolling(window=20).std().fillna(1)
    enhanced_intraday_factor /= std_log_returns
    enhanced_intraday_factor /= combined_spike
    enhanced_intraday_factor = np.clip(enhanced_intraday_factor, -10, 10)
    
    # New Momentum Component
    momentum = df['close'] - df['close'].shift(n)
    enhanced_intraday_factor += momentum
    
    # High-Low Volatility
    high_low_volatility = df['high'] - df['low']
    avg_high_low_volatility = high_low_volatility.rolling(window=p).mean()
    
    # Volume Weighted High-Low Volatility
    volume_weighted_high_low_volatility = high_low_volatility * df['volume']
    avg_volume_weighted_high_low_volatility = volume_weighted_high_low_volatility.rolling(window=q).mean()
    
    # Adjust Combined Factor by Volume Spike
    enhanced_intraday_factor /= volume_spike
    enhanced_intraday_factor = np.clip(enhanced_intraday_factor, -10, 10)
    
    # Final Alpha Factor
    final_alpha_factor = enhanced_intraday_factor + momentum + avg_volume_weighted_high_low_volatility
    
    return final_alpha_factor
```
