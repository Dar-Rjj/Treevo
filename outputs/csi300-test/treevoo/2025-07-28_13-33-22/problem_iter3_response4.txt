```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Identify Price Spikes
    price_spike = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Volume and Price Spikes
    combined_spike = volume_spike + price_spike
    
    # Scale Intraday Momentum
    scaled_momentum = intraday_momentum / combined_spike
    
    # Cap at a threshold to avoid extreme values
    threshold = 10
    scaled_momentum = scaled_momentum.clip(-threshold, threshold)
    
    return scaled_momentum
```
