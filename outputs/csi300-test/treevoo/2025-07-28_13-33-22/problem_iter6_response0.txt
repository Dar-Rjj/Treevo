```python
import pandas as pd

def heuristics_v2(df, n=10, m=20, cap_threshold=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Combine Factors
    combined_factor = (intraday_range.rolling(window=n).mean()) * (close_to_open_return + price_change)
    
    # Adjust by Volume Spikes
    volume_spike = (df['volume'] / df['volume'].rolling(window=m).mean())
    adjusted_factor = combined_factor / volume_spike
    
    # Cap at a Threshold to Avoid Extreme Values
    capped_adjusted_factor = adjusted_factor.clip(lower=-cap_threshold, upper=cap_threshold)
    
    # Weight by Volume
    volume_weight = df['volume'] / df['volume'].ewm(span=n, adjust=False).mean()
    final_factor = capped_adjusted_factor * volume_weight
    
    return final_factor
```
