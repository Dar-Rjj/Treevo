```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume and Price Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    price_spike = (df['close'] - df['close'].shift(1)).apply(lambda x: 1 if abs(x) > 0.05 * df['close'].shift(1) else 0)
    combined_spikes = volume_spike + price_spike
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=30).std()
    volatility_adjusted_momentum = intraday_momentum / (volatility + 1e-8)
    
    # Scale Intraday Momentum
    scaled_momentum = volatility_adjusted_momentum / (combined_spikes + 1e-8)
    
    # Cap at a Threshold to Avoid Extreme Values
    capped_factor = scaled_momentum.clip(lower=-3, upper=3)
    
    return capped_factor
```
