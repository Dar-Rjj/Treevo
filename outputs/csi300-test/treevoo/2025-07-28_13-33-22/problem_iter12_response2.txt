```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust for Volume and Price Changes
    # Compare Current Volume to Moving Average of Volumes
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    volume_adjustment = df['volume'] / df['volume_ma']
    volume_adjustment_capped = volume_adjustment.clip(lower=0.5, upper=2.0)  # Cap at Threshold
    
    # Compare Current Close to Previous Close
    price_change = df['close'].pct_change()
    price_change_adjustment = 1 + (price_change.abs() * 10)  # Scale by absolute price change
    price_change_adjustment_capped = price_change_adjustment.clip(lower=0.5, upper=2.0)  # Cap at Threshold
    
    # Scale Intraday Momentum
    scaled_intraday_momentum = (intraday_momentum * volume_adjustment_capped * price_change_adjustment_capped)
    
    return scaled_intraday_momentum
```
