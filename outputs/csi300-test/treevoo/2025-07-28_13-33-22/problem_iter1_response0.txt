```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Volume-Weighted Price Change
    df['price_change'] = df['close'].diff()
    df['volume_weighted_price_change'] = df['price_change'] * df['volume']
    
    # Measure Price Momentum
    momentum = df['volume_weighted_price_change'].rolling(window=20).sum()
    momentum_std = df['volume_weighted_price_change'].rolling(window=20).std()
    normalized_momentum = momentum / momentum_std
    
    # Incorporate Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], 
                      abs(x['high'] - df['close'].shift(1)), 
                      abs(df['close'].shift(1) - x['low'])), 
        axis=1
    )
    atr = df['true_range'].rolling(window=14).mean()
    
    # Combine with Momentum
    composite_factor = normalized_momentum + atr
    composite_factor = composite_factor.abs()
    
    return composite_factor
```
