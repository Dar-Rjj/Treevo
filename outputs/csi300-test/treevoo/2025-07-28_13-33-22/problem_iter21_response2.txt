```python
import pandas as pd

def heuristics_v2(df):
    # Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Overnight Return
    overnight_return = df['open'].shift(-1) - df['close']
    
    # Intraday Return
    intraday_return = df['close'] - df['open']
    
    # Price Volatility (High minus Low over a window, e.g., 5 days)
    price_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combined Momentum
    combined_momentum = intraday_momentum + overnight_return + intraday_return
    
    # Volume Impact
    volume_ema = df['volume'].ewm(span=5, adjust=False).mean()
    volume_impact = df['volume'] / volume_ema
    combined_momentum_weighted = combined_momentum * volume_impact
    
    # Adjust for Price Volatility
    volatility_threshold = 0.1  # Example threshold
    capped_volatility = price_volatility.where(price_volatility < volatility_threshold, volatility_threshold)
    adjusted_momentum = combined_momentum_weighted / capped_volatility
    
    # Final Alpha Factor
    alpha_factor = adjusted_momentum
    
    return alpha_factor
```
