def heuristics_v2(df, n=5, m=5, k=5, p=10, q=10, r=10, cap_threshold=3.0):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'].diff()
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    combined_intraday_features = df['open'] - intraday_midpoint - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = price_change.abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    enhanced_intraday_factor *= avg_abs_price_change
    
    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=k).mean()
    volume_adjusted_factor = df['volume'].div(avg_volume)
    volume_adjusted_factor = volume_adjusted_factor.apply(lambda x: 1 if x > 1 else 0.5)
    enhanced_intraday_factor *= volume_adjusted_factor
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'].div(df['volume'].shift(1))
    price_spike_factor = price_change.div(price_change.shift(1).abs())

    avg_amount = df['amount'].rolling(window=r).mean()
    recent_amount_trend = df['amount'].div(avg_amount)
    enhanced_intraday_factor *= recent_amount_trend
