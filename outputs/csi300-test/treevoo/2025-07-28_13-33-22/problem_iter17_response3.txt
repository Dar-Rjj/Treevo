```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10, k=10):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (np.abs(df['close'] - df['close'].shift(1)) > 0.05 * df['close'].shift(1)).astype(int)
    
    # Introduce Volatility Adjustment
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['std_log_return'] = df['log_return'].rolling(window=m).std()
    df['intraday_momentum_vol_adj'] = df['intraday_momentum'] / df['std_log_return']
    
    # Scale Intraday Momentum
    df['spike_factor'] = df['volume_spike'] + df['price_spike']
    df['spike_factor'] = df['spike_factor'].clip(upper=2)
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['combined_intraday_features'] = df['open'] - (df['intraday_midpoint'] - df['opening_gap'])
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Recent Volatility
    df['abs_price_change'] = np.abs(df['close'] - df['close'].shift(1))
    df['recent_volatility'] = df['abs_price_change'].rolling(window=m).mean()
    df['recent_volatility_factor'] = df['enhanced_intraday_factor'] * df['recent_volatility']
    
    # Volume Adjusted Factor
    df['avg_volume'] = df['volume'].rolling(window=k).mean()
    df['volume_adjusted_factor'] = (df['volume'] / df['avg_volume']).apply(lambda x: 1 if x > 1 else 0.5)
    df['volume_adjusted_factor'] = df['recent_volatility_factor'] * df['volume_adjusted_factor']
    
    # Weighted Volatility
    df['weighted_volatility'] = df['recent_volatility'] * df['volume_adjusted_factor']
    
    # Adjust for Volume Spikes
    df['final_factor'] = df['volume_adjusted_factor'] * df['spike_factor']
    df['final_factor'] = df['final_factor'].clip(upper=3)
    
    return df['final_factor']
```
