```python
def heuristics_v2(df):
    # Calculate the price difference between high and low
    price_diff = df['high'] - df['low']
    
    # Calculate the percentage change in close price
    pct_change_close = df['close'].pct_change()
    
    # Calculate the ratio of amount to volume
    amount_volume_ratio = df['amount'] / df['volume']
    
    # Heuristic: Combine the three components
    factor_values = (price_diff * pct_change_close) / (amount_volume_ratio + 1e-6)
    
    return factor_values
```
