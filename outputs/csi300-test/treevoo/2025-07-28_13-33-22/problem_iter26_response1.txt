```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (df['close'] > df['close'].shift(1)).astype(int)
    df['combined_spikes'] = df['volume_spike'] + df['price_spike']
    
    # Introduce Volatility Adjustment
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=30).std()
    
    # Scale Intraday Momentum
    df['scaled_intraday_momentum'] = df['intraday_range'] / (df['combined_spikes'] + 1e-6)
    df['scaled_intraday_momentum'] = df['scaled_intraday_momentum'].clip(lower=-10, upper=10)
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['combined_intraday_feature'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = np.abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=5).mean()
    df['enhanced_feature'] = df['combined_intraday_feature'] * df['volatility'] / (df['avg_abs_price_change'] + 1e-6)
    
    # Adjust for Combined Volume and Price Spikes
    df['volume_spike_factor'] = df['volume'] / (df['volume'].shift(1) + 1e-6)
    df['price_spike_factor'] = np.abs(df['close'] - df['close'].shift(1)) / (np.abs(df['close'].shift(1) - df['close'].shift(2)) + 1e-6)
    df['combined_spike_factor'] = (df['volume_spike_factor'] + df['price_spike_factor']) / 2
    df['adjusted_feature'] = df['enhanced_feature'] / (df['combined_spike_factor'] + 1e-6)
    df['adjusted_feature'] = df['adjusted_feature'].clip(lower=-10, upper=10)
    
    # Weight by Volume and Recent Amount Trend
    df['recent_amount_trend'] = df['amount'].rolling(window=10).mean()
    df['weighted_feature'] = (df['adjusted_feature'] * df['volume'] / (df['volume_spike'] + 1e-6)) * df['recent_amount_trend']
    df['weighted_feature'] = df['weighted_feature'].clip(lower=-10, upper=10)
    
    return df['weighted_feature']
```
