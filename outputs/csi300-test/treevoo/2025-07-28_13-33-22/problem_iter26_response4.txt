```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10, k=10):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'].shift(1) - df['open']
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = (avg_intraday_range * (close_to_open_return + price_change)) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = np.abs(price_change)
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    recent_volatility = enhanced_intraday_factor * avg_abs_price_change
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = np.where(df['volume'] > df['volume'].shift(1), 1, 0.5)
    price_spike_factor = np.where(abs_price_change > abs_price_change.shift(1), 1, 0.5)
    combined_spike_factor = volume_spike_factor * price_spike_factor
    scaled_recent_volatility = np.minimum(recent_volatility * combined_spike_factor, 10)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns = log_returns.rolling(window=k).std()
    adjusted_intraday_momentum = enhanced_intraday_factor / std_log_returns
    
    # Apply Non-Linear Transformation
    softmax_values = np.exp(adjusted_intraday_momentum) / np.sum(np.exp(adjusted_intraday_momentum))
    tanh_values = np.tanh(adjusted_intraday_momentum)
    transformed_momentum = softmax_values * tanh_values
    
    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=k).mean()
    volume_weight = np.where(df['volume'] > avg_volume, 1, 0.5)
    volume_adjusted_factor = transformed_momentum * volume_weight
    
    # Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    avg_transaction_intensity = transaction_intensity.rolling(window=k).mean()
    normalized_transaction_intensity = transaction_intensity / avg_transaction_intensity
    intensity_adjusted_factor = volume_adjusted_factor * normalized_transaction_intensity
    
    # Final Factor
    final_factor = intensity_adjusted_factor * scaled_recent_volatility
    
    # Incorporate Accumulated Volume
    accumulated_volume = df['volume'].rolling(window=n).sum()
    normalized_final_factor = final_factor / accumulated_volume
    
    # Apply Moving Average Smoothing
    smoothed_factor = normalized_final_factor.rolling(window=n).mean()
    
    return smoothed_factor
```
