```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Spikes
    volume_spikes = (df['volume'] / df['volume'].shift(1) > 2).astype(int)
    price_spikes = (df['close'] / df['close'].shift(1) > 1.05).astype(int)
    combined_spikes = volume_spikes + price_spikes
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Scale Intraday Momentum
    scaled_intraday_momentum = intraday_range / (combined_spikes + 1)  # +1 to avoid division by zero
    capped_scaled_intraday_momentum = np.clip(scaled_intraday_momentum, a_min=None, a_max=np.percentile(scaled_intraday_momentum, 95))
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=20).mean()
    
    # Adjust for Volume Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(upper=3)
    adjusted_feature = (capped_scaled_intraday_momentum * intraday_features * close_to_open_return * avg_abs_price_change * volume_spike_factor)
    adjusted_feature = np.clip(adjusted_feature, a_min=None, a_max=np.percentile(adjusted_feature, 95))
    
    # Weight by Volume
    final_factor = adjusted_feature * df['volume']
    
    return final_factor
```
