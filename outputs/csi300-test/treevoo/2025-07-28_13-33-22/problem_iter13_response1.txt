```python
import pandas as pd

def heuristics_v2(df, n=20, m=10, volume_spike_threshold=3.0):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Average Intraday Range over n periods
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Combine Factors
    combined_factors = avg_intraday_range * (close_to_open_return + price_change)
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    enhanced_factor = combined_factors * avg_abs_price_change
    
    # Adjust for Volume Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(upper=volume_spike_threshold)
    adjusted_factor = enhanced_factor * volume_spike_factor
    
    # Weight by Volume
    weighted_factor = adjusted_factor * df['volume']
    
    return weighted_factor
```
