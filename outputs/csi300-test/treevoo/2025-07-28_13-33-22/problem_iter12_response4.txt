```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    # Identify Volume Spikes
    avg_vol_5d = df['volume'].rolling(window=5).mean().shift(1)
    volume_spike = (df['volume'] - avg_vol_5d) / avg_vol_5d
    
    # Identify Price Spikes
    avg_close_5d = df['close'].rolling(window=5).mean().shift(1)
    price_spike = (df['close'] - avg_close_5d) / avg_close_5d
    
    # Combine Volume and Price Spikes
    combined_spike = volume_spike * price_spike
    
    # Scale Intraday Momentum
    intraday_momentum = intraday_range / combined_spike
    threshold = 3.0  # Cap at a Threshold to Avoid Extreme Values
    intraday_momentum = intraday_momentum.clip(lower=-threshold, upper=threshold)
    
    # Adjust for Volume Impact
    vol_ratio = df['volume'] / avg_vol_5d
    intraday_momentum = intraday_momentum * vol_ratio
    
    # Weight by Trade Amount
    avg_amount_10d = df['amount'].rolling(window=10).mean().shift(1)
    amount_weight = df['amount'] / avg_amount_10d
    factor_values = intraday_momentum * amount_weight
    
    return factor_values
```
