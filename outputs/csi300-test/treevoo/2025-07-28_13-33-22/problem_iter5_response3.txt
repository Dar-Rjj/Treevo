```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    volume_spike = (df['volume'] - df['volume'].shift(1)).clip(lower=0)
    price_spike = (df['close'] - df['close'].shift(1)).abs().clip(lower=0)
    
    combined_spike = volume_spike + price_spike
    
    # Scale Intraday Momentum
    scaled_momentum = intraday_momentum / (combined_spike + 1e-9)  # avoid division by zero
    capped_momentum = scaled_momentum.clip(lower=-5, upper=5)  # cap at a threshold to avoid extreme values
    
    return capped_momentum
```
