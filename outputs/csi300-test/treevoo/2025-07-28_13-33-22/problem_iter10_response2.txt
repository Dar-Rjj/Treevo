```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff().fillna(0)
    
    # Combine Momentum and Range
    momentum_and_range = (close_to_open_return + price_change) * intraday_range
    
    # Adjust by Volume Spike
    volume_spike = df['volume'] / df['volume'].shift(1)
    threshold = 5  # Cap at a threshold to avoid extreme values
    adjusted_volume_spike = volume_spike.clip(upper=threshold)
    
    # Weight by Relative Volume
    n_periods = 20
    average_volume = df['volume'].rolling(window=n_periods).mean()
    relative_volume = df['volume'] / average_volume
    final_factor = momentum_and_range * relative_volume * adjusted_volume_spike
    
    return final_factor
```
