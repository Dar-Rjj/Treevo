import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Spikes
    df['volume_spike'] = df['volume'].diff().abs()
    df['price_spike'] = df['close'].diff().abs()
    
    # Introduce Volatility Adjustment
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['std_log_returns'] = df['log_returns'].rolling(window=30).std()
    df['adjusted_intraday_momentum'] = (df['close'] - df['open']) / df['std_log_returns']
    
    # Scale Intraday Momentum
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    df['scaled_momentum'] = df['adjusted_intraday_momentum'] / df['combined_spike']
    df['scaled_momentum'] = df['scaled_momentum'].clip(lower=-5, upper=5)
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Combined Intraday Features
    df['combined_intraday_features'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=10).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = df['price_change'].abs()
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=10).mean()
    df['enhanced_intraday_factor'] *= df['avg_abs_price_change']
    
    # Adjust for Combined Volume and Price Spikes
    df['vol_spike_factor'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['price_spike_factor'] = df['price_change'] / df['price_change'].rolling(window=10).mean()
    df['combined_spike_factor'] = df['vol_spike_factor'] + df['price_spike_factor']
    df['enhanced_intraday_factor'] /= df['combined_spike_factor']
    
    # Weight by Volume
    df['avg_volume'] = df['volume'].rolling(window=10).mean()
    df['volume_weight'] = np.where(df['volume'] > df['avg_volume'], 1, 0.5)
    df['enhanced_intraday_factor'] *= df['volume_weight']
    
    # Incorporate Trading Direction and Recent Amount Trend
    df['net_change_amount'] = df['amount'] - df['amount'].shift(1)
    df['trading_direction'] = np.where(df['net_change_amount'] > 0, 1, -1)
    df['recent_amount_trend'] = df['amount'].rolling(window=10).sum()
    df['final_factor'] = df['enhanced_intraday_factor'] * df['trading_direction'] * df['recent_amount_trend']
    
    return df['final_factor']
