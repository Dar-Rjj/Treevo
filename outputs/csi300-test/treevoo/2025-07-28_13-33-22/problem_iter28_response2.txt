import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, p=5):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1)).astype(int)
    price_spike = (df['close'] > df['close'].shift(1)).astype(int)
    combined_spikes = volume_spike + price_spike
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close']) - np.log(df['close'].shift(1))
    rolling_std = log_returns.rolling(window=n).std()
    adjusted_intraday_momentum = intraday_momentum / rolling_std
    
    # Scale Intraday Momentum
    scaled_intraday_momentum = adjusted_intraday_momentum * combined_spikes
    scaled_intraday_momentum = scaled_intraday_momentum.clip(lower=-3, upper=3)  # Cap at a threshold
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_feature = (df['open'] - ((df['high'] + df['low']) / 2) - opening_gap)
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    enhanced_intraday_feature = intraday_feature * avg_abs_price_change
    
    # Adjust Combined Factor by Volume Spike
    adjusted_factor = enhanced_intraday_feature / (combined_spikes + 1e-6)  # Avoid division by zero
    adjusted_factor = adjusted_factor.clip(lower=-3, upper=3)  # Cap at a threshold
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=n).mean()
    weighted_factor = adjusted_factor * relative_volume
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    
    # Volume-Adjusted Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    sum_transaction_intensity = transaction_intensity.rolling(window=n).sum()
    
    # Price Trend Reversal
    price_trend_reversal = (df['close'] - df['close'].shift(1)) * (transaction_intensity / transaction_intensity.rolling(window=n).mean())
    
    # Intraday Volatility Standard Deviation
    intraday_range = df['high'] - df['low']
    intraday_volatility_std = intraday_range.rolling(window=n).std()
    
    # Combined Alpha Factor
    close_to_open_return = (df['close'] - df['open']) / df['open']
    alpha_factor = (intraday_range.rolling(window=n).mean() * (close_to_open_return + (df['close'] - df['close'].shift(1)))) + daily_momentum + price_trend_reversal
    alpha_factor = alpha_factor * (intraday_volatility_std * close_to_open_return)
    
    # Add Recent Amount Trend
    recent_amount_trend = df['amount'].rolling(window=p).mean()
    final_alpha_factor = alpha_factor * recent_amount_trend
    
    return final_alpha_factor
