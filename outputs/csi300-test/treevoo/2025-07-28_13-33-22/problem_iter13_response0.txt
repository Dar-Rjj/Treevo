```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) + (df['close'] + df['open'])

    # Compare Current Volume to Previous Volume and Calculate Volume Spike Factor
    volume_spike_factor = df['volume'].pct_change().fillna(0)

    # Compare Current Price Change to Previous Price Change and Calculate Price Spike Factor
    price_spike_factor = df['close'].pct_change().fillna(0)
    
    # Combine and Scale Intraday Momentum
    combined_factor = (intraday_momentum * (1 + volume_spike_factor) * (1 + price_spike_factor))
    
    # Cap at Threshold to Avoid Extreme Values
    threshold = 3
    adjusted_factor = combined_factor.clip(lower=-threshold, upper=threshold)

    return adjusted_factor
```
