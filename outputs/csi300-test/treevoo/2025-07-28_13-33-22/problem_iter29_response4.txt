import pandas as pd

def heuristics_v2(df, n=20, m=5, k=10, p=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (df['close'] > df['close'].shift(1)).astype(int)
    
    # Introduce Volatility Adjustment
    df['log_returns'] = (df['close'] / df['close'].shift(1)).apply(lambda x: 0 if pd.isna(x) else x).apply(np.log)
    df['volatility'] = df['log_returns'].rolling(window=n).std()
    
    # Scale Intraday Momentum
    df['spike_factor'] = df['volume_spike'] + df['price_spike']
    df['intraday_momentum'] = df['intraday_range'] / (df['spike_factor'] + 1e-6)
    df['intraday_momentum'] = df['intraday_momentum'].clip(lower=0, upper=3 * df['intraday_momentum'].mean())
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['combined_intraday'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = (df['close'] - df['close'].shift(1)).abs()
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=m).mean()
    
    # Adjust for Volume Spikes and Weight by Volume
    df['volume_spike_factor'] = (df['volume'] / df['volume'].shift(1)).clip(lower=1, upper=3)
    df['adjusted_combined_feature'] = (df['combined_intraday'] * df['volume_spike_factor']).clip(lower=-3, upper=3)
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=n).mean()
    df['weighted_adjusted_feature'] = df['adjusted_combined_feature'] * df['relative_volume']
    
    # Incorporate Combined Volume and Price Spikes
    df['volume_spike_factor_k'] = (df['volume'] / df['volume'].rolling(window=k).mean()).clip(lower=1, upper=3)
    df['price_spike_factor_k'] = (df['close'] - df['close'].shift(1)) / (df['close'] - df['close'].shift(1)).rolling(window=k).mean()
    df['combined_spike_factor'] = df['volume_spike_factor_k'] + df['price_spike_factor_k']
    df['scaled_combined_feature'] = df['weighted_adjusted_feature'] * df['combined_spike_factor'].clip(lower=1, upper=3)
    
    # Add Recent Amount Trend
    df['recent_amount_trend'] = df['amount'].rolling(window=p).sum()
    df['enhanced_feature'] = df['scaled_combined_feature'] * df['recent_amount_trend']
    
    # Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change'])
    
    # Adjust Intraday Momentum
    df['adjusted_intraday_momentum'] = df['intraday_momentum'] / (df['volatility'] + 1e-6)
    
    # Final Factor
    df['final_factor'] = df['enhanced_intraday_factor'] * df['avg_abs_price_change'] * df['weighted_adjusted_feature'] * df['adjusted_intraday_momentum']
    
    # Apply Moving Average Smoothing
    df['smoothed_final_factor'] = df['final_factor'].rolling(window=n).mean()
    
    return df['smoothed_final_factor']
