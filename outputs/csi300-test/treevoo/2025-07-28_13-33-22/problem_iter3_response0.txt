```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Identify Volume Spikes
    df['volume_spike'] = df['volume'].diff()
    
    # Identify Price Spikes
    df['price_spike'] = df['close'].diff().abs()
    
    # Adjust by Volume and Price Spikes
    df['adjusted_spike'] = (df['volume_spike'] + df['price_spike']).clip(upper=10)
    
    # Scale Intraday Momentum
    df['scaled_momentum'] = df['intraday_momentum'] / df['adjusted_spike']
    
    # Apply Smoothing Function
    df['smoothed_scaled_momentum'] = df['scaled_momentum'].rolling(window=5).mean()
    
    return df['smoothed_scaled_momentum']
```
