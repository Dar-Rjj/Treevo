```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 20-Day and 50-Day Exponential Moving Averages (EMA)
    df['20_day_ema'] = df['close'].ewm(span=20, adjust=False).mean()
    df['50_day_ema'] = df['close'].ewm(span=50, adjust=False).mean()

    # Compute Trend Signal
    df['trend_signal'] = df['20_day_ema'] - df['50_day_ema']

    # Calculate 20-Day Volume Moving Average
    df['20_day_vol_sma'] = df['volume'].rolling(window=20).mean()

    # Volume Confirmation
    def volume_weight(row):
        if row['volume'] > row['20_day_vol_sma']:
            return 1.0
        else:
            return -1.0

    df['volume_weight'] = df.apply(volume_weight, axis=1)

    # Generate Final Alpha Factor
    df['alpha_factor'] = df['trend_signal'] * df['volume_weight']

    return df['alpha_factor']
```
