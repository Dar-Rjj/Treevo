```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Weighted Average Log Return using Volume as Weights
    n_days = 20  # Define the lookback period
    volumes = df['volume'].rolling(window=n_days).sum()
    weighted_log_returns = (df['log_return'].rolling(window=n_days) * df['volume']).sum() / volumes
    
    # Incorporate Short-Term Price Reversal, Volatility, and Volume Change
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Adjust Weighted Average Log Return with High-Low Range, Daily Volatility, and Volume Change
    adjusted_weighted_log_return = weighted_log_returns + df['high_low_range'] + df['daily_volatility'] - df['volume_change']
    
    return adjusted_weighted_log_return.dropna()
```
