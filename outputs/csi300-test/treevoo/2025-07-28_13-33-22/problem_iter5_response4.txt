```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_feature = (df['open'] - intraday_midpoint) - opening_gap
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1) * 2).astype(int)
    
    # Adjust Combined Feature by Volume Spike
    factor_values = combined_feature.copy()
    spike_detected = volume_spike == 1
    factor_values[spike_detected] = combined_feature[spike_detected] / df['volume'][spike_detected]
    factor_values[~spike_detected] = combined_feature[~spike_detected] * df['volume'][~spike_detected]
    
    # Cap at a Threshold to Avoid Extreme Values
    threshold = 3
    factor_values = factor_values.clip(lower=-threshold, upper=threshold)
    
    return factor_values
```
