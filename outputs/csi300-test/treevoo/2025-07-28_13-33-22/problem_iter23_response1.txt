def heuristics_v2(data):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
        data (pd.DataFrame): A DataFrame with index as date, and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
        
    Returns:
        pd.Series: A Series representing the factor values, indexed by date.
    """
    # Calculate the average of high and low prices
    avg_price = (data['high'] + data['low']) / 2
    
    # Calculate the weighted average price using the volume
    weighted_avg_price = (data['close'] * data['volume']).cumsum() / data['volume'].cumsum()
    
    # Calculate the difference between the average price and the weighted average price
    price_diff = avg_price - weighted_avg_price
    
    # Calculate the amount-volume ratio
    amount_volume_ratio = data['amount'] / data['volume']
    
    # Calculate the moving average of the amount-volume ratio
    moving_avg_amount_volume_ratio = amount_volume_ratio.rolling(window=10).mean()
    
    # Calculate the final factor value
    factor = price_diff * moving_avg_amount_volume_ratio
    
    return factor
