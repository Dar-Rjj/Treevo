import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=10, k=10):
    # Intraday Range
    intraday_range = df['high'] - df['low']

    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Price Change
    price_change = df['close'].diff()

    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2

    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)

    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap

    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features

    # Recent Volatility
    abs_price_change = price_change.abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    volatility_adjusted_factor = enhanced_intraday_factor * avg_abs_price_change

    # Volume Adjusted Factor
    avg_volume = df['volume'].rolling(window=k).mean()
    volume_ratio = df['volume'] / avg_volume
    volume_adjusted_factor = np.where(volume_ratio > 1, 1, 0.5)
    adjusted_enhanced_intraday_factor = volatility_adjusted_factor * volume_adjusted_factor

    # Adjust for Combined Volume and Price Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1)).fillna(1)
    price_spike = (price_change / price_change.shift(1)).fillna(1)
    combined_spike = volume_spike + price_spike

    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    std_log_returns = log_returns.rolling(window=k).std()
    volatility_adjusted_factor = adjusted_enhanced_intraday_factor / (1 + std_log_returns)

    # Scale Enhanced Intraday Factor
    scaled_enhanced_intraday_factor = volatility_adjusted_factor / combined_spike
    scaled_enhanced_intraday_factor = np.clip(scaled_enhanced_intraday_factor, -3, 3)

    # Non-Linear Transformation to Smooth Out Extreme Values
    softmax_factor = np.exp(scaled_enhanced_intraday_factor) / np.sum(np.exp(scaled_enhanced_intraday_factor))
    tanh_factor = np.tanh(scaled_enhanced_intraday_factor)

    return pd.Series(tanh_factor, index=df.index, name='heuristic_factor_v2')
