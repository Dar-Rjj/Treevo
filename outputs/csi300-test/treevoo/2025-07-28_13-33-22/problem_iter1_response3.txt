```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1)).fillna(1)
    volume_spike[volume_spike < 1] = 1  # Ensure that we only consider spikes, not drops
    
    # Adjust Intraday Momentum by Volume Spikes
    adjusted_momentum = (intraday_momentum / volume_spike).clip(lower=0, upper=5)  # Cap at a threshold to avoid extreme values
    
    return adjusted_momentum
```
