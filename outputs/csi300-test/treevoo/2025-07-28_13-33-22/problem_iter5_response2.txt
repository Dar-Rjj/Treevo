```python
import pandas as pd

def heuristics_v2(df, n=10, spike_threshold=5, cap_threshold=10):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = avg_intraday_range + (close_to_open_return + price_change)
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    
    # Adjust by Volume Spikes
    volume_spike = df['volume'] / df['volume'].shift(1)
    is_spike = volume_spike > spike_threshold
    volume_spike[~is_spike] = 1  # If not a spike, set to 1 (no adjustment)
    
    adjusted_factor = combined_factor * volume_weight / volume_spike
    adjusted_factor = adjusted_factor.clip(lower=-cap_threshold, upper=cap_threshold)
    
    return adjusted_factor
```
