```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, p=7, threshold=3.0):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)

    # Combine Intraday Features
    mid_price = (df['high'] + df['low']) / 2
    intraday_feature = df['open'] - mid_price - opening_gap

    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']

    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)

    # Combine Factors
    intraday_momentum = intraday_range.rolling(window=n).mean() * (close_to_open_return + price_change)

    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=n).std()
    adjusted_intraday_momentum = intraday_momentum / volatility

    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    recent_volatility = abs_price_change.rolling(window=m).mean()
    scaled_feature = adjusted_intraday_momentum * recent_volatility

    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'] / df['volume'].shift(1)
    price_spike_factor = abs_price_change / abs_price_change.shift(1)
    combined_spike_factor = volume_spike_factor + price_spike_factor
    spike_adjusted_feature = scaled_feature / combined_spike_factor

    # Weight by Volume
    volume_weighted_feature = spike_adjusted_feature * df['volume']

    # Add Recent Amount Trend
    recent_amount_trend = df['amount'].rolling(window=p).mean()
    enhanced_feature = volume_weighted_feature * recent_amount_trend

    # Final Volatility Adjustment
    final_volatility = log_returns.rolling(window=n).std()
    final_adjusted_feature = enhanced_feature / final_volatility

    # Cap at a Threshold to Avoid Extreme Values
    final_alpha = np.clip(final_adjusted_feature, -threshold, threshold)

    return final_alpha
```
