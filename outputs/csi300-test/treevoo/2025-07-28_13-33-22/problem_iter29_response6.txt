```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Identify Volume and Price Spikes
    df['volume_spike'] = (df['volume'] / df['volume'].shift(1)) > 1.5
    df['price_spike'] = (df['close'] / df['close'].shift(1)) > 1.05
    df['combined_spikes'] = (df['volume_spike'] + df['price_spike']) / 2
    
    # Introduce Dynamic Volatility Adjustment
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['rolling_volatility'] = df['log_returns'].rolling(window=20).std()
    
    # Scale Intraday Momentum
    df['scaled_intraday_momentum'] = df['intraday_range'] / (df['combined_spikes'] + 1e-6)
    df['scaled_intraday_momentum'] = df['scaled_intraday_momentum'].clip(lower=-3, upper=3)
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['combined_intraday_feature'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = abs(df['close'] - df['close'].shift(1))
    df['recent_volatility'] = df['abs_price_change'].rolling(window=10).mean()
    
    # Adjust for Dynamic Volume Spikes
    df['volume_spike_factor'] = df['volume'] / df['volume'].shift(1)
    df['adjusted_combined_feature'] = df['combined_intraday_feature'] * df['volume_spike_factor'].clip(lower=0.5, upper=2)
    
    # Weight by Volume
    df['weighted_feature'] = df['adjusted_combined_feature'] * df['volume']
    
    # Incorporate Accumulated Volume
    df['accumulated_volume'] = df['volume'].rolling(window=20).sum()
    df['normalized_feature'] = df['weighted_feature'] / df['accumulated_volume']
    
    # Integrate Long-Term Momentum
    df['long_term_momentum'] = df['close'].rolling(window=60).mean()
    df['final_feature'] = df['normalized_feature'] + df['long_term_momentum']
    
    # Volume-Adjusted Transaction Intensity
    df['transaction_intensity'] = df['amount'] / df['volume']
    df['intensity_sum'] = df['transaction_intensity'].rolling(window=20).sum()
    
    # Price Trend Reversal
    df['trend_reversal'] = (df['close'] - df['close'].shift(1)) * (df['transaction_intensity'] / df['intensity_sum'])
    
    # Intraday Volatility Standard Deviation
    df['intraday_volatility'] = df['intraday_range'].rolling(window=20).std()
    
    # Apply Adaptive Moving Average Smoothing
    df['smoothed_final_feature'] = df['final_feature'].ewm(span=10, adjust=False).mean()
    
    return df['smoothed_final_feature']
```
