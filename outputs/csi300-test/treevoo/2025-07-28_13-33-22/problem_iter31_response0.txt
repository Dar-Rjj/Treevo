import pandas as pd

def heuristics_v2(df, n=10, m=10, k=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    intraday_features = (df['open'] - (df['high'] + df['low']) / 2) - opening_gap
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factor = avg_intraday_range * (opening_gap + price_change)
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    volatility_factor = abs_price_change.rolling(window=m).mean()
    enhanced_factor = combined_factor * volatility_factor
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'] / df['volume'].rolling(window=k).mean()
    price_spike_factor = abs_price_change / abs_price_change.rolling(window=k).mean()
    spike_adjusted_factor = enhanced_factor * (volume_spike_factor + price_spike_factor)
    
    # Weight by Volume
    weighted_factor = spike_adjusted_factor * df['volume']
    
    return weighted_factor
