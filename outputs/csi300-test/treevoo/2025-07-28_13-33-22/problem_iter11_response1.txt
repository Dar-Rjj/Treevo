```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low'] + df['close'] + df['open']
    
    # Adjust for Combined Volume and Price Spikes
    # Volume Spike Factor
    volume_spike_factor = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_spike_factor = volume_spike_factor.replace([np.inf, -np.inf], 0).fillna(0)
    
    # Price Spike Factor
    price_change = df['close'] - df['close'].shift(1)
    price_spike_factor = (price_change - price_change.shift(1)) / price_change.shift(1)
    price_spike_factor = price_spike_factor.replace([np.inf, -np.inf], 0).fillna(0)
    
    # Combine Volume and Price Spike Factors
    combined_spike = volume_spike_factor + price_spike_factor
    
    # Introduce Volatility Adjustment
    # Calculate Log Returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Standard Deviation of Log Returns
    volatility = log_returns.rolling(window=30).std().fillna(0)  # Fixed number of days = 30
    inverse_volatility = 1 / volatility
    
    # Adjust Intraday Momentum by Volatility
    adjusted_intraday_momentum = intraday_momentum * inverse_volatility
    
    # Scale Intraday Momentum
    scaled_intraday_momentum = adjusted_intraday_momentum / (combined_spike + 1e-6)  # Avoid division by zero
    
    # Cap at a Threshold to Avoid Extreme Values
    threshold = 5
    capped_scaled_intraday_momentum = np.clip(scaled_intraday_momentum, -threshold, threshold)
    
    return capped_scaled_intraday_momentum
```
