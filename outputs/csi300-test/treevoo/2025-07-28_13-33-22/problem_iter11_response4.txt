```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(1)
    
    # Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['combined_intraday_features'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Recent Volatility
    df['abs_price_change'] = abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=m).mean()
    df['enhanced_intraday_factor'] *= df['avg_abs_price_change']
    
    # Volume Weight
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']
    df['enhanced_intraday_factor'] *= df['volume_weight']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] / df['close'].shift(1)
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = df['intraday_momentum'] * df['volume_change'] + df['intraday_momentum']
    df['amount_weight'] = df['amount'] / df['amount'].rolling(window=5).mean()
    df['combined_factors'] *= df['amount_weight']
    
    # Adjust for Volume Impact
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['combined_factors'] = df['intraday_momentum'] * df['volume_ratio']
    
    # Final Factor
    df['final_factor'] = df['enhanced_intraday_factor'] + df['combined_factors']
    
    return df['final_factor']
```
