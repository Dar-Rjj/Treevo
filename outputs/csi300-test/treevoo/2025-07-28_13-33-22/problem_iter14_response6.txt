```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_spike'] = (df['volume'] > 1.5 * df['avg_volume_5d']).astype(int)
    
    df['avg_close_5d'] = df['close'].rolling(window=5).mean()
    df['price_spike'] = (df['close'] > 1.5 * df['avg_close_5d']).astype(int)
    
    # Combine Volume and Price Spikes
    df['combined_spike'] = df['volume_spike'] * df['price_spike']
    
    # Scale Intraday Momentum
    df['intraday_momentum'] = df['intraday_range'] / (df['combined_spike'] + 1)  # avoid division by zero
    df['intraday_momentum'] = df['intraday_momentum'].clip(upper=5)  # cap at a threshold
    
    # Adjust for Volume Impact
    df['volume_ratio'] = df['volume'] / df['avg_volume_5d']
    df['intraday_momentum'] *= df['volume_ratio']
    
    # Weight by Trade Amount
    df['avg_amount_10d'] = df['amount'].rolling(window=10).mean()
    df['amount_ratio'] = df['amount'] / df['avg_amount_10d']
    df['intraday_momentum'] *= df['amount_ratio']
    
    # Introduce Volatility Adjustment
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_return'].rolling(window=20).std()
    df['intraday_momentum'] /= df['volatility']
    
    return df['intraday_momentum']
```
