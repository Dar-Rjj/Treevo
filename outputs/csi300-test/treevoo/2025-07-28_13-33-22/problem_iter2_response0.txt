```python
import pandas as pd

def heuristics_v2(df, n=5, threshold=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Average Intraday Range over n periods
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    
    # Combine Factors
    combined_factor = (close_to_open_return + price_change) * avg_intraday_range
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1)).fillna(1)
    
    # Scale Combined Factor
    scaled_combined_factor = combined_factor / volume_spike
    scaled_combined_factor = scaled_combined_factor.clip(-threshold, threshold)
    
    # Weight by Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    volume_weight.fillna(1, inplace=True)
    
    # Final Alpha Factor
    alpha_factor = scaled_combined_factor * volume_weight
    
    return alpha_factor
```
