```python
import pandas as pd

def heuristics_v2(df, n=10, m=10, k=10):
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open'].shift(1)
    
    # Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['combined_intraday_features'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Recent Volatility
    df['abs_price_change'] = (df['close'] - df['close'].shift(1)).abs()
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=m).mean()
    df['enhanced_intraday_factor'] *= df['avg_abs_price_change']
    
    # Volume Adjusted Factor
    df['avg_volume'] = df['volume'].rolling(window=k).mean()
    df['volume_adjusted_factor'] = df['volume'] / df['avg_volume']
    df['volume_adjusted_factor'] = df['volume_adjusted_factor'].apply(lambda x: 1 if x > 1 else 0.5)
    df['enhanced_intraday_factor'] *= df['volume_adjusted_factor']
    
    # Volume and Price Spike Adjustment
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (df['close'] > df['close'].shift(1)).astype(int)
    df['combined_spike'] = df['volume_spike'] + df['price_spike']
    df['enhanced_intraday_factor'] /= df['combined_spike']
    df['enhanced_intraday_factor'] = df['enhanced_intraday_factor'].clip(upper=3)  # Cap at 3 to avoid extreme values
    
    # Volume Weight
    df['volume_weight'] = df['volume'] / df['avg_volume']
    
    # Transaction Intensity
    df['transaction_intensity'] = df['amount'] / df['volume']
    df['avg_transaction_intensity'] = df['transaction_intensity'].rolling(window=n).mean()
    
    # Volume and Transaction Intensity Adjustment
    df['volume_weight'] = df['volume'] / df['avg_volume']
    df['transaction_intensity_weight'] = df['transaction_intensity'] / df['avg_transaction_intensity']
    df['final_factor'] = df['enhanced_intraday_factor'] * df['volume_weight'] * df['transaction_intensity_weight']
    
    return df['final_factor']
```
