import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1)).astype(int)
    price_spike = (df['close'] > df['close'].shift(1)).astype(int)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    vol_adjustment = log_returns.rolling(window=20).std()
    intraday_momentum_vol_adj = intraday_momentum / (vol_adjustment + 1e-8)  # Avoid division by zero
    
    # Scale Intraday Momentum
    spike_adjustment = volume_spike + price_spike
    intraday_momentum_scaled = intraday_momentum_vol_adj * np.clip(spike_adjustment, 0, 5)  # Cap at 5
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday_features = df['open'] - ((df['high'] + df['low']) / 2 - opening_gap)
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Combine Factors
    combined_factors = intraday_range * (close_to_open_return + df['close'] - df['close'].shift(1))
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=5).mean()
    combined_factors_enhanced = combined_factors * avg_abs_price_change
    
    # Adjust Combined Factor by Volume Spike
    combined_factors_adjusted = combined_factors_enhanced / (volume_spike + 1e-8)  # Avoid division by zero
    combined_factors_capped = np.clip(combined_factors_adjusted, -5, 5)  # Cap at Â±5
    
    # Weight by Relative Volume
    relative_volume = df['volume'] / df['volume'].rolling(window=20).mean()
    final_factor = combined_factors_capped * relative_volume
    
    return final_factor
