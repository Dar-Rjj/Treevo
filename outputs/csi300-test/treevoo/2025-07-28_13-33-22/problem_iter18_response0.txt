import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['price_spike'] = (df['close'] > df['close'].shift(1)).astype(int)
    
    # Introduce Volatility Adjustment
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    df['intraday_momentum_adjusted'] = df['intraday_momentum'] * (1 / df['volatility'])
    
    # Scale Intraday Momentum
    df['spike_combined'] = df['volume_spike'] + df['price_spike']
    df['spike_combined_capped'] = np.clip(df['spike_combined'], 0, 1)
    df['intraday_momentum_scaled'] = df['intraday_momentum_adjusted'] * df['spike_combined_capped']
    
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['intraday_features_combined'] = df['open'] - (df['intraday_midpoint'] - df['opening_gap'])
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    df['combined_factor'] = df['intraday_range'] * (df['close_to_open_return'] + df['price_change'])
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=5).mean()
    df['combined_factor_enhanced'] = df['combined_factor'] * df['avg_abs_price_change']
    
    # Adjust Combined Factor by Volume Spike
    df['combined_factor_adjusted'] = df['combined_factor_enhanced'] / (df['volume_spike'] + 1)  # Avoid division by zero
    df['combined_factor_adjusted_capped'] = np.clip(df['combined_factor_adjusted'], -3, 3)
    
    # Weight by Relative Volume
    df['relative_volume'] = df['volume'] / df['volume'].rolling(window=20).mean()
    df['final_factor'] = df['combined_factor_adjusted_capped'] * df['relative_volume']
    
    # Apply Moving Average to Smooth the Final Factor
    df['final_factor_smoothed'] = df['final_factor'].rolling(window=5).mean()
    
    return df['final_factor_smoothed']
