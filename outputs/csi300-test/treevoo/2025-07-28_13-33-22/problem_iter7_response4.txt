```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Volume-Weighted Moving Average of Intraday Returns
    weighted_intraday_return = intraday_return * df['volume']
    sum_weighted_returns = weighted_intraday_return.rolling(window=5).sum()
    sum_volumes = df['volume'].rolling(window=5).sum()
    volume_weighted_avg = sum_weighted_returns / sum_volumes
    
    # Calculate Daily Price Change
    daily_price_change = (df['close'] - df['open']) / df['open']
    
    # Calculate Momentum Score
    momentum_score = 0.5 * volume_weighted_avg + daily_price_change
    
    # Apply Hysteresis
    hysteresis = momentum_score.copy()
    hysteresis[momentum_score > 0.05] = 1
    hysteresis[momentum_score < -0.05] = -1
    hysteresis[(momentum_score <= 0.05) & (momentum_score >= -0.05)] = 0
    
    return hysteresis
```
