def heuristics_v2(df):
    # Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Factor
    combined_factor = intraday_momentum + close_to_open_return + price_change
    
    # Volume Weight
    avg_volume = df['volume'].mean()
    volume_weight = df['volume'] / avg_volume
    weighted_combined_factor = combined_factor * volume_weight
    
    # Adjust by Volume and Price Spikes
    # Identify Volume Spikes and cap at threshold
    volume_spike_threshold = 3
    volume_spikes = (df['volume'] > volume_spike_threshold * avg_volume).astype(int) * volume_spike_threshold * avg_volume
    adjusted_volume = df['volume'] - volume_spikes + volume_spike_threshold * avg_volume
    
    # Identify Price Spikes and cap at threshold
    price_spike_threshold = 0.05 * df['close'].rolling(window=20).std()
    price_spikes = (abs(df['close'] - df['close'].shift(1)) > price_spike_threshold).astype(int) * price_spike_threshold
    adjusted_price_change = abs(df['close'] - df['close'].shift(1)) - price_spikes + price_spike_threshold
    
    # Combine Volume and Price Spikes
    combined_spike = (adjusted_volume + adjusted_price_change) / 2
    
    # Final Alpha Factor
    final_alpha_factor = weighted_combined_factor / combined_spike
    
    return final_alpha_factor
