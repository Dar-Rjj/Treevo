```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    n = 5  # Number of days for momentum calculation

    # Calculate daily log returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))

    # Compute weighted average log return using volume as weights
    df['weighted_log_returns'] = (df['log_returns'].rolling(window=n).apply(
        lambda x: np.sum(x * df['volume'].iloc[-len(x):] / df['volume'].iloc[-len(x):].sum()),
        raw=False
    ))

    # Calculate high-low range over short-term
    df['high_low_range'] = df['high'] - df['low']

    # Calculate daily volatility
    df['daily_volatility'] = (df['high'] - df['low']) / df['close'].shift(1)

    # Calculate volume change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)

    # Subtract combined reversal, volatility, and volume change from weighted average log return
    df['alpha_factor'] = df['weighted_log_returns'] - (df['high_low_range'] + df['daily_volatility'] + df['volume_change'])

    return df['alpha_factor']
```
