import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) + (df['close'] + df['open'])
    
    # Adjust for Volume and Price Spikes
    
    # Calculate Volume Spike Factor
    volume_spike_factor = df['volume'] / df['volume'].shift(1)
    
    # Calculate Price Spike Factor
    price_change = df['close'] - df['open']
    prev_price_change = df['close'].shift(1) - df['open'].shift(1)
    price_spike_factor = price_change / prev_price_change
    
    # Combine and Scale Intraday Momentum
    adjusted_intraday_momentum = intraday_momentum * volume_spike_factor * price_spike_factor
    
    # Cap at Threshold to Avoid Extreme Values
    threshold = 100
    capped_adjusted_momentum = adjusted_intraday_momentum.clip(lower=-threshold, upper=threshold)
    
    return capped_adjusted_momentum
