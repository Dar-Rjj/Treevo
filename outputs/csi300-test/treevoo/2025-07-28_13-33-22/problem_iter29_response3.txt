```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, p=10, threshold=3.0):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Intraday Features
    intraday_midpoint = (df['high'] + df['low']) / 2
    opening_gap = df['open'] - df['close'].shift(1)
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = np.abs(price_change)
    ema_abs_price_change = abs_price_change.ewm(span=m, adjust=False).mean()
    recent_volatility = enhanced_intraday_factor * ema_abs_price_change
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = df['volume'] / df['volume'].shift(1)
    price_spike_factor = np.abs(price_change) / np.abs(price_change).shift(1)
    combined_spike_factor = volume_spike_factor * price_spike_factor
    adjusted_recent_volatility = np.minimum(recent_volatility * combined_spike_factor, threshold)
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    log_returns_std = log_returns.rolling(window=p).std()
    volatility_adjustment = log_returns_std
    adjusted_volatility = adjusted_recent_volatility / volatility_adjustment
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = np.where(df['volume'] > avg_volume, 1.0, 0.5)
    
    # Final Factor
    final_factor = enhanced_intraday_factor * adjusted_volatility * volume_weight
    recent_amount_trend = df['amount'].rolling(window=p).mean()
    final_factor *= recent_amount_trend
    
    return final_factor
```
