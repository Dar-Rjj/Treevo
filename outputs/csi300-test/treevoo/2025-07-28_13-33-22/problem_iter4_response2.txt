```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate daily log returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Define the number of days to look back for the weighted average
    n = 10
    
    # Compute the weighted average log return
    df['weighted_log_return'] = (df['log_return'].rolling(window=n).apply(
        lambda x: (x * df.loc[x.index, 'volume']).sum() / df.loc[x.index, 'volume'].sum(), raw=False
    ))
    
    # Calculate high-low range
    df['high_low_range'] = df['high'] - df['low']
    
    # Subtract high-low range from weighted average log return to get the final factor
    df['factor'] = df['weighted_log_return'] - df['high_low_range']
    
    return df['factor'].dropna()
```
