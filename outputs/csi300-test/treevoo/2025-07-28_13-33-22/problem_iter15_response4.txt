import pandas as pd

def heuristics_v2(df, n=20):
    # Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['close_to_open'] = df['open'] - df['close'].shift(1)
    
    # Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Volatility
    df['intraday_volatility'] = df['intraday_range'].rolling(window=n).std()
    
    # Combined Factor
    avg_intraday_range = df['intraday_range'].rolling(window=n).mean()
    combined_factor = (avg_intraday_range + 
                       (df['close_to_open'] + df['price_change']) + 
                       (df['intraday_volatility'] * df['close_to_open']))
    
    # Daily Momentum
    daily_momentum = df['close'] - df['close'].shift(n)
    combined_factor += daily_momentum
    
    # Volume and Price Spike Adjustment
    volume_spike = (df['volume'] > 1.5 * df['volume'].shift(1))
    price_spike = (df['close'] > 1.5 * df['close'].shift(1)) | (df['close'] < 0.67 * df['close'].shift(1))
    combined_spike = volume_spike + price_spike
    combined_factor /= combined_spike.replace(0, 1)
    combined_factor = combined_factor.clip(lower=-3, upper=3)  # Cap at a threshold to avoid extreme values
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    
    # Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    avg_transaction_intensity = transaction_intensity.rolling(window=n).mean()
    transaction_intensity_adjustment = (transaction_intensity / avg_transaction_intensity).clip(lower=0.5, upper=2)
    
    # Adjusted Combined Factor
    adjusted_combined_factor = combined_factor * volume_weight * transaction_intensity_adjustment
    
    # Incorporate Short-Term Reversal
    short_term_reversal = (df['close'] - df['close'].shift(n)) / df['intraday_range'].rolling(window=n).mean()
    adjusted_combined_factor += short_term_reversal
    
    return adjusted_combined_factor.dropna()
