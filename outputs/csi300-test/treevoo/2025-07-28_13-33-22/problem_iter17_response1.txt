import pandas as pd

def heuristics_v2(df, n=20, m=10, threshold=5):
    # Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    combined_intraday_features = (df['open'] - intraday_midpoint) - opening_gap
    
    # Intraday Volatility
    intraday_volatility = intraday_range.rolling(window=n).std()
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = abs(price_change)
    recent_volatility = abs_price_change.rolling(window=m).mean()
    enhanced_intraday_factor = enhanced_intraday_factor * recent_volatility
    
    # Volume and Price Spike Adjustment
    volume_spike = df['volume'] > df['volume'].shift(1) * 1.5
    price_spike = abs(df['close'] - df['close'].shift(1)) > abs_price_change.rolling(window=m).mean() * 2
    combined_spike = volume_spike + price_spike
    adjusted_factor = enhanced_intraday_factor / (combined_spike + 1)
    adjusted_factor = adjusted_factor.clip(lower=-threshold, upper=threshold)
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = df['volume'] / avg_volume
    
    # Transaction Intensity
    transaction_intensity = df['amount'] / df['volume']
    avg_transaction_intensity = transaction_intensity.rolling(window=n).mean()
    transaction_intensity = transaction_intensity / avg_transaction_intensity
    
    # Volume, Transaction Intensity, and Adjusted Volatility Factor
    volume_adjusted_factor = volume_weight.apply(lambda x: 1 if x > 1 else 0.5)
    adjusted_factor = adjusted_factor * (volume_adjusted_factor * transaction_intensity)
    
    # Weighted Volatility
    weighted_volatility = recent_volatility * volume_adjusted_factor
    
    # Adjust for Volume Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(lower=1, upper=5)
    adjusted_factor = adjusted_factor * volume_spike_factor
    
    return adjusted_factor
