```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Weighted Average Log Return
    n = 20  # Number of days to look back
    df['weighted_log_return'] = (df['log_return'].rolling(window=n).sum() * df['volume'].rolling(window=n).mean()) / df['volume'].rolling(window=n).sum()
    
    # Adjust for Short-Term Dynamics
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate True Range
    df['true_range'] = df[['high', 'low']].sub(df['close'].shift(1), axis=0).abs().max(axis=1)
    
    # Calculate Daily Volatility
    df['daily_volatility'] = (df['high'] - df['low']) / 2
    
    # Calculate Volume Change
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Subtract Adjustments from Weighted Average Log Return
    df['adjusted_factor'] = df['weighted_log_return'] - (df['high_low_range'] + df['true_range'] + df['daily_volatility'] + df['volume_change'])
    
    return df['adjusted_factor']
```
