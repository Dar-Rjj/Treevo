```python
import pandas as pd

def heuristics_v2(df, n=10, spike_threshold=2, cap_threshold=5, momentum_window=20):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Combine Factors
    combined_factors = (intraday_range.rolling(window=n).mean() * 
                        (close_to_open_return + price_change))
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] / df['volume'].shift(1)).fillna(1)
    
    # Adjust Combined Factor by Volume Spike
    adjusted_combined_factors = (combined_factors / volume_spike).clip(-cap_threshold, cap_threshold)
    
    # Calculate Daily Returns
    daily_returns = df['close'].pct_change().fillna(0)
    
    # Calculate Simple Moving Average of Daily Returns
    sma_daily_returns = daily_returns.rolling(window=n).mean()
    
    # Adjust for Volume Volatility
    daily_volume_changes = df['volume'].diff().fillna(0)
    volume_volatility = daily_volume_changes.rolling(window=n).std()
    normalized_combined_factors = adjusted_combined_factors / volume_volatility
    
    # Apply Momentum Indicator
    momentum = normalized_combined_factors.rolling(window=momentum_window).sum()
    
    # Finalize the Momentum Factor
    final_alpha_factor = momentum
    
    return final_alpha_factor
```
