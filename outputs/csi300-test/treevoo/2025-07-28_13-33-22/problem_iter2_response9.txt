```python
def heuristics_v2(data, n=10, cap_threshold=5):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = data['close'] - data['open']
    
    # Calculate Price Change
    price_change = data['close'].diff()
    
    # Combine Factors
    combined_factor = (intraday_range.rolling(window=n).mean()) * (close_to_open_return + price_change)
    
    # Identify Volume Spikes
    volume_spike = data['volume'] / data['volume'].shift(1)
    
    # Adjust Combined Factor by Volume Spike
    adjusted_factor = combined_factor / volume_spike
    
    # Cap at a Threshold to Avoid Extreme Values
    adjusted_factor = adjusted_factor.clip(lower=-cap_threshold, upper=cap_threshold)
    
    return adjusted_factor
```
