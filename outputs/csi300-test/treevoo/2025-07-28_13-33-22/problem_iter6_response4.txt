```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate Volume Change
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Momentum and Volume Change
    combined_intraday = intraday_momentum * volume_change + intraday_momentum
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Weight by Trade Amount
    amount_5_day_sum = df['amount'].rolling(window=5).sum()
    weight_by_amount = df['amount'] / amount_5_day_sum
    
    # Weight by Volume
    volume_n_period_sum = df['volume'].rolling(window=10).sum()
    weight_by_volume = df['volume'] / volume_n_period_sum
    
    # Combine Weights
    combined_weights = combined_intraday * weight_by_amount * weight_by_volume
    
    # Final Combined Factor
    final_factor = (close_to_open_return + price_change) * combined_weights
    
    return final_factor
```
