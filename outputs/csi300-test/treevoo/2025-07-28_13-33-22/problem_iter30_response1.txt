    intraday_range = df['high'] - df['low']

    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_intraday_feature = (df['open'] - intraday_midpoint) - opening_gap
    
    # Enhance with Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    avg_abs_price_change = abs_price_change.rolling(window=10).mean()
    enhanced_feature = combined_intraday_feature / (avg_abs_price_change + 1)  # +1 to avoid division by zero
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1))
    price_spike_factor = (df['close'] - df['close'].shift(1)) / (df['close'].shift(1) - df['close'].shift(2))
    combined_spike_factor = (volume_spike_factor + price_spike_factor) / 2
    adjusted_feature = enhanced_feature * combined_spike_factor
    
    # Weight by Volume and Recent Amount Trend
    recent_amount_trend = df['amount'].rolling(window=10).mean()
    weighted_feature = (adjusted_feature * df['volume']) / (volume_spike_factor + 1)  # +1 to avoid division by zero
    weighted_feature = weighted_feature.clip(upper=5)  # Cap at a threshold
    final_weighted_feature = weighted_feature * recent_amount_trend
    
    # Incorporate Moving Averages
    short_term_ma = df['close'].rolling(window=10).mean()
    long_term_ma = df['close'].rolling(window=60).mean()
    ma_diff = short_term_ma - long_term_ma

    momentum = df['close'] - df['close'].shift(30)

    ad_line = ((df['close'] - df['low']) / (df['high'] - df['low'])) * df['volume']
    ad_line = ad_line.rolling(window=10).sum()
    ad_adjusted_feature = final_weighted_feature * (ad_line + 1)  # +1 to avoid division by zero
