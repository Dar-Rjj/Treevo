```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Log Returns
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute Weighted Average Log Return
    n = 5  # Number of days to look back for the weighted average
    weights = df['volume'].rolling(window=n).apply(lambda x: x / x.sum(), raw=False)
    df['weighted_log_return'] = (df['log_return'].rolling(window=n) * weights).sum()
    
    # Incorporate Short-Term Price Reversal
    df['high_low_range'] = df['high'] - df['low']
    df['alpha_factor'] = df['weighted_log_return'] - df['high_low_range']
    
    return df['alpha_factor'].dropna()
```
