```python
import pandas as pd

def heuristics_v2(df, n=20, m=10, threshold=3.0):
    # Calculate Intraday Return
    df['IntradayRange'] = df['high'] - df['low']
    
    # Close-to-Open Return
    df['CloseToOpenReturn'] = df['close'] - df['open'].shift(1)
    
    # Price Change
    df['PriceChange'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['IntradayMidpoint'] = (df['high'] + df['low']) / 2
    
    # Opening Gap
    df['OpeningGap'] = df['open'] - df['close'].shift(1)
    
    # Combined Intraday Features
    df['CombinedIntradayFeatures'] = (df['open'] - df['IntradayMidpoint']) - df['OpeningGap']
    
    # Enhanced Intraday Factor
    df['AvgIntradayRange'] = df['IntradayRange'].rolling(window=n).mean()
    df['EnhancedIntradayFactor'] = df['AvgIntradayRange'] * (df['CloseToOpenReturn'] + df['PriceChange']) + df['CombinedIntradayFeatures']
    
    # Recent Volatility
    df['AbsPriceChange'] = abs(df['PriceChange'])
    df['RecentVolatility'] = df['AbsPriceChange'].rolling(window=m).mean()
    
    # Volume Weight
    df['AvgVolume'] = df['volume'].rolling(window=n).mean()
    df['VolumeWeight'] = df['volume'] / df['AvgVolume']
    df['VolumeWeight'] = df['VolumeWeight'].apply(lambda x: 1 if x > 1 else 0.5)
    
    # Adjust for Combined Volume and Price Spikes
    df['VolumeSpikeFactor'] = df['volume'] / df['volume'].shift(1)
    df['PriceSpikeFactor'] = df['PriceChange'] / df['PriceChange'].shift(1)
    df['SpikeAdjustment'] = (df['VolumeSpikeFactor'] + df['PriceSpikeFactor']) / 2
    df['SpikeAdjustment'] = df['SpikeAdjustment'].clip(lower=-threshold, upper=threshold)
    
    # Adjusted Intraday Momentum
    df['AdjustedIntradayMomentum'] = (df['high'] - df['low']) + (df['close'] - df['open'])
    
    # Final Factor
    df['VolumeAdjustedFactor'] = df['VolumeWeight'] * df['SpikeAdjustment']
    df['FinalFactor'] = (df['EnhancedIntradayFactor'] * df['RecentVolatility'] * df['VolumeAdjustedFactor']) + df['AdjustedIntradayMomentum']
    
    return df['FinalFactor']
```
