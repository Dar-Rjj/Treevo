def heuristics_v2(data, n=10, m=20, k=5, p=10):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']
    
    # Calculate Opening Gap
    opening_gap = data['open'] - data['close'].shift(1)
    
    # Combine Intraday Features
    mid_price = (data['high'] + data['low']) / 2
    intraday_features = data['open'] - mid_price - opening_gap
    
    # Calculate Price Change
    price_change = data['close'] - data['close'].shift(1)
    
    # Combine Factors
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    combined_factors = avg_intraday_range * (opening_gap + price_change)
    
    # Enhance with Recent Volatility
    abs_price_change = abs(data['close'] - data['close'].shift(1))
    volatility_factor = abs_price_change.rolling(window=m).mean()
    enhanced_factors = combined_factors * volatility_factor
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = data['volume'] / data['volume'].rolling(window=k).mean()
    price_spike_factor = abs_price_change / abs_price_change.rolling(window=k).mean()
    combined_spike_factor = volume_spike_factor * price_spike_factor
    adjusted_features = enhanced_factors * combined_spike_factor
    
    # Weight by Volume
    volume_weighted_features = adjusted_features * data['volume']
    
    # Add Recent Amount Trend
    recent_amount_trend = data['amount'].rolling(window=p).sum()
    final_feature = volume_weighted_features * recent_amount_trend
    
    return final_feature
