import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    df['intraday_feature'] = (df['open'] - df['intraday_midpoint']) - df['opening_gap']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Average Intraday Range over n periods
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    
    # Combine Factors
    df['combined_factor'] = (df['close_to_open_return'] + df['price_change']) * df['avg_intraday_range']
    
    # Weight by Volume
    df['avg_volume'] = df['volume'].rolling(window=n).mean()
    df['volume_weight'] = df['volume'] / df['avg_volume']
    df['weighted_factor'] = df['combined_factor'] * df['volume_weight']
    
    # Return the alpha factor
    return df['weighted_factor'].dropna()
