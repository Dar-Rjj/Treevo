```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Features
    df['intraday_momentum'] = df['high'] - df['low']
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Momentum and Reversal
    df['combined_feature'] = (df['intraday_momentum'] - (df['open'] - df['intraday_midpoint'])) - df['opening_gap']
    
    # Identify Volume Spikes
    df['volume_spike'] = df['volume'] > df['volume'].shift(1) * 2  # Example criteria for a volume spike
    
    # Adjust Combined Feature by Volume Spike
    threshold = 10  # Example cap threshold
    adjusted_feature = []
    for i in range(len(df)):
        if df['volume_spike'].iloc[i]:
            value = df['combined_feature'].iloc[i] / df['volume'].iloc[i]
            adjusted_feature.append(min(value, threshold))
        else:
            adjusted_feature.append(df['combined_feature'].iloc[i] * df['volume'].iloc[i])
    
    return pd.Series(adjusted_feature, index=df.index)
```
