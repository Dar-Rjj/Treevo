```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the weighted average of open and close prices using volume as weights
    weighted_avg = (df['open'] * df['volume'] + df['close'] * df['volume']) / (2 * df['volume'])
    
    # Calculate the difference between high and low, a measure of intraday volatility
    intraday_range = df['high'] - df['low']
    
    # Measure buying pressure by comparing amount traded to volume
    buying_pressure = df['amount'] / df['volume']
    
    # Combine the above into a single factor
    factor = (weighted_avg - df['close'].shift(1)) / intraday_range * buying_pressure
    
    return factor
```
