```python
import pandas as pd

def heuristics_v2(df, n=20, m=10):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Combined Intraday Features
    intraday_midpoint = (df['high'] + df['low']) / 2
    opening_gap = df['open'] - df['close'].shift(1)
    combined_intraday_features = (df['open'] - intraday_midpoint) + opening_gap
    
    # Enhanced Intraday Factor
    avg_intraday_range = intraday_range.rolling(window=n).mean()
    enhanced_intraday_factor = avg_intraday_range * (close_to_open_return + price_change) + combined_intraday_features
    
    # Recent Volatility
    abs_price_change = (df['close'] - df['close'].shift(1)).abs()
    avg_abs_price_change = abs_price_change.rolling(window=m).mean()
    recent_volatility = enhanced_intraday_factor * avg_abs_price_change
    
    # Adjust for Combined Volume and Price Spikes
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).clip(lower=0, upper=5)
    price_spike_factor = (abs_price_change / abs_price_change.shift(1)).clip(lower=0, upper=5)
    adjusted_recent_volatility = recent_volatility * (volume_spike_factor * price_spike_factor)
    adjusted_recent_volatility = adjusted_recent_volatility.clip(lower=-1, upper=1)
    
    # Volume Weight
    avg_volume = df['volume'].rolling(window=n).mean()
    volume_weight = (df['volume'] / avg_volume).apply(lambda x: 1 if x > 1 else 0.5)
    
    # Trend Confirmation
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd = short_ema - long_ema
    signal_line = macd.ewm(span=9, adjust=False).mean()
    trend_confirmation = (macd > signal_line).astype(float).apply(lambda x: 1 if x == 1 else 0.5)
    
    # Final Factor
    final_factor = (enhanced_intraday_factor * adjusted_recent_volatility) * volume_weight * trend_confirmation
    
    return final_factor
```
