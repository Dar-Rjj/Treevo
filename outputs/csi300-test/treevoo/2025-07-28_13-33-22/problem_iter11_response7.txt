```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open'].shift(1)
    
    # Calculate Price Change
    price_change = df['close'] - df['close'].shift(1)
    
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Momentum and Range
    combined_momentum_range = intraday_range * (close_to_open_return + price_change)
    
    # Combine Intraday Momentum and Reversal
    combined_intraday_reversal = (intraday_momentum - (df['open'] - intraday_midpoint)) - opening_gap
    
    # Identify Volume Spikes
    volume_spike = (df['volume'] > df['volume'].shift(1) * 2).astype(int)
    
    # Adjust Combined Feature by Volume Spike
    adjusted_feature = combined_momentum_range * combined_intraday_reversal
    adjusted_feature[volume_spike == 1] = adjusted_feature / df['volume']
    adjusted_feature = adjusted_feature.clip(lower=-10, upper=10)  # Cap at a Threshold to Avoid Extreme Values
    
    # Weight by Relative Volume
    average_volume = df['volume'].rolling(window=20).mean()
    relative_volume = df['volume'] / average_volume
    weighted_factor = adjusted_feature * relative_volume
    
    return weighted_factor
```
