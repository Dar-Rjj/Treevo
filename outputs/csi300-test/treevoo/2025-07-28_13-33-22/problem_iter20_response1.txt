```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=10, k=10, p=50, cap_threshold=5):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust by Volume and Price Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)) * 1
    df['price_spike'] = (df['close'] > df['close'].shift(1)) * 1
    
    # Introduce Volatility Adjustment
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=n).std()
    df['intraday_momentum'] = df['intraday_range'] / df['volatility']
    
    # Scale Intraday Momentum
    df['combined_spikes'] = df['volume_spike'] + df['price_spike']
    df['scaled_intraday_momentum'] = df['intraday_momentum'] * np.minimum(cap_threshold, df['combined_spikes'])
    
    # Calculate Opening Gap
    df['opening_gap'] = df['open'] - df['close'].shift(1)
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'] - df['open']
    
    # Calculate Price Change
    df['price_change'] = df['close'] - df['close'].shift(1)
    
    # Intraday Midpoint
    df['intraday_midpoint'] = (df['high'] + df['low']) / 2
    
    # Combined Intraday Features
    df['combined_intraday_features'] = df['open'] - df['intraday_midpoint'] - df['opening_gap']
    
    # Enhanced Intraday Factor
    df['avg_intraday_range'] = df['intraday_range'].rolling(window=n).mean()
    df['enhanced_intraday_factor'] = df['avg_intraday_range'] * (df['close_to_open_return'] + df['price_change']) + df['combined_intraday_features']
    
    # Enhance with Recent Volatility
    df['abs_price_change'] = np.abs(df['close'] - df['close'].shift(1))
    df['avg_abs_price_change'] = df['abs_price_change'].rolling(window=m).mean()
    df['enhanced_factor'] = df['enhanced_intraday_factor'] * df['avg_abs_price_change']
    
    # Weight by Relative Volume
    df['avg_volume'] = df['volume'].rolling(window=k).mean()
    df['relative_volume'] = np.where(df['volume'] > df['avg_volume'], 1, 0.5)
    df['weighted_enhanced_factor'] = df['enhanced_factor'] * df['relative_volume']
    
    # Incorporate Trading Volume Trend
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['smoothed_volume_trend'] = df['volume_diff'].rolling(window=k).mean()
    df['volume_trend_adjusted_factor'] = df['weighted_enhanced_factor'] * df['smoothed_volume_trend']
    
    # Incorporate Price Trend
    df['sma'] = df['close'].rolling(window=p).mean()
    df['price_trend_adjusted_factor'] = df['volume_trend_adjusted_factor'] * (df['close'] - df['sma'])
    
    # Final Factor
    df['final_factor'] = df['price_trend_adjusted_factor']
    df['final_factor'] = np.clip(df['final_factor'], -cap_threshold, cap_threshold)
    
    return df['final_factor'].dropna()
```
