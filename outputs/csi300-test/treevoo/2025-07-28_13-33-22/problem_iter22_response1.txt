```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Identify Spikes
    volume_spikes = (df['volume'] > df['volume'].shift(1)).astype(int)
    price_spikes = (df['close'] > df['close'].shift(1)).astype(int)
    combined_spikes = volume_spikes + price_spikes
    
    # Introduce Volatility Adjustment
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=30).std()
    
    # Scale Intraday Momentum
    scaled_intraday_momentum = intraday_range / (combined_spikes + 1)  # +1 to avoid division by zero
    scaled_intraday_momentum = np.clip(scaled_intraday_momentum, -5, 5)  # Cap at a threshold
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate Opening Gap
    opening_gap = df['open'] - df['close'].shift(1)
    
    # Combine Intraday Features
    combined_feature = df['open'] - intraday_midpoint - opening_gap
    
    # Enhance with Recent Volatility
    abs_price_change = np.abs(df['close'] - df['close'].shift(1))
    avg_abs_price_change = abs_price_change.rolling(window=10).mean()
    combined_feature += avg_abs_price_change
    
    # Adjust for Volume Spikes and Weight by Volume
    volume_spike_factor = (df['volume'] / df['volume'].shift(1)).fillna(1)
    volume_spike_factor = np.clip(volume_spike_factor, 0.5, 2)  # Cap at a threshold
    adjusted_feature = combined_feature * volume_spike_factor * df['volume']
    
    return adjusted_feature
```
