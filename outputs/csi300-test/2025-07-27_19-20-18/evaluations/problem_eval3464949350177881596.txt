import pandas as pd

def heuristics_v2(df):
    momentum = df['close'] - df['close'].shift(20)
    delta = df['close'].diff()
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    rsi_period = 14
    avg_gain = up.rolling(window=rsi_period).mean()
    avg_loss = down.abs().rolling(window=rsi_period).mean()
    rs = avg_gain / avg_loss
    rsi = 100.0 - (100.0 / (1.0 + rs))
    heuristics_matrix = momentum * rsi
    return heuristics_matrix
