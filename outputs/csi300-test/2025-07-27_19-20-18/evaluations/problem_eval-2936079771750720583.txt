import pandas as pd

def heuristics_v2(df):
    def calculate_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def calculate_adl(high, low, close, volume):
        adl = ((close - low) - (high - close)) / (high - low) * volume
        return adl.cumsum()

    def calculate_bollinger_bands(series, window=20, num_std=2):
        ma = series.rolling(window=window).mean()
        std = series.rolling(window=window).std()
        upper_band = ma + (std * num_std)
        lower_band = ma - (std * num_std)
        bb_value = (series - lower_band) / (upper_band - lower_band)
        return bb_value

    close = df['close']
    high = df['high']
    low = df['low']
    volume = df['volume']

    rsi = calculate_rsi(close, window=14)
    adl = calculate_adl(high, low, close, volume)
    bb_value = calculate_bollinger_bands(close, window=20, num_std=2)

    heuristics_matrix = (rsi - 50) * (adl / adl.shift(1)) * bb_value

    return heuristics_matrix
