import pandas as pd
import numpy as np

def heuristics_v2(df):
    def rsi_series(price_series, window=14):
        delta = price_series.diff(1)
        gain = (delta.where(delta > 0, 0)).fillna(0)
        loss = (-delta.where(delta < 0, 0)).fillna(0)
        avg_gain = gain.rolling(window=window).mean()
        avg_loss = loss.rolling(window=window).mean()
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    def atr_series(high, low, close, window=14):
        tr = np.maximum.reduce([high - low, np.abs(high - close.shift(1)), np.abs(low - close.shift(1))])
        atr = tr.rolling(window=window).mean()
        atr_change = atr.pct_change().fillna(0)
        return atr_change

    df['rsi'] = df['close'].rolling(window=14).apply(rsi_series, raw=False)
    df['atr_change'] = atr_series(df['high'], df['low'], df['close'])
    heuristics_matrix = 0.6 * df['rsi'] + 0.4 * df['atr_change']
    
    return heuristics_matrix
