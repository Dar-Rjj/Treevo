def heuristics_v2(df):
    # Calculate the percentage change in close prices
    pct_change_close = df['close'].pct_change()
    
    # Calculate the relative difference between high and low prices
    rel_diff_high_low = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate the rate of change in volumes
    roc_volume = df['volume'].pct_change()
    
    # Aggregate the three components into a heuristic score
    heuristics_matrix = (pct_change_close + rel_diff_high_low + roc_volume) / 3
    
    return heuristics_matrix
