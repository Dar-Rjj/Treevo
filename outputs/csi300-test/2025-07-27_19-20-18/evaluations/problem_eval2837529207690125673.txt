import pandas as pd
import numpy as np

def heuristics_v2(df):
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=20, adjust=False).mean()
    volume_ratio_sqrt = (df['volume'] / ema_volume).apply(lambda x: max(0.0001, x)).apply(np.sqrt)
    heuristics_matrix = ema_close * volume_ratio_sqrt
    return heuristics_matrix
