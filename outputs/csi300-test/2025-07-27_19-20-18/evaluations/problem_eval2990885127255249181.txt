import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=10):
        return series.rolling(window=window).mean()

    def calculate_momentum(series, period=5):
        return series / series.shift(period) - 1

    def calculate_volatility(series, window=30):
        return series.rolling(window=window).std()

    close = df['close']
    high = df['high']
    low = df['low']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=10)
    ma_long = calculate_moving_average(close, window=50)
    momentum = calculate_momentum(close, period=5)
    vol = calculate_volatility(close, window=30)

    range_factor = (high - low) / (high + low)

    heuristics_matrix = (ma_short - ma_long) * momentum / vol * range_factor * (volume / volume.shift(1))

    return heuristics_matrix
