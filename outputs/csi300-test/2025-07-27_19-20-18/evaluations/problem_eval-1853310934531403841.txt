import pandas as pd

def heuristics_v2(df):
    range_change = (df['high'] - df['low']).pct_change()
    avg_range_change = range_change.rolling(window=5).mean()
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    heuristics_matrix = avg_range_change * volume_ratio
    return heuristics_matrix
