import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=14):
        return series.rolling(window=window).mean()

    def calculate_momentum(series, period=10):
        return series / series.shift(period) - 1

    def calculate_average_volume(series, window=10):
        return series.rolling(window=window).mean()

    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=7)
    ma_long = calculate_moving_average(close, window=21)
    momentum = calculate_momentum(close, period=5)
    avg_vol = calculate_average_volume(volume, window=10)

    daily_return = close / close.shift(1) - 1
    heuristics_matrix = (ma_short - ma_long) * momentum * (volume / avg_vol) * daily_return

    return heuristics_matrix
