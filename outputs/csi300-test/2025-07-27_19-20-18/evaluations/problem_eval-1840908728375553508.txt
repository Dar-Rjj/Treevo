import pandas as pd

def heuristics_v2(df):
    # Calculate daily percentage change in close prices
    daily_change = df['close'].pct_change().fillna(0)
    
    # Calculate 5-day moving average of daily changes
    ma_daily_change = daily_change.rolling(window=5).mean()
    
    # Calculate 10-day moving average of volume
    ma_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the trend in volume as the difference between current volume and 10-day MA
    volume_trend = df['volume'] - ma_volume
    
    # Combine the moving average of daily changes with the volume trend
    heuristics_matrix = (ma_daily_change + volume_trend) / 2
    
    return heuristics_matrix
