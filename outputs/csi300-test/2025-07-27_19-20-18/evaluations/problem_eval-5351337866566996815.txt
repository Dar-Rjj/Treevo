import pandas as pd

def heuristics_v2(df):
    close = df['close']
    amount = df['amount']
    ema_amount_20 = amount.ewm(span=20, adjust=False).mean()
    
    roc_close_5 = (close / close.shift(5) - 1)
    amount_ratio = amount / ema_amount_20
    
    heuristics_matrix = roc_close_5 * amount_ratio
    return heuristics_matrix
