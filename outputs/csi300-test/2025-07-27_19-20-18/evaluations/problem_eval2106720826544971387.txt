import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=14):
        return series.rolling(window=window).mean()

    def calculate_momentum(series, period=7):
        return series / series.shift(period) - 1

    def calculate_volatility(series, window=15):
        return series.rolling(window=window).std()

    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=7)
    ma_long = calculate_moving_average(close, window=20)
    momentum = calculate_momentum(close, period=7)
    vol = calculate_volatility(close, window=15)

    heuristics_matrix = (ma_short - ma_long) * momentum / vol * (volume / volume.shift(1))

    return heuristics_matrix
