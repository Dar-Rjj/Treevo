import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=14):
        return series.rolling(window=window).mean()

    def calculate_momentum(series, period=10):
        return series / series.shift(period) - 1

    def calculate_volatility(series, window=20):
        return series.rolling(window=window).std()

    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=7)
    ma_long = calculate_moving_average(close, window=50)
    momentum = calculate_momentum(close, period=15)
    vol = calculate_volatility(close, window=25)

    price_diff = (df['high'] - df['low']) / df['close']
    
    heuristics_matrix = (ma_short - ma_long) * momentum / vol * (price_diff) * (volume / volume.shift(1))

    return heuristics_matrix
