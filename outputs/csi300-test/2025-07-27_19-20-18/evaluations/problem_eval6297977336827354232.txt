import pandas as pd

def heuristics_v2(df):
    # Calculate daily percentage change in close prices
    daily_change = df['close'].pct_change().fillna(0)
    
    # Calculate 5-day exponential moving average of daily changes
    ema_daily_change = daily_change.ewm(span=5, adjust=False).mean()
    
    # Calculate 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate the volume momentum as the ratio of current volume to 10-day EMA
    volume_momentum = df['volume'] / ema_volume
    
    # Combine the EMA of daily changes with the volume momentum
    heuristics_matrix = (ema_daily_change + volume_momentum) / 2
    
    return heuristics_matrix
