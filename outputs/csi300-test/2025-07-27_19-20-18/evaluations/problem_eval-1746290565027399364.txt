def heuristics_v2(df):
    high = df['high']
    low = df['low']
    close = df['close']
    
    # Predicted close price using a simple average of high and low
    predicted_close = (high + low) / 2
    
    # Calculate the difference between actual and predicted close prices
    surprise = close - predicted_close
    
    # Heuristic matrix is the surprise factor
    heuristics_matrix = surprise
    
    return heuristics_matrix
