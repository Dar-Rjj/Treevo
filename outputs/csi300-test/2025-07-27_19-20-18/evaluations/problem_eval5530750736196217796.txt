import pandas as pd

def heuristics_v2(df):
    avg_high_low = (df['high'] + df['low']).rolling(window=10).mean()
    close_to_avg_ratio = df['close'] / avg_high_low
    amount_change = df['amount'].pct_change(periods=10)
    heuristics_matrix = close_to_avg_ratio * amount_change
    return heuristics_matrix
