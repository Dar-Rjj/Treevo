import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=14):
        return series.rolling(window=window).mean()

    def calculate_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def calculate_volume_ratio(series, short_window=5, long_window=30):
        short_ma = calculate_moving_average(series, window=short_window)
        long_ma = calculate_moving_average(series, window=long_window)
        return series / long_ma

    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=5)
    ma_long = calculate_moving_average(close, window=30)
    rsi = calculate_rsi(close, window=14)
    vol_ratio = calculate_volume_ratio(volume)

    heuristics_matrix = (ma_short - ma_long) * rsi * vol_ratio

    return heuristics_matrix
