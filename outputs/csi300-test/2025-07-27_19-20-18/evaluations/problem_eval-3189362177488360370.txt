import pandas as pd

def heuristics_v2(df):
    def calculate_exponential_moving_average(series, span=10):
        return series.ewm(span=span, adjust=False).mean()

    def calculate_rsi(series, window=14):
        delta = series.diff()
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window).mean()
        roll_down = down.abs().rolling(window).mean()
        rs = roll_up / roll_down
        rsi = 100.0 - (100.0 / (1.0 + rs))
        return rsi

    close = df['close']
    volume = df['volume']

    ema_short = calculate_exponential_moving_average(close, span=5)
    ema_long = calculate_exponential_moving_average(close, span=30)
    rsi = calculate_rsi(close, window=14)

    heuristics_matrix = ((ema_short / ema_long) * rsi) * (volume / volume.shift(1))

    return heuristics_matrix
