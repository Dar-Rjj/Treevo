import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=14):
        return series.rolling(window=window).mean()

    close = df['close']
    high = df['high']
    low = df['low']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=5)
    ma_long = calculate_moving_average(close, window=30)
    
    high_low_ratio = high / low
    volume_change = volume / volume.shift(1) - 1

    heuristics_matrix = (ma_short - ma_long) * high_low_ratio * volume_change

    return heuristics_matrix
