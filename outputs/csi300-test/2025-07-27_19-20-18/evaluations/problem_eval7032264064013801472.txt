def heuristics_v2(df):
    short_term_return = df['close'].pct_change(periods=5)
    long_term_return = df['close'].pct_change(periods=30)
    volume_change = df['volume'] / df['volume'].shift(1)

    heuristics_matrix = (short_term_return / long_term_return) * volume_change

    return heuristics_matrix
