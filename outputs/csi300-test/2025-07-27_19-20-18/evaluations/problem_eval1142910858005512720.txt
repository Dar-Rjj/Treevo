import pandas as pd
import numpy as np

def heuristics_v2(df):
    sma_close = df['close'].rolling(window=30).mean()
    sma_volume = df['volume'].rolling(window=30).mean()
    volume_ratio_sqrt = (df['volume'] / sma_volume).apply(lambda x: max(0.0001, x)).apply(np.sqrt)
    heuristics_matrix = sma_close * volume_ratio_sqrt
    return heuristics_matrix
