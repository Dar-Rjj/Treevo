import pandas as pd

def heuristics_v2(df):
    def calculate_rsi(series, window=14):
        delta = series.diff(1)
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    def calculate_volume_weighted_ma(series, volume, window=20):
        return (series * volume).rolling(window=window).sum() / volume.rolling(window=window).sum()

    def calculate_roc(series, period=10):
        return (series / series.shift(period) - 1) * 100

    close = df['close']
    volume = df['volume']

    rsi = calculate_rsi(close, window=14)
    vwma = calculate_volume_weighted_ma(close, volume, window=20)
    roc = calculate_roc(close, period=10)

    heuristics_matrix = (rsi - 50) * (vwma - close) * roc

    return heuristics_matrix
