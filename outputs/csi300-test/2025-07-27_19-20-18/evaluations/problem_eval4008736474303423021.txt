import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=14):
        return series.rolling(window=window).mean()

    def calculate_rsi(series, period=14):
        delta = series.diff()
        gain = delta.where(delta > 0, 0)
        loss = -delta.where(delta < 0, 0)
        avg_gain = gain.rolling(window=period).mean()
        avg_loss = loss.rolling(window=period).mean()
        rs = avg_gain / avg_loss
        rsi = 100 - (100 / (1 + rs))
        return rsi

    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=5)
    ma_long = calculate_moving_average(close, window=30)
    rsi = calculate_rsi(close, period=14)
    vol_change = (volume / volume.shift(1)) - 1

    heuristics_matrix = (ma_short / ma_long) * rsi * vol_change

    return heuristics_matrix
