import pandas as pd
import numpy as np

def heuristics_v2(df):
    ema_close_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_amount = df['amount'].ewm(span=10, adjust=False).mean()
    amount_ratio_sqrt = ((df['amount'] / ema_amount).apply(lambda x: max(0.0001, x))).apply(np.sqrt)
    heuristics_matrix = (ema_close_5 - ema_close_10) * amount_ratio_sqrt
    return heuristics_matrix
