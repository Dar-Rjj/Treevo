import pandas as pd

def calculate_rsi(series, window=14):
    delta = series.diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
    RS = gain / loss
    RSI = 100 - (100 / (1 + RS))
    return RSI

def heuristics_v2(df):
    amount = df['amount']
    rsi = calculate_rsi(amount)
    roc_amount = amount.pct_change(periods=10)
    heuristics_matrix = roc_amount * rsi
    return heuristics_matrix
