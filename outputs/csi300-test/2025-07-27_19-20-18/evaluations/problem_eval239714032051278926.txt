import pandas as pd

def calculate_moving_average(series, window=14):
    return series.rolling(window=window).mean()

def calculate_rate_of_change(series, period=10):
    return (series - series.shift(period)) / series.shift(period)

def calculate_rsi(series, window=14):
    delta = series.diff(1)
    up, down = delta.copy(), delta.copy()
    up[up < 0] = 0
    down[down > 0] = 0
    average_gain = up.rolling(window=window).mean()
    average_loss = abs(down.rolling(window=window).mean())
    rs = average_gain / average_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=5)
    ma_long = calculate_moving_average(close, window=30)
    roc = calculate_rate_of_change(close, period=10)
    rsi = calculate_rsi(close, window=14)

    ma_ratio = ma_short / ma_long
    vol_adjustment = volume / volume.shift(1)

    heuristics_matrix = (roc * ma_ratio * rsi) * vol_adjustment

    return heuristics_matrix
