import pandas as pd

def heuristics_v2(df):
    high = df['high']
    low = df['low']
    close = df['close']
    volume = df['volume']

    range_10d = (high.rolling(window=10).max() - low.rolling(window=10).min()) / close.rolling(window=10).mean()
    ema_volume = volume.ewm(span=10, adjust=False).mean()

    heuristics_matrix = range_10d * ema_volume

    return heuristics_matrix
