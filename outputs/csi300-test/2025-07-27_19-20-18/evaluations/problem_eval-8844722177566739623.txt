import pandas as pd

def heuristics_v2(df):
    # Calculate 20-day EMA of close prices
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate 10-day EMA of open prices
    ema_open = df['open'].ewm(span=10, adjust=False).mean()
    
    # Calculate the price trend as the difference between 20-day EMA of close and 10-day EMA of open
    price_trend = ema_close - ema_open
    
    # Calculate 5-day SMA of volume
    sma_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the volume trend as the ratio of current day's volume to 5-day SMA of volume
    volume_trend = df['volume'] / sma_volume
    
    # Combine the price trend with the volume trend
    heuristics_matrix = price_trend * volume_trend
    
    return heuristics_matrix
