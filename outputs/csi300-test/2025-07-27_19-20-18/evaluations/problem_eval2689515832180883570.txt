import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=7):
        return series.rolling(window=window).mean()

    def calculate_momentum(series, period=14):
        return series / series.shift(period) - 1

    def calculate_volatility(series, window=10):
        return series.rolling(window=window).std()

    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=7)
    ma_long = calculate_moving_average(close, window=21)
    momentum = calculate_momentum(close, period=14)
    vol = calculate_volatility(close, window=10)

    heuristics_matrix = (ma_short - ma_long) * momentum / (vol + 1e-6) * (volume / (volume.shift(1) + 1e-6))

    return heuristics_matrix
