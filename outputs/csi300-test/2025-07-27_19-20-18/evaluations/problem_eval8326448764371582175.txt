import pandas as pd
import numpy as np

def heuristics_v2(df):
    log_returns = np.log(df['close'] / df['close'].shift(1))
    avg_log_returns_5 = log_returns.rolling(window=5).mean()
    rsi = 100 - (100 / (1 + (df['close'].diff(1).clip(lower=0).rolling(window=14).mean() / df['close'].diff(1).clip(upper=0).rolling(window=14).mean().abs())))
    heuristics_matrix = avg_log_returns_5 * rsi
    return heuristics_matrix
