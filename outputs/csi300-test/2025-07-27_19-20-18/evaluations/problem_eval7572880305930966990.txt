import pandas as pd

def heuristics_v2(df):
    recent_high = df['high'].rolling(window=10).max()
    recent_low = df['low'].rolling(window=10).min()
    avg_volume_long = df['volume'].rolling(window=30).mean()
    volume_change = df['volume'] / avg_volume_long - 1
    heuristics_matrix = (df['close'] - recent_low) / (recent_high - recent_low) * volume_change
    return heuristics_matrix
