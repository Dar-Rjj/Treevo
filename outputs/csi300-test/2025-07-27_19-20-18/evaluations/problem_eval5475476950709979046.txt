import pandas as pd

def calculate_moving_average(series, window=14):
    return series.rolling(window=window).mean()

def calculate_momentum(series, period=10):
    return series / series.shift(period) - 1

def calculate_volatility(series, window=20):
    return series.rolling(window=window).std()

def calculate_rsi(series, window=14):
    delta = series.diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=window, min_periods=1).mean()
    avg_loss = loss.rolling(window=window, min_periods=1).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

def heuristics_v2(df):
    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=7)
    ma_long = calculate_moving_average(close, window=21)
    momentum = calculate_momentum(close, period=7)
    vol = calculate_volatility(close, window=14)
    rsi = calculate_rsi(close, window=14)

    heuristics_matrix = (ma_short - ma_long) * momentum / vol * rsi * (volume / volume.shift(1))

    return heuristics_matrix
