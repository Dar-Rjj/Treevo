import pandas as pd

def heuristics_v2(df):
    # Calculate the difference between high and low for each day
    diff = df['high'] - df['low']
    
    # Calculate the weighted sum using volume as weights for the last 10 days
    weighted_diff = (diff * df['volume']).rolling(window=10).sum()
    
    # Calculate the total volume over the same window to normalize the weight
    total_volume = df['volume'].rolling(window=10).sum()
    
    # Final heuristic factor
    heuristics_matrix = (weighted_diff / total_volume).dropna()
    
    return heuristics_matrix
