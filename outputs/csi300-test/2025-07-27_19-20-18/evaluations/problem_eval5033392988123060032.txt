import pandas as pd

def heuristics_v2(df):
    def calculate_percentage_change(series, period=10):
        return (series - series.shift(period)) / series.shift(period)

    def calculate_vwap(high, low, close, volume):
        typical_price = (high + low + close) / 3
        return (typical_price * volume).cumsum() / volume.cumsum()

    def calculate_rsi(series, window=14):
        delta = series.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))

    close = df['close']
    high = df['high']
    low = df['low']
    volume = df['volume']

    pct_change = calculate_percentage_change(close, period=10)
    vwap = calculate_vwap(high, low, close, volume)
    rsi = calculate_rsi(close, window=14)

    heuristics_matrix = pct_change * (vwap / close) * (100 - rsi)

    return heuristics_matrix
