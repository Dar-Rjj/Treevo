import pandas as pd

def heuristics_v2(df):
    def calculate_relative_strength(series_high, series_low, window=14):
        return (series_high - series_low).rolling(window=window).mean()

    def calculate_volume_change(volume, period=5):
        return volume / volume.shift(period) - 1

    high = df['high']
    low = df['low']
    volume = df['volume']

    rel_strength = calculate_relative_strength(high, low, window=14)
    vol_change = calculate_volume_change(volume, period=5)

    heuristics_matrix = rel_strength * vol_change

    return heuristics_matrix
