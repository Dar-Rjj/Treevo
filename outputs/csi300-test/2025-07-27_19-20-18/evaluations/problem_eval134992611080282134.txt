import pandas as pd

def heuristics_v2(df):
    close_high_ratio = df['close'] / df['high']
    close_high_roc = close_high_ratio / close_high_ratio.shift(10) - 1
    volume_pct_change = df['volume'].pct_change()
    heuristics_matrix = close_high_roc * volume_pct_change
    return heuristics_matrix
