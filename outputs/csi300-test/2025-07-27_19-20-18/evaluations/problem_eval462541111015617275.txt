import pandas as pd

def heuristics_v2(df):
    def calculate_moving_average(series, window=14):
        return series.rolling(window=window).mean()

    def calculate_rsi(series, window=14):
        delta = series.diff(1)
        up, down = delta.copy(), delta.copy()
        up[up < 0] = 0
        down[down > 0] = 0
        roll_up = up.rolling(window).mean()
        roll_down = down.abs().rolling(window).mean()
        rs = roll_up / roll_down
        return 100 - (100 / (1 + rs))

    close = df['close']
    volume = df['volume']

    ma_short = calculate_moving_average(close, window=5)
    ma_long = calculate_moving_average(close, window=30)
    rsi = calculate_rsi(close, window=14)
    volume_change = volume / volume.shift(1)

    heuristics_matrix = ((ma_short - ma_long) * rsi * volume_change)

    return heuristics_matrix
