import pandas as pd
import numpy as np

def heuristics_v2(df):
    ema_close = df['close'].ewm(span=15, adjust=False).mean()
    ema_amount = df['amount'].ewm(span=15, adjust=False).mean()
    amount_ratio_sqrt = (df['amount'] / ema_amount).apply(lambda x: max(0.0001, x)).apply(np.sqrt)
    heuristics_matrix = ema_close * amount_ratio_sqrt
    return heuristics_matrix
