def heuristics_v2(df):
    returns = df['close'].pct_change(periods=10)
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    volume_ratio = avg_volume_10 / avg_volume_30
    heuristics_matrix = returns * volume_ratio
    return heuristics_matrix
