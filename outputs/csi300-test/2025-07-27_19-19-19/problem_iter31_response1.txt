```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period (e.g., last 20 days)
    N = 20
    avg_price_change = daily_price_change.rolling(window=N).mean()
    
    # Calculate Volume Adjusted Close-Open Difference
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_close_open = (df['close'] - df['open']) / avg_volume_20
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_va = (high_low_range * volume_adjusted_close_open).rolling(window=10).sum()
    
    # Calculate Volume-Weighted Close Price
    volume_weighted_close = (df['volume'] * df['close']) / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    scaling_high_low = 1.2
    high_low_momentum_score = combined_hl_va * scaling_high_low
    
    scaling_volume_weighted = 0.6
    volume_weighted_score = volume_weighted_close * scaling_volume_weighted
    
    scaling_momentum = 1.8
    momentum_score = avg_price_change * scaling_momentum
    
    # Integrate Indicators
    # Short-Term Price Direction (SPDI)
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    
    # Medium-Term Price Range (MPR)
    moving_high_20 = df['high'].rolling(window=20).max()
    moving_low_20 = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - moving_low_20) / (moving_high_20 - moving_low_20) > 0.5).astype(int)
    
    # Volume Growth (VGF)
    vgf = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    
    # Integrated Momentum, Volume, and Trend Indicators (IMVTI)
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume + imvti
    
    # Apply Smoothing
    alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.4, adjust=False).mean()
    
    return alpha_factor_smoothed
```
