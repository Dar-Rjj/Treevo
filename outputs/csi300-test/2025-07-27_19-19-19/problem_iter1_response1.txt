```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 10-Day Price Momentum
    sma_10 = df['close'].rolling(window=10).mean()
    price_momentum_10 = df['close'] - sma_10
    
    # Calculate 30-Day Price Momentum
    sma_30 = df['close'].rolling(window=30).mean()
    price_momentum_30 = df['close'] - sma_30
    
    # Calculate Price Momentum Score
    price_momentum_score = price_momentum_10 + price_momentum_30
    
    # Calculate 10-Day and 30-Day Volume Moving Averages
    volume_sma_10 = df['volume'].rolling(window=10).mean()
    volume_sma_30 = df['volume'].rolling(window=30).mean()
    
    # Calculate Volume Momentum
    volume_momentum = volume_sma_10 - volume_sma_30
    
    # Final Factor
    final_factor = price_momentum_score * volume_momentum
    
    return final_factor
```
