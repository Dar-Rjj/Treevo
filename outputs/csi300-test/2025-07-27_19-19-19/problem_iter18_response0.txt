```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Combined Change
    df['Close_Price_Diff'] = df['close'].diff()
    df['Volume_Change'] = df['volume'].diff()
    df['Combined_Daily_Change'] = df['Close_Price_Diff'] + df['Volume_Change']

    # Aggregate Changes Over a Period (N=5)
    N = 5
    df['Sum_Combined_Change'] = df['Combined_Daily_Change'].rolling(window=N).sum()
    df['Average_Daily_Change'] = df['Sum_Combined_Change'] / N

    # Compute Short-Term Price Direction (SPDI)
    df['5_day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = np.where(df['close'] > df['5_day_MA_Close'], 1, 0)

    # Compute Medium-Term Price Range (MPR)
    df['20_day_High'] = df['high'].rolling(window=20).max()
    df['20_day_Low'] = df['low'].rolling(window=20).min()
    df['20_day_Midpoint'] = (df['20_day_High'] + df['20_day_Low']) / 2
    df['MPR'] = np.where(df['close'] > df['20_day_Midpoint'], 1, 0)

    # Calculate Volume Growth Factor (VGF)
    df['3_day_MA_Volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = np.where(df['volume'] > df['3_day_MA_Volume'], 1, 0)

    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3

    # Combine Price and Volume Factors
    df['Momentum_Score'] = df['Average_Daily_Change'] * 1.5
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['Volume_Sum_N'] = df['volume'].rolling(window=N).sum()
    df['Volume_Weighted_Price'] = (df['Volume_Price_Product'] / df['Volume_Sum_N']) * 0.5

    # Compute Long-Term Trend Indicator (LTTI)
    df['60_day_MA_Close'] = df['close'].rolling(window=60).mean()
    df['LTTI'] = np.where(df['close'] > df['60_day_MA_Close'], 1, 0)

    # Compute Volatility Factor (VF)
    df['20_day_STD_Close'] = df['close'].rolling(window=20).std()
    df['VF'] = df['20_day_STD_Close'] / df['close']

    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Momentum_Score'] + df['Volume_Weighted_Price'] + df['IMVTI'] + df['LTTI'] - df['VF']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.3).mean()

    return df['Final_Alpha_Factor']
```
