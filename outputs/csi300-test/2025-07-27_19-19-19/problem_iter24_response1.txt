```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Adjusted_Close_Open'] = (df['Close_Open_Diff'] * df['volume']) / df['Avg_Volume_10']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    N = 5
    df['Aggregate_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum() / N
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_HL_VACO'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']
    df['Combined_HL_VACO_Sum'] = df['Combined_HL_VACO'].rolling(window=5).sum()
    
    # Calculate Short-Term Price Direction (SPDI)
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA']).astype(int)
    
    # Calculate Medium-Term Price Range (MPR)
    df['20_day_High'] = df['high'].rolling(window=20).max()
    df['20_day_Low'] = df['low'].rolling(window=20).min()
    df['20_day_Range_Mid'] = (df['20_day_High'] + df['20_day_Low']) / 2
    df['MPR'] = (df['close'] > df['20_day_Range_Mid']).astype(int)
    
    # Calculate Volume Growth (VGF)
    df['3_day_Avg_Volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_Avg_Volume']).astype(int)
    
    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Generate Intermediate Alpha Scores
    HL_Momentum_Score = df['Combined_HL_VACO_Sum'] * 1.0
    Momentum_Score = df['Aggregate_Price_Change'] * 1.5
    
    # Calculate Volume-Weighted Close Price (VWCP)
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['VWCP'] = df['Volume_Price_Product'].rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    Volume_Weighted_Score = df['VWCP'] * 0.5
    
    # Combine Price and Volume Factors
    Combined_Momentum_Volume_Score = Momentum_Score + Volume_Weighted_Score
    
    # Generate Final Alpha Factor
    Final_Alpha_Factor = HL_Momentum_Score + Combined_Momentum_Volume_Score + df['IMVTI']
    Final_Alpha_Factor = Final_Alpha_Factor.ewm(span=5, adjust=False).mean()
    
    return Final_Alpha_Factor
```
