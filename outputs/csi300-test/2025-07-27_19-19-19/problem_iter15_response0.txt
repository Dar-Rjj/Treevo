```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=5, scaling_high_low=1.0, scaling_volume_weighted=0.5, scaling_momentum=1.5, scaling_acceleration=0.8, smoothing_factor=0.3):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    daily_price_change_percentage = daily_price_change / df['close'].shift(1)
    
    # Aggregate Price Change Over a Period
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open = (close_open_diff * df['volume']) / avg_volume_10
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_high_low = (high_low_range * volume_adjusted_close_open).rolling(window=M).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close = volume_price_product / df['volume'].rolling(window=N).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low * scaling_high_low
    volume_weighted_score = volume_weighted_close * scaling_volume_weighted
    
    # Incorporate Price Acceleration
    price_acceleration = daily_price_change.diff()
    scaled_price_acceleration = price_acceleration * scaling_acceleration
    
    # Combine Price and Volume Factors
    momentum_score = average_price_change * scaling_momentum
    combined_scores = momentum_score + volume_weighted_score + scaled_price_acceleration
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_scores
    final_alpha_factor = final_alpha_factor.ewm(alpha=smoothing_factor).mean()
    
    return final_alpha_factor
```
