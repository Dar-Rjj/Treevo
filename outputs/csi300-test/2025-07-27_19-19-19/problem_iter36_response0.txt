import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (N=20)
    df['Avg_Price_Change'] = df['Daily_Price_Change'].rolling(window=20).mean()
    
    # Calculate Volume Adjusted Close-Open Difference
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    df['Volume_Adjusted_Close_Open'] = (df['close'] - df['open']) / avg_volume_30
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_HL_VACO'] = (df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']).rolling(window=15).sum()
    
    # Calculate Volume Weighted Average
    total_volume_30 = df['volume'].rolling(window=30).sum()
    df['Volume_Weighted_Close'] = (df['volume'] * df['close']) / total_volume_30
    
    # Generate Intermediate Alpha Scores
    df['HL_Momentum_Score'] = df['Combined_HL_VACO'] * 1.5
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Close'] * 0.7
    df['Momentum_Score'] = df['Avg_Price_Change'] * 1.9
    
    # Integrate Trend Indicators
    df['SPDI'] = df['close'].rolling(window=5).mean().apply(lambda x: 1 if x > df['close'] else -1)
    moving_high_20 = df['high'].rolling(window=20).max()
    moving_low_20 = df['low'].rolling(window=20).min()
    df['MPR'] = ((df['close'] - moving_low_20) / (moving_high_20 - moving_low_20)).apply(lambda x: 1 if x > 0.5 else -1)
    df['VGF'] = df['volume'].gt(df['volume'].rolling(window=10).mean()).astype(int).replace({True: 1, False: -1})
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['Combined_MV_Scores'] = df['Momentum_Score'] + df['Volume_Weighted_Score']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['HL_Momentum_Score'] + df['Combined_MV_Scores'] + df['IMVTI']
    
    # Apply Smoothing
    df['Smoothed_Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.5).mean()
    
    return df['Smoothed_Final_Alpha_Factor']
