```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (e.g., 5, 10, 20 days)
    df['Avg_Price_Change'] = df['Daily_Price_Change'].rolling(window=5).mean()
    
    # Volume Adjusted Close-Open Difference
    df['Volume_Adjusted_CO_Diff'] = (df['close'] - df['open']) / df['volume'].rolling(window=20).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_High_Low_Vol'] = (df['High_Low_Range'] * df['Volume_Adjusted_CO_Diff']).rolling(window=10).sum()
    
    # Volume Weighted Average
    df['Volume_Weighted_Close'] = (df['volume'] * df['close']) / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    df['High_Low_Momentum_Score'] = df['Combined_High_Low_Vol'] * 1.2
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Close'] * 0.6
    df['Momentum_Score'] = df['Avg_Price_Change'] * 1.8
    
    # Integrate Trend Indicators
    df['SPDI'] = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    df['MPR'] = ((df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / 2 < df['close']).astype(int)
    df['VGF'] = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['Combined_Mom_Vol_Scores'] = df['Momentum_Score'] + df['Volume_Weighted_Score']
    
    # Generate Final Alpha Factor
    df['Intermediate_Alpha_Factor'] = df['High_Low_Momentum_Score'] + df['Combined_Mom_Vol_Scores']
    df['Final_Alpha_Factor'] = df['Intermediate_Alpha_Factor'] + df['IMVTI']
    
    # Apply Smoothing
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.4, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
```
