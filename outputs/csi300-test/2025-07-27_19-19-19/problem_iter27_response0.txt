```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Adjusted_Close_Open'] = df['Close_Open_Diff'] * (df['volume'] / df['Avg_Volume_10'])

    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'] - df['close'].shift(1)

    # Aggregate Price Change Over a Period
    N = 5
    df['Aggregate_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum() / N

    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_HL_Range_VA_CO'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']
    df['Combined_HL_Range_VA_CO_Sum'] = df['Combined_HL_Range_VA_CO'].rolling(window=5).sum()

    # Calculate Short-Term Price Direction (SPDI)
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA']).astype(int)

    # Calculate Medium-Term Price Range (MPR)
    df['20_day_MH'] = df['high'].rolling(window=20).max()
    df['20_day_ML'] = df['low'].rolling(window=20).min()
    df['Mid_Point'] = (df['20_day_MH'] + df['20_day_ML']) / 2
    df['MPR'] = (df['close'] > df['Mid_Point']).astype(int)

    # Calculate Volume Growth (VGF)
    df['3_day_MA_Vol'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_MA_Vol']).astype(int)

    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3

    # Generate Intermediate Alpha Scores
    scaling_factor_hl_momentum = 1.2
    scaling_factor_momentum = 1.7
    scaling_factor_volume_weighted = 0.6

    df['HL_Momentum_Score'] = df['Combined_HL_Range_VA_CO_Sum'] * scaling_factor_hl_momentum
    df['Momentum_Score'] = df['Aggregate_Price_Change'] * scaling_factor_momentum

    # Calculate Volume-Weighted Close Price (VWCP)
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['Sum_Volumes_N'] = df['volume'].rolling(window=N).sum()
    df['VWCP'] = df['Volume_Price_Product'] / df['Sum_Volumes_N']

    # Calculate Volume Weighted Score
    df['Volume_Weighted_Score'] = df['VWCP'] * scaling_factor_volume_weighted

    # Combine Price and Volume Factors
    df['Combined_Momentum_Volume_Score'] = df['Momentum_Score'] + df['Volume_Weighted_Score']

    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['HL_Momentum_Score'] + df['Combined_Momentum_Volume_Score'] + df['IMVTI']

    # Apply Smoothing
    smoothing_factor = 0.45
    df['Smoothed_Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=smoothing_factor).mean()

    return df['Smoothed_Final_Alpha_Factor'].dropna()
```
