import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open_diff = (close_open_diff * df['volume']) / avg_volume
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    daily_price_change_percentage = daily_price_change / df['close'].shift(1)
    
    # Aggregate Price Change Over a Period (N=5 for example)
    N = 5
    accumulated_price_change = daily_price_change_percentage.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_score = high_low_range * volume_adjusted_close_open_diff
    combined_5_day_score = combined_score.rolling(window=5).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close_price = volume_price_product / df['volume'].rolling(window=N).sum()
    
    # Generate Intermediate Alpha Scores
    scaling_factor_high_low = 1.0
    high_low_momentum_score = combined_5_day_score * scaling_factor_high_low
    
    scaling_factor_volume_weighted = 0.5
    volume_weighted_score = volume_weighted_close_price * scaling_factor_volume_weighted
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + volume_weighted_score
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.3, adjust=False).mean()
    
    return final_alpha_factor
