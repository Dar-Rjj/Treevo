import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume-Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open = close_open_diff * (df['volume'] / avg_volume_10)
    
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    daily_price_change_percentage = daily_price_change / df['close'].shift(1)
    
    # Aggregate Price Change Over a Period
    sum_daily_price_changes = daily_price_change.rolling(window=5).sum()
    average_price_change = sum_daily_price_changes / 5
    
    # Combine High-Low Range and Volume-Adjusted Close-Open Difference
    combined_high_low_vol_adj = high_low_range * volume_adjusted_close_open
    combined_high_low_vol_adj_5 = combined_high_low_vol_adj.rolling(window=5).sum()
    
    # Compute Short-Term Price Direction (SPDI)
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    
    # Compute Medium-Term Price Range (MPR)
    moving_high_20 = df['high'].rolling(window=20).max()
    moving_low_20 = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - moving_low_20) / (moving_high_20 - moving_low_20) > 0.5).astype(int)
    
    # Calculate Volume Growth (VGF)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    
    # Integrate Indicators
    imvti = (spdi + mpr + vgf) / 3
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low_vol_adj_5 * 1.0
    momentum_score = average_price_change * 1.5
    volume_weighted_price = (df['volume'] * df['close']) / df['volume'].rolling(window=5).sum()
    
    # Incorporate Trading Volume Ratio (TVR)
    tvr = (df['volume'] > df['volume'].rolling(window=10).mean()).astype(int)
    
    # Combine Price and Volume Factors
    combined_momentum_volume = momentum_score + volume_weighted_price
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume + (tvr * 0.2)
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.3).mean()
    
    return final_alpha_factor
