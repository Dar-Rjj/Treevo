```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Compute Daily Combined Change
    df['Close_Price_Diff'] = df['close'].diff()
    df['Volume_Change'] = df['volume'].diff()
    df['Combined_Daily_Change'] = df['Close_Price_Diff'] + df['Volume_Change']
    
    # Aggregate Changes Over a Period
    df['Sum_Combined_Change'] = df['Combined_Daily_Change'].rolling(window=N).sum()
    df['Avg_Daily_Change'] = df['Sum_Combined_Change'] / N
    
    # Compute Short-Term Price Direction (SPDI)
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA']).astype(int)
    
    # Compute Medium-Term Price Range (MPR)
    df['20_day_High'] = df['high'].rolling(window=20).max()
    df['20_day_Low'] = df['low'].rolling(window=20).min()
    df['Range_Midpoint'] = (df['20_day_High'] + df['20_day_Low']) / 2
    df['MPR'] = (df['close'] > df['Range_Midpoint']).astype(int)
    
    # Calculate Volume Growth (VGF)
    df['3_day_Avg_Volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_Avg_Volume']).astype(int)
    
    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['Momentum_Score'] = df['Avg_Daily_Change'] * 1.5
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['Volume_Sum'] = df['volume'].rolling(window=N).sum()
    df['Volume_Weighted_Price'] = df['Volume_Price_Product'] / df['Volume_Sum']
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Price'] * 0.5
    
    # Compute Long-Term Trend Indicator (LTTI)
    df['60_day_MA'] = df['close'].rolling(window=60).mean()
    df['LTTI'] = (df['close'] > df['60_day_MA']).astype(int)
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Momentum_Score'] + df['Volume_Weighted_Score'] + df['IMVTI'] + df['LTTI']
    
    # Apply Smoothing
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.3, adjust=False).mean()
    
    return df['Final_Alpha_Factor']
```
