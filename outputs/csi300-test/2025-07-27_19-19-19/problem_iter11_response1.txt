```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Compute Daily Price and Volume Changes
    df['CloseDiff'] = df['close'].diff()
    df['VolumeChange'] = df['volume'].diff()
    df['CombinedDailyChange'] = df['CloseDiff'] + df['VolumeChange']
    
    # Aggregate Changes Over a Period (N days)
    df['SumCombinedDailyChanges'] = df['CombinedDailyChange'].rolling(window=N).sum()
    df['AverageDailyChange'] = df['SumCombinedDailyChanges'] / N
    
    # Calculate Short-Term Price Direction (SPDI)
    df['5DayMA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5DayMA']).astype(int)
    
    # Compute Medium-Term Price Range (MPR)
    df['20DayHigh'] = df['close'].rolling(window=20).max()
    df['20DayLow'] = df['close'].rolling(window=20).min()
    df['MidRange'] = (df['20DayHigh'] + df['20DayLow']) / 2
    df['MPR'] = (df['close'] > df['MidRange']).astype(int)
    
    # Calculate Volume Growth (VGF)
    df['3DayVolumeMA'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3DayVolumeMA']).astype(int)
    
    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Calculate Volume-Weighted Close Price
    df['VolumePriceProduct'] = df['volume'] * df['close']
    df['VolumeSum'] = df['volume'].rolling(window=N).sum()
    df['VolumeWeightedPrice'] = df['VolumePriceProduct'].rolling(window=N).sum() / df['VolumeSum']
    
    # Combine Price and Volume Factors
    scaling_factor = 1.0
    df['MomentumScore'] = df['AverageDailyChange'] * scaling_factor
    df['VolumeWeightedScore'] = df['VolumeWeightedPrice'] * scaling_factor
    
    # Generate Final Alpha Factor
    df['FinalAlphaFactor'] = df['IMVTI'] + df['MomentumScore'] + df['VolumeWeightedScore']
    
    # Apply Exponential Smoothing
    df['FinalAlphaFactor'] = df['FinalAlphaFactor'].ewm(alpha=0.3, adjust=False).mean()
    
    return df['FinalAlphaFactor']
```
