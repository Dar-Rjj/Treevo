import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_close_open_diff = close_open_diff * (df['volume'] / avg_volume)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_vol = (high_low_range * volume_adjusted_close_open_diff).rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close = volume_price_product.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_hl_vol * 1.2
    volume_weighted_score = volume_weighted_close * 0.6
    momentum_score = average_price_change * 1.8
    
    # Integrate Indicators
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    mpr_high = df['high'].rolling(window=20).max()
    mpr_low = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - mpr_low) > (mpr_high - mpr_low) / 2).astype(int)
    vgf = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume_scores = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_scores + imvti
    smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.4).mean()
    
    return smoothed_alpha_factor
