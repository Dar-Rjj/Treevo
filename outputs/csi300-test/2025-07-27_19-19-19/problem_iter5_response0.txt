import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Direction Indicator (SPDI)
    df['5_day_MA_close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA_close']).astype(int)

    # Calculate Medium-Term Price Direction Indicator (MPDI)
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['range_mid'] = (df['20_day_high'] + df['20_day_low']) / 2
    df['MPDI'] = (df['close'] > df['range_mid']).astype(int)

    # Integrate Short and Medium-Term Indicators
    df['integrated_indicator'] = (df['SPDI'] + df['MPDI']) / 2

    # Calculate 10-Day and 30-Day Price Change
    df['10_day_SMA'] = df['close'].rolling(window=10).mean()
    df['30_day_SMA'] = df['close'].rolling(window=30).mean()
    df['10_day_price_change'] = df['close'] - df['10_day_SMA']
    df['30_day_price_change'] = df['close'] - df['30_day_SMA']

    # Calculate Price Momentum Score
    df['price_momentum_score'] = df['10_day_price_change'] + df['30_day_price_change']

    # Calculate 10-Day and 30-Day Volume Moving Average
    df['10_day_vol_MA'] = df['volume'].rolling(window=10).mean()
    df['30_day_vol_MA'] = df['volume'].rolling(window=30).mean()

    # Calculate Volume Momentum
    df['volume_momentum'] = df['10_day_vol_MA'] - df['30_day_vol_MA']

    # Calculate Volume Growth Factor (VGF)
    df['3_day_vol_MA'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_vol_MA']).astype(int)

    # Combine Integrated Indicators and Volume Factors
    df['momentum_volatility_score'] = df['integrated_indicator'] * 1.5
    df['volume_weighted_score'] = (df['close'] * df['volume']) * 0.5

    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['momentum_volatility_score'] + df['volume_weighted_score']) * df['price_momentum_score'] * df['volume_momentum']

    # Apply Smoothing
    df['smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.3).mean()

    return df['smoothed_alpha_factor'].dropna()
