```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor: Volume Weighted Close-to-Open Return.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with the calculated factor values indexed by date.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
