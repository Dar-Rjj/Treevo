```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period (N days)
    accumulated_price_change = daily_price_change.rolling(N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(20).mean()
    volume_adjusted_close_open_diff = close_open_diff * (df['volume'] / avg_volume)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_vol = (high_low_range * volume_adjusted_close_open_diff).rolling(10).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close_price = volume_price_product.rolling(20).sum() / df['volume'].rolling(20).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_hl_vol * 1.2
    volume_weighted_score = volume_weighted_close_price * 0.6
    momentum_score = average_price_change * 1.8
    
    # Integrate Indicators
    spdi = (df['close'] > df['close'].rolling(5).mean()).astype(int)
    mpr = ((df['close'] > df['close'].rolling(20).max()) * 0.5 + (df['close'] < df['close'].rolling(20).min()) * 0.5).astype(int)
    vgf = (df['volume'] > df['volume'].rolling(5).mean()).astype(int)
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume + imvti
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.4).mean()
    
    return final_alpha_factor_smoothed
```
