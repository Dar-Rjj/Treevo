```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, momentum_scale=1.5, volume_scale=0.5, high_low_scale=0.3, smoothing_factor=0.3):
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    df['Daily_Price_Change_Percentage'] = df['Daily_Price_Change'] / df['close'].shift(1)
    
    # Aggregate Price Change Over a Period
    df['Accumulated_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum()
    df['Average_Price_Change'] = df['Accumulated_Price_Change'] / N
    
    # Calculate Volume Weighted Average
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['Volume_Sum'] = df['volume'].rolling(window=N).sum()
    df['Volume_Weighted_Close_Price'] = df['Volume_Price_Product'].rolling(window=N).sum() / df['Volume_Sum']
    
    # Integrate High and Low Prices
    df['High_Low_Range'] = df['high'] - df['low']
    df['Normalized_High_Low_Range'] = df['High_Low_Range'] / df['close'].shift(1)
    
    # Combine Price and Volume Factors
    df['Momentum_Score'] = df['Average_Price_Change'] * momentum_scale
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Close_Price'] * volume_scale
    df['High_Low_Score'] = df['Normalized_High_Low_Range'] * high_low_scale
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Momentum_Score'] + df['Volume_Weighted_Score'] + df['High_Low_Score']
    df['Smoothed_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=smoothing_factor).mean()
    
    return df['Smoothed_Alpha_Factor']
```
