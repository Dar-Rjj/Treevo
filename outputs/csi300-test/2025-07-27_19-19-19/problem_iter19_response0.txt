```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period (e.g., 5, 10, 20 days)
    N1, N2, N3 = 5, 10, 20
    avg_price_change_N1 = daily_price_change.rolling(window=N1).mean()
    avg_price_change_N2 = daily_price_change.rolling(window=N2).mean()
    avg_price_change_N3 = daily_price_change.rolling(window=N3).mean()
    
    # Calculate Volume Adjusted Close-Open Difference
    volume_adjusted_close_open_diff = (df['close'] - df['open']) / df['volume'].rolling(window=10).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference over the last 5 days
    combined_high_low_vol_adj = (high_low_range * volume_adjusted_close_open_diff).rolling(window=5).sum()
    
    # Calculate Volume Weighted Average
    volume_weighted_close = (df['volume'] * df['close']) / df['volume'].rolling(window=N1).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low_vol_adj * 1.0
    volume_weighted_score = volume_weighted_close * 0.5
    momentum_score = avg_price_change_N1 * 1.5
    
    # Integrate Trend Indicators
    spdi = (df['close'].rolling(window=5).mean() > df['close']).astype(int)
    mpr = ((df['high'].rolling(window=20).max() - df['low'].rolling(window=20).min()) / 2 + df['low'].rolling(window=20).min() < df['close']).astype(int)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume_scores = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_scores + imvti
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.3).mean()
    
    return final_alpha_factor_smoothed
```
