```python
import pandas as pd

def heuristics_v2(df, N=20, smoothing_factor=0.3):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    daily_price_change_pct = daily_price_change / df['close'].shift(1)
    
    # Aggregate Price Change Over a Period
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    avg_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open_diff = (close_open_diff * df['volume']) / avg_volume
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_high_low_vol = (high_low_range * volume_adjusted_close_open_diff).rolling(window=5).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close_price = volume_price_product.rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low_vol * 1.0
    volume_weighted_score = volume_weighted_close_price * 0.5
    momentum_score = avg_price_change * 1.5
    
    # Integrate Indicators
    short_term_price_direction = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    twenty_day_high = df['high'].rolling(window=20).max()
    twenty_day_low = df['low'].rolling(window=20).min()
    medium_term_price_range = ((df['close'] - twenty_day_low) / (twenty_day_high - twenty_day_low) > 0.5).astype(int)
    volume_growth = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    imvti = (short_term_price_direction + medium_term_price_range + volume_growth) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume_scores = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_scores + imvti
    final_alpha_factor = final_alpha_factor.ewm(alpha=smoothing_factor).mean()
    
    return final_alpha_factor
```
