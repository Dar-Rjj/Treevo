```python
import pandas as pd

def heuristics_v2(df):
    # Calculate 5-day moving average of close prices
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    
    # Short-Term Price Direction Indicator (SPDI)
    df['spdi'] = (df['close'] > df['5_day_ma_close']).astype(int)
    
    # Calculate 20-day moving high and low
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    
    # Medium-Term Price Direction Indicator (MPDI)
    mid_range = (df['20_day_high'] + df['20_day_low']) / 2
    df['mpdi'] = (df['close'] > mid_range).astype(int)
    
    # Volume Growth Factor (VGF)
    df['3_day_ma_volume'] = df['volume'].rolling(window=3).mean()
    df['vgf'] = (df['volume'] > df['3_day_ma_volume']).astype(int)
    
    # High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Volume Adjusted Close-Open Difference
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['vol_adj_close_open_diff'] = (df['close'] - df['open']) / df['10_day_avg_volume']
    
    # Combine Indicators
    combined_indicator = df['spdi'] * df['mpdi'] * df['vgf'] * df['vol_adj_close_open_diff']
    
    # Final Indicator Adjustment
    def final_adjustment(x):
        if x > 0.66:
            return 1
        elif x > 0.33:
            return 0.5
        else:
            return 0
    
    df['smppi'] = combined_indicator.apply(final_adjustment)
    
    return df['smppi']
```
