```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=10).mean()
    df['Volume_Adjusted_CO'] = (df['Close_Open_Diff'] * df['volume']) / avg_volume

    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    df['Daily_Price_Change_Pct'] = df['Daily_Price_Change'] / df['close'].shift(1)

    # Aggregate Price Change Over a Period
    N = 5
    df['Accumulated_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum()
    df['Average_Price_Change'] = df['Accumulated_Price_Change'] / N

    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    weighted_volume_adjusted_co = df['Volume_Adjusted_CO'].rolling(window=5).sum()
    df['Combined_HL_VA'] = df['High_Low_Range'] * weighted_volume_adjusted_co

    # Calculate Short-Term Price Direction (SPDI)
    df['5_day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA_Close']).astype(int)

    # Calculate Medium-Term Price Range (MPR)
    df['20_day_Moving_High'] = df['close'].rolling(window=20).max()
    df['20_day_Moving_Low'] = df['close'].rolling(window=20).min()
    df['MPR'] = ((df['close'] - df['20_day_Moving_Low']) / (df['20_day_Moving_High'] - df['20_day_Moving_Low']) > 0.5).astype(int)

    # Calculate Volume Growth (VGF)
    df['3_day_MA_Volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_MA_Volume']).astype(int)

    # Calculate Volume-Weighted Close Price (VWCP)
    N_days = 5
    df['Volume_Price_Product'] = df['close'] * df['volume']
    sum_volume = df['volume'].rolling(window=N_days).sum()
    df['VWCP'] = df['Volume_Price_Product'].rolling(window=N_days).sum() / sum_volume

    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3

    # Generate Intermediate Alpha Scores
    scaling_factor_hl = 1.0
    scaling_factor_vwcp = 0.5
    scaling_factor_momentum = 1.5

    df['High_Low_Momentum_Score'] = df['Combined_HL_VA'] * scaling_factor_hl
    df['Volume_Weighted_Score'] = df['VWCP'] * scaling_factor_vwcp
    df['Momentum_Score'] = df['Average_Price_Change'] * scaling_factor_momentum

    # Generate Final Alpha Factor
    df['Intermediate_Alpha_Scores'] = df['High_Low_Momentum_Score'] + df['Volume_Weighted_Score'] + df['Momentum_Score']
    df['Final_Alpha_Factor'] = df['Intermediate_Alpha_Scores'] + df['IMVTI']

    # Apply Smoothing
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.3).mean()

    return df['Final_Alpha_Factor']
```
