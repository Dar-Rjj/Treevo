```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (e.g., 5, 10, 20 days)
    N = 10
    df['Avg_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).mean()
    
    # Calculate Volume Adjusted Close-Open Difference
    df['Volume_Adjusted_Close_Open'] = (df['close'] - df['open']) / df['volume'].rolling(window=10).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_High_Low_Volume'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']
    df['Sum_Combined_High_Low_Volume'] = df['Combined_High_Low_Volume'].rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    df['Volume_Weighted_Close'] = df['volume'] * df['close']
    df['Volume_Weighted_Avg'] = df['Volume_Weighted_Close'] / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = df['Sum_Combined_High_Low_Volume'] * 1.2
    volume_weighted_score = df['Volume_Weighted_Avg'] * 0.7
    momentum_score = df['Avg_Price_Change'] * 1.8
    
    # Integrate Trend Indicators
    df['SPDI'] = (df['close'].rolling(window=5).mean() > df['close']).astype(int)
    df['MPR'] = ((df['close'].rolling(window=20).max() + df['close'].rolling(window=20).min()) / 2 < df['close']).astype(int)
    df['VGF'] = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume_score = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_score + df['IMVTI']
    
    # Apply Smoothing
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.4).mean()
    
    return final_alpha_factor
```
