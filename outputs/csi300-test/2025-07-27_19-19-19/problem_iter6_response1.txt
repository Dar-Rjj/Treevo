import pandas as pd

def heuristics_v2(df, N=20, momentum_scaling=1.5, volume_weighting_scaling=0.5, smoothing_factor=0.3):
    # Compute Daily Price and Volume Change
    df['close_diff'] = df['close'].diff()
    df['volume_diff'] = df['volume'].diff()
    df['combined_daily_change'] = df['close_diff'] + df['volume_diff']
    
    # Aggregate Changes Over a Period
    df['accumulated_daily_changes'] = df['combined_daily_change'].rolling(window=N).sum()
    df['average_daily_change'] = df['accumulated_daily_changes'] / N
    
    # Calculate Volume-Weighted Close Price
    df['volume_price_product'] = df['volume'] * df['close']
    df['volume_sum'] = df['volume'].rolling(window=N).sum()
    df['volume_weighted_price'] = df['volume_price_product'].rolling(window=N).sum() / df['volume_sum']
    
    # Combine Price and Volume Factors
    df['momentum_score'] = df['average_daily_change'] * momentum_scaling
    df['volume_weighted_score'] = df['volume_weighted_price'] * volume_weighting_scaling
    df['final_alpha_factor'] = df['momentum_score'] + df['volume_weighted_score']
    
    # Apply Smoothing
    df['smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=smoothing_factor).mean()
    
    return df['smoothed_alpha_factor']
