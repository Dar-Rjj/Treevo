import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, long_term_window=100, smoothing_factor=0.5):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    df['Accumulated_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum()
    df['Average_Price_Change'] = df['Accumulated_Price_Change'] / N
    
    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Avg_Volume'] = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Close_Open_Diff'] = df['Close_Open_Diff'] * (df['volume'] / df['Avg_Volume'])
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_High_Low_Range_VA'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open_Diff']
    df['Sum_Combined_HLR_VA'] = df['Combined_High_Low_Range_VA'].rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['Volume_Sum_20'] = df['volume'].rolling(window=20).sum()
    df['Volume_Weighted_Close_Price'] = df['Volume_Price_Product'] / df['Volume_Sum_20']
    
    # Generate Intermediate Alpha Scores
    df['High_Low_Momentum_Score'] = df['Sum_Combined_HLR_VA'] * 1.5
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Close_Price'] * 0.7
    df['Momentum_Score'] = df['Average_Price_Change'] * 1.9
    
    # Integrate Indicators
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA']).astype(int)
    
    df['20_day_MA_high'] = df['high'].rolling(window=20).max()
    df['20_day_MA_low'] = df['low'].rolling(window=20).min()
    df['MPR'] = ((df['close'] - df['20_day_MA_low']) / (df['20_day_MA_high'] - df['20_day_MA_low']) > 0.5).astype(int)
    
    df['5_day_MA_vol'] = df['volume'].rolling(window=5).mean()
    df['VGF'] = (df['volume'] > df['5_day_MA_vol']).astype(int)
    
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['Combined_Momentum_Volume_Score'] = df['Momentum_Score'] + df['Volume_Weighted_Score']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['High_Low_Momentum_Score'] + df['Combined_Momentum_Volume_Score'] + df['IMVTI']
    
    # Apply Smoothing
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=smoothing_factor).mean()
    
    # Incorporate Long-Term Trend
    df['100_day_MA'] = df['close'].rolling(window=long_term_window).mean()
    adjustment = (df['close'] > df['100_day_MA']).apply(lambda x: 1.15 if x else 0.85)
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * adjustment
    
    return df['Final_Alpha_Factor']
