import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    N = 5
    df['Average_Price_Change_5'] = df['Daily_Price_Change'].rolling(window=N).mean()
    
    N = 10
    df['Average_Price_Change_10'] = df['Daily_Price_Change'].rolling(window=N).mean()
    
    N = 20
    df['Average_Price_Change_20'] = df['Daily_Price_Change'].rolling(window=N).mean()
    
    # Calculate Volume Adjusted Close-Open Difference
    df['Volume_Adjusted_Close_Open_Diff'] = (df['close'] - df['open']) / df['volume'].rolling(window=20).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_High_Low_Volume_Adjusted'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open_Diff']
    df['Sum_Combined_10'] = df['Combined_High_Low_Volume_Adjusted'].rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    df['Volume_Weighted_Close'] = (df['volume'] * df['close']) / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    scaling_factor_high_low = 1.2
    df['High_Low_Momentum_Score'] = df['Sum_Combined_10'] * scaling_factor_high_low
    
    scaling_factor_volume_weighted = 0.6
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Close'] * scaling_factor_volume_weighted
    
    scaling_factor_momentum = 1.8
    df['Momentum_Score'] = df['Average_Price_Change_20'] * scaling_factor_momentum
    
    # Integrate Trend Indicators
    df['SPDI'] = (df['close'].rolling(window=5).mean() > df['close']).astype(int)
    
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['MPR'] = ((df['close'] - df['20_day_low']) / (df['20_day_high'] - df['20_day_low']) > 0.5).astype(int)
    
    df['VGF'] = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['Combined_Momentum_Volume_Score'] = df['Momentum_Score'] + df['Volume_Weighted_Score']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['High_Low_Momentum_Score'] + df['Combined_Momentum_Volume_Score'] + df['IMVTI']
    
    # Apply Smoothing
    df['Smoothed_Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.4).mean()
    
    return df['Smoothed_Final_Alpha_Factor']
