```python
import pandas as pd

def heuristics_v2(df, N=20, scaling_high_low=1.2, scaling_volume_weighted=0.6, scaling_momentum=1.8, smoothing_factor=0.4):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_close_open = close_open_diff / (df['volume'] / avg_volume)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_voladj = high_low_range * volume_adjusted_close_open
    combined_hl_voladj_sum = combined_hl_voladj.rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close = volume_price_product.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_hl_voladj_sum * scaling_high_low
    volume_weighted_score = volume_weighted_close * scaling_volume_weighted
    momentum_score = average_price_change * scaling_momentum
    
    # Integrate Indicators
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    mp_high = df['high'].rolling(window=20).max()
    mp_low = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - mp_low) / (mp_high - mp_low) > 0.5).astype(int)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    intermediate_alpha_scores = high_low_momentum_score + combined_momentum_volume
    final_alpha_factor = intermediate_alpha_scores + imvti
    final_alpha_factor = final_alpha_factor.ewm(alpha=smoothing_factor).mean()
    
    # Incorporate Long-Term Trend
    long_term_trend = df['close'].rolling(window=100).mean()
    trend_adjustment = (df['close'] > long_term_trend).apply(lambda x: 0.1 if x else -0.1)
    final_alpha_factor = final_alpha_factor * (1 + trend_adjustment)
    
    return final_alpha_factor
```
