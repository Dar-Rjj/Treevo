```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Close-Open Difference
    df['close_open_diff'] = df['close'] - df['open']
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['volume_adjusted_close_open_diff'] = df['close_open_diff'] * (df['volume'] / df['avg_volume_10d'])
    
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Aggregate Price Change Over a Period (N = 5 for example)
    N = 5
    df['aggregate_price_change'] = df['daily_price_change'].rolling(window=N).sum() / N
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['combined_hl_vol_adj'] = df['high_low_range'] * df['volume_adjusted_close_open_diff']
    df['sum_combined_hl_vol_adj'] = df['combined_hl_vol_adj'].rolling(window=5).sum()
    
    # Calculate Short-Term Price Direction (SPDI)
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['spdi'] = np.where(df['close'] > df['5_day_ma_close'], 1, -1)
    
    # Calculate Medium-Term Price Range (MPR)
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['price_range_position'] = (df['close'] - df['20_day_low']) / (df['20_day_high'] - df['20_day_low'])
    df['mpr'] = np.where(df['price_range_position'] > 0.5, 1, -1)
    
    # Calculate Volume Growth (VGF)
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['vgf'] = np.where(df['volume'] > df['3_day_avg_volume'], 1, -1)
    
    # Integrate Indicators
    df['imvti'] = (df['spdi'] + df['mpr'] + df['vgf']) / 3
    
    # Generate Intermediate Alpha Scores
    scaling_factor_hl_momentum = 1.3
    df['high_low_momentum_score'] = df['sum_combined_hl_vol_adj'] * scaling_factor_hl_momentum
    
    scaling_factor_momentum = 1.8
    df['momentum_score'] = df['aggregate_price_change'] * scaling_factor_momentum
    
    # Calculate Volume-Weighted Close Price (VWCP)
    df['volume_price_product'] = df['volume'] * df['close']
    df['sum_volume_last_N_days'] = df['volume'].rolling(window=N).sum()
    df['vwcp'] = df['volume_price_product'] / df['sum_volume_last_N_days']
    
    scaling_factor_vwcp = 0.7
    df['volume_weighted_score'] = df['vwcp'] * scaling_factor_vwcp
    
    # Combine Price and Volume Factors
    df['combined_momentum_volume'] = df['momentum_score'] + df['volume_weighted_score']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['high_low_momentum_score'] + df['combined_momentum_volume'] + df['imvti']
    
    # Apply Smoothing
    smoothing_factor = 0.5
    df['smoothed_final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=smoothing_factor).mean()
    
    return df['smoothed_final_alpha_factor']
```
