import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the difference in money flow from previous day
    diff_money_flow = money_flow.diff()
    
    # Calculate the cumulative sum of the money flow
    cum_money_flow = diff_money_flow.cumsum()
    
    # Normalize the cumulative money flow by the rolling standard deviation
    normalized_cum_money_flow = cum_money_flow / cum_money_flow.rolling(window=10).std()
    
    # Return the factor
    return normalized_cum_money_flow
