import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (e.g., 5, 10, 20 days)
    for period in [5, 10, 20]:
        df[f'avg_price_change_{period}'] = df['daily_price_change'].rolling(window=period).mean()
    
    # Volume Adjusted Close-Open Difference
    df['volume_adjusted_close_open_diff'] = (df['close'] - df['open']) / df['volume'].rolling(window=30).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['combined_high_low_vol_adj'] = df['high_low_range'] * df['volume_adjusted_close_open_diff']
    df['sum_combined_high_low_vol_adj'] = df['combined_high_low_vol_adj'].rolling(window=15).sum()
    
    # Volume Weighted Average
    df['volume_weighted_close'] = (df['volume'] * df['close']) / df['volume'].rolling(window=30).sum()
    
    # Generate Intermediate Alpha Scores
    df['high_low_momentum_score'] = df['sum_combined_high_low_vol_adj'] * 2.0
    df['volume_weighted_score'] = df['volume_weighted_close'] * 0.8
    df['momentum_score'] = df['avg_price_change_20'] * 2.1
    
    # Integrate Trend Indicators
    df['7_day_ma_close'] = df['close'].rolling(window=7).mean()
    df['spdi'] = np.where(df['close'] > df['7_day_ma_close'], 1, -1)
    
    df['30_day_high'] = df['high'].rolling(window=30).max()
    df['30_day_low'] = df['low'].rolling(window=30).min()
    df['mpri'] = np.where((df['close'] - df['30_day_low']) / (df['30_day_high'] - df['30_day_low']) > 0.5, 1, -1)
    
    df['7_day_ma_volume'] = df['volume'].rolling(window=7).mean()
    df['vgf'] = np.where(df['volume'] > df['7_day_ma_volume'], 1, -1)
    
    df['imvti'] = (df['spdi'] + df['mpri'] + df['vgf']) / 3
    
    # Combine Price and Volume Factors
    df['combined_mom_vol_scores'] = df['momentum_score'] + df['volume_weighted_score']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['high_low_momentum_score'] + df['combined_mom_vol_scores'] + df['imvti']
    
    # Apply Smoothing
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(alpha=0.6).mean()
    
    # Incorporate Long-Term Trend
    df['200_day_ma_close'] = df['close'].rolling(window=200).mean()
    df['final_alpha_factor_adjusted'] = np.where(
        df['close'] > df['200_day_ma_close'],
        df['final_alpha_factor_smoothed'] * 1.2,
        df['final_alpha_factor_smoothed'] * 0.8
    )
    
    return df['final_alpha_factor_adjusted']
