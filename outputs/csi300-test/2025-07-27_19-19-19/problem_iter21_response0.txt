```python
import pandas as pd

def heuristics_v2(df, N=10, short_term_ma=5, medium_term_range=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    accumulated_price_change = daily_price_change.rolling(N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(20).mean()
    volume_adjusted_close_open = close_open_diff * (df['volume'] / avg_volume)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_vol_adj = (high_low_range * volume_adjusted_close_open).rolling(10).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close = volume_price_product.rolling(20).sum() / df['volume'].rolling(20).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_hl_vol_adj * 1.2
    volume_weighted_score = volume_weighted_close * 0.6
    momentum_score = average_price_change * 1.8
    
    # Integrate Indicators
    short_term_ma_close = df['close'].rolling(short_term_ma).mean()
    spdi = (df['close'] > short_term_ma_close).astype(int)
    
    moving_high = df['high'].rolling(medium_term_range).max()
    moving_low = df['low'].rolling(medium_term_range).min()
    mpr = ((df['close'] - moving_low) / (moving_high - moving_low) > 0.5).astype(int)
    
    avg_volume_5 = df['volume'].rolling(5).mean()
    vgf = (df['volume'] > avg_volume_5).astype(int)
    
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_vol = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_vol + imvti
    smoothed_alpha_factor = final_alpha_factor.rolling(7).mean()
    
    return smoothed_alpha_factor
```
