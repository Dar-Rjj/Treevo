```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average for 5 and 10 days for close price
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the exponential moving average for 5 and 10 days for close price
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    
    # Price momentum: Today's close price - Close price 10 days ago
    df['momentum'] = df['close'] - df['close'].shift(10)
    
    # Price range: (High - Low) / Open
    df['range'] = (df['high'] - df['low']) / df['open']
    
    # Volume weighted average price (VWAP): (High + Low + Close) / 3, multiplied by volume, then averaged
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    df['vwap'] = df['vwap'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate the alpha factor
    # Alpha factor is a combination of SMA crossover, EMA crossover, momentum, price range, and VWAP
    df['alpha_factor'] = (df['SMA_5'] - df['SMA_10']) + (df['EMA_5'] - df['EMA_10']) + df['momentum'] + df['range'] + (df['close'] - df['vwap'])
    
    return df['alpha_factor']
```
