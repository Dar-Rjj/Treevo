```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average (SMA) for the last 5 and 10 days
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_10'] = df['close'].rolling(window=10).mean()

    # Calculate the relative strength index (RSI) for the last 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI'] = 100 - (100 / (1 + rs))

    # Calculate the percentage price change from the previous close
    df['pct_change'] = df['close'].pct_change()

    # Heuristic: if the short-term SMA is above the long-term SMA and RSI is below 70, it might indicate a bullish trend with room to grow
    df['alpha_factor'] = (df['SMA_5'] > df['SMA_10']) & (df['RSI'] < 70)

    # Convert boolean alpha factor to float (1 for True, 0 for False)
    df['alpha_factor'] = df['alpha_factor'].astype(int)

    return df['alpha_factor']
```
