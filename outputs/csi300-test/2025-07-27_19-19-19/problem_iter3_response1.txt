```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Direction Indicator (SPDI)
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = 0
    df.loc[df['close'] > df['5_day_ma_close'], 'SPDI'] = 1
    df.loc[df['close'] < df['5_day_ma_close'], 'SPDI'] = -1

    # Calculate Medium-Term Price Direction Indicator (MPDI)
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['20_day_range'] = df['20_day_high'] - df['20_day_low']
    df['MPDI'] = 0
    df.loc[df['close'] > (df['20_day_high'] + df['20_day_low']) / 2, 'MPDI'] = 1
    df.loc[df['close'] <= (df['20_day_high'] + df['20_day_low']) / 2, 'MPDI'] = -1

    # Calculate Volume Growth Factor (VGF)
    df['3_day_ma_volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = 0
    df.loc[df['volume'] > df['3_day_ma_volume'], 'VGF'] = 1
    df.loc[df['volume'] <= df['3_day_ma_volume'], 'VGF'] = -1

    # Integrate Indicators
    df['EPVTI'] = (df['SPDI'] + df['MPDI'] + df['VGF']) / 3

    # Final Indicator Adjustment
    df['EPVTI_final'] = 0
    df.loc[df['EPVTI'] > 0.66, 'EPVTI_final'] = 1
    df.loc[(df['EPVTI'] > 0.33) & (df['EPVTI'] <= 0.66), 'EPVTI_final'] = 0.5
    df.loc[df['EPVTI'] <= 0.33, 'EPVTI_final'] = 0

    return df['EPVTI_final']
```
