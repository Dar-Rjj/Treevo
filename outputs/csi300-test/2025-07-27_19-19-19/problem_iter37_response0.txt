import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (N=5, 10, 20)
    periods = [5, 10, 20]
    for period in periods:
        df[f'avg_price_change_{period}'] = df['daily_price_change'].rolling(window=period).mean()

    # Calculate Volume Adjusted Close-Open Difference
    df['volume_adjusted_close_open_diff'] = (df['close'] - df['open']) / df['volume'].rolling(window=20).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['combined_high_low_volume'] = df['high_low_range'] * df['volume_adjusted_close_open_diff']
    df['sum_combined_high_low_volume_10'] = df['combined_high_low_volume'].rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    df['volume_weighted_close'] = (df['volume'] * df['close']) / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    df['high_low_momentum_score'] = df['sum_combined_high_low_volume_10'] * 1.2
    df['volume_weighted_score'] = df['volume_weighted_close'] * 0.6
    df['momentum_score'] = df['avg_price_change_5'] * 1.8
    
    # Integrate Trend Indicators
    df['5_day_ma_close'] = df['close'].rolling(window=5).mean()
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['5_day_ma_volume'] = df['volume'].rolling(window=5).mean()
    
    # Short-Term Price Direction (SPDI)
    df['spdi'] = (df['close'] > df['5_day_ma_close']).astype(int)
    
    # Medium-Term Price Range (MPR)
    mid_range = (df['20_day_high'] + df['20_day_low']) / 2
    df['mpr'] = (df['close'] > mid_range).astype(int)
    
    # Volume Growth (VGF)
    df['vgf'] = (df['volume'] > df['5_day_ma_volume']).astype(int)
    
    # Integrated Momentum, Volume, and Trend Indicators (IMVTI)
    df['imvti'] = (df['spdi'] + df['mpr'] + df['vgf']) / 3
    
    # Combine Price and Volume Factors
    df['combined_momentum_volume_scores'] = df['momentum_score'] + df['volume_weighted_score']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = (df['high_low_momentum_score'] + 
                                df['combined_momentum_volume_scores'] + 
                                df['imvti'])
    
    # Apply Smoothing
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(alpha=0.4).mean()
    
    return df['final_alpha_factor_smoothed']
