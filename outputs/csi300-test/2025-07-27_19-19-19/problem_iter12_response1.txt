```python
import pandas as pd

def heuristics_v2(df, N=20):
    # Compute Daily Price and Volume Change
    df['Close_Price_Diff'] = df['close'].diff()
    df['Volume_Change'] = df['volume'].diff()
    df['Combined_Daily_Change'] = df['Close_Price_Diff'] + df['Volume_Change']
    
    # Aggregate Changes Over a Period
    df['Accumulated_Changes'] = df['Combined_Daily_Change'].rolling(window=N).sum()
    df['Average_Daily_Change'] = df['Accumulated_Changes'] / N
    
    # Compute Short-Term Price Direction (SPDI)
    df['5_day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA_Close']).astype(int)
    
    # Compute Medium-Term Price Range (MPR)
    df['20_day_MA_High'] = df['high'].rolling(window=20).max()
    df['20_day_MA_Low'] = df['low'].rolling(window=20).min()
    df['Mid_Range'] = (df['20_day_MA_High'] + df['20_day_MA_Low']) / 2
    df['MPR'] = (df['close'] > df['Mid_Range']).astype(int)
    
    # Calculate Volume Growth (VGF)
    df['3_day_MA_Volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_MA_Volume']).astype(int)
    
    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Calculate Volume-Weighted Close Price
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['Volume_Sum_N'] = df['volume'].rolling(window=N).sum()
    df['Volume_Weighted_Price'] = df['Volume_Price_Product'].rolling(window=N).sum() / df['Volume_Sum_N']
    
    # Combine Price and Volume Factors
    df['Momentum_Score'] = df['Average_Daily_Change'] * 1.5
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Price'] * 0.5
    
    # Compute Long-Term Trend Indicator (LTTI)
    df['60_day_MA_Close'] = df['close'].rolling(window=60).mean()
    df['LTTI'] = (df['close'] > df['60_day_MA_Close']).astype(int)
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Momentum_Score'] + df['Volume_Weighted_Score'] + df['IMVTI'] + df['LTTI']
    
    # Apply Smoothing
    df['Smoothed_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.3).mean()
    
    return df['Smoothed_Alpha_Factor']
```
