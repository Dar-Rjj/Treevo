import pandas as pd

def heuristics_v2(df, N=10, long_term_window=100, smoothing_factor=0.4):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    df['Accumulated_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum()
    df['Average_Price_Change'] = df['Accumulated_Price_Change'] / N
    
    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Close_Open'] = df['Close_Open_Diff'] * (df['volume'] / avg_volume_20)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_HL_VA'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']
    df['Sum_Combined_HL_VA'] = df['Combined_HL_VA'].rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    df['Volume_Price_Product'] = df['volume'] * df['close']
    volume_sum_20 = df['volume'].rolling(window=20).sum()
    df['Volume_Weighted_Close_Price'] = df['Volume_Price_Product'] / volume_sum_20
    
    # Generate Intermediate Alpha Scores
    df['High_Low_Momentum_Score'] = 1.2 * df['Sum_Combined_HL_VA']
    df['Volume_Weighted_Score'] = 0.6 * df['Volume_Weighted_Close_Price']
    df['Momentum_Score'] = 1.8 * df['Average_Price_Change']
    
    # Integrate Indicators
    df['SPDI'] = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['MPR'] = (df['close'] > (df['20_day_high'] + df['20_day_low']) / 2).astype(int)
    df['VGF'] = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['Combined_Momentum_Volume_Score'] = df['Momentum_Score'] + df['Volume_Weighted_Score']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['High_Low_Momentum_Score'] + df['Combined_Momentum_Volume_Score'] + df['IMVTI']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=smoothing_factor).mean()
    
    # Incorporate Long-Term Trend
    df['100_day_MA'] = df['close'].rolling(window=long_term_window).mean()
    long_term_trend_adjustment = (df['close'] > df['100_day_MA']).astype(float) * 0.1 - 0.1
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * (1 + long_term_trend_adjustment)
    
    return df['Final_Alpha_Factor'].dropna()
