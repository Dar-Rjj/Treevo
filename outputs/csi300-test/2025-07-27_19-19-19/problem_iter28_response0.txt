```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Close-Open Difference
    volume_adjusted_diff = (df['close'] - df['open']) / df['volume']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period (last N days)
    N = 5
    aggregated_price_change = daily_price_change.rolling(window=N).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_vol_diff = (high_low_range * volume_adjusted_diff).rolling(window=5).sum()
    
    # Integrate Indicators
    # Short-Term Price Direction (SPDI)
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    
    # Medium-Term Price Range (MPR)
    moving_high = df['high'].rolling(window=20).max()
    moving_low = df['low'].rolling(window=20).min()
    mpr = (df['close'] > (moving_high + moving_low) / 2).astype(int)
    
    # Volume Growth (VGF)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    
    # Integrated Momentum-Volume and Trend Indicator (IMVTI)
    imvti = (spdi + mpr + vgf) / 3
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_hl_vol_diff * 1.0
    momentum_score = aggregated_price_change * 1.5
    
    # Combine Price and Volume Factors
    combined_alpha_score = high_low_momentum_score + momentum_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = combined_alpha_score + imvti
    
    # Apply Smoothing
    smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.3, adjust=False).mean()
    
    return smoothed_alpha_factor
```
