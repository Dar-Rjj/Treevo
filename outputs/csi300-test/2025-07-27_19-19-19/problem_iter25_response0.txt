```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    volume_adjusted_diff = close_open_diff / (df['volume'] / avg_volume_10d)
    
    # Compute Daily Price Change
    daily_price_change = df['close'] - df['close'].shift(1)
    
    # Aggregate Price Change Over a Period (N=5 days for example)
    N = 5
    aggregate_price_change = daily_price_change.rolling(window=N).sum() / N
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_high_low = (high_low_range * volume_adjusted_diff).rolling(window=5).sum()
    
    # Integrate Indicators
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    moving_high_20d = df['high'].rolling(window=20).max()
    moving_low_20d = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - moving_low_20d) > (moving_high_20d - moving_low_20d) / 2).astype(int)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    imvti = (spdi + mpr + vgf) / 3
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low * 1.0
    momentum_score = aggregate_price_change * 1.5
    
    # Combine Price and Volume Factors
    combined_momentum_and_volume = momentum_score + high_low_momentum_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = combined_momentum_and_volume + imvti
    smoothed_final_alpha_factor = final_alpha_factor.ewm(alpha=0.3, adjust=False).mean()
    
    return smoothed_final_alpha_factor
```
