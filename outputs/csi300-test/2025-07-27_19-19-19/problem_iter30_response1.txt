```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period (last 20 days)
    N = 20
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_close_open = close_open_diff * (df['volume'] / avg_volume_20)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference (last 10 days)
    combined_hl_vol_adj_co = (high_low_range * volume_adjusted_close_open).rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close = volume_price_product.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_hl_vol_adj_co * 1.5
    volume_weighted_score = volume_weighted_close * 0.7
    momentum_score = average_price_change * 1.9
    
    # Integrate Indicators
    spdi = np.where(df['close'] > df['close'].rolling(window=5).mean(), 1, -1)
    mpr = np.where((df['close'] > (df['high'].rolling(window=20).max() + df['low'].rolling(window=20).min()) / 2), 1, -1)
    vgf = np.where(df['volume'] > df['volume'].rolling(window=5).mean(), 1, -1)
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume + imvti
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.5).mean()
    
    # Incorporate Long-Term Trend
    long_term_trend = df['close'].rolling(window=100).mean()
    adjustment_factor = np.where(df['close'] > long_term_trend, 1.15, 0.85)
    final_alpha_factor *= adjustment_factor
    
    return final_alpha_factor
```
