```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Adjust by Volume
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open_diff = close_open_diff * (df['volume'] / avg_volume_10)
    
    # Weight for the volume adjusted close-open difference
    weight = 0.5  # This can be tuned
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_factor = high_low_range * (weight * volume_adjusted_close_open_diff)
    
    # Sum the results over the last 5 days
    factor_values = combined_factor.rolling(window=5).sum()
    
    return factor_values
```
