```python
import pandas as pd

def heuristics_v2(df, N=5, M=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adjusted_diff = close_open_diff * (df['volume'] / avg_volume_10)
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    aggregate_price_change = daily_price_change.rolling(window=N).sum() / N
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_high_low = (high_low_range * volume_adjusted_diff).rolling(window=5).sum()
    
    # Calculate Short-Term Price Direction (SPDI)
    ma_5_close = df['close'].rolling(window=5).mean()
    spdi = (df['close'] > ma_5_close).astype(int)
    
    # Calculate Medium-Term Price Range (MPR)
    moving_high_20 = df['high'].rolling(window=M).max()
    moving_low_20 = df['low'].rolling(window=M).min()
    mpr = ((df['close'] - moving_low_20) / (moving_high_20 - moving_low_20) > 0.5).astype(int)
    
    # Calculate Volume Growth (VGF)
    ma_3_volume = df['volume'].rolling(window=3).mean()
    vgf = (df['volume'] > ma_3_volume).astype(int)
    
    # Integrate Indicators
    imvti = (spdi + mpr + vgf) / 3
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low * 1.0
    momentum_score = aggregate_price_change * 1.5
    
    # Calculate Volume-Weighted Close Price (VWCP)
    volume_price_product = df['volume'] * df['close']
    sum_volume_N_days = df['volume'].rolling(window=N).sum()
    vwcp = volume_price_product / sum_volume_N_days
    volume_weighted_score = vwcp * 0.5
    
    # Combine Price and Volume Factors
    combined_momentum_volume = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume + imvti
    smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.3, adjust=False).mean()
    
    return smoothed_alpha_factor
```
