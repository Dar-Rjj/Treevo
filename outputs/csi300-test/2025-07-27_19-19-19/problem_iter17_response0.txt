```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['HLR'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['daily_change'] = df['close'] - df['close'].shift(1)
    
    # Aggregate Price Change Over a Period (Last N Days, e.g., 10 days)
    df['avg_price_change'] = df['daily_change'].rolling(window=10).mean()
    
    # Calculate Volume-Adjusted Close-Open Difference
    df['CO_diff'] = df['close'] - df['open']
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['vol_adj_CO_diff'] = df['CO_diff'] / df['avg_volume_10']
    
    # Combine High-Low Range and Volume-Adjusted Close-Open Difference
    df['HLR_vol_adj_CO_diff'] = df['HLR'] * df['vol_adj_CO_diff']
    df['combined_HLR_vol_adj_CO_diff'] = df['HLR_vol_adj_CO_diff'].rolling(window=5).sum()
    
    # Compute Short-Term Price Direction (SPDI)
    df['MA_5_close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['MA_5_close']).astype(int)
    
    # Compute Medium-Term Price Range (MPR)
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['range'] = (df['close'] - df['20_day_low']) / (df['20_day_high'] - df['20_day_low'])
    df['MPR'] = (df['range'] > 0.5).astype(int)
    
    # Integrate Indicators to get IMVTI
    df['IMVTI'] = (df['SPDI'] + df['MPR']) / 2
    
    # Generate Intermediate Alpha Scores
    df['HL_momentum_score'] = df['combined_HLR_vol_adj_CO_diff'] * 1.0
    
    # Calculate Volume-Weighted Score
    df['volume_price_product'] = df['volume'] * df['close']
    df['volume_weighted_close'] = df['volume_price_product'] / df['volume'].rolling(window=10).sum()
    
    # Incorporate Trading Volume Ratio (TVR)
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['TVR'] = (df['volume'] > df['avg_volume_10']).astype(int)
    
    # Calculate Volume Growth (VGF)
    df['3_day_avg_volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_avg_volume']).astype(int)
    
    # Integrate Enhanced Indicators to get EIMVTI
    df['EIMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['momentum_score'] = df['avg_price_change'] * 1.5
    df['combined_momentum_volume_score'] = df['momentum_score'] + df['volume_weighted_close']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['HL_momentum_score'] + df['combined_momentum_volume_score'] + df['EIMVTI']
    df['smoothed_final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.3).mean()
    
    return df['smoothed_final_alpha_factor']
```
