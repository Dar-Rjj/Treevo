```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']

    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adj_close_open = (close_open_diff * df['volume']) / avg_volume_10

    # Compute Daily Price Change
    daily_price_change = df['close'].diff()

    # Aggregate Price Change Over a Period (N = 10)
    N = 10
    aggregate_price_change = daily_price_change.rolling(window=N).sum() / N

    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_signal = high_low_range * volume_adj_close_open
    combined_signal_5_days = combined_signal.rolling(window=5).sum()

    # Calculate Short-Term Price Direction (SPDI)
    ma_5 = df['close'].rolling(window=5).mean()
    spdi = (df['close'] > ma_5).astype(int)

    # Calculate Medium-Term Price Range (MPR)
    rolling_high_20 = df['close'].rolling(window=20).max()
    rolling_low_20 = df['close'].rolling(window=20).min()
    mpr = ((df['close'] - rolling_low_20) / (rolling_high_20 - rolling_low_20) > 0.5).astype(int)

    # Calculate Volume Growth (VGF)
    ma_volume_3 = df['volume'].rolling(window=3).mean()
    vgf = (df['volume'] > ma_volume_3).astype(int)

    # Integrate Indicators
    imvti = (spdi + mpr + vgf) / 3

    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_signal_5_days * 1.0
    momentum_score = aggregate_price_change * 1.5

    # Calculate Volume-Weighted Close Price (VWCP)
    volume_price_product = df['volume'] * df['close']
    sum_volume_N_days = df['volume'].rolling(window=N).sum()
    vwcp = volume_price_product / sum_volume_N_days

    # Calculate Volume Weighted Score
    volume_weighted_score = vwcp * 0.5

    # Generate Final Alpha Factor
    final_alpha_factor = (high_low_momentum_score + 
                          momentum_score + 
                          imvti + 
                          volume_weighted_score)

    # Apply Smoothing
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.3).mean()

    return final_alpha_factor_smoothed
```
