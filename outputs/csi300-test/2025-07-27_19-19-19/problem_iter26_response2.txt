```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open = close_open_diff * (df['volume'] / avg_volume)
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period
    agg_price_change = daily_price_change.rolling(window=N).sum()
    avg_price_change = agg_price_change / N
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_vol = (high_low_range * volume_adjusted_close_open).rolling(window=5).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close = volume_price_product / df['volume'].rolling(window=20).sum()
    
    # Integrate Indicators
    # Short-Term Price Direction (SPDI)
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    
    # Medium-Term Price Range (MPR)
    mpr_high = df['high'].rolling(window=20).max()
    mpr_low = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - mpr_low) / (mpr_high - mpr_low) > 0.5).astype(int)
    
    # Volume Growth (VGF)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    
    # IMVTI
    imvti = (spdi + mpr + vgf) / 3
    
    # Generate Intermediate Alpha Scores
    hl_momentum_score = combined_hl_vol * 1.2
    volume_weighted_score = volume_weighted_close * 0.6
    momentum_score = avg_price_change * 1.8
    
    # Combine Price and Volume Factors
    combined_scores = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = hl_momentum_score + combined_scores + imvti
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.4).mean()
    
    # Incorporate Long-Term Trend
    long_term_trend = df['close'].rolling(window=100).mean()
    trend_adjustment = (df['close'] > long_term_trend).astype(int) * 0.1 - (df['close'] <= long_term_trend).astype(int) * 0.1
    final_alpha_factor = final_alpha_factor * (1 + trend_adjustment)
    
    return final_alpha_factor
```
