```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period (last 10 days)
    N = 10
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_close_open = close_open_diff * (df['volume'] / avg_volume_20)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference over the last 10 days
    combined_high_low_vol_adj = (high_low_range * volume_adjusted_close_open).rolling(window=N).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_sum_20 = df['volume'].rolling(window=20).sum()
    volume_weighted_close_price = volume_price_product / volume_sum_20
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low_vol_adj * 1.2
    volume_weighted_score = volume_weighted_close_price * 0.6
    momentum_score = average_price_change * 1.8
    
    # Integrate Indicators
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    moving_high_20 = df['high'].rolling(window=20).max()
    moving_low_20 = df['low'].rolling(window=20).min()
    mpr = ((df['close'] > (moving_high_20 + moving_low_20) / 2)).astype(int)
    vgf = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume_scores = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_scores + imvti
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.4).mean()
    
    # Incorporate Long-Term Trend
    ma_100 = df['close'].rolling(window=100).mean()
    trend_adjustment = (df['close'] > ma_100).astype(float) * 0.1 - (df['close'] <= ma_100).astype(float) * 0.1
    final_alpha_factor = final_alpha_factor * (1 + trend_adjustment)
    
    return final_alpha_factor
```
