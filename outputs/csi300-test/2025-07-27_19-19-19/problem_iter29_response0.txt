```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    
    # Aggregate Price Change Over a Period (N=10)
    N = 10
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_adjusted_close_open_diff = close_open_diff / avg_volume_20_days * df['volume']
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_high_low_vol_diff = (high_low_range * volume_adjusted_close_open_diff).rolling(window=N).sum()
    
    # Calculate Short-Term Price Direction (SPDI)
    short_term_ma = df['close'].rolling(window=5).mean()
    spdi = (df['close'] > short_term_ma).astype(int)
    
    # Calculate Medium-Term Price Range (MPR)
    moving_high_20_days = df['high'].rolling(window=20).max()
    moving_low_20_days = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - moving_low_20_days) / (moving_high_20_days - moving_low_20_days) > 0.5).astype(int)
    
    # Calculate Volume Growth (VGF)
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    vgf = (df['volume'] > avg_volume_5_days).astype(int)
    
    # Integrate Indicators
    imvti = (spdi + mpr + vgf) / 3
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low_vol_diff * 1.2
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close_price = volume_price_product / df['volume'].rolling(window=20).sum()
    volume_weighted_score = volume_weighted_close_price * 0.6
    momentum_score = average_price_change * 1.8
    
    # Combine Price and Volume Factors
    combined_momentum_volume_score = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_score + imvti
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.45, adjust=False).mean()
    
    # Incorporate Long-Term Trend
    long_term_ma_100_days = df['close'].rolling(window=100).mean()
    trend_adjustment = (df['close'] > long_term_ma_100_days).astype(int) * 0.1 - 0.1
    final_alpha_factor = final_alpha_factor * (1 + trend_adjustment)
    
    return final_alpha_factor
```
