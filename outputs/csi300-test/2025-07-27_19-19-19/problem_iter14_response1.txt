```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, P=5):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    daily_price_change_percent = daily_price_change / df['close'].shift(1)
    
    # Aggregate Price Change Over a Period (N days)
    accumulated_price_change = daily_price_change.rolling(window=N).sum()
    average_price_change = accumulated_price_change / N
    
    # Calculate Volume Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open_diff = (close_open_diff * df['volume']) / avg_volume_10
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    combined_hl_vol_adj = (high_low_range * volume_adjusted_close_open_diff).rolling(window=P).sum()
    
    # Calculate Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    volume_weighted_close = volume_price_product.rolling(window=M).sum() / df['volume'].rolling(window=M).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_hl_vol_adj * 1.0
    volume_weighted_score = volume_weighted_close * 0.5
    momentum_score = average_price_change * 1.5
    
    # Integrate Indicators
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    twenty_day_high = df['high'].rolling(window=20).max()
    twenty_day_low = df['low'].rolling(window=20).min()
    mpr = ((df['close'] - twenty_day_low) / (twenty_day_high - twenty_day_low) > 0.5).astype(int)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    
    imvti = (spdi + mpr + vgf) / 3
    
    # Combine Price and Volume Factors
    combined_momentum_volume_scores = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_scores + imvti
    smoothed_final_alpha_factor = final_alpha_factor.ewm(alpha=0.3).mean()
    
    return smoothed_final_alpha_factor
```
