```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (N=5, 10, 20)
    periods = [5, 10, 20]
    for N in periods:
        df[f'price_change_{N}_days'] = df['daily_price_change'].rolling(window=N).mean()
    
    # Calculate Volume Adjusted Close-Open Difference
    df['volume_adjusted_close_open'] = (df['close'] - df['open']) / df['volume'].rolling(window=20).mean()
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['combined_high_low_volume'] = (df['high_low_range'] * df['volume_adjusted_close_open']).rolling(window=10).sum()
    
    # Calculate Volume Weighted Average
    df['volume_weighted_close'] = (df['volume'] * df['close']) / df['volume'].rolling(window=20).sum()
    
    # Generate Intermediate Alpha Scores
    df['high_low_momentum_score'] = df['combined_high_low_volume'] * 1.2
    df['volume_weighted_score'] = df['volume_weighted_close'] * 0.6
    df['momentum_score'] = df['price_change_20_days'] * 1.8
    
    # Integrate Trend Indicators
    df['spdi'] = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    df['mpr'] = ((df['close'] > (df['high'].rolling(window=20).max() + df['low'].rolling(window=20).min()) / 2)).astype(int)
    df['vgf'] = (df['volume'] > df['volume'].rolling(window=5).mean()).astype(int)
    df['imvti'] = (df['spdi'] + df['mpr'] + df['vgf']) / 3
    
    # Combine Price and Volume Factors
    df['combined_momentum_volume'] = df['momentum_score'] + df['volume_weighted_score']
    
    # Generate Final Alpha Factor
    df['final_alpha_factor'] = df['high_low_momentum_score'] + df['combined_momentum_volume'] + df['imvti']
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.4).mean()
    
    return df['final_alpha_factor']
```
