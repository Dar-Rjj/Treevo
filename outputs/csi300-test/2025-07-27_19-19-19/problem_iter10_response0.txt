```python
import pandas as pd

def heuristics_v2(df, N=5, momentum_scaling=1.0, volume_scaling=1.0, ema_smoothing=0.2):
    """
    Generate an enhanced momentum-volume interaction alpha factor.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume) and index (date)
    - N: Number of days to aggregate changes over
    - momentum_scaling: Scaling factor for the momentum score
    - volume_scaling: Scaling factor for the volume-weighted score
    - ema_smoothing: Smoothing factor for the exponential moving average
    
    Returns:
    - pandas Series indexed by (date) representing the factor values
    """
    
    # Compute Daily Price and Volume Changes
    df['close_diff'] = df['close'].diff()
    df['volume_diff'] = df['volume'].diff()
    df['combined_daily_change'] = df['close_diff'] + df['volume_diff']
    
    # Aggregate Changes Over a Period (N days)
    df['sum_combined_changes'] = df['combined_daily_change'].rolling(window=N).sum()
    df['avg_daily_change'] = df['sum_combined_changes'] / N
    
    # Calculate Volume-Weighted Close Price
    df['volume_price_product'] = df['volume'] * df['close']
    df['volume_weighted_price'] = df['volume_price_product'].rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Combine Price and Volume Factors
    df['momentum_score'] = df['avg_daily_change'] * momentum_scaling
    df['volume_weighted_score'] = df['volume_weighted_price'] * volume_scaling
    df['combined_score'] = df['momentum_score'] + df['volume_weighted_score']
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['combined_score'].ewm(alpha=ema_smoothing).mean()
    
    return df['alpha_factor']
```
