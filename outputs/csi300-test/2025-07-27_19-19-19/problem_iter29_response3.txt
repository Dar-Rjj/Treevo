```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Volume Adjusted Close-Open Difference
    df['Volume_Adjusted_CO_Diff'] = (df['close'] - df['open']) / df['volume']

    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # Aggregate Price Change Over a Period (N=5)
    N = 5
    df['Aggregate_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).mean()

    # Combine High-Low Range and Volume Adjusted Close-Open Difference over the last 5 days
    df['Combined_HL_VA'] = (df['High_Low_Range'] * df['Volume_Adjusted_CO_Diff']).rolling(window=5).sum()

    # Short-Term Price Direction (SPDI)
    df['5_day_MA_Close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA_Close']).astype(int)

    # Medium-Term Price Range (MPR)
    df['20_day_Moving_High'] = df['close'].rolling(window=20).max()
    df['20_day_Moving_Low'] = df['close'].rolling(window=20).min()
    df['MPR'] = (df['close'] > (df['20_day_Moving_High'] + df['20_day_Moving_Low']) / 2).astype(int)

    # Volume Growth (VGF)
    df['3_day_MA_Volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_MA_Volume']).astype(int)

    # Generate IMVTI
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3

    # Calculate Volume Weighted Average
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['Volume_Sum_20_days'] = df['volume'].rolling(window=20).sum()
    df['Volume_Weighted_Close'] = (df['Volume_Price_Product']).rolling(window=20).sum() / df['Volume_Sum_20_days']

    # Generate Intermediate Alpha Scores
    df['High_Low_Momentum_Score'] = df['Combined_HL_VA'] * 1.0
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Close'] * 0.6
    df['Momentum_Score'] = df['Aggregate_Price_Change'] * 1.5

    # Combine Price and Volume Factors
    df['Combined_Momentum_Volume'] = df['Momentum_Score'] + df['High_Low_Momentum_Score']

    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['High_Low_Momentum_Score'] + df['Combined_Momentum_Volume'] + df['IMVTI']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.35, adjust=False).mean()

    return df['Final_Alpha_Factor']
```
