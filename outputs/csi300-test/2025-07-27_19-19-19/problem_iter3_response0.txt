```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Direction Indicator (SPDI)
    df['5_day_ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['SPDI'] = (df['close'] > df['5_day_ema_close']).astype(int)

    # Calculate Medium-Term Price Direction Indicator (MPDI)
    df['20_day_ema_high'] = df['high'].ewm(span=20, adjust=False).mean()
    df['20_day_ema_low'] = df['low'].ewm(span=20, adjust=False).mean()
    df['range_midpoint'] = (df['20_day_ema_high'] + df['20_day_ema_low']) / 2
    df['MPDI'] = (df['close'] > df['range_midpoint']).astype(int)

    # Calculate Volume Growth Factor (VGF)
    df['3_day_ema_volume'] = df['volume'].ewm(span=3, adjust=False).mean()
    df['VGF'] = (df['volume'] > df['3_day_ema_volume']).astype(int)

    # Integrate Indicators to get IMVTI
    df['IMVTI'] = (df['SPDI'] + df['MPDI'] + df['VGF']) / 3
    
    # Final Indicator Adjustment
    conditions = [
        (df['IMVTI'] > 0.66),
        (df['IMVTI'] > 0.33) & (df['IMVTI'] <= 0.66),
        (df['IMVTI'] <= 0.33)
    ]
    choices = [1, 0.5, 0]
    df['IMVTI'] = pd.np.select(conditions, choices, default=0)

    return df['IMVTI']
```
