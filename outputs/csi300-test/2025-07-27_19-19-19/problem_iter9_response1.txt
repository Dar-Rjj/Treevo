```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20):
    # Compute Short-Term Price Direction (SPDI)
    df['5_day_EMA'] = df['close'].ewm(span=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_EMA']).astype(int)

    # Compute Medium-Term Price Range (MPR)
    highest_high = df['high'].rolling(window=20).max()
    lowest_low = df['low'].rolling(window=20).min()
    df['Price_Range'] = highest_high - lowest_low
    df['Current_Price_Range'] = (df['close'] - lowest_low) / df['Price_Range']
    conditions = [
        (df['Current_Price_Range'] >= 2/3),
        (df['Current_Price_Range'] >= 1/3),
        (df['Current_Price_Range'] < 1/3)
    ]
    choices = [1.5, 1, 0.5]
    df['MPR'] = np.select(conditions, choices, default=0)

    # Calculate Volume Growth (VGF)
    df['3_day_Vol_MA'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_Vol_MA']).astype(int)

    # Aggregate Price Change Over N Days (APCN)
    df['Daily_Price_Change'] = df['close'].pct_change()
    df['APCN'] = df['Daily_Price_Change'].rolling(window=n_days).sum() / n_days

    # Calculate Volume-Weighted Close Price (VWCP)
    df['Volume_Price_Product'] = df['volume'] * df['close']
    total_volume = df['volume'].rolling(window=n_days).sum()
    df['VWCP'] = (df['Volume_Price_Product'].rolling(window=n_days).sum()) / total_volume

    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3

    # Combine Price and Volume Factors
    momentum_score = df['APCN'] * 1.5
    volume_weighted_score = df['VWCP'] * 0.5

    # Generate Final Alpha Factor (FAF)
    df['FAF'] = df['IMVTI'] + momentum_score + volume_weighted_score
    df['Final_Factor'] = df['FAF'].ewm(alpha=0.2).mean()

    return df['Final_Factor']
```
