```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Compute Daily Price Change
    daily_price_change = df['close'].diff()
    daily_price_change_percentage = daily_price_change / df['close'].shift(1)
    
    # Aggregate Price Change Over a Period
    sum_daily_price_changes = daily_price_change.rolling(window=N).sum()
    average_price_change = sum_daily_price_changes / N
    
    # Calculate Volume-Adjusted Close-Open Difference
    close_open_diff = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_adjusted_close_open_diff = close_open_diff * (df['volume'] / avg_volume)
    
    # Combine High-Low Range and Volume-Adjusted Close-Open Difference
    combined_high_low_vol_adj = (high_low_range * volume_adjusted_close_open_diff).rolling(window=5).sum()
    
    # Compute Short-Term Price Direction (SPDI)
    spdi = (df['close'] > df['close'].rolling(window=5).mean()).astype(int)
    
    # Compute Medium-Term Price Range (MPR)
    mpr = ((df['close'] > (df['high'].rolling(window=20).max() + df['low'].rolling(window=20).min()) / 2)).astype(int)
    
    # Calculate Volume Growth (VGF)
    vgf = (df['volume'] > df['volume'].rolling(window=3).mean()).astype(int)
    
    # Integrate Indicators
    imvti = (spdi + mpr + vgf) / 3
    
    # Calculate Volume-Weighted Close Price (VWCP)
    volume_price_product = df['volume'] * df['close']
    vwcp = volume_price_product.rolling(window=N).sum() / df['volume'].rolling(window=N).sum()
    
    # Generate Intermediate Alpha Scores
    high_low_momentum_score = combined_high_low_vol_adj * 1.0
    volume_weighted_score = vwcp * 0.5
    
    # Combine Price and Volume Factors
    momentum_score = average_price_change * 1.5
    combined_momentum_volume_scores = momentum_score + volume_weighted_score
    
    # Generate Final Alpha Factor
    final_alpha_factor = high_low_momentum_score + combined_momentum_volume_scores
    smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.3, adjust=False).mean()
    
    return smoothed_alpha_factor
```
