```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']

    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    df['Avg_Volume_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Adjusted_Close_Open'] = df['Close_Open_Diff'] / df['Avg_Volume_10']

    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()

    # Aggregate Price Change Over a Period (N = 5 days)
    N = 5
    df['Aggregate_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum() / N

    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['HLR_VACO'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']
    df['Combined_HLR_VACO'] = df['HLR_VACO'].rolling(window=5).sum()

    # Short-Term Price Direction (SPDI)
    df['MA_5'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['MA_5']).astype(int)

    # Medium-Term Price Range (MPR)
    df['MAH_20'] = df['high'].rolling(window=20).max()
    df['MAL_20'] = df['low'].rolling(window=20).min()
    df['Mid_Range_20'] = (df['MAH_20'] + df['MAL_20']) / 2
    df['MPR'] = (df['close'] > df['Mid_Range_20']).astype(int)

    # Volume Growth (VGF)
    df['MA_Vol_3'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['MA_Vol_3']).astype(int)

    # Integrate Indicators (IMVTI)
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3

    # Intermediate Alpha Scores
    df['High_Low_Momentum_Score'] = df['Combined_HLR_VACO'] * 1.0
    df['Momentum_Score'] = df['Aggregate_Price_Change'] * 1.5

    # Volume-Weighted Close Price (VWCP)
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['VWCP'] = df['Volume_Price_Product'] / df['volume'].rolling(window=N).sum()
    df['Volume_Weighted_Score'] = df['VWCP'] * 0.5

    # Combine Price and Volume Factors
    df['Combined_Price_Volume'] = df['Momentum_Score'] + df['Volume_Weighted_Score']

    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['High_Low_Momentum_Score'] + df['Combined_Price_Volume'] + df['IMVTI']

    # Apply Smoothing with Exponential Moving Average
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.3, adjust=False).mean()

    return df['Final_Alpha_Factor']
```
