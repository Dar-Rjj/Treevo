import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=5):
    # Compute Short-Term Price Direction (SPDI)
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA']).astype(int)

    # Compute Medium-Term Price Range (MPR)
    df['20_day_high'] = df['high'].rolling(window=20).max()
    df['20_day_low'] = df['low'].rolling(window=20).min()
    df['MPR'] = ((df['close'] - df['20_day_low']) / (df['20_day_high'] - df['20_day_low']) > 0.5).astype(int)

    # Calculate Volume Growth (VGF)
    df['3_day_vol_MA'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_vol_MA']).astype(int)

    # Integrate Indicators
    df['Integrated_Indicator'] = (df['SPDI'] + df['MPR']) / 2

    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].pct_change()

    # Aggregate Price Change Over N Days (APCN)
    df['APCN'] = df['Daily_Price_Change'].rolling(window=n_days).mean()

    # Calculate Volume-Weighted Close Price (VWCP)
    df['Volume_Price_Product'] = df['volume'] * df['close']
    df['VWCP'] = df['Volume_Price_Product'].rolling(window=n_days).sum() / df['volume'].rolling(window=n_days).sum()

    # Combine Price and Volume Factors
    momentum_score = df['APCN'] * 1.5
    volume_weighted_score = df['VWCP'] * 0.5

    # Generate Final Alpha Factor (FAF)
    df['FAF'] = df['Integrated_Indicator'] + momentum_score + volume_weighted_score

    # Apply Exponential Smoothing
    df['Final_Alpha_Factor'] = df['FAF'].ewm(alpha=0.3, adjust=False).mean()

    return df['Final_Alpha_Factor']
