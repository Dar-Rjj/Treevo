```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate 10-day and 30-day simple moving averages of the close price
    sma_10_close = data['close'].rolling(window=10).mean()
    sma_30_close = data['close'].rolling(window=30).mean()

    # Calculate 10-Day and 30-Day Price Change
    price_change_10 = data['close'] - sma_10_close
    price_change_30 = data['close'] - sma_30_close

    # Calculate Price Momentum Score
    price_momentum_score = price_change_10 + price_change_30

    # Calculate 10-day and 30-day moving averages of volume
    sma_10_volume = data['volume'].rolling(window=10).mean()
    sma_30_volume = data['volume'].rolling(window=30).mean()

    # Calculate Volume Momentum
    volume_momentum = sma_10_volume - sma_30_volume

    # Final Factor: Multiply Price Momentum Score by Volume Momentum
    final_factor = price_momentum_score * volume_momentum

    return final_factor
```
