```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Direction Indicator (SPDI)
    df['5_day_MA_close'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA_close']).astype(int)
    
    # Calculate Medium-Term Price Direction Indicator (MPDI)
    df['20_day_High'] = df['high'].rolling(window=20).max()
    df['20_day_Low'] = df['low'].rolling(window=20).min()
    df['20_day_Range'] = (df['20_day_High'] + df['20_day_Low']) / 2
    df['MPDI'] = (df['close'] > df['20_day_Range']).astype(int)
    
    # Calculate Volume Growth Factor (VGF)
    df['3_day_MA_volume'] = df['volume'].rolling(window=3).mean()
    df['VGF'] = (df['volume'] > df['3_day_MA_volume']).astype(int)
    
    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPDI'] + df['VGF']) / 3
    
    # Final Indicator Adjustment
    df['IMVTI_final'] = df['IMVTI'].apply(lambda x: 1 if x > 0.66 else (0.5 if x > 0.33 else 0))
    
    return df['IMVTI_final']
```
