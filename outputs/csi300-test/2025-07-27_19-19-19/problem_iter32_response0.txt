```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (e.g., 10 days)
    N = 10
    df['Accumulated_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum()
    df['Average_Price_Change'] = df['Accumulated_Price_Change'] / N
    
    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Close_Open'] = df['Close_Open_Diff'] * (df['volume'] / avg_volume)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_HL_VO'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']
    df['Sum_Combined_HL_VO'] = df['Combined_HL_VO'].rolling(window=N).sum()
    
    # Calculate Short-Term Price Direction (SPDI)
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA']).astype(int)
    
    # Calculate Medium-Term Price Range (MPR)
    df['20_day_MH'] = df['high'].rolling(window=20).max()
    df['20_day_ML'] = df['low'].rolling(window=20).min()
    mid_range = (df['20_day_MH'] + df['20_day_ML']) / 2
    df['MPR'] = (df['close'] > mid_range).astype(int)
    
    # Calculate Volume Growth (VGF)
    df['5_day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['VGF'] = (df['volume'] > df['5_day_Avg_Volume']).astype(int)
    
    # Integrate Indicators
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Generate Intermediate Alpha Scores
    scaling_factor_1 = 1.2
    df['High_Low_Momentum_Score'] = df['Sum_Combined_HL_VO'] * scaling_factor_1
    
    # Volume Weighted Average
    volume_price_product = df['volume'] * df['close']
    df['Volume_Weighted_Close'] = volume_price_product / df['volume'].rolling(window=20).sum()
    
    scaling_factor_2 = 0.6
    df['Volume_Weighted_Score'] = df['Volume_Weighted_Close'] * scaling_factor_2
    
    scaling_factor_3 = 1.8
    df['Momentum_Score'] = df['Average_Price_Change'] * scaling_factor_3
    
    # Combine Price and Volume Factors
    df['Combined_MV_Score'] = df['Momentum_Score'] + df['Volume_Weighted_Score']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['High_Low_Momentum_Score'] + df['Combined_MV_Score'] + df['IMVTI']
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'].ewm(alpha=0.45, adjust=False).mean()
    
    # Incorporate Long-Term Trend
    df['100_day_MA'] = df['close'].rolling(window=100).mean()
    long_term_trend_adjustment = (df['close'] > df['100_day_MA']).astype(float) * 0.1 - 0.1
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * (1 + long_term_trend_adjustment)
    
    return df['Final_Alpha_Factor']
```
