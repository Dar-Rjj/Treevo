```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Aggregate Price Change Over a Period (N=10)
    N = 10
    df['Accumulated_Price_Change'] = df['Daily_Price_Change'].rolling(window=N).sum()
    df['Average_Price_Change'] = df['Accumulated_Price_Change'] / N
    
    # Calculate Volume Adjusted Close-Open Difference
    df['Close_Open_Diff'] = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['Volume_Adjusted_Close_Open'] = df['Close_Open_Diff'] * (df['volume'] / avg_volume_20)
    
    # Combine High-Low Range and Volume Adjusted Close-Open Difference
    df['Combined_High_Low_Volume'] = df['High_Low_Range'] * df['Volume_Adjusted_Close_Open']
    df['Sum_Combined_High_Low_Volume'] = df['Combined_High_Low_Volume'].rolling(window=N).sum()
    
    # Generate Intermediate Alpha Scores
    scaling_factor_hl = 1.2
    scaling_factor_momentum = 1.8
    df['High_Low_Momentum_Score'] = df['Sum_Combined_High_Low_Volume'] * scaling_factor_hl
    df['Momentum_Score'] = df['Average_Price_Change'] * scaling_factor_momentum
    
    # Integrate Indicators
    df['5_day_MA'] = df['close'].rolling(window=5).mean()
    df['SPDI'] = (df['close'] > df['5_day_MA']).astype(int)
    
    df['20_day_High'] = df['high'].rolling(window=20).max()
    df['20_day_Low'] = df['low'].rolling(window=20).min()
    df['MPR'] = ((df['close'] - df['20_day_Low']) > (df['20_day_High'] - df['close'])).astype(int)
    
    df['5_day_Volume_MA'] = df['volume'].rolling(window=5).mean()
    df['VGF'] = (df['volume'] > df['5_day_Volume_MA']).astype(int)
    
    df['IMVTI'] = (df['SPDI'] + df['MPR'] + df['VGF']) / 3
    
    # Combine Price and Volume Factors
    df['Combined_Price_Volume_Factors'] = df['Momentum_Score'] + df['High_Low_Momentum_Score']
    
    # Generate Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Price_Volume_Factors'] + df['IMVTI']
    
    # Incorporate Long-Term Trend
    df['100_day_MA'] = df['close'].rolling(window=100).mean()
    df['Final_Alpha_Factor'] = df['Final_Alpha_Factor'] * (1 + (df['close'] > df['100_day_MA']).astype(int) * 0.1 - (df['close'] <= df['100_day_MA']).astype(int) * 0.1)
    
    return df['Final_Alpha_Factor'].dropna()
```
