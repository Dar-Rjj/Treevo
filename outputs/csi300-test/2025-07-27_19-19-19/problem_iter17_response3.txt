```python
import pandas as pd

def heuristics_v2(df):
    """
    This function calculates a novel alpha factor based on the given DataFrame.
    The alpha factor is a combination of different market features and aims to predict future stock returns.
    
    :param df: A pandas DataFrame with (date) as index and columns (open, high, low, close, amount, volume).
    :return: A pandas Series representing the factor values, indexed by (date).
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the exponential moving average of the money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the momentum of the closing price
    momentum_close = df['close'].pct_change(periods=10)
    
    # Calculate the relative strength of the closing price
    rs_close = (df['close'] - df['close'].rolling(window=14).min()) / (df['close'].rolling(window=14).max() - df['close'].rolling(window=14).min())
    
    # Combine the factors into a single alpha factor
    alpha_factor = (ema_money_flow.rank(pct=True) + momentum_close.rank(pct=True) + rs_close.rank(pct=True)) / 3
    
    return alpha_factor
```
