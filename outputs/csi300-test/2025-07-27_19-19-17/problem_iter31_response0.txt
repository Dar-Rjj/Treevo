```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['Prev_Close'] = df['close'].shift(1)
    df['High_Low'] = df['high'] - df['low']
    df['High_PrevClose'] = (df['high'] - df['Prev_Close']).abs()
    df['Low_PrevClose'] = (df['low'] - df['Prev_Close']).abs()
    df['TR'] = df[['High_Low', 'High_PrevClose', 'Low_PrevClose']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday High-Low Spread
    df['Intraday_HighLow_Spread'] = df['high'] - df['low']

    # Calculate Open-to-Close Return
    df['Open_to_Close_Return'] = df['close'] - df['open']

    # Integrate Returns with weights
    intraday_weight = 0.5
    close_open_weight = 0.5
    df['Integrated_Returns'] = (intraday_weight * df['Intraday_HighLow_Spread'] + 
                                close_open_weight * df['Open_to_Close_Return'])

    # Smooth the Integrated Returns using EMA (window size = 5 days)
    df['Smoothed_Integrated_Returns'] = df['Integrated_Returns'].ewm(span=5, adjust=False).mean()

    # Adjust for Volume and Amount Confirmation
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Adjusted_Integrated_Returns'] = df['Smoothed_Integrated_Returns'] * df['Volume_Ratio'] * df['Amount_Ratio']

    # Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Price_Momentum'] = df['SMA_10'] - df['SMA_50']

    # Incorporate ATR in Integrated Returns
    df['Adjusted_ATR'] = df['ATR'] * df['Adjusted_Integrated_Returns']

    # Adjust with Volume Factor
    df['Volume_Diff'] = df['volume'] - df['volume'].shift(1)
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Volume_Factor'] = df['Volume_Diff'] / df['Avg_Volume_20']
    df['Adjusted_ATR_Volume'] = df['Adjusted_ATR'] + df['Volume_Factor']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['Integrated_Momentum'] = df['Intraday_HighLow_Spread'] * df['Open_to_Close_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['Combined_Factor'] = (df['Adjusted_Integrated_Returns'] + 
                             df['Adjusted_ATR_Volume'] + 
                             df['Price_Momentum'] + 
                             df['Smoothed_Integrated_Momentum'])

    # Confirm with Volume
    df['Max_Volume_10'] = df['volume'].rolling(window=10).max()
    df['Volume_Confirmed_Factor'] = (df['Combined_Factor'] / np.sqrt(df['volume'])) * (df['volume'] / df['Max_Volume_10'])

    return df['Volume_Confirmed_Factor']
```
