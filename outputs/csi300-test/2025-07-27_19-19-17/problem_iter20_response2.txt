```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR) for each day
    df['Previous Close'] = df['close'].shift(1)
    df['TR'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x['high'] - x['low'], 
                      abs(x['high'] - x['Previous Close']), 
                      abs(x['low'] - x['Previous Close'])), axis=1
    )
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrate Returns with weights
    weight_intraday = 0.6
    weight_close_open = 0.4
    df['Integrated_Returns'] = (weight_intraday * df['Intraday_High_Low_Spread'] + 
                                weight_close_open * df['Close_Open_Return'])
    
    # Smooth the Integrated Returns using EMA
    df['Smoothed_Integrated_Returns'] = df['Integrated_Returns'].ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount Confirmation
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Adjusted_Integrated_Returns'] = df['Smoothed_Integrated_Returns']
    
    # If both Volume and Amount conditions are met
    df.loc[(df['volume'] > df['volume'].shift(1)) & (df['amount'] > df['amount'].shift(1)), 
           'Adjusted_Integrated_Returns'] *= df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Price Momentum
    df['Short_Term_SMA'] = df['close'].rolling(window=10).mean()
    df['Long_Term_SMA'] = df['close'].rolling(window=50).mean()
    df['SMA_Difference'] = df['Short_Term_SMA'] - df['Long_Term_SMA']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Smoothed_Integrated_Returns'] + df['Adjusted_Integrated_Returns']) * \
                               (df['SMA_Difference'] * df['ATR'])
    
    return df['Final_Alpha_Factor']
```
