```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['TR'] = np.maximum(
        np.maximum(
            df['high'] - df['low'],
            np.abs(df['high'] - df['close'].shift(1))
        ),
        np.abs(df['low'] - df['close'].shift(1))
    )

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']

    # Calculate Open-to-Close Return
    df['Open_to_Close_Return'] = df['close'] - df['open']

    # Integrate Returns
    df['Integrated_Returns'] = 0.5 * df['Intraday_Spread'] + 0.5 * df['Open_to_Close_Return']

    # Smooth the Integrated Returns using EMA with a 5-day window
    df['Smoothed_Integrated_Returns'] = df['Integrated_Returns'].ewm(span=5, adjust=False).mean()

    # Adjust for Volume and Amount Confirmation
    df['Volume_Adjusted_Returns'] = df['Smoothed_Integrated_Returns']
    df.loc[df['volume'] > df['volume'].shift(1), 'Volume_Adjusted_Returns'] *= (df['volume'] / df['volume'].shift(1))
    df.loc[df['amount'] > df['amount'].shift(1), 'Volume_Adjusted_Returns'] *= (df['amount'] / df['amount'].shift(1))
    df.loc[(df['volume'] > df['volume'].shift(1)) & (df['amount'] > df['amount'].shift(1)), 'Volume_Adjusted_Returns'] *= (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))

    # Price Momentum
    df['Short_term_SMA'] = df['close'].rolling(window=10).mean()
    df['Long_term_SMA'] = df['close'].rolling(window=50).mean()
    df['Price_Momentum'] = df['Short_term_SMA'] - df['Long_term_SMA']

    # Incorporate ATR in Integrated Returns
    df['ATR_Adjusted_Returns'] = df['ATR'] * df['Volume_Adjusted_Returns']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['Integrated_Momentum'] = df['Intraday_Spread'] * df['Open_to_Close_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['Combined_Factor'] = df['Volume_Adjusted_Returns'] + df['ATR_Adjusted_Returns'] + df['Price_Momentum'] + df['Smoothed_Integrated_Momentum']

    # Confirm with Volume
    max_volume = df['volume'].rolling(window=10).max()
    df['Final_Alpha_Factor'] = df['Combined_Factor'] / np.sqrt(df['volume']) * (df['volume'] / max_volume)

    return df['Final_Alpha_Factor']
```
