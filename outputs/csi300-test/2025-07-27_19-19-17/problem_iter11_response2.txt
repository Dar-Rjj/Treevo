```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['Integrated_Returns'] = df['Intraday_High_Low_Spread'] + df['Close_Open_Return']
    
    # Adjust for Volume and Amount Confirmation
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Adjusted_Integrated_Returns'] = df['Integrated_Returns'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Calculate Average True Range (ATR) with Intraday Return
    df['True_Range'] = df[['high', 'low', df['close'].shift(1)]].max(axis=1) - df[['high', 'low', df['close'].shift(1)]].min(axis=1)
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Incorporate Intraday and Open-to-Close Returns
    df['Intraday_Return'] = df['high'] - df['low']
    df['Open_Close_Return'] = df['close'] - df['open']
    df['Integrated_Returns'] = df['Intraday_Return'] + df['Open_Close_Return']
    
    # Combine ATR, Integrated Returns, and Volume Adjustment
    df['ATR_Adjusted_Return'] = df['ATR_14'] * df['Integrated_Returns']
    df['Volume_Diff'] = df['volume'] - df['volume'].shift(1)
    df['Avg_Volume_20'] = df['volume'].rolling(window=20).mean()
    df['Normalized_Volume_Diff'] = df['Volume_Diff'] / df['Avg_Volume_20']
    df['ATR_Adjusted_Return'] += df['Normalized_Volume_Diff']
    
    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Price_Momentum'] = df['SMA_10'] - df['SMA_50']
    df['Price_Momentum_Adjusted'] = df['Price_Momentum'] * df['ATR_Adjusted_Return']
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Integrated_Momentum'] = df['Intraday_High_Low_Spread'] * df['Open_Close_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    df['Combined_Factor'] = df['ATR_Adjusted_Return'] + df['Smoothed_Integrated_Momentum']
    df['Volume_Adjustment'] = df['volume'] / df['volume'].rolling(window=10).max()
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['Volume_Adjustment']
    
    return df['Final_Alpha_Factor']
```
