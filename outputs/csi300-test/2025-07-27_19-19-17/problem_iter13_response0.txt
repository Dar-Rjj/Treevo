```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['prev_close'])
    df['tr3'] = abs(df['low'] - df['prev_close'])
    df['tr'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['atr'] = df['tr'].rolling(window=14).mean()

    # Calculate Integrated Return
    df['integrated_return'] = (df['high'] - df['low']) + (df['close'] - df['open'])

    # Weighted Integrated Return by Volume
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['weighted_integrated_return_vol'] = 0
    df.loc[df['volume'] > df['volume'].shift(1), 'weighted_integrated_return_vol'] = \
        df['integrated_return'] * (df['volume'] / df['avg_volume_10'])

    # Weighted Integrated Return by Amount
    df['avg_amount_10'] = df['amount'].rolling(window=10).mean()
    df['weighted_integrated_return_amt'] = 0
    df.loc[df['amount'] > df['amount'].shift(1), 'weighted_integrated_return_amt'] = \
        df['integrated_return'] * (df['amount'] / df['avg_amount_10'])

    # Price Momentum: Short-term and Long-term SMAs
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_diff'] = df['sma_10'] - df['sma_50']

    # Final Alpha Factor
    df['final_alpha_factor'] = (df['weighted_integrated_return_vol'] + df['weighted_integrated_return_amt']) * (df['sma_diff'] * df['atr'])

    return df['final_alpha_factor']
```
