```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_return = intraday_return + open_to_close_return
    
    # Confirm with Volume
    volume_weight = (df['volume'] > df['volume'].shift(1)) * (df['volume'] / df['volume'].rolling(window=10).mean())
    weighted_integrated_return = integrated_return * volume_weight
    
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Integrate High-Low and Close-Open for Momentum
    integrated_momentum = high_low_spread * open_to_close_return
    smoothed_integrated_momentum = integrated_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Factor
    final_factor = weighted_integrated_return + smoothed_integrated_momentum
    final_factor /= df['volume']
    
    return final_factor
```
