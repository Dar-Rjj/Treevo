```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_return = intraday_return + open_to_close_return
    
    # Confirm with Volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).mean()
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Apply Weight
    factor_values = integrated_return * volume_ratio * volume_confirmation
    
    return factor_values
```
