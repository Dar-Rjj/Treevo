```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['Integrated_Returns'] = df['Intraday_HL_Spread'] + df['Close_Open_Return']
    
    # Smooth the Integrated Returns using EMA
    df['Smoothed_Integrated_Returns'] = df['Integrated_Returns'].ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount
    avg_volume = df['volume'].rolling(window=10).mean()
    avg_amount = df['amount'].rolling(window=10).mean()
    df['Volume_Adjustment'] = df['volume'] / avg_volume
    df['Amount_Adjustment'] = df['amount'] / avg_amount
    df['Adjusted_Integrated_Returns'] = (df['Smoothed_Integrated_Returns'] / (df['volume'] * df['amount'])) * df['Volume_Adjustment'] * df['Amount_Adjustment']
    
    # Calculate Average True Range (ATR)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = abs(df['high'] - df['close'].shift(1))
    df['TR3'] = abs(df['low'] - df['close'].shift(1))
    df['True_Range'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Incorporate Volume and Amount Factors
    df['Volume_Diff'] = df['volume'] - df['volume'].shift(1)
    df['Amount_Diff'] = df['amount'] - df['amount'].shift(1)
    df['Volume_Amount_Diff'] = df['Volume_Diff'] + df['Amount_Diff']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    avg_amount_20 = df['amount'].rolling(window=20).mean()
    df['Normalized_Volume_Amount_Diff'] = df['Volume_Amount_Diff'] / (avg_volume_20 + avg_amount_20)
    
    # Combine Smoothed Integrated Returns with ATR and Volume-Amount Factor
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_Diff'] = df['SMA_10'] - df['SMA_50']
    df['Factor'] = df['SMA_Diff'] * df['ATR'] + df['Normalized_Volume_Amount_Diff'] * df['Adjusted_Integrated_Returns']
    
    return df['Factor']
```
