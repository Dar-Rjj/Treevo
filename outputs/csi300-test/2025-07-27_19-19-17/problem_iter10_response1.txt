```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['Integrated_Returns'] = df['Intraday_High_Low_Spread'] + df['Close_Open_Return']

    # Adjust for Volume and Amount Confirmation
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Check if volume and amount are greater than the previous day's
    df['Volume_Confirmation'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    df['Amount_Confirmation'] = (df['amount'] > df['amount'].shift(1)).astype(int)
    
    # Adjust Integrated Returns
    df['Adjusted_Integrated_Returns'] = (df['Integrated_Returns'] * 
                                         df['Volume_Ratio'] * 
                                         df['Amount_Ratio'] * 
                                         df['Volume_Confirmation'] * 
                                         df['Amount_Confirmation'])
    
    # Return the final factor values
    return df['Adjusted_Integrated_Returns']
```
