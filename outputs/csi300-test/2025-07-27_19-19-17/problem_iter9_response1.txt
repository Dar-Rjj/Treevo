```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['close_open_return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['integrated_returns'] = df['high_low_spread'] + df['close_open_return']
    
    # Smooth the Integrated Returns using EMA
    df['ema_returns'] = df['integrated_returns'].ewm(span=5, adjust=False).mean()
    
    # Volume Confirmation
    df['volume_increase'] = df['volume'] > df['volume'].shift(1)
    df['max_volume_10d'] = df['volume'].rolling(window=10).max()
    df['volume_adjusted_returns'] = (df['ema_returns'] / np.sqrt(df['volume'])) * (df['volume'] / df['max_volume_10d'])
    
    # Weighting Factor
    df['weighting_factor'] = df['amount'] / df['volume']
    
    # Apply Weighting Factor and Final Adjustments
    df['final_returns'] = (df['volume_adjusted_returns'] * df['weighting_factor']) / np.sqrt(df['volume']) * (df['volume'] / df['max_volume_10d'])
    
    return df['final_returns']
```
