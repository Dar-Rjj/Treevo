```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = (df['close'] - df['open']) / df['open']
    
    # Integrate Returns
    integrated_return = intraday_return + open_to_close_return
    
    # Confirm with Volume and Amount
    volume_condition = df['volume'] > df['volume'].shift(1)
    amount_condition = df['amount'] > df['amount'].shift(1)
    
    # Apply Weight
    avg_volume = df['volume'].rolling(window=10).mean()
    avg_amount = df['amount'].rolling(window=10).mean()
    weight = (df['volume'] / avg_volume) * (df['amount'] / avg_amount)
    
    factor_values = integrated_return * weight * volume_condition * amount_condition
    
    return factor_values
```
