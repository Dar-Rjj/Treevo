```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Smooth the Integrated Returns using EMA
    ema_window = 5
    smoothed_integrated_returns = integrated_returns.ewm(span=ema_window, adjust=False).mean()
    
    # Adjust for Volume and Amount
    volume_adjustment = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))
    adjusted_smoothed_returns = smoothed_integrated_returns * volume_adjustment
    
    # Calculate Average True Range (ATR)
    true_range = np.maximum(np.maximum(df['high'] - df['low'], df['high'] - df['close'].shift(1)), df['close'].shift(1) - df['low'])
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate Volume and Amount Factors
    volume_difference = df['volume'] - df['volume'].shift(1)
    amount_difference = df['amount'] - df['amount'].shift(1)
    volume_amount_factor = volume_difference + amount_difference
    
    # Combine Smoothed Integrated Returns with ATR and Volume-Amount Factor
    combined_factor = (adjusted_smoothed_returns * atr) + volume_amount_factor
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    sma_difference = short_term_sma - long_term_sma
    price_momentum = sma_difference * atr
    
    # Final Alpha Factor
    final_alpha_factor = combined_factor + price_momentum
    
    return final_alpha_factor
```
