```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = 0.5 * intraday_high_low_spread + 0.5 * close_open_return
    
    # Adjust for Volume and Amount Confirmation
    volume_diff = df['volume'].diff().fillna(0)
    amount_diff = df['amount'].diff().fillna(0)
    volume_adjustment = (df['volume'] / df['volume'].shift(1).fillna(1)) * (df['amount'] / df['amount'].shift(1).fillna(1))
    integrated_returns *= volume_adjustment
    
    # Calculate True Range (TR) for each day
    tr = pd.Series(0, index=df.index)
    tr = tr.combine(df['high'] - df['low'], max)
    tr = tr.combine(df['high'] - df['close'].shift(1).fillna(0), max)
    tr = tr.combine(df['low'] - df['close'].shift(1).fillna(0), max)
    
    # Calculate Average True Range (ATR)
    atr = tr.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    adjusted_atr = integrated_returns * atr
    
    # Adjust with Volume Factor
    avg_volume = df['volume'].rolling(window=20).mean()
    normalized_volume_diff = volume_diff / avg_volume
    adjusted_atr += normalized_volume_diff
    
    # Calculate Price Momentum
    short_sma = df['close'].rolling(window=10).mean()
    long_sma = df['close'].rolling(window=50).mean()
    sma_difference = short_sma - long_sma
    price_momentum = sma_difference * adjusted_atr
    
    # Integrate High-Low and Close-Open for Additional Momentum
    additional_momentum = intraday_high_low_spread * close_open_return
    smoothed_momentum = additional_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    combined_factor = adjusted_atr + price_momentum + smoothed_momentum
    max_volume_10_days = df['volume'].rolling(window=10).max()
    volume_adjusted_factor = combined_factor * (df['volume'] / max_volume_10_days) / np.sqrt(df['volume'])
    
    # Confirm with Volume
    volume_confirmation = (df['volume'] > df['volume'].shift(1)).astype(int)
    final_alpha_factor = volume_adjusted_factor * volume_confirmation
    
    return final_alpha_factor
```
