```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Adjust for Volume and Amount Confirmation
    volume_adjustment = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))
    adjusted_integrated_returns = integrated_returns * volume_adjustment
    
    # Calculate Average True Range (ATR)
    true_range = pd.concat([
        df['high'] - df['low'],
        abs(df['high'] - df['close'].shift(1)),
        abs(df['low'] - df['close'].shift(1))
    ], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate Intraday and Open-to-Close Returns
    intraday_return = intraday_high_low_spread
    open_to_close_return = close_open_return
    integrated_returns_atr = (intraday_return + open_to_close_return) * atr
    
    # Combine ATR, Integrated Returns, and Volume Adjustment
    combined_factor = integrated_returns_atr * adjusted_integrated_returns
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    sma_difference = short_term_sma - long_term_sma
    price_momentum = sma_difference * atr
    
    # Integrate High-Low and Close-Open for Additional Momentum
    integrated_momentum = intraday_high_low_spread * open_to_close_return
    smoothed_momentum = integrated_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    final_factor = (combined_factor + price_momentum + smoothed_momentum) / np.sqrt(df['volume'])
    final_factor = final_factor * (df['volume'] / df['volume'].rolling(window=10).max())
    
    return final_factor
```
