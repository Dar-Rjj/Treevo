```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Adjust for Volume and Amount Confirmation
    volume_adjustment = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))
    adjusted_integrated_returns = integrated_returns * volume_adjustment
    
    # Calculate Average True Range (ATR) with Intraday Return
    true_range = pd.concat([
        df['high'] - df['low'],
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ], axis=1).max(axis=1)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_return = atr * adjusted_integrated_returns
    
    # Adjust with Volume Factor
    volume_diff = df['volume'] - df['volume'].shift(1)
    avg_volume = df['volume'].rolling(window=20).mean()
    normalized_volume_diff = volume_diff / avg_volume
    atr_adjusted_return += normalized_volume_diff
    
    # Calculate Price Momentum
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=50).mean()
    price_momentum = (sma_short - sma_long) * atr
    
    # Integrate High-Low and Close-Open for Additional Momentum
    additional_momentum = intraday_high_low_spread * close_open_return
    smoothed_momentum = additional_momentum.ewm(span=5).mean()
    
    # Final Alpha Factor
    combined_factor = atr_adjusted_return + price_momentum + smoothed_momentum
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).max()
    final_alpha_factor = combined_factor * volume_ratio / np.sqrt(df['volume'])
    
    return final_alpha_factor
```
