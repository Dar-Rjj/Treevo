```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Adjust for Volume and Amount Confirmation
    volume_factor = df['volume'] / df['volume'].shift(1)
    amount_factor = df['amount'] / df['amount'].shift(1)
    adjusted_integrated_returns = integrated_returns * volume_factor * amount_factor
    
    # Calculate True Range (TR) for each day
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = atr * adjusted_integrated_returns
    
    # Adjust with Volume Factor
    volume_difference = df['volume'] - df['volume'].shift(1)
    average_volume = df['volume'].rolling(window=20).mean()
    normalized_volume_difference = volume_difference / average_volume
    atr_adjusted_returns_with_volume = atr_adjusted_returns + normalized_volume_difference
    
    # Calculate Price Momentum
    sma_short = df['close'].rolling(window=10).mean()
    sma_long = df['close'].rolling(window=50).mean()
    price_momentum = sma_short - sma_long
    momentum_adjusted_returns = price_momentum * atr_adjusted_returns_with_volume
    
    # Integrate High-Low and Close-Open for Additional Momentum
    intraday_high_low_spread = df['high'] - df['low']
    open_to_close_return = df['close'] - df['open']
    integrated_momentum = intraday_high_low_spread * open_to_close_return
    smoothed_integrated_momentum = integrated_momentum.ewm(span=7).mean()
    
    # Final Alpha Factor
    combined_factor = (atr_adjusted_returns_with_volume +
                       momentum_adjusted_returns +
                       smoothed_integrated_momentum)
    
    # Confirm with Volume
    volume_condition = (df['volume'] > df['volume'].shift(1))
    max_volume_last_15_days = df['volume'].rolling(window=15).max()
    final_alpha_factor = combined_factor.divide(np.sqrt(df['volume'])) * (df['volume'] / max_volume_last_15_days)
    final_alpha_factor = final_alpha_factor.where(volume_condition, 0)
    
    # Enhanced Final Alpha Factor
    open_price_trend = df['open'] - df['open'].rolling(window=10).mean()
    enhanced_final_alpha_factor = (momentum_adjusted_returns +
                                   smoothed_integrated_momentum +
                                   open_price_trend)
    enhanced_final_alpha_factor = enhanced_final_alpha_factor.where(volume_condition, 0)
    
    return enhanced_final_alpha_factor
```
