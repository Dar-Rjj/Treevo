```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = abs(df['high'] - df['prev_close'])
    df['low_prev_close'] = abs(df['low'] - df['prev_close'])
    df['tr'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Calculate Intraday and Open-to-Close Returns
    df['intraday_return'] = df['high'] - df['low']
    df['open_to_close_return'] = df['close'] - df['open']
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Adjust with Volume Factor
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['adjusted_volume_diff'] = np.where(df['volume'] > df['volume'].shift(1), 
                                           df['volume_diff'] / df['avg_volume_20'], 
                                           0)
    
    # Combine ATR, Integrated Returns, and Volume Adjustment
    df['weighted_atr'] = df['atr'] * df['integrated_return'] + df['adjusted_volume_diff']
    
    # Calculate Price Momentum
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_diff'] = df['sma_10'] - df['sma_50']
    df['price_momentum'] = df['sma_diff'] * df['weighted_atr']
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['integrated_momentum'] = df['intraday_high_low_spread'] * df['open_to_close_return']
    df['smoothed_integrated_momentum'] = df['integrated_momentum'].ewm(span=5).mean()
    
    # Final Alpha Factor
    df['combined_factor'] = df['weighted_atr'] + df['smoothed_integrated_momentum']
    df['sqrt_volume'] = np.sqrt(df['volume'])
    df['max_volume_10'] = df['volume'].rolling(window=10).max()
    df['final_alpha_factor'] = (df['combined_factor'] / df['sqrt_volume']) * (df['volume'] / df['max_volume_10']) + df['weighted_atr'] + df['price_momentum']
    
    return df['final_alpha_factor']
```
