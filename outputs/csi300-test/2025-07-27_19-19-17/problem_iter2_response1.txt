```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['TR'] = df[['High', 'Low']].apply(lambda x: max(x['High'] - x['Low'], 
                                                        abs(x['High'] - df['Close'].shift(1)), 
                                                        abs(x['Low'] - df['Close'].shift(1))), axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Calculate Intraday Return
    df['Intraday_Return'] = df['High'] - df['Low']
    
    # Calculate Open-to-Close Return
    df['Open_to_Close_Return'] = df['Close'] - df['Open']
    
    # Integrate Returns
    df['Integrated_Return'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    
    # Confirm with Volume
    average_volume = df['Volume'].rolling(window=10).mean()
    df['Volume_Weight'] = df['Volume'] / average_volume
    df['Volume_Weighted_Return'] = df['Integrated_Return'] * df['Volume_Weight']
    
    # Combine with Price Momentum
    df['SMA_10'] = df['Close'].rolling(window=10).mean()
    df['SMA_50'] = df['Close'].rolling(window=50).mean()
    df['Momentum'] = (df['SMA_10'] - df['SMA_50']) * df['ATR']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Volume_Weighted_Return'] * df['Momentum']
    
    return df['Alpha_Factor']
```
