```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range (TR) for each day
    df['Prev_Close'] = df['close'].shift(1)
    df['High_Low'] = df['high'] - df['low']
    df['High_Prev_Close'] = abs(df['high'] - df['Prev_Close'])
    df['Low_Prev_Close'] = abs(df['low'] - df['Prev_Close'])
    df['TR'] = df[['High_Low', 'High_Prev_Close', 'Low_Prev_Close']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday Return
    df['Intraday_Return'] = df['high'] - df['low']

    # Calculate Open-to-Close Return
    df['Open_to_Close_Return'] = df['close'] - df['open']

    # Integrate Returns
    df['Integrated_Return'] = df['Intraday_Return'] + df['Open_to_Close_Return']

    # Confirm with Volume
    df['Volume_Confirmation'] = df['volume'] > df['volume'].shift(1)
    df['10_day_Avg_Volume'] = df['volume'].rolling(window=10).mean()
    df['Weighted_Integrated_Return'] = df['Integrated_Return'] * (df['volume'] / df['10_day_Avg_Volume'])
    df['Weighted_Integrated_Return'] = df['Weighted_Integrated_Return'] * df['Volume_Confirmation']

    # Combine with Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_Difference'] = df['SMA_10'] - df['SMA_50']

    # Final Alpha Factor
    df['Alpha_Factor'] = df['Weighted_Integrated_Return'] * (df['SMA_Difference'] * df['ATR'])

    return df['Alpha_Factor']
```
