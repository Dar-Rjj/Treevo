```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['Previous Close'] = df['close'].shift(1)
    df['True Range'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x[0] - x[1], abs(x[0] - x[2]), abs(x[1] - x[2])), axis=1
    )
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['True Range'].rolling(window=14).mean()
    
    # Calculate Intraday Return and Open-to-Close Return
    df['Intraday Return'] = df['high'] - df['low']
    df['Open-to-Close Return'] = df['close'] - df['open']
    df['Integrated Return'] = df['Intraday Return'] + df['Open-to-Close Return']
    
    # Volume and Amount Adjustments
    df['Volume Adjustment'] = np.where(
        df['volume'] > df['volume'].shift(1),
        df['Integrated Return'] * (df['volume'] / df['volume'].rolling(window=10).mean()),
        0
    )
    df['Amount Adjustment'] = np.where(
        df['amount'] > df['amount'].shift(1),
        df['Integrated Return'] * (df['amount'] / df['amount'].rolling(window=10).mean()),
        0
    )
    
    # Combine ATR, Integrated Returns, and Volume/Amount Adjustments
    df['Adjusted ATR'] = df['ATR'] * df['Integrated Return']
    df['Combined Factor'] = df['Adjusted ATR'] + df['Volume Adjustment'] + df['Amount Adjustment']
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['Intraday High-Low Spread'] = df['high'] - df['low']
    df['Integrated Momentum'] = df['Intraday High-Low Spread'] * df['Open-to-Close Return']
    df['Smoothed Integrated Momentum'] = df['Integrated Momentum'].ewm(span=5).mean()
    
    # Final Alpha Factor
    df['Final Alpha Factor'] = df['Combined Factor'] + df['Smoothed Integrated Momentum']
    
    # Confirm with Volume and Amount
    df['Final Alpha Factor'] = np.where(
        df['volume'] > df['volume'].shift(1),
        df['Final Alpha Factor'] / np.sqrt(df['volume']) * (df['volume'] / df['volume'].rolling(window=10).max()),
        df['Final Alpha Factor']
    )
    
    return df['Final Alpha Factor']
```
