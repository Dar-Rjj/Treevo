```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['Prev_Close'] = df['close'].shift(1)
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0] - x[1], abs(x[0] - df['Prev_Close'][x.name]), abs(x[1] - df['Prev_Close'][x.name])), axis=1)

    # Calculate Average True Range (ATR) - 14-day average TR
    df['ATR_14'] = df['TR'].rolling(window=14).mean()

    # Calculate Integrated Return
    df['Intraday_Spread'] = df['high'] - df['low']
    df['Close_Open_Return'] = df['close'] - df['open']
    df['Integrated_Return'] = 0.5 * df['Intraday_Spread'] + 0.5 * df['Close_Open_Return']

    # Adjust and Smooth Integrated Return
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)

    conditions = [
        (df['volume'] > df['volume'].shift(1)) & (df['amount'] > df['amount'].shift(1)),
        (df['volume'] > df['volume'].shift(1)),
        (df['amount'] > df['amount'].shift(1))
    ]
    choices = [
        df['Volume_Ratio'] * df['Amount_Ratio'],
        df['Volume_Ratio'],
        df['Amount_Ratio']
    ]
    df['Adjusted_Integrated_Return'] = df['Integrated_Return'] * np.select(conditions, choices, default=1)
    
    # Apply Exponential Moving Average (EMA) on Adjusted Integrated Return
    df['Smoothed_Adjusted_IR'] = df['Adjusted_Integrated_Return'].ewm(span=5, adjust=False).mean()

    # Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Momentum_Diff'] = df['SMA_10'] - df['SMA_50']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (df['Smoothed_Adjusted_IR'] + df['Integrated_Return']) * (df['Momentum_Diff'] * df['ATR_14'])
    df['Final_Alpha_Factor'] += (df['volume'] - df['volume'].shift(1)) + (df['amount'] - df['amount'].shift(1))

    return df['Final_Alpha_Factor'].dropna()
```
