```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Smooth the Integrated Returns using EMA
    ema_window = 5
    smoothed_integrated_returns = integrated_returns.ewm(span=ema_window).mean()
    
    # Adjust for Volume and Amount
    volume_ratio = df['volume'] / df['volume'].shift(1)
    amount_ratio = df['amount'] / df['amount'].shift(1)
    volume_amount_adjusted_returns = smoothed_integrated_returns * volume_ratio * amount_ratio
    
    # Calculate True Range (TR) for each day
    high_prev_close = df['high'] - df['close'].shift(1)
    low_prev_close = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([df['high'] - df['low'], high_prev_close, low_prev_close], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    atr_window = 14
    atr = true_range.rolling(window=atr_window).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = volume_amount_adjusted_returns * atr
    
    # Adjust with Volume Factor
    volume_diff = df['volume'] - df['volume'].shift(1)
    avg_volume_window = 20
    avg_volume = df['volume'].rolling(window=avg_volume_window).mean()
    normalized_volume_diff = volume_diff / avg_volume
    volume_factor_adjusted_returns = atr_adjusted_returns + normalized_volume_diff
    
    # Calculate Price Momentum
    sma_short_term = df['close'].rolling(window=10).mean()
    sma_long_term = df['close'].rolling(window=50).mean()
    price_momentum = sma_short_term - sma_long_term
    adjusted_price_momentum = price_momentum * volume_factor_adjusted_returns
    
    # Integrate High-Low and Close-Open for Additional Momentum
    intraday_high_low_spread = df['high'] - df['low']
    open_to_close_return = df['close'] - df['open']
    integrated_momentum = intraday_high_low_spread * open_to_close_return
    smoothed_integrated_momentum = integrated_momentum.ewm(span=ema_window).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (volume_factor_adjusted_returns + 
                          adjusted_price_momentum + 
                          smoothed_integrated_momentum)
    
    # Confirm with Volume
    max_volume_window = 10
    max_volume = df['volume'].rolling(window=max_volume_window).max()
    volume_adjusted_factor = final_alpha_factor * (df['volume'] / max_volume)
    
    return volume_adjusted_factor
```
