```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return: (High - Low)
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return: (Close - Open)
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Integrate Returns: (Intraday Return + Open-to-Close Return)
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Confirm with Volume and Amount
    # Check Volume Increase: If Volume > Previous Day's Volume
    # Check Amount Increase: If Amount > Previous Day's Amount
    df['volume_increase'] = df['volume'] > df['volume'].shift(1)
    df['amount_increase'] = df['amount'] > df['amount'].shift(1)
    
    # Apply Weight: Multiply Integrated Return by (Current Volume / Previous Day's Volume) * (Current Amount / Previous Day's Amount)
    df['weight'] = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))
    df['weighted_return'] = df['integrated_return'] * df['weight']
    
    # Only keep the weighted return when both volume and amount increase
    df['heuristic_factor'] = df.apply(lambda row: row['weighted_return'] if row['volume_increase'] and row['amount_increase'] else 0, axis=1)
    
    return df['heuristic_factor']
```
