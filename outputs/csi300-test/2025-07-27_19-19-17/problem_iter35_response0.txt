```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns: Weighted Sum of Intraday High-Low Spread and Close-Open Return
    integrated_returns = 0.5 * intraday_spread + 0.5 * close_open_return
    
    # Smooth the Integrated Returns using EMA with a 14-day window
    smoothed_returns = integrated_returns.ewm(span=14, adjust=False).mean()
    
    # Adjust for Volume Confirmation
    volume_adjusted_returns = smoothed_returns.where(df['volume'] > df['volume'].shift(1), 
                                                     smoothed_returns * (df['volume'] / df['volume'].shift(1)))
    
    # Calculate True Range (TR) for each day
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 21-day window
    atr = true_range.rolling(window=21, min_periods=1).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = atr / volume_adjusted_returns
    
    # Calculate Price Momentum
    short_sma = df['close'].rolling(window=15, min_periods=1).mean()
    long_sma = df['close'].rolling(window=60, min_periods=1).mean()
    price_momentum = short_sma - long_sma
    
    # Final Alpha Factor: Combine Weighted Integrated Return, ATR, and Price Momentum
    combined_factor = 0.7 * atr_adjusted_returns + 0.3 * price_momentum
    
    # Confirm with Volume
    final_factor = combined_factor.where(df['volume'] > df['volume'].shift(1), 
                                         combined_factor / (df['volume'] / df['volume'].shift(1)))
    
    return final_factor
```
