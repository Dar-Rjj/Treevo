```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = abs(df['high'] - df['prev_close'])
    df['low_prev_close'] = abs(df['low'] - df['prev_close'])
    df['TR'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']

    # Calculate Open-to-Close Return
    df['open_close_return'] = df['close'] - df['open']

    # Integrate Returns
    df['integrated_return'] = df['intraday_return'] + df['open_close_return']

    # Smooth the Integrated Returns with EMA
    df['smoothed_integrated_return'] = df['integrated_return'].ewm(span=5, adjust=False).mean()

    # Confirm with Volume and Amount
    df['volume_confirmation'] = (df['volume'] > df['volume'].shift(1)) & (df['amount'] > df['amount'].shift(1))
    df['volume_adjustment'] = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))

    # Adjust integrated return with volume and amount
    df['volume_adjusted_return'] = df['smoothed_integrated_return'] * df['volume_adjustment']

    # Incorporate ATR in Integrated Returns
    df['atr_integrated_return'] = df['ATR'] * df['volume_adjusted_return']

    # Adjust with Volume Factor
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['normalized_volume_diff'] = df['volume_diff'] / df['avg_volume_20']
    df['adjusted_atr_integrated_return'] = df['atr_integrated_return'] + df['normalized_volume_diff']

    # Calculate Price Momentum
    df['SMA_short'] = df['close'].rolling(window=10).mean()
    df['SMA_long'] = df['close'].rolling(window=50).mean()
    df['price_momentum'] = df['SMA_short'] - df['SMA_long']

    # Multiply the SMA difference by the adjusted ATR
    df['price_momentum_adj'] = df['price_momentum'] * df['ATR']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['intraday_spread'] = df['high'] - df['low']
    df['intraday_momentum'] = df['intraday_spread'] * df['open_close_return']
    df['smoothed_intraday_momentum'] = df['intraday_momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['combined_factor'] = (df['adjusted_atr_integrated_return'] 
                             + df['price_momentum_adj'] 
                             + df['smoothed_intraday_momentum'])

    # Confirm with Volume
    df['final_alpha_factor'] = np.where(df['volume_confirmation'], 
                                        df['combined_factor'] / np.sqrt(df['volume']) * (df['volume'] / df['volume'].rolling(window=10).max()), 
                                        0)

    return df['final_alpha_factor']
```
