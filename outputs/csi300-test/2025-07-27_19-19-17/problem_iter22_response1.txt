```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute True Range (TR)
    df['TR'] = df[['High', 'Low']].apply(lambda x: max(x['High'] - x['Low'], 
                                                        abs(x['High'] - df['Close'].shift(1)), 
                                                        abs(x['Low'] - df['Close'].shift(1))), axis=1)
    
    # Average TR over 14 days to get ATR
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Integrated Return: Intraday Return + Open-to-Close Return
    df['Intraday_Return'] = df['High'] - df['Low']
    df['Open_to_Close_Return'] = df['Close'] - df['Open']
    df['Integrated_Return'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    
    # Volume Adjustment: Volume Difference / Average Volume (20 days)
    df['Volume_Difference'] = df['Volume'] - df['Volume'].shift(1)
    df['Avg_Volume_20'] = df['Volume'].rolling(window=20).mean()
    df['Volume_Adjustment'] = df['Volume_Difference'] / df['Avg_Volume_20']
    
    # Adjusted ATR: ATR * Integrated Return + Volume Adjustment
    df['Adjusted_ATR'] = df['ATR'] * df['Integrated_Return'] + df['Volume_Adjustment']
    
    # Calculate Price Momentum
    df['SMA_10'] = df['Close'].rolling(window=10).mean()
    df['SMA_50'] = df['Close'].rolling(window=50).mean()
    df['SMA_Difference'] = df['SMA_10'] - df['SMA_50']
    
    # Smoothed Integrated Momentum: EMA (Intraday High-Low Spread * Open-to-Close Return, 5 days)
    df['Intraday_High_Low_Spread'] = df['High'] - df['Low']
    df['Integrated_Momentum'] = df['Intraday_High_Low_Spread'] * df['Open_to_Close_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5).mean()
    
    # Combined Factor: Adjusted ATR + SMA Difference * Adjusted ATR
    df['Combined_Factor'] = df['Adjusted_ATR'] + df['SMA_Difference'] * df['Adjusted_ATR']
    
    # Confirm with Volume and Amount
    df['Volume_Check'] = df['Volume'] > df['Volume'].shift(1)
    df['Amount_Check'] = df['Amount'] > df['Amount'].shift(1)
    df['Final_Adjustment'] = df['Combined_Factor'] * (df['Volume'] / df['Volume'].shift(1)) * (df['Amount'] / df['Amount'].shift(1))
    
    # Final Alpha Factor: Combined Factor + Volume-Adjusted Momentum
    df['Final_Alpha_Factor'] = df['Combined_Factor'] + df['Smoothed_Integrated_Momentum'] * df['Final_Adjustment']
    
    return df['Final_Alpha_Factor'].dropna()
```
