```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = 0.5 * intraday_high_low_spread + 0.5 * open_to_close_return
    
    # Smooth the Integrated Returns using EMA
    smooth_integrated_returns = integrated_returns.ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume Confirmation
    volume_adjusted_returns = np.where(
        df['volume'] > df['volume'].shift(1),
        smooth_integrated_returns * (df['volume'] / df['volume'].shift(1)),
        smooth_integrated_returns
    )
    
    # Calculate True Range (TR) for each day
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = atr * smooth_integrated_returns
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    price_momentum = short_term_sma - long_term_sma
    
    # Integrate High-Low and Close-Open for Additional Momentum
    additional_momentum = intraday_high_low_spread * open_to_close_return
    
    # Smooth the Integrated Momentum using EMA
    smooth_additional_momentum = additional_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    combined_factor = 0.4 * volume_adjusted_returns + 0.3 * atr_adjusted_returns + 0.2 * price_momentum + 0.1 * smooth_additional_momentum
    
    # Confirm with Volume
    final_alpha_factor = np.where(
        df['volume'] > df['volume'].shift(1),
        combined_factor / np.sqrt(df['volume']),
        combined_factor
    )
    
    return pd.Series(final_alpha_factor, index=df.index)
```
