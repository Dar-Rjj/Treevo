```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['close_open_return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['integrated_returns'] = df['intraday_high_low_spread'] + df['close_open_return']
    
    # Smooth the Integrated Returns using EMA
    ema_window = 5
    df['smoothed_integrated_returns'] = df['integrated_returns'].ewm(span=ema_window, adjust=False).mean()
    
    # Adjust for Volume and Amount
    df['volume_adj_factor'] = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))
    df['adjusted_smoothed_returns'] = df['smoothed_integrated_returns'] * df['volume_adj_factor']
    
    # Calculate Average True Range (ATR)
    df['true_range'] = np.maximum.reduce([df['high'] - df['low'], 
                                          abs(df['high'] - df['close'].shift(1)), 
                                          abs(df['low'] - df['close'].shift(1))])
    atr_window = 14
    df['atr'] = df['true_range'].rolling(window=atr_window, min_periods=1).mean()
    
    # Combine Smoothed Integrated Returns with ATR and Volume-Amount Factor
    df['combined_factor'] = df['adjusted_smoothed_returns'] * df['atr']
    
    # Calculate Price Momentum
    short_term_sma_window = 10
    long_term_sma_window = 50
    df['sma_short'] = df['close'].rolling(window=short_term_sma_window, min_periods=1).mean()
    df['sma_long'] = df['close'].rolling(window=long_term_sma_window, min_periods=1).mean()
    df['price_momentum'] = (df['sma_short'] - df['sma_long']) * df['atr']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] + df['price_momentum']
    
    return df['alpha_factor']
```
