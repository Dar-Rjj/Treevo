```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['Integrated_Returns'] = df['High_Low_Spread'] + df['Close_Open_Return']
    
    # Adjust for Volume and Amount Confirmation
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Adjusted_Integrated_Returns'] = df['Integrated_Returns'] * df['Volume_Ratio'] * df['Amount_Ratio']
    
    # Calculate Average True Range (ATR) with Intraday Return
    df['True_Range'] = np.maximum(
        np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))),
        abs(df['low'] - df['close'].shift(1))
    )
    df['ATR_14'] = df['True_Range'].rolling(window=14).mean()
    
    # Incorporate Intraday and Open-to-Close Returns
    df['Intraday_Return'] = df['high'] - df['low']
    df['Open_to_Close_Return'] = df['close'] - df['open']
    df['Integrated_Returns_2'] = df['Intraday_Return'] + df['Open_to_Close_Return']
    
    # Combine ATR, Integrated Returns, and Volume Adjustment
    df['ATR_Adjusted'] = df['ATR_14'] * df['Integrated_Returns_2']
    df['Volume_Diff'] = df['volume'] - df['volume'].shift(1)
    df['ATR_Volume_Adjusted'] = df['ATR_Adjusted'] + df['Volume_Diff']
    
    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Price_Momentum'] = df['SMA_10'] - df['SMA_50']
    df['Momentum_Adjusted_ATR'] = df['Price_Momentum'] * df['ATR_Volume_Adjusted']
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['High_Low_Spread_2'] = df['high'] - df['low']
    df['Integrated_Momentum'] = df['High_Low_Spread_2'] * df['Open_to_Close_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    df['Combined_Factor'] = df['Momentum_Adjusted_ATR'] + df['Smoothed_Integrated_Momentum']
    df['Volume_Adjusted_Combined_Factor'] = df['Combined_Factor'] / np.sqrt(df['volume']) * (df['volume'] / df['volume'].rolling(window=10).max())
    
    # Final Alpha Factor
    final_alpha_factor = df['Volume_Adjusted_Combined_Factor'].copy()
    
    return final_alpha_factor
```
