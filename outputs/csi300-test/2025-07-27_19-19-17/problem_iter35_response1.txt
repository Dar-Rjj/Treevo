import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['TR'] = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), abs(df['low'] - df['close'].shift(1)))
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday_HL_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['Integrated_Returns'] = 0.5 * df['Intraday_HL_Spread'] + 0.5 * df['Close_Open_Return']
    
    # Smooth the Integrated Returns using EMA with a 7-day window
    df['Smoothed_Integrated_Returns'] = df['Integrated_Returns'].ewm(span=7).mean()
    
    # Adjust for Volume Confirmation
    df['Volume_Adjusted_Returns'] = np.where(df['volume'] > df['volume'].shift(1), df['Smoothed_Integrated_Returns'] * (df['volume'] / df['volume'].shift(1)), df['Smoothed_Integrated_Returns'])
    
    # Incorporate ATR in Integrated Returns
    df['ATR_Adjusted_Returns'] = df['ATR'] * df['Volume_Adjusted_Returns']
    
    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_60'] = df['close'].rolling(window=60).mean()
    df['Price_Momentum'] = (df['SMA_10'] - df['SMA_60']) * df['ATR_Adjusted_Returns']
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['Intraday_HL_Spread_Multiplied'] = df['Intraday_HL_Spread'] * df['Close_Open_Return']
    df['Smoothed_Integrated_Momentum'] = df['Intraday_HL_Spread_Multiplied'].ewm(span=5).mean()
    
    # Final Alpha Factor
    df['Combined_Factor'] = df['Volume_Adjusted_Returns'] + df['ATR_Adjusted_Returns'] + df['Price_Momentum'] + df['Smoothed_Integrated_Momentum']
    
    # Confirm with Volume
    max_volume_10_days = df['volume'].rolling(window=10).max()
    df['Final_Alpha_Factor'] = df['Combined_Factor'] / np.sqrt(df['volume']) * (df['volume'] / max_volume_10_days)
    
    return df['Final_Alpha_Factor']
