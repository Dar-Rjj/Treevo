```python
import pandas as pd

def heuristics_v2(df, atr_window=14, short_sma_window=10, long_sma_window=50):
    # Compute True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = (df['high'] - df['prev_close']).abs()
    df['low_prev_close'] = (df['low'] - df['prev_close']).abs()
    df['true_range'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['atr'] = df['true_range'].rolling(window=atr_window).mean()
    
    # Calculate Simple Moving Averages (SMA)
    df['sma_short'] = df['close'].rolling(window=short_sma_window).mean()
    df['sma_long'] = df['close'].rolling(window=long_sma_window).mean()
    
    # Compute the difference between the two SMAs
    df['sma_diff'] = df['sma_short'] - df['sma_long']
    
    # Combine with ATR
    df['factor_value'] = df['sma_diff'] * df['atr']
    
    return df['factor_value']
```
