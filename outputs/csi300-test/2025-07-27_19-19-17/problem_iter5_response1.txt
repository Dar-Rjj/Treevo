import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['TR'] = np.maximum(
        np.maximum(df['high'] - df['low'], df['high'] - df['close'].shift(1)),
        df['close'].shift(1) - df['low']
    )
    
    # Calculate Average True Range (ATR) over 14 days
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Incorporate Volume Factor
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['normalized_volume_diff'] = df['volume_diff'] / df['avg_volume_20']
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Confirm with Volume
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['weighted_integrated_return'] = df['integrated_return'] * df['volume_weight']
    
    # Combine ATR and Volume Factor with Price Momentum
    df['sma_short'] = df['close'].rolling(window=10).mean()
    df['sma_long'] = df['close'].rolling(window=50).mean()
    df['sma_diff'] = df['sma_short'] - df['sma_long']
    df['combined_alpha_factor'] = df['sma_diff'] * df['ATR'] + df['normalized_volume_diff']
    
    # Calculate Intraday High-Low Spread
    df['intraday_high_low_spread'] = df['high'] - df['low']
    
    # Integrate High-Low and Close-Open for Momentum
    df['integrated_momentum'] = df['intraday_high_low_spread'] * df['open_to_close_return']
    df['smoothed_integrated_momentum'] = df['integrated_momentum'].ewm(span=5).mean()
    
    # Final Factor
    df['final_factor'] = df['weighted_integrated_return'] + df['smoothed_integrated_momentum'] + df['combined_alpha_factor']
    df['final_factor'] = df['final_factor'] / df['volume']
    
    return df['final_factor'].dropna()
