import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrate Returns
    weight_high_low = 0.5
    weight_close_open = 0.5
    df['Integrated_Returns'] = (weight_high_low * df['High_Low_Spread']) + (weight_close_open * df['Close_Open_Return'])
    
    # Adjust for Volume and Amount Confirmation
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    
    df['Adjusted_Integrated_Returns'] = df['Integrated_Returns']
    df.loc[df['volume'] > df['volume'].shift(1), 'Adjusted_Integrated_Returns'] *= df['Volume_Ratio']
    df.loc[df['amount'] > df['amount'].shift(1), 'Adjusted_Integrated_Returns'] *= df['Amount_Ratio']
    df.loc[(df['volume'] > df['volume'].shift(1)) & (df['amount'] > df['amount'].shift(1)), 'Adjusted_Integrated_Returns'] *= (df['Volume_Ratio'] * df['Amount_Ratio'])
    
    # Calculate Average True Range (ATR)
    df['True_Range'] = df[['high', 'low', 'close']].diff(axis=1).abs().max(axis=1)
    df['ATR'] = df['True_Range'].rolling(window=14).mean()
    
    # Incorporate Intraday and Open-to-Close Returns
    df['Intraday_Return'] = df['high'] - df['low']
    df['Open_Close_Return'] = df['close'] - df['open']
    df['Integrated_Returns_2'] = df['Intraday_Return'] + df['Open_Close_Return']
    
    # Adjust with Volume Factor
    df['Volume_Diff'] = df['volume'] - df['volume'].shift(1)
    avg_volume = df['volume'].rolling(window=20).mean()
    df['Normalized_Volume_Diff'] = df['Volume_Diff'] / avg_volume
    
    # Combine ATR, Integrated Returns, and Volume Adjustment
    df['ATR_Adjusted'] = (df['ATR'] * df['Integrated_Returns_2']) + df['Normalized_Volume_Diff']
    
    # Calculate Price Momentum
    df['SMA_Short'] = df['close'].rolling(window=10).mean()
    df['SMA_Long'] = df['close'].rolling(window=50).mean()
    df['SMA_Diff'] = df['SMA_Short'] - df['SMA_Long']
    df['Price_Momentum'] = df['SMA_Diff'] * df['ATR_Adjusted']
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['Integrated_Momentum'] = df['High_Low_Spread'] * df['Close_Open_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    df['Weighted_Integrated_Return'] = df['Adjusted_Integrated_Returns']
    df['Combined_Factor'] = df['Weighted_Integrated_Return'] + df['Smoothed_Integrated_Momentum']
    
    max_volume = df['volume'].rolling(window=10).max()
    df['Volume_Adjusted_Combined_Factor'] = df['Combined_Factor'] / np.sqrt(df['volume']) * (df['volume'] / max_volume)
    
    # Combine Adjusted ATR, Price Momentum, and Volume-Adjusted Momentum
    df['Final_Alpha_Factor'] = df['ATR_Adjusted'] + df['Price_Momentum'] + df['Volume_Adjusted_Combined_Factor']
    
    return df['Final_Alpha_Factor']
