```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['IntradayHighLowSpread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['CloseOpenReturn'] = df['close'] - df['open']
    
    # Integrate Returns
    df['IntegratedReturns'] = df['IntradayHighLowSpread'] + df['CloseOpenReturn']
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    df['SmoothedIntegratedReturns'] = df['IntegratedReturns'].ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount
    df['VolumeAdjustment'] = df['volume'] / df['volume'].shift(1)
    df['AmountAdjustment'] = df['amount'] / df['amount'].shift(1)
    df['AdjustedSmoothedIntegratedReturns'] = df['SmoothedIntegratedReturns'] * df['VolumeAdjustment'] * df['AmountAdjustment']
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['TrueRange'] = df[['high', 'low']].apply(lambda x: max(x['high'] - x['low'], abs(x['high'] - df['close'].shift(1)), abs(x['low'] - df['close'].shift(1))), axis=1)
    df['ATR'] = df['TrueRange'].rolling(window=14).mean()
    
    # Incorporate Volume and Amount Factors
    df['VolumeDiff'] = df['volume'] - df['volume'].shift(1)
    df['AmountDiff'] = df['amount'] - df['amount'].shift(1)
    df['VolumeAmountSum'] = df['VolumeDiff'] + df['AmountDiff']
    
    # Combine Smoothed Integrated Returns with ATR and Volume-Amount Factor
    df['SMA10'] = df['close'].rolling(window=10).mean()
    df['SMA50'] = df['close'].rolling(window=50).mean()
    df['SMADifference'] = df['SMA10'] - df['SMA50']
    df['FinalFactor'] = (df['SMADifference'] * df['ATR'] + df['VolumeAmountSum']) * df['AdjustedSmoothedIntegratedReturns']
    
    return df['FinalFactor']
```
