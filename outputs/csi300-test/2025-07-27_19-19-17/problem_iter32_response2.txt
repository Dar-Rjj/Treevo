```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift(1)).abs()
    df['tr3'] = (df['low'] - df['close'].shift(1)).abs()
    df['tr'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['atr'] = df['tr'].rolling(window=14).mean()

    # Calculate Intraday and Open-to-Close Returns
    df['intraday_return'] = df['high'] - df['low']
    df['open_to_close_return'] = df['close'] - df['open']
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']

    # Volume Factor
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adj'] = df['volume_diff'] / df['avg_volume_20']

    # Adjust ATR by Integrated Return and Volume Adjustment
    df['adj_atr'] = (df['atr'] * df['integrated_return']) + (df['volume_adj'] / df['volume_adj'].std())

    # Price Momentum
    df['sma_short'] = df['close'].rolling(window=10).mean()
    df['sma_long'] = df['close'].rolling(window=50).mean()
    df['price_momentum'] = df['sma_short'] - df['sma_long']
    df['price_mom_adj_atr'] = df['price_momentum'] * df['adj_atr']

    # Additional Momentum from High-Low and Close-Open
    df['hl_spread'] = df['high'] - df['low']
    df['mom_integrated'] = df['hl_spread'] * df['open_to_close_return']
    df['ema_smoothed'] = df['mom_integrated'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['weighted_int_return'] = df['adj_atr']
    df['volume_adjust'] = df['volume'] / df['volume'].rolling(window=10).max()
    df['final_factor'] = df['weighted_int_return'] + df['ema_smoothed'] * df['volume_adjust']

    # Combine Adjusted ATR, Price Momentum, and Volume-Adjusted Momentum
    df['final_alpha_factor'] = df['adj_atr'] + df['price_mom_adj_atr'] + df['final_factor']

    return df['final_alpha_factor']
```
