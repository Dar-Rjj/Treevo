```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    
    # Integrate Returns
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Check Volume Increase
    df['volume_increase'] = df['volume'] > df['volume'].shift(1)
    
    # Check Amount Increase
    df['amount_increase'] = df['amount'] > df['amount'].shift(1)
    
    # Average Volume and Amount over the last 10 days
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    df['avg_amount_10d'] = df['amount'].rolling(window=10).mean()
    
    # Apply Weight
    df['factor'] = (
        df['integrated_return'] * 
        (df['volume'] / df['avg_volume_10d']) * 
        (df['amount'] / df['avg_amount_10d'])
    )
    
    # Filter out where volume or amount did not increase
    df['factor'] = df.apply(
        lambda row: row['factor'] if row['volume_increase'] and row['amount_increase'] else 0,
        axis=1
    )
    
    return df['factor']
```
