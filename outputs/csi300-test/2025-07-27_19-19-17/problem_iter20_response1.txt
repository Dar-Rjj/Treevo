```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns (Weighted Sum)
    integrated_returns = 0.5 * intraday_high_low_spread + 0.5 * close_open_return
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    ema_integrated_returns = integrated_returns.ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount Confirmation
    volume_adjustment = df['volume'] / df['volume'].shift(1)
    amount_adjustment = df['amount'] / df['amount'].shift(1)
    combined_adjustment = volume_adjustment * amount_adjustment
    adjusted_integrated_returns = ema_integrated_returns * combined_adjustment
    
    # Calculate True Range (TR) for each day
    tr = pd.DataFrame({
        'high_low': df['high'] - df['low'],
        'high_prev_close': abs(df['high'] - df['close'].shift(1)),
        'low_prev_close': abs(df['low'] - df['close'].shift(1))
    }).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    atr = tr.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = adjusted_integrated_returns * atr
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    price_momentum = short_term_sma - long_term_sma
    price_momentum_atr = price_momentum * atr
    
    # Integrate High-Low and Close-Open for Additional Momentum
    additional_momentum = intraday_high_low_spread * close_open_return
    ema_additional_momentum = additional_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (
        atr_adjusted_returns +
        price_momentum_atr +
        ema_additional_momentum
    )
    
    # Confirm with Volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=10).max()
    final_alpha_factor_volume_adjusted = final_alpha_factor * volume_ratio
    
    return final_alpha_factor_volume_adjusted
```
