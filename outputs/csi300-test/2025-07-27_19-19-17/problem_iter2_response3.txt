import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['TR'] = np.maximum(
        np.maximum(
            df['high'] - df['low'], 
            np.abs(df['high'] - df['close'].shift(1))
        ), 
        np.abs(df['low'] - df['close'].shift(1))
    )
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Calculate Simple Moving Averages (SMA) of Close price
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    
    # Compute the difference between the two SMAs
    sma_diff = short_term_sma - long_term_sma
    
    # Multiply the difference by the ATR
    intermediate_atr_momentum = sma_diff * df['ATR']
    
    # Calculate Intraday High-Low Spread
    intraday_spread = df['high'] - df['low']
    
    # Measure Intraday Momentum
    intraday_momentum = np.where(
        (df['close'] - df['open']) > intraday_spread, 
        1, 
        np.where((df['close'] - df['open']) < intraday_spread, -1, 0)
    )
    
    # Combine Intermediate ATR Momentum and Intraday Momentum
    combined_momentum = np.where(
        intraday_momentum > 0, 
        intermediate_atr_momentum, 
        np.where(intraday_momentum < 0, -intermediate_atr_momentum, 0)
    )
    
    # Calculate Daily Volume Change
    daily_volume_change = df['volume'].pct_change()
    
    # Define threshold for significant volume surge
    surge_threshold = 0.2  # Example threshold
    volume_surge = daily_volume_change > surge_threshold
    
    # Apply Volume Surge
    final_factor = np.where(
        volume_surge, 
        combined_momentum * 1.5, 
        combined_momentum
    )
    
    return pd.Series(final_factor, index=df.index, name='alpha_factor')
