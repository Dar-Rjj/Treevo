```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['Previous Close'] = df['close'].shift(1)
    df['High-Low'] = df['high'] - df['low']
    df['High-PrevClose'] = (df['high'] - df['Previous Close']).abs()
    df['Low-PrevClose'] = (df['low'] - df['Previous Close']).abs()
    df['TR'] = df[['High-Low', 'High-PrevClose', 'Low-PrevClose']].max(axis=1)

    # Calculate Average True Range (ATR) over a 21-day window
    df['ATR'] = df['TR'].rolling(window=21).mean()

    # Calculate Intraday High-Low Spread
    df['IntradaySpread'] = df['high'] - df['low']

    # Calculate Close-Open Return
    df['Close-Open_Return'] = df['close'] - df['open']

    # Integrate Returns
    integrated_returns = df['IntradaySpread'] + df['Close-Open_Return']

    # Smooth the Integrated Returns with EMA (7 days)
    df['Smoothed_Integrated_Returns'] = integrated_returns.ewm(span=7, adjust=False).mean()

    # Adjust for Volume Confirmation
    df['Volume_Adjustment'] = df['volume'] / df['volume'].shift(1)
    df['Adjusted_Integrated_Returns'] = df['Smoothed_Integrated_Returns'] * np.where(df['Volume_Adjustment'] > 1, df['Volume_Adjustment'], 1)

    # Incorporate ATR in Integrated Returns
    df['ATR_Adjusted_Returns'] = df['ATR'] * df['Adjusted_Integrated_Returns']

    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_60'] = df['close'].rolling(window=60).mean()
    df['Price_Momentum'] = df['SMA_10'] - df['SMA_60']
    df['Momentum_Adjusted_ATR'] = df['Price_Momentum'] * df['ATR_Adjusted_Returns']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['Integrated_Momentum'] = df['IntradaySpread'] * df['Close-Open_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['ATR_Adjusted_Returns'] + df['Momentum_Adjusted_ATR'] + df['Smoothed_Integrated_Momentum']

    # Confirm with Volume
    df['Volume_Confirmation'] = df['volume'] / df['volume'].shift(1)
    df['Max_Volume_10D'] = df['volume'].rolling(window=10).max()
    df['Volume_Adjusted_Factor'] = df['Final_Alpha_Factor'] * np.sqrt(df['volume']) * (df['volume'] / df['Max_Volume_10D'])

    # Final Alpha Factor
    final_alpha_factor = df['Volume_Adjusted_Factor']
    return final_alpha_factor
```
