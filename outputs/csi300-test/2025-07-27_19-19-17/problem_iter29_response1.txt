import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['PrevClose'] = df['close'].shift(1)
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], 
                                                               abs(x['high'] - df.loc[x.name, 'PrevClose']), 
                                                               abs(x['low'] - df.loc[x.name, 'PrevClose'])), axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Calculate Intraday High-Low Spread
    df['IntradaySpread'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['OpenCloseReturn'] = df['close'] - df['open']
    
    # Integrate Returns with weights
    weight_intraday = 0.6
    weight_openclose = 0.4
    df['IntegratedReturns'] = (weight_intraday * df['IntradaySpread']) + (weight_openclose * df['OpenCloseReturn'])
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    df['SmoothedReturns'] = df['IntegratedReturns'].ewm(span=5).mean()
    
    # Adjust for Volume and Amount Confirmation
    df['VolumeFactor'] = np.where(df['volume'] > df['volume'].shift(1), df['volume'] / df['volume'].shift(1), 1)
    df['AmountFactor'] = np.where(df['amount'] > df['amount'].shift(1), df['amount'] / df['amount'].shift(1), 1)
    df['FinalAdjustment'] = df['SmoothedReturns'] * df['VolumeFactor'] * df['AmountFactor']
    
    # Price Momentum
    df['SMA10'] = df['close'].rolling(window=10).mean()
    df['SMA50'] = df['close'].rolling(window=50).mean()
    df['PriceMomentum'] = df['SMA10'] - df['SMA50']
    
    # Incorporate ATR in Integrated Returns
    df['ATR_Returns'] = df['ATR'] * df['FinalAdjustment']
    
    # Adjust with Volume Factor
    df['VolumeDiff'] = df['volume'] - df['volume'].shift(1)
    df['AvgVolume'] = df['volume'].rolling(window=20).mean()
    df['VolumeAdjustedFactor'] = df['ATR_Returns'] + (df['VolumeDiff'] / df['AvgVolume'])
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['AdditionalMomentum'] = df['IntradaySpread'] * df['OpenCloseReturn']
    df['SmoothedMomentum'] = df['AdditionalMomentum'].ewm(span=5).mean()
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['FinalAdjustment'] + df['ATR_Returns'] + df['PriceMomentum'] + df['SmoothedMomentum']
    
    # Confirm with Volume
    df['MaxVolume10'] = df['volume'].rolling(window=10).max()
    df['AlphaFactor'] = df['AlphaFactor'] / np.sqrt(df['volume']) * (df['volume'] / df['MaxVolume10'])
    
    return df['AlphaFactor']
