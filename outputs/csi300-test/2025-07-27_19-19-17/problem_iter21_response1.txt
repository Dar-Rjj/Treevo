```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Smooth the Integrated Returns using EMA (5 days)
    smoothed_integrated_returns = integrated_returns.ewm(span=5).mean()
    
    # Adjust for Volume and Amount
    volume_factor = df['volume'] / df['volume'].shift(1)
    amount_factor = df['amount'] / df['amount'].shift(1)
    adjusted_integrated_returns = smoothed_integrated_returns * volume_factor * amount_factor
    
    # Calculate True Range (TR) for each day
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    weighted_integrated_returns = adjusted_integrated_returns * atr
    
    # Adjust with Volume Factor
    volume_difference = df['volume'] - df['volume'].shift(1)
    average_volume = df['volume'].rolling(window=20).mean()
    volume_adjustment = volume_difference / average_volume
    adjusted_atr = weighted_integrated_returns + volume_adjustment
    
    # Calculate Price Momentum
    short_sma = df['close'].rolling(window=10).mean()
    long_sma = df['close'].rolling(window=50).mean()
    sma_difference = short_sma - long_sma
    price_momentum = sma_difference * adjusted_atr
    
    # Integrate High-Low and Close-Open for Additional Momentum
    additional_momentum = intraday_high_low_spread * close_open_return
    smoothed_additional_momentum = additional_momentum.ewm(span=5).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (adjusted_atr + price_momentum + smoothed_additional_momentum)
    
    # Confirm with Volume
    volume_adjusted_factor = final_alpha_factor / np.sqrt(df['volume']) * (df['volume'] / df['volume'].rolling(window=10).max())
    
    return volume_adjusted_factor
```
