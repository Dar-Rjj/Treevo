```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = abs(df['high'] - df['prev_close'])
    df['low_prev_close'] = abs(df['low'] - df['prev_close'])
    df['true_range'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['atr'] = df['true_range'].rolling(window=14).mean()

    # Calculate Intraday Return and Open-to-Close Return
    df['intraday_return'] = df['high'] - df['low']
    df['open_to_close_return'] = df['close'] - df['open']

    # Integrate Returns
    df['integrated_return'] = 0.6 * df['intraday_return'] + 0.4 * df['open_to_close_return']

    # Adjust with Volume Factor
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['adjusted_volume'] = df['volume_diff'] / df['avg_volume_20']

    # Combine ATR, Integrated Returns, and Volume Adjustment
    df['combined_factor'] = df['atr'] * df['integrated_return'] + df['adjusted_volume']

    # Calculate Price Momentum
    df['sma_short'] = df['close'].rolling(window=10).mean()
    df['sma_long'] = df['close'].rolling(window=50).mean()
    df['price_momentum'] = df['sma_short'] - df['sma_long']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['integrated_momentum'] = df['intraday_high_low_spread'] * df['open_to_close_return']
    df['smoothed_integrated_momentum'] = df['integrated_momentum'].ewm(span=5).mean()

    # Final Alpha Factor
    df['final_alpha'] = df['combined_factor'] + df['price_momentum'] + df['smoothed_integrated_momentum']

    # Confirm with Volume
    df['volume_adjusted_final_alpha'] = df['final_alpha'] / np.sqrt(df['volume']) * (df['volume'] / df['volume'].rolling(window=10).max())

    return df['volume_adjusted_final_alpha']
```
