```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']

    # Integrate Returns
    integrated_returns = high_low_spread + close_open_return

    # Smooth the Integrated Returns using EMA
    ema_window = 5
    smoothed_returns = integrated_returns.ewm(span=ema_window, adjust=False).mean()

    # Volume and Amount Confirmation
    volume_increase = df['volume'] > df['volume'].shift(1)
    amount_increase = df['amount'] > df['amount'].shift(1)

    # Calculate Average Volume and Amount over last 10 days
    avg_volume_10d = df['volume'].rolling(window=10).mean()
    avg_amount_10d = df['amount'].rolling(window=10).mean()

    # Adjust for Volume and Amount
    adjusted_factor = (smoothed_returns / (df['volume'] * df['amount'])) * \
                      (df['volume'] / avg_volume_10d) * (df['amount'] / avg_amount_10d)

    # Apply Volume and Amount confirmation
    adjusted_factor = adjusted_factor.where(volume_increase & amount_increase, other=0)

    return adjusted_factor
```
