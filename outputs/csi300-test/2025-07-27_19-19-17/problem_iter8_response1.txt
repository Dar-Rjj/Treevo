```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = (df['high'] - df['prev_close']).abs()
    df['low_prev_close'] = (df['low'] - df['prev_close']).abs()
    df['tr'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['atr'] = df['tr'].rolling(window=14).mean()
    
    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Confirm with Volume and Amount
    df['volume_weight'] = df['volume'] / df['volume'].rolling(window=10).mean()
    df['amount_weight'] = df['amount'] / df['amount'].rolling(window=10).mean()
    
    # Apply weights if current volume and amount are greater than previous day's
    df['weighted_integrated_return_volume'] = df.apply(
        lambda row: row['integrated_return'] * row['volume_weight'] 
        if row['volume'] > df.shift(1).loc[row.name, 'volume'] 
        else 0, axis=1
    )
    
    df['weighted_integrated_return_amount'] = df.apply(
        lambda row: row['integrated_return'] * row['amount_weight'] 
        if row['amount'] > df.shift(1).loc[row.name, 'amount'] 
        else 0, axis=1
    )
    
    # Combine with Price Momentum
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_diff'] = df['sma_10'] - df['sma_50']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_integrated_return_volume'] + df['weighted_integrated_return_amount']) * (df['sma_diff'] * df['atr'])
    
    return df['alpha_factor']
```
