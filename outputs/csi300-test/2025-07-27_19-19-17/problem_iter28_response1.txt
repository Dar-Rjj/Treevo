```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns (Weighted Sum of Intraday High-Low Spread and Close-Open Return)
    integrated_returns = 0.5 * intraday_high_low_spread + 0.5 * close_open_return
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    smoothed_returns = integrated_returns.ewm(span=5).mean()
    
    # Adjust for Volume Confirmation
    volume_adj_factor = df['volume'].div(df['volume'].shift(1))
    adjusted_returns = smoothed_returns * np.where(df['volume'] > df['volume'].shift(1), volume_adj_factor, 1)
    
    # Calculate True Range (TR) for each day
    high_prev_close = df['high'] - df['close'].shift(1)
    low_prev_close = abs(df['low'] - df['close'].shift(1))
    tr = pd.concat([intraday_high_low_spread, high_prev_close, low_prev_close], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    atr = tr.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = adjusted_returns * atr
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    price_momentum = short_term_sma - long_term_sma
    
    # Final Alpha Factor: Combine Weighted Integrated Return, ATR, and Price Momentum
    final_alpha = 0.6 * atr_adjusted_returns + 0.4 * price_momentum
    
    # Confirm with Volume
    volume_confirmed_alpha = final_alpha / np.sqrt(np.where(df['volume'] > df['volume'].shift(1), df['volume'], 1))
    
    return volume_confirmed_alpha
```
