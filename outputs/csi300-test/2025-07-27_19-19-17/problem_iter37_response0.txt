```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = (df['high'] - df['close'].shift()).abs()
    df['tr3'] = (df['low'] - df['close'].shift()).abs()
    df['TR'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday and Open-to-Close Returns
    df['intraday_return'] = df['high'] - df['low']
    df['open_to_close_return'] = df['close'] - df['open']

    # Integrate Returns
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']

    # Adjust with Volume Factor
    df['volume_diff'] = df['volume'] - df['volume'].shift()
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjustment'] = df['volume_diff'] / df['avg_volume_20']

    # Combine ATR, Integrated Returns, and Volume Adjustment
    df['adjusted_atr'] = df['ATR'] * df['integrated_return'] + df['volume_adjustment']

    # Calculate Price Momentum
    df['sma_short'] = df['close'].rolling(window=10).mean()
    df['sma_long'] = df['close'].rolling(window=50).mean()
    df['sma_diff'] = df['sma_short'] - df['sma_long']

    # Adjusted ATR with Price Momentum
    df['price_momentum'] = df['sma_diff'] * df['adjusted_atr']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['integrated_momentum'] = df['intraday_high_low_spread'] * df['open_to_close_return']

    # Smooth the Integrated Momentum with EMA
    df['smoothed_integrated_momentum'] = df['integrated_momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['weighted_integrated_return'] = df['integrated_return'] * df['adjusted_atr']
    df['combined_factor'] = df['weighted_integrated_return'] + df['smoothed_integrated_momentum']

    # Confirm with Volume
    df['volume_adjusted_factor'] = df['combined_factor'] / np.sqrt(df['volume']) * (df['volume'] / df['volume'].rolling(window=10).max())

    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_atr'] + df['price_momentum'] + df['volume_adjusted_factor']

    return df['alpha_factor']
```
