```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns (Weighted Sum of Intraday High-Low Spread and Close-Open Return)
    integrated_returns = 0.5 * intraday_high_low_spread + 0.5 * close_open_return
    
    # Smooth the Integrated Returns (EMA with 14-day window)
    ema_integrated_returns = integrated_returns.ewm(span=14).mean()
    
    # Adjust for Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1))
    adjusted_returns = ema_integrated_returns * (df['volume'] / df['volume'].shift(1)).where(volume_confirmation, 1)
    
    # Calculate True Range (TR) for each day
    high_low = df['high'] - df['low']
    high_close_prev = abs(df['high'] - df['close'].shift(1))
    low_close_prev = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([high_low, high_close_prev, low_close_prev], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 21-day window
    atr = true_range.rolling(window=21).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = adjusted_returns / atr
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=15).mean()
    long_term_sma = df['close'].rolling(window=60).mean()
    price_momentum = short_term_sma - long_term_sma
    
    # Final Alpha Factor (Combine Weighted Integrated Return, ATR, and Price Momentum)
    final_factor = 0.5 * atr_adjusted_returns + 0.5 * price_momentum
    
    # Confirm with Volume
    final_factor = final_factor.where(volume_confirmation, final_factor / df['volume'])
    
    return final_factor
```
