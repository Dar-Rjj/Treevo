import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['high_low'] = df['high'] - df['low']
    df['high_prev_close'] = abs(df['high'] - df['prev_close'])
    df['low_prev_close'] = abs(df['low'] - df['prev_close'])
    df['TR'] = df[['high_low', 'high_prev_close', 'low_prev_close']].max(axis=1)
    
    # Average True Range (ATR) over a window, e.g., 14 days
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Integrated Returns
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Volume Factor
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume'] = df['volume'].rolling(window=20).mean()
    df['normalized_volume_diff'] = df['volume_diff'] / df['avg_volume']
    
    # Adjust ATR with Integrated Returns and Volume Adjustment
    df['adjusted_ATR'] = df['ATR'] * df['integrated_return'] + df['normalized_volume_diff']
    
    # Price Momentum
    df['SMA_short'] = df['close'].rolling(window=10).mean()
    df['SMA_long'] = df['close'].rolling(window=50).mean()
    df['SMA_diff'] = df['SMA_short'] - df['SMA_long']
    df['price_momentum'] = df['SMA_diff'] * df['adjusted_ATR']
    
    # High-Low and Close-Open for Additional Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['integrated_momentum'] = df['intraday_high_low_spread'] * df['open_to_close_return']
    df['smoothed_integrated_momentum'] = df['integrated_momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha'] = df['adjusted_ATR'] + df['price_momentum'] + df['smoothed_integrated_momentum']
    
    # Volume Confirmation
    df['volume_confirmation'] = df['volume'] > df['volume'].shift(1)
    df['final_alpha'] *= df['volume_confirmation']
    max_volume = df['volume'].rolling(window=10).max()
    df['final_alpha'] *= (df['volume'] / max_volume) / np.sqrt(df['volume'])
    
    return df['final_alpha'].dropna()
