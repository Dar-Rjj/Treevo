```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = 0.5 * (intraday_high_low_spread + close_open_return)
    
    # Smooth the Integrated Returns with EMA
    smoothed_integrated_returns = integrated_returns.ewm(span=5).mean()
    
    # Adjust for Volume Confirmation
    volume_confirmation = (df['volume'] > df['volume'].shift(1))
    adjusted_integrated_returns = smoothed_integrated_returns * (df['volume'] / df['volume'].shift(1))
    adjusted_integrated_returns[~volume_confirmation] = 0
    
    # Calculate True Range (TR) for each day
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = atr * adjusted_integrated_returns
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    price_momentum = short_term_sma - long_term_sma
    
    # Integrate High-Low and Close-Open for Additional Momentum
    additional_momentum = intraday_high_low_spread * close_open_return
    
    # Smooth the Integrated Momentum with EMA
    smoothed_additional_momentum = additional_momentum.ewm(span=5).mean()
    
    # Final Alpha Factor
    combined_factor = 0.5 * (atr_adjusted_returns + price_momentum + smoothed_additional_momentum)
    
    # Confirm with Volume
    volume_adjusted_factor = combined_factor / np.sqrt(df['volume'])
    final_alpha_factor = volume_adjusted_factor.where(volume_confirmation, 0)
    
    return final_alpha_factor
```
