```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['TR'] = df[['High', 'Low', 'Close']].apply(lambda x: max(x['High'] - x['Low'], 
                                                                  abs(x['High'] - df['Close'].shift(1)), 
                                                                  abs(x['Low'] - df['Close'].shift(1))), axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['High'] - df['Low']

    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['Close'] - df['Open']

    # Integrate Returns
    df['Integrated_Returns'] = df['Intraday_Spread'] + df['Close_Open_Return']

    # Smooth the Integrated Returns with EMA
    df['Smoothed_Integrated_Returns'] = df['Integrated_Returns'].ewm(span=5, adjust=False).mean()

    # Adjust with Volume and Amount
    volume_adjustment = (df['Volume'] / df['Volume'].shift(1)) * (df['Amount'] / df['Amount'].shift(1))
    df['Adjusted_Integrated_Returns'] = df['Smoothed_Integrated_Returns'] * volume_adjustment

    # Incorporate ATR in Integrated Returns
    df['ATR_Adjusted_Returns'] = df['ATR'] * df['Adjusted_Integrated_Returns']

    # Calculate Price Momentum
    df['SMA_10'] = df['Close'].rolling(window=10).mean()
    df['SMA_50'] = df['Close'].rolling(window=50).mean()
    df['Price_Momentum'] = (df['SMA_10'] - df['SMA_50']) * df['ATR_Adjusted_Returns']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['Integrated_Momentum'] = df['Intraday_Spread'] * df['Close_Open_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['Combined_Factor'] = df['ATR_Adjusted_Returns'] + df['Price_Momentum'] + df['Smoothed_Integrated_Momentum']

    # Confirm with Volume
    max_volume = df['Volume'].rolling(window=10).max()
    df['Final_Alpha_Factor'] = (df['Combined_Factor'] / np.sqrt(df['Volume'])) * (df['Volume'] / max_volume)

    return df['Final_Alpha_Factor']
```
