```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['high_low'] = df['high'] - df['low']
    df['high_close_prev'] = abs(df['high'] - df['close'].shift(1))
    df['low_close_prev'] = abs(df['low'] - df['close'].shift(1))
    df['tr'] = df[['high_low', 'high_close_prev', 'low_close_prev']].max(axis=1)

    # Calculate Average True Range (ATR) over a window, e.g., 14 days
    df['atr'] = df['tr'].rolling(window=14).mean()

    # Calculate Intraday Return
    df['intraday_return'] = df['high'] - df['low']

    # Calculate Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']

    # Integrate Returns
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']

    # Adjust with Volume and Amount Factors
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['norm_volume_diff'] = df['volume_diff'] / df['avg_volume_20']

    # Check Volume and Amount conditions
    df['volume_adj'] = np.where(df['volume'] > df['volume'].shift(1), 
                                (df['volume'] / df['volume'].shift(1)), 1)
    df['amount_adj'] = np.where(df['amount'] > df['amount'].shift(1),
                                (df['amount'] / df['amount'].shift(1)), 1)
    df['final_adj'] = np.where((df['volume'] > df['volume'].shift(1)) & (df['amount'] > df['amount'].shift(1)),
                               (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1)), 1)

    # Combine ATR, Integrated Returns, and Volume-Amount Adjustment
    df['adjusted_atr'] = df['atr'] * df['integrated_return'] + df['norm_volume_diff']

    # Calculate Price Momentum
    df['sma_short'] = df['close'].rolling(window=10).mean()
    df['sma_long'] = df['close'].rolling(window=50).mean()
    df['sma_diff'] = df['sma_short'] - df['sma_long']
    df['price_momentum'] = df['sma_diff'] * df['adjusted_atr']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['momentum'] = df['intraday_high_low_spread'] * df['open_to_close_return']
    df['smoothed_momentum'] = df['momentum'].ewm(span=5).mean()

    # Final Alpha Factor
    df['weighted_integrated_return'] = df['integrated_return'] * df['final_adj']
    df['combined_factor'] = df['weighted_integrated_return'] + df['smoothed_momentum']

    # Adjust for Volume
    df['max_volume_10'] = df['volume'].rolling(window=10).max()
    df['final_alpha'] = df['combined_factor'] * (df['volume'] / df['max_volume_10'])

    return df['final_alpha']
```
