```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['close_open_return'] = df['close'] - df['open']
    
    # Integrate Returns
    df['integrated_returns'] = df['high_low_spread'] + df['close_open_return']
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    df['ema_integrated_returns'] = df['integrated_returns'].ewm(span=5, adjust=False).mean()
    
    # Confirm with Volume
    df['volume_confirmation'] = df['volume'] > df['volume'].shift(1)
    
    # Adjust for Volume
    df['average_volume_10_days'] = df['volume'].rolling(window=10).mean()
    df['volume_adjusted_factor'] = df['ema_integrated_returns'] / df['volume'] * (df['volume'] / df['average_volume_10_days'])
    
    # Final alpha factor
    alpha_factor = df['volume_adjusted_factor'].where(df['volume_confirmation'], 0)
    
    return alpha_factor
```
