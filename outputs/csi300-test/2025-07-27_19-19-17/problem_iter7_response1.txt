```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return (High - Low)
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return (Close - Open)
    open_close_return = df['close'] - df['open']
    
    # Integrate Returns (Intraday Return + Open-to-Close Return)
    integrated_return = intraday_return + open_close_return
    
    # Check if Volume > Previous Day's Volume
    volume_confirmation = df['volume'] > df['volume'].shift(1)
    
    # Calculate the weight (Current Volume / Average Volume over last 10 days)
    avg_volume_10_days = df['volume'].rolling(window=10).mean()
    volume_weight = df['volume'] / avg_volume_10_days
    
    # Apply Weight to Integrated Return where Volume Confirmation is True
    factor_values = integrated_return * volume_weight * volume_confirmation
    
    return factor_values
```
