```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns with weights
    weight_high_low = 0.5
    weight_close_open = 0.5
    integrated_returns = (weight_high_low * intraday_high_low_spread) + (weight_close_open * close_open_return)
    
    # Adjust for Volume and Amount Confirmation
    adjusted_returns = integrated_returns.copy()
    
    # Check Volume
    volume_condition = df['volume'] > df['volume'].shift(1)
    adjusted_returns[volume_condition] *= df['volume'][volume_condition] / df['volume'].shift(1)[volume_condition]
    
    # Check Amount
    amount_condition = df['amount'] > df['amount'].shift(1)
    adjusted_returns[amount_condition] *= df['amount'][amount_condition] / df['amount'].shift(1)[amount_condition]
    
    # Final Adjustment if both Volume and Amount conditions are met
    both_conditions_met = volume_condition & amount_condition
    adjusted_returns[both_conditions_met] *= (df['volume'][both_conditions_met] / df['volume'].shift(1)[both_conditions_met]) * (df['amount'][both_conditions_met] / df['amount'].shift(1)[both_conditions_met])
    
    return adjusted_returns
```
