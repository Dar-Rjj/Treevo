```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Measure Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Compare Intraday Momentum to High-Low Spread
    df['momentum_indicator'] = (df['intraday_momentum'] > df['high_low_spread']).astype(int)
    
    # Calculate Daily Volume Change
    df['volume_change'] = df['volume'].pct_change()
    
    # Set Threshold for significant volume surge
    volume_surge_threshold = 0.3  # Example threshold, can be adjusted
    
    # Apply Threshold to identify volume surges
    df['volume_surge'] = (df['volume_change'] > volume_surge_threshold).astype(int)
    
    # Generate Factor by combining Intraday Momentum and Volume Surge
    df['alpha_factor'] = (df['momentum_indicator'] * df['volume_surge']).replace({0: -1, 1: 1})
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
