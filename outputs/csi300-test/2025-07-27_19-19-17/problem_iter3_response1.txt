import pandas as pd

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()
    
    # Compute True Range (TR)
    df['TR'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x[0] - x[1], abs(x[0] - df['close'].shift(1)), abs(x[1] - df['close'].shift(1))), axis=1
    )
    
    # Calculate Average True Range (ATR) over a window, e.g., 14 days
    atr_window = 14
    df['ATR'] = df['TR'].rolling(window=atr_window).mean()
    
    # Calculate volume difference between current and previous day
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    
    # Normalize the volume difference by average volume over a window, e.g., 20 days
    vol_norm_window = 20
    df['avg_volume'] = df['volume'].rolling(window=vol_norm_window).mean()
    df['norm_volume_diff'] = df['volume_diff'] / df['avg_volume']
    
    # Calculate Simple Moving Averages (SMA) of Close price
    short_term_sma_window = 10
    long_term_sma_window = 50
    df['SMA_short'] = df['close'].rolling(window=short_term_sma_window).mean()
    df['SMA_long'] = df['close'].rolling(window=long_term_sma_window).mean()
    
    # Compute the difference between the two SMAs
    df['SMA_diff'] = df['SMA_short'] - df['SMA_long']
    
    # Multiply the SMA difference by the ATR
    df['SMA_ATR_product'] = df['SMA_diff'] * df['ATR']
    
    # Adjust the product by adding the normalized volume difference
    df['alpha_factor'] = df['SMA_ATR_product'] + df['norm_volume_diff']
    
    # Return the final alpha factor
    return df['alpha_factor']
