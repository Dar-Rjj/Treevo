```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = (df['close'] - df['open']) / df['open']
    
    # Integrate High-Low and Close-Open for Momentum
    integrated_momentum = intraday_high_low_spread * close_open_return
    
    # Smooth the Integrated Momentum using Exponential Moving Average (EMA)
    ema_window = 5
    smoothed_momentum = integrated_momentum.ewm(span=ema_window, adjust=False).mean()
    
    # Adjust for Volume
    adjusted_momentum = smoothed_momentum / df['volume']
    
    return adjusted_momentum
```
