```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['prev_close'] = df['close'].shift(1)
    df['tr1'] = df['high'] - df['low']
    df['tr2'] = abs(df['high'] - df['prev_close'])
    df['tr3'] = abs(df['low'] - df['prev_close'])
    df['true_range'] = df[['tr1', 'tr2', 'tr3']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['atr'] = df['true_range'].rolling(window=14).mean()

    # Calculate Intraday High-Low Spread
    df['intraday_high_low'] = df['high'] - df['low']

    # Calculate Close-Open Return
    df['close_open_return'] = df['close'] - df['open']

    # Integrate Returns
    df['integrated_returns'] = df['intraday_high_low'] + df['close_open_return']

    # Smooth the Integrated Returns with EMA (5 days)
    df['smoothed_integrated_returns'] = df['integrated_returns'].ewm(span=5, adjust=False).mean()

    # Adjust with Volume and Amount
    df['volume_adjustment'] = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))
    df['adjusted_smoothed_returns'] = df['smoothed_integrated_returns'] * df['volume_adjustment']

    # Incorporate ATR in Integrated Returns
    df['atrr_weighted_returns'] = df['atr'] * df['integrated_returns']

    # Calculate Price Momentum
    df['sma_short'] = df['close'].rolling(window=10).mean()
    df['sma_long'] = df['close'].rolling(window=50).mean()
    df['price_momentum'] = df['sma_short'] - df['sma_long']
    df['atrr_price_momentum'] = df['price_momentum'] * df['atrr_weighted_returns']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['integrated_momentum'] = df['intraday_high_low'] * df['close_open_return']
    df['smoothed_integrated_momentum'] = df['integrated_momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['combined_factor'] = (
        df['adjusted_smoothed_returns'] +
        df['atrr_weighted_returns'] +
        df['atrr_price_momentum'] +
        df['smoothed_integrated_momentum']
    )

    # Confirm with Volume
    df['volume_adjusted_factor'] = df['combined_factor'] / np.sqrt(df['volume'])
    df['volume_adjusted_factor'] *= (df['volume'] / df['volume'].rolling(window=10).max())

    final_alpha_factor = df['volume_adjusted_factor']

    return final_alpha_factor
```
