import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['TR'] = df[['high', 'low', 'close']].apply(lambda x: max(x['high'] - x['low'], 
                                                                abs(x['high'] - x['close'].shift(1)), 
                                                                abs(x['low'] - x['close'].shift(1))), axis=1)
    
    # Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Intraday Return
    df['intraday_return'] = df['high'] - df['low']
    
    # Open-to-Close Return
    df['open_to_close_return'] = df['close'] - df['open']
    
    # Integrated Returns
    df['integrated_return'] = df['intraday_return'] + df['open_to_close_return']
    
    # Volume Adjustment
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['normalized_volume_diff'] = df['volume_diff'] / df['average_volume']
    
    # Adjusted ATR
    df['adjusted_ATR'] = df['ATR'] * df['integrated_return'] + df['normalized_volume_diff']
    
    # Price Momentum
    df['SMA_short'] = df['close'].rolling(window=10).mean()
    df['SMA_long'] = df['close'].rolling(window=50).mean()
    df['price_momentum'] = df['SMA_short'] - df['SMA_long']
    df['weighted_price_momentum'] = df['price_momentum'] * df['adjusted_ATR']
    
    # Additional Momentum Integration
    df['intraday_high_low_spread'] = df['high'] - df['low']
    df['integrated_momentum'] = df['intraday_high_low_spread'] * df['open_to_close_return']
    df['smoothed_integrated_momentum'] = df['integrated_momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    df['combined_factor'] = df['weighted_price_momentum'] + df['smoothed_integrated_momentum']
    df['final_alpha_factor'] = df['adjusted_ATR'] + df['weighted_price_momentum'] + df['smoothed_integrated_momentum'] / df['volume']
    
    return df['final_alpha_factor'].dropna()
