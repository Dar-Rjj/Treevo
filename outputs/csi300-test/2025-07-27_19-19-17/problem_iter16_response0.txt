```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['close_open_return'] = df['close'] - df['open']
    
    # Integrate Returns with weights
    weight_high_low = 0.5
    weight_close_open = 0.5
    df['integrated_returns'] = (weight_high_low * df['high_low_spread']) + (weight_close_open * df['close_open_return'])
    
    # Adjust for Volume and Amount Confirmation
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['amount_ratio'] = df['amount'] / df['amount'].shift(1)
    
    # Apply volume and amount confirmation
    df['adjusted_returns'] = df['integrated_returns'] * df['volume_ratio'] * df['amount_ratio']
    
    # Return the factor values as a Series
    return df['adjusted_returns']
```
