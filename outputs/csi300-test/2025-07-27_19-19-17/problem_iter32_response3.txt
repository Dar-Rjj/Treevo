```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = 0.5 * intraday_high_low_spread + 0.5 * close_open_return
    
    # Smooth the Integrated Returns using EMA with 7-day window
    smoothed_integrated_returns = integrated_returns.ewm(span=7).mean()
    
    # Adjust for Volume Confirmation
    volume_confirmation = df['volume'] > df['volume'].shift(1)
    adjusted_integrated_returns = smoothed_integrated_returns * (df['volume'] / df['volume'].shift(1)).where(volume_confirmation, 1)
    
    # Calculate True Range (TR) for each day
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_integrated_returns = atr * adjusted_integrated_returns
    
    # Integrate High-Low and Close-Open for Additional Momentum
    additional_momentum = intraday_high_low_spread * close_open_return
    smoothed_additional_momentum = additional_momentum.ewm(span=5).mean()
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=15).mean()
    long_term_sma = df['close'].rolling(window=60).mean()
    price_momentum = short_term_sma - long_term_sma
    
    # Final Alpha Factor
    combined_factor = atr_adjusted_integrated_returns + price_momentum + smoothed_additional_momentum
    
    # Confirm with Volume
    volume_adjusted_combined_factor = combined_factor / np.sqrt(df['volume'])
    
    return volume_adjusted_combined_factor
```
