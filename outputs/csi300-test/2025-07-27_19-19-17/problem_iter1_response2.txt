import pandas as pd

def heuristics_v2(df, short_window=10, long_window=50, atr_window=14):
    # Compute True Range (TR)
    df['Previous Close'] = df['close'].shift(1)
    df['TR1'] = df['high'] - df['low']
    df['TR2'] = (df['high'] - df['Previous Close']).abs()
    df['TR3'] = (df['low'] - df['Previous Close']).abs()
    df['TR'] = df[['TR1', 'TR2', 'TR3']].max(axis=1)
    
    # Calculate Average True Range (ATR)
    df['ATR'] = df['TR'].rolling(window=atr_window).mean()
    
    # Calculate Simple Moving Averages (SMA)
    df['SMA_short'] = df['close'].rolling(window=short_window).mean()
    df['SMA_long'] = df['close'].rolling(window=long_window).mean()
    
    # Compute the difference between the two SMAs
    df['SMA_diff'] = df['SMA_short'] - df['SMA_long']
    
    # Multiply the difference by the ATR to get the final alpha factor
    df['alpha_factor'] = df['SMA_diff'] * df['ATR']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
