```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns with weights
    weight_intraday_high_low = 0.6
    weight_close_open = 0.4
    integrated_returns = (weight_intraday_high_low * intraday_high_low_spread) + \
                         (weight_close_open * close_open_return)
    
    # Adjust for Volume and Amount Confirmation
    volume_adjustment = df['volume'] / df['volume'].shift(1)
    amount_adjustment = df['amount'] / df['amount'].shift(1)
    
    # Apply volume and amount adjustments
    adjusted_returns = integrated_returns * volume_adjustment * amount_adjustment
    
    # Add a smoothing factor to the final adjusted return
    adjusted_returns_shifted = adjusted_returns.shift(1).fillna(0)  # Shift to get previous day's return
    smoothing_factor = 0.1 * (adjusted_returns_shifted - adjusted_returns)
    final_adjusted_returns = adjusted_returns + smoothing_factor
    
    return final_adjusted_returns
```
