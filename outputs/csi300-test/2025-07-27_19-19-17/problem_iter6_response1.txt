```python
import pandas as pd

def heuristics_v2(df, atr_window=14, volume_amount_window=20, short_sma_window=10, long_sma_window=50):
    # Compute True Range (TR)
    df['TR'] = df[['high', 'low']].apply(lambda x: x['high'] - x['low'], axis=1)
    df['prev_close'] = df['close'].shift(1)
    df['TR'] = df[['TR', 'high', 'prev_close', 'low']].apply(
        lambda x: max(x['TR'], abs(x['high'] - x['prev_close']), abs(x['low'] - x['prev_close'])), axis=1)
    
    # Calculate ATR
    df['ATR'] = df['TR'].rolling(window=atr_window).mean()
    
    # Incorporate Volume and Amount Factors
    df['volume_diff'] = df['volume'] - df['volume'].shift(1)
    df['amount_diff'] = df['amount'] - df['amount'].shift(1)
    df['volume_amount_diff_sum'] = df['volume_diff'] + df['amount_diff']
    df['avg_volume'] = df['volume'].rolling(window=volume_amount_window).mean()
    df['avg_amount'] = df['amount'].rolling(window=volume_amount_window).mean()
    df['norm_volume_amount_diff'] = df['volume_amount_diff_sum'] / (df['avg_volume'] + df['avg_amount'])
    
    # Combine ATR and Volume-Amount Factor with Price Momentum
    df['SMA_short'] = df['close'].rolling(window=short_sma_window).mean()
    df['SMA_long'] = df['close'].rolling(window=long_sma_window).mean()
    df['SMA_diff'] = df['SMA_short'] - df['SMA_long']
    df['ATR_SMA_product'] = df['SMA_diff'] * df['ATR']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['ATR_SMA_product'] + df['norm_volume_amount_diff']
    
    return df['alpha_factor']
```
