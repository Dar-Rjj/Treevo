```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['High_Low'] = df['high'] - df['low']
    df['High_Close'] = np.abs(df['high'] - df['close'].shift(1))
    df['Low_Close'] = np.abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['High_Low', 'High_Close', 'Low_Close']].max(axis=1)

    # Calculate Average True Range (ATR) over 14 days
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday Return and Open-to-Close Return
    df['Intraday_Return'] = df['high'] - df['low']
    df['Open_to_Close_Return'] = df['close'] - df['open']

    # Integrate Returns
    df['Integrated_Return'] = df['Intraday_Return'] + df['Open_to_Close_Return']

    # Adjust with Volume and Amount Factors
    df['Volume_Diff'] = df['volume'] - df['volume'].shift(1)
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)

    # Adjust Integrated Returns
    conditions = [
        (df['volume'] > df['volume'].shift(1)) & (df['amount'] > df['amount'].shift(1)),
        (df['volume'] > df['volume'].shift(1)),
        (df['amount'] > df['amount'].shift(1))
    ]
    choices = [
        df['Volume_Ratio'] * df['Amount_Ratio'],
        df['Volume_Ratio'],
        df['Amount_Ratio']
    ]
    df['Adjusted_Integrated_Return'] = np.select(conditions, choices, default=1.0) * df['Integrated_Return']

    # Combine ATR and Integrated Returns
    df['Weighted_Integrated_Return'] = df['ATR'] * df['Adjusted_Integrated_Return'] + df['Volume_Diff']

    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Price_Momentum'] = df['SMA_10'] - df['SMA_50']
    df['Price_Momentum_Adjusted'] = df['Price_Momentum'] * df['Weighted_Integrated_Return']

    # Integrate High-Low and Close-Open for Additional Momentum
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Integrated_Momentum'] = df['Intraday_High_Low_Spread'] * df['Open_to_Close_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    df['Combined_Factor'] = df['Weighted_Integrated_Return'] + df['Smoothed_Integrated_Momentum']

    # Adjust for Volume
    df['Max_Volume_10'] = df['volume'].rolling(window=10).max()
    df['Volume_Adjusted_Combined_Factor'] = df['Combined_Factor'] / np.sqrt(df['volume']) * (df['volume'] / df['Max_Volume_10'])

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Weighted_Integrated_Return'] + df['Price_Momentum_Adjusted'] + df['Volume_Adjusted_Combined_Factor']

    return df['Final_Alpha_Factor']
```
