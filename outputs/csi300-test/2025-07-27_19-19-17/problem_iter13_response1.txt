```python
import pandas as pd

def heuristics_v2(df):
    # Calculate True Range (TR) for each day
    df['Previous Close'] = df['close'].shift(1)
    df['High-Low'] = df['high'] - df['low']
    df['High-PrevClose'] = abs(df['high'] - df['Previous Close'])
    df['Low-PrevClose'] = abs(df['low'] - df['Previous Close'])
    df['TR'] = df[['High-Low', 'High-PrevClose', 'Low-PrevClose']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Intraday Return
    df['Intraday_Return'] = df['high'] - df['low']

    # Calculate Open-to-Close Return
    df['Open-to-Close_Return'] = df['close'] - df['open']

    # Integrate Returns
    df['Integrated_Return'] = df['Intraday_Return'] + df['Open-to-Close_Return']

    # Confirm with Volume and Amount
    df['Volume_Weight'] = 0
    df['Amount_Weight'] = 0

    # Volume Weight
    df.loc[df['volume'] > df['volume'].shift(1), 'Volume_Weight'] = df['Integrated_Return'] * (df['volume'] / df['volume'].rolling(window=10).mean())

    # Amount Weight
    df.loc[df['amount'] > df['amount'].shift(1), 'Amount_Weight'] = df['Integrated_Return'] * (df['amount'] / df['amount'].rolling(window=10).mean())

    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Volume_Weight'] + df['Amount_Weight']) * df['ATR']

    return df['Alpha_Factor'].dropna()
```
