```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['Prev_Close'] = df['close'].shift(1)
    df['TR'] = df[['high', 'low', 'close']].max(axis=1) - df[['high', 'low', 'close']].min(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Calculate Intraday High-Low Spread
    df['Intraday_Spread'] = df['high'] - df['low']
    
    # Calculate Close-Open Return
    df['Close_Open_Return'] = df['close'] - df['open']
    
    # Integrated Return
    df['Integrated_Return'] = df['Intraday_Spread'] + df['Close_Open_Return']
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    df['Smoothed_Integrated_Return'] = df['Integrated_Return'].ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume and Amount Confirmation
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    df['Amount_Ratio'] = df['amount'] / df['amount'].shift(1)
    df['Adjusted_Integrated_Return'] = (df['Integrated_Return'] * df['Volume_Ratio'] * df['Amount_Ratio'])
    df['Smoothing_Factor'] = 0.1 * (df['Adjusted_Integrated_Return'].shift(1) - df['Adjusted_Integrated_Return'])
    df['Final_Adjusted_Return'] = df['Adjusted_Integrated_Return'] + df['Smoothing_Factor']
    
    # Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Momentum'] = df['SMA_10'] - df['SMA_50']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (df['Smoothed_Integrated_Return'] + df['Final_Adjusted_Return']) * (df['Momentum'] * df['ATR']) + (df['volume'] - df['volume'].shift(1)) + (df['amount'] - df['amount'].shift(1))
    
    return df['Alpha_Factor']
```
