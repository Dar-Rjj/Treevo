```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['High-Low'] = df['high'] - df['low']
    df['High-PrevClose'] = abs(df['high'] - df['close'].shift(1))
    df['Low-PrevClose'] = abs(df['low'] - df['close'].shift(1))
    df['TR'] = df[['High-Low', 'High-PrevClose', 'Low-PrevClose']].max(axis=1)

    # Calculate Average True Range (ATR) over a 14-day window
    df['ATR'] = df['TR'].rolling(window=14).mean()

    # Calculate Integrated Return
    df['IntegratedReturn'] = (df['high'] - df['low']) + (df['close'] - df['open'])

    # Weighted Integrated Return
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    avg_amount_10 = df['amount'].rolling(window=10).mean()
    
    df['WeightedIntegratedReturn_Volume'] = np.where(
        df['volume'] > df['volume'].shift(1),
        df['IntegratedReturn'] * (df['volume'] / avg_volume_10), 
        0
    )
    
    df['WeightedIntegratedReturn_Amount'] = np.where(
        df['amount'] > df['amount'].shift(1),
        df['IntegratedReturn'] * (df['amount'] / avg_amount_10), 
        0
    )
    
    df['WeightedIntegratedReturn'] = df['WeightedIntegratedReturn_Volume'] + df['WeightedIntegratedReturn_Amount']

    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['PriceMomentum'] = df['SMA_10'] - df['SMA_50']

    # Final Alpha Factor
    df['AlphaFactor'] = df['WeightedIntegratedReturn'] * df['PriceMomentum'] * df['ATR']

    return df['AlphaFactor']
```
