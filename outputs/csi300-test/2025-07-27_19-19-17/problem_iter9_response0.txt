```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Calculate Open-to-Close Return
    open_to_close_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_return = intraday_return + open_to_close_return
    
    # Confirm with Volume and Amount
    # If Volume > Previous Day's Volume
    volume_weight = (df['volume'] / df['volume'].rolling(window=10).mean()).fillna(1)
    volume_weighted_return = integrated_return * (df['volume'] > df['volume'].shift(1)) * volume_weight
    
    # If Amount > Previous Day's Amount
    amount_weight = (df['amount'] / df['amount'].rolling(window=10).mean()).fillna(1)
    amount_weighted_return = integrated_return * (df['amount'] > df['amount'].shift(1)) * amount_weight
    
    # Combine Weights
    combined_weighted_return = (volume_weighted_return + amount_weighted_return) / 2.0
    
    return combined_weighted_return
```
