```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate True Range (TR)
    df['Prev_Close'] = df['close'].shift(1)
    df['High_Low'] = df['high'] - df['low']
    df['High_PrevClose'] = np.abs(df['high'] - df['Prev_Close'])
    df['Low_PrevClose'] = np.abs(df['low'] - df['Prev_Close'])
    df['TR'] = df[['High_Low', 'High_PrevClose', 'Low_PrevClose']].max(axis=1)
    
    # Calculate ATR over a window of 14 days
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Calculate Intraday Return and Open-to-Close Return
    df['Intraday_Return'] = df['high'] - df['low']
    df['Open_to_Close_Return'] = df['close'] - df['open']
    df['Integrated_Return'] = df['Intraday_Return'] * df['Open_to_Close_Return']
    
    # Adjust with Volume Factor
    df['Volume_Diff'] = df['volume'] - df['volume'].shift(1)
    df['Smoothed_Volume_Diff'] = df['Volume_Diff'].ewm(span=5, adjust=False).mean()
    
    # Combine ATR, Integrated Returns, and Volume Adjustment
    df['Adjusted_ATR'] = df['ATR'] * df['Integrated_Return'] + df['Smoothed_Volume_Diff']
    
    # Calculate Price Momentum
    df['SMA_10'] = df['close'].rolling(window=10).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['SMA_Diff'] = df['SMA_10'] - df['SMA_50']
    df['Price_Momentum'] = df['SMA_Diff'] * df['Adjusted_ATR']
    
    # Integrate High-Low and Close-Open for Additional Momentum
    df['Intraday_High_Low_Spread'] = df['high'] - df['low']
    df['Integrated_Momentum'] = df['Intraday_High_Low_Spread'] * df['Open_to_Close_Return']
    df['Smoothed_Integrated_Momentum'] = df['Integrated_Momentum'].ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    df['Combined_Factor'] = df['Price_Momentum'] + df['Smoothed_Integrated_Momentum']
    df['Volume_Confirmation'] = df['volume'] > df['volume'].shift(1)
    df['Volume_Adjusted_Combined_Factor'] = df['Combined_Factor'] / np.sqrt(df['volume']) * (df['volume'] / df['volume'].rolling(window=10).max())
    df['Final_Alpha_Factor'] = df['Adjusted_ATR'] + df['Price_Momentum'] + df['Volume_Adjusted_Combined_Factor']
    
    return df['Final_Alpha_Factor']
```
