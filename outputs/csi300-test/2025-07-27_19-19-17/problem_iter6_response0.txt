```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    ema_window = 5
    smoothed_returns = integrated_returns.ewm(span=ema_window, adjust=False).mean()
    
    # Confirm with Volume
    volume_confirmation = df['volume'] > df['volume'].shift(1)
    
    # Adjust for Volume
    adjusted_returns = smoothed_returns / np.sqrt(df['volume'])
    max_volume_10d = df['volume'].rolling(window=10).max()
    volume_adjusted_factor = adjusted_returns * (df['volume'] / max_volume_10d)
    
    # Apply Volume Confirmation
    final_factor = volume_adjusted_factor.where(volume_confirmation, other=np.nan)
    
    return final_factor
```
