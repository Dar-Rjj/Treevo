```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns (Weighted Sum of Intraday High-Low Spread and Close-Open Return)
    integrated_returns = 0.5 * intraday_high_low_spread + 0.5 * close_open_return
    
    # Smooth the Integrated Returns using EMA with a 5-day window
    ema_integrated_returns = integrated_returns.ewm(span=5, adjust=False).mean()
    
    # Adjust for Volume Confirmation
    volume_confirmation = df['volume'] > df['volume'].shift(1)
    adjusted_integrated_returns = ema_integrated_returns * (df['volume'] / df['volume'].shift(1))
    adjusted_integrated_returns[~volume_confirmation] = ema_integrated_returns[~volume_confirmation]
    
    # Calculate True Range (TR)
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR) over a 14-day window
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_integrated_returns = atr * adjusted_integrated_returns
    
    # Combine Factors (Weighted Integrated Return, ATR, and Price Momentum)
    price_momentum = df['close'].pct_change()
    combined_factor = 0.6 * atr_integrated_returns + 0.4 * price_momentum
    
    # Confirm with Volume
    volume_adjusted_combined_factor = combined_factor / np.sqrt(df['volume'])
    
    return volume_adjusted_combined_factor
```
