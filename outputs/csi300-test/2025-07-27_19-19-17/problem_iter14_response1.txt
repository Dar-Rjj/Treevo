```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    intraday_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-Open Return
    close_open_return = df['close'] - df['open']
    
    # Integrate Returns
    integrated_returns = intraday_high_low_spread + close_open_return
    
    # Adjust for Volume and Amount Confirmation
    volume_factor = (df['volume'] / df['volume'].shift(1)) * (df['amount'] / df['amount'].shift(1))
    adjusted_integrated_returns = integrated_returns * volume_factor
    
    # Calculate True Range (TR) for each day
    tr1 = df['high'] - df['low']
    tr2 = (df['high'] - df['close'].shift(1)).abs()
    tr3 = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate Average True Range (ATR)
    atr = true_range.rolling(window=14).mean()
    
    # Incorporate ATR in Integrated Returns
    atr_adjusted_returns = adjusted_integrated_returns * atr
    
    # Adjust with Volume Factor
    volume_diff = df['volume'] - df['volume'].shift(1)
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    normalized_volume_diff = volume_diff / avg_volume_20_days
    final_atr_adjusted_returns = atr_adjusted_returns + normalized_volume_diff
    
    # Calculate Price Momentum
    short_term_sma = df['close'].rolling(window=10).mean()
    long_term_sma = df['close'].rolling(window=50).mean()
    price_momentum = short_term_sma - long_term_sma
    atr_price_momentum = price_momentum * final_atr_adjusted_returns
    
    # Integrate High-Low and Close-Open for Additional Momentum
    intraday_high_low_spread = df['high'] - df['low']
    open_close_return = df['close'] - df['open']
    integrated_momentum = intraday_high_low_spread * open_close_return
    smoothed_integrated_momentum = integrated_momentum.ewm(span=5).mean()
    
    # Final Alpha Factor
    combined_factor = final_atr_adjusted_returns + atr_price_momentum + smoothed_integrated_momentum
    max_volume_10_days = df['volume'].rolling(window=10).max()
    volume_adjusted_factor = combined_factor / np.sqrt(df['volume']) * (df['volume'] / max_volume_10_days)
    
    return volume_adjusted_factor
```
