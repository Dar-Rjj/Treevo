```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=50, alpha=0.3, beta=0.4, gamma=0.3):
    # Obtain Close Prices and Volumes
    close_prices = df['close']
    volumes = df['volume']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = momentum.quantile(0.95)
    lower_threshold = momentum.quantile(0.05)
    
    # Clip Log Returns to Thresholds
    momentum_clipped = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volume_relative = volumes / volumes.rolling(window=M).mean()
    momentum_adjusted = momentum_clipped * volume_relative
    
    # Determine Absolute Price Changes
    abs_price_changes = np.abs(close_prices - close_prices.shift(1))
    
    # Calculate Advanced Volatility Measures
    std_volatility = abs_price_changes.rolling(window=M).std()
    ema_volatility = abs_price_changes.ewm(span=M).mean()
    iqr_volatility = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weighted_momentum = alpha * momentum_adjusted
    weighted_volume = beta * (volume_relative - 1)
    weighted_volatility = gamma * (ema_volatility + iqr_volatility)
    
    final_factor = weighted_momentum + weighted_volume + weighted_volatility
    
    # Ensure Weights Sum to 1
    final_factor = (final_factor - final_factor.min()) / (final_factor.max() - final_factor.min())
    
    return final_factor
```
