```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days_momentum=20, m_days_volatility=20):
    # Obtain close prices
    close_prices = df['close']
    
    # Compute log return over N days
    log_returns = np.log(close_prices).diff(n_days_momentum)
    
    # Calculate momentum
    momentum = log_returns
    
    # Retrieve volumes
    volume = df['volume']
    
    # Calculate relative volume (Volume / Moving Average of Volume)
    moving_avg_volume = volume.rolling(window=n_days_momentum).mean()
    relative_volume = volume / moving_avg_volume
    
    # Incorporate volume into momentum
    volume_adjusted_momentum = momentum * relative_volume
    
    # Determine absolute price changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate standard deviation of absolute price changes over M days
    volatility = abs_price_changes.rolling(window=m_days_volatility).std()
    
    # Final factor calculation
    alpha_factor = volume_adjusted_momentum + volatility
    
    # Ensure interpretability and novelty
    # The factor is a sum of volume-adjusted momentum and price change volatility,
    # which provides a balance between momentum and risk.
    
    return alpha_factor
```
