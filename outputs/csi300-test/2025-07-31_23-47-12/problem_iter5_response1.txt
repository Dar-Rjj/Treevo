```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, p=60, m=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Momentum
    log_returns = np.log(close_prices).diff(n)
    momentum = log_returns
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=p).mean()
    volume_ratio = volumes / mean_volume
    volume_adjusted_momentum = momentum * volume_ratio
    
    # Incorporate Price Change Volatility
    abs_price_changes = close_prices.diff().abs()
    volatility = abs_price_changes.rolling(window=m).std()
    scaled_volatility = volatility * 0.1  # Scaling factor to balance the components
    
    # Final Factor Calculation
    weight_momentum = 0.7
    weight_volatility = 0.3
    final_factor = (weight_momentum * volume_adjusted_momentum) + (weight_volatility * scaled_volatility)
    
    return final_factor
```
