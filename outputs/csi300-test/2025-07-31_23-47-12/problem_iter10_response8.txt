```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20, alpha=0.5, beta=0.3, gamma=0.2):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    upper_threshold = cum_log_returns.quantile(0.95)
    lower_threshold = cum_log_returns.quantile(0.05)
    momentum = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    volatility_measures = (std_dev_abs_price_changes + ema_abs_price_changes + iqr_abs_price_changes) / 3
    
    # Integrate Market Sentiment (Placeholder for actual sentiment data)
    sentiment_index = pd.Series(np.random.randn(len(df)), index=df.index)
    weighted_momentum_sentiment = (momentum * alpha) + (sentiment_index * (1 - alpha))
    
    # Refined Volatility Measures
    high_low_spread = df['high'] - df['low']
    open_close_spread = df['open'] - df['close']
    std_high_low_spread = high_low_spread.rolling(window=M).std()
    std_open_close_spread = open_close_spread.rolling(window=M).std()
    
    refined_volatility = (volatility_measures + std_high_low_spread + std_open_close_spread) / 3
    
    # Final Factor Calculation
    final_factor = (weighted_momentum_sentiment * beta) + (volume_adjusted_momentum * (1 - beta - gamma)) + (refined_volatility * gamma)
    
    return final_factor
```
