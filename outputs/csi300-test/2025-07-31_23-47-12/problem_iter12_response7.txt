```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, P=50):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    
    # Compute Momentum
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    
    # Apply Dynamic Thresholds
    upper_threshold = cumulative_log_returns.mean() + 2 * cumulative_log_returns.std()
    lower_threshold = cumulative_log_returns.mean() - 2 * cumulative_log_returns.std()
    clipped_cumulative_returns = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=P).mean()
    volume_ratio = volumes / mean_volume
    volume_adjusted_momentum = clipped_cumulative_returns * volume_ratio
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    ema_volatility = absolute_price_changes.ewm(span=N, adjust=False).mean()
    
    # Final Factor Calculation
    # Assign Weights to Each Component
    weight_momentum = 0.4
    weight_volume = 0.3
    weight_volatility = 0.3
    
    # Sum of Weighted Components
    factor_values = (weight_momentum * volume_adjusted_momentum +
                     weight_volume * volume_ratio +
                     weight_volatility * ema_volatility)
    
    return factor_values
```
