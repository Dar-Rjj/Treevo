```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=20, momentum_thresholds=(-0.05, 0.05), volume_weight=0.3, volatility_weight=0.7):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(n_days)
    
    # Compute Momentum
    cumulative_log_returns = log_returns.cumsum()
    upper_threshold, lower_threshold = momentum_thresholds
    clipped_cumulative_log_returns = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=n_days).mean()
    volume_adjusted_momentum = (volumes / mean_volume) * clipped_cumulative_log_returns
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_volatility = absolute_price_changes.rolling(window=m_days).std()
    ema_volatility = absolute_price_changes.ewm(span=m_days, adjust=False).mean()
    iqr_volatility = absolute_price_changes.rolling(window=m_days).quantile(0.75) - absolute_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Final Factor Calculation
    weighted_momentum = volume_adjusted_momentum
    weighted_volatility = (std_dev_volatility + ema_volatility + iqr_volatility) / 3
    final_factor = (volume_weight * weighted_momentum) + (volatility_weight * weighted_volatility)
    
    return final_factor
```
