import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=30, k_days=15, l_days=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=n_days).sum()
    
    # Define Dynamic Thresholds
    rolling_mean = log_returns.rolling(window=n_days).mean()
    rolling_std = log_returns.rolling(window=n_days).std()
    upper_threshold = rolling_mean + 2 * rolling_std
    lower_threshold = rolling_mean - 2 * rolling_std
    
    # Clip Log Returns to Thresholds
    clipped_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=n_days).mean()
    volume_adjusted_momentum = (volumes / mean_volume) * clipped_log_returns
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_volatility = abs_price_changes.rolling(window=m_days).std()
    ema_volatility = abs_price_changes.ewm(span=k_days, adjust=False).mean()
    iqr_volatility = abs_price_changes.rolling(window=l_days).quantile(0.75) - abs_price_changes.rolling(window=l_days).quantile(0.25)
    
    # Final Factor Calculation
    weights = np.array([0.4, 0.3, 0.3])
    combined_factor = (weights[0] * volume_adjusted_momentum +
                       weights[1] * std_dev_volatility +
                       weights[2] * ema_volatility)
    
    # Integrate Market Breadth and Economic Indicators
    advance_decline_line = (df['close'] > df['open']).astype(int).cumsum()
    new_highs = (df['high'] == df['high'].rolling(window=252).max()).astype(int)
    new_lows = (df['low'] == df['low'].rolling(window=252).min()).astype(int)
    market_breadth = advance_decline_line + new_highs - new_lows
    combined_factor += 0.2 * market_breadth
    
    # Incorporate Key Economic Indicators (Assuming these are available in the DataFrame)
    if 'interest_rate' in df.columns:
        combined_factor += 0.1 * df['interest_rate']
    if 'gdp_growth_rate' in df.columns:
        combined_factor += 0.1 * df['gdp_growth_rate']
    if 'inflation_rate' in df.columns:
        combined_factor += 0.1 * df['inflation_rate']
    
    return combined_factor.dropna()
