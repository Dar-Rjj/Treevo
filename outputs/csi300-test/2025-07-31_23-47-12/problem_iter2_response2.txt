```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Momentum
    daily_momentum = df['close'].diff()
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'].diff()
    
    # Combine Price and Volume Momentum
    combined_price_momentum = daily_momentum * df['close']
    combined_volume_momentum = volume_momentum * df['volume']
    
    # Calculate Intersection Influence
    intersection_influence = (combined_price_momentum + combined_volume_momentum) / (df['close'] + df['volume'])
    
    # Calculate Volatility (Average True Range)
    def calculate_atr(high, low, close, period=14):
        high_low = high - low
        high_close = np.abs(high - close.shift())
        low_close = np.abs(low - close.shift())
        tr = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1)
        atr = tr.rolling(window=period).mean()
        return atr
    
    atr = calculate_atr(df['high'], df['low'], df['close'])
    volatility_adjusted_influence = intersection_influence / atr
    volatility_adjusted_influence = np.where(atr > 0, volatility_adjusted_influence, 0)
    
    # Calculate Trend Strength
    short_sma = df['close'].rolling(window=10).mean()
    long_sma = df['close'].rolling(window=50).mean()
    trend_strength = short_sma - long_sma
    
    # Adjust Intersection Influence by Trend
    trend_adjusted_influence = np.where(trend_strength > 0, volatility_adjusted_influence * trend_strength, 0)
    
    # Refine Intersection Score
    refined_intersection_score = trend_adjusted_influence - trend_adjusted_influence.min()
    refined_intersection_score = np.where(refined_intersection_score >= 0, refined_intersection_score, 0)
    
    return pd.Series(refined_intersection_score, index=df.index)
```
