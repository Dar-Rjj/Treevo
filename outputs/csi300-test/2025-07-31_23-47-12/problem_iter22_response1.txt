```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, momentum_weight=0.5, volume_weight=0.3, volatility_weight=0.2):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    upper_threshold = cum_log_returns.quantile(0.75)
    lower_threshold = cum_log_returns.quantile(0.25)
    momentum = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    relative_volume = volumes / mean_volume
    volume_adjusted_momentum = momentum * relative_volume
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = abs_price_changes.rolling(window=M).std()
    ema = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    volatility = (std_dev + ema + iqr) / 3
    
    # Final Factor Calculation
    weights_sum = momentum_weight + volume_weight + volatility_weight
    factor = (momentum_weight * volume_adjusted_momentum + 
              volume_weight * relative_volume + 
              volatility_weight * volatility) / weights_sum
    
    return factor
```
