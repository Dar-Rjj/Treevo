```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Cumulative Log Returns Over N Days
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    
    # Compute Volume-Adjusted Momentum
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_relative = volumes / mean_volume
    volume_adjusted_momentum = cumulative_log_returns * volume_relative
    
    # Incorporate Price Change Volatility
    absolute_price_changes = abs(close_prices.diff())
    volatility = absolute_price_changes.rolling(window=M).std()
    
    # Assign Dynamic Weights to Volume-Adjusted Momentum and Volatility
    weight_momentum = 0.6
    weight_volatility = 0.4
    assert weight_momentum + weight_volatility == 1.0, "Weights must sum to 1"
    
    # Final Factor Calculation
    factor_values = (weight_momentum * volume_adjusted_momentum) + (weight_volatility * volatility)
    
    return factor_values
```
