```python
import pandas as pd

def heuristics_v2(df, period=14):
    # Compute Up Move and Down Move
    df['Up Move'] = df['High'] - df['Low'].shift(1)
    df['Down Move'] = df['Low'] - df['High'].shift(1)
    
    # Determine Positive and Negative Directional Indicators
    df['+DM'] = df['Up Move'].apply(lambda x: max(x, 0))
    df['-DM'] = df['Down Move'].apply(lambda x: max(x, 0))
    
    # Calculate True Range
    df['High-Low'] = df['High'] - df['Low']
    df['High-Close'] = (df['High'] - df['Close'].shift(1)).abs()
    df['Close-Low'] = (df['Close'].shift(1) - df['Low']).abs()
    df['True Range'] = df[['High-Low', 'High-Close', 'Close-Low']].max(axis=1)
    
    # Sum Positive and Negative DMs over the period
    df['+DM_sum'] = df['+DM'].rolling(window=period).sum()
    df['-DM_sum'] = df['-DM'].rolling(window=period).sum()
    
    # Sum of True Ranges over the period
    df['TR_sum'] = df['True Range'].rolling(window=period).sum()
    
    # Compute Directional Indicator
    df['+DI'] = 100 * (df['+DM_sum'] / df['TR_sum'])
    df['-DI'] = 100 * (df['-DM_sum'] / df['TR_sum'])
    
    # Derive Cumulative Directional Movement
    df['CDM'] = (df['+DI'] - df['-DI']) * df['Volume']
    
    # Generate Alpha Factor
    alpha_factor = df['CDM']
    
    return alpha_factor
```
