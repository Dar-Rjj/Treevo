import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=10):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(n_days)
    
    # Compute Momentum
    momentum = log_returns.cumsum()
    
    # Define dynamic thresholds
    upper_threshold = momentum.quantile(0.95)
    lower_threshold = momentum.quantile(0.05)
    momentum_clipped = np.clip(momentum, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_mean = volumes.rolling(window=m_days).mean()
    volume_adjusted_momentum = momentum_clipped * (volumes / volume_mean)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = abs_price_changes.rolling(window=m_days).std()
    ema = abs_price_changes.ewm(span=m_days, adjust=False).mean()
    iqr = abs_price_changes.rolling(window=m_days).quantile(0.75) - abs_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Final Factor Calculation
    weights = np.array([0.4, 0.3, 0.3])
    factor = (weights[0] * volume_adjusted_momentum + 
              weights[1] * std_dev + 
              weights[2] * ema)
    
    # Ensure Weights Sum to 1
    factor = factor / weights.sum()
    
    # Integrate Market Breadth and Economic Indicators
    # For simplicity, assume we have a DataFrame with market breadth indicators and economic indicators
    # Example: advance_decline_line, new_highs, new_lows, interest_rates, gdp_growth_rate
    # market_breadth_df = ...
    # economic_indicators_df = ...
    
    # Add market breadth and economic indicators to the final factor
    # factor += 0.1 * market_breadth_df['advance_decline_line'] + 0.1 * (economic_indicators_df['interest_rates'] + economic_indicators_df['gdp_growth_rate'])
    
    return factor
