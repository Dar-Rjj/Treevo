```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Momentum
    df['High_Low_Momentum'] = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Volume Surge Indicator
    df['Volume_Change'] = df['volume'].pct_change()
    volatility = df['close'].rolling(window=20).std()
    
    # Adaptive Threshold for Volume Surge
    volume_surge_threshold = 15 * (volatility > 0.02) + 10 * (volatility <= 0.02)
    df['Volume_Surge'] = df['Volume_Change'] > (volume_surge_threshold / 100)
    
    # Combine High-Low Momentum, Volume Surge, and Volatility
    df['Factor'] = df.apply(
        lambda row: row['High_Low_Momentum'] * (1 + 0.1 * volatility.loc[row.name]) if row['Volume_Surge'] else 0,
        axis=1
    )
    
    return df['Factor']
```
