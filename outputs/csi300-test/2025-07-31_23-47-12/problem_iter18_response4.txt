```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=60, M=20, alpha=0.1):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    upper_threshold = cum_log_returns.quantile(0.95)
    lower_threshold = cum_log_returns.quantile(0.05)
    momentum = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = abs_price_changes.rolling(window=M).std()
    ema_abs_changes = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Integrate Market Breadth Indicators
    advances = (df['close'] > df['open']).astype(int)
    declines = (df['close'] < df['open']).astype(int)
    advance_decline_line = (advances - declines).cumsum()
    
    new_highs = (df['high'] == df['high'].rolling(window=N).max()).astype(int)
    new_lows = (df['low'] == df['low'].rolling(window=N).min()).astype(int)
    net_new_highs_lows = new_highs - new_lows
    
    # Final Factor Calculation
    weighted_components = {
        'momentum': momentum,
        'volume_adjusted_momentum': volume_adjusted_momentum,
        'volatility_std': std_dev,
        'volatility_ema': ema_abs_changes,
        'volatility_iqr': iqr,
        'advance_decline_line': advance_decline_line,
        'net_new_highs_lows': net_new_highs_lows
    }
    
    weights = {
        'momentum': 0.2,
        'volume_adjusted_momentum': 0.3,
        'volatility_std': 0.1,
        'volatility_ema': 0.1,
        'volatility_iqr': 0.1,
        'advance_decline_line': 0.1,
        'net_new_highs_lows': 0.1
    }
    
    factor_values = sum(weights[comp] * weighted_components[comp] for comp in weighted_components)
    
    return factor_values
```
