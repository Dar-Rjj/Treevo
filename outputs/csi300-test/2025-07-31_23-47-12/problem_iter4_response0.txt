```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, m_days=20, momentum_weight=0.5, volatility_weight=0.5):
    # Calculate Logarithmic Return Momentum
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(n_days)
    
    # Adjust for Volume
    volumes = df['volume']
    avg_volume = volumes.rolling(window=n_days).mean()
    relative_volume = volumes / avg_volume
    volume_adjusted_momentum = log_returns * relative_volume
    
    # Calculate Price Change Volatility
    price_changes = close_prices.diff().abs()
    volatility = price_changes.rolling(window=m_days).std()
    
    # Combine Momentum and Volatility
    alpha_factor = (momentum_weight * volume_adjusted_momentum) + (volatility_weight * volatility)
    
    # Ensure Non-Negative Results
    alpha_factor = alpha_factor.clip(lower=0)
    
    return alpha_factor
```
