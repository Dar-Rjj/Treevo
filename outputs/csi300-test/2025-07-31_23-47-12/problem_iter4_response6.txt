```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Obtain close prices and volumes
    close_prices = df['close']
    volumes = df['volume']
    
    # Calculate multi-day logarithmic returns (N days)
    N = 5
    log_returns = np.log(close_prices).diff(N)
    
    # Compute average volume over N days
    avg_volume = volumes.rolling(window=N).mean()
    
    # Multiply logarithmic returns by average volume
    volume_adjusted_returns = log_returns * avg_volume
    
    # Determine absolute price changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate standard deviation of absolute price changes over M days
    M = 20
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    
    # Adjust for recent volatility
    P = 10
    Q = 20
    moving_avg_close = close_prices.rolling(window=P).mean()
    std_close = close_prices.rolling(window=Q).std()
    recent_volatility = std_close / moving_avg_close
    
    # Dynamic threshold for volume impact
    volatility_threshold = 0.02
    volume_impact_factor = np.where(recent_volatility > volatility_threshold, 0.8, 1.2)
    
    # Final factor calculation
    adjusted_momentum = volume_adjusted_returns * volume_impact_factor
    volatility_component = std_abs_price_changes
    alpha_factor = adjusted_momentum + volatility_component
    
    return alpha_factor
```
