```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Calculate Momentum
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_relative = volumes / volumes.rolling(window=N).mean()
    volume_adjusted_momentum = log_returns * volume_relative
    
    # Incorporate Price Change Volatility
    abs_price_changes = close_prices.diff().abs()
    volatility = abs_price_changes.rolling(window=M).std()
    
    # Final Factor Calculation
    momentum_weight = 0.7
    volatility_weight = 0.3
    factor = (momentum_weight * volume_adjusted_momentum) + (volatility_weight * volatility)
    
    return factor
```
