```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=14):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    upper_threshold = 0.05
    lower_threshold = -0.05
    momentum_clipped = np.clip(momentum, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = (volumes / mean_volume) * momentum_clipped
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_volatility = abs_price_changes.rolling(window=M).std()
    ema_volatility = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr_volatility = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Integrate Market Breadth Indicators
    advance_decline_line = (close_prices > close_prices.shift(1)).astype(int).cumsum() - (close_prices < close_prices.shift(1)).astype(int).cumsum()
    new_highs = (close_prices == close_prices.rolling(window=252).max()).astype(int)
    new_lows = (close_prices == close_prices.rolling(window=252).min()).astype(int)
    
    # Incorporate Key Economic Indicators
    # For simplicity, we assume these indicators are provided as external data
    interest_rates = df.get('interest_rates', 1.0)  # Placeholder: Use 1.0 if not provided
    gdp_growth_rate = df.get('gdp_growth_rate', 1.0)  # Placeholder: Use 1.0 if not provided
    
    # Final Factor Calculation
    weights = {
        'momentum': 0.3,
        'volatility': 0.2,
        'market_breadth': 0.2,
        'economic_indicators': 0.3
    }
    
    factor = (
        weights['momentum'] * volume_adjusted_momentum +
        weights['volatility'] * (std_volatility + ema_volatility + iqr_volatility) / 3 +
        weights['market_breadth'] * (advance_decline_line + new_highs - new_lows) +
        weights['economic_indicators'] * (interest_rates + gdp_growth_rate) / 2
    )
    
    return factor
```
