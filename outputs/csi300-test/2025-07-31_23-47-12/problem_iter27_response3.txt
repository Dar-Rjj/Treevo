import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=20, interest_rate=0.02, gdp_growth=0.03):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(n_days)
    
    # Calculate Momentum
    cum_sum_log_returns = log_returns.cumsum()
    upper_threshold = np.percentile(cum_sum_log_returns, 75)
    lower_threshold = np.percentile(cum_sum_log_returns, 25)
    clipped_log_returns = np.clip(cum_sum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=n_days).mean()
    volume_adjusted_momentum = clipped_log_returns * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_abs_price_changes = abs_price_changes.rolling(window=m_days).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=m_days, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=m_days).quantile(0.75) - abs_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Integrate Market Breadth and Economic Indicators
    advance_decline_line = (df['close'] > df['close'].shift(1)).cumsum() - (df['close'] < df['close'].shift(1)).cumsum()
    new_highs = (df['high'] == df['high'].rolling(window=20).max()).astype(int)
    new_lows = (df['low'] == df['low'].rolling(window=20).min()).astype(int)
    economic_indicators = (interest_rate + gdp_growth) / 2
    
    # Final Factor Calculation
    weighted_components = (
        0.4 * volume_adjusted_momentum +
        0.3 * (std_dev_abs_price_changes + ema_abs_price_changes + iqr_abs_price_changes) / 3 +
        0.2 * (advance_decline_line + new_highs - new_lows) / 3 +
        0.1 * economic_indicators
    )
    
    return weighted_components
