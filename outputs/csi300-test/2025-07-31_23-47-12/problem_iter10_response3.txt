import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=30, volume_window=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    momentum = log_returns.rolling(window=n_days).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = momentum.quantile(0.95)
    lower_threshold = momentum.quantile(0.05)
    
    # Clip Log Returns to Thresholds
    momentum = np.clip(momentum, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=volume_window).mean()
    volume_adjusted_momentum = momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_volatility = absolute_price_changes.rolling(window=m_days).std()
    ema_volatility = absolute_price_changes.ewm(span=m_days).mean()
    iqr_volatility = absolute_price_changes.rolling(window=m_days).quantile(0.75) - absolute_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weight_momentum = 0.4
    weight_volume = 0.3
    weight_volatility = 0.3
    
    final_factor = (weight_momentum * volume_adjusted_momentum +
                    weight_volume * (volumes / mean_volume) +
                    weight_volatility * (std_dev_volatility + ema_volatility + iqr_volatility) / 3)
    
    return final_factor
