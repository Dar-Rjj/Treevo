```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, P=10, M=30, momentum_weight=0.6, volatility_weight=0.4):
    # Ensure the sum of weights is 1
    assert momentum_weight + volatility_weight == 1, "Weights must sum to 1"
    
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Momentum
    log_returns = np.log(close_prices).diff(N)
    momentum = log_returns
    
    # Adjust for Volume
    volumes = df['volume']
    avg_volume = volumes.rolling(window=P).mean()
    volume_ratio = volumes / avg_volume
    volume_adjusted_momentum = momentum * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    price_change_volatility = abs_price_changes.rolling(window=M).std()
    
    # Final Factor Calculation
    final_factor = (momentum_weight * volume_adjusted_momentum) - (volatility_weight * price_change_volatility)
    
    return final_factor
```
