```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(N)
    
    # Retrieve Volumes and Calculate Volume Relative to its Own Mean
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_ratio = volumes / mean_volume
    
    # Incorporate Volume into Momentum
    momentum = log_returns * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Standard Deviation over M Days for Volatility
    volatility = abs_price_changes.rolling(window=M).std()
    
    # Final Factor Calculation: Combine Momentum and Volatility Component
    final_factor = momentum / (volatility + 1e-8)  # Adding a small constant to avoid division by zero
    
    return final_factor
```
