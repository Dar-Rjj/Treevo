```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate True Range
    true_range = pd.concat([
        (df['high'] - df['low']),
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ], axis=1).max(axis=1)
    
    # Calculate Average True Range over 10 Days
    atr_10 = true_range.rolling(window=10).mean()
    
    # Normalize the High-Low Range by the Average True Range
    normalized_high_low = high_low_range / atr_10
    
    # Calculate Volume Trend
    volume_trend = df['volume']
    volume_ma_5 = volume_trend.rolling(window=5).mean()
    
    # Normalize the Volume by its 5-Day Moving Average
    normalized_volume = volume_trend / volume_ma_5
    
    # Combine the Factors
    alpha_factor = normalized_high_low * normalized_volume
    
    return alpha_factor
```
