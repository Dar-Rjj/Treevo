```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, m_days=30, threshold_upper=0.05, threshold_lower=-0.05):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(n_days)
    
    # Calculate Momentum
    cum_log_returns = log_returns.cumsum()
    
    # Apply Dynamic Thresholds
    clipped_cum_log_returns = cum_log_returns.clip(lower=threshold_lower, upper=threshold_upper)
    
    # Adjust for Volume
    volume = df['volume']
    mean_volume = volume.rolling(window=n_days).mean()
    volume_adjusted_momentum = (volume / mean_volume) * clipped_cum_log_returns
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_abs_price_changes = abs_price_changes.rolling(window=m_days).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=m_days, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=m_days).quantile(0.75) - abs_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weights = np.array([0.4, 0.3, 0.3])
    weighted_components = (
        weights[0] * volume_adjusted_momentum +
        weights[1] * std_dev_abs_price_changes +
        weights[2] * (ema_abs_price_changes + iqr_abs_price_changes)
    )
    
    # Ensure Weights Sum to 1
    assert np.isclose(weights.sum(), 1), "Weights must sum to 1"
    
    # Integrate Market Breadth Indicators
    advance_decline_line = (df['close'] > df['open']).astype(int).cumsum()
    new_highs = (close_prices == close_prices.rolling(window=m_days).max()).astype(int)
    new_lows = (close_prices == close_prices.rolling(window=m_days).min()).astype(int)
    
    # Final Factor Calculation
    factor_values = weighted_components + 0.05 * (advance_decline_line + new_highs - new_lows)
    
    return factor_values
```
