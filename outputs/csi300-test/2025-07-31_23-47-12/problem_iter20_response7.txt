import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, K=5, L=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Returns
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    mean_log_returns = log_returns.rolling(window=N).mean()
    std_log_returns = log_returns.rolling(window=N).std()
    upper_threshold = mean_log_returns + 2 * std_log_returns
    lower_threshold = mean_log_returns - 2 * std_log_returns
    clipped_cum_log_returns = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = clipped_cum_log_returns * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=K).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=L).quantile(0.75) - abs_price_changes.rolling(window=L).quantile(0.25)
    
    # Final Factor Calculation
    weights = [0.3, 0.4, 0.3]  # Weights for momentum, volume, and volatility
    combined_factor = (weights[0] * volume_adjusted_momentum +
                       weights[1] * std_abs_price_changes +
                       weights[2] * ema_abs_price_changes)
    
    return pd.Series(combined_factor, index=df.index)
