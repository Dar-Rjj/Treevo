```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, upper_threshold=0.05, lower_threshold=-0.05, volume_window=20, ema_span=20, interest_rate=None, gdp_growth_rate=None):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff(n_days)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    clipped_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=volume_window).mean()
    volume_adjusted_momentum = (volumes / mean_volume) * clipped_log_returns
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    ema_volatility = abs_price_changes.ewm(span=ema_span, adjust=False).mean()
    
    # Final Factor Calculation
    weights = {'momentum': 0.4, 'volume': 0.3, 'volatility': 0.3}
    combined_factor = (weights['momentum'] * volume_adjusted_momentum +
                       weights['volume'] * (volumes / mean_volume) +
                       weights['volatility'] * ema_volatility)
    
    if interest_rate is not None and gdp_growth_rate is not None:
        combined_factor += interest_rate * 0.1 + gdp_growth_rate * 0.1
        combined_factor *= 0.8  # Ensure the factor remains within a reasonable range
    
    return combined_factor
```
