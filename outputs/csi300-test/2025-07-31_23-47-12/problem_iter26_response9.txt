```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, momentum_threshold=0.05, volume_lookback=20, ema_span=10, interest_rate=0.01, gdp_growth=0.02):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(n_days)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    clipped_cum_log_returns = np.clip(cum_log_returns, -momentum_threshold, momentum_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=volume_lookback).mean()
    volume_adjusted_momentum = clipped_cum_log_returns * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    ema_volatility = absolute_price_changes.ewm(span=ema_span, adjust=False).mean()
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.2, 0.1]  # Weights for momentum, volume, EMA volatility, and economic indicators
    factor = (
        weights[0] * volume_adjusted_momentum +
        weights[1] * (volumes / mean_volume) +
        weights[2] * ema_volatility +
        weights[3] * (interest_rate + gdp_growth)
    )
    
    return factor
```
