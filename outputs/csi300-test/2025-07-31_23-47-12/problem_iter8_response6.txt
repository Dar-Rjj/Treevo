```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain Close Prices
    close = df['close']
    
    # Compute Log Return over N Days
    log_return = np.log(close) - np.log(close.shift(1))
    momentum = log_return.rolling(window=N).sum()
    
    # Adjust for Volume
    volume = df['volume']
    moving_avg_volume = volume.rolling(window=N).mean()
    volume_ratio = volume / moving_avg_volume
    weighted_momentum = momentum * volume_ratio
    
    # Incorporate Price Change Volatility
    abs_price_changes = np.abs(close - close.shift(1))
    ema_volatility = abs_price_changes.ewm(span=M, adjust=False).mean()
    
    # Combine Weighted Momentum and Volatility Component
    weight_momentum = 0.6
    weight_volatility = 0.4
    final_factor = (weighted_momentum * weight_momentum) + (ema_volatility * weight_volatility)
    
    return final_factor
```
