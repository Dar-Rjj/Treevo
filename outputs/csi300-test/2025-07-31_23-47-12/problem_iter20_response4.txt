import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, K=5, L=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    rolling_mean = log_returns.rolling(window=N).mean()
    rolling_std = log_returns.rolling(window=N).std()
    upper_threshold = rolling_mean + 2 * rolling_std
    lower_threshold = rolling_mean - 2 * rolling_std
    
    # Clip Log Returns to Thresholds
    clipped_log_returns = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_mean = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = clipped_log_returns * (volumes / volume_mean)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=K).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=L).quantile(0.75) - abs_price_changes.rolling(window=L).quantile(0.25)
    
    # Final Factor Calculation
    # Weights for the components
    weights = [0.4, 0.3, 0.3]
    combined_factor = (weights[0] * volume_adjusted_momentum + 
                       weights[1] * std_abs_price_changes + 
                       weights[2] * iqr_abs_price_changes)
    
    # Integrate Market Breadth and Economic Indicators
    # For simplicity, we will use a dummy market breadth indicator
    advance_decline_line = (close_prices > close_prices.shift(1)).rolling(window=N).sum() - (close_prices < close_prices.shift(1)).rolling(window=N).sum()
    new_highs = (close_prices == close_prices.rolling(window=N).max()).rolling(window=N).sum()
    new_lows = (close_prices == close_prices.rolling(window=N).min()).rolling(window=N).sum()
    
    # Dummy economic indicators
    interest_rates = 0.05  # Example value
    gdp_growth_rate = 0.02  # Example value
    
    # Combine all factors
    final_factor = combined_factor + 0.1 * advance_decline_line + 0.1 * new_highs - 0.1 * new_lows + 0.1 * interest_rates + 0.1 * gdp_growth_rate
    
    return final_factor
