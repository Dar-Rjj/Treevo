```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, p_days=60, m_days=30, momentum_weight=0.7, volatility_weight=0.3):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    momentum = log_returns.rolling(window=n_days).sum()
    smoothed_momentum = momentum.ewm(span=n_days, adjust=False).mean()
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=p_days).mean()
    relative_volume = volumes / mean_volume
    volume_adjusted_momentum = smoothed_momentum * relative_volume
    
    # Incorporate Price Change Volatility
    abs_price_changes = close_prices.diff().abs()
    volatility = abs_price_changes.rolling(window=m_days).std()
    # Normalize Volatility Component (Ensure No Normalization Used)
    # In this case, we do not normalize the volatility component to [0, 1] range.
    
    # Final Factor Calculation
    factor_values = (momentum_weight * volume_adjusted_momentum) + (volatility_weight * volatility)
    
    return factor_values
```
