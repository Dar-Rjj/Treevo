```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(N)
    
    # Calculate Momentum
    momentum = log_returns
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Incorporate Volume into Momentum
    mean_volume = volumes.rolling(window=N).mean()
    volume_ratio = volumes / mean_volume
    volume_adjusted_momentum = momentum * volume_ratio
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Standard Deviation of Absolute Price Changes over M Days
    price_change_volatility = absolute_price_changes.rolling(window=M).std()
    
    # Final Factor Calculation
    alpha_factor = volume_adjusted_momentum - price_change_volatility
    
    return alpha_factor
```
