```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, interest_rates=None, gdp_growth_rate=None):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    upper_threshold = cum_log_returns.quantile(0.95)
    lower_threshold = cum_log_returns.quantile(0.05)
    clipped_momentum = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.mean()
    volume_adjusted_momentum = clipped_momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate EMA Volatility
    ema_volatility = abs_price_changes.ewm(span=M).mean()
    
    # Final Factor Calculation
    weights = [0.6, 0.3, 0.1]  # Weights for momentum, volume, and EMA volatility components
    combined_factor = (
        weights[0] * volume_adjusted_momentum + 
        weights[1] * ema_volatility + 
        weights[2] * (interest_rates if interest_rates is not None else 0) +
        (gdp_growth_rate if gdp_growth_rate is not None else 0)
    )
    
    return combined_factor
```
