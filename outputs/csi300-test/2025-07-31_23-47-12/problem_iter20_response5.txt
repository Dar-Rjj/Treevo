```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=30, K=10, L=50):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    rolling_mean_lr = log_returns.rolling(window=N).mean()
    rolling_std_lr = log_returns.rolling(window=N).std()
    upper_threshold = rolling_mean_lr + 2 * rolling_std_lr
    lower_threshold = rolling_mean_lr - 2 * rolling_std_lr
    clipped_momentum = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    rolling_mean_vol = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = (volumes / rolling_mean_vol) * clipped_momentum
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_changes = abs_price_changes.ewm(span=K, min_periods=K).mean()
    iqr_abs_changes = abs_price_changes.rolling(window=L).quantile(0.75) - abs_price_changes.rolling(window=L).quantile(0.25)
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.3]  # Weights for momentum, volatility, and IQR
    combined_factor = (
        weights[0] * volume_adjusted_momentum +
        weights[1] * std_abs_changes +
        weights[2] * iqr_abs_changes
    )
    
    return combined_factor
```
