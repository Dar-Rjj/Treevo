```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, m_days=20, momentum_weight=0.7, volatility_weight=0.3):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(n_days)
    
    # Calculate Momentum
    momentum = log_returns
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Calculate Volume Ratio (Current Volume / Moving Average Volume)
    volume_ratio = volumes / volumes.rolling(window=n_days).mean()
    
    # Incorporate Volume into Momentum
    volume_adjusted_momentum = momentum * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Standard Deviation over M Days for Volatility
    volatility = abs_price_changes.rolling(window=m_days).std()
    
    # Final Factor Calculation
    factor = (momentum_weight * volume_adjusted_momentum) + (volatility_weight * volatility)
    
    return factor
```
