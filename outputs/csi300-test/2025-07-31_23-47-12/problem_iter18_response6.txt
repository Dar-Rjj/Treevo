```python
import pandas as pd
import numpy as np
from scipy.stats import kurtosis, skew

def heuristics_v2(data: pd.DataFrame, N: int, M: int) -> pd.Series:
    # Obtain Close Prices
    close_prices = data['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = 0.05
    lower_threshold = -0.05
    clipped_cum_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = data['volume']
    avg_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = clipped_cum_log_returns * (volumes / avg_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_volatility = abs_price_changes.rolling(window=M).std()
    ema_volatility = abs_price_changes.ewm(span=M, min_periods=M).mean()
    iqr_volatility = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    kurtosis_volatility = abs_price_changes.rolling(window=M).apply(kurtosis, raw=False)
    skewness_volatility = abs_price_changes.rolling(window=M).apply(skew, raw=False)
    
    # Integrate Market Breadth and Economic Indicators
    # For simplicity, we will use dummy values for economic indicators
    interest_rates = pd.Series(np.random.randn(len(data)), index=data.index)
    gdp_growth_rate = pd.Series(np.random.randn(len(data)), index=data.index)
    inflation_rate = pd.Series(np.random.randn(len(data)), index=data.index)
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.1, 0.1, 0.05, 0.05]
    factor_values = (
        weights[0] * volume_adjusted_momentum +
        weights[1] * std_dev_volatility +
        weights[2] * ema_volatility +
        weights[3] * iqr_volatility +
        weights[4] * kurtosis_volatility +
        weights[5] * skewness_volatility
    )
    
    # Ensure Weights Sum to 1
    assert sum(weights) == 1.0
    
    return factor_values
```
