```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Calculate Log Returns
    log_returns = np.log(df['close']).diff().fillna(0)
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = cum_log_returns.mean() + 2 * cum_log_returns.std()
    lower_threshold = cum_log_returns.mean() - 2 * cum_log_returns.std()
    
    # Clip Log Returns to Thresholds
    clipped_momentum = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    mean_volume = df['volume'].rolling(window=N).mean()
    relative_volume = df['volume'] / mean_volume
    volume_adjusted_momentum = clipped_momentum * relative_volume
    
    # Determine Absolute Price Changes
    abs_price_changes = df['close'].diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Final Factor Calculation
    # Weights for the components
    weights = {
        'momentum': 0.4,
        'volume': 0.3,
        'volatility_std': 0.15,
        'volatility_ema': 0.1,
        'volatility_iqr': 0.05
    }
    
    # Ensure Weights Sum to 1
    assert sum(weights.values()) == 1, "Weights must sum to 1"
    
    # Combine Weighted Components
    factor = (weights['momentum'] * volume_adjusted_momentum +
              weights['volume'] * relative_volume +
              weights['volatility_std'] * std_abs_price_changes +
              weights['volatility_ema'] * ema_abs_price_changes +
              weights['volatility_iqr'] * iqr_abs_price_changes)
    
    return factor
```
