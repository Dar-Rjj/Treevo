```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Price Momentum
    close = data['close']
    momentum = (close - close.shift(10)) / close.shift(10)
    avg_momentum = momentum.rolling(window=5).mean()

    # Incorporate Volume Shock
    volume = data['volume']
    volume_diff = volume - volume.shift(2)
    
    # Adjust Momentum based on Volume Shock
    def adjust_momentum(avg_mom, vol_diff):
        if vol_diff > 0:
            return avg_mom * 1.1
        else:
            return avg_mom * 0.9
        
    adjusted_momentum = avg_momentum.combine(volume_diff, adjust_momentum)
    
    return adjusted_momentum
```
