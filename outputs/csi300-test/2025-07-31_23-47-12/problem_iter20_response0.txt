```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    
    # Define and Apply Dynamic Thresholds
    upper_threshold = cum_log_returns.quantile(0.75)
    lower_threshold = cum_log_returns.quantile(0.25)
    clipped_cum_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = (clipped_cum_log_returns * (volumes / mean_volume)).fillna(0)
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = absolute_price_changes.rolling(window=M).std()
    ema_abs_price_changes = absolute_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.3]  # Weights for momentum, standard deviation, and EMA
    combined_factor = (weights[0] * volume_adjusted_momentum +
                       weights[1] * std_abs_price_changes +
                       weights[2] * ema_abs_price_changes)
    
    return combined_factor
```
