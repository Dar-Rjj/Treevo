```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=60, interest_rate_data=None, gdp_data=None, inflation_data=None):
    # Calculate Log Returns
    df['log_return'] = np.log(df['close']).diff(n_days)
    
    # Compute Momentum
    df['cumulative_log_return'] = df['log_return'].rolling(window=n_days).sum()
    upper_threshold = 0.05
    lower_threshold = -0.05
    df['clipped_log_return'] = df['cumulative_log_return'].clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    mean_volume = df['volume'].mean()
    df['volume_adjusted_momentum'] = (df['volume'] / mean_volume) * df['clipped_log_return']
    
    # Determine Absolute Price Changes
    df['abs_price_change'] = df['close'].diff().abs()
    
    # Calculate Advanced Volatility Measures
    df['std_abs_price_change'] = df['abs_price_change'].rolling(window=m_days).std()
    df['ema_abs_price_change'] = df['abs_price_change'].ewm(span=m_days, adjust=False).mean()
    df['high_low_range'] = df['high'] - df['low']
    
    # Integrate Market Breadth Indicators
    df['advances'] = (df['close'] > df['open']).astype(int)
    df['declines'] = (df['close'] < df['open']).astype(int)
    df['advance_decline_line'] = (df['advances'] - df['declines']).cumsum()
    
    # Count Number of New Highs and Lows
    df['new_high'] = (df['high'] == df['high'].rolling(window=m_days).max()).astype(int)
    df['new_low'] = (df['low'] == df['low'].rolling(window=m_days).min()).astype(int)
    
    # Incorporate Key Economic Indicators
    if interest_rate_data is not None:
        df['change_interest_rate'] = interest_rate_data.diff()
    
    if gdp_data is not None:
        df['gdp_growth_rate'] = gdp_data.pct_change(4)  # Assuming quarterly data
    
    if inflation_data is not None:
        df['inflation_rate'] = inflation_data.pct_change(12)  # Year-over-year change
    
    # Final Factor Calculation
    weights = {
        'volume_adjusted_momentum': 0.3,
        'std_abs_price_change': 0.2,
        'ema_abs_price_change': 0.1,
        'high_low_range': 0.1,
        'advance_decline_line': 0.1,
        'new_high': 0.05,
        'new_low': 0.05,
        'change_interest_rate': 0.05,
        'gdp_growth_rate': 0.025,
        'inflation_rate': 0.025
    }
    
    factor = (
        df['volume_adjusted_momentum'] * weights['volume_adjusted_momentum'] +
        df['std_abs_price_change'] * weights['std_abs_price_change'] +
        df['ema_abs_price_change'] * weights['ema_abs_price_change'] +
        df['high_low_range'] * weights['high_low_range'] +
        df['advance_decline_line'] * weights['advance_decline_line'] +
        df['new_high'] * weights['new_high'] +
        df['new_low'] * weights['new_low']
    )
    
    if interest_rate_data is not None:
        factor += df['change_interest_rate'] * weights['change_interest_rate']
    
    if gdp_data is not None:
        factor += df['gdp_growth_rate'] * weights['gdp_growth_rate']
    
    if inflation_data is not None:
        factor += df['inflation_rate'] * weights['inflation_rate']
    
    return factor
```
