import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    cum_sum_log_returns = log_returns.cumsum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = 0.05
    lower_threshold = -0.05
    
    # Clip Log Returns to Thresholds
    momentum = np.clip(cum_sum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.mean()
    volume_adjusted_momentum = momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_volatility = absolute_price_changes.rolling(window=M).std()
    ema_volatility = absolute_price_changes.ewm(span=M, adjust=False).mean()
    iqr_volatility = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.2, 0.1]  # Weights for momentum, volume, standard deviation, and EMA
    factor = (weights[0] * volume_adjusted_momentum + 
              weights[1] * std_volatility + 
              weights[2] * ema_volatility + 
              weights[3] * iqr_volatility)
    
    # Ensure Weights Sum to 1
    assert np.isclose(sum(weights), 1.0)
    
    # Review and Validate Each Component
    # Integrate Market Breadth and Economic Indicators
    # This step is placeholder for further integration if needed
    
    return factor
