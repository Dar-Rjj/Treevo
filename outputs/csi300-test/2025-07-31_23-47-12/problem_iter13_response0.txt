```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20, sentiment_col='sentiment', sector_col='sector_return', interest_rate_col='interest_rate', gdp_growth_col='gdp_growth'):
    # Obtain Close Prices
    close = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close).diff().fillna(0)
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = 1.5 * log_returns.std()
    lower_threshold = -1.5 * log_returns.std()
    
    # Clip Log Returns to Thresholds
    clipped_cum_log_returns = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volume = df['volume']
    mean_volume = volume.rolling(window=M).mean()
    volume_adjusted_momentum = clipped_cum_log_returns * (volume / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Integrate Sentiment and Sector Performance
    sentiment_scores = df[sentiment_col]
    sector_returns = df[sector_col]
    sector_performance_index = sector_returns.rolling(window=M).mean()
    
    # Incorporate Economic Indicators
    interest_rates = df[interest_rate_col]
    gdp_growth_rate = df[gdp_growth_col]
    
    # Final Factor Calculation
    combined_factor = (
        0.4 * volume_adjusted_momentum + 
        0.3 * (std_abs_price_changes + ema_abs_price_changes + iqr_abs_price_changes) / 3 +
        0.2 * (sentiment_scores + sector_performance_index) / 2 +
        0.1 * (interest_rates + gdp_growth_rate) / 2
    )
    
    return combined_factor
```
