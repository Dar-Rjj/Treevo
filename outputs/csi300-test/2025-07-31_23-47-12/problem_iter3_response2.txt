```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain close prices
    close_prices = df['close']
    
    # Calculate log return over N days
    log_returns = np.log(close_prices).diff(N)
    
    # Retrieve volumes and calculate relative volume
    volumes = df['volume']
    avg_volume = volumes.rolling(window=N).mean()
    relative_volume = volumes / avg_volume
    
    # Adjust momentum for volume
    volume_adjusted_momentum = log_returns * relative_volume
    
    # Determine absolute price changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate standard deviation of price changes over M days
    price_change_volatility = abs_price_changes.rolling(window=M).std()
    
    # Final factor calculation: weighted sum of momentum and volatility component
    alpha_factor = 0.7 * volume_adjusted_momentum + 0.3 * price_change_volatility
    
    return alpha_factor
```
