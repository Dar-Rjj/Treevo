import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Returns over N Days
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = momentum.quantile(0.75)
    lower_threshold = momentum.quantile(0.25)
    
    # Clip Log Returns to Thresholds
    momentum = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_relative = volumes / volumes.rolling(window=N).mean()
    momentum_adjusted = momentum * volume_relative
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_volatility = absolute_price_changes.rolling(window=M).std()
    ema_volatility = absolute_price_changes.ewm(span=M, adjust=False).mean()
    iqr_volatility = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weights = [0.4, 0.3, 0.3]  # Weights for momentum, standard deviation, and IQR
    final_factor = (weights[0] * momentum_adjusted +
                    weights[1] * std_volatility +
                    weights[2] * iqr_volatility)
    
    return final_factor

# Example usage
df = pd.DataFrame({
    'date': pd.date_range(start='2023-01-01', periods=100),
    'open': np.random.rand(100),
    'high': np.random.rand(100),
    'low': np.random.rand(100),
    'close': np.random.rand(100),
    'amount': np.random.rand(100) * 1000,
    'volume': np.random.randint(100, 1000, 100)
})
df.set_index('date', inplace=True)

factor_values = heuristics_v2(df)
print(factor_values)
