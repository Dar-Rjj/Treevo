```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    
    # Define and apply dynamic thresholds
    upper_threshold = momentum.quantile(0.95)
    lower_threshold = momentum.quantile(0.05)
    momentum = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_adjusted_momentum = momentum * (volumes / volumes.rolling(window=N).mean())
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = absolute_price_changes.rolling(window=M).std()
    ema_abs_changes = absolute_price_changes.ewm(span=M).mean()
    
    # Inter-Quartile Range (IQR)
    iqr = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Incorporate High-Low Price Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Average True Range (ATR)
    tr1 = abs(df['high'] - df['low'])
    tr2 = abs(df['high'] - df['close'].shift())
    tr3 = abs(df['low'] - df['close'].shift())
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    atr = true_range.rolling(window=M).mean()
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.2, 0.1]
    factor = (
        weights[0] * volume_adjusted_momentum +
        weights[1] * std_dev +
        weights[2] * ema_abs_changes +
        weights[3] * iqr
    )
    
    # Ensure Weights Sum to 1
    factor = factor / sum(weights)
    
    return factor
```
