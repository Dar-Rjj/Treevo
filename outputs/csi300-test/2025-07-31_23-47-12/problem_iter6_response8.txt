```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=10):
    # Obtain close prices and calculate log returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff()
    
    # Compute momentum (log return over N days)
    momentum = log_returns.rolling(window=n_days).sum()
    
    # Retrieve volumes and compute moving average volume over N days
    volumes = df['volume']
    avg_volume = volumes.rolling(window=n_days).mean()
    volume_ratio = volumes / avg_volume
    
    # Adjust momentum for volume
    volume_adjusted_momentum = momentum * volume_ratio
    
    # Determine absolute price changes and calculate standard deviation over M days
    abs_price_changes = close_prices.diff().abs()
    price_change_volatility = abs_price_changes.rolling(window=m_days).std()
    
    # Define dynamic weights for momentum and volatility
    # Example: Simple equal weighting
    weight_momentum = 0.5
    weight_volatility = 0.5
    
    # Ensure weights are positive and sum to 1
    assert weight_momentum + weight_volatility == 1.0
    assert weight_momentum >= 0 and weight_volatility >= 0
    
    # Combine weighted momentum and volatility component
    final_factor = (weight_momentum * volume_adjusted_momentum) - (weight_volatility * price_change_volatility)
    
    return final_factor
```
