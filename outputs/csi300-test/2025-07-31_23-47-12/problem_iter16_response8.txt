```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    
    # Compute Momentum
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    upper_threshold = 0.05
    lower_threshold = -0.05
    clipped_cumulative_log_returns = np.clip(cumulative_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_ratio = volumes / mean_volume
    volume_adjusted_momentum = clipped_cumulative_log_returns * volume_ratio
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = absolute_price_changes.rolling(window=M).std()
    ema = absolute_price_changes.ewm(span=M, min_periods=M).mean()
    iqr = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    mad = absolute_price_changes.rolling(window=M).apply(lambda x: np.median(np.abs(x - np.median(x))), raw=True)
    kurtosis = absolute_price_changes.rolling(window=M).kurt()
    
    # Final Factor Calculation
    weights = [0.3, 0.3, 0.1, 0.1, 0.1, 0.1]  # Weights for momentum, volume-adjusted momentum, and volatility measures
    factor_values = (weights[0] * cumulative_log_returns +
                     weights[1] * volume_adjusted_momentum +
                     weights[2] * std_dev +
                     weights[3] * ema +
                     weights[4] * iqr +
                     weights[5] * mad +
                     kurtosis)
    
    # Ensure Weights Sum to 1
    factor_values = factor_values / sum(weights)
    
    # Integrate Market Breadth and Economic Indicators
    advance_decline_line = (df['close'].diff() > 0).astype(int).cumsum()
    new_highs = (df['close'] == df['high']).astype(int).rolling(window=N).sum()
    new_lows = (df['close'] == df['low']).astype(int).rolling(window=N).sum()
    
    # Assume interest rates and GDP growth rate are available in the DataFrame
    interest_rates = df.get('interest_rates', 0)
    gdp_growth_rate = df.get('gdp_growth_rate', 0)
    
    # Combine all components
    final_factor = (factor_values + 
                    0.1 * advance_decline_line + 
                    0.1 * new_highs - 
                    0.1 * new_lows + 
                    0.1 * interest_rates + 
                    0.1 * gdp_growth_rate)
    
    return final_factor
```
