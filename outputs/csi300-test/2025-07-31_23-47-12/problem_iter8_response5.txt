import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, p_days=60, m_days=30, w_momentum=0.5, w_volume=0.3, w_volatility=0.2):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff(n_days)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    upper_threshold = 0.05
    lower_threshold = -0.05
    clipped_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=p_days).mean()
    volume_ratio = volumes / mean_volume
    momentum = clipped_log_returns * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=m_days).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=m_days, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=m_days).quantile(0.75) - abs_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Final Factor Calculation
    factor = (w_momentum * momentum + 
              w_volume * volume_ratio + 
              w_volatility * (std_abs_price_changes + ema_abs_price_changes + iqr_abs_price_changes) / 3)
    
    return factor
