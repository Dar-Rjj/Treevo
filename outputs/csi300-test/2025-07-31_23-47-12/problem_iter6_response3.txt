```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, P=20, M=20, momentum_weight=0.7, volatility_weight=0.3):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(N)
    
    # Calculate Momentum
    momentum = log_returns
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=P).mean()
    volume_ratio = volumes / mean_volume
    volume_adjusted_momentum = momentum * volume_ratio
    
    # Incorporate Price Change Volatility
    abs_price_changes = close_prices.diff().abs()
    price_change_volatility = abs_price_changes.rolling(window=M).std()
    
    # Scale Volatility Component
    scaled_volatility = price_change_volatility * volatility_weight
    
    # Final Factor Calculation
    factor_values = (volume_adjusted_momentum * momentum_weight) + scaled_volatility
    
    return factor_values
```
