```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=20, momentum_weight=0.7, volatility_weight=0.3):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Momentum
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    momentum = log_returns.rolling(window=n_days).sum()
    
    # Adjust for Volume
    volumes = df['volume']
    volume_ratio = volumes / volumes.rolling(window=n_days).mean()
    adjusted_momentum = momentum * volume_ratio
    
    # Incorporate Price Change Volatility
    price_changes = close_prices - close_prices.shift(1)
    abs_price_changes = np.abs(price_changes)
    volatility = abs_price_changes.rolling(window=m_days).std()
    
    # Weighting Factors
    assert momentum_weight + volatility_weight == 1, "Weights must sum to 1"
    weighted_momentum = adjusted_momentum * momentum_weight
    weighted_volatility = volatility * volatility_weight
    
    # Final Factor Calculation
    factor_values = weighted_momentum + weighted_volatility
    
    return factor_values
```
