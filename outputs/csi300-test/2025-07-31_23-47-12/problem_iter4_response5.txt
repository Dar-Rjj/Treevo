```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, L=10, M=5, W1=0.7, W2=0.3):
    # Obtain close prices and volumes
    close_prices = df['close']
    volumes = df['volume']
    
    # Calculate logarithmic momentum over a dynamic lookback period N
    log_returns = np.log(close_prices).diff(N)
    
    # Calculate relative volume by dividing the current volume by its moving average over a dynamic lookback period L
    avg_volume = volumes.rolling(window=L).mean()
    rel_volume = volumes / avg_volume
    
    # Adjust logarithmic momentum by relative volume
    vol_adjusted_mom = log_returns * rel_volume
    
    # Determine the absolute price changes and calculate their standard deviation over a dynamic lookback period M to get price change volatility
    abs_price_changes = np.abs(close_prices.diff())
    price_volatility = abs_price_changes.rolling(window=M).std()
    
    # Final factor calculation: a weighted sum of volatility-adjusted momentum and price change volatility
    final_factor = (W1 * vol_adjusted_mom) + (W2 * price_volatility)
    
    # Ensure non-negative results
    final_factor = final_factor.clip(lower=0)
    
    return final_factor
```
