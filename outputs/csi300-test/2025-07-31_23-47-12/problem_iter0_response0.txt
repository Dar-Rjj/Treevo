```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a volume-weighted close-to-open return.
    
    :param df: A pandas DataFrame with columns (open, high, low, close, amount, volume)
               and index (date).
    :return: A pandas Series representing the factor values.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
