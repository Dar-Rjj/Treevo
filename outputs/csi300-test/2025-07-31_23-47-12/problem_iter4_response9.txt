```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, m_days=20, momentum_weight=0.7, volatility_weight=0.3):
    """
    Compute a novel alpha factor: Volume Adjusted Momentum with Price Change Volatility.
    
    Parameters:
    - df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                         and index as dates.
    - n_days (int): Number of days to compute the momentum over.
    - m_days (int): Number of days to compute the volatility over.
    - momentum_weight (float): Weight assigned to the momentum component in the final factor.
    - volatility_weight (float): Weight assigned to the volatility component in the final factor.
    
    Returns:
    - pd.Series: A Series with the computed factor values indexed by date.
    """
    
    # Calculate Momentum
    log_returns = np.log(df['close']).diff()
    momentum = log_returns.rolling(window=n_days).sum()
    
    # Adjust for Volume
    volume_ratio = df['volume'] / df['volume'].rolling(window=n_days).mean()
    adjusted_momentum = momentum * volume_ratio
    
    # Incorporate Price Change Volatility
    price_changes = df['close'].diff().abs()
    volatility = price_changes.rolling(window=m_days).std()
    
    # Final Factor Calculation
    assert momentum_weight + volatility_weight == 1, "Momentum and volatility weights must sum to 1."
    combined_factor = (momentum_weight * adjusted_momentum) + (volatility_weight * volatility)
    
    return combined_factor
```
