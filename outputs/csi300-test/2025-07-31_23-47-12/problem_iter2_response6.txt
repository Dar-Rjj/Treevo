import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    price_momentum = df['close'] - df['close'].shift(1)
    
    # Calculate Volume Momentum
    volume_momentum = df['volume'] - df['volume'].shift(1)
    
    # Integrate Price and Volume Momentum
    integrated_price_momentum = price_momentum * df['close']
    integrated_volume_momentum = volume_momentum * df['volume']
    
    # Combine Integrated Price and Volume Momentum
    combined_momentum = integrated_price_momentum + integrated_volume_momentum
    
    # Calculate Volatility
    volatility = (df['high'] - df['low']) / df['close']
    
    # Combine Momentum and Volatility
    combined_momentum_volatility = combined_momentum * volatility
    
    # Calculate Liquidity Measure
    average_price = df['amount'] / df['volume']
    
    # Refine Intersection Score
    refined_score = combined_momentum_volatility * average_price - combined_momentum_volatility.min()
    
    # Output Final Alpha Factor
    final_alpha_factor = refined_score.clip(lower=0)
    
    return final_alpha_factor
