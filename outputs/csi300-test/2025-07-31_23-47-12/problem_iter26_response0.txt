import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, K=10, alpha=0.6, beta=0.3, gamma=0.1):
    """
    Generate a novel and interpretable alpha factor using Volume Adjusted Momentum with Price Change Volatility.
    
    Parameters:
    - df: pandas DataFrame with columns (open, high, low, close, amount, volume) and index (date).
    - N: Number of days to compute log returns.
    - K: Number of days to compute exponential moving average for volatility.
    - alpha, beta, gamma: Weights for momentum, volume, and volatility components respectively.
    
    Returns:
    - alpha_factor: pandas Series indexed by (date) representing the factor values.
    """
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N).dropna()
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    upper_threshold = cum_log_returns.quantile(0.75)
    lower_threshold = cum_log_returns.quantile(0.25)
    clipped_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.mean()
    volume_adjusted_momentum = clipped_log_returns * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    ema_volatility = abs_price_changes.ewm(span=K).mean()
    
    # Final Factor Calculation
    alpha_factor = (alpha * volume_adjusted_momentum + 
                    beta * ema_volatility + 
                    gamma * 0.0)  # Placeholder for key economic indicators (e.g., interest rates, GDP growth rate)
    return alpha_factor.dropna()
