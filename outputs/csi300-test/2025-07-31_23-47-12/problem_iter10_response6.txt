import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(n_days)
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Calculate Moving Average Volume over N Days
    moving_avg_volume = volumes.rolling(window=n_days).mean()
    
    # Calculate Volume Ratio (Current Volume / Moving Average Volume)
    volume_ratio = volumes / moving_avg_volume
    
    # Adjust Momentum for Volume
    momentum = log_returns * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = np.abs(close_prices.diff())
    
    # Calculate Exponential Moving Average of Absolute Differences over M Days
    ema_volatility = abs_price_changes.ewm(span=m_days, adjust=False).mean()
    
    # Assign Dynamic Weights to Momentum and Volatility
    weight_momentum = 0.6
    weight_volatility = 0.4
    
    # Ensure Weights are Positive and Sum to 1
    assert weight_momentum + weight_volatility == 1.0, "Weights must sum to 1"
    
    # Combine Weighted Momentum and Volatility Component
    factor_values = weight_momentum * momentum + weight_volatility * ema_volatility
    
    # Ensure Interpretability and Novelty
    return pd.Series(factor_values, index=df.index, name='Volume_Adjusted_Momentum_with_Price_Change_Volatility')
