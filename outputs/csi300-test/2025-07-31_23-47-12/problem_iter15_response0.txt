```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Momentum
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    upper_threshold = cumulative_log_returns.quantile(0.95)
    lower_threshold = cumulative_log_returns.quantile(0.05)
    clipped_cumulative_returns = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = clipped_cumulative_returns * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = absolute_price_changes.rolling(window=M).std()
    ema_abs_price_changes = absolute_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Weighted Components
    weights = [0.4, 0.3, 0.3]  # Weights for momentum, volume, and volatility
    combined_factor = (weights[0] * volume_adjusted_momentum + 
                       weights[1] * std_abs_price_changes + 
                       weights[2] * (ema_abs_price_changes + iqr_abs_price_changes) / 2)
    
    # Ensure Weights Sum to 1
    assert np.isclose(np.sum(weights), 1.0)
    
    # Return the final factor values
    return combined_factor
```
