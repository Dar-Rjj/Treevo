```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Cumulative Sum of Log Returns
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = 0.05
    lower_threshold = -0.05
    
    # Clip Log Returns to Thresholds
    dynamic_momentum = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Calculate Volume Relative to Mean
    volume_relative_to_mean = volumes / volumes.rolling(window=N).mean()
    
    # Incorporate Volume into Momentum
    adjusted_momentum = dynamic_momentum * volume_relative_to_mean
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Compute Standard Deviation of Absolute Price Changes Over M Days
    std_dev_abs_price_changes = absolute_price_changes.rolling(window=M).std()
    
    # Calculate Exponential Moving Average (EMA) of Absolute Price Changes
    ema_abs_price_changes = absolute_price_changes.ewm(span=M, adjust=False).mean()
    
    # Calculate Inter-Quartile Range (IQR) of Absolute Price Changes
    iqr_abs_price_changes = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weights = [0.4, 0.3, 0.3]  # Weights for momentum, volume, and volatility components
    final_factor = (weights[0] * adjusted_momentum + 
                    weights[1] * volume_relative_to_mean + 
                    weights[2] * (std_dev_abs_price_changes + ema_abs_price_changes + iqr_abs_price_changes) / 3)
    
    return final_factor
```
