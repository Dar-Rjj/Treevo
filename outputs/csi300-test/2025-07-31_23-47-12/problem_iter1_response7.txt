import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=10, m_days=30):
    """
    Generate a volume-adjusted momentum alpha factor with price change volatility.

    Parameters:
    df (pd.DataFrame): DataFrame containing market features (open, high, low, close, amount, volume)
                       with date as the index.
    n_days (int): Number of days for momentum calculation.
    m_days (int): Number of days for volatility calculation.

    Returns:
    pd.Series: Series of factor values indexed by date.
    """

    # Obtain Close Prices
    close_prices = df['close']

    # Calculate Momentum
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    momentum = log_returns.rolling(window=n_days).sum()

    # Adjust for Volume
    volumes = df['volume']
    normalized_volume = (volumes / volumes.mean())
    volume_adjusted_momentum = momentum * normalized_volume

    # Incorporate Price Change Volatility
    price_changes = close_prices.diff()
    absolute_price_changes = price_changes.abs()
    volatility = absolute_price_changes.rolling(window=m_days).std()

    # Final Factor Calculation
    factor = 0.7 * volume_adjusted_momentum + 0.3 * volatility
    return factor
