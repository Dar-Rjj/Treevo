```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=30, volume_weight=0.5, momentum_weight=0.3, volatility_weight=0.2):
    # Ensure the weights sum to 1
    assert (volume_weight + momentum_weight + volatility_weight) == 1.0, "Weights must sum to 1.0"
    
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    upper_threshold = cum_log_returns.quantile(0.75)
    lower_threshold = cum_log_returns.quantile(0.25)
    clipped_cum_log_returns = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_relative_to_mean = volumes / volumes.rolling(window=M).mean()
    momentum = clipped_cum_log_returns * volume_relative_to_mean
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    volatility = (std_abs_price_changes + ema_abs_price_changes + iqr_abs_price_changes) / 3
    
    # Final Factor Calculation
    factor = (momentum * momentum_weight + 
              volume_relative_to_mean * volume_weight + 
              volatility * volatility_weight)
    
    return factor
```
