```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days_momentum=20, m_days_volatility=20, momentum_threshold=3, volume_weight=0.5, volatility_weight=0.3, momentum_weight=0.2):
    # Ensure the sum of weights is 1
    assert np.isclose(volume_weight + volatility_weight + momentum_weight, 1), "Weights must sum to 1"
    
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    cumulative_log_returns = log_returns.rolling(window=n_days_momentum).sum()
    upper_threshold = momentum_threshold * cumulative_log_returns.std()
    lower_threshold = -momentum_threshold * cumulative_log_returns.std()
    clipped_cumulative_log_returns = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=n_days_momentum).mean()
    relative_volume = volumes / mean_volume
    volume_adjusted_momentum = clipped_cumulative_log_returns * relative_volume
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_abs_changes = absolute_price_changes.rolling(window=m_days_volatility).std()
    ema_abs_changes = absolute_price_changes.ewm(span=m_days_volatility, adjust=False).mean()
    iqr_abs_changes = absolute_price_changes.rolling(window=m_days_volatility).quantile(0.75) - absolute_price_changes.rolling(window=m_days_volatility).quantile(0.25)
    combined_volatility = (std_dev_abs_changes + ema_abs_changes + iqr_abs_changes) / 3
    
    # Final Factor Calculation
    factor_values = (volume_weight * volume_adjusted_momentum +
                     volatility_weight * combined_volatility +
                     momentum_weight * clipped_cumulative_log_returns)
    
    return factor_values
```
