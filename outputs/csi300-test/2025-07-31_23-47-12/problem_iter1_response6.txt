```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread Ratio
    high_low_spread_ratio = (df['high'] - df['low']) / df['close']
    
    # Calculate Open-Close Trend
    open_close_trend = (df['close'] - df['open']) / df['open']
    
    # Weight by Volume and Price Movement
    weighted_volume_price_movement = df['volume'] * high_low_spread_ratio
    
    # Combine Indicators
    factor = weighted_volume_price_movement * open_close_trend
    
    return factor
```
