```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, alpha=0.7, beta=0.3):
    # Ensure the sum of weights is 1 and they are positive
    assert 0 < alpha < 1 and 0 < beta < 1 and alpha + beta == 1, "Alpha and Beta must be positive and sum to 1"
    
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(N)
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Calculate Moving Average Volume over N Days
    moving_avg_volume = volumes.rolling(window=N).mean()
    
    # Calculate Volume Ratio (Current Volume / Moving Average Volume)
    volume_ratio = volumes / moving_avg_volume
    
    # Multiply by Momentum
    momentum = log_returns * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Final Factor Calculation
    final_factor = alpha * momentum + beta * abs_price_changes
    
    return final_factor
```
