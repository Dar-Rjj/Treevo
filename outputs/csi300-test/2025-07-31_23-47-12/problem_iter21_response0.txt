```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Momentum
    cum_log_returns = log_returns.rolling(window=N).sum()
    
    # Define dynamic thresholds for clipping
    upper_threshold = cum_log_returns.quantile(0.95)
    lower_threshold = cum_log_returns.quantile(0.05)
    clipped_cum_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_relative = volumes / volumes.rolling(window=N).mean()
    volume_adjusted_momentum = clipped_cum_log_returns * volume_relative
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine weighted components
    weight_momentum = 0.4
    weight_vol_std = 0.3
    weight_vol_ema = 0.2
    weight_vol_iqr = 0.1
    
    total_weight = weight_momentum + weight_vol_std + weight_vol_ema + weight_vol_iqr
    assert np.isclose(total_weight, 1.0), "Weights do not sum to 1"
    
    factor = (weight_momentum * volume_adjusted_momentum +
              weight_vol_std * std_abs_price_changes +
              weight_vol_ema * ema_abs_price_changes +
              weight_vol_iqr * iqr_abs_price_changes)
    
    return factor
```
