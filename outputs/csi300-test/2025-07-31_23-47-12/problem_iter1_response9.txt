import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Daily Return
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    
    # Measure Intraday Volatility
    df['prev_close'] = df['close'].shift(1)
    df['intraday_volatility'] = df['intraday_range'] / df['prev_close']
    
    # Adjust Daily Return for Intraday Volatility
    df['volatility_adjusted_return'] = df['daily_return'] / df['intraday_volatility']
    
    return df['volatility_adjusted_return']
