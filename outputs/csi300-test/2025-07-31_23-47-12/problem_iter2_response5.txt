```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Momentum
    df['High_Low_Momentum'] = df['high'] - df['low']
    df['Smoothed_High_Low_Momentum'] = df['High_Low_Momentum'].rolling(window=5).mean()
    
    # Calculate Daily Volatility
    df['Daily_Volatility'] = df['close'].rolling(window=5).std()
    
    # Enhanced Volume Surge Indicator
    df['Volume_Change'] = df['volume'].pct_change()
    df['Volume_Surge'] = df['Volume_Change'] > 0.1
    
    # Combine High-Low Momentum, Daily Volatility, and Volume Surge
    df['Factor'] = 0
    df.loc[df['Volume_Surge'], 'Factor'] = (df['Smoothed_High_Low_Momentum'] / df['Daily_Volatility']).fillna(0)
    
    return df['Factor']
```
