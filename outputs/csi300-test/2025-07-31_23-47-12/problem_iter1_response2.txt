```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Momentum
    high_low_diff = df['high'] - df['low']
    high_low_momentum = high_low_diff.rolling(window=5).mean()

    # Volume Surge Indicator
    volume_change = df['volume'].pct_change()
    volume_surge = (volume_change > 0.10)

    # Combine High-Low Momentum and Volume Surge
    factor = high_low_momentum * volume_surge
    factor = factor.fillna(0)  # Set NaN values to 0, in case of no surge

    return factor
```
