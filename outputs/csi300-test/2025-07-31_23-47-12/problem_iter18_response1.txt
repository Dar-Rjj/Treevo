import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=30, k_days=10, l_days=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Cumulative Sum of Log Returns
    cum_log_returns = log_returns.rolling(window=n_days).sum()
    
    # Define Upper and Lower Thresholds
    rolling_mean = log_returns.rolling(window=n_days).mean()
    rolling_std = log_returns.rolling(window=n_days).std()
    upper_threshold = rolling_mean + 2 * rolling_std
    lower_threshold = rolling_mean - 2 * rolling_std
    
    # Clip Log Returns to Thresholds
    clipped_log_returns = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_mean = volumes.rolling(window=n_days).mean()
    volume_adjusted_momentum = clipped_log_returns * (volumes / volume_mean)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = abs_price_changes.rolling(window=m_days).std()
    ema = abs_price_changes.ewm(span=k_days, adjust=False).mean()
    iqr = abs_price_changes.rolling(window=l_days).quantile(0.75) - abs_price_changes.rolling(window=l_days).quantile(0.25)
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.3]  # Weights for Momentum, EMA, and IQR
    combined_factor = (weights[0] * volume_adjusted_momentum +
                       weights[1] * ema +
                       weights[2] * iqr)
    
    return combined_factor.dropna()
