import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, k_days=10, interest_rate=0.05, gdp_growth_rate=0.02):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff(n_days)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    upper_threshold = cum_log_returns.quantile(0.75)
    lower_threshold = cum_log_returns.quantile(0.25)
    clipped_cum_log_returns = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_relative = volumes / volumes.rolling(window=n_days).mean()
    momentum_adjusted = clipped_cum_log_returns * volume_relative
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    ema_abs_price_changes = abs_price_changes.ewm(span=k_days, adjust=False).mean()
    
    # Final Factor Calculation
    weights = np.array([0.5, 0.3, 0.2])  # Weights for momentum, volume, and volatility
    weighted_momentum = momentum_adjusted * weights[0]
    weighted_volatility = ema_abs_price_changes * weights[1]
    economic_indicator = (interest_rate + gdp_growth_rate) * weights[2]
    
    alpha_factor = weighted_momentum + weighted_volatility + economic_indicator
    return alpha_factor
