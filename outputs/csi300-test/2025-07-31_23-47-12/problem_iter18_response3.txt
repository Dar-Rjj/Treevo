```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5, momentum_weight=0.5, volume_weight=0.3, volatility_weight=0.2):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    cum_log_returns = log_returns.cumsum()
    
    # Define and Apply Dynamic Thresholds
    upper_threshold = cum_log_returns.quantile(0.9)
    lower_threshold = cum_log_returns.quantile(0.1)
    momentum = np.clip(cum_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.mean()
    volume_adjustment = (volumes / mean_volume) * momentum
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=M*4).quantile(0.75) - abs_price_changes.rolling(window=M*4).quantile(0.25)
    
    # Combine Weighted Components
    weighted_momentum = momentum * momentum_weight
    weighted_volume = volume_adjustment * volume_weight
    weighted_volatility = (std_dev_abs_price_changes + ema_abs_price_changes + iqr_abs_price_changes) / 3 * volatility_weight
    
    final_factor = weighted_momentum + weighted_volume + weighted_volatility
    
    return final_factor
```
