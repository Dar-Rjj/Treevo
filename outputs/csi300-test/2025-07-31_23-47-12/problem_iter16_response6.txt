```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=20, momentum_threshold=0.05, volume_weight=0.5, ema_alpha=0.1, iqr_weight=0.3):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(n_days)
    
    # Compute Momentum
    cumulative_log_returns = log_returns.rolling(window=n_days).sum()
    upper_threshold = cumulative_log_returns.quantile(1 - momentum_threshold)
    lower_threshold = cumulative_log_returns.quantile(momentum_threshold)
    clipped_momentum = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=n_days).mean()
    volume_adjusted_momentum = clipped_momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    ema_volatility = abs_price_changes.ewm(alpha=ema_alpha, min_periods=m_days).mean()
    iqr_volatility = abs_price_changes.rolling(window=m_days).quantile(0.75) - abs_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Final Factor Calculation
    weighted_momentum = clipped_momentum
    weighted_volume_adjusted_momentum = volume_adjusted_momentum * volume_weight
    weighted_ema_volatility = ema_volatility * (1 - iqr_weight)
    weighted_iqr_volatility = iqr_volatility * iqr_weight
    
    final_factor = weighted_momentum + weighted_volume_adjusted_momentum - weighted_ema_volatility - weighted_iqr_volatility
    
    return final_factor
```
