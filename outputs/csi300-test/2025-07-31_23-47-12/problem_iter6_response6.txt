```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain close prices
    close_prices = df['close']
    
    # Compute log returns over N days
    log_returns = np.log(close_prices).diff(N)
    
    # Calculate cumulative sum of log returns
    cum_log_returns = log_returns.cumsum()
    
    # Retrieve volumes and calculate volume relative to mean
    volumes = df['volume']
    volume_mean = volumes.rolling(window=N).mean()
    volume_ratio = volumes / volume_mean
    
    # Incorporate volume into momentum
    volume_adjusted_momentum = cum_log_returns * volume_ratio
    
    # Determine absolute price changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate standard deviation of absolute price changes over M days
    price_change_volatility = abs_price_changes.rolling(window=M).std()
    
    # Final factor calculation with equal weights for simplicity
    factor = 0.5 * volume_adjusted_momentum + 0.5 * price_change_volatility
    
    return factor
```
