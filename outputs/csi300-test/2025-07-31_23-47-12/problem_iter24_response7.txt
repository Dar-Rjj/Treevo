```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, vol_window=20, momentum_thresholds=(-0.1, 0.1), interest_rate=0.05, gdp_growth=0.02):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    momentum = log_returns.rolling(window=n_days).sum()
    momentum = momentum.clip(lower=momentum_thresholds[0], upper=momentum_thresholds[1])
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=vol_window).mean()
    volume_adjusted_momentum = momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    ema_volatility = abs_price_changes.ewm(span=vol_window, adjust=False).mean()
    
    # Final Factor Calculation
    weights = [0.4, 0.3, 0.3]  # Weights for momentum, volume, and volatility
    combined_factor = (weights[0] * volume_adjusted_momentum + 
                       weights[1] * ema_volatility + 
                       weights[2] * (interest_rate + gdp_growth))
    
    return combined_factor
```
