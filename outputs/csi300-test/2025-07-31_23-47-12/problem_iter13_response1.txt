import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=30):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff().fillna(0)
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    
    # Apply Dynamic Thresholds
    upper_threshold = 0.05
    lower_threshold = -0.05
    momentum = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    mean_volume = df['volume'].mean()
    volume_adjusted_momentum = momentum * (df['volume'] / mean_volume)
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = absolute_price_changes.rolling(window=M).std()
    ema_abs_price_changes = absolute_price_changes.ewm(span=M).mean()
    iqr_abs_price_changes = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Final Factor Calculation
    weights = np.array([0.4, 0.3, 0.3])  # Weights for momentum, volume, and volatility
    factor = (weights[0] * volume_adjusted_momentum +
              weights[1] * std_abs_price_changes +
              weights[2] * ema_abs_price_changes +
              iqr_abs_price_changes)
    
    return pd.Series(factor, index=df.index)
