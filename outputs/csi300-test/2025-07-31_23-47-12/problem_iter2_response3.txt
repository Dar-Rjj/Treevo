```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Logarithmic Returns
    log_returns = np.log(close_prices).diff(n_days)
    
    # Retrieve Volumes and Average Volume over N Days
    volumes = df['volume']
    avg_volume = volumes.rolling(window=n_days).mean()
    
    # Incorporate Volume into Momentum
    volume_adjusted_momentum = log_returns * avg_volume
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Standard Deviation over M Days
    price_change_volatility = abs_price_changes.rolling(window=m_days).std()
    
    # Final Factor Calculation
    final_factor = (volume_adjusted_momentum + price_change_volatility) / 2
    return final_factor
```
