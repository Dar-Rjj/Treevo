```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N1=5, N2=20, N3=60, P=10, M=30, volume_threshold=0.1):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over Multiple Periods (N1, N2, N3)
    log_returns_N1 = np.log(close_prices / close_prices.shift(N1))
    log_returns_N2 = np.log(close_prices / close_prices.shift(N2))
    log_returns_N3 = np.log(close_prices / close_prices.shift(N3))
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Compute Volume Change over P Days
    volume_change = volumes.pct_change(P)
    
    # Determine if Volume Change Exceeds Threshold
    volume_factor = (volume_change > volume_threshold).astype(float)
    
    # Multiply by Momentum if Volume Change Exceeds Threshold, else set to 0
    momentum_N1 = log_returns_N1 * volume_factor
    momentum_N2 = log_returns_N2 * volume_factor
    momentum_N3 = log_returns_N3 * volume_factor
    
    # Sum Weighted Momentum
    total_momentum = momentum_N1 + momentum_N2 + momentum_N3
    
    # Determine Absolute Price Changes
    absolute_changes = close_prices.diff().abs()
    
    # Calculate Standard Deviation over M Days
    volatility = absolute_changes.rolling(window=M).std()
    
    # Final Factor Calculation
    factor = total_momentum + volatility
    
    return factor
```
