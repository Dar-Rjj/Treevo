import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff()
    
    # Calculate Cumulative Sum of Log Returns
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = 0.05
    lower_threshold = -0.05
    
    # Clip Log Returns to Thresholds
    clipped_cumulative_log_returns = np.clip(cumulative_log_returns, lower_threshold, upper_threshold)
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Calculate Volume Relative to Mean
    volume_relative_to_mean = volumes / volumes.rolling(window=N).mean()
    
    # Multiply by Momentum
    momentum_adjusted = clipped_cumulative_log_returns * volume_relative_to_mean
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Compute Standard Deviation of Absolute Price Changes
    std_dev_price_changes = absolute_price_changes.rolling(window=M).std()
    
    # Calculate Exponential Moving Average (EMA) of Absolute Price Changes
    ema_price_changes = absolute_price_changes.ewm(span=M, adjust=False).mean()
    
    # Calculate Inter-Quartile Range (IQR) of Absolute Price Changes
    iqr_price_changes = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weight_momentum = 0.4
    weight_volatility = 0.3
    weight_ema = 0.2
    weight_iqr = 0.1
    
    final_factor = (
        weight_momentum * momentum_adjusted +
        weight_volatility * std_dev_price_changes +
        weight_ema * ema_price_changes +
        weight_iqr * iqr_price_changes
    )
    
    return final_factor
