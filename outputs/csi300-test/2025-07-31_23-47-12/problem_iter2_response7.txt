```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Change Momentum
    df['Close_10_days_ago'] = df['close'].shift(10)
    df['Momentum_Ratio'] = (df['close'] - df['Close_10_days_ago']) / df['Close_10_days_ago']

    # Incorporate Recent Price Changes
    df['Close_5_days_ago'] = df['close'].shift(5)
    df['Short_Term_Momentum_Ratio'] = (df['close'] - df['Close_5_days_ago']) / df['Close_5_days_ago']
    
    # Calculate Volatility
    df['High_Low_Range'] = df['high'] - df['low']
    df['Avg_High_Low_Range'] = df['High_Low_Range'].rolling(window=10).mean()

    # Incorporate Volume Shock
    df['Volume_2_days_ago'] = df['volume'].shift(2)
    df['Volume_Change'] = abs(df['volume'] - df['Volume_2_days_ago'])

    # Adjust for Significant Volume Shifts
    df['Adjusted_Short_Term_Momentum'] = df['Short_Term_Momentum_Ratio'].copy()
    df.loc[df['volume'] > df['Volume_2_days_ago'], 'Adjusted_Short_Term_Momentum'] *= 1.1
    df.loc[df['volume'] < df['Volume_2_days_ago'], 'Adjusted_Short_Term_Momentum'] *= 0.9

    # Combine Factors
    df['Combined_Factor'] = (df['Momentum_Ratio'] + df['Adjusted_Short_Term_Momentum']) * df['Avg_High_Low_Range']

    return df['Combined_Factor'].dropna()
```
