import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    momentum = log_returns.rolling(window=n_days).sum()
    
    # Apply Dynamic Thresholds
    upper_threshold = momentum.quantile(0.75)
    lower_threshold = momentum.quantile(0.25)
    momentum = np.clip(momentum, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_ratio = volumes / volumes.mean()
    adjusted_momentum = momentum * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_volatility = abs_price_changes.rolling(window=m_days).std()
    ema_volatility = abs_price_changes.ewm(span=m_days).mean()
    iqr_volatility = abs_price_changes.rolling(window=m_days).quantile(0.75) - abs_price_changes.rolling(window=m_days).quantile(0.25)
    
    # Final Factor Calculation
    weights = np.array([0.4, 0.3, 0.3])  # Weights for adjusted_momentum, std_volatility, and ema_volatility
    final_factor = (weights[0] * adjusted_momentum + 
                    weights[1] * std_volatility + 
                    weights[2] * ema_volatility)
    
    return final_factor
