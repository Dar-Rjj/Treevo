```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=20, m_days=10):
    # Obtain close prices
    close_prices = df['close']
    
    # Compute log return over N days
    log_returns = np.log(close_prices).diff(n_days)
    
    # Retrieve volumes and calculate volume relative to mean
    volumes = df['volume']
    avg_volume = volumes.rolling(window=n_days).mean()
    volume_ratio = volumes / avg_volume
    
    # Adjust momentum for volume
    momentum_adjusted = log_returns * volume_ratio
    
    # Determine absolute price changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate standard deviation of absolute price changes over M days
    price_volatility = abs_price_changes.rolling(window=m_days).std()
    
    # Final factor calculation: sum of weighted momentum and volatility component
    alpha_factor = momentum_adjusted - price_volatility  # Subtracting here to ensure the factor is not overly complex
    
    # Ensure interpretability and novelty by capping extreme values
    alpha_factor = np.clip(alpha_factor, -3, 3)  # Capping at 3 standard deviations
    
    return alpha_factor.dropna()
```
