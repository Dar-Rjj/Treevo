```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(N)
    
    # Calculate Cumulative Sum of Log Returns
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = cumulative_log_returns.quantile(0.95)
    lower_threshold = cumulative_log_returns.quantile(0.05)
    
    # Clip Log Returns to Thresholds
    momentum = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Calculate Volume Relative to Mean
    volume_relative = volumes / volumes.rolling(window=N).mean()
    
    # Incorporate Volume into Momentum
    adjusted_momentum = momentum * volume_relative
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Compute Standard Deviation of Absolute Price Changes Over M Days
    std_dev_volatility = absolute_price_changes.rolling(window=M).std()
    
    # Calculate Exponential Moving Average (EMA) of Absolute Price Changes
    ema_volatility = absolute_price_changes.ewm(span=M, adjust=False).mean()
    
    # Calculate Inter-Quartile Range (IQR) of Absolute Price Changes
    iqr_volatility = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weights = [0.4, 0.3, 0.1, 0.2]  # Weights for momentum, volume, std_dev_volatility, and ema_volatility
    final_factor = (adjusted_momentum * weights[0] + 
                    volume_relative * weights[1] + 
                    std_dev_volatility * weights[2] + 
                    ema_volatility * weights[3])
    
    return final_factor
```
