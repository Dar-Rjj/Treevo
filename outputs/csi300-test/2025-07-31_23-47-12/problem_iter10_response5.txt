```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, momentum_weight=0.5, volume_weight=0.3, volatility_weight=0.2):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Dynamic Momentum
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    upper_threshold = cumulative_log_returns.quantile(0.75)
    lower_threshold = cumulative_log_returns.quantile(0.25)
    dynamic_momentum = np.clip(cumulative_log_returns, lower_threshold, upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    relative_volume = volumes / volumes.rolling(window=N).mean()
    volume_adjusted_momentum = dynamic_momentum * relative_volume
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_volatility = absolute_price_changes.rolling(window=M).std()
    ema_volatility = absolute_price_changes.ewm(span=M, adjust=False).mean()
    iqr_volatility = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Final Factor Calculation
    factor = (momentum_weight * volume_adjusted_momentum +
              volume_weight * relative_volume +
              volatility_weight * (std_dev_volatility + ema_volatility + iqr_volatility) / 3)
    
    # Normalize weights to sum to 1
    total_weight = momentum_weight + volume_weight + volatility_weight
    factor /= total_weight
    
    return factor
```
