```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days=30, m_days=20):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Returns
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    momentum = log_returns.rolling(window=n_days).sum()
    
    # Define Dynamic Thresholds
    upper_threshold = momentum.quantile(0.95)
    lower_threshold = momentum.quantile(0.05)
    momentum_clipped = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.mean()
    volume_adjusted_momentum = momentum_clipped * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_volatility = abs_price_changes.rolling(window=m_days).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=m_days, adjust=False).mean()
    high_low_range = df['high'] - df['low']
    
    # Combine Weighted Components
    weights = {'momentum': 0.4, 'volume': 0.3, 'volatility': 0.3}
    combined_factor = (
        weights['momentum'] * volume_adjusted_momentum +
        weights['volume'] * (volumes / mean_volume) +
        weights['volatility'] * (std_volatility + ema_abs_price_changes + high_low_range)
    )
    
    # Ensure Weights Sum to 1
    assert sum(weights.values()) == 1.0
    
    return combined_factor
```
