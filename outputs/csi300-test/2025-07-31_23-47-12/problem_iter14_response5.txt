```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, P=60, M=20, economic_weights=None):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    
    # Apply Dynamic Thresholds
    upper_threshold = momentum.quantile(0.95)
    lower_threshold = momentum.quantile(0.05)
    momentum = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=P).mean()
    volume_ratio = volumes / mean_volume
    adjusted_momentum = momentum * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M).mean()
    iqr = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    high_low_range = df['high'] - df['low']
    
    # Integrate Economic Indicators
    if economic_weights is None:
        economic_weights = {'interest_rate': 0.3, 'gdp_growth_rate': 0.4, 'inflation': 0.3}
    
    economic_factor = sum(df[col] * weight for col, weight in economic_weights.items() if col in df.columns)
    
    # Final Factor Calculation
    weighted_components = {
        'adjusted_momentum': 0.4,
        'std_dev': 0.2,
        'ema_abs_price_changes': 0.1,
        'iqr': 0.1,
        'high_low_range': 0.1,
        'economic_factor': 0.1
    }
    
    final_factor = (adjusted_momentum * weighted_components['adjusted_momentum'] +
                    std_dev * weighted_components['std_dev'] +
                    ema_abs_price_changes * weighted_components['ema_abs_price_changes'] +
                    iqr * weighted_components['iqr'] +
                    high_low_range * weighted_components['high_low_range'] +
                    economic_factor * weighted_components['economic_factor'])
    
    return final_factor
```
