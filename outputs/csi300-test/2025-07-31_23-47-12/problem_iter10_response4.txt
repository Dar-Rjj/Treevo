import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, P=30, M=10, momentum_weight=0.5, volume_weight=0.3, volatility_weight=0.2):
    # Calculate Log Returns
    df['log_returns'] = np.log(df['close']).diff(N)
    
    # Compute Momentum
    df['cumulative_log_returns'] = df['log_returns'].rolling(window=N).sum()
    
    # Define dynamic thresholds for clipping
    upper_threshold = df['cumulative_log_returns'].quantile(0.95)
    lower_threshold = df['cumulative_log_returns'].quantile(0.05)
    
    df['momentum'] = df['cumulative_log_returns'].clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    mean_volume = df['volume'].rolling(window=P).mean()
    df['volume_ratio'] = df['volume'] / mean_volume
    df['volume_adjusted_momentum'] = df['momentum'] * df['volume_ratio']
    
    # Determine Absolute Price Changes
    df['absolute_price_changes'] = df['close'].diff().abs()
    
    # Calculate Advanced Volatility Measures
    df['volatility_std'] = df['absolute_price_changes'].rolling(window=M).std()
    df['volatility_ema'] = df['absolute_price_changes'].ewm(span=M, adjust=False).mean()
    df['volatility_iqr'] = df['absolute_price_changes'].rolling(window=M).quantile(0.75) - df['absolute_price_changes'].rolling(window=M).quantile(0.25)
    
    # Final Factor Calculation
    df['final_factor'] = (
        momentum_weight * df['volume_adjusted_momentum'] +
        volume_weight * df['volume_ratio'] +
        volatility_weight * (df['volatility_std'] + df['volatility_ema'] + df['volatility_iqr']) / 3
    )
    
    return df['final_factor']
