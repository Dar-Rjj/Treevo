```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Momentum
    df['high_low_diff'] = df['high'] - df['low']
    df['high_low_momentum'] = df['high_low_diff'].rolling(window=5).mean()

    # Volume Surge Indicator
    df['volume_change'] = df['volume'].pct_change()
    surge_threshold = 0.10
    df['volume_surge'] = (df['volume_change'] > surge_threshold).astype(int)

    # Combine High-Low Momentum and Volume Surge
    df['factor'] = df['high_low_momentum'] * df['volume_surge']

    # Market Sentiment Indicator
    df['close_change'] = df['close'].pct_change()
    df['market_sentiment'] = (df['close_change'] > 0).astype(int)
    
    # Adjust Factor for Market Sentiment
    df.loc[df['market_sentiment'] == 1, 'factor'] *= 1.1
    df.loc[df['market_sentiment'] == 0, 'factor'] *= 0.9

    return df['factor']
```
