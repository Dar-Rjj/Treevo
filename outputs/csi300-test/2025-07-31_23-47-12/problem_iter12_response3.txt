```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n_days_mom=20, n_days_vol_avg=10, m_days_vol_ema=5):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(n_days_mom)
    
    # Calculate Momentum
    momentum = log_returns
    
    # Retrieve Volumes
    volumes = df['volume']
    
    # Calculate Moving Average Volume over N Days
    avg_volume = volumes.rolling(window=n_days_mom).mean()
    
    # Calculate Volume Ratio (Current Volume / Moving Average Volume)
    volume_ratio = volumes / avg_volume
    
    # Incorporate Volume into Momentum
    vol_adjusted_momentum = momentum * volume_ratio
    
    # Determine Absolute Price Changes
    abs_price_changes = np.abs(close_prices.diff())
    
    # Calculate Exponential Moving Average of Absolute Differences over M Days
    ema_abs_diff = abs_price_changes.ewm(span=m_days_vol_ema, min_periods=m_days_vol_ema).mean()
    
    # Assign Dynamic Weights to Momentum and Volatility
    weight_mom = 0.6  # Example: 60% weight on momentum
    weight_vol = 0.4  # Example: 40% weight on volatility, ensure they sum to 1
    assert weight_mom + weight_vol == 1, "Weights must sum to 1"
    
    # Final Factor Calculation
    factor_values = (weight_mom * vol_adjusted_momentum) + (weight_vol * ema_abs_diff)
    
    return factor_values
```
