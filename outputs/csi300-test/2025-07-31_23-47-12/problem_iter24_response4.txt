```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, K=5, momentum_weight=0.4, volume_weight=0.3, volatility_weight=0.3):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns over N Days
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    cumulative_log_returns = log_returns.cumsum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = cumulative_log_returns.quantile(0.9)
    lower_threshold = cumulative_log_returns.quantile(0.1)
    
    # Clip Log Returns to Thresholds
    momentum = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.rolling(window=N).mean()
    volume_adjusted_momentum = momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures (EMA of Absolute Price Changes over K Days)
    ema_volatility = abs_price_changes.ewm(span=K).mean()
    
    # Final Factor Calculation
    factor = (momentum_weight * volume_adjusted_momentum + 
              volume_weight * (volumes / mean_volume) + 
              volatility_weight * ema_volatility)
    
    # Ensure Weights Sum to 1
    assert np.isclose(momentum_weight + volume_weight + volatility_weight, 1), "Weights must sum to 1"
    
    return factor
```
