import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20, K=10, L=20, sector_data=None, economic_data=None):
    # Obtain close prices
    close_prices = df['close']
    
    # Calculate log returns over N days
    log_returns = np.log(close_prices).diff(N)
    
    # Compute rolling mean and standard deviation of log returns
    rolling_mean = log_returns.rolling(window=N).mean()
    rolling_std = log_returns.rolling(window=N).std()
    
    # Define upper and lower thresholds for dynamic clipping
    upper_threshold = rolling_mean + rolling_std
    lower_threshold = rolling_mean - rolling_std
    
    # Clip log returns to the thresholds
    clipped_log_returns = np.clip(log_returns, lower_threshold, upper_threshold)
    
    # Compute momentum as the cumulative sum of clipped log returns
    momentum = clipped_log_returns.cumsum()
    
    # Adjust for volume
    volumes = df['volume']
    rolling_volume_mean = volumes.rolling(window=N).mean()
    volume_adjustment = (volumes / rolling_volume_mean) * momentum
    
    # Determine absolute price changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate advanced volatility measures
    std_volatility = abs_price_changes.rolling(window=M).std()
    ema_volatility = abs_price_changes.ewm(span=K).mean()
    iqr_volatility = abs_price_changes.rolling(window=L).quantile(0.75) - abs_price_changes.rolling(window=L).quantile(0.25)
    
    # Integrate sector-specific data if provided
    if sector_data is not None:
        sector_volatility = sector_data['volatility']
        std_volatility *= sector_volatility
        ema_volatility *= sector_volatility
        iqr_volatility *= sector_volatility
    
    # Integrate macroeconomic data if provided
    if economic_data is not None:
        interest_rates = economic_data['interest_rates']
        gdp_growth_rate = economic_data['gdp_growth_rate']
        std_volatility *= (1 + interest_rates)
        ema_volatility *= (1 + interest_rates)
        iqr_volatility *= (1 + gdp_growth_rate)
    
    # Combine weighted components
    weights = [0.4, 0.3, 0.3]  # Weights for momentum, volume, and volatility respectively
    final_factor = (weights[0] * volume_adjustment +
                    weights[1] * std_volatility +
                    weights[2] * ema_volatility)
    
    return final_factor
