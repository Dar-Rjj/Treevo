```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff()
    
    # Calculate Cumulative Sum of Log Returns
    cum_log_returns = log_returns.rolling(window=N).sum()
    
    # Define Upper and Lower Thresholds
    upper_threshold = cum_log_returns.quantile(0.95)
    lower_threshold = cum_log_returns.quantile(0.05)
    
    # Clip Log Returns to Thresholds
    clipped_cum_log_returns = cum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Retrieve Volumes
    volume = df['volume']
    
    # Calculate Volume Relative to Mean
    volume_relative_to_mean = volume / volume.mean()
    
    # Multiply by Momentum
    volume_adjusted_momentum = clipped_cum_log_returns * volume_relative_to_mean
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Compute Standard Deviation of Absolute Price Changes
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    
    # Calculate Exponential Moving Average (EMA) of Absolute Price Changes
    ema_abs_price_changes = abs_price_changes.ewm(span=M).mean()
    
    # Calculate Inter-Quartile Range (IQR) of Absolute Price Changes
    iqr_abs_price_changes = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weights = [0.4, 0.3, 0.2, 0.1]  # Weights for momentum, EMA, Std, and IQR
    final_factor = (
        weights[0] * volume_adjusted_momentum +
        weights[1] * ema_abs_price_changes +
        weights[2] * std_abs_price_changes +
        weights[3] * iqr_abs_price_changes
    )
    
    # Ensure Weights Sum to 1
    final_factor = final_factor / sum(weights)
    
    # Integrate Market Breadth and Economic Indicators
    # Calculate Advance-Decline Line
    ad_line = (df['close'] > df['close'].shift(1)).astype(int).cumsum()
    
    # Incorporate New Highs and Lows
    new_highs = (df['close'] == df['close'].rolling(window=20).max()).astype(int)
    new_lows = (df['close'] == df['close'].rolling(window=20).min()).astype(int)
    
    # Use Key Economic Indicators
    # Assuming interest_rates and gdp_growth_rate are available in the DataFrame
    interest_rates = df.get('interest_rates', 0)
    gdp_growth_rate = df.get('gdp_growth_rate', 0)
    
    # Final Factor Calculation
    final_factor = final_factor + 0.1 * (ad_line + new_highs - new_lows + interest_rates + gdp_growth_rate)
    
    return final_factor
```
