```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=20, momentum_weight=0.5, volume_weight=0.3, volatility_weight=0.2):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    cumulative_log_returns = log_returns.cumsum()
    upper_threshold = cumulative_log_returns.quantile(0.9)
    lower_threshold = cumulative_log_returns.quantile(0.1)
    clipped_log_returns = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.mean()
    volume_adjusted_momentum = (volumes / mean_volume) * clipped_log_returns
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_volatility = absolute_price_changes.rolling(window=M).std()
    ema_volatility = absolute_price_changes.ewm(span=M, adjust=False).mean()
    iqr_volatility = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Combine Volatility Metrics
    combined_volatility = (std_volatility + ema_volatility + iqr_volatility) / 3
    
    # Final Factor Calculation
    factor_values = (momentum_weight * volume_adjusted_momentum +
                     volume_weight * (volumes / mean_volume) +
                     volatility_weight * combined_volatility)
    
    # Ensure Weights Sum to 1
    assert np.isclose(momentum_weight + volume_weight + volatility_weight, 1.0), "Weights must sum to 1"
    
    return factor_values
```
