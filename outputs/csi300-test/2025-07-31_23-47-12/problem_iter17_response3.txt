```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20, momentum_weight=0.5, volume_weight=0.3, volatility_weight=0.2):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    log_returns_N_days = log_returns.rolling(window=N).sum()
    
    # Compute Momentum
    momentum = log_returns_N_days.ewm(span=N, adjust=False).mean()
    
    # Define Upper and Lower Thresholds
    upper_threshold = momentum.quantile(0.75)
    lower_threshold = momentum.quantile(0.25)
    
    # Clip Log Returns to Thresholds
    momentum = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_relative = volumes / volumes.rolling(window=M).mean()
    volume_adjusted_momentum = momentum * volume_relative
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_volatility = absolute_price_changes.rolling(window=M).std()
    ema_volatility = absolute_price_changes.ewm(span=M, adjust=False).mean()
    iqr_volatility = absolute_price_changes.rolling(window=M).quantile(0.75) - absolute_price_changes.rolling(window=M).quantile(0.25)
    
    # Final Factor Calculation
    combined_volatility = (std_dev_volatility + ema_volatility + iqr_volatility) / 3
    final_factor = (momentum_weight * volume_adjusted_momentum) + (volatility_weight * combined_volatility)
    
    # Ensure Weights Sum to 1
    weights_sum = momentum_weight + volume_weight + volatility_weight
    if not np.isclose(weights_sum, 1.0, atol=1e-6):
        raise ValueError("Weights must sum to 1.0")
    
    # Integrate Market Breadth and Economic Indicators
    advance_decline_line = (df['close'] > df['close'].shift(1)).astype(int).cumsum()
    new_highs = (df['close'] == df['high']).rolling(window=M).sum()
    new_lows = (df['close'] == df['low']).rolling(window=M).sum()
    
    market_breadth_indicator = (new_highs - new_lows) / M
    final_factor = final_factor + (market_breadth_indicator * 0.1)  # Adding a small weight for the market breadth indicator
    
    return final_factor
```
