```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=5):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Log Returns
    log_returns = np.log(close_prices) - np.log(close_prices.shift(1))
    
    # Compute Momentum
    cum_sum_log_returns = log_returns.rolling(window=N).sum()
    upper_threshold = cum_sum_log_returns.quantile(0.75)
    lower_threshold = cum_sum_log_returns.quantile(0.25)
    clipped_momentum = cum_sum_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    mean_volume = volumes.mean()
    volume_adjusted_momentum = clipped_momentum * (volumes / mean_volume)
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev_volatility = abs_price_changes.rolling(window=M).std()
    ema_volatility = abs_price_changes.ewm(span=M).mean()
    iqr_volatility = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Calculate Average True Range (ATR)
    high_low_diff = df['high'] - df['low']
    high_close_diff = (df['high'] - df['close'].shift(1)).abs()
    low_close_diff = (df['low'] - df['close'].shift(1)).abs()
    true_range = pd.concat([high_low_diff, high_close_diff, low_close_diff], axis=1).max(axis=1)
    atr = true_range.rolling(window=M).mean()
    
    # Calculate Bollinger Bands Width
    middle_band = close_prices.rolling(window=M).mean()
    std_dev = close_prices.rolling(window=M).std()
    upper_band = middle_band + 2 * std_dev
    lower_band = middle_band - 2 * std_dev
    bollinger_width = (upper_band - lower_band) / middle_band
    
    # Combine Weighted Momentum, Volume, and Volatility Components
    weights = np.array([0.4, 0.3, 0.1, 0.1, 0.05, 0.05])
    factor = (weights[0] * volume_adjusted_momentum +
              weights[1] * std_dev_volatility +
              weights[2] * ema_volatility +
              weights[3] * iqr_volatility +
              weights[4] * atr +
              weights[5] * bollinger_width)
    
    # Ensure Weights Sum to 1
    assert np.isclose(weights.sum(), 1), "Weights must sum to 1"
    
    return factor
```
