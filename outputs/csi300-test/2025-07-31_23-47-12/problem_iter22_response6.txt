import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20, K=10, L=5):
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff()
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    
    # Define and update thresholds
    upper_threshold = momentum.rolling(window=M * 4).mean() + 2 * momentum.rolling(window=M * 4).std()
    lower_threshold = momentum.rolling(window=M * 4).mean() - 2 * momentum.rolling(window=M * 4).std()
    
    # Clip log returns to thresholds
    clipped_momentum = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volumes = df['volume']
    volume_relative = volumes / volumes.rolling(window=N).mean()
    adjusted_momentum = clipped_momentum * volume_relative
    
    # Determine Absolute Price Changes
    absolute_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_dev = absolute_price_changes.rolling(window=M).std()
    ema = absolute_price_changes.ewm(span=K).mean()
    iqr = absolute_price_changes.rolling(window=L).quantile(0.75) - absolute_price_changes.rolling(window=L).quantile(0.25)
    
    # Final Factor Calculation
    weights = np.array([0.3, 0.3, 0.4])  # Weights for momentum, volume, and volatility
    final_factor = (weights[0] * adjusted_momentum + 
                    weights[1] * (volumes / volumes.rolling(window=N).mean()) + 
                    weights[2] * (std_dev + ema + iqr) / 3)
    
    return final_factor
