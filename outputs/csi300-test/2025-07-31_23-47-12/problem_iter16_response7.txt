import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=10, interest_rate=None, gdp_growth_rate=None):
    # Calculate Log Returns
    log_returns = np.log(df['close']).diff().fillna(0)
    
    # Compute Momentum
    cumulative_log_returns = log_returns.rolling(window=N).sum()
    upper_threshold = cumulative_log_returns.quantile(0.75)
    lower_threshold = cumulative_log_returns.quantile(0.25)
    momentum = cumulative_log_returns.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volume_mean = df['volume'].rolling(window=N).mean()
    volume_adjusted_momentum = momentum * (df['volume'] / volume_mean)
    
    # Determine Absolute Price Changes
    abs_price_changes = df['close'].diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_abs_price_changes = abs_price_changes.rolling(window=M).std()
    ema_abs_price_changes = abs_price_changes.ewm(span=M, adjust=False).mean()
    iqr_abs_price_changes = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Market Breadth and Economic Indicators
    advance_decline_line = (df['close'] > df['close'].shift(1)).rolling(window=N).sum() - (df['close'] < df['close'].shift(1)).rolling(window=N).sum()
    
    # Combine Weighted Components
    weights = {
        'momentum': 0.3,
        'volume_adjusted_momentum': 0.2,
        'std_abs_price_changes': 0.1,
        'ema_abs_price_changes': 0.1,
        'iqr_abs_price_changes': 0.1,
        'advance_decline_line': 0.2
    }
    
    factor_values = (
        weights['momentum'] * momentum +
        weights['volume_adjusted_momentum'] * volume_adjusted_momentum +
        weights['std_abs_price_changes'] * std_abs_price_changes +
        weights['ema_abs_price_changes'] * ema_abs_price_changes +
        weights['iqr_abs_price_changes'] * iqr_abs_price_changes +
        weights['advance_decline_line'] * advance_decline_line
    )
    
    # Incorporate Key Economic Indicators
    if interest_rate is not None:
        factor_values += 0.05 * interest_rate
    if gdp_growth_rate is not None:
        factor_values += 0.05 * gdp_growth_rate
    
    return factor_values
