import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=30, momentum_thresholds=(-0.05, 0.05), vol_weight=0.3, mom_weight=0.4, econ_weight=0.3):
    """
    Generate a novel alpha factor based on Volume Adjusted Momentum with Price Change Volatility and Market Indicators.
    
    :param df: DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    :param N: Number of days to compute log returns
    :param M: Number of days to compute advanced volatility measures
    :param momentum_thresholds: Tuple (lower, upper) for clipping momentum
    :param vol_weight: Weight for the volatility component
    :param mom_weight: Weight for the momentum component
    :param econ_weight: Weight for the economic indicators component
    :return: Series of alpha factor values
    """
    
    # Calculate Log Returns
    close_prices = df['close']
    log_returns = np.log(close_prices).diff(N)
    
    # Compute Momentum
    momentum = log_returns.rolling(window=N).sum()
    lower_threshold, upper_threshold = momentum_thresholds
    momentum_clipped = momentum.clip(lower=lower_threshold, upper=upper_threshold)
    
    # Adjust for Volume
    volume = df['volume']
    mean_volume = volume.rolling(window=M).mean()
    volume_adjustment = volume / mean_volume
    adjusted_momentum = momentum_clipped * volume_adjustment
    
    # Determine Absolute Price Changes
    abs_price_changes = close_prices.diff().abs()
    
    # Calculate Advanced Volatility Measures
    std_volatility = abs_price_changes.rolling(window=M).std()
    ema_volatility = abs_price_changes.ewm(span=M, min_periods=M).mean()
    iqr_volatility = abs_price_changes.rolling(window=M).quantile(0.75) - abs_price_changes.rolling(window=M).quantile(0.25)
    
    # Integrate Market Breadth Indicators
    advance_decline_line = (df['close'] > df['close'].shift(1)).cumsum() - (df['close'] < df['close'].shift(1)).cumsum()
    
    # Incorporate New Highs and Lows
    new_highs = (df['close'] > df['high'].rolling(window=M).max()).astype(int)
    new_lows = (df['close'] < df['low'].rolling(window=M).min()).astype(int)
    market_breadth = new_highs - new_lows
    
    # Final Factor Calculation
    volatility_component = (std_volatility + ema_volatility + iqr_volatility) / 3
    total_weight = vol_weight + mom_weight + econ_weight
    weighted_volatility = vol_weight / total_weight * volatility_component
    weighted_momentum = mom_weight / total_weight * adjusted_momentum
    weighted_econ = econ_weight / total_weight * (advance_decline_line + market_breadth) / 2
    final_factor = weighted_volatility + weighted_momentum + weighted_econ
    
    return final_factor
