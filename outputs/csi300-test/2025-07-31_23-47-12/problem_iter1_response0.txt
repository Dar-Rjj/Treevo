```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-to-Low Range
    high_to_low_range = df['high'] - df['low']
    
    # Calculate High-to-Low Range Ratio
    high_to_low_range_ratio = high_to_low_range / df['close']
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff(periods=1)
    
    # Combine Factors
    alpha_factor = high_to_low_range_ratio * price_momentum
    
    return alpha_factor
```
