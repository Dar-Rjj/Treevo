```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=20, m=20, momentum_weight=0.6, volatility_weight=0.4):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Momentum
    # Compute Log Return over N Days
    log_returns = np.log(close_prices).diff(n)
    
    # Adjust for Volume
    # Retrieve Volumes
    volumes = df['volume']
    # Calculate Volume Ratio (Current Volume / Moving Average Volume)
    moving_average_volume = volumes.rolling(window=n).mean()
    volume_ratio = volumes / moving_average_volume
    
    # Incorporate Volume into Momentum
    volume_adjusted_momentum = log_returns * volume_ratio
    
    # Incorporate Price Change Volatility
    # Determine Absolute Price Changes
    price_changes = close_prices.diff().abs()
    # Calculate Standard Deviation over M Days
    price_change_volatility = price_changes.rolling(window=m).std()
    
    # Final Factor Calculation
    # Combine Weighted Momentum and Volatility Component
    final_factor = (momentum_weight * volume_adjusted_momentum) + (volatility_weight * price_change_volatility)
    
    return final_factor
```
