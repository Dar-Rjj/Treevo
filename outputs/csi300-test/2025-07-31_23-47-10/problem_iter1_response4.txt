```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    
    # Calculate 10-Day Moving Average of Volume
    df['volume_10_day_ma'] = df['volume'].rolling(window=10).mean()
    
    # Calculate Volume Intensity
    df['volume_intensity'] = df['volume'] / df['volume_10_day_ma']
    
    # Generate Weighted Momentum Reversal Score
    df['weighted_momentum_reversal'] = (df['daily_price_change'] * -1) * df['volume_intensity']
    
    return df['weighted_momentum_reversal']
```
