```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_adj = (volume_trend > 0) * 1.5 + (volume_trend <= 0) * 0.5
    
    # Adjust High-Low Spread by Volume Trend
    adj_high_low_spread = high_low_spread * volume_trend_adj
    
    # Calculate Short-Term, Medium-Term, and Long-Term Price Trends
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    
    short_term_trend = df['close'] - short_term_ema
    medium_term_trend = df['close'] - medium_term_ema
    long_term_trend = df['close'] - long_term_ema
    
    # Incorporate Short-Term Price Trend
    short_term_trend_adj = (short_term_trend > 0) * 1.2 + (short_term_trend <= 0) * 0.8
    # Incorporate Medium-Term Price Trend
    medium_term_trend_adj = (medium_term_trend > 0) * 1.1 + (medium_term_trend <= 0) * 0.9
    # Incorporate Long-Term Price Trend
    long_term_trend_adj = (long_term_trend > 0) * 1.3 + (long_term_trend <= 0) * 0.7
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_adj = (volatility > volatility.quantile(0.5)) * 1.4 + (volatility <= volatility.quantile(0.5)) * 0.6
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5d = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5d / long_term_ema_20d
    relative_strength_adj = (relative_strength > 1) * 1.5 + (relative_strength <= 1) * 0.5
    
    # Combine all factors
    alpha_factor = (adj_high_low_spread 
                    * short_term_trend_adj 
                    * medium_term_trend_adj 
                    * long_term_trend_adj 
                    * volatility_adj 
                    * relative_strength_adj)
    
    return alpha_factor
```
