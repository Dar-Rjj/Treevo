```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    
    # Calculate Short-Term Price Trend
    close_10_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - close_10_day_ema
    
    # Calculate Medium-Term Price Trend
    close_30_day_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - close_30_day_ema
    
    # Calculate Long-Term Price Trend
    close_50_day_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - close_50_day_ema
    
    # Calculate Dynamic Volatility
    dynamic_volatility = df['close'].rolling(window=10).std()
    
    # Integrate Momentum and Relative Strength
    close_5_day_ema = df['close'].ewm(span=5, adjust=False).mean()
    close_20_day_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_5_day_ema / close_20_day_ema
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend > 0] *= 1.5
    adjusted_spread[volume_trend <= 0] *= 0.5
    
    # Incorporate Short-Term Price Trend
    adjusted_spread[short_term_price_trend > 0] *= 1.2
    adjusted_spread[short_term_price_trend <= 0] *= 0.8
    
    # Incorporate Medium-Term Price Trend
    adjusted_spread[medium_term_price_trend > 0] *= 1.1
    adjusted_spread[medium_term_price_trend <= 0] *= 0.9
    
    # Incorporate Long-Term Price Trend
    adjusted_spread[long_term_price_trend > 0] *= 1.3
    adjusted_spread[long_term_price_trend <= 0] *= 0.7
    
    # Incorporate Dynamic Volatility
    adjusted_spread[dynamic_volatility > dynamic_volatility.mean()] *= 1.4
    adjusted_spread[dynamic_volatility <= dynamic_volatility.mean()] *= 0.6
    
    # Incorporate Momentum and Relative Strength
    adjusted_spread[relative_strength > 1] *= 1.8
    adjusted_spread[relative_strength <= 1] *= 0.4
    
    return adjusted_spread
```
