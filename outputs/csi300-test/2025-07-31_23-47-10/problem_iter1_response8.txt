```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    momentum = df['close'] - df['close'].shift(5)
    momentum_normalized = momentum / df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Reversal
    reversal = df['close'] - df['close'].shift(20)
    reversal_normalized = reversal / df['close'].rolling(window=20).mean()
    
    # Combine Momentum and Reversal
    combined_mr = 0.5 * (momentum_normalized + reversal_normalized)
    
    # Calculate Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(10)) / df['volume'].rolling(window=10).mean()
    
    # Combine with Volume Trend
    alpha_factor = combined_mr * volume_trend
    
    return alpha_factor
```
