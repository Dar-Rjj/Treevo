```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    open_close_spread = df['open'] - df['close']
    
    # Average of High-Low and Open-Close Spreads
    avg_spread = (high_low_spread + open_close_spread) / 2
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    is_volume_trend_positive = volume_trend > 0
    is_volume_trend_negative = volume_trend < 0
    
    # Adjust for positive and negative trends in the combined spread
    adjusted_spread = avg_spread.where(is_volume_trend_positive, avg_spread * 1.5).where(~is_volume_trend_negative, avg_spread * 0.5)
    
    # Calculate Daily Volatility
    daily_volatility = df['close'].rolling(window=10).std()
    volatility_threshold = daily_volatility.mean()
    is_volatility_high = daily_volatility > volatility_threshold
    
    # Incorporate Volatility
    adjusted_spread_vol = adjusted_spread.where(is_volatility_high, adjusted_spread * 1.2).where(~is_volatility_high, adjusted_spread * 0.8)
    
    # Final Adjustment
    final_factor = adjusted_spread_vol.where(is_volume_trend_positive, adjusted_spread_vol * 1.5).where(~is_volume_trend_negative, adjusted_spread_vol * 0.5)
    
    return final_factor
```
