```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_positive = price_trend > 0
    
    # Integrate Momentum and Relative Strength
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_ema_5 / close_ema_20
    
    # Adaptive Volatility
    close_std_20 = df['close'].rolling(window=20).std()
    close_ma_20 = df['close'].rolling(window=20).mean()
    volatility = close_std_20 / close_ma_20
    
    # Dynamic Weight Adjustment
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_adjusted = high_low_spread * np.where(volume_trend_positive, 1.5, 0.5)
    price_trend_adjusted = np.where(price_trend_positive, 1.2, 0.8)
    final_factor = spread_adjusted * price_trend_adjusted * relative_strength_weight * volatility_weight
    
    return final_factor
```
