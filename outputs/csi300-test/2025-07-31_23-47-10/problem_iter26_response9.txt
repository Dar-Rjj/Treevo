import pandas as pd

def heuristics_v2(df):
    # Calculate the relative position of the closing price within the high-low range
    df['close_position'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the ratio of the closing price to the open price
    df['close_open_ratio'] = df['close'] / df['open']
    
    # Calculate the volume weighted amount
    df['volume_weighted_amount'] = df['amount'] / df['volume']
    
    # Combine the factors with a simple average to create an alpha factor
    df['alpha_factor'] = (df['close_position'] + df['close_open_ratio'] + df['volume_weighted_amount']) / 3
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
