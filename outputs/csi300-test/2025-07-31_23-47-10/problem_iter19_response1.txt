```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - short_term_ema
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - medium_term_ema
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - long_term_ema
    
    # Calculate Dynamic Volatility (Standard Deviation over 10 days)
    dynamic_volatility = df['close'].rolling(window=10).std()
    volatility_percentile = dynamic_volatility > dynamic_volatility.median()
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread.copy()
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread[volume_trend > 0] *= 1.5
    adjusted_high_low_spread[volume_trend <= 0] *= 0.5
    
    # Incorporate Short-Term Price Trend
    adjusted_high_low_spread[short_term_price_trend > 0] *= 1.2
    adjusted_high_low_spread[short_term_price_trend <= 0] *= 0.8
    
    # Incorporate Medium-Term Price Trend
    adjusted_high_low_spread[medium_term_price_trend > 0] *= 1.1
    adjusted_high_low_spread[medium_term_price_trend <= 0] *= 0.9
    
    # Incorporate Long-Term Price Trend
    adjusted_high_low_spread[long_term_price_trend > 0] *= 1.3
    adjusted_high_low_spread[long_term_price_trend <= 0] *= 0.7
    
    # Incorporate Dynamic Volatility
    adjusted_high_low_spread[volatility_percentile] *= 1.4
    adjusted_high_low_spread[~volatility_percentile] *= 0.6
    
    # Incorporate Momentum and Relative Strength
    adjusted_high_low_spread[relative_strength > 1] *= 1.7
    adjusted_high_low_spread[relative_strength <= 1] *= 0.3
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread
    
    return alpha_factor
```
