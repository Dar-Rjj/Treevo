```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    
    # Determine if positive or negative volume trend
    volume_trend_direction = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Calculate Price Trend using EMA
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    
    # Determine if positive or negative price trend
    price_trend_direction = np.where(price_trend > 0, 1.2, 0.8)
    
    # Combine Spread, Volume, and Price Trends
    adjusted_spread = high_low_spread * volume_trend
    adjusted_spread_with_volume_trend = adjusted_spread * volume_trend_direction
    final_alpha_factor = adjusted_spread_with_volume_trend * price_trend_direction
    
    return final_alpha_factor
```
