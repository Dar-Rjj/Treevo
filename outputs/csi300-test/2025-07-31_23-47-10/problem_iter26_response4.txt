import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA_10']
    
    # Calculate Short-Term Price Trend
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Trend'] = df['close'] - df['EMA_10']
    
    # Calculate Medium-Term Price Trend
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Price_Trend'] = df['close'] - df['EMA_30']
    
    # Calculate Long-Term Price Trend
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Price_Trend'] = df['close'] - df['EMA_50']
    
    # Calculate Dynamic Volatility
    df['Volatility'] = df['close'].rolling(window=10).std()
    df['Volatility_Score'] = 0
    df.loc[df['Volatility'] > df['Volatility'].quantile(0.8), 'Volatility_Score'] = 2.5
    df.loc[(df['Volatility'] > df['Volatility'].quantile(0.6)) & (df['Volatility'] <= df['Volatility'].quantile(0.8)), 'Volatility_Score'] = 1.5
    df.loc[(df['Volatility'] > df['Volatility'].quantile(0.4)) & (df['Volatility'] <= df['Volatility'].quantile(0.6)), 'Volatility_Score'] = 1.0
    df.loc[df['Volatility'] <= df['Volatility'].quantile(0.4), 'Volatility_Score'] = 0.5
    
    # Integrate Momentum and Relative Strength
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Relative_Strength'] = df['EMA_5'] / df['EMA_20']
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    df['Adjusted_Spread'] = df['High-Low_Spread']
    df.loc[df['Volume_Trend'] > 0, 'Adjusted_Spread'] *= 1.5
    df.loc[df['Volume_Trend'] <= 0, 'Adjusted_Spread'] *= 0.5
    
    df['Adjusted_Short_Term_Trend'] = df['Short_Term_Price_Trend']
    df.loc[df['Short_Term_Price_Trend'] > 0, 'Adjusted_Short_Term_Trend'] *= 1.2
    df.loc[df['Short_Term_Price_Trend'] <= 0, 'Adjusted_Short_Term_Trend'] *= 0.8
    
    df['Adjusted_Medium_Term_Trend'] = df['Medium_Term_Price_Trend']
    df.loc[df['Medium_Term_Price_Trend'] > 0, 'Adjusted_Medium_Term_Trend'] *= 1.1
    df.loc[df['Medium_Term_Price_Trend'] <= 0, 'Adjusted_Medium_Term_Trend'] *= 0.9
    
    df['Adjusted_Long_Term_Trend'] = df['Long_Term_Price_Trend']
    df.loc[df['Long_Term_Price_Trend'] > 0, 'Adjusted_Long_Term_Trend'] *= 1.3
    df.loc[df['Long_Term_Price_Trend'] <= 0, 'Adjusted_Long_Term_Trend'] *= 0.7
    
    df['Adjusted_Volatility'] = df['Volatility_Score']
    
    df['Adjusted_Relative_Strength'] = 1
    df.loc[df['Relative_Strength'] > 1, 'Adjusted_Relative_Strength'] = 1.7
    df.loc[df['Relative_Strength'] <= 1, 'Adjusted_Relative_Strength'] = 0.3
