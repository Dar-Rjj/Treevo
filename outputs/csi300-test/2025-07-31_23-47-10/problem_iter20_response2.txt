```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_10_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_ma
    volume_trend_sign = np.where(volume_trend > 0, 'positive', 'negative')

    # Calculate Short-Term Price Trend (EMA over 10 days)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - short_term_ema
    short_term_price_trend_sign = np.where(short_term_price_trend > 0, 'positive', 'negative')

    # Calculate Medium-Term Price Trend (EMA over 30 days)
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - medium_term_ema
    medium_term_price_trend_sign = np.where(medium_term_price_trend > 0, 'positive', 'negative')

    # Calculate Long-Term Price Trend (EMA over 50 days)
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - long_term_ema
    long_term_price_trend_sign = np.where(long_term_price_trend > 0, 'positive', 'negative')

    # Calculate Dynamic Volatility (Standard Deviation over 20 days)
    volatility = df['close'].rolling(window=20).std()
    volatility_quantile = volatility.quantile(0.75)
    volatility_label = np.where(volatility > volatility_quantile, 'high', 'low')

    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    relative_strength_label = np.where(relative_strength > 1, 'strong', 'weak')

    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread.copy()
    adjusted_high_low_spread[volume_trend_sign == 'positive'] *= 1.5
    adjusted_high_low_spread[volume_trend_sign == 'negative'] *= 0.5

    adjusted_high_low_spread[short_term_price_trend_sign == 'positive'] *= 1.2
    adjusted_high_low_spread[short_term_price_trend_sign == 'negative'] *= 0.8

    adjusted_high_low_spread[medium_term_price_trend_sign == 'positive'] *= 1.1
    adjusted_high_low_spread[medium_term_price_trend_sign == 'negative'] *= 0.9

    adjusted_high_low_spread[long_term_price_trend_sign == 'positive'] *= 1.3
    adjusted_high_low_spread[long_term_price_trend_sign == 'negative'] *= 0.7

    adjusted_high_low_spread[volatility_label == 'high'] *= 1.4
    adjusted_high_low_spread[volatility_label == 'low'] *= 0.6

    adjusted_high_low_spread[relative_strength_label == 'strong'] *= 1.5
    adjusted_high_low_spread[relative_strength_label == 'weak'] *= 0.5

    # Consider Market Context
    # Assume market trend is given as 'bullish' or 'bearish'
    market_trend = 'bullish'  # Replace with actual market trend
    if market_trend == 'bullish':
        adjusted_high_low_spread[long_term_price_trend_sign == 'positive'] *= 1.5
    elif market_trend == 'bearish':
        adjusted_high_low_spread[long_term_price_trend_sign == 'positive'] *= 0.5

    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread

    return alpha_factor
```
