import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day Exponential Moving Average (EMA) of Volume
    ema_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - ema_volume_10
    
    # Determine if the trend is positive or negative
    volume_trend_sign = np.where(volume_trend > 0, 1, -1)
    
    # Adjust for positive and negative trends
    adjusted_trend = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Combine High-Low Spread with Adjusted Volume Trend
    factor = high_low_spread * adjusted_trend
    
    return pd.Series(factor, index=df.index, name='heuristics_v2')
