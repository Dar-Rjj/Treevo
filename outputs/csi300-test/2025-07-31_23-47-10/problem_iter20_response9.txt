```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    
    # Determine if positive or negative trend
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    short_term_trend_positive = short_term_trend > 0
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30
    medium_term_trend_positive = medium_term_trend > 0
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    long_term_trend_positive = long_term_trend > 0
    
    # Calculate Dynamic Volatility (Standard deviation over 10 days)
    std_10 = df['close'].rolling(window=10).std()
    volatility_score = np.where(std_10 > std_10.quantile(0.75), 2.0,
                                np.where(std_10 > std_10.quantile(0.5), 1.0, 0.5))
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    relative_strength_factor = np.where(relative_strength > 1, 1.7, 0.3)
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread.copy()
    adjusted_high_low_spread[volume_trend_positive] *= 1.5
    adjusted_high_low_spread[~volume_trend_positive] *= 0.5
    
    adjusted_high_low_spread[short_term_trend_positive] *= 1.2
    adjusted_high_low_spread[~short_term_trend_positive] *= 0.8
    
    adjusted_high_low_spread[medium_term_trend_positive] *= 1.1
    adjusted_high_low_spread[~medium_term_trend_positive] *= 0.9
    
    adjusted_high_low_spread[long_term_trend_positive] *= 1.3
    adjusted_high_low_spread[~long_term_trend_positive] *= 0.7
    
    adjusted_high_low_spread *= volatility_score
    adjusted_high_low_spread *= relative_strength_factor
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread
    
    return alpha_factor
```
