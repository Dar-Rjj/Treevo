import pandas as pd

def heuristics_v2(df):
    # Calculate the daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate the daily range
    df['range'] = df['high'] - df['low']
    
    # Calculate the buying pressure (amount / volume)
    df['buying_pressure'] = df['amount'] / df['volume']
    
    # Calculate the relative buying pressure (normalized by range)
    df['relative_buying_pressure'] = df['buying_pressure'] / df['range']
    
    # Calculate the momentum (average of the last 5 days' returns)
    df['momentum'] = df['return'].rolling(window=5).mean()
    
    # Generate the alpha factor
    df['alpha_factor'] = (df['relative_buying_pressure'] * df['momentum']) / df['range']
    
    # Return the alpha factor
    return df['alpha_factor'].dropna()
