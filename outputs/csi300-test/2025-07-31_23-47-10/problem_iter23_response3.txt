```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    vol_10d_ma = df['volume'].rolling(window=10).mean()
    daily_vol_trend = df['volume'] - vol_10d_ma
    vol_trend_positive = (daily_vol_trend > 0).astype(int)

    # Calculate Short-Term Price Trend (EMA over 10 days)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - short_term_ema
    short_term_trend_positive = (short_term_trend > 0).astype(int)

    # Calculate Medium-Term Price Trend (EMA over 30 days)
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - medium_term_ema
    medium_term_trend_positive = (medium_term_trend > 0).astype(int)

    # Calculate Long-Term Price Trend (EMA over 50 days)
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - long_term_ema
    long_term_trend_positive = (long_term_trend > 0).astype(int)

    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_score = np.where(volatility > np.percentile(volatility, 80), 2.5,
                                np.where((volatility > np.percentile(volatility, 60)) & (volatility <= np.percentile(volatility, 80)), 1.5,
                                         np.where((volatility > np.percentile(volatility, 40)) & (volatility <= np.percentile(volatility, 60)), 1.0, 0.5)))

    # Integrate Momentum and Relative Strength
    short_term_ema_5d = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5d / long_term_ema_20d
    momentum_score = np.where(relative_strength > 1, 1.7, 0.3)

    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread * (1.5 if vol_trend_positive else 0.5)
    adjusted_short_term_trend = short_term_trend * (1.2 if short_term_trend_positive else 0.8)
    adjusted_medium_term_trend = medium_term_trend * (1.1 if medium_term_trend_positive else 0.9)
    adjusted_long_term_trend = long_term_trend * (1.3 if long_term_trend_positive else 0.7)

    # Incorporate Dynamic Volatility
    final_factor = (adjusted_high_low_spread + adjusted_short_term_trend + adjusted_medium_term_trend + adjusted_long_term_trend) * volatility_score * momentum_score

    return final_factor
```
