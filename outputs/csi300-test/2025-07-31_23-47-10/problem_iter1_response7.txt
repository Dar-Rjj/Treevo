import pandas as pd

def heuristics_v2(df):
    # Compute 20-Day and 50-Day Moving Averages of Close Prices
    df['20_day_MA'] = df['close'].rolling(window=20).mean()
    df['50_day_MA'] = df['close'].rolling(window=50).mean()
    
    # Calculate Long-Term Price Momentum
    df['long_term_momentum'] = df['20_day_MA'] - df['50_day_MA']
    
    # Determine Daily Price Change
    df['daily_price_change'] = df['open'] - df['close']

    # Compute 5-Day Sum of Daily Price Changes
    df['5_day_sum_daily_change'] = df['daily_price_change'].rolling(window=5).sum()

    # Determine High-Low Range over 5 Days
    df['5_day_high'] = df['high'].rolling(window=5).max()
    df['5_day_low'] = df['low'].rolling(window=5).min()
    df['high_low_range'] = df['5_day_high'] - df['5_day_low']

    # Calculate Short-Term Reversal Signal
    df['reversal_signal'] = (df['5_day_sum_daily_change'] / df['high_low_range']) * df['volume']

    # Combine the long-term momentum and short-term reversal signal
    df['price_momentum_reversal'] = df['long_term_momentum'] + df['reversal_signal']

    # Return the factor values
    return df['price_momentum_reversal'].dropna()
