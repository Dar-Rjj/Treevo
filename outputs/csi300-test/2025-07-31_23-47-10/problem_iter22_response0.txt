```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Short-Term Price Trend
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_10d_ema
    short_term_trend_positive = (short_term_trend > 0).astype(int)
    
    # Calculate Medium-Term Price Trend
    close_30d_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_30d_ema
    medium_term_trend_positive = (medium_term_trend > 0).astype(int)
    
    # Calculate Long-Term Price Trend
    close_50d_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_50d_ema
    long_term_trend_positive = (long_term_trend > 0).astype(int)
    
    # Calculate Dynamic Volatility
    close_10d_std = df['close'].rolling(window=10).std()
    volatility_75th = close_10d_std.quantile(0.75)
    volatility_50th = close_10d_std.quantile(0.5)
    volatility_score = np.where(close_10d_std > volatility_75th, 2.0,
                                np.where((close_10d_std <= volatility_75th) & (close_10d_std > volatility_50th), 1.0, 0.5))
    
    # Integrate Momentum and Relative Strength
    close_5d_ema = df['close'].ewm(span=5, adjust=False).mean()
    close_20d_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_5d_ema / close_20d_ema
    momentum_factor = np.where(relative_strength > 1, 
                               np.where(short_term_trend_positive, 1.2, 0.8),
                               np.where(short_term_trend_positive, 1.1, 0.9))
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread * (volume_trend_positive * 1.5 + (1 - volume_trend_positive) * 0.5)
    combined_factor = (adjusted_spread * 
                       (short_term_trend_positive * 1.2 + (1 - short_term_trend_positive) * 0.8) *
                       (medium_term_trend_positive * 1.1 + (1 - medium_term_trend_positive) * 0.9) *
                       (long_term_trend_positive * 1.3 + (1 - long_term_trend_positive) * 0.7) *
                       volatility_score *
                       momentum_factor)
    
    # Final Alpha Factor
    alpha_factor = combined_factor * np.where(relative_strength > 1, 1.7, 0.3)
    
    return alpha_factor
```
