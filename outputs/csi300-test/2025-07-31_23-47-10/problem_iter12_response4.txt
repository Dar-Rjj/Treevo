```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_5d_ma = df['volume'].rolling(window=5).mean()
    volume_trend = df['volume'] - volume_5d_ma
    
    # Calculate Short-Term Price Trend
    close_5d_ema = df['close'].ewm(span=5, adjust=False).mean()
    short_term_trend = df['close'] - close_5d_ema
    
    # Calculate Long-Term Price Trend
    close_20d_ema = df['close'].ewm(span=20, adjust=False).mean()
    long_term_trend = df['close'] - close_20d_ema
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate Relative Volatility
    relative_volatility = np.where(volatility > volatility.quantile(0.5), 1.3, 0.7)
    
    # Combine Spread, Volume, Price Trends, Volatility, and Market Sentiment
    alpha_factor = (high_low_spread * 
                    (np.where(volume_trend > 0, 1.5, 0.5) * volume_trend) * 
                    (np.where(short_term_trend > 0, 1.2, 0.8)) * 
                    (np.where(long_term_trend > 0, 1.1, 0.9)) * 
                    relative_volatility)
    
    # Incorporate Market Sentiment (Assuming a column 'sentiment' is available in the DataFrame)
    if 'sentiment' in df.columns:
        sentiment_adjustment = np.where(df['sentiment'] > 0, 1.4, 0.6)
        alpha_factor *= sentiment_adjustment
    
    return alpha_factor
```
