```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day Exponential Moving Average (EMA) of Volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - ema_volume
    
    # Determine if positive or negative trend
    volume_trend_factor = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Combine High-Low Spread and Volume Trend
    combined_factor = (high_low_spread * volume_trend) * volume_trend_factor
    
    return pd.Series(combined_factor, index=df.index, name='alpha_factor')
```
