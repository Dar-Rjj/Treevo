```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Determine if positive or negative trend for volume
    volume_trend_positive = (volume_trend > 0).astype(int)
    volume_trend_negative = (volume_trend <= 0).astype(int)
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    
    # Determine if positive or negative trend for price
    price_trend_positive = (price_trend > 0).astype(int)
    price_trend_negative = (price_trend <= 0).astype(int)
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Combine Spread, Volume, Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread.copy()
    adjusted_high_low_spread *= (volume_trend_positive * 1.5 + volume_trend_negative * 0.5)
    adjusted_high_low_spread *= (price_trend_positive * 1.2 + price_trend_negative * 0.8)
    
    # Adjust for Volatility
    volatility_high = (volatility > volatility.quantile(0.5)).astype(int)
    volatility_low = (volatility <= volatility.quantile(0.5)).astype(int)
    adjusted_high_low_spread *= (volatility_high * 1.3 + volatility_low * 0.7)
    
    return adjusted_high_low_spread
```
