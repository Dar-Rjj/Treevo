```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = volume_10d_ma - df['volume']
    
    # Determine if positive or negative trend
    is_positive_trend = (volume_trend > 0)
    
    # Combine Spread and Volume Trend
    factor = high_low_spread * volume_trend
    
    # Adjust for positive and negative trends
    factor[is_positive_trend] *= 1.5
    factor[~is_positive_trend] *= 0.5
    
    return factor
```
