```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - short_term_ema
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - medium_term_ema
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - long_term_ema
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_score = np.where(volatility > np.percentile(volatility.dropna(), 80), 2.5,
                                np.where(volatility > np.percentile(volatility.dropna(), 60), 1.5,
                                         np.where(volatility > np.percentile(volatility.dropna(), 40), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread * np.where(daily_volume_trend > 0, 1.5, 0.5)
    adjusted_short_term_trend = short_term_price_trend * np.where(short_term_price_trend > 0, 1.2, 0.8)
    adjusted_medium_term_trend = medium_term_price_trend * np.where(medium_term_price_trend > 0, 1.1, 0.9)
    adjusted_long_term_trend = long_term_price_trend * np.where(long_term_price_trend > 0, 1.3, 0.7)
    adjusted_volatility = volatility * volatility_score
    adjusted_relative_strength = np.where(relative_strength > 1, 1.7, 0.3)
    
    # Final Alpha Factor
    alpha_factor = (adjusted_high_low_spread + 
                    adjusted_short_term_trend + 
                    adjusted_medium_term_trend + 
                    adjusted_long_term_trend + 
                    adjusted_volatility) * adjusted_relative_strength
    
    return alpha_factor
```
