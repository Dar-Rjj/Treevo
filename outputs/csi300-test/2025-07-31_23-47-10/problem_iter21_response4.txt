```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA_10']
    df['Volume_Trend_Sign'] = np.where(df['Volume_Trend'] > 0, 1, 0)
    
    # Calculate Short-Term Price Trend
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Trend'] = df['close'] - df['EMA_10']
    df['Short_Term_Price_Trend_Sign'] = np.where(df['Short_Term_Price_Trend'] > 0, 1, 0)
    
    # Calculate Medium-Term Price Trend
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Price_Trend'] = df['close'] - df['EMA_30']
    df['Medium_Term_Price_Trend_Sign'] = np.where(df['Medium_Term_Price_Trend'] > 0, 1, 0)
    
    # Calculate Long-Term Price Trend
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Price_Trend'] = df['close'] - df['EMA_50']
    df['Long_Term_Price_Trend_Sign'] = np.where(df['Long_Term_Price_Trend'] > 0, 1, 0)
    
    # Calculate Dynamic Volatility
    df['Volatility'] = df['close'].rolling(window=20).std()
    volatility_75th = df['Volatility'].quantile(0.75)
    volatility_25th = df['Volatility'].quantile(0.25)
    df['Dynamic_Volatility'] = np.where(df['Volatility'] > volatility_75th, 'high', 
                                        np.where(df['Volatility'] < volatility_25th, 'low', 'medium'))
    
    # Integrate Momentum and Relative Strength
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Relative_Strength'] = df['EMA_5'] / df['EMA_20']
    df['Relative_Strength_Sign'] = np.where(df['Relative_Strength'] > 1, 'strong', 'weak')
    
    # Combine Factors
    df['Factor'] = (df['High_Low_Spread'] * 
                    (np.where(df['Volume_Trend_Sign'] == 1, 1.5, 0.5)) * 
                    (np.where(df['Short_Term_Price_Trend_Sign'] == 1, 1.2, 0.8)) * 
                    (np.where(df['Medium_Term_Price_Trend_Sign'] == 1, 1.1, 0.9)) * 
                    (np.where(df['Long_Term_Price_Trend_Sign'] == 1, 1.3, 0.7)) * 
                    (np.where(df['Dynamic_Volatility'] == 'high', 1.4, 
                              np.where(df['Dynamic_Volatility'] == 'low', 0.6, 1))) * 
                    (np.where(df['Relative_Strength_Sign'] == 'strong', 1.5, 0.5)))
    
    # Consider Market Context
    # Assuming a market trend is provided, otherwise, it can be calculated or input
    # For simplicity, we'll assume a bullish trend
    market_trend = 'bullish'
    if market_trend == 'bullish':
        df['Factor'] *= 1.5
    elif market_trend == 'bearish':
        df['Factor'] *= 0.5
    
    return df['Factor']
```
