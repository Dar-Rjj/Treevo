```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_signal = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Calculate Volatility
    close_std_20 = df['close'].rolling(window=20).std()
    close_mean_20 = df['close'].rolling(window=20).mean()
    volatility = close_std_20 / close_mean_20
    
    # Combine High-Low Spread, Volume Trend, and Volatility
    combined_factor = high_low_spread * volatility * volume_trend_signal
    
    # Refine Moving Averages for Sharper Signals
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    ema_crossover_signal = ema_10 - ema_20
    ema_signal = np.where(ema_crossover_signal > 0, 1, -1)
    
    # Final Alpha Factor
    alpha_factor = combined_factor * ema_signal
    
    return alpha_factor
```
