```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10day_ma
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Short-Term Price Trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - ema_10
    short_term_trend_positive = short_term_price_trend > 0
    
    # Calculate Medium-Term Price Trend
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - ema_30
    medium_term_trend_positive = medium_term_price_trend > 0
    
    # Calculate Long-Term Price Trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - ema_50
    long_term_trend_positive = long_term_price_trend > 0
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_score = np.where(volatility > volatility.quantile(0.8), 2.5,
                                np.where((volatility > volatility.quantile(0.6)) & (volatility <= volatility.quantile(0.8)), 1.5,
                                         np.where((volatility > volatility.quantile(0.4)) & (volatility <= volatility.quantile(0.6)), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    momentum_factor = np.where(relative_strength > 1, 1.7, 0.3)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread = high_low_spread * np.where(volume_trend_positive, 1.5, 0.5)
    
    # Incorporate Short-Term Price Trend
    adjusted_high_low_spread *= np.where(short_term_trend_positive, 1.2, 0.8)
    
    # Incorporate Medium-Term Price Trend
    adjusted_high_low_spread *= np.where(medium_term_trend_positive, 1.1, 0.9)
    
    # Incorporate Long-Term Price Trend
    adjusted_high_low_spread *= np.where(long_term_trend_positive, 1.3, 0.7)
    
    # Incorporate Dynamic Volatility
    adjusted_high_low_spread *= volatility_score
    
    # Incorporate Momentum and Relative Strength
    adjusted_high_low_spread *= momentum_factor
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread
    
    return alpha_factor
```
