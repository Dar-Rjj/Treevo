```python
import pandas as pd
import numpy as np

def heuristics_v2(df, benchmark_index_close):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_positive = (volume_trend > 0).astype(float)

    # Calculate Price Trend
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10d_ema
    price_trend_positive = (price_trend > 0).astype(float)

    # Integrate Momentum and Relative Strength
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema / long_term_ema
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)

    # Adaptive Volatility
    close_20d_std = df['close'].rolling(window=20).std()
    close_20d_ma = df['close'].rolling(window=20).mean()
    volatility = close_20d_std / close_20d_ma
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)

    # Cross-Asset Correlation
    correlation = df['close'].rolling(window=20).corr(benchmark_index_close)
    correlation_weight = np.where(correlation > 0.5, 1.3, 0.7)

    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    alpha_factor = high_low_spread * volume_trend
    alpha_factor *= np.where(volume_trend_positive, 1.5, 0.5)
    alpha_factor *= np.where(price_trend_positive, 1.2, 0.8)
    alpha_factor *= relative_strength_weight
    alpha_factor *= volatility_weight
    alpha_factor *= correlation_weight

    return alpha_factor
```
