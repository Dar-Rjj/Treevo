```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Momentum
    df['close_momentum'] = df['close'].diff()
    
    # Calculate 10-Day Volatility
    df['volatility'] = df['close'].rolling(window=10).std()
    
    # Combine Range, Momentum, and Volatility
    df['combined_metric'] = (df['daily_range'] * df['close_momentum']) / df['volatility']
    
    # Adjust for Trading Activity
    df['combined_metric_adjusted'] = df['combined_metric'] / df['volume']
    
    # Calculate Relative Strength
    df['relative_strength'] = df['close'] / df['close'].rolling(window=50).mean()
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['combined_metric_adjusted'] * df['relative_strength']
    
    return df['alpha_factor']
```
