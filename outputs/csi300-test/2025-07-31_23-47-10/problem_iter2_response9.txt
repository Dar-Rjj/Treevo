```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10day_ma
    volume_trend_adjusted = volume_trend.apply(lambda x: 1.5 if x > 0 else 0.5)

    # Incorporate Volatility
    daily_volatility = df['close'].rolling(window=20).std()
    normalized_high_low_spread = high_low_spread / daily_volatility

    # Combine Spread and Volume Trend
    combined_spread_volume = normalized_high_low_spread * volume_trend
    adjusted_spread_volume = combined_spread_volume * volume_trend_adjusted

    # Consider Market Sentiment
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    market_sentiment = daily_return.apply(lambda x: 1 if x > 0 else -1)

    # Final Alpha Factor
    alpha_factor = adjusted_spread_volume * market_sentiment

    return alpha_factor
```
