```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    daily_volume_trend = df['volume'] - volume_ema_10
    
    # Calculate Price Momentum
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_momentum = df['close'] - close_ema_10
    
    # Determine if volume trend is positive or negative
    volume_trend_positive = (daily_volume_trend > 0).astype(int)
    volume_trend_negative = (daily_volume_trend < 0).astype(int)
    
    # Determine if price momentum is positive or negative
    price_momentum_positive = (price_momentum > 0).astype(int)
    price_momentum_negative = (price_momentum < 0).astype(int)
    
    # Combine Spread, Volume Trend, and Price Momentum
    combined_factor = high_low_spread * daily_volume_trend
    combined_factor *= 1.5 * volume_trend_positive + 0.5 * volume_trend_negative
    combined_factor *= 1.5 * price_momentum_positive + 0.5 * price_momentum_negative
    
    return combined_factor
```
