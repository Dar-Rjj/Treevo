```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']

    # Calculate Daily Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA_10']
    df['Volume_Trend_Positive'] = df['Volume_Trend'] > 0

    # Calculate Short-Term Price Trend (EMA over 10 days)
    df['Close_EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Trend'] = df['close'] - df['Close_EMA_10']
    df['Short_Term_Price_Trend_Positive'] = df['Short_Term_Price_Trend'] > 0

    # Calculate Medium-Term Price Trend (EMA over 30 days)
    df['Close_EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Price_Trend'] = df['close'] - df['Close_EMA_30']
    df['Medium_Term_Price_Trend_Positive'] = df['Medium_Term_Price_Trend'] > 0

    # Calculate Long-Term Price Trend (EMA over 50 days)
    df['Close_EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Price_Trend'] = df['close'] - df['Close_EMA_50']
    df['Long_Term_Price_Trend_Positive'] = df['Long_Term_Price_Trend'] > 0

    # Calculate Dynamic Volatility
    df['Volatility'] = df['close'].rolling(window=10).std()
    vol_percentile = df['Volatility'].rank(pct=True)
    df['Volatility_Score'] = np.where(vol_percentile > 0.8, 2.5,
                                      np.where((vol_percentile > 0.6) & (vol_percentile <= 0.8), 1.5,
                                               np.where((vol_percentile > 0.4) & (vol_percentile <= 0.6), 1.0, 0.5)))

    # Integrate Momentum and Relative Strength
    df['Short_Term_EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Long_Term_EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Relative_Strength'] = df['Short_Term_EMA_5'] / df['Long_Term_EMA_20']

    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    df['Adjusted_Spread'] = df['High_Low_Spread']
    df.loc[df['Volume_Trend_Positive'], 'Adjusted_Spread'] *= 1.5
    df.loc[~df['Volume_Trend_Positive'], 'Adjusted_Spread'] *= 0.5

    df['Adjusted_Spread'] *= np.where(df['Short_Term_Price_Trend_Positive'], 1.2, 0.8)
    df['Adjusted_Spread'] *= np.where(df['Medium_Term_Price_Trend_Positive'], 1.1, 0.9)
    df['Adjusted_Spread'] *= np.where(df['Long_Term_Price_Trend_Positive'], 1.3, 0.7)
    df['Adjusted_Spread'] *= df['Volatility_Score']
    df['Adjusted_Spread'] *= np.where(df['Relative_Strength'] > 1, 1.7, 0.3)

    # Consider Dynamic Market Context
    # Example: If market trend is bullish, increase weight of long-term trend to 1.5
    # This part is not implemented in the code below. It requires additional data and logic.
    # Similarly, sector performance is not considered in this example.

    # Final Alpha Factor
    df['Alpha_Factor'] = df['Adjusted_Spread']
    return df['Alpha_Factor']
```
