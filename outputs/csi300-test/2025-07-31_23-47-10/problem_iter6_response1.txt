```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    
    # Adjust for positive and negative trends
    volume_trend_adjusted = volume_trend.apply(lambda x: 1.5 * x if x > 0 else 0.5 * x)
    
    # Calculate Price Volatility
    returns = df['close'].pct_change()
    price_volatility = returns.rolling(window=22).std()
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].rolling(window=5).mean()
    
    # Calculate Relative Strength
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = df['close'] / ema_20
    
    # Combine High-Low Spread, Volume Trend, and Volatility
    combined_factor = (high_low_spread * volume_trend_adjusted) / (price_volatility + 1e-6)
    
    # Balance Short-Term Momentum and Relative Strength
    balanced_factor = (combined_factor + short_term_momentum) * relative_strength
    
    return balanced_factor
```
