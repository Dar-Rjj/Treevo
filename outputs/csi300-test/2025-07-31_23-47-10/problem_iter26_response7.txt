```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate an alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index (date) representing the alpha factor values.
    """
    # Calculate the percentage change in close price
    close_change = df['close'].pct_change()
    
    # Calculate the average of the open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the weighted average price using volume
    wavg_price = (df['amount'] * avg_price) / df['volume']
    
    # Calculate the standard deviation of the high and low prices over a rolling window
    std_price = df[['high', 'low']].std(axis=1).rolling(window=20).std()
    
    # Compute the factor as a combination of the above metrics
    factor = (wavg_price * close_change) / std_price
    
    return factor
```
