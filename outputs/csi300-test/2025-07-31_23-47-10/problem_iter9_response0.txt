```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = (df['volume'] - volume_ma_10)
    volume_trend_factor = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = (df['close'] - close_ema_10)
    price_trend_factor = np.where(price_trend > 0, 1.2, 0.8)
    
    # Combine Spread, Volume, and Price Trends
    combined_trends = high_low_spread * volume_trend_factor * price_trend_factor
    
    # Final Alpha Factor
    alpha_factor = pd.Series(combined_trends, index=df.index, name='alpha_factor')
    
    return alpha_factor
```
