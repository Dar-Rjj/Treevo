```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = (volume_trend > 0).astype(int)

    # Calculate Short-Term Price Trend (EMA over 10 days)
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_ema_10
    short_term_trend_positive = (short_term_trend > 0).astype(int)

    # Calculate Medium-Term Price Trend (EMA over 30 days)
    close_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_ema_30
    medium_term_trend_positive = (medium_term_trend > 0).astype(int)

    # Calculate Long-Term Price Trend (EMA over 50 days)
    close_ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_ema_50
    long_term_trend_positive = (long_term_trend > 0).astype(int)

    # Calculate Dynamic Volatility (Standard Deviation over 10 days)
    volatility = df['close'].rolling(window=10).std()
    high_volatility = (volatility > volatility.mean()).astype(int)

    # Integrate Momentum and Relative Strength
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_ema_5 / close_ema_20

    # Combine Adjusted Factors
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_positive == 1] *= 1.5
    adjusted_spread[volume_trend_positive == 0] *= 0.5

    adjusted_spread[short_term_trend_positive == 1] *= 1.2
    adjusted_spread[short_term_trend_positive == 0] *= 0.8

    adjusted_spread[medium_term_trend_positive == 1] *= 1.1
    adjusted_spread[medium_term_trend_positive == 0] *= 0.9

    adjusted_spread[long_term_trend_positive == 1] *= 1.3
    adjusted_spread[long_term_trend_positive == 0] *= 0.7

    adjusted_spread[high_volatility == 1] *= 1.4
    adjusted_spread[high_volatility == 0] *= 0.6

    adjusted_spread[relative_strength > 1] *= 1.7
    adjusted_spread[relative_strength <= 1] *= 0.3

    # Final Alpha Factor
    alpha_factor = adjusted_spread

    return alpha_factor
```
