```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Price Trend
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10d_ema
    price_trend_positive = (price_trend > 0).astype(int)
    
    # Adjust High-Low Spread based on Volume Trend
    adjusted_high_low_spread = high_low_spread * ((volume_trend_positive * 1.5) + ((1 - volume_trend_positive) * 0.5))
    
    # Incorporate Price Trend
    final_spread = adjusted_high_low_spread * ((price_trend_positive * 1.2) + ((1 - price_trend_positive) * 0.8))
    
    # Combine Trends
    alpha_factor = final_spread
    
    return alpha_factor
```
