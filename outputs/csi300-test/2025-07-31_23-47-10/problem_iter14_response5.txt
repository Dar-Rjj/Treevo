```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    ema_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_trend = df['volume'] - ema_volume_10
    volume_trend_positive = (volume_trend > 0).astype(int)
    volume_trend_negative = (volume_trend < 0).astype(int)
    
    # Calculate Price Trend
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - ema_close_20
    price_trend_positive = (price_trend > 0).astype(int)
    price_trend_negative = (price_trend < 0).astype(int)
    
    # Calculate Momentum Trend
    momentum_trend = df['close'] - df['close'].shift(10)
    momentum_trend_positive = (momentum_trend > 0).astype(int)
    momentum_trend_negative = (momentum_trend < 0).astype(int)
    
    # Calculate Volatility (Average True Range over 14 days)
    true_range = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), 
                            abs(df['low'] - df['close'].shift(1)))
    atr_14 = true_range.rolling(window=14).mean()
    volatility_high = (atr_14 > atr_14.mean()).astype(int)
    volatility_low = (atr_14 <= atr_14.mean()).astype(int)
    
    # Calculate Relative Strength
    gain = df['close'] - df['close'].shift(1)
    loss = df['close'].shift(1) - df['close']
    gain = gain.where(gain > 0, 0)
    loss = loss.where(loss > 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean().replace(0, 0.0001)  # to avoid division by zero
    rs = avg_gain / avg_loss
    rs_high = (rs > rs.mean()).astype(int)
    rs_low = (rs <= rs.mean()).astype(int)
    
    # Combine all trends
    combined_factor = high_low_spread * (volume_trend + (volume_trend_positive * 0.5) - (volume_trend_negative * 0.5))
    combined_factor = combined_factor * (price_trend_positive * 1.2 + price_trend_negative * 0.8)
    combined_factor = combined_factor * (momentum_trend_positive * 1.3 + momentum_trend_negative * 0.7)
    combined_factor = combined_factor * (volatility_high * 1.1 + volatility_low * 0.9)
    combined_factor = combined_factor * (rs_high * 1.1 + rs_low * 0.9)
    
    return combined_factor
```
