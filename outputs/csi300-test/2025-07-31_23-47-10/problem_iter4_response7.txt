```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_moving_avg
    
    # Adjust for positive and negative trends
    volume_trend_adjusted = daily_volume_trend.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine Spread and Volume Trend
    factor_values = high_low_spread * daily_volume_trend * volume_trend_adjusted
    
    return factor_values
```
