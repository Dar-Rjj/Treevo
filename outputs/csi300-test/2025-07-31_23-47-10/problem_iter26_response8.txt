```python
import pandas as pd

def heuristics_v2(df):
    """
    The function takes a pandas DataFrame with (open, high, low, close, amount, volume) prices and calculates a novel
    alpha factor that could help predict future stock returns.
    
    :param df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    :return: pandas Series representing the factor values
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the difference between the typical price and the close price
    price_diff = typical_price - df['close']
    
    # Calculate the weighted money flow using the price difference
    weighted_money_flow = money_flow * price_diff
    
    # Calculate the 10-day moving average of the weighted money flow
    ma_weighted_money_flow = weighted_money_flow.rolling(window=10).mean()
    
    # Normalize the 10-day moving average by the standard deviation to get the final factor
    factor = ma_weighted_money_flow / weighted_money_flow.rolling(window=10).std()
    
    return factor
```
