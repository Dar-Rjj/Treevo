```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    volume_trend_factor = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread * volume_trend_factor
    
    # Calculate Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Calculate Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    price_momentum = df['close'] - close_5_day_ma
    momentum_factor = np.where(price_momentum > 0, 1.2, 0.8)
    
    # Adjust for Price Momentum
    adjusted_spread = adjusted_spread * momentum_factor
    
    # Calculate Relative Strength
    close_252_day_ma = df['close'].rolling(window=252).mean()
    relative_strength = df['close'] / close_252_day_ma
    adjusted_spread = adjusted_spread * relative_strength
    
    # Calculate Market Breadth (assuming a DataFrame with all stocks' close prices is available)
    market_breadth = (df['close'] > df['close'].rolling(window=50).mean()).mean(axis=1)
    market_breadth_factor = np.where(market_breadth > 0.5, 1.1, 0.9)
    adjusted_spread = adjusted_spread * market_breadth_factor
    
    # Calculate Adaptive Moving Average
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    adaptive_ma_factor = np.where(ema_10 > ema_30, 1.2, 0.8)
    adjusted_spread = adjusted_spread * adaptive_ma_factor
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread * volatility
    
    return alpha_factor
```
