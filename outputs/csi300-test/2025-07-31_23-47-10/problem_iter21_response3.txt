```python
import pandas as pd
import numpy as np
from scipy.stats import rankdata

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma

    # Calculate Short-Term Price Trend
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_10d_ema

    # Calculate Medium-Term Price Trend
    close_30d_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_30d_ema

    # Calculate Long-Term Price Trend
    close_50d_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_50d_ema

    # Calculate Dynamic Volatility
    close_20d_std = df['close'].rolling(window=20).std()
    volatility_75th_percentile = close_20d_std.quantile(0.75)
    volatility_25th_percentile = close_20d_std.quantile(0.25)
    dynamic_volatility = np.where(close_20d_std > volatility_75th_percentile, 'high',
                                  np.where(close_20d_std < volatility_25th_percentile, 'low', 'medium'))

    # Integrate Momentum and Relative Strength
    close_5d_ema = df['close'].ewm(span=5, adjust=False).mean()
    close_20d_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_5d_ema / close_20d_ema

    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread.copy()
    adjusted_high_low_spread[volume_trend > 0] *= 1.5
    adjusted_high_low_spread[volume_trend <= 0] *= 0.5

    adjusted_short_term_trend = short_term_trend.copy()
    adjusted_short_term_trend[short_term_trend > 0] *= 1.2
    adjusted_short_term_trend[short_term_trend <= 0] *= 0.8

    adjusted_medium_term_trend = medium_term_trend.copy()
    adjusted_medium_term_trend[medium_term_trend > 0] *= 1.1
    adjusted_medium_term_trend[medium_term_trend <= 0] *= 0.9

    adjusted_long_term_trend = long_term_trend.copy()
    adjusted_long_term_trend[long_term_trend > 0] *= 1.3
    adjusted_long_term_trend[long_term_trend <= 0] *= 0.7

    adjusted_dynamic_volatility = np.where(dynamic_volatility == 'high', 1.4,
                                           np.where(dynamic_volatility == 'low', 0.6, 1.0))

    adjusted_relative_strength = np.where(relative_strength > 1, 1.5, 0.5)

    # Final Alpha Factor
    alpha_factor = (adjusted_high_low_spread * adjusted_dynamic_volatility *
                    adjusted_relative_strength * adjusted_short_term_trend *
                    adjusted_medium_term_trend * adjusted_long_term_trend)

    return pd.Series(alpha_factor, index=df.index)
```
