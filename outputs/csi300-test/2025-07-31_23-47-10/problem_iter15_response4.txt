```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    volume_trend_sign = np.sign(volume_trend)
    
    # Calculate Price Trend
    close_10_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10_day_ema
    price_trend_sign = np.sign(price_trend)
    
    # Integrate Momentum and Relative Strength
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema / long_term_ema
    
    # Adaptive Volatility
    std_close_20_days = df['close'].rolling(window=20).std()
    moving_avg_close_20_days = df['close'].rolling(window=20).mean()
    volatility = std_close_20_days / moving_avg_close_20_days
    
    # Dynamic Weight Adjustment
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    volume_trend_multiplier = np.where(volume_trend_sign > 0, 1.5, 0.5)
    price_trend_multiplier = np.where(price_trend_sign > 0, 1.2, 0.8)
    
    adjusted_spread = high_low_spread * volume_trend_multiplier * price_trend_multiplier * relative_strength_weight * volatility_weight
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    
    return alpha_factor
```
