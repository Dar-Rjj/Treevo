```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Determine if positive or negative volume trend
    volume_trend_sign = np.where(volume_trend > 0, 1, -1)
    
    # Adjust volume trend for positive and negative trends
    volume_trend_adjusted = np.where(volume_trend_sign > 0, volume_trend * 1.5, volume_trend * 0.5)
    
    # Calculate Short-Term Price Momentum (EMA over 5 days)
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    short_momentum = df['close'] - short_ema
    
    # Calculate Long-Term Price Momentum (EMA over 20 days)
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    long_momentum = df['close'] - long_ema
    
    # Combine Short- and Long-Term Momentum
    combined_momentum = 0.6 * short_momentum + 0.4 * long_momentum
    
    # Incorporate Combined Price Momentum
    combined_momentum_adjusted = np.where(combined_momentum > 0, combined_momentum * 1.2, combined_momentum * 0.8)
    
    # Combine Spread, Volume, and Combined Price Momentum
    spread_volume_momentum = high_low_spread * volume_trend_adjusted + combined_momentum_adjusted
    
    # Adaptive Volatility Adjustment
    std_dev_20 = df['close'].rolling(window=20).std()
    threshold = 0.1  # Example threshold, can be adjusted
    volatility_adjustment = np.where(std_dev_20 > threshold, -0.1, 0.1)
    
    # Adjust Final Alpha Factor
    final_alpha_factor = spread_volume_momentum + volatility_adjustment
    
    return final_alpha_factor
```
