```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the simple moving average of the close prices for a 5-day window
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the simple moving average of the volume for a 10-day window
    df['SMA_volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the price to volume ratio by dividing the 5-day SMA of close by the 10-day SMA of volume
    df['Price_to_Volume_Ratio'] = df['SMA_5'] / df['SMA_volume_10']
    
    # Calculate the percentage change in the 'Price_to_Volume_Ratio' over a 5 day period
    factor_values = df['Price_to_Volume_Ratio'].pct_change(periods=5)
    
    return factor_values.dropna()
```
