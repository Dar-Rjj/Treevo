```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_close = df['close'].rolling(window=5).mean()
    short_term_returns = (df['close'] - df['close'].shift(5)) / short_term_close

    # Calculate Long-Term Price Reversal
    long_term_close = df['close'].rolling(window=20).mean()
    long_term_returns = (df['close'] - df['close'].shift(20)) / long_term_close

    # Combine Momentum and Reversal
    combined_momentum_reversal = (short_term_returns + long_term_returns) * 0.5

    # Calculate Volume Trend
    volume_trend = (df['volume'] - df['volume'].shift(10)) / df['volume'].rolling(window=10).mean()

    # Final Alpha Factor
    final_alpha_factor = combined_momentum_reversal * volume_trend
    
    return final_alpha_factor
```
