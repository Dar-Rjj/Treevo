```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_sign = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread = high_low_spread * volume_trend_sign
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_sign = np.where(price_trend > 0, 1.2, 0.8)
    
    # Incorporate Price Trend
    adjusted_spread = adjusted_high_low_spread * price_trend_sign
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_threshold = np.percentile(volatility, 50)
    volatility_factor = np.where(volatility > volatility_threshold, 1.3, 0.7)
    
    # Incorporate Dynamic Volatility
    final_alpha_factor = adjusted_spread * volatility_factor
    
    return final_alpha_factor
```
