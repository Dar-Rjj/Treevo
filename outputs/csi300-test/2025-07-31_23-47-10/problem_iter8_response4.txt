```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10d_ma
    
    # Calculate Price Momentum
    close_12d_ema = df['close'].ewm(span=12, adjust=False).mean()
    price_momentum = df['close'] - close_12d_ema
    
    # Combine Spread, Volume Trend, and Price Momentum
    combined_factor = high_low_spread * daily_volume_trend
    
    # Adjust for Volume Trend
    combined_factor = combined_factor * np.where(daily_volume_trend > 0, 1.5, 0.5)
    
    # Adjust for Price Momentum
    combined_factor = combined_factor + np.where(price_momentum > 0, combined_factor * 0.1, -combined_factor * 0.1)
    
    # Final Adjustment
    combined_factor = combined_factor + np.where((daily_volume_trend > 0) & (price_momentum > 0), combined_factor * 0.05, 0)
    combined_factor = combined_factor + np.where((daily_volume_trend < 0) & (price_momentum < 0), -combined_factor * 0.05, 0)
    
    return pd.Series(combined_factor, index=df.index, name='heuristics_v2')
```
