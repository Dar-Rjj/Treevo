```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10d_ma
    volume_trend_direction = np.where(daily_volume_trend > 0, 'positive', 'negative')
    
    # Calculate Price Trend
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10d_ema
    price_trend_direction = np.where(price_trend > 0, 'positive', 'negative')
    
    # Adjust High-Low Spread with Volume Trend
    adjusted_high_low_spread = high_low_spread * np.where(volume_trend_direction == 'positive', 1.5, 0.5)
    
    # Incorporate Price Trend
    adjusted_high_low_spread = np.where(price_trend_direction == 'positive', 
                                        adjusted_high_low_spread * 1.2, 
                                        adjusted_high_low_spread * 0.8)
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread + daily_volume_trend + price_trend
    
    return alpha_factor
```
