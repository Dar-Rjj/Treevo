```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Short-Term Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - close_ema_10
    short_term_trend_positive = short_term_price_trend > 0
    
    # Calculate Medium-Term Price Trend
    close_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - close_ema_30
    medium_term_trend_positive = medium_term_price_trend > 0
    
    # Calculate Long-Term Price Trend
    close_ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - close_ema_50
    long_term_trend_positive = long_term_price_trend > 0
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    vol_80th = volatility.quantile(0.8)
    vol_60th = volatility.quantile(0.6)
    vol_40th = volatility.quantile(0.4)
    volatility_score = np.where(volatility > vol_80th, 2.5,
                                np.where((volatility > vol_60th) & (volatility <= vol_80th), 1.5,
                                         np.where((volatility > vol_40th) & (volatility <= vol_60th), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_ema_5 / close_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread * np.where(volume_trend_positive, 1.5, 0.5)
    adjusted_spread *= np.where(short_term_trend_positive, 1.2, 0.8)
    adjusted_spread *= np.where(medium_term_trend_positive, 1.1, 0.9)
    adjusted_spread *= np.where(long_term_trend_positive, 1.3, 0.7)
    adjusted_spread *= volatility_score
    adjusted_spread *= np.where(relative_strength > 1, 1.7, 0.3)
    
    # Consider Dynamic Market Context (Placeholder for now, can be expanded based on market data)
    market_trend_bullish = True  # Placeholder
    sector_outperforming = True  # Placeholder
    if market_trend_bullish:
        adjusted_spread *= 1.5
    else:
        adjusted_spread *= 0.5
    if sector_outperforming:
        adjusted_spread *= 1.2
    else:
        adjusted_spread *= 0.8
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    
    return alpha_factor
```
