```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'].rolling(window=10).mean() - df['volume']
    volume_trend_positive = volume_trend > 0
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    short_term_trend_positive = short_term_trend > 0
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30
    medium_term_trend_positive = medium_term_trend > 0
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    long_term_trend_positive = long_term_trend > 0
    
    # Calculate Dynamic Volatility (standard deviation over 10 days)
    volatility = df['close'].rolling(window=10).std()
    high_volatility = volatility > volatility.quantile(0.5)
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    strong_momentum = relative_strength > 1
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    combined_factor = high_low_spread.copy()
    
    # Adjust High-Low Spread by Volume Trend
    combined_factor[volume_trend_positive] *= 1.5
    combined_factor[~volume_trend_positive] *= 0.5
    
    # Incorporate Short-Term Price Trend
    combined_factor[short_term_trend_positive] *= 1.2
    combined_factor[~short_term_trend_positive] *= 0.8
    
    # Incorporate Medium-Term Price Trend
    combined_factor[medium_term_trend_positive] *= 1.1
    combined_factor[~medium_term_trend_positive] *= 0.9
    
    # Incorporate Long-Term Price Trend
    combined_factor[long_term_trend_positive] *= 1.3
    combined_factor[~long_term_trend_positive] *= 0.7
    
    # Incorporate Dynamic Volatility
    combined_factor[high_volatility] *= 1.4
    combined_factor[~high_volatility] *= 0.6
    
    # Incorporate Momentum and Relative Strength
    combined_factor[strong_momentum] *= 1.5
    combined_factor[~strong_momentum] *= 0.5
    
    return combined_factor
```
