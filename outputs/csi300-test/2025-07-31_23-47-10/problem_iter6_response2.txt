```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10_day_ma
    
    # Determine if positive or negative trend
    volume_trend_positive = daily_volume_trend > 0
    volume_trend_negative = daily_volume_trend < 0
    
    # Combine Spread and Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_positive] *= 1.5
    adjusted_spread[volume_trend_negative] *= 0.5
    
    return adjusted_spread
```
