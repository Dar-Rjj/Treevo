```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_ema_10
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    close_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_ema_30
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    close_ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_ema_50
    
    # Calculate Dynamic Volatility (Standard Deviation over 10 days)
    volatility = df['close'].rolling(window=10).std()
    volatility_score = np.where(volatility > volatility.quantile(0.8), 2.5,
                                np.where((volatility > volatility.quantile(0.6)) & (volatility <= volatility.quantile(0.8)), 1.5,
                                         np.where((volatility > volatility.quantile(0.4)) & (volatility <= volatility.quantile(0.6)), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    short_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_ema_5 / long_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread * np.where(volume_trend > 0, 1.5, 0.5)
    adjusted_short_term_trend = np.where(short_term_trend > 0, 1.2, 0.8)
    adjusted_medium_term_trend = np.where(medium_term_trend > 0, 1.1, 0.9)
    adjusted_long_term_trend = np.where(long_term_trend > 0, 1.3, 0.7)
    adjusted_volatility = volatility_score
    adjusted_relative_strength = np.where(relative_strength > 1, 1.7, 0.3)
    
    # Consider Dynamic Market Context
    # For simplicity, let's assume the market trend is bullish and the sector is outperforming
    market_trend_weight = 1.5
    sector_performance_weight = 1.2
    
    # Final Alpha Factor
    alpha_factor = (adjusted_high_low_spread * 
                    adjusted_short_term_trend * 
                    adjusted_medium_term_trend * 
                    (adjusted_long_term_trend * market_trend_weight) * 
                    adjusted_volatility * 
                    (adjusted_relative_strength * sector_performance_weight))
    
    return alpha_factor
```
