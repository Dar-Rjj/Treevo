```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    volume_trend = df['volume'] - volume_ema
    
    # Determine if positive or negative volume trend
    volume_trend_factor = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Calculate Price Trend
    close_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema
    
    # Determine if positive or negative price trend
    price_trend_factor = np.where(price_trend > 0, 1.2, 0.8)
    
    # Combine Spread, Volume, and Price Trends
    adjusted_high_low_spread = high_low_spread * volume_trend
    adjusted_high_low_spread = adjusted_high_low_spread * volume_trend_factor
    final_factor = adjusted_high_low_spread * price_trend_factor
    
    return final_factor
```
