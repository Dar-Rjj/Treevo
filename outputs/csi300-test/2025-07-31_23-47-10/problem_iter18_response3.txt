```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Price Trends
    close_5d_ema = df['close'].ewm(span=5, adjust=False).mean()
    short_term_price_trend = df['close'] - close_5d_ema
    short_term_price_trend_positive = (short_term_price_trend > 0).astype(int)
    
    close_20d_ema = df['close'].ewm(span=20, adjust=False).mean()
    long_term_price_trend = df['close'] - close_20d_ema
    long_term_price_trend_positive = (long_term_price_trend > 0).astype(int)
    
    # Integrate Momentum and Relative Strength
    relative_strength = close_5d_ema / close_20d_ema
    relative_strength_above_1 = (relative_strength > 1).astype(int)
    
    # Calculate Open-Price Momentum
    open_price_momentum = df['close'] - df['open']
    open_price_momentum_positive = (open_price_momentum > 0).astype(int)
    
    # Adaptive Volatility
    close_20d_std = df['close'].rolling(window=20).std()
    close_20d_ma = df['close'].rolling(window=20).mean()
    volatility = close_20d_std / close_20d_ma
    
    # Dynamic Weight Adjustment
    relative_strength_weight = relative_strength_above_1 * 1.5 + (1 - relative_strength_above_1) * 0.5
    volatility_weight = (volatility > 1).astype(int) * 1.2 + (volatility <= 1).astype(int) * 0.8
    open_momentum_weight = open_price_momentum_positive * 1.1 + (1 - open_price_momentum_positive) * 0.9
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_volume = high_low_spread * (volume_trend_positive * 1.5 + (1 - volume_trend_positive) * 0.5)
    short_term_weighted = short_term_price_trend * (short_term_price_trend_positive * 1.2 + (1 - short_term_price_trend_positive) * 0.8)
    long_term_weighted = long_term_price_trend * (long_term_price_trend_positive * 1.3 + (1 - long_term_price_trend_positive) * 0.7)
    
    # Final Alpha Factor
    alpha_factor = (spread_volume + short_term_weighted + long_term_weighted) * relative_strength_weight * volatility_weight * open_momentum_weight
    
    return alpha_factor
```
