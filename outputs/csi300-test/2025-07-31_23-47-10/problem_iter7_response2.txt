```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    
    # Determine if positive or negative volume trend
    volume_trend_adjustment = 1.5 * (daily_volume_trend > 0) + 0.5 * (daily_volume_trend <= 0)
    
    # Calculate Price Momentum
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    price_momentum = df['close'] - ema_12
    
    # Adjust for positive and negative momentum
    momentum_adjustment = 1.1 * (price_momentum > 0) + 0.9 * (price_momentum <= 0)
    
    # Combine Spread, Volume Trend, and Price Momentum
    factor_value = (high_low_spread * volume_trend_adjustment) * momentum_adjustment
    
    return factor_value
```
