```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma
    
    # Adjust for positive and negative trends
    volume_trend_adjusted = volume_trend.apply(lambda x: x * 1.5 if x > 0 else x * 0.5)
    
    # Combine Spread and Volume Trend
    factor_values = high_low_spread * volume_trend_adjusted
    
    return factor_values
```
