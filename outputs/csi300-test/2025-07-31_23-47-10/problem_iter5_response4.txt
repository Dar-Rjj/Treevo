```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Determine if trend is positive or negative
    volume_trend_sign = (volume_trend > 0).astype(int)
    
    # Combine Spread and Volume Trend
    combined_factor = high_low_spread * volume_trend
    combined_factor_adjusted = combined_factor * (1.5 * volume_trend_sign + 0.5 * (1 - volume_trend_sign))
    
    return combined_factor_adjusted
```
