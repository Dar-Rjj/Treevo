```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    volume_trend_direction = (daily_volume_trend > 0).astype(int) * 3 - 1  # 2 for positive, -1 for negative
    
    # Incorporate Price Momentum
    close_pct_change_5 = df['close'].pct_change(periods=5)
    momentum_adjustment = (close_pct_change_5 > 0).astype(float) * 1.2 + (close_pct_change_5 <= 0).astype(float) * 0.8
    price_momentum_factor = close_pct_change_5 * momentum_adjustment
    
    # Combine High-Low Spread and Volume Trend
    combined_factor = high_low_spread * volume_trend_direction
    combined_factor = combined_factor * momentum_adjustment
    
    # Incorporate Volatility Adjustment
    volatility = df['close'].rolling(window=20).std()
    volatility_threshold = 0.02  # 2% threshold
    volatility_adjustment = (volatility > volatility_threshold).astype(float) * 1.3 + (volatility <= volatility_threshold).astype(float) * 0.7
    final_factor = combined_factor * volatility_adjustment
    
    return final_factor
```
