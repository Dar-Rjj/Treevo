```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    volume_trend = volume_ema_10 - df['volume']
    
    # Determine if positive or negative volume trend
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = close_ema_10 - df['close']
    
    # Determine if positive or negative price trend
    price_trend_positive = (price_trend > 0).astype(int)
    
    # Calculate Momentum Trend
    momentum_trend = df['close'] - df['close'].shift(10)
    
    # Determine if positive or negative momentum trend
    momentum_trend_positive = (momentum_trend > 0).astype(int)
    
    # Combine Spread, Volume, Price, and Momentum Trends
    adjusted_spread = high_low_spread * (volume_trend + 1)  # Add 1 to avoid zero division
    
    # Adjust for positive and negative trends
    adjusted_spread *= (volume_trend_positive * 1.5 + (1 - volume_trend_positive) * 0.5)
    
    # Incorporate Price Trend
    adjusted_spread *= (price_trend_positive * 1.2 + (1 - price_trend_positive) * 0.8)
    
    # Incorporate Momentum Trend
    adjusted_spread *= (momentum_trend_positive * 1.3 + (1 - momentum_trend_positive) * 0.7)
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    
    return alpha_factor
```
