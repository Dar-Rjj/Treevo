```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_factor = (volume_trend > 0).astype(int) * 1.5 + (volume_trend < 0).astype(int) * 0.5
    
    # Calculate Short-Term Price Trend
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - short_term_ema
    short_term_trend_factor = (short_term_trend > 0).astype(int) * 1.2 + (short_term_trend < 0).astype(int) * 0.8
    
    # Calculate Medium-Term Price Trend
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - medium_term_ema
    medium_term_trend_factor = (medium_term_trend > 0).astype(int) * 1.15 + (medium_term_trend < 0).astype(int) * 0.85
    
    # Calculate Long-Term Price Trend
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - long_term_ema
    long_term_trend_factor = (long_term_trend > 0).astype(int) * 1.3 + (long_term_trend < 0).astype(int) * 0.7
    
    # Calculate Dynamic Volatility
    volatility_10d = df['close'].rolling(window=10).std()
    volatility_75th = volatility_10d.quantile(0.75)
    volatility_25th = volatility_10d.quantile(0.25)
    volatility_factor = (volatility_10d > volatility_75th).astype(int) * 1.4 + (volatility_10d < volatility_25th).astype(int) * 0.6
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5d = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5d / long_term_ema_20d
    momentum_factor = (relative_strength > 1).astype(int) * 1.6 + (relative_strength <= 1).astype(int) * 0.4
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread * volume_trend_factor * short_term_trend_factor * medium_term_trend_factor * long_term_trend_factor * volatility_factor * momentum_factor
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread
    
    return alpha_factor
```
