```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Short-Term Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - close_ema_10
    price_trend_positive = short_term_price_trend > 0
    
    # Adjust High-Low Spread based on Volume Trend
    adjusted_high_low_spread = high_low_spread.copy()
    adjusted_high_low_spread[volume_trend_positive] *= 1.5
    adjusted_high_low_spread[~volume_trend_positive] *= 0.5
    
    # Incorporate Short-Term Price Trend
    alpha_factor = adjusted_high_low_spread.copy()
    alpha_factor[price_trend_positive] *= 1.2
    alpha_factor[~price_trend_positive] *= 0.8
    
    return alpha_factor
```
