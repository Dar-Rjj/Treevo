```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Calculate Short-Term Price Trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    
    # Calculate Medium-Term Price Trend
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30
    
    # Calculate Long-Term Price Trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_score = np.where(volatility > volatility.quantile(0.8), 2.5,
                                np.where((volatility > volatility.quantile(0.6)) & (volatility <= volatility.quantile(0.8)), 1.5,
                                         np.where((volatility > volatility.quantile(0.4)) & (volatility <= volatility.quantile(0.6)), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread = high_low_spread * np.where(volume_trend > 0, 1.5, 0.5)
    
    # Incorporate Short-Term Price Trend
    adjusted_short_term_trend = np.where(short_term_trend > 0, 1.2, 0.8) * short_term_trend
    
    # Incorporate Medium-Term Price Trend
    adjusted_medium_term_trend = np.where(medium_term_trend > 0, 1.1, 0.9) * medium_term_trend
    
    # Incorporate Long-Term Price Trend
    adjusted_long_term_trend = np.where(long_term_trend > 0, 1.3, 0.7) * long_term_trend
    
    # Incorporate Dynamic Volatility
    adjusted_volatility = volatility_score * volatility
    
    # Incorporate Momentum and Relative Strength
    adjusted_relative_strength = np.where(relative_strength > 1, 1.7, 0.3) * relative_strength
    
    # Combine all factors
    alpha_factor = (adjusted_high_low_spread +
                    adjusted_short_term_trend +
                    adjusted_medium_term_trend +
                    adjusted_long_term_trend +
                    adjusted_volatility +
                    adjusted_relative_strength)
    
    return alpha_factor
```
