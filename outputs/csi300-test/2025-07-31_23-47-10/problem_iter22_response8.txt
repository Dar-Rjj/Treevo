```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    df['Volume_MA10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA10']
    df['Volume_Trend_Sign'] = np.where(df['Volume_Trend'] > 0, 'Positive', 'Negative')
    
    # Calculate Short-Term Price Trend
    df['Close_EMA10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Trend'] = df['close'] - df['Close_EMA10']
    df['Short_Term_Price_Trend_Sign'] = np.where(df['Short_Term_Price_Trend'] > 0, 'Positive', 'Negative')
    
    # Calculate Medium-Term Price Trend
    df['Close_EMA30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Price_Trend'] = df['close'] - df['Close_EMA30']
    df['Medium_Term_Price_Trend_Sign'] = np.where(df['Medium_Term_Price_Trend'] > 0, 'Positive', 'Negative')
    
    # Calculate Long-Term Price Trend
    df['Close_EMA50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Price_Trend'] = df['close'] - df['Close_EMA50']
    df['Long_Term_Price_Trend_Sign'] = np.where(df['Long_Term_Price_Trend'] > 0, 'Positive', 'Negative')
    
    # Calculate Dynamic Volatility
    df['Volatility'] = df['close'].rolling(window=10).std()
    quantiles = df['Volatility'].quantile([0.4, 0.6, 0.8])
    df['Volatility_Score'] = np.where(
        df['Volatility'] > quantiles[0.8], 2.5,
        np.where(
            (df['Volatility'] > quantiles[0.6]) & (df['Volatility'] <= quantiles[0.8]), 1.5,
            np.where(
                (df['Volatility'] > quantiles[0.4]) & (df['Volatility'] <= quantiles[0.6]), 1.0, 0.5
            )
        )
    )
    
    # Integrate Momentum and Relative Strength
    df['Close_EMA5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Close_EMA20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Relative_Strength'] = df['Close_EMA5'] / df['Close_EMA20']
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    df['Adjusted_Spread'] = df['High-Low_Spread'] * np.where(df['Volume_Trend_Sign'] == 'Positive', 1.5, 0.5)
    df['Adjusted_Short_Term_Trend'] = np.where(df['Short_Term_Price_Trend_Sign'] == 'Positive', 1.2, 0.8)
    df['Adjusted_Medium_Term_Trend'] = np.where(df['Medium_Term_Price_Trend_Sign'] == 'Positive', 1.1, 0.9)
    df['Adjusted_Long_Term_Trend'] = np.where(df['Long_Term_Price_Trend_Sign'] == 'Positive', 1.3, 0.7)
    df['Adjusted_Relative_Strength'] = np.where(df['Relative_Strength'] > 1, 1.7, 0.3)
    
    # Final Alpha Factor
    df['Alpha_Factor'] = (
        df['Adjusted_Spread'] * 
        df['Adjusted_Short_Term_Trend'] * 
        df['Adjusted_Medium_Term_Trend'] * 
        df['Adjusted_Long_Term_Trend'] * 
        df['Volatility_Score'] * 
        df['Adjusted_Relative_Strength']
    )
    
    return df['Alpha_Factor']
```
