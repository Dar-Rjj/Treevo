```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Calculate Short-Term Price Trend (EMA 10 days)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - short_term_ema
    
    # Calculate Medium-Term Price Trend (EMA 30 days)
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - medium_term_ema
    
    # Calculate Long-Term Price Trend (EMA 50 days)
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - long_term_ema
    
    # Calculate Dynamic Volatility (Standard Deviation 10 days)
    volatility = df['close'].rolling(window=10).std()
    is_high_volatility = volatility > volatility.quantile(0.5)
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend > 0] *= 1.5
    adjusted_spread[volume_trend <= 0] *= 0.5
    
    # Incorporate Short-Term Price Trend
    if short_term_trend.iloc[-1] > 0:
        adjusted_spread *= 1.2
    else:
        adjusted_spread *= 0.8
    
    # Incorporate Medium-Term Price Trend
    if medium_term_trend.iloc[-1] > 0:
        adjusted_spread *= 1.1
    else:
        adjusted_spread *= 0.9
    
    # Incorporate Long-Term Price Trend
    if long_term_trend.iloc[-1] > 0:
        adjusted_spread *= 1.3
    else:
        adjusted_spread *= 0.7
    
    # Incorporate Dynamic Volatility
    if is_high_volatility.iloc[-1]:
        adjusted_spread *= 1.4
    else:
        adjusted_spread *= 0.6
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread * relative_strength
    
    return alpha_factor
```
