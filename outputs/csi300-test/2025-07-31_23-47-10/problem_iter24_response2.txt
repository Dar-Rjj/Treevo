import pandas as pd

def heuristics_v2(df):
    # Calculate the average of (open, high, low, close) prices
    df['avg_price'] = df[['open', 'high', 'low', 'close']].mean(axis=1)
    
    # Compute the change in average price and volume over the last 5 days
    df['avg_price_change'] = df['avg_price'].pct_change(periods=5)
    df['volume_change'] = df['volume'].pct_change(periods=5)
    
    # Calculate the relative strength index (RSI) for the average price
    delta = df['avg_price'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Generate a factor based on RSI and volume change
    df['alpha_factor'] = (df['rsi'] - 50) * df['volume_change']
    
    return df['alpha_factor']
