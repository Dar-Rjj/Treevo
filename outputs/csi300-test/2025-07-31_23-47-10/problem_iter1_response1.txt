```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20):
    # Calculate High-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Log Return
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Momentum Indicator (sum of log returns over past N days)
    df['momentum_indicator'] = df['log_return'].rolling(window=N).sum()
    
    # Weight by High-Low Range
    df['alpha_factor'] = df['high_low_range'] * df['momentum_indicator']
    
    return df['alpha_factor']
```
