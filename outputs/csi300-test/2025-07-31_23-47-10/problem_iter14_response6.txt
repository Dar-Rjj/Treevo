import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = volume_trend > 0
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_positive = price_trend > 0
    
    # Integrate Momentum and Relative Strength
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema / long_term_ema
    
    # Adaptive Volatility
    volatility = df['close'].rolling(window=20).std() / df['close'].rolling(window=20).mean()
    
    # Dynamic Weight Adjustment
    rs_weight = np.where(relative_strength > 1, 1.5, 0.5)
    vol_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_weighted = high_low_spread * (np.where(volume_trend_positive, 1.5, 0.5))
    trend_weighted = spread_weighted * (np.where(price_trend_positive, 1.2, 0.8))
    final_factor = trend_weighted * rs_weight * vol_weight
    
    return pd.Series(final_factor, index=df.index)
