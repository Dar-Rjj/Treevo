```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    ma_20 = df['close'].rolling(window=20).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    long_term_momentum = ma_20 - ma_50
    
    # Calculate Short-Term Price Reversal
    daily_price_change = df['close'] - df['open']
    sum_5_day_price_change = daily_price_change.rolling(window=5).sum()
    high_5_days = df['high'].rolling(window=5).max()
    low_5_days = df['low'].rolling(window=5).min()
    high_low_range = high_5_days - low_5_days
    short_term_reversal_signal = (sum_5_day_price_change / high_low_range) * df['volume']

    # Combine Long-Term and Short-Term Signals
    signal_product = long_term_momentum * short_term_reversal_signal
    final_score = long_term_momentum + short_term_reversal_signal  # Sum of Long-Term and Short-Term Signal Products

    return final_score
```
