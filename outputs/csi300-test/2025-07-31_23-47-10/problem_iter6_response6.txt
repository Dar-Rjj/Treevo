```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Adjust for positive and negative volume trends
    volume_trend_adjusted = volume_trend.apply(lambda x: 1.5 if x > 0 else (0.5 if x < 0 else 1))
    
    # Calculate Price Trend
    close_ma_10 = df['close'].rolling(window=10).mean()
    price_trend = df['close'] - close_ma_10
    
    # Adjust for positive and negative price trends
    price_trend_adjusted = price_trend.apply(lambda x: 1.2 if x > 0 else (0.8 if x < 0 else 1))
    
    # Combine Features
    combined_factor = (high_low_spread * volume_trend) * volume_trend_adjusted * price_trend_adjusted
    
    return combined_factor
```
