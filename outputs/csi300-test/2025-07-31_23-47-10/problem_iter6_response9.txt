```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Determine if positive or negative trend
    volume_trend_direction = (volume_trend > 0).astype(float)
    
    # Calculate Daily Volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Combine Spread, Volume Trend, and Volatility
    combined_spread_volume = high_low_spread * volume_trend
    combined_spread_volume_trend_adjusted = combined_spread_volume * (1.5 * volume_trend_direction + 0.5 * (1 - volume_trend_direction))
    
    # Incorporate Volatility
    volatility_threshold = volatility.median()
    combined_spread_volume_volatility = combined_spread_volume_trend_adjusted * (1.2 * (volatility > volatility_threshold) + 0.8 * (volatility <= volatility_threshold))
    
    # Incorporate Additional Price Levels
    open_close_spread = df['open'] - df['close']
    average_spread = (high_low_spread + open_close_spread) / 2
    combined_average_spread_volume = average_spread * volume_trend
    combined_average_spread_volume_trend_adjusted = combined_average_spread_volume * (1.5 * volume_trend_direction + 0.5 * (1 - volume_trend_direction))
    
    # Final Adjustment
    final_factor = combined_spread_volume_volatility + combined_average_spread_volume_trend_adjusted
    final_factor = final_factor * (1.5 * volume_trend_direction + 0.5 * (1 - volume_trend_direction))
    
    return final_factor
```
