```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA_10']
    
    # Calculate Short-Term Price Trend
    df['Close_EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Trend'] = df['close'] - df['Close_EMA_10']
    
    # Calculate Medium-Term Price Trend
    df['Close_EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Trend'] = df['close'] - df['Close_EMA_30']
    
    # Calculate Long-Term Price Trend
    df['Close_EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Trend'] = df['close'] - df['Close_EMA_50']
    
    # Calculate Dynamic Volatility
    df['Volatility'] = df['close'].rolling(window=10).std()
    vol_percentile = df['Volatility'].rank(pct=True)
    df['Volatility_Score'] = np.where(vol_percentile > 0.8, 2.5,
                                       np.where((vol_percentile > 0.6) & (vol_percentile <= 0.8), 1.5,
                                                np.where((vol_percentile > 0.4) & (vol_percentile <= 0.6), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    df['Close_EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Close_EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Relative_Strength'] = df['Close_EMA_5'] / df['Close_EMA_20']
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    df['Adjusted_High_Low_Spread'] = df['High_Low_Spread'] * np.where(df['Volume_Trend'] > 0, 1.5, 0.5)
    df['Adjusted_Short_Term_Trend'] = df['Adjusted_High_Low_Spread'] * np.where(df['Short_Term_Trend'] > 0, 1.2, 0.8)
    df['Adjusted_Medium_Term_Trend'] = df['Adjusted_Short_Term_Trend'] * np.where(df['Medium_Term_Trend'] > 0, 1.1, 0.9)
    df['Adjusted_Long_Term_Trend'] = df['Adjusted_Medium_Term_Trend'] * np.where(df['Long_Term_Trend'] > 0, 1.3, 0.7)
    df['Adjusted_Volatility'] = df['Adjusted_Long_Term_Trend'] * df['Volatility_Score']
    df['Adjusted_Relative_Strength'] = df['Adjusted_Volatility'] * np.where(df['Relative_Strength'] > 1, 1.7, 0.3)
    
    # Consider Dynamic Market Context (assuming we have a column 'market_trend' and 'sector_performance')
    if 'market_trend' in df.columns and 'sector_performance' in df.columns:
        df['Final_Alpha_Factor'] = df['Adjusted_Relative_Strength'] * \
                                  np.where(df['market_trend'] == 'bullish', 1.5, 0.5) * \
                                  np.where(df['sector_performance'] == 'outperforming', 1.2, 0.8)
    else:
        df['Final_Alpha_Factor'] = df['Adjusted_Relative_Strength']
    
    return df['Final_Alpha_Factor']
```
