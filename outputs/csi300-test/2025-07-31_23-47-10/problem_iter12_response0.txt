```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=10).mean()
    volume_trend_positive = volume_trend > 0
    
    # Calculate Price Trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_10
    price_trend_positive = price_trend > 0
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    momentum_positive = momentum > 0
    
    # Calculate Volatility using Average True Range (ATR)
    true_range = np.maximum(np.maximum(df['high'] - df['low'], 
                                       abs(df['high'] - df['close'].shift(1))), 
                            abs(df['low'] - df['close'].shift(1)))
    atr_10 = true_range.rolling(window=10).mean()
    volatility_high = atr_10 > atr_10.mean()
    
    # Combine Spread, Volume, Price Trends, Momentum, and Volatility
    adjusted_spread = high_low_spread * np.where(volume_trend_positive, 1.5, 0.5)
    adjusted_spread = adjusted_spread * np.where(price_trend_positive, 1.2, 0.8)
    
    # Adjust for Momentum
    momentum_weight = 1.3 if momentum_positive.all() else 0.7
    if volatility_high.any():
        momentum_weight += 0.1
    else:
        momentum_weight -= 0.1
    adjusted_spread = adjusted_spread * np.where(momentum_positive, momentum_weight, 1 / momentum_weight)
    
    return adjusted_spread
```
