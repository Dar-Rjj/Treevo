```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_sign = volume_trend.apply(lambda x: 1 if x > 0 else -1)
    
    # Calculate Volatility
    close_price_volatility = df['close'].rolling(window=20).std()
    normalized_high_low_spread = high_low_spread / close_price_volatility
    
    # Calculate Market Sentiment
    close_price_ma_5 = df['close'].rolling(window=5).mean()
    market_sentiment = (df['close'] > close_price_ma_5).astype(int) * 1.2 + (df['close'] <= close_price_ma_5).astype(int) * 0.8
    
    # Combine Adjusted Spread, Volume Trend, and Market Sentiment
    adjusted_spread = normalized_high_low_spread * (volume_trend_sign * 1.5).apply(lambda x: 1.5 if x > 0 else 0.5)
    final_factor = adjusted_spread * market_sentiment
    
    return final_factor
```
