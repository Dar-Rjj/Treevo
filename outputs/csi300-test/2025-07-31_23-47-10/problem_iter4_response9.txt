```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_adjusted = volume_trend.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Calculate Price Volatility
    daily_returns = df['close'].pct_change()
    price_volatility = daily_returns.rolling(window=20).std()
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].rolling(window=5).mean()
    
    # Combine High-Low Spread, Volume Trend, and Volatility
    combined_factor = (high_low_spread * volume_trend * volume_trend_adjusted) / price_volatility
    
    # Balance Short-Term Momentum
    balanced_factor = combined_factor + (short_term_momentum * 0.5)
    
    # Final Alpha Factor
    final_alpha_factor = balanced_factor
    
    return final_alpha_factor
```
