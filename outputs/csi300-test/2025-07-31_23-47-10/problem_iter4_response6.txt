```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_returns = (df['close'] - df['close'].shift(5)) / df['close'].rolling(window=5).mean()
    
    # Calculate Long-Term Price Reversal
    long_term_returns = (df['close'] - df['close'].shift(20)) / df['close'].rolling(window=20).mean()
    
    # Combine Momentum and Reversal
    combined_momentum_reversal = 0.5 * (short_term_returns + long_term_returns)
    
    # Calculate Volume Trend
    volume_change = (df['volume'] - df['volume'].shift(10)) / df['volume'].rolling(window=10).mean()
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_reversal * volume_change
    
    return final_alpha_factor
```
