```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    vol_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - vol_10_day_ma
    
    # Determine if positive or negative trend
    volume_trend_direction = (volume_trend > 0).astype(int)
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    short_term_trend_direction = (short_term_trend > 0).astype(int)
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30
    medium_term_trend_direction = (medium_term_trend > 0).astype(int)
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    long_term_trend_direction = (long_term_trend > 0).astype(int)
    
    # Calculate Dynamic Volatility (standard deviation over 10 days)
    volatility = df['close'].rolling(window=10).std()
    volatility_direction = (volatility > volatility.median()).astype(int)
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread * (1.5 * volume_trend_direction + 0.5 * (1 - volume_trend_direction))
    
    # Incorporate Short-Term Price Trend
    adjusted_spread *= (1.2 * short_term_trend_direction + 0.8 * (1 - short_term_trend_direction))
    
    # Incorporate Medium-Term Price Trend
    adjusted_spread *= (1.1 * medium_term_trend_direction + 0.9 * (1 - medium_term_trend_direction))
    
    # Incorporate Long-Term Price Trend
    adjusted_spread *= (1.3 * long_term_trend_direction + 0.7 * (1 - long_term_trend_direction))
    
    # Incorporate Dynamic Volatility
    adjusted_spread *= (1.4 * volatility_direction + 0.6 * (1 - volatility_direction))
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread + volume_trend + short_term_trend + medium_term_trend + long_term_trend + volatility
    return alpha_factor
```
