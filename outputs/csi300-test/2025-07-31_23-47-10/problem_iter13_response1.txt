```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Short-Term Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - close_ema_10
    short_term_price_trend_positive = (short_term_price_trend > 0).astype(int)
    
    # Calculate Long-Term Price Trend
    close_ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - close_ema_50
    long_term_price_trend_positive = (long_term_price_trend > 0).astype(int)
    
    # Combine Spread, Volume, and Price Trends
    spread_volume_trend = high_low_spread * volume_trend
    adjusted_spread_volume_trend = (
        spread_volume_trend * (volume_trend_positive * 1.5 + (1 - volume_trend_positive) * 0.5)
    )
    
    adjusted_short_term_trend = (
        adjusted_spread_volume_trend * (short_term_price_trend_positive * 1.2 + (1 - short_term_price_trend_positive) * 0.8)
    )
    
    final_alpha_factor = (
        adjusted_short_term_trend * (long_term_price_trend_positive * 1.3 + (1 - long_term_price_trend_positive) * 0.7)
    )
    
    return final_alpha_factor
```
