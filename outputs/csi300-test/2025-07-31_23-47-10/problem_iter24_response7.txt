```python
import pandas as pd

def heuristics_v2(df):
    """
    This function takes a pandas DataFrame with columns (open, high, low, close, amount, volume)
    and computes a custom alpha factor based on the following heuristics:
    1. Calculate the daily return.
    2. Calculate the daily volatility.
    3. Compute the ratio of the daily return to the daily volatility.
    4. Normalize the ratio by subtracting the mean and dividing by the standard deviation.
    5. Return the normalized ratio as the factor value.
    
    Parameters:
    df (pd.DataFrame): DataFrame with market data, indexed by date.
    
    Returns:
    pd.Series: Factor values, indexed by date.
    """
    # Calculate daily return
    df['return'] = df['close'].pct_change()
    
    # Calculate daily volatility
    df['volatility'] = df[['open', 'high', 'low', 'close']].std(axis=1)
    
    # Compute the ratio of the daily return to the daily volatility
    df['ratio'] = df['return'] / df['volatility']
    
    # Normalize the ratio
    df['factor'] = (df['ratio'] - df['ratio'].mean()) / df['ratio'].std()
    
    return df['factor']
```
