```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Log Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Short-Term and Long-Term Exponential Moving Average of Log Returns
    df['short_term_ema'] = df['log_returns'].ewm(span=10, adjust=False).mean()
    df['long_term_ema'] = df['log_returns'].ewm(span=25, adjust=False).mean()
    
    # Determine Adaptive Momentum
    df['adaptive_momentum'] = np.where(df['short_term_ema'] > df['long_term_ema'], 
                                       df['short_term_ema'] - df['long_term_ema'], 0)
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['log_returns'].pow(2).rolling(window=20).mean()
    
    # Determine Trend Strength
    df['trend_strength'] = np.where(df['short_term_ema'] > df['long_term_ema'], 1, 0)
    
    # Combine Adaptive Momentum, Trend Strength, and Volatility
    df['factor'] = (df['adaptive_momentum'] * df['trend_strength']) / df['realized_volatility']
    
    return df['factor']
```
