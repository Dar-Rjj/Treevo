```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the price change from the previous day
    df['price_change'] = df['close'].pct_change()
    
    # Calculate the average true range (ATR) over 14 days
    df['h-l'] = abs(df['high'] - df['low'])
    df['h-yc'] = abs(df['high'] - df['close'].shift())
    df['l-yc'] = abs(df['low'] - df['close'].shift())
    df['tr'] = df[['h-l', 'h-yc', 'l-yc']].max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the relative strength index (RSI) over 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate a custom volume-weighted close
    df['vwc'] = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Combine the factors to create a novel alpha factor
    df['alpha_factor'] = (df['price_change'] * df['atr_14']) / (df['rsi_14'] + 1) + df['vwc'].pct_change()
    
    return df['alpha_factor']
```
