```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - volume_10d_ma
    
    # Determine if the trend is positive or negative
    volume_trend_sign = (volume_trend > 0).astype(int)
    
    # Combine Spread and Volume Trend
    combined_factor = high_low_spread * volume_trend
    
    # Adjust for positive and negative trends
    adjustment_factor = 1.5 * volume_trend_sign + 0.5 * (1 - volume_trend_sign)
    adjusted_factor = combined_factor * adjustment_factor
    
    return adjusted_factor
```
