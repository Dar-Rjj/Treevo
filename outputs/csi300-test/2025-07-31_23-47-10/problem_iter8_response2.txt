```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ema_10 = df['volume'].ewm(span=10, adjust=False).mean()
    daily_volume_trend = df['volume'] - volume_ema_10
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_positive = price_trend > 0
    
    # Combine Spread, Volume, and Price Trends
    adjusted_spread = high_low_spread * (daily_volume_trend * 1.5 if volume_trend_positive else daily_volume_trend * 0.5)
    final_factor = adjusted_spread * (1.2 if price_trend_positive else 0.8)
    
    return final_factor
```
