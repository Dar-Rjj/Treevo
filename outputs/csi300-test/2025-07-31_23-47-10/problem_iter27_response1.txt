```python
import pandas as pd
import numpy as np
from scipy.stats import percentileofscore

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_signal = (volume_trend > 0).astype(int)
    
    # Calculate Short-Term Price Trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    short_term_trend_signal = (short_term_trend > 0).astype(int)
    
    # Calculate Medium-Term Price Trend
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30
    medium_term_trend_signal = (medium_term_trend > 0).astype(int)
    
    # Calculate Long-Term Price Trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    long_term_trend_signal = (long_term_trend > 0).astype(int)
    
    # Calculate Dynamic Volatility
    std_10 = df['close'].rolling(window=10).std()
    volatility_percentile = [percentileofscore(std_10, x) for x in std_10]
    volatility_score = np.where(np.array(volatility_percentile) > 80, 2.5,
                                np.where((np.array(volatility_percentile) > 60), 1.5,
                                         np.where((np.array(volatility_percentile) > 40), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread * (1.5 * volume_trend_signal + 0.5 * (1 - volume_trend_signal))
    
    # Incorporate Short-Term Price Trend
    adjusted_spread *= (1.2 * short_term_trend_signal + 0.8 * (1 - short_term_trend_signal))
    
    # Incorporate Medium-Term Price Trend
    adjusted_spread *= (1.1 * medium_term_trend_signal + 0.9 * (1 - medium_term_trend_signal))
    
    # Incorporate Long-Term Price Trend
    adjusted_spread *= (1.3 * long_term_trend_signal + 0.7 * (1 - long_term_trend_signal))
    
    # Incorporate Dynamic Volatility
    adjusted_spread *= volatility_score
    
    # Incorporate Momentum and Relative Strength
    adjusted_spread *= (1.7 * (relative_strength > 1) + 0.3 * (relative_strength <= 1))
    
    # Consider Dynamic Market Context
    # For simplicity, we assume the market trend is bullish and the sector is outperforming
    market_trend_factor = 1.5  # Bullish
    sector_performance_factor = 1.2  # Outperforming
    adjusted_spread *= (market_trend_factor * long_term_trend_signal + sector_performance_factor * short_term_trend_signal)
    
    return adjusted_spread
```
