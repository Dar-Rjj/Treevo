```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10day_ma
    volume_trend_adjusted = np.where(volume_trend > 0, 1.5 * volume_trend, 0.5 * volume_trend)
    
    # Calculate Price Trend
    close_10day_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10day_ema
    price_trend_adjusted = np.where(price_trend > 0, 1.2 * price_trend, 0.8 * price_trend)
    
    # Combine Spread, Volume, and Price Trends
    spread_volume_trend = high_low_spread * volume_trend_adjusted
    alpha_factor = spread_volume_trend + price_trend_adjusted
    
    return alpha_factor
```
