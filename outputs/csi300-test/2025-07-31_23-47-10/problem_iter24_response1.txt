```python
import pandas as pd
import numpy as np
from scipy.stats import rankdata

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - short_term_ema
    short_term_positive = short_term_price_trend > 0
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - medium_term_ema
    medium_term_positive = medium_term_price_trend > 0
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - long_term_ema
    long_term_positive = long_term_price_trend > 0
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_score = np.where(volatility > np.percentile(volatility, 80), 2.5,
                                np.where((volatility > np.percentile(volatility, 60)) & (volatility <= np.percentile(volatility, 80)), 1.5,
                                         np.where((volatility > np.percentile(volatility, 40)) & (volatility <= np.percentile(volatility, 60)), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread * np.where(volume_trend_positive, 1.5, 0.5)
    adjusted_spread *= np.where(short_term_positive, 1.2, 0.8)
    adjusted_spread *= np.where(medium_term_positive, 1.1, 0.9)
    adjusted_spread *= np.where(long_term_positive, 1.3, 0.7)
    adjusted_spread *= volatility_score
    adjusted_spread *= np.where(relative_strength > 1, 1.7, 0.3)
    
    # Consider Dynamic Market Context (Simplified for this example)
    # For a more complete solution, integrate market and sector analysis
    adjusted_spread *= np.where(long_term_positive, 1.5, 0.5)  # Simplified market trend
    adjusted_spread *= np.where(short_term_positive, 1.2, 0.8)  # Simplified sector performance
    
    # Adapt Weights Based on Cross-Sectional Rank
    ranks = rankdata(adjusted_spread, method='ordinal')
    weights = 1 / (ranks + 1)
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread * weights
    
    return pd.Series(alpha_factor, index=df.index)
```
