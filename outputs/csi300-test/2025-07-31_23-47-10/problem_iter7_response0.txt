```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'].rolling(window=10).mean() - df['volume']
    
    # Determine if volume trend is positive or negative
    volume_trend_sign = np.sign(volume_trend)
    
    # Calculate Price Trend using EMA
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_10
    
    # Determine if price trend is positive or negative
    price_trend_sign = np.sign(price_trend)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread = high_low_spread * (volume_trend_sign * 0.5 + 1) * (price_trend_sign * 0.2 + 1)
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread
    
    return alpha_factor
```
