```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day Exponential Moving Average (EMA) of Volume
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - volume_ema
    
    # Determine if positive or negative trend
    volume_trend_indicator = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Combine Spread and Volume Trend
    adjusted_high_low_spread = high_low_spread * volume_trend_indicator
    
    # Output the factor values
    return adjusted_high_low_spread
```
