```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_10d_avg = df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - volume_10d_avg
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend > 0] *= 1.5
    adjusted_spread[volume_trend <= 0] *= 0.5
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    
    return alpha_factor
```
