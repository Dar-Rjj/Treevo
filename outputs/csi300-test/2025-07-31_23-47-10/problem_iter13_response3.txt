import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High-Low Spread'] = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    df['Volume Moving Average 10D'] = df['volume'].rolling(window=10).mean()
    df['Volume Trend'] = df['volume'] - df['Volume Moving Average 10D']
    
    # Calculate Short-Term EMA
    df['EMA 5D'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Short-Term EMA Trend'] = df['close'] - df['EMA 5D']
    
    # Calculate Long-Term EMA
    df['EMA 20D'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Long-Term EMA Trend'] = df['close'] - df['EMA 20D']
    
    # Combine Spread, Volume, and EMAs
    alpha_factor = df['High-Low Spread'].copy()
    
    # Adjust for volume trend
    alpha_factor = np.where(df['Volume Trend'] > 0, alpha_factor * 1.5, alpha_factor * 0.5)
    
    # Incorporate Short-Term EMA
    alpha_factor = np.where(df['Short-Term EMA Trend'] > 0, alpha_factor * 1.2, alpha_factor * 0.8)
    
    # Incorporate Long-Term EMA
    alpha_factor = np.where(df['Long-Term EMA Trend'] > 0, alpha_factor * 1.3, alpha_factor * 0.7)
    
    return pd.Series(alpha_factor, index=df.index)
