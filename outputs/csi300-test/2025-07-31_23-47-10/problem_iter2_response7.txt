```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=20, M=20):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Log Return
    log_return = np.log(df['close'] / df['close'].shift(1))
    
    # Compute Momentum Indicator
    momentum_indicator = log_return.rolling(window=N).sum()
    
    # Calculate Volatility
    volatility = log_return.rolling(window=M).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = momentum_indicator / volatility
    
    # Weight by High-Low Range
    factor_values = high_low_range * adjusted_momentum
    
    return factor_values
```
