```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    
    # Determine if positive or negative trend
    volume_trend_factor = (volume_trend > 0) * 1.5 + (volume_trend < 0) * 0.5
    
    # Combine Spread and Volume Trend
    adjusted_spread = high_low_spread * volume_trend_factor
    
    return adjusted_spread
```
