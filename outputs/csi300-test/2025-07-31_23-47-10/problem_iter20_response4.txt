```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    
    # Calculate Short-Term Price Trend
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - short_term_ema
    
    # Calculate Medium-Term Price Trend
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - medium_term_ema
    
    # Calculate Long-Term Price Trend
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - long_term_ema
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_weight = np.where(volatility > volatility.quantile(0.5), 1.4, 0.6)
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread * np.where(volume_trend > 0, 1.5, 0.5)
    short_term_trend_weight = np.where(short_term_trend > 0, 1.2, 0.8)
    medium_term_trend_weight = np.where(medium_term_trend > 0, 1.1, 0.9)
    long_term_trend_weight = np.where(long_term_trend > 0, 1.3, 0.7)
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)
    
    # Final Alpha Factor
    alpha_factor = (adjusted_spread * short_term_trend_weight * 
                    medium_term_trend_weight * long_term_trend_weight * 
                    volatility_weight * relative_strength_weight)
    
    return alpha_factor
```
