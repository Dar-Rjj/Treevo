```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10day_ma
    
    # Determine if positive or negative trend
    volume_trend_flag = (volume_trend > 0).astype(int)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread * (1.5 * volume_trend_flag + 0.5 * (1 - volume_trend_flag))
    
    # Calculate Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread * volatility
    
    return alpha_factor
```
