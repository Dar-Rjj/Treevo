```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    vol_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_vol_trend = df['volume'] - vol_10_day_ma
    volume_trend_positive = daily_vol_trend > 0
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    short_term_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - short_term_ema
    short_term_trend_positive = short_term_trend > 0
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    medium_term_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - medium_term_ema
    medium_term_trend_positive = medium_term_trend > 0
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    long_term_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - long_term_ema
    long_term_trend_positive = long_term_trend > 0
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    vol_80th_percentile = np.percentile(volatility.dropna(), 80)
    vol_60th_percentile = np.percentile(volatility.dropna(), 60)
    vol_40th_percentile = np.percentile(volatility.dropna(), 40)
    
    def volatility_score(x):
        if x > vol_80th_percentile:
            return 2.5
        elif x > vol_60th_percentile:
            return 1.5
        elif x > vol_40th_percentile:
            return 1.0
        else:
            return 0.5
    
    dynamic_volatility = volatility.apply(volatility_score)
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_positive] *= 1.5
    adjusted_spread[~volume_trend_positive] *= 0.5
    
    adjusted_spread[short_term_trend_positive] *= 1.2
    adjusted_spread[~short_term_trend_positive] *= 0.8
    
    adjusted_spread[medium_term_trend_positive] *= 1.1
    adjusted_spread[~medium_term_trend_positive] *= 0.9
    
    adjusted_spread[long_term_trend_positive] *= 1.3
    adjusted_spread[~long_term_trend_positive] *= 0.7
    
    adjusted_spread *= dynamic_volatility
    
    adjusted_spread[relative_strength > 1] *= 1.7
    adjusted_spread[relative_strength <= 1] *= 0.3
    
    # Consider Dynamic Market Context
    market_trend_bullish = long_term_trend.mean() > 0
    sector_outperforming = short_term_trend.mean() > 0
    
    if market_trend_bullish:
        adjusted_spread[long_term_trend_positive] *= 1.5
        adjusted_spread[~long_term_trend_positive] *= 0.5
    
    if sector_outperforming:
        adjusted_spread[short_term_trend_positive] *= 1.2
        adjusted_spread[~short_term_trend_positive] *= 0.8
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    
    return alpha_factor
```
