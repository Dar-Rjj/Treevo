```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = volume_trend > 0
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_positive = price_trend > 0
    
    # Calculate Momentum Trend
    close_ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    close_ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    momentum_trend = close_ema_12 - close_ema_26
    momentum_trend_positive = momentum_trend > 0
    
    # Adjust High-Low Spread based on Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_positive] *= 1.5
    adjusted_spread[~volume_trend_positive] *= 0.5
    
    # Incorporate Price Trend
    adjusted_spread[price_trend_positive] *= 1.2
    adjusted_spread[~price_trend_positive] *= 0.8
    
    # Incorporate Momentum Trend
    final_alpha_factor = adjusted_spread.copy()
    final_alpha_factor[momentum_trend_positive] *= 1.3
    final_alpha_factor[~momentum_trend_positive] *= 0.7
    
    return final_alpha_factor
```
