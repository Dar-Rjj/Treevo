```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ema = df['volume'].ewm(span=10, adjust=False).mean()
    volume_trend = df['volume'] - volume_ema
    volume_trend_positive = (volume_trend > 0).astype(float)
    
    # Calculate Price Trend
    close_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema
    price_trend_positive = (price_trend > 0).astype(float)
    
    # Calculate Momentum Trend
    momentum = df['close'] - df['close'].shift(10)
    momentum_positive = (momentum > 0).astype(float)
    
    # Adjust High-Low Spread based on Volume Trend
    adjusted_spread = high_low_spread * (volume_trend_positive * 1.5 + (1 - volume_trend_positive) * 0.5)
    
    # Incorporate Price Trend
    adjusted_spread_price = adjusted_spread * (price_trend_positive * 1.2 + (1 - price_trend_positive) * 0.8)
    
    # Incorporate Momentum Trend
    final_alpha_factor = adjusted_spread_price * (momentum_positive * 1.3 + (1 - momentum_positive) * 0.7)
    
    return final_alpha_factor
```
