import pandas as pd
import numpy as np

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: Series with index as date and values as the computed alpha factor.
    """
    # Calculate the relative position of the closing price within the high-low range
    df['close_pos'] = (df['close'] - df['low']) / (df['high'] - df['low'])
    
    # Calculate the relative strength of the trading volume
    df['vol_strength'] = df['volume'] / df['amount']
    
    # Calculate the momentum of the closing price
    df['momentum'] = df['close'].pct_change()
    
    # Combine the heuristics to form the final alpha factor
    df['alpha_factor'] = (df['close_pos'] + df['vol_strength'] * df['momentum']) / 2
    
    return df['alpha_factor']
