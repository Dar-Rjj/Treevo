```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate daily returns using close price
    df['daily_return'] = df['close'].pct_change()
    
    # Initialize the weighted sum
    weighted_sum = 0
    
    # Aggregate volume weighted returns over the last N days
    for i in range(N):
        # Calculate volume weights
        volume_weights = df['volume'].shift(i) / df['volume'].rolling(window=N).sum()
        
        # Add to the weighted sum
        weighted_sum += df['daily_return'].shift(i) * volume_weights
    
    # Finalize the alpha factor
    df['volume_adjusted_momentum'] = weighted_sum / N
    
    return df['volume_adjusted_momentum']
```
