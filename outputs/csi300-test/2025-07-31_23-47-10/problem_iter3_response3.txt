```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - volume_10day_ma
    
    # Combine Spread and Volume Trend
    adjusted_trend = volume_trend.apply(lambda x: 1.5 if x > 0 else 0.5)
    combined_factor = high_low_spread * volume_trend * adjusted_trend
    
    return combined_factor
```
