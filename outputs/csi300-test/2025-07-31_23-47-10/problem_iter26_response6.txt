import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10_day_ma
    volume_trend_positive = daily_volume_trend > 0
    
    # Calculate Short-Term Price Trend
    close_10_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_price_trend = df['close'] - close_10_day_ema
    short_term_trend_positive = short_term_price_trend > 0
    
    # Calculate Medium-Term Price Trend
    close_30_day_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_price_trend = df['close'] - close_30_day_ema
    medium_term_trend_positive = medium_term_price_trend > 0
    
    # Calculate Long-Term Price Trend
    close_50_day_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_price_trend = df['close'] - close_50_day_ema
    long_term_trend_positive = long_term_price_trend > 0
    
    # Calculate Dynamic Volatility
    volatility_10_day = df['close'].rolling(window=10).std()
    volatility_80th = volatility_10_day.quantile(0.8)
    volatility_60th = volatility_10_day.quantile(0.6)
    volatility_40th = volatility_10_day.quantile(0.4)
    
    volatility_score = np.where(volatility_10_day > volatility_80th, 2.5,
                                np.where((volatility_10_day > volatility_60th) & (volatility_10_day <= volatility_80th), 1.5,
                                         np.where((volatility_10_day > volatility_40th) & (volatility_10_day <= volatility_60th), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    close_5_day_ema = df['close'].ewm(span=5, adjust=False).mean()
    close_20_day_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_5_day_ema / close_20_day_ema
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread
    adjusted_high_low_spread = np.where(volume_trend_positive, adjusted_high_low_spread * 1.5, adjusted_high_low_spread * 0.5)
    adjusted_high_low_spread = np.where(short_term_trend_positive, adjusted_high_low_spread * 1.2, adjusted_high_low_spread * 0.8)
    adjusted_high_low_spread = np.where(medium_term_trend_positive, adjusted_high_low_spread * 1.1, adjusted_high_low_spread * 0.9)
    adjusted_high_low_spread = np.where(long_term_trend_positive, adjusted_high_low_spread * 1.3, adjusted_high_low_spread * 0.7)
    adjusted_high_low_spread = adjusted_high_low_spread * volatility_score
    adjusted_high_low_spread = np.where(relative_strength > 1, adjusted_high_low_spread * 1.7, adjusted_high_low_spread * 0.3)
    
    # Consider Dynamic Market Context (simplified for this example)
    market_trend_bullish = True  # Assume bullish for simplicity
    sector_outperforming = True   # Assume outperforming for simplicity
    
    if market_trend_bullish:
        adjusted_high_low_spread = np.where(long_term_trend_positive, adjusted_high_low_spread * 1.5, adjusted_high_low_spread * 0.5)
    else:
        adjusted_high_low_spread = np.where(long_term_trend_positive, adjusted_high_low_spread * 0.5, adjusted_high_low_spread * 1.5)
    
    if sector_outperforming:
        adjusted_high_low_spread = np.where(short_term_trend_positive, adjusted_high_low_spread * 1.2, adjusted_high_low_spread * 0.8)
    else:
