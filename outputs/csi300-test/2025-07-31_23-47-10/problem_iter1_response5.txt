```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Log Returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Calculate Short-Term and Long-Term Average Log Returns
    df['short_term_avg'] = df['log_returns'].ewm(span=10, adjust=False).mean()
    df['long_term_avg'] = df['log_returns'].ewm(span=25, adjust=False).mean()
    
    # Determine Trend Strength
    df['trend_strength'] = (df['short_term_avg'] > df['long_term_avg']).astype(int)
    
    # Calculate Realized Volatility
    df['realized_volatility'] = df['log_returns']**2
    df['realized_volatility'] = df['realized_volatility'].rolling(window=20).mean()
    
    # Combine Momentum and Volatility
    df['momentum_signal'] = df['short_term_avg'] - df['long_term_avg']
    df['factor_value'] = (df['momentum_signal'] * df['trend_strength']) / df['realized_volatility']
    
    return df['factor_value'].dropna()
```
