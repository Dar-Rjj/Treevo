```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_trend = df['volume'].rolling(window=10).mean() - df['volume']
    volume_trend_adjusted = np.where(volume_trend > 0, 1.5, 0.5)

    # Calculate Short-Term EMA (5 days)
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    short_term_ema_diff = short_term_ema - df['close']
    short_term_ema_adjusted = np.where(short_term_ema_diff > 0, 1.2, 0.8)

    # Calculate Long-Term EMA (20 days)
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    long_term_ema_diff = long_term_ema - df['close']
    long_term_ema_adjusted = np.where(long_term_ema_diff > 0, 1.3, 0.7)

    # Combine Spread, Volume, and EMAs
    combined_factor = high_low_spread * volume_trend_adjusted * short_term_ema_adjusted * long_term_ema_adjusted

    return pd.Series(combined_factor, index=df.index, name='alpha_factor')
```
