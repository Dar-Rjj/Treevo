```python
import pandas as pd

def heuristics_v2(df):
    # Obtain Close Prices
    close_prices = df['close']
    
    # Calculate Returns
    daily_returns = close_prices.pct_change()
    returns_5_days = daily_returns.rolling(window=5).sum()
    
    # Calculate Volume Shock
    volume = df['volume']
    volume_change = volume.pct_change()
    volume_shock = (volume_change > 0.5).astype(int)  # Threshold set at 1.5x (i.e., 50% increase)
    
    # Combine Factors
    combined_factor = returns_5_days * volume_shock
    
    return combined_factor
```
