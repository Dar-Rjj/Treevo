```python
import pandas as pd
import talib

def heuristics_v2(data):
    # Calculate High-Low Spread
    high_low_spread = data['high'] - data['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = data['volume'].rolling(window=10).mean()
    volume_trend = (data['volume'] > volume_ma_10).astype(float)
    volume_trend[volume_trend == 1] = 1.5
    volume_trend[volume_trend == 0] = 0.5
    
    # Calculate Price Trend
    close_ema_10 = data['close'].ewm(span=10, adjust=False).mean()
    price_trend = (data['close'] > close_ema_10).astype(float)
    price_trend[price_trend == 1] = 1.2
    price_trend[price_trend == 0] = 0.8
    
    # Calculate Relative Strength Index (RSI)
    rsi = talib.RSI(data['close'], timeperiod=14)
    rsi_factor = pd.Series(1, index=data.index)
    rsi_factor[rsi > 70] = 0.5
    rsi_factor[rsi < 30] = 1.5
    
    # Combine Spread, Volume, and Price Trends
    adjusted_spread = high_low_spread * volume_trend
    adjusted_spread_with_price = adjusted_spread * price_trend
    alpha_factor = adjusted_spread_with_price * rsi_factor
    
    return alpha_factor
```
