import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    vol_mavg_10 = df['volume'].rolling(window=10).mean()
    daily_vol_trend = df['volume'] - vol_mavg_10
    vol_trend_factor = np.where(daily_vol_trend > 0, 1.5, 0.5)
    
    # Calculate Short-Term Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_ema_10
    short_term_trend_factor = np.where(short_term_trend > 0, 1.2, 0.8)
    
    # Calculate Medium-Term Price Trend
    close_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_ema_30
    medium_term_trend_factor = np.where(medium_term_trend > 0, 1.1, 0.9)
    
    # Calculate Long-Term Price Trend
    close_ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_ema_50
    long_term_trend_factor = np.where(long_term_trend > 0, 1.3, 0.7)
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_percentile = df['close'].rolling(window=10).std().rank(pct=True)
    volatility_score = np.select(
        [volatility_percentile > 0.8, volatility_percentile > 0.6, volatility_percentile > 0.4],
        [2.5, 1.5, 1.0],
        default=0.5
    )
    
    # Integrate Momentum and Relative Strength
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_ema_5 / close_ema_20
    momentum_factor = np.where(relative_strength > 1, 1.7, 0.3)
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread * vol_trend_factor
    combined_factor = (
        adjusted_high_low_spread * 
        short_term_trend_factor * 
        medium_term_trend_factor * 
        long_term_trend_factor * 
        volatility_score * 
        momentum_factor
    )
    
    # Consider Dynamic Market Context
    # Placeholder for market trend and sector performance analysis
    # For simplicity, we'll assume the market is bullish and the sector is outperforming
    market_trend_weight = 1.5
    sector_performance_weight = 1.2
    final_alpha_factor = combined_factor * market_trend_weight * sector_performance_weight
    
    return pd.Series(final_alpha_factor, index=df.index)
