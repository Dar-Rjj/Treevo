```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Adjust High-Low Spread Based on Volume Trend
    adjusted_high_low_spread = high_low_spread.where(volume_trend >= 0, high_low_spread * 0.5)
    adjusted_high_low_spread = adjusted_high_low_spread.where(volume_trend < 0, high_low_spread * 1.5)
    
    # Incorporate Close-to-Open Momentum
    close_open_momentum = df['close'] - df['open']
    
    # Combine Adjusted High-Low Spread and Momentum
    combined_metric = adjusted_high_low_spread + close_open_momentum
    
    # Calculate Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Generate Final Alpha Factor
    alpha_factor = combined_metric * rsi
    
    return alpha_factor
```
