```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    df['Volume_MA10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA10']
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    df['Close_EMA10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Trend'] = df['close'] - df['Close_EMA10']
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    df['Close_EMA30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Price_Trend'] = df['close'] - df['Close_EMA30']
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    df['Close_EMA50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Price_Trend'] = df['close'] - df['Close_EMA50']
    
    # Calculate Dynamic Volatility (Standard Deviation over 10 days)
    df['Volatility'] = df['close'].rolling(window=10).std()
    median_volatility = df['Volatility'].median()
    df['Volatility_Adjusted'] = df['Volatility'].apply(lambda x: 1.4 if x > median_volatility else 0.6)
    
    # Integrate Momentum and Relative Strength
    df['Close_EMA5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['Close_EMA20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Relative_Strength'] = df['Close_EMA5'] / df['Close_EMA20']
    
    # Calculate MACD and Signal Line
    df['MACD'] = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    df['Signal_Line'] = df['MACD'].ewm(span=9, adjust=False).mean()
    df['MACD_Signal'] = df['MACD'] > df['Signal_Line']
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    df['Adjusted_Spread'] = df['High-Low_Spread'] * df['Volume_Trend'].apply(lambda x: 1.5 if x > 0 else 0.5)
    df['Adjusted_Spread'] *= df['Short_Term_Price_Trend'].apply(lambda x: 1.2 if x > 0 else 0.8)
    df['Adjusted_Spread'] *= df['Medium_Term_Price_Trend'].apply(lambda x: 1.1 if x > 0 else 0.9)
    df['Adjusted_Spread'] *= df['Long_Term_Price_Trend'].apply(lambda x: 1.3 if x > 0 else 0.7)
    df['Adjusted_Spread'] *= df['Volatility_Adjusted']
    df['Adjusted_Spread'] *= df['Relative_Strength'].apply(lambda x: 1.5 if x > 1 else 0.5)
    df['Adjusted_Spread'] *= df['MACD_Signal'].apply(lambda x: 1.6 if x else 0.4)
    
    # Final Alpha Factor
    alpha_factor = df['Adjusted_Spread']
    
    return alpha_factor
```
