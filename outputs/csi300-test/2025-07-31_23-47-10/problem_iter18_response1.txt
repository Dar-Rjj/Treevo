```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_ma
    volume_trend_signal = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Calculate Price Trend
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10_ema
    price_trend_signal = np.where(price_trend > 0, 1.2, 0.8)
    
    # Integrate Momentum
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    momentum = short_term_ema - long_term_ema
    
    # Adaptive Volatility
    volatility = df['close'].rolling(window=20).std() / df['close'].rolling(window=20).mean()
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Dynamic Weight Adjustment
    momentum_weight = np.where(momentum > 0, 1.5, 0.5)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_volume_trend = high_low_spread * volume_trend_signal
    adjusted_spread_volume_trend = spread_volume_trend * price_trend_signal
    final_factor = adjusted_spread_volume_trend * momentum_weight * volatility_weight
    
    return final_factor
```
