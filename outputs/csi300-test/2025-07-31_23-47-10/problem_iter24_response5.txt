import pandas as pd

def heuristics_v2(df):
    # Calculate the return from open to close
    df['return_open_to_close'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the return from high to low
    df['return_high_to_low'] = (df['low'] - df['high']) / df['high']
    
    # Calculate the ratio of volume to amount
    df['volume_to_amount_ratio'] = df['volume'] / df['amount']
    
    # Calculate the average price
    df['average_price'] = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the weighted average of the returns and the volume-to-amount ratio
    df['alpha_factor'] = (df['return_open_to_close'] * 0.3 +
                          df['return_high_to_low'] * 0.2 +
                          df['volume_to_amount_ratio'] * 0.1 +
                          df['average_price'] * 0.4)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
