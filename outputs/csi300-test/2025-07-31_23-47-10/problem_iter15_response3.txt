```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10_ma
    volume_trend_positive = (daily_volume_trend > 0).astype(int)
    
    # Calculate Price Trend
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10_ema
    price_trend_positive = (price_trend > 0).astype(int)
    
    # Integrate Momentum and Relative Strength
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema / long_term_ema
    
    # Adaptive Volatility
    close_20_ma = df['close'].rolling(window=20).mean()
    volatility = df['close'].rolling(window=20).std() / close_20_ma
    volatility_high = (volatility > 1).astype(int)
    
    # Dynamic Weight Adjustment
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    volume_trend_multiplier = volume_trend_positive * 1.5 + (1 - volume_trend_positive) * 0.5
    price_trend_multiplier = price_trend_positive * 1.2 + (1 - price_trend_positive) * 0.8
    combined_factor = high_low_spread * volume_trend_multiplier * price_trend_multiplier * relative_strength_weight * volatility_weight
    
    return combined_factor
```
