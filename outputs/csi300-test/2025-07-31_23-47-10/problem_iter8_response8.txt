```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    
    # Calculate Volatility
    volatility = df['close'].rolling(window=10).std()
    
    # Combine Spread, Volume, Price Trends, and Volatility
    adjusted_spread = high_low_spread * (volume_trend > 0) * 1.5 + high_low_spread * (volume_trend < 0) * 0.5
    adjusted_price_trend = (price_trend > 0) * 1.2 + (price_trend < 0) * 0.8
    adjusted_volatility = (volatility > volatility.mean()) * 1.3 + (volatility <= volatility.mean()) * 0.7
    
    alpha_factor = adjusted_spread * adjusted_price_trend * adjusted_volatility
    
    return alpha_factor
```
