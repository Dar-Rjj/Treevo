```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Calculate Short-Term Price Trend
    short_term_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - short_term_ema_10
    
    # Calculate Medium-Term Price Trend
    medium_term_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - medium_term_ema_30
    
    # Calculate Long-Term Price Trend
    long_term_ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - long_term_ema_50
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_score = np.where(volatility > np.percentile(volatility, 80), 2.5,
                                np.where((volatility > np.percentile(volatility, 60)) & (volatility <= np.percentile(volatility, 80)), 1.5,
                                         np.where((volatility > np.percentile(volatility, 40)) & (volatility <= np.percentile(volatility, 60)), 1.0, 0.5)))
    
    # Integrate Momentum and Relative Strength
    short_term_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema_5 / long_term_ema_20
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread * np.where(volume_trend > 0, 1.5, 0.5)
    adjusted_spread *= np.where(short_term_trend > 0, 1.2, 0.8)
    adjusted_spread *= np.where(medium_term_trend > 0, 1.1, 0.9)
    adjusted_spread *= np.where(long_term_trend > 0, 1.3, 0.7)
    adjusted_spread *= volatility_score
    adjusted_spread *= np.where(relative_strength > 1, 1.7, 0.3)
    
    # Consider Dynamic Market Context
    # Assuming market trend and sector performance are provided as additional columns in the DataFrame
    # For simplicity, let's assume they are not available and we set them to 1.0
    market_trend_weight = 1.0  # Adjust this based on actual market trend
    sector_performance_weight = 1.0  # Adjust this based on actual sector performance
    
    adjusted_spread *= np.where(market_trend_weight > 0, 1.5, 0.5)
    adjusted_spread *= np.where(sector_performance_weight > 0, 1.2, 0.8)
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    return pd.Series(alpha_factor, index=df.index)
```
