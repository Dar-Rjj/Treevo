```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Calculate Intraday Open-Close Difference
    intraday_open_close_diff = df['close'] - df['open']
    
    # Combine Intraday Movements
    combined_intraday_movements = intraday_high_low_range * intraday_open_close_diff
    
    # Calculate Volume Intensity
    volume_moving_average = df['volume'].rolling(window=10).mean()
    volume_intensity = df['volume'] / volume_moving_average
    
    # Generate Weighted Intraday Momentum Volatility Score
    weighted_score = combined_intraday_movements * volume_intensity
    
    return weighted_score
```
