```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-Open Difference
    close_open_diff = df['close'] - df['open']
    
    # Combine with Close-Open Difference and Intraday Range
    combined_movement = close_open_diff * intraday_range
    
    # Multiply by Volume
    final_factor = combined_movement * df['volume']
    
    return final_factor
```
