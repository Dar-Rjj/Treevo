```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_adj = (volume_trend > 0) * 1.5 + (volume_trend < 0) * 0.5
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread = high_low_spread * volume_trend_adj
    
    # Calculate Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()
    
    # Calculate Momentum
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    momentum = ema_50 - ema_200
    
    # Combine Spread, Volume Trend, Volatility, and Momentum
    alpha_factor = adjusted_high_low_spread * volatility * np.sign(momentum)
    
    return alpha_factor
```
