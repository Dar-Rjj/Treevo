```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10_day_ma
    volume_trend_factor = np.where(daily_volume_trend > 0, 1.5, 0.5)

    # Calculate Volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = log_returns.rolling(window=20).std()

    # Calculate Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    price_momentum = df['close'] - close_5_day_ma
    momentum_factor = np.where(price_momentum > 0, 1.2, 0.8)

    # Combine Spread, Volume Trend, Volatility, and Price Momentum
    adjusted_spread = high_low_spread * volume_trend_factor
    final_adjusted_spread = adjusted_spread * momentum_factor
    alpha_factor = final_adjusted_spread * volatility

    return alpha_factor
```
