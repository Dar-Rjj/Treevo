```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - volume_10_day_ma
    
    # Determine if positive or negative trend
    positive_trend = (volume_trend > 0).astype(float)
    negative_trend = (volume_trend <= 0).astype(float)
    
    # Combine Spread and Volume Trend
    combined_factor = high_low_spread * (positive_trend * 1.5 + negative_trend * 0.5)
    
    return combined_factor
```
