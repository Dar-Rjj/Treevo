```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_sign = np.sign(volume_trend)
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_sign = np.sign(price_trend)
    
    # Calculate Momentum
    momentum = (df['close'] / df['close'].shift(10)) - 1
    momentum_sign = np.sign(momentum)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_sign > 0] *= 1.5
    adjusted_spread[volume_trend_sign < 0] *= 0.5
    
    # Incorporate Price Trend
    adjusted_spread[price_trend_sign > 0] *= 1.2
    adjusted_spread[price_trend_sign < 0] *= 0.8
    
    # Adjust for Momentum
    final_factor = adjusted_spread.copy()
    final_factor[momentum_sign > 0] *= 1.3
    final_factor[momentum_sign < 0] *= 0.7
    
    return final_factor
```
