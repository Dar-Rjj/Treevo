```python
import pandas as pd

def heuristics_v2(df):
    """
    Generates a novel alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    # Calculate price movement
    df['price_movement'] = df['close'] - df['open']
    
    # Calculate normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].mean()) / df['volume'].std()
    
    # Calculate the difference between high and low
    df['high_low_diff'] = df['high'] - df['low']
    
    # Calculate the weighted price movement
    df['weighted_price_movement'] = df['price_movement'] * df['norm_volume']
    
    # Calculate the final factor
    df['alpha_factor'] = df['weighted_price_movement'] / df['high_low_diff']
    
    return df['alpha_factor']
```
