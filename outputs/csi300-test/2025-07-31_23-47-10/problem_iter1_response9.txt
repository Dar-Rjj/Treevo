```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Close Momentum
    close_momentum = df['close'].diff(1)
    
    # Combine Range and Momentum
    combined_metric = daily_range * close_momentum
    
    # Adjust by Trading Activity
    adjusted_metric = combined_metric / df['volume']
    
    return adjusted_metric
```
