```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10ma
    
    # Determine if positive or negative trend for volume
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Price Trend
    close_10ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10ema
    
    # Determine if positive or negative trend for price
    price_trend_positive = (price_trend > 0).astype(int)
    
    # Integrate Momentum and Relative Strength
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema / long_term_ema
    
    # Adaptive Volatility
    close_20std = df['close'].rolling(window=20).std()
    close_20ma = df['close'].rolling(window=20).mean()
    volatility = close_20std / close_20ma
    
    # Dynamic Weight Adjustment
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Incorporate Liquidity Measures
    true_range = df['high'] - df['low']
    atr_14 = true_range.rolling(window=14).mean()
    atr_weight = np.where(atr_14 > 1, 1.3, 0.7)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_volume_trend = high_low_spread * np.where(volume_trend_positive, 1.5, 0.5)
    weighted_price_trend = price_trend * np.where(price_trend_positive, 1.2, 0.8)
    
    # Final Alpha Factor
    alpha_factor = (spread_volume_trend + weighted_price_trend) * relative_strength_weight * volatility_weight * atr_weight
    
    return alpha_factor
```
