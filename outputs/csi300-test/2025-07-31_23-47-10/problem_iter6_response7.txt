import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day weighted moving average of volume
    df['volume_wma_10'] = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate Daily Volume Trend
    daily_volume_trend = df['volume'] - df['volume_wma_10']
    volume_trend_multiplier = (daily_volume_trend > 0) * 1.5 + (daily_volume_trend < 0) * 0.5
    
    # Calculate 14-day relative strength
    df['gain'] = df['close'].diff().clip(lower=0)
    df['loss'] = -df['close'].diff().clip(upper=0)
    avg_gain = df['gain'].rolling(window=14).mean()
    avg_loss = df['loss'].rolling(window=14).mean()
    
    # Calculate RSI
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Adjust for Relative Strength
    rsi_adjustment = (rsi > 70) * 0.8 + (rsi < 30) * 1.2 + (rsi >= 30) & (rsi <= 70) * 1
    adjusted_spread = high_low_spread * volume_trend_multiplier * rsi_adjustment
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    return alpha_factor
