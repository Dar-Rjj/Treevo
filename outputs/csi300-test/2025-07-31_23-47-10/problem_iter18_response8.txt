```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Short-Term Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_ema_10
    short_term_trend_positive = (short_term_trend > 0).astype(int)
    
    # Calculate Medium-Term Price Trend
    close_ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_ema_30
    medium_term_trend_positive = (medium_term_trend > 0).astype(int)
    
    # Calculate Long-Term Price Trend
    close_ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_ema_50
    long_term_trend_positive = (long_term_trend > 0).astype(int)
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_high = (volatility > volatility.quantile(0.5)).astype(int)
    
    # Integrate Momentum and Relative Strength
    close_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_ema_5 / close_ema_20
    relative_strength_positive = (relative_strength > 1).astype(int)
    
    # Calculate MACD and Signal Line
    macd = df['close'].ewm(span=12, adjust=False).mean() - df['close'].ewm(span=26, adjust=False).mean()
    signal_line = macd.ewm(span=9, adjust=False).mean()
    macd_bullish = (macd > signal_line).astype(int)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread * (1.5 * volume_trend_positive + 0.5 * (1 - volume_trend_positive))
    
    # Incorporate Short-Term Price Trend
    adjusted_spread *= (1.2 * short_term_trend_positive + 0.8 * (1 - short_term_trend_positive))
    
    # Incorporate Medium-Term Price Trend
    adjusted_spread *= (1.1 * medium_term_trend_positive + 0.9 * (1 - medium_term_trend_positive))
    
    # Incorporate Long-Term Price Trend
    adjusted_spread *= (1.3 * long_term_trend_positive + 0.7 * (1 - long_term_trend_positive))
    
    # Incorporate Dynamic Volatility
    adjusted_spread *= (1.4 * volatility_high + 0.6 * (1 - volatility_high))
    
    # Incorporate Momentum and Relative Strength
    adjusted_spread *= (1.5 * relative_strength_positive + 0.5 * (1 - relative_strength_positive))
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread + 0.1 * (macd - signal_line)
    
    return alpha_factor
```
