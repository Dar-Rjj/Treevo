```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_direction = (volume_trend > 0).astype(int)
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_direction = (price_trend > 0).astype(int)
    
    # Integrate Momentum and Relative Strength
    short_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_ema_5 / long_ema_20
    
    # Adaptive Volatility
    close_std_20 = df['close'].rolling(window=20).std()
    close_ma_20 = df['close'].rolling(window=20).mean()
    volatility = close_std_20 / close_ma_20
    
    # Dynamic Weight Adjustment
    rs_weight = (relative_strength > 1) * 1.5 + (relative_strength < 1) * 0.5
    vol_weight = (volatility > 1) * 1.2 + (volatility < 1) * 0.8
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_adjusted = high_low_spread * (volume_trend_direction * 1.5 + (1 - volume_trend_direction) * 0.5)
    price_trend_adjusted = (price_trend_direction * 1.2 + (1 - price_trend_direction) * 0.8)
    
    final_factor = spread_adjusted * price_trend_adjusted * rs_weight * vol_weight
    
    return final_factor
```
