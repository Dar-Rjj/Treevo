```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    vol_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_vol_trend = df['volume'] - vol_10_day_ma
    vol_trend_positive = daily_vol_trend > 0
    
    # Calculate Price Trend
    close_10_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10_day_ema
    price_trend_positive = price_trend > 0
    
    # Integrate Momentum and Relative Strength
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema / long_term_ema
    
    # Adaptive Volatility
    close_20_day_std = df['close'].rolling(window=20).std()
    close_20_day_ma = df['close'].rolling(window=20).mean()
    adaptive_volatility = close_20_day_std / close_20_day_ma
    high_volatility = adaptive_volatility > 1
    
    # Dynamic Weight Adjustment
    rs_weight = np.where(relative_strength > 1, 1.5, 0.5)
    vol_weight = np.where(high_volatility, 1.2, 0.8)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_volume_trend = high_low_spread * np.where(vol_trend_positive, 1.5, 0.5)
    adjusted_price_trend = np.where(price_trend_positive, 1.2, 0.8)
    final_alpha_factor = spread_volume_trend * adjusted_price_trend * rs_weight * vol_weight
    
    return final_alpha_factor
```
