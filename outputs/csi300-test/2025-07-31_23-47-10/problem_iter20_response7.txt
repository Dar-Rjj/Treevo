```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['High_Low_Spread'] = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    df['Volume_MA_10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA_10']
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    df['EMA_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Short_Term_Price_Trend'] = df['close'] - df['EMA_10']
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    df['EMA_30'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Price_Trend'] = df['close'] - df['EMA_30']
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    df['EMA_50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Price_Trend'] = df['close'] - df['EMA_50']
    
    # Calculate Dynamic Volatility (Standard Deviation over 10 days)
    df['Volatility_10'] = df['close'].rolling(window=10).std()
    volatility_percentile = df['Volatility_10'].quantile(0.5)
    df['Dynamic_Volatility'] = np.where(df['Volatility_10'] > volatility_percentile, 1.4, 0.6)
    
    # Integrate Momentum and Relative Strength
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['Relative_Strength'] = df['EMA_5'] / df['EMA_20']
    
    # Adjust High-Low Spread by Volume Trend
    df['Adjusted_Spread'] = df['High_Low_Spread'] * np.where(df['Volume_Trend'] > 0, 1.5, 0.5)
    
    # Incorporate Short-Term Price Trend
    df['Adjusted_Spread'] = df['Adjusted_Spread'] * np.where(df['Short_Term_Price_Trend'] > 0, 1.2, 0.8)
    
    # Incorporate Medium-Term Price Trend
    df['Adjusted_Spread'] = df['Adjusted_Spread'] * np.where(df['Medium_Term_Price_Trend'] > 0, 1.1, 0.9)
    
    # Incorporate Long-Term Price Trend
    df['Adjusted_Spread'] = df['Adjusted_Spread'] * np.where(df['Long_Term_Price_Trend'] > 0, 1.3, 0.7)
    
    # Incorporate Dynamic Volatility
    df['Adjusted_Spread'] = df['Adjusted_Spread'] * df['Dynamic_Volatility']
    
    # Incorporate Momentum and Relative Strength
    df['Adjusted_Spread'] = df['Adjusted_Spread'] * np.where(df['Relative_Strength'] > 1, 1.5, 0.5)
    
    # Final Alpha Factor
    alpha_factor = df['Adjusted_Spread']
    
    return alpha_factor
```
