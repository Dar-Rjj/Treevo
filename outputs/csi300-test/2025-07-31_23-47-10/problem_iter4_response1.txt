```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Log Return
    log_return = np.log(df['close']).diff()
    
    # Compute Momentum Indicator (Sum of log returns over past N days, e.g., N=20)
    N = 20
    momentum = log_return.rolling(window=N).sum()
    
    # Calculate Volatility (Standard Deviation of log returns over past M days, e.g., M=20)
    M = 20
    volatility = log_return.rolling(window=M).std()
    
    # Adjust Momentum by Volatility
    adjusted_momentum = momentum / volatility
    
    # Calculate Daily Volume Trend
    volume_10day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10day_ma
    volume_trend_factor = np.where(volume_trend > 0, 1.5, 0.5)
    
    # Combine Volume Trend and Momentum
    combined_momentum_volume = adjusted_momentum * volume_trend_factor
    
    # Weight by High-Low Range
    final_factor = high_low_range * combined_momentum_volume
    
    return final_factor
```
