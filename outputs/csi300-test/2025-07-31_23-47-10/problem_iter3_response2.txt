```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_moving_avg = df['volume'].rolling(window=10).mean()
    
    # Calculate daily volume trend
    volume_trend = df['volume'] - volume_moving_avg
    
    # Determine if the volume trend is positive or negative
    volume_trend_sign = (volume_trend > 0).astype(int)
    
    # Adjust for positive and negative trends
    adjusted_trend = volume_trend_sign * 1.5 + (1 - volume_trend_sign) * 0.5
    adjusted_high_low_spread = high_low_spread * adjusted_trend * volume_trend
    
    return adjusted_high_low_spread
```
