```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_ma
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_10_ema
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    close_30_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_30_ema
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    close_50_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_50_ema
    
    # Calculate Dynamic Volatility (Standard Deviation over 20 days)
    volatility = df['close'].rolling(window=20).std()
    high_volatility = volatility > volatility.quantile(0.75)
    low_volatility = volatility < volatility.quantile(0.25)
    
    # Integrate Momentum and Relative Strength
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_term_ema / long_term_ema
    strong_rs = relative_strength > 1
    weak_rs = relative_strength <= 1
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_high_low_spread = high_low_spread.copy()
    adjusted_high_low_spread[volume_trend > 0] *= 1.5
    adjusted_high_low_spread[volume_trend <= 0] *= 0.5
    
    adjusted_high_low_spread[short_term_trend > 0] *= 1.2
    adjusted_high_low_spread[short_term_trend <= 0] *= 0.8
    
    adjusted_high_low_spread[medium_term_trend > 0] *= 1.1
    adjusted_high_low_spread[medium_term_trend <= 0] *= 0.9
    
    adjusted_high_low_spread[long_term_trend > 0] *= 1.3
    adjusted_high_low_spread[long_term_trend <= 0] *= 0.7
    
    adjusted_high_low_spread[high_volatility] *= 1.4
    adjusted_high_low_spread[low_volatility] *= 0.6
    
    adjusted_high_low_spread[strong_rs] *= 1.5
    adjusted_high_low_spread[weak_rs] *= 0.5
    
    # Consider Market Context
    market_index = df['close'].ewm(span=200, adjust=False).mean()
    bullish_market = df['close'] > market_index
    bearish_market = df['close'] < market_index
    
    # Adjust Long-Term Price Trend based on market context
    adjusted_high_low_spread[(bullish_market & (long_term_trend > 0))] *= 1.5
    adjusted_high_low_spread[(bearish_market & (long_term_trend <= 0))] *= 0.5
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread
    
    return alpha_factor
```
