import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    vol_10d_avg = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - vol_10d_avg
    volume_trend_sign = np.sign(volume_trend)
    
    # Calculate Price Trend
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10d_ema
    price_trend_sign = np.sign(price_trend)
    
    # Integrate Momentum and Relative Strength
    short_ema = df['close'].ewm(span=5, adjust=False).mean()
    long_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_ema / long_ema
    
    # Adaptive Volatility
    volatility = df['close'].rolling(window=20).std() / df['close'].rolling(window=20).mean()
    
    # Dynamic Weight Adjustment
    rel_str_weight = np.where(relative_strength > 1, 1.5, 0.5)
    vol_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Market Sentiment Index
    market_sentiment = (df['close'] - df['open']) + (df['high'] - df['low'])
    sentiment_weight = np.where(market_sentiment > 0, 1.3, 0.7)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_adjusted = high_low_spread * np.where(volume_trend > 0, 1.5, 0.5)
    trend_adjusted = np.where(price_trend > 0, 1.2, 0.8)
    dynamic_weights = rel_str_weight * vol_weight * sentiment_weight
    
    # Final Alpha Factor
    alpha_factor = (spread_adjusted * trend_adjusted * dynamic_weights)
    
    return alpha_factor
