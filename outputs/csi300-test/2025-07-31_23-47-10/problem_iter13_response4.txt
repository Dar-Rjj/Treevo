```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_10_day_ma
    volume_trend_positive = (daily_volume_trend > 0).astype(int)
    
    # Calculate Price Trend
    close_10_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10_day_ema
    price_trend_positive = (price_trend > 0).astype(int)
    
    # Integrate Momentum and Relative Strength
    close_20_day_ema = df['close'].ewm(span=20, adjust=False).mean()
    close_50_day_ema = df['close'].ewm(span=50, adjust=False).mean()
    momentum_trend = close_20_day_ema - close_50_day_ema
    momentum_positive = (momentum_trend > 0).astype(int)
    
    # Adaptive Volatility
    close_20_day_std = df['close'].rolling(window=20).std()
    volatility_adjustment = np.where(close_20_day_std > close_20_day_std.mean(), 1.2, 0.8)
    
    # Combine Spread, Volume, and Price Trends
    adjusted_spread = high_low_spread * daily_volume_trend
    adjusted_spread = adjusted_spread * (volume_trend_positive * 1.5 + (1 - volume_trend_positive) * 0.5)
    adjusted_spread = adjusted_spread * (price_trend_positive * 1.2 + (1 - price_trend_positive) * 0.8)
    adjusted_spread = adjusted_spread * (momentum_positive * 1.3 + (1 - momentum_positive) * 0.7)
    adjusted_spread = adjusted_spread * volatility_adjustment
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    return alpha_factor
```
