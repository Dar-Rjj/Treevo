```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Volume Trend
    daily_volume_trend = df['volume'] - volume_ma_10
    
    # Determine if positive or negative trend
    trend_indicator = (daily_volume_trend > 0).astype(int) * 1.5 + (daily_volume_trend <= 0).astype(int) * 0.5
    
    # Combine Spread and Volume Trend
    combined_factor = high_low_spread * daily_volume_trend * trend_indicator
    
    return combined_factor
```
