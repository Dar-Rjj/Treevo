```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_day_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_day_ma
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Volatility
    log_returns = np.log(df['close']).diff()
    volatility = log_returns.rolling(window=20).std()
    
    # Calculate Price Momentum
    close_5_day_ma = df['close'].rolling(window=5).mean()
    price_momentum = df['close'] - close_5_day_ma
    price_momentum_positive = (price_momentum > 0).astype(int)
    
    # Calculate Relative Strength
    close_252_day_ma = df['close'].rolling(window=252).mean()
    relative_strength = df['close'] / close_252_day_ma
    
    # Calculate Market Breadth
    close_50_day_ma = df['close'].rolling(window=50).mean()
    above_50_day_ma = (df['close'] > close_50_day_ma).astype(int)
    market_breadth = above_50_day_ma.rolling(window=50).mean()
    
    # Calculate Adaptive Moving Average
    close_10_day_ema = df['close'].ewm(span=10, adjust=False).mean()
    close_30_day_ema = df['close'].ewm(span=30, adjust=False).mean()
    adaptive_moving_average_bullish = (close_10_day_ema > close_30_day_ema).astype(int)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread = high_low_spread * (1.5 * volume_trend_positive + 0.5 * (1 - volume_trend_positive))
    
    # Adjust for Price Momentum
    adjusted_high_low_spread = adjusted_high_low_spread * (1.2 * price_momentum_positive + 0.8 * (1 - price_momentum_positive))
    
    # Incorporate Relative Strength
    adjusted_high_low_spread = adjusted_high_low_spread * relative_strength
    
    # Incorporate Market Breadth
    adjusted_high_low_spread = adjusted_high_low_spread * (1.1 * (market_breadth > 0.5).astype(int) + 0.9 * (market_breadth <= 0.5).astype(int))
    
    # Final Alpha Factor
    alpha_factor = adjusted_high_low_spread * volatility
    
    return alpha_factor
```
