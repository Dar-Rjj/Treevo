```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    volume_trend_label = np.where(volume_trend > 0, 'positive', 'negative')
    
    # Calculate Short-Term Price Trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    short_term_trend_label = np.where(short_term_trend > 0, 'positive', 'negative')
    
    # Calculate Medium-Term Price Trend
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30
    medium_term_trend_label = np.where(medium_term_trend > 0, 'positive', 'negative')
    
    # Calculate Long-Term Price Trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    long_term_trend_label = np.where(long_term_trend > 0, 'positive', 'negative')
    
    # Calculate Dynamic Volatility
    volatility_20 = df['close'].rolling(window=20).std()
    volatility_quantile_75 = df['close'].rolling(window=20).std().quantile(0.75)
    volatility_quantile_25 = df['close'].rolling(window=20).std().quantile(0.25)
    volatility_label = np.where(volatility_20 > volatility_quantile_75, 'high',
                                np.where(volatility_20 < volatility_quantile_25, 'low', 'neutral'))
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    relative_strength_label = np.where(relative_strength > 1, 'strong', 'weak')
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_label == 'positive'] *= 1.5
    adjusted_spread[volume_trend_label == 'negative'] *= 0.5
    
    adjusted_spread[short_term_trend_label == 'positive'] *= 1.2
    adjusted_spread[short_term_trend_label == 'negative'] *= 0.8
    
    adjusted_spread[medium_term_trend_label == 'positive'] *= 1.1
    adjusted_spread[medium_term_trend_label == 'negative'] *= 0.9
    
    adjusted_spread[long_term_trend_label == 'positive'] *= 1.3
    adjusted_spread[long_term_trend_label == 'negative'] *= 0.7
    
    adjusted_spread[volatility_label == 'high'] *= 1.4
    adjusted_spread[volatility_label == 'low'] *= 0.6
    
    adjusted_spread[relative_strength_label == 'strong'] *= 1.5
    adjusted_spread[relative_strength_label == 'weak'] *= 0.5
    
    # Consider Market Context (simplified for this example)
    market_trend = 'bullish'  # Assume a bullish market trend
    if market_trend == 'bullish':
        adjusted_spread[long_term_trend_label == 'positive'] *= 1.5
    elif market_trend == 'bearish':
        adjusted_spread[long_term_trend_label == 'negative'] *= 0.5
    
    return adjusted_spread
```
