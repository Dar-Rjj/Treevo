```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_sign = np.where(volume_trend > 0, 1, -1)

    # Calculate Volatility
    close_20d_std = df['close'].rolling(window=20).std()
    normalized_high_low_spread = high_low_spread / close_20d_std

    # Calculate Long-Term Price Momentum
    close_20d_ma = df['close'].rolling(window=20).mean()
    close_50d_ma = df['close'].rolling(window=50).mean()
    long_term_momentum = close_20d_ma - close_50d_ma

    # Calculate Short-Term Price Reversal
    daily_price_change = df['close'] - df['open']
    sum_5d_price_change = daily_price_change.rolling(window=5).sum()
    high_5d = df['high'].rolling(window=5).max()
    low_5d = df['low'].rolling(window=5).min()
    high_low_range_5d = high_5d - low_5d
    short_term_reversal_signal = (sum_5d_price_change / high_low_range_5d) * df['volume']

    # Calculate Market Sentiment
    close_5d_ma = df['close'].rolling(window=5).mean()
    market_sentiment = np.where(df['close'] > close_5d_ma, 1.2, 0.8)
    
    # Combine Adjusted Spread, Volume Trend, Momentum, and Market Sentiment
    adjusted_spread = normalized_high_low_spread * volume_trend_sign
    adjusted_spread = np.where(volume_trend_sign > 0, adjusted_spread * 1.5, adjusted_spread * 0.5)
    combined_factor = adjusted_spread * long_term_momentum + short_term_reversal_signal * market_sentiment

    return combined_factor
```
