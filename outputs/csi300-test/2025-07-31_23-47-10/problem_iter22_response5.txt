    df['High-Low_Spread'] = df['high'] - df['low']

    df['Volume_MA10'] = df['volume'].rolling(window=10).mean()
    df['Volume_Trend'] = df['volume'] - df['Volume_MA10']
    df['Volume_Trend_Positive'] = df['Volume_Trend'] > 0

    df['Close_EMA10'] = df['close'].ewm(span=10, adjust=False).mean()

    df['Medium_Term_Price_Trend'] = df['close'] - df['Close_EMA30']
    df['Medium_Term_Price_Trend_Positive'] = df['Medium_Term_Price_Trend'] > 0
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    df['Close_EMA50'] = df['close'].ewm(span=50, adjust=False).mean()
    df['Long_Term_Price_Trend'] = df['close'] - df['Close_EMA50']
    df['Long_Term_Price_Trend_Positive'] = df['Long_Term_Price_Trend'] > 0

    df['Volatility'] = df['close'].rolling(window=20).std()
