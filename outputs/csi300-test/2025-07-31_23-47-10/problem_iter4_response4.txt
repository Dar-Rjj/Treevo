```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Determine if the trend is positive or negative
    volume_trend_indicator = volume_trend.apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Spread and Volume Trend
    adjusted_spread = high_low_spread * volume_trend_indicator
    adjusted_spread = adjusted_spread * (volume_trend_indicator.replace({1: 1.5, -1: 0.5}))
    
    return adjusted_spread
```
