```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10
    volume_trend_sign = np.where(daily_volume_trend > 0, 1.5, 0.5)
    
    # Calculate Price Trend
    close_ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema_10
    price_trend_sign = np.where(price_trend > 0, 1.2, 0.8)
    
    # Integrate Momentum and Relative Strength
    short_ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    long_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = short_ema_5 / long_ema_20
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)
    
    # Adaptive Volatility
    volatility = df['close'].rolling(window=20).std() / df['close'].rolling(window=20).mean()
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    adjusted_spread = high_low_spread * (daily_volume_trend * volume_trend_sign) * price_trend_sign * relative_strength_weight * volatility_weight
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    
    return alpha_factor
```
