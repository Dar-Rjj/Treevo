```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Log Returns
    df['log_returns'] = np.log(df['close']).diff()

    # Calculate Short-Term and Long-Term EMA of Log Returns
    df['ema_short'] = df['log_returns'].ewm(span=10, adjust=False).mean()
    df['ema_long'] = df['log_returns'].ewm(span=25, adjust=False).mean()

    # Determine Trend Strength
    df['trend_score'] = (df['ema_short'] > df['ema_long']).astype(int)

    # Adjust Momentum Signal with Trend Strength
    df['adjusted_momentum'] = df['ema_short'] * df['trend_score']

    # Calculate Squared Log Returns
    df['squared_log_returns'] = df['log_returns'] ** 2

    # Calculate 20-day SMA and EMA of Squared Log Returns
    df['sma_squared'] = df['squared_log_returns'].rolling(window=20).mean()
    df['ema_squared'] = df['squared_log_returns'].ewm(span=20, adjust=False).mean()

    # Combine SMA and EMA of Squared Log Returns
    df['combined_volatility'] = (df['sma_squared'] + df['ema_squared']) / 2

    # Combine Adaptive Momentum and Volatility
    df['alpha_factor'] = (df['adjusted_momentum'] * df['combined_volatility']) / df['volume']

    return df['alpha_factor'].dropna()
```
