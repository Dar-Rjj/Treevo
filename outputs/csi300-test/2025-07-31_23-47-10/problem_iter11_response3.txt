```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10_ma
    volume_trend_sign = np.sign(volume_trend)
    
    # Calculate Price Trend
    close_10_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_10_ema
    price_trend_sign = np.sign(price_trend)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_high_low_spread = high_low_spread * (1.5 * (volume_trend_sign > 0) + 0.5 * (volume_trend_sign < 0))
    
    # Incorporate Price Trend
    final_alpha_factor = adjusted_high_low_spread * (1.2 * (price_trend_sign > 0) + 0.8 * (price_trend_sign < 0))
    
    return final_alpha_factor
```
