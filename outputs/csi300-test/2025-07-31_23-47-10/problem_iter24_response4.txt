```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    daily_volume_trend = df['volume'] - volume_ma_10

    # Calculate Short-Term Price Trend
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10

    # Calculate Medium-Term Price Trend
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30

    # Calculate Long-Term Price Trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50

    # Calculate Dynamic Volatility
    volatility_20 = df['close'].rolling(window=20).std()
    high_volatility = volatility_20 > volatility_20.quantile(0.75)
    low_volatility = volatility_20 < volatility_20.quantile(0.25)

    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    strong_rs = relative_strength > 1
    weak_rs = relative_strength <= 1

    # Consider Market Context
    market_index = df['close']  # Assuming close price is used as a proxy for the market index
    market_trend_200 = market_index.rolling(window=200).mean()
    bullish_market = market_index > market_trend_200
    bearish_market = market_index < market_trend_200

    # Combine all factors
    adjusted_high_low_spread = high_low_spread * (1.5 if daily_volume_trend > 0 else 0.5)
    adjusted_short_term_trend = short_term_trend * (1.2 if short_term_trend > 0 else 0.8)
    adjusted_medium_term_trend = medium_term_trend * (1.1 if medium_term_trend > 0 else 0.9)
    adjusted_long_term_trend = long_term_trend * (1.3 if long_term_trend > 0 else 0.7)
    if bullish_market:
        adjusted_long_term_trend *= 1.5
    elif bearish_market:
        adjusted_long_term_trend *= 0.5

    adjusted_volatility = volatility_20 * (1.4 if high_volatility else 0.6)
    adjusted_relative_strength = relative_strength * (1.5 if strong_rs else 0.5)

    # Final Alpha Factor
    alpha_factor = (
        adjusted_high_low_spread +
        adjusted_short_term_trend +
        adjusted_medium_term_trend +
        adjusted_long_term_trend +
        adjusted_volatility +
        adjusted_relative_strength
    )

    return alpha_factor
```
