```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_positive = (volume_trend > 0).astype(int)
    
    # Calculate Short-Term Price Trend (5-day EMA)
    close_5d_ema = df['close'].ewm(span=5, adjust=False).mean()
    short_term_trend = df['close'] - close_5d_ema
    short_term_trend_positive = (short_term_trend > 0).astype(int)
    
    # Calculate Long-Term Price Trend (20-day EMA)
    close_20d_ema = df['close'].ewm(span=20, adjust=False).mean()
    long_term_trend = df['close'] - close_20d_ema
    long_term_trend_positive = (long_term_trend > 0).astype(int)
    
    # Integrate Relative Strength
    relative_strength = close_5d_ema / close_20d_ema
    relative_strength_above_1 = (relative_strength > 1).astype(int)
    
    # Adaptive Volatility
    close_20d_std = df['close'].rolling(window=20).std()
    close_20d_ma = df['close'].rolling(window=20).mean()
    volatility = close_20d_std / close_20d_ma
    volatility_above_1 = (volatility > 1).astype(int)
    
    # Dynamic Weight Adjustment
    relative_strength_weight = np.where(relative_strength > 1, 1.5, 0.5)
    volatility_weight = np.where(volatility > 1, 1.2, 0.8)
    
    # Combine Spread, Volume, Price Trends, and Dynamic Weights
    spread_adjusted = high_low_spread * np.where(volume_trend_positive, 1.5, 0.5)
    short_term_trend_adjusted = short_term_trend * np.where(short_term_trend_positive, 1.2, 0.8)
    long_term_trend_adjusted = long_term_trend * np.where(long_term_trend_positive, 1.3, 0.7)
    
    # Final Alpha Factor
    alpha_factor = (
        spread_adjusted +
        short_term_trend_adjusted +
        long_term_trend_adjusted
    ) * relative_strength_weight * volatility_weight
    
    return alpha_factor
```
