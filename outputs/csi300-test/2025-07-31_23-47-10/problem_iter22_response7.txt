import numpy as np

    # Calculate High-Low Spread
    df['High-Low_Spread'] = df['high'] - df['low']

    df['Daily_Volume_Trend'] = df['volume'] - df['Vol_10MA']
    df['Volume_Trend'] = np.where(df['Daily_Volume_Trend'] > 0, 1.5, 0.5)

    df['Short_Term_Trend'] = np.where(df['Short_Term_Price_Trend'] > 0, 1.2, 0.8)

    df['Close_30EMA'] = df['close'].ewm(span=30, adjust=False).mean()
    df['Medium_Term_Price_Trend'] = df['close'] - df['Close_30EMA']
    df['Medium_Term_Trend'] = np.where(df['Medium_Term_Price_Trend'] > 0, 1.1, 0.9)

    df['Long_Term_Trend'] = np.where(df['Long_Term_Price_Trend'] > 0, 1.3, 0.7)

    df['Low_Vol'] = df['Vol_20STD'] < df['Vol_20STD'].quantile(0.25)
