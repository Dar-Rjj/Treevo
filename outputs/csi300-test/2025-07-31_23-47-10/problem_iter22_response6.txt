```python
import pandas as pd

def heuristics_v2(df):
    """
    This function takes a pandas DataFrame with columns (open, high, low, close, amount, volume) and computes a novel alpha factor.
    The alpha factor is based on the following heuristics:
    1. The difference between the high and low prices is used to measure intraday volatility.
    2. The ratio of the closing price to the opening price is used to gauge the day's trend.
    3. The volume is normalized by the mean and standard deviation of the volume over the past 30 days to account for liquidity.
    4. The factor is computed as a weighted sum of these three components.
    
    :param df: pandas DataFrame with columns (open, high, low, close, amount, volume)
    :return: pandas Series with the factor values
    """
    # Compute intraday volatility
    intraday_volatility = df['high'] - df['low']
    
    # Compute the day's trend
    day_trend = df['close'] / df['open']
    
    # Normalize the volume
    volume_mean = df['volume'].rolling(window=30).mean()
    volume_std = df['volume'].rolling(window=30).std()
    normalized_volume = (df['volume'] - volume_mean) / volume_std
    
    # Compute the alpha factor
    alpha_factor = 0.5 * intraday_volatility + 0.3 * day_trend + 0.2 * normalized_volume
    
    return alpha_factor
```
