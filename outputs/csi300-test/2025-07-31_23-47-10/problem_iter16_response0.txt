import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'].rolling(window=10).mean() - df['volume']
    volume_trend_direction = np.where(volume_trend > 0, 1, -1)
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    short_term_trend_direction = np.where(short_term_trend > 0, 1, -1)
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    long_term_trend_direction = np.where(long_term_trend > 0, 1, -1)
    
    # Calculate Adaptive Volatility
    rolling_std = df['close'].rolling(window=20).std()
    ema_slope = (ema_10 - ema_10.shift(1)) / ema_10.shift(1) * 100
    adaptive_volatility = np.where(ema_slope > 0.5, rolling_std * 1.2, rolling_std * 0.8)
    volatility_direction = np.where(adaptive_volatility > adaptive_volatility.mean(), 1, -1)
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread * (np.where(volume_trend > 0, 1.5, 0.5))
    
    # Incorporate Short-Term Price Trend
    adjusted_spread *= np.where(short_term_trend > 0, 1.2, 0.8)
    
    # Incorporate Long-Term Price Trend
    adjusted_spread *= np.where(long_term_trend > 0, 1.3, 0.7)
    
    # Incorporate Adaptive Volatility
    adjusted_spread *= np.where(volatility_direction == 1, 1.4, 0.6)
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    
    return pd.Series(alpha_factor, index=df.index)
