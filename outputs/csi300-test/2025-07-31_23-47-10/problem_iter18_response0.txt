```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    
    # Determine if positive or negative trend
    volume_trend_pos = (volume_trend > 0).astype(float)
    
    # Calculate Short-Term Price Trend
    close_10d_ema = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - close_10d_ema
    
    # Determine if positive or negative trend
    short_term_trend_pos = (short_term_trend > 0).astype(float)
    
    # Calculate Medium-Term Price Trend
    close_30d_ema = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - close_30d_ema
    
    # Determine if positive or negative trend
    medium_term_trend_pos = (medium_term_trend > 0).astype(float)
    
    # Calculate Long-Term Price Trend
    close_50d_ema = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - close_50d_ema
    
    # Determine if positive or negative trend
    long_term_trend_pos = (long_term_trend > 0).astype(float)
    
    # Calculate Dynamic Volatility
    volatility = df['close'].rolling(window=10).std()
    volatility_high = (volatility > volatility.quantile(0.5)).astype(float)
    
    # Integrate Momentum and Relative Strength
    close_5d_ema = df['close'].ewm(span=5, adjust=False).mean()
    close_20d_ema = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = close_5d_ema / close_20d_ema
    relative_strength_pos = (relative_strength > 1).astype(float)
    
    # Combine Spread, Volume, Multi-Period Price Trends, and Volatility
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_pos == 1] *= 1.5
    adjusted_spread[volume_trend_pos == 0] *= 0.5
    
    adjusted_spread[short_term_trend_pos == 1] *= 1.2
    adjusted_spread[short_term_trend_pos == 0] *= 0.8
    
    adjusted_spread[medium_term_trend_pos == 1] *= 1.1
    adjusted_spread[medium_term_trend_pos == 0] *= 0.9
    
    adjusted_spread[long_term_trend_pos == 1] *= 1.3
    adjusted_spread[long_term_trend_pos == 0] *= 0.7
    
    adjusted_spread[volatility_high == 1] *= 1.4
    adjusted_spread[volatility_high == 0] *= 0.6
    
    adjusted_spread[relative_strength_pos == 1] *= 1.7
    adjusted_spread[relative_strength_pos == 0] *= 0.3
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    return alpha_factor
```
