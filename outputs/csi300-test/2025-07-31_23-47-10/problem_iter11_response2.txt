```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']

    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10

    # Calculate Price Trend using EMA
    close_ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    price_trend = df['close'] - close_ema_20

    # Adjust for positive and negative trends
    adjusted_volume_trend = volume_trend.apply(lambda x: 1.5 if x > 0 else 0.5)
    adjusted_price_trend = price_trend.apply(lambda x: 1.2 if x > 0 else 0.8)

    # Combine Spread, Volume, and Price Trends
    combined_factor = (high_low_spread * adjusted_volume_trend * adjusted_price_trend)

    return combined_factor
```
