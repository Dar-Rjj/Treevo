import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Long-Term Price Reversal
    long_term_reversal = df['close'] - df['close'].shift(20)
    
    # Combine Momentum and Reversal
    combined_momentum_reversal = (short_term_momentum + long_term_reversal) * 0.5
    
    # Calculate Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(10)
    
    # Integrate Volume Trend
    integrated_volume_trend = volume_trend / df['volume'].rolling(window=10).mean()
    
    # Generate Final Alpha Factor
    final_alpha_factor = combined_momentum_reversal * integrated_volume_trend
    
    return final_alpha_factor
