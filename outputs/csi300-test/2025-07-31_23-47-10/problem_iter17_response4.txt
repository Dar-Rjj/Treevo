```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_ma_10 = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_ma_10
    
    # Calculate Short-Term Price Trend (EMA over 10 days)
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    short_term_trend = df['close'] - ema_10
    
    # Calculate Medium-Term Price Trend (EMA over 30 days)
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    medium_term_trend = df['close'] - ema_30
    
    # Calculate Long-Term Price Trend (EMA over 50 days)
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    long_term_trend = df['close'] - ema_50
    
    # Calculate Dynamic Volatility (Standard Deviation over 10 days)
    volatility = df['close'].rolling(window=10).std()
    volatility_high = volatility > volatility.quantile(0.5)
    
    # Integrate Momentum and Relative Strength
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = ema_5 / ema_20
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend > 0] *= 1.5
    adjusted_spread[volume_trend <= 0] *= 0.5
    
    # Incorporate Short-Term Price Trend
    adjusted_spread[short_term_trend > 0] *= 1.2
    adjusted_spread[short_term_trend <= 0] *= 0.8
    
    # Incorporate Medium-Term Price Trend
    adjusted_spread[medium_term_trend > 0] *= 1.1
    adjusted_spread[medium_term_trend <= 0] *= 0.9
    
    # Incorporate Long-Term Price Trend
    adjusted_spread[long_term_trend > 0] *= 1.3
    adjusted_spread[long_term_trend <= 0] *= 0.7
    
    # Incorporate Dynamic Volatility
    adjusted_spread[volatility_high] *= 1.4
    adjusted_spread[~volatility_high] *= 0.6
    
    # Incorporate Relative Strength
    adjusted_spread[relative_strength > 1] *= 1.5
    adjusted_spread[relative_strength <= 1] *= 0.5
    
    # Final Alpha Factor
    alpha_factor = adjusted_spread
    return alpha_factor
```
