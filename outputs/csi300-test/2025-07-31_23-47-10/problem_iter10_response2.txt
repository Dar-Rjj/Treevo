```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_trend = df['volume'] - df['volume'].rolling(window=10).mean()
    volume_trend_positive = volume_trend > 0
    
    # Calculate Price Trend
    close_ema = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - close_ema
    price_trend_positive = price_trend > 0
    
    # Adjust High-Low Spread by Volume Trend
    adjusted_spread = high_low_spread.copy()
    adjusted_spread[volume_trend_positive] *= 1.5
    adjusted_spread[~volume_trend_positive] *= 0.5
    
    # Incorporate Price Trend
    final_factor = adjusted_spread.copy()
    final_factor[price_trend_positive] *= 1.2
    final_factor[~price_trend_positive] *= 0.8
    
    return final_factor
```
