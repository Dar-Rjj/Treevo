```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate 10-day moving average of volume
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    
    # Calculate Daily Volume Trend
    daily_volume_trend = df['volume'] - volume_10d_ma
    
    # Adjust for positive and negative trends
    adjusted_trend = daily_volume_trend.apply(lambda x: 1.5 if x > 0 else 0.5)
    
    # Combine Spread and Volume Trend
    factor_values = high_low_spread * daily_volume_trend * adjusted_trend
    
    return factor_values
```
