import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Daily Volume Trend
    volume_10d_ma = df['volume'].rolling(window=10).mean()
    volume_trend = df['volume'] - volume_10d_ma
    volume_trend_direction = np.where(volume_trend > 0, 'positive', 'negative')
    
    # Adjust for positive and negative trends
    adjusted_volume_trend = np.where(volume_trend_direction == 'positive', 1.5, 0.5)
    
    # Calculate Price Trend
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    price_trend = df['close'] - ema_10d
    price_trend_direction = np.where(price_trend > 0, 'positive', 'negative')
    
    # Adjust for positive and negative trends
    adjusted_price_trend = np.where(price_trend_direction == 'positive', 1.2, 0.8)
    
    # Combine Spread, Volume, and Price Trends
    adjusted_spread = high_low_spread * adjusted_volume_trend
    combined_alpha_factor = (adjusted_spread + adjusted_price_trend) / 2  # Averaging the two factors
    
    return combined_alpha_factor
