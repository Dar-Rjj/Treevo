import pandas as pd

def heuristics_v2(df):
    close_roc_20d = df['close'].pct_change(20)
    sum_amount_10d = df['amount'].rolling(window=10).sum()
    total_avg_amount = df['amount'].mean()
    rel_diff_amount = (sum_amount_10d - total_avg_amount) / total_avg_amount
    heuristics_matrix = (close_roc_20d + rel_diff_amount) / 2
    return heuristics_matrix
