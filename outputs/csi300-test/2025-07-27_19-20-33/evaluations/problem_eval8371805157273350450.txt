import pandas as pd

def heuristics_v2(df):
    close_volatility = df['close'].rolling(window=20).std()
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    rel_diff_vol = (df['volume'] - avg_vol_10d) / avg_vol_10d
    heuristics_matrix = (close_volatility + rel_diff_vol) / 2
    return heuristics_matrix
