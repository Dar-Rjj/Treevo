def heuristics_v2(df):
    median_true_range_15d = (df['high'] - df['low']).rolling(window=15).median()
    sum_volume_7d = df['volume'].rolling(window=7).sum()
    total_avg_volume = df['volume'].mean()
    rel_diff_volume = (sum_volume_7d - total_avg_volume) / total_avg_volume
    heuristics_matrix = (median_true_range_15d + rel_diff_volume) / 2
    return heuristics_matrix
