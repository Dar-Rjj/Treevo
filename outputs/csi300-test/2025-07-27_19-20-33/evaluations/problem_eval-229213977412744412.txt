def heuristics_v2(df):
    ema_range_14d = (df['high'] - df['low']).ewm(span=14, adjust=False).mean()
    sum_vol_7d = df['volume'].rolling(window=7).sum()
    total_max_vol = df['volume'].max()
    ratio_vol = sum_vol_7d / total_max_vol
    heuristics_matrix = (ema_range_14d + ratio_vol) / 2
    return heuristics_matrix
