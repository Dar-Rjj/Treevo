def heuristics_v2(df):
    def compute_alpha(row):
        return (row['high'] - row['low']) / row['close']

    # Compute daily alpha factor values
    df['alpha_factor'] = df.apply(compute_alpha, axis=1)
    
    # Calculate the 5-day moving average of the daily alpha factors
    five_day_avg = df['alpha_factor'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the close prices
    twenty_day_close_avg = df['close'].rolling(window=20).mean()
    
    # Generate the heuristic matrix (factor values)
    heuristics_matrix = (five_day_avg / twenty_day_close_avg).rename('heuristic_value')
    
    return heuristics_matrix
