def heuristics_v2(df):
    median_range_20d = (df['high'] - df['low']).rolling(window=20).median()
    avg_vol_14d = df['volume'].rolling(window=14).mean()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (avg_vol_14d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (median_range_20d + rel_diff_vol) / 2
    return heuristics_matrix
