import pandas as pd

def heuristics_v2(df):
    median_range_10d = (df['high'] - df['low']).rolling(window=10).median()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    total_median_vol = df['volume'].median()
    rel_diff_vol = (sum_vol_5d - total_median_vol) / total_median_vol
    heuristics_matrix = (median_range_10d + rel_diff_vol) / 2
    return heuristics_matrix
