def heuristics_v2(df):
    std_close_15d = df['close'].rolling(window=15).std()
    sum_amount_7d = df['amount'].rolling(window=7).sum()
    total_median_amount = df['amount'].median()
    rel_diff_amount = (sum_amount_7d - total_median_amount) / total_median_amount
    heuristics_matrix = (std_close_15d + rel_diff_amount) / 2
    return heuristics_matrix
