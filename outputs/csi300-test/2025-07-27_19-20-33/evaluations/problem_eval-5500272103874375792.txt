def heuristics_v2(df):
    avg_true_range_10d = (df['high'] - df['low']).rolling(window=10).mean()
    sum_amount_5d = df['amount'].rolling(window=5).sum()
    total_avg_amount = df['amount'].mean()
    rel_diff_amount = (sum_amount_5d - total_avg_amount) / total_avg_amount
    heuristics_matrix = (avg_true_range_10d * 0.7 + rel_diff_amount * 0.3)
    return heuristics_matrix
