def heuristics_v2(df):
    mean_range_10d = (df['high'] - df['low']).rolling(window=10).mean()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_5d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (mean_range_10d + rel_diff_vol) / 2
    return heuristics_matrix
