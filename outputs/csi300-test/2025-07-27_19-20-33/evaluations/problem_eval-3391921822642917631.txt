def heuristics_v2(df):
    std_range_15d = (df['high'] - df['low']).rolling(window=15).std()
    avg_vol_7d = df['volume'].rolling(window=7).mean()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (avg_vol_7d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (std_range_15d + rel_diff_vol) / 2
    return heuristics_matrix
