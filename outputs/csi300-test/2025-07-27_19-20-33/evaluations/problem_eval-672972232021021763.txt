def heuristics_v2(df):
    avg_range_20d = (df['high'] - df['low']).rolling(window=20).mean()
    sum_vol_10d = df['volume'].rolling(window=10).sum()
    sum_vol_30d = df['volume'].rolling(window=30).sum()
    rel_diff_vol = (sum_vol_10d - sum_vol_30d) / sum_vol_30d
    heuristics_matrix = 0.6 * avg_range_20d + 0.4 * rel_diff_vol
    return heuristics_matrix
