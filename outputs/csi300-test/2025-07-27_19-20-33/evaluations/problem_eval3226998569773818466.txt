def heuristics_v2(df):
    # Calculate the Rate of Change (ROC) for the closing price with a 14-period lookback
    roc = df['close'].pct_change(periods=14)
    
    # Calculate the Money Flow Multiplier
    mf_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    
    # Filter out potential divide-by-zero errors
    mf_multiplier = mf_multiplier.replace([np.inf, -np.inf], np.nan).fillna(0)
    
    # Calculate the Money Flow Volume
    mf_volume = mf_multiplier * df['volume']
    
    # Compute the Accumulation/Distribution Line
    adl = mf_volume.cumsum()
    
    # Compute the heuristic factor as a weighted sum of ROC and ADL
    heuristics_matrix = 0.7 * roc + 0.3 * (adl - adl.shift(1))  # Use the daily change in ADL
    
    return heuristics_matrix
