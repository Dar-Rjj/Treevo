def heuristics_v2(df):
    avg_range_10d = (df['high'] - df['low']).rolling(window=10).mean()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    total_avg_vol = df['volume'].mean()
    log_diff_vol = np.log(sum_vol_5d + 1) - np.log(total_avg_vol + 1)
    heuristics_matrix = 0.7 * avg_range_10d + 0.3 * log_diff_vol
    return heuristics_matrix
