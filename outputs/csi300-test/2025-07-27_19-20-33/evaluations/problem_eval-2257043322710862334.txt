def heuristics_v2(df):
    avg_true_range_10d = (df['high'] - df['low']).rolling(window=10).mean()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    rel_diff_vol = (sum_vol_5d - avg_vol_20d) / avg_vol_20d
    heuristics_matrix = (0.7 * avg_true_range_10d + 0.3 * rel_diff_vol)
    return heuristics_matrix
