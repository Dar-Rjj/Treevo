import pandas as pd

def heuristics_v2(df):
    ema_range_20d = (df['high'] - df['low']).ewm(span=20, adjust=False).mean()
    ema_vol_10d = df['volume'].ewm(span=10, adjust=False).mean()
    total_ema_vol = df['volume'].ewm(span=len(df), adjust=False).mean()
    rel_diff_vol = (ema_vol_10d - total_ema_vol) / total_ema_vol
    heuristics_matrix = (ema_range_20d + rel_diff_vol) / 2
    return heuristics_matrix
