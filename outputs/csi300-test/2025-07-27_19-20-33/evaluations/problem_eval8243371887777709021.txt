def heuristics_v2(df):
    weighted_price_change_10d = (df['close'] - df['open']).rolling(window=10).mean()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_5d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (weighted_price_change_10d + rel_diff_vol) / 2
    return heuristics_matrix
