import pandas as pd

def heuristics_v2(df):
    ema_close_20d = df['close'].ewm(span=20, adjust=False).mean()
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    rel_diff_vol = (df['volume'] - avg_vol_10d) / avg_vol_10d
    heuristics_matrix = (ema_close_20d + rel_diff_vol) / 2
    return heuristics_matrix
