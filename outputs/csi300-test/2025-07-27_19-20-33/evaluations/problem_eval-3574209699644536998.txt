import pandas as pd

def heuristics_v2(df):
    close_momentum = df['close'].pct_change(10)
    max_vol_5d = df['volume'].rolling(window=5).max()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (max_vol_5d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (close_momentum + rel_diff_vol) / 2
    return heuristics_matrix
