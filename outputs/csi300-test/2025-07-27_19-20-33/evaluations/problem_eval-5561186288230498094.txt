def heuristics_v2(df):
    mean_range_14d = (df['high'] - df['low']).rolling(window=14).mean()
    sum_vol_7d = df['volume'].rolling(window=7).sum()
    total_mean_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_7d - total_mean_vol) / total_mean_vol
    heuristics_matrix = (mean_range_14d + rel_diff_vol) / 2
    return heuristics_matrix
