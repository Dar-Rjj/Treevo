import pandas as pd

def heuristics_v2(df):
    close_momentum = df['close'].pct_change(20)
    avg_vol_10d = df['volume'].rolling(window=10).mean()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (avg_vol_10d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (close_momentum + rel_diff_vol) / 2
    return heuristics_matrix
