def heuristics_v2(df):
    avg_true_range_14d = (df['high'] - df['low']).rolling(window=14).mean()
    sum_amount_7d = df['amount'].rolling(window=7).sum()
    total_avg_amount = df['amount'].mean()
    rel_diff_amount = (sum_amount_7d - total_avg_amount) / total_avg_amount
    heuristics_matrix = (avg_true_range_14d + rel_diff_amount) / 2
    return heuristics_matrix
