def heuristics_v2(df):
    avg_range_15d = (df['high'] - df['low']).rolling(window=15).mean()
    sum_vol_7d = df['volume'].rolling(window=7).sum()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_7d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (avg_range_15d + rel_diff_vol) / 2
    return heuristics_matrix
