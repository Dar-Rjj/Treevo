def heuristics_v2(df):
    avg_range_20d = (df['high'] - df['low']).rolling(window=20).mean()
    sum_vol_10d = df['volume'].rolling(window=10).sum()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_10d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (avg_range_20d + rel_diff_vol) / 2
    return heuristics_matrix
