import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    daily_return = df['close'].pct_change()
    
    # Calculate volatility (standard deviation of daily returns over the last 30 days)
    volatility = daily_return.rolling(window=30).std()
    
    # Calculate price momentum (difference between current close and 5-day moving average of close)
    momentum = df['close'] - df['close'].rolling(window=5).mean()
    
    # Calculate volume trend (percentage change in volume)
    volume_trend = df['volume'].pct_change()
    
    # Dynamically adjust weights based on the recent trend of daily returns
    weight_momentum = (daily_return > 0).rolling(window=30).sum() / 30
    weight_volatility = (1 - weight_momentum)
    weight_volume = (volume_trend > 0).rolling(window=30).sum() / 30
    
    # Compute the heuristic factor
    heuristics_matrix = (weight_momentum * momentum) + (weight_volatility * volatility) + (weight_volume * volume_trend)
    
    # Return the resulting series
    return heuristics_matrix
