def heuristics_v2(df):
    weighted_avg_range_20d = ((df['high'] - df['low']) * df['volume']).rolling(window=20).sum() / df['volume'].rolling(window=20).sum()
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    rel_diff_vol = (df['volume'] - avg_vol_20d) / avg_vol_20d
    heuristics_matrix = (weighted_avg_range_20d + rel_diff_vol) / 2
    return heuristics_matrix
