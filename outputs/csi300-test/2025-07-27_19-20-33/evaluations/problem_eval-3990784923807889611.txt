import pandas as pd

def heuristics_v2(df):
    price_change_ema_20d = (df['close'] - df['open']).ewm(span=20, adjust=False).mean()
    max_vol_30d = df['volume'].rolling(window=30).max()
    vol_ratio = df['volume'] / max_vol_30d
    heuristics_matrix = (price_change_ema_20d + vol_ratio) / 2
    return heuristics_matrix
