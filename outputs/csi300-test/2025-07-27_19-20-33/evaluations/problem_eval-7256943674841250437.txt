def heuristics_v2(df):
    median_range_15d = (df['high'] - df['low']).rolling(window=15).median()
    max_vol_5d = df['volume'].rolling(window=5).max()
    total_median_vol = df['volume'].median()
    rel_diff_vol = (max_vol_5d - total_median_vol) / total_median_vol
    heuristics_matrix = (median_range_15d + rel_diff_vol) / 2
    return heuristics_matrix
