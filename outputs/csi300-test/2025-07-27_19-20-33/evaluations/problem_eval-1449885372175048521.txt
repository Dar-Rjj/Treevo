def heuristics_v2(df):
    mean_range_14d = (df['high'] - df['low']).rolling(window=14).mean()
    avg_vol_7d = df['volume'].rolling(window=7).mean()
    total_median_vol = df['volume'].median()
    rel_diff_vol = (avg_vol_7d - total_median_vol) / total_median_vol
    heuristics_matrix = (mean_range_14d + rel_diff_vol) / 2
    return heuristics_matrix
