import pandas as pd

def heuristics_v2(df):
    weighted_avg_close_10d = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_5d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (weighted_avg_close_10d + rel_diff_vol) / 2
    return heuristics_matrix
