import pandas as pd

def heuristics_v2(df):
    std_range_10d = (df['high'] - df['low']).rolling(window=10).std()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    total_std_vol = df['volume'].std()
    rel_diff_vol = (sum_vol_5d - total_std_vol) / total_std_vol
    heuristics_matrix = (std_range_10d + rel_diff_vol) / 2
    return heuristics_matrix
