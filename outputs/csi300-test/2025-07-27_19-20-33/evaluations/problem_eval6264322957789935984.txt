import pandas as pd

def heuristics_v2(df):
    avg_return_30d = df['close'].pct_change().rolling(window=30).mean()
    std_vol_10d = df['volume'].rolling(window=10).std()
    total_avg_vol = df['volume'].mean()
    vol_ratio = std_vol_10d / total_avg_vol
    heuristics_matrix = (avg_return_30d + vol_ratio) / 2
    return heuristics_matrix
