import pandas as pd

def heuristics_v2(df):
    avg_true_range_10d = df[['high', 'low', 'close']].apply(lambda x: x - x.shift(1)).abs().max(axis=1).rolling(window=10).mean()
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (avg_vol_5d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (0.3 * avg_true_range_10d + 0.7 * rel_diff_vol)
    return heuristics_matrix
