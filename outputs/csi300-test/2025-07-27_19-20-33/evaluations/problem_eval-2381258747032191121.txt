def heuristics_v2(df):
    mean_range_20d = (df['high'] - df['low']).rolling(window=20).mean()
    sum_vol_10d = df['volume'].rolling(window=10).sum()
    total_mean_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_10d - total_mean_vol) / total_mean_vol
    heuristics_matrix = (mean_range_20d + rel_diff_vol) / 2
    return heuristics_matrix
