import pandas as pd

def heuristics_v2(df):
    daily_return = df['close'].pct_change()
    ema_20d = daily_return.ewm(span=20, adjust=False).mean()
    vol_std_14d = df['volume'].rolling(window=14).std()
    total_avg_vol = df['volume'].mean()
    rel_std_vol = (vol_std_14d / total_avg_vol)
    heuristics_matrix = (ema_20d + rel_std_vol) / 2
    return heuristics_matrix
