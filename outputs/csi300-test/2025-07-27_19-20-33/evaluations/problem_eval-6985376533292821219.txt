import pandas as pd

def heuristics_v2(df):
    ewm_range_15d = (df['high'] - df['low']).ewm(span=15).mean()
    ewm_vol_7d = df['volume'].ewm(span=7).mean()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (ewm_vol_7d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (ewm_range_15d + rel_diff_vol) / 2
    return heuristics_matrix
