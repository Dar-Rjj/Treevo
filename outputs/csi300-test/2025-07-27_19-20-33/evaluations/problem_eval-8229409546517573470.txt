def heuristics_v2(df):
    ema_true_range_10d = (df['high'] - df['low']).ewm(span=10).mean()
    ema_amount_5d = df['amount'].ewm(span=5).mean()
    total_avg_amount = df['amount'].mean()
    rel_diff_amount = (ema_amount_5d - total_avg_amount) / total_avg_amount
    heuristics_matrix = (ema_true_range_10d + rel_diff_amount) / 2
    return heuristics_matrix
