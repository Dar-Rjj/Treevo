import pandas as pd

def heuristics_v2(df):
    close_ema = df['close'].ewm(span=10, adjust=False).mean()
    close_ema_momentum = close_ema.pct_change(10)
    avg_vol_5d_ema = df['volume'].ewm(span=5, adjust=False).mean()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol_ema = (avg_vol_5d_ema - total_avg_vol) / total_avg_vol
    heuristics_matrix = (close_ema_momentum + rel_diff_vol_ema) / 2
    return heuristics_matrix
