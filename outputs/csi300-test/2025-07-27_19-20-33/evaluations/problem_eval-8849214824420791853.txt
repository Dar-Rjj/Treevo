import pandas as pd

def heuristics_v2(df):
    ema_range_15d = (df['high'] - df['low']).ewm(span=15, adjust=False).mean()
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    rel_diff_vol = (df['volume'] - avg_vol_30d) / avg_vol_30d
    heuristics_matrix = (ema_range_15d + rel_diff_vol) / 2
    return heuristics_matrix
