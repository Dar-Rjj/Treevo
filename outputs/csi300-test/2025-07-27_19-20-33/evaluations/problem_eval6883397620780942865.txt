import pandas as pd

def heuristics_v2(df):
    ema_20d = df['close'].ewm(span=20, adjust=False).mean()
    high_15d = df['high'].rolling(window=15).max()
    low_15d = df['low'].rolling(window=15).min()
    high_low_ratio = high_15d / low_15d
    heuristics_matrix = (ema_20d + high_low_ratio) / 2
    return heuristics_matrix
