import pandas as pd

def heuristics_v2(df):
    # Calculate volatility factor
    df['Volatility'] = (df['high'] - df['low']) / df['close']
    # Calculate momentum factor using a 5-day moving average
    df['Momentum'] = df['close'] / df['close'].rolling(window=5).mean()
    # Combine the factors into a matrix
    heuristics_matrix = pd.concat([df['Volatility'], df['Momentum']], axis=1)
    # Return the first column as a Series for simplicity
    return heuristics_matrix
