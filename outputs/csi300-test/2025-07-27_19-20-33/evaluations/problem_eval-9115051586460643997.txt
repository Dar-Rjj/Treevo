import pandas as pd

def heuristics_v2(df):
    ema_range_10d = (df['high'] - df['low']).ewm(span=10, adjust=False).mean()
    weighted_vol_5d = (df['volume'].rolling(window=5).sum() * 0.2)
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (weighted_vol_5d - total_avg_vol) / total_avg_vol
    heuristics_matrix = (ema_range_10d + rel_diff_vol) / 2
    return heuristics_matrix
