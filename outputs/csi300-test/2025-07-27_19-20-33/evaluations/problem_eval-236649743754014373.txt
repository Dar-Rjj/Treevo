import pandas as pd

def heuristics_v2(df):
    weighted_prices = (df['close'] * (1 / df['volume'])).rolling(window=20).mean()
    daily_return = df['close'].pct_change()
    ema_daily_return = daily_return.ewm(span=20, adjust=False).mean()
    heuristics_matrix = (weighted_prices + ema_daily_return) / 2
    return heuristics_matrix
