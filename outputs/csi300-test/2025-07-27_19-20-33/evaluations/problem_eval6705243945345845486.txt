import pandas as pd

def heuristics_v2(df):
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    ema_return_20d = df['daily_return'].ewm(span=20, adjust=False).mean()
    max_vol_30d = df['volume'].rolling(window=30).max()
    vol_ratio = df['volume'] / max_vol_30d
    heuristics_matrix = (ema_return_20d + vol_ratio) / 2
    return heuristics_matrix
