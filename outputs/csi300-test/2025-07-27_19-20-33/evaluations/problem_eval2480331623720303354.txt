def heuristics_v2(df):
    mean_abs_diff_10d = (df['close'] - df['open']).abs().rolling(window=10).mean()
    sum_vol_5d = df['volume'].rolling(window=5).sum()
    total_mean_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_5d - total_mean_vol) / total_mean_vol
    heuristics_matrix = (mean_abs_diff_10d + rel_diff_vol) / 2
    return heuristics_matrix
