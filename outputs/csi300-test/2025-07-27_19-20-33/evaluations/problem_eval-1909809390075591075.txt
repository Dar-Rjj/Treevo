def heuristics_v2(df):
    ema_true_range_10d = (df['high'] - df['low']).ewm(span=10, adjust=False).mean()
    weights = np.arange(1, 6)  # Weights for the last 5 days
    weighted_sum_amount_5d = (df['amount'].rolling(window=5).apply(lambda x: (x * (6 - np.arange(1, 6))).sum(), raw=True) / 15)
    total_avg_amount = df['amount'].mean()
    rel_diff_amount = (weighted_sum_amount_5d - total_avg_amount) / total_avg_amount
    heuristics_matrix = (ema_true_range_10d + rel_diff_amount) / 2
    return heuristics_matrix
