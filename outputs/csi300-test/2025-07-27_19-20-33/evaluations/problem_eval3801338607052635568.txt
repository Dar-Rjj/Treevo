import pandas as pd

def heuristics_v2(df):
    # Calculate daily return
    daily_return = df['close'].pct_change()
    
    # Calculate RSI
    gain = daily_return.where(daily_return > 0, 0)
    loss = -daily_return.where(daily_return < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate positive and negative trading days
    positive_days = (daily_return > 0).rolling(window=30).sum()
    negative_days = (daily_return < 0).rolling(window=30).sum()
    
    # Compute the sentiment indicator
    sentiment_indicator = positive_days / (positive_days + negative_days)
    
    # Compute the heuristic factor
    heuristics_matrix = 0.5 * rsi + 0.5 * sentiment_indicator
    
    # Return the resulting series
    return heuristics_matrix
