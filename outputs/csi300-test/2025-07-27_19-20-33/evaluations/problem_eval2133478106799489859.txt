def heuristics_v2(df):
    avg_range_15d = (df['high'] - df['low']).rolling(window=15).mean()
    sum_vol_7d = df['volume'].rolling(window=7).sum()
    total_avg_vol = df['volume'].mean()
    rel_diff_vol = (sum_vol_7d - total_avg_vol) / total_avg_vol
    vol_std_30d = df['volume'].rolling(window=30).std().fillna(df['volume'].expanding().std())
    weight = 1 / (1 + np.exp(-vol_std_30d))
    heuristics_matrix = (avg_range_15d * weight + rel_diff_vol * (1 - weight))
    return heuristics_matrix
