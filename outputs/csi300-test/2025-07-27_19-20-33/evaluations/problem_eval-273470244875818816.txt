import pandas as pd

def heuristics_v2(df):
    close_ema = df['close'].ewm(span=10, adjust=False).mean()
    min_vol_5d = df['volume'].rolling(window=5).min()
    total_med_vol = df['volume'].median()
    rel_diff_vol = (min_vol_5d - total_med_vol) / total_med_vol
    heuristics_matrix = (close_ema + rel_diff_vol) / 2
    return heuristics_matrix
