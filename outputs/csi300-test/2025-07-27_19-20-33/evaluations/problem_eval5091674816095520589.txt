def heuristics_v2(df):
    heuristics_matrix = pd.Series(index=df.index, dtype='float64')
    # Example factor 1: Simple price momentum over the last 5 days
    heuristics_matrix['mom_5d'] = df['close'].pct_change(periods=5)
    # Example factor 2: Volatility as standard deviation of daily returns over the last 20 days
    heuristics_matrix['vol_20d'] = df['close'].pct_change().rolling(window=20).std()
    # Example factor 3: Volume relative to the 30-day average volume
    heuristics_matrix['volume_ratio_30d'] = df['volume'] / df['volume'].rolling(window=30).mean()
    return heuristics_matrix
