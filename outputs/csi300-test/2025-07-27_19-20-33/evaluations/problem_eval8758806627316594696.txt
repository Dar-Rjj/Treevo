def heuristics_v2(df):
    daily_return = (df['close'] - df['open']) / df['open']
    std_daily_return_30d = daily_return.rolling(window=30).std()
    avg_vol_30d = df['volume'].rolling(window=30).mean()
    rel_diff_vol = (df['volume'] - avg_vol_30d) / avg_vol_30d
    heuristics_matrix = (std_daily_return_30d + rel_diff_vol) / 2
    return heuristics_matrix
