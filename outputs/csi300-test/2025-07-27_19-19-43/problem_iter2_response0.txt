```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices, indicating the price movement during the day
    price_movement = df['close'] - df['open']
    
    # Calculate the average of the high and low prices, giving a sense of the midpoint of the trading range for the day
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the daily volatility as the difference between the high and low prices
    daily_volatility = df['high'] - df['low']
    
    # Calculate an intensity factor that captures the magnitude of the price change relative to the midpoint.
    # This aims to capture how significant the daily price move is compared to the day's trading range.
    intensity_factor = price_movement / (avg_price + 1e-7)
    
    # Calculate a normalized trading range that incorporates the daily volatility
    # This factor adjusts the trading range by the intensity of the price movement
    normalized_trading_range = daily_volatility / (avg_price + 1e-7)
    
    # Combine the intensity factor and the normalized trading range
    combined_factor = 0.5 * intensity_factor + 0.5 * normalized_trading_range
    
    return combined_factor
```
