```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns to capture the momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute the 30-day and 60-day rolling standard deviation of log returns as a measure of volatility
    df['volatility_30'] = df['log_return'].rolling(window=30).std()
    df['volatility_60'] = df['log_return'].rolling(window=60).std()
    
    # Calculate the mean reversion factor using the difference between the current close and the 60-day moving average
    df['ma_60'] = df['close'].rolling(window=60).mean()
    df['mean_reversion'] = (df['close'] - df['ma_60']) / df['ma_60']
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Incorporate market sentiment using the ratio of the amount traded to the 30-day average amount
    market_sentiment = df['amount'] / (df['amount'].rolling(window=30).mean() + 1e-7)
    
    # Combine the momentum, volatility, mean reversion, price difference to volume, volume ratio, and market sentiment factors
    alpha_factor = (
        (df['log_return'] / (df['volatility_30'] + 1e-7)) +
        0.5 * (df['mean_reversion'] / (df['volatility_60'] + 1e-7)) +
        0.3 * price_diff_to_volume +
        0.1 * volume_ratio +
        0.1 * market_sentiment
    )
    
    return alpha_factor
```
