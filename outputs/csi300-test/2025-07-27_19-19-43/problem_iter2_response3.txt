```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the percentage change in close price over the last 10 days
    momentum = df['close'].pct_change(periods=10)
    
    # Compute the daily volatility, using the standard deviation of the last 30 days' close prices
    volatility = df['close'].rolling(window=30).std()
    
    # Calculate the relative strength by comparing today's close price to the lowest close price in the last 30 days
    # This gives an idea of how strong the stock is performing compared to its recent lows
    relative_strength = (df['close'] - df['close'].rolling(window=30).min()) / df['close']
    
    # Combine the three factors. The momentum provides a measure of the trend, 
    # the volatility reflects the risk, and the relative strength indicates the current strength of the stock.
    alpha_factor = (momentum - volatility + relative_strength) * (df['close'] - df['open']) * np.log1p(df['volume'] / df['volume'].rolling(window=30).mean())
    
    return alpha_factor
```
