```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day rolling standard deviation of daily returns as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the RSI (Relative Strength Index) with a 14-day window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the MACD (Moving Average Convergence Divergence) with 12 and 26-day EMA
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    df['macd'] = ema_12 - ema_26
    df['signal_line'] = df['macd'].ewm(span=9, adjust=False).mean()
    
    # Calculate the mean reversion factor using the 50-day moving average
    df['50_day_ma'] = df['close'].rolling(window=50).mean()
    df['mean_reversion'] = df['close'] - df['50_day_ma']
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the adaptive weights based on the standard deviation of the factors
    std_daily_return = df['daily_return'].rolling(window=30).std()
    std_price_diff_to_volume = price_diff_to_volume.rolling(window=30).std()
    std_volume_ratio = volume_ratio.rolling(window=30).std()
    std_rsi = df['rsi'].rolling(window=30).std()
    std_macd_signal = (df['macd'] - df['signal_line']).rolling(window=30).std()
    std_mean_reversion = df['mean_reversion'].rolling(window=30).std()
    
    total_std = (std_daily_return + std_price_diff_to_volume + std_volume_ratio + std_rsi + std_macd_signal + std_mean_reversion)
    
    w_daily_return = std_daily_return / total_std
    w_price_diff_to_volume = std_price_diff_to_volume / total_std
    w_volume_ratio = std_volume_ratio / total_std
    w_rsi = std_rsi / total_std
    w_macd_signal = std_macd_signal / total_std
    w_mean_reversion = std_mean_reversion / total_std
    
    # Combine the factors with adaptive weights
    alpha_factor = (w_daily_return * (df['daily_return'] / (df['volatility'] + 1e-7)) +
                    w_price_diff_to_volume * price_diff_to_volume +
                    w_volume_ratio * volume_ratio +
                    w_rsi * (df['rsi'] - 50) +
                    w_macd_signal * (df['macd'] - df['signal_line']) +
                    w_mean_reversion * (df['mean_reversion'] / df['volatility']))
    
    return alpha_factor
```
