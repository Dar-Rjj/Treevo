```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 60-day rolling standard deviation of daily returns as a measure of long-term volatility
    df['long_term_volatility'] = df['daily_return'].rolling(window=60).std()
    
    # Calculate the difference between close and open prices, then divide by average volume over the last 60 days
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=60).mean() + 1e-7)
    
    # Combine the momentum and price difference to volume factors
    alpha_factor = (df['daily_return'] / (df['long_term_volatility'] + 1e-7)) + 0.8 * price_diff_to_volume
    
    return alpha_factor
```
