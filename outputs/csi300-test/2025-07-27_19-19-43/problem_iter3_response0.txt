```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily range (high - low)
    daily_range = df['high'] - df['low']
    
    # Calculate the relative strength as the close minus the low, divided by the daily range
    relative_strength = (df['close'] - df['low']) / (daily_range + 1e-7)
    
    # Calculate the 30-day rolling standard deviation of the close price for volatility
    volatility = df['close'].rolling(window=30).std()
    
    # Calculate the momentum as the difference between the current close and the 30-day moving average of the close
    momentum = df['close'] - df['close'].rolling(window=30).mean()
    
    # Combine these factors in a weighted sum, with more weight on relative strength and momentum
    alpha_factor = 0.4 * relative_strength + 0.4 * momentum + 0.2 * (1 / volatility)
    
    return alpha_factor
```
