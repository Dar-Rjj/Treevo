```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day rolling standard deviation of daily returns as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the Relative Strength Index (RSI) for a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / (avg_loss + 1e-7)
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the MACD (Moving Average Convergence Divergence)
    short_ema = df['close'].ewm(span=12, adjust=False).mean()
    long_ema = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = short_ema - long_ema
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    df['macd_histogram'] = macd_line - signal_line
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the 5-day and 10-day moving averages
    df['ma_5'] = df['close'].rolling(window=5).mean()
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the difference between the 5-day and 10-day moving averages
    ma_difference = df['ma_5'] - df['ma_10']
    
    # Calculate the price change over a 5-day period
    df['price_change_5'] = df['close'].pct_change(5)
    
    # Calculate the volume change over a 5-day period
    df['volume_change_5'] = df['volume'].pct_change(5)
    
    # Combine the momentum, volatility, RSI, MACD, price difference to volume, volume ratio, moving average difference, price change, and volume change factors
    alpha_factor = (
        (df['daily_return'] / (df['volatility'] + 1e-7)) +
        0.5 * (df['rsi'] - 50) / 50 +
        0.3 * df['macd_histogram'] +
        0.1 * price_diff_to_volume +
        0.1 * volume_ratio +
        0.2 * (ma_difference / (df['close'] + 1e-7)) +
        0.1 * df['price_change_5'] +
        0.1 * df['volume_change_5']
    )
    
    return alpha_factor
```
