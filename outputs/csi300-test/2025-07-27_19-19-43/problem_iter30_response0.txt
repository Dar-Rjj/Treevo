```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 10-day rolling standard deviation of daily returns as a measure of short-term volatility
    df['volatility'] = df['daily_return'].rolling(window=10).std()
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=10).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 10-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=10).mean() + 1e-7)
    
    # Calculate the 10-day moving average of the closing price to capture the trend
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the 3-day moving average of the closing price to capture the very short-term trend
    df['ma_3'] = df['close'].rolling(window=3).mean()
    
    # Calculate the percentage change in the 3-day moving average
    df['ma_3_change'] = df['ma_3'].pct_change()
    
    # Combine the momentum, volatility, price difference to volume, volume ratio, and trend factors
    alpha_factor = (df['daily_return'] / (df['volatility'] + 1e-7)) + \
                   0.6 * price_diff_to_volume + \
                   0.2 * volume_ratio + \
                   0.2 * df['ma_3_change']
    
    return alpha_factor
```
