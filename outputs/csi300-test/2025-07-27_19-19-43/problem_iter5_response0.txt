```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the logarithmic return to capture the momentum
    df['log_return'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Compute the 30-day rolling standard deviation of log returns as a measure of volatility
    df['volatility'] = df['log_return'].rolling(window=30).std()
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Adaptive weights based on the 30-day rolling correlation between log returns and volume
    adaptive_weights = df['log_return'].rolling(window=30).corr(df['volume']).fillna(0.5)
    
    # Combine the momentum, volatility, price difference to volume, and volume ratio factors with adaptive weights
    alpha_factor = (adaptive_weights * (df['log_return'] / (df['volatility'] + 1e-7))) + (1 - adaptive_weights) * (0.7 * price_diff_to_volume + 0.3 * volume_ratio)
    
    return alpha_factor
```
