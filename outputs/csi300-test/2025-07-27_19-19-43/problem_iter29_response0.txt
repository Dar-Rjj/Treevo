```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day rolling standard deviation of daily returns as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the difference between close and open prices, then divide by the current day's volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'] + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the 60-day moving average of the close price to integrate long-term trends
    df['long_term_trend'] = df['close'].rolling(window=60).mean()
    
    # Combine the momentum, volatility, price difference to volume, volume ratio, and long-term trend factors
    alpha_factor = (df['daily_return'] / (df['volatility'] + 1e-7)) + 0.5 * price_diff_to_volume + 0.3 * volume_ratio + 0.2 * (df['close'] / (df['long_term_trend'] + 1e-7))
    
    return alpha_factor
```
