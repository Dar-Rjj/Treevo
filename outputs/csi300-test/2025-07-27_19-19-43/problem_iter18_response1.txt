```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day and 60-day rolling standard deviation of daily returns as a measure of short-term and long-term volatility
    df['volatility_30'] = df['daily_return'].rolling(window=30).std()
    df['volatility_60'] = df['daily_return'].rolling(window=60).std()
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Incorporate high/low price information
    price_range = (df['high'] - df['low']) / df['close']
    
    # Combine the momentum, short-term and long-term volatility, price difference to volume, volume ratio, and price range factors
    alpha_factor = (df['daily_return'] / (df['volatility_30'] + 1e-7)) + 0.6 * price_diff_to_volume + 0.2 * volume_ratio + 0.2 * price_range
    
    return alpha_factor
```
