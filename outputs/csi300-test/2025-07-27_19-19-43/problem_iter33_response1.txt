```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day exponentially weighted moving average (EWMA) of daily returns for responsiveness
    ewma_daily_return = df['daily_return'].ewm(span=30, adjust=False).mean()
    
    # Calculate the difference between close and open prices, then divide by the 30-day EWMA volume
    ewma_volume = df['volume'].ewm(span=30, adjust=False).mean()
    price_diff_to_volume = (df['close'] - df['open']) / (ewma_volume + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day EWMA volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (ewma_volume + 1e-7)
    
    # Combine the EWMA momentum, price difference to volume, and volume ratio factors
    alpha_factor = ewma_daily_return + 0.7 * price_diff_to_volume + 0.3 * volume_ratio
    
    return alpha_factor
```
