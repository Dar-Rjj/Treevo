```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and the average of last 5 days' close prices
    avg_close_5_days = df['close'].rolling(window=5).mean()
    delta_close_5_days = df['close'] - avg_close_5_days
    
    # Calculate the money flow ratio using the amount and volume, indicating the strength of buying/selling
    money_flow_ratio = (df['amount'] / df['volume']).diff(1)
    
    # Combine the two signals, giving a higher weight to the relative change in close price
    alpha_factor = 0.7 * delta_close_5_days + 0.3 * money_flow_ratio

    return alpha_factor
```
