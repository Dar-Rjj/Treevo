```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day rolling standard deviation of daily returns as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the average of high, low, and open prices, then subtract the close price, and divide by the 30-day average volume
    price_diff_to_volume = (df[['high', 'low', 'open']].mean(axis=1) - df['close']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Apply a simple moving average (SMA) to the daily return for smoothing
    df['smoothed_daily_return'] = df['daily_return'].rolling(window=5).mean()
    
    # Combine the smoothed momentum, volatility, price difference to volume, and volume ratio factors
    alpha_factor = (df['smoothed_daily_return'] / (df['volatility'] + 1e-7)) + 0.7 * price_diff_to_volume + 0.3 * volume_ratio
    
    return alpha_factor
```
