```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 30-day rolling average of the daily range (high - low)
    avg_daily_range = (df['high'] - df['low']).rolling(window=30).mean()
    
    # Momentum factor: difference between close and open prices, adjusted by the average daily range
    price_diff_to_range = (df['close'] - df['open']) / (avg_daily_range + 1e-7)
    
    # Volatility factor: standard deviation of close prices over a 30-day window
    volatility = df['close'].rolling(window=30).std()
    
    # Relative strength factor: ratio of the current close to the 30-day moving average of the close
    relative_strength = df['close'] / (df['close'].rolling(window=30).mean() + 1e-7)
    
    # Volume-adjusted momentum: price difference divided by the 30-day average volume
    vol_adj_momentum = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Combine the factors in a weighted sum to create the alpha factor
    alpha_factor = 0.4 * price_diff_to_range + 0.3 * relative_strength + 0.2 * vol_adj_momentum + 0.1 * (1 / volatility)
    
    return alpha_factor
```
