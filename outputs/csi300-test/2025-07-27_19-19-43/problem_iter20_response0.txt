```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day rolling standard deviation of daily returns as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the Relative Strength Index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the MACD line and signal line
    df['ema_12'] = df['close'].ewm(span=12, adjust=False).mean()
    df['ema_26'] = df['close'].ewm(span=26, adjust=False).mean()
    df['macd_line'] = df['ema_12'] - df['ema_26']
    df['signal_line'] = df['macd_line'].ewm(span=9, adjust=False).mean()
    df['macd_histogram'] = df['macd_line'] - df['signal_line']
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the 50-day simple moving average to capture the trend
    df['sma_50'] = df['close'].rolling(window=50).mean()
    trend_factor = df['close'] / df['sma_50']
    
    # Calculate the 200-day simple moving average for long-term trend
    df['sma_200'] = df['close'].rolling(window=200).mean()
    long_trend_factor = df['close'] / df['sma_200']
    
    # Incorporate market sentiment or external economic data (example: hypothetical market sentiment score)
    if 'sentiment_score' in df.columns:
        sentiment_factor = (df['sentiment_score'] - df['sentiment_score'].mean()) / (df['sentiment_score'].std() + 1e-7)
    else:
        sentiment_factor = 0
    
    # Calculate the Aroon indicator to identify new trends
    df['aroon_up'] = df['high'].rolling(window=25).apply(lambda x: (x.argmax() + 1) / 25 * 100, raw=True)
    df['aroon_down'] = df['low'].rolling(window=25).apply(lambda x: (x.argmin() + 1) / 25 * 100, raw=True)
    aroon_factor = (df['aroon_up'] - df['aroon_down']) / 100
    
    # Calculate the On-Balance Volume (OBV) to capture buying and selling pressure
    df['obv'] = (df['close'] > df['close'].shift(1)).astype(int) * df['volume'] - (df['close'] < df['close'].shift(1)).astype(int) * df['volume']
    obv_factor = df['obv'].pct_change()
    
    # Combine the factors with optimized weights
    alpha_factor = (df['daily_return'] / (df['volatility'] + 1e-7)) + 0.5 * (100 - df['rsi']) + 0.3 * df['macd_histogram'] + 0.1 * price_diff_to_volume + 0.05 * volume_ratio + 0.2 * (trend_factor - 1) + 0.1 * (long_trend_factor - 1) + 0.2 * sentiment_factor + 0.1 * aroon_factor + 0.1 * obv_factor
    
    return alpha_factor
```
