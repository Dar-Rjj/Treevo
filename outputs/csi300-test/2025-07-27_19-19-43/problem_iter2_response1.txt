```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 30-day rolling standard deviation of daily returns as a measure of volatility
    df['volatility'] = df['daily_return'].rolling(window=30).std()
    
    # Calculate the relative strength by comparing the current close price to the 50-day moving average
    df['ma_50'] = df['close'].rolling(window=50).mean()
    df['relative_strength'] = df['close'] / df['ma_50']
    
    # Incorporate the volume influence by weighing the daily return with the log of normalized volume
    df['volume_influence'] = np.log1p(df['volume'] / df['volume'].rolling(window=30).mean())
    
    # Combine the momentum, volatility, and relative strength factors
    alpha_factor = (df['daily_return'] * df['volume_influence']) + (df['relative_strength'] / df['volatility'])
    
    return alpha_factor
```
