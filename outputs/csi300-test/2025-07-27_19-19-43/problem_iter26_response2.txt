```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return to capture the momentum
    df['daily_return'] = df['close'].pct_change()
    
    # Compute the 20-day and 60-day rolling standard deviation of daily returns as a measure of volatility
    df['volatility_20'] = df['daily_return'].rolling(window=20).std()
    df['volatility_60'] = df['daily_return'].rolling(window=60).std()
    
    # Calculate the difference between close and open prices, then divide by average volume
    price_diff_to_volume = (df['close'] - df['open']) / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the ratio of the current day's volume to the 30-day average volume, which reflects relative trading activity
    volume_ratio = df['volume'] / (df['volume'].rolling(window=30).mean() + 1e-7)
    
    # Calculate the 30-day rolling mean of the high and low prices
    df['high_low_mean_30'] = (df['high'] + df['low']).rolling(window=30).mean()
    
    # Calculate the ratio of the close price to the 30-day rolling mean of high and low prices
    close_to_high_low_mean = df['close'] / (df['high_low_mean_30'] + 1e-7)
    
    # Combine the momentum, volatility, price difference to volume, volume ratio, and close to high/low mean factors
    alpha_factor = (df['daily_return'] / (df['volatility_20'] + df['volatility_60'] + 1e-7)) + 0.6 * price_diff_to_volume + 0.2 * volume_ratio + 0.2 * close_to_high_low_mean
    
    return alpha_factor
```
