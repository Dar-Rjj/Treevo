def heuristics_v2(df):
    import pandas as pd
    
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_range
    
    # Compute the Inverse of Volume-to-Range Ratio
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    adjusted_intraday_volatility = intraday_range * inverse_volume_to_range_ratio
    
    return pd.Series(adjusted_intraday_volatility, index=df.index)
