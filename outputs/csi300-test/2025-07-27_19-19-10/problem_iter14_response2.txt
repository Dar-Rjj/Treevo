```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    prev_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / daily_range
    adjusted_intraday_high_low_range = daily_range * (1 / volume_to_range_ratio)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Incorporate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    sum_abs_price_changes = df['close'].diff().abs().rolling(window=5).sum()
    momentum_factor = price_change / sum_abs_price_changes
    combined_momentum_volatility = momentum_factor + adjusted_intraday_high_low_range
    
    # Apply SMA Trend Adjustment
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    trend_adjustment = np.where(df['close'] > ema_10, 1.2, 0.8)
    final_alpha_factor_trend_adjusted = combined_momentum_volatility * trend_adjustment
    
    # Compute Price Difference Momentum
    price_difference_momentum = df['close'] - df['close'].shift(1)
    last_5_days_price_diff_momentum = price_difference_momentum.rolling(window=5).sum()
    
    # Combine Factors
    combined_factors = (
        daily_range + 
        volume_adjusted_daily_range + 
        weighted_intraday_return + 
        last_5_days_price_diff_momentum + 
        prev_day_return + 
        adjusted_intraday_high_low_range
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * trend_adjustment
    
    return final_alpha_factor
```
