```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    prev_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / daily_range
    adjusted_intraday_range = daily_range * (1 / volume_to_range_ratio)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Incorporate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    abs_price_changes = (df['close'] - df['close'].shift(1)).abs().rolling(window=5).sum()
    momentum_factor = price_change / abs_price_changes
    combined_momentum = momentum_factor + adjusted_intraday_range
    
    # Apply Moving Average Trend Adjustment
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    trend_adjustment = np.where(df['close'] > ema_10d, 1.2, 0.8)
    
    # Compute Price Difference Momentum
    price_diff = df['close'] - df['close'].shift(1)
    price_diff_last_5_days = [price_diff.shift(i) for i in range(5)]
    
    # Calculate Exponential Moving Average (EMA) of Price Difference
    ema_price_diff = df['close'].diff().ewm(alpha=0.2, adjust=False).mean()
    
    # Introduce Price Momentum Smoothing
    sma_price_diff = df['close'].diff().rolling(window=5).mean()
    
    # Combine Factors
    combined_factors = (
        daily_range + 
        volume_adjusted_daily_range + 
        weighted_intraday_return + 
        sum(price_diff_last_5_days) + 
        ema_price_diff + 
        prev_day_return + 
        adjusted_intraday_range + 
        sma_price_diff
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * trend_adjustment
    
    return final_alpha_factor
```
