import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio and its inverse
    volume_to_range_ratio = df['volume'] / intraday_range
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    adjusted_intraday_range = intraday_range * inverse_volume_to_range_ratio
    
    # Incorporate Price Momentum
    price_change = df['close'].diff()
    sum_abs_price_changes = price_change.rolling(window=5).apply(lambda x: abs(x).sum(), raw=True)
    normalized_price_change = price_change / sum_abs_price_changes
    momentum_adjusted_volatility = adjusted_intraday_range + normalized_price_change
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Compute Volume Adjusted Daily Range
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * average_volume_5_days
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Compute Price Difference Momentum
    price_difference_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Combine Factors
    combined_factors = (
        momentum_adjusted_volatility + 
        weighted_intraday_return + 
        previous_day_return + 
        volume_adjusted_daily_range + 
        price_difference_momentum
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors
    
    return final_alpha_factor
