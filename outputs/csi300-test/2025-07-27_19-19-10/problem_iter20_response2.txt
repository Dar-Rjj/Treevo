```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio and its inverse
    volume_to_range_ratio = df['volume'] / intraday_range
    inv_volume_to_range_ratio = 1 / volume_to_range_ratio
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    adjusted_intraday_range = intraday_range * inv_volume_to_range_ratio
    
    # Incorporate Price Momentum
    price_change = df['close'].diff(1)
    abs_price_changes = price_change.abs().rolling(window=5).sum()
    normalized_price_change = price_change / abs_price_changes
    momentum_adjusted_volatility = adjusted_intraday_range + normalized_price_change
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    prev_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Compute Price Difference Momentum
    price_diff_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Combine Factors
    combined_factors = (daily_range + 
                        volume_adjusted_daily_range + 
                        weighted_intraday_return + 
                        price_diff_momentum + 
                        prev_day_return + 
                        momentum_adjusted_volatility)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors + volume_adjusted_daily_range
    
    return final_alpha_factor
```
