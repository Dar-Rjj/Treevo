```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    df['volume_adjusted_daily_range'] = df['daily_range'] * (df['volume'].rolling(window=5).sum() / 5)
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['previous_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['adjusted_intraday_high_low_range'] = df['daily_range'] * (1 / (df['volume'] / df['daily_range']))
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Incorporate Price Momentum
    abs_price_changes = (df['close'] - df['close'].shift(1)).abs()
    sum_abs_price_changes = abs_price_changes.rolling(window=5).sum()
    df['price_momentum'] = (df['close'] - df['close'].shift(1)) / sum_abs_price_changes
    
    # Apply Moving Average Trend Adjustment
    df['10_day_ema'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = df.apply(lambda x: 1.2 if x['close'] > x['10_day_ema'] else 0.8, axis=1)
    
    # Compute Price Difference Momentum
    df['price_difference_momentum'] = df['close'] - df['close'].shift(1)
    
    # Combine Factors
    df['combined_factors'] = (
        df['daily_range'] + 
        df['volume_adjusted_daily_range'] + 
        df['weighted_intraday_return'] + 
        df['price_difference_momentum'].rolling(window=5).sum().shift(1) + 
        df['previous_day_return'] + 
        df['adjusted_intraday_high_low_range']
    )
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_factors'] * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
