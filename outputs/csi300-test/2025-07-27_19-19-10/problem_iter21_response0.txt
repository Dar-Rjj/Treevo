```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Daily Range
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    df['vol_adj_daily_range'] = df['daily_range'] * (df['avg_vol_5d'] / df['volume'])
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['vol_to_range_ratio'] = df['volume'] / df['daily_range']
    df['inv_vol_to_range_ratio'] = 1 / df['vol_to_range_ratio']
    df['adj_intraday_range'] = df['daily_range'] * df['inv_vol_to_range_ratio']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['sum_abs_price_change_5d'] = df['price_change'].abs().rolling(window=5).sum()
    df['momentum_factor'] = df['price_change'] / df['sum_abs_price_change_5d']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    df['price_diff_momentum_5d'] = df['price_diff_momentum'].rolling(window=5).sum()
    
    # Calculate Exponential Moving Average (EMA) of Price Difference
    df['ema_price_diff'] = df['price_diff_momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Combine Factors
    combined_factors = (
        df['daily_range'] +
        df['vol_adj_daily_range'] +
        df['weighted_intraday_return'] +
        df['price_diff_momentum_5d'] +
        df['ema_price_diff'] +
        df['prev_day_return'] +
        df['adj_intraday_range'] +
        df['momentum_factor']
    )
    
    # Final Alpha Factor
    final_alpha = combined_factors + df['vol_adj_daily_range']
    
    return final_alpha.dropna()
```
