```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    df['avg_vol_5d'] = df['volume'].rolling(window=5).mean()
    df['vol_adj_daily_range'] = df['daily_range'] * df['avg_vol_5d']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = np.where(df['volume_change'] > 0, 1, 0)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['intraday_high_low_range'] = df['high'] - df['low']
    df['volume_to_range_ratio'] = df['volume'] / df['intraday_high_low_range']
    df['inverse_vol_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adj_intraday_high_low_range'] = df['intraday_high_low_range'] * df['inverse_vol_to_range_ratio']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['abs_price_change_5d_sum'] = df['price_change'].abs().rolling(window=5).sum()
    df['price_momentum'] = df['price_change'] / df['abs_price_change_5d_sum']
    df['combined_momentum'] = df['price_momentum'] + df['adj_intraday_high_low_range']
    
    # Apply Moving Average Trend Adjustment
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = np.where(df['close'] > df['ema_10d'], 1.2, 0.8)
    
    # Compute Price Difference Momentum
    df['price_diff_mom'] = df['close'] - df['close'].shift(1)
    df['price_diff_mom_5d'] = df['price_diff_mom'].rolling(window=5).sum()
    
    # Combine Factors
    df['combined_factors'] = (df['daily_range'] + 
                              df['vol_adj_daily_range'] + 
                              df['weighted_intraday_return'] + 
                              df['prev_day_return'] + 
                              df['price_diff_mom_5d'] + 
                              df['combined_momentum'])
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_factors'] * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
