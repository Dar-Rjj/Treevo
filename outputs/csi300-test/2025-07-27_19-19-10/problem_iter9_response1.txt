```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['intraday_range']
    
    # Multiply Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['adjusted_intraday_volatility'] = df['intraday_range'] * (1 / df['volume_to_range_ratio'])
    
    # Calculate Price Change
    df['price_change'] = df['close'].diff()
    
    # Sum of Absolute Price Changes over the last 5 days
    df['abs_price_change_sum'] = df['price_change'].abs().rolling(window=5).sum()
    
    # Divide Current Price Change by Sum
    df['momentum_factor'] = df['price_change'] / df['abs_price_change_sum']
    
    # Combine with Adjusted Intraday Volatility
    df['combined_factor'] = df['adjusted_intraday_volatility'] + df['momentum_factor']
    
    # Calculate 10-Day Simple Moving Average (SMA)
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Apply Additional Weighting Based on Moving Average
    conditions = [
        (df['close'] > df['sma_10']),
        (df['close'] <= df['sma_10'])
    ]
    choices = [1.2, 0.8]
    df['weighting'] = pd.Series(pd.np.select(conditions, choices, default=0.0), index=df.index)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] * df['weighting']
    
    return df['alpha_factor']
```
