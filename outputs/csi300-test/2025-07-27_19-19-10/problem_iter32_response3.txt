import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    df['Volume_Adjusted_Daily_Range'] = df['Daily_Range'] * (df['volume'].rolling(window=5).mean())
    
    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['Previous_Day_Return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    df['Volume_Spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['Volume_to_Range_Ratio'] = df['volume'] / df['Daily_Range']
    df['Adjusted_Intraday_High_Low_Range'] = df['Daily_Range'] * (1 / df['Volume_to_Range_Ratio'])
    
    # Weighted Intraday Return
    df['Weighted_Intraday_Return'] = df['Intraday_Return'] * df['Volume_Spike']
    
    # Incorporate Price Momentum
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Sum_Absolute_Price_Changes'] = df['Price_Change'].abs().rolling(window=5).sum()
    df['Normalized_Price_Change'] = df['Price_Change'] / df['Sum_Absolute_Price_Changes']
    df['Adjusted_Intraday_Volatility'] = df['Adjusted_Intraday_High_Low_Range'] + df['Normalized_Price_Change']
    
    # Apply Moving Average Trend Adjustment
    df['10_Day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Trend_Adjustment'] = np.where(df['close'] > df['10_Day_EMA'], 1.2, 0.8)
    
    # Compute Price Difference Momentum
    df['Price_Difference_Momentum'] = df['close'] - df['close'].shift(1)
    df['Price_Difference_Momentum_5D'] = df['Price_Difference_Momentum'].rolling(window=5).sum()
    
    # Calculate Exponential Moving Average (EMA) of Price Difference
    df['EMA_Price_Difference'] = df['Price_Difference_Momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Combine Factors
    df['Combined_Factors'] = (
        df['Daily_Range'] +
        df['Volume_Adjusted_Daily_Range'] +
        df['Weighted_Intraday_Return'] +
        df['Price_Difference_Momentum_5D'] +
        df['EMA_Price_Difference'] +
        df['Previous_Day_Return'] +
        df['Adjusted_Intraday_High_Low_Range'] +
        df['Adjusted_Intraday_Volatility']
    )
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factors'] * df['Trend_Adjustment']
    
    return df['Final_Alpha_Factor']
