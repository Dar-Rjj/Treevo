```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['intraday_high_low_range']
    
    # Adjust Intraday High-Low Range
    df['inverse_volume_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adjusted_intraday_high_low_range'] = df['intraday_high_low_range'] * df['inverse_volume_to_range_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > 0).astype(int)
    
    # Combine Factors
    df['combined_factors'] = df['daily_range'] + df['adjusted_intraday_high_low_range']
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    df['combined_factors'] += df['weighted_intraday_return'] + df['prev_day_return']
    
    # Compute Price Difference Momentum
    df['price_difference_momentum'] = df['close'] - df['close'].shift(1)
    df['price_difference_momentum_5_days'] = df['price_difference_momentum'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] + df['price_difference_momentum_5_days']
    
    return df['alpha_factor']
```
