```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_range
    adjusted_intraday_volatility = intraday_range * (1 / volume_to_range_ratio)
    
    # Incorporate Price Momentum
    price_change = df['close'].diff()
    momentum_sum = price_change.abs().rolling(window=5).sum()
    momentum_factor = price_change / momentum_sum
    combined_momentum_adjusted_volatility = adjusted_intraday_volatility + momentum_factor
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 0).astype(int)
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5_days
    
    # Final Alpha Factor
    weighted_intraday_return = intraday_return * volume_spike
    final_alpha_factor = (combined_momentum_adjusted_volatility
                          + weighted_intraday_return
                          + previous_day_return
                          + daily_range
                          + volume_adjusted_daily_range)
    
    # Apply Additional Weighting Based on Moving Average
    sma_10 = df['close'].rolling(window=10).mean()
    final_alpha_factor = final_alpha_factor.where(df['close'] > sma_10, final_alpha_factor * 0.8).where(df['close'] <= sma_10, final_alpha_factor * 1.2)
    
    return final_alpha_factor
```
