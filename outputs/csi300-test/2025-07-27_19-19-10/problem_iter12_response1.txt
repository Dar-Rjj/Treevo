```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio and its inverse
    volume_to_range_ratio = df['volume'] / intraday_range
    inv_volume_to_range_ratio = 1 / volume_to_range_ratio
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    adj_intraday_volatility = intraday_range * inv_volume_to_range_ratio
    
    # Incorporate Price Momentum
    price_change = df['close'].diff()
    abs_price_change_sum_5d = price_change.abs().rolling(window=5).sum()
    momentum_factor = price_change / abs_price_change_sum_5d
    combined_factor = adj_intraday_volatility + momentum_factor.fillna(0)
    
    # Apply SMA Trend Adjustment
    ema_10d = df['close'].ewm(span=10, adjust=False).mean()
    trend_adjustment = (df['close'] > ema_10d).astype(float) * 1.2 + (df['close'] <= ema_10d).astype(float) * 0.8
    
    final_alpha_factor = combined_factor * trend_adjustment
    
    return final_alpha_factor
```
