```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Adjust by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_range
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_volatility = intraday_range * inverse_volume_to_range_ratio
    
    # Incorporate Price Momentum
    price_change = df['close'].diff(1)
    abs_price_changes = price_change.abs().rolling(window=5).sum()
    momentum_factor = price_change / abs_price_changes
    
    # Combine with Adjusted Intraday Volatility
    factor_value = adjusted_intraday_volatility + momentum_factor
    
    return factor_value
```
