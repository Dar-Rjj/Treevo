import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio and its inverse
    volume_to_range_ratio = df['volume'] / intraday_range
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    adjusted_intraday_range = intraday_range * inverse_volume_to_range_ratio
    
    # Calculate Price Momentum
    price_change = df['close'].diff(1)
    sum_abs_price_changes = price_change.abs().rolling(window=5).sum()
    price_momentum = price_change / sum_abs_price_changes
    
    # Combine with Adjusted Intraday Volatility
    adjusted_intraday_volatility = adjusted_intraday_range + price_momentum
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_spike = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Adjust Intraday Return by Volume-to-Range Ratio
    adjusted_intraday_return = intraday_return * inverse_volume_to_range_ratio
    
    # Compute Price Difference Momentum
    price_difference_momentum = df['close'].diff(1).rolling(window=5).sum()
    
    # Combine Indicators
    combined_indicators = (
        weighted_intraday_return +
        price_difference_momentum +
        previous_day_return +
        adjusted_intraday_return
    )
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = intraday_range * avg_volume_5_days
    
    # Final Alpha Factor
    alpha_factor = (
        combined_indicators +
        volume_adjusted_daily_range +
        adjusted_intraday_range
    )
    
    return alpha_factor
