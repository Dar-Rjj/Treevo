```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    df['avg_vol_5_days'] = df['volume'].rolling(window=5).mean()
    df['vol_adj_daily_range'] = df['daily_range'] * df['avg_vol_5_days']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > 0).astype(int)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['vol_to_range_ratio'] = df['volume'] / df['daily_range']
    df['inv_vol_to_range_ratio'] = 1 / df['vol_to_range_ratio']
    df['adj_intraday_range'] = df['daily_range'] * df['inv_vol_to_range_ratio']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['abs_price_changes_5_days'] = df['price_change'].rolling(window=5).sum().abs()
    df['momentum_factor'] = df['price_change'] / df['abs_price_changes_5_days']
    df['momentum_adj_intraday_volatility'] = df['momentum_factor'] + df['adj_intraday_range']
    
    # Apply Moving Average Trend Adjustment
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = (df['close'] > df['ema_10']).astype(float) * 1.2 + (df['close'] <= df['ema_10']).astype(float) * 0.8
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    df['last_5_days_price_diff_momentum'] = df['price_diff_momentum'].rolling(window=5).sum()
    
    # Combine Factors
    df['combined_factors'] = (
        df['daily_range'] + 
        df['vol_adj_daily_range'] + 
        df['weighted_intraday_return'] + 
        df['last_5_days_price_diff_momentum'] + 
        df['prev_day_return'] + 
        df['momentum_adj_intraday_volatility']
    )
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_factors'] * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
