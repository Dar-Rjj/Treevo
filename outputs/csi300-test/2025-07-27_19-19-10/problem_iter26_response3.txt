```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_high_low_range
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    adjusted_intraday_high_low_range = intraday_high_low_range * (1 / volume_to_range_ratio)
    
    # Incorporate Price Momentum
    price_change = df['close'].diff(1)
    sum_abs_price_changes = price_change.abs().rolling(window=5).sum()
    momentum_factor = price_change / sum_abs_price_changes
    combined_momentum_adjusted_volatility = adjusted_intraday_high_low_range + momentum_factor
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Compute Price Difference Momentum
    price_diff_momentum = df['close'].diff().rolling(window=5).sum()
    
    # Calculate Exponential Moving Average (EMA) of Price Difference
    ema_price_diff = df['close'].diff().ewm(alpha=0.2, min_periods=1).mean()
    
    # Introduce Relative Strength Index (RSI)
    close_diff = df['close'].diff()
    up_changes = close_diff.clip(lower=0)
    down_changes = -close_diff.clip(upper=0)
    avg_up = up_changes.rolling(window=14).mean()
    avg_down = down_changes.rolling(window=14).mean()
    rsi = 100 - (100 / (1 + (avg_up / avg_down)))
    
    # Combine Factors
    factor_values = (
        daily_range +
        volume_adjusted_daily_range +
        weighted_intraday_return +
        price_diff_momentum +
        ema_price_diff +
        previous_day_return +
        combined_momentum_adjusted_volatility +
        rsi
    )
    
    return factor_values
```
