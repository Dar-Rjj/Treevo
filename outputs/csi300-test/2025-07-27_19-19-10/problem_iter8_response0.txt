```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_range
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_range = intraday_range * inverse_volume_to_range_ratio
    
    # Incorporate Price Momentum
    price_change = df['close'].diff()
    abs_price_change_sum = price_change.rolling(window=5).sum().abs()
    momentum_factor = price_change / abs_price_change_sum
    combined_momentum_volatility = adjusted_intraday_range + momentum_factor
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Final Alpha Factor
    final_alpha_factor = (combined_momentum_volatility 
                          + intraday_return * volume_spike 
                          + previous_day_return 
                          + daily_range 
                          + volume_adjusted_daily_range)
    
    return final_alpha_factor
```
