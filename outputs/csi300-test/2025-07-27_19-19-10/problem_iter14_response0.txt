```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust by Inverse of Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['intraday_range']
    df['inverse_volume_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adjusted_intraday_volatility'] = df['intraday_range'] * df['inverse_volume_to_range_ratio']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'].diff()
    df['sum_abs_price_changes_5d'] = df['price_change'].abs().rolling(window=5).sum()
    df['momentum_factor'] = df['price_change'] / df['sum_abs_price_changes_5d']
    df['combined_momentum_volatility'] = df['adjusted_intraday_volatility'] + df['momentum_factor']
    
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > 0).astype(int)
    
    # Compute Volume Adjusted Daily Range
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['vol_adjusted_daily_range'] = df['daily_range'] * df['avg_volume_5d']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        df['combined_momentum_volatility'] +
        (df['intraday_return'] * df['volume_spike']) +
        df['prev_day_return'] +
        (df['daily_range'] + df['vol_adjusted_daily_range'])
    )
    
    # Apply Additional Weighting Based on Moving Average
    df['10_day_sma'] = df['close'].rolling(window=10).mean()
    df['final_alpha_factor'] = df['final_alpha_factor'] * np.where(
        df['close'] > df['10_day_sma'], 1.2, 0.8
    )
    
    return df['final_alpha_factor']
```
