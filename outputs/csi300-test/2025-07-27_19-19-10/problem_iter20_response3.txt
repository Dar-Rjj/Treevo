```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / (df['high'] - df['low'])
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_volatility = (df['high'] - df['low']) * inverse_volume_to_range_ratio
    
    # Incorporate Price Momentum
    price_change = df['close'].diff()
    abs_price_change_sum_5d = price_change.rolling(window=5).sum().abs()
    momentum_factor = price_change / abs_price_change_sum_5d
    combined_momentum_volatility = adjusted_intraday_volatility + momentum_factor
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    volume_spike = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Compute Price Difference Momentum
    price_diff_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Final Alpha Factor
    alpha_factor = (
        daily_range +
        adjusted_intraday_volatility +
        weighted_intraday_return +
        price_diff_momentum +
        previous_day_return +
        combined_momentum_volatility +
        volume_adjusted_daily_range
    )
    
    return alpha_factor
```
