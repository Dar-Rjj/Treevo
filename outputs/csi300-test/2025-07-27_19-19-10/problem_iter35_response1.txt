```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_daily_range'] = df['daily_range'] * df['avg_volume_5d']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = df['volume_change'].apply(lambda x: 1 if x > 0 else 0)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['daily_range']
    df['adjusted_intraday_range'] = df['daily_range'] * (1 / df['volume_to_range_ratio'])
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['abs_price_change_sum_5d'] = df['close'].diff().abs().rolling(window=5).sum()
    df['price_momentum'] = df['price_change'] / df['abs_price_change_sum_5d']
    df['adjusted_intraday_volatility'] = df['adjusted_intraday_range'] + df['price_momentum']
    
    # Apply Moving Average Trend Adjustment
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = df.apply(lambda row: 1.2 if row['close'] > row['ema_10'] else 0.8, axis=1)
    
    # Compute Price Difference Momentum
    df['price_diff_5d'] = df['close'] - df['close'].shift(1)
    df['price_diff_momentum_5d'] = df['price_diff_5d'].rolling(window=5).sum()
    
    # Combine Factors
    df['combined_factors'] = (
        df['daily_range'] +
        df['volume_adjusted_daily_range'] +
        df['weighted_intraday_return'] +
        df['price_diff_momentum_5d'] +
        df['prev_day_return'] +
        df['adjusted_intraday_volatility']
    )
    
    # Final Alpha Factor with Additional Weighting Based on Moving Average
    df['final_alpha_factor'] = df['combined_factors'] * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
