```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_range
    
    # Adjust Intraday High-Low Range
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_range = intraday_range * inverse_volume_to_range_ratio
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['previous_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Combine Factors
    combined_factors = df['daily_range'] + adjusted_intraday_range + intraday_return * volume_spike + df['previous_day_return']
    
    # Compute Price Difference Momentum
    price_difference_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Incorporate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    abs_price_changes = (df['close'] - df['close'].shift(1)).abs().rolling(window=5).sum()
    price_momentum = price_change / abs_price_changes
    adjusted_intraday_volatility = combined_factors + price_momentum
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = df['daily_range'] * avg_volume_5_days
    
    # Apply Additional Weighting Based on Moving Average
    sma_10 = df['close'].rolling(window=10).mean()
    final_alpha_factor = adjusted_intraday_volatility + volume_adjusted_daily_range
    final_alpha_factor = final_alpha_factor.where(df['close'] > sma_10, final_alpha_factor * 0.8).where(df['close'] <= sma_10, final_alpha_factor * 1.2)
    
    return final_alpha_factor
```
