import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['intraday_range']
    
    # Compute Inverse of Volume-to-Range Ratio
    df['inverse_volume_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    
    # Adjust Intraday by Volume-to-Range Ratio
    df['adjusted_intraday_range'] = df['intraday_range'] * df['inverse_volume_to_range_ratio']
    
    # Compute Volume Adjusted Daily Range
    df['avg_vol_5_days'] = df['volume'].rolling(window=5).mean()
    df['vol_adj_daily_range'] = df['daily_range'] * df['avg_vol_5_days']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > 0).astype(int)
    
    # Combine Factors
    df['combined_factors'] = df['daily_range'] + df['vol_adj_daily_range'] + df['adjusted_intraday_range']
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    df['combined_factors'] += df['weighted_intraday_return'] + df['prev_day_return']
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    df['price_diff_momentum_sum'] = df['price_diff_momentum'].rolling(window=5).sum()
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factors'] + df['price_diff_momentum_sum']
    
    return df['alpha_factor']
