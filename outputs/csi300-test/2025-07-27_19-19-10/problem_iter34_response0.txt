```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    df['ShortTermPriceMomentum'] = (df['close'] / df['close'].shift(5)) - 1
    df['MediumTermPriceMomentum'] = (df['close'] / df['close'].shift(20)) - 1
    df['LongTermPriceMomentum'] = (df['close'] / df['close'].shift(60)) - 1
    
    # Calculate Volume Momentum
    df['ShortTermVolumeMomentum'] = (df['volume'] / df['volume'].shift(5)) - 1
    df['MediumTermVolumeMomentum'] = (df['volume'] / df['volume'].shift(20)) - 1
    df['LongTermVolumeMomentum'] = (df['volume'] / df['volume'].shift(60)) - 1
    
    # Calculate Price Volatility
    df['ShortTermPriceVolatility'] = df['close'].pct_change().rolling(window=5).std()
    df['MediumTermPriceVolatility'] = df['close'].pct_change().rolling(window=20).std()
    df['LongTermPriceVolatility'] = df['close'].pct_change().rolling(window=60).std()
    
    # Calculate Volume Volatility
    df['ShortTermVolumeVolatility'] = df['volume'].rolling(window=5).std()
    df['MediumTermVolumeVolatility'] = df['volume'].rolling(window=20).std()
    df['LongTermVolumeVolatility'] = df['volume'].rolling(window=60).std()
    
    # Calculate Price-Volume Correlation
    short_term_price_vol_corr = df[['close', 'volume']].pct_change().rolling(window=5).corr().unstack().iloc[::2]
    medium_term_price_vol_corr = df[['close', 'volume']].pct_change().rolling(window=20).corr().unstack().iloc[::2]
    long_term_price_vol_corr = df[['close', 'volume']].pct_change().rolling(window=60).corr().unstack().iloc[::2]
    
    df['ShortTermPriceVolumeCorrelation'] = short_term_price_vol_corr.values
    df['MediumTermPriceVolumeCorrelation'] = medium_term_price_vol_corr.values
    df['LongTermPriceVolumeCorrelation'] = long_term_price_vol_corr.values
    
    # Calculate Price-Volume Ratio
    df['ShortTermPriceVolumeRatio'] = (df['close'] / df['volume']) / (df['close'].shift(5) / df['volume'].shift(5))
    df['MediumTermPriceVolumeRatio'] = (df['close'] / df['volume']) / (df['close'].shift(20) / df['volume'].shift(20))
    df['LongTermPriceVolumeRatio'] = (df['close'] / df['volume']) / (df['close'].shift(60) / df['volume'].shift(60))
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        df['ShortTermPriceMomentum'] * 0.1 +
        df['MediumTermPriceMomentum'] * 0.2 +
        df['LongTermPriceMomentum'] * 0.3 +
        df['ShortTermVolumeMomentum'] * 0.1 +
        df['MediumTermVolumeMomentum'] * 0.1 +
        df['LongTermVolumeMomentum'] * 0.1 +
        df['ShortTermPriceVolatility'] * 0.05 +
        df['MediumTermPriceVolatility'] * 0.1 +
        df['LongTermPriceVolatility'] * 0.1 +
        df['ShortTermVolumeVolatility'] * 0.05 +
        df['MediumTermVolumeVolatility'] * 0.05 +
        df['LongTermVolumeVolatility'] * 0.05 +
        df['ShortTermPriceVolumeCorrelation'] * 0.05 +
        df['MediumTermPriceVolumeCorrelation'] * 0.05 +
        df['LongTermPriceVolumeCorrelation'] * 0.05 +
        df['ShortTermPriceVolumeRatio'] * 0.05 +
        df['MediumTermPriceVolumeRatio'] * 0.05 +
        df['LongTermPriceVolumeRatio'] * 0.05
    )
    
    return alpha_factor
```
