```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['vol_adj_daily_range'] = df['daily_range'] * df['avg_volume_5d']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = df['volume_change'].apply(lambda x: 1 if x > 0 else 0)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['daily_range']
    df['inverse_vol_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adj_intraday_high_low_range'] = df['daily_range'] * df['inverse_vol_to_range_ratio']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['sum_abs_price_changes_5d'] = df['price_change'].abs().rolling(window=5).sum()
    df['momentum_factor'] = df['price_change'] / df['sum_abs_price_changes_5d']
    df['combined_momentum_volatility'] = df['adj_intraday_high_low_range'] + df['momentum_factor']
    
    # Apply Moving Average Trend Adjustment
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = df.apply(lambda row: 1.2 if row['close'] > row['ema_10d'] else 0.8, axis=1)
    
    # Compute Price Difference Momentum
    df['price_diff'] = df['close'] - df['close'].shift(1)
    
    # Calculate Exponential Moving Average (EMA) of Price Difference
    df['ema_price_diff_5d'] = df['price_diff'].ewm(alpha=0.2, adjust=False).mean()
    
    # Introduce Price Momentum Smoothing
    df['sma_price_diff_5d'] = df['price_diff'].rolling(window=5).mean()
    
    # Combine Factors
    df['combined_factors'] = (
        df['daily_range'] + 
        df['vol_adj_daily_range'] + 
        df['weighted_intraday_return'] + 
        df['price_diff'].rolling(window=5).sum() + 
        df['ema_price_diff_5d'] + 
        df['prev_day_return'] + 
        df['adj_intraday_high_low_range'] + 
        df['sma_price_diff_5d']
    )
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_factors'] * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
