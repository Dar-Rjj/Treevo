import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on the given market data.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A series of alpha factor values indexed by date.
    """
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 10-day moving average of the typical price
    typical_price_ma_10 = typical_price.rolling(window=10).mean()
    
    # Calculate the 10-day moving average of the money flow
    money_flow_ma_10 = money_flow.rolling(window=10).mean()
    
    # Normalize the money flow by the typical price
    normalized_money_flow = money_flow / typical_price
    
    # Normalize the 10-day moving average of money flow by the 10-day moving average of the typical price
    normalized_money_flow_ma_10 = money_flow_ma_10 / typical_price_ma_10
    
    # Calculate the alpha factor
    alpha_factor = normalized_money_flow - normalized_money_flow_ma_10
    
    return alpha_factor
