```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    prev_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / daily_range
    inv_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_hl_range = daily_range * inv_volume_to_range_ratio
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Incorporate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    abs_price_changes_5d = (df['close'] - df['close'].shift(1)).abs().rolling(window=5).sum()
    momentum_factor = price_change / abs_price_changes_5d
    combined_momentum_volatility = adjusted_intraday_hl_range + momentum_factor
    
    # Apply Moving Average Trend Adjustment
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    trend_adjustment = np.where(df['close'] > ema_10, 1.2, 0.8)
    final_alpha_factor = combined_momentum_volatility * trend_adjustment
    
    # Compute Price Difference Momentum
    price_diff_5d = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Combine Factors
    combined_factors = (
        daily_range + 
        volume_adjusted_daily_range + 
        weighted_intraday_return + 
        price_diff_5d + 
        prev_day_return + 
        adjusted_intraday_hl_range
    )
    
    # Final Alpha Factor with additional weighting based on moving average
    final_alpha_factor = (combined_factors * trend_adjustment).dropna()
    
    return final_alpha_factor
```
