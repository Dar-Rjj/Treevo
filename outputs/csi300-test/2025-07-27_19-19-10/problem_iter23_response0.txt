```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['intraday_range']
    df['adjusted_intraday_range'] = df['intraday_range'] * (1 / df['volume_to_range_ratio'])
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'].diff()
    df['abs_price_changes_sum_5d'] = df['price_change'].abs().rolling(window=5).sum()
    df['momentum_factor'] = df['price_change'] / df['abs_price_changes_sum_5d']
    df['adjusted_intraday_volatility'] = df['adjusted_intraday_range'] + df['momentum_factor']
    
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Compute Volume Adjusted Daily Range
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_daily_range'] = df['daily_range'] * df['avg_volume_5d']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    df['price_diff_momentum_5d'] = df['price_diff_momentum'].rolling(window=5).sum()
    
    # Apply Moving Average Trend Adjustment
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = np.where(df['close'] > df['ema_10'], 1.2, 0.8)
    
    # Combine Factors
    df['combined_factors'] = (
        df['daily_range'] +
        df['volume_adjusted_daily_range'] +
        df['weighted_intraday_return'] +
        df['price_diff_momentum_5d'] +
        df['prev_day_return'] +
        df['adjusted_intraday_volatility']
    )
    
    # Final Alpha Factor
    df['final_alpha_factor'] = df['combined_factors'] * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
