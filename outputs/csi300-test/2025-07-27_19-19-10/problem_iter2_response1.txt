```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust Intraday by Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_range
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_range = intraday_range * inverse_volume_to_range_ratio
    
    # Compute Volume Adjusted Daily Range
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * average_volume_5_days
    
    # Compute Price Difference Momentum
    price_diff_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Combine Indicators
    alpha_factor = adjusted_intraday_range + volume_adjusted_daily_range + price_diff_momentum
    
    return alpha_factor.dropna()
```
