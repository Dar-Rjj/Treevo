```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Compute Price Difference Momentum
    df['price_diff'] = df['close'] - df['close'].shift(1)
    df['price_diff_momentum'] = df['price_diff'].rolling(window=5).sum()
    
    # Calculate Average True Range
    df['true_range'] = df.apply(lambda row: max(row['high'] - row['low'], 
                                                abs(row['high'] - row['close'].shift(1)), 
                                                abs(row['close'].shift(1) - row['low'])), axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=5).mean()
    
    # Combine Indicators
    df['combined_indicators'] = df['weighted_intraday_return'] + df['price_diff_momentum'] + df['prev_day_return'] + df['avg_true_range']
    
    # Final Alpha Factor
    alpha_factor = df['combined_indicators']
    
    return alpha_factor
```
