```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Close-to-Open Return
    close_to_open_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Weight Intraday Return by Volume Spike
    weighted_intraday_return = intraday_return * volume_spike
    
    # Weight Close-to-Open Return by Volume
    weighted_close_to_open_return = close_to_open_return * df['volume']
    
    # Combine Factors
    combined_factors = weighted_intraday_return + weighted_close_to_open_return
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors
    
    return final_alpha_factor
```
