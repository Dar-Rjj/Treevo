```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price (average of high, low, and close)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume weighted typical price
    df['vwap_typical_price'] = (df['typical_price'] * df['volume']) / df['volume'].rolling(window=10).mean()
    
    # Calculate the average amount over the last 10 days
    df['avg_amount_10d'] = df['amount'].rolling(window=10).mean()
    
    # Alpha factor: VWAP Typical Price divided by 10-day Average Amount
    alpha_factor = df['vwap_typical_price'] / df['avg_amount_10d']
    
    return alpha_factor
```
