```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Adjust Intraday by Volume-to-Range Ratio
    df['vol_to_range_ratio'] = df['volume'] / df['daily_range']
    df['inverse_vol_to_range_ratio'] = 1 / df['vol_to_range_ratio']
    df['adjusted_intraday_return'] = df['intraday_return'] * df['inverse_vol_to_range_ratio']
    
    # Compute Price Difference Momentum for the last 5 days
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    df['last_5_days_price_diff_momentum'] = df['price_diff_momentum'].rolling(window=5).sum()
    
    # Calculate Average True Range
    df['true_range'] = df[['high' - 'low', 'high' - df['close'].shift(1), df['close'].shift(1) - 'low']].max(axis=1)
    df['average_true_range'] = df['true_range'].rolling(window=5).mean()
    
    # Combine Indicators
    df['combined_indicators'] = (
        df['weighted_intraday_return'] +
        df['last_5_days_price_diff_momentum'] +
        df['prev_day_return'] +
        df['adjusted_intraday_return'] +
        df['average_true_range']
    )
    
    # Final Alpha Factor
    alpha_factor = df['combined_indicators']
    
    return alpha_factor
```
