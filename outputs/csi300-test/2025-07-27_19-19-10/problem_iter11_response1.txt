```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5d
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    prev_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / daily_range
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_high_low_range = daily_range * inverse_volume_to_range_ratio
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Compute Price Difference Momentum
    price_difference_momentum = df['close'].diff().rolling(window=5).sum()
    
    # Incorporate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    sum_abs_price_changes = df['close'].diff().abs().rolling(window=5).sum()
    incorporate_price_momentum = price_change / sum_abs_price_changes + adjusted_intraday_high_low_range
    
    # Apply Additional Weighting Based on Moving Average
    sma_10 = df['close'].rolling(window=10).mean()
    moving_average_weight = 1.2 if df['close'][-1] > sma_10[-1] else 0.8
    
    # Combine Factors
    combined_factors = (
        daily_range + 
        volume_adjusted_daily_range + 
        weighted_intraday_return + 
        price_difference_momentum + 
        prev_day_return + 
        adjusted_intraday_high_low_range + 
        incorporate_price_momentum
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * moving_average_weight
    
    return final_alpha_factor
```
