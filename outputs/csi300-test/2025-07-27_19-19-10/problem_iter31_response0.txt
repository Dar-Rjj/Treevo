```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    prev_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Compute Price Difference Momentum
    price_diff = df['close'] - df['close'].shift(1)
    price_diff_momentum = price_diff.rolling(window=5).sum()
    
    # Calculate EMA of Price Difference
    ema_price_diff = price_diff.ewm(alpha=0.2, adjust=False).mean()
    
    # Combine Factors
    factor = (
        daily_range +
        weighted_intraday_return +
        price_diff_momentum +
        ema_price_diff +
        prev_day_return
    )
    
    return factor
```
