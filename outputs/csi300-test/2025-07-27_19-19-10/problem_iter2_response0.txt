```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']

    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)

    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)

    # Compute Daily Range
    daily_range = df['high'] - df['low']

    # Compute Price Difference Momentum for the last 5 days
    price_diff_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()

    # Compute Volume Adjusted Daily Range
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * avg_volume_5_days

    # Combine Indicators
    combined_indicators = daily_range + volume_adjusted_daily_range + price_diff_momentum

    # Final Alpha Factor
    final_alpha_factor = (intraday_return * volume_spike) + previous_day_return + combined_indicators

    return final_alpha_factor
```
