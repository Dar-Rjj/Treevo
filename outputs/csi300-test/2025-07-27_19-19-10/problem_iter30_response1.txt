import pandas as pd

def heuristics_v2(df):
    # Calculate the log return of the close price
    df['log_return'] = (df['close'].apply(np.log) - df['close'].shift(1).apply(np.log)).fillna(0)
    
    # Calculate the relative range
    df['range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate the volume weighted average price
    df['vwap'] = (df['amount'] / df['volume']).fillna(0)
    
    # Calculate a simple moving average of the log return
    df['sma_log_return_5'] = df['log_return'].rolling(window=5).mean().fillna(0)

    # Calculate the normalized volume
    df['norm_volume'] = (df['volume'] - df['volume'].mean()) / df['volume'].std()

    # Define the alpha factor as a combination of the components
    df['alpha_factor'] = (df['log_return'] + df['range'] + 
                          df['vwap'] * 0.5 + df['sma_log_return_5'] + 
                          df['norm_volume'])
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
