```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday High-Low Range
    df['intraday_high_low_range'] = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / (df['intraday_high_low_range'] + 1e-6)
    
    # Adjust Intraday High-Low Range
    df['inverse_volume_to_range_ratio'] = 1 / (df['volume_to_range_ratio'] + 1e-6)
    df['adjusted_intraday_high_low_range'] = df['intraday_high_low_range'] * df['inverse_volume_to_range_ratio']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = (df['volume_change'] > 0).astype(int)
    
    # Combine Factors
    df['combined_factors'] = df['daily_range'] + df['adjusted_intraday_high_low_range']
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    df['combined_factors'] += df['weighted_intraday_return'] + df['prev_day_return']
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    
    # Store in a list for last 5 days
    df['price_diff_list'] = df['price_diff_momentum'].rolling(window=5, min_periods=1).apply(lambda x: list(x), raw=False)
    
    # Sum of Absolute Price Changes over the last 5 days
    df['sum_abs_price_changes'] = df['price_diff_momentum'].abs().rolling(window=5, min_periods=1).sum()
    
    # Divide Current Price Change by Sum
    df['price_momentum_factor'] = df['price_diff_momentum'] / (df['sum_abs_price_changes'] + 1e-6)
    
    # Add Result to Adjusted Intraday Volatility
    df['combined_factors'] += df['price_momentum_factor']
    
    # Compute Volume Adjusted Daily Range
    df['avg_volume_5_days'] = df['volume'].rolling(window=5, min_periods=1).mean()
    df['volume_adjusted_daily_range'] = df['daily_range'] * df['avg_volume_5_days']
    
    # Add Volume Adjusted Daily Range
    df['combined_factors'] += df['volume_adjusted_daily_range']
    
    # Apply Additional Weighting Based on Moving Average
    df['sma_10'] = df['close'].rolling(window=10, min_periods=1).mean()
    df['additional_weighting'] = (df['close'] > df['sma_10']).map({True: 1.2, False: 0.8})
    df['final_alpha_factor'] = df['combined_factors'] * df['additional_weighting']
    
    return df['final_alpha_factor']
```
