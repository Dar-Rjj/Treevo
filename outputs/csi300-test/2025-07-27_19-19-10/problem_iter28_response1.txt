```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Adjust Intraday High-Low Range
    df['volume_to_range_ratio'] = df['volume'] / df['daily_range']
    df['adjusted_intraday_range'] = df['daily_range'] * (1 / df['volume_to_range_ratio'])
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Compute Price Difference Momentum
    df['price_diff'] = df['close'] - df['close'].shift(1)
    df['price_diff_momentum'] = df['price_diff'].rolling(window=5).sum()
    
    # Calculate EMA of Price Difference
    df['ema_price_diff'] = df['price_diff'].ewm(alpha=0.2, min_periods=5).mean()
    
    # Combine Factors
    df['alpha_factor'] = (df['daily_range'] + 
                          df['weighted_intraday_return'] + 
                          df['price_diff_momentum'] + 
                          df['ema_price_diff'] + 
                          df['prev_day_return'] + 
                          df['adjusted_intraday_range']).fillna(0)
    
    return df['alpha_factor']
```
