```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    daily_range = df['high'] - df['low']
    
    # Adjust by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / daily_range
    adjusted_daily_range = daily_range * (1 / volume_to_range_ratio)
    
    # Incorporate Price Momentum
    price_change = df['close'].diff()
    abs_price_changes_last_5_days = df['close'].diff().abs().rolling(window=5).sum()
    price_momentum = price_change / abs_price_changes_last_5_days
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Volume Spike Detection
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 0).astype(int)
    
    # Apply SMA Trend Adjustment
    sma_10 = df['close'].rolling(window=10).mean()
    trend_adjustment = (df['close'] > sma_10).astype(float) * 1.2 + (df['close'] <= sma_10).astype(float) * 0.8
    
    # Compute Price Difference Momentum
    price_diff_momentum = df['close'].diff().rolling(window=5).sum()
    
    # Combine Factors
    combined_factors = (adjusted_daily_range + 
                        intraday_return + 
                        volume_spike + 
                        price_momentum + 
                        price_diff_momentum)
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors * trend_adjustment
    
    return final_alpha_factor
```
