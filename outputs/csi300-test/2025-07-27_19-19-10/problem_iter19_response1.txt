```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio and adjust by its inverse
    volume_to_range_ratio = df['volume'] / intraday_range
    adjusted_intraday_volatility = intraday_range * (1 / volume_to_range_ratio)
    
    # Incorporate Price Momentum
    price_change = df['close'].diff()
    abs_price_changes_5d = price_change.abs().rolling(window=5).sum()
    momentum_factor = price_change / abs_price_changes_5d
    
    # Combine with Adjusted Intraday Volatility
    combined_momentum_volatility = momentum_factor + adjusted_intraday_volatility
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'].diff()
    volume_spike = (volume_change > 0).astype(int)
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = intraday_range * avg_volume_5d
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_momentum_volatility +
        (intraday_return * volume_spike) +
        previous_day_return +
        volume_adjusted_daily_range +
        adjusted_intraday_volatility
    )
    
    # Apply Additional Weighting Based on Moving Average
    sma_10 = df['close'].rolling(window=10).mean()
    final_alpha_factor = final_alpha_factor * (1.2 if df['close'] > sma_10 else 0.8)
    
    return final_alpha_factor
```
