```python
import pandas as pd

def heuristics_v2(df):
    # Daily Range
    daily_range = df['high'] - df['low']
    
    # Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range * (avg_volume_5d / 5)
    
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike
    volume_spike = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Adjusted Intraday High-Low Range
    adjusted_intraday_high_low_range = daily_range * (1 / (df['volume'] / daily_range))
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Price Difference Momentum
    price_difference_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Final Alpha Factor
    final_alpha_factor = (
        daily_range + 
        volume_adjusted_daily_range + 
        weighted_intraday_return + 
        price_difference_momentum + 
        previous_day_return + 
        adjusted_intraday_high_low_range
    )
    
    return final_alpha_factor
```
