```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['Daily_Range'] = df['high'] - df['low']
    
    # Compute Price Difference Momentum for the last 5 days
    df['Price_Diff_Momentum'] = df['close'].diff(1)
    df['Last_5_Days_Price_Diff_Sum'] = df['Price_Diff_Momentum'].rolling(window=5).sum()
    
    # Compute Volume Adjusted Daily Range
    df['Avg_Volume_5_Days'] = df['volume'].rolling(window=5).mean()
    df['Volume_Adjusted_Daily_Range'] = df['Daily_Range'] * df['Avg_Volume_5_Days']
    
    # Combine Indicators
    df['Alpha_Factor'] = (df['Daily_Range'] + 
                          df['Volume_Adjusted_Daily_Range'] + 
                          df['Last_5_Days_Price_Diff_Sum'])
    
    # Return the alpha factor as a pandas Series
    return df['Alpha_Factor']
```
