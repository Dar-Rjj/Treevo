```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df_shifted = df.shift(1)
    previous_day_return = (df_shifted['close'] - df_shifted['open']) / df_shifted['open']
    
    # Enhanced Volume Spike Detection
    volume_t = df['volume']
    volume_t_1 = df_shifted['volume']
    volume_t_2 = df.shift(2)['volume']
    volume_spike = ((volume_t > 1.5 * volume_t_1) & (volume_t > 2 * volume_t_2)).astype(int)
    
    # Combine Factors
    weighted_intraday_return = intraday_return * volume_spike
    combined_factor = weighted_intraday_return + previous_day_return
    
    return combined_factor
```
