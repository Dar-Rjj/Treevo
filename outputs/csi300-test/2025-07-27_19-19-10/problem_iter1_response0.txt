```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum Indicator
    short_term_window = 5
    long_term_window = 20
    
    df['short_term_ma'] = df['close'].rolling(window=short_term_window).mean()
    df['long_term_ma'] = df['close'].rolling(window=long_term_window).mean()
    df['price_momentum'] = df['short_term_ma'] - df['long_term_ma']

    # Calculate Volume Trend Indicator
    alpha = 0.1
    df['ema_volume'] = df['volume'].ewm(alpha=alpha, adjust=False).mean()
    df['volume_trend'] = df['ema_volume'] - df['volume']

    # Combine Price Momentum and Volume Trend Indicators
    price_momentum_weight = 0.7
    volume_trend_weight = 0.3
    df['combined_indicator'] = (df['price_momentum'] * price_momentum_weight) + (df['volume_trend'] * volume_trend_weight)

    # Filter and Rank the Combined Indicator
    threshold = 0.05
    significant_values = df[df['combined_indicator'] > threshold]
    ranked_df = significant_values.sort_values(by='combined_indicator', ascending=False)
    
    return ranked_df['combined_indicator']
```
