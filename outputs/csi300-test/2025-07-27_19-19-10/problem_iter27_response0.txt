import pandas as pd

def heuristics_v2(df):
    # Calculate the return from open to close
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the range of the day
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate the average volume over a 30-day window
    df['avg_vol_30'] = df['volume'].rolling(window=30).mean()
    
    # Calculate the ratio of today's volume to the 30-day average volume
    df['vol_ratio'] = df['volume'] / df['avg_vol_30']
    
    # Calculate the amount-weighted intraday return
    df['amount_weighted_return'] = df['intraday_return'] * (df['amount'] / df['amount'].rolling(window=30).mean())
    
    # Generate the alpha factor
    df['alpha_factor'] = (df['intraday_return'] + df['daily_range'] / df['open'] + 
                          0.5 * (df['vol_ratio'] - 1) + df['amount_weighted_return'])
    
    # Drop any rows with NaN values
    df.dropna(inplace=True)
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
