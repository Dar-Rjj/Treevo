```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    df['Daily_Range'] = df['high'] - df['low']

    # Compute Volume Adjusted Daily Range
    df['Volume_Adjusted_Daily_Range'] = df['Daily_Range'] * df['volume'].rolling(window=5).mean()

    # Calculate Intraday Return
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']

    # Calculate Previous Day Return
    df['Previous_Day_Return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)

    # Detect Volume Spike
    df['Volume_Spike'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)

    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['Volume_to_Range_Ratio'] = df['volume'] / df['Daily_Range']
    df['Adjusted_Intraday_Volatility'] = df['Daily_Range'] * (1 / df['Volume_to_Range_Ratio'])

    # Weighted Intraday Return
    df['Weighted_Intraday_Return'] = df['Intraday_Return'] * df['Volume_Spike']

    # Incorporate Price Momentum
    df['Price_Change'] = df['close'] - df['close'].shift(1)
    df['Sum_Abs_Price_Changes'] = df['Price_Change'].rolling(window=5).apply(lambda x: abs(x).sum(), raw=True)
    df['Momentum_Adjustment'] = df['Price_Change'] / df['Sum_Abs_Price_Changes']
    df['Adjusted_Intraday_Volatility'] += df['Momentum_Adjustment']

    # Apply Moving Average Trend Adjustment
    df['10Day_EMA'] = df['close'].ewm(span=10, adjust=False).mean()
    df['Trend_Adjustment'] = df.apply(lambda x: 1.2 if x['close'] > x['10Day_EMA'] else 0.8, axis=1)
    
    # Compute Price Difference Momentum
    price_diff_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()

    # Calculate Exponential Moving Average (EMA) of Price Difference
    ema_price_diff = (df['close'] - df['close'].shift(1)).ewm(alpha=0.2, adjust=False).mean()

    # Combine Factors
    df['Combined_Factors'] = (df['Daily_Range'] + 
                              df['Volume_Adjusted_Daily_Range'] + 
                              df['Weighted_Intraday_Return'] + 
                              price_diff_momentum + 
                              ema_price_diff + 
                              df['Previous_Day_Return'] + 
                              df['Adjusted_Intraday_Volatility'])

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['Combined_Factors'] * df['Trend_Adjustment']

    return df['Final_Alpha_Factor']
```
