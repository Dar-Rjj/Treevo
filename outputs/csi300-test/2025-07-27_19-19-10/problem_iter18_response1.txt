import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['DailyRange'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['PrevDayReturn'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['VolumeSpike'] = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Weighted Intraday Return
    df['WeightedIntradayReturn'] = df['IntradayReturn'] * df['VolumeSpike']
    
    # Adjust Intraday by Volume-to-Range Ratio
    df['VolumeToRangeRatio'] = df['volume'] / df['DailyRange']
    df['InverseVolumeToRangeRatio'] = 1 / df['VolumeToRangeRatio']
    df['AdjustedIntradayReturn'] = df['IntradayReturn'] * df['InverseVolumeToRangeRatio']
    
    # Compute Price Difference Momentum (last 5 days)
    df['PriceDiffMomentum'] = df['close'] - df['close'].shift(1)
    df['Last5DaysPriceDiffMomentum'] = df['PriceDiffMomentum'].rolling(window=5).sum()
    
    # Calculate Average True Range
    df['TrueRange'] = df[['high', 'low', 'close']].apply(
        lambda x: max(x[0] - x[1], abs(x[0] - x[2].shift(1)), abs(x[1] - x[2].shift(1))), axis=1
    )
    df['AverageTrueRange'] = df['TrueRange'].rolling(window=5).mean()
    
    # Introduce Directional Movement
    df['DirectionalIndicator'] = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Combine Indicators
    df['CombinedIndicators'] = (
        df['WeightedIntradayReturn'] +
        df['Last5DaysPriceDiffMomentum'] +
        df['PrevDayReturn'] +
        df['AdjustedIntradayReturn'] +
        df['AverageTrueRange'] +
        df['DirectionalIndicator']
    )
    
    # Final Alpha Factor
    alpha_factor = df['CombinedIndicators']
    
    return alpha_factor
