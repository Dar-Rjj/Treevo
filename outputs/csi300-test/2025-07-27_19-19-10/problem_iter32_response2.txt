```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_adjusted_daily_range'] = df['daily_range'] * df['avg_volume_5d']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = np.where(df['volume_change'] > 0, 1, 0)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['daily_range']
    df['inverse_volume_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adjusted_intraday_volatility'] = df['daily_range'] * df['inverse_volume_to_range_ratio']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['sum_abs_price_changes_5d'] = df['price_change'].rolling(window=5).apply(lambda x: np.sum(np.abs(x)), raw=True)
    df['normalized_price_change'] = df['price_change'] / df['sum_abs_price_changes_5d']
    df['momentum_adjusted_volatility'] = df['adjusted_intraday_volatility'] + df['normalized_price_change']
    
    # Apply Moving Average Trend Adjustment
    df['ema_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = np.where(df['close'] > df['ema_10d'], 1.2, 0.8)
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Exponential Moving Average (EMA) of Price Difference
    df['ema_price_diff_5d'] = df['price_diff_momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Introduce Price Momentum Smoothing
    df['sma_price_diff_5d'] = df['price_diff_momentum'].rolling(window=5).mean()
    
    # Combine Factors
    df['final_alpha_factor'] = (
        df['daily_range'] +
        df['volume_adjusted_daily_range'] +
        df['weighted_intraday_return'] +
        df['price_diff_momentum'].rolling(window=5).sum() +
        df['ema_price_diff_5d'] +
        df['prev_day_return'] +
        df['adjusted_intraday_volatility'] +
        df['sma_price_diff_5d']
    )
    
    # Final Alpha Factor with Additional Weighting Based on Moving Average
    df['final_alpha_factor'] = df['final_alpha_factor'] * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
