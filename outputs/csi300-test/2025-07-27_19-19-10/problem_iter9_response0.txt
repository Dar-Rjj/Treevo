```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Close-to-Open Return
    df['prev_close_to_open_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = df['volume_change'].apply(lambda x: 1 if x > 0 else 0)
    
    # Weight Intraday Return by Volume Spike
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Weight Close-to-Open Return by Volume
    df['weighted_close_to_open_return'] = df['prev_close_to_open_return'] * df['volume']
    
    # Integrate Trade Amount
    df['amount_change'] = df['amount'] - df['amount'].shift(1)
    df['amount_integral'] = df['amount_change'].apply(lambda x: 1 if x > 0 else 0)
    
    # Combine Factors
    df['combined_factors'] = df['weighted_intraday_return'] + df['weighted_close_to_open_return'] + df['amount_integral']
    
    # Final Alpha Factor
    alpha_factor = df['combined_factors']
    
    return alpha_factor
```
