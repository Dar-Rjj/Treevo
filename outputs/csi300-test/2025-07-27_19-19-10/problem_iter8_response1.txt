```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Compute Volume Adjusted Daily Range
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    df['vol_adj_daily_range'] = df['daily_range'] * avg_volume_5_days
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    df['volume_spike'] = ((df['volume'] - df['volume'].shift(1)) > 0).astype(int)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Adjust Intraday by Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['daily_range']
    df['inv_vol_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adj_intraday_high_low_range'] = df['daily_range'] * df['inv_vol_to_range_ratio']
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'].diff().rolling(window=5).sum()
    
    # Combine Indicators
    df['combined_indicators'] = (
        df['weighted_intraday_return'] +
        df['vol_adj_daily_range'] +
        df['price_diff_momentum'] +
        df['prev_day_return'] +
        df['adj_intraday_high_low_range']
    )
    
    # Final Alpha Factor
    final_alpha_factor = df['combined_indicators']
    
    return final_alpha_factor
```
