```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    df['volume_spike'] = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['volume_to_range_ratio'] = df['volume'] / df['daily_range']
    df['inverse_volume_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adjusted_daily_range'] = df['daily_range'] * df['inverse_volume_to_range_ratio']
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    df['price_diff_momentum_5_days'] = df['price_diff_momentum'].rolling(window=5).sum()
    
    # Calculate Exponential Moving Average (EMA) of Price Difference
    df['ema_price_diff'] = df['price_diff_momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Combine Factors
    df['alpha_factor'] = (
        df['daily_range'] +
        df['weighted_intraday_return'] +
        df['price_diff_momentum_5_days'] +
        df['ema_price_diff'] +
        df['prev_day_return'] +
        df['adjusted_daily_range']
    )
    
    return df['alpha_factor']
```
