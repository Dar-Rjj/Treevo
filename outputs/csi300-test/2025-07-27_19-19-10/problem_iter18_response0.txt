```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Average True Range (ATR)
    df['TrueRange'] = df[['High', 'Low']].apply(lambda x: max(x[0], x[1]), axis=1) - df[['High', 'Low']].apply(lambda x: min(x[0], x[1]), axis=1)
    df['AbsHighPrevClose'] = abs(df['High'] - df['Close'].shift(1))
    df['AbsLowPrevClose'] = abs(df['Low'] - df['Close'].shift(1))
    df['TR'] = df[['TrueRange', 'AbsHighPrevClose', 'AbsLowPrevClose']].max(axis=1)
    df['ATR'] = df['TR'].rolling(window=5).mean()

    # Volume-Adjusted ATR
    df['Volume_Adjusted_ATR'] = df['ATR'] * (df['Volume'].rolling(window=5).sum() / 5)

    # Intraday Volatility
    df['Intraday_Volatility'] = (df['High'] - df['Low']) / df['Low']

    # Reversed Volume Spike
    df['Reversed_Volume_Spike'] = np.where((df['Volume'].shift(1) - df['Volume']) > 0, 1, 0)

    # Adjusted Intraday High-Low Ratio
    df['Adjusted_Intraday_High_Low_Ratio'] = df['Intraday_Volatility'] * (1 / (df['Volume'] / (df['High'] - df['Low'])))

    # Weighted Intraday Volatility
    df['Weighted_Intraday_Volatility'] = df['Intraday_Volatility'] * df['Reversed_Volume_Spike']

    # Price Difference Trend
    df['Price_Diff'] = df['Close'] - df['Close'].shift(1)
    df['Price_Difference_Trend'] = df['Price_Diff'].rolling(window=5).sum()

    # Momentum of Returns
    df['Momentum_of_Returns'] = (df['Close'] - df['Open']) / df['Open']

    # Final Alpha Factor
    df['Final_Alpha_Factor'] = df['ATR'] + df['Volume_Adjusted_ATR'] + df['Weighted_Intraday_Volatility'] + df['Price_Difference_Trend'] + df['Momentum_of_Returns'] + df['Adjusted_Intraday_High_Low_Ratio']

    return df['Final_Alpha_Factor']
```
