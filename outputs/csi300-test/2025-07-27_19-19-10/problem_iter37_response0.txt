```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Daily Range
    daily_range = df['high'] - df['low']
    
    # Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Previous Day Return
    prev_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike
    volume_spike = (df['volume'] - df['volume'].shift(1)).apply(lambda x: 1 if x > 0 else 0)
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Price Difference Momentum
    price_diff_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # EMA of Price Difference
    ema_alpha = 0.2
    ema_price_diff = (df['close'] - df['close'].shift(1)).ewm(alpha=ema_alpha, adjust=False).mean()
    
    # Intraday Volume Impact
    intraday_volume_impact = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Intraday Price Volatility
    intraday_price_volatility = intraday_return.rolling(window=5).std()
    
    # Combine Factors
    factor = (
        daily_range +
        weighted_intraday_return +
        price_diff_momentum +
        ema_price_diff +
        prev_day_return +
        intraday_volume_impact -
        intraday_price_volatility
    )
    
    return factor
```
