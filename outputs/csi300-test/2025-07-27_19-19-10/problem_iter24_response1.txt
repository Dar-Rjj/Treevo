```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_high_low_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_high_low_range
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    inverse_volume_to_range_ratio = 1 / volume_to_range_ratio
    adjusted_intraday_volatility = intraday_high_low_range * inverse_volume_to_range_ratio
    
    # Incorporate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    sum_abs_price_changes = price_change.rolling(window=5).sum().abs()
    momentum_factor = price_change / sum_abs_price_changes
    
    # Combine with Adjusted Intraday Volatility
    combined_momentum_volatility = (momentum_factor + adjusted_intraday_volatility) / 2
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Volume Spike Detection
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_spike = (volume_change > 0).astype(int)
    
    # Weight Intraday Return by Volume Spike
    weighted_intraday_return = intraday_return * volume_spike
    
    # Final Alpha Factor
    final_alpha_factor = (
        combined_momentum_volatility +
        weighted_intraday_return +
        previous_day_return +
        (df['high'] - df['low']) / df['volume'] +
        adjusted_intraday_volatility
    )
    
    return final_alpha_factor
```
