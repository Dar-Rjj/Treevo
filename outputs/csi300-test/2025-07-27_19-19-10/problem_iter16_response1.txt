```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Range
    daily_range = df['high'] - df['low']
    
    # Calculate Volume Adjusted Daily Range
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_adjusted_daily_range = daily_range / avg_volume_5d
    
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    previous_day_return = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    volume_spike = (df['volume'] > df['volume'].shift(1)).astype(int)
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / daily_range
    adjusted_intraday_volatility = daily_range * (1 / volume_to_range_ratio)
    
    # Incorporate Price Momentum
    price_change = df['close'] - df['close'].shift(1)
    sum_abs_price_changes_5d = abs(df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    momentum_factor = price_change / sum_abs_price_changes_5d
    combined_momentum_and_volatility = adjusted_intraday_volatility + momentum_factor
    
    # Weighted Intraday Return
    weighted_intraday_return = intraday_return * volume_spike
    
    # Compute Price Difference Momentum
    price_difference_momentum = (df['close'] - df['close'].shift(1)).rolling(window=5).sum()
    
    # Combine Factors
    combined_factors = (
        daily_range +
        volume_adjusted_daily_range +
        weighted_intraday_return +
        price_difference_momentum +
        previous_day_return +
        combined_momentum_and_volatility
    )
    
    # Final Alpha Factor
    final_alpha_factor = combined_factors

    return final_alpha_factor
```
