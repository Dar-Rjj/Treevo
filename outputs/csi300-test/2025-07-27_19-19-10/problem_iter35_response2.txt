```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Volume Adjusted Daily Range
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_adj_daily_range'] = df['daily_range'] * df['avg_volume_5d']
    
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Previous Day Return
    df['prev_day_return'] = (df['close'].shift(1) - df['open'].shift(1)) / df['open'].shift(1)
    
    # Detect Volume Spike
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    df['volume_spike'] = df['volume_change'].apply(lambda x: 1 if x > 0 else 0)
    
    # Weighted Intraday Return
    df['weighted_intraday_return'] = df['intraday_return'] * df['volume_spike']
    
    # Adjust Intraday High-Low Range by Inverse of Volume-to-Range Ratio
    df['intraday_high_low_range'] = df['high'] - df['low']
    df['volume_to_range_ratio'] = df['volume'] / df['intraday_high_low_range']
    df['inverse_vol_to_range_ratio'] = 1 / df['volume_to_range_ratio']
    df['adjusted_intraday_range'] = df['intraday_high_low_range'] * df['inverse_vol_to_range_ratio']
    
    # Incorporate Price Momentum
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['sum_abs_price_changes_5d'] = df['price_change'].abs().rolling(window=5).sum()
    df['momentum_adjusted_intraday_range'] = df['adjusted_intraday_range'] + df['price_change'] / df['sum_abs_price_changes_5d']
    
    # Apply Moving Average Trend Adjustment
    df['ema_10'] = df['close'].ewm(span=10, adjust=False).mean()
    df['trend_adjustment'] = df.apply(lambda x: 1.2 if x['close'] > x['ema_10'] else 0.8, axis=1)
    
    # Compute Price Difference Momentum
    df['price_diff_momentum'] = df['close'] - df['close'].shift(1)
    df['last_5_days_price_diff_momentum'] = df['price_diff_momentum'].rolling(window=5).sum()
    
    # Calculate EMA of Price Difference
    df['ema_price_diff'] = df['price_diff_momentum'].ewm(alpha=0.2, adjust=False).mean()
    
    # Calculate Accumulated Volume
    df['accumulated_volume'] = df['volume'].rolling(window=5).sum()
    
    # Calculate Volume Adjusted Return
    df['volume_adj_return'] = df['prev_day_return'] * df['accumulated_volume']
    
    # Combine Factors
    combined_factors = (
        df['daily_range'] 
        + df['volume_adj_daily_range'] 
        + df['weighted_intraday_return'] 
        + df['last_5_days_price_diff_momentum']
        + df['prev_day_return']
        + df['momentum_adjusted_intraday_range']
        + df['ema_price_diff']
        + df['volume_adj_return']
    )
    
    # Final Alpha Factor
    df['final_alpha_factor'] = combined_factors * df['trend_adjustment']
    
    return df['final_alpha_factor']
```
