```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Range
    intraday_range = df['high'] - df['low']
    
    # Compute Volume-to-Range Ratio
    volume_to_range_ratio = df['volume'] / intraday_range
    
    # Adjust by Volume-to-Range Ratio
    adjusted_intraday_volatility = intraday_range * (1 / volume_to_range_ratio)
    
    # Calculate Price Change
    price_change = df['close'].diff()
    
    # Normalize Price Change over the last 5 days
    abs_price_changes_sum = price_change.abs().rolling(window=5).sum()
    normalized_price_change = price_change / abs_price_changes_sum
    
    # Combine with Adjusted Intraday Volatility
    factor_values = adjusted_intraday_volatility + normalized_price_change
    
    return pd.Series(factor_values, index=df.index)
```
