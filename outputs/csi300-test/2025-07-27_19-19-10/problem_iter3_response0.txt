```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Parameters
    short_term_window = 5
    long_term_window = 20
    vroc_period = 14
    ema_alpha = 0.1
    volatility_window = 10
    
    # Compute Short-Term Weighted Moving Average of Close Prices
    weights = np.arange(1, short_term_window + 1)
    df['Short_Term_WMA'] = df['close'].rolling(window=short_term_window).apply(lambda x: (x * weights).sum() / weights.sum(), raw=True)
    
    # Compute Long-Term Simple Moving Average of Close Prices
    df['Long_Term_SMA'] = df['close'].rolling(window=long_term_window).mean()
    
    # Calculate the Difference between Short-Term WMA and Long-Term SMA
    df['Price_Momentum'] = df['Short_Term_WMA'] - df['Long_Term_SMA']
    
    # Calculate Volume Rate of Change (VROC) Indicator
    df['Volume_ROC'] = df['volume'].pct_change(periods=vroc_period)
    
    # Calculate the Exponential Moving Average (EMA) of VROC
    df['VROC_EMA'] = df['Volume_ROC'].ewm(alpha=ema_alpha, adjust=False).mean()
    
    # Calculate Historical Volatility
    df['Volatility'] = df['close'].rolling(window=volatility_window).std()
    
    # Combine Adjusted Price Momentum and VROC Indicators
    df['Combined_Indicator'] = df['Price_Momentum'] * df['VROC_EMA']
    
    # Assign Dynamic Weights to Each Indicator
    df['Momentum_Weight'] = df['Volatility'] / df['Volatility'].max()
    df['VROC_Weight'] = 1 - df['Momentum_Weight']
    df['Combined_Indicator_Weighted'] = (df['Price_Momentum'] * df['Momentum_Weight']) + (df['VROC_EMA'] * df['VROC_Weight'])
    
    # Filter and Rank the Combined Indicator
    market_return = df['close'].pct_change().mean()
    threshold = market_return * 2  # Example threshold, can be adjusted based on market conditions
    significant_values = df[df['Combined_Indicator_Weighted'] > threshold]
    
    # Rank the Selected Values
    ranked_values = significant_values['Combined_Indicator_Weighted'].rank(ascending=False, method='first')
    
    return ranked_values
```
