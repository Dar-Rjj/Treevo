```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day exponential moving averages of close prices to capture momentum
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EMAs to gauge momentum
    momentum_factor = ema_10_close - ema_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the 10-day exponential moving average of high and low prices
    ema_10_high = df['high'].ewm(span=10, adjust=False).mean()
    ema_10_low = df['low'].ewm(span=10, adjust=False).mean()
    
    # Compute the price range factor as the difference between the 10-day EMAs of high and low prices
    price_range_factor = ema_10_high - ema_10_low
    
    # Calculate the relative strength (RS) as the ratio of the 10-day EMA of close to the 30-day EMA of close
    relative_strength = ema_10_close / ema_30_close
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, price range, and relative strength
    factor_value = (momentum_factor * liquidity_factor * price_range_factor * relative_strength) / (volatility_factor + 1e-6)
    
    return factor_value
```
