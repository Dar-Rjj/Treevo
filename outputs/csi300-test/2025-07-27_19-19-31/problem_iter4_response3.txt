```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum: 50-day and 100-day return
    momentum_50 = df['close'].pct_change(50)
    momentum_100 = df['close'].pct_change(100)

    # Volatility: Average True Range (ATR) over the last 14 days
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr_14 = tr.rolling(window=14).mean()

    # Liquidity: 50-day average volume
    avg_volume_50 = df['volume'].rolling(window=50).mean()

    # Long-term trend: 200-day Exponential Moving Average (EMA)
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Relative Strength: 50-day return against a benchmark (e.g., S&P 500)
    benchmark_close = df['benchmark_close']  # Assuming 'benchmark_close' is a column in the DataFrame
    benchmark_momentum_50 = benchmark_close.pct_change(50)
    relative_strength_50 = momentum_50 - benchmark_momentum_50

    # Alpha factor: Combine momentum, volatility, liquidity, long-term trend, and relative strength
    alpha_factor = (relative_strength_50 / (atr_14 + 1e-7)) * (avg_volume_50 / (ema_200 + 1e-7))

    return alpha_factor
```
