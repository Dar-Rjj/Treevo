```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day exponentially weighted moving average (EWMA) of close prices to capture momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponentially weighted moving average (EWMA) of close prices for a longer-term trend
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day exponentially weighted moving average (EWMA) of volume to measure liquidity
    ewma_10_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day EWMA volume
    liquidity_factor = df['volume'] / ewma_10_volume
    
    # Combine the factors: momentum, adjusted by volatility and liquidity
    factor_value = (momentum_factor * liquidity_factor) / (volatility_factor + 1e-6)
    
    return factor_value
```
