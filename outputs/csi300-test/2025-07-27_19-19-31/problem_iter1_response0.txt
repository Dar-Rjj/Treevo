```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the weighted average of the amount and volume
    weighted_avg = (df['amount'] * df['volume']) / (df['volume'].rolling(window=5).mean() + 1e-7)
    
    # Alpha factor: Weighted average of the amount and volume, scaled by the price difference
    alpha_factor = price_diff * weighted_avg
    
    return alpha_factor
```
