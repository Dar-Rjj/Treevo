```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close to get the daily price change
    daily_price_change = df['close'].diff(1)
    # Calculate the 10-day EMA of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    # Calculate the factor as a ratio of daily price change to the 10-day EMA of volume
    alpha_factor = daily_price_change / (ema_volume + 1e-7)
    return alpha_factor
```
