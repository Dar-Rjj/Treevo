```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate adaptive moving averages of close prices to capture momentum
    ma_10_close = df['close'].rolling(window=10, min_periods=5).mean()
    ma_30_close = df['close'].rolling(window=30, min_periods=15).mean()

    # Compute the difference between the two moving averages to gauge momentum
    momentum_factor = ma_10_close - ma_30_close

    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10, min_periods=5).std()

    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10, min_periods=5).mean()

    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume

    # Incorporate price change and volume change
    price_change = df['close'].pct_change()
    volume_change = df['volume'].pct_change()

    # Combine the factors: momentum, adjusted by volatility and liquidity, and incorporate price and volume changes
    factor_value = (momentum_factor * liquidity_factor + price_change) / (volatility_factor + 1e-6) * (1 + volume_change)

    return factor_value
```
