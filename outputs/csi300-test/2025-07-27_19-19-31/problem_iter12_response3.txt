```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of close prices to capture momentum
    ma_10_close = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of close prices for longer-term trend
    ma_30_close = df['close'].rolling(window=30).mean()
    
    # Compute the difference between the two moving averages to gauge momentum
    momentum_factor = ma_10_close - ma_30_close
    
    # Calculate the 10-day exponentially weighted moving standard deviation of close prices for a more reactive volatility measure
    ewm_volatility = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the 10-day exponentially weighted moving average of volume to capture recent trading activity spikes
    ewma_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a trading activity spike factor as the ratio of current volume to the 10-day EWMA volume
    trading_activity_spike = df['volume'] / ewma_volume
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and trading activity spike
    factor_value = (momentum_factor * liquidity_factor * trading_activity_spike) / (ewm_volatility + 1e-6)
    
    return factor_value
```
