```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day moving average of close prices to capture momentum
    ma_10_close = df['close'].rolling(window=10).mean()
    ma_30_close = df['close'].rolling(window=30).mean()
    
    # Compute the difference between the two moving averages to gauge short-term momentum
    momentum_factor = ma_10_close - ma_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent short-term volatility
    volatility_10 = df['close'].rolling(window=10).std()
    
    # Calculate the 50-day standard deviation of close prices to represent medium-term volatility
    volatility_50 = df['close'].rolling(window=50).std()
    
    # Combine short-term and medium-term volatility for a more robust measure
    dynamic_volatility = (volatility_10 + volatility_50) / 2
    
    # Calculate the 10-day and 50-day average volume to measure liquidity
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    avg_volume_50 = df['volume'].rolling(window=50).mean()
    
    # Combine short-term and medium-term average volume for a more robust measure
    avg_volume_combined = (avg_volume_10 + avg_volume_50) / 2
    
    # Calculate a liquidity factor as the ratio of current volume to the combined average volume
    liquidity_factor = df['volume'] / avg_volume_combined
    
    # Calculate the 200-day EMA to capture long-term trend
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()
    
    # Calculate the 50-day EMA to capture short-term trend
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()
    
    # Price-to-EMA ratio: Compare current price to long-term EMA
    price_to_ema_200 = df['close'] / ema_200
    
    # Trend Strength: Difference between short-term EMA and long-term EMA
    trend_strength = ema_50 - ema_200
    
    # Combine the factors: momentum, adjusted by dynamic volatility, liquidity, and trend strength
    factor_value = (momentum_factor * liquidity_factor * price_to_ema_200 * trend_strength) / (dynamic_volatility + 1e-7)
    
    return factor_value
```
