```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change over a 10-day period
    price_change_10d = df['close'] - df['close'].shift(10)
    
    # Calculate the relative volume, which is the current volume divided by the mean of the previous 30 days' volumes
    relative_volume_30d = df['volume'] / df['volume'].rolling(window=30).mean()
    
    # Calculate the 50-day moving average
    ma_50d = df['close'].rolling(window=50).mean()
    
    # Calculate the distance from the current close price to the 50-day moving average
    distance_to_ma = df['close'] - ma_50d
    
    # Calculate the ATR (Average True Range) for a 14-day period as a measure of volatility
    tr = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    atr_14d = tr.rolling(window=14).mean()
    
    # Combine the factors
    factor_value = (price_change_10d * relative_volume_30d) + (distance_to_ma / atr_14d)
    
    return factor_value
```
