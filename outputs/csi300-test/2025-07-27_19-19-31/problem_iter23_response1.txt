```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Exponentially Weighted Moving Average of close prices with a span of 10 days for smoothing
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day moving average of close prices for longer-term trend
    ma_30_close = df['close'].rolling(window=30).mean()
    
    # Compute the difference between EWMA and simple moving average to gauge momentum
    momentum_factor = ewma_10_close - ma_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent recent volatility
    recent_volatility = df['close'].rolling(window=10).std()
    
    # Calculate the 60-day standard deviation of close prices to represent longer-term volatility
    long_term_volatility = df['close'].rolling(window=60).std()
    
    # Volatility factor as the ratio of recent to long-term volatility
    volatility_factor = recent_volatility / (long_term_volatility + 1e-6)
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / (avg_volume + 1e-6)
    
    # Incorporate high/low price differences to capture intraday dynamics
    price_diff = df['high'] - df['low']
    normalized_price_diff = price_diff / (df['close'] + 1e-6)
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and price difference
    factor_value = (momentum_factor * liquidity_factor * (1 + normalized_price_diff)) / (volatility_factor + 1e-6)
    
    return factor_value
```
