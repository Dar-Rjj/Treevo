```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day exponentially weighted moving average (EWMA) of close prices to capture recent momentum
    ewma_5_close = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 10-day exponentially weighted moving average (EWMA) of close prices for shorter-term trend
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Compute the difference between the 5-day and 10-day EWMA to gauge recent momentum
    recent_momentum_factor = ewma_5_close - ewma_10_close
    
    # Calculate the 30-day exponentially weighted moving average (EWMA) of close prices for longer-term trend
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the 10-day and 30-day EWMA to gauge medium-term momentum
    medium_momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day exponentially weighted standard deviation of close prices to represent volatility
    ewm_volatility_factor = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 5-day exponentially weighted average volume to measure recent liquidity
    ewma_5_volume = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 5-day EWMA volume
    liquidity_factor = df['volume'] / ewma_5_volume
    
    # Integrate the log returns of the close price to capture short-term price movements
    log_returns = np.log(df['close']).diff()
    
    # Combine the factors: recent momentum, medium-term momentum, adjusted by volatility and liquidity, and incorporate log returns
    factor_value = ((recent_momentum_factor + medium_momentum_factor) * liquidity_factor * log_returns) / (ewm_volatility_factor + 1e-6)
    
    return factor_value
```
