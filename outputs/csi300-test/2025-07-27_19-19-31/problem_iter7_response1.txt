```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day Exponentially Weighted Moving Average (EWMA) of close prices to capture momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day Exponentially Weighted Moving Average (EWMA) of close prices for longer-term trend
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day standard deviation of log returns to represent volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility_factor = log_returns.rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the 10-day Exponentially Weighted Moving Average (EWMA) of amount to capture transaction activity
    ewma_10_amount = df['amount'].ewm(span=10, adjust=False).mean()
    
    # Calculate a dynamic liquidity factor that considers transaction activity
    dynamic_liquidity_factor = (df['volume'] * df['amount']) / (avg_volume * ewma_10_amount + 1e-6)
    
    # Combine the factors: momentum, adjusted by volatility and dynamic liquidity
    factor_value = (momentum_factor * dynamic_liquidity_factor) / (volatility_factor + 1e-6)
    
    return factor_value
```
