```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day exponentially weighted moving average (EWMA) of close prices to capture adaptive momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponentially weighted moving average (EWMA) of close prices for longer-term trend
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge adaptive momentum
    adaptive_momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day exponentially weighted standard deviation of close prices to represent adaptive volatility
    adaptive_volatility_factor = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day exponentially weighted average volume to measure adaptive liquidity
    ewma_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day EWMA volume
    adaptive_liquidity_factor = df['volume'] / ewma_volume
    
    # Combine the factors: adaptive momentum, adjusted by adaptive volatility and adaptive liquidity
    factor_value = (adaptive_momentum_factor * adaptive_liquidity_factor) / (adaptive_volatility_factor + 1e-6)
    
    return factor_value
```
