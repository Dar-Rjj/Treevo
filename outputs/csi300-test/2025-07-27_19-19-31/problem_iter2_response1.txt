```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    price_diff = df['close'] - df['open']
    
    # Calculate the relative volume, which is the current volume divided by the mean of the previous 10 days' volumes
    relative_volume = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # Calculate the 5-day moving average of the closing price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the closing price
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the volatility as the standard deviation of the past 20 days' closing prices
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate the Relative Strength Index (RSI) over the past 14 days
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rsi = 100 - (100 / (1 + gain / loss))
    
    # Combine the factors into a single alpha factor
    alpha_factor = (price_diff * relative_volume * (ma_5 - ma_20) / volatility) + rsi
    
    return alpha_factor
```
