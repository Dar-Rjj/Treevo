```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day Exponentially Weighted Moving Averages (EWMA) of close prices to capture momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day rolling standard deviation of close prices to represent recent volatility
    recent_volatility = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Incorporate high and low prices to capture intraday volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Calculate the 10-day rolling sum of amount to capture market sentiment
    market_sentiment = df['amount'].rolling(window=10).sum()
    
    # Combine the factors: momentum, adjusted by recent volatility, liquidity, intraday volatility, and market sentiment
    factor_value = (momentum_factor * liquidity_factor * market_sentiment) / (recent_volatility + 1e-6) * intraday_volatility
    
    return factor_value
```
