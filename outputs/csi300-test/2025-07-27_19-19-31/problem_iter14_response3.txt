```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day exponentially weighted moving average (EWMA) of close prices to capture short-term and long-term trends
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day exponentially weighted moving standard deviation of close prices for a more reactive volatility measure
    ewm_volatility = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day rolling average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Integrate market cap into the factor
    market_cap_factor = df['amount'] / df['close']
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and market cap
    factor_value = (momentum_factor * liquidity_factor * market_cap_factor) / (ewm_volatility + 1e-6)
    
    return factor_value
```
