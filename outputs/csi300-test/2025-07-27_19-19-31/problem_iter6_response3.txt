```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day exponentially smoothed close prices to capture short-term trend
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponentially smoothed close prices for longer-term trend
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EMAs to gauge momentum
    momentum_factor = ema_10_close - ema_30_close
    
    # Calculate the 10-day exponentially smoothed standard deviation of close prices to represent volatility
    volatility_factor = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day exponentially smoothed average volume to measure liquidity
    avg_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day EMA volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Incorporate seasonality by adding a seasonal adjustment factor
    # Assume the data is daily and use a 5-day week as the seasonality period
    df['date'] = pd.to_datetime(df.index)
    df['day_of_week'] = df['date'].dt.dayofweek
    seasonal_factor = df.groupby('day_of_week')['close'].transform(lambda x: (x - x.mean()) / x.std())
    
    # Calculate the money flow index (MFI) to incorporate more price/volume features
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfi = positive_money_flow.rolling(window=14).sum() / (positive_money_flow.rolling(window=14).sum() + negative_money_flow.rolling(window=14).sum())
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, seasonality, and MFI
    factor_value = (momentum_factor * liquidity_factor * (1 + mfi) * (1 + seasonal_factor)) / (volatility_factor + 1e-6)
    
    return factor_value
```
