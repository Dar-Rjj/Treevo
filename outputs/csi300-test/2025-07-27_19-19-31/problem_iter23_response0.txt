```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day Exponentially Weighted Moving Averages (EWMA) of close prices
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge short-term momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day rolling standard deviation of close prices for recent volatility
    recent_volatility = df['close'].rolling(window=10).std()
    
    # Calculate the 5-day and 20-day average volume to measure liquidity changes
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    
    # Calculate a liquidity factor as the ratio of 5-day average volume to 20-day average volume
    liquidity_factor = avg_volume_5 / avg_volume_20
    
    # Incorporate the high/low price differences to capture intraday volatility
    high_low_diff = df['high'] - df['low']
    intraday_volatility = high_low_diff.rolling(window=10).mean()
    
    # Combine the factors: momentum, adjusted by recent and intraday volatility, and liquidity
    factor_value = (momentum_factor * liquidity_factor) / ((recent_volatility + intraday_volatility) + 1e-6)
    
    return factor_value
```
