```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day moving averages of close prices
    ma_10_close = df['close'].rolling(window=10).mean()
    ma_30_close = df['close'].rolling(window=30).mean()
    
    # Compute the difference between the two moving averages to gauge momentum
    momentum_factor = ma_10_close - ma_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the 10-day rate of change of close prices
    price_change_rate = (df['close'] / df['close'].shift(10)) - 1
    
    # Calculate the 10-day correlation between volume and close prices
    volume_price_corr = df['close'].rolling(window=10).corr(df['volume'])
    
    # Calculate the 10-day adaptive moving average (AMA)
    ama_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, price change rate, and volume-price correlation
    factor_value = (momentum_factor * liquidity_factor * price_change_rate * volume_price_corr) / (volatility_factor + 1e-6)
    
    return factor_value
```
