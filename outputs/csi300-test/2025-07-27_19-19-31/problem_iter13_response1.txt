```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day Exponentially Weighted Moving Average (EWMA) of close prices to capture momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day rolling standard deviation of the log return to represent volatility
    log_return = np.log(df['close'] / df['close'].shift(1))
    volatility_factor = log_return.rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Incorporate seasonality by calculating the 6-month (approximately 120 trading days) rolling mean
    seasonality_factor = df['close'].rolling(window=120).mean()
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and seasonality
    factor_value = (momentum_factor * liquidity_factor * seasonality_factor) / (volatility_factor + 1e-6)
    
    return factor_value
```
