```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day exponentially smoothed close prices to capture short-term and long-term trends
    ema_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EMAs to gauge momentum
    momentum_factor = ema_10_close - ema_30_close
    
    # Calculate the 10-day exponentially smoothed standard deviation of close prices to represent volatility
    volatility_factor = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day exponentially smoothed average volume to measure liquidity
    avg_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day EMA volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Adaptive window for high and low prices
    adaptive_window = (df['high'] - df['low']).rolling(window=50).mean()
    
    # Calculate the 10-day exponentially smoothed adaptive window
    ema_adaptive_window = adaptive_window.ewm(span=10, adjust=False).mean()
    
    # Factor based on the adaptive window
    adaptive_factor = adaptive_window / ema_adaptive_window
    
    # Incorporate the amount traded as a factor
    amount_factor = df['amount'].ewm(span=10, adjust=False).mean() / df['amount']
    
    # Calculate the 10-day exponentially smoothed open prices to capture the opening trend
    ema_10_open = df['open'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponentially smoothed high and low prices to capture price range
    ema_10_high = df['high'].ewm(span=10, adjust=False).mean()
    ema_10_low = df['low'].ewm(span=10, adjust=False).mean()
    
    # Factor based on the price range
    price_range_factor = (ema_10_high - ema_10_low) / (ema_10_close + 1e-6)
    
    # Add a factor based on the exponential smoothing of the amount
    amount_trend_factor = df['amount'].ewm(span=10, adjust=False).mean() / df['amount']
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, adaptive window, amount traded, price range, and amount trend
    factor_value = (momentum_factor * liquidity_factor * adaptive_factor * amount_factor * price_range_factor * amount_trend_factor) / (volatility_factor + 1e-6)
    
    return factor_value
```
