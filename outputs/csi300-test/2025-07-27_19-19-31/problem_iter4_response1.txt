```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of close prices to capture short-term momentum
    ma_5_close = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of close prices for medium-term trend
    ma_20_close = df['close'].rolling(window=20).mean()
    
    # Compute the difference between the two moving averages to gauge short-term momentum
    short_momentum_factor = ma_5_close - ma_20_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 5-day average volume to measure recent liquidity
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 5-day average volume
    liquidity_factor = df['volume'] / (avg_volume_5 + 1e-6)
    
    # Combine the factors: short-term momentum, adjusted by volatility and liquidity
    factor_value = (short_momentum_factor * liquidity_factor) / (volatility_factor + 1e-6)
    
    return factor_value
```
