```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close as a measure of momentum
    df['momentum'] = df['close'].diff()
    # Calculate the volume weighted by the closing price, which can be seen as a form of value-weighted trading activity
    df['volume_weighted_price'] = (df['close'] * df['volume'])
    # A heuristic factor that combines momentum with volume weighted price, which could indicate investor sentiment
    alpha_factor = (df['volume_weighted_price'].rolling(window=5).mean() + df['momentum']) / df['close']
    return alpha_factor
```
