```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of the close price
    ma_close = df['close'].rolling(window=10).mean()
    # Calculate the 10-day standard deviation (volatility) of the close price
    std_close = df['close'].rolling(window=10).std()
    # Calculate the 20-day momentum of the close price
    momentum = df['close'] / df['close'].shift(20)
    # Calculate the 10-day average of the volume
    avg_volume = df['volume'].rolling(window=10).mean()
    # Calculate the relative volume, which is the current volume divided by the 10-day average volume
    relative_volume = df['volume'] / avg_volume
    # Calculate the 10-day moving average of the high-low range
    ma_high_low_range = (df['high'] - df['low']).rolling(window=10).mean()
    # Combine the factors: volatility, momentum, and liquidity (relative volume), and market breadth (average high-low range)
    factor_value = (std_close * momentum * relative_volume) / (ma_high_low_range + 1e-6)  # add small constant to avoid division by zero
    return factor_value
```
