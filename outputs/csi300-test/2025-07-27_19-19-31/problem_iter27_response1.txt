```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive moving average of close prices to capture short-term momentum
    ma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of close prices for longer-term trend
    ma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two moving averages to gauge momentum
    momentum_factor = ma_10_close - ma_30_close
    
    # Calculate the 10-day exponential standard deviation of close prices to represent short-term volatility
    volatility_factor = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day exponential average volume to measure liquidity
    avg_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Integrate intraday data: ratio of (high - open) to (open - low) to capture intraday sentiment
    intraday_sentiment = (df['high'] - df['open']) / (df['open'] - df['low'])
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and intraday sentiment
    factor_value = (momentum_factor * liquidity_factor * intraday_sentiment) / (volatility_factor + 1e-6)
    
    return factor_value
```
