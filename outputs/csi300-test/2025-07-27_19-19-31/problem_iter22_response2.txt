```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day moving averages of close prices to capture short-term momentum
    ma_5_close = df['close'].rolling(window=5).mean()
    ma_10_close = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day and 60-day moving averages of close prices for longer-term trend
    ma_30_close = df['close'].rolling(window=30).mean()
    ma_60_close = df['close'].rolling(window=60).mean()
    
    # Compute the difference between the short-term and long-term moving averages to gauge momentum
    short_term_momentum = ma_5_close - ma_10_close
    long_term_momentum = ma_30_close - ma_60_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Incorporate price change rates
    price_change_rate = df['close'].pct_change(periods=1)
    
    # Consider volume-price relationship
    volume_price_ratio = df['volume'] / df['close']
    
    # Combine the factors: short-term and long-term momentum, adjusted by volatility, liquidity, price change rate, and volume-price relationship
    factor_value = (short_term_momentum + long_term_momentum) * liquidity_factor / (volatility_factor + 1e-6) * price_change_rate * volume_price_ratio
    
    return factor_value
```
