```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of close prices to capture short-term and medium-term momentum
    ma_5_close = df['close'].rolling(window=5).mean()
    ma_20_close = df['close'].rolling(window=20).mean()
    
    # Compute the difference between the two moving averages to gauge short-term momentum
    short_momentum_factor = ma_5_close - ma_20_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the 5-day average range (high - low) to represent market breadth
    market_breadth = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the 10-day moving average of the amount to capture transaction size dynamics
    avg_amount = df['amount'].rolling(window=10).mean()
    
    # Calculate the transaction size factor as the ratio of current amount to the 10-day average amount
    transaction_size_factor = df['amount'] / avg_amount
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and transaction size, and consider market breadth
    factor_value = (short_momentum_factor * liquidity_factor * transaction_size_factor) / (volatility_factor + 1e-6) + market_breadth
    
    return factor_value
```
