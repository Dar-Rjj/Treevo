```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day Exponentially Weighted Moving Average (EWMA) of close prices for adaptive momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the correlation between volume and price over the last 10 days
    volume_price_corr = df[['close', 'volume']].rolling(window=10).corr().unstack().iloc[::2]['volume']
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and volume-price correlation
    factor_value = (momentum_factor * liquidity_factor * volume_price_corr) / (volatility_factor + 1e-6)
    
    return factor_value
```
