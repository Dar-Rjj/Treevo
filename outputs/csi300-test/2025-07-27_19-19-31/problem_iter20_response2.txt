```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day exponentially weighted moving average (EWMA) of close prices to capture momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponentially weighted moving average (EWMA) of close prices for longer-term trend
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day exponentially weighted standard deviation of close prices to represent volatility
    volatility_factor = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day exponentially weighted average volume to measure liquidity
    avg_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day EWMA volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the 10-day exponentially weighted moving average of amount to capture transaction activity
    ewma_10_amount = df['amount'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponentially weighted standard deviation of amount to represent transaction variability
    amount_volatility_factor = df['amount'].ewm(span=10, adjust=False).std()
    
    # Incorporate price change and volume change
    price_change = df['close'].pct_change()
    volume_change = df['volume'].pct_change()
    
    # Calculate the 10-day exponentially weighted moving average of high and low prices to capture intraday volatility
    ewma_10_high = df['high'].ewm(span=10, adjust=False).mean()
    ewma_10_low = df['low'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day exponentially weighted range (high - low) to represent intraday volatility
    intraday_volatility_factor = ewma_10_high - ewma_10_low
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, and transaction activity, and incorporate price and volume changes, and intraday volatility
    factor_value = (momentum_factor * liquidity_factor * ewma_10_amount * (1 + price_change)) / (volatility_factor + amount_volatility_factor + intraday_volatility_factor + 1e-6) * (1 + volume_change)
    
    return factor_value
```
