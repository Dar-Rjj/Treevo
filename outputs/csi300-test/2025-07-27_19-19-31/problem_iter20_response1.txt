```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day Exponentially Weighted Moving Average (EWMA) of close prices to capture adaptive momentum
    ewma_10_close = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day Exponentially Weighted Moving Average (EWMA) of close prices for longer-term trend
    ewma_30_close = df['close'].ewm(span=30, adjust=False).mean()
    
    # Compute the difference between the two EWMA to gauge adaptive momentum
    momentum_factor = ewma_10_close - ewma_30_close
    
    # Calculate the 10-day exponentially weighted standard deviation of close prices to represent dynamic volatility
    volatility_factor = df['close'].ewm(span=10, adjust=False).std()
    
    # Calculate the 10-day exponentially weighted average volume to measure dynamic liquidity
    avg_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day exponentially weighted average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Incorporate transaction activity as the ratio of amount to volume
    transaction_activity = df['amount'] / df['volume']
    
    # Combine the factors: adaptive momentum, adjusted by dynamic volatility, dynamic liquidity, and transaction activity
    factor_value = (momentum_factor * liquidity_factor * transaction_activity) / (volatility_factor + 1e-6)
    
    return factor_value
```
