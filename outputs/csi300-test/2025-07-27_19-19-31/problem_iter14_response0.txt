```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day moving average of close prices to capture momentum
    ma_10_close = df['close'].rolling(window=10).mean()
    ma_30_close = df['close'].rolling(window=30).mean()
    
    # Compute the difference between the two moving averages to gauge momentum
    momentum_factor = ma_10_close - ma_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Calculate the 10-day moving average of volume to capture volume trends
    ma_10_volume = df['volume'].rolling(window=10).mean()
    
    # Compute the difference between the current volume and 10-day moving average volume to gauge volume trend
    volume_trend_factor = df['volume'] - ma_10_volume
    
    # Calculate the 1-day price change rate
    price_change_rate = df['close'].pct_change()
    
    # Combine the factors: momentum, adjusted by volatility, liquidity, volume trend, and price change rate
    factor_value = (momentum_factor * liquidity_factor * volume_trend_factor * price_change_rate) / (volatility_factor + 1e-6)
    
    return factor_value
```
