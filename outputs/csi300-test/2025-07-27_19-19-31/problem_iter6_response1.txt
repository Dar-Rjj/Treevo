```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum: 50-day and 100-day return
    momentum_50 = df['close'].pct_change(50)
    momentum_100 = df['close'].pct_change(100)

    # Dynamic Volatility: Standard deviation of daily returns over the last 50 days, adjusted for recent volatility
    daily_returns = df['close'].pct_change()
    volatility_50 = daily_returns.rolling(window=50).std()
    recent_volatility = daily_returns.rolling(window=10).std()
    dynamic_volatility = (volatility_50 + recent_volatility) / 2

    # Liquidity: 50-day average volume and 10-day average volume
    avg_volume_50 = df['volume'].rolling(window=50).mean()
    avg_volume_10 = df['volume'].rolling(window=10).mean()
    liquidity_factor = (avg_volume_50 + avg_volume_10) / 2

    # Long-term trend: 200-day EMA
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Short-term trend: 50-day EMA
    ema_50 = df['close'].ewm(span=50, adjust=False).mean()

    # Price-to-EMA ratio: Compare current price to long-term EMA
    price_to_ema_200 = df['close'] / ema_200

    # Trend Strength: Difference between short-term EMA and long-term EMA
    trend_strength = ema_50 - ema_200

    # Alpha factor: Combine momentum, dynamic volatility, liquidity, and long-term trend
    alpha_factor = (momentum_50 - momentum_100) / (dynamic_volatility + 1e-7) * (liquidity_factor / (ema_200 + 1e-7)) * price_to_ema_200 * trend_strength

    return alpha_factor
```
