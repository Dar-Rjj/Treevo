```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum: 50-day and 100-day return
    momentum_50 = df['close'].pct_change(50)
    momentum_100 = df['close'].pct_change(100)

    # Volatility: Standard deviation of daily returns over the last 50 days
    daily_returns = df['close'].pct_change()
    volatility_50 = daily_returns.rolling(window=50).std()

    # Liquidity: 50-day average volume
    avg_volume_50 = df['volume'].rolling(window=50).mean()

    # Long-term trend: 200-day EMA
    ema_200 = df['close'].ewm(span=200, adjust=False).mean()

    # Alpha factor: Combine momentum, volatility, and liquidity
    alpha_factor = (momentum_50 - momentum_100) / (volatility_50 + 1e-7) * (avg_volume_50 / (ema_200 + 1e-7))

    return alpha_factor
```
