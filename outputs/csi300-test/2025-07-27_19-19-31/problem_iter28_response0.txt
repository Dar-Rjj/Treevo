def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of close prices to capture short-term momentum
    ma_10_close = df['close'].rolling(window=10).mean()
    
    # Calculate the 30-day moving average of close prices for longer-term trend
    ma_30_close = df['close'].rolling(window=30).mean()
    
    # Compute the difference between the two moving averages to gauge momentum
    momentum_factor = ma_10_close - ma_30_close
    
    # Calculate the 10-day standard deviation of close prices to represent short-term volatility
    volatility_factor = df['close'].rolling(window=10).std()
    
    # Calculate the 10-day average volume to measure liquidity
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a liquidity factor as the ratio of current volume to the 10-day average volume
    liquidity_factor = df['volume'] / avg_volume
    
    # Integrate intraday data: ratio of (high - open) to (open - low) to capture intraday sentiment
    intraday_sentiment = (df['high'] - df['open']) / (df['open'] - df['low'])
    
    # Incorporate market depth: calculate the 10-day moving average of the amount and volume
    ma_10_amount = df['amount'].rolling(window=10).mean()
    ma_10_volume = df['volume'].rolling(window=10).mean()
    
    # Create a market depth factor as the ratio of the 10-day moving average of amount to volume
    market_depth_factor = ma_10_amount / ma_10_volume
    
    # Calculate the 10-day rolling sum of the amount to capture total transaction value
    rolling_sum_amount = df['amount'].rolling(window=10).sum()
    
    # Create a transaction value factor as the ratio of the 10-day rolling sum of amount to the 10-day average volume
    transaction_value_factor = rolling_sum_amount / avg_volume
    
    # Combine the factors using non-linear combinations
    combined_momentum = np.sqrt(np.abs(momentum_factor))
    combined_volatility = np.log(volatility_factor + 1)
    combined_liquidity = np.power(liquidity_factor, 0.5)
    combined_intraday_sentiment = np.sign(intraday_sentiment) * np.log(np.abs(intraday_sentiment) + 1)
    combined_market_depth = np.log(market_depth_factor + 1)
    combined_transaction_value = np.log(transaction_value_factor + 1)
    
    # Final factor value
    factor_value = (combined_momentum * combined_liquidity * combined_intraday_sentiment * combined_market_depth * combined_transaction_value) / (combined_volatility + 1e-6)
    
    return factor_value
