import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['hlv'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_difference = df['hlv'].rolling(window=20).sum()
    total_volume_20_days = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_average = long_term_weighted_price_difference / total_volume_20_days
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=20).sum() / 20
    
    # Calculate 20-Day Standard Deviation of Daily Returns
    daily_log_returns = np.log(df['close'] / df['close'].shift(1))
    volatility = daily_log_returns.rolling(window=20).std()
    
    # Final Alpha Factor
    final_alpha_factor = (long_term_cumulative_weighted_average 
                          - long_term_average_daily_return) / volatility
    
    return final_alpha_factor.dropna()
