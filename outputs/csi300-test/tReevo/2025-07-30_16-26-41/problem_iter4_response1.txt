import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['high_low_vol'].rolling(window=20).sum()

    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(20)

    # Calculate the Long-Term Cumulative Weighted Average
    total_volume_20_days = df['volume'].rolling(window=20).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume_20_days

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).sum() / 20
    adjusted_long_term_cumulative_weighted_avg = long_term_cumulative_weighted_avg - long_term_avg_daily_return

    # Adjust for Volatility
    daily_returns_pct = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility_20_days = daily_returns_pct.rolling(window=20).std()
    adjusted_for_volatility = adjusted_long_term_cumulative_weighted_avg / volatility_20_days

    # Incorporate Volume-Weighted Intraday Momentum
    intraday_momentum = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_intraday_momentum = intraday_momentum * rolling_avg_volume

    # Determine Reversal Potential
    reversal_potential = np.where(df['close'] > df['open'], 1, -1)
    volume_weighted_reversal_potential = reversal_potential * df['volume']

    # Integrate Volume-Weighted Intraday Momentum and Reversal Potential
    number_of_trades = df['volume']  # Assuming volume as a proxy for the number of trades
    fraction = np.where(number_of_trades < 10000, 0.5, 0.8)
    integrated_momentum = volume_weighted_intraday_momentum - (volume_weighted_reversal_potential * fraction)

    # Final Alpha Factor
    final_alpha_factor = adjusted_for_volatility + integrated_momentum

    return final_alpha_factor
