```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 20-day calculations
    df['HL_Vol_20'] = (df['high'] - df['low']) * df['volume']
    long_term_vol_weighted_diff = df['HL_Vol_20'].rolling(window=20).sum()
    total_volume_220 = df['volume'].rolling(window=220).sum()
    long_term_cumulative_weighted_avg = long_term_vol_weighted_diff / total_volume_220
    
    long_term_open_to_close_mom = df['close'] - df['open'].shift(20)
    long_term_avg_daily_return = df['close'].diff().rolling(window=20).sum() / 20
    daily_returns = df['close'].pct_change()
    long_term_volatility = daily_returns.rolling(window=20).std()
    
    # Final 20-day alpha factor
    long_term_alpha = (long_term_cumulative_weighted_avg + long_term_open_to_close_mom - long_term_avg_daily_return) / long_term_volatility

    # 5-day calculations
    short_term_vol_weighted_diff = (df['high'] - df['low']) * df['volume']
    short_term_vol_weighted_diff = short_term_vol_weighted_diff.rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    short_term_cumulative_weighted_avg = short_term_vol_weighted_diff / total_volume_5
    
    short_term_open_to_close_mom = df['close'] - df['open'].shift(5)
    short_term_avg_daily_return = df['close'].diff().rolling(window=5).sum() / 5
    short_term_volatility = daily_returns.rolling(window=5).std()
    
    # Final 5-day alpha factor
    short_term_alpha = (short_term_cumulative_weighted_avg + short_term_open_to_close_mom - short_term_avg_daily_return) / short_term_volatility

    # Combine 20-day and 5-day metrics
    combined_alpha = (0.7 * long_term_alpha) + (0.3 * short_term_alpha)

    # Adjust by 5-day average volume
    five_day_avg_vol = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha * five_day_avg_vol

    return final_alpha_factor.dropna()
```
