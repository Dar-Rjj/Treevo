```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Long-Term (10-day) Metrics
    df['high_low_vol_10'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff_10 = df['high_low_vol_10'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    long_term_cumulative_weighted_avg_10 = long_term_weighted_price_diff_10 / total_volume_10
    
    long_term_open_close_momentum_10 = df['close'] - df['open'].shift(10)
    
    daily_returns_10 = df['close'].pct_change()
    long_term_avg_daily_return_10 = daily_returns_10.rolling(window=10).mean()
    
    long_term_volatility_10 = daily_returns_10.rolling(window=10).std()
    
    alpha_factor_10 = (long_term_cumulative_weighted_avg_10 + long_term_open_close_momentum_10 - long_term_avg_daily_return_10) / long_term_volatility_10

    # Short-Term (5-day) Metrics
    df['high_low_vol_5'] = (df['high'] - df['low']) * df['volume']
    short_term_weighted_price_diff_5 = df['high_low_vol_5'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    short_term_cumulative_weighted_avg_5 = short_term_weighted_price_diff_5 / total_volume_5
    
    short_term_open_close_momentum_5 = df['close'] - df['open'].shift(5)
    
    daily_returns_5 = df['close'].pct_change()
    short_term_avg_daily_return_5 = daily_returns_5.rolling(window=5).mean()
    
    short_term_volatility_5 = daily_returns_5.rolling(window=5).std()
    
    alpha_factor_5 = (short_term_cumulative_weighted_avg_5 + short_term_open_close_momentum_5 - short_term_avg_daily_return_5) / short_term_volatility_5

    # Combine 10-Day and 5-Day Metrics
    combined_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5

    # Integrate Trading Volume Effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5

    return final_alpha_factor
```
