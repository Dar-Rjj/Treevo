```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Open to Close Price Movement
    open_prices = df['open']
    close_prices = df['close']
    price_movement = (close_prices - open_prices) / open_prices
    
    # Determine Volume Rank for the Day
    daily_volume_sum = df.groupby(df.index.date)['volume'].sum()
    volume_rank = df['volume'].rank(ascending=False, method='first')
    
    # Apply Exponential Moving Average to Volume Ranks
    alpha = 0.2  # Smoothing Factor
    ema_volume_rank = volume_rank.ewm(alpha=alpha, adjust=False).mean()
    
    # Combine Price Movement and EMA of Volume Ranks
    combined_factor = price_movement * ema_volume_rank
    
    return combined_factor
```
