```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day metrics
    df['high_low_vol_10'] = (df['high'] - df['low']) * df['volume']
    df['total_vol_10'] = df['volume'].rolling(window=10).sum()
    df['weighted_price_diff_10'] = df['high_low_vol_10'].rolling(window=10).sum() / df['total_vol_10']
    
    df['open_10_lag'] = df['open'].shift(10)
    df['long_term_momentum_10'] = df['close'] - df['open_10_lag']
    
    df['daily_return_10'] = df['close'].diff()
    df['avg_daily_return_10'] = df['daily_return_10'].rolling(window=10).mean()
    
    df['daily_return_pct_10'] = df['close'].pct_change()
    df['volatility_10'] = df['daily_return_pct_10'].rolling(window=10).std()
    
    df['long_term_alpha_10'] = (df['weighted_price_diff_10'] - df['avg_daily_return_10']) / df['volatility_10']
    
    # Calculate 5-day metrics
    df['high_low_vol_5'] = (df['high'] - df['low']) * df['volume']
    df['total_vol_5'] = df['volume'].rolling(window=5).sum()
    df['weighted_price_diff_5'] = df['high_low_vol_5'].rolling(window=5).sum() / df['total_vol_5']
    
    df['open_5_lag'] = df['open'].shift(5)
    df['short_term_momentum_5'] = df['close'] - df['open_5_lag']
    
    df['daily_return_5'] = df['close'].diff()
    df['avg_daily_return_5'] = df['daily_return_5'].rolling(window=5).mean()
    
    df['daily_return_pct_5'] = df['close'].pct_change()
    df['volatility_5'] = df['daily_return_pct_5'].rolling(window=5).std()
    
    df['short_term_alpha_5'] = (df['weighted_price_diff_5'] - df['avg_daily_return_5']) / df['volatility_5']
    
    # Combine 10-day and 5-day metrics
    df['combined_alpha'] = 0.6 * df['long_term_alpha_10'] + 0.4 * df['short_term_alpha_5']
    
    # Integrate trading volume
    df['avg_vol_5'] = df['volume'].rolling(window=5).mean()
    df['final_alpha_factor'] = df['combined_alpha'] * df['avg_vol_5']
    
    return df['final_alpha_factor'].dropna()
```
