import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day metrics
    df['PriceDiffVol_10'] = ((df['high'] - df['low']) * df['volume']).rolling(10).sum()
    df['TotalVolume_10'] = df['volume'].rolling(10).sum()
    df['WeightedPriceDiff_10'] = df['PriceDiffVol_10'] / df['TotalVolume_10']
    
    df['Momentum_10'] = df['close'] - df['open'].shift(10)
    
    df['AvgDailyReturn_10'] = (df['close'] - df['close'].shift(1)).rolling(10).sum() / 10
    df['StdDev_10'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1).rolling(10).std()
    
    # 5-day metrics
    df['PriceDiffVol_5'] = ((df['high'] - df['low']) * df['volume']).rolling(5).sum()
    df['TotalVolume_5'] = df['volume'].rolling(5).sum()
    df['WeightedPriceDiff_5'] = df['PriceDiffVol_5'] / df['TotalVolume_5']
    
    df['Momentum_5'] = df['close'] - df['open'].shift(5)
    
    df['AvgDailyReturn_5'] = (df['close'] - df['close'].shift(1)).rolling(5).sum() / 5
    df['StdDev_5'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1).rolling(5).std()
    
    # Combine 10-Day and 5-Day Metrics
    df['AlphaFactor_10'] = (df['WeightedPriceDiff_10'] - df['AvgDailyReturn_10']) / df['StdDev_10']
    df['AlphaFactor_5'] = (df['WeightedPriceDiff_5'] - df['AvgDailyReturn_5']) / df['StdDev_5']
    
    # Adaptive Window Adjustment
    df['VolumeEMA_20'] = df['volume'].ewm(span=20, adjust=False).mean()
    df['VolumeAvg_5'] = df['volume'].rolling(5).mean()
    
    df['5DayMetricWeight'] = 0.4
    df.loc[df['VolumeAvg_5'] > df['VolumeEMA_20'], '5DayMetricWeight'] = 0.5
    df.loc[df['VolumeAvg_5'] < df['VolumeEMA_20'], '5DayMetricWeight'] = 0.3
    
    # Final Alpha Factor with Exponential Weights
    df['CombinedAlphaFactor'] = (df['AlphaFactor_10'] * 0.8 + df['AlphaFactor_5'] * 0.2) * df['VolumeAvg_5']
    df['FinalAlphaFactor'] = (df['CombinedAlphaFactor'] * df['5DayMetricWeight'] * 1.5 + 
                              df['CombinedAlphaFactor'] * (1 - df['5DayMetricWeight']) * 0.6)
    
    return df['FinalAlphaFactor'].dropna()
