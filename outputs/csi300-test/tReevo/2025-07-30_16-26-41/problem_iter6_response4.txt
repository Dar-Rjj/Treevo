import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    vol_weighted_price_diff = df['high_low_vol'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    volume_weighted_price_diff = vol_weighted_price_diff / total_volume
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_avg = volume_weighted_price_diff
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).mean()
    adjusted_long_term_cumulative_avg = long_term_cumulative_avg - long_term_avg_daily_return
    
    # Adjust for Long-Term Volatility
    daily_returns_pct = df['close'].pct_change()
    long_term_volatility = daily_returns_pct.rolling(window=20).std()
    adjusted_for_volatility = adjusted_long_term_cumulative_avg / long_term_volatility
    
    # Incorporate Intraday Reversal Signal
    intraday_range = df['high'] - df['low']
    intraday_midpoint = (df['open'] + df['close']) / 2
    intraday_direction = np.where(df['open'] > df['close'], -1, 1)
    intraday_momentum_reversal_score = intraday_range * intraday_direction
    intraday_volatility = df['high'] - df['low']
    adjusted_intraday_score = intraday_momentum_reversal_score * df['volume'] / intraday_volatility
    
    # Final Alpha Factor
    alpha_factor = adjusted_for_volatility + adjusted_intraday_score
    
    return alpha_factor
