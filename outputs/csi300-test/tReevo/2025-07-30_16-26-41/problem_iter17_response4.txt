import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df):
    # Volume-Weighted Moving Averages (VWMA)
    def vwma(price, volume, window):
        return (price * volume).rolling(window=window).sum() / volume.rolling(window=window).sum()
    
    df['VWMA_5'] = vwma(df['close'], df['volume'], 5)
    df['VWMA_20'] = vwma(df['close'], df['volume'], 20)
    df['VWMA_60'] = vwma(df['close'], df['volume'], 60)
    
    # Exponential Moving Averages (EMA)
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['EMA_60'] = df['close'].ewm(span=60, adjust=False).mean()
    
    # Relative Strength Index (RSI) with Exponential Smoothing
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/5, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/5, adjust=False).mean()
    rs = gain / loss
    df['RSI_5'] = 100 - (100 / (1 + rs))
    
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    df['RSI_14'] = 100 - (100 / (1 + rs))
    
    # Volume-Weighted Average Price (VWAP)
    df['VWAP_5'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    df['VWAP_20'] = (df['amount'] / df['volume']).rolling(window=20).mean()
    
    # Volatility Metrics
    df['Returns'] = df['close'].pct_change()
    df['Std_20'] = df['Returns'].rolling(window=20).std()
    
    tr = df[['high', 'low', 'close']].shift(1).max(axis=1) - df[['high', 'low', 'close']].shift(1).min(axis=1)
    df['ATR_14'] = tr.rolling(window=14).mean()
    
    # Combined Momentum and Volatility Factor
    df['Momentum_Volatility_Factor'] = (df['VWMA_5'] - df['VWMA_60']) / df['Std_20']
    df['Momentum_Volatility_Factor'] = df['Momentum_Volatility_Factor'] * (100 - df['RSI_14']) / 100
    df['Momentum_Volatility_Factor'] = df['Momentum_Volatility_Factor'] * df['ATR_14']
    
    # Liquidity and Recent Market Conditions
    df['Volume_Ratio'] = df['volume'] / df['volume'].rolling(window=5).mean()
    df['Final_Factor'] = df['Momentum_Volatility_Factor'] * df['Volume_Ratio']
    
    return df['Final_Factor'].dropna()
