```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-Day Weighted Price Difference
    df['high_low_diff'] = (df['high'] - df['low']) * df['volume']
    ten_day_weighted_price_diff = df['high_low_diff'].rolling(window=10).sum()
    ten_day_total_volume = df['volume'].rolling(window=10).sum()
    
    # Determine 10-Day Open to Close Momentum
    ten_day_open_close_momentum = df['close'] - df['open'].shift(10)
    
    # Calculate 10-Day Cumulative Weighted Average
    ten_day_cumulative_weighted_avg = ten_day_weighted_price_diff / ten_day_total_volume
    
    # Subtract 10-Day Average Daily Return
    ten_day_daily_returns = df['close'] - df['close'].shift(1)
    ten_day_average_daily_return = ten_day_daily_returns.rolling(window=10).mean()
    
    # Adjust for 10-Day Dynamic Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ten_day_volatility = daily_returns.ewm(span=10, adjust=False).std()
    
    # Final 10-Day Alpha Factor
    ten_day_alpha_factor = (ten_day_cumulative_weighted_avg - ten_day_average_daily_return) / ten_day_volatility
    
    # Recalculate Metrics Using a 5-Day Window
    five_day_weighted_price_diff = df['high_low_diff'].rolling(window=5).sum()
    five_day_total_volume = df['volume'].rolling(window=5).sum()
    five_day_open_close_momentum = df['close'] - df['open'].shift(5)
    five_day_cumulative_weighted_avg = five_day_weighted_price_diff / five_day_total_volume
    five_day_daily_returns = df['close'] - df['close'].shift(1)
    five_day_average_daily_return = five_day_daily_returns.rolling(window=5).mean()
    five_day_volatility = daily_returns.ewm(span=5, adjust=False).std()
    five_day_alpha_factor = (five_day_cumulative_weighted_avg - five_day_average_daily_return) / five_day_volatility
    
    # Combine 10-Day and 5-Day Metrics
    combined_alpha_factor = 0.6 * ten_day_alpha_factor + 0.4 * five_day_alpha_factor
    
    # Integrate Trading Volume Effectively
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    ten_day_avg_volume = df['volume'].rolling(window=10).mean()
    normalized_five_day_avg_volume = five_day_avg_volume / ten_day_avg_volume
    final_alpha_factor = combined_alpha_factor * normalized_five_day_avg_volume
    
    return final_alpha_factor
```
