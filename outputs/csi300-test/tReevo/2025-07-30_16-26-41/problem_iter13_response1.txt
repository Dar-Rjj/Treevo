import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['HighLowVol'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['HighLowVol'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    dynamic_long_term_cumulative_weighted_average = long_term_weighted_price_diff / total_volume_10
    
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(10)
    long_term_average_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.rolling(window=10).std()
    
    long_term_alpha_factor = (dynamic_long_term_cumulative_weighted_average - long_term_average_daily_return) / long_term_volatility
    
    # 5-day calculations
    short_term_weighted_price_diff = (df['high'] - df['low']) * df['volume']
    short_term_weighted_price_diff = short_term_weighted_price_diff.rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    dynamic_short_term_cumulative_weighted_average = short_term_weighted_price_diff / total_volume_5
    
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(5)
    short_term_average_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    short_term_volatility = daily_returns.rolling(window=5).std()
    
    short_term_alpha_factor = (dynamic_short_term_cumulative_weighted_average - short_term_average_daily_return) / short_term_volatility
    
    # Combine 10-day and 5-day metrics
    combined_alpha_factor = 0.6 * long_term_alpha_factor + 0.4 * short_term_alpha_factor
    
    # Integrate trading volume effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    combined_alpha_factor *= avg_volume_5
    
    # Incorporate recent data with exponential weights
    combined_alpha_factor = 0.8 * long_term_alpha_factor + 0.2 * short_term_alpha_factor
    combined_alpha_factor *= avg_volume_5
    
    return combined_alpha_factor
