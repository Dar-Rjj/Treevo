import pandas as pd
import numpy as np

def heuristics_v2(df):
    """
    Generate a novel alpha factor based on Exponential Moving Average Crossover with Volume-Weighted Momentum.
    
    Parameters:
    df (pd.DataFrame): DataFrame with index as date and columns as market features (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: Factor values indexed by date.
    """
    # Calculate short-term and long-term EMAs
    short_window = 10
    long_window = 50
    
    df['short_ema'] = df['close'].ewm(span=short_window, adjust=False).mean()
    df['long_ema'] = df['close'].ewm(span=long_window, adjust=False).mean()
    
    # Calculate Volume-Weighted Momentum
    n = 20
    df['price_diff'] = df['close'] - df['close'].shift(n)
    df['volume_sum'] = df['volume'].rolling(window=n).sum()
    df['vol_weighted_momentum'] = df['price_diff'] * (df['volume_sum'] / n)
    
    # Combine the EMAs and Volume-Weighted Momentum
    df['ema_diff'] = df['short_ema'] - df['long_ema']
    df['combined_factor'] = df['ema_diff'] * df['vol_weighted_momentum']
    
    # Adjust for Liquidity
    df['average_volume'] = df['volume'].rolling(window=n).mean()
    df['final_factor'] = df['combined_factor'] * df['average_volume']
    
    # Simplify the calculations
    df['final_factor'] = df['final_factor'].fillna(0)  # Fill NaNs with 0
    
    return df['final_factor']
