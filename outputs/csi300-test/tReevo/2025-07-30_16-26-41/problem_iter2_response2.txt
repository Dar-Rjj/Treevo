import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['price_diff'] = (df['high'] - df['low']) * df['volume']
    long_term_price_diff = df['price_diff'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    open_shifted_20 = df['open'].shift(20)
    close_current = df['close']
    long_term_momentum = close_current - open_shifted_20
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cum_weighted_avg = long_term_price_diff / total_volume
    
    # Subtract Long-Term Average Daily Return
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).mean()
    
    # Adjust for Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns.rolling(window=20).std()
    
    # Final Alpha Factor
    final_alpha_factor = (long_term_cum_weighted_avg - long_term_avg_daily_return) / volatility
    
    return final_alpha_factor.dropna()
