```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Logarithmic Daily Returns using Close price
    df['log_returns'] = np.log(df['close']).diff()
    
    # Set Lambda (decay factor) for EWMA
    lambda_ = 0.94
    
    # Calculate Exponential Weighted Moving Average (EWMA) of Daily Returns
    def ewma(x, alpha):
        return x.ewm(alpha=alpha, adjust=False).mean().iloc[-1]

    df['ewma_returns'] = df['log_returns'].rolling(window=len(df), min_periods=1).apply(lambda x: ewma(x, 1 - lambda_), raw=False)
    
    # Calculate Historical Volatility
    df['abs_log_returns'] = df['log_returns'].abs()
    df['volatility'] = df['abs_log_returns'].rolling(window=len(df), min_periods=1).apply(lambda x: np.sqrt(np.sum(x**2)), raw=False)
    
    # Calculate Inverse of Historical Volatility
    df['inverse_volatility'] = 1 / df['volatility']
    
    # Combine EWMA and Inverse of Volatility
    df['alpha_factor'] = df['ewma_returns'] * df['inverse_volatility']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
