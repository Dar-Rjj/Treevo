```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    total_volume = df['volume'].rolling(window=5).sum()
    short_term_weighted_price_diff = weighted_price_diff.rolling(window=5).sum() / total_volume
    
    # Determine Short-Term Open to Close Momentum
    open_to_close_momentum = df['close'] - df['open'].shift(5)
    
    # Calculate the Short-Term Cumulative Weighted Average
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff + open_to_close_momentum
    
    # Subtract Short-Term Average Daily Return
    average_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    short_term_adjusted_cumulative_avg = short_term_cumulative_weighted_avg - average_daily_return
    
    # Adjust for Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns.rolling(window=5).std()
    
    # Final Alpha Factor
    alpha_factor = short_term_adjusted_cumulative_avg / volatility
    
    return alpha_factor
```
