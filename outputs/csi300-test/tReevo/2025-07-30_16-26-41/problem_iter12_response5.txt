```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 7-day metrics
    df['HLV_7'] = (df['high'] - df['low']) * df['volume']
    df['HLV_7_sum'] = df['HLV_7'].rolling(window=7).sum()
    df['volume_7_sum'] = df['volume'].rolling(window=7).sum()
    df['open_7_lag'] = df['open'].shift(7)
    df['long_term_momentum_7'] = df['close'] - df['open_7_lag']
    df['long_term_weighted_price_diff_7'] = df['HLV_7_sum'] / df['volume_7_sum']
    df['daily_return_7'] = df['close'].pct_change()
    df['avg_daily_return_7'] = df['daily_return_7'].rolling(window=7).mean()
    df['std_dev_7'] = df['daily_return_7'].rolling(window=7).std()

    # Final 7-Day Alpha Factor
    df['alpha_7'] = (df['long_term_weighted_price_diff_7'] - df['avg_daily_return_7']) / df['std_dev_7']

    # Calculate 3-day metrics
    df['HLV_3'] = (df['high'] - df['low']) * df['volume']
    df['HLV_3_sum'] = df['HLV_3'].rolling(window=3).sum()
    df['volume_3_sum'] = df['volume'].rolling(window=3).sum()
    df['open_3_lag'] = df['open'].shift(3)
    df['short_term_momentum_3'] = df['close'] - df['open_3_lag']
    df['short_term_weighted_price_diff_3'] = df['HLV_3_sum'] / df['volume_3_sum']
    df['avg_daily_return_3'] = df['daily_return_7'].rolling(window=3).mean()
    df['std_dev_3'] = df['daily_return_7'].rolling(window=3).std()

    # Final 3-Day Alpha Factor
    df['alpha_3'] = (df['short_term_weighted_price_diff_3'] - df['avg_daily_return_3']) / df['std_dev_3']

    # Combine 7-day and 3-day metrics
    df['combined_alpha'] = 0.6 * df['alpha_7'] + 0.4 * df['alpha_3']

    # Integrate trading volume effectively
    df['volume_3_avg'] = df['volume'].rolling(window=3).mean()
    df['volume_20_avg'] = df['volume'].rolling(window=20).mean()
    df['volume_trend_indicator'] = (df['volume_3_avg'] - df['volume_20_avg']) / df['volume_20_avg']

    # Adjust the combined alpha factor by volume trend indicator
    df['final_alpha'] = df['combined_alpha'] * df['volume_trend_indicator']

    # Return the final alpha factor
    return df['final_alpha']
```
