```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-Day Metrics
    df['daily_return'] = df['close'].pct_change()
    df['weighted_price_diff_10'] = ((df['high'] - df['low']) * df['volume']).rolling(window=10).sum()
    df['total_volume_10'] = df['volume'].rolling(window=10).sum()
    df['cumulative_weighted_avg_10'] = df['weighted_price_diff_10'] / df['total_volume_10']
    df['open_close_momentum_10'] = df['close'] - df['open'].shift(10)
    df['average_daily_return_10'] = df['close'].diff().rolling(window=10).mean()
    df['volatility_10'] = df['daily_return'].rolling(window=10).std()

    # 5-Day Metrics
    df['weighted_price_diff_5'] = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    df['total_volume_5'] = df['volume'].rolling(window=5).sum()
    df['cumulative_weighted_avg_5'] = df['weighted_price_diff_5'] / df['total_volume_5']
    df['open_close_momentum_5'] = df['close'] - df['open'].shift(5)
    df['average_daily_return_5'] = df['close'].diff().rolling(window=5).mean()
    df['volatility_5'] = df['daily_return'].rolling(window=5).std()

    # Final 10-Day Alpha Factor
    df['alpha_factor_10'] = (df['cumulative_weighted_avg_10'] - df['average_daily_return_10']) / df['volatility_10']

    # Final 5-Day Alpha Factor
    df['alpha_factor_5'] = (df['cumulative_weighted_avg_5'] - df['average_daily_return_5']) / df['volatility_5']

    # Combine 10-Day and 5-Day Metrics
    df['combined_alpha_factor'] = 0.6 * df['alpha_factor_10'] + 0.4 * df['alpha_factor_5']

    # Integrate Trading Volume Effectively
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['final_alpha_factor'] = df['combined_alpha_factor'] * df['avg_volume_5']

    return df['final_alpha_factor'].dropna()
```
