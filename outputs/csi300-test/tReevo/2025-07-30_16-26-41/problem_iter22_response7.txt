```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily log returns
    df['log_ret'] = np.log(df['close']).diff()
    
    # Calculate the average of (high + low) / 2
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate the moving average of average prices over a 20-day window
    df['avg_price_ma_20'] = df['avg_price'].rolling(window=20).mean()
    
    # Calculate the volatility of log returns over a 20-day window
    df['volatility'] = df['log_ret'].rolling(window=20).std()
    
    # Calculate the ratio of close price to the 20-day moving average of close price
    df['price_to_ma_ratio'] = df['close'] / df['close'].rolling(window=20).mean()
    
    # Heuristic factor: Momentum and Volatility Adjusted Price-to-MA Ratio
    df['heur_factor'] = (df['log_ret'].rolling(window=5).sum() / df['volatility']) * df['price_to_ma_ratio']
    
    # Drop NaN values resulting from rolling calculations
    df.dropna(inplace=True)
    
    # Return the heuristic factor as a pandas Series
    return df['heur_factor']
```
