```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Exponential Weighted Dynamic Long-Term Price Difference
    ew_long_price_diff = (df['high'] - df['low']) * df['volume']
    ew_long_price_diff = ew_long_price_diff.ewm(span=10).mean()
    ew_long_volume = df['volume'].ewm(span=10).mean()
    
    # Determine Exponential Weighted Dynamic Long-Term Open to Close Momentum
    ew_long_momentum = (df['close'] - df['open'].shift(10)).ewm(span=10).mean()
    
    # Calculate the Exponential Weighted Dynamic Long-Term Cumulative Average
    ew_long_cumulative_avg = ew_long_price_diff / ew_long_volume
    
    # Subtract Exponential Weighted Dynamic Long-Term Average Daily Return
    ew_long_daily_return = (df['close'] - df['close'].shift(1)).ewm(span=10).mean()
    ew_long_average_daily_return = ew_long_daily_return / 10
    
    # Adjust for Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ew_long_volatility = daily_returns.rolling(window=10).std().ewm(span=10).mean()
    
    # Final Exponential Weighted Dynamic Long-Term Alpha Factor
    long_term_alpha = (ew_long_cumulative_avg - ew_long_average_daily_return) / ew_long_volatility
    
    # Recalculate Exponential Weighted Short-Term Metrics
    ew_short_price_diff = (df['high'] - df['low']) * df['volume']
    ew_short_price_diff = ew_short_price_diff.ewm(span=5).mean()
    ew_short_volume = df['volume'].ewm(span=5).mean()
    
    # Determine Exponential Weighted Short-Term Open to Close Momentum
    ew_short_momentum = (df['close'] - df['open'].shift(5)).ewm(span=5).mean()
    
    # Calculate the Exponential Weighted Short-Term Cumulative Average
    ew_short_cumulative_avg = ew_short_price_diff / ew_short_volume
    
    # Subtract Exponential Weighted Short-Term Average Daily Return
    ew_short_daily_return = (df['close'] - df['close'].shift(1)).ewm(span=5).mean()
    ew_short_average_daily_return = ew_short_daily_return / 5
    
    # Adjust for Volatility
    ew_short_volatility = daily_returns.rolling(window=5).std().ewm(span=5).mean()
    
    # Final Exponential Weighted Dynamic Short-Term Alpha Factor
    short_term_alpha = (ew_short_cumulative_avg - ew_short_average_daily_return) / ew_short_volatility
    
    # Combine Long-Term and Short-Term Metrics
    combined_alpha = 0.6 * long_term_alpha + 0.4 * short_term_alpha
    
    # Integrate Trading Volume Effectively
    ew_5_day_volume = df['volume'].ewm(span=5).mean()
    combined_alpha *= ew_5_day_volume
    
    # Consider Volume Spikes
    volume_spike_factor = df['volume'] / ew_5_day_volume
    final_alpha = combined_alpha * volume_spike_factor
    
    return final_alpha
```
