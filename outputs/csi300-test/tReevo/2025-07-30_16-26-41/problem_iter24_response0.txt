```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative range of the day
    df['range'] = df['high'] - df['low']
    df['relative_range'] = df['range'] / df['close']

    # Calculate the relative movement of the day
    df['movement'] = df['close'] - df['open']
    df['relative_movement'] = df['movement'] / df['close']

    # Calculate the relative volume
    mean_volume = df['volume'].rolling(window=20).mean()
    df['relative_volume'] = df['volume'] / mean_volume

    # Calculate the weighted amount
    df['weighted_amount'] = (df['amount'] * df['close']) / df['volume']

    # Alpha factor: a combination of these relative measures
    alpha_factor = (df['relative_range'] + 
                    df['relative_movement'] + 
                    df['relative_volume'] + 
                    df['weighted_amount']) / 4

    return alpha_factor
```
