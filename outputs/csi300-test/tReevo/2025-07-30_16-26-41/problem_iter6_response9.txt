```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    total_volume = df['volume'].rolling(window=25).sum()
    long_term_weighted_price_diff = weighted_price_diff.rolling(window=25).sum() / total_volume

    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(25)

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff

    # Subtract Long-Term Average Daily Return
    daily_return = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_return.rolling(window=25).sum() / 25
    long_term_cumulative_weighted_avg -= long_term_avg_daily_return

    # Incorporate Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(25)
    combined_volume_momentum = 0.4 * volume_trend + 0.6 * long_term_cumulative_weighted_avg

    # Incorporate Price Trend
    price_trend = df['close'] - df['close'].shift(25)
    final_factor = 0.3 * price_trend + 0.7 * combined_volume_momentum

    # Adjust for Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns.rolling(window=25).std()
    final_alpha_factor = final_factor / volatility

    return final_alpha_factor
```
