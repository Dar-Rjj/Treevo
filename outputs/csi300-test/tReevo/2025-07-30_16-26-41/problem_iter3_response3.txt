```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    sum_weighted_price_diff = weighted_price_diff.rolling(window=20).sum()
    sum_volume = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff = sum_weighted_price_diff / sum_volume

    # Determine Long-Term Open to Close Momentum
    open_20_days_ago = df['open'].shift(20)
    close_today = df['close']
    long_term_momentum = close_today - open_20_days_ago

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff + long_term_momentum

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'].pct_change()
    long_term_average_daily_return = daily_returns.rolling(window=20).mean()
    long_term_cumulative_weighted_avg -= long_term_average_daily_return

    # Adjust for Volatility
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    final_alpha_factor = long_term_cumulative_weighted_avg / volatility

    return final_alpha_factor
```
