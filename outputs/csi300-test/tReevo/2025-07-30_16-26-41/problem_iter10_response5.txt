import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Long-term calculations (20 days)
    long_term_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=20).sum()
    total_volume_220 = df['volume'].rolling(window=220).sum()
    long_term_momentum = df['close'] - df['open'].shift(20)
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume_220
    long_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=20).sum() / 20
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.rolling(window=20).std()
    
    # Final Long-Term Alpha Factor
    long_term_alpha_factor = (long_term_cumulative_weighted_avg - long_term_avg_daily_return) / long_term_volatility
    
    # Short-term calculations (5 days)
    short_term_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    short_term_momentum = df['close'] - df['open'].shift(5)
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff / total_volume_5
    short_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    short_term_volatility = daily_returns.rolling(window=5).std()
    
    # Final Short-Term Alpha Factor
    short_term_alpha_factor = (short_term_cumulative_weighted_avg - short_term_avg_daily_return) / short_term_volatility
    
    # Combine Long-Term and Short-Term Metrics
    combined_alpha_factor = 0.7 * long_term_alpha_factor + 0.3 * short_term_alpha_factor
    
    # Integrate Trading Volume Effectively
    volume_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5
    
    return final_alpha_factor.dropna()
