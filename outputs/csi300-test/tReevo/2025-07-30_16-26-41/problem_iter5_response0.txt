```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    total_weighted_price_diff = weighted_price_diff.rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff = total_weighted_price_diff / total_volume

    # Determine Long-Term Open to Close Momentum
    open_t_minus_20 = df['open'].shift(20)
    close_t = df['close']
    long_term_momentum = close_t - open_t_minus_20

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff * long_term_momentum

    # Subtract Long-Term Average Daily Return
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).mean()
    long_term_cumulative_weighted_avg_adjusted = long_term_cumulative_weighted_avg - long_term_avg_daily_return

    # Adjust for Volatility
    std_dev_20_days = daily_returns.rolling(window=20).std()
    final_alpha_factor = long_term_cumulative_weighted_avg_adjusted / std_dev_20_days

    return final_alpha_factor
```
