```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['hlv'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['hlv'].rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume
    
    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(10)
    
    # Calculate the Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=10).sum() / 10
    
    # Adjust for Volatility
    daily_pct_change = daily_returns / df['close'].shift(1)
    ten_day_volatility = daily_pct_change.rolling(window=10).std()
    
    # Final Alpha Factor
    alpha_factor = ((long_term_cumulative_weighted_avg + long_term_open_to_close_momentum) 
                    - long_term_avg_daily_return) / ten_day_volatility
    
    return alpha_factor
```
