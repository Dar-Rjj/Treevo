import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term Weighted Price Difference (10-day)
    hl_diff = (df['high'] - df['low']) * df['volume']
    total_volume_10 = df['volume'].rolling(window=10).sum()
    dynamic_long_term_weighted_price_diff = hl_diff.rolling(window=10).sum() / total_volume_10
    
    # Determine Dynamic Long-Term Open to Close Momentum (10-day)
    dynamic_long_term_momentum = df['close'] - df['open'].shift(10)
    
    # Calculate the Dynamic Long-Term Cumulative Weighted Average (10-day)
    dynamic_long_term_cumulative_avg = dynamic_long_term_weighted_price_diff * dynamic_long_term_momentum
    
    # Subtract Dynamic Long-Term Average Daily Return (10-day)
    daily_returns_10 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    dynamic_long_term_avg_daily_return = daily_returns_10.rolling(window=10).mean()
    dynamic_long_term_cumulative_avg -= dynamic_long_term_avg_daily_return
    
    # Adjust for Volatility (10-day)
    std_dev_10 = daily_returns_10.rolling(window=10).std()
    adjusted_factor_10 = dynamic_long_term_cumulative_avg / std_dev_10
    
    # Recalculate the above metrics using a 5-day window
    hl_diff_5 = (df['high'] - df['low']) * df['volume']
    total_volume_5 = df['volume'].rolling(window=5).sum()
    dynamic_short_term_weighted_price_diff = hl_diff_5.rolling(window=5).sum() / total_volume_5
    dynamic_short_term_momentum = df['close'] - df['open'].shift(5)
    dynamic_short_term_cumulative_avg = dynamic_short_term_weighted_price_diff * dynamic_short_term_momentum
    daily_returns_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    dynamic_short_term_avg_daily_return = daily_returns_5.rolling(window=5).mean()
    dynamic_short_term_cumulative_avg -= dynamic_short_term_avg_daily_return
    std_dev_5 = daily_returns_5.rolling(window=5).std()
    adjusted_factor_5 = dynamic_short_term_cumulative_avg / std_dev_5
    
    # Combine 10-day and 5-day metrics
    combined_factor = 0.6 * adjusted_factor_10 + 0.4 * adjusted_factor_5
    
    # Integrate Trading Volume Effectively
    volume_weighted_price_diff_5 = (hl_diff_5.rolling(window=5).sum()) / total_volume_5
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_enhanced_alpha_factor = combined_factor * avg_volume_5
    
    return final_enhanced_alpha_factor
