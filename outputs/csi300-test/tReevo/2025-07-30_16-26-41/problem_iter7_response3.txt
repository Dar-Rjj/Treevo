import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    total_weighted_price_diff = weighted_price_diff.rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    long_term_weighted_price_diff = total_weighted_price_diff / total_volume

    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(10)

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_avg = long_term_weighted_price_diff

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=10).sum() / 10

    # Adjust for Volatility
    daily_returns_pct = df['close'].pct_change()
    volatility = daily_returns_pct.rolling(window=10).std()

    # Final Alpha Factor
    alpha_factor = (long_term_cumulative_avg - long_term_average_daily_return) / volatility

    return alpha_factor
