```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-High Ratio and Low-Open Ratio
    close_high_ratio = df['close'] / df['high']
    low_open_ratio = df['low'] / df['open']
    
    # Assign Weights and Combine Ratios
    weight_close_high = 0.6
    weight_low_open = 0.4
    combined_ratios = (close_high_ratio * weight_close_high) + (low_open_ratio * weight_low_open)
    
    # Apply Volume Filter
    median_volume = df['volume'].median()
    trend_strength = combined_ratios.where(df['volume'] > median_volume, 0)
    
    # Incorporate Momentum
    momentum_20_day = df['close'] - df['close'].shift(20)
    momentum_10_day = df['close'] - df['close'].shift(10)
    
    # Assign Weights and Combine Momentum Factors
    weight_momentum_20 = 0.7
    weight_momentum_10 = 0.3
    combined_momentum = (momentum_20_day * weight_momentum_20) + (momentum_10_day * weight_momentum_10)
    
    # Incorporate Volatility
    volatility_20_day = df['close'].rolling(window=20).std()
    volatility_10_day = df['close'].rolling(window=10).std()
    
    # Assign Weights and Combine Volatility Factors
    weight_volatility_20 = 0.6
    weight_volatility_10 = 0.4
    combined_volatility = (volatility_20_day * weight_volatility_20) + (volatility_10_day * weight_volatility_10)
    
    # Final Alpha Factor
    alpha_factor = trend_strength + combined_momentum - combined_volatility
    
    return alpha_factor
```
