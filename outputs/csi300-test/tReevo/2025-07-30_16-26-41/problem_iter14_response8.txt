```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Helper function to calculate the weighted price difference
    def weighted_price_diff(data, window):
        return (data['high'] - data['low']) * data['volume'].rolling(window=window).sum() / data['volume'].rolling(window=window).sum()
    
    # Helper function to calculate the open to close momentum
    def open_to_close_momentum(data, window):
        return data['close'] - data['open'].shift(window)
    
    # Helper function to calculate the average daily return
    def avg_daily_return(data, window):
        return (data['close'] - data['close'].shift(1)).rolling(window=window).sum() / window
    
    # Helper function to calculate the standard deviation of daily returns
    def std_daily_return(data, window):
        return (data['close'] - data['close'].shift(1)) / data['close'].shift(1).rolling(window=window).std()
    
    # 10-day metrics
    df['10_day_weighted_price_diff'] = weighted_price_diff(df, 10)
    df['10_day_open_to_close_momentum'] = open_to_close_momentum(df, 10)
    df['10_day_avg_daily_return'] = avg_daily_return(df, 10)
    df['10_day_std_daily_return'] = std_daily_return(df, 10)
    df['10_day_alpha_factor'] = (df['10_day_weighted_price_diff'] - df['10_day_avg_daily_return']) / df['10_day_std_daily_return']
    
    # 5-day metrics
    df['5_day_weighted_price_diff'] = weighted_price_diff(df, 5)
    df['5_day_open_to_close_momentum'] = open_to_close_momentum(df, 5)
    df['5_day_avg_daily_return'] = avg_daily_return(df, 5)
    df['5_day_std_daily_return'] = std_daily_return(df, 5)
    df['5_day_alpha_factor'] = (df['5_day_weighted_price_diff'] - df['5_day_avg_daily_return']) / df['5_day_std_daily_return']
    
    # Combine 10-day and 5-day metrics
    df['combined_alpha_factor'] = 0.6 * df['10_day_alpha_factor'] + 0.4 * df['5_day_alpha_factor']
    
    # Integrate trading volume effectively
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['combined_alpha_factor'] *= df['5_day_avg_volume']
    
    # Incorporate recent data with exponential weights
    df['10_day_exp_weighted_alpha_factor'] = df['10_day_alpha_factor'].ewm(alpha=0.8).mean()
    df['5_day_exp_weighted_alpha_factor'] = df['5_day_alpha_factor'].ewm(alpha=0.2).mean()
    df['final_alpha_factor'] = 0.6 * df['10_day_exp_weighted_alpha_factor'] + 0.4 * df['5_day_exp_weighted_alpha_factor']
    df['final_alpha_factor'] *= df['5_day_avg_volume']
    
    return df['final_alpha_factor']
```
