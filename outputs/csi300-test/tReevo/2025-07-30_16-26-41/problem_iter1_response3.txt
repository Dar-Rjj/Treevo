import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Intraday Volume Proportion
    total_volume = df['volume']
    intraday_volume = (df['close'] - df['open']) / (df['high'] - df['low']) * df['volume']
    intraday_volume_proportion = intraday_volume / total_volume
    
    # Multiply High-Low Spread by Intraday Volume Proportion
    factor_values = high_low_spread * intraday_volume_proportion
    
    return factor_values
