```python
import pandas as pd

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate Short-Term Weighted Price Difference
    weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    short_term_weighted_avg = weighted_price_diff / total_volume

    # Determine Short-Term Open to Close Momentum
    open_to_close_momentum = df['close'] - df['open'].shift(5)

    # Subtract Short-Term Average Daily Return
    short_term_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5

    # Adjust for Volatility
    range_based_volatility = ((df['high'] - df['low']) / df['close']).rolling(window=5).sum() / 5

    # Final Alpha Factor
    alpha_factor = (short_term_weighted_avg + open_to_close_momentum - short_term_daily_return) / range_based_volatility

    return alpha_factor.dropna()
```
