import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_high_low_vol_sum = df['high_low_vol'].rolling(window=10).sum()
    long_term_volume_sum = df['volume'].rolling(window=10).sum()
    
    # Determine Long-Term Open to Close Momentum
    open_10_days_ago = df['open'].shift(10)
    close_today = df['close']
    long_term_momentum = close_today - open_10_days_ago
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_weighted_price_diff = long_term_high_low_vol_sum / long_term_volume_sum
    
    # Subtract Long-Term Average Daily Return
    daily_return = df['close'].diff()
    long_term_average_daily_return = daily_return.rolling(window=10).mean()
    
    # Adjust for Volatility: 10-Day Standard Deviation of Daily Returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=10).std()
    
    # Final Alpha Factor
    alpha_factor = (long_term_weighted_price_diff - long_term_average_daily_return) / volatility
    
    return alpha_factor
