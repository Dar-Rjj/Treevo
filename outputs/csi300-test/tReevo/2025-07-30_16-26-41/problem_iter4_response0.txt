```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    open_20_days_ago = df['open'].shift(20)
    long_term_momentum = df['close'] - open_20_days_ago
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_weighted_avg = (weighted_price_diff / total_volume) * long_term_momentum
    
    # Subtract Long-Term Average Daily Return
    daily_returns = (df['close'] - df['close'].shift(1)).rolling(window=20).sum()
    avg_daily_return = daily_returns / 20
    adjusted_long_term_weighted_avg = long_term_weighted_avg - avg_daily_return
    
    # Adjust for Advanced Volatility
    daily_returns_vol = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev = daily_returns_vol.rolling(window=20).std()
    
    true_range = np.maximum.reduce([df['high'] - df['low'], 
                                    np.abs(df['high'] - df['close'].shift(1)), 
                                    np.abs(df['low'] - df['close'].shift(1))])
    atr = true_range.rolling(window=20).mean()
    
    combined_volatility = (std_dev + atr) / 2
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_long_term_weighted_avg / combined_volatility
    
    return final_alpha_factor
```
