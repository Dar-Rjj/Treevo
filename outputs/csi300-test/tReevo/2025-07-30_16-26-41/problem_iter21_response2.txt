import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Short-term Momentum Analysis
    sma_5 = df['close'].rolling(window=5).mean()
    short_momentum = df['close'] - sma_5
    
    # Adaptive Volatility Measure
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=10).std()
    
    # Volume Trend Indicator
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_trend = df['volume'] / avg_volume_20
    
    # Dynamic Weight Adjustment
    momentum_weight = 1 / (1 + volatility)
    volatility_weight = volatility
    volume_trend_weight = 1 - momentum_weight
    
    # Combine the factors with dynamic weights
    combined_factor = (short_momentum * momentum_weight +
                       volatility * volatility_weight +
                       volume_trend * volume_trend_weight)
    
    # Incorporate EMA for Smoothing
    ema_combined_factor = combined_factor.ewm(span=10, adjust=False).mean()
    
    # Liquidity Consideration
    liquidity_adjustment = 1 / (df['high'] - df['low']).rolling(window=30).mean()
    
    # Final Alpha Factor Synthesis
    alpha_factor = ema_combined_factor * liquidity_adjustment
    
    return alpha_factor.dropna()
