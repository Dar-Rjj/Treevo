import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) metrics
    df['High_Low_Volume_10'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_10'] = df['volume'].rolling(window=10).sum()
    df['Weighted_Price_Diff_10'] = df['High_Low_Volume_10'].rolling(window=10).sum() / df['Total_Volume_10']
    
    df['Long_Term_Momentum_10'] = df['close'] - df['open'].shift(10)
    df['Cumulative_Weighted_Avg_10'] = df['Weighted_Price_Diff_10']
    
    daily_returns_10 = df['close'] - df['close'].shift(1)
    df['Avg_Daily_Return_10'] = daily_returns_10.rolling(window=10).sum() / 10
    df['Adjusted_Cumulative_Weighted_Avg_10'] = df['Cumulative_Weighted_Avg_10'] - df['Avg_Daily_Return_10']
    
    ewm_std_10 = daily_returns_10.ewm(span=10, adjust=False).std()
    df['10_Day_Alpha_Factor'] = df['Adjusted_Cumulative_Weighted_Avg_10'] / ewm_std_10
    
    # Calculate Short-Term (5-day) metrics
    df['High_Low_Volume_5'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_5'] = df['volume'].rolling(window=5).sum()
    df['Weighted_Price_Diff_5'] = df['High_Low_Volume_5'].rolling(window=5).sum() / df['Total_Volume_5']
    
    df['Short_Term_Momentum_5'] = df['close'] - df['open'].shift(5)
    df['Cumulative_Weighted_Avg_5'] = df['Weighted_Price_Diff_5']
    
    daily_returns_5 = df['close'] - df['close'].shift(1)
    df['Avg_Daily_Return_5'] = daily_returns_5.rolling(window=5).sum() / 5
    df['Adjusted_Cumulative_Weighted_Avg_5'] = df['Cumulative_Weighted_Avg_5'] - df['Avg_Daily_Return_5']
    
    ewm_std_5 = daily_returns_5.ewm(span=5, adjust=False).std()
    df['5_Day_Alpha_Factor'] = df['Adjusted_Cumulative_Weighted_Avg_5'] / ewm_std_5
    
    # Combine 10-Day and 5-Day Metrics with Dynamic Weights
    relative_volatility_ratio = ewm_std_5 / ewm_std_10
    df['10_Day_Weight'] = 1 / (1 + relative_volatility_ratio)
    df['5_Day_Weight'] = relative_volatility_ratio / (1 + relative_volatility_ratio)
    
    df['Combined_Alpha_Factor'] = (df['10_Day_Alpha_Factor'] * df['10_Day_Weight']) + \
                                  (df['5_Day_Alpha_Factor'] * df['5_Day_Weight'])
    
    # Integrate Trading Volume Effectively
    df['5_Day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['Final_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['5_Day_Avg_Volume']
    
    return df['Final_Alpha_Factor']
