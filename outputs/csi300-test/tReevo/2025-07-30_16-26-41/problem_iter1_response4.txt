```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position within Intraday Range
    closing_position = (df['close'] - df['low']) / intraday_range
    
    # Calculate Opening Position within Intraday Range
    opening_position = (df['open'] - df['low']) / intraday_range
    
    # Determine Reversal Potential
    reversal_potential = closing_position - opening_position
    reversal_potential = reversal_potential.apply(lambda x: 1 if x > 0 else -1)
    
    # Weight by Volume for Confidence
    factor = reversal_potential * df['volume']
    
    return factor
```
