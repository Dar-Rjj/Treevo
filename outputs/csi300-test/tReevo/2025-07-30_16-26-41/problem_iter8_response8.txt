import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['hlv'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff = df['hlv'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()

    # Determine Long-Term Open to Close Momentum
    open_t_minus_20 = df['open'].shift(20)
    close_t = df['close']
    long_term_momentum = close_t - open_t_minus_20

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_weighted_avg = weighted_price_diff / total_volume

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'].pct_change()
    avg_daily_return = daily_returns.rolling(window=20).mean()

    # Adjust for Volatility
    std_dev_20_day = daily_returns.rolling(window=20).std()

    # Final Alpha Factor
    alpha_factor = (long_term_weighted_avg - avg_daily_return) / std_dev_20_day

    return alpha_factor
