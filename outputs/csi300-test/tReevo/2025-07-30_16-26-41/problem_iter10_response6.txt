```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) Weighted Price Difference
    long_term_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=10).sum()
    long_term_total_volume = df['volume'].rolling(window=10).sum()
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(10)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / long_term_total_volume
    
    # Subtract Long-Term Average Daily Return
    long_term_daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_average_daily_return = long_term_daily_returns.rolling(window=10).mean()
    
    # Adjust for Volatility
    long_term_volatility = long_term_daily_returns.rolling(window=10).std()
    
    # Final Alpha Factor (10-day)
    long_term_alpha_factor = (long_term_cumulative_weighted_avg - long_term_average_daily_return) / long_term_volatility
    
    # Recalculate using a 5-day window
    short_term_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    short_term_total_volume = df['volume'].rolling(window=5).sum()
    
    short_term_momentum = df['close'] - df['open'].shift(5)
    
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff / short_term_total_volume
    
    short_term_daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    short_term_average_daily_return = short_term_daily_returns.rolling(window=5).mean()
    
    short_term_volatility = short_term_daily_returns.rolling(window=5).std()
    
    short_term_alpha_factor = (short_term_cumulative_weighted_avg - short_term_average_daily_return) / short_term_volatility
    
    # Combine 10-day and 5-day metrics
    combined_alpha_factor = 0.6 * long_term_alpha_factor + 0.4 * short_term_alpha_factor
    
    # Integrate Trading Volume Effectively
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Final Alpha Factor
    final_alpha_factor = combined_alpha_factor * five_day_avg_volume
    
    return final_alpha_factor
```
