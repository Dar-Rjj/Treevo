```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term Weighted Price Difference (20-day)
    df['Weighted_Price_Diff_20'] = ((df['high'] - df['low']) * df['volume']).rolling(window=20).sum()
    total_volume_20 = df['volume'].rolling(window=20).sum()
    
    # Determine Dynamic Long-Term Open to Close Momentum (20-day)
    df['Open_to_Close_20'] = df['close'] - df['open'].shift(20)
    
    # Calculate the Dynamic Long-Term Cumulative Weighted Average (20-day)
    df['Cumulative_Weighted_Avg_20'] = df['Weighted_Price_Diff_20'] / total_volume_20
    
    # Subtract Dynamic Long-Term Average Daily Return (20-day)
    daily_returns_20 = (df['close'] - df['close'].shift(1)).rolling(window=20).sum() / 20
    
    # Adjust for Volatility (20-day)
    df['Daily_Returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility_20 = df['Daily_Returns'].rolling(window=20).std()
    
    # Final 20-Day Alpha Factor
    alpha_factor_20 = (df['Cumulative_Weighted_Avg_20'] - daily_returns_20) / volatility_20
    
    # Recalculate Metrics Using a 5-Day Window
    # Calculate Dynamic Short-Term Weighted Price Difference (5-day)
    df['Weighted_Price_Diff_5'] = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    
    # Determine Dynamic Short-Term Open to Close Momentum (5-day)
    df['Open_to_Close_5'] = df['close'] - df['open'].shift(5)
    
    # Calculate the Dynamic Short-Term Cumulative Weighted Average (5-day)
    df['Cumulative_Weighted_Avg_5'] = df['Weighted_Price_Diff_5'] / total_volume_5
    
    # Subtract Dynamic Short-Term Average Daily Return (5-day)
    daily_returns_5 = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # Adjust for Volatility (5-day)
    volatility_5 = df['Daily_Returns'].rolling(window=5).std()
    
    # Final 5-Day Alpha Factor
    alpha_factor_5 = (df['Cumulative_Weighted_Avg_5'] - daily_returns_5) / volatility_5
    
    # Combine 20-Day and 5-Day Metrics with Dynamic Weights
    combined_alpha_factor = 0.7 * alpha_factor_20 + 0.3 * alpha_factor_5
    
    # Integrate Trading Volume Effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    combined_alpha_factor *= avg_volume_5
    
    # Incorporate Recent Data with Exponential Smoothing
    exp_weight_20 = 0.8
    exp_weight_5 = 0.2
    
    # Apply Exponential Weights to 20-Day and 5-Day Metrics
    combined_alpha_factor = (exp_weight_20 * alpha_factor_20 + exp_weight_5 * alpha_factor_5) * avg_volume_5
    
    return combined_alpha_factor.dropna()
```
