```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day metrics
    df['HL_vol_10'] = (df['high'] - df['low']) * df['volume']
    total_HL_vol_10 = df['HL_vol_10'].rolling(window=10).sum()
    total_vol_10 = df['volume'].rolling(window=10).sum()
    dynamic_long_term_weighted_price_diff_10 = total_HL_vol_10 / total_vol_10
    
    long_term_open_close_momentum_10 = df['close'] - df['open'].shift(10)
    
    average_daily_return_10 = df['close'].diff().rolling(window=10).sum() / 10
    daily_returns_10 = df['close'].pct_change()
    std_dev_10 = daily_returns_10.rolling(window=10).std()
    
    alpha_factor_10 = (dynamic_long_term_weighted_price_diff_10 + long_term_open_close_momentum_10 - average_daily_return_10) / std_dev_10
    
    # Calculate 5-day metrics
    df['HL_vol_5'] = (df['high'] - df['low']) * df['volume']
    total_HL_vol_5 = df['HL_vol_5'].rolling(window=5).sum()
    total_vol_5 = df['volume'].rolling(window=5).sum()
    dynamic_short_term_weighted_price_diff_5 = total_HL_vol_5 / total_vol_5
    
    short_term_open_close_momentum_5 = df['close'] - df['open'].shift(5)
    
    average_daily_return_5 = df['close'].diff().rolling(window=5).sum() / 5
    daily_returns_5 = df['close'].pct_change()
    std_dev_5 = daily_returns_5.rolling(window=5).std()
    
    alpha_factor_5 = (dynamic_short_term_weighted_price_diff_5 + short_term_open_close_momentum_5 - average_daily_return_5) / std_dev_5
    
    # Combine 10-day and 5-day metrics
    combined_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5
    
    # Integrate trading volume effectively
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_vol_5
    
    return final_alpha_factor
```
