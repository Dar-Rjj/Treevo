```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['daily_hl_diff'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['daily_hl_diff'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    df['long_term_open_close_momentum'] = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=20).mean()
    
    # Adjust for Volatility
    daily_returns_pct = daily_returns / df['close'].shift(1)
    volatility = daily_returns_pct.rolling(window=20).std()
    
    # Final Alpha Factor
    alpha_factor = (long_term_cumulative_weighted_avg - long_term_average_daily_return) / volatility
    
    return alpha_factor
```
