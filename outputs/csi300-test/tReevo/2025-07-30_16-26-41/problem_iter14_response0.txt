import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day and 5-day metrics
    def calculate_metrics(df, window):
        df['PriceDiff'] = (df['high'] - df['low']) * df['volume']
        df[f'WeightedPriceDiff_{window}'] = df['PriceDiff'].rolling(window=window).sum() / df['volume'].rolling(window=window).sum()
        df[f'OpenToCloseMomentum_{window}'] = df['close'] - df['open'].shift(window)
        df[f'AvgDailyReturn_{window}'] = df['close'].diff().rolling(window=window).sum() / window
        df[f'StdDev_{window}'] = df['close'].pct_change().rolling(window=window).std()

        # Adjust for volatility
        df[f'AlphaFactor_{window}'] = (df[f'WeightedPriceDiff_{window}'] - df[f'AvgDailyReturn_{window}']) / df[f'StdDev_{window}']

        # Apply exponential weights
        exp_weights = 0.8 if window == 10 else 0.2
        df[f'ExponentialAlphaFactor_{window}'] = df[f'AlphaFactor_{window}'] * (exp_weights ** np.arange(len(df)))

        return df

    # Calculate 10-day and 5-day metrics
    df = calculate_metrics(df, 10)
    df = calculate_metrics(df, 5)

    # Combine 10-day and 5-day metrics
    df['CombinedAlphaFactor'] = 0.6 * df['ExponentialAlphaFactor_10'] + 0.4 * df['ExponentialAlphaFactor_5']

    # Integrate trading volume effectively
    df['5DayAvgVolume'] = df['volume'].rolling(window=5).mean()
    df['CombinedAlphaFactor'] *= df['5DayAvgVolume']

    # Incorporate additional market features
    df['VPT_10'] = (df['volume'] * (df['close'] - df['close'].shift(1))).rolling(window=10).sum()
    df['VPT_5'] = (df['volume'] * (df['close'] - df['close'].shift(1))).rolling(window=5).sum()
    df['CombinedAlphaFactor'] += df['VPT_10'] + df['VPT_5']

    return df['CombinedAlphaFactor']
