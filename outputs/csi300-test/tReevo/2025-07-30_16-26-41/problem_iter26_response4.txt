```python
import pandas as pd

def heuristics_v2(df):
    # Long-Term Alpha Factor (10-day)
    long_term_weighted_price_diff = (df['high'] - df['low']).rolling(window=10).sum() * df['volume'].rolling(window=10).sum()
    long_term_total_volume = df['volume'].rolling(window=10).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / long_term_total_volume
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(10)
    long_term_daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_avg_daily_return = long_term_daily_returns.rolling(window=10).mean()
    long_term_volatility = long_term_daily_returns.rolling(window=10).std()
    long_term_alpha_factor = (long_term_cumulative_weighted_avg - long_term_avg_daily_return) / long_term_volatility

    # Short-Term Alpha Factor (3-day)
    short_term_weighted_price_diff = (df['high'] - df['low']).rolling(window=3).sum() * df['volume'].rolling(window=3).sum()
    short_term_total_volume = df['volume'].rolling(window=3).sum()
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff / short_term_total_volume
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(3)
    short_term_daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    short_term_avg_daily_return = short_term_daily_returns.rolling(window=3).mean()
    short_term_volatility = short_term_daily_returns.rolling(window=3).std()
    short_term_alpha_factor = (short_term_cumulative_weighted_avg - short_term_avg_daily_return) / short_term_volatility

    # Combine 10-Day and 3-Day Metrics
    combined_alpha_factor = 0.7 * long_term_alpha_factor + 0.3 * short_term_alpha_factor

    # Integrate Dynamic Trading Volume
    three_day_average_volume = df['volume'].rolling(window=3).mean()
    combined_alpha_factor_adjusted = combined_alpha_factor * three_day_average_volume

    # Adjust for Recent Market Sentiment
    two_day_avg_return = (df['close'] - df['open']).rolling(window=2).mean() / 2
    final_alpha_factor = combined_alpha_factor_adjusted + two_day_avg_return

    return final_alpha_factor
```
