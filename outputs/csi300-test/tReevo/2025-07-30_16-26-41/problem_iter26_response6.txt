import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    df['long_term_weighted_price_diff'] = df['high_low_vol'].rolling(window=10).sum()
    df['long_term_total_volume'] = df['volume'].rolling(window=10).sum()
    df['long_term_cumulative_weighted_avg'] = df['long_term_weighted_price_diff'] / df['long_term_total_volume']
    
    # Determine Long-Term Open to Close Momentum (10-day)
    df['long_term_open_to_close_momentum'] = df['close'] - df['open'].shift(10)
    
    # Subtract Long-Term Average Daily Return (10-day)
    df['long_term_daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['long_term_avg_daily_return'] = df['long_term_daily_return'].rolling(window=10).mean()
    
    # Adjust for Adaptive Volatility (10-day)
    df['long_term_daily_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['long_term_ewm_std'] = df['long_term_daily_returns'].ewm(span=10, adjust=False).std()
    
    # Final 10-Day Alpha Factor
    df['long_term_alpha_factor'] = (df['long_term_cumulative_weighted_avg'] - df['long_term_avg_daily_return']) / df['long_term_ewm_std']
    
    # Recalculate Metrics Using a 5-Day Window
    df['short_term_weighted_price_diff'] = df['high_low_vol'].rolling(window=5).sum()
    df['short_term_total_volume'] = df['volume'].rolling(window=5).sum()
    df['short_term_cumulative_weighted_avg'] = df['short_term_weighted_price_diff'] / df['short_term_total_volume']
    
    # Determine Short-Term Open to Close Momentum (5-day)
    df['short_term_open_to_close_momentum'] = df['close'] - df['open'].shift(5)
    
    # Subtract Short-Term Average Daily Return (5-day)
    df['short_term_daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['short_term_avg_daily_return'] = df['short_term_daily_return'].rolling(window=5).mean()
    
    # Adjust for Adaptive Volatility (5-day)
    df['short_term_daily_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['short_term_ewm_std'] = df['short_term_daily_returns'].ewm(span=5, adjust=False).std()
    
    # Final 5-Day Alpha Factor
    df['short_term_alpha_factor'] = (df['short_term_cumulative_weighted_avg'] - df['short_term_avg_daily_return']) / df['short_term_ewm_std']
    
    # Combine 10-Day and 5-Day Metrics with Dynamic Weights
    df['relative_volatility_ratio'] = df['short_term_ewm_std'] / df['long_term_ewm_std']
    df['long_term_metric_weight'] = 1 / (1 + df['relative_volatility_ratio'])
    df['short_term_metric_weight'] = df['relative_volatility_ratio'] / (1 + df['relative_volatility_ratio'])
    df['combined_alpha_factor'] = df['long_term_metric_weight'] * df['long_term_alpha_factor'] + df['short_term_metric_weight'] * df['short_term_alpha_factor']
    
    # Integrate Trading Volume Effectively
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['combined_alpha_factor'] *= df['5_day_avg_volume']
    
    # Incorporate Liquidity
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['final_alpha_factor'] = df['combined_alpha_factor'] * df['10_day_avg_volume']
    
    return df['final_alpha_factor']
