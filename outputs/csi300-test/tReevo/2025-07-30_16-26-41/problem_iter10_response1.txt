```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['HLV'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['HLV'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_daily_return = daily_returns.rolling(window=20).sum() / 20
    
    # Adjust for Volatility
    daily_returns_pct = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns_pct.rolling(window=20).std()
    
    # Incorporate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    short_term_daily_return = daily_returns.rolling(window=5).sum() / 5
    
    # Incorporate Trading Volume Trends
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    volume_trend_indicator = (avg_volume_5 - avg_volume_20) / avg_volume_20
    
    # Incorporate Market Sentiment
    high_low_ratio = (df['high'] - df['low']) / (df['high'].shift(20) - df['low'].shift(20))
    
    # Final Alpha Factor
    alpha_factor = (
        long_term_cumulative_weighted_avg
        - long_term_daily_return
        + (short_term_momentum - short_term_daily_return)
        * volume_trend_indicator
        * high_low_ratio
    ) / long_term_volatility
    
    return alpha_factor.dropna()
```
