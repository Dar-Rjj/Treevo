import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['open'] + df['close']) / 2
    
    # Determine Intraday Direction
    intraday_direction = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Compute Intraday Momentum Reversal Score
    intraday_momentum_reversal = intraday_range * intraday_direction
    
    # Weight by Volume
    weighted_momentum_reversal = intraday_momentum_reversal * df['volume']
    
    # Calculate Rolling Average Volume
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    
    # Incorporate Volume-Based Strengthening
    volume_factor = (df['volume'] > rolling_avg_volume).astype(float)
    volume_factor[volume_factor == 0] = 1.0
    volume_factor[volume_factor == 1] = 1.5
    final_score = weighted_momentum_reversal * volume_factor
    
    return final_score
