import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) Weighted Price Difference
    long_term_weighted_price_diff = (df['high'] - df['low']) * df['volume']
    long_term_total_volume = df['volume'].rolling(window=10).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff.rolling(window=10).sum() / long_term_total_volume
    
    # Determine Long-Term Open to Close Momentum (10-day)
    long_term_open_close_momentum = df['close'] - df['open'].shift(10)
    
    # Subtract Long-Term Average Daily Return (10-day)
    long_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    long_term_adjusted = long_term_cumulative_weighted_avg - long_term_avg_daily_return
    
    # Adjust for Adaptive Volatility (10-day)
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.ewm(span=10, adjust=False).std()
    long_term_alpha_factor = long_term_adjusted / long_term_volatility
    
    # Recalculate Metrics Using a 5-Day Window
    short_term_weighted_price_diff = (df['high'] - df['low']) * df['volume']
    short_term_total_volume = df['volume'].rolling(window=5).sum()
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff.rolling(window=5).sum() / short_term_total_volume
    
    # Determine Short-Term Open to Close Momentum (5-day)
    short_term_open_close_momentum = df['close'] - df['open'].shift(5)
    
    # Subtract Short-Term Average Daily Return (5-day)
    short_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    short_term_adjusted = short_term_cumulative_weighted_avg - short_term_avg_daily_return
    
    # Adjust for Adaptive Volatility (5-day)
    short_term_volatility = daily_returns.ewm(span=5, adjust=False).std()
    short_term_alpha_factor = short_term_adjusted / short_term_volatility
    
    # Combine 10-Day and 5-Day Metrics with Dynamic Weights
    relative_volatility_ratio = short_term_volatility / long_term_volatility
    long_term_weight = 1 / (1 + relative_volatility_ratio)
    short_term_weight = relative_volatility_ratio / (1 + relative_volatility_ratio)
    
    combined_alpha_factor = long_term_weight * long_term_alpha_factor + short_term_weight * short_term_alpha_factor
    
    # Integrate Trading Volume Effectively
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * five_day_avg_volume
    
    return final_alpha_factor
