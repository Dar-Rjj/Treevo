```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['open'] + df['close']) / 2
    
    # Determine Intraday Direction
    df['intraday_direction'] = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Compute Intraday Momentum Reversal Score
    df['intraday_momentum_reversal_score'] = df['intraday_range'] * df['intraday_direction']
    
    # Weight by Volume
    df['volume_weighted_score'] = df['intraday_momentum_reversal_score'] * df['volume']
    
    # Incorporate Volume-Based Strengthening
    df['rolling_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_score'] = df.apply(lambda row: row['volume_weighted_score'] * 1.5 if row['volume'] > row['rolling_avg_volume'] else row['volume_weighted_score'], axis=1)
    
    # Incorporate Historical Trend
    df['20_day_ma_close'] = df['close'].rolling(window=20).mean()
    df['historical_trend_multiplier'] = df.apply(lambda row: 1.2 if row['close'] > row['20_day_ma_close'] else 0.8, axis=1)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['adjusted_score'] * df['historical_trend_multiplier']
    
    return df['alpha_factor']
```
