```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day window
    df['HLV'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['HLV'].rolling(window=10).sum()
    long_term_total_volume = df['volume'].rolling(window=10).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / long_term_total_volume
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(10)
    long_term_daily_return = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = long_term_daily_return.rolling(window=10).mean()
    long_term_daily_returns = df['close'].pct_change()
    long_term_volatility = long_term_daily_returns.rolling(window=10).std()
    
    # Final 10-day alpha factor
    long_term_alpha_factor = (long_term_cumulative_weighted_avg - long_term_average_daily_return) / long_term_volatility
    
    # 5-day window
    short_term_weighted_price_diff = df['HLV'].rolling(window=5).sum()
    short_term_total_volume = df['volume'].rolling(window=5).sum()
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff / short_term_total_volume
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(5)
    short_term_daily_return = df['close'] - df['close'].shift(1)
    short_term_average_daily_return = short_term_daily_return.rolling(window=5).mean()
    short_term_daily_returns = df['close'].pct_change()
    short_term_volatility = short_term_daily_returns.rolling(window=5).std()

    # Final 5-day alpha factor
    short_term_alpha_factor = (short_term_cumulative_weighted_avg - short_term_average_daily_return) / short_term_volatility
    
    # Combine 10-day and 5-day metrics
    combined_alpha_factor = 0.6 * long_term_alpha_factor + 0.4 * short_term_alpha_factor
    
    # 5-day average volume
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    
    # Adjust the combined alpha factor
    adjusted_alpha_factor = combined_alpha_factor * five_day_avg_volume
    
    # Incorporate recent data with exponential weights
    long_term_exp_weight = 0.8
    short_term_exp_weight = 0.2
    exp_weighted_alpha_factor = (long_term_exp_weight * long_term_alpha_factor + 
                                 short_term_exp_weight * short_term_alpha_factor) * five_day_avg_volume
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * long_term_alpha_factor + 0.4 * short_term_alpha_factor
    final_alpha_factor = final_alpha_factor * five_day_avg_volume
    final_alpha_factor = (long_term_exp_weight * final_alpha_factor) + (short_term_exp_weight * exp_weighted_alpha_factor)
    
    return final_alpha_factor
```
