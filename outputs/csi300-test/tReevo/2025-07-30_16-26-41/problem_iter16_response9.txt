```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-Day Metrics
    df['High_Low_Volume'] = (df['high'] - df['low']) * df['volume']
    df['10D_Weighted_Price_Diff'] = df['High_Low_Volume'].rolling(window=10).sum()
    df['10D_Total_Volume'] = df['volume'].rolling(window=10).sum()
    df['10D_Cumulative_Weighted_Avg'] = df['10D_Weighted_Price_Diff'] / df['10D_Total_Volume']

    df['10D_Open_to_Close_Momentum'] = df['close'] - df['open'].shift(10)
    df['10D_Avg_Daily_Return'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    df['10D_Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['10D_Volatility'] = df['10D_Daily_Return'].rolling(window=10).std()

    df['10D_Factor'] = (df['10D_Cumulative_Weighted_Avg'] - df['10D_Avg_Daily_Return']) / df['10D_Volatility']

    # 5-Day Metrics
    df['5D_Weighted_Price_Diff'] = df['High_Low_Volume'].rolling(window=5).sum()
    df['5D_Total_Volume'] = df['volume'].rolling(window=5).sum()
    df['5D_Cumulative_Weighted_Avg'] = df['5D_Weighted_Price_Diff'] / df['5D_Total_Volume']

    df['5D_Open_to_Close_Momentum'] = df['close'] - df['open'].shift(5)
    df['5D_Avg_Daily_Return'] = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    df['5D_Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5D_Volatility'] = df['5D_Daily_Return'].rolling(window=5).std()

    df['5D_Factor'] = (df['5D_Cumulative_Weighted_Avg'] - df['5D_Avg_Daily_Return']) / df['5D_Volatility']

    # Combine 10-Day and 5-Day Metrics
    df['Combined_Factor'] = 0.6 * df['10D_Factor'] + 0.4 * df['5D_Factor']

    # Adjust by 5-Day Average Volume
    df['5D_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['Final_Alpha_Factor'] = df['Combined_Factor'] * df['5D_Avg_Volume']

    return df['Final_Alpha_Factor']
```
