import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['PriceDiff_10'] = (df['high'] - df['low']) * df['volume']
    df['WeightedPriceDiff_10'] = df['PriceDiff_10'].rolling(window=10).sum()
    df['TotalVolume_10'] = df['volume'].rolling(window=10).sum()
    df['CumulativeWeightedAverage_10'] = df['WeightedPriceDiff_10'] / df['TotalVolume_10']
    df['OpenCloseMomentum_10'] = df['close'] - df['open'].shift(10)
    df['DailyReturn_10'] = df['close'] - df['close'].shift(1)
    df['AverageDailyReturn_10'] = df['DailyReturn_10'].rolling(window=10).mean()
    df['Volatility_10'] = df['DailyReturn_10'].pct_change().rolling(window=10).std()
    df['AlphaFactor_10'] = (df['CumulativeWeightedAverage_10'] - df['AverageDailyReturn_10']) / df['Volatility_10']

    # 5-day calculations
    df['PriceDiff_5'] = (df['high'] - df['low']) * df['volume']
    df['WeightedPriceDiff_5'] = df['PriceDiff_5'].rolling(window=5).sum()
    df['TotalVolume_5'] = df['volume'].rolling(window=5).sum()
    df['CumulativeWeightedAverage_5'] = df['WeightedPriceDiff_5'] / df['TotalVolume_5']
    df['OpenCloseMomentum_5'] = df['close'] - df['open'].shift(5)
    df['DailyReturn_5'] = df['close'] - df['close'].shift(1)
    df['AverageDailyReturn_5'] = df['DailyReturn_5'].rolling(window=5).mean()
    df['Volatility_5'] = df['DailyReturn_5'].pct_change().rolling(window=5).std()
    df['AlphaFactor_5'] = (df['CumulativeWeightedAverage_5'] - df['AverageDailyReturn_5']) / df['Volatility_5']

    # Combine 10-day and 5-day metrics
    df['CombinedAlphaFactor'] = 0.4 * df['AlphaFactor_10'] + 0.6 * df['AlphaFactor_5']

    # Integrate trading volume
    df['AverageVolume_5'] = df['volume'].rolling(window=5).mean()
    df['FinalAlphaFactor'] = df['CombinedAlphaFactor'] * df['AverageVolume_5']

    return df['FinalAlphaFactor'].dropna()
