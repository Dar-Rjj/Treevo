import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Volume-Weighted Price (VWP)
    df['VWP'] = (df['volume'] * df['close']) / df['volume'].rolling(window=10).sum()
    
    # Compute Short-Term Momentum
    df['Short_Term_Momentum'] = (df['VWP'] - df['VWP'].shift(10)) / df['VWP'].shift(10)
    
    # Compute Long-Term Momentum
    df['Long_Term_Momentum'] = (df['VWP'] - df['VWP'].shift(20)) / df['VWP'].shift(20)
    
    # Adaptive Volatility
    df['Volatility'] = df['VWP'].rolling(window=30).std()
    
    # Combine Momentum and Volatility
    df['Alpha_Factor'] = (df['Short_Term_Momentum'] + df['Long_Term_Momentum']) / df['Volatility']
    
    # Liquidity Adjustment
    df['Liquidity'] = (df['close'] - df['open']) / (df['high'] - df['low'])
    
    # Adjust Alpha Factor
    df['Alpha_Factor_Adjusted'] = df['Alpha_Factor'] * df['Liquidity']
    
    # Final Alpha Factor
    final_alpha_factor = df['Alpha_Factor_Adjusted']
    
    return final_alpha_factor
