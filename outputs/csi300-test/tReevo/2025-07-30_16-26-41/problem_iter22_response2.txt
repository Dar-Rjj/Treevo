```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate short-term and long-term EMA of close prices
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Assume sentiment and news impact are additional columns in the DataFrame
    df['sentiment_EMA_5'] = df['sentiment'].ewm(span=5, adjust=False).mean()
    df['news_impact_EMA_5'] = df['news_impact'].ewm(span=5, adjust=False).mean()
    df['sentiment_EMA_20'] = df['sentiment'].ewm(span=20, adjust=False).mean()
    df['news_impact_EMA_20'] = df['news_impact'].ewm(span=20, adjust=False).mean()

    # Volume-Weighted Close Price
    total_volume = df['volume'].rolling(window=10).sum()
    df['VW_close'] = (df['close'] * df['volume']).rolling(window=10).sum() / total_volume

    # Volume-Weighted High-Low Difference
    df['VW_HL_diff'] = ((df['high'] - df['low']) * df['volume']).rolling(window=10).sum()

    # Sector-Specific EMA
    sector_ema_5 = df.groupby('sector')['close'].transform(lambda x: x.ewm(span=5, adjust=False).mean())
    sector_ema_20 = df.groupby('sector')['close'].transform(lambda x: x.ewm(span=20, adjust=False).mean())

    # Sector Momentum
    df['sector_momentum'] = df.groupby('sector')['close'].transform(lambda x: x.pct_change(10)).fillna(0)

    # Time-Series Momentum
    df['momentum'] = df['close'].pct_change(10).fillna(0)

    # 5-Day Average Volume
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()

    # 10-Day and 5-Day High-Low Range
    df['HL_range_10'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    df['HL_range_5'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()

    # Combine 10-Day and 5-Day Alpha Factors
    alpha_factor_10 = 0.6 * (df['EMA_5'] - df['EMA_20']) + 0.4 * (df['VW_close'] - df['close'])
    alpha_factor_5 = 0.6 * (df['sentiment_EMA_5'] - df['sentiment_EMA_20']) + 0.4 * (df['news_impact_EMA_5'] - df['news_impact_EMA_20'])

    # Adjust by 5-Day Average Volume
    alpha_factor_10 = alpha_factor_10 * df['avg_volume_5']
    alpha_factor_5 = alpha_factor_5 * df['avg_volume_5']

    # Adjust by 10-Day and 5-Day High-Low Range
    alpha_factor_10 = alpha_factor_10 / df['HL_range_10']
    alpha_factor_5 = alpha_factor_5 / df['HL_range_5']

    # Final Alpha Factor
    final_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5

    return final_alpha_factor
```
