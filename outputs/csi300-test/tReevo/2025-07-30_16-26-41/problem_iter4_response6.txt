```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    total_weighted_price_diff = weighted_price_diff.rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_avg = total_weighted_price_diff / total_volume
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).mean()
    
    # Adjust for Volatility
    daily_returns_volatility = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility_20_day = daily_returns_volatility.rolling(window=20).std()
    
    # Final Alpha Factor
    alpha_factor = (long_term_cumulative_avg + long_term_momentum - long_term_avg_daily_return) / volatility_20_day
    
    return alpha_factor
```
