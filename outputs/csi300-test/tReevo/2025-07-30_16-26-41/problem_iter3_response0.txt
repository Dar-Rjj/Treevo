```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_diff_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['high_low_diff_vol'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume

    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(20)

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'].pct_change()
    long_term_average_daily_return = daily_returns.rolling(window=20).mean()

    # Adjust for Volatility
    daily_returns_shifted = df['close'].pct_change().shift(-1)
    volatility = daily_returns_shifted.rolling(window=20).std()

    # Final Alpha Factor
    alpha_factor = (long_term_cumulative_weighted_avg + long_term_open_to_close_momentum) - long_term_average_daily_return
    alpha_factor = alpha_factor / volatility

    return alpha_factor
```
