import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference (20-day)
    df['hlv_20'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['hlv_20'].rolling(window=20).sum()
    total_volume_20 = df['volume'].rolling(window=22).sum()  # use 22 days to avoid index shift issue

    # Determine Long-Term Open to Close Momentum (20-day)
    long_term_momentum = df['close'] - df['open'].shift(20)

    # Calculate the Long-Term Cumulative Weighted Average (20-day)
    long_term_cumulative_avg = long_term_weighted_price_diff / total_volume_20

    # Subtract Long-Term Average Daily Return (20-day)
    long_term_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=20).sum() / 20

    # Adjust for Volatility (20-day)
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.rolling(window=20).std()

    # Final 20-Day Alpha Factor
    long_term_alpha = (long_term_cumulative_avg - long_term_daily_return) / long_term_volatility

    # Recalculate Metrics Using a 5-Day Window
    df['hlv_5'] = (df['high'] - df['low']) * df['volume']
    short_term_weighted_price_diff = df['hlv_5'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()

    # Determine Short-Term Open to Close Momentum (5-day)
    short_term_momentum = df['close'] - df['open'].shift(5)

    # Calculate the Short-Term Cumulative Weighted Average (5-day)
    short_term_cumulative_avg = short_term_weighted_price_diff / total_volume_5

    # Subtract Short-Term Average Daily Return (5-day)
    short_term_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5

    # Adjust for Volatility (5-day)
    short_term_volatility = daily_returns.rolling(window=5).std()

    # Final 5-Day Alpha Factor
    short_term_alpha = (short_term_cumulative_avg - short_term_daily_return) / short_term_volatility

    # Combine 20-Day and 5-Day Metrics
    combined_alpha = 0.6 * long_term_alpha + 0.4 * short_term_alpha

    # Integrate Trading Volume Effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    combined_alpha *= avg_volume_5

    # Incorporate Relative Strength
    relative_strength = df['close'] / df['close'].shift(20)
    final_alpha = combined_alpha * relative_strength

    return final_alpha
