```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Closing Position within Intraday Range
    closing_position = (df['close'] - df['low']) / intraday_range
    
    # Calculate Opening Position within Intraday Range
    opening_position = (df['open'] - df['low']) / intraday_range
    
    # Determine Reversal Potential
    reversal_potential = (closing_position > opening_position).astype(int) * 1 - \
                         (closing_position <= opening_position).astype(int) * 1
    reversal_potential *= df['volume']
    
    # Integrate Volume-Weighted Intraday Momentum
    close_to_close_return = df['close'].pct_change()
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_momentum = close_to_close_return * rolling_avg_volume
    
    # Combine with Reversal Potential
    fraction = 0.5 + 0.3 * (df['amount'] > 10000).astype(int)
    alpha_factor = volume_weighted_momentum + reversal_potential - fraction * reversal_potential
    
    return alpha_factor
```
