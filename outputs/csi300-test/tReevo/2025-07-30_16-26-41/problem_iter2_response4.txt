```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Volatility
    daily_volatility = df['high'] - df['low']
    
    # Calculate Weighted Daily Volatility
    weighted_daily_volatility = daily_volatility * df['volume']
    
    # Calculate Smoothed Weighted Volatility
    smoothed_weighted_volatility = (weighted_daily_volatility.rolling(window=20).sum() / 
                                    df['volume'].rolling(window=20).sum())
    
    # Determine Long-Term Price Trend
    long_term_trend = df['close'] - df['close'].shift(20)
    
    # Adjust for Price Direction
    price_direction = (df['close'] > df['close'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Momentum from Open to Close
    momentum_open_to_close = df['close'] - df['open'].shift(1)
    
    # Combine Momentum and Trend
    combined_momentum_trend = (momentum_open_to_close * price_direction) + long_term_trend
    
    # Final Alpha Factor
    final_alpha_factor = combined_momentum_trend * smoothed_weighted_volatility
    
    return final_alpha_factor
```
