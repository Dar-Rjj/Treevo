```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_high_low_vol_sum = df['high_low_vol'].rolling(window=20).sum()
    long_term_volume_sum = df['volume'].rolling(window=20).sum()

    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close = df['close'] - df['open'].shift(20)

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_weighted_avg = long_term_high_low_vol_sum / long_term_volume_sum

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_daily_return = daily_returns.rolling(window=20).sum() / 20

    # Adjust for Volatility
    daily_log_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_std_dev = daily_log_returns.rolling(window=20).std()

    # Final Alpha Factor
    alpha_factor = (long_term_weighted_avg + long_term_open_to_close - long_term_daily_return) / long_term_std_dev

    return alpha_factor
```
