```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Helper function to calculate the weighted price difference
    def weighted_price_diff(df, window):
        high_low_diff = df['high'] - df['low']
        volume = df['volume']
        return (high_low_diff * volume).rolling(window=window).sum() / volume.rolling(window=window).sum()
    
    # Calculate Long-Term Weighted Price Difference (10-day)
    long_term_weighted_diff = weighted_price_diff(df, 10)
    
    # Determine Long-Term Open to Close Momentum (10-day)
    long_term_momentum = df['close'] - df['open'].shift(10)
    
    # Calculate the Long-Term Cumulative Weighted Average (10-day)
    long_term_cumulative_avg = long_term_weighted_diff
    
    # Subtract Long-Term Average Daily Return (10-day)
    long_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    
    # Adjust for Volatility (10-day)
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.rolling(window=10).std()
    
    # Final 10-Day Alpha Factor
    alpha_10_day = (long_term_cumulative_avg - long_term_avg_daily_return) / long_term_volatility
    
    # Recalculate Metrics Using a 5-Day Window
    short_term_weighted_diff = weighted_price_diff(df, 5)
    short_term_momentum = df['close'] - df['open'].shift(5)
    short_term_cumulative_avg = short_term_weighted_diff
    short_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    short_term_volatility = daily_returns.rolling(window=5).std()
    
    # Final 5-Day Alpha Factor
    alpha_5_day = (short_term_cumulative_avg - short_term_avg_daily_return) / short_term_volatility
    
    # Combine 10-Day and 5-Day Metrics
    combined_alpha_factor = 0.6 * alpha_10_day + 0.4 * alpha_5_day
    
    # Integrate Trading Volume Effectively
    avg_volume_5_day = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5_day
    
    return final_alpha_factor
```
