```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['high_low_vol_10'] = (df['high'] - df['low']) * df['volume']
    total_high_low_vol_10 = df['high_low_vol_10'].rolling(window=10).sum()
    total_vol_10 = df['volume'].rolling(window=10).sum()
    
    long_term_weighted_price_diff_10 = total_high_low_vol_10 / total_vol_10
    long_term_open_to_close_momentum_10 = df['close'] - df['open'].shift(10)
    
    long_term_cumulative_weighted_avg_10 = long_term_weighted_price_diff_10
    
    long_term_avg_daily_return_10 = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    long_term_volatility_10 = df['close'].pct_change().rolling(window=10).std()
    
    final_10_day_alpha_factor = (long_term_cumulative_weighted_avg_10 - long_term_avg_daily_return_10) / long_term_volatility_10
    
    # 3-day calculations
    df['high_low_vol_3'] = (df['high'] - df['low']) * df['volume']
    total_high_low_vol_3 = df['high_low_vol_3'].rolling(window=3).sum()
    total_vol_3 = df['volume'].rolling(window=3).sum()
    
    short_term_weighted_price_diff_3 = total_high_low_vol_3 / total_vol_3
    short_term_open_to_close_momentum_3 = df['close'] - df['open'].shift(3)
    
    short_term_cumulative_weighted_avg_3 = short_term_weighted_price_diff_3
    
    short_term_avg_daily_return_3 = (df['close'] - df['close'].shift(1)).rolling(window=3).sum() / 3
    short_term_volatility_3 = df['close'].pct_change().rolling(window=3).std()
    
    final_3_day_alpha_factor = (short_term_cumulative_weighted_avg_3 - short_term_avg_daily_return_3) / short_term_volatility_3
    
    # Combine 10-day and 3-day metrics
    combined_alpha_factor = 0.7 * final_10_day_alpha_factor + 0.3 * final_3_day_alpha_factor
    
    # Integrate dynamic trading volume
    avg_vol_3 = df['volume'].rolling(window=3).mean()
    adjusted_combined_alpha_factor = combined_alpha_factor * avg_vol_3
    
    return adjusted_combined_alpha_factor
```
