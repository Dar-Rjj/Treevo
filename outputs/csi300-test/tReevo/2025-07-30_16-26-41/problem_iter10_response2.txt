import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic Short-Term Weighted Price Difference
    df['hl_vol'] = (df['high'] - df['low']) * df['volume']
    sum_hl_vol_5 = df['hl_vol'].rolling(window=5).sum()
    sum_volume_5 = df['volume'].rolling(window=5).sum()
    dynamic_weighted_price_diff = sum_hl_vol_5 / sum_volume_5
    
    # Determine Dynamic Short-Term Open to Close Momentum
    open_5_days_ago = df['open'].shift(5)
    close_today = df['close']
    dynamic_open_to_close_momentum = close_today - open_5_days_ago
    
    # Calculate the Dynamic Short-Term Cumulative Weighted Average
    dynamic_cumulative_weighted_avg = dynamic_weighted_price_diff

    # Subtract Dynamic Short-Term Average Daily Return
    daily_returns = df['close'].diff()
    dynamic_avg_daily_return = daily_returns.rolling(window=5).mean()
    adjusted_cumulative_weighted_avg = dynamic_cumulative_weighted_avg - dynamic_avg_daily_return
    
    # Adjust for Volatility
    daily_log_returns = df['close'].pct_change()
    std_dev_5_days = daily_log_returns.rolling(window=5).std()
    alpha_factor = adjusted_cumulative_weighted_avg / std_dev_5_days
    
    # Integrate Trading Volume Effectively
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    final_alpha_factor = alpha_factor * avg_volume_5_days
    
    return final_alpha_factor
