import pandas as pd

def heuristics_v2(df):
    # Long-term calculations (30 days)
    long_term_days = 30
    df['High_Low_Volume_30'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['High_Low_Volume_30'].rolling(window=long_term_days).sum()
    total_volume_30 = df['volume'].rolling(window=long_term_days).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume_30
    
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(long_term_days)
    
    long_term_daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = long_term_daily_returns.rolling(window=long_term_days).sum() / long_term_days
    
    long_term_volatility = long_term_daily_returns.rolling(window=long_term_days).std()
    
    long_term_alpha_factor = (long_term_cumulative_weighted_avg - long_term_average_daily_return) / long_term_volatility
    
    # Short-term calculations (5 days)
    short_term_days = 5
    df['High_Low_Volume_5'] = (df['high'] - df['low']) * df['volume']
    short_term_weighted_price_diff = df['High_Low_Volume_5'].rolling(window=short_term_days).sum()
    total_volume_5 = df['volume'].rolling(window=short_term_days).sum()
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff / total_volume_5
    
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(short_term_days)
    
    short_term_daily_returns = df['close'] - df['close'].shift(1)
    short_term_average_daily_return = short_term_daily_returns.rolling(window=short_term_days).sum() / short_term_days
    
    short_term_volatility = short_term_daily_returns.rolling(window=short_term_days).std()
    
    short_term_alpha_factor = (short_term_cumulative_weighted_avg - short_term_average_daily_return) / short_term_volatility
    
    # Combine long-term and short-term alpha factors
    combined_alpha_factor = 0.7 * long_term_alpha_factor + 0.3 * short_term_alpha_factor
    
    # Integrate trading volume effectively
    volume_weighted_price_diff_5 = (df['high'] - df['low']) * df['volume']
    volume_weighted_price_diff_5 = volume_weighted_price_diff_5.rolling(window=short_term_days).sum()
    average_volume_5 = df['volume'].rolling(window=short_term_days).mean()
    
    final_combined_alpha_factor = combined_alpha_factor * average_volume_5
    
    return final_combined_alpha_factor
