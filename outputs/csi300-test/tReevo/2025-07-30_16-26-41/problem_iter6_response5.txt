import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['price_diff_vol'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff = df['price_diff_vol'].rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    long_term_weighted_price_diff = weighted_price_diff / total_volume
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(10)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff
    
    # Subtract Long-Term Average Daily Return
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=10).mean()
    
    adjusted_long_term_cumulative_weighted_avg = long_term_cumulative_weighted_avg - long_term_average_daily_return
    
    # Adjust for Volatility
    std_dev_10_day = daily_returns.rolling(window=10).std()
    final_alpha_factor = adjusted_long_term_cumulative_weighted_avg / std_dev_10_day
    
    return final_alpha_factor
