```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-Day Weighted Price Difference
    df['price_diff_vol_10'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff_10 = df['price_diff_vol_10'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    cumulative_weighted_avg_10 = weighted_price_diff_10 / total_volume_10
    
    # Determine 10-Day Open to Close Momentum
    momentum_10 = df['close'] - df['open'].shift(10)
    
    # Subtract 10-Day Average Daily Return
    avg_daily_return_10 = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    
    # Adjust for 10-Day Volatility
    daily_returns_10 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev_10 = daily_returns_10.rolling(window=10).std()
    
    # Final 10-Day Alpha Factor
    alpha_factor_10 = (cumulative_weighted_avg_10 + momentum_10 - avg_daily_return_10) / std_dev_10
    
    # Recalculate Metrics Using a 5-Day Window
    df['price_diff_vol_5'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff_5 = df['price_diff_vol_5'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    cumulative_weighted_avg_5 = weighted_price_diff_5 / total_volume_5
    
    # Determine 5-Day Open to Close Momentum
    momentum_5 = df['close'] - df['open'].shift(5)
    
    # Subtract 5-Day Average Daily Return
    avg_daily_return_5 = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # Adjust for 5-Day Volatility
    daily_returns_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev_5 = daily_returns_5.rolling(window=5).std()
    
    # Final 5-Day Alpha Factor
    alpha_factor_5 = (cumulative_weighted_avg_5 + momentum_5 - avg_daily_return_5) / std_dev_5
    
    # Combine 10-Day and 5-Day Metrics
    combined_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5
    
    # Integrate Trading Volume Effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5
    
    return final_alpha_factor
```
