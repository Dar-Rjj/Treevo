```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-Day Metrics
    df['Weighted_Price_Diff_10'] = ((df['high'] - df['low']) * df['volume']).rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    df['Cumulative_Weighted_Avg_10'] = df['Weighted_Price_Diff_10'] / total_volume_10
    df['Open_Close_Momentum_10'] = df['close'] - df['open'].shift(10)
    daily_returns_10 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Average_Daily_Return_10'] = daily_returns_10.rolling(window=10).mean()
    df['Volatility_10'] = daily_returns_10.rolling(window=10).std()
    df['Alpha_Factor_10'] = (df['Cumulative_Weighted_Avg_10'] + df['Open_Close_Momentum_10'] - df['Average_Daily_Return_10']) / df['Volatility_10']
    
    # 5-Day Metrics
    df['Weighted_Price_Diff_5'] = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    df['Cumulative_Weighted_Avg_5'] = df['Weighted_Price_Diff_5'] / total_volume_5
    df['Open_Close_Momentum_5'] = df['close'] - df['open'].shift(5)
    daily_returns_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Average_Daily_Return_5'] = daily_returns_5.rolling(window=5).mean()
    df['Volatility_5'] = daily_returns_5.rolling(window=5).std()
    df['Alpha_Factor_5'] = (df['Cumulative_Weighted_Avg_5'] + df['Open_Close_Momentum_5'] - df['Average_Daily_Return_5']) / df['Volatility_5']
    
    # Combine 10-Day and 5-Day Metrics
    df['Combined_Alpha_Factor'] = 0.6 * df['Alpha_Factor_10'] + 0.4 * df['Alpha_Factor_5']
    
    # Adjust for 5-Day Average Volume
    df['Average_Volume_5'] = df['volume'].rolling(window=5).mean()
    df['Final_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['Average_Volume_5']
    
    return df['Final_Alpha_Factor'].dropna()
```
