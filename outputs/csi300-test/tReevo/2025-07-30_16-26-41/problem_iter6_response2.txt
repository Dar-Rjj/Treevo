import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Multi-Period Volume-Weighted Price Difference
    df['price_diff_vol_weighted'] = (df['high'] - df['low']) * df['volume']
    total_price_diff_vol_weighted = df['price_diff_vol_weighted'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    multi_period_vol_weighted_price_diff = total_price_diff_vol_weighted / total_volume

    # Determine Multi-Period Open to Close Momentum
    multi_period_open_to_close_momentum = df['close'] - df['open'].shift(20)

    # Calculate the Multi-Period Cumulative Weighted Average
    multi_period_momentum_factor = multi_period_vol_weighted_price_diff

    # Subtract Multi-Period Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    avg_daily_return = daily_returns.rolling(window=20).mean()
    adjusted_multi_period_momentum = multi_period_momentum_factor - avg_daily_return

    # Adjust for Multi-Period Volatility
    daily_returns_pct = daily_returns / df['close'].shift(1)
    multi_period_volatility = daily_returns_pct.rolling(window=20).std()
    adjusted_multi_period_momentum /= multi_period_volatility

    # Incorporate Intraday Reversal Signal
    intraday_range = df['high'] - df['low']
    intraday_midpoint = (df['open'] + df['close']) / 2
    intraday_direction = np.where(df['open'] > df['close'], -1, 1)
    intraday_momentum_reversal_score = intraday_range * intraday_direction
    intraday_volatility = df['high'] - df['low']
    weighted_intraday_score = (intraday_momentum_reversal_score * df['volume']) / intraday_volatility

    # Combine Multi-Period and Intraday Signals
    final_alpha_factor = adjusted_multi_period_momentum + weighted_intraday_score

    return final_alpha_factor
