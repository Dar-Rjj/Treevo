import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-Day Weighted Price Difference
    df['price_diff_vol_10'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff_10 = df['price_diff_vol_10'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    weighted_avg_10 = weighted_price_diff_10 / total_volume_10
    
    # 10-Day Open to Close Momentum
    open_to_close_mom_10 = df['close'] - df['open'].shift(10)
    
    # 10-Day Average Daily Return
    daily_return_10 = (df['close'] - df['close'].shift(1)).fillna(0).rolling(window=10).sum() / 10
    
    # 10-Day Dynamic Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    ewm_vol_10 = daily_returns.ewm(span=10, adjust=False).std()
    
    # 10-Day Alpha Factor
    alpha_factor_10 = (weighted_avg_10 - daily_return_10) / ewm_vol_10
    
    # 5-Day Weighted Price Difference
    weighted_price_diff_5 = (df['high'] - df['low']) * df['volume']
    total_volume_5 = df['volume'].rolling(window=5).sum()
    weighted_avg_5 = weighted_price_diff_5.rolling(window=5).sum() / total_volume_5
    
    # 5-Day Open to Close Momentum
    open_to_close_mom_5 = df['close'] - df['open'].shift(5)
    
    # 5-Day Average Daily Return
    daily_return_5 = (df['close'] - df['close'].shift(1)).fillna(0).rolling(window=5).sum() / 5
    
    # 5-Day Dynamic Volatility
    ewm_vol_5 = daily_returns.ewm(span=5, adjust=False).std()
    
    # 5-Day Alpha Factor
    alpha_factor_5 = (weighted_avg_5 - daily_return_5) / ewm_vol_5
    
    # Combine 10-Day and 5-Day Metrics
    combined_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5
    
    # Integrate Trading Volume Effectively
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    avg_vol_10 = df['volume'].rolling(window=10).mean()
    normalized_vol_5 = avg_vol_5 / avg_vol_10
    final_alpha_factor = combined_alpha_factor * normalized_vol_5
    
    return final_alpha_factor
