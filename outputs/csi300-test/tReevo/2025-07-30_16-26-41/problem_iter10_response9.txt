import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    long_term_weighted_price_diff = (df['high'] - df['low']) * df['volume']
    long_term_total_volume = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff = long_term_weighted_price_diff.rolling(window=20).sum() / long_term_total_volume
    
    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff
    
    # Subtract Long-Term Average Daily Return
    long_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=20).sum() / 20
    long_term_cumulative_weighted_avg -= long_term_avg_daily_return
    
    # Adjust for Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.rolling(window=20).std()
    long_term_alpha_factor = long_term_cumulative_weighted_avg / long_term_volatility
    
    # Calculate Short-Term Weighted Price Difference
    short_term_weighted_price_diff = (df['high'] - df['low']) * df['volume']
    short_term_total_volume = df['volume'].rolling(window=5).sum()
    short_term_weighted_price_diff = short_term_weighted_price_diff.rolling(window=5).sum() / short_term_total_volume
    
    # Determine Short-Term Open to Close Momentum
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(5)
    
    # Calculate the Short-Term Cumulative Weighted Average
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff
    
    # Subtract Short-Term Average Daily Return
    short_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    short_term_cumulative_weighted_avg -= short_term_avg_daily_return
    
    # Adjust for Volatility
    short_term_volatility = daily_returns.rolling(window=5).std()
    short_term_alpha_factor = short_term_cumulative_weighted_avg / short_term_volatility
    
    # Combine Long-Term and Short-Term Alpha Factors
    combined_alpha_factor = 0.7 * long_term_alpha_factor + 0.3 * short_term_alpha_factor
    
    # Integrate Trading Volume Effectively
    volume_weighted_price_diff = (df['high'] - df['low']) * df['volume']
    five_day_average_volume = df['volume'].rolling(window=5).mean()
    adjusted_combined_alpha_factor = combined_alpha_factor * five_day_average_volume
    
    return adjusted_combined_alpha_factor
