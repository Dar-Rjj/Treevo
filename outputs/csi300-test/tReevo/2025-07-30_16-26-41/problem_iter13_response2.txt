import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term (10-day) Metrics
    df['price_diff_vol_10'] = (df['high'] - df['low']) * df['volume']
    df['sum_price_diff_vol_10'] = df['price_diff_vol_10'].rolling(window=10).sum()
    df['sum_vol_10'] = df['volume'].rolling(window=10).sum()
    df['open_t-10'] = df['open'].shift(10)
    df['close_t'] = df['close']
    df['long_term_momentum_10'] = df['close_t'] - df['open_t-10']
    df['long_term_cumulative_weighted_avg_10'] = df['sum_price_diff_vol_10'] / df['sum_vol_10']
    df['daily_returns'] = df['close'].pct_change()
    df['avg_daily_return_10'] = df['daily_returns'].rolling(window=10).mean()
    df['std_daily_return_10'] = df['daily_returns'].rolling(window=10).std()
    
    # Calculate 10-Day Alpha Factor
    df['alpha_factor_10'] = (df['long_term_cumulative_weighted_avg_10'] - df['avg_daily_return_10']) / df['std_daily_return_10']

    # Calculate Dynamic Short-Term (5-day) Metrics
    df['price_diff_vol_5'] = (df['high'] - df['low']) * df['volume']
    df['sum_price_diff_vol_5'] = df['price_diff_vol_5'].rolling(window=5).sum()
    df['sum_vol_5'] = df['volume'].rolling(window=5).sum()
    df['open_t-5'] = df['open'].shift(5)
    df['short_term_momentum_5'] = df['close_t'] - df['open_t-5']
    df['short_term_cumulative_weighted_avg_5'] = df['sum_price_diff_vol_5'] / df['sum_vol_5']
    df['avg_daily_return_5'] = df['daily_returns'].rolling(window=5).mean()
    df['std_daily_return_5'] = df['daily_returns'].rolling(window=5).std()
    
    # Calculate 5-Day Alpha Factor
    df['alpha_factor_5'] = (df['short_term_cumulative_weighted_avg_5'] - df['avg_daily_return_5']) / df['std_daily_return_5']

    # Combine 10-Day and 5-Day Metrics
    df['combined_alpha_factor'] = 0.6 * df['alpha_factor_10'] + 0.4 * df['alpha_factor_5']

    # Integrate Trading Volume Effectively
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['combined_alpha_factor'] = df['combined_alpha_factor'] * df['avg_volume_5']

    # Incorporate Recent Data with Exponential Weights
    df['exp_weighted_10'] = 0.8 * df['alpha_factor_10'].ewm(span=10, adjust=False).mean()
    df['exp_weighted_5'] = 0.2 * df['alpha_factor_5'].ewm(span=5, adjust=False).mean()
    df['final_alpha_factor'] = df['exp_weighted_10'] + df['exp_weighted_5']
    
    return df['final_alpha_factor'].dropna()
