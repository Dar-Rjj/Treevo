```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the 14-day EMA of money flow
    ema_14_money_flow = money_flow.ewm(span=14).mean()
    
    # Calculate the 28-day EMA of money flow
    ema_28_money_flow = money_flow.ewm(span=28).mean()
    
    # Generate the alpha factor
    alpha_factor = (ema_14_money_flow - ema_28_money_flow) / ema_28_money_flow
    
    return alpha_factor
```
