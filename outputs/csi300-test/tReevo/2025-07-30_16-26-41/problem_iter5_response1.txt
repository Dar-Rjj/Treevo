```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['hl_vol'] = (df['high'] - df['low']) * df['volume']
    total_hl_vol_20 = df['hl_vol'].rolling(window=20).sum()
    total_volume_20 = df['volume'].rolling(window=20).sum()
    weighted_price_diff = total_hl_vol_20 / total_volume_20
    
    # Determine Long-Term Open to Close Momentum
    open_t_minus_20 = df['open'].shift(20)
    close_t = df['close']
    long_term_momentum = close_t - open_t_minus_20
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_momentum * weighted_price_diff
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).mean()
    adjusted_long_term_momentum = long_term_cumulative_weighted_avg - long_term_avg_daily_return
    
    # Adjust for Volatility
    daily_returns_pct = daily_returns / df['close'].shift(1)
    volatility_20 = daily_returns_pct.rolling(window=20).std()
    final_alpha_factor = adjusted_long_term_momentum / volatility_20
    
    return final_alpha_factor
```
