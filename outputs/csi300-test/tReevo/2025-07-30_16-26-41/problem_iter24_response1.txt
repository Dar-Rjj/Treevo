```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 20-day calculations
    df['hl_vol_20'] = (df['high'] - df['low']) * df['volume']
    hl_vol_sum_20 = df['hl_vol_20'].rolling(window=20).sum()
    vol_sum_20 = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff_20 = hl_vol_sum_20 / vol_sum_20
    long_term_open_to_close_momentum_20 = df['close'] - df['open'].shift(20)
    long_term_cumulative_weighted_avg_20 = long_term_weighted_price_diff_20 / vol_sum_20
    avg_daily_return_20 = (df['close'] - df['close'].shift(1)).rolling(window=20).sum() / 20
    daily_returns = df['close'].pct_change()
    std_dev_20 = daily_returns.rolling(window=20).std()
    alpha_factor_20 = (long_term_cumulative_weighted_avg_20 + long_term_open_to_close_momentum_20 - avg_daily_return_20) / std_dev_20

    # 5-day calculations
    df['hl_vol_5'] = (df['high'] - df['low']) * df['volume']
    hl_vol_sum_5 = df['hl_vol_5'].rolling(window=5).sum()
    vol_sum_5 = df['volume'].rolling(window=5).sum()
    short_term_weighted_price_diff_5 = hl_vol_sum_5 / vol_sum_5
    short_term_open_to_close_momentum_5 = df['close'] - df['open'].shift(5)
    short_term_cumulative_weighted_avg_5 = short_term_weighted_price_diff_5 / vol_sum_5
    avg_daily_return_5 = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    std_dev_5 = daily_returns.rolling(window=5).std()
    alpha_factor_5 = (short_term_cumulative_weighted_avg_5 + short_term_open_to_close_momentum_5 - avg_daily_return_5) / std_dev_5

    # Combine 20-day and 5-day metrics
    combined_alpha_factor = 0.6 * alpha_factor_20 + 0.4 * alpha_factor_5

    # Integrate dynamic trading volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5

    return final_alpha_factor
```
