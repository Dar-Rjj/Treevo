```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['High_Low_Volume_10'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_10'] = df['volume'].rolling(window=10).sum()
    df['Weighted_Price_Diff_10'] = df['High_Low_Volume_10'].rolling(window=10).sum() / df['Total_Volume_10']
    
    df['Open_to_Close_Momentum_10'] = df['close'] - df['open'].shift(10)
    df['Daily_Return_10'] = df['close'] - df['close'].shift(1)
    df['Average_Daily_Return_10'] = df['Daily_Return_10'].rolling(window=10).mean()
    
    df['10_Day_Std_Dev'] = df['Daily_Return_10'].rolling(window=10).std()
    
    # Final 10-Day Alpha Factor
    df['10_Day_Alpha_Factor'] = (df['Weighted_Price_Diff_10'] - df['Average_Daily_Return_10']) / df['10_Day_Std_Dev']
    
    # 5-day calculations
    df['High_Low_Volume_5'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_5'] = df['volume'].rolling(window=5).sum()
    df['Weighted_Price_Diff_5'] = df['High_Low_Volume_5'].rolling(window=5).sum() / df['Total_Volume_5']
    
    df['Open_to_Close_Momentum_5'] = df['close'] - df['open'].shift(5)
    df['Daily_Return_5'] = df['close'] - df['close'].shift(1)
    df['Average_Daily_Return_5'] = df['Daily_Return_5'].rolling(window=5).mean()
    
    df['5_Day_Std_Dev'] = df['Daily_Return_5'].rolling(window=5).std()
    
    # Final 5-Day Alpha Factor
    df['5_Day_Alpha_Factor'] = (df['Weighted_Price_Diff_5'] - df['Average_Daily_Return_5']) / df['5_Day_Std_Dev']
    
    # Combine 10-Day and 5-Day Metrics
    df['Combined_Alpha_Factor'] = 0.6 * df['10_Day_Alpha_Factor'] + 0.4 * df['5_Day_Alpha_Factor']
    
    # Integrate Trading Volume Effectively
    df['5_Day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['Final_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['5_Day_Avg_Volume']
    
    return df['Final_Alpha_Factor'].dropna()
```
