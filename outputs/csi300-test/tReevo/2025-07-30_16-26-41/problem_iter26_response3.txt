```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term (10-day) Metrics
    long_term_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=10).sum()
    total_volume_10_day = df['volume'].rolling(window=10).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume_10_day
    
    long_term_open_to_close_momentum = (df['close'] - df['open'].shift(10)).fillna(0)
    
    long_term_average_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    daily_returns_10_day = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns_10_day.rolling(window=10).std()
    
    long_term_alpha_factor = (long_term_cumulative_weighted_avg + long_term_open_to_close_momentum - long_term_average_daily_return) / long_term_volatility

    # Calculate Short-Term (3-day) Metrics
    short_term_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=3).sum()
    total_volume_3_day = df['volume'].rolling(window=3).sum()
    short_term_cumulative_weighted_avg = short_term_weighted_price_diff / total_volume_3_day
    
    short_term_open_to_close_momentum = (df['close'] - df['open'].shift(3)).fillna(0)
    
    short_term_average_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=3).sum() / 3
    daily_returns_3_day = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    short_term_volatility = daily_returns_3_day.rolling(window=3).std()
    
    short_term_alpha_factor = (short_term_cumulative_weighted_avg + short_term_open_to_close_momentum - short_term_average_daily_return) / short_term_volatility

    # Combine 10-Day and 3-Day Metrics
    combined_alpha_factor = 0.7 * long_term_alpha_factor + 0.3 * short_term_alpha_factor

    # Integrate Dynamic Trading Volume Effectively
    average_volume_3_day = df['volume'].rolling(window=3).mean()
    adjusted_combined_alpha_factor = combined_alpha_factor * average_volume_3_day

    # Adjust for Recent Market Sentiment
    recent_market_sentiment = (df['close'] - df['open']).rolling(window=2).sum() / 2
    final_alpha_factor = adjusted_combined_alpha_factor + recent_market_sentiment

    return final_alpha_factor
```
