import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Long-term (10-day) calculations
    df['high_low_vol_10'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff_10 = df['high_low_vol_10'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    long_term_cumulative_weighted_avg_10 = long_term_weighted_price_diff_10 / total_volume_10
    
    long_term_open_to_close_momentum_10 = df['close'] - df['open'].shift(10)
    
    daily_returns_10 = df['close'] - df['close'].shift(1)
    long_term_average_daily_return_10 = daily_returns_10.rolling(window=10).mean()
    
    volatility_10 = daily_returns_10.rolling(window=10).std()
    
    alpha_factor_10 = (long_term_cumulative_weighted_avg_10 - long_term_average_daily_return_10) / volatility_10
    
    # Short-term (5-day) calculations
    df['high_low_vol_5'] = (df['high'] - df['low']) * df['volume']
    short_term_weighted_price_diff_5 = df['high_low_vol_5'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    short_term_cumulative_weighted_avg_5 = short_term_weighted_price_diff_5 / total_volume_5
    
    short_term_open_to_close_momentum_5 = df['close'] - df['open'].shift(5)
    
    daily_returns_5 = df['close'] - df['close'].shift(1)
    short_term_average_daily_return_5 = daily_returns_5.rolling(window=5).mean()
    
    volatility_5 = daily_returns_5.rolling(window=5).std()
    
    alpha_factor_5 = (short_term_cumulative_weighted_avg_5 - short_term_average_daily_return_5) / volatility_5
    
    # Combine 10-day and 5-day metrics
    combined_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5
    
    # Integrate trading volume effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5
    
    return final_alpha_factor.dropna()
