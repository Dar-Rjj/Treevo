```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    high_low_diff = (df['high'] - df['low']) * df['volume']
    total_high_low_vol = high_low_diff.rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    long_term_weighted_price_diff = total_high_low_vol / total_volume

    # Determine Long-Term Open to Close Momentum
    open_t_minus_10 = df['open'].shift(10)
    close_t = df['close']
    long_term_open_close_momentum = close_t - open_t_minus_10

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=10).mean()

    # Adjust for Volatility
    daily_returns_for_vol = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns_for_vol.rolling(window=10).std()

    # Final Alpha Factor
    alpha_factor = (long_term_cumulative_weighted_avg + long_term_open_close_momentum - long_term_avg_daily_return) / volatility
    
    return alpha_factor
```
