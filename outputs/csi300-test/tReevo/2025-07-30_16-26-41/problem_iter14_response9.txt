import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    long_term_weighted_price_diff = ((df['high'] - df['low']) * df['volume']).rolling(window=20).sum()
    total_volume_20_days = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume_20_days
    
    # Subtract Long-Term Average Daily Return
    long_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=20).sum() / 20
    
    # Adjust for Long-Term Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.rolling(window=20).std()
    
    # Incorporate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Subtract Short-Term Average Daily Return
    short_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # Adjust for Short-Term Volatility
    short_term_volatility = daily_returns.rolling(window=5).std()
    
    # Incorporate Trading Volume Trends
    average_volume_20_days = df['volume'].rolling(window=20).mean()
    average_volume_5_days = df['volume'].rolling(window=5).mean()
    volume_trend_indicator = (average_volume_5_days - average_volume_20_days) / average_volume_20_days
    
    # Incorporate Market Sentiment
    high_low_ratio = (df['high'] - df['low']) / (df['high'].shift(20) - df['low'].shift(20))
    
    # Final Alpha Factor
    alpha_factor = (
        long_term_cumulative_weighted_avg
        - long_term_avg_daily_return
    ) / long_term_volatility + (
        short_term_momentum
        - short_term_avg_daily_return
    ) / short_term_volatility * volume_trend_indicator * high_low_ratio
    
    return alpha_factor.dropna()
