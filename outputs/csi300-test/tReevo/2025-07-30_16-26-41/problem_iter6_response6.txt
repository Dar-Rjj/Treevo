import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    intraday_midpoint = (df['open'] + df['close']) / 2
    
    # Determine Intraday Direction
    intraday_direction = (df['close'] > df['open']).astype(int) * 2 - 1  # 1 for bullish, -1 for bearish
    
    # Compute Intraday Momentum Reversal Score
    intraday_momentum_reversal_score = intraday_range * intraday_direction
    
    # Weight by Volume
    weighted_score = intraday_momentum_reversal_score * df['volume']
    
    # Incorporate Volume-Based Strengthening
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    strengthened_score = weighted_score.where(df['volume'] <= rolling_avg_volume, weighted_score * 1.5)
    
    # Incorporate Historical Trend
    close_20_day_ma = df['close'].rolling(window=20).mean().shift(1)  # Avoid lookahead bias
    trend_adjustment = (df['close'] > close_20_day_ma).astype(int) * 2 - 1  # 1 for bullish, -1 for bearish
    trend_adjusted_score = strengthened_score + trend_adjustment * 0.5
    
    # Final Alpha Factor
    final_alpha_factor = trend_adjusted_score
    
    return final_alpha_factor
