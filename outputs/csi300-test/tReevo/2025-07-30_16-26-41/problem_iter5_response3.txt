```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['weighted_price_diff'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['weighted_price_diff'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume

    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).sum() / 20

    # Adjust for Volatility
    daily_returns_pct = df['close'].pct_change()
    volatility = daily_returns_pct.rolling(window=20).std()

    # Final Alpha Factor
    final_alpha_factor = (long_term_cumulative_weighted_avg + long_term_momentum - long_term_avg_daily_return) / volatility

    return final_alpha_factor.dropna()
```
