import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Volume-Weighted Price Difference
    df['price_diff_vol'] = (df['high'] - df['low']) * df['volume']
    total_price_diff_vol = df['price_diff_vol'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=22).sum()

    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_weighted_avg = total_price_diff_vol / total_volume

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).sum() / 20
    adjusted_momentum = long_term_momentum - long_term_avg_daily_return

    # Adjust for Volatility
    daily_returns_pct = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns_pct.rolling(window=20).std()
    final_alpha_factor = adjusted_momentum / volatility

    return final_alpha_factor.dropna()
