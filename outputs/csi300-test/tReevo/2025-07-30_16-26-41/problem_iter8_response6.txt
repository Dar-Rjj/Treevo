```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = weighted_price_diff.rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    
    # Determine Long-Term Open to Close Momentum
    open_to_close_momentum = df['close'] - df['open'].shift(10)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_avg = long_term_weighted_price_diff / total_volume
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=10).sum() / 10
    
    # Adjust for Volatility
    daily_returns_pct = df['close'].pct_change()
    volatility = daily_returns_pct.rolling(window=10).std()
    
    # Incorporate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    short_term_average_daily_return = daily_returns.rolling(window=5).sum() / 5
    
    # Incorporate Trading Volume Trends
    avg_volume_10_day = df['volume'].rolling(window=10).mean()
    avg_volume_5_day = df['volume'].rolling(window=5).mean()
    volume_trend_indicator = (avg_volume_5_day - avg_volume_10_day) / avg_volume_10_day
    
    # Incorporate Market Sentiment
    high_low_ratio = (df['high'] - df['low']) / (df['high'].shift(10) - df['low'].shift(10))
    
    # Final Alpha Factor
    alpha_factor = (
        (long_term_cumulative_avg - long_term_average_daily_return) / volatility
        + short_term_momentum - short_term_average_daily_return
        * volume_trend_indicator * high_low_ratio
    )
    
    return alpha_factor
```
