import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 30-day calculations
    df['long_term_weighted_diff'] = (df['high'] - df['low']) * df['volume']
    long_term_sum_vol = df['volume'].rolling(window=30).sum()
    long_term_weighted_price_diff = df['long_term_weighted_diff'].rolling(window=30).sum()
    long_term_cumulative_avg = long_term_weighted_price_diff / long_term_sum_vol
    
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(30)
    
    long_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=30).sum() / 30
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.rolling(window=30).std()
    
    long_term_alpha_factor = (long_term_cumulative_avg + long_term_open_to_close_momentum - long_term_avg_daily_return) / long_term_volatility
    
    # 5-day calculations
    df['short_term_weighted_diff'] = (df['high'] - df['low']) * df['volume']
    short_term_sum_vol = df['volume'].rolling(window=5).sum()
    short_term_weighted_price_diff = df['short_term_weighted_diff'].rolling(window=5).sum()
    short_term_cumulative_avg = short_term_weighted_price_diff / short_term_sum_vol
    
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(5)
    
    short_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    short_term_volatility = daily_returns.rolling(window=5).std()
    
    short_term_alpha_factor = (short_term_cumulative_avg + short_term_open_to_close_momentum - short_term_avg_daily_return) / short_term_volatility
    
    # Combine 30-day and 5-day metrics
    combined_alpha_factor = (0.7 * long_term_alpha_factor) + (0.3 * short_term_alpha_factor)
    
    # Adjust by 5-day average volume
    five_day_avg_volume = df['volume'].rolling(window=5).mean()
    adjusted_combined_alpha_factor = combined_alpha_factor * five_day_avg_volume
    
    # Incorporate relative strength
    thirty_day_relative_strength = df['close'] / df['close'].shift(30)
    final_alpha_factor = adjusted_combined_alpha_factor * thirty_day_relative_strength
    
    return final_alpha_factor
