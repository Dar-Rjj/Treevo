import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate 10-day metrics
    df['High_Low_Volume_10'] = (df['high'] - df['low']) * df['volume']
    total_volume_10 = df['volume'].rolling(window=10).sum()
    weighted_price_diff_10 = df['High_Low_Volume_10'].rolling(window=10).sum() / total_volume_10
    open_to_close_momentum_10 = df['close'] - df['open'].shift(10)
    average_daily_return_10 = df['close'].diff().rolling(window=10).sum() / 10
    daily_returns = df['close'].pct_change()
    std_dev_10 = daily_returns.rolling(window=10).std()

    # Final 10-day alpha factor
    alpha_factor_10 = (weighted_price_diff_10 + open_to_close_momentum_10 - average_daily_return_10) / std_dev_10

    # Calculate 5-day metrics
    df['High_Low_Volume_5'] = (df['high'] - df['low']) * df['volume']
    total_volume_5 = df['volume'].rolling(window=5).sum()
    weighted_price_diff_5 = df['High_Low_Volume_5'].rolling(window=5).sum() / total_volume_5
    open_to_close_momentum_5 = df['close'] - df['open'].shift(5)
    average_daily_return_5 = df['close'].diff().rolling(window=5).sum() / 5
    std_dev_5 = daily_returns.rolling(window=5).std()

    # Final 5-day alpha factor
    alpha_factor_5 = (weighted_price_diff_5 + open_to_close_momentum_5 - average_daily_return_5) / std_dev_5

    # Combine 10-day and 5-day metrics with adaptive weights
    adaptive_weights_10 = 0.8
    adaptive_weights_5 = 0.2
    combined_alpha_factor = (adaptive_weights_10 * alpha_factor_10 + adaptive_weights_5 * alpha_factor_5)

    # Integrate trading volume effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    combined_alpha_factor *= avg_volume_5

    # Incorporate intraday volatility
    intraday_volatility_5 = (df['high'] - df['low']).rolling(window=5).mean()
    combined_alpha_factor /= intraday_volatility_5

    # Apply exponential smoothing
    combined_alpha_factor = (0.8 * alpha_factor_10 + 0.2 * alpha_factor_5) * avg_volume_5 / intraday_volatility_5

    return combined_alpha_factor
