```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term (10-day) Metrics
    df['High_Low_Volume_10'] = (df['high'] - df['low']) * df['volume']
    df['Dynamic_Long_Term_Price_Diff_10'] = df['High_Low_Volume_10'].rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    df['Dynamic_Long_Term_Momentum_10'] = df['close'] - df['open'].shift(10)
    df['Dynamic_Long_Term_Cumulative_Avg_10'] = df['Dynamic_Long_Term_Price_Diff_10']
    df['Dynamic_Long_Term_Daily_Return_10'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    df['Dynamic_Long_Term_Adjusted_Cumulative_Avg_10'] = df['Dynamic_Long_Term_Cumulative_Avg_10'] - df['Dynamic_Long_Term_Daily_Return_10']
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['10_Day_Std_Dev'] = df['Daily_Return'].rolling(window=10).std()
    df['Final_10_Day_Factor'] = df['Dynamic_Long_Term_Adjusted_Cumulative_Avg_10'] / df['10_Day_Std_Dev']

    # Calculate Dynamic Short-Term (5-day) Metrics
    df['High_Low_Volume_5'] = (df['high'] - df['low']) * df['volume']
    df['Dynamic_Short_Term_Price_Diff_5'] = df['High_Low_Volume_5'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    df['Dynamic_Short_Term_Momentum_5'] = df['close'] - df['open'].shift(5)
    df['Dynamic_Short_Term_Cumulative_Avg_5'] = df['Dynamic_Short_Term_Price_Diff_5']
    df['Dynamic_Short_Term_Daily_Return_5'] = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    df['Dynamic_Short_Term_Adjusted_Cumulative_Avg_5'] = df['Dynamic_Short_Term_Cumulative_Avg_5'] - df['Dynamic_Short_Term_Daily_Return_5']
    df['5_Day_Std_Dev'] = df['Daily_Return'].rolling(window=5).std()
    df['Final_5_Day_Factor'] = df['Dynamic_Short_Term_Adjusted_Cumulative_Avg_5'] / df['5_Day_Std_Dev']

    # Combine 10-Day and 5-Day Metrics
    df['Combined_Alpha_Factor'] = 0.6 * df['Final_10_Day_Factor'] + 0.4 * df['Final_5_Day_Factor']

    # Integrate Trading Volume Effectively
    df['5_Day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['Adjusted_Combined_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['5_Day_Avg_Volume']

    # Incorporate Recent Data with Exponential Weights
    df['Exponential_Weighted_10_Day'] = df['Final_10_Day_Factor'].ewm(span=10, adjust=False).mean()
    df['Exponential_Weighted_5_Day'] = df['Final_5_Day_Factor'].ewm(span=5, adjust=False).mean()
    df['Final_Combined_Alpha_Factor'] = 0.8 * df['Exponential_Weighted_10_Day'] + 0.2 * df['Exponential_Weighted_5_Day']
    df['Final_Alpha_Factor'] = df['Final_Combined_Alpha_Factor'] * df['5_Day_Avg_Volume']

    return df['Final_Alpha_Factor'].dropna()
```
