```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) Metrics
    df['hlv_10'] = (df['high'] - df['low']) * df['volume']
    long_term_hlv_sum = df['hlv_10'].rolling(window=10).sum()
    long_term_volume_sum = df['volume'].rolling(window=10).sum()
    
    long_term_weighted_price_diff = long_term_hlv_sum / long_term_volume_sum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(10)
    long_term_average_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    long_term_volatility = df['close'].pct_change().rolling(window=10).std()
    
    final_10_day_alpha_factor = (long_term_weighted_price_diff + long_term_open_to_close_momentum - long_term_average_daily_return) / long_term_volatility
    
    # Calculate Short-Term (3-day) Metrics
    short_term_hlv_sum = (df['high'] - df['low']) * df['volume'].rolling(window=3).sum()
    short_term_volume_sum = df['volume'].rolling(window=3).sum()
    
    short_term_weighted_price_diff = short_term_hlv_sum / short_term_volume_sum
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(3)
    short_term_average_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=3).sum() / 3
    short_term_volatility = df['close'].pct_change().rolling(window=3).std()
    
    final_3_day_alpha_factor = (short_term_weighted_price_diff + short_term_open_to_close_momentum - short_term_average_daily_return) / short_term_volatility
    
    # Combine 10-Day and 3-Day Metrics
    combined_alpha_factor = 0.7 * final_10_day_alpha_factor + 0.3 * final_3_day_alpha_factor
    
    # Integrate Dynamic Trading Volume Effectively
    three_day_average_volume = df['volume'].rolling(window=3).mean()
    combined_alpha_factor *= three_day_average_volume
    
    # Adjust for Recent Market Sentiment
    two_day_average_return = (df['close'] - df['open']).rolling(window=2).sum() / 2
    final_alpha_factor = combined_alpha_factor + two_day_average_return
    
    return final_alpha_factor
```
