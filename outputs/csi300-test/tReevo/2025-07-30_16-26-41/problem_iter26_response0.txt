import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) metrics
    df['high_low_vol_10'] = (df['high'] - df['low']) * df['volume']
    long_term_high_low_vol_sum = df['high_low_vol_10'].rolling(window=10).sum()
    long_term_volume_sum = df['volume'].rolling(window=10).sum()
    long_term_weighted_price_diff = long_term_high_low_vol_sum / long_term_volume_sum
    
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(10)
    
    long_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    long_term_volatility = daily_returns.ewm(span=10, adjust=False).std()
    
    long_term_alpha_factor = (long_term_weighted_price_diff - long_term_avg_daily_return) / long_term_volatility
    
    # Calculate Short-Term (5-day) metrics
    df['high_low_vol_5'] = (df['high'] - df['low']) * df['volume']
    short_term_high_low_vol_sum = df['high_low_vol_5'].rolling(window=5).sum()
    short_term_volume_sum = df['volume'].rolling(window=5).sum()
    short_term_weighted_price_diff = short_term_high_low_vol_sum / short_term_volume_sum
    
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(5)
    
    short_term_avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    short_term_volatility = daily_returns.ewm(span=5, adjust=False).std()
    
    short_term_alpha_factor = (short_term_weighted_price_diff - short_term_avg_daily_return) / short_term_volatility
    
    # Combine 10-Day and 5-Day Metrics with Dynamic Weights
    relative_volatility_ratio = short_term_volatility / long_term_volatility
    ten_day_weight = 1 / (1 + relative_volatility_ratio)
    five_day_weight = relative_volatility_ratio / (1 + relative_volatility_ratio)
    
    combined_alpha_factor = (ten_day_weight * long_term_alpha_factor) + (five_day_weight * short_term_alpha_factor)
    
    # Integrate Trading Volume Effectively
    ten_day_avg_volume = df['volume'].rolling(window=10).mean()
    
    # Incorporate Liquidity
    final_alpha_factor = combined_alpha_factor * ten_day_avg_volume
    
    return final_alpha_factor
