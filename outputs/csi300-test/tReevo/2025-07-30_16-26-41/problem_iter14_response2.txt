```python
import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index (date) and columns (open, high, low, close, amount, volume).
    
    Returns:
    pd.Series: A Series with index (date) representing the factor values.
    """
    # Calculate the average of open and close prices
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the percentage change in average price
    avg_price_pct_change = avg_price.pct_change().fillna(0)
    
    # Calculate the ratio of amount to volume, which could be seen as a price level for transactions
    price_level = df['amount'] / df['volume']
    
    # Calculate the z-score of the price level
    price_level_zscore = (price_level - price_level.mean()) / price_level.std()
    
    # Combine the two factors
    factor = (avg_price_pct_change + price_level_zscore) / 2
    
    return factor
```
