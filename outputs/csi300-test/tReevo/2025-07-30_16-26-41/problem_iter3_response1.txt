```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['daily_price_diff_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['daily_price_diff_vol'].rolling(window=20).sum()

    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(20)

    # Calculate the Long-Term Cumulative Weighted Average
    total_volume_20_days = df['volume'].rolling(window=20).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume_20_days

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=20).sum() / 20

    momentum_factor = long_term_cumulative_weighted_avg + long_term_open_to_close_momentum - long_term_average_daily_return

    # Adjust for Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_returns.rolling(window=20).std()

    # Final Alpha Factor
    alpha_factor = momentum_factor / volatility

    return alpha_factor
```
