```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    high_low_diff = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = high_low_diff.rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=20).sum() / 20
    
    # Adjust for Volatility
    daily_log_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_log_returns.rolling(window=20).std()
    
    # Incorporate Volume-Weighted Intraday Momentum
    intraday_momentum = df['close'] - df['close'].shift(1)
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    volume_weighted_intraday_momentum = intraday_momentum * rolling_avg_volume
    
    # Combine Long-Term and Intraday Momentum
    fraction = np.where(df['volume'] < df['volume'].quantile(0.5), 0.5, 0.8)
    combined_momentum = (long_term_cumulative_weighted_avg + volume_weighted_intraday_momentum 
                        - fraction * long_term_average_daily_return)
    
    # Final Alpha Factor
    alpha_factor = (combined_momentum / volatility).fillna(0)
    
    return alpha_factor
```
