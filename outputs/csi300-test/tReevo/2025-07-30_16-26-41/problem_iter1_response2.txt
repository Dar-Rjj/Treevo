```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Range
    intraday_price_range = df['high'] - df['low']
    
    # Calculate Adjusted Price Range
    adjusted_price_range = intraday_price_range * df['volume']
    
    # Consider the Sign of Close-Open Difference
    sign = (df['close'] > df['open']).astype(int) - (df['close'] < df['open']).astype(int)
    adjusted_price_range *= sign
    
    # Assign Neutral Adjusted Price Range for equal Close and Open
    adjusted_price_range[df['close'] == df['open']] = 0
    
    return adjusted_price_range
```
