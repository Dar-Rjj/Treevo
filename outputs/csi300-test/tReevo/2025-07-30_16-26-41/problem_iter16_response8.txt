```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) Metrics
    df['High_Low_Volume'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['High_Low_Volume'].rolling(window=10).sum()
    total_volume_10_days = df['volume'].rolling(window=10).sum()
    long_term_weighted_avg = long_term_weighted_price_diff / total_volume_10_days
    
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(10)
    
    long_term_daily_return = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = long_term_daily_return.rolling(window=10).sum() / 10
    
    daily_returns = df['close'].pct_change()
    long_term_volatility = daily_returns.rolling(window=10).std()
    
    long_term_alpha_factor = (long_term_weighted_avg + long_term_open_to_close_momentum - long_term_avg_daily_return) / long_term_volatility

    # Calculate Short-Term (5-day) Metrics
    short_term_weighted_price_diff = df['High_Low_Volume'].rolling(window=5).sum()
    total_volume_5_days = df['volume'].rolling(window=5).sum()
    short_term_weighted_avg = short_term_weighted_price_diff / total_volume_5_days
    
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(5)
    
    short_term_daily_return = df['close'] - df['close'].shift(1)
    short_term_avg_daily_return = short_term_daily_return.rolling(window=5).sum() / 5
    
    short_term_volatility = daily_returns.rolling(window=5).std()
    
    short_term_alpha_factor = (short_term_weighted_avg + short_term_open_to_close_momentum - short_term_avg_daily_return) / short_term_volatility

    # Combine 10-Day and 5-Day Metrics
    combined_alpha_factor = 0.6 * long_term_alpha_factor + 0.4 * short_term_alpha_factor
    
    # Integrate Trading Volume Effectively
    avg_volume_5_days = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5_days

    return final_alpha_factor
```
