import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['high_low_vol'].rolling(window=22).sum()
    total_volume = df['volume'].rolling(window=22).sum()
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff / total_volume
    
    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(22)
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_average_daily_return = daily_returns.rolling(window=22).sum() / 22
    adjusted_momentum = long_term_open_to_close_momentum - long_term_average_daily_return
    
    # Incorporate Volume Trend
    volume_trend = df['volume'] - df['volume'].shift(22)
    
    # Assign Weights
    weight_volume_trend = 0.3
    weight_long_term_cumulative_weighted_avg = 0.7
    combined_factor = (weight_volume_trend * volume_trend) + (weight_long_term_cumulative_weighted_avg * long_term_cumulative_weighted_avg)
    
    # Adjust for Volatility
    daily_returns_for_volatility = df['close'].pct_change()
    volatility = daily_returns_for_volatility.rolling(window=22).std()
    final_alpha_factor = combined_factor / volatility
    
    return final_alpha_factor
