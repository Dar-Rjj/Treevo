```python
import pandas as pd

def heuristics_v2(df):
    # 10-day calculations
    df['Weighted_Price_Diff_10'] = (df['high'] - df['low']) * df['volume']
    total_weighted_diff_10 = df['Weighted_Price_Diff_10'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    long_term_cumulative_avg_10 = total_weighted_diff_10 / total_volume_10
    
    long_term_momentum_10 = df['close'] - df['open'].shift(10)
    
    daily_return_10 = df['close'] - df['close'].shift(1)
    avg_daily_return_10 = daily_return_10.rolling(window=10).mean()
    
    daily_ret_pct_10 = daily_return_10 / df['close'].shift(1)
    std_dev_10 = daily_ret_pct_10.rolling(window=10).std()
    
    alpha_factor_10 = (long_term_cumulative_avg_10 + long_term_momentum_10 - avg_daily_return_10) / std_dev_10
    
    # 5-day calculations
    df['Weighted_Price_Diff_5'] = (df['high'] - df['low']) * df['volume']
    total_weighted_diff_5 = df['Weighted_Price_Diff_5'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    short_term_cumulative_avg_5 = total_weighted_diff_5 / total_volume_5
    
    short_term_momentum_5 = df['close'] - df['open'].shift(5)
    
    daily_return_5 = df['close'] - df['close'].shift(1)
    avg_daily_return_5 = daily_return_5.rolling(window=5).mean()
    
    daily_ret_pct_5 = daily_return_5 / df['close'].shift(1)
    std_dev_5 = daily_ret_pct_5.rolling(window=5).std()
    
    alpha_factor_5 = (short_term_cumulative_avg_5 + short_term_momentum_5 - avg_daily_return_5) / std_dev_5
    
    # Combine 10-day and 5-day metrics
    combined_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5
    
    # Integrate trading volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5
    
    return final_alpha_factor
```
