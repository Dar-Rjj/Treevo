```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['10_day_weighted_price_diff'] = (df['high'] - df['low']) * df['volume']
    df['10_day_total_volume'] = df['volume'].rolling(window=10).sum()
    df['10_day_weighted_price_sum'] = df['10_day_weighted_price_diff'].rolling(window=10).sum()
    df['10_day_cumulative_weighted_avg'] = df['10_day_weighted_price_sum'] / df['10_day_total_volume']
    
    df['10_day_momentum'] = df['close'] - df['open'].shift(10)
    df['10_day_avg_daily_return'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    df['10_day_daily_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['10_day_volatility'] = df['10_day_daily_returns'].rolling(window=10).std()

    df['10_day_alpha_factor'] = (df['10_day_cumulative_weighted_avg'] - df['10_day_avg_daily_return']) / df['10_day_volatility']

    # 5-day calculations
    df['5_day_weighted_price_diff'] = (df['high'] - df['low']) * df['volume']
    df['5_day_total_volume'] = df['volume'].rolling(window=5).sum()
    df['5_day_weighted_price_sum'] = df['5_day_weighted_price_diff'].rolling(window=5).sum()
    df['5_day_cumulative_weighted_avg'] = df['5_day_weighted_price_sum'] / df['5_day_total_volume']
    
    df['5_day_momentum'] = df['close'] - df['open'].shift(5)
    df['5_day_avg_daily_return'] = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    df['5_day_daily_returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['5_day_volatility'] = df['5_day_daily_returns'].rolling(window=5).std()

    df['5_day_alpha_factor'] = (df['5_day_cumulative_weighted_avg'] - df['5_day_avg_daily_return']) / df['5_day_volatility']

    # Combine 10-day and 5-day metrics
    df['combined_alpha_factor'] = 0.6 * df['10_day_alpha_factor'] + 0.4 * df['5_day_alpha_factor']

    # Adjust for volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['final_alpha_factor'] = df['combined_alpha_factor'] * df['5_day_avg_volume'] * df['10_day_avg_volume']

    return df['final_alpha_factor'].dropna()
```
