import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term Weighted Price Difference
    df['HLV'] = (df['high'] - df['low']) * df['volume']
    hl_sum_10 = df['HLV'].rolling(window=10).sum()
    vol_sum_10 = df['volume'].rolling(window=10).sum()
    dynamic_long_term_weighted_price_diff = hl_sum_10 / vol_sum_10

    # Determine Dynamic Long-Term Open to Close Momentum
    open_t_minus_10 = df['open'].shift(10)
    close_t = df['close']
    dynamic_long_term_momentum = close_t - open_t_minus_10

    # Calculate the Dynamic Long-Term Cumulative Weighted Average
    dynamic_long_term_cumulative_weighted_avg = dynamic_long_term_weighted_price_diff

    # Subtract Dynamic Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    dynamic_long_term_avg_daily_return = daily_returns.rolling(window=10).mean()
    dynamic_long_term_adjusted = dynamic_long_term_cumulative_weighted_avg - dynamic_long_term_avg_daily_return

    # Adjust for Volatility
    daily_returns_pct = daily_returns / df['close'].shift(1)
    volatility_10 = daily_returns_pct.rolling(window=10).std()
    final_alpha_factor_10 = dynamic_long_term_adjusted / volatility_10

    # Incorporate Recent Data
    # Recalculate the above metrics using a 5-day window
    hl_sum_5 = df['HLV'].rolling(window=5).sum()
    vol_sum_5 = df['volume'].rolling(window=5).sum()
    dynamic_short_term_weighted_price_diff = hl_sum_5 / vol_sum_5

    open_t_minus_5 = df['open'].shift(5)
    dynamic_short_term_momentum = close_t - open_t_minus_5

    dynamic_short_term_cumulative_weighted_avg = dynamic_short_term_weighted_price_diff

    daily_returns_5 = df['close'] - df['close'].shift(1)
    dynamic_short_term_avg_daily_return = daily_returns_5.rolling(window=5).mean()
    dynamic_short_term_adjusted = dynamic_short_term_cumulative_weighted_avg - dynamic_short_term_avg_daily_return

    daily_returns_pct_5 = daily_returns_5 / df['close'].shift(1)
    volatility_5 = daily_returns_pct_5.rolling(window=5).std()
    final_alpha_factor_5 = dynamic_short_term_adjusted / volatility_5

    # Combine 10-day and 5-day metrics
    final_alpha_factor = 0.6 * final_alpha_factor_10 + 0.4 * final_alpha_factor_5

    # Integrate Trading Volume Effectively
    avg_vol_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor *= avg_vol_5

    return final_alpha_factor
