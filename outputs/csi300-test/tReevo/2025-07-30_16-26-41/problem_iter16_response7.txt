```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['price_diff_vol_10'] = (df['high'] - df['low']) * df['volume']
    df['sum_price_diff_vol_10'] = df['price_diff_vol_10'].rolling(window=10).sum()
    df['sum_volume_10'] = df['volume'].rolling(window=10).sum()
    df['weighted_price_diff_10'] = df['sum_price_diff_vol_10'] / df['sum_volume_10']
    
    df['open_to_close_momentum_10'] = df['close'] - df['open'].shift(10)
    
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['avg_daily_return_10'] = df['daily_return'].rolling(window=10).mean()
    
    df['volatility_10'] = df['daily_return'].rolling(window=10).std()
    
    df['alpha_factor_10'] = (df['weighted_price_diff_10'] - df['avg_daily_return_10']) / df['volatility_10']
    
    # 5-day calculations
    df['price_diff_vol_5'] = (df['high'] - df['low']) * df['volume']
    df['sum_price_diff_vol_5'] = df['price_diff_vol_5'].rolling(window=5).sum()
    df['sum_volume_5'] = df['volume'].rolling(window=5).sum()
    df['weighted_price_diff_5'] = df['sum_price_diff_vol_5'] / df['sum_volume_5']
    
    df['open_to_close_momentum_5'] = df['close'] - df['open'].shift(5)
    
    df['avg_daily_return_5'] = df['daily_return'].rolling(window=5).mean()
    
    df['volatility_5'] = df['daily_return'].rolling(window=5).std()
    
    df['alpha_factor_5'] = (df['weighted_price_diff_5'] - df['avg_daily_return_5']) / df['volatility_5']
    
    # Combine 10-day and 5-day metrics
    df['combined_alpha_factor'] = 0.6 * df['alpha_factor_10'] + 0.4 * df['alpha_factor_5']
    
    # Adjust by 5-day average volume
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['final_alpha_factor'] = df['combined_alpha_factor'] * df['avg_volume_5']
    
    return df['final_alpha_factor'].dropna()
```
