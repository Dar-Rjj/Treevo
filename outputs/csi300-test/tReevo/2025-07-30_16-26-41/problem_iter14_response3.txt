import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['Daily_Return_10'] = df['close'].pct_change()
    df['Weighted_Price_Diff_10'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_10'] = df['volume'].rolling(window=10).sum()
    df['Cumulative_Weighted_Price_Diff_10'] = df['Weighted_Price_Diff_10'].rolling(window=10).sum()
    df['Cumulative_Weighted_Avg_10'] = df['Cumulative_Weighted_Price_Diff_10'] / df['Total_Volume_10']
    df['Long_Term_Open_Close_Momentum_10'] = df['close'] - df['open'].shift(10)
    df['Avg_Daily_Return_10'] = df['Daily_Return_10'].rolling(window=10).mean()
    df['Std_Dev_10'] = df['Daily_Return_10'].rolling(window=10).std()
    df['Alpha_Factor_10'] = (df['Cumulative_Weighted_Avg_10'] + df['Long_Term_Open_Close_Momentum_10'] - df['Avg_Daily_Return_10']) / df['Std_Dev_10']

    # 5-day calculations
    df['Weighted_Price_Diff_5'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_5'] = df['volume'].rolling(window=5).sum()
    df['Cumulative_Weighted_Price_Diff_5'] = df['Weighted_Price_Diff_5'].rolling(window=5).sum()
    df['Cumulative_Weighted_Avg_5'] = df['Cumulative_Weighted_Price_Diff_5'] / df['Total_Volume_5']
    df['Short_Term_Open_Close_Momentum_5'] = df['close'] - df['open'].shift(5)
    df['Avg_Daily_Return_5'] = df['Daily_Return_10'].rolling(window=5).mean()
    df['Std_Dev_5'] = df['Daily_Return_10'].rolling(window=5).std()
    df['Alpha_Factor_5'] = (df['Cumulative_Weighted_Avg_5'] + df['Short_Term_Open_Close_Momentum_5'] - df['Avg_Daily_Return_5']) / df['Std_Dev_5']

    # Combine 10-day and 5-day metrics
    df['Combined_Alpha_Factor'] = 0.6 * df['Alpha_Factor_10'] + 0.4 * df['Alpha_Factor_5']

    # Adjust for 5-day average volume
    df['5_Day_Avg_Volume'] = df['volume'].rolling(window=5).mean()
    df['Adjusted_Combined_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['5_Day_Avg_Volume']

    # Apply exponential weights
    df['Exponentially_Weighted_Combined_Alpha_Factor'] = 0.8 * df['Alpha_Factor_10'] + 0.2 * df['Alpha_Factor_5']
    df['Final_Alpha_Factor'] = df['Exponentially_Weighted_Combined_Alpha_Factor'] * df['5_Day_Avg_Volume']

    return df['Final_Alpha_Factor'].dropna()
