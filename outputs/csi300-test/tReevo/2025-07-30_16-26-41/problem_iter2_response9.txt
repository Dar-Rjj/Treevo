import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Intraday Midpoint
    df['intraday_midpoint'] = (df['open'] + df['close']) / 2
    
    # Determine Intraday Direction
    df['intraday_direction'] = (df['close'] > df['open']).astype(int) * 2 - 1
    
    # Compute Intraday Momentum Reversal Score
    df['momentum_reversal_score'] = df['intraday_range'] * df['intraday_direction']
    
    # Weight by Volume
    df['weighted_score'] = df['momentum_reversal_score'] * df['volume']
    
    # Incorporate Volume-Based Strengthening
    df['rolling_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_score'] = df.apply(lambda row: row['weighted_score'] * 1.5 if row['volume'] > row['rolling_avg_volume'] else row['weighted_score'], axis=1)
    
    return df['adjusted_score']
