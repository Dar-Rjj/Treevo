```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate short-term and long-term EMA for close prices, sentiment, and news impact
    df['short_ema_close'] = df['close'].ewm(span=5).mean()
    df['long_ema_close'] = df['close'].ewm(span=20).mean()
    
    # Assuming 'sentiment' and 'news_impact' are columns in the DataFrame
    df['short_ema_sentiment'] = df['sentiment'].ewm(span=5).mean()
    df['long_ema_sentiment'] = df['sentiment'].ewm(span=20).mean()
    
    df['short_ema_news'] = df['news_impact'].ewm(span=5).mean()
    df['long_ema_news'] = df['news_impact'].ewm(span=20).mean()
    
    # Volume-Weighted Close Price
    total_volume = df['volume'].rolling(window=5).sum()
    df['vwap_close_short'] = (df['close'] * df['volume']).rolling(window=5).sum() / total_volume
    df['vwap_close_long'] = (df['close'] * df['volume']).rolling(window=20).sum() / total_volume
    
    # Volume-Weighted High-Low Difference
    df['vwap_high_low'] = ((df['high'] - df['low']) * df['volume']).rolling(window=5).sum()
    
    # Time-Series Momentum
    df['momentum'] = df['close'].pct_change(20)
    
    # Sector Rotation (assuming 'sector' is a column in the DataFrame)
    sector_performance = df.groupby('sector')['close'].apply(lambda x: x.pct_change(20).mean()).reset_index(name='sector_momentum')
    df = df.merge(sector_performance, on='sector', how='left')
    
    # Exponential Smoothing
    df['smoothed_short_ema_close'] = df['short_ema_close'].ewm(alpha=0.1).mean()
    df['smoothed_long_ema_close'] = df['long_ema_close'].ewm(alpha=0.1).mean()
    
    df['smoothed_short_ema_sentiment'] = df['short_ema_sentiment'].ewm(alpha=0.1).mean()
    df['smoothed_long_ema_sentiment'] = df['long_ema_sentiment'].ewm(alpha=0.1).mean()
    
    df['smoothed_short_ema_news'] = df['short_ema_news'].ewm(alpha=0.1).mean()
    df['smoothed_long_ema_news'] = df['long_ema_news'].ewm(alpha=0.1).mean()
    
    # Combine all factors
    df['alpha_factor'] = (
        df['smoothed_short_ema_close'] + 
        df['smoothed_long_ema_close'] + 
        df['smoothed_short_ema_sentiment'] + 
        df['smoothed_long_ema_sentiment'] + 
        df['smoothed_short_ema_news'] + 
        df['smoothed_long_ema_news'] + 
        df['vwap_close_short'] + 
        df['vwap_close_long'] + 
        df['vwap_high_low'] + 
        df['momentum'] + 
        df['sector_momentum']
    )
    
    return df['alpha_factor']
```
