```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-High Ratio
    close_high_ratio = df['close'] / df['high']
    
    # Calculate Low-Open Ratio
    low_open_ratio = df['low'] / df['open']
    
    # Assign Weights
    weight_close_high = 0.6
    weight_low_open = 0.4
    
    # Multiply Ratios by Weights
    weighted_close_high = close_high_ratio * weight_close_high
    weighted_low_open = low_open_ratio * weight_low_open
    
    # Sum of Weighted Ratios
    trend_strength = weighted_close_high + weighted_low_open
    
    # Apply Volume Filter
    median_volume = df['volume'].median()
    trend_strength_filtered = trend_strength[df['volume'] > median_volume]
    
    return trend_strength_filtered
```
