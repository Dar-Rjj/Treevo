```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term Weighted Price Difference
    df['hl_vol_10'] = (df['high'] - df['low']) * df['volume']
    sum_hl_vol_10 = df['hl_vol_10'].rolling(window=10).sum()
    sum_vol_10 = df['volume'].rolling(window=10).sum()
    weighted_price_diff_10 = sum_hl_vol_10 / sum_vol_10

    # Determine Dynamic Long-Term Open to Close Momentum
    open_t_10 = df['open'].shift(10)
    close_t = df['close']
    open_close_momentum_10 = close_t - open_t_10

    # Calculate the Dynamic Long-Term Cumulative Weighted Average
    dynamic_weighted_avg_10 = weighted_price_diff_10

    # Subtract Dynamic Long-Term Average Daily Return
    daily_returns_10 = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    adjusted_weighted_avg_10 = dynamic_weighted_avg_10 - daily_returns_10

    # Adjust for Volatility
    daily_return_10 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_10 = daily_return_10.rolling(window=10).std()
    final_alpha_10 = adjusted_weighted_avg_10 / std_10

    # Incorporate Recent Data using a 5-day window
    df['hl_vol_5'] = (df['high'] - df['low']) * df['volume']
    sum_hl_vol_5 = df['hl_vol_5'].rolling(window=5).sum()
    sum_vol_5 = df['volume'].rolling(window=5).sum()
    weighted_price_diff_5 = sum_hl_vol_5 / sum_vol_5

    open_t_5 = df['open'].shift(5)
    close_t = df['close']
    open_close_momentum_5 = close_t - open_t_5

    dynamic_weighted_avg_5 = weighted_price_diff_5
    daily_returns_5 = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    adjusted_weighted_avg_5 = dynamic_weighted_avg_5 - daily_returns_5

    daily_return_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_5 = daily_return_5.rolling(window=5).std()
    final_alpha_5 = adjusted_weighted_avg_5 / std_5

    # Combine 10-day and 5-day metrics
    combined_alpha = 0.6 * final_alpha_10 + 0.4 * final_alpha_5

    # Integrate Trading Volume Effectively
    volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha = combined_alpha * volume_5

    return final_alpha
```
