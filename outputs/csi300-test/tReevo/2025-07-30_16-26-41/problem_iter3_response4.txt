```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_high_low_vol_sum = df['high_low_vol'].rolling(window=20).sum()
    total_volume_20_days = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff = long_term_high_low_vol_sum / total_volume_20_days
    
    # Determine Long-Term Open to Close Momentum
    open_t_minus_20 = df['open'].shift(20)
    close_t = df['close']
    long_term_open_to_close_momentum = close_t - open_t_minus_20
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff * long_term_open_to_close_momentum
    
    # Subtract Long-Term Average Daily Return
    daily_return = df['close'].pct_change()
    long_term_avg_daily_return = daily_return.rolling(window=20).mean()
    adjusted_long_term_cumulative_weighted_avg = long_term_cumulative_weighted_avg - long_term_avg_daily_return
    
    # Adjust for Volatility
    daily_returns = df['close'].pct_change().fillna(0)
    volatility = daily_returns.rolling(window=20).std()
    final_alpha_factor = adjusted_long_term_cumulative_weighted_avg / volatility
    
    return final_alpha_factor
```
