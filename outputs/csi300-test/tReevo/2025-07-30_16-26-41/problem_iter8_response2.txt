import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Weighted Price Difference
    df['High_Low_Volume'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff = df['High_Low_Volume'].rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    short_term_weighted_price_diff = weighted_price_diff / total_volume
    
    # Determine Short-Term Open to Close Momentum
    open_5_days_ago = df['open'].shift(5)
    close_today = df['close']
    short_term_momentum = close_today - open_5_days_ago
    
    # Calculate the Short-Term Cumulative Weighted Average
    cumulative_weighted_average = short_term_weighted_price_diff
    
    # Subtract Short-Term Average Daily Return
    daily_returns = close_today - df['close'].shift(1)
    short_term_avg_daily_return = daily_returns.rolling(window=5).mean()
    
    # Adjust for Volatility
    daily_return_pct = daily_returns / df['close'].shift(1)
    five_day_std_dev = daily_return_pct.rolling(window=5).std()
    
    # Final Alpha Factor
    alpha_factor = (cumulative_weighted_average - short_term_avg_daily_return) / five_day_std_dev
    
    return alpha_factor
