```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    sum_high_low_vol_20 = df['high_low_vol'].rolling(window=20).sum()
    sum_vol_20 = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff = sum_high_low_vol_20 / sum_vol_20
    
    # Determine Long-Term Open to Close Momentum
    long_term_open_to_close_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff * long_term_open_to_close_momentum
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).mean()
    long_term_cumulative_weighted_avg -= long_term_avg_daily_return
    
    # Adjust for Volatility
    daily_returns_pct = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    vol_20 = daily_returns_pct.rolling(window=20).std()
    
    # Final Alpha Factor
    alpha_factor = long_term_cumulative_weighted_avg / vol_20
    
    return alpha_factor
```
