```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term (10-day) Metrics
    df['HLV_10'] = (df['high'] - df['low']) * df['volume']
    df['HLV_10_sum'] = df['HLV_10'].rolling(window=10).sum()
    df['Volume_10_sum'] = df['volume'].rolling(window=10).sum()
    df['Dynamic_LTCumulativeWeightedAverage_10'] = df['HLV_10_sum'] / df['Volume_10_sum']
    
    df['Dynamic_LTOpenToCloseMomentum_10'] = df['close'] - df['open'].shift(10)
    
    df['Daily_Returns_10'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Dynamic_LT_AverageDailyReturn_10'] = df['Daily_Returns_10'].rolling(window=10).mean()
    df['Dynamic_LTStdDev_10'] = df['Daily_Returns_10'].rolling(window=10).std()
    
    df['Alpha_Factor_10'] = (df['Dynamic_LTCumulativeWeightedAverage_10'] - df['Dynamic_LT_AverageDailyReturn_10']) / df['Dynamic_LTStdDev_10']
    
    # Calculate Dynamic Short-Term (5-day) Metrics
    df['HLV_5'] = (df['high'] - df['low']) * df['volume']
    df['HLV_5_sum'] = df['HLV_5'].rolling(window=5).sum()
    df['Volume_5_sum'] = df['volume'].rolling(window=5).sum()
    df['Dynamic_STCumulativeWeightedAverage_5'] = df['HLV_5_sum'] / df['Volume_5_sum']
    
    df['Dynamic_STOpenToCloseMomentum_5'] = df['close'] - df['open'].shift(5)
    
    df['Daily_Returns_5'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['Dynamic_ST_AverageDailyReturn_5'] = df['Daily_Returns_5'].rolling(window=5).mean()
    df['Dynamic_STStdDev_5'] = df['Daily_Returns_5'].rolling(window=5).std()
    
    df['Alpha_Factor_5'] = (df['Dynamic_STCumulativeWeightedAverage_5'] - df['Dynamic_ST_AverageDailyReturn_5']) / df['Dynamic_STStdDev_5']
    
    # Combine 10-Day and 5-Day Alpha Factors
    df['Combined_Alpha_Factor'] = 0.6 * df['Alpha_Factor_10'] + 0.4 * df['Alpha_Factor_5']
    
    # Integrate Trading Volume Effectively
    df['5Day_Average_Volume'] = df['volume'].rolling(window=5).mean()
    df['Final_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['5Day_Average_Volume']
    
    return df['Final_Alpha_Factor']
```
