import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    weighted_price_diff = (df['high'] - df['low']) * df['volume']
    weighted_price_diff_sum = weighted_price_diff.rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    long_term_weighted_price_diff = weighted_price_diff_sum / total_volume

    # Determine Long-Term Open to Close Momentum
    open_t_minus_10 = df['open'].shift(10)
    close_t = df['close']
    long_term_open_to_close_momentum = close_t - open_t_minus_10

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff

    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'].pct_change()
    long_term_average_daily_return = daily_returns.rolling(window=10).sum() / 10
    long_term_cumulative_weighted_avg -= long_term_average_daily_return

    # Adjust for Volatility
    daily_returns_std = daily_returns.rolling(window=10).std()
    alpha_factor = long_term_cumulative_weighted_avg / daily_returns_std

    return alpha_factor
