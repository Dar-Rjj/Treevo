```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a volume-weighted close-to-open return.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as date.
    
    Returns:
    pd.Series: A Series with the alpha factor values, indexed by date.
    """
    # Calculate the close-to-open return
    close_to_open_return = df['open'] / df['close'].shift(1) - 1
    
    # Weight the return by volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Normalize the factor to make it comparable across different stocks or time periods
    normalized_factor = volume_weighted_return / volume_weighted_return.rolling(window=20).std()
    
    return normalized_factor
```
