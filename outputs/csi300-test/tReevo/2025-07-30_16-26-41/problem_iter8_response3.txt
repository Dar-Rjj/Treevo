import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['price_diff_vol_weighted'] = (df['high'] - df['low']) * df['volume']
    long_term_price_diff = df['price_diff_vol_weighted'].rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    
    # Determine Long-Term Open to Close Momentum
    open_10_days_ago = df['open'].shift(10)
    close_today = df['close']
    long_term_momentum = close_today - open_10_days_ago
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_price_diff / total_volume
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=10).mean()
    
    # Adjust for Volatility
    daily_pct_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility = daily_pct_returns.rolling(window=10).std()
    
    # Final Alpha Factor
    alpha_factor = (long_term_cumulative_weighted_avg + long_term_momentum - long_term_avg_daily_return) / volatility
    
    return alpha_factor.dropna()
