```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['price_diff_vol_10'] = (df['high'] - df['low']) * df['volume']
    df['total_vol_10'] = df['volume'].rolling(window=10).sum()
    df['weighted_price_diff_10'] = df['price_diff_vol_10'].rolling(window=10).sum() / df['total_vol_10']
    df['long_term_momentum_10'] = df['close'] - df['open'].shift(10)
    df['cumulative_weighted_avg_10'] = (df['weighted_price_diff_10']).rolling(window=10).mean()
    df['avg_daily_return_10'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    df['daily_returns_10'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['ewm_std_10'] = df['daily_returns_10'].ewm(span=10, adjust=False).std()

    # 5-day calculations
    df['price_diff_vol_5'] = (df['high'] - df['low']) * df['volume']
    df['total_vol_5'] = df['volume'].rolling(window=5).sum()
    df['weighted_price_diff_5'] = df['price_diff_vol_5'].rolling(window=5).sum() / df['total_vol_5']
    df['short_term_momentum_5'] = df['close'] - df['open'].shift(5)
    df['cumulative_weighted_avg_5'] = (df['weighted_price_diff_5']).rolling(window=5).mean()
    df['avg_daily_return_5'] = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    df['daily_returns_5'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['ewm_std_5'] = df['daily_returns_5'].ewm(span=5, adjust=False).std()

    # Final 10-day alpha factor
    df['alpha_10'] = (df['cumulative_weighted_avg_10'] - df['avg_daily_return_10']) / df['ewm_std_10']

    # Final 5-day alpha factor
    df['alpha_5'] = (df['cumulative_weighted_avg_5'] - df['avg_daily_return_5']) / df['ewm_std_5']

    # Relative Volatility Ratio
    df['relative_volatility_ratio'] = df['ewm_std_5'] / df['ewm_std_10']

    # Dynamic Weights
    df['weight_10'] = 1 / (1 + df['relative_volatility_ratio'])
    df['weight_5'] = df['relative_volatility_ratio'] / (1 + df['relative_volatility_ratio'])

    # Combined Alpha Factor
    df['combined_alpha'] = (df['alpha_10'] * df['weight_10']) + (df['alpha_5'] * df['weight_5'])

    # Volume and Liquidity Adjustments
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['liquidity_adjustment_factor'] = df['avg_volume_5'] / df['avg_volume_20']

    # Final Alpha Factor
    df['final_alpha'] = (df['combined_alpha'] * df['avg_volume_5']) * df['liquidity_adjustment_factor']

    return df['final_alpha'].dropna()
```
