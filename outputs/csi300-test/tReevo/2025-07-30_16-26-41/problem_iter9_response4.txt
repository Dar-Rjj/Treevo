import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic Long-Term Weighted Price Difference
    df['hl_diff_vol_10'] = (df['high'] - df['low']) * df['volume']
    dynamic_long_weighted_price_diff_10 = df['hl_diff_vol_10'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    
    # Determine Dynamic Long-Term Open to Close Momentum
    dynamic_long_momentum_10 = df['close'] - df['open'].shift(10)
    
    # Calculate the Dynamic Long-Term Cumulative Weighted Average
    dynamic_long_cumulative_avg_10 = dynamic_long_weighted_price_diff_10 / total_volume_10
    
    # Subtract Dynamic Long-Term Average Daily Return
    dynamic_long_avg_daily_return_10 = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    
    # Adjust for Volatility
    daily_returns_10 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility_10 = daily_returns_10.rolling(window=10).std()
    
    # Final Alpha Factor (10-day)
    alpha_factor_10 = (dynamic_long_cumulative_avg_10 - dynamic_long_avg_daily_return_10) / volatility_10
    
    # Incorporate Recent Data
    # Recalculate the above metrics using a 5-day window
    df['hl_diff_vol_5'] = (df['high'] - df['low']) * df['volume']
    dynamic_short_weighted_price_diff_5 = df['hl_diff_vol_5'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    
    # Determine Dynamic Short-Term Open to Close Momentum
    dynamic_short_momentum_5 = df['close'] - df['open'].shift(5)
    
    # Calculate the Dynamic Short-Term Cumulative Weighted Average
    dynamic_short_cumulative_avg_5 = dynamic_short_weighted_price_diff_5 / total_volume_5
    
    # Subtract Dynamic Short-Term Average Daily Return
    dynamic_short_avg_daily_return_5 = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # Adjust for Volatility
    daily_returns_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility_5 = daily_returns_5.rolling(window=5).std()
    
    # Final Alpha Factor (5-day)
    alpha_factor_5 = (dynamic_short_cumulative_avg_5 - dynamic_short_avg_daily_return_5) / volatility_5
    
    # Combine 10-day and 5-day metrics
    combined_alpha_factor = 0.6 * alpha_factor_10 + 0.4 * alpha_factor_5
    
    # Integrate Trading Volume Effectively
    volume_weighted_price_diff_5 = (df['high'] - df['low']) * df['volume']
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    
    # Adjust the Final Combined Alpha Factor
    final_alpha_factor = combined_alpha_factor * avg_volume_5
    
    return final_alpha_factor
