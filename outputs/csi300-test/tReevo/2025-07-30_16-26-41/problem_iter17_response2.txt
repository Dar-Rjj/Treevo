```python
import pandas as pd

def heuristics_v2(df):
    # Short-Term Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    short_term_ema = df['close'].ewm(span=5, adjust=False).mean()
    
    # Long-Term Momentum
    long_term_ma = df['close'].rolling(window=20).mean()
    long_term_ema = df['close'].ewm(span=20, adjust=False).mean()
    
    # Volume-Weighted Close Price
    vwap = (df['amount'] / df['volume']).rolling(window=10).mean()
    close_to_vwap_ratio = df['close'] / vwap
    
    # Volume-Weighted Moving Average
    vwma = (df['close'] * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    close_to_vwma_ratio = df['close'] / vwma
    
    # Exponential Volatility
    high_low_range_ema = (df['high'] - df['low']).ewm(span=10, adjust=False).mean()
    current_high_low_ratio = (df['high'] - df['low']) / high_low_range_ema
    
    # Exponential Momentum
    close_change_ema = df['close'].diff().ewm(span=10, adjust=False).mean()
    current_close_change_ratio = df['close'].diff() / close_change_ema
    
    # Volume Trend
    volume_sma = df['volume'].rolling(window=10).mean()
    volume_trend_ratio = df['volume'] / volume_sma
    
    # Recent Market Conditions
    recent_high = df['high'].rolling(window=5).max()
    recent_low = df['low'].rolling(window=5).min()
    close_to_recent_high_ratio = df['close'] / recent_high
    close_to_recent_low_ratio = df['close'] / recent_low
    recent_close_change = df['close'].pct_change(5)
    
    # Combine all factors into a single alpha factor
    alpha_factor = (
        short_term_ma + 
        short_term_ema + 
        long_term_ma + 
        long_term_ema + 
        close_to_vwap_ratio + 
        close_to_vwma_ratio + 
        current_high_low_ratio + 
        current_close_change_ratio + 
        volume_trend_ratio + 
        close_to_recent_high_ratio + 
        close_to_recent_low_ratio + 
        recent_close_change
    ) / 12  # Normalize by the number of factors
    
    return alpha_factor
```
