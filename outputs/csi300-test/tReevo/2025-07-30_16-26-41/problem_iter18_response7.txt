```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term (10-day) Metrics
    df['hl_vol_10'] = (df['high'] - df['low']) * df['volume']
    df['hl_vol_sum_10'] = df['hl_vol_10'].rolling(window=10).sum()
    df['vol_sum_10'] = df['volume'].rolling(window=10).sum()
    df['weighted_price_diff_10'] = df['hl_vol_sum_10'] / df['vol_sum_10']
    
    df['long_term_momentum_10'] = df['close'] - df['open'].shift(10)
    df['daily_return_10'] = df['close'] - df['close'].shift(1)
    df['avg_daily_return_10'] = df['daily_return_10'].rolling(window=10).mean()
    df['std_daily_return_10'] = df['daily_return_10'].rolling(window=10).std()
    
    df['long_term_alpha_10'] = (df['weighted_price_diff_10'] - df['avg_daily_return_10']) / df['std_daily_return_10']
    
    # Calculate Short-Term (5-day) Metrics
    df['hl_vol_5'] = (df['high'] - df['low']) * df['volume']
    df['hl_vol_sum_5'] = df['hl_vol_5'].rolling(window=5).sum()
    df['vol_sum_5'] = df['volume'].rolling(window=5).sum()
    df['weighted_price_diff_5'] = df['hl_vol_sum_5'] / df['vol_sum_5']
    
    df['short_term_momentum_5'] = df['close'] - df['open'].shift(5)
    df['daily_return_5'] = df['close'] - df['close'].shift(1)
    df['avg_daily_return_5'] = df['daily_return_5'].rolling(window=5).mean()
    df['std_daily_return_5'] = df['daily_return_5'].rolling(window=5).std()
    
    df['short_term_alpha_5'] = (df['weighted_price_diff_5'] - df['avg_daily_return_5']) / df['std_daily_return_5']
    
    # Combine 10-Day and 5-Day Metrics
    df['combined_alpha'] = 0.6 * df['long_term_alpha_10'] + 0.4 * df['short_term_alpha_5']
    
    # Integrate Trading Volume Effectively
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['adjusted_combined_alpha'] = df['combined_alpha'] * df['avg_volume_5']
    
    # Incorporate Price Change Volatility
    df['high_low_range_10'] = df['high'] - df['low'].shift(9)
    df['high_low_range_5'] = df['high'] - df['low'].shift(4)
    
    df['final_alpha'] = df['adjusted_combined_alpha'] / df['high_low_range_10'] / df['high_low_range_5']
    
    return df['final_alpha'].dropna()
```
