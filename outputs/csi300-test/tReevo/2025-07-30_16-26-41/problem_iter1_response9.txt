```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a novel alpha factor based on weighted momentum change from open to close.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'] and index as date.
    
    Returns:
    pd.Series: A Series with the alpha factor values indexed by date.
    """
    # Calculate Weighted Price Difference
    daily_price_diff = (df['high'] - df['low']) * df['volume']
    total_volume_20_days = df['volume'].rolling(window=20).sum()
    weighted_price_diff = daily_price_diff.rolling(window=20).sum() / total_volume_20_days
    
    # Determine Open to Close Momentum
    open_to_close_momentum = df['close'] - df['open'].shift(1)
    
    # Calculate the Cumulative Weighted Average
    cumulative_weighted_avg = (open_to_close_momentum * df['volume']).rolling(window=20).sum() / total_volume_20_days
    
    # Subtract Average Daily Return
    avg_daily_return = (df['close'] - df['close'].shift(1)).rolling(window=20).mean()
    
    # Final Alpha Factor
    alpha_factor = cumulative_weighted_avg - avg_daily_return
    
    return alpha_factor
```
