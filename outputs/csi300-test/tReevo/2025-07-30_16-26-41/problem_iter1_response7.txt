```python
import pandas as pd

def heuristics_v2(df, n_days=10):
    # Calculate High-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Price Direction Indicator
    price_direction = (df['close'] > df['close'].shift(1)).astype(int) * 2 - 1
    
    # Calculate Volume Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Calculate Smoothed High-Low Range
    smoothed_high_low_range = (volume_weighted_high_low_range.rolling(window=n_days).sum() / 
                               df['volume'].rolling(window=n_days).sum())
    
    # Adjust for Price Direction
    adjusted_smoothed_high_low_range = smoothed_high_low_range * price_direction
    
    return adjusted_smoothed_high_low_range
```
