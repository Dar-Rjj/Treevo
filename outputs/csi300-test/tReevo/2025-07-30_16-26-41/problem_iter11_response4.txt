```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['HLV'] = (df['high'] - df['low']) * df['volume']
    long_term_sum_hlv = df['HLV'].rolling(window=10).sum()
    long_term_sum_volume = df['volume'].rolling(window=10).sum()
    
    df['long_term_weighted_price_diff'] = long_term_sum_hlv / long_term_sum_volume
    df['long_term_open_to_close_momentum'] = df['close'] - df['open'].shift(10)
    
    df['long_term_cumulative_weighted_avg'] = df['long_term_weighted_price_diff'] / long_term_sum_volume
    daily_returns_10 = df['close'].diff()
    long_term_avg_daily_return = daily_returns_10.rolling(window=10).sum() / 10
    df['long_term_volatility'] = df['close'].pct_change().rolling(window=10).std()
    
    df['long_term_alpha'] = (df['long_term_cumulative_weighted_avg'] - long_term_avg_daily_return) / df['long_term_volatility']
    
    # 5-day calculations
    short_term_sum_hlv = df['HLV'].rolling(window=5).sum()
    short_term_sum_volume = df['volume'].rolling(window=5).sum()
    
    df['short_term_weighted_price_diff'] = short_term_sum_hlv / short_term_sum_volume
    df['short_term_open_to_close_momentum'] = df['close'] - df['open'].shift(5)
    
    df['short_term_cumulative_weighted_avg'] = df['short_term_weighted_price_diff'] / short_term_sum_volume
    daily_returns_5 = df['close'].diff()
    short_term_avg_daily_return = daily_returns_5.rolling(window=5).sum() / 5
    df['short_term_volatility'] = df['close'].pct_change().rolling(window=5).std()
    
    df['short_term_alpha'] = (df['short_term_cumulative_weighted_avg'] - short_term_avg_daily_return) / df['short_term_volatility']
    
    # Combine 10-day and 5-day metrics
    df['combined_alpha'] = 0.6 * df['long_term_alpha'] + 0.4 * df['short_term_alpha']
    
    # Adjust by 5-day average volume
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['final_alpha'] = df['combined_alpha'] * df['5_day_avg_volume']
    
    return df['final_alpha'].dropna()
```
