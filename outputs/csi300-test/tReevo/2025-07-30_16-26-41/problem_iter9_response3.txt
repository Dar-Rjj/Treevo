```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Dynamic Long-Term (10-day) Metrics
    df['high_low_vol_10'] = (df['high'] - df['low']) * df['volume']
    total_vol_10 = df['volume'].rolling(window=10).sum()
    dynamic_long_term_weighted_price_diff_10 = df['high_low_vol_10'].rolling(window=10).sum()
    dynamic_long_term_cumulative_weighted_avg_10 = dynamic_long_term_weighted_price_diff_10 / total_vol_10
    dynamic_long_term_open_to_close_momentum_10 = df['close'] - df['open'].shift(10)
    dynamic_long_term_avg_daily_return_10 = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    daily_returns_10 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev_10 = daily_returns_10.rolling(window=10).std()
    final_10_day_alpha_factor = (dynamic_long_term_cumulative_weighted_avg_10 + 
                                 dynamic_long_term_open_to_close_momentum_10 - 
                                 dynamic_long_term_avg_daily_return_10) / std_dev_10

    # Calculate Dynamic Short-Term (5-day) Metrics
    df['high_low_vol_5'] = (df['high'] - df['low']) * df['volume']
    total_vol_5 = df['volume'].rolling(window=5).sum()
    dynamic_short_term_weighted_price_diff_5 = df['high_low_vol_5'].rolling(window=5).sum()
    dynamic_short_term_cumulative_weighted_avg_5 = dynamic_short_term_weighted_price_diff_5 / total_vol_5
    dynamic_short_term_open_to_close_momentum_5 = df['close'] - df['open'].shift(5)
    dynamic_short_term_avg_daily_return_5 = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    daily_returns_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_dev_5 = daily_returns_5.rolling(window=5).std()
    final_5_day_alpha_factor = (dynamic_short_term_cumulative_weighted_avg_5 + 
                                dynamic_short_term_open_to_close_momentum_5 - 
                                dynamic_short_term_avg_daily_return_5) / std_dev_5

    # Combine 10-Day and 5-Day Metrics
    combined_alpha_factor = 0.6 * final_10_day_alpha_factor + 0.4 * final_5_day_alpha_factor

    # Integrate Trading Volume Effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5

    return final_alpha_factor
```
