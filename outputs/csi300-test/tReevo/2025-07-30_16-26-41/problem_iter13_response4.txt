```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate the 10-day metrics
    df['high_low_vol_10'] = (df['high'] - df['low']) * df['volume']
    df['sum_high_low_vol_10'] = df['high_low_vol_10'].rolling(window=10).sum()
    df['sum_volume_10'] = df['volume'].rolling(window=10).sum()
    df['dynamic_long_term_weighted_price_diff_10'] = df['sum_high_low_vol_10'] / df['sum_volume_10']
    df['dynamic_long_term_open_close_momentum_10'] = df['close'] - df['open'].shift(10)
    df['dynamic_long_term_cumulative_weighted_average_10'] = (df['dynamic_long_term_weighted_price_diff_10'] + df['dynamic_long_term_open_close_momentum_10']) / df['sum_volume_10']
    df['dynamic_long_term_avg_daily_return_10'] = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    daily_returns_10 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volatility_10'] = daily_returns_10.rolling(window=10).std()
    df['alpha_factor_10'] = (df['dynamic_long_term_cumulative_weighted_average_10'] - df['dynamic_long_term_avg_daily_return_10']) / df['volatility_10']

    # Calculate the 5-day metrics
    df['high_low_vol_5'] = (df['high'] - df['low']) * df['volume']
    df['sum_high_low_vol_5'] = df['high_low_vol_5'].rolling(window=5).sum()
    df['sum_volume_5'] = df['volume'].rolling(window=5).sum()
    df['dynamic_short_term_weighted_price_diff_5'] = df['sum_high_low_vol_5'] / df['sum_volume_5']
    df['dynamic_short_term_open_close_momentum_5'] = df['close'] - df['open'].shift(5)
    df['dynamic_short_term_cumulative_weighted_average_5'] = (df['dynamic_short_term_weighted_price_diff_5'] + df['dynamic_short_term_open_close_momentum_5']) / df['sum_volume_5']
    df['dynamic_short_term_avg_daily_return_5'] = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    daily_returns_5 = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volatility_5'] = daily_returns_5.rolling(window=5).std()
    df['alpha_factor_5'] = (df['dynamic_short_term_cumulative_weighted_average_5'] - df['dynamic_short_term_avg_daily_return_5']) / df['volatility_5']

    # Combine 10-day and 5-day metrics
    df['combined_alpha_factor'] = 0.6 * df['alpha_factor_10'] + 0.4 * df['alpha_factor_5']

    # Integrate trading volume effectively
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['combined_alpha_factor_vol_adjusted'] = df['combined_alpha_factor'] * df['avg_volume_5']

    # Incorporate recent data with exponential weights
    df['combined_alpha_factor_exp_weighted'] = 0.8 * df['alpha_factor_10'] + 0.2 * df['alpha_factor_5']
    df['final_alpha_factor'] = df['combined_alpha_factor_exp_weighted'] * df['avg_volume_5']

    return df['final_alpha_factor']
```
