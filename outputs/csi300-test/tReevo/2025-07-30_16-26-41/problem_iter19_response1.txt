import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Adaptive EMA Windows
    short_ema_window = 5
    long_ema_window = 20
    
    # Calculate EMA of close prices
    df['short_ema_close'] = df['close'].ewm(span=short_ema_window, adjust=False).mean()
    df['long_ema_close'] = df['close'].ewm(span=long_ema_window, adjust=False).mean()
    
    # Volume-Weighted Metrics
    # Volume-Weighted Close Price
    total_volume = df['volume'].sum()
    df['volume_weighted_close'] = (df['close'] * df['volume']) / total_volume
    df['short_ema_volume_weighted_close'] = df['volume_weighted_close'].ewm(span=short_ema_window, adjust=False).mean()
    df['long_ema_volume_weighted_close'] = df['volume_weighted_close'].ewm(span=long_ema_window, adjust=False).mean()
    
    # Volume-Weighted High-Low Difference
    df['volume_weighted_high_low_diff'] = (df['high'] - df['low']) * df['volume']
    df['volume_weighted_high_low_diff_sum'] = df['volume_weighted_high_low_diff'].rolling(window=short_ema_window).sum()
    
    # Dynamic Factor Weighting
    # Time-Series Momentum
    df['momentum'] = df['close'].pct_change().rolling(window=short_ema_window).mean()
    momentum_sign = np.sign(df['momentum'])
    df['adjusted_short_ema_close'] = df['short_ema_close'] * momentum_sign
    df['adjusted_long_ema_close'] = df['long_ema_close'] * momentum_sign
    
    # Sector Rotation (Assuming sector data is available)
    # For simplicity, we'll assume a fixed sector performance factor
    # In practice, this would be dynamic and based on recent sector performance
    sector_performance_factor = 1.0  # Placeholder for actual sector performance factor
    df['adjusted_short_ema_close'] *= sector_performance_factor
    df['adjusted_long_ema_close'] *= sector_performance_factor
    
    # Exponential Smoothing
    alpha = 0.2  # Smoothing factor
    df['smoothed_short_ema_close'] = df['adjusted_short_ema_close'].ewm(alpha=alpha, adjust=False).mean()
    df['smoothed_long_ema_close'] = df['adjusted_long_ema_close'].ewm(alpha=alpha, adjust=False).mean()
    
    # Combine all factors to create the final alpha factor
    df['alpha_factor'] = (
        df['smoothed_short_ema_close'] - 
        df['smoothed_long_ema_close'] + 
        df['short_ema_volume_weighted_close'] - 
        df['long_ema_volume_weighted_close'] + 
        df['volume_weighted_high_low_diff_sum']
    )
    
    return df['alpha_factor']
