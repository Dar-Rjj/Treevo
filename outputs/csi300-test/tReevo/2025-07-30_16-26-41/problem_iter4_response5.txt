```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['price_diff_vol'] = (df['high'] - df['low']) * df['volume']
    long_term_price_diff = df['price_diff_vol'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_weighted_avg = long_term_price_diff / total_volume
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).sum() / 20
    
    # Adjust for Volatility
    daily_returns_pct = daily_returns / df['close'].shift(1)
    volatility = daily_returns_pct.rolling(window=20).std()
    
    # Incorporate Intraday Momentum and Reversal
    intraday_range = df['high'] - df['low']
    closing_position = (df['close'] - df['low']) / intraday_range
    opening_position = (df['open'] - df['low']) / intraday_range
    reversal_potential = np.where(closing_position > opening_position, 1, -1)
    volume_weighted_reversal = reversal_potential * df['volume']
    
    # Calculate Intraday Momentum
    close_to_close_return = df['close'] - df['close'].shift(1)
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    intraday_momentum = close_to_close_return * rolling_avg_volume
    
    # Combine with Reversal Potential
    fraction = np.where(df['amount'] < 1000, 0.5, 0.8)  # Assuming amount is a proxy for number of trades
    combined_intraday = intraday_momentum + (volume_weighted_reversal * -fraction)
    
    # Final Alpha Factor
    alpha_factor = (
        long_term_weighted_avg 
        - long_term_avg_daily_return 
        / volatility 
        + combined_intraday
    )
    
    return alpha_factor
```
