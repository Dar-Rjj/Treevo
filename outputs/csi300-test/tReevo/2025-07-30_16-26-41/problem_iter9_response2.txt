import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Dynamic Long-Term Weighted Price Difference
    high_low_diff = (df['high'] - df['low']) * df['volume']
    sum_high_low_diff_10 = high_low_diff.rolling(window=10).sum()
    sum_volume_10 = df['volume'].rolling(window=10).sum()
    dynamic_long_term_weighted_price_diff = sum_high_low_diff_10 / sum_volume_10

    # Determine Dynamic Long-Term Open to Close Momentum
    open_lag_10 = df['open'].shift(10)
    dynamic_long_term_momentum = df['close'] - open_lag_10

    # Calculate the Dynamic Long-Term Cumulative Weighted Average
    dynamic_long_term_cumulative_avg = dynamic_long_term_weighted_price_diff / sum_volume_10

    # Subtract Dynamic Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    avg_daily_return_10 = daily_returns.rolling(window=10).mean()
    dynamic_long_term_cumulative_avg -= avg_daily_return_10

    # Adjust for Volatility
    daily_returns_pct = daily_returns / df['close'].shift(1)
    std_dev_10 = daily_returns_pct.rolling(window=10).std()
    adjusted_factor_10 = dynamic_long_term_cumulative_avg / std_dev_10

    # Recalculate the above metrics using a 5-day window
    sum_high_low_diff_5 = high_low_diff.rolling(window=5).sum()
    sum_volume_5 = df['volume'].rolling(window=5).sum()
    dynamic_long_term_weighted_price_diff_5 = sum_high_low_diff_5 / sum_volume_5
    dynamic_long_term_cumulative_avg_5 = dynamic_long_term_weighted_price_diff_5 / sum_volume_5
    avg_daily_return_5 = daily_returns.rolling(window=5).mean()
    dynamic_long_term_cumulative_avg_5 -= avg_daily_return_5
    std_dev_5 = daily_returns_pct.rolling(window=5).std()
    adjusted_factor_5 = dynamic_long_term_cumulative_avg_5 / std_dev_5

    # Combine 10-day and 5-day metrics
    combined_factor = 0.6 * adjusted_factor_10 + 0.4 * adjusted_factor_5

    # Integrate Trading Volume Effectively
    volume_weighted_price_diff_5 = sum_high_low_diff_5 / sum_volume_5
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_factor * avg_volume_5

    return final_alpha_factor
