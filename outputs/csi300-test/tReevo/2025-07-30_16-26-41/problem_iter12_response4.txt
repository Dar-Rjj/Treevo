```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day metrics
    df['hl_vol_10'] = (df['high'] - df['low']) * df['volume']
    df['vol_sum_10'] = df['volume'].rolling(window=10).sum()
    df['price_diff_sum_10'] = df['hl_vol_10'].rolling(window=10).sum()
    df['cum_weighted_avg_10'] = df['price_diff_sum_10'] / df['vol_sum_10']
    df['open_t-10'] = df['open'].shift(10)
    df['close_t'] = df['close']
    df['long_term_momentum_10'] = df['close_t'] - df['open_t-10']
    df['avg_daily_return_10'] = df['close'].diff().rolling(window=10).sum() / 10
    df['daily_return'] = df['close'].pct_change()
    df['std_dev_10'] = df['daily_return'].rolling(window=10).std()
    df['alpha_factor_10'] = (df['cum_weighted_avg_10'] - df['avg_daily_return_10']) / df['std_dev_10']

    # 5-day metrics
    df['hl_vol_5'] = (df['high'] - df['low']) * df['volume']
    df['vol_sum_5'] = df['volume'].rolling(window=5).sum()
    df['price_diff_sum_5'] = df['hl_vol_5'].rolling(window=5).sum()
    df['cum_weighted_avg_5'] = df['price_diff_sum_5'] / df['vol_sum_5']
    df['open_t-5'] = df['open'].shift(5)
    df['close_t'] = df['close']
    df['short_term_momentum_5'] = df['close_t'] - df['open_t-5']
    df['avg_daily_return_5'] = df['close'].diff().rolling(window=5).sum() / 5
    df['std_dev_5'] = df['daily_return'].rolling(window=5).std()
    df['alpha_factor_5'] = (df['cum_weighted_avg_5'] - df['avg_daily_return_5']) / df['std_dev_5']

    # Combine 10-day and 5-day metrics
    df['combined_alpha_factor'] = 0.6 * df['alpha_factor_10'] + 0.4 * df['alpha_factor_5']

    # Adjust for 5-day average volume
    df['avg_volume_5'] = df['volume'].rolling(window=5).mean()
    df['adjusted_alpha_factor'] = df['combined_alpha_factor'] * df['avg_volume_5']

    # Apply exponential weights
    df['exp_weighted_alpha_factor_10'] = df['alpha_factor_10'] * 0.8
    df['exp_weighted_alpha_factor_5'] = df['alpha_factor_5'] * 0.2
    df['final_alpha_factor'] = 0.6 * df['exp_weighted_alpha_factor_10'] + 0.4 * df['exp_weighted_alpha_factor_5']
    df['final_adjusted_alpha_factor'] = df['final_alpha_factor'] * df['avg_volume_5']

    return df['final_adjusted_alpha_factor']
```
