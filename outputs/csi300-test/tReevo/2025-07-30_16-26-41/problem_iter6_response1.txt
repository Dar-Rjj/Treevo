import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    high_low_diff = (df['high'] - df['low']) * df['volume']
    total_high_low_vol = high_low_diff.rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff = total_high_low_vol / total_volume

    # Determine Long-Term Open to Close Momentum
    open_20_days_ago = df['open'].shift(20)
    close_today = df['close']
    long_term_momentum = close_today - open_20_days_ago

    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff

    # Subtract Long-Term Average Daily Return
    daily_returns = (df['close'] - df['close'].shift(1)).rolling(window=20).sum() / 20
    long_term_cumulative_weighted_avg -= daily_returns

    # Adjust for Volatility
    daily_returns_std = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    volatility_20_day = daily_returns_std.rolling(window=20).std()
    long_term_cumulative_weighted_avg /= volatility_20_day

    # Incorporate Cumulative Intraday Movement
    intraday_price_range = (df['high'] - df['low']).rolling(window=20).sum()
    volume_weighted_intraday_price_diff = high_low_diff.rolling(window=20).sum()
    cumulative_intraday_movement = volume_weighted_intraday_price_diff / intraday_price_range

    # Calculate Cumulative Volume-Weighted Close-Open Difference
    close_open_diff = (df['close'] - df['open']) * df['volume']
    cumulative_close_open_diff = close_open_diff.rolling(window=20).sum()

    # Final Alpha Factor
    final_alpha_factor = long_term_cumulative_weighted_avg + cumulative_intraday_movement + cumulative_close_open_diff

    return final_alpha_factor
