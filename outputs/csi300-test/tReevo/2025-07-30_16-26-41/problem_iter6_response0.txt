import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['HL_Vol_Weighted'] = (df['high'] - df['low']) * df['volume']
    sum_HL_Vol_Weighted = df['HL_Vol_Weighted'].rolling(window=20).sum()
    sum_volume_20 = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    long_term_momentum = df['close'] - df['open'].shift(20)
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_weighted_price_diff = sum_HL_Vol_Weighted / sum_volume_20
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).sum() / 20
    
    # Adjust for Volatility
    daily_returns_percentage = daily_returns / df['close'].shift(1)
    volatility = daily_returns_percentage.rolling(window=20).std()
    
    # Incorporate Volume-Weighted Intraday Momentum
    close_to_close_return = df['close'] - df['close'].shift(1)
    rolling_avg_volume = df['volume'].rolling(window=20).mean()
    intraday_momentum = close_to_close_return * rolling_avg_volume
    
    # Adjust for Trade Frequency
    num_trades_threshold = 1000
    trade_frequency = np.where(df['amount'] < num_trades_threshold, 0.5, 0.8)
    adjusted_long_term_avg_daily_return = long_term_avg_daily_return * trade_frequency
    
    # Combine Long-Term and Intraday Momentum
    combined_momentum = long_term_weighted_price_diff + intraday_momentum - adjusted_long_term_avg_daily_return
    final_alpha_factor = combined_momentum / volatility
    
    return final_alpha_factor.dropna()
