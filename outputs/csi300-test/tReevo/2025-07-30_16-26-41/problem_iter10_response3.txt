import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Weighted Price Difference
    df['HighLowVolume'] = (df['high'] - df['low']) * df['volume']
    sum_high_low_volume = df['HighLowVolume'].rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    short_term_weighted_price_diff = sum_high_low_volume / total_volume

    # Determine Short-Term Open to Close Momentum
    open_lag_5 = df['open'].shift(5)
    close_today = df['close']
    short_term_momentum = close_today - open_lag_5

    # Calculate the Short-Term Cumulative Weighted Average
    short_term_cumulative_avg = short_term_weighted_price_diff

    # Subtract Short-Term Average Daily Return
    daily_returns = df['close'].diff()
    short_term_avg_daily_return = daily_returns.rolling(window=5).sum() / 5
    adjusted_cumulative_avg = short_term_cumulative_avg - short_term_avg_daily_return

    # Adjust for Volatility
    daily_returns_std = daily_returns.rolling(window=5).std()
    final_alpha_factor = adjusted_cumulative_avg / daily_returns_std

    return final_alpha_factor.dropna()
