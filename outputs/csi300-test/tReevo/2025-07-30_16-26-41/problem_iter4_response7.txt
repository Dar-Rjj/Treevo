import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['HLV'] = (df['high'] - df['low']) * df['volume']
    long_term_weighted_price_diff = df['HLV'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    
    # Determine Long-Term Open to Close Momentum
    open_t_minus_20 = df['open'].shift(20)
    close_t = df['close']
    long_term_open_close_momentum = close_t - open_t_minus_20
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = (long_term_weighted_price_diff / total_volume).shift(1)  # Shift to avoid look-ahead bias
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'].pct_change()
    long_term_avg_daily_return = daily_returns.rolling(window=20).mean()
    
    # Adjust for Volatility
    daily_returns = df['close'].pct_change()
    vol_20_day = daily_returns.rolling(window=20).std()
    
    # Incorporate Cumulative Intraday Movement
    intraday_range = (df['high'] - df['low']).rolling(window=20).sum()
    volume_weighted_close_open_diff = ((df['close'] - df['open']) * df['volume']).rolling(window=20).sum()
    cumulative_intraday_movement = volume_weighted_close_open_diff / intraday_range
    
    # Final Alpha Factor
    alpha_factor = (
        (long_term_cumulative_weighted_avg - long_term_avg_daily_return) / vol_20_day +
        cumulative_intraday_movement
    )
    
    return alpha_factor
