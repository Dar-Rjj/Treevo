import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['high_low_vol'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff = df['high_low_vol'].rolling(window=20).sum()
    total_volume = df['volume'].rolling(window=20).sum()
    long_term_weighted_price_diff = weighted_price_diff / total_volume
    
    # Determine Long-Term Open to Close Momentum
    df['open_lag_20'] = df['open'].shift(20)
    long_term_momentum = df['close'] - df['open_lag_20']
    
    # Calculate the Long-Term Cumulative Weighted Average
    long_term_cumulative_weighted_avg = long_term_weighted_price_diff
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    long_term_avg_daily_return = daily_returns.rolling(window=20).sum() / 20
    adjusted_long_term_cumulative = long_term_cumulative_weighted_avg - long_term_avg_daily_return
    
    # Adjust for Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    twenty_day_std = daily_returns.rolling(window=20).std()
    volatility_adjusted_factor = adjusted_long_term_cumulative / twenty_day_std
    
    # Incorporate Cumulative Intraday Movement
    intraday_price_range = (df['high'] - df['low']).rolling(window=20).sum()
    vol_weighted_close_open_diff = ((df['close'] - df['open']) * df['volume']).rolling(window=20).sum()
    cum_intraday_movement = vol_weighted_close_open_diff / intraday_price_range
    
    # Final Alpha Factor
    final_alpha_factor = volatility_adjusted_factor + cum_intraday_movement
    
    return final_alpha_factor
