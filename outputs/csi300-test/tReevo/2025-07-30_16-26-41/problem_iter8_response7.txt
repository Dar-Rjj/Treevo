```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Volume-Weighted Price Difference
    df['price_diff_vol_weighted'] = (df['high'] - df['low']) * df['volume']
    total_price_diff_vol_weighted = df['price_diff_vol_weighted'].rolling(window=5).sum()
    total_volume = df['volume'].rolling(window=5).sum()
    short_term_vol_weighted_price_diff = total_price_diff_vol_weighted / total_volume
    
    # Determine Short-Term Open to Close Momentum
    short_term_open_to_close_momentum = df['close'] - df['open'].shift(5)
    
    # Calculate the Short-Term Cumulative Weighted Average
    short_term_cumulative_weighted_avg = short_term_vol_weighted_price_diff + short_term_open_to_close_momentum
    
    # Subtract Short-Term Average Daily Return
    daily_returns = df['close'] - df['close'].shift(1)
    short_term_average_daily_return = daily_returns.rolling(window=5).mean()
    adjusted_cumulative_weighted_avg = short_term_cumulative_weighted_avg - short_term_average_daily_return
    
    # Adjust for Volatility
    daily_returns_pct = daily_returns / df['close'].shift(1)
    five_day_volatility = daily_returns_pct.rolling(window=5).std()
    final_alpha_factor = adjusted_cumulative_weighted_avg / five_day_volatility
    
    return final_alpha_factor.dropna()
```
