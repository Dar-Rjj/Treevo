```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Momentum
    df['close_to_close_return'] = df['close'].pct_change()
    df['rolling_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['momentum'] = df['close_to_close_return'] * df['rolling_avg_volume']
    
    # Incorporate Short-Term Reversal
    df['high_low_range'] = df['high'] - df['low']
    def fraction(trades):
        if trades < 1000:  # Example threshold, adjust as needed
            return 0.5
        else:
            return 0.8
    df['fraction'] = df['amount'].apply(fraction)  # Assuming 'amount' is the number of trades here
    df['reversal_adjustment'] = -df['high_low_range'] * df['fraction']
    
    # Combine Momentum and Reversal
    df['alpha_factor'] = df['momentum'] + df['reversal_adjustment']
    
    return df['alpha_factor'].dropna()

# alpha_factor_series = heuristics_v2(prices_df)
```

Note: Ensure that 'amount' in the DataFrame represents the number of trades. If it does not, you may need to use a different column or adjust the input data accordingly. The threshold for the number of trades (set to 1000 in the example) should be adjusted based on the specific dataset and market conditions.
