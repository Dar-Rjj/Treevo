```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Long-Term Weighted Price Difference
    df['price_diff_vol'] = (df['high'] - df['low']) * df['volume']
    weighted_price_diff = df['price_diff_vol'].rolling(window=10).sum()
    total_volume = df['volume'].rolling(window=10).sum()
    
    # Determine Long-Term Open to Close Momentum
    open_10_days_ago = df['open'].shift(10)
    close_today = df['close']
    open_to_close_momentum = close_today - open_10_days_ago
    
    # Calculate the Long-Term Cumulative Weighted Average
    cumulative_weighted_average = (weighted_price_diff / total_volume)
    
    # Subtract Long-Term Average Daily Return
    daily_returns = df['close'].pct_change()
    long_term_avg_daily_return = daily_returns.rolling(window=10).mean()
    
    # Adjust for Volatility
    std_dev_10_day = daily_returns.rolling(window=10).std()
    
    # Final Alpha Factor
    alpha_factor = (cumulative_weighted_average + open_to_close_momentum - long_term_avg_daily_return) / std_dev_10_day
    
    return alpha_factor.dropna()
```
