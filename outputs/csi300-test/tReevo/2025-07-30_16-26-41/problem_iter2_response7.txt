```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Intraday Price Range
    intraday_range = df['high'] - df['low']

    # Calculate Volume-Weighted Close-Open Difference
    vol_weighted_diff = (df['close'] - df['open']) * df['volume']

    # Determine Intraday Trend Strength
    trend_strength = vol_weighted_diff / intraday_range

    # Consider Direction of Intraday Movement
    trend_strength = trend_strength.where(df['close'] > df['open'], -trend_strength)
    trend_strength = trend_strength.where(df['close'] != df['open'], 0)

    # Incorporate Cumulative Volume-Weighted Momentum
    cumulative_vol_weighted_diff = vol_weighted_diff.rolling(window=n).sum()
    cumulative_intraday_range = intraday_range.rolling(window=n).sum()

    cumulative_trend_strength = cumulative_vol_weighted_diff / cumulative_intraday_range

    # Consider Direction of Cumulative Intraday Movement
    cumulative_trend_strength = cumulative_trend_strength.where(
        cumulative_vol_weighted_diff > 0, -cumulative_trend_strength
    )
    cumulative_trend_strength = cumulative_trend_strength.where(
        cumulative_vol_weighted_diff != 0, 0
    )

    return cumulative_trend_strength
```
