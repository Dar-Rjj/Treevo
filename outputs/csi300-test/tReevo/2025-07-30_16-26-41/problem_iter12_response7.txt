```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Short-Term Weighted Price Difference (5-day)
    df['High_Low_Volume_5'] = (df['high'] - df['low']) * df['volume']
    short_term_weighted_price_diff_5 = df['High_Low_Volume_5'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    short_term_cumulative_weighted_avg_5 = short_term_weighted_price_diff_5 / total_volume_5
    
    # Determine Short-Term Open to Close Momentum (5-day)
    short_term_open_close_momentum_5 = df['close'] - df['open'].shift(5)
    
    # Subtract Short-Term Average Daily Return (5-day)
    daily_returns_5 = df['close'] - df['close'].shift(1)
    short_term_avg_daily_return_5 = daily_returns_5.rolling(window=5).sum() / 5
    
    # Adjust for Volatility (5-day)
    daily_returns_5_percentage = daily_returns_5 / df['close'].shift(1)
    short_term_volatility_5 = daily_returns_5_percentage.rolling(window=5).std()
    
    # Final 5-Day Alpha Factor
    alpha_factor_5 = (short_term_cumulative_weighted_avg_5 - short_term_avg_daily_return_5) / short_term_volatility_5
    
    # Recalculate the above metrics using a 3-day window
    df['High_Low_Volume_3'] = (df['high'] - df['low']) * df['volume']
    short_term_weighted_price_diff_3 = df['High_Low_Volume_3'].rolling(window=3).sum()
    total_volume_3 = df['volume'].rolling(window=3).sum()
    short_term_cumulative_weighted_avg_3 = short_term_weighted_price_diff_3 / total_volume_3
    
    # Determine Very Short-Term Open to Close Momentum (3-day)
    short_term_open_close_momentum_3 = df['close'] - df['open'].shift(3)
    
    # Subtract Very Short-Term Average Daily Return (3-day)
    daily_returns_3 = df['close'] - df['close'].shift(1)
    short_term_avg_daily_return_3 = daily_returns_3.rolling(window=3).sum() / 3
    
    # Adjust for Volatility (3-day)
    daily_returns_3_percentage = daily_returns_3 / df['close'].shift(1)
    short_term_volatility_3 = daily_returns_3_percentage.rolling(window=3).std()
    
    # Final 3-Day Alpha Factor
    alpha_factor_3 = (short_term_cumulative_weighted_avg_3 - short_term_avg_daily_return_3) / short_term_volatility_3
    
    # Combine 5-day and 3-day metrics
    combined_alpha_factor = 0.7 * alpha_factor_5 + 0.3 * alpha_factor_3
    
    # Integrate Trading Volume Effectively
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    final_alpha_factor = combined_alpha_factor * avg_volume_5
    
    return final_alpha_factor
```
