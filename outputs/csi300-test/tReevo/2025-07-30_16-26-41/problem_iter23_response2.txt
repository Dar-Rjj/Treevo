import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Dynamic, Volume-Weighted Momentum
    df['daily_return'] = (df['close'] - df['open']) / df['open']
    df['momentum'] = (df['daily_return'] * df['volume']).rolling(window=10).sum()
    
    # Adaptive Volatility
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    df['adjusted_momentum'] = df['momentum'] / df['volatility']
    
    # Sector-Specific Exponential Moving Average (EMA)
    # Assuming 'sector' is a column in the DataFrame and we have a dictionary of spans for each sector
    sector_spans = {
        'Technology': 5,
        'Healthcare': 7,
        'Finance': 9,
        'Consumer': 11,
        'Industrial': 13,
        'Other': 15
    }
    def calculate_ema_sector(group):
        sector = group.name
        span = sector_spans.get(sector, sector_spans['Other'])
        return group['close'].ewm(span=span).mean()
    
    df['ema_sector'] = df.groupby('sector').apply(calculate_ema_sector).reset_index(level=0, drop=True)
    df['normalized_momentum'] = df['adjusted_momentum'] / df['ema_sector']
    
    # Liquidity Adjustment
    df['avg_volume'] = df['volume'].rolling(window=30).mean()
    df['liquidity_adjusted_momentum'] = df['normalized_momentum'] * df['avg_volume']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['liquidity_adjusted_momentum'] * 1.0
    
    # Return the alpha factor as a Series
    return df['alpha_factor']
