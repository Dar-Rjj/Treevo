import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given market features.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with (date) as index, and columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series indexed by (date) representing the factor values.
    """
    
    # Calculate the percentage change in close price
    df['pct_change_close'] = df['close'].pct_change()
    
    # Calculate the average true range over 14 days
    df['true_range'] = df[['high', 'low']].apply(lambda x: x.max() - x.min(), axis=1)
    df['avg_true_range'] = df['true_range'].rolling(window=14).mean()
    
    # Calculate the money flow multiplier
    df['mf_multiplier'] = (df['close'] - df['low']) - (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Calculate the money flow volume
    df['mf_volume'] = df['mf_multiplier'] * df['volume']
    
    # Calculate the money flow ratio
    df['mf_ratio'] = df['mf_volume'].rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Combine the factors to generate the final alpha factor
    df['alpha_factor'] = df['pct_change_close'] * df['avg_true_range'] * df['mf_ratio']
    
    return df['alpha_factor']
