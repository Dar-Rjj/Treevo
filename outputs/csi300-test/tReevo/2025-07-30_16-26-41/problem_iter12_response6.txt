import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-Day Weighted Price Difference
    df['Weighted_Price_Diff_10'] = (df['high'] - df['low']) * df['volume']
    sum_weighted_diff_10 = df['Weighted_Price_Diff_10'].rolling(window=10).sum()
    total_volume_10 = df['volume'].rolling(window=10).sum()
    
    # 10-Day Open to Close Momentum
    open_t_minus_10 = df['open'].shift(10)
    close_t = df['close']
    momentum_10 = close_t - open_t_minus_10
    
    # 10-Day Cumulative Weighted Average
    weighted_avg_10 = sum_weighted_diff_10 / total_volume_10
    
    # 10-Day Average Daily Return
    daily_return_10 = (df['close'] - df['close'].shift(1)).rolling(window=10).sum() / 10
    
    # 10-Day Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    std_10 = daily_returns.rolling(window=10).std()
    
    # Final 10-Day Alpha Factor
    alpha_10 = (weighted_avg_10 - daily_return_10) / std_10

    # 5-Day Weighted Price Difference
    sum_weighted_diff_5 = df['Weighted_Price_Diff_10'].rolling(window=5).sum()
    total_volume_5 = df['volume'].rolling(window=5).sum()
    
    # 5-Day Open to Close Momentum
    open_t_minus_5 = df['open'].shift(5)
    momentum_5 = close_t - open_t_minus_5
    
    # 5-Day Cumulative Weighted Average
    weighted_avg_5 = sum_weighted_diff_5 / total_volume_5
    
    # 5-Day Average Daily Return
    daily_return_5 = (df['close'] - df['close'].shift(1)).rolling(window=5).sum() / 5
    
    # 5-Day Volatility
    std_5 = daily_returns.rolling(window=5).std()
    
    # Final 5-Day Alpha Factor
    alpha_5 = (weighted_avg_5 - daily_return_5) / std_5

    # Combine 10-Day and 5-Day Metrics
    combined_alpha = 0.6 * alpha_10 + 0.4 * alpha_5

    # 5-Day Average Volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    
    # Adjust the Combined Alpha Factor by 5-Day Average Volume
    final_alpha = combined_alpha * avg_volume_5
    
    return final_alpha.dropna()
