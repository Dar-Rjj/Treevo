import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-day calculations
    df['High_Low_Vol_10'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_10'] = df['volume'].rolling(window=10).sum()
    df['Weighted_Price_Diff_10'] = df['High_Low_Vol_10'].rolling(window=10).sum() / df['Total_Volume_10']
    
    df['Open_Close_Mom_10'] = df['close'] - df['open'].shift(10)
    df['Daily_Return_10'] = df['close'] - df['close'].shift(1)
    df['Avg_Daily_Return_10'] = df['Daily_Return_10'].rolling(window=10).sum() / 10
    
    df['Volatility_10'] = df['Daily_Return_10'].rolling(window=10).std()
    df['Alpha_Factor_10'] = (df['Weighted_Price_Diff_10'] - df['Avg_Daily_Return_10']) / df['Volatility_10']

    # 5-day calculations
    df['High_Low_Vol_5'] = (df['high'] - df['low']) * df['volume']
    df['Total_Volume_5'] = df['volume'].rolling(window=5).sum()
    df['Weighted_Price_Diff_5'] = df['High_Low_Vol_5'].rolling(window=5).sum() / df['Total_Volume_5']
    
    df['Open_Close_Mom_5'] = df['close'] - df['open'].shift(5)
    df['Daily_Return_5'] = df['close'] - df['close'].shift(1)
    df['Avg_Daily_Return_5'] = df['Daily_Return_5'].rolling(window=5).sum() / 5
    
    df['Volatility_5'] = df['Daily_Return_5'].rolling(window=5).std()
    df['Alpha_Factor_5'] = (df['Weighted_Price_Diff_5'] - df['Avg_Daily_Return_5']) / df['Volatility_5']

    # Combine 10-day and 5-day metrics with dynamic weights
    df['Combined_Alpha_Factor'] = 0.6 * df['Alpha_Factor_10'] + 0.4 * df['Alpha_Factor_5']

    # Adjust for average volume
    df['Average_Volume_5'] = df['volume'].rolling(window=5).mean()
    df['Adjusted_Combined_Alpha_Factor'] = df['Combined_Alpha_Factor'] * df['Average_Volume_5']

    # Apply exponential smoothing
    df['Exponential_10'] = 0.8 * df['Alpha_Factor_10'] + 0.2 * df['Alpha_Factor_5']
    df['Final_Alpha_Factor'] = df['Exponential_10'] * df['Average_Volume_5']

    return df['Final_Alpha_Factor'].dropna()
