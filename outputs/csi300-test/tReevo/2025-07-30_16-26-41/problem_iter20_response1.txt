```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate EMAs
    df['ema_short_close'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_long_close'] = df['close'].ewm(span=20, adjust=False).mean()
    
    # Volume-Weighted Metrics
    total_volume_10 = df['volume'].rolling(window=10).sum()
    df['vwc_price'] = (df['close'] * df['volume']).rolling(window=10).sum() / total_volume_10
    df['vwhl_diff'] = ((df['high'] - df['low']) * df['volume']).rolling(window=10).sum()
    
    # Momentum
    df['momentum_close'] = df['close'] - df['close'].shift(10)
    
    # Exponential Smoothing
    df['smooth_ema_short_close'] = df['ema_short_close'].ewm(span=5, adjust=False).mean()
    df['smooth_ema_long_close'] = df['ema_long_close'].ewm(span=20, adjust=False).mean()
    
    # 5-Day Average Volume
    df['avg_vol_5'] = df['volume'].rolling(window=5).mean()
    
    # High-Low Range
    df['hl_range_10'] = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    df['hl_range_5'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Combine Alpha Factors
    df['alpha_factor_10_day'] = 0.6 * df['smooth_ema_long_close'] + 0.4 * df['vwc_price']
    df['alpha_factor_5_day'] = 0.6 * df['smooth_ema_short_close'] + 0.4 * df['vwhl_diff']
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['alpha_factor_10_day'] * 0.6 + df['alpha_factor_5_day'] * 0.4) * df['avg_vol_5'] / df['hl_range_10'] / df['hl_range_5']
    
    return df['final_alpha_factor']
```
