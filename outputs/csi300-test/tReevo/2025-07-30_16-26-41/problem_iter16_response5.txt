```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # 10-Day Metrics
    df['10_day_weighted_price_diff'] = (df['high'] - df['low']) * df['volume']
    df['10_day_weighted_price_diff'] = df['10_day_weighted_price_diff'].rolling(window=10).sum()
    df['10_day_total_volume'] = df['volume'].rolling(window=10).sum()
    df['10_day_cumulative_weighted_avg'] = df['10_day_weighted_price_diff'] / df['10_day_total_volume']
    
    # 10-Day Open to Close Momentum
    df['10_day_momentum'] = df['close'] - df['open'].shift(10)
    
    # 10-Day Average Daily Return
    df['10_day_avg_daily_return'] = df['close'].diff().rolling(window=10).sum() / 10
    
    # 10-Day Dynamic Volatility
    daily_returns = df['close'].pct_change()
    df['10_day_ew_std'] = daily_returns.ewm(span=10, adjust=False).std()
    
    # Final 10-Day Alpha Factor
    df['10_day_alpha_factor'] = (df['10_day_cumulative_weighted_avg'] - df['10_day_avg_daily_return']) / df['10_day_ew_std']
    
    # 5-Day Metrics
    df['5_day_weighted_price_diff'] = (df['high'] - df['low']) * df['volume']
    df['5_day_weighted_price_diff'] = df['5_day_weighted_price_diff'].rolling(window=5).sum()
    df['5_day_total_volume'] = df['volume'].rolling(window=5).sum()
    df['5_day_cumulative_weighted_avg'] = df['5_day_weighted_price_diff'] / df['5_day_total_volume']
    
    # 5-Day Open to Close Momentum
    df['5_day_momentum'] = df['close'] - df['open'].shift(5)
    
    # 5-Day Average Daily Return
    df['5_day_avg_daily_return'] = df['close'].diff().rolling(window=5).sum() / 5
    
    # 5-Day Dynamic Volatility
    df['5_day_ew_std'] = daily_returns.ewm(span=5, adjust=False).std()
    
    # Final 5-Day Alpha Factor
    df['5_day_alpha_factor'] = (df['5_day_cumulative_weighted_avg'] - df['5_day_avg_daily_return']) / df['5_day_ew_std']
    
    # Combine 10-Day and 5-Day Metrics
    df['combined_alpha_factor'] = 0.6 * df['10_day_alpha_factor'] + 0.4 * df['5_day_alpha_factor']
    
    # Integrate Trading Volume Effectively
    df['5_day_avg_volume'] = df['volume'].rolling(window=5).mean()
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    df['normalized_5_day_avg_volume'] = df['5_day_avg_volume'] / df['10_day_avg_volume']
    
    # Adjust the Combined Alpha Factor
    df['final_alpha_factor'] = df['combined_alpha_factor'] * df['normalized_5_day_avg_volume']
    
    return df['final_alpha_factor']
```
