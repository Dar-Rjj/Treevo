```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Change Factors
    df['Daily_Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['2Day_Return'] = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    df['5Day_Return'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Evaluate Momentum Reversal
    df['Extreme_Movement'] = (df['high'] - df['low']) / df['low']
    df['Momentum_Reversal_Signal'] = (df['high'] - df['close']) / (df['high'] - df['low'])
    
    # Measure Historical Volatility
    df['20Day_Volatility'] = df['Daily_Return'].rolling(window=20).std()
    df['50Day_Volatility'] = df['Daily_Return'].rolling(window=50).std()
    
    # Dynamic Volatility Adjustment
    df['EMA_20Day_Volatility'] = df['20Day_Volatility'].ewm(alpha=0.2).mean()
    df['EMA_50Day_Volatility'] = df['50Day_Volatility'].ewm(alpha=0.1).mean()
    
    # Calculate Volume Change Factors
    df['Daily_Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['5Day_Volume_Change'] = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)
    
    # Integrate Volume with Price
    df['Volume_Weighted_Price'] = (df['close'] * df['volume'] + df['close'].shift(1) * df['volume'].shift(1)) / (df['volume'] + df['volume'].shift(1))
    
    # Consider Sector Performance
    # Assuming 'sector_index' is a column in the DataFrame representing the sector index
    df['Sector_Index_Return'] = (df['sector_index'] - df['sector_index'].shift(1)) / df['sector_index'].shift(1)
    df['Sector_Outperformance'] = df['Daily_Return'] - df['Sector_Index_Return']
    
    # Simplify Calculations and Adjust for Liquidity
    df['SMA_20'] = df['close'].rolling(window=20).mean()
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Liquidity_Factor'] = df['volume'] / (df['high'] - df['low'])
    
    # Balance Factors to Avoid Overfitting
    df['Momentum_Volatility_Ratio'] = df['5Day_Return'] / df['20Day_Volatility']
    df['Composite_Alpha_Factor'] = (df['Momentum_Volatility_Ratio'] * df['Volume_Weighted_Price']) + (df['Sector_Outperformance'] * df['Liquidity_Factor'])
    
    return df['Composite_Alpha_Factor'].dropna()
```
