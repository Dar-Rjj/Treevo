import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    df['range_over_open'] = df['daily_range'] / df['open']

    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_sma_5'] = df['open_to_close_return'].rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / avg_volume_20

    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['price_volume_trend_30'] = df['price_volume_trend'].rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.6 * df['range_over_open'] + 0.4 * df['open_to_close_sma_5']
    df['exponential_smoothed_combined'] = df['combined_momentum_volatility'].ewm(alpha=0.2).mean()

    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.7 * df['price_volume_trend_30'] + 0.3 * df['volume_adjusted_intraday_movement']
    df['exponential_smoothed_volume_sensitive'] = df['volume_sensitive_momentum'].ewm(alpha=0.2).mean()

    # Final Alpha Factor
    df['final_alpha_factor'] = 0.6 * df['exponential_smoothed_combined'] + 0.4 * df['exponential_smoothed_volume_sensitive']
    final_alpha_factor = df['final_alpha_factor'].ewm(alpha=0.2).mean()

    return final_alpha_factor
