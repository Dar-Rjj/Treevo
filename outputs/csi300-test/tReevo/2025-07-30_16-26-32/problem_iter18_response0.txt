import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate the daily range (High - Low) and normalize by the opening price
    df['DailyRange'] = (df['high'] - df['low']) / df['open']
    
    # Calculate the open-to-close return
    df['OpenToCloseReturn'] = (df['close'] - df['open']) / df['open']
    
    # Smooth with 5-day exponential moving average (EMA)
    df['SmoothedOpenToCloseReturn'] = df['OpenToCloseReturn'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['VolumeWeightedOpenToCloseReturn'] = df['OpenToCloseReturn'] * df['volume']
    df['SmoothedVolumeWeightedOpenToCloseReturn'] = df['VolumeWeightedOpenToCloseReturn'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['DailyPriceChange'] = df['close'] - df['close'].shift(1)
    df['PriceVolumeTrend'] = df['DailyPriceChange'] * df['volume']
    df['PriceVolumeTrend30Day'] = df['PriceVolumeTrend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['IntradayMovement'] = df['close'] - df['open']
    df['20DayAvgVolume'] = df['volume'].rolling(window=20).mean()
    df['VolumeAdjustedIntradayMovement'] = df['IntradayMovement'] / df['20DayAvgVolume']
    
    # Combined Momentum and Volatility Factor
    combined_factor = 0.4 * df['DailyRange'] + 0.3 * df['SmoothedOpenToCloseReturn'] + 0.3 * df['SmoothedVolumeWeightedOpenToCloseReturn']
    df['CombinedMomentumVolatility'] = combined_factor.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = 0.4 * df['PriceVolumeTrend30Day'] + 0.3 * df['SmoothedVolumeWeightedOpenToCloseReturn'] + 0.3 * df['VolumeAdjustedIntradayMovement']
    df['VolumeSensitiveMomentum'] = volume_sensitive_factor.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * df['CombinedMomentumVolatility'] + 0.4 * df['VolumeSensitiveMomentum']
    df['FinalAlphaFactor'] = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return df['FinalAlphaFactor']
