```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    range_over_open = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma5 = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_vol_20

    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.6 * range_over_open + 0.4 * open_to_close_sma5
    combined_mom_vol_smoothed = combined_mom_vol.ewm(span=5, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * price_volume_trend + 0.3 * volume_adjusted_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    final_alpha = 0.5 * combined_mom_vol_smoothed + 0.5 * volume_sensitive_momentum_smoothed
    final_alpha_smoothed = final_alpha.ewm(span=5, adjust=False).mean()

    return final_alpha_smoothed
```
