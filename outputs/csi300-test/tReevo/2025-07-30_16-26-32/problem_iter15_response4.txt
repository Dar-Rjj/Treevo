```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.rolling(window=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = open_to_close_return * df['volume']
    volume_weighted_momentum = volume_weighted_return.rolling(window=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_factor = (0.4 * daily_range + 0.3 * open_to_close_momentum + 0.3 * volume_weighted_momentum)
    combined_factor_smoothed = combined_factor.ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = (0.7 * price_volume_trend + 0.3 * volume_weighted_momentum)
    volume_sensitive_factor_smoothed = volume_sensitive_factor.ewm(alpha=0.2, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (0.6 * combined_factor_smoothed + 0.4 * volume_sensitive_factor_smoothed)
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean()
    
    return final_alpha_factor
```
