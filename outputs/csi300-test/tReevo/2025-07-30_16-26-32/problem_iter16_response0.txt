```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close_return = open_to_close_return.rolling(window=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    smooth_volume_weighted_return = volume_weighted_return.rolling(window=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = price_change * df['volume']
    aggregate_price_volume_trend = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * daily_range + 0.3 * smooth_open_to_close_return + 0.3 * smooth_volume_weighted_return
    exp_smooth_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.2).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * aggregate_price_volume_trend + 0.3 * smooth_volume_weighted_return
    exp_smooth_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.2).mean()
    
    # New Momentum-Based Factor
    ten_day_return = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    smooth_ten_day_return = ten_day_return.rolling(window=5).mean()
    
    # New Volume-Based Factor
    ten_day_avg_volume = df['volume'].rolling(window=10).mean()
    fifty_day_avg_volume = df['volume'].rolling(window=50).mean()
    normalized_volume = ten_day_avg_volume / fifty_day_avg_volume
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * exp_smooth_combined_momentum_volatility + 0.3 * exp_smooth_volume_sensitive_momentum + 0.1 * smooth_ten_day_return + 0.1 * normalized_volume
    exp_smooth_final_alpha_factor = final_alpha_factor.ewm(alpha=0.2).mean()
    
    return exp_smooth_final_alpha_factor
```
