```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Price Momentum
    momentum = df['close'].shift(1) - df['close'].shift(21)
    
    # Calculate 7-day Exponential Moving Average of Volume
    ema_volume = df['volume'].ewm(span=7, adjust=False).mean()
    
    # Normalize Momentum by EMA
    adjusted_momentum = momentum / ema_volume
    
    # Apply Signum Function to Adjusted Momentum
    factor = np.sign(adjusted_momentum)
    
    return pd.Series(factor, index=df.index, name='heuristics_v2')
```
