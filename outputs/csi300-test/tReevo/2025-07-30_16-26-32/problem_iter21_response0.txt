```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smoothed_momentum = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    smoothed_volume_weighted_return = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    price_volume_product = price_change * df['volume']
    pvt_indicator = price_volume_product.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_20_day_volume
    
    # Combined Momentum and Volatility Factor
    adaptive_weights_momentum_volatility = {
        'intraday_high_to_low': 0.4,
        'open_to_close_momentum': 0.3,
        'volume_weighted_momentum': 0.3
    }
    combined_momentum_volatility = (
        adaptive_weights_momentum_volatility['intraday_high_to_low'] * normalized_range +
        adaptive_weights_momentum_volatility['open_to_close_momentum'] * smoothed_momentum +
        adaptive_weights_momentum_volatility['volume_weighted_momentum'] * smoothed_volume_weighted_return
    )
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    adaptive_weights_volume_sensitive = {
        'price_volume_trend': 0.4,
        'volume_weighted_open_to_close': 0.3,
        'volume_adjusted_intraday': 0.3
    }
    volume_sensitive_momentum = (
        adaptive_weights_volume_sensitive['price_volume_trend'] * pvt_indicator +
        adaptive_weights_volume_sensitive['volume_weighted_open_to_close'] * smoothed_volume_weighted_return +
        adaptive_weights_volume_sensitive['volume_adjusted_intraday'] * volume_adjusted_intraday
    )
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    adaptive_weights_final = {
        'combined_momentum_volatility': 0.5,
        'volume_sensitive_momentum': 0.5
    }
    final_alpha_factor = (
        adaptive_weights_final['combined_momentum_volatility'] * smoothed_combined_momentum_volatility +
        adaptive_weights_final['volume_sensitive_momentum'] * smoothed_volume_sensitive_momentum
    )
    final_smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_smoothed_alpha_factor
```
