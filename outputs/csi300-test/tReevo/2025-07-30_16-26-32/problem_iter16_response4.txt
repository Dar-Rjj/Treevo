```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_open_to_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['intraday_movement_vol_adj'] = (df['close'] - df['open']) / avg_volume_20
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume Weighted Momentum
    df['vol_weighted_open_to_close'] = df['open_to_close_return'] * df['volume']
    df['sma_vol_weighted_open_to_close'] = df['vol_weighted_open_to_close'].rolling(window=5).mean()
    
    # Volume-Weighted Price-Volume Trend
    df['vol_weighted_pvt'] = df['price_change'] * df['volume']
    df['vwpvt_60'] = df['vol_weighted_pvt'].rolling(window=60).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * df['daily_range'] + 0.3 * df['sma_open_to_close'] + 0.3 * df['sma_vol_weighted_open_to_close']
    df['combined_momentum_volatility'] = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * df['pvt_30'] + 0.3 * df['vwpvt_60'] + 0.2 * df['intraday_movement_vol_adj']
    df['volume_sensitive_momentum'] = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * df['combined_momentum_volatility'] + 0.4 * df['volume_sensitive_momentum']
    df['final_alpha_factor'] = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor']
```
