import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    sma_5_open_to_close = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * sma_5_open_to_close
    exp_smoothing_combined = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * price_volume_trend + 0.3 * volume_adjusted_intraday
    exp_smoothing_volume_sensitive = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = 0.5 * exp_smoothing_combined + 0.5 * exp_smoothing_volume_sensitive
    final_exp_smoothing = final_alpha.ewm(alpha=0.1, adjust=False).mean()
    
    return final_exp_smoothing
