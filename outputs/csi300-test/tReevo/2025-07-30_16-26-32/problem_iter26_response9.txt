```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    df['range_adjusted_close'] = df['daily_range'] / df['close'].shift(1)
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_5'] = df['open_to_close_return'].ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_return_5'] = df['volume_weighted_return'].ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['daily_price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday'] = df['intraday_movement'] / df['sma_volume_20']
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.4 * df['range_adjusted_close'] + \
                                         0.3 * df['ema_open_to_close_5'] + \
                                         0.3 * df['ema_volume_weighted_return_5']
    df['smoothed_combined_momentum_volatility'] = df['combined_momentum_volatility'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.4 * df['pvt_30'] + \
                                      0.3 * df['ema_volume_weighted_return_5'] + \
                                      0.3 * df['volume_adjusted_intraday']
    df['smoothed_volume_sensitive_momentum'] = df['volume_sensitive_momentum'].ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.5 * df['smoothed_combined_momentum_volatility'] + \
                               0.5 * df['smoothed_volume_sensitive_momentum']
    df['final_smoothed_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_smoothed_alpha_factor']
```
