```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_5_open_to_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    avg_volume = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_movement'] = (df['close'] - df['open']) / avg_volume
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * df['daily_range'] + 0.4 * df['sma_5_open_to_close']
    df['combined_momentum_volatility_smoothed'] = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * df['pvt_30'] + 0.3 * df['volume_adjusted_movement']
    df['volume_sensitive_momentum_smoothed'] = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * df['combined_momentum_volatility_smoothed'] + 0.4 * df['volume_sensitive_momentum_smoothed']
    df['final_alpha_factor'] = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor']
```
