```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_5d'] = df['open_to_close_return'].ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_open_to_close_5d'] = df['volume_weighted_open_to_close_return'].ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30d'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['sma_volume_20d']
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.4 * df['daily_range'] + 0.3 * df['ema_open_to_close_5d'] + 0.3 * df['ema_volume_weighted_open_to_close_5d']
    df['adaptive_smoothing_cm_v'] = df['combined_momentum_volatility'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.4 * df['pvt_30d'] + 0.3 * df['ema_volume_weighted_open_to_close_5d'] + 0.3 * df['volume_adjusted_intraday_movement']
    df['adaptive_smoothing_vsm'] = df['volume_sensitive_momentum'].ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.5 * df['adaptive_smoothing_cm_v'] + 0.5 * df['adaptive_smoothing_vsm']
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor_smoothed']
```
