```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) * df['volume']
    smooth_volume_weighted_return = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    pvt = daily_price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_20_day_volume
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.3 * daily_range + 0.4 * smooth_open_to_close + 0.3 * smooth_volume_weighted_return
    combined_momentum_volatility_smooth = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * pvt + 0.3 * smooth_volume_weighted_return + 0.3 * volume_adjusted_movement
    volume_sensitive_momentum_smooth = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility_smooth + 0.5 * volume_sensitive_momentum_smooth
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_alpha_factor
```
