```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open'] - 1)
    smoothed_open_to_close_momentum = open_to_close_return.rolling(window=10).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    normalized_intraday_movement = intraday_movement / df['volume'].rolling(window=20).mean()

    # Price-Volume Trend Indicator
    price_volume_trend = (df['close'].diff() * df['volume']).rolling(window=60).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * intraday_high_low_range + 0.5 * smoothed_open_to_close_momentum
    combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.15).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.6 * price_volume_trend + 0.4 * normalized_intraday_movement
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.15).mean()

    # Market Trend Indicator
    long_term_ma = df['close'].rolling(window=200).mean()
    market_trend_indicator = (df['close'] > long_term_ma).astype(int)

    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility + 0.3 * volume_sensitive_momentum + 0.2 * market_trend_indicator
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.15).mean()

    return final_alpha_factor
```
