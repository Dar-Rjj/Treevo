```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    range_over_open = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma5 = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume

    # Price-Volume Trend Indicator
    pvt = (df['close'] - df['close'].shift(1)) * df['volume']
    pvt_30 = pvt.rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * range_over_open + 0.4 * open_to_close_sma5
    combined_momentum_volatility_exp = combined_momentum_volatility.ewm(span=10, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * pvt_30 + 0.3 * volume_adjusted_movement
    volume_sensitive_momentum_exp = volume_sensitive_momentum.ewm(span=10, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_momentum_volatility_exp + 0.4 * volume_sensitive_momentum_exp
    final_alpha_factor_exp = final_alpha_factor.ewm(span=10, adjust=False).mean()

    return final_alpha_factor_exp
```
