```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_momentum'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_oc_return'] = df['open_to_close_return'] * df['volume']
    df['volume_weighted_momentum'] = df['volume_weighted_oc_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pv_trend_30d'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_20d_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_move'] = df['intraday_movement'] / df['avg_20d_volume']
    
    # Combined Momentum and Volatility Factor
    df['combined_mom_vol'] = 0.4 * df['daily_range'] + 0.3 * df['open_to_close_momentum'] + 0.3 * df['volume_weighted_momentum']
    df['smoothed_combined_mom_vol'] = df['combined_mom_vol'].ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum_factor'] = 0.4 * df['pv_trend_30d'] + 0.3 * df['volume_weighted_momentum'] + 0.3 * df['volume_adjusted_intraday_move']
    df['smoothed_volume_sensitive_mom'] = df['volume_sensitive_momentum_factor'].ewm(alpha=0.2, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.6 * df['smoothed_combined_mom_vol'] + 0.4 * df['smoothed_volume_sensitive_mom']
    df['final_smoothed_alpha'] = df['final_alpha_factor'].ewm(alpha=0.2, adjust=False).mean()
    
    return df['final_smoothed_alpha']
```
