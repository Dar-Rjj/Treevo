```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Ratio
    daily_range = df['high'] - df['low']
    intraday_high_low_ratio = (daily_range / df['open']).fillna(0)
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    sma_5_open_to_close = open_to_close_return.rolling(window=5).mean().fillna(0)
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    volume_adjusted_movement = intraday_movement * df['volume'].fillna(0)
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff().fillna(0)  # Subtract yesterday's close
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum().fillna(0)
    
    # Combine the factors into a single alpha factor
    alpha_factor = (
        intraday_high_low_ratio +
        sma_5_open_to_close +
        volume_adjusted_movement +
        price_volume_trend
    )
    
    return alpha_factor
```
