```python
import pandas as pd
import numpy as np
from talib import EMA, SMA

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema = EMA(open_to_close_return, timeperiod=5)

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = open_to_close_return * df['volume']
    volume_weighted_ema = EMA(volume_weighted_return, timeperiod=5)

    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    average_volume_20d = SMA(df['volume'], timeperiod=20)
    volume_adjusted_movement = intraday_movement / average_volume_20d

    # Combined Momentum and Volatility Factor
    market_sentiment = (df['close'] - df['open']) / df['open']
    weight_intraday_range = 1 / (1 + np.exp(-market_sentiment))
    weight_open_close_momentum = 1 - weight_intraday_range
    combined_momentum_volatility = (weight_intraday_range * normalized_range) + (weight_open_close_momentum * open_to_close_ema)
    combined_momentum_volatility_smoothed = EMA(combined_momentum_volatility, timeperiod=10)

    # Volume-Sensitive Momentum Factor
    weight_price_volume_trend = 1 / (1 + np.exp(-market_sentiment))
    weight_volume_weighted_momentum = 1 - weight_price_volume_trend
    volume_sensitive_momentum = (weight_price_volume_trend * price_volume_trend) + (weight_volume_weighted_momentum * volume_weighted_ema) + (weight_volume_weighted_momentum * volume_adjusted_movement)
    volume_sensitive_momentum_smoothed = EMA(volume_sensitive_momentum, timeperiod=10)

    # Final Alpha Factor
    final_alpha_factor = (0.5 * combined_momentum_volatility_smoothed) + (0.5 * volume_sensitive_momentum_smoothed)
    final_alpha_factor_smoothed = EMA(final_alpha_factor, timeperiod=10)

    return final_alpha_factor_smoothed
```
