```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    sma_5d_open_to_close = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20d
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = daily_price_change * df['volume']
    pvt_30d = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * sma_5d_open_to_close
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * pvt_30d + 0.3 * volume_adjusted_movement
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    alpha_factor = 0.5 * smoothed_combined_momentum_volatility + 0.5 * smoothed_volume_sensitive_momentum
    return alpha_factor
```
