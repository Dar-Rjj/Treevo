```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_ret = (df['close'] / df['open'] - 1)
    open_to_close_sma_5 = open_to_close_ret.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    historical_volatility = df['close'].pct_change().rolling(window=20).std()
    weight_intraday_range = historical_volatility / (historical_volatility + 0.01)
    combined_mom_vol = weight_intraday_range * daily_range + (1 - weight_intraday_range) * open_to_close_sma_5
    combined_mom_vol_smoothed = combined_mom_vol.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_weight = df['volume'] / df['volume'].rolling(window=20).mean()
    volume_sensitive_mom = volume_weight * price_volume_trend + (1 - volume_weight) * volume_adjusted_movement
    volume_sensitive_mom_smoothed = volume_sensitive_mom.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    market_sentiment = (df['close'] - df['close'].shift(1)).rolling(window=20).mean()
    positive_sentiment = market_sentiment > 0
    final_alpha = positive_sentiment.astype(int) * combined_mom_vol_smoothed + (1 - positive_sentiment.astype(int)) * volume_sensitive_mom_smoothed
    final_alpha_smoothed = final_alpha.ewm(alpha=0.1).mean()
    
    return final_alpha_smoothed
```
