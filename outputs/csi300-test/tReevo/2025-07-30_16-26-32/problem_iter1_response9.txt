```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['intraday_price_movement'] = df['high'] - df['low']
    
    # Calculate Average Intraday Price Movement (Last 10 days)
    df['avg_intraday_price_movement'] = df['intraday_price_movement'].rolling(window=10).mean()
    
    # Detect Volume Spike
    df['daily_volume_change'] = df['volume'].diff()
    avg_daily_volume_change = df['daily_volume_change'].mean()
    volume_spike_threshold = 2 * avg_daily_volume_change
    df['volume_spike'] = (df['daily_volume_change'] > volume_spike_threshold).astype(int)
    
    # Combine Intraday Price Movement and Volume Spike
    df['combined_factor'] = df['intraday_price_movement'] * df['volume_spike']
    
    # Sum of Combined Values (Last 5 days)
    combined_factor_5d = df['combined_factor'].rolling(window=5).sum()
    
    return combined_factor_5d
```
