```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_ema5'] = df['open_to_close_return'].ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_oc_return'] = df['open_to_close_return'] * df['volume']
    df['volume_weighted_oc_ema5'] = df['volume_weighted_oc_return'].ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff()
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pv_trend_30d'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_move'] = df['intraday_movement'] / df['avg_volume_20d']
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.3 * df['daily_range'] + 0.4 * df['open_to_close_ema5'] + 0.3 * df['volume_weighted_oc_ema5']
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * df['pv_trend_30d'] + 0.3 * df['volume_weighted_oc_ema5'] + 0.3 * df['volume_adjusted_intraday_move']
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility_smoothed + 0.5 * volume_sensitive_momentum_smoothed
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_factor_smoothed
```
