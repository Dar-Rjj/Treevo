```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smoothed_open_to_close = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) * df['volume']
    smoothed_volume_weighted_return = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1) * df['volume']
    pvt_30_days = price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20_days = df['volume'].rolling(window=20).mean()
    adjusted_intraday_movement = intraday_movement / avg_volume_20_days
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = (0.4 * normalized_range + 
                                    0.3 * smoothed_open_to_close + 
                                    0.3 * smoothed_volume_weighted_return)
    smoothed_combined_mv = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = (0.4 * pvt_30_days + 
                                 0.3 * smoothed_volume_weighted_return + 
                                 0.3 * adjusted_intraday_movement)
    smoothed_volume_sensitive = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha = 0.6 * smoothed_combined_mv + 0.4 * smoothed_volume_sensitive
    final_alpha_smoothed = final_alpha.ewm(alpha=0.1).mean()
    
    return final_alpha_smoothed
```
