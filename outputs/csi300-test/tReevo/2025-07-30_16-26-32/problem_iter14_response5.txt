```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_open_to_close_5'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    average_volume = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = (df['close'] - df['open']) / average_volume
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume Weighted Momentum
    df['volume_weighted_open_to_close_return'] = df['open_to_close_return'] * df['volume']
    df['sma_volume_weighted_open_to_close_5'] = df['volume_weighted_open_to_close_return'].rolling(window=5).mean()
    
    # Volume-Weighted Price-Volume Trend
    df['volume_weighted_pvt'] = df['price_change'] * df['volume']
    df['vwpvt_60'] = df['volume_weighted_pvt'].rolling(window=60).sum()
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.4 * df['daily_range'] + 0.3 * df['sma_open_to_close_5'] + 0.3 * df['sma_volume_weighted_open_to_close_5']
    df['exp_smooth_combined_momentum_volatility'] = df['combined_momentum_volatility'].ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.3 * df['pvt_30'] + 0.4 * df['vwpvt_60'] + 0.3 * df['volume_adjusted_intraday_movement']
    df['exp_smooth_volume_sensitive_momentum'] = df['volume_sensitive_momentum'].ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.6 * df['exp_smooth_combined_momentum_volatility'] + 0.4 * df['exp_smooth_volume_sensitive_momentum']
    final_alpha_factor = df['final_alpha_factor'].ewm(alpha=0.1).mean()
    
    return final_alpha_factor
```
