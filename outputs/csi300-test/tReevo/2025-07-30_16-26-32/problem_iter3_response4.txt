import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['smoothed_momentum'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume Adjusted Intraday Movement
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    df['intraday_movement'] = (df['close'] - df['open']) / avg_vol_20
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['pvt'] = (df['price_change'] * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.5 * df['daily_range'] + 0.5 * df['smoothed_momentum']
    df['combined_factor'] = combined_mom_vol.ewm(span=20).mean()
    
    # Volume-Sensitive Momentum Factor
    df['vol_sens_mom'] = 0.5 * df['pvt'] + 0.5 * df['intraday_movement']
    df['vol_sens_mom_smoothed'] = df['vol_sens_mom'].ewm(span=20).mean()
    
    # Final Alpha Factor
    final_factor = 0.5 * df['combined_factor'] + 0.5 * df['vol_sens_mom_smoothed']
    df['final_alpha'] = final_factor.ewm(span=20).mean()
    
    return df['final_alpha']
