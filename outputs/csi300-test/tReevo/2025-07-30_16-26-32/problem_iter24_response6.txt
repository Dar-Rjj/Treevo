import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close = open_to_close_return.ewm(span=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close = open_to_close_return * df['volume']
    smooth_volume_weighted_open_to_close = volume_weighted_open_to_close.ewm(span=5).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    twenty_day_avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / twenty_day_avg_volume

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.3 * normalized_range + 0.4 * smooth_open_to_close + 0.3 * smooth_volume_weighted_open_to_close
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * price_volume_trend + 0.3 * smooth_volume_weighted_open_to_close + 0.3 * volume_adjusted_intraday_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1).mean()

    # Sentiment and Macroeconomic Indicators (Assuming these are available in the DataFrame)
    sentiment_score = df.get('sentiment_score', 0)  # If not available, default to 0
    macroeconomic_data = df.get('macroeconomic_data', 0)  # If not available, default to 0

    # Final Alpha Factor
    final_alpha_factor = 0.4 * combined_momentum_volatility_smoothed + 0.4 * volume_sensitive_momentum_smoothed + 0.2 * (sentiment_score + macroeconomic_data)
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1).mean()

    return final_alpha_factor_smoothed
