```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    intraday_price_movement = df['high'] - df['low']
    
    # Intraday Volatility Index
    true_range = df['high'] - df['low']
    intraday_volatility_index = true_range.rolling(window=5).mean()
    
    # Volume Confirmation
    volume_change = df['volume'] - df['volume'].shift(1)
    volume_trend_factor = volume_change.apply(lambda x: 1.5 if x > 0 else (0.5 if x <= 0 else 1))
    adjusted_volume_change = volume_change * volume_trend_factor
    
    # Combine Intraday Momentum, Volume, and Volatility
    alpha_factor = (intraday_price_movement * adjusted_volume_change) / intraday_volatility_index
    
    return alpha_factor
```
