```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Price Momentum
    df['close_t1'] = df['close'].shift(1)
    df['close_tn'] = df['close'].shift(5)
    df['price_momentum'] = (df['close_t1'] - df['close_tn']) / df['close_tn']
    
    # Incorporate Volume Trend
    df['volume_t1'] = df['volume'].shift(1)
    df['avg_volume_tn'] = df['volume'].rolling(window=5).mean()
    df['volume_trend_factor'] = df.apply(lambda row: 1.5 if row['volume_t1'] > row['avg_volume_tn'] else 0.5, axis=1)
    
    df['alpha_factor'] = df['price_momentum'] * df['volume_trend_factor']
    
    return df['alpha_factor'].dropna()
```
