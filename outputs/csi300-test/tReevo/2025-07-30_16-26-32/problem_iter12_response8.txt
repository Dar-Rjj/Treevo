```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    oc_return = df['close'] / df['open'] - 1
    oc_sma_5 = oc_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = price_change * df['volume']
    pvt_30 = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    dynamic_weight_mom_vol = 0.6  # Example: dynamically assigned weight
    combined_mom_vol = (daily_range * dynamic_weight_mom_vol + 
                        oc_sma_5 * (1 - dynamic_weight_mom_vol))
    combined_mom_vol_smoothed = combined_mom_vol.ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    dynamic_weight_vol_mom = 0.7  # Example: dynamically assigned weight
    volume_sensitive_mom = (price_volume_trend * dynamic_weight_vol_mom + 
                            volume_adjusted_movement * (1 - dynamic_weight_vol_mom))
    volume_sensitive_mom_smoothed = volume_sensitive_mom.ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    dynamic_weight_final = 0.5  # Example: dynamically assigned weight
    final_alpha_factor = (combined_mom_vol_smoothed * dynamic_weight_final + 
                          volume_sensitive_mom_smoothed * (1 - dynamic_weight_final))
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_factor_smoothed
```
