```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    smooth_volume_weighted = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1) * df['volume']
    price_volume_trend = price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_20_day_volume
    
    # Combined Momentum and Volatility Factor
    adaptive_weights_momentum_volatility = np.array([0.4, 0.3, 0.3])
    combined_momentum_volatility = (daily_range * adaptive_weights_momentum_volatility[0] +
                                    smooth_open_to_close * adaptive_weights_momentum_volatility[1] +
                                    smooth_volume_weighted * adaptive_weights_momentum_volatility[2])
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    adaptive_weights_volume_sensitive = np.array([0.4, 0.3, 0.3])
    volume_sensitive_momentum = (price_volume_trend * adaptive_weights_volume_sensitive[0] +
                                 smooth_volume_weighted * adaptive_weights_volume_sensitive[1] +
                                 volume_adjusted_movement * adaptive_weights_volume_sensitive[2])
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    adaptive_weights_final = np.array([0.5, 0.5])
    final_alpha_factor = (smoothed_combined_momentum_volatility * adaptive_weights_final[0] +
                          smoothed_volume_sensitive_momentum * adaptive_weights_final[1])
    final_smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_smoothed_alpha_factor
```
