```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_move = intraday_movement / avg_volume
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    price_volume_trend = (price_change * df['volume']).rolling(window=20).sum()
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.7 * intraday_high_low_range + 0.3 * smooth_open_to_close
    exp_smooth_combined_mom_vol = combined_mom_vol.ewm(span=10, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_mom = 0.6 * price_volume_trend + 0.4 * volume_adjusted_intraday_move
    exp_smooth_volume_mom = volume_sensitive_mom.ewm(span=10, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * exp_smooth_combined_mom_vol + 0.5 * exp_smooth_volume_mom
    final_alpha_factor_exp_smooth = final_alpha_factor.ewm(span=10, adjust=False).mean()
    
    return final_alpha_factor_exp_smooth
```
