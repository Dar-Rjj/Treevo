```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_return = (df['close'] - df['open']) / df['open']
    open_close_sma = open_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = (df['close'] - df['open']) / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * daily_range + 0.5 * open_close_sma
    combined_mom_vol_smoothed = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * price_volume_trend + 0.5 * volume_adjusted_movement
    volume_sensitive_mom_smoothed = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha = 0.5 * combined_mom_vol_smoothed + 0.5 * volume_sensitive_mom_smoothed
    final_alpha_smoothed = final_alpha.ewm(alpha=0.1).mean()
    
    return final_alpha_smoothed
```
