```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_return = (df['close'] - df['open']) / df['open']
    ema_open_close_5 = open_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    vol_weighted_return = (df['close'] - df['open']) * df['volume']
    ema_vol_weighted_return_5 = vol_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_vol_trend = price_change * df['volume']
    pvt_30 = price_vol_trend.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * daily_range + 0.3 * ema_open_close_5 + 0.3 * ema_vol_weighted_return_5
    smoothed_combined_mom_vol = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * pvt_30 + 0.3 * ema_vol_weighted_return_5 + 0.3 * volume_adjusted_intraday
    smoothed_volume_sensitive_mom = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha = 0.6 * smoothed_combined_mom_vol + 0.4 * smoothed_volume_sensitive_mom
    final_smoothed_alpha = final_alpha.ewm(alpha=0.1).mean()
    
    return final_smoothed_alpha
```
