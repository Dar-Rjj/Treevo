import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['intraday_price_movement'] = df['high'] - df['low']
    
    # Intraday Volatility Index (True Range smoothed with Moving Average)
    df['true_range'] = df['high'] - df['low']
    df['volatility_index'] = df['true_range'].rolling(window=10).mean()
    
    # Volume Confirmation (Volume Change)
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Combine Intraday Momentum and Volume
    df['intraday_momentum_with_volume'] = (df['intraday_price_movement'] * df['volume_change']) / df['volatility_index']
    
    return df['intraday_momentum_with_volume']
