```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['intraday_price_move'] = df['high'] - df['low']

    # Calculate Rolling Average of Intraday Price Movement (Last 10 days)
    df['rolling_avg_intraday_price_move'] = df['intraday_price_move'].rolling(window=10).mean()

    # Detect Volume Spike
    # Calculate Daily Volume Change
    df['daily_volume_change'] = df['volume'].diff()

    # Calculate Average Daily Volume Change (Last 10 days)
    df['avg_daily_volume_change'] = df['daily_volume_change'].rolling(window=10).mean()

    # Set a Threshold (e.g., 2x average daily volume change)
    df['volume_spike_indicator'] = (df['daily_volume_change'] > 2 * df['avg_daily_volume_change']).astype(int)

    # Combine Intraday Price Movement and Volume Spike
    df['combined_value'] = df['intraday_price_move'] * df['volume_spike_indicator']

    # Sum of Combined Values (Last 5 days)
    df['sum_combined_values'] = df['combined_value'].rolling(window=5).sum()

    # Final Alpha Factor
    df['alpha_factor'] = df['sum_combined_values'] * df['volume']

    return df['alpha_factor']
```
