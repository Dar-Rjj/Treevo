```python
import pandas as pd

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Compute 10-Day Moving Average of High-Low Spread
    ma_10 = high_low_spread.rolling(window=10).mean()
    
    # Compute 50-Day Moving Average of the 10-Day Moving Average
    ma_50 = ma_10.rolling(window=50).mean()
    
    return ma_50
```
