import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_momentum = (df['close'] - df['open']) / df['open']
    open_close_momentum_smoothed = open_close_momentum.ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    volume_weighted_return_smoothed = volume_weighted_return.ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1) * df['volume']
    price_volume_trend = price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    vol_adjusted_intraday_move = intraday_movement / avg_vol_20
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = (
        0.4 * daily_range + 
        0.3 * open_close_momentum_smoothed + 
        0.3 * volume_weighted_return_smoothed
    )
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = (
        0.4 * price_volume_trend + 
        0.3 * volume_weighted_return_smoothed + 
        0.3 * vol_adjusted_intraday_move
    )
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.6 * combined_momentum_volatility_smoothed + 
        0.4 * volume_sensitive_momentum_smoothed
    )
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_factor_smoothed
