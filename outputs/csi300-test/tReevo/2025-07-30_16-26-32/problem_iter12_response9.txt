```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_sma_5'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['intraday_movement'] = (df['close'] - df['open']) / avg_volume_20
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_factor = 0.6 * df['daily_range'] + 0.4 * df['open_to_close_sma_5']
    combined_factor = combined_factor.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = 0.7 * df['pvt_30'] + 0.3 * df['intraday_movement']
    volume_sensitive_factor = volume_sensitive_factor.ewm(alpha=0.1, adjust=False).mean()
    
    # Additional Derivatives
    df['daily_return'] = df['close'].pct_change()
    df['price_acceleration'] = df['daily_return'].diff().rolling(window=5).mean()
    df['volume_acceleration'] = df['volume'].diff().rolling(window=5).mean()
    
    # Market Conditions Adjustment
    # Assuming 'benchmark' is a column in the DataFrame representing the benchmark index or sector ETF
    df['market_trend'] = df['benchmark'].rolling(window=10).mean()
    bearish = df['market_trend'] < df['benchmark']
    bullish = df['market_trend'] > df['benchmark']
    
    # Dynamic Weights Adjustment
    combined_factor_weight = 0.6 * (bearish) + 0.4 * (bullish)
    volume_sensitive_factor_weight = 0.4 * (bearish) + 0.6 * (bullish)
    
    # Final Alpha Factor
    final_alpha = combined_factor_weight * combined_factor + volume_sensitive_factor_weight * volume_sensitive_factor
    final_alpha = final_alpha.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha
```
