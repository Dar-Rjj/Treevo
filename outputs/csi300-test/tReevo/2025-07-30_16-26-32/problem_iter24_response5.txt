import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema = open_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close = (df['close'] - df['open']) / df['open'] * df['volume']
    volume_weighted_open_to_close_ema = volume_weighted_open_to_close.ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    price_volume_trend = daily_price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_20_day_volume
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * daily_range + 0.3 * open_to_close_ema + 0.3 * volume_weighted_open_to_close_ema
    combined_momentum_volatility_ema = combined_momentum_volatility.ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * price_volume_trend + 0.3 * volume_weighted_open_to_close_ema + 0.3 * volume_adjusted_intraday_movement
    volume_sensitive_momentum_ema = volume_sensitive_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility_ema + 0.5 * volume_sensitive_momentum_ema
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean()
    
    return final_alpha_factor
