import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema_5 = open_to_close_return.ewm(span=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = open_to_close_return * df['volume']
    volume_weighted_ema_5 = volume_weighted_return.ewm(span=5).mean()

    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    adjusted_intraday_movement = intraday_movement / avg_volume_20

    # Combined Momentum and Volatility Factor
    w1, w2, w3 = 0.33, 0.33, 0.34  # Dynamic Weights
    combined_momentum_volatility = (w1 * normalized_range + 
                                    w2 * open_to_close_ema_5 + 
                                    w3 * volume_weighted_ema_5)
    alpha1 = 0.5  # Exponential Smoothing Alpha
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=alpha1).mean()

    # Volume-Sensitive Momentum Factor
    w4, w5, w6 = 0.33, 0.33, 0.34  # Dynamic Weights
    volume_sensitive_momentum = (w4 * price_volume_trend + 
                                 w5 * volume_weighted_ema_5 + 
                                 w6 * adjusted_intraday_movement)
    alpha2 = 0.5  # Exponential Smoothing Alpha
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=alpha2).mean()

    # Final Alpha Factor
    w7, w8 = 0.5, 0.5  # Dynamic Weights
    final_alpha_factor = (w7 * combined_momentum_volatility_smoothed + 
                          w8 * volume_sensitive_momentum_smoothed)
    alpha3 = 0.5  # Exponential Smoothing Alpha
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=alpha3).mean()

    return final_alpha_factor_smoothed
