```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum_ema5 = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    volume_weighted_return_ema5 = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1) * df['volume']
    pvt_30 = price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20
    
    # Combined Momentum and Volatility Factor
    w1, w2, w3 = 0.3, 0.3, 0.4  # Adaptive weights
    combined_momentum_volatility = (w1 * daily_range + 
                                    w2 * open_to_close_momentum_ema5 + 
                                    w3 * volume_weighted_return_ema5)
    combined_momentum_volatility_ema = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    w4, w5, w6 = 0.4, 0.3, 0.3  # Adaptive weights
    volume_sensitive_momentum = (w4 * pvt_30 + 
                                 w5 * volume_weighted_return_ema5 + 
                                 w6 * volume_adjusted_movement)
    volume_sensitive_momentum_ema = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    w7, w8 = 0.5, 0.5  # Adaptive weights
    final_alpha_factor = (w7 * combined_momentum_volatility_ema + 
                          w8 * volume_sensitive_momentum_ema)
    final_alpha_factor_ema = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_alpha_factor_ema
```
