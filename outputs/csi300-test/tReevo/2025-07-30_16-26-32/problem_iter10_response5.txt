```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_5_open_to_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = (df['close'] - df['open'])
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['avg_volume_20']
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    intraday_high_low_weight = 0.6  # Example weight, can be dynamically assigned
    open_to_close_momentum_weight = 0.4  # Example weight, can be dynamically assigned
    df['combined_factor'] = (intraday_high_low_weight * df['daily_range']) + \
                            (open_to_close_momentum_weight * df['sma_5_open_to_close'])
    df['combined_factor'] = df['combined_factor'].ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    pvt_weight = 0.7  # Example weight, can be dynamically assigned
    volume_adjusted_movement_weight = 0.3  # Example weight, can be dynamically assigned
    df['volume_sensitive_factor'] = (pvt_weight * df['pvt_30']) + \
                                    (volume_adjusted_movement_weight * df['volume_adjusted_intraday_movement'])
    df['volume_sensitive_factor'] = df['volume_sensitive_factor'].ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    combined_momentum_volatility_weight = 0.5  # Example weight, can be dynamically assigned
    volume_sensitive_momentum_weight = 0.5  # Example weight, can be dynamically assigned
    df['final_alpha_factor'] = (combined_momentum_volatility_weight * df['combined_factor']) + \
                               (volume_sensitive_momentum_weight * df['volume_sensitive_factor'])
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor']
```
