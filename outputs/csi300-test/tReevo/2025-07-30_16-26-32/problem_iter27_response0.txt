```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    ema_open_to_close = open_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close = (df['close'] - df['open']) * df['volume']
    ema_volume_weighted_open_to_close = volume_weighted_open_to_close.ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    pvt_30_days = daily_price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_20_day_volume
    
    # Combined Momentum and Volatility Factor
    weight_intraday_range = 0.4  # Example: Market sentiment-based weight
    weight_open_to_close_momentum = 0.3
    weight_volume_weighted_momentum = 0.3
    combined_momentum_volatility = (
        weight_intraday_range * daily_range +
        weight_open_to_close_momentum * ema_open_to_close +
        weight_volume_weighted_momentum * ema_volume_weighted_open_to_close
    )
    adaptive_smoothing_alpha = 0.2  # Example: Adaptive smoothing alpha
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=adaptive_smoothing_alpha, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    weight_price_volume_trend = 0.4  # Example: Market sentiment-based weight
    weight_volume_weighted_open_to_close = 0.3
    weight_volume_adjusted_intraday = 0.3
    volume_sensitive_momentum = (
        weight_price_volume_trend * pvt_30_days +
        weight_volume_weighted_open_to_close * ema_volume_weighted_open_to_close +
        weight_volume_adjusted_intraday * volume_adjusted_intraday_movement
    )
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=adaptive_smoothing_alpha, adjust=False).mean()
    
    # Final Alpha Factor
    weight_combined_momentum_volatility = 0.6  # Example: Market sentiment-based weight
    weight_volume_sensitive_momentum = 0.4
    final_alpha_factor = (
        weight_combined_momentum_volatility * smoothed_combined_momentum_volatility +
        weight_volume_sensitive_momentum * smoothed_volume_sensitive_momentum
    )
    final_smoothed_alpha_factor = final_alpha_factor.ewm(alpha=adaptive_smoothing_alpha, adjust=False).mean()
    
    return final_smoothed_alpha_factor
```
