```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Weight by Volume
    df['weighted_high_low_spread'] = df['high_low_spread'] * df['volume']
    
    # Compute 10-Day and 50-Day Weighted Moving Averages of High-Low Spreads
    df['10_day_wma'] = df['weighted_high_low_spread'].rolling(window=10).mean()
    df['50_day_wma'] = df['weighted_high_low_spread'].rolling(window=50).mean()
    
    # Compute 10-Day and 50-Day Simple Moving Averages of Close Prices
    df['10_day_sma_close'] = df['close'].rolling(window=10).mean()
    df['50_day_sma_close'] = df['close'].rolling(window=50).mean()
    
    # Trend Strength Indicator
    df['trend_strength'] = df['10_day_sma_close'] - df['50_day_sma_close']
    df['bullish_bearish'] = df['trend_strength'].apply(lambda x: 1 if x > 0 else -1)
    
    # Volatility Indicator
    df['volatility'] = df['high_low_spread'].rolling(window=10).std()
    
    # Combine Trend Strength and Volatility
    df['trend_volatility'] = df['bullish_bearish'] * (1 + df['volatility'])
    
    # Adjusted Volume-Weighted High-Low Spread
    df['adjusted_spread'] = df['10_day_wma'] - df['50_day_wma']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['trend_volatility'] * df['adjusted_spread']
    
    return df['alpha_factor'].dropna()
```
