import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_open_to_close_5'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['sma_volume_weighted_return_5'] = df['volume_weighted_return'].rolling(window=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['price_volume_trend_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Dynamic Weights
    df['weight_intraday_range'] = df['daily_range'].rolling(window=5).mean().rank(pct=True)
    df['weight_open_close_momentum'] = df['sma_open_to_close_5'].rank(pct=True)
    df['weight_volume_weighted_momentum'] = df['sma_volume_weighted_return_5'].rank(pct=True)
    df['weight_price_volume_trend'] = df['price_volume_trend_30'].rank(pct=True)
    
    # Dynamic Weighted Momentum and Volatility Factor
    df['dynamic_factor'] = (df['daily_range'] * df['weight_intraday_range'] +
                            df['sma_open_to_close_5'] * df['weight_open_close_momentum'] +
                            df['sma_volume_weighted_return_5'] * df['weight_volume_weighted_momentum'])
    df['smoothed_dynamic_factor'] = df['dynamic_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_factor'] = (df['price_volume_trend_30'] * df['weight_price_volume_trend'] +
                                     df['sma_volume_weighted_return_5'] * df['weight_volume_weighted_momentum'])
    df['smoothed_volume_sensitive_factor'] = df['volume_sensitive_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['smoothed_dynamic_factor'] * df['weight_dynamic_factor'] +
                                df['smoothed_volume_sensitive_factor'] * df['weight_volume_sensitive_factor'])
    df['smoothed_final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['smoothed_final_alpha_factor']
