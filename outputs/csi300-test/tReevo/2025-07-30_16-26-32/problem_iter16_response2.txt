```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = df['close'] / df['open'] - 1
    df['sma_open_to_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['avg_volume_20']
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.6 * df['daily_range'] + 0.4 * df['sma_open_to_close']
    df['combined_momentum_volatility_exp'] = df['combined_momentum_volatility'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.7 * df['pvt_30'] + 0.3 * df['volume_adjusted_intraday_movement']
    df['volume_sensitive_momentum_exp'] = df['volume_sensitive_momentum'].ewm(alpha=0.1, adjust=False).mean()
    
    # Additional Price Derivatives
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['volatility_10'] = df['close'].pct_change().rolling(window=10).std()
    
    # Additional Volume Derivatives
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volatility_volume_10'] = df['volume'].pct_change().rolling(window=10).std()
    
    # Enhanced Final Alpha Factor
    df['enhanced_alpha_factor'] = 0.4 * df['combined_momentum_volatility_exp'] + \
                                 0.3 * df['volume_sensitive_momentum_exp'] + \
                                 0.2 * df['sma_10'] + \
                                 0.1 * df['volatility_10']
    df['enhanced_alpha_factor_exp'] = df['enhanced_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.8 * df['enhanced_alpha_factor_exp'] + 0.2 * df['volatility_volume_10']
    df['final_alpha_factor_exp'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor_exp']
```
