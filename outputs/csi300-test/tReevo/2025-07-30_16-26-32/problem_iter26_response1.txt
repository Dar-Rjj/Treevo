```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_return_ema_5'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = (df['close'] - df['open']) * df['volume']
    df['volume_weighted_return_ema_5'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = (df['price_change'] * df['volume']).rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['vol_adj_intraday_move'] = df['intraday_movement'] / df['avg_volume_20']
    
    # Combined Momentum and Volatility Factor
    adaptive_weights = {
        'intraday_high_low_range': 0.4,
        'open_to_close_momentum': 0.3,
        'volume_weighted_open_to_close': 0.3
    }
    combined_factor = (adaptive_weights['intraday_high_low_range'] * df['daily_range'] +
                       adaptive_weights['open_to_close_momentum'] * df['open_to_close_return_ema_5'] +
                       adaptive_weights['volume_weighted_open_to_close'] * df['volume_weighted_return_ema_5'])
    df['combined_mom_vol_factor'] = combined_factor.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    adaptive_weights_vol = {
        'price_volume_trend': 0.4,
        'volume_weighted_open_to_close': 0.3,
        'vol_adj_intraday_move': 0.3
    }
    volume_sensitive_factor = (adaptive_weights_vol['price_volume_trend'] * df['price_volume_trend'] +
                               adaptive_weights_vol['volume_weighted_open_to_close'] * df['volume_weighted_return_ema_5'] +
                               adaptive_weights_vol['vol_adj_intraday_move'] * df['vol_adj_intraday_move'])
    df['volume_sensitive_mom_factor'] = volume_sensitive_factor.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_weights = {
        'combined_mom_vol': 0.5,
        'volume_sensitive_mom': 0.5
    }
    df['final_alpha_factor'] = (final_weights['combined_mom_vol'] * df['combined_mom_vol_factor'] +
                                final_weights['volume_sensitive_mom'] * df['volume_sensitive_mom_factor'])
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.1).mean()

    return df['final_alpha_factor']
```
