```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the typical price
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow
    money_flow = typical_price * df['volume']
    
    # Calculate the exponential moving average of money flow
    ema_money_flow = money_flow.ewm(span=14, adjust=False).mean()
    
    # Calculate the momentum of money flow
    momentum = (money_flow - money_flow.shift(14)) / money_flow.shift(14)
    
    # Calculate the final factor value
    factor_value = (ema_money_flow - ema_money_flow.shift(14)) * momentum
    
    return factor_value
```
