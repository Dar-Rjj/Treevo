```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum_ema5 = open_to_close_return.ewm(span=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) * df['volume']
    volume_weighted_return_ema5 = volume_weighted_return.ewm(span=5).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_trend = daily_price_change * df['volume']
    price_volume_trend_30d = price_volume_trend.rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    volume_sma20 = df['volume'].rolling(window=20).mean()
    adjusted_intraday_movement = intraday_movement / volume_sma20

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = (0.3 * normalized_range + 
                                    0.4 * open_to_close_momentum_ema5 + 
                                    0.3 * volume_weighted_return_ema5)
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = (0.4 * price_volume_trend_30d + 
                                 0.3 * volume_weighted_return_ema5 + 
                                 0.3 * adjusted_intraday_movement)
    
    # Final Alpha Factor
    final_alpha_factor = (0.5 * combined_momentum_volatility + 
                          0.5 * volume_sensitive_momentum)
    
    # Apply Adaptive Exponential Smoothing
    alpha = 0.1  # Fixed alpha for exponential smoothing
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=alpha, adjust=False).mean()
    
    return final_alpha_factor_smoothed
```
