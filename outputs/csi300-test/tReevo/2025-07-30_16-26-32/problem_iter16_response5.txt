```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    ema_open_to_close = open_to_close_return.ewm(span=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Volume Weighted Momentum
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    ema_volume_weighted_return = volume_weighted_return.ewm(span=5).mean()
    
    # Adaptive Weights
    def adaptive_weights(series, lookback=20):
        return (series.rolling(window=lookback).mean() / series.rolling(window=lookback).std()).fillna(0)
    
    weight_intraday_range = adaptive_weights(daily_range)
    weight_open_to_close_momentum = adaptive_weights(ema_open_to_close)
    weight_volume_weighted_momentum = adaptive_weights(ema_volume_weighted_return)
    weight_price_volume_trend = adaptive_weights(price_volume_trend)
    weight_volume_adjusted_movement = adaptive_weights(volume_adjusted_movement)
    
    # Adaptive Momentum and Volatility Factor
    adaptive_momentum_volatility = (
        weight_intraday_range * daily_range +
        weight_open_to_close_momentum * ema_open_to_close +
        weight_volume_weighted_momentum * ema_volume_weighted_return
    ).ewm(span=20).mean()
    
    # Adaptive Volume-Sensitive Momentum Factor
    adaptive_volume_sensitive_momentum = (
        weight_price_volume_trend * price_volume_trend +
        weight_volume_adjusted_movement * volume_adjusted_movement
    ).ewm(span=20).mean()
    
    # Final Alpha Factor
    final_factor = (
        adaptive_weights(adaptive_momentum_volatility) * adaptive_momentum_volatility +
        adaptive_weights(adaptive_volume_sensitive_momentum) * adaptive_volume_sensitive_momentum
    ).ewm(span=20).mean()
    
    return final_factor
```
