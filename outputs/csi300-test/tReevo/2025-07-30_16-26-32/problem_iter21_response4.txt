```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    df['normalized_range'] = df['daily_range'] / df['open']

    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_5d'] = df['open_to_close_return'].ewm(span=5).mean()

    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['ema_volume_weighted_return_5d'] = df['volume_weighted_return'].ewm(span=5).mean()

    # Price-Volume Trend Indicator
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['daily_price_change'] * df['volume']
    df['pvt_30d'] = df['price_volume_trend'].rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['sma_volume_20d']

    # Combined Momentum and Volatility Factor
    weights_momentum_volatility = {
        'normalized_range': 0.3,  # Placeholder for adaptive weights
        'ema_open_to_close_5d': 0.4,
        'ema_volume_weighted_return_5d': 0.3
    }
    combined_momentum_volatility = (df['normalized_range'] * weights_momentum_volatility['normalized_range'] +
                                    df['ema_open_to_close_5d'] * weights_momentum_volatility['ema_open_to_close_5d'] +
                                    df['ema_volume_weighted_return_5d'] * weights_momentum_volatility['ema_volume_weighted_return_5d'])
    df['combined_momentum_volatility'] = combined_momentum_volatility.ewm(alpha=0.1).mean()

    # Volume-Sensitive Momentum Factor
    weights_volume_sensitive = {
        'pvt_30d': 0.4,  # Placeholder for adaptive weights
        'ema_volume_weighted_return_5d': 0.3,
        'volume_adjusted_intraday_movement': 0.3
    }
    volume_sensitive_momentum = (df['pvt_30d'] * weights_volume_sensitive['pvt_30d'] +
                                 df['ema_volume_weighted_return_5d'] * weights_volume_sensitive['ema_volume_weighted_return_5d'] +
                                 df['volume_adjusted_intraday_movement'] * weights_volume_sensitive['volume_adjusted_intraday_movement'])
    df['volume_sensitive_momentum'] = volume_sensitive_momentum.ewm(alpha=0.1).mean()

    # Final Alpha Factor
    final_weights = {
        'combined_momentum_volatility': 0.6,  # Placeholder for adaptive weights
        'volume_sensitive_momentum': 0.4
    }
    final_alpha_factor = (df['combined_momentum_volatility'] * final_weights['combined_momentum_volatility'] +
                          df['volume_sensitive_momentum'] * final_weights['volume_sensitive_momentum'])
    df['final_alpha_factor'] = final_alpha_factor.ewm(alpha=0.1).mean()

    return df['final_alpha_factor']
```
