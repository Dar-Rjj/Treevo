```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Returns
    df['daily_return'] = df['close'].pct_change()

    # Aggregate Returns for Momentum
    last_5_days_returns = df['daily_return'].rolling(window=5).sum()
    average_volume_last_5_days = df['volume'].rolling(window=5).mean()
    momentum_factor = last_5_days_returns * average_volume_last_5_days

    # Calculate Reversal Factor
    reversal_factor = df['high'] - df['low']

    # Combine Momentum and Reversal
    combined_factor = momentum_factor - reversal_factor

    # Final Alpha Factor
    final_alpha_factor = combined_factor * df['volume']
    
    return final_alpha_factor
```
