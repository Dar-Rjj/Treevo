```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_return'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = (df['close'] - df['open']) * df['volume']
    df['ema_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['agg_price_volume_trend'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_20_vol'] = df['volume'].rolling(window=20).mean()
    df['vol_adj_intraday_movement'] = df['intraday_movement'] / df['avg_20_vol']
    
    # Mean Reversion Factor
    df['sma_60'] = df['close'].rolling(window=60).mean()
    df['price_deviation'] = df['sma_60'] - df['close']
    df['ema_price_deviation'] = df['price_deviation'].ewm(span=5).mean()
    
    # Liquidity Measure
    df['daily_dollar_volume'] = df['close'] * df['volume']
    df['ema_daily_dollar_volume'] = df['daily_dollar_volume'].ewm(span=20).mean()
    
    # Sentiment Analysis (Assuming sentiment scores are available)
    # For the sake of this example, we'll create a dummy sentiment score
    df['positive_sentiment'] = np.random.normal(0.5, 0.1, len(df))
    df['negative_sentiment'] = np.random.normal(0.5, 0.1, len(df))
    df['agg_sentiment_score'] = df['positive_sentiment'] - df['negative_sentiment']
    
    # Combined Momentum and Volatility Factor
    df['combined_mom_vol'] = 0.4 * df['daily_range'] + 0.3 * df['ema_open_to_close_return'] + 0.3 * df['ema_volume_weighted_return']
    df['smoothed_combined_mom_vol'] = df['combined_mom_vol'].ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_mom'] = 0.4 * df['agg_price_volume_trend'] + 0.3 * df['ema_volume_weighted_return'] + 0.3 * df['vol_adj_intraday_movement']
    df['smoothed_volume_sensitive_mom'] = df['volume_sensitive_mom'].ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (0.3 * df['smoothed_combined_mom_vol'] + 
                                0.3 * df['smoothed_volume_sensitive_mom'] + 
                                0.2 * df['ema_price_deviation'] + 
                                0.1 * df['ema_daily_dollar_volume'] + 
                                0.1 * df['agg_sentiment_score'])
    
    final_alpha_factor = df['final_alpha_factor'].ewm(alpha=0.1).mean()
    
    return final_alpha_factor
```
