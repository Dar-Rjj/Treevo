```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']

    # Open to Close Momentum
    df['oc_return'] = (df['close'] - df['open']) / df['open']
    df['oc_sma5'] = df['oc_return'].rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = (df['close'] - df['open'])
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_movement'] = df['intraday_movement'] / avg_volume_20

    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff()
    df['pv_trend'] = df['price_change'] * df['volume']
    df['pv_trend_30'] = df['pv_trend'].rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * df['daily_range'] + 0.4 * df['oc_sma5']
    combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * df['pv_trend_30'] + 0.3 * df['volume_adjusted_movement']
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_momentum_volatility + 0.4 * volume_sensitive_momentum
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()

    return final_alpha_factor.dropna()
```
