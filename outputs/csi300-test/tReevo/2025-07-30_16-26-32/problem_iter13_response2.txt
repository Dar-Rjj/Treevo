```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    df['intraday_high_to_low_range'] = df['daily_range'] / df['open']

    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['smoothed_open_to_close_return'] = df['open_to_close_return'].rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    df['avg_volume_20_day'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = (df['close'] - df['open']) / df['avg_volume_20_day']

    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['aggregated_price_volume_trend'] = df['price_volume_trend'].rolling(window=30).sum()

    # Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['open_to_close_return'] * df['volume']
    df['smoothed_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=5).mean()

    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility_factor'] = (0.4 * df['intraday_high_to_low_range'] +
                                                 0.3 * df['smoothed_open_to_close_return'] +
                                                 0.3 * df['smoothed_volume_weighted_momentum'])
    df['exponentially_smoothed_combined_momentum_volatility_factor'] = df['combined_momentum_volatility_factor'].ewm(alpha=0.1, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum_factor'] = (0.7 * df['aggregated_price_volume_trend'] +
                                              0.3 * df['volume_adjusted_intraday_movement'])
    df['exponentially_smoothed_volume_sensitive_momentum_factor'] = df['volume_sensitive_momentum_factor'].ewm(alpha=0.1, adjust=False).mean()

    # Final Alpha Factor
    df['final_alpha_factor'] = (0.6 * df['exponentially_smoothed_combined_momentum_volatility_factor'] +
                                0.4 * df['exponentially_smoothed_volume_sensitive_momentum_factor'])
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()

    return df['final_alpha_factor']
```
