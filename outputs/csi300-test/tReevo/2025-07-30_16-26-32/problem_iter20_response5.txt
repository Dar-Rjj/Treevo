import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_5'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = (df['close'] - df['open']) * df['volume']
    df['ema_volume_weighted_return_5'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['daily_price_change'] = df['close'].diff(1) * df['volume']
    df['price_volume_trend_30'] = df['daily_price_change'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = (df['close'] - df['open'])
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['vol_adj_intraday_move'] = df['intraday_movement'] / df['avg_volume_20']
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.4 * df['daily_range'] + 0.3 * df['ema_open_to_close_5'] + 0.3 * df['ema_volume_weighted_return_5']
    df['combined_mom_vol_smoothed'] = combined_mom_vol.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * df['price_volume_trend_30'] + 0.3 * df['ema_volume_weighted_return_5'] + 0.3 * df['vol_adj_intraday_move']
    df['volume_sensitive_mom_smoothed'] = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * df['combined_mom_vol_smoothed'] + 0.4 * df['volume_sensitive_mom_smoothed']
    df['final_alpha_factor'] = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor']
