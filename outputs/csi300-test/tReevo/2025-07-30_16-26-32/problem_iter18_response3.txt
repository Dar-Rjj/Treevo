import pandas as pd

def heuristics_v2(df):
    """
    Generate a novel and interpretable alpha factor based on the given DataFrame.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'].
    
    Returns:
    pd.Series: A Series representing the alpha factor values.
    """
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the average price
    avg_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the volume weighted by the close price
    volume_weighted_close = df['close'] * df['volume']
    
    # Calculate the amount ratio
    amount_ratio = df['amount'] / df['volume']
    
    # Calculate the price change
    price_change = df['close'] - df['open']
    
    # Generate the alpha factor
    alpha_factor = (price_change / price_range) * (volume_weighted_close / df['close'].rolling(window=20).mean()) * (amount_ratio / df['close'])
    
    return alpha_factor
