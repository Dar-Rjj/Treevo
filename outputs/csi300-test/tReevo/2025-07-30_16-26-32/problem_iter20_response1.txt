```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_return = (df['close'] / df['open']) - 1
    open_close_sma = open_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open']) / df['volume'].rolling(window=20).mean()
    
    # Price-Volume Trend Indicator
    price_volume_trend = (df['close'] - df['close'].shift(1)) * df['volume']
    price_volume_trend_30 = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * open_close_sma
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * price_volume_trend_30 + 0.3 * intraday_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = 0.6 * combined_momentum_volatility_smoothed + 0.4 * volume_sensitive_momentum_smoothed
    final_alpha_smoothed = final_alpha.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_smoothed
```
