```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Weight by Volume
    weighted_high_low_spread = high_low_spread * df['volume']
    
    # Compute 10-Day Weighted Moving Average
    wma_10 = weighted_high_low_spread.rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    
    # Compute 50-Day Weighted Moving Average
    wma_50 = wma_10.rolling(window=50).mean()
    
    # Calculate 10-Day Simple Moving Average of Close Prices
    sma_10_close = df['close'].rolling(window=10).mean()
    
    # Calculate 50-Day Simple Moving Average of Close Prices
    sma_50_close = df['close'].rolling(window=50).mean()
    
    # Trend Strength Indicator
    trend_strength = sma_10_close - sma_50_close
    bullish = (trend_strength > 0).astype(int)
    
    # Volatility Indicator
    volatility = high_low_spread.rolling(window=10).std()
    
    # Composite Alpha Factor
    composite_alpha = np.where(bullish, 
                               (sma_10_close - sma_50_close) * (1 + volatility), 
                               (sma_10_close - sma_50_close) * (1 - volatility))
    
    # Incorporate Volume-Weighted High-Low Spread
    alpha_factor = composite_alpha + wma_10 - wma_50
    
    return alpha_factor
```
