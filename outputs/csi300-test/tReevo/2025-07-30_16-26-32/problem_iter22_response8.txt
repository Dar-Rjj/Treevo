```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) * df['volume']
    volume_weighted_ema = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1)
    price_volume_trend = daily_price_change * df['volume']
    price_volume_trend_30d = price_volume_trend.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20d
    
    # Combined Momentum and Volatility Factor
    weight_intraday_range = 0.4
    weight_open_to_close_momentum = 0.3
    weight_volume_weighted_momentum = 0.3
    combined_momentum_volatility = (
        weight_intraday_range * normalized_range +
        weight_open_to_close_momentum * open_to_close_ema +
        weight_volume_weighted_momentum * volume_weighted_ema
    )
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    weight_price_volume_trend = 0.4
    weight_volume_weighted_open_to_close = 0.3
    weight_volume_adjusted_intraday = 0.3
    volume_sensitive_momentum = (
        weight_price_volume_trend * price_volume_trend_30d +
        weight_volume_weighted_open_to_close * volume_weighted_ema +
        weight_volume_adjusted_intraday * volume_adjusted_intraday
    )
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    weight_combined_momentum_volatility = 0.6
    weight_volume_sensitive_momentum = 0.4
    final_alpha_factor = (
        weight_combined_momentum_volatility * combined_momentum_volatility_smoothed +
        weight_volume_sensitive_momentum * volume_sensitive_momentum_smoothed
    )
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_alpha_factor_smoothed
```
