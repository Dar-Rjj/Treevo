```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['oc_return'] = (df['close'] - df['open']) / df['open']
    df['oc_sma_5'] = df['oc_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    average_volume_20 = df['volume'].rolling(window=20).mean()
    df['intraday_move_vol_adj'] = (df['close'] - df['open']) / average_volume_20
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff()
    df['pvt'] = (df['price_change'] * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    df['mom_vol_factor'] = 0.6 * df['daily_range'] + 0.4 * df['oc_sma_5']
    df['exp_smooth_mom_vol'] = df['mom_vol_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['vol_sensitive_mom'] = 0.7 * df['pvt'] + 0.3 * df['intraday_move_vol_adj']
    df['exp_smooth_vol_mom'] = df['vol_sensitive_mom'].ewm(alpha=0.1, adjust=False).mean()
    
    # Relative Strength Indicator
    df['sma_50'] = df['close'].rolling(window=50).mean()
    df['sma_10'] = df['close'].rolling(window=10).mean()
    df['rsi'] = df['sma_10'] / df['sma_50']
    
    # Adaptive Volatility
    df['std_20'] = df['close'].rolling(window=20).std()
    df['sma_close_20'] = df['close'].rolling(window=20).mean()
    df['adaptive_vol'] = df['std_20'] / df['sma_close_20']
    
    # Dynamic Volume Adjustment
    df['avg_vol_20'] = df['volume'].rolling(window=20).mean()
    df['vol_ratio'] = df['volume'] / df['avg_vol_20']
    
    # Final Alpha Factor
    df['alpha_factor'] = (
        0.3 * df['exp_smooth_mom_vol'] +
        0.2 * df['exp_smooth_vol_mom'] +
        0.2 * df['rsi'] +
        0.3 * df['adaptive_vol']
    )
    df['final_alpha'] = df['alpha_factor'] * df['vol_ratio']
    df['final_alpha_exp_smooth'] = df['final_alpha'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_exp_smooth'].dropna()
```
