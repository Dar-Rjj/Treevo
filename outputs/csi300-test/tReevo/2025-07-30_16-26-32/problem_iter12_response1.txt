```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_return = df['close'] / df['open'] - 1
    open_close_sma_5 = open_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * open_close_sma_5
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * price_volume_trend + 0.3 * volume_adjusted_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_momentum_volatility_smoothed + 0.4 * volume_sensitive_momentum_smoothed
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_alpha_factor_smoothed
```
