```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema = open_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    volume_weighted_ema = volume_weighted_return.ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    price_volume_trend = daily_price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_vol_20
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = (daily_range * 0.33 + 
                        open_to_close_ema * 0.33 + 
                        volume_weighted_ema * 0.33)
    combined_mom_vol_smoothed = combined_mom_vol.ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_mom = (price_volume_trend * 0.33 + 
                            volume_weighted_ema * 0.33 + 
                            volume_adjusted_intraday * 0.33)
    volume_sensitive_mom_smoothed = volume_sensitive_mom.ewm(alpha=0.2, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = (combined_mom_vol_smoothed * 0.5 + 
                   volume_sensitive_mom_smoothed * 0.5)
    final_alpha_smoothed = final_alpha.ewm(alpha=0.2, adjust=False).mean()

    return final_alpha_smoothed
```
