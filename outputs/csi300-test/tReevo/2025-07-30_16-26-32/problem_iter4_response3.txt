```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_close_return'] = (df['close'] / df['open']) - 1
    df['smoothed_open_close_return'] = df['open_close_return'].ewm(span=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = (df['close'] - df['open']) / df['volume'].rolling(window=30).mean()
    
    # Price-Volume Trend Indicator
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = (df['daily_price_change'] * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.6 * df['daily_range'] + 0.4 * df['smoothed_open_close_return']
    df['smoothed_combined_momentum_volatility'] = df['combined_momentum_volatility'].ewm(span=5).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.7 * df['price_volume_trend'] + 0.3 * df['intraday_movement']
    df['smoothed_volume_sensitive_momentum'] = df['volume_sensitive_momentum'].ewm(span=5).mean()
    
    # Integrated Alpha Factor
    df['integrated_alpha_factor'] = 0.6 * df['smoothed_combined_momentum_volatility'] + 0.4 * df['smoothed_volume_sensitive_momentum']
    df['final_alpha_factor'] = df['integrated_alpha_factor'].ewm(span=5).mean()
    
    return df['final_alpha_factor']
```
