```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_daily_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20d

    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * normalized_daily_range + 0.5 * smooth_open_to_close
    combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * price_volume_trend + 0.5 * volume_adjusted_intraday
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility + 0.5 * volume_sensitive_momentum
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()

    return final_alpha_factor
```
