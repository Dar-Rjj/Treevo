```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Price Movement
    df['intraday_movement'] = df['high'] - df['low']
    
    # Calculate Average Intraday Price Movement (Last 10 days)
    df['avg_intraday_movement_10d'] = df['intraday_movement'].rolling(window=10).mean()
    
    # Detect Volume Spike
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    avg_volume_change = df['volume_change'].rolling(window=10).mean()
    threshold = 2 * avg_volume_change
    df['volume_spike'] = (df['volume_change'] > threshold).astype(int)
    
    # Combine Intraday Price Movement and Volume Spike
    df['combined_value'] = df['intraday_movement'] * df['volume_spike']
    df['sum_combined_5d'] = df['combined_value'].rolling(window=5).sum()
    
    # Calculate Relative Strength
    df['return_10d'] = df['close'].pct_change(periods=10)
    df['return_50d'] = df['close'].pct_change(periods=50)
    df['relative_strength'] = df['return_10d'] / df['return_50d']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['sum_combined_5d'] * df['relative_strength']
    
    return df['alpha_factor']
```
