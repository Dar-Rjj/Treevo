```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_return = df['close'] / df['open'] - 1
    open_close_momentum = open_close_return.ewm(span=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open']) / df['volume'].rolling(window=5).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.5 * daily_range + 0.5 * open_close_momentum
    combined_mom_vol_smoothed = combined_mom_vol.ewm(span=10).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_mom = 0.5 * price_volume_trend + 0.5 * intraday_movement
    volume_sensitive_mom_smoothed = volume_sensitive_mom.ewm(span=10).mean()
    
    # Final Alpha Factor
    final_factor = 0.5 * combined_mom_vol_smoothed + 0.5 * volume_sensitive_mom_smoothed
    final_factor = final_factor.ewm(span=10).mean()
    
    return final_factor
```
