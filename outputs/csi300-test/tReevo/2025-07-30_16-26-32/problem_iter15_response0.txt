import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_to_low_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.rolling(window=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = open_to_close_return * df['volume']
    volume_weighted_momentum = volume_weighted_return.rolling(window=5).mean()

    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = (
        0.4 * intraday_high_to_low_range +
        0.3 * open_to_close_momentum +
        0.3 * volume_weighted_momentum
    )
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = (
        0.7 * price_volume_trend +
        0.3 * volume_weighted_momentum
    )
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = (
        0.6 * combined_momentum_volatility_smoothed +
        0.4 * volume_sensitive_momentum_smoothed
    )
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()

    return final_alpha_factor_smoothed
