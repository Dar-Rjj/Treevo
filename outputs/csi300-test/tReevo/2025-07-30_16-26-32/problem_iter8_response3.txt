```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_5_open_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['intraday_movement'] = (df['close'] - df['open']) / avg_volume_20
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff()
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.6 * df['daily_range'] + 0.4 * df['sma_5_open_close']
    df['exp_smoothed_mv'] = df['combined_momentum_volatility'].ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.7 * df['pvt_30'] + 0.3 * df['intraday_movement']
    df['exp_smoothed_vs'] = df['volume_sensitive_momentum'].ewm(alpha=0.1).mean()
    
    # Sector-Specific Factor
    # Assuming `sector_index` is a column in the DataFrame
    df['sector_sma_21'] = df['sector_index'].rolling(window=21).mean()
    df['relative_strength'] = df['sector_sma_21'] / df['market_index']
    
    # Macroeconomic Factor
    # Assuming `gdp_growth_rate` and `federal_funds_rate` are columns in the DataFrame
    df['macroeconomic_factor'] = df['gdp_growth_rate'] - df['federal_funds_rate']
    
    # Final Alpha Factor
    df['final_alpha'] = 0.4 * df['exp_smoothed_mv'] + \
                        0.3 * df['exp_smoothed_vs'] + \
                        0.2 * df['relative_strength'] + \
                        0.1 * df['macroeconomic_factor']
    
    # Cross-Sectional Rank Adjustment
    df['rank'] = df.groupby(df.index)['final_alpha'].rank(pct=True)
    
    return df['rank']
```
