```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema5 = open_to_close_return.ewm(span=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) * df['volume']
    volume_weighted_ema5 = volume_weighted_return.ewm(span=5).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    average_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / average_volume_20

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.3 * normalized_range + 0.4 * open_to_close_ema5 + 0.3 * volume_weighted_ema5
    combined_momentum_volatility_ema = combined_momentum_volatility.ewm(alpha=0.1).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * price_volume_trend + 0.3 * volume_weighted_ema5 + 0.3 * volume_adjusted_intraday
    volume_sensitive_momentum_ema = volume_sensitive_momentum.ewm(alpha=0.1).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_momentum_volatility_ema + 0.4 * volume_sensitive_momentum_ema
    final_alpha_factor_ema = final_alpha_factor.ewm(alpha=0.1).mean()

    return final_alpha_factor_ema
```
