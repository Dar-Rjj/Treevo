```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma5 = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = price_change * df['volume']
    price_volume_trend_30 = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * open_to_close_sma5
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * price_volume_trend_30 + 0.3 * volume_adjusted_intraday_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close = open_to_close_return * df['volume']
    volume_weighted_open_to_close_sma5 = volume_weighted_open_to_close.rolling(window=5).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.4 * combined_momentum_volatility_smoothed +
        0.3 * volume_sensitive_momentum_smoothed +
        0.3 * volume_weighted_open_to_close_sma5
    )
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean()
    
    return final_alpha_factor
```
