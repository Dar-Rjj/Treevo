```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    norm_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    ema_open_close = open_to_close_return.ewm(span=5, adjust=False).mean()

    # Volume-Weighted Open-to-Close Return
    vol_weighted_return = open_to_close_return * df['volume']
    ema_vol_weighted = vol_weighted_return.ewm(span=5, adjust=False).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    pvt = daily_price_change.rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    vol_adjusted_movement = intraday_movement / avg_volume_20

    # Combined Momentum and Volatility Factor
    combined_factor = (0.4 * norm_range + 0.3 * ema_open_close + 0.3 * ema_vol_weighted)
    smoothed_combined_factor = combined_factor.ewm(alpha=0.1, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = (0.4 * pvt + 0.3 * ema_vol_weighted + 0.3 * vol_adjusted_movement)
    smoothed_volume_sensitive_factor = volume_sensitive_factor.ewm(alpha=0.1, adjust=False).mean()

    # Final Alpha Factor
    final_alpha = (0.6 * smoothed_combined_factor + 0.4 * smoothed_volume_sensitive_factor)
    final_smoothed_alpha = final_alpha.ewm(alpha=0.1, adjust=False).mean()

    return final_smoothed_alpha
```
