```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_ret = (df['close'] / df['open'] - 1)
    open_to_close_sma = open_to_close_ret.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_trend = daily_price_change * df['volume']
    aggregated_pvt = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * open_to_close_sma
    smoothed_combined_mv = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * aggregated_pvt + 0.3 * volume_adjusted_movement
    smoothed_vsm = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * smoothed_combined_mv + 0.4 * smoothed_vsm
    final_smoothed_alpha = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_smoothed_alpha
```
