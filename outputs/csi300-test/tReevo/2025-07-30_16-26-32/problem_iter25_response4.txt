```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_return'] = df['open_to_close_return'].ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30_day'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20_day'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['sma_volume_20_day']
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.3 * df['daily_range'] + 0.4 * df['ema_open_to_close_return'] + 0.3 * df['ema_volume_weighted_return']
    combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * df['pvt_30_day'] + 0.3 * df['ema_volume_weighted_return'] + 0.3 * df['volume_adjusted_intraday_movement']
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.2, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility + 0.5 * volume_sensitive_momentum
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean()
    
    return final_alpha_factor
```
