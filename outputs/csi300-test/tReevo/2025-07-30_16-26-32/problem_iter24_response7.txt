import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema = open_to_close_return.ewm(span=5, min_periods=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    volume_weighted_ema = volume_weighted_return.ewm(span=5, min_periods=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30, min_periods=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20, min_periods=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20
    
    # Combined Momentum and Volatility Factor
    def dynamic_weights_momentum_volatility(df):
        recent_volatility = df['high'].rolling(window=10, min_periods=10).std()
        recent_momentum = open_to_close_return.rolling(window=10, min_periods=10).mean()
        recent_volume_trend = df['volume'].rolling(window=10, min_periods=10).mean()
        
        w1 = 1 / (1 + recent_volatility)
        w2 = 1 / (1 + np.abs(recent_momentum))
        w3 = 1 / (1 + recent_volume_trend)
        
        w_sum = w1 + w2 + w3
        w1 /= w_sum
        w2 /= w_sum
        w3 /= w_sum
        
        return w1, w2, w3
    
    w1, w2, w3 = dynamic_weights_momentum_volatility(df)
    combined_momentum_volatility = w1 * daily_range + w2 * open_to_close_ema + w3 * volume_weighted_ema
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    def dynamic_weights_volume_sensitive(df):
        price_volume_corr = df['close'].pct_change().rolling(window=10, min_periods=10).corr(df['volume'].pct_change())
        recent_volume_trend = df['volume'].rolling(window=10, min_periods=10).mean()
        recent_intraday_volume = volume_adjusted_intraday.rolling(window=10, min_periods=10).mean()
        
        w4 = 1 / (1 + np.abs(price_volume_corr))
        w5 = 1 / (1 + recent_volume_trend)
        w6 = 1 / (1 + recent_intraday_volume)
        
        w_sum = w4 + w5 + w6
        w4 /= w_sum
        w5 /= w_sum
        w6 /= w_sum
        
        return w4, w5, w6
    
    w4, w5, w6 = dynamic_weights_volume_sensitive(df)
    volume_sensitive_momentum = w4 * price_volume_trend + w5 * volume_weighted_ema + w6 * volume_adjusted_intraday
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    def dynamic_weights_final(df):
        recent_momentum_volatility = combined_momentum_volatility.rolling(window=10, min_periods=10).mean()
        recent_volume_sensitivity = volume_sensitive_momentum.rolling(window=10, min_periods=10).mean()
        
        f7 = 1 / (1 + recent_momentum_volatility)
        f8 = 1 / (1 + recent_volume_sensitivity)
        
        f_sum = f7 + f8
        f7 /= f_sum
        f8 /= f_sum
        
        return f7, f8
    
    f7, f8 = dynamic_weights_final(df)
    final_alpha_factor = f7 * combined_momentum_volatility_smoothed + f8 * volume_sensitive_momentum_smoothed
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_factor_smoothed
