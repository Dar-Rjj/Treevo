```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Range (High - Low)
    daily_range = df['high'] - df['low']
    
    # Open to Close Momentum (Close - Open) / Open
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close_return = (open_to_close_return * df['volume']).ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff().fillna(0)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_20_day_volume
    
    # Combined Momentum and Volatility Factor
    combined_factor = 0.4 * daily_range + 0.3 * open_to_close_momentum + 0.3 * volume_weighted_open_to_close_return
    combined_factor = combined_factor.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * price_volume_trend + 0.3 * volume_weighted_open_to_close_return + 0.2 * volume_adjusted_intraday_movement
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha = 0.6 * combined_factor + 0.4 * volume_sensitive_momentum
    final_alpha = final_alpha.ewm(alpha=0.1).mean()
    
    return final_alpha
```
