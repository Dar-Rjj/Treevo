import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = (df['price_change'] * df['volume']).rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday'] = df['intraday_movement'] / df['sma_volume_20']
    
    # Combined Momentum and Volatility Factor
    weights_momentum_volatility = {
        'daily_range': 0.3,
        'ema_open_to_close': 0.3,
        'ema_volume_weighted_return': 0.4
    }
    df['combined_momentum_volatility'] = (
        df['daily_range'] * weights_momentum_volatility['daily_range'] +
        df['ema_open_to_close'] * weights_momentum_volatility['ema_open_to_close'] +
        df['ema_volume_weighted_return'] * weights_momentum_volatility['ema_volume_weighted_return']
    )
    df['ema_combined_momentum_volatility'] = df['combined_momentum_volatility'].ewm(span=5).mean()
    
    # Volume-Sensitive Momentum Factor
    weights_volume_sensitive = {
        'price_volume_trend': 0.3,
        'ema_volume_weighted_return': 0.3,
        'volume_adjusted_intraday': 0.4
    }
    df['volume_sensitive_momentum'] = (
        df['price_volume_trend'] * weights_volume_sensitive['price_volume_trend'] +
        df['ema_volume_weighted_return'] * weights_volume_sensitive['ema_volume_weighted_return'] +
        df['volume_adjusted_intraday'] * weights_volume_sensitive['volume_adjusted_intraday']
    )
    df['ema_volume_sensitive_momentum'] = df['volume_sensitive_momentum'].ewm(span=5).mean()
    
    # Additional Volatility Measure
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['historical_volatility'] = df['log_returns'].rolling(window=10).std()
    
    # Final Alpha Factor
    weights_final = {
        'ema_combined_momentum_volatility': 0.4,
        'ema_volume_sensitive_momentum': 0.4,
        'historical_volatility': 0.2
    }
    df['final_alpha_factor'] = (
        df['ema_combined_momentum_volatility'] * weights_final['ema_combined_momentum_volatility'] +
        df['ema_volume_sensitive_momentum'] * weights_final['ema_volume_sensitive_momentum'] +
        df['historical_volatility'] * weights_final['historical_volatility']
    )
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(span=5).mean()
    
    return df['final_alpha_factor']
