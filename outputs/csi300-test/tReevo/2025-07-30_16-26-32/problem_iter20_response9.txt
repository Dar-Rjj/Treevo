import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    adj_daily_range = daily_range / df['open']

    # Open to Close Momentum
    oc_return = (df['close'] - df['open']) / df['open']
    oc_ema_5 = oc_return.ewm(span=5).mean()

    # Volume-Weighted Open-to-Close Return
    vol_weighted_oc = oc_return * df['volume']
    vol_weighted_oc_ema_5 = vol_weighted_oc.ewm(span=5).mean()

    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    pv_trend = (price_change * df['volume']).rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    vol_adj_intraday = intraday_movement / avg_vol_20

    # Combined Momentum and Volatility Factor
    combined_factor = (0.4 * adj_daily_range + 0.3 * oc_ema_5 + 0.3 * vol_weighted_oc_ema_5).ewm(alpha=0.1).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = (0.4 * pv_trend + 0.3 * vol_weighted_oc_ema_5 + 0.3 * vol_adj_intraday).ewm(alpha=0.1).mean()

    # Final Alpha Factor
    final_alpha = (0.6 * combined_factor + 0.4 * volume_sensitive_momentum).ewm(alpha=0.1).mean()

    return final_alpha
