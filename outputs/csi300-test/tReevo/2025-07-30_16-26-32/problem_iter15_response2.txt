```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.rolling(window=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = open_to_close_return * df['volume']
    volume_weighted_open_to_close = volume_weighted_return.rolling(window=5).mean()

    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_factor = (0.4 * intraday_high_low_range + 
                       0.3 * open_to_close_momentum + 
                       0.3 * volume_weighted_open_to_close)
    combined_factor_smoothed = combined_factor.ewm(span=5, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = (0.7 * price_volume_trend + 
                                 0.3 * volume_weighted_open_to_close)
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = (0.6 * combined_factor_smoothed + 
                          0.4 * volume_sensitive_momentum_smoothed)
    final_alpha_factor_smoothed = final_alpha_factor.ewm(span=5, adjust=False).mean()

    return final_alpha_factor_smoothed
```
