```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema5 = open_to_close_return.ewm(span=5, adjust=False).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close_return = open_to_close_return * df['volume']
    volume_weighted_open_to_close_ema5 = volume_weighted_open_to_close_return.ewm(span=5, adjust=False).mean()

    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    price_volume_trend = price_change * df['volume']
    pv_trend_30 = price_volume_trend.rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    volume_sma20 = df['volume'].rolling(window=20).mean()
    adjusted_intraday_movement = intraday_movement / volume_sma20

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * normalized_range + 0.3 * open_to_close_ema5 + 0.3 * volume_weighted_open_to_close_ema5
    combined_momentum_volatility_ema = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * pv_trend_30 + 0.3 * volume_weighted_open_to_close_ema5 + 0.3 * adjusted_intraday_movement
    volume_sensitive_momentum_ema = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility_ema + 0.5 * volume_sensitive_momentum_ema
    final_alpha_factor_ema = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()

    return final_alpha_factor_ema
```
