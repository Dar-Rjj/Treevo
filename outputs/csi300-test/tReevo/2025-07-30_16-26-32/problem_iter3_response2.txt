```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smoothed_open_to_close = open_to_close_return.ewm(span=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff().fillna(0)
    price_volume_trend = price_change * df['volume']
    aggregated_price_volume_trend = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * smoothed_open_to_close
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * aggregated_price_volume_trend + 0.3 * volume_adjusted_movement
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * smoothed_combined_momentum_volatility + 0.5 * smoothed_volume_sensitive_momentum
    final_smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_smoothed_alpha_factor
```
