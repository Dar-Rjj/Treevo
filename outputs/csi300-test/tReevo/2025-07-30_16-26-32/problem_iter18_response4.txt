```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_oc_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_oc'] = df['volume_weighted_oc_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['daily_price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['daily_price_change'] * df['volume']
    df['pv_trend_30_day'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = (df['close'] - df['open'])
    df['avg_volume_20_day'] = df['volume'].rolling(window=20).mean()
    df['vol_adj_intraday_move'] = df['intraday_movement'] / df['avg_volume_20_day']
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.4 * df['daily_range'] + 0.3 * df['ema_open_to_close'] + 0.3 * df['ema_volume_weighted_oc']
    df['combined_mom_vol'] = combined_mom_vol.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_mom = 0.4 * df['pv_trend_30_day'] + 0.3 * df['ema_volume_weighted_oc'] + 0.3 * df['vol_adj_intraday_move']
    df['volume_sensitive_mom'] = volume_sensitive_mom.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = 0.6 * df['combined_mom_vol'] + 0.4 * df['volume_sensitive_mom']
    df['final_alpha_factor'] = final_alpha.ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor']
```
