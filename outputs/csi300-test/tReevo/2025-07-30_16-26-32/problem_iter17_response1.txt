```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_daily_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close = open_to_close_return.ewm(span=5, min_periods=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close = open_to_close_return * df['volume']
    smooth_volume_weighted_open_to_close = volume_weighted_open_to_close.ewm(span=5, min_periods=5).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30, min_periods=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20day_volume = df['volume'].rolling(window=20, min_periods=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_20day_volume

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * normalized_daily_range + 0.3 * smooth_open_to_close + 0.3 * smooth_volume_weighted_open_to_close
    smooth_combined_momentum_volatility = combined_momentum_volatility.ewm(span=5, min_periods=5).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * price_volume_trend + 0.3 * smooth_volume_weighted_open_to_close + 0.3 * volume_adjusted_intraday_movement
    smooth_volume_sensitive_momentum = volume_sensitive_momentum.ewm(span=5, min_periods=5).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.5 * smooth_combined_momentum_volatility + 0.5 * smooth_volume_sensitive_momentum
    final_smooth_alpha_factor = final_alpha_factor.ewm(span=5, min_periods=5).mean()

    return final_smooth_alpha_factor
```
