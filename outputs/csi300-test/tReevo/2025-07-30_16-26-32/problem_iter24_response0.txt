```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['smoothed_open_to_close_return'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['smoothed_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['price_volume_trend_30d'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['volume_20d_sma'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['volume_20d_sma']
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * df['daily_range'] + 0.3 * df['smoothed_open_to_close_return'] + 0.3 * df['smoothed_volume_weighted_return']
    df['combined_momentum_volatility'] = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * df['price_volume_trend_30d'] + 0.3 * df['smoothed_volume_weighted_return'] + 0.3 * df['volume_adjusted_intraday_movement']
    df['volume_sensitive_momentum'] = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Sentiment and Economic Composite Factor
    # Assuming external data is available in the DataFrame
    df['market_sentiment_score'] = df['external_market_sentiment_data']
    df['economic_impact_score'] = df['external_economic_data']
    sentiment_economic_factor = 0.5 * df['market_sentiment_score'] + 0.5 * df['economic_impact_score']
    df['sentiment_economic_factor'] = sentiment_economic_factor.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * df['combined_momentum_volatility'] + 0.3 * df['volume_sensitive_momentum'] + 0.1 * df['sentiment_economic_factor']
    df['final_alpha_factor'] = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return df['final_alpha_factor']
```
