```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_close_return'] = df['close'] - df['open']
    df['sma_open_close_return'] = df['open_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['avg_volume_20']
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['price_volume_trend_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Price Volatility Factor
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    df['volatility'] = df['log_returns'].rolling(window=20).std()
    
    # Combined Momentum, Volatility, and Volume Factor
    combined_factor = 0.4 * df['daily_range'] + 0.3 * df['sma_open_close_return'] + 0.3 * df['volatility']
    combined_factor = combined_factor.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * df['price_volume_trend_30'] + 0.3 * df['volume_adjusted_intraday_movement']
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_factor + 0.4 * volume_sensitive_momentum
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_alpha_factor
```
