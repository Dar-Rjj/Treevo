```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Ratio
    df['daily_range'] = df['high'] - df['low']
    df['intraday_high_to_low_ratio'] = df['daily_range'] / df['open']

    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_sma_5'] = df['open_to_close_return'].rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['avg_volume_20']

    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()

    # Momentum and Volume Combined Factor
    df['momentum_factor_10'] = df['close'].pct_change(periods=10)
    df['volume_factor_10'] = df['volume'].pct_change(periods=10)
    df['combined_factor'] = 0.7 * df['momentum_factor_10'] + 0.3 * df['volume_factor_10']

    # Cumulative Return Over Volume Surges
    df['volume_change'] = df['volume'].pct_change()
    df['vol_change_ma_10'] = df['volume_change'].rolling(window=10).mean()
    df['vol_change_std_10'] = df['volume_change'].rolling(window=10).std()
    df['surge_threshold'] = df['vol_change_ma_10'] + 2 * df['vol_change_std_10']
    surge_days = df[df['volume_change'] > df['surge_threshold']].index
    df['close_to_close_return'] = df['close'].pct_change()
    cumulative_returns = df.loc[surge_days, 'close_to_close_return'].expanding().sum()
    df['cumulative_factor'] = cumulative_returns.reindex(df.index).fillna(0) * np.exp(-np.arange(len(df)) / 60)

    # Final Alpha Factor
    alpha_factor = (
        df['intraday_high_to_low_ratio'] +
        df['open_to_close_sma_5'] +
        df['volume_adjusted_intraday_movement'] +
        df['pvt_30'] +
        df['combined_factor'] +
        df['cumulative_factor']
    )

    return alpha_factor
```
