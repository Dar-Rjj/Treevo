```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Ratio
    daily_range = df['high'] - df['low']
    intraday_high_low_ratio = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    open_to_close_momentum_sma = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume_20_day

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff().fillna(0)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combine all factors into a single alpha factor
    alpha_factor = (
        intraday_high_low_ratio +
        open_to_close_momentum_sma +
        volume_adjusted_intraday_movement +
        price_volume_trend
    ) / 4

    return alpha_factor
```
