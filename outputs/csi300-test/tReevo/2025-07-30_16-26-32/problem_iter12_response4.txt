```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    sma_5 = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1)
    price_volume_trend = price_change * df['volume']
    pvt_10 = price_volume_trend.rolling(window=10).sum()
    pvt_20 = price_volume_trend.rolling(window=20).sum()
    pvt_30 = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * daily_range + 0.5 * sma_5
    exponential_smoothing_alpha = 2 / (1 + 20)  # Simple volatility-based alpha
    combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=exponential_smoothing_alpha).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * pvt_10 + 0.5 * volume_adjusted_movement
    exponential_smoothing_alpha_volume = 2 / (1 + 50)  # Simple volume-based alpha
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=exponential_smoothing_alpha_volume).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility + 0.5 * volume_sensitive_momentum
    final_alpha_factor = final_alpha_factor.ewm(span=20, adjust=False).mean()
    
    return final_alpha_factor
```
