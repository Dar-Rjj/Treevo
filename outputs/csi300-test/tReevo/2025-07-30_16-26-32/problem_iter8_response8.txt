```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_return = df['close'] / df['open'] - 1
    smooth_open_close_return = open_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open']) / df['volume'].rolling(window=30).mean()
    
    # Price-Volume Trend Indicator
    pvt = (df['close'].diff() * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * daily_range + 0.5 * smooth_open_close_return
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * pvt + 0.5 * intraday_movement
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * smoothed_combined_momentum_volatility + 0.5 * smoothed_volume_sensitive_momentum
    final_smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_smoothed_alpha_factor
```
