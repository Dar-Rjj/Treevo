```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    otc_5d_ema = open_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    vol_weighted_otc_return = (df['close'] - df['open']) * df['volume']
    vwo_5d_ema = vol_weighted_otc_return.ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    pvt = (price_change * df['volume']).rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    adjusted_intraday_movement = intraday_movement / avg_vol_20d
    
    # Combined Momentum and Volatility Factor
    adaptive_weights_momentum_volatility = np.array([1/3, 1/3, 1/3])
    combined_momentum_volatility = (daily_range * adaptive_weights_momentum_volatility[0] +
                                    otc_5d_ema * adaptive_weights_momentum_volatility[1] +
                                    vwo_5d_ema * adaptive_weights_momentum_volatility[2])
    cmsmooth = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    adaptive_weights_volume_sensitive = np.array([1/3, 1/3, 1/3])
    volume_sensitive_momentum = (pvt * adaptive_weights_volume_sensitive[0] +
                                 vwo_5d_ema * adaptive_weights_volume_sensitive[1] +
                                 adjusted_intraday_movement * adaptive_weights_volume_sensitive[2])
    vssmooth = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (cmsmooth * 0.5 + vssmooth * 0.5)
    final_smooth = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_smooth
```
