```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma_5 = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_vol_20

    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * normalized_range + 0.5 * open_to_close_sma_5
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * price_volume_trend + 0.5 * volume_adjusted_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()

    # Trend Following Factor
    sma_50 = df['close'].rolling(window=50).mean()
    sma_200 = df['close'].rolling(window=200).mean()
    trend_following_factor = sma_50 - sma_200

    # Final Alpha Factor
    final_alpha = 0.4 * combined_momentum_volatility_smoothed + \
                  0.3 * volume_sensitive_momentum_smoothed + \
                  0.3 * trend_following_factor
    final_alpha_smoothed = final_alpha.ewm(alpha=0.1, adjust=False).mean()

    return final_alpha_smoothed
```
