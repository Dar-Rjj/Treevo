```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    intraday_momentum = daily_range.rolling(window=5).mean()

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma = open_to_close_return.rolling(window=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    volume_weighted_sma = volume_weighted_return.rolling(window=5).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    pvt_30d = daily_price_change.rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_factor = 0.4 * daily_range + 0.3 * open_to_close_sma + 0.3 * volume_weighted_sma
    combined_factor = combined_factor.ewm(alpha=0.2, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = 0.7 * pvt_30d + 0.3 * volume_weighted_sma
    volume_sensitive_factor = volume_sensitive_factor.ewm(alpha=0.2, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_factor + 0.4 * volume_sensitive_factor
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean()

    return final_alpha_factor
```
