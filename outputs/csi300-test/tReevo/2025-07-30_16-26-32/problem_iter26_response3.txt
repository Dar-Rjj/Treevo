```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = open_to_close_return * df['volume']
    volume_weighted_ema = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20
    
    # Combined Momentum and Volatility Factor
    combined_factor = 0.3 * intraday_high_low_range + 0.4 * open_to_close_momentum + 0.3 * volume_weighted_ema
    combined_factor_smoothed = combined_factor.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = 0.4 * price_volume_trend + 0.3 * volume_weighted_ema + 0.3 * volume_adjusted_intraday
    volume_sensitive_factor_smoothed = volume_sensitive_factor.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha = 0.5 * combined_factor_smoothed + 0.5 * volume_sensitive_factor_smoothed
    final_alpha_smoothed = final_alpha.ewm(alpha=0.1).mean()
    
    return final_alpha_smoothed
```
