```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_ema5'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['volume_weighted_ema5'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['daily_price_change'] = df['close'].diff().fillna(0)
    df['price_volume_trend'] = df['daily_price_change'] * df['volume']
    df['pvt_30d'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['volume_20d_sma'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_movement'] = df['intraday_movement'] / df['volume_20d_sma']
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.4 * df['daily_range'] + 0.3 * df['open_to_close_ema5'] + 0.3 * df['volume_weighted_ema5']
    df['combined_momentum_volatility_smoothed'] = df['combined_momentum_volatility'].ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.4 * df['pvt_30d'] + 0.3 * df['volume_weighted_ema5'] + 0.3 * df['volume_adjusted_movement']
    df['volume_sensitive_momentum_smoothed'] = df['volume_sensitive_momentum'].ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.6 * df['combined_momentum_volatility_smoothed'] + 0.4 * df['volume_sensitive_momentum_smoothed']
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.1).mean()
    
    return df['final_alpha_factor']
```
