```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.ewm(span=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume_20d

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * intraday_high_low_range + 0.5 * open_to_close_momentum
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(span=10).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * price_volume_trend + 0.5 * volume_adjusted_intraday_movement
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(span=10).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.5 * smoothed_combined_momentum_volatility + 0.5 * smoothed_volume_sensitive_momentum
    final_alpha_factor_smoothed = final_alpha_factor.ewm(span=10).mean()

    return final_alpha_factor_smoothed
```
