```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_range_ratio = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    open_to_close_momentum = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_vol_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_vol_20

    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = (0.5 * intraday_range_ratio + 0.5 * open_to_close_momentum).ewm(alpha=0.2).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = (0.5 * price_volume_trend + 0.5 * volume_adjusted_intraday_movement).ewm(alpha=0.2).mean()

    # Final Alpha Factor
    final_alpha_factor = (0.5 * combined_momentum_volatility + 0.5 * volume_sensitive_momentum).ewm(alpha=0.2).mean()

    return final_alpha_factor
```
