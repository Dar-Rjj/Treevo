import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['smoothed_open_to_close_return'] = df['open_to_close_return'].ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['smoothed_volume_weighted_return'] = df['volume_weighted_return'].ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pv_trend_30_days'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_20d_volume'] = df['volume'].rolling(window=20).mean()
    df['adjusted_intraday_movement'] = df['intraday_movement'] / df['avg_20d_volume']
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * df['daily_range'] + 0.3 * df['smoothed_open_to_close_return'] + 0.3 * df['smoothed_volume_weighted_return']
    df['combined_momentum_volatility'] = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * df['pv_trend_30_days'] + 0.3 * df['smoothed_volume_weighted_return'] + 0.3 * df['adjusted_intraday_movement']
    df['volume_sensitive_momentum'] = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = 0.6 * df['combined_momentum_volatility'] + 0.4 * df['volume_sensitive_momentum']
    df['final_alpha'] = final_alpha.ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha']
