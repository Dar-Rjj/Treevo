```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    df['normalized_range'] = df['daily_range'] / df['open']

    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['smoothed_open_to_close_return'] = df['open_to_close_return'].rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume'] = df['volume'].rolling(window=10).mean()  # Using 10-day average volume
    df['volume_adjusted_movement'] = df['intraday_movement'] / df['avg_volume']

    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].pct_change()
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pv_trend_30d'] = df['price_volume_trend'].rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.6 * df['normalized_range'] + 0.4 * df['smoothed_open_to_close_return']
    df['combined_mom_vol_exp_smoothed'] = combined_mom_vol.ewm(alpha=0.2, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_mom = 0.7 * df['pv_trend_30d'] + 0.3 * df['volume_adjusted_movement']
    df['volume_sensitive_mom_exp_smoothed'] = volume_sensitive_mom.ewm(alpha=0.2, adjust=False).mean()

    # Final Alpha Factor
    final_alpha = 0.6 * df['combined_mom_vol_exp_smoothed'] + 0.4 * df['volume_sensitive_mom_exp_smoothed']
    df['final_alpha_factor'] = final_alpha.ewm(alpha=0.2, adjust=False).mean()

    return df['final_alpha_factor']
```
