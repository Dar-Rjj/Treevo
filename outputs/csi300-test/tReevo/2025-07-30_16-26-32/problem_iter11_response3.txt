```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    smoothed_open_to_close_momentum = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_20_day_volume

    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['open']
    price_volume_trend = daily_price_change * df['volume']
    aggregated_price_volume_trend = price_volume_trend.rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_factor = (0.6 * intraday_high_low_range) + (0.4 * smoothed_open_to_close_momentum)
    combined_factor_smoothed = combined_factor.ewm(alpha=0.2, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = (0.7 * aggregated_price_volume_trend) + (0.3 * volume_adjusted_intraday_movement)
    volume_sensitive_factor_smoothed = volume_sensitive_factor.ewm(alpha=0.2, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = (0.6 * combined_factor_smoothed) + (0.4 * volume_sensitive_factor_smoothed)
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean()

    return final_alpha_factor_smoothed
```
