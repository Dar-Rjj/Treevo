```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    open_to_close_momentum = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_20_day_volume
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = price_change * df['volume']
    aggregated_price_volume_trend = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    weights = {'range': 0.5, 'momentum': 0.5}  # Example dynamic weights
    combined_factor = (daily_range * weights['range']) + (open_to_close_momentum * weights['momentum'])
    combined_factor_smoothed = combined_factor.ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    weights_vol_sensitive = {'pv_trend': 0.6, 'vol_adj_mov': 0.4}  # Example dynamic weights
    vol_sensitive_factor = (aggregated_price_volume_trend * weights_vol_sensitive['pv_trend']) + (volume_adjusted_intraday_movement * weights_vol_sensitive['vol_adj_mov'])
    vol_sensitive_factor_smoothed = vol_sensitive_factor.ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_weights = {'combined': 0.7, 'vol_sensitive': 0.3}  # Example dynamic weights
    final_alpha = (combined_factor_smoothed * final_weights['combined']) + (vol_sensitive_factor_smoothed * final_weights['vol_sensitive'])
    final_alpha_smoothed = final_alpha.ewm(alpha=0.2, adjust=False).mean().ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_smoothed
```
