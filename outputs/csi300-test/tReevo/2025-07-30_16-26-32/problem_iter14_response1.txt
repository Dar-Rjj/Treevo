import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_5_open_to_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['avg_volume_20']
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff()
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume Weighted Momentum
    df['volume_weighted_momentum'] = df['open_to_close_return'] * df['volume']
    df['sma_5_volume_weighted_momentum'] = df['volume_weighted_momentum'].rolling(window=5).mean()
    
    # Market Sentiment Factor
    df['sentiment_score'] = 0.4 * (df['high'] - df['low']) + 0.3 * df['close'] + 0.3 * df['volume']
    df['exp_smoothing_sentiment'] = df['sentiment_score'].ewm(alpha=0.1, adjust=False).mean()
    
    # Combined Momentum and Volatility Factor
    df['combined_factor'] = 0.4 * df['daily_range'] + 0.3 * df['sma_5_open_to_close'] + 0.3 * df['sma_5_volume_weighted_momentum']
    df['exp_smoothing_combined'] = df['combined_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.7 * df['pvt_30'] + 0.3 * df['volume_adjusted_intraday_movement']
    df['exp_smoothing_volume_sensitive'] = df['volume_sensitive_momentum'].ewm(alpha=0.1, adjust=False).mean()
    
    # Dynamic Weighting Mechanism
    # This is a simplified version, in practice, you might use more sophisticated methods
    df['historical_return'] = df['close'].pct_change()
    df['dynamic_weight_1'] = df['historical_return'].rolling(window=20).mean()
    df['dynamic_weight_2'] = df['historical_return'].rolling(window=20).std()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (df['dynamic_weight_1'] * df['exp_smoothing_combined'] + 
                                df['dynamic_weight_2'] * df['exp_smoothing_volume_sensitive'] + 
                                (1 - df['dynamic_weight_1'] - df['dynamic_weight_2']) * df['exp_smoothing_sentiment'])
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor']
