```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume = df['volume'].rolling(window=30).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume

    # Price-Volume Trend Indicator
    price_change = df['close'].pct_change()
    price_volume_trend = price_change * df['volume']
    price_volume_trend_30d = price_volume_trend.rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.6 * intraday_high_low_range + 0.4 * open_to_close_momentum
    combined_mom_vol_smoothed = combined_mom_vol.ewm(span=10, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_mom = 0.7 * price_volume_trend_30d + 0.3 * volume_adjusted_intraday_movement
    volume_sensitive_mom_smoothed = volume_sensitive_mom.ewm(span=10, adjust=False).mean()

    # Final Alpha Factor
    alpha_factor = 0.6 * combined_mom_vol_smoothed + 0.4 * volume_sensitive_mom_smoothed
    final_alpha_factor = alpha_factor.ewm(span=10, adjust=False).mean()

    return final_alpha_factor
```
