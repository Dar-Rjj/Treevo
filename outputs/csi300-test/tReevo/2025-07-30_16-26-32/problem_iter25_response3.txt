```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_5'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_return_5'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_movement'] = df['intraday_movement'] / df['sma_volume_20']
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.3 * df['daily_range'] + 0.4 * df['ema_open_to_close_5'] + 0.3 * df['ema_volume_weighted_return_5']
    df['combined_momentum_volatility_smoothed'] = combined_momentum_volatility.ewm(alpha=0.5).mean()  # Adaptive smoothing
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * df['pvt_30'] + 0.3 * df['ema_volume_weighted_return_5'] + 0.3 * df['volume_adjusted_movement']
    df['volume_sensitive_momentum_smoothed'] = volume_sensitive_momentum.ewm(alpha=0.5).mean()  # Adaptive smoothing
    
    return df['volume_sensitive_momentum_smoothed']
```
