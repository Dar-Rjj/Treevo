import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close'] = df['open_to_close_return'].ewm(span=5).mean()
    
    # Volume Adjusted Intraday Movement
    average_volume = df['volume'].rolling(window=20).mean()
    df['intraday_movement'] = (df['close'] - df['open']) / average_volume
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff().fillna(0)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30_day_sum'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume Weighted Momentum
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted'] = df['volume_weighted_return'].ewm(span=5).mean()
    
    # Adaptive Momentum and Volatility Factor
    adaptive_weights = {
        'range_weight': 0.3,
        'momentum_weight': 0.4,
        'volume_weighted_momentum_weight': 0.3
    }
    df['adaptive_factor'] = (df['daily_range'] * adaptive_weights['range_weight'] +
                             df['ema_open_to_close'] * adaptive_weights['momentum_weight'] +
                             df['ema_volume_weighted'] * adaptive_weights['volume_weighted_momentum_weight'])
    df['adaptive_factor_smoothed'] = df['adaptive_factor'].ewm(alpha=0.1).mean()
    
    # Adaptive Volume-Sensitive Momentum Factor
    adaptive_weights_vol_sensitive = {
        'pvt_weight': 0.6,
        'intraday_movement_weight': 0.4
    }
    df['volume_sensitive_factor'] = (df['pvt_30_day_sum'] * adaptive_weights_vol_sensitive['pvt_weight'] +
                                     df['intraday_movement'] * adaptive_weights_vol_sensitive['intraday_movement_weight'])
    df['volume_sensitive_factor_smoothed'] = df['volume_sensitive_factor'].ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_weights = {
        'adaptive_momentum_volatility_weight': 0.5,
        'adaptive_volume_sensitive_weight': 0.5
    }
    df['final_alpha_factor'] = (df['adaptive_factor_smoothed'] * final_weights['adaptive_momentum_volatility_weight'] +
                                df['volume_sensitive_factor_smoothed'] * final_weights['adaptive_volume_sensitive_weight'])
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(alpha=0.1).mean()
    
    return df['final_alpha_factor_smoothed']
