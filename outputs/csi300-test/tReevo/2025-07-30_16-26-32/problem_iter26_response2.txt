```python
import pandas as pd
import numpy as np

def heuristics(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_5'] = df['open_to_close_return'].ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_open_to_close'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_open_to_close_5'] = df['volume_weighted_open_to_close'].ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    df['daily_price_change'] = df['close'].diff()
    df['price_volume_trend'] = df['daily_price_change'] * df['volume']
    df['pv_trend_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['sma_volume_20']
    
    # Dynamic Volatility Measure
    df['returns'] = df['close'].pct_change()
    df['volatility_20'] = df['returns'].rolling(window=20).std() * np.sqrt(252)
    
    # Market Sentiment Indicator
    df['ma_close_10'] = df['close'].rolling(window=10).mean()
    df['ma_volume_10'] = df['volume'].rolling(window=10).mean()
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = (
        0.3 * df['daily_range'] + 
        0.4 * df['ema_open_to_close_5'] + 
        0.3 * df['ema_volume_weighted_open_to_close_5']
    )
    df['combined_momentum_volatility_smooth'] = df['combined_momentum_volatility'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = (
        0.3 * df['pv_trend_30'] + 
        0.4 * df['ema_volume_weighted_open_to_close_5'] + 
        0.3 * df['volume_adjusted_intraday_movement']
    )
    df['volume_sensitive_momentum_smooth'] = df['volume_sensitive_momentum'].ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = (
        0.5 * df['combined_momentum_volatility_smooth'] + 
        0.5 * df['volume_sensitive_momentum_smooth']
    )
    df['final_alpha_factor'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor']
```
