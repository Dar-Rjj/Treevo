```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_close_momentum = (df['close'] - df['open'])
    open_close_momentum_sma5 = open_close_momentum.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_factor = 0.7 * daily_range + 0.3 * open_close_momentum_sma5
    combined_factor = combined_factor.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = 0.8 * price_volume_trend + 0.2 * volume_adjusted_movement
    volume_sensitive_factor = volume_sensitive_factor.ewm(alpha=0.1, adjust=False).mean()
    
    # Short-Term Trend Following Factor
    sma_20 = df['close'].rolling(window=20).mean()
    sma_50 = df['close'].rolling(window=50).mean()
    short_term_trend = sma_20 - sma_50
    
    # Final Alpha Factor
    final_factor = 0.5 * combined_factor + 0.4 * volume_sensitive_factor + 0.1 * short_term_trend
    final_factor = final_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_factor
```
