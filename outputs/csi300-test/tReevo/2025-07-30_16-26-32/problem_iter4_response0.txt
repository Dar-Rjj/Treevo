```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_low_range = daily_range / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma_5 = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = price_change * df['volume']
    pvt_30 = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.7 * intraday_high_low_range + 0.3 * open_to_close_sma_5
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.6 * pvt_30 + 0.4 * volume_adjusted_intraday_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Relative Strength Indicator
    close_benchmark = df['close'] / df['close'].iloc[0]  # Assuming a benchmark index starting at 1
    relative_strength = close_benchmark.rolling(window=14).mean()
    
    # Average True Range
    true_range = df[['high', 'low']].diff(axis=1).max(axis=1)
    true_range.iloc[0] = df['high'].iloc[0] - df['low'].iloc[0]
    atr_14 = true_range.rolling(window=14).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (
        0.3 * combined_momentum_volatility_smoothed +
        0.3 * volume_sensitive_momentum_smoothed +
        0.2 * relative_strength +
        0.2 * atr_14
    )
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()

    return final_alpha_factor
```
