```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_ret = (df['close'] - df['open']) / df['open']
    smoothed_open_to_close = open_to_close_ret.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open']) / df['volume']
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1)
    price_volume_trend = daily_price_change * df['volume']
    aggregated_pvt = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.5 * daily_range + 0.5 * smoothed_open_to_close
    exp_smoothed_mom_vol = combined_mom_vol.ewm(alpha=0.2, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    combined_vol_sens_mom = 0.5 * aggregated_pvt + 0.5 * volume_adjusted_intraday
    exp_smoothed_vol_sens_mom = combined_vol_sens_mom.ewm(alpha=0.2, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha = 0.5 * exp_smoothed_mom_vol + 0.5 * exp_smoothed_vol_sens_mom
    final_exp_smoothed = final_alpha.ewm(alpha=0.2, adjust=False).mean()
    
    return final_exp_smoothed
```
