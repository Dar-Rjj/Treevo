```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['sma_open_to_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday'] = df['intraday_movement'] / df['avg_volume_20d']
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff()
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30d'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume Weighted Momentum
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['sma_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=5).mean()
    
    # Combined Momentum and Volatility Factor
    df['combined_mom_vol'] = 0.4 * df['daily_range'] + 0.3 * df['sma_open_to_close'] + 0.3 * df['sma_volume_weighted_return']
    df['combined_mom_vol_smooth'] = df['combined_mom_vol'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_mom'] = 0.7 * df['pvt_30d'] + 0.3 * df['volume_adjusted_intraday']
    df['volume_sensitive_mom_smooth'] = df['volume_sensitive_mom'].ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.6 * df['combined_mom_vol_smooth'] + 0.4 * df['volume_sensitive_mom_smooth']
    df['final_alpha_factor_smooth'] = df['final_alpha_factor'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor_smooth']
```
