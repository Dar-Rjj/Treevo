import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smoothed_momentum = open_to_close_return.ewm(span=5, adjust=False).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    smoothed_volume_weighted_return = volume_weighted_return.ewm(span=5, adjust=False).mean()

    # Price-Volume Trend Indicator
    price_change = df['close'].diff()
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    adjusted_intraday_movement = intraday_movement / avg_volume_20

    # Combined Momentum and Volatility Factor
    # Assigning adaptive weights based on market sentiment (simplified using fixed weights for demonstration)
    weight_range = 0.4
    weight_momentum = 0.3
    weight_volume_weighted = 0.3
    combined_factor = (weight_range * normalized_range + 
                       weight_momentum * smoothed_momentum + 
                       weight_volume_weighted * smoothed_volume_weighted_return)

    # Volume-Sensitive Momentum Factor
    # Assigning adaptive weights based on market sentiment (simplified using fixed weights for demonstration)
    weight_pvt = 0.4
    weight_vol_weighted_ret = 0.3
    weight_adj_intraday_move = 0.3
    volume_sensitive_factor = (weight_pvt * price_volume_trend + 
                               weight_vol_weighted_ret * smoothed_volume_weighted_return + 
                               weight_adj_intraday_move * adjusted_intraday_movement)

    # Final Alpha Factor
    # Assigning adaptive weights based on market sentiment (simplified using fixed weights for demonstration)
    weight_combined = 0.6
    weight_volume_sensitive = 0.4
    final_factor = (weight_combined * combined_factor + 
                    weight_volume_sensitive * volume_sensitive_factor).ewm(alpha=0.1, adjust=False).mean()

    return final_factor
