```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smoothed_open_to_close = open_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open']) / df['volume'].rolling(window=20).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = price_change * df['volume']
    smoothed_price_volume_trend = price_volume_trend.ewm(span=30, adjust=False).mean()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * smoothed_open_to_close
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(span=30, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * smoothed_price_volume_trend + 0.3 * intraday_movement
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(span=30, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * smoothed_combined_momentum_volatility + 0.4 * smoothed_volume_sensitive_momentum
    final_alpha_factor = final_alpha_factor.ewm(span=30, adjust=False).mean()
    
    return final_alpha_factor
```
