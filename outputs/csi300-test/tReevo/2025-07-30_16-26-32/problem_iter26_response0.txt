```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_ema = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_open_to_close = open_to_close_return * df['volume']
    volume_weighted_open_to_close_ema = volume_weighted_open_to_close.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    price_volume_trend = daily_price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume_20_day
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.3 * daily_range + 0.4 * open_to_close_ema + 0.3 * volume_weighted_open_to_close_ema
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * price_volume_trend + 0.3 * volume_weighted_open_to_close_ema + 0.3 * volume_adjusted_intraday_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_momentum_volatility_smoothed + 0.5 * volume_sensitive_momentum_smoothed
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_alpha_factor_smoothed
```
