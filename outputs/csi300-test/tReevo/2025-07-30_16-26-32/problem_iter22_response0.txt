```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = df['high'] - df['low']

    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['ema_open_to_close_7'] = df['open_to_close_return'].ewm(span=7).mean()

    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_return'] = df['open_to_close_return'] * df['volume']
    df['ema_volume_weighted_return_7'] = df['volume_weighted_return'].ewm(span=7).mean()

    # Price-Volume Trend Indicator
    df['price_change'] = df['close'].diff()
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pv_trend_30'] = df['price_volume_trend'].rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['sma_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['sma_volume_20']

    # Combined Momentum and Volatility Factor
    adaptive_weights = df['open_to_close_return'].abs() * (df['volume'] / df['volume'].rolling(window=20).mean())
    df['combined_factor'] = (df['daily_range'] * adaptive_weights + 
                             df['ema_open_to_close_7'] * adaptive_weights + 
                             df['ema_volume_weighted_return_7'] * adaptive_weights) / 3
    df['ema_combined_factor'] = df['combined_factor'].ewm(span=7).mean()

    # Volume-Sensitive Momentum Factor
    adaptive_weights_pv = df['intraday_movement'].abs() * (df['volume'] / df['volume'].rolling(window=20).mean())
    df['volume_sensitive_momentum'] = (df['pv_trend_30'] * adaptive_weights_pv + 
                                       df['ema_volume_weighted_return_7'] * adaptive_weights_pv + 
                                       df['volume_adjusted_intraday_movement'] * adaptive_weights_pv) / 3
    df['ema_volume_sensitive_momentum'] = df['volume_sensitive_momentum'].ewm(span=7).mean()

    # Final Alpha Factor
    final_adaptive_weights = df['ema_combined_factor'].abs() * (df['volume'] / df['volume'].rolling(window=20).mean())
    df['final_alpha_factor'] = (df['ema_combined_factor'] * final_adaptive_weights + 
                                df['ema_volume_sensitive_momentum'] * final_adaptive_weights) / 2
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(span=7).mean()

    return df['final_alpha_factor_smoothed']
```
