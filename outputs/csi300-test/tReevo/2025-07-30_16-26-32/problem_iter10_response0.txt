```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma_5 = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open']) / df['volume']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = price_change * df['volume']
    price_volume_trend_30 = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    recent_volatility = daily_range.rolling(window=10).std()
    recent_momentum = open_to_close_return.rolling(window=10).mean()
    weight_daily_range = 1 / (1 + recent_volatility)
    weight_open_to_close = 1 - weight_daily_range
    combined_mom_vol = (weight_daily_range * daily_range) + (weight_open_to_close * open_to_close_sma_5)
    combined_mom_vol_smoothed = combined_mom_vol.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    recent_volume_trend = df['volume'].pct_change().rolling(window=10).mean()
    weight_price_vol_trend = 1 / (1 + np.abs(recent_volume_trend))
    weight_volume_intraday = 1 - weight_price_vol_trend
    volume_sensitive_mom = (weight_price_vol_trend * price_volume_trend_30) + (weight_volume_intraday * volume_adjusted_intraday)
    volume_sensitive_mom_smoothed = volume_sensitive_mom.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    recent_performance = combined_mom_vol.rolling(window=10).mean()
    recent_volume = df['volume'].rolling(window=10).mean()
    weight_combined_mom_vol = 1 / (1 + np.abs(recent_performance))
    weight_volume_sensitive_mom = 1 / (1 + np.abs(recent_volume))
    final_alpha = (weight_combined_mom_vol * combined_mom_vol_smoothed) + (weight_volume_sensitive_mom * volume_sensitive_mom_smoothed)
    final_alpha_smoothed = final_alpha.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_smoothed
```
