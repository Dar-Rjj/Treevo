```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_to_low_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma_5 = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume_20

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1)
    price_volume_trend = daily_price_change * df['volume']
    aggregated_price_volume_trend = price_volume_trend.rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.5 * intraday_high_to_low_range + 0.5 * open_to_close_sma_5
    combined_momentum_volatility_smoothed = combined_momentum_volatility.ewm(alpha=0.1).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.6 * aggregated_price_volume_trend + 0.4 * volume_adjusted_intraday_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(alpha=0.1).mean()

    # Market Trend Adjustment
    long_term_trend = df['close'].rolling(window=100).mean()
    short_term_trend = df['close'].rolling(window=20).mean()
    market_trend_difference = short_term_trend - long_term_trend

    # Final Alpha Factor
    final_alpha_factor = (
        0.4 * combined_momentum_volatility_smoothed +
        0.4 * volume_sensitive_momentum_smoothed +
        0.2 * market_trend_difference
    )
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1).mean()

    return final_alpha_factor_smoothed
```
