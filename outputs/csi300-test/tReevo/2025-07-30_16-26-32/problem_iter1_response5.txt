```python
import pandas as pd

def heuristics_v2(df, n=14):
    # Calculate Price Differences
    df['diff'] = df['close'].diff()
    
    # Separate Positive and Negative Gains
    df['pos_gain'] = df['diff'].apply(lambda x: max(x, 0))
    df['neg_gain'] = df['diff'].apply(lambda x: abs(min(x, 0)))
    
    # Calculate Weighted Sum of Positive and Negative Gains
    df['weighted_pos_gain'] = df['pos_gain'] * df['volume']
    df['weighted_neg_gain'] = df['neg_gain'] * df['volume']
    
    # Calculate Total Volume
    total_volume = df['volume'].rolling(window=n).sum()
    
    # Calculate Average Gain and Loss
    avg_weighted_pos_gain = df['weighted_pos_gain'].rolling(window=n).sum() / total_volume
    avg_weighted_neg_gain = df['weighted_neg_gain'].rolling(window=n).sum() / total_volume
    
    # Calculate Volume-Weighted Relative Strength (VRS)
    vrs = (avg_weighted_pos_gain / avg_weighted_neg_gain) + 1
    
    # Final VRSI Calculation
    vrsi = 100 - (100 / vrs)
    
    return vrsi.dropna()
```
