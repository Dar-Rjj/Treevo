```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_sma_5'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = (df['close'] - df['open'])
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday_movement'] = df['intraday_movement'] / df['avg_volume_20']
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    df['combined_momentum_volatility'] = 0.5 * df['daily_range'] + 0.5 * df['open_to_close_sma_5']
    df['combined_momentum_volatility_smoothed'] = df['combined_momentum_volatility'].ewm(alpha=0.2, adjust=False).mean()
    df['combined_momentum_volatility_smoothed'] = df['combined_momentum_volatility_smoothed'].ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.5 * df['pvt_30'] + 0.5 * df['volume_adjusted_intraday_movement']
    df['volume_sensitive_momentum_smoothed'] = df['volume_sensitive_momentum'].ewm(alpha=0.2, adjust=False).mean()
    df['volume_sensitive_momentum_smoothed'] = df['volume_sensitive_momentum_smoothed'].ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.5 * df['combined_momentum_volatility_smoothed'] + 0.5 * df['volume_sensitive_momentum_smoothed']
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor'].ewm(alpha=0.2, adjust=False).mean()
    df['final_alpha_factor_smoothed'] = df['final_alpha_factor_smoothed'].ewm(alpha=0.1, adjust=False).mean()
    
    return df['final_alpha_factor_smoothed']
```
