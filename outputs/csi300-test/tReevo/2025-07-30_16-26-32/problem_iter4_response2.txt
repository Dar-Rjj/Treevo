```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    range_to_open = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open']) - 1
    smoothed_momentum = open_to_close_return.ewm(span=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20d = df['volume'].rolling(window=20).mean()
    normalized_intraday_move = intraday_movement / avg_volume_20d

    # Price-Volume Trend Indicator
    daily_price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_factor = 0.5 * range_to_open + 0.5 * smoothed_momentum
    smoothed_combined_factor = combined_factor.ewm(span=10).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.5 * price_volume_trend + 0.5 * normalized_intraday_move
    smoothed_volume_momentum = volume_sensitive_momentum.ewm(span=10).mean()

    # Final Alpha Factor
    final_factor = 0.5 * smoothed_combined_factor + 0.5 * smoothed_volume_momentum
    final_smoothed_factor = final_factor.ewm(span=10).mean()

    return final_smoothed_factor
```
