```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    intraday_high_to_low = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] / df['open'] - 1)
    open_to_close_momentum = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_volume_20_day

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_trend = (daily_price_change * df['volume']).rolling(window=30).sum()

    # Combined Momentum and Volatility Factor
    combined_mom_vol = (intraday_high_to_low * 0.5 + open_to_close_momentum * 0.5).ewm(alpha=0.2).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_mom = (price_volume_trend * 0.6 + volume_adjusted_intraday_movement * 0.4).ewm(alpha=0.2).mean()

    # Final Alpha Factor
    final_alpha_factor = (combined_mom_vol * 0.7 + volume_sensitive_mom * 0.3).ewm(alpha=0.2).mean()

    return final_alpha_factor
```
