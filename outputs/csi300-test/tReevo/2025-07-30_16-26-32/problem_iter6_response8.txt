```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close = open_to_close_return.rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume
    
    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_trend = (price_change * df['volume']).rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.6 * daily_range + 0.4 * smooth_open_to_close
    exp_smooth_combined = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * price_volume_trend + 0.3 * volume_adjusted_movement
    exp_smooth_volume_sensitive = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Relative Strength Indicator
    # Assuming 'benchmark' is a column in the DataFrame representing the benchmark index close price
    relative_strength = df['close'] / df['benchmark']
    smooth_relative_strength = relative_strength.rolling(window=14).mean()
    
    # Average True Range
    true_range = pd.concat([
        df['high'] - df['low'],
        (df['high'] - df['close'].shift(1)).abs(),
        (df['low'] - df['close'].shift(1)).abs()
    ], axis=1).max(axis=1)
    average_true_range = true_range.rolling(window=14).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.4 * exp_smooth_combined + 0.3 * exp_smooth_volume_sensitive + 0.2 * smooth_relative_strength + 0.1 * average_true_range
    final_smooth_alpha = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_smooth_alpha
```
