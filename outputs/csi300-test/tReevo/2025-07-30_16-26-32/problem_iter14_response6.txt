```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    otc_return = (df['close'] - df['open']) / df['open']
    otc_sma_5 = otc_return.rolling(window=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_otc = otc_return * df['volume']
    vwo_otc_sma_5 = volume_weighted_otc.rolling(window=5).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1)
    price_volume_trend = daily_price_change * df['volume']
    pvt_30 = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * daily_range + 0.3 * otc_sma_5 + 0.3 * vwo_otc_sma_5
    combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * pvt_30 + 0.3 * vwo_otc_sma_5
    volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_momentum_volatility + 0.4 * volume_sensitive_momentum
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_factor
```
