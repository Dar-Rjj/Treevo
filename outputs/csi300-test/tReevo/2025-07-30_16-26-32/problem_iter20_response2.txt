```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_ret'] = (df['close'] - df['open']) / df['open']
    df['open_to_close_ema5'] = df['open_to_close_ret'].ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    df['volume_weighted_open_to_close'] = df['open_to_close_ret'] * df['volume']
    df['volume_weighted_ema5'] = df['volume_weighted_open_to_close'].ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['pvt_30d'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    df['volume_sma20'] = df['volume'].rolling(window=20).mean()
    df['adjusted_intraday_movement'] = df['intraday_movement'] / df['volume_sma20']
    
    # Combined Momentum and Volatility Factor
    df['combined_mom_vol'] = 0.4 * df['daily_range'] + 0.3 * df['open_to_close_ema5'] + 0.3 * df['volume_weighted_ema5']
    df['combined_mom_vol_ema'] = df['combined_mom_vol'].ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    df['volume_sensitive_momentum'] = 0.4 * df['pvt_30d'] + 0.3 * df['volume_weighted_ema5'] + 0.3 * df['adjusted_intraday_movement']
    df['volume_sensitive_momentum_ema'] = df['volume_sensitive_momentum'].ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    df['final_alpha_factor'] = 0.6 * df['combined_mom_vol_ema'] + 0.4 * df['volume_sensitive_momentum_ema']
    final_alpha_factor = df['final_alpha_factor'].ewm(alpha=0.1).mean()
    
    return final_alpha_factor
```
