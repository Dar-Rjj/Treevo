```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_sma = open_to_close_return.rolling(window=5).mean()

    # Volume Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20_day = df['volume'].rolling(window=20).mean()
    volume_adjusted_movement = intraday_movement / avg_volume_20_day

    # Price-Volume Trend Indicator
    price_change = df['close'] - df['close'].shift(1)
    price_volume_product = price_change * df['volume']
    price_volume_trend = price_volume_product.rolling(window=30).sum()

    # Volume Weighted Momentum
    volume_weighted_return = open_to_close_return * df['volume']
    volume_weighted_sma = volume_weighted_return.rolling(window=5).mean()

    # Combined Momentum and Volatility Factor
    combined_factor = 0.4 * normalized_range + 0.3 * open_to_close_sma + 0.3 * volume_weighted_sma
    combined_factor_smoothed = combined_factor.ewm(span=5, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * price_volume_trend + 0.3 * volume_adjusted_movement
    volume_sensitive_momentum_smoothed = volume_sensitive_momentum.ewm(span=5, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_factor_smoothed + 0.4 * volume_sensitive_momentum_smoothed
    final_alpha_factor_smoothed = final_alpha_factor.ewm(span=5, adjust=False).mean()

    return final_alpha_factor_smoothed
```
