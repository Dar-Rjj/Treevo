```python
import pandas as pd

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smooth_open_to_close_return = open_to_close_return.rolling(window=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) / df['open'] * df['volume']
    smooth_volume_weighted_return = volume_weighted_return.rolling(window=5).mean()
    
    # Price-Volume Trend Indicator
    price_change = df['close'].diff(1)
    price_volume_trend = price_change * df['volume']
    agg_price_volume_trend = price_volume_trend.rolling(window=30).sum()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * daily_range + 0.3 * smooth_open_to_close_return + 0.3 * smooth_volume_weighted_return
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * agg_price_volume_trend + 0.3 * smooth_volume_weighted_return
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * smoothed_combined_momentum_volatility + 0.4 * smoothed_volume_sensitive_momentum
    final_smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_smoothed_alpha_factor
```
