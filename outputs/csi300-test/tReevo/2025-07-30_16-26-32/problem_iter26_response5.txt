```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    normalized_range = daily_range / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    ema_open_to_close_5 = open_to_close_return.ewm(span=5, adjust=False).mean()
    
    # Volume-Weighted Open-to-Close Return
    vol_weighted_return = (df['close'] - df['open']) * df['volume']
    ema_vol_weighted_return_5 = vol_weighted_return.ewm(span=5, adjust=False).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1)
    price_vol_trend = (daily_price_change * df['volume']).rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday = intraday_movement / avg_volume_20
    
    # Combined Momentum and Volatility Factor
    combined_mom_vol = 0.4 * normalized_range + 0.3 * ema_open_to_close_5 + 0.3 * ema_vol_weighted_return_5
    combined_mom_vol_smoothed = combined_mom_vol.ewm(alpha=0.1, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    vol_sensitive_mom = 0.4 * price_vol_trend + 0.3 * ema_vol_weighted_return_5 + 0.3 * volume_adjusted_intraday
    vol_sensitive_mom_smoothed = vol_sensitive_mom.ewm(alpha=0.1, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.5 * combined_mom_vol_smoothed + 0.5 * vol_sensitive_mom_smoothed
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return final_alpha_factor_smoothed
```
