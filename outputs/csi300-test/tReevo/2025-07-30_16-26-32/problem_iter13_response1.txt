import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    df['daily_range'] = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    df['open_to_close_return'] = (df['close'] - df['open']) / df['open']
    df['smoothed_open_to_close'] = df['open_to_close_return'].rolling(window=5).mean()
    
    # Volume Adjusted Intraday Movement
    df['intraday_movement'] = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    df['volume_adjusted_intraday'] = df['intraday_movement'] / avg_volume_20
    
    # Price-Volume Trend Indicator
    df['price_change'] = df['close'] - df['close'].shift(1)
    df['price_volume_trend'] = df['price_change'] * df['volume']
    df['aggregated_pvt'] = df['price_volume_trend'].rolling(window=30).sum()
    
    # Volume Weighted Momentum
    df['volume_weighted_return'] = (df['close'] - df['open']) / df['open'] * df['volume']
    df['smoothed_volume_weighted'] = df['volume_weighted_return'].rolling(window=5).mean()
    
    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * df['daily_range'] + 0.3 * df['smoothed_open_to_close'] + 0.3 * df['smoothed_volume_weighted']
    df['combined_momentum_volatility'] = combined_momentum_volatility.ewm(span=5, adjust=False).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.7 * df['aggregated_pvt'] + 0.3 * df['volume_adjusted_intraday']
    df['volume_sensitive_momentum'] = volume_sensitive_momentum.ewm(span=5, adjust=False).mean()
    
    # Final Alpha Factor
    final_alpha_factor = 0.6 * df['combined_momentum_volatility'] + 0.4 * df['volume_sensitive_momentum']
    df['final_alpha_factor'] = final_alpha_factor.ewm(span=5, adjust=False).mean()
    
    return df['final_alpha_factor']
