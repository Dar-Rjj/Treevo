```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = df['high'] - df['low']
    range_adjusted_by_open = daily_range / df['open']

    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    otc_ema_5 = open_to_close_return.ewm(span=5).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_otc_return = open_to_close_return * df['volume']
    vw_otc_ema_5 = volume_weighted_otc_return.ewm(span=5).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff()
    price_volume_product = daily_price_change * df['volume']
    pvt_30 = price_volume_product.rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    va_im = intraday_movement / avg_volume_20

    # Combined Momentum and Volatility Factor
    combined_factor = 0.4 * range_adjusted_by_open + 0.3 * otc_ema_5 + 0.3 * vw_otc_ema_5
    combined_factor_smoothed = combined_factor.ewm(alpha=0.1).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = 0.4 * pvt_30 + 0.3 * vw_otc_ema_5 + 0.3 * va_im
    volume_sensitive_factor_smoothed = volume_sensitive_factor.ewm(alpha=0.1).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.6 * combined_factor_smoothed + 0.4 * volume_sensitive_factor_smoothed
    final_alpha_factor = final_alpha_factor.ewm(alpha=0.1).mean()

    return final_alpha_factor
```
