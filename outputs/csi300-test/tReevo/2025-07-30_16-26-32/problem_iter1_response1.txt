```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Reversal
    intraday_reversal = (df['high'] - df['low']) / (df['close'] - df['open'])
    
    # Compute Volume Acceleration
    df['volume_t-1'] = df['volume'].shift(1)
    df['volume_t-2'] = df['volume'].shift(2)
    volume_acceleration = (df['volume'] - 2 * df['volume_t-1'] + df['volume_t-2']) / df['volume_t-2']
    
    # Adjust for Sign Change in Close Price
    df['close_t-1'] = df['close'].shift(1)
    sign_change = (df['close'] > df['close_t-1']).astype(int) * 2 - 1
    
    # Combine Intraday Reversal and Volume Acceleration
    alpha_factor = sign_change * (intraday_reversal * volume_acceleration)
    
    return alpha_factor.dropna()
```
