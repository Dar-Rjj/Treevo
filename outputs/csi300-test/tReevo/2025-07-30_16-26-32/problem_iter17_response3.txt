```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    normalized_daily_range = (daily_range - daily_range.mean()) / daily_range.std()
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    smoothed_open_to_close_return = open_to_close_return.ewm(span=5, adjust=False).mean()

    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = open_to_close_return * df['volume']
    smoothed_volume_weighted_return = volume_weighted_return.ewm(span=5, adjust=False).mean()

    # Price-Volume Trend Indicator
    daily_price_change = df['close'].diff(1) * df['volume']
    price_volume_trend = daily_price_change.rolling(window=30).sum()

    # Volume-Adjusted Intraday Movement
    intraday_movement = (df['close'] - df['open'])
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    adjusted_intraday_movement = intraday_movement / avg_20_day_volume

    # Combined Momentum and Volatility Factor
    combined_momentum_volatility = 0.4 * normalized_daily_range + 0.3 * smoothed_open_to_close_return + 0.3 * smoothed_volume_weighted_return
    smoothed_combined_momentum_volatility = combined_momentum_volatility.ewm(alpha=0.2, adjust=False).mean()

    # Volume-Sensitive Momentum Factor
    volume_sensitive_momentum = 0.4 * price_volume_trend + 0.3 * smoothed_volume_weighted_return + 0.3 * adjusted_intraday_movement
    smoothed_volume_sensitive_momentum = volume_sensitive_momentum.ewm(alpha=0.2, adjust=False).mean()

    # Final Alpha Factor
    final_alpha_factor = 0.6 * smoothed_combined_momentum_volatility + 0.4 * smoothed_volume_sensitive_momentum
    final_smoothed_alpha_factor = final_alpha_factor.ewm(alpha=0.2, adjust=False).mean()

    return final_smoothed_alpha_factor
```
