```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Intraday High-to-Low Range
    daily_range = (df['high'] - df['low']) / df['open']
    
    # Open to Close Momentum
    open_to_close_return = (df['close'] - df['open']) / df['open']
    open_to_close_momentum = open_to_close_return.ewm(span=5).mean()
    
    # Volume-Weighted Open-to-Close Return
    volume_weighted_return = (df['close'] - df['open']) * df['volume']
    volume_weighted_open_to_close = volume_weighted_return.ewm(span=5).mean()
    
    # Price-Volume Trend Indicator
    daily_price_change = (df['close'] - df['close'].shift(1)) * df['volume']
    price_volume_trend = daily_price_change.rolling(window=30).sum()
    
    # Volume-Adjusted Intraday Movement
    intraday_movement = df['close'] - df['open']
    avg_20_day_volume = df['volume'].rolling(window=20).mean()
    volume_adjusted_intraday_movement = intraday_movement / avg_20_day_volume
    
    # Combined Momentum and Volatility Factor
    combined_factor = (0.4 * daily_range + 0.3 * open_to_close_momentum + 0.3 * volume_weighted_open_to_close)
    combined_factor_smoothed = combined_factor.ewm(alpha=0.1).mean()
    
    # Volume-Sensitive Momentum Factor
    volume_sensitive_factor = (0.4 * price_volume_trend + 0.3 * volume_weighted_open_to_close + 0.3 * volume_adjusted_intraday_movement)
    volume_sensitive_factor_smoothed = volume_sensitive_factor.ewm(alpha=0.1).mean()
    
    # Final Alpha Factor
    final_alpha_factor = (0.6 * combined_factor_smoothed + 0.4 * volume_sensitive_factor_smoothed)
    final_alpha_factor_smoothed = final_alpha_factor.ewm(alpha=0.1).mean()
    
    return final_alpha_factor_smoothed
```
