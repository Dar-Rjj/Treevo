```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Define recent and older weights
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}
    
    # Apply weights based on the recency of the data
    # For simplicity, we assume that the most recent 5 days are "recent" and the rest are "older"
    recent_mask = df.index.isin(df.index[-5:])
    df['weighted_intraday_range'] = df['intraday_range'] * (df.index.map(lambda x: recent_weights['intraday_range'] if x in df.index[-5:] else older_weights['intraday_range']))
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * (df.index.map(lambda x: recent_weights['close_to_open_return'] if x in df.index[-5:] else older_weights['close_to_open_return']))
    
    # Combine Weighted Intraday Range and Close-to-Open Return
    df['alpha_factor'] = df['weighted_intraday_range'] + df['weighted_close_to_open_return']
    
    return df['alpha_factor']
```
