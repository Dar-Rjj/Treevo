```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open'].shift(-1)) / df['close']
    
    # Combine Intraday Range and Close-to-Open Return
    combined_factor = close_to_open_return / intraday_range
    
    # Apply Exponentially Weighted Moving Average (EWMA) to Intraday Range
    ewma_span = 20
    ewma_intraday_range = intraday_range.ewm(span=ewma_span, min_periods=ewma_span).mean()
    
    # Adjust Factor Value with Adaptive Weighting
    adaptive_weighted_factor = combined_factor * ewma_intraday_range
    
    # Final Factor Value
    final_factor = adaptive_weighted_factor.shift(1)  # Shift by 1 day to predict future returns
    
    return final_factor
```
