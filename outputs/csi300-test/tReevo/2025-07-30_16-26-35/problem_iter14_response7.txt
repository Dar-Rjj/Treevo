import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Determine recent and older weights
    recent_weights = [0.6, 0.4]  # [Intraday Range, Close-to-Open Return]
    older_weights = [0.5, 0.5]   # [Intraday Range, Close-to-Open Return]
    
    # Combine Intraday Range and Close-to-Open Return with weighted combination
    def weighted_combination(row, recent=True):
        if recent:
            return row['intraday_range'] * recent_weights[0] + row['close_to_open_return'] * recent_weights[1]
        else:
            return row['intraday_range'] * older_weights[0] + row['close_to_open_return'] * older_weights[1]
    
    # Apply the weighting function to create the factor
    df['recent_factor'] = df.apply(weighted_combination, axis=1, recent=True)
    df['older_factor'] = df.apply(weighted_combination, axis=1, recent=False)
    
    # Determine the final factor by combining recent and older factors
    # Assuming 'recent' is defined as within the last 30 days for this example
    df['final_factor'] = df['recent_factor'].where(df.index.isin(df.index[-30:]), df['older_factor'])
    
    return df['final_factor']
