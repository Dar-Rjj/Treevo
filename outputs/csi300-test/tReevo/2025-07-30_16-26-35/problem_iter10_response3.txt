```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['prev_close'] = df['close'].shift(1)
    close_to_open_return = np.log(df['open'] / df['prev_close'])
    
    # Combine Intraday Range and Close-to-Open Return with weights
    recent_weights = 0.6 * intraday_range + 0.4 * close_to_open_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_open_return
    
    # Determine the factor values
    factor_values = pd.Series(index=df.index, dtype='float64')
    for i in range(len(df)):
        if i < len(df) // 2:
            factor_values.iloc[i] = recent_weights.iloc[i]
        else:
            factor_values.iloc[i] = older_weights.iloc[i]
    
    return factor_values
```
