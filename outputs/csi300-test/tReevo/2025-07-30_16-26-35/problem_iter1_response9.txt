```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = (df['high'] - df['low']) / df['low']
    
    # Weight by Trade Intensity
    trade_intensity = df['volume'] / df['amount']
    weighted_intraday_return = intraday_return * trade_intensity
    
    # Incorporate Close-to-PreviousClose Trend
    df['close_shifted'] = df['close'].shift(1)
    prev_close_to_close_return = (df['close'] - df['close_shifted']) / df['close_shifted']
    
    # Combine the results
    factor_values = weighted_intraday_return + prev_close_to_close_return
    factor_values = factor_values.dropna()  # Drop NaN values to ensure clean output
    
    return factor_values
```
