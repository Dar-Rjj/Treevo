```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return (logarithmic)
    df['close_to_open_return'] = np.log(df['close'] / df['open'])
    
    # Calculate Volume-Weighted Intraday Range
    df['volume_weighted_intraday_range'] = df['intraday_range'] * df['volume']
    
    # Calculate 20-Day Average of Volume-Weighted Intraday Range
    df['avg_20_day_volume_weighted_intraday_range'] = df['volume_weighted_intraday_range'].rolling(window=20).mean()
    
    # Calculate Logarithmic Momentum Score
    df['log_momentum_score'] = np.log(df['close']).diff().rolling(window=20).sum()
    
    # Normalize by 20-Day Average of Volume-Weighted Intraday Range
    df['normalized_log_momentum_score'] = df['log_momentum_score'] / df['avg_20_day_volume_weighted_intraday_range']
    
    # Combine Volume-Weighted Intraday Range and Logarithmic Momentum
    df['factor'] = df['normalized_log_momentum_score'] * df['close_to_open_return']
    
    # Drop rows with NaN values resulting from the rolling calculations
    df.dropna(inplace=True)
    
    return df['factor']
```
