import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['high'] - df['low']) / df['low']
    
    # Weight by Trade Intensity
    df['trade_intensity'] = df['volume'] / df['amount']
    df['weighted_intraday_return'] = df['intraday_return'] * df['trade_intensity']
    
    # Incorporate Close-to-PreviousClose Trend
    df['prev_close_to_close_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['trend_factor'] = df['weighted_intraday_return'] + df['prev_close_to_close_return']
    
    # Integrate Volatility
    df['daily_return'] = df['close'].pct_change()
    df['volatility'] = df['daily_return'].rolling(window=20).std()
    df['trade_intensity_volatility'] = df['trade_intensity'] * df['volatility']
    
    # Consider Multi-Period Returns
    df['short_term_ma'] = df['daily_return'].rolling(window=5).mean()
    df['long_term_ma'] = df['daily_return'].rolling(window=20).mean()
    
    # Final Factor Value
    df['combined_factor'] = df['trend_factor'] + df['volatility']
    df['multi_period_adjustment'] = (df['long_term_ma'] - df['short_term_ma']) * df['combined_factor']
    
    # Incorporate Long-Term Trends
    df['long_term_return'] = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    df['final_factor_value'] = df['multi_period_adjustment'] + df['long_term_return']
    
    return df['final_factor_value'].dropna()
