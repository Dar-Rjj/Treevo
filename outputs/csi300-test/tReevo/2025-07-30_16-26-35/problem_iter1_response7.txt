```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-to-Low Range
    daily_range = df['high'] - df['low']
    
    # Compute Weighted High-to-Low Range
    weighted_range = daily_range * df['volume']
    
    # Calculate Rolling Sum of Weighted High-to-Low Ranges over a 20-day Window
    rolling_sum = weighted_range.rolling(window=20).sum()
    
    # Normalize Rolling Sum by the Maximum of the 20-day Moving Sum
    normalized_rolling_sum = rolling_sum / rolling_sum.rolling(window=20).max()
    
    return normalized_rolling_sum
```
