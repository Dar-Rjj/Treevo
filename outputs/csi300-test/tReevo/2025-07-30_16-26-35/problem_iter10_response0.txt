import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range: High - Low
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: (Close - Open) / Open
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Shift the close price to get the previous day's close
    df['prev_close'] = df['close'].shift(1)
    df['close_to_open_return'] = (df['open'] - df['prev_close']) / df['prev_close']
    
    # Combine Intraday Range and Close-to-Open Return with recent weights
    df['factor_value'] = 0.7 * df['intraday_range'] + 0.3 * df['close_to_open_return']
    
    # Return the factor values as a pandas Series
    return df['factor_value']
