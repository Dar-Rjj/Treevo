```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range (High - Low)
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: ln(Close / Prev_Close)
    prev_close = df['close'].shift(1)
    close_to_open_return = np.log(df['open'] / prev_close)
    
    # Incorporate Volume Shock
    ma_volume = df['volume'].rolling(window=20).mean()
    volume_shock = (df['volume'] - ma_volume) / ma_volume
    
    # Combine Intraday Range, Close-to-Open Return, and Volume Shock
    # Assign weights to recent data
    recent_weights = [0.5, 0.3, 0.2]
    older_weights = [0.4, 0.4, 0.2]
    
    # Compute the weighted combination
    recent_combination = (
        recent_weights[0] * intraday_range + 
        recent_weights[1] * close_to_open_return + 
        recent_weights[2] * volume_shock
    )
    
    older_combination = (
        older_weights[0] * intraday_range + 
        older_weights[1] * close_to_open_return + 
        older_weights[2] * volume_shock
    )
    
    # Determine the final factor by combining recent and older combinations
    # Here we use a simple approach to combine them, but more sophisticated methods can be used
    factor = recent_combination + older_combination
    
    # Apply Exponential Weighting with a decay factor of 0.95
    decay_factor = 0.95
    days_ago = np.arange(len(factor))
    factor = factor * (decay_factor ** days_ago)
    
    return factor
```
