import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Define dynamic weights based on recency
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}

    # Define a function to adjust weights based on volume
    def adjust_weights(volume, avg_volume, base_weights):
        if volume > avg_volume:
            return {k: v * 1.1 for k, v in base_weights.items()}
        elif volume < avg_volume:
            return {k: v * 0.9 for k, v in base_weights.items()}
        else:
            return base_weights
    
    # Calculate average volume
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Combine Intraday Range and Close-to-Open Return with dynamic weighting
    factor_values = []
    for i in range(len(df)):
        if i < 20:  # Use recent weights for the first 20 days
            weights = recent_weights
        else:
            weights = adjust_weights(df.iloc[i]['volume'], avg_volume.iloc[i], older_weights)
        
        factor_value = (df.iloc[i]['intraday_range'] * weights['intraday_range'] +
                       df.iloc[i]['close_to_open_return'] * weights['close_to_open_return'])
        factor_values.append(factor_value)
    
    # Create a pandas Series with the factor values
    factor_series = pd.Series(factor_values, index=df.index, name='heuristics_v2_factor')
    
    return factor_series
