```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Assign weights for recent and older data
    recent_weight_intraday = 0.6
    older_weight_intraday = 0.4
    recent_weight_close_to_open = 0.3
    older_weight_close_to_open = 0.2
    recent_weight_vwap = 0.1
    older_weight_vwap = 0.4
    
    # Compute Weighted Intraday Range
    df['weighted_intraday_range'] = df['intraday_range'] * recent_weight_intraday
    df['weighted_intraday_range'] = df['weighted_intraday_range'].fillna(older_weight_intraday * df['intraday_range'])
    
    # Compute Weighted Close-to-Open Return
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * recent_weight_close_to_open
    df['weighted_close_to_open_return'] = df['weighted_close_to_open_return'].fillna(older_weight_close_to_open * df['close_to_open_return'])
    
    # Compute Weighted VWAP
    df['weighted_vwap'] = df['vwap'] * recent_weight_vwap
    df['weighted_vwap'] = df['weighted_vwap'].fillna(older_weight_vwap * df['vwap'])
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_intraday_range'] + 
                          df['weighted_close_to_open_return'] + 
                          df['weighted_vwap'])
    
    return df['alpha_factor']
```
