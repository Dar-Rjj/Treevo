```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Compute Weighted Intraday Range
    df['Weighted_Intraday_Range'] = df['Intraday_Range'] * 0.4
    
    # Compute Weighted Close-to-Open Return
    df['Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * 0.2
    
    # Compute Weighted VWAP
    df['Weighted_VWAP'] = df['VWAP'] * 0.2
    
    # Compute Weighted Momentum
    df['Weighted_Momentum'] = df['Momentum'] * 0.2
    
    # Combine all weighted factors to get the final alpha factor
    df['Alpha_Factor'] = (
        df['Weighted_Intraday_Range'] + 
        df['Weighted_Close_to_Open_Return'] + 
        df['Weighted_VWAP'] + 
        df['Weighted_Momentum']
    )
    
    return df['Alpha_Factor']
```
