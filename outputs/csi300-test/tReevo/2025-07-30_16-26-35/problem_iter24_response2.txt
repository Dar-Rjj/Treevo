```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Assign weights to recent data
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.6, 'close_to_open_return': 0.4}
    
    # Combine Intraday Range and Close-to-Open Return
    df['combined_factor'] = df['intraday_range'] * recent_weights['intraday_range'] + df['close_to_open_return'] * recent_weights['close_to_open_return']
    df['combined_factor'] = df['combined_factor'].fillna(df['intraday_range'] * older_weights['intraday_range'] + df['close_to_open_return'] * older_weights['close_to_open_return'])
    
    # Compute Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust Alpha Factor based on Volume Ratio
    df['adjusted_alpha_factor'] = df['combined_factor'] * (1 + (df['volume_ratio'] - 1) * 0.5)
    
    # Incorporate Volume Momentum
    df['volume_momentum'] = (df['volume_ratio'] - 1).clip(lower=0)
    df['final_alpha_factor'] = df['adjusted_alpha_factor'] + df['volume_momentum'] * 0.2
    
    return df['final_alpha_factor']
```
