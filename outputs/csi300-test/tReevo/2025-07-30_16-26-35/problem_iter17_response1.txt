```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['prev_close'] = df['close'].shift(1)
    df['close_to_open_return'] = (df['open'] - df['prev_close']) / df['prev_close']
    
    # Define the weights for recent and older data
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}

    # Determine the threshold to separate recent and older data
    threshold_index = len(df) // 2

    # Apply weights to recent data
    recent_factors = (
        df.iloc[threshold_index:]['intraday_range'] * recent_weights['intraday_range'] +
        df.iloc[threshold_index:]['close_to_open_return'] * recent_weights['close_to_open_return']
    )

    # Apply weights to older data
    older_factors = (
        df.iloc[:threshold_index]['intraday_range'] * older_weights['intraday_range'] +
        df.iloc[:threshold_index]['close_to_open_return'] * older_weights['close_to_open_return']
    )

    # Combine the factors
    combined_factors = pd.concat([older_factors, recent_factors]).sort_index()

    return combined_factors
```
