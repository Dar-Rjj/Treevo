```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Define the weights for recent and older data
    recent_weights = 0.7  # weight for intraday range in recent data
    older_weights = 0.5  # weight for intraday range in older data
    
    # Determine the cut-off point for recent vs older data
    cutoff_index = len(df) // 2
    
    # Initialize the factor series
    factor = pd.Series(index=df.index, dtype=float)
    
    # Calculate the factor for recent and older data
    factor.iloc[:cutoff_index] = (older_weights * intraday_range + (1 - older_weights) * close_to_open_return).iloc[:cutoff_index]
    factor.iloc[cutoff_index:] = (recent_weights * intraday_range + (1 - recent_weights) * close_to_open_return).iloc[cutoff_index:]
    
    return factor
```
