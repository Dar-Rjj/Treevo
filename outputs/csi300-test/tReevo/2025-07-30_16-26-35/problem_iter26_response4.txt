```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Define recent and older weights
    recent_intraday_range = df['intraday_range'].rolling(window=2).mean() * 0.6 + df['intraday_range'].shift(2).rolling(window=5).mean() * 0.4
    recent_close_to_open_return = df['close_to_open_return'].rolling(window=2).mean() * 0.4 + df['close_to_open_return'].shift(2).rolling(window=5).mean() * 0.3
    recent_vwap = df['vwap'].rolling(window=2).mean() * 0.2 + df['vwap'].shift(2).rolling(window=5).mean() * 0.5
    recent_intraday_momentum = df['intraday_momentum'].rolling(window=2).mean() * 0.2 + df['intraday_momentum'].shift(2).rolling(window=5).mean() * 0.5
    
    # Combine with dynamic weighting
    initial_alpha_factor = (recent_intraday_range + recent_close_to_open_return + recent_vwap + recent_intraday_momentum)
    
    # Incorporate Recency and Volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = initial_alpha_factor * (1 + (df['volume_ratio'] - 1) * 0.7)
    
    # Incorporate Advanced Volume Momentum
    volume_momentum = (df['volume_ratio'] - 1).clip(lower=0)
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.3
    
    return final_alpha_factor
```
