```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday Range and Close-to-Open Return with weights
    df['combined_factor'] = df['intraday_range'] * 0.7 + df['close_to_open_return'] * 0.3
    
    # Momentum and Volume Integration
    df['10_day_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    df['10_day_avg_volume'] = df['volume'].rolling(window=10).mean()
    
    # Final alpha factor
    df['alpha_factor'] = df['combined_factor'] + df['10_day_momentum'] + df['10_day_avg_volume']
    
    return df['alpha_factor']
```
