```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = (df['high'] - df['low']) / df['open']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Calculate VWAP
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Momentum
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Volume Impact
    avg_vol_5d = df['volume'].rolling(window=5).mean()
    avg_vol_20d = df['volume'].rolling(window=20).mean()
    volume_impact = df['volume'] / avg_vol_5d
    
    # Dynamic Weighting
    recent_weights = [0.4, 0.2, 0.2, 0.2, 0.2]  # Intraday Range, Close-to-Open, VWAP, Momentum, Volume Impact
    older_weights = [0.3, 0.2, 0.2, 0.3, 0.2]
    
    weights = pd.DataFrame(index=df.index, columns=['intraday_range', 'close_to_open', 'vwap', 'momentum', 'volume_impact'])
    weights.loc[df['volume'] > avg_vol_20d, :] = recent_weights
    weights.loc[df['volume'] <= avg_vol_20d, :] = older_weights
    
    # Apply Weights
    weighted_intraday_range = intraday_range * weights['intraday_range']
    weighted_close_to_open = close_to_open_return * weights['close_to_open']
    weighted_vwap = vwap * weights['vwap']
    weighted_momentum = momentum * weights['momentum']
    weighted_volume_impact = volume_impact * weights['volume_impact']
    
    # Final Alpha Factor
    alpha_factor = (weighted_intraday_range + weighted_close_to_open + 
                    weighted_vwap + weighted_momentum + weighted_volume_impact)
    
    return alpha_factor
```
