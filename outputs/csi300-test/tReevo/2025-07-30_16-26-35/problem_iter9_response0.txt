import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range: (High - Low) / Previous Day's Close
    df['intraday_range'] = (df['high'] - df['low']) / df['close'].shift(1)
    
    # Calculate Close-to-Open Return: (Close - Open) / Open
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Exponentially Weighted Combination of Intraday Range and Close-to-Open Return
    def weighted_combination(x, recent_weights, older_weights, span):
        weights = np.where(np.arange(len(x)) < span, recent_weights, older_weights)
        return np.sum(weights * x) / np.sum(weights)
    
    span = 5  # Assuming a 5-day window for exponential weighting
    recent_weights = np.array([0.7, 0.3])
    older_weights = np.array([0.5, 0.5])
    
    # Apply the weighted combination function
    df['alpha_factor'] = df[['intraday_range', 'close_to_open_return']].apply(
        lambda x: weighted_combination(x, recent_weights, older_weights, span), axis=1
    )
    
    return df['alpha_factor']
