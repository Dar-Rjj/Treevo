import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    price_sum = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    df['vwap'] = (price_sum * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Assign weights based on the recency of the data
    recent_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.3, 'vwap': 0.2}
    older_weights = {'intraday_range': 0.4, 'close_to_open_return': 0.4, 'vwap': 0.2}
    
    # Determine the cut-off for "recent" and "older" data
    midpoint = len(df) // 2
    
    # Initialize the factor column
    df['factor'] = 0.0
    
    # Apply weighted combination for recent data
    for col, weight in recent_weights.items():
        df.loc[df.index < midpoint, 'factor'] += df.loc[df.index < midpoint, col] * weight
    
    # Apply weighted combination for older data
    for col, weight in older_weights.items():
        df.loc[df.index >= midpoint, 'factor'] += df.loc[df.index >= midpoint, col] * weight
    
    return df['factor']
