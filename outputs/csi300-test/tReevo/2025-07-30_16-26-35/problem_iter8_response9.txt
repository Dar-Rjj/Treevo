```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Determine recent weights and older weights
    recent_weights = (0.7, 0.3)
    older_weights = (0.5, 0.5)
    
    # Assign adaptive weights based on recent data (e.g., last 30 days)
    recent_data_threshold = 30
    if len(df) < recent_data_threshold:
        weights = recent_weights
    else:
        weights = older_weights
    
    # Compute the weighted combination of Intraday Range and Close-to-Open Return
    combined_score = (weights[0] * intraday_range + weights[1] * close_to_open_return)
    
    # Short-Term Trend Analysis
    ma_20 = df['close'].rolling(window=20).mean()
    ma_100 = df['close'].rolling(window=100).mean()
    
    # Adjust the combined score based on moving average comparison
    combined_score += 0.1 * (ma_20 > ma_100) - 0.1 * (ma_20 < ma_100)
    
    return combined_score
```
