```python
import pandas as pd

def heuristics_v2(df):
    # Determine Day Over Day Returns
    df['returns'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Compute 10-Day Moving Average of Returns
    df['momentum'] = df['returns'].rolling(window=10).mean()
    
    # Identify Volume Spikes
    df['volume_spike'] = (df['volume'] > df['volume'].shift(1) * 1.20).astype(int)
    
    # Calculate Volume-Weighted Spike Index
    df['volume_weighted_spike'] = df['volume'] * df['volume_spike']
    
    # Adjust Momentum by Volume Shock
    adjustment_factor = -0.1  # Negative adjustment for volume shock
    df['adjusted_momentum'] = df['momentum'] - (df['volume_weighted_spike'] * adjustment_factor / df['volume'])
    
    # Generate Final Alpha Factor
    df['alpha_factor'] = df['adjusted_momentum'] * df['volume']
    
    return df['alpha_factor']
```
