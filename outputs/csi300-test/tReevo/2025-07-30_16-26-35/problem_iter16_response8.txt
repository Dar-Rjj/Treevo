```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['IntradayRange'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)

    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Incorporate Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Dynamic VWAP Based on Recency and Volume
    df['VolumeRatio'] = df['volume'] / df['volume'].shift(1)
    conditions = [
        (df['VolumeRatio'] > 1),
        (df['VolumeRatio'] <= 1)
    ]
    choices_recent = [0.7 * df['VWAP'] + 0.3 * df['VWAP'].shift(1), 0.4 * df['VWAP'] + 0.6 * df['VWAP'].shift(1)]
    df['DynamicVWAP'] = pd.np.select(conditions, choices_recent, default=0.5 * df['VWAP'] + 0.5 * df['VWAP'].shift(1))
    
    # Combine Intraday Range, Close-to-Open Return, and Dynamic VWAP
    df['WeightedIntradayRange'] = 0.5 * df['IntradayRange'] + 0.5 * df['IntradayRange'].shift(1)
    df['WeightedCloseToOpenReturn'] = 0.3 * df['CloseToOpenReturn'] + 0.2 * df['CloseToOpenReturn'].shift(1)
    df['WeightedVWAP'] = 0.2 * df['DynamicVWAP'] + 0.3 * df['DynamicVWAP'].shift(1)
    
    # Adjust Weights Based on Volatility and Volume Trends
    high_range = df['IntradayRange'] > df['IntradayRange'].mean()
    low_range = df['IntradayRange'] <= df['IntradayRange'].mean()
    
    # Adjust weights based on conditions
    weight_intraday_high = 0.1
    weight_intraday_low = -0.1
    weight_close_open = -0.05
    weight_vwap = -0.05
    weight_momentum_positive = 0.05
    weight_momentum_negative = -0.05
    
    # Apply adjustments
    df['AdjustedWeightedIntradayRange'] = df['WeightedIntradayRange'] + \
        (weight_intraday_high * (high_range & (df['VolumeRatio'] > 1))) + \
        (weight_intraday_low * (low_range & (df['VolumeRatio'] <= 1))) + \
        (weight_momentum_positive * (df['Momentum'] > 0)) + \
        (weight_momentum_negative * (df['Momentum'] < 0))
    
    df['AdjustedWeightedCloseToOpenReturn'] = df['WeightedCloseToOpenReturn'] + \
        (weight_close_open * (high_range & (df['VolumeRatio'] > 1))) + \
        (-weight_close_open * (low_range & (df['VolumeRatio'] <= 1))) + \
        (-weight_momentum_positive * (df['Momentum'] > 0)) + \
        (-weight_momentum_negative * (df['Momentum'] < 0))
    
    df['AdjustedWeightedVWAP'] = df['WeightedVWAP'] + \
        (weight_vwap * (high_range & (df['VolumeRatio'] > 1))) + \
        (-weight_vwap * (low_range & (df['VolumeRatio'] <= 1)))
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['AdjustedWeightedIntradayRange'] + df['AdjustedWeightedCloseToOpenReturn'] + df['AdjustedWeightedVWAP']
    
    return df['AlphaFactor']
```
