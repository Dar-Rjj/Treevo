import pandas as pd

def heuristics_v2(df):
    # Calculate Long-Term Price Momentum
    df['Return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['20DayMomentum'] = df['Return'].rolling(window=20).mean()
    
    # Incorporate Volume Shock
    df['VolumeSpike'] = (df['volume'] > df['volume'].shift(1) * 1.3).astype(int)
    df['VolumeWeightedSpikeIndex'] = df['volume'] * df['VolumeSpike']
    
    # Adjust Long-Term Momentum by Volume Shock
    adjustment_factor = -0.5  # Example negative adjustment factor
    df['AdjustedMomentum'] = df['20DayMomentum'] - adjustment_factor * df['VolumeWeightedSpikeIndex']
    
    # Incorporate Intraday Volatility
    df['IntradayVolatility'] = (df['high'] - df['low']) / df['low']
    
    # Integrate Intraday Volatility into Final Alpha Factor
    df['FinalAlphaFactor'] = df['AdjustedMomentum'] * df['volume'] * df['IntradayVolatility']
    
    return df['FinalAlphaFactor']
