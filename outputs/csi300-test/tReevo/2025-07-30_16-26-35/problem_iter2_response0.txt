import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate Intraday Range
    daily_high_low_spread = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_open_spread = df['close'] - df['open']
    
    # Adjust for Intraday Volatility
    adjusted_return = close_open_spread / daily_high_low_spread
    
    # Integrate Volume and Price Changes
    # Assuming 'volume' is in the DataFrame and represents the total volume per day,
    # we calculate the sum of absolute price changes per tick using a rolling window.
    # Here, we approximate the sum of absolute price changes by the difference between
    # consecutive close prices. This approximation assumes that each day's close to
    # the next day's open is a good proxy for the intra-day price movement.
    abs_price_changes = (df['close'].shift(-1) - df['close']).abs()
    synchronization_metric = abs_price_changes / df['volume']
    
    # Weight Intraday Volatility-Adjusted Return
    weighted_adjusted_return = adjusted_return * synchronization_metric
    
    # Adjust for Market Sentiment
    market_return = df['close'] - df['close'].shift(1)
    sentiment_adjustment = (market_return > 0).astype(float) * 1.1 + (market_return <= 0).astype(float) * 0.9
    final_factor = weighted_adjusted_return * sentiment_adjustment
    
    return final_factor.dropna()
