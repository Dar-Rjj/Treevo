```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] / df['open'].shift(1)) - 1
    
    # Adaptive Weighting of Recent Data
    n = 5  # Number of days for moving average
    m = 10  # Number of days for exponential moving average
    moving_avg_close_to_open = close_to_open_return.rolling(window=n).mean()
    ema_intraday_range = intraday_range.ewm(span=m, adjust=False).mean()
    
    # Combine Adjusted Intraday Range and Momentum
    adjusted_momentum = (close_to_open_return / ema_intraday_range) * moving_avg_close_to_open
    
    return adjusted_momentum.dropna()
```
