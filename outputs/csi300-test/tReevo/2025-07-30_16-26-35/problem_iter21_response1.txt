import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate the 20-day average volume
    df['average_volume_20'] = df['volume'].rolling(window=20).mean()
    
    # Dynamic Weighting Based on Volume
    def apply_weights(row):
        if row['volume'] > row['average_volume_20']:
            # Apply Recent Weights
            return 0.7 * row['intraday_range'] + 0.3 * row['close_to_open_return']
        else:
            # Apply Older Weights
            return 0.5 * row['intraday_range'] + 0.5 * row['close_to_open_return']
    
    # Generate the alpha factor
    df['alpha_factor'] = df.apply(apply_weights, axis=1)
    
    return df['alpha_factor']
