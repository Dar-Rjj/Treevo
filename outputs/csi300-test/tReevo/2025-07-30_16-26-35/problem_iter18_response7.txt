```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Dynamic Weighting
    recent_intraday_range = 0.6 * intraday_range + 0.4 * intraday_range.shift(1)
    recent_close_to_open_return = 0.3 * close_to_open_return + 0.2 * close_to_open_return.shift(1)
    recent_vwap = 0.1 * vwap + 0.4 * vwap.shift(1)
    
    # Initial Alpha Factor
    initial_alpha_factor = recent_intraday_range + recent_close_to_open_return + recent_vwap
    
    # Incorporate Recency and Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = initial_alpha_factor * (1 + (volume_ratio - 1) * 0.5)
    
    # Volume Momentum
    volume_momentum = (volume_ratio - 1).clip(lower=0)
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.1
    
    return final_alpha_factor
```
