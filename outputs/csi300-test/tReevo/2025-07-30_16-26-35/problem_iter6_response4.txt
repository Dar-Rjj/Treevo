```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    total_volume = df['volume'].sum()
    vwap = df['vwap'].sum() / total_volume
    
    # Integrate Amount
    df['amount_normalized'] = df['amount'] / vwap
    
    # Assign dynamic weights based on recent data
    recent_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.3, 'amount_normalized': 0.2}
    older_weights = {'intraday_range': 0.4, 'close_to_open_return': 0.4, 'amount_normalized': 0.2}
    
    # Combine Intraday Range, Close-to-Open Return, and Amount
    df['combined_factor'] = (
        recent_weights['intraday_range'] * df['intraday_range'] +
        recent_weights['close_to_open_return'] * df['close_to_open_return'] +
        recent_weights['amount_normalized'] * df['amount_normalized']
    )
    
    # Long-Term Trend Analysis
    df['100_day_ma'] = df['close'].rolling(window=100).mean()
    df['20_day_ma'] = df['close'].rolling(window=20).mean()
    
    trend_adjustment = (df['20_day_ma'] > df['100_day_ma']).astype(int) * 0.1 - (df['20_day_ma'] < df['100_day_ma']).astype(int) * 0.1
    df['combined_factor'] += trend_adjustment
    
    return df['combined_factor']
```
