```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low'])
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Volume Impact
    df['volume_impact'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Calculate Daily Volatility
    daily_returns = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['volatility'] = daily_returns.rolling(window=20).std()
    
    # Dynamically Adjust Weights Based on Volatility
    df['weight_intraday_range'] = np.where(df['volatility'] > df['volatility'].mean(), 0.6, 0.4)
    df['weight_close_to_open_return'] = np.where(df['volatility'] > df['volatility'].mean(), 0.2, 0.4)
    df['weight_volume_impact'] = 0.2
    
    # Combine Intraday Range, Close-to-Open Return, and Volume Impact
    df['alpha_factor'] = (df['weight_intraday_range'] * df['intraday_range'] +
                          df['weight_close_to_open_return'] * df['close_to_open_return'] +
                          df['weight_volume_impact'] * df['volume_impact'])
    
    return df['alpha_factor']
```
