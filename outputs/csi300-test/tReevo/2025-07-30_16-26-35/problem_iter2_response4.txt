import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']

    # Calculate Intraday Volatility (20-day average of intraday ranges)
    df['intraday_volatility'] = df['intraday_range'].rolling(window=20).mean()

    # Calculate Adjusted Daily Return
    df['daily_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    df['adjusted_daily_return'] = df['daily_return'] * df['volume']

    # Calculate Rolling Sum of Adjusted Daily Returns (20-day window)
    df['rolling_sum_adjusted_return'] = df['adjusted_daily_return'].rolling(window=20).sum()

    # Calculate Final Momentum Score
    df['momentum_score'] = df['rolling_sum_adjusted_return'] / df['intraday_volatility']

    # Drop rows with NaN values that were created by the rolling calculations
    df.dropna(inplace=True)

    # Return the momentum score as a Series
    return df['momentum_score']
