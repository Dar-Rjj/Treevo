import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Consider Volume Impact
    df['average_volume'] = df['volume'].rolling(window=20).mean()
    df['volume_relative_strength'] = df['volume'] / df['average_volume']
    
    # Incorporate Moving Averages
    df['5_day_sma'] = df['close'].rolling(window=5).mean()
    df['20_day_sma'] = df['close'].rolling(window=20).mean()
    df['sma_difference'] = df['5_day_sma'] - df['20_day_sma']
    
    # Combine Factors
    recent_weights = 0.7 * df['intraday_range'] + 0.2 * df['close_to_open_return'] + 0.1 * df['volume_relative_strength']
    older_weights = 0.5 * df['intraday_range'] + 0.3 * df['close_to_open_return'] + 0.2 * df['sma_difference']
    
    # Weighted Combination
    df['alpha_factor'] = 0.6 * recent_weights + 0.4 * older_weights
    
    return df['alpha_factor']
