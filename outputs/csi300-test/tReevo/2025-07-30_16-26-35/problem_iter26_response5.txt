```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Dynamic Weighting
    recent_intraday_range = intraday_range.ewm(span=10).mean() * 0.7 + intraday_range.ewm(span=20).mean() * 0.3
    recent_close_to_open_return = close_to_open_return.ewm(span=5).mean() * 0.5 + close_to_open_return.ewm(span=10).mean() * 0.2
    recent_vwap = vwap.ewm(span=5).mean() * 0.3 + vwap.ewm(span=10).mean() * 0.4
    recent_intraday_momentum = intraday_momentum.ewm(span=5).mean() * 0.3 + intraday_momentum.ewm(span=10).mean() * 0.4
    
    # Initial Alpha Factor
    initial_alpha_factor = (recent_intraday_range + 
                            recent_close_to_open_return + 
                            recent_vwap + 
                            recent_intraday_momentum)
    
    # Incorporate Recency and Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = initial_alpha_factor * (1 + (volume_ratio - 1) * 0.8)
    
    # Advanced Volume Momentum
    volume_momentum = (volume_ratio - 1).clip(lower=0)
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.4
    
    return final_alpha_factor
```
