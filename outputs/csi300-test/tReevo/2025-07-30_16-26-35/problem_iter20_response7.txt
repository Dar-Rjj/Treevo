```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Momentum
    df['momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Dynamic Weighting for Intraday Range
    intraday_range_recent_weight = 0.4
    intraday_range_older_weight = 0.3
    df['weighted_intraday_range'] = df['intraday_range'] * intraday_range_recent_weight
    df['weighted_intraday_range'] = df['weighted_intraday_range'].fillna(0) + df['intraday_range'].shift(5).fillna(0) * intraday_range_older_weight
    
    # Dynamic Weighting for Close-to-Open Return
    close_to_open_recent_weight = 0.2
    close_to_open_older_weight = 0.2
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * close_to_open_recent_weight
    df['weighted_close_to_open_return'] = df['weighted_close_to_open_return'].fillna(0) + df['close_to_open_return'].shift(5).fillna(0) * close_to_open_older_weight
    
    # Dynamic Weighting for VWAP
    vwap_recent_weight = 0.2
    vwap_older_weight = 0.2
    df['weighted_vwap'] = df['vwap'] * vwap_recent_weight
    df['weighted_vwap'] = df['weighted_vwap'].fillna(0) + df['vwap'].shift(5).fillna(0) * vwap_older_weight
    
    # Dynamic Weighting for Momentum
    momentum_recent_weight = 0.2
    momentum_older_weight = 0.3
    df['weighted_momentum'] = df['momentum'] * momentum_recent_weight
    df['weighted_momentum'] = df['weighted_momentum'].fillna(0) + df['momentum'].shift(5).fillna(0) * momentum_older_weight
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_intraday_range'] + df['weighted_close_to_open_return'] + df['weighted_vwap'] + df['weighted_momentum'])
    
    return df['alpha_factor']
```
