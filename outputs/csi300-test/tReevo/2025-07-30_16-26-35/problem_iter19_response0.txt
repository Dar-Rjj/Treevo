```python
import pandas as pd

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Determine weights based on the date
    recent_weights = 0.7 * intraday_range + 0.3 * close_to_open_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_open_return
    
    # Compute Volume Momentum
    volume_momentum = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_momentum = volume_momentum.fillna(0)  # Handle NaN values for the first row
    
    # Final Alpha Factor
    factor_values = (
        (recent_weights + older_weights) / 2 +
        0.6 * ((recent_weights + older_weights) / 2) + 
        0.4 * volume_momentum
    )
    
    return factor_values
```
