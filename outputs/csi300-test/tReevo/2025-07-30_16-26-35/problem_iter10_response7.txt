import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Compute recent volatility
    recent_volatility = df['close'].rolling(window=5).std()
    
    # Dynamic Weighted Combination
    dynamic_weights_intraday = 0.8 * (recent_volatility > recent_volatility.mean()) + 0.6 * (recent_volatility <= recent_volatility.mean())
    dynamic_weights_close_to_open = 1 - dynamic_weights_intraday
    
    combined_factor = (intraday_range * dynamic_weights_intraday) + (close_to_open_return * dynamic_weights_close_to_open)
    
    # Incorporate Price Momentum
    momentum_5_day = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    momentum_integrated_factor = (combined_factor * 0.7) + (momentum_5_day * 0.3)
    
    # Incorporate Volume Shocks
    volume_shock = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    final_alpha_factor = (momentum_integrated_factor * 0.8) + (volume_shock * 0.2)
    
    return final_alpha_factor.dropna()
