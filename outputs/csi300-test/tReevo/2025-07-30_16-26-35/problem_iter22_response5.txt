```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low'])

    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)

    # Compute 20-day average volume
    df['avg_volume_20'] = df['volume'].rolling(window=20).mean()

    # Weighted Combination based on recent volume trend
    def weighted_combination(row):
        if row['volume'] > row['avg_volume_20']:
            return 0.7 * row['intraday_range'] + 0.3 * row['close_to_open_return']
        else:
            return 0.5 * row['intraday_range'] + 0.5 * row['close_to_open_return']

    df['weighted_combination'] = df.apply(weighted_combination, axis=1)

    # Integrate Price Dynamics
    df['momentum'] = (df['close'] - df['open']) / df['open']
    df['reversal'] = (df['open'] - df['close']) / df['open']

    # Final Alpha Factor
    df['alpha_factor'] = df['weighted_combination'] + df['momentum'] - df['reversal']

    return df['alpha_factor']
```
