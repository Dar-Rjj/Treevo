```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)

    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']

    # Calculate Momentum
    df['Momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Calculate Volume Impact
    df['Volume_Impact'] = df['volume'] / df['volume'].rolling(window=5).mean()

    # Dynamic Weighting
    recent_weights = [0.4, 0.2, 0.2, 0.2, 0.2]  # Intraday Range, Close-to-Open Return, VWAP, Momentum, Volume Impact
    older_weights = [0.3, 0.2, 0.2, 0.3, 0.2]

    # Apply weights
    df['Weighted_Intraday_Range'] = df['Intraday_Range'] * (recent_weights[0] if df.index[-1] >= df.index[-5] else older_weights[0])
    df['Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * (recent_weights[1] if df.index[-1] >= df.index[-5] else older_weights[1])
    df['Weighted_VWAP'] = df['VWAP'] * (recent_weights[2] if df.index[-1] >= df.index[-5] else older_weights[2])
    df['Weighted_Momentum'] = df['Momentum'] * (recent_weights[3] if df.index[-1] >= df.index[-5] else older_weights[3])
    df['Weighted_Volume_Impact'] = df['Volume_Impact'] * (recent_weights[4] if df.index[-1] >= df.index[-5] else older_weights[4])

    # Combine all weighted factors to get the final alpha factor
    df['Alpha_Factor'] = (
        df['Weighted_Intraday_Range'] + 
        df['Weighted_Close_to_Open_Return'] + 
        df['Weighted_VWAP'] + 
        df['Weighted_Momentum'] + 
        df['Weighted_Volume_Impact']
    )

    return df['Alpha_Factor']
```
