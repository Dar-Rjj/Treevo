import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['close']) / df['close']
    
    # Determine Market Trend
    df['trend'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'trend'] = 1  # Up
    df.loc[df['close'] < df['close'].shift(1), 'trend'] = -1  # Down
    
    # Dynamic Weighting Based on Market Trends and Recent Data Emphasis
    weights = {
        1: {'intraday_range': 0.6, 'close_to_open_return': 0.4},
        -1: {'intraday_range': 0.7, 'close_to_open_return': 0.3},
        0: {'intraday_range': 0.5, 'close_to_open_return': 0.5}
    }
    
    df['recent_factor'] = (df['intraday_range'] * 0.8 + 
                           df['close_to_open_return'] * 0.2)
    df['older_factor'] = (df['intraday_range'] * 0.5 + 
                          df['close_to_open_return'] * 0.5)
    
    # Compute Volume Impact
    df['volume_impact'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Final Dynamic Factor
    df['interim_factor'] = (df['recent_factor'] * 0.7 + 
                            df['older_factor'] * 0.3)
    df['adjusted_volume_impact'] = df['volume_impact'] * 0.2
    df['final_factor'] = (df['interim_factor'] * 0.8 + 
                          df['adjusted_volume_impact'])
    
    return df['final_factor']
