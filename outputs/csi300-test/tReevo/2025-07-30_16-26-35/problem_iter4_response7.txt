```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['low']
    close_price_volatility = df['close'].rolling(window=20).std()
    
    # Combine Intraday Range and Return
    normalized_intraday_range = intraday_range / intraday_volatility
    combined_factor = normalized_intraday_range * close_to_open_return
    
    # Weight by Volume
    volume_weighted_factor = combined_factor * df['volume']
    
    # Integrate Multi-Period Momentum
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    momentum = short_term_ma - long_term_ma
    
    # Consider Market Sentiment
    daily_volatility = (df['high'] - df['low']) / df['low']
    market_sentiment = momentum * (1 / daily_volatility)
    
    # Final Alpha Factor
    alpha_factor = volume_weighted_factor * market_sentiment
    
    return alpha_factor
```
