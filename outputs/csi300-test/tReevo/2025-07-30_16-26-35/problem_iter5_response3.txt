```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open'].shift(-1)) / df['close']
    
    # Combine Intraday Range and Close-to-Open Return
    combined_factor = close_to_open_return / intraday_range
    
    # Apply Adaptive Weighting using exponentially weighted moving average of Intraday Range
    ewma_intraday_range = intraday_range.ewm(span=20).mean()
    adaptive_weighted_factor = combined_factor * (intraday_range / ewma_intraday_range)
    
    return adaptive_weighted_factor
```
