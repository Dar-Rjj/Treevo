```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Determine Market Trend
    df['trend'] = 0
    df.loc[df['close'] > df['close'].shift(1), 'trend'] = 1  # Trending Up
    df.loc[df['close'] < df['close'].shift(1), 'trend'] = -1  # Trending Down
    
    # Define weights based on market trend
    def get_weights(trend):
        if trend == 1:
            return 0.6, 0.4
        elif trend == -1:
            return 0.7, 0.3
        else:
            return 0.5, 0.5
    
    df['intraday_weight'], df['c_to_o_weight'] = zip(*df['trend'].apply(get_weights))
    
    # Combine Intraday Range and Close-to-Open Return with dynamic weighting
    df['recent_factor'] = (df['intraday_range'] * df['intraday_weight'] * 0.8) + (df['close_to_open_return'] * df['c_to_o_weight'] * 0.2)
    df['older_factor'] = (df['intraday_range'] * df['intraday_weight'] * 0.5) + (df['close_to_open_return'] * df['c_to_o_weight'] * 0.5)
    
    # Final Factor
    df['final_factor'] = (df['recent_factor'] * 0.7) + (df['older_factor'] * 0.3)
    
    return df['final_factor']
```
