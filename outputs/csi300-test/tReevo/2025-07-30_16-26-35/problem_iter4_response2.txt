```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Average Daily Volume
    avg_daily_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate Intraday Momentum
    close_prices = df['close']
    intraday_momentum = close_prices.pct_change(periods=5)
    
    # Market Sentiment
    high_low_sentiment = (df['high'] + df['low']) / 2
    close_sentiment = df['close']
    market_sentiment = (high_low_sentiment + close_sentiment) / 2
    
    # Combine Intraday Range, Return, and Volume
    combined_factor = (intraday_range * close_to_open_return) / avg_daily_volume
    
    # Integrate Momentum and Sentiment
    final_factor = combined_factor + intraday_momentum + market_sentiment
    
    return final_factor
```
