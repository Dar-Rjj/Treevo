```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Define dynamic weights based on the recency of the data
    recent_weights_intraday = 0.7
    older_weights_intraday = 0.5
    recent_weights_close_to_open = 0.3
    older_weights_close_to_open = 0.5
    
    # Apply dynamic weighting
    weighted_intraday_range = intraday_range * recent_weights_intraday
    weighted_intraday_range.iloc[:-7] = intraday_range.iloc[:-7] * older_weights_intraday  # Assuming 7 days for 'older' threshold
    
    weighted_close_to_open_return = close_to_open_return * recent_weights_close_to_open
    weighted_close_to_open_return.iloc[:-7] = close_to_open_return.iloc[:-7] * older_weights_close_to_open  # Assuming 7 days for 'older' threshold
    
    # Combine Weighted Intraday Range and Weighted Close-to-Open Return
    alpha_factor = weighted_intraday_range + weighted_close_to_open_return
    
    return alpha_factor
```
