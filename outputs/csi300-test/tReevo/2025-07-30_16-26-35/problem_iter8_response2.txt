import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Adaptive Weighting
    def adaptive_weighting(row, index):
        if index < 5:
            return row['intraday_range'] * 0.7 + row['close_to_open_return'] * 0.3
        elif index < 25:
            return row['intraday_range'] * 0.5 + row['close_to_open_return'] * 0.5
        else:
            return row['intraday_range'] * 0.3 + row['close_to_open_return'] * 0.7
    
    df['adjusted_return'] = df.apply(lambda row: adaptive_weighting(row, df.index.get_loc(row.name)), axis=1)
    
    # Incorporate Enhanced Market Sentiment
    df['market_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    def adjust_for_market_trend(row):
        if row['market_return'] > 0.01:
            return row['adjusted_return'] * 1.10
        elif -0.01 <= row['market_return'] <= 0.01:
            return row['adjusted_return']
        else:
            return row['adjusted_return'] * 0.90
    
    df['adjusted_return'] = df.apply(adjust_for_market_trend, axis=1)
    
    # Dynamic Volatility Measure
    df['std_close_20'] = df['close'].rolling(window=20).std()
    
    def adjust_for_volatility(row):
        if row['std_close_20'] > 0.05:
            return row['adjusted_return'] * 0.8
        elif 0.02 <= row['std_close_20'] <= 0.05:
            return row['adjusted_return']
        else:
            return row['adjusted_return'] * 1.2
    
    df['adjusted_return'] = df.apply(adjust_for_volatility, axis=1)
    
    # Drop intermediate columns to clean up the DataFrame
    df.drop(columns=['intraday_range', 'close_to_open_return', 'market_return', 'std_close_20'], inplace=True)
    
    # Return the factor values as a Series
    return df['adjusted_return']
