import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Incorporate Momentum
    df['momentum'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Dynamic VWAP Based on Recency and Volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['recent_vwap'] = 0.7 * df['vwap'] + 0.3 * df['vwap'].shift(1)
    df['older_vwap'] = 0.4 * df['vwap'] + 0.6 * df['vwap'].shift(1)
    df['dynamic_vwap'] = df.apply(lambda row: row['recent_vwap'] if row['volume_ratio'] > 1 else row['older_vwap'], axis=1)
    
    # Combine Intraday Range, Close-to-Open Return, and Dynamic VWAP
    df['weighted_intraday_range'] = 0.5 * df['intraday_range'] + 0.5 * df['intraday_range'].shift(1)
    df['weighted_close_to_open_return'] = 0.3 * df['close_to_open_return'] + 0.2 * df['close_to_open_return'].shift(1)
    df['weighted_vwap'] = 0.2 * df['dynamic_vwap'] + 0.3 * df['dynamic_vwap'].shift(1)
    
    # Adjust Weights Based on Volatility and Volume Trends
    def adjust_weights(row):
        if row['intraday_range'] > df['intraday_range'].mean() and row['volume_ratio'] > 1:
            return 0.6 * row['weighted_intraday_range'] + 0.2 * row['weighted_close_to_open_return'] + 0.2 * row['weighted_vwap']
        elif row['intraday_range'] < df['intraday_range'].mean() and row['volume_ratio'] <= 1:
            return 0.2 * row['weighted_intraday_range'] + 0.4 * row['weighted_close_to_open_return'] + 0.4 * row['weighted_vwap']
        else:
            return 0.5 * row['weighted_intraday_range'] + 0.3 * row['weighted_close_to_open_return'] + 0.2 * row['weighted_vwap']
    
    df['alpha_factor'] = df.apply(adjust_weights, axis=1)
    
    return df['alpha_factor']
