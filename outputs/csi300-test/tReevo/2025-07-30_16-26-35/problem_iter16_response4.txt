```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Compute Weighted Intraday Range
    weighted_intraday_range = intraday_range * 0.5
    
    # Compute Weighted Close-to-Open Return
    weighted_close_to_open_return = close_to_open_return * 0.3
    
    # Compute Weighted VWAP
    weighted_vwap = vwap * 0.2
    
    # Final Alpha Factor
    alpha_factor = (weighted_intraday_range + 
                    weighted_close_to_open_return + 
                    weighted_vwap)
    
    return alpha_factor
```
