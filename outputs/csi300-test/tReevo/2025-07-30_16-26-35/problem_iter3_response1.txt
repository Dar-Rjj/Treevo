```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Average Daily Volume
    avg_daily_volume = df['volume'].mean()
    
    # Normalize by Intraday Range
    normalized_close_to_open_return = close_to_open_return / intraday_range
    
    # Multiply by Close-to-Open Return
    adjusted_return = normalized_close_to_open_return * close_to_open_return
    
    # Adjust by Average Daily Volume
    factor_values = adjusted_return * (df['volume'] / avg_daily_volume)
    
    return factor_values
```
