```python
import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Consider Volume Impact
    df['Avg_Volume_20D'] = df['volume'].rolling(window=20).mean()
    df['Volume_Relative_Strength'] = df['volume'] / df['Avg_Volume_20D']
    
    # Incorporate Moving Averages
    df['5D_SMA'] = df['close'].rolling(window=5).mean()
    df['20D_SMA'] = df['close'].rolling(window=20).mean()
    df['SMA_Difference'] = df['5D_SMA'] - df['20D_SMA']
    
    # Incorporate Momentum Indicators
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['RSI_14D'] = 100 - (100 / (1 + rs))
    
    df['12D_EMA'] = df['close'].ewm(span=12, adjust=False).mean()
    df['26D_EMA'] = df['close'].ewm(span=26, adjust=False).mean()
    df['MACD_Line'] = df['12D_EMA'] - df['26D_EMA']
    
    # Combine Factors
    recent_weights = [0.4, 0.3, 0.1, 0.1, 0.1]
    older_weights = [0.5, 0.2, 0.1, 0.1, 0.05, 0.05]
    
    recent_factors = [df['Intraday_Range'], df['Close_to_Open_Return'], 
                      df['Volume_Relative_Strength'], df['SMA_Difference'], df['RSI_14D']]
    older_factors = [df['Intraday_Range'], df['Close_to_Open_Return'], 
                     df['Volume_Relative_Strength'], df['SMA_Difference'], df['RSI_14D'], df['MACD_Line']]
    
    recent_factor_scores = (np.array(recent_factors) * np.array(recent_weights)).sum(axis=0)
    older_factor_scores = (np.array(older_factors) * np.array(older_weights)).sum(axis=0)
    
    df['Factor_Scores'] = recent_factor_scores + older_factor_scores
    
    # Dynamic Weight Adjustment
    df['EMA_Intraday_Range'] = df['Intraday_Range'].ewm(span=10, adjust=False).mean()
    df['EMA_Close_to_Open_Return'] = df['Close_to_Open_Return'].ewm(span=10, adjust=False).mean()
    
    recent_volatility = df['EMA_Intraday_Range'].std()
    close_to_open_volatility = df['EMA_Close_to_Open_Return'].std()
    
    if recent_volatility > close_to_open_volatility:
        weight_adjustment = [0.05, -0.05, 0, 0, 0, 0]
    else:
        weight_adjustment = [-0.05, 0.05, 0, 0, 0, 0]
    
    adjusted_recent_weights = np.array(recent_weights) + np.array(weight_adjustment[:len(recent_weights)])
    adjusted_older_weights = np.array(older_weights) + np.array(weight_adjustment[:len(older_weights)])
    
    adjusted_recent_factor_scores = (np.array(recent_factors) * adjusted_recent_weights).sum(axis=0)
    adjusted_older_factor_scores = (np.array(older_factors) * adjusted_older_weights).sum(axis=0)
    
    df['Adjusted_Factor_Scores'] = adjusted_recent_factor_scores + adjusted_older_factor_scores
    
    return df['Adjusted_Factor_Scores']
```
