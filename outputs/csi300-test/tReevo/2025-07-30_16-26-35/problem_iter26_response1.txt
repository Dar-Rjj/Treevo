```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Volume Impact
    average_volume_20_days = df['volume'].rolling(window=20).mean()
    volume_impact = df['volume'] / average_volume_20_days
    
    # Compute Dynamic Weights
    recent_weights = {
        'intraday_range': 0.4,
        'close_to_open_return': 0.2,
        'vwap': 0.2,
        'long_term_momentum': 0.2,
        'medium_term_momentum': 0.2,
        'volume_impact': 0.2
    }
    older_weights = {
        'intraday_range': 0.3,
        'close_to_open_return': 0.2,
        'vwap': 0.2,
        'long_term_momentum': 0.3,
        'medium_term_momentum': 0.3,
        'volume_impact': 0.2
    }
    
    # Apply Dynamic Weights
    weighted_intraday_range = intraday_range * (recent_weights['intraday_range'] * 0.5 + older_weights['intraday_range'] * 0.5)
    weighted_close_to_open_return = close_to_open_return * (recent_weights['close_to_open_return'] * 0.5 + older_weights['close_to_open_return'] * 0.5)
    weighted_vwap = vwap * (recent_weights['vwap'] * 0.5 + older_weights['vwap'] * 0.5)
    weighted_long_term_momentum = long_term_momentum * (recent_weights['long_term_momentum'] * 0.5 + older_weights['long_term_momentum'] * 0.5)
    weighted_medium_term_momentum = medium_term_momentum * (recent_weights['medium_term_momentum'] * 0.5 + older_weights['medium_term_momentum'] * 0.5)
    weighted_volume_impact = volume_impact * (recent_weights['volume_impact'] * 0.5 + older_weights['volume_impact'] * 0.5)
    
    # Combine Weighted Components
    alpha_factor = (weighted_intraday_range + weighted_close_to_open_return + weighted_vwap +
                    weighted_long_term_momentum + weighted_medium_term_momentum + weighted_volume_impact)
    
    # Adjust Alpha Factor based on Volume Ratio
    volume_ratio = df['volume'] / average_volume_20_days
    adjusted_alpha_factor = alpha_factor * (1 + (volume_ratio - 1) * 0.7)
    final_alpha_factor = adjusted_alpha_factor + (volume_ratio - 1).clip(lower=0) * 0.3
    
    return final_alpha_factor
```
