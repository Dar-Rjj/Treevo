```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']

    # Determine the weights based on recent and older data
    recent_weights = 0.7 * intraday_range + 0.3 * close_to_open_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_open_return

    # Combine Intraday Range and Close-to-Open Return with weighted combination
    alpha_factor = recent_weights.shift(1) + older_weights

    return alpha_factor
```
