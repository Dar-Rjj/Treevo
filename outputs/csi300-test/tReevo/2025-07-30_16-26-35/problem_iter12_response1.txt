```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Dynamic Weighting Based on Recency and Volume
    recent_weights = 0.6 * df['Intraday_Range'] + 0.3 * df['Close_to_Open_Return'] + 0.1 * df['Volume_Ratio']
    older_weights = 0.4 * df['Intraday_Range'] + 0.4 * df['Close_to_Open_Return'] + 0.2 * df['Volume_Ratio']
    
    # Final Alpha Factor
    df['Final_Alpha_Factor'] = (recent_weights + older_weights) / 2
    
    return df['Final_Alpha_Factor']
```
