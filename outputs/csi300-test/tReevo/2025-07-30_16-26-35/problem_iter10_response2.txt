```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range: High - Low
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: (Close - Open) / Open
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Define recent and older periods
    recent_period = 30
    older_period = 60
    
    # Apply adaptive weighted combination
    recent_weights = (0.7 * intraday_range + 0.3 * close_to_open_return).rolling(window=recent_period).mean()
    older_weights = (0.5 * intraday_range + 0.5 * close_to_open_return).rolling(window=older_period).mean()
    
    # Combine the two weighted series
    combined_factor = (recent_weights * 0.6 + older_weights * 0.4).dropna()
    
    return combined_factor
```
