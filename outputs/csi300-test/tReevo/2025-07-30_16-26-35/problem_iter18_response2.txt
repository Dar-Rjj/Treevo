```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Combine Intraday Range, Close-to-Open Return, and VWAP with dynamic weighting
    recent_intraday_range = df['intraday_range'].rolling(window=2).mean() * 0.7 + df['intraday_range'].shift(2).rolling(window=8).mean() * 0.3
    recent_close_to_open_return = df['close_to_open_return'].rolling(window=2).mean() * 0.4 + df['close_to_open_return'].shift(2).rolling(window=8).mean() * 0.2
    recent_vwap = df['vwap'].rolling(window=2).mean() * 0.1 + df['vwap'].shift(2).rolling(window=8).mean() * 0.5
    
    # Initial Alpha Factor
    initial_alpha_factor = (recent_intraday_range + recent_close_to_open_return + recent_vwap)
    
    # Incorporate Recency and Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = initial_alpha_factor * (1 + (volume_ratio - 1) * 0.6)
    
    # Volume Momentum
    volume_momentum = (volume_ratio - 1).apply(lambda x: max(0, x))
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.2
    
    return final_alpha_factor
```
