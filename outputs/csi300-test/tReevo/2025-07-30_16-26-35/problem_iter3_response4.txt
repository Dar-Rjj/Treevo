```python
import pandas as pd

def heuristics_v2(data):
    # Calculate Intraday Range
    intraday_range = data['high'] - data['low']

    # Calculate Close-to-Open Return
    close_to_open_return = (data['close'] / data['open']) - 1

    # Normalize Close-to-Open Return by Intraday Range
    normalized_return = close_to_open_return / intraday_range

    # Calculate Average Volume
    avg_volume = data['volume'].rolling(window=5).mean()

    # Volume-Weighted Momentum
    volume_weighted_momentum = normalized_return * avg_volume

    # Integrate Multi-Period Averages
    vwm_5d_avg = volume_weighted_momentum.rolling(window=5).mean()
    vwm_20d_avg = volume_weighted_momentum.rolling(window=20).mean()

    # Final Alpha Factor
    alpha_factor = (vwm_5d_avg + vwm_20d_avg) / 2

    return alpha_factor
```
