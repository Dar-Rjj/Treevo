import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Combine Intraday Range and Close-to-Open Return with weights
    recent_weights = 0.6 * df['intraday_range'] + 0.4 * df['close_to_open_return']
    older_weights = 0.5 * df['intraday_range'] + 0.5 * df['close_to_open_return']
    df['intraday_volatility_adjusted_return'] = recent_weights.fillna(older_weights)
    
    # Integrate Momentum
    # Simple Moving Average (SMA) of Close prices
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Relative Strength Index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Combine SMA and RSI with weights
    df['momentum'] = 0.5 * (df['close'] / df['sma_20']) + 0.5 * (df['rsi_14'] / 100)
    
    # Integrate Volume Trends
    # Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).cumsum() / (df['volume'].cumsum())
    
    # Combine Intraday Volatility Adjusted Return, Momentum, and Volume Trends with weights
    df['alpha_factor'] = (
        0.4 * df['intraday_volatility_adjusted_return'] +
        0.3 * df['momentum'] +
        0.2 * df['volume_change'] +
        0.1 * df['vwap']
    )
    
    return df['alpha_factor']
