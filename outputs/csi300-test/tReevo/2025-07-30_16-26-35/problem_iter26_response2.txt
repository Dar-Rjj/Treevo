```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative difference between the close and open price
    rel_diff = (df['close'] - df['open']) / df['open']
    
    # Calculate the money flow index (MFI) as a measure of buying/selling pressure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(rel_diff > 0, 0).rolling(window=14).sum()
    negative_money_flow = raw_money_flow.where(rel_diff < 0, 0).rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Calculate the factor value as a combination of the above metrics
    factor_value = rel_diff * mfi
    
    return factor_value
```
