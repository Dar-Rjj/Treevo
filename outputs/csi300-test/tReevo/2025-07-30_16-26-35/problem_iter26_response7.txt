```python
import pandas as pd

def heuristics_v2(data):
    # Ensure the data is sorted by date
    data = data.sort_index()
    
    # Calculate Intraday Range (High - Low)
    intraday_range = data['high'] - data['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (data['close'] - data['open']) / data['open']
    
    # Determine weights for recent and older data
    recent_weights = 0.6 * intraday_range + 0.4 * close_to_open_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_open_return
    
    # Combine recent and older weights to create the final factor
    combined_factor = recent_weights.shift(1).fillna(older_weights)
    
    return combined_factor
```
