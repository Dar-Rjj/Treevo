```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['prev_close'] = df['close'].shift(1)
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Combine Intraday Range and Close-to-Open Return with weights
    df['combined_return'] = 0.7 * df['intraday_range'] + 0.3 * df['close_to_open_return']
    
    # Adjust for older data
    df['combined_return'] = df['combined_return'].where(df.index.isin(df.index[-5:]), 
                                                         0.5 * df['intraday_range'] + 0.5 * df['close_to_open_return'])
    
    # Calculate Volume Shock
    df['prev_volume'] = df['volume'].shift(1)
    df['volume_shock'] = (df['volume'] - df['prev_volume']) / df['prev_volume']
    
    # Integrate Volume Shock with Combined Return
    df['final_factor'] = 0.8 * df['combined_return'] + 0.2 * df['volume_shock']
    
    return df['final_factor']
```
