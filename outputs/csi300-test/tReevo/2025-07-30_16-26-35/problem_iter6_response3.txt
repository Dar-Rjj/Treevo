```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Combine Intraday Range and Close-to-Open Return with weights
    recent_weights = 0.7 * df['intraday_range'] + 0.3 * df['close_to_open_return']
    older_weights = 0.5 * df['intraday_range'] + 0.5 * df['close_to_open_return']
    df['combined_factor'] = recent_weights.fillna(older_weights)
    
    # Filter Days with High Volume
    df['20_day_avg_volume'] = df['volume'].rolling(window=20).mean()
    df['high_volume'] = (df['volume'] > 1.5 * df['20_day_avg_volume'])
    
    # Calculate Multi-Period Return
    df['geometric_return'] = (df['close'] / df['close'].shift(5)) ** (1/5) - 1
    
    # Integrate Volume Shocks
    df.loc[df['high_volume'], 'adjusted_geometric_return'] = df['geometric_return'] * 1.1
    df['adjusted_geometric_return'] = df['adjusted_geometric_return'].fillna(df['geometric_return'])
    
    # Final Alpha Factor
    df['alpha_factor'] = df['combined_factor'] * df['adjusted_geometric_return']
    
    return df['alpha_factor']
```
