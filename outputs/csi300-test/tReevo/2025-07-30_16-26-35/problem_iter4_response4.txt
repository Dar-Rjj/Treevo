```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Compute Volume-Weighted High-to-Low Range
    df['volume_weighted_range'] = df['volume'] * df['high_low_range']
    
    # Calculate Rolling Sum of Volume-Weighted High-to-Low Ranges over a 20-day window
    df['rolling_volatility_sum'] = df['volume_weighted_range'].rolling(window=20).sum()
    
    # Calculate Intraday Volatility (20-day moving average of high-to-low ranges)
    df['intraday_volatility'] = df['high_low_range'].rolling(window=20).mean()
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = np.log(df['close'] / df['open'])
    
    # Normalize by Intraday Volatility
    df['normalized_return'] = df['close_to_open_return'] / df['intraday_volatility']
    
    # Calculate 20-day Moving Average of Close-to-Open Returns
    df['ma_close_to_open_return'] = df['close_to_open_return'].rolling(window=20).mean()
    
    # Calculate Momentum Factor
    df['momentum_factor'] = df['ma_close_to_open_return'] - df['close_to_open_return']
    
    # Final Alpha Factor
    df['alpha_factor'] = df['rolling_volatility_sum'] * df['normalized_return'] * df['momentum_factor']
    
    # Drop rows with NaN values resulting from the rolling calculations
    df.dropna(inplace=True)
    
    return df['alpha_factor']
```
