import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']

    # Compute Recent Intraday Range with dynamic weighting
    recent_intraday_range = 0.6 * df['Intraday_Range'] + 0.4 * df['Intraday_Range'].shift(1)

    # Compute Recent Close-to-Open Return with dynamic weighting
    recent_close_to_open_return = 0.3 * df['Close_to_Open_Return'] + 0.2 * df['Close_to_Open_Return'].shift(1)

    # Compute Recent VWAP with dynamic weighting
    recent_vwap = 0.1 * df['VWAP'] + 0.4 * df['VWAP'].shift(1)

    # Initial Alpha Factor
    initial_alpha_factor = recent_intraday_range + recent_close_to_open_return + recent_vwap

    # Incorporate Recency and Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Adjust Alpha Factor based on Volume Ratio
    adjusted_alpha_factor = initial_alpha_factor * (1 + (volume_ratio - 1) * 0.5)
    
    # Incorporate Volume Momentum
    volume_momentum = (volume_ratio - 1).clip(lower=0)
    
    # Final Alpha Factor
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.1
    
    return final_alpha_factor
