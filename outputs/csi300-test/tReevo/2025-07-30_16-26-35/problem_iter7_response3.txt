```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Incorporate Volume Shock
    df['volume_ma'] = df['volume'].rolling(window=20).mean()
    df['volume_shock'] = (df['volume'] - df['volume_ma']) / df['volume_ma']
    
    # Define weights for recent and older data
    recent_weights = [0.5, 0.3, 0.2]  # weights for intraday range, close-to-open return, volume shock
    older_weights = [0.4, 0.4, 0.2]
    
    # Combine Intraday Range, Close-to-Open Return, and Volume Shock
    df['recent_factor'] = (recent_weights[0] * df['intraday_range'] +
                           recent_weights[1] * df['close_to_open_return'] +
                           recent_weights[2] * df['volume_shock'])
    
    df['older_factor'] = (older_weights[0] * df['intraday_range'] +
                          older_weights[1] * df['close_to_open_return'] +
                          older_weights[2] * df['volume_shock'])
    
    # Exponential Weighting
    decay_factor = 0.95
    df['factor'] = df['recent_factor'] * (decay_factor ** np.arange(len(df)))
    df['factor'] += df['older_factor'] * (decay_factor ** np.arange(len(df)))
    
    # Normalize the factor
    df['factor'] = (df['factor'] - df['factor'].mean()) / df['factor'].std()
    
    return df['factor']
```
