import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    df['close_to_open_return'].fillna(0, inplace=True)  # Handle the first row
    
    # Define weights
    recent_weight_intraday = 0.7
    recent_weight_close_to_open = 0.3
    older_weight_intraday = 0.5
    older_weight_close_to_open = 0.5
    
    # Adaptive Exponential Weighting
    def adaptive_exponential_weighting(series, recent_weight, older_weight):
        alpha = 0.9
        weighted_series = series.copy()
        for i in range(1, len(weighted_series)):
            if i < 10:  # Define 'recent' data as the last 10 days
                weighted_series.iloc[i] = (recent_weight * series.iloc[i] + 
                                         alpha * weighted_series.iloc[i-1])
            else:
                weighted_series.iloc[i] = (older_weight * series.iloc[i] + 
                                         alpha * weighted_series.iloc[i-1])
        return weighted_series
    
    # Apply weights to intraday range and close-to-open return
    df['weighted_intraday_range'] = adaptive_exponential_weighting(
        df['intraday_range'], recent_weight_intraday, older_weight_intraday)
    df['weighted_close_to_open_return'] = adaptive_exponential_weighting(
        df['close_to_open_return'], recent_weight_close_to_open, older_weight_close_to_open)
    
    # Final Alpha Factor
    df['intraday_volatility_adjusted_return'] = (df['weighted_intraday_range'] + 
                                                 df['weighted_close_to_open_return'])
    
    return df['intraday_volatility_adjusted_return']
