```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily High-to-Low Range
    daily_range = df['high'] - df['low']
    
    # Compute Weighted High-to-Low Range
    weighted_daily_range = df['volume'] * daily_range
    
    # Calculate Rolling Sum of Weighted High-to-Low Ranges
    rolling_sum_weighted_range = weighted_daily_range.rolling(window=20).sum()
    
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] / df['open']) - 1
    
    # Combine Intraday Range and Return
    combined_intraday_return = close_to_open_return / intraday_range
    
    # Adjust for Momentum
    # Calculate 20-day Moving Average of Close-to-Open Return
    moving_avg_close_to_open = close_to_open_return.rolling(window=20).mean()
    
    # Calculate Momentum Factor
    momentum_factor = moving_avg_close_to_open - close_to_open_return
    
    # Final Alpha Factor
    alpha_factor = (rolling_sum_weighted_range * combined_intraday_return) * momentum_factor
    
    return alpha_factor
```
