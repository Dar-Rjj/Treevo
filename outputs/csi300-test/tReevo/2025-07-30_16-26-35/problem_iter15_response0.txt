import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Weighted Combination of Intraday Range and Close-to-Open Return
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}
    
    df['weighted_combination'] = (
        df['intraday_range'] * recent_weights['intraday_range'] + 
        df['close_to_open_return'] * recent_weights['close_to_open_return']
    )
    
    # Integrate Momentum and Volume Trends
    # Calculate 20-day Simple Moving Average (SMA) of Close prices
    df['sma_20'] = df['close'].rolling(window=20).mean()
    
    # Calculate 14-day Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi_14'] = 100 - (100 / (1 + rs))
    
    # Calculate Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Combine Momentum, Volume, and Intraday Volatility Adjusted Return
    momentum_weight = 0.3
    volume_weight = 0.3
    intraday_volatility_weight = 0.4
    
    df['momentum_trend'] = (df['close'] - df['sma_20']) / df['sma_20']
    df['combined_factor'] = (
        df['weighted_combination'] * intraday_volatility_weight +
        df['momentum_trend'] * momentum_weight +
        df['volume_change'] * volume_weight
    )
    
    return df['combined_factor']
