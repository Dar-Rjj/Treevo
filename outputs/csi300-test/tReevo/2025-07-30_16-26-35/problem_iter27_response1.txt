```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Define Time Window
    recent_days = 10
    older_days = 30
    
    # Compute Moving Averages
    df['recent_sma'] = df['close'].rolling(window=recent_days).mean()
    df['older_sma'] = df['close'].rolling(window=older_days).mean()
    
    # Weighted Combination
    df['weighted_intraday_range'] = np.where(df.index.isin(df.index[-recent_days:]), 
                                              0.7 * df['intraday_range'], 
                                              0.5 * df['intraday_range'])
    df['weighted_close_to_open_return'] = np.where(df.index.isin(df.index[-recent_days:]), 
                                                    0.3 * df['close_to_open_return'], 
                                                    0.5 * df['close_to_open_return'])
    
    # Calculate Alpha Factor
    df['alpha_factor'] = (df['weighted_intraday_range'] + df['weighted_close_to_open_return']) * (df['recent_sma'] / df['older_sma'])
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor']
```
