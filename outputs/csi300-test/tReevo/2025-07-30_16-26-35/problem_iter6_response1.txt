```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    prev_close = df['close'].shift(1)
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Determine weights based on recency
    recent_weights = 0.7 * intraday_range + 0.3 * close_to_open_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_open_return
    
    # Apply dynamic weighting
    combined_returns = pd.Series(index=df.index, dtype=float)
    for i in range(1, len(df)):
        if i < 22:
            combined_returns.iloc[i] = recent_weights.iloc[i]
        else:
            combined_returns.iloc[i] = older_weights.iloc[i]
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range.rolling(window=22).mean()
    
    # Calculate Adjusted Momentum
    adjusted_momentum = combined_returns.rolling(window=20).sum() / intraday_volatility
    
    return adjusted_momentum
```
