```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Incorporate Volume and Amount
    df['Liquidity_Measure'] = df['volume'] / df['amount']

    # Determine if the data is recent or not
    df['Recent'] = df.index.isin(df.index[-5:])

    # Dynamic Weighting Based on Recency
    df['Weighted_Factor'] = 0.0
    df.loc[df['Recent'], 'Weighted_Factor'] = 0.6 * df['Intraday_Range'] + 0.3 * df['Close_to_Open_Return'] + 0.1 * df['Liquidity_Measure']
    df.loc[~df['Recent'], 'Weighted_Factor'] = 0.4 * df['Intraday_Range'] + 0.4 * df['Close_to_Open_Return'] + 0.2 * df['Liquidity_Measure']

    # Assess Market Conditions
    # Calculate Daily Volatility
    df['Daily_Volatility'] = df['close'].pct_change().rolling(window=5).std()

    # Determine Market Trend
    df['Market_Trend'] = df['close'].rolling(window=5).mean().pct_change()

    # Apply Market Trend Multipliers
    df['Trend_Multiplier'] = 1.0
    df.loc[df['Market_Trend'] > 0, 'Trend_Multiplier'] = 1.2
    df.loc[df['Market_Trend'] < 0, 'Trend_Multiplier'] = 0.8

    # Final Alpha Factor Adjustment
    df['Final_Factor'] = df['Weighted_Factor'] * df['Trend_Multiplier']

    return df['Final_Factor']
```
