```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Compute weighted combination based on recency
    df['recent_weights_intraday'] = 0.6
    df['recent_weights_close_to_open'] = 0.4
    df['older_weights_intraday'] = 0.5
    df['older_weights_close_to_open'] = 0.5
    
    # Apply weights
    df['weighted_intraday'] = df['intraday_range'] * df['recent_weights_intraday']
    df['weighted_close_to_open'] = df['close_to_open_return'] * df['recent_weights_close_to_open']
    
    # Adjust for older data
    df.loc[df.index < df.index[-1] - pd.DateOffset(days=7), 'weighted_intraday'] = \
        df.loc[df.index < df.index[-1] - pd.DateOffset(days=7), 'intraday_range'] * df['older_weights_intraday']
    df.loc[df.index < df.index[-1] - pd.DateOffset(days=7), 'weighted_close_to_open'] = \
        df.loc[df.index < df.index[-1] - pd.DateOffset(days=7), 'close_to_open_return'] * df['older_weights_close_to_open']
    
    # Combine the weighted values
    df['combined_factor'] = df['weighted_intraday'] + df['weighted_close_to_open']
    
    return df['combined_factor'].dropna()
```
