```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate 5-Day Moving Averages
    ma_intraday_range = intraday_range.rolling(window=5).mean()
    ma_close_to_open_return = close_to_open_return.rolling(window=5).mean()
    
    # Dynamically Adjust Weights Based on Volume
    volume_ma = df['volume'].rolling(window=5).mean()
    weights = []
    for i in range(len(df)):
        if df['volume'].iloc[i] > volume_ma.iloc[i]:
            weights.append((0.7, 0.3))
        else:
            weights.append((0.5, 0.5))
    
    # Combine Intraday Range and Close-to-Open Return with Weights
    weighted_factors = (intraday_range * [w[0] for w in weights]) + (close_to_open_return * [w[1] for w in weights])
    
    # Integrate Multi-Period Trend
    multi_period_trend = (ma_intraday_range * [w[0] for w in weights]) + (ma_close_to_open_return * [w[1] for w in weights])
    
    # Simplify with Key Price Movements
    # Assuming support and resistance levels are available
    # For simplicity, we use a fixed value. In practice, these should be dynamically determined.
    support = df['low'].min()
    resistance = df['high'].max()
    
    key_price_adjustment = []
    for i in range(len(df)):
        if df['high'].iloc[i] > resistance or df['low'].iloc[i] < support:
            key_price_adjustment.append(1.1)  # Increase factor if key level is breached
        else:
            key_price_adjustment.append(1.0)
    
    # Final Alpha Factor
    alpha_factor = (weighted_factors + multi_period_trend) * key_price_adjustment
    
    return alpha_factor
```
