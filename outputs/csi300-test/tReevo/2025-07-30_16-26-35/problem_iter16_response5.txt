```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low'])
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Assign weights based on recency
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}
    
    # Combine Intraday Range and Close-to-Open Return
    def combine_intraday_and_close_to_open(row, weights):
        return (weights['intraday_range'] * row['intraday_range']) + (weights['close_to_open_return'] * row['close_to_open_return'])
    
    # Determine if the data is recent or old
    recent_threshold = df.index.max() - pd.DateOffset(days=30)  # Example: Recent data is from the last 30 days
    df['combined_factor'] = df.apply(lambda row: combine_intraday_and_close_to_open(row, recent_weights) if row.name > recent_threshold else combine_intraday_and_close_to_open(row, older_weights), axis=1)
    
    # Integrate Market Trends
    df['10_day_momentum'] = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Liquidity Factor
    df['turnover_ratio'] = (df['volume'] * df['close']) / (df['amount'] / df['close'])  # Assuming amount is in dollars
    df['10_day_avg_turnover_ratio'] = df['turnover_ratio'].rolling(window=10).mean()
    
    # Final Alpha Factor Calculation
    df['alpha_factor'] = df['combined_factor'] + df['10_day_momentum']
    df['alpha_factor'] = df['alpha_factor'] * df['10_day_avg_turnover_ratio']
    
    return df['alpha_factor']
```
