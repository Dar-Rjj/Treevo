```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Determine recent or older weights
    df['recent_weight_intraday'] = 0.7
    df['recent_weight_close_open'] = 0.3
    df['older_weight_intraday'] = 0.5
    df['older_weight_close_open'] = 0.5
    
    # Combine Intraday Range and Close-to-Open Return with weights
    df['weighted_intraday_range'] = df.apply(lambda row: row['intraday_range'] * row['recent_weight_intraday'] if row.name == df.index[-1] else row['intraday_range'] * row['older_weight_intraday'], axis=1)
    df['weighted_close_to_open_return'] = df.apply(lambda row: row['close_to_open_return'] * row['recent_weight_close_open'] if row.name == df.index[-1] else row['close_to_open_return'] * row['older_weight_close_open'], axis=1)
    
    # Final Alpha Factor
    df['alpha_factor'] = df['weighted_intraday_range'] + df['weighted_close_to_open_return']
    
    return df['alpha_factor']
```
