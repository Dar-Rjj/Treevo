```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Compute Liquidity Measure
    df['liquidity'] = df['volume'] / df['amount']
    
    # Define weights based on recency and liquidity
    recent_weights = [0.6, 0.3, 0.1]
    older_weights = [0.5, 0.4, 0.1]
    
    # Weighted Combination
    df['recent_factor'] = (recent_weights[0] * df['intraday_range'] + 
                            recent_weights[1] * df['close_to_open_return'] + 
                            recent_weights[2] * df['liquidity'])
    
    df['older_factor'] = (older_weights[0] * df['intraday_range'].shift(1) + 
                          older_weights[1] * df['close_to_open_return'].shift(1) + 
                          older_weights[2] * df['liquidity'].shift(1))
    
    # Combine the factors
    df['alpha_factor'] = df['recent_factor'] + df['older_factor'].shift(1)
    
    return df['alpha_factor'].dropna()
```
