```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']

    # Calculate Multi-Period Momentum
    momentum_1d = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    momentum_2d = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    momentum_5d = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Calculate Multi-Period Liquidity
    liquidity_1d = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    liquidity_2d = (df['volume'] - df['volume'].shift(2)) / df['volume'].shift(2)
    liquidity_5d = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)

    # Combine components with weights
    weighted_intraday_range = 0.4 * intraday_range
    weighted_close_to_open_return = 0.2 * close_to_open_return
    weighted_vwap = 0.1 * vwap
    weighted_momentum = 0.067 * momentum_1d + 0.067 * momentum_2d + 0.067 * momentum_5d
    weighted_liquidity = 0.05 * liquidity_1d + 0.05 * liquidity_2d + 0.05 * liquidity_5d

    # Final Alpha Factor
    alpha_factor = (weighted_intraday_range + weighted_close_to_open_return +
                    weighted_vwap + weighted_momentum + weighted_liquidity)

    return alpha_factor
```
