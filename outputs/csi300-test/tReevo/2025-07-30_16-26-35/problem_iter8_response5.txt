```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Define weights for recent and older data
    recent_weights = [0.7, 0.3]  # [Intraday Range, Close-to-Open Return]
    older_weights = [0.5, 0.5]   # [Intraday Range, Close-to-Open Return]
    
    # Determine if the data is recent or older
    df['is_recent'] = df.index >= df.index[-1] - pd.Timedelta(days=30)
    
    # Compute the weighted combination
    df['weighted_score'] = (
        df['is_recent'] * (recent_weights[0] * df['intraday_range'] + recent_weights[1] * df['close_to_open_return'])
        + (~df['is_recent']) * (older_weights[0] * df['intraday_range'] + older_weights[1] * df['close_to_open_return'])
    )
    
    # Long-Term Trend Analysis
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['ma_100'] = df['close'].rolling(window=100).mean()
    
    # Adjust the final combined score based on the trend
    df['trend_adjustment'] = df.apply(
        lambda row: 0.1 if row['ma_20'] > row['ma_100'] else -0.1,
        axis=1
    )
    
    df['final_factor'] = df['weighted_score'] + df['trend_adjustment']
    
    return df['final_factor']
```
