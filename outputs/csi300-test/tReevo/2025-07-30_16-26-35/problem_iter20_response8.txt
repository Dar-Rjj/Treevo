import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range: High - Low
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: (Close - Open) / Open
    close_to_open_return = (df['close'] - df.shift(1)['open']) / df.shift(1)['open']
    
    # Calculate Volume-Weighted Average Price (VWAP): (High + Low + Close) / 3 * Volume
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Intraday Momentum: (Close - Open)
    intraday_momentum = df['close'] - df['open']

    # Compute Recent Intraday Range with Dynamic Weighting
    recent_intraday_range = intraday_range.ewm(span=10, adjust=False).mean() * 0.5 + \
                             intraday_range.ewm(span=30, adjust=False).mean() * 0.5
    
    # Compute Recent Close-to-Open Return with Dynamic Weighting
    recent_close_to_open_return = close_to_open_return.ewm(span=10, adjust=False).mean() * 0.3 + \
                                  close_to_open_return.ewm(span=30, adjust=False).mean() * 0.2
    
    # Compute Recent VWAP with Dynamic Weighting
    recent_vwap = vwap.ewm(span=10, adjust=False).mean() * 0.1 + \
                  vwap.ewm(span=30, adjust=False).mean() * 0.4
    
    # Compute Recent Intraday Momentum with Dynamic Weighting
    recent_intraday_momentum = intraday_momentum.ewm(span=10, adjust=False).mean() * 0.1 + \
                               intraday_momentum.ewm(span=30, adjust=False).mean() * 0.4
    
    # Initial Alpha Factor: (Recent Intraday Range + Recent Close-to-Open Return + Recent VWAP + Recent Intraday Momentum)
    initial_alpha_factor = (recent_intraday_range + recent_close_to_open_return + recent_vwap + recent_intraday_momentum)
    
    # Compute Volume Ratio: Volume_t / Volume_t-1
    volume_ratio = df['volume'] / df.shift(1)['volume']
    
    # Adjust Alpha Factor based on Volume Ratio
    adjusted_alpha_factor = initial_alpha_factor * (1 + (volume_ratio - 1) * 0.5)
    
    # Incorporate Advanced Volume Momentum
    volume_momentum = (volume_ratio - 1).clip(lower=0)
    
    # Final Alpha Factor: Adjusted Alpha Factor + Volume Momentum * 0.1
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.1
    
    return final_alpha_factor
