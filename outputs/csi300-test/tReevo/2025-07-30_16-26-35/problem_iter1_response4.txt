```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily Return using Close price
    df['daily_return'] = df['close'].pct_change()

    # Apply Moving Average on Returns for Short-Term and Long-Term
    short_term_window = 5
    long_term_window = 20

    df['short_term_momentum'] = df['daily_return'].rolling(window=short_term_window).mean()
    df['long_term_momentum'] = df['daily_return'].rolling(window=long_term_window).mean()

    # Calculate Price Momentum (use the difference between short-term and long-term momentum)
    df['price_momentum'] = df['short_term_momentum'] - df['long_term_momentum']

    # Compute Daily Volume Change
    df['daily_volume_change'] = df['volume'].pct_change()

    # Apply Moving Average on Volume Changes for Short-Term and Long-Term
    df['short_term_volume_trend'] = df['daily_volume_change'].rolling(window=short_term_window).mean()
    df['long_term_volume_trend'] = df['daily_volume_change'].rolling(window=long_term_window).mean()

    # Calculate Volume Synchronized Trend (use the difference between short-term and long-term volume trends)
    df['volume_synchronized_trend'] = df['short_term_volume_trend'] - df['long_term_volume_trend']

    # Combine Price Momentum and Volume Synchronized Trend
    df['combined_indicator'] = df['price_momentum'] * df['volume_synchronized_trend']

    # Filter by Price Movement
    df['price_range'] = (df['high'] - df['low']) / df['close']

    # Final Factor Value
    df['factor_value'] = df['combined_indicator'] * df['price_range']

    return df['factor_value'].dropna()
```
