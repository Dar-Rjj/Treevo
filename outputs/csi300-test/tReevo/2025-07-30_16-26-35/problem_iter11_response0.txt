```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range (High - Low)
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return (Close - Previous Day's Close) / Previous Day's Close
    df['prev_close'] = df['close'].shift(1)
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Intraday Range and Close-to-Open Return
    df['recent_intraday_weighted'] = 0.7 * df['intraday_range']
    df['recent_close_to_open_weighted'] = 0.3 * df['close_to_open_return']
    df['older_intraday_weighted'] = 0.5 * df['intraday_range']
    df['older_close_to_open_weighted'] = 0.5 * df['close_to_open_return']
    
    # Combine Intraday Range and Close-to-Open Return with Weights
    df['alpha_factor'] = df['recent_intraday_weighted'] + df['recent_close_to_open_weighted']
    df['alpha_factor'] = df['alpha_factor'].fillna(df['older_intraday_weighted'] + df['older_close_to_open_weighted'])
    
    return df['alpha_factor']
```
