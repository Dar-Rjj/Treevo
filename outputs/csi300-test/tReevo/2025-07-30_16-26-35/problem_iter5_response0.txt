import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = (df['high'] - df['low']) / df['close']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Define weights for recent and older data
    recent_weights = [0.7, 0.3]  # 0.7 for intraday range, 0.3 for close to open return
    older_weights = [0.5, 0.5]   # Equal weight for both
    
    # Function to apply weights based on the date
    def apply_weights(row):
        if row.name < df.index[-10]:  # Older data
            return older_weights[0] * row['intraday_range'] + older_weights[1] * row['close_to_open_return']
        else:  # Recent data
            return recent_weights[0] * row['intraday_range'] + recent_weights[1] * row['close_to_open_return']

    # Combine intraday range and close-to-open return with respective weights
    df['alpha_factor'] = df.apply(apply_weights, axis=1)
    
    # Drop the last row which is NaN due to the shift operation
    df.dropna(inplace=True)
    
    return df['alpha_factor']
