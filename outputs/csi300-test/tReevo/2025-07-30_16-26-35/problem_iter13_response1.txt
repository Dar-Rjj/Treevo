import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Dynamic Weighting Based on Recent Trends
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}
    
    # Apply dynamic weighting
    df['recent_factor'] = recent_weights['intraday_range'] * df['intraday_range'] + \
                          recent_weights['close_to_open_return'] * df['close_to_open_return']
    df['older_factor'] = older_weights['intraday_range'] * df['intraday_range'] + \
                         older_weights['close_to_open_return'] * df['close_to_open_return']
    
    # Integrate Market Sentiment
    df['volume_change'] = df['volume'].pct_change()
    df['price_volatility'] = df['close'].rolling(window=10).std()
    
    # Final factor with market sentiment
    df['final_factor'] = df['recent_factor'] + df['volume_change'] + df['price_volatility']
    
    return df['final_factor']
