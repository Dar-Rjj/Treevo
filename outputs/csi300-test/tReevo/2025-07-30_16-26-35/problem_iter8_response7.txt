```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range: High - Low
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: (Close - Open) / Open
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Determine recent and older periods
    # For simplicity, we'll use the median date to split the data into two halves
    median_date = df.index[int(len(df) / 2)]
    recent_period = df.index > median_date
    
    # Recent Weights: 0.7 for Intraday Range, 0.3 for Close-to-Open Return
    recent_factor = 0.7 * intraday_range + 0.3 * close_to_open_return
    recent_factor[~recent_period] = 0  # Set to 0 for older periods
    
    # Older Weights: 0.5 for Intraday Range, 0.5 for Close-to-Open Return
    older_factor = 0.5 * intraday_range + 0.5 * close_to_open_return
    older_factor[recent_period] = 0  # Set to 0 for recent periods
    
    # Combine the factors
    factor_values = recent_factor + older_factor
    
    return factor_values
```
