```python
import pandas as pd

def heuristics_v2(df, lookback_period=10):
    # Calculate Daily Range
    df['daily_range'] = df['high'] - df['low']
    
    # Calculate Upper and Lower Breakout Levels
    df['upper_breakout'] = df['close'] + (df['daily_range'] * 0.5)
    df['lower_breakout'] = df['close'] - (df['daily_range'] * 0.5)
    
    # Calculate Distance to Breakout Levels
    df['distance_to_upper'] = df['upper_breakout'] - df['close']
    df['distance_to_lower'] = df['close'] - df['lower_breakout']
    
    # Assign Weight Based on Volume and Price
    df['weighted_distance_upper'] = df['distance_to_upper'] * df['volume'] * df['close']
    df['weighted_distance_lower'] = df['distance_to_lower'] * df['volume'] * df['close']
    
    # Calculate Final Alpha Factor
    df['alpha_factor'] = df['weighted_distance_upper'] - df['weighted_distance_lower']
    alpha_factor = df['alpha_factor'].rolling(window=lookback_period).sum()
    
    return alpha_factor
```
