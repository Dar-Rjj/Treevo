```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] / df['open']) - 1
    
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']
    
    # Normalize Intraday Range by Intraday Volatility
    normalized_range = intraday_range / intraday_volatility
    
    # Multiply by Close-to-Open Return
    combined_factor = normalized_range * close_to_open_return
    
    # Weight by Volume
    volume_weighted_factor = combined_factor * df['volume']
    
    # Integrate Momentum: Use Close prices over multiple periods
    momentum = df['close'].pct_change(periods=5)  # 5-day momentum as an example
    momentum.fillna(0, inplace=True)
    
    # Consider Market Sentiment: Use High and Low prices
    market_sentiment = (df['high'] + df['low']) / 2 - df['close']
    market_sentiment = market_sentiment.rolling(window=5).mean()  # 5-day moving average for sentiment
    market_sentiment.fillna(0, inplace=True)
    
    # Final Alpha Factor
    final_factor = (volume_weighted_factor * momentum) + market_sentiment
    
    return final_factor
```
