```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate Short-Term Momentum
    short_term_momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Calculate Long-Term Momentum
    long_term_momentum = (df['close'] - df['close'].shift(20)) / df['close'].shift(20)
    
    # Compute Weighted Metrics
    weighted_intraday_range = intraday_range * 0.3
    weighted_close_to_open_return = close_to_open_return * 0.2
    weighted_vwap = vwap * 0.2
    weighted_short_term_momentum = short_term_momentum * 0.15
    weighted_long_term_momentum = long_term_momentum * 0.15
    
    # Final Alpha Factor
    alpha_factor = (weighted_intraday_range + weighted_close_to_open_return + 
                    weighted_vwap + weighted_short_term_momentum + 
                    weighted_long_term_momentum)
    
    return alpha_factor
```
