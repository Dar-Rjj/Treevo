```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['IntradayRange'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['CloseToOpenReturn'] = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['VWAP'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Assign weights for recent and older data
    recent_weight = 0.6
    older_weight = 0.4
    recent_intraday_range = df['IntradayRange'] * recent_weight + df['IntradayRange'].shift(1) * older_weight
    recent_close_to_open_return = df['CloseToOpenReturn'] * 0.3 + df['CloseToOpenReturn'].shift(1) * 0.2
    recent_vwap = df['VWAP'] * 0.1 + df['VWAP'].shift(1) * 0.4
    
    # Initial Alpha Factor
    initial_alpha_factor = recent_intraday_range + recent_close_to_open_return + recent_vwap
    
    # Incorporate Recency and Volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = initial_alpha_factor * (1 + (volume_ratio - 1) * 0.5)
    
    # Incorporate Volume Momentum
    volume_momentum = (volume_ratio - 1).clip(lower=0)
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.1
    
    return final_alpha_factor
```
