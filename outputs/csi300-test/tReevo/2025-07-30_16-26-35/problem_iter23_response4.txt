import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute the previous day's close price
    df['prev_close'] = df['close'].shift(1)
    
    # Volume Weighting
    df['volume_weight'] = 1.5 * (df['volume'] > df['volume'].mean()) + 0.5 * (df['volume'] <= df['volume'].mean())
    
    # Combine Intraday Range and Close-to-Open Return with dynamic weights
    df['uptrend'] = (df['close'] > df['prev_close']).astype(int)
    df['downtrend'] = (df['close'] < df['prev_close']).astype(int)
    
    # Initialize the factor
    df['factor'] = 0
    
    # Dynamic weighting based on market conditions
    df.loc[df['uptrend'] == 1, 'factor'] = (0.8 * df['intraday_range'] + 0.2 * df['close_to_open_return']) * df['volume_weight']
    df.loc[df['downtrend'] == 1, 'factor'] = (0.4 * df['intraday_range'] + 0.6 * df['close_to_open_return']) * df['volume_weight']
    
    # For neutral trend, use the average of recent and older weights
    df.loc[(df['uptrend'] == 0) & (df['downtrend'] == 0), 'factor'] = (
        0.7 * df['intraday_range'] + 0.3 * df['close_to_open_return']) * df['volume_weight']
    
    return df['factor']
