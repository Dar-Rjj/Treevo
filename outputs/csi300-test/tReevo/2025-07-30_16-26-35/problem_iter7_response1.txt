```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range: High - Low
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: (Open - Previous Day's Close) / Previous Day's Close
    close_to_open_return = (df['open'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Integrate Volume Shock: (Current Day's Volume - Previous Day's Volume) / Previous Day's Volume
    volume_change = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Exponential Weighting for Intraday Range
    recent_weights_intraday_range = 0.7
    older_weights_intraday_range = 0.5
    weighted_intraday_range = intraday_range.ewm(alpha=recent_weights_intraday_range, adjust=False).mean() * recent_weights_intraday_range + \
                              intraday_range.ewm(alpha=older_weights_intraday_range, adjust=False).mean() * (1 - recent_weights_intraday_range)
    
    # Exponential Weighting for Close-to-Open Return
    recent_weights_close_to_open = 0.3
    older_weights_close_to_open = 0.5
    weighted_close_to_open_return = close_to_open_return.ewm(alpha=recent_weights_close_to_open, adjust=False).mean() * recent_weights_close_to_open + \
                                   close_to_open_return.ewm(alpha=older_weights_close_to_open, adjust=False).mean() * (1 - recent_weights_close_to_open)
    
    # Exponential Weighting for Volume Change
    recent_weights_volume_change = 0.5
    older_weights_volume_change = 0.5
    weighted_volume_change = volume_change.ewm(alpha=recent_weights_volume_change, adjust=False).mean() * recent_weights_volume_change + \
                            volume_change.ewm(alpha=older_weights_volume_change, adjust=False).mean() * (1 - recent_weights_volume_change)
    
    # Combine Weighted Intraday Range, Close-to-Open Return, and Volume Change
    final_alpha_factor = (weighted_intraday_range + 
                          weighted_close_to_open_return + 
                          weighted_volume_change)
    
    return final_alpha_factor
```
