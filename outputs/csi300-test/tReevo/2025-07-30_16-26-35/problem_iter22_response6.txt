```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the relative strength index (RSI) for the closing prices
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the average true range (ATR) using high, low and close prices
    tr = pd.DataFrame(index=df.index)
    tr['hl'] = df['high'] - df['low']
    tr['hc'] = abs(df['high'] - df['close'].shift())
    tr['lc'] = abs(df['low'] - df['close'].shift())
    tr['tr'] = tr.max(axis=1)
    atr = tr['tr'].rolling(window=14).mean()
    
    # Calculate the money flow index (MFI) using typical price and volume
    tp = (df['high'] + df['low'] + df['close']) / 3
    mf = tp * df['volume']
    positive_mf = mf.where(df['close'] > df['close'].shift(), 0)
    negative_mf = mf.where(df['close'] < df['close'].shift(), 0)
    mfr = positive_mf.rolling(window=14).sum() / negative_mf.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + mfr))
    
    # Combine the three indicators to create a novel alpha factor
    alpha_factor = (rsi - 50) / 50 * atr + mfi
    
    return alpha_factor
```
