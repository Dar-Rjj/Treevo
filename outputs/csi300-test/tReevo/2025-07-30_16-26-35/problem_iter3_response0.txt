```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Volume Weighted Average Price (VWAP)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    vwap = (typical_price * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Calculate Intraday Volatility
    intraday_volatility = intraday_range.rolling(window=20).std()
    
    # Normalize by Intraday Volatility
    normalized_intraday_range = intraday_range / intraday_volatility
    normalized_vwap = (vwap - vwap.rolling(window=20).mean()) / vwap.rolling(window=20).std()
    
    # Multiply by Close-to-Open Return
    combined_factor = normalized_intraday_range * close_to_open_return * normalized_vwap
    
    # Incorporate Momentum
    ma_5 = combined_factor.rolling(window=5).mean()
    ma_20 = combined_factor.rolling(window=20).mean()
    momentum = ma_5 - ma_20
    
    return momentum
```
