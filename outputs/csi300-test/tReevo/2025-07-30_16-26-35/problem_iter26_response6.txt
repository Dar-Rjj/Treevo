```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open'].shift(-1)) / df['open'].shift(-1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Compute Recent Intraday Range with dynamic weighting
    recent_intraday_range = df['intraday_range'].rolling(window=2).mean() * 0.5 + df['intraday_range'].shift(2).rolling(window=2).mean() * 0.3
    
    # Compute Recent Close-to-Open Return with dynamic weighting
    recent_close_to_open_return = df['close_to_open_return'].rolling(window=2).mean() * 0.3 + df['close_to_open_return'].shift(2).rolling(window=2).mean() * 0.2
    
    # Compute Recent VWAP with dynamic weighting
    recent_vwap = df['vwap'].rolling(window=2).mean() * 0.1 + df['vwap'].shift(2).rolling(window=2).mean() * 0.4
    
    # Compute Recent Intraday Momentum with dynamic weighting
    recent_intraday_momentum = df['intraday_momentum'].rolling(window=2).mean() * 0.1 + df['intraday_momentum'].shift(2).rolling(window=2).mean() * 0.4
    
    # Combine to get the Initial Alpha Factor
    initial_alpha_factor = (recent_intraday_range + 
                            recent_close_to_open_return + 
                            recent_vwap + 
                            recent_intraday_momentum)
    
    # Incorporate Recency and Volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = initial_alpha_factor * (1 + (df['volume_ratio'] - 1) * 0.6)
    
    # Incorporate Advanced Volume Metrics
    volume_momentum = np.maximum(0, df['volume_ratio'] - 1)
    final_alpha_factor = (adjusted_alpha_factor + 
                          volume_momentum * 0.2)
    
    # Incorporate Additional Volume-Derived Metrics
    volume_growth = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    final_alpha_factor += volume_growth * 0.1
    
    # Incorporate Price and Volume Interaction
    price_volume_corr = df[['high', 'low', 'close']].apply(lambda x: x.corr(df['volume']), axis=0).mean()
    final_alpha_factor += price_volume_corr * 0.1
    
    return final_alpha_factor
```
