```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']

    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']

    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']

    # Calculate Momentum
    momentum_1day = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    momentum_2day = (df['close'] - df['close'].shift(2)) / df['close'].shift(2)
    momentum_5day = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)

    # Calculate Liquidity
    volume_change_1day = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    volume_change_2day = (df['volume'] - df['volume'].shift(2)) / df['volume'].shift(2)
    volume_change_5day = (df['volume'] - df['volume'].shift(5)) / df['volume'].shift(5)

    # Compute Weighted Factors
    weighted_intraday_range = 0.4 * intraday_range
    weighted_close_to_open_return = 0.2 * close_to_open_return
    weighted_vwap = 0.1 * vwap
    weighted_momentum = 0.1 * momentum_1day + 0.1 * momentum_2day + 0.1 * momentum_5day
    weighted_liquidity = 0.05 * volume_change_1day + 0.05 * volume_change_2day + 0.05 * volume_change_5day

    # Final Alpha Factor
    alpha_factor = (weighted_intraday_range + 
                    weighted_close_to_open_return + 
                    weighted_vwap + 
                    weighted_momentum + 
                    weighted_liquidity)

    return alpha_factor
```
