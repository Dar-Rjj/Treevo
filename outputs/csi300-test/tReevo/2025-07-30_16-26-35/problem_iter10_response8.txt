```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Determine market condition (trending or ranging) for dynamic weighting
    # This is a simplified heuristic: if the 10-day moving average of the close price is increasing, consider the market trending
    df['ma_10'] = df['close'].rolling(window=10).mean()
    df['trend'] = (df['ma_10'] > df['ma_10'].shift(1)).astype(int)
    
    # Define weights based on market condition
    df['weight_intrange'] = df['trend'].apply(lambda x: 0.6 if x == 1 else 0.4)
    df['weight_clopen'] = 1 - df['weight_intrange']
    
    # Combine Intraday Range and Close-to-Open Return with dynamic weights
    factor_value = (intraday_range * df['weight_intrange']) + (close_to_open_return * df['weight_clopen'])
    
    return factor_value
```
