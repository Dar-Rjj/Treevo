```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Intraday Volatility Adjusted Return.
    
    Parameters:
    df (pd.DataFrame): A DataFrame with index as date and columns as open, high, low, close, amount, volume.
    
    Returns:
    pd.Series: A Series with index as date and values as the calculated alpha factor.
    """
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'].pct_change()
    
    # Combine Intraday Range and Return
    # Normalize by Intraday Range
    normalized_return = close_to_open_return / (intraday_range + 1e-8)  # Adding a small epsilon to avoid division by zero
    
    return normalized_return
```
