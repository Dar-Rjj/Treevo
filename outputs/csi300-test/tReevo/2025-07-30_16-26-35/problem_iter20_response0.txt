```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Shift the close_to_open_return to align with the correct day
    df['close_to_open_return'] = df['close_to_open_return'].shift(-1)
    
    # Define recent and older weights
    recent_weights = {'intraday_range': 0.6, 'close_to_open_return': 0.4}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}
    
    # Apply weighted combination
    factor_values = pd.Series(index=df.index)
    
    for i, row in df.iterrows():
        if i < df.index[-5]:
            # Older days
            factor_values[i] = (older_weights['intraday_range'] * row['intraday_range']) + \
                               (older_weights['close_to_open_return'] * row['close_to_open_return'])
        else:
            # Recent days
            factor_values[i] = (recent_weights['intraday_range'] * row['intraday_range']) + \
                               (recent_weights['close_to_open_return'] * row['close_to_open_return'])

    return factor_values
```
