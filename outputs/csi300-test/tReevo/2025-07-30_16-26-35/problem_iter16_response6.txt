```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Compute Liquidity Measure
    df['Liquidity'] = df['volume'] / df['amount']
    
    # Dynamic Weighting
    weights_recent = 0.6 * df['Intraday_Range'] + 0.3 * df['Close_to_Open_Return'] + 0.1 * df['Liquidity']
    weights_older = 0.4 * df['Intraday_Range'] + 0.4 * df['Close_to_Open_Return'] + 0.2 * df['Liquidity']
    
    # Apply dynamic weighting based on recency
    recent_mask = df.index > df.index[-5]
    df['Alpha_Factor'] = 0
    df.loc[recent_mask, 'Alpha_Factor'] = weights_recent[recent_mask]
    df.loc[~recent_mask, 'Alpha_Factor'] = weights_older[~recent_mask]
    
    # Assess Market Conditions
    df['Daily_Volatility'] = df['close'].rolling(window=5).std()
    df['Market_Trend'] = df['close'].rolling(window=5).mean().pct_change()
    
    # Determine Trend and apply multipliers
    trend_multiplier = df['Market_Trend'].apply(lambda x: 1.2 if x > 0 else 0.8)
    df['Alpha_Factor'] *= trend_multiplier
    
    return df['Alpha_Factor']
```
