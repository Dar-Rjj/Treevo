```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range: High - Low
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: (Close - Open) / Open
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Combine Intraday Range and Close-to-Open Return with recent weights
    combined_factor = (intraday_range * 0.7) + (close_to_open_return * 0.3)
    
    return combined_factor
```
