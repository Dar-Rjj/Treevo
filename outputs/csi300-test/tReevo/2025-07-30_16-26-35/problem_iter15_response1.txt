```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()

    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Assign weights based on recency
    recent_weights_intraday = 0.7
    recent_weights_closetoopen = 0.3
    older_weights_intraday = 0.5
    older_weights_closetoopen = 0.5
    
    # Determine the split point for "recent" and "older" data
    split_point = int(len(df) * 0.5)  # You can adjust the split point as needed
    
    # Apply dynamic weighting
    weighted_factor = pd.Series(index=df.index, dtype=float)
    weighted_factor.iloc[:split_point] = (older_weights_intraday * intraday_range.iloc[:split_point] +
                                          older_weights_closetoopen * close_to_open_return.iloc[:split_point])
    weighted_factor.iloc[split_point:] = (recent_weights_intraday * intraday_range.iloc[split_point:] +
                                           recent_weights_closetoopen * close_to_open_return.iloc[split_point:])
    
    return weighted_factor
```
