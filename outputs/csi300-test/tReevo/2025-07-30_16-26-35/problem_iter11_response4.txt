```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']
    
    # Assign weights based on recency
    df['Intraday_Volatility_Adjusted_Return'] = 0.7 * df['Intraday_Range'] + 0.3 * df['Close_to_Open_Return']
    df['Intraday_Volatility_Adjusted_Return_shifted'] = 0.5 * df['Intraday_Range'].shift(1) + 0.5 * df['Close_to_Open_Return'].shift(1)
    
    # Combine the two weighted factors
    df['Final_Factor'] = df['Intraday_Volatility_Adjusted_Return'] + df['Intraday_Volatility_Adjusted_Return_shifted']
    
    # Return the final factor as a pandas Series
    return df['Final_Factor'].dropna()
```
