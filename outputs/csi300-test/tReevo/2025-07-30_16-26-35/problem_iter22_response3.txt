```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Assign weights based on recency
    recent_weights = {'Intraday_Range': 0.6, 'Close_to_Open_Return': 0.4}
    older_weights = {'Intraday_Range': 0.5, 'Close_to_Open_Return': 0.5}
    
    # Weighted Combination of Intraday Range and Close-to-Open Return
    df['Combined_Factor'] = df.apply(lambda row: (row['Intraday_Range'] * recent_weights['Intraday_Range'] + 
                                                  row['Close_to_Open_Return'] * recent_weights['Close_to_Open_Return']) if row.name == df.index[-1] else 
                                                 (row['Intraday_Range'] * older_weights['Intraday_Range'] + 
                                                  row['Close_to_Open_Return'] * older_weights['Close_to_Open_Return']), axis=1)
    
    # Compute Volume Ratio
    df['Volume_Ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust Alpha Factor based on Volume Ratio
    df['Adjusted_Alpha_Factor'] = df['Combined_Factor'] * (1 + (df['Volume_Ratio'] - 1) * 0.5)
    
    # Compute Volume Momentum
    df['Volume_Momentum'] = df['Volume_Ratio'] - 1
    df['Volume_Momentum'] = df['Volume_Momentum'].apply(lambda x: max(0, x))
    
    # Final Alpha Factor with Volume Momentum
    df['Final_Alpha_Factor'] = df['Adjusted_Alpha_Factor'] + df['Volume_Momentum'] * 0.1
    
    return df['Final_Alpha_Factor']
```
