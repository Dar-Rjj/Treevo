import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Assign weights to recent and older data
    recent_weights = 0.7 * intraday_range + 0.3 * close_to_open_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_open_return
    
    # Combine Intraday Range and Close-to-Open Return
    factor_values = recent_weights * 0.6 + older_weights * 0.4
    
    return pd.Series(factor_values, index=df.index)
