import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Compute Liquidity Measure
    liquidity = df['volume'] / df['amount']
    
    # Calculate Daily Volatility using Standard Deviation of Recent Returns
    df['daily_return'] = df['close'].pct_change()
    daily_volatility = df['daily_return'].rolling(window=20).std().fillna(0)
    high_volatility_threshold = daily_volatility.mean() + daily_volatility.std()
    
    # Determine Market Trend using Moving Average of Recent Close Prices
    moving_average = df['close'].rolling(window=50).mean()
    market_trend_positive = df['close'] > moving_average
    market_trend_multiplier = np.where(market_trend_positive, 1.2, 0.8)
    
    # Weighted Combination based on Volatility
    weights = np.where(daily_volatility > high_volatility_threshold,
                       0.6 * intraday_range + 0.3 * close_to_open_return + 0.1 * liquidity,
                       0.4 * intraday_range + 0.4 * close_to_open_return + 0.2 * liquidity)
    
    # Apply Market Trend Multiplier
    final_factor = weights * market_trend_multiplier
    
    return pd.Series(final_factor, index=df.index)
