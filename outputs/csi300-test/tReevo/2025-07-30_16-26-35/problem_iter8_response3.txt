import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Assign weights to recent and older data
    recent_weights = [0.7, 0.3]
    older_weights = [0.5, 0.5]
    
    # Combine Intraday Range and Close-to-Open Return with dynamic weighting
    df['combined_return'] = (
        df['intraday_range'] * recent_weights[0] + 
        df['close_to_open_return'] * recent_weights[1]
    )
    
    # Incorporate Price Momentum
    df['5_day_momentum'] = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Integrate Momentum with Combined Return
    momentum_weight = 0.3
    combined_return_weight = 0.7
    df['momentum_integrated_factor'] = (
        df['combined_return'] * combined_return_weight + 
        df['5_day_momentum'] * momentum_weight
    )
    
    # Incorporate Volume Shocks
    df['volume_shock'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Integrate Volume Shock with Momentum Integrated Factor
    volume_shock_weight = 0.2
    momentum_integrated_factor_weight = 0.8
    df['final_factor'] = (
        df['momentum_integrated_factor'] * momentum_integrated_factor_weight + 
        df['volume_shock'] * volume_shock_weight
    )
    
    return df['final_factor'].dropna()
