```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Compute Weighted Intraday Range
    weighted_intraday_range = intraday_range.rolling(window=2).apply(
        lambda x: x[0] * 0.6 + x[1] * 0.4, raw=True
    )
    
    # Compute Weighted Close-to-Open Return
    weighted_close_to_open_return = close_to_open_return.rolling(window=2).apply(
        lambda x: x[0] * 0.3 + x[1] * 0.2, raw=True
    )
    
    # Compute Weighted VWAP
    weighted_vwap = vwap.rolling(window=2).apply(
        lambda x: x[0] * 0.1 + x[1] * 0.4, raw=True
    )
    
    # Final Alpha Factor
    alpha_factor = (
        weighted_intraday_range + 
        weighted_close_to_open_return + 
        weighted_vwap
    )
    
    return alpha_factor
```
