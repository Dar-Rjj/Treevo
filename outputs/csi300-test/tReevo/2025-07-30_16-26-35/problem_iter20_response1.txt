```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range: High - Low
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return: (Close - Open) / Open
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Determine recent and older days
    recent_days = df.index.isin(df.index[-5:])
    
    # Define weights
    recent_weights = {'intraday_range': 0.7, 'close_to_open_return': 0.3}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.5}
    
    # Initialize the factor series
    factor = pd.Series(index=df.index)
    
    # Calculate the factor for recent and older days
    factor[recent_days] = (recent_weights['intraday_range'] * intraday_range[recent_days]) + \
                          (recent_weights['close_to_open_return'] * close_to_open_return[recent_days])
    factor[~recent_days] = (older_weights['intraday_range'] * intraday_range[~recent_days]) + \
                           (older_weights['close_to_open_return'] * close_to_open_return[~recent_days])
    
    return factor
```
