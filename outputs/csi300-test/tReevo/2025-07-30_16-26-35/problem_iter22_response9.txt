```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']

    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Compute Recent Intraday Range
    recent_intraday_range = df['intraday_range'].ewm(span=10, adjust=False).mean()
    older_intraday_range = df['intraday_range'].ewm(span=50, adjust=False).mean()
    recent_intraday_range = 0.6 * recent_intraday_range + 0.4 * older_intraday_range
    
    # Compute Recent Close-to-Open Return
    recent_close_to_open_return = df['close_to_open_return'].ewm(span=10, adjust=False).mean()
    older_close_to_open_return = df['close_to_open_return'].ewm(span=50, adjust=False).mean()
    recent_close_to_open_return = 0.4 * recent_close_to_open_return + 0.3 * older_close_to_open_return
    
    # Compute Recent VWAP
    recent_vwap = df['vwap'].ewm(span=10, adjust=False).mean()
    older_vwap = df['vwap'].ewm(span=50, adjust=False).mean()
    recent_vwap = 0.2 * recent_vwap + 0.5 * older_vwap
    
    # Compute Recent Intraday Momentum
    recent_intraday_momentum = df['intraday_momentum'].ewm(span=10, adjust=False).mean()
    older_intraday_momentum = df['intraday_momentum'].ewm(span=50, adjust=False).mean()
    recent_intraday_momentum = 0.2 * recent_intraday_momentum + 0.6 * older_intraday_momentum
    
    # Combine Recent Intraday Range, Recent Close-to-Open Return, Recent VWAP, and Recent Intraday Momentum
    initial_alpha_factor = (
        recent_intraday_range +
        recent_close_to_open_return +
        recent_vwap +
        recent_intraday_momentum
    )
    
    # Compute Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Adjust Alpha Factor based on Volume Ratio
    adjusted_alpha_factor = initial_alpha_factor * (1 + (df['volume_ratio'] - 1) * 0.6)
    
    # Incorporate Advanced Volume Momentum
    volume_momentum = (df['volume_ratio'] - 1).clip(lower=0)
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.1
    
    return final_alpha_factor
```
