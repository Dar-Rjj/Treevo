```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Daily Volume Change
    df['volume_change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Exponential Moving Average (EMA) of Intraday Range and Close-to-Open Return
    ema_span = 5
    df['ema_intraday_range'] = df['intraday_range'].ewm(span=ema_span, adjust=False).mean()
    df['ema_close_to_open_return'] = df['close_to_open_return'].ewm(span=ema_span, adjust=False).mean()
    
    # Adaptive Weighting Based on Recent Volatility
    df['weight_intraday_range'] = 0.5 + 0.5 * (df['ema_intraday_range'] / (df['ema_intraday_range'] + df['ema_close_to_open_return']))
    df['weight_close_to_open_return'] = 1 - df['weight_intraday_range']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weight_intraday_range'] * df['intraday_range'] + 
                           df['weight_close_to_open_return'] * df['close_to_open_return']) * df['volume_change']
    
    return df['alpha_factor']
```
