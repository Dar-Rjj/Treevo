```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute dynamic weights based on recent volume and momentum
    df['volume_momentum'] = df['volume'].pct_change()
    df['intraday_range_weight'] = 0.5  # Initial weight
    df['close_to_open_weight'] = 0.5  # Initial weight
    
    # Adjust weights based on recent volume
    df.loc[df['volume'] > df['volume'].shift(1), 'intraday_range_weight'] += 0.1
    df.loc[df['volume'] < df['volume'].shift(1), 'intraday_range_weight'] -= 0.1
    
    # Adjust weights based on momentum
    df.loc[df['close'] > df['close'].shift(1), 'intraday_range_weight'] += 0.1
    df.loc[df['close'] < df['close'].shift(1), 'intraday_range_weight'] -= 0.1
    
    # Ensure weights are between 0 and 1
    df['intraday_range_weight'] = df['intraday_range_weight'].clip(lower=0, upper=1)
    df['close_to_open_weight'] = 1 - df['intraday_range_weight']
    
    # Combine Intraday Range and Close-to-Open Return with dynamic weights
    df['alpha_factor'] = (df['intraday_range'] * df['intraday_range_weight']) + \
                         (df['close_to_open_return'] * df['close_to_open_weight'])
    
    return df['alpha_factor']
```
