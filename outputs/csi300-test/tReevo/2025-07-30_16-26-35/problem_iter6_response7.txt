```python
import pandas as pd

def heuristics_v2(df, n=10, m=5):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open']
    
    # Calculate Average Daily Volume
    avg_daily_volume = df['volume'].rolling(window=n).mean()
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['close'] / df['close'].shift(m)) - 1
    
    # Calculate Market Sentiment
    market_sentiment = (df['high'] + df['low']) / (2 * df['close'])
    
    # Combine Intraday Range, Return, and Volume
    combined_factor = (intraday_range * close_to_open_return) / avg_daily_volume
    
    # Integrate Momentum and Sentiment
    final_factor = combined_factor * intraday_momentum * market_sentiment
    
    return final_factor
```
