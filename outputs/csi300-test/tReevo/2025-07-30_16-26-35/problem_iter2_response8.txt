```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Return
    df['IntradayReturn'] = (df['high'] - df['low']) / df['low']
    
    # Weight by Trade Intensity
    df['TradeIntensity'] = df['volume'] / df['amount']
    df['WeightedIntradayReturn'] = df['IntradayReturn'] * df['TradeIntensity']
    
    # Incorporate Close-to-PreviousClose Trend
    df['PrevClose'] = df['close'].shift(1)
    df['CloseToPrevCloseReturn'] = (df['close'] - df['PrevClose']) / df['PrevClose']
    df['CombinedReturn'] = df['WeightedIntradayReturn'] + df['CloseToPrevCloseReturn']
    
    # Integrate Volatility
    df['DailyReturn'] = (df['close'] - df['PrevClose']) / df['PrevClose']
    window_size_volatility = 20
    df['Volatility'] = df['DailyReturn'].rolling(window=window_size_volatility).std()
    df['AdjustedVolatility'] = df['Volatility'] * df['TradeIntensity']
    
    # Consider Multi-Period Returns
    short_term_window = 5
    long_term_window = 20
    df['ShortTermMA'] = df['DailyReturn'].rolling(window=short_term_window).mean()
    df['LongTermMA'] = df['DailyReturn'].rolling(window=long_term_window).mean()
    df['MultiPeriodAdjustment'] = (df['ShortTermMA'] - df['LongTermMA'])
    
    # Final Factor Value
    df['FinalFactorValue'] = (df['CombinedReturn'] + df['AdjustedVolatility']) * df['MultiPeriodAdjustment']
    
    return df['FinalFactorValue']
```
