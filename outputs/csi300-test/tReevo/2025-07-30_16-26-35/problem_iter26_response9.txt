import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']).shift(-1) / df['open'].shift(-1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Compute Volume Ratio
    volume_ratio = df['volume'] / df['volume'].shift(1)
    
    # Enhanced Adaptive Weighting
    w_recent_intraday_range = 0.5 + 0.2 * (volume_ratio - 1)
    w_older_intraday_range = 1 - w_recent_intraday_range
    recent_intraday_range = intraday_range * w_recent_intraday_range + intraday_range.shift(1) * w_older_intraday_range
    
    w_recent_close_to_open = 0.3 + 0.2 * (volume_ratio - 1)
    w_older_close_to_open = 1 - w_recent_close_to_open
    recent_close_to_open_return = close_to_open_return * w_recent_close_to_open + close_to_open_return.shift(1) * w_older_close_to_open
    
    w_recent_vwap = 0.1 + 0.2 * (volume_ratio - 1)
    w_older_vwap = 1 - w_recent_vwap
    recent_vwap = vwap * w_recent_vwap + vwap.shift(1) * w_older_vwap
    
    w_recent_intraday_momentum = 0.1 + 0.2 * (volume_ratio - 1)
    w_older_intraday_momentum = 1 - w_recent_intraday_momentum
    recent_intraday_momentum = intraday_momentum * w_recent_intraday_momentum + intraday_momentum.shift(1) * w_older_intraday_momentum
    
    # Initial Alpha Factor
    initial_alpha_factor = (recent_intraday_range + recent_close_to_open_return + recent_vwap + recent_intraday_momentum)
    
    # Adjust Alpha Factor based on Volume Ratio
    adjusted_alpha_factor = initial_alpha_factor * (1 + (volume_ratio - 1) * 0.8)
    
    # Incorporate Advanced Volume Momentum
    volume_momentum = (volume_ratio - 1).apply(lambda x: max(0, x))
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.4
    
    return final_alpha_factor
