```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=10, m=5, p=3, q=2):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] / df['open']).shift(1) - 1
    
    # Calculate Average Daily Volume
    avg_daily_volume = df['volume'].rolling(window=n).mean()
    
    # Calculate Multi-Period Intraday Momentum
    momentum_returns = df['close'].pct_change().rolling(window=m).sum()
    
    # Calculate Market Sentiment
    sentiment_high = df['high'].rolling(window=p).max()
    sentiment_low = df['low'].rolling(window=p).min()
    sentiment_close = df['close'].rolling(window=q).mean()
    market_sentiment = (sentiment_close - sentiment_low) / (sentiment_high - sentiment_low)
    
    # Combine Intraday Range, Return, and Volume
    combined_factor = intraday_range * close_to_open_return / avg_daily_volume
    
    # Integrate Momentum and Sentiment
    final_factor = combined_factor + momentum_returns * market_sentiment
    
    return final_factor
```
