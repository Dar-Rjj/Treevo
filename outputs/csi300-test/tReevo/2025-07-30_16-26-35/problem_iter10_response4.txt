```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute 20-day Moving Average of Volume
    df['volume_20d_ma'] = df['volume'].rolling(window=20).mean()
    
    # Assign liquidity factor based on current day's volume compared to 20-day MA
    df['liquidity_factor'] = df.apply(
        lambda row: 1.2 if row['volume'] > row['volume_20d_ma'] else 0.8,
        axis=1
    )
    
    # Determine recent and older weights
    recent_weights = {'intraday_range': 0.6, 'close_to_open_return': 0.3, 'liquidity_factor': 0.1}
    older_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.4, 'liquidity_factor': 0.1}

    # Apply weights to create the final alpha factor
    df['alpha_factor'] = df.apply(
        lambda row: (
            recent_weights['intraday_range'] * row['intraday_range'] +
            recent_weights['close_to_open_return'] * row['close_to_open_return'] +
            recent_weights['liquidity_factor'] * row['liquidity_factor']
        ) if row.name < 20 else (
            older_weights['intraday_range'] * row['intraday_range'] +
            older_weights['close_to_open_return'] * row['close_to_open_return'] +
            older_weights['liquidity_factor'] * row['liquidity_factor']
        ),
        axis=1
    )
    
    return df['alpha_factor']
```
