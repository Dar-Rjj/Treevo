```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Calculate Intraday Momentum
    df['intraday_momentum'] = df['close'] - df['open']
    
    # Compute Recent Intraday Range
    recent_intraday_range = 0.5 * df['intraday_range'] + 0.3 * df['intraday_range'].shift(1)
    
    # Compute Recent Close-to-Open Return
    recent_close_to_open_return = 0.3 * df['close_to_open_return'] + 0.2 * df['close_to_open_return'].shift(1)
    
    # Compute Recent VWAP
    recent_vwap = 0.1 * df['vwap'] + 0.4 * df['vwap'].shift(1)
    
    # Compute Recent Intraday Momentum
    recent_intraday_momentum = 0.1 * df['intraday_momentum'] + 0.4 * df['intraday_momentum'].shift(1)
    
    # Initial Alpha Factor
    initial_alpha_factor = recent_intraday_range + recent_close_to_open_return + recent_vwap + recent_intraday_momentum
    
    # Incorporate Recency and Volume
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = initial_alpha_factor * (1 + (df['volume_ratio'] - 1) * 0.6)
    
    # Compute Volume Momentum
    volume_momentum = (df['volume_ratio'] - 1).clip(lower=0)
    
    # Final Alpha Factor with Volume Momentum
    final_alpha_factor = adjusted_alpha_factor + volume_momentum * 0.2
    
    # Compute Volume Growth
    volume_growth = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    
    # Final Alpha Factor with Volume Growth
    final_alpha_factor += volume_growth * 0.1
    
    return final_alpha_factor
```
