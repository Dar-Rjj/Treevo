import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Dynamic Weighting Based on Recency
    def dynamic_weighting(index, column):
        if index in df.index[-5:]:
            return 0.7 * df.loc[index, 'intraday_range'] + 0.3 * df.loc[index, column]
        else:
            return 0.5 * df.loc[index, 'intraday_range'] + 0.5 * df.loc[index, column]
    
    # Apply the dynamic weighting
    factor_values = df.apply(lambda row: dynamic_weighting(row.name, 'close_to_open_return'), axis=1)
    
    return pd.Series(factor_values, index=df.index)
