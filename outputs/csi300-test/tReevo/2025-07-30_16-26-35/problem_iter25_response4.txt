```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Moving Average of Volume
    df['volume_ma'] = df['volume'].rolling(window=5).mean()
    
    # Dynamically Adjust Weights Based on Volume
    df['intraday_weight'] = np.where(df['volume'] > df['volume_ma'], 0.7, 0.5)
    df['close_to_open_weight'] = 1 - df['intraday_weight']
    
    # Combine Intraday Range and Close-to-Open Return
    df['factor_value'] = df['intraday_weight'] * df['intraday_range'] + \
                         df['close_to_open_weight'] * df['close_to_open_return']
    
    # Integrate Multi-Period Trend
    df['intraday_range_ma'] = df['intraday_range'].rolling(window=5).mean()
    df['close_to_open_return_ma'] = df['close_to_open_return'].rolling(window=5).mean()
    
    df['factor_value_trend'] = df['intraday_weight'] * df['intraday_range_ma'] + \
                               df['close_to_open_weight'] * df['close_to_open_return_ma']
    
    # Simplify with Key Price Movements
    df['support'] = df['low'].rolling(window=5).min()
    df['resistance'] = df['high'].rolling(window=5).max()
    
    # Identify Key Level Breaches
    df['high_breaks_resistance'] = df['high'] > df['resistance'].shift(1)
    df['low_breaks_support'] = df['low'] < df['support'].shift(1)
    
    # Incorporate Key Level Breaches into the factor
    df['factor_value_breaches'] = df['factor_value_trend'] + \
                                  df['high_breaks_resistance'].astype(int) - \
                                  df['low_breaks_support'].astype(int)
    
    return df['factor_value_breaches'].dropna()
```
