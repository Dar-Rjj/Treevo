```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'].shift(-1) - df['open']) / df['open']
    
    # Integrate Volume Shocks
    df['Volume_Change'] = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)
    df['Volume_Shock'] = 1.0
    df.loc[df['Volume_Change'] > 0.2, 'Volume_Shock'] = 1.5
    df.loc[df['Volume_Change'] < -0.2, 'Volume_Shock'] = 0.5
    
    # Combine Intraday Range and Close-to-Open Return
    recent_weights = (0.7 * df['Intraday_Range'] + 0.3 * df['Close_to_Open_Return'])
    older_weights = (0.5 * df['Intraday_Range'] + 0.5 * df['Close_to_Open_Return'])
    df['Factor'] = (recent_weights + older_weights) / 2 * df['Volume_Shock']
    
    # Incorporate Moving Averages
    df['Short_MA'] = df['close'].rolling(window=10).mean()
    df['Long_MA'] = df['close'].rolling(window=50).mean()
    
    # Compute Moving Average Crossover
    df['MA_Crossover'] = 1.0
    df.loc[df['Short_MA'] > df['Long_MA'], 'MA_Crossover'] = 1.2
    df.loc[df['Short_MA'] < df['Long_MA'], 'MA_Crossover'] = 0.8
    
    # Adjust Factor with Moving Average Crossover
    df['Factor'] = df['Factor'] * df['MA_Crossover']
    
    return df['Factor'].shift(1)  # Shift to predict future returns
```
