```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open'].shift(1)) / df['open'].shift(1)
    
    # Assign adaptive weights to recent data
    recent_mask = (df.index >= df.index[-5])  # Considering the last 5 days as recent
    older_mask = ~recent_mask
    
    # Recent Weights: 0.7 for Intraday Range, 0.3 for Close-to-Open Return
    recent_factor = 0.7 * intraday_range[recent_mask] + 0.3 * close_to_open_return[recent_mask]
    
    # Older Weights: 0.5 for Intraday Range, 0.5 for Close-to-Open Return
    older_factor = 0.5 * intraday_range[older_mask] + 0.5 * close_to_open_return[older_mask]
    
    # Combine factors
    factor_values = pd.concat([recent_factor, older_factor]).sort_index()
    
    return factor_values
```
