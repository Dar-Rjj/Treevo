```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Compute Daily Returns
    df['Daily_Return'] = np.log(df['close']).diff()

    # Calculate Price Momentum Indicator
    df['10D_Price_Momentum'] = df['Daily_Return'].rolling(window=10).sum()
    df['20D_Price_Momentum'] = df['Daily_Return'].rolling(window=20).sum()

    # Integrate Volume Trend
    df['Volume_Growth'] = np.log(df['volume']).diff()

    df['10D_Volume_Growth'] = df['Volume_Growth'].rolling(window=10).sum()
    df['20D_Volume_Growth'] = df['Volume_Growth'].rolling(window=20).sum()

    # Combine Momentum and Volume Trends
    df['Combined_Factor_10D'] = df['10D_Price_Momentum'] * df['10D_Volume_Growth']
    df['Combined_Factor_20D'] = df['20D_Price_Momentum'] * df['20D_Volume_Growth']

    # Smooth Combined Factor
    df['5D_SMA_Combined_Factor_10D'] = df['Combined_Factor_10D'].rolling(window=5).mean()
    df['10D_SMA_Combined_Factor_10D'] = df['Combined_Factor_10D'].rolling(window=10).mean()

    df['5D_SMA_Combined_Factor_20D'] = df['Combined_Factor_20D'].rolling(window=5).mean()
    df['10D_SMA_Combined_Factor_20D'] = df['Combined_Factor_20D'].rolling(window=10).mean()

    # Final Alpha
    alpha_5D = df['5D_SMA_Combined_Factor_10D']
    alpha_10D = df['10D_SMA_Combined_Factor_20D']

    return alpha_5D, alpha_10D
```
