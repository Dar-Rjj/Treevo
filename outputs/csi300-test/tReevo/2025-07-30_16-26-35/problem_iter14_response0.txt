```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    vwap = ((df['high'] + df['low'] + df['close']) / 3) * df['volume']
    
    # Define weights for recent and older data
    weights_intraday_range = 0.6 * intraday_range + 0.4 * intraday_range.shift(1)
    weights_close_to_open_return = 0.3 * close_to_open_return + 0.2 * close_to_open_return.shift(1)
    weights_vwap = 0.1 * vwap + 0.4 * vwap.shift(1)
    
    # Combine the weighted components
    alpha_factor = (weights_intraday_range + 
                    weights_close_to_open_return + 
                    weights_vwap)
    
    # Incorporate recency and volume
    volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_alpha_factor = alpha_factor * (1 + (volume_ratio - 1) * 0.5)
    
    return adjusted_alpha_factor
```
