import pandas as pd

def heuristics_v2(data):
    # Calculate Intraday Range
    data['Intraday_Range'] = data['high'] - data['low']
    
    # Calculate Close-to-Open Return using previous day's close
    data['Close_to_Open_Return'] = (data['close'] - data['open'].shift(1)) / data['open'].shift(1)
    
    # Assign weights and combine to form the alpha factor
    data['Alpha_Factor'] = 0.7 * data['Intraday_Range'] + 0.3 * data['Close_to_Open_Return']
    
    # Return the alpha factor as a Series
    return data['Alpha_Factor']
