```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open']
    
    # Compute Daily Volatility
    df['daily_return'] = df['close'].pct_change()
    volatility_window = 20
    df['daily_volatility'] = df['daily_return'].rolling(window=volatility_window).std()
    
    # Assign dynamic weights based on recent data
    weights_intraday = np.where(df.index.isin(df.index[-5:]), 0.8, 0.5)
    weights_close_to_open = np.where(df.index.isin(df.index[-5:]), 0.2, 0.5)
    
    # Combine Intraday Range and Close-to-Open Return with weights
    df['weighted_intraday_range'] = df['intraday_range'] * weights_intraday
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * weights_close_to_open
    df['combined_factor'] = df['weighted_intraday_range'] + df['weighted_close_to_open_return']
    
    # Adjust by Daily Volatility
    df['final_factor_value'] = df['combined_factor'] / df['daily_volatility']
    
    return df['final_factor_value']
```
