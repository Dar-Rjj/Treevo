```python
import pandas as pd

def heuristics_v2(df):
    # Ensure the DataFrame is sorted by date
    df = df.sort_index()
    
    # Calculate Intraday Range
    intraday_range = (df['high'] - df['low']) / df['low']
    
    # Calculate Close-to-Open Return
    close_to_open_return = (df['close'] - df.shift(1)['close']) / df.shift(1)['close']
    
    # Compute Intraday Volatility Adjusted Return
    recent_weights = 0.7 * intraday_range + 0.3 * close_to_open_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_open_return
    
    # Combine using dynamic weighting based on recency
    # Assume 'recent' is the most recent 5% of the data, and 'older' is the rest
    num_recent = int(len(df) * 0.05)
    factor = pd.Series(index=df.index, dtype=float)
    
    for i in range(len(df)):
        if i >= len(df) - num_recent:
            factor.iloc[i] = recent_weights.iloc[i]
        else:
            factor.iloc[i] = older_weights.iloc[i]
    
    return factor
```
