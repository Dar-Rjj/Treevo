```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    intraday_range = df['high'] - df['low']
    
    # Calculate Close-to-Previous Day's Close Return
    close_to_close_return = (df['close'].diff() / df['close'].shift()).fillna(0)
    
    # Determine the weights for recent and older data
    recent_weights = 0.7 * intraday_range + 0.3 * close_to_close_return
    older_weights = 0.5 * intraday_range + 0.5 * close_to_close_return
    
    # Combine the weighted values
    combined_factor = recent_weights * 0.6 + older_weights * 0.4
    
    return combined_factor
```
