import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).cumsum() / df['volume'].cumsum()
    
    # Determine recent and older weights
    recent_weights = {'intraday_range': 0.5, 'close_to_open_return': 0.3, 'vwap': 0.2}
    older_weights = {'intraday_range': 0.4, 'close_to_open_return': 0.4, 'vwap': 0.2}
    
    # Combine Intraday Range and Close-to-Open Return with VWAP
    df['factor_value'] = (recent_weights['intraday_range'] * df['intraday_range'] + 
                          recent_weights['close_to_open_return'] * df['close_to_open_return'] + 
                          recent_weights['vwap'] * df['vwap'])
    
    # Apply older weights if the data is considered "older"
    # For simplicity, let's assume the last 5 days are considered "recent", adjust this as needed
    df.loc[df.index < df.index[-5], 'factor_value'] = (older_weights['intraday_range'] * df['intraday_range'] + 
                                                      older_weights['close_to_open_return'] * df['close_to_open_return'] + 
                                                      older_weights['vwap'] * df['vwap'])
    
    return df['factor_value']
