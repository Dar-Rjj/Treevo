```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    df['High_Low_Spread'] = df['high'] - df['low']
    df['Close_Open_Spread'] = df['close'] - df['open']
    df['Intraday_Momentum'] = df['High_Low_Spread'] * df['Close_Open_Spread']

    # Calculate Volume Synchronization
    df['Price_Change_Abs'] = df['close'].diff().abs()
    df['Sum_Price_Change_Abs'] = df['Price_Change_Abs'].cumsum()
    df['Synchronization_Metric'] = df['Sum_Price_Change_Abs'] / df['volume']

    # Weight Momentum by Volume Synchronization
    df['VSM'] = df['Intraday_Momentum'] * df['Synchronization_Metric']

    # Adjust for Market Sentiment
    df['Market_Return'] = df['close'].diff()
    df['VSM_Adjusted'] = df.apply(lambda row: row['VSM'] * 1.1 if row['Market_Return'] > 0 else row['VSM'] * 0.9, axis=1)

    return df['VSM_Adjusted']
```
