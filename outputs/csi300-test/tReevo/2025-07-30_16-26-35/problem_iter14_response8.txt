```python
import pandas as pd
import talib

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = (df['close'] - df['open']) / df['open'].shift(1)
    
    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df['high'] + df['low'] + df['close']) / 3 * df['volume']
    
    # Calculate Relative Strength Index (RSI)
    df['rsi'] = talib.RSI(df['close'], timeperiod=14)
    
    # Assign weight based on RSI
    df['rsi_weight'] = 1
    df.loc[df['rsi'] > 70, 'rsi_weight'] = 0.5
    df.loc[df['rsi'] < 30, 'rsi_weight'] = 1.5
    
    # Assuming sentiment data is available in the dataframe
    # Calculate net sentiment score
    df['net_sentiment'] = df['positive_news'] - df['negative_news']
    
    # Assign weight based on sentiment
    df['sentiment_weight'] = 1
    df.loc[df['net_sentiment'] > 0, 'sentiment_weight'] = 1.5
    df.loc[df['net_sentiment'] < 0, 'sentiment_weight'] = 0.5
    
    # Combine weights
    df['total_weight'] = df['rsi_weight'] * df['sentiment_weight']
    
    # Compute Weighted Intraday Range
    df['weighted_intraday_range'] = df['intraday_range'] * df['total_weight']
    
    # Compute Weighted Close-to-Open Return
    df['weighted_close_to_open_return'] = df['close_to_open_return'] * df['total_weight']
    
    # Compute Weighted VWAP
    df['weighted_vwap'] = df['vwap'] * df['total_weight']
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['weighted_intraday_range'] + 
                           df['weighted_close_to_open_return'] + 
                           df['weighted_vwap'])
    
    return df['alpha_factor']
```
