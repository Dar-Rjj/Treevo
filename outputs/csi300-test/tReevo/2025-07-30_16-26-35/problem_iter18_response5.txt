```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['Intraday_Range'] = df['high'] - df['low']

    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['close'] - df['open']) / df['open']

    # Calculate Average Volume
    df['Average_Volume_20D'] = df['volume'].rolling(window=20).mean()

    # Compute Volume Relative Strength
    df['Volume_Relative_Strength'] = df['volume'] / df['Average_Volume_20D']

    # Calculate 5-Day and 20-Day Simple Moving Averages
    df['SMA_5D'] = df['close'].rolling(window=5).mean()
    df['SMA_20D'] = df['close'].rolling(window=20).mean()

    # Calculate SMA Difference
    df['SMA_Difference'] = df['SMA_5D'] - df['SMA_20D']

    # Determine recent vs older data
    recent_weights = [0.4, 0.3, 0.2, 0.1]
    older_weights = [0.6, 0.2, 0.1, 0.1]

    # Combine factors
    df['Recent_Factor'] = (
        df['Intraday_Range'] * recent_weights[0] +
        df['Close_to_Open_Return'] * recent_weights[1] +
        df['Volume_Relative_Strength'] * recent_weights[2] +
        df['SMA_Difference'] * recent_weights[3]
    )

    df['Older_Factor'] = (
        df['Intraday_Range'] * older_weights[0] +
        df['Close_to_Open_Return'] * older_weights[1] +
        df['Volume_Relative_Strength'] * older_weights[2] +
        df['SMA_Difference'] * older_weights[3]
    )

    # Determine the final factor value
    df['Factor_Value'] = df['Recent_Factor'].fillna(0) + df['Older_Factor'].fillna(0)

    return df['Factor_Value']
```
