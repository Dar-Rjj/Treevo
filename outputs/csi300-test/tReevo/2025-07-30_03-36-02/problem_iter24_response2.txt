```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility (High - Low)
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume ((High - Low) * Volume)
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum (Close - Open)
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Weighted Volatility and Momentum ((High - Low) * Volume + (Close - Open))
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor).mean()
    
    # Ensure Positive Values (Smoothed Value + Epsilon)
    epsilon = 1e-10
    positive_smoothed_value = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation (log(Smoothed Value + Epsilon))
    log_transformed_value = np.log(positive_smoothed_value)
    
    return log_transformed_value
```
