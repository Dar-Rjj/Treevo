```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_window = 5  # Define the rolling window size
    intraday_momentum = daily_momentum.rolling(window=rolling_window).sum()
    
    # Integrate Weighted Volatility and Momentum
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Ensure Positive Values
    small_constant = 1e-6
    smoothed_positive_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(smoothed_positive_value)
    
    # Factor Output
    factor_values = log_transformed_value
    
    return factor_values
```
