```python
import pandas as pd

def heuristics_v2(df):
    # Compute Daily High-Low Difference
    df['high_low_diff'] = df['high'] - df['low']
    
    # Normalize by Previous Close
    df['intraday_momentum'] = df['high_low_diff'] / df['close'].shift(1)
    
    # Calculate Volume Ratio
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    
    # Apply Volume Correction
    df['factor'] = df['intraday_momentum'] * df['volume_ratio']
    
    return df['factor']
```
