import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    momentum_5 = close_open_diff.rolling(window=5).sum()
    momentum_10 = close_open_diff.rolling(window=10).sum()
    momentum_20 = close_open_diff.rolling(window=20).sum()
    
    # Integrate Momentum and Volatility
    integrated_5 = weighted_volatility + momentum_5
    integrated_10 = weighted_volatility + momentum_10
    integrated_20 = weighted_volatility + momentum_20
    
    # Apply Dynamic Exponential Smoothing
    def exponential_smoothing(series, alpha):
        smoothed = series.copy()
        for i in range(1, len(smoothed)):
            smoothed.iloc[i] = alpha * series.iloc[i] + (1 - alpha) * smoothed.iloc[i-1]
        return smoothed
    
    smoothed_5 = exponential_smoothing(integrated_5, 0.9)
    smoothed_10 = exponential_smoothing(integrated_10, 0.8)
    smoothed_20 = exponential_smoothing(integrated_20, 0.7)
    
    # Ensure Values are Positive
    smoothed_5 += 1e-6
    smoothed_10 += 1e-6
    smoothed_20 += 1e-6
    
    # Apply Logarithmic Transformation
    log_smoothed_5 = np.log(smoothed_5)
    log_smoothed_10 = np.log(smoothed_10)
    log_smoothed_20 = np.log(smoothed_20)
    
    # Factor Output
    factor_output = (log_smoothed_5 + log_smoothed_10 + log_smoothed_20) / 3
    
    return factor_output
