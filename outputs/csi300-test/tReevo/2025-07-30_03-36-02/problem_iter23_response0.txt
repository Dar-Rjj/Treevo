```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()  # Assuming a 5-day rolling window
    
    # Integrate Momentum and Volatility
    integrated_signal = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_signal = integrated_signal.ewm(alpha=1 - smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive (add a small constant to avoid log of zero or negative)
    epsilon = 1e-8
    positive_smoothed_signal = smoothed_signal + epsilon
    
    # Apply Logarithmic Transformation
    log_transformed_signal = np.log(positive_smoothed_signal)
    
    return log_transformed_signal
```
