```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']

    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    intraday_momentum = daily_momentum.rolling(window=5).sum()

    # Integrate Momentum and Weighted Volatility
    integrated_factor = weighted_volatility + intraday_momentum

    # Apply Exponential Smoothing
    alpha = 0.2  # Alpha parameter for Exponential Smoothing
    smoothed_factor = integrated_factor.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_factor = smoothed_factor + abs(smoothed_factor.min()) + 1
    
    # Apply Logarithmic Transformation
    factor_values = np.log(smoothed_factor)

    return factor_values
```
