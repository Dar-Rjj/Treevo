import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.5
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor).mean()
    
    # Ensure Values are Positive
    small_constant = 1e-6
    positive_smoothed_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    # Incorporate Previous Day's Integrated Value
    lagged_value = log_transformed_value.shift(1)
    combined_value = 0.7 * log_transformed_value + 0.3 * lagged_value
    
    # Factor Output
    factor_output = combined_value
    
    return factor_output
