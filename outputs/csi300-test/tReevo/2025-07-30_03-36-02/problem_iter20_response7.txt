```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_window = 5
    intraday_momentum = daily_momentum.rolling(window=rolling_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + intraday_momentum
    smoothing_factor_1 = 0.9
    smoothed_factor = integrated_factor.ewm(alpha=smoothing_factor_1, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_factor = smoothed_factor + 1e-6
    
    # Apply Logarithmic Transformation
    log_smoothed_factor = np.log(smoothed_factor)
    
    # Incorporate Mean Reversion
    long_term_ma_window = 20
    long_term_ma = df['close'].rolling(window=long_term_ma_window).mean()
    mean_reversion_deviation = (df['close'] - long_term_ma).abs()
    
    # Combine Mean Reversion Deviation with Integrated Factor
    combined_factor = log_smoothed_factor + mean_reversion_deviation
    smoothing_factor_2 = 0.8
    final_factor = combined_factor.ewm(alpha=smoothing_factor_2, adjust=False).mean()
    
    return final_factor
```
