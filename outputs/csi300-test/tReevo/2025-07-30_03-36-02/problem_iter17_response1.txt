```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Weight by Volume
    df['weighted_intraday_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['close_open_diff'] = df['close'] - df['open']
    df['intraday_momentum'] = df['close_open_diff'].rolling(window=5).sum()
    
    # Integrate Weighted Volatility and Momentum
    df['combined_metric'] = df['weighted_intraday_volatility'] + df['intraday_momentum']
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    df['smoothed_value'] = df['combined_metric'].ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Positive Values
    df['smoothed_value'] = df['smoothed_value'] + 1e-6
    
    # Apply Logarithmic Transformation
    df['log_transformed_value'] = np.log(df['smoothed_value'])
    
    # Final Alpha Factor
    alpha_factor = df['log_transformed_value']
    
    return alpha_factor
```
