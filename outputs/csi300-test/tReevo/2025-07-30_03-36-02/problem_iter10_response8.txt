```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Dynamic Volume Weighting
    dynamic_weight = df['volume'] / (df['volume'] + df['volume'].shift(1))
    
    # Apply Dynamic Weight to Intraday Volatility
    weighted_volatility = intraday_volatility * dynamic_weight
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Weighted Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    smoothing_factor = 0.1
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor).mean()
    
    # Incorporate Recent Price Trends
    recent_price_trend = df['close'] - df['close'].shift(1)
    
    # Combine Smoothed Integration and Recent Price Trend
    factor = smoothed_value + recent_price_trend
    
    return factor
```
