```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    momentum_window = 5  # Example: Use a 5-day window for summing momentum
    intraday_momentum = daily_momentum.rolling(window=momentum_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    alpha = 0.2  # Alpha parameter for exponential smoothing
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    small_constant = 1e-6
    positive_smoothed_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    return log_transformed_value
```
