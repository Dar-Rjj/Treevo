```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_term_ema = df['log_returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_term_ema = df['log_returns'].ewm(span=20, adjust=False).mean()
    
    # Compute the Difference Between Short-Term and Long-Term EMA
    momentum_diff = short_term_ema - long_term_ema
    
    # Volume-Weight the Momentum Difference
    volume_weighted_momentum = momentum_diff * df['volume']
    
    return volume_weighted_momentum
```
