```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Apply Exponential Moving Average (EMA) to Momentum
    ema_momentum = intraday_momentum.ewm(span=10, adjust=False).mean()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + ema_momentum
    
    # Apply Exponential Smoothing
    smoothed_factor = integrated_factor.ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
