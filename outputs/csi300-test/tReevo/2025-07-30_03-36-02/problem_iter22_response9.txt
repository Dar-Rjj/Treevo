```python
import pandas as pd
import numpy as np

def heuristics_v2(df, rolling_window=10, smoothing_factor=0.9, epsilon=1e-6):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=rolling_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive
    positive_smoothed_value = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation
    factor_values = np.log(positive_smoothed_value)
    
    return factor_values
```
