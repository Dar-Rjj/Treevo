```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    rolling_momentum = close_open_diff.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + rolling_momentum
    
    # Apply Fixed Exponential Smoothing
    alpha = 0.9
    smoothed_factor = integrated_factor.ewm(alpha=1-alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_factor = smoothed_factor + 1e-6  # Small constant to avoid log of zero or negative
    
    # Apply Logarithmic Transformation
    log_transformed_factor = np.log(smoothed_factor)
    
    return log_transformed_factor
```
