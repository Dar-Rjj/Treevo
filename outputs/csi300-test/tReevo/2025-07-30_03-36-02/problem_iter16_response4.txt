import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor).mean()
    
    # Ensure Values are Positive
    small_constant = 1e-6
    positive_smoothed_value = smoothed_value + small_constant
    
    # Adaptive Weighting Based on Recent Performance
    short_ma_momentum = rolling_momentum.rolling(window=10).mean()
    long_ma_volatility = weighted_volatility.rolling(window=20).mean()
    
    # Assign Weights
    recent_performance = (short_ma_momentum > 0).astype(int)
    adaptive_weights = (recent_performance * 0.7) + ((1 - recent_performance) * 0.3)
    
    # Combine Weights
    weighted_smoothed_value = positive_smoothed_value * adaptive_weights
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(weighted_smoothed_value)
    
    return log_transformed_value
