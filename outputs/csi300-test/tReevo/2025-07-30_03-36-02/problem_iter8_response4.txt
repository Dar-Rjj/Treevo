```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Combine Weighted Volatility and Momentum
    combined_signal = weighted_volatility + intraday_momentum
    
    # Apply Logarithmic Transformation
    log_transformed = np.log1p(combined_signal)
    
    # Integrate Additional Market Signals
    # Calculate Average Volume (20-day moving average)
    avg_volume = df['volume'].rolling(window=20).mean()
    
    # Calculate Price Change (Percentage change from previous close)
    price_change = df['close'].pct_change()
    
    # Combine All Signals
    integrated_signal = log_transformed + avg_volume + price_change
    
    # Apply Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_signal = integrated_signal.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    return smoothed_signal
```
