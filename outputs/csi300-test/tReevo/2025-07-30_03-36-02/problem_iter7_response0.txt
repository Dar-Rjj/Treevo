```python
import pandas as pd
import numpy as np
from statsmodels.tsa.holtwinters import SimpleExpSmoothing

def heuristics_v2(df, alpha=0.6):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    smoothed_factor = SimpleExpSmoothing(integrated_factor).fit(smoothing_level=alpha).fittedvalues
    
    # Apply Logarithmic Transformation
    log_transformed_factor = np.log1p(smoothed_factor)
    
    return log_transformed_factor
```
