```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_factor = integrated_factor.ewm(alpha=1 - smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-6
    positive_smoothed_factor = smoothed_factor + epsilon
    
    # Apply Logarithmic Transformation
    final_factor = np.log(positive_smoothed_factor)
    
    return final_factor
```
