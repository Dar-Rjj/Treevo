```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Daily Return
    df['daily_return'] = df['close'].pct_change()

    # Calculate 5-Day Exponential Moving Average (EMA) of Daily Return
    alpha = 2 / (5 + 1)
    df['ema_5'] = df['daily_return'].ewm(alpha=alpha, adjust=False).mean()

    # Calculate 10-Day Simple Moving Average (SMA) of Daily Return
    df['sma_10'] = df['daily_return'].rolling(window=10).mean()

    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']

    # Calculate Relative Strength
    df['relative_strength'] = df['ema_5'] / df['sma_10']

    # Assume we have a column 'market_return' in the DataFrame representing the S&P 500 or another market index's daily return
    # If not, you need to add this column to the DataFrame with the appropriate market index returns
    df['market_ema_5'] = df['market_return'].ewm(alpha=alpha, adjust=False).mean()
    
    # Adjust for Market Volatility
    df['adjusted_ema_5'] = df['ema_5'] - df['market_ema_5']

    # Integrate Factors
    df['alpha_factor'] = (df['relative_strength'] * df['adjusted_ema_5']) + df['intraday_volatility']

    # Final Alpha Factor
    return df['alpha_factor']
```
