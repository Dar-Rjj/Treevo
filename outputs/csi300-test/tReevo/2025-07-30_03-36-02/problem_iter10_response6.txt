```python
import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df, alpha=0.1, smoothing_window=5):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=smoothing_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + rolling_momentum
    integrated_factor = integrated_factor.ewm(alpha=alpha, adjust=False).mean()
    
    # Apply Logarithmic Transformation
    integrated_factor = np.log1p(integrated_factor.clip(lower=0))
    
    return integrated_factor
```
