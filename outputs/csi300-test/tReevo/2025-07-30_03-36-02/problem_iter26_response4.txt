```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_intraday_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Weighted Volatility and Momentum
    integrated_value = weighted_intraday_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    alpha = 0.9
    smoothed_values = integrated_value.ewm(alpha=alpha).mean()
    
    # Add a Small Constant
    epsilon = 1e-10
    adjusted_smoothed_values = smoothed_values + epsilon
    
    # Apply Logarithmic Transformation
    alpha_factor = np.log(adjusted_smoothed_values)
    
    return alpha_factor
```
