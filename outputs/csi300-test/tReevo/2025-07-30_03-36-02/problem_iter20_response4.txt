```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()
    
    # Integrate Weighted Volatility and Momentum
    integrated_value = weighted_volatility + rolling_momentum
    
    # Ensure values are positive
    integrated_value += 1e-6  # Small constant to avoid log of zero or negative
    
    # Define Adaptive Smoothing Factor (e.g., based on historical data)
    smoothing_factor = 0.2  # Example value, can be adjusted or optimized
    
    # Apply Exponential Smoothing Formula
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor).mean()
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(smoothed_value)
    
    # Generate Final Alpha Factor
    alpha_factor = log_transformed_value
    
    return alpha_factor
```
