import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    
    # Sum of Close-Open Differences over a Rolling Window (e.g., 5 days)
    momentum_sum = daily_momentum.rolling(window=5).sum()
    
    # Compute Additional Momentum Features
    short_window = 3
    long_window = 10
    momentum_avg_short = daily_momentum.rolling(window=short_window).mean()
    momentum_avg_long = daily_momentum.rolling(window=long_window).mean()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + momentum_sum
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-6
    smoothed_value_pos = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation
    log_smoothed_value = np.log(smoothed_value_pos)
    
    # Incorporate Rolling Metrics
    rolling_window = 20
    rolling_mean = log_smoothed_value.rolling(window=rolling_window).mean()
    rolling_std = log_smoothed_value.rolling(window=rolling_window).std()
    
    # Compute Z-Score of Logarithmically Transformed Value
    z_score = (log_smoothed_value - rolling_mean) / rolling_std
    
    # Factor Output
    return z_score
