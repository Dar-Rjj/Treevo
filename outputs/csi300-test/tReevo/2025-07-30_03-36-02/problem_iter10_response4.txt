import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_window = 5
    rolling_momentum = daily_momentum.rolling(window=rolling_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_signal = weighted_volatility + rolling_momentum
    
    # Apply Adaptive Exponential Smoothing
    alpha = 0.1  # Initial smoothing parameter
    recent_volatility = intraday_volatility.rolling(window=rolling_window).mean()
    adaptive_alpha = 1 / (1 + recent_volatility)
    smoothed_signal = integrated_signal.ewm(alpha=alpha, adjust=False).mean() * adaptive_alpha
    
    # Ensure Positive Values
    smoothed_positive_signal = smoothed_signal + 1e-6  # Add a small constant to avoid negative values
    
    # Apply Logarithmic Transformation
    factor_values = np.log(smoothed_positive_signal)
    
    return factor_values
