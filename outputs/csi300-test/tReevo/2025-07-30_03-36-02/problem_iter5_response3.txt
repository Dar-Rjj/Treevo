```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Volume-Weighted Intraday Volatility
    volume_weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Weighted Volatility and Momentum
    integrated_factor = volume_weighted_volatility + intraday_momentum
    integrated_factor_log = np.log1p(integrated_factor)
    
    return integrated_factor_log
```
