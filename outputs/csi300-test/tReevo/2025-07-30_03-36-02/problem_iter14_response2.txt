```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Weighted Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-6
    smoothed_value_positive = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation
    factor_values = np.log(smoothed_value_positive)
    
    return factor_values
```
