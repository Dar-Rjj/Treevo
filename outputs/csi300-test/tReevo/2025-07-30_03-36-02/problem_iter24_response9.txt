```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()  # Example: 5-day window
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9  # Smoothing factor
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    small_constant = 1e-6
    positive_smoothed_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    # Factor Output
    return log_transformed_value
```
