```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']

    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=10).sum()

    # Integrate Momentum and Weighted Volatility
    integrated_value = weighted_volatility + rolling_momentum

    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_values = integrated_value.ewm(alpha=smoothing_factor, adjust=False).mean()

    # Ensure Values are Positive
    small_constant = 1e-6
    positive_values = smoothed_values + small_constant

    # Apply Logarithmic Transformation
    log_transformed_values = np.log(positive_values)

    return log_transformed_values
```
