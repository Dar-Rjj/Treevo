```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    momentum_rolling_sum = daily_momentum.rolling(window=10).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + momentum_rolling_sum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-6
    positive_smoothed_value = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    # Mean Reversion Adjustment
    long_term_ma = df['close'].rolling(window=50).mean()
    adjusted_factor = log_transformed_value - long_term_ma
    
    return adjusted_factor
```
