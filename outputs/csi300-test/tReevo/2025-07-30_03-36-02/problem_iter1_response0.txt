```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Momentum
    momentum = df['close'].diff(1)
    
    # Calculate Volume Adjusted Momentum
    volume_adjusted_momentum = momentum * df['volume']
    
    # Combine Intraday Volatility and Volume Adjusted Momentum
    factor = intraday_volatility + volume_adjusted_momentum
    
    return factor
```
