```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Logarithmic Returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate Short-Term Exponential Moving Average (EMA)
    short_term_ema = df['log_returns'].ewm(span=5, adjust=False).mean()
    
    # Calculate Medium-Term Exponential Moving Average (EMA)
    medium_term_ema = df['log_returns'].ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term Exponential Moving Average (EMA)
    long_term_ema = df['log_returns'].ewm(span=20, adjust=False).mean()
    
    # Compute the Momentum Difference Between EMAs
    short_vs_medium_diff = short_term_ema - medium_term_ema
    medium_vs_long_diff = medium_term_ema - long_term_ema
    
    # Combine Momentum Differences
    combined_momentum = 0.5 * short_vs_medium_diff + 0.5 * medium_vs_long_diff
    
    # Volume-Weight the Combined Momentum
    volume_weighted_momentum = combined_momentum * df['volume']
    
    return volume_weighted_momentum
```
