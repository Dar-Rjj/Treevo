```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Multi-Level Intraday Momentum
    daily_momentum = df['close'] - df['open']
    
    # Calculate Short-Term EMA of Momentum
    short_term_ema = daily_momentum.ewm(span=5, adjust=False).mean()
    
    # Calculate Medium-Term EMA of Momentum
    medium_term_ema = daily_momentum.ewm(span=10, adjust=False).mean()
    
    # Calculate Long-Term EMA of Momentum
    long_term_ema = daily_momentum.ewm(span=20, adjust=False).mean()
    
    # Integrate Multi-Level Momentum with Weighted Volatility
    integrated_short_term = weighted_volatility + short_term_ema
    integrated_medium_term = weighted_volatility + medium_term_ema
    integrated_long_term = weighted_volatility + long_term_ema
    
    # Combine all terms for the final factor
    final_factor = (integrated_short_term + integrated_medium_term + integrated_long_term) / 3
    
    return final_factor
```
