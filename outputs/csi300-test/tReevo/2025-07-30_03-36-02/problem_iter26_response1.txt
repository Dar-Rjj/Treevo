```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=10, alpha=0.9, small_constant=1e-8):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=window).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Fixed Exponential Smoothing
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    log_smoothed_value = np.log(smoothed_value)
    
    return log_smoothed_value
```
