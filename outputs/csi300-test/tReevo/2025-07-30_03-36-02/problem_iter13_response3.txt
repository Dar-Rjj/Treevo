```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Weight by Volume
    df['weighted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['close_open_diff'] = df['close'] - df['open']
    df['rolling_momentum'] = df['close_open_diff'].rolling(window=10).sum()
    
    # Integrate Momentum and Volatility
    df['integrated_value'] = df['weighted_volatility'] + df['rolling_momentum']
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    df['smoothed_value'] = df['integrated_value'].ewm(alpha=smoothing_factor).mean()
    
    # Ensure Values are Positive
    df['positive_value'] = df['smoothed_value'] + 1e-6
    
    # Apply Logarithmic Transformation
    df['log_transformed_value'] = np.log(df['positive_value'])
    
    # Factor Output
    factor_values = df['log_transformed_value']
    
    return factor_values
```
