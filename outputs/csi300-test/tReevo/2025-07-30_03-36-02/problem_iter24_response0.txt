```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    IV = df['high'] - df['low']
    
    # Weight by Volume
    W_IV = IV * df['volume']
    
    # Calculate Intraday Momentum
    Rolling_M = (df['close'] - df['open']).rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    epsilon = 1e-10
    Positive_W_IV = W_IV + epsilon
    Sum = Positive_W_IV + Rolling_M
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    ES = Sum.ewm(alpha=alpha, adjust=False).mean()
    
    # Apply Logarithmic Transformation
    Log_ES = np.log(ES + epsilon)
    
    # Factor Output
    Alpha_Factor = Log_ES
    return Alpha_Factor
```
