```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=10, smoothing_factor=0.9, epsilon=1e-10):
    """
    Calculate the integrated and transformed intraday momentum and volatility factor.
    
    Parameters:
        df (pd.DataFrame): A DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                          indexed by date.
        window (int): The rolling window size for momentum calculation.
        smoothing_factor (float): The smoothing factor for exponential smoothing.
        epsilon (float): A small constant to avoid log of zero or negative value.
        
    Returns:
        pd.Series: A Series representing the alpha factor values indexed by date.
    """
    # Calculate Intraday Volatility
    IV = df['high'] - df['low']
    
    # Weight by Volume
    W_IV = IV * df['volume']
    
    # Calculate Intraday Momentum
    M = df['close'] - df['open']
    RM = M.rolling(window=window).sum()
    
    # Integrate Momentum and Volatility
    Sum = W_IV + RM
    
    # Apply Dynamic Exponential Smoothing
    ES = Sum.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive
    Positive_ES = ES + epsilon
    
    # Apply Logarithmic Transformation
    Log_ES = np.log(Positive_ES)
    
    # Factor Output
    Alpha_Factor = Log_ES
    
    return Alpha_Factor
```
