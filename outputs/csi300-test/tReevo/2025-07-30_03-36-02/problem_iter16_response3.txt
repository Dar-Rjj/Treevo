```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=5, smoothing_factor=0.9):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    momentum = close_open_diff.rolling(window=window).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothed_value = integrated_value.ewm(alpha=1 - smoothing_factor).mean()
    
    # Ensure Values are Positive
    positive_smoothed_value = smoothed_value + 1e-6
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    return log_transformed_value
```
