import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()  # Example: 5-day rolling window
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    integrated_value += 1e-6  # Add a small constant to avoid log of zero or negative
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9  # Smoothing factor
    smoothed_values = [integrated_value.iloc[0]]  # Initialize with the first value
    for i in range(1, len(integrated_value)):
        smoothed_values.append(alpha * integrated_value.iloc[i] + (1 - alpha) * smoothed_values[-1])
    smoothed_series = pd.Series(smoothed_values, index=integrated_value.index)
    
    # Apply Logarithmic Transformation
    log_transformed = np.log(smoothed_series)
    
    return log_transformed
