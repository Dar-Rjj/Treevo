```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = (df['high'] - df['low']) ** 2
    
    # Volume-Weighted Intraday Return
    close_to_open_diff = df['close'] - df['open']
    volume_weighted_intraday_return = close_to_open_diff * df['volume']
    
    # Adjusted Volume Based on Intraday Range
    intraday_range = df['high'] - df['low']
    adjusted_volume = df['volume'] / intraday_range
    
    # Integrate Intraday Volatility and Volume-Weighted Intraday Return
    integrated_factor = intraday_volatility + volume_weighted_intraday_return
    
    # Combine Factors
    final_factor = integrated_factor * adjusted_volume
    
    return final_factor
```
