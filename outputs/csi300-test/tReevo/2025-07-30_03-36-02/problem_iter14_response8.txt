import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility with Adaptive Weighting
    adaptive_weight_volatility = 0.5
    adaptive_weight_momentum = 0.5
    integrated_factor = (weighted_volatility * adaptive_weight_volatility) + (rolling_momentum * adaptive_weight_momentum)
    
    # Apply Exponential Smoothing
    alpha = 0.3
    smoothed_factor = integrated_factor.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Positive Values
    epsilon = 1e-6
    positive_smoothed_factor = smoothed_factor + epsilon
    
    # Apply Logarithmic Transformation
    factor_output = np.log(positive_smoothed_factor)
    
    return factor_output
