```python
import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Volume-to-Price Volatility Ratio
    volume_aggregated = df['volume'].rolling(window=N).sum()
    price_range = (df['high'] - df['low']).rolling(window=N).mean()
    volume_to_price_volatility_ratio = volume_aggregated / price_range
    
    # Compute Moving Average of Volume-to-Price Volatility Ratio
    moving_average_ratio = volume_to_price_volatility_ratio.rolling(window=N).mean()
    
    # Synchronization Signal
    synchronization_signal = volume_to_price_volatility_ratio - moving_average_ratio
    
    # Combine with Intraday High-Low Spread
    factor_values = high_low_spread * (synchronization_signal > 0) - high_low_spread * (synchronization_signal <= 0)
    
    return factor_values
```
