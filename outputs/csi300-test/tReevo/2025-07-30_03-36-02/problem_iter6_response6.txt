```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    momentum = df['close'] - df['close'].shift(1)  # Using 1 day lag for momentum
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + momentum + close_open_diff
    
    # Apply Exponential Smoothing
    alpha = 0.5  # Smoothing factor
    smoothed_factor = integrated_factor.ewm(alpha=alpha).mean()
    
    return smoothed_factor
```
