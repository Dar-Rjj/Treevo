```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + daily_momentum
    
    # Apply Exponential Smoothing
    alpha = 0.2  # Smoothing factor, can be tuned
    integrated_factor = integrated_factor.ewm(alpha=alpha, adjust=False).mean()
    
    # Apply Logarithmic Transformation
    integrated_factor = np.log1p(integrated_factor)
    
    return integrated_factor
```
