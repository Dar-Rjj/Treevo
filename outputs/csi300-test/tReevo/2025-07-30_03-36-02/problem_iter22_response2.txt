```python
import pandas as pd
import numpy as np

def heuristics_v2(df, rolling_window=5, smoothing_factor=0.9, small_constant=1e-6):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_close_open_diff = df['close'] - df['open']
    intraday_momentum = daily_close_open_diff.rolling(window=rolling_window).sum()
    
    # Integrate Additional Market Signals
    intraday_range = df['high'] - df['low']
    intraday_volume_change = (df['volume'] / df['volume'].shift(1) - 1) * 100
    integrated_signals = weighted_volatility + intraday_momentum + intraday_range + intraday_volume_change
    
    # Apply Dynamic Exponential Smoothing
    smoothed_value = integrated_signals.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive
    positive_smoothed_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    return log_transformed_value
```
