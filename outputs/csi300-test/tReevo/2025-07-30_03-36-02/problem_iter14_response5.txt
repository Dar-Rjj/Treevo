```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=10, smoothing_factor=0.9, regime_threshold=20):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=window).sum()
    
    # Integrate Momentum and Weighted Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothed_value = integrated_value.ewm(alpha=smoothing_factor).mean()
    
    # Ensure Values are Positive
    positive_value = smoothed_value + 1e-6
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_value)
    
    # Incorporate Market Regime Adaptability
    # Assuming we have a market indicator 'vix' in the dataframe
    high_vol_regime = (df['vix'] > regime_threshold).astype(int)
    low_vol_regime = (df['vix'] <= regime_threshold).astype(int)
    
    # Adjust Factor Weights Based on Current Regime
    factor_weights = high_vol_regime * 1.5 + low_vol_regime * 0.5
    adjusted_factor = log_transformed_value * factor_weights
    
    return adjusted_factor
```
