```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    intraday_momentum = close_open_diff.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    # Assign weights based on recent performance (e.g., 50% for Volatility, 50% for Momentum)
    combined_factor = 0.5 * weighted_volatility + 0.5 * intraday_momentum
    
    # Apply Exponential Smoothing
    smoothed_factor = combined_factor.ewm(alpha=0.3).mean()
    
    # Ensure Positive Values
    positive_smoothed_factor = smoothed_factor + 1e-6
    
    # Factor Output
    # Take the Natural Logarithm of the Positive Value
    final_factor = np.log(positive_smoothed_factor)
    
    return final_factor
```
