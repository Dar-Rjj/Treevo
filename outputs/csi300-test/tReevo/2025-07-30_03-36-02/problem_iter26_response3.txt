```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    rolling_momentum = close_open_diff.rolling(window=10).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + rolling_momentum
    
    # Define Initial Smoothing Factor
    alpha = 0.9
    # Adjust Smoothing Factor Based on Recent Volatility
    recent_volatility = intraday_volatility.rolling(window=5).std()
    adjusted_alpha = 1 / (1 + recent_volatility)
    
    # Apply Exponential Smoothing
    smoothed_factor = integrated_factor.ewm(alpha=adjusted_alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_factor = smoothed_factor + 1e-6
    
    # Calculate Additional Price-Derived Features
    short_term_ma = df['close'].rolling(window=5).mean()
    long_term_ma = df['close'].rolling(window=20).mean()
    moving_average_crossover = short_term_ma - long_term_ma
    
    price_rate_of_change = df['close'].pct_change(10)
    
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow_multiplier = (typical_price - df['low']) - (df['high'] - typical_price) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    accumulation_distribution_line = money_flow_volume.rolling(window=10).sum()
    
    # Combine Integrated Factors and Additional Features
    combined_factor = pd.concat([smoothed_factor, moving_average_crossover, price_rate_of_change, accumulation_distribution_line], axis=1)
    combined_factor.columns = ['Smoothed_Factor', 'MA_Crossover', 'Price_Rate_of_Change', 'Accumulation_Distribution']
    
    # Sum the combined factors
    combined_factor_sum = combined_factor.sum(axis=1)
    
    # Apply Logarithmic Transformation
    log_combined_factor = np.log(combined_factor_sum)
    
    return log_combined_factor
```
