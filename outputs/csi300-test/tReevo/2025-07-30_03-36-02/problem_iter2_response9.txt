```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Price Momentum
    price_momentum = df['close'].diff()
    
    # Combine Intraday Volatility and Price Momentum
    combined_factor = intraday_volatility + price_momentum
    
    # Weight by Volume
    weighted_factor = combined_factor * df['volume']
    
    return weighted_factor
```
