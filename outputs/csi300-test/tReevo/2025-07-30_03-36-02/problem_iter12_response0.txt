import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_window_size = 10
    intraday_momentum = daily_momentum.rolling(window=rolling_window_size).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Define base smoothing factor
    base_smoothing_factor = 0.9
    
    # Calculate recent volatility (5-day moving average)
    recent_volatility = intraday_volatility.rolling(window=5).mean()
    
    # Adjust smoothing factor based on recent volatility
    max_recent_volatility = recent_volatility.max()
    if max_recent_volatility > 0:
        adjusted_smoothing_factor = base_smoothing_factor * (1 - (recent_volatility / max_recent_volatility))
    else:
        adjusted_smoothing_factor = base_smoothing_factor
    
    # Apply dynamic exponential smoothing
    smoothed_value = adjusted_smoothing_factor * integrated_value + (1 - adjusted_smoothing_factor) * integrated_value.shift(1)
    
    # Ensure values are positive
    smoothed_value = smoothed_value + 1e-6
    
    # Apply logarithmic transformation
    log_smoothed_value = np.log(smoothed_value)
    
    return log_smoothed_value
