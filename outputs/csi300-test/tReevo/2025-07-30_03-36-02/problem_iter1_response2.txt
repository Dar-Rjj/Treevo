```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate a trading pressure alpha factor.
    
    :param df: A pandas DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume'], indexed by date.
    :return: A pandas Series representing the factor values, indexed by date.
    """
    # Calculate buying and selling pressure
    buying_pressure = df['open'] - df['low']
    selling_pressure = df['high'] - df['close']
    
    # Net trading pressure
    net_trading_pressure = buying_pressure - selling_pressure
    
    # Adjust for volume intensity
    volume_intensity = df['volume']
    trading_pressure_factor = net_trading_pressure / volume_intensity
    
    return trading_pressure_factor
```
