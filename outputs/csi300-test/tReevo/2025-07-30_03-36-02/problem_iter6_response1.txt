import pandas as pd

def heuristics_v2(df, window_size=5):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Simple Moving Average to Momentum
    moving_avg_momentum = intraday_momentum.rolling(window=window_size).mean()
    
    # Factor Output
    final_factor = integrated_value + moving_avg_momentum
    
    return final_factor
