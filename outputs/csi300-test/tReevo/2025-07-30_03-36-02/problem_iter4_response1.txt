import pandas as pd

def heuristics_v2(df, alpha=0.5):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Volatility
    integrated_mv = weighted_volatility + intraday_momentum
    
    # Incorporate Average Prices
    average_price = (df['high'] + df['low'] + df['close'] + df['open']) / 4
    adjusted_factor = integrated_mv / average_price
    
    # Apply Exponential Smoothing to Momentum
    smoothed_momentum = [intraday_momentum[0]]
    for i in range(1, len(intraday_momentum)):
        smoothed_value = alpha * intraday_momentum[i] + (1 - alpha) * smoothed_momentum[-1]
        smoothed_momentum.append(smoothed_value)
    
    # Final Factor Output
    factor_values = adjusted_factor + pd.Series(smoothed_momentum, index=df.index)
    
    return factor_values
