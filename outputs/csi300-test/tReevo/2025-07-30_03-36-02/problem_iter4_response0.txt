import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight Intraday Volatility by Volume
    weighted_intraday_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Weight Intraday Momentum by Volume
    weighted_intraday_momentum = intraday_momentum * df['volume']
    
    # Integrate Weighted Volatility and Momentum
    integrated_factor = weighted_intraday_volatility + weighted_intraday_momentum
    
    return integrated_factor
