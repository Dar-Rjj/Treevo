import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility_by_volume = intraday_volatility * df['volume']
    
    # Explore Alternative Weighting Methods for Volatility
    weighted_volatility_by_amount = intraday_volatility * df['amount']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Apply Exponential Moving Average (EMA) to Momentum
    ema_momentum = intraday_momentum.ewm(span=10, adjust=False).mean()
    
    # Incorporate Additional Price Features
    daily_range = df['high'] - df['low']
    range_adjusted_momentum = intraday_momentum / daily_range
    ema_range_adjusted_momentum = range_adjusted_momentum.ewm(span=10, adjust=False).mean()
    
    # Integrate Momentum and Volatility
    integrated_signal = weighted_volatility_by_volume + ema_momentum
    # Apply Exponential Smoothing
    smoothed_integrated_signal = integrated_signal.ewm(span=10, adjust=False).mean()
    
    # Combine All Components
    combined_signal = weighted_volatility_by_amount + ema_momentum + ema_range_adjusted_momentum
    # Apply Exponential Smoothing
    final_factor = combined_signal.ewm(span=10, adjust=False).mean()
    
    return final_factor
