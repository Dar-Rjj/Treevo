```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    rolling_momentum = close_open_diff.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    smoothing_factor = 0.9
    smoothed_value = integrated_value.ewm(alpha=1 - smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive
    small_constant = 1e-6
    positive_smoothed_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    factor_values = np.log(positive_smoothed_value)
    
    return factor_values
```
