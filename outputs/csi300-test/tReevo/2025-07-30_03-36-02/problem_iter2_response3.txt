```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Momentum
    intraday_momentum = df['high'] - df['low']
    
    # Calculate Close-to-Open Difference
    close_to_open_diff = df['close'] - df['open']
    
    # Calculate Volume Adjusted by Intraday Range
    intraday_range = df['high'] - df['low']
    adjusted_volume = df['volume'] / (intraday_range + 1e-6)  # Adding a small positive value to avoid division by zero
    
    # Combine Factors
    combined_factors = (intraday_momentum + close_to_open_diff) * adjusted_volume
    
    return combined_factors
```
