import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['Daily_Price_Change'] = df['close'].diff()
    
    # Calculate Absolute Price Change
    df['Abs_Price_Change'] = df['Daily_Price_Change'].abs().rolling(window=5).sum()
    
    # Calculate Volume Weighted Daily Return
    df['Volume_Weighted_Daily_Return'] = (df['Daily_Price_Change'] / df['close'].shift(1)) * df['volume']
    
    # Aggregate Over Window
    df['Cum_Volume'] = df['volume'].rolling(window=5).sum()
    df['Volume_Weighted_Momentum'] = df['Volume_Weighted_Daily_Return'].rolling(window=5).sum() / df['Cum_Volume']
    
    return df['Volume_Weighted_Momentum'].dropna()
