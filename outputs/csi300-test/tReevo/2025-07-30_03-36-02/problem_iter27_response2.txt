```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    rolling_momentum = close_open_diff.rolling(window=5).sum()
    
    # Integrate Weighted Volatility and Momentum
    integrated_metric = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = integrated_metric.ewm(alpha=smoothing_factor).mean()
    
    # Ensure Values are Positive
    smoothed_positive = smoothed_value + 1e-6  # Add a small constant to avoid log of zero or negative
    
    # Apply Logarithmic Transformation
    log_transformed = np.log(smoothed_positive)
    
    return log_transformed
```
