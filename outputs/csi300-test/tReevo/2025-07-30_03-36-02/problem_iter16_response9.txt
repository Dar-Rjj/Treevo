import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_diff = df['close'] - df['open']
    rolling_sum_momentum = daily_diff.rolling(window=20).sum()
    avg_short_momentum = daily_diff.rolling(window=5).mean()
    avg_long_momentum = daily_diff.rolling(window=20).mean()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_sum_momentum
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-6
    smoothed_value = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation
    factor_values = np.log(smoothed_value)
    
    return factor_values
