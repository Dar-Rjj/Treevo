```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Short-Term Price Momentum
    short_momentum = df['close'] - df['close'].shift(5)
    short_momentum_smoothed = short_momentum.ewm(span=5, adjust=False).mean() * 0.9
    
    # Calculate Long-Term Price Momentum
    long_momentum = df['close'] - df['close'].shift(20)
    long_momentum_smoothed = long_momentum.ewm(span=20, adjust=False).mean() * 0.95
    
    # Calculate Volume Trend
    volume_trend = df['volume'].rolling(window=10).mean()
    
    # Combine Factors
    combined_factor = (short_momentum_smoothed - long_momentum_smoothed) / (volume_trend + 1e-6)
    
    return combined_factor
```
