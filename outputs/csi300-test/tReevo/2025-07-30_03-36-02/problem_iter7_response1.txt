```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Combine Weighted Volatility and Momentum
    combined_factor = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_factor = combined_factor.ewm(alpha=1-smoothing_factor).mean()
    
    # Apply Logarithmic Transformation
    transformed_factor = np.log(smoothed_factor)
    
    return transformed_factor
```
