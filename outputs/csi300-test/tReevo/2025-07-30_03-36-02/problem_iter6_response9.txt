```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Multi-Timeframe Intraday Momentum
    daily_momentum = df['close'] - df['open']
    
    # Calculate Short-Term, Medium-Term, and Long-Term EMA of Momentum
    short_term_ema = daily_momentum.ewm(span=5, adjust=False).mean()
    medium_term_ema = daily_momentum.ewm(span=10, adjust=False).mean()
    long_term_ema = daily_momentum.ewm(span=20, adjust=False).mean()
    
    # Integrate Multi-Timeframe Momentum with Weighted Volatility
    integrated_factor_short = weighted_volatility + short_term_ema
    integrated_factor_medium = weighted_volatility + medium_term_ema
    integrated_factor_long = weighted_volatility + long_term_ema
    
    # Combine the factors for the final output
    integrated_factor = (integrated_factor_short + integrated_factor_medium + integrated_factor_long) / 3
    
    return integrated_factor
```
