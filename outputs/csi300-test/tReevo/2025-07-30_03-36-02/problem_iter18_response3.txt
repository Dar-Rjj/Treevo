import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_intraday_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_intraday_volatility + intraday_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.5
    smoothed_values = [integrated_value[0]]  # Initialize with the first value
    for i in range(1, len(integrated_value)):
        smoothed_value = smoothing_factor * integrated_value[i] + (1 - smoothing_factor) * smoothed_values[-1]
        smoothed_values.append(smoothed_value)
    
    # Ensure Values are Positive
    epsilon = 1e-10
    positive_smoothed_values = [value + epsilon for value in smoothed_values]
    
    # Apply Logarithmic Transformation
    log_transformed_values = np.log(positive_smoothed_values)
    
    # Incorporate Previous Day's Integrated Value
    combined_log_values = [log_transformed_values[0]]  # Initialize with the first value
    for i in range(1, len(log_transformed_values)):
        combined_value = 0.7 * log_transformed_values[i] + 0.3 * combined_log_values[-1]
        combined_log_values.append(combined_value)
    
    # Factor Output
    alpha_factor = pd.Series(combined_log_values, index=df.index)
    
    return alpha_factor
