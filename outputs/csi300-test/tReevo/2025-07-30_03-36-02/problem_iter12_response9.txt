```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()  # Rolling window of 5 days
    
    # Integrate Momentum and Weighted Volatility
    integrated_metric = weighted_volatility + rolling_momentum
    smoothing_factor = 0.9
    smoothed_value = integrated_metric.ewm(alpha=1-smoothing_factor).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-8
    positive_smoothed_value = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    return log_transformed_value
```
