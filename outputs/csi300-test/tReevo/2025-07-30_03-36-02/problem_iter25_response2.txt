```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_values = []
    for i, value in enumerate(integrated_value):
        if i == 0:
            smoothed_value = value
        else:
            smoothed_value = smoothing_factor * value + (1 - smoothing_factor) * smoothed_values[-1]
        smoothed_values.append(smoothed_value)
    smoothed_series = pd.Series(smoothed_values, index=df.index)
    
    # Ensure Values are Positive
    smoothed_series += 0.001
    
    # Apply Logarithmic Transformation
    log_transformed = np.log(smoothed_series)
    
    return log_transformed
```
