```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Momentum and Weighted Volatility
    integrated_factor = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing with Alpha
    alpha = 0.3  # You can adjust this value based on your preference
    smoothed_factor = integrated_factor.ewm(alpha=alpha).mean()
    
    return smoothed_factor
```
