```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']

    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    momentum = close_open_diff.rolling(window=5).sum()  # Sum of Close-Open Differences over a Timeframe (e.g., 5 days)
    
    # Integrate Momentum and Weighted Volatility
    integrated_factor = weighted_volatility + momentum
    
    # Apply Exponential Smoothing
    alpha = 0.2  # Smoothing factor
    smoothed_factor = integrated_factor.ewm(alpha=alpha, adjust=False).mean()
    
    # Apply Logarithmic Transformation
    positive_values = smoothed_factor + abs(smoothed_factor.min()) + 1  # Ensure values are positive
    final_factor = np.log(positive_values)
    
    return final_factor
```
