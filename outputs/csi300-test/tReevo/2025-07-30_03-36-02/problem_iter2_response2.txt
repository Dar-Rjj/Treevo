import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Calculate Absolute Price Change
    df['abs_price_change'] = df['daily_price_change'].abs().rolling(window=5).sum()
    
    # Calculate Volume Weighted Daily Return
    df['volume_weighted_return'] = (df['daily_price_change'] / df['close'].shift(1)) * df['volume']
    
    # Aggregate Over 5-day Window
    df['aggregated_vol_weighted_return'] = df['volume_weighted_return'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Calculate Price Momentum
    df['price_momentum'] = df['close'] / df['close'].shift(5)
    
    # Combine Factors
    df['alpha_factor'] = df['aggregated_vol_weighted_return'] * df['price_momentum']
    
    # Return the alpha factor as a pandas Series
    return df['alpha_factor'].dropna()
