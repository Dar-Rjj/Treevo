import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    momentum = close_open_diff.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + momentum
    
    # Apply Dynamic Exponential Smoothing
    base_smoothing_factor = 0.9
    recent_volatility = intraday_volatility.rolling(window=5).mean()
    dynamic_smoothing_factor = base_smoothing_factor * (1 + 0.1 * (recent_volatility / recent_volatility.mean()))
    smoothed_factor = integrated_factor.ewm(alpha=(1 - dynamic_smoothing_factor)).mean()
    
    # Ensure Values are Positive
    smoothed_factor_pos = smoothed_factor + 1e-6
    
    # Apply Logarithmic Transformation
    log_smoothed_factor = np.log(smoothed_factor_pos)
    
    return log_smoothed_factor
