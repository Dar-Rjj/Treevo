import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    
    # Define Rolling Window Sizes
    window_sizes = [5, 10, 20]
    rolling_momentum = {f'rolling_{win}': close_open_diff.rolling(window=win).sum() for win in window_sizes}
    
    # Integrate Momentum and Volatility
    integrated_values = {f'integrated_{win}': weighted_volatility + rolling_momentum[f'rolling_{win}'] for win in window_sizes}
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factors = [0.9, 0.8, 0.7]
    smoothed_values = {}
    for win in window_sizes:
        alpha = smoothing_factors[window_sizes.index(win)]
        smoothed = integrated_values[f'integrated_{win}'].ewm(alpha=alpha, adjust=False).mean()
        smoothed_values[f'smoothed_{win}'] = smoothed
    
    # Ensure Values are Positive
    small_constant = 1e-6
    positive_smoothed_values = {key: value + small_constant for key, value in smoothed_values.items()}
    
    # Apply Logarithmic Transformation
    log_transformed_values = {key: np.log(value) for key, value in positive_smoothed_values.items()}
    
    # Factor Output
    factor_output = pd.concat(log_transformed_values.values(), axis=1).mean(axis=1)
    
    return factor_output
