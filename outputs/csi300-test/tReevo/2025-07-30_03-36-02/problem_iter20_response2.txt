```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    
    # Define Rolling Window Sizes
    rolling_windows = [5, 10, 20]
    
    # Calculate Rolling Sum for Each Window Size
    rolling_momentums = [daily_momentum.rolling(window).sum() for window in rolling_windows]
    
    # Integrate Momentum and Volatility
    integrated_values = [weighted_volatility + momentum for momentum in rolling_momentums]
    
    # Define Adaptive Smoothing Factors
    smoothing_factors = [0.9, 0.8, 0.7]
    
    # Apply Exponential Smoothing Formula for Each Integrated Value
    smoothed_values = []
    for i, (value, alpha) in enumerate(zip(integrated_values, smoothing_factors)):
        smoothed_value = value.ewm(alpha=alpha).mean()
        smoothed_values.append(smoothed_value)
    
    # Ensure Values are Positive
    epsilon = 1e-6
    positive_smoothed_values = [np.log(value + epsilon) for value in smoothed_values]
    
    # Combine the logarithmic transformed values into a single factor
    combined_factor = sum(positive_smoothed_values) / len(positive_smoothed_values)
    
    return combined_factor
```
