```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Weight by Volume
    df['weighted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['close_open_diff'] = df['close'] - df['open']
    momentum_window = 5  # You can adjust the window size
    df['intraday_momentum'] = df['close_open_diff'].rolling(window=momentum_window).sum()
    
    # Integrate Momentum and Volatility
    df['integrated_metric'] = df['weighted_volatility'] + df['intraday_momentum']
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    df['smoothed_value'] = df['integrated_metric'].ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive and Transform
    epsilon = 1e-6  # Small constant to avoid log of zero or negative
    df['factor'] = np.log(df['smoothed_value'] + epsilon)
    
    return df['factor']
```
