```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    momentum = close_open_diff.rolling(window=5).sum()
    
    # Integrate Momentum and Weighted Volatility
    integrated_value = weighted_volatility + momentum
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_value += 1e-6  # Small constant to avoid log of zero or negative
    
    # Apply Logarithmic Transformation
    transformed_value = np.log(smoothed_value)
    
    # Calculate Rolling Metrics
    rolling_mean = transformed_value.rolling(window=20).mean()
    rolling_std = transformed_value.rolling(window=20).std()
    
    # Factor Output: Combined Rolling Metrics and Integrated, Transformed Intraday Momentum and Volatility
    factor = (transformed_value - rolling_mean) / rolling_std
    
    return factor
```
