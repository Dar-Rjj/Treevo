```python
import pandas as pd
import numpy as np

def heuristics_v2(df, n=5, alpha=0.9, epsilon=1e-6):
    """
    Calculate the integrated and transformed intraday momentum and volatility factor.
    
    Parameters:
    - df: pandas DataFrame with columns ['open', 'high', 'low', 'close', 'volume']
    - n: int, number of days for the rolling window
    - alpha: float, smoothing factor for exponential smoothing
    - epsilon: float, small constant to avoid log of zero or negative values
    
    Returns:
    - pandas Series, the factor values
    """
    # Calculate Intraday Volatility (High - Low)
    intraday_volatility = df['high'] - df['low']
    
    # Volume-Weighted Intraday Volatility
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum (Close - Open) and its rolling sum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=n).sum()
    
    # Integrate Weighted Volatility and Momentum
    integrated_factor = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothed_factor = integrated_factor.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_factor = smoothed_factor + epsilon
    
    # Apply Logarithmic Transformation
    transformed_factor = np.log(smoothed_factor)
    
    return transformed_factor
```
