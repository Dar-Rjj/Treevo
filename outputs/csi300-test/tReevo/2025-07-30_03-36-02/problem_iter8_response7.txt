```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Normalize Volatility by Close Price
    normalized_volatility = intraday_volatility / df['close']
    
    # Weight by Volume
    weighted_volatility = normalized_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Combine Weighted Volatility and Momentum
    combined_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = combined_value.ewm(alpha=1 - smoothing_factor).mean()
    
    # Apply Logarithmic Transformation
    transformed_value = np.log(smoothed_value)
    
    # Factor Output
    factor_values = transformed_value
    
    return factor_values
```
