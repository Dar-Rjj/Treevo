```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=10, smoothing_factor=0.9):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=window).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothed_values = [integrated_value[0]]
    for i in range(1, len(integrated_value)):
        smoothed_value = smoothing_factor * integrated_value[i] + (1 - smoothing_factor) * smoothed_values[-1]
        smoothed_values.append(smoothed_value)
    
    # Ensure Values are Positive
    smoothed_values = np.array(smoothed_values) + 1e-6  # Add a small constant to avoid log of zero or negative
    
    # Apply Logarithmic Transformation
    transformed_values = np.log(smoothed_values)
    
    # Factor Output
    factor_series = pd.Series(transformed_values, index=df.index, name='factor')
    
    return factor_series
```
