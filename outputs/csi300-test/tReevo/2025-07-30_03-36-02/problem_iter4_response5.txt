import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Short-Term Momentum
    short_term_momentum = df['close'] - df['close'].shift(5)
    
    # Calculate Medium-Term Momentum
    medium_term_momentum = df['close'] - df['close'].shift(10)
    
    # Calculate Long-Term Momentum
    long_term_momentum = df['close'] - df['close'].shift(20)
    
    # Weight by Volume
    volume_weighted_short_term_momentum = short_term_momentum * df['volume']
    volume_weighted_medium_term_momentum = medium_term_momentum * df['volume']
    volume_weighted_long_term_momentum = long_term_momentum * df['volume']
    
    # Combine Intraday Volatility and Volume-Adjusted Multi-Period Momentum
    combined_factor = (intraday_volatility 
                       + volume_weighted_short_term_momentum 
                       + volume_weighted_medium_term_momentum 
                       + volume_weighted_long_term_momentum)
    
    # Final Alpha Factor
    alpha_factor = combined_factor
    
    return alpha_factor
