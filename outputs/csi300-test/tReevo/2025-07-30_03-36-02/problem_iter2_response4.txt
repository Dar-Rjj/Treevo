import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']
    
    # Compute Daily Price Change
    df['daily_price_change'] = df['close'].diff()
    
    # Volume-Weighted Intraday Momentum
    df['volume_weighted_momentum'] = (df['daily_price_change'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Aggregate Volume and Price Range over N periods
    N = 5
    df['cumulative_volume'] = df['volume'].rolling(window=N).sum()
    df['price_range'] = df[['high', 'low']].rolling(window=N).mean().diff(axis=1)['high']
    df['volume_to_price_volatility_ratio'] = df['cumulative_volume'] / df['price_range']
    
    # Moving Average of Volume-to-Price Volatility Ratio
    df['ratio_ma'] = df['volume_to_price_volatility_ratio'].rolling(window=N).mean()
    
    # Synchronization Signal
    df['synchronization_signal'] = (df['volume_to_price_volatility_ratio'] - df['ratio_ma']).apply(lambda x: 1 if x > 0 else -1)
    
    # Combine Intraday High-Low Spread and Volume-Weighted Intraday Momentum
    df['combined_factor'] = df['synchronization_signal'] * df['high_low_spread'] + df['volume_weighted_momentum']
    
    return df['combined_factor'].dropna()
