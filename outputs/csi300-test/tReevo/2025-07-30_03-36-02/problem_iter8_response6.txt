```python
import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday Volatility
    intraday_volatility = data['high'] - data['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * data['volume']
    
    # Calculate Intraday Momentum
    momentum = data['close'] - data['open']
    
    # Apply Exponential Moving Average (EMA) to Momentum
    ema_momentum = momentum.ewm(span=10, adjust=False).mean()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + ema_momentum
    
    # Apply Exponential Smoothing
    alpha_factor = integrated_factor.ewm(alpha=0.1, adjust=False).mean()
    
    return alpha_factor
```
