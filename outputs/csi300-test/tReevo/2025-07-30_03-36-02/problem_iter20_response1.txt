import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    
    # Define Rolling Window Sizes
    rolling_windows = [5, 10, 20]
    
    # Calculate Rolling Sum for Each Window Size
    rolling_momentum = {window: close_open_diff.rolling(window=window).sum() for window in rolling_windows}
    
    # Define Adaptive Smoothing Factors
    smoothing_factors = [0.9, 0.8, 0.7]
    
    # Integrate Momentum and Volatility and Apply Dynamic Exponential Smoothing
    integrated_values = []
    for i, window in enumerate(rolling_windows):
        integrated_value = weighted_volatility + rolling_momentum[window]
        smoothed_value = integrated_value.ewm(alpha=smoothing_factors[i]).mean()
        integrated_values.append(smoothed_value)
    
    # Combine the integrated values
    final_factor = sum(integrated_values)
    
    # Ensure Values are Positive
    final_factor += 1e-6
    
    # Apply Logarithmic Transformation
    final_factor = np.log(final_factor)
    
    return final_factor
