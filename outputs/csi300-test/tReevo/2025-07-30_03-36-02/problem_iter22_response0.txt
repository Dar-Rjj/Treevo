import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    momentum = df['close'] - df['open']
    rolling_momentum = momentum.rolling(window=10).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + rolling_momentum
    
    # Apply Adaptive Exponential Smoothing
    alpha = 0.9  # Initial smoothing factor
    smoothed_value = [integrated_value[0]]
    for i in range(1, len(integrated_value)):
        alpha = 0.9 - 0.8 * (intraday_volatility[i] / intraday_volatility[:i].max())
        smoothed_value.append(alpha * integrated_value[i] + (1 - alpha) * smoothed_value[-1])
    
    # Ensure Values are Positive
    smoothed_value = np.array(smoothed_value) + 1e-6
    
    # Incorporate Additional Market Data
    price_average = (df['high'] + df['low'] + df['close']) / 3
    integrated_final = 0.4 * smoothed_value + 0.3 * rolling_momentum + 0.3 * price_average
    
    # Apply Logarithmic Transformation
    log_transformed = np.log(integrated_final)
    
    # Factor Output
    factor = pd.Series(log_transformed, index=df.index, name='alpha_factor')
    return factor
