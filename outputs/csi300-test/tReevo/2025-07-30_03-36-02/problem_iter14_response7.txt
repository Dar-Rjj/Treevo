import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_window = 5  # Define a rolling window, e.g., 5 days
    intraday_momentum = daily_momentum.rolling(window=rolling_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9  # Initial smoothing factor
    smoothed_values = [integrated_value[0]]  # Initialize with the first value
    
    for i in range(1, len(integrated_value)):
        if i < 10:
            smoothed_value = alpha * integrated_value[i] + (1 - alpha) * smoothed_values[-1]
        else:
            recent_variance = np.var(integrated_value.iloc[i-10:i])
            if recent_variance > 0.01:  # Adjust based on recent variance
                alpha = 0.9  # Higher smoothing for high variance
            else:
                alpha = 0.6  # Lower smoothing for low variance
            smoothed_value = alpha * integrated_value[i] + (1 - alpha) * smoothed_values[-1]
        smoothed_values.append(smoothed_value)
    
    # Ensure values are positive
    smoothed_series = pd.Series(smoothed_values, index=integrated_value.index)
    smoothed_series = smoothed_series + 1e-6  # Add a small constant to avoid log of zero or negative
    
    # Apply Logarithmic Transformation
    final_factor = np.log(smoothed_series)
    
    return final_factor
