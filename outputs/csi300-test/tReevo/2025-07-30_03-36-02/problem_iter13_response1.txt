import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=10).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9  # Initial smoothing factor
    smoothed_factor = [integrated_factor.iloc[0]]
    for i in range(1, len(integrated_factor)):
        smoothed_factor.append(smoothing_factor * smoothed_factor[-1] + (1 - smoothing_factor) * integrated_factor.iloc[i])
    smoothed_factor = pd.Series(smoothed_factor, index=integrated_factor.index)
    
    # Ensure Values are Positive
    smoothed_factor = smoothed_factor + 1e-6  # Add a small constant to avoid log of zero or negative
    
    # Adjust for Market Conditions
    market_volatility = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()
    market_momentum = df['close'].pct_change().rolling(window=10).mean()
    market_condition_indicator = market_volatility + market_momentum.abs()
    
    # Adjust Smoothing Factor Based on Market Condition
    threshold = market_condition_indicator.median()
    smoothing_factor_adjusted = np.where(market_condition_indicator > threshold, smoothing_factor * 0.8, smoothing_factor * 1.2)
    smoothed_factor_adjusted = []
    for i in range(len(integrated_factor)):
        if i == 0:
            smoothed_factor_adjusted.append(integrated_factor.iloc[0])
        else:
            smoothed_factor_adjusted.append(smoothing_factor_adjusted[i] * smoothed_factor_adjusted[-1] + (1 - smoothing_factor_adjusted[i]) * integrated_factor.iloc[i])
    smoothed_factor_adjusted = pd.Series(smoothed_factor_adjusted, index=integrated_factor.index)
    
    # Apply Logarithmic Transformation
    final_factor = np.log(smoothed_factor_adjusted)
    
    return final_factor
