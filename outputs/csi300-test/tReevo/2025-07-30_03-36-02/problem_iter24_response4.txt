```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = df['close'] - df['open']
    
    # Integrate Weighted Volatility and Momentum
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = integrated_value.ewm(alpha=1 - smoothing_factor).mean()
    
    # Add a Small Constant to Ensure Positive Values
    epsilon = 1e-6
    smoothed_value += epsilon
    
    # Apply Logarithmic Transformation
    log_smoothed_value = np.log(smoothed_value)
    
    return log_smoothed_value
```
