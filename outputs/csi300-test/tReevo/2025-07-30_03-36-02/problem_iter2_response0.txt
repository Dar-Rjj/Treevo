```python
import pandas as pd

def heuristics_v2(df, n=5):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Calculate Multi-Period Momentum
    close_t = df['close']
    close_t_n = df['close'].shift(n)
    multi_period_momentum = close_t - close_t_n
    
    # Weight by Volume
    volume_weighted_momentum = multi_period_momentum * df['volume']
    
    # Combine Intraday Volatility and Volume-Adjusted Multi-Period Momentum
    combined_factor = intraday_volatility + volume_weighted_momentum
    
    return combined_factor
```
