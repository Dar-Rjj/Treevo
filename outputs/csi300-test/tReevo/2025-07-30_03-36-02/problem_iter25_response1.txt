```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Weight by Volume
    df['weighted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['close_open_diff'] = df['close'] - df['open']
    df['rolling_momentum'] = df['close_open_diff'].rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    df['combined_value'] = df['weighted_volatility'] + df['rolling_momentum']
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    df['smoothed_value'] = df['combined_value'].ewm(alpha=(1 - alpha), adjust=False).mean()
    
    # Ensure Values are Positive
    df['adjusted_value'] = df['smoothed_value'] + 0.0001
    
    # Apply Logarithmic Transformation
    df['transformed_value'] = np.log(df['adjusted_value'])
    
    # Factor Output
    return df['transformed_value']
```
