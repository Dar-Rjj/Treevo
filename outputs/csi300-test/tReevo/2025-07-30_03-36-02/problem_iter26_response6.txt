```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_iv = intraday_volatility * df['volume']

    # Calculate Intraday Momentum
    daily_diff = df['close'] - df['open']
    rolling_momentum = daily_diff.rolling(window=5).sum()
    abs_rolling_momentum = rolling_momentum.abs()
    scaled_momentum = abs_rolling_momentum * 0.1  # Example scaling factor, adjust as needed

    # Integrate Momentum and Volatility
    integrated_sum = weighted_iv + scaled_momentum

    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    smoothed_values = [integrated_sum.iloc[0]]
    for i in range(1, len(integrated_sum)):
        smoothed_value = alpha * integrated_sum.iloc[i] + (1 - alpha) * smoothed_values[-1]
        smoothed_values.append(smoothed_value)
    smoothed_series = pd.Series(smoothed_values, index=integrated_sum.index)

    # Ensure Values are Positive
    positive_smoothed_series = smoothed_series + 1e-6  # Small constant to avoid log of zero or negative

    # Apply Logarithmic Transformation
    log_transformed_series = np.log(positive_smoothed_series)

    # Factor Output
    alpha_factor = log_transformed_series
    
    return alpha_factor
```
