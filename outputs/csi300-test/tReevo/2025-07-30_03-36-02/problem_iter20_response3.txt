```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    IV = df['high'] - df['low']
    
    # Weight by Volume
    W_IV = IV * df['volume']
    
    # Calculate Intraday Momentum
    M = df['close'] - df['open']
    Rolling_M = M.rolling(window=5).sum()
    
    # Integrate Weighted Volatility and Momentum
    Sum = W_IV + Rolling_M
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.9
    ES = Sum.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-10
    Positive_ES = ES + epsilon
    
    # Apply Logarithmic Transformation
    Log_ES = np.log(Positive_ES)
    
    # Factor Output
    Alpha_Factor = Log_ES
    
    return Alpha_Factor
```
