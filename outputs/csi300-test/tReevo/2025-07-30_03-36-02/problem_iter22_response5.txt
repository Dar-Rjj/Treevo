```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    df['IV'] = df['high'] - df['low']
    
    # Weight by Volume
    df['W_IV'] = df['IV'] * df['volume']
    
    # Calculate Intraday Momentum
    df['M'] = df['close'] - df['open']
    df['Rolling_M'] = df['M'].rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    df['Sum'] = df['W_IV'] + df['Rolling_M']
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.2  # Define Adaptive Smoothing Factor based on Market Conditions
    df['ES'] = df['Sum'].ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    epsilon = 1e-6
    df['Positive_ES'] = df['ES'] + epsilon
    
    # Apply Logarithmic Transformation
    df['Log_ES'] = np.log(df['Positive_ES'])
    
    # Alpha Factor Output
    alpha_factor = df['Log_ES']
    
    return alpha_factor
```
