import pandas as pd
import numpy as np
from scipy.stats import zscore

def heuristics_v2(df, macroeconomic_data, sector_map, alpha=0.5, beta=0.3):
    # Calculate Intraday Volatility
    df['intraday_volatility'] = df['high'] - df['low']
    
    # Weight by Volume
    df['weighted_volatility'] = df['intraday_volatility'] * df['volume']
    
    # Calculate Intraday Momentum
    df['momentum'] = df['close'] - df['open']
    df['sum_momentum'] = df.groupby(df.index.date)['momentum'].transform('sum')
    
    # Adjust for Sector-Specific Volatility
    df['sector'] = df.index.map(sector_map)
    sector_volatility = df.groupby('sector')['intraday_volatility'].std()
    df['sector_volatility'] = df['sector'].map(sector_volatility)
    df['normalized_volatility'] = (df['intraday_volatility'] / df['sector_volatility']) * df['volume']
    
    # Integrate Momentum and Weighted Volatility
    df['integrated_factor'] = df['weighted_volatility'] + df['sum_momentum']
    df['smoothed_factor'] = df['integrated_factor'].ewm(alpha=alpha).mean()
    df['positive_factor'] = np.log1p(df['smoothed_factor'])
    
    # Incorporate External Macroeconomic Indicators
    gdp_weight = 0.4
    inflation_weight = 0.3
    interest_rate_weight = 0.3
    macroeconomic_data['gdp_growth'] = (macroeconomic_data['gdp_growth'] - macroeconomic_data['gdp_growth'].mean()) / macroeconomic_data['gdp_growth'].std()
    macroeconomic_data['inflation_rate'] = (macroeconomic_data['inflation_rate'] - macroeconomic_data['inflation_rate'].mean()) / macroeconomic_data['inflation_rate'].std()
    macroeconomic_data['interest_rate'] = (macroeconomic_data['interest_rate'] - macroeconomic_data['interest_rate'].mean()) / macroeconomic_data['interest_rate'].std()
    
    df = df.merge(macroeconomic_data, left_index=True, right_index=True, how='left')
    df['adjusted_factor'] = (
        df['positive_factor'] +
        df['gdp_growth'] * gdp_weight +
        df['inflation_rate'] * inflation_weight +
        df['interest_rate'] * interest_rate_weight
    )
    df['final_factor'] = df['adjusted_factor'].ewm(alpha=beta).mean()
    
    return df['final_factor']
