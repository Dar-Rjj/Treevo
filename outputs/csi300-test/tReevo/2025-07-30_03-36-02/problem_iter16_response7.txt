```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=10):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=window).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    rolling_std = integrated_factor.rolling(window=window).std()
    median_rolling_std = rolling_std.rolling(window=window).median()
    smoothing_factor = 0.9 * (1 + 0.1 * (rolling_std / median_rolling_std))
    smoothed_factor = integrated_factor.ewm(alpha=smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_factor = smoothed_factor + 1e-6  # Small constant to avoid log of zero or negative
    
    # Apply Logarithmic Transformation
    log_transformed_factor = np.log(smoothed_factor)
    
    return log_transformed_factor
```
