```python
import pandas as pd

def heuristics_v2(data: pd.DataFrame) -> pd.Series:
    # Calculate Intraday Volatility
    intraday_volatility = data['high'] - data['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * data['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = data['close'] - data['open']
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    alpha = 0.1
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    return smoothed_value
```
