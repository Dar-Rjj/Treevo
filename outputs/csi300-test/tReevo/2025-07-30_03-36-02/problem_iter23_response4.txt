```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    rolling_momentum = daily_momentum.rolling(window=5).sum()
    
    # Integrate Momentum and Volatility
    combined_metric = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_values = [combined_metric.iloc[0]]
    for i in range(1, len(combined_metric)):
        smoothed_value = smoothing_factor * combined_metric.iloc[i] + (1 - smoothing_factor) * smoothed_values[-1]
        smoothed_values.append(smoothed_value)
    smoothed_series = pd.Series(smoothed_values, index=df.index)
    
    # Ensure values are positive
    epsilon = 1e-6
    smoothed_series_positive = smoothed_series + epsilon
    
    # Apply Logarithmic Transformation
    log_transformed = np.log(smoothed_series_positive)
    
    return log_transformed
```
