```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Incorporate Price Change
    daily_return = (df['close'] / df['close'].shift(1)) - 1
    adjusted_volatility = weighted_volatility + daily_return
    
    # Factor Output
    return adjusted_volatility
```
