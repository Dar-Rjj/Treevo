import pandas as pd
import numpy as np
from scipy.stats import boxcox

def heuristics_v2(df, rolling_window=10, alpha_smoothing=0.6):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    momentum_rolling_sum = daily_momentum.rolling(window=rolling_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + momentum_rolling_sum
    integrated_factor = integrated_factor.ewm(alpha=alpha_smoothing).mean()
    
    # Apply Logarithmic Transformation
    # Ensure values are positive by adding a small constant
    epsilon = 1e-6
    integrated_factor += epsilon
    factor_log_transformed = np.log(integrated_factor)
    
    return factor_log_transformed
