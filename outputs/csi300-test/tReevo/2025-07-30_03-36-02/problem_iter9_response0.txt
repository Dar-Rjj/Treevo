```python
import pandas as pd
import numpy as np

def heuristics_v2(df, alpha=0.3, smoothing_window=10):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    momentum_sum = daily_momentum.rolling(window=smoothing_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + momentum_sum
    
    # Apply Exponential Smoothing
    integrated_factor_smoothed = integrated_factor.ewm(alpha=alpha, adjust=False).mean()
    
    # Apply Logarithmic Transformation
    positive_values = np.maximum(integrated_factor_smoothed, 0) + 1e-6  # Ensure values are positive
    log_transformed = np.log(positive_values)
    
    return pd.Series(log_transformed, index=df.index, name='heuristics_v2')
```
