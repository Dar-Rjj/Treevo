import pandas as pd
import numpy as np

def heuristics_v2(df, short_window=5, long_window=20, alpha=0.3, epsilon=1e-8):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    momentum = close_open_diff.rolling(window=short_window).sum()
    
    # Additional Momentum Features
    short_avg_momentum = close_open_diff.rolling(window=short_window).mean()
    long_avg_momentum = close_open_diff.rolling(window=long_window).mean()
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + momentum
    
    # Apply Exponential Smoothing
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_value += epsilon
    
    # Apply Logarithmic Transformation
    factor_values = np.log(smoothed_value)
    
    return factor_values
