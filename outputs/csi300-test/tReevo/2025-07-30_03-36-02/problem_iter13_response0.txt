```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=10, alpha=0.9, epsilon=1e-8):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    rolling_momentum = close_open_diff.rolling(window=window).sum()
    
    # Integrate Momentum and Volatility
    integrated_values = weighted_volatility + rolling_momentum
    
    # Apply Dynamic Exponential Smoothing
    smoothed_values = integrated_values.ewm(alpha=alpha).mean()
    
    # Ensure Values are Positive
    positive_smoothed_values = smoothed_values + epsilon
    
    # Apply Logarithmic Transformation
    log_transformed_values = np.log(positive_smoothed_values)
    
    return log_transformed_values
```
