```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    intraday_momentum = daily_momentum.rolling(window=5).sum()  # Timeframe of 5 days for example
    
    # Integrate Weighted Volatility and Momentum
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    smoothing_factor = 0.9
    smoothed_value = integrated_value.ewm(alpha=1 - smoothing_factor, adjust=False).mean()
    
    # Ensure Values are Positive
    small_constant = 1e-6
    smoothed_positive_value = smoothed_value + small_constant
    
    # Apply Logarithmic Transformation
    final_factor = np.log(smoothed_positive_value)
    
    return final_factor
```
