```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    intraday_momentum = close_open_diff.rolling(window=10).sum()  # Timeframe of 10 days
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Exponential Smoothing
    alpha = 0.9  # Smoothing Factor
    smoothed_value = integrated_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_value += 1e-6  # Small constant to avoid log of zero or negative
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(smoothed_value)
    
    return log_transformed_value
```
