```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_return = df['high'] - df['low']
    
    # Volume Weighted Close-to-Open Difference
    close_open_diff = df['close'] - df['open']
    volume_weighted_close_open_diff = close_open_diff * df['volume']
    
    # Adjusted Volume Based on Intraday Range
    intraday_range = df['high'] - df['low']
    adjusted_volume = df['volume'] / intraday_range
    
    # Combine Factors
    factor_sum = intraday_return + volume_weighted_close_open_diff
    final_factor = factor_sum * adjusted_volume
    
    return final_factor
```
