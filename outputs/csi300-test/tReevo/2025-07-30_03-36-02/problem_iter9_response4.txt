```python
import pandas as pd
import numpy as np

def heuristics_v2(df, alpha=0.6, smoothing_window=5):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    daily_momentum = df['close'] - df['open']
    momentum = daily_momentum.rolling(window=smoothing_window).sum()
    
    # Integrate Momentum and Volatility
    integrated_factor = weighted_volatility + momentum
    
    # Apply Exponential Smoothing
    integrated_factor = integrated_factor.ewm(alpha=alpha).mean()
    
    # Ensure values are positive for logarithmic transformation
    integrated_factor = np.maximum(integrated_factor, 1e-10)
    
    # Apply Logarithmic Transformation
    log_transformed_factor = np.log(integrated_factor)
    
    return log_transformed_factor
```
