```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Intraday Volatility
    intraday_volatility = df['high'] - df['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * df['volume']
    
    # Calculate Intraday Momentum
    close_open_diff = df['close'] - df['open']
    rolling_window = 10  # Define the rolling window size
    intraday_momentum = close_open_diff.rolling(window=rolling_window).sum()
    
    # Integrate Momentum and Volatility
    combined_value = weighted_volatility + intraday_momentum
    
    # Apply Fixed Exponential Smoothing
    alpha = 0.9
    smoothed_value = combined_value.ewm(alpha=alpha, adjust=False).mean()
    
    # Ensure Values are Positive
    smoothed_value = smoothed_value + 0.0001
    
    # Apply Logarithmic Transformation
    log_smoothed_value = np.log(smoothed_value)
    
    return log_smoothed_value
```
