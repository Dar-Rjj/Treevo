```python
import pandas as pd
import numpy as np

def heuristics_v2(data):
    # Calculate Intraday Volatility
    intraday_volatility = data['high'] - data['low']
    
    # Weight by Volume
    weighted_volatility = intraday_volatility * data['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = data['close'] - data['open']
    
    # Integrate Momentum and Volatility
    integrated_value = weighted_volatility + intraday_momentum
    
    # Apply Dynamic Exponential Smoothing
    alpha = 0.5
    smoothed_value = integrated_value.copy()
    for i in range(1, len(smoothed_value)):
        smoothed_value.iloc[i] = alpha * integrated_value.iloc[i] + (1 - alpha) * smoothed_value.iloc[i - 1]
    
    # Ensure Values are Positive
    epsilon = 1e-8
    positive_smoothed_value = smoothed_value + epsilon
    
    # Apply Logarithmic Transformation
    log_transformed_value = np.log(positive_smoothed_value)
    
    # Incorporate Previous Day's Integrated Value
    log_previous_day = log_transformed_value.shift(1).fillna(method='bfill')
    final_factor = 0.7 * log_transformed_value + 0.3 * log_previous_day
    
    return final_factor
```
