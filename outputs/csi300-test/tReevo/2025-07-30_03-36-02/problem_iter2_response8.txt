import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate Intraday High-Low Spread
    df['high_low_spread'] = df['high'] - df['low']

    # Calculate Volume-Weighted Average Price (VWAP)
    df['vwap'] = (df[['open', 'high', 'low', 'close']].mean(axis=1) * df['volume']).rolling(window=N).sum() / df['volume'].rolling(window=N).sum()

    # Calculate Volume-to-Price Volatility Ratio
    df['volume_agg'] = df['volume'].rolling(window=N).sum()
    df['price_range'] = df['high'] - df['low']
    df['avg_price_range'] = df['price_range'].rolling(window=N).mean()
    df['volatility_ratio'] = df['volume_agg'] / df['avg_price_range']

    # Compute Moving Average of Volume-to-Price Volatility Ratio
    df['volatility_ratio_ma'] = df['volatility_ratio'].rolling(window=N).mean()

    # Synchronization Signal
    df['synchronization_signal'] = df['volatility_ratio'] - df['volatility_ratio_ma']
    df['synchronization_weight'] = df['synchronization_signal'].apply(lambda x: 1 if x > 0 else -1)

    # Combine with Intraday High-Low Spread
    df['alpha_factor'] = df['high_low_spread'] * df['synchronization_weight']

    return df['alpha_factor']
