import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adaptive Volatility Calculation
    df['Volatility'] = df[['high', 'low', 'close']].rolling(window=20).std().mean(axis=1)
    
    # Determine adaptive window size based on volatility
    def adjust_window_size(volatility, threshold_high=0.05, threshold_low=0.01, min_window=10, max_window=50):
        if volatility > threshold_high:
            return min_window
        elif volatility < threshold_low:
            return max_window
        else:
            return int((max_window - min_window) / (threshold_high - threshold_low) * (volatility - threshold_low) + min_window)
    
    df['Adaptive_Window'] = df['Volatility'].apply(adjust_window_size)
    
    # Rolling Statistics with Adaptive Window
    def rolling_stats(series, window_series):
        return series.rolling(window=window_series).agg(['mean', 'std'])

    df_expanded = df.join(df.groupby('date')['Adaptive_Window'].transform('first'), rsuffix='_expanded')
    rolling_stats_df = df_expanded.groupby('date').apply(lambda x: rolling_stats(x['Volume_Weighted_Return'], x['Adaptive_Window_expanded']))
    
    df = df.merge(rolling_stats_df, left_index=True, right_index=True, suffixes=('', '_rolling'))
    
    # Volatility Adjustment
    df['Volatility_Adjusted_Return'] = df['Volume_Weighted_Return'] / df['std_rolling']
    
    # Final Alpha Factor
    alpha_factor = df['Volatility_Adjusted_Return']
    
    return alpha_factor
