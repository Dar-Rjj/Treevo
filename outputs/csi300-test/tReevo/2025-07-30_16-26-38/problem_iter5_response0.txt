```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['NextDayOpen'] = df['open'].shift(-1)
    close_to_open_return = (df['NextDayOpen'] - df['close']) / df['close']
    
    # Volume Weighting
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Calculate Intraday Momentum
    intraday_momentum = (df['high'] - df['low']) / df['close']
    
    # Adjust for Volatility
    rolling_std = df['close'].rolling(window=20).std()
    adjusted_intraday_momentum = intraday_momentum / rolling_std
    
    # Combine with Volume-Weighted Close-to-Open Return
    factor = volume_weighted_return + adjusted_intraday_momentum
    
    return factor
```
