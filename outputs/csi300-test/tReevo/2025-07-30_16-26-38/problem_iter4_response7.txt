import pandas as pd

def heuristics_v2(df):
    # Calculate Day’s High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']

    # Adaptive rolling window size
    window_sizes = [5, 10, 20]

    for window in window_sizes:
        # Calculate Previous Days’ High-to-Low Range Cumulatively
        df[f'cum_high_low_range_{window}'] = df['high_low_range'].rolling(window=window).sum()

        # Calculate Volume-Weighted Average of Open and Close Prices
        df['volume_weighted_avg'] = (df['open'] + df['close']) / 2

        # Calculate Volume-Adjusted Cumulative High-to-Low Range
        df[f'vol_adj_cum_range_{window}'] = df[f'cum_high_low_range_{window}'] * df['volume_weighted_avg']

        # Calculate Rolling Standard Deviation of Daily High-to-Low Range
        df[f'high_low_range_std_{window}'] = df['high_low_range'].rolling(window=window).std()

        # Adjust for Volatility
        df[f'vol_adj_cum_range_vol_adj_{window}'] = df[f'vol_adj_cum_range_{window}'] / df[f'high_low_range_std_{window}']

        # Calculate Daily Price Momentum
        df['price_momentum'] = df['close'].diff()

        # Calculate Volume-Weighted Price Momentum
        df['vol_weighted_momentum'] = df['price_momentum'] * df['volume']

        # Integrate Momentum into Adjusted Cumulative High-to-Low Range
        df[f'final_factor_{window}'] = df[f'vol_adj_cum_range_vol_adj_{window}'] + df['vol_weighted_momentum']

    # Signal Generation
    df['signal'] = df[f'final_factor_{window_sizes[-1]}'].apply(lambda x: 1 if x > 0 else -1)

    return df['signal']
