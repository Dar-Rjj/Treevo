```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    df['intraday_return'] = (df['close'] - df['open']) / df['open']
    
    # Calculate High-Low Volatility
    df['high_low_volatility'] = (df['high'] - df['low']) / df['low']
    
    # Determine Volume Trend
    df['volume_ratio'] = df['volume'] / df['volume'].shift(1)
    df['volume_trend'] = df['volume_ratio'] - 1
    
    # Combine Factors
    df['combined_factor'] = (
        0.4 * df['intraday_return'] + 
        0.3 * df['high_low_volatility'] +
        0.3 * df['volume_trend']
    )
    
    # Weigh by Average Volume
    average_volume = df['volume'].mean()
    df['weighted_factor'] = df['combined_factor'] * (df['volume'] / average_volume)
    
    return df['weighted_factor']
```
