```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the forward close-to-open return
    df['fwd_c_to_o_return'] = df['open'].shift(-1) - df['close']
    
    # Volume weighting
    df['volume_weighted_return'] = df['fwd_c_to_o_return'] * df['volume']
    
    return df['volume_weighted_return']
```
