import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility
    df['volatility'] = df[['high', 'low', 'close']].pct_change().rolling(window=30).std().mean(axis=1)
    
    # Adjust Window Size based on Volatility
    def adjust_window_size(vol):
        if vol > vol.median():
            return 5  # Smaller window for high volatility
        else:
            return 60  # Larger window for low volatility
    
    df['window_size'] = df['volatility'].apply(adjust_window_size)
    
    # Rolling Statistics with Adaptive Window
    volume_weighted_returns = []
    for i in range(len(df)):
        window = int(df.iloc[i]['window_size'])
        if i + window < len(df):
            mean = df.iloc[i:i+window]['volume_weighted_return'].mean()
            std = df.iloc[i:i+window]['volume_weighted_return'].std()
            volume_weighted_returns.append((mean, std))
        else:
            volume_weighted_returns.append((np.nan, np.nan))
    
    df[['adaptive_mean', 'adaptive_std']] = pd.DataFrame(volume_weighted_returns, index=df.index)
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['adaptive_mean'] / df['adaptive_std']).fillna(0)
    
    return df['alpha_factor']
