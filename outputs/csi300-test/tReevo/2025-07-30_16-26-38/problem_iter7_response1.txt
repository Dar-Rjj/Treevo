```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Determine Volatility using High, Low, and Close prices
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    volatility = typical_price.rolling(window=20).std()
    
    # Adjust Window Size based on Volatility
    def adjust_window_size(vol):
        if vol > volatility.quantile(0.75):
            return 5  # high volatility, decrease window size
        elif vol < volatility.quantile(0.25):
            return 30  # low volatility, increase window size
        else:
            return 10  # medium volatility, use a moderate window size
    
    adaptive_windows = volatility.apply(adjust_window_size)
    
    # Rolling Statistics with Adaptive Windows
    rolling_mean = volume_weighted_return.rolling(window=adaptive_windows, min_periods=1).mean()
    rolling_std = volume_weighted_return.rolling(window=adaptive_windows, min_periods=1).std()
    
    # Standardize the rolling mean by its standard deviation
    factor_values = (rolling_mean / rolling_std).fillna(0)
    
    return factor_values
```
