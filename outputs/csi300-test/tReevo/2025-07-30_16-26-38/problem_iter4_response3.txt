import pandas as pd

def heuristics_v2(df, window_size=10):
    # Calculate Intraday Return
    df['intraday_return'] = df['close'] - df['open']
    
    # Volume-Weighted Intraday Return
    df['volume_weighted_return'] = df['intraday_return'] * df['volume']
    
    # Calculate Day’s High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Previous Days’ High-to-Low Range Cumulatively
    df['cumulative_high_low_range'] = df['high_low_range'].rolling(window=window_size).sum()
    
    # Calculate Volume-Weighted Average of Open and Close Prices
    df['volume_weighted_avg_price'] = (df['open'] + df['close']) / 2
    
    # Calculate Volume-Adjusted Cumulative High-to-Low Range
    df['volume_adjusted_cumulative_high_low'] = df['cumulative_high_low_range'] * df['volume_weighted_avg_price'] * df['volume']
    
    # Calculate Rolling Mean of Volume-Weighted Intraday Return
    df['rolling_mean_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=window_size).mean()
    
    # Calculate Rolling Standard Deviation of Volume-Weighted Intraday Return
    df['rolling_std_volume_weighted_return'] = df['volume_weighted_return'].rolling(window=window_size).std()
    
    # Generate Alpha Signal
    df['standardized_volume_weighted_return'] = (df['volume_weighted_return'] - df['rolling_mean_volume_weighted_return']) / df['rolling_std_volume_weighted_return']
    
    df['alpha_signal'] = df['standardized_volume_weighted_return'].apply(lambda x: 1 if x > 0 else -1)
    
    # Enhance Alpha Signal with High-to-Low Range
    df['standardized_volume_adjusted_cumulative_high_low'] = (df['volume_adjusted_cumulative_high_low'] - df['rolling_mean_volume_weighted_return']) / df['rolling_std_volume_weighted_return']
    
    df['enhanced_alpha_signal'] = df['standardized_volume_weighted_return'] + df['standardized_volume_adjusted_cumulative_high_low']
    df['enhanced_alpha_signal'] = df['enhanced_alpha_signal'].apply(lambda x: 1 if x > 0 else -1)
    
    return df['enhanced_alpha_signal']
