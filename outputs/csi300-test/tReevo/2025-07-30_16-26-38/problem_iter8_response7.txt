```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Forward Close-to-Open Return
    df['forward_c_to_o_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['vol_weighted_fwd_c_to_o_return'] = df['forward_c_to_o_return'] * df['volume']
    
    return df['vol_weighted_fwd_c_to_o_return']
```
