```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']

    # Determine Volatility
    df['range'] = df['high'] - df['low']
    df['volatility'] = df['range'].rolling(window=20).std()
    
    # Adjust Window Size based on Volatility
    def adjust_window_size(row):
        if row['volatility'] > df['volatility'].mean():
            return 10  # Decrease window size if volatility is high
        else:
            return 30  # Increase window size if volatility is low
    
    df['window_size'] = df.apply(adjust_window_size, axis=1)
    
    # Initialize the final alpha factor series
    alpha_factor = pd.Series(index=df.index, dtype='float64')
    
    # Calculate Rolling Statistics with Adaptive Window
    for i in range(len(df)):
        window = int(df['window_size'].iloc[i])
        start = max(i - window + 1, 0)
        end = i + 1
        alpha_factor.iloc[i] = df['volume_weighted_return'].iloc[start:end].mean() / df['volume_weighted_return'].iloc[start:end].std()
    
    return alpha_factor.dropna()
```
