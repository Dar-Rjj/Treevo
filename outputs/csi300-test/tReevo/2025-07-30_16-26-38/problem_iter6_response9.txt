```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return using (High - Low) / Low
    df['Intraday_Return'] = (df['high'] - df['low']) / df['low']
    
    # Measure Overnight Return using (Today's Close - Yesterday's Close) / Yesterday's Close
    df['ONR'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    
    # Combine Intraday with Momentum
    intraday_weight = df['Intraday_Return'].apply(lambda x: 1.5 if x > 0.01 else 1)
    df['Adjusted_Intraday'] = df['Intraday_Return'] * intraday_weight
    
    # Combine ONR with Reversal Expectation
    onr_adjustment = df['ONR'].apply(lambda x: -1 if x > 0.03 else 0.5)
    df['Adjusted_ONR'] = df['ONR'] + onr_adjustment
    
    # Final Composite Score
    df['Composite_Score'] = df['Adjusted_Intraday'] + df['Adjusted_ONR']
    
    # Incorporate Volume
    df['Final_Score'] = df['Composite_Score'] * df['volume']

    return df['Final_Score'].dropna()
```
