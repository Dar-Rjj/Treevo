```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['next_open'] = df['open'].shift(-1)
    df['close_to_open_return'] = (df['next_open'] - df['close']) / df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Adjust for Volatility
    df['true_range'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['daily_volatility'] = df['true_range'].rolling(window=20).std()
    df['volatility_adjusted_return'] = df['volume_weighted_return'] / df['daily_volatility']
    
    # Adaptive Window Rolling Average
    df['close_std'] = df['close'].rolling(window=20).std()
    df['adaptive_window'] = (df['close_std'] * 20 / df['close_std'].mean()).astype(int).clip(lower=5, upper=60)
    df['factor_value'] = df['volatility_adjusted_return'].rolling(window=df['adaptive_window']).mean()
    
    return df['factor_value']
```
