```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility
    df['daily_range'] = df['high'] - df['low']
    initial_window = 20  # Fixed Initial Window for volatility
    df['volatility'] = df['daily_range'].rolling(window=initial_window).std()
    
    # Adjust Window Size based on Volatility
    def adjust_window(volatility, high_threshold=0.5, low_threshold=0.1, base_window=10):
        if volatility > high_threshold:
            return max(2, base_window - 3)  # Decrease window size but not below 2
        elif volatility < low_threshold:
            return min(20, base_window + 3)  # Increase window size but not above 20
        else:
            return base_window
    
    df['window_size'] = df['volatility'].apply(adjust_window)
    
    # Rolling Statistics with Adaptive Window
    rolling_mean = df.groupby(df.index)['volume_weighted_return'].transform(lambda x: x.rolling(window=x.name[1]['window_size']).mean())
    rolling_std = df.groupby(df.index)['volume_weighted_return'].transform(lambda x: x.rolling(window=x.name[1]['window_size']).std())
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['volume_weighted_return'] - rolling_mean) / rolling_std
    
    return df['alpha_factor']
```
