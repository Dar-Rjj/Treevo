```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['Daily_Price_Momentum'] = df['close'].diff(1)
    
    # Measure Volume Impact Factor
    df['Volume_Impact_Factor'] = df['volume'] / df['volume'].shift(1)
    
    # Combine Price Momentum and Volume Impact
    df['Combined_Factor'] = df['Daily_Price_Momentum'] * df['Volume_Impact_Factor']
    
    # Smooth with Moving Average
    df['Final_Alpha_Factor'] = df['Combined_Factor'].rolling(window=5).mean()
    
    return df['Final_Alpha_Factor']
```
