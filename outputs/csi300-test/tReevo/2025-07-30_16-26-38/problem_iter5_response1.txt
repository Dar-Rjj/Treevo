import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Calculate 20-day rolling volatility (standard deviation of Close prices)
    df['rolling_volatility'] = df['close'].rolling(window=20).std()
    
    # Adjust Close-to-Open Return for Volatility
    df['adjusted_close_to_open_return'] = df['close_to_open_return'] / df['rolling_volatility']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['adjusted_close_to_open_return'] * df['volume']
    
    # Return the volume weighted return as the alpha factor
    return df['volume_weighted_return']
