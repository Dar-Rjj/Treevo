```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']

    # Determine Volatility (Rolling Standard Deviation over a Fixed Window, e.g., 20 days)
    df['Volatility'] = df[['high', 'low', 'close']].rolling(window=20).std().mean(axis=1)

    # Adjust Window Size based on Volatility
    df['Window_Size'] = np.where(df['Volatility'] > df['Volatility'].mean(), 5, 30)

    # Rolling Statistics with Adaptive Window
    df['Rolling_Mean'] = df.groupby('Window_Size')['Volume_Weighted_Return'].transform(lambda x: x.rolling(x.name).mean())
    df['Rolling_Std'] = df.groupby('Window_Size')['Volume_Weighted_Return'].transform(lambda x: x.rolling(x.name).std())

    # Normalize the factor using the rolling mean and standard deviation
    df['Alpha_Factor'] = (df['Volume_Weighted_Return'] - df['Rolling_Mean']) / df['Rolling_Std']

    # Drop rows with NaN values created by the rolling window calculations
    df = df.dropna()

    return df['Alpha_Factor']
```
