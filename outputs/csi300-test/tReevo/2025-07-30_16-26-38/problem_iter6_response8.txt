```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Weight by Volume
    volume_weighted_close_to_open = close_to_open_return * df['volume']
    
    # Calculate High-to-Low Range
    high_low_range = df['high'] - df['low']
    
    # Calculate Volume-Weighted High-Low Range
    volume_weighted_high_low_range = high_low_range * df['volume']
    
    # Combine Volume-Weighted Close-to-Open Return and High-Low Range
    combined_factor = volume_weighted_close_to_open + volume_weighted_high_low_range
    
    # Signal Generation
    signal = combined_factor.apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    return signal
```
