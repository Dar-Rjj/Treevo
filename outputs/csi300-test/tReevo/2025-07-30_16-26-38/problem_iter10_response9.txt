```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adjust for Advanced Volatility
    df['Log_Returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['EMA_Volatility'] = df['Log_Returns'].ewm(span=20, adjust=False).std()
    df['Volatility_Adjusted_Return'] = df['Volume_Weighted_Return'] / df['EMA_Volatility']
    
    # Incorporate Intraday Data
    df['Intraday_Range'] = df['high'] - df['low']
    df['Intraday_Adjusted_Return'] = df['Volatility_Adjusted_Return'] * df['Intraday_Range']
    
    # Consider Liquidity Factors
    df['Average_Daily_Volume'] = df['volume'].rolling(window=30).mean()
    df['Volume_Ratio'] = df['volume'] / df['Average_Daily_Volume']
    df['Liquidity_Adjusted_Return'] = df['Intraday_Adjusted_Return'] * df['Volume_Ratio']
    
    # Combine Factors
    df['Alpha_Factor'] = df['Liquidity_Adjusted_Return']
    
    return df['Alpha_Factor']
```
