import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']

    # Volume Weighting
    df['Volume_Weighted_CtoO'] = df['Close_to_Open_Return'] * df['volume']

    # Determine Volatility
    df['Volatility'] = df[['high', 'low', 'close']].rolling(window=30).std().mean(axis=1)

    # Calculate Volume Trend
    df['Volume_Trend'] = df['volume'].rolling(window=30).mean().pct_change()

    # Adaptive Window Calculation
    def adjust_window_size(row):
        base_window = 60
        if row['Volatility'] > df['Volatility'].median():
            window_adj_volatility = max(1, int(base_window * 0.8))
        else:
            window_adj_volatility = min(len(df), int(base_window * 1.2))
        
        if row['Volume_Trend'] > 0:
            window_adj_volume = int(window_adj_volatility * 1.2)
        else:
            window_adj_volume = int(window_adj_volatility * 0.8)
        
        return min(len(df), max(1, window_adj_volume))

    df['Adaptive_Window'] = df.apply(adjust_window_size, axis=1)

    # Rolling Statistics
    rolling_mean = df['Volume_Weighted_CtoO'].rolling(window=df['Adaptive_Window']).mean()
    rolling_std = df['Volume_Weighted_CtoO'].rolling(window=df['Adaptive_Window']).std()

    # Final Factor
    df['Alpha_Factor'] = (rolling_mean / rolling_std).fillna(0)

    return df['Alpha_Factor']
