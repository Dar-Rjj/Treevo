```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close = df['close']
    next_open = df['open'].shift(-1)
    close_to_open_return = (next_open - close) / close
    
    # Volume Weighting
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
