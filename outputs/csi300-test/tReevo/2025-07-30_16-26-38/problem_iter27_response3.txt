import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['vol_weighted_c2o_return'] = df['close_to_open_return'] * df['volume']
    
    # Adaptive Window Calculation
    df['high_low_close_avg'] = (df['high'] + df['low'] + df['close']) / 3
    fixed_window = 20  # Fixed window for volatility calculation
    df['volatility'] = df['high_low_close_avg'].rolling(window=fixed_window).std()
    
    # Adjust Window Size based on Volatility
    min_window = 5
    max_window = 50
    df['adaptive_window'] = ((max_window - min_window) / (df['volatility'].max() - df['volatility'].min())) * (df['volatility'] - df['volatility'].min()) + min_window
    df['adaptive_window'] = df['adaptive_window'].fillna(fixed_window).astype(int)
    
    # Momentum Component
    df['momentum'] = df['close'].rolling(window=fixed_window).mean()
    
    # Combine with Volume Weighted Close-to-Open Return
    df['combined_factor'] = df['vol_weighted_c2o_return'] + df['momentum']
    
    # Rolling Statistics
    def rolling_stats(series, window_series):
        mean_list = [series.iloc[i:i+window].mean() if not np.isnan(window) else series.iloc[i] for i, window in enumerate(window_series)]
        std_list = [series.iloc[i:i+window].std() if not np.isnan(window) else 0 for i, window in enumerate(window_series)]
        return pd.Series(mean_list, index=series.index), pd.Series(std_list, index=series.index)
    
    rolling_mean, rolling_std = rolling_stats(df['combined_factor'], df['adaptive_window'])
    
    # Final Factor
    df['final_factor'] = (df['combined_factor'] - rolling_mean) / rolling_std
    
    return df['final_factor'].dropna()
