```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Return
    intraday_range = (df['high'] - df['low']) / df['low']
    true_intraday_movement = (df['close'] - df['open']) / df['open']

    # Measure Overnight Return
    df['ONR'] = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)

    # Combine Intraday with Momentum
    intraday_momentum = true_intraday_movement.apply(lambda x: 1.5 * x if x > 0.01 else x)

    # Combine ONR with Reversal Expectation
    onr_reversal = df['ONR'].apply(lambda x: -1 if x > 0.03 else 0.5)

    # Volume-Adjusted Momentum
    volume_weighted_avg = (df['open'] + df['close']) / 2
    volume_adjusted_momentum = volume_weighted_avg * df['volume']
    volume_adjusted_momentum = volume_adjusted_momentum.apply(lambda x: 1.2 * x if x > df['volume'].mean() else x)

    # Final Composite Score
    composite_factor = intraday_momentum + onr_reversal + volume_adjusted_momentum

    return composite_factor
```
