import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Smooth out volume using a 5-day moving average
    df['Smoothed_Volume'] = df['volume'].rolling(window=5).mean()
    
    # Calculate Volume Weighted Close-to-Open Return
    df['Volume_Weighted_Close_to_Open_Return'] = df['Close_to_Open_Return'] * df['Smoothed_Volume']
    
    # Calculate Adaptive Rolling Volatility
    df['HL_C_diff'] = df[['high', 'low', 'close']].std(axis=1)
    recent_volatility = df['HL_C_diff'].rolling(window=20).std()
    
    # Adjust window size based on recent volatility
    def adjust_window(volatility, default_window=20):
        if volatility > recent_volatility.median():
            return max(1, int(default_window * 0.8))
        else:
            return min(len(df), int(default_window * 1.2))
    
    adaptive_volatility = df['HL_C_diff'].rolling(window=df['HL_C_diff'].apply(lambda x: adjust_window(x))).std()
    
    # Volatility Adjustment
    df['Volatility_Adjusted_Volume_Weighted_Return'] = df['Volume_Weighted_Close_to_Open_Return'] / adaptive_volatility
    
    # Integrate Trend Information
    df['SMA_50'] = df['close'].rolling(window=50).mean()
    df['Trend_Indicator'] = (df['close'] > df['SMA_50']).astype(int) * 2 - 1  # Positive or Negative Trend
    
    # Combine Factors
    df['Alpha_Factor'] = df['Volatility_Adjusted_Volume_Weighted_Return'] * df['Trend_Indicator']
    
    # Drop rows with NaN values
    df.dropna(inplace=True)
    
    # Return the alpha factor as a pandas Series
    return df['Alpha_Factor']
