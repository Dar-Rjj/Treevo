```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] - df['open']
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Incorporate Recent Price Changes (5-day Price Momentum)
    price_momentum_5d = df['close'] - df['close'].shift(5)
    combined_result = volume_weighted_return + price_momentum_5d
    
    # Apply Exponential Smoothing (Exponential Moving Average over a 5-day window)
    ema_window = 5
    alpha_factor = combined_result.ewm(span=ema_window, adjust=False).mean()
    
    return alpha_factor
```
