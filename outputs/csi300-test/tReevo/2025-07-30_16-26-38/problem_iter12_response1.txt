```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the forward close-to-open return
    df['forward_close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume weighting
    df['volume_weighted_return'] = df['forward_close_to_open_return'] * df['volume']
    
    # Return the volume weighted return as the alpha factor
    return df['volume_weighted_return']
```
