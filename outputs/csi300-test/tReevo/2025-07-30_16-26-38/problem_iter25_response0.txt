import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility
    df['high_low_close_avg'] = (df['high'] + df['low'] + df['close']) / 3
    df['volatility'] = df['high_low_close_avg'].rolling(window=20).std()
    
    # Adjust Window Size Based on Volatility
    def adaptive_window(volatility, high_vol_threshold, low_vol_threshold, base_window, max_window, min_window):
        if volatility > high_vol_threshold:
            return max(min_window, base_window - 5)
        elif volatility < low_vol_threshold:
            return min(max_window, base_window + 5)
        else:
            return base_window
    
    # Parameters for Adaptive Window
    high_vol_threshold = df['volatility'].quantile(0.75)
    low_vol_threshold = df['volatility'].quantile(0.25)
    base_window = 20
    max_window = 40
    min_window = 10
    
    df['adaptive_window'] = df['volatility'].apply(lambda x: adaptive_window(x, high_vol_threshold, low_vol_threshold, base_window, max_window, min_window))
    
    # Calculate Rolling Statistics Using Adaptive Window
    def rolling_stats(data, window_col):
        result = []
        for i in range(len(data)):
            if i < data[window_col].iloc[i]:
                result.append(np.nan)
            else:
                start = i - data[window_col].iloc[i] + 1
                subset = data.iloc[start:i+1]
                mean = subset.mean()
                std = subset.std()
                result.append((mean, std))
        return result
    
    rolling_stats_result = rolling_stats(df[['volume_weighted_return', 'adaptive_window']], 'adaptive_window')
    df['rolling_mean'] = [x[0]['volume_weighted_return'] if not isinstance(x, float) else np.nan for x in rolling_stats_result]
    df['rolling_std'] = [x[1]['volume_weighted_return'] if not isinstance(x, float) else np.nan for x in rolling_stats_result]
    
    # Final Alpha Factor
    df['alpha_factor'] = (df['volume_weighted_return'] - df['rolling_mean']) / df['rolling_std']
    
    return df['alpha_factor']
