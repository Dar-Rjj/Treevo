import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['close'].shift(-1) - df['open'].shift(-1)
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility using High, Low, and Close prices
    df['range'] = df['high'] - df['low']
    df['volatility'] = df['range'].rolling(window=20).std()
    
    # Adaptive Window Calculation
    def adjust_window_size(volatility, low_vol_threshold, high_vol_threshold, low_win_size, high_win_size):
        if volatility < low_vol_threshold:
            return low_win_size
        elif volatility > high_vol_threshold:
            return high_win_size
        else:
            return (low_win_size + high_win_size) // 2
    
    df['window_size'] = df['volatility'].apply(adjust_window_size, 
                                              args=(df['volatility'].quantile(0.25), df['volatility'].quantile(0.75), 60, 10))
    
    # Rolling Statistics with Adaptive Window
    df['rolling_mean'] = df.groupby('window_size')['volume_weighted_return'].transform(lambda x: x.rolling(x.name).mean())
    df['rolling_std'] = df.groupby('window_size')['volume_weighted_return'].transform(lambda x: x.rolling(x.name).std())
    
    # Normalize Rolling Mean by Rolling Standard Deviation
    df['alpha_factor'] = df['rolling_mean'] / df['rolling_std']
    
    # Generate Final Alpha Factor
    alpha_factor = df['alpha_factor']
    
    return alpha_factor
