```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Volatility Calculation
    def volatility(x):
        return x.std()
    df['Volatility'] = df[['high', 'low', 'close']].rolling(window=30).apply(volatility, raw=False).mean(axis=1)
    
    # Adaptive Window Size Adjustment
    def adaptive_window_size(vol, low_vol_threshold=0.05, high_vol_threshold=0.15):
        if vol > high_vol_threshold:
            return 5
        elif vol < low_vol_threshold:
            return 60
        else:
            return 30
    
    # Apply the adaptive window size
    df['Adaptive_Window'] = df['Volatility'].apply(adaptive_window_size)
    
    # Rolling Statistics
    df['Rolling_Mean'] = df.groupby('Adaptive_Window')['Volume_Weighted_Return'].transform(lambda x: x.rolling(window=x.name, min_periods=1).mean())
    df['Rolling_Std'] = df.groupby('Adaptive_Window')['Volume_Weighted_Return'].transform(lambda x: x.rolling(window=x.name, min_periods=1).std())
    
    # Standardize the Rolling Mean
    df['Alpha_Factor'] = (df['Rolling_Mean'] - df['Rolling_Mean'].mean()) / df['Rolling_Mean'].std()

    return df['Alpha_Factor']
```
