import pandas as pd

def heuristics_v2(df):
    # Calculate Intraday Range
    df['intraday_range'] = df['high'] - df['low']
    
    # Calculate Closing Position in Intraday Range
    df['closing_position'] = (df['close'] - df['low']) / df['intraday_range']
    
    # Determine Momentum Direction and Calculate Intraday Momentum Score
    def calculate_momentum_score(row):
        if row['closing_position'] > 0.5:
            # Bullish, penalize stocks that close near the high end of the range
            return 1 - row['closing_position']
        else:
            # Bearish, reward stocks that close near the low end of the range
            return row['closing_position'] - 0.5
    
    df['momentum_score'] = df.apply(calculate_momentum_score, axis=1)
    
    # Incorporate Volume
    df['weighted_momentum_score'] = df['momentum_score'] * df['volume']
    
    # Return the weighted momentum score as a Series
    return df['weighted_momentum_score']
