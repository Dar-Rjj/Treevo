```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Volatility Calculation
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=20).std().mean(axis=1)
    
    # Adaptive Window Size Determination
    def adaptive_window_size(volatility, high_threshold, low_threshold, window_high, window_low, window_mid):
        if volatility > high_threshold:
            return window_high
        elif volatility < low_threshold:
            return window_low
        else:
            return window_mid
    
    # Define thresholds and windows
    high_vol_threshold = df['volatility'].quantile(0.75)
    low_vol_threshold = df['volatility'].quantile(0.25)
    window_high = 10
    window_low = 30
    window_mid = 20
    
    # Apply adaptive window size
    df['window_size'] = df['volatility'].apply(lambda x: adaptive_window_size(x, high_vol_threshold, low_vol_threshold, window_high, window_low, window_mid))
    
    # Rolling Statistics with Adaptive Window Size
    df['rolling_mean'] = df.groupby('window_size')['volume_weighted_return'].transform(lambda x: x.rolling(window=x.name).mean())
    df['rolling_std'] = df.groupby('window_size')['volume_weighted_return'].transform(lambda x: x.rolling(window=x.name).std())
    
    # Final factor value
    df['alpha_factor'] = df['rolling_mean'] / (df['rolling_std'] + 1e-6)  # Add a small constant to avoid division by zero
    
    return df['alpha_factor']
```
