import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility
    df['volatility'] = df[['high', 'low', 'close']].std(axis=1).rolling(window=20).mean()
    
    # Adjust Window Size Based on Volatility
    def adjust_window(vol, low_vol_threshold=0.5, high_vol_threshold=1.5, base_window=50):
        if vol < low_vol_threshold:
            return int(base_window * 1.5)  # Increase window size for low volatility
        elif vol > high_vol_threshold:
            return int(base_window * 0.5)  # Decrease window size for high volatility
        else:
            return base_window  # Keep the base window for medium volatility
    
    df['window_size'] = df['volatility'].apply(adjust_window)
    
    # Initialize the rolling statistics with adaptive window
    rolling_mean = []
    rolling_std = []
    
    for i in range(len(df)):
        window = df['window_size'].iloc[i]
        if i + window > len(df):
            break
        window_data = df['volume_weighted_return'].iloc[i:i+window]
        rolling_mean.append(window_data.mean())
        rolling_std.append(window_data.std())
    
    # Assign the rolling statistics back to the dataframe
    df['rolling_mean'] = pd.Series(rolling_mean, index=df.index[:len(rolling_mean)])
    df['rolling_std'] = pd.Series(rolling_std, index=df.index[:len(rolling_std)])
    
    # The alpha factor is the z-score of the volume weighted close-to-open return
    df['alpha_factor'] = (df['volume_weighted_return'] - df['rolling_mean']) / df['rolling_std']
    
    return df['alpha_factor'].dropna()
