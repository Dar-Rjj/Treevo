```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    volume_weighted_return = close_to_open_return * df['volume']
    
    # Adjust for Volatility
    rolling_volatility = df['close'].rolling(window=20).std()
    volatility_adjusted_return = volume_weighted_return / rolling_volatility
    
    # Incorporate Intraday Data
    intraday_range = df['high'] - df['low']
    adjusted_alpha_factor = volatility_adjusted_return * intraday_range
    
    # Incorporate Price Momentum
    moving_average_5 = df['close'].rolling(window=5).mean()
    momentum_5 = df['close'] - moving_average_5
    
    # Combine Factors
    alpha_factor = adjusted_alpha_factor + momentum_5
    
    return alpha_factor.dropna()
```
