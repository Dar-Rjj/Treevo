import pandas as pd

def heuristics_v2(df, N=10):
    # Calculate daily close-to-close return
    df['Close_Return'] = df['close'].pct_change()
    
    # Initialize AVT to 0
    avt = 0
    
    # Calculate Accumulated Volume Trend (AVT)
    avt_series = []
    for i in range(len(df)):
        if i == 0:
            avt_series.append(0)  # No previous day to compare on the first day
        else:
            volume_diff = df['volume'].iloc[i] - df['volume'].iloc[i-1]
            if df['volume'].iloc[i] > df['volume'].iloc[i-1]:
                avt += volume_diff
            else:
                avt -= volume_diff
            avt_series.append(avt)
            
    # Add AVT to the dataframe
    df['AVT'] = avt_series
    
    # Determine sign of AVT and apply weight
    df['AVT_Sign'] = df['AVT'].apply(lambda x: 1 if x > 0 else -1 if x < 0 else 0)
    
    # Combine Close-to-Close Return and AVT, weighted by the sign of AVT
    df['Alpha_Factor'] = df['Close_Return'] * df['AVT_Sign']
    
    # Return the Alpha Factor
    return df['Alpha_Factor']
