```python
import pandas as pd
import numpy as np

def heuristics_v2(df, window=20):
    # Calculate Forward Close-to-Open Return
    df['FwdCtoO'] = df['open'].shift(-1) - df['close']
    
    # Calculate Exponential Moving Average (EMA) of Volume
    df['Volume_EMA'] = df['volume'].ewm(span=window, adjust=False).mean()
    
    # Adjust Volume by Dividing it by the EMA of Volume
    df['Adjusted_Volume'] = df['volume'] / df['Volume_EMA']
    
    # Multiply by Forward Close-to-Open Return
    df['Volume_Weighted_FwdCtoO'] = df['Adjusted_Volume'] * df['FwdCtoO']
    
    # Calculate Rolling Mean and Standard Deviation
    df['Rolling_Mean'] = df['Volume_Weighted_FwdCtoO'].rolling(window=window).mean()
    df['Rolling_Std'] = df['Volume_Weighted_FwdCtoO'].rolling(window=window).std()
    
    # The final factor can be the rolling mean or a combination of rolling mean and standard deviation
    # Here we use the rolling mean for simplicity
    factor = df['Rolling_Mean']
    
    return factor
```
