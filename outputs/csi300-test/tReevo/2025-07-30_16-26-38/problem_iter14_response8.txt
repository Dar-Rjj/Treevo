```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']

    # Adaptive Window Calculation based on Volatility
    fixed_initial_window = 20
    df['volatility'] = df['close'].rolling(window=fixed_initial_window).std()
    adaptive_window = (30 - (df['volatility'] / df['volatility'].mean() * 10)).astype(int)
    adaptive_window = np.clip(adaptive_window, a_min=5, a_max=40)

    # Rolling Statistics with Adaptive Window
    df['rolling_mean'] = df['volume_weighted_return'].rolling(window=adaptive_window).mean()
    df['rolling_std'] = df['volume_weighted_return'].rolling(window=adaptive_window).std()

    # Normalize the Volume Weighted Close-to-Open Return
    df['normalized_value'] = (df['volume_weighted_return'] - df['rolling_mean']) / df['rolling_std']

    # Generate Alpha Factor
    df['alpha_factor'] = np.where(df['normalized_value'] > 0, 1, -1) * df['normalized_value'].abs()

    return df['alpha_factor']
```
