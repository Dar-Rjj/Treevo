import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Dynamic Volatility Calculation
    # High-Low Range
    df['High_Low_Range'] = df['high'] - df['low']
    
    # True Range
    df['True_Range'] = df.apply(lambda x: max(x['high'] - x['low'], 
                                               abs(x['high'] - x['close'].shift(1)), 
                                               abs(x['low'] - x['close'].shift(1))), axis=1)
    
    # Rolling Standard Deviation of True Range
    std_true_range = df['True_Range'].rolling(window=20).std()
    
    # Adjust Window Size
    df['Rolling_Window'] = np.where(std_true_range > std_true_range.rolling(window=20).mean(), 5, 30)
    
    # Incorporate Additional Market Features
    # Dollar Volume
    df['Dollar_Volume'] = df['volume'] * df['close']
    
    # Average Dollar Volume
    df['Avg_Dollar_Volume'] = df['Dollar_Volume'].rolling(window=20).mean()
    
    # Rolling Statistics with Adaptive Window
    def adaptive_rolling_stat(series, window_series):
        return series.rolling(window=window_series, min_periods=1).apply(lambda x: x.mean(), raw=True)
    
    df['Rolling_Mean_Volume_Weighted_Return'] = adaptive_rolling_stat(df['Volume_Weighted_Return'], df['Rolling_Window'])
    df['Rolling_Std_Volume_Weighted_Return'] = adaptive_rolling_stat(df['Volume_Weighted_Return'], df['Rolling_Window'])
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Rolling_Mean_Volume_Weighted_Return'] / df['Rolling_Std_Volume_Weighted_Return']
    
    return df['Alpha_Factor']
