```python
import pandas as pd
import numpy as np
from scipy.stats import linregress

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['vol_weighted_close_to_open_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility
    df['daily_range'] = df['high'] - df['low']
    df['volatility'] = df['daily_range'].rolling(window=20).std()
    
    # Adjust Window Size Based on Volatility
    def adjust_window_size(volatility):
        if volatility > df['volatility'].mean():
            return 10
        else:
            return 30
    
    df['window_size'] = df['volatility'].apply(adjust_window_size)
    
    # Rolling Statistics with Adaptive Window
    def rolling_stats(series, window_sizes):
        result = pd.DataFrame(index=series.index)
        for i, row in series.iterrows():
            window = int(window_sizes.loc[i])
            rolling_mean = series.iloc[:i+1].rolling(window=window).mean().iloc[-1]
            rolling_std = series.iloc[:i+1].rolling(window=window).std().iloc[-1]
            result.at[i, 'rolling_mean'] = rolling_mean
            result.at[i, 'rolling_std'] = rolling_std
        return result
    
    stats = rolling_stats(df['vol_weighted_close_to_open_return'], df['window_size'])
    df['rolling_mean'] = stats['rolling_mean']
    df['rolling_std'] = stats['rolling_std']
    
    # Incorporate Momentum Factors
    df['momentum_5'] = df['close'].pct_change(5)
    df['momentum_10'] = df['close'].pct_change(10)
    df['momentum_20'] = df['close'].pct_change(20)
    
    # Calculate Trend
    def calculate_trend(high, low, m):
        x = np.arange(m)
        high_slope, _, _, _, _ = linregress(x, high[-m:])
        low_slope, _, _, _, _ = linregress(x, low[-m:])
        return (high_slope + low_slope) / 2
    
    df['trend'] = df.apply(lambda row: calculate_trend(df.loc[row.name-pd.Timedelta(days=10):row.name, 'high'], 
                                                       df.loc[row.name-pd.Timedelta(days=10):row.name, 'low'], 10), axis=1)
    
    # Combine All Factors into Final Alpha Factor
    df['alpha_factor'] = (df['rolling_mean'] + df['momentum_5'] + df['momentum_10'] + df['momentum_20'] + df['trend']) / df['rolling_std']
    
    return df['alpha_factor']
```
