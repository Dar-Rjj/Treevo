```python
import pandas as pd

def heuristics_v2(df, n=10):
    """
    Calculate the Intraday Volatility Weighted by Volume.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
    n (int): Number of periods to use for average volume calculation
    
    Returns:
    pd.Series: Series with the calculated factor values
    """
    # Determine High-Low Spread
    high_low_spread = df['high'] - df['low']
    
    # Calculate Open-Close Spread
    open_close_spread = df['close'] - df['open']
    
    # Combine Price Movements
    squared_high_low = high_low_spread ** 2
    squared_open_close = open_close_spread ** 2
    sum_of_squared_spreads = squared_high_low + squared_open_close
    
    # Calculate Average Volume
    avg_volume = df['volume'].rolling(window=n).mean()
    
    # Weight Volatility by Volume
    weighted_volatility = sum_of_squared_spreads * avg_volume
    factor_values = weighted_volatility ** 0.5
    
    return factor_values
```
