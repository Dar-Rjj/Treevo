```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    """
    Calculate the Adaptive Volume-Weighted Price and Volatility Alpha Factor.
    
    Parameters:
    df (pd.DataFrame): DataFrame with columns ['open', 'high', 'low', 'close', 'amount', 'volume']
                       and index as date.
                       
    Returns:
    pd.Series: Alpha factor values indexed by date.
    """
    # Compute Close-to-Open Return
    close_to_open_return = df['open'].shift(-1) / df['close'] - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']

    # Calculate Historical Volatility
    log_returns = np.log(df['close']).diff()
    historical_volatility = log_returns.rolling(window=20).std() * np.sqrt(252)
    
    # Select Dynamic Window Size based on Volatility
    min_window, max_window = 5, 50
    window_size = (max_window - min_window) * (1 - historical_volatility / historical_volatility.max()) + min_window
    window_size = window_size.rolling(window=1).mean().fillna(method='bfill').astype(int)

    # Compute Adaptive Moving Average of Close Prices
    adaptive_ma = df['close'].rolling(window=window_size, min_periods=min_window).mean()

    # Determine Intraday High-Low Range Volatility
    intraday_volatility = df['high'] - df['low']

    # Integrate Components for Alpha Factor
    alpha_factor = (volume_weighted_return + adaptive_ma) + intraday_volatility

    return alpha_factor
```
