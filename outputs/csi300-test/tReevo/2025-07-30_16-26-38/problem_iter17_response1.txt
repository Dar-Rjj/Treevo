import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adaptive Volatility Calculation
    df['Volatility'] = df[['high', 'low', 'close']].rolling(window=20).std().mean(axis=1)
    
    # Adjust Window Size Based on Volatility
    def get_adaptive_window(vol):
        if vol > df['Volatility'].quantile(0.75):
            return 10  # Decrease window size for high volatility
        elif vol < df['Volatility'].quantile(0.25):
            return 30  # Increase window size for low volatility
        else:
            return 20  # Default window size
    
    # Rolling Statistics with Adaptive Windows
    rolling_mean = [np.nan] * len(df)
    rolling_std = [np.nan] * len(df)
    
    for i in range(20, len(df)):
        window = get_adaptive_window(df['Volatility'].iloc[i])
        if i >= window:
            rolling_mean[i] = df['Volume_Weighted_Return'].iloc[i-window:i].mean()
            rolling_std[i] = df['Volume_Weighted_Return'].iloc[i-window:i].std()
    
    df['Rolling_Mean'] = rolling_mean
    df['Rolling_Std'] = rolling_std
    
    # Market Microstructure Feature: Incorporate Bid-Ask Spread
    df['Bid_Ask_Spread'] = (df['high'] - df['low']) / 2
    df['Spread_Adjusted_Return'] = df['Volume_Weighted_Return'] / df['Bid_Ask_Spread']
    
    # Regime Change Detection
    df['Regime'] = (df['high'] - df['low']).pct_change().apply(lambda x: 1 if abs(x) > 0.05 else 0)
    
    # Adjust Alpha Factor Based on Detected Regimes
    def adjust_alpha_factor(row):
        if row['Regime'] == 1:  # Regime is changing
            return row['Rolling_Mean'] * 0.9  # Apply smoothing
        else:  # Regime is stable
            return row['Rolling_Mean']
    
    df['Alpha_Factor'] = df.apply(adjust_alpha_factor, axis=1)
    
    return df['Alpha_Factor']
