```python
import pandas as pd

def heuristics_v2(df, n=10):
    # Calculate Daily Price Momentum
    df['daily_momentum'] = df['close'] - df['close'].shift(1)
    
    # Calculate Volume-Weighted Price Momentum
    df['vol_weighted_momentum'] = df['volume'] * df['daily_momentum']
    
    # Sum of Last N Days' Volume-Weighted Price Momenta
    df['alpha_factor'] = df['vol_weighted_momentum'].rolling(window=n).sum()
    
    return df['alpha_factor'].dropna()
```
