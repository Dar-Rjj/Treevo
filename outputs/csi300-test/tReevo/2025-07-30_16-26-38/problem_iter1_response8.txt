import pandas as pd

def heuristics_v2(df, window_size=10):
    # Calculate Day's High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Calculate Previous Daysâ€™ High-to-Low Range Cumulatively
    df['cumulative_high_to_low_range'] = df['high_to_low_range'].rolling(window=window_size).sum()
    
    # Calculate Volume-Weighted Average of Open and Close Prices
    df['volume_weighted_avg_price'] = (df['open'] + df['close']) / 2
    
    # Calculate Volume-Adjusted Cumulative High-to-Low Range
    df['volume_adjusted_cumulative_range'] = df['cumulative_high_to_low_range'] * df['volume_weighted_avg_price'] * df['volume']
    
    # Normalize by Rolling Standard Deviation
    df['normalized_factor'] = df['volume_adjusted_cumulative_range'] / df['volume_adjusted_cumulative_range'].rolling(window=window_size).std()
    
    return df['normalized_factor']
