import pandas as pd

def heuristics_v2(df):
    # Calculate Dayâ€™s High-to-Low Range
    df['high_to_low_range'] = df['high'] - df['low']
    
    # Define adaptive rolling window sizes
    window_sizes = [5, 10, 20]
    
    # Calculate Adaptive Cumulative High-to-Low Range
    for window in window_sizes:
        df[f'cum_high_to_low_{window}d'] = df['high_to_low_range'].rolling(window=window).sum()
    
    # Calculate Volume-Weighted Average of Open and Close Prices
    df['volume_weighted_avg'] = (df['open'] + df['close']) / 2
    
    # Calculate Volume-Adjusted Cumulative High-to-Low Range
    for window in window_sizes:
        df[f'volume_adjusted_cum_range_{window}d'] = df[f'cum_high_to_low_{window}d'] * df['volume_weighted_avg'] * df['volume']
    
    # Calculate Rolling Standard Deviation of Daily High-to-Low Range
    for window in window_sizes:
        df[f'rolling_std_{window}d'] = df['high_to_low_range'].rolling(window=window).std()
    
    # Adjust for Volatility
    for window in window_sizes:
        df[f'volatility_adjusted_range_{window}d'] = df[f'volume_adjusted_cum_range_{window}d'] / df[f'rolling_std_{window}d']
    
    # Incorporate Recent Price Trend
    df['5_day_momentum'] = df['close'] - df['close'].shift(5)
    
    # Combine all the volatility-adjusted ranges with momentum
    df['combined_signal'] = 0
    for window in window_sizes:
        df['combined_signal'] += df[f'volatility_adjusted_range_{window}d']
    df['combined_signal'] += df['5_day_momentum']
    
    # Signal Generation
    df['signal'] = df['combined_signal'].apply(lambda x: 1 if x > 0 else -1)
    
    return df['signal']
