```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Incorporate Intraday Momentum
    df['Intraday_Momentum'] = (df['high'] - df['low']) / df['close']
    
    # Adjust for Intraday Volatility
    differences = [
        (df['high'] - df['low']),
        (df['high'] - df['close']),
        (df['low'] - df['close']),
        (df['high'] - df['open']),
        (df['low'] - df['open'])
    ]
    df['Intraday_Volatility'] = pd.concat(differences, axis=1).mean(axis=1)
    df['Adjusted_Intraday_Momentum'] = df['Intraday_Momentum'] / df['Intraday_Volatility']
    
    # Final Composite Factor
    df['Novel_Alpha_Factor'] = df['Volume_Weighted_Return'] + df['Adjusted_Intraday_Momentum']
    
    return df['Novel_Alpha_Factor']
```
