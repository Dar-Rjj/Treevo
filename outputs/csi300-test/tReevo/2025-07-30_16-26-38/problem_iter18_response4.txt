import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility
    df['hlc_volatility'] = df[['high', 'low', 'close']].std(axis=1, ddof=0)
    initial_window = 20
    rolling_std = df['hlc_volatility'].rolling(window=initial_window).std()
    
    # Adaptive Window Calculation
    def adjust_window(rolling_std, volatility):
        if volatility > rolling_std:
            return max(5, initial_window - 5)  # Decrease window size
        else:
            return min(50, initial_window + 5)  # Increase window size
    
    df['adaptive_window'] = df.apply(lambda x: adjust_window(rolling_std, x['hlc_volatility']), axis=1)
    
    # Rolling Statistics with Adaptive Window
    def rolling_stats(s, window_series):
        result = pd.Series(index=s.index, dtype='float64')
        for i in range(len(s)):
            if i < len(s) - 1:  # avoid future leakage
                window = int(window_series.iloc[i])
                result.iloc[i] = s.iloc[max(0, i-window+1):i+1].mean()
        return result
    
    df['rolling_mean'] = rolling_stats(df['volume_weighted_return'], df['adaptive_window'])
    df['rolling_std'] = rolling_stats(df['volume_weighted_return'] - df['rolling_mean'], df['adaptive_window'])
    
    # Factor Value
    df['alpha_factor'] = (df['volume_weighted_return'] - df['rolling_mean']) / df['rolling_std']
    
    return df['alpha_factor'].shift()  # shift to avoid lookahead bias
