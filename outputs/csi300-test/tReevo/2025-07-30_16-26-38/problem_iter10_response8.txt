```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the forward close-to-open return
    df['ForwardCtoOReturn'] = df['open'].shift(-1) - df['close']
    
    # Volume-weighted forward close-to-open return
    df['VolumeWeightedFwdCtoOReturn'] = df['volume'] * df['ForwardCtoOReturn']
    
    return df['VolumeWeightedFwdCtoOReturn']
```
