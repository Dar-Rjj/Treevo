import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_CtO_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Dynamic Volatility Calculation
    df['HL_C_Mean'] = (df['high'] + df['low'] + df['close']) / 3
    df['Volatility'] = df['HL_C_Mean'].rolling(window=20).std()
    upper_threshold = df['Volatility'].quantile(0.75)
    lower_threshold = df['Volatility'].quantile(0.25)
    
    # Adaptive Window Size Determination
    df['Window_Size'] = np.where(df['Volatility'] > upper_threshold, 10,
                                 np.where(df['Volatility'] < lower_threshold, 30, 20))
    
    # Momentum Factor Integration
    df['Price_Momentum'] = df['close'].rolling(window=df['Window_Size'], min_periods=1).mean()
    df['Volume_Momentum'] = df['volume'].rolling(window=df['Window_Size'], min_periods=1).mean()
    
    # Liquidity Factor Integration
    df['Avg_Daily_Volume'] = df['volume'].rolling(window=df['Window_Size'], min_periods=1).mean()
    df['Market_Cap'] = df['close'] * df['volume']
    df['Turnover_Ratio'] = df['volume'] / df['Market_Cap']
    
    # Advanced Rolling Statistics
    df['Rolling_Mean_VW_CtO_Return'] = df.groupby('Window_Size')['Volume_Weighted_CtO_Return'].transform(lambda x: x.rolling(window=x.name, min_periods=1).mean())
    df['Rolling_Std_VW_CtO_Return'] = df.groupby('Window_Size')['Volume_Weighted_CtO_Return'].transform(lambda x: x.rolling(window=x.name, min_periods=1).std())
    df['Rolling_Z_Score_VW_CtO_Return'] = (df['Volume_Weighted_CtO_Return'] - df['Rolling_Mean_VW_CtO_Return']) / df['Rolling_Std_VW_CtO_Return']
    df['Rolling_Sharpe_Ratio_VW_CtO_Return'] = df['Rolling_Mean_VW_CtO_Return'] / df['Rolling_Std_VW_CtO_Return']
    
    # Final Alpha Factor
    df['Alpha_Factor'] = df['Rolling_Sharpe_Ratio_VW_CtO_Return'] + df['Price_Momentum'] + df['Turnover_Ratio']
    
    return df['Alpha_Factor'].copy()
