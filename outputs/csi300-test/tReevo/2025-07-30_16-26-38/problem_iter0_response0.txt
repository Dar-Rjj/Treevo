```python
import pandas as pd

def heuristics_v2(df):
    """
    Calculate the Volume Weighted Close-to-Open Return.
    
    Parameters:
    - df: pandas DataFrame with (open, high, low, close, amount, volume) as columns and date as index.
    
    Returns:
    - A pandas Series indexed by date, representing the factor values.
    """
    # Calculate Close-to-Open Return
    close_to_open_return = df['close'] / df['open'].shift(-1) - 1
    
    # Weight by Volume
    volume_weighted_return = close_to_open_return * df['volume']
    
    return volume_weighted_return
```
