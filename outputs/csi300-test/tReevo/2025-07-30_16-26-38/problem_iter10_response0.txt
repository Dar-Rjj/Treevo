```python
import pandas as pd

def heuristics_v2(df):
    # Calculate the forward close-to-open return
    df['forward_c_to_o'] = df['open'].shift(-1) - df['close']
    
    # Volume weighting
    df['volume_weighted_forward_c_to_o'] = df['volume'] * df['forward_c_to_o']
    
    return df['volume_weighted_forward_c_to_o']
```
