import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Calculate Rolling Volatility (using High, Low, and Close prices)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['rolling_volatility'] = df['typical_price'].rolling(window=20).std()
    
    # Adaptive Window for Volume Weighting
    df['smoothed_volume'] = df['volume'].rolling(window=5).mean()
    df['volume_weighted_return'] = df['close_to_open_return'] * df['smoothed_volume']
    
    # Volatility Adjustment
    df['volatility_adjusted_factor'] = df['volume_weighted_return'] / df['rolling_volatility']
    
    # Drop rows with NaN values created by the rolling window calculations
    df.dropna(inplace=True)
    
    # Return the factor values as a pandas Series
    return df['volatility_adjusted_factor']
