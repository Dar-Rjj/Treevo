import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Volume_Weighted_Return'] * df['volume']
    
    # Dynamic Volatility Measures
    # Rolling Standard Deviation
    df['Rolling_STD'] = df[['high', 'low', 'close']].std(axis=1).rolling(window=20).mean()
    
    # Average True Range (ATR)
    df['TR'] = np.maximum(np.maximum(df['high'] - df['low'], abs(df['high'] - df['close'].shift(1))), 
                          abs(df['low'] - df['close'].shift(1)))
    df['ATR'] = df['TR'].rolling(window=14).mean()
    
    # Volume Trend
    df['Volume_Trend'] = df['volume'].rolling(window=30).mean()
    
    # Combine Volatility Measures
    df['Combined_Volatility'] = 0.4 * df['Rolling_STD'] + 0.4 * df['ATR'] + 0.2 * df['Volume_Trend']
    
    # Adaptive Window Calculation
    def adjust_window(x):
        if x > df['Combined_Volatility'].quantile(0.75):
            return 5  # High volatility, decrease window size
        elif x < df['Combined_Volatility'].quantile(0.25):
            return 30  # Low volatility, increase window size
        else:
            return 10  # Medium volatility, moderate window size
    
    df['Adaptive_Window'] = df['Combined_Volatility'].apply(adjust_window)
    
    # Rolling Statistics with Adaptive Window
    def rolling_stats(group):
        window = group['Adaptive_Window'].iloc[0]
        return pd.Series({
            'Rolling_Mean': group['Volume_Weighted_Return'].rolling(window=window).mean().iloc[-1],
            'Rolling_STD': group['Volume_Weighted_Return'].rolling(window=window).std().iloc[-1]
        })
    
    df_grouped = df.groupby('date').apply(rolling_stats)
    
    # The final factor is the z-score of the rolling mean
    df_grouped['Alpha_Factor'] = (df_grouped['Rolling_Mean'] - df_grouped['Rolling_Mean'].mean()) / df_grouped['Rolling_Mean'].std()
    
    return df_grouped['Alpha_Factor']
