```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Forward Close-to-Open Return
    df['forward_c_o_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_c_o_return'] = df['forward_c_o_return'] * df['volume']
    
    return df['volume_weighted_c_o_return']
```
