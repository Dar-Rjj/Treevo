import pandas as pd

def heuristics_v2(df, window_size=10):
    # Calculate Day’s High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Calculate Previous Days’ High-to-Low Range Cumulatively
    df['cumulative_high_low_range'] = df['high_low_range'].rolling(window=window_size).sum()
    
    # Calculate Volume-Weighted Average of Open and Close Prices
    df['volume_weighted_avg_price'] = (df['open'] + df['close']) / 2
    
    # Calculate Volume-Adjusted Cumulative High-to-Low Range
    df['volume_adjusted_cumulative_high_low_range'] = df['cumulative_high_low_range'] * df['volume_weighted_avg_price'] * df['volume']
    
    # Calculate Rolling Standard Deviation
    df['rolling_std'] = df['volume_adjusted_cumulative_high_low_range'].rolling(window=window_size).std()
    
    # Generate Alpha Signal
    df['alpha_signal'] = df['volume_adjusted_cumulative_high_low_range'] / df['rolling_std']
    
    # Compare with Zero
    df['alpha_signal'] = df['alpha_signal'].apply(lambda x: 1 if x > 0 else (-1 if x < 0 else 0))
    
    return df['alpha_signal']
