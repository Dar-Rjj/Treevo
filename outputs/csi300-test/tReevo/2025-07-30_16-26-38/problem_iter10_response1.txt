import pandas as pd

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    close_to_open_return = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    volume_weighted_return = close_to_open_return * df['volume']

    # Adjust for Volatility
    rolling_volatility = df['close'].rolling(window=20).std()
    volatility_adjusted_return = volume_weighted_return / rolling_volatility
    
    # Incorporate Intraday Data
    intraday_range = df['high'] - df['low']
    intraday_adjusted_return = volatility_adjusted_return * intraday_range
    
    # Incorporate Price Momentum
    five_day_moving_average = df['close'].rolling(window=5).mean()
    momentum = five_day_moving_average - df['close']
    
    # Integrate Trading Volume Ratio
    trading_volume_ratio = df['volume'] / df['volume'].shift(1)
    adjusted_return_with_volume = intraday_adjusted_return * trading_volume_ratio
    
    # Combine Factors
    alpha_factor = adjusted_return_with_volume + intraday_range + momentum + trading_volume_ratio
    
    return alpha_factor.dropna()
