import pandas as pd
import numpy as np

def heuristics_v2(df, initial_window=30, threshold=0.05):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = (df['open'].shift(-1) - df['close']) / df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Volume_Weighted_Return'] * df['volume']
    
    # Determine Volatility
    df['Daily_Range'] = df['high'] - df['low']
    df['Volatility'] = df['Daily_Range'].rolling(window=initial_window).std()
    
    # Adjust Window Size Based on Volatility
    def adjust_window(volatility):
        if volatility > threshold:
            return max(1, int(initial_window * 0.8))
        else:
            return min(len(df), int(initial_window * 1.2))
    
    dynamic_window = df['Volatility'].apply(adjust_window)
    
    # Rolling Statistics with Adaptive Window
    rolling_mean = []
    rolling_std = []
    for i in range(len(df)):
        window = dynamic_window.iloc[i]
        if i + window <= len(df):
            mean = df['Volume_Weighted_Return'].iloc[i:i+window].mean()
            std = df['Volume_Weighted_Return'].iloc[i:i+window].std()
        else:
            mean = np.nan
            std = np.nan
        rolling_mean.append(mean)
        rolling_std.append(std)
    
    df['Rolling_Mean'] = rolling_mean
    df['Rolling_Std'] = rolling_std
    
    # Normalize the factor value
    df['Factor_Value'] = (df['Volume_Weighted_Return'] - df['Rolling_Mean']) / df['Rolling_Std']
    
    return df['Factor_Value']
