```python
import pandas as pd
import numpy as np

def heuristics_v2(df, N=10, M=10):
    # Calculate Price Momentum
    ## Determine Logarithmic Returns using Close price
    df['log_returns'] = np.log(df['close']).diff()
    
    ## Sum of Logarithmic Returns Over N Days
    df['price_momentum'] = df['log_returns'].rolling(window=N).sum()
    
    # Calculate Volume Acceleration
    ## Compute Volume Changes using Percentage Change
    df['volume_change'] = df['volume'].pct_change()
    
    ## Sum of Volume Changes Over M Days
    df['volume_acceleration'] = df['volume_change'].rolling(window=M).sum()
    
    # Combine Momentum and Acceleration
    df['alpha_factor'] = df['price_momentum'] * df['volume_acceleration']
    
    return df['alpha_factor']
```
