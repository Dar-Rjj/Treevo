import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['Close_to_Open_Return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['Volume_Weighted_Return'] = df['Close_to_Open_Return'] * df['volume']
    
    # Adjust for Volatility
    df['Rolling_Volatility'] = df['close'].rolling(window=20).std()
    df['Vol_Adjusted_Return'] = df['Volume_Weighted_Return'] / df['Rolling_Volatility']
    
    # Incorporate High-Frequency Data
    df['Intraday_Range'] = df['high'] - df['low']
    df['HF_Adjusted_Return'] = df['Vol_Adjusted_Return'] * df['Intraday_Range']
    
    # Incorporate High-Low Price Ratio
    df['High_Low_Price_Ratio'] = df['high'] / df['low']
    df['Final_Factor'] = df['HF_Adjusted_Return'] * df['High_Low_Price_Ratio']
    
    # Final Alpha Factor
    return df['Final_Factor']
