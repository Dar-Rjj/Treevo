```python
import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Adjust for Volatility
    df['rolling_volatility'] = df['close'].rolling(window=20).std()
    df['vol_adjusted_return'] = df['volume_weighted_return'] / df['rolling_volatility']
    
    # Incorporate Intraday Data
    df['intraday_range'] = df['high'] - df['low']
    df['intraday_adjusted_factor'] = df['intraday_range'] * df['vol_adjusted_return']
    
    # Incorporate Price Momentum
    df['5_day_moving_avg'] = df['close'].rolling(window=5).mean()
    df['5_day_momentum'] = df['5_day_moving_avg'] - df['close']
    
    # Combine Factors
    df['alpha_factor'] = df['vol_adjusted_return'] + df['intraday_adjusted_factor'] + df['5_day_momentum']
    
    return df['alpha_factor']
```
