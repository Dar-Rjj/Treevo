def heuristics_v2(df):
    # Calculate Forward Close-to-Open Return
    df['forward_c_to_o_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_fwd_c_to_o'] = df['volume'] * df['forward_c_to_o_return']
    
    return df['volume_weighted_fwd_c_to_o']
