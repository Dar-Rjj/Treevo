```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Daily Price Momentum
    df['price_momentum'] = df['close'].diff()

    # Calculate Volume Weighted Average Price (VWAP)
    df['typical_price'] = (df['high'] + df['low'] + df['close']) / 3
    df['vwap'] = (df['typical_price'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Integrate VWAP and Momentum
    df['integrated_value'] = df['price_momentum'] * df['vwap']

    # Filter by Moving Average
    df['20_day_sma'] = df['integrated_value'].rolling(window=20).mean()
    df['signal'] = df['integrated_value'] - df['20_day_sma']

    # Generate the final factor values
    factor_values = df['signal']
    
    return factor_values
```
