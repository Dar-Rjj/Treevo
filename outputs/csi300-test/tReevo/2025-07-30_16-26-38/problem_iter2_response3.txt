```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Day’s High-to-Low Range
    df['high_low_range'] = df['high'] - df['low']
    
    # Define adaptive rolling window sizes
    windows = [5, 10, 20]
    
    # Initialize a DataFrame to store the results
    results = pd.DataFrame(index=df.index)
    
    for window in windows:
        # Calculate Previous Days’ High-to-Low Range Cumulatively
        df[f'cum_high_low_range_{window}'] = df['high_low_range'].rolling(window=window, min_periods=1).sum()
        
        # Calculate Volume-Weighted Average of Open and Close Prices
        df['volume_weighted_avg'] = (df['open'] + df['close']) / 2
        
        # Calculate Volume-Adjusted Cumulative High-to-Low Range
        df[f'vol_adj_cum_high_low_range_{window}'] = df[f'cum_high_low_range_{window}'] * df['volume_weighted_avg'] * df['volume']
        
        # Calculate Rolling Standard Deviation of Daily High-to-Low Range
        df[f'roll_std_high_low_range_{window}'] = df['high_low_range'].rolling(window=window, min_periods=1).std()
        
        # Adjust for Volatility
        df[f'vol_adjusted_signal_{window}'] = df[f'vol_adj_cum_high_low_range_{window}'] / df[f'roll_std_high_low_range_{window}']
        
        # Store the signal in the results DataFrame
        results[f'factor_{window}'] = df[f'vol_adjusted_signal_{window}']
    
    # Generate the final factor by averaging the signals from different windows
    df['final_factor'] = results.mean(axis=1)
    
    return df['final_factor']
```
