import pandas as pd
import numpy as np

def heuristics_v2(df):
    # Calculate Close-to-Open Return
    df['close_to_open_return'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['volume_weighted_return'] = df['close_to_open_return'] * df['volume']
    
    # Determine Volatility
    df['volatility'] = df[['high', 'low', 'close']].rolling(window=30).std().mean(axis=1)
    
    # Adjust Window Size based on Volatility
    def adaptive_window_size(volatility, high_vol_threshold=0.05, low_vol_threshold=0.01, high_window=20, low_window=60, mid_window=40):
        if volatility > high_vol_threshold:
            return high_window
        elif volatility < low_vol_threshold:
            return low_window
        else:
            return mid_window
    
    df['window_size'] = df['volatility'].apply(adaptive_window_size)
    
    # Rolling Statistics with Adaptive Window
    df['rolling_mean'] = df.groupby('window_size')['volume_weighted_return'].transform(lambda x: x.rolling(window=x.name).mean())
    df['rolling_std'] = df.groupby('window_size')['volume_weighted_return'].transform(lambda x: x.rolling(window=x.name).std())
    
    # Standardize the factor
    df['factor'] = (df['volume_weighted_return'] - df['rolling_mean']) / df['rolling_std']
    
    return df['factor']
