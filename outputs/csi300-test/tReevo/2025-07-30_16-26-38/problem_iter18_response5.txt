```python
import pandas as pd
import numpy as np

def heuristics_v2(data: pd.DataFrame, initial_window: int = 20, fixed_volume_window: int = 10) -> pd.Series:
    # Calculate Close-to-Open Return
    data['Close_to_Open_Return'] = data['open'].shift(-1) - data['close']
    
    # Volume Weighting
    data['Volume_Weighted_Return'] = data['Close_to_Open_Return'] * data['volume']
    
    # Adaptive Window Calculation
    # Determine Volatility
    data['True_Range'] = data[['high', 'low', 'close']].apply(lambda x: np.max(x) - np.min(x), axis=1)
    data['Volatility'] = data['True_Range'].rolling(window=initial_window).std()
    
    # Adjust Window Size
    data['Adaptive_Window'] = np.where(data['Volatility'] > data['Volatility'].mean(), initial_window - 5, initial_window + 5)
    data['Adaptive_Window'] = np.clip(data['Adaptive_Window'], 5, 30)  # Ensure window size is within a reasonable range
    
    # Rolling Statistics
    data['Rolling_Mean_VW_Return'] = data.groupby('ticker')['Volume_Weighted_Return'].transform(
        lambda x: x.rolling(window=data['Adaptive_Window'][x.index[0]], min_periods=1).mean())
    data['Rolling_Std_VW_Return'] = data.groupby('ticker')['Volume_Weighted_Return'].transform(
        lambda x: x.rolling(window=data['Adaptive_Window'][x.index[0]], min_periods=1).std())
    
    # Trading Volume Trends
    data['Rolling_Mean_Volume'] = data['volume'].rolling(window=fixed_volume_window).mean()
    data['Volume_Trend'] = data['Rolling_Mean_Volume'].pct_change()
    
    # Adjust Volume-Weighted Close-to-Open Return
    multiplier = np.where(data['Volume_Trend'] > 0, 1.2, 0.8)
    data['Adjusted_VW_Return'] = data['Rolling_Mean_VW_Return'] * multiplier
    
    return data['Adjusted_VW_Return']
```
