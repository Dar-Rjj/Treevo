```python
import pandas as pd

def heuristics_v2(df):
    # Calculate Forward Close-to-Open Return
    df['FwdCtoO'] = df['open'].shift(-1) - df['close']
    
    # Volume Weighting
    df['VolumeWeightedReturn'] = df['volume'] * df['FwdCtoO']
    
    # Calculate Daily Volatility (True Range)
    df['PrevClose'] = df['close'].shift(1)
    df['HighLow'] = df['high'] - df['low']
    df['HighPrevClose'] = (df['high'] - df['PrevClose']).abs()
    df['PrevCloseLow'] = (df['PrevClose'] - df['low']).abs()
    df['TrueRange'] = df[['HighLow', 'HighPrevClose', 'PrevCloseLow']].max(axis=1)
    
    # Simple Volatility Adjustment
    df['VolAdjMultiplier'] = 1 / (df['TrueRange'] + 1e-6)  # Avoid division by zero
    df['VolAdjMultiplier'] = df['VolAdjMultiplier'] / df['VolAdjMultiplier'].rolling(window=5).mean()  # Normalize
    
    # Final Alpha Factor
    df['AlphaFactor'] = df['VolumeWeightedReturn'] * df['VolAdjMultiplier']
    
    return df['AlphaFactor']
```
