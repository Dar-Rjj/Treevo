```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Calculate exponential smoothing for the close price
    df['exp_smooth_close'] = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the difference between today's close and the exponentially smoothed close
    df['close_diff_exp_smooth'] = df['close'] - df['exp_smooth_close']
    
    # Calculate the average volume over the past 3 days
    df['avg_volume_3d'] = df['volume'].rolling(window=3).mean()
    
    # Calculate the ratio of today's volume to the average volume over the past 3 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_3d']
    
    # Calculate the trend component by taking the difference between the current close and the simple moving average
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    df['trend'] = df['close'] - df['sma_5d']
    
    # Combine the log returns, volume ratio, and trend into a single factor
    factor = df['log_returns'] + (df['amount'] * df['vol_ratio']) - df['trend']
    
    return factor.dropna()
```
