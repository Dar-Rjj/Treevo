```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the adaptive moving average of the volume based on volatility
    rolling_volatility = df['close'].rolling(window=10).std()
    adaptive_window = (rolling_volatility / rolling_volatility.mean() * 5).apply(lambda x: max(3, int(x)))
    df['adaptive_volume_ma'] = df['volume'].rolling(window=adaptive_window).mean()
    
    # Calculate the weighted moving average of the close price over the past 10 days with more weight on recent data
    weights = [i**2 for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Calculate the ratio of today's volume to the adaptive moving average volume
    df['vol_ratio'] = df['volume'] / df['adaptive_volume_ma']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Calculate a momentum factor using the change in close price over the last 5 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Combine the close price difference, volume ratio, trend, and momentum into a single factor
    factor = df['close_diff'] + (df['amount'] * df['vol_ratio']) - df['trend'] + df['momentum_5d']
    
    return factor.dropna()
```
