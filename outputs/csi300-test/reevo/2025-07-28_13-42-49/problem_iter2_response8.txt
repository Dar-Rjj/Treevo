```python
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=10).fillna(0)
    
    # Calculate the volatility factor
    volatility = df['close'].pct_change().rolling(window=20).std().fillna(0)
    
    # Calculate the market depth factor
    avg_price = (df['high'] + df['low']) / 2
    volume_weighted_avg_price = (avg_price * df['volume']).rolling(window=10).sum() / df['volume'].rolling(window=10).sum()
    market_depth = (df['close'] - volume_weighted_avg_price) / df['close']
    
    # Combine all factors
    alpha_factor = momentum * (1 / (volatility + 1e-7)) * market_depth
    return alpha_factor
```
