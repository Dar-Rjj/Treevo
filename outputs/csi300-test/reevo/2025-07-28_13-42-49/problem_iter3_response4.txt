def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day moving average of close prices
    df['ma_10d'] = df['close'].rolling(window=10).mean()
    df['ma_30d'] = df['close'].rolling(window=30).mean()
    
    # Momentum factor: difference between 10-day and 30-day moving averages
    df['momentum'] = df['ma_10d'] - df['ma_30d']
    
    # Volatility factor: standard deviation of close prices over the past 20 days
    df['volatility'] = df['close'].rolling(window=20).std()
    
    # Volume factor: ratio of today's volume to the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['volume_factor'] = df['volume'] / df['avg_volume_5d']
    
    # Adaptive window for weighted moving average: adjust the window size based on volatility
    def adaptive_window(price_series):
        std = price_series.rolling(window=20).std().iloc[-1]
        return int(20 + (std * 100))  # Adjust window size based on the level of volatility
    
    df['wma_adaptive'] = df['close'].rolling(window=df['close'].apply(lambda x: adaptive_window(df['close']))).mean()
    
    # Trend component: difference between the current close and the adaptive weighted moving average
    df['trend'] = df['close'] - df['wma_adaptive']
    
    # Market sentiment: ratio of today's high to today's low
    df['sentiment'] = df['high'] / df['low']
    
    # Combine all factors into a single alpha factor
    factor = (df['momentum'] * 0.4) + (df['volatility'] * 0.2) + (df['volume_factor'] * 0.3) + (df['trend'] * 0.1) + (df['sentiment'] * 0.05)
    
    return factor.dropna()
