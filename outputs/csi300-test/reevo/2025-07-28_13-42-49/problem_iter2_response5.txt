```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate moving averages for close price over 5 and 10 periods
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the relative strength by dividing the two moving averages
    relative_strength = (ma_5 - ma_10) / (ma_10 + 1e-7)
    
    # Momentum factor: difference in close price compared to 5 and 10 days ago
    momentum_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    momentum_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Volume pressure factor: ratio of current volume to average volume over 5 and 10 days
    volume_pressure_5 = df['volume'] / df['volume'].rolling(window=5).mean()
    volume_pressure_10 = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # High-low range over 5 and 10 days
    range_5 = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()) / df['close']
    range_10 = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / df['close']
    
    # Combined alpha factor
    alpha_factor = (relative_strength * momentum_5 * momentum_10 * volume_pressure_5 * volume_pressure_10 * range_5 * range_10)
    
    return alpha_factor
```
