def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the weighted moving average of the close price over the past 10 days with dynamic weights
    def dynamic_weights(prices):
        return [i**2 for i in range(1, len(prices) + 1)]
    
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(dynamic_weights(prices), prices)) / sum(dynamic_weights(prices)), raw=True)
    
    # Calculate the ratio of today's volume to the average volume over the past 5 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Incorporate a multi-timeframe analysis
    df['close_diff_20d'] = df['close'].diff(periods=20)
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['wma_30d'] = df['close'].rolling(window=30).apply(lambda prices: sum(w * p for w, p in zip(dynamic_weights(prices), prices)) / sum(dynamic_weights(prices)), raw=True)
    df['trend_30d'] = df['close'] - df['wma_30d']
    
    # Combine the close price differences, volume ratios, and trends into a single factor
    factor = (df['close_diff'] + df['close_diff_20d']) + (df['amount'] * (df['vol_ratio'] + (df['volume'] / df['avg_volume_20d'])) / 2) - (df['trend'] + df['trend_30d']) / 2
    
    return factor.dropna()
