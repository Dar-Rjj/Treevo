```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the adaptive average volume using a combination of 5-day and 10-day rolling windows
    df['avg_volume_adaptive'] = (df['volume'].rolling(window=5).mean() + df['volume'].rolling(window=10).mean()) / 2
    
    # Calculate the weighted moving average of the close price over the past 10 days with dynamic weights
    weights = [i**2 for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Calculate the ratio of today's volume to the adaptive average volume
    df['vol_ratio'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Calculate the momentum as the rate of change over the past 5, 20, and 50 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    df['momentum_50d'] = df['close'].pct_change(periods=50)
    
    # Calculate the volatility as the standard deviation of the close price over the past 10, 30, and 60 days
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    df['volatility_30d'] = df['close'].rolling(window=30).std()
    df['volatility_60d'] = df['close'].rolling(window=60).std()
    
    # Calculate the liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Calculate the relative strength index (RSI) over a 14-day period
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    df['rsi'] = 100 - (100 / (1 + rs))
    
    # Calculate the money flow ratio (MFR) over a 14-day period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(), 0)
    positive_mf_avg = positive_money_flow.rolling(window=14).mean()
    negative_mf_avg = negative_money_flow.rolling(window=14).mean()
    df['mfr'] = positive_mf_avg / negative_mf_avg
    
    # Calculate the high-low spread and open-close spread
    df['hl_spread'] = (df['high'] - df['low']) / df['close']
    df['oc_spread'] = (df['open'] - df['close']) / df['close']
    
    # Combine the close price difference, volume ratio, trend, multiple momentums, volatilities, liquidity, RSI, MFR, and market sentiments into a single factor
    factor = (
        df['close_diff'] +
        (df['amount'] * df['vol_ratio']) -
        df['trend'] +
        df['momentum_5d'] +
        df['momentum_20d'] +
        df['momentum_50d'] -
        df['volatility_10d'] -
        df['volatility_30d'] -
        df['volatility_60d'] +
        df['liquidity'] +
        df['rsi'] +
        df['mfr'] +
        df['hl_spread'] +
        df['oc_spread']
    )
    
    return factor.dropna()
```
