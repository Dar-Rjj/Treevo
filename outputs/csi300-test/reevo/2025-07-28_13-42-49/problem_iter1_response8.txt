```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open prices
    delta_close_open = df['close'] - df['open']
    
    # Calculate the average volume over the past 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the 5-day average volume
    volume_ratio = df['volume'] / avg_volume_5d
    
    # Factor: (Close - Open) * (Volume Ratio)
    factor = delta_close_open * volume_ratio
    return factor
```
