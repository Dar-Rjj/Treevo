```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the short-term momentum as the 5-day return
    df['momentum_5d'] = df['close'].pct_change(5)
    
    # Calculate the medium-term momentum as the 10-day return
    df['momentum_10d'] = df['close'].pct_change(10)
    
    # Calculate the long-term momentum as the 20-day return
    df['momentum_20d'] = df['close'].pct_change(20)
    
    # Calculate the volatility as the standard deviation of log returns over the past 30 days
    df['log_returns'] = np.log(df['close']).diff()
    df['volatility_30d'] = df['log_returns'].rolling(window=30).std()
    
    # Calculate the adaptive weighted moving average of the close price over the past 10 days with dynamic weights
    def adaptive_weights(prices):
        diff = prices.diff().abs()
        total_diff = diff.sum()
        if total_diff == 0:
            return [i for i in range(1, 11)]
        else:
            return (diff / total_diff * 10).values
    
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(adaptive_weights(prices), prices)) / sum(adaptive_weights(prices)), raw=True)
    
    # Calculate the ratio of today's volume to the average volume over the past 7 days
    df['avg_volume_7d'] = df['volume'].rolling(window=7).mean()
    df['vol_ratio'] = df['volume'] / df['avg_volume_7d']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Integrate market sentiment by calculating the average amount over the past 5 days
    df['avg_amount_5d'] = df['amount'].rolling(window=5).mean()
    
    # Combine the short-term, medium-term, and long-term momentum, volatility, and trend into a single factor
    # Adjust the factor with the volume ratio to incorporate market sentiment
    factor = (df['momentum_5d'] + df['momentum_10d'] + df['momentum_20d']) / (df['volatility_30d'] + 1e-6) + (df['amount'] * df['vol_ratio']) - df['trend'] * (df['amount'] / (df['avg_amount_5d'] + 1e-6))
    
    return factor.dropna()
```
