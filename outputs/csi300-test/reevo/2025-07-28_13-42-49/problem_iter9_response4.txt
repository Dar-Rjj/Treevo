```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns of the close price
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the exponential moving average of the close price over the past 5 days
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the adaptive window for volume based on volatility
    vol_window = (df['high'] - df['low']).rolling(window=5).std() * 5
    vol_window = vol_window.fillna(5).astype(int)
    df['avg_volume_adaptive'] = df['volume'].rolling(window=vol_window, min_periods=1).mean()
    
    # Calculate the ratio of today's volume to the adaptive average volume
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the EMA
    df['trend'] = df['close'] - df['ema_5d']
    
    # Calculate the liquidity component as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Combine the log returns, volume ratio, trend, and liquidity into a single factor
    factor = df['log_returns'] + (df['vol_ratio_adaptive'] * df['liquidity']) - df['trend']
    
    return factor.dropna()
```
