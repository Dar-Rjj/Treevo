def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Adaptive window mean for volume (5 and 10 days)
    def adaptive_window_mean(series, start, end):
        return (series.rolling(window=start).mean() + series.rolling(window=end).mean()) / 2

    df['avg_volume_adaptive'] = adaptive_window_mean(df['volume'], 5, 10)
    
    # Weighted moving average of the close price over the past 10 days
    weights = [i for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Ratio of today's volume to the adaptive average volume
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # 5-day momentum
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Mean reversion: 30-day rolling mean
    df['mean_reversion_30d'] = df['close'] - df['close'].rolling(window=30).mean()
    
    # Liquidity adjustment
    df['liquidity_adjustment'] = df['amount'] / df['volume']
    
    # Dynamic realized volatility (standard deviation of log returns) over the past 20 days
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    df['realized_volatility'] = df['log_returns'].rolling(window=20).std()
    
    # Combine the close price difference, volume ratio, trend, momentum, mean reversion, liquidity adjustment, and realized volatility into a single factor
    factor = (
        df['close_diff'] 
        + (df['amount'] * df['vol_ratio_adaptive']) 
        - df['trend'] 
        + df['momentum_5d'] 
        + df['mean_reversion_30d'] 
        + df['liquidity_adjustment'] 
        - df['realized_volatility']
    )
    
    return factor.dropna()
