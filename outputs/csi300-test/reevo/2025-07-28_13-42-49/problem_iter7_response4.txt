def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over an adaptive window (e.g., 5-10 days based on recent volatility)
    df['avg_volume_adaptive'] = df['volume'].rolling(window=5 + int(5 * df['close'].rolling(window=30).std() / df['close'].std())).mean()
    
    # Calculate the weighted moving average of the close price over the past 10 days with dynamic weights
    weights = [i**2 for i in range(1, 11)]  # Dynamic weights
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Calculate the ratio of today's volume to the average volume over the adaptive window
    df['vol_ratio'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Calculate the momentum factor using the percentage change over a 5-day period
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the liquidity factor as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Combine the close price difference, volume ratio, trend, momentum, and liquidity into a single factor
    factor = df['close_diff'] + (df['amount'] * df['vol_ratio']) - df['trend'] + df['momentum_5d'] * df['liquidity']
    
    return factor.dropna()
