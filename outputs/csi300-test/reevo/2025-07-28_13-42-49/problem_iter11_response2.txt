```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the exponential moving average of the close price over the past 5 days
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the trend component by taking the difference between the current close and the exponential moving average
    df['trend'] = df['close'] - df['ema_5d']
    
    # Calculate the average volume over the past 3 days
    df['avg_volume_3d'] = df['volume'].rolling(window=3).mean()
    
    # Calculate the ratio of today's volume to the average volume over the past 3 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_3d']
    
    # Combine the log returns, volume ratio, and trend into a single factor
    factor = df['log_return'] + (df['amount'] * df['vol_ratio']) - df['trend']
    
    return factor.dropna()
```
