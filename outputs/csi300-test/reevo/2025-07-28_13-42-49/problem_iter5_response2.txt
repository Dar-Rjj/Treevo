```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum over the past 5 days and 20 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    df['momentum_20d'] = df['close'].pct_change(periods=20)
    
    # Calculate the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the weighted moving average of the close price over the past 10 days with adaptive weights
    weights = [i for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Calculate the ratio of today's volume to the average volume over the past 5 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Calculate the volatility over the past 5 days
    df['volatility_5d'] = df['close'].rolling(window=5).std()
    
    # Incorporate market sentiment using the difference between the high and low prices
    df['market_sentiment'] = df['high'] - df['low']
    
    # Combine the momentum, volume ratio, trend, and market sentiment into a single factor
    factor = (df['momentum_5d'] + df['momentum_20d']) + (df['amount'] * df['vol_ratio']) - df['trend'] - df['volatility_5d'] + df['market_sentiment']
    
    return factor.dropna()
```
