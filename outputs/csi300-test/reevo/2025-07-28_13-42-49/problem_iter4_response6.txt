```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using 10-day, 30-day, and 60-day close price returns
    df['momentum_10d'] = df['close'].pct_change(10)
    df['momentum_30d'] = df['close'].pct_change(30)
    df['momentum_60d'] = df['close'].pct_change(60)
    
    # Calculate the volatility factor using the standard deviation of the last 10 days' and 30 days' close prices
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    df['volatility_30d'] = df['close'].rolling(window=30).std()
    
    # Calculate the average volume over the past 5 days and 20 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the weighted moving average of the close price over the past 10 days and 20 days
    weights_10d = [i for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights_10d, prices)) / sum(weights_10d), raw=True)
    
    weights_20d = [i for i in range(1, 21)]
    df['wma_20d'] = df['close'].rolling(window=20).apply(lambda prices: sum(w * p for w, p in zip(weights_20d, prices)) / sum(weights_20d), raw=True)
    
    # Calculate the ratio of today's volume to the average volume over the past 5 days and 20 days
    df['vol_ratio_5d'] = df['volume'] / df['avg_volume_5d']
    df['vol_ratio_20d'] = df['volume'] / df['avg_volume_20d']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving averages
    df['trend_10d'] = df['close'] - df['wma_10d']
    df['trend_20d'] = df['close'] - df['wma_20d']
    
    # Integrate market sentiment by calculating the high-low spread and open-close spread
    df['hl_spread'] = (df['high'] - df['low']) / df['close']
    df['oc_spread'] = (df['open'] - df['close']) / df['close']
    
    # Combine the momentum, volatility, volume ratios, trends, and market sentiments into a single factor
    factor = (
        (df['momentum_10d'] + df['momentum_30d'] + df['momentum_60d']) -
        (df['volatility_10d'] + df['volatility_30d'] + df['trend_10d'] + df['trend_20d']) +
        (df['amount'] * df['vol_ratio_5d']) + (df['amount'] * df['vol_ratio_20d']) +
        df['hl_spread'] + df['oc_spread']
    )
    
    return factor.dropna()
```
