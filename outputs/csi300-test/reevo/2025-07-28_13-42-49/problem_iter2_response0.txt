def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the percentage change in close price over 14 days
    df['momentum_14d'] = df['close'].pct_change(periods=14).fillna(0)
    
    # Calculate the average true range (volatility) over 14 days
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x) - min(x), axis=1)
    df['atr_14d'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the volume-weighted close price change over 5 days
    df['vw_close_diff'] = (df['close'].diff() * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # Combine the momentum, average true range, and volume-weighted close price change into a single factor
    factor = df['momentum_14d'] + (df['amount'] / df['volume']) - (df['close'] / df['atr_14d']) + df['vw_close_diff']
    
    return factor.dropna()
