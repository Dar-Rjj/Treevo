```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum over the past 60 days using log returns
    df['log_close'] = np.log(df['close'])
    df['momentum_60d'] = df['log_close'].pct_change(periods=60)
    
    # Calculate the volatility as the standard deviation of the log returns over the past 90 days
    df['log_returns'] = df['log_close'].diff()
    df['volatility_90d'] = df['log_returns'].rolling(window=90).std()
    
    # Calculate the average volume over the past 10 days
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the exponentially weighted moving average of the close price over the past 30 days
    df['ewma_30d'] = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the ratio of today's volume to the average volume over the past 10 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_10d']
    
    # Calculate the trend component by taking the difference between the current close and the exponentially weighted moving average
    df['trend'] = df['close'] - df['ewma_30d']
    
    # Combine the momentum, volatility, and trend into a single factor
    df['factor'] = df['momentum_60d'] / df['volatility_90d'] + (df['amount'] * df['vol_ratio']) - df['trend']
    
    return df['factor'].dropna()
```
