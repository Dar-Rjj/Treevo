```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low prices, indicating price range
    price_range = df['high'] - df['low']
    
    # Calculate the average of open and close prices, representing a central tendency
    average_price = (df['open'] + df['close']) / 2
    
    # Calculate the difference between the average price and the low price, normalized by the price range
    # This factor represents how the central price is positioned relative to the day's low and high
    alpha_factor = (average_price - df['low']) / (price_range + 1e-7)
    
    return alpha_factor
```
