```python
import numpy as np
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the average volume over an adaptive window (e.g., 3, 5, 10 days)
    df['avg_volume_3d'] = df['volume'].rolling(window=3).mean()
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    df['avg_volume_10d'] = df['volume'].rolling(window=10).mean()
    
    # Calculate the simple moving average of the close price over an adaptive window
    df['sma_3d'] = df['close'].rolling(window=3).mean()
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    df['sma_10d'] = df['close'].rolling(window=10).mean()
    
    # Calculate the ratio of today's volume to the average volume over the past 3 and 5 days
    df['vol_ratio_3d'] = df['volume'] / df['avg_volume_3d']
    df['vol_ratio_5d'] = df['volume'] / df['avg_volume_5d']
    
    # Calculate the trend component by taking the difference between the current close and the simple moving averages
    df['trend_3d'] = df['close'] - df['sma_3d']
    df['trend_5d'] = df['close'] - df['sma_5d']
    df['trend_10d'] = df['close'] - df['sma_10d']
    
    # Calculate the momentum as the rate of change over the past 3 and 5 days
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the volatility as the standard deviation of the close price over the past 3 and 5 days
    df['volatility_3d'] = df['close'].rolling(window=3).std()
    df['volatility_5d'] = df['close'].rolling(window=5).std()
    
    # Calculate the liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Dynamic weights for combining key indicators
    df['weight_3d'] = (df['avg_volume_3d'] / df['avg_volume_5d']).clip(lower=0.5, upper=2.0)
    df['weight_5d'] = (df['avg_volume_5d'] / df['avg_volume_10d']).clip(lower=0.5, upper=2.0)
    
    # Market sentiment or external economic indicators (assuming we have a sentiment score column)
    if 'sentiment_score' in df.columns:
        df['sentiment_factor'] = (df['sentiment_score'] - df['sentiment_score'].mean()) / df['sentiment_score'].std()
    else:
        df['sentiment_factor'] = 0.0
    
    # Combine the key indicators into a single factor with dynamic weights
    factor = (
        df['log_returns'] +
        (df['amount'] * df['vol_ratio_3d'] * df['weight_3d']) -
        df['trend_3d'] * df['weight_3d'] +
        df['momentum_3d'] * df['weight_3d'] -
        df['volatility_3d'] * df['weight_3d'] +
        df['liquidity'] * df['weight_3d'] +
        0.5 * df['sentiment_factor']
    )
    
    return factor.dropna()
```
