```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over the past 3 days
    df['avg_volume_3d'] = df['volume'].rolling(window=3).mean()
    
    # Calculate the simple moving average of the close price over the past 5 days
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the average volume over the past 3 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_3d']
    
    # Calculate the trend component by taking the difference between the current close and the simple moving average
    df['trend'] = df['close'] - df['sma_5d']
    
    # Calculate the momentum as the percentage change in close price over the past 5 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the mean reversion factor as the difference between the current close and the simple moving average
    df['mean_reversion'] = df['sma_5d'] - df['close']
    
    # Detect volume spikes by comparing today's volume to the average volume over the past 3 days
    df['vol_spike'] = (df['volume'] > 1.5 * df['avg_volume_3d']).astype(int)
    
    # Combine the close price difference, volume ratio, trend, momentum, mean reversion, and volume spike into a single factor
    factor = (
        df['close_diff'] + 
        (df['amount'] * df['vol_ratio']) - 
        df['trend'] + 
        df['momentum_5d'] + 
        df['mean_reversion'] + 
        df['vol_spike']
    )
    
    return factor.dropna()
```
