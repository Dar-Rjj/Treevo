```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the difference between today's close and the close 10 days ago
    df['momentum_10d'] = df['close'].diff(10)
    
    # Calculate the exponential moving average of the volume over the past 5 days
    df['ema_volume_5d'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the smoothed price change as the difference between the EMA of close prices over the last 10 days and the EMA over the last 20 days
    df['ema_close_10d'] = df['close'].ewm(span=10, adjust=False).mean()
    df['ema_close_20d'] = df['close'].ewm(span=20, adjust=False).mean()
    df['smoothed_price_change'] = df['ema_close_10d'] - df['ema_close_20d']
    
    # Combine the momentum, normalized amount by EMA volume, and smoothed price change into a single factor
    factor = df['momentum_10d'] + (df['amount'] / df['ema_volume_5d']) + df['smoothed_price_change']
    
    return factor.dropna()
```
