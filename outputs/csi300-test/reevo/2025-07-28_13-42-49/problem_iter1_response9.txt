```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average of the absolute high-low range over the last 5 days
    df['avg_range_5d'] = df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()
    
    # Alpha factor: (today's close - yesterday's close) / 5-day average range
    alpha_factor = df['close_diff'] / (df['avg_range_5d'] + 1e-7)
    
    return alpha_factor
```
