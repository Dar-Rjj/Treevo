```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over the past 3 days
    df['avg_volume_3d'] = df['volume'].rolling(window=3).mean()
    
    # Calculate the simple moving average of the close price over the past 10 days
    df['sma_10d'] = df['close'].rolling(window=10).mean()
    
    # Calculate the ratio of today's volume to the average volume over the past 3 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_3d']
    
    # Calculate the trend component by taking the difference between the current close and the simple moving average
    df['trend'] = df['close'] - df['sma_10d']
    
    # Calculate the momentum as the rate of change over the past 5 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the volatility as the standard deviation of the close price over the past 10 days
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Calculate the liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Calculate the average true range (ATR) over the past 14 days
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    df['atr_14d'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the relative strength index (RSI) over the past 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))
    
    # Combine the key indicators into a single factor
    factor = (
        df['close_diff'] +
        (df['amount'] * df['vol_ratio']) -
        df['trend'] +
        df['momentum_5d'] -
        df['volatility_10d'] +
        df['liquidity'] -
        df['atr_14d'] +
        (50 - df['rsi_14d'])
    )
    
    return factor.dropna()
```
