```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:

    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Adaptive window for average volume, using exponentially weighted moving average (EWMA)
    df['ewma_volume_3d'] = df['volume'].ewm(span=3, adjust=False).mean()
    df['ewma_volume_5d'] = df['volume'].ewm(span=5, adjust=False).mean()
    df['avg_volume_adaptive'] = 0.7 * df['ewma_volume_3d'] + 0.3 * df['ewma_volume_5d']
    
    # Exponentially weighted moving average of the close price over the past 5 days
    df['ewma_close_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Volume ratio with adaptive window
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Trend component
    df['trend'] = df['close'] - df['ewma_close_5d']
    
    # Momentum as the rate of change over the past 3 days
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Volatility as the exponentially weighted standard deviation of the close price over the past 5 days
    df['volatility_5d_ewm'] = df['close'].ewm(span=5, adjust=False).std()
    
    # Liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Market sentiment (using the difference between high and low prices)
    df['market_sentiment'] = (df['high'] - df['low']) / df['close']
    
    # Mean reversion signal (using the difference between the close price and its 5-day EWMA, normalized by the 5-day EWMA volatility)
    df['mean_reversion'] = (df['close'] - df['ewma_close_5d']) / df['volatility_5d_ewm']
    
    # Relative Strength Index (RSI) with a 14-day period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))
    
    # Dynamic weights based on recent performance
    recent_performance = df['log_returns'].rolling(window=21).sum()
    dynamic_weights = (recent_performance - recent_performance.min()) / (recent_performance.max() - recent_performance.min())
    
    # Combine the key indicators into a single factor
    factor = (
        dynamic_weights * 0.25 * df['log_returns'] +
        (1 - dynamic_weights) * 0.05 * (df['amount'] * df['vol_ratio_adaptive']) -
        dynamic_weights * 0.10 * df['trend'] +
        (1 - dynamic_weights) * 0.15 * df['momentum_3d'] -
        dynamic_weights * 0.20 * df['volatility_5d_ewm'] +
        (1 - dynamic_weights) * 0.10 * df['liquidity'] +
        dynamic_weights * 0.05 * df['market_sentiment'] -
        (1 - dynamic_weights) * 0.05 * df['mean_reversion'] +
        dynamic_weights * 0.05 * (50 - df['rsi_14d'])
    )
    
    return factor.dropna()
```
