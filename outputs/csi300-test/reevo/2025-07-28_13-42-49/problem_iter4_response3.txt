```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily return
    df['daily_return'] = df['close'].pct_change()
    
    # Calculate the exponential moving average for close price over 5 and 10 periods
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the relative strength by dividing the two EMAs
    relative_strength = (ema_5 - ema_10) / (ema_10 + 1e-7)
    
    # Momentum factor: difference in close price compared to 5 and 10 days ago
    momentum_5 = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    momentum_10 = (df['close'] - df['close'].shift(10)) / df['close'].shift(10)
    
    # Volume pressure factor: ratio of current volume to average volume over 5 and 10 days
    volume_pressure_5 = df['volume'] / df['volume'].rolling(window=5).mean()
    volume_pressure_10 = df['volume'] / df['volume'].rolling(window=10).mean()
    
    # High-low range over 5 and 10 days
    range_5 = (df['high'].rolling(window=5).max() - df['low'].rolling(window=5).min()) / df['close']
    range_10 = (df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()) / df['close']
    
    # Volatility: standard deviation of daily returns over 5 and 10 days
    volatility_5 = df['daily_return'].rolling(window=5).std()
    volatility_10 = df['daily_return'].rolling(window=10).std()
    
    # Market cap factor: assuming 'amount' is the market cap
    market_cap_factor = df['amount'] / df['amount'].rolling(window=5).mean()
    
    # Combined alpha factor
    alpha_factor = (relative_strength * momentum_5 * momentum_10 * volume_pressure_5 * volume_pressure_10 * range_5 * range_10 * volatility_5 * volatility_10 * market_cap_factor)
    
    return alpha_factor.dropna()
```
