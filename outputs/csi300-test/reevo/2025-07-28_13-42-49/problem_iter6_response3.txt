```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the dynamic price change as the 5-day and 10-day percentage change in the close price
    df['dynamic_price_change_5d'] = df['close'].pct_change(5)
    df['dynamic_price_change_10d'] = df['close'].pct_change(10)
    
    # Calculate the volume spike as the ratio of today's volume to the average volume over the past 20 days
    df['avg_volume_20d'] = df['volume'].rolling(window=20).mean()
    df['volume_spike'] = df['volume'] / df['avg_volume_20d']
    
    # Calculate the liquidity adjustment using the amount and volume
    df['liquidity_adjustment'] = df['amount'] / df['volume']
    
    # Calculate the trend component by taking the difference between the current close and the simple moving average over the past 20 days
    df['sma_20d'] = df['close'].rolling(window=20).mean()
    df['trend'] = df['close'] - df['sma_20d']
    
    # Calculate the momentum factor using the 10-day, 30-day, and 60-day close price returns
    df['momentum_10d'] = df['close'].pct_change(10)
    df['momentum_30d'] = df['close'].pct_change(30)
    df['momentum_60d'] = df['close'].pct_change(60)
    
    # Combine the dynamic price changes, volume spike, liquidity adjustment, trend, and momentum into a single factor
    factor = (
        (df['dynamic_price_change_5d'] + df['dynamic_price_change_10d']) * df['volume_spike'] +
        df['liquidity_adjustment'] - df['trend'] +
        (df['momentum_10d'] + df['momentum_30d'] + df['momentum_60d'])
    )
    
    return factor.dropna()
```
