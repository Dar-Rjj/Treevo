```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using 10-day and 30-day close price returns
    df['momentum_10d'] = df['close'].pct_change(10)
    df['momentum_30d'] = df['close'].pct_change(30)

    # Calculate the volatility factor using the standard deviation of the last 10 days' close prices
    df['volatility_10d'] = df['close'].rolling(window=10).std()

    # Calculate the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()

    # Calculate the weighted moving average of the close price over the past 10 days
    weights = [i for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)

    # Calculate the ratio of today's volume to the average volume over the past 5 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_5d']

    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']

    # Integrate market sentiment by calculating the high-low spread
    df['hl_spread'] = (df['high'] - df['low']) / df['close']

    # Calculate the percentage change in the amount over 10 days
    df['amount_pct_change_10d'] = df['amount'].pct_change(10)

    # Calculate the volume-weighted close price change over 5 days
    df['vw_close_diff'] = (df['close'].diff() * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Combine the momentum, volatility, volume ratio, trend, market sentiment, amount percentage change, and volume-weighted close price change into a single factor
    factor = (df['momentum_10d'] + df['momentum_30d']) - df['volatility_10d'] + df['vol_ratio'] + df['trend'] + df['hl_spread'] + df['amount_pct_change_10d'] + df['vw_close_diff']

    return factor.dropna()
```
