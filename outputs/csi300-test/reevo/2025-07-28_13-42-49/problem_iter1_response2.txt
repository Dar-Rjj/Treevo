def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price, then scale it by the average of high and low prices
    diff_close_open = df['close'] - df['open']
    avg_high_low = (df['high'] + df['low']) / 2
    alpha_factor = diff_close_open / (avg_high_low + 1e-7)
    
    # Weight the alpha factor by the logarithm of the volume to incorporate trading volume impact
    log_volume = np.log(df['volume'] + 1e-7)
    weighted_alpha_factor = alpha_factor * log_volume
    
    return weighted_alpha_factor
