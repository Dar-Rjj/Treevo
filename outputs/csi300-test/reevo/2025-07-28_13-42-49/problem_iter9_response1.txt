```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log return of close prices
    df['log_return'] = np.log(df['close']).diff()
    
    # Calculate the exponential moving average (EMA) of volume over 5 days
    df['ema_volume_5d'] = df['volume'].ewm(span=5, adjust=False).mean()
    
    # Calculate the weighted moving average (WMA) of the close price over the past 10 days with dynamic weights
    weights = np.arange(1, 11)
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Calculate the ratio of today's volume to the EMA volume over the past 5 days
    df['vol_ratio'] = df['volume'] / df['ema_volume_5d']
    
    # Calculate the trend component by taking the difference between the current close and the WMA
    df['trend'] = df['close'] - df['wma_10d']
    
    # Combine the log return, volume ratio, and trend into a single factor
    factor = df['log_return'] + (df['amount'] * df['vol_ratio']) - df['trend']
    
    return factor.dropna()
```
