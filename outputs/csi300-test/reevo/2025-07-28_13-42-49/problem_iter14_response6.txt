```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_returns'] = np.log(df['close']) - np.log(df['close'].shift(1))
    
    # Calculate the average volume over an adaptive window (last 3 to 5 days)
    df['avg_volume_adaptive'] = df['volume'].rolling(window=3).mean() + df['volume'].rolling(window=5).mean()
    df['avg_volume_adaptive'] /= 2
    
    # Calculate the simple moving average of the close price over the past 5 days
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    
    # Calculate the exponential moving average of the close price over the past 5 days
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the ratio of today's volume to the average adaptive volume
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the exponential moving average
    df['trend'] = df['close'] - df['ema_5d']
    
    # Calculate the momentum as the rate of change over the past 3 days
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Calculate the volatility as the standard deviation of the log returns over the past 5 days
    df['volatility_5d'] = df['log_returns'].rolling(window=5).std()
    
    # Calculate the liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Integrate market sentiment by calculating the high-low spread
    df['hl_spread'] = (df['high'] - df['low']) / df['close']
    
    # Combine the key indicators into a single factor with dynamic weights
    factor = (
        0.3 * df['log_returns'] +
        0.2 * (df['amount'] * df['vol_ratio_adaptive']) -
        0.1 * df['trend'] +
        0.2 * df['momentum_3d'] -
        0.1 * df['volatility_5d'] +
        0.1 * df['liquidity'] +
        0.05 * df['hl_spread']
    )
    
    return factor.dropna()
```
