```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns of the close price
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate the exponential moving average of the log returns over the past 3 days
    df['ema_log_returns_3d'] = df['log_returns'].ewm(span=3, adjust=False).mean()
    
    # Calculate the adaptive window for volume based on the volatility of the last 5 days
    vol_window = 5
    df['volatility'] = df['log_returns'].rolling(window=vol_window).std()
    df['adaptive_vol_window'] = (df['volatility'] * vol_window).astype(int)
    df['avg_volume_adaptive'] = df['volume'].rolling(window=df['adaptive_vol_window']).mean()
    
    # Calculate the ratio of today's volume to the adaptive average volume
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the exponential moving average of the close price
    df['sma_close_5d'] = df['close'].rolling(window=5).mean()
    df['trend'] = df['close'] - df['sma_close_5d']
    
    # Combine the log returns, volume ratio, and trend into a single factor
    factor = df['ema_log_returns_3d'] + (df['amount'] * df['vol_ratio_adaptive']) - df['trend']
    
    return factor.dropna()
```
