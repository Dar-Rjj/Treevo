```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Dynamic window for average volume, based on the last 3 to 5 days
    dynamic_window = df['volume'].rolling(window=3).mean() * 0.5 + df['volume'].rolling(window=5).mean() * 0.5
    df['avg_volume_dynamic'] = dynamic_window
    
    # Simple moving average of the close price over the past 5 days
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    
    # Volume ratio with dynamic window
    df['vol_ratio_dynamic'] = df['volume'] / df['avg_volume_dynamic']
    
    # Trend component
    df['trend'] = df['close'] - df['sma_5d']
    
    # Momentum as the rate of change over the past 3 days
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Volatility as the standard deviation of the close price over the past 5 days
    df['volatility_5d'] = df['close'].rolling(window=5).std()
    
    # Liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Market sentiment (using the difference between high and low prices)
    df['market_sentiment'] = (df['high'] - df['low']) / df['close']
    
    # Mean reversion signal (using the difference between the close price and its 5-day average, normalized by the 5-day volatility)
    df['mean_reversion'] = (df['close'] - df['sma_5d']) / df['volatility_5d']
    
    # Combine the key indicators into a single factor
    factor = (
        df['log_returns'] +
        (df['amount'] * df['vol_ratio_dynamic']) -
        df['trend'] +
        df['momentum_3d'] -
        df['volatility_5d'] +
        df['liquidity'] +
        df['market_sentiment'] -
        df['mean_reversion']
    )
    
    return factor.dropna()
```
