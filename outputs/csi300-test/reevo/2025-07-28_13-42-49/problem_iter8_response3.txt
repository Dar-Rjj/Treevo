```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the adaptive average volume over a 5 to 10 day window
    def adaptive_window_mean(series, start, end):
        return (series.rolling(window=start).mean() + series.rolling(window=end).mean()) / 2

    df['avg_volume_adaptive'] = adaptive_window_mean(df['volume'], 5, 10)
    
    # Calculate the weighted moving average of the close price over the past 10 days
    weights = [i for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Calculate the ratio of today's volume to the adaptive average volume
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Incorporate short-term momentum by calculating the 5-day return
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the average amount over an adaptive window (e.g., 5 to 10 days)
    df['avg_amount_adaptive'] = adaptive_window_mean(df['amount'], 5, 10)
    
    # Calculate the ratio of today's amount to the adaptive average amount
    df['amount_ratio_adaptive'] = df['amount'] / df['avg_amount_adaptive']
    
    # Combine the close price difference, volume ratio, trend, momentum, and amount ratio into a single factor
    factor = df['close_diff'] + (df['amount'] * df['vol_ratio_adaptive']) - df['trend'] + df['momentum_5d'] + df['amount_ratio_adaptive']
    
    return factor.dropna()
```
