def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the average volume over the past 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the weighted moving average of the close price over the past 10 days
    weights = [i for i in range(1, 11)]
    df['wma_10d'] = df['close'].rolling(window=10).apply(lambda prices: sum(w * p for w, p in zip(weights, prices)) / sum(weights), raw=True)
    
    # Calculate the ratio of today's volume to the average volume over the past 5 days
    df['vol_ratio'] = df['volume'] / df['avg_volume_5d']
    
    # Calculate the trend component by taking the difference between the current close and the weighted moving average
    df['trend'] = df['close'] - df['wma_10d']
    
    # Calculate the dynamic volatility as the standard deviation of the close price difference over the past 10 days
    df['dynamic_volatility'] = df['close_diff'].rolling(window=10).std()
    
    # Integrate market sentiment or macroeconomic data (assuming a 'sentiment' column is available)
    df['sentiment_factor'] = df['sentiment'] * df['vol_ratio']
    
    # Combine the close price difference, volume ratio, trend, dynamic volatility, and sentiment factor into a single factor
    factor = df['close_diff'] + (df['amount'] * df['vol_ratio']) - df['trend'] - df['dynamic_volatility'] + df['sentiment_factor']
    
    return factor.dropna()
