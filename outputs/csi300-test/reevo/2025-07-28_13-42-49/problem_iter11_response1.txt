```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_return'] = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate exponential moving average (EMA) of the close price over the past 5 days
    df['ema_5d'] = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the trend component by taking the difference between the current close and the EMA
    df['trend'] = df['close'] - df['ema_5d']
    
    # Calculate the volume ratio using a dynamic window
    df['avg_volume_dyn'] = df['volume'].rolling(window=int(len(df) * 0.05)).mean()
    df['vol_ratio'] = df['volume'] / df['avg_volume_dyn']
    
    # Combine the log return, volume ratio, and trend into a single factor
    factor = df['log_return'] + (df['amount'] * df['vol_ratio']) - df['trend']
    
    return factor.dropna()
```
