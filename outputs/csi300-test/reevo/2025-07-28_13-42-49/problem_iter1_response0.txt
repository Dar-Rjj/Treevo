def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open, indicating the price movement over the day
    daily_price_movement = df['close'] - df['open']
    
    # Calculate the trading range of the stock for the day
    trading_range = df['high'] - df['low']
    
    # Compute the ratio of the daily price movement to the trading range, capturing how much of the
    # total possible price movement (the range from low to high) actually occurred. This ratio may
    # indicate days where the closing price significantly changes relative to the day's volatility.
    daily_movement_ratio = daily_price_movement / (trading_range + 1e-7)
    
    # Calculate a smoothed version of the volume, which might be useful in identifying trends
    smoothed_volume = df['volume'].rolling(window=5).mean()
    
    # The final factor multiplies the daily price movement/trading range ratio by the smoothed
    # volume, under the heuristic that large volume on big relative moves could indicate more
    # significant trend changes or market belief shifts.
    alpha_factor = daily_movement_ratio * smoothed_volume
    return alpha_factor
