```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the moving average of the close price over a 10-day window
    df['sma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the weighted sum of the close price and volume, with more weight on the close price
    df['weighted_sum'] = (df['close'] * 0.7 + df['volume'] * 0.3)
    
    # Generate the alpha factor by normalizing the close price difference by the weighted sum
    alpha_factor = df['close_diff'] / (df['weighted_sum'] + 1e-7)
    
    return alpha_factor
```
