```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate moving averages for close price over 5 and 10 periods
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the relative strength by dividing the two moving averages
    relative_strength = (ma_5 - ma_10) / (ma_10 + 1e-7)
    
    # Momentum factor: difference in close price compared to 5 days ago
    momentum = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Volume pressure factor: ratio of current volume to average volume over 5 days
    volume_pressure = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Volatility factor: standard deviation of the last 5 days' closing prices
    volatility = df['close'].rolling(window=5).std()
    
    # Market sentiment factor: ratio of the amount traded to the volume
    market_sentiment = df['amount'] / df['volume']
    
    # Combined alpha factor
    alpha_factor = relative_strength * momentum * volume_pressure * (1 / (volatility + 1e-7)) * market_sentiment
    
    return alpha_factor
```
