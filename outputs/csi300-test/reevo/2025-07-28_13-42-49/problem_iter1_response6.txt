def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the money flow ratio
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()
    
    # Calculate the normalized price change
    price_change = df['close'].pct_change().fillna(0)
    normalized_price_change = (price_change - price_change.min()) / (price_change.max() - price_change.min() + 1e-7)
    
    # Combine both factors
    alpha_factor = (money_ratio - 1) * normalized_price_change
    return alpha_factor
