```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the adaptive average volume over the past 3 to 5 days
    df['avg_volume_adaptive'] = (df['volume'].rolling(window=3).mean() + df['volume'].rolling(window=5).mean()) / 2
    
    # Calculate the simple moving average of the close price over the past 10 days
    df['sma_10d'] = df['close'].rolling(window=10).mean()
    
    # Calculate the ratio of today's volume to the adaptive average volume
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the simple moving average
    df['trend'] = df['close'] - df['sma_10d']
    
    # Calculate the momentum as the rate of change over the past 5 days
    df['momentum_5d'] = df['close'].pct_change(periods=5)
    
    # Calculate the volatility as the standard deviation of the close price over the past 14 days
    df['volatility_14d'] = df['close'].rolling(window=14).std()
    
    # Calculate the liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Calculate the average true range (ATR) over the past 14 days
    df['tr'] = df[['high', 'low', 'close']].apply(lambda x: max(x[0], x[1]) - min(x[0], x[1]), axis=1)
    df['atr_14d'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the relative strength index (RSI) over the past 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    df['rsi_14d'] = 100 - (100 / (1 + rs))
    
    # Adaptive weights based on the magnitude of the indicators
    def adaptive_weight(x):
        return 1 / (1 + np.exp(-x))
    
    # Calculate the adaptive weights for each indicator
    df['weight_close_diff'] = adaptive_weight(np.abs(df['close_diff']))
    df['weight_vol_ratio'] = adaptive_weight(df['vol_ratio_adaptive'])
    df['weight_trend'] = adaptive_weight(np.abs(df['trend']))
    df['weight_momentum'] = adaptive_weight(np.abs(df['momentum_5d']))
    df['weight_volatility'] = adaptive_weight(df['volatility_14d'])
    df['weight_liquidity'] = adaptive_weight(df['liquidity'])
    df['weight_atr'] = adaptive_weight(df['atr_14d'])
    df['weight_rsi'] = adaptive_weight(50 - df['rsi_14d'])
    
    # Combine the key indicators into a single factor with adaptive weights
    factor = (
        df['weight_close_diff'] * df['close_diff'] +
        df['weight_vol_ratio'] * (df['amount'] * df['vol_ratio_adaptive']) -
        df['weight_trend'] * df['trend'] +
        df['weight_momentum'] * df['momentum_5d'] -
        df['weight_volatility'] * df['volatility_14d'] +
        df['weight_liquidity'] * df['liquidity'] -
        df['weight_atr'] * df['atr_14d'] +
        df['weight_rsi'] * (50 - df['rsi_14d'])
    )
    
    return factor.dropna()
```
