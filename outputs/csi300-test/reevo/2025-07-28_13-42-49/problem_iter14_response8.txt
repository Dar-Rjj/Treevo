```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate log returns
    df['log_returns'] = np.log(df['close']).diff()
    
    # Calculate the adaptive average volume over a dynamic window (last 3 to 7 days)
    avg_volume_window = (df['volume'].rolling(window=3).mean() + df['volume'].rolling(window=5).mean() + df['volume'].rolling(window=7).mean()) / 3
    df['avg_volume_adaptive'] = avg_volume_window
    
    # Calculate the simple moving average of the close price over the past 5 days
    df['sma_5d'] = df['close'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the adaptive average volume
    df['vol_ratio_adaptive'] = df['volume'] / df['avg_volume_adaptive']
    
    # Calculate the trend component by taking the difference between the current close and the simple moving average
    df['trend'] = df['close'] - df['sma_5d']
    
    # Calculate the momentum as the rate of change over the past 3 days
    df['momentum_3d'] = df['close'].pct_change(periods=3)
    
    # Calculate the volatility as the standard deviation of the close price over the past 5 days
    df['volatility_5d'] = df['close'].rolling(window=5).std()
    
    # Calculate the liquidity as the ratio of amount to volume
    df['liquidity'] = df['amount'] / df['volume']
    
    # Dynamic weights for the key indicators
    weight_log_returns = 0.3
    weight_vol_ratio = 0.2
    weight_trend = -0.1
    weight_momentum = 0.2
    weight_volatility = -0.1
    weight_liquidity = 0.3
    
    # Combine the key indicators into a single factor with dynamic weights
    factor = (
        weight_log_returns * df['log_returns'] +
        weight_vol_ratio * (df['amount'] * df['vol_ratio_adaptive']) +
        weight_trend * df['trend'] +
        weight_momentum * df['momentum_3d'] +
        weight_volatility * df['volatility_5d'] +
        weight_liquidity * df['liquidity']
    )
    
    return factor.dropna()
```
