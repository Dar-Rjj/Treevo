```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low to understand the daily range
    daily_range = df['high'] - df['low']
    
    # Calculate the percentage of daily range that the closing price covers, which indicates the strength of the day's move
    close_percentage_of_range = (df['close'] - df['low']) / (daily_range + 1e-7)
    
    # Calculate a weighted moving average for the amount and volume over the last 5 days
    df['amount_wma_5'] = df['amount'].rolling(window=5).mean()
    df['volume_wma_5'] = df['volume'].rolling(window=5).mean()
    
    # Calculate a weighted moving average for the amount and volume over the last 20 days
    df['amount_wma_20'] = df['amount'].rolling(window=20).mean()
    df['volume_wma_20'] = df['volume'].rolling(window=20).mean()
    
    # Construct an alpha factor based on the ratio of the current day's amount and volume to their respective 5-day WMA
    amount_ratio_5 = df['amount'] / df['amount_wma_5']
    volume_ratio_5 = df['volume'] / df['volume_wma_5']
    
    # Construct an alpha factor based on the ratio of the current day's amount and volume to their respective 20-day WMA
    amount_ratio_20 = df['amount'] / df['amount_wma_20']
    volume_ratio_20 = df['volume'] / df['volume_wma_20']
    
    # Calculate momentum by comparing the current close price to the 20-day simple moving average
    df['close_sma_20'] = df['close'].rolling(window=20).mean()
    momentum = (df['close'] - df['close_sma_20']) / df['close_sma_20']
    
    # Combine the factors to create a final alpha score
    alpha_factor = (close_percentage_of_range * amount_ratio_5 * volume_ratio_5 * 
                    amount_ratio_20 * volume_ratio_20 * momentum)
    
    return alpha_factor
```
