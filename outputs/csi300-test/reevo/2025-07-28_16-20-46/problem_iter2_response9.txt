```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    df['close_5_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the trend strength as the difference between today's close and the 5-day moving average
    df['trend_strength'] = df['close'] - df['close_5_ma']
    
    # Calculate the 5-day rolling sum of volume
    df['volume_5_sum'] = df['volume'].rolling(window=5).sum()
    
    # Calculate the volume impact factor, which is the ratio of today's volume to the 5-day rolling sum
    df['volume_impact'] = df['volume'] / (df['volume_5_sum'] + 1e-7)
    
    # Calculate the 5-day standard deviation of the close price to measure price action consistency
    df['close_5_std'] = df['close'].rolling(window=5).std()
    
    # Heuristic factor: Combine trend strength, volume impact, and inverse of price action consistency
    heuristic_factor = df['trend_strength'] * df['volume_impact'] / (df['close_5_std'] + 1e-7)
    
    return heuristic_factor
```
