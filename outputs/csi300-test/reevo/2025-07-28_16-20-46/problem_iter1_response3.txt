```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the ratio of volume to amount, representing the average price per unit volume
    avg_price_per_volume = df['amount'] / df['volume']
    
    # Combine the two signals, normalizing by the high-low spread to avoid large fluctuations
    factor = (diff_close_open * avg_price_per_volume) / (df['high'] - df['low'] + 1e-7)
    
    return factor
```
