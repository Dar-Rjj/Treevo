```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate exponential moving averages for 5 and 20 days
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the Relative Strength (RS) as the ratio of the current close to the 20-day EMA
    df['RS'] = df['close'] / df['EMA_20']

    # Calculate the price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the volatility as the standard deviation of the last 5 days' closing prices
    df['volatility'] = df['close'].rolling(window=5).std()

    # Normalize the RS by the volatility
    df['normalized_RS'] = df['RS'] / (df['volatility'] + 1e-7)

    # Calculate the weighted momentum as the difference between the 5-day EMA and the 20-day EMA
    df['momentum'] = df['EMA_5'] - df['EMA_20']

    # Weighted combination of factors
    df['alpha_factor'] = (0.3 * df['normalized_RS'] + 
                          0.2 * df['price_change'] + 
                          0.4 * df['volume_change'] + 
                          0.1 * df['momentum'])

    return df['alpha_factor']
```
