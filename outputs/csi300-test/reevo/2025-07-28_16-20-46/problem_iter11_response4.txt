```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Volume-weighted price
    df['vwp'] = (df['close'] * df['volume']).cumsum() / df['volume'].cumsum()

    # Calculate the 5-day and 20-day exponential moving average of volume-weighted prices to capture momentum
    df['EMA_5_vwp'] = df['vwp'].ewm(span=5, adjust=False).mean()
    df['EMA_20_vwp'] = df['vwp'].ewm(span=20, adjust=False).mean()

    # Momentum factor: difference between 5-day and 20-day EMA of volume-weighted prices
    df['momentum_factor'] = df['EMA_5_vwp'] - df['EMA_20_vwp']

    # Calculate the 20-day standard deviation of volume-weighted prices to capture volatility
    df['volatility_factor'] = df['vwp'].rolling(window=20).std()

    # Calculate the 5-day rolling sum of volume to capture liquidity
    df['liquidity_factor'] = df['volume'].rolling(window=5).sum()

    # Calculate relative strength as the ratio of current vwp to 20-day EMA of vwp
    df['relative_strength'] = df['vwp'] / df['EMA_20_vwp']

    # Market breadth: difference between 5-day and 20-day moving averages of (high - low)
    df['range_5'] = (df['high'] - df['low']).rolling(window=5).mean()
    df['range_20'] = (df['high'] - df['low']).rolling(window=20).mean()
    df['market_breadth'] = df['range_5'] - df['range_20']

    # Adaptive weights for the factors based on entropy pooling
    df['momentum_weight'] = df['volatility_factor'].rank(pct=True)
    df['liquidity_weight'] = df['liquidity_factor'].rank(pct=True)
    df['relative_strength_weight'] = df['relative_strength'].rank(pct=True)
    df['market_breadth_weight'] = df['market_breadth'].rank(pct=True)

    # Combine the factors with adaptive weights
    heuristic_factor = (df['momentum_factor'] * df['momentum_weight']) + \
                       (df['liquidity_factor'] * df['liquidity_weight']) + \
                       (df['relative_strength'] * df['relative_strength_weight']) + \
                       (df['market_breadth'] * df['market_breadth_weight'])

    return heuristic_factor
```
