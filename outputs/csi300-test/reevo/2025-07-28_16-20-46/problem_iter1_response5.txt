```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the change in volume from the previous day
    df['volume_change'] = df['volume'].pct_change()
    # Calculate a simple moving average of the closing price over the last 5 days
    df['sma_5'] = df['close'].rolling(window=5).mean()
    # Calculate the standard deviation of the high price over the last 5 days
    df['std_high_5'] = df['high'].rolling(window=5).std()
    # Heuristics factor: combine the relative close difference, volume change, and std of high
    factor = (df['close_diff'] / (df['std_high_5'] + 1e-7)) * df['volume_change']
    return factor.dropna()
```
