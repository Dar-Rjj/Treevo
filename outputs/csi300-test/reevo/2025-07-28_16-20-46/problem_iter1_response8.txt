```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between closing price and opening price
    daily_diff = df['close'] - df['open']
    
    # Calculate the average volume over the past 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate a smoothed amount which is the 3-day moving average of the trading amount
    smooth_amount = df['amount'].rolling(window=3).mean()
    
    # Generate the heuristic: (daily price change / average volume) * smoothed amount
    factor = (daily_diff / (avg_volume_5d + 1e-7)) * smooth_amount
    
    return factor
```
