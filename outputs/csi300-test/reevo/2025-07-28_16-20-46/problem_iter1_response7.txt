```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the open, representing the daily return momentum
    daily_momentum = df['close'] - df['open']
    
    # Incorporate volume as a measure of strength for the move (volume weighted momentum)
    vol_weighted_momentum = daily_momentum * df['volume']
    
    # Calculate the average of the high and low prices to represent the central tendency of the day's trading range
    avg_price = (df['high'] + df['low']) / 2
    
    # Factor that combines momentum with the volume, adjusted by the average price level.
    factor = vol_weighted_momentum / (avg_price + 1e-7)
    
    return factor
```
