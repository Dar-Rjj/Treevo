```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate simple moving averages for 5 and 20 days
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate the Relative Strength (RS) as the ratio of the current close to the 20-day SMA
    df['RS'] = df['close'] / df['SMA_20']

    # Calculate the price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the volatility as the standard deviation of the last 5 days' closing prices
    df['volatility'] = df['close'].rolling(window=5).std()

    # Normalize the RS by the volatility
    df['normalized_RS'] = df['RS'] / (df['volatility'] + 1e-7)

    # Calculate the momentum as the difference between the 5-day SMA and the 20-day SMA
    df['momentum'] = df['SMA_5'] - df['SMA_20']

    # Calculate the 5-day and 20-day exponential moving average (EMA) of the closing prices
    df['EMA_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['EMA_20'] = df['close'].ewm(span=20, adjust=False).mean()

    # Calculate the 5-day and 20-day rate of change (ROC)
    df['ROC_5'] = df['close'].pct_change(periods=5)
    df['ROC_20'] = df['close'].pct_change(periods=20)

    # Calculate the 5-day and 20-day money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    df['MFI_5'] = (positive_money_flow.rolling(window=5).sum() / (positive_money_flow.rolling(window=5).sum() + negative_money_flow.rolling(window=5).sum())) * 100
    df['MFI_20'] = (positive_money_flow.rolling(window=20).sum() / (positive_money_flow.rolling(window=20).sum() + negative_money_flow.rolling(window=20).sum())) * 100

    # Combine the normalized RS, price change, volume change, momentum, EMA, ROC, and MFI to form the alpha factor
    df['alpha_factor'] = (
        0.2 * df['normalized_RS'] +
        0.2 * df['price_change'] +
        0.1 * df['volume_change'] +
        0.1 * df['momentum'] +
        0.1 * (df['EMA_5'] - df['EMA_20']) +
        0.1 * df['ROC_5'] +
        0.1 * df['ROC_20'] +
        0.1 * (df['MFI_5'] - df['MFI_20'])
    )

    return df['alpha_factor']
```
