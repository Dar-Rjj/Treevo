```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, to capture the momentum
    df['momentum'] = df['close'].diff()
    
    # Calculate the 5-day Simple Moving Average (SMA) of the closing prices
    df['5_day_SMA'] = df['close'].rolling(window=5).mean()
    
    # Calculate the percentage change in volume compared to the 10-day average volume
    df['avg_volume_10'] = df['volume'].rolling(window=10).mean()
    df['volume_change_ratio'] = (df['volume'] - df['avg_volume_10']) / (df['avg_volume_10'] + 1e-7)
    
    # The factor is a combination of the momentum and the impact of unusual volume changes.
    # It's intended to capture situations where significant price movement (momentum) is accompanied by unusual trading activity (volume).
    alpha_factor = df['momentum'] * df['volume_change_ratio']
    
    return alpha_factor
```
