```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate multi-period price changes
    df['price_change_1d'] = df['close'].pct_change(1)
    df['price_change_5d'] = df['close'].pct_change(5)
    df['price_change_20d'] = df['close'].pct_change(20)

    # Calculate multi-period volume changes
    df['volume_change_1d'] = df['volume'].pct_change(1)
    df['volume_change_5d'] = df['volume'].pct_change(5)
    df['volume_change_20d'] = df['volume'].pct_change(20)

    # Calculate the high-low range as a measure of volatility
    df['range'] = df['high'] - df['low']

    # Calculate the relative change in the high-low range
    df['range_change_1d'] = df['range'].pct_change(1)
    df['range_change_5d'] = df['range'].pct_change(5)
    df['range_change_20d'] = df['range'].pct_change(20)

    # Calculate the average of the high and low prices to represent the central tendency of the day's trading range
    df['avg_price'] = (df['high'] + df['low']) / 2

    # Calculate the weighted sum of these changes with a bias towards volume and incorporating range
    df['alpha_factor'] = (
        0.3 * df['price_change_1d'] + 
        0.2 * df['price_change_5d'] + 
        0.1 * df['price_change_20d'] +
        0.6 * df['volume_change_1d'] + 
        0.4 * df['volume_change_5d'] + 
        0.2 * df['volume_change_20d'] +
        0.1 * df['range_change_1d'] + 
        0.05 * df['range_change_5d'] + 
        0.025 * df['range_change_20d']
    ) / 2.875

    # Incorporate the average price level as a scaling factor
    df['alpha_factor'] = df['alpha_factor'] * df['avg_price']

    # Add a lag to the alpha factor to capture delayed effects
    df['alpha_factor'] = df['alpha_factor'].shift(1)

    return df['alpha_factor']
```
