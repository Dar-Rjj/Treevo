```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate simple moving averages for 5 and 20 days
    df['SMA_5'] = df['close'].rolling(window=5).mean()
    df['SMA_20'] = df['close'].rolling(window=20).mean()

    # Calculate the Relative Strength (RS) as the ratio of the current close to the 20-day SMA
    df['RS'] = df['close'] / df['SMA_20']

    # Calculate the price change and volume change
    df['price_change'] = df['close'].pct_change()
    df['volume_change'] = df['volume'].pct_change()

    # Calculate the 5-day and 20-day volatility (standard deviation of the closing prices)
    df['volatility_5'] = df['close'].rolling(window=5).std()
    df['volatility_20'] = df['close'].rolling(window=20).std()

    # Calculate the volatility ratio as a measure of recent volatility relative to longer-term volatility
    df['volatility_ratio'] = df['volatility_5'] / (df['volatility_20'] + 1e-7)

    # Normalize the RS by the 5-day volatility
    df['normalized_RS'] = df['RS'] / (df['volatility_5'] + 1e-7)

    # Calculate the mean reversion factor as the difference between the current close and the 20-day SMA
    df['mean_reversion'] = (df['close'] - df['SMA_20']) / (df['SMA_20'] + 1e-7)

    # Combine the normalized RS, price change, volume change, and mean reversion to form the alpha factor
    df['alpha_factor'] = (
        0.3 * df['normalized_RS'] +
        0.2 * df['price_change'] +
        0.4 * df['volume_change'] +
        0.1 * df['mean_reversion']
    ) / (df['volatility_ratio'] + 1e-7)

    return df['alpha_factor']
```
