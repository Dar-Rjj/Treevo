```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    df['close_5_ma'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day moving average of the close price for relative strength
    df['close_20_ma'] = df['close'].rolling(window=20).mean()
    
    # Calculate the relative strength as the difference between today's close and the 20-day moving average
    df['relative_strength'] = df['close'] - df['close_20_ma']
    
    # Calculate the 5-day rolling sum of volume
    df['volume_5_sum'] = df['volume'].rolling(window=5).sum()
    
    # Calculate the volume impact factor, which is the ratio of today's volume to the 5-day rolling sum
    df['volume_impact'] = df['volume'] / (df['volume_5_sum'] + 1e-7)
    
    # Identify volume spikes by comparing today's volume to the 5-day rolling sum
    df['volume_spike'] = (df['volume'] > 1.5 * df['volume_5_sum']).astype(int)
    
    # Calculate the 5-day standard deviation of the close price to measure price action stability
    df['close_5_std'] = df['close'].rolling(window=5).std()
    
    # Heuristic factor: Combine relative strength, volume impact, volume spike, and inverse of price action stability
    heuristic_factor = (df['relative_strength'] * df['volume_impact'] * (df['volume_spike'] + 1)) / (df['close_5_std'] + 1e-7)
    
    return heuristic_factor
```
