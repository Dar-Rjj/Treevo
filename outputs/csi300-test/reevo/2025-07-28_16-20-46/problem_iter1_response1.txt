```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate the exponential moving average of the close price with a 7-day span
    df['ema_7'] = df['close'].ewm(span=7, adjust=False).mean()
    
    # Calculate the ratio of the close price to the 7-day EMA
    ema_ratio = df['close'] / df['ema_7']
    
    # Calculate the money flow index (MFI) as a measure of buying/selling pressure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow.where(df['close'] > df['close'].shift(1), 0)
    negative_money_flow = raw_money_flow.where(df['close'] < df['close'].shift(1), 0)
    money_ratio = positive_money_flow.rolling(window=14).sum() / negative_money_flow.rolling(window=14).sum()
    mfi = 100 - (100 / (1 + money_ratio))
    
    # Alpha factor: Combine the EMA ratio and MFI
    alpha_factor = ema_ratio * (50 - mfi)
    
    return alpha_factor
```
