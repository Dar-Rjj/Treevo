```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close price
    df['close_diff'] = df['close'].diff()
    # Calculate the volume weighted average price (VWAP) over the last 5 days
    df['VWAP_5'] = (df['amount'].rolling(window=5).sum() / df['volume'].rolling(window=5).sum())
    # Heuristic factor: Today's close price change divided by VWAP of the past 5 days, to see if recent price changes are substantial compared to the average
    heuristic_factor = df['close_diff'] / (df['VWAP_5'] + 1e-7)
    return heuristic_factor
```
