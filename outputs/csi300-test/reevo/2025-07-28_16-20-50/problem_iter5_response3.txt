```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    import numpy as np

    # Calculate Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=5).mean()

    # Calculate multi-period momentum
    df['mom_5'] = df['close'].pct_change(periods=5)
    df['mom_10'] = df['close'].pct_change(periods=10)

    # Exponential smoothing of the close price
    df['exp_smooth_close'] = df['close'].ewm(span=5, adjust=False).mean()

    # Market depth: difference between high and low
    df['price_range'] = df['high'] - df['low']
    df['normalized_price_range'] = df['price_range'] / df['close']

    # Seasonality factor: day of the week
    df['day_of_week'] = df.index.dayofweek + 1  # Convert Monday to 1
    df['seasonality'] = df['day_of_week'] * df['close'].pct_change(periods=7)

    # Robust statistical method: Median of VWAP over a rolling window
    df['median_vwap'] = df['vwap'].rolling(window=5).median()

    # Combine factors
    factor = (
        df['vwap'] +
        df['mom_5'] +
        df['mom_10'] +
        df['exp_smooth_close'] +
        df['normalized_price_range'] +
        df['seasonality'] +
        df['median_vwap']
    ) / 7  # Average of the combined factors

    return factor
```
