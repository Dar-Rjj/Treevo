def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative part of the amount over volume
    df['positive_amount_vol'] = (df['amount'] / df['volume']).clip(lower=0)
    df['negative_amount_vol'] = (df['amount'] / df['volume']).clip(upper=0)

    # Sum of positive and absolute negative parts with a rolling window of 5
    pos_sum = df['positive_amount_vol'].rolling(window=5).sum()
    neg_sum_abs = df['negative_amount_vol'].abs().rolling(window=5).sum()

    # Factor: ratio of positive sum to absolute negative sum
    factor = pos_sum / (neg_sum_abs + 1e-7)

    # Calculate the relative change in close price
    df['close_rel_change'] = df['close'].pct_change()

    # Calculate the average of high and low prices
    df['avg_price'] = (df['high'] + df['low']) / 2

    # Calculate the price volatility using the standard deviation of the close price
    df['price_volatility'] = df['close'].pct_change().rolling(window=10).std()

    # Calculate the directional volume
    df['directional_volume'] = (df['close'] - df['open']) * df['volume']

    # Create a composite factor that incorporates the relative change in close price, average price, and directional volume
    df['composite_factor'] = (df['close_rel_change'] / df['avg_price']) * (df['directional_volume'] / (df['price_volatility'] + 1e-7))

    # Enhance the factor by incorporating the calculated factor from positive and negative amount over volume
    df['enhanced_factor'] = df['composite_factor'] * factor

    return df['enhanced_factor']
