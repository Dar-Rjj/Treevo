def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative part of the amount over volume
    df['positive_amount_vol'] = (df['amount'] / df['volume']).clip(lower=0)
    df['negative_amount_vol'] = (df['amount'] / df['volume']).clip(upper=0)

    # Sum of positive and absolute negative parts with dynamically adjusted window sizes
    pos_sum = df['positive_amount_vol'].rolling(window=int(len(df) * 0.05)).sum()
    neg_sum_abs = df['negative_amount_vol'].abs().rolling(window=int(len(df) * 0.05)).sum()

    # Factor: ratio of positive sum to absolute negative sum
    sentiment_factor = pos_sum / (neg_sum_abs + 1e-7)

    # Calculate the volatility using the close price
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    volatility = df['log_returns'].rolling(window=int(len(df) * 0.1)).std() * np.sqrt(252)

    # Calculate the VWAP with dynamically adjusted window size
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=int(len(df) * 0.1)).mean()

    # Exponential smoothing on the VWAP with adaptive span
    span = int(len(df) * 0.08)
    df['vwap_smoothed'] = df['vwap'].ewm(span=span, adjust=False).mean()

    # Momentum factor with dynamically adjusted periods
    momentum = df['close'].pct_change(periods=int(len(df) * 0.1))

    # Mean reversion factor with dynamically adjusted periods
    mean_reversion = -df['close'].pct_change(periods=int(len(df) * 0.07))

    # Long-term trend strength factor: difference between long and short exponential moving averages
    df['ema_long'] = df['close'].ewm(span=int(len(df) * 0.1), adjust=False).mean()
    df['ema_short'] = df['close'].ewm(span=int(len(df) * 0.05), adjust=False).mean()
    long_trend_strength = df['ema_short'] - df['ema_long']

    # Adaptive weighting based on recent market conditions
    recent_volatility = df['log_returns'].rolling(window=int(len(df) * 0.04)).std() * np.sqrt(252)
    adaptive_weight = 1 / (1 + np.exp(-recent_volatility))

    # Non-linear transformation of the factors
    sentiment_factor_nonlinear = np.sign(sentiment_factor) * np.log1p(np.abs(sentiment_factor))
    vwap_ratio_nonlinear = np.sign(df['vwap'] / df['vwap_smoothed']) * np.log1p(np.abs(df['vwap'] / df['vwap_smoothed']))

    # Additional factor: relative strength index (RSI) with dynamically adjusted window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=int(len(df) * 0.02)).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=int(len(df) * 0.02)).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Incorporate liquidity
    liquidity = df['volume'].rolling(window=int(len(df) * 0.05)).mean()

    vwap_ratio_nonlinear_std = (vwap_ratio_nonlinear - vwap_ratio_nonlinear.mean()) / vwap_ratio_nonlinear.std()
    momentum_std = (momentum - momentum.mean()) / momentum.std()
    mean_reversion_std = (mean_reversion - mean_reversion.mean()) / mean_reversion.std()
    long_trend_strength_std = (long_trend_strength - long_trend_strength.mean()) / long_trend_strength.std()
