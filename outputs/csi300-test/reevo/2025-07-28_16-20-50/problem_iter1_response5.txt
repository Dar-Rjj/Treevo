```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price, which represents the day's net change
    daily_net_change = df['close'] - df['open']
    
    # Calculate the average of the high and low prices, which can be seen as a simple measure of the day's midpoint
    daily_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate the ratio of the volume to the amount, representing the effective value per share traded
    value_per_share_traded = df['amount'] / df['volume']
    
    # The alpha factor is created by combining the normalized daily net change relative to the daily midpoint
    # with the logarithm of the value per share traded. This combination aims to capture both the price dynamics
    # in the context of the day's trading range and the economic significance of the trading activity.
    alpha_factor = (daily_net_change / (daily_midpoint + 1e-7)) * np.log(value_per_share_traded + 1)
    
    return alpha_factor
```
