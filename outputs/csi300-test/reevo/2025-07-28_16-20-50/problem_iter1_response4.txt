```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    diff_close_open = df['close'] - df['open']
    
    # Calculate the average volume over the past 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the momentum as the change in closing price over the past 5 days
    momentum_5d = df['close'].pct_change(periods=5)
    
    # Generate a factor that combines these features
    factor = (diff_close_open * momentum_5d) / (avg_volume_5d + 1e-7)
    
    return factor
```
