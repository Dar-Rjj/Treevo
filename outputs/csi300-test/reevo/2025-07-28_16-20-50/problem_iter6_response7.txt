```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 21-day volume-weighted average price (VWAP)
    df['vwap_21'] = (df['close'] * df['volume']).rolling(window=21).sum() / df['volume'].rolling(window=21).sum()
    
    # Calculate the ratio of today's close price to the 21-day VWAP
    long_term_momentum_factor = df['close'] / df['vwap_21']
    
    # Calculate the 50-day simple moving average of the close price to capture long-term trends
    df['sma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the mean reversion factor by taking the difference between the current close and the 50-day SMA
    long_term_mean_reversion_factor = df['close'] - df['sma_50']
    
    # Calculate the 7-day volume-weighted average price (VWAP)
    df['vwap_7'] = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Calculate the ratio of today's close price to the 7-day VWAP
    short_term_momentum_factor = df['close'] / df['vwap_7']
    
    # Calculate the 21-day simple moving average of the close price for short-term mean reversion
    df['sma_21'] = df['close'].rolling(window=21).mean()
    
    # Calculate the short-term mean reversion factor by taking the difference between the current close and the 21-day SMA
    short_term_mean_reversion_factor = df['close'] - df['sma_21']
    
    # Calculate the 14-day exponentially weighted moving average (EWMA) of the close price
    df['ewma_14'] = df['close'].ewm(span=14, adjust=False).mean()
    
    # Calculate the EWMA-based momentum factor
    ewma_momentum_factor = df['close'] / df['ewma_14']
    
    # Calculate the 14-day exponentially weighted moving average (EWMA) of the volume
    df['volume_ewma_14'] = df['volume'].ewm(span=14, adjust=False).mean()
    
    # Calculate the volume intensity factor
    volume_intensity_factor = df['volume'] / df['volume_ewma_14']
    
    # Combine the long-term and short-term momentum and mean reversion factors with the EWMA and volume intensity factors
    combined_factor = (long_term_momentum_factor + long_term_mean_reversion_factor +
                       short_term_momentum_factor + short_term_mean_reversion_factor +
                       ewma_momentum_factor + volume_intensity_factor) / 6
    
    return combined_factor
```
