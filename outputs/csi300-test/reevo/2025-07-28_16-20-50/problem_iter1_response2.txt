```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the money flow multiplier
    mfm = (2 * df['close'] - df['high'] - df['low']) / (df['high'] - df['low'] + 1e-7)
    # Calculate the money flow volume
    mf_volume = mfm * df['volume']
    # Calculate a simple moving average of the money flow volume over 10 days
    sma_mf_volume = mf_volume.rolling(window=10).mean()
    # Subtract the 10-day simple moving average of the actual volume from the 10-day SMA of the money flow volume
    factor = sma_mf_volume - df['volume'].rolling(window=10).mean()
    return factor
```
