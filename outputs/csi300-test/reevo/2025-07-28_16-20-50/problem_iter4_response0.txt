```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import numpy as np
    import pandas as pd

    # Calculate the daily net change
    daily_net_change = df['close'] - df['open']
    
    # Calculate the daily midpoint
    daily_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate the value per share traded
    value_per_share_traded = df['amount'] / df['volume']
    
    # Calculate the momentum factor using the 5-day return
    momentum = df['close'].pct_change(periods=5).fillna(0)
    
    # Calculate the volatility as the exponential moving average of the past 5 days' returns
    volatility = df['close'].pct_change().ewm(span=5, adjust=False).std().fillna(0)
    
    # Calculate trade intensity as the ratio of current volume to the 5-day exponential moving average volume
    trade_intensity = df['volume'] / df['volume'].ewm(span=5, adjust=False).mean().fillna(1)
    
    # Incorporate market depth: relative difference between high and low prices
    market_depth = (df['high'] - df['low']) / df['close']
    
    # Calculate relative strength index (RSI) for the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(alpha=1/14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the factors: normalized daily net change, logarithm of value per share traded, momentum, trade intensity, market depth, and RSI
    alpha_factor = (daily_net_change / (daily_midpoint + 1e-7)) * np.log(value_per_share_traded + 1) * (momentum + 1) * (trade_intensity + 1) * (1 / (market_depth + 1e-7)) * (rsi / 100)
    
    return alpha_factor
```
