```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative part of the daily net change
    df['daily_net_change'] = df['close'] - df['open']
    df['positive_net_change'] = df['daily_net_change'].clip(lower=0)
    df['negative_net_change'] = df['daily_net_change'].clip(upper=0)

    # Calculate the positive and negative part of the amount over volume
    df['positive_amount_vol'] = (df['amount'] / df['volume']).clip(lower=0)
    df['negative_amount_vol'] = (df['amount'] / df['volume']).clip(upper=0)

    # Sum of positive and absolute negative parts of the daily net change
    pos_net_change_sum = df['positive_net_change'].rolling(window=5).sum()
    neg_net_change_sum_abs = df['negative_net_change'].abs().rolling(window=5).sum()

    # Sum of positive and absolute negative parts of the amount over volume
    pos_amount_vol_sum = df['positive_amount_vol'].rolling(window=5).sum()
    neg_amount_vol_sum_abs = df['negative_amount_vol'].abs().rolling(window=5).sum()

    # Factor: ratio of positive sum to absolute negative sum for both net change and amount over volume
    net_change_factor = pos_net_change_sum / (neg_net_change_sum_abs + 1e-7)
    amount_vol_factor = pos_amount_vol_sum / (neg_amount_vol_sum_abs + 1e-7)

    # Combine the two factors
    combined_factor = net_change_factor * amount_vol_factor

    return combined_factor
```
