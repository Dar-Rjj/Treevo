```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price range and volume-weighted average price (VWAP)
    df['price_range'] = df['high'] - df['low']
    df['VWAP'] = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the multi-period momentum as the difference between the current close and the VWAP
    df['momentum'] = df['close'] - df['VWAP']

    # Exponential smoothing on the momentum
    df['smooth_momentum'] = df['momentum'].ewm(span=10, adjust=False).mean()

    # Seasonality adjustment (e.g., monthly or weekly patterns)
    df['day_of_week'] = df.index.dayofweek  # Assuming index is datetime
    seasonal_factor = df.groupby('day_of_week')['smooth_momentum'].transform('mean')

    # Final factor combining smooth momentum with seasonality
    final_factor = (df['smooth_momentum'] - seasonal_factor) / (df['price_range'] + 1e-7)

    return final_factor
```
