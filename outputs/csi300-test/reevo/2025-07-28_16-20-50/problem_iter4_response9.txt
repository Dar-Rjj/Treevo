```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    price_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate a weighted momentum using the amount and volume
    momentum = (df['amount'] * df['volume']).rolling(window=7).mean()
    
    # Calculate the price volatility using the standard deviation of the price difference
    price_volatility = price_diff.rolling(window=10).std()
    
    # Incorporate longer-term trends using exponential smoothing
    trend = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the relative strength index (RSI) as a measure of market sentiment
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Create a factor that combines the price movement, average price, momentum, volatility, trend, and RSI
    factor = (price_diff / avg_price) * momentum / (price_volatility + 1e-7) * trend * rsi
    
    return factor
```
