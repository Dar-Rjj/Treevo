def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns to capture percentage change and smooth out the return series
    log_returns = np.log(df['close'] / df['close'].shift(1))
    
    # Calculate 5-day and 20-day rolling standard deviation of log returns as a measure of short and medium term volatility
    vol_5d = log_returns.rolling(window=5).std()
    vol_20d = log_returns.rolling(window=20).std()
    
    # Trend strength, calculated as the cumulative sum of the sign of daily returns over the last 20 days
    trend_strength = log_returns.rolling(window=20).apply(np.sign, raw=True).cumsum()
    
    # Liquidity factor, combining volume with the absolute value of log returns (to reflect both volume and price movement intensity)
    liquidity_factor = (df['volume'] * abs(log_returns)).rolling(window=5).mean()
    
    # Alpha factor: a composite score that combines the inverse of short-term volatility, trend strength, and normalized liquidity
    alpha_factor = (1 / (vol_5d + 1e-7)) * trend_strength * (liquidity_factor / (vol_20d + 1e-7))
    
    return alpha_factor
