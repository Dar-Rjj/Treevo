```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:

    # Calculate the positive and negative part of the amount over volume
    df['positive_amount_vol'] = (df['amount'] / df['volume']).clip(lower=0)
    df['negative_amount_vol'] = (df['amount'] / df['volume']).clip(upper=0)

    # Sum of positive and absolute negative parts with dynamic window
    pos_sum = df['positive_amount_vol'].rolling(window=df['positive_amount_vol'].rolling(window=5).mean().astype(int)).sum()
    neg_sum_abs = df['negative_amount_vol'].abs().rolling(window=df['negative_amount_vol'].abs().rolling(window=5).mean().astype(int)).sum()

    # Factor: ratio of positive sum to absolute negative sum
    sentiment_factor = pos_sum / (neg_sum_abs + 1e-7)

    # Calculate the volatility using the close price
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    volatility = df['log_returns'].rolling(window=20).std() * np.sqrt(252)

    # Calculate the VWAP
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=20).mean()

    # Exponential smoothing on the VWAP
    df['vwap_smoothed'] = df['vwap'].ewm(span=20, adjust=False).mean()

    # Momentum factor
    momentum = df['close'].pct_change(periods=20)

    # Mean reversion factor
    mean_reversion = -df['close'].pct_change(periods=5)

    # Trend strength factor: difference between 20-day and 5-day exponential moving averages
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    trend_strength = df['ema_5'] - df['ema_20']

    # Adaptive weighting based on recent market conditions
    recent_volatility = df['log_returns'].rolling(window=10).std() * np.sqrt(252)
    weight_sentiment = 0.3 * (1 / (recent_volatility + 1e-7))
    weight_volatility = 0.2 * (1 / (volatility + 1e-7))
    weight_vwap = 0.2 * (df['vwap'] / df['vwap_smoothed'])
    weight_momentum = 0.15 * (momentum / (recent_volatility + 1e-7))
    weight_mean_reversion = 0.15 * (mean_reversion / (recent_volatility + 1e-7))
    weight_trend_strength = 0.05 * (trend_strength / (recent_volatility + 1e-7))

    # Normalize the weights
    total_weight = (weight_sentiment + weight_volatility + weight_vwap + 
                    weight_momentum + weight_mean_reversion + weight_trend_strength)
    weight_sentiment /= total_weight
    weight_volatility /= total_weight
    weight_vwap /= total_weight
    weight_momentum /= total_weight
    weight_mean_reversion /= total_weight
    weight_trend_strength /= total_weight

    # Combine factors with adaptive weights
    alpha_factor = (weight_sentiment * sentiment_factor + 
                    weight_volatility * (1/volatility) + 
                    weight_vwap * (df['vwap'] / df['vwap_smoothed']) + 
                    weight_momentum * momentum + 
                    weight_mean_reversion * mean_reversion + 
                    weight_trend_strength * trend_strength)

    return alpha_factor
```
