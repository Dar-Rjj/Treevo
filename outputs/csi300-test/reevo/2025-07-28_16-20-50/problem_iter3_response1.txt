```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate rolling mean and standard deviation of close prices
    rolling_mean = df['close'].rolling(window=5).mean()
    rolling_std = df['close'].rolling(window=5).std()

    # Calculate the normalized volume-weighted amount
    df['amount_vol_normalized'] = (df['amount'] / df['volume']) / (df['high'] - df['low'])

    # Calculate the positive and negative part of the normalized amount over volume
    df['positive_amount_vol_norm'] = df['amount_vol_normalized'].clip(lower=0)
    df['negative_amount_vol_norm'] = df['amount_vol_normalized'].clip(upper=0)

    # Sum of positive and absolute negative parts
    pos_sum = df['positive_amount_vol_norm'].rolling(window=5).sum()
    neg_sum_abs = df['negative_amount_vol_norm'].abs().rolling(window=5).sum()

    # Factor: ratio of positive sum to absolute negative sum
    factor_pos_neg_ratio = pos_sum / (neg_sum_abs + 1e-7)

    # Calculate multi-period returns
    df['return_2d'] = df['close'].pct_change(periods=2)
    df['return_5d'] = df['close'].pct_change(periods=5)

    # Combine momentum, volatility, and liquidity
    factor = (df['return_5d'] + (df['return_2d'] / rolling_std)) * factor_pos_neg_ratio

    return factor
```
