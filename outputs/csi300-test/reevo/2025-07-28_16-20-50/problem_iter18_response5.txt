```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the positive and negative part of the amount over volume
    df['positive_amount_vol'] = (df['amount'] / df['volume']).clip(lower=0)
    df['negative_amount_vol'] = (df['amount'] / df['volume']).clip(upper=0)

    # Sum of positive and absolute negative parts with a shorter window
    pos_sum = df['positive_amount_vol'].rolling(window=3).sum()
    neg_sum_abs = df['negative_amount_vol'].abs().rolling(window=3).sum()

    # Factor: ratio of positive sum to absolute negative sum
    sentiment_factor = pos_sum / (neg_sum_abs + 1e-7)

    # Calculate the volatility using the close price with a shorter window
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    volatility = df['log_returns'].rolling(window=5).std() * np.sqrt(252)

    # Calculate the VWAP with a shorter window
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=5).mean()

    # Exponential smoothing on the VWAP with a shorter span
    df['vwap_smoothed'] = df['vwap'].ewm(span=5, adjust=False).mean()

    # Momentum factor with a shorter period
    momentum = df['close'].pct_change(periods=3)

    # Mean reversion factor with a shorter period
    mean_reversion = -df['close'].pct_change(periods=2)

    # Trend strength factor: difference between 5-day and 2-day exponential moving averages
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    df['ema_2'] = df['close'].ewm(span=2, adjust=False).mean()
    trend_strength = df['ema_2'] - df['ema_5']

    # Adaptive weighting based on recent market conditions with a shorter window
    recent_volatility = df['log_returns'].rolling(window=3).std() * np.sqrt(252)
    adaptive_weight = 1 / (1 + np.exp(-recent_volatility))

    # Combine factors with adaptive weights
    alpha_factor = (adaptive_weight * sentiment_factor + 
                    (1 - adaptive_weight) * (1/volatility) + 
                    0.2 * (df['vwap'] / df['vwap_smoothed']) + 
                    0.15 * momentum + 
                    0.15 * mean_reversion + 
                    0.05 * trend_strength)

    return alpha_factor
```
