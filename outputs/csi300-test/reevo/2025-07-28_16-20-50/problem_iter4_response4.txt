```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the relative change in close price
    df['close_rel_change'] = df['close'].pct_change()
    
    # Calculate the average of high and low prices
    df['avg_price'] = (df['high'] + df['low']) / 2
    
    # Calculate a weighted momentum using the amount and volume
    df['momentum'] = (df['amount'] * df['volume']).shift(1).rolling(window=7).mean()
    
    # Calculate the price volatility using the standard deviation of the close price
    df['price_volatility'] = df['close'].pct_change().rolling(window=10).std()
    
    # Calculate the relative change in volume
    df['volume_rel_change'] = df['volume'].pct_change()
    
    # Create a factor that combines the relative change in close price, average price, momentum, and volatility
    df['factor'] = (df['close_rel_change'] / df['avg_price']) * df['momentum'] / (df['price_volatility'] + 1e-7)
    
    # Enhance the factor by incorporating the relative change in volume
    df['enhanced_factor'] = df['factor'] * df['volume_rel_change']
    
    return df['enhanced_factor']
```
