```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 14-day volume-weighted average price (VWAP)
    df['vwap_14'] = (df['close'] * df['volume']).rolling(window=14).sum() / df['volume'].rolling(window=14).sum()
    
    # Calculate the 28-day volume-weighted average price (VWAP)
    df['vwap_28'] = (df['close'] * df['volume']).rolling(window=28).sum() / df['volume'].rolling(window=28).sum()
    
    # Calculate the 7-day simple moving average of the close price for short-term trend
    df['sma_7'] = df['close'].rolling(window=7).mean()
    
    # Calculate the 56-day simple moving average of the close price for long-term trend
    df['sma_56'] = df['close'].rolling(window=56).mean()
    
    # Calculate the ratio of today's close price to the 14-day VWAP
    momentum_factor_14 = df['close'] / df['vwap_14']
    
    # Calculate the ratio of today's close price to the 28-day VWAP
    momentum_factor_28 = df['close'] / df['vwap_28']
    
    # Calculate the mean reversion factor by taking the difference between the current close and the 7-day SMA
    mean_reversion_factor_7 = df['close'] - df['sma_7']
    
    # Calculate the mean reversion factor by taking the difference between the current close and the 56-day SMA
    mean_reversion_factor_56 = df['close'] - df['sma_56']
    
    # Combine the momentum and mean reversion factors, giving more weight to the longer-term factors
    combined_factor = (momentum_factor_14 + momentum_factor_28 * 2 + 
                       mean_reversion_factor_7 + mean_reversion_factor_56 * 2) / 6
    
    return combined_factor
```
