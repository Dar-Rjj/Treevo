    volatility = df['close'].rolling(window=df['close'].rolling(window=5).std().astype(int), min_periods=1).std()

    vwap_window = df['volume'].rolling(window=5).mean().astype(int)

    mean_reversion = df['close'] - df['close'].rolling(window=5, min_periods=1).mean()

    neg_sum_abs = df['negative_amount_vol'].abs().rolling(window=5, min_periods=1).sum()
    liquidity_factor = pos_sum / (neg_sum_abs + 1e-7)
