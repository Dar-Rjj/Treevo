```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    price_diff = df['close'] - df['open']
    
    # Calculate the average of high and low prices
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate a weighted momentum using the amount and volume
    momentum = ((df['amount'] * df['volume']).shift(1)).rolling(window=7).mean()
    
    # Create a factor that combines the price movement, average price, and momentum
    factor = (price_diff / avg_price) * momentum
    
    return factor
```
