def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily net change
    daily_net_change = df['close'] - df['open']
    
    # Calculate the daily midpoint
    daily_midpoint = (df['high'] + df['low']) / 2
    
    # Calculate the value per share traded
    value_per_share_traded = df['amount'] / df['volume']
    
    # Calculate the 5-day return for momentum
    momentum = df['close'].pct_change(periods=5).fillna(0)
    
    # Calculate the volatility as the standard deviation of the past 5 days' returns
    volatility = df['close'].pct_change().rolling(window=5).std().fillna(0)
    
    # Calculate trade intensity as the ratio of current volume to the 5-day average volume
    trade_intensity = df['volume'] / df['volume'].rolling(window=5).mean().fillna(1)
    
    # Calculate the price range
    price_range = df['high'] - df['low']
    
    # Calculate the logarithm of the value per share traded
    log_value_per_share = np.log(value_per_share_traded + 1)
    
    # Combine the factors: normalized daily net change, logarithm of value per share traded, momentum, and trade intensity
    alpha_factor = (daily_net_change / (price_range + 1e-7)) * log_value_per_share * (momentum + 1) * (trade_intensity + 1) / (volatility + 1e-7)
    
    # Smooth the alpha factor with a 5-day rolling mean
    smoothed_alpha_factor = alpha_factor.rolling(window=5).mean().fillna(0)
    
    return smoothed_alpha_factor
