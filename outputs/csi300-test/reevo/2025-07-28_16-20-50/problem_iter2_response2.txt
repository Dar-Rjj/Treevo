```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the weighted moving average of closing prices, using volume as weights for a 7-day window
    df['wma_7'] = (df['close'] * df['volume']).rolling(window=7).sum() / df['volume'].rolling(window=7).sum()
    
    # Calculate the ratio of today's close price to the 7-day weighted moving average
    momentum_factor = df['close'] / df['wma_7']
    
    # Calculate the 21-day simple moving average of the close price
    df['sma_21'] = df['close'].rolling(window=21).mean()
    
    # Calculate the mean reversion factor by taking the difference between the current close and the 21-day SMA
    mean_reversion_factor = df['close'] - df['sma_21']
    
    # Combine the momentum and mean reversion factors
    combined_factor = momentum_factor + mean_reversion_factor
    
    return combined_factor
```
