```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the Volume Weighted Average Price (VWAP)
    df['vwap'] = (df['amount'] / df['volume'])

    # Calculate 5 and 20 period Simple Moving Averages (SMA)
    df['sma_5'] = df['close'].rolling(window=5).mean()
    df['sma_20'] = df['close'].rolling(window=20).mean()

    # Calculate the positive and negative part of the amount over volume
    df['positive_amount_vol'] = (df['amount'] / df['volume']).clip(lower=0)
    df['negative_amount_vol'] = (df['amount'] / df['volume']).clip(upper=0)

    # Sum of positive and absolute negative parts
    pos_sum = df['positive_amount_vol'].rolling(window=5).sum()
    neg_sum_abs = df['negative_amount_vol'].abs().rolling(window=5).sum()

    # Factor: ratio of positive sum to absolute negative sum
    ratio_pos_neg = pos_sum / (neg_sum_abs + 1e-7)

    # Momentum factor: 5-day SMA minus 20-day SMA
    momentum = df['sma_5'] - df['sma_20']

    # Mean reversion factor: VWAP minus close price
    mean_reversion = df['vwap'] - df['close']

    # Volatility factor: 20-day standard deviation of the close price
    volatility = df['close'].rolling(window=20).std()

    # Leverage weighted factors
    leverage_weighted_momentum = momentum / (volatility + 1e-7)
    leverage_weighted_mean_reversion = mean_reversion / (volatility + 1e-7)

    # Inter-period relationship: 5-day return
    df['return_5'] = df['close'].pct_change(periods=5)
    inter_period_relationship = df['return_5']

    # Combine factors with weights
    factor = 0.3 * ratio_pos_neg + 0.2 * leverage_weighted_momentum + 0.2 * leverage_weighted_mean_reversion + 0.3 * inter_period_relationship

    return factor
```
