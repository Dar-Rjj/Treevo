```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:

    # Calculate the positive and negative part of the amount over volume
    df['positive_amount_vol'] = (df['amount'] / df['volume']).clip(lower=0)
    df['negative_amount_vol'] = (df['amount'] / df['volume']).clip(upper=0)

    # Sum of positive and absolute negative parts with dynamic window
    pos_sum = df['positive_amount_vol'].rolling(window=5, min_periods=1).sum()
    neg_sum_abs = df['negative_amount_vol'].abs().rolling(window=5, min_periods=1).sum()

    # Factor: ratio of positive sum to absolute negative sum
    sentiment_factor = pos_sum / (neg_sum_abs + 1e-7)

    # Calculate the volatility using the close price with dynamic window
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1))
    volatility = df['log_returns'].rolling(window=20, min_periods=1).std() * np.sqrt(252)

    # Calculate the VWAP with dynamic window
    df['vwap'] = (df['amount'] / df['volume']).rolling(window=20, min_periods=1).mean()

    # Exponential smoothing on the VWAP
    df['vwap_smoothed'] = df['vwap'].ewm(span=20, adjust=False).mean()

    # Momentum factor with adaptive weighting
    momentum = df['close'].pct_change(periods=20)
    momentum_weight = 0.15 + 0.05 * (momentum.rank(pct=True) - 0.5)

    # Mean reversion factor with adaptive weighting
    mean_reversion = -df['close'].pct_change(periods=5)
    mean_reversion_weight = 0.15 + 0.05 * (mean_reversion.rank(pct=True) - 0.5)

    # Trend strength factor: difference between 20-day and 5-day exponential moving averages
    df['ema_20'] = df['close'].ewm(span=20, adjust=False).mean()
    df['ema_5'] = df['close'].ewm(span=5, adjust=False).mean()
    trend_strength = df['ema_5'] - df['ema_20']

    # Adaptive weighting for trend strength
    trend_strength_weight = 0.05 + 0.02 * (trend_strength.rank(pct=True) - 0.5)

    # Combine factors with adaptive weights
    alpha_factor = (0.3 * sentiment_factor + 
                    0.2 * (1/volatility) + 
                    0.2 * (df['vwap'] / df['vwap_smoothed']) + 
                    momentum_weight * momentum + 
                    mean_reversion_weight * mean_reversion + 
                    trend_strength_weight * trend_strength)

    return alpha_factor
```
