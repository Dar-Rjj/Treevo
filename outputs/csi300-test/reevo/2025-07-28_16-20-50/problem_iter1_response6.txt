```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close price and yesterday's close price
    daily_return = df['close'] - df['close'].shift(1)
    
    # Calculate the rolling 5-day average of the difference between high and low price
    avg_high_low_diff_5d = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Calculate the weighted volume, where weight is the daily return
    weighted_volume = daily_return * df['volume']
    
    # Alpha factor: ratio of the 5-day average of high-low difference to the weighted volume
    alpha_factor = avg_high_low_diff_5d / (weighted_volume + 1e-7)
    return alpha_factor
```
