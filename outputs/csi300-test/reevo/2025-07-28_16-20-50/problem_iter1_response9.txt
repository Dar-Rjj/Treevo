```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low, and the difference between close and open
    hl_diff = df['high'] - df['low']
    co_diff = df['close'] - df['open']
    
    # Create a factor that emphasizes stocks with strong intraday momentum (high difference from open to close relative to the day range)
    # And adjust this value based on the volume traded, assuming higher volume supports stronger price movements
    factor_values = (co_diff / (hl_diff + 1e-7)) * (df['volume'] / df['volume'].rolling(window=5).mean())
    
    return factor_values
```
