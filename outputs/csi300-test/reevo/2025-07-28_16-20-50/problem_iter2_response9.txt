```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Relative strength (momentum) as the difference between today's close and 20-day moving average
    relative_strength = df['close'] - ma_20
    
    # Volatility as the standard deviation of the daily returns over the last 20 days
    daily_returns = df['close'] - df['close'].shift(1)
    volatility = daily_returns.rolling(window=20).std()
    
    # Weighted volume, where weight is the daily return
    weighted_volume = daily_returns * df['volume']
    
    # Alpha factor: Combine relative strength, 5-day moving average, and volatility adjusted by weighted volume
    alpha_factor = (relative_strength / (ma_5 + 1e-7)) * (volatility / (weighted_volume + 1e-7))
    return alpha_factor
```
