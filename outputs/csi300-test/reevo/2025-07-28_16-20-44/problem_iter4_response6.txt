```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the average of the open and close prices, which can represent the central tendency of the day
    mid_point = (df['open'] + df['close']) / 2
    
    # Calculate the money flow ratio using the amount (dollar volume) and volume
    money_flow_ratio = df['amount'] / df['volume']
    
    # Calculate the momentum as the difference between the current close and the close 5 days ago
    df['momentum'] = df['close'].diff(periods=5)
    
    # Calculate the ATR (Average True Range) as a measure of volatility over the past 14 days
    df['tr'] = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift()), abs(df['low'] - df['close'].shift())], axis=1).max(axis=1)
    df['atr_14'] = df['tr'].rolling(window=14).mean()
    
    # Calculate the relative strength as the ratio of the current close to the 20-day moving average
    df['ma_20'] = df['close'].rolling(window=20).mean()
    df['relative_strength'] = df['close'] / df['ma_20']
    
    # Calculate the price change from today's open to today's close
    intraday_change = df['close'] - df['open']
    
    # Calculate the price change from previous day's close to today's open
    overnight_change = df['open'] - df['close'].shift(1)
    
    # Calculate the 5-day simple moving average of the close price
    df['ma_5'] = df['close'].rolling(window=5).mean()
    
    # Calculate the 50-day simple moving average of the close price
    df['ma_50'] = df['close'].rolling(window=50).mean()
    
    # Calculate the trend strength as the ratio of the 5-day moving average to the 50-day moving average
    df['trend_strength'] = df['ma_5'] / df['ma_50']
    
    # Create an interaction effect factor by combining the normalized distance from the low to the midpoint,
    # scaled by the money flow ratio, momentum, ATR, relative strength, intraday change, overnight change, and trend strength.
    # The idea is that a higher mid-point in relation to the day's range, combined with strong money flow, positive momentum,
    # lower volatility, higher relative strength, positive intraday and overnight changes, and a strong trend, could be indicative of stronger future returns.
    factor = ((mid_point - df['low']) / (df['high'] - df['low'] + 1e-7)) * money_flow_ratio * df['momentum'] / df['atr_14'] * df['relative_strength'] * intraday_change * overnight_change * df['trend_strength']

    return factor
```
