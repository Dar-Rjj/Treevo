```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using multiple periods
    momentum_short = df['close'].pct_change(periods=5)
    momentum_medium = df['close'].pct_change(periods=20)
    momentum_long = df['close'].pct_change(periods=60)
    momentum = (momentum_short + momentum_medium + momentum_long) / 3

    # Calculate the liquidity factor using the average volume over multiple periods
    liquidity_short = df['volume'].rolling(window=5).mean()
    liquidity_medium = df['volume'].rolling(window=20).mean()
    liquidity_long = df['volume'].rolling(window=60).mean()
    liquidity = (liquidity_short + liquidity_medium + liquidity_long) / 3

    # Calculate the market sentiment using the money flow ratio over multiple periods
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_short = money_flow_ratio.rolling(window=5).mean()
    market_sentiment_medium = money_flow_ratio.rolling(window=20).mean()
    market_sentiment_long = money_flow_ratio.rolling(window=60).mean()
    market_sentiment = (market_sentiment_short + market_sentiment_medium + market_sentiment_long) / 3

    # Calculate the volatility factor using the rolling standard deviation of the close price over the last 20 days
    volatility = df['close'].rolling(window=20).std()

    # Calculate the buying pressure as (high - close) / (high - low + 1e-7), indicating strength of buying relative to range
    buying_pressure = (df['high'] - df['close']) / (df['high'] - df['low'] + 1e-7)

    # Normalize the factors
    momentum_normalized = (momentum - momentum.mean()) / momentum.std()
    liquidity_normalized = (liquidity - liquidity.mean()) / liquidity.std()
    market_sentiment_normalized = (market_sentiment - market_sentiment.mean()) / market_sentiment.std()
    volatility_normalized = (volatility - volatility.mean()) / volatility.std()
    buying_pressure_normalized = (buying_pressure - buying_pressure.mean()) / buying_pressure.std()

    # Create a combined factor by multiplying the normalized factors
    factor = (momentum_normalized + 1) * (liquidity_normalized + 1) * (market_sentiment_normalized + 1) * (1 / (volatility_normalized + 1)) * (buying_pressure_normalized + 1)

    return factor
```
