```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the logarithmic return over the last 10 days
    log_return_10 = (df['close'].shift(10) / df['close']).apply(np.log)
    momentum = -log_return_10  # Negation to make positive momentum values higher
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Create a combined factor by multiplying the momentum, liquidity, and market sentiment
    # Avoid normalization as per instruction
    factor = momentum * liquidity * market_sentiment
    
    return factor
```
