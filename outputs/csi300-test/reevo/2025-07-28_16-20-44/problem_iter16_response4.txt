```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return over varying lookback periods (10-30 days)
    momentum_10 = np.log(df['close']).diff(periods=10).fillna(0)
    momentum_20 = np.log(df['close']).diff(periods=20).fillna(0)
    momentum_30 = np.log(df['close']).diff(periods=30).fillna(0)
    momentum_combined = (momentum_10 + momentum_20 + momentum_30) / 3

    # Calculate the liquidity factor using the average volume over varying lookback periods (10-40 days)
    liquidity_10 = df['volume'].rolling(window=10).mean().fillna(0)
    liquidity_20 = df['volume'].rolling(window=20).mean().fillna(0)
    liquidity_30 = df['volume'].rolling(window=30).mean().fillna(0)
    liquidity_40 = df['volume'].rolling(window=40).mean().fillna(0)
    liquidity_combined = (liquidity_10 + liquidity_20 + liquidity_30 + liquidity_40) / 4

    # Calculate the market sentiment using the average money flow ratio over varying lookback periods (5-15 days)
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_5 = money_flow_ratio.rolling(window=5).mean().fillna(0)
    market_sentiment_10 = money_flow_ratio.rolling(window=10).mean().fillna(0)
    market_sentiment_15 = money_flow_ratio.rolling(window=15).mean().fillna(0)
    market_sentiment_combined = (market_sentiment_5 + market_sentiment_10 + market_sentiment_15) / 3

    # Calculate the volatility factor using the standard deviation of the log returns over varying lookback periods (20-60 days)
    volatility_20 = np.log(df['close']).diff().rolling(window=20).std().fillna(0)
    volatility_40 = np.log(df['close']).diff().rolling(window=40).std().fillna(0)
    volatility_60 = np.log(df['close']).diff().rolling(window=60).std().fillna(0)
    volatility_combined = (volatility_20 + volatility_40 + volatility_60) / 3

    # Calculate the intraday dynamics factor using the ratio of high to low prices over varying lookback periods (5-10 days)
    intraday_dynamics_5 = (df['high'] / df['low']).rolling(window=5).mean().fillna(1)
    intraday_dynamics_10 = (df['high'] / df['low']).rolling(window=10).mean().fillna(1)
    intraday_dynamics_combined = (intraday_dynamics_5 + intraday_dynamics_10) / 2

    # Adjust the factor weights based on historical performance
    momentum_weight = 0.35
    liquidity_weight = 0.20
    market_sentiment_weight = 0.20
    volatility_weight = -0.15
    intraday_dynamics_weight = 0.10

    # Combine the factors with their respective weights
    factor = (momentum_combined * momentum_weight + 
              liquidity_combined * liquidity_weight + 
              market_sentiment_combined * market_sentiment_weight + 
              intraday_dynamics_combined * intraday_dynamics_weight + 
              volatility_combined * volatility_weight)

    return factor
```
