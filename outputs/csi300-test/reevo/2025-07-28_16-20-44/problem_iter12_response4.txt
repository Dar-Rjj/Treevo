```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns over the last 10 days
    log_returns = np.log(df['close']).diff(periods=10)
    
    # Calculate the volatility of the log returns over the last 30 days using exponential smoothing
    alpha = 0.1
    volatility = log_returns.ewm(alpha=alpha, adjust=False).std()
    
    # Calculate the liquidity factor using the exponentially smoothed average volume over the last 20 days
    liquidity = df['volume'].ewm(span=20, adjust=False).mean()
    
    # Calculate the market sentiment using the exponentially smoothed average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.ewm(span=5, adjust=False).mean()
    
    # Calculate the RSI (Relative Strength Index) over the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).ewm(span=14, adjust=False).mean()
    loss = (-delta.where(delta < 0, 0)).ewm(span=14, adjust=False).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the MACD (Moving Average Convergence Divergence)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd = macd_line - signal_line
    
    # Calculate the Aroon Oscillator
    aroon_up = df['high'].rolling(window=25).apply(lambda x: float((x.argmax() + 1) / 25 * 100), raw=True)
    aroon_down = df['low'].rolling(window=25).apply(lambda x: float((x.argmin() + 1) / 25 * 100), raw=True)
    aroon_oscillator = aroon_up - aroon_down
    
    # Dynamic weighting based on the standard deviation of each factor
    std_log_returns = log_returns.std()
    std_volatility = volatility.std()
    std_liquidity = liquidity.std()
    std_market_sentiment = market_sentiment.std()
    std_rsi = rsi.std()
    std_macd = macd.std()
    std_aroon_oscillator = aroon_oscillator.std()
    
    total_std = std_log_returns + std_volatility + std_liquidity + std_market_sentiment + std_rsi + std_macd + std_aroon_oscillator
    weight_log_returns = std_log_returns / total_std
    weight_volatility = std_volatility / total_std
    weight_liquidity = std_liquidity / total_std
    weight_market_sentiment = std_market_sentiment / total_std
    weight_rsi = std_rsi / total_std
    weight_macd = std_macd / total_std
    weight_aroon_oscillator = std_aroon_oscillator / total_std
    
    # Create a combined factor by balancing the log returns, volatility, liquidity, and market sentiment
    # The idea is that stocks with positive log returns, low volatility, high liquidity, strong market sentiment, 
    # favorable RSI, and positive MACD may outperform
    factor = (log_returns + abs(log_returns.min())) * weight_log_returns * (1 / (volatility + 1)) * weight_volatility * (liquidity / liquidity.mean()) * weight_liquidity * (market_sentiment / market_sentiment.mean()) * weight_market_sentiment * (rsi / 100) * weight_rsi * (macd + abs(macd.min())) * weight_macd * (aroon_oscillator / aroon_oscillator.abs().max()) * weight_aroon_oscillator
    
    return factor
```
