```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price normalized by the average of the true range (high-low) over 10 days
    avg_true_range_10 = df[['high', 'low']].rolling(window=10).apply(lambda x: (x['high']-x['low']).mean(), raw=False)
    factor = (df['close'] - df['open']) / (avg_true_range_10 + 1e-7)
    return factor
```
