```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate momentum indicators
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_20 = df['close'].pct_change(periods=20)
    
    # Calculate liquidity indicators
    liquidity = (df['high'] - df['low']) / df['volume']
    
    # Calculate volatility
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate the distance from the close price to the midpoint, which indicates the closing strength
    mid_point = (df['high'] + df['low']) / 2
    close_to_mid = abs(df['close'] - mid_point)
    
    # Calculate the money flow ratio using the amount (dollar volume) and volume
    money_flow_ratio = df['amount'] / df['volume']
    
    # Create a factor that combines momentum, liquidity, and volatility
    factor = (momentum_10 + momentum_20) * (1 / (liquidity + 1e-7)) * (1 / (volatility + 1e-7))
    
    # Add an interaction effect by considering the distance from the close to the midpoint, scaled by the money flow ratio
    factor *= ((df['close'] - df['low']) / (df['high'] - df['low'] + 1e-7)) * money_flow_ratio * (1 / (close_to_mid + 1e-7))

    return factor
```
