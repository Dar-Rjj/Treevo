```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 20-day price momentum as the ratio of today's close to the 20-day moving average of close prices
    df['price_momentum_20'] = df['close'] / df['close'].rolling(window=20).mean()
    
    # Calculate the 20-day volume trend as the 20-day moving average of volume
    df['volume_trend_20'] = df['volume'].rolling(window=20).mean()
    
    # Calculate the 20-day relative strength as the ratio of today's close to the 20-day minimum of close prices
    df['relative_strength_20'] = df['close'] / df['close'].rolling(window=20).min()
    
    # Calculate the 20-day volatility (ATR) over a 20-day period
    df['tr_20'] = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift()), abs(df['low'] - df['close'].shift())], axis=1).max(axis=1)
    df['atr_20'] = df['tr_20'].rolling(window=20).mean()
    
    # Calculate the 20-day buying pressure as (high - close) / (high - low + 1e-7), indicating strength of buying relative to range
    df['buying_pressure_20'] = (df['high'] - df['close']) / (df['high'] - df['low'] + 1e-7)
    
    # Combine the factors into a single alpha factor
    # The idea is that stocks with positive long-term momentum, high liquidity, and strong buying pressure may outperform
    alpha_factor = (df['price_momentum_20'] * df['relative_strength_20'] * df['buying_pressure_20']) * (df['volume_trend_20'] / df['atr_20'])
    
    return alpha_factor
```
