```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the recent momentum factor using the log return over a 10-day lookback period
    momentum_recent = np.log(df['close']).diff(periods=10)
    
    # Calculate the recent liquidity factor using the average volume over a 10-day lookback period
    liquidity_recent = df['volume'].rolling(window=10).mean()
    
    # Calculate the recent market sentiment using the average money flow ratio over a 5-day lookback period
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_recent = money_flow_ratio.rolling(window=5).mean()
    
    # Calculate the recent volatility factor using the standard deviation of the log returns over a 20-day lookback period
    volatility_recent = np.log(df['close']).diff().rolling(window=20).std()
    
    # Calculate the recent intraday dynamics factor using the ratio of high to low prices over the last 5 days
    intraday_dynamics_recent = (df['high'] / df['low']).rolling(window=5).mean()
    
    # Normalize each recent factor by dividing by its mean to give equal weightage
    momentum_normalized = (momentum_recent + 1) / (momentum_recent + 1).mean()
    liquidity_normalized = liquidity_recent / liquidity_recent.mean()
    market_sentiment_normalized = market_sentiment_recent / market_sentiment_recent.mean()
    volatility_normalized = (volatility_recent + 1e-6) / (volatility_recent + 1e-6).mean()
    intraday_dynamics_normalized = intraday_dynamics_recent / intraday_dynamics_recent.mean()
    
    # Create a combined factor by multiplying the normalized recent factors
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, lower volatility, and positive intraday dynamics may outperform
    factor = (momentum_normalized * liquidity_normalized * market_sentiment_normalized * intraday_dynamics_normalized) / volatility_normalized
    
    return factor
```
