```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using logarithmic returns over the last 10 days
    log_returns = np.log(df['close']).diff(periods=1)
    momentum = log_returns.rolling(window=10).sum()
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Calculate the volatility factor using the standard deviation of log returns over the last 30 days
    volatility = log_returns.rolling(window=30).std()
    
    # Incorporate a value factor using the ratio of the close price to the average amount over the last 10 days
    value = df['close'] / (df['amount'].rolling(window=10).mean())
    
    # Create a combined factor by multiplying the unnormalized factors
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, and low volatility may outperform
    factor = (momentum + 1) * (liquidity) * (market_sentiment) / (volatility + 1) * (value)
    
    return factor
```
