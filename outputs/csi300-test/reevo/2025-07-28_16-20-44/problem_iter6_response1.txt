```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day price momentum as the ratio of today's close to the 5-day moving average of close prices
    df['price_momentum_5'] = df['close'] / df['close'].rolling(window=5).mean()

    # Calculate the buying pressure as (high - close) / (high - low + 1e-7), indicating strength of buying relative to range
    df['buying_pressure'] = (df['high'] - df['close']) / (df['high'] - df['low'] + 1e-7)

    # Calculate the volume trend as the 5-day moving average of volume
    df['volume_trend_5'] = df['volume'].rolling(window=5).mean()

    # Calculate the relative strength as the ratio of today's close to the 5-day minimum of close prices
    df['relative_strength_5'] = df['close'] / df['close'].rolling(window=5).min()

    # Calculate the average true range (ATR) over a 5-day period
    df['tr_5'] = pd.concat([df['high'] - df['low'], abs(df['high'] - df['close'].shift()), abs(df['low'] - df['close'].shift())], axis=1).max(axis=1)
    df['atr_5'] = df['tr_5'].rolling(window=5).mean()

    # The alpha factor is the product of price momentum, buying pressure, and relative strength, weighted by volume trend and ATR
    alpha_factor = (df['price_momentum_5'] * df['buying_pressure'] * df['relative_strength_5']) * (df['volume_trend_5'] / df['atr_5'])

    return alpha_factor
```
