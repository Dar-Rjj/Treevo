```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day moving averages for close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()

    # Calculate the relative strength by comparing the short-term to long-term MA
    relative_strength = (ma_5 - ma_10) / ma_10

    # Calculate the volume-weighted average price (VWAP)
    vwap = (df['volume'] * (df['high'] + df['low']) / 2).cumsum() / df['volume'].cumsum()

    # Calculate the divergence between the close price and VWAP
    vwap_divergence = (df['close'] - vwap) / vwap

    # Calculate the 5-day and 10-day standard deviation for close price as a measure of volatility
    vol_5 = df['close'].rolling(window=5).std()
    vol_10 = df['close'].rolling(window=10).std()

    # Calculate the momentum indicator using the difference in close price over 5 days
    momentum = df['close'].diff(5)

    # Calculate the Chaikin Money Flow (CMF) over 20 periods
    money_flow_multiplier = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low'])
    money_flow_volume = money_flow_multiplier * df['volume']
    cmf = money_flow_volume.rolling(window=20).sum() / df['volume'].rolling(window=20).sum()

    # Combine the relative strength, VWAP divergence, volatility, momentum, and CMF into a single factor
    factor = relative_strength + vwap_divergence + (vol_5 - vol_10) + momentum + cmf

    return factor
```
