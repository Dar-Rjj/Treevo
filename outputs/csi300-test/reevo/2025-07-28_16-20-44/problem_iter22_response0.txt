```python
def heuristics_v2(df: pd.DataFrame, sector: str) -> pd.Series:
    # Dynamic lookback periods based on sector
    momentum_lookback = 20 if sector == 'Tech' else (30 if sector == 'Finance' else 25)
    liquidity_lookback = 40 if sector == 'Tech' else (50 if sector == 'Finance' else 35)
    market_sentiment_lookback = 15 if sector == 'Tech' else (10 if sector == 'Finance' else 7)
    volatility_lookback = 50 if sector == 'Tech' else (40 if sector == 'Finance' else 30)
    rsi_lookback = 14
    macd_ema_short = 12
    macd_ema_long = 26
    macd_signal = 9

    # Calculate the momentum factor using the log return over a dynamic lookback period
    momentum = np.log(df['close']).diff(periods=momentum_lookback)
    
    # Calculate the liquidity factor using the average volume over a dynamic lookback period
    liquidity = df['volume'].rolling(window=liquidity_lookback).mean()
    
    # Calculate the market sentiment using the average money flow ratio over a dynamic lookback period
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=market_sentiment_lookback).mean()
    
    # Calculate the volatility factor using the standard deviation of the log returns over a dynamic lookback period
    volatility = np.log(df['close']).diff().rolling(window=volatility_lookback).std()
    
    # Calculate the intraday dynamics factor using the ratio of high to low prices over the last 5 days
    intraday_dynamics = (df['high'] / df['low']).rolling(window=5).mean()
    
    # Calculate the RSI (Relative Strength Index) over the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=rsi_lookback).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=rsi_lookback).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the MACD (Moving Average Convergence Divergence)
    ema_12 = df['close'].ewm(span=macd_ema_short, adjust=False).mean()
    ema_26 = df['close'].ewm(span=macd_ema_long, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=macd_signal, adjust=False).mean()
    macd = macd_line - signal_line
    
    # Adaptive weights for each factor
    momentum_weight = 1 / (momentum.abs().mean() + 1e-6)
    liquidity_weight = 1 / (liquidity.mean() + 1e-6)
    market_sentiment_weight = 1 / (market_sentiment.mean() + 1e-6)
    volatility_weight = 1 / (volatility.mean() + 1e-6)
    intraday_dynamics_weight = 1 / (intraday_dynamics.mean() + 1e-6)
    rsi_weight = 1 / (rsi.mean() + 1e-6)
    macd_weight = 1 / (macd.abs().mean() + 1e-6)
    
    # Combine the factors with adaptive weights
    factor = (momentum * momentum_weight + 
              liquidity * liquidity_weight + 
              market_sentiment * market_sentiment_weight + 
              intraday_dynamics * intraday_dynamics_weight + 
              rsi * rsi_weight + 
              macd * macd_weight) / (volatility * volatility_weight)
    
    return factor
```
