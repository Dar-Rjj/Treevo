```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    
    # Calculate a weighted moving average of volume with window size 5, 
    # where more recent volumes have higher weights
    df['weighted_volume'] = df['volume'].rolling(window=5).apply(
        lambda x: (x * np.arange(1, 6)).sum() / x.sum(), raw=True
    )
    
    # Calculate the 5-day momentum
    df['momentum_5d'] = df['close'] - df['close'].shift(5)
    
    # Calculate the ratio of today's high to today's low as a liquidity measure
    df['liquidity'] = df['high'] / df['low']
    
    # Combine the factors into one alpha factor
    # Momentum per unit of traded interest, adjusted by liquidity
    alpha_factor = (df['close_diff'] + df['momentum_5d']) / (df['weighted_volume'] + 1e-7) * df['liquidity']
    
    return alpha_factor
```
