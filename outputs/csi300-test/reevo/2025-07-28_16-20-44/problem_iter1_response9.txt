```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then scale by average volume over the last 5 days.
    daily_return = (df['close'] - df['close'].shift(1)) / df['close'].shift(1)
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    factor = daily_return * avg_volume_5d
    return factor
```
