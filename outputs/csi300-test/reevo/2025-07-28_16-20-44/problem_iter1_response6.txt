```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate the average of the high and low price
    avg_high_low = (df['high'] + df['low']) / 2
    
    # Calculate a momentum-like factor based on the volume and amount
    volume_amount_momentum = (df['volume'].shift(1) * df['amount'].shift(1)) / (df['volume'] * df['amount'] + 1e-7)
    
    # Combine the factors into one alpha factor
    alpha_factor = close_open_diff * avg_high_low * volume_amount_momentum
    
    return alpha_factor
```
