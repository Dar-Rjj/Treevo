```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return over a dynamic lookback period
    momentum_period = 10 + (df['close'].rolling(window=5).std() * 100).astype(int)
    momentum = np.log(df['close']).diff(periods=momentum_period)
    
    # Calculate the liquidity factor using the average volume over a dynamic lookback period
    liquidity_period = 20 + (df['volume'].rolling(window=5).std() * 100).astype(int)
    liquidity = df['volume'].rolling(window=liquidity_period).mean()
    
    # Calculate the market sentiment using the average money flow ratio over a dynamic lookback period
    money_flow_ratio = df['amount'] / df['volume']
    sentiment_period = 5 + (money_flow_ratio.rolling(window=5).std() * 100).astype(int)
    market_sentiment = money_flow_ratio.rolling(window=sentiment_period).mean()
    
    # Calculate the volatility factor using the standard deviation of the log returns over the last 30 days
    volatility = np.log(df['close']).diff().rolling(window=30).std()
    
    # Combine factors with dynamic weights based on their predictive power
    momentum_weight = 0.4
    liquidity_weight = 0.3
    market_sentiment_weight = 0.2
    volatility_weight = 0.1
    
    # Create a combined factor
    factor = (momentum * momentum_weight +
              liquidity * liquidity_weight +
              market_sentiment * market_sentiment_weight -
              volatility * volatility_weight)
    
    return factor
```
