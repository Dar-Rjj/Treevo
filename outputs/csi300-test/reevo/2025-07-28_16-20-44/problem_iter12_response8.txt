```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns over the last 10 days
    log_returns = np.log(df['close']).diff(periods=10)
    
    # Calculate the volatility of the log returns over the last 30 days
    volatility = log_returns.rolling(window=30).std()
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Calculate the RSI (Relative Strength Index) over the last 14 days
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the MACD (Moving Average Convergence Divergence)
    ema_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_12 - ema_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd = macd_line - signal_line
    
    # Calculate the Accumulation/Distribution Line (ADL)
    adl = ((df['close'] - df['low']) - (df['high'] - df['close'])) / (df['high'] - df['low']) * df['volume']
    adl = adl.cumsum()
    
    # Calculate the On-Balance Volume (OBV)
    obv = (np.sign(df['close'].diff()) * df['volume']).cumsum()
    
    # Dynamic weights based on the standard deviation of each factor
    weight_log_returns = 1 / (volatility + 1)
    weight_volatility = 1 / (volatility + 1)
    weight_liquidity = liquidity / liquidity.mean()
    weight_market_sentiment = market_sentiment / market_sentiment.mean()
    weight_rsi = rsi / 100
    weight_macd = macd + abs(macd.min())
    weight_adl = (adl - adl.mean()) / adl.std()
    weight_obv = (obv - obv.mean()) / obv.std()
    
    # Create a combined factor by balancing the log returns, volatility, liquidity, and market sentiment
    # The idea is that stocks with positive log returns, low volatility, high liquidity, strong market sentiment,
    # favorable RSI, positive MACD, and strong ADL/ OBV may outperform
    factor = (log_returns * weight_log_returns) * (1 / (volatility * weight_volatility)) * (liquidity * weight_liquidity) * (market_sentiment * weight_market_sentiment) * (rsi * weight_rsi) * (macd * weight_macd) * (adl * weight_adl) * (obv * weight_obv)
    
    return factor
```
