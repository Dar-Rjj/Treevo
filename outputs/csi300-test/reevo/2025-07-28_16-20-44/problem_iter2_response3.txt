```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the midpoint of the day's trading range
    mid_point = (df['high'] + df['low']) / 2
    
    # Calculate the money flow ratio using the amount (dollar volume) and volume
    money_flow_ratio = df['amount'] / df['volume']
    
    # Calculate the distance from the close price to the midpoint, which indicates the closing strength
    close_to_mid = abs(df['close'] - mid_point)
    
    # Create an interaction effect factor by combining the normalized distance from the low to the close,
    # scaled by the money flow ratio.
    # The idea is that a higher close in relation to the day's range, combined with strong money flow,
    # could be indicative of stronger demand.
    factor = ((df['close'] - df['low']) / (df['high'] - df['low'] + 1e-7)) * money_flow_ratio * (1 / (close_to_mid + 1e-7))

    return factor
```
