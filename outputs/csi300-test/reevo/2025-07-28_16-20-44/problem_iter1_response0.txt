```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    delta_close_open = df['close'] - df['open']
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the standard deviation of VWAP over a 5-day window
    vwap_std = (df['amount'] / df['volume']).rolling(window=5).std()
    
    # Alpha factor: (Close - Open) normalized by the standard deviation of VWAP
    alpha_factor = delta_close_open / (vwap_std + 1e-7)

    return alpha_factor
```
