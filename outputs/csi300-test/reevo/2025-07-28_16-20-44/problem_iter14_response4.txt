```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return over a dynamic lookback period (15-30 days)
    momentum_15 = np.log(df['close']).diff(periods=15)
    momentum_20 = np.log(df['close']).diff(periods=20)
    momentum_30 = np.log(df['close']).diff(periods=30)
    momentum_combined = (momentum_15 + momentum_20 + momentum_30) / 3

    # Calculate the liquidity factor using the average volume over a dynamic lookback period (15-40 days)
    liquidity_15 = df['volume'].rolling(window=15).mean()
    liquidity_30 = df['volume'].rolling(window=30).mean()
    liquidity_40 = df['volume'].rolling(window=40).mean()
    liquidity_combined = (liquidity_15 + liquidity_30 + liquidity_40) / 3

    # Calculate the market sentiment using the average money flow ratio over a dynamic lookback period (7-14 days)
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_7 = money_flow_ratio.rolling(window=7).mean()
    market_sentiment_14 = money_flow_ratio.rolling(window=14).mean()
    market_sentiment_combined = (market_sentiment_7 + market_sentiment_14) / 2

    # Calculate the volatility factor using the standard deviation of the log returns over a dynamic lookback period (20-60 days)
    volatility_20 = np.log(df['close']).diff().rolling(window=20).std()
    volatility_40 = np.log(df['close']).diff().rolling(window=40).std()
    volatility_60 = np.log(df['close']).diff().rolling(window=60).std()
    volatility_combined = (volatility_20 + volatility_40 + volatility_60) / 3

    # Calculate the intraday dynamics factor using the ratio of high to low prices over the last 5-10 days
    intraday_dynamics_5 = (df['high'] / df['low']).rolling(window=5).mean()
    intraday_dynamics_10 = (df['high'] / df['low']).rolling(window=10).mean()
    intraday_dynamics_combined = (intraday_dynamics_5 + intraday_dynamics_10) / 2

    # Weight factors by their predictive power (weights are based on historical performance and can be adjusted)
    momentum_weight = 0.35
    liquidity_weight = 0.20
    market_sentiment_weight = 0.20
    volatility_weight = -0.15
    intraday_dynamics_weight = 0.10

    # Combine the factors with their respective weights
    factor = (momentum_combined * momentum_weight + 
              liquidity_combined * liquidity_weight + 
              market_sentiment_combined * market_sentiment_weight + 
              intraday_dynamics_combined * intraday_dynamics_weight + 
              volatility_combined * volatility_weight)

    return factor
```
