```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the change in close price over the last 10 days
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Calculate the value factor using the average of high and low prices over the last 20 days
    value_factor = (df['high'] + df['low']) / 2
    value_factor = value_factor.rolling(window=20).mean()
    
    # Incorporate a macroeconomic indicator, such as a moving average of an interest rate (assuming it's available in the DataFrame)
    interest_rate_moving_avg = df['interest_rate'].rolling(window=20).mean()
    
    # Create a combined factor by multiplying the normalized momentum, liquidity, market sentiment, and value factor
    # Also, include the macroeconomic indicator to adjust the factor based on the current economic condition
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, and favorable value may outperform, adjusted for economic conditions
    factor = (momentum + 1) * (liquidity / liquidity.mean()) * (market_sentiment / market_sentiment.mean()) * (value_factor / value_factor.mean()) * (interest_rate_moving_avg / interest_rate_moving_avg.mean())
    
    return factor
```
