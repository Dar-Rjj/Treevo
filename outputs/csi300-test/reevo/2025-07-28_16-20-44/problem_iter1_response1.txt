```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and previous day's close
    df['close_diff'] = df['close'].diff()
    # Calculate a weighted moving average of volume with window size 5, where more recent volumes have higher weights
    df['weighted_volume'] = df['volume'].rolling(window=5).apply(lambda x: (x * 5).sum() / x.sum(), raw=True)
    # Alpha factor: ratio of close difference to weighted volume, which could capture momentum per unit of traded interest
    alpha_factor = df['close_diff'] / (df['weighted_volume'] + 1e-7)
    return alpha_factor
```
