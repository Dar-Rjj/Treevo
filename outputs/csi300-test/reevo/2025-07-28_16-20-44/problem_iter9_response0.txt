```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the multi-period log returns for the last 5 and 10 days
    log_return_5 = np.log(df['close']).diff(periods=5)
    log_return_10 = np.log(df['close']).diff(periods=10)
    
    # Calculate the momentum factor using the average of the 5 and 10 day log returns
    momentum = (log_return_5 + log_return_10) / 2
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Calculate the volatility as the standard deviation of the last 20 days' close prices
    volatility = df['close'].rolling(window=20).std()
    
    # Create a combined factor by multiplying the momentum, liquidity, and market sentiment, and dividing by volatility
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, and lower volatility may outperform
    factor = (momentum + 1) * (liquidity / df['volume'].mean()) * (market_sentiment / money_flow_ratio.mean()) / (volatility / df['close'].std())
    
    return factor
```
