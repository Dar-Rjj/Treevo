```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the change in close price over the last 10 days
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the volatility factor using the standard deviation of the close price over the last 20 days
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Create a combined factor by dynamically weighting the momentum, volatility, liquidity, and market sentiment
    # The idea is that stocks with positive momentum, low volatility, high liquidity, and strong market sentiment may outperform
    factor = (momentum + 1) * (1 / (volatility + 1)) * (liquidity / liquidity.mean()) * (market_sentiment / market_sentiment.mean())
    
    return factor
```
