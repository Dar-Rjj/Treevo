```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the change in close price over the last 10 days
    momentum = df['close'].pct_change(periods=10)
    
    # Calculate the volatility factor using the standard deviation of the close price over the last 20 days
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Create a dynamic weighting for each factor
    momentum_weight = 1 / (momentum.abs() + 1e-6)
    volatility_weight = 1 / (volatility + 1e-6)
    liquidity_weight = 1 / (liquidity + 1e-6)
    sentiment_weight = 1 / (market_sentiment + 1e-6)
    
    # Normalize the weights
    total_weight = momentum_weight + volatility_weight + liquidity_weight + sentiment_weight
    momentum_weight /= total_weight
    volatility_weight /= total_weight
    liquidity_weight /= total_weight
    sentiment_weight /= total_weight
    
    # Combine the factors using the dynamic weights
    factor = (momentum * momentum_weight) + (volatility * -volatility_weight) + (liquidity * liquidity_weight) + (market_sentiment * sentiment_weight)
    
    return factor
```
