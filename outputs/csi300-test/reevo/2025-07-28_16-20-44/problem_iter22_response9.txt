```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return over a dynamic lookback period (10-30 days) with EMA smoothing
    momentum_10 = np.log(df['close']).diff(periods=10).ewm(span=10, adjust=False).mean()
    momentum_20 = np.log(df['close']).diff(periods=20).ewm(span=20, adjust=False).mean()
    momentum_30 = np.log(df['close']).diff(periods=30).ewm(span=30, adjust=False).mean()
    momentum = (momentum_10 + momentum_20 + momentum_30) / 3
    
    # Calculate the liquidity factor using the exponential moving average volume over a dynamic lookback period (10-40 days)
    liquidity_10 = df['volume'].ewm(span=10, adjust=False).mean()
    liquidity_20 = df['volume'].ewm(span=20, adjust=False).mean()
    liquidity_30 = df['volume'].ewm(span=30, adjust=False).mean()
    liquidity_40 = df['volume'].ewm(span=40, adjust=False).mean()
    liquidity = (liquidity_10 + liquidity_20 + liquidity_30 + liquidity_40) / 4
    
    # Calculate the market sentiment using the exponential moving average money flow ratio over a dynamic lookback period (5-15 days)
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_5 = money_flow_ratio.ewm(span=5, adjust=False).mean()
    market_sentiment_10 = money_flow_ratio.ewm(span=10, adjust=False).mean()
    market_sentiment_15 = money_flow_ratio.ewm(span=15, adjust=False).mean()
    market_sentiment = (market_sentiment_5 + market_sentiment_10 + market_sentiment_15) / 3
    
    # Calculate the volatility factor using the standard deviation of the log returns over a dynamic lookback period (20-60 days) with EMA
    volatility_20 = np.log(df['close']).diff().ewm(span=20, adjust=False).std()
    volatility_40 = np.log(df['close']).diff().ewm(span=40, adjust=False).std()
    volatility_60 = np.log(df['close']).diff().ewm(span=60, adjust=False).std()
    volatility = (volatility_20 + volatility_40 + volatility_60) / 3
    
    # Calculate the intraday dynamics factor using the ratio of high to low prices over the last 5-10 days with EMA
    intraday_dynamics_5 = (df['high'] / df['low']).ewm(span=5, adjust=False).mean()
    intraday_dynamics_10 = (df['high'] / df['low']).ewm(span=10, adjust=False).mean()
    intraday_dynamics = (intraday_dynamics_5 + intraday_dynamics_10) / 2
    
    # Adaptive, inversely proportional weights for each factor
    momentum_weight = 1 / (momentum.abs().mean() + 1e-6)
    liquidity_weight = 1 / (liquidity.mean() + 1e-6)
    market_sentiment_weight = 1 / (market_sentiment.mean() + 1e-6)
    volatility_weight = 1 / (volatility.mean() + 1e-6)
    intraday_dynamics_weight = 1 / (intraday_dynamics.mean() + 1e-6)
    
    # Combine the factors with adaptive, inversely proportional weights
    factor = (momentum * momentum_weight + 
              liquidity * liquidity_weight + 
              market_sentiment * market_sentiment_weight + 
              intraday_dynamics * intraday_dynamics_weight) / (volatility * volatility_weight)
    
    return factor
```
