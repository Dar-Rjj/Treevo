```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change from previous day's close to today's open
    overnight_change = df['open'] - df['close'].shift(1)
    
    # Calculate the average trading volume over the last 5 days
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the 5-day average volume
    volume_ratio = df['volume'] / (avg_volume_5d + 1e-7)
    
    # Calculate the price change from today's open to today's close
    intraday_change = df['close'] - df['open']
    
    # Combine the factors into one alpha factor
    alpha_factor = overnight_change * volume_ratio * intraday_change
    
    return alpha_factor
```
