```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the change in close price over the last 10, 30, and 60 days
    momentum_short = df['close'].pct_change(periods=10)
    momentum_medium = df['close'].pct_change(periods=30)
    momentum_long = df['close'].pct_change(periods=60)
    momentum = (momentum_short + momentum_medium + momentum_long) / 3

    # Calculate the liquidity factor using the average volume over the last 20 and 50 days
    liquidity_short = df['volume'].rolling(window=20).mean()
    liquidity_long = df['volume'].rolling(window=50).mean()
    liquidity = (liquidity_short + liquidity_long) / 2

    # Calculate the market sentiment using the average money flow ratio over the last 5, 10, and 20 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_short = money_flow_ratio.rolling(window=5).mean()
    market_sentiment_medium = money_flow_ratio.rolling(window=10).mean()
    market_sentiment_long = money_flow_ratio.rolling(window=20).mean()
    market_sentiment = (market_sentiment_short + market_sentiment_medium + market_sentiment_long) / 3

    # Calculate the volatility factor using the rolling standard deviation of the close price over the last 20 and 50 days
    volatility_short = df['close'].rolling(window=20).std()
    volatility_long = df['close'].rolling(window=50).std()
    volatility = (volatility_short + volatility_long) / 2

    # Create a combined factor by combining the momentum, liquidity, market sentiment, and inversely the volatility
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, and low volatility may outperform
    factor = (momentum + 1) * (liquidity / liquidity.mean()) * (market_sentiment / market_sentiment.mean()) * (1 / (volatility / volatility.mean()))

    return factor
```
