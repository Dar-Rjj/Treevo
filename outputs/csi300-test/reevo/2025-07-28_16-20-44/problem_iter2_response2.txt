def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    delta_close_open = df['close'] - df['open']
    
    # Calculate the volume weighted average price (VWAP)
    vwap = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Calculate the 5-day standard deviation of the close prices as a measure of volatility
    close_std = df['close'].rolling(window=5).std()
    
    # Alpha factor: (Close - Open) normalized by the 5-day standard deviation of the close prices
    alpha_factor = delta_close_open / (close_std + 1e-7)

    return alpha_factor
