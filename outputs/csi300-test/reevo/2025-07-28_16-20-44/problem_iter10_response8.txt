def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return over a dynamic lookback period (10 to 30 days)
    momentum_10 = np.log(df['close']).diff(periods=10)
    momentum_20 = np.log(df['close']).diff(periods=20)
    momentum_30 = np.log(df['close']).diff(periods=30)
    
    # Combine the momentum factors with a weighted average
    momentum = (momentum_10 * 0.4 + momentum_20 * 0.3 + momentum_30 * 0.3)
    
    # Calculate the liquidity factor using the average volume over a dynamic lookback period (10 to 30 days)
    liquidity_10 = df['volume'].rolling(window=10).mean()
    liquidity_20 = df['volume'].rolling(window=20).mean()
    liquidity_30 = df['volume'].rolling(window=30).mean()
    
    # Combine the liquidity factors with a weighted average
    liquidity = (liquidity_10 * 0.4 + liquidity_20 * 0.3 + liquidity_30 * 0.3)
    
    # Calculate the market sentiment using the average money flow ratio over a dynamic lookback period (5 to 10 days)
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_5 = money_flow_ratio.rolling(window=5).mean()
    market_sentiment_10 = money_flow_ratio.rolling(window=10).mean()
    
    # Combine the market sentiment factors with a weighted average
    market_sentiment = (market_sentiment_5 * 0.6 + market_sentiment_10 * 0.4)
    
    # Calculate the volatility factor using the standard deviation of the log returns over a dynamic lookback period (10 to 30 days)
    volatility_10 = np.log(df['close']).diff().rolling(window=10).std()
    volatility_20 = np.log(df['close']).diff().rolling(window=20).std()
    volatility_30 = np.log(df['close']).diff().rolling(window=30).std()
    
    # Combine the volatility factors with a weighted average
    volatility = (volatility_10 * 0.4 + volatility_20 * 0.3 + volatility_30 * 0.3)
    
    # Create a combined factor by multiplying the momentum, liquidity, and market sentiment, and dividing by volatility
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, and lower volatility may outperform
    factor = ((momentum + 1) * (liquidity / liquidity.mean()) * (market_sentiment / market_sentiment.mean()) / (volatility + 1e-6))
    
    return factor
