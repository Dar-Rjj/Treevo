```python
import numpy as np
import pandas as pd

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return with exponential decay
    def exp_decay_momentum(data, span):
        return data.ewm(span=span, adjust=False).mean()
    
    momentum_10 = np.log(df['close']).diff().transform(lambda x: exp_decay_momentum(x, 10))
    momentum_20 = np.log(df['close']).diff().transform(lambda x: exp_decay_momentum(x, 20))
    momentum_30 = np.log(df['close']).diff().transform(lambda x: exp_decay_momentum(x, 30))
    momentum = (momentum_10 + momentum_20 + momentum_30) / 3
    
    # Calculate the liquidity factor using the average volume with exponential decay
    liquidity_10 = df['volume'].transform(lambda x: exp_decay_momentum(x, 10))
    liquidity_20 = df['volume'].transform(lambda x: exp_decay_momentum(x, 20))
    liquidity_30 = df['volume'].transform(lambda x: exp_decay_momentum(x, 30))
    liquidity_40 = df['volume'].transform(lambda x: exp_decay_momentum(x, 40))
    liquidity = (liquidity_10 + liquidity_20 + liquidity_30 + liquidity_40) / 4
    
    # Calculate the market sentiment using the average money flow ratio with exponential decay
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_5 = money_flow_ratio.transform(lambda x: exp_decay_momentum(x, 5))
    market_sentiment_10 = money_flow_ratio.transform(lambda x: exp_decay_momentum(x, 10))
    market_sentiment_15 = money_flow_ratio.transform(lambda x: exp_decay_momentum(x, 15))
    market_sentiment = (market_sentiment_5 + market_sentiment_10 + market_sentiment_15) / 3
    
    # Calculate the volatility factor using the standard deviation of the log returns with exponential decay
    volatility_20 = np.log(df['close']).diff().transform(lambda x: x.ewm(span=20, adjust=False).std())
    volatility_40 = np.log(df['close']).diff().transform(lambda x: x.ewm(span=40, adjust=False).std())
    volatility_60 = np.log(df['close']).diff().transform(lambda x: x.ewm(span=60, adjust=False).std())
    volatility = (volatility_20 + volatility_40 + volatility_60) / 3
    
    # Calculate the intraday dynamics factor using the ratio of high to low prices with exponential decay
    intraday_dynamics_5 = (df['high'] / df['low']).transform(lambda x: exp_decay_momentum(x, 5))
    intraday_dynamics_10 = (df['high'] / df['low']).transform(lambda x: exp_decay_momentum(x, 10))
    intraday_dynamics = (intraday_dynamics_5 + intraday_dynamics_10) / 2
    
    # Combine the factors using a weighted sum
    factor = (momentum * 0.3 + liquidity * 0.2 + market_sentiment * 0.2 + intraday_dynamics * 0.2) / (volatility + 1e-6)
    
    return factor
```
