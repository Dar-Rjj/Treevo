```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low
    range_diff = df['high'] - df['low']
    
    # Calculate the buying pressure as the amount traded at the close price divided by the total volume
    buying_pressure = (df['close'] * df['volume']) / df['amount']
    
    # Calculate the factor as a combination of the range difference and buying pressure
    factor = (buying_pressure - 1) * range_diff
    return factor
```
