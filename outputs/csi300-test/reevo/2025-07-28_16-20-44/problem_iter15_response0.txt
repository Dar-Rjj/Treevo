```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return over a dynamic lookback period (10-30 days)
    momentum = np.log(df['close']).diff(periods=20).fillna(0)
    
    # Calculate the liquidity factor using the average volume over a dynamic lookback period (10-40 days)
    liquidity = df['volume'].rolling(window=30).mean().fillna(0)
    
    # Calculate the market sentiment using the average money flow ratio over a dynamic lookback period (5-15 days)
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=10).mean().fillna(0)
    
    # Calculate the volatility factor using the standard deviation of the log returns over a dynamic lookback period (20-50 days)
    volatility = np.log(df['close']).diff().rolling(window=40).std().fillna(0)
    
    # Calculate the intraday dynamics factor using the ratio of high to low prices over the last 5 days
    intraday_dynamics = (df['high'] / df['low']).rolling(window=5).mean().fillna(1)
    
    # Adjust the factor weights dynamically based on recent performance
    momentum_weight = 1.0
    liquidity_weight = 1.0
    market_sentiment_weight = 1.0
    volatility_weight = 1.0
    intraday_dynamics_weight = 1.0
    
    # Create a combined factor by combining the factors with their respective weights
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, lower volatility, and positive intraday dynamics may outperform
    factor = (momentum * momentum_weight + 
              liquidity * liquidity_weight + 
              market_sentiment * market_sentiment_weight + 
              intraday_dynamics * intraday_dynamics_weight) / (volatility * volatility_weight + 1e-6)
    
    return factor
```
