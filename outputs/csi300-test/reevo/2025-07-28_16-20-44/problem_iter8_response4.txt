```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using logarithmic returns over the last 10 days
    momentum = (df['close'].pct_change(periods=10) + 1).apply(np.log)
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Calculate the volatility factor using the standard deviation of the last 20 days' returns
    volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Create a combined factor by incorporating momentum, liquidity, market sentiment, and volatility
    # The idea is that stocks with positive momentum, high liquidity, strong market sentiment, and lower volatility may outperform
    factor = momentum * (liquidity / liquidity.mean()) * (market_sentiment / market_sentiment.mean()) / volatility
    
    return factor
```
