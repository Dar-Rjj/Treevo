```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the log returns over the last 10 days
    log_returns = np.log(df['close']).diff(periods=10)
    
    # Calculate the volatility of the log returns over the last 30 days
    volatility = log_returns.rolling(window=30).std()
    
    # Calculate the liquidity factor using the average volume over the last 20 days
    liquidity = df['volume'].rolling(window=20).mean()
    
    # Calculate the market sentiment using the average money flow ratio over the last 5 days
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment = money_flow_ratio.rolling(window=5).mean()
    
    # Create a combined factor by balancing the log returns, volatility, liquidity, and market sentiment
    # The idea is that stocks with positive log returns, low volatility, high liquidity, and strong market sentiment may outperform
    factor = (log_returns + abs(log_returns.min())) * (1 / volatility) * (liquidity / liquidity.mean()) * (market_sentiment / market_sentiment.mean())
    
    return factor
```
