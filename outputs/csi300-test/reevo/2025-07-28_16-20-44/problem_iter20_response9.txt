def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the log return over a dynamic lookback period (10-30 days)
    momentum_10 = np.log(df['close']).diff(periods=10)
    momentum_20 = np.log(df['close']).diff(periods=20)
    momentum_30 = np.log(df['close']).diff(periods=30)
    momentum = (momentum_10 + momentum_20 + momentum_30) / 3
    
    # Calculate the liquidity factor using the average volume over a dynamic lookback period (10-40 days)
    liquidity_10 = df['volume'].rolling(window=10).mean()
    liquidity_20 = df['volume'].rolling(window=20).mean()
    liquidity_30 = df['volume'].rolling(window=30).mean()
    liquidity_40 = df['volume'].rolling(window=40).mean()
    liquidity = (liquidity_10 + liquidity_20 + liquidity_30 + liquidity_40) / 4
    
    # Calculate the market sentiment using the average money flow ratio over a dynamic lookback period (5-15 days)
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_5 = money_flow_ratio.rolling(window=5).mean()
    market_sentiment_10 = money_flow_ratio.rolling(window=10).mean()
    market_sentiment_15 = money_flow_ratio.rolling(window=15).mean()
    market_sentiment = (market_sentiment_5 + market_sentiment_10 + market_sentiment_15) / 3
    
    # Calculate the volatility factor using the standard deviation of the log returns over a dynamic lookback period (20-50 days)
    volatility_20 = np.log(df['close']).diff().rolling(window=20).std()
    volatility_40 = np.log(df['close']).diff().rolling(window=40).std()
    volatility_60 = np.log(df['close']).diff().rolling(window=60).std()
    volatility = (volatility_20 + volatility_40 + volatility_60) / 3
    
    # Calculate the intraday dynamics factor using the ratio of high to low prices over the last 5-10 days
    intraday_dynamics_5 = (df['high'] / df['low']).rolling(window=5).mean()
    intraday_dynamics_10 = (df['high'] / df['low']).rolling(window=10).mean()
    intraday_dynamics = (intraday_dynamics_5 + intraday_dynamics_10) / 2
    
    # Define dynamic weights for each factor
    weight_momentum = 0.3 * (momentum.ewm(span=10, adjust=False).mean() + 1)
    weight_liquidity = 0.2 * (liquidity.ewm(span=20, adjust=False).mean() / liquidity.mean())
    weight_market_sentiment = 0.2 * (market_sentiment.ewm(span=15, adjust=False).mean() / market_sentiment.mean())
    weight_intraday_dynamics = 0.2 * (intraday_dynamics.ewm(span=10, adjust=False).mean() / intraday_dynamics.mean())
    weight_volatility = 0.1 * (1 / (volatility.ewm(span=30, adjust=False).mean() + 1e-6))
    
    # Combine factors with dynamic weights
    combined_factor = (weight_momentum * (momentum + 1) + 
                       weight_liquidity * liquidity + 
                       weight_market_sentiment * market_sentiment + 
                       weight_intraday_dynamics * intraday_dynamics) / (weight_volatility * (volatility + 1e-6))
    
    # Apply alternative smoothing technique (Holt-Winters Exponential Smoothing)
    smoothed_factor = combined_factor.ewm(span=10, adjust=False).mean().ewm(span=10, adjust=False).mean()
    
    return smoothed_factor
