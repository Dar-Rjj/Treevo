def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum factor using the change in close price over the last 10, 30, and 60 days with weights
    momentum_short = df['close'].pct_change(periods=10)
    momentum_medium = df['close'].pct_change(periods=30)
    momentum_long = df['close'].pct_change(periods=60)
    momentum = (momentum_short * 0.5 + momentum_medium * 0.3 + momentum_long * 0.2)

    # Calculate the liquidity factor using the average volume over the last 20 and 50 days with weights
    liquidity_short = df['volume'].rolling(window=20).mean()
    liquidity_long = df['volume'].rolling(window=50).mean()
    liquidity = (liquidity_short * 0.6 + liquidity_long * 0.4)

    # Calculate the market sentiment using the average money flow ratio over the last 5, 10, and 20 days with weights
    money_flow_ratio = df['amount'] / df['volume']
    market_sentiment_short = money_flow_ratio.rolling(window=5).mean()
    market_sentiment_medium = money_flow_ratio.rolling(window=10).mean()
    market_sentiment_long = money_flow_ratio.rolling(window=20).mean()
    market_sentiment = (market_sentiment_short * 0.4 + market_sentiment_medium * 0.3 + market_sentiment_long * 0.3)

    # Calculate the volatility factor using the rolling standard deviation of the close price over the last 20 and 50 days with weights
    volatility_short = df['close'].rolling(window=20).std()
    volatility_long = df['close'].rolling(window=50).std()
    volatility = (volatility_short * 0.5 + volatility_long * 0.5)

    # Calculate the trading range as (high - low) / close, indicating the daily price range relative to the closing price
    trading_range = (df['high'] - df['low']) / df['close']

    # Calculate the return skew as the skewness of the daily returns over the last 30 days
    return_skew = df['close'].pct_change().rolling(window=30).skew()

    # Create a combined factor by combining the momentum, liquidity, market sentiment, and inversely the volatility
    # Additionally, include the trading range and return skew to capture intraday market dynamics and asymmetry in returns
    factor = (momentum + 1) * (liquidity / liquidity.mean()) * (market_sentiment / market_sentiment.mean()) * (1 / (volatility / volatility.mean())) * (1 / (trading_range / trading_range.mean())) * (1 / (return_skew / return_skew.abs().mean()))

    return factor
