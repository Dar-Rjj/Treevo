def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by today's volume
    momentum_adjusted = (df['close'].diff() / df['volume']).shift(1)
    # Calculate the weighted price using amount and volume to identify the money flow
    money_flow = (df['amount'] / df['volume']).rolling(window=3).mean()
    # Calculate the relative strength index (RSI) of the close prices with a 14-day period
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    # Combine momentum_adjusted, money_flow, and RSI into a single alpha factor
    alpha_factor = (momentum_adjusted + money_flow + rsi) / 3
    # Smooth the alpha factor with a 5-period simple moving average
    alpha_factor_smoothed = alpha_factor.rolling(window=5).mean()
    return alpha_factor_smoothed
