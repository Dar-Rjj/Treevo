```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price to understand the general trend for the day
    daily_price_diff = df['close'] - df['open']
    
    # Calculate an average of 'high' prices over a 5-day window to understand the recent peak prices
    high_rolling_avg = df['high'].rolling(window=5).mean()
    
    # Calculate the change in volume compared to the previous day to capture interest in the stock
    volume_change = df['volume'] - df['volume'].shift(1)
    
    # The alpha factor is computed as the weighted sum of these metrics, capturing both short-term momentum and long-term trend
    alpha_factor = (daily_price_diff + high_rolling_avg.diff() + 0.5 * volume_change) / 3
    return alpha_factor
```
