```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by today's volume
    momentum_adjusted = (df['close'].diff() / df['volume']).shift(1)
    
    # Calculate the weighted price using amount and volume to identify the money flow
    money_flow = (df['amount'] / df['volume']).rolling(window=3).mean()
    
    # Calculate the ratio of the current day's range to the 5-day average range
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=5).mean()
    range_ratio = daily_range / avg_daily_range
    
    # Calculate the 5-day rolling standard deviation of the daily range for volatility
    daily_range_volatility = daily_range.rolling(window=5).std()
    
    # Incorporate the volatility into the range ratio to adjust for market conditions
    adjusted_range_ratio = range_ratio / (daily_range_volatility + 1e-7)
    
    # Calculate the 5-day return to capture momentum
    five_day_return = df['close'].pct_change(periods=5)
    
    # Combine all factors with a simple moving average of 5 periods to smooth out the noise
    alpha_factor = ((momentum_adjusted + money_flow + adjusted_range_ratio + five_day_return) / 4).rolling(window=5).mean()
    
    return alpha_factor
```
