```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of the close price to capture momentum
    ma_10 = df['close'].rolling(window=10).mean()
    # Calculate the 30-day moving average of the close price for a longer-term trend
    ma_30 = df['close'].rolling(window=30).mean()
    # Momentum factor as the difference between the 10-day and 30-day moving averages
    momentum_factor = ma_10 - ma_30
    
    # Calculate the daily return
    daily_return = df['close'].pct_change()
    # Calculate the 10-day rolling standard deviation of returns to capture volatility
    volatility = daily_return.rolling(window=10).std()
    
    # Volume-weighted close price
    vw_close = df['close'] * df['volume']
    # Average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    # Factor as the ratio of the volume-weighted close to the 10-day average volume
    volume_factor = (vw_close / (avg_volume + 1e-7)).fillna(0)
    
    # Combine the momentum, volatility, and volume factors
    factor = momentum_factor + (volatility * 10) + (volume_factor * 5)
    
    return factor
```
