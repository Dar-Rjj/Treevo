```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum with a 5-day and 10-day window, then take the average
    momentum_5 = df['close'].pct_change(5)
    momentum_10 = df['close'].pct_change(10)
    momentum = (momentum_5 + momentum_10) / 2

    # Calculate the volume-weighted price (VWP) using a 5-day window
    vwp = (df['amount'] / df['volume']).rolling(window=5).mean()

    # Combine momentum and VWP to create a volume-adjusted momentum
    vol_adj_momentum = momentum * vwp

    # Calculate the ratio of the current day's range to the 10-day average range
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=10).mean()
    range_ratio = daily_range / avg_daily_range

    # Integrate volatility by calculating the standard deviation of the close prices over a 10-day window
    volatility = df['close'].rolling(window=10).std()

    # Dynamic weighting: use the inverse of volatility to weight the factors
    inv_volatility = 1 / volatility

    # Combine all factors with dynamic weighting and apply a 5-period simple moving average for smoothing
    alpha_factor = ((vol_adj_momentum + range_ratio) * inv_volatility).rolling(window=5).mean()

    return alpha_factor
```
