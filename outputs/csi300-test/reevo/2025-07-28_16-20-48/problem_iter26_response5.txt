```python
import pandas as pd
import numpy as np

def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponential moving average of close prices for a 10-day window
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the momentum factor: difference between today's close and the EMA value, then divide by today's volume
    momentum_adjusted = (df['close'] - ema_10) / df['volume']
    
    # Calculate the money flow over a 5-day window
    money_flow_5 = (df['amount'] / df['volume']).rolling(window=5).mean()
    
    # Adaptive volatility: use the standard deviation of the last 10 days
    adaptive_volatility = df['close'].rolling(window=10).std()
    
    # Additional feature: market sentiment (e.g., percentage change in volume)
    volume_change = df['volume'].pct_change().fillna(0)
    
    # Incorporate logarithmic returns as a measure of robust, responsive signals
    log_returns = np.log(df['close'] / df['close'].shift(1)).fillna(0)
    
    # Dynamic weights based on the adaptive volatility
    weight_momentum = 1 / (1 + adaptive_volatility)
    weight_money_flow = 0.5 / (1 + adaptive_volatility)
    weight_volume_change = 0.3 / (1 + adaptive_volatility)
    weight_log_returns = 0.7 / (1 + adaptive_volatility)
    
    # Combine all factors with a weighted sum
    alpha_factor = (weight_momentum * momentum_adjusted +
                    weight_money_flow * money_flow_5 +
                    weight_volume_change * volume_change +
                    weight_log_returns * log_returns)
    
    # Smooth the alpha factor using an exponential moving average with a span of 5
    alpha_factor_smoothed = alpha_factor.ewm(span=5, adjust=False).mean()
    
    return alpha_factor_smoothed
```
