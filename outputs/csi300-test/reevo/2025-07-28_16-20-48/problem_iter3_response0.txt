```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum by looking at the change in close price over a 5-day period
    momentum = (df['close'] - df['close'].shift(5)).shift(1)
    
    # Volume-weighted price to account for the importance of volume in price movements
    vwp = (df['amount'] / df['volume']).rolling(window=7).mean()
    
    # Adaptive window for moving average based on the ATR (Average True Range) to capture volatility
    true_range = df[['high', 'low']].diff(axis=1).iloc[:,0].abs() + (df['close'] - df['high']).abs() + (df['close'] - df['low']).abs()
    atr = true_range.rolling(window=14).mean()
    adaptive_window = (atr / atr.mean()).apply(lambda x: 5 if x < 1.5 else (10 if x < 2.5 else 15))
    
    # Dynamic Moving Average using the adaptive window
    dynamic_ma = df['close'].rolling(window=adaptive_window).mean().fillna(method='bfill')
    
    # Sentiment Factor (assuming a simple version where positive returns are positive sentiment, negative returns are negative)
    sentiment = (df['close'] - df['open']) / df['open']
    
    # Integrate all the factors: momentum, volume-weighted price, dynamic moving average, and sentiment
    alpha_factor = (momentum + vwp - dynamic_ma + sentiment).rolling(window=5).mean()
    
    return alpha_factor
```
