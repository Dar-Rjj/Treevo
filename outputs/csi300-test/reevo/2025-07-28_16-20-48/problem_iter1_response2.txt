```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the typical price as the average of (high + low + close)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    
    # Calculate the money flow ratio as the ratio of positive and negative money flows
    positive_money_flow = ((typical_price > typical_price.shift(1)) * typical_price * df['volume']).rolling(window=14).sum()
    negative_money_flow = ((typical_price < typical_price.shift(1)) * typical_price * df['volume']).rolling(window=14).sum()
    money_flow_ratio = positive_money_flow / (negative_money_flow + 1e-7)
    
    # Calculate the Money Flow Index (MFI) as an indicator of buying and selling pressure
    mfi = 100 - (100 / (1 + money_flow_ratio))
    
    # Factor: Money Flow Index (MFI) to detect overbought/oversold levels
    return mfi
```
