```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day moving average of the close price for trend indication
    close_ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 50-day moving average of the close price for a longer-term trend
    close_ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the volatility (standard deviation) of the close price over the last 10 days
    volatility = df['close'].rolling(window=10).std()
    
    # Calculate the volume-weighted close price
    vw_close = df['close'] * df['volume']
    
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate the factor as a combination of the above metrics
    factor = (vw_close / (avg_volume + 1e-7)) * (close_ma_10 / close_ma_50) * (volatility + 1)
    return factor
```
