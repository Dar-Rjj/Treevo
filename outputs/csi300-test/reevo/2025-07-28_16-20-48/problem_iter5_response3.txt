```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the exponentially weighted moving average of close prices for a 5-day window
    ewma_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the difference between today's close and the 5-day EWMA, then divide by today's volume
    momentum_adjusted = (df['close'] - ewma_5) / df['volume']
    
    # Calculate the volume-weighted price using amount and volume to identify the money flow over a 3-day window
    volume_weighted_price = (df['amount'] / df['volume']).rolling(window=3).mean()
    money_flow = (df['close'] * volume_weighted_price).rolling(window=3).mean()
    
    # Calculate the ratio of the current day's range to the 5-day average range
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=5).mean()
    range_ratio = daily_range / avg_daily_range
    
    # Adaptive window for volatility: use the exponentially weighted standard deviation of the last 10 days
    volatility = df['close'].ewm(span=10, adjust=False).std()
    
    # Combine all factors with a weighted sum, where the weights are based on the adaptive volatility
    alpha_factor = (momentum_adjusted + 0.5 * money_flow + 0.3 * range_ratio) / (1 + volatility)
    
    # Smooth the alpha factor using an exponential moving average with a span of 5
    alpha_factor_smoothed = alpha_factor.ewm(span=5, adjust=False).mean()
    
    return alpha_factor_smoothed
```
