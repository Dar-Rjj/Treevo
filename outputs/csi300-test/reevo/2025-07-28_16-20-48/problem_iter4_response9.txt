```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adjusted momentum by dividing today's close change by yesterday's volume
    adjusted_momentum = (df['close'].diff() / df['volume'].shift(1)).shift(1)
    
    # Calculate the 3-day exponential moving average of the money flow
    money_flow = (df['amount'] / df['volume']).ewm(span=3, adjust=False).mean()
    
    # Calculate the ratio of the current day's range to the 5-day exponential moving average range for volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.ewm(span=5, adjust=False).mean()
    range_ratio = daily_range / avg_daily_range
    
    # Combine all factors with a simple moving average of 5 periods to smooth out the noise
    alpha_factor = ((adjusted_momentum + money_flow + range_ratio) / 3).rolling(window=5, min_periods=1).mean()
    
    return alpha_factor
```
