```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the price change adjusted by volume
    momentum_adjusted = (df['close'].diff() / df['volume']).shift(1)
    
    # Compute the weighted price using amount and volume, reflecting the money flow
    money_flow = (df['amount'] / df['volume']).rolling(window=3).mean()
    
    # Incorporate a dynamic weighting factor based on the relative strength of the price
    close_ema_short = df['close'].ewm(span=5, adjust=False).mean()
    close_ema_long = df['close'].ewm(span=20, adjust=False).mean()
    relative_strength = (close_ema_short - close_ema_long) / close_ema_long
    dynamic_weight = 1 + (relative_strength - relative_strength.min()) / (relative_strength.max() - relative_strength.min())
    
    # Combine the factors with dynamic weighting
    combined_factor = (momentum_adjusted * dynamic_weight + money_flow) / 2
    
    # Apply different smoothing techniques to reduce noise
    smoothed_factor = combined_factor.rolling(window=5).mean().ewm(span=5, adjust=False).mean()
    
    return smoothed_factor
```
