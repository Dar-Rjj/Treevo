```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the exponential moving average of the close price
    df['ema_close'] = df['close'].ewm(span=5, adjust=False).mean()
    # Calculate the ratio of the close price to the exponential moving average
    factor = (df['close'] / df['ema_close']) * df['close_diff']
    return factor
```
