```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, then divide by today's volume
    momentum_adjusted = (df['close'].diff() / df['volume']).shift(1)
    
    # Calculate the weighted price using amount and volume to identify the money flow
    money_flow = (df['amount'] / df['volume']).rolling(window=3).mean()
    
    # Calculate the relative strength index (RSI) to measure the speed and change of price movements
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day moving average of the close price for trend indication
    close_ma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 50-day moving average of the close price for a longer-term trend
    close_ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate the trend factor as the ratio of short-term to long-term moving averages
    trend_factor = close_ma_10 / close_ma_50
    
    # Combine momentum-adjusted, money flow, RSI, and trend factor with a simple moving average of 5 periods to smooth out the noise
    alpha_factor = ((momentum_adjusted + money_flow + rsi + trend_factor) / 4).rolling(window=5).mean()
    
    return alpha_factor
```
