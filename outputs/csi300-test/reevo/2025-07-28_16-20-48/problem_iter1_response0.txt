```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open as a percentage of the open price
    price_diff_percent = (df['close'] - df['open']) / df['open']

    # Calculate the average amount over the last 5 days
    avg_amount_5d = df['amount'].rolling(window=5).mean()

    # Factor: Weighted sum of the percentage price change and the normalized 5-day average amount
    factor = 0.7 * price_diff_percent + 0.3 * (avg_amount_5d / df['amount'])

    return factor
```
