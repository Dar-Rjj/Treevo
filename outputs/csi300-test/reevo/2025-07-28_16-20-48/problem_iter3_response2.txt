```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum and adjust it with the volume
    momentum_adjusted = (df['close'].pct_change().rolling(window=5, min_periods=1).mean() * df['volume']).shift(1)
    
    # Calculate a volume-weighted price to identify the money flow over a 3-day window
    volume_weighted_price = (df['amount'] / df['volume']).rolling(window=3, min_periods=1).mean()
    
    # Calculate the ratio of the current day's range to the 10-day average range for volatility
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=10, min_periods=1).mean()
    range_ratio = daily_range / avg_daily_range
    
    # Incorporate a simple moving average of 5 periods to smooth out the close price
    sma_5 = df['close'].rolling(window=5, min_periods=1).mean()
    
    # Combine all factors using a dynamic weighting approach
    alpha_factor = (momentum_adjusted + 2 * volume_weighted_price + range_ratio + sma_5) / 4
    
    return alpha_factor
```
