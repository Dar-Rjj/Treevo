```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the volume-weighted close price
    vw_close = df['close'] * df['volume']
    # Calculate the average volume over the last 10 days
    avg_volume = df['volume'].rolling(window=10).mean()
    # Calculate the factor as the ratio of the volume-weighted close to the 10-day average volume
    factor = (vw_close / (avg_volume + 1e-7)).fillna(0)
    return factor
```
