```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate a momentum factor based on the 5-day and 20-day moving averages of close prices
    ma5 = df['close'].rolling(window=5).mean()
    ma20 = df['close'].rolling(window=20).mean()
    momentum_factor = ma5 - ma20
    
    # Calculate a weighted sum of the high, low, and close prices; weight by volume
    avg_price = (df['high'] + df['low'] + 2 * df['close']) / 4
    price_volume_weighted = (avg_price * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()

    # Calculate a volatility factor using the standard deviation of the last 10 days' closing prices
    volatility_factor = df['close'].rolling(window=10).std()

    # Combine the momentum factor, price-volume weighted average, and volatility factor
    alpha_factor = momentum_factor + (price_volume_weighted - df['close']) - volatility_factor

    return alpha_factor
```
