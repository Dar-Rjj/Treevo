```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the difference between today's close and the 5-day moving average, then divide by today's volume
    momentum = (df['close'] - df['close'].rolling(window=5).mean()) / df['volume']
    
    # Calculate the volume-weighted price using amount and volume to identify the money flow, with a 7-day rolling window
    money_flow = (df['amount'] / df['volume']).rolling(window=7).mean()
    
    # Calculate the ratio of the current day's range to the 10-day average range
    daily_range = df['high'] - df['low']
    avg_daily_range = daily_range.rolling(window=10).mean()
    range_ratio = daily_range / avg_daily_range
    
    # Adaptive volatility factor: standard deviation of returns over the past 20 days
    volatility = df['close'].pct_change().rolling(window=20).std()
    
    # Market sentiment: ratio of the close price to the 30-day simple moving average
    market_sentiment = df['close'] / df['close'].rolling(window=30).mean()
    
    # Combine all factors with dynamic weighting
    alpha_factor = (momentum + 2 * money_flow + 3 * range_ratio + 4 * volatility + 5 * market_sentiment) / 15
    
    # Smooth the alpha factor with a 5-day exponential moving average
    alpha_factor_smoothed = alpha_factor.ewm(span=5, adjust=False).mean()
    
    return alpha_factor_smoothed
```
