```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close, capturing daily price movement
    daily_price_change = df['close'].diff()
    # Calculate the average volume over the last 5 days as a measure of recent liquidity
    average_volume_5_days = df['volume'].rolling(5).mean()
    # Calculate the daily price range as an indicator of volatility
    daily_range = df['high'] - df['low']
    # Combine these elements into an alpha factor
    alpha_factor = (daily_price_change / daily_range) * average_volume_5_days
    return alpha_factor
```
