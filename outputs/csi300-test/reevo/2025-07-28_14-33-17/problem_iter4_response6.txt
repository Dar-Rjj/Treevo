```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:

    # Calculate the adaptive momentum with a lookback period based on volatility
    daily_returns = df['close'].pct_change()
    volatility_period = (daily_returns.rolling(window=20).std() * 100).round().astype(int)
    momentum = df['close'].pct_change(periods=volatility_period)

    # Calculate the adaptive RSI with a lookback period based on volatility
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=volatility_period, min_periods=1).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=volatility_period, min_periods=1).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))

    # Calculate the exponential moving average (EMA) of the close price with an adaptive span
    ema = df['close'].ewm(span=volatility_period, adjust=False).mean()

    # Calculate the adaptive volatility using the standard deviation of the returns
    volatility = daily_returns.rolling(window=volatility_period, min_periods=1).std()

    # Calculate the volume weighted by the amount which may represent the financial flow
    volume_weighted_by_amount = df['volume'] * (df['amount'] / df['volume'].rolling(window=20).mean())

    # Calculate a novel factor: a blend of adaptive momentum, RSI, EMA, and volatility, all adjusted by the volume weighted by amount
    factor = (momentum + (rsi - 50) / 50 + (df['close'] - ema) / df['close'].rolling(window=20).std() + 1/volatility) * volume_weighted_by_amount

    return factor
```
