```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate average trading volume in the last 5 days
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the ratio of today's volume to the 5-day average volume
    volume_ratio = df['volume'] / df['avg_volume_5d']
    
    # Calculate the percentage change in close price from the previous day
    close_change = df['close'].pct_change()
    
    # Calculate the weighted momentum factor
    momentum_factor = close_change * volume_ratio
    
    return momentum_factor
```
