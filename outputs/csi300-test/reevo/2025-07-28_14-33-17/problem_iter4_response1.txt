```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the adaptive rolling average of 'close' and 'volume'
    window_close = df['volume'].rolling(window=10).mean().apply(lambda x: 5 if x < 1000 else 20)
    avg_close = df['close'].rolling(window=window_close).mean()
    window_volume = df['amount'].rolling(window=10).mean().apply(lambda x: 5 if x < 100000 else 20)
    avg_volume = df['volume'].rolling(window=window_volume).mean()
    
    # Calculate a momentum signal based on the adaptive average close price and volume
    momentum_signal = (df['close'] - avg_close) * (df['volume'] - avg_volume)
    
    # Calculate the relative strength index (RSI) for the 'close' price with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the adaptive standard deviation of the close prices
    std_30 = df['close'].rolling(window=30).std().fillna(0)
    std_adaptive = df['close'].rolling(window=df['volume'].rolling(window=10).mean().apply(int)).std().fillna(0)
    
    # Calculate the 10-day exponential moving average of the price differences
    df['price_diff'] = df['close'].diff().fillna(0)
    ema_10 = df['price_diff'].ewm(span=10, adjust=False).mean()
    
    # Calculate a market sentiment indicator based on high-low range
    sentiment = (df['high'] - df['low']) / df['close']
    
    # Combine the momentum signal with RSI, volatility adjustment, and market sentiment to create the alpha factor
    alpha_factor = (momentum_signal * rsi * sentiment) / std_adaptive + ema_10
    
    return alpha_factor
```
