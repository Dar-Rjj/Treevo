```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily price change
    price_change = df['close'] - df['open']
    
    # Calculate the daily volume change
    volume_change = df['volume'].diff().fillna(0)
    
    # Calculate the amount of money traded
    money_traded = df['amount'] * df['close']
    
    # Calculate the momentum factor
    momentum = df['close'].pct_change(periods=5).fillna(0)
    
    # Combine the factors to form an alpha factor
    alpha_factor = (price_change / (df['high'] - df['low'] + 1e-7)) * (volume_change / (money_traded + 1e-7)) * momentum
    
    return alpha_factor
```
