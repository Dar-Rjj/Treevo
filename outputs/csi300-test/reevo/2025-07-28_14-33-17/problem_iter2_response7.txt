```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling average of 'close' and 'volume'
    avg_close = df['close'].rolling(window=10).mean()
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a momentum signal based on the average close price and volume
    momentum_signal = (df['close'] - avg_close) * (df['volume'] - avg_volume)
    
    # Calculate the relative strength index (RSI) for the 'close' price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 30-day standard deviation of the close prices
    std_30 = df['close'].rolling(window=30).std().fillna(0)
    
    # Calculate the 10-day exponential moving average of the price differences
    df['price_diff'] = df['close'].diff().fillna(0)
    ema_10 = df['price_diff'].ewm(span=10, adjust=False).mean()
    
    # Combine the momentum signal with RSI and volatility adjustment to create the alpha factor
    alpha_factor = (momentum_signal * rsi) / std_30 + ema_10
    
    return alpha_factor
```
