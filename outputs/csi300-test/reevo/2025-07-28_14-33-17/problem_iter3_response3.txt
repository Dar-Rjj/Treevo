```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the relative strength index (RSI)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0)
    loss = -delta.where(delta < 0, 0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the exponential moving average (EMA) of the close price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the difference between the EMA and the close price
    diff_ema_close = ema_close - df['close']
    
    # Calculate the volume weighted by the amount which may represent the financial flow
    volume_weighted_by_amount = df['volume'] * (df['amount'] / df['volume'].rolling(window=20).mean())
    
    # Calculate a novel factor: RSI combined with the difference between the EMA and the close price, then multiplied by the adjusted volume
    factor = (rsi + diff_ema_close) * volume_weighted_by_amount
    
    return factor
```
