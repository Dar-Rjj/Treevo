def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling average and standard deviation of 'close' for volatility
    avg_close_10 = df['close'].rolling(window=10).mean()
    std_close_10 = df['close'].rolling(window=10).std()
    volatility = std_close_10 / avg_close_10

    # Adaptive window for volume-based signals using a 30-day average
    avg_volume_30 = df['volume'].rolling(window=30).mean()
    std_volume_30 = df['volume'].rolling(window=30).std()
    volume_volatility = std_volume_30 / avg_volume_30

    # Momentum signal based on the exponential moving average (EMA) of close price and volume
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_volume_30 = df['volume'].ewm(span=30, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_10) * (df['volume'] - ema_volume_30)

    # Relative Strength Index (RSI) for the 'close' price with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.ewm(com=13, adjust=False).mean()
    avg_loss = loss.ewm(com=13, adjust=False).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=21).sum() / negative_flow.rolling(window=21).sum()

    # Trend following indicator: 50-day and 200-day exponential moving averages
    trend_following_short = df['close'].ewm(span=50, adjust=False).mean()
    trend_following_long = df['close'].ewm(span=200, adjust=False).mean()

    # Market sentiment: 10-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=10).sum()

    # Daily price range as an indicator of volatility
    daily_range = df['high'] - df['low']

    # Dynamic weighting based on recent performance
    def dynamic_weighting(signal, window):
        return (signal - signal.rolling(window=window).mean()) / signal.rolling(window=window).std()

    momentum_weight = dynamic_weighting(momentum_signal, 10)
    rsi_weight = dynamic_weighting(rsi, 10)
    money_ratio_weight = dynamic_weighting(money_ratio, 10)
    volatility_weight = dynamic_weighting(volatility, 10)
    volume_volatility_weight = dynamic_weighting(volume_volatility, 10)
    trend_short_weight = dynamic_weighting(df['close'] - trend_following_short, 10)
    trend_long_weight = dynamic_weighting(trend_following_short - trend_following_long, 10)
    market_sentiment_weight = dynamic_weighting(market_sentiment, 10)
    daily_range_weight = dynamic_weighting(daily_range, 10)

    # Combine all the signals into a final alpha factor with dynamic weights
    alpha_factor = (
        momentum_signal * momentum_weight * 0.3 +
        rsi * rsi_weight * 0.2 +
        money_ratio * money_ratio_weight * 0.15 +
        volatility * volatility_weight * 0.1 +
        volume_volatility * volume_volatility_weight * 0.05 +
        (df['close'] - trend_following_short) * trend_short_weight * 0.05 +
        (trend_following_short - trend_following_long) * trend_long_weight * 0.05 +
        market_sentiment * market_sentiment_weight * 0.05 +
        daily_range * daily_range_weight * 0.05
    ).fillna(0)

    return alpha_factor
