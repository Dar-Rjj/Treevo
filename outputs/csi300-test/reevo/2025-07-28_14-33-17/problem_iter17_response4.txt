```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Dynamic window for close price and volume based on recent volatility
    window_close = 30 + (df['close'].rolling(window=10).std() * 10).apply(int)
    window_volume = 45 + (df['volume'].rolling(window=10).std() * 10).apply(int)

    avg_close = df['close'].rolling(window=window_close).mean()
    std_close = df['close'].rolling(window=window_close).std()
    volatility = std_close / avg_close

    avg_volume = df['volume'].rolling(window=window_volume).mean()
    std_volume = df['volume'].rolling(window=window_volume).std()
    volume_volatility = std_volume / avg_volume

    # Momentum signal with adaptive EMA and volume
    ema_window = 20 + (volatility.rolling(window=10).mean() * 10).apply(int)
    ema_close = df['close'].ewm(span=ema_window, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close) * (df['volume'] - avg_volume)

    # Adaptive RSI
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    rsi_window = 14 + (volatility.rolling(window=10).mean() * 5).apply(int)
    avg_gain = gain.rolling(window=rsi_window).mean()
    avg_loss = loss.rolling(window=rsi_window).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Adaptive Money Flow Ratio
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    mfr_window = 30 + (volume_volatility.rolling(window=10).mean() * 10).apply(int)
    money_ratio = positive_flow.rolling(window=mfr_window).sum() / negative_flow.rolling(window=mfr_window).sum()

    # Trend following indicator: 120-day EMA
    trend_following = df['close'].ewm(span=120, adjust=False).mean()

    # Market sentiment: 30-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=30).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # Price range expansion factor
    price_range = df['high'] - df['low']
    price_range_expansion = price_range.rolling(window=30).mean() / avg_close

    # Seasonality factor: 5-day average return
    weekly_return = df['close'].pct_change(periods=5).rolling(window=20).mean()

    # Intraday volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']

    # Bollinger Bands %B
    bollinger_bands_high = avg_close + (2 * std_close)
    bollinger_bands_low = avg_close - (2 * std_close)
    bollinger_bands_percent_b = (df['close'] - bollinger_bands_low) / (bollinger_bands_high - bollinger_bands_low)

    # Combine all the signals into a final alpha factor with weighted combinations
    alpha_factor = (
        momentum_signal * 0.38 +
        rsi * 0.22 +
        money_ratio * 0.2 +
        volatility * 0.08 +
        volume_volatility * 0.05 +
        (df['close'] - trend_following) * 0.05 +
        market_sentiment * 0.025 +
        liquidity_measure * 0.025 +
        price_range_expansion * 0.025 +
        weekly_return * 0.025 +
        intraday_volatility * 0.025 +
        bollinger_bands_percent_b * 0.025
    ).fillna(0)

    return alpha_factor
```
