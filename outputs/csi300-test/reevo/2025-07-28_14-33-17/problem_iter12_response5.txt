```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive window for close price and volume
    avg_close_35 = df['close'].rolling(window=35).mean()
    std_close_35 = df['close'].rolling(window=35).std()
    volatility = std_close_35 / avg_close_35

    avg_volume_45 = df['volume'].rolling(window=45).mean()
    std_volume_45 = df['volume'].rolling(window=45).std()
    volume_volatility = std_volume_45 / avg_volume_45

    # Momentum signal based on the exponential moving average (EMA) and adaptive volume
    ema_close_22 = df['close'].ewm(span=22, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_22) * (df['volume'] - avg_volume_45)

    # Relative Strength Index (RSI) with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure with an adaptive window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=30).sum() / negative_flow.rolling(window=30).sum()

    # Trend following indicator: 110-day Exponential Moving Average (EMA)
    trend_following = df['close'].ewm(span=110, adjust=False).mean()

    # Market sentiment: 30-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=30).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # High-Low range factor
    high_low_range = (df['high'] - df['low']) / df['close']

    # New factor: Price change over the last 5 days
    price_change_5d = df['close'].pct_change(periods=5)

    # New factor: Price change over the last 10 days
    price_change_10d = df['close'].pct_change(periods=10)

    # New factor: Price change over the last 20 days
    price_change_20d = df['close'].pct_change(periods=20)

    # Combine all the signals into a final alpha factor with weighted combinations
    alpha_factor = (
        momentum_signal * 0.3 +
        rsi * 0.2 +
        money_ratio * 0.2 +
        volatility * 0.1 +
        volume_volatility * 0.05 +
        (df['close'] - trend_following) * 0.05 +
        market_sentiment * 0.05 +
        liquidity_measure * 0.05 +
        high_low_range * 0.05 +
        price_change_5d * 0.03 +
        price_change_10d * 0.02 +
        price_change_20d * 0.01
    ).fillna(0)

    return alpha_factor
```
