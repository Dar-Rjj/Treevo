def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Dynamic window for close price and volume based on the ATR (Average True Range)
    atr_30 = df[['high', 'low', 'close']].rolling(window=30).apply(lambda x: ta.ATR(x, timeperiod=14), raw=False)
    avg_close_30 = df['close'].rolling(window=30).mean()
    std_close_30 = df['close'].rolling(window=30).std()
    volatility = std_close_30 / avg_close_30

    avg_volume_45 = df['volume'].rolling(window=45).mean()
    std_volume_45 = df['volume'].rolling(window=45).std()
    volume_volatility = std_volume_45 / avg_volume_45

    # Momentum signal based on the exponential moving average (EMA) and adaptive volume
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_20) * (df['volume'] - avg_volume_45)

    # Relative Strength Index (RSI) with a dynamic window
    rsi_window = 14 + (atr_30 / avg_close_30).fillna(14).astype(int)
    rsi = df['close'].rolling(window=rsi_window).apply(lambda x: talib.RSI(x, rsi_window[-1]), raw=True)

    # Money Flow Ratio as a measure of buying to selling pressure with a dynamic window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio_window = 30 + (atr_30 / avg_close_30).fillna(30).astype(int)
    money_ratio = positive_flow.rolling(window=money_ratio_window).sum() / negative_flow.rolling(window=money_ratio_window).sum()

    # Trend following indicator: 120-day Exponential Moving Average (EMA)
    trend_following = df['close'].ewm(span=120, adjust=False).mean()

    # Market sentiment: 30-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=30).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # Price range expansion factor
    price_range = df['high'] - df['low']
    price_range_expansion = price_range.rolling(window=30).mean() / avg_close_30

    # Seasonality factor: 5-day average return
    weekly_return = df['close'].pct_change(periods=5).rolling(window=20).mean()

    # Intraday volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']

    # Bollinger Bands %B
    bollinger_bands_high = avg_close_30 + (2 * std_close_30)
    bollinger_bands_low = avg_close_30 - (2 * std_close_30)
    bollinger_bands_percent_b = (df['close'] - bollinger_bands_low) / (bollinger_bands_high - bollinger_bands_low)

    # Sector-specific indicator: high-low range as a percentage of close price
    sector_specific_indicator = (df['high'] - df['low']) / df['close']

    # Adaptive sentiment factor: 30-day change in volume
    volume_change_30 = df['volume'].pct_change(periods=30)

    # Combine all the signals into a final alpha factor with refined weighted combinations
    alpha_factor = (
        momentum_signal * 0.4 +
        rsi * 0.2 +
        money_ratio * 0.15 +
        volatility * 0.1 +
        volume_volatility * 0.08 +
        (df['close'] - trend_following) * 0.05 +
        market_sentiment * 0.03 +
        liquidity_measure * 0.02 +
        price_range_expansion * 0.02 +
        weekly_return * 0.02 +

        bollinger_bands_percent_b * 0.01 +
        sector_specific_indicator * 0.01 +
