```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the smoothed close price and volume
    smoothed_close = df['close'].ewm(span=10, adjust=False).mean()
    smoothed_volume = df['volume'].ewm(span=10, adjust=False).mean()

    # Calculate a momentum signal based on the smoothed close price and volume
    momentum_signal = (df['close'] - smoothed_close) * (df['volume'] - smoothed_volume)

    # Calculate the relative strength index (RSI) for the 'close' price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Smooth the RSI
    smoothed_rsi = rsi.ewm(span=10, adjust=False).mean()

    # Combine the momentum signal with smoothed RSI to create the alpha factor
    alpha_factor = momentum_signal * smoothed_rsi
    return alpha_factor
```
