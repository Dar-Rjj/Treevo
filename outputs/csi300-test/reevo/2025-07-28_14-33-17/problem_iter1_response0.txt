```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's close and yesterday's close
    df['close_diff'] = df['close'].diff()
    # Calculate the exponential moving average of volume
    df['ema_volume'] = df['volume'].ewm(span=5, adjust=False).mean()
    # Combine the two factors
    factor = df['close_diff'] / (df['ema_volume'] + 1e-7)
    return factor
```
