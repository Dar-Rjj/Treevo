def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Dynamic window for close price and volume based on market volatility
    avg_close_window = df['close'].rolling(window=30).std().apply(lambda x: 15 if x < 0.05 else 30)
    avg_close = df['close'].rolling(window=avg_close_window, min_periods=15).mean()
    std_close = df['close'].rolling(window=avg_close_window, min_periods=15).std()
    volatility = std_close / avg_close

    avg_volume_window = df['volume'].rolling(window=45).std().apply(lambda x: 20 if x < 0.05 else 45)
    avg_volume = df['volume'].rolling(window=avg_volume_window, min_periods=20).mean()
    std_volume = df['volume'].rolling(window=avg_volume_window, min_periods=20).std()
    volume_volatility = std_volume / avg_volume

    # Momentum signal based on the exponential moving average (EMA) and adaptive volume
    ema_close_20 = df['close'].ewm(span=20, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_20) * (df['volume'] - avg_volume)

    # Relative Strength Index (RSI) with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain_window = gain.rolling(window=14).std().apply(lambda x: 7 if x < 0.05 else 14)
    avg_loss_window = loss.rolling(window=14).std().apply(lambda x: 7 if x < 0.05 else 14)
    avg_gain = gain.rolling(window=avg_gain_window, min_periods=7).mean()
    avg_loss = loss.rolling(window=avg_loss_window, min_periods=7).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure with an adaptive window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=30).sum() / negative_flow.rolling(window=30).sum()

    # Trend following indicator: 120-day Exponential Moving Average (EMA)
    trend_following = df['close'].ewm(span=120, adjust=False).mean()

    # Market sentiment: 30-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=30).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # Price range expansion factor
    price_range = df['high'] - df['low']
    price_range_expansion = price_range.rolling(window=30).mean() / avg_close

    # Seasonality factor: 5-day average return
    weekly_return = df['close'].pct_change(periods=5).rolling(window=20).mean()

    # Intraday volatility
    intraday_volatility = (df['high'] - df['low']) / df['close']

    # New indicator: Bollinger Bands %B
    bollinger_bands_high = avg_close + (2 * std_close)
    bollinger_bands_low = avg_close - (2 * std_close)
    bollinger_bands_percent_b = (df['close'] - bollinger_bands_low) / (bollinger_bands_high - bollinger_bands_low)

    # Combine all the signals into a final alpha factor with weighted combinations
    alpha_factor = (
        momentum_signal * 0.38 +
        rsi * 0.22 +
        money_ratio * 0.2 +
        volatility * 0.08 +
        volume_volatility * 0.05 +
        (df['close'] - trend_following) * 0.05 +
        market_sentiment * 0.025 +
        liquidity_measure * 0.025 +
        price_range_expansion * 0.025 +
        weekly_return * 0.025 +
        intraday_vol
