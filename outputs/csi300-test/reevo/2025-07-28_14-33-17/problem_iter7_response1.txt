```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive window for close price and volume
    avg_close_20 = df['close'].rolling(window=20).mean()
    std_close_20 = df['close'].rolling(window=20).std()
    volatility = std_close_20 / avg_close_20

    avg_volume_30 = df['volume'].rolling(window=30).mean()
    std_volume_30 = df['volume'].rolling(window=30).std()
    volume_volatility = std_volume_30 / avg_volume_30

    # Momentum signal based on the exponential moving average (EMA) and adaptive volume
    ema_close_15 = df['close'].ewm(span=15, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_15) * (df['volume'] - avg_volume_30)

    # Relative Strength Index (RSI) with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure with an adaptive window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=20).sum() / negative_flow.rolling(window=20).sum()

    # Trend following indicator: 100-day Exponential Moving Average (EMA)
    trend_following = df['close'].ewm(span=100, adjust=False).mean()

    # Market sentiment: 20-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=20).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # New factor: High-Low range
    high_low_range = (df['high'] - df['low']) / df['close']

    # Combine all the signals into a final alpha factor with weighted combinations
    alpha_factor = (
        momentum_signal * 0.3 +
        rsi * 0.2 +
        money_ratio * 0.2 +
        volatility * 0.1 +
        volume_volatility * 0.05 +
        (df['close'] - trend_following) * 0.05 +
        market_sentiment * 0.05 +
        liquidity_measure * 0.05 +
        high_low_range * 0.05
    ).fillna(0)

    return alpha_factor
```
