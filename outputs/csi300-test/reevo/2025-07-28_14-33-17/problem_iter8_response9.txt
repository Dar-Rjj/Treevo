def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling average and standard deviation of 'close' for volatility
    avg_close_10 = df['close'].rolling(window=10).mean()
    std_close_10 = df['close'].rolling(window=10).std()
    volatility = std_close_10 / avg_close_10

    # Adaptive window for volume-based signals using a 20-day average
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    std_volume_20 = df['volume'].rolling(window=20).std()
    volume_volatility = std_volume_20 / avg_volume_20

    # Momentum signal based on the exponential moving average (EMA) of close price and volume
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_10) * (df['volume'] - ema_volume_20)

    # Relative Strength Index (RSI) for the 'close' price with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=14).sum() / negative_flow.rolling(window=14).sum()

    # Trend following indicator: 50-day exponential moving average (EMA)
    trend_following = df['close'].ewm(span=50, adjust=False).mean()

    # Market sentiment: 10-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=10).sum()

    # Price range indicator: 10-day high minus 10-day low
    price_range = df['high'].rolling(window=10).max() - df['low'].rolling(window=10).min()

    # Volume surge: 10-day EMA of volume divided by 20-day EMA of volume
    ema_volume_10 = df['volume'].ewm(span=10, adjust=False).mean()
    ema_volume_20 = df['volume'].ewm(span=20, adjust=False).mean()
    volume_surge = ema_volume_10 / ema_volume_20

    # Dynamic weights based on the performance of each signal
    momentum_weight = (momentum_signal > 0).astype(int).rolling(window=10).mean()
    rsi_weight = (rsi < 70).astype(int).rolling(window=10).mean()
    money_ratio_weight = (money_ratio > 1).astype(int).rolling(window=10).mean()
    volatility_weight = (volatility < 0.05).astype(int).rolling(window=10).mean()
    volume_volatility_weight = (volume_volatility < 0.05).astype(int).rolling(window=10).mean()
    trend_weight = (df['close'] - trend_following > 0).astype(int).rolling(window=10).mean()
    market_sentiment_weight = (market_sentiment > market_sentiment.shift(1)).astype(int).rolling(window=10).mean()
    price_range_weight = (price_range > 0).astype(int).rolling(window=10).mean()
    volume_surge_weight = (volume_surge > 1).astype(int).rolling(window=10).mean()

        volatility_weight +

        trend_weight + 
        market_sentiment_weight +
        price_range_weight +
        volume_surge_weight
