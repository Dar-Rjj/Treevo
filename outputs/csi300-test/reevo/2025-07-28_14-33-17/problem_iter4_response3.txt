```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum using the close price with a 20-day lookback period
    momentum = df['close'].pct_change(periods=20)
    
    # Calculate the RSI (Relative Strength Index) with a 14-day lookback period
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the exponential moving average (EMA) of the close price with a 30-day span
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the volatility using the standard deviation of the last 20 days' returns
    daily_returns = df['close'].pct_change()
    volatility = daily_returns.rolling(window=20).std()
    
    # Calculate the volume weighted by the amount which may represent the financial flow
    volume_weighted_by_amount = df['volume'] * (df['amount'] / df['volume'].rolling(window=20).mean())
    
    # Calculate the money flow index (MFI) with a 14-day lookback period
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_money_flow = money_flow.where(typical_price > typical_price.shift(1), 0).rolling(window=14).sum()
    negative_money_flow = money_flow.where(typical_price < typical_price.shift(1), 0).rolling(window=14).sum()
    mfi = 100 - (100 / (1 + (positive_money_flow / negative_money_flow)))
    
    # Calculate the Chaikin Money Flow (CMF) with a 20-day lookback period
    cmf = (money_flow / df['volume']).rolling(window=20).sum()
    
    # Calculate a novel factor: a blend of momentum, RSI, EMA, MFI, CMF, and volatility, all adjusted by the volume weighted by amount
    factor = (momentum + (rsi - 50) / 50 + (df['close'] - ema_30) / df['close'].rolling(window=20).std() + 
              (mfi - 50) / 50 + cmf + 1/volatility) * volume_weighted_by_amount
    
    return factor
```
