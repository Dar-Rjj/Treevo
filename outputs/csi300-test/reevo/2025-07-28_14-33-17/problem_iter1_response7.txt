```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate moving average of close price over a 10-day period to find trend direction
    df['ma_10'] = df['close'].rolling(window=10).mean()
    
    # Calculate the difference between the closing price and its 10-day moving average, normalized by volume
    alpha_factor = (df['close'] - df['ma_10']) / (df['volume'] + 1e-7)
    
    return alpha_factor
```
