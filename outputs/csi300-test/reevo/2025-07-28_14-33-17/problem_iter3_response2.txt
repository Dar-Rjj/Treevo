```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    
    # Calculate the average of open and close price
    avg_price = (df['open'] + df['close']) / 2
    
    # Calculate the difference between the average price and the low, high prices
    diff_low = avg_price - df['low']
    diff_high = df['high'] - avg_price
    
    # Calculate the volume weighted by the amount which may represent the financial flow
    volume_weighted_by_amount = df['volume'] * (df['amount'] / df['volume'].rolling(window=20).mean())
    
    # Calculate momentum factor using a 20-day lookback period
    momentum = df['close'] - df['close'].shift(20)
    
    # Calculate RSI (Relative Strength Index) with a 14-day window
    delta = df['close'].diff()
    up = delta.clip(lower=0)
    down = -delta.clip(upper=0)
    ema_up = up.ewm(com=13, adjust=False).mean()
    ema_down = down.ewm(com=13, adjust=False).mean()
    rs = ema_up/ema_down
    rsi = 100 - (100/(1 + rs))
    
    # Calculate EMA (Exponential Moving Average) with a 20-day window
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Combine all signals into a composite alpha factor
    composite_factor = ((diff_low + diff_high) / (diff_low + diff_high).max()) * volume_weighted_by_amount
    composite_factor += momentum + rsi + (df['close'] - ema_20)
    
    return composite_factor
```
