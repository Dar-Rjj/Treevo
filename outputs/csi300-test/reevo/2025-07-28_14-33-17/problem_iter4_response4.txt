```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor over different time frames
    momentum_short = df['close'].pct_change(5)
    momentum_medium = df['close'].pct_change(20)
    momentum_long = df['close'].pct_change(60)

    # Volatility factor: standard deviation of daily returns over the past 20 days
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=20).std()

    # Liquidity factor: volume ratio compared to the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    liquidity = df['volume'] / avg_volume_20

    # Relative strength (RS) factor: ratio of close price to its 60-day moving average
    ma_60 = df['close'].rolling(window=60).mean()
    relative_strength = df['close'] / ma_60

    # Money flow factor: ratio of amount to volume, smoothed over 10 days
    money_flow = (df['amount'] / df['volume']).ewm(span=10, adjust=False).mean()

    # Combined alpha factor: weighted sum of multiple factors
    factor = (
        momentum_short * 0.2 + 
        momentum_medium * 0.2 + 
        momentum_long * 0.2 + 
        (1 / volatility) * 0.1 +
        liquidity * 0.1 +
        relative_strength * 0.1 +
        money_flow * 0.1
    ).fillna(0)

    return factor
```
