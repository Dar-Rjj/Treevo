```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling average of 'close' and 'volume'
    avg_close = df['close'].rolling(window=10).mean()
    avg_volume = df['volume'].rolling(window=10).mean()
    
    # Calculate a momentum signal based on the average close price and volume
    momentum_signal = (df['close'] - avg_close) * (df['volume'] - avg_volume)
    
    # Calculate the relative strength index (RSI) for the 'close' price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Combine the momentum signal with RSI to create the alpha factor
    alpha_factor = momentum_signal * rsi
    return alpha_factor
```
