```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive window for close price and volume with seasonal adjustment (90-day window)
    avg_close_90 = df['close'].rolling(window=90).mean()
    std_close_90 = df['close'].rolling(window=90).std()
    volatility = std_close_90 / avg_close_90

    avg_volume_90 = df['volume'].rolling(window=90).mean()
    std_volume_90 = df['volume'].rolling(window=90).std()
    volume_volatility = std_volume_90 / avg_volume_90

    # Momentum signal based on the exponential moving average (EMA) and adaptive volume
    ema_close_30 = df['close'].ewm(span=30, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_30) * (df['volume'] - avg_volume_90)

    # Relative Strength Index (RSI) with an adaptive window (40-day window)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=40).mean()
    avg_loss = loss.rolling(window=40).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure with an adaptive window (60-day window)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=60).sum() / negative_flow.rolling(window=60).sum()

    # Trend following indicator: 200-day Exponential Moving Average (EMA)
    trend_following = df['close'].ewm(span=200, adjust=False).mean()

    # Market sentiment: 40-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=40).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # Price range expansion factor
    price_range = df['high'] - df['low']
    price_range_expansion = price_range.rolling(window=90).mean() / avg_close_90

    # MACD for trend confirmation
    ema_close_12 = df['close'].ewm(span=12, adjust=False).mean()
    ema_close_26 = df['close'].ewm(span=26, adjust=False).mean()
    macd_line = ema_close_12 - ema_close_26
    signal_line = macd_line.ewm(span=9, adjust=False).mean()
    macd_histogram = macd_line - signal_line

    # Combine all the signals into a final alpha factor with balanced weights
    alpha_factor = (
        momentum_signal * 0.3 +
        rsi * 0.2 +
        money_ratio * 0.2 +
        volatility * 0.05 +
        volume_volatility * 0.03 +
        (df['close'] - trend_following) * 0.07 +
        market_sentiment * 0.05 +
        liquidity_measure * 0.05 +
        price_range_expansion * 0.05 +
        macd_histogram * 0.05
    ).fillna(0)

    return alpha_factor
```
