```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor: 3-day return
    momentum = df['close'].pct_change(3)
    
    # Volatility factor: standard deviation of daily returns over the past 10 days
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=10).std()
    
    # Liquidity factor: volume ratio compared to the 5-day average volume
    avg_volume_5 = df['volume'].rolling(window=5).mean()
    liquidity = df['volume'] / avg_volume_5
    
    # Market sentiment factor: difference between high and low over a 5-day period
    high_low_diff = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Combined factor: weighted sum of momentum, inverse of volatility, liquidity, and market sentiment
    factor = (momentum * 0.4 + (1 / volatility) * 0.2 + liquidity * 0.2 + high_low_diff * 0.2).fillna(0)
    
    return factor
```
