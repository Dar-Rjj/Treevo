def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 50-day and 200-day moving averages of 'close' to capture long-term trends
    ma_50 = df['close'].rolling(window=50).mean()
    ma_200 = df['close'].rolling(window=200).mean()
    
    # Calculate the 50-day and 200-day moving averages of 'volume' to capture long-term liquidity
    vol_ma_50 = df['volume'].rolling(window=50).mean()
    vol_ma_200 = df['volume'].rolling(window=200).mean()
    
    # Calculate the daily price range as an indicator of volatility
    daily_range = df['high'] - df['low']
    
    # Calculate the relative strength index (RSI) for the 'close' price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate a momentum signal based on the long-term moving averages of close and volume
    momentum_signal = (df['close'] - ma_50) * (df['volume'] - vol_ma_50)
    
    # Combine the momentum signal, RSI, and volatility to create the alpha factor
    alpha_factor = momentum_signal * rsi * daily_range
    return alpha_factor
