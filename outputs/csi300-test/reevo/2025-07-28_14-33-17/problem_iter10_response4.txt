```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive window for close price and volume with seasonal adjustment (45-day window)
    avg_close_45 = df['close'].rolling(window=45).mean()
    std_close_45 = df['close'].rolling(window=45).std()
    volatility = std_close_45 / avg_close_45

    avg_volume_45 = df['volume'].rolling(window=45).mean()
    std_volume_45 = df['volume'].rolling(window=45).std()
    volume_volatility = std_volume_45 / avg_volume_45

    # Momentum signal based on the exponential moving average (EMA) and adaptive volume
    ema_close_18 = df['close'].ewm(span=18, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_18) * (df['volume'] - avg_volume_45)

    # Relative Strength Index (RSI) with an adaptive window (20-day window)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=20).mean()
    avg_loss = loss.rolling(window=20).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure with an adaptive window (30-day window)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=30).sum() / negative_flow.rolling(window=30).sum()

    # Trend following indicator: 120-day Exponential Moving Average (EMA)
    trend_following = df['close'].ewm(span=120, adjust=False).mean()

    # Market sentiment: 25-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=25).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # Price range expansion factor
    price_range = df['high'] - df['low']
    price_range_expansion = price_range.rolling(window=45).mean() / avg_close_45

    # Additional indicators
    ema_close_50 = df['close'].ewm(span=50, adjust=False).mean()
    ema_close_200 = df['close'].ewm(span=200, adjust=False).mean()
    golden_cross = (ema_close_50 > ema_close_200).astype(int) - (ema_close_50 < ema_close_200).astype(int)

    # Volume-based momentum
    volume_momentum = (df['volume'] - df['volume'].shift(1)) / df['volume'].shift(1)

    # Inter-market analysis: Correlation with S&P 500
    sp500 = df['sp500'] if 'sp500' in df.columns else pd.Series(0, index=df.index)
    correlation_sp500 = df['close'].rolling(window=50).corr(sp500)

    # Price change over the last 5 days
    price_change_5 = df['close'] - df['close'].shift(5)

    # Combine all the signals into a final alpha factor with weighted combinations
    alpha_factor = (
        momentum_signal * 0.3 +
        rsi * 0.15 +
        money_ratio * 0.15 +
        volatility * 0.1 +
        volume_volatility * 0.05 +
        (df['close'] - trend_following) * 0.05 +
        market_sentiment * 0.05 +
        liquidity_measure * 0.05 +
        price_range_expansion * 0.05 +
        golden_cross * 0.05 +
        volume_momentum * 0.05 +
        correlation_sp500 * 0.05 +
        price_change_5 * 0.05
    ).fillna(0)

    return alpha_factor
```
