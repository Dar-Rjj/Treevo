```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and open price
    close_open_diff = df['close'] - df['open']
    
    # Calculate a weighted average price using volume
    weighted_avg_price = (df['close'] * df['volume']) / df['volume'].rolling(window=5).mean()
    
    # Combine the two features to form an alpha factor
    alpha_factor = close_open_diff * weighted_avg_price
    return alpha_factor
```
