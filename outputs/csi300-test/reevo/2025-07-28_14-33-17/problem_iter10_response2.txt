```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive window for close price and volume
    avg_close_40 = df['close'].rolling(window=40).mean()
    std_close_40 = df['close'].rolling(window=40).std()
    volatility = std_close_40 / avg_close_40

    avg_volume_50 = df['volume'].rolling(window=50).mean()
    std_volume_50 = df['volume'].rolling(window=50).std()
    volume_volatility = std_volume_50 / avg_volume_50

    # Momentum signal based on the exponential moving average (EMA) and adaptive volume
    ema_close_30 = df['close'].ewm(span=30, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_30) * (df['volume'] - avg_volume_50)

    # Relative Strength Index (RSI) with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure with an adaptive window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=40).sum() / negative_flow.rolling(window=40).sum()

    # Trend following indicator: 150-day Exponential Moving Average (EMA)
    trend_following = df['close'].ewm(span=150, adjust=False).mean()

    # Market sentiment: 40-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=40).sum()

    # Liquidity measure: Volume-to-Amount ratio
    liquidity_measure = df['volume'] / df['amount']

    # Price range expansion factor
    price_range = df['high'] - df['low']
    price_range_expansion = price_range.rolling(window=40).mean() / avg_close_40

    # Combine all the signals into a final alpha factor with weighted combinations
    alpha_factor = (
        momentum_signal * 0.4 +
        rsi * 0.2 +
        money_ratio * 0.2 +
        volatility * 0.08 +
        volume_volatility * 0.07 +
        (df['close'] - trend_following) * 0.05 +
        market_sentiment * 0.02 +
        liquidity_measure * 0.02 +
        price_range_expansion * 0.06
    ).fillna(0)

    return alpha_factor
```
