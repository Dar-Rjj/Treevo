```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the daily price change
    df['price_change'] = df['close'] - df['open']
    
    # Calculate the 10-day exponential moving average of the price changes
    ema_10_price_change = df['price_change'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day standard deviation of the close prices to represent volatility
    std_30_close = df['close'].rolling(window=30).std().fillna(0)
    
    # Calculate the volume rate of change over 10 days
    roc_10_volume = df['volume'].pct_change(periods=10).fillna(0)
    
    # Novel alpha factor: combine momentum with volatility and volume rate of change
    alpha_factor = (ema_10_price_change / std_30_close) * roc_10_volume
    
    return alpha_factor
```
