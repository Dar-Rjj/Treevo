```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the rolling average and standard deviation of 'close' for volatility with an adaptive window
    avg_close_10 = df['close'].rolling(window=10).mean()
    std_close_10 = df['close'].rolling(window=10).std()
    volatility = std_close_10 / avg_close_10

    # Adaptive window for volume-based signals using a 30-day EMA
    avg_volume_30 = df['volume'].ewm(span=30, adjust=False).mean()
    std_volume_30 = df['volume'].ewm(span=30, adjust=False).std()
    volume_volatility = std_volume_30 / avg_volume_30

    # Momentum signal based on the exponential moving average (EMA) of close price and volume
    ema_close_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_volume_30 = df['volume'].ewm(span=30, adjust=False).mean()
    momentum_signal = (df['close'] - ema_close_10) * (df['volume'] - ema_volume_30)

    # Relative Strength Index (RSI) for the 'close' price with an adaptive window
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))

    # Money Flow Ratio as a measure of buying to selling pressure with an adaptive window
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    money_flow = typical_price * df['volume']
    positive_flow = money_flow.where(typical_price > typical_price.shift(1), 0)
    negative_flow = money_flow.where(typical_price < typical_price.shift(1), 0)
    money_ratio = positive_flow.rolling(window=21).sum() / negative_flow.rolling(window=21).sum()

    # Trend following indicators: 50-day and 200-day exponential moving averages
    trend_following_short = df['close'].ewm(span=50, adjust=False).mean()
    trend_following_long = df['close'].ewm(span=200, adjust=False).mean()

    # Market sentiment: 10-day sum of the amount traded
    market_sentiment = df['amount'].rolling(window=10).sum()

    # Daily price range as an indicator of volatility
    daily_range = df['high'] - df['low']

    # Combine all the signals into a final alpha factor with balanced weights
    alpha_factor = (
        momentum_signal * 0.3 +
        rsi * 0.2 +
        money_ratio * 0.15 +
        volatility * 0.1 +
        volume_volatility * 0.05 +
        (df['close'] - trend_following_short) * 0.05 +
        (trend_following_short - trend_following_long) * 0.05 +
        market_sentiment * 0.05 +
        daily_range * 0.05
    ).fillna(0)

    return alpha_factor
```
