```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    import pandas as pd
    
    # Calculate the Relative Strength Index (RSI)
    delta = df['close'].diff(1)
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the Exponential Moving Average (EMA) of the close price
    ema_close = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the momentum factor
    momentum = df['close'] - df['close'].shift(20)
    
    # Calculate the volatility factor
    volatility = df['close'].rolling(window=20).std()
    
    # Calculate the volume weighted by the amount
    volume_weighted_by_amount = df['volume'] * (df['amount'] / df['volume'].rolling(window=20).mean())
    
    # Combine RSI, EMA, momentum, and volatility into a single alpha factor
    # Normalize each component by its rolling standard deviation for stability
    rsi_normalized = (rsi - rsi.rolling(window=20).mean()) / rsi.rolling(window=20).std()
    ema_normalized = (ema_close - ema_close.rolling(window=20).mean()) / ema_close.rolling(window=20).std()
    momentum_normalized = (momentum - momentum.rolling(window=20).mean()) / momentum.rolling(window=20).std()
    volatility_normalized = (volatility - volatility.rolling(window=20).mean()) / volatility.rolling(window=20).std()
    
    # Combine normalized factors
    factor = (rsi_normalized + ema_normalized + momentum_normalized + volatility_normalized) * volume_weighted_by_amount
    
    return factor
```
