```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Momentum factor: 5-day return
    momentum = df['close'].pct_change(5)
    
    # Volatility factor: standard deviation of daily returns over the past 20 days
    returns = df['close'].pct_change()
    volatility = returns.rolling(window=20).std()
    
    # Liquidity factor: volume ratio compared to the 20-day average volume
    avg_volume_20 = df['volume'].rolling(window=20).mean()
    liquidity = df['volume'] / avg_volume_20
    
    # Combined factor: weighted sum of momentum, inverse of volatility, and liquidity
    factor = (momentum * 0.4 + (1 / volatility) * 0.3 + liquidity * 0.3).fillna(0)
    
    return factor
```
