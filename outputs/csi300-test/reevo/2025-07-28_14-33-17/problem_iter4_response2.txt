```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day exponential moving average of 'close' and 'volume'
    ema_close = df['close'].ewm(span=10, adjust=False).mean()
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Calculate a momentum signal based on the EMA of close and volume
    momentum_signal = (df['close'] - ema_close) * (df['volume'] - ema_volume)
    
    # Calculate the relative strength index (RSI) for the 'close' price
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).fillna(0)
    loss = (-delta.where(delta < 0, 0)).fillna(0)
    avg_gain = gain.rolling(window=14).mean()
    avg_loss = loss.rolling(window=14).mean()
    rs = avg_gain / avg_loss
    rsi = 100 - (100 / (1 + rs))
    
    # Calculate the 10-day standard deviation of the close prices
    std_10 = df['close'].rolling(window=10).std().fillna(0)
    
    # Calculate the 5-day exponential moving average of the price differences
    df['price_diff'] = df['close'].diff().fillna(0)
    ema_5 = df['price_diff'].ewm(span=5, adjust=False).mean()
    
    # Combine the momentum signal with RSI and volatility adjustment to create the alpha factor
    alpha_factor = (momentum_signal * rsi) / std_10 + ema_5
    
    return alpha_factor
```
