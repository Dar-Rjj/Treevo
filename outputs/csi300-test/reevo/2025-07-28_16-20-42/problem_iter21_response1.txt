def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day exponential moving averages of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with equal weights
    weighted_prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the ratio of (high - low) to the 10-day exponential moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (ema_10 + 1e-7)
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Adaptive momentum and mean reversion weights
    momentum_weight = (ema_5 - ema_10) / (ema_10 + 1e-7)
    mean_reversion_weight = (weighted_prices - df['close']) / (volatility_ratio + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (momentum_weight + mean_reversion_weight +
                    (ema_10 - df['close']) / (realized_volatility + 1e-7) +
                    volume_factor)
    return alpha_factor
