```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.15 * df['open'] + 0.35 * df['high'] + 0.35 * df['low'] + 0.15 * df['close'])
    
    # Calculate the ratio of (high - low) to the 10-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_10 + 1e-7)
    
    # Calculate the 3-day and 10-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Incorporate momentum by calculating the 3-day and 10-day return
    momentum_3 = df['close'].pct_change(periods=3)
    momentum_10 = df['close'].pct_change(periods=10)
    
    # Incorporate liquidity by calculating the turnover ratio (volume / amount)
    liquidity = df['volume'] / (df['amount'] + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        (weighted_prices - sma_3) / (volatility_ratio + 1e-7) +
        (ema_3 - df['close']) / (realized_volatility + 1e-7) +
        0.5 * (momentum_3 + momentum_10) +
        0.1 * liquidity
    )
    return alpha_factor
```
