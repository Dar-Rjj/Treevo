```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()

    # Calculate momentum as the difference between the 5-day and 20-day moving averages
    momentum = ma_5 - ma_20

    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])

    # Calculate the ratio of volume to the moving amount for liquidity
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)

    # Calculate the 5-day standard deviation of the close price for volatility
    std_5 = df['close'].rolling(window=5).std()

    # Adaptive weighting based on the recent trend
    adaptive_weight = (momentum + 1e-7) / (volatility_ratio + 1e-7)

    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - ma_5) * adaptive_weight / (std_5 + 1e-7)
    return alpha_factor
```
