def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Adaptive window for moving averages and volatility
    window = 3 + (df['close'].pct_change().abs().rolling(window=10).mean() * 10).astype(int)
    
    # Calculate the adaptive simple moving average of the close price
    sma_adaptive = df['close'].rolling(window=window, min_periods=3).mean()
    
    # Calculate the adaptive exponential moving average of the close price
    ema_adaptive = df['close'].ewm(span=window, adjust=False).mean()
    
    # Calculate the adaptive realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility_adaptive = log_returns.rolling(window=window, min_periods=3).std()
    
    # Calculate the ratio of (high - low) to the adaptive simple moving average of the close price for volatility
    volatility_ratio_adaptive = (df['high'] - df['low']) / (sma_adaptive + 1e-7)
    
    # Incorporate volume data for robustness with an adaptive average
    avg_volume_adaptive = df['volume'].rolling(window=window, min_periods=3).mean()
    volume_factor_adaptive = (df['volume'] - avg_volume_adaptive) / (avg_volume_adaptive + 1e-7)
    
    # Calculate the adaptive mean reversion factor
    mean_reversion_adaptive = (df['close'] - sma_adaptive) / (sma_adaptive + 1e-7)
    
    # Momentum factor
    momentum = (df['close'] / df['close'].shift(window) - 1).fillna(0)
    
    # Liquidity factor
    liquidity = df['amount'] / df['volume']
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        (ema_adaptive - df['close']) / (realized_volatility_adaptive + 1e-7) +
        (df['close'] - sma_adaptive) / (volatility_ratio_adaptive + 1e-7) +
        volume_factor_adaptive +
        mean_reversion_adaptive +
        momentum +
        liquidity
    )
    return alpha_factor
