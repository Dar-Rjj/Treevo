```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the true range as the difference between high and low for a given day
    true_range = df['high'] - df['low']
    
    # Calculate the 5-day average volume
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day momentum based on the close price
    momentum = df['close'] - df['close'].shift(5)
    
    # Calculate the 5-day volatility using the standard deviation of the close prices
    volatility = df['close'].rolling(window=5).std()
    
    # Combine true range, average volume, momentum, and volatility to create a new factor
    factor_values = (true_range + avg_volume + momentum) / volatility
    
    return factor_values
```
