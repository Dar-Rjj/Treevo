```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price to capture short-term trend
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Dynamic weights for open, high, low, and close prices based on market conditions
    open_weight = (df['high'] - df['open']) / (df['high'] - df['low'] + 1e-7)
    high_weight = (df['high'] - df['low']) / (df['high'] - df['low'] + 1e-7)
    low_weight = (df['low'] - df['close']) / (df['high'] - df['low'] + 1e-7)
    close_weight = (df['close'] - df['open']) / (df['high'] - df['low'] + 1e-7)
    
    # Normalize the weights to ensure they sum to 1
    total_weight = open_weight + high_weight + low_weight + close_weight
    open_weight /= total_weight
    high_weight /= total_weight
    low_weight /= total_weight
    close_weight /= total_weight
    
    # Calculate the weighted sum of the open, high, low, and close prices
    weighted_prices = (open_weight * df['open'] + high_weight * df['high'] + low_weight * df['low'] + close_weight * df['close'])
    
    # Calculate the daily range to measure volatility
    daily_range = df['high'] - df['low']
    
    # Calculate the ratio of daily range to the 5-day moving average to normalize volatility
    volatility_ratio = daily_range / (ma_5 + 1e-7)
    
    # Calculate the volume-weighted amount to reflect trading activity
    vol_weighted_amount = df['amount'] / (df['volume'] + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = ((weighted_prices - ma_5) * vol_weighted_amount) / (volatility_ratio + 1e-7)
    
    return alpha_factor
```
