```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving averages of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.15 * df['open'] + 0.35 * df['high'] + 0.35 * df['low'] + 0.15 * df['close'])
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Momentum factor: difference between 3-day and 10-day SMAs
    momentum = sma_3 - sma_10
    
    # Liquidity factor: volume divided by the absolute difference between high and low
    liquidity = df['volume'] / (df['high'] - df['low'] + 1e-7)
    
    # Mean reversion factor: difference between the close price and the 3-day EMA
    mean_reversion = df['close'] - ema_3
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - sma_3) / (volatility_ratio + 1e-7) + (ema_3 - df['close']) / (realized_volatility + 1e-7) + momentum + liquidity + mean_reversion
    return alpha_factor
```
