```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the open price, suggesting momentum
    momentum = df['close'] - df['open']
    
    # Compute the average of high and low for a typical price representation
    avg_price = (df['high'] + df['low']) / 2
    
    # Consider the volume weighted amount, reflecting transaction activity
    vol_weighted_amount = df['amount'] / df['volume']
    
    # Calculate the ratio of the current close to the previous close, indicating price change
    close_ratio = df['close'] / df['close'].shift(1)
    
    # Calculate the recent volatility over the last 5 days using the range (high - low)
    recent_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Create a factor that combines momentum with the relative position of the close to the average price
    # and adjusts it by the close ratio, volume-weighted amount, and recent volatility
    # This factor aims to capture the strength of the price movement adjusted by where the closing occurred in relation to the day's average price
    # and the rate of change in the close price, along with the transaction activity and recent market volatility
    factor = (momentum * vol_weighted_amount * close_ratio) / ((avg_price - df['low']) + 1e-7) / (recent_volatility + 1e-7)
    
    return factor
```
