```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 30-day exponential moving average of the close price
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 3-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 3-day mean reversion factor
    mean_reversion = (df['close'] - sma_3) / (sma_3 + 1e-7)
    
    # Calculate the 3-day and 10-day momentum factors
    momentum_3 = df['close'] / df['close'].shift(3) - 1
    momentum_10 = df['close'] / df['close'].shift(10) - 1
    
    # Calculate relative strength
    relative_strength = (ema_3 - ema_10) / (ema_10 + 1e-7)
    
    # Calculate adaptive smoothing factor
    adaptive_smoothing = (ema_3 - ema_30) / (ema_30 + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        (ema_3 - df['close']) / (realized_volatility + 1e-7) +
        (df['close'] - sma_3) / (volatility_ratio + 1e-7) +
        volume_factor +
        mean_reversion +
        momentum_3 +
        momentum_10 +
        relative_strength +
        adaptive_smoothing
    )
    return alpha_factor
```
