def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 14-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_14 = df['close'].rolling(window=14).mean()
    
    # Adaptive weights based on the ratio of today's volume to the 14-day average volume
    avg_volume = df['volume'].rolling(window=14).mean()
    adaptive_weights = df['volume'] / (avg_volume + 1e-7)
    
    # Weighted sum of open, high, low, and close prices with adaptive weights
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close']) * adaptive_weights
    
    # Volatility ratio using the 3-day simple moving average
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # 3-day and 14-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    ema_14 = df['close'].ewm(span=14, adjust=False).mean()
    
    # 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Momentum factor: difference between the 3-day and 14-day EMA
    momentum = (ema_3 - ema_14) / (realized_volatility + 1e-7)
    
    # Mean reversion factor: difference between the current price and the 14-day SMA
    mean_reversion = (sma_14 - df['close']) / (volatility_ratio + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - sma_3) / (volatility_ratio + 1e-7) + momentum + mean_reversion
    return alpha_factor
