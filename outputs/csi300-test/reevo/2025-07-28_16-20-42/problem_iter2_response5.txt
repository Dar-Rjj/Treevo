```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the momentum as the difference between close and open prices
    momentum = df['close'] - df['open']
    
    # Compute the average of high and low for a typical price representation
    avg_price = (df['high'] + df['low']) / 2
    
    # Calculate the volume-weighted amount, reflecting transaction activity
    vol_weighted_amount = df['amount'] / df['volume']
    
    # Calculate the recent volatility over the last 5 days using the range (high - low)
    recent_volatility = (df['high'] - df['low']).rolling(window=5).mean()
    
    # Create a factor that combines momentum with the relative position of the close to the average price
    # This factor aims to capture the strength of the price movement adjusted by where the closing occurred in relation to the day's average price
    # Adjust for recent volatility to normalize the factor
    factor = (momentum * vol_weighted_amount) / ((avg_price - df['low']) + 1e-7) / (recent_volatility + 1e-7)
    
    return factor
```
