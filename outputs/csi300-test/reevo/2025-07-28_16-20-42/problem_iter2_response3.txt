```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day momentum (difference between current close and 5-day prior close)
    df['momentum_5d'] = df['close'] - df['close'].shift(5)
    
    # Calculate the 10-day volatility (standard deviation of the last 10 days' closing prices)
    df['volatility_10d'] = df['close'].rolling(window=10).std()
    
    # Calculate the 5-day average volume
    df['avg_volume_5d'] = df['volume'].rolling(window=5).mean()
    
    # Calculate the 5-day volume trend (current volume - 5-day average volume)
    df['volume_trend_5d'] = df['volume'] - df['avg_volume_5d']
    
    # Define a factor based on the ratio of the close price to the 10-day moving average of the close
    df['close_to_mavg_ratio_10d'] = df['close'] / df['close'].rolling(window=10).mean()
    
    # Final alpha factor: weighted sum of 5-day momentum, 10-day volatility, 5-day volume trend, and close to 10-day moving average ratio
    factor = 0.3 * df['momentum_5d'] + 0.2 * df['volatility_10d'] + 0.3 * df['volume_trend_5d'] + 0.2 * df['close_to_mavg_ratio_10d']
    
    return factor
```
