def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with dynamic weights based on volume
    total_volume = df['volume'].sum()
    volume_ratio = df['volume'] / total_volume
    weighted_prices = (0.1 * volume_ratio * df['open'] + 0.25 * volume_ratio * df['high'] + 0.25 * volume_ratio * df['low'] + 0.4 * volume_ratio * df['close'])
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Incorporate momentum by calculating the 7-day return
    momentum = df['close'] / df['close'].shift(7) - 1
    
    # Final alpha factor as a combination of the above metrics, incorporating momentum
    alpha_factor = (weighted_prices - sma_3) / (volatility_ratio + 1e-7) + (ema_3 - df['close']) / (realized_volatility + 1e-7) + momentum
    return alpha_factor
