def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the ratio of volume to the moving amount
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Calculate the 5-day average trading amount
    avg_amount = df['amount'].rolling(window=5).mean()
    
    # Calculate the 5-day average volume
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the trend strength using the difference between the 5-day EMA and the 10-day EMA
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    trend_strength = ema_5 - ema_10
    
    # Calculate the liquidity as the ratio of average amount to average volume
    liquidity = avg_amount / (avg_volume + 1e-7)
    
    # Final alpha factor as a combination of the above metrics with adaptive weights
    alpha_factor = (weighted_prices - ma_5) / (volatility_ratio + 1e-7) + \
                   (ema_5 - df['close']) / (realized_volatility + 1e-7) + \
                   (df['volume'] / avg_amount) + \
                   trend_strength + \
                   liquidity
    return alpha_factor
