```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low price over the average of open and close, 
    # aiming to gauge the day's trading range normalized by the central price level.
    daily_range = (df['high'] - df['low']) / ((df['open'] + df['close'])/2 + 1e-7)
    
    # Calculate volume-weighted moving average of the close price for a 5-day window to capture short-term momentum adjusted for volume.
    volume_weighted_close = (df['close'] * df['volume']).rolling(window=5).sum() / df['volume'].rolling(window=5).sum()
    
    # The final alpha factor is obtained by comparing today's close price against its 5-day VWAP,
    # adjusted by the relative daily range to give an idea on how the stock performed in comparison
    # to its recent past, incorporating both price volatility and volume information.
    factor = (df['close'] - volume_weighted_close) * daily_range
    
    return factor
```
