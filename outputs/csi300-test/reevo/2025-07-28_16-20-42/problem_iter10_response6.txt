```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the ratio of (high - low) to the 5-day moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Calculate the 5-day and 20-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Calculate the 5-day average trading amount
    avg_amount = df['amount'].rolling(window=5).mean()
    
    # Calculate trend strength
    trend_strength = (df['close'] - ma_20) / (ma_20 + 1e-7)
    
    # Calculate volume trend
    volume_trend = (df['volume'] - df['volume'].shift(5)) / (df['volume'].shift(5) + 1e-7)
    
    # Calculate liquidity as the ratio of amount to volume
    liquidity = df['amount'] / (df['volume'] + 1e-7)
    
    # Calculate market depth as the ratio of (high - low) to volume
    market_depth = (df['high'] - df['low']) / (df['volume'] + 1e-7)
    
    # Calculate momentum
    momentum = (df['close'] - df['close'].shift(20)) / (df['close'].shift(20) + 1e-7)
    
    # Calculate the 5-day money flow index (MFI)
    typical_price = (df['high'] + df['low'] + df['close']) / 3
    raw_money_flow = typical_price * df['volume']
    positive_money_flow = raw_money_flow[df['close'] > df['close'].shift(1)].fillna(0)
    negative_money_flow = raw_money_flow[df['close'] < df['close'].shift(1)].fillna(0)
    mfi_positive = positive_money_flow.rolling(window=5).sum()
    mfi_negative = negative_money_flow.rolling(window=5).sum()
    money_flow_index = 100 - (100 / (1 + mfi_positive / (mfi_negative + 1e-7)))
    
    # Calculate the 14-day relative strength index (RSI)
    delta = df['close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / (loss + 1e-7)
    rsi = 100 - (100 / (1 + rs))
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        0.2 * ((weighted_prices - ma_5) / (volatility_ratio + 1e-7)) +
        0.1 * ((ema_5 - df['close']) / (realized_volatility + 1e-7)) +
        0.1 * (df['volume'] / avg_amount) +
        0.2 * trend_strength +
        0.1 * volume_trend +
        0.1 * (liquidity - liquidity.rolling(window=5).mean()) +
        0.1 * market_depth +
        0.1 * momentum +
        0.05 * (money_flow_index - 50) / 50 +
        0.05 * (rsi - 50) / 50
    )
    return alpha_factor
```
