```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with equal weights
    weighted_prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Momentum factor: difference between the 3-day and 10-day SMA
    momentum_factor = sma_3 - sma_10
    
    # Mean reversion factor: difference between the close price and 3-day EMA
    mean_reversion_factor = df['close'] - ema_3
    
    # Adaptive weights for momentum and mean reversion
    adaptive_weights = (momentum_factor + 1e-7) / (realized_volatility + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = ((weighted_prices - sma_3) / (volatility_ratio + 1e-7) + 
                    (mean_reversion_factor / (realized_volatility + 1e-7)) +
                    adaptive_weights * volume_factor)
    return alpha_factor
```
