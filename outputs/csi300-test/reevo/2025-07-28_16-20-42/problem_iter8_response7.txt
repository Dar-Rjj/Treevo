def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day exponential moving averages of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with dynamic weights
    weighted_prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the ratio of (high - low) to the 5-day EMA of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (ema_5 + 1e-7)
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Incorporate market trends by calculating the 20-day momentum
    momentum_20 = df['close'] / df['close'].shift(20) - 1
    
    # Incorporate liquidity by calculating the turnover rate
    turnover_rate = df['volume'] / (df['amount'] + 1e-7)
    
    # Advanced volatility measure: Parkinson's volatility
    parkinson_volatility = np.sqrt((1 / (4 * 5 * np.log(2))) * ((np.log(df['high'] / df['low']) ** 2).rolling(window=5).sum()))
    
    # Adaptive weights based on trend strength and realized volatility
    adaptive_weight_trend = (momentum_20 > 0).astype(int) * 0.5 + (realized_volatility < 0.05).astype(int) * 0.5
    adaptive_weight_volatility = (realized_volatility < 0.05).astype(int)
    
    # Final alpha factor as a combination of the above metrics with adaptive weights
    alpha_factor = (
        (weighted_prices - ema_5) / (volatility_ratio + 1e-7) * adaptive_weight_trend +
        (ema_5 - df['close']) / (realized_volatility + 1e-7) * (1 - adaptive_weight_trend) +
        0.5 * momentum_20 * adaptive_weight_trend +
        0.5 * turnover_rate * (1 - adaptive_weight_trend) +
        0.5 * parkinson_volatility * adaptive_weight_volatility
    )
    return alpha_factor
