def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving averages of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.25 * df['high'] + 0.25 * df['low'] + 0.4 * df['close'])
    
    # Calculate the ratio of (high - low) to the 10-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_10 + 1e-7)
    
    # Calculate the 3-day and 10-day exponential moving averages of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 3-day and 10-day realized volatilities using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility_3 = log_returns.rolling(window=3).std()
    realized_volatility_10 = log_returns.rolling(window=10).std()
    
    # Momentum factor as the difference between the 3-day and 10-day EMA
    momentum_factor = (ema_3 - ema_10) / (realized_volatility_10 + 1e-7)
    
    # Liquidity factor as the ratio of volume to the 10-day SMA
    liquidity_factor = df['volume'] / (sma_10 + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - sma_10) / (volatility_ratio + 1e-7) + momentum_factor + liquidity_factor
    return alpha_factor
