def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day simple moving average of the close price for short-term trend
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate the 20-day simple moving average of the close price for long-term trend
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    adaptive_weights = (df['volume'] / df['volume'].rolling(window=5).mean()) + 1
    weighted_prices = (df['open'] * adaptive_weights + 
                      df['high'] * adaptive_weights + 
                      df['low'] * adaptive_weights + 
                      df['close'] * adaptive_weights) / (4 * adaptive_weights)
    
    # Calculate the ratio of (high - low) to the 5-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_5 + 1e-7)
    
    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 5-day mean reversion factor
    mean_reversion = (weighted_prices - sma_5) / (sma_5 + 1e-7)
    
    # Final alpha factor as a balanced combination of the above metrics
    alpha_factor = (
        0.5 * (ema_5 - df['close']) / (realized_volatility + 1e-7) +
        0.3 * (weighted_prices - sma_5) / (volatility_ratio + 1e-7) +
        0.1 * volume_factor +
        0.1 * mean_reversion +
        0.05 * (sma_5 - sma_20) / (sma_20 + 1e-7)  # Combine short and long-term trends
    )
    return alpha_factor
