def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    total_price = df[['open', 'high', 'low', 'close']].sum(axis=1)
    adaptive_weights_open = (df['open'] / total_price).fillna(0.25)
    adaptive_weights_high = (df['high'] / total_price).fillna(0.25)
    adaptive_weights_low = (df['low'] / total_price).fillna(0.25)
    adaptive_weights_close = (df['close'] / total_price).fillna(0.25)
    weighted_prices = (adaptive_weights_open * df['open'] + 
                       adaptive_weights_high * df['high'] + 
                       adaptive_weights_low * df['low'] + 
                       adaptive_weights_close * df['close'])
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 10-day momentum factor based on the change in close price
    momentum_factor = (df['close'] - df['close'].shift(10)) / (sma_3 + 1e-7)
    
    # Calculate the 30-day trend factor based on the change in close price
    trend_factor = (df['close'] - df['close'].shift(30)) / (sma_3 + 1e-7)
    
    # Calculate the 10-day mean reversion factor based on the difference between close price and 30-day moving average
    sma_30 = df['close'].rolling(window=30).mean()
    mean_reversion_factor = (df['close'] - sma_30) / (sma_3 + 1e-7)
    
    # Adaptive weights for momentum and mean reversion
    momentum_weight = 1 / (1 + np.exp(-(sma_3 - sma_10)))
    mean_reversion_weight = 1 - momentum_weight
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        (weighted_prices - sma_3) / (volatility_ratio + 1e-7) + 
        (ema_3 - df['close']) / (realized_volatility + 1e-7) +
        volume_factor + 
        momentum_weight * 0.8 * momentum_factor +  # Adjusted weight for momentum
        0.6 * trend_factor +  # Adjusted weight for trend
        mean_reversion_weight * 0.4 * mean_reversion_factor  # New mean reversion factor
    )
    return alpha_factor
