def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate adaptive weights based on recent price changes
    adaptive_weights_open = 0.1 + (df['open'] - df['open'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    adaptive_weights_high = 0.3 + (df['high'] - df['high'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    adaptive_weights_low = 0.3 + (df['low'] - df['low'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    adaptive_weights_close = 0.3 + (df['close'] - df['close'].shift(1)) / (df['high'] - df['low'] + 1e-7)
    weighted_prices = (adaptive_weights_open * df['open'] + 
                       adaptive_weights_high * df['high'] + 
                       adaptive_weights_low * df['low'] + 
                       adaptive_weights_close * df['close'])
    
    # Calculate the ratio of (high - low) to the 5-day moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Calculate the 5-day average trading volume
    avg_volume = df['volume'].rolling(window=5).mean()
    
    # Calculate the true range for an advanced volatility measure
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate the average true range (ATR) as an advanced volatility measure
    atr = true_range.rolling(window=14).mean()
    
    # Adjust the factor weights based on recent market trends
    market_trend = (df['close'] > df['close'].shift(1)).astype(int).rolling(window=5).sum() / 5
    
    # Final alpha factor as a combination of the above metrics with dynamic weightings
    alpha_factor = (
        (weighted_prices - ma_5) / (volatility_ratio + 1e-7) +
        (ema_5 - df['close']) / (realized_volatility + 1e-7) +
        (df['volume'] / avg_volume) +
        0.2 * (market_trend - 0.5) * (atr - realized_volatility)
    )
    return alpha_factor
