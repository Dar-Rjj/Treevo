```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the change in volume from the previous day to the current day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the average trading range (high - low) over the last 5 days
    df['avg_trading_range'] = df['high'].rolling(window=5).mean() - df['low'].rolling(window=5).mean()
    
    # Define a factor based on the ratio of the close price to the 5-day moving average of the close
    df['close_to_mavg_ratio'] = df['close'] / df['close'].rolling(window=5).mean()
    
    # Final alpha factor: weighted sum of volume change, average trading range, and close to moving average ratio
    factor = 0.4 * df['volume_change'] + 0.3 * df['avg_trading_range'] + 0.3 * df['close_to_mavg_ratio']
    
    return factor
```
