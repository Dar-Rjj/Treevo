def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day simple moving averages of the close price for trend analysis
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()

    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])

    # Calculate the ratio of (high - low) to the 5-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_5 + 1e-7)

    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()

    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()

    # Calculate the liquidity factor as the ratio of volume to the 5-day simple moving average of volume
    volume_sma_5 = df['volume'].rolling(window=5).mean()
    liquidity_factor = df['volume'] / (volume_sma_5 + 1e-7)

    # Calculate the mean reversion factor as the difference between the close price and its 10-day simple moving average
    mean_reversion_factor = (df['close'] - sma_10) / (sma_10 + 1e-7)

    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - sma_5) / (volatility_ratio + 1e-7) + \
                   (ema_5 - df['close']) / (realized_volatility + 1e-7) + \
                   liquidity_factor + \
                   mean_reversion_factor

    return alpha_factor
