```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between the close and the open price, suggesting momentum
    momentum = df['close'] - df['open']
    
    # Compute the average of high and low for a typical price representation
    avg_price = (df['high'] + df['low']) / 2
    
    # Consider the volume weighted amount, reflecting transaction activity
    vol_weighted_amount = df['amount'] / df['volume']
    
    # Create a factor that combines momentum with the relative position of the close to the average price
    # This factor aims to capture the strength of the price movement adjusted by where the closing occurred in relation to the day's average price
    factor = (momentum * vol_weighted_amount) / ((avg_price - df['low']) + 1e-7)
    
    return factor
```
