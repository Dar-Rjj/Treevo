```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 20-day, and 50-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Momentum factor: difference between the 5-day and 20-day moving averages
    momentum_short_term = ma_5 - ma_20
    momentum_long_term = ma_20 - ma_50
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Volatility factor: ratio of the range (high - low) to the 5-day moving average
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Volume trend: 10-day exponential moving average of volume
    ema_volume = df['volume'].ewm(span=10, adjust=False).mean()
    
    # Liquidity factor: ratio of current volume to the 10-day EMA of volume
    liquidity_ratio = df['volume'] / (ema_volume + 1e-7)
    
    # Adaptive weighting based on market trends
    adaptive_weight_short_term = (momentum_short_term > 0).astype(int) * 0.5 + (volatility_ratio < 0.1).astype(int) * 0.5
    adaptive_weight_long_term = (momentum_long_term > 0).astype(int) * 0.5 + (volatility_ratio < 0.1).astype(int) * 0.5
    adaptive_weight = (adaptive_weight_short_term + adaptive_weight_long_term) / 2
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (adaptive_weight * (weighted_prices - ma_5) / (volatility_ratio + 1e-7)) + ((1 - adaptive_weight) * liquidity_ratio)
    return alpha_factor
```
