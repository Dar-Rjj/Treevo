def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price to capture short and medium-term trends
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()

    # Momentum indicator: Difference between the 5-day and 20-day moving averages
    momentum = ma_5 - ma_20

    # Dynamic weights for open, high, low, and close prices based on 5-day realized volatility
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    dynamic_weight_open = 0.1 + 0.1 * (realized_volatility / realized_volatility.max())
    dynamic_weight_high = 0.3 + 0.1 * (realized_volatility / realized_volatility.max())
    dynamic_weight_low = 0.3 + 0.1 * (realized_volatility / realized_volatility.max())
    dynamic_weight_close = 0.3 + 0.1 * (realized_volatility / realized_volatility.max())

    weighted_prices = (dynamic_weight_open * df['open'] +
                       dynamic_weight_high * df['high'] +
                       dynamic_weight_low * df['low'] +
                       dynamic_weight_close * df['close'])

    # Volatility ratio: (high - low) divided by the 5-day moving average of the close price
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)

    # 5-day exponential moving average of the close price for smoother trend following
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()

    # 5-day realized volatility using the standard deviation of log returns
    realized_volatility = log_returns.rolling(window=5).std()

    # Adapted liquidity factor: Volume divided by the 5-day moving average of volume
    liquidity_factor = df['volume'] / df['volume'].rolling(window=5).mean()

    # Sector-specific trends: Calculate the 20-day sector momentum
    if 'sector' in df.columns:
        sector_momentum = df.groupby('sector')['close'].transform(lambda x: x / x.shift(20) - 1)
    else:
        sector_momentum = 0

    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        (weighted_prices - ma_5) / (volatility_ratio + 1e-7) +
        (ema_5 - df['close']) / (realized_volatility + 1e-7) +
        momentum * liquidity_factor +
        sector_momentum
    )
    return alpha_factor
