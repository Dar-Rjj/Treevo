```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate the momentum (difference between today's close and 20-day moving average)
    momentum = df['close'] - ma_20
    
    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the volume-price ratio
    volume_price_ratio = df['volume'] / (df['close'] + 1e-7)
    
    # Calculate the volatility (standard deviation of log returns over a 5-day window)
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1)).fillna(0)
    volatility = df['log_returns'].rolling(window=5).std().fillna(0)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - ma_5) * momentum * volume_price_ratio / (volatility + 1e-7)
    return alpha_factor
```
