```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the ratio of volume to the moving amount as a measure of liquidity
    liquidity_ratio = df['volume'] / (df['amount'].rolling(window=5).mean() + 1e-7)
    
    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Calculate the 5-day average trading amount
    avg_amount = df['amount'].rolling(window=5).mean()
    
    # Calculate the trend strength using the difference between the 5-day EMA and the 20-day EMA
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    trend_strength = (ema_5 - ema_20) / (realized_volatility + 1e-7)
    
    # Calculate the volume trend by comparing the current volume to the 5-day average volume
    volume_trend = (df['volume'] - df['volume'].rolling(window=5).mean()) / (df['volume'].rolling(window=5).std() + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - ma_5) / (liquidity_ratio + 1e-7) + (ema_5 - df['close']) / (realized_volatility + 1e-7) + trend_strength + volume_trend
    return alpha_factor
```
