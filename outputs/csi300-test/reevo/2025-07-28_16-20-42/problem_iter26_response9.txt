def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with dynamic weights
    weighted_prices = (df['open'] * 0.2 + df['high'] * 0.25 + df['low'] * 0.25 + df['close'] * 0.3)
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for short-term volatility
    short_volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 3-day mean reversion factor
    mean_reversion = (weighted_prices - sma_3) / (sma_3 + 1e-7)
    
    # Long-term trend factor
    long_term_trend = (sma_10 - df['close']) / (sma_10 + 1e-7)
    
    # Adaptive weight for combining short and long-term trends based on realized volatility
    adaptive_weight = 1 / (realized_volatility + 1e-7)
    
    # Final alpha factor as a balanced combination of the above metrics
    alpha_factor = (
        adaptive_weight * 0.3 * (ema_3 - df['close']) +
        adaptive_weight * 0.2 * (weighted_prices - sma_3) / (short_volatility_ratio + 1e-7) +
        0.2 * volume_factor +
        0.15 * mean_reversion +
        0.15 * long_term_trend
    )
    return alpha_factor
