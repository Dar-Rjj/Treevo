```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving average of the close price to capture short and long-term trends
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    adaptive_weights = (sma_10 - sma_3) / (sma_10 + 1e-7)
    weighted_prices = (0.1 * adaptive_weights * df['open'] + 0.3 * adaptive_weights * df['high'] 
                       + 0.3 * adaptive_weights * df['low'] + 0.3 * adaptive_weights * df['close'])
    
    # Calculate the ratio of (high - low) to the 10-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_10 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Momentum factor: difference between the 3-day and 10-day SMA
    momentum_factor = sma_3 - sma_10
    
    # Liquidity factor: volume divided by amount
    liquidity_factor = df['volume'] / (df['amount'] + 1e-7)
    
    # Mean reversion factor: difference between close price and 10-day SMA
    mean_reversion_factor = df['close'] - sma_10
    
    # Final alpha factor as a combination of the above metrics with balanced weights
    alpha_factor = (weighted_prices - sma_3) / (volatility_ratio + 1e-7) + (ema_3 - df['close']) / (realized_volatility + 1e-7) + \
                   0.5 * momentum_factor + 0.2 * liquidity_factor + 0.3 * mean_reversion_factor
    return alpha_factor
```
