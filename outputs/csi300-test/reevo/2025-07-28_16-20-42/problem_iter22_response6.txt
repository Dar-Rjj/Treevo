def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    adaptive_weights_open = (df['open'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    adaptive_weights_high = (df['high'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    adaptive_weights_low = (df['low'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    adaptive_weights_close = (df['close'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    weighted_prices = (adaptive_weights_open * df['open'] + 
                       adaptive_weights_high * df['high'] + 
                       adaptive_weights_low * df['low'] + 
                       adaptive_weights_close * df['close'])
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Balance short-term EMAs with longer-term trends
    sma_9 = df['close'].rolling(window=9).mean()
    trend_factor = (sma_9 - sma_3) / (sma_9 + 1e-7)
    
    # Momentum factor: 3-day return
    momentum_3 = df['close'].pct_change(periods=3)
    
    # Incorporate a longer-term trend factor
    sma_21 = df['close'].rolling(window=21).mean()
    long_trend_factor = (sma_21 - sma_9) / (sma_21 + 1e-7)
    
    # Adaptive weights for momentum and mean reversion
    momentum_weight = 0.5 * (ema_3 > df['close']).astype(float)
    mean_reversion_weight = 0.5 * (ema_3 <= df['close']).astype(float)
    
    # Final alpha factor as a combination of the above metrics with adaptive weights
    alpha_factor = (
        momentum_weight * ((weighted_prices - sma_3) / (volatility_ratio + 1e-7)) +
        mean_reversion_weight * ((ema_3 - df['close']) / (realized_volatility + 1e-7)) +
        volume_factor +
        trend_factor +
        0.5 * momentum_3 +
        long_trend_factor
    )
    return alpha_factor
