```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 20-day moving averages of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_20 = df['close'].rolling(window=20).mean()

    # Calculate a weighted sum of the open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])

    # Calculate the true range for advanced volatility measure
    tr1 = df['high'] - df['low']
    tr2 = abs(df['high'] - df['close'].shift(1))
    tr3 = abs(df['low'] - df['close'].shift(1))
    true_range = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
    
    # Calculate the average true range (ATR) as an advanced volatility measure
    atr = true_range.rolling(window=14).mean()

    # Calculate the momentum factor as the difference between the 5-day and 20-day moving averages
    momentum_factor = ma_5 - ma_20

    # Calculate the liquidity factor as the ratio of volume to amount
    liquidity_factor = df['volume'] / (df['amount'] + 1e-7)

    # Calculate the 5-day and 20-day exponential moving averages of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()

    # Refine the momentum calculation using EMA
    refined_momentum_factor = ema_5 - ema_20

    # Adjust the factor weights based on market conditions
    market_trend = (df['close'] > df['close'].shift(1)).astype(int).rolling(window=5).sum() / 5
    alpha_factor = (weighted_prices - ma_5) / (atr + 1e-7) + 0.5 * refined_momentum_factor + 0.3 * liquidity_factor
    alpha_factor = alpha_factor * (1 + 0.2 * (market_trend - 0.5))

    return alpha_factor
```
