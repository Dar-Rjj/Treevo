def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day and 10-day moving average of the close price
    ma_5 = df['close'].rolling(window=5).mean()
    ma_10 = df['close'].rolling(window=10).mean()
    
    # Price change rate over 5 days
    price_change_rate = (df['close'] - df['close'].shift(5)) / df['close'].shift(5)
    
    # Volume trend as a ratio of current volume to the 5-day average volume
    avg_volume_5d = df['volume'].rolling(window=5).mean()
    volume_trend = df['volume'] / (avg_volume_5d + 1e-7)
    
    # Weighted sum of open, high, low, and close prices
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Volatility ratio
    volatility_ratio = (df['high'] - df['low']) / (ma_5 + 1e-7)
    
    # Adaptive weighting for combining factors
    weight_price_change_rate = np.clip(price_change_rate, -1, 1).abs()
    weight_volume_trend = np.clip(volume_trend, -1, 1).abs()
    
    # Normalize weights
    total_weight = weight_price_change_rate + weight_volume_trend + 1
    weight_price_change_rate /= total_weight
    weight_volume_trend /= total_weight
    base_weight = 1 / total_weight
    
    # Final alpha factor
    alpha_factor = (weighted_prices - ma_5) / (volatility_ratio + 1e-7)
    alpha_factor += (price_change_rate * weight_price_change_rate + volume_trend * weight_volume_trend) * base_weight
    return alpha_factor
