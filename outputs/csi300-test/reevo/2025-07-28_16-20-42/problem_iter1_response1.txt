```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between close and open price, indicating the price movement of the day
    price_diff = df['close'] - df['open']
    
    # Calculate the average of the absolute differences between high and low prices over the last 5 days.
    # This represents the recent volatility (or uncertainty in the market).
    recent_volatility = df[['high', 'low']].diff(axis=1).abs().iloc[:,0].rolling(window=5).mean()
    
    # Factor: The ratio of daily price change to recent average volatility, aimed at capturing momentum with respect to recent market conditions.
    factor = price_diff / (recent_volatility + 1e-7)
    
    return factor
```
