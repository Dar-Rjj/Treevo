```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate momentum factor by taking the 10 day difference in close price
    momentum_factor = df['close'].diff(10)
    
    # Calculate volatility over a 5-day window
    volatility_factor = df['close'].rolling(window=5).std()
    
    # Calculate trading volume intensity as a percentage of the average volume over a 14-day period
    avg_volume_14d = df['volume'].rolling(window=14).mean()
    volume_intensity = (df['volume'] - avg_volume_14d) / avg_volume_14d

    # Generate the alpha factor as a weighted sum of these components, with weights chosen to balance their contributions
    alpha_factor = (momentum_factor * 0.4) + (volatility_factor * 0.3) + (volume_intensity * 0.3)

    return alpha_factor
```
