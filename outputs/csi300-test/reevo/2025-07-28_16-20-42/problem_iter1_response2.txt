```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 1-day simple moving average of the close prices
    sma_1d = df['close'].rolling(window=1).mean()
    
    # Calculate the 5-day simple moving average of the close prices
    sma_5d = df['close'].rolling(window=5).mean()
    
    # Calculate the relative position of the current close price compared to the 5-day SMA
    relative_position = (df['close'] - sma_5d) / sma_5d
    
    # Calculate the ratio of today's volume to the 5-day mean volume as a measure of volume shock
    volume_shock = df['volume'] / df['volume'].rolling(window=5).mean()
    
    # Combine the relative position and the volume shock to create a factor
    alpha_factor = relative_position * volume_shock
    
    return alpha_factor
```
