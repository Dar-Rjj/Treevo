def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 30-day exponential moving average of the close price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    ema_30 = df['close'].ewm(span=30, adjust=False).mean()
    
    # Calculate the 10-day simple moving average of the close price
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 10-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=10).std()
    
    # Calculate the ratio of (high - low) to the 10-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_10 + 1e-7)
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=10).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 10-day mean reversion factor
    mean_reversion = (sma_10 - df['close']) / (sma_10 + 1e-7)
    
    # Calculate the 10-day momentum factor
    momentum = df['close'] / df['close'].shift(10) - 1
    
    # Calculate relative strength
    relative_strength = (ema_10 - ema_30) / (ema_30 + 1e-7)
    
    # Calculate the 10-day skewness of the close price
    skewness = log_returns.rolling(window=10).skew()
    
    # Calculate the 10-day kurtosis of the close price
    kurtosis = log_returns.rolling(window=10).kurt()
    
    # Seasonality factor (assuming daily data, weekday can be used)
    if isinstance(df.index, pd.DatetimeIndex):
        seasonality = df.index.dayofweek
    else:
        seasonality = pd.Series(0, index=df.index)
    
    # Final alpha factor as a weighted combination of the above metrics
    alpha_factor = (
        0.15 * (ema_10 - df['close']) / (realized_volatility + 1e-7) +
        0.15 * (df['close'] - sma_10) / (volatility_ratio + 1e-7) +
        0.15 * volume_factor +
        0.15 * mean_reversion +
        0.15 * momentum +
        0.1 * relative_strength +
        0.05 * skewness +
        0.05 * kurtosis +
        0.05 * seasonality
    )
    return alpha_factor
