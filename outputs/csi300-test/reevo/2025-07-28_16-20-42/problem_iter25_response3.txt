def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day and 10-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with equal weights
    weighted_prices = (df['open'] + df['high'] + df['low'] + df['close']) / 4
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day and 10-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 3-day and 10-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility_3 = log_returns.rolling(window=3).std()
    realized_volatility_10 = log_returns.rolling(window=10).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 3-day mean reversion factor
    mean_reversion_3 = (weighted_prices - sma_3) / (sma_3 + 1e-7)
    mean_reversion_10 = (weighted_prices - sma_10) / (sma_10 + 1e-7)
    
    # Adaptive weights for combining short and long-term trends
    adaptive_weight = 1 / (realized_volatility_3 + realized_volatility_10 + 1e-7)
    
    # Final alpha factor as a balanced combination of the above metrics
    alpha_factor = (
        adaptive_weight * (ema_3 - df['close']) / (realized_volatility_3 + 1e-7) +
        adaptive_weight * (ema_10 - df['close']) / (realized_volatility_10 + 1e-7) +
        0.5 * volume_factor +
        0.5 * mean_reversion_3 +
        0.5 * mean_reversion_10
    )
    return alpha_factor
