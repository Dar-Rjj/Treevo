def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day simple moving average of the close price
    sma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    adaptive_weights_open = (df['open'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    adaptive_weights_high = (df['high'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    adaptive_weights_low = (df['low'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    adaptive_weights_close = (df['close'] / df[['open', 'high', 'low', 'close']].sum(axis=1)).fillna(0.25)
    weighted_prices = (adaptive_weights_open * df['open'] + 
                       adaptive_weights_high * df['high'] + 
                       adaptive_weights_low * df['low'] + 
                       adaptive_weights_close * df['close'])
    
    # Calculate the ratio of (high - low) to the 5-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_5 + 1e-7)
    
    # Calculate the 5-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=5).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 5-day momentum factor as the difference between the 5-day EMA and SMA
    momentum_factor = ema_5 - sma_5
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = ((weighted_prices - sma_5) / (volatility_ratio + 1e-7) + 
                    (ema_5 - df['close']) / (realized_volatility + 1e-7) +
                    volume_factor + 
                    momentum_factor)
    return alpha_factor
