```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    total_price = df[['open', 'high', 'low', 'close']].sum(axis=1)
    adaptive_weights_open = (df['open'] / total_price).fillna(0.25)
    adaptive_weights_high = (df['high'] / total_price).fillna(0.25)
    adaptive_weights_low = (df['low'] / total_price).fillna(0.25)
    adaptive_weights_close = (df['close'] / total_price).fillna(0.25)
    weighted_prices = (adaptive_weights_open * df['open'] + 
                       adaptive_weights_high * df['high'] + 
                       adaptive_weights_low * df['low'] + 
                       adaptive_weights_close * df['close'])
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Calculate the 3-day exponential moving average of the close price
    ema_3 = df['close'].ewm(span=3, adjust=False).mean()
    
    # Calculate the 5-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=5).std()
    
    # Incorporate volume data for robustness
    avg_volume = df['volume'].rolling(window=3).mean()
    volume_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Incorporate liquidity by calculating the turnover ratio (volume / amount)
    liquidity = df['volume'] / (df['amount'] + 1e-7)
    
    # Mean reversion factor: difference between close price and 3-day SMA
    mean_reversion_factor = (df['close'] - sma_3) / (sma_3 + 1e-7)
    
    # Adaptive momentum factor: weighted sum of different momentum periods
    momentum_3 = df['close'].pct_change(periods=3)
    momentum_5 = df['close'].pct_change(periods=5)
    adaptive_momentum = 0.5 * momentum_3 + 0.5 * momentum_5
    
    # Trend strength factor: rolling correlation between close price and its 3-day EMA
    trend_strength = df['close'].rolling(window=5).corr(ema_3)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = ((weighted_prices - sma_3) / (volatility_ratio + 1e-7) + 
                    (ema_3 - df['close']) / (realized_volatility + 1e-7) +
                    volume_factor + 
                    0.1 * liquidity +
                    0.3 * mean_reversion_factor +
                    adaptive_momentum +
                    0.2 * trend_strength)
    return alpha_factor
```
