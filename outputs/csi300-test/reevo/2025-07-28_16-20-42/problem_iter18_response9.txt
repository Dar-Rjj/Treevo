def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day simple moving average of the close price for longer-term trend
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with fixed weights
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the ratio of (high - low) to the 10-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_10 + 1e-7)
    
    # Calculate the 10-day exponential moving average of the close price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=10).std()
    
    # Momentum factor: difference between the 10-day EMA and 10-day SMA
    momentum_factor = ema_10 - sma_10
    
    # Mean reversion factor: difference between the 10-day EMA and the current close price, normalized by realized volatility
    mean_reversion_factor = (ema_10 - df['close']) / (realized_volatility + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - sma_10) / (volatility_ratio + 1e-7) + \
                   momentum_factor + \
                   mean_reversion_factor
    return alpha_factor
