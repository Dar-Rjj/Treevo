def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day, 10-day, and 20-day simple moving average of the close price
    sma_5 = df['close'].rolling(window=5).mean()
    sma_10 = df['close'].rolling(window=10).mean()
    sma_20 = df['close'].rolling(window=20).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights based on volume
    avg_volume = df['volume'].rolling(window=20).mean()
    adaptive_weights = df['volume'] / (avg_volume + 1e-7)
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close']) * adaptive_weights
    
    # Calculate the ratio of (high - low) to the 20-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_20 + 1e-7)
    
    # Calculate the 5-day and 20-day exponential moving average of the close price
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    ema_20 = df['close'].ewm(span=20, adjust=False).mean()
    
    # Calculate the 5-day and 20-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility_5 = log_returns.rolling(window=5).std()
    realized_volatility_20 = log_returns.rolling(window=20).std()
    
    # Incorporate momentum by calculating the 5-day, 10-day, and 20-day return
    momentum_5 = df['close'].pct_change(periods=5)
    momentum_10 = df['close'].pct_change(periods=10)
    momentum_20 = df['close'].pct_change(periods=20)
    
    # Incorporate liquidity by calculating the turnover ratio (volume / amount)
    liquidity = df['volume'] / (df['amount'] + 1e-7)
    
    # Mean reversion factor: difference between close price and 20-day SMA
    mean_reversion_factor = df['close'] - sma_20
    
    # Adaptive momentum factor: weighted sum of different momentum periods
    adaptive_momentum = (momentum_5 + 2 * momentum_10 + 3 * momentum_20) / 6
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        (weighted_prices - sma_5) / (volatility_ratio + 1e-7) +
        (ema_5 - df['close']) / (realized_volatility_5 + 1e-7) +
        0.5 * adaptive_momentum +
        0.1 * liquidity +
        0.3 * mean_reversion_factor
    )
    return alpha_factor
