```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day momentum as the difference between the current close and the close 10 days ago
    df['momentum_10d'] = df['close'] - df['close'].shift(10)
    
    # Calculate the 20-day volatility as the standard deviation of the last 20 days' close prices
    df['volatility_20d'] = df['close'].rolling(window=20).std()
    
    # Calculate the change in volume from the previous day to the current day
    df['volume_change'] = df['volume'] - df['volume'].shift(1)
    
    # Calculate the average trading range (high - low) over the last 5 days
    df['avg_trading_range'] = (df['high'].rolling(window=5).mean() - df['low'].rolling(window=5).mean()) / df['close'].rolling(window=5).mean()
    
    # Define a factor based on the ratio of the close price to the 5-day moving average of the close
    df['close_to_mavg_ratio'] = df['close'] / df['close'].rolling(window=5).mean()
    
    # Final alpha factor: weighted sum of momentum, volatility, volume change, and other factors
    factor = 0.3 * df['momentum_10d'] + 0.2 * df['volatility_20d'] + 0.2 * df['volume_change'] + 0.1 * df['avg_trading_range'] + 0.2 * df['close_to_mavg_ratio']
    
    return factor
```
