def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 3-day simple moving average of the close price
    sma_3 = df['close'].rolling(window=3).mean()
    
    # Calculate the 10-day simple moving average of the close price for trend strength
    sma_10 = df['close'].rolling(window=10).mean()
    
    # Calculate the 3-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=3).std()
    
    # Calculate the ratio of (high - low) to the 3-day simple moving average of the close price for dynamic volatility
    dynamic_volatility = (df['high'] - df['low']) / (sma_3 + 1e-7)
    
    # Incorporate volume data for liquidity
    avg_volume = df['volume'].rolling(window=3).mean()
    liquidity_factor = (df['volume'] - avg_volume) / (avg_volume + 1e-7)
    
    # Calculate the 3-day mean reversion factor
    mean_reversion = (sma_3 - df['close']) / (sma_3 + 1e-7)
    
    # Calculate the 10-day momentum factor based on the change in close price
    momentum_factor = (df['close'] - df['close'].shift(10)) / (sma_10 + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (
        (df['close'] - sma_3) / (realized_volatility + 1e-7) +
        mean_reversion +
        liquidity_factor +
        momentum_factor
    )
    return alpha_factor
