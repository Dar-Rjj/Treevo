```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between today's and yesterday's close price
    df['close_diff'] = df['close'].diff().fillna(0)
    
    # Calculate the change in volume
    df['volume_change'] = df['volume'].pct_change().fillna(0)
    
    # Calculate the volatility (standard deviation of log returns over a 5-day window)
    df['log_returns'] = np.log(df['close'] / df['close'].shift(1)).fillna(0)
    df['volatility'] = df['log_returns'].rolling(window=5).std().fillna(0)
    
    # Alpha factor: (change in close price) * (volume change) / (volatility + 1e-7)
    alpha_factor = (df['close_diff'] * df['volume_change']) / (df['volatility'] + 1e-7)
    
    return alpha_factor
```
