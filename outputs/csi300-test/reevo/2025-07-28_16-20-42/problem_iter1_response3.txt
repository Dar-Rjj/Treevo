```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the difference between high and low for a given day as true range
    true_range = df['high'] - df['low']
    # Calculate average volume over the past 5 days
    avg_volume = df['volume'].rolling(window=5).mean()
    # Calculate momentum based on the close price of 5 days difference
    momentum = df['close'] - df['close'].shift(5)
    # Combine true range, average volume, and momentum to create a new factor
    factor_values = (true_range + avg_volume) * momentum
    return factor_values
```
