```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 1-day simple moving average of the close price
    sma_1 = df['close'].rolling(window=1).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with optimized weights
    weighted_prices = (0.1 * df['open'] + 0.45 * df['high'] + 0.45 * df['low'] + 0.1 * df['close'])
    
    # Calculate the ratio of (high - low) to the 1-day simple moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (sma_1 + 1e-7)
    
    # Calculate the 1-day exponential moving average of the close price
    ema_1 = df['close'].ewm(span=1, adjust=False).mean()
    
    # Calculate the 1-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=1).std()
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - sma_1) / (volatility_ratio + 1e-7) + (ema_1 - df['close']) / (realized_volatility + 1e-7)
    return alpha_factor
```
