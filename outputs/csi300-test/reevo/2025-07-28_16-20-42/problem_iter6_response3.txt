```python
def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 5-day moving average of the close price to capture trend
    ma_5 = df['close'].rolling(window=5).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with different weights
    weighted_prices = (0.1 * df['open'] + 0.3 * df['high'] + 0.3 * df['low'] + 0.3 * df['close'])
    
    # Calculate the daily range to measure volatility
    daily_range = df['high'] - df['low']
    
    # Calculate the ratio of daily range to the 5-day moving average to normalize volatility
    volatility_ratio = daily_range / (ma_5 + 1e-7)
    
    # Calculate the volume-weighted amount to reflect trading activity
    vol_weighted_amount = df['amount'] / df['volume']
    
    # Calculate the 5-day exponential moving average of the close price for smoother trend
    ema_5 = df['close'].ewm(span=5, adjust=False).mean()
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = ((weighted_prices - ema_5) * vol_weighted_amount) / (volatility_ratio + 1e-7)
    return alpha_factor
```
