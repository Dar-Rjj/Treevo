def heuristics_v2(df: pd.DataFrame) -> pd.Series:
    # Calculate the 10-day and 50-day moving average of the close price for longer-term trends
    ma_10 = df['close'].rolling(window=10).mean()
    ma_50 = df['close'].rolling(window=50).mean()
    
    # Calculate a weighted sum of the open, high, low, and close prices with adaptive weights
    adaptive_weights_open = 1 / (df['open'].rolling(window=10).std() + 1e-7)
    adaptive_weights_high = 1 / (df['high'].rolling(window=10).std() + 1e-7)
    adaptive_weights_low = 1 / (df['low'].rolling(window=10).std() + 1e-7)
    adaptive_weights_close = 1 / (df['close'].rolling(window=10).std() + 1e-7)
    total_weight = adaptive_weights_open + adaptive_weights_high + adaptive_weights_low + adaptive_weights_close
    weighted_prices = (adaptive_weights_open * df['open'] + adaptive_weights_high * df['high'] + 
                       adaptive_weights_low * df['low'] + adaptive_weights_close * df['close']) / total_weight
    
    # Calculate the ratio of (high - low) to the 10-day moving average of the close price for volatility
    volatility_ratio = (df['high'] - df['low']) / (ma_10 + 1e-7)
    
    # Calculate the 10-day exponential moving average of the close price
    ema_10 = df['close'].ewm(span=10, adjust=False).mean()
    
    # Calculate the 10-day realized volatility using the standard deviation of log returns
    log_returns = np.log(df['close'] / df['close'].shift(1))
    realized_volatility = log_returns.rolling(window=10).std()
    
    # Momentum factor: difference between the 10-day EMA and 50-day MA
    momentum_factor = ema_10 - ma_50
    
    # Liquidity factor: ratio of volume to amount
    liquidity_factor = df['volume'] / (df['amount'] + 1e-7)
    
    # Adaptive mean reversion factor: difference between the 10-day EMA and the current close price
    mean_reversion_factor = (ema_10 - df['close']) / (realized_volatility + 1e-7)
    
    # Final alpha factor as a combination of the above metrics
    alpha_factor = (weighted_prices - ma_10) / (volatility_ratio + 1e-7) + \
                   mean_reversion_factor + \
                   momentum_factor + \
                   liquidity_factor
    return alpha_factor
